```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Price_Velocity'] = df['close'].diff(1)
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = (df['close'] / df['close'].shift(7)) - 1
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = (df['close'] / df['close'].shift(30)) - 1
    
    # Calculate Daily Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['Adjusted_Momentum'] = (df['Short_Term_Momentum'] - df['Long_Term_Momentum']) * df['Volume_Change']
    
    # Introduce Volume Weighted Price Momentum
    for i in range(1, 8):
        df[f'Volume_Weighted_Price_{i}'] = df['volume'].shift(i-1) * df['close'].shift(i-1)
    df['Volume_Weighted_Price_Sum'] = df[[f'Volume_Weighted_Price_{i}' for i in range(1, 8)]].sum(axis=1)
    df['Volume_Weighted_Price_Momentum'] = df['Volume_Weighted_Price_Sum'] / df['volume'].rolling(window=7).sum()
    
    # Synthesize Final Factor
    df['Final_Factor'] = df['Adjusted_Momentum'] + df['Volume_Weighted_Price_Momentum']
    
    return df['Final_Factor']
```
