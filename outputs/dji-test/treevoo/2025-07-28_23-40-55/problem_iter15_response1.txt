```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate the Price Change from Previous Day
    df['Price_Change_1D'] = df['close'].diff()
    
    # Calculate the Price Change Over a Longer Period (e.g., 20 days)
    df['Price_Change_20D'] = df['close'] - df['close'].shift(n)
    
    # Calculate the Volume Change from Previous Day
    df['Volume_Change_1D'] = df['volume'].diff()
    
    # Calculate the Average Volume Over a Fixed Period (e.g., 20 days)
    df['Avg_Volume_20D'] = df['volume'].rolling(window=n).mean()
    
    # Calculate the Volume-Weighted Price Change over the last n days
    df['Vol_Weighted_Price_Change_20D'] = (df['close'].diff() * df['volume']).rolling(window=n).sum()
    
    # Calculate the Price-Volatility Ratio
    price_volatility = df['close'].rolling(window=n).std()
    df['Price_Volatility_Ratio_20D'] = price_volatility / df['Avg_Volume_20D']
    
    # Calculate the Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate the Average Daily Price Range Over a Fixed Period (e.g., 20 days)
    df['Avg_Daily_Price_Range_20D'] = df['Daily_Price_Range'].rolling(window=n).mean()
    
    # Calculate the High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate the Intraday Price Movement
    df['Intraday_Price_Movement'] = df['close'] - df['open']
    
    # Calculate the Directional Price Movement
    df['Directional_Price_Movement'] = df['Intraday_Price_Movement'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Calculate the Intraday Price Volatility over the last n days
    df['Intraday_Price_Volatility_20D'] = df['Intraday_Price_Movement'].rolling(window=n).std()
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        df['Price_Change_1D'] + 
        df['Price_Change_20D'] + 
        df['Volume_Change_1D'] + 
        df['Vol_Weighted_Price_Change_20D'] + 
        df['Price_Volatility_Ratio_20D'] + 
        df['Avg_Daily_Price_Range_20D'] + 
        df['High_Low_Ratio'] + 
        df['Intraday_Price_Movement'] + 
        df['Directional_Price_Movement'] + 
        df['Intraday_Price_Volatility_20D']
    )
    
    return alpha_factor
```
