```python
def heuristics_v2(df):
    import pandas as pd
    
    # Short-Term Price Momentum (5 days)
    short_term_days = 5
    df['Close_diff_5d'] = df['close'].diff(short_term_days)
    df['Short_Term_Price_Momentum'] = df['Close_diff_5d'].rolling(window=short_term_days).sum() / short_term_days

    # Long-Term Price Momentum (20 days)
    long_term_days = 20
    df['Close_diff_20d'] = df['close'].diff(long_term_days)
    df['Long_Term_Price_Momentum'] = df['Close_diff_20d'].rolling(window=long_term_days).sum() / long_term_days

    # Volume Trend
    df['Daily_Volume_Change'] = df['volume'].diff()
    df['Total_Volume_Change_5d'] = df['Daily_Volume_Change'].rolling(window=short_term_days).sum() / short_term_days

    # Combine Factors
    df['Alpha_Factor'] = (df['Short_Term_Price_Momentum'] 
                          * df['Long_Term_Price_Momentum'] 
                          * df['Total_Volume_Change_5d'])

    return df['Alpha_Factor']
```
