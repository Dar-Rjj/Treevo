```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term EMA (12-day span) on Close price
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate long-term EMA (26-day span) on Close price
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Subtract Long-Term EMA from Short-Term EMA
    ema_diff = short_ema - long_ema
    
    # Multiply the difference by Volume to get the Adjusted Momentum
    adjusted_momentum = ema_diff * df['volume']
    
    return adjusted_momentum
```
