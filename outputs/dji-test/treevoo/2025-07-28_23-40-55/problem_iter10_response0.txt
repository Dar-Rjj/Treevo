```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Change
    df['price_velocity'] = df['close'].diff()

    # Calculate Short-Term EMA (12-day span)
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = df['ema_diff'] * df['volume_change']
    
    # Synthesize Final Factor
    df['final_factor'] = df['price_velocity'] + df['adjusted_momentum'] + df['ema_diff']
    
    return df['final_factor']
```
