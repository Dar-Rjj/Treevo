```python
def heuristics_v2(data):
    import pandas as pd
    
    # Calculate Short-Term Momentum
    short_term_momentum = data['close'].pct_change(6)
    
    # Calculate Long-Term Momentum
    long_term_momentum = data['close'].pct_change(29)
    
    # Calculate Daily Volume Change
    volume_change = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_adjustment = (short_term_momentum - long_term_momentum) * volume_change
    
    # Calculate Intraday High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = data['close'] - data['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    intraday_momentum = high_low_range + open_close_momentum
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = momentum_adjustment + intraday_momentum
    
    # Apply Volume Filter to Identify High Volume Days
    volume_rolling_mean = data['volume'].rolling(window=20).mean()
    high_volume_threshold = 1.5 * volume_rolling_mean  # Example threshold, can be adjusted
    high_volume_days = data['volume'] > high_volume_threshold
    
    # Refine the Final Factor: Only Include High Volume Days
    refined_alpha_factor = final_alpha_factor.where(high_volume_days, other=0)
    
    return refined_alpha_factor
```
