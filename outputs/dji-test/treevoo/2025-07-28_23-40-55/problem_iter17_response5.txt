```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Short-Term and Long-Term Momentum
    df['short_term_momentum'] = (df['close'].shift(1) - df['close'].shift(7)) / 7
    df['long_term_momentum'] = (df['close'].shift(1) - df['close'].shift(30)) / 30
    
    # Calculate Weighted EMAs
    df['weighted_close'] = df['close'] * df['volume']
    df['short_wema'] = df['weighted_close'].ewm(span=12, adjust=False).mean()
    df['long_wema'] = df['weighted_close'].ewm(span=26, adjust=False).mean()
    
    # WEMA Difference
    df['wema_diff'] = df['short_wema'] - df['long_wema']
    
    # Weighted Momentum Signal
    df['avg_volume_26'] = df['volume'].rolling(window=26).mean()
    df['weighted_momentum_signal'] = df['wema_diff'] * (df['volume'] / df['avg_volume_26'])
    
    # Volume-Adjusted Volatility
    df['volatility'] = df['high'].sub(df['low']).rolling(window=26).std()
    df['volume_adjusted_volatility'] = df['volatility'] * (df['volume'] / df['avg_volume_26'])
    
    # Combined Alpha Factor
    df['combined_alpha_factor'] = df['weighted_momentum_signal'] / df['volume_adjusted_volatility']
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['momentum_differential'] = df['short_term_momentum'] - df['long_term_momentum']
    df['adjusted_momentum'] = df['momentum_differential'] * df['volume_change']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume'])
    df['sum_vwap_7'] = df['vwap'].rolling(window=7).sum()
    
    # Synthesize Final Factor
    df['synthesize_final_factor'] = df['adjusted_momentum'] + df['sum_vwap_7']
    
    # Final Enhanced Factor
    df['final_enhanced_factor'] = df['synthesize_final_factor'] + df['combined_alpha_factor']
    
    return df['final_enhanced_factor']
```
