import pandas as pd

def heuristics_v2(df):
    # Short-Term WEMA (12-day span, Close price * Volume)
    short_term_wema = df['close'] * df['volume']
    short_term_wema = short_term_wema.ewm(span=12, adjust=False).mean()
    
    # Long-Term WEMA (26-day span, Close price * Volume)
    long_term_wema = df['close'] * df['volume']
    long_term_wema = long_term_wema.ewm(span=26, adjust=False).mean()
    
    # WEMA Difference
    wema_diff = short_term_wema - long_term_wema
    
    # Weighted Momentum Signal
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    weighted_momentum_signal = wema_diff * (df['volume'] / avg_volume_26)
    
    # Volume-Adjusted Volatility
    volatility = df['high'] - df['low']
    volatility = volatility.rolling(window=26).std()
    volume_adjusted_volatility = volatility * (df['volume'] / avg_volume_26)
    
    # Combined Alpha Factor
    combined_alpha_factor = weighted_momentum_signal / volume_adjusted_volatility
    
    return combined_alpha_factor
