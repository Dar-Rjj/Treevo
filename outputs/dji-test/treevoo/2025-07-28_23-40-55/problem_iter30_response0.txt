```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['high_low_range'] + df['open_close_momentum']

    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=5).mean()
    high_volume_threshold = df['volume_rolling_mean'].quantile(0.75)
    high_volume_days = df[df['volume'] > high_volume_threshold]

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(1)

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(30)

    # Adjust Price Momentum by Volume Shock
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_price_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']

    # Calculate Daily Volume Change
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Intraday Momentum by Volatility
    df['volatility'] = (df['high'] - df['low']) / df['close']
    df['adjusted_intraday_momentum'] = df['integrated_momentum'] * df['volatility']

    # Calculate Price Velocity
    df['price_velocity'] = df['close'] - df['close'].shift(1)

    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)

    # Calculate Short-Term EMA
    df['short_term_ema'] = df['close'].ewm(span=12, adjust=False).mean()

    # Calculate Long-Term EMA
    df['long_term_ema'] = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    df['ema_difference'] = df['short_term_ema'] - df['long_term_ema']

    # Adjust EMA Difference by Volume Shock
    df['adjusted_ema_difference'] = df['ema_difference'] * df['volume_change']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_price_momentum'] + 
                                df['adjusted_intraday_momentum'] + 
                                df['price_acceleration'] + 
                                df['adjusted_ema_difference'])

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Incorporate Intraday Return
    df['final_alpha_factor'] *= df['intraday_return']

    # Refine the Final Factor
    final_factor = df.loc[high_volume_days.index, 'final_alpha_factor']

    # Adjust Momentum by High-Low Volatility
    final_factor = final_factor * df.loc[high_volume_days.index, 'volatility']

    return final_factor
```
