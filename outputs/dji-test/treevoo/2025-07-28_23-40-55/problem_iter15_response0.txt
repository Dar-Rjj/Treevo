import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (high_low_range + open_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > volume_threshold
    filtered_integrated_momentum = integrated_momentum[high_volume_days].reindex(integrated_momentum.index, fill_value=0)
    
    # Calculate Short-Term EMA (7-day span, Close price)
    short_term_ema = df['close'].ewm(span=7).mean()
    
    # Calculate Long-Term EMA (21-day span, Close price)
    long_term_ema = df['close'].ewm(span=21).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)).rolling(window=5).sum() / 5
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(30)).rolling(window=30).sum() / 30
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=5).sum() / 5
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * total_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = filtered_integrated_momentum + adjusted_momentum
    
    # Calculate EMAs for final adjusted momentum signal
    short_term_ema_final = df['close'].ewm(span=10).mean()
    long_term_ema_final = df['close'].ewm(span=40).mean()
    ema_difference_final = short_term_ema_final - long_term_ema_final
    adjusted_momentum_signal = (ema_difference_final * df['volume']) + (df['high'] - df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum * df['close']) + adjusted_momentum_signal
    
    return final_alpha_factor
