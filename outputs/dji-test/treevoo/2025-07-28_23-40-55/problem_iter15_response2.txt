```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = 1.5 * volume_rolling_mean
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_alpha_factor = integrated_momentum[high_volume_days]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    momentum_with_return = filtered_alpha_factor * intraday_return[high_volume_days]
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    final_alpha_factor = momentum_with_return * high_low_volatility[high_volume_days]
    
    return final_alpha_factor
```
