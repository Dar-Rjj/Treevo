```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['short_term_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['long_term_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_difference'] = df['short_term_ema'] - df['long_term_ema']
    
    # Generate Composite Momentum Signal
    df['composite_momentum'] = df['ema_difference'] * df['volume'] + df['price_acceleration']
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12d'] = df['volume'].rolling(window=12).mean()
    df['vol_weighted_ema_diff'] = df['ema_difference'] * df['avg_volume_12d']
    
    # Adjust Momentum by Volume Shock
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_momentum'] = (df['short_term_ema'] - df['long_term_ema']) * df['volume_change']
    
    # Synthesize Price-Acceleration and Volume-Adjusted Momentum with EMA
    df['final_factor'] = df['price_acceleration'] + df['adjusted_momentum'] + df['vol_weighted_ema_diff']
    
    return df['final_factor']
```
