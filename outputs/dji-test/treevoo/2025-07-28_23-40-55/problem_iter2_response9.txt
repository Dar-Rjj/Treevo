def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(6)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(29)
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_adjusted_by_volume = (short_term_momentum - long_term_momentum) * volume_change
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_high_low_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = volume_rolling_mean * 1.5  # Example threshold
    high_volume_days = df['volume'] >= high_volume_threshold
    filtered_integrated_momentum = integrated_momentum.where(high_volume_days, 0)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = momentum_adjusted_by_volume + filtered_integrated_momentum
    return final_alpha_factor
