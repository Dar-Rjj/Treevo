```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_alpha_factor = integrated_momentum * high_volume_days
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    alpha_factor_with_return = filtered_alpha_factor * intraday_return
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    final_alpha_factor = alpha_factor_with_return * high_low_volatility
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_by_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Introduce Volume Weighted Price Momentum
    vwap_7_days = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Introduce Volume Weighted Price Oscillator
    vwap_14_days = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    vwap_oscillator = vwap_7_days - vwap_14_days.shift(7)
    
    # Synthesize Final Factor
    final_factor = final_alpha_factor + momentum_by_volume_shock + vwap_oscillator
    
    return final_factor
```
