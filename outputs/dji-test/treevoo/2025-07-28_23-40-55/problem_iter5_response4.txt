```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_integrated_momentum = integrated_momentum.where(high_volume_days, 0)
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_price_diff = df['close'] - df['close'].shift(5)
    short_term_price_momentum = short_term_price_diff.sum() / 5
    
    # Calculate Long-Term Price Momentum
    long_term_price_diff = df['close'] - df['close'].shift(20)
    long_term_price_momentum = long_term_price_diff.sum() / 20
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=5).sum() / 5
    
    # Combine Factors
    combined_factors = ema_difference * short_term_price_momentum * total_volume_change
    
    # Calculate Daily Volume Change
    relative_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_adjusted_by_volume_shock = (short_term_price_momentum - long_term_price_momentum) * relative_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_factor = filtered_integrated_momentum + momentum_adjusted_by_volume_shock
    
    return synthesized_factor
```
