```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_intraday_momentum = (high_low_range + open_close_momentum) / 2

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=30).mean()
    volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > volume_threshold

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].shift(1) - df['close'].shift(7)
    long_term_momentum = df['close'].shift(1) - df['close'].shift(30)

    # Adjust Price Momentum by Volume Shock
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_price_momentum = (short_term_momentum - long_term_momentum) * volume_change

    # Calculate High-Low Volatility
    high_low_volatility = high_low_range

    # Adjust Intraday Momentum by Volatility
    adjusted_intraday_momentum = integrated_intraday_momentum * high_low_volatility

    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)

    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)

    # Calculate Short-Term and Long-Term EMA
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    ema_difference = ema_short - ema_long

    # Adjust EMA Difference by Volume Shock
    adjusted_ema_difference = ema_difference * volume_change

    # Synthesize Final Alpha Factor
    final_factor = (
        adjusted_price_momentum + 
        adjusted_intraday_momentum + 
        price_acceleration + 
        adjusted_ema_difference
    )

    # Apply Volume Filter to the Final Factor
    final_factor = final_factor[high_volume_days]

    return final_factor
```
