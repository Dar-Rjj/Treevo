```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Calculate 5-Day Moving Average Momentum
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    ma_momentum = df['close'] - df['5_day_ma']
    combined_momentum = integrated_momentum + ma_momentum
    
    # Apply Volume and Amount Filter
    # Use Volume
    volume_mean = df['volume'].rolling(window=5).mean()
    high_volume_days = df[df['volume'] > 1.5 * volume_mean]
    
    # Use Amount
    amount_mean = df['amount'].rolling(window=5).mean()
    high_amount_days = high_volume_days[high_volume_days['amount'] > 1.5 * amount_mean]
    
    # Finalize Alpha Factor
    alpha_factor = combined_momentum[high_amount_days.index]

    return alpha_factor
```
