import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate Volume-Weighted Moving Averages
    def vwma(price, volume, span):
        return (price * volume).rolling(window=span).sum() / volume.rolling(window=span).sum()

    df['short_vwma'] = vwma(df['close'], df['volume'], 12)
    df['long_vwma'] = vwma(df['close'], df['volume'], 26)

    # Adjusted Momentum
    df['adjusted_momentum'] = (df['short_vwma'] - df['long_vwma']) * df['volume']

    return df['adjusted_momentum']
