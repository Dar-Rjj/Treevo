```python
def heuristics_v2(data):
    # Calculate daily close price change
    data['price_velocity'] = data['close'].diff()

    # Calculate price acceleration
    data['price_acceleration'] = data['price_velocity'].diff()
    
    # Drop the first two rows as they will have NaN values due to differencing
    data.dropna(inplace=True)
    
    # Return the price_acceleration as the alpha factor
    return data['price_acceleration']
```
