```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Momentum
    integrated_momentum = (intraday_range + open_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean.mean() + volume_rolling_mean.std()
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_momentum = integrated_momentum * high_volume_days
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    final_alpha_factor = (filtered_momentum * high_low_volatility)
    
    return final_alpha_factor
```
