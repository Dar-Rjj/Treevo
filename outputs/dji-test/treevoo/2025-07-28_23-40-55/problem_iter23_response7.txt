```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()

    # Adjusted Momentum
    ema_diff = short_ema - long_ema
    adjusted_momentum = ema_diff * df['volume']

    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Generate Composite Momentum Signal
    composite_momentum = adjusted_momentum * (price_velocity - price_velocity.shift(1))

    # Adjust Momentum by Volume Shock
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_composite_momentum = (short_ema - long_ema) * volume_change

    # Introduce Volume Weighted Price Momentum
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap_sum = vwap.rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    vwap_momentum = vwap_sum.diff()

    # Incorporate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=7).mean()

    # Synthesize Intermediate Alpha Factor
    intermediate_factor = (adjusted_composite_momentum + vwap_momentum) + avg_high_low_range

    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['open'] - df['close']

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=7).mean()
    high_volume_days = df['volume'] > volume_rolling_mean
    filtered_intermediate_factor = intermediate_factor[high_volume_days]

    # Calculate High-Low Volatility
    volatility = df['high'] - df['low']

    # Adjust Momentum by Volatility
    adjusted_alpha_factor = filtered_intermediate_factor / volatility

    # Incorporate Amount in the Alpha Factor
    amount_rolling_avg = df['amount'].rolling(window=7).mean()
    amount_ratio = df['amount'] / amount_rolling_avg
    final_adjusted_alpha_factor = adjusted_alpha_factor * amount_ratio

    # Incorporate Price and Volume Correlation
    price_vol_corr = df['close'].rolling(window=10).corr(df['volume'])
    final_alpha_factor = final_adjusted_alpha_factor * price_vol_corr

    return final_alpha_factor
```
