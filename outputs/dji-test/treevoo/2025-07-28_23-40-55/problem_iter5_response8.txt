```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / 5
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / 20
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change_5_days = daily_volume_change.rolling(window=5).sum()
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Combine Factors
    combined_factor = (short_term_momentum * 
                       long_term_momentum * 
                       total_volume_change_5_days * 
                       price_acceleration)
    
    return combined_factor
```
