```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Price Change from Previous Day
    df['price_change_1'] = df['close'].diff(1)
    
    # Calculate the Price Change Over a Longer Period
    df['price_change_20'] = df['close'].diff(20)
    
    # Calculate the Volume Change from Previous Day
    df['volume_change_1'] = df['volume'].diff(1)
    
    # Calculate the Average Volume Over a Fixed Period
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the Volume-Weighted Price Change
    df['vw_price_change'] = (df['close'].diff() * df['volume']).rolling(window=20).sum()
    
    # Calculate the Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the Intraday Price Movement
    df['intraday_movement'] = df['close'] - df['open']
    
    # Calculate the Directional Price Movement
    df['directional_movement'] = (df['close'] > df['open']).astype(int) - (df['close'] < df['open']).astype(int)

    # Combine all factors into a single alpha factor
    # This is a simple linear combination of the features
    df['alpha_factor'] = (
        0.3 * df['price_change_1'] +
        0.2 * df['price_change_20'] +
        0.1 * df['volume_change_1'] +
        0.1 * df['avg_volume_20'] +
        0.1 * df['vw_price_change'] +
        0.1 * df['daily_range'] +
        0.1 * df['high_low_ratio'] +
        0.1 * df['intraday_movement'] +
        0.1 * df['directional_movement']
    )
    
    return df['alpha_factor']
```
