```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span)
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Short-Term Price Momentum (5-day time frame)
    df['short_momentum'] = (df['close'] - df['close'].shift(5)) / 5
    
    # Calculate Long-Term Price Momentum (20-day time frame)
    df['long_momentum'] = (df['close'] - df['close'].shift(20)) / 20
    
    # Calculate Volume Trend (Daily Volume Change)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['total_volume_change'] = df['volume_change'].rolling(window=5).sum() / 5
    
    # Combine Factors
    df['alpha_factor'] = df['ema_diff'] * df['short_momentum'] * df['total_volume_change']
    
    # Return the final alpha factor
    return df['alpha_factor'].dropna()
```
