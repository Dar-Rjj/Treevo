import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Ratio
    df['price_movement_ratio'] = df['high'] / df['low']
    
    # Evaluate Weighted Average Volume Over Period
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    df['momentum_20d'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    def adjust_for_volume_impact(row):
        if row['volume'] > row['avg_volume_20d']:
            return row['momentum_20d'] * 1.1
        else:
            return row['momentum_20d']
    
    df['adjusted_momentum_20d'] = df.apply(adjust_for_volume_impact, axis=1)
    
    # Assess Positive vs Negative Momentum Contribution
    def assess_momentum_contribution(row):
        if row['momentum_20d'] > 0:
            return row['adjusted_momentum_20d'] * 1.5
        else:
            return row['adjusted_momentum_20d']
    
    df['final_momentum_20d'] = df.apply(assess_momentum_contribution, axis=1)
    
    # Integrate Accumulated Momentum Impact
    df['accumulated_momentum_20d'] = df['final_momentum_20d'].rolling(window=20).sum() * 0.6 + df['final_momentum_20d'].rolling(window=40).sum() * 0.4
    
    return df['accumulated_momentum_20d']
