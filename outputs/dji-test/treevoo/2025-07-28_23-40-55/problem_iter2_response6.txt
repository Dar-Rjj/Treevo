```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].pct_change(7)
    long_term_momentum = df['close'].pct_change(30)
    
    # Adjust Momentum by Volume Shock
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    momentum_adjusted_by_volume = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Evaluate Weighted Average Volume Over Period
    avg_volume_30_days = df['volume'].rolling(window=30).mean()
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['close'] - df['close'].shift(30)) / avg_volume_30_days
    
    # Assess Positive vs Negative Momentum Contribution
    positive_momentum_contribution = volume_adjusted_momentum * 1.5
    negative_momentum_contribution = volume_adjusted_momentum
    adjusted_momentum = pd.Series(index=df.index, dtype='float64')
    adjusted_momentum[df['close'] > df['close'].shift(30)] = positive_momentum_contribution
    adjusted_momentum[df['close'] <= df['close'].shift(30)] = negative_momentum_contribution
    
    # Integrate Accumulated Momentum Impact
    final_alpha = 0.4 * momentum_adjusted_by_volume + 0.6 * adjusted_momentum
    
    return final_alpha
```
