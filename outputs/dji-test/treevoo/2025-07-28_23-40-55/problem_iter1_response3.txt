```python
import pandas as pd

def heuristics(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate the 20-day average volume
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Adjust for Volume Impact
    def adjust_for_volume(row):
        if row['volume'] > row['avg_volume_20d']:
            return row['price_momentum'] * 1.1
        else:
            return row['price_momentum']
    
    df['adjusted_momentum'] = df.apply(adjust_for_volume, axis=1)
    
    return df['adjusted_momentum']
```
