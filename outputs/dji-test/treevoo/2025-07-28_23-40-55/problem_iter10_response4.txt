```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Exponential Moving Average (EMA)
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12_days'] = df['volume'].rolling(window=12).mean()
    df['vol_weighted_ema_diff'] = df['ema_diff'] * df['avg_volume_12_days']
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['vol_weighted_ema_diff'] * df['volume']
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['close'] - df['close'].shift(1)
    df['velocity_change'] = df['price_acceleration'] - df['price_acceleration'].shift(1)
    
    # Calculate Momentum by Volume Shock
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['mom_by_vol_shock'] = (df['ema_short'] - df['ema_long']) * df['volume_change']
    
    # Synthesize Momentum and Volume-Adjusted Price-Acceleration
    df['alpha_factor'] = df['momentum_signal'] + df['mom_by_vol_shock'] + df['velocity_change']
    
    return df['alpha_factor']
```
