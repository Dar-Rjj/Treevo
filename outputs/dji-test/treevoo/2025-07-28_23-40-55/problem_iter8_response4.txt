```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Price Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    df['vol_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['adjusted_momentum'] = df.apply(lambda row: row['momentum'] * 1.1 if row['volume'] > row['vol_20d_avg'] else row['momentum'], axis=1)
    
    # Adjust for Amount Impact
    df['amount_20d_avg'] = df['amount'].rolling(window=20).mean()
    df['final_momentum'] = df.apply(lambda row: row['adjusted_momentum'] * 1.05 if row['amount'] > row['amount_20d_avg'] else row['adjusted_momentum'], axis=1)
    
    # Calculate Price Velocity
    df['price_velocity'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)
    
    # Calculate Short-Term EMA (12-day span)
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Generate Composite Momentum Signal
    df['composite_momentum_signal'] = df['ema_diff'] * df['volume']
    
    # Incorporate Integrated Alpha Factors
    df['alpha_factor'] = df['final_momentum'] + df['composite_momentum_signal'] + df['price_acceleration']
    
    return df['alpha_factor']
```
