```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term EMA
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # EMA Difference
    df['EMA_diff'] = df['EMA_12'] - df['EMA_26']
    
    # Volume-Weighted EMA Difference
    df['avg_volume_12'] = df['volume'].rolling(window=12).mean()
    df['vol_weighted_EMA_diff'] = df['EMA_diff'] * df['avg_volume_12']
    
    # Momentum Signal
    df['momentum_signal'] = df['vol_weighted_EMA_diff'] * df['volume']
    
    # Price Velocity
    df['price_velocity'] = df['close'] - df['close'].shift(1)
    
    # Price Acceleration
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)
    
    # Short-Term and Long-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    df['long_term_momentum'] = df['close'] - df['close'].shift(30)
    
    # Daily Volume Change
    df['daily_vol_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['vol_adjusted_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['daily_vol_change']
    
    # Synthesize Momentum and Volume-Adjusted Price-Acceleration
    df['synthesized_factor'] = df['momentum_signal'] + df['price_acceleration'] + df['vol_adjusted_momentum']
    
    # Volume Weighted Average Price
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['VWAP_7'] = df['VWAP'].rolling(window=7).sum()
    
    # Incorporate Volume Weighted Price Momentum
    df['final_enhanced_factor'] = df['synthesized_factor'] + df['VWAP_7']
    
    return df['final_enhanced_factor']
```
