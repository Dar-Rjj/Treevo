```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = high_low_range + open_close_momentum
    
    # Apply Volume Filter
    rolling_mean_volume = df['volume'].rolling(window=10).mean()
    volume_threshold = 1.5 * rolling_mean_volume
    high_volume_days = df['volume'] > volume_threshold
    
    # Filter Integrated Momentum by High Volume Days
    filtered_integrated_momentum = integrated_momentum[high_volume_days].reindex(df.index, fill_value=0)
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    long_term_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Adjust Price Momentum by Volume Shock
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_price_momentum = (short_term_momentum - long_term_momentum) * volume_change
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Intraday Momentum by Volatility
    high_low_volatility = df['high'] - df['low']
    adjusted_intraday_momentum = filtered_integrated_momentum * high_low_volatility
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Calculate Short-Term and Long-Term Exponential Moving Average (EMA)
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = ema_short - ema_long
    
    # Adjust EMA Difference by Volume Shock
    adjusted_ema_difference = ema_difference * volume_change
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_price_momentum + 
                          adjusted_intraday_momentum + 
                          price_acceleration + 
                          adjusted_ema_difference)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    final_alpha_factor *= intraday_return
    
    # Refine the Final Factor
    final_alpha_factor = final_alpha_factor.where(high_volume_days, 0)
    
    # Adjust Final Alpha Factor by High-Low Volatility
    final_alpha_factor *= high_low_volatility
    
    return final_alpha_factor
```
