```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Velocity'] = df['close'].diff()

    # Calculate Price Acceleration
    df['Acceleration'] = df['Velocity'].diff()

    # Calculate Short-Term EMA (12-day span)
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()

    # Calculate Long-Term EMA (26-day span)
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    df['EMA_Diff'] = df['EMA_12'] - df['EMA_26']

    # Generate Composite Momentum Signal
    df['Composite_Momentum'] = df['EMA_Diff'] * df['volume'] + df['Acceleration']

    # Calculate Volume-Weighted EMA Difference
    df['Volume_Weighted_EMA_Diff'] = df['EMA_Diff'] * df['volume'].rolling(window=12).mean()

    # Adjust Momentum by Volume Shock
    df['Volume_Change'] = df['volume'].pct_change()
    df['Adjusted_Momentum'] = df['Composite_Momentum'] * df['Volume_Change']

    # Synthesize Price-Acceleration and Volume-Adjusted Momentum with EMA
    df['Synthesized_Momentum'] = df['Acceleration'] + df['Adjusted_Momentum'] + df['Volume_Weighted_EMA_Diff']

    # Calculate Intraday High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    df['Open_Close_Momentum'] = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    df['Integrated_Momentum'] = df['High_Low_Range'] + df['Open_Close_Momentum']

    # Apply Volume Filter
    df['Volume_Rolling_Mean'] = df['volume'].rolling(window=12).mean()
    high_volume_threshold = df['Volume_Rolling_Mean'].quantile(0.8)
    df['Filtered_Integrated_Momentum'] = df['Integrated_Momentum'].where(df['volume'] > high_volume_threshold, 0)

    # Calculate High-Low Volatility
    df['Volatility'] = df['high'] - df['low']

    # Adjust Momentum by Volatility
    df['Adjusted_Alpha'] = df['Filtered_Integrated_Momentum'] / df['Volatility']

    # Incorporate Amount in the Alpha Factor
    df['Average_Amount'] = df['amount'].rolling(window=12).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']

    # Finalize Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Alpha'] * df['Amount_Ratio']

    return df['Final_Alpha_Factor']
```
