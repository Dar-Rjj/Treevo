```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = high_low_range + open_close_momentum

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = 1.5 * volume_rolling_mean
    high_volume_days = df['volume'] > high_volume_threshold

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = (df['close'] / df['close'].shift(1)) - 1
    long_term_momentum = (df['close'] / df['close'].shift(30)) - 1

    # Adjust Price Momentum by Volume Shock
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_change

    # Adjust Intraday Momentum by Volatility
    volatility = df['high'] - df['low']
    adjusted_intraday_momentum = integrated_momentum * volatility

    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)

    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)

    # Calculate EMA
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema

    # Adjust EMA Difference by Volume Shock
    adjusted_ema_difference = ema_difference * volume_change

    # Synthesize Final Alpha Factor
    final_alpha_factor = adjusted_momentum + adjusted_intraday_momentum + price_acceleration + adjusted_ema_difference

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Incorporate Intraday Return
    final_alpha_factor *= intraday_return

    # Refine the Final Factor for High Volume Days
    final_alpha_factor[~high_volume_days] = 0

    # Adjust Momentum by High-Low Volatility
    final_alpha_factor *= volatility

    return final_alpha_factor
```
