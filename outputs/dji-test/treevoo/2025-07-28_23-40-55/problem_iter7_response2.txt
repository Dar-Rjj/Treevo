```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    df['volume_avg_20'] = df['volume'].rolling(window=20).mean()
    df['momentum_volume_adj'] = df.apply(lambda row: row['momentum'] * 1.1 if row['volume'] > row['volume_avg_20'] else row['momentum'], axis=1)
    
    # Adjust for Amount Impact
    df['amount_avg_20'] = df['amount'].rolling(window=20).mean()
    df['factor'] = df.apply(lambda row: row['momentum_volume_adj'] * 1.05 if row['amount'] > row['amount_avg_20'] else row['momentum_volume_adj'], axis=1)

    return df['factor']
```
