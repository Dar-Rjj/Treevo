```python
import pandas as pd

def heuristics_v2(df):
    # Calculate intraday volatility measures
    df['daily_range'] = df['high'] - df['low']
    df['intraday_true_range'] = df[['high', 'close', 'open']].apply(lambda x: max(x['high'] - x['close'], x['high'] - x['open'], x['close'] - x['open']), axis=1)
    
    # Analyze trading activity through volume
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_rel_to_ma'] = df['volume'] / df['volume_20_day_ma']
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * 100
    
    # Construct indicators based on price movements
    df['C_minus_O'] = df['close'] - df['open']
    df['C_minus_L_div_H_minus_L'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Incorporate amount traded
    df['amount_per_share'] = df['amount'] / df['volume']
    df['amount_close_ratio'] = df['amount'] / df['close']
    
    # Explore momentum in price and volume
    df['momentum_price_10'] = df['close'].pct_change(periods=10)
    df['momentum_volume_10'] = df['volume'].pct_change(periods=10)
    
    # Utilize technical patterns
    df['doji_pattern'] = ((df['close'] - df['open']).abs() / (df['high'] - df['low']) < 0.1) & (df['high'] - df['low'] > 0.5 * df['close'])
    df['hammer_pattern'] = (df['close'] > (df['open'] + df['close']) / 2) & (df['high'] - df['low'] > 3 * (df['open'] - df['close']))
    
    # Leverage historical returns
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Integrate trend indicators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_crossover'] = df['SMA_50'] > df['SMA_200']
    
    # Accumulation/Distribution Line
    df['A_D_Line'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['A_D_Line'] = df['A_D_Line'].cumsum()
    df['A_D_Oscillator'] = df['A_D_Line'] - df['A_D_Line'].shift(1)
    
    # Combine all the features into a single factor
    alpha_factor = (
        0.1 * df['daily_range'] +
        0.1 * df['intraday_true_range'] +
        0.1 * df['volume_rel_to_ma'] +
        0.1 * df['volume_spike'] +
        0.1 * df['C_minus_L_div_H_minus_L'] +
        0.1 * df['amount_per_share'] +
        0.1 * df['momentum_price_10'] +
        0.1 * df['momentum_volume_10'] +
        0.1 * df['SMA_crossover'] +
        0.1 * df['A_D_Oscillator']
    )
    
    return alpha_factor
```
