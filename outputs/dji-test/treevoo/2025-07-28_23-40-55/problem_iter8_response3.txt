```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_integrated_momentum = integrated_momentum[high_volume_days]
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] - df['close'].shift(7)).rolling(window=7).sum() / 7
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(25)).rolling(window=25).sum() / 25
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'].pct_change()
    total_volume_change = daily_volume_change.rolling(window=7).sum()
    
    # Calculate Weighted EMA Difference
    weighted_ema_difference = ema_difference * 1.5
    
    # Combine Factors
    combined_factor = weighted_ema_difference * short_term_momentum * total_volume_change
    
    # Calculate Daily Volume Change
    relative_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * relative_volume_change
    
    # Synthesize Momentum and Volume Shock
    final_alpha_factor = filtered_integrated_momentum + adjusted_momentum
    
    return final_alpha_factor
```
