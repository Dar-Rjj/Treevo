```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Price_Velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['Price_Acceleration'] = df['Price_Velocity'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['EMA_Short_Term'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['EMA_Long_Term'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['EMA_Difference'] = df['EMA_Short_Term'] - df['EMA_Long_Term']
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = (df['close'].shift(1) - df['close'].shift(7)) / df['close'].shift(1)
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = (df['close'].shift(1) - df['close'].shift(30)) / df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['Adjusted_Momentum'] = (df['Short_Term_Momentum'] - df['Long_Term_Momentum']) * df['Volume_Change']
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['Open_to_Close_Momentum'] = (df['open'] - df['close']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['Integrated_Momentum'] = df['Intraday_High_Low_Range'] + df['Open_to_Close_Momentum']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Velocity'] + 
        df['Price_Acceleration'] + 
        df['Adjusted_Momentum'] + 
        df['EMA_Difference'] + 
        df['Integrated_Momentum']
    )
    
    # Apply Volume Filter
    rolling_mean_volume = df['volume'].rolling(window=20).mean()
    high_volume_threshold = rolling_mean_volume * 1.5
    df['High_Volume'] = df['volume'] > high_volume_threshold
    
    # Refine the Final Factor to only include high volume days
    df['Refined_Final_Factor'] = df['Final_Alpha_Factor'] * df['High_Volume']
    
    return df['Refined_Final_Factor'].dropna()
```
