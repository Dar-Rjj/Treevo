```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    price_velocity = df['close'].diff()

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    long_term_momentum = df['close'] - df['close'].shift(30)

    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_change

    # Introduce Volume Weighted Price Momentum
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap_momentum = vwap.rolling(window=7).sum() / df['volume'].rolling(window=7).sum()

    # Incorporate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=7).mean()

    # Integrate Price Action with Open and Close
    open_close_spread = df['open'] - df['close']

    # Synthesize Final Factor
    final_factor = (adjusted_momentum + vwap_momentum + avg_high_low_range + open_close_spread)

    return final_factor
```
