import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean * 1.5  # Example threshold
    high_volume_days = df['volume'] >= high_volume_threshold
    filtered_integrated_momentum = integrated_momentum[high_volume_days]
    
    # Calculate Short-Term EMA
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = (df['close'] - df['close'].shift(7)).fillna(0)
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = (df['close'] - df['close'].shift(25)).fillna(0)
    
    # Calculate Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(7)).fillna(0)
    volume_trend = volume_trend / 7  # Adjust for the number of days
    
    # Combine Factors
    combined_factor = (ema_difference * short_term_price_momentum * volume_trend).fillna(0)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor
    
    return final_alpha_factor
