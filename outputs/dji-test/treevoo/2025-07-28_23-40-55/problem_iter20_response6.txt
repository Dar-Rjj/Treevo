```python
def heuristics_v2(df):
    # Calculate the price range for each day
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) / price_range
    
    # Calculate the volume weighted average price
    vwap = (df['close'] + df['high'] + df['low']) / 3 * df['volume']
    vwap = vwap.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine the factors
    factor_value = buying_pressure * vwap
    
    return factor_value
```
