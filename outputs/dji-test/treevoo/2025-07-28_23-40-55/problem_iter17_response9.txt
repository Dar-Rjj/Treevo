```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = (df['high_low_range'] + df['open_close_momentum']) / 2
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=12).mean()
    high_volume_threshold = df['volume_rolling_mean'].quantile(0.75)
    df['high_volume_days'] = (df['volume'] > high_volume_threshold).astype(int)
    df['filtered_integrated_momentum'] = df['integrated_momentum'] * df['high_volume_days']
    
    # Calculate Short-Term EMA
    df['short_term_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    df['long_term_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_term_ema'] - df['long_term_ema']
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12_days'] = df['volume'].rolling(window=12).mean()
    df['volume_weighted_ema_diff'] = df['ema_diff'] * df['avg_volume_12_days']
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['volume_weighted_ema_diff'] * df['volume']
    
    # Calculate Price Acceleration
    df['price_acceleration'] = (df['close'] - df['close'].shift(1)) - (df['close'].shift(1) - df['close'].shift(2))
    
    # Calculate Short-Term Price Momentum
    df['short_term_price_momentum'] = df['close'].diff().rolling(window=7).sum() / 7
    
    # Calculate Long-Term Price Momentum
    df['long_term_price_momentum'] = df['close'].diff().rolling(window=25).sum() / 25
    
    # Calculate Volume Trend
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend'] = df['daily_volume_change'].rolling(window=7).sum()
    
    # Calculate Weighted EMA Difference
    df['weighted_ema_diff'] = df['ema_diff'] * 1.8
    
    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_term_price_momentum'] - df['long_term_price_momentum']) * df['volume_trend']
    
    # Synthesize Momentum and Volume Shock
    df['synthesized_momentum'] = df['filtered_integrated_momentum'] + df['adjusted_momentum']
    
    # Combine Momentum, Volume-Weighted EMA Difference, and Enhanced Price-Acceleration
    df['combined_factor'] = df['momentum_signal'] + df['volume_weighted_ema_diff'] + df['price_acceleration']
    
    # Final Combination
    df['final_combination'] = df['combined_factor'] * 1.5 * df['short_term_price_momentum'] * df['volume_trend']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['synthesized_momentum'] * df['close']) + df['final_combination']
    
    return df['alpha_factor']
```
