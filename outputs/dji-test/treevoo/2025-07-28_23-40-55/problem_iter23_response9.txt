```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Adjusted Momentum
    momentum = (short_ema - long_ema) * df['volume']
    
    # Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Open-to-Close Momentum
    otc_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + otc_momentum
    
    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=10).mean()
    high_volume_days = df['volume'] > 1.5 * volume_mean
    filtered_momentum = integrated_momentum.where(high_volume_days, 0)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    final_filtered_momentum = filtered_momentum * intraday_return
    
    # High-Low Volatility
    hl_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    adjusted_momentum = final_filtered_momentum * hl_volatility
    
    # Price Velocity
    price_velocity = df['close'].diff()
    
    # Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum_vol_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Volume Weighted Price Momentum
    vw_price_momentum = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Volume Weighted Price Oscillator
    vwpo_t6_t = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    vwpo_t14_t7 = (df['close'] * df['volume']).rolling(window=7, center=True).sum().shift(7) / df['volume'].rolling(window=7, center=True).sum().shift(7)
    vwpo = vwpo_t6_t - vwpo_t14_t7
    
    # Synthesize Final Factor
    final_factor = adjusted_momentum + vw_price_momentum + vwpo
    
    return final_factor
```
