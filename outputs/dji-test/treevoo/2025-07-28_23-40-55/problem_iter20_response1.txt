```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = high_low_range + open_close_momentum
    
    # Apply Volume Filter
    rolling_mean_volume = df['volume'].rolling(window=20).mean()
    high_volume_days = df[df['volume'] > 1.5 * rolling_mean_volume]
    filtered_alpha_factor = integrated_momentum[high_volume_days.index]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    integrated_momentum_return = filtered_alpha_factor * intraday_return
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    preliminary_alpha_factor = integrated_momentum_return * high_low_volatility
    
    # Calculate Short-Term EMA (7-day span, Close price)
    short_ema = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate Long-Term EMA (25-day span, Close price)
    long_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_ema - long_ema
    
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = (df['close'] - df['close'].shift(7)) / 7
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = (df['close'] - df['close'].shift(25)) / 25
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=7).sum() / 7
    
    # Calculate Weighted EMA Difference
    weighted_ema_difference = ema_difference * 1.8
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_price_momentum - long_term_price_momentum) * total_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = preliminary_alpha_factor + adjusted_momentum
    
    # Calculate EMAs
    short_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    long_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # EMA Difference
    ema_difference_long = short_ema_10 - long_ema_30
    
    # Adjusted Momentum Signal
    adjusted_momentum_signal = ema_difference_long * df['volume'] + df['high'] - df['low']
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum * df['close']) + adjusted_momentum_signal
    return final_alpha_factor
```
