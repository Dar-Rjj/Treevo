```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term EMA
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_vol_12d = df['volume'].rolling(window=12).mean()
    vol_weighted_ema_diff = ema_diff * avg_vol_12d
    
    # Calculate Momentum Signal
    momentum_signal = vol_weighted_ema_diff * df['volume']
    
    # Calculate Intraday High-Low Range and Open-to-Close Momentum
    intraday_range = df['high'] - df['low']
    open_close_mom = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_close_mom) / 2
    
    # Apply Volume Filter
    volume_roll_mean = df['volume'].rolling(window=12).mean()
    high_volume_threshold = volume_roll_mean.quantile(0.75)
    filtered_integrated_momentum = integrated_momentum.where(df['volume'] > high_volume_threshold, 0)
    
    # Adjust Momentum by Volatility
    high_low_volatility = df['high'] - df['low']
    volatility_adjusted_momentum = filtered_integrated_momentum * high_low_volatility
    
    # Calculate Price Acceleration
    velocity_t = df['close'] - df['close'].shift(1)
    velocity_t_minus_1 = df['close'].shift(1) - df['close'].shift(2)
    price_acceleration = velocity_t - velocity_t_minus_1
    
    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']
    
    # Calculate Short-Term and Long-Term Momentum
    short_mom = df['close'] - df['close'].shift(7)
    long_mom = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_mom - long_mom) * volume_change
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = momentum_signal + adjusted_price_acceleration + volatility_adjusted_momentum + adjusted_momentum
    
    return final_alpha_factor
```
