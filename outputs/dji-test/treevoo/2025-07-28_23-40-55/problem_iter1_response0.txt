```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate 10-Day Moving Average of High-Low Spread
    df['10_day_ma_high_low'] = df['high_low_spread'].rolling(window=10).mean()
    
    # Calculate 30-Day Moving Average of High-Low Spread
    df['30_day_ma_high_low'] = df['high_low_spread'].rolling(window=30).mean()
    
    # Derive Momentum by Comparing Averages
    df['momentum'] = df['10_day_ma_high_low'] - df['30_day_ma_high_low']
    
    # Calculate 10-Day Average Volume
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Apply Volume Weighted Adjustment
    df['alpha_factor'] = df['momentum'] * (df['volume'] / df['10_day_avg_volume'])
    
    return df['alpha_factor']
```
