import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Weighted EMAs
    short_term_wema = df['close'] * df['volume'].rolling(window=12).mean()
    long_term_wema = df['close'] * df['volume'].rolling(window=26).mean()

    # WEMA Difference
    wema_diff = short_term_wema - long_term_wema

    # Volume-Averaged over 26 days
    avg_volume_26 = df['volume'].rolling(window=26).mean()

    # Weighted Momentum Signal
    weighted_momentum_signal = wema_diff * (df['volume'] / avg_volume_26)

    # Volume-Adjusted Volatility
    volatility = df['high'] - df['low']
    vol_adj_volatility = volatility.rolling(window=26).std() * (df['volume'] / avg_volume_26)

    # Combined Alpha Factor
    combined_alpha_factor = weighted_momentum_signal / vol_adj_volatility

    # Daily Close Price Change
    daily_close_change = df['close'].diff(1)

    # Momentum Differentials
    short_term_momentum = df['close'].shift(1) - df['close'].shift(7)
    long_term_momentum = df['close'].shift(1) - df['close'].shift(30)
    momentum_differential = short_term_momentum - long_term_momentum

    # Daily Volume Change
    daily_volume_change = (df['volume'].diff(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    adjusted_momentum = momentum_differential * daily_volume_change

    # Volume Weighted Price Momentum
    vw_price = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vw_price_sum = vw_price.rolling(window=7).sum()

    # Synthesize Final Factor
    synthesized_final_factor = adjusted_momentum + vw_price_sum

    # Final Enhanced Factor
    final_enhanced_factor = synthesized_final_factor + combined_alpha_factor

    return final_enhanced_factor
