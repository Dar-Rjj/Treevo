```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_intraday_momentum = intraday_range + open_close_momentum
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_days = (df['volume'] > volume_rolling_mean * 1.5)
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Price Momentum by Volume Shock
    price_momentum_adjusted = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Calculate High-Low Volatility
    high_low_volatility = intraday_range
    
    # Adjust Intraday Momentum by Volatility
    intraday_momentum_adjusted = integrated_intraday_momentum * high_low_volatility
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Calculate Short-Term and Long-Term EMA
    ema_short_term = df['close'].ewm(span=12, adjust=False).mean()
    ema_long_term = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = ema_short_term - ema_long_term
    
    # Adjust EMA Difference by Volume Shock
    ema_difference_adjusted = ema_difference * daily_volume_change
    
    # Calculate Intraday Price Reversal
    intraday_price_reversal = df['close'] - df['close'].shift(1)
    
    # Synthesize Final Alpha Factor
    final_factor = (price_momentum_adjusted 
                    + intraday_momentum_adjusted 
                    + price_acceleration 
                    + ema_difference_adjusted 
                    + intraday_price_reversal)
    
    # Apply Volume Filter to the final factor
    final_factor_filtered = final_factor[high_volume_days]
    
    return final_factor_filtered
```
