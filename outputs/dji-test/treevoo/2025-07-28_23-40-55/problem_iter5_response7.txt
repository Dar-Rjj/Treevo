```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term and Long-Term Exponential Moving Averages (EMA)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_difference'] = short_ema - long_ema
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Combine and Adjust for Volume Impact
    df['combined_value'] = df['ema_difference'] * df['price_momentum']
    
    # Calculate the average volume over the past 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Adjust based on Volume
    df['volume_adjusted_combined_value'] = df.apply(
        lambda row: row['combined_value'] * 1.1 if row['volume'] > avg_volume.loc[row.name] else row['combined_value'], 
        axis=1
    )
    
    # Synthesize Combined Alpha Factor
    df['alpha_factor'] = df['price_velocity'] + df['price_acceleration'] + df['volume_adjusted_combined_value']
    
    return df['alpha_factor']
```
