```python
import pandas as pd

def heuristics_v2(df):
    # Calculate EMAs
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # EMA Difference
    ema_diff = short_ema - long_ema
    
    # Adjusted Momentum Signal
    adj_mom_signal = ema_diff * df['volume'] + df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = adj_mom_signal * df['close']
    
    return alpha_factor
```
