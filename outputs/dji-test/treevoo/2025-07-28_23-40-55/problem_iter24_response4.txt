```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_intraday_momentum = intraday_range + open_close_momentum

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=5).mean()
    high_volume_threshold = volume_rolling_mean.quantile(0.7)
    high_volume_days = df[df['volume'] > high_volume_threshold].index

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(6)
    long_term_momentum = df['close'] - df['close'].shift(29)

    # Adjust Price Momentum by Volume Shock
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_price_momentum = (short_term_momentum - long_term_momentum) * volume_change

    # Adjust Intraday Momentum by Volatility
    volatility = df['high'] - df['low']
    adjusted_intraday_momentum = integrated_intraday_momentum * volatility

    # Calculate Price Velocity and Acceleration
    price_velocity = df['close'] - df['close'].shift(1)
    price_acceleration = price_velocity - price_velocity.shift(1)

    # Calculate Short-Term and Long-Term EMA
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference and Adjust by Volume Shock
    ema_difference = ema_short - ema_long
    adjusted_ema_difference = ema_difference * volume_change

    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_price_momentum + adjusted_intraday_momentum + price_acceleration + adjusted_ema_difference)

    # Incorporate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor *= intraday_return

    # Refine the Final Factor: Only Include High Volume Days
    final_alpha_factor = final_alpha_factor.loc[high_volume_days]

    # Adjust Momentum by High-Low Volatility
    final_alpha_factor *= volatility

    return final_alpha_factor
```
