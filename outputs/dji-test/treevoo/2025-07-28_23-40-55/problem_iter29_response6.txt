```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date
    """
    
    # Calculate the price range for each day
    price_range = df['high'] - df['low']
    
    # Calculate the midpoint of the high and low prices
    midpoint_price = (df['high'] + df['low']) / 2
    
    # Calculate the normalized closing price relative to the midpoint
    norm_close_midpoint = (df['close'] - midpoint_price) / price_range
    
    # Calculate the volume-weighted close price
    vol_weighted_close = df['close'] * df['volume']
    
    # Calculate the moving average of the volume-weighted close price
    ma_vol_weighted_close = vol_weighted_close.rolling(window=5).mean()
    
    # Calculate the difference between the volume-weighted close price and its moving average
    diff_vol_ma = vol_weighted_close - ma_vol_weighted_close
    
    # Calculate the final alpha factor
    alpha_factor = norm_close_midpoint * diff_vol_ma
    
    return alpha_factor
```
