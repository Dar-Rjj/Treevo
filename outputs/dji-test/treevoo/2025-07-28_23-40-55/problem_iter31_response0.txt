```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_to_close_momentum
    
    # Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].shift(1) - df['close'].shift(7)
    long_term_momentum = df['close'].shift(1) - df['close'].shift(30)
    
    # Adjust Price Momentum by Volume Shock
    volume_shock = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    price_momentum_adjusted = (short_term_momentum - long_term_momentum) * volume_shock
    
    # Daily Volume Change
    daily_volume_change = volume_shock
    
    # Exponential Moving Averages
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # EMA Difference
    ema_difference = ema_12 - ema_26
    
    # Adjust EMA Difference by Volume Shock
    ema_difference_adjusted = ema_difference * daily_volume_change
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = price_momentum_adjusted + integrated_momentum + ema_difference_adjusted
    
    # Refine the Final Factor (Only Include High Volume Days)
    high_volume_days = df['volume'] > df['volume'].rolling(window=30).mean() * 1.5
    final_alpha_factor[~high_volume_days] = np.nan
    
    return final_alpha_factor
```
