import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_close_momentum
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    is_high_volume = df['volume'] > high_volume_threshold
    filtered_alpha_factor = integrated_momentum[is_high_volume]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    incorporated_momentum = filtered_alpha_factor * intraday_return[is_high_volume]
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    adjusted_momentum = incorporated_momentum * high_low_volatility[is_high_volume]
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(periods=7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(periods=30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_adjusted_by_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Introduce Volume Weighted Price Momentum
    volume_weighted_price = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    volume_weighted_price_momentum = volume_weighted_price.pct_change()
    
    # Introduce Volume Weighted Price Oscillator
    volume_weighted_price_oscillator = volume_weighted_price - volume_weighted_price.shift(7)
    
    # Introduce Volume-Weighted Price Momentum Ratio
    volume_weighted_price_momentum_ratio = volume_weighted_price / volume_weighted_price.shift(7)
    
    # Synthesize Final Factor
    final_factor = (
        adjusted_momentum +
        volume_weighted_price_momentum +
        volume_weighted_price_oscillator +
        volume_weighted_price_momentum_ratio
    )
    
    return final_factor
