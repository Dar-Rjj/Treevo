```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Volume-Weighted EMA Difference
    df['volume_weighted_ema_diff'] = df['ema_diff'] * df['volume'].rolling(window=12).mean()
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['volume_weighted_ema_diff'] * df['volume']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_range'] + df['open_to_close_momentum']
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=12).mean()
    high_volume_threshold = df['volume_rolling_mean'].quantile(0.75)
    df['filtered_alpha_factor'] = df.apply(lambda x: x['integrated_momentum'] if x['volume'] > high_volume_threshold else 0, axis=1)
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    df['volatility_adjusted_momentum'] = df['filtered_alpha_factor'] * df['high_low_volatility']
    
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].pct_change(periods=7)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].pct_change(periods=30)
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['volume_adjusted_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']
    
    # Synthesize Momentum, Volume-Adjusted Price-Acceleration, and Volatility-Adjusted Momentum
    df['final_factor'] = df['momentum_signal'] + df['price_acceleration'] * df['volume_change'] + df['volatility_adjusted_momentum']
    
    return df['final_factor']
```
