```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the short-term (12-day) EMA using close price
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate the long-term (26-day) EMA using close price
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the EMA difference
    ema_diff = short_ema - long_ema
    
    # Calculate the momentum signal by multiplying the EMA difference with volume
    momentum_signal = ema_diff * df['volume']
    
    return momentum_signal
```
