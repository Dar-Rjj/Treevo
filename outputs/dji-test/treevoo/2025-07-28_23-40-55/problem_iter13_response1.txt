```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Close Price Difference (Price Velocity)
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Short-Term Exponential Moving Average (EMA) with 12-day span
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA) with 26-day span
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Calculate Price Momentum over 20 days
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate average volume over the past 20 days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Adjust for Volume Impact
    df['volume_adjusted_ema_diff'] = df['ema_diff'] * df['price_momentum']
    df['volume_adjusted_ema_diff'] = df['volume_adjusted_ema_diff'].where(df['volume'] <= df['avg_volume_20'], 
                                                                          df['volume_adjusted_ema_diff'] * 1.1)
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12'] = df['volume'].rolling(window=12).mean()
    df['volume_weighted_ema_diff'] = df['ema_diff'] * df['avg_volume_12']
    
    # Adjust Momentum by Volume Shock
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    df['adjusted_momentum'] = (df['ema_short'] - df['ema_long']) * df['volume_change']
    
    # Synthesize Combined Alpha Factor
    df['alpha_factor'] = (df['price_velocity'] + 
                          df['volume_adjusted_ema_diff'] + 
                          df['volume_weighted_ema_diff'] + 
                          df['adjusted_momentum'])
    
    return df['alpha_factor']
```
