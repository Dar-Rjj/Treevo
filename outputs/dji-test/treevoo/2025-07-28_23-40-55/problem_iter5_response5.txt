import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Ratio
    price_movement_ratio = df['high'] / df['low']

    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']

    # Evaluate Weighted Average Volume Over Period
    avg_volume_20_days = df['volume'].rolling(window=20).mean()

    # Calculate 10-Day Moving Average of High-Low Spread
    ma_high_low_spread_10 = high_low_spread.rolling(window=10).mean()

    # Calculate 30-Day Moving Average of High-Low Spread
    ma_high_low_spread_30 = high_low_spread.rolling(window=30).mean()

    # Derive Momentum from Close Prices
    momentum_close = df['close'] - df['close'].shift(20)

    # Derive Momentum by Comparing Averages
    momentum_avg = ma_high_low_spread_10 - ma_high_low_spread_30

    # Adjust for Volume Impact
    volume_adjustment = df['volume'] > avg_volume_20_days
    momentum_avg[volume_adjustment] *= 1.1

    # Apply Volume-Weighted Adjustment to Combined Momentum
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    combined_momentum = (momentum_close + momentum_avg) * (df['volume'] / avg_volume_10_days)

    # Assess Positive vs Negative Momentum Contribution
    combined_momentum[momentum_close > 0] *= 1.5

    # Integrate Accumulated Momentum Impact
    factor_values = combined_momentum

    return factor_values
