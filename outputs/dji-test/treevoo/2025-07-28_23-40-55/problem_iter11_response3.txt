```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Price_Velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['Price_Acceleration'] = df['Price_Velocity'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['Short_Term_EMA'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['Long_Term_EMA'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['EMA_Difference'] = df['Short_Term_EMA'] - df['Long_Term_EMA']
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['Open_to_Close_Momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['Integrated_Momentum'] = df['Intraday_High_Low_Range'] + df['Open_to_Close_Momentum']
    
    # Calculate Short-Term Price Momentum (5-day)
    df['Short_Term_Price_Momentum'] = df['close'].diff(periods=5).rolling(window=5).sum() / 5
    
    # Calculate Long-Term Price Momentum (20-day)
    df['Long_Term_Price_Momentum'] = df['close'].diff(periods=20).rolling(window=20).sum() / 20
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['Adjusted_Momentum'] = (df['Short_Term_Price_Momentum'] - df['Long_Term_Price_Momentum']) * df['Daily_Volume_Change']
    
    # Apply Volume Filter
    df['Volume_Rolling_Mean'] = df['volume'].rolling(window=20).mean()
    df['High_Volume_Days'] = df['volume'] > df['Volume_Rolling_Mean']
    df['Filtered_Adjusted_Momentum'] = df['Adjusted_Momentum'] * df['High_Volume_Days']
    
    # Combine and Adjust for Volume Impact
    df['Volume_Comparison'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_EMA_Difference'] = df['EMA_Difference'] * df['Volume_Comparison'].apply(lambda x: 1.1 if x > 1 else 1)
    
    # Calculate Volume-Weighted EMA Difference
    df['Volume_Weighted_EMA_Difference'] = df['EMA_Difference'] * df['volume'].rolling(window=12).mean()
    
    # Synthesize Combined Alpha Factor
    df['Alpha_Factor'] = (
        df['Price_Velocity'] +
        df['Price_Acceleration'] +
        df['Volume_Adjusted_EMA_Difference'] +
        df['Filtered_Adjusted_Momentum'] +
        df['Volume_Weighted_EMA_Difference']
    )
    
    return df['Alpha_Factor']
```
