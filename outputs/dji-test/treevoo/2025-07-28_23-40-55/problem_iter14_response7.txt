import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Price_Velocity'] = df['close'] - df['close'].shift(1)
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = (df['close'] / df['close'].shift(7) - 1) * 100
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = (df['close'] / df['close'].shift(30) - 1) * 100
    
    # Calculate Daily Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['Adjusted_Momentum'] = (df['Short_Term_Momentum'] - df['Long_Term_Momentum']) * df['Volume_Change']
    
    # Introduce Volume Weighted Price Momentum
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['VWAP_7D_Sum'] = df['VWAP'].rolling(window=7).sum()
    
    # Incorporate High-Low Range Volatility
    df['High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range_7D'] = df['High_Low_Range'].rolling(window=7).mean()
    
    # Adjust for Amount Impact
    df['Amount_20D_Avg'] = df['amount'].rolling(window=20).mean()
    df['Adjusted_Momentum_with_Amount'] = df['Adjusted_Momentum'].where(df['amount'] > df['Amount_20D_Avg'], df['Adjusted_Momentum'] * 1.1)
    
    # Calculate Price Acceleration
    df['Price_Acceleration'] = df['Price_Velocity'] - df['Price_Velocity'].shift(1)
    
    # Calculate EMA
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['EMA_Diff'] = df['EMA_12'] - df['EMA_26']
    
    # Generate Composite Momentum Signal
    df['Composite_Momentum_Signal'] = df['EMA_Diff'] * df['volume']
    
    # Incorporate Integrated Alpha Factors
    df['Integrated_Alpha_Factors'] = df['Adjusted_Momentum_with_Amount'] + df['Composite_Momentum_Signal'] + df['Price_Acceleration']
    
    # Calculate Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Volatility_20D_Avg'] = df['Volatility'].rolling(window=20).mean()
    
    # Adjust for Volatility
    df['Final_Alpha_Factor'] = df['Integrated_Alpha_Factors'].where(df['Volatility'] > df['Volatility_20D_Avg'], df['Integrated_Alpha_Factors'] * 0.9)
    
    return df['Final_Alpha_Factor'].dropna()
