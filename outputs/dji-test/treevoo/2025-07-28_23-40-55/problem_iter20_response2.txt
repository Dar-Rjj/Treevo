```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns [open, high, low, close, volume].
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the closing location within the day's range
    close_location = (df['close'] - df['low']) / daily_range
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=5)
    
    # Combine factors into a single alpha factor
    alpha_factor = (close_location + volume_change + momentum) / 3

    return alpha_factor
```
