import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Change
    daily_close_change = df['close'] - df['close'].shift(1)
    
    # Calculate Momentum Differentials
    short_term_momentum = df['close'].shift(1) - df['close'].shift(7)
    long_term_momentum = df['close'].shift(1) - df['close'].shift(30)
    momentum_differential = short_term_momentum - long_term_momentum
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = momentum_differential * volume_change
    
    # Introduce Volume Weighted Price Momentum
    vwap = (df['high'] + df['low'] + df['close']) / 3
    vwap_weighted = vwap * df['volume']
    vwap_sum_7_days = vwap_weighted.rolling(window=7).sum()
    
    # Synthesize Final Factor
    final_factor = adjusted_momentum + vwap_sum_7_days
    
    return final_factor
