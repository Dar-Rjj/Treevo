```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_volume_12 = df['volume'].rolling(window=12).mean()
    volume_weighted_ema_diff = ema_diff * avg_volume_12
    
    # Calculate Momentum Signal
    momentum_signal = volume_weighted_ema_diff * df['volume']
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()
    
    # Calculate Short-Term Momentum
    short_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    long_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_adjusted = (short_momentum - long_momentum) * volume_change
    
    # Synthesize Momentum and Volume-Adjusted Price-Acceleration
    final_factor = momentum_signal + momentum_adjusted * price_acceleration
    
    return final_factor
```
