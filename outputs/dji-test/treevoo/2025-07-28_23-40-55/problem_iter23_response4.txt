```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term EMA
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    ema_diff = short_ema - long_ema

    # Volume-Weighted EMA Difference
    avg_vol_12 = df['volume'].rolling(window=12).mean()
    vol_weighted_ema_diff = ema_diff * avg_vol_12

    # Momentum Signal
    momentum_signal = vol_weighted_ema_diff * df['volume']

    # Intraday High-Low Range
    high_low_range = df['high'] - df['low']

    # Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (high_low_range + open_close_momentum) / 2

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=12).mean()
    high_volume_days = df['volume'] > 1.5 * volume_rolling_mean
    filtered_alpha = integrated_momentum[high_volume_days]

    # Adjust Momentum by Volatility
    high_low_volatility = high_low_range
    adjusted_alpha = filtered_alpha * high_low_volatility

    # Calculate Price Velocity
    price_velocity = df['close'].diff()

    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()

    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    long_term_momentum = df['close'] - df['close'].shift(30)

    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    momentum_adj_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change

    # Introduce Volume Weighted Price Momentum
    weighted_prices = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    volume_weighted_price_momentum = weighted_prices - df['close'].shift(7)

    # Synthesize Final Alpha Factor
    final_alpha = momentum_signal + adjusted_price_acceleration + adjusted_alpha + volume_weighted_price_momentum

    return final_alpha
```
