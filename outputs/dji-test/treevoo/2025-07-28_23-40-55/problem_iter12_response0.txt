```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Short-Term EMA (8-day span)
    df['short_ema'] = df['close'].ewm(span=8, adjust=False).mean()
    
    # Calculate Long-Term EMA (20-day span)
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Adjust Momentum by Volume Shock
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_momentum'] = df['ema_diff'] * df['volume_change']
    
    # Synthesize Final Factor
    df['final_factor'] = 0.5 * df['price_velocity'] + 0.3 * df['adjusted_momentum'] + 0.2 * df['ema_diff']
    
    return df['final_factor']
```
