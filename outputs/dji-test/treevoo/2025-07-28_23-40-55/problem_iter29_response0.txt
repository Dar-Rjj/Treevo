```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Open-to-Close Momentum
    df['Open_Close_Momentum'] = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Momentum
    df['Integrated_Momentum'] = df['Intraday_High_Low_Range'] + df['Open_Close_Momentum']
    
    # Volume Filter
    df['Volume_Rolling_Mean'] = df['volume'].rolling(window=30).mean()
    high_volume_threshold = df['Volume_Rolling_Mean'].quantile(0.75)
    df['High_Volume_Days'] = (df['volume'] > high_volume_threshold).astype(int)
    
    # Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['close'].shift(1) - df['close'].shift(7)
    df['Long_Term_Momentum'] = df['close'].shift(1) - df['close'].shift(30)
    
    # Adjust Price Momentum by Volume Shock
    df['Volume_Shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Price_Momentum'] = (df['Short_Term_Momentum'] - df['Long_Term_Momentum']) * df['Volume_Shock']
    
    # Daily Volume Change
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Intraday Momentum by Volatility
    df['Adjusted_Intraday_Momentum'] = df['Integrated_Momentum'] * df['Intraday_High_Low_Range']
    
    # Price Velocity
    df['Price_Velocity'] = df['close'] - df['close'].shift(1)
    
    # Price Acceleration
    df['Price_Acceleration'] = df['Price_Velocity'] - df['Price_Velocity'].shift(1)
    
    # Exponential Moving Averages
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # EMA Difference
    df['EMA_Difference'] = df['EMA_12'] - df['EMA_26']
    
    # Adjust EMA Difference by Volume Shock
    df['Adjusted_EMA_Difference'] = df['EMA_Difference'] * df['Daily_Volume_Change']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Price_Momentum'] + df['Adjusted_Intraday_Momentum'] + df['Price_Acceleration']
    
    # Add Adjusted EMA Difference
    df['Final_Alpha_Factor'] += df['Adjusted_EMA_Difference']
    
    # Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    df['Final_Alpha_Factor'] *= df['Intraday_Return']
    
    # Refine the Final Factor (Only Include High Volume Days)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['High_Volume_Days']
    
    # Adjust Momentum by High-Low Volatility
    df['Final_Alpha_Factor'] *= df['Intraday_High_Low_Range']
    
    return df['Final_Alpha_Factor'].dropna()
```
