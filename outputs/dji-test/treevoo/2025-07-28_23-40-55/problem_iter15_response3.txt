```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_volume_12 = df['volume'].rolling(window=12).mean()
    volume_weighted_ema_diff = ema_diff * avg_volume_12
    
    # Calculate Momentum Signal
    momentum_signal = volume_weighted_ema_diff * df['volume']
    
    # Calculate Price Acceleration
    price_velocity = df['close'] - df['close'].shift(1)
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Calculate Short-Term Price Momentum (7-day time frame)
    close_diff_7 = df['close'] - df['close'].shift(7)
    short_term_price_momentum = close_diff_7.rolling(window=7).sum() / 7
    
    # Calculate Long-Term Price Momentum (25-day time frame)
    close_diff_25 = df['close'] - df['close'].shift(25)
    long_term_price_momentum = close_diff_25.rolling(window=25).sum() / 25
    
    # Calculate Volume Trend
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_trend = volume_change.rolling(window=7).sum() / 7
    
    # Calculate Price Acceleration Over 7 Days
    price_acceleration_7 = df['close'] - df['close'].shift(7)
    
    # Combine Momentum, Volume-Weighted EMA Difference, and Enhanced Price-Acceleration
    combined_factor = momentum_signal + volume_weighted_ema_diff + price_acceleration_7
    
    # Final Combination
    final_factor = 1.5 * combined_factor * short_term_price_momentum * volume_trend
    
    return final_factor
```
