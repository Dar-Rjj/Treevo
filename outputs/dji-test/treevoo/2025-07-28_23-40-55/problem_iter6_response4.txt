```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (10-day span)
    df['short_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA (30-day span)
    df['long_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Short-Term Price Momentum (7 days)
    short_term_days = 7
    df['short_momentum'] = (df['close'] - df['close'].shift(short_term_days)) / short_term_days
    
    # Calculate Long-Term Price Momentum (25 days)
    long_term_days = 25
    df['long_momentum'] = (df['close'] - df['close'].shift(long_term_days)) / long_term_days
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['total_volume_change'] = df['volume_change'].rolling(window=short_term_days).sum() / short_term_days
    
    # Calculate Weighted EMA Difference
    weight_factor = 1.5
    df['weighted_ema_diff'] = df['ema_diff'] * weight_factor
    
    # Combine Factors
    df['alpha_factor'] = df['weighted_ema_diff'] * df['short_momentum'] * df['total_volume_change']
    
    return df['alpha_factor']
```
