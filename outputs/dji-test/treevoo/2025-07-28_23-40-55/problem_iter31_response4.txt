```python
import pandas as pd

def heuristics_v2(df):
    # Momentum factors
    df['1_day_return'] = df['close'].pct_change()
    df['20_day_return'] = df['close'].pct_change(20)
    df['60_day_return'] = df['close'].pct_change(60)
    
    # Volatility factor
    df['20_day_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Volume factors
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Support and resistance factor
    df['20_day_moving_avg'] = df['close'].rolling(window=20).mean()
    df['days_above_20_day_ma'] = (df['close'] > df['20_day_moving_avg']).rolling(window=20).sum()
    
    # Intraday sentiment factor
    df['up_day'] = (df['close'] > df['open']).astype(int)
    df['proportion_up_days'] = df['up_day'].rolling(window=20).mean()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['1_day_return'] + 
        df['20_day_return'] + 
        df['60_day_return'] - 
        df['20_day_volatility'] + 
        df['5_day_avg_volume'] + 
        df['20_day_avg_volume'] + 
        df['days_above_20_day_ma'] + 
        df['proportion_up_days']
    )
    
    return df['alpha_factor'].dropna()
```
