```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term and Long-Term EMA
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12'] = df['volume'].rolling(window=12).mean()
    df['volume_weighted_ema_diff'] = df['ema_diff'] * df['avg_volume_12']
    
    # Calculate Short-Term and Long-Term Momentum
    df['short_momentum'] = df['close'].shift(1) - df['close'].shift(7)
    df['long_momentum'] = df['close'].shift(1) - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_momentum'] - df['long_momentum']) * df['volume_change']
    
    # Synthesize Momentum and Volume-Adjusted Price-Acceleration
    df['synthesized_factor'] = df['adjusted_momentum'] + df['price_acceleration'] * df['avg_volume_12']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap_sum_7'] = df['vwap'].rolling(window=7).sum()
    df['volume_weighted_price_momentum'] = df['vwap_sum_7'] / df['volume'].rolling(window=7).sum()
    
    # Incorporate Volume Weighted Price Momentum
    df['final_synthesized_factor'] = df['synthesized_factor'] + df['volume_weighted_price_momentum']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_range'] + df['open_close_momentum']
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=10).mean()
    volume_threshold = df['volume_rolling_mean'].quantile(0.8)
    df['filtered_alpha'] = df['integrated_momentum'].where(df['volume'] > volume_threshold, 0)
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    
    # Adjust Integrated Momentum by Volatility
    df['interim_alpha'] = df['filtered_alpha'] * df['high_low_volatility']
    
    # Introduce New Factor: Price and Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=10).corr(df['volume'])
    
    # Combine Synthesized Factor with Volatility-Adjusted Momentum and Price-Volume Correlation
    df['final_alpha'] = df['final_synthesized_factor'] + df['interim_alpha'] + df['price_volume_corr']
    
    return df['final_alpha']
```
