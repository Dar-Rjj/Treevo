```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_close_momentum) / 2

    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = volume_mean * 1.5  # Setting a threshold for high volume
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_alpha_factor = integrated_momentum[high_volume_days]

    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']

    # Adjust Momentum by Volatility
    adjusted_momentum = filtered_alpha_factor * high_low_volatility
    adjusted_momentum = adjusted_momentum.reindex(df.index, fill_value=0)

    return adjusted_momentum
```
