```python
def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the midpoint of the high and low
    df['mid_point'] = (df['high'] + df['low']) / 2
    
    # Calculate the percentage difference between the close price and the midpoint
    df['close_to_mid_ratio'] = (df['close'] - df['mid_point']) / df['range']
    
    # Calculate the percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the factor as a combination of the close to mid ratio and the volume change
    df['factor'] = df['close_to_mid_ratio'] * df['volume_change'].fillna(0)
    
    return df['factor']
```
