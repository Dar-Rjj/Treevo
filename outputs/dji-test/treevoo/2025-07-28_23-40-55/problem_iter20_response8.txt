```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term EMA
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    # Calculate Long-Term EMA
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    # Calculate Volume-Weighted EMA Difference
    vol_weighted_ema_diff = ema_diff * df['volume'].rolling(window=12).mean()
    # Calculate Momentum Signal
    momentum_signal = vol_weighted_ema_diff * df['volume']

    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    # Calculate Open-to-Close Momentum
    open_to_close_mom = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_mom) / 2

    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=12).mean()
    high_volume_days = df['volume'] > 1.5 * volume_mean
    filtered_alpha_factor = integrated_momentum * high_volume_days

    # Adjust Momentum by Volatility
    volatility = df['high'] - df['low']
    adjusted_filtered_alpha_factor = filtered_alpha_factor * volatility

    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()
    
    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']

    # Calculate Short-Term and Long-Term Momentum
    short_term_mom = df['close'].shift(1) - df['close'].shift(7)
    long_term_mom = df['close'].shift(1) - df['close'].shift(30)

    # Calculate Daily Volume Change
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    adjusted_momentum_by_vol_shock = (short_term_mom - long_term_mom) * daily_vol_change

    # Introduce Volume Weighted Price Momentum
    weighted_prices = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()

    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        momentum_signal +
        adjusted_price_acceleration +
        adjusted_filtered_alpha_factor +
        adjusted_momentum_by_vol_shock +
        weighted_prices
    )

    return final_alpha_factor
```
