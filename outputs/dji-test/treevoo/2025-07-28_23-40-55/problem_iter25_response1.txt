```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (high_low_range + open_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    volume_threshold = volume_rolling_mean * 1.5  # Set threshold to 1.5 times the rolling mean
    high_volume_days = df['volume'] > volume_threshold
    filtered_alpha_factor = integrated_momentum[high_volume_days].reindex(df.index, fill_value=0)
    
    # Adjust Momentum by Volatility
    high_low_volatility = df['high'] - df['low']
    preliminary_alpha_factor = filtered_alpha_factor * high_low_volatility
    
    # Calculate Short-Term Price Momentum
    short_term_window = 7
    close_price_diff_7d = df['close'] - df['close'].shift(short_term_window)
    sum_close_diff_7d = close_price_diff_7d.rolling(window=short_term_window).sum()
    avg_daily_return_7d = sum_close_diff_7d / short_term_window
    
    # Calculate Long-Term Price Momentum
    long_term_window = 25
    close_price_diff_25d = df['close'] - df['close'].shift(long_term_window)
    sum_close_diff_25d = close_price_diff_25d.rolling(window=long_term_window).sum()
    avg_daily_return_25d = sum_close_diff_25d / long_term_window
    
    # Adjust Momentum by Volume Shock
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change_7d = daily_volume_change.rolling(window=short_term_window).sum() / short_term_window
    adjusted_momentum = (avg_daily_return_7d - avg_daily_return_25d) * total_volume_change_7d
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = preliminary_alpha_factor + adjusted_momentum
    
    # Calculate EMAs
    short_term_ema = df['close'].ewm(span=10).mean()
    long_term_ema = df['close'].ewm(span=30).mean()
    
    # EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Adjusted Momentum Signal
    adjusted_momentum_signal = ema_difference * df['volume'] + (df['high'] - df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum * df['close']) + adjusted_momentum_signal
    
    return final_alpha_factor
```
