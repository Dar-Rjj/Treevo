```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=7).mean()
    high_volume_threshold = volume_rolling_mean * 1.5  # 50% above the rolling mean
    is_high_volume = df['volume'] > high_volume_threshold
    filtered_integrated_momentum = integrated_momentum[is_high_volume].reindex(df.index, fill_value=0)
    
    # Calculate Short-Term and Long-Term EMA
    short_term_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_return = (df['close'] - df['close'].shift(7)) / 7
    
    # Calculate Long-Term Price Momentum
    long_term_return = (df['close'] - df['close'].shift(25)) / 25
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'].pct_change()
    volume_trend = daily_volume_change.rolling(window=7).sum()
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_return - long_term_return) * volume_trend
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = (filtered_integrated_momentum + adjusted_momentum)
    
    # Adjusted Momentum Signal
    adjusted_momentum_signal = ema_difference * df['volume'] + (df['high'] - df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum * df['close']) + adjusted_momentum_signal
    
    return final_alpha_factor
```
