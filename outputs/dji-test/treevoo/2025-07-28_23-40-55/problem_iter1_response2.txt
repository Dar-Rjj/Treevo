```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement Ratio
    df['price_movement_ratio'] = df['high'] / df['low']
    
    # Evaluate Weighted Average Volume Over Period
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['close_diff'] = df['close'] - df['close'].shift(n)
    df['volume_adjusted_momentum'] = df['close_diff'] / df['avg_volume']
    
    # Assess Positive vs Negative Momentum Contribution
    df['adjusted_momentum'] = df.apply(lambda row: row['volume_adjusted_momentum'] * 1.5 if row['close_diff'] > 0 else row['volume_adjusted_momentum'], axis=1)
    
    # Integrate Accumulated Momentum Impact
    # Using multiple time horizons (e.g., 10, 20, 30 days) and summing them up with weights
    time_horizons = [10, 20, 30]
    weights = [1, 2, 3]  # Assign greater weight to longer time horizons
    accumulated_momentum = pd.Series(0, index=df.index)
    
    for i, horizon in enumerate(time_horizons):
        momentum = (df['close'] - df['close'].shift(horizon)) / df['volume'].rolling(window=horizon).mean()
        momentum_adj = momentum.apply(lambda x: x * 1.5 if x > 0 else x)
        accumulated_momentum += momentum_adj * weights[i]
    
    return accumulated_momentum
```
