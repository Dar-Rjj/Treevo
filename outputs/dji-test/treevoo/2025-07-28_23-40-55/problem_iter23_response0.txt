```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / daily_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / daily_range
    
    # Heuristic: The net pressure is the difference between buying and selling pressure
    net_pressure = buying_pressure - selling_pressure
    
    # Integrate the volume into the factor, assuming higher volume means more conviction
    volume_weighted_net_pressure = net_pressure * df['volume']
    
    return volume_weighted_net_pressure
```
