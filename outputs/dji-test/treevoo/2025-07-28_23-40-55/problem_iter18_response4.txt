```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the day's range
    df['day_range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['day_range']

    # Calculate the directional movement by comparing the current close to the previous close
    df['directional_movement'] = df['close'].diff()

    # Calculate the volume weighted close
    df['volume_weighted_close'] = df['close'] * df['volume']

    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_position'] + df['directional_movement'] + df['volume_weighted_close']).shift(1)

    return df['alpha_factor']
```
