```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Close Price Difference (Price Velocity)
    df['price_velocity'] = df['close'].diff()
    
    # Compute Daily Velocity Difference (Price Acceleration)
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['short_term_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['long_term_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_difference'] = df['short_term_ema'] - df['long_term_ema']
    
    # Generate Composite Momentum Signal
    df['composite_momentum_signal'] = (df['ema_difference'] * df['volume']) + df['price_acceleration']
    
    return df['composite_momentum_signal']
```
