```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Price Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = momentum.where(df['volume'] < avg_volume_20, momentum * 1.15)
    
    # Adjust for Amount Impact
    avg_amount_20 = df['amount'].rolling(window=20).mean()
    amount_adjusted_momentum = volume_adjusted_momentum.where(df['amount'] < avg_amount_20, volume_adjusted_momentum * 1.1)
    
    # Calculate Price Velocity
    velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    acceleration = velocity - velocity.shift(1)
    
    # Calculate Short-Term EMA (12-day span)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Generate Composite Momentum Signal
    composite_momentum_signal = ema_difference * df['volume']
    
    # Incorporate Integrated Alpha Factors
    integrated_alpha_factors = amount_adjusted_momentum + composite_momentum_signal + acceleration
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=20).std()
    
    # Adjust for Volatility
    avg_volatility_20 = volatility.rolling(window=20).mean()
    volatility_adjusted_factors = integrated_alpha_factors.where(volatility < avg_volatility_20, integrated_alpha_factors * 0.9)
    
    # Finalize Alpha Factor
    alpha_factor = volatility_adjusted_factors
    
    return alpha_factor
```
