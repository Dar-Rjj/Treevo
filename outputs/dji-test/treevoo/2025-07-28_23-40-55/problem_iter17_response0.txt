```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the closing price over a 10-day window
    df['close_10_day_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of the closing price over a 10-day window
    df['close_10_day_std'] = df['close'].rolling(window=10).std()
    
    # Compute the z-score of the current closing price relative to its 10-day mean and standard deviation
    df['z_score_close'] = (df['close'] - df['close_10_day_mean']) / df['close_10_day_std']
    
    # Calculate the momentum of the closing price over a 5-day period
    df['momentum_5_day'] = df['close'] - df['close'].shift(5)
    
    # Calculate the relative strength index (RSI) with a 14-day lookback
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14_day'] = 100 - (100 / (1 + rs))
    
    # Combine the z-score, momentum, and RSI into a single alpha factor
    df['alpha_factor'] = 0.5 * df['z_score_close'] + 0.3 * df['momentum_5_day'] + 0.2 * (df['rsi_14_day'] - 50)
    
    return df['alpha_factor']
```
