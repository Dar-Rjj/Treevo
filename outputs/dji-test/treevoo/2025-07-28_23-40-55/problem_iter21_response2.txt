```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_to_close_momentum = (df['close'] - df['open']) / df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_days = df['volume'] > volume_rolling_mean * 1.5
    filtered_alpha_factor = integrated_momentum[high_volume_days]

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Incorporate Intraday Return
    incorporated_momentum = filtered_alpha_factor * intraday_return

    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']

    # Adjust Momentum by Volatility
    adjusted_momentum = incorporated_momentum * high_low_volatility

    # Calculate Price Velocity
    price_velocity = df['close'].diff()

    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(7)

    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(30)

    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    momentum_adjusted_by_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change

    # Introduce Volume Weighted Price Momentum
    vwap_7d = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    volume_weighted_price_momentum = vwap_7d.diff()

    # Introduce Volume Weighted Price Oscillator
    vwap_14d = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    volume_weighted_price_oscillator = vwap_7d - vwap_14d.shift(7)

    # Synthesize Final Factor
    final_factor = adjusted_momentum + volume_weighted_price_momentum + volume_weighted_price_oscillator

    return final_factor
```
