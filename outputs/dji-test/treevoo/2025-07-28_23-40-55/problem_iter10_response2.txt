```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span) and Long-Term EMA (26-day span)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the Volume-Weighted EMA Difference
    ema_diff = (short_ema - long_ema) * df['volume']
    
    return ema_diff
```
