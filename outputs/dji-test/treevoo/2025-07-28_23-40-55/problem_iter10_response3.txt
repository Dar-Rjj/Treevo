```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA
    df['short_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA
    df['long_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Short-Term Price Momentum
    df['7_day_momentum'] = (df['close'] - df['close'].shift(7)) / 7
    
    # Calculate Long-Term Price Momentum
    df['25_day_momentum'] = (df['close'] - df['close'].shift(25)) / 25
    
    # Calculate Volume Trend
    df['daily_vol_change'] = df['volume'] - df['volume'].shift(1)
    df['vol_trend'] = df['daily_vol_change'].rolling(window=7).sum()
    
    # Combine Factors
    df['combined_factor'] = 1.5 * df['ema_diff'] * df['7_day_momentum'] * df['vol_trend']
    
    return df['combined_factor']
```
