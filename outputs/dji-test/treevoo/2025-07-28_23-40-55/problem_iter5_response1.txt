```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Volume-Weighted EMA Difference
    df['volume_weighted_ema_diff'] = df['ema_diff'] * df['volume'].rolling(window=12).mean()
    
    # Adjust Momentum by Volume Shock
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_momentum'] = (df['short_ema'] - df['long_ema']) * df['volume_change']
    
    # Synthesize Price-Acceleration and Volume-Adjusted Momentum with EMA
    df['final_factor'] = df['price_acceleration'] + df['adjusted_momentum'] + df['volume_weighted_ema_diff']
    
    return df['final_factor']
```
