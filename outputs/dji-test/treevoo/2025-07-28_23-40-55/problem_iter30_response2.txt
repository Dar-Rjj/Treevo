```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['close']
    
    # Identify High Volume Days
    volume_rolling_median = df['volume'].rolling(window=30).median()
    high_volume_threshold = 1.5 * volume_rolling_median
    is_high_volume_day = df['volume'] > high_volume_threshold
    
    # Combine Intraday Volatility and Open-to-Close Return
    combined_momentum = intraday_volatility + open_to_close_return
    
    # Retain Only High Volume Days' Combined Momentum
    high_volume_filtered_momentum = combined_momentum.where(is_high_volume_day, 0)
    
    # Adjusted Momentum by Intraday Volatility
    adjusted_momentum = high_volume_filtered_momentum * intraday_volatility
    
    # EMA Difference
    ema_short_term = df['close'].ewm(span=15, adjust=False).mean()
    ema_long_term = df['close'].ewm(span=45, adjust=False).mean()
    ema_difference = ema_short_term - ema_long_term
    
    # Accumulated Volume for the last 10 days
    accumulated_volume = df['volume'].rolling(window=10).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (ema_difference * adjusted_momentum) * accumulated_volume
    
    return final_alpha_factor
```
