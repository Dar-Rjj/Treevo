import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()

    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(30)

    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Adjust Momentum by Volume Shock
    df['momentum_adjusted'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']

    # Synthesize Price-Acceleration and Volume-Adjusted Momentum
    df['final_factor'] = df['price_acceleration'] + df['momentum_adjusted']
    
    return df['final_factor']
