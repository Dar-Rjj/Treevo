def heuristics_v2(df):
    # Calculate Price Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_above_avg = df['volume'] > avg_volume_20
    adjusted_momentum = momentum * (1.1 * volume_above_avg + 1.0 * ~volume_above_avg)
    
    # Adjust for Amount Impact
    avg_amount_20 = df['amount'].rolling(window=20).mean()
    amount_above_avg = df['amount'] > avg_amount_20
    final_adjusted_momentum = adjusted_momentum * (1.05 * amount_above_avg + 1.0 * ~amount_above_avg)
    
    return final_adjusted_momentum
