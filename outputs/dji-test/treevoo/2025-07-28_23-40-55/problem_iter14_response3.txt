```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term EMA
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_vol = df['volume'].rolling(window=12).mean()
    vol_weighted_ema_diff = ema_diff * avg_vol
    
    # Calculate Momentum Signal
    momentum_signal = vol_weighted_ema_diff * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_to_close_momentum
    
    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=12).mean()
    high_volume_days = df['volume'] > 1.5 * volume_mean
    filtered_alpha_factor = integrated_momentum[high_volume_days]
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    volatility_adjusted_momentum = filtered_alpha_factor * high_low_volatility
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()
    
    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_with_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Synthesize Momentum, Volume-Adjusted Price-Acceleration, and Volatility-Adjusted Momentum
    final_factor = momentum_signal + adjusted_price_acceleration + volatility_adjusted_momentum + momentum_with_volume_shock
    
    return final_factor
```
