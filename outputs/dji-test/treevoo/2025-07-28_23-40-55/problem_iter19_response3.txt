```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Price_Velocity'] = df['close'].diff()

    # Calculate Short-Term and Long-Term EMA
    df['Short_Term_EMA'] = df['close'].ewm(span=12, adjust=False).mean()
    df['Long_Term_EMA'] = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    df['EMA_Difference'] = df['Short_Term_EMA'] - df['Long_Term_EMA']

    # Generate Composite Momentum Signal
    df['Composite_Momentum'] = (df['EMA_Difference'] * df['volume']) + df['Price_Velocity'].diff()

    # Adjust Momentum by Volume Shock
    df['Volume_Shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Momentum'] = df['EMA_Difference'] * df['Volume_Shock']

    # Introduce Volume Weighted Price Momentum
    df['Volume_Weighted_Price'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['Volume_Weighted_Price_Momentum'] = df['Volume_Weighted_Price'].rolling(window=7).sum()

    # Incorporate High-Low Range Volatility
    df['Daily_High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Volatility'] = df['Daily_High_Low_Range'].rolling(window=7).mean()

    # Synthesize Final Factor
    df['Final_Factor'] = (df['Adjusted_Momentum'] + df['Volume_Weighted_Price_Momentum']) + df['High_Low_Range_Volatility']

    # Apply Volume Filter
    df['Volume_Rolling_Mean'] = df['volume'].rolling(window=20).mean()
    high_volume_threshold = df['Volume_Rolling_Mean'].quantile(0.8)
    df['High_Volume'] = df['volume'] > high_volume_threshold
    df = df[df['High_Volume']]
    
    return df['Final_Factor']
```
