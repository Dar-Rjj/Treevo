```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_days = df[volume_rolling_mean > df['volume']].index
    filtered_alpha_factor = integrated_momentum[high_volume_days]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    preliminary_alpha_factor = filtered_alpha_factor * intraday_return
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    adjusted_momentum = preliminary_alpha_factor * high_low_volatility
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] - df['close'].shift(7)) / 7
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(25)) / 25
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=7).sum() / 7
    
    # Calculate Weighted EMA Difference
    weighted_ema_difference = ema_difference * 1.8
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum_by_volume_shock = (short_term_momentum - long_term_momentum) * total_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum_and_volume_shock = adjusted_momentum + adjusted_momentum_by_volume_shock
    
    # Calculate EMAs
    short_term_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # EMA Difference
    ema_difference_10_30 = short_term_ema_10 - long_term_ema_30
    
    # Adjusted Momentum Signal
    adjusted_momentum_signal = ema_difference_10_30 * df['volume'] + df['high'] - df['low']
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum_and_volume_shock * df['close']) + adjusted_momentum_signal
    
    return final_alpha_factor
```
