```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span)
    df['Short_EMA'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['Long_EMA'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['EMA_Diff'] = df['Short_EMA'] - df['Long_EMA']
    
    # Calculate Price Momentum (t - t-20)
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    # Calculate the average volume over the past 20 days
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Combine EMA Difference and Price Momentum
    df['Combined_Signal'] = df['EMA_Diff'] * df['Price_Momentum']
    
    # Adjust based on Volume
    df['Volume_Adjusted_Signal'] = df.apply(
        lambda row: row['Combined_Signal'] * 1.1 if row['volume'] > row['Avg_Volume_20'] else row['Combined_Signal'],
        axis=1
    )
    
    # Return the final factor values
    return df['Volume_Adjusted_Signal']
```
