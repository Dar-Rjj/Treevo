```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['Open_Close_Momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['Integrated_Momentum'] = (df['High_Low_Range'] + df['Open_Close_Momentum']) / 2
    
    # Apply Volume Filter
    df['Volume_Rolling_Mean'] = df['volume'].rolling(window=10).mean()
    volume_threshold = df['volume'].quantile(0.8)  # Example threshold: 80th percentile
    high_volume_days = df['volume'] > volume_threshold
    df_high_volume = df[high_volume_days]
    
    # Calculate Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['close'] - df['close'].shift(7)
    df['Long_Term_Momentum'] = df['close'] - df['close'].shift(30)
    
    # Compute Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Price Momentum by Volume Shock
    df['Adjusted_Price_Momentum'] = (df['Short_Term_Momentum'] - df['Long_Term_Momentum']) * df['Volume_Change']
    
    # Adjust Intraday Momentum by Volatility
    df['Volatility'] = df['high'] - df['low']
    df['Adjusted_Intraday_Momentum'] = df['Integrated_Momentum'] * df['Volatility']
    
    # Calculate Price Velocity
    df['Velocity'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    df['Acceleration'] = df['Velocity'] - df['Velocity'].shift(1)
    
    # Calculate Short-Term and Long-Term EMA
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['EMA_Difference'] = df['EMA_12'] - df['EMA_26']
    
    # Adjust EMA Difference by Volume Shock
    df['Adjusted_EMA_Difference'] = df['EMA_Difference'] * df['Volume_Change']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Price_Momentum'] 
                                + df['Adjusted_Intraday_Momentum'] 
                                + df['Acceleration']
                                + df['Adjusted_EMA_Difference'])
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    df['Integrated_Final_Momentum'] = df['Integrated_Momentum'] * df['Intraday_Return']
    
    # Refine the Final Factor
    df['Refined_Final_Factor'] = df['Final_Alpha_Factor'] * df['Intraday_Return']
    
    # Adjust Momentum by High-Low Volatility
    df['Final_Factor'] = df['Refined_Final_Factor'] * df['Volatility']
    
    return df_high_volume['Final_Factor']
```
