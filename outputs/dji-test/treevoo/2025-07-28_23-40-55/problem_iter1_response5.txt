```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day moving average return
    df['10_day_moving_avg_return'] = df['close'].pct_change(10)
    
    # Calculate current day's log return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Momentum Factor: 10-Day Moving Average Return - Current Day's Log Return
    df['momentum_factor'] = df['10_day_moving_avg_return'] - df['log_return']

    # Intraday Range Percentage
    df['intraday_range_pct'] = (df['high'] - df['low']) / df['open']
    
    # Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['low']
    
    # Volume Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_returns'] * df['volume']
    
    # Intraday Sentiment Index
    df['intraday_sentiment_index'] = df['intraday_range_pct'] + df['volume_weighted_intraday_momentum']
    
    # Combine Momentum and Sentiment
    df['combined_factor'] = df['momentum_factor'] + df['intraday_sentiment_index']
    
    # Apply a Non-Linear Transformation (Sigmoid Function)
    df['final_factor'] = 1 / (1 + np.exp(-df['combined_factor']))
    
    return df['final_factor']
```
