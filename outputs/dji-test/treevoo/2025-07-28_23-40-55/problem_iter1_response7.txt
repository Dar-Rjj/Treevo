```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Calculate Momentum (14-Day Moving Average of Close)
    momentum = df['close'] - df['close'].rolling(window=14).mean()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum / high_low_volatility
    
    return adjusted_momentum
```
