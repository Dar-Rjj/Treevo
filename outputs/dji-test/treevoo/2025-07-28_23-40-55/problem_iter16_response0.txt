```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_close_momentum) / 2
    
    # Apply Volume Filter
    volume_threshold = df['volume'].quantile(0.75)  # Set a Threshold for High Volume
    high_volume_days = df['volume'] > volume_threshold
    filtered_momentum = integrated_momentum[high_volume_days]
    
    # Adjust Momentum by Volatility
    volatility = df['high'] - df['low']  # High-Low Volatility
    adjusted_momentum = filtered_momentum * (1 / volatility)
    
    # Generate Final Alpha Factor
    alpha_factor = pd.Series(index=df.index, data=0.0)
    alpha_factor.loc[high_volume_days] = adjusted_momentum
    
    return alpha_factor
```
