import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_high_low_range + open_to_close_momentum) / 2

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = df['volume'].quantile(0.75)
    is_high_volume = df['volume'] > high_volume_threshold
    filtered_by_volume = integrated_momentum.where(is_high_volume, 0)

    # Apply Amount Filter
    amount_rolling_mean = df['amount'].rolling(window=10).mean()
    high_amount_threshold = df['amount'].quantile(0.75)
    is_high_amount = df['amount'] > high_amount_threshold
    final_alpha_factor = filtered_by_volume.where(is_high_amount, 0)

    return final_alpha_factor
