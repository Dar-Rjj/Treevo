```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Calculate Short-Term Price Momentum
    short_term_days = 5
    short_term_momentum = (df['close'] - df['close'].shift(short_term_days)).rolling(window=short_term_days).sum() / short_term_days
    
    # Calculate Long-Term Price Momentum
    long_term_days = 20
    long_term_momentum = (df['close'] - df['close'].shift(long_term_days)).rolling(window=long_term_days).sum() / long_term_days
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_change
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = 1.5  # Example threshold
    high_volume_days = df['volume'] > high_volume_threshold * volume_rolling_mean
    filtered_momentum = adjusted_momentum[high_volume_days]
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Combine Factors
    combined_factor = ema_difference * short_term_momentum * volume_change
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (filtered_momentum + integrated_momentum) * combined_factor
    final_alpha_factor = final_alpha_factor[high_volume_days]
    
    return final_alpha_factor
```
