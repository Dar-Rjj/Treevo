import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted EMAs
    df['CloseVol'] = df['close'] * df['volume']
    short_term_wema = df['CloseVol'].ewm(span=12, adjust=False).mean()
    long_term_wema = df['CloseVol'].ewm(span=26, adjust=False).mean()
    
    # WEMA Difference
    wema_diff = short_term_wema - long_term_wema
    
    # Weighted Momentum Signal
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    weighted_momentum_signal = wema_diff * (df['volume'] / avg_volume_26)
    
    # Volume-Adjusted Volatility
    volatility = (df['high'] - df['low']).rolling(window=26).std()
    adjusted_volatility = volatility * (df['volume'] / avg_volume_26)
    
    # Calculate Enhanced Price Momentum
    enhanced_price_momentum = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    enhanced_price_momentum = np.where(df['volume'] > avg_volume_20, 
                                       enhanced_price_momentum * 1.15, 
                                       enhanced_price_momentum)
    
    # Adjust for Amount Impact
    avg_amount_20 = df['amount'].rolling(window=20).mean()
    enhanced_price_momentum = np.where(df['amount'] > avg_amount_20, 
                                       enhanced_price_momentum * 1.1, 
                                       enhanced_price_momentum)
    
    # Calculate Price Velocity and Acceleration
    price_velocity = df['close'] - df['close'].shift(1)
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Combined Momentum Factor
    combined_momentum_factor = weighted_momentum_signal + enhanced_price_momentum + price_acceleration
    
    # Final Alpha Factor Calculation
    final_alpha_factor = combined_momentum_factor / adjusted_volatility
    
    # Adjust for Volatility
    avg_volatility_20 = adjusted_volatility.rolling(window=20).mean()
    final_alpha_factor = np.where(adjusted_volatility > avg_volatility_20, 
                                  final_alpha_factor * 0.9, 
                                  final_alpha_factor)
    
    return final_alpha_factor.dropna()
