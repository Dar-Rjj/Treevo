```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span)
    df['EMA_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['EMA_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['EMA_diff'] = df['EMA_short'] - df['EMA_long']
    
    # Calculate Volume-Weighted EMA Difference
    df['avg_volume_12'] = df['volume'].rolling(window=12).mean()
    df['vol_weighted_EMA_diff'] = df['EMA_diff'] * df['avg_volume_12']
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['vol_weighted_EMA_diff'] * df['volume']
    
    # Calculate Price Velocity
    df['price_velocity'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)
    
    # Calculate Short-Term Momentum
    df['short_momentum'] = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    df['long_momentum'] = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_momentum'] - df['long_momentum']) * df['volume_change']
    
    # Synthesize Momentum and Volume-Adjusted Price-Acceleration
    df['synthesized_factor'] = df['momentum_signal'] + df['price_acceleration'] * df['volume_change']
    
    # Calculate Volume Weighted Average Price
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['volume_weighted_price_momentum'] = df['vwap'].rolling(window=7).sum()
    
    # Incorporate Volume Weighted Price Momentum
    df['enhanced_factor'] = df['synthesized_factor'] + df['volume_weighted_price_momentum']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_range'] + df['open_close_momentum']
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=20).mean()
    high_volume_threshold = df['volume_rolling_mean'].quantile(0.8)
    df['filtered_alpha'] = df['integrated_momentum'].where(df['volume'] > high_volume_threshold, 0)
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    
    # Adjust Integrated Momentum by Volatility
    df['interim_alpha'] = df['filtered_alpha'] * df['high_low_volatility']
    
    # Combine Synthesized Factor with Volatility-Adjusted Momentum
    df['final_enhanced_factor'] = df['enhanced_factor'] + df['interim_alpha']
    
    return df['final_enhanced_factor']
```
