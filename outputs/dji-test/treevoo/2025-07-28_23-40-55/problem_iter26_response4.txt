```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the range between high and low
    range_high_low = df['high'] - df['low']
    
    # Define a momentum factor as the difference between the close and the average of open and close
    momentum_factor = df['close'] - avg_open_close
    
    # Define a volatility factor as the ratio of the range to the volume
    volatility_factor = range_high_low / df['volume']
    
    # Combine the factors with a heuristic: if the momentum is positive, use the momentum factor; otherwise, use the inverse of the volatility factor
    alpha_factor = momentum_factor.where(momentum_factor > 0, 1 / volatility_factor)
    
    return alpha_factor
```
