```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_to_close_momentum
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    filtered_integrated_momentum = integrated_momentum[high_volume_days].reindex(integrated_momentum.index, fill_value=0)
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_ema - long_ema
    
    # Calculate Short-Term Price Momentum
    short_term_return = (df['close'] - df['close'].shift(5)).fillna(0)
    short_term_momentum = short_term_return.rolling(window=5).sum() / 5
    
    # Calculate Long-Term Price Momentum
    long_term_return = (df['close'] - df['close'].shift(20)).fillna(0)
    long_term_momentum = long_term_return.rolling(window=20).sum() / 20
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=5).sum() / 5
    
    # Combine Factors
    combined_factors = ema_difference * short_term_momentum * total_volume_change
    
    # Adjust Momentum by Volume Shock
    volume_shock = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_shock
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = filtered_integrated_momentum + adjusted_momentum
    
    # Calculate Price-Velocity Indicator
    short_price_velocity = df['close'].pct_change(periods=5).fillna(0)
    long_price_velocity = df['close'].pct_change(periods=20).fillna(0)
    
    # Integrate Price-Velocity Indicator
    weighted_price_velocity = 0.7 * short_price_velocity + 0.3 * long_price_velocity
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - 2 * df['volume'].shift(1) + df['volume'].shift(2)
    
    # Incorporate Volume Acceleration
    final_factor = 0.6 * synthesized_momentum + 0.4 * (combined_factors + volume_acceleration * weighted_price_velocity)
    
    return final_factor
```
