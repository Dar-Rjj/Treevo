```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor using a combination of price and volume data.
    
    Parameters:
    df (pandas.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pandas.Series: Alpha factor values indexed by date.
    """
    
    # Calculate the ratio of close to open prices
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high to low prices
    high_to_low = df['high'] / df['low']
    
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the normalized volume
    volume_normalized = df['volume'] / df['volume'].mean()
    
    # Combine the factors
    factor = (close_to_open * high_to_low) * (daily_range + volume_normalized)
    
    return factor
```
