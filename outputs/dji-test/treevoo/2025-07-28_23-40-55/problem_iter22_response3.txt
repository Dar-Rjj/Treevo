import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 10-day moving averages of close prices
    df['MA5'] = df['close'].rolling(window=5).mean()
    df['MA10'] = df['close'].rolling(window=10).mean()
    
    # Determine the price trend
    df['Trend'] = (df['MA5'] > df['MA10']).astype(int)
    
    # Compute the Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Define overbought and oversold levels
    df['Overbought'] = (df['RSI'] > 70).astype(int)
    df['Oversold'] = (df['RSI'] < 30).astype(int)
    
    # Calculate the Average True Range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['ATR'] = tr.rolling(window=14).mean()
    
    # Calculate daily volume change as a percentage
    df['Volume_Change'] = df['volume'].pct_change() * 100
    
    # Identify significant volume spikes
    df['Volume_Spike_Positive'] = ((df['Volume_Change'] > 0) & (df['close'] > df['close'].shift(1))).astype(int)
    df['Volume_Spike_Negative'] = ((df['Volume_Change'] > 0) & (df['close'] < df['close'].shift(1))).astype(int)
    
    # Compute the Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Cum_VP'] = (df['Typical_Price'] * df['volume']).cumsum()
    df['Cum_Volume'] = df['volume'].cumsum()
    df['VWAP'] = df['Cum_VP'] / df['Cum_Volume']
    
    # Compare close price to VWAP for intraday momentum
    df['Momentum_Up'] = (df['close'] > df['VWAP']).astype(int)
    df['Momentum_Down'] = (df['close'] < df['VWAP']).astype(int)
    
    # Calculate the money flow multiplier
    df['Money_Flow_Multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Sum the multipliers over time to form the AD Line
    df['AD_Line'] = (df['Money_Flow_Multiplier'] * df['volume']).cumsum()
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Trend'] 
        - df['Overbought'] 
        + df['Oversold'] 
        - df['ATR'] 
        + df['Volume_Spike_Positive'] 
        - df['Volume_Spike_Negative'] 
        + df['Momentum_Up'] 
        - df['Momentum_Down'] 
        + df['AD_Line'].pct_change()
    )
    
    return df['Alpha_Factor']
