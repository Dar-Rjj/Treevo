def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_to_close_momentum
    
    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=20).mean()
    high_volume_threshold = volume_mean * 1.5  # Define a threshold for high volume
    high_volume_days = df[df['volume'] > high_volume_threshold]
    
    # Filter Alpha Factor on High Volume Days
    filtered_alpha_factor = integrated_momentum.loc[high_volume_days.index]
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    final_alpha_factor = filtered_alpha_factor / high_low_volatility
    return final_alpha_factor
