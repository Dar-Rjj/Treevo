```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change between the close and open price
    pct_change_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate the difference between high and low price
    hl_diff = df['high'] - df['low']
    
    # Calculate a simple moving average of the close price over 5 days
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the distance of the close price from its 5-day SMA
    distance_from_sma = df['close'] - sma_5
    
    # Heuristic: The alpha factor is the product of the percentage change, the HL difference, and the distance from SMA
    alpha_factor = pct_change_close_open * hl_diff * distance_from_sma
    
    return alpha_factor
```
