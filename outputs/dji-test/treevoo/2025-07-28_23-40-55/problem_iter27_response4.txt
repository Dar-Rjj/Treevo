def heuristics_v2(df):
    # Calculate the relative position of the closing price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the relative change in volume
    df['volume_change'] = df['volume'].pct_change()

    # Calculate a weighted average of the close position and volume change
    df['alpha_factor'] = 0.7 * df['close_position'] + 0.3 * df['volume_change']

    return df['alpha_factor']
