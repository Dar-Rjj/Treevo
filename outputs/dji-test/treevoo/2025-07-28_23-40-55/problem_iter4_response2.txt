```python
def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Momentum
    df['momentum_short_term'] = df['close'].pct_change(periods=6)  # t-1 / t-7
    df['momentum_long_term'] = df['close'].pct_change(periods=29)  # t-1 / t-30
    
    # Compute Relative Volume Change
    df['relative_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Combined Momentum by Volume Shock
    df['adjusted_momentum'] = (df['momentum_short_term'] - df['momentum_long_term']) * df['relative_volume_change']
    
    return df['adjusted_momentum']
```
