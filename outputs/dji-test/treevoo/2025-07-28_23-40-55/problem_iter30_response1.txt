```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_days = df[df['volume'] > 1.5 * volume_rolling_mean]
    filtered_alpha_factor = integrated_momentum[high_volume_days.index]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    adjusted_momentum = filtered_alpha_factor * intraday_return
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    volatility_adjusted_momentum = adjusted_momentum * high_low_volatility
    
    # Calculate Short-Term EMA
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_volume_12_days = df['volume'].rolling(window=12).mean()
    volume_weighted_ema_difference = ema_difference * avg_volume_12_days
    
    # Calculate Momentum Signal
    momentum_signal = volume_weighted_ema_difference * df['volume']
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    momentum_by_volume_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Calculate Price-Velocity Interaction
    price_velocity_interaction = price_velocity * price_acceleration
    
    # Adjust Price-Velocity Interaction by Volume
    adjusted_price_velocity_interaction = price_velocity_interaction * df['volume']
    
    # Calculate Price-Volatility Interaction
    price_volatility_interaction = price_velocity * high_low_volatility
    
    # Adjust Price-Volatility Interaction by Volume
    adjusted_price_volatility_interaction = price_volatility_interaction * df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        volatility_adjusted_momentum + 
        momentum_signal + 
        adjusted_price_acceleration + 
        adjusted_price_velocity_interaction + 
        adjusted_price_volatility_interaction + 
        momentum_by_volume_shock
    )
    
    return final_alpha_factor
```
