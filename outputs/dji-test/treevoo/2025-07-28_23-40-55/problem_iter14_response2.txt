```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_12'] - df['ema_26']
    
    # Generate Composite Momentum Signal
    df['composite_momentum'] = df['ema_diff'] * df['volume'] + df['price_velocity'].diff()
    
    # Adjust Momentum by Volume Shock
    df['volume_shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_momentum'] = (df['ema_12'] - df['ema_26']) * df['volume_shock']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap_7_sum'] = df['vwap'].rolling(window=7).sum()
    
    # Incorporate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=7).mean()
    
    # Synthesize Final Factor
    df['final_factor'] = (df['adjusted_momentum'] + df['vwap_7_sum']) + df['avg_high_low_range']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_range'] + df['open_close_momentum']
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=7).mean()
    high_volume_threshold = df['volume_rolling_mean'].quantile(0.95)
    df['filtered_alpha_factor'] = df['integrated_momentum'].where(df['volume'] > high_volume_threshold, 0)
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    df['adjusted_alpha_factor'] = df['filtered_alpha_factor'] / df['high_low_volatility']
    
    # Incorporate Amount in the Alpha Factor
    df['amount_rolling_avg'] = df['amount'].rolling(window=7).mean()
    df['amount_ratio'] = df['amount'] / df['amount_rolling_avg']
    
    # Finalize Alpha Factor
    df['final_alpha_factor'] = df['adjusted_alpha_factor'] * df['amount_ratio']
    
    return df['final_alpha_factor']
```
