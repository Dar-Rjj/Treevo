```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = (df['high_low_range'] + df['open_close_momentum']) / 2
    
    # Apply Volume Filter
    rolling_mean_volume = df['volume'].rolling(window=20).mean()
    high_volume_threshold = rolling_mean_volume * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    df = df[high_volume_days]
    
    # Calculate Short-Term and Long-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(6)
    df['long_term_momentum'] = df['close'] - df['close'].shift(29)
    
    # Calculate Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Price Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']
    
    # Adjust Intraday Momentum by Volatility
    df['volatility'] = df['high'] - df['low']
    df['adjusted_intraday_momentum'] = df['integrated_momentum'] * df['volatility']
    
    # Calculate Price Velocity and Acceleration
    df['price_velocity'] = df['close'] - df['close'].shift(1)
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)
    
    # Calculate Short-Term and Long-Term EMA
    df['short_term_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    df['long_term_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_difference'] = df['short_term_ema'] - df['long_term_ema']
    
    # Adjust EMA Difference by Volume Shock
    df['adjusted_ema_difference'] = df['ema_difference'] * df['volume_change']
    
    # Synthesize Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = df['adjusted_momentum'] + df['adjusted_intraday_momentum']
    
    # Calculate High-Low Range Volatility
    df['high_low_volatility'] = df['high_low_range'].rolling(window=7).mean()
    df['intermediate_alpha_factor'] += df['high_low_volatility']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3
    df['vwap_weighted_price'] = df['vwap'] * df['volume']
    df['volume_weighted_price_momentum'] = df['vwap_weighted_price'].rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    df['intermediate_alpha_factor'] += df['volume_weighted_price_momentum']
    
    # Generate Composite Momentum Signal
    df['composite_momentum_signal'] = df['ema_difference'] * df['volume'] + df['price_acceleration']
    
    # Adjust Momentum by Volume Shock
    df['adjusted_composite_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_composite_momentum'] + df['integrated_momentum'] + df['price_acceleration']
    df['final_alpha_factor'] += df['adjusted_ema_difference']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    df['final_alpha_factor'] *= df['intraday_return']
    
    # Refine the Final Factor
    df['final_alpha_factor'] = df['final_alpha_factor'][high_volume_days]
    
    # Adjust Momentum by High-Low Volatility
    df['final_alpha_factor'] *= df['volatility']
    
    return df['final_alpha_factor']
```
