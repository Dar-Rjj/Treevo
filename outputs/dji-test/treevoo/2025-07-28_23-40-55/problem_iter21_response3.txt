```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted EMAs
    df['Close_Volume'] = df['close'] * df['volume']
    short_term_wema = df['Close_Volume'].ewm(span=12, adjust=False).mean()
    long_term_wema = df['Close_Volume'].ewm(span=26, adjust=False).mean()
    
    # WEMA Difference
    wema_diff = short_term_wema - long_term_wema
    
    # Weighted Momentum Signal
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    weighted_momentum_signal = wema_diff * (df['volume'] / avg_volume_26)
    
    # Volume-Adjusted Volatility
    volatility = df['high'] - df['low']
    vol_26 = volatility.rolling(window=26).std()
    volume_adjusted_volatility = vol_26 * (df['volume'] / avg_volume_26)
    
    # Combined Alpha Factor
    combined_alpha_factor = weighted_momentum_signal / volume_adjusted_volatility
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_high_low_range + open_to_close_momentum
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=26).mean()
    high_volume_days = df['volume'] > (1.5 * volume_rolling_mean)
    filtered_integrated_momentum = integrated_momentum.where(high_volume_days, 0)
    
    # Calculate Short-Term EMA
    short_term_ema = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_term_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_term_ema - long_term_ema
    
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = df['close'].diff(7).sum() / 7
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = df['close'].diff(25).sum() / 25
    
    # Adjust Momentum by Volume Shock
    volume_change = df['volume'].pct_change()
    adjusted_momentum = (short_term_price_momentum - long_term_price_momentum) * volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = filtered_integrated_momentum + adjusted_momentum
    
    # Introduce Volume Weighted Price Momentum
    vwap = (df['high'] + df['low'] + 2 * df['close']) / 4
    vwap_weighted = vwap * df['volume']
    vwap_sum_7 = vwap_weighted.rolling(window=7).sum()
    
    # Synthesize Final Factor
    final_factor = synthesized_momentum + vwap_sum_7
    
    # Final Enhanced Factor
    enhanced_factor = final_factor + combined_alpha_factor
    
    return enhanced_factor
```
