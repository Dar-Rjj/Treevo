```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Ratio
    df['price_movement_ratio'] = df['high'] / df['low']
    
    # Evaluate Weighted Average Volume Over Period
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Impact
    df['adjusted_momentum'] = df['momentum']
    df.loc[df['volume'] > df['avg_volume_20'], 'adjusted_momentum'] *= 1.1
    
    # Adjust for Amount Impact
    df['avg_amount_20'] = df['amount'].rolling(window=20).mean()
    df.loc[df['amount'] > df['avg_amount_20'], 'adjusted_momentum'] *= 1.05
    
    # Assess Positive vs Negative Momentum Contribution
    df.loc[df['momentum'] > 0, 'adjusted_momentum'] *= 1.5
    
    # Integrate Accumulated Momentum Impact
    df['long_term_momentum'] = df['adjusted_momentum'].shift(40) * 0.3 + df['adjusted_momentum'].shift(60) * 0.2
    df['integrated_momentum'] = df['adjusted_momentum'] + df['long_term_momentum']
    
    return df['integrated_momentum']
```
