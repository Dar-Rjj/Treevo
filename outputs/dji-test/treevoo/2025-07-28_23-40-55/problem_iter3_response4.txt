```python
def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'].diff()
    
    # Calculate Momentum Indicator
    n = 10  # Define the period for momentum calculation
    df['momentum_indicator'] = df['close'].pct_change(periods=n).rolling(window=n).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['price_velocity'] + df['price_acceleration'] + df['momentum_indicator']) / 3
    
    return df['alpha_factor']
```
