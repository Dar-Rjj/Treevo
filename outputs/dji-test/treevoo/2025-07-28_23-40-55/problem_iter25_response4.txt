```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Momentum Filter by Volume
    volume_rolling_mean = df['volume'].rolling(window=30).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    high_volume_days = df['volume'] > high_volume_threshold
    
    combined_momentum = open_to_close_return * intraday_volatility
    filtered_momentum = combined_momentum[high_volume_days].reindex(df.index, fill_value=0)
    
    # Adjust Combined Momentum by Intraday Volatility
    adjusted_momentum = filtered_momentum * intraday_volatility
    
    # EMA Difference
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    ema_difference = short_term_ema - long_term_ema
    
    # 10-Day Accumulated Volume
    accumulated_volume = df['volume'].rolling(window=10).sum()
    
    # Final Alpha Factor
    alpha_factor = ema_difference * adjusted_momentum * accumulated_volume
    
    return alpha_factor
```
