```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / 5
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / 20
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    total_volume_change = daily_volume_change.rolling(window=5).sum()
    
    # Combine Factors
    combined_factor = short_term_momentum * long_term_momentum * total_volume_change
    
    return combined_factor
```
