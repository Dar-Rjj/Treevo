```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Daily Price Oscillation
    df['price_oscillation'] = (df['high'] - df['low']) / df['open']
    
    # Evaluate Weighted Average Volume Over Period
    df['avg_volume'] = df['volume'].rolling(window=n_days).sum() / n_days
    
    # Calculate Volume-Adjusted Momentum
    df['momentum'] = (df['close'] - df['open'].shift(n_days)) * df['avg_volume']
    
    # Assess Positive vs Negative Momentum Contribution
    def momentum_contribution(row, n_days):
        if row['close'] - row['open'].shift(n_days) > 0:
            return row['momentum'] + 0.5 * row['momentum']
        else:
            return row['momentum'] - 0.25 * row['momentum']
    
    df['adjusted_momentum'] = df.apply(momentum_contribution, args=(n_days,), axis=1)
    
    # Integrate Accumulated Momentum Impact
    time_horizons = [10, 20, 30]
    weights = [1, 2, 3]  # Assign greater weight to longer time horizons
    accumulated_momentum = pd.Series(index=df.index, dtype=float)
    
    for n, w in zip(time_horizons, weights):
        df[f'adjusted_momentum_{n}'] = df.apply(momentum_contribution, args=(n,), axis=1)
        accumulated_momentum += df[f'adjusted_momentum_{n}'] * w
    
    return accumulated_momentum
```
