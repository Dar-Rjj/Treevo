```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted close price
    df['weighted_close'] = df['close'] * df['volume']
    
    # Short-Term WEMA (12-day span)
    short_term_wema = df['weighted_close'].ewm(span=12, adjust=False).mean()
    
    # Long-Term WEMA (26-day span)
    long_term_wema = df['weighted_close'].ewm(span=26, adjust=False).mean()
    
    # WEMA Difference
    wema_diff = short_term_wema - long_term_wema
    
    # Average Volume over 26 days
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    
    # Weighted Momentum Signal
    momentum_signal = wema_diff * (df['volume'] / avg_volume_26)
    
    return momentum_signal
```
