```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    df['price_acceleration'] = df['price_velocity'] - df['price_velocity'].shift(1)
    
    # Calculate Short-Term Momentum (7 days)
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum (30 days)
    df['long_term_momentum'] = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_term_momentum'] - df['long_term_momentum']) * df['volume_change']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap_7_day_sum'] = df['vwap'].rolling(window=7).sum()
    
    # Synthesize Price-Acceleration and Volume-Adjusted Momentum
    df['synthesized_factor'] = df['price_acceleration'] + df['adjusted_momentum']
    
    # Incorporate Volume Weighted Price Momentum
    df['final_enhanced_factor'] = df['synthesized_factor'] + df['vwap_7_day_sum']
    
    return df['final_enhanced_factor'].dropna()
```
