```python
def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    - data: pandas DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the price range
    price_range = data['high'] - data['low']
    
    # Calculate the buying pressure
    buying_pressure = (data['close'] - data['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (data['high'] - data['close']) / price_range
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (data['volume'] * (data['high'] + data['low'] + data['close']) / 3).cumsum() / data['volume'].cumsum()
    
    # Calculate the momentum factor
    momentum = (data['close'] - data['close'].shift(1)) / data['close'].shift(1)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (buying_pressure - selling_pressure) * vwap * momentum
    
    return alpha_factor
```
