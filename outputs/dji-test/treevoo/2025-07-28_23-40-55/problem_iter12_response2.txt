```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span, Close price)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span, Close price)
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Subtract Long-Term EMA from Short-Term EMA
    ema_diff = short_ema - long_ema
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = ema_diff * (df['volume'] / (df['high'] - df['low']))
    
    return volume_adjusted_momentum
```
