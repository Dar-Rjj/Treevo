```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = high_low_range + open_close_momentum
    
    # Apply Volume Filter
    volume_threshold = df['volume'].quantile(0.75)  # Set a threshold for high volume
    high_volume_days = df[df['volume'] > volume_threshold]
    filtered_momentum = integrated_momentum[high_volume_days.index]
    
    # Calculate High-Low Volatility
    high_low_volatility = high_low_range
    
    # Adjust Momentum by Volatility
    adjusted_momentum = filtered_momentum * high_low_volatility
    
    return adjusted_momentum
```
