```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in the average price
    avg_price_change = avg_price.pct_change()
    
    # Calculate the difference between the closing and opening prices
    price_diff = df['close'] - df['open']
    
    # Calculate the percentage change in the volume
    vol_change = df['volume'].pct_change()
    
    # Alpha factor is a combination of the average price change, price difference, and volume change
    alpha_factor = (avg_price_change * price_diff) + (vol_change * price_diff)
    
    return alpha_factor
```
