```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Introduce Volume Weighted Price Momentum
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap_7d_sum = vwap.rolling(window=7).sum()
    
    # Incorporate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=7).mean()
    
    # Synthesize Final Factor
    final_factor = adjusted_momentum + vwap_7d_sum + avg_high_low_range
    
    return final_factor
```
