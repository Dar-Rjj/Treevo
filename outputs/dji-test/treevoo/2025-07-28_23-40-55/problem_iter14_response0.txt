```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(7)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(30)
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_change
    
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_intraday_momentum = high_low_range + open_close_momentum
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Calculate Short-Term EMA
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Volume-Adjusted EMA Difference
    volume_adjusted_ema_difference = ema_difference * (df['volume'] / (df['high'] - df['low']))
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = adjusted_momentum + integrated_intraday_momentum + price_acceleration + volume_adjusted_ema_difference
    
    # Apply Volume Filter
    volume_mean = df['volume'].rolling(window=20).mean()
    high_volume_days = df['volume'] > 1.5 * volume_mean
    final_alpha_factor[~high_volume_days] = 0
    
    return final_alpha_factor
```
