```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = (intraday_range + open_to_close_momentum) / 2
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean * 1.5
    is_high_volume = df['volume'] > high_volume_threshold
    filtered_alpha_factor = integrated_momentum.where(is_high_volume, 0)
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] / df['close'].shift(7) - 1
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] / df['close'].shift(30) - 1
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_factor = filtered_alpha_factor + adjusted_momentum
    
    return synthesized_factor
```
