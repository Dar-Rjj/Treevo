```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = high_low_range + open_close_momentum
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    high_volume_days = df['volume'] > 1.5 * volume_rolling_mean
    filtered_alpha_factor = integrated_momentum[high_volume_days]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    preliminary_alpha_factor = filtered_alpha_factor * intraday_return
    
    # Calculate High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    adjusted_momentum = preliminary_alpha_factor * high_low_volatility
    
    # Calculate Short-Term and Long-Term EMA
    short_term_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=25, adjust=False).mean()
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].pct_change(periods=6)
    long_term_momentum = df['close'].pct_change(periods=30)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum_shock = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Synthesize Momentum and Volume Shock
    synthesized_momentum = adjusted_momentum + adjusted_momentum_shock
    
    # Introduce Volume Weighted Price Momentum
    volume_weighted_price_momentum = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Introduce Volume Weighted Price Oscillator
    vwpo_short_term = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    vwpo_long_term = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    vwpo = vwpo_short_term - vwpo_long_term
    
    # Calculate Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Calculate EMAs
    ema_short_term = df['close'].ewm(span=10, adjust=False).mean()
    ema_long_term = df['close'].ewm(span=30, adjust=False).mean()
    ema_difference_final = ema_short_term - ema_long_term
    
    # Adjusted Momentum Signal
    adjusted_momentum_signal = ema_difference_final * df['volume'] + df['high'] - df['low']
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_momentum * df['close']) + adjusted_momentum_signal + volume_weighted_price_momentum + vwpo
    
    return final_alpha_factor
```
