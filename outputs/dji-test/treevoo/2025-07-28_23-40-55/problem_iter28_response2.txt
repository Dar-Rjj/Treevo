```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    
    # Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']

    # High Volume Filtered Momentum
    # Calculate Volume Rolling Mean
    volume_rolling_mean = df['volume'].rolling(window=20).mean()
    # Set a Threshold for High Volume
    high_volume_threshold = volume_rolling_mean * 1.5
    is_high_volume = df['volume'] > high_volume_threshold
    # Combine Intraday Volatility and Open-to-Close Return
    combined_momentum = intraday_volatility + open_to_close_return
    # Retain Only High Volume Days' Combined Momentum
    high_volume_filtered_momentum = combined_momentum.where(is_high_volume, np.nan)
    
    # Adjusted Momentum by Intraday Volatility
    adjusted_momentum = high_volume_filtered_momentum * intraday_volatility
    
    # EMA Difference
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    ema_difference = short_term_ema - long_term_ema
    
    # Final Alpha Factor
    final_alpha_factor = ema_difference * adjusted_momentum * df['volume'].rolling(window=10).sum()

    return final_alpha_factor
```
