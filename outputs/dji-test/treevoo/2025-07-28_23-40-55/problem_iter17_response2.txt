```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted EMAs
    df['wema_short'] = (df['close'] * df['volume']).ewm(span=12).mean()
    df['wema_long'] = (df['close'] * df['volume']).ewm(span=26).mean()

    # WEMA Difference
    df['wema_diff'] = df['wema_short'] - df['wema_long']

    # Weighted Momentum Signal
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    df['weighted_momentum_signal'] = df['wema_diff'] * (df['volume'] / avg_volume_26)

    # Volume-Adjusted Volatility
    volatility = df[['high', 'low']].diff(axis=1).iloc[:, 0].rolling(window=26).std()
    df['volume_adjusted_volatility'] = volatility * (df['volume'] / avg_volume_26)

    # Combined Alpha Factor
    df['combined_alpha_factor'] = df['weighted_momentum_signal'] / df['volume_adjusted_volatility']

    # Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']

    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_high_low_range'] + df['open_to_close_momentum']

    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=26).mean()
    high_volume_threshold = volume_rolling_mean * 1.5  # Threshold for high volume days
    df['filtered_integrated_momentum'] = df['integrated_momentum'].where(df['volume'] > high_volume_threshold, 0)

    # Short-Term EMA
    df['short_term_ema'] = df['close'].ewm(span=7).mean()

    # Long-Term EMA
    df['long_term_ema'] = df['close'].ewm(span=25).mean()

    # EMA Difference
    df['ema_diff'] = df['short_term_ema'] - df['long_term_ema']

    # Short-Term Price Momentum
    df['short_term_price_momentum'] = df['close'].diff(periods=7).mean()

    # Long-Term Price Momentum
    df['long_term_price_momentum'] = df['close'].diff(periods=25).mean()

    # Adjust Momentum by Volume Shock
    df['adjusted_momentum'] = (df['short_term_price_momentum'] - df['long_term_price_momentum']) * df['volume'].pct_change()

    # Synthesize Momentum and Volume Shock
    df['synthesized_momentum'] = df['filtered_integrated_momentum'] + df['adjusted_momentum']

    # Volume Weighted Price Momentum
    df['volume_weighted_avg_price'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['sum_weighted_prices'] = df['volume_weighted_avg_price'].rolling(window=7).sum()

    # Synthesize Final Factor
    df['final_factor'] = df['synthesized_momentum'] + df['sum_weighted_prices']

    # Final Enhanced Factor
    df['enhanced_factor'] = df['final_factor'] + df['combined_alpha_factor']

    return df['enhanced_factor']
```
