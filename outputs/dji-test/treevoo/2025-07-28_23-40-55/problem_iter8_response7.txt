```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['Velocity'] = df['close'].diff()

    # Calculate Price Acceleration
    df['Acceleration'] = df['Velocity'].diff()

    # Calculate Short-Term EMA (12-day span)
    df['EMA_Short'] = df['close'].ewm(span=12, adjust=False).mean()

    # Calculate Long-Term EMA (26-day span)
    df['EMA_Long'] = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    df['EMA_Difference'] = df['EMA_Short'] - df['EMA_Long']

    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)

    # Adjust for Volume Impact
    df['Volume_Average_20Days'] = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted'] = df.apply(lambda x: x['EMA_Difference'] * 1.1 if x['volume'] > x['Volume_Average_20Days'] else x['EMA_Difference'], axis=1)

    # Calculate Volume-Weighted EMA Difference
    df['Volume_Average_12Days'] = df['volume'].rolling(window=12).mean()
    df['Volume_Weighted_EMA_Diff'] = df['EMA_Difference'] * df['Volume_Average_12Days']

    # Adjust Momentum by Volume Shock
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1)
    df['Adjusted_Momentum'] = (df['Momentum']) * df['Volume_Change']

    # Synthesize Combined Alpha Factor
    df['Alpha_Factor'] = df['Velocity'] + df['Acceleration'] + df['Volume_Adjusted'] + df['Volume_Weighted_EMA_Diff'] + df['Adjusted_Momentum']

    return df['Alpha_Factor']
```
