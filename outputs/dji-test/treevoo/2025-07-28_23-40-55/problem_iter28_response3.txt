```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrated Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_close_momentum
    
    # Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=5).mean()
    high_volume = df['volume'] > volume_rolling_mean
    
    # Short-Term and Long-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Adjust Price Momentum by Volume Shock
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_price_momentum = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Adjust Intraday Momentum by Volatility
    adjusted_intraday_momentum = integrated_momentum * (df['high'] - df['low'])
    
    # Price Velocity
    price_velocity = df['close'] - df['close'].shift(1)
    
    # Price Acceleration
    price_acceleration = price_velocity - price_velocity.shift(1)
    
    # Exponential Moving Averages
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # EMA Difference
    ema_diff = ema_12 - ema_26
    
    # Adjust EMA Difference by Volume Shock
    adjusted_ema_diff = ema_diff * daily_volume_change
    
    # Synthesize Final Alpha Factor
    final_alpha = (adjusted_price_momentum + adjusted_intraday_momentum + price_acceleration + adjusted_ema_diff)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday Return
    final_alpha *= intraday_return
    
    # Refine the Final Factor (Only Include High Volume Days)
    final_alpha[~high_volume] = 0
    
    # Adjust Momentum by High-Low Volatility
    final_alpha *= (df['high'] - df['low'])
    
    return final_alpha
```
