```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term EMA
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_diff = short_ema - long_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_volume = df['volume'].rolling(window=10).mean()
    vol_weighted_ema_diff = ema_diff * avg_volume
    
    # Calculate Momentum Signal
    momentum_signal = vol_weighted_ema_diff * df['volume']
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_momentum = intraday_range + open_to_close_momentum
    
    # Apply Volume Filter
    high_volume_threshold = 1.5  # Example threshold
    volume_mean = df['volume'].rolling(window=10).mean()
    high_volume_days = (df['volume'] > high_volume_threshold * volume_mean)
    filtered_alpha_factor = integrated_momentum.where(high_volume_days, 0)
    
    # Adjust Momentum by Volatility
    volatility = df['high'] - df['low']
    adjusted_filtered_alpha = filtered_alpha_factor * volatility
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()
    
    # Adjust Price Acceleration by Volume
    adjusted_price_acceleration = price_acceleration * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].shift(1) - df['close'].shift(5)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].shift(1) - df['close'].shift(40)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * daily_volume_change
    
    # Calculate Price-Velocity Interaction
    price_velocity_interaction = price_velocity * price_acceleration
    
    # Adjust Price-Velocity Interaction by Volume
    adjusted_price_velocity_interaction = price_velocity_interaction * df['volume']
    
    # Calculate Price-Volatility Interaction
    price_volatility_interaction = price_velocity * volatility
    
    # Adjust Price-Volatility Interaction by Volume
    adjusted_price_volatility_interaction = price_volatility_interaction * df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        momentum_signal + 
        adjusted_price_acceleration + 
        adjusted_filtered_alpha + 
        adjusted_price_velocity_interaction + 
        adjusted_price_volatility_interaction
    )
    
    return final_alpha_factor
```
