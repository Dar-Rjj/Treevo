```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Velocity
    df['price_velocity'] = df['close'].diff()
    
    # Calculate Short-Term EMA (12-day span)
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Generate Composite Momentum Signal
    df['price_acceleration'] = df['price_velocity'].diff()
    df['composite_momentum'] = df['ema_diff'] * df['volume'] + df['price_acceleration']
    
    # Adjust Momentum by Volume Shock
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_momentum'] = df['composite_momentum'] * df['volume_change']
    
    # Introduce Volume Weighted Price Momentum
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap_7_sum'] = df['vwap'].rolling(window=7).sum()
    
    # Incorporate High-Low Range Volatility
    df['daily_range'] = df['high'] - df['low']
    df['avg_range_7'] = df['daily_range'].rolling(window=7).mean()
    
    # Synthesize Intermediate Alpha Factor
    df['intermediate_factor'] = df['adjusted_momentum'] + df['vwap_7_sum'] + df['avg_range_7']
    
    # Calculate Intraday High-Low Range and Open-to-Close Momentum
    df['intraday_range'] = df['high'] - df['low']
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    df['integrated_momentum'] = df['intraday_range'] + df['open_close_momentum']
    
    # Apply Volume Filter
    df['volume_rolling_mean'] = df['volume'].rolling(window=7).mean()
    df['high_volume'] = df['volume'] > df['volume_rolling_mean'] * 1.5
    df['filtered_intermediate_factor'] = df['intermediate_factor'].where(df['high_volume'])
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    
    # Adjust Momentum by Volatility
    df['adjusted_alpha_factor'] = df['filtered_intermediate_factor'] / df['high_low_volatility']
    
    # Incorporate Amount in the Alpha Factor
    df['average_amount'] = df['amount'].rolling(window=7).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount']
    df['final_alpha_factor'] = df['adjusted_alpha_factor'] * df['amount_ratio']
    
    # Incorporate Price and Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=10).corr(df['volume'])
    df['final_alpha_factor'] = 0.8 * df['final_alpha_factor'] + 0.2 * df['price_volume_corr']
    
    return df['final_alpha_factor'].dropna()
```
