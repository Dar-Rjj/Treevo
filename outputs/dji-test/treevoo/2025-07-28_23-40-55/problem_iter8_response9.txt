```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(6)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(29)
    
    # Calculate Daily Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Momentum by Volume Shock
    adjusted_momentum = (short_term_momentum - long_term_momentum) * volume_change
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Integrate Intraday and Open-to-Close Momentum
    integrated_intraday_momentum = (intraday_range + open_close_momentum) / 2
    
    # Calculate Price Velocity
    price_velocity = df['close'].diff()
    
    # Calculate Price Acceleration
    price_acceleration = price_velocity.diff()
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = short_term_ema - long_term_ema
    
    # Calculate Volume-Weighted EMA Difference
    avg_volume = df['volume'].rolling(window=12).mean()
    volume_weighted_ema_difference = ema_difference * avg_volume
    
    # Synthesize Final Alpha Factor
    final_factor = (adjusted_momentum + integrated_intraday_momentum + price_acceleration + volume_weighted_ema_difference)
    
    # Apply Volume Filter
    volume_rolling_mean = df['volume'].rolling(window=10).mean()
    high_volume_threshold = volume_rolling_mean.quantile(0.8)
    high_volume_days = df[df['volume'] > high_volume_threshold]
    
    final_factor = final_factor[final_factor.index.isin(high_volume_days.index)]
    
    return final_factor
```
