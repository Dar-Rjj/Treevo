```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term EMA (12-day span)
    df['ema_short'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate Long-Term EMA (26-day span)
    df['ema_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['ema_short'] - df['ema_long']
    
    # Calculate Volume-Weighted EMA Difference
    df['vol_ave_12'] = df['volume'].rolling(window=12).mean()
    df['vol_weighted_ema_diff'] = df['ema_diff'] * df['vol_ave_12']
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['vol_weighted_ema_diff'] * df['volume']
    
    # Calculate Price Acceleration
    df['price_velocity_t'] = df['close'] - df['close'].shift(1)
    df['price_acceleration'] = df['price_velocity_t'] - df['price_velocity_t'].shift(1)
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].diff(periods=7).rolling(window=7).sum() / 7
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].diff(periods=25).rolling(window=25).sum() / 25
    
    # Calculate Volume Trend
    df['daily_vol_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend'] = df['daily_vol_change'].rolling(window=7).sum()
    
    # Combine Momentum and Volume-Adjusted Price-Acceleration
    df['combined_momentum'] = df['momentum_signal'] + df['price_acceleration']
    
    # Final Combination
    df['final_factor'] = (df['combined_momentum'] * 1.5) * df['short_term_momentum'] * df['volume_trend']
    
    return df['final_factor']
```
