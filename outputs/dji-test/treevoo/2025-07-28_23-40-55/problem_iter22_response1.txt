import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Differences
    short_term_momentum = (df['close'] - df['close'].shift(7)) / 7
    long_term_momentum = (df['close'] - df['close'].shift(30)) / 30
    adjusted_momentum = (short_term_momentum - long_term_momentum) * (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weighted EMAs
    df['weighted_close'] = df['close'] * df['volume']
    short_term_wema = df['weighted_close'].ewm(span=12, adjust=False).mean()
    long_term_wema = df['weighted_close'].ewm(span=26, adjust=False).mean()
    wema_signal = (short_term_wema - long_term_wema) * (df['volume'] / df['volume'].rolling(window=26).mean())
    
    # Volatility
    high_low_diff = df['high'] - df['low']
    volatility = high_low_diff.rolling(window=26).std()
    
    # Volume-Adjusted Volatility
    avg_volume_26 = df['volume'].rolling(window=26).mean()
    volume_adjusted_volatility = volatility * (df['volume'] / avg_volume_26)
    
    # Combined Alpha Factor
    combined_alpha_factor = wema_signal / volume_adjusted_volatility
    
    # Synthesize Final Factor
    final_factor = adjusted_momentum + combined_alpha_factor
    
    return final_factor
