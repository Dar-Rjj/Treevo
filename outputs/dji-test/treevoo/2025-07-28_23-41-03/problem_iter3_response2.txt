```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust by Volume and Amount
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['normalized_volume'] = df['avg_volume'] / df['avg_amount']
    
    # Combine High-Low Range Momentum and Adjusted Volume
    df['factor'] = df['high_low_momentum'] / df['normalized_volume']
    
    return df['factor']
```
