```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    sma_diff_100_200 = sma_100 - sma_200
    sma_diff_50_100 = sma_50 - sma_100

    # Calculate 20-Day Weighted Moving Average of Returns
    daily_return = df['close'].pct_change()
    weighted_returns = (daily_return * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Adjust for Price Volatility
    price_range = df['high'] - df['low']
    avg_price_range = price_range.rolling(window=20).mean()
    adjusted_weighted_returns = weighted_returns - avg_price_range

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Determine Reversal Signal
    reversal_signal = np.where(intraday_return > 0, -1, 1)

    # Multiply Subtracted SMA by Reversal Signal
    combined_sma_reversal = (sma_diff_100_200 + sma_diff_50_100) * reversal_signal

    # Compute Average True Range (ATR)
    true_range = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    atr = true_range.rolling(window=14).mean()

    # Multiply ATR by Combined Signal
    atr_combined_signal = atr * combined_sma_reversal

    # Smooth Result with Exponential Moving Average (EMA)
    ema_atr_combined_signal = atr_combined_signal.ewm(span=20, adjust=False).mean()

    # High-Low Range Momentum with Volume Adjustment
    high_low_range = df['high'] - df['low']
    high_low_momentum = high_low_range - high_low_range.shift(1)
    avg_volume = df['volume'].rolling(window=30).mean()
    high_low_momentum_adjusted = high_low_momentum / avg_volume

    # Final Alpha Factor Combination
    w1 = 0.4
    w2 = 0.3
    w3 = 0.3
    alpha_factor = (combined_sma_reversal + ema_atr_combined_signal) * w1 + high_low_momentum_adjusted * w2 + adjusted_weighted_returns * w3

    return alpha_factor
```
