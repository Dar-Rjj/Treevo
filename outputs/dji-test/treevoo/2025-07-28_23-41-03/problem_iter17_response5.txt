import pandas as pd

def heuristics_v2(df):
    # Intraday and Short-Term Price Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open']
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    s_t_momentum = (sma_50 - sma_100) + (sma_100 - sma_200)
    combined_s_t_momentum = s_t_momentum + intraday_momentum
    
    # Volume-Adjusted Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_return = open_to_close_return * avg_volume
    
    # Volume-Weighted Price Movement
    price_movement = (df['high'] - df['low']) * df['volume']
    
    # Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    prev_high_low_range = high_low_range.shift(1)
    adjusted_high_low_range = high_low_range - prev_high_low_range
    
    # Intraday Reversal Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    reversal_signal = intraday_return.apply(lambda x: -1 if x > 0 else 1)
    atr = (df['high'] - df['low']).abs() + (df['close'].shift(1) - df['high']).abs() + (df['close'].shift(1) - df['low']).abs()
    atr_reversal = atr * reversal_signal
    ema_reversal = atr_reversal.ewm(span=10, adjust=False).mean()
    
    # High-Low Range Momentum with Volume Adjustment
    high_low_momentum = high_low_range - high_low_range.shift(1)
    avg_vol = df['volume'].rolling(window=20).mean()
    high_low_vol_adjusted = high_low_momentum / avg_vol
    
    # Final Alpha Factor Combination
    w1 = 0.4
    w2 = 0.3
    w3 = 0.3
    final_alpha_factor = (combined_s_t_momentum * w1) + (volume_adjusted_return * w2) + (high_low_vol_adjusted * w3)
    
    return final_alpha_factor
