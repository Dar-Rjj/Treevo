```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume-weighted average of buying and selling pressure
    volume_weighted_avg = (buying_pressure * df['volume'] - selling_pressure * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Calculate the momentum based on the close price
    momentum = df['close'] / df['close'].shift(10) - 1
    
    # Combine the factors into a single alpha factor
    alpha_factor = (volume_weighted_avg + momentum) / 2
    
    return alpha_factor
```
