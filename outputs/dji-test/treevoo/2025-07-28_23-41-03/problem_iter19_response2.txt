```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to high (C/H) as a measure of day's performance
    df['C_H_ratio'] = df['close'] / df['high']
    
    # Compute the difference between high and low prices adjusted by volume (VWAP-like metric)
    df['HL_diff'] = df['high'] - df['low']
    df['VWAP_adj_diff'] = df['HL_diff'] * df['volume']
    
    # Measure the gap between opening and closing prices (O-C) to gauge market momentum
    df['O_C_gap'] = df['open'] - df['close']
    
    # Develop a simple moving average (SMA) of close prices
    df['SMA_close_10'] = df['close'].rolling(window=10).mean()
    df['SMA_deviation'] = df['close'] - df['SMA_close_10']
    
    # Construct a volume-weighted moving average (VWMA) of close prices
    df['VWMA_close_10'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['VWMA_deviation'] = df['close'] - df['VWMA_close_10']
    
    # Monitor the volume at significant price levels (support, resistance) to detect potential breakouts or bounces
    df['price_level_volume'] = (df['close'] - df['low']) / (df['high'] - df['low']) * df['volume']
    
    # Combine multiple indicators to create a composite factor
    df['composite_factor'] = (
        0.3 * df['C_H_ratio'] + 
        0.2 * df['VWAP_adj_diff'] + 
        0.15 * df['O_C_gap'] + 
        0.15 * df['SMA_deviation'] + 
        0.1 * df['VWMA_deviation'] + 
        0.1 * df['price_level_volume']
    )
    
    return df['composite_factor']
```
