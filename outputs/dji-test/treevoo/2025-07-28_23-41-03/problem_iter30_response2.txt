def heuristics_v2(df):
    # Price Movement Analysis
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_Diff'] = df['SMA_5'] - df['SMA_20']

    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_20'] = df['close'].pct_change(periods=20)
    
    df['High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range_5'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Avg_High_Low_Range_20'] = df['High_Low_Range'].rolling(window=20).mean()
    df['High_Low_Range_Ratio'] = df['Avg_High_Low_Range_5'] / df['Avg_High_Low_Range_20']

    # Volume Analysis
    df['VMA_5'] = df['volume'].rolling(window=5).mean()
    df['VMA_20'] = df['volume'].rolling(window=20).mean()
    df['VMA_Diff'] = df['VMA_5'] - df['VMA_20']
    
    df['VROC_1'] = df['volume'].pct_change(periods=1)
    df['VROC_5'] = df['volume'].pct_change(periods=5)
    df['VROC_20'] = df['volume'].pct_change(periods=20)

    # Combined Price and Volume Indicators
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['TP'] = (df['high'] + df['low'] + df['close']) / 3
    df['MF'] = df['TP'] * df['volume']
    df['MFI'] = 100 - (100 / (1 + (df['MF'].rolling(window=14).apply(lambda x: x[df['close'].diff() > 0].sum()) /
                                  df['MF'].rolling(window=14).apply(lambda x: x[df['close'].diff() < 0].sum()))))

    # Momentum and Trend Indicators
    df['RSI_14'] = df['close'].rolling(window=14).apply(lambda x: 100 - (100 / (1 + (x.diff()[x.diff() > 0].mean() / x.diff()[x.diff() < 0].abs().mean()))))
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD'] = df['EMA_12'] - df['EMA_26']
    df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['MACD_Histogram'] = df['MACD'] - df['Signal_Line']

    df['BB_Lower'] = df['SMA_20'] - 2 * df['close'].rolling(window=20).std()
