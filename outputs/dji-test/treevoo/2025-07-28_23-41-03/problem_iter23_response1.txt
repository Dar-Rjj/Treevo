```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['100_day_SMA'] = df['close'].rolling(window=100).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()

    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)

    # Momentum Signals
    df['momentum_100_200'] = df['100_day_SMA'] - df['200_day_SMA']
    df['momentum_50_100'] = df['50_day_SMA'] - df['100_day_SMA']

    # High-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Volume Adjustment
    n_days = 10
    df['avg_volume'] = df['volume'].rolling(window=n_days).sum() / n_days

    # Adjusted High-Low Range Momentum
    df['adj_high_low_range_momentum'] = df['high_low_range_momentum'] / df['avg_volume']

    # Combined Momentum and Reversal
    df['combined_momentum_reversal_100_200'] = df['momentum_100_200'] * df['reversal_signal']
    df['combined_momentum_reversal_50_100'] = df['momentum_50_100'] * df['reversal_signal']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_momentum_reversal_100_200'] + 
                                df['combined_momentum_reversal_50_100'] + 
                                df['adj_high_low_range_momentum'])

    return df['final_alpha_factor']
```
