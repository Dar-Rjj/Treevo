```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Analyze price patterns
    df['daily_range'] = df['high'] - df['low']
    df['high_low_ratio'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Measure price momentum
    df['5_day_ma_return'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_ma_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Evaluate volume trends
    df['5_day_ma_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    df['vol_ratio'] = df['5_day_ma_volume'] / df['20_day_ma_volume']
    
    # Assess trading activity
    df['5_day_ma_amount'] = df['amount'].rolling(window=5).mean()
    df['20_day_ma_amount'] = df['amount'].rolling(window=20).mean()
    df['amt_ratio'] = df['5_day_ma_amount'] / df['20_day_ma_amount']
    
    # Combine indicators
    df['composite_factor'] = (df['5_day_ma_return'] + df['20_day_ma_return'] 
                              + df['vol_ratio'] + df['amt_ratio'])
    df['weighted_composite_factor'] = df['composite_factor'] * df['high_low_ratio']
    
    # Evaluate recent performance
    df['5_day_cumulative_return'] = (df['close'] / df['close'].shift(5) - 1)
    df['20_day_cumulative_return'] = (df['close'] / df['close'].shift(20) - 1)
    
    # Analyze volatility
    df['20_day_std_return'] = df['daily_return'].rolling(window=20).std()
    
    # Identify reversal signals
    df['RSI'] = 100 - (100 / (1 + (df['daily_return'].rolling(window=14).apply(lambda x: np.mean(x[x > 0])) / -np.mean(df['daily_return'].rolling(window=14).apply(lambda x: x[x < 0])))))
    df['RSI_signal'] = ((df['RSI'] > 70).astype(int) - (df['RSI'] < 30).astype(int))
    
    # Detect trend continuation
    df['MACD_line'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    df['Signal_line'] = df['MACD_line'].ewm(span=9, adjust=False).mean()
    df['MACD_signal'] = ((df['MACD_line'] > df['Signal_line']).astype(int) - (df['MACD_line'] < df['Signal_line']).astype(int))
    
    # Evaluate market sentiment
    df['5_day_change_vol'] = df['5_day_ma_volume'].pct_change(periods=5)
    df['vol_change_signal'] = (df['5_day_change_vol'] > 0).astype(int)
    
    # Finalize alpha factor
    df['alpha_factor'] = (df['weighted_composite_factor'] 
                          + df['RSI_signal'] 
                          + df['MACD_signal'] 
                          + df['vol_change_signal'])
    
    return df['alpha_factor']
```
