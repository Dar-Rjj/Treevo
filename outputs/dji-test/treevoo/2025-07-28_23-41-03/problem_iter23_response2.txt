```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate Rate of Change (ROC)
    df['ROC_10'] = df['close'].pct_change(periods=10)
    df['ROC_30'] = df['close'].pct_change(periods=30)
    
    # Determine the Direction of Price Movement
    df['Positive_Direction'] = df['close'] > df['open']
    df['Negative_Direction'] = df['close'] < df['open']
    
    # Calculate the Average True Range (ATR)
    df['TR'] = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Compute the Daily Price Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Identify Breakouts
    df['Breakout_SMA_50'] = (df['high'] > df['SMA_50']) & (df['close'] > df['open'])
    df['Breakout_SMA_200'] = (df['high'] > df['SMA_200']) & (df['close'] > df['open'])
    
    # Detect Reversals
    df['Reversal_SMA_50'] = (df['low'] < df['SMA_50']) & (df['close'] < df['open'])
    df['Reversal_SMA_200'] = (df['low'] < df['SMA_200']) & (df['close'] < df['open'])
    
    # Evaluate Strength of Trend
    df['Trend_Up'] = (df['ROC_10'] > 0) & (df['ROC_30'] > 0)
    df['Trend_Down'] = (df['ROC_10'] < 0) & (df['ROC_30'] < 0)
    
    # Analyze Volume Patterns
    df['Volume_Increase'] = df['volume'] > df['volume'].shift(1)
    df['Volume_Decrease'] = df['volume'] < df['volume'].shift(1)
    
    # Examine Amount Trends
    df['Amount_Increase'] = df['amount'] > df['amount'].shift(1)
    df['Amount_Decrease'] = df['amount'] < df['amount'].shift(1)
    
    # Correlate Price and Volume Movements
    df['Price_Volume_Up'] = (df['close'] > df['close'].shift(1)) & (df['Volume_Increase'])
    df['Price_Volume_Down'] = (df['close'] < df['close'].shift(1)) & (df['Volume_Decrease'])
    
    # Combine Momentum and Volume
    df['Momentum_Volume_Up'] = (df['ROC_10'] > 0) & (df['Volume_Increase'])
    df['Momentum_Volume_Down'] = (df['ROC_30'] < 0) & (df['Volume_Decrease'])
    
    # Integrate Volatility and Amount
    threshold = df['ATR_14'].mean()  # Example threshold, can be adjusted
    df['Volatility_Amount_Up'] = (df['ATR_14'] > threshold) & (df['Amount_Increase'])
    df['Volatility_Amount_Down'] = (df['ATR_14'] < threshold) & (df['Amount_Decrease'])
    
    # Construct Composite Alpha Factor
    df['Composite_Alpha_Factor'] = (df['ROC_10'] > 0) & (df['ATR_14'] > threshold) & (df['Volume_Increase'])
    
    return df['Composite_Alpha_Factor']
```
