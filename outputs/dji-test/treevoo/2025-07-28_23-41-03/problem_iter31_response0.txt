```python
import pandas as pd

def heuristics_v2(df):
    # Momentum and Volume-Weighted Price Movement Factors
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['momentum_factor'] = df['10_day_SMA'] - df['20_day_SMA']
    
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    df['10_day_EMA_VWAP'] = df['VWAP'].ewm(span=10, adjust=False).mean()
    df['50_day_EMA_VWAP'] = df['VWAP'].ewm(span=50, adjust=False).mean()
    df['vwap_ema_ratio'] = df['10_day_EMA_VWAP'] / df['50_day_EMA_VWAP']
    
    df['momentum_vwap_factor'] = df['momentum_factor'] * df['vwap_ema_ratio']
    
    # Hybrid Volatility and RSI Factor
    df['ATR'] = df['high'] - df['low']
    df['14_day_ATR_SMA'] = df['ATR'].rolling(window=14).mean()
    df['daily_price_change'] = df['close'].diff()
    df['gain'] = df['daily_price_change'].apply(lambda x: max(x, 0))
    df['loss'] = df['daily_price_change'].apply(lambda x: -min(x, 0))
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['RSI'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))
    
    df['volatility_rsi_factor'] = df['14_day_ATR_SMA'] + df['RSI']
    
    # Combined MFI and Momentum Factor
    df['TP'] = (df['high'] + df['low'] + df['close']) / 3
    df['MF'] = df['TP'] * df['volume']
    df['PMF'] = df['MF'].where(df['TP'] > df['TP'].shift(1), 0)
    df['NMF'] = df['MF'].where(df['TP'] < df['TP'].shift(1), 0)
    df['14_day_PMF'] = df['PMF'].rolling(window=14).sum()
    df['14_day_NMF'] = df['NMF'].rolling(window=14).sum()
    df['MR'] = df['14_day_PMF'] / df['14_day_NMF']
    df['MFI'] = 100 - (100 / (1 + df['MR']))
    
    df['mfi_momentum_factor'] = df['MFI'] * df['momentum_factor']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['momentum_vwap_factor'] + df['volatility_rsi_factor'] + df['mfi_momentum_factor']
    
    return df['alpha_factor']
```
