import pandas as pd
import numpy as np

def heuristics_v2(df, N=30, w1=0.7, w2=0.3):
    # Price Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # 50-day EMA - 100-day EMA + 100-day EMA - 200-day EMA
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    df['ema_200'] = df['close'].ewm(span=200, adjust=False).mean()
    df['combined_ema_momentum'] = (df['ema_50'] - df['ema_100']) + (df['ema_100'] - df['ema_200'])
    
    # Add Intraday Momentum
    df['total_momentum'] = df['combined_ema_momentum'] + df['intraday_momentum']
    
    # Volume-Adjusted Factors
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    df['volume_adjusted_return'] = df['intraday_momentum'] * (df['volume'] / df['average_volume'])
    df['volume_weighted_price_movement'] = (df['high'] - df['low']) * df['volume']
    
    # Adjusted High-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    df['adjusted_high_low_range_momentum'] = df['high_low_range'] - df['prev_high_low_range']
    df['adjusted_high_low_range_momentum_vol_adj'] = df['adjusted_high_low_range_momentum'] / df['volume'].shift(1)
    
    # Final Alpha Factor Combination
    df['momentum_and_volume_adjusted_factors'] = (
        df['total_momentum'] + df['volume_adjusted_return'] + df['volume_weighted_price_movement']
    )
    df['final_alpha_factor'] = (
        w1 * df['momentum_and_volume_adjusted_factors'] + 
        w2 * df['adjusted_high_low_range_momentum_vol_adj']
    )
    
    return df['final_alpha_factor'].dropna()
