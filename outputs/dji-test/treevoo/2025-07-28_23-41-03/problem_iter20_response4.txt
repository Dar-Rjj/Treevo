```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Volume-Weighted Average Return
    df['weighted_return'] = df['daily_return'] * df['volume']
    df['20_day_vol_weighted_return'] = df['weighted_return'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Adjust for Price Volatility
    df['20_day_avg_price_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_momentum'] = df['20_day_vol_weighted_return'] - df['20_day_avg_price_range']

    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['low']

    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']

    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factors'] = df['adjusted_momentum'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return']

    # Price Momentum Component
    df['50_sma'] = df['close'].rolling(window=50).mean()
    df['100_sma'] = df['close'].rolling(window=100).mean()
    df['200_sma'] = df['close'].rolling(window=200).mean()
    df['price_momentum'] = (df['50_sma'] - df['100_sma']) + (df['100_sma'] - df['200_sma'])

    # Intraday Reversal Momentum and High-Low Range Components
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)
    df['true_range'] = df[['high', 'low']].diff(axis=1).max(axis=1).abs().fillna(0)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['atr_reversal'] = df['atr_14'] * df['reversal_signal']
    df['ema_short'] = df['atr_reversal'].ewm(span=10, adjust=False).mean()
    df['high_low_range'] = df['high'] - df['low']
    df['adjusted_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['high_low_range_momentum'] = df['adjusted_high_low_range'] / df['volume'].rolling(window=20).mean()

    # Final Alpha Factor Combination
    w1, w2 = 0.7, 0.3
    df['alpha_factor'] = (df['combined_factors'] + df['price_momentum']) * w1 + df['high_low_range_momentum'] * w2

    return df['alpha_factor']
```
