```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)
    
    # Calculate Momentum Component
    df['previous_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].sub(df['previous_close'], axis=0).abs().max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    df['momentum_component'] = df['reversal_signal'] * df['atr']
    df['smoothed_momentum'] = df['momentum_component'].ewm(span=5, adjust=False).mean()
    
    # Calculate 5-Day Price Momentum
    df['5_day_price_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 5-Day Volume Surge
    df['5_day_volume_surge'] = df['volume'] - df['volume'].shift(5)
    
    # Combine Momentum and Volume Indicators
    df['weighted_5_day_price_momentum'] = 0.5 * df['5_day_price_momentum']
    df['weighted_5_day_volume_surge'] = 0.5 * df['5_day_volume_surge']
    
    # Integrate Intraday Reversal and 5-Day Momentum
    df['integrated_momentum'] = df['smoothed_momentum'] + df['weighted_5_day_price_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['integrated_momentum'] + df['weighted_5_day_volume_surge']
    
    return df['alpha_factor']
```
