```python
import pandas as pd

def heuristics_v2(df):
    # Price Momentum
    ## Intraday Momentum
    df['Intraday_Momentum'] = df['close'] - df['open']
    
    ## Simple Moving Averages (SMA)
    df['50_SMA'] = df['close'].rolling(window=50).mean()
    df['100_SMA'] = df['close'].rolling(window=100).mean()
    df['200_SMA'] = df['close'].rolling(window=200).mean()
    
    ## Combined Momentum
    df['Combined_Momentum'] = (df['50_SMA'] - df['100_SMA']) + (df['100_SMA'] - df['200_SMA'])
    df['Combined_Momentum'] += df['Intraday_Momentum']

    # Volume-Adjusted Factors
    ## Volume-Adjusted Return
    average_volume = df['volume'].rolling(window=20).mean()  # Use last N days, here N is 20
    df['Volume_Adjusted_Return'] = (df['close'] - df['open']) * (df['volume'] / average_volume)

    ## Volume-Weighted Price Movement
    df['Volume_Weighted_Price_Movement'] = (df['high'] - df['low']) * df['volume']

    # High-Low Range Momentum
    ## High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    ## Adjusted High-Low Range
    df['Adjusted_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    ## High-Low Range Momentum
    df['High_Low_Range_Momentum'] = df['Adjusted_High_Low_Range'] / average_volume

    # Final Alpha Factor Combination
    w1 = 0.7  # Weight for momentum and volume-adjusted factors
    w2 = 0.3  # Weight for high-low range momentum
    df['Alpha_Factor'] = (w1 * df['Combined_Momentum'] + df['Volume_Adjusted_Return'] + df['Volume_Weighted_Price_Movement']) + \
                         (w2 * df['High_Low_Range_Momentum'])

    return df['Alpha_Factor']
```
