```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Combined Momentum
    combined_momentum = (sma_50 - sma_100) + (sma_100 - sma_200) + intraday_momentum
    
    # Volume-Adjusted Return
    avg_volume = df['volume'].rolling(window=N).mean()
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    # Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Volume-Adjusted Intraday Range
    volume_adjusted_intraday_range = (df['high'] - df['low']) * (df['volume'] / avg_volume)
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjusted High-Low Range
    adjusted_high_low_range = high_low_range - high_low_range.shift(1)
    
    # High-Low Range Momentum
    high_low_range_momentum = adjusted_high_low_range / avg_volume
    
    # Volume-Adjusted High-Low Range Momentum
    volume_adjusted_high_low_range_momentum = (high_low_range_momentum * df['volume']) / avg_volume
    
    # 20-Day Weighted Moving Average of Returns
    daily_price_return = df['close'] / df['close'].shift(1) - 1
    weighted_returns = (daily_price_return * df['volume']).rolling(window=N).sum()
    total_volume = df['volume'].rolling(window=N).sum()
    weighted_moving_average_of_returns = weighted_returns / total_volume
    
    # 20-Day Average Daily Price Range
    average_daily_price_range = (df['high'] - df['low']).rolling(window=N).sum() / N
    
    # 20-Day Standard Deviation of Returns
    average_return = daily_price_return.rolling(window=N).mean()
    std_dev_returns = np.sqrt(((daily_price_return - average_return)**2).rolling(window=N).sum() / N)
    
    # Adjust 20-Day Weighted Moving Average of Returns
    adjusted_weighted_moving_average_of_returns = (weighted_moving_average_of_returns - average_daily_price_range) / std_dev_returns
    
    # Intraday Return Ratio
    intraday_return_ratio = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Weighted Open-to-Close Return
    weighted_open_to_close_return = (df['close'] - df['open']) * df['volume']
    
    # Final Alpha Factor Combination
    final_alpha_factor = (
        combined_momentum + 
        volume_adjusted_return + 
        volume_weighted_price_movement + 
        volume_adjusted_intraday_range + 
        high_low_range_momentum / avg_volume + 
        volume_adjusted_high_low_range_momentum / avg_volume + 
        adjusted_weighted_moving_average_of_returns + 
        intraday_return_ratio + 
        weighted_open_to_close_return
    )
    
    return final_alpha_factor
```
