```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Ratio
    intraday_return_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Weighted Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    weighted_open_to_close_return = open_to_close_return * df['volume']
    
    # Calculate Volume Flow
    volume_flow = df['volume'] * (df['high'] - df['low'])
    
    # Combine Factors
    combined_factor = intraday_return_ratio + weighted_open_to_close_return + volume_flow
    
    return combined_factor
```
