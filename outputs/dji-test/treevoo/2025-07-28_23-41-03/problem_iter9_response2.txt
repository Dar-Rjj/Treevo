```python
import pandas as pd

def heuristics_v2(df, n=5, m=5):
    # Calculate Intraday Momentum
    high_to_close = df['high'] - df['close']
    open_to_low = df['open'] - df['low']
    
    # Integrate Volume Thrust
    daily_volume_increase = df['volume'].diff().shift(-1)
    weighted_high_to_close = high_to_close * daily_volume_increase
    weighted_open_to_low = open_to_low * daily_volume_increase
    
    intraday_momentum = weighted_high_to_close + weighted_open_to_low
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    high_low_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Integrate Volume and Amount
    avg_volume = df['volume'].rolling(window=n).mean()
    avg_amount = df['amount'].rolling(window=m).mean()
    adjusted_volume = avg_volume * avg_amount
    
    # Combine High-Low Range Momentum and Adjusted Volume
    alpha_factor = high_low_range_momentum * adjusted_volume
    
    return alpha_factor
```
