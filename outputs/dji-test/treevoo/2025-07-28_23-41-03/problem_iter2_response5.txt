import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movements
    high_to_low_change = df['high'] - df['low']
    open_to_close_change = df['close'] - df['open']

    # Integrate Volume Thrust
    volume_trend = df['volume'] / df['volume'].shift(1)
    smoothed_volume = df['volume'].ewm(span=7).mean()
    
    # Weigh High-to-Low and Open-to-Close Price Changes by Volume Trend
    weighted_high_to_low = high_to_low_change * (volume_trend - 1)
    weighted_open_to_close = open_to_close_change * (volume_trend - 1)

    # Analyze Intraday Price and Volume Relationships
    intraday_price_range_percent = (df['high'] - df['low']) / df['low'] * 100
    vwap = ((df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()

    # Analyze Volume Direction
    positive_volume_flow = df.loc[df['close'] > df['open'], 'volume']
    negative_volume_flow = df.loc[df['close'] <= df['open'], 'volume']

    # Combine all components into a single alpha factor
    alpha_factor = (weighted_high_to_low + weighted_open_to_close) * vwap * (positive_volume_flow - negative_volume_flow)

    return alpha_factor
