```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_return'] = df['daily_price_return'] * df['volume']
    df['sum_weighted_return'] = df['weighted_return'].rolling(window=22).sum()
    df['sum_volume'] = df['volume'].rolling(window=22).sum()
    df['20_day_wma_returns'] = df['sum_weighted_return'] / df['sum_volume']
    
    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    df['20_day_avg_price_range'] = df['price_range'].rolling(window=22).mean()
    df['final_factor_value'] = df['20_day_wma_returns'] - df['20_day_avg_price_range']
    
    # Combine Factors
    # Compute Average True Range (ATR)
    df['tr1'] = abs(df['high'] - df['low'])
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=22).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Add to 20-Day Weighted Moving Average of Returns
    df['combined_factor'] = df['20_day_wma_returns'] + df['atr_reversal']
    
    # Apply Exponential Moving Average (EMA) with Short Period
    df['final_alpha'] = df['combined_factor'].ewm(span=7, adjust=False).mean()
    
    return df['final_alpha']
```
