```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Combined Momentum
    combined_momentum = (sma_50 - sma_100) + (sma_100 - sma_200) + intraday_momentum
    
    # Volume-Adjusted Return
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    # Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Volume-Adjusted Intraday Range
    volume_adjusted_intraday_range = (df['high'] - df['low']) * (df['volume'] / avg_volume)
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjusted High-Low Range
    adjusted_high_low_range = high_low_range - high_low_range.shift(1)
    
    # High-Low Range Momentum
    high_low_range_momentum = adjusted_high_low_range / avg_volume
    
    # Volume-Adjusted High-Low Range Momentum
    volume_adjusted_high_low_range_momentum = (high_low_range_momentum * df['volume']) / avg_volume
    
    # 20-Day Weighted Moving Average of Returns
    daily_price_return = (df['close'] / df['close'].shift(1)) - 1
    weighted_sum_returns = (daily_price_return * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    weighted_avg_returns = weighted_sum_returns / total_volume
    
    # 20-Day Average Daily Price Range
    avg_daily_price_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Adjust for Price Volatility
    adjusted_weighted_avg_returns = weighted_avg_returns - avg_daily_price_range
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Reversal Signal
    reversal_signal = np.where(intraday_return > 0, -1, 1)
    
    # Multiply Subtracted SMA by Reversal Signal
    subtracted_sma = (sma_100 - sma_200) * reversal_signal
    subtracted_sma_50_100 = (sma_50 - sma_100) * reversal_signal
    
    # Average True Range (ATR)
    tr = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    atr = tr.rolling(window=20).mean()
    
    # Multiply ATR by Combined Signal
    atr_combined_signal = atr * (subtracted_sma + subtracted_sma_50_100)
    
    # Smooth Result with Exponential Moving Average (EMA)
    ema_atr_signal = atr_combined_signal.ewm(span=20).mean()
    
    # Final Alpha Factor Combination
    w1, w2, w3, w4, w5, w6, w7 = 1, 1, 1, 1, 1, 1, 1  # Weights can be adjusted
    final_alpha_factor = (combined_momentum * w1) + \
                         (high_low_range_momentum / avg_volume * w2) + \
                         (volume_adjusted_high_low_range_momentum / avg_volume * w3) + \
                         (adjusted_weighted_avg_returns * w4) + \
                         (intraday_return * w5) + \
                         (volume_adjusted_return * w6) + \
                         (ema_atr_signal * w7)
    
    return final_alpha_factor
```
