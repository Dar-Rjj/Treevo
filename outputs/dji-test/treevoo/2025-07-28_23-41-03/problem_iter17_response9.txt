```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the net buying pressure
    net_buying_pressure = buying_pressure - selling_pressure
    
    # Calculate the volume weighted net buying pressure
    volume_weighted_net_bp = net_buying_pressure * df['volume']
    
    # Calculate the factor value
    factor_value = volume_weighted_net_bp.rolling(window=5).mean()
    
    return factor_value
```
