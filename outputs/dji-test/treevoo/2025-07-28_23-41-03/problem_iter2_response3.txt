```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range Percentage
    df['intraday_range'] = (df['high'] - df['low']) / df['low'] * 100
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (df['vwap_price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Analyze Volume Direction
    df['volume_flow'] = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else -x['volume'], axis=1)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Average True Range (ATR)
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high' - 'low', 'high' - 'prev_close', 'prev_close' - 'low']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Momentum Component
    df['momentum_component'] = df['reversal_signal'] * df['atr']
    df['smooth_momentum'] = df['momentum_component'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smooth_momentum']
    
    return alpha_factor
```
