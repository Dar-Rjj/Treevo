```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Components
    df['intraday_momentum'] = df['close'] - df['open']
    
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    
    df['combined_momentum'] = (df['sma_10'] - df['sma_20']) + (df['sma_50'] - df['sma_100'])
    df['combined_momentum'] += df['intraday_momentum']
    
    # Volume-Adjusted Factors
    avg_volume = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_return'] = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap_ema_10'] = df['vwap'].ewm(span=10, adjust=False).mean()
    df['vwap_ema_50'] = df['vwap'].ewm(span=50, adjust=False).mean()
    df['vwap_ema_ratio'] = df['vwap_ema_10'] / df['vwap_ema_50']
    
    # High-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['adjusted_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['high_low_range_momentum'] = df['adjusted_high_low_range'] / avg_volume
    
    # Volume-Adjusted High-Low Range Momentum
    df['volume_adjusted_high_low_range_momentum'] = (df['high_low_range_momentum'] * df['volume']) / avg_volume
    
    # Calculate 20-Day Weighted Moving Average of Returns
    daily_returns = df['close'].pct_change()
    weighted_returns = (daily_returns * df['volume']).rolling(window=20).sum()
    sum_volumes = df['volume'].rolling(window=220).sum()
    df['weighted_moving_avg_returns'] = weighted_returns / sum_volumes
    
    # Adjust for Price Volatility
    df['avg_daily_price_range'] = df['high_low_range'].rolling(window=20).mean()
    avg_return = daily_returns.rolling(window=20).mean()
    std_dev_returns = np.sqrt(((daily_returns - avg_return) ** 2).rolling(window=20).sum() / 20)
    df['adjusted_weighted_returns'] = (df['weighted_moving_avg_returns'] - df['avg_daily_price_range']) / std_dev_returns
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_open_to_close_return'] = (df['close'] - df['open']) * df['volume']
    
    # Final Alpha Factor Combination
    w1, w2, w3, w4, w5, w6 = 0.2, 0.1, 0.1, 0.3, 0.1, 0.2
    df['final_alpha_factor'] = (df['combined_momentum'] + df['vwap_ema_ratio']) * w1
    df['final_alpha_factor'] += (df['high_low_range_momentum'] / avg_volume) * w2
    df['final_alpha_factor'] += (df['volume_adjusted_high_low_range_momentum'] / avg_volume) * w3
    df['final_alpha_factor'] += df['adjusted_weighted_returns'] * w4
    df['final_alpha_factor'] += df['intraday_return_ratio'] * w5
    df['final_alpha_factor'] += df['weighted_open_to_close_return'] * w6
    
    return df['final_alpha_factor']
```
