import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    sma_diff_100_200 = sma_100 - sma_200
    sma_diff_50_100 = sma_50 - sma_100
    
    # Calculate Intraday Price Range Percentage
    intraday_range_pct = (df['high'] - df['low']) / df['low'] * 100
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Analyze Volume Direction
    positive_volume_flow = df.loc[df['close'] > df['open'], 'volume']
    negative_volume_flow = df.loc[df['close'] <= df['open'], 'volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    
    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(df['close'] > df['open'], 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(df['close'] <= df['open'], 0).rolling(window=14).sum()
    money_flow_ratio = positive_money_flow / negative_money_flow
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Analyze Intraday Momentum
    momentum = (df['close'] - df['open']) / df['open'] * 100
    positive_momentum = momentum.where(momentum > 0, 0)
    negative_momentum = momentum.where(momentum <= 0, 0)
    
    # Calculate Price Momentum Over 5 Days
    price_momentum_5d = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Surge Over 5 Days
    volume_surge_5d = df['volume'] - df['volume'].shift(5)
    
    # Combine Momentum and Volume Indicators
    weighted_price_momentum = price_momentum_5d * 0.5
    weighted_volume_surge = volume_surge_5d * 0.5
    
    # Sum the Weighted Indicators with SMA Differences
    alpha_factor = (weighted_price_momentum + 
                    weighted_volume_surge + 
                    sma_diff_100_200 + 
                    sma_diff_50_100)
    
    return alpha_factor
