```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['SMA_diff_100_200'] = df['SMA_100'] - df['SMA_200']
    df['SMA_diff_50_100'] = df['SMA_50'] - df['SMA_100']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Combine Momentum and Reversal Signals
    df['combined_signal_100_200'] = df['SMA_diff_100_200'] * df['reversal_signal']
    df['combined_signal_50_100'] = df['SMA_diff_50_100'] * df['reversal_signal']
    
    # Compute Average True Range (ATR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['TR'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Integrate ATR with Combined Signal
    df['ATR_combined_100_200'] = df['ATR'] * df['combined_signal_100_200']
    df['ATR_combined_50_100'] = df['ATR'] * df['combined_signal_50_100']
    
    # Smooth Result with Exponential Moving Average (EMA)
    df['EMA_ATR_combined_100_200'] = df['ATR_combined_100_200'].ewm(span=7, adjust=False).mean()
    df['EMA_ATR_combined_50_100'] = df['ATR_combined_50_100'].ewm(span=7, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = 0.5 * (df['EMA_ATR_combined_100_200'] + df['EMA_ATR_combined_50_100'])
    
    return df['alpha_factor']
```
