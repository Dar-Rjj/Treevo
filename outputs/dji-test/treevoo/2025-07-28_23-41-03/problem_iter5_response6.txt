import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['100_day_SMA'] = df['close'].rolling(window=100).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    df['SMA_diff_100_200'] = df['100_day_SMA'] - df['200_day_SMA']
    df['SMA_diff_50_100'] = df['50_day_SMA'] - df['100_day_SMA']

    # Calculate Intraday Momentum
    df['intraday_high_return'] = (df['high'] - df['open']) / df['open']
    df['intraday_low_return'] = (df['low'] - df['open']) / df['open']
    df['intraday_close_return'] = (df['close'] - df['open']) / df['open']
    df['average_intraday_return'] = (df['intraday_high_return'] + df['intraday_low_return'] + df['intraday_close_return']) / 3

    # Calculate Daily Volatility
    df['log_returns'] = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    df['daily_volatility'] = df['log_returns'].rolling(window=50).std()

    # Combine Intraday Momentum and Daily Volatility
    df['intraday_volatility_factor'] = df['average_intraday_return'] * df['daily_volatility']
    df['adjusted_intraday_volatility'] = df['intraday_volatility_factor'] / df['volume']

    # Calculate Price Momentum Over 5 Days
    df['price_momentum_5d'] = df['close'] - df['close'].shift(5)

    # Calculate Volume Surge Over 5 Days
    df['volume_surge_5d'] = df['volume'] - df['volume'].shift(5)

    # Combine Momentum and Volume Indicators
    df['weighted_price_momentum'] = 0.5 * df['price_momentum_5d']
    df['weighted_volume_surge'] = 0.5 * df['volume_surge_5d']

    # Sum the Weighted Indicators with SMA Differences
    df['combined_factor'] = df['weighted_price_momentum'] + df['weighted_volume_surge'] + df['SMA_diff_100_200'] + df['SMA_diff_50_100']

    # Introduce Weighting by Trading Volume
    df['volume_20_day_MA'] = df['volume'].rolling(window=20).mean()
    df['final_factor'] = df['combined_factor'] * df['adjusted_intraday_volatility'] / df['volume_20_day_MA']

    return df['final_factor']
