```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['SMA_diff_100_200'] = df['SMA_100'] - df['SMA_200']
    df['SMA_diff_50_100'] = df['SMA_50'] - df['SMA_100']
    
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_returns'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['average_daily_price_range'] = (df['high'] - df['low']).rolling(window=22).mean()
    df['adjusted_weighted_returns'] = df['weighted_returns'] - df['average_daily_price_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Multiply Subtracted SMA by Reversal Signal
    df['SMA_diff_100_200_reversed'] = df['SMA_diff_100_200'] * df['reversal_signal']
    df['SMA_diff_50_100_reversed'] = df['SMA_diff_50_100'] * df['reversal_signal']
    
    # Compute Average True Range (ATR)
    df['tr'] = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    df['ATR'] = df['tr'].rolling(window=14).mean()
    
    # Multiply ATR by Combined Signal
    df['ATR_signal'] = df['ATR'] * (df['SMA_diff_100_200_reversed'] + df['SMA_diff_50_100_reversed'])
    
    # Smooth Result with Exponential Moving Average (EMA)
    df['EMA_ATR_signal'] = df['ATR_signal'].ewm(span=10, adjust=False).mean()
    
    # High-Low Range Momentum with Volume Adjustment
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['average_volume_30'] = df['volume'].rolling(window=30).mean()
    df['high_low_range_adjusted'] = df['high_low_range_momentum'] / df['average_volume_30']
    
    # Final Alpha Factor Combination
    w1 = 0.6
    w2 = 0.4
    df['final_alpha_factor'] = (df['SMA_diff_50_100_reversed'] + df['SMA_diff_100_200_reversed'] + df['EMA_ATR_signal']) * w1 + (df['high_low_range_adjusted'] + df['adjusted_weighted_returns']) * w2
    
    return df['final_alpha_factor']
```
