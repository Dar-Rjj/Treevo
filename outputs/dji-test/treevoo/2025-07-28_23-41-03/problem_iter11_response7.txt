```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday and Short-Term Price Momentum
    intraday_momentum = df['close'] - df['open']
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    combined_sma = (sma_50 - sma_100) + (sma_100 - sma_200)
    short_term_momentum = combined_sma + intraday_momentum
    
    # Volume-Adjusted Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    avg_volume = df['volume'].rolling(window=20).mean()  # Last 20 days volume
    volume_adjusted_return = open_to_close_return * avg_volume
    
    # Volume-Weighted Price Movement
    vwp_movement = (df['high'] - df['low']) * df['volume']
    
    # Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    prev_high_low_range = high_low_range.shift(1)
    adjusted_high_low_range = high_low_range - prev_high_low_range
    
    # Intraday Reversal Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    reversal_signal = np.sign(intraday_return) * -1  # Reverse the signal
    atr = (df['high'] - df['low']).abs() + (df['high'] - df['close'].shift(1)).abs() + (df['low'] - df['close'].shift(1)).abs()
    atr_reversal = atr * reversal_signal
    ema_atr_reversal = atr_reversal.ewm(span=10, adjust=False).mean()
    
    # High-Low Range Momentum with Volume Adjustment
    hl_range_momentum = high_low_range - high_low_range.shift(1)
    avg_volume_hl = df['volume'].rolling(window=20).mean()  # Last 20 days volume
    adjusted_hl_range_momentum = hl_range_momentum / avg_volume_hl
    
    # Final Alpha Factor Combination
    w1 = 0.4  # Weight for Intraday and Short-Term Price Momentum
    w2 = 0.3  # Weight for Volume-Adjusted Factors
    w3 = 0.3  # Weight for High-Low Range Momentum
    alpha_factor = (short_term_momentum * w1) + \
                   (volume_adjusted_return * w2) + \
                   (vwp_movement * w2) + \
                   (adjusted_hl_range_momentum * w3) + \
                   (ema_atr_reversal * w1)
    
    return alpha_factor
```
