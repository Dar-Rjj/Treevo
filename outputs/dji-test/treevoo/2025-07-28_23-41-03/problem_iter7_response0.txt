```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    weighted_returns = (df['daily_return'].shift(1) * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    df['wma_20d'] = weighted_returns / total_volume
    
    # Adjust for Price Volatility
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_wma_20d'] = df['wma_20d'] - avg_daily_range
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_oc_return'] = (df['close'] - df['open']) * df['volume']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['alpha_factor'] = df['adjusted_wma_20d'] + df['intraday_return_ratio'] + df['weighted_oc_return']
    
    return df['alpha_factor']
```
