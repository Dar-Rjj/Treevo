```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Average True Range (ATR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Exponential Moving Average (EMA) of ATR * Reversal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=5, adjust=False).mean()
    
    # Calculate Simple Moving Averages
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff'] = df['sma_50'] - df['sma_200']
    
    # Calculate Price Momentum Over 5 Days
    df['price_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Surge Over 5 Days
    df['volume_surge'] = df['volume'] - df['volume'].shift(5)
    
    # Combine Momentum and Volume Indicators
    df['combined_mom_vol'] = 0.5 * df['price_momentum'] + 0.5 * df['volume_surge']
    
    # Final Combined Alpha Factor
    df['alpha_factor_part1'] = df['sma_diff'] + df['ema_atr_reversal'] + df['combined_mom_vol']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=5).mean()
    df['weighted_open_to_close'] = df['open_to_close_return'] * avg_volume
    
    # Calculate Volume Flow
    df['volume_flow'] = df['volume'] * (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Combine All Factors
    df['combined_factors'] = (df['intraday_return'] + df['weighted_open_to_close'] + df['volume_flow'] + 
                              df['high_low_range_momentum']) / average_volume
    
    return df['alpha_factor_part1'] + df['combined_factors']
```
