```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between the close and open prices
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted by the close price
    vol_weighted_close = df['volume'] * df['close']
    
    # Combine the above calculations into a novel factor
    factor = close_open_diff / avg_high_low + vol_weighted_close
    
    return factor
```
