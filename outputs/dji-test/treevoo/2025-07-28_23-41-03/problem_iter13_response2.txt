```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Volume-Weighted Average Return
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_volume_weighted_return'] = df['volume_weighted_return'] - df['avg_daily_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['open']) / (df['close'] - df['low'])
    
    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factors'] = df['adjusted_volume_weighted_return'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return']
    
    # Calculate Simple Moving Averages (SMA)
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_50_100_diff'] = df['sma_50'] - df['sma_100']
    df['sma_100_200_diff'] = df['sma_100'] - df['sma_200']
    
    # Combine SMAs
    df['sma_combined'] = df['sma_50_100_diff'] + df['sma_100_200_diff']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate ATR
    df['atr'] = (df['high'] - df['low']).abs() + (df['high'] - df['close'].shift(1)).abs() + (df['low'] - df['close'].shift(1)).abs()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Calculate Exponential Moving Average (EMA) of ATR * Reversal Signal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=10, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] + df['sma_combined'] + df['ema_atr_reversal']
    
    return df['alpha_factor']
```
