```python
def heuristics_v2(df):
    # Calculate the difference between high and low price for the day
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the difference between close and open price for the day
    df['daily_oc_diff'] = df['close'] - df['open']
    
    # Calculate the ratio of daily range to average volume
    df['range_volume_ratio'] = df['daily_range'] / df['volume']
    
    # Calculate the percentage change in close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Heuristic factor: weighted sum of the daily range, daily open-close difference, and the percentage change in close price
    df['heuristic_factor'] = 0.5 * df['daily_range'] + 0.3 * df['daily_oc_diff'] + 0.2 * df['close_pct_change']
    
    return df['heuristic_factor']
```
