```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, w1=0.5, w2=0.3, w3=0.2):
    # Calculate Simple Moving Averages
    df['50_SMA'] = df['close'].rolling(window=50).mean()
    df['100_SMA'] = df['close'].rolling(window=100).mean()
    df['200_SMA'] = df['close'].rolling(window=200).mean()

    # Price Momentum
    df['Intraday_Momentum'] = df['close'] - df['open']
    df['Combined_Momentum'] = (df['50_SMA'] - df['100_SMA']) + (df['100_SMA'] - df['200_SMA']) + df['Intraday_Momentum']

    # Volume-Adjusted Factors
    df['avg_volume'] = df['volume'].rolling(window=N).mean()
    df['Volume_Adjusted_Return'] = (df['close'] - df['open']) * (df['volume'] / df['avg_volume'])
    df['Volume_Weighted_Price_Movement'] = (df['high'] - df['low']) * df['volume']
    df['Volume_Adjusted_Intraday_Range'] = (df['high'] - df['low']) * (df['volume'] / df['avg_volume'])

    # High-Low Range Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Adjusted_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    df['High_Low_Range_Momentum'] = df['Adjusted_High_Low_Range'] / df['avg_volume']

    # Volume-Adjusted High-Low Range Momentum
    df['Volume_Adjusted_High_Low_Range_Momentum'] = (df['High_Low_Range_Momentum'] * df['volume']) / df['avg_volume']

    # Final Alpha Factor Combination
    alpha_factor = (
        w1 * df['Combined_Momentum'] + 
        w2 * df['High_Low_Range_Momentum'] / df['avg_volume'] +
        w3 * df['Volume_Adjusted_High_Low_Range_Momentum'] / df['avg_volume']
    )

    return alpha_factor
```
