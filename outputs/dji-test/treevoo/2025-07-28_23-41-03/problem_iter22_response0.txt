```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the C/H ratio
    df['C_H_ratio'] = df['close'] / df['high']
    
    # Compute a VWAP-like metric: (high - low) * volume
    df['VWAP_like'] = (df['high'] - df['low']) * df['volume']
    
    # Measure the gap between opening and closing prices (O-C)
    df['O_C_gap'] = df['open'] - df['close']
    
    # Simple Moving Average (SMA) of close prices over 5 days
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Volume-Weighted Moving Average (VWMA) of close prices over 5 days
    df['VWMA_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Composite alpha factor
    df['composite_factor'] = (
        0.4 * df['C_H_ratio'] + 
        0.3 * df['VWAP_like'] + 
        0.2 * df['O_C_gap'] + 
        0.1 * (df['close'] - df['SMA_5']) + 
        0.1 * (df['close'] - df['VWMA_5'])
    )
    
    return df['composite_factor']
```
