```python
def heuristics_v2(data, n=10):
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range - high_low_range.shift(1)
    
    # Adjust by Volume
    average_volume = data['volume'].rolling(window=n).sum() / n
    
    # Combine High-Low Range Momentum and Volume Adjustment
    factor = high_low_momentum / average_volume
    
    return factor
```
