```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['Daily_Price_Return'] = df['close'].pct_change()
    
    # Calculate 20-Day Volume-Weighted Average Return
    df['Volume_Weighted_Avg_Return'] = (df['Daily_Price_Return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility: Calculate 20-Day Average Daily Price Range
    df['Avg_Daily_Price_Range'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Subtract 20-Day Average Daily Price Range from 20-Day Volume-Weighted Average Return
    df['Adjusted_Volume_Weighted_Avg_Return'] = df['Volume_Weighted_Avg_Return'] - df['Avg_Daily_Price_Range']
    
    # Calculate Intraday Return Ratio
    df['Intraday_Return_Ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['Open_to_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Weighted_Open_to_Close_Return'] = df['Open_to_Close_Return'] * df['volume']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['Alpha_Factor'] = df['Adjusted_Volume_Weighted_Avg_Return'] + df['Intraday_Return_Ratio'] + df['Weighted_Open_to_Close_Return']
    
    return df['Alpha_Factor']
```
