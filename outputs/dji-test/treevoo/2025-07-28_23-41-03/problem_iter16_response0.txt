```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to high (C/H) as a measure of day's performance
    df['C/H'] = df['close'] / df['high']
    
    # Compute the difference between high and low prices adjusted by volume (VWAP-like metric)
    df['HL_Vol_Adjusted'] = (df['high'] - df['low']) * df['volume']
    
    # Measure the gap between opening and closing prices (O-C) to gauge market momentum
    df['O-C'] = df['open'] - df['close']
    
    # Develop a simple moving average (SMA) of close prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_Deviation'] = df['close'] - df['SMA_5']
    
    # Construct a volume-weighted moving average (VWMA) of close prices
    df['VWMA_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWMA_Deviation'] = df['close'] - df['VWMA_5']
    
    # Identify significant price levels (support, resistance) using previous high and low points
    df['Support'] = df['low'].rolling(window=10).min()
    df['Resistance'] = df['high'].rolling(window=10).max()
    
    # Monitor the volume at these levels to detect potential breakouts or bounces
    df['Volume_Support'] = df['volume'] * (df['low'] == df['Support'])
    df['Volume_Resistance'] = df['volume'] * (df['high'] == df['Resistance'])
    
    # Alpha factor: Combine multiple indicators
    df['Alpha_Factor'] = (
        0.3 * (df['C/H'] - df['C/H'].shift(1)) +  # Trend in C/H
        0.2 * df['HL_Vol_Adjusted'] +            # VWAP-like metric
        0.2 * df['O-C'] +                         # Open-Close gap
        0.15 * df['SMA_Deviation'] +             # Deviation from SMA
        0.15 * df['VWMA_Deviation']               # Deviation from VWMA
    )
    
    return df['Alpha_Factor'].dropna()
```
