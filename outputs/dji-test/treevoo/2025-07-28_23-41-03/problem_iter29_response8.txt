```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['50SMA'] = df['close'].rolling(window=50).mean()
    df['100SMA'] = df['close'].rolling(window=100).mean()
    df['200SMA'] = df['close'].rolling(window=200).mean()
    
    # Combined Momentum
    df['CombinedMomentum'] = (df['50SMA'] - df['100SMA']) + (df['100SMA'] - df['200SMA'])
    df['IntradayMomentum'] = df['close'] - df['open']
    df['CombinedMomentum'] += df['IntradayMomentum']

    # Volume-Adjusted Factors
    n_days = 20
    df['AvgVolume'] = df['volume'].rolling(window=n_days).mean()
    df['VolAdjReturn'] = (df['close'] - df['open']) * (df['volume'] / df['AvgVolume'])
    df['VolWeightedPriceMove'] = (df['high'] - df['low']) * df['volume']
    df['VolAdjIntradayRange'] = (df['high'] - df['low']) * (df['volume'] / df['AvgVolume'])

    # High-Low Range Momentum
    df['HighLowRange'] = df['high'] - df['low']
    df['AdjHighLowRange'] = df['HighLowRange'] - df['HighLowRange'].shift(1)
    df['HighLowRangeMom'] = df['AdjHighLowRange'] / df['volume']

    # Volume-Adjusted High-Low Range Momentum
    df['VolAdjHighLowRangeMom'] = (df['HighLowRangeMom'] * df['volume']) / df['AvgVolume']

    # Calculate 20-Day Weighted Moving Average of Returns
    df['DailyRet'] = df['close'].pct_change()
    df['WMA'] = (df['DailyRet'] * df['volume']).rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()

    # Adjust for Price Volatility
    df['20DayAvgPriceRange'] = (df['high'] - df['low']).rolling(window=n_days).mean()
    df['AdjWMA'] = df['WMA'] - df['20DayAvgPriceRange']

    # Intraday Return Ratio and Weighted Open-to-Close Return
    df['IntradayRetRatio'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['WeightedOpenCloseRet'] = (df['close'] - df['open']) * df['volume']

    # Final Alpha Factor Combination
    w1, w2, w3, w4, w5, w6 = 0.25, 0.2, 0.15, 0.2, 0.1, 0.1
    df['AlphaFactor'] = (w1 * df['CombinedMomentum'] +
                         w2 * (df['HighLowRangeMom'] / df['volume']) +
                         w3 * (df['VolAdjHighLowRangeMom'] / df['AvgVolume']) +
                         w4 * df['AdjWMA'] +
                         w5 * df['IntradayRetRatio'] +
                         w6 * df['WeightedOpenCloseRet'])

    return df['AlphaFactor']
```
