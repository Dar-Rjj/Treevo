```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Volume-Weighted Average Return
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_momentum'] = df['volume_weighted_return'] - df['avg_daily_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factor'] = df['adjusted_momentum'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return']
    
    # Price Momentum Component
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    df['price_momentum'] = (df['sma_50'] - df['sma_100']) + (df['sma_100'] - df['sma_200'])
    
    # Intraday Reversal Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    df['reversal_momentum'] = df['atr_14'] * df['reversal_signal']
    
    # Calculate Exponential Moving Average (EMA) of the reversal momentum
    df['reversal_ema'] = df['reversal_momentum'].ewm(span=14, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['price_momentum'] + df['reversal_ema']
    
    return df['alpha_factor']
```
