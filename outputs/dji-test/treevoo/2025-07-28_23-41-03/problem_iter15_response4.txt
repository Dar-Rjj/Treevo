```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_return'] = df['daily_return'].rolling(window=20).apply(
        lambda x: (x * df.loc[x.index, 'volume']).sum() / df.loc[x.index, 'volume'].sum(), raw=False)

    # Adjust for Price Volatility
    df['avg_price_range'] = (df['high'] - df['low']).rolling(window=22).mean()
    df['adjusted_weighted_return'] = df['weighted_return'] - df['avg_price_range']

    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['open']

    # Calculate Weighted Open-to-Close Return
    df['weighted_open_close_return'] = (df['close'] - df['open']) * df['volume']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate High-Low Range Momentum
    df['high_low_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=30).mean()

    # Calculate Price Trend
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['price_trend'] = df['sma_50'] - df['close']

    # Combine Adjusted Momentum, Intraday, and High-Low Range Factors
    df['combined_factors'] = df['adjusted_weighted_return'] + df['intraday_return_ratio'] + df['weighted_open_close_return'] + df['high_low_momentum']

    # Price Momentum Component
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['price_momentum'] = (df['sma_50'] - df['sma_100']) + (df['sma_100'] - df['sma_200'])

    # Intraday Reversal Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    df['atr'] = (df['high'] - df['low']).abs().rolling(window=14).mean()
    df['intraday_reversal_momentum'] = df['atr'] * df['reversal_signal']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_factors'] + df['price_momentum'] + df['intraday_reversal_momentum']) / df['avg_volume'] * df['price_trend']

    return df['final_alpha_factor']
```
