```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    df['SMA_100_200'] = df['SMA_100'] - df['SMA_200']
    df['SMA_50_100'] = df['SMA_50'] - df['SMA_100']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1

    # Calculate Reversal Signal
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)

    # Combine Momentum and Reversal Signals
    df['combined_signal_100_200'] = df['SMA_100_200'] * df['reversal_signal']
    df['combined_signal_50_100'] = df['SMA_50_100'] * df['reversal_signal']

    # Compute Average True Range (ATR)
    df['TR'] = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Integrate ATR with Combined Signal
    df['ATR_combined_100_200'] = df['ATR'] * df['combined_signal_100_200']
    df['ATR_combined_50_100'] = df['ATR'] * df['combined_signal_50_100']

    # Smooth Result with Exponential Moving Average (EMA)
    df['smoothed_ATR_100_200'] = df['ATR_combined_100_200'].ewm(span=7).mean()
    df['smoothed_ATR_50_100'] = df['ATR_combined_50_100'].ewm(span=7).mean()

    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_returns'] = (df['daily_return'] * df['volume']).rolling(window=22).sum() / df['volume'].rolling(window=22).sum()

    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    df['avg_price_range'] = df['price_range'].rolling(window=20).mean()
    df['adjusted_weighted_returns'] = df['weighted_returns'] - df['avg_price_range']

    # Combine Factors
    df['final_factor'] = (df['smoothed_ATR_100_200'] + df['smoothed_ATR_50_100'] + 
                          df['weighted_returns'] + df['adjusted_weighted_returns']).ewm(span=7).mean()

    return df['final_factor']
```
