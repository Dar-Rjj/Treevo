import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to high (C/H)
    df['C_H_ratio'] = df['close'] / df['high']
    
    # Compute the difference between high and low prices adjusted by volume (VWAP-like metric)
    df['HL_V_adj'] = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Measure the gap between opening and closing prices (O-C)
    df['O_C_gap'] = df['open'] - df['close']
    
    # Calculate a simple moving average (SMA) of close prices
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_deviation'] = df['close'] - df['SMA_20']
    
    # Construct a volume-weighted moving average (VWMA) of close prices
    df['VWMA_20'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWMA_deviation'] = df['close'] - df['VWMA_20']
    
    # Identify significant price levels (support, resistance) using previous high and low points
    df['support'] = df['low'].rolling(window=20).min()
    df['resistance'] = df['high'].rolling(window=20).max()
    df['support_resistance_vol'] = (df['close'] - df['support']) * (df['resistance'] - df['close']) * df['volume']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['C_H_ratio'] + 
        df['HL_V_adj'] + 
        df['O_C_gap'] + 
        df['SMA_deviation'] + 
        df['VWMA_deviation'] + 
        df['support_resistance_vol']
    )
    
    return df['alpha_factor'].dropna()
