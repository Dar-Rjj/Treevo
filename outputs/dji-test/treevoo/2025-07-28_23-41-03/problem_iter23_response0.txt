```python
def heuristics_v2(data):
    # Calculate the average of the open and close prices
    avg_price = (data['open'] + data['close']) / 2
    
    # Calculate the high-low range
    high_low_range = data['high'] - data['low']
    
    # Calculate the ratio of the volume to the high-low range
    volume_to_range_ratio = data['volume'] / high_low_range
    
    # The factor is defined as the product of the average price and the volume-to-range ratio
    alpha_factor = avg_price * volume_to_range_ratio
    
    return alpha_factor
```
