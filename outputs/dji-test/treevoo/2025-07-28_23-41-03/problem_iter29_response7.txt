```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']

    # Calculate Combined EMA Momentum
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    combined_ema_momentum = (ema_50 - ema_100) + (ema_100 - ema_200)

    # Add Intraday Momentum to Combined EMA Momentum
    total_momentum = combined_ema_momentum + intraday_momentum

    # Calculate Volume-Adjusted Return
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)

    # Calculate Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']

    # Calculate Adjusted High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    adjusted_high_low_range_momentum = high_low_range - high_low_range.shift(1)
    adjusted_high_low_range_momentum /= df['volume'].shift(1)

    # Final Alpha Factor Combination
    w1 = 0.7  # Weight for Momentum and Volume-Adjusted Factors
    w2 = 0.3  # Weight for Adjusted High-Low Range Momentum
    final_alpha_factor = (total_momentum + volume_adjusted_return + volume_weighted_price_movement) * w1
    final_alpha_factor += adjusted_high_low_range_momentum * w2

    return final_alpha_factor
```
