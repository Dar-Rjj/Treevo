```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to high (C/H)
    df['C/H'] = df['close'] / df['high']
    
    # Compute the difference between high and low prices adjusted by volume (VWAP-like metric)
    df['HL_Volume_Adjusted'] = (df['high'] - df['low']) * df['volume']
    
    # Measure the gap between opening and closing prices (O-C)
    df['O-C'] = df['open'] - df['close']
    
    # Develop a simple moving average (SMA) of close prices
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Construct a volume-weighted moving average (VWMA) of close prices
    df['VWMA_50'] = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    
    # Identify significant price levels (support, resistance) using previous high and low points
    df['Support'] = df['low'].rolling(window=50).min()
    df['Resistance'] = df['high'].rolling(window=50).max()
    
    # Combine C/H ratio, VWAP-like metric, and O-C gap into a single factor
    df['Composite_Factor'] = df['C/H'] + df['HL_Volume_Adjusted'] + df['O-C']
    
    # Incorporate SMA and VWMA crossovers with support and resistance levels
    df['SMA_Crossover'] = (df['close'] > df['SMA_50']).astype(int)
    df['VWMA_Crossover'] = (df['close'] > df['VWMA_50']).astype(int)
    df['Support_Bounce'] = (df['close'] > df['Support']).astype(int)
    df['Resistance_Breakout'] = (df['close'] < df['Resistance']).astype(int)
    
    # Create a composite signal that considers both price and volume dynamics
    df['Alpha_Factor'] = df['Composite_Factor'] + df['SMA_Crossover'] + df['VWMA_Crossover'] + df['Support_Bounce'] + df['Resistance_Breakout']
    
    return df['Alpha_Factor']
```
