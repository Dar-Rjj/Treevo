```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff()
    
    # Calculate the money flow, which is the product of typical price, volume, and price change direction
    money_flow = avg_price * df['volume'] * price_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate the cumulative money flow over a rolling window of 14 days
    cmf = money_flow.rolling(window=14).sum()
    
    # Normalize the cumulative money flow by the total volume over the same period
    total_volume = df['volume'].rolling(window=14).sum()
    normalized_cmf = cmf / total_volume
    
    # The final alpha factor is the normalized cumulative money flow
    return normalized_cmf
```
