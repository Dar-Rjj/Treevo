```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Volume-Weighted Average Return
    df['weighted_return'] = df['daily_return'] * df['volume']
    df['sum_weighted_return_20d'] = df['weighted_return'].rolling(window=20).sum()
    df['sum_volume_20d'] = df['volume'].rolling(window=20).sum()
    df['vol_weighted_avg_return_20d'] = df['sum_weighted_return_20d'] / df['sum_volume_20d']

    # Adjust for Price Volatility
    df['price_range_20d'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_vol_weighted_return_20d'] = df['vol_weighted_avg_return_20d'] - df['price_range_20d']

    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']

    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']

    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factors'] = df['adjusted_vol_weighted_return_20d'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return']

    # Price Momentum Component
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['sma_diff_100_200'] = df['sma_100'] - df['sma_200']
    df['sma_diff_50_100'] = df['sma_50'] - df['sma_100']
    df['sma_combined'] = df['sma_diff_50_100'] + df['sma_diff_100_200']

    # Intraday Reversal Momentum Component
    df['intraday_return'] = (df['open'] - df['close']) / df['open']
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1).shift(1)
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()
    df['atr_reversal'] = df['atr_14d'] * df['reversal_signal']
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=9, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] + df['sma_combined'] + df['ema_atr_reversal']
    
    return df['alpha_factor']
```
