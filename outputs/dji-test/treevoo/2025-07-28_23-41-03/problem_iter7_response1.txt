```python
def heuristics_v2(df, n=5):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Average Volume over past n days
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Combine Adjustments
    df['factor'] = df['range_momentum'] / df['avg_volume']
    
    return df['factor']
```
