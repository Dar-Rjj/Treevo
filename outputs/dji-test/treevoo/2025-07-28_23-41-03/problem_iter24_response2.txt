import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Momentum Signals
    momentum_100_200 = sma_100 - sma_200
    momentum_50_100 = sma_50 - sma_100
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Reversal Signal
    reversal_signal = -1 * (intraday_return > 0) + 1 * (intraday_return <= 0)
    
    # Combine Momentum and Reversal
    combined_momentum_reversal = (momentum_100_200 * reversal_signal) + (momentum_50_100 * reversal_signal)
    
    # High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    high_low_momentum = high_low_range - high_low_range.shift(1)
    
    # Volume Adjustment
    n_days = 20
    volume_adjustment = df['volume'].rolling(window=n_days).sum() / n_days
    
    # Adjusted High-Low Range Momentum
    adjusted_high_low_momentum = high_low_momentum / volume_adjustment
    
    # Price Change Momentum
    price_change_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combined Price and Volume Momentum
    combined_price_volume_momentum = price_change_momentum * volume_adjustment
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_reversal + adjusted_high_low_momentum + combined_price_volume_momentum
    
    return final_alpha_factor
