```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    def weighted_moving_average_returns(returns, volumes, window=20):
        return (returns * volumes).rolling(window).sum() / volumes.rolling(window).sum()
    
    df['wma_returns'] = weighted_moving_average_returns(df['daily_return'], df['volume'])
    
    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    df['avg_price_range'] = df['price_range'].rolling(window=20).mean()
    df['adjusted_wma_returns'] = df['wma_returns'] - df['avg_price_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['open']) / (df['close'] - df['low'])
    
    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']
    
    # Calculate Volume Flow
    df['volume_flow'] = df['volume'] * (df['high'] - df['low']) / df['close']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['final_factor'] = df['adjusted_wma_returns'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return'] + df['volume_flow']
    
    return df['final_factor']
```
