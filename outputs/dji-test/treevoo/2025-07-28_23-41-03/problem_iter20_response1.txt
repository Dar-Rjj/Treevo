```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    delta = df['close'].diff()
    positive_money_flow = money_flow.where(delta > 0, 0)
    negative_money_flow = money_flow.where(delta < 0, 0)
    
    # Calculate the money ratio
    positive_sum = positive_money_flow.rolling(window=14).sum()
    negative_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_sum / negative_sum
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the Heuristics Alpha Factor
    factor = mfi - 50
    
    return factor
```
