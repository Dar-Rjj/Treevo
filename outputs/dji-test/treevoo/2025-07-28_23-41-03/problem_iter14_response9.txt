import pandas as pd

def heuristics_v2(df):
    # Intraday and Short-Term Price Momentum
    intraday_momentum = df['close'] - df['open']
    
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    sma_diff_100_200 = sma_100 - sma_200
    sma_diff_50_100 = sma_50 - sma_100
    
    combined_smas = (sma_diff_50_100 + sma_diff_100_200) + intraday_momentum
    
    # Volume-Adjusted Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_return = open_to_close_return * avg_volume_10
    
    # Volume-Weighted Price Movement
    vwap = (df['high'] + df['low'] + df['close']) / 3
    volume_weighted_movement = vwap * df['volume']
    
    # Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    high_low_range_shift = high_low_range.shift(1)
    adjusted_high_low_range = high_low_range - high_low_range_shift
    
    # Intraday Reversal Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    reversal_signal = intraday_return.apply(lambda x: -1 if x > 0 else 1)
    
    atr = (df['high'] - df['low']).abs() + (df['high'] - df['close'].shift(1)).abs() + (df['low'] - df['close'].shift(1)).abs()
    atr = atr.rolling(window=14).mean()
    
    reversal_momentum = atr * reversal_signal
    ema_reversal = reversal_momentum.ewm(span=7, adjust=False).mean()
    
    # High-Low Range Momentum with Volume Adjustment
    high_low_momentum = high_low_range - high_low_range.shift(1)
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    high_low_momentum_vol_adj = high_low_momentum / avg_volume_5
    
    # Final Alpha Factor Combination
    w1 = 0.4
    w2 = 0.3
    w3 = 0.3
    final_alpha = (combined_smas * w1) + (volume_adjusted_return * w2) + (high_low_momentum_vol_adj * w3)
    
    return final_alpha
