```python
import pandas as pd

def heuristics_v2(df, N=10, n=5, m=5):
    # Calculate Intraday Return Ratio
    intraday_return_ratio = df['high'] / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Weight by Average Volume of Last N Days
    avg_volume_N = df['volume'].rolling(window=N).mean()
    weighted_open_to_close_return = open_to_close_return * avg_volume_N
    
    # Calculate Volume Flow
    volume_flow = df['volume'] * (df['high'] - df['low'])
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Calculate Average Volume and Amount
    avg_volume_n = df['volume'].rolling(window=n).mean()
    avg_amount_m = df['amount'].rolling(window=m).mean()
    
    # Adjust by Volume and Amount
    combined_factors = (intraday_return_ratio + weighted_open_to_close_return + volume_flow) * high_low_range_momentum
    adjusted_factor = combined_factors / (avg_volume_n + avg_amount_m)
    
    return adjusted_factor
```
