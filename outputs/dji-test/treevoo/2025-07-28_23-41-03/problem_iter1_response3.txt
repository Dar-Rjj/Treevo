```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    def calculate_reversal_signal(intraday_return):
        if intraday_return > 0:
            return -1
        else:
            return 1
    df['reversal_signal'] = df['intraday_return'].apply(calculate_reversal_signal)
    
    # Calculate Average True Range (ATR)
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_signal'] = df['atr'] * df['reversal_signal']
    
    # Smooth with Exponential Moving Average (EMA)
    df['ema_atr_signal'] = df['atr_signal'].ewm(span=5, adjust=False).mean()
    
    return df['ema_atr_signal']
```
