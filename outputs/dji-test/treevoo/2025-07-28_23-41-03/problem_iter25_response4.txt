```python
import pandas as pd

def heuristics_v2(df, w1=0.7, w2=0.3, n=20):
    # Price Momentum
    intraday_momentum = df['close'] - df['open']
    
    df['50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    combined_ema_momentum = (df['50_ema'] - df['100_ema']) + (df['100_ema'] - df['200_ema'])
    
    momentum_factor = combined_ema_momentum + intraday_momentum
    
    # Volume-Adjusted Factors
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_adjusted_return = intraday_momentum * (df['volume'] / avg_volume)
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Adjusted High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    prev_high_low_range = high_low_range.shift(1)
    adjusted_high_low_range_momentum = high_low_range - prev_high_low_range
    adjusted_high_low_range_vol = adjusted_high_low_range_momentum / df['volume'].shift(1)
    
    # Final Alpha Factor Combination
    alpha_factor = (momentum_factor + volume_adjusted_return + volume_weighted_price_movement) * w1 + adjusted_high_low_range_vol * w2
    
    return alpha_factor
```
