import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']

    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Combine Momentum
    combined_momentum = (sma_50 - sma_100) + (sma_100 - sma_200) + intraday_momentum

    # Calculate Volume-Adjusted Factors
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    volume_adjusted_intraday_range = (df['high'] - df['low']) * (df['volume'] / avg_volume)

    # High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    adjusted_high_low_range = high_low_range - high_low_range.shift(1)
    high_low_range_momentum = adjusted_high_low_range / avg_volume

    # Volume-Adjusted High-Low Range Momentum
    volume_adjusted_high_low_range_momentum = (high_low_range_momentum * df['volume']) / avg_volume

    # 20-Day Weighted Moving Average of Returns
    daily_returns = df['close'].pct_change()
    weighted_returns = (daily_returns * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Adjust for Price Volatility
    avg_daily_price_range = (df['high'] - df['low']).rolling(window=20).mean()
    adjusted_weighted_returns = weighted_returns - avg_daily_price_range

    # Intraday Return Ratio
    intraday_return_ratio = df['high'] / df['low']

    # Weighted Open-to-Close Return
    weighted_open_to_close_return = (df['close'] - df['open']) * df['volume']

    # Final Alpha Factor Combination
    w1, w2, w3, w4, w5, w6 = 0.2, 0.2, 0.2, 0.2, 0.1, 0.1  # Example weights
    final_factor = (
        combined_momentum * w1 +
        (high_low_range_momentum / avg_volume) * w2 +
        (volume_adjusted_high_low_range_momentum / avg_volume) * w3 +
        (adjusted_weighted_returns) * w4 +
        (intraday_return_ratio) * w5 +
        (weighted_open_to_close_return) * w6
    )

    return final_factor
