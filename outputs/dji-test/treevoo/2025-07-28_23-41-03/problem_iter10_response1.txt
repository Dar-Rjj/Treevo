```python
import pandas as pd

def heuristics_v2(df):
    # Price Momentum Component
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    price_momentum = sma_50 - sma_200
    
    # Intraday Reversal Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    reversal_signal = intraday_return.apply(lambda x: -1 if x > 0 else 1)
    
    true_range = pd.concat([
        (df['high'] - df['low']),
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    
    atr = true_range.rolling(window=14).mean()
    reversal_momentum = atr * reversal_signal
    
    ema_reversal_momentum = reversal_momentum.ewm(span=7, adjust=False).mean()
    
    # Combine the two components
    alpha_factor = price_momentum + ema_reversal_momentum
    
    return alpha_factor
```
