import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    diff_sma_100_200 = sma_100 - sma_200
    diff_sma_50_100 = sma_50 - sma_100

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Reversal Signal
    reversal_signal = np.where(intraday_return > 0, -1, 1)

    # Combine Momentum and Reversal Signals
    combined_signal_100_200 = diff_sma_100_200 * reversal_signal
    combined_signal_50_100 = diff_sma_50_100 * reversal_signal

    # Compute Average True Range (ATR)
    tr = pd.DataFrame(index=df.index)
    tr['H-L'] = df['high'] - df['low']
    tr['H-PC'] = abs(df['high'] - df['close'].shift(1))
    tr['L-PC'] = abs(df['low'] - df['close'].shift(1))
    tr['TR'] = tr.max(axis=1)
    atr = tr['TR'].rolling(window=14).mean()

    # Integrate ATR with Combined Signal
    integrated_signal_100_200 = atr * combined_signal_100_200
    integrated_signal_50_100 = atr * combined_signal_50_100

    ema_integrated_signal_100_200 = integrated_signal_100_200.ewm(span=10, adjust=False).mean()
    ema_integrated_signal_50_100 = integrated_signal_50_100.ewm(span=10, adjust=False).mean()

    # Calculate 20-Day Weighted Moving Average of Returns
    daily_price_return = df['close'].pct_change()
    weighted_average_return = (daily_price_return * df['volume']).rolling(window=22).sum() / df['volume'].rolling(window=22).sum()

    # Adjust for Price Volatility
    daily_price_range = df['high'] - df['low']
    average_daily_price_range = daily_price_range.rolling(window=22).mean()
    adjusted_factor_value = weighted_average_return - average_daily_price_range

    # Final Alpha Factor
    final_alpha_factor = ema_integrated_signal_100_200 + ema_integrated_signal_50_100 + adjusted_factor_value

    return final_alpha_factor
