```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Volume of Last N Days
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume-Adjusted Open-to-Close Return
    volume_adjusted_return = open_to_close_return * avg_volume_n_days
    
    # Calculate Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Calculate Adjusted High-Low Range
    high_low_range_today = df['high'] - df['low']
    high_low_range_yesterday = high_low_range_today.shift(1)
    adjusted_high_low_range = high_low_range_today - high_low_range_yesterday
    
    # Calculate Volume-Weighted Average
    avg_volume = df['volume'].rolling(window=n).sum() / n
    avg_amount = df['amount'].rolling(window=m).sum() / m
    total_avg = avg_volume + avg_amount
    
    # Combine All Factors
    factor = (intraday_momentum + volume_adjusted_return + volume_weighted_price_movement) * adjusted_high_low_range / total_avg
    
    return factor
```
