```python
import pandas as pd

def heuristics_v2(df):
    # 20-Day Weighted Moving Average of Returns
    df['Daily_Price_Return'] = df['close'].pct_change()
    df['Weighted_Return'] = df['Daily_Price_Return'] * df['volume']
    df['Sum_Weighted_Return'] = df['Weighted_Return'].rolling(window=20).sum()
    df['Sum_Volume'] = df['volume'].rolling(window=20).sum()
    df['WMA_Return'] = df['Sum_Weighted_Return'] / df['Sum_Volume']

    # Adjust for Price Volatility
    df['Price_Range'] = df['high'] - df['low']
    df['Avg_Price_Range'] = df['Price_Range'].rolling(window=20).mean()
    df['Adjusted_WMA_Return'] = df['WMA_Return'] - df['Avg_Price_Range']

    # Intraday Return Ratio
    df['Intraday_Return_Ratio'] = df['high'] / df['low']

    # Weighted Open-to-Close Return
    df['Open_Close_Return'] = (df['close'] - df['open']) * df['volume']

    # High-Low Range Momentum
    df['HL_Range_Today'] = df['high'] - df['low']
    df['HL_Range_Yesterday'] = df['high'].shift(1) - df['low'].shift(1)
    df['HL_Range_Momentum'] = df['HL_Range_Today'] / df['HL_Range_Yesterday']

    # Price Trend
    df['50_Day_SMA'] = df['close'].rolling(window=50).mean()
    df['Price_Trend'] = df['50_Day_SMA'] - df['close']

    # Final Alpha Factor
    df['Sum_Volume_30'] = df['volume'].rolling(window=30).sum()
    df['Alpha_Factor'] = (df['Adjusted_WMA_Return'] + df['Intraday_Return_Ratio'] + 
                          df['Open_Close_Return'] + df['HL_Range_Momentum'] + 
                          df['Price_Trend']) / df['Sum_Volume_30']

    return df['Alpha_Factor']
```
