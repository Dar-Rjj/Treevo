```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Averages (EMA)
    df['50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Subtract Longer EMA from Shorter EMA
    df['price_momentum_1'] = df['100_ema'] - df['200_ema']
    df['price_momentum_2'] = df['50_ema'] - df['100_ema']
    
    # Combine the two price momentum components
    df['price_momentum'] = df['price_momentum_1'] + df['price_momentum_2']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df['true_range'].combine_first((df['high'] - df.shift(1)['close']).abs())
    df['true_range'] = df['true_range'].combine_first((df['low'] - df.shift(1)['close']).abs())
    
    # Multiply True Range by Reversal Signal
    df['intraday_reversal'] = df['true_range'] * df['reversal_signal']
    
    # Apply Simple Moving Average (SMA) to the intraday reversal
    df['intraday_reversal_sma'] = df['intraday_reversal'].rolling(window=10).mean()
    
    # Combine the price momentum and intraday reversal components
    df['alpha_factor'] = df['price_momentum'] + df['intraday_reversal_sma']
    
    return df['alpha_factor']
```
