```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return using close prices
    df['daily_return'] = df['close'].diff()
    
    # Analyze intra-day price movements
    df['range'] = df['high'] - df['low']
    df['gap_opening'] = df['open'] - df['close'].shift(1)
    df['closing_direction'] = df['close'] - df['open']
    
    # Explore volume-based features
    df['volume_change'] = df['volume'].diff()
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Combine price and volume features
    df['price_volume_momentum'] = df['daily_return'] * df['volume_change']
    df['volatility_adjusted_volume'] = (df['daily_return'] / df['range']) * df['volume']
    
    # Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate multi-day patterns
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['trend_reversal'] = df['5_day_ma'].diff().apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    df['cumulative_volume_price_change'] = (df['daily_return'] * df['volume']).rolling(window=10).sum()
    
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['trend_following_factor'] = df['close'] - df['20_day_ma']
    
    # Generate the final alpha factor
    df['alpha_factor'] = (
        df['price_volume_momentum'] + 
        df['volatility_adjusted_volume'] + 
        df['cumulative_volume_price_change'] + 
        df['trend_following_factor']
    )
    
    return df['alpha_factor']
```
