```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Return Ratio
    intraday_return_ratio = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Weighted Open-to-Close Return
    avg_volume = df['volume'].rolling(window=N).mean()
    weighted_open_to_close_return = open_to_close_return * avg_volume
    
    # Calculate Volume Flow
    volume_flow = (df['high'] - df['low']) * df['volume']
    
    # Combine Factors
    combined_factor = (intraday_return_ratio + weighted_open_to_close_return + volume_flow)
    
    return combined_factor
```
