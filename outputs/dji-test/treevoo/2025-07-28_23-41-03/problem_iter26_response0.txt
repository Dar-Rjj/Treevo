import pandas as pd

def heuristics_v2(df, n=10):
    # Momentum: Difference between today's close and the close n days ago
    df['Momentum'] = df['close'] - df['close'].shift(n)
    
    # Trend Reversal: Low of today is higher than the high n days ago
    df['Trend_Reversal'] = (df['low'] > df['high'].shift(n)).astype(int)
    
    # Volume Change: Change in volume from one day to the next
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Strength of Moves: Percentage change in closing prices multiplied by the daily volume
    df['Strength_of_Moves'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Financial Flows: Today's transaction amount compared to the average over the past n days
    df['Average_Transaction_Amount'] = df['amount'].rolling(window=n).mean()
    df['Financial_Flows'] = df['amount'] - df['Average_Transaction_Amount']
    
    # Composite Alpha: Blend price and volume signals
    df['Composite_Alpha'] = df['Momentum'] + df['Strength_of_Moves']
    
    # Comprehensive Alpha: Integrate financial flows into the combined price and volume signal
    df['Comprehensive_Alpha'] = df['Composite_Alpha'] + df['Financial_Flows']
    
    return df['Comprehensive_Alpha']
