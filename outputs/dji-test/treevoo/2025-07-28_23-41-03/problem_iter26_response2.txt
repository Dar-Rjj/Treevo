```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Calculate EMAs
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Combined EMA Momentum
    combined_ema_momentum = (ema_50 - ema_100) + (ema_100 - ema_200)
    
    # Add Intraday Momentum
    total_momentum = combined_ema_momentum + intraday_momentum
    
    # Volume-Adjusted Return
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    # Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Adjusted High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    adjusted_high_low_range = high_low_range - high_low_range.shift(1)
    adjusted_high_low_momentum = adjusted_high_low_range / df['volume'].shift(1)
    
    # Final Alpha Factor Combination
    w1 = 0.7  # Weight for momentum and volume-adjusted factors
    w2 = 0.3  # Weight for adjusted high-low range momentum
    final_factor = (total_momentum + volume_adjusted_return + volume_weighted_price_movement) * w1 + adjusted_high_low_momentum * w2
    
    return final_factor
```
