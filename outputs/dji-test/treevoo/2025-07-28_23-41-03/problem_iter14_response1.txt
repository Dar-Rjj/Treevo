```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Price Momentum Component
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    price_momentum = (ema_100 - ema_200) + (ema_50 - ema_100)

    # Intraday Reversal Momentum Component
    intraday_return = df['open'] / df['close'] - 1
    reversal_signal = np.sign(intraday_return)
    true_range = pd.Series(np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    ), index=df.index)
    true_range_reversal = true_range * reversal_signal
    sma_true_range_reversal = true_range_reversal.rolling(window=5).mean()

    # Volume-Adjusted Momentum Indicator
    intraday_return_ratio = (df['high'] - df['low']) / df['low']
    open_to_close_return = (df['close'] - df['open']) / df['open']
    avg_volume_n_days = df['volume'].rolling(window=5).mean()
    weighted_open_to_close_return = open_to_close_return * avg_volume_n_days
    volume_flow = (df['close'] - df['open']) * df['volume']
    high_low_range = df['high'] - df['low']
    high_low_range_momentum = high_low_range - high_low_range.shift(1)
    avg_volume = df['volume'].rolling(window=5).sum() / 5
    volume_adjusted_momentum = (intraday_return_ratio + weighted_open_to_close_return + volume_flow + high_low_range_momentum) / avg_volume

    # Simple Moving Average (SMA) Combination
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_combination = sma_50 - sma_200

    # Combined Alpha Factor
    alpha_factor = 0.3 * price_momentum + 0.4 * volume_adjusted_momentum + 0.2 * sma_combination + 0.1 * sma_true_range_reversal
    return alpha_factor
```
