```python
def heuristics_v2(df):
    # Calculate the relative position of closing price within the range [low, high]
    df['rel_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the average of the high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the closing price to the average price
    df['price_ratio'] = df['close'] / df['avg_price']
    
    # Calculate the factor: (relative position of close * price ratio)
    factor = df['rel_pos'] * df['price_ratio']
    
    return factor
```
