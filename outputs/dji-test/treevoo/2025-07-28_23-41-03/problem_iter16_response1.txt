import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range Momentum
    df['High-Low_Range'] = df['high'] - df['low']
    df['High-Low_Momentum'] = df['High-Low_Range'] - df['High-Low_Range'].shift(1)

    # Calculate 20-Day Weighted Moving Average of Returns
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    weighted_returns = (daily_returns * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    df['WMA_Returns'] = weighted_returns / total_volume

    # Adjust for Price Volatility
    avg_daily_price_range = (df['high'] - df['low']).rolling(window=20).mean()
    df['Adjusted_WMA_Returns'] = df['WMA_Returns'] - avg_daily_price_range

    # Calculate 20-Day Weighted Moving Average of Intraday Return Ratio
    intraday_return_ratio = (df['high'] - df['low']) / df['low']
    weighted_intraday_ratios = (intraday_return_ratio * df['volume']).rolling(window=20).sum()
    df['WMA_Intraday_Ratios'] = weighted_intraday_ratios / total_volume

    # Combine Adjusted Momentum and Intraday Factors
    combined_factor = (df['Adjusted_WMA_Returns'] + df['WMA_Intraday_Ratios']) * df['High-Low_Momentum']

    return combined_factor.dropna()
