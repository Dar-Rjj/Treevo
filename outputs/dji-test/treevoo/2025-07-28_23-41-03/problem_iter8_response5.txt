```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_price_range'] = (df['high'] - df['low']).rolling(window=22).mean()
    df['adjusted_weighted_return'] = df['weighted_return'] - df['avg_price_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_openclose_return'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_momentum'] = df['high_low_range'].diff()
    
    # Adjust by Volume
    n = 20  # Number of days for volume adjustment
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Price Trend
    m = 50  # Number of days for SMA
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_trend'] = df['sma_close'] - df['close']
    
    # Combine Adjusted Momentum, Intraday, and High-Low Range Factors
    df['combined_factor'] = df['adjusted_weighted_return'] + df['intraday_return_ratio'] + df['weighted_openclose_return'] + df['high_low_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_factor'] / df['avg_volume']) * df['price_trend']
    
    return df['alpha_factor']
```
