```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Price Momentum Component
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    price_momentum = (sma_50 - sma_100) + (sma_100 - sma_200)
    
    # Intraday Reversal Momentum Component
    intraday_return = (df['open'] - df['close']) / df['close']
    reversal_signal = np.where(intraday_return > 0, -1, 1)
    
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()  # 14-day ATR for example
    
    atr_reversal = atr * reversal_signal
    ema_atr_reversal = atr_reversal.ewm(span=7, adjust=False).mean()  # 7-day EMA for example
    
    # Combine both components
    alpha_factor = price_momentum + ema_atr_reversal
    
    return alpha_factor
```
