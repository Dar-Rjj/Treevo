```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)
    
    # Calculate Average True Range (ATR)
    df['previous_close'] = df['close'].shift(1)
    df['high_low_range'] = df['high'] - df['low']
    df['high_close_range'] = abs(df['high'] - df['previous_close'])
    df['low_close_range'] = abs(df['low'] - df['previous_close'])
    df['true_range'] = df[['high_low_range', 'high_close_range', 'low_close_range']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Exponential Moving Average (EMA) of ATR * Reversal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=7, min_periods=7).mean()
    
    # Calculate Simple Moving Averages
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff_100_200'] = df['sma_100'] - df['sma_200']
    df['sma_diff_50_100'] = df['sma_50'] - df['sma_100']
    
    # Calculate Price Momentum Over 5 Days
    df['price_momentum_5d'] = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Surge Over 5 Days
    df['volume_surge_5d'] = df['volume'] - df['volume'].shift(5)
    
    # Combine Momentum and Volume Indicators
    price_momentum_weighted = 0.5 * df['price_momentum_5d']
    volume_surge_weighted = 0.5 * df['volume_surge_5d']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = (
        price_momentum_weighted +
        volume_surge_weighted +
        df['sma_diff_100_200'] +
        df['sma_diff_50_100'] +
        df['ema_atr_reversal']
    )
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    average_volume = df['volume'].rolling(window=30).mean()
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * average_volume
    
    # Calculate Volume Flow
    df['volume_flow'] = df['volume'] * (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust by Volume and Amount
    n = 30
    m = 30
    average_volume = df['volume'].rolling(window=n).sum() / n
    average_amount = df['amount'].rolling(window=m).sum() / m
    df['combined_factor'] = (df['intraday_return'] + df['weighted_open_to_close_return'] + df['volume_flow'] + df['high_low_range_momentum']) / (average_volume + average_amount)
    
    return df['combined_factor']
```
