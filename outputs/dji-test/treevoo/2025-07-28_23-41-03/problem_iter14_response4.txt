```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    sma_diff_100_200 = sma_100 - sma_200
    sma_diff_50_100 = sma_50 - sma_100

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Reversal Signal
    reversal_signal = np.where(intraday_return > 0, -1, 1)

    # Combine Momentum and Reversal Signals
    combined_signal_100_200 = sma_diff_100_200 * reversal_signal
    combined_signal_50_100 = sma_diff_50_100 * reversal_signal

    # Compute Average True Range (ATR)
    tr = np.maximum.reduce([df['high'] - df['low'], 
                            abs(df['high'] - df['close'].shift(1)), 
                            abs(df['low'] - df['close'].shift(1))])
    atr = tr.rolling(window=14).mean()

    # Integrate ATR with Combined Signal
    integrated_signal_100_200 = atr * combined_signal_100_200
    integrated_signal_50_100 = atr * combined_signal_50_100

    # Smooth Result with Exponential Moving Average (EMA)
    ema_integrated_signal_100_200 = integrated_signal_100_200.ewm(span=7).mean()
    ema_integrated_signal_50_100 = integrated_signal_50_100.ewm(span=7).mean()

    # High-Low Range Momentum with Volume Adjustment
    high_low_range = df['high'] - df['low']
    high_low_momentum = high_low_range - high_low_range.shift(1)
    avg_volume = df['volume'].rolling(window=20).mean()
    high_low_momentum_adjusted = high_low_momentum / avg_volume

    # Final Alpha Factor Combination
    price_intraday_reversal_momentum = (sma_diff_50_100 + sma_diff_100_200) + (ema_integrated_signal_50_100 + ema_integrated_signal_100_200)
    alpha_factor = (price_intraday_reversal_momentum * 0.6) + (high_low_momentum_adjusted * 0.4)

    # Additional Layer: Volume-Momentum Interaction
    volume_momentum = df['volume'].rolling(window=20).mean()
    interaction_term = volume_momentum * high_low_momentum
    final_alpha_factor = (price_intraday_reversal_momentum * 0.6) + (high_low_momentum_adjusted * 0.3) + (interaction_term * 0.1)

    return final_alpha_factor
```
