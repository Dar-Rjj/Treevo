```python
def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'].diff()
    
    # Adjust by Volume: Get the average volume over the past n days
    n = 5  # Number of days to look back for volume average
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Combine High-Low Range Momentum and Volume Adjustment
    df['factor'] = df['high_low_range_momentum'] / df['avg_volume']
    
    return df['factor']
```
