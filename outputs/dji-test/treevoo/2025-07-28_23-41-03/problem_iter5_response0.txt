```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    sma_diff_100_200 = sma_100 - sma_200
    sma_diff_50_100 = sma_50 - sma_100
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    reversal_signal = np.where(intraday_return > 0, -1, 1)
    
    # Combine Momentum and Reversal Signals
    combined_signal_100_200 = sma_diff_100_200 * reversal_signal
    combined_signal_50_100 = sma_diff_50_100 * reversal_signal
    
    # Compute Average True Range (ATR)
    tr = np.maximum.reduce([df['high'] - df['low'],
                            abs(df['high'] - df['close'].shift(1)),
                            abs(df['low'] - df['close'].shift(1))])
    atr = tr.rolling(window=14).sum()  # 14-day period for ATR
    
    # Integrate ATR with Combined Signal
    integrated_signal_100_200 = atr * combined_signal_100_200
    integrated_signal_50_100 = atr * combined_signal_50_100
    
    # Smooth Result with Exponential Moving Average (EMA)
    ema_integrated_signal_100_200 = integrated_signal_100_200.ewm(span=7, adjust=False).mean()
    ema_integrated_signal_50_100 = integrated_signal_50_100.ewm(span=7, adjust=False).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range - high_low_range.shift(1)
    
    # Adjust by Volume and Amount
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    normalized_volume = avg_volume / avg_amount
    
    # Combine High-Low Range Momentum and Adjusted Volume
    combined_high_low_vol = high_low_momentum / normalized_volume
    
    # Final Alpha Factor
    final_alpha = (ema_integrated_signal_100_200 + 
                   ema_integrated_signal_50_100 + 
                   combined_high_low_vol)
    
    return final_alpha
```
