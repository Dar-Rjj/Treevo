```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
        
    Returns:
        pd.Series: Series with the calculated alpha factor values indexed by date.
    """
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the average price of the day
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / daily_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / daily_range
    
    # Calculate the volume-weighted buying pressure
    volume_weighted_buying_pressure = buying_pressure * df['volume']
    
    # Calculate the volume-weighted selling pressure
    volume_weighted_selling_pressure = selling_pressure * df['volume']
    
    # Calculate the net volume-weighted pressure
    net_volume_weighted_pressure = volume_weighted_buying_pressure - volume_weighted_selling_pressure
    
    # Calculate the final alpha factor
    alpha_factor = (net_volume_weighted_pressure / df['volume']) * avg_price
    
    return alpha_factor
```
