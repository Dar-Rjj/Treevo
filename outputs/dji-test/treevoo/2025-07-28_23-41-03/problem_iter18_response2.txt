import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between high and low prices to capture volatility
    df['rel_diff_high_low'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the return for the day
    df['return'] = df['close'] / df['open'] - 1
    
    # Calculate the weighted average of high, low, and close prices using volume
    df['weighted_avg_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Create the alpha factor by combining the above metrics
    df['alpha_factor'] = (df['weighted_avg_price'] * df['return'] * df['rel_diff_high_low'])
    
    # Ensure the alpha factor is a Series indexed by date
    alpha_factor = df['alpha_factor']
    
    return alpha_factor
