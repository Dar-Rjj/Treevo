```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    reversal_signal = intraday_return.apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Weighted Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    weighted_open_to_close_return = open_to_close_return * df['volume']
    
    # Calculate Momentum Component
    high_low_diff = df['high'] - df['low']
    high_prev_close_diff = (df['high'] - df['close'].shift(1)).abs()
    low_prev_close_diff = (df['low'] - df['close'].shift(1)).abs()
    tr = pd.concat([high_low_diff, high_prev_close_diff, low_prev_close_diff], axis=1).max(axis=1)
    atr = tr.rolling(window=14).mean()  # 14-day ATR
    momentum_component = atr * reversal_signal
    momentum_component_smoothed = momentum_component.ewm(span=5, adjust=False).mean()  # Short EMA period
    
    # Calculate Volume Flow
    volume_flow = df['volume'] * ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Combine Factors
    alpha_factor = weighted_open_to_close_return + momentum_component_smoothed + volume_flow
    
    return alpha_factor
```
