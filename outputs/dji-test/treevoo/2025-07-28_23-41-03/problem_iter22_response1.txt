```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['50_SMA'] = df['close'].rolling(window=50).mean()
    df['100_SMA'] = df['close'].rolling(window=100).mean()
    df['200_SMA'] = df['close'].rolling(window=200).mean()

    # Momentum Signals
    df['Momentum_100_200'] = df['100_SMA'] - df['200_SMA']
    df['Momentum_50_100'] = df['50_SMA'] - df['100_SMA']

    # Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Reversal Signal
    df['Reversal_Signal'] = df['Intraday_Return'].apply(lambda x: -1 if x > 0 else 1)

    # Combine Momentum and Reversal
    df['Combined_Momentum_100_200'] = df['Momentum_100_200'] * df['Reversal_Signal']
    df['Combined_Momentum_50_100'] = df['Momentum_50_100'] * df['Reversal_Signal']

    # High-Low Range Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)

    # Volume Adjustment
    n_days = 50
    df['Average_Volume'] = df['volume'].rolling(window=n_days).sum() / n_days

    # Adjusted High-Low Range Momentum
    df['Adjusted_High_Low_Range_Momentum'] = df['High_Low_Range_Momentum'] / df['Average_Volume']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        (df['50_SMA'] - df['100_SMA'] + df['100_SMA'] - df['200_SMA']) * 
        df['Reversal_Signal'] +
        df['Adjusted_High_Low_Range_Momentum']
    )

    return df['Final_Alpha_Factor'].dropna()
```
