```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Volume-Weighted Average Return
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Adjust for Price Volatility
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_volume_weighted_return'] = df['volume_weighted_return'] - df['avg_daily_range']

    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['close']

    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factor'] = df['adjusted_volume_weighted_return'] + df['intraday_return_ratio']

    # Intraday Reversal Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)

    # Calculate True Range
    df['true_range'] = pd.concat([
        (df['high'] - df['low']),
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)

    # Calculate Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()

    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr_14'] * df['reversal_signal']

    # Calculate Exponential Moving Average (EMA) of ATR * Reversal Signal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['ema_atr_reversal']

    return df['alpha_factor']
```
