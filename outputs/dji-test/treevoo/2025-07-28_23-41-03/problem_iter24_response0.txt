```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the overnight price movement
    df['overnight_movement'] = df['open'] - df['close'].shift(1)
    
    # Calculate the closing price momentum
    df['momentum_close'] = df['close'] - df['close'].shift(1)
    
    # Heuristic: High intraday range and positive overnight movement suggest a continuation of the trend
    df['alpha_factor'] = (df['intraday_range'] * df['overnight_movement'] > 0) * df['momentum_close']
    
    return df['alpha_factor']
```
