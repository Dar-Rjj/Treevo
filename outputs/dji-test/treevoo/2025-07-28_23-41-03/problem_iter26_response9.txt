def heuristics_v2(df):
    # Calculate the rolling mean of the close price over the last 5 days
    df['close_rolling_mean_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between the close price and its 5-day rolling mean
    df['close_diff_rolling_mean_5'] = df['close'] - df['close_rolling_mean_5']
    
    # Calculate the average true range (ATR) over a 14-day period
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['close_diff_rolling_mean_5'] / df['atr_14'] * (100 - df['rsi_14'])
    
    return df['alpha_factor']
