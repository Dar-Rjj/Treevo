```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    diff_100_200 = sma_100 - sma_200
    diff_50_100 = sma_50 - sma_100
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine the components to form the alpha factor
    factor = (diff_50_100 + diff_100_200) / high_low_range
    
    return factor
```
