```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages for 50, 100, and 200 days
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    crossover_100_200 = sma_100 - sma_200
    crossover_50_100 = sma_50 - sma_100

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Combine SMA Crossover and High-Low Range
    combined_crossover_100_200 = crossover_100_200 / high_low_range
    combined_crossover_50_100 = crossover_50_100 / high_low_range

    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range.diff()

    # Adjust by Volume: Average Volume over the past n days
    n = 50  # number of days to average volume over
    avg_volume = df['volume'].rolling(window=n).sum() / n

    # Combine All Factors
    combined_factors = (combined_crossover_100_200 + combined_crossover_50_100 + high_low_momentum) / avg_volume

    return combined_factors
```
