```python
import pandas as pd

def heuristics_v2(df):
    # Price Momentum - Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Price Momentum - Exponential Moving Averages (EMA)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Price Momentum - Combined Momentum
    combined_momentum = (ema_50 - ema_100) + (ema_100 - ema_200)
    combined_momentum += intraday_momentum
    
    # Volume-Adjusted Factors - Volume-Adjusted Return
    avg_volume = df['volume'].rolling(window=20).mean()  # Assuming N = 20 for average volume
    volume_adjusted_return = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    # Volume-Adjusted Factors - Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    
    # Volume-Adjusted Factors - Volume-Adjusted Closing Price
    volume_adjusted_close = df['close'] * (df['volume'] / avg_volume)
    
    # High-Low Range Momentum - High-Low Range
    high_low_range = df['high'] - df['low']
    
    # High-Low Range Momentum - Adjusted High-Low Range
    adjusted_high_low_range = high_low_range - high_low_range.shift(1)
    
    # High-Low Range Momentum - High-Low Range Momentum
    high_low_range_momentum = adjusted_high_low_range / df['volume'].shift(1)
    
    # Final Alpha Factor Combination
    w1 = 0.8  # Weight for Momentum and Volume-Adjusted Factors
    w2 = 0.2  # Weight for High-Low Range Momentum
    alpha_factor = (combined_momentum + volume_adjusted_return + volume_weighted_price_movement) * w1
    alpha_factor += (high_low_range_momentum / df['volume'].shift(1)) * w2
    
    return alpha_factor
```
