```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to high (C/H) as a measure of day's performance
    df['C/H'] = df['close'] / df['high']
    
    # Compute the difference between high and low prices adjusted by volume (VWAP-like metric)
    df['HL_Volume_Adjusted'] = (df['high'] - df['low']) * df['volume']
    
    # Measure the gap between opening and closing prices (O-C) to gauge market momentum
    df['O-C'] = df['open'] - df['close']
    
    # Develop a simple moving average (SMA) of close prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Construct a volume-weighted moving average (VWMA) of close prices
    df['VWMA_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWMA_20'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine C/H ratio, HL_Volume_Adjusted, and O-C gap into a single factor
    df['Composite_Factor'] = 0.4 * df['C/H'] + 0.3 * df['HL_Volume_Adjusted'] + 0.3 * df['O-C']
    
    # Incorporate SMA and VWMA crossovers with support and resistance levels
    df['SMA_Crossover'] = df['SMA_5'] > df['SMA_20']
    df['VWMA_Crossover'] = df['VWMA_5'] > df['VWMA_20']
    
    # Examine the sequence of price and volume changes over time
    df['Price_Volume_Sequence'] = (df['close'].diff() > 0) & (df['volume'].diff() > 0)
    
    # Create a composite signal that considers both price and volume dynamics
    df['Composite_Signal'] = 0.5 * df['SMA_Crossover'].astype(int) + 0.5 * df['VWMA_Crossover'].astype(int)
    
    # Final alpha factor
    df['Alpha_Factor'] = 0.5 * df['Composite_Factor'] + 0.5 * df['Composite_Signal']
    
    return df['Alpha_Factor']
```
