```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily return using close prices
    df['daily_return'] = df['close'].diff()
    
    # Calculate the range (high - low)
    df['range'] = df['high'] - df['low']
    
    # Measure the gap opening (open - previous day's close)
    df['gap_opening'] = df['open'] - df['close'].shift(1)
    
    # Examine the closing direction (close - open)
    df['closing_direction'] = df['close'] - df['open']
    
    # Calculate daily change in volume
    df['volume_change'] = df['volume'].diff()
    
    # Measure the ratio of volume to amount
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Analyze the relationship between volume and price change
    df['volume_price_relationship'] = df['volume'] * df['daily_return']
    
    # Create a price-volume momentum indicator
    df['price_volume_momentum'] = df['daily_return'] * df['volume_change']
    
    # Develop a volatility-adjusted volume factor
    df['volatility_adjusted_volume'] = (df['daily_return'] / df['range']) * df['volume']
    
    # Detect changes in the direction of the 5-day moving average of close prices
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['trend_reversal'] = (df['5_day_ma'] > df['5_day_ma'].shift(1)) != (df['5_day_ma'].shift(1) > df['5_day_ma'].shift(2))
    df['trend_reversal'] = df['trend_reversal'].astype(int)
    
    # Sum the product of daily return and volume over the past 10 days
    df['cumulative_volume_price_change'] = (df['daily_return'] * df['volume']).rolling(window=10).sum()
    
    # Combine all features into a single alpha factor
    alpha_factor = (
        df['daily_return'] +
        df['range'] +
        df['gap_opening'] +
        df['closing_direction'] +
        df['volume_change'] +
        df['volume_to_amount_ratio'] +
        df['volume_price_relationship'] +
        df['price_volume_momentum'] +
        df['volatility_adjusted_volume'] +
        df['trend_reversal'] +
        df['cumulative_volume_price_change']
    )
    
    return alpha_factor
```
