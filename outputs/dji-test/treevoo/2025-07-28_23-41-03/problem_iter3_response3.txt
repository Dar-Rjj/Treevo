```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Compute Intraday Returns
    data['intraday_return_high'] = (data['high'] - data['open']) / data['open']
    data['intraday_return_low'] = (data['low'] - data['open']) / data['open']
    data['intraday_return_close'] = (data['close'] - data['open']) / data['open']

    # Average of Intraday Returns
    data['avg_intraday_return'] = (data['intraday_return_high'] + data['intraday_return_low'] + data['intraday_return_close']) / 3

    # Compute Log Returns
    data['log_return'] = np.log(data['close']).diff()

    # Calculate Standard Deviation of Log Returns
    data['daily_volatility'] = data['log_return'].rolling(window=20).std()

    # Combine Intraday Momentum and Daily Volatility
    data['combined_factor'] = data['avg_intraday_return'] * data['daily_volatility']

    # Adjust for Volume
    data['volume_adjusted_factor'] = data['combined_factor'] / data['volume']

    # Compute Volume Moving Average
    data['vol_moving_avg'] = data['volume'].rolling(window=20).mean()

    # Adjust Combined Factor by Volume Moving Average
    data['final_factor'] = data['volume_adjusted_factor'] / data['vol_moving_avg']

    # Return the final factor
    return data['final_factor']
```
