```python
import pandas as pd

def heuristics_v2(df, n=10, m=30):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'].diff()
    
    # Adjust by Volume: Calculate the n-day average volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Price Trend: SMA of Close prices over m days
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_trend'] = df['close'] - df['sma_close']
    
    # Combine High-Low Range Momentum, Volume Adjustment, and Price Trend
    df['factor'] = (df['high_low_range_momentum'] / df['avg_volume']) * df['price_trend']
    
    # Return the factor as a Series
    return df['factor']
```
