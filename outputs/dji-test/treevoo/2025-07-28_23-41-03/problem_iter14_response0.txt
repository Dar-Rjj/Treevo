def heuristics_v2(df):
    """
    A function to generate a novel and interpretable alpha factor.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by 'date'.
        
    Returns:
        pd.Series: A Series indexed by 'date' representing the alpha factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-period EMA of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-period EMA of the volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Generate the alpha factor
    alpha_factor = ema_money_flow / ema_volume
    
    return alpha_factor
