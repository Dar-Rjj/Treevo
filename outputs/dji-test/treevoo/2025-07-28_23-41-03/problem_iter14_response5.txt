```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = np.sign(df['intraday_return']).apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Exponential Moving Average (EMA) of ATR * Reversal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=7, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'].diff()
    
    # Adjust by Volume and Amount
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    df['combined_avg'] = df['avg_volume_10'] + df['avg_amount_10']
    
    # Combine High-Low Range Momentum and Adjusted Volume-Amount
    df['adjusted_high_low_momentum'] = df['high_low_range_momentum'] / df['combined_avg']
    
    # Calculate Simple Moving Averages
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff_100_200'] = df['sma_100'] - df['sma_200']
    df['sma_diff_50_100'] = df['sma_50'] - df['sma_100']
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_returns_20'] = (df['daily_price_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_open_close_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Combine Momentum and Volume Indicators
    df['alpha_factor'] = (0.3 * df['weighted_returns_20'] +
                          0.2 * df['intraday_return_ratio'] +
                          0.1 * df['weighted_open_close_return'] +
                          0.2 * df['sma_diff_100_200'] +
                          0.2 * df['sma_diff_50_100'] +
                          df['ema_atr_reversal'] +
                          df['adjusted_high_low_momentum'])
    
    return df['alpha_factor']
```
