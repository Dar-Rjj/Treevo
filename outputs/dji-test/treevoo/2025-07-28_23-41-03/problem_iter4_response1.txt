```python
def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_to_low_momentum = df['high'] - df['low']
    open_to_close_momentum = df['close'] - df['open']

    # Calculate Daily Volume Increase
    daily_volume_increase = df['volume'].pct_change()

    # Weigh High-to-Low Price Change by Daily Volume Increase
    weighted_high_to_low_momentum = high_to_low_momentum * daily_volume_increase

    # Weigh Open-to-Close Price Change by Daily Volume Increase
    weighted_open_to_close_momentum = open_to_close_momentum * daily_volume_increase

    # Combine the two momentum factors
    intraday_momentum_with_volume_thrust = weighted_high_to_low_momentum + weighted_open_to_close_momentum
    
    return intraday_momentum_with_volume_thrust
```
