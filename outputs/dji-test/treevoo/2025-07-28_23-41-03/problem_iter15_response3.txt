```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_return'] = df['daily_return'] * df['volume']
    df['sum_weighted_returns'] = df['weighted_return'].rolling(window=20).sum()
    df['sum_volumes'] = df['volume'].rolling(window=20).sum()
    df['wma_returns'] = df['sum_weighted_returns'] / df['sum_volumes']

    # Adjust for Price Volatility (20-Day Average True Range)
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1).shift(1)
    df['atr_20'] = df['true_range'].rolling(window=22).mean()
    df['adjusted_wma_returns'] = df['wma_returns'] - df['atr_20']

    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['low']

    # Calculate Weighted Open-to-Close Return
    df['weighted_open_to_close_return'] = (df['close'] - df['open']) * df['volume']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate High-Low Range Momentum
    df['high_low_momentum'] = df['high_low_range'].diff()

    # Adjust by Volume (Average Volume over past n days)
    df['avg_volume'] = df['volume'].rolling(window=20).mean()

    # Calculate Price Trend
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['price_trend'] = df['sma_50'] - df['sclose']

    # Combine Adjusted Momentum, Intraday, and High-Low Range Factors
    df['combined_factors'] = df['adjusted_wma_returns'] + df['intraday_return_ratio'] + df['weighted_open_to_close_return'] + df['high_low_momentum']

    # Price Momentum Component
    df['sma_50_100'] = df['sma_50'] - df['sma_100']
    df['sma_100_200'] = df['sma_100'] - df['sma_200']
    df['price_momentum'] = df['sma_50_100'] + df['sma_100_200']

    # Intraday Reversal Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    df['atr'] = df[['high', 'low', 'close']].diff().abs().sum(axis=1).shift(1).rolling(window=22).mean()
    df['intraday_reversal_momentum'] = df['atr'] * df['reversal_signal']

    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_factors'] + df['price_momentum'] + df['intraday_reversal_momentum']) / df['avg_volume'] * df['price_trend']
    
    return df['alpha_factor']
```
