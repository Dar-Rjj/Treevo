```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 20-Day Weighted Moving Average of Returns
    def weighted_moving_average_returns(df, window=20):
        weighted_sum = (df['daily_return'] * df['volume']).rolling(window=window).sum()
        volume_sum = df['volume'].rolling(window=window).sum()
        return weighted_sum / volume_sum
    
    df['wma_returns'] = weighted_moving_average_returns(df)

    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']

    # Calculate 20-Day Average Daily Price Range
    df['avg_daily_price_range'] = df['daily_price_range'].rolling(window=20).mean()

    # Final Factor Value
    df['factor_value'] = df['wma_returns'] - df['avg_daily_price_range']

    return df['factor_value']
```
