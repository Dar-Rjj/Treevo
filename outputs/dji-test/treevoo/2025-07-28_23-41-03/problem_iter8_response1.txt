```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Volume-Weighted Average Return
    df['vol_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_vol_weighted_return'] = df['vol_weighted_return'] - df['avg_daily_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['weighted_open_close_return'] = df['open_close_return'] * df['volume']
    
    # Combine Adjusted Momentum and Intraday Factors
    df['combined_factors'] = df['adjusted_vol_weighted_return'] + df['intraday_return_ratio'] + df['weighted_open_close_return']
    
    # Calculate Simple Moving Averages (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff_100_200'] = sma_100 - sma_200
    df['sma_diff_50_100'] = sma_50 - sma_100
    
    # Combine SMAs
    df['sma_combined'] = df['sma_diff_50_100'] + df['sma_diff_100_200']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)
    
    # Calculate ATR
    df['tr'] = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    atr = df['tr'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = atr * df['reversal_signal']
    
    # Calculate Exponential Moving Average (EMA)
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=9, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] + df['sma_combined'] + df['ema_atr_reversal']
    
    return df['alpha_factor']
```
