```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['100_day_SMA'] = df['close'].rolling(window=100).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()

    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)

    # Momentum Signals
    df['long_momentum'] = df['100_day_SMA'] - df['200_day_SMA']
    df['short_momentum'] = df['50_day_SMA'] - df['100_day_SMA']

    # High-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Volume Adjustment
    n_days = 50  # You can adjust the number of days
    df['average_volume'] = df['volume'].rolling(window=n_days).sum() / n_days

    # Adjusted High-Low Range Momentum
    df['adjusted_high_low_range_momentum'] = df['high_low_range_momentum'] / df['average_volume']

    # Combined Momentum and Reversal
    df['combined_long_momentum'] = df['long_momentum'] * df['reversal_signal']
    df['combined_short_momentum'] = df['short_momentum'] * df['reversal_signal']

    # Price Change Momentum
    df['price_change_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combined Price and Volume Momentum
    df['combined_price_volume_momentum'] = df['price_change_momentum'] * df['average_volume']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_long_momentum'] + df['combined_short_momentum'] +
                                df['adjusted_high_low_range_momentum'] + 
                                df['combined_price_volume_momentum'])

    return df['final_alpha_factor']
```
