```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range Percentage
    intraday_range = (df['high'] - df['low']) / df['low'] * 100
    
    # Calculate Volume-Weighted Average Price (VWAP)
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (avg_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Analyze Volume Direction
    positive_volume_flow = df['volume'].where(df['close'] > df['open'], 0)
    negative_volume_flow = df['volume'].where(df['close'] <= df['open'], 0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (intraday_range + vwap + (positive_volume_flow - negative_volume_flow))
    
    return alpha_factor
```
