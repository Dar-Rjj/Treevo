```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'] / df['close'].shift(5) - 1
    
    # Calculate Long-Term Return
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Momentum Indicator
    momentum_indicator = long_term_return - short_term_return
    momentum_sign = momentum_indicator.apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Price Fluctuation
    price_fluctuation = df['high'] - df['low']
    
    # Integrate Volume into the Factor
    volume_sum_short_term = df['volume'].rolling(window=5).sum()
    volume_weighted_momentum = momentum_sign * volume_sum_short_term
    
    # Combine Momentum and Price Fluctuation
    factor = (volume_weighted_momentum / price_fluctuation).dropna()
    
    return factor
```
