```python
import pandas as pd

def heuristics_v2(df):
    # 20-Day Weighted Moving Average of Returns
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    df['weighted_return'] = df['daily_return'] * df['volume']
    df['weighted_return_sum_20'] = df['weighted_return'].rolling(window=20).sum()
    df['volume_sum_20'] = df['volume'].rolling(window=20).sum()
    df['wma_returns_20'] = df['weighted_return_sum_20'] / df['volume_sum_20']

    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    df['avg_price_range_20'] = df['price_range'].rolling(window=20).mean()
    df['adjusted_wma_returns_20'] = df['wma_returns_20'] - df['avg_price_range_20']

    # Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']

    # Weighted Open-to-Close Return
    df['weighted_oc_return'] = (df['close'] - df['open']) * df['volume']

    # High-Low Range Momentum
    df['hl_momentum'] = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))

    # Adjust by Volume
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()

    # Price Trend
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['price_trend'] = df['sma_50'] - df['close']

    # Combine Factors
    df['combined_factors'] = (df['adjusted_wma_returns_20'] + 
                              df['intraday_return_ratio'] + 
                              df['weighted_oc_return'] + 
                              df['hl_momentum'] + 
                              df['price_trend'])

    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] / df['avg_volume_30']

    return df['alpha_factor']
```
