```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Indicator using Close Prices
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    momentum_factor = df['SMA_10'] - df['SMA_20']
    
    # Volume-Weighted Price Movement Factor
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['EMA_VWAP_10'] = df['VWAP'].ewm(span=10, adjust=False).mean()
    df['EMA_VWAP_50'] = df['VWAP'].ewm(span=50, adjust=False).mean()
    vwap_factor = df['EMA_VWAP_10'] / df['EMA_VWAP_50']
    
    # Volatility Factor using High and Low Prices
    df['ATR'] = df['high'] - df['low']
    volatility_factor = df['ATR'].rolling(window=14).mean()
    
    # Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff()
    df['gain'] = df['price_change'].apply(lambda x: max(x, 0))
    df['loss'] = df['price_change'].apply(lambda x: -min(x, 0))
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['RS'] = df['avg_gain'] / df['avg_loss']
    rsi_factor = 100 - (100 / (1 + df['RS']))
    
    # Money Flow Index (MFI)
    df['TP'] = (df['high'] + df['low'] + df['close']) / 3
    df['MF'] = df['TP'] * df['volume']
    df['PMF'] = np.where(df['TP'] > df['TP'].shift(1), df['MF'], 0)
    df['NMF'] = np.where(df['TP'] < df['TP'].shift(1), df['MF'], 0)
    df['positive_money_flow'] = df['PMF'].rolling(window=14).sum()
    df['negative_money_flow'] = df['NMF'].rolling(window=14).sum()
    df['MR'] = df['positive_money_flow'] / df['negative_money_flow']
    mfi_factor = 100 - (100 / (1 + df['MR']))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (momentum_factor + vwap_factor + volatility_factor + rsi_factor + mfi_factor) / 5
    return alpha_factor
```
