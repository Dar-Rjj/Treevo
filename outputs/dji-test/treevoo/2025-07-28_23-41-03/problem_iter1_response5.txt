```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Over 5 Days
    price_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Surge Over 5 Days
    volume_surge = df['volume'] - df['volume'].shift(5)
    
    # Combine Momentum and Volume Indicators with weights
    weighted_price_momentum = 0.5 * price_momentum
    weighted_volume_surge = 0.5 * volume_surge
    
    # Sum the Weighted Indicators
    factor = weighted_price_momentum + weighted_volume_surge
    
    return factor
```
