import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum_high_low = df['high'] - df['low']
    intraday_momentum_open_close = df['close'] - df['open']

    # Volume Trend Analysis
    volume_trend = df['volume'].pct_change()
    volume_trend_smoothed = df['volume'].ewm(span=7).mean().pct_change()

    # Combine with Intraday Momentum
    high_low_weighted = intraday_momentum_high_low * volume_trend
    open_close_weighted = intraday_momentum_open_close * volume_trend_smoothed

    # Final Alpha Factor
    alpha_factor = (high_low_weighted + open_close_weighted) / 2

    return alpha_factor
