```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(n)
    
    # Identify Trend Reversals
    df['Trend_Reversal'] = (df['low'] > df['high'].shift(n)).astype(int)
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    
    # Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Reversal Signal
    df['Reversal_Signal'] = df['Intraday_Return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Measure Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Strength of Moves
    df['Strength_of_Moves'] = (df['close'] / df['close'].shift(1) - 1) * df['volume']
    
    # Volume Adjustment
    df['Volume_Adjustment'] = df['volume'].rolling(window=n).sum() / n
    
    # Gauge Financial Flows
    avg_transaction_amount = df['amount'].rolling(window=n).mean()
    df['Financial_Flows'] = (df['amount'] - avg_transaction_amount) / avg_transaction_amount
    
    # Generate Composite Alphas
    df['Combined_Momentum_and_Reversal'] = ((df['SMA_100'] - df['SMA_200']) * df['Reversal_Signal']) + ((df['SMA_50'] - df['SMA_100']) * df['Reversal_Signal'])
    
    # High-Low Range Momentum
    df['High_Low_Range_Momentum'] = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    # Adjusted High-Low Range Momentum
    df['Adjusted_High_Low_Range_Momentum'] = df['High_Low_Range_Momentum'] / df['Volume_Adjustment']
    
    # Combined Price and Volume Momentum
    df['Combined_Price_Volume_Momentum'] = df['Momentum'] * df['Volume_Adjustment']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Combined_Momentum_and_Reversal'] + df['Adjusted_High_Low_Range_Momentum'] + df['Combined_Price_Volume_Momentum']
    
    return df['Alpha_Factor']
```
