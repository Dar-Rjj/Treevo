```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Heuristics factor: (daily range / average price) * volume change
    factor = (daily_range / avg_price) * vol_change
    
    return factor
```
