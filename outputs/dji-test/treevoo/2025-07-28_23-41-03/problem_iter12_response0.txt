```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'].diff()
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_returns'] = (df['daily_price_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_daily_price_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['adjusted_weighted_returns'] = df['weighted_returns'] - df['avg_daily_price_range']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_high_low_momentum'] = df['high_low_range_momentum'] / df['avg_volume']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = df['high'] / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_open_close_return'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate 20-Day Weighted Moving Average of Intraday Return Ratio
    df['weighted_intraday_return_ratio'] = (df['intraday_return_ratio'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine Adjusted Momentum, Intraday Factors, and Volume-Weighted Intraday Return
    df['combined_factors'] = df['adjusted_high_low_momentum'] + df['adjusted_weighted_returns'] + df['weighted_intraday_return_ratio']
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] * df['weighted_intraday_return_ratio']
    
    return df['alpha_factor']
```
