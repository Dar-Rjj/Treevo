```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Reversal Signal
    df['volume_weighted_signal'] = df['intraday_return'].apply(lambda x: -1 * df['volume'] if x > 0 else 1 * df['volume'])
    
    # Calculate ATR Component
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    atr = df['true_range'].rolling(window=14).mean()
    
    # Calculate Momentum Component
    df['momentum_component'] = atr * df['volume_weighted_signal']
    
    # Exponential Moving Average (EMA) with a short period
    ema_short_period = df['momentum_component'].ewm(span=5, adjust=False).mean()
    
    return ema_short_period
```
