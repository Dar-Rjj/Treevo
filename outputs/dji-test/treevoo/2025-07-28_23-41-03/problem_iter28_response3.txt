```python
import pandas as pd
import talib

def heuristics_v2(df):
    # Momentum Indicators
    sma_short = df['close'].rolling(window=5).mean()
    sma_long = df['close'].rolling(window=20).mean()
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume Indicators
    obv = talib.OBV(df['close'], df['volume'])
    
    # Volatility Indicators
    atr = talib.ATR(df['high'], df['low'], df['close'], timeperiod=14)
    
    # Oscillators
    rsi = talib.RSI(df['close'], timeperiod=14)
    
    # Price Patterns
    bullish_engulfing = talib.CDLENGULFING(df['open'], df['high'], df['low'], df['close'])
    bearish_engulfing = -bullish_engulfing
    bullish_harami = talib.CDLHARAMI(df['open'], df['high'], df['low'], df['close'])
    bearish_harami = -bullish_harami
    
    # Generate Alpha Factors
    factor_1 = (sma_short - sma_long) / df['close']
    factor_2 = (ema_short - ema_long) / df['close']
    factor_3 = obv / df['volume']
    factor_4 = atr / df['high']
    factor_5 = rsi
    factor_6 = (df['close'] - df['open']) / (df['high'] - df['low'])
    factor_7 = ((bullish_engulfing > 0) | (bearish_engulfing < 0)).astype(int)
    factor_8 = ((bullish_harami > 0) | (bearish_harami < 0)).astype(int)
    
    # Combine all factors
    combined_factor = (factor_1 + factor_2 + factor_3 + factor_4 + factor_5 + factor_6 + factor_7 + factor_8) / 8
    
    return combined_factor
```
