```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the closing position within the daily range
    df['close_position'] = (df['close'] - df['low']) / df['daily_range']
    
    # Calculate the normalized volume, which is not for normalization but for relative comparison
    df['volume_change'] = df['volume'].pct_change()
    
    # Generate alpha factor by combining the above signals
    df['alpha_factor'] = (
        df['intraday_return'] * 0.4 +
        (df['close_position'] - 0.5) * 0.3 +  # Shifted to measure the extreme positions
        df['volume_change'] * 0.3
    )
    
    return df['alpha_factor']
```
