```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = np.where(df['intraday_return'] > 0, -1, 1)
    
    # Calculate Average True Range (ATR)
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Multiply ATR by Reversal Signal
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    
    # Exponential Moving Average (EMA) of ATR * Reversal
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=7, adjust=False).mean()
    
    # Calculate Simple Moving Averages
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff_100_200'] = df['sma_100'] - df['sma_200']
    df['sma_diff_50_100'] = df['sma_50'] - df['sma_100']
    
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_returns'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust for Price Volatility
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['volatility_adjusted_returns'] = df['weighted_returns'] - df['avg_daily_range']
    
    # Calculate Intraday Return Ratio
    df['intraday_return_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Weighted Open-to-Close Return
    df['weighted_open_to_close_return'] = (df['close'] - df['open']) * df['volume']
    
    # Combine Momentum and Volume Indicators
    df['combined_alpha'] = (
        0.3 * df['weighted_returns'] +
        0.2 * df['intraday_return_ratio'] +
        0.1 * df['weighted_open_to_close_return'] +
        0.2 * df['sma_diff_100_200'] +
        0.2 * df['sma_diff_50_100'] +
        df['ema_atr_reversal']
    )
    
    return df['combined_alpha']
```
