```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index and the alpha factor values.
    """
    
    # Calculate the relative range of the day
    range_day = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative body of the candle
    body_candle = (df['close'] - df['open']) / df['open']
    
    # Calculate the weighted average price
    wap = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the price momentum
    price_momentum = df['close'].pct_change()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (range_day * 0.4) + (body_candle * 0.3) + (wap * 0.2) + (price_momentum * 0.1)
    
    return alpha_factor
```
