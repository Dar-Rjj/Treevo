```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Returns
    intraday_return_high = (df['high'] - df['open']) / df['open']
    intraday_return_low = (df['low'] - df['open']) / df['open']
    intraday_return_close = (df['close'] - df['open']) / df['open']

    # Average of Intraday Returns
    intraday_momentum = (intraday_return_high + intraday_return_low + intraday_return_close) / 3

    # Compute Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))

    # Calculate Standard Deviation of Log Returns
    daily_volatility = log_returns.rolling(window=20).std()

    # Combine Intraday Momentum and Daily Volatility
    combined_factor = intraday_momentum * daily_volatility

    # Adjust for Volume
    adjusted_factor = combined_factor / df['volume']

    return adjusted_factor
```
