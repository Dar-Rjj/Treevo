```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range Percentage
    df['PriceRangePct'] = (df['high'] - df['low']) / df['low'] * 100
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['TP_Volume'] = df['TypicalPrice'] * df['volume']
    vwap = df['TP_Volume'].sum() / df['volume'].sum()
    df['VWAP'] = vwap
    
    # Analyze Volume Direction
    df['VolumeDirection'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else -row['volume'], axis=1)
    
    # Calculate Intraday Volatility
    df['IntradayVolatility'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Money Flow Index (MFI)
    df['TypicalPrice'] = (df['high'] + df['low'] + df['close']) / 3
    df['RawMoneyFlow'] = df['TypicalPrice'] * df['volume']
    df['PositiveMoneyFlow'] = df[df['close'] > df['close'].shift(1)]['RawMoneyFlow'].rolling(window=14).sum()
    df['NegativeMoneyFlow'] = df[df['close'] <= df['close'].shift(1)]['RawMoneyFlow'].rolling(window=14).sum().abs()
    df['MoneyFlowRatio'] = df['PositiveMoneyFlow'] / df['NegativeMoneyFlow']
    df['MFI'] = 100 - (100 / (1 + df['MoneyFlowRatio']))
    
    # Analyze Intraday Momentum
    df['Momentum'] = (df['close'] - df['open']) / df['open'] * 100
    df['MomentumCategory'] = df['Momentum'].apply(lambda x: 'Positive' if x > 0 else 'Negative')
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (df['PriceRangePct'] + df['VWAP'] + df['VolumeDirection'] + 
                    df['IntradayVolatility'] + df['MFI'] + df['Momentum'])
    
    return alpha_factor
```
