```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Weighted Moving Average of Returns
    df['weighted_return'] = df['daily_price_return'] * df['volume']
    df['sum_weighted_return'] = df['weighted_return'].rolling(window=22).sum()
    df['sum_volume'] = df['volume'].rolling(window=22).sum()
    df['wma_returns'] = df['sum_weighted_return'] / df['sum_volume']
    
    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    df['avg_price_range'] = df['price_range'].rolling(window=22).mean()
    df['adjusted_wma_returns'] = df['wma_returns'] - df['avg_price_range']
    
    # Introduce Volume-Weighted Price Change
    df['volume_weighted_price'] = (df['close'] * df['volume']).rolling(window=22).sum() / df['sum_volume']
    df['volume_weighted_price_change'] = df['close'] - df['volume_weighted_price']
    
    # Compute Average True Range (ATR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['close'].shift(1) - df['high']).abs()
    df['tr3'] = (df['close'].shift(1) - df['low']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=22).mean()
    
    # Combine Factors
    df['atr_reversal'] = df['atr'] * df['reversal_signal']
    df['ema_atr_reversal'] = df['atr_reversal'].ewm(span=10, adjust=False).mean()
    
    # Simple Moving Averages
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Subtract Longer SMA from Shorter SMA
    df['sma_diff_100_200'] = df['sma_100'] - df['sma_200']
    df['sma_diff_50_100'] = df['sma_50'] - df['sma_100']
    
    # Combine Momentum and Volume Indicators
    df['combined_factor'] = (
        0.3 * df['wma_returns'] +
        0.2 * df['sma_diff_100_200'] +
        0.2 * df['sma_diff_50_100'] +
        df['ema_atr_reversal']
    )
    
    # Incorporate into Combined Factor
    df['final_factor'] = (
        df['reversal_signal'] +
        df['wma_returns'] +
        df['adjusted_wma_returns'] +
        df['volume_weighted_price_change'] +
        df['combined_factor']
    )
    
    return df['final_factor']
```
