```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator using Close Prices
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['momentum_factor'] = df['10_day_SMA'] - df['20_day_SMA']
    
    # Volume-Weighted Price Movement Factor
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['10_day_EMA_VWAP'] = df['VWAP'].ewm(span=10, adjust=False).mean()
    df['50_day_EMA_VWAP'] = df['VWAP'].ewm(span=50, adjust=False).mean()
    df['VWAP_ratio'] = df['10_day_EMA_VWAP'] / df['50_day_EMA_VWAP']
    
    # Volatility Factor using High and Low Prices
    df['ATR'] = df['high'] - df['low']
    df['14_day_SMA_ATR'] = df['ATR'].rolling(window=14).mean()
    
    # Relative Strength Index (RSI) based on Close Prices
    df['price_change'] = df['close'].diff()
    df['gain'] = df['price_change'].apply(lambda x: max(x, 0))
    df['loss'] = df['price_change'].apply(lambda x: max(-x, 0))
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['RS'] = df['avg_gain'] / df['avg_loss']
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Money Flow Index (MFI) using High, Low, Close, and Volume
    df['TP'] = (df['high'] + df['low'] + df['close']) / 3
    df['MF'] = df['TP'] * df['volume']
    df['PMF'] = df.apply(lambda x: x['MF'] if x['TP'] > x.shift(1)['TP'] else 0, axis=1)
    df['NMF'] = df.apply(lambda x: x['MF'] if x['TP'] < x.shift(1)['TP'] else 0, axis=1)
    df['14_day_PMF'] = df['PMF'].rolling(window=14).sum()
    df['14_day_NMF'] = df['NMF'].rolling(window=14).sum()
    df['MR'] = df['14_day_PMF'] / df['14_day_NMF']
    df['MFI'] = 100 - (100 / (1 + df['MR']))
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['momentum_factor'] + df['VWAP_ratio'] + df['14_day_SMA_ATR'] + df['RSI'] + df['MFI']) / 5
    
    return df['alpha_factor']
```
