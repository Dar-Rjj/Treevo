```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Momentum
    high_to_low_change = df['high'] - df['low']
    open_to_close_change = df['close'] - df['open']
    
    # Integrate Volume Thrust
    daily_volume_increase = (df['volume'] / df['volume'].shift(1)) - 1
    
    weighted_high_to_low = high_to_low_change * daily_volume_increase
    weighted_open_to_close = open_to_close_change * daily_volume_increase
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    high_low_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Adjust by Volume and Amount
    avg_volume = df['volume'].rolling(window=n).mean()
    avg_amount = df['amount'].rolling(window=m).mean()
    
    normalized_volume = avg_volume / avg_amount
    
    # Combine High-Low Range Momentum and Adjusted Volume
    adjusted_momentum = high_low_range_momentum / normalized_volume
    
    # Combine all components
    alpha_factor = (weighted_high_to_low + weighted_open_to_close) * adjusted_momentum
    
    return alpha_factor
```
