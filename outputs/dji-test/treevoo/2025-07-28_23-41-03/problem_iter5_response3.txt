```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()

    # Subtract Longer SMA from Shorter SMA
    df['SMA_100_200'] = df['SMA_100'] - df['SMA_200']
    df['SMA_50_100'] = df['SMA_50'] - df['SMA_100']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Reversal Signal
    df['reversal_signal'] = df['intraday_return'].apply(lambda x: -1 if x > 0 else 1)

    # Combine Momentum and Reversal Signals
    df['combined_signal_100_200'] = df['SMA_100_200'] * df['reversal_signal']
    df['combined_signal_50_100'] = df['SMA_50_100'] * df['reversal_signal']

    # Compute Average True Range (ATR)
    df['TR'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['low'] - df['close'].shift(1)]].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Integrate ATR with Combined Signal
    df['ATR_combined_100_200'] = df['ATR'] * df['combined_signal_100_200']
    df['ATR_combined_50_100'] = df['ATR'] * df['combined_signal_50_100']

    # Smooth Result with Exponential Moving Average (EMA)
    df['smoothed_ATR_100_200'] = df['ATR_combined_100_200'].ewm(span=7, adjust=False).mean()
    df['smoothed_ATR_50_100'] = df['ATR_combined_50_100'].ewm(span=7, adjust=False).mean()

    # Calculate Price Momentum Over 5 Days
    df['price_momentum_5d'] = df['close'] - df['close'].shift(5)

    # Calculate Volume Surge Over 5 Days
    df['volume_surge_5d'] = df['volume'] - df['volume'].shift(5)

    # Combine Momentum and Volume Indicators
    df['weighted_price_momentum'] = df['price_momentum_5d'] * 0.5
    df['weighted_volume_surge'] = df['volume_surge_5d'] * 0.5

    # Integrate Weighted Indicators with SMAs
    df['integrated_indicators'] = (df['weighted_price_momentum'] 
                                   + df['weighted_volume_surge'] 
                                   + df['SMA_100_200'] 
                                   + df['SMA_50_100'])

    # Final Alpha Factor
    df['alpha_factor'] = (df['smoothed_ATR_100_200'] 
                          + df['smoothed_ATR_50_100'] 
                          + df['integrated_indicators'])

    return df['alpha_factor']
```
