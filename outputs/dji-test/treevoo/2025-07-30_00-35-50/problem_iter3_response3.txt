```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    close_t10 = df['close'].shift(10)
    price_momentum = df['close'] - close_t10
    
    # Adjust for Volume Trend
    moving_avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjustment = df['volume'] - moving_avg_volume
    
    # Adjust for Amount Trend
    moving_avg_amount = df['amount'].rolling(window=10).mean()
    amount_adjustment = df['amount'] - moving_avg_amount
    
    # Combine Price Momentum, Volume Adjustment, and Amount Adjustment
    factor = price_momentum * volume_adjustment * amount_adjustment
    factor[factor < 0] = 0
    
    return factor
```
