```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Confirm with Volume
    confirmed_with_volume = high_low_spread * (volume_change > 0) * volume_change
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Confirm with Price
    confirmed_with_price = confirmed_with_volume * (price_change > 0) * price_change
    
    # Apply Exponential Moving Average (EMA) with a period of 5 days
    ema_period = 5
    factor = confirmed_with_price.ewm(span=ema_period, adjust=False).mean()
    
    return factor
```
