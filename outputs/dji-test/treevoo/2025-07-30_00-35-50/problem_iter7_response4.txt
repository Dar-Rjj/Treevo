import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['high_low_range_prev'] = df['high_low_range'].shift(1)
    df['price_range_momentum'] = df['high_low_range'] / df['high_low_range_prev']
    
    # Calculate Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Price Change
    df['price_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['momentum_volume_adjusted'] = df['price_range_momentum'] * df['volume_change']
    df['final_alpha_factor'] = df['momentum_volume_adjusted'] + df['price_change']
    
    # Return the final alpha factor
    return df['final_alpha_factor'].dropna()
