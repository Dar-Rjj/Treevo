```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap']
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap']
    
    # Apply Exponential Decay to Adjustment (half-life = 1 day)
    df['Adjusted_Daily_Return'] = df['Adjusted_Daily_Return'].ewm(halflife=1).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].ewm(span=5).mean().diff()
    
    # Calculate High-to-Low Price Range Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Momentum'] = df['High_Low_Range'].diff()
    
    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['Volume_Change'] = df['volume'].diff()
    df['Amount_EMA'] = df['amount'].ewm(span=5).mean()
    df['Amount_Acceleration'] = df['Amount_EMA'].diff()
    
    df['Volume_Adjusted_High_Low'] = df['High_Low_Range'] * df['Volume_Change']
    df['Volume_Adjusted_High_Low'] = np.where(df['Volume_Change'] > 0, 
                                              df['Volume_Adjusted_High_Low'] * df['Amount_Acceleration'], 
                                              df['Volume_Adjusted_High_Low'])
    
    # Smoothing Filter
    df['Smoothed_Volume_Adjusted_High_Low'] = df['Volume_Adjusted_High_Low'].rolling(window=10).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['Integrated_Momentum'] = (df['Price_Momentum'] + df['High_Low_Range_Momentum']) * df['Smoothed_Volume_Adjusted_High_Low']
    
    # Evaluate Momentum Over Time
    df['Sum_Daily_Returns'] = df['Daily_Return'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = (df['Sum_Daily_Returns'] * df['volume']) / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    df['Volume_SMA'] = df['volume'].rolling(window=10).mean()
    volume_spike_threshold = 1.5
    df['Volume_Weighted_Momentum'] = np.where(df['volume'] > volume_spike_threshold * df['Volume_SMA'], 
                                              1.2 * df['Volume_Weighted_Momentum'], 
                                              df['Volume_Weighted_Momentum'])
    
    # Adjust for Volatility
    df['Historical_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    volatility_threshold = 2.0
    df['Volume_Weighted_Momentum'] = np.where(np.abs(df['close'].diff()) > volatility_threshold * df['Historical_Volatility'], 
                                              1.3 * df['Volume_Weighted_Momentum'], 
                                              df['Volume_Weighted_Momentum'])
    
    # Incorporate Price Confirmation
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Combined_Factor'] = np.where((df['Price_Change'] > 0) & (df['Volume_Change'] > 0), 
                                     df['High_Low_Range'] * (df['Price_Change'] + df['Volume_Change']), 
                                     df['High_Low_Range'])
    
    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Weighted Momentum
    df['Alpha_Factor'] = df['Adjusted_Daily_Return'] * df['Integrated_Momentum'] * df['Volume_Weighted_Momentum']
    
    # Ensure Positive Factor
    df['Alpha_Factor'] = np.where(df['Alpha_Factor'] < 0, 0, df['Alpha_Factor'])
    
    return df['Alpha_Factor']
```
