```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (10-day window)
    df['momentum_10d'] = df['price_change'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_10d'] * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    volume_ma_10d = df['volume'].rolling(window=10).mean()
    df['volume_spike_indicator'] = np.where(df['volume'] > 1.5 * volume_ma_10d, 1.2, 1.0)
    df['adjusted_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['volume_spike_indicator']
    
    # Adjust for Volatility
    df['historical_volatility'] = df['price_change'].rolling(window=10).std()
    df['volatility_adjustment'] = np.where(np.abs(df['price_change']) > 2 * df['historical_volatility'], 1.3, 1.0)
    df['final_volume_weighted_momentum'] = df['adjusted_volume_weighted_momentum'] * df['volatility_adjustment']
    
    # Adjust for Long-Term Price Momentum
    df['long_term_price_momentum'] = df['close'] - df['close'].shift(20)
    df['combined_momentum'] = df['long_term_price_momentum'] * df['final_volume_weighted_momentum']
    
    # Apply Softmax to Ensure Positive Factor
    df['combined_momentum'] = np.exp(df['combined_momentum']) / (1 + np.exp(df['combined_momentum']))
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['integrated_factor'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Calculate 5-day EMA for Close Price
    df['ema_close_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_close_5d'].diff()
    
    # Calculate 5-day EMA for Volume
    df['ema_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume_5d'].diff()
    
    # Combine Short-Term Factors
    df['short_term_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_factor']
    
    # Calculate 10-day EMA for Close Price
    df['ema_close_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_price_momentum_10d'] = df['ema_close_10d'].diff()
    
    # Calculate 10-day EMA for Volume
    df['ema_volume_10d'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Volume Acceleration
    df['long_term_volume_acceleration'] = df['ema_volume_10d'].diff()
    
    # Combine Long-Term Factors
    df['long_term_factor'] = df['long_term_volume_acceleration'] * df['long_term_price_momentum_10d']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['short_term_factor'] + df['long_term_factor']
    
    return df['alpha_factor']
```
