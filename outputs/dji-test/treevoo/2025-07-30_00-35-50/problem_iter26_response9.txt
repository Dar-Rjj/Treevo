```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] / df['close'].shift(1)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += abs(df['price_gap']) / df['close'].shift(1)

    # Evaluate Momentum Over Time (10-day window)
    df['momentum_10d'] = df['adjusted_daily_return'].rolling(window=10).sum()

    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_10d'] * df['volume'] / df['volume'].rolling(window=10).mean()

    # Adjust for Volume Spike Indicator
    df['volume_ma_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * 1.2 * (df['volume'] > 1.5 * df['volume_ma_10d'])

    # Adjust for Volatility
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * 1.3 * (abs(df['close'].diff()) > 2 * df['volatility_10d'])

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Integrate Price Range Momentum with Volume Change
    df['integrated_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']

    # Combine Adjusted Daily Return with Price Range Momentum
    df['combined_momentum'] = df['adjusted_daily_return'] * df['volume'] + df['integrated_price_range_momentum']

    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5d_price'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5d_price'] - df['ema_5d_price'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5d_volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5d_volume'] - df['ema_5d_volume'].shift(1)

    # Adjust for Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)

    # Adjust for Volume Trend
    df['volume_trend_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_adjustment'] = df['volume'] - df['volume_trend_20d']

    # Combine Factors
    final_alpha = (df['volume_acceleration'] * df['price_momentum'] +
                   df['combined_momentum'] +
                   df['long_term_momentum'] * df['volume_weighted_momentum'])
    final_alpha *= df['volume_trend_adjustment']

    # Ensure Positive Factor
    final_alpha = final_alpha.clip(lower=0)

    # Incorporate High and Low Prices
    df['intraday_range'] = df['high'] - df['low']
    df['normalized_intraday_range'] = df['intraday_range'] / df['close']

    # Adjust for Intraday Volatility
    df['intraday_volatility_10d'] = df['intraday_range'].rolling(window=10).std()
    final_alpha = final_alpha * 1.4 * (df['intraday_range'] > 2 * df['intraday_volatility_10d'])

    return final_alpha
```
