```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_to_low_range'] - df['high_to_low_range'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_ratio']
    
    # Combine and Smooth
    df['smoothed_alpha'] = df['integrated_momentum'].rolling(window=5).mean()
    
    # Output the Smoothed Alpha Factor
    return df['smoothed_alpha']
```
