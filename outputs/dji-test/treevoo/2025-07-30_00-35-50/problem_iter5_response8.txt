```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['Close_t-10'] = df['close'].shift(10)
    df['Price_Momentum'] = df['close'] - df['Close_t-10']

    # Adjust for Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Adjustment'] = df['volume'] - df['Volume_MA_10']

    # Adjust for Amount Trend
    df['Amount_MA_10'] = df['amount'].rolling(window=10).mean()
    df['Amount_Adjustment'] = df['amount'] - df['Amount_MA_10']

    # Combine Price Momentum, Volume Adjustment, and Amount Adjustment
    df['Combined_Factor'] = (df['Price_Momentum'] * df['Volume_Adjustment'] * df['Amount_Adjustment']).clip(lower=0)

    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Identify Positive and Negative Returns
    positive_return_days = df[df['Daily_Log_Return'] > 0]
    negative_return_days = df[df['Daily_Log_Return'] <= 0]

    # Calculate Sum of Upward Volume and Downward Volume
    df['Upward_Volume'] = positive_return_days['volume']
    df['Downward_Volume'] = negative_return_days['volume']

    # Compute Trend Reversal Signal
    total_volume = df['Upward_Volume'].sum() + df['Downward_Volume'].sum()
    upward_volume_ratio = df['Upward_Volume'].sum() / total_volume
    downward_volume_ratio = df['Downward_Volume'].sum() / total_volume

    df['Trend_Reversal_Signal'] = (upward_volume_ratio > downward_volume_ratio) * df['Daily_Log_Return']

    # Combine Price Momentum and Trend Reversal Signal
    df['Final_Factor'] = (df['Combined_Factor'] + df['Trend_Reversal_Signal']).clip(lower=0)

    return df['Final_Factor']
```
