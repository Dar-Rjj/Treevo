```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap with Exponential Decay
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            return row['daily_return'] - (row['price_gap'] * np.exp(-1))
        else:
            return row['daily_return'] + (row['price_gap'] * np.exp(-1))

    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)

    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum'] = df['adjusted_daily_return'].rolling(window=momentum_window).sum()

    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()

    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    df['volume_sma'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike_multiplier'] = np.where(df['volume'] > volume_threshold * df['volume_sma'], 1.2, 1.0)
    df['volume_weighted_momentum'] *= df['volume_spike_multiplier']

    # Adjust for Volatility
    volatility_window = 10
    df['volatility'] = df['adjusted_daily_return'].rolling(window=volatility_window).std()
    volatility_threshold = 2
    df['volatility_multiplier'] = np.where(np.abs(df['close'].pct_change()) > volatility_threshold * df['volatility'], 1.3, 1.0)
    df['volume_weighted_momentum'] *= df['volatility_multiplier']

    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] - df['price_range'].shift(1)

    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']

    # Combine Adjusted Daily Return with Integrated Momentum
    df['combined_momentum'] = df['adjusted_daily_return'] * df['volume'] + df['integrated_momentum']

    # Calculate 5-day Exponential Moving Average for Price
    ema_window = 5
    df['ema_close'] = df['close'].ewm(span=ema_window).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_close'] - df['ema_close'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_window).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume'] - df['ema_volume'].shift(1)

    # Combine Factors
    df['final_factor'] = (df['volume_acceleration'] * df['price_momentum'] + 
                           df['combined_momentum'] * df['volume_weighted_momentum'])

    # Ensure Positive Factor
    df['final_factor'] = np.where(df['final_factor'] < 0, np.exp(df['final_factor']) / (1 + np.exp(df['final_factor'])), df['final_factor'])

    return df['final_factor']
```
