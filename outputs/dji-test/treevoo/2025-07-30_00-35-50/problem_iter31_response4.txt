```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['price_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_daily_return'] = (df['close'] - df['close'].shift(1) - df['price_gap']) * np.exp(-1/1)
    
    # Calculate Adjusted High-Low Spread
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_high_low_spread'] = (df['high'] - df['low']) * df['volume_diff'].apply(lambda x: x if x > 0 else 0)
    
    # Apply Simple Moving Average (SMA) to Adjusted High-Low Spread
    df['sma_adjusted_high_low_spread'] = df['adjusted_high_low_spread'].rolling(window=5).mean()
    
    # Calculate 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_trend_adjustment'] = df['volume'] - df['10_day_avg_volume']
    
    # Combine Price Momentum and Volume Trend Adjustment
    df['combined_price_volume'] = df['10_day_price_momentum'] * df['volume_trend_adjustment']
    df['combined_price_volume'] = df['combined_price_volume'].apply(lambda x: max(x, 0))
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_returns = df['daily_log_return'] > 0
    negative_returns = df['daily_log_return'] <= 0
    
    # Calculate Sum of Upward and Downward Volumes
    df['upward_volume'] = df['volume'] * positive_returns
    df['downward_volume'] = df['volume'] * negative_returns
    total_volume = df['volume'].sum()
    upward_volume_sum = df['upward_volume'].sum()
    downward_volume_sum = df['downward_volume'].sum()
    
    # Compute Trend Reversal Signal
    upward_volume_ratio = upward_volume_sum / total_volume
    downward_volume_ratio = downward_volume_sum / total_volume
    trend_reversal_signal = (upward_volume_ratio > downward_volume_ratio) * df['daily_log_return']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_daily_return'] + 
        df['sma_adjusted_high_low_spread'] + 
        df['combined_price_volume'] + 
        trend_reversal_signal
    )
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['final_alpha_factor']
```
