import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple daily return
    df['simple_return'] = df['close'].pct_change()
    
    # Logarithmic daily return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Volume change percentage
    df['volume_change'] = df['volume'].pct_change()
    
    # Volume surprise factor
    df['volume_surprise'] = df['volume'] - df['volume'].rolling(window=20).mean().shift(1)
    
    # True range
    df['true_range'] = (df[['high', 'low']].diff(axis=1)['high'] - df[['high', 'low']].diff(axis=1)['low']).abs().max(axis=1)
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Average true range
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Accumulation/Distribution Line (ADL)
    df['adl'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['adl'] = df['adl'].cumsum()
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (positive_money_flow_sum / negative_money_flow_sum)))
    
    # Rate of Change (ROC) for Close prices
    df['roc_25'] = df['close'].pct_change(25)
    
    # On-balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Daily traded amount change
    df['amount_change'] = df['amount'].pct_change()
    
    # Ratio of traded amount to volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Historical Volatility
    df['historical_volatility'] = df['log_return'].rolling(window=21).std() * np.sqrt(252)
    
    # Weighted combination of simple daily return, logarithmic daily return, and ROC
    df['composite_alpha_signal'] = 0.4 * df['simple_return'] + 0.3 * df['log_return'] + 0.3 * df['roc_25']
    
    # Composite volume-based signal
    df['composite_volume_signal'] = 0.5 * df['volume_change'] + 0.5 * df['volume_surprise']
    
    # Composite technical indicator
    df['composite_tech_indicator'] = 0.3 * df['adl'] + 0.3 * df['mfi'] + 0.4 * df['obv']
    
    # Lagged simple daily return
    df['lagged_simple_return'] = df['simple_return'].shift(1)
    
    # Lead-lag relationship between volume and price
    df['volume_price_correlation'] = df['close'].rolling(window=20).corr(df['volume'].shift(1))
    
    # Tick rule
    df['tick_rule'] = np.sign(df['close'] - df['close'].shift(1)) * df['volume']
    
    # Intraday return
    df['intraday_return'] = (df['close'] / df['open']) - 1
