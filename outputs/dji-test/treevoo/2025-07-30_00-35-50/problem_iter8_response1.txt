import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] - df['price_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_factor'] = (df['price_range_momentum'] * df['volume_change']) + df['price_change']
    
    # Confirm with Volume
    df['confirmed_factor'] = df['integrated_factor'] * df['volume_change']
    df['confirmed_factor'] = df['confirmed_factor'].where(df['volume_change'] > 0, 0)
    
    # Smoothing Filter
    df['final_alpha_factor'] = df['confirmed_factor'].rolling(window=5).mean()
    
    return df['final_alpha_factor'].dropna()
