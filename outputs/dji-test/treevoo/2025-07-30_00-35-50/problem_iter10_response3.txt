```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    window = 10
    df['momentum'] = df['price_change'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=window).mean()
    
    # Integrate Volume Spike Indicator
    volume_threshold = 1.5
    multiplier_vol_spike = 1.2
    avg_volume = df['volume'].rolling(window=window).mean()
    df['volume_spike'] = (df['volume'] > volume_threshold * avg_volume).astype(int)
    df['volume_weighted_momentum'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * multiplier_vol_spike if row['volume_spike'] == 1 else row['volume_weighted_momentum'],
        axis=1
    )
    
    # Adjust for Volatility
    historical_volatility = df['price_change'].rolling(window=window).std()
    volatility_threshold = 2.0
    multiplier_vol_adjust = 1.3
    df['volatility_adjusted_factor'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * multiplier_vol_adjust if abs(row['price_change']) > volatility_threshold * historical_volatility.loc[row.name] else row['volume_weighted_momentum'],
        axis=1
    )
    
    return df['volatility_adjusted_factor']
```
