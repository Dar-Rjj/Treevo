import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day rolling standard deviation of the closing price
    df['std_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day rolling standard deviation of the closing price
    df['std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] / df['close'].shift(30) - 1
    
    # Calculate the mean-reversion factor
    df['mean_reversion'] = (df['close'] - df['ma_30']) / df['std_30']
    
    # Combine the factors to create the final heuristic
    df['heuristic'] = df['momentum'] + df['mean_reversion'] + (df['vwap'] - df['ma_10']) / df['std_10']
    
    return df['heuristic'].dropna()
