import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume' and index as date.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average true range over the last 14 days
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over the last 14 days
    df['gain'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['daily_return'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rsi'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))
    
    # Calculate the factor as a combination of RSI and ATR
    df['factor'] = (df['rsi'] / 100) * (df['avg_true_range'] / df['close'])
    
    return df['factor']
