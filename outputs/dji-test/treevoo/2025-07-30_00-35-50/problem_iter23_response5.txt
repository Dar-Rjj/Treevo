```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap'] / df['close'].shift(1)
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap'] / df['close'].shift(1)
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].ewm(span=5).mean().diff()
    
    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Price Range Momentum
    df['High_Low_Range_Momentum'] = df['High_Low_Range'].diff()
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].diff()
    
    # Calculate Amount Acceleration
    df['Amount_EMA'] = df['amount'].ewm(span=5).mean()
    df['Amount_Acceleration'] = df['Amount_EMA'].diff()
    
    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['Combined_High_Low_Spread'] = df['High_Low_Range'] * df['Volume_Change']
    df['Combined_High_Low_Spread'] = np.where(df['Volume_Change'] > 0, df['Combined_High_Low_Spread'], 0)
    df['Adjusted_High_Low_Spread'] = df['Combined_High_Low_Spread'] * df['Amount_Acceleration']
    
    # Smoothing Filter
    df['Smoothed_Adjusted_High_Low_Spread'] = df['Adjusted_High_Low_Spread'].rolling(window=10).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['Integrated_Momentum'] = (df['Price_Momentum'] + df['High_Low_Range_Momentum']) * df['Smoothed_Adjusted_High_Low_Spread']
    
    # Adjust by Volume Change
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Change
    df['Combined_Factor'] = df['Adjusted_Daily_Return'] * df['Integrated_Momentum'] * df['Volume_Ratio']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_returns = df[df['Daily_Log_Return'] > 0]
    negative_returns = df[df['Daily_Log_Return'] <= 0]
    
    # Calculate Sum of Upward Volume
    sum_upward_volume = positive_returns['volume'].sum()
    
    # Calculate Sum of Downward Volume
    sum_downward_volume = negative_returns['volume'].sum()
    
    # Compute Trend Reversal Signal
    total_volume = sum_upward_volume + sum_downward_volume
    upward_ratio = sum_upward_volume / total_volume
    downward_ratio = sum_downward_volume / total_volume
    trend_reversal_signal = np.where(upward_ratio > downward_ratio, df['Daily_Log_Return'], -df['Daily_Log_Return'])
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + trend_reversal_signal
    df['Final_Alpha_Factor'] = np.where(df['Final_Alpha_Factor'] < 0, 0, df['Final_Alpha_Factor'])
    
    return df['Final_Alpha_Factor'].dropna()
```
