```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] / df['close'].shift(1)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] / df['close'].shift(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum_sum'] = df['adjusted_daily_return'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_sum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    df['volume_moving_avg'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = df['volume'] / df['volume_moving_avg']
    df.loc[df['volume_spike'] > volume_spike_threshold, 'volume_weighted_momentum'] *= 1.2
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'].diff()
    df['high_low_spread_adjusted'] = df['high_low_spread']
    df.loc[df['volume_change'] > 0, 'high_low_spread_adjusted'] *= df['volume_change'] / df['volume_moving_avg']
    
    # Incorporate Price Confirmation
    df['price_change'] = df['close'].diff()
    df['final_factor'] = df['high_low_spread_adjusted']
    df.loc[(df['price_change'] > 0) & (df['volume_change'] > 0), 'final_factor'] *= (df['price_change'] + df['volume_change'])
    
    return df['final_factor'].dropna()
```
