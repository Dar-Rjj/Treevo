```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is a combination of intraday price movement, volume, and volatility.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the intraday return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the 5-day rolling average of volume
    rolling_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day rolling standard deviation of the close price
    rolling_std_close = df['close'].rolling(window=5).std()
    
    # Calculate the alpha factor
    alpha_factor = (intraday_return * rolling_avg_volume) / (intraday_range + rolling_std_close)
    
    # Normalize the factor to avoid extreme values
    alpha_factor = (alpha_factor - alpha_factor.mean()) / alpha_factor.std()
    
    return alpha_factor
```
