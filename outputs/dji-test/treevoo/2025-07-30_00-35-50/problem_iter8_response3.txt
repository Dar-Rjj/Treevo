```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum_sum'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_sum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    volume_moving_avg = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike_indicator'] = np.where(df['volume'] > volume_spike_threshold * volume_moving_avg, 1.2, 1.0)
    df['volume_weighted_momentum'] *= df['volume_spike_indicator']
    
    # Adjust for Volatility
    volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=volatility_window).std()
    volatility_threshold = 2.0
    df['volatility_adjustment'] = np.where(abs(df['daily_price_change']) > volatility_threshold * df['historical_volatility'], 1.3, 1.0)
    df['volume_weighted_momentum'] *= df['volatility_adjustment']
    
    # Adjust for Long-Term Price Momentum
    long_term_momentum_days = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_momentum_days)
    
    # Adjust for Volume Trend
    volume_trend_days = 20
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=volume_trend_days).mean()
    
    # Adjust for Amount Trend
    amount_trend_days = 20
    df['amount_trend'] = df['amount'] - df['amount'].rolling(window=amount_trend_days).mean()
    
    # Combine All Adjustments
    df['combined_adjustment'] = df['long_term_momentum'] * df['volume_weighted_momentum'] * df['volume_trend'] * df['amount_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['factor'] = np.where(df['combined_adjustment'] < 0, 
                            np.exp(df['combined_adjustment']) / (1 + np.exp(df['combined_adjustment'])), 
                            df['combined_adjustment'])
    
    return df['factor']
```
