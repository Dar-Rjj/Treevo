```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day moving average of close price
    df['close_5day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Close Price Momentum
    df['close_momentum'] = df['close_5day_ma'] - df['close_5day_ma'].shift(1)
    
    # Calculate High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate 5-day moving average of volume
    df['volume_5day_ma'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume_5day_ma'] - df['volume_5day_ma'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Confirm High-Low Spread with Volume Change
    df['confirmed_high_low_spread'] = df['high_low_spread'] * df['volume_change']
    
    # Calculate 10-day Moving Average Volume
    df['volume_10day_ma'] = df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume_10day_ma']
    
    # Combine Close Price Momentum, High-Low Spread, and Volume Dynamics
    df['combined_alpha'] = (df['close_momentum'] * df['volume_acceleration']) + \
                            (df['confirmed_high_low_spread'] * df['volume_trend'])
    
    # Apply Simple Moving Average (SMA) to the combined alpha factor
    df['final_alpha'] = df['combined_alpha'].rolling(window=5).mean()
    
    # Ensure Positive Factor
    df['final_alpha'] = df['final_alpha'].apply(lambda x: max(x, 0))
    
    return df['final_alpha']
```
