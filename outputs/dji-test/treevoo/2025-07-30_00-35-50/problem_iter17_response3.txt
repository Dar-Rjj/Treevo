```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = row['price_gap']
        else:
            adjustment = -row['price_gap']
        return row['daily_return'] - adjustment * np.exp(-1)  # Exponential decay with half-life = 1 day
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-Weighted Price Change
    df['volume_weighted_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['integrated_factor'] = df['price_range_momentum'] * df['volume_change'] + df['volume_weighted_price_change']
    
    # Combine Integrated Factor with Volume Change
    df['combined_integrated_factor'] = df['integrated_factor'] * df['volume_change']
    df['combined_integrated_factor'] = df['combined_integrated_factor'].where(df['volume_change'] > 0, 0)
    
    # Smoothing Filter
    df['sma_combined_integrated_factor'] = df['combined_integrated_factor'].rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_close_5'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_close_5'] - df['ema_close_5'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_volume_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume_5'] - df['ema_volume_5'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_factor'] + df['close_price_momentum']
    
    # Final Alpha Factor with Volume Weighting
    total_volume = df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['combined_factors'] / total_volume
    df['final_alpha_factor'] = df['final_alpha_factor'].where(df['final_alpha_factor'] > 0, 0)
    
    return df['final_alpha_factor']
```
