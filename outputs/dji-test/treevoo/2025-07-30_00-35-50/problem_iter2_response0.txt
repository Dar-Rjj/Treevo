```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close Price Momentum
    close_5day_ma = df['close'].rolling(window=5).mean()
    close_price_momentum = close_5day_ma - close_5day_ma.shift(1)
    
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Acceleration
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_acceleration = volume_5day_ma - volume_5day_ma.shift(1)
    
    # Confirm High-Low Spread with Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_high_low_spread = high_low_spread * volume_change
    
    # Combine Close Price Momentum and High-Low Spread with Volume Changes
    combined_alpha_factor = (close_price_momentum * volume_acceleration + 
                             confirmed_high_low_spread)
    
    # Apply Simple Moving Average (SMA) to the Combined Alpha Factor
    smoothed_alpha_factor = combined_alpha_factor.rolling(window=5).mean()
    
    return smoothed_alpha_factor
```
