```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Combine High-Low Spread with Volume Change if Volume Change is positive
    combined = high_low_spread * (volume_change > 0) * volume_change
    
    # Apply Simple Moving Average (SMA) with a period of 5 days
    sma_5 = combined.rolling(window=5).mean()
    
    return sma_5
```
