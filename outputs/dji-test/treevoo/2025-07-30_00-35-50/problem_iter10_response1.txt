```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    previous_high_to_low_range = high_to_low_range.shift(1)
    price_range_momentum = (high_to_low_range / previous_high_to_low_range) - 1
    
    # Adjust by Volume and Price Change
    current_volume = df['volume']
    previous_volume = current_volume.shift(1)
    volume_change = (current_volume - previous_volume) / previous_volume
    
    current_close = df['close']
    previous_close = current_close.shift(1)
    price_change = current_close - previous_close
    
    # Integrate Momentum, Volume, and Price Change
    factor = (price_range_momentum * volume_change) + price_change
    
    return factor
```
