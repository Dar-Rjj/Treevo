```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = -row['price_gap']
        else:
            adjustment = row['price_gap']
        return row['daily_return'] + adjustment * np.exp(-np.log(2))
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_daily_return'] * df['volume']
    
    return df['volume_weighted_momentum']
```
