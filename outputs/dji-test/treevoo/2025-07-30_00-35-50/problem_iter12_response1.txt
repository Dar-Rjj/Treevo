import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df.apply(
        lambda row: row['daily_return'] - row['price_gap'] if row['price_gap'] > 0 else row['daily_return'] + row['price_gap'],
        axis=1
    )
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'].pct_change()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Integrate Adjusted Daily Return and Price Range Momentum
    df['integrated_alpha'] = df['adjusted_daily_return'] + df['adjusted_price_range_momentum']
    
    # Calculate Trend Continuity
    df['trend_continuity'] = df['daily_return'].shift(1) - df['daily_return'].shift(2)
    
    # Combine Trend Continuity with Integrated Alpha
    df['final_alpha_factor'] = df['integrated_alpha'] * df['trend_continuity']
    
    return df['final_alpha_factor'].dropna()
