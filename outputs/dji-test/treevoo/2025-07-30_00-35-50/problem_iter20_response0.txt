import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the relative volume change
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Calculate the range of the day
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the average true range over 14 days
    df['atr_14'] = df['daily_range'].rolling(window=14).mean()
    
    # Calculate the normalized daily range
    df['norm_daily_range'] = df['daily_range'] / df['atr_14']
    
    # Calculate the momentum factor
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the volatility factor
    df['volatility_20'] = df['close'].rolling(window=20).std()
    
    # Combine factors to create the final alpha factor
    df['alpha_factor'] = (
        df['close_pct_change'] * 0.3 +
        df['volume_pct_change'] * 0.2 +
        df['norm_daily_range'] * 0.2 +
        df['momentum_10'] * 0.15 +
        df['volatility_20'] * 0.15
    )
    
    return df['alpha_factor']
