import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term (5-day) and long-term (20-day) moving averages of close prices
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    # Subtract the 5-day moving average from the 20-day moving average to capture short-term versus long-term trends
    df['ma_diff'] = df['20_day_ma'] - df['5_day_ma']
    
    # Compute the ratio of today's close price to the average close price over a past period (e.g., 10 days)
    df['avg_close_10_day'] = df['close'].rolling(window=10).mean()
    df['price_strength_ratio'] = df['close'] / df['avg_close_10_day']
    
    # Measure the volatility using the standard deviation of log returns over a specified period (e.g., 10 days)
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Evaluate the accumulation/distribution line by considering the daily volume and the position of the closing price within the day's range
    df['money_flow_multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    df['acc_dist_line'] = df['money_flow_volume'].rolling(window=30).sum()
    
    # Investigate the change in trade volume over different time horizons (e.g., comparing current day to 1-week, 1-month ago)
    df['volume_change_1w'] = (df['volume'] - df['volume'].shift(7)) / df['volume'].shift(7)
    df['volume_change_1m'] = (df['volume'] - df['volume'].shift(21)) / df['volume'].shift(21)
    
    # Create a composite signal by combining elements from the above factors
    df['composite_signal'] = (df['ma_diff'] + df['price_strength_ratio'] - df['volatility'] + df['acc_dist_line'] + df['volume_change_1w'] + df['volume_change_1m']) / 6
    
    return df['composite_signal']
