```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    df['momentum_10d'] = df['price_change'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_10d'] * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    df['volume_ma_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_spike_multiplier'] = np.where(df['volume'] > 1.5 * df['volume_ma_10d'], 1.2, 1.0)
    df['adjusted_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['volume_spike_multiplier']
    
    # Adjust for Volatility
    df['volatility_10d'] = df['price_change'].rolling(window=10).std()
    df['volatility_multiplier'] = np.where(np.abs(df['price_change']) > 2 * df['volatility_10d'], 1.3, 1.0)
    df['adjusted_volume_weighted_momentum'] = df['adjusted_volume_weighted_momentum'] * df['volatility_multiplier']
    
    # Adjust for Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume_ma_10d']
    
    # Combine All Adjustments
    df['combined_factor'] = df['long_term_momentum'] * df['adjusted_volume_weighted_momentum'] * df['volume_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['alpha_factor'] = np.exp(df['combined_factor']) / (1 + np.exp(df['combined_factor']))
    
    return df['alpha_factor']
```
