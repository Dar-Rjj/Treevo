import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time (10-day window)
    df['momentum_10d'] = df['price_change'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum_10d'] * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    volume_multiplier = 1.2
    df['volume_spike'] = (df['volume'] > volume_threshold * df['volume'].rolling(window=10).mean()).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + (volume_multiplier - 1) * df['volume_spike'])
    
    # Adjust for Volatility
    volatility_threshold = 2.0
    volatility_multiplier = 1.3
    df['historical_volatility'] = df['price_change'].rolling(window=10).std()
    df['volatility_adjustment'] = (np.abs(df['price_change']) > volatility_threshold * df['historical_volatility']).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + (volatility_multiplier - 1) * df['volatility_adjustment'])
    
    # Adjust for Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['long_term_momentum']
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['volume_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['volume_weighted_momentum'] = np.exp(df['volume_weighted_momentum']) / (1 + np.exp(df['volume_weighted_momentum']))
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    df['price_range_momentum'] = df['price_range'] - df['price_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Calculate 5-day EMA for Close Price
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Short-Term Factors
    df['short_term_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_momentum']
    
    # Calculate 10-day EMA for Close Price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['long_term_price_momentum'] = df['ema_10_close'] - df['ema_10_close'].shift(1)
    
    # Calculate 10-day EMA for Volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['long_term_volume_acceleration'] = df['ema_10_volume'] - df['ema_10_volume'].shift(1)
    
    # Combine Long-Term Factors
    df['long_term_factor'] = df['long_term_volume_acceleration'] * df['long_term_price_momentum'] + df['short_term_factor']
    
    # Incorporate Open Price into the Factor
    df['open_price_momentum'] = df['open'] - df['open'].shift(1)
    df['alpha_factor'] = df['long_term_factor'] + df['open_price_momentum']
    
    return df['alpha_factor']
