import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_low_range / high_low_range.shift(1)
    
    # Confirm High-Low Spread with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    combined_high_low_spread = high_low_range * volume_change
    combined_high_low_spread = combined_high_low_spread.where(volume_change > 0, 0)
    
    # Apply Simple Moving Average (SMA) for smoothing
    smoothed_combined_high_low_spread = combined_high_low_spread.rolling(window=5).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    final_alpha_factor = price_range_momentum * smoothed_combined_high_low_spread
    
    return final_alpha_factor
