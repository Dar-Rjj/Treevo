```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day Exponential Moving Average (EMA) of Close price
    df['close_ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close_ema_5'] - df['close_ema_5'].shift(1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Price Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Amount Acceleration
    df['amount_ema_5'] = df['amount'].ewm(span=5, adjust=False).mean()
    df['amount_acceleration'] = df['amount_ema_5'] - df['amount_ema_5'].shift(1)
    
    # Multiply High-Low Spread by Volume Change and ensure positive volume change
    df['high_low_volume'] = df['high_low_range'] * df['volume_change']
    df['high_low_volume'] = df['high_low_volume'] * (df['volume_change'] > 0)
    
    # Adjust with Amount Acceleration
    df['adjusted_high_low_volume'] = df['high_low_volume'] * df['amount_acceleration']
    
    # Apply Simple Moving Average (SMA) to the adjusted high-low spread
    df['sma_adjusted_high_low_volume'] = df['adjusted_high_low_volume'].rolling(window=5).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['integrated_momentum'] = (df['price_momentum'] + df['range_momentum']) * df['sma_adjusted_high_low_volume']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    decay_factor = 0.5
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * decay_factor
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * decay_factor
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate 10-day Moving Average Volume
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['integrated_close_momentum'] = df['close_price_momentum'] * (df['volume'] - df['volume_ma_10'])
    
    # Combine Volume Weighted and Price Range Momentum
    df['combined_momentum'] = df['integrated_momentum'] + df['integrated_close_momentum']
    
    # Final Alpha Factor Integration
    df['last_5_days_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['alpha_factor'] = (df['combined_momentum'] * df['adjusted_daily_return'] * df['volume']) / df['last_5_days_volume_sum']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
```
