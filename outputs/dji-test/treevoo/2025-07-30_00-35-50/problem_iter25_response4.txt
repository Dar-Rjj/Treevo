```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] / df['close'].shift(1)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] / df['close'].shift(1)

    # Evaluate Momentum Over Time (10-day window)
    momentum_window = 10
    df['sum_daily_returns'] = df['adjusted_daily_return'].rolling(window=momentum_window).sum()

    # Compute Volume Weighted Momentum
    df['average_volume'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_weighted_momentum'] = (df['sum_daily_returns'] * df['volume']) / df['average_volume']

    # Adjust for Volume Spike Indicator
    df['volume_ma'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike_indicator'] = 1.0
    df.loc[df['volume'] > 1.5 * df['volume_ma'], 'volume_spike_indicator'] = 1.2
    df['volume_weighted_momentum'] *= df['volume_spike_indicator']

    # Adjust for Volatility
    df['historical_volatility'] = df['daily_return'].rolling(window=momentum_window).std()
    df['absolute_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['volatility_adjustment'] = 1.0
    df.loc[df['absolute_price_change'] > 2 * df['historical_volatility'], 'volatility_adjustment'] = 1.3
    df['volume_weighted_momentum'] *= df['volatility_adjustment']

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Integrate Price Range Momentum with Volume Change
    df['integrated_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']

    # Combine Adjusted Daily Return with Price Range Momentum
    df['combined_factor'] = (df['adjusted_daily_return'] * df['volume']) + df['integrated_price_range_momentum']

    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)

    # Combine Factors
    df['final_factor'] = (df['volume_acceleration'] * df['price_momentum']) + df['combined_factor'] + (df['volume_weighted_momentum'] * df['price_momentum'])

    # Ensure Positive Factor
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))

    return df['final_factor']
```
