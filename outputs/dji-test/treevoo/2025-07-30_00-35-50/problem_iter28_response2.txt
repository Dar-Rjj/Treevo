```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (10-day window)
    momentum_window = 10
    df['momentum'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    multiplier_volume_spike = 1.2
    df['moving_avg_volume'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = (df['volume'] > volume_spike_threshold * df['moving_avg_volume']).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + (df['volume_spike'] * (multiplier_volume_spike - 1)))
    
    # Adjust for Volatility
    volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=volatility_window).std()
    volatility_threshold = 2
    multiplier_volatility = 1.3
    df['volatility_adjustment'] = (np.abs(df['daily_price_change']) > volatility_threshold * df['historical_volatility']).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + (df['volatility_adjustment'] * (multiplier_volatility - 1)))
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_window)
    df['factor'] = df['long_term_momentum'] * df['volume_weighted_momentum']
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    df['factor'] = df['factor'] * df['volume_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['factor'] = np.exp(df['factor']) / (1 + np.exp(df['factor']))
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    df['prev_price_range'] = df['price_range'].shift(1)
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['prev_price_range']
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'].pct_change()
    df['price_change'] = df['close'].pct_change()
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Combine with Factor
    df['factor'] += df['integrated_momentum']
    
    # Calculate 5-day EMA for Close Price
    ema_window = 5
    df['ema_close'] = df['close'].ewm(span=ema_window, adjust=False).mean()
    df['previous_ema_close'] = df['ema_close'].shift(1)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_close'] - df['previous_ema_close']
    
    # Calculate 5-day EMA for Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_window, adjust=False).mean()
    df['previous_ema_volume'] = df['ema_volume'].shift(1)
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume'] - df['previous_ema_volume']
    
    # Combine Short-Term Factors
    df['short_term_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_momentum']
    
    # Final Alpha Factor
    df['open_momentum'] = df['open'].diff()
    df['final_alpha_factor'] = df['short_term_factor'] + df['open_momentum']
    
    return df['final_alpha_factor']
```
