```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df.apply(
        lambda row: row['Daily_Return'] - row['Price_Gap'] if row['Price_Gap'] > 0 else row['Daily_Return'] + row['Price_Gap'],
        axis=1
    )
    
    # Evaluate Momentum Over Time (Sum of Daily Returns over a Window, e.g., 10 days)
    df['Momentum'] = df['Adjusted_Daily_Return'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Momentum'] * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    avg_volume = df['volume'].rolling(window=10).mean()
    df['Volume_Spike'] = df.apply(
        lambda row: row['Volume_Weighted_Momentum'] * 1.2 if row['volume'] > 1.5 * avg_volume.loc[row.name] else row['Volume_Weighted_Momentum'],
        axis=1
    )
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust High-Low Spread with Volume
    df['Adjusted_High_Low_Spread'] = df.apply(
        lambda row: row['High_Low_Spread'] * (row['Volume_Change'] / avg_volume.loc[row.name]) if row['Volume_Change'] > 0 else row['High_Low_Spread'],
        axis=1
    )
    
    # Incorporate Price Confirmation
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Spread, Volume, and Price
    df['Combined_Factor'] = df.apply(
        lambda row: row['Adjusted_High_Low_Spread'] * (row['Price_Change'] + row['Volume_Change']) if row['Price_Change'] > 0 and row['Volume_Change'] > 0 else row['Adjusted_High_Low_Spread'],
        axis=1
    )
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Volume_Spike'] + df['Combined_Factor']
    
    return df['Alpha_Factor']
```
