```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = np.where(
        df['Price_Gap'] > 0,
        (df['Daily_Return'] - df['Price_Gap']) * np.exp(-1),
        (df['Daily_Return'] + df['Price_Gap']) * np.exp(-1)
    )
    
    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['Price_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust by Volume and Volume-Weighted Price Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Weighted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['Integrated_Factor'] = (df['Price_Range_Momentum'] * df['Volume_Change']) + df['Volume_Weighted_Price_Change']
    
    # Combine Integrated Factor with Volume Change
    df['Combined_Factor'] = df['Integrated_Factor'] * np.where(df['Volume_Change'] > 0, df['Volume_Change'], 0)
    
    # Smoothing Filter
    df['Smoothed_Combined_Factor'] = df['Combined_Factor'].rolling(window=5).mean()
    
    # Calculate 5-day EMA for Price
    df['EMA_5_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_5_Close'] - df['EMA_5_Close'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['EMA_5_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_5_Volume'] - df['EMA_5_Volume'].shift(1)
    
    # Combine Factors
    df['Final_Factor'] = (df['Volume_Acceleration'] * df['Price_Momentum']) + df['Smoothed_Combined_Factor']
    
    # Final Alpha Factor with Volume Weighting
    df['Alpha_Factor'] = df['Final_Factor'] / df['volume'].rolling(window=5).sum()
    df['Alpha_Factor'] = np.where(df['Alpha_Factor'] < 0, 0, df['Alpha_Factor'])
    
    return df['Alpha_Factor'].dropna()
```
