```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap']
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += abs(df['price_gap'])
    
    # Calculate Price Momentum
    df['ema_close_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['price_momentum'] = df['ema_close_5'] - df['ema_close_5'].shift(1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Amount Acceleration
    df['ema_amount_5'] = df['amount'].ewm(span=5, adjust=False).mean()
    df['amount_acceleration'] = df['ema_amount_5'] - df['ema_amount_5'].shift(1)
    
    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['high_low_spread_vol_adj'] = (df['high_low_momentum'] * df['volume_change']) * df['amount_acceleration']
    
    # Apply Simple Moving Average (SMA) for smoothing
    df['sma_high_low_spread_vol_adj'] = df['high_low_spread_vol_adj'].rolling(window=5).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['integrated_momentum'] = df['price_momentum'] * df['sma_high_low_spread_vol_adj']
    
    # Adjust by Volume Change
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Change
    df['factor'] = df['adjusted_daily_return'] * df['integrated_momentum'] * df['volume_change_ratio']
    
    # Ensure Positive Factor
    df['factor'] = df['factor'].apply(lambda x: max(x, 0))
    
    return df['factor']
```
