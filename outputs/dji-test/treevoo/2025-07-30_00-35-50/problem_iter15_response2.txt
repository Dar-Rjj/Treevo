import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range - high_to_low_range.shift(1)
    
    # Adjust by Volume and Price Change
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    integrated_factor = (price_range_momentum * volume_change) + price_change
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    combined_factor = integrated_factor / volume_change
    combined_factor = combined_factor.where(volume_change > 0, 0)  # Ensure positive volume change
    
    # Smoothing Filter
    final_alpha_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
