```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    decay_factor = 0.5  # half-life of 1 day
    df['adjusted_daily_return'] = df['daily_return']
    df['adjusted_daily_return'] = np.where(
        df['price_gap'] > 0,
        df['daily_return'] - (df['price_gap'] * np.exp(-decay_factor)),
        df['daily_return'] + (df['price_gap'] * np.exp(-decay_factor))
    )
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['volume_weighted_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume-Weighted and Price Range Momentum
    df['volume_weighted_adjusted_return'] = df['adjusted_daily_return'] * df['volume']
    
    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = df['volume_weighted_adjusted_return'] / df['volume'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Price Range Momentum
    df['volume_weighted_price_range_momentum'] = df['price_range_momentum'] * df['volume']
    total_volume = df['volume'] + df['volume'].shift(1)
    df['normalized_volume_weighted_price_range_momentum'] = df['volume_weighted_price_range_momentum'] / total_volume
    
    # Integrate Volume-Weighted Price Range Momentum and Adjusted Daily Return
    df['final_alpha_factor'] = df['final_alpha_factor'] + df['normalized_volume_weighted_price_range_momentum']
    
    return df['final_alpha_factor'].dropna()
```
