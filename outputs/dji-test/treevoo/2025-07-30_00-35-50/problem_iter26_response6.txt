```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided DataFrame.
    The factor is designed to help predict future stock returns using only current and past data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the average volume over the last 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 20-day exponential moving average of the daily return
    ema_20d = daily_return.ewm(span=20, adjust=False).mean()
    
    # Calculate the standard deviation of the daily return over the last 10 days
    std_dev_10d = daily_return.rolling(window=10).std()
    
    # Calculate the factor as a combination of these metrics
    factor = (typical_price * avg_volume_10d * ema_20d) / (std_dev_10d + 1e-6)
    
    return factor
```
