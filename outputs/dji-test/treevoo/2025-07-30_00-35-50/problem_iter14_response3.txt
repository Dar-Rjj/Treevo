```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'].apply(lambda x: x * np.exp(-np.log(2) * 1))
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'].apply(lambda x: x * np.exp(-np.log(2) * 1))

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].diff()

    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'].diff(periods=10)

    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()

    # Integrate Momentum and Volume Change
    df['integrated_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    df['integrated_close_momentum'] = df['close_price_momentum'] * (df['volume'] - df['volume'].rolling(window=10).mean())

    # Combine Volume Weighted and Price Range Momentum
    df['volume_weighted_return'] = df['adjusted_daily_return'] * df['volume']

    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_close'].diff()

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'].diff()

    # Combine Factors
    df['combined_factor'] = (df['volume_acceleration'] * df['price_momentum']) + \
                             (df['integrated_range_momentum'] + df['integrated_close_momentum'])

    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = df['combined_factor'] / df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))

    return df['final_alpha_factor'].dropna()
```
