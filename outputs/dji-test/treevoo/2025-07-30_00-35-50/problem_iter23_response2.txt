```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the close-to-open return
    close_to_open_return = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the 10-day moving average of the VWAP
    vwap_10_day_ma = vwap.rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the close-to-open return
    close_to_open_volatility = close_to_open_return.rolling(window=10).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (vwap / vwap_10_day_ma) * (intraday_range / df['close']) * (1 / close_to_open_volatility)
    
    return alpha_factor
```
