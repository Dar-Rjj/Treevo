```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time
    window = 10
    df['momentum_sum'] = df['daily_price_change'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum_sum'] * df['volume']) / df['volume'].rolling(window=window).mean()
    
    # Integrate Volume Spike Indicator
    volume_threshold = 1.5
    multiplier = 1.2
    df['volume_moving_avg'] = df['volume'].rolling(window=window).mean()
    df['volume_spike_indicator'] = (df['volume'] > volume_threshold * df['volume_moving_avg']).astype(float) * (multiplier - 1) + 1
    
    df['volume_weighted_momentum_adjusted'] = df['volume_weighted_momentum'] * df['volume_spike_indicator']
    
    return df['volume_weighted_momentum_adjusted']
```
