```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap with Exponential Decay (half-life = 1 day)
    df['adjusted_daily_return'] = df['daily_return'] + df['price_gap'].ewm(halflife=1, ignore_na=True).mean()
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].pct_change()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['volume_adjusted_momentum'] = df['price_range_momentum'] * (1 + df['volume_change'])
    
    # Calculate Accumulated Volume Impact
    df['accumulated_volume'] = df['volume'].rolling(window=5, min_periods=1).sum()
    df['volume_weighted_return'] = df['adjusted_daily_return'] * (df['accumulated_volume'] / df['volume'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['volume_weighted_return'] + 
                                df['volume_adjusted_momentum'] + 
                                df['accumulated_volume'])
    
    return df['final_alpha_factor']
```
