```python
import pandas as pd

def heuristics_v2(df):
    # Subthought 1.1: Calculate the difference between high and low prices of the day t
    price_range = df['high'] - df['low']
    
    # Subthought 1.2: Evaluate the change in close price from day t-1 to day t
    close_change = df['close'].diff()
    
    # Subthought 1.3: Analyze the trend of open prices over the last 5 trading days
    open_trend_5d = df['open'].rolling(window=5).mean()
    
    # Subthought 1.4: Investigate the relationship between the open and close prices on day t
    open_close_ratio = df['close'] / df['open']
    
    # Subthought 2.1: Measure the daily volume change from day t-1 to day t
    volume_change = df['volume'].diff()
    
    # Subthought 2.2: Look into the total trade amount's impact, considering its change from day t-1
    amount_change = df['amount'].diff()
    
    # Subthought 2.3: Examine the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Subthought 2.4: Consider the ratio of day t's volume to the average volume over the past 5 days
    volume_ratio_5d = df['volume'] / avg_volume_5d
    
    # Subthought 3.1: Integrate price volatility (high-low) with volume change for a new indicator
    price_volatility_volume = price_range * volume_change
    
    # Subthought 3.2: Mix the momentum of close prices with the trend in trading volume
    momentum_volume_trend = close_change * df['volume'].pct_change()
    
    # Subthought 3.3: Combine the relationship between open and close prices with the volume ratio
    open_close_volume_ratio = (df['close'] - df['open']) * volume_ratio_5d
    
    # Subthought 4.1: Identify significant gaps between the open and previous close prices
    gap_open_close = df['open'] - df['close'].shift(1)
    
    # Subthought 4.2: Detect unusual spikes in volume compared to the recent average
    volume_spike = (df['volume'] > avg_volume_5d * 1.5).astype(int)
    
    # Subthought 4.3: Look for consistent patterns where the close is higher/lower than the open for several consecutive days
    close_higher_than_open = (df['close'] > df['open']).rolling(window=5).sum()
    
    # Subthought 4.4: Analyze the instances where the closing price is exactly at the high or low of the day
    close_at_extreme = ((df['close'] == df['high']) | (df['close'] == df['low'])).astype(int)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        price_range * 0.2 +
        close_change * 0.2 +
        open_trend_5d * 0.1 +
        open_close_ratio * 0.1 +
        volume_change * 0.1 +
        amount_change * 0.05 +
        volume_ratio_5d * 0.05 +
        price_volatility_volume * 0.05 +
        momentum_volume_trend * 0.05 +
        open_close_volume_ratio * 0.05 +
        gap_open_close * 0.05 +
        volume_spike * 0.05 +
        close_higher_than_open * 0.05 +
        close_at_extreme * 0.05
    )
    
    return alpha_factor
```
