import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap with Exponential Decay
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            return row['daily_return'] - (row['price_gap'] * decay_factor)
        elif row['price_gap'] < 0:
            return row['daily_return'] + (-row['price_gap'] * decay_factor)
        else:
            return row['daily_return']
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Evaluate Momentum Over Time
    momentum_window = 10
    df['momentum'] = df['adjusted_daily_return'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    volume_spike_multiplier = 1.2
    
    df['volume_moving_avg'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = df['volume'] > df['volume_moving_avg'] * volume_spike_threshold
    df['volume_weighted_momentum'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * volume_spike_multiplier if row['volume_spike'] else row['volume_weighted_momentum'],
        axis=1
    )
    
    # Adjust for Volatility
    volatility_window = 10
    df['volatility'] = df['daily_return'].rolling(window=volatility_window).std()
    volatility_threshold = 2
    volatility_multiplier = 1.3
    
    df['volatility_adjusted_momentum'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * volatility_multiplier if abs(row['daily_return']) > row['volatility'] * volatility_threshold else row['volume_weighted_momentum'],
        axis=1
    )
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] - df['price_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Range Momentum with Volume Change
    df['integrated_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Adjusted Daily Return with Price Range Momentum
    df['combined_factor'] = df['adjusted_daily_return'] * df['volume'] + df['integrated_price_range_momentum']
    
    # Calculate 5-day Exponential Moving Average for Price
    ema_window = 5
    df['ema_price'] = df['close'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_price'] - df['ema_price'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume'] - df['ema_volume'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = df['volume_acceleration'] * df['price_momentum'] + df['combined_factor']
    df['combined_factors'] = df['combined_factors'] * df['volume_weighted_momentum']
    
    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = df['combined_factors'] / df['volume'].rolling(window=ema_window).sum()
    
    # Ensure Positive Factor
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))
    
    # Integrate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['open']
    df['intraday_momentum'] = df['intraday_momentum'] * df['volume']
    
    # Adjust for Intraday Momentum
    df
