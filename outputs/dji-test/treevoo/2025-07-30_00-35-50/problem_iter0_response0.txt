import pandas as pd

def heuristics_v2(df):
    # Calculate the close-to-open return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Return the factor as a pandas Series
    return df['volume_weighted_return']
