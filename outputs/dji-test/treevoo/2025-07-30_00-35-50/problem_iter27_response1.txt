```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    range_momentum = high_low_range - high_low_range.shift(1)
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    integrated_factor = (range_momentum * volume_change) + price_change
    
    # Confirm with Volume
    volume_change_adjusted = volume_change + 1
    combined_factor = integrated_factor / volume_change_adjusted
    
    # Smoothing Filter: Apply Exponential Moving Average (EMA) with a period of 5 days
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    # Calculate Adjusted Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    price_gap = df['open'] - df['close'].shift(1)
    adjusted_daily_return = daily_return - price_gap * (price_gap > 0) + price_gap * (price_gap < 0)
    
    # Integrate Smoothed Volume-Adjusted High-Low Spread with Adjusted Daily Return
    integrated_smoothed_factor = adjusted_daily_return * smoothed_factor
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    final_factor = integrated_smoothed_factor * volume_ratio
    
    # Ensure Positive Factor
    final_factor = final_factor.apply(lambda x: max(x, 0))
    
    return final_factor
```
