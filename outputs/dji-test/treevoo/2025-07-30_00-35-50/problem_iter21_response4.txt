```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * decay_factor
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * decay_factor

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(10)

    # Adjust by Volume and Volume-Weighted Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']

    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['integrated_factor'] = df['range_momentum'] * df['volume_change'] + df['volume_weighted_price_change']

    # Combine Integrated Factor with Volume Change
    df['combined_integrated_factor'] = df['integrated_factor'] * df['volume_change']
    df['combined_integrated_factor'] = df['combined_integrated_factor'].where(df['volume_change'] > 0, 0)

    # Smoothing Filter: Apply Simple Moving Average (SMA)
    df['sma_5'] = df['combined_integrated_factor'].rolling(window=5).mean()

    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_close'] = df['close'].ewm(span=5).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)

    # Combine Factors
    df['final_combined_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['sma_5']

    # Final Alpha Factor with Volume Weighting
    df['sum_last_5_volumes'] = df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['final_combined_factor'] / df['sum_last_5_volumes']
    df['final_alpha_factor'] = df['final_alpha_factor'].clip(lower=0)

    # Incorporate Trading Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=5).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / df['volume_volatility']

    # Introduce Price Trend Indicator
    df['ema_20_close'] = df['close'].ewm(span=20).mean()
    df['price_trend'] = df['ema_20_close'] - df['ema_20_close'].shift(1)
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['price_trend']
    df['final_alpha_factor'] = df['final_alpha_factor'].clip(lower=0)

    return df['final_alpha_factor']
```
