```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap with Exponential Decay
    half_life = 1
    decay_factor = 0.5 ** (1 / half_life)
    df['Adjustment'] = df['Price_Gap'].apply(lambda x: -x if x > 0 else x)
    df['Adjusted_Daily_Return'] = df['Daily_Return'] + df['Adjustment'] * decay_factor

    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['Price_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)

    # Adjust by Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Integrate Price Range Momentum with Volume Change
    df['Integrated_Price_Range_Momentum'] = df['Price_Range_Momentum'] * df['Volume_Change']

    # Combine Adjusted Daily Return with Integrated Price Range Momentum
    df['Combined_Factor'] = df['Adjusted_Daily_Return'] * df['volume'] + df['Integrated_Price_Range_Momentum']

    # Calculate 5-day Exponential Moving Average for Price
    df['EMA_5_Price'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_5_Price'] - df['EMA_5_Price'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['EMA_5_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_5_Volume'] - df['EMA_5_Volume'].shift(1)

    # Combine Factors
    df['Combined_Factor'] += df['Volume_Acceleration'] * df['Price_Momentum']

    # Final Alpha Factor with Volume Weighting
    df['Final_Alpha_Factor'] = df['Combined_Factor'] / df['volume'].rolling(window=5).sum()

    # Ensure Positive Factor
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].apply(lambda x: max(x, 0))

    return df['Final_Alpha_Factor']
```
