```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Evaluate Short-Term Momentum
    short_window = 5
    df['short_term_momentum'] = df['price_change'].rolling(window=short_window).sum()
    
    # Compute Volume Weighted Short-Term Momentum
    df['volume_weighted_momentum'] = (df['short_term_momentum'] * df['volume']) / df['volume'].rolling(window=short_window).mean()
    
    # Adjust for Volume Spike Indicator
    vol_spike_threshold = 1.5
    df['vol_spike_multiplier'] = np.where(df['volume'] > df['volume'].rolling(window=10).mean() * vol_spike_threshold, 1.2, 1.0)
    df['volume_weighted_momentum'] *= df['vol_spike_multiplier']
    
    # Adjust for Volatility
    df['historical_volatility'] = df['price_change'].rolling(window=short_window).std()
    vol_adjustment_threshold = 2.0
    df['vol_adjustment_multiplier'] = np.where(np.abs(df['price_change']) > df['historical_volatility'] * vol_adjustment_threshold, 1.3, 1.0)
    df['volume_weighted_momentum'] *= df['vol_adjustment_multiplier']
    
    # Adjust for Long-Term Price Momentum
    long_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_window)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_window).mean()
    
    # Combine All Adjustments
    df['combined_factor'] = df['long_term_momentum'] * df['volume_weighted_momentum'] * df['volume_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['final_factor'] = np.exp(df['combined_factor']) / (1 + np.exp(df['combined_factor']))
    
    return df['final_factor']
```
