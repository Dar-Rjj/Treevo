import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine High-Low Spread with Volume Change
    combined_factor = high_low_spread * (volume_change > 0) * volume_change
    
    # Apply Simple Moving Average (SMA) to Combined Factor
    sma_combined_factor = combined_factor.rolling(window=5).mean()
    
    # Calculate Price Range Momentum
    price_range_momentum = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    # Adjust by Volume and Price Change
    price_change = df['close'] - df['close'].shift(1)
    integrated_factor = price_range_momentum * volume_change + price_change
    
    # Confirm with Volume
    confirmed_factor = integrated_factor * (volume_change > 0)
    
    # Smoothing Filter
    final_factor = confirmed_factor.rolling(window=5).mean()
    
    return final_factor
