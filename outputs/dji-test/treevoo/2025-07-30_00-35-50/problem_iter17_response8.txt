```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10_daily_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 20-day moving average of the close price
    df['ma_20_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the close price
    df['std_20_close'] = df['close'].rolling(window=20).std()
    
    # Calculate the 10-day exponentially weighted moving average of the volume
    df['ewma_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength (RS) as the ratio of the 10-day MA of the daily return to its 10-day standard deviation
    df['rs_factor'] = df['ma_10_daily_return'] / df['std_10_daily_return']
    
    # Calculate the volatility factor as the ratio of the 20-day MA of the close price to its 20-day standard deviation
    df['volatility_factor'] = df['ma_20_close'] / df['std_20_close']
    
    # Combine the RS factor and the volatility factor
    df['combined_factor'] = df['rs_factor'] * df['volatility_factor']
    
    # Normalize the combined factor by dividing it by the 10-day EWMA of the volume
    df['normalized_combined_factor'] = df['combined_factor'] / df['ewma_10_volume']
    
    # Return the normalized combined factor as the final alpha factor
    return df['normalized_combined_factor']
```
