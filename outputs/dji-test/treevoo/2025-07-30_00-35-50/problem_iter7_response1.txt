import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # 10-day Moving Average Volume
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Integrate Momentum and Volume Change
    df['integrated_range_momentum'] = df['range_momentum'] * df['volume_change']
    df['integrated_close_momentum'] = df['close_momentum'] * (df['volume'] - df['avg_volume_10'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['integrated_range_momentum'] + df['integrated_close_momentum']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
