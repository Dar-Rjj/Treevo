```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    close_ema = df['close'].ewm(span=5, adjust=False).mean()
    price_momentum = close_ema - close_ema.shift(1)
    
    # Calculate Volume Acceleration
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_acceleration = volume_ema - volume_ema.shift(1)
    
    # Calculate Amount Acceleration
    amount_ema = df['amount'].ewm(span=5, adjust=False).mean()
    amount_acceleration = amount_ema - amount_ema.shift(1)
    
    # Combine Factors
    factor = (volume_acceleration * price_momentum) + amount_acceleration
    
    return factor
```
