```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = row['price_gap'] * np.exp(-np.log(2) / 1)
            return row['daily_return'] - adjustment
        elif row['price_gap'] < 0:
            adjustment = row['price_gap'] * np.exp(-np.log(2) / 1)
            return row['daily_return'] + adjustment
        return row['daily_return']
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)

    # Evaluate Momentum Over Time (e.g., 10 days)
    df['momentum'] = df['adjusted_daily_return'].rolling(window=10).sum()

    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=10).mean()

    # Adjust for Volume Spike Indicator
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_spike_indicator'] = np.where(df['volume'] > 1.5 * df['volume_moving_avg'], 1.2, 1)
    df['volume_weighted_momentum'] *= df['volume_spike_indicator']

    # Adjust for Volatility
    df['historical_volatility'] = df['adjusted_daily_return'].rolling(window=10).std()
    df['volatility_adjustment'] = np.where(abs(df['close'] - df['close'].shift(1)) > 2 * df['historical_volatility'], 1.3, 1)
    df['volume_weighted_momentum'] *= df['volatility_adjustment']

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)

    # Integrate Momentum, Volume, and Price Change
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']

    # Combine Adjusted Daily Return with Integrated Momentum
    df['combined_factor'] = df['adjusted_daily_return'] * df['volume'] + df['integrated_momentum']

    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_price'] - df['ema_5_price'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)

    # Combine Factors
    df['combined_factors'] = (df['volume_acceleration'] * df['price_momentum'] + 
                              df['combined_factor'] * df['volume_weighted_momentum'])

    # Final Alpha Factor with Volume Weighting
    df['final_alpha'] = df['combined_factors'] / df['volume'].rolling(window=5).sum()

    # Ensure Positive Factor
    df['final_alpha'] = df['final_alpha'].apply(lambda x: max(x, 0))

    return df['final_alpha']
```
