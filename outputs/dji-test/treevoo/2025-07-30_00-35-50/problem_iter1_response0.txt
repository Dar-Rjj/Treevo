```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day moving average of close price
    df['close_5d_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference in 5-day moving average of close price
    df['price_momentum'] = df['close_5d_ma'] - df['close_5d_ma'].shift(1)
    
    # Calculate 5-day moving average of volume
    df['volume_5d_ma'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the difference in 5-day moving average of volume
    df['volume_acceleration'] = df['volume_5d_ma'] - df['volume_5d_ma'].shift(1)
    
    # Calculate the final factor by multiplying price momentum and volume acceleration
    df['factor'] = df['price_momentum'] * df['volume_acceleration']
    
    return df['factor']
```
