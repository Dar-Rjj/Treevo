```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Evaluate Momentum Over Time
    momentum_window = 10
    df['momentum_10d'] = df['daily_return'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_10d_sum'] = df['volume'].rolling(window=momentum_window).sum()
    df['average_volume_10d'] = df['volume_10d_sum'] / momentum_window
    df['volume_weighted_momentum'] = (df['momentum_10d'] * df['volume']) / df['average_volume_10d']
    
    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    multiplier_volume_spike = 1.2
    df['volume_moving_avg_10d'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike_indicator'] = (df['volume'] > (volume_threshold * df['volume_moving_avg_10d'])).astype(int)
    df['volume_adjusted_momentum'] = df['volume_weighted_momentum'] * (1 + (df['volume_spike_indicator'] * (multiplier_volume_spike - 1)))
    
    # Adjust for Volatility
    volatility_multiplier = 1.3
    df['historical_volatility_10d'] = df['daily_return'].rolling(window=momentum_window).std()
    df['price_change_abs'] = (df['close'] - df['open']).abs()
    df['volatility_indicator'] = (df['price_change_abs'] > (2 * df['historical_volatility_10d'])).astype(int)
    df['volatility_adjusted_momentum'] = df['volume_adjusted_momentum'] * (1 + (df['volatility_indicator'] * (volatility_multiplier - 1)))
    
    # Combine Factors
    df['price_momentum'] = df['momentum_10d']
    df['final_factor'] = df['volatility_adjusted_momentum'] * df['price_momentum']
    
    # Ensure Positive Factor
    df['final_factor'] = df['final_factor'].apply(lambda x: exp(x) / (1 + exp(x)) if x < 0 else x)
    
    return df['final_factor']
```
