import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is a combination of price and volume information, designed to be interpretable
    and predictive of future stock returns.
    
    :param df: A pandas DataFrame with (date) as index and columns (open, high, low, close, volume).
    :return: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 20-day moving average of the daily return
    df['ma_20_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate the 20-day moving average of the closing price
    df['ma_20_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day volatility of the closing price
    df['volatility_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the 20-day volume weighted average price (VWAP)
    df['cumulative_volume'] = df['volume'].rolling(window=20).sum()
    df['cumulative_price_volume'] = (df['close'] * df['volume']).rolling(window=20).sum()
    df['vwap_20'] = df['cumulative_price_volume'] / df['cumulative_volume']
    
    # Calculate the distance between the current close and the 20-day VWAP
    df['distance_to_vwap'] = df['close'] - df['vwap_20']
    
    # Calculate the factor as a combination of the above components
    df['factor'] = (df['ma_20_daily_return'] + 
                    (df['close'] - df['ma_20_close']) / df['volatility_20'] + 
                    df['distance_to_vwap'])
    
    # Return the factor as a pandas Series
    return df['factor'].dropna()
