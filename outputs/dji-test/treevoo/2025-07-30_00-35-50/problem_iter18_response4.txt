```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range / high_to_low_range.shift(1)
    
    # Adjust by Volume and Price Change
    volume_change = df['volume'] / df['volume'].shift(1)
    price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    integrated_factor = (price_range_momentum * volume_change) + price_change
    
    # Calculate 5-day EMA for Close Price
    ema_5_close = df['close'].ewm(span=5, adjust=False).mean()
    short_term_price_momentum = ema_5_close - ema_5_close.shift(1)
    
    # Calculate 5-day EMA for Volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    short_term_volume_acceleration = ema_5_volume - ema_5_volume.shift(1)
    
    # Combine Short-Term Factors
    short_term_combined = (short_term_volume_acceleration * short_term_price_momentum) + integrated_factor
    
    # Calculate 10-day EMA for Close Price
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    long_term_price_momentum = ema_10_close - ema_10_close.shift(1)
    
    # Calculate 10-day EMA for Volume
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    long_term_volume_acceleration = ema_10_volume - ema_10_volume.shift(1)
    
    # Combine Long-Term Factors
    long_term_combined = (long_term_volume_acceleration * long_term_price_momentum) + short_term_combined
    
    # Calculate 20-day EMA for Close Price
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    medium_term_price_momentum = ema_20_close - ema_20_close.shift(1)
    
    # Calculate 20-day EMA for Volume
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    medium_term_volume_acceleration = ema_20_volume - ema_20_volume.shift(1)
    
    # Combine Medium-Term Factors
    medium_term_combined = (medium_term_volume_acceleration * medium_term_price_momentum) + long_term_combined
    
    # Final Alpha Factor
    final_alpha_factor = medium_term_combined
    
    return final_alpha_factor
```
