```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Combine High-Low Spread with Volume Change
    combined_factor = high_low_spread * volume_change
    combined_factor = combined_factor.where(volume_change > 0, 0)
    
    # Apply Simple Moving Average (SMA) with a period of 5 days
    sma_period = 5
    smoothed_alpha_factor = combined_factor.rolling(window=sma_period).mean()
    
    return smoothed_alpha_factor
```
