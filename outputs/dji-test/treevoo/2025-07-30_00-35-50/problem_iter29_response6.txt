```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return'] - df['price_gap'].apply(lambda x: x if x > 0 else 0) + df['price_gap'].apply(lambda x: x if x < 0 else 0)
    
    # Evaluate Short-Term Momentum
    short_window = 5
    df['short_term_momentum'] = df['adjusted_daily_return'].rolling(window=short_window).sum()
    
    # Compute Volume Weighted Short-Term Momentum
    df['volume_weighted_short_term_momentum'] = df['short_term_momentum'] * df['volume'] / df['volume'].rolling(window=short_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    multiplier_volume_spike = 1.2
    df['volume_sma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_spike_indicator'] = (df['volume'] > df['volume_sma_10'] * volume_threshold).astype(int)
    df['volume_weighted_short_term_momentum'] = df['volume_weighted_short_term_momentum'] * (1 + (df['volume_spike_indicator'] * (multiplier_volume_spike - 1)))
    
    # Adjust for Volatility
    volatility_threshold = 2
    multiplier_volatility = 1.3
    df['historical_volatility'] = df['adjusted_daily_return'].rolling(window=short_window).std()
    df['volatility_adjustment'] = (np.abs(df['adjusted_daily_return']) > df['historical_volatility'] * volatility_threshold).astype(int)
    df['volume_weighted_short_term_momentum'] = df['volume_weighted_short_term_momentum'] * (1 + (df['volatility_adjustment'] * (multiplier_volatility - 1)))
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['average_volume'] = df['volume'].rolling(window=short_window).mean()
    df['adjusted_high_low_spread'] = df['high_low_spread'] * (df['volume_change'] / df['average_volume']).apply(lambda x: x if x > 0 else 0)
    
    # Incorporate Price Confirmation
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['combined_high_low_spread'] = df['adjusted_high_low_spread'] * ((df['price_change'] + df['volume_change']) > 0).astype(int)
    
    # Adjust for Long-Term Price Momentum
    df['long_term_price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Combine All Adjustments
    df['combined_factor'] = df['long_term_price_momentum'] * df['volume_weighted_short_term_momentum'] * (1 + df['volume_trend'] / df['volume'])
    df['combined_factor'] = df['combined_factor'] + df['combined_high_low_spread'] * (df['combined_factor'] > 0).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'].apply(lambda x: np.exp(x) / (1 + np.exp(x)) if x < 0 else x)
    
    return df['alpha_factor']
```
