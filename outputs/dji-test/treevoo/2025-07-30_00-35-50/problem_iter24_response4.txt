import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    df['volume_spike_multiplier'] = 1.2
    df.loc[df['volume'] > volume_spike_threshold * df['volume'].rolling(window=momentum_window).mean(), 'volume_weighted_momentum'] *= df['volume_spike_multiplier']
    
    # Adjust for Volatility
    historical_volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=historical_volatility_window).std()
    volatility_threshold = 2
    df['volatility_multiplier'] = 1.3
    df.loc[abs(df['daily_price_change']) > volatility_threshold * df['historical_volatility'], 'volume_weighted_momentum'] *= df['volatility_multiplier']
    
    # Adjust for Long-Term Price Momentum
    long_term_momentum_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_momentum_window)
    df['adjusted_momentum'] = df['volume_weighted_momentum'] * df['long_term_momentum']
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_momentum_window).mean()

    intraday_volatility_window = 10
    df['intraday_volatility'] = df['intraday_range'].rolling(window=intraday_volatility_window).std()
    intraday_volatility_threshold = 2
    df['intraday_volatility_multiplier'] = 1.4
    df.loc[df['intraday_range'] > intraday_volatility_threshold * df['intraday_volatility'], 'adjusted_momentum'] *= df['intraday_volatility_multiplier']
