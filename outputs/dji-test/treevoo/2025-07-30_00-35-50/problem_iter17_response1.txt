import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', and 'volume'.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['10_day_ma'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day volatility (standard deviation) of the daily return
    df['30_day_volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the 10-day moving average of the volume
    df['10_day_volume_ma'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the volume momentum as the ratio of the current volume to the 10-day moving average of the volume
    df['volume_momentum'] = df['volume'] / df['10_day_volume_ma']
    
    # Calculate the high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the 20-day moving average of the high-low range
    df['20_day_high_low_ma'] = df['high_low_range'].rolling(window=20).mean()
    
    # Calculate the alpha factor as a combination of the 10-day moving average return, 
    # the 30-day volatility, the volume momentum, and the 20-day moving average of the high-low range
    df['alpha_factor'] = (df['10_day_ma'] / df['30_day_volatility']) * df['volume_momentum'] * (df['high_low_range'] / df['20_day_high_low_ma'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
