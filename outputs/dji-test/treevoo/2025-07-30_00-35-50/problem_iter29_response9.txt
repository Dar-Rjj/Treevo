import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap with Exponential Decay
    def exponential_decay(x, half_life=1):
        return np.exp(-np.log(2) / half_life * x)
    
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap'] * exponential_decay(1)
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap'] * exponential_decay(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    df['Momentum'] = df['Adjusted_Daily_Return'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Momentum'] * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for Volume Spike Indicator
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    df['Volume_Weighted_Momentum_Adjusted'] = df['Volume_Weighted_Momentum']
    df.loc[df['volume'] > 1.5 * volume_moving_avg, 'Volume_Weighted_Momentum_Adjusted'] *= 1.2
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    avg_volume = df['volume'].rolling(window=10).mean()
    df['Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * (df['Volume_Change'] / avg_volume)
    df['Adjusted_High_Low_Spread'] = df['Adjusted_High_Low_Spread'].fillna(0)
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['Integrated_Factor'] = df['Volume_Weighted_Momentum_Adjusted'] + df['Adjusted_High_Low_Spread']
    
    # Combine Integrated Factor with Volume Change
    df['Integrated_Factor_Volume_Adjusted'] = df['Integrated_Factor'] * df['Volume_Change'].apply(lambda x: max(x, 0))
    
    # Smoothing Filter (5-day SMA)
    df['SMA_5'] = df['Integrated_Factor_Volume_Adjusted'].rolling(window=5).mean()
    
    # Calculate 5-day EMA for Close Price
    df['EMA_5_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_5_Close'] - df['EMA_5_Close'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['EMA_5_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_5_Volume'] - df['EMA_5_Volume'].shift(1)
    
    # Combine Factors
    df['Combined_Factor'] = df['Volume_Acceleration'] * df['Price_Momentum'] + df['SMA_5']

    df['Sum_Volumes_5'] = df['volume'].rolling(window=5).sum()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] / df['Sum_Volumes_5']
