import pandas as pd

def heuristics_v2(df, lookback_period=30):
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Directionality
    df['close_minus_open'] = df['close'] - df['open']
    df['high_minus_low'] = df['high'] - df['low']
    df['price_directionality'] = df['close_minus_open'] / df['high_minus_low']
    
    # Combine Volume and Price Information
    df['combined_factor'] = df['volume_change'] * df['price_directionality']
    
    # Adjust for Liquidity
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_shock_factor'] = df['combined_factor'] / df['avg_volume']
    
    # Return the volume shock factor
    return df['volume_shock_factor']
