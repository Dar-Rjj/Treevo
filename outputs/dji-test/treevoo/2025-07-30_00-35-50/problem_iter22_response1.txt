```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum'] = df['price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_ma = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * volume_ma).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + 0.2 * df['volume_spike'])
    
    # Adjust for Volatility
    volatility = df['price_change'].rolling(window=momentum_window).std()
    high_volatility = (np.abs(df['price_change']) > 2 * volatility).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + 0.3 * high_volatility)
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_window)
    
    # Adjust for Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    
    # Combine All Adjustments
    factor = df['long_term_momentum'] * df['volume_weighted_momentum'] * (1 + volume_trend / df['volume'])
    
    # Ensure Positive Factor
    factor = np.exp(factor) / (1 + np.exp(factor))
    
    return factor
```
