```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term (5-day) and long-term (20-day) moving averages of close prices
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    # Subtract the 5-day moving average from the 20-day moving average to capture trend differences
    df['trend_difference'] = df['20_day_ma'] - df['5_day_ma']
    
    # Compute the ratio of today's close price to the average close price over the past 10 days
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['price_strength_ratio'] = df['close'] / df['10_day_avg_close']
    
    # Measure the volatility using the standard deviation of log returns over the past 10 days
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Evaluate the accumulation/distribution line by considering daily volume and the position of the closing price within the day's range
    df['money_flow_multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    df['accumulation_distribution_line'] = df['money_flow_volume'].rolling(window=30).sum()
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (df['trend_difference'] + df['price_strength_ratio'] - df['volatility'] + df['accumulation_distribution_line']).dropna()
    
    return df['alpha_factor']
```
