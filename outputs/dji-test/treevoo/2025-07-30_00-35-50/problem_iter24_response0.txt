```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    adjustment = df['Price_Gap'].apply(lambda x: x if x < 0 else -x)
    df['Adjusted_Return'] = df['Daily_Return'] + (adjustment * np.exp(-1))

    # Calculate High-to-Low Price Range
    df['Price_Range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['Price_Range_Momentum'] = df['Price_Range'] - df['Price_Range'].shift(1)

    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].shift(10)

    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Calculate Volume-Weighted Price Change
    df['Volume_Weighted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']

    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['Integrated_Factor'] = df['Price_Range_Momentum'] * df['Volume_Change'] + df['Volume_Weighted_Price_Change']

    # Combine Integrated Factor with Volume Change
    df['Combined_Factor'] = df['Integrated_Factor'] * df['Volume_Change']
    df['Combined_Factor'] = df['Combined_Factor'].where(df['Volume_Change'] > 0, 0)

    # Smoothing Filter
    df['SMA_Combined_Factor'] = df['Combined_Factor'].rolling(window=5).mean()

    # Calculate 5-day Exponential Moving Average for Price
    df['EMA_Close_5'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_Close_5'] - df['EMA_Close_5'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['EMA_Volume_5'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_Volume_5'] - df['EMA_Volume_5'].shift(1)

    # Combine Factors
    df['Combined_Factors'] = df['Volume_Acceleration'] * df['Price_Momentum'] + df['Integrated_Factor']

    # Final Alpha Factor with Volume Weighting
    sum_volume = df['volume'].rolling(window=5).sum()
    df['Final_Alpha_Factor'] = df['Combined_Factors'] / sum_volume
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].where(df['Final_Alpha_Factor'] > 0, 0)

    # Introduce Volatility-Adjusted Momentum
    df['Historical_Volatility'] = df['close'].rolling(window=5).std()
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] / df['Historical_Volatility']

    # Enhance Volume-Weighted Adjustments
    df['Volume_Weighted_Price_Gap'] = df['Price_Gap'] * df['volume']
    df['Final_Alpha_Factor'] += df['Volume_Weighted_Price_Gap'] * np.exp(-1)

    return df['Final_Alpha_Factor']
```
