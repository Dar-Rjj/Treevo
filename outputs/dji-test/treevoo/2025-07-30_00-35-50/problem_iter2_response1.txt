```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].shift(10) - df['close']
    
    # Adjust for Volume Trend
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_adjustment'] = df['volume'] - df['volume_ma_10']
    
    # Combine Price Momentum and Volume Adjustment
    df['combined_factor'] = df['price_momentum'] * df['volume_adjustment']
    df['combined_factor'] = df['combined_factor'].apply(lambda x: max(x, 0))
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_returns = df[df['log_return'] > 0]
    negative_returns = df[df['log_return'] <= 0]
    
    # Calculate Sum of Upward Volume
    df['upward_volume_sum'] = df['volume'].where(df['log_return'] > 0, 0)
    upward_volume_sum = df['upward_volume_sum'].sum()
    
    # Calculate Sum of Downward Volume
    df['downward_volume_sum'] = df['volume'].where(df['log_return'] <= 0, 0)
    downward_volume_sum = df['downward_volume_sum'].sum()
    
    # Compute Trend Reversal Signal
    total_volume = upward_volume_sum + downward_volume_sum
    upward_ratio = upward_volume_sum / total_volume
    downward_ratio = downward_volume_sum / total_volume
    reversal_signal = (upward_ratio > downward_ratio) * df['log_return']
    
    # Combine Price Momentum with Trend Reversal Signal
    df['final_factor'] = df['combined_factor'] + reversal_signal
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))
    
    return df['final_factor']
```
