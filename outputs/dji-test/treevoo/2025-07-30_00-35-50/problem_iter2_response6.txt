```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = -row['price_gap']
        else:
            adjustment = row['price_gap']
        return row['daily_return'] + (adjustment * decay_factor)
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate Price Range Momentum
    df['prev_price_range'] = df['price_range'].shift(1)
    df['price_range_momentum'] = df['price_range'] / df['prev_price_range']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum, Volume Change, and Adjusted Daily Return
    df['momentum_volume'] = df['price_range_momentum'] * df['volume_change']
    df['final_alpha'] = df['momentum_volume'] + df['adjusted_daily_return']
    
    # Final Alpha Factor
    alpha_factor = df['final_alpha']
    
    return alpha_factor
```
