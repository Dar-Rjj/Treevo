import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap'] * np.exp(-1)
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap'] * np.exp(-1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['Sum_Daily_Returns'] = df['Adjusted_Daily_Return'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Sum_Daily_Returns'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    volume_multiplier = 1.2
    df['Volume_MA'] = df['volume'].rolling(window=momentum_window).mean()
    df['Volume_Spike'] = df['volume'] > volume_spike_threshold * df['Volume_MA']
    df['Volume_Adjusted_Momentum'] = df['Volume_Weighted_Momentum'] * df['Volume_Spike'].apply(lambda x: volume_multiplier if x else 1)
    
    # Adjust for Volatility
    volatility_window = 10
    df['Historical_Volatility'] = df['Daily_Return'].rolling(window=volatility_window).std()
    volatility_threshold = 2
    volatility_multiplier = 1.3
    df['Volatility_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] * (df['close'] - df['close'].shift(1)).abs().gt(volatility_threshold * df['Historical_Volatility']).apply(lambda x: volatility_multiplier if x else 1)
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    average_volume = df['volume'].rolling(window=momentum_window).mean()
    df['Volume_Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * (df['Volume_Change'] / average_volume).fillna(0)
    
    # Incorporate Price Confirmation
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Combined_Factor'] = df['Volume_Adjusted_High_Low_Spread']
    df.loc[(df['Price_Change'] > 0) & (df['Volume_Change'] > 0), 'Combined_Factor'] *= (df['Price_Change'] + df['Volume_Change'])
    
    return df['Combined_Factor']
