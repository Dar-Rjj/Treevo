import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily returns
    df['ma_10'] = df['return'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the daily returns
    df['ma_30'] = df['return'].rolling(window=30).mean()
    
    # Calculate the standard deviation of the daily returns over the last 30 days
    df['std_30'] = df['return'].rolling(window=30).std()
    
    # Calculate the 10-day exponential moving average of the daily returns
    df['ema_10'] = df['return'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the daily returns
    df['ema_30'] = df['return'].ewm(span=30, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over the last 14 days
    df['rsi_14'] = df['return'].rolling(window=14).apply(lambda x: 100 - (100 / (1 + (x[x > 0].mean() / -x[x < 0].mean()))), raw=True)
    
    # Calculate the momentum factor over the last 12 days
    df['momentum_12'] = df['close'] / df['close'].shift(12) - 1
    
    # Calculate the price range
    df['price_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the factor as a combination of the above metrics
    df['alpha_factor'] = (
        0.25 * (df['ema_10'] - df['ema_30']) / df['std_30'] +
        0.25 * (df['ma_10'] - df['ma_30']) / df['std_30'] +
        0.25 * (df['rsi_14'] / 100 - 0.5) +
        0.25 * df['momentum_12']
    )
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
