import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    def adjust_daily_return(daily_return, price_gap):
        if price_gap > 0:
            return daily_return - price_gap * np.exp(-1)
        else:
            return daily_return + price_gap * np.exp(-1)
    
    df['adjusted_daily_return'] = df.apply(lambda row: adjust_daily_return(row['daily_return'], row['price_gap']), axis=1)

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_momentum'] = df['high_low_range'].diff()

    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(20)

    # Adjust for Volume Trend
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_adjustment'] = df['volume'] - df['volume_20_day_ma']

    # Adjust for Amount Trend
    df['amount_20_day_ma'] = df['amount'].rolling(window=20).mean()
    df['amount_adjustment'] = df['amount'] - df['amount_20_day_ma']

    # Combine Price Momentum, Volume Adjustment, and Amount Adjustment
    df['combined_price_momentum'] = (df['close_price_momentum'] * df['volume_adjustment'] * df['amount_adjustment']).clip(lower=0)

    # Integrate Momentum and Volume Change
    df['integrated_range_momentum'] = df['high_low_range_momentum'] * (df['volume'] - df['volume_20_day_ma'])

    # Combine Integrated Range Momentum and Integrated Close Price Momentum
    df['integrated_momentum'] = df['integrated_range_momentum'] + df['combined_price_momentum']

    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Identify Positive and Negative Returns
    positive_returns = df[df['daily_log_return'] > 0]
    negative_returns = df[df['daily_log_return'] <= 0]

    # Calculate Sum of Upward Volume
    sum_upward_volume = positive_returns['volume'].sum()

    # Calculate Sum of Downward Volume
    sum_downward_volume = negative_returns['volume'].sum()

    # Compute Enhanced Trend Reversal Signal
    total_volume = sum_upward_volume + sum_downward_volume
    upward_volume_ratio = sum_upward_volume / total_volume
    downward_volume_ratio = sum_downward_volume / total_volume
    reversal_signal = (upward_volume_ratio > downward_volume_ratio) * df['daily_log_return']

    # Introduce Trend Confirmation
    df['log_return_20_day_ma'] = df['daily_log_return'].rolling(window=20).mean()
    trend_confirmation = np.where(df['daily_log_return'] > df['log_return_20_day_ma'], 1, -1)
    enhanced_reversal_signal = reversal_signal * trend_confirmation

    # Final Alpha Factor with Volume Weighting and Trend Reversal
    df['final_factor'] = df['integrated_momentum'] * enhanced_reversal_signal
    df['final_factor'] = df['final_factor'].clip(lower=0)

    return df['final_factor']
