```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * np.exp(-1)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * np.exp(-1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust by Volume and Volume-Weighted Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['integrated_factor'] = df['price_range_momentum'] * df['volume_change'] + df['volume_weighted_price_change']
    
    # Combine Integrated Factor with Volume Change
    df['combined_integrated_factor'] = df['integrated_factor'] * df['volume_change'].apply(lambda x: max(x, 0))
    
    # Smoothing Filter
    df['smoothed_factor'] = df['combined_integrated_factor'].rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Factors
    df['final_combined_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['smoothed_factor']
    
    # Final Alpha Factor with Volume Weighting
    df['alpha_factor'] = df['final_combined_factor'] / df['volume'].rolling(window=5).sum() + df['price_range_momentum'] + df['close_price_momentum']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
```
