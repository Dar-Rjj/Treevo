import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using open, high, low, close, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the relative range
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / df['close'].shift(1)
    
    # Calculate the relative volume change
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Calculate the price momentum
    df['momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the weighted sum of these features
    factor = (
        0.4 * df['relative_range'] + 
        0.3 * df['volume_change'] + 
        0.3 * df['momentum']
    )
    
    return factor.dropna()
