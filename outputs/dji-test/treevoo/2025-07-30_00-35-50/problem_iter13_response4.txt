```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_low_range.pct_change()
    
    # Adjust by Volume and Price Change
    volume_change = df['volume'].pct_change()
    price_change = df['close'].pct_change()
    
    # Integrate Momentum, Volume, and Price Change
    integrated_factor = (price_range_momentum * volume_change) + price_change
    
    # Calculate 5-day EMA for Close Price
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    price_momentum = close_ema_5.diff()
    
    # Calculate 5-day EMA for Volume
    volume_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    volume_acceleration = volume_ema_5.diff()
    
    # Combine Short-Term Factors
    short_term_factor = (volume_acceleration * price_momentum) + integrated_factor
    
    # Calculate 10-day EMA for Close Price
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = close_ema_10.diff()
    
    # Calculate 10-day EMA for Volume
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Volume Acceleration
    long_term_volume_acceleration = volume_ema_10.diff()
    
    # Combine Long-Term Factors
    long_term_factor = (long_term_volume_acceleration * long_term_price_momentum)
    
    # Final Alpha Factor
    final_alpha_factor = short_term_factor + long_term_factor
    
    return final_alpha_factor
```
