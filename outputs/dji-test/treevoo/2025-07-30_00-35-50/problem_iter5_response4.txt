```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].diff()
    
    # Calculate Price Gap
    price_gap = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    adjustment = np.where(price_gap > 0, -price_gap, price_gap)
    adjustment_decay = adjustment * np.exp(-1 / 1)  # Exponential decay with half-life of 1 day
    adjusted_daily_return = daily_return + adjustment_decay
    
    # Calculate Price Directionality
    C_O = df['close'] - df['open']
    H_L = df['high'] - df['low']
    price_directionality = C_O / H_L
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Combine Volume and Price Information
    combined_vol_price = volume_change * price_directionality
    
    # Adjust for Liquidity
    avg_volume = df['volume'].rolling(window=20).mean()  # Lookback period of 20 days
    liquidity_adjusted = combined_vol_price / avg_volume
    
    # Integrate Momentum, Volume Change, and Adjusted Daily Return
    integrated_momentum = price_range * volume_change + adjusted_daily_return
    
    # Incorporate Price Momentum
    five_day_return = df['close'] - df['close'].shift(5)
    final_alpha_factor = integrated_momentum + five_day_return + liquidity_adjusted
    
    return final_alpha_factor
```
