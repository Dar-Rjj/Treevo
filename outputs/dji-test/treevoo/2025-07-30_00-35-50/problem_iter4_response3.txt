```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_to_low_range'] - df['high_to_low_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['factor'] = df['price_range_momentum'] * df['volume_change']
    
    return df['factor']
```
