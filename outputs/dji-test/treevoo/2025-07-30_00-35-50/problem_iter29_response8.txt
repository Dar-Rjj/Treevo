```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap']
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap']
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Momentum
    df['close_ema_5'] = df['close'].ewm(span=5).mean()
    df['price_momentum'] = df['close_ema_5'] - df['close_ema_5'].shift(1)
    
    # Calculate High-to-Low Price Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Volume Change and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_momentum'] = (df['high_low_range_momentum'] * df['volume_change']) + df['price_change']
    
    # Calculate Amount Acceleration
    df['amount_ema_5'] = df['amount'].ewm(span=5).mean()
    df['amount_acceleration'] = df['amount_ema_5'] - df['amount_ema_5'].shift(1)
    
    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['combined_high_low_spread'] = df['high_low_range'] * df['volume_change']
    df['combined_high_low_spread'] = df['combined_high_low_spread'].where(df['volume_change'] > 0, 0)
    df['combined_high_low_spread'] *= df['amount_acceleration']
    
    # Smoothing Filter
    df['sma_10'] = df['combined_high_low_spread'].rolling(window=10).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['integrated_momentum_smoothed'] = df['price_momentum'] * df['sma_10']
    
    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Ratio
    df['final_factor'] = df['adjusted_daily_return'] * df['integrated_momentum_smoothed'] * df['volume_ratio']
    
    # Ensure Positive Factor
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))
    
    return df['final_factor']
```
