import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].diff()
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df.apply(
        lambda row: row['Daily_Return'] - row['Price_Gap'] if row['Price_Gap'] > 0 else row['Daily_Return'] + row['Price_Gap'],
        axis=1
    )
    
    # Evaluate Momentum Over Time
    df['Momentum_10D'] = df['Adjusted_Daily_Return'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['Volume_10D_Avg'] = df['volume'].rolling(window=10).mean()
    df['Volume_Weighted_Momentum'] = (df['Momentum_10D'] * df['volume']) / df['Volume_10D_Avg']
    
    # Adjust for Volume Spike Indicator
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Volume_10D_Avg']).astype(int)
    df['Volume_Weighted_Momentum_Adjusted'] = df['Volume_Weighted_Momentum'] * (1 + 0.2 * df['Volume_Spike'])
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'].diff()
    df['Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * (df['Volume_Change'] / df['Volume_10D_Avg'])
    
    # Adjust for Volatility
    df['Historical_Volatility'] = df['Adjusted_Daily_Return'].rolling(window=10).std()
    df['Volatility_Adjustment'] = (np.abs(df['Adjusted_Daily_Return']) > 2 * df['Historical_Volatility']).astype(int)
    df['Volume_Weighted_Momentum_Final'] = df['Volume_Weighted_Momentum_Adjusted'] * (1 + 0.3 * df['Volatility_Adjustment'])
    
    # Adjust for Long-Term Price Momentum
    df['Long_Term_Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Combine All Adjustments
    df['Alpha_Factor'] = (
        df['Long_Term_Price_Momentum'] *
        df['Volume_Weighted_Momentum_Final'] *
        df['Adjusted_High_Low_Spread'] *
        df['Volume_Trend']
    )
    
    # Apply Softmax to Ensure Positive Factor
    df['Alpha_Factor'] = np.exp(df['Alpha_Factor']) / (1 + np.exp(df['Alpha_Factor']))
    
    return df['Alpha_Factor']
