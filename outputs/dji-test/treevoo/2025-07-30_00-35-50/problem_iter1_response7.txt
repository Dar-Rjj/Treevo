```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_adjustment'] = df['volume'] - df['volume_moving_avg']
    
    # Combine Price Momentum and Volume Adjustment
    df['factor'] = df['price_momentum'] * df['volume_adjustment']
    
    # Ensure Positive Factor
    df['factor'] = df['factor'].apply(lambda x: max(x, 0))
    
    return df['factor']
```
