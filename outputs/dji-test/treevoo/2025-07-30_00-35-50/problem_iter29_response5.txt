```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    df['volume_moving_average'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = (df['volume'] > df['volume_moving_average'] * volume_spike_threshold).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + 0.2 * df['volume_spike'])
    
    # Adjust for Volatility
    volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=volatility_window).std()
    df['volatility_multiplier'] = (np.abs(df['daily_price_change']) > 2 * df['historical_volatility']).astype(int)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + 0.3 * df['volatility_multiplier'])
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_window)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    
    # Combine All Adjustments
    df['combined_adjustments'] = df['long_term_momentum'] * df['volume_weighted_momentum'] * df['volume_trend']
    df['combined_adjustments'] = np.where(df['combined_adjustments'] < 0, 
                                          np.exp(df['combined_adjustments']) / (1 + np.exp(df['combined_adjustments'])), 
                                          df['combined_adjustments'])
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['daily_price_change']
    
    # Calculate 5-day EMA for Close Price
    ema_window = 5
    df['ema_close'] = df['close'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_close'] - df['ema_close'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume'] - df['ema_volume'].shift(1)
    
    # Combine Short-Term Factors
    df['short_term_factors'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_momentum']
    
    # Incorporate Open Price into the Factor
    df['open_momentum'] = df['open'] - df['open'].shift(1)
    df['short_term_factors'] += df['open_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['short_term_factors'] * df['volume_trend'] + df['long_term_momentum']
    df['final_alpha_factor'] = np.where(df['final_alpha_factor'] < 0, 
                                        np.exp(df['final_alpha_factor']) / (1 + np.exp(df['final_alpha_factor'])), 
                                        df['final_alpha_factor'])
    
    return df['final_alpha_factor']
```
