```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = -row['price_gap']
        else:
            adjustment = row['price_gap']
        decay_factor = 0.5  # half-life of 1 day
        return row['daily_return'] + adjustment * decay_factor
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Adjust by Volume Change
    df['volume_weighted_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume-Weighted and Price Range Momentum
    df['volume_weighted_adjusted_daily_return'] = df['adjusted_daily_return'] * df['volume']
    
    # Final Alpha Factor with Volume Weighting
    df['rolling_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['alpha_factor'] = (df['volume_weighted_adjusted_daily_return'] + df['volume_weighted_price_range_momentum']) / df['rolling_volume_sum']
    
    return df['alpha_factor']
```
