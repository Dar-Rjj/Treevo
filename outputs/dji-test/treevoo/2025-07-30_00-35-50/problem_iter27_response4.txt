```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor using open, high, low, close, and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of the calculated alpha factor, indexed by date.
    """
    # Calculate the daily price range
    price_range = df['high'] - df['low']
    
    # Calculate the daily price change
    price_change = df['close'] - df['open']
    
    # Calculate the normalized volume
    average_volume = df['volume'].rolling(window=20).mean()
    normalized_volume = df['volume'] / average_volume
    
    # Calculate the directional movement
    positive_directional_movement = df['high'].diff().fillna(0)
    negative_directional_movement = df['low'].diff().fillna(0)
    directional_movement = positive_directional_movement - negative_directional_movement
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_change / price_range) * normalized_volume * directional_movement
    
    return alpha_factor
```
