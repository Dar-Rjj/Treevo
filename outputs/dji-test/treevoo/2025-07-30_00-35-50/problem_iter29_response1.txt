```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['price_gap'] = df['open'] - df['close'].shift(1)
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['adjusted_daily_return'] = df['daily_return'] * np.exp(-1 / 1) + df['price_gap']
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Adjusted High-Low Spread
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_high_low_spread'] = (df['high'] - df['low']) * np.where(df['volume_change'] > 0, df['volume_change'], 0)
    
    # Apply Simple Moving Average (SMA) to Adjusted High-Low Spread
    df['sma_adjusted_high_low_spread'] = df['adjusted_high_low_spread'].rolling(window=5).mean()
    
    # Calculate 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_trend_adjustment'] = df['volume'] - df['10_day_avg_volume']
    
    # Combine Price Momentum and Volume Trend Adjustment
    df['combined_price_vol_momentum'] = df['10_day_price_momentum'] * df['volume_trend_adjustment']
    df['combined_price_vol_momentum'] = np.where(df['combined_price_vol_momentum'] < 0, 0, df['combined_price_vol_momentum'])
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    df['positive_return_days'] = df['daily_log_return'] > 0
    df['negative_return_days'] = df['daily_log_return'] <= 0
    
    # Calculate Sum of Upward and Downward Volumes
    df['upward_volume'] = df['volume'] * df['positive_return_days']
    df['downward_volume'] = df['volume'] * df['negative_return_days']
    
    # Compute Trend Reversal Signal
    total_volume = df['volume'].rolling(window=10).sum()
    upward_volume_ratio = df['upward_volume'].rolling(window=10).sum() / total_volume
    downward_volume_ratio = df['downward_volume'].rolling(window=10).sum() / total_volume
    df['trend_reversal_signal'] = np.where(upward_volume_ratio > downward_volume_ratio, 1, -1) * df['daily_log_return']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_daily_return'] + 
        df['price_range_momentum'] + 
        df['sma_adjusted_high_low_spread'] + 
        df['combined_price_vol_momentum'] + 
        df['trend_reversal_signal']
    )
    df['final_alpha_factor'] = np.where(df['final_alpha_factor'] < 0, 0, df['final_alpha_factor'])
    
    return df['final_alpha_factor']
```
