```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    df['price_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_daily_return'] = df['daily_return'] - df['price_gap'] * (df['price_gap'] > 0) + df['price_gap'] * (df['price_gap'] < 0)

    # Calculate Price Range Momentum
    df['current_range'] = df['high'] - df['low']
    df['previous_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['price_range_momentum'] = df['current_range'] - df['previous_range']

    # Volume Change Factor
    df['volume_change_factor'] = df['volume'] / df['volume'].shift(1)

    # Combine Factors
    df['alpha_factor'] = df['adjusted_daily_return'] * df['price_range_momentum'] * df['volume_change_factor']

    return df['alpha_factor']
```
