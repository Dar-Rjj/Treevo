import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['price_change'] = df['close'] / df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_factor'] = (df['price_range_momentum'] * df['volume_change']) + df['price_change']
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_price_5'] = df['close'].ewm(span=5).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_price_5'] - df['ema_price_5'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_volume_5'] = df['volume'].ewm(span=5).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume_5'] - df['ema_volume_5'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = (df['volume_acceleration'] * df['price_momentum']) + df['integrated_factor']
    
    # Final Alpha Factor
    alpha_factor = df['combined_factors']
    
    return alpha_factor
