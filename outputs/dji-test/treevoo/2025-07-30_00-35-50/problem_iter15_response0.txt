import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range - high_to_low_range.shift(1)
    
    # Adjust by Volume and Volume-Weighted Price Change
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_weighted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    integrated_factor = (price_range_momentum * volume_change) + volume_weighted_price_change
    
    # Confirm with Volume
    combined_factor = integrated_factor * volume_change
    combined_factor = combined_factor.where(volume_change > 0, 0)
    
    # Smoothing Filter
    sma_period = 5
    final_alpha_factor = combined_factor.rolling(window=sma_period).mean()
    
    return final_alpha_factor
