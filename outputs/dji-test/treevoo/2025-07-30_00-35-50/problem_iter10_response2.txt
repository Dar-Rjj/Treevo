```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df.apply(
        lambda row: row['daily_return'] - row['price_gap'] if row['price_gap'] > 0 else row['daily_return'] + row['price_gap'],
        axis=1
    )
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['volume_adjusted_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Integrate Adjusted Daily Return and Price Range Momentum
    df['alpha_factor'] = df['adjusted_daily_return'] + df['volume_adjusted_price_range_momentum']
    
    # Final Alpha Factor
    alpha_factor = df['alpha_factor']
    
    return alpha_factor
```
