```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    decay_factor = 0.5  # half-life of 1 day
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * np.exp(-decay_factor)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * np.exp(-decay_factor)

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] / df['high_low_range'].shift(1)

    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Combine High-Low Spread with Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['high_low_spread_vol_adjusted'] = df['high_low_spread'] * (df['volume_change'] > 0) * df['volume_change']

    # Apply Simple Moving Average (SMA) to High-Low Spread with Volume Change
    df['sma_high_low_spread_vol_adjusted'] = df['high_low_spread_vol_adjusted'].rolling(window=5).mean()

    # Adjust by Volume Change
    df['volume_weighted_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']

    # Combine Volume-Weighted and Price Range Momentum
    df['volume_weighted_daily_return'] = df['adjusted_daily_return'] * df['volume']

    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = (df['volume_weighted_daily_return'] + df['volume_weighted_price_range_momentum']) / df['volume'].rolling(window=5).sum()

    # Integrate Volume-Weighted Price Range Momentum and Adjusted Daily Return
    df['final_alpha_factor'] = df['final_alpha_factor'] + df['sma_high_low_spread_vol_adjusted']

    return df['final_alpha_factor']
```
