```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time
    window = 10
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['sum_daily_price_changes'] * df['volume']) / df['volume'].rolling(window=window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_ma = df['volume'].rolling(window=window).mean()
    spike_threshold = 1.5
    df['volume_spike_multiplier'] = np.where(df['volume'] > spike_threshold * volume_ma, 1.2, 1.0)
    df['volume_weighted_momentum'] *= df['volume_spike_multiplier']
    
    # Adjust for Volatility
    historical_volatility = df['daily_price_change'].rolling(window=window).std()
    volatility_threshold = 2.0
    df['volatility_multiplier'] = np.where(np.abs(df['daily_price_change']) > volatility_threshold * historical_volatility, 1.3, 1.0)
    df['volume_weighted_momentum'] *= df['volatility_multiplier']
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_price_momentum'] = df['close'] - df['close'].shift(long_term_window)
    
    # Adjust for Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    
    # Combine All Adjustments
    factor = df['long_term_price_momentum'] * df['volume_weighted_momentum'] * volume_trend
    
    # Apply Softmax to Ensure Positive Factor
    factor = np.exp(factor) / (1 + np.exp(factor))
    
    return factor
```
