```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (10-day window)
    momentum_window = 10
    df['momentum'] = df['price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    df['volume_spike'] = df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    df['volume_weighted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * 1.2 if row['volume_spike'] > volume_threshold else row['volume_weighted_momentum'], axis=1)
    
    # Adjust for Volatility
    volatility_window = 10
    df['volatility'] = df['price_change'].rolling(window=volatility_window).std()
    df['abs_price_change'] = df['price_change'].abs()
    df['volume_weighted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * 1.3 if row['abs_price_change'] > 2 * row['volatility'] else row['volume_weighted_momentum'], axis=1)
    
    # Adjust for Long-Term Price Momentum
    long_term_momentum_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_momentum_window)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_momentum_window).mean()
    
    # Combine All Adjustments
    df['adjusted_momentum'] = df['volume_weighted_momentum'] * df['long_term_momentum'] * df['volume_trend']
    df['adjusted_momentum'] = df['adjusted_momentum'].apply(lambda x: np.exp(x) / (1 + np.exp(x)) if x < 0 else x)
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    df['prev_price_range'] = df['price_range'].shift(1)
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] - df['prev_price_range']
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['integrated_factor'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Calculate 5-day EMA for Close Price
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Short-Term Factors
    df['short_term_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_factor']
    
    # Calculate 10-day EMA for Close Price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['long_term_price_momentum'] = df['ema_10_close'] - df['ema_10_close'].shift(1)
    
    # Calculate 10-day EMA for Volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['long_term_volume_acceleration'] = df['ema_10_volume'] - df['ema_10_volume'].shift(1)
    
    # Combine Long-Term Factors
    df['long_term_factor'] = df['long_term_volume_acceleration'] * df['long_term_price_momentum'] + df['short_term_factor']
    
    # Final Alpha Factor
    alpha_factor = df['long_term_factor']
    
    return alpha_factor
```
