import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time
    window = 10
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=window).sum()
    
    # Calculate Average Volume Over Time
    df['sum_volume'] = df['volume'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['sum_daily_price_changes'] * df['sum_volume']) / df['sum_volume']
    
    # Integrate Volume Spike Indicator
    df['volume_moving_average'] = df['volume'].rolling(window=window).mean()
    threshold = 1.5
    df['volume_spike'] = df['volume'] > (threshold * df['volume_moving_average'])
    
    # Adjust Volume Weighted Momentum based on Volume Spike
    multiplier = 1.2
    df['adjusted_volume_weighted_momentum'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * multiplier if row['volume_spike'] else row['volume_weighted_momentum'],
        axis=1
    )
    
    # Return the factor values
    return df['adjusted_volume_weighted_momentum'].dropna()
