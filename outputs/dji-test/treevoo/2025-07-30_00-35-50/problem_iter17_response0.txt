```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    half_life = 1
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'].apply(lambda x: x * np.exp(-1 / half_life))
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'].apply(lambda x: -x * np.exp(-1 / half_life))

    # Evaluate Momentum Over Time
    window = 10
    df['momentum'] = df['daily_return'].rolling(window=window).sum()

    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=window).mean()

    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    volume_multiplier = 1.2
    df['volume_sma'] = df['volume'].rolling(window=window).mean()
    df['volume_spike_indicator'] = (df['volume'] > volume_threshold * df['volume_sma']).astype(float)
    df['volume_weighted_momentum'] *= (1 + (df['volume_spike_indicator'] * (volume_multiplier - 1)))

    # Adjust for Volatility
    volatility_threshold = 2
    volatility_multiplier = 1.3
    df['historical_volatility'] = df['daily_return'].rolling(window=window).std()
    df['volatility_adjustment'] = (np.abs(df['close'] - df['close'].shift(1)) > volatility_threshold * df['historical_volatility']).astype(float)
    df['volume_weighted_momentum'] *= (1 + (df['volatility_adjustment'] * (volatility_multiplier - 1)))

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Integrate Price Range Momentum with Volume Change
    df['integrated_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']

    # Combine Adjusted Daily Return with Price Range Momentum
    df['combined_factor'] = (df['adjusted_daily_return'] * df['volume']) + df['integrated_price_range_momentum']

    # Calculate 5-day Exponential Moving Average for Price
    ema_window = 5
    df['ema_price'] = df['close'].ewm(span=ema_window, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_price'] - df['ema_price'].shift(1)

    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_window, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_volume'] - df['ema_volume'].shift(1)

    # Combine Factors
    df['final_factor'] = (df['volume_acceleration'] * df['price_momentum']) + df['combined_factor'] + (df['volume_weighted_momentum'] * df['price_momentum'])

    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = df['final_factor'] / df['volume'].rolling(window=ema_window).sum()

    # Ensure Positive Factor
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))

    return df['final_alpha_factor']
```
