```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine High-Low Spread with Volume Change
    combined_factor = high_low_spread * (volume_change > 0) * volume_change
    
    # Apply Simple Moving Average (SMA) with a period of 5 days
    sma_5_days = combined_factor.rolling(window=5).mean()
    
    return sma_5_days
```
