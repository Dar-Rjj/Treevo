import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['Price_Gap'] > 0:
            adjustment = row['Price_Gap']
        else:
            adjustment = -row['Price_Gap']
        return row['Daily_Return'] - adjustment * np.exp(-1)
    
    df['Adjusted_Daily_Return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate Price and High-to-Low Range Momentum
    df['Close_EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Price_Momentum'] = df['Close_EMA_5'] - df['Close_EMA_5'].shift(1)
    
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_EMA_5'] = df['amount'].ewm(span=5, adjust=False).mean()
    df['Amount_Acceleration'] = df['Amount_EMA_5'] - df['Amount_EMA_5'].shift(1)
    
    df['Combined_High_Low_Spread'] = df['High_Low_Range'] * df['Volume_Change']
    df['Combined_High_Low_Spread'] = df['Combined_High_Low_Spread'] * df['Amount_Acceleration'].apply(lambda x: max(x, 0))
    
    # Smoothing Filter
    df['Smoothed_Volume_Adjusted_High_Low_Spread'] = df['Combined_High_Low_Spread'].rolling(window=10).mean()
    
    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['Integrated_Momentum'] = (df['Price_Momentum'] + df['High_Low_Range_Momentum']) * df['Smoothed_Volume_Adjusted_High_Low_Spread']
    
    # Adjust by Volume Weighted Momentum
    window = 10
    df['Sum_Daily_Returns'] = df['Daily_Return'].rolling(window=window).sum()
    df['Average_Volume'] = df['volume'].rolling(window=window).mean()
    df['Volume_Weighted_Momentum'] = df['Sum_Daily_Returns'] * df['volume'] / df['Average_Volume']
    
    # Adjust for Volume Spike Indicator
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Spike_Indicator'] = df['volume'] / df['Volume_MA_10']
    df['Volume_Weighted_Momentum'] = df['Volume_Weighted_Momentum'] * df['Volume_Spike_Indicator'].apply(lambda x: 1.2 if x > 1.5 else 1)
    
    # Adjust for Volatility
    df['Historical_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    df['Abs_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Volatility_Adjustment'] = df['Abs_Price_Change'] / df['Historical_Volatility']
    df['Volume_Weighted_Momentum'] = df['Volume_Weighted_Momentum'] * df['Volatility_Adjustment'].apply(lambda x: 1.3 if x > 2 else 1)
    
    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Weighted Momentum
    df['Alpha_Factor'] = df['Adjusted_Daily_Return'] * df['Integrated_Momentum'] * df['Volume_Weighted_Momentum']
    
    # Final Alpha Factor Integration
    df['Alpha_Factor'] = df['Alpha_Factor'].apply(lambda x: max(x, 0))
    
    return df['Alpha_Factor']
