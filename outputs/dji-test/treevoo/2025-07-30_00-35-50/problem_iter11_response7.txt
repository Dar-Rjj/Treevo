```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df.apply(lambda row: row['daily_return'] - row['price_gap'] if row['price_gap'] > 0 else row['daily_return'] + row['price_gap'], axis=1)
    
    # Calculate Current Period High-to-Low Range
    df['current_range'] = df['high'] - df['low']
    
    # Calculate Previous Period High-to-Low Range
    df['prev_range'] = df['current_range'].shift(1)
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['current_range'] - df['prev_range']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Adjusted Daily Return, Price Range Momentum, and Volume Change
    df['combined_factor'] = (df['adjusted_daily_return'] * df['range_momentum']) * df['volume_change']
    
    # Apply Softmax to Ensure Positive Factor
    df['factor'] = df['combined_factor'].apply(lambda x: np.exp(x) / (1 + np.exp(x)) if x < 0 else x)
    
    return df['factor']
```
