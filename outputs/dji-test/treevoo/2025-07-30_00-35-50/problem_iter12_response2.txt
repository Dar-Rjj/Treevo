```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap with Exponential Decay
    half_life = 1
    df['decay_factor'] = np.exp(-np.log(2) / half_life)
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * df['decay_factor']
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * df['decay_factor']
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].pct_change()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Integrate Momentum and Volume Change
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume Weighted and Price Range Momentum
    df['volume_weighted_return'] = df['adjusted_daily_return'] * df['volume']
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_price'].diff()
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'].diff()
    
    # Combine Factors
    df['combined_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_momentum'] + df['volume_weighted_return']
    
    # Calculate 20-day Exponential Moving Average for Price
    df['ema_20_price'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_price_momentum'] = df['ema_20_price'].diff()
    
    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = (df['combined_factor'] + df['long_term_price_momentum']) / df['volume'].rolling(window=5).sum()
    
    return df['final_alpha_factor']
```
