```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['price_range'] - df['price_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Range Momentum with Volume Change
    df['integrated_momentum'] = df['range_momentum'] * df['volume_change']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    decay_factor = 0.5  # half-life of 1 day
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * np.exp(-1/decay_factor)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * np.exp(-1/decay_factor)
    
    # Combine Adjusted Daily Return with Price Range Momentum
    df['combined_factor'] = (df['adjusted_daily_return'] * df['volume']) + df['integrated_momentum']
    
    # Final Alpha Factor
    df['last_5_days_volume'] = df['volume'].rolling(window=5).sum()
    df['alpha_factor'] = df['combined_factor'] / df['last_5_days_volume']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
```
