```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Multiply High-Low Spread by Volume Change
    combined_signal = high_low_spread * volume_change
    
    # Apply SMA (5 days)
    smoothed_signal = combined_signal.rolling(window=5).mean()
    
    return smoothed_signal
```
