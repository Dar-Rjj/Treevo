```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    
    # Combine High-Low Spread with Volume Change
    combined_high_low = high_low_spread * (volume_change > 0) * volume_change
    
    # Incorporate Price Trend
    price_change = df['close'].diff()
    
    # Integrate Combined High-Low Spread, Volume Change, and Price Change
    integrated_high_low = combined_high_low * (price_change.abs())
    
    # Apply Exponential Moving Average (EMA)
    ema_factor = integrated_high_low.ewm(alpha=0.2).mean()
    
    return ema_factor
```
