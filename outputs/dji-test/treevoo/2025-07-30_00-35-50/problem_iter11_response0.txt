```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * np.exp(-1)
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * np.exp(-1)
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['price_range'] - df['price_range'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['integrated_range_momentum'] = df['range_momentum'] * df['volume_change']
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['integrated_close_momentum'] = df['close_momentum'] * (df['volume'] - df['avg_volume_10d'])
    
    # Combine Volume Weighted and Price Range Momentum
    df['volume_weighted_adjusted_return'] = df['adjusted_daily_return'] * df['volume']
    
    # Final Alpha Factor with Volume Weighting
    df['last_5d_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = (df['volume_weighted_adjusted_return'] + 
                                df['integrated_range_momentum'] + 
                                df['integrated_close_momentum']) / df['last_5d_volume_sum']
    
    # Ensure Positive Factor
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['final_alpha_factor']
```
