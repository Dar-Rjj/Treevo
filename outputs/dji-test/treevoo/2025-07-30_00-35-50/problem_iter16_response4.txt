```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            return row['daily_return'] - (row['price_gap'] * decay_factor)
        elif row['price_gap'] < 0:
            return row['daily_return'] + (row['price_gap'] * decay_factor)
        else:
            return row['daily_return']
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)

    # Calculate High-to-Low Price Range
    df['high_to_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_to_low_range'].pct_change()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()

    # Adjust by Volume Change
    df['volume_weighted_price_range_momentum'] = df['price_range_momentum'] * df['volume_change']

    # Combine Volume-Weighted and Price Range Momentum
    df['volume_weighted_adjusted_daily_return'] = df['adjusted_daily_return'] * df['volume']

    # Integrate Volume-Weighted Price Range Momentum and Adjusted Daily Return
    df['alpha_factor'] = df['volume_weighted_price_range_momentum'] + df['volume_weighted_adjusted_daily_return']

    return df['alpha_factor']
```
