```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 20-day moving average of the typical price
    typical_price_ma_20 = typical_price.rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the typical price
    typical_price_std_20 = typical_price.rolling(window=20).std()
    
    # Calculate the Z-score of the typical price
    z_score_typical_price = (typical_price - typical_price_ma_20) / typical_price_std_20
    
    # Calculate the 10-day exponentially weighted moving average of the volume
    ewm_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the volume ratio
    volume_ratio = df['volume'] / ewm_volume_10
    
    # Combine the Z-score of the typical price and the volume ratio to create the alpha factor
    alpha_factor = z_score_typical_price * volume_ratio
    
    return alpha_factor
```
