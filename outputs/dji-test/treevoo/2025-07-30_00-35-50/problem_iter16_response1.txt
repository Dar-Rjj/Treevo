```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range.pct_change()
    
    # Adjust by Volume and Price Change
    volume_change = df['volume'].pct_change()
    price_change = df['close'].pct_change()
    
    # Integrate Momentum, Volume, and Price Change
    integrated_factor = price_range_momentum * volume_change + price_change
    
    # Calculate 5-day EMA for Close Price
    ema_5_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = ema_5_close.diff()
    
    # Calculate 5-day EMA for Volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term Volume Acceleration
    short_term_volume_acceleration = ema_5_volume.diff()
    
    # Combine Short-Term Factors
    short_term_factor = short_term_volume_acceleration * short_term_price_momentum + integrated_factor
    
    # Calculate 10-day EMA for Close Price
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = ema_10_close.diff()
    
    # Calculate 10-day EMA for Volume
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Volume Acceleration
    long_term_volume_acceleration = ema_10_volume.diff()
    
    # Combine Long-Term Factors
    long_term_factor = long_term_volume_acceleration * long_term_price_momentum + short_term_factor
    
    # Final Alpha Factor
    final_alpha_factor = long_term_factor
    
    return final_alpha_factor
```
