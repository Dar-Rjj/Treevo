```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust High-Low Spread with Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_high_low_spread = high_low_spread * (volume_change / avg_volume)
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Spread, Volume, and Price
    factor = high_low_spread
    factor[(price_change > 0) & (volume_change > 0)] = adjusted_high_low_spread * (price_change + volume_change)
    
    return factor
```
