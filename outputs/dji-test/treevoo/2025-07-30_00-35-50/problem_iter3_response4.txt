import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['high_low_range'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Price Change
    df['price_change'] = df['close'].pct_change()
    
    # Integrate Momentum, Volume, and Price Change
    df['momentum_volume_price'] = (df['range_momentum'] * df['volume_change']) + df['price_change']
    
    # Final Alpha Factor
    alpha_factor = df['momentum_volume_price']
    
    return alpha_factor
