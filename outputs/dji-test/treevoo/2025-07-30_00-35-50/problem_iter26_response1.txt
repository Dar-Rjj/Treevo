import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].diff()
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap with Exponential Decay (half-life = 1 day)
    df['Adjusted_Return'] = df['Daily_Return'] - df['Price_Gap'].apply(lambda x: x * np.exp(-1))
    
    # Calculate High-to-Low Price Range
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['Price_Range_Momentum'] = df['Price_Range'] - df['Price_Range'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-Weighted Price Change
    df['Volume_Weighted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['Integrated_Factor'] = df['Price_Range_Momentum'] * df['Volume_Change'] + df['Volume_Weighted_Price_Change']
    
    # Combine Integrated Factor with Volume Change
    df['Combined_Factor'] = df['Integrated_Factor'] * np.where(df['Volume_Change'] > 0, df['Volume_Change'], 1)
    
    # Smoothing Filter (SMA over 5 days)
    df['Smoothed_Combined_Factor'] = df['Combined_Factor'].rolling(window=5).mean()
    
    # Calculate 5-day EMA for Price
    df['EMA_5_Price'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_5_Price'] - df['EMA_5_Price'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['EMA_5_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_5_Volume'] - df['EMA_5_Volume'].shift(1)
    
    # Combine Factors
    df['Combined_Factors'] = df['Volume_Acceleration'] * df['Price_Momentum'] + df['Smoothed_Combined_Factor']
    
    # Final Alpha Factor with Volume Weighting
    df['Final_Alpha_Factor'] = df['Combined_Factors'] / df['volume'].rolling(window=5).sum()
    df['Final_Alpha_Factor'] = np.where(df['Final_Alpha_Factor'] < 0, 0, df['Final_Alpha_Factor'])
    
    # Incorporate Trading Volume Volatility
    df['Volume_Volatility'] = df['volume'].rolling(window=5).std()
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] / df['Volume_Volatility']
    
    # Introduce Price Trend Indicator
    df['EMA_20_Price'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend'] = df['EMA_20_Price'] - df['EMA_20_Price'].shift(1)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Price_Trend']
    df['Final_Alpha_Factor'] = np.where(df['Final_Alpha_Factor'] < 0, 0, df['Final_Alpha_Factor'])
    
    return df['Final_Alpha_Factor'].dropna()
