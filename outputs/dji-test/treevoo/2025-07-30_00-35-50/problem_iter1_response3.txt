import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Calculate Average Volume Over Time (e.g., 10 days)
    df['sum_volume'] = df['volume'].rolling(window=momentum_window).sum()
    df['avg_volume'] = df['sum_volume'] / momentum_window
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['sum_daily_price_changes'] * df['sum_volume']) / df['avg_volume']
    
    # Identify Volume Spike
    df['volume_moving_avg'] = df['volume'].rolling(window=momentum_window).mean()
    volume_spike_threshold = 1.5
    df['volume_spike'] = df['volume'] > volume_spike_threshold * df['volume_moving_avg']
    
    # Integrate Volume Spike Indicator
    multiplier = 1.2
    df['volume_weighted_momentum_with_spike'] = df['volume_weighted_momentum'] * (df['volume_spike'] * (multiplier - 1) + 1)
    
    # Return the factor values
    return df['volume_weighted_momentum_with_spike']
