import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['price_gap'] > 0:
            adjustment = -row['price_gap']
        else:
            adjustment = row['price_gap']
        return row['daily_return'] + adjustment * np.exp(-1)
    
    df['adjusted_daily_return'] = df.apply(adjust_daily_return, axis=1)
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['momentum_volume_weighted'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume Weighted and Price Range Momentum
    df['final_alpha_factor'] = df['adjusted_daily_return'] * df['volume']
    
    # Return the final alpha factor
    return df['final_alpha_factor']
