```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return']
    decay_factor = 0.5
    for i in range(1, len(df)):
        if df['price_gap'].iloc[i] > 0:
            df['adjusted_daily_return'].iloc[i] -= df['price_gap'].iloc[i] * (decay_factor ** (i-1))
        elif df['price_gap'].iloc[i] < 0:
            df['adjusted_daily_return'].iloc[i] += df['price_gap'].iloc[i] * (decay_factor ** (i-1))
    
    # Evaluate Momentum Over Time
    window = 10
    df['sum_adjusted_daily_return'] = df['adjusted_daily_return'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['sum_adjusted_daily_return'] * df['volume'] / df['volume'].rolling(window=window).mean()
    
    # Adjust for Volume Spike Indicator
    df['volume_moving_avg'] = df['volume'].rolling(window=window).mean()
    df['volume_weighted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * 1.2 if row['volume'] > 1.5 * row['volume_moving_avg'] else row['volume_weighted_momentum'], axis=1)
    
    # Adjust for Volatility
    df['historical_volatility'] = df['adjusted_daily_return'].rolling(window=window).std()
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['volume_weighted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * 1.3 if row['abs_price_change'] > 2 * row['historical_volatility'] else row['volume_weighted_momentum'], axis=1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Combine Adjusted Daily Return with Integrated Momentum
    df['combined_momentum'] = df['adjusted_daily_return'] * df['volume'] + df['integrated_momentum']
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_close'] = df['close'].ewm(span=5).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_close'] - df['ema_5_close'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Factors
    df['factor'] = (df['volume_acceleration'] * df['price_momentum'] + 
                    df['combined_momentum'] + 
                    df['volume_weighted_momentum'] * df['price_momentum'])
    
    # Final Alpha Factor with Volume Weighting
    df['final_factor'] = df['factor'] / df['volume'].rolling(window=5).sum()
    
    # Ensure Positive Factor
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))
    
    return df['final_factor']
```
