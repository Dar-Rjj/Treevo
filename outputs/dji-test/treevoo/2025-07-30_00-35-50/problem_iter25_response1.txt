import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range / high_to_low_range.shift(1)
    
    # Adjust by Volume and Price Change
    volume_change = df['volume'] / df['volume'].shift(1)
    price_change = df['close'] / df['close'].shift(1) - 1
    
    # Integrate Momentum, Volume, and Price Change
    integrated_momentum = price_range_momentum * volume_change + price_change
    
    # Calculate Short-Term Factors
    short_ema_close = df['close'].ewm(span=5, adjust=False).mean()
    short_price_momentum = short_ema_close - short_ema_close.shift(1)
    
    short_ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    short_volume_acceleration = short_ema_volume - short_ema_volume.shift(1)
    
    short_term_factors = short_volume_acceleration * short_price_momentum + integrated_momentum
    
    # Calculate Long-Term Factors
    long_ema_close = df['close'].ewm(span=10, adjust=False).mean()
    long_price_momentum = long_ema_close - long_ema_close.shift(1)
    
    long_ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    long_volume_acceleration = long_ema_volume - long_ema_volume.shift(1)
    
    long_term_factors = long_volume_acceleration * long_price_momentum + short_term_factors
    
    # Calculate Medium-Term Factors
    medium_ema_close = df['close'].ewm(span=20, adjust=False).mean()
    medium_price_momentum = medium_ema_close - medium_ema_close.shift(1)
    
    medium_ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    medium_volume_acceleration = medium_ema_volume - medium_ema_volume.shift(1)
    
    medium_term_factors = medium_volume_acceleration * medium_price_momentum + long_term_factors
    
    # Final Alpha Factor
    final_alpha_factor = medium_term_factors
    
    return final_alpha_factor
