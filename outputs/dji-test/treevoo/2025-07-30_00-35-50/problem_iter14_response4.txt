import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time
    momentum_window = 10
    df['momentum_10d'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum_10d'] * df['volume']) / df['volume'].rolling(window=momentum_window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    df['volume_moving_avg'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = df['volume'] / df['volume_moving_avg']
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum'] * np.where(df['volume_spike'] > volume_spike_threshold, 1.2, 1)
    
    # Adjust for Volatility
    volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=volatility_window).std()
    df['price_change_volatility_adj'] = np.where(np.abs(df['daily_price_change']) > 2 * df['historical_volatility'], 1.3, 1)
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum_adj'] * df['price_change_volatility_adj']
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_price_momentum'] = df['close'] - df['close'].shift(long_term_window)
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum_adj'] * df['long_term_price_momentum']
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum_adj'] * df['volume_trend']
    
    # Apply Softmax to Ensure Positive Factor
    df['softmax_adj'] = np.exp(df['volume_weighted_momentum_adj']) / (1 + np.exp(df['volume_weighted_momentum_adj']))
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum_adj'] * df['softmax_adj']
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Calculate 5-day EMA for Close Price
    ema_short_window = 5
    df['ema_close_5d'] = df['close'].ewm(span=ema_short_window, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum_5d'] = df['ema_close_5d'] - df['ema_close_5d'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['ema_volume_5d'] = df['volume'].ewm(span=ema_short_window, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration_5d'] = df['ema_volume_5d'] - df['ema_volume_5d'].shift(1)
    
    # Combine Short-Term Factors
    df['short_term_factor'] = df['volume_acceleration_5d'] * df['price_momentum_5d'] + df['integrated_momentum']
