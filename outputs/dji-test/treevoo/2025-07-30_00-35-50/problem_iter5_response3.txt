```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'].pct_change()
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['volume_10d_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = df['volume'] - df['volume_10d_ma']
    
    # Adjust for Amount Trend
    df['amount_10d_ma'] = df['amount'].rolling(window=10).mean()
    df['amount_trend'] = df['amount'] - df['amount_10d_ma']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    df['integrated_range_momentum'] = df['price_range_momentum'] * df['volume_change']
    df['integrated_close_momentum'] = df['close_price_momentum'] * df['volume_trend']
    
    # Integrate Adjustments
    df['final_factor'] = (df['integrated_range_momentum'] + df['integrated_close_momentum']) * df['amount_trend']
    
    # Ensure Positive Factor
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))
    
    return df['final_factor']
```
