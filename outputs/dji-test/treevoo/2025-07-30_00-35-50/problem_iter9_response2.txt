```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'] / df['price_range'].shift(1)
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_momentum'] = (df['price_range_momentum'] * df['volume_change']) + df['price_change']
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_price'] - df['ema_5_price'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Factors
    df['short_term_factor'] = (df['volume_acceleration'] * df['price_momentum']) + df['integrated_momentum']
    
    # Calculate 10-day Exponential Moving Average for Price
    df['ema_10_price'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_price_momentum'] = df['ema_10_price'] - df['ema_10_price'].shift(1)
    
    # Calculate 10-day Exponential Moving Average for Volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Volume Acceleration
    df['long_term_volume_acceleration'] = df['ema_10_volume'] - df['ema_10_volume'].shift(1)
    
    # Long-Term Factor Combination
    df['long_term_factor'] = (df['long_term_volume_acceleration'] * df['long_term_price_momentum'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['short_term_factor'] + df['long_term_factor']
    
    return df['alpha_factor']
```
