```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjustment'] = 0
    df.loc[df['price_gap'] > 0, 'adjustment'] = df['price_gap'] * np.exp(-1)
    df.loc[df['price_gap'] < 0, 'adjustment'] = -df['price_gap'] * np.exp(-1)
    
    df['adjusted_daily_return'] = df['daily_return'] - df['adjustment']
    
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'].pct_change()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change().fillna(0)
    
    # Integrate Momentum and Volume Change
    df['momentum_volume'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume Weighted and Price Range Momentum
    df['volume_weighted_momentum'] = df['adjusted_daily_return'] * df['volume']
    
    # Final Alpha Factor with Volume Weighting
    df['last_5_days_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['volume_weighted_momentum'] / df['last_5_days_volume_sum']
    
    return df['final_alpha_factor']
```
