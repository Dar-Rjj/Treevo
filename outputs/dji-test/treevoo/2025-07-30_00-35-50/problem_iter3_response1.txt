import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Directionality
    df['close_minus_open'] = df['close'] - df['open']
    df['high_minus_low'] = df['high'] - df['low']
    df['price_directionality'] = df['close_minus_open'] / df['high_minus_low']
    
    # Combine Volume and Price Information
    df['volume_price_info'] = df['volume_change'] * df['price_directionality']
    
    # Adjust for Liquidity
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['adjusted_volume_price_info'] = df['volume_price_info'] / df['avg_volume']
    
    # Incorporate Price Momentum
    df['5_day_return'] = df['close'] - df['close'].shift(5)
    df['alpha_factor'] = df['adjusted_volume_price_info'] + df['5_day_return']
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
