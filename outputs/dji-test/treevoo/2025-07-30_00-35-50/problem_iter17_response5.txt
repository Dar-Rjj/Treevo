import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Multiply High-Low Spread by Volume Change if Volume Change > 0
    combined_factor = high_low_range * volume_change.where(volume_change > 0, 0)
    
    # Apply Simple Moving Average (SMA) to Combined Factor
    sma_combined_factor = combined_factor.rolling(window=5).mean()
    
    # Calculate Price Range Momentum
    price_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Calculate Volume-Weighted Price Change
    price_change = df['close'] - df['close'].shift(1)
    volume_weighted_price_change = price_change * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    integrated_factor = (price_range_momentum * volume_change) + volume_weighted_price_change
    
    # Confirm with Volume
    confirmed_factor = integrated_factor * volume_change.where(volume_change > 0, 0)
    
    # Smoothing Filter: Apply Simple Moving Average (SMA)
    final_alpha_factor = confirmed_factor.rolling(window=5).mean()
    
    return final_alpha_factor
