```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['daily_return'] = df['close'].pct_change()
    df['price_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_daily_return'] = df['daily_return'] + df['price_gap'] * np.exp(-1)

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].pct_change()

    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Adjust for Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * np.where(df['volume_change'] > 0, df['volume_change'], 0)

    # Apply Simple Moving Average (SMA) to High-Low Spread with Volume Change
    df['sma_high_low_spread'] = df['volume_adjusted_high_low_spread'].rolling(window=5).mean()

    # Calculate 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)

    # Adjust for Volume Trend
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_trend_adjustment'] = df['volume'] - df['10_day_avg_volume']
    df['combined_price_momentum_vol_trend'] = df['10_day_price_momentum'] * df['volume_trend_adjustment']
    df['positive_factor'] = df['combined_price_momentum_vol_trend'].apply(lambda x: max(x, 0))

    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Identify Positive and Negative Returns
    positive_returns = df[df['log_return'] > 0]
    negative_returns = df[df['log_return'] <= 0]

    # Calculate Sum of Upward Volume
    sum_upward_volume = positive_returns['volume'].sum()

    # Calculate Sum of Downward Volume
    sum_downward_volume = negative_returns['volume'].sum()

    # Compute Trend Reversal Signal
    total_volume = sum_upward_volume + sum_downward_volume
    upward_volume_ratio = sum_upward_volume / total_volume
    downward_volume_ratio = sum_downward_volume / total_volume
    trend_reversal_signal = np.where(upward_volume_ratio > downward_volume_ratio, df['log_return'], 0)

    # Calculate Accumulated Volume Impact
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    df['volume_weighted_daily_return'] = df['daily_return'] * df['accumulated_volume']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_daily_return'] +
        df['sma_high_low_spread'] +
        df['positive_factor'] +
        trend_reversal_signal +
        df['volume_weighted_daily_return']
    )
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: max(x, 0))

    return df['final_alpha_factor']
```
