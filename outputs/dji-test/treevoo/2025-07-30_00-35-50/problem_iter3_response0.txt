```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    price_range_momentum = high_to_low_range / high_to_low_range.shift(1)
    
    # Adjust by Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    
    # Incorporate Close Price Trend
    close_price_trend = df['close'] / df['close'].shift(1)
    
    # Integrate Momentum, Volume Change, and Close Price Trend
    alpha_factor = (price_range_momentum * volume_change * close_price_trend).dropna()
    
    return alpha_factor
```
