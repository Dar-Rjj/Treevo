```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    adjustment = df['price_gap'].apply(lambda x: x if x < 0 else -x)
    adjustment = adjustment * np.exp(-1)  # Apply Exponential Decay with half-life of 1 day
    df['adjusted_daily_return'] = df['daily_return'] - adjustment
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-Weighted Price Change
    df['volume_weighted_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['integrated_factor'] = df['range_momentum'] * df['volume_change'] + df['volume_weighted_price_change']
    
    # Combine Integrated Factor with Volume Change
    df['combined_factor'] = df['integrated_factor'] * df['volume_change'].apply(lambda x: max(x, 0))
    
    # Smoothing Filter with SMA
    df['sma_combined_factor'] = df['combined_factor'].rolling(window=5).mean()
    
    # Calculate 5-day EMA for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_price'] - df['ema_5_price'].shift(1)
    
    # Calculate 5-day EMA for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = df['volume_acceleration'] * df['price_momentum'] + df['integrated_factor']
    
    # Final Alpha Factor with Volume Weighting
    df['final_alpha'] = df['combined_factors'] / df['volume'].rolling(window=5).sum()
    df['final_alpha'] = df['final_alpha'].apply(lambda x: max(x, 0))
    
    # Incorporate Trading Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=5).std()
    df['final_alpha_adjusted'] = df['final_alpha'] / df['volume_volatility']
    
    # Introduce Price Trend Indicator
    df['ema_20_price'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = df['ema_20_price'] - df['ema_20_price'].shift(1)
    df['final_alpha_trend'] = df['final_alpha_adjusted'] * df['price_trend']
    df['final_alpha_trend'] = df['final_alpha_trend'].apply(lambda x: max(x, 0))
    
    return df['final_alpha_trend']
```
