```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    df['Adjusted_Daily_Return'] = df['Daily_Return'] - df['Price_Gap']
    df['Adjusted_Daily_Return'] = df['Adjusted_Daily_Return'] * np.exp(-1 / 1)  # Exponential decay with half-life of 1 day

    # Evaluate Momentum Over Time
    window = 10
    df['Sum_Adjusted_Returns'] = df['Adjusted_Daily_Return'].rolling(window=window).sum()

    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = (df['Sum_Adjusted_Returns'] * df['volume']) / df['volume'].rolling(window=window).mean()

    # Adjust for Volume Spike Indicator
    volume_spike_threshold = 1.5
    df['Volume_MA'] = df['volume'].rolling(window=window).mean()
    df['Volume_Spike'] = (df['volume'] > df['Volume_MA'] * volume_spike_threshold).astype(float)
    df['Volume_Weighted_Momentum'] *= (1 + df['Volume_Spike'] * 0.2)

    # Adjust for Volatility
    df['Historical_Volatility'] = df['Adjusted_Daily_Return'].rolling(window=window).std()
    volatility_threshold = 2
    df['Volatility_Adjustment'] = (np.abs(df['close'] - df['close'].shift(1)) > df['Historical_Volatility'] * volatility_threshold).astype(float)
    df['Volume_Weighted_Momentum'] *= (1 + df['Volatility_Adjustment'] * 0.3)

    # Calculate Price and High-to-Low Range Momentum
    df['Close_EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Price_Momentum'] = df['Close_EMA_5'] - df['Close_EMA_5'].shift(1)
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)

    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_EMA_5'] = df['amount'].ewm(span=5, adjust=False).mean()
    df['Amount_Acceleration'] = df['Amount_EMA_5'] - df['Amount_EMA_5'].shift(1)
    df['Combined_High_Low_Spread'] = df['High_Low_Momentum'] * df['Volume_Change'].apply(lambda x: max(x, 0))
    df['Combined_High_Low_Spread'] = df['Combined_High_Low_Spread'] * df['Amount_Acceleration']

    # Smoothing Filter
    df['SMA_Combined_Spread'] = df['Combined_High_Low_Spread'].rolling(window=window).mean()

    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['Integrated_Momentum'] = df['Price_Momentum'] * df['SMA_Combined_Spread']

    # Incorporate Price Confirmation
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Price_Confirmed'] = df['High_Low_Momentum'] * (df['Price_Change'] + df['Volume_Change'])
    df['Price_Confirmed'] = df['Price_Confirmed'].apply(lambda x: max(x, 0))

    # Combine Factors
    df['Alpha_Factor'] = df['Adjusted_Daily_Return'] * df['Integrated_Momentum'] * df['Volume_Weighted_Momentum']

    # Ensure Positive Factor
    df['Alpha_Factor'] = df['Alpha_Factor'].apply(lambda x: np.exp(x) / (1 + np.exp(x)))

    return df['Alpha_Factor']
```
