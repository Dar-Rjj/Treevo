```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    momentum_window = 10
    df['momentum'] = df['daily_price_change'].rolling(window=momentum_window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['momentum'] * df['volume'] / df['volume'].rolling(window=momentum_window).mean()
    
    # Integrate Volume Spike Indicator
    volume_spike_threshold = 1.5
    volume_spike_multiplier = 1.2
    df['volume_moving_avg'] = df['volume'].rolling(window=momentum_window).mean()
    df['volume_spike'] = (df['volume'] > volume_spike_threshold * df['volume_moving_avg']).astype(float) * (volume_spike_multiplier - 1)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + df['volume_spike'])
    
    # Adjust for Volatility
    volatility_window = 10
    df['historical_volatility'] = df['daily_price_change'].rolling(window=volatility_window).std()
    volatility_threshold = 2
    volatility_multiplier = 1.3
    df['volatility_adjustment'] = (df['daily_price_change'].abs() > volatility_threshold * df['historical_volatility']).astype(float) * (volatility_multiplier - 1)
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'] * (1 + df['volatility_adjustment'])
    
    # Return the final factor
    return df['volume_weighted_momentum'].dropna()
```
