```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df['daily_return'] - df['price_gap']
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    df['adjusted_daily_return'] = df['adjusted_daily_return'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'].pct_change()

    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['volume_adjusted_price_range_momentum'] = df['price_range_momentum'] * (1 + df['volume_change'])

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_daily_return'] + df['volume_adjusted_price_range_momentum']

    return df['final_alpha_factor'].dropna()
```
