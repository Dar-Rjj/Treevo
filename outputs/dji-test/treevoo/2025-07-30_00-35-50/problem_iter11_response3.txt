```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap with Exponential Decay
    def exponential_decay(x, half_life=1):
        return x * 0.5 ** (1 / half_life)
    
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap'].apply(exponential_decay)
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += (-df['Price_Gap']).apply(exponential_decay)

    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']

    # Calculate Price Range Momentum
    df['Price_Range_Momentum'] = df['High_Low_Range'].pct_change()

    # Calculate Volume Change and Price Change
    df['Volume_Change'] = df['volume'].pct_change()
    df['Price_Change'] = df['close'].pct_change()

    # Integrate Momentum, Volume, and Price Change
    df['Integrated_Factor'] = df['Price_Range_Momentum'] * df['Volume_Change'] + df['Price_Change']

    # Combine Volume Weighted and Price Range Momentum
    df['Volume_Weighted_Adjusted_Return'] = df['Adjusted_Daily_Return'] * df['volume']

    # Calculate 5-day EMA for Close Price
    df['EMA_5_Close'] = df['close'].ewm(span=5).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['EMA_5_Close'].pct_change()

    # Calculate 5-day EMA for Volume
    df['EMA_5_Volume'] = df['volume'].ewm(span=5).mean()

    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['EMA_5_Volume'].pct_change()

    # Short-Term Factor Combination
    df['Short_Term_Factor'] = df['Volume_Acceleration'] * df['Price_Momentum'] + df['Integrated_Factor']

    # Calculate 10-day EMA for Close Price
    df['EMA_10_Close'] = df['close'].ewm(span=10).mean()

    # Calculate Long-Term Price Momentum
    df['Long_Term_Price_Momentum'] = df['EMA_10_Close'].pct_change()

    # Calculate 10-day EMA for Volume
    df['EMA_10_Volume'] = df['volume'].ewm(span=10).mean()

    # Calculate Long-Term Volume Acceleration
    df['Long_Term_Volume_Acceleration'] = df['EMA_10_Volume'].pct_change()

    # Long-Term Factor Combination
    df['Long_Term_Factor'] = df['Long_Term_Volume_Acceleration'] * df['Long_Term_Price_Momentum'] + df['Short_Term_Factor']

    # Final Alpha Factor with Volume Weighting
    df['Final_Alpha_Factor'] = df['Long_Term_Factor'] / df['volume'].rolling(window=5).sum()

    return df['Final_Alpha_Factor'].dropna()
```
