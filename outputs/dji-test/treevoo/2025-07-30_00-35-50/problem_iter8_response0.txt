```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['DailyReturn'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['PriceGap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_daily_return(row):
        if row['PriceGap'] > 0:
            return row['DailyReturn'] - row['PriceGap']
        else:
            return row['DailyReturn'] + row['PriceGap']
    
    df['AdjustedDailyReturn'] = df.apply(adjust_daily_return, axis=1)
    
    # Apply Exponential Decay to Adjustment (half-life = 1 day)
    df['AdjustedDailyReturn'] *= np.exp(-1 / 1)
    
    # Calculate High-to-Low Price Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['PriceRangeMomentum'] = df['HighLowRange'].pct_change()
    
    # Adjust by Volume Change
    df['VolumeChange'] = df['volume'].pct_change()
    
    # Integrate Momentum and Volume Change
    df['IntegratedMomentum'] = df['PriceRangeMomentum'] * df['VolumeChange']
    
    # Combine Volume Weighted and Price Range Momentum
    df['VolumeWeightedReturn'] = df['AdjustedDailyReturn'] * df['volume']
    
    # Calculate 5-day Exponential Moving Average for Price
    df['5DayEMA_Close'] = df['close'].ewm(span=5, min_periods=5).mean()
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['5DayEMA_Close'].diff()
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['5DayEMA_Volume'] = df['volume'].ewm(span=5, min_periods=5).mean()
    
    # Calculate Volume Acceleration
    df['VolumeAcceleration'] = df['5DayEMA_Volume'].diff()
    
    # Combine Factors
    df['CombinedFactor'] = df['VolumeAcceleration'] * df['PriceMomentum'] + df['IntegratedMomentum'] + df['VolumeWeightedReturn']
    
    # Final Alpha Factor with Volume Weighting
    df['FinalAlphaFactor'] = df['CombinedFactor'] / df['volume'].rolling(window=5).sum()
    
    return df['FinalAlphaFactor']
```
