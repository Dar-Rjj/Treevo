```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided market data.
    The factor is designed to help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the close price
    df['close_5d_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price
    df['close_20d_ma'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day moving standard deviation of the close price
    df['close_5d_std'] = df['close'].rolling(window=5).std()
    
    # Calculate the 20-day moving standard deviation of the close price
    df['close_20d_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the momentum factor as the difference between the 5-day and 20-day moving averages
    df['momentum_factor'] = df['close_5d_ma'] - df['close_20d_ma']
    
    # Calculate the volatility factor as the ratio of 5-day and 20-day moving standard deviations
    df['volatility_factor'] = df['close_5d_std'] / df['close_20d_std']
    
    # Calculate the alpha factor as a combination of the momentum and volatility factors
    df['alpha_factor'] = df['momentum_factor'] / df['volatility_factor']
    
    return df['alpha_factor']
```
