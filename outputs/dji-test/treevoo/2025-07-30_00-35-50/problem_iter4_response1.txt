```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine and Smooth
    combined_factor = high_low_spread * volume_change
    smoothed_factor = combined_factor.rolling(window=5).mean()
    
    return smoothed_factor
```
