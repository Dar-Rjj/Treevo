import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    df['adjusted_daily_return'] = df['daily_return']
    df.loc[df['price_gap'] > 0, 'adjusted_daily_return'] -= df['price_gap'] * decay_factor
    df.loc[df['price_gap'] < 0, 'adjusted_daily_return'] += df['price_gap'] * decay_factor
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['price_range'].pct_change()
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Volume-Weighted Price Range Momentum
    df['volume_weighted_price_range_momentum'] = (df['price_range_momentum'] * df['volume']) / (df['volume'] + df['volume'].shift(1))
    
    # Integrate Volume-Weighted Price Range Momentum and Adjusted Daily Return
    df['alpha_factor'] = df['adjusted_daily_return'] + df['volume_weighted_price_range_momentum']
    
    return df['alpha_factor'].dropna()
