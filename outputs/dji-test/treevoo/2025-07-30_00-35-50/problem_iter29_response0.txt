import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided historical stock data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the volume relative to the 10-day average volume
    df['volume_rel'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (df['momentum'] 
                          + (df['ma_10_close'] - df['ma_30_close']) / df['close']
                          + df['volume_rel'] * df['price_range'] / df['close'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
