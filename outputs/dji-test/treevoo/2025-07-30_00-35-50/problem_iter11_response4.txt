```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Evaluate Momentum Over Time (Sum of Daily Price Changes over a Window, e.g., 10 days)
    window = 10
    df['momentum'] = df['price_change'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=window).mean()
    
    # Adjust for Volume Spike Indicator
    volume_threshold = 1.5
    spike_multiplier = 1.2
    df['vol_spike_indicator'] = df['volume'] / df['volume'].rolling(window=window).mean()
    df['volume_weighted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * spike_multiplier if row['vol_spike_indicator'] > volume_threshold else row['volume_weighted_momentum'], axis=1)
    
    # Adjust for Volatility
    volatility_multiplier = 1.3
    df['historical_volatility'] = df['price_change'].rolling(window=window).std()
    df['volatility_adjusted_momentum'] = df.apply(lambda row: row['volume_weighted_momentum'] * volatility_multiplier if abs(row['price_change']) > 2 * row['historical_volatility'] else row['volume_weighted_momentum'], axis=1)
    
    # Adjust for Long-Term Price Momentum
    long_term_window = 20
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_window)
    
    # Adjust for Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=long_term_window).mean()
    
    # Adjust for Amount Trend
    df['amount_trend'] = df['amount'] - df['amount'].rolling(window=long_term_window).mean()
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['prev_high_low_range']
    
    # Adjust by Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['integrated_momentum'] = df['price_range_momentum'] * df['volume_change'] + df['price_change']
    
    # Combine All Adjustments
    df['factor'] = (
        df['long_term_momentum'] * 
        df['volatility_adjusted_momentum'] * 
        df['volume_trend'] * 
        df['amount_trend'] + 
        df['integrated_momentum']
    )
    
    # Apply Softmax to Ensure Positive Factor
    df['factor'] = df['factor'].apply(lambda x: np.exp(x) / (1 + np.exp(x)) if x < 0 else x)
    
    return df['factor']
```
