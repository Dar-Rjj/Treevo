```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Adjust for Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * (df['Volume_Change'] > 0) * df['Volume_Change']
    
    # Apply Simple Moving Average (SMA) with a period of 5 days
    df['SMA_Adjusted_High_Low_Spread'] = df['Adjusted_High_Low_Spread'].rolling(window=5).mean()
    
    # Calculate 10-Day Price Momentum
    df['10_Day_Price_Momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['10_Day_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend_Adjustment'] = df['volume'] - df['10_Day_MA_Volume']
    
    # Combine Price Momentum and Volume Trend Adjustment
    df['Combined_Momentum_Volume_Adjustment'] = df['10_Day_Price_Momentum'] * df['Volume_Trend_Adjustment']
    df['Combined_Momentum_Volume_Adjustment'] = df['Combined_Momentum_Volume_Adjustment'].apply(lambda x: max(x, 0))
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_return_days = df['Daily_Log_Return'] > 0
    negative_return_days = df['Daily_Log_Return'] <= 0
    
    # Calculate Sum of Upward Volume and Downward Volume
    df['Upward_Volume'] = df['volume'].where(positive_return_days, 0)
    df['Downward_Volume'] = df['volume'].where(negative_return_days, 0)
    df['Total_Volume'] = df['Upward_Volume'] + df['Downward_Volume']
    
    # Compute Trend Reversal Signal
    df['Upward_Volume_Ratio'] = df['Upward_Volume'] / df['Total_Volume']
    df['Downward_Volume_Ratio'] = df['Downward_Volume'] / df['Total_Volume']
    df['Trend_Reversal_Signal'] = (df['Upward_Volume_Ratio'] > df['Downward_Volume_Ratio']).astype(int) * df['Daily_Log_Return']
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap'] * np.exp(-1)
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap'] * np.exp(-1)
    
    # Final Volume Weighted Momentum
    df['Final_Volume_Weighted_Momentum'] = df['volume'] * df['Combined_Momentum_Volume_Adjustment'] * df['Adjusted_Daily_Return']
    
    # Integrate Momentum, Volume, and Price Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Integrated_Momentum'] = df['10_Day_Price_Momentum'] * df['Volume_Change'] + df['Price_Change']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Combined_Momentum_Volume_Adjustment'] + 
                                df['Trend_Reversal_Signal'] + 
                                df['Final_Volume_Weighted_Momentum'] + 
                                df['Integrated_Momentum'])
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].apply(lambda x: max(x, 0))
    
    return df['Final_Alpha_Factor']
```
