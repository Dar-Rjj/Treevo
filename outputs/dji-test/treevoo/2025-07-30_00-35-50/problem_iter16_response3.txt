```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Evaluate Momentum Over Time
    window = 10
    df['momentum'] = df['price_change'].rolling(window=window).sum()
    
    # Compute Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['momentum'] * df['volume']) / df['volume'].rolling(window=window).mean()
    
    # Adjust for Volume Spike Indicator
    df['volume_ma'] = df['volume'].rolling(window=window).mean()
    volume_spike_threshold = 1.5
    df['volume_spike_indicator'] = np.where(df['volume'] > volume_spike_threshold * df['volume_ma'], 1.2, 1.0)
    df['volume_weighted_momentum'] *= df['volume_spike_indicator']
    
    # Adjust for Volatility
    df['volatility'] = df['price_change'].rolling(window=window).std()
    volatility_threshold = 2.0
    df['volatility_indicator'] = np.where(np.abs(df['price_change']) > volatility_threshold * df['volatility'], 1.3, 1.0)
    df['volume_weighted_momentum'] *= df['volatility_indicator']
    
    # Adjust for Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] - df['prev_high_low_range']
    
    # Integrate Momentum, Volume, and Price Change
    df['integrated_momentum'] = df['price_range_momentum'] + df['price_change']
    
    # Combine All Adjustments
    df['factor'] = (df['long_term_momentum'] * df['volume_weighted_momentum']) + df['integrated_momentum']
    
    # Apply Softmax to Ensure Positive Factor
    df['factor'] = np.where(df['factor'] < 0, np.exp(df['factor']) / (1 + np.exp(df['factor'])), df['factor'])
    
    return df['factor']
```
