```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    df['price_range_momentum'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Adjust by Volume Change
    df['prev_volume'] = df['volume'].shift(1)
    df['volume_change'] = df['volume'] / df['prev_volume']
    
    # Integrate Momentum and Volume Change
    df['alpha_factor'] = df['price_range_momentum'] * df['volume_change']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
