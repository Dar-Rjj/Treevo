```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)

    # Adjust Daily Return by Price Gap with Exponential Decay
    df['adjusted_return'] = df['daily_return']
    half_life = 1
    decay_factor = np.exp(-np.log(2) / half_life)
    df.loc[df['price_gap'] > 0, 'adjusted_return'] -= df['price_gap'] * decay_factor
    df.loc[df['price_gap'] < 0, 'adjusted_return'] += df['price_gap'] * decay_factor

    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Calculate Volume and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)

    # Integrate Momentum, Volume, and Price Change
    df['integrated_factor'] = (df['range_momentum'] * df['volume_change']) + df['price_change']

    # Confirm with Volume
    df['confirmed_factor'] = df['integrated_factor'] * df['volume_change']
    df['confirmed_factor'] = df['confirmed_factor'].where(df['volume_change'] > 0, 0)

    # Smoothing Filter
    df['sma_confirmed_factor'] = df['confirmed_factor'].rolling(window=5).mean()

    # Calculate 5-day EMA for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['ema_5_price'] - df['ema_5_price'].shift(1)

    # Calculate 5-day EMA for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)

    # Combine Factors
    df['combined_factor'] = (df['volume_acceleration'] * df['price_momentum']) + df['sma_confirmed_factor']

    # Calculate 20-day EMA for Price
    df['ema_20_price'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['ema_20_price'] - df['ema_20_price'].shift(1)

    # Final Alpha Factor with Volume Weighting
    df['final_alpha_factor'] = df['combined_factor'] + df['long_term_momentum']
    df['final_alpha_factor'] /= df['volume'].rolling(window=5).sum()

    return df['final_alpha_factor']
```
