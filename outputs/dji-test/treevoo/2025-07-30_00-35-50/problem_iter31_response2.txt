```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['adjusted_daily_return'] = df.apply(
        lambda row: row['daily_return'] - row['price_gap'] if row['price_gap'] > 0 else row['daily_return'] + row['price_gap'],
        axis=1
    )
    
    # Evaluate Momentum Over Time (e.g., 10 days)
    df['sum_daily_returns'] = df['adjusted_daily_return'].rolling(window=10).sum()
    
    # Compute Volume Weighted Momentum
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_weighted_momentum'] = (df['sum_daily_returns'] * df['volume']) / df['average_volume']
    
    # Adjust for Volume Spike Indicator
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_spike_indicator'] = df['volume'] > 1.5 * df['volume_moving_avg']
    df['volume_weighted_momentum'] = df.apply(
        lambda row: row['volume_weighted_momentum'] * 1.2 if row['volume_spike_indicator'] else row['volume_weighted_momentum'],
        axis=1
    )
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_high_low_spread'] = df.apply(
        lambda row: row['high_low_spread'] * (row['volume_change'] / row['average_volume']) if row['volume_change'] > 0 else row['high_low_spread'],
        axis=1
    )
    
    # Integrate Momentum, Volume, and Volume-Weighted Price Change
    df['integrated_factor'] = (df['sum_daily_returns'] * df['volume_change']) + df['volume_weighted_momentum']
    
    # Combine Integrated Factor with Volume Change
    df['integrated_factor_with_volume'] = df['integrated_factor'] * df['volume_change']
    df['integrated_factor_with_volume'] = df['integrated_factor_with_volume'].where(df['volume_change'] > 0, 0)
    
    # Smoothing Filter
    df['sma_integrated_factor'] = df['integrated_factor_with_volume'].rolling(window=5).mean()
    
    # Calculate 5-day Exponential Moving Average for Price
    df['ema_5_price'] = df['close'].ewm(span=5, adjust=False).mean()
    df['price_momentum'] = df['ema_5_price'] - df['ema_5_price'].shift(1)
    
    # Calculate 5-day Exponential Moving Average for Volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_acceleration'] = df['ema_5_volume'] - df['ema_5_volume'].shift(1)
    
    # Combine Factors
    df['combined_factor'] = df['volume_acceleration'] * df['price_momentum'] + df['sma_integrated_factor']
    
    # Final Alpha Factor with Volume Weighting
    df['last_5_days_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['combined_factor'] / df['last_5_days_volume_sum']
    df['final_alpha_factor'] = df['final_alpha_factor'].where(df['final_alpha_factor'] > 0, 0)
    
    # Introduce Price Trend Indicator
    df['ema_20_price'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = df['ema_20_price'] - df['ema_20_price'].shift(1)
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['price_trend']
    df['final_alpha_factor'] = df['final_alpha_factor'].where(df['final_alpha_factor'] > 0, 0)
    
    return df['final_alpha_factor']
```
