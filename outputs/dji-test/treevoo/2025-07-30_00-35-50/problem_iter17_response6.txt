```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    df['Adjusted_Daily_Return'] = (df['Daily_Return'] - df['Price_Gap']).ewm(span=1, adjust=False).mean()

    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    df['Price_Range_Momentum'] = df['High_Low_Range'] / df['Prev_High_Low_Range']

    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Adjust for Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['High_Low_Spread_Adjusted'] = df['High_Low_Spread'] * (df['Volume_Change']).clip(lower=0)

    # Apply Simple Moving Average (SMA) to High-Low Spread with Volume Change
    df['SMA_High_Low_Spread_Vol'] = df['High_Low_Spread_Adjusted'].rolling(window=5).mean()

    # Calculate 10-Day Price Momentum
    df['10_Day_Price_Momentum'] = df['close'] - df['close'].shift(10)

    # Adjust for Volume Trend
    df['10_Day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend_Adjustment'] = df['volume'] - df['10_Day_Avg_Volume']
    
    # Combine Price Momentum and Volume Trend Adjustment
    df['Combined_Momentum_Volume_Adjustment'] = (df['10_Day_Price_Momentum'] * df['Volume_Trend_Adjustment']).clip(lower=0)

    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))

    # Identify Positive and Negative Returns
    positive_returns = df[df['Daily_Log_Return'] > 0]
    negative_returns = df[df['Daily_Log_Return'] <= 0]

    # Calculate Sum of Upward Volume and Downward Volume
    df['Upward_Volume'] = np.where(df['Daily_Log_Return'] > 0, df['volume'], 0)
    df['Downward_Volume'] = np.where(df['Daily_Log_Return'] <= 0, df['volume'], 0)
    df['Sum_Upward_Volume'] = df['Upward_Volume'].rolling(window=5).sum()
    df['Sum_Downward_Volume'] = df['Downward_Volume'].rolling(window=5).sum()

    # Compute Trend Reversal Signal
    df['Total_Volume'] = df['Sum_Upward_Volume'] + df['Sum_Downward_Volume']
    df['Upward_Volume_Ratio'] = df['Sum_Upward_Volume'] / df['Total_Volume']
    df['Downward_Volume_Ratio'] = df['Sum_Downward_Volume'] / df['Total_Volume']
    df['Trend_Reversal_Signal'] = np.where(df['Upward_Volume_Ratio'] > df['Downward_Volume_Ratio'], 1, 0) * df['Daily_Log_Return']

    # Calculate Accumulated Volume Impact
    df['Accumulated_Volume_Impact'] = df['volume'].rolling(window=5).sum()

    # Synthesize Final Alpha Factor
    df['Final_Factor'] = (df['Adjusted_Daily_Return'] + 
                          df['SMA_High_Low_Spread_Vol'] + 
                          df['Combined_Momentum_Volume_Adjustment'] + 
                          df['Trend_Reversal_Signal']) * (df['volume'] / df['Accumulated_Volume_Impact'])
    df['Final_Factor'] = df['Final_Factor'].clip(lower=0)

    return df['Final_Factor']
```
