```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'].shift(10) - df['close']
    
    # Adjust for Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_adjustment = df['volume'] - volume_10_day_ma
    
    # Combine Price Momentum and Volume Adjustment
    combined_factor = price_momentum * volume_adjustment
    combined_factor = combined_factor.apply(lambda x: max(x, 0))
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_returns = daily_log_return > 0
    negative_returns = daily_log_return <= 0
    
    # Calculate Sum of Upward Volume
    upward_volume = df['volume'][positive_returns].rolling(window=10).sum()
    
    # Calculate Sum of Downward Volume
    downward_volume = df['volume'][negative_returns].rolling(window=10).sum()
    
    # Compute Trend Reversal Signal
    total_volume = upward_volume + downward_volume
    upward_volume_ratio = upward_volume / total_volume
    downward_volume_ratio = downward_volume / total_volume
    trend_reversal_signal = (upward_volume_ratio > downward_volume_ratio).astype(int) * daily_log_return
    
    # Final Alpha Factor
    final_factor = combined_factor * trend_reversal_signal
    final_factor = final_factor.apply(lambda x: max(x, 0))
    
    return final_factor
```
