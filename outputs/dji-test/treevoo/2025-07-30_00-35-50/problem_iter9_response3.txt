```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Adjust for Volume Trend
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_adjustment = df['volume'] - volume_ma_20
    
    # Adjust for Amount Trend
    amount_ma_20 = df['amount'].rolling(window=20).mean()
    amount_adjustment = df['amount'] - amount_ma_20
    
    # Combine Price Momentum, Volume Adjustment, and Amount Adjustment
    combined_factor = (price_momentum * volume_adjustment) * amount_adjustment
    combined_factor = np.maximum(combined_factor, 0)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_return_days = daily_log_return > 0
    negative_return_days = daily_log_return <= 0
    
    # Calculate Sum of Upward Volume
    upward_volume = df.loc[positive_return_days, 'volume'].sum()
    
    # Calculate Sum of Downward Volume
    downward_volume = df.loc[negative_return_days, 'volume'].sum()
    
    # Compute Enhanced Trend Reversal Signal
    total_volume = upward_volume + downward_volume
    upward_volume_ratio = upward_volume / total_volume
    downward_volume_ratio = downward_volume / total_volume
    reversal_signal = (upward_volume_ratio > downward_volume_ratio) * daily_log_return
    
    # Introduce Trend Confirmation
    log_return_ma_20 = daily_log_return.rolling(window=20).mean()
    trend_confirmation = (daily_log_return > log_return_ma_20) * daily_log_return
    reversal_signal *= trend_confirmation
    
    # Combine Price Momentum and Enhanced Trend Reversal Signal
    final_factor = combined_factor + reversal_signal
    final_factor = np.maximum(final_factor, 0)
    
    return final_factor
```
