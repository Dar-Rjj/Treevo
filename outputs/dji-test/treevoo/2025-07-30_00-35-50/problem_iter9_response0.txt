```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['close_t_minus_20'] = df['close'].shift(20)
    df['price_momentum'] = df['close'] - df['close_t_minus_20']
    
    # Calculate Volume Trend Score
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_score'] = df['volume'] / df['volume_20_day_ma']
    
    # Calculate Amount Trend Score
    df['amount_20_day_ma'] = df['amount'].rolling(window=20).mean()
    df['amount_trend_score'] = df['amount'] / df['amount_20_day_ma']
    
    # Combine Price Momentum, Volume Trend Score, and Amount Trend Score
    df['combined_factor'] = df['price_momentum'] * df['volume_trend_score'] * df['amount_trend_score']
    
    # Apply Softmax to Ensure Positive Factor
    def softmax(x):
        if x < 0:
            return np.exp(x) / (1 + np.exp(x))
        return x
    
    df['final_factor'] = df['combined_factor'].apply(softmax)
    
    return df['final_factor'].dropna()
```
