```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    df['Adjusted_Daily_Return'] = df['Daily_Return']
    df.loc[df['Price_Gap'] > 0, 'Adjusted_Daily_Return'] -= df['Price_Gap']
    df.loc[df['Price_Gap'] < 0, 'Adjusted_Daily_Return'] += df['Price_Gap']
    df['Adjusted_Daily_Return'] *= np.exp(-1 / 1)  # Apply exponential decay with half-life of 1 day

    # Calculate Integrated Momentum
    df['Price_Momentum'] = df['close'].ewm(span=5).mean() - df['close'].ewm(span=5).mean().shift(1)
    df['High_Low_Range'] = df['high'] - df['low']
    df['Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    df['Integrated_Momentum'] = df['Price_Momentum'] + df['Range_Momentum']

    # Confirm High-Low Spread with Volume and Adjusted by Amount Acceleration
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_EMA'] = df['amount'].ewm(span=5).mean()
    df['Amount_Acceleration'] = df['Amount_EMA'] - df['Amount_EMA'].shift(1)
    df['High_Low_Spread_Adjusted'] = df['High_Low_Range'] * df['Volume_Change'].clip(lower=0) * df['Amount_Acceleration']

    # Smoothing Filter
    df['SMA_High_Low_Spread'] = df['High_Low_Spread_Adjusted'].rolling(window=10).mean()

    # Integrate Momentum and Smoothed Volume-Adjusted High-Low Spread
    df['Integrated_Momentum_Scaled'] = df['Integrated_Momentum'] * df['SMA_High_Low_Spread']

    # Evaluate Momentum Over Time
    df['Sum_Daily_Returns'] = df['Daily_Return'].rolling(window=10).sum()

    # Compute Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Sum_Daily_Returns'] * df['volume'] / df['volume'].rolling(window=10).mean()

    # Adjust for Volume Spike Indicator
    df['Volume_MA'] = df['volume'].rolling(window=10).mean()
    df['Volume_Spike_Indicator'] = (df['volume'] > 1.5 * df['Volume_MA']).astype(int)
    df['Volume_Weighted_Momentum'] *= (1 + 0.2 * df['Volume_Spike_Indicator'])

    # Adjust for Volatility
    df['Historical_Volatility'] = df['Daily_Return'].rolling(window=10).std()
    df['Absolute_Price_Change'] = (df['close'] - df['close'].shift(1)).abs()
    df['Volatility_Spike_Indicator'] = (df['Absolute_Price_Change'] > 2 * df['Historical_Volatility']).astype(int)
    df['Volume_Weighted_Momentum'] *= (1 + 0.3 * df['Volatility_Spike_Indicator'])

    # Incorporate Price Confirmation
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Price_Confirmation'] = (df['Price_Change'] > 0) & (df['Volume_Change'] > 0)
    df.loc[df['Price_Confirmation'], 'High_Low_Spread_Adjusted'] *= (df['Price_Change'] + df['Volume_Change'])

    # Combine Adjusted Daily Return, Integrated Momentum, and Volume Weighted Momentum
    df['Final_Factor'] = df['Adjusted_Daily_Return'] * df['Integrated_Momentum_Scaled'] * df['Volume_Weighted_Momentum']
    
    # Ensure Positive Factor
    df['Final_Factor'] = df['Final_Factor'].clip(lower=0)

    return df['Final_Factor']
```
