```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day Exponential Moving Average for Close price
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate 10-Day Price Momentum
    df['price_momentum_10'] = df['close'] - df['close'].shift(10)
    
    # Adjust for Volume Trend
    df['volume_trend_10'] = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Trend Adjustment
    df['combined_momentum'] = df['price_momentum_10'] * df['volume_trend_10']
    df['combined_momentum'] = df['combined_momentum'].apply(lambda x: max(x, 0))
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify Positive and Negative Returns
    positive_returns = (df['daily_log_return'] > 0)
    negative_returns = (df['daily_log_return'] <= 0)
    
    # Calculate Sum of Upward Volume
    df['upward_volume'] = df['volume'] * positive_returns
    df['sum_upward_volume'] = df['upward_volume'].rolling(window=10).sum()
    
    # Calculate Sum of Downward Volume
    df['downward_volume'] = df['volume'] * negative_returns
    df['sum_downward_volume'] = df['downward_volume'].rolling(window=10).sum()
    
    # Compute Trend Reversal Signal
    total_volume = df['sum_upward_volume'] + df['sum_downward_volume']
    upward_volume_ratio = df['sum_upward_volume'] / total_volume
    downward_volume_ratio = df['sum_downward_volume'] / total_volume
    reversal_signal = (upward_volume_ratio > downward_volume_ratio) * df['daily_log_return']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    decay_factor = 0.5
    df['adjusted_daily_return'] = df['daily_return'] - df['price_gap'] * np.exp(-decay_factor)
    
    # Final Volume Weighted Momentum
    df['final_volume_weighted_momentum'] = df['volume'] * df['combined_momentum'] * df['adjusted_daily_return']
    
    # Integrate Momentum, Volume, and Price Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['integrated_factor'] = df['price_momentum_10'] * df['volume_change'] + df['price_change']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] + reversal_signal + df['final_volume_weighted_momentum'] + df['integrated_factor']
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
```
