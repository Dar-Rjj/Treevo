```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust Daily Return by Price Gap
    def adjust_return(row):
        if row['price_gap'] > 0:
            adjustment = row['price_gap'] * np.exp(-1)  # Apply exponential decay with half-life of 1 day
            return row['daily_return'] - adjustment
        elif row['price_gap'] < 0:
            adjustment = row['price_gap'] * np.exp(-1)  # Apply exponential decay with half-life of 1 day
            return row['daily_return'] + adjustment
        else:
            return row['daily_return']
    
    df['adjusted_daily_return'] = df.apply(adjust_return, axis=1)
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Range Momentum
    df['price_range_momentum'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_momentum'] = df['price_range_momentum'] * df['volume_change']
    
    # Combine Volume-Weighted and Price Range Momentum
    df['volume_weighted_return'] = df['adjusted_daily_return'] * df['volume']
    
    # Integrate Volume-Weighted Price Range Momentum and Adjusted Daily Return
    df['integrated_momentum'] = df['volume_weighted_return'] + df['volume_weighted_momentum']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Confirm Volume Increase
    df['volume_confirmed_spread'] = df['high_low_spread'] * df['volume_change']
    df['volume_confirmed_spread'] = df['volume_confirmed_spread'].where(df['volume_change'] > 0, 0)
    
    # Combine Momentum and Volume Confirmed High-Low Spread
    df['combined_factor'] = df['integrated_momentum'] + df['volume_confirmed_spread']
    
    # Apply Simple Moving Average (SMA) to Combined Factor
    df['final_alpha_factor'] = df['combined_factor'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
