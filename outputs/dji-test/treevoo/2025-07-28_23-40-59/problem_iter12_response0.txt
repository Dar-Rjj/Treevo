```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=30, k_days=30, p_days=60):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Compute Price Change over N Days
    n_day_return = df['close'].pct_change(n_days)
    
    # Compare Latest Volume to Average
    avg_volume = df['volume'].rolling(window=m_days).mean()
    volume_support = np.where(df['volume'] > avg_volume, 1, -1)
    
    # Compare Latest Amount to Average
    avg_amount = df['amount'].rolling(window=k_days).mean()
    amount_support = np.where(df['amount'] > avg_amount, 1, -1)
    
    # Compute Volume-Weighted Momentum
    p_day_returns = df['close'].pct_change(p_days)
    p_day_volumes = df['volume'].rolling(window=p_days).sum()
    volume_weighted_momentum = (p_day_returns * p_day_volumes).sum()
    
    # Combine all the factors
    factor = (
        irwap_diff +
        volume_adjusted_log_return +
        n_day_return +
        volume_support +
        amount_support +
        volume_weighted_momentum
    )
    
    return factor
```
