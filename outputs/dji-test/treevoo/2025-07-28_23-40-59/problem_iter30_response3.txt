import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_100'] = df['close'].rolling(window=100).mean()

    # Calculate Exponential Moving Averages
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_100'] = df['close'].ewm(span=100, adjust=False).mean()

    # Calculate Rate of Change (ROC) Indicators
    df['ROC_10'] = df['close'].pct_change(periods=10)
    df['ROC_20'] = df['close'].pct_change(periods=20)
    df['ROC_50'] = df['close'].pct_change(periods=50)

    # Calculate Bollinger Bands
    std_20 = df['close'].rolling(window=20).std()
    df['Upper_Band'] = df['SMA_20'] + 2 * std_20
    df['Lower_Band'] = df['SMA_20'] - 2 * std_20

    # Calculate Historical Volatility (HV_20)
    df['HV_20'] = df['close'].pct_change().rolling(window=20).std() * (252 ** 0.5)

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Calculate Chaikin Money Flow (CMF_20)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate Accumulation/Distribution Line (A/D Line)
    df['AD_Line'] = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low']) * df['volume']
    df['AD_Line'] = df['AD_Line'].cumsum()

    # Calculate True Range (TR)
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1)

    # Calculate Average True Range (ATR)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    df['ATR_20'] = df['TR'].rolling(window=20).mean()

    # Calculate William's %R
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    df['Williams_%R_14'] = -100 * (highest_high - df['close']) / (highest_high - lowest_low)
    highest_high = df['high'].rolling(window=20).max()
    lowest_low = df['low'].rolling(window=20).min()
    df['Williams_%R_20'] = -100 * (highest_high - df['close']) / (highest_high - lowest_low)

    # Create Composite Alpha Factors
    df['Trend_Strength'] = df['SMA_20'] - df['EMA_20']
    df['Volatility_Factor'] = df['Upper_Band'] - df['Lower_Band']
    df['Volume_Confirmation_Factor'] = df['OBV'] * df['CMF_20'] * df['AD_Line']
    df['Price_Action_Factor'] = (df['close'] - df['open']) / df['ATR_20']
    df['Momentum_and_Volatility_Factor'] = df['ROC_20'] * df['HV_20']

    # Final Alpha Factor
    alpha_factor = df['Trend_Strength'] + df['Volatility_Factor
