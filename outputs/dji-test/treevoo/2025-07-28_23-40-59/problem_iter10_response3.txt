```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, LMA_period=200, SMA_period=50, EMA_period=10):
    # Calculate N-Day Price Change using Close Prices
    price_change = df['close'].pct_change(N)
    
    # Calculate M-Day Average Volume
    avg_volume = df['volume'].rolling(window=M).mean()
    
    # Compare Latest Volume to Average
    volume_score = np.where(df['volume'] > avg_volume, 1, -1)
    
    # Calculate LMA (200-Period Close Price Moving Average)
    lma = df['close'].rolling(window=LMA_period).mean()
    
    # Calculate SMA (50-Period Close Price Moving Average)
    sma = df['close'].rolling(window=SMA_period).mean()
    
    # Compute LMA-SMA Difference
    lma_sma_diff = lma - sma
    
    # Apply EMA Smoothing (10-Period Exponential Moving Average)
    smoothed_diff = lma_sma_diff.ewm(span=EMA_period, adjust=False).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = price_change * volume_score * smoothed_diff
    
    return alpha_factor
```
