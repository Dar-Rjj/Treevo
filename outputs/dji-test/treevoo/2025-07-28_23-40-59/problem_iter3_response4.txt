```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    short_term_momentum = df['log_return'].rolling(window=7).sum()

    # Calculate Long-Term Momentum
    long_term_momentum = df['log_return'].rolling(window=25).sum()

    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum

    # Filter by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_condition = df['volume'] > 1.5 * avg_volume_20
    combined_momentum_filtered = combined_momentum * volume_condition.astype(int)

    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum_filtered / price_volatility

    return adjusted_combined_momentum.dropna()
```
