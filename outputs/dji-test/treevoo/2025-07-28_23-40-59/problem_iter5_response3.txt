```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change().rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change().rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_momentum = combined_momentum / (price_volatility + 1e-8)  # Add a small constant to avoid division by zero
    
    # Confirm with Volume Trends
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_trend_score = np.where(df['volume'] > avg_volume, 1, -1)
    
    # Apply Volume Weighted Filter
    volume_weighted_diff = (adjusted_momentum * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    smoothed_factor = volume_weighted_diff.ewm(span=10).mean()
    
    # Final Factor Calculation
    final_factor = smoothed_factor * volume_trend_score
    
    return final_factor
```
