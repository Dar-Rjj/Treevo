import pandas as pd

def heuristics_v2(df, n_days=20, spike_threshold=2.0, spike_factor=1.5):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Volume-Weighted Daily Return
    df['volume_weighted_daily_return'] = df['daily_price_change'] * df['volume']
    df['volume_weighted_cumulative_sum'] = df['volume_weighted_daily_return'].rolling(window=n_days).sum()
    df['total_volume_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['volume_weighted_daily_return'] = df['volume_weighted_cumulative_sum'] / df['total_volume_n_days']
    
    # Compute Volume-Averaged Daily Return
    df['volume_averaged_daily_return'] = (df['daily_price_change'] * df['volume']).rolling(window=n_days).sum() / df['total_volume_n_days']
    
    # Incorporate Volume Spike Indicator
    df['volume_std_dev'] = df['volume'].rolling(window=n_days).std()
    df['volume_spike'] = (df['volume'] > (spike_threshold * df['volume_std_dev'])).astype(int)
    
    # Adjust Volume-Weighted Daily Return for Volume Spikes
    df['adjusted_volume_weighted_daily_return'] = df['volume_weighted_daily_return'] * (1 + (df['volume_spike'] * (spike_factor - 1)))
    
    # Combine Volume-Weighted and Volume-Averaged Returns
    df['combined_factor'] = (df['adjusted_volume_weighted_daily_return'] + df['volume_averaged_daily_return']) / 2
    
    return df['combined_factor'].dropna()
