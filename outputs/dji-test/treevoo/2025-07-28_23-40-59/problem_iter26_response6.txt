```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    adj_daily_return = daily_return / intraday_spread
    
    # Smooth the Adjusted Return
    smooth_adj_return = adj_daily_return.rolling(window=5).mean()
    
    # Combine Short and Long-Term Momentum
    short_momentum = (np.log(df['close']) - np.log(df['close'].shift(1))).rolling(window=7).sum()
    long_momentum = (np.log(df['close']) - np.log(df['close'].shift(1))).rolling(window=25).sum()
    momentum_diff = short_momentum - long_momentum
    
    # Introduce Price Volatility Component
    price_volatility = momentum_diff * (1 / (df['high'] - df['low']).rolling(window=10).std())
    
    # Confirm with Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Smooth Intraday Strength
    intraday_strength = ((df['close'] - df['open']) / (df['high'] - df['low'])).rolling(window=5).mean()
    
    # Compare Latest Volume to M-day Average
    avg_volume_m = df['volume'].rolling(window=20).mean()
    volume_score = np.where(df['volume'] > avg_volume_m, 1, -1)
    
    # Compare Latest Amount to K-day Average
    avg_amount_k = df['amount'].rolling(window=20).mean()
    amount_score = np.where(df['amount'] > avg_amount_k, 1, -1)
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    lma_sma_diff = lma - sma
    
    # Introduce Open-Price Based Momentum
    open_momentum = (np.log(df['open']) - np.log(df['open'].shift(1))).rolling(window=7).sum()
    
    # Combine Open-Price Based Momentum
    open_long_momentum_diff = open_momentum - long_momentum
    
    # Final Factor Calculation
    final_factor = (irwap_diff + smooth_adj_return + price_volatility 
                    + intraday_strength + volume_trend + amount_score 
                    + lma_sma_diff + open_long_momentum_diff)
    
    return final_factor
```
