```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']).diff()
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff()
    
    # Compute Volume Change
    volume_change = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    weighted_log_returns = daily_log_return.ewm(span=10).mean()
    
    # Volume Change Momentum
    volume_change_momentum = volume_change.ewm(span=10).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = daily_log_return.rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = daily_log_return.rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Filter by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_filter = df['volume'] > 1.5 * avg_volume_20_days
    combined_momentum[~volume_filter] = 0
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    combined_momentum /= price_volatility
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    weighted_daily_ranges = daily_range.ewm(span=10).mean()
    
    # Calculate Price-Average Trading Volume
    price_avg_trading_volume = df['volume'].rolling(window=20).mean()
    
    # Introduce Price-Average Trading Volume Adjusted Momentum
    volume_adjusted_momentum = combined_momentum * price_avg_trading_volume
    
    # Combine Factors
    final_factor = (weighted_log_returns * 
                    volume_change_momentum * 
                    weighted_daily_ranges * 
                    volume_adjusted_momentum)
    
    return final_factor
```
