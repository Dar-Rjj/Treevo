```python
import pandas as pd

def heuristics_v2(df, n=10, m=30, k=30):
    # Compute Price Change over N Days
    close_prices = df['close']
    n_day_return = close_prices.pct_change(n).dropna()
    
    # Compare Latest Volume to Average
    average_volume = df['volume'].rolling(window=m).mean()
    latest_volume = df['volume']
    volume_support = (latest_volume > average_volume).astype(int) - (latest_volume < average_volume).astype(int)
    
    # Compare Latest Amount to Average
    average_amount = df['amount'].rolling(window=k).mean()
    latest_amount = df['amount']
    amount_support = (latest_amount > average_amount).astype(int) - (latest_amount < average_amount).astype(int)
    
    # Combine the signals
    combined_signal = n_day_return + volume_support + amount_support
    return combined_signal
```
