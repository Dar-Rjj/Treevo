```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['Close_Volume_Product'] = df['close'] * df['volume']
    IRWAP = df['Close_Volume_Product'].rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    df['IRWAP'] = IRWAP
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Log_Return'] * df['volume']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Analysis'] = df['Intraday_High_Low_Spread'] + df['IRWAP_Difference']
    
    # Calculate Price Momentum
    N = 10
    df['Price_Momentum'] = df['close'].pct_change(N)
    
    # Compare Latest Volume to Moving Average
    M = 20
    df['MA_Volume'] = df['volume'].rolling(window=M).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['MA_Volume'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    LMA_period = 200
    SMA_period = 50
    df['LMA'] = df['close'].rolling(window=LMA_period).mean()
    df['SMA'] = df['close'].rolling(window=SMA_period).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['LMA_SMA_Diff'] = (df['SMA'] - df['LMA']) * df['volume']
    df['LMA_SMA_VW_EMA'] = df['LMA_SMA_Diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_window = 7
    long_term_window = 25
    df['Short_Term_Momentum'] = df['Log_Return'].rolling(window=short_term_window).sum()
    df['Long_Term_Momentum'] = df['Log_Return'].rolling(window=long_term_window).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Filter by Volume
    df['Average_Volume_20D'] = df['volume'].rolling(window=20).mean()
    df['Volume_Filter'] = np.where(df['volume'] > 1.5 * df['Average_Volume_20D'], 1, 0)
    
    # Introduce Price Volatility Component
    volatility_window = 10
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=volatility_window).std()
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['WMA_Daily_Range'] = df['Daily_Range'].ewm(span=10, adjust=False).mean()
    
    # Combine Factors
    alpha_factor = (df['LMA_SMA_VW_EMA'] *
                    df['Adjusted_Combined_Momentum'] *
                    df['WMA_Daily_Range'])
    
    return alpha_factor
```
