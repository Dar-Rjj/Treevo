```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Indicator
    intraday_range = df['high'] - df['low']
    
    # Integrate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Indicators for Intraday Analysis
    intraday_analysis = intraday_range + (vwap - df['close'])
    
    # Calculate Volume-Adjusted Log Return
    high_low_spread = df['high'] - df['low']
    adjusted_volume = df['volume'] / high_low_spread
    log_return = np.log(df['close'] / df['close'].shift(1))
    volume_adjusted_log_return = log_return * adjusted_volume
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(5)  # N=5 days
    
    # Confirm with Volume Trends
    avg_volume_10 = df['volume'].rolling(window=10).mean()  # M=10 days
    volume_trend_score = np.where(df['volume'] > avg_volume_10, 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    vol_weighted_diff = (lma - sma) * df['volume']
    smoothed_vol_weighted_diff = vol_weighted_diff.ewm(span=10).mean()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_analysis +
        volume_adjusted_log_return +
        price_momentum +
        volume_trend_score * smoothed_vol_weighted_diff
    )
    
    return alpha_factor
```
