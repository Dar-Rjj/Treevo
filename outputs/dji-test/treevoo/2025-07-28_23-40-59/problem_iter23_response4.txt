import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=50, K=50, LMA_period=200, SMA_period=50, smoothing_span=10):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Weighted Daily Return
    df['Volume_Weighted_Return'] = (df['Daily_Price_Change'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Introduce Enhanced Price Directional Component
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Sign_Close_Open_Diff'] = np.sign(df['Close_Open_Diff'])
    
    # Incorporate Dynamic Volume Spike Indicator
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    df['Volume_Spike'] = (df['volume'] > volume_mean + 2 * volume_std).astype(int)
    df['Adjusted_Volume_Weighted_Return'] = df['Volume_Weighted_Return'] * (1 + (df['volume'] - volume_mean) / volume_std).where(df['Volume_Spike'] == 1, 1)
    
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['IRWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['IRWAP_Diff'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['High_Low_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Log_Return'] * df['Adjusted_Volume']
    
    # Combine Adjusted Volume, Log Return, and Enhanced Price Direction
    df['Combined_Factor'] = df['Volume_Adjusted_Log_Return'] * df['Sign_Close_Open_Diff'] * np.sign(df['high'] - df['close'])
    
    # Smooth the Combined Adjusted Volume, Log Return, and Enhanced Price Direction
    df['Smoothed_Combined_Factor'] = df['Combined_Factor'].rolling(window=5).mean()
    
    # Compare Latest Volume to M-day Average
    df['M_day_Avg_Volume'] = df['volume'].rolling(window=M).mean()
    df['Volume_Score'] = np.where(df['volume'] > df['M_day_Avg_Volume'], 1, -1)
    
    # Compare Latest Amount to K-day Average
    df['K_day_Avg_Amount'] = df['amount'].rolling(window=K).mean()
    df['Amount_Score'] = np.where(df['amount'] > df['K_day_Avg_Amount'], 1, -1)
    
    # Calculate Price Momentum
    df['N_day_Return'] = df['close'].pct_change(periods=N)
    
    # Compare Latest Volume to Moving Average
    df['M_day_Moving_Avg_Volume'] = df['volume'].rolling(window=M).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['M_day_Moving_Avg_Volume'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['LMA'] = df['close'].rolling(window=LMA_period).mean()
    df['SMA'] = df['close'].rolling(window=SMA_period).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['Volume_Weighted_Diff'] = (df['LMA'] - df['SMA']) * df['volume']
    df['Volume_Weighted_Sum'] = df['Volume_Weighted_Diff'].rolling(window=1).sum()
    
    # Smooth with Exponential Moving Average Over 10 periods
    df['Final_Factor'] = df['Volume_Weighted_Sum'].ewm(span=smoothing_span, adjust=False).mean()
    
    return df['Final_Factor']
