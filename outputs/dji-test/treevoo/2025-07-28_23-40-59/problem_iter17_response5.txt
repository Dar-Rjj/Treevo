```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Integrate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap'] = vwap
    
    # VWAP Difference
    df['vwap_diff'] = df['vwap'] - df['close']
    
    # Combine Indicators for Intraday Analysis
    df['intraday_analysis'] = df['intraday_range'] + df['vwap_diff']
    
    # Calculate Volume-Adjusted Log Return
    df['high_low_spread'] = df['high'] - df['low']
    adjusted_volume = df['volume'] / df['high_low_spread']
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['log_return'] * adjusted_volume
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Confirm with Volume Trends
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_score'] = np.where(df['volume'] > df['avg_volume_20'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['lma'] = df['close'].rolling(window=200).mean()
    df['sma'] = df['close'].rolling(window=50).mean()
    
    # Adjust Daily Return by Intraday Volatility
    df['adjusted_return'] = (df['close'] - df['close'].shift(1)) / df['intraday_range']
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()
    
    # Calculate Short-Term and Medium-Term Momentum
    df['short_term_momentum'] = df['log_return'].rolling(window=7).sum()
    df['medium_term_momentum'] = df['log_return'].rolling(window=14).sum()
    df['combined_momentum'] = df['short_term_momentum'] - df['medium_term_momentum']
    
    # Introduce Price Volatility Component
    df['price_volatility'] = df['high_low_spread'].rolling(window=10).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['price_volatility']
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['lma_sma_diff'] = (df['lma'] - df['sma']) * df['volume']
    df['volume_weighted_lma_sma_diff'] = df['lma_sma_diff'].ewm(span=10).mean()
    
    # Final Factor Calculation
    df['final_factor'] = (df['adjusted_combined_momentum'] 
                          + df['smoothed_adjusted_return'] 
                          + df['volume_weighted_lma_sma_diff'])
    
    return df['final_factor']
```
