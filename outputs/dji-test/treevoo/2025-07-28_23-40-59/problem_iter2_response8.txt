import pandas as pd

def heuristics_v2(df, n_days=5, m_days=20):
    # Calculate Intraday Range Indicator
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Daily VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(periods=n_days)
    
    # Calculate Average Volume over M Days
    df['avg_volume'] = df['volume'].rolling(window=m_days).mean()
    
    # Determine if Volume Trend is Supported
    df['volume_trend_score'] = df.apply(
        lambda row: 1 if row['volume'] > row['avg_volume'] else -1,
        axis=1
    )
    
    # Merge Indicators
    df['factor'] = df['intraday_range'] * df['vwap'] * df['price_momentum'] * df['volume_trend_score']
    
    return df['factor']
