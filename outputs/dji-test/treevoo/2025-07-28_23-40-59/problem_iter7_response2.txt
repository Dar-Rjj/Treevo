```python
import pandas as pd

def heuristics_v2(df):
    # Calculate N-Day Price Change (Price Momentum)
    n = 10
    df['N_Day_Price_Change'] = df['close'].pct_change(periods=n)
    
    # Calculate M-Day Average Volume and Compare Latest Volume to Average
    m = 30
    df['M_Day_Avg_Volume'] = df['volume'].rolling(window=m).mean()
    df['Volume_Score'] = df.apply(lambda row: 1 if row['volume'] > row['M_Day_Avg_Volume'] else -1, axis=1)
    
    # Calculate LMA (200-Period Moving Average of Close Prices)
    lma_period = 200
    df['LMA'] = df['close'].rolling(window=lma_period).mean()
    
    # Calculate SMA (50-Period Moving Average of Close Prices)
    sma_period = 50
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    
    # Compute LMA-SMA Difference
    df['LMA_SMA_Diff'] = df['LMA'] - df['SMA']
    
    # Apply Volume-Weighted Filter
    df['Volume_Weighted_Factor'] = df['LMA_SMA_Diff'] * df['Volume_Score'] * df['volume']
    
    # EMA Smoothing (10-Period Exponential Moving Average)
    ema_period = 10
    df['Alpha_Factor'] = df['Volume_Weighted_Factor'].ewm(span=ema_period, adjust=False).mean()
    
    return df['Alpha_Factor']
```
