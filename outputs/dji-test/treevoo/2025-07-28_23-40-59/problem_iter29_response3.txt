import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    df['Momentum_5D'] = df['Daily_Return'].rolling(window=5).sum()

    # Analyze High-Low Range for Volatility Indicators
    df['True_Range'] = df[['high', 'low', 'close']].join(
        df['close'].shift(1), rsuffix='_Prev').apply(
        lambda x: max(x['high'] - x['low'], 
                      abs(x['high'] - x['close_Prev']), 
                      abs(x['low'] - x['close_Prev'])), axis=1)
    df['ATR_14D'] = df['True_Range'].rolling(window=14).mean()

    # Investigate Trading Volume Patterns
    df['Volume_20D_Avg'] = df['volume'].rolling(window=20).mean()
    df['High_Volume_Day'] = df['volume'] > df['Volume_20D_Avg']
    df['Volume_Signal'] = np.where(df['High_Volume_Day'] & (df['close'] > df['open']), 1,
                                   np.where(df['High_Volume_Day'] & (df['close'] < df['open']), -1, 0))

    # Explore Price Gaps at Market Open
    df['Upward_Gap'] = df['open'] > df['close'].shift(1)
    df['Downward_Gap'] = df['open'] < df['close'].shift(1)

    # Categorize Day-to-Day Price Movement Trends
    df['Rate_of_Change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Price_Trend_Score'] = pd.cut(df['Rate_of_Change'], bins=[-np.inf, -0.02, -0.005, 0.005, 0.02, np.inf], labels=[-2, -1, 0, 1, 2])

    # Evaluate Volume-Price Correlation
    df['Volume_Price_Correlation'] = df['volume'].rolling(window=30).corr(df['close'])

    # Calculate Volume-Weighted Daily Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open']) * df['volume']
    df['Cumulative_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=20).sum()
    df['Total_Volume'] = df['volume'].rolling(window=20).sum()
    df['Volume_Weighted_Daily_Return'] = df['Cumulative_Volume_Weighted_Return'] / df['Total_Volume']

    # Introduce Enhanced Price Directional Component
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Close_Open_Diff_Sign'] = np.sign(df['Close_Open_Diff'])

    df['Volume_STD'] = df['volume'].rolling(window=20).std()
    df['Volume_Mean'] = df['volume'].rolling(window=20).mean()
    df['Volume_Spike'] = df['volume'] > (df['Volume_Mean'] + 2 * df['Volume_STD'])
    df['Dynamic_Factor'] = 1 + (df['volume'] - df['Volume_Mean']) / df['Volume_STD']
    df.loc[~df['Volume_Spike'], 'Dynamic_Factor'] = 1
