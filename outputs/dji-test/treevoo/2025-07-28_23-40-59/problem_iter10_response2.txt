```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return over N days (e.g., 5, 10, 20)
    N = 20
    df['Daily_Return_Volume_Weighted'] = df['Daily_Price_Change'] * df['volume']
    df['Cumulative_Return_Volume_Weighted'] = df['Daily_Return_Volume_Weighted'].rolling(window=N).sum()
    df['Total_Volume'] = df['volume'].rolling(window=N).sum()
    df['Volume_Averaged_Daily_Return'] = df['Cumulative_Return_Volume_Weighted'] / df['Total_Volume']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['close'] - df['close'].shift(7)
    
    # Incorporate Volume Change
    df['Average_Volume_Last_7_Days'] = df['volume'].rolling(window=7).mean()
    df['Volume_Factor'] = df['volume'] / df['Average_Volume_Last_7_Days']
    
    # Adjust for Volatility
    df['Short_Term_Volatility'] = df['close'].rolling(window=7).std()
    df['Inverse_Volatility'] = 1 / df['Short_Term_Volatility']
    df['Volatility_Adjusted_Momentum'] = df['Raw_Momentum'] * df['Inverse_Volatility']
    
    # Apply Volume Factor to Final Momentum
    df['Volume_Adjusted_Momentum'] = df['Volatility_Adjusted_Momentum'] * df['Volume_Factor']
    
    # Combine Factors
    df['Intermediate_Alpha_Factor'] = df['Volume_Adjusted_Momentum'] * df['Volume_Averaged_Daily_Return']
    
    # Final Alpha Factor
    final_alpha_factor = df['Intermediate_Alpha_Factor']

    return final_alpha_factor
```
