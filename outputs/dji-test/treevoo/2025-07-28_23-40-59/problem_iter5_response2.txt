```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume Weighted Filter
    volume_weighted_diff = diff_lma_sma * df['volume']
    volume_weighted_sum = volume_weighted_diff.rolling(window=10).sum()
    smoothed_volume_weighted = volume_weighted_sum.ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_daily_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return over the last 5 days
    smoothed_adjusted_return = adjusted_daily_return.rolling(window=5).mean()
    
    # Output the smoothed value as the alpha factor
    alpha_factor = smoothed_adjusted_return * smoothed_volume_weighted
    
    return alpha_factor
```
