```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA)
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA)
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume Weighted Filter
    vol_weighted_diff = (diff_lma_sma * df['volume']).ewm(span=10).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=7).sum()
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = np.log(df['close']).diff().rolling(window=14).sum()
    
    # Combine Short and Medium-Term Momentum
    combined_momentum = short_term_momentum - medium_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_momentum = combined_momentum / (price_volatility + 1e-6)  # Avoid division by zero
    
    # Confirm with Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend_score = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Final Factor Calculation
    ema_20 = df['close'].ewm(span=20).mean()
    ema_5 = df['close'].ewm(span=5).mean()
    trend_strength = np.where(ema_5 > ema_20, 1.5, -1.5)
    
    final_factor = (vol_weighted_diff + smoothed_adjusted_return + adjusted_momentum * volume_trend_score) + trend_strength
    
    return final_factor.dropna()
```
