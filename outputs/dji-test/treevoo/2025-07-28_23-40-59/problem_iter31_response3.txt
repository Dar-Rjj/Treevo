```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns for short-term momentum
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    df['Short_Term_Momentum'] = df['Daily_Return'].rolling(window=5).sum()

    # Calculate True Range and Average True Range for long-term volatility
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'],
                                                               abs(x['high'] - df['close'].shift(1)),
                                                               abs(x['low'] - df['close'].shift(1))), axis=1)
    df['Average_True_Range_14'] = df['True_Range'].rolling(window=14).mean()

    # Identify unusually high volume days
    df['Volume_20_Day_Avg'] = df['volume'].rolling(window=20).mean()
    df['High_Volume'] = df['volume'] > df['Volume_20_Day_Avg']

    # Examine price changes on high volume days
    df['Price_Change'] = df['close'] - df['open']
    df['Strong_Buy_Signal'] = (df['High_Volume'] & (df['Price_Change'] > 0))
    df['Potential_Sell_Off'] = (df['High_Volume'] & (df['Price_Change'] < 0))

    # Detect upward and downward gaps
    df['Upward_Gap'] = (df['open'] > df['close'].shift(1)) & ((df['open'] - df['close'].shift(1)) / df['close'].shift(1) > 0.01)
    df['Downward_Gap'] = (df['open'] < df['close'].shift(1)) & ((df['open'] - df['close'].shift(1)) / df['close'].shift(1) < -0.01)

    # Evaluate Volume-Price Correlation
    df['Volume_Price_Correlation'] = df['volume'].rolling(window=30).corr(df['close'])

    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (df['Short_Term_Momentum'] +
                          df['Average_True_Range_14'] * 10 +
                          df['Strong_Buy_Signal'].astype(int) * 5 -
                          df['Potential_Sell_Off'].astype(int) * 5 +
                          df['Upward_Gap'].astype(int) * 10 -
                          df['Downward_Gap'].astype(int) * 10 +
                          df['Volume_Price_Correlation'] * 10)

    return df['Alpha_Factor']
```
