```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA)
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA)
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume Weighted Filter
    vol_weighted_diff = diff_lma_sma * df['volume']
    sum_vol_weighted_diff = vol_weighted_diff.rolling(window=1).sum()
    smoothed_diff = sum_vol_weighted_diff.ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Output the smoothed value as the alpha factor
    alpha_factor = smoothed_diff * smoothed_adjusted_return
    return alpha_factor
```
