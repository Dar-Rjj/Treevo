```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Incorporate Volume Change
    volume_factor = df['volume'] / df['volume'].rolling(window=7).mean()
    momentum_adjusted_volume = raw_momentum / volume_factor
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=7).std()
    volatility_adjusted_momentum = momentum_adjusted_volume / volatility
    
    # Integrate Intraday Indicators
    intraday_range = df['high'] - df['low']
    daily_vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    alpha_factor = volatility_adjusted_momentum * intraday_range * daily_vwap
    
    return alpha_factor
```
