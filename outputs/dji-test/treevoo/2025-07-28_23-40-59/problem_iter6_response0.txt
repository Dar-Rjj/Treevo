import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Raw Momentum
    short_term_raw_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Raw Momentum
    long_term_raw_momentum = df['close'] - df['close'].shift(25)
    
    # Incorporate Volume Change
    volume_factor = df['volume'] / df['volume'].rolling(window=10).mean()
    adjusted_short_term_momentum = short_term_raw_momentum / volume_factor
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=10).std()
    volatility_adjusted_momentum = adjusted_short_term_momentum / (short_term_volatility + 1e-8)
    
    # Incorporate High-Low Range
    daily_high_low_range = df['high'] - df['low']
    avg_high_low_range = daily_high_low_range.rolling(window=10).mean()
    high_low_adjusted_momentum = volatility_adjusted_momentum / (avg_high_low_range + 1e-8)
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_raw_momentum - long_term_raw_momentum
    
    # Filter by Volume
    min_volume_threshold = df['volume'].rolling(window=25).mean()
    filtered_combined_momentum = combined_momentum[df['volume'] > min_volume_threshold]
    
    return filtered_combined_momentum
