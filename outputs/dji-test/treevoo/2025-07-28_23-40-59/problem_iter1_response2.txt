```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods using close price
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods using close price
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff = lma - sma
    
    # Multiply the difference by volume to apply a volume-weighted filter
    vol_weighted_diff = diff * df['volume']
    
    # Sum of volume-weighted differences
    sum_vol_weighted_diff = vol_weighted_diff.rolling(window=1).sum()
    
    # Smooth with Exponential Moving Average (EMA) over 10 periods
    ema_smoothed = sum_vol_weighted_diff.ewm(span=10, adjust=False).mean()
    
    return ema_smoothed
```
