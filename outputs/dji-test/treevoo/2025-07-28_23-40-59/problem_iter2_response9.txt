```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return
    df['Price_Volume'] = df['Daily_Price_Change'] * df['volume']
    df['Volume_Sum_N'] = df['volume'].rolling(window=n).sum()
    df['Volume_Averaged_Return'] = df['Price_Volume'].rolling(window=n).sum() / df['Volume_Sum_N']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Volume Change
    df['Volume_Percentage_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_Log_Return'] = df['Daily_Log_Return'].ewm(span=n, adjust=False).mean()
    
    # Volume Change Momentum
    df['Weighted_Volume_Change'] = df['Volume_Percentage_Change'].ewm(span=n, adjust=False).mean()
    
    # Combine Factors
    df['Intermediate_Alpha_Factor'] = df['Weighted_Log_Return'] * df['Weighted_Volume_Change']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] * df['Volume_Averaged_Return']
    
    return df['Final_Alpha_Factor'].dropna()
```
