```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods using Close price
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods using Close price
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff = lma - sma
    
    # Apply Volume Weighted Filter: Multiply by Volume and take the sum, then smooth with EMA over 10 periods
    volume_weighted_diff = (diff * df['volume']).ewm(span=10).mean()
    
    # Calculate Intraday Range (High - Low)
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return with a 5-day simple moving average
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Output the final alpha factor
    alpha_factor = volume_weighted_diff * smoothed_adjusted_return
    return alpha_factor
```
