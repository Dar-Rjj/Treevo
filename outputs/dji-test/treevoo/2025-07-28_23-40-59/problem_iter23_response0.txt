```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate the average daily return over a 20-day window for momentum factor
    df['momentum_factor'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate the percentage change in price from the high and low of the previous day for reversal factor
    df['reversal_factor'] = (df['high'] - df['low'].shift(1)) / df['low'].shift(1)
    
    # Calculate volume-weighted close
    df['volume_weighted_close'] = (df['close'] * df['volume']) / df['volume'].rolling(window=5).sum()
    
    # Calculate strength factor
    df['strength_factor'] = (df['volume_weighted_close'] - df['close']) / df['close']
    
    # Calculate the standard deviation of daily returns over a 20-day window for volatility factor
    df['volatility_factor'] = df['daily_return'].rolling(window=20).std()
    
    # Combine the momentum factor with the volatility factor
    df['momentum_and_volatility_combined'] = df['momentum_factor'] * df['volatility_factor']
    
    # Combine the reversal factor with the strength factor
    df['reversal_and_strength_combined'] = df['reversal_factor'] * df['strength_factor']
    
    # Integrate all factors to get the final alpha score
    df['integrated_alpha'] = (df['momentum_and_volatility_combined'] + df['reversal_and_strength_combined']) / 2
    
    return df['integrated_alpha']
```
