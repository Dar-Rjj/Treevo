```python
def heuristics_v2(df):
    # Calculate the price range for each day
    df['range'] = df['high'] - df['low']
    
    # Calculate the absolute difference between open and close
    df['abs_diff_open_close'] = (df['open'] - df['close']).abs()
    
    # Calculate the percentage of the range that the absolute difference represents
    df['factor'] = df['abs_diff_open_close'] / df['range']
    
    # Return the factor as a pandas Series
    return df['factor']
```
