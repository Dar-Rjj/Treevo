```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, volume].
    
    Returns:
    pd.Series: A Series with the index as date and the values as the calculated alpha factor.
    """
    
    # Compute the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Calculate a weighted sum of the average price, price range, and volume change
    # The weights are chosen to give more importance to the price range and less to the volume change
    factor = 0.4 * avg_price + 0.5 * price_range + 0.1 * vol_change
    
    return factor
```
