```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust Log Return by Intraday Range
    adjusted_log_return = log_return / intraday_range
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_range
    
    # Generate Hybrid Factor
    hybrid_factor = adjusted_log_return * adjusted_volume
    
    return hybrid_factor
```
