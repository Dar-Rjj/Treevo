```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Smoothed Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Compute Volume-Weighted Daily Return
    N = 10  # Example: 10 days
    df['volume_weighted_return'] = (df['price_change'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Incorporate Dynamic Volume Spike Indicator
    rolling_std = df['volume'].rolling(window=20).std()
    rolling_mean = df['volume'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume'] > rolling_mean + 2 * rolling_std).astype(int)
    df['dynamic_factor'] = 1 + (df['volume'] - rolling_mean) / rolling_std
    df['volume_weighted_return'] = df['volume_weighted_return'] * df['dynamic_factor']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    df['adjusted_return'] = df['daily_return'] / df['intraday_range']
    
    # Smooth the Adjusted Return
    df['smoothed_adjusted_return'] = df['adjusted_return'].ewm(span=5).mean()
    
    # Combine Volume-Weighted Price Momentum with Intraday Volatility-Adjusted Returns
    df['interim_alpha'] = df['volume_weighted_return'] + df['smoothed_adjusted_return']
    
    # Calculate Smoothed Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(10)
    df['smoothed_momentum'] = df['close'] - df['close'].shift(21)
    df['combined_momentum'] = (df['raw_momentum'] + df['smoothed_momentum']) / 2
    
    # Incorporate Volume Change
    df['volume_factor'] = df['volume'] / df['volume'].rolling(window=21).mean()
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=21).std()
    df['adjusted_momentum'] = df['combined_momentum'] / df['volatility'] * df['volume_factor']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_diff'] = (df['close'] - df['vwap']) / df['volatility']
    
    # Calculate Market Sentiment
    df['avg_amount'] = df['amount'].rolling(window=21).mean()
    df['sentiment_ratio'] = df['amount'] / df['avg_amount']
    
    # Combine Indicators
    df['alpha_factor'] = (df['interim_alpha'] + 
                          df['adjusted_momentum'] + 
                          df['vwap_diff'] + 
                          df['sentiment_ratio'])
    
    return df['alpha_factor']
```
