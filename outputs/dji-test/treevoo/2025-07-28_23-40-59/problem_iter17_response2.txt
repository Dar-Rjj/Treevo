```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['volume_weighted_close'] = df['close'] * df['volume']
    total_volume = df['volume'].rolling(window=1).sum()
    df['irwap'] = df['volume_weighted_close'].rolling(window=1).sum() / total_volume
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    df['lma'] = df['close'].rolling(window=200).mean()
    df['sma'] = df['close'].rolling(window=50).mean()
    df['lma_sma_diff'] = df['lma'] - df['sma']
    
    # Apply Volume Weighted Filter
    df['vol_weighted_diff'] = df['lma_sma_diff'] * df['volume']
    df['vol_weighted_sum'] = df['vol_weighted_diff'].rolling(window=1).sum()
    df['ewma_vol_weighted'] = df['vol_weighted_sum'].ewm(span=10, adjust=False).mean()
    
    # Adjust Daily Return by Intraday Volatility
    df['daily_return'] = df['close'].pct_change()
    df['adjusted_return'] = df['daily_return'] / df['high_low_spread']
    
    # Smooth the Adjusted Return
    df['smoothed_adj_return'] = df['adjusted_return'].rolling(window=5).mean()
    
    # Calculate Short-Term and Long-Term Momentum
    df['log_ret'] = np.log(df['close']).diff()
    df['short_term_mom'] = df['log_ret'].rolling(window=7).sum()
    df['long_term_mom'] = df['log_ret'].rolling(window=25).sum()
    df['combined_mom'] = df['short_term_mom'] - df['long_term_mom']
    
    # Introduce Price Volatility Component
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['adj_combined_mom'] = df['combined_mom'] / df['price_volatility']
    
    # Confirm with Volume Trends
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_trend'] = np.where(df['volume'] > df['avg_volume'], 1, -1)
    
    # Introduce Open-Price Based Momentum
    df['open_log_ret'] = np.log(df['open']).diff()
    df['open_short_term_mom'] = df['open_log_ret'].rolling(window=7).sum()
    df['open_combined_mom'] = df['open_short_term_mom'] - df['long_term_mom']
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['irwap_diff'] +
        df['smoothed_adj_return'] +
        df['adj_combined_mom'] +
        df['volume_trend'] +
        df['open_combined_mom']
    )
    
    return df['final_factor']
```
