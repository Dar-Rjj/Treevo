```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Indicator
    intraday_range = df['high'] - df['low']
    
    # Integrate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Indicators for Intraday Analysis
    vwap_difference = vwap - df['close']
    
    # Calculate Volume-Adjusted Log Return
    high_low_spread = df['high'] - df['low']
    adjusted_volume = df['volume'] / high_low_spread
    log_return = np.log(df['close'] / df['close'].shift(1))
    volume_adjusted_log_return = log_return * adjusted_volume
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Averaged Daily Return over N days (e.g., 5, 10, 20)
    n_days = 5
    volume_averaged_daily_return = (daily_price_change * df['volume']).rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Incorporate Volume Change
    average_volume_last_7_days = df['volume'].rolling(window=7).mean()
    volume_factor = df['volume'] / average_volume_last_7_days
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=7).std()
    volatility_adjusted_momentum = raw_momentum / short_term_volatility * volume_factor
    
    # Confirm with Volume Trends
    m_days = 20
    average_volume_m_days = df['volume'].rolling(window=m_days).mean()
    volume_trend_score = np.where(df['volume'] > average_volume_m_days, 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    lma_sma_difference = lma - sma
    
    # Apply Volume Weighted Filter
    volume_weighted_difference = lma_sma_difference * df['volume']
    smoothed_volume_weighted_difference = volume_weighted_difference.ewm(span=10).mean()
    
    # Final Alpha Factor
    intermediate_alpha_factor = (intraday_range + vwap_difference + volume_adjusted_log_return + 
                                 volume_averaged_daily_return + volatility_adjusted_momentum + 
                                 volume_trend_score + smoothed_volume_weighted_difference)
    
    final_alpha_factor = intermediate_alpha_factor * volume_averaged_daily_return
    
    return final_alpha_factor
```
