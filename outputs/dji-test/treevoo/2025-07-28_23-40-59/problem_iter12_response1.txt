def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / high_low_spread
    
    # Introduce Enhanced Price Directional Component
    close_open_diff = df['close'] - df['open']
    sign_close_open = np.sign(close_open_diff)
    
    # Calculate Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate High-Close Difference
    high_close_diff = df['high'] - df['close']
    sign_high_close = np.sign(high_close_diff)
    
    # Combine Adjusted Volume, Log Return, and Enhanced Price Direction
    alpha_factor = (log_return * adjusted_volume * sign_close_open * sign_high_close).dropna()
    
    return alpha_factor
