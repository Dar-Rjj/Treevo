```python
import pandas as pd

def heuristics_v2(df):
    N = 20  # Price Momentum period
    M = 5   # Volume Trend confirmation period
    LMA_PERIOD = 200  # Long-Term Moving Average period
    SMA_PERIOD = 50   # Short-Term Moving Average period
    VWAP_SMOOTHING = 10  # Smoothing for VWAP

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(N)

    # Confirm with Volume Trends
    df['Avg_Volume'] = df['volume'].rolling(window=M).mean()
    df['Volume_Trend_Score'] = df.apply(lambda row: 1 if row['volume'] > row['Avg_Volume'] else -1, axis=1)
    
    # Calculate Long-Term and Short-Term Moving Averages
    df['LMA'] = df['close'].rolling(window=LMA_PERIOD).mean()
    df['SMA'] = df['close'].rolling(window=SMA_PERIOD).mean()
    
    # Calculate Difference Between LMA and SMA
    df['LMA_SMA_Diff'] = df['LMA'] - df['SMA']
    
    # Apply Volume Weighted Filter
    df['Volume_Weighted_Diff'] = df['LMA_SMA_Diff'] * df['volume']
    df['Sum_Volume_Weighted_Diff'] = df['Volume_Weighted_Diff'].rolling(window=VWAP_SMOOTHING).sum()
    df['EMA_Sum_Volume_Weighted_Diff'] = df['Sum_Volume_Weighted_Diff'].ewm(span=VWAP_SMOOTHING).mean()

    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Compute Volume-Averaged Daily Return
    df['Volume_Averaged_Return'] = (df['Daily_Price_Change'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()

    # Calculate Intraday Range Indicator
    df['Intraday_Range'] = df['high'] - df['low']

    # Compute Daily VWAP
    df['Close_Volume'] = df['close'] * df['volume']
    df['Daily_VWAP'] = df['Close_Volume'].rolling(window=N).sum() / df['volume'].rolling(window=N).sum()

    # Integrate Indicators
    df['Alpha_Factor'] = (df['Price_Momentum'] * df['Volume_Trend_Score']) + \
                         df['EMA_Sum_Volume_Weighted_Diff'] + \
                         df['Volume_Averaged_Return'] + \
                         df['Intraday_Range'] + \
                         df['Daily_VWAP']

    return df['Alpha_Fctor']
```
