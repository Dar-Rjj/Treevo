```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change(periods=20)  # N = 20 days for example

    # Confirm with Volume Trends
    avg_volume_5 = df['volume'].rolling(window=5).mean()  # M = 5 days for example
    volume_trend = np.where(df['volume'] > avg_volume_5, 1, -1)

    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()

    # Calculate Difference Between LMA and SMA
    diff_lma_sma = lma - sma

    # Apply Volume Weighted Filter
    vol_weighted_diff = diff_lma_sma * df['volume']
    sum_vol_weighted_diff = vol_weighted_diff.rolling(window=10).sum()
    ema_sum_vol_weighted_diff = sum_vol_weighted_diff.ewm(span=10).mean()

    # Incorporate Volatility Measure
    log_returns = np.log(df['close'] / df['close'].shift(1))
    historical_volatility = log_returns.rolling(window=20).std()  # P = 20 days for example
    adjusted_alpha = ema_sum_vol_weighted_diff / (1 + historical_volatility)

    # Integrate Open and High Prices for Enhanced Momentum
    open_high_range = (df['high'] - df['open']) / df['open']
    enhanced_momentum = price_momentum + open_high_range

    # Combine all components
    final_factor = enhanced_momentum * volume_trend * adjusted_alpha

    return final_factor
```
