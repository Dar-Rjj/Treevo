```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['Close_Volume_Product'] = df['close'] * df['volume']
    total_close_volume_product = df['Close_Volume_Product'].rolling(window=1, min_periods=1).sum()
    total_volume = df['volume'].rolling(window=1, min_periods=1).sum()
    df['IRWAP'] = total_close_volume_product / total_volume
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['Intraday_High_Low_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Adjusted_Volume'] * df['Log_Return']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Analysis'] = df['Intraday_High_Low_Spread'] + df['IRWAP_Difference']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(periods=5)
    
    # Compare Latest Volume to Moving Average
    df['MA_Volume_10'] = df['volume'].rolling(window=10, min_periods=1).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['MA_Volume_10'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['LMA_200'] = df['close'].rolling(window=200, min_periods=1).mean()
    df['SMA_50'] = df['close'].rolling(window=50, min_periods=1).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['LMA_SMA_Difference'] = (df['LMA_200'] - df['SMA_50']) * df['volume']
    df['Volume_Weighted_Difference_EMA'] = df['LMA_SMA_Difference'].ewm(span=10, min_periods=1).mean()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Intraday_Analysis'] 
        + df['Volume_Adjusted_Log_Return'] 
        + df['Price_Momentum'] 
        + df['Volume_Support'] 
        + df['Volume_Weighted_Difference_EMA']
    )
    
    return df['Alpha_Factor']
```
