```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return using Close price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Volume Change using Percentage Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns with Exponential Weighting
    span = 20  # Example span, can be adjusted
    df['weighted_log_returns'] = df['log_return'].ewm(span=span, adjust=False).mean()
    
    # Volume Change Momentum with Exponential Weighting
    df['weighted_volume_change'] = df['volume_change'].ewm(span=span, adjust=False).mean()
    
    # Combine Factors by Multiplying Weighted Moving Average of Log Returns by Volume Change Momentum
    df['alpha_factor'] = df['weighted_log_returns'] * df['weighted_volume_change']
    
    return df['alpha_factor']
```
