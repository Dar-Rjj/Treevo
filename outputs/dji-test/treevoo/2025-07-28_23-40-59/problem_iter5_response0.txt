```python
import pandas as pd

def heuristics_v2(df, N=5, lookback_vol=7):
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return over N days
    volume_averaged_return = (daily_price_change * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(lookback_vol)
    
    # Incorporate Volume Change
    avg_volume_last_7_days = df['volume'].rolling(window=lookback_vol).mean()
    volume_factor = df['volume'] / avg_volume_last_7_days
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=lookback_vol).std()
    weighted_raw_momentum = raw_momentum / short_term_volatility
    final_momentum = weighted_raw_momentum * volume_factor
    
    # Combine Factors
    intermediate_alpha_factor = volume_averaged_return * final_momentum
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_averaged_return
    
    return final_alpha_factor
```
