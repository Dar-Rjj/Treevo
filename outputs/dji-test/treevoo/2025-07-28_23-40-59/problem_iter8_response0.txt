```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Volume-Weighted Daily Return
    df['volume_weighted_return'] = df['daily_price_change'] * df['volume']
    df['cumulative_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=n).sum()
    df['total_volume'] = df['volume'].rolling(window=n).sum()
    df['volume_weighted_daily_return'] = df['cumulative_volume_weighted_return'] / df['total_volume']
    
    # Incorporate Dynamic Volume Spike Indicator
    volume_mean = df['volume'].rolling(window=n).mean()
    volume_std_dev = df['volume'].rolling(window=n).std()
    df['is_volume_spike'] = (df['volume'] > (volume_mean + 2 * volume_std_dev))
    dynamic_factor = 1 + (df['volume'] - volume_mean) / volume_std_dev
    df['adjusted_volume_weighted_return'] = df['volume_weighted_daily_return'].where(~df['is_volume_spike'], df['volume_weighted_daily_return'] * dynamic_factor)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Adjust Daily Return by Intraday Volatility
    df['adjusted_daily_return'] = df['daily_return'] / df['intraday_range']
    
    # Smooth the Adjusted Return
    df['smoothed_adjusted_return'] = df['adjusted_daily_return'].rolling(window=5).mean()
    
    # Combine Volume-Weighted Price Momentum with Intraday Volatility-Adjusted Returns
    df['alpha_factor'] = df['adjusted_volume_weighted_return'] + df['smoothed_adjusted_return']
    
    return df['alpha_factor']
```
