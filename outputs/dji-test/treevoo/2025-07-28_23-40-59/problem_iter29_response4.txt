```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap_numerator = (df['close'] * df['volume']).rolling(window=1).sum()
    irwap_denominator = df['volume'].rolling(window=1).sum()
    irwap = irwap_numerator / irwap_denominator
    
    # Evaluate IRWAP Difference
    irwap_difference = df['close'] - irwap
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = (df['close'] / df['open'].shift(1)) - 1
    adjusted_return = daily_return / intraday_high_low_spread
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=7).mean()
    
    # Calculate Short-Term Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_momentum = log_returns.rolling(window=5).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = log_returns.rolling(window=20).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = (short_term_momentum - long_term_momentum) * 1.5
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=14).std()
    adjusted_combined_momentum = combined_momentum / (price_volatility + 1)
    
    # Compare Latest Volume to Moving Average
    m_day_moving_avg_volume = df['volume'].rolling(window=30).mean()
    volume_support = np.where(df['volume'] > m_day_moving_avg_volume, 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    volume_weighted_diff = (lma - sma) * df['volume']
    volume_weighted_diff_smoothed = volume_weighted_diff.ewm(span=10).mean()
    
    # Introduce Price-Volatility Interaction
    price_change = (df['high'] - df['low']) / df['open']
    price_volatility_interaction = price_change * price_volatility
    
    # Combine Factors
    final_alpha_factor = (
        volume_weighted_diff_smoothed *
        adjusted_combined_momentum *
        smoothed_adjusted_return *
        volume_support *
        price_volatility_interaction
    )
    
    return final_alpha_factor
```
