```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA)
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA)
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume Weighted Filter
    volume_weighted_diff = diff_lma_sma * df['volume']
    sum_volume_weighted_diff = volume_weighted_diff.ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_daily_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_daily_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change().rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change().rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_trend_score = (df['volume'] > avg_volume).astype(int) * 2 - 1  # +1 if true, -1 if false
    
    # Apply Volume-Weighted Filter to Combined Momentum
    volume_weighted_momentum = adjusted_combined_momentum * df['volume']
    smoothed_volume_weighted_momentum = volume_weighted_momentum.ewm(span=10, adjust=False).mean()
    
    # Dynamic Momentum Indicator
    raw_momentum = df['close'] - df['close'].shift(7)
    volume_factor = df['volume'] / df['volume'].rolling(window=7).mean()
    short_term_price_volatility = df['close'].rolling(window=7).std()
    dynamic_momentum = (raw_momentum / short_term_price_volatility) * volume_factor
    
    # Final Factor Calculation
    final_factor = (smoothed_adjusted_return + smoothed_volume_weighted_momentum) * volume_trend_score * dynamic_momentum
    
    return final_factor
```
