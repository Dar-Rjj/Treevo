```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = ((df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()).fillna(0)
    
    # Evaluate IRWAP Difference
    irwap_difference = df['close'] - irwap
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = (df['close'] / df['open'].shift(1)) - 1
    adjusted_return = daily_return / intraday_high_low_spread
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=7).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=5).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = np.log(df['close']).diff().rolling(window=20).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = (short_term_momentum - long_term_momentum) * 1.5
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=14).std()
    adjusted_combined_momentum = combined_momentum / (price_volatility + 1)
    
    # Confirm with Volume Trends
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_trend_confirmation = np.where(df['volume'] > avg_volume_30, 1, -1)
    
    # Final Factor Calculation
    final_factor = (irwap_difference + smoothed_adjusted_return + adjusted_combined_momentum) * volume_trend_confirmation
    
    return final_factor
```
