```python
import pandas as pd

def heuristics_v2(df, N=20, spike_threshold=2.0, adjustment_factor=1.5):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()

    # Compute Volume-Weighted Daily Return
    df['volume_weighted_return'] = df['price_change'] * df['volume']
    df['cumulative_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=N).sum()
    df['total_volume'] = df['volume'].rolling(window=N).sum()
    df['volume_weighted_daily_return'] = df['cumulative_volume_weighted_return'] / df['total_volume']

    # Incorporate Volume Spike Indicator
    df['volume_std_dev'] = df['volume'].rolling(window=N).std()
    df['is_volume_spike'] = df['volume'] > (df['volume_std_dev'] * spike_threshold)
    
    # Adjust Volume-Weighted Daily Return for Volume Spikes
    df['adjusted_volume_weighted_daily_return'] = df.apply(
        lambda row: row['volume_weighted_daily_return'] * adjustment_factor if row['is_volume_spike'] else row['volume_weighted_daily_return'],
        axis=1
    )

    return df['adjusted_volume_weighted_daily_return']
```
