```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return with an Exponential Moving Average (EMA) over 5 days
    smoothed_adjusted_return = adjusted_return.ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term Momentum using Log Returns over a 7-day window
    log_returns = np.log(df['close']).diff()
    short_term_momentum = log_returns.rolling(window=7).sum()
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_momentum = short_term_momentum / (price_volatility + 1e-8)
    
    # Confirm with Volume Trends
    avg_volume = df['volume'].rolling(window=20).mean()
    latest_volume = df['volume']
    volume_trend_score = np.where(latest_volume > avg_volume, 1, -1)
    
    # Final Factor Calculation
    rolling_volatility_of_volume = df['volume'].rolling(window=10).std()
    weight = 1 / (rolling_volatility_of_volume + 1e-8)
    final_factor = (adjusted_momentum + smoothed_adjusted_return + volume_trend_score) * weight
    
    return final_factor
```
