```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Weighted Daily Return
    df['Volume_Weighted_Return'] = (df['Daily_Price_Change'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Incorporate Intraday Range Indicator
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Integrate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    df['VWAP_Difference'] = df['close'] - df['VWAP']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Factor'] = 1.0
    df.loc[df['Intraday_Range'] > df['Intraday_Range'].mean(), 'Intraday_Factor'] = 1.2
    df.loc[df['Intraday_Range'] < df['Intraday_Range'].mean(), 'Intraday_Factor'] = 0.8
    
    # Adjust based on VWAP Difference
    df['VWAP_Factor'] = 1.0
    df.loc[df['VWAP_Difference'] > 0, 'VWAP_Factor'] = 1.1
    df.loc[df['VWAP_Difference'] < 0, 'VWAP_Factor'] = 0.9
    
    # Adjust Volume-Weighted Daily Return for Combined Intraday Analysis
    df['Adjusted_Volume_Weighted_Return'] = df['Volume_Weighted_Return'] * df['Intraday_Factor'] * df['VWAP_Factor']
    
    # Incorporate Enhanced Volume Spike Indicator
    df['Volume_Spike'] = df['volume'] > (df['volume'].rolling(window=N).std() * 2 + df['volume'].rolling(window=N).mean())
    df['Volume_Spike_Factor'] = 1.0
    df.loc[df['Volume_Spike'] & (df['Daily_Price_Change'] > 0), 'Volume_Spike_Factor'] = 1.5
    df.loc[df['Volume_Spike'] & (df['Daily_Price_Change'] < 0), 'Volume_Spike_Factor'] = 0.7
    
    # Final adjustment for volume spike
    df['Final_Adjusted_Return'] = df['Adjusted_Volume_Weighted_Return'] * df['Volume_Spike_Factor']
    
    return df['Final_Adjusted_Return']
```
