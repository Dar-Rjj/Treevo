```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price from the previous day
    avg_price_change = avg_price.pct_change()
    
    # Calculate the ratio of volume to the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Combine the two signals
    factor = avg_price_change * volume_ratio
    
    return factor
```
