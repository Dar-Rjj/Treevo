```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Daily VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the difference between Close price and VWAP
    df['vwap_diff'] = df['close'] - df['vwap']
    
    # Combine Indicators for Intraday Analysis
    df['intraday_analysis'] = df['intraday_range'] + df['vwap_diff']
    
    # Calculate Volume-Adjusted Log Return
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['log_return'] * df['adjusted_volume']
    
    # Calculate Price Momentum
    N = 5
    df['price_momentum'] = df['close'] - df['close'].shift(N)
    
    # Confirm with Volume Trends
    M = 30
    df['avg_volume'] = df['volume'].rolling(window=M).mean()
    df['volume_trend_score'] = np.where(df['volume'] > df['avg_volume'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    LMA = 200
    SMA = 50
    df['lma'] = df['close'].rolling(window=LMA).mean()
    df['sma'] = df['close'].rolling(window=SMA).mean()
    
    # Calculate Difference Between LMA and SMA
    df['lma_sma_diff'] = df['lma'] - df['sma']
    
    # Apply Volume Weighted Filter
    df['volume_weighted_diff'] = df['lma_sma_diff'] * df['volume']
    df['smoothed_volume_weighted_diff'] = df['volume_weighted_diff'].ewm(span=10).mean()
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (
        df['intraday_analysis'] + 
        df['volume_adjusted_log_return'] + 
        df['price_momentum'] * df['volume_trend_score'] + 
        df['smoothed_volume_weighted_diff']
    )
    
    return df['alpha_factor']
```
