```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['close_volume_sum'] = df['close'] * df['volume']
    df['irwap'] = df['close_volume_sum'].rolling(window=1, min_periods=1).sum() / df['volume'].rolling(window=1, min_periods=1).sum()
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Adjust Daily Return by Intraday Volatility
    df['daily_return'] = (df['close'] / df['open'].shift(1)) - 1
    df['adjusted_return'] = df['daily_return'] / df['intraday_spread']
    
    # Smooth the Adjusted Return
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=7, min_periods=1).mean()
    
    # Combine Short and Long-Term Momentum
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['short_term_momentum'] = df['log_returns'].rolling(window=5, min_periods=1).sum()
    df['long_term_momentum'] = df['log_returns'].rolling(window=20, min_periods=1).sum()
    df['combined_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Introduce Price Volatility Component
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=14, min_periods=1).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / (df['price_volatility'] + 1)
    
    # Confirm with Volume Trends
    df['avg_volume_30'] = df['volume'].rolling(window=30, min_periods=1).mean()
    df['volume_trend'] = np.where(df['volume'] > df['avg_volume_30'], 1, -1)
    
    # Introduce Price-Volatility Interaction
    df['price_change'] = (df['high'] - df['low']) / df['open']
    df['interaction'] = df['price_change'] * df['price_volatility']
    
    # Smooth Intraday Strength
    df['intraday_strength'] = (df['close'] - df['open']) / df['intraday_spread']
    df['smoothed_intraday_strength'] = df['intraday_strength'].rolling(window=5, min_periods=1).mean()
    
    # Filter by Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20, min_periods=1).mean()
    df['volume_filtered'] = np.where(df['volume'] > 1.5 * df['avg_volume_20'], 1, 0)
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['weighted_daily_range'] = df['daily_range'].ewm(span=10, adjust=False).mean()
    
    # Volume Change Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['weighted_volume_change'] = df['volume_change'].ewm(span=10, adjust=False).mean()
    
    # Combine Factors
    df['final_alpha_factor'] = (
        df['smoothed_adjusted_return'] * 
        df['weighted_volume_change'] * 
        df['weighted_daily_range'] * 
        df['adjusted_combined_momentum'] + 
        df['interaction']
    ) * df['volume_filtered']
    
    return df['final_alpha_factor']
```
