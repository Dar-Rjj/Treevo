import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow for each day
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']

    # Compute a simple moving average of the money flow over a 14-day window
    money_flow_sma = money_flow.rolling(window=14).mean()

    # Calculate the money flow ratio
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sma = positive_money_flow.rolling(window=14).mean()
    negative_money_flow_sma = negative_money_flow.rolling(window=14).mean()
    money_ratio = positive_money_flow_sma / negative_money_flow_sma

    # Generate the final factor value
    factor_value = money_ratio - money_flow_sma

    return factor_value
