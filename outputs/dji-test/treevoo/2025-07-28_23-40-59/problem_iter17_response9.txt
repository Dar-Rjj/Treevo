```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return
    volume_averaged_daily_return = (daily_price_change * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Calculate Intraday Range Indicator
    intraday_range = df['high'] - df['low']
    
    # Integrate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    vwap_diff = vwap - df['close']
    
    # Combine Indicators for Intraday Analysis
    intraday_analysis = intraday_range + vwap_diff
    
    # Calculate Volume-Adjusted Log Return
    high_low_spread = df['high'] - df['low']
    adjusted_volume = df['volume'] / high_low_spread
    log_return = np.log(df['close'] / df['close'].shift(1))
    volume_adjusted_log_return = log_return * adjusted_volume
    
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Adjust for Volatility and Volume
    short_term_volatility = df['close'].rolling(window=7).std()
    volume_factor = df['volume'] / df['volume'].rolling(window=7).mean()
    weighted_raw_momentum = raw_momentum / (short_term_volatility * volume_factor)
    
    # Combine Factors
    intermediate_alpha_factor = weighted_raw_momentum * volume_averaged_daily_return
    
    # Calculate M-Day Exponential Moving Average (EMA) Volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Compare Latest Volume to EMA Volume
    volume_comparison_score = np.where(df['volume'] > ema_volume, 1, -1)
    
    # Calculate K-Day Exponential Moving Average (EMA) Amount
    ema_amount = df['amount'].ewm(span=5, adjust=False).mean()
    
    # Compare Latest Amount to EMA Amount
    amount_comparison_score = np.where(df['amount'] > ema_amount, 1, -1)
    
    # Calculate LMA and SMA
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    
    # Compute LMA-SMA Difference
    lma_sma_diff = lma - sma
    
    # Apply EMA Smoothing
    smoothed_lma_sma_diff = lma_sma_diff.ewm(span=10, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_averaged_daily_return * volume_comparison_score * amount_comparison_score * smoothed_lma_sma_diff
    
    return final_alpha_factor
```
