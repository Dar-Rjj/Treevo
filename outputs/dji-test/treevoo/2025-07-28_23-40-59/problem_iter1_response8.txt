```python
def heuristics_v2(df, N=10, M=10):
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(periods=N)
    
    # Calculate Average Volume over M Days
    df['Avg_Volume'] = df['volume'].rolling(window=M).mean()
    
    # Compare with Current Volume and assign scores
    df['Volume_Score'] = (df['volume'] > df['Avg_Volume']).astype(int)
    df['Volume_Score'] = df['Volume_Score'].replace({1: 1, 0: -1})
    
    # Combine Price Momentum with Volume Score
    df['Alpha_Factor'] = df['Price_Momentum'] * df['Volume_Score']
    
    return df['Alpha_Factor']
```
