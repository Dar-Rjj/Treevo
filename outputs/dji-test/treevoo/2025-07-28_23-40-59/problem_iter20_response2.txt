```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the daily return based on the close price
    daily_return = df['close'].pct_change()
    
    # Calculate a weighted sum of average price and daily return
    factor_value = 0.5 * avg_price + 0.5 * daily_return
    
    return factor_value
```
