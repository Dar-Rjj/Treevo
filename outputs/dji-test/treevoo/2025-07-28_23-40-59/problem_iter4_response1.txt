```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Daily VWAP
    total_value = df['close'] * df['volume']
    cum_total_value = total_value.cumsum()
    cum_volume = df['volume'].cumsum()
    vwap = cum_total_value / cum_volume
    
    # Merge Indicators: combine Intraday Range and VWAP
    # For simplicity, we create a composite indicator by multiplying the two.
    # This is to capture both the momentum (intraday range) and activity intensity (VWAP).
    alpha_factor = intraday_range * vwap
    
    return alpha_factor
```
