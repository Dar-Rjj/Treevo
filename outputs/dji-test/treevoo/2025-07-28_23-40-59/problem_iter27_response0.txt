import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['Close_Volume_Product'] = df['close'] * df['volume']
    df['IRWAP'] = df['Close_Volume_Product'] / df['volume']
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['Intraday_High_Low_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Adjusted_Volume'] * df['Log_Return']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Analysis'] = (df['Intraday_High_Low_Spread'] + df['IRWAP_Difference']) / 2
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(20)
    
    # Compare Latest Volume to Moving Average
    df['MA_Volume_30'] = df['volume'].rolling(window=30).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['MA_Volume_30'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['LMA'] = df['close'].rolling(window=200).mean()
    df['SMA'] = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['Volume_Weighted_Diff'] = (df['LMA'] - df['SMA']) * df['volume']
    df['EMA_Volume_Weighted_Diff'] = df['Volume_Weighted_Diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']).diff()
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_MA_Log_Returns'] = df['Daily_Log_Return'].ewm(span=10, adjust=False).mean()
    
    # Volume Change Momentum
    df['Volume_Change_Momentum'] = df['Volume_Change'].ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Filter by Volume
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Filter'] = np.where(df['volume'] > 1.5 * df['Average_Volume_20'], 1, 0)
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['Weighted_MA_Daily_Ranges'] = df['Daily_Range'].ewm(span=10, adjust=False).mean()
    
    # Introduce Relative Strength Index (RSI)
    df['Change'] = df['close'].diff()
    gain = df['Change'].apply(lambda x: x if x > 0 else 0)
    loss = df['Change'].apply(lambda x: -x if x < 0 else 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
