```python
import pandas as pd

def heuristics_v2(df, N=21):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Averaged Daily Return
    df['Volume_Adjusted_Return'] = df['Daily_Price_Change'] * df['volume']
    df['Volume_Sum_N'] = df['volume'].rolling(window=N).sum()
    df['Volume_Averaged_Return'] = df['Volume_Adjusted_Return'].rolling(window=N).sum() / df['Volume_Sum_N']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['close'] - df['close'].shift(7)
    
    # Adjust for Volatility and Volume
    # Short-Term Price Volatility
    df['Short_Term_Volatility'] = df['close'].rolling(window=7).std()
    
    # Volume Factor
    df['Avg_Volume_7'] = df['volume'].rolling(window=7).mean()
    df['Volume_Factor'] = df['volume'] / df['Avg_Volume_7']
    
    # Weigh Raw Momentum by Inverse of Volatility and Volume Factor
    df['Weighted_Raw_Momentum'] = df['Raw_Momentum'] / (df['Short_Term_Volatility'] * df['Volume_Factor'])
    
    # Combine Factors
    df['Alpha_Factor'] = df['Weighted_Raw_Momentum'] * df['Volume_Averaged_Return']
    
    return df['Alpha_Factor'].dropna()
```
