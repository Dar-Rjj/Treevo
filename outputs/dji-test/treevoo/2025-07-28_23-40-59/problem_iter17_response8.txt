```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Volume Change
    volume_change = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    wma_log_returns = daily_log_return.ewm(span=10).mean()
    
    # Volume Change Momentum
    wma_volume_change = volume_change.ewm(span=10).mean()
    
    # Calculate Long-Term Moving Average (LMA)
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA)
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    lma_sma_diff = lma - sma
    
    # Apply Volume Weighted Filter
    volume_weighted_filter = (lma_sma_diff * df['volume']).ewm(span=10).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=7).sum()
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = np.log(df['close']).diff().rolling(window=14).sum()
    
    # Combine Short and Medium-Term Momentum
    combined_momentum = short_term_momentum - medium_term_momentum
    
    # Filter by Volume
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_filter = (df['volume'] > 1.5 * avg_volume_20d).astype(int)
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    wma_daily_ranges = daily_range.ewm(span=10).mean()
    
    # Confirm with Volume Trends
    volume_trend_score = np.where(df['volume'] > avg_volume_20d, 1, -1)
    
    # Combine Factors
    final_factor = (wma_log_returns * wma_volume_change * wma_daily_ranges * adjusted_combined_momentum * volume_trend_score * volume_weighted_filter)
    
    return final_factor.dropna()
```
