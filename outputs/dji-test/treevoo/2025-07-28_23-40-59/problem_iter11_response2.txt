```python
import pandas as pd

def heuristics_v2(df, n_days=20, volume_spike_threshold=2.0, momentum_lookback=14):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Compute Volume-Weighted Daily Return
    df['volume_weighted_return'] = df['daily_price_change'] * df['volume']
    df['volume_sum'] = df['volume'].rolling(window=n_days).sum()
    df['volume_weighted_daily_return'] = df['volume_weighted_return'].rolling(window=n_days).sum() / df['volume_sum']

    # Incorporate Dynamic Volume Spike Indicator
    volume_std = df['volume'].rolling(window=n_days).std()
    volume_mean = df['volume'].rolling(window=n_days).mean()
    df['is_volume_spike'] = (df['volume'] > volume_mean + volume_spike_threshold * volume_std)
    df['dynamic_factor'] = 1 + (df['volume'] - volume_mean) / volume_std
    df.loc[df['is_volume_spike'], 'volume_weighted_daily_return'] *= df['dynamic_factor']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Smoothed Momentum
    df['raw_momentum_7d'] = df['close'] - df['close'].shift(7)
    df['smoothed_momentum'] = (df['close'] - df['close'].shift(momentum_lookback)) / 2 + df['raw_momentum_7d'] / 2

    # Incorporate Volume Change
    df['average_volume_14d'] = df['volume'].rolling(window=momentum_lookback).mean()
    df['volume_factor'] = df['volume'] / df['average_volume_14d']

    # Adjust for Volatility
    df['price_volatility'] = df['close'].rolling(window=momentum_lookback).std()
    df['adjusted_smoothed_momentum'] = df['smoothed_momentum'] / df['price_volatility'] * df['volume_factor']

    # Integrate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_diff'] = (df['close'] - df['vwap']) / df['price_volatility']

    # Combine Volume-Weighted Price Momentum with Smoothed Momentum
    df['combined_momentum'] = df['volume_weighted_daily_return'] + df['adjusted_smoothed_momentum']

    # Combine All Indicators
    df['alpha_factor'] = df['combined_momentum'] + df['intraday_range'] * df['volume_factor'] + df['vwap_diff']
    
    return df['alpha_factor']
```
