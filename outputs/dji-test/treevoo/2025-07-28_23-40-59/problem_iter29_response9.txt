```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices, using volume as weights
    weighted_avg = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Create a factor by dividing the difference of close and open by the weighted average of high and low
    factor_values = diff_close_open / weighted_avg
    
    return factor_values
```
