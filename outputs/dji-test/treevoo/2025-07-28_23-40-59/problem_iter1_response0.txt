```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Incorporate Volume Change
    avg_volume_7_days = df['volume'].rolling(window=7).mean()
    volume_factor = df['volume'] / avg_volume_7_days
    adjusted_momentum = raw_momentum / volume_factor
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=7).std()
    final_momentum = adjusted_momentum / short_term_volatility
    
    return final_momentum
```
