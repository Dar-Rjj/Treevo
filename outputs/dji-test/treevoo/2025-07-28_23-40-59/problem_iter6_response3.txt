```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Volume Factor
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_factor = df['volume'] / avg_volume_10_days
    
    # Calculate Short-Term Price Volatility
    short_term_volatility = (df['high'] - df['low']).rolling(window=7).std()
    
    # Adjust for Volatility and Apply Volume Factor
    adjusted_momentum = raw_momentum * (1 / short_term_volatility) * volume_factor
    
    return adjusted_momentum
```
