```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['IRWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = df['Intraday_Spread'].rolling(window=10).std()
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Smooth the Daily Return
    df['Smoothed_Daily_Return'] = df['Daily_Return'].rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = df['close'].apply(np.log).diff().rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = df['close'].apply(np.log).diff().rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Adjust Combined Momentum by Price Volatility
    df['Adjusted_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Confirm with Volume Trends
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend_Confirmation'] = np.where(df['volume'] > df['Volume_MA_20'], 1, -1)
    
    # Incorporate Open-Price Based Momentum
    df['Open_Price_Momentum'] = df['open'].apply(np.log).diff().rolling(window=7).sum()
    
    # Combine Open-Price Based Momentum
    df['Open_Price_Combined'] = df['Open_Price_Momentum'] - df['Long_Term_Momentum']
    
    # Introduce Volume-Weighted Close Price (VWCP)
    df['VWCP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Evaluate VWCP Difference
    df['VWCP_Difference'] = df['close'] - df['VWCP']
    
    # Final Factor Calculation
    df['Final_Alpha_Factor'] = (
        df['IRWAP_Difference'] + 
        df['Smoothed_Daily_Return'] + 
        df['Adjusted_Momentum'] + 
        df['Volume_Trend_Confirmation'] + 
        df['Open_Price_Combined'] + 
        df['VWCP_Difference']
    )
    
    return df['Final_Alpha_Factor']
```
