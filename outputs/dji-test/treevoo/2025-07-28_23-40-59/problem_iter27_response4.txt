```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']).diff()
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Compare Latest Volume to M-day Average
    m = 10
    avg_volume_m = df['volume'].rolling(window=m).mean()
    volume_score = np.where(df['volume'] > avg_volume_m, 1, -1)
    
    # Compare Latest Amount to K-day Average
    k = 10
    avg_amount_k = df['amount'].rolling(window=k).mean()
    amount_score = np.where(df['amount'] > avg_amount_k, 1, -1)
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    ma_diff = sma - lma
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / intraday_spread
    smoothed_adjusted_return = adjusted_return.rolling(window=7).mean()
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=14).sum()
    long_term_momentum = np.log(df['close']).diff().rolling(window=30).sum()
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend_confirmation = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Introduce Open-Price Based Momentum
    open_price_momentum = np.log(df['open']).diff().rolling(window=14).sum()
    open_close_difference = (df['open'] - df['close']) / df['close']
    
    # Final Factor Calculation
    final_factor = (
        irwap_diff + 
        smoothed_adjusted_return + 
        adjusted_combined_momentum + 
        volume_trend_confirmation + 
        (open_price_momentum - long_term_momentum) + 
        open_close_difference
    )
    
    return final_factor
```
