```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_difference = df['close'] - irwap
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    lma_sma_diff = lma - sma
    
    # Apply Volume-Weighted Filter
    volume_weighted_diff = (lma_sma_diff * df['volume']).rolling(window=10).mean()
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_daily_return = daily_return / intraday_spread
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_daily_return.rolling(window=5).mean()
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=7).sum()
    long_term_momentum = np.log(df['close']).diff().rolling(window=25).sum()
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_trend_score = np.where(df['volume'] > avg_volume, 1, -1)
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']).diff()
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Compare Latest Amount to Average
    avg_amount = df['amount'].rolling(window=20).mean()
    amount_trend_score = np.where(df['amount'] > avg_amount, 1, -1)
    
    # Calculate Intraday Strength
    intraday_strength = (df['close'] - df['open']) / intraday_spread
    
    # Smooth Intraday Strength
    smoothed_intraday_strength = intraday_strength.rolling(window=5).mean()
    
    # Final Factor Calculation
    final_factor = (
        irwap_difference + 
        smoothed_adjusted_return + 
        adjusted_combined_momentum + 
        smoothed_intraday_strength + 
        volume_weighted_diff + 
        volume_adjusted_log_return
    )
    
    return final_factor
```
