```python
import pandas as pd
import numpy as np

def heuristics_v2(df, M=30, K=30):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['irwap_numerator'] = df['close'] * df['volume']
    total_volume = df['volume'].sum()
    irwap = df['irwap_numerator'].sum() / total_volume
    df['irwap_difference'] = df['close'] - irwap

    # Calculate Adjusted Volume
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']

    # Integrate Volume-Adjusted Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['adjusted_volume'] * df['log_return']

    # Compare Latest Volume to M-day Average
    df['m_day_avg_volume'] = df['volume'].rolling(M).mean()
    df['volume_score'] = np.where(df['volume'] > df['m_day_avg_volume'], 1, -1)

    # Compare Latest Amount to K-day Average
    df['k_day_avg_amount'] = df['amount'].rolling(K).mean()
    df['amount_score'] = np.where(df['amount'] > df['k_day_avg_amount'], 1, -1)

    # Combine all the factors
    df['heuristic_factor'] = df['volume_adjusted_log_return'] + df['volume_score'] + df['amount_score']

    return df['heuristic_factor']
```
