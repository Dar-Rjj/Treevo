```python
import pandas as pd

def heuristics_v2(df):
    # Momentum from Close Prices
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    df['Short_Term_Momentum'] = df['Daily_Return'].rolling(window=5).sum()

    # High-Low Range for Volatility Indicators
    df['True_Range'] = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df.shift(1).loc[x.name, 'close']), abs(x['low'] - df.shift(1).loc[x.name, 'close'])), axis=1)
    df['Average_True_Range_14'] = df['True_Range'].rolling(window=14).mean()

    # Trading Volume Patterns
    df['Volume_20DMA'] = df['volume'].rolling(window=20).mean()
    df['High_Volume_Day'] = df['volume'] > 1.5 * df['Volume_20DMA']
    
    df['Price_Change'] = df['close'] - df['open']
    df['Strong_Buy_Signal'] = (df['High_Volume_Day'] & (df['Price_Change'] > 0)).astype(int)
    df['Potential_Sell_Off'] = (df['High_Volume_Day'] & (df['Price_Change'] < 0)).astype(int)

    # Price Gaps at Market Open
    df['Gap_Up'] = (df['open'] - df['close'].shift(1)) > 0.02 * df['close'].shift(1)
    df['Gap_Down'] = (df['open'] - df['close'].shift(1)) < -0.02 * df['close'].shift(1)

    # Day-to-Day Price Movement Trends
    df['Rate_Of_Change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Trend_Score'] = pd.cut(df['Rate_Of_Change'], bins=[-1, -0.05, 0, 0.05, 1], labels=[-2, -1, 1, 2]).astype(float)

    # Evaluate Volume-Price Correlation
    df['Volume_Price_Correlation_30'] = df['volume'].rolling(window=30).corr(df['close'])

    # Combine all the factors into a single alpha factor
    alpha_factor = (
        0.3 * df['Short_Term_Momentum'] +
        0.2 * (df['Average_True_Range_14'] - df['Average_True_Range_14'].mean()) /
        df['Average_True_Range_14'].std() +
        0.2 * (df['Trend_Score']) +
        0.1 * (df['Strong_Buy_Signal'] - df['Potential_Sell_Off']) +
        0.2 * df['Volume_Price_Correlation_30']
    )

    return alpha_factor
```
