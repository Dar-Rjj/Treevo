```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume Weighted Filter: Multiply by Volume and then smooth with EMA over 10 periods
    volume_weighted_diff = (diff_lma_sma * df['volume']).ewm(span=10).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return using the last 5 days' average
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Final factor: Multiply the smoothed adjusted return by the volume weighted difference
    factor = smoothed_adjusted_return * volume_weighted_diff
    
    return factor
```
