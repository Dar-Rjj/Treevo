```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    n = 50  # Example: 50-day price change
    df['Price_Momentum'] = df['close'].pct_change(n)
    
    # Confirm with Volume Trends
    m = 20  # Example: 20-day average volume
    df['Avg_Volume_20D'] = df['volume'].rolling(window=m).mean()
    df['Volume_Trend_Score'] = np.where(df['volume'] > df['Avg_Volume_20D'], 1, -1)
    
    # Calculate LMA and SMA
    df['LMA'] = df['close'].rolling(window=200).mean()
    df['SMA'] = df['close'].rolling(window=50).mean()
    df['LMA_SMA_Diff'] = df['LMA'] - df['SMA']
    
    # Apply Volume-Weighted Filter
    df['Volume_Adjusted_Diff'] = df['LMA_SMA_Diff'] * df['volume']
    df['Volume_Adjusted_Diff_Smoothed'] = df['Volume_Adjusted_Diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_Log_Returns'] = df['Daily_Log_Return'].ewm(span=n, adjust=False).mean() * n
    
    # Volume Change Momentum
    df['Weighted_Volume_Change'] = df['Volume_Change'].ewm(span=n, adjust=False).mean() * n
    
    # Calculate Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=7).sum()
    df['Long_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Filter by Volume
    df['Filtered_Combined_Momentum'] = np.where(df['volume'] > 1.5 * df['volume'].rolling(window=20).mean(), 
                                                df['Combined_Momentum'], 0)
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['Adjusted_Combined_Momentum'] = df['Filtered_Combined_Momentum'] / (df['Price_Volatility'] + 1e-6)
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['Weighted_Daily_Range'] = df['Daily_Range'].ewm(span=n, adjust=False).mean()
    
    # Combine Factors
    df['Final_Alpha_Factor'] = (df['Weighted_Log_Returns'] * df['Weighted_Volume_Change'] * 
                                df['Weighted_Daily_Range'] * df['Adjusted_Combined_Momentum'])
    
    return df['Final_Alpha_Factor']
```
