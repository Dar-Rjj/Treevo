```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Moving Average (SMA) over 50 periods using Close price
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Intraday Range (High - Low)
    intraday_range = df['high'] - df['low']
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return with a 5-day SMA
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum as a Rolling Sum of Log Returns over 7 days
    log_returns = np.log(df['close']).diff()
    short_term_momentum = log_returns.rolling(window=7).sum()
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_momentum = short_term_momentum / (price_volatility + 1e-6)  # Avoid division by zero
    
    # Confirm with Volume Trends
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_confirmation = np.where(df['volume'] > avg_volume, 1, -1)
    
    # Final Factor Calculation
    factor_value = (sma_50 + smoothed_adjusted_return + adjusted_momentum * volume_weighted_confirmation)
    
    return factor_value
```
