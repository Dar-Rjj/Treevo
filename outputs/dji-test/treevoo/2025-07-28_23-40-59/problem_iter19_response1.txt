```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'] - df['close'].shift(1)
    intraday_range = df['high'] - df['low']
    adjusted_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return with a 5-day Simple Moving Average
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=7).sum()
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = np.log(df['close']).diff().rolling(window=14).sum()
    
    # Combine Short and Medium-Term Momentum
    combined_momentum = short_term_momentum - medium_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_weighted_confirmation = np.where(df['volume'] > avg_volume_20_days, 1, -1)
    
    # Final Factor Calculation
    final_factor = (adjusted_combined_momentum + smoothed_adjusted_return + diff_lma_sma) * volume_weighted_confirmation
    
    return final_factor
```
