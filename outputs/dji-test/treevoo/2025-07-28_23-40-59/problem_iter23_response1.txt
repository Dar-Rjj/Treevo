```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the daily price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume-weighted buying pressure
    vol_weighted_buying_pressure = buying_pressure * df['volume']
    
    # Calculate the volume-weighted selling pressure
    vol_weighted_selling_pressure = selling_pressure * df['volume']
    
    # Final factor: normalized difference between volume-weighted buying and selling pressures
    alpha_factor = (vol_weighted_buying_pressure - vol_weighted_selling_pressure) / avg_price
    
    return alpha_factor
```
