```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Price Momentum
    N = 20  # Number of days for price momentum
    price_momentum = close_prices.pct_change(periods=N)
    
    # Confirm with Volume Trends
    M = 20  # Number of days for average volume
    avg_volume = df['volume'].rolling(window=M).mean()
    latest_volume = df['volume']
    volume_trend_score = np.where(latest_volume > avg_volume, 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    LMA = close_prices.rolling(window=200).mean()
    SMA = close_prices.rolling(window=50).mean()
    
    # Calculate Difference Between LMA and SMA
    diff_LMA_SMA = LMA - SMA
    
    # Apply Volume Weighted Filter
    volume_weighted_diff = diff_LMA_SMA * df['volume']
    
    # Sum of Volume-Weighted Differences
    sum_volume_weighted_diff = volume_weighted_diff.rolling(window=10).sum()
    
    # Smooth with Exponential Moving Average Over 10 periods
    EMA_smoothed = sum_volume_weighted_diff.ewm(span=10, adjust=False).mean()
    
    # Combine all the factors
    alpha_factor = price_momentum * volume_trend_score * EMA_smoothed
    
    return alpha_factor
```
