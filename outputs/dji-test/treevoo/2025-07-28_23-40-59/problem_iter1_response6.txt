```python
def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Multiply Daily Price Change with Volume
    df['weighted_change'] = df['price_change'] * df['volume']
    
    # Compute the cumulative sum of weighted changes and volumes over N days
    df['cum_weighted_change'] = df['weighted_change'].rolling(window=N).sum()
    df['cum_volume'] = df['volume'].rolling(window=N).sum()
    
    # Divide cumulative weighted change by cumulative volume to get the average
    df['volume_averaged_momentum'] = df['cum_weighted_change'] / df['cum_volume']
    
    return df['volume_averaged_momentum']
```
