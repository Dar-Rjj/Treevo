```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return
    volume_weighted_return = daily_price_change * df['volume']
    cumulative_volume_weighted_return = volume_weighted_return.rolling(window=N).sum()
    total_volume = df['volume'].rolling(window=N).sum()
    volume_averaged_daily_return = cumulative_volume_weighted_return / total_volume
    
    return volume_averaged_daily_return
```
