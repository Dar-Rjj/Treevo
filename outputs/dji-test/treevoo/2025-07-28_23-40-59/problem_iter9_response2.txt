```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate N-day Return
    n_day_return = close_prices.pct_change(periods=N)
    
    # Calculate M-day Exponential Moving Average (EMA) Volume
    ema_volume = df['volume'].ewm(span=M, adjust=False).mean()
    
    # Determine Support from Volume
    latest_volume = df['volume']
    volume_support = np.where(latest_volume > ema_volume, 1, -1)
    
    # Calculate K-day Exponential Moving Average (EMA) Amount
    ema_amount = df['amount'].ewm(span=K, adjust=False).mean()
    
    # Determine Support from Amount
    latest_amount = df['amount']
    amount_support = np.where(latest_amount > ema_amount, 1, -1)
    
    # Combine all factors
    factor = n_day_return * volume_support * amount_support
    
    return factor
```
