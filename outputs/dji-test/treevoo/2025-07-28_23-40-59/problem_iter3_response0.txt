```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    span = 10  # you can adjust the span
    df['wma_log_returns'] = df['log_return'].ewm(span=span, min_periods=span).mean()

    # Volume Change Momentum
    df['wma_volume_change'] = df['volume_change'].ewm(span=span, min_periods=span).mean()

    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']

    # Weighted Moving Average of Daily Ranges
    df['wma_daily_range'] = df['daily_range'].ewm(span=span, min_periods=span).mean()

    # Combine Factors
    df['alpha_factor'] = (df['wma_log_returns'] * 
                           df['wma_volume_change'] * 
                           df['wma_daily_range'])

    # Output the Final Alpha Factor
    return df['alpha_factor']
```
