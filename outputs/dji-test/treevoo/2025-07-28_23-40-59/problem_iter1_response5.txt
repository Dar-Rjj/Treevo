```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Adjust Daily Return by Intraday Volatility
    df['adjusted_return'] = df['daily_return'] / df['intraday_range']

    # Smooth the Adjusted Return with a 5-day simple moving average
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()

    # Output the smoothed value as the alpha factor
    return df['smoothed_adjusted_return']
```
