```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=20, k=20):
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['Close_Vol'] = df['close'] * df['volume']
    df['IRWAP'] = df['Close_Vol'] / df['volume']
    
    # Evaluate IRWAP Difference
    df['IRWAP_Diff'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['Intraday_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Adjusted_Volume'] * df['Log_Return']
    
    # Compare Latest Volume to M-day Average
    df['M_day_Avg_Volume'] = df['volume'].rolling(window=m).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['M_day_Avg_Volume'], 1, -1)
    
    # Compare Latest Amount to K-day Average
    df['K_day_Avg_Amount'] = df['amount'].rolling(window=k).mean()
    df['Amount_Support'] = np.where(df['amount'] > df['K_day_Avg_Amount'], 1, -1)
    
    # Generate the final factor
    df['Factor_Value'] = (
        df['Volume_Adjusted_Log_Return'] 
        + df['IRWAP_Diff'] 
        + df['Volume_Support'] 
        + df['Amount_Support']
    )
    
    return df['Factor_Value'].copy()
```
