```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Dynamics Analysis
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['IRWAP'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    df['Price_Volatility'] = df['Intraday_High_Low_Spread'].rolling(window=10).std()

    # Momentum and Return Adjustments
    df['Daily_Return'] = df['close'].pct_change()
    df['Smoothed_Daily_Return'] = df['Daily_Return'].rolling(window=5).mean()
    df['Short_Term_Momentum'] = np.log(df['close']).diff().rolling(window=7).sum()
    df['Long_Term_Momentum'] = np.log(df['close']).diff().rolling(window=25).sum()
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    df['Adjusted_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']

    # Confirm with Volume Trends
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend_Confirmation'] = np.where(df['volume'] > df['Avg_Volume_20'], 1, -1)

    # Incorporate Open-Price Based Momentum
    df['Open_Price_Momentum'] = np.log(df['open']).diff().rolling(window=7).sum()
    df['Open_Price_Combined_Momentum'] = df['Open_Price_Momentum'] - df['Long_Term_Momentum']

    # Introduce Volume-Weighted Close Price (VWCP)
    df['VWCP'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['VWCP_Difference'] = df['close'] - df['VWCP']

    # Final Factor Calculation
    df['Final_Factor'] = (
        df['IRWAP_Difference'] + 
        df['Smoothed_Daily_Return'] + 
        df['Adjusted_Momentum'] + 
        df['Volume_Trend_Confirmation'] + 
        df['Open_Price_Combined_Momentum'] + 
        df['VWCP_Difference']
    )

    return df['Final_Factor'].dropna()
```
