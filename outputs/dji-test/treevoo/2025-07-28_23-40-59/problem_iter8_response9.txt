import pandas as pd

def heuristics_v2(df):
    # Compute Price Change over N Days
    n = 7
    df['N-day Return'] = df['close'].pct_change(n)

    # Intraday Range and VWAP Analysis
    df['Intraday Range'] = df['high'] - df['low']
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP Difference'] = df['close'] - df['VWAP']

    # Combine Indicators for Intraday Analysis
    df['Intraday Analysis'] = df['Intraday Range'] + df['VWAP Difference']

    # Compare Latest Volume to Average
    m = 30
    df['M-day Avg Volume'] = df['volume'].rolling(window=m).mean()
    df['Volume Support'] = df.apply(lambda x: 1 if x['volume'] > x['M-day Avg Volume'] else -1, axis=1)

    # Dynamic Momentum Indicator with Adjustments
    df['Raw Momentum'] = df['close'] - df['close'].shift(7)
    df['Volume Factor'] = df['volume'] / df['volume'].rolling(window=7).mean()
    df['Volatility'] = df['close'].rolling(window=7).std()
    df['Adjusted Momentum'] = (df['Raw Momentum'] / df['Volatility']) * df['Volume Factor']

    # Calculate Volume-Adjusted Log Return
    df['High-Low Spread'] = df['high'] - df['low']

    df['Volume-Adjusted Log Return'] = df['Log Return'] * df['Adjusted Volume']
