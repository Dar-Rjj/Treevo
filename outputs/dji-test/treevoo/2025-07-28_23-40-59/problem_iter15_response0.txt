```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['weighted_close'] = df['close'] * df['volume']
    total_weighted_close = df['weighted_close'].rolling(window=1).sum()
    total_volume = df['volume'].rolling(window=1).sum()
    df['irwap'] = total_weighted_close / total_volume
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Calculate Adjusted Volume
    df['adjusted_volume'] = df['volume'] / df['intraday_spread']

    # Integrate Volume-Adjusted Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['adjusted_volume'] * df['log_return']
    
    # Compute Price Change over N Days
    N = 5
    df['n_day_return'] = df['close'].pct_change(periods=N)
    
    # Compare Latest Volume to Moving Average
    M = 20
    df['ma_volume'] = df['volume'].rolling(window=M).mean()
    df['volume_support'] = (df['volume'] > df['ma_volume']).astype(int) * 2 - 1
    
    # Compare Latest Amount to Exponential Moving Average
    K = 10
    df['ema_amount'] = df['amount'].ewm(span=K, adjust=False).mean()
    df['amount_support'] = (df['amount'] > df['ema_amount']).astype(int) * 2 - 1
    
    # Combine all components into a single alpha factor
    df['alpha_factor'] = (
        df['irwap_diff'] + 
        df['volume_adjusted_log_return'] + 
        df['n_day_return'] + 
        df['volume_support'] + 
        df['amount_support']
    )
    
    return df['alpha_factor']
```
