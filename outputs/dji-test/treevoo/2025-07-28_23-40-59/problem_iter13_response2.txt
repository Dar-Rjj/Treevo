```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_difference'] = df['vwap'] - df['close']
    
    # Combine Indicators for Intraday Analysis
    df['intraday_indicator'] = df['intraday_range'] + df['vwap_difference']
    
    # Calculate Volume-Adjusted Log Return
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['vol_adj_log_return'] = df['log_return'] * df['adjusted_volume']
    
    # Calculate Price Momentum
    N = 10  # Number of days for momentum
    df['price_momentum'] = df['close'] - df['close'].shift(N)
    
    # Confirm with Volume Trends
    M = 30  # Number of days for average volume
    df['avg_volume'] = df['volume'].rolling(window=M).mean()
    df['volume_trend_score'] = np.where(df['volume'] > df['avg_volume'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    LMA_window = 200
    SMA_window = 50
    df['lma'] = df['close'].rolling(window=LMA_window).mean()
    df['sma'] = df['close'].rolling(window=SMA_window).mean()
    df['ma_difference'] = df['sma'] - df['lma']
    
    # Apply Volume Weighted Filter
    df['vol_weighted_ma_diff'] = df['ma_difference'] * df['volume']
    df['smoothed_vol_weighted_ma_diff'] = df['vol_weighted_ma_diff'].ewm(span=10).mean()
    
    # Final Alpha Factor
    alpha_factor = (
        df['intraday_indicator'] 
        + df['vol_adj_log_return'] 
        + df['price_momentum'] 
        + df['volume_trend_score'] 
        + df['smoothed_vol_weighted_ma_diff']
    )
    
    return alpha_factor.dropna()
```
