```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['irwap'] = (df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Adjust Daily Return by Intraday Volatility
    df['daily_return'] = df['close'].pct_change()
    df['adjusted_return'] = df['daily_return'] / df['intraday_spread']
    
    # Smooth the Adjusted Return
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['short_term_momentum'] = df['log_return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['log_return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Introduce Price Volatility Component
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['adjusted_momentum'] = df['combined_momentum'] * (1 / df['price_volatility'])
    
    # Confirm with Volume Trends
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_score'] = np.where(df['volume'] > df['avg_volume_20'], 1, -1)
    
    # Calculate Intraday Strength
    df['intraday_strength'] = (df['close'] - df['open']) / df['intraday_spread']
    
    # Smooth Intraday Strength
    df['smoothed_intraday_strength'] = df['intraday_strength'].rolling(window=5).mean()
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['irwap_diff'] + 
        df['smoothed_adjusted_return'] + 
        df['adjusted_momentum'] + 
        df['smoothed_intraday_strength'] * df['volume_score']
    )
    
    return df['final_factor'].dropna()
```
