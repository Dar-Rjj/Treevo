```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff = sma - lma
    
    # Calculate Daily Price Range (High - Low)
    daily_range = df['high'] - df['low']
    
    # Multiply by Volume to get Volume-Adjusted Range
    volume_adjusted_range = daily_range * df['volume']
    
    # Sum of Volume-Adjusted Ranges
    sum_volume_adjusted_range = volume_adjusted_range.rolling(window=10).sum()
    
    # Smooth with Exponential Moving Average over 10 periods
    smoothed_range = sum_volume_adjusted_range.ewm(span=10, adjust=False).mean()
    
    # Final alpha factor
    alpha_factor = diff * smoothed_range
    
    return alpha_factor
```
