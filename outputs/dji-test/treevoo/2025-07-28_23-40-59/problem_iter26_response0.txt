```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['Volume_Adjusted_Close'] = df['close'] * df['volume']
    df['Sum_Volume_Adjusted_Close'] = df['Volume_Adjusted_Close'].rolling(window=1).sum()
    df['IRWAP'] = df['Sum_Volume_Adjusted_Close'] / df['volume'].rolling(window=1).sum()
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['Intraday_High_Low_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Volume_Adjusted_Log_Return'] = df['Adjusted_Volume'] * df['Log_Return']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Analysis'] = df['Intraday_High_Low_Spread'] + df['IRWAP_Difference']
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = df['Log_Return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = df['Log_Return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Confirm with Volume Trends
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend_Score'] = np.where(df['volume'] > df['Average_Volume_20'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['LMA'] = df['close'].rolling(window=200).mean()
    df['SMA'] = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['Volume_Weighted_Diff'] = (df['LMA'] - df['SMA']) * df['volume']
    df['Smoothed_Volume_Weighted_Diff'] = df['Volume_Weighted_Diff'].ewm(span=10).mean()
    
    # Adjust Daily Return by Intraday Volatility
    df['Daily_Return'] = df['close'] / df['close'].shift(1) - 1
    df['Intraday_Volatility_Adjusted_Return'] = df['Daily_Return'] / df['Intraday_High_Low_Spread']
    df['Smoothed_Adjusted_Return'] = df['Intraday_Volatility_Adjusted_Return'].rolling(window=5).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_Log_Returns'] = df['Daily_Log_Return'].ewm(span=10).mean()
    
    # Volume Change Momentum
    df['Volume_Change_Momentum'] = df['Volume_Change'].ewm(span=10).mean()
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['Weighted_Daily_Ranges'] = df['Daily_Range'].ewm(span=10).mean()
    
    # Combine Factors
    df['Final_Factor'] = (
        df['Weighted_Log_Returns'] *
        df['Volume_Change_Momentum'] *
        df['Weighted_Daily_Ranges'] *
        df['Adjusted_Combined_Momentum']
    )
    
    return df['Final_Factor']
```
