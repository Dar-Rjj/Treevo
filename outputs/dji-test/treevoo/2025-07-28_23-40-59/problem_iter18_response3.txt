```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    lma = df['close'].rolling(window=200).mean()
    sma = df['close'].rolling(window=50).mean()
    lma_sma_diff = lma - sma
    
    # Apply Volume Weighted Filter
    volume_weighted_diff = lma_sma_diff * df['volume']
    sum_volume_weighted_diff = volume_weighted_diff.rolling(window=10).sum()
    smoothed_volume_weighted_diff = sum_volume_weighted_diff.ewm(span=10, adjust=False).mean()
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / intraday_spread
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum and Long-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=7).sum()
    long_term_momentum = np.log(df['close']).diff().rolling(window=25).sum()
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    average_volume = df['volume'].rolling(window=20).mean()
    volume_trend_score = np.where(df['volume'] > average_volume, 1, -1)
    
    # Introduce Open-Price Based Momentum
    open_price_momentum = np.log(df['open']).diff().rolling(window=7).sum()
    open_price_combined_momentum = open_price_momentum - long_term_momentum
    
    # Integrate Intraday Range and Volume
    intraday_range = df['high'] - df['low']
    volume_ratio = df['volume'] / average_volume
    integrated_intraday_range_volume = intraday_range * volume_ratio
    
    # Final Factor Calculation
    final_factor = (
        irwap_diff + 
        smoothed_adjusted_return + 
        adjusted_combined_momentum + 
        smoothed_volume_weighted_diff + 
        open_price_combined_momentum + 
        integrated_intraday_range_volume
    ) * volume_trend_score
    
    return final_factor
```
