```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Weighted Daily Return
    df['volume_weighted_return'] = (df['daily_price_change'] * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Introduce Enhanced Price Directional Component
    df['close_open_diff'] = df['close'] - df['open']
    df['close_open_sign'] = np.sign(df['close_open_diff'])
    
    # Incorporate Dynamic Volume Spike Indicator
    mean_volume = df['volume'].rolling(window=N).mean()
    std_dev_volume = df['volume'].rolling(window=N).std()
    df['volume_spike'] = (df['volume'] > (mean_volume + 2 * std_dev_volume)).astype(int)
    df['dynamic_factor'] = 1 + (df['volume'] - mean_volume) / std_dev_volume
    df.loc[df['volume_spike'] == 0, 'dynamic_factor'] = 1
    df['adjusted_volume_weighted_return'] = df['volume_weighted_return'] * df['dynamic_factor']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Adjusted Volume
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']
    
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Combine Adjusted Volume, Log Return, and Enhanced Price Direction
    df['combined_component'] = df['log_return'] * df['adjusted_volume'] * df['close_open_sign'] * np.sign(df['high'] - df['close'])
    
    # Smooth the Combined Adjusted Volume, Log Return, and Enhanced Price Direction
    df['smoothed_combined_component'] = df['combined_component'].ewm(span=5).mean()
    
    # Incorporate Intraday Price Reversal
    df['intraday_price_reversal'] = (df['high'] - df['close']) / (df['close'] - df['low'])
    df['intraday_price_reversal_sign'] = np.sign(df['intraday_price_reversal'])
    
    # Combine Components into Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_volume_weighted_return'] + df['smoothed_combined_component']) * df['intraday_price_reversal_sign']
    
    return df['final_alpha_factor']
```
