```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between LMA and SMA
    diff_lma_sma = lma - sma
    
    # Apply Volume-Weighted Filter
    volume_weighted_diff = diff_lma_sma * df['volume']
    volume_weighted_sum = volume_weighted_diff.rolling(window=10).sum()
    smoothed_momentum = volume_weighted_sum.ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Factor
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_factor = df['volume'] / avg_volume_14
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=14).std()
    adjusted_smoothed_momentum = smoothed_momentum / (short_term_volatility + 1e-8) * volume_factor
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_daily_return = daily_return / (intraday_range + 1e-8)
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_daily_return.rolling(window=5).mean()
    
    # Integrate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_difference = (df['close'] - vwap) / (short_term_volatility + 1e-8)
    
    # Combine Indicators
    alpha_factor = (
        adjusted_smoothed_momentum +
        intraday_range * volume_factor +
        vwap_difference +
        smoothed_adjusted_return
    )
    
    return alpha_factor
```
