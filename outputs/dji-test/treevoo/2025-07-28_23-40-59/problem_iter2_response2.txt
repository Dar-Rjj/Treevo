```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Raw Momentum
    short_term_raw_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Raw Momentum
    long_term_raw_momentum = df['close'] - df['close'].shift(20)
    
    # Incorporate Volume Change
    volume_factor = df['volume'] / df['volume'].rolling(window=7).mean()
    short_term_adjusted_momentum = short_term_raw_momentum / volume_factor
    
    # Adjust for Volatility
    short_term_volatility = df['close'].rolling(window=7).std()
    volatility_adjusted_momentum = short_term_adjusted_momentum / short_term_volatility
    
    # Combine Short and Long-Term Momentum
    combined_momentum = volatility_adjusted_momentum - long_term_raw_momentum
    
    # Filter by Volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    filtered_momentum = combined_momentum.where(df['volume'] > avg_20_day_volume, 0)
    
    return filtered_momentum
```
