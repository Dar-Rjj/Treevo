```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Compute Volume-Averaged Daily Return
    df['Price_Change_Volume'] = df['Daily_Price_Change'] * df['volume']
    df['Volume_Sum'] = df['volume'].rolling(window=N).sum()
    df['Volume_Averaged_Return'] = df['Price_Change_Volume'].rolling(window=N).sum() / df['Volume_Sum']
    
    # Calculate Intraday Range Indicator
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Compute Daily VWAP
    df['VWAP_Numerator'] = (df['close'] * df['volume']).rolling(window=N).sum()
    df['VWAP'] = df['VWAP_Numerator'] / df['Volume_Sum']
    
    # Integrate Indicators
    df['Alpha_Factor'] = df['Intraday_Range'] + df['VWAP']
    
    return df['Alpha_Factor'].dropna()
```
