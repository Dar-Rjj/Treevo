```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term and Short-Term Moving Averages
    df['LMA'] = df['close'].rolling(window=200).mean()
    df['SMA'] = df['close'].rolling(window=50).mean()
    
    # Compute LMA-SMA Difference and Apply Volume-Weighted Filter
    df['LMA_SMA_Diff'] = (df['LMA'] - df['SMA']) * df['volume']
    df['Volume_Weighted_Sum'] = df['LMA_SMA_Diff'].rolling(window=10).sum()
    df['Volume_Weighted_EMA'] = df['Volume_Weighted_Sum'].ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    df['Adjusted_Return'] = df['Daily_Return'] / df['Intraday_Range']
    
    # Smooth the Adjusted Return
    df['Smoothed_Adjusted_Return'] = df['Adjusted_Return'].rolling(window=5).mean()
    
    # Confirm with Volume Trends
    df['Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Score'] = np.where(df['volume'] > df['Average_Volume'], 1, -1)
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(periods=30)
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_Log_Return'] = df['Log_Return'].ewm(span=10, adjust=False).mean()
    
    # Volume Change Momentum
    df['Volume_Change_Momentum'] = df['Volume_Change'].ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['Log_Return'].rolling(window=7).sum()
    df['Long_Term_Momentum'] = df['Log_Return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Filter by Volume
    df['Volume_Filter'] = np.where(df['volume'] > df['volume'].rolling(window=20).mean() * 1.5, 1, 0)
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['Weighted_Daily_Range'] = df['Daily_Range'].ewm(span=10, adjust=False).mean()
    
    # Combine Factors
    df['Alpha_Factor'] = (
        df['Weighted_Log_Return'] * 
        df['Volume_Change_Momentum'] * 
        df['Weighted_Daily_Range'] * 
        df['Adjusted_Combined_Momentum']
    )
    
    return df['Alpha_Factor']
```
