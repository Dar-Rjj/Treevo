```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Smoothed Momentum
    df['raw_momentum_7'] = df['close'] - df['close'].shift(7)
    df['momentum_14'] = df['close'] - df['close'].shift(14)
    df['smoothed_momentum'] = (df['raw_momentum_7'] + df['momentum_14']) / 2
    
    # Incorporate Volume Change
    df['avg_volume_14'] = df['volume'].rolling(window=14).mean()
    df['volume_factor'] = df['volume'] / df['avg_volume_14']
    
    # Adjust for Volatility
    df['short_term_volatility'] = df['close'].rolling(window=14).std()
    df['adjusted_momentum'] = df['smoothed_momentum'] * (1 / df['short_term_volatility'])
    df['final_momentum'] = df['adjusted_momentum'] * df['volume_factor']
    
    # Calculate Intraday Range Indicator
    df['intraday_range'] = (df['high'] - df['low']) * df['volume_factor']
    
    # Integrate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_diff'] = (df['close'] - df['daily_vwap']) * (1 / df['short_term_volatility'])
    
    # Combine Indicators
    df['alpha_factor'] = df['final_momentum'] + df['intraday_range'] + df['vwap_diff']
    
    return df['alpha_factor']
```
