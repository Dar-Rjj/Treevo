```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['weighted_log_return'] = df['log_return'].ewm(span=20, adjust=False).mean()
    
    # Volume Change Momentum
    df['weighted_volume_change'] = df['volume_change'].ewm(span=20, adjust=False).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['log_return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['log_return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Filter by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['volume_filtered'] = (df['volume'] > 1.5 * avg_volume_20)
    
    # Introduce Price Volatility Component
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['price_volatility']
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    df['weighted_daily_range'] = df['daily_range'].ewm(span=20, adjust=False).mean()
    
    # Calculate Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Weighted Moving Average of Open-Close Price Differences
    df['weighted_open_close_diff'] = df['open_close_diff'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    alpha_factor = (df['weighted_log_return'] * 
                    df['weighted_volume_change'] * 
                    df['weighted_daily_range'] * 
                    df['adjusted_combined_momentum'] * 
                    df['weighted_open_close_diff'])
    
    # Output the Final Alpha Factor
    return alpha_factor
```
