```python
import pandas as pd

def heuristics_v2(df, n=10, m=10, k=10):
    # Calculate N-Day Price Change using Close Prices
    df['n_day_price_change'] = df['close'].pct_change(n)
    
    # Compute M-Day Exponential Moving Average (EMA) Volume
    df['ema_volume'] = df['volume'].ewm(span=m, adjust=False).mean()
    
    # Compare Latest Volume to EMA Volume and Assign Score
    df['volume_score'] = df.apply(lambda row: 1 if row['volume'] > row['ema_volume'] else -1, axis=1)
    
    # Calculate K-Day Exponential Moving Average (EMA) Amount
    df['ema_amount'] = df['amount'].ewm(span=k, adjust=False).mean()
    
    # Compare Latest Amount to EMA Amount and Assign Score
    df['amount_score'] = df.apply(lambda row: 1 if row['amount'] > row['ema_amount'] else -1, axis=1)
    
    # Calculate LMA (200-Period Close Price Moving Average)
    df['lma'] = df['close'].rolling(window=200).mean()
    
    # Calculate SMA (50-Period Close Price Moving Average)
    df['sma'] = df['close'].rolling(window=50).mean()
    
    # Compute LMA-SMA Difference
    df['lma_sma_diff'] = df['lma'] - df['sma']
    
    # Apply EMA Smoothing (10-Period Exponential Moving Average)
    df['factor'] = df['lma_sma_diff'].ewm(span=10, adjust=False).mean() * df['volume_score'] * df['amount_score']
    
    return df['factor']
```
