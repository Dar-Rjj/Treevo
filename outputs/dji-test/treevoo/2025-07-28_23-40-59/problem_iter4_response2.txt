```python
def heuristics_v2(df):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Volume Factor
    avg_volume_7_days = df['volume'].rolling(window=7).mean()
    volume_factor = df['volume'] / avg_volume_7_days
    
    # Calculate Short-Term Price Volatility
    short_term_volatility = df['close'].rolling(window=7).std()
    
    # Weigh Raw Momentum by Inverse of Volatility
    adjusted_momentum = raw_momentum / short_term_volatility
    
    # Apply Volume Factor to Final Momentum
    final_momentum = adjusted_momentum * volume_factor
    
    return final_momentum
```
