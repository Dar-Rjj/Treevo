```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap_numerator = df['close'] * df['volume']
    irwap_denominator = df['volume']
    irwap = irwap_numerator.cumsum() / irwap_denominator.cumsum()
    
    # Evaluate IRWAP Difference
    irwap_difference = df['close'] - irwap
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / intraday_spread
    
    # Integrate Volume-Adjusted Log Return
    log_return = np.log(df['close']).diff()
    volume_adjusted_log_return = adjusted_volume * log_return
    
    # Construct the final alpha factor
    alpha_factor = irwap_difference + volume_adjusted_log_return
    
    return alpha_factor
```
