```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()

    # Compute Volume Change
    df['volume_change'] = df['volume'].pct_change()

    # Weighted Moving Average of Log Returns
    df['wma_log_return'] = df['log_return'].ewm(span=10, min_periods=10).mean()

    # Volume Change Momentum
    df['wma_volume_change'] = df['volume_change'].ewm(span=10, min_periods=10).mean()

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['log_return'].rolling(window=7, min_periods=7).sum()

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['log_return'].rolling(window=25, min_periods=25).sum()

    # Combine Short and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] - df['long_term_momentum']

    # Filter by Volume
    avg_volume = df['volume'].rolling(window=20, min_periods=20).mean()
    df['volume_filtered'] = df['volume'] >= 1.5 * avg_volume

    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10, min_periods=10).std()
    df['adj_combined_momentum'] = df['combined_momentum'] / price_volatility

    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']

    # Weighted Moving Average of Daily Ranges
    df['wma_daily_range'] = df['daily_range'].ewm(span=10, min_periods=10).mean()

    # Combine Factors
    df['final_alpha_factor'] = df['wma_log_return'] * df['wma_volume_change'] * df['wma_daily_range'] * df['adj_combined_momentum'] * df['volume_filtered']

    return df['final_alpha_factor']
```
