import pandas as pd

def heuristics_v2(df, n=5, lma_periods=200, sma_periods=50, ewm_span=10):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute Volume-Averaged Daily Return
    df['Volume_Averaged_Daily_Return'] = (df['Daily_Price_Change'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    df['LMA'] = df['close'].rolling(window=lma_periods).mean()
    df['SMA'] = df['close'].rolling(window=sma_periods).mean()
    
    # Calculate Difference Between LMA and SMA
    df['LMA_SMA_Difference'] = df['LMA'] - df['SMA']
    
    # Apply Volume-Weighted Filter
    df['Volume_Weighted_Difference'] = df['LMA_SMA_Difference'] * df['volume']
    
    # Sum of Volume-Weighted Differences
    df['Volume_Weighted_Sum'] = df['Volume_Weighted_Difference'].rolling(window=n).sum()
    
    # Smooth with Exponential Moving Average
    df['Factor'] = df['Volume_Weighted_Sum'].ewm(span=ewm_span).mean()
    
    return df['Factor']
