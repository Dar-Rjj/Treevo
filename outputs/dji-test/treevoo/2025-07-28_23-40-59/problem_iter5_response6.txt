import pandas as pd

def heuristics_v2(df, n_days=5, m_days=10):
    # Compute Price Change over N Days
    close_prices = df['close']
    n_day_return = (close_prices / close_prices.shift(n_days)) - 1
    
    # Compare Latest Volume to Average
    avg_volume = df['volume'].rolling(window=m_days).mean()
    latest_volume = df['volume']
    volume_support = (latest_volume > avg_volume).astype(int) * 2 - 1  # Positive if latest > average, negative otherwise
    
    # Dynamic Momentum Indicator
    raw_momentum = close_prices - close_prices.shift(7)
    volume_factor = latest_volume / avg_volume.rolling(window=7).mean()
    
    # Short-Term Price Volatility (standard deviation of Close prices over the past 7 days)
    short_term_volatility = close_prices.rolling(window=7).std()
    inv_volatility = 1 / short_term_volatility
    
    # Weigh Raw Momentum by Inverse of Volatility and Apply Volume Factor
    adjusted_momentum = (raw_momentum * inv_volatility * volume_factor) * volume_support
    
    return adjusted_momentum
