import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate short-term momentum (5-day rolling sum of log returns)
    short_term_momentum = df['log_return'].rolling(window=5).sum()
    
    # Calculate long-term momentum (20-day rolling sum of log returns)
    long_term_momentum = df['log_return'].rolling(window=20).sum()
    
    # Combine short and long-term momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Filter by volume (minimum volume threshold: average volume over 20 days)
    min_volume_threshold = df['volume'].rolling(window=20).mean()
    filtered_momentum = combined_momentum[df['volume'] > min_volume_threshold]
    
    # Return the factor values
    return filtered_momentum.dropna()
