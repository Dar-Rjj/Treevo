```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['IRWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Evaluate IRWAP Difference
    df['IRWAP_Difference'] = df['close'] - df['IRWAP']
    
    # Calculate Adjusted Volume
    df['Adjusted_Volume'] = df['volume'] / df['Intraday_High_Low_Spread']
    
    # Integrate Volume-Adjusted Log Return
    df['Log_Return'] = np.log(df['close']).diff()
    df['Volume_Adjusted_Log_Return'] = df['Adjusted_Volume'] * df['Log_Return']
    
    # Combine Indicators for Intraday Analysis
    df['Intraday_Analysis'] = df['Intraday_High_Low_Spread'] + df['IRWAP_Difference']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(10)
    
    # Compare Latest Volume to Moving Average
    df['Volume_MA'] = df['volume'].rolling(window=20).mean()
    df['Volume_Support'] = np.where(df['volume'] > df['Volume_MA'], 1, -1)
    
    # Calculate Long-Term (LMA) and Short-Term (SMA) Moving Averages
    df['LMA'] = df['close'].rolling(window=200).mean()
    df['SMA'] = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['Volume_Weighted_Diff'] = (df['SMA'] - df['LMA']) * df['volume']
    df['Volume_Weighted_EMA'] = df['Volume_Weighted_Diff'].ewm(span=10).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']).diff()
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    df['Weighted_Log_Returns'] = df['Daily_Log_Return'].ewm(span=10).mean()
    
    # Volume Change Momentum
    df['Volume_Change_Momentum'] = df['Volume_Change'].ewm(span=10).mean()
    
    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = df['Daily_Log_Return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['Combined_Momentum'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Filter by Volume
    df['Volume_Filter'] = df['volume'] > 1.5 * df['volume'].rolling(window=20).mean()
    
    # Introduce Price Volatility Component
    df['Price_Volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # Introduce Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Combine Factors
    df['Final_Factor'] = (df['Weighted_Log_Returns'] * df['Volume_Change_Momentum'] * 
                          df['Adjusted_Combined_Momentum'] * df['RSI'])
    
    return df['Final_Factor']
```
