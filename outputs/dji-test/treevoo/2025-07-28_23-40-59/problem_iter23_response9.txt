```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = df['close'].pct_change()
    adjusted_return = daily_return / (df['high'] - df['low'])
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    log_returns = np.log(df['close']).diff()
    short_term_momentum = log_returns.rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = log_returns.rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Confirm with Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Calculate Intraday Strength
    intraday_strength = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Smooth Intraday Strength
    smoothed_intraday_strength = intraday_strength.rolling(window=5).mean()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Volume-Weighted Daily Return
    volume_weighted_daily_return = (daily_price_change * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Introduce Enhanced Price Directional Component
    close_open_diff = df['close'] - df['open']
    close_open_sign = np.sign(close_open_diff)
    
    # Incorporate Dynamic Volume Spike Indicator
    volume_std_dev = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    is_volume_spike = df['volume'] > (volume_mean + 2 * volume_std_dev)
    dynamic_factor = 1 + (df['volume'] - volume_mean) / volume_std_dev
    volume_weighted_daily_return[is_volume_spike] *= dynamic_factor[is_volume_spike]
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / (df['high'] - df['low'])
    
    # Calculate Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Combine Adjusted Volume, Log Return, and Enhanced Price Direction
    combined_adjusted_volume_log_return = log_return * adjusted_volume * close_open_sign * np.sign(df['high'] - df['close'])
    
    # Smooth the Combined Adjusted Volume, Log Return, and Enhanced Price Direction
    smoothed_combined_adjusted_volume_log_return = combined_adjusted_volume_log_return.ewm(span=5).mean()
    
    # Final Factor Calculation
    final_factor = (
        irwap_diff +
        smoothed_adjusted_return +
        adjusted_combined_momentum +
        smoothed_intraday_strength +
        volume_weighted_daily_return +
        smoothed_combined_adjusted_volume_log_return
    )
    
    return final_factor
```
