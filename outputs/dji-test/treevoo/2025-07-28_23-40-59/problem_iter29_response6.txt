def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as dates.
    
    Returns:
    pd.Series: A Series indexed by date representing the alpha factor values.
    """
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the close-open difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate the volume-weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Heuristic: High-Low Range Adjusted by Close-Open Difference and Volume-Weighted Close
    heuristic_factor = (high_low_range + close_open_diff) / (volume_weighted_close + 1e-6)
    
    return heuristic_factor
