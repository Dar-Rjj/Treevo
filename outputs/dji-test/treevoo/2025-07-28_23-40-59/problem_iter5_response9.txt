```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Daily VWAP
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_difference = df['close'] - vwap
    
    # Combine Indicators for Intraday Analysis
    intraday_analysis = intraday_range + vwap_difference
    
    # Calculate Volume-Adjusted Log Return
    high_low_spread = df['high'] - df['low']
    adjusted_volume = df['volume'] / high_low_spread
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    volume_adjusted_log_return = log_return * adjusted_volume
    
    # Final alpha factor
    alpha_factor = intraday_analysis + volume_adjusted_log_return
    
    return alpha_factor
```
