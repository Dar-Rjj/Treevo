```python
import pandas as pd

def heuristics_v2(df, n_days=10, m_days=30):
    # Compute Price Change over N Days
    close_prices = df['close']
    n_day_returns = close_prices.pct_change(periods=n_days)
    
    # Compare Latest Volume to Average
    avg_volume = df['volume'].rolling(window=m_days).mean()
    latest_volume = df['volume']
    
    volume_support = (latest_volume > avg_volume) * 1 - (latest_volume < avg_volume) * 1
    
    # Combine the two factors
    factor_values = n_day_returns * volume_support
    
    return factor_values
```
