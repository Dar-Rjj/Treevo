import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['irwap'] = (df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Calculate Adjusted Volume
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']
    
    # Integrate Volume-Adjusted Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['adjusted_volume'] * df['log_return']
    
    # Compare Latest Volume to M-day Average
    m_day_avg_vol = df['volume'].rolling(window=20).mean()
    df['volume_score'] = np.where(df['volume'] > m_day_avg_vol, 1, -1)
    
    # Compare Latest Amount to K-day Average
    k_day_avg_amount = df['amount'].rolling(window=20).mean()
    df['amount_score'] = np.where(df['amount'] > k_day_avg_amount, 1, -1)
    
    # Calculate Long-Term Moving Average (LMA) and Short-Term Moving Average (SMA)
    df['lma'] = df['close'].rolling(window=200).mean()
    df['sma'] = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Difference Between LMA and SMA
    df['vol_weighted_diff_lma_sma'] = (df['lma'] - df['sma']) * df['volume']
    df['smoothed_vol_weighted_diff'] = df['vol_weighted_diff_lma_sma'].ewm(span=10).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['log_return'].rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['log_return'].rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Filter by Volume
    avg_vol_20_days = df['volume'].rolling(window=20).mean()
    df['filtered_combined_momentum'] = np.where(df['volume'] > 1.5 * avg_vol_20_days, df['combined_momentum'], np.nan)
    
    # Adjust Daily Return by Intraday Volatility
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['adjusted_return'] = df['daily_return'] / df['high_low_spread']
    
    # Smooth the Adjusted Return
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=7).mean()
    
    # Introduce Price Volatility Component
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=10).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['price_volatility']
    
    # Confirm with Volume Trends
    df['avg_vol_20_days'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_confirmation'] = np.where(df['volume'] > df['avg_vol_20_days'], 1, -1)
    
    # Introduce Open-Price Based Momentum
    df['open_price_momentum'] = np.log(df['open'] / df['open'].shift(14)).rolling(window=14).sum()
    
    # Combine Open-Price Based Momentum
    df['open_close_momentum'] = df['open_price_momentum'] - df['long_term_momentum']
    
    # Introduce Open-Close Price Difference
    df['open_close_diff'] = (df['open'] - df['close']) / df['close']
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['irwap_diff'] + 
        df['smoothed_adjusted_return'] + 
        df['adjusted_combined_momentum'] + 
        df['volume_trend_confirmation'] + 
        df['open_close_momentum'] + 
        df['open_close_diff']
    ) * df['smoothed_vol_weighted_diff']
    
    return
