```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Compute Weighted Intraday Volatility
    weighted_intraday_volatility = df['volume'] * intraday_range
    
    # Aggregate Intraday Volatility (Smooth with Moving Average)
    smoothed_intraday_volatility = weighted_intraday_volatility.rolling(window=20).mean()
    
    # Combine with Close-to-Open Return
    factor = (close_to_open_return / smoothed_intraday_volatility).abs()
    
    return factor
```
