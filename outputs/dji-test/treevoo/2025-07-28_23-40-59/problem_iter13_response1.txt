```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=5, m_days=10, k_days=10):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    df['weighted_close'] = df['close'] * df['volume']
    total_weighted_close = df['weighted_close'].rolling(window=n_days).sum()
    total_volume = df['volume'].rolling(window=n_days).sum()
    df['irwap'] = total_weighted_close / total_volume
    
    # Evaluate IRWAP Difference
    df['irwap_diff'] = df['close'] - df['irwap']
    
    # Calculate Adjusted Volume
    df['adjusted_volume'] = df['volume'] / df['high_low_spread']
    
    # Integrate Volume-Adjusted Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_adjusted_log_return'] = df['adjusted_volume'] * df['log_return']
    
    # Compute Price Change over N Days
    df['n_day_return'] = df['close'].pct_change(n_days)
    
    # Compare Latest Volume to Average
    df['m_day_avg_volume'] = df['volume'].rolling(window=m_days).mean()
    df['volume_support'] = np.where(df['volume'] > df['m_day_avg_volume'], 1, -1)
    
    # Compare Latest Amount to Average
    df['k_day_avg_amount'] = df['amount'].rolling(window=k_days).mean()
    df['amount_support'] = np.where(df['amount'] > df['k_day_avg_amount'], 1, -1)
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (
        df['irwap_diff'] + 
        df['volume_adjusted_log_return'] + 
        df['n_day_return'] + 
        df['volume_support'] + 
        df['amount_support']
    )
    
    return df['alpha_factor']
```
