```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Moving Average (LMA) over 200 periods
    lma = df['close'].rolling(window=200).mean()
    
    # Calculate Short-Term Moving Average (SMA) over 50 periods
    sma = df['close'].rolling(window=50).mean()
    
    # Compute LMA-SMA Difference
    lma_sma_diff = lma - sma
    
    # Apply Volume-Weighted Filter to LMA-SMA Difference
    volume_weighted_diff = lma_sma_diff * df['volume']
    volume_weighted_sum = volume_weighted_diff.ewm(span=10).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Intraday Volatility
    adjusted_daily_return = daily_return / intraday_range
    
    # Smooth the Adjusted Return with a 5-day simple moving average
    smoothed_adjusted_return = adjusted_daily_return.rolling(window=5).mean()
    
    # Confirm with Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff()
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']).diff()
    
    # Compute Volume Change
    volume_change = df['volume'].pct_change()
    
    # Weighted Moving Average of Log Returns
    weighted_log_returns = log_return.ewm(span=5).mean()
    
    # Volume Change Momentum
    volume_change_momentum = volume_change.ewm(span=5).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = log_return.rolling(window=7).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = log_return.rolling(window=25).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = short_term_momentum - long_term_momentum
    
    # Filter by Volume
    min_vol_threshold = 1.5 * df['volume'].rolling(window=20).mean()
    combined_momentum_filtered = np.where(df['volume'] > min_vol_threshold, combined_momentum, 0)
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_momentum = combined_momentum_filtered / price_volatility
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Weighted Moving Average of Daily Ranges
    weighted_daily_ranges = daily_range.ewm(span=5).mean()
    
    # Dynamic Momentum Indicator
    raw_momentum = df['close'] - df['close'].shift(7)
    volume_factor = df['volume'] / df['volume'].rolling(window=7).mean()
    short_term_price_volatility = df['close'].rolling(window=7).std()
    dynamic_momentum = (raw_momentum / short_term_price_volatility) * volume_factor
    
    # Final Factor Calculation
    final_factor = (smoothed_adjusted_return * volume_weighted_sum * volume_trend *
                    weighted_log_returns * volume_change_momentum * weighted_daily_ranges * 
                    adjusted_combined_momentum * dynamic_momentum)
    
    return final_factor
```
