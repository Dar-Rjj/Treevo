```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Intraday Range Weighted Average Price (IRWAP)
    irwap = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Evaluate IRWAP Difference
    irwap_diff = df['close'] - irwap
    
    # Adjust Daily Return by Intraday Volatility
    daily_return = (df['close'] / df['open']) - 1
    adjusted_return = daily_return / intraday_spread
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = np.log(df['close']).diff().rolling(window=5).sum()
    
    # Calculate Long-Term Momentum
    long_term_momentum = np.log(df['close']).diff().rolling(window=20).sum()
    
    # Combine Short and Long-Term Momentum
    combined_momentum = (short_term_momentum - long_term_momentum) * 1.2
    
    # Introduce Price Volatility Component
    price_volatility = (df['high'] - df['low']).rolling(window=14).std()
    adjusted_combined_momentum = combined_momentum / (price_volatility + 1)
    
    # Confirm with Volume Trends
    volume_trend = df['volume'].rolling(window=30).mean()
    volume_trend_confirmation = np.where(df['volume'] > volume_trend, 1, -1)
    
    # Introduce Price-Volatility Interaction
    price_change = (df['high'] - df['low']) / df['close']
    interaction = price_change * price_volatility
    
    # Final Factor Calculation
    final_factor = (irwap_diff 
                    + smoothed_adjusted_return 
                    + adjusted_combined_momentum 
                    + volume_trend_confirmation 
                    + interaction)
    
    return final_factor
```
