```python
def heuristics_v2(df):
    # Calculate the weighted price using open, high, low, and close
    df['weighted_price'] = (df['open'] + 2 * df['high'] + 2 * df['low'] + df['close']) / 6
    
    # Calculate the price change from the previous day
    df['price_change'] = df['weighted_price'].diff()
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'].diff()
    
    # Calculate the price momentum as the sum of the last 5 days' price changes
    df['price_momentum'] = df['price_change'].rolling(window=5).sum()
    
    # Calculate the volume momentum as the sum of the last 5 days' volume changes
    df['volume_momentum'] = df['volume_change'].rolling(window=5).sum()
    
    # Heuristic: The factor is the product of price momentum and volume momentum
    df['alpha_factor'] = df['price_momentum'] * df['volume_momentum']
    
    return df['alpha_factor']
```
