```python
import pandas as pd

def heuristics_v2(df, n=10, m=20, k=20, lma_period=200, sma_period=50, ema_smooth_period=10):
    # Calculate N-Day Price Change
    price_change = df['close'].pct_change(n)
    
    # Compute M-Day Exponential Moving Average (EMA) Volume
    ema_volume = df['volume'].ewm(span=m, adjust=False).mean()
    volume_score = (df['volume'] - ema_volume).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate K-Day Exponential Moving Average (EMA) Amount
    ema_amount = df['amount'].ewm(span=k, adjust=False).mean()
    amount_score = (df['amount'] - ema_amount).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate LMA-SMA Difference
    lma = df['close'].rolling(window=lma_period).mean()
    sma = df['close'].rolling(window=sma_period).mean()
    lma_sma_diff = lma - sma
    
    # Apply EMA Smoothing to LMA-SMA Difference
    smoothed_diff = lma_sma_diff.ewm(span=ema_smooth_period, adjust=False).mean()
    
    # Combine all the components into a single factor
    combined_factor = price_change * volume_score * amount_score * smoothed_diff
    
    return combined_factor
```
