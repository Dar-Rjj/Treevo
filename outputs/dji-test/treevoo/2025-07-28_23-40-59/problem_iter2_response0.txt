```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Log Return
    log_return = np.log(data['close']) - np.log(data['close'].shift(1))
    
    # Adjust Log Return by Intraday Volatility
    adjusted_log_return = log_return / intraday_range
    
    # Calculate Adjusted Volume
    adjusted_volume = data['volume'] / intraday_range
    
    # Multiply Adjusted Log Return by Adjusted Volume to generate the hybrid factor
    hybrid_factor = adjusted_log_return * adjusted_volume
    
    return hybrid_factor
```
