```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
        
    Returns:
        pd.Series: A Series of alpha factor values indexed by date.
    """
    
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the intraday range
    close_to_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate the volume weighted price
    vwp = (df['close'] * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Calculate the momentum of the volume weighted price
    vwp_momentum = vwp.pct_change(periods=10)
    
    # Combine the factors into a single alpha factor
    alpha_factor = close_to_range * vwp_momentum
    
    return alpha_factor
```
