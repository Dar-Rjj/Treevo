```python
def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Opening Price Trend
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Opening Movements
    combined_movement = intraday_movement + opening_trend
    
    # Adjust for Volume
    factor_values = combined_movement * df['volume']
    
    return factor_values
```
