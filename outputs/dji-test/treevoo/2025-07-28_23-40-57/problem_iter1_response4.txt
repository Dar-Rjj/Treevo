```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term EMA (12 days) and long-term EMA (26 days)
    df['EMA_short'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_long'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the difference between EMAs
    df['EMA_diff'] = df['EMA_short'] - df['EMA_long']
    
    # Calculate Volume Rate of Change (VROC) over 14 days
    df['VROC'] = df['volume'].pct_change(periods=14)
    
    # Combine the trend indicator and VROC
    df['alpha_factor'] = df['EMA_diff'] * df['VROC']
    
    return df['alpha_factor']
```
