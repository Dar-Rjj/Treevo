```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Trend
    previous_close = df['close'].shift(1)
    opening_trend = df['open'] - previous_close
    
    # Combine Movements and Weight by Volume
    combined_movements = intraday_movement + opening_trend
    factor_values = combined_movements * df['volume']
    
    return factor_values
```
