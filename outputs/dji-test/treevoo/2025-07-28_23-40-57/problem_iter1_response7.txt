```python
def heuristics_v2(df):
    # Calculate Intraday Returns
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Trend
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Combine Intraday Returns and Volume Trend
    df['alpha_factor'] = df['intraday_return'] * df['volume_ratio']
    
    return df['alpha_factor']
```
