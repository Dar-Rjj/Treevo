import pandas as pd

def heuristics_v2(df, n_days=10, short_term_days=5):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Price Impact
    volume_sum = df['volume'].rolling(window=n_days).sum()
    high_avg = df['high'].rolling(window=n_days).mean()
    low_avg = df['low'].rolling(window=n_days).mean()
    price_impact = (high_avg - low_avg) * volume_sum
    
    # Adjust Intraday Return by Volume-Weighted Price Impact
    df['adjusted_intraday_return'] = df['intraday_return'] / price_impact
    factor_1 = df['adjusted_intraday_return']
    
    # Calculate Cumulative Volume-Weighted Intraday Return
    weighted_intraday_return = df['intraday_return'] * df['volume']
    cumulative_weighted_return = weighted_intraday_return.rolling(window=n_days).sum()
    factor_2 = cumulative_weighted_return
    
    # Calculate Short-Term Volume Variability
    volume_volatility = df['volume'].rolling(window=short_term_days).std()
    
    # Final Factor
    final_factor = factor_1 + factor_2
    
    return final_factor
