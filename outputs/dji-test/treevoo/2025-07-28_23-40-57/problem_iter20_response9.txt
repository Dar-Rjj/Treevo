```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Opening Price Trend
    opening_price_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Intraday Movement (VWIM)
    vwim = intraday_high_low_spread * df['volume']
    vwim_5d_ma = vwim.rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend (AWOT)
    awot = opening_price_trend * df['amount']
    awot_5d_ma = awot.rolling(window=5).mean()
    
    # Combine Adjusted Movements and Trends (AMT)
    amt = (vwim + awot) - vwim_5d_ma - awot_5d_ma
    
    # Adjust by Cumulative Volume (CV)
    n_days = 5
    cv = df['volume'].rolling(window=n_days).sum()
    adjusted_amt = amt / cv
    
    # Generate Factor 1
    volume_sum_10d = df['volume'].rolling(window=10).sum()
    high_avg_10d = df['high'].rolling(window=10).mean()
    low_avg_10d = df['low'].rolling(window=10).mean()
    volume_impact_score = (high_avg_10d - low_avg_10d) / volume_sum_10d
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Confirm with Volume Trend
    av_15d = df['volume'].rolling(window=15).mean()
    vr = df['volume'] / av_15d
    
    # Compute Final Factor
    if vr > 1.3:
        final_factor = factor_1 * adjusted_amt
    else:
        final_factor = 0.6 * (factor_1 + adjusted_amt)
    
    # Calculate Cumulative Volume-Weighted Momentum
    m_days = 20
    cvwm = (adjusted_amt * df['volume']).rolling(window=m_days).sum()
    
    # Enhanced Volume Trend Confirmation
    enhanced_vr = df['volume'] / av_15d
    if enhanced_vr > 1.5:
        combined_momentum = price_momentum * intraday_momentum
        aggregate_momentum = cvwm.sum()
        final_factor = combined_momentum * aggregate_momentum
    else:
        final_factor = 0.3 * (price_momentum + intraday_momentum)
    
    return final_factor
```
