```python
def heuristics_v2(df):
    # Calculate the average of (open, close) prices and subtract it by the volume
    avg_open_close = (df['open'] + df['close']) / 2
    factor = avg_open_close - df['volume']
    
    # Introduce a multiplier based on the difference between high and low
    # The idea is that a wider range (high - low) might indicate higher volatility or interest
    range_multiplier = df['high'] - df['low']
    
    # Final alpha factor
    factor *= range_multiplier
    
    return factor
```
