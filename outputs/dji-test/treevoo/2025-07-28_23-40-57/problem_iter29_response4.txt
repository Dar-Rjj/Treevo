def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    - pd.Series: A Series with the alpha factor values, indexed by date.
    """
    # Calculate the midpoint of the high and low prices
    midpoint = (data['high'] + data['low']) / 2
    
    # Calculate the difference between the close price and the midpoint
    close_mid_diff = data['close'] - midpoint
    
    # Calculate the average volume over a 5-day window
    avg_volume_5d = data['volume'].rolling(window=5).mean()
    
    # Calculate the relative change in volume compared to the 5-day average
    rel_vol_change = (data['volume'] - avg_volume_5d) / avg_volume_5d
    
    # Combine the two components to form the final alpha factor
    alpha_factor = close_mid_diff * rel_vol_change
    
    return alpha_factor
