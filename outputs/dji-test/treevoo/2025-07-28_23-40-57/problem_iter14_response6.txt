import pandas as pd

def heuristics_v2(df, n_days=10, m_days=30):
    # Calculate Intraday Volume Proportion
    df['intraday_volume'] = df['volume'] - (df['volume'].shift(1) + df['volume'].diff())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['price_range'] = df['high'] - df['low']
    df['intraday_vol_intensity'] = df['intraday_volume'] / df['price_range']
    
    # Calculate Volume Impact Score
    volume_sum_n_days = df['volume'].rolling(window=n_days).sum()
    avg_high_n_days = df['high'].rolling(window=n_days).mean()
    avg_low_n_days = df['low'].rolling(window=n_days).mean()
    price_range_avg = avg_high_n_days - avg_low_n_days
    df['volume_impact_score'] = price_range_avg / volume_sum_n_days
    
    # Multiply Intraday Return by Volume Impact Score
    df['factor_intraday'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_vol_intensity'] = df['intraday_vol_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Volume
    cum_volume = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['momentum'] / cum_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['cum_vol_weighted_momentum'] = (df['adjusted_momentum'] * df['volume']).rolling(window=n_days).sum()
    
    # Aggregate Momentum Over M Days
    df['aggregated_momentum'] = df['cum_vol_weighted_momentum'].rolling(window=m_days).sum()
    
    return df['aggregated_momentum']
