```python
def heuristics_v2(df, N=5, M=3):
    # Calculate Intraday Breakout Momentum
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Intraday_Breakout_Momentum'] = df['High_Low_Spread'] / df['close'].shift(1)
    
    # Adjust by Volume
    df['Cumulative_Volume'] = df['volume'].rolling(window=N).sum()
    df['Adjusted_Intraday_Breakout_Momentum'] = df['Intraday_Breakout_Momentum'] / df['Cumulative_Volume']
    
    # Aggregate Momentum Over M Days
    df['Aggregated_Momentum'] = df['Adjusted_Intraday_Breakout_Momentum'].rolling(window=M).sum()
    
    # Calculate Volume Trend
    df['Average_Volume_5D'] = df['volume'].rolling(window=5).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_5D']
    
    # Combine Aggregated Momentum and Volume Trend
    df['Alpha_Factor'] = df['Aggregated_Momentum'] * df['Volume_Ratio']
    
    return df['Alpha_Factor']
```
