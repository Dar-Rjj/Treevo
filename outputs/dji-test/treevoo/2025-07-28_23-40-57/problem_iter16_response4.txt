```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day moving average of close prices
    df['MA20'] = df['close'].rolling(window=20).mean()
    
    # Signal: Close price - MA20
    df['signal'] = df['close'] - df['MA20']
    
    # Daily volatility: High - Low
    df['daily_volatility'] = df['high'] - df['low']
    
    # 20-day moving average of volume
    df['VMA20'] = df['volume'].rolling(window=20).mean()
    
    # Volume ratio: Volume / VMA20
    df['volume_ratio'] = df['volume'] / df['VMA20']
    
    # Change in closing price from t-1 to t
    df['price_change'] = df['close'].diff()
    
    # Ratio of close price at t to the open price at t
    df['intraday_ratio'] = df['close'] / df['open']
    
    # Heuristic factor: Combine all indicators
    df['heuristic_factor'] = (
        df['signal'] +
        df['daily_volatility'] +
        (df['volume_ratio'] - 1) * 100 +
        df['price_change'] +
        (df['intraday_ratio'] - 1) * 100
    )
    
    return df['heuristic_factor']
```
