import pandas as pd

def heuristics_v2(df, n_days=10, short_term=20, long_term=50):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=n_days).sum()
    avg_high = df['high'].rolling(window=n_days).mean()
    avg_low = df['low'].rolling(window=n_days).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Multiply Intraday Return by Volume Impact Score
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(n_days)
    intraday_movement = df['high'] - df['low']
    
    # Adjust for Combined Movements
    combined_movements = (intraday_movement + (df['close'] - df['open'])) * df['volume']
    
    # Confirm with Volume Trend
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume
    price_and_intraday_momentum = price_momentum * intraday_return
    factor_2 = price_and_intraday_momentum.where(volume_ratio > 1, 0)
    
    # Combine Factors
    final_factor_1 = factor_1 * factor_2
    
    # Calculate Cumulative Volume Weighted Intraday Return
    cumulative_volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=n_days).sum()
    
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=short_term).mean()
    long_term_ma = df['close'].rolling(window=long_term).mean()
    
    # Calculate Moving Average Crossover
    moving_average_crossover = short_term_ma - long_term_ma
    
    # Synthesize Final Factors
    final_factor_2 = cumulative_volume_weighted_intraday_return * moving_average_crossover
    final_factor = final_factor_1 * final_factor_2
    
    return final_factor
