```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    prev_close = df['close'].shift(1)
    momentum = high_low_spread / prev_close
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    avg_volume = df['volume'].rolling(window=10).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume
    
    if_volume_higher = (volume_ratio > 1.2)
    combined_momentum = if_volume_higher * (price_momentum * intraday_return) + \
                       (~if_volume_higher) * 0.5 * (price_momentum + intraday_return)
    
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday = (intraday_movement * df['volume']).rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_opening = (opening_trend * df['amount']).rolling(window=5).mean()
    
    # Combine Weighted Movements and Trends
    combined_weighted = volume_weighted_intraday + amount_weighted_opening - \
                        volume_weighted_intraday.rolling(window=5).mean() - \
                        amount_weighted_opening.rolling(window=5).mean()
    
    # Introduce Relative Strength Component
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    relative_strength = ma_5 / ma_20
    
    # Incorporate RS into Momentum Calculation
    adjusted_price_momentum = price_momentum * relative_strength
    adjusted_intraday_return = intraday_return * relative_strength
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + combined_weighted + adjusted_price_momentum + adjusted_intraday_return
    return alpha_factor
```
