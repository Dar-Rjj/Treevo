```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Opening Price Trend
    df['opening_price_trend'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    df['volume_weighted_intraday_movement'] = df['intraday_high_low_spread'] * df['volume']
    df['volume_weighted_intraday_moving_avg'] = df['volume_weighted_intraday_movement'].rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    df['amount_weighted_opening_trend'] = df['opening_price_trend'] * df['amount']
    df['amount_weighted_opening_trend_moving_avg'] = df['amount_weighted_opening_trend'].rolling(window=5).mean()
    
    # Combine Adjusted Movements and Trends
    df['combined_movements_and_trends'] = (df['volume_weighted_intraday_movement'] + df['amount_weighted_opening_trend']) - (df['volume_weighted_intraday_moving_avg'] + df['amount_weighted_opening_trend_moving_avg'])
    
    # Adjust by Volume
    N = 15
    df['cumulative_volume'] = df['volume'].rolling(window=N).sum()
    df['adjusted_momentum'] = df['combined_movements_and_trends'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=N).sum()
    
    # Confirm with Volume Trend
    df['average_volume'] = df['volume'].rolling(window=15).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = df.apply(lambda row: row['adjusted_momentum'] * row['combined_movements_and_trends'] if row['volume_ratio'] > 1.3 else 0.6 * (row['adjusted_momentum'] + row['combined_movements_and_trends']), axis=1)
    
    return df['alpha_factor']
```
