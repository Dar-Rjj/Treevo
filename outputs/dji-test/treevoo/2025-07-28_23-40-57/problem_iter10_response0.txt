import pandas as pd

def heuristics_v2(df, M=5):
    # Intraday Movement
    df['Intraday_Movement'] = df['high'] - df['low']
    
    # Opening Price Change
    df['Opening_Price_Change'] = df['open'] - df['close'].shift(1)
    
    # Adjust for Intraday Breakout
    df['Adjusted_Breakout'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Momentum
    df['Combined_Momentum'] = (df['Intraday_Movement'] + 
                               df['Opening_Price_Change'] + 
                               df['Adjusted_Breakout']) * df['volume']
    
    # Aggregated Momentum Over M Days
    df['Aggregated_Momentum'] = df['Combined_Momentum'].rolling(window=M).sum()
    
    # Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Final Factor
    df['Final_Factor'] = df['Aggregated_Momentum'] * df['Volume_Ratio']
    
    return df['Final_Factor']
