```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    intraday_volume_proportion = (df['volume'] - df['volume'].shift(1)) / (df['volume'] - df['open'].shift(1) * df['close'].shift(1))
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = intraday_volume_proportion / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    n_days = 10
    sum_volume = df['volume'].rolling(window=n_days).sum()
    avg_high = df['high'].rolling(window=n_days).mean()
    avg_low = df['low'].rolling(window=n_days).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Multiply Intraday Return by Volume Impact Score
    weighted_intraday_return = intraday_return * volume_impact_score
    
    # Calculate Weighted Intraday Volume Intensity
    weighted_intraday_volume_intensity = intraday_volume_intensity * weighted_intraday_return
    
    return weighted_intraday_volume_intensity
```
