```python
import pandas as pd

def heuristics_v2(df):
    n = 10
    m = 5
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    volume_sum_n_days = df['volume'].rolling(window=n).sum()
    avg_high_n_days = df['high'].rolling(window=n).mean()
    avg_low_n_days = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high_n_days - avg_low_n_days) / volume_sum_n_days
    
    # Intermediate Factor (Factor 1)
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(n)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    adjusted_momentum = price_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=n).sum()
    
    # Confirm with Volume Trend
    average_volume_5_days = df['volume'].rolling(window=m).mean()
    current_day_volume = df['volume']
    volume_ratio = current_day_volume / average_volume_5_days
    
    # Combine Price and Intraday Momentum
    combined_momentum = price_momentum * intraday_return
    aggregated_momentum = volume_weighted_momentum.rolling(window=m).sum()
    
    # Final Factor
    final_factor = pd.Series(0, index=df.index)
    final_factor[volume_ratio > 1] = factor_1 * aggregated_momentum
    
    return final_factor
```
