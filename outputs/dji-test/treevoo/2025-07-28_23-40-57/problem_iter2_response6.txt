```python
import pandas as pd

def heuristics_v2(df, short_term=12, long_term=26, vroc_period=14, n_days=10, m_days=30):
    # Calculate Exponential Moving Averages (EMA) of Close Prices
    df['short_ema'] = df['close'].ewm(span=short_term, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=long_term, adjust=False).mean()
    
    # Calculate the difference between EMAs
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['breakout_momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust Trend Indicator by Volume
    df['vroc'] = df['volume'].pct_change(periods=vroc_period)
    df['adjusted_trend'] = df['ema_diff'] * df['vroc']
    
    # Adjust Breakout Momentum by Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n_days).sum()
    df['adjusted_breakout'] = df['breakout_momentum'] / df['cumulative_volume']
    
    # Aggregate Combined Momentum Over M Days
    df['combined_momentum'] = df['adjusted_trend'] + df['adjusted_breakout']
    df['alpha_factor'] = df['combined_momentum'].rolling(window=m_days).sum()
    
    return df['alpha_factor'].dropna()
```
