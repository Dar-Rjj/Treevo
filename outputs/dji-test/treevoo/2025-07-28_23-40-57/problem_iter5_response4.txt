```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Momentum using previous close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    recent_close = df['close']
    close_10_days_ago = df['close'].shift(10)
    price_momentum = recent_close - close_10_days_ago
    
    # Adjust by Volume
    N = 5  # Number of days to sum the volume
    cumulative_volume = df['volume'].rolling(window=N).sum()
    adjusted_momentum = price_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    past_5_days_volume_avg = df['volume'].rolling(window=5).mean()
    current_day_volume = df['volume']
    volume_ratio = current_day_volume / past_5_days_volume_avg
    
    # Combine Price and Intraday Momentum if Volume Ratio > 1
    combined_momentum = (price_momentum * momentum).where(volume_ratio > 1, 0)
    
    # Aggregate Momentum Over M Days
    M = 10  # Number of days to sum the combined momentum
    aggregated_momentum = combined_momentum.rolling(window=M).sum()
    
    return aggregated_momentum
```
