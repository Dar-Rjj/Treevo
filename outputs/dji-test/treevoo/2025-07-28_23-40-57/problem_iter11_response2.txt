```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between close price and the average of high and low
    price_diff = df['close'] - avg_price
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Heuristic alpha factor: product of price difference and volume change
    alpha_factor = price_diff * volume_change
    
    return alpha_factor
```
