```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Opening Price Trend
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement (VWIM)
    vwim = intraday_spread * df['volume']
    vwim_5d = vwim.rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend (AWOT)
    awot = opening_trend * df['amount']
    awot_5d = awot.rolling(window=5).mean()
    
    # Combine Adjusted Movements and Trends (AMT)
    amt = (vwim + awot) - vwim_5d - awot_5d
    
    # Adjust by Cumulative Volume (CV)
    cv = df['volume'].rolling(window=15).sum()
    amt_adj = amt / cv
    
    # Calculate Cumulative Volume-Weighted Momentum (CVWM)
    cvwm = (amt_adj * df['volume']).rolling(window=15).sum()
    
    # Confirm with Volume Trend
    av = df['volume'].rolling(window=15).mean()
    vr = df['volume'] / av
    
    # Compute Final Factor based on VR
    final_factor = pd.Series(index=df.index, dtype=float)
    for i in range(len(df)):
        if vr.iloc[i] > 1.3:
            final_factor.iloc[i] = amt_adj.iloc[i] * cvwm.iloc[i]
        else:
            final_factor.iloc[i] = 0.6 * (amt_adj.iloc[i] + cvwm.iloc[i])
    
    return final_factor
```
