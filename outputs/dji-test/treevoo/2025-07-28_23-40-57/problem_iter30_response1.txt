```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    volume_proportion = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Intraday Volume Intensity
    price_range = df['high'] - df['low']
    intraday_volume_intensity = volume_proportion / price_range
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=n).sum()
    avg_high = df['high'].rolling(window=n).mean()
    avg_low = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Multiply Intraday Return by Volume Impact Score
    weighted_intraday_return = intraday_return * volume_impact_score
    
    # Calculate Weighted Intraday Volume Intensity
    weighted_intraday_volume_intensity = intraday_volume_intensity * intraday_return
    
    # Calculate Intraday and Breakout Momentum
    intraday_momentum = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    adjusted_momentum = intraday_momentum / cumulative_volume
    
    # Incorporate Volume Adjusted Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    adjusted_breakout_intensity = breakout_intensity / cumulative_volume
    combined_intensity = (weighted_intraday_volume_intensity + adjusted_breakout_intensity) * intraday_return
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_return.rolling(window=m).std()
    
    # Adjust Combined Intensity by Intraday Volatility
    adjusted_combined_intensity = combined_intensity / intraday_volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_combined_intensity * intraday_return
    
    return alpha_factor
```
