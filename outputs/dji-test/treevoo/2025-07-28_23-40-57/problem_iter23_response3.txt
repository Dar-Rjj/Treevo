```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the closing location relative to the daily range
    close_location = (df['close'] - df['low']) / daily_range
    
    # Calculate the difference between the closing price and the opening price
    open_close_diff = df['close'] - df['open']
    
    # Calculate the factor value
    factor = close_location * open_close_diff
    
    # Return the factor as a pandas Series
    return factor
```
