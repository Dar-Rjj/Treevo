```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Trend
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Adjust for Combined Movements
    combined_movements = (intraday_movement + opening_trend) * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    n_days = 10
    volume_sum = df['volume'].rolling(window=n_days).sum()
    avg_high = df['high'].rolling(window=n_days).mean()
    avg_low = df['low'].rolling(window=n_days).mean()
    volume_impact_score = (avg_high - avg_low) / volume_sum
    
    # Combine Adjusted Movements and Intraday Return
    adjusted_factor = combined_movements * intraday_return * volume_impact_score
    
    # Calculate Cumulative Volume Weighted Intraday Return
    cumulative_weighted_return = (intraday_return * df['volume']).rolling(window=n_days).sum()
    
    # Calculate Short-Term Average Volume
    short_term_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average
    long_term_ma = df['close'].rolling(window=50).mean()
    
    # Calculate Short-Term Moving Average
    short_term_ma = df['close'].rolling(window=20).mean()
    
    # Calculate Moving Average Crossover
    ma_crossover = long_term_ma - short_term_ma
    
    # Synthesize Factors
    factor_1 = adjusted_factor
    factor_2 = cumulative_weighted_return
    factor_3 = ma_crossover
    final_factor = factor_1 * factor_2 * factor_3
    
    return final_factor
```
