```python
import pandas as pd

def heuristics_v2(df, n_days=10, short_term_days=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Volume Impact Score
    volume_sum = df['volume'].rolling(window=n_days).sum()
    avg_high = df['high'].rolling(window=n_days).mean()
    avg_low = df['low'].rolling(window=n_days).mean()
    volume_impact_score = (avg_high - avg_low) / volume_sum

    # Multiply Intraday Return by Volume Impact Score
    factor_1a = intraday_return * volume_impact_score

    # Calculate Volume-Weighted Price Impact
    volume_weighted_price_impact = (avg_high - avg_low) * volume_sum

    # Adjust Intraday Return by Volume-Weighted Price Impact
    factor_1b = intraday_return / volume_weighted_price_impact

    # Final Factor 1
    final_factor_1 = factor_1a * factor_1b

    # Calculate Cumulative Volume-Weighted Intraday Return
    cumulative_volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=n_days).sum()
    final_factor_2 = cumulative_volume_weighted_intraday_return

    # Calculate Short-Term Volume Variability
    short_term_volume_variability = df['volume'].rolling(window=short_term_days).std()

    # Combine Factors
    combined_factor = final_factor_1 + final_factor_2 - short_term_volume_variability

    return combined_factor
```
