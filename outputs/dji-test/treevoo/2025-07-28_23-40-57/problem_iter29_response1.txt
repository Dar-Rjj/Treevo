```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price trend using the close prices
    df['price_trend'] = df['close'].diff()
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].diff()
    
    # Compute the price range
    df['price_range'] = df['high'] - df['low']
    
    # Heuristic: High volume changes often indicate significant moves
    # We will multiply the price trend by the absolute value of the volume change
    df['factor'] = df['price_trend'] * df['volume_change'].abs()
    
    # Return the alpha factor as a pandas Series
    return df['factor']
```
