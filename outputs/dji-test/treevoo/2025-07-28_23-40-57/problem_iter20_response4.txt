```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    df['intraday_volume_proportion'] = df['volume'] - (df['volume'].shift(1) + df['volume'].shift(-1))
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    n_days = 10
    df['volume_sum_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high_n_days'] = df['high'].rolling(window=n_days).mean()
    df['avg_low_n_days'] = df['low'].rolling(window=n_days).mean()
    df['volume_impact_score'] = (df['avg_high_n_days'] - df['avg_low_n_days']) / df['volume_sum_n_days']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['intraday_breakout_momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Cumulative Volume
    N_days = 20
    df['cumulative_volume'] = df['volume'].rolling(window=N_days).sum()
    df['adjusted_momentum'] = df['intraday_breakout_momentum'] / df['cumulative_volume']
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['intraday_volume_proportion']
    
    # Generate Factor 1
    df['factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Confirm with Volume Trend
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['current_day_volume'] = df['volume']
    df['volume_ratio'] = df['current_day_volume'] / df['avg_volume_5_days']
    df['factor_2'] = df['intraday_breakout_momentum'] * df['intraday_return'] * (df['volume_ratio'] > 1)
    
    # Adjust by Enhanced Volume Trend
    df['avg_volume_15_days'] = df['volume'].rolling(window=15).mean()
    df['enhanced_volume_ratio'] = df['current_day_volume'] / df['avg_volume_15_days']
    df['enhanced_factor_2'] = df.apply(lambda row: row['intraday_breakout_momentum'] * row['intraday_return'] if row['enhanced_volume_ratio'] > 1.5 else 0.3 * (row['intraday_breakout_momentum'] + row['intraday_return']), axis=1)
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['cumulative_volume_weighted_momentum'] = (df['adjusted_momentum'] * df['volume']).rolling(window=20).sum()
    
    # Final Factor
    df['final_factor'] = df['factor_1'] * df['cumulative_volume_weighted_momentum']
    
    return df['final_factor']
```
