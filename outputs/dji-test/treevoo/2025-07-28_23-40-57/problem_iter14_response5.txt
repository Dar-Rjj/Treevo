```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the average of high and low
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the distance between close and the average of high and low
    df['close_to_avg_dist'] = (df['close'] - df['avg_high_low']).abs()
    
    # Calculate the percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Generate the alpha factor by combining the above elements
    df['alpha_factor'] = (df['close_change'] * df['close_to_avg_dist']) + (df['volume_change'] * df['avg_high_low'])
    
    return df['alpha_factor'].dropna()
```
