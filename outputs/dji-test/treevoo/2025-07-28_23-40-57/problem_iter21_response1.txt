```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in close prices
    df['Close_Change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Evaluate the daily volatility
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Measure the gap opening effect
    df['Gap_Opening'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Identify volume trend changes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Correlate volume with price movement
    df['Price_Volume_Correlation'] = (df['close'] - df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Investigate the impact of large transactions
    df['Amount_Change'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Link transaction size to price efficiency
    df['Transaction_Price_Efficiency'] = (df['close'] - (df['amount'] / df['volume'])) / df['close'].shift(1)
    
    # Examine the ratio of amount to close price as a liquidity indicator
    df['Liquidity_Ratio'] = df['amount'] / df['close']
    
    # Combine all sub-factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Close_Change'] * 0.2 +
        df['Daily_Volatility'] * 0.2 +
        df['Gap_Opening'] * 0.1 +
        df['Volume_Change'] * 0.1 +
        df['Price_Volume_Correlation'] * 0.1 +
        df['Amount_Change'] * 0.1 +
        df['Transaction_Price_Efficiency'] * 0.1 +
        df['Liquidity_Ratio'] * 0.1
    )
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
```
