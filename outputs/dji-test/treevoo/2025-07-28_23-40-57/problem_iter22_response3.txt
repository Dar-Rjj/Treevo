```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Volume Proportion
    df['overnight_volume'] = df['volume'].shift(1) - df['volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['intraday_vol_intensity'] = df['overnight_volume'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    sum_volume_n_days = df['volume'].rolling(window=n).sum()
    avg_high_n_days = df['high'].rolling(window=n).mean()
    avg_low_n_days = df['low'].rolling(window=n).mean()
    df['volume_impact_score'] = (avg_high_n_days - avg_low_n_days) / sum_volume_n_days
    
    # Multiply Intraday Return by Volume Impact Score
    df['intermediate_factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(n)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    df['adjusted_momentum'] = df['price_momentum'] / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['cum_vol_weighted_momentum'] = (df['adjusted_momentum'] * df['volume']).rolling(window=n).sum()
    
    # Confirm with Volume Trend
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5_days']
    
    def combine_price_intraday_momentum(row):
        if row['volume_ratio'] > 1:
            return row['price_momentum'] * row['intraday_return']
        else:
            return 0
    df['combined_momentum'] = df.apply(combine_price_intraday_momentum, axis=1)
    
    # Aggregate Momentum Over M Days
    df['aggregated_momentum'] = df['cum_vol_weighted_momentum'].rolling(window=m).sum()
    
    # Final Factor
    df['final_factor'] = df['intermediate_factor_1'] * df['aggregated_momentum']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_vol_intensity'] = df['intraday_vol_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['intraday_breakout_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Volume
    df['adjusted_intraday_breakout_momentum'] = df['intraday_breakout_momentum'] / cumulative_volume
    
    # Aggregate Momentum Over M Days
    df['aggregated_intraday_breakout_momentum'] = df['adjusted_intraday_breakout_momentum'].rolling(window=m).sum()
    
    return df['final_factor']
```
