import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    intraday_volume_proportion = (df['volume'] - df['volume'].shift(1)) / (df['open'].shift(-1) - df['close'])
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = intraday_volume_proportion / (df['high'] - df['low'])

    # Calculate Weighted Intraday Volume Intensity
    weighted_intraday_volume_intensity = intraday_volume_intensity * ((df['close'] - df['open']) / df['open'])

    # Incorporate Momentum with Intraday Volume Intensity
    five_day_avg_return = (df['close'].rolling(window=5).mean() - df['open'].rolling(window=5).mean()) / 5
    momentum_adjusted_intensity = weighted_intraday_volume_intensity * five_day_avg_return

    return momentum_adjusted_intensity
