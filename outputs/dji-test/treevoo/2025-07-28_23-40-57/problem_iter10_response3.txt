```python
def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    open_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday = intraday_movement * df['volume']
    volume_weighted_intraday_5d = volume_weighted_intraday.rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_open_trend = open_trend * df['amount']
    amount_weighted_open_trend_5d = amount_weighted_open_trend.rolling(window=5).mean()
    
    # Combine Weighted Movements and Trends
    combined_factor = (volume_weighted_intraday + amount_weighted_open_trend 
                       - volume_weighted_intraday_5d 
                       - amount_weighted_open_trend_5d)
    
    return combined_factor
```
