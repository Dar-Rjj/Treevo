import pandas as pd

def heuristics_v2(df):
    # Calculate the intraday return
    df['intraday_return'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Multiply the intraday return by volume
    df['volume_confirmed_return'] = df['intraday_return'] * df['volume']
    
    # Sum over a fixed period (e.g., 10 days)
    df['factor_value'] = df['volume_confirmed_return'].rolling(window=10).sum()
    
    # Return the factor values as a pandas Series
    return df['factor_value']
