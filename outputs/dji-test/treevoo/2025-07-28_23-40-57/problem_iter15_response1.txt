```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    df['intraday_volume'] = df['volume'] - (df['volume'].shift(1) + df['volume'].shift(-1))
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['price_range'] = df['high'] - df['low']
    df['intraday_vol_intensity'] = df['intraday_volume'] / df['price_range']
    
    # Calculate Volume Impact Score
    n_days = 10
    df['volume_sum_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high_n_days'] = df['high'].rolling(window=n_days).mean()
    df['avg_low_n_days'] = df['low'].rolling(window=n_days).mean()
    df['vol_impact_score'] = (df['avg_high_n_days'] - df['avg_low_n_days']) / df['volume_sum_n_days']
    
    # Multiply Intraday Return by Volume Impact Score
    df['factor_1'] = df['intraday_return'] * df['vol_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_vol_intensity'] = df['intraday_vol_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Volume
    n_days_cumulative = 10
    df['cumulative_volume'] = df['volume'].rolling(window=n_days_cumulative).sum()
    df['adjusted_momentum'] = df['momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['daily_vol_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_vol_weighted_momentum'] = df['daily_vol_weighted_momentum'].rolling(window=n_days).sum()
    
    # Aggregate Momentum Over M Days
    m_days = 5
    df['aggregated_momentum'] = df['cumulative_vol_weighted_momentum'].rolling(window=m_days).sum()
    
    # Introduce Volume-Weighted Momentum Trend
    k_days = 5
    df['moving_avg_daily_vol_weighted_momentum'] = df['daily_vol_weighted_momentum'].rolling(window=k_days).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['factor_1'] + df['weighted_intraday_vol_intensity'] + 
                          df['aggregated_momentum'] + df['moving_avg_daily_vol_weighted_momentum'])
    
    return df['alpha_factor'].dropna()
```
