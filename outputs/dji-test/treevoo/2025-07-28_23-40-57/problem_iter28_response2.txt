```python
import pandas as pd

def heuristics_v2(df):
    # Sub-Sub-Thought: Calculating the Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Sub-Sub-Thought: Evaluating the Daily Volatility
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Sub-Sub-Thought: Identifying Volume Trend Changes
    df['Volume_Trend_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Sub-Sub-Thought: Correlating Volume with Price Movement
    df['Volume_Price_Correlation'] = (df['close'] - df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Generate Factor 1
    df['Volume_Impact_Score'] = df['Volume_Price_Correlation'] * df['Volume_Trend_Change']
    df['Factor_1'] = df['Intraday_Return'] * df['Volume_Impact_Score']
    
    # Calculate Price and Intraday Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Confirm with Volume Trend
    df['Average_Volume'] = df['volume'].rolling(window=5).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    df['Factor_2'] = df.apply(lambda row: row['Price_Momentum'] * row['Intraday_Momentum'] if row['Volume_Ratio'] > 1.2 else 0, axis=1)
    
    # Sub-Sub-Thought: Linking Transaction Size to Price Efficiency
    df['Transaction_Efficiency'] = (df['close'] - (df['amount'] / df['volume'])) / df['close'].shift(1)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Factor_1'] * df['Factor_2']
    
    return df['Alpha_Factor'].dropna()
```
