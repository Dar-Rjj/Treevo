```python
import pandas as pd

def heuristics_v2(df, M=5):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Opening Price Trend
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Adjust for Intraday Breakout
    adjusted_breakout = (intraday_movement) / df['close'].shift(1)
    
    # Combine Intraday, Opening, and Breakout Movements
    combined_movements = (intraday_movement + opening_trend + adjusted_breakout) * df['volume']
    
    # Aggregate Adjusted Momentum Over M Days
    aggregated_momentum = combined_movements.rolling(window=M).sum()
    
    # Calculate Volume Trend
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Final Factor
    final_factor = aggregated_momentum * volume_ratio
    
    return final_factor
```
