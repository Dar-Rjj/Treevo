```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    df['volume_proportion'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Volume Intensity
    df['volume_intensity'] = df['volume_proportion'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    high_avg = df['high'].rolling(window=n).mean()
    low_avg = df['low'].rolling(window=n).mean()
    volume_sum = df['volume'].rolling(window=n).sum()
    df['volume_impact_score'] = (high_avg - low_avg) / volume_sum
    
    # Multiply Intraday Return by Volume Impact Score
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_volume_intensity'] = df['volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    prev_close = df['close'].shift(1)
    df['momentum'] = (df['high'] - df['low']) / prev_close
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    df['adjusted_momentum'] = df['momentum'] / cumulative_volume
    
    # Final Alpha Factor
    alpha_factor = df['weighted_intraday_return'] + df['weighted_volume_intensity'] + df['adjusted_momentum']
    
    return alpha_factor.dropna()
```
