```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Price Impact
    n_days = 10
    volume_sum = df['volume'].rolling(window=n_days).sum()
    high_avg = df['high'].rolling(window=n_days).mean()
    low_avg = df['low'].rolling(window=n_days).mean()
    price_impact = (high_avg - low_avg) * volume_sum
    
    # Adjust Intraday Return by Volume-Weighted Price Impact
    adjusted_intraday_return = intraday_return / (price_impact + 1e-6)  # Add small constant to avoid division by zero
    factor_1 = adjusted_intraday_return
    
    # Calculate Cumulative Volume-Weighted Intraday Return
    volume_weighted_return = intraday_return * df['volume']
    cumulative_volume_weighted_return = volume_weighted_return.rolling(window=n_days).sum()
    factor_2 = cumulative_volume_weighted_return
    
    # Calculate Short-Term Price Volatility
    short_term_volatility = df['close'].rolling(window=5).std()
    factor_3 = short_term_volatility
    
    # Calculate Relative Volume Strength
    avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume_strength = df['volume'] / avg_volume
    factor_4 = relative_volume_strength
    
    # Synthesize Factors
    final_factor = (factor_1 + factor_2 + factor_4) / (1 + factor_3)
    
    return final_factor
```
