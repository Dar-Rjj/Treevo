import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    intraday_volume_proportion = df['volume'].shift(1) - df['volume']
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = intraday_volume_proportion / (df['high'] - df['low'])
    
    # Multiply Intraday Return by Intraday Volume Intensity
    intermediate_factor_1 = intraday_return * intraday_volume_intensity
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust Price Momentum by Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    adjusted_momentum = price_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = adjusted_momentum * df['volume']
    sum_volume_weighted_momentum = volume_weighted_momentum.rolling(window=10).sum()
    
    # Confirm with Volume Trend
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume
    combined_momentum = (price_momentum * intraday_return).where(volume_ratio > 1, 0)
    aggregated_momentum = combined_momentum.rolling(window=10).sum()
    
    final_factor = intermediate_factor_1 * aggregated_momentum
    
    # Calculate Intraday and Breakout Momentum
    intraday_breakout_momentum = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Cumulative Volume
    adjusted_intraday_breakout_momentum = intraday_breakout_momentum / cumulative_volume
    
    # Aggregate Momentum Over M Days
    aggregated_breakout_momentum = adjusted_intraday_breakout_momentum.rolling(window=10).sum()
    
    # Final Factor
    final_factor = final_factor + aggregated_breakout_momentum
    
    return final_factor
