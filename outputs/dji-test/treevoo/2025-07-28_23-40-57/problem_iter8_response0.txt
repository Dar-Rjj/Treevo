def heuristics_v2(df):
    import pandas as pd
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / average_volume_5_days
    
    combined_momentum = pd.Series(0, index=df.index)
    m = 10  # Number of days to aggregate momentum
    for i in range(len(df)):
        if volume_ratio.iloc[i] > 1:
            combined_momentum.iloc[i] = price_momentum.iloc[i] * momentum.iloc[i]
    
    aggregate_momentum = combined_momentum.rolling(window=m).sum()
    
    return aggregate_momentum
