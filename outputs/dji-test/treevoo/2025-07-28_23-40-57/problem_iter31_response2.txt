```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in close prices
    roc_close = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate daily volatility
    daily_volatility = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate gap opening effect
    gap_opening = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate volume trend changes
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Calculate correlation of volume with price movement
    volume_price_corr = (df['close'] - df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Calculate impact of large transactions
    transaction_impact = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Calculate linking transaction size to price efficiency
    price_efficiency = (df['close'] - (df['amount'] / df['volume'])) / df['close'].shift(1)
    
    # Calculate ratio of amount to close price as a liquidity indicator
    liquidity_indicator = df['amount'] / df['close']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        roc_close + 
        daily_volatility + 
        gap_opening + 
        volume_trend + 
        volume_price_corr + 
        transaction_impact + 
        price_efficiency + 
        liquidity_indicator
    )
    
    return alpha_factor
```
