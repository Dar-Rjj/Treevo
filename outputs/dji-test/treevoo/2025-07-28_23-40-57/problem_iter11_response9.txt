```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=n).sum()
    avg_high = df['high'].rolling(window=n).mean()
    avg_low = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Multiply Intraday Return by Volume Impact Score
    factor_1 = df['intraday_return'] * volume_impact_score
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide High-Low Spread by Previous Close
    df['momentum'] = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(n)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    adjusted_momentum = price_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    cumulative_vol_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=n).sum()
    
    # Confirm with Volume Trend
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume_5d
    
    # Initialize Final Factor
    final_factor = pd.Series(index=df.index, data=0.0)
    
    # Combine Factors
    for i in range(n, len(df)):
        if volume_ratio.iloc[i] > 1:
            aggregated_momentum = cumulative_vol_weighted_momentum.iloc[i-n:i].sum()
            final_factor.iloc[i] = factor_1.iloc[i] * aggregated_momentum
        else:
            final_factor.iloc[i] = 0.0
    
    return final_factor
```
