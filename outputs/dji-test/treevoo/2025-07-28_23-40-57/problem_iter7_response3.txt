```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    recent_close = df['close'].iloc[-1]
    close_10_days_ago = df['close'].iloc[-11]
    price_momentum = recent_close - close_10_days_ago

    # Confirm with Volume Increase
    recent_volume = df['volume'].iloc[-1]
    volume_10_days_ago = df['volume'].iloc[-11]
    volume_ratio = recent_volume / volume_10_days_ago
    if volume_ratio <= 1:
        return pd.Series(0, index=[df.index[-1]])

    # Calculate EMAs
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate EMA Difference
    ema_difference = short_ema.iloc[-1] - long_ema.iloc[-1]

    # Calculate Volume Rate of Change (VROC)
    vroc = (recent_volume / df['volume'].shift(14).iloc[-1]) - 1

    # Combine Indicators
    if price_momentum > 0 and ema_difference > 0:
        factor_value = ema_difference * vroc
    else:
        factor_value = 0

    return pd.Series(factor_value, index=[df.index[-1]])
```
