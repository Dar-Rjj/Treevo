```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Evaluate the Daily Volatility
    df['DailyVolatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Measure the Gap Opening Effect
    df['GapOpeningEffect'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Identify Volume Trend Changes
    df['VolumeTrendChange'] = df['volume'] - df['volume'].shift(1)
    
    # Correlate Volume with Price Movement
    df['VolumePriceCorrelation'] = (df['close'] - df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume Impact Score
    df['VolumeImpactScore'] = (df['high'].rolling(window=10).mean() - df['low'].rolling(window=10).mean()) / df['volume'].rolling(window=10).sum()
    
    # Generate Factor 1
    df['Factor1'] = df['IntradayReturn'] * df['VolumeImpactScore']
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Intraday Return
    df['IntradayReturn_HL'] = (df['high'] - df['low']) / df['low']
    
    # Compute Average Volume (Past 5 Days)
    df['AvgVolume5D'] = df['volume'].rolling(window=5).mean()
    
    # Get Current Day Volume
    df['CurrentDayVolume'] = df['volume']
    
    # Compute Volume Ratio
    df['VolumeRatio'] = df['CurrentDayVolume'] / df['AvgVolume5D']
    
    # Confirm with Volume Trend
    df['Factor2'] = df.apply(lambda x: x['PriceMomentum'] * x['IntradayReturn_HL'] if x['VolumeRatio'] > 1.2 else 0, axis=1)
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['Factor1'] * df['Factor2']
    
    # Investigate the Impact of Large Transactions
    df['LargeTransactionImpact'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Link Transaction Size to Price Efficiency
    df['TransactionSizeEfficiency'] = (df['close'] - (df['amount'] / df['volume'])) / df['close'].shift(1)
    
    # Examining the Ratio of Amount to Close Price as a Liquidity Indicator
    df['LiquidityIndicator'] = df['amount'] / df['close']
    
    # Analyzing the Impact of Large Orders on Intraday Returns
    df['LargeOrderIntradayImpact'] = df['IntradayReturn'] * (df['amount'] / df['volume'])
    
    return df['AlphaFactor']
```
