import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Change in Intraday Range over Time (Daily Delta)
    df['daily_delta_intraday_range'] = df['intraday_range'].diff()
    
    # Identify Gaps at Market Open
    df['opening_gap_up'] = df['open'] - df['close'].shift(1)
    df['opening_gap_down'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Analyze Volume Spikes
    df['volume_spike'] = df['volume'] > df['volume'].rolling(window=10).mean() * 1.5
    
    # Calculate Trading Amount Volatility
    df['trading_amount_volatility'] = df['amount'].pct_change().rolling(window=10).std()
    
    # Correlate Trading Amount with Price Changes
    df['trading_amount_to_price_change_ratio'] = (df['amount'] / df['amount'].shift(1)) / (df['close'] / df['close'].shift(1))
    
    # Integrate Multiple Indicators for Composite Alpha Factors
    df['composite_alpha_score'] = (df['daily_delta_intraday_range'] + df['opening_gap_up'] - df['opening_gap_down'] + df['vwap'].diff()) / 4
    
    # Create a Combined Factor Using Volume Spikes and Trading Amount
    df['enhanced_alpha_factor'] = (df['volume_spike'] * df['trading_amount_to_price_change_ratio']).replace({True: 1, False: 0})
    
    # Return the final alpha factor
    return df['enhanced_alpha_factor']
