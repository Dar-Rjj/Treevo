```python
def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = df['intraday_range'] / df['close'].shift(1)
    
    # Adjusted Intraday Return with Volume
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume']
    
    return df['adjusted_intraday_return']
```
