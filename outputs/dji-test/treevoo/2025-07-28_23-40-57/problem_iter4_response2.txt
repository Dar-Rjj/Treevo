import pandas as pd

def heuristics_v2(df, period=10):
    # Calculate High-Low Range Return
    df['high_low_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the sum of traded amount over the period
    df['amount_sum'] = df['amount'].rolling(window=period, min_periods=1).sum()
    
    # Assign weights to each period
    weights = [i / period for i in range(1, period + 1)]
    
    # Calculate weighted high-low returns
    for i in range(period):
        if i == 0:
            df[f'weighted_return_{i}'] = df['high_low_return'].shift(i) * weights[i]
        else:
            df[f'weighted_return_{i}'] = df['high_low_return'].shift(i) * weights[i] / df['amount_sum'].shift(i)
    
    # Sum the weighted returns to get the final alpha factor
    df['alpha_factor'] = df[[f'weighted_return_{i}' for i in range(period)]].sum(axis=1)
    
    return df['alpha_factor']
