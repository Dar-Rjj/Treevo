import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday = intraday_movement * df['volume']
    volume_weighted_intraday_mavg5 = volume_weighted_intraday.rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_opening = opening_trend * df['amount']
    amount_weighted_opening_mavg5 = amount_weighted_opening.rolling(window=5).mean()
    
    # Combine Adjusted Movements and Trends
    combined_adjusted = (volume_weighted_intraday + amount_weighted_opening) - volume_weighted_intraday_mavg5 - amount_weighted_opening_mavg5
    
    # Calculate Volume Impact Score
    sum_volume_10 = df['volume'].rolling(window=10).sum()
    avg_high_10 = df['high'].rolling(window=10).mean()
    avg_low_10 = df['low'].rolling(window=10).mean()
    volume_impact_score = (avg_high_10 - avg_low_10) / sum_volume_10
    
    # Multiply Intraday Return by Volume Impact Score for intermediate factor (Factor 1a)
    factor_1a = intraday_return * volume_impact_score
    
    # Calculate Volume-Weighted Price Impact
    price_impact = (avg_high_10 - avg_low_10) * sum_volume_10
    
    # Adjust Intraday Return by Volume-Weighted Price Impact for intermediate factor (Factor 1b)
    factor_1b = intraday_return / price_impact
    
    # Final Factor 1 = Factor 1a * Factor 1b
    final_factor_1 = factor_1a * factor_1b
    
    # Calculate Cumulative Volume-Weighted Intraday Return for Final Factor 2
    volume_weighted_intraday_return = intraday_return * df['volume']
    final_factor_2 = volume_weighted_intraday_return.rolling(window=10).sum()
    
    # Calculate Short-Term Volume Variability
    short_term_vol_variability = df['volume'].rolling(window=5).std()
    
    # Combine Factors
    combined_factor = final_factor_1 + final_factor_2 - short_term_vol_variability
    
    return combined_factor
