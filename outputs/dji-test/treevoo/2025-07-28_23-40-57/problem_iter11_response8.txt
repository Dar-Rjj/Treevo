```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Calculate Intraday Returns
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Adjust by Volume
    n_days = 10
    sum_volume = df['volume'].rolling(window=n_days).sum()
    avg_volume = sum_volume / n_days
    volume_adjusted_momentum = momentum * avg_volume
    cumulative_volume = df['volume'].rolling(window=n_days).sum()
    final_adjusted_momentum = volume_adjusted_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    cumulative_vol_weighted_momentum = (final_adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    past_10_days_volume = df['volume'].rolling(window=10).mean()
    current_day_volume = df['volume']
    volume_ratio = current_day_volume / past_10_days_volume
    
    # Combine Price and Intraday Momentum
    combined_momentum = price_momentum * intraday_return
    aggregated_momentum = cumulative_vol_weighted_momentum.rolling(window=10).sum()
    
    # Final factor
    alpha_factor = pd.Series(index=df.index, dtype='float64')
    alpha_factor[volume_ratio > 1.2] = aggregated_momentum
    alpha_factor[volume_ratio <= 1.2] = 0.5 * (price_momentum + intraday_return)
    
    return alpha_factor
```
