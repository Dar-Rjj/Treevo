import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    recent_close = df['close'].iloc[-1]
    close_10_days_ago = df['close'].iloc[-11]
    price_momentum = recent_close - close_10_days_ago

    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Confirm with Volume Trend
    avg_volume_5_days = df['volume'].rolling(window=5).mean().iloc[-1]
    current_day_volume = df['volume'].iloc[-1]
    volume_ratio = current_day_volume / avg_volume_5_days

    if volume_ratio > 1:
        factor_3 = price_momentum * intraday_return.iloc[-1]
    else:
        factor_3 = 0

    # Calculate Volume-Adjusted Intraday Momentum
    intraday_return_open_close = (df['close'] - df['open']) / df['open']
    sum_volume_10_days = df['volume'].rolling(window=10).sum().iloc[-1]
    avg_high_10_days = df['high'].rolling(window=10).mean().iloc[-1]
    avg_low_10_days = df['low'].rolling(window=10).mean().iloc[-1]
    volume_impact_score = (avg_high_10_days - avg_low_10_days) / sum_volume_10_days
    factor_4 = intraday_return_open_close.iloc[-1] * volume_impact_score

    # Calculate Cumulative Volume Weighted Intraday Return
    weighted_intraday_return = intraday_return_open_close * df['volume']
    factor_5 = weighted_intraday_return.rolling(window=10).sum().iloc[-1]

    # Synthesize Factors
    final_factor = factor_3 * (factor_4 + factor_5)

    return pd.Series(final_factor, index=[df.index[-1]])
