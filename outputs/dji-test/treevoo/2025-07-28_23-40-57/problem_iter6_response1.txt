import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Volume Impact Score
    df['Volume_Sum'] = df['volume'].rolling(window=n).sum()
    df['Avg_High'] = df['high'].rolling(window=n).mean()
    df['Avg_Low'] = df['low'].rolling(window=n).mean()
    df['Volume_Impact_Score'] = (df['Avg_High'] - df['Avg_Low']) / df['Volume_Sum']
    
    # Calculate Volatility Adjusted Intraday Return
    df['Close_Std'] = df['close'].rolling(window=n).std()
    df['Volatility_Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['Close_Std']
    
    # Multiply Volatility Adjusted Intraday Return by Volume Impact Score
    df['Factor_Value'] = df['Volatility_Adjusted_Intraday_Return'] * df['Volume_Impact_Score']
    
    return df['Factor_Value'].dropna()
