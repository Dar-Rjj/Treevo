import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Divide by Previous Close
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Calculate Price and Intraday Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(5)
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust by Volume
    N = 20
    df['cumulative_volume'] = df['volume'].rolling(window=N).sum()
    df['adjusted_momentum'] = df['momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_vol_weighted_mom'] = df['volume_weighted_momentum'].rolling(window=N).sum()
    
    # Confirm with Volume Trend
    df['avg_volume_15'] = df['volume'].rolling(window=15).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_15']
    
    def combine_momentums(row):
        if row['volume_ratio'] > 1.5:
            return row['cumulative_vol_weighted_mom']
        else:
            return 0.3 * (row['price_momentum'] + row['intraday_return'])
    
    df['alpha_factor'] = df.apply(combine_momentums, axis=1)
    
    return df['alpha_factor']
