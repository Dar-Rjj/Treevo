```python
import pandas as pd

def heuristics_v2(df, N=10, M=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Adjust by Volume
    volume_sum_N = df['volume'].rolling(window=N).sum()
    volume_adjusted_momentum = momentum * df['volume']
    
    # Calculate Cumulative Volume-Weighted Momentum over N days
    cumulative_vol_weighted_momentum = volume_adjusted_momentum.rolling(window=N).sum() / volume_sum_N
    
    # Aggregate Momentum Over M Days
    aggregated_momentum = cumulative_vol_weighted_momentum.rolling(window=M).sum()
    
    return aggregated_momentum
```
