```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    df['overnight_volume'] = df['volume'].shift(1) * df['close'].shift(1) / df['open']
    df['intraday_volume_proportion'] = df['volume'] - df['overnight_volume']
    
    # Calculate Intraday Volume Intensity
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / df['high_low_range']
    
    # Calculate Volume Impact Score
    n_days = 10
    df['volume_sum_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['high_avg_n_days'] = df['high'].rolling(window=n_days).mean()
    df['low_avg_n_days'] = df['low'].rolling(window=n_days).mean()
    df['price_range_avg_n_days'] = df['high_avg_n_days'] - df['low_avg_n_days']
    df['volume_impact_score'] = df['price_range_avg_n_days'] / df['volume_sum_n_days']
    
    # Generate Factor 1
    df['factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=n_days).sum()
    
    # Aggregate Momentum Over M Days
    m_days = 10
    df['aggregated_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=m_days).sum()
    
    # Confirm with Volume Trend
    df['average_volume_past_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_past_5_days']
    
    # Combine Intraday and Breakout Momentum
    df['factor_2'] = df['momentum'] * df['aggregated_momentum'] * (df['volume_ratio'] > 1)
    
    # Final Factor
    df['final_factor'] = df['factor_1'] + df['factor_2']
    
    return df['final_factor']
```
