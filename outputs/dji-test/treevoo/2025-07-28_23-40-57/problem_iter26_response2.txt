```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume_10_days = df['volume'].rolling(window=10).sum()
    avg_high_10_days = df['high'].rolling(window=10).mean()
    avg_low_10_days = df['low'].rolling(window=10).mean()
    volume_impact_score = (avg_high_10_days - avg_low_10_days) / sum_volume_10_days
    
    # Generate Intermediate Factor 1
    intermediate_factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price Momentum
    recent_close = df['close']
    close_10_days_ago = df['close'].shift(10)
    price_momentum = recent_close - close_10_days_ago
    
    # Adjust Price Momentum by Volume
    adjusted_momentum = price_momentum / sum_volume_10_days
    
    # Calculate Cumulative Volume-Weighted Momentum
    cumulative_volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    avg_volume_past_5_days = df['volume'].rolling(window=5).mean()
    current_day_volume = df['volume']
    volume_ratio = current_day_volume / avg_volume_past_5_days
    
    combined_momentum = price_momentum * intraday_return
    aggregated_momentum = cumulative_volume_weighted_momentum
    
    final_factor = pd.Series(index=df.index, dtype=float)
    final_factor[volume_ratio > 1.2] = intermediate_factor_1 * aggregated_momentum
    final_factor[volume_ratio <= 1.2] = 0
    
    # Explore Intraday Price Dynamics
    intraday_range = df['high'] - df['low']
    daily_delta_intraday_range = intraday_range.diff()
    
    opening_gap_up = df['open'] - df['close'].shift(1)
    opening_gap_down = df['close'].shift(1) - df['open']
    
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    significant_volume_increase = df['volume'].pct_change() > 1.5
    volume_spike_days = significant_volume_increase & (df['close'] - df['close'].shift(1)).abs() > 0.05
    
    trading_amount_volatility = df['amount'].pct_change().rolling(window=10).std()
    amount_to_price_change_ratio = df['amount'].pct_change() / df['close'].pct_change()
    
    composite_alpha_score = intraday_range + opening_gap_up - opening_gap_down + vwap
    enhanced_alpha_factor = composite_alpha_score * volume_spike_days + amount_to_price_change_ratio * trading_amount_volatility
    
    # Final Alpha Factor
    final_alpha_factor = final_factor * enhanced_alpha_factor
    
    return final_alpha_factor
```
