```python
def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=n).sum()
    avg_high = df['high'].rolling(window=n).mean()
    avg_low = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Multiply Intraday Return by Volume Impact Score for Factor 1
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Cumulative Volume Weighted Intraday Return for Factor 2
    volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=n).sum()
    
    # Combine Factors for the final factor
    final_factor = factor_1 * volume_weighted_intraday_return
    
    return final_factor
```
