```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    close_to_open_change = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_volume_sum = (df['close'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['open'] * df['volume']) / 4
    volume_sum = df['volume']
    vwap = price_volume_sum.rolling(window=n).sum() / volume_sum.rolling(window=n).sum()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    volume_sum_n_days = df['volume'].rolling(window=n).sum()
    avg_high_n_days = df['high'].rolling(window=n).mean()
    avg_low_n_days = df['low'].rolling(window=n).mean()
    high_low_diff = avg_high_n_days - avg_low_n_days
    volume_impact_score = high_low_diff / volume_sum_n_days
    
    # Calculate High-Low Spread for Breakout Momentum
    high_low_spread = df['high'] - df['low']
    
    # Adjust by Previous Close for Breakout Momentum
    prev_close = df['close'].shift(1)
    breakout_momentum = high_low_spread / prev_close
    
    # Adjust by Volume for Breakout Momentum
    sum_volume_n_days = df['volume'].rolling(window=n).sum()
    average_volume = sum_volume_n_days / n
    volume_adjusted_breakout_momentum = breakout_momentum * average_volume
    
    # Introduce Volatility Measure
    std_close_prices = df['close'].rolling(window=n).std()
    normalized_high_low_spread = high_low_spread / std_close_prices
    
    # Combine Indicators
    adjusted_intraday_range = intraday_range / vwap
    combined_indicator = adjusted_intraday_range * close_to_open_change
    combined_indicator *= volume_change
    combined_indicator *= intraday_return * volume_impact_score
    combined_indicator *= volume_adjusted_breakout_momentum
    combined_indicator *= normalized_high_low_spread
    
    return combined_indicator
```
