```python
def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    avg_high_n_days = df['high'].rolling(window=n).mean()
    avg_low_n_days = df['low'].rolling(window=n).mean()
    sum_volume_n_days = df['volume'].rolling(window=n).sum()
    volume_impact_score = (avg_high_n_days - avg_low_n_days) / sum_volume_n_days
    
    # Multiply Intraday Return by Volume Impact Score to get Factor 1
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Cumulative Volume Weighted Intraday Return (Factor 2)
    weighted_intraday_return = intraday_return * df['volume']
    factor_2 = weighted_intraday_return.rolling(window=n).sum()
    
    # Combine Factors
    final_factor = factor_1 * factor_2
    
    return final_factor
```
