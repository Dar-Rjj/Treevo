```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    volume_sum_10d = df['volume'].rolling(window=10).sum()
    avg_high_10d = df['high'].rolling(window=10).mean()
    avg_low_10d = df['low'].rolling(window=10).mean()
    price_range_10d = avg_high_10d - avg_low_10d
    volume_impact_score = price_range_10d / volume_sum_10d
    
    # Generate Factor 1
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Confirm with Volume Trend
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume_5d
    
    factor_2 = price_momentum * intraday_momentum
    factor_2.loc[volume_ratio <= 1.2] = 0  # Set to 0 if volume ratio is not significant
    
    # Final Factor
    final_factor = factor_1 * factor_2
    
    return final_factor
```
