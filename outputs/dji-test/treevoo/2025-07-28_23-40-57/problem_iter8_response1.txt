```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    df['close_to_open_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices_sum = (df['close'] + df['high'] + df['low'] + df['open']) * df['volume']
    vwap = prices_sum.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['vwap'] = vwap
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=10).sum()
    avg_high = df['high'].rolling(window=10).mean()
    avg_low = df['low'].rolling(window=10).mean()
    df['volume_impact_score'] = (avg_high - avg_low) / sum_volume
    
    # Adjust Intraday Price Range by VWAP
    df['adjusted_intraday_range'] = df['intraday_range'] / df['vwap']
    
    # Multiply by Close-to-Open Price Change
    df['combined_indicator_1'] = df['adjusted_intraday_range'] * df['close_to_open_change']
    
    # Incorporate Volume Change
    df['combined_indicator_2'] = df['combined_indicator_1'] * df['volume_change']
    
    # Multiply Intraday Return by Volume Impact Score
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Final Alpha Factor
    df['final_alpha_factor'] = df['combined_indicator_2'] + df['adjusted_intraday_return']
    
    return df['final_alpha_factor']
```
