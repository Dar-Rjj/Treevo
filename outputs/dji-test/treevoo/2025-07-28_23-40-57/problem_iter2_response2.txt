```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    df['intraday_volume_proportion'] = df['volume'] - (df['volume'].shift(1) + df['open'])
    
    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / (df['high'] - df['low'])
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * (df['close'] - df['open'])
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    df['close_to_open_price_change'] = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['close', 'high', 'low', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Normalize Intraday Price Range by VWAP
    df['normalized_intraday_price_range'] = df['intraday_price_range'] / df['vwap']
    
    # Multiply by Close-to-Open Price Change
    df['combined_indicator'] = df['normalized_intraday_price_range'] * df['close_to_open_price_change']
    
    # Weight by Intraday Volume Intensity
    df['alpha_factor'] = df['combined_indicator'] * df['weighted_intraday_volume_intensity']
    
    return df['alpha_factor']
```
