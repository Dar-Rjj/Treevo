```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Closing Price Stability
    closing_stability = df['close'] - df['low']
    
    # Combine Intraday, Opening, and Closing Movements
    combined_movement = intraday_movement + opening_trend + closing_stability
    
    # Weight by Volume and Amount
    weighted_factor = combined_movement * (df['volume'] + df['amount'])
    
    return weighted_factor
```
