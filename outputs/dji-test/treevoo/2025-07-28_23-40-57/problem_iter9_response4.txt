```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Price Impact
    n_days = 10
    df['volume_sum'] = df['volume'].rolling(window=n_days).sum()
    df['high_avg'] = df['high'].rolling(window=n_days).mean()
    df['low_avg'] = df['low'].rolling(window=n_days).mean()
    df['price_impact'] = (df['high_avg'] - df['low_avg']) * df['volume_sum']
    
    # Adjust Intraday Return by Volume-Weighted Price Impact
    df['factor_1'] = df['intraday_return'] / df['price_impact']

    # Calculate Cumulative Volume-Weighted Intraday Return
    df['vol_weighted_return'] = df['intraday_return'] * df['volume']
    df['factor_2'] = df['vol_weighted_return'].rolling(window=n_days).sum()

    # Calculate Short-Term Volume Variability
    short_term_days = 5
    df['volume_std'] = df['volume'].rolling(window=short_term_days).std()

    # Calculate Short-Term Price Volatility
    df['price_volatility'] = df['close'].rolling(window=short_term_days).std()

    # Synthesize Factors
    df['final_factor'] = (df['factor_1'] + df['factor_2']) / (1 + df['price_volatility'])

    return df['final_factor'].dropna()
```
