```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday = intraday_movement * df['volume']
    volume_weighted_intraday_ma = volume_weighted_intraday.rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_opening = opening_trend * df['amount']
    amount_weighted_opening_ma = amount_weighted_opening.rolling(window=5).mean()
    
    # Combine Weighted Movements and Trends
    combined_movement_trend = (volume_weighted_intraday + amount_weighted_opening) - (volume_weighted_intraday_ma + amount_weighted_opening_ma)
    
    # Confirm with Volume and Price Trend
    price_momentum = df['close'] - df['close'].shift(5)
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10d
    
    # Final Alpha Factor
    alpha_factor = pd.Series(index=df.index)
    for i in range(len(df)):
        if volume_ratio.iloc[i] > 1.2:
            alpha_factor.iloc[i] = price_momentum.iloc[i] * intraday_return.iloc[i]
        else:
            alpha_factor.iloc[i] = 0.5 * (price_momentum.iloc[i] + intraday_return.iloc[i])
    
    return alpha_factor + combined_movement_trend
```
