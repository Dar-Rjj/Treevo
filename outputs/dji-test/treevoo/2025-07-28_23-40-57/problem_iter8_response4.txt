```python
import pandas as pd

def heuristics_v2(df, n=10, short_term_n=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    volume_sum_n_days = df['volume'].rolling(window=n).sum()
    avg_high_n_days = df['high'].rolling(window=n).mean()
    avg_low_n_days = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high_n_days - avg_low_n_days) / volume_sum_n_days
    
    # Intermediate Factor 1a: Multiply Intraday Return by Volume Impact Score
    factor_1a = intraday_return * volume_impact_score
    
    # Calculate Volume-Weighted Price Impact
    volume_weighted_price_impact = (avg_high_n_days - avg_low_n_days) * volume_sum_n_days
    
    # Adjust Intraday Return by Volume-Weighted Price Impact
    adjusted_intraday_return = intraday_return / volume_weighted_price_impact
    
    # Intermediate Factor 1b: Adjusted Intraday Return
    factor_1b = adjusted_intraday_return
    
    # Final Factor 1: Combine Factor 1a and Factor 1b
    final_factor_1 = factor_1a * factor_1b
    
    # Calculate Cumulative Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    final_factor_2 = volume_weighted_intraday_return.rolling(window=n).sum()
    
    # Calculate Short-Term Volume Variability
    short_term_vol_std = df['volume'].rolling(window=short_term_n).std()
    
    # Normalize Short-Term Volume Variability
    mean_volume = df['volume'].rolling(window=short_term_n).mean()
    normalized_vol_variability = short_term_vol_std / mean_volume
    
    # Final Factor 3: Normalized Short-Term Volume Variability
    final_factor_3 = normalized_vol_variability
    
    # Combine Factors
    combined_factor = final_factor_1 + final_factor_2 - final_factor_3
    
    return combined_factor
```
