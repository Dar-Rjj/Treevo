```python
def heuristics_v2(df):
    # Calculate Intraday Returns
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Trend
    df['volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    df['volume_ratio'] = df['volume'] / df['volume_sum_5d']
    
    # Calculate Amount Trend
    df['amount_sum_5d'] = df['amount'].rolling(window=5).sum()
    df['amount_ratio'] = df['amount'] / df['amount_sum_5d']
    
    # Combine Intraday Returns, Volume Trend, and Amount Trend
    df['alpha_factor'] = df['intraday_return'] * df['volume_ratio'] * df['amount_ratio']
    
    return df['alpha_factor']
```
