```python
import pandas as pd

def heuristics_v2(df, n_days=10, m_days=5):
    # Calculate Intraday Volume Proportion
    df['overnight_volume'] = df['volume'].shift(1) + (df['volume'] - df['volume'].shift(1))
    df['intraday_volume_proportion'] = df['volume'] - df['overnight_volume']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volume Intensity
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / df['high_low_range']

    # Calculate Volume Impact Score
    sum_volume_n_days = df['volume'].rolling(window=n_days).sum()
    avg_high_n_days = df['high'].rolling(window=n_days).mean()
    avg_low_n_days = df['low'].rolling(window=n_days).mean()
    df['volume_impact_score'] = (avg_high_n_days - avg_low_n_days) / sum_volume_n_days

    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']

    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['prev_close'] = df['close'].shift(1)
    df['intraday_breakout_momentum'] = df['high_low_spread'] / df['prev_close']

    # Adjust by Cumulative Volume
    cumulative_volume_n_days = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['intraday_breakout_momentum'] / cumulative_volume_n_days

    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['intraday_volume_proportion']

    # Calculate Cumulative Volume-Weighted Momentum
    df['cumulative_volume_weighted_momentum'] = df['volume_adjusted_momentum'] * df['volume']
    df['cumulative_volume_weighted_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=m_days).sum()

    # Aggregate Momentum Over M Days
    df['aggregated_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=m_days).sum()

    return df['aggregated_momentum']
```
