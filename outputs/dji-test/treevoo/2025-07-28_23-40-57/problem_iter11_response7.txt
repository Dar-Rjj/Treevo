```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Divide by Previous Close
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Impact Score
    n_days = 10
    df['volume_sum'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high'] = df['high'].rolling(window=n_days).mean()
    df['avg_low'] = df['low'].rolling(window=n_days).mean()
    df['volume_impact_score'] = (df['avg_high'] - df['avg_low']) / df['volume_sum']
    
    # Multiply Intraday Return by Volume Impact Score
    df['factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Adjust by Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['factor_2'] = df['volume_weighted_intraday_return'].rolling(window=n_days).sum()
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=n_days).sum()
    
    # Confirm with Volume Trend
    volume_trend_days = 5
    df['average_volume'] = df['volume'].rolling(window=volume_trend_days).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Combine Factors if Volume Ratio > 1
    m_days = 10
    df['final_factor'] = df.apply(lambda row: row['factor_1'] * row['factor_2'] if row['volume_ratio'] > 1 else 0, axis=1)
    df['aggregate_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=m_days).sum()
    
    return df['final_factor']
```
