```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    intraday_volume_proportion = df['volume'] - (df['volume'].shift(1) + df['open'] * 0)
    
    # Calculate Intraday Volume Intensity
    high_low_range = df['high'] - df['low']
    intraday_volume_intensity = intraday_volume_proportion / high_low_range
    
    # Calculate Weighted Intraday Volume Intensity
    close_open_return = (df['close'] - df['open']) / df['open']
    weighted_intraday_volume_intensity = intraday_volume_intensity * close_open_return
    
    return weighted_intraday_volume_intensity
```
