```python
def heuristics_v2(df):
    # Calculate the percentage change of close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the percentage of the range (high-low) that the close price is above the open price
    df['range_above_open'] = (df['close'] - df['open']) / df['high_low_diff']
    
    # Calculate the weighted average of the high and low prices using volume as the weight
    df['volume_weighted_avg'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Generate the alpha factor by combining the different components
    df['alpha_factor'] = (
        df['close_pct_change'] +
        df['high_low_diff'] * df['range_above_open'] +
        df['volume_weighted_avg']
    )
    
    return df['alpha_factor']
```
