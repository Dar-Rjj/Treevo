```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in average price
    pct_change_avg_price = avg_price.pct_change()
    
    # Calculate the percentage change in price range
    pct_change_price_range = price_range.pct_change()
    
    # Calculate the interaction between the percentage change in average price and the percentage change in price range
    factor = pct_change_avg_price * pct_change_price_range
    
    return factor
```
