```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5, short_window=12, long_window=26):
    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']

    # Divide by Previous Close
    df['Breakout_Momentum'] = df['High-Low_Spread'] / df['close'].shift(1)

    # Adjust by Volume
    df['Cumulative_Volume'] = df['volume'].rolling(window=N).sum()
    df['Volume_Adjusted_Momentum'] = df['Breakout_Momentum'] / df['Cumulative_Volume']

    # Incorporate Trend Strength
    df['Short_EMA'] = df['close'].ewm(span=short_window, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=long_window, adjust=False).mean()
    df['MACD'] = df['Short_EMA'] - df['Long_EMA']
    df['Trend_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] * df['MACD']

    # Aggregate Momentum Over M Days
    df['Aggregated_Momentum'] = df['Trend_Adjusted_Momentum'].rolling(window=M).sum()

    return df['Aggregated_Momentum']
```
