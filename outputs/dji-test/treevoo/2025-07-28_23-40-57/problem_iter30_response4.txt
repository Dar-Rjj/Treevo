```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volume Proportion
    df['intraday_volume_proportion'] = df['volume'] - (df['volume'].shift(1) + df['open'].diff().abs())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    n_days = 10
    df['sum_volume_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high_low_n_days'] = (df['high'] + df['low']).rolling(window=n_days).mean() / 2
    df['volume_impact_score'] = (df['avg_high_low_n_days'].diff(n_days-1)) / df['sum_volume_n_days']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['intraday_breakout_momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Cumulative Volume
    N_days = 20
    df['cumulative_volume'] = df['volume'].rolling(window=N_days).sum()
    df['adjusted_momentum'] = df['intraday_breakout_momentum'] / df['cumulative_volume']
    
    # Generate Factor 1
    df['factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Confirm with Volume Trend
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    current_vol = df['volume']
    vol_ratio = current_vol / avg_vol_5d
    
    # Compute Factor 2
    df['factor_2'] = 0
    df.loc[vol_ratio > 1, 'factor_2'] = df.loc[vol_ratio > 1, 'intraday_breakout_momentum'] * df.loc[vol_ratio > 1, 'intraday_return']
    
    # Adjust by Enhanced Volume Trend
    avg_vol_15d = df['volume'].rolling(window=15).mean()
    enhanced_vol_ratio = current_vol / avg_vol_15d
    
    # Aggregate Momentum Over M Days
    M_days = 10
    df['cumulative_vol_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['aggregated_momentum'] = df['cumulative_vol_weighted_momentum'].rolling(window=M_days).sum()
    
    # Combine Momentum
    df.loc[enhanced_vol_ratio > 1.5, 'combined_momentum'] = df.loc[enhanced_vol_ratio > 1.5, 'aggregated_momentum']
    df.loc[enhanced_vol_ratio <= 1.5, 'combined_momentum'] = 0.4 * (df.loc[enhanced_vol_ratio <= 1.5, 'intraday_breakout_momentum'] + df.loc[enhanced_vol_ratio <= 1.5, 'intraday_return'])
    
    # Final Factor
    df['final_factor'] = df['factor_1'] * df['combined_momentum'] + df['factor_2']
    
    return df['final_factor']
```
