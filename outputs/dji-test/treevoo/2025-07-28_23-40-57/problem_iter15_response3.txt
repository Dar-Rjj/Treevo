```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate Intraday Movement (High - Low)
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Change (Open - Previous Day's Close)
    opening_price_change = df['open'] - df['close'].shift(1)
    
    # Adjust for Intraday Breakout (High - Low) / Previous Day's Close
    adjusted_breakout = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Momentum
    combined_momentum = (intraday_movement + opening_price_change + adjusted_breakout) * df['volume']
    
    # High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Momentum divided by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Sum of Volumes Over N Days
    sum_of_volumes = df['volume'].rolling(window=N).sum()
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = momentum * (sum_of_volumes / N)
    
    # Cumulative Volume-Weighted Momentum over a fixed period (e.g., 10 days)
    cumulative_volume_weighted_momentum = volume_adjusted_momentum.rolling(window=10).sum()
    
    # Aggregated Momentum Over M Days
    aggregated_momentum = combined_momentum.rolling(window=M).mean()
    
    # Volume Ratio
    average_volume_past_5_days = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / average_volume_past_5_days
    
    # Weighted Average Volume
    weighted_average_volume = (df['volume'] + sum_of_volumes) / (N + 1)
    
    # Final Factor
    final_factor = (aggregated_momentum * volume_ratio) + cumulative_volume_weighted_momentum + (combined_momentum * weighted_average_volume)
    
    return final_factor
```
