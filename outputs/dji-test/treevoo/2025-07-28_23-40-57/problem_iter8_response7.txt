```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close to get Momentum
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5_days
    confirmed_momentum = volume_weighted_momentum * (volume_ratio > 1)
    
    # Calculate EMAs
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    ema_difference = ema_short - ema_long
    
    # Calculate Volume Rate of Change (VROC)
    vroc = df['volume'] / df['volume'].shift(14) - 1
    
    # Combine Indicators
    combined_indicator = ema_difference * vroc
    alpha_factor = (price_momentum > 0) & (ema_difference > 0)
    alpha_factor = combined_indicator.where(alpha_factor, 0)
    
    return alpha_factor
```
