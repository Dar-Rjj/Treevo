```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range Return
    df['high_low_return'] = (df['high'] - df['low']) / df['open']
    
    # Sum of Traded Amount over a period
    df['cumulative_amount'] = df['amount'].cumsum()
    
    # Normalize Weights
    df['weight'] = df['amount'] / df['cumulative_amount']
    
    # Multiply Each Periodâ€™s High-Low Return by its Weight
    df['weighted_return'] = df['high_low_return'] * df['weight']
    
    # Final Alpha Factor: Sum the Weighted Returns
    alpha_factor = df['weighted_return'].rolling(window=len(df), min_periods=1).sum()
    
    return alpha_factor
```
