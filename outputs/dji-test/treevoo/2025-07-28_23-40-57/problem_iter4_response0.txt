```python
def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    close_to_open_change = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume'].sum()
    price_volume_sum = (df['close'] + df['high'] + df['low'] + df['open']) * df['volume']
    vwap = price_volume_sum.sum() / total_volume
    
    # Adjust Intraday Price Range by VWAP
    adjusted_intraday_range = intraday_price_range / vwap
    
    # Multiply by Close-to-Open Price Change
    final_alpha_factor = adjusted_intraday_range * close_to_open_change
    
    return final_alpha_factor
```
