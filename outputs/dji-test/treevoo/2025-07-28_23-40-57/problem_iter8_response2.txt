```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    sum_volume = df['volume'].rolling(window=n).sum()
    avg_high = df['high'].rolling(window=n).mean()
    avg_low = df['low'].rolling(window=n).mean()
    volume_impact_score = (avg_high - avg_low) / sum_volume
    
    # Combine for intermediate factor (Factor 1)
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(n)
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Confirm with Volume Trend
    avg_volume = df['volume'].rolling(window=m).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume
    factor_2 = price_momentum * intraday_momentum
    factor_2[volume_ratio <= 1] = 0
    
    # Combine Factors
    final_factor = factor_1 * factor_2
    return final_factor
```
