```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    close_to_open_change = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap_numerator = (df['close'] + df['high'] + df['low'] + df['open']) * df['volume']
    vwap_denominator = df['volume']
    vwap = vwap_numerator.cumsum() / vwap_denominator.cumsum()
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change(fill_method=None)
    
    # Combine Indicators
    adjusted_intraday_range = intraday_range / vwap
    combined_indicator = adjusted_intraday_range * close_to_open_change
    integrated_indicator = combined_indicator * (1 + volume_change)
    
    return integrated_indicator
```
