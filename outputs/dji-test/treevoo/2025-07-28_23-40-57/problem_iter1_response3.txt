```python
def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Opening Movements
    combined_movement = intraday_movement + opening_trend
    
    # Weight by Volume
    volume_weighted_factor = combined_movement * df['volume']
    
    return volume_weighted_factor
```
