```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['low']
    
    # Confirm with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10d
    
    # Combine Price and Intraday Momentum
    combined_momentum = pd.Series(index=df.index, dtype=float)
    for i in range(len(df)):
        if volume_ratio[i] > 1.2:
            combined_momentum[i] = price_momentum[i] * intraday_returns[i]
        else:
            combined_momentum[i] = 0.5 * (price_momentum[i] + intraday_returns[i])
    
    return combined_momentum
```
