```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    df['close_to_open_change'] = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['close'] + df['high'] + df['low'] + df['open']) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    n_days = 10
    df['sum_volume_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high_n_days'] = df['high'].rolling(window=n_days).mean()
    df['avg_low_n_days'] = df['low'].rolling(window=n_days).mean()
    df['price_vol_impact_score'] = (df['avg_high_n_days'] - df['avg_low_n_days']) / df['sum_volume_n_days']
    
    # Combine Intraday Price Indicators
    df['adj_intraday_price_range'] = df['intraday_price_range'] / df['vwap']
    df['combined_indicator'] = df['adj_intraday_price_range'] * df['close_to_open_change'] * df['intraday_return']
    
    # Calculate Cumulative Volume Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['cum_vol_weighted_intraday_return'] = df['vol_weighted_intraday_return'].rolling(window=n_days).sum()
    
    # Integrate Volume Components
    df['integrated_indicator'] = df['combined_indicator'] * df['volume_change']
    df['combined_vol_component'] = df['integrated_indicator'] * df['price_vol_impact_score']
    
    # Final Alpha Factor Calculation
    df['final_alpha_factor'] = df['combined_indicator'] * df['cum_vol_weighted_intraday_return'] * df['combined_vol_component']
    
    return df['final_alpha_factor']
```
