```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    volume_sum_10 = df['volume'].rolling(window=10).sum()
    high_avg_10 = df['high'].rolling(window=10).mean()
    low_avg_10 = df['low'].rolling(window=10).mean()
    price_diff = high_avg_10 - low_avg_10
    volume_impact_score = price_diff / volume_sum_10
    
    # Generate Factor 1
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Price Momentum
    recent_close = df['close']
    close_10_days_ago = df['close'].shift(10)
    price_momentum = recent_close - close_10_days_ago
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    adjusted_momentum = price_momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    cumulative_vol_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume_5_days
    
    # If Volume Ratio > 1, combine Price and Intraday Momentum
    aggregated_momentum = cumulative_vol_weighted_momentum.rolling(window=10).sum()
    factor_2 = pd.Series(0, index=df.index)
    factor_2[volume_ratio > 1] = (price_momentum * intraday_return * aggregated_momentum)[volume_ratio > 1]
    
    # Final Factor
    final_factor = factor_1 * factor_2
    
    return final_factor
```
