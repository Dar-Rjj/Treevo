```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Opening Price Change
    df['opening_price_change'] = df['open'] - df['close'].shift(1)
    
    # Adjust for Intraday Breakout
    df['adjusted_breakout'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = (df['intraday_movement'] + df['opening_price_change'] + df['adjusted_breakout']) * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Divide by Previous Close
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Sum of Volumes Over N Days (e.g., 10 days)
    n_days = 10
    df['sum_of_volumes'] = df['volume'].rolling(window=n_days).sum()
    
    # Multiply Momentum by Average Volume
    df['volume_adjusted_momentum'] = df['momentum'] * (df['sum_of_volumes'] / n_days)
    
    # Cumulative Volume-Weighted Momentum over a fixed period (e.g., 10 days)
    df['cumulative_volume_weighted_momentum'] = df['volume_adjusted_momentum'].rolling(window=n_days).sum()
    
    # Aggregated Momentum Over M Days (e.g., 10 days)
    m_days = 10
    df['aggregated_momentum'] = df['combined_momentum'].rolling(window=m_days).sum() / m_days
    
    # Volume Ratio
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5_days']
    
    # Final Factor
    df['final_factor'] = (df['aggregated_momentum'] * df['volume_ratio']) + df['cumulative_volume_weighted_momentum']
    
    return df['final_factor'].dropna()
```
