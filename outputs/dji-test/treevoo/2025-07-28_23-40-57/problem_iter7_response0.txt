```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Average Volume (Past 5 Days)
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Get Current Day Volume
    df['current_volume'] = df['volume']
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['current_volume'] / df['avg_volume_5d']
    
    # Combine Price and Intraday Momentum
    def combine_momentum(row):
        if row['volume_ratio'] > 1:
            return row['price_momentum'] * row['intraday_return']
        else:
            return 0
    
    df['factor_value'] = df.apply(combine_momentum, axis=1)
    
    return df['factor_value']
```
