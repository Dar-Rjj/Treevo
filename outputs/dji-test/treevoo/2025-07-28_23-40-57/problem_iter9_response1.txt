import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday and Breakout Momentum
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Sum of Volumes Over N Days
    df['sum_volume_N_days'] = df['volume'].rolling(window=N).sum()
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['momentum'] * (df['sum_volume_N_days'] / N)
    
    # Calculate Cumulative Volume-Weighted Momentum over a fixed period (e.g., 10 days)
    df['cumulative_volume_weighted_momentum'] = df['volume_adjusted_momentum'].rolling(window=N).sum()
    
    # Aggregate Momentum Over M Days
    df['aggregated_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=M).mean()
    
    return df['aggregated_momentum'].dropna()
