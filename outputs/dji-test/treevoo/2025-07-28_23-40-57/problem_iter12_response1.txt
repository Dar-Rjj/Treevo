```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['low']
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=15).sum()
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=15).sum()
    
    # Confirm with Volume Trend
    avg_volume_15_days = df['volume'].rolling(window=15).mean()
    current_volume = df['volume']
    volume_ratio = current_volume / avg_volume_15_days
    
    # Combine Price and Intraday Momentum
    combined_momentum = price_momentum * intraday_returns
    
    # Aggregate Momentum Over M Days
    aggregated_momentum = combined_momentum.rolling(window=15).sum()
    
    # Final Factor Calculation
    factor = pd.Series(0, index=df.index)
    factor[volume_ratio > 1.3] = aggregated_momentum + volume_weighted_momentum
    factor[volume_ratio <= 1.3] = 0.6 * (price_momentum + intraday_returns)
    
    return factor
```
