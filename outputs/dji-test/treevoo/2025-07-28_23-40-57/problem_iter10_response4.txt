```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Return
    previous_close = df['close'].shift(1)
    intraday_return = intraday_range / previous_close
    
    # Adjusted Intraday Return with Volume
    adjusted_intraday_return = intraday_return * df['volume']
    
    return adjusted_intraday_return
```
