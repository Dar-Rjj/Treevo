```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Momentum
    previous_close = df['close'].shift(1)
    momentum = high_low_spread / previous_close
    
    # Sum of Volumes Over N Days
    cumulative_volume = df['volume'].rolling(window=N).sum()
    
    # Adjusted Momentum
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum over a fixed period (e.g., 10 days)
    volume_weighted_momentum = adjusted_momentum * df['volume']
    cumulative_volume_weighted_momentum = volume_weighted_momentum.rolling(window=N).sum()
    
    # Aggregate Momentum Over M Days
    aggregated_momentum = cumulative_volume_weighted_momentum.rolling(window=M).sum()
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_opening_trend = opening_trend * df['amount']
    
    # Integrate Closing Price Reversal
    closing_reversal = df['close'] - df['close'].shift(1)
    
    # Combine Volume-Weighted, Amount-Weighted, and Closing Reversal
    combined_factor = (
        volume_weighted_momentum + 
        amount_weighted_opening_trend + 
        closing_reversal
    )
    
    return combined_factor
```
