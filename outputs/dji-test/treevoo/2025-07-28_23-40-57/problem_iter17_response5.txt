import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    momentum = high_low_spread / df['close'].shift(1)
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(5)
    intraday_returns = (df['high'] - df['low']) / df['low']
    
    # Adjust by Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=20).sum()
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=20).sum()
    
    # Confirm with Volume Trend
    avg_volume_15_days = df['volume'].rolling(window=15).mean()
    volume_ratio = df['volume'] / avg_volume_15_days
    combined_momentum = 0.3 * (price_momentum + intraday_returns)
    if volume_ratio > 1.5:
        combined_momentum = price_momentum * intraday_returns
        combined_momentum += volume_weighted_momentum.rolling(window=20).sum()
    
    # Calculate Intraday Volume Proportion
    overnight_volume = df['volume'].shift(1)
    intraday_volume_proportion = df['volume'] - overnight_volume
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    high_low_range = df['high'] - df['low']
    intraday_volume_intensity = intraday_volume_proportion / high_low_range
    
    # Calculate Weighted Intraday Volume Intensity
    weighted_intraday_volume_intensity = intraday_volume_intensity * intraday_return
    
    # Calculate Intraday and Breakout Momentum
    intraday_breakout_momentum = (intraday_volume_intensity * adjusted_momentum).rolling(window=20).sum()
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + intraday_breakout_momentum
    return alpha_factor
