import pandas as pd

def heuristics_v2(df, n_days=10, m_days=5):
    # Calculate Intraday Volume Proportion
    df['intraday_volume'] = df['volume'] - (df['volume'].shift(1) + df['volume'])
    df['intraday_volume_proportion'] = df['intraday_volume'] / df['volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / df['high_low_range']
    
    # Calculate Volume Impact Score
    df['sum_volume_n_days'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high_n_days'] = df['high'].rolling(window=n_days).mean()
    df['avg_low_n_days'] = df['low'].rolling(window=n_days).mean()
    df['price_range_n_days'] = df['avg_high_n_days'] - df['avg_low_n_days']
    df['volume_impact_score'] = df['price_range_n_days'] / df['sum_volume_n_days']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['prev_close'] = df['close'].shift(1)
    df['breakout_momentum'] = df['high_low_spread'] / df['prev_close']
    
    # Adjust by Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['breakout_momentum'] / df['cumulative_volume']
    
    # Generate Factor 1
    df['factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Confirm with Volume Trend
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['current_volume'] = df['volume']
    df['volume_ratio'] = df['current_volume'] / df['avg_volume_5_days']
    df['factor_2'] = df.apply(lambda row: row['intraday_return'] * row['adjusted_momentum'] if row['volume_ratio'] > 1 else 0, axis=1)
    
    # Adjust by Enhanced Volume Trend
    df['avg_volume_15_days'] = df['volume'].rolling(window=15).mean()
    df['enhanced_volume_ratio'] = df['current_volume'] / df['avg_volume_15_days']
    df['aggregate_momentum'] = df['adjusted_momentum'].rolling(window=m_days).sum()
    df['final_factor'] = df.apply(lambda row: row['factor_1'] * row['aggregate_momentum'] if row['enhanced_volume_ratio'] > 1.5 else 0.3 * (row['breakout_momentum'] + row['intraday_return']), axis=1)
    
    return df['final_factor']
