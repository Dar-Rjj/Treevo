```python
def heuristics_v2(data):
    # Calculate Price Momentum
    recent_close = data['close']
    close_10_days_ago = data['close'].shift(10)
    price_momentum = recent_close - close_10_days_ago
    
    # Confirm with Volume Increase
    recent_volume = data['volume']
    volume_10_days_ago = data['volume'].shift(10)
    volume_ratio = recent_volume / volume_10_days_ago
    
    # Apply the confirmation condition
    factor = price_momentum.where(volume_ratio > 1, 0)
    
    return factor
```
