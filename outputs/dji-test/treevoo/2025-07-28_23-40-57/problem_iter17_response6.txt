```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', and 'volume'
    
    Returns:
    pd.Series: Alpha factor values indexed by date
    """
    # Calculate the percentage change between the close and the open
    close_open_change = (df['close'] - df['open']) / df['open']
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the relative volume
    relative_volume = df['volume'] / avg_volume_5d
    
    # Combine the factors to create the final alpha factor
    alpha_factor = close_open_change * daily_range * relative_volume
    
    return alpha_factor
```
