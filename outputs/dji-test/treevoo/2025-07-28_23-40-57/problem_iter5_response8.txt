```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Momentum
    previous_close = df['close'].shift(1)
    momentum = high_low_spread / previous_close
    
    # Sum of Volumes Over N Days
    cumulative_volume = df['volume'].rolling(window=N).sum()
    
    # Adjust Momentum by Cumulative Volume
    adjusted_momentum = momentum / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    volume_weighted_momentum = (adjusted_momentum * df['volume']).rolling(window=N).sum()
    
    # Aggregate Momentum Over M Days
    aggregated_momentum = volume_weighted_momentum.rolling(window=M).sum()
    
    # Calculate Opening Price Trend Impact
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday_movement = (df['close'] - df['open']) * df['volume']
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_opening_trend = opening_trend * df['amount']
    
    # Combine Volume-Weighted and Amount-Weighted Movements
    combined_movement = volume_weighted_intraday_movement + amount_weighted_opening_trend
    
    # Final factor
    factor = aggregated_momentum + combined_movement
    
    return factor
```
