import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    n_days = 10
    volume_sum = df['volume'].rolling(window=n_days).sum()
    high_avg = df['high'].rolling(window=n_days).mean()
    low_avg = df['low'].rolling(window=n_days).mean()
    volume_impact_score = (high_avg - low_avg) / volume_sum
    
    # Multiply Intraday Return by Volume Impact Score
    factor_1 = intraday_return * volume_impact_score
    
    # Calculate Cumulative Volume Weighted Intraday Return
    cumulative_volume_weighted_return = (intraday_return * df['volume']).rolling(window=n_days).sum()
    factor_2 = cumulative_volume_weighted_return
    
    # Calculate Short-Term Average Volume
    short_term_period = 5
    short_term_avg_volume = df['volume'].rolling(window=short_term_period).mean()
    
    # Synthesize Factors
    final_factor = factor_1 * factor_2
    
    return final_factor
