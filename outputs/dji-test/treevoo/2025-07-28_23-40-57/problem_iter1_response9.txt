```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close
    breakout_momentum = high_low_spread / df['close'].shift(1)
    
    # Sum of Volumes Over N Days
    cumulative_volume = df['volume'].rolling(window=N).sum()
    
    # Adjust Breakout Momentum by Cumulative Volume
    adjusted_breakout_momentum = breakout_momentum / cumulative_volume
    
    # Aggregate Momentum Over M Days
    factor_values = adjusted_breakout_momentum.rolling(window=M).sum()
    
    return factor_values
```
