```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Price Change
    close_to_open_change = df['close'] - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['close', 'high', 'low', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Normalize Intraday Price Range by VWAP
    normalized_intraday_range = intraday_range / vwap
    
    # Multiply by Close-to-Open Price Change
    combined_indicator = normalized_intraday_range * close_to_open_change
    
    # Calculate Intraday Return
    prev_close = df['close'].shift(1)
    intraday_return = (df['high'] - df['low']) / prev_close
    
    # Calculate Cumulative Volume Weighted Intraday Return over 10 days
    cum_vol_weighted_return = (intraday_return * df['volume']).rolling(window=10).sum()
    
    # Multiply by Cumulative Volume Weighted Intraday Return
    final_combined_indicator = combined_indicator * cum_vol_weighted_return
    
    # Calculate Short-Term Average Volume over 5 days
    short_term_avg_volume = df['volume'].rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = final_combined_indicator * (short_term_avg_volume / df['volume'])
    
    return alpha_factor
```
