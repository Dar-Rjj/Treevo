```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    intraday_volume_proportion = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Intraday Volume Intensity
    price_range = df['high'] - df['low']
    intraday_volume_intensity = intraday_volume_proportion / price_range
    
    # Calculate Volume Impact Score
    volume_sum_n_days = df['volume'].rolling(window=n).sum()
    average_high = df['high'].rolling(window=n).mean()
    average_low = df['low'].rolling(window=n).mean()
    volume_impact_score = (average_high - average_low) / volume_sum_n_days
    
    # Multiply Intraday Return by Volume Impact Score
    weighted_intraday_return = intraday_return * volume_impact_score
    
    # Calculate Weighted Intraday Volume Intensity
    weighted_intraday_volume_intensity = intraday_volume_intensity * intraday_return
    
    # Calculate Intraday and Breakout Momentum
    intraday_breakout_momentum = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    adjusted_intraday_breakout_momentum = intraday_breakout_momentum / cumulative_volume
    
    # Incorporate Volume-Adjusted Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    adjusted_breakout_intensity = breakout_intensity / cumulative_volume
    
    # Combine Intraday and Breakout Intensities
    combined_intensity = (weighted_intraday_volume_intensity + adjusted_breakout_intensity) * intraday_return
    
    return combined_intensity
```
