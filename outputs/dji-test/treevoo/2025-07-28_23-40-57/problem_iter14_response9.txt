import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Divide by Previous Close for Breakout Momentum
    breakout_momentum = high_low_spread / df['close'].shift(1)
    
    # Adjust by Volume for Breakout Momentum
    volume_sum_n_days = df['volume'].rolling(window=10).sum()
    volume_adjusted_breakout_momentum = breakout_momentum * (volume_sum_n_days / 10)
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] + df['high'] + df['low'] + df['open']) / 4 * df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Cumulative Volume-Weighted Momentum
    final_adjusted_momentum = volume_adjusted_breakout_momentum * df['volume']
    cumulative_volume_weighted_momentum = final_adjusted_momentum.rolling(window=10).sum()
    
    # Confirm with Volume Trend
    average_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / average_volume
    
    if volume_ratio > 1.2:
        combined_momentum = price_momentum * intraday_return
        aggregated_momentum = cumulative_volume_weighted_momentum.rolling(window=10).sum()
        indicator = combined_momentum + aggregated_momentum
    else:
        indicator = 0.5 * (price_momentum + intraday_return)
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Adjust Intraday Price Range by VWAP
    adjusted_intraday_price_range = intraday_price_range * vwap
    
    # Multiply by Close-to-Open Price Change
    close_to_open_price_change = df['close'] - df['open']
    combined_indicator = adjusted_intraday_price_range * close_to_open_price_change
    
    # Incorporate Volume Change
    combined_indicator *= volume_change
    
    # Multiply Intraday Return by Volume Impact Score
    volume_impact_score = df['volume'] / df['volume'].rolling(window=10).mean()
    intraday_return_with_volume_impact = intraday_return * volume_impact_score
    
    # Incorporate Volume-Adjusted Breakout Momentum
    combined_factor = combined_indicator + volume_adjusted_breakout_momentum
    
    # Calculate Final Alpha Factor
    integrated_indicator = combined_factor + intraday_return_with_volume_impact
    
    return integrated_indicator
