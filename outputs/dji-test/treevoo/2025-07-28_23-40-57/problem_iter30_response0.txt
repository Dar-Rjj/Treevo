import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    df['intraday_volume_proportion'] = df['volume'].shift(1) - df['volume']
    
    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['intraday_volume_proportion'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    df['sum_of_volume'] = df['volume'].rolling(window=n).sum()
    df['avg_high_low_diff'] = (df['high'].rolling(window=n).mean() - df['low'].rolling(window=n).mean())
    df['volume_impact_score'] = df['avg_high_low_diff'] / df['sum_of_volume']
    
    # Multiply Intraday Return by Volume Impact Score
    df['intermediate_factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(n)
    
    # Adjust Price Momentum by Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum()
    df['adjusted_momentum'] = df['price_momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['aggregated_momentum'] = df['volume_weighted_momentum'].rolling(window=m).sum()
    
    # Confirm with Volume Trend
    df['avg_volume_past_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_past_5_days']
    
    # Combine Price and Intraday Momentum if Volume Ratio > 1
    df['combined_momentum'] = df['price_momentum'] * df['intraday_return']
    df.loc[df['volume_ratio'] > 1, 'final_factor'] = df['intermediate_factor_1'] * df['aggregated_momentum']
    df['final_factor'] = df.get('final_factor', 0)
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['intraday_breakout_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Volume
    df['adjusted_momentum_2'] = df['intraday_breakout_momentum'] / df['cumulative_volume']
    
    # Aggregate Momentum Over M Days
    df['aggregated_momentum_2'] = df['adjusted_momentum_2'].rolling(window=m).sum()
    
    # Calculate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=n).std()
    
    # Adjust Final Factor by Price Volatility
    df['final_factor_adjusted'] = df['final_factor'] / df['price_volatility']
    
    # Ensure Non-Negativity
    df['final_factor_adjusted'] = df['final_factor_adjusted'].apply(lambda x: max(x, 0))
    
    return df['final_factor_adjusted']
