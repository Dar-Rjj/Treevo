import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Confirm with Volume Increase
    df['volume_ratio'] = df['volume'] / df['volume'].shift(10)
    df['volume_increase'] = (df['volume_ratio'] > 1).astype(int)
    
    # Calculate EMAs
    df['short_ema'] = df['close'].ewm(span=12, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate EMA Difference
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Calculate Intraday Volume Proportion
    df['intraday_volume'] = df['volume'] - (df['volume'].shift(1) + df['volume'])
    df['overnight_volume'] = df['volume'].shift(1) + df['volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['price_range'] = df['high'] - df['low']
    df['intraday_volume_intensity'] = df['intraday_volume'] / df['price_range']
    
    # Calculate Volume Impact Score
    n_days = 10
    df['sum_volume'] = df['volume'].rolling(window=n_days).sum()
    df['avg_high'] = df['high'].rolling(window=n_days).mean()
    df['avg_low'] = df['low'].rolling(window=n_days).mean()
    df['price_spread'] = df['avg_high'] - df['avg_low']
    df['volume_impact_score'] = df['price_spread'] / df['sum_volume']
    
    # Multiply Intraday Return by Volume Impact Score
    df['factor'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['previous_close'] = df['close'].shift(1)
    df['momentum'] = df['high_low_spread'] / df['previous_close']
    
    # Adjust by Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['momentum'] / df['cumulative_volume']
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['cumulative_vol_weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['cumulative_vol_weighted_momentum'] = df['cumulative_vol_weighted_momentum'].rolling(window=n_days).sum()
    
    # Aggregate Momentum Over M Days
    m_days = 10
    df['aggregated_momentum'] = df['cumulative_vol_weighted_momentum'].rolling(window=m_days).sum()
    
    # Combine Indicators
    df['final_factor'] = df['ema_diff'] * df['aggregated_momentum']
    df['final_factor'] = np.where((df['price_momentum'] > 0) & (df['ema_diff'] > 0), df['final_factor'], 0)
    
    return df['final_factor']
