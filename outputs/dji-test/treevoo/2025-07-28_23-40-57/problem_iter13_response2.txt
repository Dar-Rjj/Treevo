def heuristics_v2(df, n=10):
    # Calculate Intraday Volume Proportion
    df['intraday_volume'] = df['volume'] - (df['volume'].shift(1) + df['volume'].shift(-1))
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Intensity
    df['price_range'] = df['high'] - df['low']
    df['intraday_volume_intensity'] = df['intraday_volume'] / df['price_range']
    
    # Calculate Volume Impact Score
    volume_sum = df['volume'].rolling(window=n).sum()
    avg_high = df['high'].rolling(window=n).mean()
    avg_low = df['low'].rolling(window=n).mean()
    price_diff = avg_high - avg_low
    df['volume_impact_score'] = price_diff / volume_sum
    
    # Multiply Intraday Return by Volume Impact Score
    df['factor'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Calculate Intraday and Breakout Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['momentum'] = df['high_low_spread'] / df['close'].shift(1)
    
    # Adjust by Volume
    cumulative_volume = df['volume'].rolling(window=n).sum()
    df['adjusted_momentum'] = df['momentum'] / cumulative_volume
    
    # Final Factor
    df['final_factor'] = df['factor'] + df['weighted_intraday_volume_intensity'] + df['adjusted_momentum']
    
    return df['final_factor']
