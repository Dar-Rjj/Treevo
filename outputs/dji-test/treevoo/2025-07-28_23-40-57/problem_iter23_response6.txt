```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Proportion
    df['overnight_volume'] = df['volume'].shift(1) - df['volume']
    
    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['overnight_volume'] / (df['high'] - df['low'])
    
    # Calculate Volume Impact Score
    volume_sum_10d = df['volume'].rolling(window=10).sum()
    avg_high_10d = df['high'].rolling(window=10).mean()
    avg_low_10d = df['low'].rolling(window=10).mean()
    df['volume_impact_score'] = (avg_high_10d - avg_low_10d) / volume_sum_10d
    
    # Intermediate Factor 1
    df['intermediate_factor_1'] = df['intraday_return'] * df['volume_impact_score']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust Price Momentum by Volume
    cumulative_volume = df['volume'].rolling(window=10).sum()
    df['adjusted_momentum'] = df['price_momentum'] / cumulative_volume
    
    # Calculate Cumulative Volume-Weighted Momentum
    df['cumulative_volume_weighted_momentum'] = (df['adjusted_momentum'] * df['volume']).rolling(window=10).sum()
    
    # Confirm with Volume Trend
    average_volume_5d = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / average_volume_5d
    
    # Combined Momentum
    df['combined_momentum'] = df['price_momentum'] * df['intraday_return']
    df['aggregated_momentum'] = df['cumulative_volume_weighted_momentum'].rolling(window=10).sum()
    
    # Final Factor
    df['final_factor'] = df.apply(lambda row: row['intermediate_factor_1'] * row['aggregated_momentum'] if row['volume_ratio'] > 1 else 0, axis=1)
    
    # Weighted Intraday Volume Intensity
    df['weighted_intraday_volume_intensity'] = df['intraday_volume_intensity'] * df['intraday_return']
    
    # Intraday and Breakout Momentum
    df['intraday_breakout_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjusted Momentum
    df['adjusted_momentum_final'] = df['intraday_breakout_momentum'] / cumulative_volume
    
    # Aggregate Momentum Over M Days
    df['aggregated_momentum_final'] = (df['adjusted_momentum_final'] * df['volume']).rolling(window=10).sum()
    
    return df['final_factor']
```
