```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Volume Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Sum over a fixed period (e.g., 10 days) to get the cumulative volume weighted intraday return
    df['cum_vol_weighted_intraday_return'] = df['vol_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate Short-Term Average Volume
    df['short_term_avg_volume'] = df['volume'].rolling(window=5).sum()
    
    # Final factor: Cumulative Volume Weighted Intraday Return divided by Short-Term Average Volume
    factor = df['cum_vol_weighted_intraday_return'] / df['short_term_avg_volume']
    
    return factor.dropna()
```
