import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Opening Trend
    opening_trend = df['open'] - df['close'].shift(1)
    
    # Adjust for Combined Movements
    combined_movements = (intraday_movement + opening_trend) * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact Score
    n_days = 10
    volume_sum_n_days = df['volume'].rolling(window=n_days).sum()
    avg_high_n_days = df['high'].rolling(window=n_days).mean()
    avg_low_n_days = df['low'].rolling(window=n_days).mean()
    high_low_diff = avg_high_n_days - avg_low_n_days
    volume_impact_score = high_low_diff / volume_sum_n_days
    
    # Combine Adjusted Movements and Intraday Return
    adjusted_movements_intraday_return = combined_movements * intraday_return * volume_impact_score
    
    # Calculate Cumulative Volume Weighted Intraday Return
    cumulative_volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=n_days).sum()
    
    # Calculate Short-Term Average Volume
    short_term_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Synthesize Factors
    factor_1 = adjusted_movements_intraday_return
    factor_2 = cumulative_volume_weighted_intraday_return
    final_factor = factor_1 * factor_2
    
    return final_factor
