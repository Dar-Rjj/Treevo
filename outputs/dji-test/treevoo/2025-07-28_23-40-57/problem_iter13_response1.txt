```python
def heuristics_v2(data):
    # Calculate Intraday Price Movement
    intraday_movement = data['high'] - data['low']
    
    # Calculate Opening Price Trend Impact
    open_trend = data['open'] - data['close'].shift(1)
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_movement = (intraday_movement * data['volume']).rolling(window=5).mean()
    
    # Calculate Amount-Weighted Opening Trend
    amount_weighted_open_trend = (open_trend * data['amount']).rolling(window=5).mean()

    # Combine Adjusted Movements and Trends
    combined_factor = (
        volume_weighted_movement + amount_weighted_open_trend 
        - volume_weighted_movement.rolling(window=5).mean() 
        - amount_weighted_open_trend.rolling(window=5).mean()
    )
    
    return combined_factor
```
