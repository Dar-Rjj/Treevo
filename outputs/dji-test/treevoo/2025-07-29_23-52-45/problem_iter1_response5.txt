```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, price_weight=0.7, volume_weight=0.3):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change().rolling(window=lookback_period).sum()
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].diff().abs().rolling(window=lookback_period).sum()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_price_momentum = price_momentum / intraday_volatility
    adjusted_volume_momentum = volume_momentum / intraday_volatility
    
    # Combine Adjusted Momentum Factors
    alpha_factor = (price_weight * adjusted_price_momentum) + (volume_weight * adjusted_volume_momentum)
    
    return alpha_factor
```
