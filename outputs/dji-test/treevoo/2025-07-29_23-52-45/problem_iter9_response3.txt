```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 12-day close price momentum
    df['momentum_12'] = df['close'] - df['close'].shift(12)
    
    # Calculate 5-day volume simple moving average
    df['sma_5_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_trend'] = df['volume'] - df['sma_5_volume']
    
    # Compute the ratio of close price to open price
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Calculate the high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute the amount-to-volume ratio
    df['amount_to_volume_ratio'] = df['amount'] / df['volume']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_12'] + 
        df['volume_trend'] + 
        df['open_close_ratio'] + 
        df['high_low_range'] + 
        df['amount_to_volume_ratio']
    )
    
    return df['alpha_factor']
```
