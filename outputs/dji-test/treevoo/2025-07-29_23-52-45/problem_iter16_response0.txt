```python
import pandas as pd

def heuristics_v2(df, lookback_period=30, short_term_days=5, long_term_days=60, price_momentum_weight=0.4, vol_weighted_short_term_weight=0.3, vol_weighted_long_term_weight=0.3):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Price Momentum
    price_momentum = df['close'].pct_change().rolling(window=lookback_period).sum()
    
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(short_term_days)) / df['close'].shift(short_term_days)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(long_term_days)) / df['close'].shift(long_term_days)
    
    # Volume-Weighted Short-Term Return
    vol_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    vol_weighted_long_term_return = long_term_return * df['volume']
    
    # Adjust Momentum by Intraday Volatility
    adjusted_price_momentum = price_momentum / intraday_volatility
    adjusted_vol_weighted_short_term_return = vol_weighted_short_term_return / intraday_volatility
    adjusted_vol_weighted_long_term_return = vol_weighted_long_term_return / intraday_volatility
    
    # Combine Adjusted Momentum Factors
    combined_factor = (
        price_momentum_weight * adjusted_price_momentum +
        vol_weighted_short_term_weight * adjusted_vol_weighted_short_term_return +
        vol_weighted_long_term_weight * adjusted_vol_weighted_long_term_return
    )
    
    return combined_factor
```
