```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    df['volume_adjusted_price_range'] = df['price_range'] * df['volume']
    
    # Momentum Indicator
    window = 10
    df['rolling_avg_volume_adjusted_price_range'] = df['volume_adjusted_price_range'].rolling(window=window).mean()
    
    # Calculate First Difference of Rolling Averages
    df['momentum_indicator'] = df['rolling_avg_volume_adjusted_price_range'].diff()
    
    # Weight Momentum by Volume
    lookback_period = 20
    df['sma_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_adjusted_momentum'] = df['momentum_indicator'] * (df['volume'] / df['sma_volume'])
    
    return df['volume_adjusted_momentum'].dropna()
```
