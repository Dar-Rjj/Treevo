import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term'] = df['long_term_return'] * df['volume']
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_short_term'] = df['volume_weighted_short_term'] / df['intraday_volatility']
    df['adjusted_long_term'] = df['volume_weighted_long_term'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term'] - df['adjusted_short_term']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=10).mean().diff()
    
    # Calculate 5-Day and 60-Day Moving Average of Close Price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_60'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['ma_5'] - df['ma_60']
    
    # Momentum Trend
    df['momentum_trend'] = df['combined_adjusted_momentum'].rolling(window=20).mean()
    df['strong_uptrend'] = df['combined_adjusted_momentum'] > df['momentum_trend']
    df['strong_downtrend'] = df['combined_adjusted_momentum'] < df['momentum_trend']
    
    # Daily Return based on Close Prices
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Identify Reversals with Close Price
    df['reversal_signal'] = (df['close'] > df['high'].shift(1)) | (df['close'] < df['low'].shift(1))
    
    # Analyze Changes in Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Detect Volume Surges
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surge'] = df['volume'] > df['avg_volume_10'] * 1.5
    
    # Develop a Price-Volume Indicator
    df['price_volume_indicator'] = df['daily_return'] + df['volume_change']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_indicator'] + df['trend_indicator'] + 0.5 * (df['close'] - df['open'])
    
    return df['final_alpha_factor']
