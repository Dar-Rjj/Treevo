```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    df['ShortTermReturn'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['LongTermReturn'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['VolumeWeightedShortTerm'] = df['ShortTermReturn'] * df['volume']
    df['VolumeWeightedLongTerm'] = df['LongTermReturn'] * df['volume']
    
    # Momentum Factor
    df['MomentumFactor'] = df['VolumeWeightedShortTerm'] - df['VolumeWeightedLongTerm']
    
    # Reversal Indicator
    df['ReversalIndicator'] = -df['MomentumFactor']
    
    # Price Change
    df['PriceChange'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['VolumeAdjustedPriceChange'] = df['PriceChange'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['SmoothedMomentum'] = df['VolumeAdjustedPriceChange'].rolling(window=10).mean()
    
    # Momentum Trend
    df['MomentumTrend'] = df['MomentumFactor'].rolling(window=20).mean()
    df['MomentumSignal'] = 'Neutral'
    df.loc[df['MomentumFactor'] > df['MomentumTrend'], 'MomentumSignal'] = 'StrongUptrend'
    df.loc[df['MomentumFactor'] < df['MomentumTrend'], 'MomentumSignal'] = 'StrongDowntrend'
    
    # Opening Gap
    df['OpeningGap'] = df['open'] - df['close'].shift(1)
    
    # Volume-Weighted Opening Gap
    df['VolumeWeightedOpeningGap'] = df['OpeningGap'] * df['volume']
    
    # Gap Momentum
    df['GapMomentum'] = df['VolumeWeightedOpeningGap'] + df['VolumeWeightedShortTerm']
    
    # Daily Volatility
    df['DailyVolatility'] = (df['high'] - df['low']) / df['close']
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['MomentumFactor'] + df['SmoothedMomentum'] + df['GapMomentum']
    
    return df['AlphaFactor']
```
