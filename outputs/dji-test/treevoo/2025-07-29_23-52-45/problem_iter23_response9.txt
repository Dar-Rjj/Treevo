```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = abs(df['high'] - df['low'])
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term / intraday_volatility
    adjusted_long_term_return = volume_weighted_long_term / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum
    
    # Simple Moving Average (SMA) of Close Prices (20 days)
    sma_close_20 = df['close'].rolling(window=20).mean()
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = (df['high'] - df['low']) * df['volume']
    
    # Smoothed Momentum Calculation
    rolling_avg_adjusted_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    first_difference = rolling_avg_adjusted_price_change - rolling_avg_adjusted_price_change.shift(1)
    
    # Momentum Trend Indicator
    combined_alpha_component = combined_adjusted_momentum + intraday_volatility + (volume_weighted_short_term - volume_weighted_long_term) + (df['close'] - sma_close_20) / df['close']
    moving_avg_combined_alpha = combined_alpha_component.rolling(window=20).mean()
    trend_signal = pd.Series(index=df.index, dtype=float)
    trend_signal[combined_alpha_component > moving_avg_combined_alpha] = 1
    trend_signal[combined_alpha_component < moving_avg_combined_alpha] = -1
    trend_signal[combined_alpha_component == moving_avg_combined_alpha] = 0
    
    # 12-Day Close Price Momentum
    close_momentum_12 = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    sma_volume_5 = df['volume'].rolling(window=5).mean()
    volume_sma_diff = df['volume'] - sma_volume_5
    
    # Open-Close Price Relationship
    open_close_ratio = df['close'] / df['open']
    
    # Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Final Combined Alpha Component
    final_combined_alpha = combined_adjusted_momentum + intraday_volatility + (volume_weighted_short_term - volume_weighted_long_term) + (df['close'] - sma_close_20) / df['close']
    
    # Final Alpha Factor
    final_alpha_factor = final_combined_alpha + close_momentum_12 + volume_sma_diff + open_close_ratio + daily_volatility + amount_to_volume_ratio
    
    return final_alpha_factor
```
