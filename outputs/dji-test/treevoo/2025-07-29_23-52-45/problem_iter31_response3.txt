import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = (df['high'] - df['low']) * df['volume']
    
    # Momentum Trend
    momentum_trend_ema = momentum_factor.ewm(span=20, adjust=False).mean()
    strong_uptrend_signal = (momentum_factor > momentum_trend_ema).astype(int)
    strong_downtrend_signal = (momentum_factor < momentum_trend_ema).astype(int)
    sideways_signal = 1 - (strong_uptrend_signal + strong_downtrend_signal)
    
    # Combine all the factors into a single factor
    combined_factor = (
        momentum_factor +
        reversal_indicator +
        volume_adjusted_price_change +
        strong_uptrend_signal -
        strong_downtrend_signal
    )
    
    return combined_factor
