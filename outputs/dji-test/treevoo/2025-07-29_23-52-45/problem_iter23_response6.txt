import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5) - 1)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(60) - 1)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Intraday Volatility
    df['intraday_volatility'] = abs(df['high'] - df['low'])
    
    # Adjusted Volume-Weighted Short-Term Return
    df['adjusted_volume_weighted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    
    # Adjusted Volume-Weighted Long-Term Return
    df['adjusted_volume_weighted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum_factor'] = 0.7 * df['adjusted_volume_weighted_short_term_return'] + 0.3 * df['adjusted_volume_weighted_long_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum_factor']
    
    # Smoothed Reversal Indicator
    df['smoothed_reversal_indicator'] = df['reversal_indicator'].rolling(window=3).mean()
    
    # Calculate 5-Day Moving Average of Close Price
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    df['60_day_ma'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['5_day_ma'] - df['60_day_ma']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_reversal_indicator'] + df['trend_indicator']
    
    # Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Intraday Volatility Ratio
    df['intraday_volatility_ratio'] = df['high'] / df['low']
    
    # Bullish Sentiment
    df['bullish_sentiment'] = df['high'] - df['close'].shift(1)
    
    # Bearish Sentiment
    df['bearish_sentiment'] = df['low'] - df['close'].shift(1)
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Activity Ratio
    df['volume_activity_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Composite Price-Volume Momentum
    df['composite_price_volume_momentum'] = df['daily_price_momentum'] * df['volume_activity_ratio']
    
    # Adjusted Intraday Volatility
    df['adjusted_intraday_volatility'] = df['intraday_volatility_ratio'] / df['volume_change']
    
    # Longer-Term Moving Average
    df['longer_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Slope of Moving Average
    df['slope_of_ma'] = df['longer_term_ma'] - df['longer_term_ma'].shift(1)
    
    # Relative Position to Moving Average
    df['relative_position_to_ma'] = df['close'] - df['longer_term_ma']
    
    # Opening Gap Analysis
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['final_alpha_factor'] + 
        df['composite_price_volume_momentum'] +
        df['adjusted_intraday_volatility'] +
        df['slope_of_ma'] +
        df['relative_position_to_ma'] +
        df['opening_gap']
    )
    
    return df['final_alpha_factor']
