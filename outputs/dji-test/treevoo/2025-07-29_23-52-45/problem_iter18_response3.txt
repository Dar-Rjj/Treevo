```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum
    lookback_period = 20
    df['price_momentum'] = df['close'].pct_change(periods=lookback_period).sum()
    
    # Calculate Short-Term Return
    short_term_days = 5
    df['short_term_return'] = (df['close'] - df['close'].shift(short_term_days)) / df['close'].shift(short_term_days)
    
    # Calculate Long-Term Return
    long_term_days = 60
    df['long_term_return'] = (df['close'] - df['close'].shift(long_term_days)) / df['close'].shift(long_term_days)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_return'] = df['long_term_return'] * df['volume']
    
    # Adjust Momentum by Intraday Range
    df['adjusted_price_momentum'] = df['price_momentum'] / df['intraday_range']
    df['adjusted_volume_weighted_short_return'] = df['volume_weighted_short_return'] / df['intraday_range']
    df['adjusted_volume_weighted_long_return'] = df['volume_weighted_long_return'] / df['intraday_range']
    
    # Combine Adjusted Momentum Factors
    price_weight = 0.4
    short_term_weight = 0.3
    long_term_weight = 0.3
    df['combined_adjusted_momentum'] = (
        price_weight * df['adjusted_price_momentum'] + 
        short_term_weight * df['adjusted_volume_weighted_short_return'] + 
        long_term_weight * df['adjusted_volume_weighted_long_return']
    )
    
    # Add Volume-Adjusted High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['volume_adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume']
    
    # Create Final Alpha Factor
    final_alpha_factor = (
        df['combined_adjusted_momentum'] + 
        df['volume_adjusted_high_low_ratio']
    )
    
    # Create Enhanced Reversal Indicator
    df['enhanced_reversal_indicator'] = -final_alpha_factor
    
    return df['enhanced_reversal_indicator']
```
