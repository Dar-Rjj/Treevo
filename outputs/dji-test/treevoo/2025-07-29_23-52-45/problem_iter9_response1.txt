```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()

    # Separate Positive and Negative Changes
    positive_changes = df['price_change'].apply(lambda x: x if x > 0 else 0)
    negative_changes = df['price_change'].apply(lambda x: abs(x) if x < 0 else 0)

    # Calculate Intraday Return
    df['intraday_return'] = df['high'] / df['low']

    # Compute Averages for Momentum Indicator
    avg_positive_changes = positive_changes.rolling(window=14).mean()
    avg_negative_changes = negative_changes.rolling(window=14).mean()

    # Calculate Relative Strength (RS)
    rs = avg_positive_changes / avg_negative_changes

    # Calculate RSI
    rsi = 100 - (100 / (1 + rs))

    # Adjust by Volume
    df['volume_change'] = df['volume'].diff()
    normalized_volume_change = df['volume_change'].apply(lambda x: 1 if x > 0 else -1)

    # Adjust RSI by Normalized Volume
    rsi_adjusted = rsi * normalized_volume_change

    # Volume-Adjusted Intraday Momentum
    volume_adjusted_intraday_momentum = df['volume'] * df['intraday_return']

    # Intraday Price Difference
    df['intraday_price_difference'] = df['close'] - df['open']

    # High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']

    # Synthesize Factors
    final_factor = (
        rsi_adjusted * 0.4 +
        df['intraday_return'] * 0.2 +
        volume_adjusted_intraday_momentum * 0.3 +
        high_low_ratio * 0.1
    )

    # Adjust for Intraday Price Difference
    final_factor += df['intraday_price_difference']

    return final_factor
```
