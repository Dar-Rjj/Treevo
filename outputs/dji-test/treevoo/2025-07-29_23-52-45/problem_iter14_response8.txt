```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()

    # Determine Price Momentum
    lookback_period = 20
    price_momentum = df['close'].pct_change().rolling(window=lookback_period).sum()
    close_to_open_return = (df['close'] - df['open']) / df['open']
    weighted_price_momentum = price_momentum * close_to_open_return

    # Adjust for Historical Volatility
    historical_volatility = df['close'].pct_change().std() * (252 ** 0.5)
    adjusted_price_momentum = weighted_price_momentum / historical_volatility

    # Integrate Volume and Price Momentum
    volume_change = df['volume'].diff()
    integrated_metric = volume_change * adjusted_price_momentum

    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)

    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']

    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']

    # Adjust Momentum by Intraday Volatility
    adjusted_volume_weighted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    adjusted_volume_weighted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    adjusted_integrated_metric = integrated_metric / intraday_volatility

    # Combine Adjusted Momentum Factors
    weight_price = 0.4
    weight_short_term = 0.3
    weight_long_term = 0.3
    combined_momentum_factor = (weight_price * adjusted_integrated_metric +
                                weight_short_term * adjusted_volume_weighted_short_term_return +
                                weight_long_term * adjusted_volume_weighted_long_term_return)

    # Reversal Indicator
    reversal_indicator = -combined_momentum_factor

    # Calculate 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()

    # Calculate 60-Day Moving Average of Close Price
    ma_60 = df['close'].rolling(window=60).mean()

    # Trend Indicator
    trend_indicator = ma_5 - ma_60

    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator

    return final_alpha_factor
```
