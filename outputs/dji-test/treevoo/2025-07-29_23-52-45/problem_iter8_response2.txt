```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_change'] = df['price_change'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] / df['low']
    
    # Compute Averages for Momentum Indicator
    df['avg_positive_change'] = df['positive_change'].rolling(window=14).mean()
    df['avg_negative_change'] = df['negative_change'].rolling(window=14).mean()
    
    # Calculate Relative Strength (RS)
    df['rs'] = df['avg_positive_change'] / df['avg_negative_change']
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Adjust by Volume
    df['volume_change'] = df['volume'].diff()
    df['normalized_volume_change'] = df['volume_change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Adjust RSI by Normalized Volume
    df['adjusted_rsi'] = df['rsi'] * df['normalized_volume_change']
    
    # Volume-Adjusted Intraday Momentum
    df['volume_adjusted_momentum'] = df['intraday_return'] * df['volume']
    
    # High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust Short-Term Return by Intraday Volatility
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    
    # Adjust Long-Term Return by Intraday Volatility
    df['adjusted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Synthesize Factors
    df['alpha_factor'] = (
        df['adjusted_rsi'] * 0.4 +
        df['intraday_return'] * 0.2 +
        df['volume_adjusted_momentum'] * 0.3 +
        df['high_low_ratio'] * 0.1 +
        df['combined_adjusted_momentum'] * 0.1
    )
    
    # Output Final Alpha Factor
    return df['alpha_factor']
```
