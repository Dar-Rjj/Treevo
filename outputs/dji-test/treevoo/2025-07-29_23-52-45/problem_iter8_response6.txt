import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5) - 1)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(60) - 1)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Enhanced Momentum Calculation
    smoothed_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    enhanced_momentum = smoothed_price_change.diff()
    
    # Momentum Trend Indicator
    momentum_trend = momentum_factor.rolling(window=20).mean()
    uptrend_signal = (momentum_factor > momentum_trend).astype(int)
    downtrend_signal = (momentum_factor < momentum_trend).astype(int)
    sideways_signal = 1 - (uptrend_signal + downtrend_signal)
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Compute Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Calculate Exponential Moving Average (EMA) of Returns
    ema_returns = daily_returns.ewm(span=10, adjust=False).mean()
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Incorporate Volume
    volume_weighted_return = daily_returns * df['volume']
    
    # Calculate EMA of Intraday Volatility
    ema_intraday_volatility = intraday_volatility.ewm(span=10, adjust=False).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    combined_factor = ema_returns + volume_weighted_return
    
    # Adjust Combined Factor by Intraday Volatility
    adjusted_combined_factor = combined_factor / ema_intraday_volatility
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    volume_adjusted_price_range = daily_price_range * df['volume']
    
    # Integrated Momentum Indicator
    rolling_avg_price_range = volume_adjusted_price_range.rolling(window=10).mean()
    integrated_momentum = rolling_avg_price_range.diff()
    
    # Integrate Volume-Adjusted Price Range Momentum
    final_momentum = adjusted_combined_factor + integrated_momentum
    
    # Final Alpha Factor
    final_alpha_factor = final_momentum
    
    return final_alpha_factor
