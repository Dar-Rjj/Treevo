```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Determine lookback periods
    sma_lookback = 20
    ema_returns_alpha = 0.1
    ema_volatility_alpha = 0.2
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Close'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Exponential Moving Average (EMA) of Returns
    df['EMA_Returns'] = df['Daily_Return'].ewm(alpha=ema_returns_alpha, adjust=False).mean()
    
    # Measure Intraday Volatility (High - Low)
    df['Intraday_Volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate EMA of Intraday Volatility
    df['EMA_Intraday_Volatility'] = df['Intraday_Volatility'].ewm(alpha=ema_volatility_alpha, adjust=False).mean()
    
    # Incorporate Volume: Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['Daily_Return'] * df['volume']
    df['Sum_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].cumsum()
    
    # Combine Momentum, Intraday Volatility, and Volume
    df['Combined_Factor'] = df['EMA_Returns'] - df['EMA_Intraday_Volatility'] + df['Sum_Volume_Weighted_Return']
    
    # Adjust Combined Factor by Intraday Volatility
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['EMA_Intraday_Volatility']
    
    # Final Alpha Factor
    alpha_factor = df['Adjusted_Combined_Factor']
    
    return alpha_factor
```
