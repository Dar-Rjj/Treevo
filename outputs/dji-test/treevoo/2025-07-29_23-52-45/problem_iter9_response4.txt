```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(120)) / df['close'].shift(120)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']).abs()
    
    # Adjust Short-Term Return by Intraday Range
    adjusted_short_term_return = volume_weighted_short_term_return / intraday_range
    
    # Adjust Long-Term Return by Intraday Range
    adjusted_long_term_return = volume_weighted_long_term_return / intraday_range
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum_factor = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum_factor
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator
    
    return final_alpha_factor
```
