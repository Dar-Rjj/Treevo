```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5) - 1)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(60) - 1)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Short-Term Return
    adjusted_volume_weighted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    
    # Adjusted Volume-Weighted Long-Term Return
    adjusted_volume_weighted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum_factor = (0.7 * adjusted_volume_weighted_short_term_return + 
                                         0.3 * adjusted_volume_weighted_long_term_return)
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum_factor
    
    # Smoothed Reversal Indicator (3-day Exponential Moving Average)
    smoothed_reversal_indicator = reversal_indicator.ewm(span=3).mean()
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5_day = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60_day = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5_day - ma_60_day
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_reversal_indicator + trend_indicator
    
    # Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(1)
    
    # Composite Price-Volume Momentum
    average_volume_20_day = df['volume'].rolling(window=20).mean()
    composite_price_volume_momentum = daily_price_momentum * (df['volume'] / average_volume_20_day)
    
    # Add Composite Price-Volume Momentum to the Final Alpha Factor
    final_alpha_factor += composite_price_volume_momentum
    
    return final_alpha_factor
```
