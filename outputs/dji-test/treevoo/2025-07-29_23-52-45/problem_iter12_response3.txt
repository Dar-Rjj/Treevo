```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Enhanced Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    df['reversal_indicator'] = df['reversal_indicator'].rolling(window=3).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Determine Price Momentum
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Volume and Price Momentum
    df['combined_volume_price_momentum'] = df['volume_change'] * df['close_to_open_return']
    
    # Adjust for Volatility - Calculate Average True Range (ATR)
    n = 14  # Typical value for ATR
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    df['true_range'] = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Final Adjusted Combined Metric
    df['final_metric'] = (df['combined_volume_price_momentum'] / df['atr']) + df['reversal_indicator']
    
    return df['final_metric']
```
