```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    df['price_momentum'] = df['close'].pct_change()
    
    # Integrate Volume and Price Momentum
    df['volume_change'] = df['volume'].pct_change()
    df['volume_surge'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['volume_weighted_short_term_return'] = ((df['close'] - df['close'].shift(5)) / df['close'].shift(5)) * df['volume']
    df['volume_weighted_long_term_return'] = ((df['close'] - df['close'].shift(60)) / df['close'].shift(60)) * df['volume']
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_volume_weighted_short_term'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_long_term'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_volume_weighted_short_term'] - df['adjusted_volume_weighted_long_term']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Moving Averages for Trend Analysis
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['60_day_ma'] = df['close'].rolling(window=60).mean()
    df['trend_indicator'] = df['5_day_ma'] - df['60_day_ma']
    df['additional_trend_indicator'] = df['20_day_ma'] - df['60_day_ma']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_indicator'] + df['trend_indicator'] + df['additional_trend_indicator']
    
    # Additional Price and Volume Factors
    df['daily_price_movement'] = df['close'] - df['open']
    df['uptrend_count'] = (df['close'] > df['open']).rolling(window=10).sum()
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_10'] = df['tr'].rolling(window=10).mean()
    df['atr_20'] = df['tr'].rolling(window=20).mean()
    df['atr_signal'] = (df['atr_10'] < df['atr_20']).astype(int)
    df['positive_volume_score'] = ((df['close'] > df['open']) & (df['volume'] > df['volume'].rolling(window=30).mean())).rolling(window=30).sum()
    df['intraday_bullishness'] = (df['high'] - df['open']) / (df['open'] - df['low'])
    df['bullishness_ratio'] = df['intraday_bullishness'].rolling(window=5).mean()
    
    # Combined Indicator
    df['combined_indicator'] = (df['combined_adjusted_momentum'] + df['uptrend_count'] + df['atr_signal'] + df['positive_volume_score'] + df['bullishness_ratio']) / 5
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_indicator'] + df['trend_indicator'] + df['additional_trend_indicator'] + df['combined_indicator']
    
    return df['final_alpha_factor'].dropna()
```
