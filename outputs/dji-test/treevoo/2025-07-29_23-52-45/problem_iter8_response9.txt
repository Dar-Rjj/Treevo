```python
import pandas as pd

def heuristics_v2(df):
    # Compute 20-day Simple Moving Average (SMA) of Close prices
    df['SMA_Close_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    df['Momentum_signal'] = df['close'] - df['SMA_Close_20']
    
    # Weight Momentum Signal by Volume
    lookback_period = 20
    df['AVG_Volume_20'] = df['volume'].rolling(window=lookback_period).mean()
    df['Normalized_Volume'] = df['volume'] / df['AVG_Volume_20']
    df['Volume_adjusted_momentum'] = df['Momentum_signal'] * df['Normalized_Volume']
    
    # Adjust for Price Volatility
    df['STD_Close_20'] = df['close'].rolling(window=lookback_period).std()
    df['Final_momentum_signal'] = df['Volume_adjusted_momentum'] / df['STD_Close_20']
    
    # Calculate Short-Term Return
    df['Short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['Long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['Volume_weighted_short_return'] = df['Short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['Volume_weighted_long_return'] = df['Long_term_return'] * df['volume']
    
    # Combined Momentum Factor
    df['Combined_momentum_factor'] = (df['Volume_weighted_short_return'] - 
                                      df['Volume_weighted_long_return'] + 
                                      df['Final_momentum_signal'])
    
    # Reversal Indicator
    df['Reversal_indicator'] = -df['Combined_momentum_factor']
    
    # Return the Reversal Indicator as the alpha factor
    return df['Reversal_indicator']
```
