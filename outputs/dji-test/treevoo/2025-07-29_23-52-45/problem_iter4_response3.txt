```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return_1 = (df['high'] - df['open']) / df['open']
    intraday_return_2 = (df['close'] - df['low']) / df['low']
    intraday_return = (intraday_return_1 + intraday_return_2) / 2

    # Volume-Adjusted Intraday Momentum
    volume_adjusted_momentum = (df['high'] - df['low']) / df['open'] * df['volume']

    # High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']

    # Combined Factor
    combined_factor = (intraday_return * 0.3) + (volume_adjusted_momentum * 0.5) + (high_low_ratio * 0.2)

    return combined_factor
```
