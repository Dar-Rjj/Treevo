import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Short-Term Return
    adjusted_volume_weighted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    
    # Adjusted Volume-Weighted Long-Term Return
    adjusted_volume_weighted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Momentum Factor
    momentum_factor = adjusted_volume_weighted_short_term_return - adjusted_volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend Indicator
    moving_average_momentum = momentum_factor.rolling(window=20).mean()
    uptrend_signal = (momentum_factor > moving_average_momentum).astype(int)
    downtrend_signal = (momentum_factor < moving_average_momentum).astype(int)
    trend_signal = uptrend_signal - downtrend_signal
    
    # Final Alpha Factor
    alpha_factor = momentum_factor + smoothed_momentum + trend_signal
    
    return alpha_factor
