```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    ema_span = 10
    ema_price_change = volume_adjusted_price_change.ewm(span=ema_span).mean()
    smoothed_momentum = ema_price_change - ema_price_change.shift(1)
    
    # Momentum Trend
    moving_average_window = 20
    momentum_trend = momentum_factor.rolling(window=moving_average_window).mean()
    
    # Advanced Filter
    # Volatility Adjustment
    volatility = df['close'].pct_change().rolling(window=30).std()
    adjusted_momentum_factor = momentum_factor / volatility
    
    # Trading Volume Anomaly
    average_volume = df['volume'].rolling(window=30).mean()
    high_volume_anomaly = (df['volume'] > 1.5 * average_volume).astype(int)
    low_volume_anomaly = (df['volume'] < 0.5 * average_volume).astype(int)
    
    # Combine all factors
    final_factor = adjusted_momentum_factor + smoothed_momentum + reversal_indicator
    final_factor = final_factor.where((high_volume_anomaly == 0) & (low_volume_anomaly == 0), 0)
    
    return final_factor
```
