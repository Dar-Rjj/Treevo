import pandas as pd

def heuristics_v2(df):
    # Determine lookback period for SMA
    sma_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Exponential Moving Average (EMA) of Returns
    ema_span_returns = 10
    df['EMA_returns'] = df['daily_return'].ewm(span=ema_span_returns, adjust=False).mean()
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Incorporate Volume
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    df['sum_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=sma_lookback).sum()
    
    # Calculate EMA of Intraday Volatility
    ema_span_volatility = 10
    df['EMA_volatility'] = df['intraday_volatility'].ewm(span=ema_span_volatility, adjust=False).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    df['combined_factor'] = df['EMA_returns'] + df['sum_volume_weighted_return']
    
    # Adjust Combined Factor by Intraday Volatility
    df['final_alpha_factor'] = df['combined_factor'] / df['EMA_volatility']
    
    # Output the Final Unnormalized Alpha Factor
    return df['final_alpha_factor']
