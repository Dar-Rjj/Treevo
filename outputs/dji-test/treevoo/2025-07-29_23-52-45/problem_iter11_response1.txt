```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = abs(df['high'] - df['low'])
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_volume_weighted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum_factor'] = df['adjusted_volume_weighted_long_term_return'] - df['adjusted_volume_weighted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum_factor']
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    lookback_period_sma = 20
    df['sma_close'] = df['close'].rolling(window=lookback_period_sma).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    df['final_alpha_factor'] = (
        df['combined_adjusted_momentum_factor'] + 
        df['intraday_volatility'] + 
        (df['volume_weighted_short_term_return'] - df['volume_weighted_long_term_return'])
    )
    
    return df['final_alpha_factor']
```
