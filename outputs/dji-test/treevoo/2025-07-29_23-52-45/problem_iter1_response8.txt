```python
import pandas as pd

def heuristics_v2(data):
    lookback_period = 20
    
    # Calculate Simple Moving Average of Close Price
    data['SMA_Close'] = data['close'].rolling(window=lookback_period).mean()
    
    # Subtract SMA from Current Close Price to get Momentum Signal
    data['Momentum_signal'] = data['close'] - data['SMA_Close']
    
    # Calculate Average Volume
    data['AVG_Volume'] = data['volume'].rolling(window=lookback_period).mean()
    
    # Normalize Volume
    data['Normalized_Volume'] = data['volume'] / data['AVG_Volume']
    
    # Multiply Momentum Signal by Normalized Volume
    data['Volume_adjusted_momentum'] = data['Momentum_signal'] * data['Normalized_Volume']
    
    return data['Volume_adjusted_momentum']
```
