import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Detect Bullish/Bearish Engulfing Patterns
    df['bullish_engulfing'] = ((df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1))).astype(int)
    df['bearish_engulfing'] = ((df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1))).astype(int)
    
    # Detect Harami Patterns
    df['body'] = abs(df['close'] - df['open'])
    df['prev_body'] = df['body'].shift(1)
    df['bullish_harami'] = ((df['body'] < df['prev_body']) & (df['close'] > df['open'])).astype(int)
    df['bearish_harami'] = ((df['body'] < df['prev_body']) & (df['close'] < df['open'])).astype(int)
    
    # Compute changes in trading volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Create composite alpha factors
    df['factor_1'] = df['daily_return'] * df['volume_change']
    df['factor_2'] = (df['bullish_engulfing'] + df['bearish_engulfing']) * df['volume_change']
    df['factor_3'] = (df['bullish_harami'] + df['bearish_harami']) * df['volume_change']
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = df['factor_1'] + df['factor_2'] + df['factor_3']
    
    return df['alpha_factor']
