import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Intraday Price Difference
    intraday_price_diff = df['close'] - df['open']
    
    # Volume-Adjusted Intraday Momentum
    volume_adjusted_momentum = intraday_return * df['volume']
    
    # High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Combined Factor
    combined_factor = (0.4 * intraday_return) + (0.6 * volume_adjusted_momentum) + (0.1 * high_low_ratio)
    
    return combined_factor
