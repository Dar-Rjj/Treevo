import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    rolling_average_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    smoothed_momentum = rolling_average_price_change - rolling_average_price_change.shift(1)
    
    # Momentum Trend
    moving_average_momentum_factor = momentum_factor.rolling(window=20).mean()
    strong_uptrend_signal = (momentum_factor > moving_average_momentum_factor).astype(int)
    strong_downtrend_signal = (momentum_factor < moving_average_momentum_factor).astype(int)
    sideways_signal = 1 - (strong_uptrend_signal + strong_downtrend_signal)
    
    # Final Alpha Factor
    alpha_factor = momentum_factor * (strong_uptrend_signal - strong_downtrend_signal) + smoothed_momentum
    return alpha_factor
