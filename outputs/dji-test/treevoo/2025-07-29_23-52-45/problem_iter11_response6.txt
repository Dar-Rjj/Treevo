```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback period
    lookback_period = 20
    
    # Calculate Price Momentum
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_signal = df['close'] - sma_close
    
    # Weight Momentum Signal by Volume
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    normalized_volume = df['volume'] / avg_volume
    volume_adjusted_momentum = momentum_signal * normalized_volume
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_return = short_term_return * df['volume']
    volume_weighted_long_return = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Returns
    adjusted_short_return = volume_weighted_short_return / intraday_volatility
    adjusted_long_return = volume_weighted_long_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_return - adjusted_short_return
    
    # Enhanced Combined Momentum Factor
    enhanced_combined_momentum = combined_adjusted_momentum + volume_adjusted_momentum
    
    # Enhanced Reversal Indicator
    enhanced_reversal_indicator = -enhanced_combined_momentum
    
    return enhanced_reversal_indicator
```
