```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['vol_weighted_short_term'] = df['short_term_return'] * df['volume']
    df['vol_weighted_long_term'] = df['long_term_return'] * df['volume']
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    df['adj_vol_weighted_short_term'] = df['vol_weighted_short_term'] / df['intraday_volatility']
    df['adj_vol_weighted_long_term'] = df['vol_weighted_long_term'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adj_momentum'] = df['adj_vol_weighted_long_term'] - df['adj_vol_weighted_short_term']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adj_momentum']
    
    # Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['vol_adj_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['vol_adj_price_change'].rolling(window=10).mean()
    df['smoothed_momentum_diff'] = df['smoothed_momentum'] - df['smoothed_momentum'].shift(1)
    
    # Calculate 5-Day Moving Average of Close Price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    df['ma_60_close'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['ma_5_close'] - df['ma_60_close']
    
    # Momentum Trend
    df['mom_trend'] = df['combined_adj_momentum'].rolling(window=20).mean()
    df['mom_trend_signal'] = 0
    df.loc[df['combined_adj_momentum'] > df['mom_trend'], 'mom_trend_signal'] = 1  # Strong Uptrend
    df.loc[df['combined_adj_momentum'] < df['mom_trend'], 'mom_trend_signal'] = -1  # Strong Downtrend
    
    # Daily Return based on Close Prices
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Identify Reversals with Close Price
    df['reversal'] = 0
    df.loc[df['close'] > df['high'].shift(1), 'reversal'] = 1
    df.loc[df['close'] < df['low'].shift(1), 'reversal'] = -1
    
    # Analyze Changes in Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Detect Volume Surges
    avg_volume = df['volume'].rolling(window=20).mean()
    df['volume_surge'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Develop a Price-Volume Indicator
    df['price_volume_indicator'] = df['daily_return'] * df['volume_change']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['reversal_indicator'] + 
                                df['trend_indicator'] + 
                                (df['close'] - df['open']))
    
    return df['final_alpha_factor']
```
