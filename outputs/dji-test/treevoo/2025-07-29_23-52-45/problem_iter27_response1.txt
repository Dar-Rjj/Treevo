```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    lookback_period = 5
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['momentum'] = df['close'].pct_change(lookback_period).mul(df['close_to_open_return'])
    
    # Integrate Volume and Price Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['weighted_momentum'] = df['momentum'] * df['volume_change']
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_short_term_return'] = df['short_term_return'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Calculate 5-Day Moving Average of Close Price
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    df['60_day_ma'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['5_day_ma'] - df['60_day_ma']
    
    # Calculate 20-Day Moving Average of Close Price
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    # Additional Trend Indicator
    df['additional_trend_indicator'] = df['20_day_ma'] - df['60_day_ma']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['reversal_indicator'] + df['trend_indicator'] + df['reversal_indicator'] + df['additional_trend_indicator']
    
    return df['alpha_factor']
```
