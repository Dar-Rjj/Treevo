import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Identify Price Reversal Patterns
    df['bullish_engulfing'] = ((df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))).astype(int)
    df['bearish_engulfing'] = ((df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))).astype(int)
    
    # Compute Average True Range
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], abs(row['high'] - df['close'].shift(1).loc[row.name]), abs(row['low'] - df['close'].shift(1).loc[row.name])), axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Compute Changes in Trading Volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Examine Volume Spike Events
    k = 2.0  # Threshold for significant volume increase
    df['volume_spike'] = (df['volume'] > k * df['volume'].shift(1)).astype(int)
    
    # Create Composite Alpha Factors
    df['factor_1'] = df['intraday_volatility'] * df['volume_change']
    df['factor_2'] = (df['bullish_engulfing'] + df['bearish_engulfing']) * df['volume_change']
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = df['factor_1'] + df['factor_2']
    
    return df['alpha_factor']
