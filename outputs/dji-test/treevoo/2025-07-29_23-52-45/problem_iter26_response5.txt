```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term / intraday_volatility
    adjusted_long_term_return = volume_weighted_long_term / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum_factor = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum_factor
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    rolling_avg_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    first_difference = rolling_avg_volume_adjusted_price_change - rolling_avg_volume_adjusted_price_change.shift(1)
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5_close = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60_close = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5_close - ma_60_close
    
    # Momentum Trend
    momentum_trend = combined_adjusted_momentum_factor.rolling(window=20).mean()
    strong_uptrend_signal = (combined_adjusted_momentum_factor > momentum_trend).astype(int)
    strong_downtrend_signal = (combined_adjusted_momentum_factor < momentum_trend).astype(int)
    neutral_signal = (combined_adjusted_momentum_factor == momentum_trend).astype(int)
    
    # 12-Day Close Price Momentum
    close_price_momentum_12 = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    sma_5_volume = df['volume'].rolling(window=5).mean()
    volume_sma_diff = df['volume'] - sma_5_volume
    
    # Open-Close Price Relationship
    open_close_ratio = df['close'] / df['open']
    
    # Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (
        reversal_indicator + 
        trend_indicator + 
        (df['close'] - df['open']) + 
        combined_adjusted_momentum_factor + 
        intraday_volatility + 
        (volume_weighted_short_term - volume_weighted_long_term) + 
        (df['close'] - df['close'].rolling(window=20).mean()) / df['close'] + 
        close_price_momentum_12 + 
        volume_sma_diff + 
        open_close_ratio + 
        daily_volatility + 
        amount_to_volume_ratio
    )
    
    return final_alpha_factor
```
