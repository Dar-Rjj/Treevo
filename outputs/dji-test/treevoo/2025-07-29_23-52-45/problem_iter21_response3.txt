import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(120)) / df['close'].shift(120)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * (df['volume'] + df['volume'].shift(1))
    volume_weighted_long_term = long_term_return * (df['volume'] + df['volume'].shift(1))
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Price Change
    price_change = (df['high'] - df['low']) / df['open']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * (df['volume'] + df['volume'].shift(1))
    
    # Smoothed Momentum Calculation
    rolling_avg_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=15).mean()
    smoothed_momentum_first_difference = rolling_avg_volume_adjusted_price_change.diff()
    
    # Momentum Trend
    moving_average_momentum_factor = momentum_factor.rolling(window=30).mean()
    strong_uptrend_signal = (momentum_factor > moving_average_momentum_factor).astype(int)
    strong_downtrend_signal = (momentum_factor < moving_average_momentum_factor).astype(int)
    sideways_or_neutral_signal = 1 - (strong_uptrend_signal + strong_downtrend_signal)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        momentum_factor +
        smoothed_momentum_first_difference +
        (strong_uptrend_signal - strong_downtrend_signal) * 10  # Emphasize trend signals
    )
    
    return alpha_factor
