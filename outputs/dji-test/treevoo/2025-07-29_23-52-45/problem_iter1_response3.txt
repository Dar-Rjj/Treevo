import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Intraday Price Difference
    intraday_price_diff = df['high'] - df['low']
    
    # Volume-Adjusted Intraday Momentum
    volume_adjusted_momentum = intraday_return * df['volume']
    
    # High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Combined Factor
    combined_factor = 0.3 * intraday_return + 0.5 * volume_adjusted_momentum + 0.2 * high_low_ratio
    
    return combined_factor
