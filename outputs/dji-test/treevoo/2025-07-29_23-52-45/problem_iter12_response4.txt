```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Price Momentum
    lookback_period = 20
    df['SMA_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_signal'] = df['close'] - df['SMA_close']
    
    # Calculate Volume Averaged Momentum
    volume_lookback_period = 20
    df['AVG_volume'] = df['volume'].rolling(window=volume_lookback_period).mean()
    df['volume_adjusted_momentum'] = df['momentum_signal'] * df['volume'] / df['AVG_volume']
    
    # Smooth Volume Averaged Momentum
    smoothing_period = 5
    df['smoothed_volume_weighted_momentum'] = df['volume_adjusted_momentum'].ewm(span=smoothing_period, adjust=False).mean()
    
    # Adjust for Close Price Volatility
    df['STD_close'] = df['close'].rolling(window=lookback_period).std()
    df['final_smoothed_volume_weighted_momentum'] = df['smoothed_volume_weighted_momentum'] / df['STD_close']
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Combine Momentum Factors with Short-Term Return
    df['combined_momentum_factor'] = 0.7 * df['final_smoothed_volume_weighted_momentum'] + 0.3 * df['volume'] * df['short_term_return']
    
    # Subtract Long-Term Return Component
    df['long_term_return_weighted'] = df['long_term_return'] * df['volume']
    df['hybrid_momentum_reversal_signal'] = df['combined_momentum_factor'] - df['long_term_return_weighted']
    
    # Reversal Indicator
    df['reversal_indicator'] = -1 * df['hybrid_momentum_reversal_signal']
    
    return df['reversal_indicator'].dropna()
```
