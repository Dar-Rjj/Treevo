```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum over a Short-Term Lookback Period (n=5)
    df['price_momentum'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Calculate Volume Momentum over a Short-Term Lookback Period (n=5)
    df['volume_momentum'] = df['volume'].diff().abs().rolling(window=5).sum()
    
    # Calculate Short-Term Return (n=5)
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return (m=60)
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['intraday_volatility']
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['intraday_volatility']
    
    # Combine Adjusted Momentum Factors with fixed weights
    price_weight = 0.7
    volume_weight = 0.3
    df['combined_momentum'] = (price_weight * df['adjusted_price_momentum'] +
                               volume_weight * df['adjusted_volume_momentum'])
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['combined_momentum'] - df['volume_weighted_long_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['momentum_factor']
    
    return df['reversal_indicator'].dropna()
```
