```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Price
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate Momentum Signal
    momentum_signal = df['close'] - sma_close
    
    # Calculate Average Volume
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    
    # Weight Momentum Signal by Volume
    volume_adjusted_momentum = momentum_signal * (df['volume'] / avg_volume)
    
    # Calculate Standard Deviation of Close Prices
    std_close = df['close'].rolling(window=lookback_period).std()
    
    # Adjust for Price Volatility
    final_momentum_signal = volume_adjusted_momentum / std_close
    
    return final_momentum_signal
```
