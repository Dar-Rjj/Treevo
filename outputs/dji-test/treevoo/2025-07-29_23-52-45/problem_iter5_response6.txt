```python
import pandas as pd

def heuristics_v2(df):
    # Determine lookback periods
    sma_lookback = 20
    ema_returns_smooth = 12
    ema_volatility_smooth = 12
    rolling_window = 10

    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Close'] = df['close'].rolling(window=sma_lookback).mean()

    # Compute Daily Returns
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Exponential Moving Average (EMA) of Returns
    df['EMA_Returns'] = df['Daily_Return'].ewm(span=ema_returns_smooth, adjust=False).mean()

    # Measure Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).abs()

    # Incorporate Volume
    df['Volume_Weighted_Return'] = df['Daily_Return'] * df['volume']
    df['Total_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=sma_lookback).sum() / df['volume'].rolling(window=sma_lookback).sum()

    # Calculate EMA of Intraday Volatility
    df['EMA_Intraday_Volatility'] = df['Intraday_Volatility'].ewm(span=ema_volatility_smooth, adjust=False).mean()

    # Combine Momentum, Intraday Volatility, and Volume
    df['Combined_Factor'] = (df['EMA_Returns'] + df['Total_Volume_Weighted_Return']) / df['EMA_Intraday_Volatility']

    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']

    # Volume Adjusted Price Range
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] * df['volume']

    # Momentum Indicator
    df['Rolling_Avg_Volume_Adjusted_Price_Range'] = df['Volume_Adjusted_Price_Range'].rolling(window=rolling_window).mean()
    df['First_Diff_Rolling_Avg'] = df['Rolling_Avg_Volume_Adjusted_Price_Range'].diff()

    # Integrate Volume-Adjusted Price Range Momentum
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['First_Diff_Rolling_Avg']

    return df['Final_Alpha_Factor']
```
