import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    df['ma_diff'] = df['ma_10'] - df['ma_30']
    
    # Calculate the 10-day standard deviation of the returns
    df['std_10'] = df['return'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the returns
    df['std_30'] = df['return'].rolling(window=30).std()
    
    # Calculate the ratio of the 10-day to 30-day standard deviation
    df['std_ratio'] = df['std_10'] / df['std_30']
    
    # Calculate the 10-day rolling sum of the volume
    df['vol_sum_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 30-day rolling sum of the volume
    df['vol_sum_30'] = df['volume'].rolling(window=30).sum()
    
    # Calculate the ratio of the 10-day to 30-day volume sums
    df['vol_ratio'] = df['vol_sum_10'] / df['vol_sum_30']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['ma_diff'] * df['std_ratio'] * df['vol_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
