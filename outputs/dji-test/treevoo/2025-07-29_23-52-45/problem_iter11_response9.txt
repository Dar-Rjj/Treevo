import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    daily_range = df['high'] - df['low']
    close_to_open_return = (df['close'] - df['open']) / df['open']
    price_momentum = daily_range * close_to_open_return
    
    # Adjust for Historical Volatility
    daily_returns = df['close'].pct_change()
    n = 30  # You can adjust the number of days for historical volatility
    historical_volatility = daily_returns.rolling(window=n).std()
    integrated_metric = price_momentum / historical_volatility
    
    # Integrate Volume and Price Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_weighted_momentum = volume_change * integrated_metric
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(120)) / df['close'].shift(120)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Adjust Short-Term Return by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    
    # Adjust Long-Term Return by Intraday Volatility
    adjusted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5_day = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60_day = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5_day - ma_60_day
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator
    
    return final_alpha_factor
