import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    lookback_period = 10
    price_momentum = df['close'].pct_change().rolling(window=lookback_period).sum()
    close_to_open_return = (df['close'] - df['open']) / df['open']
    weighted_price_momentum = price_momentum * close_to_open_return
    
    # Integrate Volume and Price Momentum
    volume_change = df['volume'].diff()
    momentum_with_volume = weighted_price_momentum * volume_change
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = short_term_return / intraday_volatility
    adjusted_long_term_return = long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Calculate 20-Day Moving Average of Close Price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Additional Trend Indicator
    additional_trend_indicator = ma_20 - ma_60
    
    # Calculate 10-Day Moving Average of Close Price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Additional Trend Indicator
    additional_trend_indicator_10 = ma_10 - ma_60
    
    # Final Alpha Factor
    final_alpha_factor = (
        reversal_indicator + trend_indicator +
        reversal_indicator + additional_trend_indicator +
        reversal_indicator + additional_trend_indicator_10
    )
    
    return final_alpha_factor
