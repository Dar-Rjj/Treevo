```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback period
    lookback_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close price
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute Price Momentum
    momentum_signal = df['close'] - sma_close
    
    # Calculate the average volume over the lookback period
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    
    # Compute Volume Adjusted Momentum Signal
    volume_adjusted_momentum = momentum_signal * (df['volume'] / avg_volume)
    
    return volume_adjusted_momentum
```
