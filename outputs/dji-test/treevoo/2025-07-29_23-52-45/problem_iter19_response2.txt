import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day moving average of the closing prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between today's closing price and the 5-day moving average
    df['momentum_factor'] = df['close'] - df['5_day_ma_close']
    
    # Calculate the daily return (close price of day t - close price of day t-1)
    df['daily_return'] = df['close'].diff()
    
    # Calculate the volume weighted average price (VWAP) over the last 5 days
    df['VWAP'] = (df['volume'] * df['close']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the ratio of the highest price to the lowest price within the same day
    df['intraday_volatility'] = df['high'] / df['low']
    
    # Calculate the change in VWAP from day t-5 to day t
    df['VWAP_change'] = df['VWAP'] - df['VWAP'].shift(5)
    
    # Calculate the cumulative sum of daily returns over the past 5 days
    df['cumulative_daily_returns'] = df['daily_return'].rolling(window=5).sum()
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_factor'] +
        df['daily_return'] +
        df['VWAP_change'] +
        df['cumulative_daily_returns'] +
        (df['intraday_volatility'] - 1)  # Normalize intraday volatility by subtracting 1
    )
    
    # Drop any rows with NaN values due to rolling window calculations
    df.dropna(inplace=True)
    
    return df['alpha_factor']
