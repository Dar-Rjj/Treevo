import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices and volume.
    
    The factor is based on the following heuristics:
    1. Calculate the average of the ratio of (close - open) / (high - low) over a 5-day rolling window.
    2. Calculate the exponential moving average (EMA) of the volume with a span of 10 days.
    3. Combine the two factors by normalizing them and taking their difference.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the ratio of (close - open) / (high - low)
    df['price_ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the 5-day rolling average of the price ratio
    df['price_ratio_rolling_avg'] = df['price_ratio'].rolling(window=5).mean()
    
    # Calculate the exponential moving average (EMA) of the volume with a span of 10 days
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Normalize the price ratio rolling average
    df['price_ratio_rolling_avg_norm'] = (df['price_ratio_rolling_avg'] - df['price_ratio_rolling_avg'].mean()) / df['price_ratio_rolling_avg'].std()
    
    # Normalize the volume EMA
    df['volume_ema_norm'] = (df['volume_ema'] - df['volume_ema'].mean()) / df['volume_ema'].std()
    
    # Combine the two normalized factors by taking their difference
    df['alpha_factor'] = df['price_ratio_rolling_avg_norm'] - df['volume_ema_norm']
    
    return df['alpha_factor']
