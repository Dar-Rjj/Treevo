```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum over a Lookback Period (e.g., 20 days)
    price_momentum = df['close'].pct_change().rolling(window=20).sum()
    
    # Calculate Short-Term Return (5 days)
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return (60 days)
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_return = long_term_return * df['volume']
    
    # Adjust Momentum by Intraday Range
    adjusted_price_momentum = price_momentum / intraday_range
    adjusted_volume_weighted_short_return = volume_weighted_short_return / intraday_range
    adjusted_volume_weighted_long_return = volume_weighted_long_return / intraday_range
    
    # Combine Adjusted Momentum Factors with fixed weights
    combined_momentum_factors = (
        0.4 * adjusted_price_momentum +
        0.3 * adjusted_volume_weighted_short_return +
        0.3 * adjusted_volume_weighted_long_return
    )
    
    # Add Volume-Adjusted High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    volume_adjusted_high_low_ratio = high_low_ratio * df['volume']
    
    # Calculate Price Volatility (20-day standard deviation)
    price_volatility = df['close'].rolling(window=20).std()
    
    # Adjust Final Alpha Factor by Price Volatility
    final_alpha_factor = combined_momentum_factors + volume_adjusted_high_low_ratio
    final_alpha_factor_adjusted = final_alpha_factor / price_volatility
    
    # Create Enhanced Reversal Indicator
    enhanced_reversal_indicator = -final_alpha_factor_adjusted
    
    return enhanced_reversal_indicator
```
