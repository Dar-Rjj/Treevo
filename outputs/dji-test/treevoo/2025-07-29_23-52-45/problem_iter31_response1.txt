import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    lookback_period = 10
    df['price_momentum'] = df['close'].pct_change().rolling(window=lookback_period).sum()
    
    # Adjusted Short-Term Return
    short_term_days = 5
    df['short_term_return'] = (df['close'] - df['close'].shift(short_term_days)) / df['close'].shift(short_term_days)
    
    # Adjusted Long-Term Return
    long_term_days = 60
    df['long_term_return'] = (df['close'] - df['close'].shift(long_term_days)) / df['close'].shift(long_term_days)
    
    # Combined Adjusted Momentum Factor
    df['combined_momentum'] = df['short_term_return'] - df['long_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_momentum']
    
    # 5-Day Moving Average of Close Price
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # 60-Day Moving Average of Close Price
    df['60_day_ma'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['5_day_ma'] - df['60_day_ma']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['reversal_indicator'] + df['trend_indicator']
    
    return df['alpha_factor'].dropna()
