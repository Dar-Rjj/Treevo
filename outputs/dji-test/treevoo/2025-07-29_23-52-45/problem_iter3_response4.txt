```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Determine High and Low Price Momentum
    df['daily_range'] = df['high'] - df['low']
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['price_momentum'] = df['daily_range'] * df['close_to_open_return']
    
    # Compute Volume Impact on Price Movement
    df['volume_impact_on_price'] = df['volume_change'] * df['price_momentum']
    
    # Generate Volatility Adjusted Trend Score
    n = 10  # Number of days for True Range calculation
    df['previous_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'previous_close']].apply(
        lambda x: max(x['high'] - x['low'], x['high'] - x['previous_close'], x['previous_close'] - x['low']), axis=1
    )
    df['true_range_n'] = df['true_range'].rolling(window=n).max()
    df['volatility_adjusted_trend_score'] = df['volume_impact_on_price'] / df['true_range_n']
    
    return df['volatility_adjusted_trend_score']
```
