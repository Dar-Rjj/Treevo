```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum over a lookback period (e.g., 10 days)
    lookback_period = 10
    df['close_pct_change'] = df['close'].pct_change()
    df['price_momentum'] = df['close_pct_change'].rolling(window=lookback_period).sum()
    
    # Calculate Short-Term Return (5 days)
    short_term_days = 5
    df['short_term_return'] = df['close'].pct_change(periods=short_term_days)
    
    # Calculate Long-Term Return (60 days)
    long_term_days = 60
    df['long_term_return'] = df['close'].pct_change(periods=long_term_days)
    
    # Calculate Average Volume for Short-Term and Long-Term
    df['avg_volume_short_term'] = df['volume'].rolling(window=short_term_days).mean()
    df['avg_volume_long_term'] = df['volume'].rolling(window=long_term_days).mean()
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['avg_volume_short_term']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['avg_volume_long_term']
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['intraday_volatility']
    df['adjusted_volume_weighted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combine Adjusted Momentum Factors
    weight_price = 0.4
    weight_short_term = 0.3
    weight_long_term = 0.3
    
    df['combined_momentum_factor'] = (
        weight_price * df['adjusted_price_momentum'] +
        weight_short_term * df['adjusted_volume_weighted_short_term_return'] +
        weight_long_term * df['adjusted_volume_weighted_long_term_return']
    )
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_momentum_factor']
    
    # Final Alpha Factor
    alpha_factor = df['combined_momentum_factor']
    
    return alpha_factor
```
