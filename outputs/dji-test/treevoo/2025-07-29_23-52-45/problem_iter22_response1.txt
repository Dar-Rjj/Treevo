```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Compute the ratio of today’s high to today’s low
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Determine the difference between today’s high and yesterday’s close
    df['high_yesterday_close_diff'] = df['high'] - df['close'].shift(1)
    
    # Calculate the difference between today’s low and yesterday’s close
    df['low_yesterday_close_diff'] = df['low'] - df['close'].shift(1)
    
    # Calculate the difference between today’s volume and yesterday’s volume
    df['volume_diff'] = df['volume'].diff()
    
    # Compute the ratio of today’s volume to the average volume over the past n days (e.g., 30 days)
    n = 30
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Multiply the difference between today’s close and yesterday’s close by the ratio of today’s volume to the average volume
    df['momentum_volume_factor'] = df['close_diff'] * df['volume_ratio']
    
    # Divide the ratio of today’s high to today’s low by the difference in volume between today and yesterday
    df['volatility_volume_factor'] = df['high_low_ratio'] / df['volume_diff'].replace(0, 1)  # Avoid division by zero
    
    # Calculate the moving average of the close price over the past n days
    df['moving_avg'] = df['close'].rolling(window=n).mean()
    
    # Determine the slope of the moving average over the past n days
    df['slope_moving_avg'] = df['moving_avg'].diff() / n
    
    # Compare today’s close price to the moving average
    df['close_vs_moving_avg'] = df['close'] - df['moving_avg']
    
    # Calculate the difference between today’s open and yesterday’s close
    df['open_gap'] = df['open'] - df['close'].shift(1)
    
    # Determine the difference between today’s close and today’s open
    df['intraday_performance'] = df['close'] - df['open']
    
    # Create a composite alpha factor
    df['alpha_factor'] = (
        0.3 * df['momentum_volume_factor'] +
        0.2 * df['volatility_volume_factor'] +
        0.15 * df['slope_moving_avg'] +
        0.15 * df['close_vs_moving_avg'] +
        0.1 * df['open_gap'] +
        0.1 * df['intraday_performance']
    )
    
    return df['alpha_factor']
```
