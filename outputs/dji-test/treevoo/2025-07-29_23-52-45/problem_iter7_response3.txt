```python
import pandas as pd

def heuristics_v2(df):
    # Compute 20-day Simple Moving Average (SMA) of Close prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    momentum_signal = df['close'] - sma_20
    
    # Weight Momentum Signal by Volume
    volume_adjusted_momentum = momentum_signal * df['volume']
    
    # Compute 20-day Standard Deviation (STD) of Close prices
    std_20 = df['close'].rolling(window=20).std()
    
    # Adjust for Price Volatility
    final_momentum_signal = volume_adjusted_momentum / std_20
    
    return final_momentum_signal
```
