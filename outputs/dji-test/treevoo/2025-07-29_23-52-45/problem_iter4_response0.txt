```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Moving Average of High-Low Range
    lookback_period = 10
    df['sma_high_low_range'] = df['high_low_range'].rolling(window=lookback_period).mean()
    
    # Determine Momentum
    df['signal'] = 0
    df.loc[df['high_low_range'] > df['sma_high_low_range'], 'signal'] = 1  # Buy
    df.loc[df['high_low_range'] < df['sma_high_low_range'], 'signal'] = -1  # Sell
    
    return df['signal']
```
