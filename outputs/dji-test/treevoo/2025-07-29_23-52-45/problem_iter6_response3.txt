```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean()
    
    # Compute First Difference of Smoothed Momentum
    smoothed_momentum_diff = smoothed_momentum.diff()
    
    # Momentum Trend Indicator
    momentum_trend_ma = momentum_factor.rolling(window=20).mean()
    
    # Define signals based on Momentum Factor and its Moving Average
    strong_uptrend_signal = (momentum_factor > momentum_trend_ma).astype(int)
    strong_downtrend_signal = (momentum_factor < momentum_trend_ma).astype(int)
    sideways_signal = ((momentum_factor <= momentum_trend_ma) & (momentum_factor >= momentum_trend_ma)).astype(int)
    
    # Combine all the signals into a single factor
    enhanced_momentum_factor = (strong_uptrend_signal - strong_downtrend_signal + smoothed_momentum_diff + reversal_indicator)
    
    return enhanced_momentum_factor
```
