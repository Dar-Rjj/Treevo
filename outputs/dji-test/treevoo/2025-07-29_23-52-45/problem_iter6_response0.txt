```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(90)) / df['close'].shift(90)
    
    # Volume-Weighted Short-Term Return
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_weighted_short_term_return = short_term_return * avg_volume_5_days
    
    # Volume-Weighted Long-Term Return
    avg_volume_90_days = df['volume'].rolling(window=90).mean()
    volume_weighted_long_term_return = long_term_return * avg_volume_90_days
    
    # Amount-Weighted Short-Term Return
    avg_amount_5_days = df['amount'].rolling(window=5).mean()
    amount_weighted_short_term_return = short_term_return * avg_amount_5_days
    
    # Amount-Weighted Long-Term Return
    avg_amount_90_days = df['amount'].rolling(window=90).mean()
    amount_weighted_long_term_return = long_term_return * avg_amount_90_days
    
    # Combined Volume-Amount Weighted Short-Term Return
    combined_short_term_return = volume_weighted_short_term_return + amount_weighted_short_term_return
    
    # Combined Volume-Amount Weighted Long-Term Return
    combined_long_term_return = volume_weighted_long_term_return + amount_weighted_long_term_return
    
    # Momentum Factor
    momentum_factor = combined_short_term_return - combined_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    return reversal_indicator
```
