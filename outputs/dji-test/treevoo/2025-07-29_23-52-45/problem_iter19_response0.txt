```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Short-Term
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    volume_weighted_short_term = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Volume Adjusted Price Change
    price_change = (df['high'] - df['low']) * df['volume']
    
    # Smoothed Momentum Calculation
    rolling_avg_price_change = price_change.rolling(window=10).mean()
    smoothed_momentum = rolling_avg_price_change - rolling_avg_price_change.shift(10)
    
    # Momentum Trend
    momentum_trend = momentum_factor.rolling(window=20).mean()
    trend_signal = pd.Series(index=df.index, name='trend_signal')
    
    for i in range(20, len(df)):
        if momentum_factor.iloc[i] > momentum_trend.iloc[i]:
            trend_signal.iloc[i] = 1  # Strong Uptrend Signal
        elif momentum_factor.iloc[i] < momentum_trend.iloc[i]:
            trend_signal.iloc[i] = -1  # Strong Downtrend Signal
        else:
            trend_signal.iloc[i] = 0  # Sideways or Neutral Signal
    
    return momentum_factor + smoothed_momentum + trend_signal
```
