```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate the relative close position within the daily range
    relative_close_position = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()

    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()

    # Calculate the momentum factor
    momentum = df['close'] / ma_30 - 1

    # Calculate the volume change
    volume_change = df['volume'].pct_change()

    # Combine the factors into a single alpha factor
    alpha_factor = (
        0.4 * relative_close_position +
        0.3 * momentum +
        0.2 * (ma_10 - ma_30) +
        0.1 * volume_change
    )

    return alpha_factor
```
