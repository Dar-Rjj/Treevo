```python
import pandas as pd

def heuristics_v2(df):
    # Determine lookback period for SMA
    lookback_period_sma = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=lookback_period_sma).mean()
    
    # Compute Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Set smoothing factor for EMA
    smoothing_factor_ema = 12
    
    # Calculate Exponential Moving Average (EMA) of Returns
    df['EMA_returns'] = df['daily_return'].ewm(span=smoothing_factor_ema, adjust=False).mean()
    
    # Measure Price Volatility
    df['high_low_range'] = df['high'] - df['low']
    
    # Set smoothing factor for EMA of High-Low Ranges
    smoothing_factor_vol_ema = 20
    
    # Calculate EMA of High-Low Ranges
    df['EMA_volatility'] = df['high_low_range'].ewm(span=smoothing_factor_vol_ema, adjust=False).mean()
    
    # Incorporate Volume
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Sum the Volume-Weighted Returns
    df['total_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=lookback_period_sma).sum()
    
    # Combine Momentum, Volatility, and Volume
    df['combined_factor'] = df['EMA_returns'] / df['EMA_volatility'] * df['total_volume_weighted_return']
    
    # Final Alpha Factor (unnormalized)
    alpha_factor = df['combined_factor']
    
    return alpha_factor
```
