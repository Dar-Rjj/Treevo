```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend Indicator
    momentum_trend = momentum_factor.rolling(window=20).mean()
    uptrend_signal = (momentum_factor > momentum_trend).astype(int)
    downtrend_signal = (momentum_factor < momentum_trend).astype(int)
    sideways_signal = ((momentum_factor <= momentum_trend) & (momentum_factor >= momentum_trend)).astype(int)
    momentum_trend_indicator = uptrend_signal - downtrend_signal
    
    # 12-Day Close Price Momentum
    close_price_momentum_12 = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    sma_volume_5 = df['volume'].rolling(window=5).mean()
    volume_sma_diff = df['volume'] - sma_volume_5
    
    # Open-Close Price Relationship
    open_close_ratio = df['close'] / df['open']
    
    # Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        momentum_factor + 
        reversal_indicator + 
        smoothed_momentum + 
        momentum_trend_indicator + 
        close_price_momentum_12 + 
        volume_sma_diff + 
        open_close_ratio + 
        daily_volatility + 
        amount_to_volume_ratio
    )
    
    return alpha_factor
```
