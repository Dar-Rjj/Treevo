```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_term'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_term'] = df['long_term_return'] * df['volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_weighted_short_term'] - df['volume_weighted_long_term']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['momentum_factor']
    
    # Calculate Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    
    # Compute First Difference
    df['smoothed_momentum_diff'] = df['smoothed_momentum'] - df['smoothed_momentum'].shift(1)
    
    # Momentum Trend Indicator
    df['momentum_trend_ma'] = df['momentum_factor'].rolling(window=20).mean()
    df['momentum_trend_signal'] = 0
    df.loc[df['momentum_factor'] > df['momentum_trend_ma'], 'momentum_trend_signal'] = 1
    df.loc[df['momentum_factor'] < df['momentum_trend_ma'], 'momentum_trend_signal'] = -1
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_factor'] + df['reversal_indicator'] + df['smoothed_momentum_diff']
    
    return df['alpha_factor']
```
