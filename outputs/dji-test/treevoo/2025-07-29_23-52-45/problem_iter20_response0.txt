```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    close_pct_change = df['close'].pct_change().fillna(0)
    close_to_open_return = (df['close'] - df['open']) / df['open']
    price_momentum = (close_pct_change.rolling(window=10).sum() * close_to_open_return).fillna(0)
    
    # Adjust for Historical Volatility
    daily_returns = df['close'].pct_change().fillna(0)
    historical_volatility = daily_returns.rolling(window=20).std().fillna(0)
    integrated_metric = price_momentum / historical_volatility
    
    # Integrate Volume and Price Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_price_momentum = volume_change * price_momentum
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Adjust Momentum by Intraday Volatility
    adjusted_price_momentum = integrated_metric / intraday_volatility
    adjusted_volume_weighted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    adjusted_volume_weighted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combine Adjusted Momentum Factors
    combined_adjusted_momentum = (
        0.4 * adjusted_price_momentum +
        0.3 * adjusted_volume_weighted_short_term_return +
        0.3 * adjusted_volume_weighted_long_term_return
    )
    
    # Combined Adjusted Momentum Factor
    combined_factor = combined_adjusted_momentum - (adjusted_volume_weighted_short_term_return - adjusted_volume_weighted_long_term_return)
    
    # Reversal Indicator
    reversal_indicator = -combined_factor
    
    # Calculate 10-Day Moving Average of Close Price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate 20-Day Moving Average of Close Price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Trend Indicator
    trend_indicator = ma_10 - ma_20
    
    # Open-Price Impact
    open_price_impact = (df['close'] - df['open']) / df['open']
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator + open_price_impact
    
    return final_alpha_factor
```
