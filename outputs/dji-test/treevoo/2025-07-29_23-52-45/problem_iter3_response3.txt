```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(90)) / df['close'].shift(90)
    
    # Calculate Average Volume for Short-Term
    avg_volume_short = df['volume'].rolling(window=5).mean()
    
    # Calculate Average Volume for Long-Term
    avg_volume_long = df['volume'].rolling(window=90).mean()
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * avg_volume_short
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * avg_volume_long
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    return reversal_indicator
```
