import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean()
    smoothed_momentum_diff = smoothed_momentum - smoothed_momentum.shift(10)
    
    # Momentum Trend
    momentum_trend = momentum_factor.rolling(window=20).mean()
    strong_uptrend_signal = (momentum_factor > momentum_trend).astype(int)
    strong_downtrend_signal = (momentum_factor < momentum_trend).astype(int)
    neutral_signal = (momentum_factor == momentum_trend).astype(int)
    
    # Additional Factors
    cumulative_volume = df['volume'].rolling(window=5).sum()
    cumulative_price_change = (df['high'] - df['low'].shift(5))
    cumulative_volume_adjusted_price_change = cumulative_price_change * cumulative_volume
    rolling_cumulative_volume_adjusted_price_change = cumulative_volume_adjusted_price_change.rolling(window=10).mean()
    rolling_cumulative_diff = rolling_cumulative_volume_adjusted_price_change - rolling_cumulative_volume_adjusted_price_change.shift(10)
    
    # Combined Indicator
    combined_indicator = (momentum_factor + rolling_cumulative_volume_adjusted_price_change) / 2
    positive_signal = (combined_indicator > 0).astype(int)
    negative_signal = (combined_indicator < 0).astype(int)
    neutral_combined_signal = (combined_indicator == 0).astype(int)
    
    # Final Alpha Factor
    alpha_factor = (strong_uptrend_signal - strong_downtrend_signal + positive_signal - negative_signal) * combined_indicator
    
    return alpha_factor
