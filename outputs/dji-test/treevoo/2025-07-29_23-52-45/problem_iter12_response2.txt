```python
import pandas as pd

def heuristics_v2(df):
    # Compute 20-day Simple Moving Average (SMA) of Close price
    sma_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    momentum_signal = df['close'] - sma_close_20
    
    # Define lookback period for volume
    lookback_period = 20
    
    # Compute 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=lookback_period).mean()
    
    # Compute Volume Ratio
    volume_ratio = df['volume'] / avg_volume_20
    
    # Weight Momentum Signal by Volume
    volume_adjusted_momentum = momentum_signal * volume_ratio
    
    # Compute 20-day standard deviation of Close price
    std_close_20 = df['close'].rolling(window=20).std()
    
    # Adjust for Price Volatility
    final_momentum_signal = volume_adjusted_momentum / std_close_20
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_return = long_term_return * df['volume']
    
    # Combined Momentum Factor
    combined_momentum_factor = volume_weighted_short_return - volume_weighted_long_return + final_momentum_signal
    
    # Reversal Indicator
    reversal_indicator = -combined_momentum_factor
    
    return reversal_indicator
```
