```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['60_day_SMA'] = df['close'].rolling(window=60).mean()

    # Calculate Price Relative to SMA
    df['close_to_5_day_SMA'] = df['close'] / df['5_day_SMA']
    df['close_to_20_day_SMA'] = df['close'] / df['20_day_SMA']
    df['close_to_60_day_SMA'] = df['close'] / df['60_day_SMA']

    # Calculate Rate of Change (ROC)
    df['10_day_ROC'] = df['close'].pct_change(periods=10)
    df['30_day_ROC'] = df['close'].pct_change(periods=30)
    df['90_day_ROC'] = df['close'].pct_change(periods=90)

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['5_day_avg_high_low_range'] = df['high_low_range'].rolling(window=5).mean()

    # Calculate Volume-Price Correlation
    df['5_day_corr_close_volume'] = df['close'].rolling(window=5).corr(df['volume'])
    df['20_day_corr_close_volume'] = df['close'].rolling(window=20).corr(df['volume'])
    df['60_day_corr_close_volume'] = df['close'].rolling(window=60).corr(df['volume'])

    # Calculate Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    df['5_day_avg_open_close_spread'] = df['open_close_spread'].rolling(window=5).mean()

    # Calculate Volatility
    df['5_day_std_close'] = df['close'].rolling(window=5).std()
    df['20_day_std_close'] = df['close'].rolling(window=20).std()
    df['60_day_std_close'] = df['close'].rolling(window=60).std()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['5_day_avg_VWAP'] = df['daily_VWAP'].rolling(window=5).mean()

    # Combine factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_to_5_day_SMA'] + 
        df['close_to_20_day_SMA'] + 
        df['close_to_60_day_SMA'] + 
        df['10_day_ROC'] + 
        df['30_day_ROC'] + 
        df['90_day_ROC'] + 
        df['5_day_avg_high_low_range'] + 
        df['5_day_corr_close_volume'] + 
        df['20_day_corr_close_volume'] + 
        df['60_day_corr_close_volume'] + 
        df['5_day_avg_open_close_spread'] + 
        df['5_day_std_close'] + 
        df['20_day_std_close'] + 
        df['60_day_std_close'] + 
        df['5_day_avg_VWAP']
    )

    return df['alpha_factor']
```
