```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Separate Positive and Negative Changes
    positive_changes = df['price_change'].where(df['price_change'] > 0, 0)
    negative_changes = -df['price_change'].where(df['price_change'] < 0, 0)
    
    # Calculate Average Positive and Negative Changes over 14 Days
    avg_positive_changes = positive_changes.rolling(window=14).mean()
    avg_negative_changes = negative_changes.rolling(window=14).mean()
    
    # Compute RSIVA
    # Calculate Relative Strength (RS)
    rs = avg_positive_changes / avg_negative_changes
    
    # Calculate RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust by Volume
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Normalize Volume Change
    normalized_volume_change = volume_change.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Multiply RSI by Normalized Volume Change
    rsiva = rsi * normalized_volume_change
    
    # Output RSIVA
    return rsiva
```
