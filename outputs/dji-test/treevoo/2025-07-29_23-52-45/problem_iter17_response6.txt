import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean()
    first_difference = smoothed_momentum - smoothed_momentum.shift(1)
    
    # Momentum Trend
    momentum_trend = momentum_factor.rolling(window=20).mean()
    
    # Signals
    uptrend_signal = (momentum_factor > momentum_trend).astype(int)
    downtrend_signal = (momentum_factor < momentum_trend).astype(int)
    neutral_signal = ((momentum_factor <= momentum_trend) & (momentum_factor >= momentum_trend)).astype(int)
    
    # Combine signals
    combined_signal = uptrend_signal - downtrend_signal + neutral_signal * 0.5
    
    # Final Alpha Factor
    alpha_factor = momentum_factor + first_difference + combined_signal
    
    return alpha_factor
