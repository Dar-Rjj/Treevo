import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['Short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['Long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['Average_volume_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_weighted_short_return'] = df['Short_term_return'] * df['Average_volume_5']
    
    # Volume-Weighted Long-Term Return
    df['Average_volume_60'] = df['volume'].rolling(window=60).mean()
    df['Volume_weighted_long_return'] = df['Long_term_return'] * df['Average_volume_60']
    
    # Calculate Price Momentum
    lookback_period = 20
    df['SMA_20'] = df['close'].rolling(window=lookback_period).mean()
    df['Momentum_signal'] = df['close'] - df['SMA_20']
    
    # Weight Momentum Signal by Volume
    df['Average_volume_20'] = df['volume'].rolling(window=lookback_period).mean()
    df['Normalized_volume'] = df['volume'] / df['Average_volume_20']
    df['Volume_adjusted_momentum'] = df['Momentum_signal'] * df['Normalized_volume']
    
    # Combined Momentum Factor
    df['Combined_momentum_factor'] = (df['Volume_weighted_short_return'] 
                                      - df['Volume_weighted_long_return'] 
                                      + df['Volume_adjusted_momentum'])
    
    # Reversal Indicator
    df['Reversal_indicator'] = -df['Combined_momentum_factor']
    
    # Calculate Price Range
    df['Price_range'] = df['high'] - df['low']
    
    # Volume-Adjusted Price Range
    df['Adjusted_price_range'] = df['Price_range'] * df['volume']
    
    # Smoothed Adjusted Price Range
    window_size = 10
    df['Smoothed_adjusted_price_range'] = (df['Adjusted_price_range'].rolling(window=window_size).sum()
                                           - df['Adjusted_price_range'].rolling(window=window_size).sum().shift(1))
    
    # Final Alpha Factor
    alpha_factor = df['Reversal_indicator'] + df['Smoothed_adjusted_price_range']
    return alpha_factor.dropna()
