```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum over a 20-day lookback period
    df['price_momentum'] = df['close'].pct_change().rolling(window=20).sum()
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_return'] = df['long_term_return'] * df['volume']
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['intraday_volatility']
    df['adjusted_volume_weighted_short_return'] = df['volume_weighted_short_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_long_return'] = df['volume_weighted_long_return'] / df['intraday_volatility']
    
    # Combine Adjusted Momentum Factors with fixed weights
    combined_momentum = (
        0.4 * df['adjusted_price_momentum'] + 
        0.3 * df['adjusted_volume_weighted_short_return'] + 
        0.3 * df['adjusted_volume_weighted_long_return']
    )
    
    # Add Volume-Adjusted High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['volume_adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum + df['volume_adjusted_high_low_ratio']
    
    # Create Enhanced Reversal Indicator
    enhanced_reversal_indicator = -final_alpha_factor
    
    return enhanced_reversal_indicator
```
