```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    df['volume_adjusted_price_range'] = df['price_range'] * df['volume']
    
    # Compute Rolling Average of Volume Adjusted Price Range (10 days)
    df['rolling_avg_volume_adjusted_price_range'] = df['volume_adjusted_price_range'].rolling(window=10).mean()
    
    # Momentum Calculation
    df['momentum_first_diff'] = df['rolling_avg_volume_adjusted_price_range'].diff()
    
    # SMA(Volume, 20)
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Volume-adjusted momentum
    df['volume_adjusted_momentum'] = df['momentum_first_diff'] * (df['volume'] / df['sma_volume_20'])
    
    # Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjusted Returns
    df['adjusted_short_term_return'] = df['short_term_return'] * (df['volume'] / df['intraday_volatility'])
    df['adjusted_long_term_return'] = df['long_term_return'] * (df['volume'] / df['intraday_volatility'])
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Integrated Momentum Indicator
    df['integrated_momentum_indicator'] = df['volume_adjusted_momentum'] + df['reversal_indicator']
    
    return df['integrated_momentum_indicator']
```
