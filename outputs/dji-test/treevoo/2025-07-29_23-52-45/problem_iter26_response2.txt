```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term'] = df['long_term_return'] * df['volume']
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_short_term_return'] = df['volume_weighted_short_term'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['volume_weighted_long_term'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Calculate 5-Day and 60-Day Moving Averages of Close Price
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['60_day_ma'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['60_day_ma'] - df['5_day_ma']
    
    # Momentum Trend
    df['momentum_trend'] = df['combined_adjusted_momentum'].rolling(window=20).mean()
    
    # Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume-Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=10).mean().diff()
    
    # Compute Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Exponential Moving Average (EMA) of Returns
    df['ema_returns'] = df['daily_return'].ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Calculate EMA of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=10, adjust=False).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    df['combined_factor'] = (df['ema_returns'] + df['volume_weighted_return']) / df['ema_intraday_volatility']
    
    # Momentum Indicator
    df['momentum_indicator'] = df['volume_adjusted_price_change'].rolling(window=10).median().diff()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_indicator'] + df['trend_indicator'] + (df['close'] - df['open'])
    
    return df['final_alpha_factor']
```
