import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is designed to predict future stock returns using current and past data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series of the calculated alpha factor, indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 21-day moving average of the daily return
    df['ma_21'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate the 5-day moving standard deviation of the daily return
    df['std_5'] = df['daily_return'].rolling(window=5).std()
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].rolling(window=21).mean()) / df['volume'].rolling(window=21).std()
    
    # Calculate the high-low range
    df['range'] = df['high'] - df['low']
    
    # Calculate the 21-day moving average of the high-low range
    df['range_ma_21'] = df['range'].rolling(window=21).mean()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['ma_21'] + df['std_5']) * df['norm_volume'] * df['range_ma_21']
    
    return df['alpha_factor']
