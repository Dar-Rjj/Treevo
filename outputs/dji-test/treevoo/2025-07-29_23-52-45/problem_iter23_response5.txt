```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Returns by Intraday Volatility
    adjusted_volume_weighted_short_term = volume_weighted_short_term / intraday_volatility
    adjusted_volume_weighted_long_term = volume_weighted_long_term / intraday_volatility
    
    # Momentum Factor
    momentum_factor = adjusted_volume_weighted_short_term - adjusted_volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend
    momentum_trend_20 = momentum_factor.rolling(window=20).mean()
    momentum_signal = pd.Series(0, index=df.index)
    momentum_signal[momentum_factor > momentum_trend_20] = 1
    momentum_signal[momentum_factor < momentum_trend_20] = -1
    
    # Additional Factors
    cumulative_volume = df['volume'].rolling(window=5).sum()
    cumulative_price_change = df['high'] - df['low'].shift(5)
    cumulative_volume_adjusted_price_change = cumulative_price_change * cumulative_volume
    rolling_avg_cumulative_volume_adjusted = cumulative_volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Combined Indicator
    combined_indicator = (momentum_factor + cumulative_volume_adjusted_price_change) / 2
    combined_signal = pd.Series(0, index=df.index)
    combined_signal[combined_indicator > 0] = 1
    combined_signal[combined_indicator < 0] = -1
    
    # 5-Day and 60-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    rsi_signal = pd.Series(0, index=df.index)
    rsi_signal[rsi > 70] = 1
    rsi_signal[rsi < 30] = -1
    
    # Stochastic Oscillator
    lowest_low_14 = df['low'].rolling(window=14).min()
    highest_high_14 = df['high'].rolling(window=14).max()
    k = 100 * (df['close'] - lowest_low_14) / (highest_high_14 - lowest_low_14)
    d = k.rolling(window=3).mean()
    stoch_signal = pd.Series(0, index=df.index)
    stoch_signal[(k > 80) & (d > 80)] = 1
    stoch_signal[(k < 20) & (d < 20)] = -1
    
    # Enhanced Final Alpha Factor
    enhanced_final_alpha_factor = final_alpha_factor + rsi + stoch_signal
    
    return enhanced_final_alpha_factor
```
