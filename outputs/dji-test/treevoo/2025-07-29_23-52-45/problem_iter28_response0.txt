```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Determine Price Momentum
    lookback_period = 10
    df['close_pct_change'] = df['close'].pct_change()
    df['price_momentum'] = df['close_pct_change'].rolling(window=lookback_period).sum()
    
    # Integrate Volume and Price Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_price_momentum'] = df['volume_change'] * df['price_momentum']
    
    # Adjusted Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Adjusted Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Combined Adjusted Momentum Factor
    df['combined_momentum'] = df['long_term_return'] - df['short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_momentum']
    
    # 5-Day Moving Average of Close Price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    
    # 60-Day Moving Average of Close Price
    df['ma_60'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['ma_5'] - df['ma_60']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['reversal_indicator'] + df['trend_indicator']
    
    return df['alpha_factor'].dropna()
```
