```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(60)) - 1
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend Indicator
    momentum_trend_signal = pd.Series(0, index=df.index)
    momentum_trend_20_ma = momentum_factor.rolling(window=20).mean()
    momentum_trend_signal[momentum_factor > momentum_trend_20_ma] = 1  # Strong Uptrend
    momentum_trend_signal[momentum_factor < momentum_trend_20_ma] = -1  # Strong Downtrend
    momentum_trend_signal[(momentum_factor <= momentum_trend_20_ma) & (momentum_factor >= momentum_trend_20_ma)] = 0  # Sideways or Neutral
    
    # 12-Day Close Price Momentum
    close_price_momentum_12 = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    volume_sma_5 = df['volume'].rolling(window=5).mean()
    volume_deviation_from_sma_5 = df['volume'] - volume_sma_5
    
    # Open-Close Price Relationship
    open_close_ratio = df['close'] / df['open']
    
    # Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        0.3 * momentum_factor +
        0.2 * smoothed_momentum +
        0.1 * reversal_indicator +
        0.1 * close_price_momentum_12 +
        0.1 * volume_deviation_from_sma_5 +
        0.1 * open_close_ratio +
        0.1 * daily_volatility +
        0.1 * amount_to_volume_ratio
    )
    
    return alpha_factor
```
