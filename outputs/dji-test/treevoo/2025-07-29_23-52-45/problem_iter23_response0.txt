```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Returns by Intraday Volatility
    adjusted_volume_weighted_short_term = volume_weighted_short_term / intraday_volatility
    adjusted_volume_weighted_long_term = volume_weighted_long_term / intraday_volatility
    
    # Momentum Factor
    momentum_factor = adjusted_volume_weighted_short_term - adjusted_volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Daily Price Movement
    daily_price_movement = df['close'] - df['open']
    
    # Uptrend Count
    uptrend_count = (df['close'] > df['open']).rolling(window=10).sum()
    
    # Average True Range (ATR)
    tr = pd.DataFrame({
        'hl': df['high'] - df['low'],
        'hc': (df['high'] - df['close'].shift(1)).abs(),
        'lc': (df['low'] - df['close'].shift(1)).abs()
    }).max(axis=1)
    atr = tr.rolling(window=10).mean()
    
    # ATR Moving Average
    atr_ma = atr.rolling(window=20).mean()
    
    # ATR Signal
    atr_signal = (atr < atr_ma).astype(int)
    
    # Positive Volume Score
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    positive_volume_score = ((df['close'] > df['open']) & (df['volume'] > avg_volume_30)).rolling(window=30).sum()
    
    # Intraday Bullishness
    intraday_bullishness = (df['high'] - df['open']) / (df['open'] - df['low'])
    
    # Bullishness Ratio
    bullishness_ratio = intraday_bullishness.rolling(window=5).mean()
    
    # Combined Indicator
    combined_indicator = (momentum_factor + daily_price_movement + uptrend_count + atr_signal + positive_volume_score + bullishness_ratio) / 6
    combined_indicator = combined_indicator.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # 60-Day Moving Average of Close Price
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator + combined_indicator
    
    return final_alpha_factor
```
