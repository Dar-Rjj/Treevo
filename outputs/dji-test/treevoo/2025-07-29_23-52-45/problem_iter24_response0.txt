import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on price and volume data.
    
    :param df: DataFrame with columns (date, open, high, low, close, volume)
    :return: Series of alpha factor values indexed by date
    """
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Identify Price Reversal Patterns
    df['bullish_engulfing'] = (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))
    df['bearish_engulfing'] = (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))
    df['doji'] = (abs(df['open'] - df['close']) / (df['high'] - df['low']) < 0.05) & ((df['high'] - df['low']) > 0.02 * (df['close']))
    
    # Compute Average True Range
    df['tr'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Analyze Volume Data
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_spike'] = df['volume'] > 2 * df['volume'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Combine Price and Volume Analysis
    df['factor_1'] = df['intraday_volatility'] * df['volume_change']
    df['factor_2'] = (df['bullish_engulfing'] | df['bearish_engulfing']) * df['volume_change']
    df['factor_3'] = df['intraday_volatility'] * df['vwap']
    
    # Composite Alpha Factor
    df['composite_factor'] = df['factor_1'] + df['factor_2'] + df['factor_3']
    
    return df['composite_factor']
