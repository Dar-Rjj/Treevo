import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5) - 1)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(60) - 1)
    
    # Calculate 10-day Simple Moving Average (SMA) of closing prices
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Evaluate the difference between today's close and the 10-day SMA
    df['close_sma_diff'] = df['close'] - df['sma_10']
    
    # Daily Price Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute the ratio of today's volume to the average volume over the past 10 days
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Measure the cumulative volume over a window, say 20 days
    df['cumulative_volume'] = df['volume'].rolling(window=20).sum()
    
    # Volume-Weighted Short-Term Return
    df['vol_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['vol_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Short-Term Return
    df['adj_vol_weighted_short_term_return'] = df['vol_weighted_short_term_return'] / df['intraday_volatility']
    
    # Adjusted Volume-Weighted Long-Term Return
    df['adj_vol_weighted_long_term_return'] = df['vol_weighted_long_term_return'] / df['intraday_volatility']
    
    # Identify gaps by comparing today's open to yesterday's close
    df['gap'] = df['open'] - df['close'].shift(1)
    df['up_gap'] = df['gap'] > 0
    df['down_gap'] = df['gap'] < 0
    
    # Determine if there is a higher high and lower low pattern in the last 5 days
    df['higher_high'] = df['high'].rolling(window=5).max().diff() > 0
    df['lower_low'] = df['low'].rolling(window=5).min().diff() < 0
    
    # Check for a bullish or bearish engulfing pattern using open, high, low, and close prices
    df['bullish_engulfing'] = (df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & (df['open'] < df['close'].shift(1)) & (df['close'] > df['open'].shift(1))
    df['bearish_engulfing'] = (df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1)) & (df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1))
    
    # Calculate the percentage change from the opening to the closing price
    df['pct_change_open_close'] = (df['close'] - df['open']) / df['open']
    
    # Create a flag for up days (close > open)
    df['up_day'] = df['close'] > df['open']
    
    # Calculate 5-Day Moving Average of Close Price
    df['sma_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    df['sma_60'] = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    df['trend_indicator'] = df['sma_5'] - df['sma_60']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['adj_vol_weighted_short_term_return'] - df['adj_vol_weighted_long_term_return']
    
    # Smoothed Reversal Indicator
    df['smoothed_reversal_indicator'] = df['reversal_indicator'].rolling(window=3).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['trend_indicator'] + df
