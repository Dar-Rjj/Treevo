```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change using High and Low prices
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Rolling Average of Adjusted Price Change with a window of 10 days
    rolling_avg = volume_adjusted_price_change.rolling(window=10).mean()
    
    # Compute First Difference of the Rolling Average
    momentum_change = rolling_avg.diff()
    
    return momentum_change
```
