import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    sma_20 = df['close'].rolling(window=20).mean()
    momentum_signal = df['close'] - sma_20
    
    # Weight Momentum Signal by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    volume_adjusted_momentum = momentum_signal * volume_ratio
    
    # Adjust for Price Volatility
    std_20 = df['close'].rolling(window=20).std()
    final_momentum_signal = volume_adjusted_momentum / std_20
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_return = long_term_return * df['volume']
    
    # Combined Momentum Factor
    combined_momentum_factor = volume_weighted_short_return - volume_weighted_long_return + final_momentum_signal
    
    # Reversal Indicator
    reversal_indicator = -combined_momentum_factor
    
    # Trend Confirmation
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    trend_confirmation = (sma_50 > sma_200).astype(int) - (sma_50 < sma_200).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_momentum_factor * trend_confirmation
    
    return alpha_factor
