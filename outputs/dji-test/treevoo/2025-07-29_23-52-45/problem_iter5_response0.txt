```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Moving Average of High-Low Range
    df['sma_high_low_range'] = df['high_low_range'].rolling(window=10).mean()
    
    # Determine Momentum
    df['momentum_signal'] = (df['high_low_range'] > df['sma_high_low_range']).astype(int) * 2 - 1
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Intraday Price Difference
    df['intraday_price_diff'] = df['close'] - df['open']
    
    # Volume-Adjusted Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # High-Low Price Ratio
    df['high_low_price_ratio'] = df['high'] / df['low']
    
    # Combined Factor
    df['combined_factor'] = (0.4 * df['intraday_return'] + 
                             0.6 * df['volume_adjusted_intraday_momentum'] + 
                             0.1 * df['high_low_price_ratio'])
    
    return df['combined_factor']
```
