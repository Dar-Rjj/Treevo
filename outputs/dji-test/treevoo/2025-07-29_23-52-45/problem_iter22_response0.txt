```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Measure Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Returns by Intraday Volatility
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum_factor'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum_factor']
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['sma_20_close'] = df['close'].rolling(window=20).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    df['combined_alpha_component'] = (df['combined_adjusted_momentum_factor'] + 
                                      df['intraday_volatility'] + 
                                      (df['volume_weighted_short_term_return'] - df['volume_weighted_long_term_return']) + 
                                      (df['close'] - df['sma_20_close']) / df['close'])
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = (df['high'] - df['low']) * df['volume']
    
    # Smoothed Momentum Calculation
    df['rolling_avg_adj_price_change'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    df['smoothed_momentum'] = df['rolling_avg_adj_price_change'] - df['rolling_avg_adj_price_change'].shift(1)
    
    # Momentum Trend Indicator
    df['moving_avg_combined_alpha'] = df['combined_alpha_component'].rolling(window=20).mean()
    df['trend_signal'] = 0
    df.loc[df['combined_alpha_component'] > df['moving_avg_combined_alpha'], 'trend_signal'] = 1  # Strong Uptrend
    df.loc[df['combined_alpha_component'] < df['moving_avg_combined_alpha'], 'trend_signal'] = -1  # Strong Downtrend
    
    # 12-Day Close Price Momentum
    df['12_day_momentum'] = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    df['sma_5_volume'] = df['volume'].rolling(window=5).mean()
    df['5_day_volume_sma'] = df['volume'] - df['sma_5_volume']
    
    # Open-Close Price Relationship
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Daily Volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    df['amount_to_volume_ratio'] = df['amount'] / df['volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_alpha_component'] + 
                                 df['12_day_momentum'] + 
                                 df['5_day_volume_sma'] + 
                                 df['open_close_ratio'] + 
                                 df['daily_volatility'] + 
                                 df['amount_to_volume_ratio'])
    
    return df['final_alpha_factor']
```
