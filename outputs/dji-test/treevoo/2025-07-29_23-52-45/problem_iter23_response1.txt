```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using open, high, low, close, and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    
    # Calculate the price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the average true range over a 10-day window
    df['atr_10'] = df['price_range'].rolling(window=10).mean()
    
    # Calculate the relative strength (RS) as the ratio of the current close to the 30-day simple moving average (SMA)
    df['sma_30'] = df['close'].rolling(window=30).mean()
    df['rs'] = df['close'] / df['sma_30']
    
    # Calculate the on-balance volume (OBV)
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Calculate the OBV momentum as the difference between the current OBV and the 5-day SMA of OBV
    df['obv_sma_5'] = df['obv'].rolling(window=5).mean()
    df['obv_momentum'] = df['obv'] - df['obv_sma_5']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['rs'] + df['obv_momentum'] / df['atr_10']) / 2
    
    return df['alpha_factor']
```
