import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Detect Bullish/Bearish Engulfing Patterns
    df['bullish_engulfing'] = ((df['close'] > df['open'].shift(1)) & 
                               (df['open'] < df['close'].shift(1)))
    df['bearish_engulfing'] = ((df['close'] < df['open'].shift(1)) & 
                               (df['open'] > df['close'].shift(1)))
    
    # Detect Doji Patterns
    df['doji'] = (abs(df['open'] - df['close']) / df['close'] < 0.005) & \
                 (abs(df['high'] - df['low']) / df['close'] > 0.01)
    
    # Compute Changes in Trading Volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Examine Volume Spike Events
    k = 2.0  # Threshold for volume spike
    df['volume_spike'] = (df['volume'] > k * df['volume'].shift(1))
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Create Composite Alpha Factors
    df['factor_1'] = df['intraday_volatility'] * df['volume_change']
    df['factor_2'] = (df['bullish_engulfing'].astype(int) - df['bearish_engulfing'].astype(int)) * df['volume_change']
    df['factor_3'] = df['intraday_volatility'] * df['vwap']
    
    # Return the composite alpha factors
    return df[['factor_1', 'factor_2', 'factor_3']]
