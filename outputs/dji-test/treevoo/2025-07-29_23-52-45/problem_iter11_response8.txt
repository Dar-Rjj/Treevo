```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    df['volume_adjusted_price_range'] = df['price_range'] * df['volume']
    df['rolling_avg_volume_adjusted_price_range'] = df['volume_adjusted_price_range'].rolling(window=10).mean()
    
    # Momentum Calculation
    df['momentum_first_diff'] = df['rolling_avg_volume_adjusted_price_range'].diff()
    
    # Volume-Weighted Momentum
    lookback_period = 20
    df['sma_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_adjusted_momentum'] = df['momentum_first_diff'] * (df['volume'] / df['sma_volume'])
    
    # Return-Based Momentum with Volatility Adjustment
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Volatility-Adjusted Returns
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Enhanced Momentum Calculation
    df['smoothed_momentum'] = df['rolling_avg_volume_adjusted_price_range'].diff().rolling(window=10).mean()
    df['momentum_trend_indicator'] = df['smoothed_momentum'].rolling(window=20).mean()
    
    # Combine Volume-Adjusted Momentum and Reversal Indicator
    df['final_alpha_factor'] = df['volume_adjusted_momentum'] + df['reversal_indicator']
    
    return df['final_alpha_factor']
```
