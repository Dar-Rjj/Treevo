```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Separate Positive and Negative Changes
    df['positive_change'] = df['price_change'].apply(lambda x: max(x, 0))
    df['negative_change'] = df['price_change'].apply(lambda x: abs(min(x, 0)))
    
    # Calculate Average Positive and Negative Changes over 14 Days
    df['avg_positive_change'] = df['positive_change'].rolling(window=14).mean()
    df['avg_negative_change'] = df['negative_change'].rolling(window=14).mean()
    
    # Compute RSIVA
    df['rs'] = df['avg_positive_change'] / (df['avg_negative_change'] + 1e-6)  # Adding a small constant to avoid division by zero
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Adjust by Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['normalized_volume_change'] = df['volume_change'].apply(lambda x: 1 if x > 0 else -1)
    df['rsiva'] = df['rsi'] * df['normalized_volume_change']
    
    # Calculate Price Momentum
    lookback_period = 20
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_signal'] = df['close'] - df['sma_close']
    
    # Weight Momentum Signal by Volume
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['normalized_volume'] = df['volume'] / (df['avg_volume'] + 1e-6)  # Adding a small constant to avoid division by zero
    df['volume_adjusted_momentum'] = df['momentum_signal'] * df['normalized_volume']
    
    # Combine RSI and Momentum
    df['vamrsi'] = df['rsiva'] + df['volume_adjusted_momentum']
    
    return df['vamrsi'].dropna()
```
