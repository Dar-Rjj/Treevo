```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    df['volume_weighted_short_return'] = df['short_term_return'] * df['volume']
    
    # Volume-Weighted Long-Term Return
    df['volume_weighted_long_return'] = df['long_term_return'] * df['volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_weighted_short_return'] - df['volume_weighted_long_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['momentum_factor']
    
    # Calculate Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    df['smoothed_momentum_diff'] = df['smoothed_momentum'].diff()
    
    # Momentum Trend Indicator
    df['momentum_trend'] = df['momentum_factor'].rolling(window=20).mean()
    df['trend_signal'] = 0
    df.loc[df['momentum_factor'] > df['momentum_trend'], 'trend_signal'] = 1
    df.loc[df['momentum_factor'] < df['momentum_trend'], 'trend_signal'] = -1
    
    # 12-Day Close Price Momentum
    df['12_day_close_momentum'] = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    df['5_day_volume_sma'] = df['volume'].rolling(window=5).mean()
    df['volume_diff_from_sma'] = df['volume'] - df['5_day_volume_sma']
    
    # Open-Close Price Relationship
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Daily Volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    df['amount_to_volume_ratio'] = df['amount'] / df['volume']
    
    # Combine all factors into a single factor
    df['alpha_factor'] = (
        df['momentum_factor'] + 
        df['reversal_indicator'] + 
        df['smoothed_momentum_diff'] + 
        df['12_day_close_momentum'] + 
        df['volume_diff_from_sma'] + 
        df['open_close_ratio'] + 
        df['daily_volatility'] + 
        df['amount_to_volume_ratio']
    )
    
    return df['alpha_factor'].dropna()
```
