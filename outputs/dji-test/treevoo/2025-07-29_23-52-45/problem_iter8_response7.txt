```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    df['volume_adjusted_price_range'] = df['daily_price_range'] * df['volume']
    df['rolling_volume_adjusted_price_range'] = df['volume_adjusted_price_range'].rolling(window=10).mean()
    
    # Momentum Calculation
    df['momentum_diff'] = df['rolling_volume_adjusted_price_range'].diff()
    
    # Weight Momentum by Volume
    lookback_period = 20
    df['sma_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_adjusted_momentum'] = df['momentum_diff'] * (df['volume'] / df['sma_volume'])
    
    # Short-Term Return Calculation
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return Calculation
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Volatility-Adjusted Returns
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Combined Adjusted Momentum Factor
    df['combined_adjusted_momentum'] = df['adjusted_long_term_return'] - df['adjusted_short_term_return']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['combined_adjusted_momentum']
    
    # Integrated Momentum Indicator
    df['integrated_momentum_indicator'] = df['volume_adjusted_momentum'] + df['reversal_indicator']
    
    return df['integrated_momentum_indicator']
```
