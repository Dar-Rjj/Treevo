```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum
    lookback_period = 20  # Example lookback period
    df['price_momentum'] = df['close'].pct_change().rolling(window=lookback_period).sum()
    
    # Calculate Short-Term, Mid-Term, and Long-Term Returns
    df['short_term_return'] = (df['close'] / df['close'].shift(5) - 1)
    df['mid_term_return'] = (df['close'] / df['close'].shift(30) - 1)
    df['long_term_return'] = (df['close'] / df['close'].shift(60) - 1)
    
    # Volume-Weighted Short-Term, Mid-Term, and Long-Term Returns
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_mid_term_return'] = df['mid_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['intraday_volatility']
    df['adjusted_volume_weighted_short_term_return'] = df['volume_weighted_short_term_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_mid_term_return'] = df['volume_weighted_mid_term_return'] / df['intraday_volatility']
    df['adjusted_volume_weighted_long_term_return'] = df['volume_weighted_long_term_return'] / df['intraday_volatility']
    
    # Reversal Indicator
    df['reversal_indicator_short_term'] = -df['adjusted_volume_weighted_short_term_return']
    df['reversal_indicator_long_term'] = -df['adjusted_volume_weighted_long_term_return']
    
    # Calculate Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    smooth_window = 10  # Example window for smoothing
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=smooth_window).mean()
    df['smoothed_momentum_diff'] = df['smoothed_momentum'].diff()
    
    # Momentum Trend Indicator
    trend_window = 20  # Example window for trend
    df['ma_short_term_return'] = df['adjusted_volume_weighted_short_term_return'].rolling(window=trend_window).mean()
    df['momentum_trend_indicator'] = 0
    df.loc[df['adjusted_volume_weighted_short_term_return'] > df['ma_short_term_return'], 'momentum_trend_indicator'] = 1
    df.loc[df['adjusted_volume_weighted_short_term_return'] < df['ma_short_term_return'], 'momentum_trend_indicator'] = -1
    
    # Combine Adjusted Momentum Factors
    fixed_weight_price = 0.4
    fixed_weight_short_term = 0.2
    fixed_weight_mid_term = 0.2
    fixed_weight_long_term = 0.2
    df['final_alpha_factor'] = (
        fixed_weight_price * df['adjusted_price_momentum'] +
        fixed_weight_short_term * df['adjusted_volume_weighted_short_term_return'] +
        fixed_weight_mid_term * df['adjusted_volume_weighted_mid_term_return'] +
        fixed_weight_long_term * df['adjusted_volume_weighted_long_term_return']
    )
    
    return df['final_alpha_factor']
```
