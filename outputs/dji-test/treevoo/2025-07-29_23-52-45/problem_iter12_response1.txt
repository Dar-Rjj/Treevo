```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 12-day close price momentum
    df['close_momentum_12'] = df['close'] - df['close'].shift(12)
    df['close_momentum_pct_12'] = df['close_momentum_12'] / df['close'].shift(12)

    # Calculate the 5-day volume simple moving average (SMA)
    df['volume_sma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_diff_sma_5'] = df['volume'] - df['volume_sma_5']
    df['volume_pct_change_sma_5'] = df['volume_diff_sma_5'] / df['volume_sma_5']

    # Compute the ratio of close price to open price
    df['close_to_open_ratio'] = df['close'] / df['open']
    df['close_to_open_sma_5'] = df['close_to_open_ratio'].rolling(window=5).mean()
    df['close_to_open_diff_sma_5'] = df['close_to_open_ratio'] - df['close_to_open_sma_5']

    # Calculate the high-low range
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_sma_5'] = df['high_low_range'].rolling(window=5).mean()
    df['high_low_range_diff_sma_5'] = df['high_low_range'] - df['high_low_range_sma_5']

    # Compute the amount-to-volume ratio
    df['amount_to_volume_ratio'] = df['amount'] / df['volume']
    df['amount_to_volume_sma_5'] = df['amount_to_volume_ratio'].rolling(window=5).mean()
    df['amount_to_volume_diff_sma_5'] = df['amount_to_volume_ratio'] - df['amount_to_volume_sma_5']

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_momentum_pct_12'] +
        df['volume_pct_change_sma_5'] +
        df['close_to_open_diff_sma_5'] +
        df['high_low_range_diff_sma_5'] +
        df['amount_to_volume_diff_sma_5']
    )

    return df['alpha_factor']
```
