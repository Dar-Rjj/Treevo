import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    adjusted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term_return - adjusted_short_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Compute Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Calculate Exponential Moving Average (EMA) of Returns
    ema_returns = daily_returns.ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume
    volume_weighted_return = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Calculate EMA of Intraday Volatility
    ema_intraday_volatility = intraday_volatility.ewm(span=10, adjust=False).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    combined_factor = ema_returns + volume_weighted_return
    
    # Adjust Combined Factor by Intraday Volatility
    adjusted_combined_factor = combined_factor / ema_intraday_volatility
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    volume_adjusted_price_range = daily_price_range * df['volume']
    
    # Momentum Indicator
    rolling_median_volume_adjusted_price_range = volume_adjusted_price_range.rolling(window=10).median()
    momentum_indicator = rolling_median_volume_adjusted_price_range.diff()
    
    # Integrate Volume-Adjusted Price Range Momentum
    integrated_momentum = adjusted_combined_factor + momentum_indicator
    
    # Additional Feature: High-Frequency Trading (HFT) Indicator
    hft_activity = ((df['high'] > df['low']) & (df['open'] < df['close'])).astype(int).diff().abs().sum()
    hft_adjusted_factor = integrated_momentum / hft_activity
    
    # Final Alpha Factor
    final_alpha_factor = hft_adjusted_factor + reversal_indicator
    
    return final_alpha_factor
