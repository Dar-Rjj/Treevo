import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Volume Adjusted Price Range
    df['volume_adjusted_price_range'] = df['price_range'] * df['volume']
    
    # Compute Rolling Average of Volume Adjusted Price Range (10 days window)
    df['rolling_avg_volume_adjusted_price_range'] = df['volume_adjusted_price_range'].rolling(window=10).mean()
    
    # Calculate First Difference of Rolling Averages
    df['momentum_indicator'] = df['rolling_avg_volume_adjusted_price_range'].diff()
    
    # Return the momentum indicator as a pandas Series
    return df['momentum_indicator']
