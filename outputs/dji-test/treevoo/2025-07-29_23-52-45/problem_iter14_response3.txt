import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(60)) - 1
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjusted Volume-Weighted Short-Term Return
    adjusted_volume_weighted_short_term = volume_weighted_short_term / intraday_volatility
    
    # Adjusted Volume-Weighted Long-Term Return
    adjusted_volume_weighted_long_term = volume_weighted_long_term / intraday_volatility
    
    # Momentum Factor
    momentum_factor = adjusted_volume_weighted_short_term - adjusted_volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend Indicator
    momentum_trend = momentum_factor.rolling(window=20).mean()
    trend_signal = pd.Series(index=df.index, dtype=float)
    trend_signal[momentum_factor > momentum_trend] = 1  # Strong Uptrend Signal
    trend_signal[momentum_factor < momentum_trend] = -1  # Strong Downtrend Signal
    trend_signal[(momentum_factor >= momentum_trend) & (momentum_factor <= momentum_trend)] = 0  # Sideways or Neutral Signal
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change().rolling(window=20).sum()
    
    # Adjust Price Momentum by Intraday Volatility
    adjusted_price_momentum = price_momentum / intraday_volatility
    
    # Combine Adjusted Momentum Factors
    final_alpha_factor = (
        0.4 * adjusted_price_momentum + 
        0.3 * adjusted_volume_weighted_short_term + 
        0.3 * adjusted_volume_weighted_long_term
    )
    
    return final_alpha_factor
