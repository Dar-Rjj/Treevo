import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A Series of the calculated factor values indexed by date.
    """
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the average volume over a 10-day rolling window
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the standard deviation of the close price over a 5-day rolling window
    df['std_close_5d'] = df['close'].rolling(window=5).std()
    
    # Calculate the factor as a combination of the above metrics
    df['factor'] = (df['pct_change_close'] * df['high_low_ratio']) / (df['avg_volume_10d'] + df['std_close_5d'])
    
    # Return the factor as a Series
    return df['factor']
