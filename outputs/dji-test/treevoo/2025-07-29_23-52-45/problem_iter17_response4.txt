```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    adjusted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term_return - adjusted_short_term_return
    
    # Enhanced Reversal Indicator
    enhanced_reversal_indicator = -combined_adjusted_momentum.rolling(window=3).mean()
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Determine Price Momentum
    price_momentum = (df['close'] - df['open']) / df['open']
    
    # Combine Volume and Price Momentum
    combined_volume_price_momentum = volume_change * price_momentum
    
    # Adjust for Volatility
    true_range = np.maximum.reduce([df['high'] - df['low'],
                                    (df['high'] - df['close'].shift(1)).abs(),
                                    (df['low'] - df['close'].shift(1)).abs()])
    average_true_range = true_range.rolling(window=14).mean()
    final_adjusted_combined_metric = combined_volume_price_momentum / average_true_range
    
    # Calculate 5-Day Moving Average of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate 60-Day Moving Average of Close Price
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_adjusted_momentum +
        enhanced_reversal_indicator +
        intraday_volatility +
        (volume_weighted_short_term_return - volume_weighted_long_term_return) +
        final_adjusted_combined_metric +
        trend_indicator
    )
    
    return final_alpha_factor
```
