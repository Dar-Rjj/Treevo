import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = (df['high'] - df['low']) * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).sum()
    
    # Momentum Trend
    ema_momentum_factor = momentum_factor.ewm(span=20, adjust=False).mean()
    
    # Strong Uptrend, Downtrend, or Sideways Signal
    strong_uptrend = momentum_factor > ema_momentum_factor
    strong_downtrend = momentum_factor < ema_momentum_factor
    sideways_signal = ~strong_uptrend & ~strong_downtrend
    
    # Combine the signals into a single factor
    factor = pd.Series(index=df.index)
    factor[strong_uptrend] = 1
    factor[strong_downtrend] = -1
    factor[sideways_signal] = 0
    
    return factor
