```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    momentum_lookback = 20
    short_term_return_lookback = 5
    long_term_return_lookback = 60
    
    # Calculate Price Momentum
    sma_close = df['close'].rolling(window=momentum_lookback).mean()
    momentum_signal = df['close'] - sma_close
    
    # Weight Momentum Signal by Volume
    avg_volume = df['volume'].rolling(window=momentum_lookback).mean()
    normalized_volume = df['volume'] / avg_volume
    volume_adjusted_momentum = momentum_signal * normalized_volume
    
    # Calculate Short-Term Return
    close_t_minus_5 = df['close'].shift(short_term_return_lookback)
    short_term_return = (df['close'] - close_t_minus_5) / close_t_minus_5
    
    # Calculate Long-Term Return
    close_t_minus_60 = df['close'].shift(long_term_return_lookback)
    long_term_return = (df['close'] - close_t_minus_60) / close_t_minus_60
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_return = long_term_return * df['volume']
    
    # Combined Momentum Factor
    combined_momentum_factor = volume_weighted_short_return - volume_weighted_long_return + volume_adjusted_momentum
    
    # Reversal Indicator
    reversal_indicator = -combined_momentum_factor
    
    return reversal_indicator
```
