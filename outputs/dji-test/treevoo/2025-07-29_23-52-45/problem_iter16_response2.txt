```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term and Long-Term Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Momentum Factor
    momentum_factor = volume_weighted_short_term - volume_weighted_long_term
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    rolling_avg_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    smoothed_momentum = rolling_avg_volume_adjusted_price_change - rolling_avg_volume_adjusted_price_change.shift(10)
    
    # Momentum Trend
    momentum_trend_20_day_ma = momentum_factor.rolling(window=20).mean()
    uptrend_signal = (momentum_factor > momentum_trend_20_day_ma).astype(int)
    downtrend_signal = (momentum_factor < momentum_trend_20_day_ma).astype(int)
    neutral_signal = 1 - (uptrend_signal + downtrend_signal)
    
    # Final Alpha Factor
    alpha_factor = (momentum_factor * smoothed_momentum * 
                    (uptrend_signal - downtrend_signal) + neutral_signal * momentum_factor)
    
    return alpha_factor
```
