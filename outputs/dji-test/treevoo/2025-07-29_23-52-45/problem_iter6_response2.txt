```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_weighted_short_term_return = short_term_return * avg_volume_5_days
    
    # Volume-Weighted Long-Term Return
    avg_volume_60_days = df['volume'].rolling(window=60).mean()
    volume_weighted_long_term_return = long_term_return * avg_volume_60_days
    
    # Combined Momentum Factor
    combined_momentum_factor = volume_weighted_short_term_return - volume_weighted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_momentum_factor
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Volume-Adjusted Price Range
    volume_adjusted_price_range = price_range * df['volume']
    
    # Smoothed Adjusted Price Range
    smoothed_adjusted_price_range = volume_adjusted_price_range.rolling(window=10).sum()
    first_difference = smoothed_adjusted_price_range.diff()
    
    # Final Alpha Factor
    alpha_factor = first_difference + reversal_indicator
    
    return alpha_factor
```
