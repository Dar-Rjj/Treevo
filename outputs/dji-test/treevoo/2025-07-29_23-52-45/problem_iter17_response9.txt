```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    adjusted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum_factor = adjusted_short_term_return - adjusted_long_term_return
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum_factor
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    combined_alpha_component = (
        combined_adjusted_momentum_factor +
        intraday_volatility +
        (volume_weighted_short_term_return - volume_weighted_long_term_return) +
        (df['close'] - sma_20) / df['close']
    )
    
    # Calculate Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean().diff()
    
    # Momentum Trend Indicator
    ma_combined_alpha = combined_alpha_component.rolling(window=20).mean()
    momentum_trend_indicator = (combined_alpha_component > ma_combined_alpha).astype(int) - (combined_alpha_component < ma_combined_alpha).astype(int)
    
    # 12-Day Close Price Momentum
    close_momentum_12 = df['close'] - df['close'].shift(12)
    
    # 5-Day Volume Simple Moving Average
    sma_5_volume = df['volume'].rolling(window=5).mean()
    volume_sma_diff = df['volume'] - sma_5_volume
    
    # Open-Close Price Relationship
    open_close_ratio = df['close'] / df['open']
    
    # Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_alpha_component +
        close_momentum_12 +
        volume_sma_diff +
        open_close_ratio +
        daily_volatility +
        amount_to_volume_ratio
    )
    
    return final_alpha_factor
```
