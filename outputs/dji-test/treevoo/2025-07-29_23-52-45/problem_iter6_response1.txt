```python
import pandas as pd

def heuristics_v2(df, n=20):
    """
    Generate a novel alpha factor based on volume and price momentum, adjusted for historical volatility.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    n (int): Number of days to use for calculating historical volatility.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Determine Price Momentum
    df['daily_range'] = df['high'] - df['low']
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['price_momentum'] = df['daily_range'] * df['close_to_open_return']
    
    # Integrate Volume and Price Momentum
    df['integrated_metric'] = df['volume_change'] * df['price_momentum']
    
    # Adjust for Historical Volatility
    df['daily_return'] = df['close'].pct_change()
    df['historical_volatility'] = df['daily_return'].rolling(window=n).std()
    df['alpha_factor'] = df['integrated_metric'] / df['historical_volatility']
    
    return df['alpha_factor'].dropna()
```
