```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs()

    # Measure Intraday Return
    df['intraday_return'] = df['high'] / df['low']

    # Separate Positive and Negative Changes
    changes = df['close'].diff()
    positive_changes = changes.where(changes > 0, 0)
    negative_changes = changes.where(changes < 0, 0).abs()

    # Calculate Averages for Momentum Indicator
    avg_positive_changes = positive_changes.rolling(window=14).mean()
    avg_negative_changes = negative_changes.rolling(window=14).mean()

    # Calculate Relative Strength (RS)
    rs = avg_positive_changes / avg_negative_changes

    # Calculate RSI
    rsi = 100 - (100 / (1 + rs))

    # Adjust by Volume
    volume_change = df['volume'].diff()
    normalized_volume_change = volume_change.apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)

    # Adjust RSI by Normalized Volume
    rsi_adjusted = rsi * normalized_volume_change

    # Volume-Adjusted Intraday Momentum
    volume_adjusted_intraday_momentum = df['intraday_return'] * df['volume']

    # Combine Momentum, Intraday Volatility, and Volume
    ema_returns = df['daily_return'].ewm(span=14, adjust=False).mean()
    ema_intraday_volatility = df['intraday_volatility'].ewm(span=14, adjust=False).mean()
    volume_weighted_return = df['daily_return'] * df['volume']
    combined_factor = (ema_returns + volume_weighted_return) / ema_intraday_volatility

    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)

    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']

    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']

    # Adjust Short-Term Return by Intraday Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / df['intraday_volatility']

    # Adjust Long-Term Return by Intraday Volatility
    adjusted_long_term_return = volume_weighted_long_term_return / df['intraday_volatility']

    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum_factor = adjusted_long_term_return - adjusted_short_term_return

    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum_factor

    # Synthesize Factors
    final_alpha_factor = (rsi_adjusted * 0.4 +
                          df['intraday_return'] * 0.2 +
                          volume_adjusted_intraday_momentum * 0.3 +
                          (df['high'] / df['low']) * 0.1 +
                          combined_adjusted_momentum_factor * 0.1)

    return final_alpha_factor
```
