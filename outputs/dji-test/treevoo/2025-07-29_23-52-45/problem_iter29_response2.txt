```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Identify Price Reversal Patterns (Bullish/Bearish Engulfing)
    df['bullish_engulfing'] = (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))
    df['bearish_engulfing'] = (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))
    
    # Examine Price Patterns (Higher High and Lower Low in the last 5 days)
    df['higher_high'] = df['high'].rolling(window=5).max() > df['high'].shift(5)
    df['lower_low'] = df['low'].rolling(window=5).min() < df['low'].shift(5)
    
    # Compute Changes in Trading Volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Examine Volume Spike Events
    k = 2.0  # Threshold for volume spike
    df['volume_spike'] = df['volume'] > k * df['volume'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Investigate Volume-related factors
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['cumulative_volume_20'] = df['volume'].rolling(window=20).sum()
    
    # Create Composite Alpha Factors
    df['factor1'] = df['intraday_volatility'] * df['volume_change']
    df['factor2'] = (df['bullish_engulfing'] | df['bearish_engulfing']) * df['volume_change']
    
    # Explore Momentum indicators
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['momentum_factor'] = df['close'] - df['sma_10']
    
    # Analyze the relationship between Open and Close Prices
    df['percentage_change_open_close'] = (df['close'] - df['open']) / df['open']
    df['up_day'] = df['close'] > df['open']
    
    # Look into Range and Volatility
    df['daily_range'] = df['high'] - df['low']
    df['avg_range_10'] = df['daily_range'].rolling(window=10).mean()
    df['range_diff'] = df['daily_range'] - df['avg_range_10']
    
    # Consider the Impact of Gaps
    df['gap_up'] = df['open'] > df['close'].shift(1)
    df['gap_down'] = df['open'] < df['close'].shift(1)
    
    # Combine all alpha factors into a single composite factor
    df['composite_factor'] = (
        df['factor1'] + 
        df['factor2'] + 
        df['momentum_factor'] + 
        df['percentage_change_open_close'] + 
        df['range_diff']
    )
    
    return df['composite_factor']
```
