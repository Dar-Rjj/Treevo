```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    df['short_term_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    df['long_term_return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    df['volume_weighted_short_term'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term'] = df['long_term_return'] * df['volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_weighted_short_term'] - df['volume_weighted_long_term']
    
    # Reversal Indicator
    df['reversal_indicator'] = -df['momentum_factor']
    
    # Price Change
    df['price_change'] = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Smoothed Momentum Calculation
    df['smoothed_momentum'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    df['first_difference'] = df['smoothed_momentum'].diff()
    
    # Momentum Trend
    df['momentum_trend'] = df['momentum_factor'].rolling(window=20).mean()
    df['uptrend_signal'] = (df['momentum_factor'] > df['momentum_trend']).astype(int)
    df['downtrend_signal'] = (df['momentum_factor'] < df['momentum_trend']).astype(int)
    df['neutral_signal'] = 1 - (df['uptrend_signal'] + df['downtrend_signal'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_factor'] + df['first_difference'] + \
                         df['uptrend_signal'] - df['downtrend_signal']
    
    return df['alpha_factor'].dropna()
```
