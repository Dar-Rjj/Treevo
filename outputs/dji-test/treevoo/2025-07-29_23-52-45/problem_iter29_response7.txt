import pandas as pd

def heuristics_v2(df):
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Volume-Weighted Returns
    volume_weighted_short_term = short_term_return * df['volume']
    volume_weighted_long_term = long_term_return * df['volume']
    
    # Measure Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Adjust Returns by Intraday Volatility
    adjusted_short_term = volume_weighted_short_term / intraday_volatility
    adjusted_long_term = volume_weighted_long_term / intraday_volatility
    
    # Combined Adjusted Momentum Factor
    combined_adjusted_momentum = adjusted_long_term - adjusted_short_term
    
    # Reversal Indicator
    reversal_indicator = -combined_adjusted_momentum
    
    # Calculate 5-Day and 60-Day Moving Averages of Close Price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Trend Indicator
    trend_indicator = ma_5 - ma_60
    
    # Momentum Trend
    momentum_trend = combined_adjusted_momentum.rolling(window=20).mean()
    strong_uptrend = (combined_adjusted_momentum > momentum_trend).astype(int)
    strong_downtrend = (combined_adjusted_momentum < momentum_trend).astype(int)
    neutral_signal = 1 - (strong_uptrend + strong_downtrend)
    
    # Price Change
    price_change = df['high'] - df['low']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = price_change * df['volume']
    
    # Smoothed Momentum Calculation
    smoothed_momentum = volume_adjusted_price_change.rolling(window=10).mean()
    first_difference = smoothed_momentum - smoothed_momentum.shift(1)
    
    # Compute Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Exponential Moving Average (EMA) of Returns
    ema_returns = daily_returns.ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume
    volume_weighted_return = (daily_returns * df['volume']).cumsum()
    
    # EMA of Intraday Volatility
    ema_intraday_volatility = intraday_volatility.ewm(span=10, adjust=False).mean()
    
    # Combine Momentum, Intraday Volatility, and Volume
    combined_factor = ema_returns + volume_weighted_return
    adjusted_combined_factor = combined_factor / ema_intraday_volatility
    
    # Momentum Indicator
    rolling_median = volume_adjusted_price_change.rolling(window=10).median()
    first_difference_median = rolling_median - rolling_median.shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = reversal_indicator + trend_indicator + first_difference + first_difference_median + (df['close'] - df['open'])
    
    return final_alpha_factor
