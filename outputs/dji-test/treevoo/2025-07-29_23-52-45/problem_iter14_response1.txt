```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    
    # Price Momentum (Sum of Percentage Change in Close over 5 days)
    price_momentum = df['close'].pct_change(5).rolling(window=5).sum()
    
    # Volume Momentum (Sum of Absolute Difference in Volume over 5 days)
    volume_momentum = df['volume'].diff().abs().rolling(window=5).sum()
    
    # Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Adjusted Price Momentum
    adjusted_price_momentum = price_momentum / intraday_volatility
    
    # Adjusted Volume Momentum
    adjusted_volume_momentum = volume_momentum / intraday_volatility
    
    # Combined Adjusted Momentum
    combined_adjusted_momentum = adjusted_price_momentum + adjusted_volume_momentum
    
    # Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = df['volume'] * short_term_return
    
    # Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = df['volume'] * long_term_return
    
    # Adjusted Volume-Weighted Short-Term Return
    adjusted_volume_weighted_short_term_return = volume_weighted_short_term_return / intraday_volatility
    
    # Adjusted Volume-Weighted Long-Term Return
    adjusted_volume_weighted_long_term_return = volume_weighted_long_term_return / intraday_volatility
    
    # Combined Adjusted Weighted Returns
    combined_adjusted_weighted_returns = adjusted_volume_weighted_short_term_return + adjusted_volume_weighted_long_term_return
    
    # Momentum Factor
    momentum_factor = combined_adjusted_momentum - combined_adjusted_weighted_returns
    
    # Reversal Indicator
    reversal_indicator = -momentum_factor
    
    return reversal_indicator
```
