```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Identify reversals with close price
    df['Reversal_Signal'] = (df['close'] > df['high'].shift(1)) | (df['close'] < df['low'].shift(1))
    
    # Compute short-term and long-term returns
    df['Short_Term_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['Long_Term_Return'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Analyze changes in volume
    df['Volume_Change'] = df['volume'].diff()
    
    # Detect volume surges
    df['Volume_Surge'] = df['volume'] > df['volume'].rolling(window=10).mean() * 2
    
    # Volume-Weighted Returns
    df['Volume_Weighted_Short_Term'] = df['Short_Term_Return'] * df['volume']
    df['Volume_Weighted_Long_Term'] = df['Long_Term_Return'] * df['volume']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjusted Volume-Weighted Returns by Intraday Volatility
    df['Adjusted_Volume_Weighted_Short_Term'] = df['Volume_Weighted_Short_Term'] / df['Intraday_Volatility']
    df['Adjusted_Volume_Weighted_Long_Term'] = df['Volume_Weighted_Long_Term'] / df['Intraday_Volatility']
    
    # Momentum and Reversal Indicators
    df['Momentum_Factor'] = df['Adjusted_Volume_Weighted_Short_Term'] - df['Adjusted_Volume_Weighted_Long_Term']
    df['Reversal_Indicator'] = -df['Momentum_Factor']
    
    # Additional Price and Volume Factors
    df['Daily_Price_Movement'] = df['close'] - df['open']
    df['Uptrend_Count'] = (df['close'] > df['open']).rolling(window=10).sum()
    
    # Average True Range (ATR)
    df['True_Range'] = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=10).mean()
    df['ATR_MA'] = df['ATR'].rolling(window=20).mean()
    df['ATR_Signal'] = df['ATR'] < df['ATR_MA']
    
    # Trading Volume Relative to Price
    df['Positive_Volume_Score'] = ((df['close'] > df['open']) & (df['volume'] > df['volume'].rolling(window=30).mean())).rolling(window=30).sum()
    
    # Intraday Bullishness
    df['Intraday_Bullishness'] = (df['high'] - df['open']).div(df['open'] - df['low'])
    df['Bullishness_Ratio'] = df['Intraday_Bullishness'].rolling(window=5).mean()
    
    # Combined Indicator
    df['Combined_Indicator'] = (df['Momentum_Factor'] + df['Uptrend_Count'] + df['ATR_Signal'].astype(int) + df['Positive_Volume_Score'] + df['Bullishness_Ratio']) / 5
    
    # Moving Averages for Trend Analysis
    df['MA_5'] = df['close'].rolling(window=5).mean()
    df['MA_60'] = df['close'].rolling(window=60).mean()
    df['Trend_Indicator'] = df['MA_5'] - df['MA_60']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Reversal_Indicator'] + df['Trend_Indicator'] + df['Combined_Indicator']
    
    return df['Final_Alpha_Factor']
```
