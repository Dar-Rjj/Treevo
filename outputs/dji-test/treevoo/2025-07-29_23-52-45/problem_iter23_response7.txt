```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return based on close prices
    df['daily_return'] = df['close'].pct_change()
    
    # Identify reversals with close price
    df['reversal'] = 0
    df.loc[df['close'] > df['high'].shift(1), 'reversal'] = 1
    df.loc[df['close'] < df['low'].shift(1), 'reversal'] = -1
    
    # Analyze changes in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Detect volume surges
    n_days = 10
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    df['volume_surge'] = (df['volume'] / df['avg_volume']) - 1
    
    # Combine price and volume for enhanced signal
    df['price_volume_indicator'] = df['daily_return'] * df['volume_surge']
    
    # Return the factor values
    return df['price_volume_indicator']
```
