```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Determine High and Low Price Momentum
    daily_range = df['high'] - df['low']
    close_to_open_return = (df['close'] - df['open']) / df['open']
    price_momentum = daily_range * close_to_open_return
    
    # Compute Volume Impact on Price Movement
    volume_impact = volume_change * price_momentum
    
    # Generate Volatility Adjusted Trend Score
    true_range = pd.concat([
        df['high'] - df['low'],
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    average_true_range = true_range.rolling(window=n).mean()
    
    volatility_adjusted_trend_score = volume_impact / average_true_range
    
    return volatility_adjusted_trend_score
```
