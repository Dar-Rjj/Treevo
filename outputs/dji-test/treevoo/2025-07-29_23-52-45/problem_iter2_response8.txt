import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    momentum_lookback = 20
    short_term_days = 5
    long_term_days = 60

    # Calculate Simple Moving Average of Close Price
    df['SMA_Close'] = df['close'].rolling(window=momentum_lookback).mean()

    # Calculate Momentum Signal
    df['Momentum_signal'] = df['close'] - df['SMA_Close']

    # Calculate Average Volume
    df['AVG_Volume'] = df['volume'].rolling(window=momentum_lookback).mean()

    # Normalize Volume
    df['Normalized_Volume'] = df['volume'] / df['AVG_Volume']

    # Calculate Volume-Adjusted Momentum
    df['Volume_adjusted_momentum'] = df['Momentum_signal'] * df['Normalized_Volume']

    # Calculate Short-Term Return
    df['Short_Term_Return'] = (df['close'] - df['close'].shift(short_term_days)) / df['close'].shift(short_term_days)

    # Calculate Long-Term Return
    df['Long_Term_Return'] = (df['close'] - df['close'].shift(long_term_days)) / df['close'].shift(long_term_days)

    # Calculate Volume-Weighted Short-Term Return
    df['Volume_Weighted_Short_Term_Return'] = df['Short_Term_Return'] * df['volume']

    # Calculate Volume-Weighted Long-Term Return
    df['Volume_Weighted_Long_Term_Return'] = df['Long_Term_Return'] * df['volume']

    # Calculate Combined Momentum Factor
    df['Combined_Momentum_Factor'] = (df['Volume_Weighted_Short_Term_Return'] 
                                      - df['Volume_Weighted_Long_Term_Return']
                                      + df['Volume_adjusted_momentum'])

    # Calculate Reversal Indicator
    df['Reversal_Indicator'] = -df['Combined_Momentum_Factor']

    # Return the Reversal Indicator as a pandas Series
    return df['Reversal_Indicator']
