import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday high-low range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the intraday open-close range
    df['oc_range'] = abs(df['open'] - df['close'])
    
    # Calculate the relative strength of the close price
    df['relative_strength'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the volume-weighted average price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the 10-day moving average of the VWAP
    df['VWAP_10d_MA'] = df['VWAP'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of the daily return
    df['daily_return_10d_std'] = df['daily_return'].rolling(window=10).std()
    
    # Heuristic factor: (Relative Strength * VWAP 10-day MA) / Intraday Range
    df['heuristic_factor'] = (df['relative_strength'] * df['VWAP_10d_MA']) / df['intraday_range']
    
    # Normalize the heuristic factor by its 10-day rolling standard deviation
    df['normalized_heuristic_factor'] = df['heuristic_factor'] / df['heuristic_factor'].rolling(window=10).std()
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['normalized_heuristic_factor']
