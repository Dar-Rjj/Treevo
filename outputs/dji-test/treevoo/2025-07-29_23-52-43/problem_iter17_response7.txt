```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    daily_high_low_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=5).max())
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    previous_day_high_low_spread = (df['high'].shift(1) - df['low'].shift(1)).fillna(0)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    combined_indicator = daily_high_low_range * volume_weighted_momentum
    
    # Calculate Momentum Difference
    momentum_difference = intraday_high_low_spread - previous_day_high_low_spread
    combined_indicator *= momentum_difference
    
    # Adjust for Price Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    long_term_trend = df['close'].rolling(window=20).mean()
    trend_difference = long_term_trend - short_term_trend
    combined_indicator += trend_difference
    
    # Adjust for Volatility
    rolling_volatility = intraday_high_low_spread.rolling(window=20).std()
    combined_indicator /= rolling_volatility
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1).fillna(1)
    spike_threshold = 2.0 * df['volume'].pct_change().rolling(window=21).quantile(0.90)
    spike_condition = (daily_volume_change > spike_threshold).astype(int)
    combined_indicator *= spike_condition
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * close_price_momentum) / price_range
    
    # Moving Average of High-Low Differences (20-day period)
    moving_avg_high_low_diff = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Final Composite Factor
    final_composite_factor = intermediate_factor + moving_avg_high_low_diff
    
    return final_composite_factor
```
