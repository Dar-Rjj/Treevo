```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Volume-Weighted High-Low Breakout Intensity
    volume_weighted_breakout = breakout_intensity * df['volume']
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Identify Volume Spikes
    volume_change = df['volume'].pct_change()
    spike_threshold = volume_change.rolling(window=21).quantile(0.75).shift() * 2.0
    spike_condition = volume_change > spike_threshold
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Moving Sum of High-Low Differences
    moving_sum_high_low = (df['high'] - df['low']).rolling(window=10).sum()
    
    # Volume-Adjusted Momentum (VAM)
    sma_close = df['close'].rolling(window=20).mean()
    momentum = sma_close.pct_change(periods=5)
    vam = momentum * df['volume']
    
    # Price Volatility Indicator (PVI)
    true_range = np.maximum.reduce([df['high'] - df['low'], 
                                    abs(df['high'] - df['close'].shift()), 
                                    abs(df['low'] - df['close'].shift())])
    atr = true_range.rolling(window=14).mean()
    pvi = true_range / atr
    smoothed_pvi = pvi.ewm(span=7).mean()
    
    # Logarithmic Return
    log_return = np.log(df['close'] / df['close'].shift())
    
    # Volume-Weighted Average True Range (VWATR)
    vwatr = atr * df['volume']
    
    # Combine Factors
    combined_factor = (
        (volume_weighted_breakout * vam) / smoothed_pvi +
        volume_weighted_intraday_return + 
        moving_sum_high_low + 
        log_return + 
        vwatr
    )
    
    return combined_factor
```
