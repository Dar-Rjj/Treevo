```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    df['combined_metrics_weighted_by_volume'] = combined_metrics * df['volume']
    
    # Adjust for Volatility
    df['daily_range'] = df['high'] - df['low']
    df['rolling_volatility'] = df['daily_range'].rolling(window=21).std()
    df['combined_metrics_adjusted'] = df['combined_metrics_weighted_by_volume'] / df['rolling_volatility']
    
    # Calculate Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['daily_high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_difference'] = df['vwap'] - df['prev_day_vwap']
    df['final_combined_indicator'] = df['momentum_difference'] * df['vwap_difference']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.8) * 1.6
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_combined_metrics'] = df['combined_metrics_adjusted'] * df['spike_condition']
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['filtered_combined_metrics'] + df['final_combined_indicator']
    
    return df['crossover_alpha_factor']
```
