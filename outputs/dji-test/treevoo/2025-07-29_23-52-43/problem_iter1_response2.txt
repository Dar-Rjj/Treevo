```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume'] / df['volume'].max())
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    factor = intraday_range * volume_weighted_momentum
    
    return factor
```
