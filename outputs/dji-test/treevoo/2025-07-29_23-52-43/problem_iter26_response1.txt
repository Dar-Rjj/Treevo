```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean().shift(1)
    avg_volume = df['volume'].rolling(window=6).mean().shift(2)
    volume_weighted_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)

    # Price Volatility
    price_volatility = df['close'].rolling(window=7).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return_high_low = (df['high'] - df['low']) / df['low']
    intraday_return_close_open = (df['close'] - df['open']) / df['open']

    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = (daily_volume_change.rolling(window=20).quantile(0.8) * 1.7).shift(1)
    volume_spike = (daily_volume_change > spike_threshold).astype(int)

    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)

    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    breakout_intensity = np.maximum(breakout_intensity, 0)

    # Moving Average of High-Low Differences (10-day period)
    high_low_diff_ma = (df['high'] - df['low']).rolling(window=10).mean()

    # Intermediate Factor
    intermediate_factor = volume_increase_ratio * (df['close'] - df['close'].rolling(window=6).mean().shift(1)) * (df['close'].rolling(window=7).std() / df['close'].shift(1))

    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return_high_low + intraday_return_close_open + breakout_intensity

    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility

    # Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Final Composite Factor
    final_composite_factor = (
        intermediate_factor +
        high_low_diff_ma +
        volume_weighted_momentum +
        price_volatility +
        adjusted_combined_metrics +
        short_term_momentum +
        long_term_momentum
    )

    # Enhance Weighting with Trade Activity
    trade_activity = df['volume'] / df['volume'].rolling(window=10).mean().shift(1)
    final_composite_factor = final_composite_factor * trade_activity

    return final_composite_factor
```
