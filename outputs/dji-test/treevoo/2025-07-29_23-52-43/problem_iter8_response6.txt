```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['previous_close'] = df['close'].shift(1)
    close_to_previous_close_return = (df['close'] - df['previous_close']) / df['previous_close']
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Calculate Daily High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    moving_sum_high_low_diff = high_low_diff.rolling(window=10).sum()
    
    # Divide Moving Sum by Close Price
    moving_sum_high_low_close_ratio = moving_sum_high_low_diff / df['close']
    
    # Identify Volume Spikes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.75) * 1.5
    volume_spike = daily_volume_change > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Filter with Spike Condition
    volume_weighted_intraday_return_filtered = volume_weighted_intraday_return.where(volume_spike, 0)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    volume_weighted_combined_metrics = combined_metrics * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = high_low_diff.rolling(window=20).std()
    adjusted_combined_metrics = volume_weighted_combined_metrics / rolling_volatility
    
    # Final Combination
    crossover_alpha_factor = (adjusted_combined_metrics + 
                              moving_sum_high_low_close_ratio + 
                              volume_weighted_intraday_return_filtered)
    
    return crossover_alpha_factor
```
