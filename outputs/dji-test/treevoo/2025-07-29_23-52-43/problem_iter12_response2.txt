```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10 days)
    df['high_low_diff_sum_10'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Compute Intraday Volatility
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high_low_diff'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Calculate Intraday Volatility Ratio
    df['intraday_vol_ratio'] = df['intraday_high_low_spread'] / df['prev_day_high_low_spread']
    
    # Calculate VWAP Ratio
    df['vwap_ratio'] = df['daily_vwap'] / df['prev_day_vwap']
    
    # Combine Intraday Volatility and VWAP Ratios
    df['combined_ratios'] = df['intraday_vol_ratio'] * df['vwap_ratio']
    
    # Introduce Open-Close Price Movement
    df['open_close_movement'] = df['open'] - df['close']
    
    # Calculate Moving Sum of Open-Close Movements (5 days)
    df['open_close_sum_5'] = df['open_close_movement'].rolling(window=5).sum()
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['high_low_diff_sum_10'] + df['open_close_sum_5'] + df['combined_ratios']
    
    return df['alpha_factor']
```
