import numpy as np

    df['High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['open']

    max_volume = df['volume'].rolling(window=20, min_periods=1).max()
    df['Volume_Weighted_Intraday_Momentum'] = (df['close'] - df['open']) * (df['volume'] / max_volume)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)

    df['Combined_Metric'] = df['High_Low_Range_Percentage'] * df['Volume_Weighted_Intraday_Momentum']

    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    
    # Calculate VWAP Difference
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    
    # Final Combined Indicator
    df['Final_Combined_Indicator'] = df['Momentum_Difference'] * df['VWAP_Difference']

    df['Short_Term_Price_Trend'] = df['close'].rolling(window=5, min_periods=1).mean()
