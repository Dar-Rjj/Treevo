```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_change_75th_percentile'] = df['volume_change'].rolling(window=21).quantile(0.75)
    multiplier = 1.5
    df['spike_threshold'] = df['volume_change_75th_percentile'] * multiplier
    
    # Filter with Spike Condition
    df['is_spike'] = df['volume_change'] > df['spike_threshold']
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(1).rolling(window=5).mean()
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Combine Price and Volume Factors
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_momentum'] / df['price_range'])
    
    # Calculate Moving Sum of High-Low Differences
    df['high_low_diff'] = df['high'] - df['low']
    df['moving_sum_high_low_diff'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Final Composite Factor
    df['crossover_momentum_volume_composite'] = df['intermediate_factor'] * df['moving_sum_high_low_diff']
    
    return df['crossover_momentum_volume_composite'].dropna()
```
