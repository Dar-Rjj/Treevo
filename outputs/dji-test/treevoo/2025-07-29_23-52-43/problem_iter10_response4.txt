```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Volume-Weighted Breakout Intensity
    volume_weighted_breakout = breakout_intensity * df['volume']
    
    # Compute Simple Moving Average (SMA) of Close Prices with a lookback period of 20 days
    sma_close = df['close'].rolling(window=20).mean()
    
    # Compute Momentum using SMA of Close Prices with a momentum period of 5 days
    momentum = sma_close.pct_change(periods=5)
    
    # Volume-Adjusted Momentum (VAM)
    vam = momentum * df['volume']
    
    # Compute True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close']).abs()
    tr3 = (df['low'] - df['close']).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Compute ATR (Average True Range) with a lookback period of 14 days
    atr = true_range.rolling(window=14).mean()
    
    # Smooth PVI using EMA with a period of 7 days
    pvi = atr.ewm(span=7).mean()
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    close_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Combine Factors
    combined_factors = (volume_weighted_breakout * vam) / pvi
    combined_factors = combined_factors * volume_increase_ratio * close_momentum / price_range
    
    # Subtract mean of the final adjusted value over a period of 30 days
    final_factor = combined_factors - combined_factors.rolling(window=30).mean()
    
    return final_factor
```
