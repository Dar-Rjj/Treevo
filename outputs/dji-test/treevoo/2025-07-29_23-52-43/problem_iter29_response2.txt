```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = (intraday_return + close_to_previous_close_return + breakout_intensity) * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = df['high'] - df['low']
    rolling_volatility = rolling_volatility.rolling(window=20).std()
    adjusted_metrics = combined_metrics / rolling_volatility
    
    # Incorporate Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    momentum_adjusted_metrics = adjusted_metrics + short_term_momentum + long_term_momentum
    
    # Enhance Weighting with Trade Activity
    average_volume = df['volume'].rolling(window=10).mean()
    trade_activity = df['volume'] / average_volume
    final_combined_indicator = momentum_adjusted_metrics * trade_activity
    
    return final_combined_indicator
```
