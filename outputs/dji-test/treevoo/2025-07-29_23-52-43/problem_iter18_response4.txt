```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Metrics
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    weighted_combined_metrics = df['volume'] * combined_metrics
    
    # Adjust for Momentum
    rolling_momentum = (df['close'] - df['close'].shift(1)).rolling(window=20).sum()
    final_alpha_factor = weighted_combined_metrics / rolling_momentum
    
    return final_alpha_factor
```
