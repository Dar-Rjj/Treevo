```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    moving_sum_high_low_diff = high_low_diff.rolling(window=10).sum()
    
    # Compute Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    previous_day_vwap = daily_vwap.shift(1)
    
    # Combine Intraday Momentum and VWAP
    vwap_difference = daily_vwap - previous_day_vwap
    combined_indicator = intraday_momentum * vwap_difference
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=22).quantile(0.75).shift(1) * 1.5
    is_spike = daily_volume_change > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Filter with Spike Condition
    filtered_volume_weighted_intraday_return = volume_weighted_intraday_return * is_spike
    
    # Synthesize Intermediate Alpha Factor
    intermediate_alpha_factor = moving_sum_high_low_diff + combined_indicator
    
    # Generate Crossover High-Low Difference Momentum and Volume-Weighted Intraday Return
    crossover_alpha = intermediate_alpha_factor + filtered_volume_weighted_intraday_return
    
    return crossover_alpha
```
