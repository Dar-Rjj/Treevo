import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as open, high, low, close, volume.
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate the relative position of the closing price within the day's range
    df['range'] = df['high'] - df['low']
    df['relative_close'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the 10-day exponential moving average of the relative close
    ema_window = 10
    df['ema_relative_close'] = df['relative_close'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate the 20-day simple moving average of the volume
    sma_volume_window = 20
    df['sma_volume'] = df['volume'].rolling(window=sma_volume_window).mean()
    
    # Calculate the factor as the ratio of the EMA of the relative close to the SMA of the volume
    df['factor'] = df['ema_relative_close'] / (df['sma_volume'] + 1e-6)  # Add small constant to avoid division by zero
    
    return df['factor']
