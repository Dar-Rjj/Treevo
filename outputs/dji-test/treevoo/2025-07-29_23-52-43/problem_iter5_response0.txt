```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    df['high_low_diff_sum_10d'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Divide by Close Price
    df['high_low_diff_sum_normalized'] = df['high_low_diff_sum_10d'] / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Determine Spike Threshold
    spike_threshold = 1.5 * df['volume_change'].rolling(window=21).quantile(0.75)
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Combine Factors
    df['alpha_factor'] = np.where(df['spike_condition'], 
                                   df['high_low_diff_sum_normalized'] * df['volume_weighted_intraday_return'], 
                                   0)
    
    return df['alpha_factor']
```
