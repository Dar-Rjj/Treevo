```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    
    # Volume-Weighted Breakout Intensity
    df['vwb_breakout_intensity'] = df['breakout_intensity'] * df['volume']

    # Volume-Adjusted Momentum (VAM)
    lookback_period = 20
    momentum_period = 5
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum'] = df['sma_close'].diff(momentum_period)
    avg_volume = df['volume'].rolling(window=momentum_period).mean()
    df['vam'] = df['momentum'] * df['volume'] / avg_volume

    # Price Volatility Indicator (PVI)
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(x[1] - df['close'].shift(1))), axis=1)
    atr_lookback_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_lookback_period).mean()
    df['pvi_smoothed'] = df['atr'].ewm(span=7).mean()

    # Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1

    # Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.8) * 1.5
    df['volume_spike'] = (df['volume_change'] > spike_threshold).astype(int)

    # Volume-Weighted Intraday Return
    df['vwi_intraday_return'] = df['intraday_return'] * df['volume']

    # Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']

    # Moving Sum of High-Low Differences
    df['sum_high_low_diff'] = df['daily_high_low_diff'].rolling(window=10).sum()

    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)

    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean()

    # Price Range
    df['price_range'] = df['high'] - df['low']

    # Combine Factors
    df['combined_factor'] = (df['vwb_breakout_intensity'] * df['vam'] 
                             / df['pvi_smoothed'] * df['vwi_intraday_return'] 
                             / df['sum_high_low_diff'] * df['volume_increase_ratio'] 
                             * df['close_momentum'] / df['price_range'])

    # Subtract mean of the final adjusted value over a period (e.g., 30 days)
    adjustment_period = 30
    df['final_factor'] = df['combined_factor'] - df['combined_factor'].rolling(window=adjustment_period).mean()

    return df['final_factor']
```
