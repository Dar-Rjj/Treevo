```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    # 1-Week Close Price Change
    df['1_week_close_change'] = df['close'] - df['close'].shift(5)
    
    # 1-Month Close Price Change
    df['1_month_close_change'] = df['close'] - df['close'].shift(20)

    # Calculate Volume Volatility
    # 1-Week Volume Standard Deviation
    df['1_week_volume_std'] = df['volume'].rolling(window=6).std()
    
    # 1-Month Volume Standard Deviation
    df['1_month_volume_std'] = df['volume'].rolling(window=21).std()

    # Combine Momentum and Volatility
    # Multiply 1-Week Close Price Change by 1-Week Volume Standard Deviation
    df['1_week_momentum_volatility'] = df['1_week_close_change'] * df['1_week_volume_std']
    
    # Multiply 1-Month Close Price Change by 1-Month Volume Standard Deviation
    df['1_month_momentum_volatility'] = df['1_month_close_change'] * df['1_month_volume_std']

    # Combine the two factors to create the final alpha factor
    df['alpha_factor'] = (df['1_week_momentum_volatility'] + df['1_month_momentum_volatility']) / 2

    return df['alpha_factor']
```
