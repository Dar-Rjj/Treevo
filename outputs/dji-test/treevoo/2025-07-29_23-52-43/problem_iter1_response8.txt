```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Increase Ratio
    df['Volume_t-1'] = df['volume'].shift(1)
    df['Volume_Increase_Ratio'] = df['volume'] / df['Volume_t-1']

    # Calculate Close Price Momentum
    df['Avg_Close_t-5_to_t-1'] = df['close'].rolling(window=5).mean().shift(1)
    df['Close_Price_Momentum'] = df['close'] - df['Avg_Close_t-5_to_t-1']

    # Adjust for Price Volatility
    df['Price_Range'] = df['high'] - df['low']

    # Combine Factors
    df['Composite_Factor'] = (df['Volume_Increase_Ratio'] * df['Close_Price_Momentum']) / df['Price_Range']

    # Drop intermediate columns
    df.drop(['Volume_t-1', 'Avg_Close_t-5_to_t-1'], axis=1, inplace=True)

    return df['Composite_Factor']
```
