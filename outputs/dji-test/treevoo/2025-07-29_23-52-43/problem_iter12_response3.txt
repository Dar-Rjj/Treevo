import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Volume Spike Detection
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['daily_volume_change'].rolling(window=21).quantile(0.75) * 1.5
    df['volume_spike'] = (df['daily_volume_change'] > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(1).rolling(window=5).mean()
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_price_momentum'] / df['price_range'])
    
    # Moving Average of High-Low Differences (10-day period)
    df['ma_high_low_diff'] = df['price_range'].rolling(window=10).mean()
    
    # Final Composite Factor
    df['final_composite_factor'] = df['intermediate_factor'] * df['ma_high_low_diff']
    
    return df['final_composite_factor'].dropna()
