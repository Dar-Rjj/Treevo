```python
import pandas as pd

def heuristics_v2(df):
    """
    Computes a composite factor based on volume and price momentum, price volatility, and price range.
    
    :param df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index (date)
    :return: pandas Series with the factor values indexed by (date)
    """
    # Volume Weighted Close Price Momentum
    avg_close_5 = df['close'].rolling(window=5).mean().shift(1)
    close_momentum = df['close'] - avg_close_5
    
    avg_volume_5 = df['volume'].rolling(window=5).mean().shift(1)
    volume_ratio = df['volume'] / avg_volume_5
    
    volume_weighted_momentum = close_momentum * volume_ratio
    
    # Price Volatility
    std_dev_close = df['close'].rolling(window=5).std()
    price_volatility = std_dev_close / df['close'].shift(1)
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Combine Factors
    final_factor = (volume_weighted_momentum * price_volatility) / price_range
    
    return final_factor
```
