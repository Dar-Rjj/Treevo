import pandas as pd

def heuristics_v2(df):
    # Volume Weighted Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    avg_volume = df['volume'].rolling(window=5).mean().shift(1)
    volume_weighted_intraday_return = intraday_return * (df['volume'] / avg_volume)

    # Volume Spike Detection
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.9) * 2.0
    volume_spike = daily_volume_change > spike_threshold

    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)

    # Close Price Momentum
    close_momentum = df['close'] - df['close'].rolling(window=5).mean()

    # Price Volatility
    price_volatility = df['close'].rolling(window=5).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Close-to-PreviousClose Return
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))

    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * close_momentum * price_volatility)

    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_returns = intraday_return + close_to_previous_close_return + breakout_intensity

    # Final Composite Factor
    final_factor = intermediate_factor + combined_returns

    return final_factor
