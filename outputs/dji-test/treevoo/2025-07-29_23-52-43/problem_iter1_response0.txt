```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Ensure Breakout Intensity is Positive
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Weight by Volume
    volume_weighted_breakout_intensity = breakout_intensity * df['volume']
    
    return volume_weighted_breakout_intensity
```
