import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values, indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day Exponential Moving Average (EMA) of the money flow
    ema_money_flow = money_flow.ewm(span=14).mean()
    
    # Calculate the 14-day Exponential Moving Average (EMA) of the volume
    ema_volume = df['volume'].ewm(span=14).mean()
    
    # Calculate the Money Flow Index (MFI)
    mfi = 100 - (100 / (1 + (ema_money_flow / ema_volume)))
    
    # Calculate the 10-day standard deviation of the typical price
    std_typical_price = typical_price.rolling(window=10).std()
    
    # Calculate the final alpha factor
    alpha_factor = mfi / std_typical_price
    
    return alpha_factor
