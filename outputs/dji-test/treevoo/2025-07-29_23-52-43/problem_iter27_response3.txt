```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Volume Weighted Intraday Return
    avg_volume_5 = df['volume'].rolling(window=5).mean().shift(1)
    df['volume_weighted_intraday_return'] = ((df['close'] - df['open']) / df['open']) * (df['volume'] / avg_volume_5)
    
    # Volume Spike Detection
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['daily_volume_change'].rolling(window=21).quantile(0.9).shift(1) * 2.0
    df['volume_spike'] = df['daily_volume_change'] > spike_threshold
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Price Volatility
    df['price_volatility'] = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Enhanced Intermediate Factor
    avg_close_5 = df['close'].rolling(window=5).mean().shift(1)
    avg_volume_t_5 = df['volume'].rolling(window=5).mean()
    avg_volume_t_6_10 = df['volume'].rolling(window=5).mean().shift(5)
    volume_trend = avg_volume_t_5 / avg_volume_t_6_10
    df['enhanced_intermediate_factor'] = (df['volume_increase_ratio'] * (df['close'] - avg_close_5) * df['price_volatility']) * volume_trend
    
    # Adjust for Volatility
    rolling_volatility = df['high'].sub(df['low']).rolling(window=21).std()
    df['adjusted_combined_metrics'] = (df['volume_weighted_intraday_return'] + df['enhanced_intermediate_factor']) / rolling_volatility
    
    # Final Composite Factor
    df['final_composite_factor'] = df['adjusted_combined_metrics']
    
    return df['final_composite_factor']
```
