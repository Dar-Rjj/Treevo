```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Determine Spike Threshold
    def calculate_spike_threshold(volume_changes, multiplier=1.5):
        percentile_75 = volume_changes.quantile(0.75)
        return percentile_75 * multiplier
    
    # Calculate the spike threshold using the past 21 days' volume changes
    df['spike_threshold'] = df['volume_change'].rolling(window=21).apply(
        lambda x: calculate_spike_threshold(x), raw=False
    )
    
    # Filter with Spike Condition
    df['is_spike'] = df['volume_change'] > df['spike_threshold']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Apply Spike Condition to the volume-weighted intraday return
    df['factor'] = df['volume_weighted_intraday_return'] * df['is_spike']
    
    return df['factor']
```
