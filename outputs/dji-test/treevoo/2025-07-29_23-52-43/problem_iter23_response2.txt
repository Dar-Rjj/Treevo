import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['Daily_High_Low_Range_Percentage'] = (df['High'] - df['Low']) / df['Open']
    
    # Calculate Volume-Weighted Intraday Momentum

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['High'] - df['Low']
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['Combined_Intraday_Metric'] = df['Daily_High_Low_Range_Percentage'] * df['Volume_Weighted_Intraday_Momentum']
    
    # Calculate Momentum Difference
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate VWAP Difference
    df['VWAP'] = (df['Close'] * df['Volume']).cumsum() / df['Volume'].cumsum()
    df['VWAP_Shifted'] = df['VWAP'].shift(1)
    df['VWAP_Difference'] = df['VWAP'] - df['VWAP_Shifted']
    
    # Calculate Open-Close Spread Difference
    df['Open_Close_Spread'] = df['Close'] - df['Open']
    df['Open_Close_Spread_Shifted'] = df['Open_Close_Spread'].shift(1)
    df['Open_Close_Spread_Difference'] = df['Open_Close_Spread'] - df['Open_Close_Spread_Shifted']
    
    # Final Combined Indicator
    df['Final_Combined_Indicator'] = (df['Momentum_Difference'] + df['VWAP_Difference'] + df['Open_Close_Spread_Difference'])

    df['Short_Term_Price_Trend'] = df['Close'].rolling(window=5).mean()

    df['Adjusted_Combined_Indicator'] = df['Final_Combined_Indicator'] * (1 + df['Trend_Adjustment'])
