```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Daily VWAP
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Compute Intraday Open-Close Spread
    df['intraday_open_close_spread'] = df['close'] - df['open']
    
    # Compute Previous Day's Open-Close Spread
    df['prev_day_open_close_spread'] = (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['positive_breakout_intensity'] = np.maximum(df['breakout_intensity'], 0)
    
    # Calculate Momentum Difference
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Calculate VWAP Difference
    df['vwap_diff'] = df['daily_vwap'] - df['prev_day_vwap']
    
    # Calculate Open-Close Spread Difference
    df['open_close_diff'] = df['intraday_open_close_spread'] - df['prev_day_open_close_spread']
    
    # Weight by Volume and Breakout Intensity
    df['weighted_breakout_intensity'] = df['volume'] * df['positive_breakout_intensity']
    
    # Combine Metrics
    df['combined_metrics'] = (df['momentum_diff'] * df['vwap_diff'] + 
                              df['open_close_diff'] + 
                              df['weighted_breakout_intensity'])
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['rolling_volatility'].rolling(window=20).std()
    df['alpha_factor'] = df['combined_metrics'] / df['rolling_volatility']
    
    return df['alpha_factor']
```
