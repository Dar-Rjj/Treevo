```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return over the past 20 days (excluding today's log return)
    df['cumulative_log_return'] = df['log_return'].rolling(window=20, min_periods=1).sum().shift(1)
    
    # Identify Volume Shock
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_shock'] = np.where(df['volume_change'] > 1.5, 1, 0)
    
    # Combine Cumulative Log Return and Volume Shock
    df['alpha_factor'] = df['cumulative_log_return'] * df['volume_shock']
    
    return df['alpha_factor']
```
