```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Compute Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Previous Day's VWAP
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Compute Intraday Open-Close Spread
    df['Intraday_Open_Close_Spread'] = df['close'] - df['open']
    
    # Compute Previous Day's Open-Close Spread
    df['Prev_Day_Open_Close_Spread'] = df['Intraday_Open_Close_Spread'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['Breakout_Intensity'] = (df['high'] - df['low']) / df['close']
    df['Breakout_Intensity'] = df['Breakout_Intensity'].clip(lower=0)
    
    # Intraday Return
    df['Intraday_Return'] = df['close'] / df['open'] - 1
    
    # Volume Spike Detection
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['Volume_Change'].rolling(window=21).quantile(0.75) * 1.5
    df['Spike_Condition'] = df['Volume_Change'] > spike_threshold
    
    # Volume Increase Ratio
    df['Volume_Increase_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Price Range
    df['Price_Range'] = df['high'] - df['low']
    
    # Intermediate Factor
    df['Intermediate_Factor'] = (df['Volume_Increase_Ratio'] * df['Close_Price_Momentum'] / df['Price_Range'])
    
    # Moving Average of High-Low Differences (10-day period)
    df['MA_High_Low_Diff'] = df['Intraday_High_Low_Spread'].rolling(window=10).mean()
    
    # Final Alpha Factor
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    df['Open_Close_Spread_Difference'] = df['Intraday_Open_Close_Spread'] - df['Prev_Day_Open_Close_Spread']
    
    df['Volume_Weighted_Breakout'] = df['volume'] * df['Breakout_Intensity']
    
    df['Combined_Metrics'] = (df['Momentum_Difference'] * df['VWAP_Difference'] +
                              df['Open_Close_Spread_Difference'] +
                              df['Volume_Weighted_Breakout'])
    
    # Adjust for Volatility
    df['Rolling_Volatility'] = df['Price_Range'].rolling(window=20).std()
    df['Adjusted_Combined_Metrics'] = df['Combined_Metrics'] / df['Rolling_Volatility']
    
    # Incorporate Intermediate Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Combined_Metrics'] + df['Intermediate_Factor'] + df['MA_High_Low_Diff']
    
    return df['Final_Alpha_Factor']
```
