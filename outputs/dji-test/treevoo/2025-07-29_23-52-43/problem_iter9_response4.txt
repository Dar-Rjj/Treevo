```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))  # Ensure non-negative
    df['volume_weighted_breakout_intensity'] = df['breakout_intensity'] * df['volume']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.75) * 2.0
    df['is_spike'] = df['volume_change'] > spike_threshold

    # Filter with Spike Condition
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume'] * df['is_spike']

    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']

    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['high_low_diff'].rolling(window=10).sum()

    # Calculate Volume-Adjusted Momentum (VAM)
    df['sma_close'] = df['close'].rolling(window=20).mean()
    df['momentum'] = df['sma_close'] / df['sma_close'].shift(5) - 1
    df['vam'] = df['momentum'] * (df['volume'] / df['volume'].rolling(window=5).mean())

    # Calculate Price Volatility Indicator (PVI)
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    df['pvi'] = df['sma_close'] / df['atr']
    df['pvi_smoothed'] = df['pvi'].ewm(span=7).mean()

    # Calculate Logarithmic Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Combine Factors
    factor = (df['volume_weighted_breakout_intensity'] * df['vam'] / df['pvi_smoothed'] 
              * df['volume_weighted_intraday_return'] 
              * df['moving_sum_high_low_diff'] 
              * df['log_return'])

    return factor
```
