import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Open-Close Log Return Ratio
    df['open_close_log_return_ratio'] = np.log(df['close'] / df['open'])
    
    # High-Low Log Return Difference
    df['high_low_log_return_diff'] = np.log(df['high'] / df['low'])
    
    # Simple Moving Averages (SMA)
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Exponential Moving Averages (EMA)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # True Range and Average True Range (ATR)
    df['true_range'] = df[['high' - 'low', 
                           (df['high'] - df['close'].shift(1)).abs(), 
                           (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['atr_21'] = df['true_range'].rolling(window=21).mean()
    
    # Historical Volatility
    df['log_returns_30'] = df['close'].pct_change().rolling(window=30).std() * np.sqrt(252)
    df['log_returns_60'] = df['close'].pct_change().rolling(window=60).std() * np.sqrt(252)
    
    # Relative Strength Index (RSI)
    def rsi(data, window=14):
        delta = data.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        gain = up.rolling(window).mean()
        loss = down.abs().rolling(window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi_14'] = rsi(df['close'], window=14)
    df['rsi_28'] = rsi(df['close'], window=28)
    
    # Stochastic Oscillator
    def stochastic(data, k_period=14, d_period=3):
        low_min = data['low'].rolling(window=k_period).min()
        high_max = data['high'].rolling(window=k_period).max()
        k = 100 * (data['close'] - low_min) / (high_max - low_min)
        d = k.rolling(window=d_period).mean()
        return k, d
    
    df['%K_14'], df['%D_14'] = stochastic(df, k_period=14, d_period=3)
    df['%K_28'], df['%D_28'] = stochastic(df, k_period=28, d_period=3)
    
    # MACD (Moving Average Convergence Divergence)
    df['macd_line'] = df['ema_12'] - df['ema_26']
    df['macd_signal'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Breakout Indicator
    df['5_day_high_breakout'] = (df['high'] > df['high'].rolling(window=5).max()).astype(int)
    df['5_day_low_breakout'] = (df['low'] < df['low'].rolling(window=5).min()).astype(int)
    
    # Gap Analysis
    df['upward_gap'] = (df['open'] > df['close'].shift(1)).astype(int)
    df['downward_gap'] = (df['open'] < df['close'].shift(1)).astype(int)
    
    # On-Balance Volume (OBV)
    df['obv'] = np.where(df['close'] > df['close'].shift(1), df['volume'],
                         np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
