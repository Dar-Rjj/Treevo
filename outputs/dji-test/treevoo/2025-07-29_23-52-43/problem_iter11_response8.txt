```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    max_volume = df['volume'].max()
    df['Volume_Weighted_Intraday_Momentum'] = (df['close'] - df['open']) * (df['volume'] / max_volume)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['Combined_Intraday_Factor'] = df['High_Low_Range_Percentage'] * df['Volume_Weighted_Intraday_Momentum']
    
    # Calculate Momentum Difference
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    
    # Calculate VWAP Difference
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    
    # Calculate Open-Close Spread Difference
    df['Open_Close_Spread_Difference'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Final Combined Indicator
    df['Final_Combined_Indicator'] = (df['Momentum_Difference'] * df['VWAP_Difference']) + df['Open_Close_Spread_Difference']
    
    # Adjust for Price Trend
    df['Short_Term_Price_Trend'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Price_Trend'] = df['close'].rolling(window=20).mean()
    price_trend_difference = df['Long_Term_Price_Trend'] - df['Short_Term_Price_Trend']
    df['Trend_Adjusted_Combined_Indicator'] = df['Final_Combined_Indicator'] + np.where(price_trend_difference > 0, 1, -1) * price_trend_difference
    
    # Adjust for Volatility
    df['Rolling_Volatility'] = df['Intraday_High_Low_Spread'].rolling(window=20).std()
    df['Volatility_Adjusted_Combined_Indicator'] = df['Trend_Adjusted_Combined_Indicator'] / df['Rolling_Volatility']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return
    df['Cumulative_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).sum().shift(1)
    
    # Identify Volume Shock
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Shock'] = np.where(df['Volume_Change'] > 1.5, 1, 0)
    
    # Combine All Factors
    df['Alpha_Factor'] = (df['Cumulative_Log_Return'] * df['Volume_Shock']) + df['Volatility_Adjusted_Combined_Indicator']
    
    return df['Alpha_Factor']
```
