```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Daily VWAP
    df['daily_vwap'] = (df['close'] * df['volume']) / df['volume']
    
    # Calculate Previous Day's VWAP
    df['prev_day_vwap'] = (df['close'].shift(1) * df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['high_low_breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['high_low_breakout_intensity'] = df['high_low_breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['volume_weighted_breakout_intensity'] = df['high_low_breakout_intensity'] * df['volume']
    
    # Combine Metrics
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_difference'] = df['daily_vwap'] - df['prev_day_vwap']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['momentum_difference'] * df['vwap_difference'] * 
                          df['volume_weighted_breakout_intensity'])
    
    return df['alpha_factor']
```
