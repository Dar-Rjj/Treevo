```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold_80th_percentile = df['daily_volume_change'].rolling(window=22).quantile(0.8).shift(1)
    multiplier = 2.0
    df['spike_threshold'] = spike_threshold_80th_percentile * multiplier
    df['volume_spike'] = df['daily_volume_change'] > df['spike_threshold']
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['volume_weighted_breakout_intensity'] = df['breakout_intensity'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['factor'] = (df['volume_weighted_intraday_return'] + df['volume_weighted_breakout_intensity']) * df['volume_spike']
    
    return df['factor']
```
