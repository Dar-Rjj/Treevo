```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    previous_close = df['close'].shift(1)
    close_to_previous_close_return = (df['close'] - previous_close) / previous_close
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Adjust for Volatility
    rolling_volatility = df['high'].sub(df['low']).rolling(window=20).std()
    adjusted_metrics = combined_metrics / rolling_volatility
    
    # Incorporate Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    final_combined_indicator = adjusted_metrics + short_term_momentum + long_term_momentum
    
    # Generate Crossover Alpha Factor
    crossover_alpha_factor = combined_metrics + final_combined_indicator
    
    return crossover_alpha_factor
```
