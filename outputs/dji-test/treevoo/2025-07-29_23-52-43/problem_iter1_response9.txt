import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Adjusted Momentum (VAM)
    close_sma_20 = df['close'].rolling(window=20).mean()
    momentum_5 = close_sma_20 - close_sma_20.shift(5)
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    vam = momentum_5 * (df['volume'] / avg_volume_5)

    # Calculate Price Volatility Indicator (PVI)
    high = df['high']
    low = df['low']
    close = df['close']
    tr = pd.DataFrame({
        'hl': high - low,
        'hc': (high - close.shift()).abs(),
        'lc': (low - close.shift()).abs()
    }).max(axis=1)
    atr_14 = tr.rolling(window=14).mean()
    pvi = tr / atr_14
    pvi_smoothed = pvi.ewm(span=7, adjust=False).mean()

    # Combine VAM and PVI
    factor = vam * pvi_smoothed
    factor_normalized = factor - factor.rolling(window=30).mean()

    return factor_normalized
