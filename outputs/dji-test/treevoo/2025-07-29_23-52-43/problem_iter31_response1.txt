import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Close-to-PreviousClose Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_prev_return'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # High-Low Breakout Intensity
    df['high_low_breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['high_low_breakout_intensity'] = df['high_low_breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Volume-Weighted Combined Metrics
    df['volume_weighted_metrics'] = df['volume'] * (df['intraday_return'] + df['close_to_prev_return'] + df['high_low_breakout_intensity'])
    
    # Rolling Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['high_low_volatility'].rolling(window=20).std()
    
    # Volatility-Adjusted Combined Metrics
    df['volatility_adjusted_metrics'] = df['volume_weighted_metrics'] / df['rolling_volatility']
    
    return df['volatility_adjusted_metrics']
