```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Volume Weighted Close Price Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    volume_weight = df['volume'] / df['volume'].rolling(window=6).mean()
    short_term_vol_weighted_momentum = short_term_momentum * volume_weight

    # Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Price Volatility
    price_volatility = df['close'].rolling(window=7).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    breakout_intensity[breakout_intensity < 0] = 0

    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)

    # Moving Average of High-Low Differences (10-day period)
    high_low_diff_ma = (df['high'] - df['low']).rolling(window=10).mean()

    # Intermediate Factor
    intermediate_factor = volume_increase_ratio * (df['close'] - df['close'].rolling(window=6).mean()) * (df['close'].rolling(window=7).std() / df['close'].shift(1))

    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_open_return + breakout_intensity

    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=7).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility

    # Final Composite Factor
    final_composite_factor = (
        intermediate_factor +
        high_low_diff_ma +
        short_term_vol_weighted_momentum +
        price_volatility +
        combined_metrics +
        adjusted_combined_metrics
    )

    # Enhance with Trade Activity and Long-Term Momentum
    trade_activity = df['volume'] / df['volume'].rolling(window=10).mean()
    enhanced_factor = (final_composite_factor * trade_activity) + long_term_momentum

    return enhanced_factor
```
