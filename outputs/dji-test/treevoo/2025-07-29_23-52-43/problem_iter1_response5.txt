import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['previous_close'] = df['close'].shift(1)
    close_to_previous_close_return = (df['close'] - df['previous_close']) / df['previous_close']
    
    # Calculate Daily Trading Range
    daily_trading_range = df['high'] - df['low']
    
    # Combine Intraday and Close-to-PreviousClose Returns
    combined_return = intraday_return + close_to_previous_close_return
    
    # Weight by Volume
    weighted_combined_return = combined_return * df['volume']
    
    # Adjust for Volatility
    # Calculate Rolling Volatility using the standard deviation of the daily trading range
    rolling_volatility = daily_trading_range.rolling(window=20).std()
    
    # Divide Combined Return by Rolling Volatility
    adjusted_factor = weighted_combined_return / rolling_volatility
    
    return adjusted_factor.dropna()
