import pandas as pd

def heuristics_v2(df):
    # Calculate Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Ensure Breakout Intensity is Positive
    breakout_intensity = breakout_intensity.apply(lambda x: max(0, x))
    
    # Multiply by Volume
    factor_values = breakout_intensity * df['volume']
    
    return factor_values
