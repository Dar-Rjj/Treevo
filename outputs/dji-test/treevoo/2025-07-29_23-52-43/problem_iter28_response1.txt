```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.9) * 2.0
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Close Price Momentum
    df['avg_close_5'] = df['close'].rolling(window=5).mean()
    df['close_momentum'] = df['close'] - df['avg_close_5']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume'] / df['volume'].shift(1)) * df['close_momentum']
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Final Composite Factor
    df['final_composite'] = df['intermediate_factor'] / df['price_range']
    
    # Confirm Momentum with Volume-Weighted Difference
    df['volume_weighted_momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].max())
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['combined_intraday'] = df['volume_weighted_momentum'] * (df['intraday_high_low_spread'] / df['open'])
    
    # Calculate VWAP Difference
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_diff'] = df['vwap'] - df['vwap'].shift(1)
    
    # Final Combined Indicator
    df['final_combined_indicator'] = df['combined_intraday'] + df['vwap_diff']
    
    # Adjust for Price Trend
    df['short_term_trend'] = df['close'].rolling(window=5).mean()
    df['long_term_trend'] = df['close'].rolling(window=20).mean()
    trend_diff = df['long_term_trend'] - df['short_term_trend']
    df['trend_adjusted_indicator'] = df['final_combined_indicator'] * trend_diff
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['price_range'].rolling(window=20).std()
    df['volatility_adjusted_indicator'] = df['trend_adjusted_indicator'] / df['rolling_volatility']
    
    # Calculate Cumulative Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['cumulative_log_return'] = df['log_return'].rolling(window=20).sum()
    
    # Introduce New Factor: Intraday Price Efficiency
    df['intraday_efficiency_ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['efficiency_adjusted_indicator'] = df['volatility_adjusted_indicator'] * df['intraday_efficiency_ratio']
    
    # Introduce New Factor: Intraday Volume Efficiency
    df['intraday_volume_efficiency'] = (df['volume'] - df['volume'].shift(1)) / (df['high'] - df['low'])
    df['final_factor'] = df['efficiency_adjusted_indicator'] * df['intraday_volume_efficiency']
    
    return df['final_factor']
```
