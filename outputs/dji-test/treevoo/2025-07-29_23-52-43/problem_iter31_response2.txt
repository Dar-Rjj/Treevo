```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = ((df['high'] - df['low']) / df['close']).apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    df['combined_metrics'] = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    
    # Weight by Volume
    df['weighted_combined_metrics'] = df['combined_metrics'] * df['volume']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['rolling_volatility'].rolling(window=21).std()
    df['volatility_adjusted_metrics'] = df['weighted_combined_metrics'] / df['rolling_volatility']
    
    # Calculate Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['daily_high_low_diff'].rolling(window=21).sum()
    
    # Compute Intraday Momentum
    df['intraday_momentum'] = df['daily_high_low_diff'] - df['daily_high_low_diff'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_vwap'] = df['intraday_momentum'] * (df['vwap'] - df['prev_vwap'])
    
    # Adjust for Momentum
    df['rolling_momentum'] = (df['close'] - df['close'].shift(1)).rolling(window=21).sum()
    df['momentum_adjusted_metrics'] = df['volatility_adjusted_metrics'] / df['rolling_momentum']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['spike_threshold'] = 1.5 * df['volume_change'].rolling(window=21).quantile(0.85)
    df['spike_condition'] = df['volume_change'] > df['spike_threshold']
    
    # Filter with Spike Condition
    df['filtered_metrics'] = df['momentum_adjusted_metrics'] * df['spike_condition']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume'] / df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=7).mean())
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['filtered_metrics'] + df['intermediate_factor']
    
    return df['crossover_alpha_factor']
```
