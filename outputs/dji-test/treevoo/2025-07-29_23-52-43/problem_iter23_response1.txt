import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=5).mean()
    momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=20).quantile(0.8) * 1.7
    volume_spike = (daily_volume_change > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(2)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']).div(df['close'].shift(1)).clip(lower=0)
    
    # Moving Average of High-Low Differences (10-day period)
    ma_high_low_diff = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Intermediate Factor
    intermediate_factor = volume_increase_ratio * (df['close'] - df['close'].rolling(window=6).mean())
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    # Calculate Daily High-Low Difference
    daily_high_low_diff = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (14-day period)
    moving_sum_high_low_diff = daily_high_low_diff.rolling(window=14).sum()
    
    # Compute Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    previous_day_vwap = (df['close'].shift(1) * df['volume'].shift(1)).cumsum() / df['volume'].shift(1).cumsum()
    
    # Combine Intraday Momentum and VWAP
    momentum_diff = intraday_momentum
    vwap_diff = vwap - previous_day_vwap
    final_combined_indicator = momentum_diff * vwap_diff
    
    # Identify Volume Spikes
    volume_spike = (daily_volume_change > spike_threshold).astype(int)
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Filter with Spike Condition
    filtered_combined_metrics = adjusted_combined_metrics * volume_spike
    
    # Generate Crossover Alpha Factor
    crossover_alpha_factor = filtered_combined_metrics * final_combined_indicator
    
    return crossover_alpha_factor
