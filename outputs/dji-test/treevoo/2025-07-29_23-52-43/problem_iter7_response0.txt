```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted High-Low Breakout Intensity.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the calculated factor values, indexed by date.
    """
    # Calculate Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Ensure Positive: If Breakout Intensity < 0, set to 0
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Multiply by Volume
    volume_weighted_breakout_intensity = breakout_intensity * df['volume']
    
    return volume_weighted_breakout_intensity
```
