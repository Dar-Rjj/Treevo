```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    prev_day_high_low_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    prev_day_vwap = vwap.shift(1)
    
    # Combine Intraday Momentum and VWAP
    momentum_difference = intraday_high_low_spread - prev_day_high_low_spread
    vwap_difference = vwap - prev_day_vwap
    intermediate_alpha_factor_1 = momentum_difference * vwap_difference
    
    # Calculate Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Adjust for Price Volatility
    price_range = df['high'] - df['low']
    
    # Generate Final Factor
    final_factor = (intermediate_alpha_factor_1 * volume_increase_ratio * close_price_change) / price_range
    
    return final_factor
```
