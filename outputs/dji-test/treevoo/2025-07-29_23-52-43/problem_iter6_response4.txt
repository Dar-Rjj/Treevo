```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_HL_Spread'] = df['Intraday_HL_Spread'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['Momentum_Difference'] = df['Intraday_HL_Spread'] - df['Prev_Day_HL_Spread']
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    df['Intermediate_Alpha_Factor_1'] = df['Momentum_Difference'] * df['VWAP_Difference']
    
    # Calculate Volume Increase Ratio
    df['Volume_Increase_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Adjust for Price Volatility
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Opening Gap
    df['Opening_Gap'] = df['open'] - df['close'].shift(1)
    
    # Adjust for Opening Gap
    df['Opening_Gap_Adjustment'] = df['Opening_Gap'] * df['Volume_Increase_Ratio']
    
    # Combine All Factors
    df['Final_Factor'] = (df['Intermediate_Alpha_Factor_1'] * df['Volume_Increase_Ratio'] * df['Opening_Gap_Adjustment']) / df['Price_Range']
    
    return df['Final_Factor']
```
