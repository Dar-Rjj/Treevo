```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['previous_close'] = df['close'].shift(1)
    close_to_prev_close_return = (df['close'] - df['previous_close']) / df['previous_close']
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_prev_close_return + breakout_intensity
    weighted_combined_metrics = combined_metrics * df['volume']
    
    # Adjust for Volatility
    daily_log_returns = np.log(df['close'] / df['previous_close']).fillna(0)
    rolling_volatility = daily_log_returns.rolling(window=20).std()
    adjusted_metrics = weighted_combined_metrics / rolling_volatility
    
    return adjusted_metrics
```
