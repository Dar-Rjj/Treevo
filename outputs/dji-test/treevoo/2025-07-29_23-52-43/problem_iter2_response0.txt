```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_difference'] = df['daily_vwap'] - df['prev_day_vwap']
    df['intermediate_alpha_factor_1'] = df['momentum_difference'] * df['vwap_difference']
    
    # Calculate Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    
    # Combine All Factors
    df['final_factor'] = (df['intermediate_alpha_factor_1'] * df['volume_increase_ratio']) / df['price_range']
    
    return df['final_factor']
```
