import pandas as pd

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=6).mean().shift(1)
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)

    # Price Volatility
    price_volatility = df['close'].rolling(window=7).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=20).quantile(0.8) * 1.7
    volume_spike = (daily_volume_change > spike_threshold).astype(int)

    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(3)

    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)

    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    moving_avg_high_low_diff = high_low_diff.rolling(window=10).mean()

    # Adjust for Volatility
    high_low_volatility = high_low_diff.rolling(window=10).std()
    adjusted_high_low_diff = moving_avg_high_low_diff / high_low_volatility

    # Final Composite Factor
    composite_factor = (
        close_momentum +
        price_volatility +
        intraday_return +
        breakout_intensity +
        adjusted_high_low_diff
    )

    return composite_factor
