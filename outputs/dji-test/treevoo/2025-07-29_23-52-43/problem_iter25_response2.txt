import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=5).mean()
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    daily_volume_change = df['volume'].pct_change()
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.75) * 1.5
    volume_spike = (daily_volume_change > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    ma_high_low_diff = high_low_diff.rolling(window=10).mean()
    
    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * (df['close'] - avg_close))
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Adjust for Volatility
    rolling_volatility = df['high'] - df['low']
    rolling_volatility = rolling_volatility.rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    # Combine Intraday Momentum and VWAP
    momentum_difference = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    vwap = (df['volume'] * (df['high'] + df['low']) / 2).cumsum() / df['volume'].cumsum()
    vwap_difference = vwap - vwap.shift(1)
    final_combined_indicator = momentum_difference * vwap_difference
    
    # Generate Crossover Alpha Factor
    crossover_alpha_factor = adjusted_combined_metrics * final_combined_indicator * volume_spike
    
    return crossover_alpha_factor
