```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean().shift(1)
    avg_volume = df['volume'].rolling(window=5).mean().shift(1)
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = 2.0 * df['volume'].pct_change().rolling(window=21).quantile(0.90)
    volume_spike = (daily_volume_change > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Moving Average of High-Low Differences (20-day period)
    moving_avg_high_low_diff = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Intermediate Factor 1
    intermediate_factor_1 = (volume_increase_ratio * (df['close'] - avg_close) / (df['high'] - df['low']))
    
    # Intermediate Factor 2
    intermediate_factor_2 = (volume_increase_ratio * (df['close'] - avg_close))
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Average Intraday Return (20-day period)
    avg_intraday_return = intraday_return.rolling(window=20).mean()
    
    # Final Composite Factor
    final_composite_factor = (
        intermediate_factor_1 + 
        intermediate_factor_2 + 
        moving_avg_high_low_diff + 
        close_momentum + 
        price_volatility + 
        combined_metrics + 
        avg_intraday_return
    )
    
    return final_composite_factor
```
