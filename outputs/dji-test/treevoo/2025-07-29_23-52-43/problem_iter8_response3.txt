import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=22).quantile(0.8).shift(1) * 2.0
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['volume_weighted_breakout_intensity'] = df['breakout_intensity'] * df['volume']
    
    # Calculate Volume-Weighted Intraday and Close-to-PreviousClose Returns
    df['intraday_return_weighted'] = (df['high'] - df['low']) / df['low'] * df['volume']
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['close_to_prev_close_return_weighted'] = df['close_to_prev_close_return'] * df['volume']
    df['combined_return'] = (df['intraday_return_weighted'] + df['close_to_prev_close_return_weighted']) / 2.0
    
    # Calculate Daily Trading Range Adjusted by Volume
    df['daily_trading_range_adjusted'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['rolling_volatility'].rolling(window=30).std()
    df['adjusted_combined_return'] = df['combined_return'] / df['rolling_volatility']
    
    # Calculate Price Volatility Indicator (PVI)
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    df['pvi'] = df['true_range'] / df['atr']
    df['smoothed_pvi'] = df['pvi'].ewm(span=7).mean()
    
    # Filter with Spike Condition
    df['factor'] = df['adjusted_combined_return'] + df['volume_weighted_breakout_intensity']
    df['factor'] = df['factor'] - df['factor'].rolling(window=30).mean()
    df['factor'] = df['factor'] * df['spike_condition']
    
    return df['factor']
