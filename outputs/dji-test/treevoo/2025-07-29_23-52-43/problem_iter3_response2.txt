```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Adjusted Momentum (VAM)
    sma_close = df['close'].rolling(window=20).mean()
    momentum = sma_close.pct_change(periods=5)
    average_volume = df['volume'].rolling(window=5).mean()
    vam = momentum * (df['volume'] / average_volume)

    # Calculate Price Volatility Indicator (PVI)
    tr = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    pvi = atr - tr
    pvi_smooth = pvi.ewm(span=7, adjust=False).mean()

    # Combine VAM and PVI
    combined_factor = vam * pvi_smooth
    combined_factor = combined_factor - combined_factor.rolling(window=30).mean()

    return combined_factor
```
