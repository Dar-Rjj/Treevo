```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the daily return
    df['ma_5d_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the daily return
    df['ma_20d_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate the 5-day standard deviation of the daily return
    df['std_5d_return'] = df['daily_return'].rolling(window=5).std()
    
    # Calculate the 20-day standard deviation of the daily return
    df['std_20d_return'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the volume-weighted return
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Calculate the 5-day moving average of the volume-weighted return
    df['ma_5d_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the volume-weighted return
    df['ma_20d_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Calculate the relative strength compared to the 5-day moving average
    df['relative_strength_5d'] = df['close'] / df['close'].rolling(window=5).mean()
    
    # Calculate the relative strength compared to the 20-day moving average
    df['relative_strength_20d'] = df['close'] / df['close'].rolling(window=20).mean()
    
    # Calculate the final alpha factor
    df['alpha_factor'] = (
        0.3 * df['ma_5d_return'] + 
        0.2 * df['ma_20d_return'] + 
        0.1 * df['std_5d_return'] - 
        0.1 * df['std_20d_return'] + 
        0.1 * df['ma_5d_volume_weighted_return'] + 
        0.1 * df['ma_20d_volume_weighted_return'] + 
        0.05 * df['relative_strength_5d'] - 
        0.05 * df['relative_strength_20d']
    )
    
    return df['alpha_factor'].dropna()
```
