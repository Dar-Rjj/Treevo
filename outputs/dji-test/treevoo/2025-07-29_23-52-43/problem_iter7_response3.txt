```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the High-Low Difference Momentum factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and a datetime index.
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences for the last 10 days
    df['sum_high_low_diff_10d'] = df['daily_high_low_diff'].rolling(window=10).sum()
    
    # Divide by Close Price
    df['factor'] = df['sum_high_low_diff_10d'] / df['close']
    
    return df['factor']
```
