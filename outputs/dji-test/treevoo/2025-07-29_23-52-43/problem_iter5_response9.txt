```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=1).max())
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    combined_factor = intraday_range * volume_weighted_momentum
    
    # Calculate Short-Term Price Trend (5-day average)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Trend (20-day average)
    long_term_trend = df['close'].rolling(window=20).mean()
    
    # Adjust Combined Factor by Price Trends
    trend_difference = long_term_trend - short_term_trend
    adjusted_combined_factor = combined_factor * (1 + trend_difference / df['close'])
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].shift(1).rolling(window=5).mean()
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Final Composite Factor
    final_composite_factor = (adjusted_combined_factor + 
                              (volume_increase_ratio * close_price_momentum) / price_range)
    
    return final_composite_factor
```
