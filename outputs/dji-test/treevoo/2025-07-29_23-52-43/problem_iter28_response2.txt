import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['previous_close'] = df['close'].shift(1)
    close_to_prev_close_return = (df['close'] - df['previous_close']) / df['previous_close']
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = ((df['close'] - df['open']) / df['open'])
    avg_volume_5 = df['volume'].rolling(window=5).mean().shift(1)
    volume_weighted_intraday_return *= (df['volume'] / avg_volume_5)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Intermediate Factor
    close_avg_5 = df['close'].rolling(window=5).mean().shift(1)
    intermediate_factor = (volume_increase_ratio * (df['close'] - close_avg_5) * price_volatility)
    
    # Final Composite Factor
    final_composite = volume_weighted_intraday_return + intermediate_factor
    
    # Adjust for Volatility
    high_low_volatility = (df['high'] - df['low']).rolling(window=5).std()
    final_composite /= high_low_volatility
    
    return final_composite
