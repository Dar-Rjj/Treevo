```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Moving Sum of High-Low Differences (14-day period)
    df['hl_diff_14d_sum'] = df['high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high_low_diff'].shift(1)
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Calculate VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['vwap'].shift(1)
    df['vwap_diff'] = df['vwap'] - df['prev_day_vwap']
    
    # Combine Intraday Momentum and VWAP
    df['final_combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['vwap_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_change_80th'] = df['volume_change'].rolling(window=22).quantile(0.8)
    spike_threshold_multiplier = 1.6
    df['spike_threshold'] = df['volume_change_80th'] * spike_threshold_multiplier
    df['spike_condition'] = df['volume_change'] > df['spike_threshold']
    
    # Filter with Spike Condition
    df['filtered_vwap_intraday_return'] = df['vwap_intraday_return'] * df['spike_condition']
    
    # Synthesize Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = df['hl_diff_14d_sum'] + df['final_combined_indicator']
    
    # Generate Crossover High-Low Difference Momentum and Volume-Weighted Intraday Return
    df['crossover_alpha_factor'] = df['intermediate_alpha_factor'] * df['filtered_vwap_intraday_return']
    
    return df['crossover_alpha_factor']
```
