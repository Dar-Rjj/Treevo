```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close_5 = df['close'].rolling(window=5).mean().shift(1)
    close_momentum = (df['close'] - avg_close_5) * (df['volume'] / df['volume'].rolling(window=5).mean().shift(1))
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Close-to-PreviousClose Return
    prev_close = df['close'].shift(1)
    close_to_prev_close_return = (df['close'] - prev_close) / prev_close
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = np.where(breakout_intensity > 0, breakout_intensity, 0)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_prev_close_return + breakout_intensity
    weighted_combined_metrics = combined_metrics * df['volume']
    
    # Final Factor
    final_factor = (volume_increase_ratio * close_momentum * price_volatility + weighted_combined_metrics)
    
    return final_factor
```
