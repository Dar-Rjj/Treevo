```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    previous_close = df['close'].shift(1)
    close_to_previous_close_return = (df['close'] - previous_close) / previous_close
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * (df['high'] - df['low']) / df['low']
    
    # Combine Metrics
    combined_metrics = volume_weighted_intraday_return + df['volume'] * (close_to_previous_close_return + breakout_intensity)
    
    # Adjust for Recent Volatility
    rolling_volatility = (df['close'] - df['open']).rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    return adjusted_combined_metrics
```
