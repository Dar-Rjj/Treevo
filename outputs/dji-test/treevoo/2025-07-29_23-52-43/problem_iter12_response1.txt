```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (14-day period)
    df['high_low_diff_sum'] = df['high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high_low_diff'].shift(1)
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    df['vwap_difference'] = df['daily_vwap'] - df['prev_day_vwap']
    
    # Combine Intraday Momentum and VWAP
    df['combined_indicator'] = df['momentum_difference'] * df['vwap_difference']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=22).quantile(0.8) * 1.6
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Synthesize Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = df['high_low_diff_sum'] + df['combined_indicator']
    
    # Generate Crossover High-Low Difference Momentum and Volume-Weighted Intraday Return
    df['alpha_factor'] = df['intermediate_alpha_factor'] + df['filtered_volume_weighted_intraday_return']
    
    return df['alpha_factor']
```
