```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    df['combined_metrics'] = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    df['weighted_combined_metrics'] = df['combined_metrics'] * df['volume']
    
    # Volume Weighted Intraday Return
    df['vol_weighted_intraday_return'] = (df['close'] - df['open']) / df['open'] * (df['volume'] / df['volume'].rolling(window=5).mean().shift(1))
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.8) * 1.6
    df['volume_spike'] = df['volume_change'] > spike_threshold
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Price Volatility
    df['price_volatility'] = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Intermediate Factor
    df['intermediate_factor'] = df['volume_increase_ratio'] * (df['close'] - df['close'].rolling(window=5).mean()) * df['price_volatility']
    
    # Adjust for Volatility
    df['rolling_volatility'] = (df['high'] - df['low']).rolling(window=5).std()
    df['adjusted_combined_metrics'] = df['weighted_combined_metrics'] / df['rolling_volatility']
    
    # Compute Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    df['prev_day_momentum'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate VWAP
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = (df['close'].shift(1) * df['volume'].shift(1)).cumsum() / df['volume'].shift(1).cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['momentum_difference'] = df['intraday_momentum'] - df['prev_day_momentum']
    df['vwap_difference'] = df['daily_vwap'] - df['prev_day_vwap']
    df['final_combined_indicator'] = df['momentum_difference'] * df['vwap_difference']
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['adjusted_combined_metrics'] * df['final_combined_indicator'] * df['intermediate_factor']
    
    return df['crossover_alpha_factor']
```
