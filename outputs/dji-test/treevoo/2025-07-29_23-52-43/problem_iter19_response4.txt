```python
import pandas as pd

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    close_momentum = df['close'] - df['close'].rolling(window=5).mean()
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_weighted_momentum = close_momentum * volume_ratio

    # Price Volatility
    price_volatility = df['close'].rolling(window=5).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)

    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_prev_close_return + breakout_intensity

    # Final Factor
    final_factor = volume_weighted_momentum * price_volatility + combined_metrics

    return final_factor
```
