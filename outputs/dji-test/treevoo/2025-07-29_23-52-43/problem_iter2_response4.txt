```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    close_to_prev_close_return = close_to_prev_close_return.fillna(0)  # Handle the first day
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_prev_close_return + breakout_intensity
    weighted_combined_metrics = combined_metrics * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    adjusted_factor = weighted_combined_metrics / rolling_volatility
    adjusted_factor = adjusted_factor.fillna(0)  # Handle the first 19 days with NaN volatility
    
    return adjusted_factor
```
