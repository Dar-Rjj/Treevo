```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['High', 'Low', 'Close']].mean(axis=1) * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Calculate Volume Increase Ratio
    df['Volume_t-1'] = df['Volume'].shift(1)
    df['Volume_Increase_Ratio'] = df['Volume'] / df['Volume_t-1']
    
    # Calculate Close Price Momentum
    df['Close_t-5_to_t-1'] = df['Close'].shift(1).rolling(window=5).mean()
    df['Close_Price_Momentum'] = df['Close'] - df['Close_t-5_to_t-1']
    
    # Adjust for Price Volatility
    df['Price_Range'] = df['High'] - df['Low']
    
    # Combine Factors
    df['Interim_Factor'] = (df['Close_Price_Momentum'] * df['Volume_Increase_Ratio']) / df['Price_Range']
    df['Final_Factor'] = df['Interim_Factor'] * df['VWAP']
    
    return df['Final_Factor']
```
