import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1) - 1
    spike_threshold = (daily_volume_change.rolling(window=21).quantile(0.9) * 2.0)
    volume_spike = daily_volume_change > spike_threshold
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].shift(1).rolling(window=5).mean()
    
    # Intermediate Factor
    intermediate_factor = volume_increase_ratio * close_price_momentum
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Final Composite Factor
    final_composite_factor = intermediate_factor * price_range
    
    return final_composite_factor
