import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday and Close-to-PreviousClose Returns
    df['combined_return'] = df['intraday_return'] + df['close_to_prev_close_return']
    
    # Weight by Volume
    df['volume_weighted_return'] = df['combined_return'] * df['volume']
    
    # Calculate Daily Trading Range Adjusted by Volume
    df['range_adjusted_by_volume'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Volatility: Calculate Rolling Volatility
    rolling_window = 20
    df['volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['volatility'].rolling(window=rolling_window).std()
    
    # Divide Combined Return by Rolling Volatility
    df['volatility_adjusted_return'] = df['volume_weighted_return'] / df['rolling_volatility']
    
    # Calculate Price Volatility Indicator (PVI)
    df['true_range'] = np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    )
    
    lookback_period = 14
    df['atr'] = df['true_range'].rolling(window=lookback_period).mean()
    
    # Smooth PVI/ATR using EMA with a period
    ema_period = 7
    df['pvi'] = df['true_range'] / df['atr']
    df['smoothed_pvi'] = df['pvi'].ewm(span=ema_period, adjust=False).mean()
    
    # Combine Volume-Weighted Returns and PVI
    df['final_factor'] = df['volatility_adjusted_return'] * df['smoothed_pvi']
    
    # Subtract mean of the product over a period
    smoothing_period = 30
    df['factor'] = df['final_factor'] - df['final_factor'].rolling(window=smoothing_period).mean()
    
    return df['factor']
