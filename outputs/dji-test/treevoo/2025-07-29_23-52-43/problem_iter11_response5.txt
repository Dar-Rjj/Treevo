```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    max_volume = df['volume'].max()
    df['Volume_Weighted_Intraday_Momentum'] = (df['close'] - df['open']) * (df['volume'] / max_volume)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['Combined_Momentum'] = df['High_Low_Range_Percentage'] * df['Volume_Weighted_Intraday_Momentum']
    
    # Calculate Momentum Difference
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    
    # Calculate VWAP Difference
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    
    # Final Combined Indicator
    df['Final_Combined_Indicator'] = df['Momentum_Difference'] * df['VWAP_Difference']
    
    # Adjust for Price Trend
    df['Short_Term_Price_Trend'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Price_Trend'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Adjustment'] = (df['Long_Term_Price_Trend'] - df['Short_Term_Price_Trend'])
    df['Adjusted_Combined_Indicator'] = df['Final_Combined_Indicator'] * (1 + df['Price_Trend_Adjustment'])
    
    # Volume Increase Ratio
    df['Volume_Increase_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['Close_Price_Momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Price Range
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Daily High-Low Difference
    df['Daily_High_Low_Difference'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['Moving_Sum_High_Low_Differences'] = df['Daily_High_Low_Difference'].rolling(window=10).sum()
    
    # Compute Intraday Volatility
    df['Intraday_Volatility'] = df['Intraday_High_Low_Spread'] / df['Prev_Day_High_Low_Spread']
    
    # Calculate VWAP Ratio
    df['VWAP_Ratio'] = df['Daily_VWAP'] / df['Prev_Day_VWAP']
    
    # Combine Intraday Volatility and VWAP Ratios
    df['Combined_Ratios'] = df['Intraday_Volatility'] * df['VWAP_Ratio']
    
    # Synthesize Final Alpha Factor
    df['Alpha_Factor'] = (
        df['High_Low_Range_Percentage'] * 
        df['Volume_Weighted_Intraday_Momentum'] * 
        (1 + df['Price_Trend_Adjustment']) + 
        df['Adjusted_Combined_Indicator'] + 
        df['Volume_Increase_Ratio'] * 
        df['Close_Price_Momentum'] / 
        df['Price_Range'] + 
        df['Moving_Sum_High_Low_Differences'] + 
        df['Combined_Ratios']
    )
    
    return df['Alpha_Factor']
```
