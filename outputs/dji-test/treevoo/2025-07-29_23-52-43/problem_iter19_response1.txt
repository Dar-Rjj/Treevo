```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Log Returns
    df['log_return_open'] = np.log(df['open']).diff()
    df['log_return_close'] = np.log(df['close']).diff()
    df['log_return_high'] = np.log(df['high']).diff()
    df['log_return_low'] = np.log(df['low']).diff()
    
    # Open-Close Log Return Ratio
    df['open_close_log_return_ratio'] = df['log_return_open'] / df['log_return_close']
    
    # High-Low Log Return Difference
    df['high_low_log_return_diff'] = df['log_return_high'] - df['log_return_low']
    
    # Simple Moving Averages (SMA)
    df['sma_5_close'] = df['close'].rolling(window=5).mean()
    df['sma_20_close'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Averages (EMA)
    df['ema_12_close'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26_close'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # True Range
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Average True Range (ATR) over 14 days
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # 30-day Historical Volatility of Close Prices
    df['hist_vol_30'] = df['log_return_close'].rolling(window=30).std() * np.sqrt(252)
    
    # On-Balance Volume (OBV)
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    
    # 20-period Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['cmf_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # 14-period Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(), 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / negative_money_flow
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Daily VWAP
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['open_close_log_return_ratio'] +
        df['high_low_log_return_diff'] +
        df['sma_5_close'] - df['sma_20_close'] +
        df['ema_12_close'] - df['ema_26_close'] -
        df['atr_14'] +
        df['hist_vol_30'] +
        df['obv'] +
        df['cmf_20'] +
        df['mfi_14'] +
        df['vwap']
    )
    
    return df['alpha_factor']
```
