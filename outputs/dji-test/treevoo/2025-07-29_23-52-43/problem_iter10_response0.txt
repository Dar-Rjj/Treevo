```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume, setting to 0 if breakout intensity is negative
    volume_weighted_breakout = breakout_intensity * df['volume']
    volume_weighted_breakout[breakout_intensity < 0] = 0
    
    # Compute Simple Moving Average (SMA) of Close Prices (20 days)
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute Momentum (5-day change in SMA)
    momentum = sma_20.diff(5)
    
    # Normalize by average volume over the same momentum period
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = momentum * (df['volume'] / avg_volume_5)
    
    # Compute True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Compute ATR (Average True Range, 14 days)
    atr_14 = tr.rolling(window=14).mean()
    
    # Adjust PVI by ATR and Smooth using EMA (7 days)
    pvi = tr / atr_14
    pvi_smoothed = pvi.ewm(span=7).mean()
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Combine Factors
    factor = (volume_weighted_breakout * volume_adjusted_momentum) / pvi_smoothed
    factor *= volume_increase_ratio
    factor *= close_price_momentum
    factor /= price_range
    factor -= factor.rolling(window=30).mean()
    
    return factor
```
