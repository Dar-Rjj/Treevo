import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is a combination of momentum and volatility signals, using only current and past data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, volume) indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day momentum
    df['momentum_10'] = df['close'].pct_change(10)
    
    # Calculate the 20-day volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the 5-day average volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day average range (high - low)
    df['avg_range_5'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['momentum_10'] / df['volatility_20']) * (df['avg_volume_5'] / df['avg_range_5'])
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
