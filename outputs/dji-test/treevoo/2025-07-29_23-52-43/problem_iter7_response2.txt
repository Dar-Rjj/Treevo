```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['High', 'Low', 'Close']].mean(axis=1) * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Calculate Volume Increase Ratio
    df['Volume_Ratio'] = df['Volume'] / df['Volume'].shift(1)
    
    # Calculate Close Price Momentum
    df['Price_Momentum'] = df['Close'] - df['Close'].rolling(window=5).mean().shift(1)
    
    # Calculate Price Range
    df['Price_Range'] = df['High'] - df['Low']
    
    # Generate Final Factor
    df['Final_Factor'] = (df['Price_Momentum'] * df['Volume_Ratio']) / df['Price_Range'] * df['VWAP']
    
    return df['Final_Factor'].dropna()
```
