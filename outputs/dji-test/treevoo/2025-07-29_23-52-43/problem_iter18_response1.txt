```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['open']
    
    # Volume-Weighted Intraday Momentum
    max_volume = df['volume'].rolling(window=5, min_periods=1).max()
    df['Volume_Weighted_Momentum'] = (df['close'] - df['open']) * (df['volume'] / max_volume)
    
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['Combined_Indicator'] = df['High_Low_Range_Percentage'] * df['Volume_Weighted_Momentum']
    
    # Momentum Difference
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    
    # Adjust for Price Trend
    df['Short_Term_Price_Trend'] = df['close'].rolling(window=5, min_periods=1).mean()
    df['Long_Term_Price_Trend'] = df['close'].rolling(window=20, min_periods=1).mean()
    df['Price_Trend_Adjustment'] = df['Long_Term_Price_Trend'] - df['Short_Term_Price_Trend']
    df['Combined_Indicator'] += np.where(df['Price_Trend_Adjustment'] > 0, 1, -1) * df['Price_Trend_Adjustment']
    
    # Adjust for Volatility
    df['Rolling_Volatility'] = df['Intraday_High_Low_Spread'].rolling(window=20, min_periods=1).std()
    df['Combined_Indicator'] /= df['Rolling_Volatility']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return
    df['Cumulative_Log_Return'] = df['Daily_Log_Return'].rolling(window=20, min_periods=1).sum() - df['Daily_Log_Return']
    
    # Identify Volume Shock
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Shock'] = np.where(df['Volume_Change'] > 1.5, 1, 0)
    
    # Introduce Volume-Smoothed Momentum
    df['Volume_Smoothed_Close'] = df.apply(lambda x: (df.loc[:x.name, 'close'] * df.loc[:x.name, 'volume']).sum() / df.loc[:x.name, 'volume'].sum(), axis=1)
    df['Volume_Smoothed_Momentum'] = (df['Volume_Smoothed_Close'] - df['Volume_Smoothed_Close'].shift(1)) / df['Volume_Smoothed_Close'].shift(1)
    
    # Combine All Factors
    df['Alpha_Factor'] = df['Cumulative_Log_Return'] * df['Volume_Shock'] + df['Combined_Indicator'] + df['Volume_Smoothed_Momentum']
    
    return df['Alpha_Factor']
```
