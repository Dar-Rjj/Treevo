```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['open']
    
    # Confirm Momentum with Volume-Weighted Difference
    df['Volume_Weighted_Momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=5).max())
    
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['Combined_Indicator'] = df['High_Low_Range_Percentage'] * df['Volume_Weighted_Momentum']
    
    # Momentum Difference
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    df['Combined_Indicator'] *= df['Momentum_Difference']
    
    # Adjust for Price Trend
    df['Short_Term_Price_Trend'] = df['close'].rolling(window=5).mean()
    df['Long_Term_Price_Trend'] = df['close'].rolling(window=20).mean()
    trend_difference = df['Long_Term_Price_Trend'] - df['Short_Term_Price_Trend']
    df['Adjusted_Combined_Indicator'] = df['Combined_Indicator'] + trend_difference
    
    # Adjust for Volatility
    df['Rolling_Volatility'] = df['Intraday_High_Low_Spread'].rolling(window=20).std()
    df['Final_Combined_Indicator'] = df['Adjusted_Combined_Indicator'] / df['Rolling_Volatility']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return
    df['Cumulative_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).sum() - df['Daily_Log_Return']
    
    # Identify Volume Shock
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Shock'] = np.where(df['Volume_Change'] > 1.5, 1, 0)
    
    # Combine All Factors
    df['Alpha_Factor'] = df['Cumulative_Log_Return'] * df['Volume_Shock'] + df['Final_Combined_Indicator']
    
    return df['Alpha_Factor']
```
