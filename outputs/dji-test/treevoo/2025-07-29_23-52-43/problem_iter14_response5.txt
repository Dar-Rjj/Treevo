```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using historical data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['close'].shift(1)) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mf_multiplier * df['volume']
    
    # Calculate the rolling sum of positive and negative money flow volumes
    window = 14  # Define the rolling window size
    pos_mf = mf_volume.where(mf_multiplier > 0, 0).rolling(window=window).sum()
    neg_mf = mf_volume.where(mf_multiplier < 0, 0).rolling(window=window).sum().abs()
    
    # Calculate the money flow ratio
    money_ratio = pos_mf / neg_mf
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the relative strength indicator (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=window).mean()
    avg_loss = loss.rolling(window=window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine MFI and RSI to create a novel alpha factor
    alpha_factor = (mfi + rsi) / 2
    
    return alpha_factor
```
