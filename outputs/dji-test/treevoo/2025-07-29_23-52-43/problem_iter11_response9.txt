```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    df['High_Low_Sum'] = df['High_Low_Diff'].rolling(window=10).sum()
    
    # Divide by Close Price
    df['High_Low_Sum_Close'] = df['High_Low_Sum'] / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['Volume_Change'].rolling(window=21).quantile(0.75).shift(1) * 1.5
    df['Spike_Condition'] = df['Volume_Change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    
    # Filter with Spike Condition
    df['Filtered_Volume_Weighted_Intraday_Return'] = df['Volume_Weighted_Intraday_Return'] * df['Spike_Condition']
    
    # Volume-Weighted Breakout Intensity
    df['Breakout_Intensity'] = (df['high'] - df['low']) / df['close']
    df['Breakout_Intensity'] = df['Breakout_Intensity'].apply(lambda x: max(x, 0))
    df['Volume_Weighted_Breakout_Intensity'] = df['Breakout_Intensity'] * df['volume']
    
    # Volume-Adjusted Momentum (VAM)
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Momentum_5'] = df['SMA_20'] - df['SMA_20'].shift(5)
    df['Avg_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['VAM'] = df['Momentum_5'] * (df['volume'] / df['Avg_Volume_5'])
    
    # Volume-Weighted Close Price Momentum
    df['Close_Price_Momentum'] = (df['close'] - df['close'].shift(5).rolling(window=5).mean()) * (df['volume'] / df['Avg_Volume_5'])
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # True Range (TR)
    df['TR'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Average True Range (ATR)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Price Volatility Indicator (PVI)
    df['PVI'] = df['ATR_14'] * df['TR'].ewm(span=7, adjust=False).mean()
    
    # Volume Increase Ratio
    df['Volume_Increase_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Factors
    df['Combined_Factor'] = (df['Volume_Weighted_Breakout_Intensity'] * df['VAM'] / df['PVI'] * df['Volume_Increase_Ratio'] * df['Close_Price_Momentum'] / (df['high'] - df['low'])) * (df['High_Low_Sum_Close'] + df['Filtered_Volume_Weighted_Intraday_Return'])
    
    # Generate Crossover Factor
    df['Crossover_Factor'] = df['Combined_Factor']
    
    return df['Crossover_Factor']
```
