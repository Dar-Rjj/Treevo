```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['open']
    
    # Adjust by Volume and Close Difference
    adjusted_breakout_intensity = breakout_intensity * df['volume']
    
    # Calculate Close Difference
    close_difference = df['close'].diff()
    
    # Apply the condition based on Close Difference
    adjusted_breakout_intensity = adjusted_breakout_intensity.where(close_difference <= 0, adjusted_breakout_intensity * 2)
    
    return adjusted_breakout_intensity
```
