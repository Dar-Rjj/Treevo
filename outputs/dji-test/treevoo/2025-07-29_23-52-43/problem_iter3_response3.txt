```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate 80th Percentile of Past 21 Days' Volume Changes
    df['volume_change_80th_percentile'] = df['volume_change'].rolling(window=21).quantile(0.8)
    
    # Set Multiplier
    multiplier = 2.0
    
    # Calculate Spike Threshold
    df['spike_threshold'] = df['volume_change_80th_percentile'] * multiplier
    
    # Determine Spike Condition
    df['is_spike'] = df['volume_change'] > df['spike_threshold']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['factor_value'] = df['volume_weighted_intraday_return'] * df['is_spike']
    
    return df['factor_value']
```
