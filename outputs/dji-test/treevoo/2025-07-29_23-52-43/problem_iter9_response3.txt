```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Calculate Moving Average of High-Low Differences (10 days)
    df['MA_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=10).mean()
    
    # Compute Intraday Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['Momentum_Diff'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    df['VWAP_Diff'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    df['Combined_Indicator'] = df['Momentum_Diff'] * df['VWAP_Diff']
    
    # Add Price Trend Confirmation
    df['5-Day_SMA'] = df['close'].rolling(window=5).mean()
    df['20-Day_SMA'] = df['close'].rolling(window=20).mean()
    df['Price_Trend'] = (df['5-Day_SMA'] > df['20-Day_SMA']).astype(int) * 2 - 1
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['MA_High_Low_Diff'] + df['Combined_Indicator']) * df['Price_Trend']
    
    return df['Final_Alpha_Factor']
```
