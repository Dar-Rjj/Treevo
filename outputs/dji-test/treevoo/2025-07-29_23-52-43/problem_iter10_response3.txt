```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.75).shift(1) * 1.5
    df['volume_spike'] = (df['volume_change'] > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_momentum']) / df['price_range']
    
    # Moving Sum of High-Low Differences (10-day period)
    df['moving_sum_high_low'] = df['high'] - df['low']
    df['moving_sum_high_low'] = df['moving_sum_high_low'].rolling(window=10).sum()
    
    # Final Composite Factor
    df['final_composite_factor'] = df['intermediate_factor'] + df['moving_sum_high_low']
    
    return df['final_composite_factor']
```
