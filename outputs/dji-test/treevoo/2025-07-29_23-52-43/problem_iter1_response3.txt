```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['prev_day_vwap'] = df['vwap'].shift(1)
    
    # Calculate Momentum Difference
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Calculate VWAP Difference
    df['vwap_diff'] = df['vwap'] - df['prev_day_vwap']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_diff'] * df['vwap_diff']
    
    return df['alpha_factor']
```
