```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    df['volume_weighted_combined'] = combined_metrics * df['volume']
    
    # Adjust for Volatility
    high_low_range = df['high'] - df['low']
    rolling_volatility = high_low_range.rolling(window=21).std()
    df['volatility_adjusted_combined'] = df['volume_weighted_combined'] / rolling_volatility
    
    # Calculate Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['daily_high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_diff'] = df['vwap'] - df['prev_vwap']
    df['final_combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Adjust for Momentum
    rolling_momentum = (df['close'] - df['close'].shift(1)).rolling(window=21).sum()
    df['momentum_adjusted_combined'] = df['volatility_adjusted_combined'] / rolling_momentum
    
    # Identify Volume Spikes
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.80) * 1.6
    df['spike_condition'] = daily_volume_change > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_factor'] = df['volume_weighted_intraday_return'].where(df['spike_condition'], 0)
    
    # Intermediate Factor
    avg_close_5 = df['close'].rolling(window=5).mean()
    df['intermediate_factor'] = (df['volume'] / df['volume'].shift(1)) * (df['close'] - avg_close_5.shift(1))
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['momentum_adjusted_combined'] + df['final_combined_indicator'] + df['intermediate_factor']
    
    return df['crossover_alpha_factor'].dropna()
```
