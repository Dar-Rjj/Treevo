import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = (intraday_return + close_to_prev_close_return + breakout_intensity) * df['volume']
    
    # Adjust for Volatility
    daily_range = df['high'] - df['low']
    rolling_volatility = daily_range.rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    # Calculate Moving Sum of High-Low Differences
    high_low_diff = df['high'] - df['low']
    moving_sum_high_low_diff = high_low_diff.rolling(window=14).sum()
    
    # Compute Intraday Momentum
    intraday_momentum = (df['high'] - df['low'])
    prev_day_intraday_momentum = (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    prev_day_vwap = (df['close'].shift(1) * df['volume'].shift(1)).cumsum() / df['volume'].shift(1).cumsum()
    
    # Combine Intraday Momentum and VWAP
    momentum_difference = intraday_momentum - prev_day_intraday_momentum
    vwap_difference = vwap - prev_day_vwap
    final_combined_indicator = momentum_difference * vwap_difference
    
    # Generate Crossover Alpha Factor
    crossover_alpha_factor = adjusted_combined_metrics + final_combined_indicator
    
    return crossover_alpha_factor
