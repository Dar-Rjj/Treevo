```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    daily_high_low_range = (df['high'] - df['low']) / df['open']
    
    # Volume-Weighted Intraday Momentum
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=5).max())
    
    # Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Combine Daily High-Low Range Percentage and Volume-Weighted Intraday Momentum
    combined_intraday_indicator = daily_high_low_range * volume_weighted_momentum
    
    # Momentum Difference
    momentum_difference = intraday_high_low_spread - intraday_high_low_spread.shift(1)
    
    # VWAP Calculation
    vwap_daily = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_daily = vwap_daily.fillna(vwap_daily.rolling(window=5).mean())
    
    # VWAP Difference
    vwap_difference = vwap_daily - vwap_daily.shift(1)
    
    # Open-Close Spread Difference
    open_close_spread_difference = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Final Combined Indicator
    final_combined_indicator = (momentum_difference + vwap_difference + open_close_spread_difference) * combined_intraday_indicator
    
    # Adjust for Price Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    long_term_trend = df['close'].rolling(window=20).mean()
    trend_adjustment = (long_term_trend - short_term_trend) / short_term_trend
    adjusted_final_indicator = final_combined_indicator * (1 + trend_adjustment)
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    volatility_adjusted_indicator = adjusted_final_indicator / rolling_volatility
    
    # Calculate Cumulative Log Return
    log_returns = np.log(df['close'] / df['close'].shift(1))
    cumulative_log_return = log_returns.rolling(window=20).sum()
    
    # Identify Volume Shock
    volume_shock = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine All Factors
    alpha_factor = cumulative_log_return + volume_shock + volatility_adjusted_indicator
    
    return alpha_factor
```
