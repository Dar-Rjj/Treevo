```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using price action and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: Alpha factor values indexed by date
    """
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Measure intraday volatility
    df['intraday_range'] = df['high'] - df['low']
    
    # Analyze closing price momentum
    df['momentum_1d'] = (df['close'] / df['close'].shift(1)) - 1
    df['momentum_5d'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Examine trading volume changes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Evaluate high and low prices
    df['high_to_low_ratio'] = df['high'] / df['low']
    
    # Analyze amount traded (assuming 'amount' is the dollar value of trades)
    if 'amount' in df.columns:
        df['amount_change'] = df['amount'] / df['amount'].shift(1)
    else:
        df['amount_change'] = 1.0  # If 'amount' is not available, set to 1.0 (no change)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['daily_return'] * 0.2 +
        df['intraday_range'] * 0.1 +
        df['momentum_1d'] * 0.2 +
        df['momentum_5d'] * 0.2 +
        (df['volume_change'] - 1) * 0.1 +
        (df['high_to_low_ratio'] - 1) * 0.1 +
        (df['amount_change'] - 1) * 0.1
    )
    
    return alpha_factor
```
