```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['High_Low_Sum_10D'] = df['High_Low_Diff'].rolling(window=10).sum()
    
    # Divide by Close Price
    df['Normalized_High_Low_Sum'] = df['High_Low_Sum_10D'] / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['Volume_Change'].rolling(window=21).quantile(0.75) * 1.5
    df['Spike_Condition'] = df['Volume_Change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    
    # Filter with Spike Condition
    df['Filtered_Volume_Weighted_Intraday_Return'] = df['Volume_Weighted_Intraday_Return'] * df['Spike_Condition']
    
    # Volume Weighted Close Price Momentum
    df['Close_Price_Momentum'] = (df['close'] - df['close'].shift(5).rolling(window=5).mean()) * (df['volume'] / df['volume'].shift(5).rolling(window=5).mean())
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Combine Factors
    df['Composite_Factor'] = (df['Normalized_High_Low_Sum'] + 
                              df['Filtered_Volume_Weighted_Intraday_Return'] + 
                              df['Close_Price_Momentum'] + 
                              df['Price_Volatility'])
    
    # Generate Crossover Factor
    df['Crossover_Factor'] = df['Composite_Factor'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['Crossover_Factor']
```
