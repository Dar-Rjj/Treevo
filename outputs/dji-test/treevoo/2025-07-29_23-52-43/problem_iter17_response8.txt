import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.75).shift(1) * 1.5
    df['spike_detected'] = (df['volume_change'] > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_momentum'] / df['price_range'])
    
    # Moving Average of High-Low Differences (10-day period)
    df['ma_high_low_diff'] = df['price_range'].rolling(window=10).mean()
    
    # Trend Strength
    df['trend_strength'] = df['price_range'].rolling(window=21).max() - df['price_range'].rolling(window=21).min()
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / (df['close'] - df['open']).replace(0, np.nan)
    
    # Volume Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * df['volume_increase_ratio']
    
    # Extended Composite Factor
    df['extended_composite_factor'] = (
        df['intermediate_factor'] + 
        df['ma_high_low_diff'] + 
        df['trend_strength'] + 
        df['intraday_volatility'] + 
        df['volume_weighted_intraday_return']
    )
    
    # Final Composite Factor
    df['final_composite_factor'] = (
        df['extended_composite_factor'] * 
        df['spike_detected'] * 
        df['volume_increase_ratio']
    )
    
    return df['final_composite_factor'].dropna()
