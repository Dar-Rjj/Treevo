```python
import pandas as pd

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean().shift(1)
    avg_volume = df['volume'].rolling(window=6).mean().shift(1)
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=7).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low'] + (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = df['volume'].pct_change().rolling(window=20).quantile(0.8).shift(1) * 1.7
    volume_spike = (volume_change > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    high_low_ma = high_low_diff.rolling(window=10).mean()
    
    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * (df['close'] - avg_close) * price_volatility)
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility
    
    # Final Composite Factor
    final_composite_factor = (
        intermediate_factor + 
        high_low_ma + 
        close_momentum + 
        price_volatility + 
        combined_metrics + 
        adjusted_combined_metrics
    )
    
    return final_composite_factor
```
