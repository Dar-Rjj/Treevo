```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['previous_close'] = df['close'].shift(1)
    close_to_previous_close_return = (df['close'] - df['previous_close']) / df['previous_close']
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Volume Increase Ratio
    df['previous_volume'] = df['volume'].shift(1)
    volume_increase_ratio = df['volume'] / df['previous_volume']
    
    # Close Price Momentum
    avg_close_5_days = df['close'].rolling(window=5).mean().shift(1)
    close_price_momentum = df['close'] - avg_close_5_days
    
    # Combine Metrics
    combined_metrics = (
        intraday_return + 
        close_to_previous_close_return + 
        breakout_intensity + 
        volume_increase_ratio + 
        close_price_momentum
    )
    
    # Adjust for Price Range
    price_range = df['high'] - df['low']
    adjusted_combined_metrics = combined_metrics / price_range
    
    return adjusted_combined_metrics
```
