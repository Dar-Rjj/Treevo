```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    volume_weighted_breakout_intensity = breakout_intensity * df['volume']

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Identify Volume Spikes
    daily_volume_change = df['volume'].diff()
    past_21_days_volume_changes = daily_volume_change.rolling(window=21).apply(np.percentile, args=(75,), raw=True)
    spike_threshold = past_21_days_volume_changes * 1.5
    spike_condition = daily_volume_change > spike_threshold

    # Filter with Spike Condition
    volume_weighted_intraday_return = intraday_return * df['volume']
    volume_weighted_intraday_return[~spike_condition] = 0

    # Calculate Daily High-Low Difference
    high_low_difference = df['high'] - df['low']
    moving_sum_high_low_diff = high_low_difference.rolling(window=10).sum()

    # Calculate Volume-Adjusted Momentum (VAM)
    sma_close_prices = df['close'].rolling(window=20).mean()
    momentum = sma_close_prices.pct_change(5)
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = momentum * (df['volume'] / avg_volume)

    # Calculate Price Volatility Indicator (PVI)
    true_range = np.maximum.reduce([df['high'] - df['low'], 
                                    abs(df['high'] - df['close'].shift(1)), 
                                    abs(df['low'] - df['close'].shift(1))])
    atr = true_range.rolling(window=14).mean()
    pvi = atr.ewm(span=7, adjust=False).mean()

    # Combine Factors
    combined_factor = (volume_weighted_breakout_intensity * volume_adjusted_momentum) / pvi
    combined_factor += volume_weighted_intraday_return
    combined_factor += moving_sum_high_low_diff

    return combined_factor
```
