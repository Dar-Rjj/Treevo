```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'].shift(1) - df['close']) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_returns = (intraday_return + close_to_prev_close_return + breakout_intensity) * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    adjusted_combined_metrics = combined_returns / rolling_volatility
    
    # Incorporate Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    medium_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Combine Momentum with Adjusted Metrics
    final_combined_metrics = adjusted_combined_metrics + short_term_momentum + medium_term_momentum + long_term_momentum
    
    # Enhance Weighting with Trade Activity
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    trade_activity_weight = df['volume'] / avg_volume_10_days
    final_combined_metrics *= trade_activity_weight
    
    return final_combined_metrics
```
