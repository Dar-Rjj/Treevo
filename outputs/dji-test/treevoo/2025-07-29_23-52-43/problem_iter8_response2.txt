```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range Percentage
    df['intraday_high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Confirm Momentum with Volume
    df['volume_weighted_momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=20, min_periods=1).max())
    df['combined_factor'] = df['intraday_high_low_range'] * df['volume_weighted_momentum']
    
    # Adjust for Price Trend
    df['short_term_trend'] = df['close'].rolling(window=5, min_periods=1).mean()
    df['long_term_trend'] = df['close'].rolling(window=20, min_periods=1).mean()
    trend_diff = df['long_term_trend'] - df['short_term_trend']
    df['combined_factor'] += trend_diff
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1).fillna(1)
    
    # Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(5).rolling(window=5, min_periods=1).mean().fillna(0)
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=22, min_periods=1).quantile(0.8) * 2.0
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['volume_weighted_breakout'] = df['breakout_intensity'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['final_factor'] = (
        df['combined_factor'] + 
        (df['volume_increase_ratio'] * df['close_price_momentum'] / df['price_range']).fillna(0) + 
        df['volume_weighted_intraday_return'] + 
        df['volume_weighted_breakout']
    )
    df['final_factor'] = df['final_factor'] * df['spike_condition'].astype(int)
    
    return df['final_factor']
```
