```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    df['high_low_diff_10d_sum'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Calculate Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high_low_diff'].shift(1)
    df['intraday_momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Compute Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    df['vwap_diff'] = df['daily_vwap'] - df['prev_day_vwap']
    
    # Combine Intraday Momentum and VWAP
    df['final_combined_indicator'] = df['intraday_momentum_diff'] * df['vwap_diff']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['high_low_diff_10d_sum'] + df['final_combined_indicator']
    
    return df['alpha_factor']
```
