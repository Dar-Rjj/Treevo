import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) / df['open']
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_intraday_return *= (df['volume'] / avg_volume.shift(1))
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.8) * 1.6
    volume_spike = (daily_volume_change > spike_threshold).astype(float)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * (df['close'] - df['close'].rolling(window=5).mean().shift(1)) * price_volatility)
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=5).std()
    combined_metrics = intermediate_factor / rolling_volatility
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_returns = intraday_return + close_to_prev_close_return + breakout_intensity
    combined_returns_weighted = combined_returns * df['volume'] * combined_metrics
    
    # Calculate Daily High-Low Difference
    daily_high_low_diff = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    moving_sum_high_low_diff = daily_high_low_diff.rolling(window=14).sum()
    
    # Compute Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    prev_day_momentum = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    prev_day_vwap = vwap.shift(1)
    
    # Combine Intraday Momentum and VWAP
    momentum_difference = intraday_momentum - prev_day_momentum
    vwap_difference = vwap - prev_day_vwap
    final_combined_indicator = momentum_difference * vwap_difference
    
    # Adjust for Momentum
    rolling_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    combined_metrics_adjusted = combined_metrics / rolling_momentum
    
    # Generate Crossover Alpha Factor
    crossover_alpha_factor = combined_metrics_adjusted * final_combined_indicator * intermediate_factor
    
    return crossover_alpha_factor.dropna()
