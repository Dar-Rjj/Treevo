```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=5).mean()
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.75) * 1.5
    volume_spike = (daily_volume_change > spike_threshold).astype(int)
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    ma_high_low_diff = high_low_diff.rolling(window=10).mean()
    
    # Intermediate Factor
    intermediate_factor = volume_increase_ratio * (df['close'] - df['close'].rolling(window=5).mean())
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_previous_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_previous_close_return + breakout_intensity
    
    # Final Composite Factor
    final_composite_factor = (
        intermediate_factor +
        ma_high_low_diff +
        close_momentum +
        price_volatility +
        combined_metrics
    )
    
    return final_composite_factor
```
