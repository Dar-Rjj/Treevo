```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = breakout_intensity.where(breakout_intensity > 0, 0)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    df['combined_metrics'] = combined_metrics * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    df['adjusted_metrics'] = df['combined_metrics'] / rolling_volatility
    
    # Incorporate Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['momentum_adjusted_metrics'] = df['adjusted_metrics'] + short_term_momentum + long_term_momentum
    
    # Enhance Weighting with Trade Activity
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    df['trade_activity_weighted_metrics'] = (df['volume'] / avg_volume_10_days) * df['momentum_adjusted_metrics']
    
    # Generate Crossover Alpha Factor
    df['daily_high_low_diff'] = df['high'] - df['low']
    moving_sum_high_low_diff = df['daily_high_low_diff'].rolling(window=21).sum()
    
    intraday_high_low_spread = df['high'] - df['low']
    prev_day_high_low_spread = (df['high'] - df['low']).shift(1)
    
    daily_vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    prev_day_vwap = (df['close'].shift(1) * df['volume'].shift(1)).cumsum() / df['volume'].shift(1).cumsum()
    
    momentum_difference = intraday_high_low_spread - prev_day_high_low_spread
    vwap_difference = daily_vwap - prev_day_vwap
    
    final_combined_indicator = momentum_difference * vwap_difference
    
    rolling_momentum = (df['close'] - df['close'].shift(1)).rolling(window=20).sum()
    df['final_metrics'] = final_combined_indicator / rolling_momentum
    
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.85) * 1.5
    volume_spike_condition = daily_volume_change > spike_threshold
    
    volume_weighted_intraday_return = df['intraday_return'] * df['volume']
    intermediate_factor = (df['volume'] / df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=7).mean())
    
    crossover_alpha_factor = (df['final_metrics'] + intermediate_factor) * volume_weighted_intraday_return * volume_spike_condition
    
    return crossover_alpha_factor
```
