```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_diff'] = df['vwap'] - df['prev_day_vwap']
    df['intermediate_alpha_factor_1'] = df['momentum_diff'] * df['vwap_diff']
    
    # Calculate Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Close Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Adjust for Price Volatility
    df['price_range'] = df['high'] - df['low']
    
    # Combine All Factors
    df['final_factor'] = (df['intermediate_alpha_factor_1'] * df['volume_increase_ratio'] * df['close_price_change']) / df['price_range']
    
    return df['final_factor']
```
