```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences over a 10-day period
    df['moving_sum_high_low_diff'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Find the 20-day price range (max High - min Low)
    df['max_high_20d'] = df['high'].rolling(window=20).max()
    df['min_low_20d'] = df['low'].rolling(window=20).min()
    df['price_range_20d'] = df['max_high_20d'] - df['min_low_20d']
    
    # Normalize by Price Range
    df['alpha_factor'] = df['moving_sum_high_low_diff'] / df['price_range_20d']
    
    return df['alpha_factor']
```
