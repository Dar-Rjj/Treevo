```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume' indexed by 'date'.
    
    Returns:
    pd.Series: A series of the alpha factor values, indexed by 'date'.
    """
    
    # Calculate the intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the normalized volume
    df['normalized_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the price momentum as the percentage change in close price over 10 days
    df['momentum'] = df['close'].pct_change(periods=10)
    
    # Calculate the volatility of the close price over a 20-day window
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Combine the factors to create the alpha factor
    df['alpha_factor'] = (df['momentum'] * df['normalized_volume']) / (df['intraday_range'] + df['volatility'])
    
    # Drop rows with NaN values resulting from rolling calculations
    df.dropna(inplace=True)
    
    return df['alpha_factor']
```
