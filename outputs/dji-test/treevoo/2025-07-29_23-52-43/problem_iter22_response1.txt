```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_90th_percentile = df['volume_change'].rolling(window=21).quantile(0.90)
    df['volume_spike'] = (df['volume_change'] > 2.0 * volume_90th_percentile).astype(int)
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Intermediate Factor
    df['intermediate_factor'] = df['volume_increase_ratio'] * df['close_momentum']
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Final Composite Factor
    df['final_composite_factor'] = df['intermediate_factor'] / df['price_range']
    
    # Drop intermediate columns if necessary
    # df.drop(columns=['intraday_return', 'volume_change', 'volume_spike', 'volume_increase_ratio', 'close_momentum', 'price_range'], inplace=True)
    
    return df['final_composite_factor']
```
