```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Daily Trading Range
    daily_trading_range = df['high'] - df['low']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Volume-Weighted Close-to-PreviousClose Return
    volume_weighted_close_to_prev_close_return = df['volume'] * close_to_prev_close_return
    
    # Sum of Volume-Weighted Returns
    sum_volume_weighted_returns = volume_weighted_intraday_return + volume_weighted_close_to_prev_close_return
    
    # Rolling Volatility
    rolling_volatility = daily_trading_range.rolling(window=20).std()
    
    # Adjust for Volatility
    combined_return = sum_volume_weighted_returns / rolling_volatility
    
    return combined_return
```
