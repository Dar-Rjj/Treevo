```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Daily VWAP
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Compute Intraday Open-Close Spread
    df['intraday_open_close_spread'] = df['close'] - df['open']
    
    # Compute Previous Day's Open-Close Spread
    df['prev_day_open_close_spread'] = (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['volume_weighted_breakout_intensity'] = df['volume'] * df['breakout_intensity']
    
    # Combine Intraday Momentum, VWAP, and Open-Close Spread
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_difference'] = df['daily_vwap'] - df['prev_day_vwap']
    df['open_close_spread_difference'] = df['intraday_open_close_spread'] - df['prev_day_open_close_spread']
    
    df['final_alpha_factor'] = (df['momentum_difference'] * df['vwap_difference']) + df['open_close_spread_difference'] + df['volume_weighted_breakout_intensity']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    df['combined_metrics'] = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    df['weighted_combined_metrics'] = df['volume'] * df['combined_metrics']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['rolling_volatility'].rolling(window=20).std()
    df['adjusted_combined_metrics'] = df['weighted_combined_metrics'] / df['rolling_volatility']
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum() - df['daily_log_return']
    
    # Identify Volume Shock
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_shock'] = df['volume_change'].apply(lambda x: 1 if x > 1.5 else 0)
    
    # Combine All Factors
    df['alpha_factor'] = (df['cumulative_log_return'] * df['volume_shock']) + df['final_alpha_factor'] + df['adjusted_combined_metrics']
    
    return df['alpha_factor']
```
