```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.75).mul(1.5)
    df['volume_spike'] = df['volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['volume_spike']
    
    # Calculate Price Momentum
    df['1_week_price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['1_month_price_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Volatility
    df['1_week_volume_volatility'] = df['volume'].rolling(window=5).std()
    df['1_month_volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Combine All Factors
    combined_factor = (
        df['1_week_price_momentum'] * df['1_week_volume_volatility'] +
        df['1_month_price_momentum'] * df['1_month_volume_volatility'] +
        df['filtered_volume_weighted_intraday_return']
    )
    
    return combined_factor
```
