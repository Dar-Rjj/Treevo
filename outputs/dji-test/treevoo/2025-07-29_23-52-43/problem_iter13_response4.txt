```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    volume_weighted_breakout_intensity = breakout_intensity * df['volume']
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Volume Spikes
    daily_volume_change = df['volume'].pct_change()
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.75) * 2.0
    spike_condition = daily_volume_change > spike_threshold
    
    # Filter with Spike Condition
    volume_weighted_breakout_intensity_filtered = volume_weighted_breakout_intensity * spike_condition
    volume_weighted_intraday_return_filtered = volume_weighted_intraday_return * spike_condition
    
    # Moving Sum of High-Low Differences
    moving_sum_high_low = (df['high'] - df['low']).rolling(window=10).sum()
    
    # Volume-Adjusted Momentum (VAM)
    sma_close = df['close'].rolling(window=20).mean()
    momentum = sma_close.pct_change(periods=5)
    vam = momentum * df['volume']
    
    # Price Volatility Indicator (PVI)
    tr = df[['high', 'low', 'close']].shift().apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    atr = tr.rolling(window=14).mean()
    pvi = tr / atr
    pvi_smoothed = pvi.ewm(span=7).mean()
    
    # Logarithmic Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Combine Factors
    combined_factor = (
        (volume_weighted_breakout_intensity_filtered * vam) / pvi_smoothed +
        volume_weighted_intraday_return_filtered +
        moving_sum_high_low +
        log_return
    )
    
    return combined_factor
```
