```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (14-day period)
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_intraday_high_low_spread'] = df['high_low_diff'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_daily_vwap'] = df['daily_vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_intraday_high_low_spread']
    df['vwap_diff'] = df['daily_vwap'] - df['prev_daily_vwap']
    df['combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Identify Volume Spikes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['daily_volume_change'].rolling(window=21).quantile(0.8) * 1.6
    df['spike_condition'] = df['daily_volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Generate Crossover High-Low Difference Momentum and Volume-Weighted Intraday Return
    df['crossover_momentum_factor'] = df['sum_high_low_diff'] * df['filtered_volume_weighted_intraday_return']
    
    return df['crossover_momentum_factor']
```
