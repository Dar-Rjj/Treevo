```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day window)
    df['high_low_diff_10d_sum'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Compute Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    df['prev_day_intraday_volatility'] = df['intraday_volatility'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_shifted'] = df['vwap'].shift(1)
    
    # Calculate Intraday Volatility Ratio
    df['intraday_volatility_ratio'] = df['intraday_volatility'] / df['prev_day_intraday_volatility']
    
    # Calculate VWAP Ratio
    df['vwap_ratio'] = df['vwap'] / df['vwap_shifted']
    
    # Combine Intraday Volatility and VWAP Ratios
    df['combined_ratio'] = df['intraday_volatility_ratio'] * df['vwap_ratio']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['high_low_diff_10d_sum'] + df['combined_ratio']
    
    return df['alpha_factor']
```
