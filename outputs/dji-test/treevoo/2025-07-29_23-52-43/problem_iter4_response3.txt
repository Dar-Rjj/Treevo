```python
import pandas as pd

def heuristics_v2(df):
    # Volume Increase Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['Close_Momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Price Range
    df['Price_Range'] = df['high'] - df['low']
    
    # Combine Factors
    df['Alpha_Factor'] = (df['Volume_Ratio'] * df['Close_Momentum']) / df['Price_Range']
    
    return df['Alpha_Factor'].dropna()
```
