import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the 20-day moving average of VWAP
    df['vwap_20_ma'] = df['vwap'].rolling(window=20).mean()
    
    # Calculate the 50-day moving average of VWAP
    df['vwap_50_ma'] = df['vwap'].rolling(window=50).mean()
    
    # Calculate the ratio of 20-day to 50-day VWAP
    df['vwap_ratio'] = df['vwap_20_ma'] / df['vwap_50_ma']
    
    # Calculate the 10-day standard deviation of close price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of close price
    df['close_30_std'] = df['close'].rolling(window=30).std()
    
    # Calculate the ratio of 10-day to 30-day standard deviation
    df['std_ratio'] = df['close_10_std'] / df['close_30_std']
    
    # Final alpha factor: combination of VWAP ratio and standard deviation ratio
    df['alpha_factor'] = (df['vwap_ratio'] - 1) * (df['std_ratio'] - 1)
    
    return df['alpha_factor']
