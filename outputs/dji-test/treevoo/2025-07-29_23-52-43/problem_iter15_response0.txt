```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = (df['volume'] / df['volume'].shift(1)).rolling(window=21).quantile(0.90).mul(2.0)
    volume_spike = daily_volume_change > spike_threshold
    
    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].shift(5).rolling(window=5).mean()
    
    # Price Range
    price_range = df['high'] - df['low']
    
    # Intermediate Factor
    intermediate_factor = (volume_increase_ratio * close_price_momentum) / price_range
    
    # Moving Average of High-Low Differences (20-day period)
    moving_avg_high_low_diff = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Final Composite Factor
    final_composite_factor = intermediate_factor * moving_avg_high_low_diff
    
    return final_composite_factor
```
