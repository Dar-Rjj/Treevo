```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity[breakout_intensity < 0] = 0
    
    # Combine Metrics, Weight by Volume
    combined_metrics = df['volume'] * (intraday_return + close_to_prev_close_return + breakout_intensity)
    
    # Calculate Rolling Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    
    # Adjust for Volatility
    adjusted_metrics = combined_metrics / rolling_volatility
    
    return adjusted_metrics
```
