```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_prev_close_return'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    df['combined_metrics'] = (df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']) * df['volume']
    
    # Adjust for Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['rolling_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['adjusted_combined_metrics'] = df['combined_metrics'] / df['rolling_volatility']
    
    # Incorporate Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['momentum_adjusted_metrics'] = (df['adjusted_combined_metrics'] + df['short_term_momentum'] + df['long_term_momentum'])
    
    # Calculate Daily High-Low Difference
    df['daily_high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['daily_high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday Momentum
    df['intraday_spread'] = df['high'] - df['low']
    df['prev_intraday_spread'] = df['intraday_spread'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_spread'] - df['prev_intraday_spread']
    df['vwap_diff'] = df['vwap'] - df['prev_vwap']
    df['final_combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['spike_threshold'] = df['volume_change'].rolling(window=21).quantile(0.8) * 1.6
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['spike_condition'] = df['volume_change'] > df['spike_threshold']
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['momentum_adjusted_metrics'] + df['final_combined_indicator'] + df['filtered_volume_weighted_intraday_return']
    
    return df['crossover_alpha_factor']
```
