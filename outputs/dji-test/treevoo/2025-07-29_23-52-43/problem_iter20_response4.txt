import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'].pct_change()
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.9) * 2.0
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Volume Increase Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Intermediate Factor
    df['intermediate_factor'] = df['volume_ratio'] * df['close_momentum']
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Final Composite Factor
    df['final_composite_factor'] = df['intermediate_factor'] + df['price_range']
    
    # Calculate Daily High-Low Range Percentage
    df['daily_high_low_range_percentage'] = (df['high'] - df['low']) / df['open']
    
    # Confirm Momentum with Volume-Weighted Difference
    df['volume_weighted_intraday_momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].max())
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['combined_intraday_indicator'] = df['daily_high_low_range_percentage'] * df['volume_weighted_intraday_momentum']
    
    # Momentum Difference
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['intraday_high_low_spread'].shift(1)
    
    # VWAP Difference
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_difference'] = df['vwap'] - df['vwap'].shift(1)
    
    # Open-Close Spread Difference
    df['open_close_spread_difference'] = (df['close'] - df['open']) - (df['close'].shift(1) - df['open'].shift(1))
    
    # Final Combined Indicator
    df['final_combined_indicator'] = df['momentum_difference'] + df['vwap_difference'] + df['open_close_spread_difference']
    
    # Adjust for Price Trend
    df['short_term_trend'] = df['close'].rolling(window=5).mean()
    df['long_term_trend'] = df['close'].rolling(window=20).mean()
    df['trend_difference'] = df['long_term_trend'] - df['short_term_trend']
    df['adjusted_combined_indicator'] = df['final_combined_indicator'] * df['trend_difference']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility_std'] = df['rolling_volatility'].rolling(window=21).std()
    df['volatility_adjusted_indicator'] = df['adjusted_combined_indicator'] / df['rolling_volatility_std']
    
    # Cumulative Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['cumulative_log_return'] = df['log_return'].rolling(window=20).sum()
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Introduce New Factor: Intraday Price Efficiency
    df['intraday_efficiency_ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['efficiency_adjusted_indicator'] = df['volatility_adjusted_indicator'] * df['intraday_efficiency_ratio']
    
    # Combine All Factors
    df['final_alpha_factor'] = df['cumulative_log_return'] + df['volume_shock'] + df['efficiency_adjusted_indicator']
    
    return df['final_alpha_factor']
