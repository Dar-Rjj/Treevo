```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['Prev_Day_High_Low_Spread'] = df['Intraday_High_Low_Spread'].shift(1)
    
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Previous Day's VWAP
    df['Prev_Day_VWAP'] = df['Daily_VWAP'].shift(1)
    
    # Compute Intraday Open-Close Spread
    df['Intraday_Open_Close_Spread'] = df['close'] - df['open']
    
    # Compute Previous Day's Open-Close Spread
    df['Prev_Day_Open_Close_Spread'] = df['Intraday_Open_Close_Spread'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['Breakout_Intensity'] = (df['high'] - df['low']) / df['close']
    df['Breakout_Intensity'] = df['Breakout_Intensity'].apply(lambda x: max(x, 0))
    
    # Weight by Volume
    df['Volume_Weighted_Breakout_Intensity'] = df['Breakout_Intensity'] * df['volume']
    
    # Combine Intraday Momentum, VWAP, and Open-Close Spread
    df['Momentum_Difference'] = df['Intraday_High_Low_Spread'] - df['Prev_Day_High_Low_Spread']
    df['VWAP_Difference'] = df['Daily_VWAP'] - df['Prev_Day_VWAP']
    df['Open_Close_Spread_Difference'] = df['Intraday_Open_Close_Spread'] - df['Prev_Day_Open_Close_Spread']
    
    df['Final_Alpha_Factor'] = (df['Momentum_Difference'] * df['VWAP_Difference'] + 
                                df['Open_Close_Spread_Difference'] + 
                                df['Volume_Weighted_Breakout_Intensity'])
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Cumulative Log Return
    df['Cumulative_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).sum().shift(1)
    
    # Identify Volume Shock
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Shock'] = df['Volume_Change'].apply(lambda x: 1 if x > 1.5 else 0)
    
    # Combine All Factors
    df['Alpha_Factor'] = (df['Cumulative_Log_Return'] * df['Volume_Shock'] + 
                          df['Final_Alpha_Factor'])
    
    return df['Alpha_Factor']
```
