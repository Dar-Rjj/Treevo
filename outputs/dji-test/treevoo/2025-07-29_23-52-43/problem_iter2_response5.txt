import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.75).iloc[-1] * 1.5
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences
    df['moving_sum_high_low_diff'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Combine Factors
    df['crossover_momentum'] = df['filtered_volume_weighted_intraday_return']
    df['volume_weighted_high_low_diff'] = df['moving_sum_high_low_diff']
    df['factor'] = df['crossover_momentum'] + df['volume_weighted_high_low_diff']
    
    return df['factor'].dropna()
