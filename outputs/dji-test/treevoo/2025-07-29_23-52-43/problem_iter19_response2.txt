```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate 14-day Moving Sum of High-Low Differences
    df['hl_diff_14_sum'] = df['high_low_diff'].rolling(window=14).sum()
    
    # Compute Intraday High-Low Spread
    df['intraday_hl_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Spread
    df['prev_day_hl_spread'] = df['high_low_diff'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_hl_spread'] - df['prev_day_hl_spread']
    df['vwap_diff'] = df['daily_vwap'] - df['prev_day_vwap']
    df['combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Identify Volume Spikes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['daily_volume_change'].rolling(window=21).quantile(0.8) * 1.6
    df['spike_condition'] = df['daily_volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['vwap_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_vwap_intraday_return'] = df['vwap_intraday_return'] * df['spike_condition']
    
    # Generate Crossover Alpha Factor
    df['crossover_alpha_factor'] = df['hl_diff_14_sum'] * df['combined_indicator'] * df['filtered_vwap_intraday_return']
    
    return df['crossover_alpha_factor']
```
