```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Confirm Momentum with Volume
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=5).max())
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    combined_intraday_factor = intraday_range * volume_weighted_momentum
    
    # Adjust for Price Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    long_term_trend = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between Short-Term and Long-Term Price Trends
    trend_difference = short_term_trend - long_term_trend
    
    # Adjust Combined Intraday Factor by Price Trends
    final_factor = combined_intraday_factor * (1 + trend_difference)
    
    return final_factor
```
