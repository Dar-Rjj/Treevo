```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on historical price and volume data.
    
    The factor is based on the following heuristics:
    1. Calculate the 20-day moving average of the close price.
    2. Calculate the 20-day exponential moving average of the volume.
    3. Compute the percentage change in the 20-day moving average of the close price.
    4. Compute the percentage change in the 20-day exponential moving average of the volume.
    5. The alpha factor is the product of the two percentage changes.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # 20-day moving average of the close price
    ma_close = df['close'].rolling(window=20).mean()
    
    # 20-day exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Percentage change in the 20-day moving average of the close price
    pct_change_ma_close = ma_close.pct_change()
    
    # Percentage change in the 20-day exponential moving average of the volume
    pct_change_ema_volume = ema_volume.pct_change()
    
    # Alpha factor is the product of the two percentage changes
    alpha_factor = pct_change_ma_close * pct_change_ema_volume
    
    return alpha_factor
```
