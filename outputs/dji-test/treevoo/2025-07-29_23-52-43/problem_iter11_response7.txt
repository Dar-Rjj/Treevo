```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on (open, high, low, close) prices and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volume weighted price
    vwp = (df['close'] * df['volume']).rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day moving average of the close price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (daily_range / df['close']) + daily_return + (vwp - df['close']) / df['close'] + (ma_20 - df['close']) / df['close'] + (rsi - 50) / 50
    
    return alpha_factor
```
