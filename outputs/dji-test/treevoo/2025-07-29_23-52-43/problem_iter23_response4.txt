```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=6).mean()
    close_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close'].shift(1)
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    ma_high_low_diff = high_low_diff.rolling(window=10).mean()
    
    # Intermediate Factor
    intermediate_factor = ((df['volume'] / df['volume'].shift(2)) * 
                            (df['close'] - df['close'].rolling(window=6).mean()))
    
    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_returns = intraday_return + close_to_prev_close_return + breakout_intensity
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * (df['high'] - df['low']) / df['low']
    
    # Combined Metrics
    combined_metrics = (volume_weighted_intraday_return + 
                        df['volume'] * (close_to_prev_close_return + breakout_intensity))
    
    # Adjust for Recent Volatility
    recent_volatility = (df['close'] - df['open']).rolling(window=10).std()
    adjusted_combined_metrics = combined_metrics / recent_volatility
    
    # Final Composite Factor
    final_factor = (intermediate_factor + ma_high_low_diff + 
                    close_momentum + price_volatility + adjusted_combined_metrics)
    
    return final_factor
```
