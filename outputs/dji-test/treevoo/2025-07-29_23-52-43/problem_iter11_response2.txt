```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences (10-day period)
    df['sum_high_low_diff_10d'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Compute Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['prev_day_vwap'] = df['daily_vwap'].shift(1)
    
    # Combine Intraday Momentum and VWAP
    df['momentum_diff'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    df['vwap_diff'] = df['daily_vwap'] - df['prev_day_vwap']
    df['combined_indicator'] = df['momentum_diff'] * df['vwap_diff']
    
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = 1.5 * df['volume_change'].rolling(window=21).quantile(0.75)
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_momentum']) / df['price_range']
    
    # Synthesize Intermediate Alpha Factor
    df['synthesized_intermediate_alpha'] = df['sum_high_low_diff_10d'] + df['combined_indicator']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Filter with Spike Condition
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Generate Crossover High-Low Difference Momentum and Volume-Weighted Intraday Return
    df['alpha_factor'] = df['synthesized_intermediate_alpha'] * df['filtered_volume_weighted_intraday_return']
    
    return df['alpha_factor']
```
