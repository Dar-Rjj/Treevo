```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_change_21 = df['volume_change'].rolling(window=21).quantile(0.75)
    spike_threshold = 1.5 * volume_change_21
    df['spike_condition'] = df['volume_change'] > spike_threshold
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['filtered_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'] * df['spike_condition']
    
    # Calculate Price Momentum
    df['1_week_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['1_month_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Volatility
    df['1_week_volatility'] = df['volume'].rolling(window=6).std()
    df['1_month_volatility'] = df['volume'].rolling(window=21).std()
    
    # Combine Momentum and Volatility
    df['combined_1_week'] = df['1_week_momentum'] * df['1_week_volatility']
    df['combined_1_month'] = df['1_month_momentum'] * df['1_month_volatility']
    
    # Synthesize Factors
    alpha_factor = (
        df['filtered_volume_weighted_intraday_return'] + 
        df['combined_1_week'] + 
        df['combined_1_month']
    )
    
    return alpha_factor
```
