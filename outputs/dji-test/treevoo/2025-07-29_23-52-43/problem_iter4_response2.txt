import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Differences over a 10-day period
    df['high_low_sum'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Divide by the Close Price on the current day
    df['factor'] = df['high_low_sum'] / df['close']
    
    # Return the factor as a Series
    return df['factor']
