```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['daily_high_low_range_pct'] = (df['high'] - df['low']) / df['open']
    
    # Volume-Weighted Intraday Momentum
    df['volume_weighted_momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].max())
    
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_intraday_high_low_spread'] = df['intraday_high_low_spread'].shift(1)
    
    # Daily VWAP
    df['vwap'] = (df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    df['prev_vwap'] = df['vwap'].shift(1)
    
    # Combine Intraday High-Low Range and Volume-Weighted Intraday Momentum
    df['combined_indicator'] = df['daily_high_low_range_pct'] * df['volume_weighted_momentum']
    
    # Momentum Difference
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_intraday_high_low_spread']
    
    # VWAP Difference
    df['vwap_difference'] = df['vwap'] - df['prev_vwap']
    
    # Final Combined Indicator
    df['final_combined_indicator'] = df['momentum_difference'] * df['vwap_difference']
    
    # Adjust for Price Trend
    df['short_term_price_trend'] = df['close'].rolling(window=5).mean()
    df['long_term_price_trend'] = df['close'].rolling(window=20).mean()
    df['price_trend_adjustment'] = df['long_term_price_trend'] - df['short_term_price_trend']
    
    # Adjust Combined Indicator by Price Trends
    df['adjusted_combined_indicator'] = df['final_combined_indicator'] + df['price_trend_adjustment']
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        (df['daily_high_low_range_pct'] * df['volume_weighted_momentum'] * df['price_trend_adjustment'] + 
         df['final_combined_indicator']) + 
        (df['volume_increase_ratio'] * df['close_price_momentum'] / df['price_range'])
    )
    
    return df['alpha_factor']
```
