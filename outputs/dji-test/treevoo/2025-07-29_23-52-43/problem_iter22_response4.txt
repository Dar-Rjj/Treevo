```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Open-Close Log Return Ratio
    df['open_close_log_return_ratio'] = (np.log(df['close']) - np.log(df['open'])) / (np.log(df['close']) + np.log(df['open']))
    
    # High-Low Log Return Difference
    df['high_low_log_return_diff'] = np.log(df['high']) - np.log(df['low'])
    
    # Simple Moving Averages (SMA)
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Averages (EMA)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # True Range and Average True Range (ATR)
    df['true_range'] = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Historical Volatility
    df['daily_volatility'] = df['log_return'].rolling(window=30).std() * np.sqrt(252)
    
    # On-Balance Volume (OBV)
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['cmf_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    positive_money_flow = (typical_price > typical_price.shift(1)) * typical_price * df['volume']
    negative_money_flow = (typical_price < typical_price.shift(1)) * typical_price * df['volume']
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / negative_money_flow_sum
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (
        df['log_return'] + 
        df['open_close_log_return_ratio'] + 
        df['high_low_log_return_diff'] + 
        (df['close'] - df['sma_5']) / df['sma_5'] + 
        (df['close'] - df['sma_20']) / df['sma_20'] + 
        (df['ema_12'] - df['ema_26']) / df['ema_26'] + 
        df['atr_14'] + 
        df['daily_volatility'] + 
        df['obv'] + 
        df['cmf_20'] + 
        df['mfi_14'] + 
        (df['close'] - df['vwap']) / df['vwap']
    )
    
    return df['alpha_factor']
```
