import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided
    (open, high, low, close) prices, amount, and volume. The factor is designed to help predict
    future stock returns.
    
    Parameters:
    df (pd.DataFrame): A pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A pandas Series representing the factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['close'].shift(1)) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mf_multiplier * df['volume']
    
    # Calculate the 14-day positive and negative money flow
    n_days = 14
    positive_mf = mf_volume.where(mf_multiplier > 0, 0).rolling(window=n_days).sum()
    negative_mf = mf_volume.where(mf_multiplier < 0, 0).rolling(window=n_days).sum()
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_mf / negative_mf.abs()
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Calculate the 5-day moving average of the Money Flow Index
    mfi_5d_ma = mfi.rolling(window=5).mean()
    
    # Calculate the 20-day simple moving average of the close price
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the close price
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate the Z-score of the close price
    z_score = (df['close'] - sma_20) / std_20
    
    # Calculate the final alpha factor
    alpha_factor = mfi_5d_ma * z_score
    
    return alpha_factor
