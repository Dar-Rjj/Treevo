```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    df['close_to_previous_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = df['breakout_intensity'].apply(lambda x: max(x, 0))
    
    # Combine Intraday and Close-to-PreviousClose Returns with Breakout Intensity
    df['combined_return'] = df['intraday_return'] + df['close_to_previous_close_return'] + df['breakout_intensity']
    
    # Weight by Volume
    df['volume_weighted_combined'] = df['volume'] * df['combined_return']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['high'] - df['low']
    df['rolling_volatility'] = df['rolling_volatility'].rolling(window=20).std()
    df['volatility_adjusted'] = df['volume_weighted_combined'] / df['rolling_volatility']
    
    # Incorporate Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['medium_term_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Combine Weighted and Adjusted Metrics with Momentum
    df['alpha_factor'] = (df['volatility_adjusted'] + 
                          df['short_term_momentum'] + 
                          df['medium_term_momentum'] + 
                          df['long_term_momentum'])
    
    return df['alpha_factor']
```
