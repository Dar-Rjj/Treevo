```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range Percentage
    df['daily_high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Confirm Momentum with Volume-Weighted Difference
    df['volume_weighted_momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=20).max())
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_high_low_spread'] = df['high'].shift(1) - df['low'].shift(1)
    df['combined_indicator'] = df['daily_high_low_range'] * df['volume_weighted_momentum']
    df['momentum_difference'] = df['intraday_high_low_spread'] - df['prev_day_high_low_spread']
    
    # Adjust for Price Trend
    df['short_term_trend'] = df['close'].rolling(window=5).mean()
    df['long_term_trend'] = df['close'].rolling(window=20).mean()
    trend_difference = df['long_term_trend'] - df['short_term_trend']
    df['combined_indicator'] += np.where(trend_difference > 0, 1, -1) * df['combined_indicator']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['intraday_high_low_spread'].rolling(window=20).std()
    df['combined_indicator'] /= df['rolling_volatility']
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    spike_threshold = df['volume_change'].rolling(window=21).quantile(0.9) * 2.0
    df['spike_condition'] = np.where(df['volume_change'] > spike_threshold, 1, 0)
    
    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Close Price Momentum
    df['close_price_momentum'] = df['close'] - df['close'].shift(5).rolling(window=5).mean()
    
    # Price Volatility
    df['price_volatility'] = df['close'].rolling(window=5).std() / df['close'].shift(1)
    
    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Close-to-PreviousClose Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # High-Low Breakout Intensity
    df['breakout_intensity'] = (df['high'] - df['low']) / df['close']
    df['breakout_intensity'] = np.where(df['breakout_intensity'] < 0, 0, df['breakout_intensity'])
    
    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * 
                                 df['close_price_momentum'] * 
                                 df['price_volatility'])
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    df['combined_returns'] = df['intraday_return'] + df['close_to_prev_close_return'] + df['breakout_intensity']
    
    # Final Composite Factor
    df['final_composite_factor'] = df['intermediate_factor'] + df['combined_returns']
    
    return df['final_composite_factor']
```
