```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Breakout Intensity
    breakout_intensity = (df['high'] - df['low']) / df['close']
    breakout_intensity = breakout_intensity.apply(lambda x: max(x, 0))
    
    # Combine Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    combined_metrics = intraday_return + close_to_prev_close_return + breakout_intensity
    combined_metrics_weighted = combined_metrics * df['volume']
    
    # Adjust for Volatility
    daily_range = df['high'] - df['low']
    rolling_volatility = daily_range.rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics_weighted / rolling_volatility
    
    return adjusted_combined_metrics
```
