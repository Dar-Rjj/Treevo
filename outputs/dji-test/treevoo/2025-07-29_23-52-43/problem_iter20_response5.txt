import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Weighted Close Price Momentum
    avg_close = df['close'].rolling(window=5).mean()
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_momentum = (df['close'] - avg_close) * (df['volume'] / avg_volume)

    # Price Volatility
    price_volatility = df['close'].rolling(window=6).std() / df['close'].shift(1)

    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Volume Spike Detection
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    spike_threshold = daily_volume_change.rolling(window=21).quantile(0.75) * 1.5
    volume_spike = daily_volume_change > spike_threshold

    # Volume Increase Ratio
    volume_increase_ratio = df['volume'] / df['volume'].shift(1)

    # High-Low Breakout Intensity
    high_low_breakout = (df['high'] - df['low']) / df['close']
    high_low_breakout[high_low_breakout < 0] = 0

    # Moving Average of High-Low Differences (10-day period)
    high_low_diff = df['high'] - df['low']
    moving_avg_high_low_diff = high_low_diff.rolling(window=10).mean()

    # Intermediate Factor
    intermediate_factor = (df['volume'] / df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=5).mean())

    # Combined Intraday, Close-to-PreviousClose Returns, and Breakout Intensity
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_metrics = intraday_return + close_to_prev_close_return + high_low_breakout

    # Calculate Intraday Return using High and Low prices
    intraday_return_high_low = (df['high'] - df['low']) / df['low']

    # Adjust for Volatility
    rolling_volatility = (df['high'] - df['low']).rolling(window=20).std()
    adjusted_combined_metrics = combined_metrics / rolling_volatility

    # Calculate Daily High-Low Difference
    daily_high_low_diff = df['high'] - df['low']

    # Calculate Moving Sum of High-Low Differences (14-day period)
    moving_sum_high_low_diff = daily_high_low_diff.rolling(window=14).sum()

    # Compute Intraday Momentum
    intraday_high_low_spread = df['high'] - df['low']
    prev_day_high_low_spread = (df['high'].shift(1) - df['low'].shift(1))
    intraday_momentum = intraday_high_low_spread - prev_day_high_low_spread

    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    prev_day_vwap = (df['close'].shift(1) * df['volume'].shift(1)).cumsum() / df['volume'].shift(1).cumsum()
    vwap_difference = daily_vwap - prev_day_vwap

    # Combine Intraday Momentum and VWAP
    final_combined_indicator = intraday_momentum * vwap_difference

    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']

    # Filter with Spike Condition
    filtered_final_combined_indicator = final_combined_indicator * volume_spike

    # Generate Crossover Alpha Factor
    crossover_alpha_factor = (adjusted_combined_metrics + filtered_final_combined_indicator) / 2

    return crossover_alpha_factor.dropna()
