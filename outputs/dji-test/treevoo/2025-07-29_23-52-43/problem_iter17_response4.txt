```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike_threshold'] = 2.0 * df['volume_change'].rolling(window=21).quantile(0.90)
    df['volume_spike'] = df['volume_change'] > df['volume_spike_threshold']

    # Volume Increase Ratio
    df['volume_increase_ratio'] = df['volume'] / df['volume'].shift(1)

    # Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)

    # Price Range
    df['price_range'] = df['high'] - df['low']

    # Intermediate Factor
    df['intermediate_factor'] = (df['volume_increase_ratio'] * df['close_momentum'] / df['price_range'])

    # Moving Average of High-Low Differences (20-day period)
    df['ma_high_low_diff'] = df['price_range'].rolling(window=20).mean()

    # Average Intraday Return (20-day period)
    df['average_intraday_return'] = df['intraday_return'].rolling(window=20).mean()

    # Final Composite Factor
    df['final_composite_factor'] = (df['intermediate_factor'] + 
                                     df['ma_high_low_diff'] + 
                                     df['average_intraday_return'])

    return df['final_composite_factor']
```
