```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Change in Intraday Price Range
    df['intraday_range_change'] = df['intraday_range'].diff()
    
    # Adjust for Volume Impact
    df['volume_adjustment'] = (df['volume'] - df['volume'].shift(1)) * df['intraday_range_change']
    
    # Aggregate Short-Term Momentum
    df['avg_intraday_range_change'] = df['volume_adjustment'].rolling(window=5).mean()
    
    # Combine Intraday Range and Volume Adjustment
    df['combined_intraday_momentum'] = df['intraday_range'] * df['avg_intraday_range_change']
    
    # Smoothing
    df['smoothed_intraday_momentum'] = df['combined_intraday_momentum'].ewm(span=5).mean()
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    df['short_ema_close'] = df['close'].ewm(span=5).mean()
    df['long_ema_close'] = df['close'].ewm(span=20).mean()
    
    # Determine Momentum
    df['momentum_signal'] = np.where(df['short_ema_close'] > df['long_ema_close'], 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    obv_changes = np.where(df['close'] > df['close'].shift(), df['volume'],
                           np.where(df['close'] < df['close'].shift(), -df['volume'], 0))
    df['obv'] = df['obv'] + obv_changes.cumsum()
    
    # Weight OBV by the Difference between Short and Long EMA
    df['ema_diff'] = np.abs(df['short_ema_close'] - df['long_ema_close'])
    df['weighted_obv'] = df['obv'] * df['ema_diff']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['smoothed_intraday_momentum'] * df['weighted_obv']
    
    return df['alpha_factor']
```
