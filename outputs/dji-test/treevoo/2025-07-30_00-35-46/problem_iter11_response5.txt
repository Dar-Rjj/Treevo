```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).median()) + 1
    
    # Combine Intraday Range and Volume Adjustment
    combined_metric = intraday_range * volume_adjustment
    
    # Smoothing with SMA
    smoothed_combined_metric = combined_metric.rolling(window=5).mean()
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Measure Volume Synchronization with Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    synchronized_product = volume_change * price_change
    synchronized_product_filtered = np.sign(synchronized_product.where(volume_ratio > volume_shock_threshold, 0))
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate MACD
    macd_line = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    momentum = (ema_short > ema_long) & (macd_line > signal_line)
    momentum_signal = momentum.astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    weight = np.abs(macd_line - signal_line)
    weighted_obv = obv * weight
    
    # Combine Momentum and OBV into a Factor
    momentum_obv_factor = weighted_obv * momentum_signal
    
    # Combine Intraday Metrics
    intraday_metrics = (intraday_range * intraday_momentum) / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + synchronized_product_filtered + momentum_obv_factor + intraday_metrics) * smoothed_combined_metric
    
    return final_alpha_factor
```
