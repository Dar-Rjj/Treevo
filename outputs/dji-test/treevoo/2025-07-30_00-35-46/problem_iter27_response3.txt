import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] / df['open']) - 1
    df['Intraday_Momentum'] = 0.5 * df['High_Low_Diff'] + 0.5 * df['Open_Close_Return']
    df['Non_Linear_Transformation'] = df['Intraday_Momentum'] ** 3

    # Analyze Price Trends and Momentum
    short_sma_period = 5
    long_sma_period = 20
    roc_period = 5
    rsi_period = 14
    short_ema_period = 5
    long_ema_period = 20

    df['SMA_Short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=long_sma_period).mean()
    df['SMA_Crossover'] = np.where(df['SMA_Short'] > df['SMA_Long'], 1, -1)

    df['ROC'] = df['close'].pct_change(periods=roc_period)
    
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=rsi_period).mean()
    avg_loss = loss.rolling(window=rsi_period).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))

    df['EMA_Short'] = df['close'].ewm(span=short_ema_period, adjust=False).mean()
    df['EMA_Long'] = df['close'].ewm(span=long_ema_period, adjust=False).mean()
    df['EMA_Crossover'] = np.where(df['EMA_Short'] > df['EMA_Long'], 1, -1)

    # Apply Integrated Volume and Amount Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Volume_Shock'] = np.where((df['Volume_Ratio'] > 2.0) & (df['Amount_Ratio'] > 2.0), 1, 0)

    # Measure Synchronized Volume and Price Changes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronized_Change'] = np.sign(df['Volume_Change'] * df['Price_Change'])
    df['Synchronized_Magnitude'] = np.abs(df['Volume_Change'] * df['Price_Change'])

    # Investigate Trading Volume
    df['Avg_Volume_Short'] = df['volume'].rolling(window=short_sma_period).mean()
    df['Avg_Volume_Long'] = df['volume'].rolling(window=long_sma_period).mean()
    df['Volume_Comparison'] = np.where(df['volume'] > df['Avg_Volume_Long'], 1, -1)

    # Combine Multiple Factors
    df['Composite_Factor'] = (
        0.3 * df['SMA_Crossover'] +
        0.2 * df['ROC'] +
        0.2 * (df['RSI'] - 50) / 50 +
        0.3 * df['EMA_Crossover'] +
        0.1 * df['Volume_Shock'] +
        0.1 * df['Synchronized_Change'] * (df['Synchronized_Magnitude'] / df['Synchronized_Magnitude'].mean()) +
        0.1 * df['Volume_Comparison']
    )

    return df['Composite_Factor']
