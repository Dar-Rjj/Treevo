import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    df['high_low_momentum'] = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    df['combined_momentum'] = (df['intraday_return'] + df['high_low_momentum']) / 2
    
    # Apply Volume-Weighted Adjustments for Trading Activity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['high_low_change'] = df['high'] - df['low']
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_ratio'] * df['amount_ratio'] * df['high_low_change']
    
    # Synthesize and Filter
    df['volume_shock_threshold'] = 1.5
    df['volume_shock'] = df['volume'] / df['volume'].shift(1).fillna(method='bfill')
    df['filtered_momentum'] = df['adjusted_combined_momentum'].where(df['volume_shock'] > df['volume_shock_threshold'], 0)
    df['smoothed_momentum'] = df['filtered_momentum'].ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Synchronize Volume Dynamics
    df['volume_change'] = df['volume'] - df['volume'].shift(1).fillna(method='bfill')
    df['price_change'] = df['close'] - df['close'].shift(1).fillna(method='bfill')
    df['synchronized_volume'] = df['volume_change'] * df['price_change']
    df['synchronized_sign'] = np.sign(df['synchronized_volume'])
    
    # EMA Cross Signal
    df['short_ema'] = df['close'].ewm(span=5).mean()
    df['long_ema'] = df['close'].ewm(span=20).mean()
    df['macd_line'] = df['short_ema'] - df['long_ema']
    df['signal_line'] = df['macd_line'].ewm(span=9).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['macd_line'] > df['signal_line'], 1, np.where(df['macd_line'] < df['signal_line'], -1, 0))
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    obv = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv += df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv -= df['volume'].iloc[i]
        df['obv'].iloc[i] = obv
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    df['obv_weight'] = np.abs(df['macd_line'] - df['signal_line'])
    df['weighted_obv'] = df['obv'] * df['obv_weight']
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    df['final_alpha_factor'] = df['smoothed_momentum'] * df['weighted_obv']
    
    return df['final_alpha_factor']
