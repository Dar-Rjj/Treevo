```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Range
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Price Change Percentage
    price_change_percentage = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Synchronize Changes
    synchronized_changes = (volume_change_ratio * price_change_percentage).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Combine the factors
    factor = intraday_range * synchronized_changes
    
    return factor
```
