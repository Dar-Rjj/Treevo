import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_return_10'] = df['return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the high-low price range
    df['std_high_low_10'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1).rolling(window=10).std()
    
    # Calculate the 10-day sum of the volume
    df['sum_volume_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the alpha factor as the ratio of the 10-day moving average of the return to the 10-day standard deviation of the high-low price range
    df['alpha_factor'] = df['ma_return_10'] / df['std_high_low_10']
    
    # Normalize the alpha factor by the 10-day sum of the volume
    df['normalized_alpha_factor'] = df['alpha_factor'] / df['sum_volume_10']
    
    # Return the normalized alpha factor
    return df['normalized_alpha_factor']
