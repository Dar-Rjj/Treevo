```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Assign weights to High-Low Difference and Open-Close Return
    weight_high_low = 0.6
    weight_open_close = 0.4
    
    # Combine High-Low and Open-Close Returns with a weighted average
    intraday_momentum = weight_high_low * high_low_diff + weight_open_close * open_close_return
    
    # Introduce non-linear transformation
    intraday_momentum = np.power(intraday_momentum, 2)
    
    # Apply Volume and Amount Adjustments
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Adjust for High-Low Spread Impact
    high_low_spread = df['high'] - df['low']
    change_in_high_low_spread = high_low_spread - high_low_spread.shift(1)
    
    # Synthesize Intraday Momentum with Adjustments
    adjusted_intraday_momentum = intraday_momentum * change_in_high_low_spread * volume_ratio * amount_ratio
    
    # Apply Volume and Amount Shock Filter
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    adjusted_intraday_momentum = adjusted_intraday_momentum.where((volume_ratio > volume_shock_threshold) & (amount_ratio > amount_shock_threshold), 0)
    
    # Introduce Smoothing Factor
    smoothed_intraday_momentum = adjusted_intraday_momentum.ewm(span=5).mean()
    
    return smoothed_intraday_momentum
```
