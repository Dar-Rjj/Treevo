```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    high_low_change = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    
    combined_momentum_adjusted = combined_momentum * volume_ratio * amount_ratio * high_low_change
    
    # Synthesize and Filter
    volume_shock_threshold = 1.5
    volume_ratio_filtered = (df['volume'] / df['volume'].shift(1)).fillna(1)
    combined_momentum_filtered = combined_momentum_adjusted * (volume_ratio_filtered > volume_shock_threshold)
    
    smoothed_combined_momentum = combined_momentum_filtered.ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Range and Volume
    range_volume = intraday_range * df['volume']
    sma_range_volume = range_volume.rolling(window=5, min_periods=1).mean()
    
    # Calculate MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    momentum = (macd_line > signal_line).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = np.zeros_like(df['close'])
    prev_close = df['close'].iloc[0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > prev_close:
            obv[i] = obv[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < prev_close:
            obv[i] = obv[i-1] - df['volume'].iloc[i]
        else:
            obv[i] = obv[i-1]
        prev_close = df['close'].iloc[i]
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = np.abs(macd_line - signal_line)
    obv_factor = obv * weight
    
    # Multiply OBV by Weight
    final_obv_factor = obv_factor
    
    # Combine Smoothed Intraday Momentum with OBV Factor
    final_alpha_factor = smoothed_combined_momentum * final_obv_factor
    
    return final_alpha_factor
```
