```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['range'] = df['high'] - df['low']
    
    # Measure Close Position in Range
    df['close_pos_in_range'] = (df['close'] - df['low']) / df['range']
    df['close_pos_in_range'].fillna(0.5, inplace=True)  # Fill with midpoint if range is zero
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + 
                  df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = 0.5 * df['high_low_diff'] + 0.5 * df['open_close_return']
    
    # Apply Volume and Amount Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['filtered_intraday_momentum'] = df['intraday_momentum'] * ((df['volume_ratio'] > 1.5) & (df['amount_ratio'] > 1.5))
    
    # Calculate Intraday Volatility
    df['abs_high_low_diff'] = abs(df['high'] - df['low'])
    df['intraday_volatility'] = df['abs_high_low_diff'].rolling(window=5).std()
    
    # Combine Factors
    factors = ['range', 'close_pos_in_range', 'vwap', 'filtered_intraday_momentum', 'intraday_volatility']
    recent_stdev = [df[factor].rolling(window=5).std() for factor in factors]
    
    weights = [1 / stdev.clip(lower=1e-6) for stdev in recent_stdev]
    total_weight = sum(weights)
    weights = [w / total_weight for w in weights]
    
    combined_factor = sum(w * df[f] for f, w in zip(factors, weights))
    
    return combined_factor
```
