```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Apply Volume Shock Filter and Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    volume_shock_threshold = 1.5
    volume_shock_filter = volume_ratio > volume_shock_threshold
    
    synchronized_product = volume_change * price_change
    synchronized_sign = synchronized_product.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Combine Intraday Momentum with Synchronized Volume and Price
    factor = intraday_momentum + synchronized_sign
    factor[~volume_shock_filter] = 0  # Apply the volume shock filter
    
    return factor
```
