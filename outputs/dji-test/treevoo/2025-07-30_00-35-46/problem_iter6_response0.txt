```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum
    # High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Returns with a weighted average
    intraday_momentum = (0.5 * high_low_diff + 0.5 * open_close_return)
    
    # Volume and Amount Shock Filter
    # Compute Volume and Amount Ratios
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Apply Dual Threshold
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    
    # Include Intraday Momentum if both ratios exceed their respective thresholds
    shock_filter = (volume_ratio > volume_shock_threshold) & (amount_ratio > amount_shock_threshold)
    
    # Final factor
    factor = intraday_momentum * shock_filter
    
    return factor
```
