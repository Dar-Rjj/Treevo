```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['close']
    
    # Measure Volume Synchronization
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Price Change Percentage
    df['price_change_percentage'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Synchronize Changes
    df['synchronized_product'] = df['volume_change_ratio'] * df['price_change_percentage']
    df['synchronized_sign'] = df['synchronized_product'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Final alpha factor
    df['alpha_factor'] = df['intraday_momentum'] * df['synchronized_sign']
    
    return df['alpha_factor']
```
