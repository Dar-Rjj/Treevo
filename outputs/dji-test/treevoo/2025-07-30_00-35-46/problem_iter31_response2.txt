```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Adjust for Volume and Amount Impact
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    adjusted_momentum = combined_momentum * (volume_ratio * amount_ratio)
    
    # Apply Volume Shock Filter
    adjusted_momentum = adjusted_momentum.where(volume_ratio > 1.5, other=0)
    
    # Smoothing with Exponential Moving Average (EMA) with Span of 5
    smoothed_momentum = adjusted_momentum.ewm(span=5).mean()
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    obv[close_diff > 0] = obv.shift(1) + df['volume']
    obv[close_diff < 0] = obv.shift(1) - df['volume']
    obv[close_diff == 0] = obv.shift(1)
    
    # Calculate MACD Line and Signal Line
    ema_12 = df['close'].ewm(span=12).mean()
    ema_26 = df['close'].ewm(span=26).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9).mean()
    
    # Weight OBV by MACD Line and Signal Line Difference
    weight = np.abs(macd_line - signal_line)
    final_obv_factor = obv * weight
    
    # Final Alpha Factor
    alpha_factor = smoothed_momentum * final_obv_factor
    return alpha_factor
```
