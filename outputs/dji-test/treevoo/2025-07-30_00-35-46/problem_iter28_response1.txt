import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] / df['open'] - 1
    daily_volatility = (df['high'] - df['low']) / df['close']
    weighted_return = open_close_return * daily_volatility
    non_linear_transform = weighted_return ** 2

    # Apply Integrated Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    historical_price_range = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    price_shock_ratio = abs_price_change / historical_price_range
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5) & (price_shock_ratio > 0.05)
    intraday_momentum = non_linear_transform.where(shock_filter, other=0)

    # Measure Synchronized Volume and Price Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_changes = np.sign(volume_change * price_change)
    synchronized_changes = synchronized_changes.where(volume_ratio > 1.5, other=0)

    # Adjust for Volume
    combined_metric = (intraday_momentum + synchronized_changes) / df['volume']

    # Calculate EMAs and Determine Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum = (short_ema > long_ema).astype(int) * 2 - 1

    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # Combine Momentum and OBV into a Factor
    weight = np.abs(short_ema - long_ema)
    obv_momentum_factor = obv * weight

    # Synthesize Final Alpha Factor
    final_alpha_factor = combined_metric + obv_momentum_factor
    final_alpha_factor = final_alpha_factor * intraday_momentum

    return final_alpha_factor
