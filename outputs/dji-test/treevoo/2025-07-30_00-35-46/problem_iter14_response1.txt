```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1)
    
    # Combine Intraday Range and Volume Adjustment
    combined_intraday = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_intraday = combined_intraday.ewm(span=5, adjust=False).mean()
    
    # Calculate Base Intraday Return
    base_intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (base_intraday_return + high_low_momentum) / 2
    
    # Apply Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    amount_ratio = df['amount'] / df['amount'].shift(1).fillna(1)
    high_low_spread_change = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    adjusted_combined_momentum = combined_momentum * volume_ratio * amount_ratio * high_low_spread_change
    
    # Synthesize and Filter
    synthesized_momentum = adjusted_combined_momentum * volume_ratio
    volume_shock_threshold = 1.5
    volume_shock_filter = (df['volume'] / df['volume'].shift(1).fillna(1)) > volume_shock_threshold
    filtered_momentum = synthesized_momentum * volume_shock_filter
    smoothed_filtered_momentum = filtered_momentum.ewm(span=5, adjust=False).mean()
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    momentum_signal = np.where(ema_short > ema_long, 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_change = df['close'].diff()
    obv[(close_change > 0)] += df['volume']
    obv[(close_change < 0)] -= df['volume']
    
    # Weight OBV by the Difference between Short and Long EMA
    ema_difference = np.abs(ema_short - ema_long)
    weighted_obv = obv * ema_difference
    
    # Multiply OBV by Weight
    final_obv_factor = weighted_obv * smoothed_filtered_momentum
    
    # Final Alpha Factor
    final_alpha_factor = final_obv_factor * momentum_signal
    
    return final_alpha_factor
```
