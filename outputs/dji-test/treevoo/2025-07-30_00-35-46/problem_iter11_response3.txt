```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = df['close'] / df['open'] - 1
    
    # Compute Historical Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Assign Weights based on Volatility
    weight_high_low = df['volatility']
    weight_open_close = 1 - weight_high_low
    
    # Combine High-Low and Open-Close Returns
    df['intraday_momentum'] = (weight_high_low * df['high_low_diff'] + 
                                weight_open_close * df['open_close_return'])
    
    # Apply Enhanced Volume and Price Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['price_shock'] = (df['close'] - df['close'].shift(1)).abs() / (df['high'] - df['low'])
    
    # Set Thresholds
    volume_threshold = 1.5
    amount_threshold = 1.5
    price_shock_threshold = 0.05
    
    # Include Intraday Momentum if conditions are met
    df['filtered_momentum'] = df.apply(
        lambda row: row['intraday_momentum'] if (row['volume_ratio'] > volume_threshold) and 
                    (row['amount_ratio'] > amount_threshold) and 
                    (row['price_shock'] > price_shock_threshold) else 0, axis=1)
    
    # Adjust for Volume
    df['volume_adjusted_momentum'] = df['filtered_momentum'] / df['volume']
    
    # Measure Volume Synchronization
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['synchronization'] = np.sign(df['volume_change'] * df['price_change'])
    
    # Introduce Smoothing Factor
    ema_alpha = 0.2
    df['smoothed_volume_ratio'] = df['volume_ratio'].ewm(alpha=ema_alpha, adjust=False).mean()
    df['smoothed_amount_ratio'] = df['amount_ratio'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Synthesize Combined Factor
    df['final_factor'] = (df['volume_adjusted_momentum'] + df['synchronization'])
    
    # Ensure Positive Values
    df['final_factor'] = df['final_factor'].apply(lambda x: max(x, 0))
    
    return df['final_factor']
```
