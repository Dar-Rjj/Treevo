```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages for close prices
    df['SMA_close_5'] = df['close'].rolling(window=5).mean()
    df['SMA_close_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate price change
    df['Price_change'] = df['close'].diff()
    
    # Calculate daily returns
    df['Daily_return'] = (df['close'] - df['open']) / df['open'] * 100
    
    # Calculate volatility
    df['Volatility_5'] = df['Daily_return'].rolling(window=5).std()
    df['Volatility_20'] = df['Daily_return'].rolling(window=20).std()
    
    # Calculate volume ratio
    df['SMA_volume_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_ratio'] = df['volume'] / df['SMA_volume_5']
    
    # Calculate true range
    df['True_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    
    # Calculate directional movements
    df['Positive_directional_movement'] = df['high'].diff()
    df['Negative_directional_movement'] = -df['low'].diff()
    
    # Calculate bullish and bearish engulfing patterns
    df['Bullish_engulfing'] = ((df['close'].shift(1) < df['open'].shift(1)) & 
                               (df['close'] > df['open']) & 
                               (df['close'] > df['open'].shift(1)) & 
                               (df['open'] < df['close'].shift(1))).astype(int)
    df['Bearish_engulfing'] = ((df['close'].shift(1) > df['open'].shift(1)) & 
                               (df['close'] < df['open']) & 
                               (df['close'] < df['open'].shift(1)) & 
                               (df['open'] > df['close'].shift(1))).astype(int)
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (df['SMA_close_5'] - df['SMA_close_20']) + \
                   df['Price_change'] - \
                   df['Volatility_5'] - \
                   df['Volatility_20'] + \
                   df['Volume_ratio'] + \
                   df['True_range'] + \
                   df['Positive_directional_movement'] - \
                   df['Negative_directional_movement'] + \
                   df['Bullish_engulfing'] - \
                   df['Bearish_engulfing']
    
    return alpha_factor
```
