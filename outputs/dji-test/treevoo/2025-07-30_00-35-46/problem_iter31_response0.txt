```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate modified PPUV
    df['mPPUV'] = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['close'] / df['volume'])
    
    # Calculate n-day relative strength (RS) factor
    df['daily_change'] = df['close'].diff()
    positive_changes = df['daily_change'].where(df['daily_change'] > 0, 0)
    negative_changes = df['daily_change'].where(df['daily_change'] < 0, 0).abs()
    df['RS'] = positive_changes.rolling(window=n).sum() / negative_changes.rolling(window=n).sum().abs()
    
    # Calculate smoothed AHLR
    df['AHLR'] = (df['high'] - df['low']) / df['close']
    df['sAHLR'] = df['AHLR'].rolling(window=m).mean()
    
    # Calculate volume stability factor
    df['volatility_factor'] = 1 / df['volume'].rolling(window=n).std()
    
    # Calculate Enhanced Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = df['close'] / df['open'] - 1
    df['daily_volatility'] = (df['high'] - df['low']).abs() / df['close']
    df['weighted_momentum'] = df['open_close_return'] * df['daily_volatility']
    df['non_linear_momentum'] = df['weighted_momentum'] ** 2
    
    # Apply Integrated Volume and Amount Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = (df['close'] * df['volume']) / (df['close'].shift(1) * df['volume'].shift(1))
    df['price_shock'] = (df['close'] - df['close'].shift(1)).abs()
    df['historical_price_range'] = (df['high'] - df['low']).rolling(window=n).max()
    df['price_shock_ratio'] = df['price_shock'] / df['historical_price_range']
    df['intraday_momentum_filtered'] = df['non_linear_momentum'].where(
        (df['volume_ratio'] > 1.5) & (df['amount_ratio'] > 1.5) & (df['price_shock_ratio'] > 0.05), 0)
    
    # Measure Synchronized Volume and Price Changes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_shock_filter'] = (df['volume'] / df['volume'].shift(1)) > 1.5
    df['synchronized_value'] = df['volume_change'] * df['price_change']
    df['synchronized_value'] = df['synchronized_value'].where(df['volume_shock_filter'], 0)
    
    # Adjust for Volume
    df['synchronized_value'] /= df['volume']
    
    # Calculate EMAs and Determine Momentum
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_bullish'] = df['short_ema'] > df['long_ema']
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = ((df['close'] > df['close'].shift(1)) * df['volume'] 
                 - (df['close'] < df['close'].shift(1)) * df['volume'])
    df['obv'] = df['obv'].cumsum()
    
    # Combine Momentum and OBV into a Factor
    df['weight'] = (df['short_ema'] - df['long_ema']).abs()
    df['obv_momentum_factor'] = df['obv'] * df['weight']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['intraday_momentum_filtered'] + df['synchronized_value'] + df['obv_momentum_factor']
    
    return df['final_alpha_factor']
```
