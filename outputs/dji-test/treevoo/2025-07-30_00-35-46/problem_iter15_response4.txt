import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns with weights
    weight_high_low = 0.5
    weight_open_close = 0.5
    combined_return = weight_high_low * high_low_diff + weight_open_close * open_close_return
    
    # Introduce Non-Linear Transformation
    non_linear_transformed = np.exp(combined_return)
    
    # Determine Volume Change and Amount Change
    volume_change = df['volume'].pct_change()
    amount_change = df['amount'].pct_change()
    
    # Compute Volume Ratio and Amount Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Set Shock Thresholds
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    # Apply Threshold Filter
    shock_filter = (volume_ratio > volume_threshold) & (amount_ratio > amount_threshold)
    
    # Introduce Smoothing Factor using EMA
    ema_alpha = 0.2
    volume_ema = volume_ratio.ewm(alpha=ema_alpha).mean()
    amount_ema = amount_ratio.ewm(alpha=ema_alpha).mean()
    
    # Final Alpha Factor
    alpha_factor = non_linear_transformed * shock_filter * (volume_ema * amount_ema)
    
    return alpha_factor
