import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum
    df['high_low_momentum'] = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    df['combined_momentum'] = 0.5 * (df['intraday_return'] + df['high_low_momentum'])
    
    # Apply Non-Linear Transformation
    df['non_linear_combined_momentum'] = df['combined_momentum'].pow(2)
    
    # Apply Volume-Weighted Adjustments
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['non_linear_combined_momentum'] * df['volume_ratio']
    
    # Integrate Amount Impact
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_adjusted_momentum'] = df['volume_adjusted_momentum'] * df['amount_ratio']
    
    # Apply Synchronized Volume and Price Changes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['synchronized_changes'] = np.sign(df['volume_change'] * df['price_change'])
    
    # Synthesize and Filter
    df['synthesized_momentum'] = df['amount_adjusted_momentum'] * df['synchronized_changes']
    
    # Apply Volume Shock Filter
    df['volume_shock'] = df['volume'] / df['volume'].shift(1) > 1.5
    df['filtered_momentum'] = df['synthesized_momentum'] * df['volume_shock']
    
    # Smoothing
    df['smoothed_momentum'] = df['filtered_momentum'].ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # EMA Cross Signal
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    df['macd_line'] = df['short_ema'] - df['long_ema']
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    df['momentum'] = (df['macd_line'] > df['signal_line']).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    df['obv_weight'] = abs(df['macd_line'] - df['signal_line'])
    df['weighted_obv'] = df['obv'] * df['obv_weight']
    
    # Combine Enhanced Intraday Momentum, Synchronized Volume, and OBV-Momentum Factor
    df['final_alpha_factor'] = df['smoothed_momentum'] + df['synchronized_changes'] + df['weighted_obv']
    
    return df['final_alpha_factor']
