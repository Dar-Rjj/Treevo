import pandas as pd
import talib

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()

    # Calculate Exponential Moving Averages (EMA)
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()

    # Calculate Rate of Change (ROC)
    df['ROC_5'] = df['close'].pct_change(periods=5) * 100
    df['ROC_10'] = df['close'].pct_change(periods=10) * 100
    df['ROC_20'] = df['close'].pct_change(periods=20) * 100
    df['ROC_50'] = df['close'].pct_change(periods=50) * 100

    # Calculate Average Directional Index (ADX)
    df['ADX'] = talib.ADX(df['high'], df['low'], df['close'], timeperiod=14)

    # Calculate Plus and Minus Directional Indicators (+DI, -DI)
    df['+DI'] = talib.PLUS_DI(df['high'], df['low'], df['close'], timeperiod=14)
    df['-DI'] = talib.MINUS_DI(df['high'], df['low'], df['close'], timeperiod=14)
    df['DI_Diff'] = df['+DI'] - df['-DI']

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = talib.OBV(df['close'], df['volume'])

    # Calculate Chaikin Money Flow (CMF)
    df['CMF_20'] = talib.ADOSC(df['high'], df['low'], df['close'], df['volume'], fastperiod=3, slowperiod=10)

    # Calculate the ratio of volume to amount
    df['Volume_to_Amount_Ratio'] = df['volume'] / df['amount']

    # Calculate Historical Volatility (HV)
    df['HV_10'] = df['close'].rolling(window=10).std() * (252**0.5)
    df['HV_20'] = df['close'].rolling(window=20).std() * (252**0.5)
    df['HV_50'] = df['close'].rolling(window=50).std() * (252**0.5)

    # Calculate Bollinger Bands
    df['Upper_Band'], df['Middle_Band'], df['Lower_Band'] = talib.BBANDS(df['close'], timeperiod=20, nbdevup=2, nbdevdn=2, matype=0)
    df['Bandwidth'] = (df['Upper_Band'] - df['Lower_Band']) / df['Middle_Band']
    df['Bandwidth_Percentage'] = df['Bandwidth'] * 100

    # Calculate True Range (TR) and Average True Range (ATR)
    df['TR'] = talib.TRANGE(df['high'], df['low'], df['close'])
    df['ATR_14'] = talib.ATR(df['high'], df['low'], df['close'], timeperiod=14)

    # Calculate Relative Strength Index (RSI)
    df['RSI_14'] = talib.RSI(df['close'], timeperiod=14)

    # Calculate Williams %R
    df['Williams_%R'] = talib.WILLR(df['high'], df['low'], df['close'], timeperiod=14)

    # Calculate Money Flow Index (MFI)
    df['MFI_14'] = talib.MFI(df['high'], df['low'], df['close'], df['volume'], timeperiod=14)
