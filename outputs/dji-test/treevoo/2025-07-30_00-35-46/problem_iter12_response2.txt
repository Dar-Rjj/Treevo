import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Assign weights to High-Low Difference and Open-Close Return
    weight_high_low = 0.6
    weight_open_close = 0.4
    
    intraday_momentum = (weight_high_low * high_low_diff + weight_open_close * open_close_return)
    
    # Apply Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    intraday_momentum_filtered = intraday_momentum.where(shock_filter, 0)
    
    # Adjust for Volume
    intraday_momentum_volume_adjusted = intraday_momentum_filtered / df['volume']
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    
    synchronized_product = volume_change * price_change
    synchronization_value = synchronized_product.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Introduce Moving Average Trend
    moving_avg_5day = df['close'].rolling(window=5).mean()
    trend_adjustment = (df['close'] - moving_avg_5day).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Synthesize Combined Factor
    combined_factor = (
        intraday_momentum_volume_adjusted +
        synchronization_value +
        trend_adjustment
    )
    
    # Ensure Positive Values
    combined_factor = combined_factor.apply(lambda x: max(x, 0))
    
    return combined_factor
