import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = (df['high_low_diff'] + df['open_close_return']) / 2
    
    # Synchronized Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sync_volume'] = np.sign(df['volume_ratio'] * df['price_change'])
    
    # EMA Cross Signal
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_momentum'] = (df['short_ema'] > df['long_ema']).astype(int)  # 1 for bullish, 0 for bearish
    
    # RSI Calculation
    df['daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['up_moves'] = df['daily_returns'].apply(lambda x: max(x, 0))
    df['down_moves'] = df['daily_returns'].apply(lambda x: -min(x, 0))
    df['avg_gain'] = df['up_moves'].rolling(window=14).mean()
    df['avg_loss'] = df['down_moves'].rolling(window=14).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss'].replace(0, np.nan)
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # On-Balance Volume (OBV)
    df['obv'] = 0
    close_diff = df['close'].diff()
    obv_increment = (close_diff > 0) * df['volume']
    obv_decrement = (close_diff < 0) * df['volume'] * -1
    df['obv'] = (obv_increment + obv_decrement).cumsum()
    
    # Combined Alpha Factor
    df['alpha_factor'] = (df['intraday_momentum'] + df['sync_volume'] * df['ema_momentum'] + df['obv']) / df['rsi']
    
    return df['alpha_factor']
