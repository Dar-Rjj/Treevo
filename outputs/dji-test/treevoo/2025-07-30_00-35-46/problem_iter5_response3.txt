import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = 0.5 * high_low_diff + 0.5 * open_close_return
    
    # Apply Volume and Amount Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    
    shock_filter = (volume_ratio > volume_shock_threshold) & (amount_ratio > amount_shock_threshold)
    intraday_momentum_filtered = intraday_momentum.where(shock_filter, 0)
    
    # Adjust for Volume
    volume_adjusted_momentum = intraday_momentum_filtered / df['volume']
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_product = volume_change * price_change
    synchronization_value = synchronized_product.apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Synthesize Combined Factor
    combined_factor = volume_adjusted_momentum + synchronization_value
    final_alpha_factor = combined_factor.clip(lower=0)
    
    return final_alpha_factor
