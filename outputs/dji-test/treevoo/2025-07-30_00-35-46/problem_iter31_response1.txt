```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Return'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] / df['open']) - 1
    df['Intraday_Momentum'] = 0.5 * (df['High_Low_Return'] + df['Open_Close_Return'])

    # Apply Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Absolute_Price_Change'] = abs(df['close'] - df['close'].shift(1))
    df['Price_Shock_Ratio'] = df['Absolute_Price_Change'] / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    df['Filtered_Intraday_Momentum'] = df.apply(lambda row: row['Intraday_Momentum'] if (row['Volume_Ratio'] > 1.5) and (row['Amount_Ratio'] > 1.5) and (row['Price_Shock_Ratio'] > 0.05) else 0, axis=1)

    # Measure Volume Synchronization
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronization_Value'] = np.sign(df['Volume_Change'] * df['Price_Change'])

    # Calculate EMAs and SMAs
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Bullish_Bearish'] = (df['Short_EMA'] > df['Long_EMA']).astype(int)

    # Analyze Price Momentum
    df['ROC'] = df['close'].pct_change(periods=5)
    df['RSI'] = 100 - (100 / (1 + (df['close'].diff().clip(lower=0).rolling(window=14).mean() / df['close'].diff().clip(upper=0).rolling(window=14).mean())))

    # Investigate Trading Volume
    df['Average_Volume_5d'] = df['volume'].rolling(window=5).mean()
    df['Volume_Signal'] = (df['volume'] > df['Average_Volume_5d']).astype(int)

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] - df['volume'].iloc[i]
        else:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1]

    # Combine Momentum and OBV into a Factor
    df['EMA_Weight'] = abs(df['Short_EMA'] - df['Long_EMA'])
    df['OBV_Momentum_Factor'] = df['OBV'] * df['EMA_Weight']

    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    df['Volume_Adjusted_Intraday_Momentum'] = df['Filtered_Intraday_Momentum'] * df['OBV_Momentum_Factor']

    # Combine Multiple Factors
    df['Composite_Alpha_Factor'] = 0.3 * df['Bullish_Bearish'] + 0.2 * df['ROC'] + 0.2 * df['RSI'] + 0.3 * df['Volume_Signal']

    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Intraday_Momentum'] + df['Synchronization_Value'] + df['Composite_Alpha_Factor']

    return df['Final_Alpha_Factor']
```
