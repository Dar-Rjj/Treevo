```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Range and Volume
    df['range_volume'] = df['intraday_range'] * df['volume']
    df['smoothed_range_volume'] = df['range_volume'].rolling(window=5).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = short_ema - long_ema
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['macd_line'] > df['signal_line'], 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] + df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] - df.loc[df.index[i], 'volume']
        else:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv']
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    df['weight'] = abs(df['macd_line'] - df['signal_line'])
    df['final_obv_factor'] = df['obv'] * df['weight']
    
    # Incorporate Market Sentiment
    df['price_change'] = (df['close'] - df['open']) / df['open']
    df['sentiment_score'] = np.where(df['price_change'] > 0.005, 1,
                                      np.where(df['price_change'] < -0.005, -1, 0))
    df['adjusted_obv_factor'] = df['final_obv_factor'] * df['sentiment_score']
    
    # Combine Smoothed Intraday Momentum with Adjusted OBV Factor
    df['final_alpha_factor'] = df['smoothed_range_volume'] * df['adjusted_obv_factor']
    
    return df['final_alpha_factor']
```
