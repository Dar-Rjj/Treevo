```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=10).mean()).fillna(1) + 1
    
    # Combine Intraday Range and Volume Adjustment
    combined_intraday_volume = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_intraday_volume = combined_intraday_volume.ewm(span=5, adjust=False).mean()
    
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    adjusted_combined_momentum = combined_momentum * volume_ratio * amount_ratio
    
    # Synthesize and Filter
    synthesized_momentum = adjusted_combined_momentum * smoothed_intraday_volume
    volume_shock_threshold = 1.5
    volume_change = df['volume'] / df['volume'].shift(1)
    synthesized_filtered_momentum = synthesized_momentum.where(volume_change > volume_shock_threshold, other=0)
    smoothed_synthesized_filtered_momentum = synthesized_filtered_momentum.ewm(span=5, adjust=False).mean()
    
    # Calculate MACD
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    momentum = (macd_line > signal_line).astype(int) - (macd_line < signal_line).astype(int)
    
    # Calculate OBV
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    weight = (macd_line - signal_line).abs()
    final_obv_factor = obv * weight
    
    # Incorporate Market Sentiment
    price_change = (df['close'] - df['open']) / df['open']
    sentiment_score = ((price_change > 0.005).astype(int) - (price_change < -0.005).astype(int))
    adjusted_obv_factor = final_obv_factor * sentiment_score
    
    # Calculate Price Velocity
    price_velocity = (df['close'] - df['close'].shift(10)) / 10
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_synthesized_filtered_momentum * adjusted_obv_factor * price_velocity
    
    return final_alpha_factor
```
