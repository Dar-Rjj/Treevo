```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Compute Historical Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Assign Weights based on Volatility
    weights = 1 / (volatility + 1e-6)
    weighted_high_low = high_low_diff * weights
    weighted_open_close = open_close_return * weights
    
    intraday_momentum = (weighted_high_low + weighted_open_close) / 2
    
    # Apply Enhanced Volume and Price Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = (df['amount'] / df['amount'].shift(1))
    price_shock = abs(df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])
    
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5) & (price_shock > 0.05)
    intraday_momentum = intraday_momentum * shock_filter
    
    # Adjust for Volume
    intraday_momentum /= df['volume']
    
    # Measure Volume Synchronization with Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronization = np.sign(volume_change * price_change) * shock_filter
    
    # Calculate Exponential Moving Averages (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_signal = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    ema_difference = abs(short_ema - long_ema)
    weighted_obv = obv * ema_difference
    
    # Combine Momentum and OBV into a Factor
    momentum_obv_factor = weighted_obv * momentum_signal
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = intraday_momentum + synchronization + momentum_obv_factor
    final_alpha_factor = final_alpha_factor.clip(lower=0)
    
    return final_alpha_factor
```
