```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 5-day exponential moving average of the volume
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the daily return
    df['std_20_return'] = df['return'].rolling(window=20).std()
    
    # Calculate the range (high - low)
    df['range'] = df['high'] - df['low']
    
    # Calculate the 10-day moving average of the range
    df['ma_10_range'] = df['range'].rolling(window=10).mean()
    
    # Calculate the relative strength of the close price compared to the 10-day moving average
    df['rs_strength'] = df['close'] / df['ma_10_close']
    
    # Calculate the normalized volume by dividing the volume by its 5-day exponential moving average
    df['norm_volume'] = df['volume'] / df['ema_5_volume']
    
    # Calculate the volatility-adjusted return by dividing the daily return by the 20-day standard deviation
    df['vol_adj_return'] = df['return'] / df['std_20_return']
    
    # Final factor: combination of strength, normalized volume, and volatility-adjusted return
    df['factor'] = df['rs_strength'] * df['norm_volume'] * df['vol_adj_return']
    
    # Return the factor as a pandas Series
    return df['factor']
```
