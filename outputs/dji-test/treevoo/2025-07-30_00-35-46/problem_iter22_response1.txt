```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] / df['open']) - 1
    weighted_high_low = 0.6 * high_low_diff
    weighted_open_close = 0.4 * open_close_return
    enhanced_intraday_momentum = (weighted_high_low + weighted_open_close) ** 2
    
    # Apply Integrated Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    
    # Measure Synchronized Volume and Price Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_changes = np.sign(volume_change * price_change)
    
    # Combine Enhanced Intraday Momentum, Synchronized Volume, and OBV-Momentum Factor
    obv = (np.sign(price_change) * df['volume']).cumsum()
    obv_momentum = obv.diff()
    final_alpha_factor = (enhanced_intraday_momentum + synchronized_changes + obv_momentum) * shock_filter
    
    return final_alpha_factor
```
