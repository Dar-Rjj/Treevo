```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adj'] = (df['volume'] / df['volume'].rolling(window=5).mean()) + 1
    
    # Combine Intraday Range and Volume Adjustment
    df['combined_range_vol'] = df['range'] * df['volume_adj']
    
    # Smoothing
    df['smoothed_momentum'] = df['combined_range_vol'].ewm(span=5).mean()
    
    # Calculate EMA Cross Signal
    df['ema_short'] = df['close'].ewm(span=5).mean()
    df['ema_long'] = df['close'].ewm(span=20).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['ema_short'] > df['ema_long'], 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'obv'] = df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'obv'] = -df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Weight OBV by Difference between Short and Long EMA
    df['obv_weight'] = np.abs(df['ema_short'] - df['ema_long'])
    
    # Multiply OBV by Weight
    df['final_obv_factor'] = df['obv'] * df['obv_weight']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['smoothed_momentum'] * df['final_obv_factor']
    
    return df['alpha_factor']
```
