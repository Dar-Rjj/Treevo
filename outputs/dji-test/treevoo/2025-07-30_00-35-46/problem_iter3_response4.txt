import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Change in High-Low Spread
    df['change_high_low_spread'] = df['high_low_spread'].diff()
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_adjusted_change'] = df['change_high_low_spread'] * df['volume_ratio']
    
    # Adjust for Amount Impact
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['amount_adjusted_change'] = df['change_high_low_spread'] * df['amount_ratio']
    
    # Aggregate Short-Term Momentum
    # Combine volume and amount adjusted changes
    df['combined_adjusted_change'] = (df['volume_adjusted_change'] + df['amount_adjusted_change']) / 2
    
    # Calculate the weighted average of the last 5 days
    def weighted_average(series, weights):
        return (series * weights).sum() / weights.sum()
    
    # Initialize the factor series
    factor_values = pd.Series(index=df.index, dtype='float64')
    
    for i in range(5, len(df)):
        window = df.iloc[i-5:i]
        combined_changes = window['combined_adjusted_change']
        volume_weights = window['volume_ratio']
        amount_weights = window['amount_ratio']
        total_weights = (volume_weights + amount_weights) / 2
        factor_values.iloc[i] = weighted_average(combined_changes, total_weights)
    
    return factor_values.dropna()
