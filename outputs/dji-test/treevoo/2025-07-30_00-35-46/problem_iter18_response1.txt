```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1)
    
    # Combine Intraday Range and Volume Adjustment
    combined_intraday_vol = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_intraday_momentum = combined_intraday_vol.ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = 0.7 * high_low_diff + 0.3 * open_close_return
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_filter = (volume_ratio > 1.2).astype(int)
    synchronized_changes = np.sign(volume_change * price_change) * volume_shock_filter
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate MACD
    macd_line = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    bullish_momentum = (ema_short > ema_long) & (macd_line > signal_line)
    
    # Calculate On-Balance Volume (OBV)
    obv = np.where(df['close'] > df['close'].shift(1), df['volume'],
                   np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    weight_obv = np.abs(macd_line - signal_line)
    weighted_obv = obv * weight_obv
    
    # Combine Momentum and OBV into a Factor
    momentum_obv_factor = weighted_obv * bullish_momentum.astype(int)
    
    # Combine Intraday Metrics
    combined_intraday_metrics = intraday_range * intraday_momentum
    
    # Adjust for Volume
    adjusted_volume = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1)
    volume_adjusted_intraday_metrics = combined_intraday_metrics / adjusted_volume
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + synchronized_changes + 
                          (momentum_obv_factor + volume_adjusted_intraday_metrics)) * smoothed_intraday_momentum
    
    return final_alpha_factor
```
