```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Enhanced Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['high_low_momentum'] = (df['high'] - df['low']) / df['open']
    df['avg_intraday_momentum'] = (df['intraday_return'] + df['high_low_momentum']) / 2
    
    # Synchronized Volume Dynamics
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['price_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_price_sync'] = df['volume_ratio'] * df['price_change']
    df['volume_shock_filter'] = df['volume_ratio'] > 1.5
    df['volume_sync'] = df['volume_price_sync'] * df['volume_shock_filter']
    
    # On-Balance Volume (OBV)
    df['obv'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] + df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] - df.loc[df.index[i], 'volume']
        else:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv']
    
    # EMA Cross Signal
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['macd_line'] = df['short_ema'] - df['long_ema']
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['macd_line'] > df['signal_line'], 1, -1)
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    df['weight'] = np.abs(df['macd_line'] - df['signal_line'])
    df['weighted_obv'] = df['obv'] * df['weight']
    
    # Apply Volume-Weighted Adjustments
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['adjusted_weighted_obv'] = df['weighted_obv'] * df['volume_ratio'] * df['amount_ratio']
    
    # Smoothing
    df['smoothed_intraday_momentum'] = df['avg_intraday_momentum'].ewm(span=5, adjust=False).mean()
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    df['final_alpha_factor'] = df['smoothed_intraday_momentum'] * df['adjusted_weighted_obv']
    
    return df['final_alpha_factor']
```
