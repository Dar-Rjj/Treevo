```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adj = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1) + 1
    
    # Combine Intraday Range and Volume Adjustment
    combined_metric = intraday_range * volume_adj
    
    # Smoothing
    smoothed_intraday = combined_metric.ewm(span=5).mean()
    
    # Calculate Intraday Momentum
    high_low_return = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = (0.5 * high_low_return + 0.5 * open_close_return).ewm(span=5).mean()
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    sync_changes = (volume_change * price_change).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Calculate EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_cross_signal = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    
    # Integrate with Intraday Momentum
    adjusted_momentum = intraday_momentum * ema_cross_signal
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0) * df['volume']
    obv = obv.cumsum()
    
    # Weight OBV by |Short EMA - Long EMA|
    weight = (short_ema - long_ema).abs()
    weighted_obv = obv * weight
    
    # Apply Dynamic Volume Shock Filter
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    volume_ratio_shock = df['volume'] / avg_vol_5d
    smoothed_intraday_filtered = smoothed_intraday.where(volume_ratio_shock > 1.8, 1)
    
    # Combine Intraday Metrics
    combined_intraday_metrics = (intraday_range * (high_low_return + open_close_return)) / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_momentum + sync_changes + weighted_obv + combined_intraday_metrics) * smoothed_intraday_filtered
    
    return final_alpha_factor
```
