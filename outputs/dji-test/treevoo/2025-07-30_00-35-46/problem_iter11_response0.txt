```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adjustment'] = (df['volume'] / df['volume'].rolling(window=5).median()) + 1
    
    # Combine Intraday Range and Volume Adjustment
    df['combined_intraday'] = df['intraday_range'] * df['volume_adjustment']
    
    # Smoothing with EMA
    df['smoothed_intraday'] = df['combined_intraday'].ewm(span=5, adjust=False).mean()
    
    # Calculate MACD
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = short_ema - long_ema
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['macd_line'] > df['signal_line'], 1, -1)
    
    # Calculate OBV
    df['obv'] = 0
    for i in range(1, len(df)):
        if df.iloc[i]['close'] > df.iloc[i-1]['close']:
            df.at[df.index[i], 'obv'] = df.iloc[i-1]['obv'] + df.iloc[i]['volume']
        elif df.iloc[i]['close'] < df.iloc[i-1]['close']:
            df.at[df.index[i], 'obv'] = df.iloc[i-1]['obv'] - df.iloc[i]['volume']
        else:
            df.at[df.index[i], 'obv'] = df.iloc[i-1]['obv']
    
    # Weight OBV by |MACD Line - Signal Line|
    df['obv_weighted'] = df['obv'] * np.abs(df['macd_line'] - df['signal_line'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['smoothed_intraday'] * df['momentum'] * df['obv_weighted']
    
    return df['alpha_factor']
```
