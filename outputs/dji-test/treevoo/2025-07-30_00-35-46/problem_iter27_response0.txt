```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate mPPUV
    df['mPPUV'] = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['close'] / df['volume'])
    
    # Calculate n-day mPPUV momentum (mPPUV_mom)
    df['mPPUV_mom'] = df['mPPUV'] - df['mPPUV'].shift(n)
    
    # Calculate the difference between the current close and the previous day's close
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Create a momentum indicator
    df['momentum_indicator'] = df['close_diff'] / ((df['close'] + df['close'].shift(1)) / 2)
    
    # Compute a n-day relative strength (RS) factor
    df['positive_change'] = df['close_diff'].apply(lambda x: x if x > 0 else 0)
    df['negative_change'] = df['close_diff'].apply(lambda x: abs(x) if x < 0 else 0)
    df['RS'] = df['positive_change'].rolling(window=n).sum() / df['negative_change'].rolling(window=n).sum()
    
    # Formulate an enhanced RS factor (eRS) as the sum of positive (high - low) changes divided by the absolute value of the sum of negative (high - low) changes over the last n days
    df['hl_change'] = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    df['positive_hl_change'] = df['hl_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_hl_change'] = df['hl_change'].apply(lambda x: abs(x) if x < 0 else 0)
    df['eRS'] = df['positive_hl_change'].rolling(window=n).sum() / df['negative_hl_change'].rolling(window=n).sum()
    
    # Generate an adjusted high-low ratio (AHLR)
    df['AHLR'] = (df['high'] - df['low']) / df['close']
    
    # Create a smoothed AHLR (sAHLR) by averaging the AHLR over the past m days
    df['sAHLR'] = df['AHLR'].rolling(window=m).mean()
    
    # Develop a trend factor (TF) as the difference between the current sAHLR and the sAHLR n days ago
    df['TF'] = df['sAHLR'] - df['sAHLR'].shift(n)
    
    # Measure the consistency of volumes by calculating the standard deviation of the last n days' volumes
    df['volume_std'] = df['volume'].rolling(window=n).std()
    
    # Derive a stability factor as the inverse of the volume standard deviation
    df['stability_factor'] = 1 / df['volume_std']
    
    # Calculate a volume momentum (VolMom) as the difference between the current volume and the volume n days ago
    df['VolMom'] = df['volume'] - df['volume'].shift(n)
    
    # Combine all factors to form the final alpha factor
    df['alpha_factor'] = (df['mPPUV_mom'] + df['momentum_indicator'] + df['RS'] + df['eRS'] + df['TF'] + df['stability_factor'] + df['VolMom']).fillna(0)
    
    return df['alpha_factor']
```
