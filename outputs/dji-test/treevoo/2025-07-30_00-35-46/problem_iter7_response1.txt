```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate an intraday momentum factor adjusted for trading volume.
    
    Args:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the calculated factor values, indexed by date.
    """
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust for Trading Activity
    adjusted_intraday_momentum = intraday_return / df['volume']
    
    return adjusted_intraday_momentum
```
