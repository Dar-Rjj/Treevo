import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] / df['open'] - 1
    
    # Assign weights to High-Low Difference and Open-Close Return
    weight_high_low = 0.6
    weight_open_close = 0.4
    weighted_average = (weight_high_low * high_low_diff + weight_open_close * open_close_return)
    
    # Apply Non-Linear Transformation (e.g., x^2)
    enhanced_intraday_momentum = weighted_average ** 2
    
    # Apply Integrated Volume and Amount Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Threshold Filter
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    
    # Measure Synchronized Volume and Price Changes
    price_change = df['close'] - df['close'].shift(1)
    synchronized_volume = (volume_change * price_change).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Apply Volume Shock Filter
    synchronized_volume_filtered = synchronized_volume * (volume_ratio > 1.5)
    
    # Combine Enhanced Intraday Momentum, Synchronized Volume, and OBV-Momentum Factor
    final_alpha_factor = enhanced_intraday_momentum + synchronized_volume_filtered
    
    return final_alpha_factor
