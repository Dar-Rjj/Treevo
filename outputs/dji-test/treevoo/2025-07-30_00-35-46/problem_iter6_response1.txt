import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['open']
    
    # Measure Volume-Price Synchronization
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    df['price_change_percentage'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['synchronized_product'] = df['volume_change_ratio'] * df['price_change_percentage']
    df['synchronized_square'] = df['synchronized_product'] ** 2
    
    # Combine the factors
    df['factor'] = df['intraday_momentum'] + df['synchronized_square']
    
    return df['factor']
