import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Components
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Apply Volume and Amount Adjustment
    high_low_spread = df['high'] - df['low']
    prev_high_low_spread = high_low_spread.shift(1)
    change_in_high_low_spread = high_low_spread - prev_high_low_spread
    
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_adjustment = volume_ratio * change_in_high_low_spread
    amount_adjustment = amount_ratio * change_in_high_low_spread
    
    adjusted_intraday_momentum = intraday_momentum * (volume_adjustment + amount_adjustment)
    
    # Adjust for Trading Activity
    trading_activity_adjustment = open_close_return / df['volume']
    adjusted_intraday_momentum *= trading_activity_adjustment
    
    # Apply Volume Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    adjusted_intraday_momentum = adjusted_intraday_momentum.where(volume_ratio > volume_shock_threshold, 0)
    
    # Smoothing
    ema_span = 5
    smoothed_intraday_momentum = adjusted_intraday_momentum.ewm(span=ema_span).mean()
    
    return smoothed_intraday_momentum
