```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Measure Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / (intraday_range)
    close_position_in_range = close_position_in_range.where(intraday_range != 0, 0)
    
    # Calculate Volume Weighted Close
    volume_weighted_close = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute recent mean and standard deviation for each factor
    rolling_window = 30  # Example window size
    intraday_range_mean = intraday_range.rolling(window=rolling_window).mean()
    intraday_range_std = intraday_range.rolling(window=rolling_window).std()
    
    close_position_in_range_mean = close_position_in_range.rolling(window=rolling_window).mean()
    close_position_in_range_std = close_position_in_range.rolling(window=rolling_window).std()
    
    volume_weighted_close_mean = volume_weighted_close.rolling(window=rolling_window).mean()
    volume_weighted_close_std = volume_weighted_close.rolling(window=rolling_window).std()
    
    # Assign weights inversely proportional to the standard deviations
    weight_intraday_range = 1 / intraday_range_std
    weight_close_position_in_range = 1 / close_position_in_range_std
    weight_volume_weighted_close = 1 / volume_weighted_close_std
    
    total_weight = weight_intraday_range + weight_close_position_in_range + weight_volume_weighted_close
    weight_intraday_range /= total_weight
    weight_close_position_in_range /= total_weight
    weight_volume_weighted_close /= total_weight
    
    # Compute weighted average of the three factors
    combined_factor = (weight_intraday_range * intraday_range +
                       weight_close_position_in_range * close_position_in_range +
                       weight_volume_weighted_close * volume_weighted_close)
    
    return combined_factor
```
