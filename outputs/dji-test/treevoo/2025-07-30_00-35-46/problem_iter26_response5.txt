```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] / df['open']) - 1
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close']
    df['Weighted_Avg'] = (df['Open_Close_Return'] * df['Daily_Volatility'] + 
                          df['High_Low_Diff'] * (1 - df['Daily_Volatility'])) / df['volume']
    
    # Apply Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Price_Shock'] = np.abs(df['close'] - df['close'].shift(1))
    df['Historical_Price_Range'] = df['close'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['Price_Shock_Ratio'] = df['Price_Shock'] / df['Historical_Price_Range']
    df['Intraday_Momentum'] = df['Weighted_Avg'].where((df['Volume_Ratio'] > 1.5) & 
                                                      (df['Amount_Ratio'] > 1.5) & 
                                                      (df['Price_Shock_Ratio'] > 0.05), 
                                                      0)
    
    # Adjust for Volume
    df['Intraday_Momentum_Adjusted'] = df['Intraday_Momentum'] / df['volume']
    
    # Measure Volume Synchronization
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronization'] = np.sign(df['Volume_Change'] * df['Price_Change'])
    df['Synchronization'] = df['Synchronization'].where(df['Volume_Ratio'] > 1.5, 0)
    
    # Trend Confirmation
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Crossover_Signal'] = np.where(df['Short_EMA'] > df['Long_EMA'], 1, -1)
    
    # Integrate with Intraday Momentum
    df['Intraday_Momentum_Final'] = df['Intraday_Momentum_Adjusted'] * df['Crossover_Signal']
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['OBV'] = obv
    df['EMA_Difference'] = np.abs(df['Short_EMA'] - df['Long_EMA'])
    df['Final_OBV_Factor'] = df['OBV'] * df['EMA_Difference']
    
    # Synthesize Final Alpha Factor
    df['Alpha_Factor'] = df['Intraday_Momentum_Final'] * df['Final_OBV_Factor']
    df['Alpha_Factor'] = df['Alpha_Factor'].where(df['Volume_Ratio'] > 1.5, 0)
    
    return df['Alpha_Factor']
```
