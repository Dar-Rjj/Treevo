```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range and Adjust for Volume
    df['range'] = df['high'] - df['low']
    df['volume_adj'] = (df['volume'] / df['volume'].rolling(window=5).mean()) + 0.01
    df['intraday_range_vol_adj'] = df['range'] * df['volume_adj']
    
    # Smoothing
    df['sma_intraday_range_vol_adj'] = df['intraday_range_vol_adj'].rolling(window=7).mean()
    
    # Calculate SMA Cross Signal
    df['sma_close_short'] = df['close'].rolling(window=7).mean()
    df['sma_close_long'] = df['close'].rolling(window=25).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['sma_close_short'] > df['sma_close_long'], 1, np.where(df['sma_close_short'] < df['sma_close_long'], -1, 0))
    
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = df['close'] - df['open']
    df['intraday_momentum'] = (df['high_low_diff'] + df['open_close_return']) / 2
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] + df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv'] - df.loc[df.index[i], 'volume']
        else:
            df.loc[df.index[i], 'obv'] = df.loc[df.index[i-1], 'obv']
    
    # Weight OBV by Difference between Short and Long SMA
    df['final_obv_factor'] = df['obv'] * np.abs(df['sma_close_short'] - df['sma_close_long'])
    
    # Apply Volume and Amount Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['shock_filter'] = (df['volume_ratio'] > 2.0) & (df['amount_ratio'] > 2.0)
    
    # Measure Volume Synchronization
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sync_value'] = np.sign(df['volume_change'] * df['price_change'])
    
    # Synthesize Combined Factor
    df['combined_factor'] = df['sma_intraday_range_vol_adj'] + df['sync_value']
    df['combined_factor'] = np.where(df['combined_factor'] > 0, df['combined_factor'], 0)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['intraday_momentum'] * np.where(df['shock_filter'], df['final_obv_factor'], 0)
    
    return df['alpha_factor']
```
