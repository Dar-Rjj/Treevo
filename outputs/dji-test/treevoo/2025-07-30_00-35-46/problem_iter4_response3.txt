```python
def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Intraday Metrics
    integrated_metric = intraday_price_movement * intraday_return
    
    # Adjust for Trading Activity
    adjusted_factor = integrated_metric / df['volume']
    
    return adjusted_factor
```
