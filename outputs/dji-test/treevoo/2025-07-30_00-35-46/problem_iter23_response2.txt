import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = df['close'] / df['open'] - 1
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Combine High-Low and Open-Close Returns with Weighted Average based on Volatility
    df['Intraday_Momentum'] = (df['High_Low_Diff'] * df['Daily_Volatility'] + df['Open_Close_Return'] * (1 - df['Daily_Volatility'])) / 2
    
    # Apply Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount'] = df['close'] * df['volume']
    df['Amount_Ratio'] = df['Amount'] / df['Amount'].shift(1)
    df['Price_Shock'] = np.abs(df['close'] - df['close'].shift(1))
    df['Price_Range'] = df['high'] - df['low']
    df['Price_Shock_Ratio'] = df['Price_Shock'] / df['Price_Range'].rolling(window=30).max()
    
    # Threshold Filter
    df['Filtered_Intraday_Momentum'] = df.apply(lambda row: row['Intraday_Momentum'] if row['Volume_Ratio'] > 1.5 and row['Amount_Ratio'] > 1.5 and row['Price_Shock_Ratio'] > 0.05 else 0, axis=1)
    
    # Adjust for Volume
    df['Volume_Adjusted_Momentum'] = df['Filtered_Intraday_Momentum'] / df['volume']
    
    # Measure Volume Synchronization
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronization'] = np.sign(df['Volume_Change'] * df['Price_Change'])
    df['Synchronization'] = df.apply(lambda row: row['Synchronization'] if row['Volume_Ratio'] > 1.5 else 0, axis=1)
    
    # Calculate EMAs and Determine Momentum
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum'] = np.where(df['Short_EMA'] > df['Long_EMA'], 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Combine Momentum and OBV into a Factor
    df['Weight'] = np.abs(df['Short_EMA'] - df['Long_EMA'])
    df['OBV_Momentum_Factor'] = df['OBV'] * df['Weight']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Momentum'] + df['Synchronization'] + df['OBV_Momentum_Factor']
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    df['Smoothed_Intraday_Momentum'] = df['Filtered_Intraday_Momentum'].ewm(span=5, adjust=False).mean()
    df['Final_Alpha_Factor'] = df['Smoothed_Intraday_Momentum'] * df['OBV_Momentum_Factor']
    
    return df['Final_Alpha_Factor']
