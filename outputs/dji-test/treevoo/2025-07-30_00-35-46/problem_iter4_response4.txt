import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Average of High-Low Difference and Open-Close Return
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Determine Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Include Intraday Momentum if Volume Ratio > 1.5
    intraday_momentum_filtered = intraday_momentum.where(volume_ratio > 1.5, 0)
    
    return intraday_momentum_filtered
