```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the High-Low Range Breakout Momentum factor.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    - A pandas Series indexed by date representing the factor values.
    """
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Determine Previous Day's High and Low
    df['prev_high'] = df['high'].shift(1)
    df['prev_low'] = df['low'].shift(1)
    
    # Calculate Today's High-Low Range Relative to Previous Day
    df['high_breakout'] = (df['high'] > df['prev_high']).astype(int)
    df['low_breakout'] = (df['low'] < df['prev_low']).astype(int)
    
    # Determine If Breakout Occurs
    df['breakout_momentum'] = 0
    df.loc[df['high_breakout'] == 1, 'breakout_momentum'] = df['close'] - df['open']
    df.loc[df['low_breakout'] == 1, 'breakout_momentum'] = df['open'] - df['close']
    
    # Return the factor values
    return df['breakout_momentum']
```
