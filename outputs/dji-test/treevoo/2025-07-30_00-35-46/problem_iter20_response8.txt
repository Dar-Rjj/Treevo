import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on (open, high, low, close) prices and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average true range (ATR) over a 14-day period
    df['h-l'] = df['high'] - df['low']
    df['h-pc'] = abs(df['high'] - df['close'].shift(1))
    df['l-pc'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-pc', 'l-pc']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the money flow ratio (MFR)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df[df['typical_price'] > df['typical_price'].shift(1)]['money_flow']
    df['negative_money_flow'] = df[df['typical_price'] < df['typical_price'].shift(1)]['money_flow']
    df['positive_money_flow'] = df['positive_money_flow'].fillna(0)
    df['negative_money_flow'] = df['negative_money_flow'].fillna(0)
    df['money_ratio'] = df['positive_money_flow'].rolling(window=14).sum() / df['negative_money_flow'].rolling(window=14).sum()
    
    # Calculate the factor
    df['factor'] = df['return'] * df['atr_14'] * df['money_ratio']
    
    return df['factor']
