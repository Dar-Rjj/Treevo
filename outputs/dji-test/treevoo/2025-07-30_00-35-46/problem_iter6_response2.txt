```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adj'] = df['volume'] / (df['volume'].rolling(window=5).median() + 1)
    
    # Combine Intraday Range and Volume Adjustment
    df['intraday_momentum'] = df['intraday_range'] * df['volume_adj']
    
    # Smoothing with Simple Moving Average (SMA) with Span of 5
    df['intraday_momentum_sma'] = df['intraday_momentum'].rolling(window=5).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = short_ema - long_ema
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    df['momentum'] = (df['macd_line'] > df['signal_line']).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    df['obv'] = df['obv'].mask(df['close'] > df['close'].shift(1), df['volume'])
    df['obv'] = df['obv'].mask(df['close'] < df['close'].shift(1), -df['volume'])
    df['obv'] = df['obv'].mask(df['close'] == df['close'].shift(1), 0)
    df['obv'] = df['obv'].cumsum()
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    df['weight'] = abs(df['macd_line'] - df['signal_line'])
    df['obv_factor'] = df['obv'] * df['weight']
    
    # Combine Smoothed Intraday Momentum with OBV Factor
    df['alpha_factor'] = df['intraday_momentum_sma'] * df['obv_factor']
    
    return df['alpha_factor']
```
