```python
def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Metrics
    combined_intraday_metric = intraday_price_range * intraday_return * df['amount']
    
    # Adjust for Volume
    volume_adjusted_factor = combined_intraday_metric / df['volume']
    
    return volume_adjusted_factor
```
