```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Adjustments for Trading Activity
    # Calculate Volume Impact
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_adjusted_momentum = combined_momentum * volume_ratio
    
    # Adjust for Amount Impact
    amount_ratio = df['amount'] / df['amount'].shift(1)
    high_low_spread_change = (df['high'] - df['low']) / df['low'].shift(1)
    amount_adjusted_momentum = volume_adjusted_momentum * amount_ratio * high_low_spread_change
    
    # Synthesize and Filter
    # Apply Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_change = df['volume'] / df['volume'].shift(1)
    volume_shock_filter = (volume_change > volume_shock_threshold)
    
    filtered_momentum = amount_adjusted_momentum.where(volume_shock_filter, other=0)
    
    # Smoothing
    smoothed_momentum = filtered_momentum.ewm(span=5, min_periods=5).mean()
    
    return smoothed_momentum
```
