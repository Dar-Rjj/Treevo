import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Intraday_Momentum'] = 0.5 * (df['High_Low_Diff'] + df['Open_Close_Return'])

    # Measure Volume Synchronization with Shock Filter
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    df['Synchronized_Product'] = df['Volume_Change'] * df['Price_Change'] * (df['Volume_Ratio'] > volume_shock_threshold).astype(int)
    df['Synchronized_Sign'] = np.sign(df['Synchronized_Product'])

    # Calculate Exponential Moving Average (EMA) Cross Signal
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum'] = np.where(df['Short_EMA'] > df['Long_EMA'], 'Bullish', 'Bearish')

    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['OBV'] = obv
    df['Weight'] = np.abs(df['Short_EMA'] - df['Long_EMA'])
    df['Momentum_and_OBV_Factor'] = df['OBV'] * df['Weight']

    # Combine Intraday Metrics
    df['Intraday_Price_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    df['Combined_Metric'] = (df['Intraday_Price_Range'] * df['Intraday_Return']) * df['amount']
    df['Volume_Adjusted_Combined_Metric'] = df['Combined_Metric'] / df['volume']

    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Intraday_Momentum'] + 
                                df['Synchronized_Sign'] + 
                                (df['Momentum_and_OBV_Factor'] + 
                                 df['Volume_Adjusted_Combined_Metric']))

    return df['Final_Alpha_Factor']
