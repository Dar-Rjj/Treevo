import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / (df['volume'].rolling(window=5).mean() + 1)
    
    # Combine Intraday Range and Volume Adjustment
    combined_intraday = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_intraday = combined_intraday.ewm(span=5).mean()
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5).mean()
    long_ema = df['close'].ewm(span=20).mean()
    
    # Determine Momentum
    momentum = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[0] = 0
    for i in range(1, len(obv)):
        if df['close'][i] > df['close'][i-1]:
            obv[i] += obv[i-1] + df['volume'][i]
        elif df['close'][i] < df['close'][i-1]:
            obv[i] = obv[i-1] - df['volume'][i]
        else:
            obv[i] = obv[i-1]
    
    # Weight OBV by the Difference between Short and Long EMA
    weight = (short_ema - long_ema).abs()
    weighted_obv = obv * weight
    
    # Multiply Smoothed Intraday Momentum by OBV Factor
    final_alpha_factor = smoothed_intraday * weighted_obv
    
    return final_alpha_factor
