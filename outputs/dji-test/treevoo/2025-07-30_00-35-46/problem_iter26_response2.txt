```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = 0.5 * high_low_diff + 0.5 * open_close_return
    intraday_momentum = np.power(intraday_momentum, 2)  # Non-linear transformation

    # Apply Synchronized Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    shock_filter = (volume_ratio > volume_threshold) & (amount_ratio > amount_threshold)
    intraday_momentum_filtered = intraday_momentum * shock_filter
    
    # Introduce Smoothing Factor
    volume_ema = df['volume'].ewm(alpha=0.2, adjust=False).mean()
    amount_ema = df['amount'].ewm(alpha=0.2, adjust=False).mean()

    # Price Change
    price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Synchronize Volume and Price Change
    synchronized_volume_price = volume_ratio * price_change

    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_momentum = short_ema - long_ema

    # On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    up_days = close_diff > 0
    down_days = close_diff < 0
    obv[up_days] = obv.shift(1)[up_days] + df['volume'][up_days]
    obv[down_days] = obv.shift(1)[down_days] - df['volume'][down_days]
    obv.fillna(method='ffill', inplace=True)

    # Combined Alpha Factor
    alpha_factor = intraday_momentum_filtered + synchronized_volume_price * ema_momentum + obv

    return alpha_factor
```
