```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns with weights
    weighted_high_low = high_low_diff * 0.6
    weighted_open_close = open_close_return * 0.4
    combined_return = weighted_high_low + weighted_open_close
    
    # Introduce Non-Linear Transformation
    intraday_momentum = combined_return ** 2
    
    # Apply Integrated Volume and Amount Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    shock_filter = (volume_ratio > volume_threshold) & (amount_ratio > amount_threshold)
    
    # Introduce Smoothing Factor
    volume_ema = volume_ratio.ewm(alpha=0.2).mean()
    amount_ema = amount_ratio.ewm(alpha=0.2).mean()
    
    # Measure Synchronized Volume and Price Changes
    price_change = df['close'] - df['close'].shift(1)
    synchronized_product = volume_change * price_change
    synchronized_sign = np.sign(synchronized_product)
    
    # Calculate Exponential Moving Averages (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    bullish_bearish = np.where(short_ema > long_ema, 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Combine Momentum and OBV into a Factor
    ema_diff = np.abs(short_ema - long_ema)
    momentum_obv_factor = obv * ema_diff
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + synchronized_sign + momentum_obv_factor) * shock_filter
    
    return final_alpha_factor
```
