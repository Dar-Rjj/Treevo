```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['range'] = df['high'] - df['low']
    
    # Measure Close Position in Range
    df['close_pos_in_range'] = (df['close'] - df['low']) / df['range'].replace(0, 1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + 
                  df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = df['close'] - df['open']
    df['intraday_momentum'] = 0.5 * df['high_low_diff'] + 0.5 * df['open_close_return']
    
    # Apply Volume and Amount Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['filtered_intraday_momentum'] = df['intraday_momentum'] * ((df['volume_ratio'] > 1.5) & (df['amount_ratio'] > 1.5))
    
    # Compute recent standard deviations
    std_range = df['range'].rolling(window=20).std()
    std_close_pos_in_range = df['close_pos_in_range'].rolling(window=20).std()
    std_vwap = df['vwap'].rolling(window=20).std()
    std_filtered_intraday_momentum = df['filtered_intraday_momentum'].rolling(window=20).std()
    
    # Assign weights inversely proportional to the standard deviations
    total_std = std_range + std_close_pos_in_range + std_vwap + std_filtered_intraday_momentum
    w_range = 1 / std_range
    w_close_pos_in_range = 1 / std_close_pos_in_range
    w_vwap = 1 / std_vwap
    w_filtered_intraday_momentum = 1 / std_filtered_intraday_momentum
    
    # Normalize weights
    w_range /= total_std
    w_close_pos_in_range /= total_std
    w_vwap /= total_std
    w_filtered_intraday_momentum /= total_std
    
    # Combine factors
    df['factor'] = (w_range * df['range'] + 
                    w_close_pos_in_range * df['close_pos_in_range'] +
                    w_vwap * df['vwap'] + 
                    w_filtered_intraday_momentum * df['filtered_intraday_momentum'])
    
    return df['factor']
```
