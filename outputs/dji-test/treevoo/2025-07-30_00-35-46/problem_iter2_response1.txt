```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Apply Volume Adjustment
    volume_mean_5_days = df['volume'].rolling(window=5).mean()
    volume_adjustment = df['volume'] / (volume_mean_5_days + 1)
    
    # Synthesize Intraday Momentum with Volume Adjustment
    adjusted_intraday_momentum = intraday_momentum * volume_adjustment
    
    # Apply Volume Shock Filter
    prev_volume = df['volume'].shift(1)
    volume_ratio = df['volume'] / prev_volume
    volume_shock_threshold = 1.5
    filtered_momentum = adjusted_intraday_momentum.where(volume_ratio > volume_shock_threshold)
    
    # Smoothing with Exponential Moving Average (EMA)
    smoothed_momentum = filtered_momentum.ewm(span=5, min_periods=1).mean()
    
    return smoothed_momentum
```
