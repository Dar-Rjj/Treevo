```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = (df['close'] / df['open']) - 1
    df['daily_volatility'] = df[['high', 'low', 'close']].std(axis=1)
    df['weighted_intraday_momentum'] = (df['high_low_diff'] * df['daily_volatility'] + 
                                        df['open_close_return'] * (1 - df['daily_volatility'])) / df['daily_volatility']

    # Apply Volume and Price Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['price_shock'] = np.abs(df['close'] - df['close'].shift(1))
    df['historical_price_range'] = df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min()
    df['price_shock_ratio'] = df['price_shock'] / df['historical_price_range']
    df['intraday_momentum_filtered'] = df['weighted_intraday_momentum'] * (
        (df['volume_ratio'] > 1.5) & (df['amount_ratio'] > 1.5) & (df['price_shock_ratio'] > 0.05))

    # Adjust for Volume
    df['volume_adjusted_intraday_momentum'] = df['intraday_momentum_filtered'] / df['volume']

    # Measure Volume Synchronization
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_synchronization'] = np.sign(df['volume_change'] * df['price_change']) * (df['volume_ratio'] > 1.5)

    # Calculate EMAs and SMAs
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['trend_signal'] = (df['short_ema'] > df['long_ema']) | (df['short_sma'] > df['long_sma'])

    # Analyze Price Momentum
    df['roc_5'] = df['close'].pct_change(periods=5)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Investigate Trading Volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_signal'] = (df['volume'] > df['avg_volume_20']).astype(int) - (df['volume'] <= df['avg_volume_20']).astype(int)

    # Combine Multiple Factors
    df['composite_alpha_factor'] = (
        0.4 * (df['short_ema'] > df['long_ema']) +
        0.4 * (df['short_sma'] > df['long_sma']) +
        0.1 * df['roc_5'] +
        0.1 * (df['rsi_14'] / 100) +
        0.1 * df['volume_signal']
    )

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['volume_adjusted_intraday_momentum'] + 
                                0.5 * df['volume_synchronization'] + 
                                df['composite_alpha_factor'])
    
    return df['final_alpha_factor']
```
