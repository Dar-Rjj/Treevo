```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return * 0.7) + (high_low_momentum * 0.3)
    
    # Apply Volume-Weighted Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_adjusted_momentum = combined_momentum * volume_ratio
    amount_adjusted_momentum = combined_momentum * amount_ratio
    
    # Synthesize and Filter
    synthesized_momentum = volume_adjusted_momentum * amount_adjusted_momentum
    
    # Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_change = df['volume'] / df['volume'].shift(1).fillna(1)
    synthesized_momentum = synthesized_momentum.where(volume_change > volume_shock_threshold, 0)
    
    # Smoothing
    smoothed_momentum = synthesized_momentum.ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Range and Volume
    range_volume = intraday_range * df['volume']
    sma_range_volume = range_volume.rolling(window=5).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12).mean()
    long_ema = df['close'].ewm(span=26).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9).mean()
    
    # Determine Momentum
    momentum = np.where(macd_line > signal_line, 'Bullish', 'Bearish')
    
    # Calculate On-Balance Volume (OBV)
    obv = np.zeros_like(df['close'])
    previous_close = df['close'].shift(1).fillna(df['close'][0])
    obv[1:] = np.where(df['close'] > previous_close, df['volume'], 0) - np.where(df['close'] < previous_close, df['volume'], 0)
    obv = obv.cumsum()
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = np.abs(macd_line - signal_line)
    final_obv_factor = obv * weight
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    final_alpha_factor = smoothed_momentum * final_obv_factor
    
    return final_alpha_factor
```
