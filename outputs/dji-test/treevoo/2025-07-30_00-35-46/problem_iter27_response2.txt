```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = df['close'] / df['open'] - 1
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Weighted Average based on Volatility
    df['Volatility_Weight'] = df['Daily_Volatility'] / df['Daily_Volatility'].rolling(window=30).mean()
    df['Intraday_Momentum'] = (df['High_Low_Diff'] * df['Volatility_Weight'] + df['Open_Close_Return'] * (1 - df['Volatility_Weight'])) / 2
    
    # Apply Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = (df['close'] * df['volume']) / ((df['close'].shift(1) * df['volume'].shift(1)))
    df['Absolute_Price_Change'] = np.abs(df['close'] - df['close'].shift(1))
    df['Historical_Price_Range'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    df['Price_Shock_Ratio'] = df['Absolute_Price_Change'] / df['Historical_Price_Range']
    
    # Threshold Filter
    df['Shock_Filter'] = (df['Volume_Ratio'] > 1.5) & (df['Amount_Ratio'] > 1.5) & (df['Price_Shock_Ratio'] > 0.05)
    df['Filtered_Intraday_Momentum'] = df['Intraday_Momentum'] * df['Shock_Filter']
    
    # Adjust for Volume
    df['Volume_Adjusted_Momentum'] = df['Filtered_Intraday_Momentum'] / df['volume']
    
    # Measure Volume Synchronization
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronization_Value'] = np.sign(df['Volume_Change'] * df['Price_Change'])
    
    # Calculate EMAs and Determine Momentum
    df['Short_EMA'] = df['close'].ewm(span=7, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Momentum'] = np.where(df['Short_EMA'] > df['Long_EMA'], 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0
    obv = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv += df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv -= df['volume'].iloc[i]
        else:
            obv += 0
        df['OBV'].iloc[i] = obv
    
    # Combine Momentum and OBV into a Factor
    df['Weight'] = np.abs(df['Short_EMA'] - df['Long_EMA'])
    df['OBV_Momentum_Factor'] = df['OBV'] * df['Weight']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Volume_Adjusted_Momentum'] + df['Synchronization_Value']) * df['OBV_Momentum_Factor']
    
    return df['Final_Alpha_Factor']
```
