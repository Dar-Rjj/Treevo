```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Measure Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['close_position_in_range'] = df['close_position_in_range'].replace([np.inf, -np.inf], np.nan)
    df['close_position_in_range'] = df['close_position_in_range'].fillna(0)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Compute recent standard deviation for each factor
    window = 20  # You can adjust the window size based on your needs
    df['intraday_range_std'] = df['intraday_range'].rolling(window=window).std()
    df['close_position_in_range_std'] = df['close_position_in_range'].rolling(window=window).std()
    df['vwap_std'] = df['vwap'].rolling(window=window).std()
    
    # Assign weights inversely proportional to the standard deviations
    total_std = df['intraday_range_std'] + df['close_position_in_range_std'] + df['vwap_std']
    df['intraday_range_weight'] = 1 / (df['intraday_range_std'] / total_std)
    df['close_position_in_range_weight'] = 1 / (df['close_position_in_range_std'] / total_std)
    df['vwap_weight'] = 1 / (df['vwap_std'] / total_std)
    
    # Ensure weights sum to 1
    df['total_weight'] = df['intraday_range_weight'] + df['close_position_in_range_weight'] + df['vwap_weight']
    df['intraday_range_weight'] = df['intraday_range_weight'] / df['total_weight']
    df['close_position_in_range_weight'] = df['close_position_in_range_weight'] / df['total_weight']
    df['vwap_weight'] = df['vwap_weight'] / df['total_weight']
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_range'] * df['intraday_range_weight'] +
                             df['close_position_in_range'] * df['close_position_in_range_weight'] +
                             df['vwap'] * df['vwap_weight'])
    
    return df['combined_factor']
```
