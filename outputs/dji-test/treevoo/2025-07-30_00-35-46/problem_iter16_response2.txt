```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    median_volume_5d = df['volume'].rolling(window=5).median().shift(1)
    volume_adjustment = (df['volume'] / (median_volume_5d + 1))
    
    # Combine Intraday Range and Volume Adjustment
    combined_metric = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_metric = combined_metric.rolling(window=5).mean()
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / (df['volume'].shift(1) + 1)
    volume_shock_filter = (volume_ratio > 1.5)
    synchronized_changes = np.sign(volume_change * price_change) * volume_shock_filter.astype(int)
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    bullish = (ema_short > ema_long) & (macd_line > signal_line)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    weight = np.abs(macd_line - signal_line)
    weighted_obv = obv * weight
    
    # Combine Momentum and OBV into a Factor
    momentum_obv_factor = weighted_obv * bullish.astype(int)
    
    # Combine Intraday Metrics
    intraday_metrics = intraday_range * intraday_momentum
    
    # Adjust for Volume
    volume_adjusted_intraday_metrics = intraday_metrics / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + synchronized_changes + (momentum_obv_factor + volume_adjusted_intraday_metrics)) * smoothed_metric
    
    return final_alpha_factor
```
