import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns using a weighted average
    weight_high_low = 0.6
    weight_open_close = 0.4
    combined_return = (weight_high_low * high_low_diff + weight_open_close * open_close_return)
    
    # Apply a non-linear transformation (e.g., x^2)
    transformed_return = combined_return ** 2
    
    # Apply Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    # Introduce Smoothing Factor using EMA
    alpha_ema = 0.2
    smoothed_volume_ratio = volume_ratio.ewm(alpha=alpha_ema).mean()
    smoothed_amount_ratio = amount_ratio.ewm(alpha=alpha_ema).mean()
    
    # Include Intraday Momentum if (Volume Ratio > Volume Threshold) AND (Amount Ratio > Amount Threshold)
    intraday_momentum = np.where((smoothed_volume_ratio > volume_threshold) & (smoothed_amount_ratio > amount_threshold), 
                                 transformed_return, 0)
    
    # Measure Synchronized Volume and Price Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    
    # Apply Volume Shock Filter (Include if Volume Ratio > 1.5)
    synchronized_changes = np.sign(volume_change * price_change) * np.where(smoothed_volume_ratio > volume_threshold, 1, 0)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv_momentum = obv.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine Integrated and Enhanced Intraday Momentum, Synchronized Volume, and OBV-Momentum Factor
    final_alpha_factor = intraday_momentum + synchronized_changes + obv_momentum
    
    return final_alpha_factor
