```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adjustment'] = (df['volume'] / df['volume'].rolling(window=10).mean()) + 1
    
    # Combine Intraday Range and Volume Adjustment
    df['combined_range_volume'] = df['range'] * df['volume_adjustment']
    
    # Smoothing
    df['smoothed_combined'] = df['combined_range_volume'].rolling(window=5).mean()
    
    # Calculate MACD
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = short_ema - long_ema
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    df['momentum'] = np.where(df['macd_line'] > df['signal_line'], 1, -1)
    
    # Calculate OBV
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    df['weight'] = np.abs(df['macd_line'] - df['signal_line'])
    df['weighted_obv'] = df['obv'] * df['weight']
    
    # Calculate Price Velocity
    df['price_velocity'] = (df['close'] - df['close'].shift(10)) / 10
    
    # Final Alpha Factor
    df['alpha_factor'] = df['smoothed_combined'] * df['momentum'] * df['weighted_obv'] * df['price_velocity']
    
    return df['alpha_factor']
```
