```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2

    # Measure Volume Synchronization with Shock Filter
    volume_change = df['volume'].diff()
    price_change = df['close'].diff()
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    synchronized_product = volume_change * price_change
    synchronized_sign = np.sign(synchronized_product)
    synchronized_volume = np.where(volume_ratio > volume_shock_threshold, synchronized_sign, 0)

    # Calculate Exponential Moving Average (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_signal = np.where(short_ema > long_ema, 1, -1)

    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] - df['volume'].iloc[i]
        else:
            obv.iloc[i] = obv.iloc[i-1]

    # Combine Momentum and OBV into a Factor
    ema_difference = np.abs(short_ema - long_ema)
    weighted_obv = obv * ema_difference
    momentum_and_obv_factor = weighted_obv * momentum_signal

    # Adjust for Trading Activity
    integrated_intraday_metrics = intraday_momentum * (df['close'] - df['open'])
    volume_adjusted_intraday_metrics = integrated_intraday_metrics / df['volume']

    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + synchronized_volume + 
                          (momentum_and_obv_factor + volume_adjusted_intraday_metrics))

    return final_alpha_factor
```
