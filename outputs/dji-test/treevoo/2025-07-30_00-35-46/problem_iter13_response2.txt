```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2

    # Measure Volume Synchronization with Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    synchronized_product = volume_change * price_change
    synchronized_sign = np.sign(synchronized_product)
    synchronized_volume = synchronized_sign.where(volume_ratio > volume_shock_threshold, 0)

    # Calculate Exponential Moving Averages (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    momentum = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)

    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    obv += df['volume'].where(df['close'] > df['close'].shift(1), 0)
    obv -= df['volume'].where(df['close'] < df['close'].shift(1), 0)

    # Weight OBV by the Difference between Short and Long EMA
    ema_diff = np.abs(short_ema - long_ema)
    weighted_obv = obv * ema_diff

    # Combine Momentum and OBV into a Factor
    momentum_and_obv_factor = weighted_obv * momentum

    # Synthesize Final Alpha Factor
    final_alpha_factor = intraday_momentum + synchronized_volume + momentum_and_obv_factor
    
    return final_alpha_factor
```
