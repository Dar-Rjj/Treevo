import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Returns by averaging
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Calculate Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Set Volume Shock Threshold
    volume_shock_threshold = 1.5
    
    # Apply Volume Shock Filter
    factor_values = intraday_momentum.where(volume_ratio > volume_shock_threshold, 0)
    
    return pd.Series(factor_values, index=df.index)
