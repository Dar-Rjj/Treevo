```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1) + 1
    
    # Combine Intraday Range and Volume Adjustment
    adjusted_intraday_range = intraday_range * volume_adjustment
    
    # Smoothing with EMA
    smoothed_intraday_range = adjusted_intraday_range.ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Apply Volume Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    factor = intraday_momentum.where(volume_ratio > 1.5, other=0)
    
    return factor
```
