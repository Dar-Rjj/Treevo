```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term SMAs for close prices
    df['SMA_short'] = df['close'].rolling(window=5).mean()
    df['SMA_long'] = df['close'].rolling(window=20).mean()
    
    # Create a binary signal based on SMA crossover
    df['SMA_signal'] = np.where(df['SMA_short'] > df['SMA_long'], 1, -1)
    
    # Calculate rate of change (ROC) over a short period
    df['ROC'] = df['close'].pct_change(periods=5)
    
    # Calculate RSI using the 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate average volume over different periods
    df['avg_volume_short'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_long'] = df['volume'].rolling(window=20).mean()
    
    # Create a binary signal based on volume
    df['volume_signal'] = np.where(df['volume'] > df['avg_volume_long'], 1, -1)
    
    # Integrate the SMA crossover signal with ROC and RSI
    df['SMA_weight'] = df['SMA_signal'] * 0.4
    df['ROC_weight'] = df['ROC'] * 0.3
    df['RSI_weight'] = (df['RSI'] - 50) / 50 * 0.3
    
    # Sum the weighted scores to create a composite alpha factor
    df['alpha_factor'] = df['SMA_weight'] + df['ROC_weight'] + df['RSI_weight']
    
    # Integrate the volume signal
    df['volume_weight'] = df['volume_signal'] * 0.2
    df['alpha_factor'] += df['volume_weight']
    
    return df['alpha_factor']
```
