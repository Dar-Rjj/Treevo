```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns using a weighted average
    weight_high_low = 0.5
    weight_open_close = 0.5
    intraday_momentum = (weight_high_low * high_low_diff) + (weight_open_close * open_close_return)
    
    # Determine Volume Change
    previous_volume = df['volume'].shift(1)
    volume_ratio = df['volume'] / previous_volume
    
    # Determine Amount Change
    previous_amount = df['amount'].shift(1)
    amount_ratio = df['amount'] / previous_amount
    
    # Set thresholds for volume and amount shock
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    
    # Apply the threshold filter
    shock_filter = (volume_ratio > volume_shock_threshold) & (amount_ratio > amount_shock_threshold)
    
    # Apply the filter to intraday momentum
    intraday_momentum_filtered = intraday_momentum.where(shock_filter, other=0)
    
    return intraday_momentum_filtered
```
