import pandas as pd

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return * 0.6) + (high_low_momentum * 0.4)
    
    # Apply Volume-Weighted Adjustments for Trading Activity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1).fillna(1)
    combined_momentum *= df['volume_ratio']
    
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1).fillna(1)
    combined_momentum *= df['amount_ratio']
    
    # Apply Volume Shock Filter
    volume_change = df['volume'] / df['volume'].shift(1).fillna(1)
    volume_shock_threshold = 1.5
    combined_momentum = combined_momentum.where(volume_change > volume_shock_threshold, 0)
    
    # Smoothing
    combined_momentum_smoothed = combined_momentum.ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Range and Volume
    intraday_range_volume = (intraday_range * df['volume']).rolling(window=5).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    momentum = (macd_line > signal_line).astype(int) - (macd_line < signal_line).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum().fillna(0)
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = (macd_line - signal_line).abs()
    obv_factor = obv * weight
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    final_alpha_factor = combined_momentum_smoothed * obv_factor
    
    return final_alpha_factor
