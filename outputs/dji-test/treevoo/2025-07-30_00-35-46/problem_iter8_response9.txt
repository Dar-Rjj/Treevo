```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = (df['high_low_diff'] + df['open_close_return']) / 2

    # Measure Volume Synchronization with Shock Filter
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    df['volume_shock'] = (df['volume_ratio'] > volume_shock_threshold).astype(int)
    df['synchronized_product'] = df['volume_change'] * df['price_change']
    df['synchronized_sign'] = np.sign(df['synchronized_product'])

    # Calculate Exponential Moving Average (EMA) Cross Signal
    df['ema_short'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_long'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum'] = (df['ema_short'] > df['ema_long']).astype(int)

    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'obv'] = df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'obv'] = -df['volume']
    df['obv'] = df['obv'].cumsum()

    # Combine Momentum and OBV into a Factor
    df['ema_diff'] = abs(df['ema_short'] - df['ema_long'])
    df['momentum_obv_factor'] = df['obv'] * df['ema_diff']

    # Combine Intraday Metrics
    df['intraday_metrics'] = df['high_low_diff'] * df['open_close_return'] * df['amount']
    df['volume_adjusted_intraday_metrics'] = df['intraday_metrics'] / df['volume']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['intraday_momentum'] +
        df['synchronized_sign'] +
        df['momentum_obv_factor'] +
        df['volume_adjusted_intraday_metrics']
    )

    return df['final_alpha_factor']
```
