```python
import pandas as pd

def heuristics_v2(df, window_size=10):
    # Calculate Intraday High-Low Range Ratio
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday High-Close and Low-Close Movements
    high_close_movement = df['high'] - df['close']
    low_close_movement = df['low'] - df['close']
    combined_movement = (high_close_movement - low_close_movement) * df['volume']
    
    # Sum over a period and divide by total volume over the period
    combined_movement_sum = combined_movement.rolling(window=window_size).sum()
    total_volume = df['volume'].rolling(window=window_size).sum()
    normalized_movement = combined_movement_sum / total_volume
    
    # Generate Final Factor
    final_factor = intraday_range * normalized_movement
    smoothed_factor = final_factor.rolling(window=window_size).mean()
    
    return smoothed_factor
```
