```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Historical Volatility
    daily_returns = (df['close'] / df['close'].shift(1) - 1).dropna()
    volatility = daily_returns.rolling(window=20).std()
    
    # Weighted Average of High-Low Difference and Open-Close Return
    weights = volatility / volatility.mean()
    intraday_momentum = (high_low_diff * weights + open_close_return * (1 - weights)) / 2
    
    # Apply Enhanced Volume and Price Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    price_shock = (df['close'] - df['close'].shift(1)).abs() / (df['high'] - df['low']).rolling(window=20).max()
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    price_shock_threshold = 0.05
    shock_filter = (volume_ratio > volume_threshold) & (amount_ratio > amount_threshold) & (price_shock > price_shock_threshold)
    
    # Adjust for Volume
    intraday_momentum = intraday_momentum / df['volume']
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    sync_product = volume_change * price_change
    sync_sign = np.sign(sync_product)
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum = (short_ema > long_ema).astype(int) * 2 - 1
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Weight OBV by |MACD Line - Signal Line|
    macd_line = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    weighted_obv = obv * (macd_line - signal_line).abs()
    
    # Calculate Price Velocity
    price_velocity = (df['close'] - df['close'].shift(10)) / 10
    
    # Integrate Intraday Metrics
    integrated_intraday_metrics = intraday_momentum * open_close_return / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_momentum + sync_sign + (weighted_obv * price_velocity + integrated_intraday_metrics))
    
    return final_alpha_factor
```
