import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] / df['open']) - 1
    intraday_momentum = 0.5 * high_low_diff + 0.5 * open_close_return
    intraday_momentum_transformed = intraday_momentum ** 3

    # Analyze Price Trends and Momentum
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    sma_crossover_signal = (short_sma > long_sma).astype(int) * 2 - 1

    roc = df['close'].pct_change(periods=5)
    
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_crossover_signal = (short_ema > long_ema).astype(int) * 2 - 1

    # Apply Enhanced Integrated Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    shock_filter = ((volume_ratio > 2.0) & (amount_ratio > 2.0)).astype(int)

    volume_shock = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    amount_shock = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    combined_shock_signal = ((volume_shock > 0) & (amount_shock > 0)).astype(int) * 2 - 1

    # Measure Synchronized Volume and Price Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_change = (volume_change * price_change).apply(np.sign)
    synchronized_magnitude = np.abs(volume_change * price_change)

    # Investigate Trading Volume
    avg_volume_short = df['volume'].rolling(window=5).mean()
    avg_volume_long = df['volume'].rolling(window=20).mean()
    volume_signal = (df['volume'] > avg_volume_long).astype(int) * 2 - 1

    # Combine Multiple Factors
    alpha_factor = (
        0.3 * sma_crossover_signal +
        0.2 * roc +
        0.2 * rsi +
        0.1 * ema_crossover_signal +
        0.1 * combined_shock_signal +
        0.1 * synchronized_change
    )

    return alpha_factor
