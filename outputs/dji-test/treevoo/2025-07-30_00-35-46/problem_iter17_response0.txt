import pandas as pd

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Volume-Weighted Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    combined_momentum *= volume_ratio * amount_ratio
    
    # Apply Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_change_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    combined_momentum = combined_momentum * (volume_change_ratio > volume_shock_threshold)
    
    # Smoothing
    smoothed_combined_momentum = combined_momentum.ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()).fillna(1) + 1
    adjusted_intraday_range = intraday_range * volume_adjustment
    
    # Smoothing
    smoothed_intraday_range = adjusted_intraday_range.ewm(span=5).mean()
    
    # Calculate EMA Cross Signal
    short_ema = df['close'].ewm(span=5).mean()
    long_ema = df['close'].ewm(span=20).mean()
    ema_cross_signal = (short_ema > long_ema).astype(int)

    final_obv_factor = weighted_obv * volume_amount_shock_filter
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1).fillna(0)
    price_change = df['close'] - df['close'].shift(1).fillna(0)
    synchronized_product = np.sign(volume_change * price_change)
    
    # Synthesize Combined Factor
    combined_factor = smoothed_intraday_range + synchronized_product
    combined_factor = combined_factor.apply(lambda x: max(x, 0))

    final_alpha_factor = combined_factor * final_obv_factor
