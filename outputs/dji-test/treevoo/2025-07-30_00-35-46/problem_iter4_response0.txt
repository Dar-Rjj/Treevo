```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Change in High-Low Spread
    df['change_high_low_spread'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Adjust for Volume Impact
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_spread'] = df['change_high_low_spread'] * df['volume_change']
    
    # Aggregate Short-Term Momentum
    df['short_term_momentum'] = df['adjusted_spread'].rolling(window=5).mean()
    
    return df['short_term_momentum']
```
