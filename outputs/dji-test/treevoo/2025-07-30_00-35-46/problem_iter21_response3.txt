import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Volume-Weighted Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    high_low_change = (df['high'] - df['low']).diff()
    
    combined_momentum *= volume_ratio
    combined_momentum *= amount_ratio * high_low_change
    
    # Apply Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_change_ratio = df['volume'] / df['volume'].shift(1, fill_value=df['volume'][0])
    combined_momentum = combined_momentum[volume_change_ratio > volume_shock_threshold]
    
    # Smoothing with Exponential Moving Average (EMA)
    smoothed_momentum = combined_momentum.ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    price_range = df['high'] - df['low']
    
    # Synchronize Volume Dynamics
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_changes = np.sign(volume_change * price_change)
    
    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5).mean()
    long_ema = df['close'].ewm(span=20).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9).mean()
    
    # Determine Momentum
    momentum = np.where(macd_line > signal_line, 1, -1)
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    obv[close_diff > 0] = df['volume']
    obv[close_diff < 0] = -df['volume']
    obv = obv.cumsum()
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = np.abs(macd_line - signal_line)
    weighted_obv = obv * weight
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    final_alpha_factor = smoothed_momentum * weighted_obv
    
    return final_alpha_factor
