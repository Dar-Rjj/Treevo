import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low'] + df['open'] - df['close']) / 2
    
    # Measure Volume Synchronization with Shock Filter
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    synchronized_volume = np.sign(df['volume_change'] * df['price_change'])
    
    # Calculate Exponential Moving Averages (EMAs)
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_trend = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    
    # On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Weighted OBV
    weighted_obv = abs(short_ema - long_ema) * df['obv']
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum + synchronized_volume + weighted_obv
    
    # Adjust for Trading Activity
    adjusted_alpha_factor = alpha_factor * (df['open'] - df['close']) / df['volume']
    
    return adjusted_alpha_factor
