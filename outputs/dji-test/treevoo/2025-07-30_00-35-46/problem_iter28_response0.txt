```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = 0.5 * (high_low_diff + open_close_return)
    
    # Volume Shock Filter
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / df['avg_volume_5d']
    volume_shock = (volume_ratio > 1.8).astype(int)
    
    # Trend Confirmation
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (short_term_ema > long_term_ema).astype(int) - (short_term_ema < long_term_ema).astype(int)
    
    # Integrate with Intraday Momentum
    trend_factor = np.where(crossover_signal == 1, 1.5, np.where(crossover_signal == -1, 0.5, 1))
    intraday_momentum *= trend_factor
    
    # On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Final OBV Factor
    final_obv_factor = df['obv'] * np.abs(short_term_ema - long_term_ema)
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum * final_obv_factor * volume_shock
    
    return alpha_factor
```
