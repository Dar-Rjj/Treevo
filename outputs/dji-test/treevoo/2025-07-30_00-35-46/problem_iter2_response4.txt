import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Averages
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    momentum = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    bullish = (short_ema > long_ema)
    bearish = (short_ema < long_ema)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine Momentum and OBV into a Factor
    weight = (short_ema - long_ema).abs()
    momentum_obv_factor = obv * weight
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Metrics
    intraday_metrics = intraday_price_range * intraday_return * df['amount']
    
    # Adjust for Volume
    volume_adjusted_intraday_metrics = intraday_metrics / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (momentum_obv_factor + volume_adjusted_intraday_metrics).fillna(0)
    
    return final_alpha_factor
