```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Calculate Volume Impact
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Adjust Combined Momentum by Volume Ratio
    combined_momentum_vol_adj = combined_momentum * volume_ratio
    
    # Adjust for Amount Impact
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Multiply by Change in High-Low Spread
    high_low_change = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    combined_momentum_amt_adj = combined_momentum_vol_adj * amount_ratio * high_low_change
    
    # Apply Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_shock = (df['volume'] / df['volume'].shift(1)) > volume_shock_threshold
    combined_momentum_filtered = combined_momentum_amt_adj.where(volume_shock, other=0)
    
    # Smoothing with Exponential Moving Average (EMA)
    smoothed_momentum = combined_momentum_filtered.ewm(span=5).mean()
    
    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    obv[close_diff > 0] += df['volume'][close_diff > 0]
    obv[close_diff < 0] -= df['volume'][close_diff < 0]
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = np.abs(macd_line - signal_line)
    weighted_obv = obv * weight
    
    # Integrate Volume-Weighted Intraday Momentum with OBV Factor
    final_alpha_factor = smoothed_momentum * weighted_obv
    
    return final_alpha_factor
```
