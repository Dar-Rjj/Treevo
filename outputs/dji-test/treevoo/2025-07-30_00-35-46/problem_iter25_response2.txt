```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate the modified PPUV (mPPUV)
    df['mPPUV'] = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['close'] / df['volume'])
    
    # Calculate the n-day relative strength (RS) factor
    df['change'] = df['close'].diff()
    positive_changes = df['change'].apply(lambda x: max(x, 0))
    negative_changes = df['change'].apply(lambda x: min(x, 0))
    df['RS'] = positive_changes.rolling(window=n).sum() / negative_changes.rolling(window=n).sum().abs()
    
    # Calculate the adjusted high-low ratio (AHLR)
    df['AHLR'] = (df['high'] - df['low']) / df['close']
    
    # Create a smoothed AHLR (sAHLR)
    df['sAHLR'] = df['AHLR'].rolling(window=m).mean()
    
    # Measure the consistency of volumes by calculating the standard deviation of the last n days' volumes
    df['volume_std'] = df['volume'].rolling(window=n).std()
    df['stability_factor'] = 1 / (df['volume_std'] + 1e-6)  # add small constant to avoid division by zero
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = df['mPPUV'] * df['RS'] * df['sAHLR'] * df['stability_factor']
    
    return df['alpha_factor']
```
