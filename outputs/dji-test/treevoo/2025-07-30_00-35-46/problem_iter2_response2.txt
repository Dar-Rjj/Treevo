import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Metrics
    combined_metric = intraday_price_range * intraday_return * df['amount']
    
    # Adjust for Volume
    adjusted_metric = combined_metric / df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['close']
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_product = volume_change * price_change
    synchronized_sign = synchronized_product.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Final Alpha Factor
    alpha_factor = adjusted_metric * intraday_momentum * synchronized_sign
    
    return alpha_factor
