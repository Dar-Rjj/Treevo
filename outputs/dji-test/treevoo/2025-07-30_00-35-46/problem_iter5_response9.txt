```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['HighMinusLow'] = df['high'] - df['low']
    df['OpenCloseReturn'] = (df['close'] - df['open']) / df['open']
    df['IntradayMomentum'] = (df['HighMinusLow'] + df['OpenCloseReturn']) / 2
    
    # Measure Volume Synchronization with Shock Filter
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    df['VolumeRatio'] = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    df['VolumeShock'] = (df['VolumeRatio'] > volume_shock_threshold).astype(int)
    df['SynchronizedProduct'] = df['VolumeChange'] * df['PriceChange']
    df['SynchronizedSign'] = df['SynchronizedProduct'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    df['SynchronizedSign'] *= df['VolumeShock']
    
    # Adjust for Trading Activity
    df['IntegratedMetric'] = df['IntradayMomentum'] * df['OpenCloseReturn']
    df['TradingActivityAdjusted'] = df['IntegratedMetric'] / df['volume']
    
    # Final Alpha Factor
    alpha_factor = df['TradingActivityAdjusted'] * df['SynchronizedSign']
    
    return alpha_factor
```
