import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns with weighted average
    weight_high_low = 0.5
    weight_open_close = 0.5
    intraday_momentum = (weight_high_low * high_low_diff) + (weight_open_close * open_close_return)
    
    # Apply a Power Function (e.g., x^2)
    intraday_momentum = np.power(intraday_momentum, 2)
    
    # Determine Volume Change and Amount Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Set thresholds for volume and amount shocks
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    # Include Intraday Momentum if (Volume Ratio > Volume Threshold) AND (Amount Ratio > Amount Threshold)
    intraday_momentum = np.where((volume_ratio > volume_threshold) & (amount_ratio > amount_threshold), 
                                 intraday_momentum, 0)
    
    # Apply Exponential Moving Average (EMA) to Volume and Amount Ratios
    ema_alpha = 0.2
    volume_ema = df['volume'].ewm(alpha=ema_alpha, adjust=False).mean()
    amount_ema = df['amount'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Measure Synchronized Volume and Price Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    
    # Apply Volume Shock Filter
    synchronized_volume = np.sign(volume_change * price_change)
    synchronized_volume = np.where(volume_ratio > 1.5, synchronized_volume, 0)
    
    # Combine Integrated Intraday Momentum, Synchronized Volume, and OBV-Momentum Factor
    obv = (np.sign(df['close'] - df['close'].shift(1)) * df['volume']).cumsum()
    obv_momentum = obv.diff()
    
    final_alpha_factor = intraday_momentum + synchronized_volume + obv_momentum
    
    return final_alpha_factor
