```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    intraday_momentum = 0.5 * (high_low_diff / df['open']) + 0.5 * open_close_return

    # Volume and Amount Shock Filter
    volume_change = df['volume'] / df['volume'].shift(1)
    amount_change = df['amount'] / df['amount'].shift(1)
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    shock_filter = (volume_change > volume_threshold) & (amount_change > amount_threshold)
    intraday_momentum = intraday_momentum * shock_filter

    # Trend Confirmation
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    crossover_signal = (short_term_ma > long_term_ma).astype(int) - (short_term_ma < long_term_ma).astype(int)
    intraday_momentum = intraday_momentum * crossover_signal

    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Adjust for Volume
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()) + 1
    intraday_range_adj = intraday_range * volume_adjustment

    # Smoothing
    smoothed_intraday = intraday_range_adj.ewm(span=5, adjust=False).mean()

    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_cross_signal = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)

    # On-Balance Volume (OBV)
    obv = np.where(df['close'] > df['close'].shift(1), df['volume'],
                  np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    
    # Weight OBV by Difference between Short and Long EMA
    weight = np.abs(short_ema - long_ema)
    weighted_obv = obv * weight

    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust Momentum based on RSI
    rsi_adjustment = np.where(rsi > 70, 0.8, np.where(rsi < 30, 1.2, 1))
    adjusted_momentum = smoothed_intraday * rsi_adjustment

    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum * weighted_obv
    
    return final_alpha_factor
```
