```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Assign weights to High-Low and Open-Close Returns
    weight_high_low = 0.6
    weight_open_close = 0.4
    
    # Combine High-Low and Open-Close Returns with a weighted average
    intraday_momentum = (weight_high_low * high_low_diff + weight_open_close * open_close_return)
    
    # Apply a non-linear transformation (e.g., x^2)
    intraday_momentum = intraday_momentum ** 2
    
    # Determine Volume Change and Compute Volume Ratio
    prev_volume = df['volume'].shift(1)
    volume_ratio = df['volume'] / prev_volume
    
    # Determine Amount Change and Compute Amount Ratio
    prev_amount = df['amount'].shift(1)
    amount_ratio = df['amount'] / prev_amount
    
    # Set thresholds for volume and amount shock
    volume_threshold = 1.5
    amount_threshold = 1.5
    
    # Filter intraday momentum based on volume and amount shock
    intraday_momentum_filtered = intraday_momentum[
        (volume_ratio > volume_threshold) & (amount_ratio > amount_threshold)
    ]
    
    # Introduce smoothing factor using Exponential Moving Average (EMA)
    ema_alpha = 0.2
    volume_ratio_smoothed = volume_ratio.ewm(alpha=ema_alpha).mean()
    amount_ratio_smoothed = amount_ratio.ewm(alpha=ema_alpha).mean()
    
    # Final alpha factor
    alpha_factor = intraday_momentum_filtered * volume_ratio_smoothed * amount_ratio_smoothed
    
    return alpha_factor
```
