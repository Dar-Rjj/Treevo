```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    prev_high_low_spread = high_low_spread.shift(1)
    
    # Compute Change in High-Low Spread
    change_high_low_spread = high_low_spread - prev_high_low_spread
    
    # Adjust for Volume Impact
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_adjusted_change = change_high_low_spread * volume_ratio
    
    # Adjust for Amount Impact
    amount_ratio = df['amount'] / df['amount'].shift(1)
    amount_adjusted_change = change_high_low_spread * amount_ratio
    
    # Synthesize and Filter
    adjusted_intraday_momentum = intraday_momentum * (volume_adjusted_change + amount_adjusted_change)
    
    # Apply Volume Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    adjusted_intraday_momentum = adjusted_intraday_momentum.where(volume_ratio > volume_shock_threshold, 0)
    
    # Smoothing with Exponential Moving Average (EMA) with Span of 5
    smoothed_momentum = adjusted_intraday_momentum.ewm(span=5).mean()
    
    return smoothed_momentum
```
