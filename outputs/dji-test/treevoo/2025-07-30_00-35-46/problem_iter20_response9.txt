import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the average true range over the last 14 days
    df['tr'] = df[['high' - 'low', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the relative intraday range to the average true range
    df['relative_range'] = df['intraday_range'] / df['atr_14']
    
    # Calculate the money flow multiplier
    df['mf_multiplier'] = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    df['mf_volume'] = df['mf_multiplier'] * df['volume']
    
    # Calculate the 14-period money flow ratio
    df['positive_mf'] = df[df['mf_multiplier'] > 0]['mf_volume'].rolling(window=14).sum()
    df['negative_mf'] = df[df['mf_multiplier'] < 0]['mf_volume'].rolling(window=14).sum().abs()
    df['money_flow_ratio'] = df['positive_mf'] / df['negative_mf']
    
    # Calculate the money flow index
    df['mfi'] = 100 - (100 / (1 + df['money_flow_ratio']))
    
    # Final alpha factor: combination of relative intraday range and money flow index
    df['alpha_factor'] = (df['relative_range'] + df['mfi']) / 2
    
    return df['alpha_factor']
