```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Determine Previous Day's High and Low
    prev_high = df['high'].shift(1)
    prev_low = df['low'].shift(1)
    
    # Calculate Today's High-Low Range Relative to Previous Day
    breakout_high = (df['high'] > prev_high)
    breakout_low = (df['low'] < prev_low)
    
    # Calculate Breakout Momentum
    breakout_momentum = pd.Series(0, index=df.index)
    breakout_momentum[breakout_high] = df.loc[breakout_high, 'close'] - df.loc[breakout_high, 'open']
    breakout_momentum[breakout_low] = df.loc[breakout_low, 'open'] - df.loc[breakout_low, 'close']
    
    # Adjust for Volume
    mean_volume = df['volume'].rolling(window=5).mean().fillna(method='bfill')
    volume_adjustment = df['volume'] / (mean_volume + 1)
    
    # Combine Breakout Momentum and Volume Adjustment
    combined_factor = breakout_momentum * volume_adjustment
    
    # Smoothing with Exponential Moving Average (EMA) with span of 5
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
