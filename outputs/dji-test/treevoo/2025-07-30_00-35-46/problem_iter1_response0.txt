```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Intraday Momentum with a Volume Shock Filter.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
                       and index as date.
    
    Returns:
    pd.Series: Series of factor values indexed by date.
    """
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Determine Volume Change
    previous_volume = df['volume'].shift(1)
    volume_ratio = df['volume'] / previous_volume
    
    # Apply Volume Shock Filter
    volume_shock_threshold = 1.5
    volume_shock_filter = volume_ratio > volume_shock_threshold
    
    # Include Intraday Momentum if Volume Ratio > Threshold
    factor_values = intraday_momentum * volume_shock_filter
    
    return factor_values
```
