```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Momentum Component
    high_low_momentum = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday and High-Low Momentum
    combined_momentum = (intraday_return + high_low_momentum) / 2
    
    # Apply Adjustments for Trading Activity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    combined_momentum_adjusted = combined_momentum * volume_ratio * amount_ratio
    
    # Synthesize and Filter
    volume_shock_threshold = 1.5
    volume_change_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    combined_momentum_filtered = combined_momentum_adjusted * (volume_change_ratio > volume_shock_threshold)
    smoothed_momentum = combined_momentum_filtered.ewm(span=5).mean()
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Range and Volume
    range_volume = intraday_range * df['volume']
    sma_range_volume = range_volume.rolling(window=5).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Determine Momentum
    momentum = (macd_line > signal_line).astype(int) - (macd_line < signal_line).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    obv = pd.Series(obv, index=df.index)
    
    # Weight OBV by the Difference between MACD Line and Signal Line
    weight = np.abs(macd_line - signal_line)
    final_obv_factor = obv * weight
    
    # Incorporate Market Sentiment
    price_change = (df['close'] - df['open']) / df['open']
    sentiment_score = np.where(price_change > 0.005, 1, np.where(price_change < -0.005, -1, 0))
    adjusted_obv_factor = final_obv_factor * sentiment_score
    
    # Combine Smoothed Intraday Momentum with Adjusted OBV Factor
    final_alpha_factor = smoothed_momentum * adjusted_obv_factor
    
    return final_alpha_factor
```
