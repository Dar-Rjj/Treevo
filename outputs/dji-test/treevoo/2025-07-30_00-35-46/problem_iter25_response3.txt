```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate modified PPUV (mPPUV)
    df['mPPUV'] = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['close'] / df['volume'])
    
    # Calculate the n-day relative strength (RS) factor
    df['close_change'] = df['close'].diff()
    positive_changes = df['close_change'].apply(lambda x: x if x > 0 else 0).rolling(window=n).sum()
    negative_changes = df['close_change'].apply(lambda x: abs(x) if x < 0 else 0).rolling(window=n).sum()
    df['RS'] = positive_changes / (negative_changes + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Calculate the adjusted high-low ratio (AHLR)
    df['AHLR'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the smoothed AHLR (sAHLR) over the past m days
    df['sAHLR'] = df['AHLR'].rolling(window=m).mean()
    
    # Calculate the volume stability factor
    df['volume_std'] = df['volume'].rolling(window=n).std()
    df['volume_stability'] = 1 / (df['volume_std'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['mPPUV'] * df['RS'] * df['sAHLR'] * df['volume_stability']
    
    return df['alpha_factor']
```
