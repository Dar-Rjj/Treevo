```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Daily_Volatility'] = df[['high', 'low', 'close']].std(axis=1)
    df['Weighted_Avg_Return'] = (df['High_Low_Diff'] * 0.5 + df['Open_Close_Return'] * 0.5) * df['Daily_Volatility']
    df['Non_Linear_Transform'] = df['Weighted_Avg_Return'].apply(lambda x: x**2)

    # Apply Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Price_Shock'] = abs(df['close'] - df['close'].shift(1))
    df['Price_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['Price_Shock_Ratio'] = df['Price_Shock'] / df['Price_Range']

    volume_threshold = 1.5
    amount_threshold = 1.5
    price_shock_threshold = 0.05

    df['Intraday_Momentum'] = df['Non_Linear_Transform'] * (
        (df['Volume_Ratio'] > volume_threshold) &
        (df['Amount_Ratio'] > amount_threshold) &
        (df['Price_Shock_Ratio'] > price_shock_threshold)
    )

    # Adjust for Volume
    df['Volume_Adjusted_Intraday'] = df['Intraday_Momentum'] / df['volume']

    # Measure Volume Synchronization
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronization'] = np.sign(df['Volume_Change'] * df['Price_Change']) * (df['Volume_Ratio'] > 1.5)

    # Calculate EMAs and Determine Momentum
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum'] = (df['Short_EMA'] > df['Long_EMA']).astype(int) * 2 - 1

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # Combine Momentum and OBV into a Factor
    df['OBV_Momentum_Factor'] = df['OBV'] * abs(df['Short_EMA'] - df['Long_EMA'])

    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Intraday'] + df['Synchronization'] + df['OBV_Momentum_Factor']

    return df['Final_Alpha_Factor']
```
