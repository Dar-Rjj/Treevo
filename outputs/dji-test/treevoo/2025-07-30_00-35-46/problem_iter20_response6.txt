import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as (date).
    
    Returns:
    - A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor: difference between 10-day and 30-day moving averages
    df['momentum_factor'] = df['ma_10'] - df['ma_30']
    
    # Calculate the 10-day standard deviation of the daily returns
    df['volatility_factor'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the volume-weighted close price
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength factor
    df['rsi_factor'] = 100 - (100 / (1 - df['daily_return'].rolling(window=14).mean()))
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = (df['momentum_factor'] / df['volatility_factor']) + (df['rsi_factor'] / 100) + (df['vwap'] / df['close'])
    
    return df['alpha_factor'].dropna()
