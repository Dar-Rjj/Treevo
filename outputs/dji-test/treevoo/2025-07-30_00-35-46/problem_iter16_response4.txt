import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range and Adjust for Volume
    intraday_range = df['high'] - df['low']
    volume_adjustment = (df['volume'] / df['volume'].rolling(window=5).mean()) + 1
    adjusted_intraday_range = intraday_range * volume_adjustment
    
    # Smoothing
    ema_adjusted_intraday_range = adjusted_intraday_range.ewm(span=5).mean()
    
    # Calculate EMA Cross Signal
    short_ema_close = df['close'].ewm(span=5).mean()
    long_ema_close = df['close'].ewm(span=20).mean()
    ema_cross_signal = (short_ema_close > long_ema_close).astype(int)
    
    # Determine Momentum
    high_low_difference = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = (0.5 * high_low_difference) + (0.5 * open_close_return)
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    obv[close_diff > 0] += df['volume'][close_diff > 0]
    obv[close_diff < 0] -= df['volume'][close_diff < 0]
    
    # Weight OBV by Difference between Short and Long EMA
    final_obv_factor = obv * abs(short_ema_close - long_ema_close)
    
    # Apply Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount = df['close'] * df['volume']
    amount_ratio = amount / amount.shift(1)
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronization_value = (volume_change * price_change).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Synthesize Combined Factor
    combined_factor = ema_adjusted_intraday_range + synchronization_value
    combined_factor = combined_factor.clip(lower=0)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * final_obv_factor * shock_filter
    
    return final_alpha_factor
