```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Metrics
    combined_metric = intraday_price_range * intraday_return
    
    # Calculate Average Amount over the last 10 days
    average_amount = df['amount'].rolling(window=10).mean()
    
    # Adjust Combined Metric by Average Amount
    adjusted_metric = combined_metric / average_amount
    
    # Calculate Average Volume over the last 10 days
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Final adjustment by Average Volume
    final_factor = adjusted_metric / average_volume
    
    return final_factor
```
