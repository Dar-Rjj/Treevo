```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = (df['close'] / df['open']) - 1
    
    # Compute Historical Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Assign Weights based on Volatility
    weight_high_low = volatility / (volatility + 1)
    weight_open_close = 1 - weight_high_low
    
    df['intraday_momentum'] = (weight_high_low * df['high_low_diff']) + (weight_open_close * df['open_close_return'])
    
    # Apply Enhanced Volume and Price Shock Filter
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    df['price_shock'] = abs(df['close'] - df['close'].shift(1))
    
    price_range = df['high'] - df['low']
    df['price_shock_ratio'] = df['price_shock'] / price_range
    
    volume_threshold = 1.5
    amount_threshold = 1.5
    price_shock_threshold = 0.05
    
    df['shock_filter'] = (df['volume_ratio'] > volume_threshold) & (df['amount_ratio'] > amount_threshold) & (df['price_shock_ratio'] > price_shock_threshold)
    
    # Adjust for Volume
    df['volume_adjusted_intraday'] = df['intraday_momentum'] / df['volume']
    
    # Measure Volume Synchronization
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['synchronized_product'] = df['volume_change'] * df['price_change']
    df['synchronization'] = np.sign(df['synchronized_product'])
    
    # Introduce Smoothing Factor
    ema_alpha = 0.2
    df['smoothed_volume_ratio'] = df['volume_ratio'].ewm(alpha=ema_alpha, adjust=False).mean()
    df['smoothed_amount_ratio'] = df['amount_ratio'].ewm(alpha=ema_alpha, adjust=False).mean()
    df['smoothed_price_shock_ratio'] = df['price_shock_ratio'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Synthesize Combined Factor
    df['combined_factor'] = (
        df['volume_adjusted_intraday'] +
        df['synchronization'] +
        df['smoothed_price_shock_ratio']
    )
    
    # Ensure Positive Values
    df['final_alpha_factor'] = df['combined_factor'].apply(lambda x: max(x, 0))
    
    return df['final_alpha_factor']
```
