```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Metrics
    df['combined_intraday_metric'] = df['intraday_price_range'] * df['intraday_return'] * df['amount']
    
    # Adjust for Volume
    df['volume_adjusted_metric'] = df['combined_intraday_metric'] / df['volume']
    
    # Measure Volume Synchronization
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['synchronized_product'] = df['volume_change'] * df['price_change']
    df['synchronization_value'] = df['synchronized_product'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Synthesize Combined Factor
    df['alpha_factor'] = df['volume_adjusted_metric'] + df['synchronization_value']
    
    # Ensure Positive Values
    df['alpha_factor'] = df['alpha_factor'].apply(lambda x: max(x, 0))
    
    return df['alpha_factor']
```
