import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Volume Adjustment
    volume_mean_5d = df['volume'].rolling(window=5).mean()
    volume_adjustment = (df['volume'] / volume_mean_5d) + 1
    
    # Adjusted Intraday Range
    adjusted_intraday_range = intraday_range * volume_adjustment
    
    # Smoothed Intraday Range
    smoothed_intraday_range = adjusted_intraday_range.ewm(span=7, adjust=False).mean()
    
    # EMA Cross Signal
    short_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Momentum
    momentum = (short_ema > long_ema).astype(int)
    
    # On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum().fillna(0)
    
    # Weight OBV by EMA Difference
    ema_diff = (short_ema - long_ema) ** 2
    final_obv_factor = obv * ema_diff
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Apply Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    intraday_momentum_filtered = intraday_momentum.where(shock_filter, 0)
    
    # Measure Volume Synchronization
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_product = np.sign(volume_change * price_change)
    
    # Synthesize Combined Factor
    combined_factor = (smoothed_intraday_range 
                       + synchronized_product
                       + intraday_momentum_filtered)
    combined_factor = combined_factor.where(combined_factor > 0, 0)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * final_obv_factor
    
    return final_alpha_factor
