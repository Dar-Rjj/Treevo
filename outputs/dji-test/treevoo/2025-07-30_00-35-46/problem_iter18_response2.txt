```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = 0.5 * (high_low_diff / df['open']) + 0.5 * open_close_return

    # Volume and Amount Shock Filter
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    amount_change_ratio = df['amount'] / df['amount'].shift(1)
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    shock_filter = (volume_change_ratio > volume_shock_threshold) & (amount_change_ratio > amount_shock_threshold)

    # Trend Confirmation
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    crossover_signal = (short_term_ma > long_term_ma).astype(int) - (short_term_ma < long_term_ma).astype(int)
    trend_adjusted_momentum = intraday_momentum * crossover_signal

    # Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Adjust for Volume
    mean_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjustment = (df['volume'] / (mean_volume_5d + 1e-9)) + 1

    # Combine Intraday Range and Volume Adjustment
    combined_factor = intraday_range * volume_adjustment

    # Smoothing
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()

    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_cross_signal = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)

    # On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # Weight OBV by Difference between Short and Long EMA
    ema_diff = np.abs(short_ema - long_ema)
    weighted_obv = obv * ema_diff

    # Apply Dynamic Volume Shock Filter
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    dynamic_volume_ratio = df['volume'] / (avg_volume_5d + 1e-9)
    dynamic_shock_filter = dynamic_volume_ratio > 1.8
    filtered_smoothed_momentum = smoothed_factor.where(dynamic_shock_filter, other=0)

    # Final Alpha Factor
    final_alpha_factor = filtered_smoothed_momentum * weighted_obv

    # Introduce Volatility Factor
    daily_returns = df['close'].pct_change()
    daily_volatility = daily_returns.rolling(window=20).std()
    final_alpha_factor = final_alpha_factor * (1 + daily_volatility)

    return final_alpha_factor
```
