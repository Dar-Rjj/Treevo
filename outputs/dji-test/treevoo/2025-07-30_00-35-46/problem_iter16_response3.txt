```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Volume Adjustment
    volume_mean_5d = df['volume'].rolling(window=5).mean()
    volume_adjustment = (df['volume'] / (volume_mean_5d + 0.01))
    
    # Adjusted Intraday Range
    adjusted_intraday_range = intraday_range * volume_adjustment
    
    # Smoothed Intraday Range
    smoothed_intraday_range = adjusted_intraday_range.ewm(span=7, adjust=False).mean()
    
    # EMA Cross Signal
    short_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_ema = df['close'].ewm(span=25, adjust=False).mean()
    
    # Momentum
    momentum = np.where(short_ema > long_ema, 1, -1)
    
    # On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Weight OBV by EMA Difference
    weight = (short_ema - long_ema) ** 2
    final_obv_factor = obv * weight
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_intraday_range * final_obv_factor
    
    return final_alpha_factor
```
