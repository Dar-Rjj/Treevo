```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_mean_5d = df['volume'].rolling(window=5).mean().shift(1) + 1
    volume_adjustment = df['volume'] / volume_mean_5d
    
    # Combine Intraday Range and Volume Adjustment
    combined_intraday = intraday_range * volume_adjustment
    
    # Smoothing with Exponential Moving Average (EMA)
    ema_combined_intraday = combined_intraday.ewm(span=5, adjust=False).mean()
    
    # Calculate EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Momentum
    momentum = (short_ema > long_ema).astype(int) - (short_ema < long_ema).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[1:] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    obv.iloc[0] = 0  # Set the first value to 0
    
    # Weight OBV by the Difference between Short and Long EMA
    weight = np.abs(short_ema - long_ema)
    final_obv_factor = obv * weight
    
    # Combine Smoothed Intraday Momentum with OBV Factor
    smoothed_obv_factor = ema_combined_intraday * final_obv_factor
    
    # Calculate Change in Intraday Price Range
    change_intraday_range = intraday_range - intraday_range.shift(1)
    
    # Adjust for Volume Impact
    adjusted_spread = change_intraday_range * (df['volume'] - df['volume'].shift(1))
    
    # Aggregate Short-Term Momentum
    short_term_momentum = adjusted_spread.rolling(window=5).mean()
    
    # Combine Intraday Range and Volume Adjustment with Short-Term Momentum
    combined_with_short_term = ema_combined_intraday * volume_adjustment * short_term_momentum
    
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    intraday_momentum = 0.5 * high_low_diff + 0.5 * open_close_return
    
    # Apply Volume and Amount Shock Filter
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    shock_filter = (volume_ratio > 1.5) & (amount_ratio > 1.5)
    intraday_momentum_shock = intraday_momentum * shock_filter
    
    # Final Alpha Factor
    final_alpha_factor = (smoothed_obv_factor * short_term_momentum * intraday_momentum_shock)
    
    return final_alpha_factor
```
