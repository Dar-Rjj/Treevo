import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Modified PPUV (mPPUV)
    df['mPPUV'] = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['close'] / df['volume'])
    
    # N-day relative strength (RS) factor
    df['close_change'] = df['close'].diff()
    positive_changes = df['close_change'].clip(lower=0).rolling(window=n).sum()
    negative_changes = df['close_change'].clip(upper=0).abs().rolling(window=n).sum()
    df['RS'] = positive_changes / negative_changes.replace(0, 1)
    
    # Smoothed AHLR (sAHLR)
    df['AHLR'] = (df['high'] - df['low']) / df['close']
    df['sAHLR'] = df['AHLR'].rolling(window=m).mean()
    
    # Volume stability factor
    df['volume_std'] = df['volume'].rolling(window=n).std()
    df['stability_factor'] = 1 / (df['volume_std'] + 1e-6)  # Adding a small value to avoid division by zero
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (df['mPPUV'] + df['RS'] + df['sAHLR'] + df['stability_factor']) / 4
    
    return df['alpha_factor'].dropna()
