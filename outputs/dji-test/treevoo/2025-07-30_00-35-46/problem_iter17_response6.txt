import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['range'] = df['high'] - df['low']
    
    # Measure Close Position in Range
    df['close_position'] = (df['close'] - df['low']) / df['range']
    df['close_position'].fillna(0.5, inplace=True)  # Handle division by zero
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + 
                  df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_return'] = df['close'] - df['open']
    df['intraday_momentum'] = 0.6 * df['high_low_diff'] + 0.4 * df['open_close_return']
    
    # Synchronized Volume
    df['prev_volume'] = df['volume'].shift(1)
    df['volume_change'] = df['volume'] - df['prev_volume']
    df['prev_close'] = df['close'].shift(1)
    df['price_change'] = df['close'] - df['prev_close']
    df['volume_ratio'] = df['volume'] / df['prev_volume']
    df['shock_filter'] = df['volume_ratio'] > 1.5
    df['synchronized_volume'] = df['volume_change'] * df['price_change']
    df['synchronized_volume'] = df['synchronized_volume'].apply(np.sign)
    df['synchronized_volume'] *= df['shock_filter'].astype(int)
    
    # EMA Cross Signal
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_signal'] = (df['ema_5'] > df['ema_20']).astype(int) * 2 - 1
    
    # On-Balance Volume (OBV)
    df['obv'] = 0
    df['obv'] += np.where(df['close'] > df['close'].shift(1), df['volume'], 0)
    df['obv'] -= np.where(df['close'] < df['close'].shift(1), df['volume'], 0)
    
    # Combined Momentum and OBV Factor
    df['weight'] = np.abs(df['ema_5'] - df['ema_20'])
    df['combined_momentum_obv'] = df['obv'] * df['weight']
    
    # Apply Volume and Amount Shock Filter
    df['amount'] = df['close'] * df['volume']
    df['prev_amount'] = df['amount'].shift(1)
    df['amount_ratio'] = df['amount'] / df['prev_amount']
    df['filtered_intraday_momentum'] = df['intraday_momentum'] * (df['volume_ratio'] > 1.5) * (df['amount_ratio'] > 1.5)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_volatility'].rolling(window=5).std()
    
    # Combine Factors
    factors = [
        'range', 'close_position', 'vwap', 'filtered_intraday_momentum', 
        'intraday_volatility', 'synchronized_volume', 'combined_momentum_obv'
    ]
    recent_std = df[factors].rolling(window=5).std().add_suffix('_std')
    df = pd.concat([df, recent_std], axis=1)
    
    weights = 1 / df[[f'{factor}_std' for factor in factors]].replace(0, np.inf)
    weights = weights.div(weights.sum(axis=1), axis=0)
    
    df['alpha_factor'] = (df[factors] * weights).sum(axis=1)
    
    return df['alpha_factor']
