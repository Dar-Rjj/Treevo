```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Synchronized Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    synchronized_volume = np.sign(volume_ratio * price_change)
    
    # EMA Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_signal = (short_ema > long_ema).astype(int)  # 1 if Bullish, 0 if Bearish
    
    # On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Combined Alpha Factor
    alpha_factor = (intraday_momentum + synchronized_volume * ema_signal) + df['obv']
    
    return alpha_factor
```
