import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Open_Close_Return'] = (df['close'] / df['open']) - 1
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close']
    df['Weighted_Momentum'] = (df['High_Low_Diff'] + df['Open_Close_Return']) * df['Daily_Volatility']
    df['Non_Linear_Transformation'] = df['Weighted_Momentum'] ** 2

    # Apply Integrated Volume and Price Shock Filter
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Price_Shock_Ratio'] = abs(df['close'] - df['close'].shift(1)) / (df['high'].max() - df['low'].min())
    df['Filter'] = ((df['Volume_Ratio'] > 1.5) & (df['Amount_Ratio'] > 1.5) & (df['Price_Shock_Ratio'] > 0.05)).astype(int)

    # Measure Synchronized Volume and Price Changes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Synchronized_Changes'] = np.sign(df['Volume_Change'] * df['Price_Change']) * df['Filter']

    # Calculate EMAs and SMAs
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_Crossover_Signal'] = (df['Short_EMA'] > df['Long_EMA']).astype(int) * 2 - 1

    # Analyze Price Momentum
    df['ROC_5'] = df['close'].pct_change(5)
    df['RSI_14'] = 100 - (100 / (1 + (df['close'].diff(1) > 0).rolling(14).mean() / (1 - (df['close'].diff(1) > 0).rolling(14).mean())))

    # Investigate Trading Volume
    df['Avg_Volume_5'] = df['volume'].rolling(5).mean()
    df['Volume_Signal'] = (df['volume'] > df['Avg_Volume_5']).astype(int) * 2 - 1

    # Combine Multiple Factors
    df['Composite_Alpha_Factor'] = (df['EMA_Crossover_Signal'] * 0.4 +
                                    df['ROC_5'] * 0.3 +
                                    (df['RSI_14'] / 100) * 0.2 +
                                    df['Volume_Signal'] * 0.1)

    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Non_Linear_Transformation'] * 0.3 +
                                df['Synchronized_Changes'] * 0.3 +
                                df['Composite_Alpha_Factor'] * 0.4)

    return df['Final_Alpha_Factor']
