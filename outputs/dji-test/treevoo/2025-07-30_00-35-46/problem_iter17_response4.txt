```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given stock price data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values, indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted by the daily return
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Calculate the 10-day moving average of the volume weighted return
    df['ma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the relative range
    df['ma_relative_range'] = df['relative_range'].rolling(window=10).mean()
    
    # The alpha factor is the ratio of the moving average of the volume weighted return to the moving average of the relative range
    alpha_factor = df['ma_volume_weighted_return'] / df['ma_relative_range']
    
    return alpha_factor
```
