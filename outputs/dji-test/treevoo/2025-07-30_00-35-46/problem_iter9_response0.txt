import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns with weights
    weight_high_low = 0.6
    weight_open_close = 0.4
    intraday_momentum = weight_high_low * high_low_diff + weight_open_close * open_close_return
    
    # Measure Volume Synchronization with Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    price_change = df['close'] - df['close'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_shock_threshold = 1.5
    volume_shock_filter = (volume_ratio > volume_shock_threshold).astype(int)
    synchronized_changes = volume_change * price_change
    synchronized_product_sign = (synchronized_changes * volume_shock_filter).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Adjust for Trading Activity
    intraday_metrics = intraday_momentum * open_close_return
    volume_smoothed = df['volume'].rolling(window=5).mean()
    adjusted_metric = intraday_metrics / volume_smoothed
    
    # Final Alpha Factor
    alpha_factor = adjusted_metric * synchronized_product_sign
    
    return alpha_factor
