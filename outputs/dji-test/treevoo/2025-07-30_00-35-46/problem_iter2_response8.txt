```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low and Open-Close Returns
    intraday_momentum = (high_low_diff + open_close_return) / 2
    
    # Adjust for Volume Impact
    volume_change = df['volume'].pct_change()
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_intraday_momentum = intraday_momentum * volume_ratio
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Change in High-Low Spread
    change_high_low_spread = high_low_spread - high_low_spread.shift(1)
    
    # Aggregate Short-Term Momentum
    short_term_momentum = adjusted_intraday_momentum + change_high_low_spread
    
    # Apply Volume Shock Filter to Final Momentum
    volume_shock_threshold = 1.5
    final_momentum = short_term_momentum.where(volume_ratio > volume_shock_threshold, 0)
    
    return final_momentum
```
