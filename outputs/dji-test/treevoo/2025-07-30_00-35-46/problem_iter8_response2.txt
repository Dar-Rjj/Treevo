```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    open_close_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = 0.5 * high_low_diff + 0.5 * open_close_return
    
    # Apply Volume and Amount Shock Filter
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    volume_shock_threshold = 1.5
    amount_shock_threshold = 1.5
    shock_filter = (volume_ratio > volume_shock_threshold) & (amount_ratio > amount_shock_threshold)
    
    # Measure Volume Synchronization with Shock Filter
    price_change = df['close'] - df['close'].shift(1)
    volume_sync = np.sign(volume_change * price_change) * shock_filter
    
    # Calculate Exponential Moving Average (EMA) Cross Signal
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum = (short_ema > long_ema).astype(int) * 2 - 1  # Bullish (+1) or Bearish (-1)
    
    # Calculate On-Balance Volume (OBV)
    obv = pd.Series(0, index=df.index)
    close_diff = df['close'].diff()
    obv[close_diff > 0] = df['volume'][close_diff > 0]
    obv[close_diff < 0] = -df['volume'][close_diff < 0]
    obv = obv.cumsum()
    
    # Combine Momentum and OBV into a Factor
    ema_diff = np.abs(short_ema - long_ema)
    weighted_obv = obv * ema_diff
    momentum_and_obv_factor = momentum * weighted_obv
    
    # Combine Intraday Metrics
    intraday_metrics = high_low_diff * open_close_return * df['amount']
    volume_adjusted_intraday_metrics = intraday_metrics / df['volume']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        intraday_momentum + 
        volume_sync + 
        (momentum_and_obv_factor + volume_adjusted_intraday_metrics)
    )
    
    return final_alpha_factor
```
