```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series of calculated alpha factor values, indexed by date.
    """
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the weighted sum of volume and price difference
    weighted_sum = (price_diff * df['volume']).rolling(window=10).sum()
    
    # Calculate the alpha factor
    alpha_factor = avg_price - weighted_sum
    
    return alpha_factor
```
