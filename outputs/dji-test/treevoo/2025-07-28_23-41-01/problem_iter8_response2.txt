import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    close_open_delta = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    reversal_score = (df['close'] - df['open']) / (df['high'] - df['low'])
    volume_weighted_reversal = reversal_score * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    momentum_adjusted_reversal = (df['close'] - df['open']) / (df['high'] - df['low'])
    momentum_adjusted_reversal *= df['volume'] * (df['close'] - df['close'].shift(1))
    momentum_adjusted_reversal += (df['close'] - df['open']).where(close_open_delta > 0, -(df['open'] - df['close']))
    
    # Identify High-Low Range Ratio
    daily_high_low_range = df['high'] - df['low']
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = (df['close'] / df['close'].shift(7)) - 1
    twenty_one_day_return = (df['close'] / df['close'].shift(21)) - 1
    combined_returns = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Introduce Trend Indicator
    fifty_day_ma = df['close'].rolling(window=50).mean()
    two_hundred_day_ma = df['close'].rolling(window=200).mean()
    trend_indicator = (fifty_day_ma > two_hundred_day_ma).astype(int) * 2 - 1
    
    # Final Alpha Factor
    alpha_factor = combined_breakout_momentum * combined_returns * volume_activity * trend_indicator
    return alpha_factor
