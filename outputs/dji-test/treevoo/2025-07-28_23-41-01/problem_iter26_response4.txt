```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    df['21_day_sum_high_low_range'] = df['high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['21_day_sum_high_low_range']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Weight Close-to-Open Return by Volume
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Combine Breakout Strength with Weighted Close-to-Open Return
    df['combined_return'] = (df['breakout_strength'] * 0.6) + (df['weighted_close_to_open_return'] * 0.4)
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Integrate Combined Return and Momentum with Volume Activity
    df['integrated_component'] = (df['combined_return'] * 0.7) + (df['price_momentum'] * 0.5) + (df['volume_activity'] * 0.8)
    
    # Derive High-Low Range Signal
    df['21_day_avg_high_low_range'] = df['high_low_range'].rolling(window=21).mean()
    df['high_low_range_signal'] = (df['high_low_range'] > df['21_day_avg_high_low_range']).astype(int)
    
    # Integrate Integrated Component with High-Low Range Signal
    df['alpha_factor'] = (df['integrated_component'] * 0.7) + (df['high_low_range_signal'] * 0.3)
    
    return df['alpha_factor']
```
