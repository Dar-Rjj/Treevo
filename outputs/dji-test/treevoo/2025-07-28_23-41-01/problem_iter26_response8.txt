```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume and Amount Weighted Close-to-Open Return
    df['weighted_close_to_open'] = (df['close'] - df['open']) * (df['volume'] + df['amount'])
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    df['volume_weighted_return'] = df['volume'] * df['daily_return']
    
    # Combine Initial Factors
    df['combined_initial_factors'] = df['close_position_in_range'] + df['volume_weighted_return']
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['intraday_range'] / df['intraday_range'].rolling(window=21).sum()
    
    # Calculate Intraday Reversal Score
    df['intraday_close_to_open_delta'] = df['close'] - df['open']
    df['intraday_reversal_score'] = (df['intraday_close_to_open_delta'] / df['intraday_range']) * df['volume']
    
    # Combine Breakout and Intraday Factors
    df['combined_breakout_intraday_factors'] = df['combined_initial_factors'] + df['breakout_strength'] + df['intraday_reversal_score']
    
    # Calculate Short-Term Momentum
    df['7_day_return'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['14_day_return'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    df['short_term_momentum'] = df['7_day_return'] + df['14_day_return']
    
    # Calculate Long-Term Momentum
    df['21_day_return'] = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    df['63_day_return'] = (df['close'] - df['close'].shift(63)) / df['close'].shift(63)
    df['long_term_momentum'] = df['21_day_return'] + df['63_day_return']
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / mean_volume
    df['adjusted_factor'] = df['combined_breakout_intraday_factors'] * df['volume_ratio']
    
    # Incorporate Momentum
    df['momentum_factor'] = df['adjusted_factor'] + df['short_term_momentum'] + df['long_term_momentum']
    
    # Integrate Volume Momentum
    df['10_day_rolling_sum_volume'] = df['volume'].rolling(window=10).sum()
    df['10_day_rolling_sum_price_changes'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Introduce Trend Indicator
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = df['50_day_ma'] > df['200_day_ma']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['momentum_factor'] * df['volume_activity']
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['trend_indicator'].apply(lambda x: 1 if x else -1)
    
    return df['final_alpha_factor']
```
