def heuristics_v2(df):
    # Calculate the relative position of the closing price within the day's range
    df['day_range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['day_range']
    
    # Calculate the volume weighted close position
    df['volume_weighted_close_position'] = df['close_position'] * df['volume']
    
    # Calculate the 10-day moving average of the volume weighted close position
    df['vwcp_10_day_ma'] = df['volume_weighted_close_position'].rolling(window=10).mean()
    
    # The alpha factor is the difference between the current volume weighted close position and its 10-day moving average
    alpha_factor = df['volume_weighted_close_position'] - df['vwcp_10_day_ma']
    
    return alpha_factor
