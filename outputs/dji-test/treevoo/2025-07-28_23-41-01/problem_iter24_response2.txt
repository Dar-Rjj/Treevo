```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-day rolling sum)
    df['recent_avg_high_low_range'] = df['high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Combine Breakout Strength with Volume-Weighted Return
    df['combined_breakout_vw_return'] = df['breakout_strength'] * df['volume_weighted_return']
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Integrate Combined Breakout-VW Return with Price Momentum and Volume Activity
    df['integrated_component'] = (df['combined_breakout_vw_return'] * 0.8) + \
                                (df['price_momentum'] * 0.5) + \
                                (df['volume_activity'] * 0.7)
    
    # Incorporate High-Low Range Signal
    n = 21
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    df['high_low_range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Integrated Component with High-Low Range Signal
    df['integrated_with_high_low_signal'] = (df['integrated_component'] * 0.6) + \
                                           (df['high_low_range_signal'] * 0.4)
    
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Integrate RSI with Integrated Component
    df['final_factor'] = (df['integrated_with_high_low_signal'] * 0.7) + (df['rsi'] * 0.3)
    
    return df['final_factor']
```
