```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    recent_avg_high_low_range = daily_high_low_range.rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = (daily_high_low_range / recent_avg_high_low_range).fillna(0)
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff().fillna(0)
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_volume = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(7).fillna(0)
    twenty_one_day_return = df['close'].pct_change(21).fillna(0)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_vol = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_vol = df['volume'].rolling(window=21).mean()
    volume_activity = (seven_day_avg_vol / twenty_one_day_avg_vol).fillna(0)
    
    # Final Alpha Factor
    final_alpha_factor = combined_breakout_volume * price_momentum * volume_activity
    
    return final_alpha_factor
```
