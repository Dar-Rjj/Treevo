```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    df['rolling_range_sum'] = df['daily_range'].rolling(window=21).sum()
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    df['recent_avg_range'] = df['rolling_range_sum'] / df['rolling_volume_sum']
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['recent_avg_range']
    
    # Calculate Price Change
    df['price_change'] = df['close'].pct_change()
    
    # Calculate Volume Weighted Momentum
    df['vol_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['alpha_factor'] = df['breakout_strength'] * df['vol_weighted_momentum']
    
    return df['alpha_factor']
```
