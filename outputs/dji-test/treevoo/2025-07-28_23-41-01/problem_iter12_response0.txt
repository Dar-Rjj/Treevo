```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    df['close_to_open_delta'] = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    df['reversal_score'] = (df['close'] - df['open']) / df['intraday_range'] * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    df['momentum_adjusted_reversal'] = (df['close'] - df['open']) / df['intraday_range'] * (df['volume'] * (df['close'] - df['close'].shift(1)))
    df['momentum_adjusted_reversal'] += df['close_to_open_delta'].apply(lambda x: x if x > 0 else -x)
    
    # Identify High-Low Range Ratio
    df['daily_high_low_range'] = df['high'] - df['low']
    df['recent_avg_high_low_range'] = df['daily_high_low_range'].rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Volume Weighted Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_strength_momentum'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Introduce Trend Indicator
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = df.apply(lambda x: 1 if x['50_day_ma'] > x['200_day_ma'] else -1, axis=1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_breakout_strength_momentum'] * df['price_momentum'] * df['volume_activity'] * df['trend_indicator']
    
    # Incorporate Intraday Volatility
    df['10_day_ema_intraday_range'] = df['intraday_range'].ewm(span=10).mean()
    df['20_day_sma_intraday_range'] = df['intraday_range'].rolling(window=20).mean()
    df['final_alpha_factor'] *= df.apply(lambda x: 1.5 if x['10_day_ema_intraday_range'] > x['20_day_sma_intraday_range'] else 1.0, axis=1)
    
    return df['final_alpha_factor']
```
