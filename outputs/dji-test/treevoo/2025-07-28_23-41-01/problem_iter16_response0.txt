```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    reversal_score = (df['close'] - df['open']) / (df['high'] - df['low'])
    volume_weighted_reversal_score = reversal_score * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    momentum_adjusted_reversal_score = (df['close'] - df['open']) / (df['high'] - df['low'])
    momentum_adjusted_reversal_score *= df['volume'] * (df['close'] - df['close'].shift(1))
    momentum_adjusted_reversal_score += np.where(close_to_open_delta > 0, close_to_open_delta, -close_to_open_delta)
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    rolling_high_low_sum = intraday_range.rolling(window=21).sum()
    rolling_volume_sum = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = intraday_range / rolling_high_low_sum
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * (price_change * df['volume'])
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(periods=7)
    twenty_one_day_return = df['close'].pct_change(periods=21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Determine Volume Impact
    mean_volume = df['volume'].mean()
    volume_ratio = df['volume'] / mean_volume
    
    # Introduce Trend Indicator
    plus_di = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    minus_di = df['high'].rolling(window=14).min() - df['low'].rolling(window=14).max()
    dmi = (plus_di - minus_di) / (plus_di + minus_di)
    adx = dmi.ewm(span=14, adjust=False).mean()
    
    # Final Synthesized Alpha Factor
    intraday_component = volume_weighted_reversal_score + momentum_adjusted_reversal_score + combined_breakout_momentum
    momentum_trend_component = combined_breakout_momentum * price_momentum * volume_activity * adx
    alpha_factor = intraday_component + momentum_trend_component
    
    return alpha_factor
```
