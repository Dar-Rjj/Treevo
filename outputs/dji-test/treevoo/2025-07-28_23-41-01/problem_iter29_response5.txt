import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Weighted Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Daily High-Low Range and Recent Average High-Low Range
    high_low_range = df['high'] - df['low']
    recent_avg_high_low_range = high_low_range.rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = high_low_range / recent_avg_high_low_range
    
    # Combine Breakout Strength with Volume-Weighted Close-to-Open Return
    combined_breakout_vw_return = breakout_strength * volume_weighted_return
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['close'].shift(7)) - 1 + (df['close'] / df['close'].shift(21)) - 1
    
    # Calculate Volume Activity
    avg_volume_7 = df['volume'].rolling(window=7).mean()
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    volume_activity = avg_volume_7 / avg_volume_21
    
    # Integrate Combined Breakout-VW Return with Price Momentum and Volume Activity
    integrated_component = (combined_breakout_vw_return * 0.8) + (price_momentum * 0.5) + (volume_activity * 0.7)
    
    # Incorporate High-Low Range Signal
    n_day_avg_high_low_range = high_low_range.rolling(window=21).mean()
    high_low_range_signal = (high_low_range > n_day_avg_high_low_range).astype(int)
    
    # Integrate Integrated Component with High-Low Range Signal
    integrated_component_with_signal = (integrated_component * 0.6) + (high_low_range_signal * 0.4)
    
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Integrate RSI with Integrated Component
    integrated_rsi = (integrated_component_with_signal * 0.7) + (rsi * 0.3)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = df['volume'] * daily_return
    
    # Combine Initial Factors
    combined_factors = close_position_in_range + weighted_daily_return
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    integrated_combined_return_momentum = (integrated_rsi * 0.7) + (high_low_range_signal * 0.3)
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_combined_return_momentum * 0.7) + (price_momentum * 0.5) + (volume_activity * 0.8)
    
    return final_alpha_factor
