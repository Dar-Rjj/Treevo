```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-Day Rolling Sum)
    df['Avg_High_Low_Range'] = df['High_Low_Range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = df['High_Low_Range'] / df['Avg_High_Low_Range']
    
    # Calculate Volume Weighted Momentum
    df['Close_Change'] = df['close'].pct_change()
    df['Volume_Weighted_Momentum'] = df['Close_Change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['Combined_Breakout_VWM'] = df['Breakout_Strength'] * df['Volume_Weighted_Momentum']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume for Close-to-Open Return
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Combine Weighted Close-to-Open Return and Breakout-VW Momentum
    df['Combined_Return_Momentum'] = (df['Weighted_Close_to_Open_Return'] * 0.4) + (df['Combined_Breakout_VWM'] * 0.6)
    
    # Calculate Price Momentum (7-day and 21-day Returns)
    df['7_day_Return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_Return'] = (df['close'] / df['close'].shift(21)) - 1
    df['Price_Momentum'] = df['7_day_Return'] + df['21_day_Return']
    
    # Calculate Volume Activity (Ratio of 7-day to 21-day Volume Averages)
    df['7_day_Volume_Avg'] = df['volume'].rolling(window=7).mean()
    df['21_day_Volume_Avg'] = df['volume'].rolling(window=21).mean()
    df['Volume_Activity'] = df['7_day_Volume_Avg'] / df['21_day_Volume_Avg']
    
    # Final Alpha Factor
    df['Integrated_Component'] = (df['Combined_Return_Momentum'] * 0.7) + (df['Price_Momentum'] * 0.5) + (df['Volume_Activity'] * 0.8)
    
    # Incorporate High-Low Range Signal
    df['High_Low_Range_Signal'] = (df['High_Low_Range'] > df['Avg_High_Low_Range']).astype(int)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    df['Alpha_Factor'] = (df['Integrated_Component'] * 0.7) + (df['High_Low_Range_Signal'] * 0.3)
    
    return df['Alpha_Factor']
```
