```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price momentum by looking at the percentage change in closing prices over the last 10 days
    momentum_10_day = df['close'].pct_change(periods=10)
    
    # Calculate the buying pressure by looking at the ratio of volume on up days to total volume over the last 5 days
    up_days_volume = df[df['close'] > df['close'].shift(1)]['volume'].rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    buying_pressure = up_days_volume / total_volume
    
    # Define a range factor which is the difference between high and low divided by the open price
    range_factor = (df['high'] - df['low']) / df['open']
    
    # Combine the three factors into a single alpha factor
    alpha_factor = momentum_10_day + buying_pressure + range_factor
    
    return alpha_factor
```
