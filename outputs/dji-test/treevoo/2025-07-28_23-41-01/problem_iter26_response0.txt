```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-Day Rolling Sum)
    df['recent_avg_high_low_range'] = df['daily_high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['close'].pct_change() * df['volume']).fillna(0)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] / df['open'].shift(-1)) - 1
    
    # Weight by Volume for Close-to-Open Return
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vw_momentum'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Combine Weighted Close-to-Open Return and Breakout-VW Momentum
    df['combined_weighted_returns'] = 0.4 * df['weighted_close_to_open_return'] + 0.6 * df['combined_breakout_vw_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['daily_return'] * df['volume']
    
    # Combine Initial Factors
    df['combined_initial_factors'] = df['close_position_in_range'] + df['weighted_daily_return']
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    df['integrated_component'] = 0.7 * df['combined_weighted_returns'] + 0.3 * df['breakout_strength']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        0.7 * df['integrated_component'] +
        0.5 * df['price_momentum'] +
        0.8 * df['volume_activity']
    )
    
    return df['final_alpha_factor']
```
