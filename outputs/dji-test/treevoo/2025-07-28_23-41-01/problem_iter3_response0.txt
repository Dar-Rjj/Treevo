import pandas as pd

def heuristics_v2(df, n_days):
    # Calculate Daily High-to-Low Range
    df['range'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Average High-to-Low Range over N Days
    def volume_weighted_average_range(data, n):
        return (data['range'] * data['volume']).rolling(window=n).sum() / data['volume'].rolling(window=n).sum()
    
    df['vwa_range'] = volume_weighted_average_range(df, n_days)
    
    # Compare with Current High-to-Low Range
    df['signal'] = 0
    df.loc[df['range'] > df['vwa_range'], 'signal'] = 1
    
    return df['signal']
