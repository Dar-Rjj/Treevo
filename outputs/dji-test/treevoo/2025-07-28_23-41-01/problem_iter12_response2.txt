```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['volume'] * df['daily_return']
    
    # Combine Factors
    df['combined_factor'] = df['close_position_in_range'] + df['weighted_daily_return']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['vwap'] = vwap
    
    # Adjust for VWAP
    df['adjusted_factor'] = df['combined_factor'] - (df['close'] - df['vwap'])
    
    # Calculate Short-Term Momentum
    short_term_period = 10
    df['short_term_momentum'] = df['close'] - df['close'].shift(short_term_period)
    
    # Calculate Long-Term Momentum
    long_term_period = 50
    df['long_term_momentum'] = df['close'] - df['close'].shift(long_term_period)
    
    # Incorporate Momentum
    df['momentum_adjusted_factor'] = df['adjusted_factor'] + df['short_term_momentum'] + df['long_term_momentum']
    
    # Calculate Price Action Signal
    df['price_action_signal'] = (df['close'] > df['open']).astype(int) - (df['close'] < df['open']).astype(int)
    
    # Final Factor
    df['final_factor'] = df['momentum_adjusted_factor'] + 0.5 * df['price_action_signal']
    
    # Adjust for Trend
    trend_period = 20
    df['moving_average'] = df['close'].rolling(window=trend_period).mean()
    df['final_factor'] = df['final_factor'] + (df['close'] - df['moving_average'])
    
    return df['final_factor']
```
