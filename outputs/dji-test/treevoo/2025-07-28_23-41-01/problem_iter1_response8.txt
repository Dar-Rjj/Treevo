```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily returns
    df['daily_returns'] = df['close'].diff()
    
    # Calculate raw price momentum
    df['price_momentum'] = df['daily_returns'].rolling(window=20).mean()
    
    # Calculate volume change and logarithmic transformation
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Aggregate volume impact
    df['volume_impact'] = df['volume_change'].rolling(window=20).mean()
    
    # Combine momentum and volume impact
    df['factor'] = 0.7 * df['price_momentum'] + 0.3 * df['volume_impact']
    
    return df['factor']
```
