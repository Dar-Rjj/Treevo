```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    rolling_sum_high_low = daily_high_low_range.rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_high_low_range = rolling_sum_high_low / 21
    
    # Calculate Breakout Strength
    breakout_strength = daily_high_low_range / recent_avg_high_low_range
    
    # Calculate Price Change
    price_change = df['close'].pct_change()
    
    # Combine Breakout Strength with Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Final Alpha Factor
    alpha_factor = breakout_strength * volume_weighted_momentum
    
    return alpha_factor
```
