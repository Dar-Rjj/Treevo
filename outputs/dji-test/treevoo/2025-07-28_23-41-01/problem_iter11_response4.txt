```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    df['close_to_open_delta'] = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    df['reversal_score'] = (df['close_to_open_delta'] / df['intraday_range']) * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    df['intraday_momentum_reversal'] = (df['close_to_open_delta'] / df['intraday_range']) * (df['volume'] * (df['close'] - df['close'].shift(1)))
    df['intraday_momentum_reversal'] = df['intraday_momentum_reversal'] + (df['close_to_open_delta'].where(df['close_to_open_delta'] > 0, 0) - (-df['close_to_open_delta']).where(df['close_to_open_delta'] <= 0, 0))
    
    # Identify High-Low Range Ratio
    df['daily_high_low_range'] = df['intraday_range']
    df['recent_avg_high_low_range'] = df['intraday_range'].rolling(window=21).sum()
    df['recent_avg_volume'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Volume Momentum
    df['volume_momentum_10d'] = df['volume'].rolling(window=10).sum()
    df['price_change_10d'] = df['price_change'].rolling(window=10).sum()
    
    # Combine Indicators
    df['combined_indicators'] = (df['breakout_strength'] 
                                 + (df['price_change'] * df['volume']) 
                                 + df['reversal_score'] 
                                 + (df['price_change_10d'] * df['volume_momentum_10d']))
    
    # Calculate Price Momentum
    df['7d_return'] = df['close'].pct_change(periods=7)
    df['21d_return'] = df['close'].pct_change(periods=21)
    df['combined_returns'] = df['7d_return'] + df['21d_return']
    
    # Calculate Volume Activity
    df['7d_vol_avg'] = df['volume'].rolling(window=7).mean()
    df['21d_vol_avg'] = df['volume'].rolling(window=21).mean()
    df['volume_activity_ratio'] = df['7d_vol_avg'] / df['21d_vol_avg']
    
    # Introduce Trend Indicator
    df['50d_MA'] = df['close'].rolling(window=50).mean()
    df['200d_MA'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = (df['50d_MA'] > df['200d_MA']).astype(int) * 2 - 1
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_indicators'] * df['combined_returns'] * df['volume_activity_ratio'] * df['trend_indicator'])
    
    return df['final_alpha_factor']
```
