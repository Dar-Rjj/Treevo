```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] / df['close'].shift(1)) - 1
    
    # Apply Exponential Decay to Volume and Amount
    decay_factor = 0.5  # Example decay factor
    df['exp_volume'] = df['volume'] * np.exp(-decay_factor * (N - 1))
    df['exp_amount'] = df['amount'] * np.exp(-decay_factor * (N - 1))
    
    # Normalize the exponential decays for volume and amount
    df['norm_exp_volume'] = df['exp_volume'] / df['exp_volume'].rolling(window=N, min_periods=1).sum()
    df['norm_exp_amount'] = df['exp_amount'] / df['exp_amount'].rolling(window=N, min_periods=1).sum()
    
    # Aggregate Weighted Returns over Past N Days
    df['weighted_return'] = df['close_to_open_return'] * (df['norm_exp_volume'] + df['norm_exp_amount']) / 2
    df['alpha_factor'] = df['weighted_return'].rolling(window=N, min_periods=1).sum()
    
    return df['alpha_factor']
```
