```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    df['avg_daily_range'] = df['daily_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['avg_daily_range']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Combine Factors
    df['combined_factor'] = df['close_position'] + df['volume_weighted_return']
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine Breakout Strength with Volume-Weighted Momentum
    df['breakout_and_momentum'] = df['breakout_strength'] + df['volume_weighted_momentum']
    
    # Calculate 21-Day Rolling Sum of Volume
    df['avg_volume'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['reversal_score'] = (df['intraday_delta'] / df['intraday_range']) * df['volume']
    
    # Combine Indicators
    df['combined_indicators'] = df['breakout_and_momentum'] + df['reversal_score']
    
    # Adjust for Trend
    df['moving_avg_close'] = df['close'].rolling(window=21).mean()
    df['trend_adjustment'] = df['close'] - df['moving_avg_close']
    
    # Adjust for VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_adjustment'] = df['close'] - df['vwap']
    
    # Calculate Momentum Indicators
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['roc_50'] = df['close'].pct_change(periods=50)
    
    # Determine Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=21).mean()
    df['weighted_sma'] = df['sma_50'] * df['volume_ratio']
    df['weighted_ema'] = df['ema_50'] * df['volume_ratio']
    df['weighted_roc'] = df['roc_50'] * df['volume_ratio']
    
    # Combine Adjusted Momentum Indicators
    df['combined_momentum'] = df['weighted_sma'] + df['weighted_ema'] + df['weighted_roc']
    
    # Incorporate Momentum
    df['momentum_adjusted_factor'] = df['combined_factor'] + df['combined_indicators'] + df['combined_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['momentum_adjusted_factor'] + df['trend_adjustment']
    
    return df['final_alpha_factor']
```
