```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=30):
    # Calculate Daily High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average High-to-Low Range over N Days
    df['avg_range_N'] = df['daily_range'].rolling(window=N).mean()
    
    # Derive Momentum Signal from High-to-Low Range
    df['range_signal'] = (df['daily_range'] > df['avg_range_N']).astype(int)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    
    # Weight by Volume for Close-to-Open Return
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['agg_volume_impact'] = df['volume_change'].rolling(window=N).sum() / N
    
    # Combine Weighted Close-to-Open Return and Volume Impact
    df['combined_momentum'] = (df['weighted_close_to_open_return'] * 0.6) + (df['agg_volume_impact'] * 0.4)
    
    # Integrate Combined Close-to-Open and Volume Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.7) + (df['range_signal'] * 0.3)
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['avg_range_N']
    
    # Calculate Volume Weighted Momentum
    df['volume_weighted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vw_momentum'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Combine Weighted Close-to-Open Return and Breakout-VW Momentum
    df['final_combined_momentum'] = (df['weighted_close_to_open_return'] * 0.4) + (df['combined_breakout_vw_momentum'] * 0.6)
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_vol'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_vol'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_vol'] / df['21_day_avg_vol']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['integrated_momentum'] * 0.8) + (df['price_momentum'] * 0.5) + (df['volume_activity'] * 0.7)
    
    # Incorporate Integrated Return and Momentum with High-Low Range Signal
    df['final_alpha_factor'] = (df['alpha_factor'] * 0.8) + (df['range_signal'] * 0.2)
    
    return df['final_alpha_factor']
```
