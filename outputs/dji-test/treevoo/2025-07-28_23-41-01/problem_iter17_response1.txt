```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the relative close position within the day's range
    df['relative_close'] = (df['close'] - df['low']) / df['daily_range']
    
    # Calculate the average volume over a 5-day window
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Heuristic: If the stock closes in the upper half of the day's range and the volume is above the 5-day average,
    # it could signal positive momentum.
    df['signal'] = (df['relative_close'] > 0.5) & (df['volume'] > df['avg_volume_5d'])
    
    # Convert the boolean signal to an integer for the alpha factor
    df['alpha_factor'] = df['signal'].astype(int)
    
    return df['alpha_factor']
```
