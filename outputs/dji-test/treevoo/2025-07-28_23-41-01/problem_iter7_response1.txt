```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Average Daily Returns
    df['avg_daily_returns'] = df['daily_returns'].rolling(window=20).mean()
    
    # Calculate Logarithmic Volume Change
    df['log_volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Average Volume Impact
    df['avg_volume_impact'] = df['log_volume_change'].rolling(window=20).mean()
    
    # Combine Factors
    df['factor'] = 0.7 * df['avg_daily_returns'] + 0.3 * df['avg_volume_impact']
    
    return df['factor']
```
