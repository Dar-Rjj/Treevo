```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Returns
    df['daily_returns'] = df['close'] - df['close'].shift(1)
    
    # Calculate Raw Price Momentum
    df['raw_price_momentum'] = df['daily_returns'].rolling(window=20).mean()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['volume_impact'] = df['volume_change'].rolling(window=20).mean()
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = 0.7 * df['raw_price_momentum'] + 0.3 * df['volume_impact']
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['final_factor'] = 0.8 * df['combined_momentum'] + 0.2 * df['range_signal']
    
    return df['final_factor']
```
