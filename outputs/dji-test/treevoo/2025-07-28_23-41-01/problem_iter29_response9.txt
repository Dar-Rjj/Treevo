```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['volume'] * df['daily_return']
    
    # Combine Initial Factors
    df['combined_initial_factors'] = df['close_position_in_range'] + df['weighted_daily_return']
    
    # Calculate Breakout Strength
    df['avg_high_low_range_21'] = df['high'].rolling(window=21).max() - df['low'].rolling(window=21).min()
    df['breakout_strength'] = (df['high'] - df['low']) / df['avg_high_low_range_21']
    
    # Calculate Intraday Reversal Score
    df['intraday_close_to_open_delta'] = df['close'] - df['open']
    df['intraday_reversal_score'] = (df['intraday_close_to_open_delta'] / df['intraday_range']) * df['volume']
    
    # Combine Breakout and Intraday Factors
    df['combined_breakout_intraday'] = df['combined_initial_factors'] + df['breakout_strength'] + df['intraday_reversal_score']
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['21_day_return'] = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Determine Volume Impact
    df['mean_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / df['mean_volume']
    df['adjusted_factor'] = df['combined_breakout_intraday'] * df['volume_ratio']
    
    # Incorporate Momentum
    df['incorporated_momentum'] = df['adjusted_factor'] + df['price_momentum']
    
    # Integrate Volume Momentum
    df['10_day_rolling_sum_volume'] = df['volume'].rolling(window=10).sum()
    df['10_day_rolling_sum_price_changes'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Introduce Trend Indicator
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = (df['50_day_ma'] > df['200_day_ma']).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_breakout_intraday'] + df['volume_weighted_close_to_open_return'] + df['incorporated_momentum']) * df['volume_ratio'] * df['trend_indicator']
    
    return df['final_alpha_factor']
```
