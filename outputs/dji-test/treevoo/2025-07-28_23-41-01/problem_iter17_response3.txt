import pandas as pd

def heuristics(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    df['avg_high_low_range'] = (df['high_low_range'].rolling(window=21).sum() / 
                                df['volume'].rolling(window=21).sum())
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['reversal_score'] = df['intraday_delta'] / df['high_low_range'] * df['volume']
    
    # Calculate Price Change
    df['price_change'] = df['close'].pct_change()
    
    # Integrate Volume Momentum
    df['integrated_vol_momentum'] = (df['volume'].rolling(window=10).sum() * 
                                     df['price_change'].rolling(window=10).sum())
    
    # Calculate Relative Price Movement
    df['relative_price_move'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Enhance Volume Momentum with Relative Price Movement
    df['enhanced_vol_momentum'] = df['integrated_vol_momentum'] * df['relative_price_move']
    
    # Combine Indicators
    df['combined_indicators'] = (df['breakout_strength'] + 
                                 df['price_change'] * df['volume'] + 
                                 df['reversal_score'] + 
                                 df['enhanced_vol_momentum'])
    
    # Calculate Volume Weighted Momentum
    df['vol_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vol_mom'] = df['breakout_strength'] * df['vol_weighted_momentum']
    
    # Calculate Price Momentum
    df['price_momentum_7d'] = (df['close'] / df['close'].shift(7)) - 1
    df['price_momentum_21d'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['price_momentum_7d'] + df['price_momentum_21d']
    
    # Calculate Volume Activity
    df['avg_vol_7d'] = df['volume'].rolling(window=7).mean()
    df['avg_vol_21d'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['avg_vol_7d'] / df['avg_vol_21d']
    
    # Simple Moving Average (SMA)
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Exponential Moving Average (EMA)
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Rate of Change (ROC)
    df['roc'] = df['close'].pct_change(periods=14)
    
    # Determine Volume Impact
    df['volume_ratio'] = df['avg_vol_7d'] / df['avg_vol_21d']
    df['weighted_sma'] = df['sma_50'] * df['volume_ratio']
    df['weighted_ema'] = df['ema_10'] * df['volume_ratio']
    df['weighted_roc'] = df['roc'] * df['volume_ratio']
    
    # Trend Indicator (DMI)
    df['+di'] = 100 * (df['high'].diff().clip(lower=0) / df['high_low_range']).ewm(span=14, adjust=False).mean()
    df['-di'] = 100 * (-df['low'].diff().clip(upper=0) / df['high_low_range']).ewm(span=14, adjust=False).mean()
    df['adx'] = 100 * (abs(df['+di'] - df['-di']) / (df['+di'] + df['-di'])).ewm(span=14, adjust=False).mean()
