```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    df['avg_high_low_range_21'] = (df['high_low_range'].rolling(window=21).sum() / 
                                   df['volume'].rolling(window=21).sum())
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['avg_high_low_range_21']
    
    # Calculate Volume Weighted Momentum
    df['price_change'] = df['close'].diff()
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_momentum'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = df['close'].pct_change(7)
    df['21_day_return'] = df['close'].pct_change(21)
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Adjust for Volatility
    df['21_day_std'] = df['close'].rolling(window=21).std()
    df['adjusted_breakout_momentum'] = df['combined_breakout_momentum'] / df['21_day_std']
    
    # Introduce Trend Indicator
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = (df['50_day_ma'] > df['200_day_ma']).astype(int) * 2 - 1
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_breakout_momentum'] * 
                                df['price_momentum'] * 
                                df['volume_activity'] * 
                                df['trend_indicator'])
    
    return df['final_alpha_factor']
```
