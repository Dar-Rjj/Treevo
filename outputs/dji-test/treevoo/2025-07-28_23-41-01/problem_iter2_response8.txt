```python
import pandas as pd

def heuristics_v2(df):
    # Compute Simple Moving Averages (SMA) for different periods
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()

    # Compute Exponential Moving Averages (EMA) for different periods
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Compute Rate of Change (ROC) over a period
    roc_14 = df['close'].pct_change(periods=14)

    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=30).mean()

    # Assign Weight to Indicators using Volume Ratio
    sma_10_weighted = sma_10 * volume_ratio
    sma_20_weighted = sma_20 * volume_ratio
    ema_10_weighted = ema_10 * volume_ratio
    ema_20_weighted = ema_20 * volume_ratio
    roc_14_weighted = roc_14 * volume_ratio

    # Combine Adjusted Momentum Indicators
    adjusted_momentum = (sma_10_weighted + sma_20_weighted + 
                         ema_10_weighted + ema_20_weighted + roc_14_weighted)

    # Identify High-Low Range Ratio
    high_low_range = df['high'] - df['low']
    recent_avg_high_low_range = high_low_range.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()

    # Calculate Breakout Strength
    breakout_strength = high_low_range / recent_avg_high_low_range

    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']

    # Synthesize Hybrid Indicator
    hybrid_indicator = (adjusted_momentum + breakout_strength + volume_weighted_momentum) / 3

    return hybrid_indicator
```
