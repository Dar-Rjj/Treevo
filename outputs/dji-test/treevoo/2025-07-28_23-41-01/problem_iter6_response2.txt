```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    daily_returns = df['close'].diff()
    daily_returns_20 = daily_returns.rolling(window=20).mean()

    # Calculate Volume Change
    volume_change = np.log(df['volume'] / df['volume'].shift(1))
    volume_impact = volume_change.rolling(window=20).sum() / 20

    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=20).mean()

    # Adjusted Volume Impact
    adjusted_volume_impact = volume_impact * (1 + average_daily_range / 100)
    adjusted_volume_impact = adjusted_volume_impact.apply(lambda x: max(x, 0))

    # Combine Factors
    factor = (daily_returns_20 * 0.6) + (adjusted_volume_impact * 0.4)

    return factor
```
