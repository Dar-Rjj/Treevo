```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_range = intraday_range.rolling(window=21).sum()
    breakout_strength = intraday_range / recent_avg_range
    
    # Calculate Intraday Reversal Score
    close_to_open_delta = df['close'] - df['open']
    intraday_reversal_score = (close_to_open_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_factors = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    seven_day_return = (df['close'] / df['close'].shift(7)) - 1
    twenty_one_day_return = (df['close'] / df['close'].shift(21)) - 1
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Incorporate Momentum
    adjusted_factor = combined_factors + price_momentum
    
    # Introduce Trend Indicator
    fifty_day_ma = df['close'].rolling(window=50).mean()
    two_hundred_day_ma = df['close'].rolling(window=200).mean()
    
    trend_indicator = (fifty_day_ma > two_hundred_day_ma).astype(int)
    trend_indicator = trend_indicator.where(trend_indicator == 1, -1)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor * trend_indicator
    
    # Calculate Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=21).mean()
    
    # Incorporate Relative Volume
    enhanced_final_alpha_factor = final_alpha_factor + relative_volume
    
    # Final Enhanced Alpha Factor
    final_enhanced_alpha_factor = enhanced_final_alpha_factor * trend_indicator
    
    return final_enhanced_alpha_factor
```
