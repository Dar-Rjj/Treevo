```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_pos_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    
    # Weighted Daily Return
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_factors = close_pos_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = ((df['close'] - df['open']) / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change(7).fillna(0) + df['close'].pct_change(21).fillna(0)
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=21).mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    factor_with_momentum = adjusted_factor + price_momentum
    
    # Introduce Trend Indicator
    trend_indicator = (df['close'].rolling(window=50).mean() > df['close'].rolling(window=200).mean()).astype(int)
    final_factor = factor_with_momentum * trend_indicator - factor_with_momentum * (1 - trend_indicator)
    
    return final_factor
```
