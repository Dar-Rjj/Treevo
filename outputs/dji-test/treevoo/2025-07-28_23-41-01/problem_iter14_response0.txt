import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    df['rolling_daily_returns'] = df['daily_returns'].rolling(window=20).sum()

    df['average_daily_returns'] = df['rolling_daily_returns'] / 20
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['aggregate_volume_impact'] = df['volume_change'].rolling(window=20).sum() / 20

    df['combined_momentum'] = (df['average_daily_returns'] * 0.7) + (df['aggregate_volume_impact'] * 0.3)

    df['high_to_low_range'] = df['high'] - df['low']
    df['n_day_average_high_to_low_range'] = df['high_to_low_range'].rolling(window=20).mean()
    df['range_signal'] = (df['high_to_low_range'] > df['n_day_average_high_to_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.8) + (df['range_signal'] * 0.2)

    df['recent_avg_high_low_range'] = df['high_to_low_range'].rolling(window=21).sum()
    df['recent_avg_volume'] = df['volume'].rolling(window=21).sum()

    df['current_high_low_range'] = df['high'] - df['low']
