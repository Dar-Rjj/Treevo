def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    volume_weighted_reversal_score = (close_to_open_delta / intraday_range) * df['volume']
    
    # Calculate Intraday Momentum Adjusted Reversal Score
    previous_close = df['close'].shift(1)
    intraday_momentum_adjustment = (df['close'] - previous_close) * df['volume']
    intraday_momentum_adjusted_reversal_score = (close_to_open_delta / intraday_range) * intraday_momentum_adjustment
    
    # Conditional Adjustment
    conditional_adjustment = close_to_open_delta.apply(lambda x: x if x > 0 else -x)
    intraday_momentum_adjusted_reversal_score += conditional_adjustment
    
    return intraday_momentum_adjusted_reversal_score
