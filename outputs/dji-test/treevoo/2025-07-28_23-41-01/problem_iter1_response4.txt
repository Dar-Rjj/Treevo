```python
import pandas as pd

def heuristics_v2(df):
    # Compute 7-day and 21-day returns
    df['7_day_return'] = df['close'].pct_change(7)
    df['21_day_return'] = df['close'].pct_change(21)
    
    # Combine the returns
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Compute 7-day and 21-day average volumes
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    
    # Combine the volume averages
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Combine price momentum and volume activity to get the final alpha factor
    df['alpha_factor'] = df['price_momentum'] * df['volume_activity']
    
    return df['alpha_factor'].dropna()
```
