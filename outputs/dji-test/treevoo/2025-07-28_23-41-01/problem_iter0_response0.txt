def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns 'open', 'high', 'low', 'close', and 'volume', indexed by 'date'.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by 'date'.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(-1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
