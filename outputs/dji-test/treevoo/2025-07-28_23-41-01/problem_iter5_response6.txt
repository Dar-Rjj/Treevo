import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Average Daily Returns over a rolling 20-day period
    df['avg_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['avg_volume_impact'] = df['volume_change'].rolling(window=20).mean()
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['avg_daily_return'] * 0.7) + (df['avg_volume_impact'] * 0.3)
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['alpha_factor'] = (df['combined_momentum'] * 0.8) + (df['range_signal'] * 0.2)
    
    return df['alpha_factor'].dropna()
