```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['daily_returns'] = df['close'].diff()
    df['rolling_daily_returns'] = df['daily_returns'].rolling(window=20).sum()
    df['raw_price_momentum'] = df['rolling_daily_returns'] / 20

    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['rolling_volume_change'] = df['volume_change'].rolling(window=20).sum()
    df['volume_impact'] = df['rolling_volume_change'] / 20

    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['raw_price_momentum'] * 0.7) + (df['volume_impact'] * 0.3)

    # Calculate Momentum Indicators
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['roc_5'] = df['close'].pct_change(periods=5)

    # Determine Volume Impact on Momentum Indicators
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['weighted_sma'] = df['sma_5'] * df['volume_ratio']
    df['weighted_ema'] = df['ema_5'] * df['volume_ratio']
    df['weighted_roc'] = df['roc_5'] * df['volume_ratio']

    # Combine Adjusted Momentum Indicators
    df['adjusted_momentum_indicators'] = df['weighted_sma'] + df['weighted_ema'] + df['weighted_roc']

    # Identify High-Low Range Ratio
    df['high_low_range'] = df['high'] - df['low']
    df['rolling_high_low_range'] = df['high_low_range'].rolling(window=21).sum()
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    df['recent_average_high_low_range'] = df['rolling_high_low_range'] / df['rolling_volume_sum']

    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['recent_average_high_low_range']

    # Incorporate High-to-Low Range Momentum
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=21).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)

    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.8) + (df['range_signal'] * 0.2)

    # Synthesize Final Indicator
    df['final_indicator'] = (df['integrated_momentum'] * 0.5) + \
                            (df['adjusted_momentum_indicators'] * 0.4) + \
                            (df['breakout_strength'] * 0.05) + \
                            (df['range_signal'] * 0.05)

    return df['final_indicator']
```
