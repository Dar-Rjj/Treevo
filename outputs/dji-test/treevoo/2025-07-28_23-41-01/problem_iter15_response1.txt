```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    reversal_score = (df['close'] - df['open']) / intraday_range * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    momentum_adjusted_reversal_score = (df['close'] - df['open']) / intraday_range * (df['volume'] * (df['close'] - df['close'].shift(1)))
    momentum_adjusted_reversal_score += (df['close'] - df['open']).where(df['close'] > df['open'], -(df['open'] - df['close']))
    
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    rolling_sum_high_low_range = daily_high_low_range.rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = daily_high_low_range / rolling_sum_high_low_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * price_change * df['volume']
    
    # Identify High-Low Range Ratio
    high_low_range_ratio = rolling_sum_high_low_range / rolling_sum_volume
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['open']) / intraday_range * df['volume']
    
    # Calculate Price Momentum
    seven_day_return = df['close'] / df['close'].shift(7) - 1
    twenty_one_day_return = df['close'] / df['close'].shift(21) - 1
    combined_returns = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Calculate Momentum Indicators
    sma_21 = df['close'].rolling(window=21).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    roc_21 = df['close'].pct_change(periods=21)
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    weighted_sma = sma_21 * volume_ratio
    weighted_ema = ema_21 * volume_ratio
    weighted_roc = roc_21 * volume_ratio
    
    # Introduce Trend Indicator
    plus_di = (df['high'] - df['high'].shift(1)).abs().rolling(window=21).sum()
    minus_di = (df['low'].shift(1) - df['low']).abs().rolling(window=21).sum()
    dmi = (plus_di - minus_di) / (plus_di + minus_di)
    adx = 100 * (dmi.abs()).ewm(span=21, adjust=False).mean()
    
    # Combine Adjusted Momentum Indicators with Trend Indicator
    adjusted_momentum_trend = (weighted_sma + weighted_ema + weighted_roc + adx)
    
    # Final Synthesized Alpha Factor
    final_factor = (reversal_score + momentum_adjusted_reversal_score + combined_breakout_momentum) * combined_returns * volume_activity * adjusted_momentum_trend
    
    return final_factor
```
