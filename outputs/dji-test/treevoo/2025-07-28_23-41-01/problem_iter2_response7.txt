```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate Daily High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average High-to-Low Range over N Days
    df['avg_N_day_range'] = df['daily_range'].rolling(window=N).mean()
    
    # Compare with Current High-to-Low Range
    df['range_momentum_signal'] = (df['daily_range'] > df['avg_N_day_range']).astype(int)
    
    # Derive High-to-Low Range Momentum Signal
    df['range_momentum_signal'] = df['range_momentum_signal'].fillna(0)
    
    # Calculate Raw Price Momentum
    df['daily_returns'] = df['close'] - df['close'].shift(1)
    df['raw_price_momentum'] = df['daily_returns'].rolling(window=N).mean()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['volume_impact'] = df['volume_change'].rolling(window=N).mean()
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum_impact'] = 0.7 * df['raw_price_momentum'] + 0.3 * df['volume_impact']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = 0.5 * df['range_momentum_signal'] + 0.5 * df['combined_momentum_impact']
    
    return df['alpha_factor']
```
