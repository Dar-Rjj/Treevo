```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume and Amount
    df['Volume_Amount_Weighted_Return'] = df['Close_to_Open_Return'] * (df['volume'] + df['amount'])
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['Close_Position_in_Range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    df['Volume_Weighted_Daily_Return'] = df['Daily_Return'] * df['volume']
    
    # Combine Initial Factors
    df['Initial_Factor'] = df['Close_Position_in_Range'] + df['Volume_Weighted_Daily_Return']
    
    # Calculate Breakout Strength
    df['High_Low_Range'] = df['high'] - df['low']
    df['Recent_Average_High_Low_Range'] = df['High_Low_Range'].rolling(window=21).sum()
    df['Breakout_Strength'] = df['High_Low_Range'] / df['Recent_Average_High_Low_Range']
    
    # Calculate Intraday Reversal Score
    df['Intraday_Close_to_Open_Delta'] = df['close'] - df['open']
    df['Intraday_Reversal_Score'] = (df['Intraday_Close_to_Open_Delta'] / df['Intraday_Range']) * df['volume']
    
    # Combine Breakout and Intraday Factors
    df['Combined_Breakout_Intraday_Factors'] = df['Initial_Factor'] + df['Breakout_Strength'] + df['Intraday_Reversal_Score']
    
    # Calculate Price Momentum
    df['7_day_Return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_Return'] = (df['close'] / df['close'].shift(21)) - 1
    df['Price_Momentum'] = df['7_day_Return'] + df['21_day_Return']
    
    # Determine Volume Impact
    df['Volume_Ratio'] = df['volume'] / df['volume'].mean()
    df['Adjusted_Factor'] = df['Combined_Breakout_Intraday_Factors'] * df['Volume_Ratio']
    
    # Incorporate Momentum
    df['Momentum_Adjusted_Factor'] = df['Adjusted_Factor'] + df['Price_Momentum']
    
    # Integrate Volume Momentum
    df['10_Day_Volume_Sum'] = df['volume'].rolling(window=10).sum()
    df['10_Day_Price_Change_Sum'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()
    
    # Calculate Volume Activity
    df['7_day_Volume_Avg'] = df['volume'].rolling(window=7).mean()
    df['21_day_Volume_Avg'] = df['volume'].rolling(window=21).mean()
    df['Volume_Activity'] = df['7_day_Volume_Avg'] / df['21_day_Volume_Avg']
    
    # Introduce Trend Indicator
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    df['200_day_MA'] = df['close'].rolling(window=200).mean()
    df['Trend_Indicator'] = df['50_day_MA'] > df['200_day_MA']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Momentum_Adjusted_Factor'] * df['Volume_Activity']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * (df['Trend_Indicator'].astype(int) * 2 - 1)
    
    return df['Final_Alpha_Factor']
```
