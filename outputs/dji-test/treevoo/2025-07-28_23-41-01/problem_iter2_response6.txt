```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    rolling_sum_range = daily_range.rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_range = rolling_sum_range / rolling_sum_volume
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Intraday Range Factor
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    reversal_score = (close_to_open_delta / intraday_range) * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Indicators
    combined_factor = (breakout_strength + volume_weighted_momentum + reversal_score) / 3
    
    return combined_factor
```
