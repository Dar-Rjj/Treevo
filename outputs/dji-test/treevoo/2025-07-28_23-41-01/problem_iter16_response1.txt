```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    daily_returns = df['close'].diff()
    rolling_returns = daily_returns.rolling(window=20).mean()
    
    # Combine Weighted Return and Price Momentum
    combined_signal = (volume_weighted_return * 0.4) + (rolling_returns * 0.6)
    
    # Incorporate High-Low Range Signal
    high_low_range = df['high'] - df['low']
    n_day_average_range = high_low_range.rolling(window=20).mean()
    high_low_range_signal = (high_low_range > n_day_average_range).astype(int)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    final_factor = (combined_signal * 0.8) + (high_low_range_signal * 0.2)
    
    return final_factor
```
