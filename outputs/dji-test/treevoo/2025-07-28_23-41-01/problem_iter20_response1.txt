```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()

    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['aggregate_volume_impact'] = df['volume_change'].rolling(window=30).sum() / 30

    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['daily_returns'] * 0.6) + (df['aggregate_volume_impact'] * 0.4)

    # Incorporate High-to-Low Range Momentum
    df['high_to_low_range'] = df['high'] - df['low']
    df['avg_high_to_low_range'] = df['high_to_low_range'].rolling(window=30).mean()
    df['range_signal'] = (df['high_to_low_range'] > df['avg_high_to_low_range']).astype(int)

    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.7) + (df['range_signal'] * 0.3)

    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_to_low_range'] / df['high_to_low_range'].rolling(window=30).mean()

    # Calculate Intraday Close-to-Open Delta
    df['intraday_close_open_delta'] = df['close'] - df['open']

    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_close_open_delta'] / df['high_to_low_range']) * df['volume']

    # Calculate Price Change
    df['price_change'] = df['close'].diff()

    # Integrate Volume Momentum
    df['rolling_volume_sum'] = df['volume'].rolling(window=10).sum()
    df['rolling_price_change_sum'] = df['price_change'].rolling(window=10).sum()
    df['volume_momentum'] = df['rolling_volume_sum'] * df['rolling_price_change_sum']

    # Calculate Relative Price Movement
    df['relative_price_movement'] = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Enhance Volume Momentum with Relative Price Movement
    df['enhanced_volume_momentum'] = df['volume_momentum'] * df['relative_price_movement']

    # Calculate Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']

    # Combine Indicators
    df['combined_indicators'] = (df['breakout_strength'] * df['volume_weighted_momentum']) + \
                                (df['price_change'] * df['volume']) + \
                                df['intraday_reversal_score'] + \
                                df['enhanced_volume_momentum']

    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']

    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']

    # Volume Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']

    # Final Alpha Factor
    df['alpha_factor'] = (df['integrated_momentum'] * df['combined_indicators']) * \
                         (df['price_momentum'] * df['volume_activity']) + \
                         df['volume_weighted_close_to_open_return']

    return df['alpha_factor']
```
