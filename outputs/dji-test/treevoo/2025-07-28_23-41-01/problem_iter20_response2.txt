```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-day rolling sum)
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Combine Breakout Strength with Volume-Weighted Return
    df['combined_breakout_vw_return'] = df['breakout_strength'] * df['volume_weighted_return']
    
    # Calculate Price Momentum (7-day and 21-day returns)
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity (7-day to 21-day volume averages ratio)
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Integrate Combined Breakout-VW Return with Price Momentum and Volume Activity
    integrated_component = (0.8 * df['combined_breakout_vw_return']) + \
                           (0.5 * df['price_momentum']) + \
                           (0.7 * df['volume_activity'])
    
    # Incorporate High-Low Range Signal
    n_days = 21
    df['avg_high_low_range_n_days'] = df['high_low_range'].rolling(window=n_days).mean()
    df['high_low_range_signal'] = (df['high_low_range'] > df['avg_high_low_range_n_days']).astype(int)
    
    # Integrate Combined Component with High-Low Range Signal
    final_factor = (0.8 * integrated_component) + (0.2 * df['high_low_range_signal'])
    
    return final_factor
```
