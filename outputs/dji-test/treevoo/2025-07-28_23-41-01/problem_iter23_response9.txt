```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns: open, high, low, close, volume.
        
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the difference between high and low prices
    hl_diff = df['high'] - df['low']
    
    # Calculate the ratio of the difference (high - low) to the close price
    hl_ratio = hl_diff / df['close']
    
    # Calculate the volume-weighted close price
    vw_close = df['close'] * df['volume']
    
    # Calculate the 10-day moving average of the volume-weighted close price
    vw_close_ma_10 = vw_close.rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume-weighted close price
    vw_close_ma_30 = vw_close.rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day to 30-day moving averages of the volume-weighted close price
    ma_ratio = vw_close_ma_10 / vw_close_ma_30
    
    # Combine the factors into a single score
    factor = (close_pct_change + hl_ratio + ma_ratio) / 3
    
    return factor
```
