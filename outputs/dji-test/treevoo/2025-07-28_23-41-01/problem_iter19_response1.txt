```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    daily_returns = df['close'].diff()
    price_momentum = daily_returns.rolling(window=20).mean()
    
    # Combine Weighted Return and Price Momentum
    weighted_volume_return = 0.4 * volume_weighted_return
    weighted_price_momentum = 0.6 * price_momentum
    combined_value = weighted_volume_return + weighted_price_momentum
    
    # Incorporate High-Low Range Signal
    high_low_range = df['high'] - df['low']
    n_day_avg_range = high_low_range.rolling(window=20).mean()
    signal = (high_low_range > n_day_avg_range).astype(int)
    
    # Integrate Combined Value and High-Low Range Signal
    weighted_combined_value = 0.8 * combined_value
    weighted_signal = 0.2 * signal
    final_factor = weighted_combined_value + weighted_signal
    
    return final_factor
```
