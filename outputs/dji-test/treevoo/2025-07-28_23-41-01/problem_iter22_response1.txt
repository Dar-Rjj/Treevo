import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = df['volume'] * daily_return
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    breakout_strength = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=21).sum()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] / df['close'].shift(7) - 1) + (df['close'] / df['close'].shift(14) - 1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] / df['close'].shift(21) - 1) + (df['close'] / df['close'].shift(63) - 1)
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['close'].shift(7) - 1) + (df['close'] / df['close'].shift(21) - 1)
    
    # Calculate Volume Activity
    volume_7day_avg = df['volume'].rolling(window=7).mean()
    volume_21day_avg = df['volume'].rolling(window=21).mean()
    volume_activity = volume_7day_avg / volume_21day_avg
    
    # Determine Volume Impact
    volume_impact = combined_breakout_intraday * volume_activity
    
    # Incorporate Momentum
    momentum_adjusted_factor = volume_impact + short_term_momentum + long_term_momentum
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=21).mean()
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted_factor + trend_adjustment
    
    return final_alpha_factor
