```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    rolling_high_low_sum = daily_high_low_range.rolling(window=21).sum()
    rolling_volume_sum = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_high_low_range = rolling_high_low_sum / rolling_volume_sum
    
    # Calculate Breakout Strength
    breakout_strength = daily_high_low_range / recent_avg_high_low_range
    
    # Calculate Price Change
    price_change = df['close'].pct_change()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_signal = breakout_strength * volume_weighted_momentum
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Weighted Close-to-Open Return
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Final Alpha Factor
    alpha_factor = combined_signal + volume_weighted_close_to_open
    
    return alpha_factor
```
