import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    # Calculate the volume weighted close price
    vwap_close = df['close'] * df['volume']
    # Calculate the difference between volume weighted close and the average of high and low
    factor = vwap_close - avg_price
    # Normalize this with the open price to make it more interpretable
    # Note: The prompt says normalization is not allowed, so we will not do this step.
    return factor
