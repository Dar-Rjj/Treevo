```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'].pct_change()).fillna(0)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday_factors = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = (df['close'] / df['close'].shift(7)) - 1
    twenty_one_day_return = (df['close'] / df['close'].shift(21)) - 1
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Determine Volume Impact
    mean_volume = df['volume'].mean()
    volume_ratio = df['volume'] / mean_volume
    adjusted_factor = combined_breakout_intraday_factors * volume_ratio
    
    # Incorporate Momentum
    momentum_adjusted_factor = adjusted_factor + price_momentum
    
    # Adjust for Trend
    moving_average_close = df['close'].rolling(window=21).mean()
    trend_adjustment = df['close'] - moving_average_close
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted_factor + trend_adjustment
    
    return final_alpha_factor
```
