def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'volume'].
        
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price
    pct_change_avg_price = avg_price.pct_change()
    
    # Calculate the volume weighted close price
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the difference between the current close price and the VWAP
    diff_close_vwap = df['close'] - vwap
    
    # Generate the alpha factor
    alpha_factor = pct_change_avg_price * diff_close_vwap
    
    return alpha_factor
