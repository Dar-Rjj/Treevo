```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['volume_impact'] = df['volume_change'].rolling(window=30).sum() / 30
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['daily_returns'] * 0.6) + (df['volume_impact'] * 0.4)
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    n = 30
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.7) + (df['range_signal'] * 0.3)
    
    # Calculate Breakout Strength
    df['recent_avg_high_low_range'] = df['high_low_range'].rolling(window=10).mean()
    df['breakout_strength'] = df['high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_delta'] / df['high_low_range']) * df['volume']
    
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Integrate Volume Momentum
    df['volume_10_day_sum'] = df['volume'].rolling(window=10).sum()
    df['price_change_10_day_sum'] = df['price_change'].rolling(window=10).sum()
    
    # Calculate Relative Price Movement
    df['relative_price_movement'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Enhance Volume Momentum with Relative Price Movement
    df['enhanced_volume_momentum'] = (df['price_change_10_day_sum'] / df['volume_10_day_sum']) * df['relative_price_movement']
    
    # Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Indicators
    df['combined_indicators'] = (df['breakout_strength'] * df['volume_weighted_momentum']) + \
                                (df['price_change'] * df['volume']) + \
                                df['intraday_reversal_score'] + \
                                df['enhanced_volume_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_volume_avg'] = df['volume'].rolling(window=7).mean()
    df['21_day_volume_avg'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_volume_avg'] / df['21_day_volume_avg']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_indicators'] * 
                                df['price_momentum'] * 
                                df['volume_activity'])
    
    return df['final_alpha_factor']
```
