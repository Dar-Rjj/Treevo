```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, decay_rate=0.95):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Apply Exponential Decay to Volume and Amount
    df['decayed_volume'] = df['volume'].apply(lambda x: x * (decay_rate ** np.arange(len(x)))[::-1])
    df['decayed_amount'] = df['amount'].apply(lambda x: x * (decay_rate ** np.arange(len(x)))[::-1])
    
    # Combine the weights (Volume and Amount) by summing them
    df['combined_weight'] = df['decayed_volume'] + df['decayed_amount']
    
    # Multiply returns by combined weights
    df['weighted_returns'] = df['close_to_open_return'] * df['combined_weight']
    
    # Aggregate weighted returns over past N days
    df['agg_weighted_returns'] = df['weighted_returns'].rolling(window=n_days).sum()
    df['agg_weights'] = df['combined_weight'].rolling(window=n_days).sum()
    
    # Normalize the aggregated weighted returns by the sum of weights
    alpha_factor = df['agg_weighted_returns'] / df['agg_weights']
    
    return alpha_factor
```
