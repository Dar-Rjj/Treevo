```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change_percent = df['close'].pct_change()
    
    # Calculate the ratio of close price to open price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the volume weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_range + close_change_percent + close_open_ratio + volume_weighted_close) / 4.0
    
    return alpha_factor
```
