import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1) - 1)
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['volume'] * df['daily_return']
    
    # Combine Factors
    df['combined_factor'] = df['close_position_in_range'] + df['weighted_daily_return']
    
    # Identify High-Low Range Ratio
    df['daily_high_low_range'] = df['high'] - df['low']
    df['recent_avg_high_low_range'] = df['daily_high_low_range'].rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_close_open_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_close_open_delta'] / df['intraday_range']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Indicators
    df['combined_indicator'] = df['breakout_strength'] + df['volume_weighted_momentum'] + df['intraday_reversal_score']
    
    # Final Factor
    df['moving_average_21'] = df['close'].rolling(window=21).mean()
    df['trend_adjustment'] = df['close'] - df['moving_average_21']
    
    # Adjust for VWAP
    total_volume = df['volume'].rolling(window=21).sum()
    df['vwap'] = (df['volume'] * df['close']).rolling(window=21).sum() / total_volume
    df['final_factor'] = df['combined_factor'] + df['trend_adjustment'] - (df['close'] - df['vwap'])
    
    return df['final_factor']
