import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Simple Moving Average (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Compute Exponential Moving Average (EMA)
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Rate of Change (ROC)
    roc_5 = df['close'].pct_change(periods=5)
    roc_10 = df['close'].pct_change(periods=10)
    
    # Determine Volume Impact
    mean_volume_21 = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / mean_volume_21
    
    # Assign Weight to Indicators
    weighted_sma_5 = sma_5 * volume_ratio
    weighted_ema_5 = ema_5 * volume_ratio
    weighted_roc_5 = roc_5 * volume_ratio
    
    # Combine Adjusted Momentum Indicators
    combined_momentum = weighted_sma_5 + weighted_ema_5 + weighted_roc_5
    
    # Identify High-Low Range Ratio
    high_low_range = df['high'] - df['low']
    avg_high_low_range_21 = high_low_range.rolling(window=21).sum() / 21
    recent_avg_high_low_range = avg_high_low_range_21 / (df['volume'].rolling(window=21).sum() / 21)
    
    # Calculate Breakout Strength
    breakout_strength = high_low_range / avg_high_low_range_21
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_strength = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    return_7 = df['close'].pct_change(periods=7)
    return_21 = df['close'].pct_change(periods=21)
    price_momentum = return_7 + return_21
    
    # Calculate Volume Activity
    avg_volume_7 = df['volume'].rolling(window=7).mean()
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    volume_activity = avg_volume_7 / avg_volume_21
    
    # Final Alpha Factor
    final_alpha_factor = combined_breakout_strength * price_momentum * volume_activity
    
    return final_alpha_factor
