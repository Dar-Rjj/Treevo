import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1) - 1)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] / df['close'].shift(7) - 1) + (df['close'] / df['close'].shift(14) - 1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] / df['close'].shift(21) - 1) + (df['close'] / df['close'].shift(63) - 1)
    
    # Combine Momentum
    combined_momentum = short_term_momentum + long_term_momentum
    
    # Incorporate Momentum
    momentum_adjusted_factor = combined_breakout_intraday + combined_momentum
    
    # Calculate Volume Activity
    volume_7_day_avg = df['volume'].rolling(window=7).mean()
    volume_21_day_avg = df['volume'].rolling(window=21).mean()
    volume_activity = volume_7_day_avg / volume_21_day_avg
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    adjusted_factor = momentum_adjusted_factor * volume_ratio
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=50).mean()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor + trend_adjustment
    
    return final_alpha_factor
