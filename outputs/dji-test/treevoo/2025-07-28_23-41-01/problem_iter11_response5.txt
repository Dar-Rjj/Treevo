```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    df['rolling_daily_returns'] = df['daily_return'].rolling(window=20).sum()
    
    # Average Daily Returns
    df['average_daily_returns'] = df['rolling_daily_returns'] / 20
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['aggregate_volume_impact'] = df['volume_change'].rolling(window=20).sum() / 20
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['average_daily_returns'] * 0.7) + (df['aggregate_volume_impact'] * 0.3)
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.8) + (df['range_signal'] * 0.2)
    
    # Calculate Recent Average High-Low Range
    df['rolling_high_low_sum'] = df['high_low_range'].rolling(window=21).sum()
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['rolling_high_low_sum']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_delta'] / df['high_low_range']) * df['volume']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Integrate Volume Momentum
    df['rolling_price_change_sum'] = df['price_change'].rolling(window=10).sum()
    df['rolling_volume_sum_10'] = df['volume'].rolling(window=10).sum()
    df['volume_momentum'] = df['rolling_price_change_sum'] / df['rolling_volume_sum_10']
    
    # Combine All Indicators
    df['alpha_factor'] = (
        df['integrated_momentum'] +
        df['breakout_strength'] +
        (df['price_change'] * df['volume']) +
        df['intraday_reversal_score'] +
        df['volume_momentum']
    )
    
    return df['alpha_factor']
```
