```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Average Daily Returns for Price Momentum
    daily_returns = df['close'].diff()
    price_momentum = daily_returns.rolling(window=20).sum() / 20
    
    # Integrate Volume-Weighted Return with Raw Price Momentum
    integrated_return = (volume_weighted_return * 0.5) + (price_momentum * 0.5)
    
    # Calculate Logarithmic Volume Change
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    
    # Aggregate Volume Impact
    volume_impact = log_volume_change.rolling(window=20).sum() / 20
    
    # Combine Integrated Return with Volume Impact
    combined_return = (integrated_return * 0.6) + (volume_impact * 0.4)
    
    # Calculate Daily High-to-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate Average High-to-Low Range over N Days
    n_day_average_range = daily_range.rolling(window=20).mean()
    
    # Derive Momentum Signal
    range_signal = (daily_range > n_day_average_range).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    final_factor = (combined_return * 0.8) + (range_signal * 0.2)
    
    return final_factor
```
