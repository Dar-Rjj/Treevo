```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'].pct_change().fillna(0))
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Intraday Reversal Score
    intraday_close_to_open_delta = (df['close'] - df['open'])
    intraday_reversal_score = (intraday_close_to_open_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday_factors = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    price_momentum_7day = (df['close'] / df['close'].shift(7)) - 1
    price_momentum_21day = (df['close'] / df['close'].shift(21)) - 1
    combined_price_momentum = price_momentum_7day + price_momentum_21day
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=21).mean()
    adjusted_factor = combined_breakout_intraday_factors * volume_ratio
    
    # Incorporate Momentum
    adjusted_factor_with_momentum = adjusted_factor + combined_price_momentum
    
    # Integrate Volume Momentum
    rolling_volume = df['volume'].rolling(window=10).sum()
    rolling_price_changes = (df['close'] - df['close'].shift()).rolling(window=10).sum()
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = (df['close'] - df['open']) * df['volume']
    
    # Combine Weighted Return and Price Momentum
    combined_value = 0.4 * volume_weighted_close_to_open_return + 0.6 * (combined_price_momentum.rolling(window=20).sum() / 20)
    
    # Incorporate High-Low Range Signal
    daily_high_low_range = df['high'] - df['low']
    n_day_average_high_low_range = daily_high_low_range.rolling(window=21).mean()
    high_low_range_signal = (daily_high_low_range > n_day_average_high_low_range).astype(int)
    
    # Final Alpha Factor
    final_combined_factors = combined_breakout_intraday_factors + adjusted_factor_with_momentum + combined_value
    final_factor = 0.8 * final_combined_factors + 0.2 * high_low_range_signal
    
    return final_factor
```
