import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Weighted by Volume
    df['volume_weighted_return'] = df['volume'] * df['daily_return']
    
    # Combine Factors
    df['combined_factor'] = df['close_position_in_range'] + df['volume_weighted_return']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    total_volume = df['volume'].sum()
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Adjust for VWAP
    df['adjusted_factor'] = df['close'] - df['vwap']
    
    # Adjust for Trend
    period = 50  # Define a suitable period for the moving average
    df['moving_average_close'] = df['close'].rolling(window=period).mean()
    df['trend_adjusted_factor'] = df['close'] - df['moving_average_close']
    
    # Final Factor
    final_factor = df['adjusted_factor'] + df['trend_adjusted_factor']
    
    return final_factor
