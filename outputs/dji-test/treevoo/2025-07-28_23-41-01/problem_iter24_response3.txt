import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = df['volume'] * daily_return
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    breakout_strength = intraday_range / intraday_range.rolling(window=21).sum()
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7) + \
                          (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / df['close'].shift(21) + \
                         (df['close'] - df['close'].shift(63)) / df['close'].shift(63)
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7) + \
                     (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Calculate Volume Activity
    volume_activity_7d = df['volume'].rolling(window=7).mean()
    volume_activity_21d = df['volume'].rolling(window=21).mean()
    volume_ratio = volume_activity_7d / volume_activity_21d
    
    # Determine Volume Impact
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    momentum_adjusted_factor = adjusted_factor + short_term_momentum + long_term_momentum
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=50).mean()
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted_factor + trend_adjustment
    
    return final_alpha_factor
