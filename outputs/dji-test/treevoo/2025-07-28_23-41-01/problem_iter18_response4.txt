```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between close and open prices
    rel_diff_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative range of the day (high - low) to (high + low)
    rel_range = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate the volume change over the previous day
    vol_change = df['volume'] - df['volume'].shift(1)
    
    # Combine the above features into a single alpha factor
    # This is a heuristic: stocks with large daily ranges, high volume changes, and significant close-to-open differences may experience mean reversion
    alpha_factor = rel_range * vol_change * rel_diff_close_open
    
    return alpha_factor
```
