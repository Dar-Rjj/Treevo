```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    df['rolling_range_sum'] = df['daily_range'].rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['rolling_range_sum']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    df['alpha_factor'] = df['breakout_strength'] * df['volume_weighted_momentum'] / df['rolling_volume_sum']
    
    return df['alpha_factor']
```
