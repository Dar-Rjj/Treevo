import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    daily_returns = df['close'].diff()
    price_momentum = daily_returns.rolling(window=20).mean()
    
    # Combine Weighted Return and Price Momentum
    combined_value = 0.4 * volume_weighted_return + 0.6 * price_momentum
    
    # Incorporate High-Low Range Signal
    daily_high_low_range = df['high'] - df['low']
    n_day_avg_high_low_range = daily_high_low_range.rolling(window=21).mean()
    high_low_signal = (daily_high_low_range > n_day_avg_high_low_range).astype(int)
    
    # Integrate Combined Value and High-Low Range Signal
    integrated_factor = 0.8 * combined_value + 0.2 * high_low_signal
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_combined_factor = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = daily_high_low_range.rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(7) + df['close'].pct_change(14)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(21) + df['close'].pct_change(63)
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=21).mean()
    volume_impact = (initial_combined_factor + breakout_strength + short_term_momentum + long_term_momentum) * volume_ratio
    
    # Integrate All Factors
    all_factors = integrated_factor + volume_impact
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=50).mean()
    
    # Final Alpha Factor
    final_alpha_factor = all_factors + trend_adjustment
    
    return final_alpha_factor
