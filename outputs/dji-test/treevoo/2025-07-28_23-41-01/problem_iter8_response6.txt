import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_pos_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Volume-Weighted Average Price (VWAP)
    total_volume = df['volume'].sum()
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['daily_return'] * df['volume']
    
    # Combine Factors
    df['combined_factor'] = df['close_pos_in_range'] + df['weighted_daily_return']
    
    # Adjust for VWAP
    df['adjusted_combined_factor'] = df['combined_factor'] - (df['close'] - df['vwap'])
    
    # Calculate Momentum Indicators
    periods = [5, 10, 20]
    
    # Simple Moving Average (SMA)
    for p in periods:
        df[f'sma_{p}'] = df['close'].rolling(window=p).mean()
    
    # Exponential Moving Average (EMA)
    for p in periods:
        df[f'ema_{p}'] = df['close'].ewm(span=p, adjust=False).mean()
    
    # Rate of Change (ROC)
    for p in periods:
        df[f'roc_{p}'] = (df['close'] / df['close'].shift(p) - 1) * 100
    
    # Determine Volume Impact
    mean_volume = df['volume'].mean()
    df['volume_ratio'] = df['volume'] / mean_volume
    
    # Assign Weight to Indicators
    for p in periods:
        df[f'sma_{p}_weighted'] = df[f'sma_{p}'] * df['volume_ratio']
        df[f'ema_{p}_weighted'] = df[f'ema_{p}'] * df['volume_ratio']
        df[f'roc_{p}_weighted'] = df[f'roc_{p}'] * df['volume_ratio']
    
    # Combine Adjusted Momentum Indicators
    df['combined_momentum_indicators'] = 0
    for p in periods:
        df['combined_momentum_indicators'] += df[f'sma_{p}_weighted']
        df['combined_momentum_indicators'] += df[f'ema_{p}_weighted']
        df['combined_momentum_indicators'] += df[f'roc_{p}_weighted']
    
    # Calculate Breakout Strength
    df['current_range'] = df['high'] - df['low']
    df['avg_range'] = df['current_range'].rolling(window=20).mean()
    df['breakout_strength'] = df['current_range'] / df['avg_range']
    
    # Calculate Volume Weighted Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_momentum'] = df['breakout_strength'] + df['volume_weighted_momentum']
    
    # Incorporate Momentum
    df['momentum_adjusted_factor'] = df['adjusted_combined_factor'] + df['combined_breakout_momentum']
    
    # Final Alpha Factor
    period = 30
    df['trend'] = df['close'].rolling(window=period).mean()
    df['alpha_factor'] = df['momentum_adjusted_factor'] - (df['close'] - df['trend'])
    
    return df['alpha_factor']
