```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    rolling_sum_range = daily_range.rolling(window=21).sum()
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    recent_avg_range = rolling_sum_range / 21
    recent_avg_volume = rolling_sum_volume / 21
    
    # Calculate Breakout Strength
    breakout_strength = (daily_range / recent_avg_range).fillna(0)
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = (df['close'] - df['open']).fillna(0)
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / daily_range * df['volume']).fillna(0)
    
    # Calculate Price Change
    price_change = df['close'].diff().fillna(0)
    
    # Combine Indicators for Breakout and Intraday Reversal
    combined_breakout_reversal = (breakout_strength + 
                                  price_change * df['volume'] + 
                                  intraday_reversal_score).fillna(0)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = (price_change * df['volume']).fillna(0)
    
    # Combine Breakout and Volume Weighted Momentum
    combined_breakout_vol_mom = (breakout_strength + volume_weighted_momentum).fillna(0)
    
    # Calculate Price Momentum
    price_momentum_7d = df['close'].pct_change(7).fillna(0)
    price_momentum_21d = df['close'].pct_change(21).fillna(0)
    price_momentum = (price_momentum_7d + price_momentum_21d).fillna(0)
    
    # Calculate Volume Activity
    volume_7d_avg = df['volume'].rolling(window=7).mean().fillna(0)
    volume_21d_avg = df['volume'].rolling(window=21).mean().fillna(0)
    volume_activity = (volume_7d_avg / volume_21d_avg).fillna(0)
    
    # Calculate Momentum Indicators
    sma_7d = df['close'].rolling(window=7).mean().fillna(0)
    ema_7d = df['close'].ewm(span=7, adjust=False).mean().fillna(0)
    roc_7d = (df['close'].pct_change(7)).fillna(0)
    
    # Determine Volume Impact
    mean_volume = df['volume'].mean()
    volume_ratio = (df['volume'] / mean_volume).fillna(0)
    
    weighted_sma = sma_7d * volume_ratio
    weighted_ema = ema_7d * volume_ratio
    weighted_roc = roc_7d * volume_ratio
    
    # Combine Adjusted Momentum Indicators
    combined_momentum_indicators = (weighted_sma + weighted_ema + weighted_roc).fillna(0)
    
    # Final Alpha Factor
    final_alpha_factor = (combined_breakout_vol_mom * 
                          (price_momentum * volume_activity) + 
                          combined_momentum_indicators).fillna(0)
    
    return final_alpha_factor
```
