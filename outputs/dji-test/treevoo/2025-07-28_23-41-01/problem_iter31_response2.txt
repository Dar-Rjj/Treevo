```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10):
    # Calculate Close-to-Open Return
    daily_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    volume_amount = df['volume'] * df['amount']
    
    # Apply Exponential Decay to Volume and Amount
    decay_factor = 0.95  # Example decay factor, can be adjusted
    exp_decay = volume_amount.ewm(span=n_days, adjust=False).mean()
    
    # Aggregate Weighted Returns over Past N Days
    weighted_returns = daily_return * exp_decay
    aggregated_returns = weighted_returns.rolling(window=n_days).sum()
    
    # Sum of Weights from Last N Days
    total_weights = exp_decay.rolling(window=n_days).sum()
    
    # Normalize the aggregated returns by the sum of weights
    factor_values = aggregated_returns / total_weights
    
    return factor_values
```
