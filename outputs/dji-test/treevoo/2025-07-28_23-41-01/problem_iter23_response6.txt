```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Apply Moving Average Filter to Smooth the Factor
    short_term_window = 5
    long_term_window = 20
    df['short_term_ma'] = df['volume_weighted_return'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['volume_weighted_return'].rolling(window=long_term_window).mean()
    
    # Subtract Long Term Moving Average from Short Term
    df['momentum_signal'] = df['short_term_ma'] - df['long_term_ma']
    
    return df['momentum_signal'].dropna()
```
