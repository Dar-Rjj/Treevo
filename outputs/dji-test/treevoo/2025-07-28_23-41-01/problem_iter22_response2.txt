```python
import pandas as pd

def heuristics_v2(df):
    # High-Low Range Ratio
    daily_range = df['high'] - df['low']
    recent_avg_range = daily_range.rolling(window=21).sum()
    
    # Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Volume Weighted Momentum
    price_change = df['close'].pct_change()
    volume_weighted_momentum = (price_change * df['volume']).fillna(0)
    
    # Combined Breakout and Volume Weighted Momentum
    combined_breakout_volume_momentum = (breakout_strength + volume_weighted_momentum) / 2
    
    # Price Momentum
    seven_day_return = df['close'].pct_change(7)
    twenty_one_day_return = df['close'].pct_change(21)
    price_momentum = (seven_day_return + twenty_one_day_return) / 2
    
    # Volume Activity
    seven_day_avg_vol = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_vol = df['volume'].rolling(window=21).mean()
    volume_activity = (seven_day_avg_vol - twenty_one_day_avg_vol) / seven_day_avg_vol
    
    # Momentum Indicators
    sma_50 = df['close'].rolling(window=50).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    roc_50 = df['close'].pct_change(50)
    
    # Adjusted Momentum Indicators
    sma_vol_adjusted = sma_50 * df['volume']
    ema_vol_adjusted = ema_50 * df['volume']
    roc_vol_adjusted = roc_50 * df['volume']
    
    # Trend Indicator (ADX)
    adx = (sma_vol_adjusted + ema_vol_adjusted + roc_vol_adjusted) / 3  # Simplified ADX for this context
    
    # Market Sentiment Factor
    open_close_diff_5 = (df['open'] - df['close']).rolling(window=5).mean()
    open_close_diff_20 = (df['open'] - df['close']).rolling(window=20).mean()
    market_sentiment = (open_close_diff_5 + open_close_diff_20) / 2
    
    # Volatility Factor
    std_dev_21 = df['close'].pct_change().rolling(window=21).std()
    std_dev_63 = df['close'].pct_change().rolling(window=63).std()
    volatility_factor = (std_dev_21 + std_dev_63) / 2
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_breakout_volume_momentum
        + (price_momentum + volume_activity) / 2
        + (sma_vol_adjusted + ema_vol_adjusted + roc_vol_adjusted) / 3
        + adx
        + market_sentiment
        - volatility_factor
    )
    
    return final_alpha_factor
```
