```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_combined_factor = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Intraday Reversal Score
    intraday_close_to_open_delta = df['close'] - df['open']
    intraday_reversal_score = (intraday_close_to_open_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = initial_combined_factor + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(7)
    twenty_one_day_return = df['close'].pct_change(21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / mean_volume
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    momentum_adjusted_factor = adjusted_factor + price_momentum
    
    # Adjust for Trend
    moving_average_close = df['close'].rolling(window=21).mean()
    trend_adjustment = df['close'] - moving_average_close
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted_factor + trend_adjustment
    
    return final_alpha_factor
```
