```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    breakout_strength_1 = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=21).sum()
    breakout_strength_2 = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    breakout_strength = breakout_strength_1 + breakout_strength_2
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    price_momentum_7 = (df['close'] / df['close'].shift(7)) - 1
    price_momentum_21 = (df['close'] / df['close'].shift(21)) - 1
    price_momentum = price_momentum_7 + price_momentum_21
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    adjusted_factor_with_momentum = adjusted_factor + price_momentum
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open = (df['close'] - df['open']) * df['volume']
    
    # Combine Weighted Return and Price Momentum
    combined_value = 0.5 * volume_weighted_close_to_open + 0.5 * (df['close'].pct_change().rolling(window=20).sum() / 20)
    
    # Incorporate High-Low Range Signal
    n_day_avg_high_low = (df['high'] - df['low']).rolling(window=20).mean()
    signal = (df['high'] - df['low']).gt(n_day_avg_high_low).astype(int)
    
    # Final Alpha Factor
    combined_factors = adjusted_factor_with_momentum + combined_value
    final_factor = 0.9 * combined_factors + 0.1 * signal
    
    return final_factor
```
