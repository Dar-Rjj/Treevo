import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Volume-Weighted Intraday Reversal Score
    volume_weighted_reversal_score = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    
    # Intraday Momentum Adjusted Reversal Score
    momentum_adjusted_reversal_score = (df['close'] - df['open']) / (df['high'] - df['low']) * (df['volume'] * (df['close'] - df['close'].shift(1)))
    momentum_adjusted_reversal_score += (df['close'] - df['open']).where(df['close'] > df['open'], -(df['open'] - df['close']))
    
    # Identify High-Low Range Ratio
    daily_high_low_range = df['high'] - df['low']
    recent_avg_high_low_range = daily_high_low_range.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = daily_high_low_range / recent_avg_high_low_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_strength_momentum = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(periods=7)
    twenty_one_day_return = df['close'].pct_change(periods=21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Introduce Trend Indicator
    fifty_day_ma = df['close'].rolling(window=50).mean()
    two_hundred_day_ma = df['close'].rolling(window=200).mean()
    trend_indicator = (fifty_day_ma > two_hundred_day_ma).astype(int) * 2 - 1
    
    # Final Alpha Factor
    alpha_factor = combined_breakout_strength_momentum * price_momentum * volume_activity * trend_indicator
    
    # Incorporate Intraday Volatility
    ten_day_ema_intraday_range = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()
    twenty_day_sma_intraday_range = df['high'].rolling(window=20).mean() - df['low'].rolling(window=20).mean()
    volatility_adjustment = 1.5 if ten_day_ema_intraday_range > twenty_day_sma_intraday_range else 1.0
    alpha_factor *= volatility_adjustment
    
    # Calculate Intraday Trading Sentiment
    trading_sentiment = (df['high'] - df['close']) / (df['high'] - df['low'])
    trading_sentiment += (df['high'] - df['close']).where(df['high'] > df['close'], -(df['close'] - df['low']))
    
    # Adjust Final Alpha Factor with Sentiment
    final_alpha_factor = alpha_factor * trading_sentiment
    final_alpha_factor = final_alpha_factor.where(trading_sentiment > 0, -final_alpha_factor)
    
    return final_alpha_factor
