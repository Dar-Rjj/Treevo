```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    rolling_sum_range = daily_range.rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / (rolling_sum_range / 21)
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_signal = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(7)
    twenty_one_day_return = df['close'].pct_change(21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_vol = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_vol = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_vol / twenty_one_day_avg_vol
    
    # Final Alpha Factor
    alpha_factor = combined_signal * (price_momentum * volume_activity)
    
    return alpha_factor
```
