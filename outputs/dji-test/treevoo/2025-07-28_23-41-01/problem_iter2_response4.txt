```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    rolling_sum_range = daily_range.rolling(window=21).sum()
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Identify High-Low Range Ratio
    high_low_ratio = (daily_range / rolling_sum_range) * (df['volume'] / rolling_sum_volume)
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / rolling_sum_range.mean()
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * volume_weighted_momentum
    
    # Calculate 7-day and 21-day Returns
    return_7d = df['close'].pct_change(7)
    return_21d = df['close'].pct_change(21)
    
    # Combine Returns
    price_momentum = return_7d + return_21d
    
    # Calculate 7-day and 21-day Average Volume
    avg_volume_7d = df['volume'].rolling(window=7).mean()
    avg_volume_21d = df['volume'].rolling(window=21).mean()
    
    # Combine Volume Averages
    volume_activity = avg_volume_7d / avg_volume_21d
    
    # Final Alpha Factor
    alpha_factor = (combined_breakout_momentum * price_momentum * volume_activity).dropna()
    
    return alpha_factor
```
