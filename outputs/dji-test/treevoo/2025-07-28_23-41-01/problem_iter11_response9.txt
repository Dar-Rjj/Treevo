```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-day rolling sum)
    recent_avg_range = daily_range.rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(periods=7)
    twenty_one_day_return = df['close'].pct_change(periods=21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / daily_range) * df['volume']
    
    # Combine Indicators for Breakout and Intraday Reversal
    combined_indicator = combined_breakout_momentum + intraday_reversal_score
    
    # Calculate Momentum Indicators
    sma_50 = df['close'].rolling(window=50).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    roc_50 = df['close'].pct_change(periods=50)
    
    # Determine Volume Impact
    mean_volume = df['volume'].mean()
    volume_ratio = df['volume'] / mean_volume
    weighted_sma = sma_50 * volume_ratio
    weighted_ema = ema_50 * volume_ratio
    weighted_roc = roc_50 * volume_ratio
    
    # Combine Adjusted Momentum Indicators
    adjusted_momentum_indicators = weighted_sma + weighted_ema + weighted_roc
    
    # Final Alpha Factor
    final_alpha_factor = (combined_breakout_momentum * 
                          (price_momentum * volume_activity) * 
                          adjusted_momentum_indicators)
    
    return final_alpha_factor
```
