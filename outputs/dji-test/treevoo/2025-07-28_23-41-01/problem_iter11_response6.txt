```python
import pandas as pd

def heuristics(df, n=10, m=20, k=60):
    # Calculate Daily Close-to-Close Returns
    close_returns = df['close'].pct_change()
    
    # Sum Last N Days' Volume
    volume_sum_n = df['volume'].rolling(window=n).sum()
    
    # Calculate Volume-Adjusted Return
    volume_adjusted_return = close_returns * volume_sum_n
    
    # Apply Moving Average on Volume-Adjusted Return
    short_term_ma = volume_adjusted_return.rolling(window=m).mean()
    long_term_ma = volume_adjusted_return.rolling(window=k).mean()
    
    # Subtract Long-Term Moving Average from Short-Term Moving Average
    factor_values = short_term_ma - long_term_ma
    
    return factor_values
```
