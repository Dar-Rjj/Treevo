```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_factor = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    rolling_high_low_sum = (df['high'] - df['low']).rolling(window=21).sum()
    rolling_volume_sum = df['volume'].rolling(window=21).sum()
    breakout_strength = (df['high'] - df['low']) / rolling_high_low_sum * rolling_volume_sum
    
    # Calculate Intraday Reversal Score
    intraday_close_to_open_delta = df['close'] - df['open']
    intraday_reversal_score = (intraday_close_to_open_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = initial_factor + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(7)
    twenty_one_day_return = df['close'].pct_change(21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / mean_volume
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    adjusted_factor_with_momentum = adjusted_factor + price_momentum
    
    # Integrate Volume Momentum
    ten_day_rolling_vol = df['volume'].rolling(window=10).sum()
    ten_day_rolling_price_change = df['close'].diff().rolling(window=10).sum()
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']
    
    # Combine Weighted Return and Price Momentum
    combined_value = 0.5 * close_to_open_return + 0.5 * price_momentum.rolling(window=20).sum() / 20
    
    # Incorporate High-Low Range Signal
    n_day_average_range = (df['high'] - df['low']).rolling(window=21).mean()
    high_low_signal = (df['high'] - df['low'] > n_day_average_range).astype(int)
    
    # Final Alpha Factor
    final_factor = 0.9 * (adjusted_factor_with_momentum + combined_value) + 0.1 * high_low_signal
    
    return final_factor
```
