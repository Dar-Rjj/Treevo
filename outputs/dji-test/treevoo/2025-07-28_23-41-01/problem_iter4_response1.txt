```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Average Daily Returns over 20-day period
    df['avg_daily_returns'] = df['daily_returns'].rolling(window=20).mean()
    
    # Calculate Volume Change
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Aggregate Volume Impact over 20-day period
    df['agg_volume_impact'] = df['volume_change'].rolling(window=20).mean()
    
    # Combine Factors
    df['factor'] = (df['avg_daily_returns'] * 0.7) + (df['agg_volume_impact'] * 0.3)
    
    return df['factor']
```
