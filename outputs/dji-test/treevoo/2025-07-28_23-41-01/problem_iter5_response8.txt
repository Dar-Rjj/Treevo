import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    rolling_sum_range = daily_range.rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_range = rolling_sum_range / 21
    
    # Identify High-Low Range Ratio
    high_low_range_ratio = daily_range / recent_avg_range
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_signal = breakout_strength * volume_weighted_momentum
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Final Alpha Factor
    alpha_factor = combined_signal + volume_weighted_return
    
    return alpha_factor
