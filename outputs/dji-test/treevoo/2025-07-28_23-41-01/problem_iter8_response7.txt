```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, close) prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the range of (high, low) prices
    price_range = df['high'] - df['low']
    
    # Calculate a relative strength indicator using volume
    rs_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Generate alpha factor
    alpha_factor = (avg_price * price_range * rs_volume) / (df['close'].shift(1) + 1e-6)
    
    return alpha_factor
```
