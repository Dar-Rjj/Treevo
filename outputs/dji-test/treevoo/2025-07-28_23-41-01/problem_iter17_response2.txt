```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=30):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Measure Volume Impact
    volume_change = df['volume'] / df['volume'].shift(1)
    log_volume_change = np.log(volume_change)
    aggregate_volume_impact = log_volume_change.rolling(window=n_days).sum() / n_days
    
    # Combine Weighted Close-to-Open Return and Volume Impact
    combined_momentum = (weighted_close_to_open_return * 0.6) + (aggregate_volume_impact * 0.4)
    
    # Incorporate High-to-Low Range Momentum
    daily_high_to_low_range = df['high'] - df['low']
    average_high_to_low_range = daily_high_to_low_range.rolling(window=n_days).mean()
    high_to_low_range_signal = (daily_high_to_low_range > average_high_to_low_range).astype(int)
    
    # Integrate Combined Close-to-Open and Volume Momentum with High-to-Low Range Signal
    integrated_momentum = (combined_momentum * 0.7) + (high_to_low_range_signal * 0.3)
    
    # Calculate Breakout Strength
    recent_avg_high_to_low_range = daily_high_to_low_range.rolling(window=n_days).mean()
    breakout_strength = (daily_high_to_low_range / recent_avg_high_to_low_range) - 1
    
    # Combine All Indicators
    final_factor = integrated_momentum + breakout_strength
    
    return final_factor
```
