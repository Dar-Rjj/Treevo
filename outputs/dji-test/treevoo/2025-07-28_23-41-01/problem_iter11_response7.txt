```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_pos_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['daily_return'] * df['volume']
    
    # Combine Factors
    df['combined_factor'] = df['close_pos_in_range'] + df['weighted_daily_return']
    
    # Identify High-Low Range Ratio
    df['hl_range_rolling_sum'] = df['intraday_range'].rolling(window=21).sum()
    df['volume_rolling_sum'] = df['volume'].rolling(window=21).sum()
    df['hl_range_ratio'] = df['hl_range_rolling_sum'] / df['volume_rolling_sum']
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['intraday_range'] / df['hl_range_rolling_sum']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['reversal_score'] = (df['intraday_delta'] / df['intraday_range']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    df['price_change'] = df['close'].diff()
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Indicators
    df['combined_indicators'] = df['breakout_strength'] + df['volume_weighted_momentum'] + df['reversal_score']
    
    # Adjust for Trend
    df['trend_adjustment'] = df['close'] - df['close'].rolling(window=20).mean()
    
    # Adjust for VWAP
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['vwap_adjustment'] = df['close'] - df['vwap']
    
    # Calculate Momentum Indicators
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['roc_14'] = df['close'].pct_change(periods=14)
    
    # Determine Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=21).mean()
    df['sma_50_vol_adj'] = df['sma_50'] * df['volume_ratio']
    df['sma_200_vol_adj'] = df['sma_200'] * df['volume_ratio']
    df['ema_50_vol_adj'] = df['ema_50'] * df['volume_ratio']
    df['roc_14_vol_adj'] = df['roc_14'] * df['volume_ratio']
    
    # Combine Adjusted Momentum Indicators
    df['combined_momentum_indicators'] = df['sma_50_vol_adj'] + df['sma_200_vol_adj'] + df['ema_50_vol_adj'] + df['roc_14_vol_adj']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vol_mom'] = df['breakout_strength'] + df['volume_weighted_momentum']
    
    # Incorporate Momentum
    df['momentum_adjusted_factor'] = df['combined_factor'] + df['combined_breakout_vol_mom']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['momentum_adjusted_factor'] + df['trend_adjustment']

    return df['final_alpha_factor']
```
