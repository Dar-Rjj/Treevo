```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine with Volume and Amount
    volume_amount_weighted = (df['volume'] * df['amount']) / (df['volume'] + df['amount'])
    
    # Final Alpha Factor
    alpha_factor = close_to_open_return * volume_amount_weighted
    
    return alpha_factor
```
