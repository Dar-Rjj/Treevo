```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    daily_returns = (df['close'] - df['open']) / df['open']
    
    # Apply Volume Weighting to Daily Returns
    volume_weighted_returns = daily_returns * df['volume']
    
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    rolling_high_low_sum = high_low_range.rolling(window=21).sum()
    rolling_volume_sum = df['volume'].rolling(window=21).sum()
    recent_avg_high_low_range = rolling_high_low_sum / rolling_volume_sum
    
    # Calculate Breakout Strength
    breakout_strength = high_low_range / recent_avg_high_low_range
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / high_low_range) * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Indicators
    factor = (
        breakout_strength 
        + (price_change * df['volume']) 
        + intraday_reversal_score
    )
    
    return factor
```
