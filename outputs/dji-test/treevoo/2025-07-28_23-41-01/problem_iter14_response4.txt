def heuristics_v2(data):
    # Calculate Daily High-Low Range
    data['daily_high_low_range'] = data['high'] - data['low']
    
    # Calculate Recent Average High-Low Range
    data['recent_avg_high_low_range'] = (data['daily_high_low_range'].rolling(window=21).sum() 
                                         / data['volume'].rolling(window=21).sum())
    
    # Calculate Breakout Strength
    data['breakout_strength'] = data['daily_high_low_range'] / data['recent_avg_high_low_range']
    
    # Calculate Intraday Close-to-Open Delta
    data['intraday_close_open_delta'] = data['close'] - data['open']
    
    # Formulate Intraday Reversal Score
    data['intraday_reversal_score'] = (data['intraday_close_open_delta'] / data['daily_high_low_range']) * data['volume']
    
    # Calculate Price Change
    data['price_change'] = data['close'] - data['close'].shift(1)
    
    # Integrate Volume Momentum
    data['integrated_vol_momentum'] = (data['volume'].rolling(window=10).sum() 
                                       * data['price_change'].rolling(window=10).sum())
    
    # Calculate Relative Price Movement
    data['relative_price_movement'] = (data['high'] - data['low']) / (data['high'] + data['low'])
    
    # Enhance Volume Momentum with Relative Price Movement
    data['enhanced_vol_momentum'] = data['integrated_vol_momentum'] * data['relative_price_movement']
    
    # Calculate Volume Weighted Momentum
    data['vol_weighted_momentum'] = data['price_change'] * data['volume']
    
    # Combine Indicators
    data['combined_breakout_vol_weighted_momentum'] = data['breakout_strength'] * data['vol_weighted_momentum']
    
    # Calculate Price Momentum
    data['price_momentum'] = ((data['close'] / data['close'].shift(7) - 1) 
                              + (data['close'] / data['close'].shift(21) - 1))
    
    # Calculate Volume Activity
    data['volume_activity'] = (data['volume'].rolling(window=7).mean() 
                               / data['volume'].rolling(window=21).mean())
    
    # Final Alpha Factor
    data['alpha_factor'] = (data['combined_breakout_vol_weighted_momentum'] 
                            * data['price_momentum'] 
                            * data['volume_activity'])
    
    return data['alpha_factor']
