import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Compute Return Using Close and Open Prices
    df['return'] = df['close'] / df['open'] - 1
    
    # Apply Volume and Amount Weights with Exponential Decay
    decay_factor = 0.95
    df['weight'] = df['volume'] * df['amount']
    for i in range(1, N+1):
        df.loc[df.index[i:], 'weight'] += decay_factor ** (i-1) * df['volume'].shift(i) * df['amount'].shift(i)
    
    # Aggregate Weighted Returns over Past N Days
    df['weighted_return'] = df['return'] * df['weight']
    df['alpha'] = df['weighted_return'].rolling(window=N).sum()
    
    return df['alpha']
