import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factor = close_position + weighted_daily_return
    
    # Calculate Breakout Strength
    avg_21_day_range = (df['high'] - df['low']).rolling(window=21).mean()
    breakout_strength = (df['high'] - df['low']) / avg_21_day_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factor + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] / df['close'].shift(7) - 1) + (df['close'] / df['close'].shift(14) - 1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] / df['close'].shift(21) - 1) + (df['close'] / df['close'].shift(63) - 1)
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['close'].shift(7) - 1) + (df['close'] / df['close'].shift(21) - 1)
    
    # Calculate Volume Activity
    avg_7_day_volume = df['volume'].rolling(window=7).mean()
    avg_21_day_volume = df['volume'].rolling(window=21).mean()
    volume_activity = avg_7_day_volume / avg_21_day_volume
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    momentum_adjusted_factor = adjusted_factor + short_term_momentum + long_term_momentum
    
    # Adjust for Trend
    moving_average_close = df['close'].rolling(window=50).mean()
    trend_adjustment = df['close'] - moving_average_close
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted_factor + trend_adjustment
    
    return final_alpha_factor
