```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Raw Price Momentum
    daily_returns = df['close'] - df['close'].shift(1)
    raw_price_momentum = daily_returns.rolling(window=20).mean()
    
    # Measure Volume Impact
    volume_change = np.log(df['volume'] / df['volume'].shift(1))
    volume_impact = volume_change.rolling(window=20).mean()
    
    # Combine Close-to-Open Return, Price Momentum, and Volume Impact
    combined_factor = 0.4 * volume_weighted_return + 0.4 * raw_price_momentum + 0.2 * volume_impact
    
    # Incorporate High-to-Low Range Momentum
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=20).mean()
    range_signal = (high_low_range > avg_high_low_range).astype(int)
    
    # Integrate Combined Return and Momentum with High-to-Low Range Signal
    final_factor = 0.8 * combined_factor + 0.2 * range_signal
    
    return final_factor
```
