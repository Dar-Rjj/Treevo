```python
import pandas as pd

def heuristics_v2(df, N=20, M=10, K=30):
    # Calculate Daily Close-to-Close Returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum Last N Days' Volume
    df['sum_volume_N_days'] = df['volume'].rolling(window=N).sum()
    
    # Calculate Volume-Adjusted Return
    df['volume_adjusted_return'] = df['daily_return'] * df['sum_volume_N_days']
    
    # Apply Moving Average on Volume-Adjusted Return
    df['MA_M_days'] = df['volume_adjusted_return'].rolling(window=M).mean()
    
    # Long-Term Moving Average
    df['MA_K_days'] = df['volume_adjusted_return'].rolling(window=K).mean()
    
    # Subtract Long-Term Moving Average from Short-Term Moving Average
    factor_values = df['MA_M_days'] - df['MA_K_days']
    
    return factor_values
```
