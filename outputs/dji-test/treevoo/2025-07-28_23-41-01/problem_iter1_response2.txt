```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    
    # Compute N-day Moving Average of Daily Returns
    ma_daily_returns = daily_returns.rolling(window=N).mean()
    
    # Calculate Volume Change
    volume_changes = df['volume'] - df['volume'].shift(1)
    
    # Compute N-day Moving Average of Volume Changes
    ma_volume_changes = volume_changes.rolling(window=N).mean()
    
    # Adjust Momentum by Volume Change Ratio
    average_volume = df['volume'].rolling(window=N).mean()
    adjusted_momentum = ma_volume_changes / average_volume
    
    # Final Alpha Factor
    alpha_factor = ma_daily_returns * adjusted_momentum
    
    return alpha_factor
```
