```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    rolling_sum_range = daily_range.rolling(window=21).sum()
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_range = rolling_sum_range / rolling_sum_volume
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Price Change
    price_change = df['close'].pct_change()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_vol_mom = breakout_strength * volume_weighted_momentum
    
    # Calculate 7-day and 21-day Returns
    return_7d = (df['close'] / df['close'].shift(7)) - 1
    return_21d = (df['close'] / df['close'].shift(21)) - 1
    combined_returns = return_7d + return_21d
    
    # Calculate 7-day and 21-day Average Volumes
    avg_volume_7d = df['volume'].rolling(window=7).mean()
    avg_volume_21d = df['volume'].rolling(window=21).mean()
    
    # Combine Volume Averages
    volume_activity = avg_volume_7d / avg_volume_21d
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=21).std()
    adjusted_factor = combined_breakout_vol_mom / volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_factor * combined_returns * volume_activity
    
    return alpha_factor
```
