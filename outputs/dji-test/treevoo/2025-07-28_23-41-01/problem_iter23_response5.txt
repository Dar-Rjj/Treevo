```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the close location value (CLV) which indicates where the close is relative to the daily range
    clv = (2 * df['close'] - df['high'] - df['low']) / daily_range
    
    # Calculate the volume weighted CLV
    volume_weighted_clv = clv * df['volume']
    
    # Sum of 10 day rolling window for volume weighted CLV
    factor_values = volume_weighted_clv.rolling(window=10).sum()
    
    return factor_values
```
