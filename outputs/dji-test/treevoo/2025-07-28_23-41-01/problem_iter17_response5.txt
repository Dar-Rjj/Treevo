```python
def heuristics_v2(df):
    # Calculate the difference between the high and low price, which represents the daily price range
    price_range = df['high'] - df['low']
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the factor as the ratio of the average price to the price range
    # This factor may capture the volatility of the stock relative to its average price
    factor = avg_price / price_range
    
    return factor
```
