```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Calculate Raw Price Momentum
    df['Daily_Return'] = df['close'] - df['close'].shift(1)
    df['Raw_Price_Momentum'] = df['Daily_Return'].rolling(window=20).mean()
    
    # Measure Volume Impact
    df['Volume_Change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['Volume_Impact'] = df['Volume_Change'].rolling(window=20).mean()
    
    # Combine Close-to-Open Return, Price Momentum, and Volume Impact
    df['Combined_Return_and_Impact'] = (0.4 * df['Volume_Weighted_Return'] +
                                        0.4 * df['Raw_Price_Momentum'] +
                                        0.2 * df['Volume_Impact'])
    
    # Incorporate High-to-Low Range Momentum
    df['High_to_Low_Range'] = df['high'] - df['low']
    df['Average_High_to_Low_Range'] = df['High_to_Low_Range'].rolling(window=20).mean()
    df['High_to_Low_Range_Signal'] = (df['High_to_Low_Range'] > df['Average_High_to_Low_Range']).astype(int)
    
    # Integrate Combined Return and Momentum with High-to-Low Range Signal
    df['Final_Factor'] = (0.8 * df['Combined_Return_and_Impact'] + 
                          0.2 * df['High_to_Low_Range_Signal'])

    return df['Final_Factor'].dropna()
```
