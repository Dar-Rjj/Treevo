```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Daily High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average High-to-Low Range over N Days
    df['N_day_avg_range'] = df['daily_range'].rolling(window=N).mean()
    
    # Compare with Current High-to-Low Range
    df['momentum_signal'] = (df['daily_range'] > df['N_day_avg_range']).astype(int)
    
    return df['momentum_signal']
```
