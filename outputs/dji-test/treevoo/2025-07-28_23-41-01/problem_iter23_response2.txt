```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    breakout_strength = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=21).sum()
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7) + (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / df['close'].shift(21) + (df['close'] - df['close'].shift(63)) / df['close'].shift(63)
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7) + (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Calculate Volume Activity
    volume_ratio = df['volume'].rolling(window=7).mean() / df['volume'].rolling(window=21).mean()
    
    # Determine Volume Impact
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    momentum_adjusted_factor = adjusted_factor + short_term_momentum + long_term_momentum
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=21).mean()
    
    # Adjusted Momentum Indicators
    sma_volume = df['close'].rolling(window=21).mean() * df['volume']
    ema_volume = df['close'].ewm(span=21, adjust=False).mean() * df['volume']
    roc_volume = ((df['close'] - df['close'].shift(21)) / df['close'].shift(21)) * df['volume']
    
    # Trend Indicator: ADX
    adx = df['high'] - df['low']
    
    # Market Sentiment Factor
    market_sentiment_5d = (df['open'] - df['close']).rolling(window=5).mean()
    market_sentiment_20d = (df['open'] - df['close']).rolling(window=20).mean()
    
    # Volatility Adjustment
    volatility_21d = df['close'].pct_change().rolling(window=21).std()
    volatility_63d = df['close'].pct_change().rolling(window=63).std()
    
    # Final Alpha Factor
    final_alpha_factor = (trend_adjustment + momentum_adjusted_factor + sma_volume + ema_volume + roc_volume + adx + 
                          market_sentiment_5d + market_sentiment_20d + volatility_21d + volatility_63d)
    
    return final_alpha_factor
```
