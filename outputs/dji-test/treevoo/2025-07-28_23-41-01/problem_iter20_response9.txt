```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of close price to volume
    df['close_to_volume'] = df['close'] / df['volume']
    
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate a combined factor using these features
    df['alpha_factor'] = (df['close_change'] * df['close_to_volume']) / df['high_low_diff']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
