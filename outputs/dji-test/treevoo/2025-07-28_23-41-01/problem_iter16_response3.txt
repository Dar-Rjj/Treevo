```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].diff()
    df['Rolling_Daily_Returns'] = df['Daily_Return'].rolling(window=20).sum()

    # Average Daily Returns
    df['Average_Daily_Return'] = df['Rolling_Daily_Returns'] / 20

    # Measure Volume Impact
    df['Volume_Change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['Aggregate_Volume_Impact'] = df['Volume_Change'].rolling(window=20).sum() / 20

    # Combine Price Momentum and Volume Impact
    df['Combined_Momentum'] = (df['Average_Daily_Return'] * 0.7) + (df['Aggregate_Volume_Impact'] * 0.3)

    # Incorporate High-to-Low Range Momentum
    df['Daily_High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range'] = df['Daily_High_Low_Range'].rolling(window=20).mean()
    df['High_Low_Signal'] = (df['Daily_High_Low_Range'] > df['Avg_High_Low_Range']).astype(int)

    # Integrate Combined Momentum with High-to-Low Range Signal
    df['Integrated_Momentum'] = (df['Combined_Momentum'] * 0.8) + (df['High_Low_Signal'] * 0.2)

    # Calculate Breakout Strength
    df['Recent_Avg_High_Low_Range'] = df['Daily_High_Low_Range'].rolling(window=21).sum()

    # Formulate Intraday Reversal Score
    df['Intraday_Close_Open_Delta'] = df['close'] - df['open']
    df['Intraday_Reversal_Score'] = (df['Intraday_Close_Open_Delta'] / df['Daily_High_Low_Range']) * df['volume']

    # Integrate Volume Momentum
    df['Rolling_Volume_Sum'] = df['volume'].rolling(window=10).sum()
    df['Rolling_Price_Change_Sum'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()

    # Combine All Indicators
    df['Alpha_Factor'] = (
        df['Integrated_Momentum'] +
        (df['Recent_Avg_High_Low_Range'] / df['Daily_High_Low_Range']) +
        (df['Rolling_Price_Change_Sum'] * df['Rolling_Volume_Sum']) +
        df['Intraday_Reversal_Score'] +
        df['Rolling_Volume_Sum']
    )

    return df['Alpha_Factor']
```
