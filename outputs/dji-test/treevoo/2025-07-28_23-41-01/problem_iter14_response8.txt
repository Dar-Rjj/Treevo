```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    current_day_high_low_range = df['high'] - df['low']
    breakout_strength = current_day_high_low_range / recent_avg_high_low_range
    
    # Calculate Intraday Reversal Score
    intraday_close_to_open_delta = df['close'] - df['open']
    intraday_reversal_score = (intraday_close_to_open_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(periods=7)
    twenty_one_day_return = df['close'].pct_change(periods=21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=21).mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    combined_with_momentum = adjusted_factor + price_momentum
    
    # Integrate Volume Momentum
    ten_day_rolling_sum_volume = df['volume'].rolling(window=10).sum()
    ten_day_rolling_sum_price_changes = df['close'].diff().rolling(window=10).sum()
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Introduce Trend Indicator
    fifty_day_ma = df['close'].rolling(window=50).mean()
    two_hundred_day_ma = df['close'].rolling(window=200).mean()
    trend_indicator = (fifty_day_ma > two_hundred_day_ma).astype(int) * 2 - 1
    
    # Final Alpha Factor
    final_alpha_factor = (combined_with_momentum * volume_activity) * trend_indicator
    
    return final_alpha_factor
```
