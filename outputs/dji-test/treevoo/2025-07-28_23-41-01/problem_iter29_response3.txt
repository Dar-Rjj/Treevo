import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_pos_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted Daily Return
    df['weighted_daily_return'] = df['daily_return'] * df['volume']
    
    # Combine Initial Factors
    df['initial_factors'] = df['close_pos_in_range'] + df['weighted_daily_return']
    
    # Calculate Breakout Strength
    df['avg_high_low_range_21'] = df['high'] - df['low']
    df['avg_high_low_range_21'] = df['avg_high_low_range_21'].rolling(window=21).sum() / 21
    df['breakout_strength'] = (df['high'] - df['low']) / df['avg_high_low_range_21']
    
    # Calculate Intraday Reversal Score
    df['intraday_reversal'] = (df['close'] - df['open']) / df['intraday_range'] * df['volume']
    
    # Combine Breakout and Intraday Factors
    df['combined_breakout_intraday'] = df['initial_factors'] + df['breakout_strength'] + df['intraday_reversal']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['daily_return'].rolling(window=7).sum() + df['daily_return'].rolling(window=21).sum()
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / mean_volume
    df['adjusted_factor'] = df['combined_breakout_intraday'] * df['volume_ratio']
    
    # Incorporate Momentum
    df['factor_with_momentum'] = df['adjusted_factor'] + df['price_momentum']
    
    # Introduce Trend Indicator
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    df['200_day_MA'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = df.apply(lambda row: row['factor_with_momentum'] if row['50_day_MA'] > row['200_day_MA'] else -row['factor_with_momentum'], axis=1)
    
    # Incorporate High-Low Range Signal
    df['21_day_avg_range'] = (df['high'] - df['low']).rolling(window=21).mean()
    df['high_low_range_signal'] = df.apply(lambda row: 1 if row['high'] - row['low'] > row['21_day_avg_range'] else 0, axis=1)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    df['integrated_component'] = df['trend_indicator'] * 0.7 + df['high_low_range_signal'] * 0.3
    
    # Final Alpha Factor
    return df['integrated_component']
