import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Close-to-Open Return
    df['return_close_open'] = (df['close'] - df['open']) / df['open']
    df['weight_volume'] = df['volume'] * 0.95**(len(df) - df.index)
    df['weight_amount'] = df['amount'] * 0.95**(len(df) - df.index)
    df['weighted_return'] = df['return_close_open'] * (df['weight_volume'] + df['weight_amount'])

    # Aggregate Weighted Returns over Past N Days
    n_days = 21
    df['sum_weighted_returns'] = df['weighted_return'].rolling(window=n_days).sum()

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']

    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Weighted by Volume
    df['weighted_daily_return'] = df['daily_return'] * df['volume']

    # Combine Initial Factors
    df['combined_initial_factor'] = df['close_position_in_range'] + df['weighted_daily_return']

    # Calculate Breakout Strength
    df['breakout_strength'] = df['intraday_range'] / df['intraday_range'].rolling(window=21).mean()

    # Calculate Intraday Reversal Score
    df['intraday_delta'] = df['close'] - df['open']
    df['intraday_reversal_score'] = (df['intraday_delta'] / df['intraday_range']) * df['volume']

    # Combine Breakout and Intraday Factors
    df['combined_breakout_intraday_factor'] = df['combined_initial_factor'] + df['breakout_strength'] + df['intraday_reversal_score']

    # Calculate Short-Term Momentum
    df['momentum_7d'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['momentum_14d'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    df['short_term_momentum'] = df['momentum_7d'] + df['momentum_14d']

    # Calculate Long-Term Momentum
    df['momentum_21d'] = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    df['momentum_63d'] = (df['close'] - df['close'].shift(63)) / df['close'].shift(63)
    df['long_term_momentum'] = df['momentum_21d'] + df['momentum_63d']

    # Calculate Price Momentum
    df['price_momentum'] = df['momentum_7d'] + df['momentum_21d']

    # Incorporate Momentum
    df['adjusted_factor'] = df['combined_breakout_intraday_factor'] + df['short_term_momentum'] + df['long_term_momentum']

    # Integrate Volume Momentum
    df['volume_10d_sum'] = df['volume'].rolling(window=10).sum()
    df['price_change_10d_sum'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()

    # Calculate Volume Activity
    df['volume_7d_avg'] = df['volume'].rolling(window=7).mean()
    df['volume_21d_avg'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['volume_7d_avg'] / df['volume_21d_avg']

    # Introduce Trend Indicator
    df['ma_50d'] = df['close'].rolling(window=50).mean()
    df['ma_200d'] = df['close'].rolling(window=200).mean()
    df['trend_indicator'] = df['ma_50d'] > df['ma_200d']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_factor'] * df['volume_activity']
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['trend_indicator'].apply(lambda x: 1 if x else -1)

    return
