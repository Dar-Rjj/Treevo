import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position + weighted_daily_return
    
    # Calculate Breakout Strength
    current_range = df['high'] - df['low']
    recent_average_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    breakout_strength = current_range / recent_average_range
    
    # Calculate Intraday Reversal Score
    intraday_delta = df['close'] - df['open']
    intraday_reversal_score = (intraday_delta / intraday_range) * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    price_momentum_7d = (df['close'] / df['close'].shift(7)) - 1
    price_momentum_21d = (df['close'] / df['close'].shift(21)) - 1
    combined_price_momentum = price_momentum_7d + price_momentum_21d
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    momentum_incorporated = adjusted_factor + combined_price_momentum
    
    # Integrate Volume Momentum
    rolling_sum_volume = df['volume'].rolling(window=10).sum()
    rolling_sum_price_changes = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()
    
    # Calculate Volume Activity
    avg_volume_7d = df['volume'].rolling(window=7).mean()
    avg_volume_21d = df['volume'].rolling(window=21).mean()
    volume_activity = avg_volume_7d / avg_volume_21d
    
    # Introduce Trend Indicator
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    trend_indicator = (ma_50 > ma_200).astype(int) * 2 - 1
    
    # Final Alpha Factor
    final_alpha_factor = momentum_incorporated * volume_activity * trend_indicator
    
    return final_alpha_factor
