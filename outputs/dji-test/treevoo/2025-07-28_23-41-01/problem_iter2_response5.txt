```python
import pandas as pd

def heuristics_v2(data, n_days=20, sma_period=10, ema_period=10, roc_period=10):
    # Calculate Daily High-to-Low Range
    data['range'] = data['high'] - data['low']
    
    # Calculate Average High-to-Low Range over N Days
    data['avg_range'] = data['range'].rolling(window=n_days).mean()
    
    # Compare with Current High-to-Low Range
    data['momentum_signal'] = (data['range'] > data['avg_range']).astype(int)
    
    # Adjust Preliminary Momentum Signal by Volume
    data['volume_ratio'] = data['volume'] / data['volume'].rolling(window=n_days).mean()
    data['adjusted_momentum_signal'] = data['momentum_signal'] * data['volume_ratio']
    
    # Calculate Momentum Indicators
    data['SMA'] = data['close'].rolling(window=sma_period).mean()
    data['EMA'] = data['close'].ewm(span=ema_period, adjust=False).mean()
    data['ROC'] = data['close'].pct_change(periods=roc_period)
    
    # Determine Volume Impact on Momentum Indicators
    data['weighted_SMA'] = data['SMA'] * data['volume_ratio']
    data['weighted_EMA'] = data['EMA'] * data['volume_ratio']
    data['weighted_ROC'] = data['ROC'] * data['volume_ratio']
    
    # Combine Adjusted Indicators and Momentum Signal
    data['alpha_factor'] = (data['weighted_SMA'] + 
                            data['weighted_EMA'] + 
                            data['weighted_ROC'] + 
                            data['adjusted_momentum_signal']) / 4
    
    return data['alpha_factor'].dropna()
```
