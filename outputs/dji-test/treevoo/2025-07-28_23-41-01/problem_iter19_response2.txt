```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']

    # Calculate Price Momentum
    daily_returns = df['close'].diff()
    price_momentum = daily_returns.rolling(window=20).mean()

    # Combine Weighted Return and Price Momentum
    combined_return_momentum = 0.4 * volume_weighted_return + 0.6 * price_momentum

    # Calculate Daily High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate N-Day Average High-to-Low Range
    n_day_avg_range = high_low_range.rolling(window=20).mean()

    # Derive High-Low Range Signal
    high_low_range_signal = (high_low_range > n_day_avg_range).astype(int)

    # Integrate Combined Return and Momentum with High-Low Range Signal
    final_factor = 0.8 * combined_return_momentum + 0.2 * high_low_range_signal

    return final_factor
```
