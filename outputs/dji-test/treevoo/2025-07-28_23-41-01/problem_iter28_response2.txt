```python
import pandas as pd
import numpy as np

def heuristics(df, N=20):
    # Compute Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Apply Weights
    decay_factor_volume = 0.95
    decay_factor_amount = 0.90
    df['weighted_return_volume'] = df['close_to_open_return'] * (df['volume'] * (decay_factor_volume ** np.arange(len(df))))
    df['weighted_return_amount'] = df['close_to_open_return'] * (df['amount'] * (decay_factor_amount ** np.arange(len(df))))
    
    # Aggregate Weighted Returns over Past N Days
    df['aggregate_weighted_return'] = (df['weighted_return_volume'] + df['weighted_return_amount']).rolling(window=N).sum()
    
    # Return the final factor values
    return df['aggregate_weighted_return']
```
