```python
def heuristics_v2(df):
    # Calculate Intraday Range Factor
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Close-to-Open Delta
    close_to_open_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    reversal_score = (close_to_open_delta / intraday_range) * df['volume']
    
    return reversal_score
```
