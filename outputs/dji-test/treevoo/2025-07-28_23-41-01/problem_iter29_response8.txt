```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted Daily Return
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = intraday_range.rolling(window=21).sum()
    breakout_strength = (intraday_range / recent_avg_high_low_range)
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = (df['close'] - df['open']) / intraday_range * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Calculate Price Momentum
    price_momentum_7d = (df['close'] / df['close'].shift(7) - 1)
    price_momentum_21d = (df['close'] / df['close'].shift(21) - 1)
    price_momentum = price_momentum_7d + price_momentum_21d
    
    # Determine Volume Impact
    volume_ratio = df['volume'].rolling(window=7).mean() / df['volume'].rolling(window=21).mean()
    adjusted_factor = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    integrated_factor = adjusted_factor + price_momentum
    
    # Introduce Trend Indicator
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    trend_indicator = (ma_50 > ma_200).astype(int) * 2 - 1
    final_alpha_factor = integrated_factor * trend_indicator
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume for Close-to-Open Return
    weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Combine Weighted Close-to-Open Return and Breakout-VW Momentum
    combined_returns = weighted_close_to_open_return * 0.4 + combined_breakout_intraday * 0.6
    
    # Final Alpha Factor
    final_alpha_factor = (combined_returns * 0.7 + price_momentum * 0.5 + volume_ratio * 0.8) * trend_indicator
    
    # Incorporate High-Low Range Signal
    high_low_signal = (intraday_range > recent_avg_high_low_range).astype(int)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    final_alpha_factor = (final_alpha_factor * 0.7 + high_low_signal * 0.3)
    
    return final_alpha_factor
```
