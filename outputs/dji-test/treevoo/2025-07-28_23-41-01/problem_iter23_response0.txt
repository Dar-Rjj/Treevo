```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range Ratio
    df['daily_range'] = df['high'] - df['low']
    df['avg_21_day_range'] = df['daily_range'].rolling(window=21).sum()
    
    # Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['avg_21_day_range']
    
    # Volume Weighted Momentum
    df['price_change'] = df['close'].pct_change()
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combined Breakout and Volume Weighted Momentum
    df['combined_breakout_vol_mom'] = df['breakout_strength'] + df['volume_weighted_momentum']
    
    # Price Momentum
    df['7d_return'] = df['close'].pct_change(7)
    df['21d_return'] = df['close'].pct_change(21)
    df['price_momentum'] = (df['7d_return'] + df['21d_return']) / 2
    
    # Volume Activity
    df['7d_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21d_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = (df['7d_avg_volume'] + df['21d_avg_volume']) / 2
    
    # Simple Moving Average (SMA) and Exponential Moving Average (EMA)
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Rate of Change (ROC)
    df['roc_14'] = df['close'].pct_change(14)
    
    # Adjusted Momentum Indicators
    df['sma_50_vol'] = df['sma_50'] * df['volume']
    df['ema_50_vol'] = df['ema_50'] * df['volume']
    df['roc_14_vol'] = df['roc_14'] * df['volume']
    
    # Trend Indicator: ADX
    df['adx'] = (df['high'] - df['low']).abs().rolling(window=14).mean()  # Simplified ADX for demonstration
    
    # Market Sentiment Factor
    df['5d_open_close_diff'] = df['open'] - df['close']
    df['5d_avg_open_close_diff'] = df['5d_open_close_diff'].rolling(window=5).mean()
    df['20d_open_close_diff'] = df['open'] - df['close']
    df['20d_avg_open_close_diff'] = df['20d_open_close_diff'].rolling(window=20).mean()
    df['market_sentiment'] = (df['5d_avg_open_close_diff'] + df['20d_avg_open_close_diff']) / 2
    
    # Volatility Factor
    df['returns'] = df['close'].pct_change()
    df['21d_std_dev'] = df['returns'].rolling(window=21).std()
    df['63d_std_dev'] = df['returns'].rolling(window=63).std()
    df['volatility_adjustment'] = (df['21d_std_dev'] + df['63d_std_dev']) / 2
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['combined_breakout_vol_mom'] +
        df['price_momentum'] * df['volume_activity'] +
        df['sma_50_vol'] + df['ema_50_vol'] + df['roc_14_vol'] +
        df['adx'] +
        df['market_sentiment'] +
        df['volatility_adjustment']
    )
    
    return df['final_alpha_factor']
```
