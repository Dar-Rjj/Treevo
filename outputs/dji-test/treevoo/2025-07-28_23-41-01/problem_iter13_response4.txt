```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate Price Momentum
    df['daily_returns'] = df['close'] - df['close'].shift(1)
    df['price_momentum'] = df['daily_returns'].rolling(window=20).mean()
    
    # Combine Weighted Return and Price Momentum
    df['combined_signal'] = 0.4 * df['weighted_return'] + 0.6 * df['price_momentum']
    
    # Incorporate High-to-Low Range Signal
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    df['final_signal'] = 0.8 * df['combined_signal'] + 0.2 * df['range_signal']
    
    return df['final_signal']
```
