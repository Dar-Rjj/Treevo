import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    weighted_daily_return = df['volume'] * daily_return
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    high_low_range_rolling_sum = intraday_range.rolling(window=21).sum()
    breakout_strength = intraday_range / high_low_range_rolling_sum
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = df['volume'] * (df['close'] - df['open'])
    
    # Integrate Volume-Weighted Return and Breakout Strength
    integrated_volume_weighted_return_and_breakout_strength = (volume_weighted_close_to_open_return * 0.4) + (breakout_strength * 0.6)
    
    # Calculate Price Momentum
    price_momentum = daily_return.rolling(window=20).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_volume_weighted_return_and_breakout_strength * 0.8) + (price_momentum * 0.2)
    
    # Derive High-Low Range Signal
    n_day_average_range = intraday_range.rolling(window=21).mean()
    high_low_range_signal = (intraday_range > n_day_average_range).astype(int)
    
    # Add High-Low Range Signal to Final Alpha Factor
    final_alpha_factor += high_low_range_signal
    
    return final_alpha_factor
