import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'].pct_change()).fillna(0)
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    breakout_strength = (df['high'] - df['low']) / (df['high'].rolling(window=21).sum() - df['low'].rolling(window=21).sum())
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'].pct_change(7)).fillna(0) + (df['close'].pct_change(14)).fillna(0)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'].pct_change(21)).fillna(0) + (df['close'].pct_change(63)).fillna(0)
    
    # Calculate Price Momentum
    price_momentum = (df['close'].pct_change(7)).fillna(0) + (df['close'].pct_change(21)).fillna(0)

    volume_7_day_avg = df['volume'].rolling(window=7).mean()
    volume_21_day_avg = df['volume'].rolling(window=21).mean()
    volume_activity = volume_7_day_avg / volume_21_day_avg
    
    # Determine Volume Impact
    volume_impact = combined_breakout_intraday * volume_activity

    adjusted_factor = volume_impact + short_term_momentum + long_term_momentum

    roc = df['close'].pct_change(periods=20).fillna(0) * df['volume']

    sentiment_5_day = (df['open'] - df['close']).rolling(window=5).mean()
    sentiment_20_day = (df['open'] - df['close']).rolling(window=20).mean()

    volatility_63_day = df['close'].pct_change().rolling(window=63).std()
