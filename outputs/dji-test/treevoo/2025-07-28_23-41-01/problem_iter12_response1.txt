```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    recent_avg_range = (daily_high_low_range.rolling(window=21).sum() / 
                        df['volume'].rolling(window=21).sum())
    
    # Calculate Breakout Strength
    breakout_strength = daily_high_low_range / recent_avg_range
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / daily_high_low_range) * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Indicators for Breakout and Intraday Reversal
    combined_breakout_reversal = (breakout_strength + 
                                  price_change * df['volume'] + 
                                  intraday_reversal_score)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Volume Weighted Momentum
    combined_breakout_vol_mom = breakout_strength + volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(periods=7)
    twenty_one_day_return = df['close'].pct_change(periods=21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_avg_vol = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_vol = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_vol / twenty_one_day_avg_vol
    
    # Calculate Momentum Indicators
    sma_5 = df['close'].rolling(window=5).mean()
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    roc_5 = df['close'].pct_change(periods=5)
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    weighted_sma = sma_5 * volume_ratio
    weighted_ema = ema_5 * volume_ratio
    weighted_roc = roc_5 * volume_ratio
    
    # Combine Adjusted Momentum Indicators
    combined_adjusted_mom = weighted_sma + weighted_ema + weighted_roc
    
    # Final Alpha Factor
    final_alpha_factor = (combined_breakout_reversal + 
                          combined_breakout_vol_mom + 
                          combined_adjusted_mom) * (price_momentum * volume_activity)
    
    return final_alpha_factor
```
