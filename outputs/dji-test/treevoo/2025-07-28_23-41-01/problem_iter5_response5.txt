import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify High-Low Range Ratio
    daily_high_low_range = df['high'] - df['low']
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate Breakout Strength
    breakout_strength = (df['high'] - df['low']) / recent_avg_high_low_range
    
    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1) - 1)
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = price_change * df['volume']
    
    # Weighted by Volume
    weighted_daily_return = df['volume'] * daily_return
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_volume = breakout_strength * volume_weighted_momentum
    
    # Combine Factors
    combined_factor = close_position_in_range + weighted_daily_return
    
    # Calculate Price Momentum
    seven_day_return = (df['close'] / df['close'].shift(7) - 1)
    twenty_one_day_return = (df['close'] / df['close'].shift(21) - 1)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Adjust for VWAP
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    adjusted_combined_factor = combined_factor - (df['close'] - vwap)
    
    # Calculate Volume Activity
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    twenty_one_day_avg_volume = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_avg_volume / twenty_one_day_avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = combined_breakout_volume * price_momentum * volume_activity
    
    return final_alpha_factor
