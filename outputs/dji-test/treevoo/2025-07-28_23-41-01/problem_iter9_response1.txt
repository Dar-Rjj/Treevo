```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    rolling_sum_range = daily_range.rolling(window=21).sum()
    rolling_sum_volume = df['volume'].rolling(window=21).sum()
    recent_avg_range = rolling_sum_range / 21
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / daily_range) * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Integrate Volume Momentum
    rolling_sum_10d_volume = df['volume'].rolling(window=10).sum()
    rolling_sum_10d_price_change = (df['close'] - df['close'].shift(1)).rolling(window=10).sum()
    volume_momentum = rolling_sum_10d_price_change * rolling_sum_10d_volume
    
    # Combine Indicators
    factor = (breakout_strength + 
              (price_change * df['volume']) + 
              intraday_reversal_score + 
              volume_momentum)
    
    return factor
```
