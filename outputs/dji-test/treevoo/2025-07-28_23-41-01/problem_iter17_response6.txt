```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_pos_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Weighted by Volume
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_pos_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_range = intraday_range.rolling(window=21).sum() / 21
    breakout_strength = intraday_range / recent_avg_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].diff()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + volume_weighted_momentum
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'].pct_change(7) + df['close'].pct_change(14)) / 2
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'].pct_change(21) + df['close'].pct_change(63)) / 2
    
    # Calculate Price Momentum
    price_momentum = (df['close'].pct_change(7) + df['close'].pct_change(21))
    
    # Calculate Volume Activity
    volume_activity = df['volume'].rolling(window=7).mean() / df['volume'].rolling(window=21).mean()
    
    # Determine Volume Impact
    volume_ratio = df['volume'] / df['volume'].mean()
    volume_impact = combined_breakout_intraday * volume_ratio
    
    # Incorporate Momentum
    adjusted_factor = volume_impact + short_term_momentum + long_term_momentum
    
    # Adjust for Trend
    trend_adjustment = df['close'] - df['close'].rolling(window=50).mean()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor + trend_adjustment
    
    return final_alpha_factor
```
