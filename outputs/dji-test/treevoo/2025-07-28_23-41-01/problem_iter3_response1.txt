```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    df['avg_range_21'] = df['daily_range'].rolling(window=21).sum()
    
    # Calculate 21-Day Rolling Sum of Volume
    df['vol_21'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    df['recent_avg_range'] = df['avg_range_21'] / df['vol_21']
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_range'] / df['recent_avg_range']
    
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Volume Weighted Momentum
    df['vol_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Calculate 21-Day Moving Average of Price
    df['price_ma_21'] = df['close'].rolling(window=21).mean()
    
    # Calculate 21-Day Moving Average of Volume
    df['vol_ma_21'] = df['volume'].rolling(window=21).mean()
    
    # Enhance Momentum with Averaged Price Change
    df['enhanced_momentum'] = (df['price_change'] / df['price_ma_21']) * df['vol_ma_21']
    
    # Combine Breakout Strength with Enhanced Momentum
    df['alpha_factor'] = df['breakout_strength'] * df['enhanced_momentum']
    
    return df['alpha_factor']
```
