def heuristics_v2(df, n_days=10):
    # Calculate daily high-to-low range
    df['range'] = df['high'] - df['low']
    
    # Calculate volume-weighted average high-to-low range over N days
    def volume_weighted_average_range(df_slice):
        ranges = df_slice['range']
        volumes = df_slice['volume']
        return (ranges * volumes).sum() / volumes.sum()
    
    df['vw_avg_range'] = df['range'].rolling(window=n_days).apply(volume_weighted_average_range, raw=False)
    
    # Compare with current high-to-low range
    df['momentum_signal'] = (df['range'] > df['vw_avg_range']).astype(int)
    
    return df['momentum_signal']
