```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].diff()
    
    # Measure Volume Impact
    df['Log_Volume_Change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['Aggregate_Volume_Impact'] = df['Log_Volume_Change'].rolling(window=30).sum() / 30
    
    # Combine Price Momentum and Volume Impact
    df['Combined_Momentum'] = df['Daily_Return'] * 0.6 + df['Aggregate_Volume_Impact'] * 0.4
    
    # Incorporate High-to-Low Range Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range'] = df['High_Low_Range'].rolling(window=30).mean()
    df['Range_Signal'] = (df['High_Low_Range'] > df['Avg_High_Low_Range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['Integrated_Momentum'] = df['Combined_Momentum'] * 0.7 + df['Range_Signal'] * 0.3
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = df['High_Low_Range'] / df['High_Low_Range'].rolling(window=30).mean()
    
    # Formulate Intraday Reversal Score
    df['Intraday_Reversal_Score'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['Volume_Weighted_Reversal'] = df['Intraday_Reversal_Score'] * df['volume']
    
    # Enhance Volume Momentum with Relative Price Movement
    df['Rolling_Sum_Volume'] = df['volume'].rolling(window=10).sum()
    df['Rolling_Sum_Price_Change'] = df['close'].diff().rolling(window=10).sum()
    df['Enhanced_Volume_Momentum'] = (df['Rolling_Sum_Price_Change'] / df['Rolling_Sum_Volume']) * (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Breakout_Strength'] + df['Volume_Weighted_Reversal'] + df['Enhanced_Volume_Momentum']) * df['Integrated_Momentum']
    
    return df['Alpha_Factor']
```
