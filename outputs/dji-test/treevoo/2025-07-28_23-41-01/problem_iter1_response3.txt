```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-Close Direction
    open_close_direction = (df['open'] > df['close']).astype(int)
    
    # Integrate Volume for Price Movement
    volume_weighted_spread = df['volume'] * high_low_spread
    
    # Adjust by Open-Close Direction
    final_factor = volume_weighted_spread.copy()
    final_factor[open_close_direction == 1] *= -1
    
    return final_factor
```
