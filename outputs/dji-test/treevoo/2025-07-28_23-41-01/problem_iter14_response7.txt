```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range
    df['avg_high_low_range_21'] = df['high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['avg_high_low_range_21']
    
    # Calculate Volume Weighted Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vol_mom'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = df['close'].pct_change(periods=7)
    df['21_day_return'] = df['close'].pct_change(periods=21)
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_volume'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_volume'] / df['21_day_avg_volume']
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    df['rolling_20_day_returns'] = df['daily_returns'].rolling(window=20).sum()
    df['average_daily_returns'] = df['rolling_20_day_returns'] / 20
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    df['aggregate_volume_impact'] = df['volume_change'].rolling(window=20).sum() / 20
    
    # Combine Price Momentum and Volume Impact
    df['combined_price_vol_impact'] = (df['average_daily_returns'] * 0.7) + (df['aggregate_volume_impact'] * 0.3)
    
    # Incorporate High-to-Low Range Momentum
    df['avg_high_low_range_N'] = df['high_low_range'].rolling(window=21).mean()
    df['range_momentum_signal'] = (df['high_low_range'] > df['avg_high_low_range_N']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_price_vol_impact'] * 0.8) + (df['range_momentum_signal'] * 0.2)
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_delta'] / df['high_low_range']) * df['volume']
    
    # Integrate Volume Momentum
    df['10_day_sum_volume'] = df['volume'].rolling(window=10).sum()
    df['10_day_sum_price_change'] = df['price_change'].rolling(window=10).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        (df['combined_breakout_vol_mom'] * 0.5) +
        (df['price_momentum'] * df['volume_activity']) +
        (df['integrated_momentum'] * 0.3) +
        (df['intraday_reversal_score'] * 0.1) +
        (df['10_day_sum_volume'] * 0.1)
    )
    
    return df['alpha_factor']
```
