```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Simple Moving Average (SMA)
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute Exponential Moving Average (EMA)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Assign Weight to Indicators
    sma_10_weighted = sma_10 * volume_ratio
    sma_20_weighted = sma_20 * volume_ratio
    ema_10_weighted = ema_10 * volume_ratio
    ema_20_weighted = ema_20 * volume_ratio
    roc_weighted = roc * volume_ratio
    
    # Compute Price Movement
    price_movement = df['high'] - df['low']
    avg_price_movement = price_movement.rolling(window=10).mean()
    
    # Assign Price Movement Weight
    price_movement_weight = np.where(price_movement > avg_price_movement, 1.2, 0.8)
    
    # Apply Price Movement Weight to indicators
    sma_10_final = sma_10_weighted * price_movement_weight
    sma_20_final = sma_20_weighted * price_movement_weight
    ema_10_final = ema_10_weighted * price_movement_weight
    ema_20_final = ema_20_weighted * price_movement_weight
    roc_final = roc_weighted * price_movement_weight
    
    # Combine Adjusted Indicators
    factor_values = (sma_10_final + sma_20_final + ema_10_final + ema_20_final + roc_final) / 5

    return factor_values
```
