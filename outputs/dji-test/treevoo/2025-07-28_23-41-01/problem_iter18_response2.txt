import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Average Daily Returns over 22 days
    df['avg_daily_returns'] = df['daily_returns'].rolling(window=22).mean()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['aggregate_volume_impact'] = df['volume_change'].rolling(window=22).mean()
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (0.6 * df['avg_daily_returns']) + (0.4 * df['aggregate_volume_impact'])
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=22).mean()
    df['range_signal'] = np.where(df['high_low_range'] > df['avg_high_low_range'], 1.5, -0.5)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (0.7 * df['combined_momentum']) + (0.3 * df['range_signal'])

    df['recent_avg_high_low_range'] = df['high_low_range'].rolling(window=22).sum()

    # Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['recent_avg_high_low_range']
    
    # Volume Weighted Momentum
