```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Simple Moving Averages (SMA) for different periods
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Compute Exponential Moving Averages (EMA) for different periods
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Rate of Change (ROC) over a period
    roc_5 = df['close'].pct_change(periods=5)
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate Volume Ratio
    mean_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / mean_volume
    
    # Assign weight to indicators based on volume ratio
    sma_5_weighted = sma_5 * volume_ratio
    sma_10_weighted = sma_10 * volume_ratio
    ema_5_weighted = ema_5 * volume_ratio
    ema_10_weighted = ema_10 * volume_ratio
    roc_5_weighted = roc_5 * volume_ratio
    roc_10_weighted = roc_10 * volume_ratio
    
    # Combine adjusted indicators
    factor_value = (sma_5_weighted + sma_10_weighted + 
                    ema_5_weighted + ema_10_weighted + 
                    roc_5_weighted + roc_10_weighted)
    
    return factor_value
```
