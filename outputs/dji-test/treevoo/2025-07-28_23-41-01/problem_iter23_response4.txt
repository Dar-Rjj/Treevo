```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: A Series with the index as date and the values as the calculated alpha factor.
    """
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the change in volume
    volume_change = df['volume'].diff()
    
    # Calculate the price change
    price_change = df['close'].diff()
    
    # Calculate the weighted average price change
    weighted_avg_price_change = (price_change * volume_change).rolling(window=5).mean()
    
    # Calculate the factor value
    factor_value = (weighted_avg_price_change / avg_price) * 100
    
    return factor_value
```
