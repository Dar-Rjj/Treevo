```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted Daily Return
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    initial_factors = close_position + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (df['high'] - df['low']).rolling(window=21).sum()
    recent_avg_volume = df['volume'].rolling(window=21).mean()
    breakout_strength = (df['high'] - df['low']) / (recent_avg_high_low_range / recent_avg_volume)
    
    # Calculate Intraday Reversal Score
    intraday_reversal = (df['close'] - df['open']) / intraday_range
    volume_weighted_intraday_reversal = intraday_reversal * df['volume']
    
    # Combine Breakout and Intraday Factors
    breakout_intraday_factors = initial_factors + breakout_strength + volume_weighted_intraday_reversal
    
    # Calculate Price Momentum
    seven_day_return = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    twenty_one_day_return = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / mean_volume
    adjusted_factor = breakout_intraday_factors * volume_ratio + price_momentum
    
    # Introduce Trend Indicator
    fifty_day_ma = df['close'].rolling(window=50).mean()
    two_hundred_day_ma = df['close'].rolling(window=200).mean()
    trend_indicator = (fifty_day_ma > two_hundred_day_ma).astype(int) * 2 - 1
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor * trend_indicator
    
    return final_alpha_factor
```
