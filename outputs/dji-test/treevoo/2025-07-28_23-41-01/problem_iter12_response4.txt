```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Average Daily Returns over a rolling 30-day period
    df['avg_daily_returns'] = df['daily_returns'].rolling(window=30).mean()
    
    # Measure Volume Impact
    df['volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['agg_volume_impact'] = df['volume_change'].rolling(window=30).mean()
    
    # Combine Price Momentum and Volume Impact
    df['combined_momentum'] = (df['avg_daily_returns'] * 0.6) + (df['agg_volume_impact'] * 0.4)
    
    # Incorporate High-to-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=30).mean()
    df['range_signal'] = (df['high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Momentum with High-to-Low Range Signal
    df['integrated_momentum'] = (df['combined_momentum'] * 0.7) + (df['range_signal'] * 0.3)
    
    # Calculate Recent Average High-Low Range
    df['recent_avg_high_low_range'] = df['high_low_range'].rolling(window=25).sum()
    df['recent_avg_volume'] = df['volume'].rolling(window=25).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Intraday Close-to-Open Delta
    df['intraday_close_open_delta'] = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    df['intraday_reversal_score'] = (df['intraday_close_open_delta'] / df['high_low_range']) * df['volume']
    
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Combine All Indicators
    df['alpha_factor'] = (
        df['integrated_momentum'] +
        df['breakout_strength'] * df['price_change'] * df['volume'] +
        df['intraday_reversal_score']
    )
    
    return df['alpha_factor']
```
