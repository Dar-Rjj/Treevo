```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    df['rolling_high_low_sum'] = df['high_low_range'].rolling(window=21).sum()
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    df['avg_high_low_range'] = df['rolling_high_low_sum'] / df['rolling_volume_sum']
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_signal'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute 21-day Moving Average of Daily Returns
    df['avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute 21-day Moving Average of Volume Changes
    df['avg_volume_change'] = df['volume_change'].rolling(window=21).mean()
    
    # Adjust Momentum by Volume Change Ratio
    df['avg_volume'] = df['volume'].rolling(window=21).mean()
    df['momentum_adjustment'] = df['avg_volume_change'] / df['avg_volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_signal'] * df['momentum_adjustment']
    
    return df['alpha_factor']
```
