```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close Position in Range
    close_position_in_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted Daily Return
    weighted_daily_return = daily_return * df['volume']
    
    # Combine Initial Factors
    combined_initial_factors = close_position_in_range + weighted_daily_return
    
    # Calculate Breakout Strength
    recent_avg_high_low_range = (intraday_range.rolling(window=21).sum() / 
                                 df['volume'].rolling(window=21).sum())
    breakout_strength = (intraday_range / recent_avg_high_low_range)
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = ((df['close'] - df['open']) / intraday_range) * df['volume']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Weight by Volume
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Combine Breakout and Intraday Factors
    combined_breakout_intraday = combined_initial_factors + breakout_strength + intraday_reversal_score
    
    # Combine All Factors
    combined_all_factors = combined_breakout_intraday + volume_weighted_close_to_open
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7) + \
                     (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Determine Volume Impact
    mean_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / mean_volume
    adjusted_factor = combined_all_factors * volume_ratio
    
    # Incorporate Momentum
    adjusted_factor_with_momentum = adjusted_factor + price_momentum
    
    # Introduce Trend Indicator
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    trend_indicator = (ma_50 > ma_200) * 2 - 1
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor_with_momentum * trend_indicator
    
    return final_alpha_factor
```
