```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range
    df['recent_avg_high_low_range'] = df['daily_high_low_range'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['daily_high_low_range'] / df['recent_avg_high_low_range']
    
    # Calculate Volume Weighted Momentum
    df['close_change'] = df['close'].pct_change()
    df['volume_weighted_momentum'] = df['close_change'] * df['volume']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume for Close-to-Open Return
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['breakout_vw_momentum'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Combine Weighted Close-to-Open Return and Breakout-VW Momentum
    df['combined_breakout_vw_momentum'] = (0.4 * df['weighted_close_to_open_return']) + (0.6 * df['breakout_vw_momentum'])
    
    # Calculate Price Momentum
    df['7_day_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['21_day_return'] = (df['close'] / df['close'].shift(21)) - 1
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_vol_avg'] = df['volume'].rolling(window=7).mean()
    df['21_day_vol_avg'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_vol_avg'] / df['21_day_vol_avg']
    
    # Final Alpha Factor
    df['integrated_component'] = (0.8 * df['combined_breakout_vw_momentum']) + (0.5 * df['price_momentum']) + (0.7 * df['volume_activity'])
    
    # Incorporate High-Low Range Signal
    n = 21  # N days for high-low range signal
    df['avg_high_low_range'] = df['daily_high_low_range'].rolling(window=n).mean()
    df['high_low_range_signal'] = (df['daily_high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Integrate Combined Return and Momentum with High-Low Range Signal
    df['alpha_factor'] = (0.8 * df['integrated_component']) + (0.2 * df['high_low_range_signal'])
    
    return df['alpha_factor']
```
