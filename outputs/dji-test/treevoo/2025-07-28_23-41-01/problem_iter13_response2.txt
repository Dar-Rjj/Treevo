```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate Recent Average High-Low Range (21-Day Rolling Sum)
    recent_avg_range = daily_range.rolling(window=21).sum()
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Volume Weighted Momentum
    price_change = df['close'].pct_change()
    volume_weighted_momentum = price_change * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    combined_breakout_momentum = breakout_strength * volume_weighted_momentum
    
    # Calculate Price Momentum
    seven_day_return = df['close'].pct_change(7)
    twenty_one_day_return = df['close'].pct_change(21)
    price_momentum = seven_day_return + twenty_one_day_return
    
    # Calculate Volume Activity
    seven_day_vol_avg = df['volume'].rolling(window=7).mean()
    twenty_one_day_vol_avg = df['volume'].rolling(window=21).mean()
    volume_activity = seven_day_vol_avg / twenty_one_day_vol_avg
    
    # Final Alpha Factor
    final_alpha_factor = combined_breakout_momentum * price_momentum * volume_activity
    
    return final_alpha_factor
```
