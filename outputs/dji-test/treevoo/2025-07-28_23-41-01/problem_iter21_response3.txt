```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted by Volume
    df['weighted_daily_return'] = df['volume'] * df['daily_return']
    
    # Combine Initial Factors
    df['initial_combined_factor'] = df['close_position_in_range'] + df['weighted_daily_return']
    
    # Calculate Breakout Strength
    df['recent_avg_high_low_range'] = df['intraday_range'].rolling(window=21).sum() / 21
    df['breakout_strength'] = df['intraday_range'] / df['recent_avg_high_low_range']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_close_to_open_return'] = df['volume'] * df['close_to_open_return']
    
    # Integrate Volume-Weighted Return and Breakout Strength
    df['integrated_vol_weighted_breakout'] = 0.4 * df['volume_weighted_close_to_open_return'] + 0.6 * df['breakout_strength']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change().rolling(window=20).mean()
    
    # Combine Weighted Return, Momentum, and Breakout Strength
    df['combined_weighted_return_momentum_breakout'] = 0.8 * df['integrated_vol_weighted_breakout'] + 0.2 * df['price_momentum']
    
    # Incorporate High-Low Range Signal
    n_days = 20
    df['daily_high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['daily_high_low_range'].rolling(window=n_days).mean()
    df['high_low_range_signal'] = (df['daily_high_low_range'] > df['avg_high_low_range']).astype(int)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_weighted_return_momentum_breakout'] + df['high_low_range_signal']
    
    return df['final_alpha_factor']
```
