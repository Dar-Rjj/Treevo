```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    df['rolling_high_low_sum'] = df['high_low_range'].rolling(window=21).sum()
    df['rolling_volume_sum'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Breakout Strength
    df['breakout_strength'] = df['high_low_range'] / df['rolling_high_low_sum']
    
    # Calculate Volume Weighted Momentum
    df['price_change'] = df['close'].diff()
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Combine Breakout Strength with Volume Weighted Momentum
    df['combined_breakout_vol_mom'] = df['breakout_strength'] * df['volume_weighted_momentum']
    
    # Calculate Price Momentum
    df['7_day_return'] = df['close'].pct_change(7)
    df['21_day_return'] = df['close'].pct_change(21)
    df['price_momentum'] = df['7_day_return'] + df['21_day_return']
    
    # Calculate Volume Activity
    df['7_day_avg_vol'] = df['volume'].rolling(window=7).mean()
    df['21_day_avg_vol'] = df['volume'].rolling(window=21).mean()
    df['volume_activity'] = df['7_day_avg_vol'] / df['21_day_avg_vol']
    
    # Calculate Simple Moving Average (SMA)
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma'] = (df['sma_5'] + df['sma_10'] + df['sma_20']) / 3
    
    # Calculate Exponential Moving Average (EMA)
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema'] = (df['ema_5'] + df['ema_10'] + df['ema_20']) / 3
    
    # Calculate Rate of Change (ROC)
    df['roc_14'] = df['close'].pct_change(14)
    
    # Determine Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=21).mean()
    df['weighted_sma'] = df['sma'] * df['volume_ratio']
    df['weighted_ema'] = df['ema'] * df['volume_ratio']
    df['weighted_roc'] = df['roc_14'] * df['volume_ratio']
    
    # Combine Adjusted Momentum Indicators
    df['combined_momentum_indicators'] = df['weighted_sma'] + df['weighted_ema'] + df['weighted_roc']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_breakout_vol_mom'] 
                          * df['price_momentum'] * df['volume_activity'] 
                          + df['combined_momentum_indicators'])
    
    return df['alpha_factor']
```
