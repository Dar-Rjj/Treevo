def heuristics_v2(df):
    """
    Calculate the Volume-Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the factor values indexed by date.
    """
    # Calculate Daily Returns
    daily_returns = (df['close'] - df['open']) / df['open']
    
    # Apply Volume Weighting
    volume_weighted_returns = daily_returns * df['volume']
    
    return volume_weighted_returns
