```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate 21-Day Rolling Sum of High-Low Range and Volume
    rolling_range_sum = daily_range.rolling(window=21).sum()
    rolling_volume_sum = df['volume'].rolling(window=21).sum()
    
    # Calculate Recent Average High-Low Range
    recent_avg_range = rolling_range_sum / rolling_volume_sum
    
    # Calculate Breakout Strength
    breakout_strength = daily_range / recent_avg_range
    
    # Calculate Intraday Close-to-Open Delta
    intraday_delta = df['close'] - df['open']
    
    # Formulate Intraday Reversal Score
    intraday_reversal_score = (intraday_delta / daily_range) * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].pct_change()
    
    # Integrate Volume Momentum
    volume_momentum = (df['volume'].rolling(window=10).sum() * price_change.rolling(window=10).sum())
    
    # Calculate Relative Price Movement
    relative_price_movement = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Enhance Volume Momentum with Relative Price Movement
    enhanced_volume_momentum = volume_momentum * relative_price_movement
    
    # Combine Indicators
    factor = (
        breakout_strength +
        price_change * df['volume'] +
        intraday_reversal_score +
        enhanced_volume_momentum
    )
    
    return factor
```
