def heuristics_v2(df):
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    df['volume_median'] = df['volume'].rolling(window=30).median()
    q1 = df['volume'].rolling(window=30).quantile(0.25)
    q3 = df['volume'].rolling(window=30).quantile(0.75)
    iqr = q3 - q1
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_median'] / iqr)))
    df['atr'] = (df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min()) / 2
    heuristics_matrix = (df['macd'] + df['volume_rsi']) / df['atr']
    return heuristics_matrix
