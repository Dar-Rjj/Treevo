def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + 5*df['close']) / 7).cumsum() / df['volume'].cumsum()
    df['atr'] = df[['high', 'low']].diff().abs().max(axis=1).rolling(window=14).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * df['vwap']) / (df['atr'] + (df['close'] - df['ema_20']).abs())
    return heuristics_matrix
