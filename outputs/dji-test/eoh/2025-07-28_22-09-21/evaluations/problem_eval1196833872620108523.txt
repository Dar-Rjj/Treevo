def heuristics_v2(df):
    df['price_change_20'] = df['close'].pct_change(periods=20)
    df['price_change_50'] = df['close'].pct_change(periods=50)
    df['volume_median'] = df['volume'].rolling(window=40).median()
    df['volume_mean'] = df['volume'].rolling(window=40).mean()
    df['volume_ratio'] = df['volume_median'] / df['volume_mean']
    df['std_dev_close'] = df['close'].rolling(window=30).std()
    weighted_price_change = (df['price_change_20'] * 0.6 + df['price_change_50'] * 0.4).shift(-1)
    heuristics_matrix = (weighted_price_change * df['volume_ratio']) / df['std_dev_close']
    return heuristics_matrix
