def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change().ewm(span=10, adjust=False).mean() * (df['volume'] / df['volume'].rolling(window=15).mean())
    df['volume_mean'] = df['volume'].rolling(window=15).mean()
    df['volume_std'] = df['volume'].rolling(window=15).std()
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_mean'] / df['volume_std'])))
    df['true_range'] = df[['high', 'low']].diff().abs().max(axis=1).rolling(window=5).sum()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_rsi']) / df['true_range']
    return heuristics_matrix
