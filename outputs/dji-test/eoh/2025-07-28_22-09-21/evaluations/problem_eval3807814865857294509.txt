def heuristics_v2(df):
    # Calculate the return from the close price
    returns = df['close'].pct_change()
    # Calculate the 10-day moving average of volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    # Calculate the volume-to-average-volume ratio
    vol_ratio = df['volume'] / avg_volume_10d
    # Combine the returns and volume ratio into a single score
    heuristics_matrix = returns * vol_ratio
    return heuristics_matrix
