import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    highs = df['high']
    lows = df['low']
    
    sma_10_close = close_prices.rolling(window=10).mean()
    high_low_diff_20 = highs.rolling(window=20).max() - lows.rolling(window=20).min()
    
    heuristics_matrix = sma_10_close * high_low_diff_20
    
    return heuristics_matrix
