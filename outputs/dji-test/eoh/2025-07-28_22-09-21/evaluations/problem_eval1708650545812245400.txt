import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_prices = df['close']
    
    ema_20_close = close_prices.ewm(span=20, adjust=False).mean()
    std_30_close = close_prices.rolling(window=30).std()
    
    heuristics_matrix = ema_20_close / std_30_close
    
    return heuristics_matrix
