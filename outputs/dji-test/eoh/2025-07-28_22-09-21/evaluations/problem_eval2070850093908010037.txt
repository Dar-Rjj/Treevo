def heuristics_v2(df):
    df['price_change_5'] = df['close'].pct_change(5)
    df['price_change_10'] = df['close'].pct_change(10)
    df['volume_std'] = df['volume'].rolling(window=14).std()
    df['volume_ratio'] = df['volume'] / df['volume_std']
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = ((0.6 * df['price_change_5'] + 0.4 * df['price_change_10']).shift(-1) * df['volume_ratio']) / df['ema_close']
    return heuristics_matrix
