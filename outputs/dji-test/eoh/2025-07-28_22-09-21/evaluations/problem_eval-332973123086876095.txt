def heuristics_v2(df):
    df['price_to_high_change'] = (df['close'] / df['high']).pct_change(periods=20)
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volatility'] = df['close'].rolling(window=60).std()
    heuristics_matrix = (df['price_to_high_change'].shift(-1) * df['avg_volume']) / df['volatility']
    return heuristics_matrix
