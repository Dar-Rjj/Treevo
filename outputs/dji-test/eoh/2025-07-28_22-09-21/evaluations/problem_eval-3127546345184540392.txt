def heuristics_v2(df):
    df['price_ratio_change'] = (df['close'] / df['open']).pct_change(periods=10)
    df['sum_volume'] = df['volume'].rolling(window=25).sum()
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['std_true_range'] = df['true_range'].rolling(window=30).std()
    heuristics_matrix = (df['price_ratio_change'].shift(-1) + df['sum_volume']) / df['std_true_range']
    return heuristics_matrix
