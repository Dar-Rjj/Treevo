def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change(periods=15)
    df['sum_volume'] = df['volume'].rolling(window=20).sum()
    df['volatility'] = df['close'].rolling(window=40).std()
    heuristics_matrix = (df['price_change'].shift(-1) * df['sum_volume']) / df['volatility']
    return heuristics_matrix
