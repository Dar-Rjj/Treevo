def heuristics_v2(df):
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    
    sma_14_close = close_prices.rolling(window=14).mean()
    sma_30_close = close_prices.rolling(window=30).mean()
    ema_7_high = high_prices.ewm(span=7, adjust=False).mean()
    ema_7_low = low_prices.ewm(span=7, adjust=False).mean()
    
    price_volatility_ratio = ema_7_high / ema_7_low
    
    heuristics_matrix = (sma_14_close - sma_30_close) * price_volatility_ratio
    
    return heuristics_matrix
