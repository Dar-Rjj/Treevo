def heuristics_v2(df):
    df['price_momentum'] = df['close'].pct_change(30)
    df['volume_strength'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    df['bb_width'] = (df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min()) / df['close'].rolling(window=28).mean()
    heuristics_matrix = (df['price_momentum'] * df['volume_strength']) / df['bb_width']
    return heuristics_matrix
