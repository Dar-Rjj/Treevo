def heuristics_v2(df):
    df['price_momentum'] = df['close'].pct_change(periods=20)
    df['volume_strength'] = (df['volume'] - df['volume'].rolling(window=20).min()) / (df['volume'].rolling(window=20).max() - df['volume'].rolling(window=20).min())
    df['atr_14'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()).shift(1)
    heuristics_matrix = (df['price_momentum'] * df['volume_strength']) / df['atr_14']
    return heuristics_matrix
