def heuristics_v2(df):
    df['wma_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['bb_width'] = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / df['close'].rolling(window=20).mean()
    df['es_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (df['wma_close'].pct_change().shift(-1) * df['bb_width']) / df['es_volume']
    return heuristics_matrix
