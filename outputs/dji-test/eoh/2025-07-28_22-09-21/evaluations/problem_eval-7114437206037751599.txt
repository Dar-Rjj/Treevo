def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['wma_volume'] = df['volume'].rolling(window=10).mean() * df['volume']
    df['std_price_change'] = df['price_change'].rolling(window=30).std()
    heuristics_matrix = (df['price_change'].shift(-1) * df['wma_volume']) / df['std_price_change']
    return heuristics_matrix
