def heuristics_v2(df):
    df['close_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    obv = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)).cumsum()
    df['returns_std'] = df['close'].pct_change().rolling(window=20).std()
    heuristics_matrix = (df['close_ema'] - df['close']) / df['returns_std']
    return heuristics_matrix
