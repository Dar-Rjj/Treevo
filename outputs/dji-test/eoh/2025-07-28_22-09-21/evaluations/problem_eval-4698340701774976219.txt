import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day simple moving average (SMA) of the close price
    sma_20d = df['close'].rolling(window=20).mean()
    # Calculate the difference between the close price and the 20-day SMA
    diff_close_sma = df['close'] - sma_20d
    # Calculate the 10-day exponential moving average (EMA) of the volume
    ema_volume_10d = df['volume'].ewm(span=10, adjust=False).mean()
    # Generate a signal based on the product of the difference and the EMA of volume
    heuristics_matrix = diff_close_sma * ema_volume_10d
    return heuristics_matrix
