def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_median'] = df['volume'].rolling(window=30).median()
    df['volume_mean'] = df['volume'].rolling(window=30).mean()
    df['volume_ratio'] = df['volume_median'] / df['volume_mean']
    df['std_dev_close'] = df['close'].rolling(window=28).std()
    df['weighted_heuristic'] = (df['price_change'].shift(-1) * 0.7 + df['volume_ratio'] * 0.3) / df['std_dev_close']
    heuristics_matrix = df['weighted_heuristic']
    return heuristics_matrix
