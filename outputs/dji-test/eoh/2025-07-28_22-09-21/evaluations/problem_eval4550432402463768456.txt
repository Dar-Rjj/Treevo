import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_prices = df['close']
    
    ema_20_close = close_prices.ewm(span=20, adjust=False).mean()
    daily_returns = close_prices.pct_change()
    std_dev_30_returns = daily_returns.rolling(window=30).std()
    
    heuristics_matrix = ema_20_close / std_dev_30_returns
    
    return heuristics_matrix
