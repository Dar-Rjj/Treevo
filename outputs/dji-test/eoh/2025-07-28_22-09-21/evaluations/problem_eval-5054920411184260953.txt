import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    volumes = df['volume']
    
    ema_10_close = close_prices.ewm(span=10, adjust=False).mean()
    ema_30_close = close_prices.ewm(span=30, adjust=False).mean()
    ema_5_volume = volumes.ewm(span=5, adjust=False).mean()
    
    heuristics_matrix = (ema_10_close - ema_30_close) * ema_5_volume
    
    return heuristics_matrix
