def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=30, adjust=False).mean()
    df['price_ema_change'] = df['close'] / df['ema_close'] - 1
    df['volume_mom'] = df['volume'].pct_change(periods=30)
    df['atr'] = (df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min()) / 2
    heuristics_matrix = (df['price_ema_change'] * df['volume_mom']) / df['atr']
    return heuristics_matrix
