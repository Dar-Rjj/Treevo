def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_consistency'] = 1 / (df['volume'].rolling(window=14).std() + 1e-6)
    df['atr'] = (df['high'] - df['low']).rolling(window=14).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_consistency']) / df['atr']
    return heuristics_matrix
