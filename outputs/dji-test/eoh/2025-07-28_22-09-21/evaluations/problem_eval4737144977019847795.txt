def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['cumulative_volume'] = df['volume'].cumsum()
    df['atr'] = df[['high', 'low']].diff().abs().max(axis=1).rolling(window=21).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * df['cumulative_volume']) / df['atr']
    return heuristics_matrix
