def heuristics_v2(df):
    df['high_low_change'] = (df['high'] - df['low']) / df['low']
    df['volume_ma'] = df['volume'].rolling(window=50).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma']
    df['true_range'] = df['high'] - df['low']
    df['atr_ema'] = df['true_range'].ewm(span=21, adjust=False).mean()
    heuristics_matrix = (df['high_low_change'].shift(-1) * df['volume_ratio']) * df['atr_ema']
    return heuristics_matrix
