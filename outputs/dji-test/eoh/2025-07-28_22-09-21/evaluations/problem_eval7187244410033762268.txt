def heuristics_v2(df):
    df['high_21'] = df['high'].rolling(window=21).max()
    df['low_21'] = df['low'].rolling(window=21).min()
    df['signed_volume'] = df['volume'] * (df['close'] > df['open']).apply(lambda x: 1 if x else -1)
    df['cum_signed_volume'] = df['signed_volume'].rolling(window=21).sum()
    df['atr_7'] = (df['high'].rolling(window=7).mean() - df['low'].rolling(window=7).mean())
    heuristics_matrix = (df['high_21'] / df['low_21']) * (df['cum_signed_volume'] / df['atr_7'])
    return heuristics_matrix
