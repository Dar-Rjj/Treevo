def heuristics_v2(df):
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    
    sma_30_close = close_prices.rolling(window=30).mean()
    sma_60_close = close_prices.rolling(window=60).mean()
    ema_15_high = high_prices.ewm(span=15, adjust=False).mean()
    ema_15_low = low_prices.ewm(span=15, adjust=False).mean()
    
    price_volatility_ratio = ema_15_high / ema_15_low
    
    heuristics_matrix = (sma_30_close - sma_60_close) * price_volatility_ratio
    
    return heuristics_matrix
