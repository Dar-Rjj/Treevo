def heuristics_v2(df):
    close_prices = df['close']
    volumes = df['volume']
    
    sma_20_close = close_prices.rolling(window=20).mean()
    sma_50_close = close_prices.rolling(window=50).mean()
    ema_10_volume = volumes.ewm(span=10, adjust=False).mean()
    
    volume_ratio = volumes / ema_10_volume
    
    heuristics_matrix = (sma_20_close - sma_50_close) * volume_ratio
    
    return heuristics_matrix
