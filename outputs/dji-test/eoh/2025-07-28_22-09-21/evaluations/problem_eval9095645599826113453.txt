def heuristics_v2(df):
    close_prices = df['close']
    low_prices = df['low']
    
    ema_12_close = close_prices.ewm(span=12, adjust=False).mean()
    ema_26_close = close_prices.ewm(span=26, adjust=False).mean()
    ema_14_low = low_prices.rolling(window=14).min().ewm(span=14, adjust=False).mean()
    
    relative_price_strength = close_prices / ema_14_low
    
    heuristics_matrix = (ema_12_close - ema_26_close) * relative_price_strength
    
    return heuristics_matrix
