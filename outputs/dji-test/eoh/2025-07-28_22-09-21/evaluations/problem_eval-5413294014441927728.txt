def heuristics_v2(df):
    m = 10  # example lookback period for min close; can be adjusted
    n = 5   # example lookback period for avg volume; can be adjusted
    df['min_close'] = df['close'].rolling(window=m).min()
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    heuristics_matrix = (df['close'] - df['min_close']) * (df['volume'] / df['avg_volume'])
    return heuristics_matrix
