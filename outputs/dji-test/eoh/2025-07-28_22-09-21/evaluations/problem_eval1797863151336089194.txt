def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=30, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=30, adjust=False).mean()
    df['momentum'] = df['close'] / df['close'].shift(30) - 1
    df['sum_high_low_diff'] = (df['high'] - df['low']).rolling(window=28).sum()
    heuristics_matrix = (df['ema_close'] * df['momentum'] * df['ema_volume']) / df['sum_high_low_diff']
    return heuristics_matrix
