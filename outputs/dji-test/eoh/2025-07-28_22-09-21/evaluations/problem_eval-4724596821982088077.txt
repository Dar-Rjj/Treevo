def heuristics_v2(df):
    df['close_ema'] = df['close'].ewm(span=14, adjust=False).mean()
    df['price_to_ema_ratio'] = df['close'] / df['close_ema']
    df['volume_roc'] = df['volume'].pct_change(5)
    df['return_std'] = df['close'].pct_change().rolling(window=20).std()
    heuristics_matrix = (df['price_to_ema_ratio'] + df['volume_roc']) / df['return_std']
    return heuristics_matrix
