def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change(periods=10)
    df['volume_median'] = df['volume'].rolling(window=45).median()
    df['volume_mean'] = df['volume'].rolling(window=28).mean()
    df['volume_ratio'] = df['volume_median'] / df['volume_mean']
    df['std_dev_close'] = df['close'].rolling(window=20).std()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_ratio']) / df['std_dev_close']
    return heuristics_matrix
