def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ema']
    df['std_dev_close'] = df['close'].rolling(window=5).std()
    df['momentum'] = df['close'].pct_change(periods=3)
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_ratio'] * df['momentum']) / df['std_dev_close']
    return heuristics_matrix
