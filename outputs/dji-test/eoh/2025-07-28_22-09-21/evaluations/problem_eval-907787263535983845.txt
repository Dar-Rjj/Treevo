def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    df['volatility'] = df['close'].rolling(window=30).std()
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['trading_range'] = df['high'] - df['low']
    df['z_score'] = (df['trading_range'] - df['trading_range'].rolling(window=50).mean()) / df['trading_range'].rolling(window=50).std()
    heuristics_matrix = (df['ema_close'].shift(-1) * df['volume_ratio']) / (df['volatility'] * df['z_score'])
    return heuristics_matrix
