def heuristics_v2(df):
    close_prices = df['close']
    
    ema_10_close = close_prices.ewm(span=10, adjust=False).mean()
    ema_30_close = close_prices.ewm(span=30, adjust=False).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    heuristics_matrix = (ema_10_close - ema_30_close) * avg_volume_20
    
    return heuristics_matrix
