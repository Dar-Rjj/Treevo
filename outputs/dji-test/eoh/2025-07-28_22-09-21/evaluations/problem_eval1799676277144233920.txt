def heuristics_v2(df):
    df['price_momentum'] = df['close'].pct_change(5)
    df['volume_ma'] = df['volume'].rolling(window=30).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma']
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_ema'] = df['true_range'].ewm(span=14, adjust=False).mean()
    heuristics_matrix = (df['price_momentum'].shift(-1) * df['volume_ratio']) / df['atr_ema']
    return heuristics_matrix
