def heuristics_v2(df):
    df['price_volatility'] = df['close'].rolling(window=14).std()
    df['volume_roc_smoothed'] = df['volume'].pct_change(periods=5).rolling(window=10).mean()
    df['atr'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()).rolling(window=14).mean()
    heuristics_matrix = (df['price_volatility'].shift(-1) * df['volume_roc_smoothed']) / df['atr']
    return heuristics_matrix
