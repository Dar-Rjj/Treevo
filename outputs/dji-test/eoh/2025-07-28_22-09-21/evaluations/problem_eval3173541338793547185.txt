def heuristics_v2(df):
    df['price_momentum'] = df['close'].pct_change(30)
    df['volume_median'] = df['volume'].rolling(window=30).median()
    q1 = df['volume'].rolling(window=30).quantile(0.25)
    q3 = df['volume'].rolling(window=30).quantile(0.75)
    iqr = q3 - q1
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_median'] / iqr)))
    df['atr'] = df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min()
    df['ema_14_close'] = df['close'].ewm(span=14, adjust=False).mean()
    heuristics_matrix = ((df['price_momentum'] * df['volume_rsi']) / df['atr']) * df['ema_14_close']
    return heuristics_matrix
