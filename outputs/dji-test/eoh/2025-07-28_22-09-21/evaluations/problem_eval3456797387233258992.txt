def heuristics_v2(df):
    df['avg_close'] = df['close'].rolling(window=5).mean()
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    df['range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    heuristics_matrix = (df['avg_close'].shift(-1) * df['volume_ema']) / df['range']
    return heuristics_matrix
