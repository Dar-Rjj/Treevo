def heuristics_v2(df):
    close_prices = df['close']
    volumes = df['volume']
    
    ema_20_close = close_prices.ewm(span=20, adjust=False).mean()
    ema_50_close = close_prices.ewm(span=50, adjust=False).mean()
    sma_10_volume = volumes.rolling(window=10).mean()
    
    volume_ratio = volumes / sma_10_volume
    
    heuristics_matrix = (ema_20_close - ema_50_close) * volume_ratio
    
    return heuristics_matrix
