def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['volume_ma_60'] = df['volume'].rolling(window=60).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma_60']
    df['returns'] = df['close'].pct_change()
    df['std_returns'] = df['returns'].rolling(window=30).std()
    heuristics_matrix = (df['ema_close'] * df['volume_ratio']) / df['std_returns']
    return heuristics_matrix
