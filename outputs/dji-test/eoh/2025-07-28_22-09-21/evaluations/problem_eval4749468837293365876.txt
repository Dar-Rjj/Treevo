import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_prices = df['close']
    highs = df['high']
    lows = df['low']
    
    ema_30_close = close_prices.ewm(span=30, adjust=False).mean()
    high_low_diff_20 = highs.rolling(window=20).max() - lows.rolling(window=20).min()
    
    heuristics_matrix = ema_30_close / high_low_diff_20
    
    return heuristics_matrix
