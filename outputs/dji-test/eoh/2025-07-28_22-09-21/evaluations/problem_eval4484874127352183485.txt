def heuristics_v2(df):
    close_prices = df['close']
    volumes = df['volume']
    
    sma_30_close = close_prices.rolling(window=30).mean()
    sma_90_close = close_prices.rolling(window=90).mean()
    ema_20_volume = volumes.ewm(span=20, adjust=False).mean()
    
    volume_ratio = volumes / ema_20_volume
    
    heuristics_matrix = (sma_30_close - sma_90_close) * volume_ratio
    
    return heuristics_matrix
