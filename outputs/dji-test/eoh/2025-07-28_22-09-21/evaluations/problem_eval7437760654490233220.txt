def heuristics_v2(df):
    df['momentum'] = df['close'].pct_change(14)
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume'].rolling(window=14).mean() / df['volume'].rolling(window=14).std())))
    df['atr'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()) / 2
    heuristics_matrix = (df['momentum'].shift(-1) * df['volume_rsi']) / df['atr']
    return heuristics_matrix
