def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=30, adjust=False).mean()
    df['price_momentum'] = df['close'] / df['ema_close']
    df['volume_median'] = df['volume'].rolling(window=30).median()
    q1 = df['volume'].rolling(window=30).quantile(0.25)
    q3 = df['volume'].rolling(window=30).quantile(0.75)
    iqr = q3 - q1
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_median'] / iqr)))
    df['atr'] = df[['high', 'low']].apply(lambda x: max(x[0], x[1]), axis=1).rolling(window=28).mean() - df[['high', 'low']].apply(lambda x: min(x[0], x[1]), axis=1).rolling(window=28).mean()
    heuristics_matrix = (df['price_momentum'] * df['volume_rsi']) / df['atr']
    return heuristics_matrix
