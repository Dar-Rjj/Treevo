def heuristics_v2(df):
    close_prices = df['close']
    
    # Calculate 20-day RSI
    delta = close_prices.diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 60-day and 90-day moving averages
    ma_60_close = close_prices.rolling(window=60).mean()
    ma_90_close = close_prices.rolling(window=90).mean()
    
    # Combine RSI with the ratio of 60-day MA to 90-day MA
    heuristics_matrix = rsi * (ma_60_close / ma_90_close)
    
    return heuristics_matrix
