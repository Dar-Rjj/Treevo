def heuristics_v2(df):
    # Calculate the difference between high and low as a measure of daily price range
    price_range = df['high'] - df['low']
    # Adjust the price range by the ratio of close price to volume
    adjusted_score = price_range * (df['close'] / df['volume'])
    # Generate the final heuristics matrix
    heuristics_matrix = adjusted_score.pct_change()
    return heuristics_matrix
