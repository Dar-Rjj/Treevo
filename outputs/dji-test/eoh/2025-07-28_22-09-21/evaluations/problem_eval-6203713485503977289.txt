def heuristics_v2(df):
    close_prices = df['close']
    
    ma_10_close = close_prices.rolling(window=10).mean()
    ma_30_close = close_prices.rolling(window=30).mean()
    daily_return = close_prices.pct_change()
    ma_5_daily_return = daily_return.rolling(window=5).mean()
    
    heuristics_matrix = (ma_10_close / ma_30_close) * ma_5_daily_return
    
    return heuristics_matrix
