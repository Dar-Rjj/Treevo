def heuristics_v2(df):
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    
    sma_20_close = close_prices.rolling(window=20).mean()
    sma_50_close = close_prices.rolling(window=50).mean()
    ema_10_high = high_prices.ewm(span=10, adjust=False).mean()
    ema_10_low = low_prices.ewm(span=10, adjust=False).mean()
    
    price_volatility_ratio = ema_10_high / ema_10_low
    
    heuristics_matrix = (sma_20_close - sma_50_close) * price_volatility_ratio
    
    return heuristics_matrix
