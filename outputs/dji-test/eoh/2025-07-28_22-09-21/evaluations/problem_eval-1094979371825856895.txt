def heuristics_v2(df):
    df['momentum'] = df['close'].pct_change(periods=10)
    df['volume_10ma'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume_10ma'] / df['volume']
    df['ema_high'] = df['high'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (df['momentum'].shift(-1) * df['volume_ratio']) / df['ema_high']
    return heuristics_matrix
