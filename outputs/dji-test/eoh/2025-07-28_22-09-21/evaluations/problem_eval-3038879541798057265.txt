import pandas as pd

def heuristics_v2(df):
    high_prices = df['high']
    low_prices = df['low']
    close_prices = df['close']
    volumes = df['volume']

    # Calculate 10-day True Range (as a proxy for volatility)
    tr = high_prices - low_prices
    tr_10 = tr.rolling(window=10).mean()

    # Calculate Money Flow Index (MFI)
    typical_price = (high_prices + low_prices + close_prices) / 3
    raw_money_flow = typical_price * volumes
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    pos_mf_14 = positive_money_flow.rolling(window=14).sum()
    neg_mf_14 = negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (pos_mf_14 / neg_mf_14)))

    # Define Heuristic Factor
    heuristics_matrix = tr_10 * mfi
    
    return heuristics_matrix
