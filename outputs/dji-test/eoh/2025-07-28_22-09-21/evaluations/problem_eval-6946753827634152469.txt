def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['vtp'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['ema_true_range'] = df[['high', 'low']].diff().abs().max(axis=1).ewm(span=14, adjust=False).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * df['vtp']) / df['ema_true_range']
    return heuristics_matrix
