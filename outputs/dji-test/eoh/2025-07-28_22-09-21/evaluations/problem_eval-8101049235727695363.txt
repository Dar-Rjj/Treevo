def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + 3*df['close']) / 5).cumsum() / df['volume'].cumsum()
    df['atr'] = df[['high', 'low']].diff().abs().max(axis=1).rolling(window=30).mean()
    avg_price = (df['high'] + df['low']) / 2
    heuristics_matrix = (df['price_change'].shift(-1) * df['vwap']) / (df['atr'] * avg_price)
    return heuristics_matrix
