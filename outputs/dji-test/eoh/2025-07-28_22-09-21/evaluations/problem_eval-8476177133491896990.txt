import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    volumes = df['volume']

    # Calculate Rate of Change (ROC)
    roc = close_prices.pct_change(periods=10)

    # Calculate Accumulation/Distribution (A/D) Line
    mfm = ((close_prices - low_prices) - (high_prices - close_prices)) / (high_prices - low_prices)
    ad_line = (mfm * volumes).cumsum()

    # Define Heuristic Factor
    heuristics_matrix = roc * ad_line
    
    return heuristics_matrix
