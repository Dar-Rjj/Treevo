def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    df['price_change'] = df['close'].pct_change()
    df['volume_rsi'] = 100 - (100 / (1 + ((df['volume'] * df['price_change']).rolling(window=14).mean() / (df['volume'] * df['price_change']).rolling(window=14).std())))
    df['atr'] = df[['high', 'low', 'close']].max(axis=1).rolling(window=14).max() - df[['high', 'low', 'close']].min(axis=1).rolling(window=14).min()
    heuristics_matrix = (df['ema_close'].shift(-1) * df['volume_rsi']) / df['atr']
    return heuristics_matrix
