def heuristics_v2(df):
    df['price_to_volume'] = df['close'] / df['volume']
    df['price_to_volume_wavg'] = df['price_to_volume'].ewm(span=42).mean()
    df['close_ema'] = df['close'].ewm(span=21).mean()
    heuristics_matrix = df['price_to_volume_wavg'] - df['close_ema']
    return heuristics_matrix
