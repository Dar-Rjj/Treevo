def heuristics_v2(df):
    df['momentum'] = df['close'].pct_change(periods=14)
    df['vol_adjusted_volume'] = df['volume'] * (df['high'] - df['low']) / df['close']
    df['ema_return'] = df['close'].pct_change().ewm(span=14, adjust=False).mean()
    heuristics_matrix = (df['momentum'] * df['vol_adjusted_volume']) / df['ema_return']
    return heuristics_matrix
