def heuristics_v2(df):
    close_prices = df['close']
    volume = df['volume']
    
    momentum_close = close_prices.pct_change(periods=20)
    cv_30_volume = volume.rolling(window=30).std() / volume.rolling(window=30).mean()
    
    heuristics_matrix = momentum_close / cv_30_volume
    
    return heuristics_matrix
