def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ema']
    df['true_range'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    heuristics_matrix = (df['ema_close'] * df['volume_ratio']) / df['true_range']
    return heuristics_matrix
