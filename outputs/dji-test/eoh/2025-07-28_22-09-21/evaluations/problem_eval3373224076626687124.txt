def heuristics_v2(df):
    df['price_ratio_change'] = (df['close'] / df['open']).pct_change(periods=20)
    df['sum_volume'] = df['volume'].rolling(window=15).sum()
    df['avg_true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1).rolling(window=40).mean()
    heuristics_matrix = (df['price_ratio_change'].shift(-1) * df['sum_volume']) / df['avg_true_range']
    return heuristics_matrix
