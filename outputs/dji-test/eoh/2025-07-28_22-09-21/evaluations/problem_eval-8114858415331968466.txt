def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['true_range'] = df['high'] - df['low']
    df['atr_std'] = df['true_range'].rolling(window=14).std()
    df['wma_price_change'] = df['price_change'].ewm(span=14, adjust=False).mean()
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    heuristics_matrix = (df['wma_price_change'] * df['atr_std']) / df['rsi']
    return heuristics_matrix
