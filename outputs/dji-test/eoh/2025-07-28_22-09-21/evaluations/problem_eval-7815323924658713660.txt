def heuristics_v2(df):
    df['ema_short'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_long'] = df['close'].ewm(span=30, adjust=False).mean()
    df['momentum_indicator'] = df['ema_short'] / df['ema_long']
    df['daily_returns'] = df['close'].pct_change()
    df['volatility_factor'] = df['daily_returns'].rolling(window=20).std()
    heuristics_matrix = df['momentum_indicator'] / df['volatility_factor']
    return heuristics_matrix
