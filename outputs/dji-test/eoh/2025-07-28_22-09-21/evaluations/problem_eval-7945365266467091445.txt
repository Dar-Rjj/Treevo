def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['pv_ratio'] = df['close'] / df['volume']
    df['pv_rolling_mean'] = df['pv_ratio'].rolling(window=30).mean()
    df['atr'] = df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min()
    heuristics_matrix = (df['price_change'].shift(-1) * df['pv_rolling_mean']) / df['atr']
    return heuristics_matrix
