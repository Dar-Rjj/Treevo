def heuristics_v2(df):
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    
    sma_20_close = close_prices.rolling(window=20).mean()
    sma_50_close = close_prices.rolling(window=50).mean()
    daily_range = high_prices - low_prices
    ema_10_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    
    range_ratio = daily_range / ema_10_daily_range
    
    heuristics_matrix = (sma_20_close - sma_50_close) * range_ratio
    
    return heuristics_matrix
