import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_prices = df['close']
    highs = df['high']
    lows = df['low']
    
    weights = np.arange(1, 21)
    wma_20_close = close_prices.rolling(window=20).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)
    high_low_sum_30 = highs.rolling(window=30).max() + lows.rolling(window=30).min()
    
    heuristics_matrix = wma_20_close / high_low_sum_30
    
    return heuristics_matrix
