def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_median'] = df['volume'].rolling(window=14).median().shift(1)
    df['volume_strength'] = (df['volume'] - df['volume_median']) / df['volume_median']
    df['atr'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()).shift(1)
    heuristics_matrix = (df['price_change'] * df['volume_strength']) / df['atr']
    return heuristics_matrix
