def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_5ma'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume_5ma'] / df['volume']
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_ratio']) / df['ema_close']
    return heuristics_matrix
