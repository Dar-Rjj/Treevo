def heuristics_v2(df):
    df['close_momentum'] = df['close'].pct_change(periods=14)
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_std'] = df['volume'].rolling(window=20).std()
    df['volume_upper_band'] = df['volume_ma'] + 2 * df['volume_std']
    df['volume_lower_band'] = df['volume_ma'] - 2 * df['volume_std']
    heuristics_matrix = (df['close_momentum'] * (df['volume_upper_band'] - df['volume_lower_band'])) / df['volume_std']
    return heuristics_matrix
