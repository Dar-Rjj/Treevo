def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    df['range'] = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    heuristics_matrix = (df['price_change'].shift(-1) * df['vwap']) / df['range']
    return heuristics_matrix
