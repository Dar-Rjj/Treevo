import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    
    # Calculate 10-day and 30-day cumulative returns
    cum_ret_10 = (close_prices / close_prices.shift(10)) - 1
    cum_ret_30 = (close_prices / close_prices.shift(30)) - 1
    
    # Calculate the ratio of 10-day to 30-day cumulative returns
    ret_ratio = cum_ret_10 / cum_ret_30
    
    # Calculate 5-day average trading volume
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    
    # Composite heuristics factor
    heuristics_matrix = ret_ratio * avg_vol_5
    
    return heuristics_matrix
