import pandas as pd

def heuristics_v2(df):
    df['volume_change'] = df['volume'].pct_change()
    df['price_change'] = df['close'].pct_change()
    df['factor'] = (df['volume_change'] * df['price_change']).rolling(window=5).mean()
    heuristics_matrix = df['factor'].dropna()
    return heuristics_matrix
