def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_median'] = df['volume'].rolling(window=60).median()
    q1 = df['volume'].rolling(window=60).quantile(0.25)
    q3 = df['volume'].rolling(window=60).quantile(0.75)
    iqr = q3 - q1
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_median'] / iqr)))
    df['atr'] = df['high'].rolling(window=56).max() - df['low'].rolling(window=56).min()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_rsi']) / df['atr']
    return heuristics_matrix
