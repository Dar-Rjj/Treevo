def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_median'] = df['volume'].rolling(window=14).median()
    q9 = df['volume'].rolling(window=14).quantile(0.9)
    q10 = df['volume'].rolling(window=14).quantile(0.1)
    idr = q9 - q10
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume_median'] / idr)))
    df['atr'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()) / 2
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_rsi']) / df['atr']
    return heuristics_matrix
