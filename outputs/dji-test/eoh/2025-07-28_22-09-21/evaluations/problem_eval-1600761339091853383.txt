def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=14).mean()
    df['std_high_low'] = df[['high', 'low']].rolling(window=14).std().mean(axis=1)
    heuristics_matrix = (df['ema_close'] * df['volume_ratio']) / df['std_high_low']
    return heuristics_matrix
