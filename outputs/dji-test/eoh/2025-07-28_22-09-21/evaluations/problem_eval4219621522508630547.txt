def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change(periods=10)
    df['volume_ema'] = df['volume'].ewm(span=30, adjust=False).mean()
    df['returns'] = df['close'].pct_change()
    df['sharpe_ratio'] = df['returns'].rolling(window=90).mean() / df['returns'].rolling(window=90).std()
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_ema']) / df['sharpe_ratio']
    return heuristics_matrix
