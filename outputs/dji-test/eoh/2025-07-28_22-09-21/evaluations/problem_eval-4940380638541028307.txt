def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_median'] = df['volume'].rolling(window=20).median()
    df['volume_mean'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume_median'] / df['volume_mean']
    df['avg_close'] = df['close'].rolling(window=14).mean()
    heuristics_matrix = (df['price_change'].shift(-1) * 0.6 + df['volume_ratio'] * 0.4) / df['avg_close']
    return heuristics_matrix
