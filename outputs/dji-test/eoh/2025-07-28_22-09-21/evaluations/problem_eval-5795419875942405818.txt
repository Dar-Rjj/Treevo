def heuristics_v2(df):
    df['price_weighted_avg'] = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    df['price_change'] = df['price_weighted_avg'].pct_change()
    q1 = df['volume'].rolling(window=14).quantile(0.25)
    q3 = df['volume'].rolling(window=14).quantile(0.75)
    iqr = q3 - q1
    df['volume_rsi'] = 100 - (100 / (1 + (df['volume'].rolling(window=14).median() / iqr)))
    df['atr'] = (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()) / 2
    heuristics_matrix = (df['price_change'].shift(-1) * df['volume_rsi']) / df['atr']
    return heuristics_matrix
