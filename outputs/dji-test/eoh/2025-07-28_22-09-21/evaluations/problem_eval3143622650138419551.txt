def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['gain'] = df['close'].diff().clip(lower=0)
    df['loss'] = -df['close'].diff().clip(upper=0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    df['volume_change'] = df['volume'].pct_change()
    df['weight'] = (df['rsi'] - 50) / 50
    df['weighted_close'] = df['close'] * df['weight']
    df['cumulative_range'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    heuristics_matrix = (df['weighted_close'] * df['volume_change']) / df['cumulative_range']
    return heuristics_matrix
