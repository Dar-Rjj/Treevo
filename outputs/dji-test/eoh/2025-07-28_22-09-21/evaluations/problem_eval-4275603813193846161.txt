import pandas as pd

def heuristics_v2(df):
    # Define a dynamic window size for the closing price WMA, influenced by daily volume
    dynamic_window = (df['volume'].rolling(window=10).mean() / df['volume']).apply(lambda x: 5 if x > 1 else 10)
    
    # Calculate the Weighted Moving Average (WMA) of the close price using the dynamic window
    wma_close = df['close'].rolling(window=dynamic_window.min()).apply(lambda x: (x * list(range(1, len(x)+1))).sum() / sum(list(range(1, len(x)+1))), raw=True)
    
    # Calculate the Simple Moving Average (SMA) of the open price over a fixed 10 day window
    sma_open = df['open'].rolling(window=10).mean()
    
    # Compute the alpha factor as the difference between SMA of open and WMA of close
    heuristics_matrix = sma_open - wma_close
    
    return heuristics_matrix
