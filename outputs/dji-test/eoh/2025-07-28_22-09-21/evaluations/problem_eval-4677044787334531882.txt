def heuristics_v2(df):
    df['high_low_change'] = (df['high'].pct_change(periods=21) - df['low'].pct_change(periods=21))
    df['close_rsi'] = 100 - (100 / (1 + (df['close'].rolling(window=14).mean() / df['close'].rolling(window=14).std())))
    df['volume_sma'] = df['volume'].rolling(window=50).mean()
    heuristics_matrix = (df['high_low_change'] * df['close_rsi']) / df['volume_sma']
    return heuristics_matrix
