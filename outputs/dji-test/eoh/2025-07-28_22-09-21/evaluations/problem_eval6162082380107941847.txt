import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of high to low prices
    high_low_ratio = df['high'] / df['low']
    # Calculate the 5-day moving average of the high-low ratio
    high_low_ratio_5d = high_low_ratio.rolling(window=5).mean()
    # Calculate the change in volume
    volume_change = df['volume'].diff()
    # Generate the heuristics matrix by combining the high-low ratio and volume change
    heuristics_matrix = high_low_ratio_5d * volume_change
    return heuristics_matrix
