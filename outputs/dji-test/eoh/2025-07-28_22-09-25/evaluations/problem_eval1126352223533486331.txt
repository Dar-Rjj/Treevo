import pandas as pd

def heuristics_v2(df):
    # Calculate EMA of the close price
    ema_close = df.close.ewm(span=20, adjust=False).mean()
    
    # Calculate the daily volume change
    volume_change = df.volume.pct_change()
    
    # Calculate the standard deviation of volume over the last 45 days
    std_volume = df.volume.rolling(window=45).std()
    
    # Volume change to volume standard deviation ratio
    vol_ratio = volume_change / std_volume
    
    # Heuristic combination: Weighted sum of EMA and volume ratio
    heuristics_matrix = 0.7 * ema_close + 0.3 * vol_ratio
    return heuristics_matrix
