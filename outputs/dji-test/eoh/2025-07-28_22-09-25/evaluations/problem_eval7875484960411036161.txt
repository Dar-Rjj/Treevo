def heuristics_v2(df):
    df['SMA_15'] = df['close'].rolling(window=15).mean()
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    vwap_10 = (df['close']*df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=90).mean()
    
    heuristics_matrix = (df['SMA_15'] * 0.4 + df['EMA_30'] * 0.3 + vwap_10 * 0.3) * df['Volume_Ratio']
    return heuristics_matrix
