def heuristics_v2(df):
    df['Momentum_30'] = df['close'] - df['close'].shift(30)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=21).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=21).mean()
    avg_gain_loss = (gain + loss) / 2
    df['RSI_21_Modified'] = 100 - (100 / (1 + avg_gain_loss))
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    sma_close_50 = df['close'].rolling(window=50).mean()
    heuristics_matrix = (df['Momentum_30'] * 0.4 + df['RSI_21_Modified'] * 0.4 + ema_volume_10 * 0.2) * (df['close'] / sma_close_50)
    return heuristics_matrix
