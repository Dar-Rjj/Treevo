import pandas as pd

def heuristics_v2(df):
    def rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def atr(df, window=14):
        tr = pd.DataFrame(index=df.index)
        tr['h-l'] = df['high'] - df['low']
        tr['h-pc'] = abs(df['high'] - df['close'].shift())
        tr['l-pc'] = abs(df['low'] - df['close'].shift())
        tr['tr'] = tr.max(axis=1)
        return tr['tr'].rolling(window=window).mean()

    rsi_value = rsi(df['close'])
    atr_value = atr(df)
    roc_atr = (df['close'] - atr_value) / df['close'].shift(periods=14)
    
    heuristics_matrix = 0.5 * rsi_value + 0.5 * roc_atr
    return heuristics_matrix
