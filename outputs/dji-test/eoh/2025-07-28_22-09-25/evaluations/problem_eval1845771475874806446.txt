def heuristics_v2(df):
    # Calculate the 14-day momentum (change in close price)
    momentum = df['close'].pct_change(periods=14)
    
    # Calculate the 14-day moving standard deviation (MSD) of the closing prices
    msd_close = df['close'].rolling(window=14).std()
    
    # Adjust the momentum by the moving standard deviation
    adjusted_momentum = momentum / msd_close
    
    # Calculate the 5-day cumulative volume
    cum_volume_5 = df['volume'].rolling(window=5).sum()
    
    # Calculate the 60-day moving average of the cumulative volume
    ma_cum_volume_60 = cum_volume_5.rolling(window=60).mean()
    
    # Adjust the cumulative volume by the moving average of cumulative volume
    volume_strength = cum_volume_5 / ma_cum_volume_60
    
    # Heuristic combination: Weighted sum of adjusted momentum and volume strength
    heuristics_matrix = 0.6 * adjusted_momentum + 0.4 * volume_strength
    return heuristics_matrix
