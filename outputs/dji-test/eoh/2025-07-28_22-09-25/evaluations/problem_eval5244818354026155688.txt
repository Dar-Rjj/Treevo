def heuristics_v2(df):
    highest_50_high = df['high'].rolling(window=50).max()
    lowest_50_low = df['low'].rolling(window=50).min()
    ewma_20_close = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (highest_50_high - lowest_50_low) / ewma_20_close
    return heuristics_matrix
