def heuristics_v2(df):
    # Calculate the 14-period standard deviation of daily returns
    df['returns'] = df['close'].pct_change()
    std_dev_returns = df['returns'].rolling(window=14).std()
    
    # Calculate the liquidity score
    df['Volume_MA_50'] = df['volume'].rolling(window=50).mean()
    liquidity_score = df['volume'] / df['Volume_MA_50']
    
    # Heuristic combination: Weighted sum of standard deviation of returns and liquidity score
    heuristics_matrix = 0.6 * std_dev_returns + 0.4 * liquidity_score
    return heuristics_matrix
