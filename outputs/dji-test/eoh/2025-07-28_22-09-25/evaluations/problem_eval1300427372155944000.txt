import pandas as pd

def heuristics_v2(df):
    def exp_weighted_moving_average(series, window=20):
        weights = (1 + 0.05) ** pd.Series(range(window))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    ewma_20 = exp_weighted_moving_average(df['close'])
    sma_40 = df['close'].rolling(window=40).mean()
    heuristics_matrix = ewma_20 - sma_40
    return heuristics_matrix
