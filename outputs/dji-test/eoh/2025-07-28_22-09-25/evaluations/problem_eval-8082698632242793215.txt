import pandas as pd

def heuristics_v2(df):
    df['Momentum'] = (df['close'].pct_change() + 1).rolling(window=5).apply(lambda x: x.prod(), raw=True) - 1
    df['VolumeActivity'] = df['volume'].pct_change().rolling(window=5).mean()
    df['Volatility'] = df['close'].pct_change().rolling(window=5).std()
    heuristics_matrix = (df['Momentum'] + df['VolumeActivity']) / (1 + df['Volatility'])
    return heuristics_matrix
