def heuristics_v2(df):
    ewma_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    sma_60_close = df['close'].rolling(window=60).mean()
    std_180_high = df['high'].rolling(window=180).std()
    heuristics_matrix = (sma_60_close - ewma_20_volume) / std_180_high
    return heuristics_matrix
