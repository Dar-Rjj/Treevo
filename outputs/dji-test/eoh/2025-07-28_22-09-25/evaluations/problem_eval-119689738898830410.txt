import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=15):
        weights = pd.Series(range(1, window+1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_15 = weighted_moving_average(df['close'])
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = wma_15 - ema_30
    return heuristics_matrix
