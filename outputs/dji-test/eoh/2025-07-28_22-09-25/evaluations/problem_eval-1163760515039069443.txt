def heuristics_v2(df):
    df['Return'] = df['close'].pct_change()
    positive_returns_sum = df[df['Return'] > 0]['Return'].rolling(window=10).sum()
    negative_returns_sum = df[df['Return'] < 0]['Return'].abs().rolling(window=10).sum()
    sentiment_ratio = (positive_returns_sum / (negative_returns_sum + 1e-6)).fillna(0)
    sma_25_close = df['close'].rolling(window=25).mean()
    adjustment_factor = df['close'] - sma_25_close
    heuristics_matrix = sentiment_ratio * adjustment_factor
    return heuristics_matrix
