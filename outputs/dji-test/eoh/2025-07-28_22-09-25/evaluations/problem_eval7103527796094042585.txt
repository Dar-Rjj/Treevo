import pandas as pd

def heuristics_v2(df):
    # Calculate short and long term moving averages
    ma_short = df['close'].rolling(window=10).mean()
    ma_long = df['close'].rolling(window=50).mean()
    
    # Calculate the price rate of change (ROC) for a 14-day period
    roc = df['close'].pct_change(periods=14)
    
    # Heuristic: If the short-term MA is above the long-term MA, give more weight to ROC
    heuristic_weights = (ma_short > ma_long).astype(int) * 2 + 1
    
    # Generate the heuristic factor
    heuristics_matrix = roc * heuristic_weights
    
    return heuristics_matrix
