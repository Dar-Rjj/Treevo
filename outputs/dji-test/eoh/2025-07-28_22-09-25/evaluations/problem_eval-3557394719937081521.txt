def heuristics_v2(df):
    df['Momentum_50'] = df['close'] - df['close'].shift(50)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    sma_volume_20 = df['volume'].rolling(window=20).mean()
    sma_close_200 = df['close'].rolling(window=200).mean()
    heuristics_matrix = (df['Momentum_50'] * 0.3 + df['RSI_14'] * 0.4 + sma_volume_20 * 0.3) * (df['close'] / sma_close_200)
    return heuristics_matrix
