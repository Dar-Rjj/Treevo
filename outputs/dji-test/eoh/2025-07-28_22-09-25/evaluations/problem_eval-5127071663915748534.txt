import pandas as pd

def heuristics_v2(df):
    # Calculate EMA for 10 and 20 days
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Volatility measure over the last 30 days
    volatility = df['close'].rolling(window=30).std()
    
    # Alpha factor combining EMAs with an adjustment for recent volatility
    heuristics_matrix = ((ema_10 - ema_20) / volatility).dropna()
    
    return heuristics_matrix
