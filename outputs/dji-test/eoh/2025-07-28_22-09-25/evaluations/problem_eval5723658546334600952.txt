def heuristics_v2(df):
    sma_5_close = df['close'].rolling(window=5).mean()
    ewma_60_volume = df['volume'].ewm(span=60).mean()
    avg_30_low = df['low'].rolling(window=30).mean()
    high_to_avg_low_ratio = df['high'] / avg_30_low
    
    heuristics_matrix = (sma_5_close * 0.5 + ewma_60_volume * 0.5) * high_to_avg_low_ratio
    return heuristics_matrix
