def heuristics_v2(df):
    df['Momentum_30'] = df['close'] - df['close'].shift(30)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    df['RSI_10'] = 100 - (100 / (1 + rs))
    sma_volume_10 = df['volume'].rolling(window=10).mean()
    sma_close_50 = df['close'].rolling(window=50).mean()
    heuristics_matrix = (df['Momentum_30'] * 0.4 + df['RSI_10'] * 0.3 + sma_volume_10 * 0.3) * (df['close'] / sma_close_50)
    return heuristics_matrix
