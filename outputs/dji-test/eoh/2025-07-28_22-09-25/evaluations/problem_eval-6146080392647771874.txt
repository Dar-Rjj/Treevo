def heuristics_v2(df):
    max_20_close = df['close'].rolling(window=20).max()
    min_20_close = df['close'].rolling(window=20).min()
    diff_20_close = max_20_close - min_20_close
    std_90_price_change = df['close'].diff().rolling(window=90).std()
    heuristics_matrix = diff_20_close / std_90_price_change
    return heuristics_matrix
