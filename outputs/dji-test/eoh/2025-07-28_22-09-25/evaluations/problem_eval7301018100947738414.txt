def heuristics_v2(df):
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Rate_of_Change'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    vwap_10 = (df['close']*df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=90).mean()
    
    heuristics_matrix = (df['Volatility'] * 0.4 + df['Rate_of_Change'] * 0.4 + vwap_10 * 0.2) * df['Volume_Ratio']
    return heuristics_matrix
