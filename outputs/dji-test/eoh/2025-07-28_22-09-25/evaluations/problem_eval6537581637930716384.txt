import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=20):
        weights = pd.Series(range(1, window+1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_20 = weighted_moving_average(df['close'])
    ema_40 = df['close'].ewm(span=40, adjust=False).mean()
    heuristics_matrix = wma_20 - ema_40
    return heuristics_matrix
