def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    sma_20_daily_return = daily_return.rolling(window=20).mean()
    ewma_40_daily_return = daily_return.ewm(span=40, adjust=False).mean()
    heuristics_matrix = sma_20_daily_return - ewma_40_daily_return
    return heuristics_matrix
