import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=30):
        weights = pd.Series(range(1, window+1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_30 = weighted_moving_average(df['close'])
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = wma_30 - ema_50
    return heuristics_matrix
