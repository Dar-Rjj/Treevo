import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day percentage change in close prices
    pct_change_5 = df['close'].pct_change(periods=5)
    
    # Calculate the 50-period weighted moving average of volume
    wma_50_volume = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Alpha factor combining 5-day percentage change and 50-period WMA of volume
    heuristics_matrix = (pct_change_5 * wma_50_volume).dropna()
    
    return heuristics_matrix
