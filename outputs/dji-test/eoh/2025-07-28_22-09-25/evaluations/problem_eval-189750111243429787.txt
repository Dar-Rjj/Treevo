def heuristics_v2(df):
    # Calculate the 20-period Simple Moving Average (SMA) for the close price
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate the 10-period Exponential Moving Average (EMA) for the volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Heuristic combination: Ratio of SMA of close to EMA of volume
    heuristics_matrix = sma_close / ema_volume
    return heuristics_matrix
