def heuristics_v2(df):
    # Calculate the 14-period RSI for the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_close = 100 - (100 / (1 + rs))
    
    # Calculate the 30-day cumulative volume
    cum_vol = df['volume'].rolling(window=30).sum()
    # Calculate the percentage change in the 30-day cumulative volume
    pct_change_cum_vol = cum_vol.pct_change()
    
    # Heuristic combination: Weighted sum of RSI and percentage change in 30-day cumulative volume
    heuristics_matrix = 0.7 * rsi_close + 0.3 * pct_change_cum_vol
    return heuristics_matrix
