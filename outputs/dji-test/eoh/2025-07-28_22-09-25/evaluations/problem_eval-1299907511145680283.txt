import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low, then multiply by volume
    raw_factor = (df['high'] - df['low']) * df['volume']
    # Apply a simple moving average over a 10-day window to smooth the signal
    heuristics_matrix = raw_factor.rolling(window=10).mean()
    return heuristics_matrix
