def heuristics_v2(df):
    sma_close_10 = df['close'].rolling(window=10).mean()
    sma_close_5 = df['close'].rolling(window=5).mean()
    std_dev_return_20 = df['close'].pct_change().rolling(window=20).std()
    heuristics_matrix = (sma_close_10 - sma_close_5) / std_dev_return_20
    return heuristics_matrix
