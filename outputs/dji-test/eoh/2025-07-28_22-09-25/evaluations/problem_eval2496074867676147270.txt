def heuristics_v2(df):
    # Calculate the 21-period Rate of Change (ROC) for the close price
    roc_close = df['close'].pct_change(periods=21)
    
    # Calculate the 21-period sum of volume
    sum_volume = df['volume'].rolling(window=21).sum()
    
    # Heuristic combination: Weighted sum of ROC and sum of volume
    heuristics_matrix = 0.6 * roc_close + 0.4 * sum_volume
    return heuristics_matrix
