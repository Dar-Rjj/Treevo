import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=30):
        weights = pd.Series(range(window, 0, -1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_30 = weighted_moving_average(df['close'])
    sma_50 = df['close'].rolling(window=50).mean()
    heuristics_matrix = wma_30 - sma_50
    return heuristics_matrix
