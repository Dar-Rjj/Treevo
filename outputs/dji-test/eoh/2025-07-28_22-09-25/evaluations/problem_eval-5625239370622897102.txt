import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the on-balance volume (OBV)
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine RSI and OBV
    combined_factor = rsi * obv
    
    # Apply a 20-day exponential moving average to smooth the signal
    heuristics_matrix = combined_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
