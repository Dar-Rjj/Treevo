def heuristics_v2(df):
    df['Return'] = df['close'].pct_change()
    std_dev = df['Return'].rolling(window=30).std()
    sma_50 = df['close'].rolling(window=50).mean()
    momentum = sma_50 - df['close']
    volume_ratio_inv = df['volume'].rolling(window=20).mean() / df['volume']
    
    heuristics_matrix = (std_dev * 0.4 + momentum * 0.6) * volume_ratio_inv
    return heuristics_matrix
