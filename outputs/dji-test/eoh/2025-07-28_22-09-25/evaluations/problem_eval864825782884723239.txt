import pandas as pd

def heuristics_v2(df):
    # Calculating 14-day momentum
    momentum_14 = df['close'] - df['close'].shift(14)
    
    # Calculating 5-day rate of change of volume
    roc_volume_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Heuristic combination: Sum of momentum and rate of change of volume
    heuristics_matrix = momentum_14 + roc_volume_5
    return heuristics_matrix
