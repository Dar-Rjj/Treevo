import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=10):
        weights = pd.Series(range(1, window+1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_10 = weighted_moving_average(df['close'])
    sma_50 = df['close'].rolling(window=50).mean()
    heuristics_matrix = wma_10 - sma_50
    return heuristics_matrix
