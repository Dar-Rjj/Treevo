def heuristics_v2(df):
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_open_5 = df['open'].ewm(span=5, adjust=False).mean()
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    price_diff = ema_close_10 - ema_open_5
    heuristics_matrix = price_diff / ema_volume_20
    return heuristics_matrix
