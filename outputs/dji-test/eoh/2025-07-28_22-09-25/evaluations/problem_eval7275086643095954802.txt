import pandas as pd

def heuristics_v2(df):
    def weighted_moving_average(series, window=20):
        weights = pd.Series(range(window, 0, -1))
        wma = series.rolling(window=window).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma
    
    wma_20 = weighted_moving_average(df['close'])
    sma_40 = df['close'].rolling(window=40).mean()
    heuristics_matrix = wma_20 - sma_40
    return heuristics_matrix
