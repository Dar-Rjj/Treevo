import pandas as pd

def heuristics_v2(df):
    # Calculate EMA of the close price
    ema_close = df.close.ewm(span=14, adjust=False).mean()
    
    # Calculate the daily volume change
    volume_change = df.volume.pct_change()
    
    # Calculate the standard deviation of volume over the last 30 days
    std_volume = df.volume.rolling(window=30).std()
    
    # Volume change to volume standard deviation ratio
    vol_ratio = volume_change / std_volume
    
    # Calculate the daily trading range
    trading_range = df.high - df.low
    
    # Calculate EMA of the trading range
    ema_trading_range = trading_range.ewm(span=14, adjust=False).mean()
    
    # Heuristic combination: Weighted sum of EMA of the close, volume ratio, and EMA of the trading range
    heuristics_matrix = 0.5 * ema_close + 0.3 * vol_ratio + 0.2 * ema_trading_range
    return heuristics_matrix
