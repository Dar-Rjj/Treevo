import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change (ROC) for the close price
    roc_close = df['close'].pct_change(periods=10)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the standard deviation of ATR over the last 30 days
    std_atr = atr.rolling(window=30).std()
    
    # ATR to ATR standard deviation ratio
    atr_ratio = atr / std_atr
    
    # Heuristic combination: Weighted sum of ROC and ATR ratio
    heuristics_matrix = 0.7 * roc_close + 0.3 * atr_ratio
    return heuristics_matrix
