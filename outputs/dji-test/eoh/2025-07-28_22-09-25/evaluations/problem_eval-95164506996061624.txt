def heuristics_v2(df):
    # Calculate the 10-period Rate of Change (ROC) for the close price
    roc_close = df['close'].pct_change(periods=10)
    
    # Calculate the 20-period Exponential Moving Average (EMA) of the volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Heuristic combination: Weighted sum of ROC and EMA of volume
    heuristics_matrix = 0.6 * roc_close + 0.4 * ema_volume
    return heuristics_matrix
