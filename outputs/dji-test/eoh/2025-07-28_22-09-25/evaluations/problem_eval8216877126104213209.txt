def heuristics_v2(df):
    # Calculate the 14-period Rate of Change (ROC) for the close price
    roc_close = df['close'].pct_change(periods=14)
    
    # Calculate the 20-period Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=20).sum() / negative_money_flow.rolling(window=20).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Heuristic combination: Weighted sum of ROC and MFI
    heuristics_matrix = 0.6 * roc_close + 0.4 * mfi
    return heuristics_matrix
