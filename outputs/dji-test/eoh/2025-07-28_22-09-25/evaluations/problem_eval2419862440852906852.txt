import pandas as pd

def heuristics_v2(df):
    # Calculating short and long period WMACDs
    short_wma = df.close.rolling(window=12).mean()
    long_wma = df.close.rolling(window=26).mean()
    wmacd = short_wma - long_wma
    
    # Adding a simple daily return as another signal
    daily_return = df.close.pct_change()
    
    # Heuristic combination: Weighted sum of WMACD and daily return
    heuristics_matrix = 0.7 * wmacd + 0.3 * daily_return
    return heuristics_matrix
