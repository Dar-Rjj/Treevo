def heuristics_v2(df):
    price_changes = df[['open', 'high', 'low', 'close']].pct_change().fillna(0)
    volume_change = (df['volume'] - df['volume'].shift(1)).fillna(0)
    weights = [0.25, 0.25, 0.25, 0.25, 0.5]  # Weights for open, high, low, close, and volume change
    heuristics_matrix = (price_changes * weights[:4]).sum(axis=1) + volume_change * weights[4]
    return heuristics_matrix
