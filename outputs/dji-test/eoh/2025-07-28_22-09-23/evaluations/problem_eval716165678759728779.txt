def heuristics_v2(df):
    df['price_change'] = df['close'].diff()
    df['price_change_sum'] = df['price_change'].rolling(window=120).sum()
    df['volume_total'] = df['volume'].rolling(window=120).sum()
    heuristics_matrix = df['price_change_sum'] / df['volume_total']
    return heuristics_matrix
