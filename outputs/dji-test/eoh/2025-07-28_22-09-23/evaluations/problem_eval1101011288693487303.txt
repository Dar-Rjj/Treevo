def heuristics_v2(df):
    vol_to_avg_vol = df['volume'] / df['volume'].rolling(window=10).mean()
    close_open_diff = df['close'] - df['open']
    std_close_short = df['close'].rolling(window=5).std()
    heuristics_matrix = vol_to_avg_vol * close_open_diff / std_close_short
    return heuristics_matrix
