def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['abs_return_diff'] = df['daily_return'].abs().rolling(window=60).sum()
    df['volume_sum'] = df['volume'].rolling(window=60).sum()
    heuristics_matrix = df['abs_return_diff'] / df['volume_sum']
    return heuristics_matrix
