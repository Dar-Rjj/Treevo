def heuristics_v2(df):
    def calculate_heuristic_diff(row):
        return row['high'] - row['low']
    
    df['heuristic_diff'] = df.apply(calculate_heuristic_diff, axis=1)
    heuristics_matrix = df['heuristic_diff'].rolling(window=60).sum() / df['close'].rolling(window=60).sum()
    return heuristics_matrix
