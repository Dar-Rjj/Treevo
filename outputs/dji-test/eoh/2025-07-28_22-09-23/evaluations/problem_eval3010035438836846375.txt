import pandas as pd

def heuristics_v2(df):
    # Calculate EMAs
    ema1 = df['close'].ewm(span=10, adjust=False).mean()
    ema2 = df['close'].ewm(span=20, adjust=False).mean()
    ema_diff = ema1 - ema2
    
    # Calculate weighted sum of the last 7 days' closing prices
    weights = [0.1, 0.15, 0.2, 0.25, 0.15, 0.1, 0.05]
    weighted_sum = df['close'].rolling(window=len(weights)).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Combine EMA and weighted sum for the final heuristic
    heuristics_matrix = ema_diff + weighted_sum.shift(1)
    return heuristics_matrix
