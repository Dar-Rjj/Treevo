def heuristics_v2(df):
    roc_short = df['close'].pct_change(periods=10)
    roc_long = df['close'].pct_change(periods=50)
    weights = list(range(1, 15))  # Weights for the WMA
    wma = df['close'].rolling(window=14).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = roc_short * rsi + roc_long * (wma / df['close'])
    return heuristics_matrix
