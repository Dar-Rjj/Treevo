def heuristics_v2(df):
    avg_high_low = (df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2
    price_volatility = df['close'] - avg_high_low
    vol_ratio = df['volume'] / df['volume'].rolling(window=30).median()
    heuristics_matrix = price_volatility + vol_ratio
    return heuristics_matrix
