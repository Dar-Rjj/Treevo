import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    avg_high_short = df['high'].rolling(window=short_window).mean()
    avg_low_short = df['low'].rolling(window=short_window).mean()
    avg_high_long = df['high'].rolling(window=long_window).mean()
    avg_low_long = df['low'].rolling(window=long_window).mean()
    short_term_signal = (avg_high_short - avg_low_short) / df['close']
    long_term_signal = (avg_high_long - avg_low_long) / df['close']
    heuristics_matrix = short_term_signal - long_term_signal
    return heuristics_matrix
