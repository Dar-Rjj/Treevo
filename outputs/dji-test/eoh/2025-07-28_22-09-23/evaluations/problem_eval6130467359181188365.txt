def heuristics_v2(df):
    df['high_max'] = df['high'].rolling(window=90).max()
    df['low_min'] = df['low'].rolling(window=90).min()
    df['volume_mean'] = df['volume'].rolling(window=90).mean()
    heuristics_matrix = (df['high_max'] - df['low_min']) / df['volume_mean']
    return heuristics_matrix
