import pandas as pd

def heuristics_v2(df):
    def calculate_roc(series, n=14):
        return (series - series.shift(n)) / series.shift(n) * 100
    
    def custom_volatility(df, n=14):
        return df['close'].rolling(window=n).std()
    
    close_roc = calculate_roc(df['close'])
    close_volatility = custom_volatility(df)
    heuristics_matrix = (close_roc + close_volatility) / 2
    return heuristics_matrix
