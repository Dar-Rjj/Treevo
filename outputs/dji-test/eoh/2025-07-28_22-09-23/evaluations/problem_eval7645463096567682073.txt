def heuristics_v2(df):
    def calculate_heuristic(row):
        return (row['high'] - row['low']) / row['close']
    
    df['heuristic'] = df.apply(calculate_heuristic, axis=1)
    heuristics_ema = df['heuristic'].ewm(span=10, adjust=False).mean()
    close_sma = df['close'].rolling(window=30).mean()
    heuristics_matrix = heuristics_ema / close_sma
    return heuristics_matrix
