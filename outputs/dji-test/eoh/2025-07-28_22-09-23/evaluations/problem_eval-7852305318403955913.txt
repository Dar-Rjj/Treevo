def heuristics_v2(df):
    df['moving_avg_10'] = df['close'].rolling(window=10).mean()
    df['relative_diff'] = (df['close'] - df['moving_avg_10']) / df['moving_avg_10']
    df['weighted_volume'] = df['volume'] * df['relative_diff']
    heuristics_matrix = df['weighted_volume'].cumsum() / df['volume'].rolling(window=30).sum()
    return heuristics_matrix
