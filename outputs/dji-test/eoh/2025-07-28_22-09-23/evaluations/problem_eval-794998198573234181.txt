def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['moving_avg_60'] = df['close'].rolling(window=60).mean()
    df['relative_diff'] = (df['close'] - df['moving_avg_60']) / df['moving_avg_60']
    df['avg_relative_diff'] = df['relative_diff'].rolling(window=60).mean()
    df['weighted_diff'] = df['daily_return'] * df['avg_relative_diff']
    heuristics_matrix = df['weighted_diff'].rolling(window=60).sum() / df['daily_return'].rolling(window=60).std()
    return heuristics_matrix
