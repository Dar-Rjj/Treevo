def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['median_60'] = df['close'].rolling(window=60).median()
    df['relative_diff'] = (df['close'] - df['median_60']) / df['median_60']
    df['weighted_diff'] = df['daily_return'] * df['relative_diff']
    df['range_60'] = df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()
    heuristics_matrix = df['weighted_diff'].rolling(window=60).sum() / df['range_60']
    return heuristics_matrix
