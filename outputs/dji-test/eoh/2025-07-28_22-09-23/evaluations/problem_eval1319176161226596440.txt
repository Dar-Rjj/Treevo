def heuristics_v2(df):
    window = 14
    # Calculate the rolling standard deviation of the closing prices
    std_close = df['close'].rolling(window=window).std()
    
    # Calculate the rate of change over a short period
    roc_short = df['close'].pct_change(periods=5)
    
    # Adjust the ROC by the rolling standard deviation
    adjusted_roc = roc_short / std_close
    
    # Calculate the volume-weighted price change
    vwp_change = (df['close'] - df['close'].shift(1)) * df['volume']
    vwp_change = vwp_change.rolling(window=window).mean()
    
    # Combine the adjusted ROC and the volume-weighted price change
    heuristics_matrix = adjusted_roc + vwp_change
    
    return heuristics_matrix
