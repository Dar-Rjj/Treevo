def heuristics_v2(df):
    def calculate_heuristic(row):
        return (row['high'] - row['low']) / row['close']
    
    df['heuristic'] = df.apply(calculate_heuristic, axis=1)
    heuristics_matrix = df['heuristic'].rolling(window=20).mean() / df['close'].rolling(window=20).mean()
    momentum = df['close'].pct_change(periods=20).fillna(0)
    heuristics_matrix = heuristics_matrix + momentum
    return heuristics_matrix
