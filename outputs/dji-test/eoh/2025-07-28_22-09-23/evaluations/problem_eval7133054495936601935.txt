def heuristics_v2(df):
    def roc(series, n=14):
        return (series / series.shift(n) - 1) * 100

    def adl_smoothed(df, n=14):
        money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
        money_flow_volume = money_flow_multiplier * df['volume']
        adl_sma = money_flow_volume.rolling(window=n).mean()
        return adl_sma.cumsum()

    volume_roc = roc(df['volume'])
    adl_line = adl_smoothed(df)
    heuristics_matrix = (volume_roc + adl_line) / 2
    return heuristics_matrix
