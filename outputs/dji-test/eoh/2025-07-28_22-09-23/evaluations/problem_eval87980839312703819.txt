def heuristics_v2(df):
    df['daily_range'] = df['high'] - df['low']
    daily_range_avg = df['daily_range'].rolling(window=30).mean()
    close_diff = df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min()
    heuristics_matrix = daily_range_avg / close_diff
    return heuristics_matrix
