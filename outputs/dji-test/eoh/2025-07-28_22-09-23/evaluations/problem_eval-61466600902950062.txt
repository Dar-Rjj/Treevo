def heuristics_v2(df):
    def roc(series, n=21):
        return (series / series.shift(n) - 1) * 100
    
    def adl_smoothed(df, n=21):
        money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
        money_flow_volume = money_flow_multiplier * df['volume']
        adl_ema = money_flow_volume.ewm(span=n, adjust=False).mean()
        return adl_ema.cumsum()
    
    close_roc = roc(df['close'])
    adl_line = adl_smoothed(df)
    heuristics_matrix = (close_roc + adl_line) / 2
    return heuristics_matrix
