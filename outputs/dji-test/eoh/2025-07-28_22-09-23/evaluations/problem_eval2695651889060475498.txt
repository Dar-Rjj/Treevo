def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    avg_positive_return = df['positive_returns'].rolling(window=90).mean()
    total_abs_returns = df['daily_return'].abs().rolling(window=90).sum()
    heuristics_matrix = avg_positive_return / total_abs_returns
    return heuristics_matrix
