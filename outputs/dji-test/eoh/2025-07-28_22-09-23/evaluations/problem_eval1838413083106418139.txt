import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Calculate weighted moving average of the last 5 days' closing prices
    weights = [0.15, 0.2, 0.25, 0.3, 0.1]
    weighted_avg = df['close'].rolling(window=len(weights)).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Combine ROC and weighted average for the final heuristic
    heuristics_matrix = roc + weighted_avg.shift(1)
    return heuristics_matrix
