def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    ema_daily_return = daily_return.ewm(span=10, adjust=False).mean()
    avg_volume_short = df['volume'].rolling(window=5).mean()
    volume_close_ratio = avg_volume_short / df['close']
    heuristics_matrix = ema_daily_return * volume_close_ratio
    return heuristics_matrix
