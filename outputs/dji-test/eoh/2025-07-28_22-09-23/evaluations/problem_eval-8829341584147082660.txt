def heuristics_v2(df):
    df['price_movement'] = (df['close'] - df['open']).abs()
    price_movement_avg = df['price_movement'].rolling(window=45).mean()
    volume_change = df['volume'].pct_change().fillna(0).rolling(window=45).sum()
    heuristics_matrix = volume_change / price_movement_avg
    return heuristics_matrix
