def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['volume_rolling_avg_60'] = df['volume'].rolling(window=60).mean()
    df['volume_ratio'] = df['volume'] / df['volume_rolling_avg_60']
    df['weighted_vol_ratio'] = df['daily_return'] * df['volume_ratio']
    heuristics_matrix = df['weighted_vol_ratio'].rolling(window=60).mean() / df['daily_return'].rolling(window=60).std()
    return heuristics_matrix
