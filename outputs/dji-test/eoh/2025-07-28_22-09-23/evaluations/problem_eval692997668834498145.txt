def heuristics_v2(df):
    df['volume_sum'] = df['volume'].rolling(window=30).sum()
    df['range_avg'] = (df['high'] - df['low']).rolling(window=30).mean()
    liquidity_ratio = df['volume_sum'] / df['range_avg']
    
    def rsi(series, n=21):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=n, min_periods=1).mean()
        avg_loss = loss.rolling(window=n, min_periods=1).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    close_rsi = rsi(df['close'])
    heuristics_matrix = liquidity_ratio * close_rsi
    return heuristics_matrix
