def heuristics_v2(df):
    df['price_change'] = df['close'].diff()
    df['price_change_10_sum'] = df['price_change'].rolling(window=10).sum()
    df['close_30_std'] = df['close'].rolling(window=30).std()
    heuristics_matrix = df['price_change_10_sum'] / df['close_30_std']
    return heuristics_matrix
