import pandas as pd
import scipy.stats as stats

def heuristics_v2(df):
    def geometric_mean(row):
        return stats.gmean([row['high'], row['low']])
    
    df['geo_mean'] = df.apply(geometric_mean, axis=1)
    geo_mean_rolling = df['geo_mean'].rolling(window=60).mean()
    close_mean_rolling = df['close'].rolling(window=60).mean()
    heuristics_matrix = geo_mean_rolling / close_mean_rolling
    return heuristics_matrix
