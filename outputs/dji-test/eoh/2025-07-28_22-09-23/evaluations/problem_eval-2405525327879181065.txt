import pandas as pd

def heuristics_v2(df):
    def rsi(series, n=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def vwma(df, window=20):
        return df['close'].rolling(window=window).mean() * df['volume'].rolling(window=window).mean()

    close_rsi = rsi(df['close'])
    close_vwma = vwma(df)
    heuristics_matrix = (close_rsi + close_vwma) / 2
    return heuristics_matrix
