def heuristics_v2(df):
    df['close_pct_change'] = df['close'].pct_change()
    heuristics_numerator = df['close_pct_change'].rolling(window=30).sum()
    heuristics_denominator = df['volume'].rolling(window=30).std()
    heuristics_matrix = heuristics_numerator / heuristics_denominator
    return heuristics_matrix
