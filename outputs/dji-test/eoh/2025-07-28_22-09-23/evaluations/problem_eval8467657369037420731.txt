def heuristics_v2(df):
    df['close_momentum'] = (df['close'] - df['close'].rolling(window=30).min()) / df['close'].rolling(window=30).min()
    df['volume_ratio'] = df['volume'].rolling(window=30).mean() / df['volume'].rolling(window=30).max()
    heuristics_matrix = df['close_momentum'] * df['volume_ratio']
    return heuristics_matrix
