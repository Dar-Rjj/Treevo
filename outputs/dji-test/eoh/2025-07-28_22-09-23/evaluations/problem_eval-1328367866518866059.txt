def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['moving_avg_30'] = df['close'].rolling(window=30).mean()
    df['relative_diff'] = (df['close'] - df['moving_avg_30']) / df['moving_avg_30']
    df['weighted_diff'] = df['daily_return'] * df['relative_diff']
    range_30 = df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min()
    heuristics_matrix = df['weighted_diff'].rolling(window=30).sum() / range_30
    return heuristics_matrix
