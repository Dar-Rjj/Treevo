def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['mean_60'] = df['close'].rolling(window=60).mean()
    df['relative_diff'] = (df['close'] - df['mean_60']) / df['mean_60']
    df['weighted_diff'] = df['daily_return'] * df['relative_diff']
    df['std_60'] = df['close'].rolling(window=60).std()
    heuristics_matrix = df['weighted_diff'].rolling(window=60).mean() * df['std_60']
    return heuristics_matrix
