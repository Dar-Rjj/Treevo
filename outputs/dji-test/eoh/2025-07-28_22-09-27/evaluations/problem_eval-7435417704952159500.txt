def heuristics_v2(df):
    # Calculate the 14-day EWM of volume
    ewma_volume_14 = df['volume'].ewm(span=14).mean()
    
    # Find the 25-day minimum volume
    min_volume_25 = df['volume'].rolling(window=25).min()
    
    # Compute the ratio of 14-day EWM volume to 25-day minimum volume
    vol_ratio = ewma_volume_14 / min_volume_25
    
    # Calculate the 14-day and 28-day SMA of the close price
    sma_close_14 = df['close'].rolling(window=14).mean()
    sma_close_28 = df['close'].rolling(window=28).mean()
    
    # Compute the difference between the 14-day and 28-day SMAs
    sma_diff = sma_close_14 - sma_close_28
    
    # Combine the two components to create the heuristic factor
    heuristics_matrix = vol_ratio + sma_diff
    
    return heuristics_matrix
