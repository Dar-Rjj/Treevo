def heuristics_v2(df):
    volume_close_ratio = df['volume'] / df['close']
    ema_50 = volume_close_ratio.ewm(span=50, adjust=False).mean()
    ema_200 = volume_close_ratio.ewm(span=200, adjust=False).mean()
    daily_return = df['close'].pct_change()
    ema_10_daily_return = daily_return.ewm(span=10, adjust=False).mean()
    heuristics_matrix = (ema_50 - ema_200) * ema_10_daily_return
    return heuristics_matrix
