import pandas as pd

def heuristics_v2(df):
    sma_10_close = df['close'].rolling(window=10).mean()
    std_20_close = df['close'].rolling(window=20).std()
    sma_5_volume = df['volume'].rolling(window=5).mean()
    
    heuristics_matrix = (sma_10_close + std_20_close + sma_5_volume) / 3
    
    return heuristics_matrix
