def heuristics_v2(df):
    window = 10
    close_prices = df['close']
    volumes = df['volume']
    
    # Calculate the percentage change in closing prices over the window
    pct_change_close = close_prices.pct_change(window)
    
    # Calculate the standard deviation of volume over the window
    std_volume = volumes.rolling(window=window).std()
    
    # Compute the heuristics factor
    heuristics_matrix = pct_change_close + std_volume
    
    return heuristics_matrix
