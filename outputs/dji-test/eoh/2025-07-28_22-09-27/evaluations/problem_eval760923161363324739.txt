import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving average of returns for enhanced momentum
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=20, adjust=False).mean()
    
    # Enhanced mean reversion factor using standard deviation for 30-day period
    sma_close = df['close'].rolling(window=30).mean()
    std_close = df['close'].rolling(window=30).std()
    mean_reversion_factor = (df['close'] - sma_close) / std_close
    
    # Adjusted liquidity as the average daily volume over 60 days
    liquidity = df['volume'].rolling(window=60).mean()
    
    # Combine factors into a heuristics matrix using a weighted sum
    heuristics_matrix = 0.5 * ema_returns - 0.3 * mean_reversion_factor + 0.2 * liquidity
    
    return heuristics_matrix
