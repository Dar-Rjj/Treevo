def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'].diff()
    
    # Calculate the volume-weighted price change
    volume_weighted_change = price_change * df['volume']
    
    # Compute the 20-day cumulative sum of the volume-weighted price changes
    heuristics_matrix = volume_weighted_change.rolling(window=20).sum()
    
    return heuristics_matrix
