def heuristics_v2(df):
    # Calculate exponential moving average of returns for momentum factor
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=20, adjust=False).mean()

    # Calculate mean reversion factor as the standard deviation from 50-day SMA
    sma_close = df['close'].rolling(window=50).mean()
    std_close = df['close'].rolling(window=50).std()
    mean_reversion_factor = (df['close'] - sma_close) / std_close

    # Adjusted liquidity calculation: average volume over 90 days divided by close price
    liquidity = (df['volume'].rolling(window=90).mean() / df['close'])

    # Combine factors into a heuristics matrix using a weighted sum
    heuristics_matrix = 0.5 * ema_returns - 0.3 * mean_reversion_factor + 0.2 * liquidity

    return heuristics_matrix
