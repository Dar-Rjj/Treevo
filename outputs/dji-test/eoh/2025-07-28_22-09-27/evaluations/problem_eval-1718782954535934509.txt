def heuristics_v2(df):
    adjusted_price = df['close'] * (df['volume'] / df['volume'].rolling(window=10).mean())
    ema_long = adjusted_price.ewm(span=20, adjust=False).mean()
    ema_short = adjusted_price.ewm(span=5, adjust=False).mean()
    heuristics_matrix = ema_short - ema_long
    return heuristics_matrix
