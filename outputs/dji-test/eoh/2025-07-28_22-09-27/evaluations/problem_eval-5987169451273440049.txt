def heuristics_v2(df):
    avg_price = (df['high'] + df['low']) / 2
    volume_avg_price_ratio = df['volume'] / avg_price
    ema_10 = volume_avg_price_ratio.ewm(span=10, adjust=False).mean()
    ema_30 = volume_avg_price_ratio.ewm(span=30, adjust=False).mean()
    heuristics_matrix = ema_10 - ema_30
    return heuristics_matrix
