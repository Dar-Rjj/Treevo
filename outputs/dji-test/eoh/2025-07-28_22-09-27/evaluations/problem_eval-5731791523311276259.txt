import pandas as pd

def heuristics_v2(df):
    # Calculate the MACD-based trend indicator
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    trend_indicator = macd_line - signal_line
    
    # Calculate the volatility-adjusted momentum
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=20, adjust=False).mean()
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=20).mean()
    vol_adj_momentum = ema_returns / atr
    
    # Calculate the change in volume over a 50-day period
    volume_change = df['volume'].pct_change(periods=50)
    
    # Combine factors into a heuristics matrix using a weighted sum
    heuristics_matrix = 0.4 * trend_indicator + 0.4 * vol_adj_momentum + 0.2 * volume_change
    
    return heuristics_matrix
