def heuristics_v2(df):
    # Calculate simple return for momentum
    returns = df['close'].pct_change().fillna(0)
    
    # Calculate volatility (standard deviation of returns)
    volatility = returns.rolling(window=20).std().fillna(0)
    
    # Calculate liquidity (volume divided by close price)
    liquidity = (df['volume'] / df['close']).fillna(0)
    
    # Combine factors into a heuristics matrix using a weighted product
    heuristics_matrix = (returns + 1) ** 0.4 * (volatility + 1) ** 0.4 * (liquidity + 1) ** 0.2 - 1
    
    return heuristics_matrix
