def heuristics_v2(df):
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['volume_weighted_tp'] = df['typical_price'] * df['volume']
    ewma_14 = df['volume_weighted_tp'].ewm(span=14).mean()
    ewma_28 = df['volume_weighted_tp'].ewm(span=28).mean()
    heuristics_matrix = ewma_14 - ewma_28
    return heuristics_matrix
