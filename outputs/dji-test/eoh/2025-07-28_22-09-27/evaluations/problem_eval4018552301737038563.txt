import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the ratio of the high-low range to the close price
    price_ratio = high_low_range / df['close']
    
    # Apply a 14-day Exponential Moving Average (EMA) to the price ratio
    heuristics_matrix = price_ratio.ewm(span=14, adjust=False).mean()
    
    return heuristics_matrix
