def heuristics_v2(df):
    # Calculate ROC for 10 and 30 day periods to capture short and medium-term momentum
    roc_10 = df['close'].pct_change(periods=10)
    roc_30 = df['close'].pct_change(periods=30)
    
    # Volatility factor: 20-day standard deviation of daily returns
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=20).std()
    
    # Buying pressure: ratio of the high price to the low price, indicating market sentiment
    buying_pressure = df['high'] / df['low']
    
    # Heuristics matrix with weights assigned to each component
    heuristics_matrix = 0.4 * roc_10 + 0.3 * roc_30 - 0.2 * volatility + 0.1 * buying_pressure
    
    return heuristics_matrix
