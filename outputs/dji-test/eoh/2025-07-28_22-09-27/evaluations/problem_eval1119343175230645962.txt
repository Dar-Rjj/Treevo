import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages
    short_term_sma = df['close'].rolling(window=5).mean()
    long_term_sma = df['close'].rolling(window=20).mean()
    
    # Calculate difference between short-term and long-term SMAs
    sma_diff = short_term_sma - long_term_sma
    
    # Calculate the ratio of current volume to 20-day average volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Calculate the 10-day rate of change in closing price
    roc_10 = df['close'].pct_change(periods=10)
    
    # Combine factors into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.3 * sma_diff + 0.4 * volume_ratio + 0.3 * roc_10
    
    return heuristics_matrix
