import pandas as pd

def heuristics_v2(df):
    # Calculate a custom weighted sum of features (example: 2*open + high - low + close)
    weighted_sum = 2*df['open'] + df['high'] - df['low'] + df['close']
    # Apply a 7-day moving average to smooth the series and potentially highlight trends
    heuristics_matrix = weighted_sum.rolling(window=7).mean()
    return heuristics_matrix
