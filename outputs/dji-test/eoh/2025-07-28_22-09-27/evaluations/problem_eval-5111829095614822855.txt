def heuristics_v2(df):
    # Calculate simple return for momentum
    returns = df['close'].pct_change()
    
    # Calculate volatility (standard deviation of returns)
    volatility = returns.rolling(window=20).std()
    
    # Calculate liquidity (volume divided by close price)
    liquidity = df['volume'] / df['close']
    
    # Calculate 20-day Simple Moving Average (SMA) of close price
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Combine factors into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.3 * returns + 0.3 * volatility + 0.2 * liquidity + 0.2 * (df['close'] - sma_20)
    
    return heuristics_matrix
