import pandas as pd

def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    short_term_volatility = daily_returns.rolling(window=30).std()
    long_term_volatility = daily_returns.rolling(window=120).std()
    heuristics_matrix = short_term_volatility / long_term_volatility
    return heuristics_matrix
