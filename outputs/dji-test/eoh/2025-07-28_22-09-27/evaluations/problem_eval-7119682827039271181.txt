def heuristics_v2(df):
    # Calculate daily range
    daily_range = df['high'] - df['low']
    # Calculate the ratio of daily range to close price
    range_to_close_ratio = daily_range / df['close']
    # Apply a 20-day EMA to the ratio
    ema_ratio = range_to_close_ratio.ewm(span=20, adjust=False).mean()
    # Calculate daily return
    returns = df['close'].pct_change()
    # Combine EMA of the ratio and returns into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.7 * ema_ratio + 0.3 * returns
    return heuristics_matrix
