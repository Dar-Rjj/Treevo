def heuristics_v2(df):
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['volume_adjusted_tp'] = df['typical_price'] * df['volume'].pow(0.5)
    ewma_10 = df['volume_adjusted_tp'].ewm(span=10).mean()
    ewma_30 = df['volume_adjusted_tp'].ewm(span=30).mean()
    heuristics_matrix = ewma_10 - ewma_30
    return heuristics_matrix
