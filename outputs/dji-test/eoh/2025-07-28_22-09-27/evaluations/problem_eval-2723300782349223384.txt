def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    price_ratio = high_low_range / df['close']
    ema_price_ratio = price_ratio.ewm(span=20, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = ema_price_ratio * ema_volume
    return heuristics_matrix
