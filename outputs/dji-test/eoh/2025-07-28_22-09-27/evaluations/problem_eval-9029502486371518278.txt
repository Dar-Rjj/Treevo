def heuristics_v2(df):
    # Calculate the percentage change in closing prices
    price_change = df['close'].pct_change()
    
    # Calculate the ratio of today's volume to the average volume over a 20-day window
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Calculate the absolute value of the difference between the current close and the 10-day moving average of closes
    ma_10 = df['close'].rolling(window=10).mean()
    price_consistency = (df['close'] - ma_10).abs()
    
    # Combine factors into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.5 * price_change + 0.3 * volume_ratio - 0.2 * price_consistency
    
    return heuristics_matrix
