import pandas as pd

def heuristics_v2(df):
    high_close_20d = df['close'].rolling(window=20).max()
    low_close_20d = df['close'].rolling(window=20).min()
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    heuristics_matrix = (high_close_20d - low_close_20d) * avg_volume_20d
    return heuristics_matrix
