def heuristics_v2(df):
    # Calculate daily return differences for momentum
    returns_diff = df['close'].diff()
    
    # Calculate high-low price range for volatility
    price_range = df['high'] - df['low']
    
    # Use volume directly as a measure of liquidity
    
    # Combine factors into a heuristics matrix
    heuristics_matrix = (returns_diff + price_range + df['volume']).rolling(window=20).mean()
    
    return heuristics_matrix
