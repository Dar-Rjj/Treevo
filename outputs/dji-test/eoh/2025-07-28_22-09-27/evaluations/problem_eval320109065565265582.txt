import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the highest and lowest prices over the last 20 days
    df['high_low_diff'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 20-day average of the lowest price
    df['avg_low_20'] = df['low'].rolling(window=20).mean()
    
    # Calculate the ratio of today's closing price to the 20-day average of the lowest price
    df['close_to_avg_low_ratio'] = df['close'] / df['avg_low_20']
    
    # Calculate the exponential moving average (EMA) of the volume over the last 20 days
    df['volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine the three factors into one using equal weights
    df['heuristics_factor'] = (df['high_low_diff'] + df['close_to_avg_low_ratio'] + df['volume_ema_20']) / 3
    
    heuristics_matrix = df['heuristics_factor'].copy()
    
    return heuristics_matrix
