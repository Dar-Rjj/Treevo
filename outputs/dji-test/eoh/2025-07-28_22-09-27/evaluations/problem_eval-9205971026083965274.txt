import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 50-day moving averages of the close price
    df['ma_close_20'] = df['close'].rolling(window=20).mean()
    df['ma_close_50'] = df['close'].rolling(window=50).mean()
    
    # Compute the ratio of the current close price to its 20-day and 50-day moving averages
    df['close_ma_ratio_20'] = df['close'] / df['ma_close_20']
    df['close_ma_ratio_50'] = df['close'] / df['ma_close_50']
    
    # Calculate the 20-day and 50-day minimum trading volumes
    df['min_volume_50'] = df['volume'].rolling(window=50).min()
    
    # Determine the ratio of the 20-day average volume to the 50-day minimum volume
    df['vol_min_ratio'] = df['volume'].rolling(window=20).mean() / df['min_volume_50']
    
    # Combine the ratios to create the heuristic factor
    heuristics_matrix = (df['close_ma_ratio_20'] + df['close_ma_ratio_50']) * df['vol_min_ratio']
    
    return heuristics_matrix
