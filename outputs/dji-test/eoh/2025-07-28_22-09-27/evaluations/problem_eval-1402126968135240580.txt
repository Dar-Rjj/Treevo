def heuristics_v2(df):
    short_window = 5
    long_window = 20
    close_prices = df['close']
    volumes = df['volume']
    
    # Calculate moving averages for close prices
    short_ma = close_prices.rolling(window=short_window).mean()
    long_ma = close_prices.rolling(window=long_window).mean()
    
    # Calculate the average volume over the long window
    avg_volume_long = volumes.rolling(window=long_window).mean()
    
    # Compute the heuristics factor
    heuristics_matrix = (short_ma - long_ma) + (volumes / avg_volume_long)
    
    return heuristics_matrix
