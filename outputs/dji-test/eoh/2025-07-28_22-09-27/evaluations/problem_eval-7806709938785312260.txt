import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day high and low close prices
    df['high_close_20'] = df['close'].rolling(window=20).max()
    df['low_close_20'] = df['close'].rolling(window=20).min()
    
    # Compute the difference between 20-day high and low close
    df['price_diff_20'] = df['high_close_20'] - df['low_close_20']
    
    # Calculate the 20-day average volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Determine the ratio of the price difference to the average volume
    df['price_vol_ratio'] = df['price_diff_20'] / df['avg_volume_20']
    
    # Find the 25-day minimum close price
    df['min_close_25'] = df['close'].rolling(window=25).min()
    
    # Compute the ratio of the current close price to its 25-day minimum
    df['close_min_ratio'] = df['close'] / df['min_close_25']
    
    # Combine the two ratios to create the heuristic factor
    heuristics_matrix = df['price_vol_ratio'] + df['close_min_ratio']
    
    return heuristics_matrix
