def heuristics_v2(df):
    adj_avg_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    sma_adj_avg_price = adj_avg_price.rolling(window=5).mean()
    volume_sma_ratio = df['volume'] / sma_adj_avg_price
    heuristics_matrix = volume_sma_ratio.ewm(span=30, adjust=False).mean()
    return heuristics_matrix
