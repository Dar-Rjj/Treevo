def heuristics_v2(df):
    # Calculate the 10-day relative strength
    df['close_10day_pct'] = df['close'].pct_change(10)
    # Calculate the percentage change in volume
    df['volume_pct_change'] = df['volume'].pct_change(1)
    # Combine the two factors
    heuristics_matrix = (df['close_10day_pct'] + df['volume_pct_change']) / 2
    return heuristics_matrix
