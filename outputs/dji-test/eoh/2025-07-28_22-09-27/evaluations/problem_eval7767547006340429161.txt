def heuristics_v2(df):
    # Calculate exponential moving average of returns with an adjusted span for momentum
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=20, adjust=False).mean()
    
    # Mean reversion factor based on deviation from a 50-day SMA
    sma_close = df['close'].rolling(window=50).mean()
    mean_reversion_factor = (df['close'] - sma_close) / sma_close
    
    # Adjusted liquidity: sum of 90-day volumes divided by close price
    liquidity = (df['volume'].rolling(window=90).sum() / df['close'])
    
    # Heuristics matrix with new weights
    heuristics_matrix = 0.5 * ema_returns - 0.3 * mean_reversion_factor + 0.2 * liquidity
    
    return heuristics_matrix
