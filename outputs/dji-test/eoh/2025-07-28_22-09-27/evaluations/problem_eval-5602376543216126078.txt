import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving average of returns for momentum
    returns = df['close'].pct_change()
    sma_returns = returns.rolling(window=20).mean()
    
    # Calculate standard deviation of returns for volatility
    std_dev = returns.rolling(window=20).std()
    
    # Calculate liquidity (volume divided by close price, then averaged over 10 days)
    liquidity = (df['volume'] / df['close']).rolling(window=10).mean()
    
    # Combine factors into a heuristics matrix using a weighted sum with dynamic weights
    heuristics_matrix = (sma_returns * 0.5 + std_dev * 0.3 + liquidity * 0.2)
    
    return heuristics_matrix
