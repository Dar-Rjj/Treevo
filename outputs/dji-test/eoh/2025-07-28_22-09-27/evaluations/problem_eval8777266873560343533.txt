import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving average of returns for momentum
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=20, adjust=False).mean()
    
    # Calculate average true range for volatility
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=20).mean()
    
    # Calculate liquidity (volume divided by close price, then averaged over 20 days)
    liquidity = (df['volume'] / df['close']).rolling(window=20).mean()
    
    # Combine factors into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.5 * ema_returns + 0.3 * atr + 0.2 * liquidity
    
    return heuristics_matrix
