import pandas as pd

def heuristics_v2(df):
    # Calculate the price change
    delta = df['close'].diff()
    
    # Separate gains and losses
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    # Calculate the 14-period RSI
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the price range over 14 days
    price_range = (df['high'] - df['low']).rolling(window=14).sum()
    
    # Combine the RSI and price range into a heuristics matrix using a simple weighted sum
    heuristics_matrix = 0.5 * rsi + 0.5 * price_range
    
    return heuristics_matrix
