def heuristics_v2(df):
    highest_high = df['high'].rolling(window=50).max()
    lowest_low = df['low'].rolling(window=50).min()
    close_to_highest_high_ratio = df['close'] / highest_high
    lowest_low_to_close_ratio = lowest_low / df['close']
    composite_factor = close_to_highest_high_ratio + lowest_low_to_close_ratio
    heuristics_matrix = composite_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
