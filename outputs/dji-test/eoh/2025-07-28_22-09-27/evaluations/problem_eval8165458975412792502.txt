def heuristics_v2(df):
    weighted_avg_price = (df['high'] + df['low'] + 2*df['close']) / 4
    volume_weighted_ratio = df['volume'] / weighted_avg_price
    heuristics_matrix_ema = volume_weighted_ratio.ewm(span=20, adjust=False).mean()
    heuristics_matrix_std = volume_weighted_ratio.rolling(window=5).std()
    heuristics_matrix = heuristics_matrix_ema + heuristics_matrix_std
    return heuristics_matrix
