import pandas as pd

def heuristics_v2(df):
    # Calculate a custom weighted sum of features (example: 3*volume + high - low)
    weighted_sum = 3*df['volume'] + df['high'] - df['low']
    # Apply a 14-day exponential moving average to smooth the series and potentially highlight trends
    heuristics_matrix = weighted_sum.ewm(span=14).mean()
    return heuristics_matrix
