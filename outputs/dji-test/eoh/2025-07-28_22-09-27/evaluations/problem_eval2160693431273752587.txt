import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving average of returns for adjusted momentum
    returns = df['close'].pct_change()
    ema_returns = returns.ewm(span=10, adjust=False).mean()
    
    # Calculate mean reversion factor as the difference between close price and 30-day SMA
    sma_close = df['close'].rolling(window=30).mean()
    mean_reversion_factor = (df['close'] - sma_close) / sma_close
    
    # Adjusted liquidity calculation: sum of volumes over 60 days divided by close price
    liquidity = (df['volume'].rolling(window=60).sum() / df['close'])
    
    # Combine factors into a heuristics matrix using a weighted sum
    heuristics_matrix = 0.4 * ema_returns - 0.4 * mean_reversion_factor + 0.2 * liquidity
    
    return heuristics_matrix
