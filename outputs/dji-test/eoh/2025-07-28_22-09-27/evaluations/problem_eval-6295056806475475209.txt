def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for momentum
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the Money Flow Index (MFI) for buying and selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine ROC and MFI into a heuristics matrix using a simple sum
    heuristics_matrix = roc + mfi
    
    # Apply a 20-day Simple Moving Average (SMA) to smooth the heuristics matrix
    heuristics_matrix = heuristics_matrix.rolling(window=20).mean()
    
    return heuristics_matrix
