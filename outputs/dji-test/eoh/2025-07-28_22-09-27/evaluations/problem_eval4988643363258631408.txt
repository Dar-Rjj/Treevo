def heuristics_v2(df):
    # Trend-following factor: Difference between 20-day and 50-day SMA of closing prices
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    trend_factor = sma_20 - sma_50
    
    # Volatility-adjusted liquidity: Sum of 90-day volumes divided by the 30-day standard deviation of returns
    daily_returns = df['close'].pct_change()
    vol_adj_liquidity = (df['volume'].rolling(window=90).sum()) / daily_returns.rolling(window=30).std()
    
    # Heuristics matrix combining trend-following and volatility-adjusted liquidity
    heuristics_matrix = trend_factor + vol_adj_liquidity
    
    return heuristics_matrix
