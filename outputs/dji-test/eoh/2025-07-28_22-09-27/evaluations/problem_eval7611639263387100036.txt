import pandas as pd

def heuristics_v2(df):
    open_close_ratio = df['open'] / df['close']
    sma_volume_5 = df['volume'].rolling(window=5).mean()
    sma_volume_20 = df['volume'].rolling(window=20).mean()
    heuristics_matrix = open_close_ratio * (sma_volume_5 - sma_volume_20)
    return heuristics_matrix
