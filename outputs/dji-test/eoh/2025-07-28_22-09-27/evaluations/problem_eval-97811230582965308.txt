import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate price volatility as standard deviation of daily returns over the last 30 days
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate volume momentum as the rate of change in volume over the last 10 days
    df['volume_momentum'] = df['volume'].pct_change(periods=10)
    
    # Calculate 50-day moving average
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the percentage difference between the current close price and its 50-day moving average
    df['price_vs_ma'] = (df['close'] - df['ma_50']) / df['ma_50']
    
    # Combine the three factors into one using equal weights
    df['heuristics_factor'] = (df['volatility'] + df['volume_momentum'] + df['price_vs_ma']) / 3
    
    heuristics_matrix = df['heuristics_factor'].copy()
    
    return heuristics_matrix
