def heuristics_v2(df):
    # Momentum factor: 20-day rate of change of close price
    momentum_factor = df['close'].pct_change(periods=20)
    
    # Volatility factor: 50-day standard deviation of daily returns
    returns = df['close'].pct_change()
    volatility_factor = returns.rolling(window=50).std()
    
    # Liquidity measure: 90-day cumulative volume
    liquidity_measure = df['volume'].rolling(window=90).sum()
    
    # Heuristics matrix combining momentum, volatility, and liquidity
    heuristics_matrix = 0.6 * momentum_factor - 0.3 * volatility_factor + 0.1 * liquidity_measure
    
    return heuristics_matrix
