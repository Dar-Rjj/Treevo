def heuristics_v2(df):
    # Calculate the average daily return over the last 10 days
    short_term_returns = df['close'].pct_change().rolling(window=10).mean()
    
    # Calculate the average daily volume over the last 10 days
    short_term_volume = df['volume'].rolling(window=10).mean()
    
    # Combine the short-term return and volume into a heuristics matrix
    heuristics_matrix = short_term_returns * short_term_volume
    
    return heuristics_matrix
