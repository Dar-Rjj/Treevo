def heuristics_v2(df):
    # Calculate the price volatility as the standard deviation of daily returns over a 30-day window
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=30).std()
    
    # Calculate the trading volume momentum as the 30-day EMA of the volume
    volume_momentum = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Combined factor: ratio of volatility to volume momentum
    combined_factor = volatility / volume_momentum
    
    # Apply a 30-day EMA to the combined factor
    heuristics_matrix = combined_factor.ewm(span=30, adjust=False).mean()
    
    return heuristics_matrix
