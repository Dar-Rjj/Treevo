def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Average True Range (ATR) over a 14-day period
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine RSI and ATR
    combined_factor = rsi + atr
    
    # Apply a 30-day EMA to the combined factor
    heuristics_matrix = combined_factor.ewm(span=30, adjust=False).mean()
    
    return heuristics_matrix
