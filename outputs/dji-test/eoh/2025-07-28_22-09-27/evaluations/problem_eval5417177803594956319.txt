def heuristics_v2(df):
    avg_high = df['high'].rolling(window=50).mean()
    avg_low = df['low'].rolling(window=50).mean()
    close_to_avg_high_ratio = df['close'] / avg_high
    avg_low_to_close_ratio = avg_low / df['close']
    composite_factor = avg_low_to_close_ratio - close_to_avg_high_ratio
    heuristics_matrix = composite_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
