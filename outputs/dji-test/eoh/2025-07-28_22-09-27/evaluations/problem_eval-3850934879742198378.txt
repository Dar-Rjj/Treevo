import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day moving averages for High and Low
    ma_high_10 = df['high'].rolling(window=10).mean()
    ma_low_10 = df['low'].rolling(window=10).mean()

    # Calculate 30-day moving average for Volume
    ma_volume_30 = df['volume'].rolling(window=30).mean()

    # Compute the heuristic as the difference between the two MA differences
    heuristics_matrix = (ma_high_10 - ma_low_10) - ma_volume_30
    
    return heuristics_matrix
