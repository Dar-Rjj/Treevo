import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the average price
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the ratio of the high-low range to the average price
    price_ratio = high_low_range / avg_price
    
    # Apply a 10-day Exponential Moving Average (EMA) to the price ratio
    heuristics_matrix = price_ratio.ewm(span=10, adjust=False).mean()
    
    return heuristics_matrix
