def heuristics_v2(df):
    momentum = df['close'] - df['close'].rolling(window=20).min()
    volatility = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    composite_factor = (momentum + volatility) / 2
    heuristics_matrix = composite_factor.rolling(window=10).mean()
    return heuristics_matrix
