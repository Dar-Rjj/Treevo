def heuristics_v2(df):
    def ema_diff(price, fast=15, slow=45):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def sma_volume(volume, periods=30):
        return volume.rolling(window=periods).mean()

    ema_signal = ema_diff(df['close'])
    sma_vol = sma_volume(df['volume'])
    combined_factor = (ema_signal + sma_vol).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=25).mean().rename('heuristic_factor')

    return heuristics_matrix
