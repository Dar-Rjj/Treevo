def heuristics_v2(df):
    def sma_line(price, fast=12, slow=26):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def roc(series, periods=14):
        return series.pct_change(periods=periods)

    sma_signal = sma_line(df['close'])
    roc_close = roc(df['close'], periods=14)
    combined_factor = (sma_signal + roc_close).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=9, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
