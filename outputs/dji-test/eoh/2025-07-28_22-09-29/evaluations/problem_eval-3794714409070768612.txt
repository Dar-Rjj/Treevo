def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for the closing price
    roc_period = 14
    df['roc'] = df['close'].pct_change(periods=roc_period)
    
    # Calculate the historical volatility as the standard deviation of returns
    volatility_window = 30
    df['volatility'] = df['close'].pct_change().rolling(window=volatility_window).std()
    
    # Create a heuristic factor by combining ROC with an inverse volatility adjustment
    heuristics_matrix = (df['roc'] / df['volatility']).rename('heuristic_factor')
    
    return heuristics_matrix
