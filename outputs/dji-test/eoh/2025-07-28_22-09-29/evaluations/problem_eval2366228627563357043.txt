def heuristics_v2(df):
    # Calculate the 12-day and 26-day Exponential Moving Averages (EMA)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Compute the MACD line
    macd = ema_12 - ema_26
    
    # Calculate the standard deviation of closing prices over a 5-day period
    close_std = df['close'].rolling(window=5).std()
    
    # Scale the MACD by the standard deviation of closing prices to get the final heuristic matrix
    heuristics_matrix = macd / close_std
    return heuristics_matrix
