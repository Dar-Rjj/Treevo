def heuristics_v2(df):
    def roc(price, periods=14):
        return (price.diff(periods) / price.shift(periods)) * 100
    
    def adl(high, low, close, volume):
        money_flow_multiplier = ((close - low) - (high - close)) / (high - low)
        money_flow_volume = money_flow_multiplier * volume
        return money_flow_volume.cumsum()
    
    roc_close = roc(df['close'], periods=14)
    adl_series = adl(df['high'], df['low'], df['close'], df['volume'])
    heuristics_matrix = (roc_close + adl_series).rename('heuristic_factor')
    
    return heuristics_matrix
