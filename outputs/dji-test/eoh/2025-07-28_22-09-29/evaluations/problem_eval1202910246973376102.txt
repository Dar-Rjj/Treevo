def heuristics_v2(df):
    def ema_diff(price, fast=10, slow=50):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def rsi_smoothed(price, window=14, smoothing_window=30):
        delta = price.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi.ewm(span=smoothing_window, adjust=False).mean()

    ema_signal = ema_diff(df['close'])
    rsi_smoothed_value = rsi_smoothed(df['close'])
    combined_factor = (ema_signal + rsi_smoothed_value).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=30, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
