def heuristics_v2(df):
    # Calculate the 10-day and 30-day moving average of close prices
    ma_close_10 = df['close'].rolling(window=10).mean()
    ma_close_30 = df['close'].rolling(window=30).mean()
    
    # Compute the momentum as the difference between the two moving averages
    momentum = ma_close_10 - ma_close_30
    
    # Calculate the change in volume over a 5-day period
    vol_change = df['volume'].diff(5)
    
    # Combine the momentum and volume change to create the heuristics matrix
    heuristics_matrix = momentum + vol_change
    
    return heuristics_matrix
