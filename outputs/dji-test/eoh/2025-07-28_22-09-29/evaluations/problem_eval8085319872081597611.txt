def heuristics_v2(df):
    def ema_diff(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def roc(series, periods=20):
        return (series / series.shift(periods) - 1) * 100

    ema_signal = ema_diff(df['close'])
    roc_high = roc(df['high'])
    combined_factor = (ema_signal + roc_high).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18, win_type='gaussian').mean(std=3).rename('heuristic_factor')

    return heuristics_matrix
