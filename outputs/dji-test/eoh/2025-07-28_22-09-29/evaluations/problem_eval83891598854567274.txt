def heuristics_v2(df):
    # Calculate 10-day EMA of the close price
    df['EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 10-day ROC of the volume
    df['ROC'] = df['volume'].pct_change(10)
    
    # Generate the heuristics factor
    df['Heuristic_Factor'] = (df['close'] - df['EMA']) * df['ROC']
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
