def heuristics_v2(df):
    def ema_diff(price, fast=10, slow=50):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def close_std(price, window=20):
        rolling_std = price.rolling(window=window).std()
        return rolling_std / price

    ema_signal = ema_diff(df['close'])
    std_signal = close_std(df['close'])
    combined_factor = (ema_signal + std_signal).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=30).mean().rename('heuristic_factor')

    return heuristics_matrix
