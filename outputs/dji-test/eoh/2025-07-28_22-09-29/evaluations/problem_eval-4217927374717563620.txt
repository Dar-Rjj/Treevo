def heuristics_v2(df):
    # Calculate the 20-day highest high and lowest low
    df['High_20'] = df['high'].rolling(window=20).max()
    df['Low_20'] = df['low'].rolling(window=20).min()
    
    # Calculate the ratio of the highest high to the lowest low
    df['HL_Ratio'] = df['High_20'] / df['Low_20']
    
    # Calculate the 20-day average trading range
    df['Avg_Trading_Range'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Calculate the 10-day rate of change in volume
    df['Volume_ROC'] = df['volume'].pct_change(periods=10)
    
    # Generate the heuristics factor
    df['Heuristic_Factor'] = df['HL_Ratio'] * df['Avg_Trading_Range'] * df['Volume_ROC']
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
