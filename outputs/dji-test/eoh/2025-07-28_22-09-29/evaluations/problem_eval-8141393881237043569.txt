import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day high and low prices
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    
    # Calculate the relative position of the current close price within the 20-day high/low range
    relative_position = (df['close'] - low_20) / (high_20 - low_20)
    
    # Calculate the short-term and long-term momentum
    short_window = 5
    long_window = 20
    short_momentum = df['close'].pct_change().rolling(window=short_window).mean()
    long_momentum = df['close'].pct_change().rolling(window=long_window).mean()
    
    # Calculate the momentum difference
    momentum_diff = short_momentum - long_momentum
    
    # Combine the relative position and momentum difference to create the heuristics matrix
    heuristics_matrix = (relative_position + momentum_diff) / 2
    
    return heuristics_matrix
