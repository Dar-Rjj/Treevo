import pandas as pd

def heuristics_v2(df):
    def macd_line(price, fast=12, slow=26):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def pct_change_9(series, periods=9):
        return series.pct_change(periods=periods)

    macd_signal = macd_line(df['close'])
    pct_change_volume = pct_change_9(df['volume'])
    combined_factor = (macd_signal + pct_change_volume).rename('combined_factor')
    heuristics_matrix = (combined_factor / combined_factor.rolling(window=7).std()).rename('heuristic_factor')

    return heuristics_matrix
