def heuristics_v2(df):
    def macd_line(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def atr(df, periods=14):
        tr = df['high'] - df['low']
        return tr.rolling(window=periods, min_periods=1).mean()

    macd_signal = macd_line(df['close'])
    atr_value = atr(df)
    combined_factor = (macd_signal + atr_value).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18, min_periods=1).mean().rename('heuristic_factor')

    return heuristics_matrix
