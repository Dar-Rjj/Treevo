import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    # Calculate the daily change in volume
    volume_diff = df['volume'].diff(1)
    # Compute a simple weighted combination for demonstration
    combined = 0.7 * price_diff + 0.3 * volume_diff
    # Apply a custom heuristic (e.g., square root for non-linearity)
    heuristics_matrix = combined.apply(lambda x: x**0.5 if x > 0 else -(-x)**0.5)
    return heuristics_matrix
