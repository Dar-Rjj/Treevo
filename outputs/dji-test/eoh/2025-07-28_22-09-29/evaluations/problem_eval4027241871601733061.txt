def heuristics_v2(df):
    def sma_line(price, fast=5, slow=20):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def roc(series, periods=10):
        return series.pct_change(periods=periods)

    sma_signal = sma_line(df['close'])
    roc_high = roc(df['high'])
    combined_factor = (sma_signal + roc_high).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=7).mean().rename('heuristic_factor')

    return heuristics_matrix
