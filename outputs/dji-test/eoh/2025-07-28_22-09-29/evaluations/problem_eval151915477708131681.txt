def heuristics_v2(df):
    short_window = 5
    long_window = 20
    df['short_momentum'] = df['close'].pct_change().rolling(window=short_window).mean()
    df['long_momentum'] = df['close'].pct_change().rolling(window=long_window).mean()
    heuristics_matrix = df['short_momentum'] - df['long_momentum']
    return heuristics_matrix
