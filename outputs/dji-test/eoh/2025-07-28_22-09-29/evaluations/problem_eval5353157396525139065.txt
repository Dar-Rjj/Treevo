import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day moving average of volume
    vol_20_ma = df['volume'].rolling(window=20).mean()
    
    # Compute the liquidity ratio
    liquidity_ratio = df['volume'] / vol_20_ma
    
    # Calculate the rate of change (ROC) over a 10-day period for the closing price
    roc_10 = df['close'].pct_change(10)
    
    # Combine the ROC and liquidity ratio to form the heuristics matrix
    heuristics_matrix = (roc_10 + liquidity_ratio) / 2
    
    return heuristics_matrix
