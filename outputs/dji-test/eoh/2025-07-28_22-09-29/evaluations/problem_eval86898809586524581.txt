def heuristics_v2(df):
    def momentum(price, period=10):
        return price / price.shift(period) - 1

    def rate_of_change(volume, periods=14):
        return (volume - volume.shift(periods)) / volume.shift(periods)

    mom_signal = momentum(df['close'])
    roc_volume = rate_of_change(df['volume'])
    combined_factor = (mom_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20).sum().rename('heuristic_factor')

    return heuristics_matrix
