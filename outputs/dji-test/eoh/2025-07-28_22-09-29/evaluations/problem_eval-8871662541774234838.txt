def heuristics_v2(df):
    def mom_vs_ema(price, span=50):
        ema = price.ewm(span=span, adjust=False).mean()
        return (price - ema) / ema

    def std_ratio(price, window=20):
        rolling_std = price.rolling(window=window).std()
        rolling_mean = price.rolling(window=window).mean()
        return rolling_std / rolling_mean

    mom_signal = mom_vs_ema(df['close'])
    ratio_signal = std_ratio(df['close'])
    combined_factor = (mom_signal + ratio_signal).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=30).mean().rename('heuristic_factor')

    return heuristics_matrix
