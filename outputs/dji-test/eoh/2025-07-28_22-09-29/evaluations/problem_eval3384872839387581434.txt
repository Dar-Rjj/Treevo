def heuristics_v2(df):
    def macd_line(price, fast=12, slow=26):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def tr(df):
        tr = df['high'] - df['low']
        return tr

    macd_signal = macd_line(df['close'])
    tr_value = tr(df)
    combined_factor = (macd_signal + tr_value).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20, min_periods=1).mean().rename('heuristic_factor')

    return heuristics_matrix
