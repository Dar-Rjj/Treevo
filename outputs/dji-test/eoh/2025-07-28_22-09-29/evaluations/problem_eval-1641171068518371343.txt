import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change (ROC) for the closing prices
    roc = df['close'].pct_change()
    
    # Calculate the rolling mean of the volume over a 10-day window
    vol_mean = df['volume'].rolling(window=10).mean()
    
    # Compute the volume trend by comparing current volume to its 10-day average
    vol_trend = df['volume'] / vol_mean
    
    # Combine ROC and volume trend to create the heuristics matrix
    heuristics_matrix = (roc + vol_trend) / 2
    
    return heuristics_matrix
