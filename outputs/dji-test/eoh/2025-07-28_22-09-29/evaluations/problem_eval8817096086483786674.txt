def heuristics_v2(df):
    def sma_diff(price, fast=10, slow=50):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    ema_signal = sma_diff(df['close'])
    roc_low = roc(df['low'])
    combined_factor = (ema_signal + roc_low).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=30, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
