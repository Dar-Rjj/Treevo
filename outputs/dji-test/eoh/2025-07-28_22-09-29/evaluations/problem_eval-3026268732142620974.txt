import pandas as pd

def heuristics_v2(df):
    def roc(series, periods=14):
        return series.pct_change(periods=periods)

    def sma(price, span=20):
        return price.rolling(window=span).mean()

    def std_dev(series, span=10):
        return series.rolling(window=span).std()
    
    roc_close = roc(df['close'])
    sma_close = sma(df['close'])
    std_vol = std_dev(df['volume'])
    
    heuristics_matrix = (roc_close * std_vol / sma_close).rename('heuristic_factor')
    
    return heuristics_matrix
