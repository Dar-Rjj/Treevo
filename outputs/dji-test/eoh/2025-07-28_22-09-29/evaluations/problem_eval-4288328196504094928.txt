import pandas as pd

def heuristics_v2(df):
    n = 10  # Number of days to look back
    df['weighted_close'] = df['close'] * df['volume']
    df['total_volume'] = df['volume'].rolling(window=n).sum()
    df['avg_weighted_close'] = df['weighted_close'].rolling(window=n).sum() / df['total_volume']
    heuristics_matrix = (df['close'] - df['avg_weighted_close']).dropna()
    return heuristics_matrix
