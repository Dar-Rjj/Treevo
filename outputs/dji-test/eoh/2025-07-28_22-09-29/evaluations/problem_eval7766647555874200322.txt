def heuristics_v2(df):
    def sma_line(price, fast=10, slow=30):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    sma_signal = sma_line(df['close'])
    roc_low = roc(df['low'])
    combined_factor = (sma_signal + roc_low).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18).mean().rename('heuristic_factor')

    return heuristics_matrix
