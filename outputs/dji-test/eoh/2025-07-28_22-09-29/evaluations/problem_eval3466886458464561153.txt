import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day high and low prices
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    
    # Calculate the relative position of the current close price within the 20-day high/low range
    relative_position = (df['close'] - low_20) / (high_20 - low_20)
    
    # Calculate the acceleration in volume
    vol_acceleration = df['volume'].diff(2)
    
    # Combine the relative position and volume acceleration to create the heuristics matrix
    heuristics_matrix = (relative_position + vol_acceleration) / 2
    
    return heuristics_matrix
