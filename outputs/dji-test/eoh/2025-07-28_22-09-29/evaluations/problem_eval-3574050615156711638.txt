def heuristics_v2(df):
    # Calculate the 9-day and 21-day Exponential Moving Averages (EMA)
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    
    # Compute the MACD line
    macd = ema_9 - ema_21
    
    # Calculate the standard deviation of closing prices over a 10-day period
    close_std = df['close'].rolling(window=10).std()
    
    # Scale the MACD by the standard deviation of closing prices to get the final heuristic matrix
    heuristics_matrix = macd / close_std
    return heuristics_matrix
