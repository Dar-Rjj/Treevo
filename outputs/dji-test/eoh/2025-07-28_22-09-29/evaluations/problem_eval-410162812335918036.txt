def heuristics_v2(df):
    # Calculate daily returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate the change in trading volume
    df['Volume_Change'] = df['volume'].diff()
    
    # Compute the product of daily return and volume change
    df['Return_Volume_Product'] = df['Daily_Return'] * df['Volume_Change']
    
    # Cumulative sum of the return-volume product
    df['Heuristic_Factor'] = df['Return_Volume_Product'].cumsum()
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
