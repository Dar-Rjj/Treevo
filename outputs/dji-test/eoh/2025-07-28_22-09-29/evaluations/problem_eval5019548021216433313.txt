def heuristics_v2(df):
    def sma_diff(price, fast=5, slow=40):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def momentum(series, periods=15):
        return series.diff(periods)

    sma_signal = sma_diff(df['close'])
    mom_high = momentum(df['high'])
    combined_factor = (sma_signal + mom_high).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20, win_type='gaussian').mean(std=4).rename('heuristic_factor')

    return heuristics_matrix
