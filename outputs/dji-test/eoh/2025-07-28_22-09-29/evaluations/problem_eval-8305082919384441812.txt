def heuristics_v2(df):
    # Calculate the 14-period Rate of Change (ROC)
    df['ROC'] = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100
    
    # Calculate the 14-day cumulative volume
    df['CumulativeVolume'] = df['volume'].rolling(window=14).sum()
    
    # Calculate the 14-day standard deviation of closing prices
    df['STD14'] = df['close'].rolling(window=14).std()
    
    # Combine ROC with the ratio of cumulative volume to 14-day standard deviation
    heuristics_matrix = df['ROC'] * (df['CumulativeVolume'] / df['STD14'])
    return heuristics_matrix
