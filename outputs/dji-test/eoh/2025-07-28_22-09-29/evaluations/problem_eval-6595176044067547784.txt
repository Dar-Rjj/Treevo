import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the standard deviation (volatility) of returns for a given window
    df['volatility'] = df['return'].rolling(window=30).std()
    
    # Calculate the exponential moving average of the daily return
    df['ema_return'] = df['return'].ewm(span=5, adjust=False).mean()
    
    # Calculate the cumulative sum of the absolute volume change
    df['cum_volume_change'] = df['volume'].diff().abs().cumsum()
    
    # Dynamic weights based on the inverse of the rolling volatility
    df['weight'] = 1 / df['volatility']
    
    # Creating a series of heuristic factors by combining EMA of returns and cumulative volume change
    heuristics_matrix = (df['ema_return'] * df['weight']).fillna(0) + (df['cum_volume_change'] * df['weight']).fillna(0)
    
    return heuristics_matrix
