def heuristics_v2(df):
    # Calculate 20-day exponential moving average of the high/low ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['HL_EMA'] = df['High_Low_Ratio'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day exponential moving average of volume
    df['Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Generate the heuristics factor
    df['Heuristic_Factor'] = df['HL_EMA'] * df['Volume_EMA']
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
