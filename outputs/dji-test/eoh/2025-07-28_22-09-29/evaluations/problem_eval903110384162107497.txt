def heuristics_v2(df):
    def ema_diff(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def normalize_volume(volume, periods=14):
        return (volume - volume.rolling(window=periods).mean()) / volume.rolling(window=periods).std()

    ema_signal = ema_diff(df['close'])
    norm_volume = normalize_volume(df['volume'])
    combined_factor = (ema_signal + norm_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20).mean().rename('heuristic_factor')

    return heuristics_matrix
