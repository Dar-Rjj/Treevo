def heuristics_v2(df):
    # Calculate the daily return and change in volume
    df['return'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the standard deviation (volatility) of returns for a given window
    df['volatility'] = df['return'].rolling(window=30).std()
    
    # Adjusted weights for the moving averages based on inverse of volatility
    df['weight'] = 1 / df['volatility']
    
    # Creating a series of heuristic factors by combining weighted moving averages of price and volume changes
    heuristics_matrix = (df['return'].rolling(window=5).mean() * df['weight']).fillna(0) + (df['volume_change'].rolling(window=5).mean() * df['weight']).fillna(0)
    
    return heuristics_matrix
