def heuristics_v2(df):
    def sma_diff(price, short_window=10, long_window=30):
        sma_short = price.rolling(window=short_window).mean()
        sma_long = price.rolling(window=long_window).mean()
        return sma_short - sma_long

    def roc(series, periods=20):
        return (series - series.shift(periods)) / series.shift(periods)

    sma_signal = sma_diff(df['close'])
    roc_volume = roc(df['volume'])
    combined_factor = (sma_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=20, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
