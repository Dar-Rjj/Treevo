def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the standard deviation of closing prices over a 5-day period
    close_std = df['close'].rolling(window=5).std()
    
    # Scale the RSI by the standard deviation of closing prices to get the final heuristic matrix
    heuristics_matrix = rsi / close_std
    return heuristics_matrix
