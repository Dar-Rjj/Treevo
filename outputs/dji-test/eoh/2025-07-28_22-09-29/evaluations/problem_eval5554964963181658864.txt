def heuristics_v2(df):
    def momentum(price, periods=20):
        return price.pct_change(periods)

    def roc(series, periods=14):
        return (series - series.shift(periods)) / series.shift(periods)

    mom_signal = momentum(df['close'])
    roc_volume = roc(df['volume'])
    combined_factor = (mom_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=20, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
