import pandas as pd

def heuristics_v2(df):
    def macd_line(price, fast=12, slow=26):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow
    
    def rsi(series, periods=14):
        delta = series.diff(1)
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window=periods).mean()
        roll_down = down.abs().rolling(window=periods).mean()
        rs = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + rs))
    
    macd_signal = macd_line(df['close'])
    rsi_volume = rsi(df['volume'])
    heuristics_matrix = (macd_signal * rsi_volume).rename('heuristic_factor')
    
    return heuristics_matrix
