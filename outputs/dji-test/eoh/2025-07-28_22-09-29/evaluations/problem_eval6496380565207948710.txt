def heuristics_v2(df):
    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    def pct_change_9(series, periods=9):
        return series.pct_change(periods=periods)

    roc_low = roc(df['low'])
    pct_change_volume = pct_change_9(df['volume'])
    combined_factor = (roc_low + pct_change_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=14).mean().rename('heuristic_factor')

    return heuristics_matrix
