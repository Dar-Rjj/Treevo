def heuristics_v2(df):
    def sma_diff(price, fast=10, slow=30):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast - sma_slow

    def momentum(series, periods=20):
        return series.diff(periods)

    sma_signal = sma_diff(df['close'])
    mom_low = momentum(df['low'])
    combined_factor = (sma_signal + mom_low).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18, win_type='gaussian').mean(std=3).rename('heuristic_factor')

    return heuristics_matrix
