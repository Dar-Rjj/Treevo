import pandas as pd

def heuristics_v2(df):
    def macd_line(price, fast=12, slow=26):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow
    
    def sma_difference(price, short=5, long=20):
        sma_short = price.rolling(window=short).mean()
        sma_long = price.rolling(window=long).mean()
        return sma_short - sma_long
    
    macd_signal = macd_line(df['close'])
    sma_diff = sma_difference(df['close'])
    heuristics_matrix = (macd_signal * sma_diff).rename('heuristic_factor')
    
    return heuristics_matrix
