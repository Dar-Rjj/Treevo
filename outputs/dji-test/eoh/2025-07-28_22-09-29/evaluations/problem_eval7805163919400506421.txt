import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day and 50-day exponential moving averages of the closing price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the ratio of the 10-day to 50-day EMA
    ema_ratio = ema_10 / ema_50
    
    # Calculate 30-day rate of change (ROC) for volume
    roc_volume = df['volume'].pct_change(periods=30)
    
    # Combine the EMA ratio and ROC of volume
    combined_factor = (ema_ratio + roc_volume).rename('combined_factor')
    
    # Smooth the combined factor using a 20-day EMA
    heuristics_matrix = combined_factor.ewm(span=20, adjust=False).mean().rename('heuristic_factor')
    
    return heuristics_matrix
