def heuristics_v2(df):
    def ema_line(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def momentum(series, periods=20):
        return series.shift(periods) - series

    ema_signal = ema_line(df['close'])
    mom_high = momentum(df['high'])
    combined_factor = (ema_signal + mom_high).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18, win_type='gaussian').mean(std=4).rename('heuristic_factor')

    return heuristics_matrix
