def heuristics_v2(df):
    short_window = 5
    long_window = 20
    df['short_ema_momentum'] = df['close'].pct_change().ewm(span=short_window, adjust=False).mean()
    df['long_ema_momentum'] = df['close'].pct_change().ewm(span=long_window, adjust=False).mean()
    heuristics_matrix = df['short_ema_momentum'] - df['long_ema_momentum']
    return heuristics_matrix
