import pandas as pd

def heuristics_v2(df):
    def ema_line(price, fast=15, slow=40):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def roc(series, periods=25):
        return series.pct_change(periods=periods)

    def sma_line(series, window=18):
        return series.rolling(window=window).mean()

    ema_signal = ema_line(df['close'])
    roc_volume = roc(df['volume'])
    combined_factor = (ema_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = sma_line(combined_factor).rename('heuristic_factor')

    return heuristics_matrix
