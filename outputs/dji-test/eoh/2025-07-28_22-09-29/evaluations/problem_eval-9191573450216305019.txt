def heuristics_v2(df):
    vol_window = 30
    change_window = 10
    df['volume_ma'] = df['volume'].rolling(window=vol_window).mean()
    df['vol_ratio'] = df['volume'] / df['volume_ma']
    df['close_change'] = (df['close'] - df.shift(change_window)['close']) / df.shift(change_window)['close']
    heuristics_matrix = df['vol_ratio'] * df['close_change']
    return heuristics_matrix
