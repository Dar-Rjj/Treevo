def heuristics_v2(df):
    def momentum(series, periods=10):
        return series.pct_change(periods)

    def volatility(series, periods=20):
        return series.rolling(window=periods).std()

    high_momentum = momentum(df['high'])
    close_volatility = volatility(df['close'])
    combined_factor = (high_momentum + close_volatility).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=14, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
