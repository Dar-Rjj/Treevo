def heuristics_v2(df):
    def macd_line(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    macd_signal = macd_line(df['close'])
    roc_high = roc(df['high'])
    combined_factor = (macd_signal + roc_high).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=18, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
