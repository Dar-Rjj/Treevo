def heuristics_v2(df):
    short_window = 5
    long_window = 20
    close_prices = df['close']
    volumes = df['volume']
    
    short_mavg_close = close_prices.rolling(window=short_window).mean()
    long_mavg_close = close_prices.rolling(window=long_window).mean()
    short_mavg_volume = volumes.rolling(window=short_window).mean()
    long_mavg_volume = volumes.rolling(window=long_window).mean()
    
    close_momentum = (short_mavg_close - long_mavg_close) / long_mavg_close
    volume_momentum = (short_mavg_volume - long_mavg_volume) / long_mavg_volume
    
    heuristics_matrix = (close_momentum * volume_momentum).dropna()
    
    return heuristics_matrix
