def heuristics_v2(df):
    # Calculate 20-day cumulative return
    df['Cumulative_Return'] = df['close'].pct_change(periods=20)
    
    # Calculate 5-day average true range
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=5).mean()
    
    # Calculate 10-day simple moving average of volume
    df['Volume_SMA'] = df['volume'].rolling(window=10).mean()
    
    # Generate the heuristics factor
    df['Heuristic_Factor'] = (df['Cumulative_Return'] / df['Average_True_Range']) * df['Volume_SMA']
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
