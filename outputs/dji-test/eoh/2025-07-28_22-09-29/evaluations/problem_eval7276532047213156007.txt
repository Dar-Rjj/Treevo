import pandas as pd

def heuristics_v2(df):
    def ema_ratio(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast / ema_slow

    def rate_of_change(series, periods=10):
        return (series / series.shift(periods)) - 1

    ema_signal = ema_ratio(df['close'])
    roc_volume = rate_of_change(df['volume'])
    combined_factor = (ema_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20).mean().rename('heuristic_factor')

    return heuristics_matrix
