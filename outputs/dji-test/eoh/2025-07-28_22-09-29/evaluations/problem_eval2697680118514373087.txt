def heuristics_v2(df):
    def ema_diff(price, fast=7, slow=21):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    def bollinger_width(price, window=10, n_std=2):
        rolling_mean = price.rolling(window=window).mean()
        rolling_std = price.rolling(window=window).std()
        upper_band = rolling_mean + (rolling_std * n_std)
        lower_band = rolling_mean - (rolling_std * n_std)
        return (upper_band - lower_band) / rolling_mean

    ema_signal = ema_diff(df['close'])
    bb_width = bollinger_width(df['close'])
    combined_factor = (ema_signal + bb_width).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=15, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
