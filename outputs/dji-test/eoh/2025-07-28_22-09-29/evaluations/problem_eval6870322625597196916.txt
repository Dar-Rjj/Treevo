def heuristics_v2(df):
    long_window = 20
    short_window = 5
    price_volatility = df['close'].rolling(window=long_window).std()
    avg_volume = df['volume'].rolling(window=short_window).mean()
    heuristics_matrix = (price_volatility / avg_volume).rename('heuristic_factor')
    return heuristics_matrix
