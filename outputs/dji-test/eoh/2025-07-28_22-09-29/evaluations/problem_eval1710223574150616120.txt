def heuristics_v2(df):
    def ema_diff(price, fast=12, slow=26):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast - ema_slow

    ema_signal = ema_diff(df['close'])
    combined_factor = (ema_signal + df['volume']).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=30).median().rename('heuristic_factor')

    return heuristics_matrix
