def heuristics_v2(df):
    def ema_ratio(price, fast=10, slow=30):
        ema_fast = price.ewm(span=fast, adjust=False).mean()
        ema_slow = price.ewm(span=slow, adjust=False).mean()
        return ema_fast / ema_slow

    def adi(high, low, close, volume, window=14):
        money_flow_multiplier = ((close - low) - (high - close)) / (high - low)
        money_flow_volume = money_flow_multiplier * volume
        adi_series = money_flow_volume.rolling(window=window).sum()
        return adi_series

    ema_signal = ema_ratio(df['close'])
    adi_volume = adi(df['high'], df['low'], df['close'], df['volume'])
    combined_factor = (ema_signal + adi_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=20).std().rename('heuristic_factor')

    return heuristics_matrix
