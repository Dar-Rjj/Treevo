def heuristics_v2(df):
    def sma(price, span=30):
        return price.rolling(window=span).mean()

    def macd_line(price, fast=10, slow=30):
        sma_fast = sma(price, fast)
        sma_slow = sma(price, slow)
        return sma_fast - sma_slow

    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    macd_signal = macd_line(df['close'])
    roc_low = roc(df['low'])
    combined_factor = (macd_signal + roc_low).rename('combined_factor')
    heuristics_matrix = combined_factor.rolling(window=18).mean().rename('heuristic_factor')

    return heuristics_matrix
