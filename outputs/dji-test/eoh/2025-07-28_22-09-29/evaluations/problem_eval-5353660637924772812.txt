def heuristics_v2(df):
    def sma_ratio(price, fast=10, slow=30):
        sma_fast = price.rolling(window=fast).mean()
        sma_slow = price.rolling(window=slow).mean()
        return sma_fast / sma_slow

    def roc(series, periods=20):
        return (series / series.shift(periods)) - 1

    sma_signal = sma_ratio(df['close'])
    roc_volume = roc(df['volume'])
    combined_factor = (sma_signal + roc_volume).rename('combined_factor')
    heuristics_matrix = combined_factor.ewm(span=18, adjust=False).mean().rename('heuristic_factor')

    return heuristics_matrix
