def heuristics_v2(df):
    # Calculate the percentage change of the closing price
    df['Close_Pct_Change'] = df['close'].pct_change()
    
    # Calculate the 50-day simple moving average of the high price
    df['SMA_High_50'] = df['high'].rolling(window=50).mean()
    
    # Calculate the 10-day exponential moving average of volume
    df['Volume_EMA_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day simple moving average of the low price
    df['SMA_Low_20'] = df['low'].rolling(window=20).mean()
    
    # Generate the heuristics factor
    df['Heuristic_Factor'] = (df['Close_Pct_Change'] + df['SMA_High_50']) * (df['Volume_EMA_10'] + df['SMA_Low_20'])
    
    heuristics_matrix = df['Heuristic_Factor'].dropna()
    return heuristics_matrix
