import pandas as pd
import numpy as np

def heuristics_v2(df):
    def roc(series, periods=20):
        return series.pct_change(periods=periods)

    def rsi(price, periods=14):
        delta = price.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.ewm(alpha=1/periods, adjust=False).mean()
        roll_down = down.abs().ewm(alpha=1/periods, adjust=False).mean()
        rs = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + rs))

    def weighted_moving_average(series, window=30):
        weights = np.arange(1, window + 1)
        return series.rolling(window=window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    roc_close = roc(df['close'])
    rsi_close = rsi(df['close'])
    combined_factor = (roc_close + rsi_close).rename('combined_factor')
    heuristics_matrix = weighted_moving_average(combined_factor).rename('heuristic_factor')

    return heuristics_matrix
