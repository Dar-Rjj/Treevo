```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Exponential smoothing for the close price
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()

    # Adaptive windows for longer-term context based on ATR
    atr_14 = (df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)).rolling(window=14).mean()
    adaptive_window = (30 + 5 * (atr_14 / atr_14.rolling(window=60).mean())).astype(int)
    df['close_ma_adaptive'] = df['close'].rolling(window=adaptive_window, min_periods=1).mean()

    # Calculate the 14-day RSI with exponential smoothing
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Calculate the 14-day ATR (Average True Range) with exponential smoothing
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = tr.ewm(alpha=1/14, adjust=False).mean()

    # Dynamic volume trend
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ema_5'] + 1e-7)

    # Factor combining the momentum, ATR-normalized volatility, and dynamic volume trend
    factor = (df['close'] - df['close_ema_10']) / df['atr_14'] * df['rsi_14'] * df['volume_relative']

    return factor
```
