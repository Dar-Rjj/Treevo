```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = (df['close'].pct_change()).fillna(0)
    
    # Calculate the 5-day and 20-day moving average of the close price to measure short and long term trends
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine the relative strength by comparing the short-term to long-term moving averages
    df['relative_strength'] = df['5_day_ma'] / df['20_day_ma']
    
    # Calculate the daily range volatility
    df['range_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate volume relative to its 10-period moving average for a smoother normalization
    df['volume_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma'] + 1e-7)
    
    # Combine the momentum, relative strength, and normalized volume for the alpha factor
    alpha_factor = df['daily_return'] * df['relative_strength'] * df['volume_relative'] / (df['range_volatility'] + 1e-7)
    
    # Smooth the alpha factor using an exponential moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_alpha_factor
```
