```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive volatility using a 10-day and 30-day window, with a condition to switch based on market conditions
    df['volatility_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volatility_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    df['adaptive_volatility'] = df['volatility_10'].where(df['close'] > df['close'].rolling(window=10).mean(), df['volatility_30'])

    # Adaptive momentum with a 10-day and 30-day moving average, switching based on the relative position of the close price
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    df['momentum_10'] = df['close'] - df['close_ma_10']
    df['momentum_30'] = df['close'] - df['close_ma_30']
    df['adaptive_momentum'] = df['momentum_10'].where(df['close'] > df['close_ma_10'], df['momentum_30'])

    # Volume adjustment factor using a 20-day rolling mean
    df['volume_adjustment'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Long-term trend using the 200-day moving average
    df['close_ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_trend'] = df['close'] - df['close_ma_200']

    # Advanced smoothing using a double exponential moving average
    df['smoothed_adaptive_momentum'] = df['adaptive_momentum'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()
    df['smoothed_long_term_trend'] = df['long_term_trend'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()

    # RSI calculation
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Dynamic thresholds for adaptive indicators using quantiles
    df['momentum_threshold'] = df['adaptive_momentum'].abs().rolling(window=10).quantile(0.75)
    df['volatility_threshold'] = df['adaptive_volatility'].rolling(window=10).quantile(0.75)
    df['rsi_threshold'] = df['rsi'].rolling(window=10).quantile(0.25)

    # Factor combining smoothed adaptive momentum, adaptive volatility, volume adjustment, smoothed long-term trend, and RSI
    factor = (
        (df['smoothed_adaptive_momentum'] / (df['adaptive_volatility'] + 1e-7)).where(df['adaptive_momentum'].abs() > df['momentum_threshold'], 0) * df['volume_adjustment'] +
        df['smoothed_long_term_trend'].where(df['adaptive_volatility'] > df['volatility_threshold'], 0) +
        (50 - df['rsi']).where(df['rsi'] < df['rsi_threshold'], 0)
    )

    return factor
```
