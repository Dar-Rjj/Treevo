```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Exponential moving averages for close price
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Long-term EMA for context
    df['close_ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_ema_200'] = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the 14-day ATR using exponential smoothing
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].ewm(span=14, adjust=False).mean()
    
    # Adaptive volume trend
    df['volume_ma_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)
    
    # Factor combining momentum, volatility, and relative volume
    factor = (df['close'] - df['close_ema_10']) / df['atr_14'] * df['rsi_14'] * df['volume_relative']
    
    return factor
```
