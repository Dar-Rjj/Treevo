```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average true range (ATR)
    atr = (df[['high', 'low']].diff().abs().max(axis=1).rolling(window=14).mean())
    
    # Calculate the price change relative to ATR
    price_change_to_atr = (df['close'] - df['open']) / (atr + 1e-7)
    
    # Calculate the volume ratio compared to its 20-day moving average
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine the price change to ATR with the volume ratio
    alpha_factor = price_change_to_atr * volume_ratio
    
    # Smooth the factor using exponential moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_alpha_factor
```
