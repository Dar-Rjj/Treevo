```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price (daily return)
    daily_return = df['close'] - df['open']
    
    # Calculate the average true range over a 10-day window
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    avg_true_range = true_range.rolling(window=10).mean()
    
    # Factor: Normalized daily return over the 10-day average true range
    factor = daily_return / (avg_true_range + 1e-7)
    
    return factor
```
