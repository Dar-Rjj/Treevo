```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive volatility with dynamic thresholds
    df['volatility_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volatility_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    df['adaptive_volatility'] = df['volatility_10'].where(df['close'] > df['close'].rolling(window=10).mean(), df['volatility_30'])
    
    # Adaptive momentum with dynamic thresholds
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    df['momentum_10'] = df['close'] - df['close_ma_10']
    df['momentum_30'] = df['close'] - df['close_ma_30']
    df['adaptive_momentum'] = df['momentum_10'].where(df['close'] > df['close_ma_10'], df['momentum_30'])

    # Volume adjustment factor using a 20-day rolling mean
    df['volume_adjustment'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Long-term trend using the 200-day moving average
    df['close_ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_trend'] = df['close'] - df['close_ma_200']

    # Advanced smoothing using double exponential moving averages
    df['smoothed_adaptive_momentum'] = df['adaptive_momentum'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()
    df['smoothed_long_term_trend'] = df['long_term_trend'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()

    # RSI calculation
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # ATR calculation
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()

    # Sentiment and macro trends
    df['sentiment_factor'] = (df['close'] - df['open']) / df['open']
    df['macro_trend'] = df['close'].ewm(span=200, adjust=False).mean()

    # Dynamic weighting of factors
    momentum_weight = (df['smoothed_adaptive_momentum'] / (df['adaptive_volatility'] + 1e-7))
    volume_weight = df['volume_adjustment']
    long_trend_weight = df['smoothed_long_term_trend']
    rsi_weight = (50 - df['rsi']) / (df['atr_14'] + 1e-7)
    sentiment_weight = df['sentiment_factor']
    macro_trend_weight = (df['close'] - df['macro_trend'])

    # Final alpha factor
    factor = (
        momentum_weight * volume_weight +
        long_trend_weight +
        rsi_weight +
        sentiment_weight +
        macro_trend_weight
    )

    return factor
```
