def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day moving average of the close price for a medium-term trend
    df['close_ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 60-day moving average of the close price for a long-term trend
    df['close_ma_60'] = df['close'].rolling(window=60).mean()
    
    # Calculate the relative strength (momentum) of the close price
    df['rsi'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / 
                                 -df['close'].diff(1).clip(upper=0).rolling(window=14).mean())))
    
    # Calculate the Average True Range (ATR) for ATR-normalized volatility
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], df['close'].shift()) - min(x['low'], df['close'].shift()), axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate volume relative to its 10-period moving average
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_10'] + 1e-7)
    
    # Factor combining the multi-period momentum, ATR-normalized volatility, and volume surge
    factor = ((df['close'] - df['close_ma_20']) / df['atr'] * df['rsi'] * df['volume_relative']) + \
             ((df['close_ma_20'] - df['close_ma_60']) / df['atr'])
    
    return factor
