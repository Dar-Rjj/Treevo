```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day moving average of the close price
    df['close_ma_3'] = df['close'].rolling(window=3).mean()
    
    # Calculate the 5-day moving average of the close price
    df['close_ma_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the relative close price compared to its 3-day and 5-day moving averages
    df['close_relative_3'] = df['close'] / (df['close_ma_3'] + 1e-7)
    df['close_relative_5'] = df['close'] / (df['close_ma_5'] + 1e-7)
    
    # Calculate the percentage change in close price
    df['close_change'] = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    
    # Calculate volume relative to its 5-period moving average
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)
    
    # Combine the price momentum, relative close price, and volume surge for alpha factor
    return df['close_change'] * (df['close_relative_3'] + df['close_relative_5']) * df['volume_relative']
```
