```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving averages of the close price for short and long term
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 50-day and 200-day EMA for longer-term context
    df['close_ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_ema_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the relative strength (momentum) of the close price over 14 days
    df['rsi_14'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).ewm(alpha=1/14, adjust=False).mean() / 
                                    -df['close'].diff(1).clip(upper=0).ewm(alpha=1/14, adjust=False).mean())))
    
    # Calculate the 14-day ATR (Average True Range) for volatility using exponential smoothing
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].ewm(alpha=1/14, adjust=False).mean()
    
    # Calculate the volume relative to its 5-period exponential moving average
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ema_5'] + 1e-7)
    
    # Calculate the 10-day and 30-day EMA momentum
    df['momentum_10d'] = df['close'] - df['close_ema_10']
    df['momentum_30d'] = df['close'] - df['close_ema_30']
    
    # Factor combining the momentum, ATR-normalized volatility, RSI, and volume surge
    factor = (df['momentum_10d'] * df['momentum_30d']) / df['atr_14'] * df['rsi_14'] * df['volume_relative']
    
    return factor
```
