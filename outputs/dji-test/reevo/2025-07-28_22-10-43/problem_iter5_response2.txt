```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponentially weighted moving averages (EWMA) of close price for adaptive smoothing
    df['close_ewma_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ewma_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute long-term EWMAs for context
    df['close_ewma_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_ewma_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI) with exponential smoothing
    delta = df['close'].diff(1)
    up = delta.clip(lower=0).ewm(alpha=1/14, adjust=False).mean()
    down = -delta.clip(upper=0).ewm(alpha=1/14, adjust=False).mean()
    rs = up / down
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the 14-day Average True Range (ATR) using exponential smoothing
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].ewm(alpha=1/14, adjust=False).mean()
    
    # Calculate the volume relative to its 5-period exponentially weighted moving average
    df['volume_ewma_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ewma_5'] + 1e-7)
    
    # Factor combining momentum, ATR-normalized volatility, and volume surge
    df['momentum'] = df['close_ewma_10'] - df['close_ewma_30']
    factor = (df['momentum'] / df['atr_14']) * df['rsi_14'] * df['volume_relative']
    
    return factor
```
