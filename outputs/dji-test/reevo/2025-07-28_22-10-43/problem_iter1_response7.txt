```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and the previous close as a percentage of the previous close
    df['close_change'] = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    # Calculate volume relative to its 5-period moving average
    df['volume_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma'] + 1e-7)
    # Combine the price momentum and volume surge for alpha factor
    return df['close_change'] * df['volume_relative']
```
