```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_change'] = df['close'].diff()
    
    # Calculate the 10-day average of close prices
    df['avg_close_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of volume
    df['volatility_vol'] = df['volume'].rolling(window=10).std()
    
    # Create a factor based on the interaction between these two features
    factor = (df['close_change'] / (df['volatility_vol'] + 1e-7)) * df['avg_close_10']
    
    return factor
```
