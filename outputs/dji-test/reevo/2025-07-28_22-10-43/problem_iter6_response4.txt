def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average true range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the dynamic price change relative to ATR
    price_change_to_atr = (df['close'] - df['open']) / (atr + 1e-7)
    
    # Calculate the volume surge relative to its 20-day moving average
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_surge = df['volume'] / (volume_ma_20 + 1e-7)
    
    # Calculate the volatility factor using the standard deviation of returns
    daily_returns = df['close'].pct_change()
    volatility_factor = daily_returns.rolling(window=20).std()
    
    # Combine the price change to ATR, volume surge, and volatility factor
    alpha_factor = price_change_to_atr * volume_surge / (volatility_factor + 1e-7)
    
    # Smooth the factor using exponential moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_alpha_factor
