```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and average of high and low prices
    close_to_avg_price = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Adjust by volume, interpreting that higher volume with a positive close-to-average-difference may indicate stronger momentum
    volume_adjusted_alpha = close_to_avg_price * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Smoothing the factor using exponential moving average to reduce noise
    smoothed_factor = volume_adjusted_alpha.ewm(span=10, adjust=False).mean()
    
    return smoothed_factor
```
