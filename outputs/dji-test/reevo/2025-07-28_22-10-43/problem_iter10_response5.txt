```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Exponential smoothing of the close price for different time horizons
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_ema_200'] = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 14-day RSI (Relative Strength Index) with exponential smoothing
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Calculate the 14-day ATR (Average True Range) with exponential smoothing
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].ewm(alpha=1/14, adjust=False).mean()

    # Calculate the volume relative to its 5-period EMA
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ema_5'] + 1e-7)

    # Dynamic volume trend indicator
    df['volume_trend'] = df['volume'].pct_change().ewm(span=5, adjust=False).mean()

    # Adaptive momentum factor
    df['momentum_10_30'] = df['close_ema_10'] - df['close_ema_30']
    df['momentum_30_50'] = df['close_ema_30'] - df['close_ema_50']
    df['momentum_50_200'] = df['close_ema_50'] - df['close_ema_200']

    # Combine factors: momentum, ATR-normalized volatility, dynamic volume trend, and volume surge
    factor = (
        (df['momentum_10_30'] + df['momentum_30_50'] + df['momentum_50_200']) / 
        (3 * df['atr_14']) * 
        df['rsi_14'] * 
        df['volume_relative'] * 
        (1 + df['volume_trend'])
    )

    # Additional adaptive volatility factor
    df['volatility_10'] = df['close'].rolling(window=10).std()
    df['volatility_30'] = df['close'].rolling(window=30).std()
    df['volatility_50'] = df['close'].rolling(window=50).std()
    df['adaptive_volatility'] = (df['volatility_10'] + df['volatility_30'] + df['volatility_50']) / 3

    # Incorporate adaptive volatility into the alpha factor
    factor *= df['adaptive_volatility']

    return factor
```
