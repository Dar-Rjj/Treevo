```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to capture daily momentum
    daily_momentum = df['close'].diff()
    
    # Calculate the average of the absolute difference between high and low over the last 5 days as a measure of volatility
    avg_volatility_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the trading volume as a percentage of the average 10-day volume to capture unusual activity
    vol_ratio = df['volume'] / df['volume'].rolling(window=10).mean()

    # Combine the factors with weights to form the final factor
    factor = (daily_momentum * 0.4) + (avg_volatility_5d * 0.3) + (vol_ratio * 0.3)
    
    return factor
```
