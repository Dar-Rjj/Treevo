```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Exponential Moving Averages (EMA) for different periods
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_ema_200'] = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 14-day RSI (Relative Strength Index)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Calculate the 14-day ATR (Average True Range)
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()

    # Calculate the volume relative to its 5-period moving average
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)

    # Factor combining the momentum, ATR-normalized volatility, and volume surge
    factor = (df['close'] - df['close_ema_10']) / df['atr_14'] * df['rsi_14'] * df['volume_relative']

    # Incorporate dynamic trends by considering the difference between short-term and long-term EMAs
    df['trend_strength'] = df['close_ema_10'] - df['close_ema_50']
    factor = factor + df['trend_strength']

    # Add market breadth indicator: New Highs - New Lows
    df['new_highs'] = df['high'].rolling(window=20).max() == df['high']
    df['new_lows'] = df['low'].rolling(window=20).min() == df['low']
    df['market_breadth'] = df['new_highs'] - df['new_lows']
    factor = factor + df['market_breadth']

    # Add sentiment indicator: Ratio of Close to 200-day EMA
    df['sentiment_ratio'] = df['close'] / df['close_ema_200']
    factor = factor * df['sentiment_ratio']

    return factor
```
