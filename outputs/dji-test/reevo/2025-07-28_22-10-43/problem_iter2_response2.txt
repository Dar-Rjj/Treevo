```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price (daily return)
    daily_return = df['close'] - df['open']
    
    # Calculate the 10-day moving average of the close price
    avg_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day average true range
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    avg_true_range = true_range.rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of volume
    volatility_vol = df['volume'].rolling(window=10).std()
    
    # Factor: Momentum, normalized daily return over the 10-day average true range, and volume interaction
    factor = (daily_return / (avg_true_range + 1e-7)) * (df['close'] / avg_close_10) * (volatility_vol + 1)
    
    return factor
```
