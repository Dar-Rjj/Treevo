```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices over the last 5 days, 
    # and divide by the average of these differences. This aims to capture volatility.
    df['high_low_diff'] = df['high'] - df['low']
    df['high_low_diff_5d_avg'] = df['high_low_diff'].rolling(window=5).mean()
    df['volatility_factor'] = (df['high_low_diff'] / df['high_low_diff_5d_avg']).shift(1)

    # Calculate the change in close price from the previous day
    df['close_change'] = df['close'].pct_change()

    # Calculate the 10-day momentum
    df['momentum_10'] = df['close'].pct_change(periods=10)

    # Calculate the 30-day mean reversion factor
    df['mean_close_30'] = df['close'].rolling(window=30).mean()
    df['mean_reversion_factor'] = (df['close'] - df['mean_close_30']) / df['close']

    # Combine the factors
    df['combined_factor'] = df['volatility_factor'] * df['close_change'].abs() * df['momentum_10'] * df['mean_reversion_factor']

    # Return the combined factor
    return df['combined_factor']
```
