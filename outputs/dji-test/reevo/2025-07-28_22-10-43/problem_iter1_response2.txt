```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    price_diff = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate the normalized volume, which is the current day's volume divided by the rolling average of the past 10 days' volumes
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    norm_volume = df['volume'] / (avg_volume_10 + 1e-7)
    
    # Factor as a function of the price difference and the normalized volume.
    # The idea is that large deviations from the average price (high-low)/2 with unusually high volume can indicate significant market activity/momentum.
    factor = price_diff * norm_volume
    return factor
```
