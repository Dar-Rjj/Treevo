```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average true range (ATR) for normalization without using a library function
    tr = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x['high'], df['high']) - min(x['low'], df['low']), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the difference between close and open prices, then divide it by the ATR
    price_change = (df['close'] - df['open'])
    alpha_factor = price_change / (atr + 1e-7)
    
    # Consider the volume to weight the factor, interpreting that higher volume with a larger price change may indicate stronger momentum
    volume_weighted_alpha = alpha_factor * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Smoothing the final factor using exponential moving average to reduce noise
    smoothed_factor = volume_weighted_alpha.ewm(span=10, adjust=False).mean()
    
    return smoothed_factor
```
