```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day moving average of the close price for smoothing
    df['close_ma_5'] = df['close'].rolling(window=5).mean()
    df['close_ma_10'] = df['close'].rolling(window=10).mean()

    # Calculate the relative close price compared to its 5-day and 10-day moving averages
    df['close_relative_5'] = df['close'] / (df['close_ma_5'] + 1e-7)
    df['close_relative_10'] = df['close'] / (df['close_ma_10'] + 1e-7)

    # Calculate the percentage change in close price from the previous day
    df['close_change'] = (df['close'].diff() / df['close'].shift(1)).fillna(0)

    # Calculate volume relative to its 5-period moving average
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)

    # Calculate the 5-day momentum
    df['momentum_5d'] = df['close'] - df['close'].shift(5)

    # Calculate the 10-day rate of change (ROC)
    df['roc_10d'] = ((df['close'] - df['close'].shift(10)) / df['close'].shift(10)).fillna(0)

    # Combine the 5-day momentum, 10-day ROC, relative close prices, and relative volume for the alpha factor
    factor = (df['momentum_5d'] * df['roc_10d'] * df['close_relative_5'] * df['close_relative_10'] * df['volume_relative'])

    return factor
```
