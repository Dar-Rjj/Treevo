```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive volatility using a 10-day window for low volatility periods and a 30-day window for high volatility
    df['volatility_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volatility_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    df['adaptive_volatility'] = df['volatility_10'].where(df['volatility_10'] < df['volatility_30'], df['volatility_30'])

    # Calculate momentum with an adaptive window that adjusts based on the 10-day and 30-day moving averages
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    df['momentum_10'] = df['close'] - df['close_ma_10']
    df['momentum_30'] = df['close'] - df['close_ma_30']
    df['adaptive_momentum'] = df['momentum_10'].where(df['close_ma_10'] > df['close_ma_30'], df['momentum_30'])

    # Volume adjustment factor using a 20-day rolling mean
    volume_adjustment = df['volume'] / df['volume'].rolling(window=20).mean()

    # Create a factor that combines adaptive volatility, adaptive momentum, and volume adjustment
    factor = (df['adaptive_momentum'] / (df['adaptive_volatility'] + 1e-7)) * volume_adjustment

    # Apply advanced smoothing using a double exponential moving average
    smoothed_factor = factor.ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()

    return smoothed_factor
```
