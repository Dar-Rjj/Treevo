```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving averages of the close price
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength (momentum) of the close price over 14 days
    df['rsi'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / 
                                 -df['close'].diff(1).clip(upper=0).rolling(window=14).mean())))
    
    # Calculate the 10-day rolling standard deviation for volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day Average True Range (ATR) for normalized volatility
    df['tr'] = df[['high', 'low', 'close']].shift().apply(lambda x: max(x['high'], x['close']) - min(x['low'], x['close']), axis=1)
    df['atr_10'] = df['tr'].rolling(window=10).mean()
    
    # Calculate volume relative to its 5-period moving average
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)
    
    # Factor combining multi-period momentum, ATR-normalized volatility, and volume surge
    factor = ((df['close'] - df['close_ma_10']) / (df['close_ma_30'] - df['close_ma_10'])) * (df['rsi'] / df['atr_10']) * df['volume_relative']
    
    return factor
```
