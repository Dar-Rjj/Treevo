def heuristics_v2(df: pd.DataFrame, sentiment: pd.Series, macro: pd.DataFrame) -> pd.Series:
    # Adaptive volatility using a 10-day and 30-day window for low and high volatility periods
    df['volatility_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volatility_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    df['adaptive_volatility'] = df['volatility_10'].where(df['volatility_10'] < df['volatility_30'], df['volatility_30'])

    # Adaptive momentum with a 10-day and 30-day moving average
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    df['momentum_10'] = df['close'] - df['close_ma_10']
    df['momentum_30'] = df['close'] - df['close_ma_30']
    df['adaptive_momentum'] = df['momentum_10'].where(df['close_ma_10'] > df['close_ma_30'], df['momentum_30'])

    # Volume adjustment factor using a 20-day rolling mean
    df['volume_adjustment'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Long-term trend using the 200-day moving average
    df['close_ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_trend'] = df['close'] - df['close_ma_200']

    # Advanced smoothing using a double exponential moving average
    df['smoothed_adaptive_momentum'] = df['adaptive_momentum'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()
    df['smoothed_long_term_trend'] = df['long_term_trend'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()

    # RSI calculation
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # ATR calculation
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()

    # Integrate sentiment
    df = df.join(sentiment, how='left')
    df['sentiment_score'] = df['sentiment'].fillna(method='ffill').fillna(0)

    # Integrate macroeconomic indicators
    df = df.join(macro, how='left')
    df['macro_score'] = df[macro.columns].fillna(method='ffill').fillna(0).sum(axis=1)

    # Use machine learning for dynamic factor weights
    features = ['smoothed_adaptive_momentum', 'adaptive_volatility', 'volume_adjustment', 'smoothed_long_term_trend', 'rsi', 'atr_14', 'sentiment_score', 'macro_score']
    X = df[features].dropna()
    from sklearn.ensemble import RandomForestRegressor
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X, df['close'].loc[X.index])
    factor_weights = model.feature_importances_
    
    # Apply dynamic factor weights
    factor = (X * factor_weights).sum(axis=1)

    return factor
