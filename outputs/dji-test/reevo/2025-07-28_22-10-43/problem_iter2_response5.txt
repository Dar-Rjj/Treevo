```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the close price
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength (momentum) of the close price
    df['rsi'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / 
                                 -df['close'].diff(1).clip(upper=0).rolling(window=14).mean())))
    
    # Calculate the 10-day rolling standard deviation for volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate volume relative to its 5-period moving average
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_relative'] = df['volume'] / (df['volume_ma_5'] + 1e-7)
    
    # Factor combining the momentum, volatility, and volume surge
    factor = (df['close'] - df['close_ma_10']) / df['volatility'] * df['rsi'] * df['volume_relative']
    
    return factor
```
