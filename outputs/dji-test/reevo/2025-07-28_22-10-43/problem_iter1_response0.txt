```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, then divide it by the average true range (ATR)
    atr = (df[['high', 'low']].diff().abs().max(axis=1).rolling(window=14).mean())
    alpha_factor = (df['close'] - df['open']) / (atr + 1e-7)
    return alpha_factor
```
