```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average true range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the price change relative to ATR
    price_change_to_atr = (df['close'] - df['open']) / (atr + 1e-7)
    
    # Calculate the volume ratio compared to its 20-day moving average
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the volatility factor using the standard deviation of returns
    daily_returns = df['close'].pct_change()
    volatility_factor = daily_returns.rolling(window=20).std()
    
    # Calculate the 50-day and 200-day moving averages for trend analysis
    close_ma_50 = df['close'].rolling(window=50).mean()
    close_ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the long-term momentum using the 200-day moving average
    long_term_momentum = df['close'] - close_ma_200
    
    # Calculate the short-term momentum using the 50-day moving average
    short_term_momentum = df['close'] - close_ma_50
    
    # Calculate the RSI (Relative Strength Index) for momentum
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the price change to ATR, volume ratio, and volatility factor
    alpha_factor = (price_change_to_atr * volume_ratio / (volatility_factor + 1e-7)) * (short_term_momentum + long_term_momentum) * rsi
    
    # Smooth the factor using exponential moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_alpha_factor
```
