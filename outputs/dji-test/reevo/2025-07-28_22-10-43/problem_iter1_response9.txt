```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price (daily return direction)
    daily_return_direction = df['close'] - df['open']
    
    # Calculate the volatility based on the high and low prices
    volatility = df['high'] - df['low']
    
    # Calculate the money flow as a ratio of close to volume
    money_flow = df['close'] / df['volume']
    
    # Combine the daily return direction, volatility, and money flow into an alpha factor
    alpha_factor = daily_return_direction * (1 + 1 / (volatility + 1e-7)) * money_flow
    
    return alpha_factor
```
