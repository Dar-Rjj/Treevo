```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate multi-period adaptive volatility
    df['volatility_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volatility_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    df['volatility_50'] = (df['high'] - df['low']).rolling(window=50).mean()
    df['adaptive_volatility'] = (df['volatility_10'] + df['volatility_30'] + df['volatility_50']) / 3

    # Calculate adaptive momentum with multi-period moving averages
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    df['close_ma_50'] = df['close'].rolling(window=50).mean()
    df['momentum_10'] = df['close'] - df['close_ma_10']
    df['momentum_30'] = df['close'] - df['close_ma_30']
    df['momentum_50'] = df['close'] - df['close_ma_50']
    df['adaptive_momentum'] = (df['momentum_10'] + df['momentum_30'] + df['momentum_50']) / 3

    # Volume adjustment factor using a 20-day rolling mean
    df['volume_adjustment'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Long-term momentum using the 200-day moving average
    df['close_ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = df['close'] - df['close_ma_200']

    # Double exponential smoothing for adaptive and long-term momentum
    df['smoothed_adaptive_momentum'] = df['adaptive_momentum'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()
    df['smoothed_long_term_momentum'] = df['long_term_momentum'].ewm(span=10, adjust=False).mean().ewm(span=5, adjust=False).mean()

    # RSI (Relative Strength Index)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # ATR (Average True Range)
    df['tr'] = pd.concat([(df['high'] - df['low']), abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    df['atr'] = df['tr'].ewm(alpha=1/14, adjust=False).mean()

    # Adaptive trend strength indicator
    df['trend_strength'] = (df['close_ma_200'] - df['close'].shift(200)) / df['close'].shift(200)

    # Create a factor that combines smoothed adaptive momentum, adaptive volatility, volume adjustment, and smoothed long-term momentum
    factor = (
        (df['smoothed_adaptive_momentum'] / (df['adaptive_volatility'] + 1e-7)) * 
        df['volume_adjustment'] + 
        df['smoothed_long_term_momentum'] + 
        (df['rsi']/100) * 
        (1 + df['trend_strength']) - 
        (df['atr']/df['close'])
    )

    return factor
```
