```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day Exponential Moving Averages (EMAs) of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the short-term momentum factor as the difference between the 5-day and 20-day EMAs
    short_momentum_factor = ema_5 - ema_20

    # Calculate the 50-day and 200-day EMAs of the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the long-term momentum factor
    long_momentum_factor = ema_50 - ema_200

    # Calculate the Average True Range (ATR) with a 14-day window
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift()), abs(x['low'] - x['close'].shift())), axis=1)
    atr = tr.rolling(window=14).mean()

    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the trade signal based on the RSI
    trade_signal = (rsi < 30).astype(int) - (rsi > 70).astype(int)

    # Combine all factors into a single alpha factor
    alpha_factor = (short_momentum_factor * long_momentum_factor * atr * trade_signal).rolling(window=5).mean()

    return alpha_factor
```
