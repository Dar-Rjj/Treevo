```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()

    # Calculate the difference between the 5-day and 20-day moving averages as a momentum factor
    momentum_factor = ma_5 - ma_20

    # Calculate the volatility (standard deviation) of the closing price over the last 20 days
    volatility = df['close'].rolling(window=20).std()

    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the trade signal based on the RSI
    trade_signal = (rsi < 30).astype(int) - (rsi > 70).astype(int)

    # Calculate the 50-day and 200-day moving averages of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Calculate the long-term momentum factor
    long_momentum_factor = ma_50 - ma_200

    # Combine all factors into a single alpha factor
    alpha_factor = (momentum_factor * long_momentum_factor * volatility * trade_signal).rolling(window=5).mean()

    return alpha_factor
```
