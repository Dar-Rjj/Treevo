```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in close price from the previous day to current day
    df['close_change'] = df['close'].diff()
    # Calculate the change in volume from the previous day to current day
    df['volume_change'] = df['volume'].diff()
    # Determine if the change in close price is proportional to the change in volume, adding a small constant (1e-7) to avoid division by zero
    factor_values = df['close_change'] / (df['volume_change'] + 1e-7)
    return factor_values
```
