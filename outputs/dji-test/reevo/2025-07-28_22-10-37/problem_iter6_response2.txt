```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of the highest and lowest prices
    avg_high_low = (df['high'] + df['low']) / 2
    diff_close_avg = df['close'] - avg_high_low

    # Calculate the momentum component by looking at the change in volume
    volume_change = df['volume'].pct_change().fillna(0)

    # Calculate the volatility using the average true range
    period = 14
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1).shift(1)
    average_true_range = df['true_range'].rolling(window=period).mean()

    # Calculate the rate of change (ROC) for the closing price over a 10-day period
    roc_10 = df['close'].pct_change(periods=10).fillna(0)

    # Calculate the On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    obv_ma = obv.rolling(window=5).mean()

    # Calculate the Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    mfi_period = 14
    pos_mf_sum = positive_money_flow.rolling(window=mfi_period).sum()
    neg_mf_sum = negative_money_flow.rolling(window=mfi_period).sum()
    mfi = 100 - (100 / (1 + (pos_mf_sum / (neg_mf_sum + 1e-7))))

    # Calculate the Relative Strength Index (RSI) for the close price
    rsi_period = 14
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).fillna(0)
    loss = -delta.where(delta < 0, 0).fillna(0)
    avg_gain = gain.rolling(window=rsi_period).mean()
    avg_loss = loss.rolling(window=rsi_period).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Calculate the Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line

    # Combine the factors with dynamic weighting
    combined_factor = (
        (diff_close_avg * volume_change / (average_true_range + 1e-7)).rolling(window=5).mean() * 0.3 +
        roc_10 * 0.2 +
        (obv - obv_ma) * 0.15 +
        mfi * 0.1 +
        rsi * 0.1 +
        macd_histogram * 0.15
    )

    return combined_factor
```
