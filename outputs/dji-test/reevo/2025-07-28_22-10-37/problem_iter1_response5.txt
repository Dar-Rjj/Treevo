```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for a given day
    daily_range = df['high'] - df['low']
    
    # Calculate the price movement from yesterday's close to today's open
    overnight_movement = abs(df['open'].shift(-1) - df['close'])
    
    # Calculate the ratio of overnight price movement to daily range
    factor = overnight_movement / (daily_range + 1e-7)
    
    return factor
```
