```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price to capture the day's price movement
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low for a more robust measure of central tendency
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Momentum: Calculate the 10-day moving average of the close price
    df['momentum_10d'] = df['close'].rolling(window=10).mean()
    # Calculate the change in momentum from the previous day to the current day
    momentum_change = df['momentum_10d'].diff()
    
    # Volatility: Calculate the 10-day standard deviation of the close price
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Relative Strength: Calculate the relative strength as the ratio of the close price to the 10-day moving average
    df['relative_strength'] = df['close'] / df['momentum_10d']
    
    # Combine all factors: The factor is based on the ratio of the closing to opening price change against the average of high and low, 
    # adjusted by the momentum change and relative strength, and inversely proportional to the volatility.
    factor_values = (close_open_diff / (avg_high_low + 1e-7)) * (1 + momentum_change) * df['relative_strength'] / (df['volatility_10d'] + 1e-7)
    
    return factor_values
```
