```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between the current close price and the close price 10 days ago
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate the average of high and low for a more robust measure of central tendency
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the close to open price change
    close_open_ratio = (df['close'] - df['open']) / (df['open'] + 1e-7)
    
    # Calculate the weighted sum of volume, where the recent volume is considered more important (weight inversely proportional to the day number)
    w_s = lambda n: 1 / (n + 1)  # Weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)
    
    # The factor is based on the combination of momentum, close-open ratio, and weighted volume
    factor_values = (momentum * close_open_ratio) / (weighted_vol + 1e-7)
    
    return factor_values
```
