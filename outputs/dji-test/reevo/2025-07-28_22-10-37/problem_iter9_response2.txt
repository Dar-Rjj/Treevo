def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day Exponential Moving Averages (EMAs) of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the difference between the 5-day and 20-day EMAs as a momentum factor
    short_momentum_factor = ema_5 - ema_20

    # Calculate the 50-day and 200-day EMAs of the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the long-term momentum factor
    long_momentum_factor = ema_50 - ema_200

    # Calculate the Average True Range (ATR) with a 14-day window
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the trade signal based on the RSI
    trade_signal = (rsi < 30).astype(int) - (rsi > 70).astype(int)

    # Calculate the volatility factor using standard deviation of returns
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=20).std()

    # Calculate the volume spike factor
    volume_spike = (df['volume'] > df['volume'].rolling(window=20).mean() * 2).astype(int)

    # Adaptive weights based on recent performance
    short_mom_performance = (short_momentum_factor.shift(1) * df['close'].pct_change()).rolling(window=20).sum()
    long_mom_performance = (long_momentum_factor.shift(1) * df['close'].pct_change()).rolling(window=20).sum()
    atr_performance = (atr.shift(1) * df['close'].pct_change()).rolling(window=20).sum()
    rsi_performance = (trade_signal.shift(1) * df['close'].pct_change()).rolling(window=20).sum()
    vol_performance = (volatility.shift(1) * df['close'].pct_change()).rolling(window=20).sum()
    vol_spike_performance = (volume_spike.shift(1) * df['close'].pct_change()).rolling(window=20).sum()

    total_performance = short_mom_performance + long_mom_performance + atr_performance + rsi_performance + vol_performance + vol_spike_performance

    weight_short_mom = short_mom_performance / total_performance
    weight_long_mom = long_mom_performance / total_performance
    weight_atr = atr_performance / total_performance
    weight_rsi = rsi_performance / total_performance
    weight_vol = vol_performance / total_performance
    weight_vol_spike = vol_spike_performance / total_performance

    # Combine all factors into a single alpha factor with adaptive weighting
    alpha_factor = (short_momentum_factor * weight_short_mom + 
                    long_momentum_factor * weight_long_mom + 
                    atr * weight_atr + 
                    trade_signal * weight_rsi + 
                    volatility * weight_vol + 
                    volume_spike * weight_vol_spike).rolling(window=5).mean()

    return alpha_factor
