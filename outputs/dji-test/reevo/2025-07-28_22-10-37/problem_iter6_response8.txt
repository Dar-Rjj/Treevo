```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the trend strength as the difference between the current close price and the average of the last 10 days' close prices
    trend_strength = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate the intra-day volatility as the range (high - low) divided by the open price
    intra_day_volatility = (df['high'] - df['low']) / (df['open'] + 1e-7)
    
    # Calculate the weighted sum of volume, with recent volume being more important (exponential weighting)
    w_s = lambda n: 0.5 ** n  # Exponential weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)
    
    # Calculate the dynamic weight for each factor
    trend_weight = 1 / (trend_strength.rolling(window=10).std() + 1e-7)
    vol_weight = 1 / (intra_day_volatility.rolling(window=10).std() + 1e-7)
    weighted_vol_weight = 1 / (weighted_vol.rolling(window=10).std() + 1e-7)
    
    # Normalize the weights to sum to 1
    total_weight = trend_weight + vol_weight + weighted_vol_weight
    trend_weight /= total_weight
    vol_weight /= total_weight
    weighted_vol_weight /= total_weight
    
    # Calculate the factor values as a combination of trend strength, intra-day volatility, and weighted volume with dynamic weights
    factor_values = (trend_strength * trend_weight + 
                     intra_day_volatility * vol_weight + 
                     1 / (weighted_vol + 1e-7) * weighted_vol_weight)
    
    return factor_values
```
