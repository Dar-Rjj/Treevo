def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the simple moving average of the closing price over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    sma_deviation = df['close'] - sma_20

    # Calculate the weighted sum of volume, where the recent volume is considered more important (weight inversely proportional to the day number)
    w_s = lambda n: 1 / (n + 1)  # Weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)

    # Final heuristic: the product of the SMA deviation and the inverse of the weighted volume, scaled by the absolute value of today's return
    todays_return_abs = abs(df['close'] - df['open'])
    alpha_factor = (sma_deviation * (1 / (weighted_vol + 1e-7))) * todays_return_abs

    return alpha_factor
