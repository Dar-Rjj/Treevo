```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day Exponential Moving Averages (EMAs) of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the difference between the 5-day and 20-day EMAs as a short-term momentum factor
    short_momentum_factor = ema_5 - ema_20

    # Calculate the 50-day and 200-day EMAs of the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the long-term momentum factor
    long_momentum_factor = ema_50 - ema_200

    # Calculate the Average True Range (ATR) with a 14-day window
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Generate a trade signal based on RSI
    trade_signal = (rsi < 30).astype(int) - (rsi > 70).astype(int)

    # Calculate the Historical Volatility (HV) with a 30-day window
    log_returns = np.log(df['close'] / df['close'].shift(1))
    hv = log_returns.rolling(window=30).std() * np.sqrt(252)

    # Calculate the Adaptive Threshold for HV
    hv_mean = hv.rolling(window=10).mean()
    hv_std = hv.rolling(window=10).std()
    hv_threshold = hv_mean + 2 * hv_std

    # Generate a volatility signal
    volatility_signal = (hv < hv_threshold).astype(int)

    # Integrate macroeconomic indicators (e.g., VIX, Interest Rates, etc.)
    # For simplicity, assume 'vix' and 'interest_rate' are columns in the DataFrame
    vix_signal = (df['vix'] < 20).astype(int)
    interest_rate_signal = (df['interest_rate'] < 2.0).astype(int)

    # Combine all factors into a single alpha factor
    alpha_factor = (short_momentum_factor * long_momentum_factor * atr * trade_signal * 
                    volatility_signal * vix_signal * interest_rate_signal).rolling(window=5).mean()

    return alpha_factor
```
