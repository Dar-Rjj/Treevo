```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of the close price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 14-day Exponential Moving Average (EMA) of the log return as a measure of momentum
    ema_period = 14
    df['momentum_ema'] = df['log_return'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate the 14-day ATR (Average True Range)
    period = 14
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    average_true_range = df['true_range'].rolling(window=period).mean()
    
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the factor as a combination of momentum, volatility, and relative strength
    factor = (diff_close_open / (average_true_range + 1e-7)) * df['momentum_ema']
    
    return factor
```
