def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average true range for a given period to represent volatility
    atr_period = 14
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    average_true_range = df['true_range'].rolling(window=atr_period).mean()
    
    # Calculate the price momentum as the rate of change (ROC) over multiple timeframes
    roc_periods = [5, 10, 20]
    price_momentum_5 = (df['close'] / df['close'].shift(roc_periods[0])) - 1
    price_momentum_10 = (df['close'] / df['close'].shift(roc_periods[1])) - 1
    price_momentum_20 = (df['close'] / df['close'].shift(roc_periods[2])) - 1
    
    # Combine the momentum factors into a single momentum factor
    combined_momentum = (price_momentum_5 + price_momentum_10 + price_momentum_20) / 3
    
    # Consider trading volume patterns by calculating the exponentially weighted sum of volumes
    w_s = lambda n: 0.5 ** n  # Exponential weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)
    
    # Construct the alpha factor incorporating the difference, ATR, combined momentum, and volume
    alpha_factor = (diff_close_open / (average_true_range + 1e-7)) * combined_momentum * (weighted_vol + 1e-7)
    
    return alpha_factor
