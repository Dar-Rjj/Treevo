```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low for a more robust measure of central tendency
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the 5-day moving average of the close price
    ma_5d_close = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day standard deviation of the close price as a measure of volatility
    vol_5d_close = df['close'].rolling(window=5).std()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single signal
    factor_values = (close_open_diff / (avg_high_low + 1e-7)) * (ma_5d_close / (df['close'] + 1e-7)) * (vol_5d_close / (df['close'] + 1e-7)) * (rsi / 50)
    
    return factor_values
```
