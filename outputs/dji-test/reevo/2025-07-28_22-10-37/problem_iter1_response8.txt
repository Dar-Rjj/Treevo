```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a measure of volatility
    daily_range = df['high'] - df['low']
    
    # Calculate the difference between open and close to determine the direction of the day
    daily_movement = df['close'] - df['open']
    
    # Calculate the ratio of daily movement to daily range, which indicates the strength of the day's move
    strength_of_move = (daily_movement / (daily_range + 1e-7))
    
    # Calculate the volume-weighted strength of move
    volume_weighted_strength = strength_of_move * df['volume']
    
    # Create a rolling window of the volume-weighted strength of move to see the trend
    rolling_mean_strength = volume_weighted_strength.rolling(window=5).mean()
    
    return rolling_mean_strength
```
