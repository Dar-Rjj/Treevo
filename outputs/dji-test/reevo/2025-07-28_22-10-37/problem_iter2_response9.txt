```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average true range for a given period to represent volatility
    period = 14
    df['true_range'] = df[['high', 'low', 'close']].shift(1).max(axis=1) - df[['high', 'low', 'close']].shift(1).min(axis=1)
    average_true_range = df['true_range'].rolling(window=period).mean()
    
    # Calculate the price momentum as the rate of change (ROC) over a 10-day period
    roc_period = 10
    price_momentum = (df['close'] / df['close'].shift(roc_period)) - 1
    
    # Consider trading volume patterns by calculating the weighted sum of volumes, with more recent volumes having higher weights
    w_s = lambda n: 1 / (n + 1)  # Weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)
    
    # Construct the alpha factor incorporating the difference, ATR, momentum, and volume
    alpha_factor = (diff_close_open / (average_true_range + 1e-7)) * price_momentum * (weighted_vol + 1e-7)
    
    return alpha_factor
```
