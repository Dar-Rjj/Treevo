def heuristics_v2(df: pd.DataFrame, macro_data: pd.DataFrame, sentiment_scores: pd.Series) -> pd.Series:
    # Calculate the 5-day and 20-day Exponential Moving Averages (EMAs) of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the difference between the 5-day and 20-day EMAs as a short-term momentum factor
    short_momentum_factor = ema_5 - ema_20

    # Calculate the 50-day and 200-day EMAs of the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the difference between the 50-day and 200-day EMAs as a long-term momentum factor
    long_momentum_factor = ema_50 - ema_200

    # Calculate the Average True Range (ATR) with a 14-day window
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the trade signal based on the RSI
    trade_signal = (rsi < 30).astype(int) - (rsi > 70).astype(int)

    # Calculate the volume spike factor
    volume_spike = (df['volume'] > df['volume'].rolling(window=14).mean() * 2).astype(int)

    # Calculate the high-low ratio as a measure of intraday volatility
    high_low_ratio = df['high'] / df['low']

    # Calculate the 20-day standard deviation of returns as a measure of historical volatility
    returns = df['close'].pct_change()
    historical_volatility = returns.rolling(window=20).std()

    # Calculate the trend factor using the 200-day EMA
    trend_factor = (df['close'] - ema_200) / ema_200

    # Calculate adaptive weights for each factor
    short_momentum_weight = (short_momentum_factor.abs() / short_momentum_factor.abs().rolling(window=20).mean()).clip(0, 1)
    long_momentum_weight = (long_momentum_factor.abs() / long_momentum_factor.abs().rolling(window=20).mean()).clip(0, 1)
    atr_weight = (atr / atr.rolling(window=20).mean()).clip(0, 1)
    trade_signal_weight = (trade_signal.abs() / trade_signal.abs().rolling(window=20).mean()).clip(0, 1)
    historical_volatility_weight = (historical_volatility / historical_volatility.rolling(window=20).mean()).clip(0, 1)
    trend_factor_weight = (trend_factor.abs() / trend_factor.abs().rolling(window=20).mean()).clip(0, 1)

    # Combine all factors into a single alpha factor with dynamic weighting
    alpha_factor = (short_momentum_factor * short_momentum_weight +
                    long_momentum_factor * long_momentum_weight +
                    atr * atr_weight +
                    trade_signal * trade_signal_weight +
                    historical_volatility * historical_volatility_weight +
                    trend_factor * trend_factor_weight) * (1 + volume_spike * 0.1) * high_low_ratio

    # Incorporate market anomalies and mean reversion
    mean_reversion_factor = (df['close'] - df['close'].rolling(window=60).mean()) / df['close'].rolling(window=60).std()
    alpha_factor += mean_reversion_factor

    # Add a seasonal adjustment factor
    seasonal_adjustment = df['close'].pct_change(252)  # Annual percentage change
    alpha_factor
