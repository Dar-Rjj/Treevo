```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for a given day (Daily Range)
    daily_range = df['high'] - df['low']
    
    # Calculate the price movement from yesterday's close to today's open (Overnight Movement)
    overnight_movement = abs(df['open'].shift(-1) - df['close'])
    
    # Calculate the ratio of overnight price movement to daily range
    overnight_to_daily_ratio = overnight_movement / (daily_range + 1e-7)
    
    # Calculate the simple moving average of the close price over 5 days (Momentum Indicator)
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the change in close price from the previous day to the current day (Volatility Indicator)
    close_change = df['close'].diff()
    
    # Calculate the 5-day Exponential Moving Average (EMA) of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 20-day Exponential Moving Average (EMA) of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the momentum factor as the difference between the 5-day and 20-day EMAs
    momentum_factor = ema_5 - ema_20
    
    # Calculate the volume surge factor
    volume_surge = df['volume'].pct_change().rolling(window=5).mean()
    
    # Combine the overnight to daily ratio, momentum, close change, and volume surge into a single factor
    factor_values = (overnight_to_daily_ratio + sma_5 + close_change + momentum_factor + volume_surge) / 5
    
    return factor_values
```
