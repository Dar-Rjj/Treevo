```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low for a more robust measure of central tendency
    avg_high_low = (df['high'] + df['low']) / 2
    
    # The factor is based on the ratio of the closing to opening price change against the average of high and low
    factor_values = close_open_diff / (avg_high_low + 1e-7)
    
    return factor_values
```
