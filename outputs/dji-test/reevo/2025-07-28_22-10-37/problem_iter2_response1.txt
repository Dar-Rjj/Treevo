```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of the highest and lowest prices for mean reversion
    avg_high_low = (df['high'] + df['low']) / 2
    mean_reversion = df['close'] - avg_high_low
    
    # Calculate the momentum component by looking at the change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Incorporate volatility using the average true range
    period = 14
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    average_true_range = df['true_range'].rolling(window=period).mean()
    
    # Calculate a short-term momentum factor
    short_momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Combine the factors with adaptive weighting
    combined_factor = (mean_reversion * average_true_range * short_momentum * volume_change).rolling(window=5).mean()
    
    return combined_factor
```
