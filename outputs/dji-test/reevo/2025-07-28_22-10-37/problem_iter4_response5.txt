```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for a given day (Daily Range)
    daily_range = df['high'] - df['low']
    
    # Calculate the price movement from yesterday's close to today's open (Overnight Movement)
    overnight_movement = abs(df['open'].shift(-1) - df['close'])
    
    # Calculate the ratio of overnight price movement to daily range
    overnight_to_daily_ratio = overnight_movement / (daily_range + 1e-7)
    
    # Calculate the simple moving average of the close price over 5 days (Momentum Indicator)
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the change in close price from the previous day to the current day (Volatility Indicator)
    close_change = df['close'].diff()
    
    # Calculate the change in volume from the previous day to the current day (Volume Indicator)
    volume_change = df['volume'].diff()
    
    # Determine if the change in close price is proportional to the change in volume, adding a small constant (1e-7) to avoid division by zero
    price_volume_ratio = close_change / (volume_change + 1e-7)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the overnight to daily ratio, momentum, price-volume ratio, and RSI into a single factor
    # Use adaptive weights based on the standard deviation of each component
    std_overnight_to_daily_ratio = overnight_to_daily_ratio.rolling(window=10).std().fillna(1)
    std_sma_5 = sma_5.rolling(window=10).std().fillna(1)
    std_price_volume_ratio = price_volume_ratio.rolling(window=10).std().fillna(1)
    std_rsi = rsi.rolling(window=10).std().fillna(1)
    
    weight_overnight_to_daily_ratio = 1 / std_overnight_to_daily_ratio
    weight_sma_5 = 1 / std_sma_5
    weight_price_volume_ratio = 1 / std_price_volume_ratio
    weight_rsi = 1 / std_rsi
    
    total_weight = weight_overnight_to_daily_ratio + weight_sma_5 + weight_price_volume_ratio + weight_rsi
    
    factor_values = (
        (overnight_to_daily_ratio * weight_overnight_to_daily_ratio) +
        (sma_5 * weight_sma_5) +
        (price_volume_ratio * weight_price_volume_ratio) +
        (rsi * weight_rsi)
    ) / total_weight
    
    return factor_values
```
