```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    price_diff = df['close'].diff()
    
    # Calculate the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of price difference to EMA volume, aiming to capture volatility adjusted by interest
    alpha_factor = price_diff / (ema_volume + 1e-7)
    
    return alpha_factor
```
