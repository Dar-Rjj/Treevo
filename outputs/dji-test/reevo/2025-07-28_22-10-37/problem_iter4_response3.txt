```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the trend strength as the difference between the current close price and the average of the last 10 days' close prices
    trend_strength = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate the intra-day volatility as the range (high - low) divided by the open price
    intra_day_volatility = (df['high'] - df['low']) / (df['open'] + 1e-7)
    
    # Calculate the weighted sum of volume, with recent volume being more important (exponential weighting)
    w_s = lambda n: 0.5 ** n  # Exponential weighting function
    weighted_vol = 0
    for i in range(10):
        weighted_vol += df['volume'].shift(i) * w_s(i)
    
    # Calculate the factor values as a combination of trend strength, intra-day volatility, and weighted volume
    factor_values = (trend_strength * intra_day_volatility) / (weighted_vol + 1e-7)
    
    return factor_values
```
