```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price, and the difference between close and open price
    price_diff_high_low = df['high'] - df['low']
    price_diff_close_open = (df['close'] - df['open']).abs()
    
    # Weighted volume factor based on the ratio of price difference to volatility
    avg_true_range = df[['high', 'low']].diff().abs().max(axis=1).rolling(window=5).mean()
    volume_factor = df['volume'] * (price_diff_high_low / (avg_true_range + 1e-7))
    
    # Momentum aspect: increase in factor value if the closing is higher than the opening
    momentum_adjustment = ((df['close'] > df['open']) * 1.0).rolling(window=3).sum()
    
    # Combine both aspects for the final heuristic score
    heuristic_score = (price_diff_close_open / (price_diff_high_low + 1e-7)) * volume_factor * momentum_adjustment
    return heuristic_score
```
