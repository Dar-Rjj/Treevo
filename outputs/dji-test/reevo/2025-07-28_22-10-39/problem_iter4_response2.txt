def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price as a measure of volatility
    diff_high_low = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Calculate the 10-day moving average of the factor
    ma_10_factor = factor.rolling(window=10).mean()
    
    # Calculate the 30-day exponential moving average of the factor
    ema_30_factor = factor.ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum of the factor over 10 days
    momentum_10 = factor - factor.shift(10)
    
    # Calculate the 60-day standard deviation of the factor as a measure of volatility
    std_60_factor = factor.rolling(window=60).std()
    
    # Calculate the 90-day rolling geometric mean of the factor
    gm_90_factor = factor.rolling(window=90).apply(lambda x: np.product(1 + x) ** (1/len(x)) - 1)
    
    # Calculate the 120-day rolling median of the factor for robustness against outliers
    median_120_factor = factor.rolling(window=120).median()
    
    # Integrate the moving average, EMA, momentum, standard deviation, geometric mean, and median to create a final factor
    final_factor = (ma_10_factor + ema_30_factor + momentum_10 + std_60_factor + gm_90_factor + median_120_factor) / 6
    
    return final_factor
