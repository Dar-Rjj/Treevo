def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate smoothed momentum by calculating the 10-day Exponential Moving Average (EMA) of the close price
    momentum_ema_10day = df['close'].ewm(span=10, adjust=False).mean()
    
    # Integrate volatility by calculating the 10-day standard deviation of the close price
    volatility_10day = df['close'].rolling(window=10).std()
    
    # Combine the factor with smoothed momentum and volatility
    combined_factor = (factor + momentum_ema_10day) / (volatility_10day + 1e-6)
    
    # Incorporate market depth by considering the ratio of volume to the EMA of volume
    vol_ema_10day = df['volume'].ewm(span=10, adjust=False).mean()
    market_depth = df['volume'] / vol_ema_10day
    
    # Final factor combining all components
    final_factor = combined_factor * market_depth
    
    # Incorporate a long-term momentum component
    momentum_ema_30day = df['close'].ewm(span=30, adjust=False).mean()
    long_term_momentum = momentum_ema_10day - momentum_ema_30day
    
    # Integrate the long-term momentum into the final factor
    final_factor_with_long_momentum = final_factor * long_term_momentum
    
    return final_factor_with_long_momentum
