```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a weighted momentum factor that considers the average of price changes over 5, 10, and 20 day periods,
    # with more weight given to the short-term price changes. This attempts to capture short to medium term trends.
    w1, w2, w3 = 0.5, 0.3, 0.2
    close_prices = df['close']
    volume = df['volume']

    # Momentum calculations
    momentum_5 = (close_prices - close_prices.shift(5)) / close_prices.shift(5)
    momentum_10 = (close_prices - close_prices.shift(10)) / close_prices.shift(10)
    momentum_20 = (close_prices - close_prices.shift(20)) / close_prices.shift(20)

    # Weighted momentum
    weighted_momentum = (w1 * momentum_5 + w2 * momentum_10 + w3 * momentum_20)

    # Volume-weighted momentum
    volume_change = (volume - volume.shift(5)) / volume.shift(5)
    volume_weighted_momentum = weighted_momentum * volume_change

    # Volatility calculation
    daily_returns = close_prices.pct_change()
    volatility = daily_returns.rolling(window=20).std()

    # Blending factors
    factor = (volume_weighted_momentum / (volatility + 1e-7))

    return factor
```
