```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over a 5-day window
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the standard deviation of high prices over a 10-day window
    df['std_high_10d'] = df['high'].rolling(window=10).std()
    
    # Create the alpha factor by combining the above components
    # The idea is to capture momentum (close_diff), liquidity (avg_volume_5d), and volatility (std_high_10d)
    alpha_factor = (df['close_diff'] / df['avg_volume_5d']) * df['std_high_10d']
    
    return alpha_factor
```
