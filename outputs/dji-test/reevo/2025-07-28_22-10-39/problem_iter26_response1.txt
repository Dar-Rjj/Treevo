```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate smoothed momentum by calculating the 10-day Exponential Moving Average (EMA) of the close price
    momentum_ema_10day = df['close'].ewm(span=10, adjust=False).mean()
    
    # Integrate volatility by calculating the 5-day standard deviation of the close price
    volatility_5day = df['close'].rolling(window=5).std()
    
    # Combine the factor with smoothed momentum and volatility
    combined_factor = (factor + momentum_ema_10day) / (volatility_5day + 1e-6)
    
    # Incorporate market depth by adding the ratio of volume to the 5-day EMA of volume
    volume_ema_5day = df['volume'].ewm(span=5, adjust=False).mean()
    market_depth = df['volume'] / volume_ema_5day
    
    # Add a new factor for recent price change
    price_change = df['close'] - df['close'].shift(1)
    
    # Integrate short-term price changes for improved responsiveness
    recent_price_change_5day = df['close'] - df['close'].shift(5)
    
    # Calculate the 5-day average directional index (ADX) for trend strength
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr = tr.rolling(window=5).mean()
    dm_pos = (df['high'] - df['high'].shift(1)).clip(lower=0)
    dm_neg = (df['low'].shift(1) - df['low']).clip(lower=0)
    di_pos = 100 * (dm_pos.rolling(window=5).sum() / atr)
    di_neg = 100 * (dm_neg.rolling(window=5).sum() / atr)
    dx = 100 * (abs(di_pos - di_neg) / (di_pos + di_neg + 1e-6))
    adx = dx.ewm(span=5, adjust=False).mean()
    
    # Adaptive volatility factor: Use the ratio of 3-day to 5-day volatility
    short_term_volatility_3day = df['close'].rolling(window=3).std()
    adaptive_volatility = short_term_volatility_3day / (volatility_5day + 1e-6)
    
    # Dynamic weighting based on the 3-day and 5-day volatility
    dynamic_weight = (volatility_5day + short_term_volatility_3day) / (volatility_5day + 1e-6)
    
    # Final factor combining all elements with dynamic weighting
    final_factor = (
        (combined_factor * market_depth +
         price_change +
         recent_price_change_5day +
         adx +
         volume_ema_5day * adaptive_volatility) * dynamic_weight
    ) / (volatility_5day + short_term_volatility_3day + 1e-6)
    
    return final_factor
```
