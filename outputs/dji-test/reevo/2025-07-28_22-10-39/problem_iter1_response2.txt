```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and the previous close to gauge recent price movement
    df['price_change'] = df['close'].diff().fillna(0)
    
    # Calculate the relative position of the close price within the high-low range of the day
    df['relative_close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the smoothed moving average of the volume
    df['volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the money flow by multiplying the average price (close) with the volume
    df['money_flow'] = df['close'] * df['volume']
    
    # Calculate the smoothed moving average of the money flow
    df['money_flow_ema'] = df['money_flow'].ewm(span=14, adjust=False).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = df['price_change'] * df['relative_close_position'] * (df['money_flow_ema'] / df['volume_ema'])
    
    return alpha_factor
```
