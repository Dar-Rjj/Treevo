def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Calculate the 10-day moving average of the close price
    ma_10_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day exponential moving average of the close price
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Integrate the momentum by subtracting the 10-day MA from the 30-day EMA
    momentum = ema_30_close - ma_10_close
    
    # Calculate the 10-day moving average of the volume
    ma_10_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day exponential moving average of the volume
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Integrate the volume momentum by subtracting the 10-day MA volume from the 30-day EMA volume
    volume_momentum = ema_30_volume - ma_10_volume
    
    # Combine the factor with the momentum and volume momentum to create a more robust alpha factor
    combined_factor = factor * momentum * volume_momentum
    
    return combined_factor
