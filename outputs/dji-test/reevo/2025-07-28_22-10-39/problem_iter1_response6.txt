```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum and rate of change as a measure of trend strength
    df['momentum'] = df['close'].pct_change()
    df['roc_10'] = df['close'].pct_change(10)
    
    # Measure volatility using the ATR (Average True Range) with a 14-day window to capture short term price volatility
    df['tr'] = df[['high', 'low', 'close']].join(df['close'].shift()).max(axis=1) - df[['high', 'low', 'close']].join(df['close'].shift()).min(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Volume confirmation: high volume on days where closing is above its 20-day moving average suggests strong buying pressure
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['vol_sig'] = (df['close'] > df['ma_20']) & (df['volume'] > df['volume'].rolling(window=20).mean())
    
    # Generate alpha factor by combining momentum, ROC, and normalized ATR (as a way to adjust for volatility risk), and volume signal
    df['alpha_factor'] = df['momentum'] * df['roc_10'] / (df['atr_14'] + 1e-7) * df['vol_sig']
    
    return df['alpha_factor']
```
