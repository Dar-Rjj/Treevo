```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the body of the candle, and the upper and lower shadows
    body = df['close'] - df['open']
    upper_shadow = df['high'] - df['close'].where(body > 0, df['open'])
    lower_shadow = df['open'].where(body < 0, df['close']) - df['low']
    
    # Measure momentum by the change in closing prices
    momentum = df['close'].pct_change()
    
    # Indicator for the balance between buying and selling pressure
    balance_pressure = (upper_shadow - lower_shadow) / (upper_shadow + lower_shadow + 1e-7)
    
    # Combine the momentum and balance of pressure with the body size,
    # this aims to capture whether strong momentum is accompanied by a dominant buying or selling pressure
    factor_value = momentum * body.abs() * balance_pressure

    return factor_value
```
