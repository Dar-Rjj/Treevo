def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a 10-day smoothed momentum
    df['smoothed_momentum'] = df['close'].pct_change(periods=10).ewm(span=5, adjust=False).mean()
    
    # Calculate the relative position of the closing price within the day's trading range
    df['relative_position'] = (df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)
    
    # Smooth the relative position with an EMA over 5 days
    df['smoothed_relative_position'] = df['relative_position'].ewm(span=5, adjust=False).mean()
    
    # Calculate the volatility clustering by using the standard deviation of the log returns over a 10-day window
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_clustering'] = df['log_returns'].rolling(window=10).std().fillna(0)
    
    # Combine the smoothed momentum, smoothed relative position, and volatility clustering into a single factor
    df['alpha_factor'] = df['smoothed_momentum'] * df['smoothed_relative_position'] * df['volatility_clustering']
    
    return df['alpha_factor']
