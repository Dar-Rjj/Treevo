```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate smoothed momentum using Exponential Moving Average (EMA)
    momentum_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Integrate volatility using the 10-day standard deviation of the log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility_10day = log_returns.rolling(window=10).std()
    
    # Combine the factor with smoothed momentum and volatility
    combined_factor = (factor + momentum_ema) / (volatility_10day + 1e-6)
    
    return combined_factor
```
