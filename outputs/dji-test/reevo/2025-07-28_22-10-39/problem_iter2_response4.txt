```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price (volatility measure)
    diff_high_low = df['high'] - df['low']
    
    # Calculate the average of open and close prices (price anchor)
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the relative position of the close price within the range of high and low (positioning)
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high (liquidity measure)
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Normalize this factor by dividing by the average of the open and close prices
    normalized_factor = factor / avg_open_close
    
    # Momentum: Calculate the percentage change in close price over the last 10 periods
    momentum = df['close'].pct_change(periods=10).fillna(0)
    
    # Volatility: Calculate the standard deviation of the last 10 days' returns
    volatility = df['close'].pct_change().rolling(window=10).std().fillna(0)
    
    # Liquidity: Calculate the sum of the last 10 days' volumes
    liquidity = df['volume'].rolling(window=10).sum().fillna(0)
    
    # Combine factors using a weighted sum
    alpha_factor = (normalized_factor * 0.4) + (momentum * 0.3) + (1 / volatility * 0.2) + (liquidity * 0.1)
    
    return alpha_factor
```
