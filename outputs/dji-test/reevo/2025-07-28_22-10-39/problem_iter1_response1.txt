```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low, then divide by the volume.
    # This factor aims to capture the momentum relative to the volume.
    avg_high_low = (df['high'] + df['low']) / 2
    return (df['close'] - avg_high_low) / (df['volume'] + 1e-7)
```
