```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5, 10, and 20 day price changes
    close_prices = df['close']
    momentum_5 = (close_prices - close_prices.shift(5)) / close_prices.shift(5)
    momentum_10 = (close_prices - close_prices.shift(10)) / close_prices.shift(10)
    momentum_20 = (close_prices - close_prices.shift(20)) / close_prices.shift(20)
    
    # Weighted momentum factor
    w1, w2, w3 = 0.5, 0.3, 0.2
    weighted_momentum = w1 * momentum_5 + w2 * momentum_10 + w3 * momentum_20
    
    # Volatility factor using standard deviation of the last 20 days' returns
    daily_returns = df['close'].pct_change().fillna(0)
    volatility = daily_returns.rolling(window=20).std()
    
    # Trend strength factor using average directional movement index (ADX)
    high_prices, low_prices = df['high'], df['low']
    adx = talib.ADX(high_prices, low_prices, close_prices, timeperiod=14)
    
    # Volume-weighted price change
    volume_weighted_price_change = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Combine all factors into a single alpha factor
    alpha_factor = weighted_momentum * (1 / volatility) * adx * volume_weighted_price_change
    
    return alpha_factor
```
