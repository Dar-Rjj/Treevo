```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with an adaptive span
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Adaptive window momentum based on the EMA's slope over a 5-day period
    momentum_adaptive = (df['close'] - ema_close.shift(5)) / (ema_close + 1e-6)
    
    # Volatility using the standard deviation of the close price over a 10-day window
    volatility_10day = df['close'].rolling(window=10).std()
    
    # Relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # New factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate the factor with adaptive momentum and normalize by volatility
    combined_factor = (factor + momentum_adaptive) / (volatility_10day + 1e-6)
    
    # Apply exponential smoothing to the combined factor for more robustness with an adaptive span
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
