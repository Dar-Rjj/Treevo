```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price momentum over a 10-day period
    df['momentum'] = df['close'].pct_change(periods=10)
    
    # Calculate relative position of the closing price within the day's trading range, smoothing with EMA over 5 days
    df['relative_position'] = ((df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)).ewm(span=5, adjust=False).mean()
    
    # Combine momentum and relative position into a single factor
    df['alpha_factor'] = df['momentum'] * df['relative_position']
    
    return df['alpha_factor']
```
