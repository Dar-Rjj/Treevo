def heuristics_v2(df: pd.DataFrame, sector_trends: pd.Series, macroeconomic_indicators: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with a span of 10 days
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the adaptive window momentum based on the EMA's slope
    momentum_adaptive = (df['close'] - ema_close) / (ema_close + 1e-6)
    
    # Calculate the long-term volatility using the standard deviation of the close price over a 50-day window
    volatility_50day = df['close'].rolling(window=50).std()
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate the factor with adaptive momentum and normalize by long-term volatility
    combined_factor = (factor + momentum_adaptive) / (volatility_50day + 1e-6)
    
    # Apply exponential smoothing to the combined factor for more robustness
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    # Additional feature: Incorporate market depth
    # Calculate the difference between high and low prices as a measure of market depth
    market_depth = df['high'] - df['low']
    
    # Adjust the final factor by the market depth
    adjusted_factor = smoothed_factor / (market_depth + 1e-6)
    
    # Incorporate multi-period momentum by calculating the 10-day and 30-day price change
    momentum_10day = df['close'].pct_change(periods=10)
    momentum_30day = df['close'].pct_change(periods=30)
    
    # Combine the multi-period momentum and the adjusted factor
    final_factor = (adjusted_factor + momentum_10day + momentum_30day) / 3
    
    # Incorporate volume trends by calculating the 20-day change in volume
    volume_trend = df['volume'].pct_change(periods=20)
    
    # Balance the final factor with the volume trend
    final_factor_with_volume = final_factor * (1 + volume_trend)
    
    # Integrate seasonality by calculating the day of the year and incorporating it into the factor
    day_of_year = df.index.dayofyear
    seasonal_adjustment = np.sin(2 * np.pi * day_of_year / 365)
    final_factor_with_seasonality = final_factor_with_volume * (1 + seasonal_adjustment)
    
    # Integrate sector-specific trends
    sector_trend = sector_trends.reindex(df.index, method='ffill')
    final_factor_with_sector_trend = final_factor_with_seasonality * (1 + sector_trend)
    
    # Integrate macroeconomic indicators
    macro_factors = macroeconomic_indicators.reindex(df.index, method='ffill')
    macro_adjustment = (macro_factors['GDP_Growth'] + macro_factors['Inflation_Rate'] + macro_factors['Interest_Rate']) / 3
    final_factor_with_macroeconomic = final_factor_with_sector_trend * (1 + macro_adjustment)
    
    return final_factor_with_macroeconomic
