```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price
    diff_high_low = df['high'] - df['low']
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / diff_high_low
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate dynamic momentum by calculating the 10-day, 30-day, and 60-day Exponential Moving Averages (EMA) of the close price
    momentum_ema_10day = df['close'].ewm(span=10, adjust=False).mean()
    momentum_ema_30day = df['close'].ewm(span=30, adjust=False).mean()
    momentum_ema_60day = df['close'].ewm(span=60, adjust=False).mean()
    dynamic_momentum = (momentum_ema_10day - momentum_ema_30day) / (momentum_ema_60day + 1e-6)
    
    # Integrate adaptive volatility by calculating the 10-day and 50-day standard deviation of the close price
    volatility_10day = df['close'].rolling(window=10).std()
    volatility_50day = df['close'].rolling(window=50).std()
    adaptive_volatility = (volatility_10day / (volatility_50day + 1e-6))
    
    # Combine the factor with dynamic momentum and adaptive volatility
    combined_factor = (factor + dynamic_momentum) / (adaptive_volatility + 1e-6)
    
    # Incorporate market depth by adding the ratio of volume to the 10-day EMA of volume
    volume_ema_10day = df['volume'].ewm(span=10, adjust=False).mean()
    market_depth = df['volume'] / volume_ema_10day
    
    # Add a new factor for recent price change
    price_change = df['close'] - df['close'].shift(1)
    
    # Integrate short-term price changes for improved responsiveness
    recent_price_change_5day = df['close'] - df['close'].shift(5)
    
    # Calculate the 14-day Average Directional Index (ADX) for sophisticated trend strength
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr = tr.rolling(window=14).mean()
    dm_pos = (df['high'] - df['high'].shift(1)).clip(lower=0)
    dm_neg = (df['low'].shift(1) - df['low']).clip(lower=0)
    di_pos = 100 * (dm_pos.rolling(window=14).sum() / atr)
    di_neg = 100 * (dm_neg.rolling(window=14).sum() / atr)
    dx = 100 * (abs(di_pos - di_neg) / (di_pos + di_neg + 1e-6))
    adx = dx.ewm(span=14, adjust=False).mean()
    
    # Integrate liquidity by using the 10-day EMA of the volume
    liquidity = volume_ema_10day
    
    # Integrate short-term volatility by using the 5-day standard deviation of the close price
    short_term_volatility_5day = df['close'].rolling(window=5).std()
    
    # Final factor combining all elements
    final_factor = (combined_factor * market_depth + 
                    price_change + 
                    0.5 * recent_price_change_5day + 
                    0.3 * adx + 
                    0.2 * liquidity * adaptive_volatility) / (volatility_10day + short_term_volatility_5day + 1e-6)
    
    return final_factor
```
