def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with an adaptive span
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Determine the adaptive window size based on volatility
    volatility_10day = df['close'].rolling(window=10).std()
    adaptive_window = 10 + (volatility_10day > volatility_10day.rolling(window=30).mean()).astype(int)
    
    # Calculate the momentum with the adaptive window
    momentum_adaptive = (df['close'] - df['close'].shift(adaptive_window)) / (df['close'].shift(adaptive_window) + 1e-6)
    
    # Normalize the momentum by the 10-day volatility
    normalized_momentum = momentum_adaptive / (volatility_10day + 1e-6)
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate the factor with normalized adaptive momentum
    combined_factor = (factor + normalized_momentum) / (volatility_10day + 1e-6)
    
    # Apply exponential smoothing to the combined factor for more robustness
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
