```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with a span of 10 days
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the adaptive window momentum based on the EMA's slope
    momentum_adaptive = (df['close'] - ema_close) / (ema_close + 1e-6)
    
    # Calculate dynamic volatility using the standard deviation of the close price over a 5-day and 20-day window
    volatility_5day = df['close'].rolling(window=5).std()
    volatility_20day = df['close'].rolling(window=20).std()
    dynamic_volatility = (volatility_5day + volatility_20day) / 2
    
    # Calculate the relative position of the close price within the range of high and low
    rel_pos_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate a weighted volume where the weight is based on the ratio of close to high
    weighted_volume = df['volume'] * (df['close'] / df['high'])
    
    # Compute a new factor as the product of the relative position and the weighted volume
    factor = rel_pos_close * weighted_volume
    
    # Integrate the factor with adaptive momentum and normalize by dynamic volatility
    combined_factor = (factor + momentum_adaptive) / (dynamic_volatility + 1e-6)
    
    # Apply adaptive smoothing to the combined factor for more robustness
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    # Incorporate market depth by considering the average of open and close prices
    market_depth = (df['open'] + df['close']) / 2
    
    # Adjust the factor with market depth
    final_factor = smoothed_factor * (market_depth / df['close'])
    
    # Use an adaptive window for the final factor to capture recent trends
    adaptive_window = df['close'].rolling(window=5).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    final_factor = final_factor * (1 + adaptive_window)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Integrate RSI into the final factor
    final_factor = final_factor * (rsi / 50)
    
    return final_factor
```
