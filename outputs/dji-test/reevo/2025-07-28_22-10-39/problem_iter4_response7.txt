```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price momentum over a 10-day period
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Calculate relative position of the closing price within the day's trading range, smoothing with EMA over 5 days
    df['relative_position'] = ((df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)).ewm(span=5, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the closing price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponential moving standard deviation of the closing price
    df['std_10'] = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the Z-score of the closing price relative to the 10-day EMA and standard deviation
    df['z_score'] = (df['close'] - df['ema_10']) / (df['std_10'] + 1e-7)
    
    # Combine momentum, relative position, and Z-score into a single factor
    df['alpha_factor'] = df['momentum_10'] * df['relative_position'] * df['z_score']
    
    return df['alpha_factor']
```
