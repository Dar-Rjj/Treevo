```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the 10-day moving average of the close price
    ma_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the distance of the current close price from its 10-day moving average
    dist_from_ma = abs(df['close'] - ma_close_10)
    
    # Calculate the 10-day rolling standard deviation of the close price to measure volatility
    vol_close_10 = df['close'].rolling(window=10).std()
    
    # Calculate the ratio of the distance from the 10-day MA to the 10-day rolling standard deviation
    ratio_dist_vol = dist_from_ma / (vol_close_10 + 1e-7)
    
    # Alpha factor: weighted sum of (difference between close and open, and ratio of distance from MA to volatility)
    alpha_factor = (df['volume'] * diff_close_open) / (ratio_dist_vol + 1e-7)
    
    return alpha_factor
```
