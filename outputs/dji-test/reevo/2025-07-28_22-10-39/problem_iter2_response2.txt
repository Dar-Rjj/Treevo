```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of the close price for trend strength
    ma_close_10 = df['close'].rolling(window=10).mean()
    ma_close_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from its 10-day moving average
    dist_from_ma_10 = abs(df['close'] - ma_close_10)
    
    # Calculate the trend strength as the difference between 10-day and 50-day moving averages
    trend_strength = ma_close_10 - ma_close_50
    
    # Calculate the 10-day standard deviation of the close price for volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the percentage change in volume over a 10-day period
    vol_change = df['volume'].pct_change(periods=10).fillna(0)
    
    # Alpha factor: combination of trend strength, volatility, and volume change
    alpha_factor = (trend_strength * vol_change) / (volatility + 1e-7) * (df['close'] - df['open']) / (dist_from_ma_10 + 1e-7)
    
    return alpha_factor
```
