```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a form of volatility
    price_range = df['high'] - df['low']
    
    # Calculate the return from open to close
    return_open_to_close = df['close'] - df['open']
    
    # Calculate a smoothed version of the return by taking the average of the last 3 days
    return_smoothed = return_open_to_close.rolling(window=3).mean()
    
    # Calculate the momentum as the difference in close prices over the last 5 days
    momentum_5_days = df['close'].diff(periods=5)
    
    # Novel alpha factor: Ratio of smoothed return to recent price range (volatility) and multiplied by momentum
    # This factor reflects the smooth momentum relative to recent volatility, adjusted for longer-term momentum
    alpha_factor = (return_smoothed / price_range) * momentum_5_days
    
    return alpha_factor
```
