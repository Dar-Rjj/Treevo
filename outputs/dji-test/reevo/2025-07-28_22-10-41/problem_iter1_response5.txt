```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the opening price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the factor as the ratio of the price difference to the average price
    # This could indicate the relative movement of the stock during the day
    factor = price_diff / (avg_price + 1e-7)
    
    return factor
```
