```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import numpy as np
    import pandas as pd

    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day exponential moving averages (EMA) of the close price to capture longer-term trends
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of log returns to capture volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the volume-weighted average price (VWAP) to incorporate liquidity
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Additional technical indicators
    df['macd_line'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    df['macd_histogram'] = df['macd_line'] - df['signal_line']
    
    # Calculate dynamic weights for each feature
    weight_log_return = 0.2 + 0.3 * (df['volatility'] / df['volatility'].max())
    weight_ema_diff_50 = 0.1 + 0.2 * (df['rsi'] / 100)
    weight_ema_diff_200 = 0.1 + 0.2 * (1 - (df['rsi'] / 100))
    weight_rsi = 0.1 + 0.2 * (df['volatility'] / df['volatility'].max())
    weight_atr = 0.1 + 0.2 * (df['rsi'] / 100)
    weight_volatility = 0.1 + 0.2 * (1 - (df['rsi'] / 100))
    weight_vwap = 0.1 + 0.2 * (df['volatility'] / df['volatility'].max())
    weight_macd_histogram = 0.1 + 0.2 * (df['rsi'] / 100)
    
    # Combine the features into a single alpha factor with dynamic weights
    factor = (weight_log_return * df['log_return']) + \
             (weight_ema_diff_50 * (df['close'] - df['ema_close_50'])) + \
             (weight_ema_diff_200 * (df['close'] - df['ema_close_200'])) + \
             (weight_rsi * (50 - df['rsi'])) + \
             (weight_atr * (df['atr'] / df['close'])) + \
             (weight_volatility * df['volatility']) + \
             (weight_vwap * (df['close'] - df['vwap'])) + \
             (weight_macd_histogram * df['macd_histogram'])
    
    # Consider market sentiment (example: using MACD histogram as a proxy for sentiment)
    factor += 0.1 * (df['macd_histogram'] / df['macd_histogram'].abs().max())
    
    return factor
```
