```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Calculate the average of the high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in volume compared to the previous day
    pct_change_volume = df['volume'].pct_change().fillna(0)
    
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the volatility over the last 10 days
    volatility_10_days = df['close'].rolling(window=10).std().fillna(0)
    
    # Calculate the momentum over the last 5 days
    momentum_5_days = df['close'].diff(periods=5).fillna(0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (diff_close_open / avg_high_low) * (1 + pct_change_volume) * (1 + daily_return) * (1 + momentum_5_days / volatility_10_days)
    return alpha_factor
```
