```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day adaptive exponential moving averages (AEMA) of the close price to capture longer-term trends
    df['aema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['aema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Adaptive adjustment for EMA
    df['aema_close_50'] = df.apply(lambda x: x['close'] * (2 / (50 + 1)) + x['aema_close_50'].shift(1) * (1 - (2 / (50 + 1))), axis=1)
    df['aema_close_200'] = df.apply(lambda x: x['close'] * (2 / (200 + 1)) + x['aema_close_200'].shift(1) * (1 - (2 / (200 + 1))), axis=1)
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of log returns to capture volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Integrate macroeconomic indicators (e.g., VIX, interest rates)
    # Assuming these are available in the DataFrame
    df['vix'] = df['vix'].fillna(method='ffill')  # Example: VIX data
    df['interest_rate'] = df['interest_rate'].fillna(method='ffill')  # Example: Interest rate data
    
    # Combine the features into a single alpha factor
    factor = (df['log_return'] * 0.5) + (df['close'] - df['aema_close_50']) + (df['close'] - df['aema_close_200']) + (50 - df['rsi']) + (df['atr'] / df['close']) + (df['volatility'] * 0.5) + (df['vix'] * -0.3) + (df['interest_rate'] * 0.2)
    return factor
```
