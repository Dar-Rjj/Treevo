```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and previous day's close
    df['close_diff'] = df['close'].diff()
    # Calculate the 10-day moving average of the high price
    df['ma_high_10'] = df['high'].rolling(window=10).mean()
    # Calculate a feature that captures the trend and momentum
    factor = (df['close_diff'] / df['close']) + (df['close'] - df['ma_high_10'])
    return factor
```
