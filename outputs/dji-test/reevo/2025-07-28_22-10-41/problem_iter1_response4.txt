```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price, which represents the daily momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate the average volume over the last 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Alpha factor: Daily momentum weighted by the ratio of current volume to average volume over 5 days
    alpha_factor = daily_momentum * (df['volume'] / avg_volume)
    
    return alpha_factor
```
