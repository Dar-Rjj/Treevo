```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the simple return for momentum
    df['simple_return'] = df['close'].pct_change()
    
    # Calculate the 50-day and 200-day moving averages of the close price to capture longer-term trends
    df['ma_close_50'] = df['close'].rolling(window=50).mean()
    df['ma_close_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    tr_high_low = df['high'] - df['low']
    tr_high_close_prev = (df['high'] - df['close'].shift(1)).abs()
    tr_low_close_prev = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = tr_high_low.combine(tr_high_close_prev, max).combine(tr_low_close_prev, max)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 50-day exponential moving average (EMA) of volume to capture volume trends
    df['ema_volume_50'] = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Combine the features into a single alpha factor
    factor = (df['simple_return'] * 100) + (df['close'] - df['ma_close_50']) + (df['close'] - df['ma_close_200']) + (50 - df['rsi']) + (df['atr'] / df['close']) + (df['volume'] - df['ema_volume_50'])
    return factor
```
