```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the closing price
    ma_10_close = df['close'].rolling(window=10).mean().fillna(0)
    
    # Calculate the 30-day moving average of the closing price
    ma_30_close = df['close'].rolling(window=30).mean().fillna(0)
    
    # Calculate the relative strength (RS) as the ratio of the 10-day MA to the 30-day MA
    relative_strength = ma_10_close / ma_30_close
    
    # Calculate the 14-day RSI (Relative Strength Index)
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    avg_gain = up.rolling(window=14).mean()
    avg_loss = down.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs)).fillna(0)
    
    # Calculate the 10-day standard deviation of the closing price for volatility
    volatility = df['close'].rolling(window=10).std().fillna(0)
    
    # Calculate the percentage change in volume compared to the previous day
    pct_change_volume = df['volume'].pct_change().fillna(0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (relative_strength * rsi) / (volatility + 1e-7) * (1 + pct_change_volume)
    return alpha_factor
```
