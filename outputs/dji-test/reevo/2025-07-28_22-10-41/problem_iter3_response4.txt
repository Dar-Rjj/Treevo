```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor as the 10-day return
    df['momentum_10'] = df['close'].pct_change(10)
    
    # Calculate the 50-day and 200-day moving averages of the close price to capture longer-term trends
    df['ma_close_50'] = df['close'].rolling(window=50).mean()
    df['ma_close_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 20-day change in volume to incorporate volume changes
    df['volume_change_20'] = df['volume'].pct_change(20)
    
    # Combine the features into a single alpha factor
    factor = (df['momentum_10'] * 0.3) + ((df['close'] - df['ma_close_50']) / df['close'] * 0.2) + ((df['close'] - df['ma_close_200']) / df['close'] * 0.2) + ((50 - df['rsi']) / 50 * 0.15) + (df['atr'] / df['close'] * 0.1) + (df['volume_change_20'] * 0.05)
    return factor
```
