import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestRegressor

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day exponential moving averages (EMA) of the close price to capture longer-term trends
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of log returns to capture volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the volume-weighted average price (VWAP) to incorporate liquidity
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the 14-day Stochastic Oscillator
    df['stoch_k'] = (df['close'] - df['low'].rolling(window=14).min()) / (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())
    
    # Integrate a market sentiment indicator (e.g., 5-day change in volume)
    df['volume_change'] = df['volume'].pct_change(5)
    
    # Dynamic and adaptive weights for RSI, ATR, VWAP, and Stochastic Oscillator
    rsi_weight = 0.5 + 0.5 * (df['rsi'] / 100)
    atr_weight = 0.5 + 0.5 * (df['atr'] / df['close'])
    vwap_weight = 0.5 + 0.5 * ((df['close'] - df['vwap']) / df['close'])
    stoch_weight = 0.5 + 0.5 * (df['stoch_k'] - 0.5)
    volume_weight = 0.5 + 0.5 * (df['volume_change'] / 2)
    
    # Combine the features into a single alpha factor with balanced and adaptive weights
    factor = (
        (df['log_return'] * 0.1) +
        (df['close'] - df['ema_close_50']) * 0.1 * (1 - rsi_weight) +
        (df['close'] - df['ema_close_200']) * 0.1 * (1 - atr_weight) +
        (50 - df['rsi']) * 0.1 * rsi_weight +
        (df['atr'] / df['close']) * 0.1 * atr_weight +
        (df['volatility'] * 0.1) * (1 - vwap_weight) +
        (df['close'] - df['vwap']) * 0.1 * vwap_weight +
        (df['stoch_k'] - 0.5) * 0.1 * stoch_weight +
        (df['volume_change'] * 0.1) * volume_weight
    )
    
    # Use machine learning for feature selection
    X = df[['log_return', 'ema_close_50', 'ema_close_200', 'rsi', 'atr', 'volatility', 'vwap', 'stoch_k', 'volume_change']]
    y = df['close'].pct_change().shift(-1)  # Future returns
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X, y)
    feature_importances = model.feature_importances
