```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import numpy as np
    import pandas as pd

    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day exponential moving averages (EMA) of the close price to capture longer-term trends
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of log returns to capture volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the Chaikin Money Flow (CMF) to incorporate volume and money flow
    mfv = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    mfv = mfv * df['volume']
    df['cmf'] = mfv.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate the 20-day simple moving average (SMA) of the close price to capture short-term trend
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day Hull Moving Average (HMA) to capture a smoother short-term trend
    df['hma'] = df['close'].rolling(window=20).apply(lambda x: (x[10:].mean() + x.mean()) / 2, raw=True)
    
    # Calculate the 20-day Commodity Channel Index (CCI) to capture overbought or oversold conditions
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    sma_typical_price = typical_price.rolling(window=20).mean()
    mean_dev = typical_price.rolling(window=20).apply(lambda x: np.fabs(x - x.mean()).mean(), raw=True)
    df['cci'] = (typical_price - sma_typical_price) / (0.015 * mean_dev)
    
    # Combine the features into a single alpha factor with different weights
    factor = (df['log_return'] * 0.3) + \
             (df['close'] - df['ema_close_50']) * 0.2 + \
             (df['close'] - df['ema_close_200']) * 0.2 + \
             (50 - df['rsi']) * 0.1 + \
             (df['atr'] / df['close']) * 0.1 + \
             (df['volatility'] * 0.3) + \
             (df['cmf'] * 0.2) + \
             (df['close'] - df['sma_close_20']) * 0.1 + \
             (df['close'] - df['hma']) * 0.05 + \
             (df['cci'] * 0.05)
    
    return factor
```
