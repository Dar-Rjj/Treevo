```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price velocity, which is the difference in price over a period divided by the volume
    price_velocity = (df['close'].diff() / df['volume']).fillna(0)
    
    # Calculate the moving average of the price velocity over 10 days
    ma_price_velocity = price_velocity.rolling(window=10).mean().fillna(0)
    
    # Calculate the price action indicator as the standard deviation of high and low prices
    price_action = (df['high'] - df['low']).rolling(window=5).std().fillna(0)
    
    # Alpha factor: Moving Average of Price Velocity divided by Price Action
    alpha_factor = ma_price_velocity / (price_action + 1e-7)
    
    return alpha_factor
```
