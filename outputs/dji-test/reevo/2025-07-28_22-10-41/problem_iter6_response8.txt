```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and previous day's close for momentum
    df['momentum'] = df['close'].diff().fillna(0)
    
    # Calculate the 50-day and 200-day moving averages of the close price to capture longer-term trends
    df['ma_50'] = df['close'].rolling(window=50).mean().fillna(0)
    df['ma_200'] = df['close'].rolling(window=200).mean().fillna(0)
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean().fillna(0)
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean().fillna(0)
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean().fillna(0)
    
    # Calculate the 100-day moving average of volume to capture volume trends
    df['ma_volume_100'] = df['volume'].rolling(window=100).mean().fillna(0)
    
    # Calculate the 30-day standard deviation of the close price to capture recent price volatility
    df['std_close_30'] = df['close'].rolling(window=30).std().fillna(0)
    
    # Combine the features into a single alpha factor
    factor = (df['momentum'] / df['close']) * 0.2 + \
             (df['close'] - df['ma_50']) * 0.2 + \
             (df['close'] - df['ma_200']) * 0.2 + \
             (50 - df['rsi']) * 0.1 + \
             (df['atr'] / df['close']) * 0.15 + \
             (df['volume'] / df['ma_volume_100']) * 0.1 + \
             (df['std_close_30'] / df['close']) * 0.05
    
    return factor
```
