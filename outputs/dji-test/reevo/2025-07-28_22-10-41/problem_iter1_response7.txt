```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    df['price_change'] = df['close'] - df['open']
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['range'] = (df['high'] - df['low']) / df['close'].shift(1)
    alpha_factor = (df['price_change'] * df['volume_ratio'] * df['range']).ewm(span=10, adjust=False).mean()
    return alpha_factor
```
