```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    df['price_diff'] = df['close'].diff()
    # Calculate the simple moving average of the last 5 days' volume
    df['volume_sma_5'] = df['volume'].rolling(window=5).mean()
    # The heuristic is the ratio of price change to the 5-day volume SMA,
    # which could indicate the relationship between price movement and the relative trading activity.
    factor_values = df['price_diff'] / (df['volume_sma_5'] + 1e-7)
    return factor_values
```
