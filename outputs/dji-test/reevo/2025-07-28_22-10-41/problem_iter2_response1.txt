def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of the high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in volume compared to the previous day
    pct_change_volume = df['volume'].pct_change().fillna(0)
    
    # Calculate the Z-score of the daily momentum
    z_score_momentum = (diff_close_open - diff_close_open.rolling(window=20).mean()) / diff_close_open.rolling(window=20).std()
    
    # Calculate the Z-score of the volume change
    z_score_volume = (pct_change_volume - pct_change_volume.rolling(window=20).mean()) / pct_change_volume.rolling(window=20).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (z_score_momentum * z_score_volume) * (1 + pct_change_volume)
    return alpha_factor
