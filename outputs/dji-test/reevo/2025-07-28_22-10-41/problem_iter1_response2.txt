```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the simple moving average of the last 5 days
    df['sma_5'] = df['close'].rolling(window=5).mean()
    alpha_factor = (df['close'] - df['sma_5']) / (df['high'] - df['low'] + 1e-7)
    return alpha_factor
```
