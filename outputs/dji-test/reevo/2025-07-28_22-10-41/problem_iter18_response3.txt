```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day exponential moving averages (EMA) of the close price to capture trends
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the volume-weighted average price (VWAP) to incorporate volume
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Define dynamic weights for each feature based on their recent performance
    df['log_return_perf'] = df['log_return'].rolling(window=10).mean().abs()
    df['ema_50_perf'] = (df['close'] - df['ema_close_50']).rolling(window=10).mean().abs()
    df['ema_200_perf'] = (df['close'] - df['ema_close_200']).rolling(window=10).mean().abs()
    df['rsi_perf'] = (50 - df['rsi']).rolling(window=10).mean().abs()
    df['vwap_perf'] = (df['close'] - df['vwap']).rolling(window=10).mean().abs()
    
    total_perf = df['log_return_perf'] + df['ema_50_perf'] + df['ema_200_perf'] + df['rsi_perf'] + df['vwap_perf']
    
    log_return_weight = df['log_return_perf'] / total_perf
    ema_50_weight = df['ema_50_perf'] / total_perf
    ema_200_weight = df['ema_200_perf'] / total_perf
    rsi_weight = df['rsi_perf'] / total_perf
    vwap_weight = df['vwap_perf'] / total_perf
    
    # Add a new feature: Money Flow Index (MFI) to incorporate both price and volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_14 = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_14 = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_14 / negative_money_flow_14
    df['mfi'] = 100 - (100 / (1 + money_ratio))
    
    # Define dynamic weight for MFI
    df['mfi_perf'] = (50 - df['mfi']).rolling(window=10).mean().abs()
    total_perf += df['mfi_perf']
    mfi_weight = df['mfi_perf'] / total_perf
    
    # Combine the features into a single alpha factor with dynamic weights
    factor = (
        (df['log_return'] * log_return_weight) +
        ((df['close'] - df['ema_close_50']) * ema_50_weight) +
        ((df['close'] - df['ema_close_200']) * ema_200_weight) +
        ((50 - df['rsi']) * rsi_weight) +
        ((df['close'] - df['vwap']) * vwap_weight) +
        ((50 - df['mfi']) * mfi_weight)
    )
    return factor
```
