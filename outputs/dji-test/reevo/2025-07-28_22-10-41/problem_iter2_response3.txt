```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the opening price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of the high and low prices for a more stable mean
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the price difference to the average price
    price_ratio = price_diff / (avg_price + 1e-7)
    
    # Calculate the volume relative to the previous day's volume
    volume_ratio = df['volume'] / (df['volume'].shift(1) + 1e-7)
    
    # Combine the price ratio and volume ratio to form the alpha factor
    # This factor might capture both the price movement and the trading activity
    alpha_factor = price_ratio * volume_ratio
    
    return alpha_factor
```
