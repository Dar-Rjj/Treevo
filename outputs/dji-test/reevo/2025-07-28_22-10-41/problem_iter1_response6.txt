```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of the high and low for the day
    close_avg_diff = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate the weighted change in volume from the previous period
    # Here, if the stock closed higher than the midpoint of the high and low, we weigh the volume change positively
    # If it closed lower, we weigh it negatively to capture momentum or lack thereof
    volume_change = df['volume'].pct_change()
    volume_weighted_change = volume_change * (1 if df['close'] > (df['high'] + df['low']) / 2 else -1)
    
    # The final factor is the product of the close-midpoint difference with the volume weighted change,
    # providing a measure that combines directional bias with liquidity flow
    factor_value = close_avg_diff * volume_weighted_change
    return factor_value
```
