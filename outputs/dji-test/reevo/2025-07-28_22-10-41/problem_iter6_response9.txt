```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of close prices to capture momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 50-day and 200-day exponential moving averages (EMA) of the close price to capture longer-term trends
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_close_200'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 14-day relative strength index (RSI) to incorporate mean reversion
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day average true range (ATR) to incorporate volatility
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['tr'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of log returns to capture volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the MACD line and signal line
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = ema_12 - ema_26
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    
    # Determine market regime based on the MACD histogram
    df['macd_histogram'] = df['macd_line'] - df['signal_line']
    df['market_regime'] = np.where(df['macd_histogram'] > 0, 'Bullish', 'Bearish')
    
    # Adaptive factor weights based on market regime
    bullish_weights = [0.5, 1, 1, 50, 1, 0.5]
    bearish_weights = [0.3, 0.8, 0.8, 70, 0.8, 0.3]
    
    df['factor'] = (
        df['log_return'] * np.where(df['market_regime'] == 'Bullish', bullish_weights[0], bearish_weights[0]) +
        (df['close'] - df['ema_close_50']) * np.where(df['market_regime'] == 'Bullish', bullish_weights[1], bearish_weights[1]) +
        (df['close'] - df['ema_close_200']) * np.where(df['market_regime'] == 'Bullish', bullish_weights[2], bearish_weights[2]) +
        (50 - df['rsi']) * np.where(df['market_regime'] == 'Bullish', bullish_weights[3], bearish_weights[3]) +
        (df['atr'] / df['close']) * np.where(df['market_regime'] == 'Bullish', bullish_weights[4], bearish_weights[4]) +
        (df['volatility'] * np.where(df['market_regime'] == 'Bullish', bullish_weights[5], bearish_weights[5]))
    )
    
    return df['factor']
```
