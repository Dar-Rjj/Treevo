```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Calculate the average of the high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in volume compared to the previous day
    pct_change_volume = df['volume'].pct_change().fillna(0)
    
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the volatility over the last 5 days
    volatility_5_days = df['close'].rolling(window=5).std().fillna(0)
    
    # Calculate the relative strength over the last 10 days
    relative_strength_10_days = df['close'] / df['close'].shift(10)
    
    # Calculate the liquidity as the ratio of volume to the average of high and low prices
    liquidity = df['volume'] / avg_high_low
    
    # Calculate the mean reversion factor as the difference between the close price and the 20-day moving average
    mean_reversion_20_days = df['close'] - df['close'].rolling(window=20).mean().fillna(0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (diff_close_open / avg_high_low) * (1 + pct_change_volume) * (1 + daily_return) / (1 + volatility_5_days) * relative_strength_10_days * liquidity * mean_reversion_20_days
    return alpha_factor
```
