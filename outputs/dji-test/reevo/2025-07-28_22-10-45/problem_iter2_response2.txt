```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a volatility measure
    volatility = df['high'] - df['low']
    
    # Calculate the closing price relative to the average of high and low
    close_to_avg = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Calculate the trading volume ratio, normalized by the 5-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the momentum factor over a 10-day period
    momentum = df['close'] / df['close'].shift(10) - 1
    
    # Combine the factors: volatility adjusted close-to-avg, RSI, momentum, and volume impact
    factor = (close_to_avg / (volatility + 1e-7)) * volume_ratio * rsi * momentum
    
    return factor
```
