def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the exponential standard deviation of the price change to measure volatility
    volatility = price_change.ewm(span=5, adjust=False).std()
    
    # Calculate the exponential moving average of the close price to measure trend strength
    trend_strength = df['close'].ewm(span=5, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 5 days
    volume_spike = (df['volume'] - df['volume'].ewm(span=5, adjust=False).mean()) / df['volume'].ewm(span=5, adjust=False).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].ewm(span=5, adjust=False).mean())
    
    # Incorporate market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].ewm(span=5, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=20, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Integrate a macroeconomic indicator: Moving average of the amount traded
    amount_moving_avg = df['amount'].ewm(span=20, adjust=False).mean()
    
    # Calculate the adaptive weights based on the exponential standard deviation of the factors
    std_weighted_price_movement = weighted_price_movement.ewm(span=5, adjust=False).std()
    std_trend_strength = trend_strength.ewm(span=5, adjust=False).std()
    std_volatility = volatility.ewm(span=5, adjust=False).std()
    std_volume_spike = volume_spike.ewm(span=5, adjust=False).std()
    std_sma_crossover = sma_crossover.ewm(span=5, adjust=False).std()
    std_rsi = rsi.ewm(span=5, adjust=False).std()
    std_amount_moving_avg = amount_moving_avg.ewm(span=5, adjust=False).std()
    
    total_std = (std_weighted_price_movement + std_trend_strength + std_volatility + std_volume_spike + std_sma_crossover + std_rsi + std_amount_moving_avg)
    
    w1 = std_weighted_price_movement / total_std
    w2 = std_trend_strength / total_std
    w3 = std_volatility / total_std
    w4 = std_volume_spike / total_std
    w5 = std_sma_crossover / total_std
    w6 = std_rsi / total_std
    w7 = std_amount_moving_avg / total_std
    
    # Combine the factors into a single alpha factor with adaptive weights
    alpha_factor = (w1 * weighted_price_movement + w2 * trend_strength - w3 * volatility + w4 * volume_spike + w5 * sma_crossover + w6 * rsi + w7 * amount_moving_avg)
    
    return alpha_factor.fillna(0)
