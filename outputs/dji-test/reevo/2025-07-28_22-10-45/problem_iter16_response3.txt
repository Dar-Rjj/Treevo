def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the standard deviation of the price change to measure volatility with an adaptive window
    volatility = price_change.rolling(window=7).std()
    
    # Calculate the exponential moving average of the close price to measure trend strength with an adaptive window
    trend_strength = df['close'].ewm(span=10, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 7 days
    volume_spike = (df['volume'] - df['volume'].ewm(span=7, adjust=False).mean()) / df['volume'].ewm(span=7, adjust=False).std()
    
    # Incorporate market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].ewm(span=10, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=30, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Add Bollinger Bands
    middle_band = df['close'].rolling(window=20).mean()
    std_dev = df['close'].rolling(window=20).std()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_band_width = (upper_band - lower_band) / middle_band
    
    # Introduce a momentum indicator: Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Combine the factors into a single alpha factor, balancing momentum and volatility
    alpha_factor = (trend_strength * 0.3 +
                    volume_spike * 0.15 +
                    sma_crossover * 0.2 +
                    rsi * 0.1 -
                    volatility * 0.1 -
                    bollinger_band_width * 0.1 +
                    roc * 0.05)
    
    return alpha_factor.fillna(0)
