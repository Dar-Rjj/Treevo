```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving average of close prices for momentum
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between 10-day and 30-day moving averages as a long-term trend indicator
    long_term_trend = df['ma_10'] - df['ma_30']
    
    # Calculate the 5-day rolling average of high and low for a short-term trend
    avg_high_5day = df['high'].rolling(window=5).mean()
    avg_low_5day = df['low'].rolling(window=5).mean()
    
    # Calculate the difference between 5-day average high and low as a short-term volatility measure
    short_term_volatility = avg_high_5day - avg_low_5day
    
    # Calculate the closing price relative to the 5-day average of high and low
    close_to_avg_5day = (df['close'] - (avg_high_5day + avg_low_5day) / 2)
    
    # Calculate the 5-day change in volume
    volume_change_5day = df['volume'] - df['volume'].shift(5)
    
    # Calculate the trading volume ratio, normalized by the 5-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Combine the factors: long-term trend, short-term volatility adjusted close-to-avg, and volume impact
    factor = (close_to_avg_5day / (short_term_volatility + 1e-7)) * volume_ratio * volume_change_5day * long_term_trend
    
    # Drop intermediate columns
    df.drop(['ma_10', 'ma_30'], axis=1, inplace=True)
    
    return factor
```
