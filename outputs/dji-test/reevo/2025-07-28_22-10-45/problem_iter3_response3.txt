```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility
    volatility = price_change.rolling(window=5).std()
    
    # Calculate the rolling mean of the close price to measure trend strength
    trend_strength = df['close'].rolling(window=5).mean()
    
    # Detect volume spikes by comparing the current volume to the rolling mean of the last 5 days
    volume_spike = (df['volume'] - df['volume'].rolling(window=5).mean()) / df['volume'].rolling(window=5).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].rolling(window=5).mean())
    
    # Incorporate a multi-day pattern by calculating the log return over the last 5 days
    log_return_5d = (df['close'] / df['close'].shift(5)).apply(np.log)
    
    # Combine the factors into a single alpha factor with adaptive weights
    alpha_factor = (0.3 * weighted_price_movement + 0.3 * trend_strength - 0.2 * volatility + 0.1 * volume_spike + 0.1 * log_return_5d)
    
    return alpha_factor.fillna(0)
```
