def heuristics_v2(df: pd.DataFrame, sector_performance: pd.Series, macro_indicators: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the adaptive span for volatility based on recent price change
    span_volatility = 5 + (price_change.rolling(window=5).std() * 10).astype(int)
    span_volatility = span_volatility.clip(lower=5, upper=20)
    
    # Calculate the exponential moving average of the price change to measure volatility
    volatility = price_change.ewm(span=span_volatility, adjust=False).std()
    
    # Calculate the adaptive span for trend strength based on recent close price
    span_trend = 5 + (df['close'].rolling(window=5).std() * 10).astype(int)
    span_trend = span_trend.clip(lower=5, upper=20)
    
    # Calculate the exponential rolling mean of the close price to measure trend strength
    trend_strength = df['close'].ewm(span=span_trend, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 5 days
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_spike = (df['volume'] - volume_ema) / df['volume'].ewm(span=5, adjust=False).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * volume_ema)
    
    # Incorporate market sentiment using a simple moving average crossover
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=30).mean()
    sentiment = (short_term_sma - long_term_sma) / df['close']
    
    # Integrate sector performance
    sector_effect = sector_performance.reindex(df.index, method='ffill')
    sector_impact = (sector_effect - sector_effect.mean()) / sector_effect.std()
    
    # Integrate macroeconomic indicators
    macro_effects = macro_indicators.reindex(df.index, method='ffill')
    macro_impact = (macro_effects - macro_effects.mean()) / macro_effects.std()
    macro_impact = macro_impact.sum(axis=1)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (weighted_price_movement + trend_strength - volatility + volume_spike + sentiment + sector_impact + macro_impact) / 7
    
    return alpha_factor.fillna(0)
