```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day rolling average of high and low for a short-term trend
    avg_high_3day = df['high'].rolling(window=3).mean()
    avg_low_3day = df['low'].rolling(window=3).mean()
    
    # Calculate the difference between 3-day average high and low as a short-term volatility measure
    short_term_volatility = avg_high_3day - avg_low_3day
    
    # Calculate the closing price relative to the 3-day average of high and low
    close_to_avg_3day = (df['close'] - (avg_high_3day + avg_low_3day) / 2)
    
    # Calculate the 5-day change in volume
    volume_change_5day = df['volume'] - df['volume'].shift(5)
    
    # Calculate the trading volume ratio, normalized by the 5-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Combine the factors: short-term volatility adjusted close-to-avg with volume impact and 5-day volume change
    factor = (close_to_avg_3day / (short_term_volatility + 1e-7)) * volume_ratio * volume_change_5day
    
    return factor
```
