```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the 5-day moving average of the closing price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    df['close_ma_5_diff'] = df['close'] - df['ma_5_close']
    
    # Calculate the change in volume compared to the 5-day moving average of volume
    df['ma_5_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_change'] = df['volume'] - df['ma_5_volume']
    
    # Alpha factor: The product of the difference between the closing price and its 5-day moving average
    # and the normalized change in volume (compared to its 5-day moving average)
    df['alpha_factor'] = df['close_ma_5_diff'] * (df['volume_change'] / (df['ma_5_volume'] + 1e-7))
    
    # Dropping intermediate columns
    df.drop(['ma_5_close', 'close_ma_5_diff', 'ma_5_volume', 'volume_change'], axis=1, inplace=True)
    
    return df['alpha_factor']
```
