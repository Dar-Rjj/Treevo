def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility
    volatility = price_change.rolling(window=5).std()
    
    # Calculate the rolling mean of the close price to measure trend strength
    short_trend_strength = df['close'].rolling(window=5).mean()
    long_trend_strength = df['close'].rolling(window=20).mean()
    
    # Detect volume spikes by comparing the current volume to the rolling mean of the last 5 days
    volume_spike = (df['volume'] - df['volume'].rolling(window=5).mean()) / df['volume'].rolling(window=5).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].rolling(window=5).mean())
    
    # Calculate momentum as the 10-day return
    momentum = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate relative strength as the ratio of the current close to the 10-day moving average
    relative_strength = df['close'] / df['close'].rolling(window=10).mean()
    
    # Incorporate market sentiment using a simple moving average of the close price over a longer period
    market_sentiment = df['close'].rolling(window=20).mean()
    
    # Calculate the liquidity factor as the ratio of volume to the average of the last 10 days
    liquidity_factor = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adaptive weighting based on the recent performance of each factor
    weight_weighted_price_movement = (weighted_price_movement.shift(5).corr(df['close'])) ** 2
    weight_short_trend_strength = (short_trend_strength.shift(5).corr(df['close'])) ** 2
    weight_long_trend_strength = (long_trend_strength.shift(5).corr(df['close'])) ** 2
    weight_volatility = (volatility.shift(5).corr(df['close'])) ** 2
    weight_volume_spike = (volume_spike.shift(5).corr(df['close'])) ** 2
    weight_momentum = (momentum.shift(5).corr(df['close'])) ** 2
    weight_relative_strength = (relative_strength.shift(5).corr(df['close'])) ** 2
    weight_market_sentiment = (market_sentiment.shift(5).corr(df['close'])) ** 2
    weight_liquidity_factor = (liquidity_factor.shift(5).corr(df['close'])) ** 2
    
    total_weight = (weight_weighted_price_movement + weight_short_trend_strength + weight_long_trend_strength +
                    weight_volatility + weight_volume_spike + weight_momentum + 
                    weight_relative_strength + weight_market_sentiment + weight_liquidity_factor)
    
    alpha_factor = (weighted_price_movement * weight_weighted_price_movement +
                    short_trend_strength * weight_short_trend_strength +
                    long_trend_strength * weight_long_trend_strength -
                    volatility * weight_volatility +
                    volume_spike * weight_volume_spike +
                    momentum * weight_momentum +
                    relative_strength * weight_relative_strength +
                    market_sentiment * weight_market_sentiment +
                    liquidity_factor * weight_liquidity_factor) / total_weight
    
    return alpha_factor.fillna(0)
