```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility with a dynamic window
    volatility = price_change.rolling(window=5 + (df['close'].rolling(window=10).std() * 2).astype(int)).std()
    
    # Calculate the rolling mean of the close price to measure trend strength with a dynamic window
    trend_strength = df['close'].rolling(window=5 + (df['close'].rolling(window=10).std() * 2).astype(int)).mean()
    
    # Detect volume spikes by comparing the current volume to the rolling mean of the last 5 days
    volume_spike = (df['volume'] - df['volume'].rolling(window=5).mean()) / df['volume'].rolling(window=5).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].rolling(window=5).mean())
    
    # Incorporate market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].rolling(window=5 + (df['close'].rolling(window=10).std() * 2).astype(int)).mean()
    long_term_sma = df['close'].rolling(window=20 + (df['close'].rolling(window=10).std() * 4).astype(int)).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14 + (df['close'].rolling(window=10).std() * 2).astype(int)).mean()
    avg_loss = loss.rolling(window=14 + (df['close'].rolling(window=10).std() * 2).astype(int)).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Add a new technical indicator: Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(price_change > 0, 0)
    negative_flow = raw_money_flow.where(price_change < 0, 0)
    money_ratio = positive_flow.rolling(window=14 + (df['close'].rolling(window=10).std() * 2).astype(int)).sum() / negative_flow.rolling(window=14 + (df['close'].rolling(window=10).std() * 2).astype(int)).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate momentum as the percentage change in the close price over a dynamic window
    momentum_window = 10 + (df['close'].rolling(window=10).std() * 2).astype(int)
    momentum = df['close'].pct_change(periods=momentum_window)
    
    # Combine the factors into a single alpha factor with weighted averages
    alpha_factor = (
        0.3 * weighted_price_movement +
        0.2 * trend_strength -
        0.1 * volatility +
        0.1 * volume_spike +
        0.1 * sma_crossover +
        0.1 * rsi +
        0.1 * mfi +
        0.1 * momentum
    )
    
    return alpha_factor.fillna(0)
```
