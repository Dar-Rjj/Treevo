```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between open and close price
    daily_diff = df['close'] - df['open']
    
    # Calculate the mean of the volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate a factor that combines the daily difference with the smoothed volume
    factor = daily_diff / (avg_volume_5d + 1e-7)
    
    return factor
```
