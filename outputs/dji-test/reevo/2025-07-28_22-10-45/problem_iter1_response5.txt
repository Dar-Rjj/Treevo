```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and previous day's close
    diff_close = df['close'].diff().fillna(0)
    # Calculate the weighted moving average of volume with a window of 5 days
    wma_volume = df['volume'].rolling(window=5, min_periods=1).mean()
    # Calculate the factor by multiplying the difference in close prices with the weighted moving average of volume
    factor = diff_close * wma_volume
    return factor
```
