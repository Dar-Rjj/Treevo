```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the normalized average true range over a 10-day window
    atr = ta.volatility.AverageTrueRange(high=df['high'], low=df['low'], close=df['close'], window=10, fillna=False).average_true_range()
    
    # Calculate the relative strength of the current day's close price compared to the 30-day SMA
    sma_30 = ta.trend.sma_indicator(close=df['close'], window=30, fillna=False)
    relative_strength = df['close'] / sma_30
    
    # Combine the three factors into a single alpha factor
    alpha_factor = (diff_close_open / atr) * relative_strength

    return alpha_factor
```
