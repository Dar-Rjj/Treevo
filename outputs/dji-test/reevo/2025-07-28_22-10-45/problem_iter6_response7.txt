```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility with a dynamic window size
    dynamic_window = 5 + (df['volume'].rolling(window=10).mean() / df['volume'])
    volatility = price_change.rolling(window=dynamic_window.astype(int), min_periods=5).std()
    
    # Calculate the rolling mean of the close price to measure trend strength with a dynamic window size
    trend_strength = df['close'].rolling(window=dynamic_window.astype(int), min_periods=5).mean()
    
    # Detect volume spikes by comparing the current volume to the rolling mean of the last 5 days
    volume_spike = (df['volume'] - df['volume'].rolling(window=5, min_periods=1).mean()) / df['volume'].rolling(window=5, min_periods=1).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].rolling(window=5, min_periods=1).mean())
    
    # Calculate momentum as the 10-day return
    momentum = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate relative strength as the ratio of the current close to the 10-day moving average
    relative_strength = df['close'] / df['close'].rolling(window=10).mean()
    
    # Calculate RSI (Relative Strength Index)
    gain = price_change.where(price_change > 0, 0)
    loss = -price_change.where(price_change < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_diff = macd - signal
    
    # Combine the factors into a single alpha factor
    alpha_factor = (weighted_price_movement + trend_strength - volatility + volume_spike + momentum + relative_strength + rsi + macd_diff) / 8
    
    return alpha_factor.fillna(0)
```
