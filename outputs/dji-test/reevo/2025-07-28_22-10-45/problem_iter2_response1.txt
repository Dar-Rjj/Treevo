```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return as the difference between close and open price
    daily_return = df['close'] - df['open']
    
    # Calculate the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the factor as the ratio of daily return to the average of high and low
    factor = daily_return / (avg_high_low + 1e-7)
    
    # Calculate the volatility as the standard deviation of the last 5 days' close prices
    df['volatility'] = df['close'].rolling(window=5).std()
    
    # Adjust the factor by incorporating volume, assuming higher volume indicates a stronger signal
    adjusted_factor = factor * np.log(df['volume'] + 1)
    
    # Further adjust the factor by scaling it with the inverse of the volatility
    final_factor = adjusted_factor / (df['volatility'] + 1e-7)
    
    # Drop intermediate columns
    df.drop(['volatility'], axis=1, inplace=True)
    
    return final_factor
```
