def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Adaptive standard deviation of the price change to measure volatility
    volatility = price_change.rolling(window=7, min_periods=1).std()
    
    # Adaptive exponential moving average of the close price to measure trend strength
    trend_strength = df['close'].ewm(span=7, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 7 days
    volume_spike = (df['volume'] - df['volume'].ewm(span=7, adjust=False).mean()) / df['volume'].ewm(span=7, adjust=False).std()
    
    # Weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].ewm(span=7, adjust=False).mean())
    
    # Market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].ewm(span=7, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=25, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.mask(typical_price <= typical_price.shift(1), 0).fillna(0)
    negative_money_flow = raw_money_flow.mask(typical_price > typical_price.shift(1), 0).fillna(0)
    money_ratio = positive_money_flow.rolling(window=14, min_periods=1).sum() / negative_money_flow.rolling(window=14, min_periods=1).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Momentum indicator: Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Chaikin Money Flow (CMF)
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    cmf = (mf_multiplier * df['volume']).rolling(window=14, min_periods=1).sum() / df['volume'].rolling(window=14, min_periods=1).sum()
    
    # Average True Range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14, min_periods=1).mean()
    
    # Adaptive window for EMA based on market conditions
    adaptive_window = (df['volume'] / df['volume'].ewm(span=7, adjust=False).mean()).clip(lower=1, upper=21)
    adaptive_trend_strength = df['close'].ewm(span=adaptive_window.astype(int), adjust=False).mean()
    
    # Combine the factors into a single alpha factor with refined weightings
    alpha_factor = (weighted_price_movement * 0.3 +
                    adaptive_trend_strength * 0.2 -
                    volatility * 0.1 +
                    volume_spike * 0.1 +
                    sma_crossover * 0.1 +
                    rsi * 0.08 +
                    mfi * 0.08 +
                    roc * 0.05 +
                    cmf * 0.05 -
                    atr * 0.05)
    
    return alpha_factor.fillna(0)
