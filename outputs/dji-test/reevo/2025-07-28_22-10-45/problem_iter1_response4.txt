```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, then divided by volume to get a price change per unit volume
    price_change_per_volume = (df['close'] - df['open']) / df['volume']
    # Calculate the mean of the high and low price as an average true range like figure
    avg_true_range = (df['high'] + df['low']) / 2
    # The final factor is computed as the rolling correlation between the price change per unit volume and the average true range
    # This aims to capture periods where price changes are strongly related to the average magnitude of price movements
    factor = price_change_per_volume.rolling(window=5).corr(avg_true_range)
    return factor
```
