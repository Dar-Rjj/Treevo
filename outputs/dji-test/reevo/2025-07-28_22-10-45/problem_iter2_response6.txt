```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a volatility measure
    volatility = df['high'] - df['low']
    
    # Calculate the closing price relative to the average of high and low
    close_to_avg = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Calculate the relative strength, using the ratio of current close to the 10-day moving average of the close
    relative_strength = df['close'] / df['close'].rolling(window=10).mean()
    
    # Calculate the trading volume ratio, normalized by the 5-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the 20-day and 50-day simple moving averages of the close price
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the distance between the 20-day and 50-day SMA
    sma_distance = sma_20 - sma_50
    
    # Combine the factors: volatility adjusted close-to-avg, relative strength, and volume impact with SMA distance
    factor = (close_to_avg / (volatility + 1e-7)) * volume_ratio * relative_strength * sma_distance
    
    return factor
```
