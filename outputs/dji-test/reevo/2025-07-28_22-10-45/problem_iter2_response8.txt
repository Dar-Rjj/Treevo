```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Volatility measure: Calculate the standard deviation of the close prices over a 5-day window
    volatility = df['close'].rolling(window=5).std()
    
    # Momentum measure: Calculate the difference between the current close and the close 5 days ago
    momentum = df['close'] - df['close'].shift(5)
    
    # Liquidity measure: Calculate the ratio of volume to the dollar amount traded (close * volume)
    liquidity = df['volume'] / (df['close'] * df['volume'])
    
    # Price change per unit volume: Calculate the difference between close and open prices, then divided by volume
    price_change_per_volume = (df['close'] - df['open']) / df['volume']
    
    # Average true range like figure: Calculate the mean of the high and low prices
    avg_true_range = (df['high'] + df['low']) / 2
    
    # Combine the measures using a rolling correlation between price change per unit volume and average true range
    # This aims to capture periods where price changes are strongly related to the average magnitude of price movements
    # Multiply the correlation by the volatility and momentum measures to incorporate their effects
    factor = (price_change_per_volume.rolling(window=5).corr(avg_true_range) 
              * volatility * momentum * liquidity)
    
    return factor
```
