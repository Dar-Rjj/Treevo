```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in close price to capture volatility
    pct_change_close = df['close'].pct_change().fillna(0)
    
    # Calculate the relative strength by comparing the close price to the 5-day moving average of the close price
    ma_close_5 = df['close'].rolling(window=5, min_periods=1).mean()
    relative_strength = df['close'] / ma_close_5
    
    # Calculate the typical price as a representation of the market context
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the factor as the rolling correlation between the percentage change in close price and the typical price
    # This aims to capture the relationship between price volatility and the overall market context
    factor = pct_change_close.rolling(window=5).corr(typical_price) * relative_strength
    
    return factor
```
