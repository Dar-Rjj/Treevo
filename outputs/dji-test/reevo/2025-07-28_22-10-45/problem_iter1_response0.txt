```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing and opening prices
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the weighted price movement relative to volume
    weighted_price_movement = (price_diff * df['volume']) / (avg_high_low * df['volume'].rolling(window=5).mean())
    
    return weighted_price_movement.fillna(0)
```
