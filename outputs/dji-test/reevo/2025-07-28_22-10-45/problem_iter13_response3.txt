def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Adaptive window for volatility based on the rolling standard deviation of the price change
    adaptive_volatility_window = (price_change.abs() / avg_high_low).rolling(window=10).mean().fillna(5).astype(int)
    volatility = price_change.rolling(window=adaptive_volatility_window).std()
    
    # Adaptive window for trend strength using EMA
    adaptive_trend_window = (volatility / avg_high_low).rolling(window=10).mean().fillna(5).astype(int)
    trend_strength = df['close'].ewm(span=adaptive_tread_window, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last adaptive window days
    adaptive_volume_window = (df['volume'] / avg_high_low).rolling(window=10).mean().fillna(5).astype(int)
    volume_spike = (df['volume'] - df['volume'].ewm(span=adaptive_volume_window, adjust=False).mean()) / df['volume'].ewm(span=adaptive_volume_window, adjust=False).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].ewm(span=adaptive_volume_window, adjust=False).mean())
    
    # Incorporate market sentiment using a simple moving average cross-over with adaptive windows
    short_term_sma = df['close'].ewm(span=adaptive_trend_window, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=2 * adaptive_trend_window, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI) with adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=adaptive_volatility_window).mean()
    avg_loss = loss.rolling(window=adaptive_volatility_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Add another technical indicator: Money Flow Index (MFI) with adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.mask(typical_price <= typical_price.shift(1), 0).fillna(0)
    negative_money_flow = raw_money_flow.mask(typical_price > typical_price.shift(1), 0).fillna(0)
    money_ratio = positive_money_flow.rolling(window=adaptive_volatility_window).sum() / negative_money_flow.rolling(window=adaptive_volatility_window).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors into a single alpha factor with adjusted weights
    alpha_factor = (weighted_price_movement * 0.3 + trend_strength * 0.2 - volatility * 0.1 + volume_spike * 0.1 + sma_crossover * 0.1 + rsi * 0.1 + mfi * 0.1)
    
    return alpha_factor.fillna(0)
