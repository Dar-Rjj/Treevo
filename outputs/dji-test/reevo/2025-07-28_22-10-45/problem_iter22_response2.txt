def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the adaptive standard deviation of the price change to measure volatility
    volatility = price_change.rolling(window=15, min_periods=1).std()
    
    # Calculate the adaptive exponential moving average of the close price to measure trend strength
    trend_strength = df['close'].ewm(span=15, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 15 days
    volume_ema = df['volume'].ewm(span=15, adjust=False).mean()
    volume_std = df['volume'].ewm(span=15, adjust=False).std()
    volume_spike = (df['volume'] - volume_ema) / volume_std
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * volume_ema)
    
    # Incorporate market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].ewm(span=10, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=30, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Add another technical indicator: Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.mask(typical_price <= typical_price.shift(1), 0).fillna(0)
    negative_money_flow = raw_money_flow.mask(typical_price > typical_price.shift(1), 0).fillna(0)
    money_ratio = positive_money_flow.rolling(window=14, min_periods=1).sum() / negative_money_flow.rolling(window=14, min_periods=1).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Introduce a momentum indicator: Rate of Change (ROC)
    roc = df['close'].pct_change(periods=20).fillna(0)
    
    # Adjust for market regimes by adding a regime factor
    regime_factor = (df['close'] - df['close'].rolling(window=100, min_periods=1).mean()) / df['close'].rolling(window=100, min_periods=1).std()
    
    # Incorporate a market sentiment indicator: Aroon Oscillator
    aroon_up = 100 * df['high'].rolling(window=25, min_periods=1).apply(lambda x: (x.argmax() + 1) / 25, raw=True)
    aroon_down = 100 * df['low'].rolling(window=25, min_periods=1).apply(lambda x: (x.argmin() + 1) / 25, raw=True)
    aroon_oscillator = aroon_up - aroon_down
    
    # Combine the factors into a single alpha factor with refined weightings and regime adjustment
    alpha_factor = (weighted_price_movement * 0.35 +
                    trend_strength * 0.15 -
                    volatility * 0.1 +
                    volume_spike * 0.08 +
                    sma_crossover * 0.08 +
                    rsi * 0.07 +
                    mfi * 0.07 +
                    roc * 0.05 +
                    regime_factor * 0.05 +
                    aroon_oscillator * 0.05)
    
    return alpha_factor.fillna(0)
