```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility over a longer period
    long_term_volatility = price_change.rolling(window=20).std()
    short_term_volatility = price_change.rolling(window=5).std()
    
    # Calculate the rolling mean of the close price to measure trend strength over a longer period
    long_term_trend_strength = df['close'].rolling(window=20).mean()
    short_term_trend_strength = df['close'].rolling(window=5).mean()
    
    # Detect volume spikes by comparing the current volume to the rolling mean of the last 5 days
    volume_spike = (df['volume'] - df['volume'].rolling(window=5).mean()) / df['volume'].rolling(window=5).std()
    
    # Calculate the weighted price movement relative to the average high and low using a longer-term average
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].rolling(window=20).mean())
    
    # Combine the factors into a single alpha factor using a weighted average
    alpha_factor = (weighted_price_movement * 0.4 + 
                    long_term_trend_strength * 0.3 - 
                    (long_term_volatility * 0.2 + short_term_volatility * 0.1) + 
                    volume_spike * 0.2)
    
    return alpha_factor.fillna(0)
```
