def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the exponentially weighted moving standard deviation of the price change to measure volatility
    volatility = price_change.ewm(span=5, adjust=False).std()
    
    # Calculate the exponentially weighted moving average of the close price to measure trend strength
    trend_strength = df['close'].ewm(span=5, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponentially weighted moving mean
    volume_spike = (df['volume'] - df['volume'].ewm(span=5, adjust=False).mean()) / df['volume'].ewm(span=5, adjust=False).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].ewm(span=5, adjust=False).mean())
    
    # Incorporate market sentiment using an exponential moving average cross-over
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_crossover = short_term_ema - long_term_ema
    
    # Add a new technical indicator: Relative Strength Index (RSI) with exponential weighting
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.ewm(span=14, adjust=False).mean()
    avg_loss = loss.ewm(span=14, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (weighted_price_movement + trend_strength - volatility + volume_spike + ema_crossover + rsi) / 6
    
    return alpha_factor.fillna(0)
