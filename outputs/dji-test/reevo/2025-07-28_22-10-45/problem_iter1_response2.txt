```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and opening price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average volume over a 10-day window
    avg_volume_10day = df['volume'].rolling(window=10).mean()
    
    # Calculate the factor as the ratio of the difference in closing and opening prices to the 10-day average volume
    factor = diff_close_open / (avg_volume_10day + 1e-7)
    
    return factor
```
