def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change (close - open)
    price_change = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the standard deviation of the price change to measure volatility with an adaptive window
    volatility = price_change.rolling(window=5).std()
    
    # Calculate the exponential moving average of the close price to measure trend strength with an adaptive window
    trend_strength = df['close'].ewm(span=5, adjust=False).mean()
    
    # Detect volume spikes by comparing the current volume to the exponential moving average of the last 5 days
    volume_spike = (df['volume'] - df['volume'].ewm(span=5, adjust=False).mean()) / df['volume'].ewm(span=5, adjust=False).std()
    
    # Calculate the weighted price movement relative to the average high and low
    weighted_price_movement = (price_change * df['volume']) / (avg_high_low * df['volume'].ewm(span=5, adjust=False).mean())
    
    # Incorporate market sentiment using a simple moving average cross-over
    short_term_sma = df['close'].ewm(span=5, adjust=False).mean()
    long_term_sma = df['close'].ewm(span=20, adjust=False).mean()
    sma_crossover = short_term_sma - long_term_sma
    
    # Add a new technical indicator: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Add another technical indicator: Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.mask(typical_price <= typical_price.shift(1), 0).fillna(0)
    negative_money_flow = raw_money_flow.mask(typical_price > typical_price.shift(1), 0).fillna(0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Introduce a dynamic indicator: Adaptive Moving Average (AMA)
    ama_alpha = 2 / (df['close'].ewm(span=10, adjust=False).std().fillna(0) + 1)
    ama = df['close'].ewm(alpha=ama_alpha, adjust=False).mean()
    
    # Add a regime detection mechanism
    market_regime = df['close'].pct_change().rolling(window=20).mean()
    regime_indicator = market_regime.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine the factors into a single alpha factor with refined weights and dynamic adjustments
    alpha_factor = (weighted_price_movement * 0.3 +
                    trend_strength * 0.2 -
                    volatility * 0.1 +
                    volume_spike * 0.1 +
                    sma_crossover * 0.1 +
                    rsi * 0.1 +
                    mfi * 0.1 +
                    ama * 0.1) * regime_indicator
    
    return alpha_factor.fillna(0)
