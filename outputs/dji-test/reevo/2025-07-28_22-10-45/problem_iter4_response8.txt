```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day rolling average of high and low for short-term and medium-term trends
    avg_high_3day = df['high'].rolling(window=3).mean()
    avg_low_3day = df['low'].rolling(window=3).mean()
    avg_high_10day = df['high'].rolling(window=10).mean()
    avg_low_10day = df['low'].rolling(window=10).mean()
    
    # Calculate the difference between 3-day and 10-day average high and low as a trend-based volatility measure
    trend_volatility = (avg_high_3day - avg_low_3day) / (avg_high_10day - avg_low_10day + 1e-7)
    
    # Calculate the closing price relative to the 10-day average of high and low
    close_to_avg_10day = (df['close'] - (avg_high_10day + avg_low_10day) / 2)
    
    # Calculate the 5-day and 20-day change in volume for short-term and medium-term volume trends
    volume_change_5day = df['volume'] - df['volume'].shift(5)
    volume_change_20day = df['volume'] - df['volume'].shift(20)
    
    # Calculate the trading volume ratio, normalized by the 20-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine the factors: trend-based volatility adjusted close-to-avg with volume impact and multi-period volume change
    factor = (close_to_avg_10day / (trend_volatility + 1e-7)) * volume_ratio * (volume_change_5day / (volume_change_20day + 1e-7))
    
    return factor
```
