```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily price change
    daily_price_change = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Adjust volume by the average of high and low prices
    adjusted_volume = df['volume'] / avg_high_low
    
    # Calculate the 5-day rolling standard deviation of the close price
    volatility = df['close'].rolling(window=5).std()
    
    # Create an alpha factor that combines the daily price change, adjusted volume, and volatility
    alpha_factor = (daily_price_change * adjusted_volume) / volatility
    
    return alpha_factor
```
