```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum with a focus on adjusting to market conditions dynamically
    short_term_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].pct_change(periods=60).ewm(span=60, adjust=False).mean()
    adaptive_momentum = 0.6 * short_term_momentum + 0.4 * long_term_momentum

    # Enhanced dynamic volatility using the exponential moving average of squared log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    squared_log_returns = log_returns**2
    dynamic_volatility = squared_log_returns.rolling(window=30).mean().ewm(span=14, adjust=False).mean()

    # Normalized volume adjusted for overall market activity
    overall_market_volume = df['volume'].sum()
    normalized_volume = (df['volume'] - df['volume'].mean()) / (df['volume'].std() + 1e-6) * (df['volume'] / overall_market_volume)

    # Market sentiment using the ratio of high to low prices and the volume weighted by the close price
    market_sentiment = (df['high'] - df['low']) / df['close'] * (df['volume'] / df['close'])

    # Cumulative trend measure using the sum of the last 10 days' closing prices
    cumulative_trend = df['close'].rolling(window=10).sum().pct_change()

    # Sector-specific insight using the relative strength of the stock's performance against its sector
    if 'sector_index' in df.columns:
        relative_strength = (df['close'] / df['sector_index']).pct_change(periods=20)
    else:
        relative_strength = 0

    # Combining the factors into a single alpha factor
    # Weights are adjusted based on their perceived importance and the addition of new factors
    factor_values = (
        0.3 * adaptive_momentum 
        - 0.2 * dynamic_volatility 
        + 0.1 * normalized_volume 
        + 0.1 * market_sentiment 
        + 0.2 * cumulative_trend 
        + 0.1 * relative_strength
    )

    return factor_values
```
