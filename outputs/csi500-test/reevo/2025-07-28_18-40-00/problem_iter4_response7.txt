```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average of (open, high, low, close)
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the weighted amount by volume
    weighted_amount = df['amount'] * df['volume']
    
    # Calculate the smoothed average price using a longer rolling window for better smoothing
    smoothed_avg_price = avg_price.rolling(window=20).mean()
    
    # Calculate the momentum factor over a longer period to capture longer-term trends
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    
    # Calculate the volume-adjusted price difference
    price_diff = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    # Combine the factors
    factor = (smoothed_avg_price * long_term_momentum + price_diff) / (weighted_amount + 1e-7)
    
    return factor
```
