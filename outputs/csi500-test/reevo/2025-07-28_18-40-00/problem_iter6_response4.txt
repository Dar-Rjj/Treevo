```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price momentum as the change in closing price over the last 10 days
    price_momentum = df['close'].pct_change(periods=10)
    
    # Calculate the money flow ratio as the average of (high - low) * volume over the last 5 days
    # This aims to capture the strength of money inflow or outflow over recent trading days
    money_flow_ratio = ((df['high'] - df['low']) * df['volume']).rolling(window=5).mean()
    
    # Calculate a simple oscillator as the difference between 5-day and 10-day moving averages
    # This may highlight reversals or confirm trends
    short_ma = df['close'].rolling(window=5).mean()
    long_ma = df['close'].rolling(window=10).mean()
    simple_oscillator = short_ma - long_ma
    
    # Calculate the relative strength index (RSI) over a 14-day period
    # This measures the magnitude of recent price changes to evaluate overbought or oversold conditions
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the dynamic volatility as the standard deviation of the last 10 days' returns
    # This aims to capture the volatility of the stock over recent trading days
    dynamic_volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the On-Balance Volume (OBV) to incorporate volume into trend analysis
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv -= (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate the MACD (Moving Average Convergence Divergence) for a more sophisticated trend indicator
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_short - ema_long
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    # Combine the factors into a single alpha factor
    # Dynamically adjust weights based on market conditions
    market_trend = df['close'].rolling(window=50).mean() - df['close'].rolling(window=200).mean()
    trend_weight = (market_trend > 0).astype(int) * 0.5 + (market_trend <= 0).astype(int) * 0.3
    factor_values = (trend_weight * 0.3 * price_momentum + 
                     0.2 * money_flow_ratio + 
                     0.3 * simple_oscillator + 
                     0.1 * rsi - 
                     0.1 * dynamic_volatility + 
                     0.1 * obv + 
                     0.1 * macd_histogram)
    
    return factor_values
```
