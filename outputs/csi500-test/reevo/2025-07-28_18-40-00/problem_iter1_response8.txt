```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the moving average of the close price over 10 days
    df['ma_10'] = df['close'].rolling(window=10).mean()
    # Calculate the standard deviation of the close price over 10 days
    df['std_10'] = df['close'].rolling(window=10).std()
    # Compute a z-score as a measure of momentum and mean reversion
    df['z_score'] = (df['close'] - df['ma_10']) / df['std_10']
    # Calculate the ratio of volume to its rolling median over 30 days, as a relative strength indicator
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=30).median()
    # Combine the factors using a weighted sum to create the alpha factor
    alpha_factor = 0.5 * df['z_score'] + 0.5 * df['volume_ratio'] * df['close_diff']
    
    return alpha_factor
```
