```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price
    price_momentum = df['close'].diff()
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    # Create an alpha factor which is the product of price momentum and normalized (divided by average) 5-day volume
    factor_values = price_momentum * (df['volume'] / avg_volume_5d)
    return factor_values
```
