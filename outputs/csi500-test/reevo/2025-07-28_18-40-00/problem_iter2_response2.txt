```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average daily range over the past 5 days as a measure of volatility
    avg_daily_range_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the 5-day return momentum, to assess recent price trend
    return_momentum_5d = df['close'].pct_change(periods=5)
    
    # Calculate the volume weighted close price over the last 5 days
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Compute the deviation of today's close from the 5-day volume weighted average
    deviation_from_volume_weighted = df['close'] - volume_weighted_close
    
    # Create an alpha factor which combines momentum, volatility, and deviation
    alpha_factor = (return_momentum_5d + deviation_from_volume_weighted) / (avg_daily_range_5d + 1e-7)
    
    return alpha_factor
```
