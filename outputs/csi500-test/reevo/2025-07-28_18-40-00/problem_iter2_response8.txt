```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Multi-period momentum
    short_momentum = df['close'].pct_change(periods=5)
    long_momentum = df['close'].pct_change(periods=20)
    
    # Liquidity indicator
    liquidity = (df['high'] - df['low']) / df['volume']
    avg_liquidity = liquidity.rolling(window=10).mean()
    
    # Trend indicators
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    trend_indicator = ema_short - ema_long
    
    # Combine the factors into a single alpha factor
    # Weights are placeholders and should be optimized through backtesting
    factor_values = 0.3 * short_momentum + 0.2 * long_momentum + 0.2 * avg_liquidity + 0.3 * trend_indicator

    return factor_values
```
