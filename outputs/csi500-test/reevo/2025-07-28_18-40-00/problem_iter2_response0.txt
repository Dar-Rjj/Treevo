```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum: Calculate the simple moving average of the close price over the last 5 days.
    momentum = df['close'].rolling(window=5).mean()
    
    # Volatility: Calculate the standard deviation of the daily price changes (close - open) over the last 5 days.
    volatility = (df['close'] - df['open']).rolling(window=5).std()
    
    # Liquidity: Calculate the average volume over the last 5 days.
    liquidity = df['volume'].rolling(window=5).mean()
    
    # Relative Price Movement: Calculate the relative difference between the close and open prices.
    relative_price_movement = (df['close'] - df['open']) / df['open']
    
    # Alpha factor: Combine momentum, volatility, and liquidity into a single factor.
    # This factor aims to capture the impact of momentum, while adjusting for volatility and liquidity.
    alpha_factor = (relative_price_movement * momentum) / (volatility + liquidity + 1e-7)
    
    return alpha_factor
```
