```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day price momentum
    price_momentum = df['close'].pct_change(periods=20)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = ((df['close'] + df['high'] + df['low']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the money flow ratio using the VWAP
    money_flow_ratio = ((vwap - df['close'].shift(1)) * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate a simple oscillator as the difference between 5-day and 20-day moving averages of VWAP
    short_ma_vwap = vwap.rolling(window=5).mean()
    long_ma_vwap = vwap.rolling(window=20).mean()
    vwap_oscillator = short_ma_vwap - long_ma_vwap
    
    # Calculate the market sentiment factor as the difference between the 5-day and 20-day average volume
    avg_volume_short = df['volume'].rolling(window=5).mean()
    avg_volume_long = df['volume'].rolling(window=20).mean()
    market_sentiment = avg_volume_short - avg_volume_long
    
    # Calculate the relative strength index (RSI) for the close prices
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.3 * price_momentum + 0.2 * money_flow_ratio + 0.2 * vwap_oscillator + 0.15 * market_sentiment + 0.15 * rsi

    return factor_values
```
