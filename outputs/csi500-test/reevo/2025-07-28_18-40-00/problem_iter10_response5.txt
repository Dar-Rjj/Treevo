def heuristics_v2(df: pd.DataFrame, macroeconomic_data: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day price momentum
    price_momentum = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day average volume using weighted moving average
    avg_volume_10d = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day volume trend
    volume_trend = df['volume'].pct_change(periods=10)
    
    # Calculate the 30-day volatility (standard deviation of daily returns)
    volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Calculate the relative price movement (close - open) / open
    relative_price_movement = (df['close'] - df['open']) / df['open']
    
    # Calculate the 50-day moving average of the close price using weighted moving average
    ma_50d = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 200-day moving average of the close price using weighted moving average
    ma_200d = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the long-term trend indicator (difference between 50-day MA and 200-day MA)
    long_term_trend = ma_50d - ma_200d
    
    # Calculate the market microstructure effect (amount/volume ratio)
    microstructure_effect = df['amount'] / df['volume']
    
    # Calculate the 20-day price range (high - low) / open
    price_range = (df['high'] - df['low']) / df['open']
    
    # Calculate the 10-day price change (close - open) / open
    price_change = (df['close'] - df['open']).pct_change(periods=10)
    
    # Integrate macroeconomic indicators
    macroeconomic_trend = macroeconomic_data['indicator'].pct_change(periods=30)
    macroeconomic_volatility = macroeconomic_data['indicator'].pct_change().rolling(window=30).std()
    
    # Combine multiple indicators using multiplicative combinations to create the alpha factor
    factor_values = (
        price_momentum * 
        volume_trend * 
        relative_price_movement * 
        long_term_trend * 
        price_range * 
        price_change * 
        (macroeconomic_trend + 1) / (macroeconomic_volatility + 1e-7)
    ) / (volatility + 1e-7) + microstructure_effect
    
    # Incorporate seasonality
    seasonality_factor = df.index.dayofweek.map({0: 1.0, 1: 1.0, 2: 1.0, 3: 1.0, 4: 1.0, 5: 0.8, 6: 0.8})
    factor_values = factor_values * seasonality_factor
    
    return factor_values
