```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive price momentum using exponential weighted moving average over the last 10 and 30 days
    price_momentum_short = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum_long = df['close'].ewm(span=30, adjust=False).mean()
    adaptive_momentum = price_momentum_short - price_momentum_long
    
    # Calculate the dynamic volatility as the standard deviation of logarithmic returns over the last 20 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=20).std()
    
    # Calculate the normalized volume as the current volume divided by the 5-day moving average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    normalized_volume = df['volume'] / avg_volume_5d
    
    # Combine the three factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.4 * adaptive_momentum + 0.4 * dynamic_volatility + 0.2 * normalized_volume

    return factor_values
```
