```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a sign of momentum
    df['momentum'] = df['close'].diff()
    
    # Calculate if the stock closed higher than it opened, indicating a bullish day
    df['bullish_day'] = (df['close'] > df['open']).astype(int)
    
    # Calculate the percentage change in volume from the previous day to gauge interest
    df['volume_interest'] = df['volume'].pct_change()

    # Combine the momentum, bullish indicator, and volume interest to create the alpha factor
    df['alpha_factor'] = df['momentum'] * df['bullish_day'] * df['volume_interest']
    
    return df['alpha_factor']
```
