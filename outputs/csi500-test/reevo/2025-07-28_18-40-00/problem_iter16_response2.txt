```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum with exponential smoothing for short and long-term trends
    short_term_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].pct_change(periods=30).ewm(span=30, adjust=False).mean()
    adaptive_momentum = 0.7 * short_term_momentum + 0.3 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over the last 14 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std().ewm(span=14, adjust=False).mean()

    # Relative volume to account for varying liquidity conditions
    volume_ma = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / volume_ma

    # Market sentiment using the ratio of high to low prices as a proxy
    market_sentiment = (df['high'] - df['low']) / df['close']

    # Price reversal indicator, which measures the difference between the open and close price
    price_reversal = (df['open'] - df['close']) / df['close']

    # Sector-specific trend: calculate the average return of the sector
    # Assuming 'sector' is a column in the DataFrame that contains the sector information
    sector_returns = df.groupby('sector')['close'].pct_change().groupby(df['sector']).mean()
    sector_trend = df['close'].pct_change() - sector_returns[df['sector']].values

    # Combining the six factors into a single alpha factor
    # Weights are adjusted based on their perceived importance and cross-validation
    factor_values = (
        0.4 * adaptive_momentum 
        - 0.2 * dynamic_volatility 
        + 0.15 * relative_volume 
        + 0.1 * market_sentiment 
        - 0.05 * price_reversal
        + 0.1 * sector_trend
    )

    return factor_values
```
