```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day price momentum
    price_momentum = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day average volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day volume trend
    volume_trend = df['volume'].pct_change(periods=10)
    
    # Calculate the 30-day volatility (standard deviation of daily returns)
    volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Calculate the relative price movement (close - open) / open
    relative_price_movement = (df['close'] - df['open']) / df['open']
    
    # Calculate the 50-day moving average of the close price
    ma_50d = df['close'].rolling(window=50).mean()
    
    # Calculate the 200-day moving average of the close price
    ma_200d = df['close'].rolling(window=200).mean()
    
    # Calculate the long-term trend indicator (difference between 50-day MA and 200-day MA)
    long_term_trend = ma_50d - ma_200d
    
    # Calculate the market microstructure effect (amount/volume ratio)
    microstructure_effect = df['amount'] / df['volume']
    
    # Combine 20-day price momentum, 10-day volume trend, relative price movement, long-term trend, and inverse of 30-day volatility to create the alpha factor
    factor_values = (price_momentum * volume_trend * relative_price_movement * long_term_trend) / (volatility + 1e-7) + microstructure_effect
    
    return factor_values
```
