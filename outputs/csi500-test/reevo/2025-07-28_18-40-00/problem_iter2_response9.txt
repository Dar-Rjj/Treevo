```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price, capturing the price movement over the day.
    daily_price_change = df['close'] - df['open']
    
    # Calculate the average of high and low, representing a midpoint of the price range for the day.
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Measure the activity level using volume, adjusted by the average of high and low prices.
    normalized_volume = df['volume'] / avg_high_low
    
    # Calculate the volatility using the standard deviation of the daily price change over the past 5 days.
    volatility = df['close'].rolling(window=5).std()
    
    # Create an alpha factor considering the price change scaled by the normalized volume and adjusted by volatility.
    # This attempts to capture the impact of trading volume on price movements while accounting for market volatility.
    alpha_factor = (daily_price_change * normalized_volume) / volatility
    
    return alpha_factor
```
