```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, this represents the daily trading range
    daily_range = df['high'] - df['low']
    
    # Calculate the difference between close and the previous day's close, representing the daily price change
    daily_change = df['close'] - df['close'].shift(1)
    
    # Compute the ratio of daily price change to daily trading range,
    # adding 1e-7 to avoid division by zero, which is a simple volatility-adjusted momentum factor
    volatility_adjusted_momentum = daily_change / (daily_range + 1e-7)
    
    return volatility_adjusted_momentum
```
