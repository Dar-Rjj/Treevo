```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive momentum as the exponential moving average (EMA) of log returns over different timeframes
    log_returns = np.log(df['close'] / df['close'].shift(1))
    ema_short = log_returns.ewm(span=10, adjust=False).mean()
    ema_long = log_returns.ewm(span=30, adjust=False).mean()
    adaptive_momentum = ema_short - ema_long

    # Calculate dynamic volatility as the standard deviation of log returns over a rolling window
    dynamic_volatility = log_returns.rolling(window=20).std()

    # Calculate normalized volume as the current volume divided by the 20-day moving average of volume
    normalized_volume = df['volume'] / df['volume'].rolling(window=20).mean()

    # Combine the three factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.5 * adaptive_momentum + 0.3 * dynamic_volatility + 0.2 * normalized_volume

    return factor_values
```
