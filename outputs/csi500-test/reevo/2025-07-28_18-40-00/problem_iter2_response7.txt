```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day price momentum
    price_momentum = df['close'].pct_change(periods=5)
    
    # Calculate the 5-day average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day volume trend
    volume_trend = df['volume'].pct_change(periods=5)
    
    # Calculate the 10-day volatility (standard deviation of daily returns)
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Combine price momentum, volume trend, and inverse of volatility to create the alpha factor
    factor_values = price_momentum * volume_trend / (volatility + 1e-7)
    
    return factor_values
```
