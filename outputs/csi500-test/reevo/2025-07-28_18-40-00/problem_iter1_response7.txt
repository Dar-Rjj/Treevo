```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open price
    price_diff = df['close'] - df['open']
    
    # Calculate the average volume over the past 5 days
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the weighted momentum factor by multiplying the price difference with the average volume
    weighted_momentum = price_diff * (avg_vol_5d / df['volume'])
    
    return weighted_momentum
```
