```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df: pd.DataFrame, macroeconomic_df: pd.DataFrame) -> pd.Series:
    # Calculate the price trend strength as the change in closing price over the last 5 days
    price_trend_strength = df['close'].pct_change(periods=5)
    
    # Calculate the money flow index as the average of (high - low) * volume over the last 3 days
    money_flow_index = ((df['high'] - df['low']) * df['volume']).rolling(window=3).mean()
    
    # Calculate the market volatility as the standard deviation of the log returns over the last 10 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    market_volatility = log_returns.rolling(window=10).std()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the accumulation/distribution line (A/D Line)
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    ad_line = ad_line.cumsum()
    
    # Calculate the On-Balance Volume (OBV)
    obv = np.where(df['close'] > df['close'].shift(1), df['volume'],
                   np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    
    # Calculate the Chaikin Oscillator
    chaikin_oscillator = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    chaikin_oscillator = chaikin_oscillator.ewm(span=3, adjust=False).mean() - \
                         chaikin_oscillator.ewm(span=10, adjust=False).mean()
    
    # Calculate the recent trend strength as the change in closing price over the last 5 days, with a focus on recent data
    recent_trend_strength = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    
    # Include macroeconomic indicators
    gdp_growth = macroeconomic_df['GDP Growth']
    inflation_rate = macroeconomic_df['Inflation Rate']
    interest_rate = macroeconomic_df['Interest Rate']
    
    # Combine all factors and macroeconomic indicators into a DataFrame for machine learning
    factors_df = pd.DataFrame({
        'Price Trend Strength': price_trend_strength,
        'Money Flow Index': money_flow_index,
        'Market Volatility': market_volatility,
        'RSI': rsi,
        'AD Line': ad_line,
        'OBV': obv,
        'Chaikin Oscillator': chaikin_oscillator,
        'Recent Trend Strength': recent_trend_strength,
        'GDP Growth': gdp_growth,
        'Inflation Rate': inflation_rate,
        'Interest Rate': interest_rate
    })
    
    # Drop NaN values
    factors_df = factors_df.dropna()
    
    # Prepare the target variable (future returns)
    future_returns = df['close'].pct_change().shift(-1).dropna()
    common_index = factors_df.index.intersection(future_returns.index)
    X = factors_df.loc[common_index]
    y = future_returns.loc[common_index]
    
    # Train a linear regression model to optimize weights
    model = LinearRegression()
    model.fit(X, y)
    weights = model.coef_
    
    # Calculate the factor values using optimized weights
    factor_values = (X * weights).sum(axis=1)
    
    return factor_values
```
