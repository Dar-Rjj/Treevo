def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive momentum as the change in closing price over the last 10, 20, and 30 days
    short_momentum = df['close'].pct_change(periods=10)
    medium_momentum = df['close'].pct_change(periods=20)
    long_momentum = df['close'].pct_change(periods=30)

    # Combine the different momentum terms into a single adaptive momentum factor
    adaptive_momentum = (short_momentum + medium_momentum + long_momentum) / 3

    # Calculate dynamic volatility as the standard deviation of log returns over the last 20 days
    log_returns = np.log(df['close']).diff()
    dynamic_volatility = log_returns.rolling(window=20).std()

    # Calculate normalized volume as the ratio of current volume to the 50-day average volume
    avg_volume = df['volume'].rolling(window=50).mean()
    normalized_volume = df['volume'] / avg_volume

    # Combine the adaptive momentum, dynamic volatility, and normalized volume into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.5 * adaptive_momentum - 0.3 * dynamic_volatility + 0.2 * normalized_volume

    return factor_values
