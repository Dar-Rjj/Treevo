def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum considering both short and long-term trends with exponential smoothing
    short_term_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].pct_change(periods=30).ewm(span=30, adjust=False).mean()
    adaptive_momentum = 0.6 * short_term_momentum + 0.4 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over the last 14 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std().ewm(span=14, adjust=False).mean()

    # Volume trend as a proxy for liquidity and market interest
    volume_trend = df['volume'].pct_change(periods=5).ewm(span=5, adjust=False).mean()

    # Market sentiment using the ratio of high to low prices as a proxy
    market_sentiment = (df['high'] - df['low']) / df['close']

    # Liquidity factor using the ratio of volume to amount
    liquidity_factor = df['volume'] / df['amount']

    # Sector-specific insight: using the close price relative to the 20-day moving average
    moving_average_20 = df['close'].rolling(window=20).mean()
    sector_insight = df['close'] / moving_average_20 - 1

    # Simple macroeconomic indicator: using the 3-month change in the 'close' price
    macroeconomic_indicator = df['close'].pct_change(periods=90)

    # Interaction terms between factors
    interaction_1 = adaptive_momentum * dynamic_volatility
    interaction_2 = volume_trend * market_sentiment
    interaction_3 = liquidity_factor * sector_insight
    interaction_4 = macroeconomic_indicator * adaptive_momentum

    # Non-linear combination of factors
    non_linear_combination = (
        (adaptive_momentum ** 2)
        - (dynamic_volatility ** 2)
        + (volume_trend ** 2)
        + (market_sentiment ** 2)
        + (liquidity_factor ** 2)
        + (sector_insight ** 2)
        + (macroeconomic_indicator ** 2)
        + (interaction_1 ** 2)
        + (interaction_2 ** 2)
        + (interaction_3 ** 2)
        + (interaction_4 ** 2)
    )

    # Adaptive weights based on the recent performance of each factor
    recent_performance = df['close'].pct_change(periods=5)
    adaptive_weights = {
        'adaptive_momentum': (recent_performance * adaptive_momentum).mean(),
        'dynamic_volatility': (recent_performance * -dynamic_volatility).mean(),
        'volume_trend': (recent_performance * volume_trend).mean(),
        'market_sentiment': (recent_performance * market_sentiment).mean(),
        'liquidity_factor': (recent_performance * liquidity_factor).mean(),
        'sector_insight': (recent_performance * sector_insight).mean(),
        'macroeconomic_indicator': (recent_performance * macroeconomic_indicator).mean(),
        'interaction_1': (recent_performance * interaction_1).mean(),
        'interaction_2': (recent_performance * interaction_2).mean(),
        'interaction_3': (recent_performance * interaction_3).mean(),
        'interaction_4': (recent_performance * interaction_4).mean(),
    }

    # Normalize the adaptive weights
    total_weight = sum(adaptive_weights.values())
    for key in adaptive_weights:
        adaptive_weights[key] /= total_weight

    # Combining the factors into a single alpha factor with adaptive weights
    factor_values = (
        adaptive_weights['adaptive_momentum'] * adaptive_momentum
        - adaptive_weights['dynamic_volatility'] * dynamic_volatility
        + adaptive_weights['volume_trend'] * volume_trend
        + adaptive_weights['market_sentiment'] * market_sentiment
        + adaptive_weights['liquidity_factor'] * liquidity_factor
        + adaptive_weights['sector_insight'] * sector_insight
        + adaptive_weights['macroeconomic_indicator'] * macroeconomic_indicator
        + adaptive_weights['interaction_1'] * interaction_1
        + adaptive_weights['interaction_2'] * interaction_2
        + adaptive_weights['interaction_3'] * interaction_3
        + adaptive_weights['interaction_4'] * interaction_4
    ) + 0.1 * non_linear_combination

    return factor_values
