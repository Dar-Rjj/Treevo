```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum considering both short and long-term trends with adaptive windows
    short_term_momentum = df['close'].pct_change(periods=5).rolling(window=10, min_periods=1).mean()
    long_term_momentum = df['close'].pct_change(periods=30).rolling(window=20, min_periods=1).mean()
    adaptive_momentum = 0.7 * short_term_momentum + 0.3 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over an adaptive window
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14, min_periods=1).std()

    # Normalized volume to account for varying liquidity conditions
    normalized_volume = (df['volume'] - df['volume'].expanding(min_periods=1).mean()) / df['volume'].expanding(min_periods=1).std()

    # Market sentiment using the average of the last 5 days' close prices
    market_sentiment = df['close'].rolling(window=5, min_periods=1).mean()

    # Incorporating factor diversification
    price_oscillator = (df['close'].rolling(window=5, min_periods=1).mean() - df['close'].rolling(window=10, min_periods=1).mean())

    # Combining the factors into a single alpha factor
    # Weights are adjusted based on their perceived importance
    factor_values = (
        0.4 * adaptive_momentum -
        0.2 * dynamic_volatility +
        0.2 * normalized_volume +
        0.1 * market_sentiment +
        0.1 * price_oscillator
    )

    return factor_values
```
