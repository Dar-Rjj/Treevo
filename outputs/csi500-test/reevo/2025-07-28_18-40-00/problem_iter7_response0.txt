```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive momentum using exponential weighted moving average (EWMA)
    adaptive_momentum = df['close'].ewm(span=10, adjust=False).mean().pct_change(periods=10)
    
    # Calculate the dynamic volatility as the standard deviation of log returns over the last 20 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=20).std()
    
    # Normalized volume factor: current volume divided by the average volume over the past 30 days
    normalized_volume = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Generate a market sentiment score, assuming a simple positive correlation with price change
    # This is represented here as the sum of changes in high and low prices
    market_sentiment = (df['high'] - df['low']).pct_change(periods=5)
    
    # Combine the factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = (
        0.4 * adaptive_momentum + 
        0.3 * dynamic_volatility + 
        0.2 * normalized_volume +
        0.1 * market_sentiment
    )

    return factor_values
```
