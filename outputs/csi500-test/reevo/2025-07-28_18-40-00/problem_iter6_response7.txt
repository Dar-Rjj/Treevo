```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price momentum as the change in closing price over an adaptive window (last 10 to 20 days)
    short_momentum = df['close'].pct_change(periods=10)
    long_momentum = df['close'].pct_change(periods=20)
    adaptive_momentum = (short_momentum + long_momentum) / 2

    # Calculate the money flow ratio as the average of (high - low) * volume over the last 5 to 10 days
    # This aims to capture the strength of money inflow or outflow over recent trading days
    money_flow_ratio_5d = ((df['high'] - df['low']) * df['volume']).rolling(window=5).mean()
    money_flow_ratio_10d = ((df['high'] - df['low']) * df['volume']).rolling(window=10).mean()
    money_flow_ratio = (money_flow_ratio_5d + money_flow_ratio_10d) / 2

    # Calculate the volatility as the standard deviation of the last 20 to 30 days' closing prices
    volatility_20d = df['close'].rolling(window=20).std()
    volatility_30d = df['close'].rolling(window=30).std()
    volatility = (volatility_20d + volatility_30d) / 2

    # Incorporate a factor that considers the relationship between volume and price change over the last 10 to 20 days
    volume_price_trend_10d = (df['volume'] * df['close'].diff()).rolling(window=10).sum()
    volume_price_trend_20d = (df['volume'] * df['close'].diff()).rolling(window=20).sum()
    volume_price_trend = (volume_price_trend_10d + volume_price_trend_20d) / 2

    # Combine the factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.4 * adaptive_momentum + 0.3 * money_flow_ratio + 0.2 * volatility + 0.1 * volume_price_trend

    return factor_values
```
