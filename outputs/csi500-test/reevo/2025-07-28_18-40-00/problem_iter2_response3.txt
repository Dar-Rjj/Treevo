```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the weighted average of high and low, using volume as the weight
    weighted_avg = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the distance from the close price to the weighted average
    dist_to_weighted_avg = abs(df['close'] - weighted_avg)
    
    # Calculate the 5-day volatility of the close price
    close_volatility_5d = df['close'].rolling(window=5).std()
    
    # Calculate the 5-day volatility of the volume
    volume_volatility_5d = df['volume'].rolling(window=5).std()
    
    # Calculate the multi-day momentum as the difference between the 5-day and 10-day close price
    multi_day_momentum = df['close'].rolling(window=5).mean() - df['close'].rolling(window=10).mean()
    
    # Alpha factor: Combine the components into a single alpha factor
    alpha_factor = (diff_close_open / (dist_to_weighted_avg + 1e-7)) * (close_volatility_5d + 1e-7) * (volume_volatility_5d + 1e-7) * (multi_day_momentum + 1e-7)
    
    return alpha_factor
```
