```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum with a combination of short, medium, and long-term trends
    short_term_momentum = df['close'].pct_change(periods=5)
    medium_term_momentum = df['close'].pct_change(periods=20)
    long_term_momentum = df['close'].pct_change(periods=60)
    adaptive_momentum = 0.4 * short_term_momentum + 0.3 * medium_term_momentum + 0.3 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over the last 14 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std()

    # Normalized volume to account for varying liquidity conditions
    normalized_volume = (df['volume'] - df['volume'].mean()) / df['volume'].std()

    # Price range as a measure of market sentiment
    price_range = (df['high'] - df['low']) / df['close']

    # Combining the four factors into a single alpha factor
    # Weights are adjusted based on their perceived importance
    factor_values = 0.3 * adaptive_momentum - 0.2 * dynamic_volatility + 0.2 * normalized_volume + 0.3 * price_range

    return factor_values
```
