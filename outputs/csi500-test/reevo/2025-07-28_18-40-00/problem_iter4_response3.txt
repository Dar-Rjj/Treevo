```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price momentum as the change in closing price over the last 10 days
    price_momentum = df['close'].pct_change(periods=10)
    
    # Calculate the money flow ratio as the average of (high - low) * volume over the last 5 days
    # This aims to capture the strength of money inflow or outflow over recent trading days
    money_flow_ratio = ((df['high'] - df['low']) * df['volume']).rolling(window=5).mean()
    
    # Calculate a simple oscillator as the difference between 5-day and 10-day moving averages
    # This may highlight reversals or confirm trends
    short_ma = df['close'].rolling(window=5).mean()
    long_ma = df['close'].rolling(window=10).mean()
    simple_oscillator = short_ma - long_ma
    
    # Calculate the relative strength index (RSI) over a 14-day period
    # This measures the magnitude of recent price changes to evaluate overbought or oversold conditions
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the volatility as the standard deviation of the last 10 days' returns
    # This aims to capture the risk or uncertainty in the market
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the trend following factor as the difference between the 50-day and 200-day moving averages
    # This aims to capture the long-term trend direction
    short_trend = df['close'].rolling(window=50).mean()
    long_trend = df['close'].rolling(window=200).mean()
    trend_following = short_trend - long_trend
    
    # Combine the six factors into a single alpha factor
    # Weighted sum based on heuristic importance assigned to each component
    factor_values = 0.2 * price_momentum + 0.1 * money_flow_ratio + 0.2 * simple_oscillator + 0.2 * rsi - 0.1 * volatility + 0.2 * trend_following

    return factor_values
```
