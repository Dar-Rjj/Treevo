```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum with exponential smoothing, considering a blend of short and long-term trends
    short_term_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].pct_change(periods=30).ewm(span=30, adjust=False).mean()
    adaptive_momentum = 0.6 * short_term_momentum + 0.4 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over the last 14 days, smoothed
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std().ewm(span=14, adjust=False).mean()

    # Volume relative strength to capture changes in trading volume
    volume_relative_strength = (df['volume'] - df['volume'].rolling(window=20).min()) / (df['volume'].rolling(window=20).max() - df['volume'].rolling(window=20).min())

    # Market sentiment using the ratio of high to low prices as a proxy, adjusted for average daily range
    market_sentiment = (df['high'] - df['low']) / df['close'] * (df['high'].rolling(window=20).mean() - df['low'].rolling(window=20).mean()) / df['close']

    # Price-Volume Trend (PVT) to incorporate both price changes and volume
    pvt = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    pvt = pvt.cumsum()

    # Incorporate macroeconomic indicators: GDP growth rate (assuming it's available in the DataFrame)
    gdp_growth_rate = df['gdp_growth_rate'].fillna(method='ffill').fillna(method='bfill')

    # Adaptive weights based on recent performance
    recent_performance = df['close'].pct_change(periods=1).rolling(window=20).mean()
    adaptive_weights = 0.5 + 0.5 * np.tanh(recent_performance)

    # Combining the factors into a single alpha factor with adaptive weights and macroeconomic indicator
    factor_values = (
        adaptive_weights * 0.3 * adaptive_momentum
        - adaptive_weights * 0.2 * dynamic_volatility
        + adaptive_weights * 0.25 * volume_relative_strength
        + adaptive_weights * 0.15 * market_sentiment
        + adaptive_weights * 0.1 * pvt
        + 0.05 * gdp_growth_rate
    )

    return factor_values
```
