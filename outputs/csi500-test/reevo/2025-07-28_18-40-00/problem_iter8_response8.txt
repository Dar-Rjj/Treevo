```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day price momentum
    price_momentum = df['close'].pct_change(periods=10)
    
    # Calculate the money flow index (MFI) over a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_flow_sum = positive_flow.rolling(window=14).sum()
    negative_flow_sum = negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_flow_sum / negative_flow_sum)))
    
    # Calculate a simple oscillator as the difference between 5-day and 10-day moving averages
    short_ma = df['close'].rolling(window=5).mean()
    long_ma = df['close'].rolling(window=10).mean()
    simple_oscillator = short_ma - long_ma
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the dynamic volatility as the standard deviation of the last 10 days' returns
    dynamic_volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day average volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day volume trend
    volume_trend = df['volume'].pct_change(periods=10)
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_diff = macd - signal
    
    # Calculate the 50-day and 200-day moving averages
    ma_50d = df['close'].rolling(window=50).mean()
    ma_200d = df['close'].rolling(window=200).mean()
    
    # Calculate the long-term trend factor
    long_term_trend = (ma_50d - ma_200d) / ma_200d
    
    # Calculate the Chaikin Money Flow (CMF) over a 20-day period
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine the factors into a single alpha factor
    # Adjust weights based on heuristic importance and market conditions
    factor_values = (
        0.2 * price_momentum + 
        0.15 * mfi + 
        0.15 * simple_oscillator + 
        0.1 * rsi - 
        0.1 * dynamic_volatility + 
        0.1 * (avg_volume_10d * volume_trend) +
        0.1 * macd_diff +
        0.1 * long_term_trend +
        0.05 * cmf
    )
    
    return factor_values
```
