```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 60-day price momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    medium_term_momentum = df['close'].pct_change(periods=20)
    long_term_momentum = df['close'].pct_change(periods=60)
    
    # Calculate the 10-day average volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day and 30-day volume trends
    volume_trend_10d = df['volume'].pct_change(periods=10)
    volume_trend_30d = df['volume'].pct_change(periods=30)
    
    # Calculate the 30-day and 60-day volatility (standard deviation of daily returns)
    volatility_30d = df['close'].pct_change().rolling(window=30).std()
    volatility_60d = df['close'].pct_change().rolling(window=60).std()
    
    # Calculate the relative price movement (close - open) / open
    relative_price_movement = (df['close'] - df['open']) / df['open']
    
    # Calculate the 50-day and 200-day moving averages of the close price
    ma_50d = df['close'].rolling(window=50).mean()
    ma_200d = df['close'].rolling(window=200).mean()
    
    # Calculate the long-term trend indicator (difference between 50-day MA and 200-day MA)
    long_term_trend = ma_50d - ma_200d
    
    # Calculate the market microstructure effect (amount/volume ratio)
    microstructure_effect = df['amount'] / df['volume']
    
    # Calculate the 20-day price range (high - low) / open
    price_range = (df['high'] - df['low']) / df['open']
    
    # Calculate the 10-day and 30-day price changes (close - open) / open
    price_change_10d = (df['close'] - df['open']).pct_change(periods=10)
    price_change_30d = (df['close'] - df['open']).pct_change(periods=30)
    
    # Non-linear transformations
    non_linear_volatility = np.sqrt(volatility_30d)
    non_linear_price_change_30d = np.log(1 + np.abs(price_change_30d))
    
    # Combine multiple indicators using weighted combinations
    factor_values = (
        0.3 * (short_term_momentum + medium_term_momentum + long_term_momentum)
        + 0.2 * (volume_trend_10d + volume_trend_30d)
        - 0.2 * (non_linear_volatility + volatility_60d)
        + 0.1 * relative_price_movement
        + 0.1 * long_term_trend
        + 0.05 * price_range
        + 0.05 * (price_change_10d + price_change_30d)
        + 0.05 * microstructure_effect
    )
    
    return factor_values
```
