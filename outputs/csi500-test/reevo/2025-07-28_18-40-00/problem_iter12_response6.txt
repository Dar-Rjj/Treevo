```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive momentum considering both short and long-term trends with exponential smoothing
    short_term_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    long_term_momentum = df['close'].pct_change(periods=30).ewm(span=30, adjust=False).mean()
    adaptive_momentum = 0.6 * short_term_momentum + 0.4 * long_term_momentum

    # Dynamic volatility using the standard deviation of logarithmic returns over the last 14 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std()

    # Normalized volume to account for varying liquidity conditions
    normalized_volume = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()

    # Market sentiment using the ratio of the number of up days to the total number of days in the past 10 days
    up_days = (df['close'] > df['open']).rolling(window=10).sum()
    total_days = 10
    market_sentiment = up_days / total_days

    # Chaikin Money Flow (CMF) over a 20-day period for additional market sentiment
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Dynamic weights based on recent market conditions
    recent_trend_strength = df['close'].pct_change(periods=5).abs().ewm(span=5, adjust=False).mean()
    recent_volatility = log_returns.rolling(window=5).std().ewm(span=5, adjust=False).mean()

    # Adjusting weights dynamically
    momentum_weight = 0.3 + 0.1 * recent_trend_strength / recent_trend_strength.max()
    volatility_weight = 0.2 + 0.1 * recent_volatility / recent_volatility.max()
    volume_weight = 0.2 + 0.1 * (1 - recent_volatility / recent_volatility.max())
    sentiment_weight = 0.1 + 0.1 * (1 - recent_trend_strength / recent_trend_strength.max())
    cmf_weight = 0.2 + 0.1 * (1 - recent_volatility / recent_volatility.max())

    # Combining the five factors into a single alpha factor with dynamic weights
    factor_values = (
        momentum_weight * adaptive_momentum -
        volatility_weight * dynamic_volatility +
        volume_weight * normalized_volume +
        sentiment_weight * market_sentiment +
        cmf_weight * cmf
    )

    return factor_values
```
