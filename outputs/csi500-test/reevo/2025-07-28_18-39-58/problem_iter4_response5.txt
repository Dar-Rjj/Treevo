```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the short-term momentum as the difference between today's close and yesterday's close
    short_term_momentum = df['close'].diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the adaptive volatility using the 5-day rolling standard deviation of the daily returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Integrate longer-term trends by calculating 20-day momentum
    long_term_momentum = df['close'].diff(20)
    
    # Calculate a liquidity measure using the amount
    liquidity = df['amount'] / df['volume']
    
    # Combine short-term momentum, volume anomaly, long-term momentum, and normalize by the true range and volatility
    factor = (short_term_momentum * volume_ratio + long_term_momentum) / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    # Incorporate the liquidity measure
    factor = factor * (liquidity / (liquidity.rolling(window=5).mean() + 1e-7))
    
    return factor
```
