```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate a 5-day moving average of the high and low to smooth out short-term volatility
    avg_high_5d = df['high'].rolling(window=5).mean()
    avg_low_5d = df['low'].rolling(window=5).mean()
    
    # Calculate the smoothed true range
    smoothed_true_range = avg_high_5d - avg_low_5d
    
    # Calculate the ratio of today's volume to the 5-day moving average volume for anomaly detection
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the change in daily momentum over the last 5 days to incorporate multi-period analysis
    momentum_change_5d = daily_momentum.rolling(window=5).sum()
    
    # Calculate the 5-day moving average of the close price to smooth out short-term noise
    avg_close_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day moving average of the amount to smooth out short-term noise
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    
    # Calculate the liquidity ratio as the 5-day moving average of the amount divided by the 5-day moving average of the volume
    liquidity_ratio = avg_amount_5d / avg_volume_5d
    
    # Combine the smoothed momentum, volume anomaly, multi-period momentum, and liquidity metrics into an alpha factor
    factor = (momentum_change_5d * volume_ratio * liquidity_ratio) / (smoothed_true_range + 1e-7)
    
    return factor
```
