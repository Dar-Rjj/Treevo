```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Calculate the 5-day, 10-day, and 30-day momentum to capture different time frames
    short_term_momentum = df['close'].diff(5)
    medium_term_momentum = df['close'].diff(10)
    long_term_momentum = df['close'].diff(30)
    
    # Calculate the 20-day moving average of the close price to smooth out the signal
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Combine daily, short-term, medium-term, and long-term momentum with volume anomaly
    combined_momentum = (daily_momentum + short_term_momentum + medium_term_momentum + long_term_momentum) * volume_ratio
    
    # Normalize by the true range and volatility
    normalized_factor = combined_momentum / (true_range + 1e-7)
    normalized_factor = normalized_factor / (volatility + 1e-7)
    
    # Incorporate liquidity measure using the 20-day simple moving average
    liquidity_measure = df['amount'] / df['volume']
    liquidity_adjusted_factor = normalized_factor * (df['close'] / sma_20) * liquidity_measure
    
    return liquidity_adjusted_factor
```
