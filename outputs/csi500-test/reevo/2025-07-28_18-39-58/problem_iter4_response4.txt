```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and the 20-day moving average of close
    daily_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the 30-day moving average of volume to identify trading activity anomalies
    avg_volume_30d = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 30-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=30).std()
    
    # Adaptive weights based on the ratio of volume anomaly to volatility
    adaptive_weights = (volume_ratio / (volatility + 1e-7))
    
    # Consider long-term trend by calculating the 60-day moving average of close prices
    ma_close_60d = df['close'].rolling(window=60).mean()
    long_term_trend = df['close'] - ma_close_60d
    
    # Combine factors: daily momentum, volume anomaly, long-term trend, normalized by true range and volatility
    factor = (daily_momentum * adaptive_weights * long_term_trend) / ((true_range + 1e-7) * (volatility + 1e-7))
    
    return factor
```
