def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily logarithmic returns
    daily_log_returns = np.log(df['close']).diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day rolling standard deviation of the daily log returns to incorporate volatility
    volatility = daily_log_returns.rolling(window=5).std()
    
    # Calculate the 10-day momentum to capture medium-term trends
    medium_term_momentum = df['close'].diff(10)
    
    # Calculate the 20-day momentum to capture longer-term trends
    long_term_momentum = df['close'].diff(20)
    
    # Calculate the adaptive weights based on the ratio of short-term to long-term momentum
    adaptive_weights = (daily_log_returns.abs() / (long_term_momentum.abs() + 1e-7)).clip(0, 1)
    
    # Calculate a short-term trend factor using the 5-day moving average of the close price
    short_term_trend = df['close'].rolling(window=5).mean().diff(1)
    
    # Combine daily log returns, medium-term momentum, and long-term momentum, adaptively weighted
    combined_factor = (daily_log_returns * adaptive_weights) + (medium_term_momentum * (1 - adaptive_weights) * 0.5) + (long_term_momentum * (1 - adaptive_weights) * 0.5)
    
    # Incorporate market microstructure effects by considering the ratio of amount to volume
    avg_price = df['amount'] / df['volume']
    avg_price_change = avg_price.diff(1)
    
    # Final factor, normalized by the true range and volatility, and incorporating market microstructure effects
    factor = (combined_factor * volume_ratio + avg_price_change) / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    return factor
