```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, representing momentum
    momentum = df['close'].diff()
    
    # Calculate the volume weighted average price over the last 5 days
    vwap_5d = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the difference between high and low, and close and open
    range_diff = (df['high'] - df['low'])
    gap_diff = (df['close'] - df['open'])
    
    # Calculate a weighted moving average of volume
    volume_wma = df['volume'].rolling(window=5).apply(lambda x: (x * (np.arange(1, len(x) + 1))).sum() / (len(x) * (len(x) + 1) / 2), raw=True)
    
    # Calculate the amount weighted average price over the last 5 days
    awap_5d = (df['close'] * df['amount']).rolling(window=5).sum() / df['amount'].rolling(window=5).sum()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (
        (momentum / (vwap_5d - df['close'] + 1e-7)) *  # Momentum adjusted by the Volume Weighted Average Price over the last 5 days
        (gap_diff / (range_diff + 1e-7)) *             # Gap difference normalized by range difference
        (volume_wma / (awap_5d - df['close'] + 1e-7))   # Volume WMA adjusted by the Amount Weighted Average Price over the last 5 days
    )
    
    return alpha_factor
```
