```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor: Calculate the difference between today's close and the 5-day moving average of close prices
    ma_close_5d = df['close'].rolling(window=5).mean()
    momentum = df['close'] - ma_close_5d

    # Relative volume: Ratio of today's volume to the 10-day moving average volume, indicating significant trading activity
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / avg_volume_10d

    # True range: Use as a normalizing factor for robustness
    true_range = df['high'] - df['low']

    # Combine factors: Momentum adjusted by relative volume, normalized by true range
    alpha_factor = (momentum * relative_volume) / (true_range + 1e-7)

    return alpha_factor
```
