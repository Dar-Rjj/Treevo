```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the 5-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std().fillna(daily_returns.std())
    
    # Calculate the 20-day momentum to capture longer-term trends
    long_term_momentum = df['close'].diff(20)
    
    # Combine daily momentum, long-term momentum, and volume anomaly
    factor = (daily_momentum + long_term_momentum) * volume_ratio
    
    # Normalize by volatility
    factor = factor / (volatility + 1e-7)
    
    return factor
```
