```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate the 3-day moving average of volume to identify trading activity anomalies
    avg_volume_3d = df['volume'].rolling(window=3).mean()
    volume_ratio = df['volume'] / avg_volume_3d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Calculate the 7-day momentum to capture medium-term trends
    medium_term_momentum = df['close'].diff(7)
    
    # Calculate the 15-day momentum to capture longer-term trends
    long_term_momentum = df['close'].diff(15)
    
    # Calculate the adaptive weights based on the ratio of short-term to long-term momentum
    adaptive_weights = (daily_momentum.abs() / (long_term_momentum.abs() + 1e-7)).clip(0, 1)
    
    # Combine daily momentum, medium-term momentum, and long-term momentum, adaptively weighted
    combined_momentum = (daily_momentum * adaptive_weights) + (medium_term_momentum * (1 - adaptive_weights) * 0.6) + (long_term_momentum * (1 - adaptive_weights) * 0.4)
    
    # Liquidity factor
    liquidity_factor = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Incorporate the 5-day and 20-day simple moving average crossover for trend signals
    sma_5d = df['close'].rolling(window=5).mean()
    sma_20d = df['close'].rolling(window=20).mean()
    trend_signal = sma_5d - sma_20d
    
    # Final factor, normalized by the true range, volatility, and incorporating liquidity and trend signal
    factor = (combined_momentum * volume_ratio * liquidity_factor * trend_signal) / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    return factor
```
