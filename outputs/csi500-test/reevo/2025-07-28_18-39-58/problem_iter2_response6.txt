```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, capturing the momentum
    momentum = df['close'].diff()
    
    # Calculate the percentage change in volume from the previous day to detect unusual trading activity
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the financial impact as the product of volume change and amount
    financial_impact = volume_change * df['amount']
    
    # Adaptive weights based on the rolling standard deviation of momentum and financial impact
    momentum_std = momentum.rolling(window=5).std().fillna(momentum.std())
    financial_impact_std = financial_impact.rolling(window=5).std().fillna(financial_impact.std())
    
    total_std = momentum_std + financial_impact_std
    
    momentum_weight = momentum_std / total_std
    financial_impact_weight = financial_impact_std / total_std
    
    # Calculate a weighted sum of price and volume changes to capture significant market movements
    factor = momentum_weight * momentum + financial_impact_weight * financial_impact
    
    return factor
```
