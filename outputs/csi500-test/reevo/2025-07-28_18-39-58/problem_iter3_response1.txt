```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and the 5-day moving average of close
    avg_close_5d = df['close'].rolling(window=5).mean()
    daily_momentum = df['close'] - avg_close_5d
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 10-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # Combine momentum, volume anomaly, and normalize by the true range and volatility
    factor = (daily_momentum * volume_ratio) / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    # Introduce a longer-term trend component by calculating the 20-day moving average of the close price
    long_term_trend = df['close'].rolling(window=20).mean()
    long_term_trend_diff = df['close'] - long_term_trend
    
    # Combine the long-term trend with the existing factor
    factor = factor * (1 + long_term_trend_diff / (df['close'] + 1e-7))
    
    return factor
```
