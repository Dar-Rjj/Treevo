```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and the 5-day moving average of close
    daily_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the 10-day moving average of volume to identify trading activity anomalies
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / (avg_volume_10d + 1e-7)
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 10-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # Adaptive weights based on the ratio of volume anomaly to volatility
    adaptive_weights = (volume_ratio / (volatility + 1e-7))
    
    # Combine momentum, volume anomaly, and normalize by the true range and volatility
    factor = (daily_momentum * adaptive_weights) / (true_range + 1e-7)
    
    # Incorporate a long-term trend component by calculating the 30-day moving average of the close price
    long_term_trend = df['close'].rolling(window=30).mean()
    trend_factor = (df['close'] - long_term_trend) / (long_term_trend + 1e-7)
    
    # Combine the short-term momentum and long-term trend factors
    final_factor = factor + 0.5 * trend_factor
    
    return final_factor
```
