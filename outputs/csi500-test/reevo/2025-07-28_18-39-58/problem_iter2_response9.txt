```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price, then scale it by the daily volume
    price_diff_scaled_by_volume = (df['close'] - df['open']) * df['volume']
    
    # Calculate a simple moving average of the scaled price difference over 10 days
    sma_10_day_price_diff_scaled = price_diff_scaled_by_volume.rolling(window=10).mean()
    
    # Calculate the Average True Range (ATR) as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_10_day = true_range.rolling(window=10).mean()
    
    # Combine the scaled price difference SMA with ATR to create a volatility-adjusted factor
    volatility_adjusted_factor = sma_10_day_price_diff_scaled / (atr_10_day + 1e-7)
    
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5_day = df['close'].rolling(window=5).mean()
    ma_20_day = df['close'].rolling(window=20).mean()
    
    # Calculate the moving average convergence divergence (MACD) for trend following
    macd = ma_5_day - ma_20_day
    
    # Generate a final factor that combines the volatility-adjusted factor with the MACD
    final_factor = volatility_adjusted_factor * macd
    
    return final_factor
```
