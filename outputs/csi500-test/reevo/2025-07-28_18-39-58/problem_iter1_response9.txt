```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a rolling mean of volume to capture the trend in trading activity
    df['rolling_volume_mean'] = df['volume'].rolling(window=5).mean()

    # Calculate the ratio of current volume to the rolling volume mean
    df['volume_trend_ratio'] = df['volume'] / (df['rolling_volume_mean'] + 1e-7)

    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()

    # Generate a factor that combines the volume trend and the momentum in close prices
    factor = df['volume_trend_ratio'] * df['close_pct_change']

    return factor
```
