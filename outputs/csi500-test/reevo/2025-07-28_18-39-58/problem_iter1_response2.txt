```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices for a true range
    true_range = df['high'] - df['low']
    
    # Calculate the average true range over a 14-day window
    avg_true_range = true_range.rolling(window=14).mean()
    
    # Calculate price momentum as the percentage change in closing price over 5 days
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the weighted volume, where weights are the percentage change of close price
    weighted_volume = df['volume'] * (df['close'].pct_change() + 1)

    # Combine the average true range, momentum, and weighted volume into an alpha factor
    alpha_factor = (momentum / avg_true_range) * weighted_volume

    return alpha_factor
```
