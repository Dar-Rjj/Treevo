```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, indicating the momentum
    daily_momentum = df['close'].diff(1)
    # Calculate the ratio of today's volume to the 5-day moving average volume, indicating trading activity anomaly
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    # Combine momentum with volume anomaly, normalizing by the true range (high - low) for added robustness
    factor = (daily_momentum * volume_ratio) / (df['high'] - df['low'] + 1e-7)
    return factor
```
