```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price
    df['price_diff'] = df['close'].diff().fillna(0)
    
    # Calculate the average trading volume over the last 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean().fillna(0)
    
    # Calculate the ratio of today's volume to the 10-day average volume
    df['volume_ratio'] = df['volume'] / (df['avg_volume_10d'] + 1e-7)
    
    # Calculate the positive and negative components of the 10-day price momentum
    df['pos_momentum'] = df['price_diff'].rolling(window=10).apply(lambda x: (x > 0).sum(), raw=True)
    df['neg_momentum'] = df['price_diff'].rolling(window=10).apply(lambda x: (x < 0).sum(), raw=True)
    
    # Incorporate longer-term trends by calculating the 30-day moving average of the close price
    df['long_term_trend'] = df['close'].rolling(window=30).mean().fillna(0)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    df['gain'] = df['price_diff'].apply(lambda x: max(x, 0))
    df['loss'] = df['price_diff'].apply(lambda x: -min(x, 0))
    df['avg_gain'] = df['gain'].rolling(window=14).mean().fillna(0)
    df['avg_loss'] = df['loss'].rolling(window=14).mean().fillna(0)
    df['rsi'] = 100 - (100 / (1 + (df['avg_gain'] / (df['avg_loss'] + 1e-7))))
    
    # Calculate the alpha factor as a weighted combination of the momentum, volume ratio, long-term trend, and RSI
    df['alpha_factor'] = (
        0.3 * (df['pos_momentum'] - df['neg_momentum']) + 
        0.3 * df['volume_ratio'] + 
        0.2 * (df['close'] - df['long_term_trend']) / df['long_term_trend'] +
        0.2 * (50 - df['rsi'])
    )
    
    return df['alpha_factor']
```
