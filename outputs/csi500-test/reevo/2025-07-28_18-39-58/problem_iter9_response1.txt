```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily log returns for a more symmetric distribution
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day rolling standard deviation of the daily log returns to incorporate volatility
    volatility = daily_log_returns.rolling(window=5).std()
    
    # Calculate the 10-day and 20-day momentum in log returns to capture medium and longer-term trends
    medium_term_momentum = np.log(df['close']) - np.log(df['close'].shift(10))
    long_term_momentum = np.log(df['close']) - np.log(df['close'].shift(20))
    
    # Calculate the adaptive weights based on the ratio of short-term to long-term momentum
    adaptive_weights = (daily_log_returns.abs() / (long_term_momentum.abs() + 1e-7)).clip(0, 1)
    
    # Combine daily, medium-term, and long-term momentum, adaptively weighted
    combined_momentum = (daily_log_returns * adaptive_weights) + (medium_term_momentum * (1 - adaptive_weights) * 0.5) + (long_term_momentum * (1 - adaptive_weights) * 0.5)
    
    # Final factor, normalized by the true range and volatility
    factor = (combined_momentum * volume_ratio) / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    return factor
```
