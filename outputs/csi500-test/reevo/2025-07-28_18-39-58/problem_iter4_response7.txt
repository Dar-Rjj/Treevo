def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor: Calculate the difference between today's close and the 10-day moving average of close prices
    ma_close_10d = df['close'].rolling(window=10).mean()
    momentum = df['close'] - ma_close_10d

    # Relative volume: Ratio of today's volume to the 20-day moving average volume, indicating significant trading activity
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20d

    # True range: Use as a volatility measure
    true_range = df['high'] - df['low']

    # Average true range over 14 days for normalizing true range
    atr_14d = true_range.rolling(window=14).mean()

    # Liquidity: Calculate the dollar volume (amount * volume) and its 30-day moving average
    dollar_volume = df['amount'] * df['volume']
    avg_dollar_volume_30d = dollar_volume.rolling(window=30).mean()

    # Liquidity ratio: Today's dollar volume divided by the 30-day moving average dollar volume
    liquidity_ratio = dollar_volume / avg_dollar_volume_30d

    # Trend strength: Moving average convergence divergence (MACD) as a trend following indicator
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal_line = macd.rolling(window=9).mean()
    trend_strength = macd - signal_line

    # Combine factors: Momentum adjusted by relative volume and liquidity, normalized by average true range, and considering trend strength
    alpha_factor = (momentum * relative_volume * liquidity_ratio * trend_strength) / (atr_14d + 1e-7)

    return alpha_factor
