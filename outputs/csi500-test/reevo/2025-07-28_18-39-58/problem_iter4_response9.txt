```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low)
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day rolling standard deviation of the daily returns to incorporate volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Calculate the 30-day momentum to capture longer-term trends
    long_term_momentum = df['close'].diff(30)
    
    # Combine daily momentum, long-term momentum, and volume anomaly
    # Weight the daily momentum more heavily to focus on short-term movements
    factor = (daily_momentum * 0.8 + long_term_momentum * 0.2) * volume_ratio
    
    # Use the true range and volatility to create a combined normalization factor
    normalization_factor = (true_range + volatility + 1e-7)
    
    # Normalize the factor
    factor = factor / normalization_factor
    
    return factor
```
