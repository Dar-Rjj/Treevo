```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average of high and low, then subtract it from close price to get a sort of volatility
    avg_high_low = (df['high'] + df['low']) / 2
    close_to_avg_diff = df['close'] - avg_high_low

    # Use the absolute value of the difference as a proxy for volatility
    volatility_proxy = abs(close_to_avg_diff)

    # Calculate a weighted momentum using the amount and close prices
    weighted_momentum = (df['close'] * df['amount']).rolling(window=5).mean().fillna(0)

    # Calculate an estimate of order book depth using the ratio of volume to the range (high - low)
    order_book_depth = df['volume'] / (df['high'] - df['low'] + 1e-7)

    # Alpha factor: Ratio of the weighted momentum adjusted by the volatility proxy to the order book depth
    alpha_factor = (weighted_momentum / (volatility_proxy + 1e-7)) / (order_book_depth + 1e-7)

    return alpha_factor
```
