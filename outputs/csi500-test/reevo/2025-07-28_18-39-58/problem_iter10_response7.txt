```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily momentum as the difference between today's close and yesterday's close
    daily_momentum = df['close'].diff(1)
    
    # Calculate the 5-day moving average of volume to identify trading activity anomalies
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Calculate the true range (high - low) for normalization
    true_range = df['high'] - df['low']
    
    # Calculate the logarithmic returns for better volatility and trend analysis
    log_returns = np.log(df['close']).diff(1)
    
    # Calculate the 5-day rolling standard deviation of the logarithmic returns to incorporate volatility
    volatility = log_returns.rolling(window=5).std()
    
    # Calculate the 20-day momentum to capture longer-term trends
    long_term_momentum = df['close'].diff(20)
    
    # Calculate a short-term trend factor using the 5-day moving average of the close price
    short_term_trend = df['close'].rolling(window=5).mean().diff(1)
    
    # Calculate a medium-term trend factor using the 20-day moving average of the close price
    medium_term_trend = df['close'].rolling(window=20).mean().diff(1)
    
    # Incorporate market microstructure effects by considering the ratio of amount to volume
    avg_price = df['amount'] / df['volume']
    avg_price_change = avg_price.diff(1)
    
    # Combine daily momentum, long-term momentum, short-term trend, medium-term trend, and market microstructure effects
    # Normalize by the true range and volatility
    factor = (daily_momentum + long_term_momentum + short_term_trend + medium_term_trend + avg_price_change) * volume_ratio
    factor = factor / (true_range + 1e-7)
    factor = factor / (volatility + 1e-7)
    
    return factor
```
