```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    # Calculate the 20-day moving average of the closing price
    df['ma_20'] = df['close'].rolling(window=20).mean()
    # Generate a momentum signal based on the change in the close price and the relationship between the two moving averages
    factor = (df['close_diff'] > 0) * (df['ma_10'] > df['ma_20'])
    return factor.astype(int)
```
