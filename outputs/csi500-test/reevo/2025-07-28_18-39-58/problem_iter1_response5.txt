```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, and average it with the volume
    avg_diff_vol = ((df['high'] - df['low']) / 2) + df['volume']
    
    # Calculate the weighted momentum using the amount and close prices
    weighted_momentum = (df['close'] * df['amount']).rolling(window=5).mean().fillna(0)
    
    # Alpha factor: Ratio of the weighted momentum to the averaged difference and volume
    alpha_factor = weighted_momentum / (avg_diff_vol + 1e-7)
    
    return alpha_factor
```
