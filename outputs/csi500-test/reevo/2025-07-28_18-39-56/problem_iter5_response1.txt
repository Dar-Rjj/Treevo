```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the exponential moving average (EMA) of the VWAP with a dynamic window
    ema_vwap = vwap.ewm(span=7, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) of the close price with a window of 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the logarithmic returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the volatility as the standard deviation of the logarithmic returns with a window of 20 days
    volatility = log_returns.rolling(window=20).std()
    
    # Calculate the factor as a combination of EMA VWAP, RSI, and volatility
    factor = (df['close'] - ema_vwap) * rsi / volatility

    return factor
```
