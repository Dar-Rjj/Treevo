def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with an adaptive window
    vwap = (df['amount'] / df['volume']).rolling(window=7, min_periods=1).mean()

    # Calculate the adaptive exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=7, adjust=False).mean()

    # Calculate the adaptive relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate the adaptive momentum as the percentage change in the close price
    short_momentum = df['close'].pct_change(periods=5)
    long_momentum = df['close'].pct_change(periods=20)
    adaptive_momentum = (short_momentum + long_momentum) / 2

    # Introduce a volatility component using the standard deviation of log returns
    volatility = log_returns.rolling(window=21, min_periods=1).std()

    # Combine the factors into a single alpha factor
    # Incorporate the EMA, RSI, logarithmic returns, adaptive momentum, VWAP, and volatility
    alpha_factor = (df['close'] - ema_close) * rsi * log_returns * adaptive_momentum * (vwap / df['close']) / (1 + volatility)

    return alpha_factor
