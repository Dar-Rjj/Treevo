```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Smooth the buying pressure over 10 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=10).mean()
    
    # Calculate the volume-weighted average price (VWAP) over a 10-day period
    vwap = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the change in VWAP from the previous day
    vwap_change = vwap.diff(1)
    
    # Calculate the 10-day simple moving average of the close price
    sma_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the close price for volatility
    std_close = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day exponential moving average of the close price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of the current close price to its 10-day EMA
    close_to_ema_ratio = df['close'] / ema_close
    
    # Calculate the 10-day momentum of the close price
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the 10-day mean reversion factor
    mean_reversion = (df['close'] - sma_close) / std_close
    
    # Final alpha factor, which is the smoothed buying pressure divided by the daily range, adjusted by the change in VWAP,
    # and further adjusted by the ratio of the current close price to its 10-day EMA, the 10-day standard deviation, the 10-day momentum,
    # and the 10-day mean reversion factor
    alpha_factor = (smooth_buying_pressure / daily_range) * vwap_change * close_to_ema_ratio * (1 + momentum) * (1 - mean_reversion)
    
    return alpha_factor
```
