```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=7).mean()
    
    # Calculate the exponential moving average of the close price with a window of 7 days
    ema_close = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate the momentum of the close price with a window of 5 days
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the logarithmic return of the close price
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Integrate VWAP, EMA, and logarithmic returns
    factor = (df['close'] - vwap) * (ema_close - df['close']) * log_return * momentum

    return factor
```
