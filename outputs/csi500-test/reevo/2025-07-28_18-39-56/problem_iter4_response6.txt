```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with a shorter window
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the exponential moving average of the close price with a shorter window
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate short-term momentum with a 3-day period
    momentum_short = df['close'].pct_change(periods=3)
    
    # Calculate long-term momentum with a 10-day period
    momentum_long = df['close'].pct_change(periods=10)
    
    # Calculate the logarithmic return of the close price
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the average true range (ATR) as a measure of volatility
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift(1))
    tr3 = np.abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = tr.rolling(window=5).mean()
    
    # Integrate VWAP, EMA, logarithmic returns, and momentum measures
    factor = (df['close'] - vwap) * (ema_close - df['close']) * log_return * (momentum_short + momentum_long) * atr

    return factor
```
