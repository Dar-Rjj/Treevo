```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the smoothed price difference (close - open) using an exponential moving average (EMA)
    price_diff = df['close'] - df['open']
    ema_price_diff = price_diff.ewm(span=5, adjust=False).mean()
    
    # Calculate the relative change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the 10-day momentum of the close price
    momentum = df['close'].pct_change(periods=10).fillna(0)
    
    # Calculate the 14-day Relative Strength Index (RSI) to capture market sentiment
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (ema_price_diff * (1 + volume_change) * (1 + momentum) * rsi).fillna(0)
    
    return alpha_factor
```
