```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price to capture short and long term trends
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Momentum factor: the difference between the 5-day and 20-day moving averages
    trend_momentum = ma_5 - ma_20
    
    # Liquidity factor: the ratio of current volume to the 10-day average volume, capturing liquidity anomalies
    avg_volume = df['volume'].rolling(window=10).mean()
    liquidity = df['volume'] / avg_volume
    
    # Price action context: the ratio of the difference between high and low to the open price
    # This provides a measure of daily price fluctuation relative to the opening price
    price_action = (df['high'] - df['low']) / df['open']
    
    # Combined alpha factor: incorporating trends, liquidity, and price action
    # The combination aims to reflect stocks with strong momentum, high liquidity, and significant daily movement
    alpha_factor = (trend_momentum * liquidity * price_action).fillna(0)
    
    return alpha_factor
```
