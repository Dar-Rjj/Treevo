```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, which represents the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure by looking at the amount of money that moved the price from open to close
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Smooth the buying pressure over 3 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=3).mean()
    
    # Final alpha factor, which is the smoothed buying pressure divided by the daily range
    # This attempts to capture the intensity of buying or selling normalized by the daily volatility
    alpha_factor = smooth_buying_pressure / daily_range
    
    return alpha_factor
```
