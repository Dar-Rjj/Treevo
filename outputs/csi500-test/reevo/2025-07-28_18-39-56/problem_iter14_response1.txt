```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with a dynamic window
    ema_close = df['close'].ewm(span=7, adjust=False).mean()

    # Calculate the adaptive relative strength index (RSI) with a dynamic window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate the adaptive momentum as the percentage change in the close price
    # with a dynamic window to capture short-term and long-term trends
    short_momentum = df['close'].pct_change(periods=5)
    long_momentum = df['close'].pct_change(periods=20)
    adaptive_momentum = (short_momentum + long_momentum) / 2

    # Calculate the volume-weighted average price (VWAP) with an adaptive window
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Calculate the trend factor as the difference between the close price and VWAP
    trend_factor = df['close'] - vwap

    # Calculate the 10-day standard deviation of the close price for volatility
    std_close_10 = df['close'].rolling(window=10).std()

    # Normalize the factors to avoid large values from dominating the alpha factor
    norm_trend_factor = trend_factor / std_close_10
    norm_ema_close = (df['close'] - ema_close) / std_close_10
    norm_rsi = rsi / 50
    norm_log_returns = log_returns
    norm_adaptive_momentum = adaptive_momentum

    # Combine the normalized factors into a single alpha factor
    # Incorporate the EMA, RSI, logarithmic returns, adaptive momentum, and trend factor
    alpha_factor = (norm_ema_close * norm_rsi * norm_log_returns * norm_adaptive_momentum * norm_trend_factor)

    return alpha_factor
```
