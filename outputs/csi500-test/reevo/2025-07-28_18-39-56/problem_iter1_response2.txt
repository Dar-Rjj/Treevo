```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, normalized by volume
    price_diff = (df['close'] - df['open']) / (df['volume'] + 1e-7)
    
    # Calculate the relative change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the weighted moving average of the price difference
    wma_price_diff = price_diff.rolling(window=5).mean()
    
    # Calculate the factor as a combination of price difference weighted moving average and volume change
    factor = wma_price_diff * (1 + volume_change)
    
    return factor
```
