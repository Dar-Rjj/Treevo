def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with a dynamic window
    vwap = (df['amount'] / df['volume']).rolling(window=7, min_periods=1).mean()

    # Calculate the adaptive exponential moving average (EMA) of the close price
    ema_close_short = df['close'].ewm(span=7, adjust=False).mean()
    ema_close_long = df['close'].ewm(span=50, adjust=False).mean()

    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14, min_periods=1).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14, min_periods=1).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Calculate logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate a smoothed momentum component using EMA over a 21-day period
    momentum = df['close'].pct_change(periods=21).ewm(span=21, adjust=False).mean()

    # Short-term factor: difference between close and short EMA, scaled by RSI, multiplied by log returns, and adjusted by momentum
    short_term_factor = (df['close'] - ema_close_short) * rsi * log_returns * (vwap / df['close']) * (1 + momentum)

    # Long-term trend component using a 50-day EMA
    long_term_trend = (df['close'] / ema_close_long)

    # Combine the short-term and long-term factors
    alpha_factor = short_term_factor * long_term_trend

    # Add a volatility component using the standard deviation of log returns over a 20-day period
    volatility = log_returns.rolling(window=20, min_periods=1).std()

    # Incorporate market microstructure by including the ratio of high to low prices
    microstructure_ratio = df['high'] / df['low']

    # Combine all components to form the final alpha factor
    final_alpha_factor = alpha_factor * (1 + volatility) * microstructure_ratio

    # Dynamic window adjustment for more adaptive and robust alpha factor generation
    vol_adjustment = 1 + (volatility - volatility.mean()) / volatility.std()
    final_alpha_factor = final_alpha_factor * vol_adjustment

    # Mean reversion component using the distance from the 20-day EMA
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    mean_reversion = (ema_close_20 - df['close']) / df['close']

    # Combine the mean reversion component with the final alpha factor
    final_alpha_factor = final_alpha_factor * (1 + mean_reversion)

    return final_alpha_factor
