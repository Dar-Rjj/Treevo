```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the exponential moving average of the VWAP with a dynamic window size
    ema_vwap = vwap.ewm(span=7, adjust=False).mean()
    
    # Calculate the logarithmic returns for the close price
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the momentum of the close price with a dynamic window size
    momentum_window = 5
    momentum = df['close'].pct_change(periods=momentum_window)
    
    # Calculate the liquidity as the ratio of volume to the difference between high and low
    liquidity = df['volume'] / (df['high'] - df['low'])
    
    # Calculate the Z-score of the close price over a rolling window to capture mean reversion
    z_score = (df['close'] - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).std()
    
    # Combine the factors into a single alpha factor
    # Incorporate liquidity, trend, and sentiment
    alpha_factor = (df['close'] - ema_vwap) * momentum * log_returns * liquidity * z_score
    
    return alpha_factor
```
