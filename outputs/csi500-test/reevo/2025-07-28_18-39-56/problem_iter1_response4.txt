```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, capturing volatility.
    volatility = df['high'] - df['low']
    
    # Calculate the percentage change in close price from the previous day, capturing momentum.
    momentum = df['close'].pct_change().fillna(0)
    
    # Calculate the ratio of current volume to the 10-day average volume, capturing volume anomaly.
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Combine the factors into a single alpha factor
    alpha_factor = (volatility * momentum * volume_ratio).fillna(0)
    
    return alpha_factor
```
