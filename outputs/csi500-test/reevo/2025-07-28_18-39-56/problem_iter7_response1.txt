```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with an adaptive window
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Calculate the exponential moving average (EMA) of the close price with a dynamic window
    ema_close = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the relative strength index (RSI) with a dynamic window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate the dynamic volatility using standard deviation of the logarithmic returns
    volatility = log_returns.rolling(window=5).std()

    # Calculate sentiment as the ratio of close to VWAP
    sentiment = df['close'] / vwap

    # Combine the factors: (Close - EMA) * RSI * Log Returns * Volatility * Sentiment
    factor = (df['close'] - ema_close) * rsi * log_returns * volatility * sentiment

    return factor
```
