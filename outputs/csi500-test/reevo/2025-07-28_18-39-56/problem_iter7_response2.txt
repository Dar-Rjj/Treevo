def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with an adaptive window
    vwap_window = 7
    vwap = (df['amount'] / df['volume']).rolling(window=vwap_window).mean()

    # Calculate the exponential moving average (EMA) of the close price with a dynamic window
    ema_window = 7
    ema_close = df['close'].ewm(span=ema_window, adjust=False).mean()

    # Calculate the relative strength index (RSI) with a dynamic window
    rsi_window = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate dynamic volatility using standard deviation of log returns over a 20-day window
    vol_window = 20
    dynamic_volatility = log_returns.rolling(window=vol_window).std()

    # Calculate the factor as the difference between the close price and the VWAP,
    # scaled by the RSI, multiplied by the logarithmic returns, and divided by the dynamic volatility
    factor = ((df['close'] - vwap) * rsi * log_returns) / dynamic_volatility

    return factor
