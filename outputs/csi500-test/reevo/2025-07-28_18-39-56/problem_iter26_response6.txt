def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np

    # Calculate the volume-weighted average price (VWAP) with a dynamic window
    vwap = (df['amount'] / df['volume']).rolling(window=7, min_periods=1).mean()

    # Calculate the adaptive exponential moving average (EMA) of the close price with a 7-day and 50-day window
    ema_close_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_close_50 = df['close'].ewm(span=50, adjust=False).mean()

    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14, min_periods=1).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14, min_periods=1).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate a smoothed momentum component using EMA over a 21-day period
    momentum = df['close'].pct_change(periods=21).ewm(span=21, adjust=False).mean()

    # Calculate the factor as the difference between the close price and the EMA of close price
    # scaled by the RSI, multiplied by the logarithmic returns, and adjusted by the smoothed momentum
    # Also, include the VWAP to balance volume and price
    factor = (df['close'] - ema_close_7) * rsi * log_returns * (vwap / df['close']) * (1 + momentum)

    # Introduce a long-term trend component using a 50-day EMA
    long_term_trend = (df['close'] / ema_close_50)

    # Combine the short-term and long-term factors
    alpha_factor = factor * long_term_trend

    # Add a volatility component using the standard deviation of log returns over a 20-day period
    volatility = log_returns.rolling(window=20, min_periods=1).std()

    # Incorporate market microstructure by including the ratio of high to low prices
    microstructure_ratio = df['high'] / df['low']

    # Combine all components to form the final alpha factor
    final_alpha_factor = alpha_factor * (1 + volatility) * microstructure_ratio

    # Dynamic window adjustment for more adaptive and robust alpha factor generation
    vol_adjustment = 1 + (volatility - volatility.mean()) / volatility.std()
    final_alpha_factor = final_alpha_factor * vol_adjustment

    # Add a mean reversion component using the distance from the 20-day simple moving average (SMA)
    sma_close_20 = df['close'].rolling(window=20, min_periods=1).mean()
    mean_reversion = (sma_close_20 - df['close']) / df['close']
    final_alpha_factor = final_alpha_factor * (1 + mean_reversion)

    # Introduce a new component: the ratio of the current close price to the 20-day high
    high_20 = df['high'].rolling(window=20, min_periods=1).max()
    high_ratio = df['close'] / high_20
    final_alpha_factor = final_alpha_factor * high_ratio

    # Introduce a new component: the ratio of the current close price to the 20-day low
    low_20 = df['low'].rolling(window=20, min_periods=1).min()
    low_ratio = df['close'] / low_20
    final_alpha_factor = final_alpha_factor * low_ratio

    # Introduce a new component: the ratio of the current volume to the 20-day moving average of volume
    volume_20 = df['volume'].rolling(window=20, min_periods=1).mean()
    volume_ratio = df['volume'] / volume_20
    final_alpha_factor = final_alpha_factor * volume_ratio

    # Additional adaptive smoothing for the final alpha factor
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=5, adjust=False).mean()

    return final_alpha_factor_smoothed
