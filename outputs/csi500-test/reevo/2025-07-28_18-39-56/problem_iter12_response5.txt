def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, which represents the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure by looking at the amount of money that moved the price from open to close
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Smooth the buying pressure over 5 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=5).mean()
    
    # Calculate the volume-weighted average price (VWAP) over a 10-day period
    vwap = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the change in VWAP from the previous day
    vwap_change = vwap.diff(1)
    
    # Calculate the 10-day simple moving average of the close price
    sma_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 50-day simple moving average of the close price for long-term trend
    sma_close_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the 10-day exponential moving average of the close price
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI) to capture market sentiment
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 5-day momentum of the close price
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day standard deviation of the close price for volatility
    std_close_10 = df['close'].rolling(window=10).std()
    
    # Calculate the 50-day standard deviation of the close price for long-term volatility
    std_close_50 = df['close'].rolling(window=50).std()
    
    # Calculate the adaptive window for RSI based on the 10-day and 50-day standard deviations
    rsi_window = (std_close_10 / std_close_50).apply(lambda x: max(14, int(x * 14)))
    
    # Recalculate the 14-day RSI with the adaptive window
    rsi_adaptive = []
    for i in range(len(df)):
        if i < rsi_window.iloc[i]:
            rsi_adaptive.append(np.nan)
        else:
            delta_i = df['close'][i-rsi_window.iloc[i]+1:i+1].diff(1)
            gain_i = (delta_i.where(delta_i > 0, 0)).mean()
            loss_i = (-delta_i.where(delta_i < 0, 0)).mean()
            rs_i = gain_i / (loss_i + 1e-7)
            rsi_i = 100 - (100 / (1 + rs_i))
            rsi_adaptive.append(rsi_i)
    rsi_adaptive = pd.Series(rsi_adaptive, index=df.index)
    
    # Final alpha factor, which is the smoothed buying pressure divided by the daily range, adjusted by the change in VWAP,
    # further adjusted by the ratio of the current close price to its 10-day SMA, the adaptive RSI, and the momentum
    alpha_factor = (smooth_buying_pressure / daily_range) * vwap_change * (df['close'] / sma_close_10) * (rsi_adaptive / 100) * momentum
    
    return alpha_factor
