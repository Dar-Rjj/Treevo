```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the exponential moving average of the VWAP with a window of 7 days
    ema_vwap = vwap.ewm(span=7, adjust=False).mean()
    
    # Calculate the momentum of the close price with a window of 5 days
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the logarithmic returns of the close price
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the volatility as the standard deviation of the logarithmic returns over a 20-day window
    volatility = log_returns.rolling(window=20).std()
    
    # Calculate the liquidity as the ratio of the close price to the volume
    liquidity = df['close'] / df['volume']
    
    # Normalize the factor by the product of volatility and liquidity
    normalized_factor = (df['close'] - ema_vwap) * momentum / (volatility * liquidity)
    
    return normalized_factor
```
