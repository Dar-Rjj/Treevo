```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for a volatility factor
    volatility = df['high'] - df['low']
    
    # Calculate the average of open and close as a smoothed price representation
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the volume weighted average price
    vw_avg_price = (df['amount'] / df['volume'])
    
    # Factor that combines the volatility, smoothed price, and volume-weighted price
    alpha_factor = (volatility * (vw_avg_price - avg_price)) / (avg_price + 1e-7)
    
    return alpha_factor
```
