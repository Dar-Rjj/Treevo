```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) with a dynamic window
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the exponential moving average of the VWAP with a dynamic window
    ema_vwap = vwap.ewm(span=7, adjust=False).mean()
    
    # Calculate the momentum of the close price with a dynamic window
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the log return of the close price
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the 5-day simple moving average of the close price
    sma_close_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day simple moving average of the close price
    sma_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 50-day simple moving average of the close price
    sma_close_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the trend factor as the difference between the 5-day and 50-day SMAs
    trend_factor = sma_close_5 - sma_close_50
    
    # Calculate the volatility factor as the standard deviation of the log returns over a 10-day period
    volatility_factor = log_return.rolling(window=10).std()
    
    # Calculate the relative change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Smooth the buying pressure over 5 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=5).mean()
    
    # Calculate the order book imbalance factor using the difference between close and open prices
    order_imbalance = (df['close'] - df['open']) / daily_range
    
    # Calculate the 14-day Relative Strength Index (RSI) to capture market sentiment
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (smooth_buying_pressure / daily_range) * (df['close'] - ema_vwap) * momentum * log_return * trend_factor * (1 + volume_change) * (rsi / 50) * order_imbalance / (volatility_factor + 1e-7)
    
    return alpha_factor
```
