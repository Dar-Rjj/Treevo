```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a measure of daily volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate the average of open and close as a smoothed price representation
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the volume weighted average price
    vw_avg_price = (df['amount'] / df['volume'])
    
    # Calculate the difference between the close and open price as a momentum indicator
    daily_momentum = df['close'] - df['open']
    
    # Calculate the 5-day simple moving average of the close price
    sma_close_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day simple moving average of the close price
    sma_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day exponential moving average of the close price
    ema_close_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 20-day exponential moving average of the close price
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (daily_volatility * (vw_avg_price - avg_price) * (ema_close_5 - ema_close_20) * rsi) / (sma_close_20 + 1e-7)
    
    return alpha_factor
```
