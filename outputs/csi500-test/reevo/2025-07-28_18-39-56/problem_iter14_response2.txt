def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) of the close price with an adaptive window
    ema_close = df['close'].ewm(span=7, adjust=False).mean()

    # Calculate the adaptive relative strength index (RSI) with a dynamic window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate the adaptive momentum as the percentage change in the close price
    # with a dynamic window to capture short-term and long-term trends
    short_momentum = df['close'].pct_change(periods=5)
    long_momentum = df['close'].pct_change(periods=20)
    adaptive_momentum = (short_momentum + long_momentum) / 2

    # Calculate the volume-weighted average price (VWAP) with an adaptive window
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Calculate the trend factor as the difference between the close price and VWAP
    trend_factor = df['close'] - vwap

    # Calculate the volatility factor as the standard deviation of the close price
    volatility = df['close'].rolling(window=20).std()

    # Combine the factors into a single alpha factor
    # Incorporate the EMA, RSI, logarithmic returns, adaptive momentum, trend factor, and volatility
    alpha_factor = ((df['close'] - ema_close) * rsi * log_returns * adaptive_momentum * trend_factor) / volatility

    return alpha_factor
