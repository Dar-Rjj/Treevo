```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Calculate the 5-day exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the 10-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 3-day period
    log_returns = np.log(df['close'] / df['close'].shift(3))

    # Calculate the 5-day standard deviation of the close price for volatility
    std_close = df['close'].rolling(window=5).std()

    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) * df['volume']

    # Smooth the buying pressure over 2 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=2).mean()

    # Final alpha factor, which incorporates momentum, volatility, and volume with market sentiment
    alpha_factor = (vwap - ema_close) * rsi * log_returns * (smooth_buying_pressure / std_close)

    return alpha_factor
```
