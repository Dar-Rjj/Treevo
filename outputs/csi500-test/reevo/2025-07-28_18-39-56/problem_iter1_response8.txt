```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open, normalized by the average of high and low.
    # This factor attempts to capture the momentum of a stock's price movement during the trading day,
    # with an emphasis on whether the stock closed higher or lower than its opening price relative to the day's range.
    return (df['close'] - df['open']) / ((df['high'] + df['low'])/2 + 1e-7)
```
