```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a measure of daily volatility
    daily_volatility = df['high'] - df['low']

    # Calculate the difference between close and open as a measure of daily momentum
    daily_momentum = df['close'] - df['open']

    # Calculate the ratio of daily momentum to daily volatility, capturing the strength of the move relative to the range
    momentum_to_volatility = daily_momentum / (daily_volatility + 1e-7)

    # Calculate the 5-day rolling average of the amount
    rolling_amount_mean = df['amount'].rolling(window=5).mean()

    # Calculate the current amount divided by the rolling mean amount, capturing unusual trading activity
    normalized_amount = df['amount'] / rolling_amount_mean

    # Calculate the 5-day rolling average of the volume
    rolling_volume_mean = df['volume'].rolling(window=5).mean()

    # Calculate the current volume divided by the rolling mean volume, capturing unusual volume activity
    normalized_volume = df['volume'] / rolling_volume_mean

    # Calculate the 5-day rolling return
    rolling_return = df['close'].pct_change(5).fillna(0)

    # Combine the factors into a single alpha factor, incorporating both momentum-to-volatility, normalized amount, normalized volume, and rolling return
    alpha_factor = (momentum_to_volatility * normalized_amount * normalized_volume * rolling_return).fillna(0)

    return alpha_factor
```
