```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Smooth the buying pressure over 5 days to reduce noise
    smooth_buying_pressure = buying_pressure.rolling(window=5).mean()
    
    # Calculate the volume-weighted average price (VWAP) over a 10-day period
    vwap = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the change in VWAP from the previous day
    vwap_change = vwap.diff(1)
    
    # Calculate the 10-day and 20-day simple moving averages of the close price for trend strength
    sma_close_10 = df['close'].rolling(window=10).mean()
    sma_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 30-day and 60-day cumulative sum of volume as a measure of liquidity
    cum_volume_30 = df['volume'].rolling(window=30).sum()
    cum_volume_60 = df['volume'].rolling(window=60).sum()
    
    # Calculate the 10-day standard deviation of the close price for volatility
    std_close_10 = df['close'].rolling(window=10).std()
    
    # Calculate the 14-day Relative Strength Index (RSI) to capture market sentiment
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the order book imbalance factor using the difference between close and open prices
    order_imbalance = (df['close'] - df['open']) / daily_range
    
    # Calculate the 10-day momentum of the close price
    momentum_10 = df['close'] / df['close'].shift(10) - 1
    
    # Calculate the 30-day money flow index (MFI) to capture market sentiment and liquidity
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow[df['close'] > df['close'].shift(1)].fillna(0).rolling(window=30).sum()
    negative_flow = raw_money_flow[df['close'] < df['close'].shift(1)].fillna(0).rolling(window=30).sum()
    mfi = 100 - (100 / (1 + (positive_flow / (negative_flow + 1e-7))))
    
    # Final alpha factor, which is the smoothed buying pressure divided by the daily range, adjusted by the change in VWAP,
    # further adjusted by the ratio of the current close price to its 10-day SMA, the 10-day standard deviation,
    # the 20-day SMA for trend strength, the 30-day and 60-day cumulative volume, the 14-day RSI for market sentiment,
    # the order book imbalance factor, the 10-day momentum, and the 30-day MFI
    alpha_factor = (smooth_buying_pressure / daily_range) * vwap_change * (df['close'] / sma_close_10) * (1 / std_close_10) * (sma_close_10 / sma_close_20) * (cum_volume_30 / df['volume']) * (rsi / 50) * order_imbalance * momentum_10 * (mfi / 100)
    
    return alpha_factor
```
