```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted price
    vwp = (df['amount'] / df['volume'])
    
    # Calculate the exponential moving average of the volume-weighted price with a window of 7 days
    ema_vwp = vwp.ewm(span=7, adjust=False).mean()
    
    # Calculate the momentum of the close price with a window of 5 days
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the factor as the difference between the close price and the EMA of volume-weighted price
    # and multiply it by the momentum to incorporate the trend
    factor = (df['close'] - ema_vwp) * momentum

    return factor
```
