```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volatility (difference between high and low)
    volatility = df['high'] - df['low']
    
    # Calculate the smoothed price (average of open and close)
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the volume-weighted average price
    vw_avg_price = (df['amount'] / df['volume'])
    
    # Calculate the price change momentum (difference between close and open)
    price_momentum = df['close'] - df['open']
    
    # Calculate the trading volume intensity (volume divided by the average volume)
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_intensity = df['volume'] / avg_volume
    
    # Combine all factors into a single alpha factor
    alpha_factor = (volatility * (vw_avg_price - avg_price) * price_momentum * volume_intensity) / (avg_price + 1e-7)
    
    return alpha_factor
```
