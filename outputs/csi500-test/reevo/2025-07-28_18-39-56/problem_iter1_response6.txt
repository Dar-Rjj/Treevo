```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative differences between high, low, and close prices.
    pos_diff = (df['high'] - df['close']).clip(lower=0)
    neg_diff = (df['low'] - df['close']).clip(lower=0)
    
    # Calculate the signed volume-weighted price change.
    signed_volume_price_change = (df['close'] - df['open']) * df['volume']
    
    # Combine the features to create a novel alpha factor.
    alpha_factor = (pos_diff - neg_diff) + signed_volume_price_change
    
    return alpha_factor
```
