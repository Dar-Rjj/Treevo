```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted price
    vwp = (df['amount'] / df['volume'])
    
    # Calculate the exponential moving average of the volume-weighted price with a window of 7 days
    ema_vwp = vwp.ewm(span=7, adjust=False).mean()
    
    # Calculate the difference between the close price and the EMA of volume-weighted price
    factor = df['close'] - ema_vwp

    return factor
```
