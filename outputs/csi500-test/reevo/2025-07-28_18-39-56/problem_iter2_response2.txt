```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the logarithmic return for momentum, capturing the stock's price change more effectively.
    log_return = np.log(df['close']).diff().fillna(0)
    
    # Calculate the difference between close and open, normalized by the average of high and low,
    # to capture the momentum of a stock's price movement during the trading day.
    intraday_momentum = (df['close'] - df['open']) / ((df['high'] + df['low']) / 2 + 1e-7)
    
    # Calculate the ratio of current volume to the 5-day average volume, capturing liquidity.
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume.fillna(1)
    
    # Incorporate market sentiment using the amount traded, which can be a proxy for the value of the transactions.
    # This factor captures the magnitude of the transaction sizes, which could indicate strong buying or selling pressure.
    market_sentiment = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor, with a focus on robustness and interpretability.
    alpha_factor = (log_return * intraday_momentum * volume_ratio * market_sentiment).fillna(0)
    
    return alpha_factor
```
