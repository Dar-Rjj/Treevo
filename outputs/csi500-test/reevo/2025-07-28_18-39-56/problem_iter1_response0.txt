```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a measure of volatility
    daily_range = df['high'] - df['low']
    
    # Calculate the difference between close and open as a measure of daily momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate the ratio of daily momentum to daily range
    momentum_to_volatility = daily_momentum / (daily_range + 1e-7)
    
    # Calculate the rolling average of the amount over the last 5 days
    rolling_amount_mean = df['amount'].rolling(window=5).mean()
    
    # Calculate the current amount divided by the rolling mean amount
    normalized_amount = df['amount'] / rolling_amount_mean
    
    # Combine the factors into a single alpha factor
    alpha_factor = momentum_to_volatility * normalized_amount

    return alpha_factor
```
