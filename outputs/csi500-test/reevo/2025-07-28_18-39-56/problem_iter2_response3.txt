```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate a smoothed version of the VWAP by applying an exponential moving average
    vwap_smooth = vwap.ewm(span=10, adjust=False).mean()
    
    # Calculate the normalized difference between the current VWAP and the smoothed VWAP
    norm_diff_vwap = (vwap - vwap_smooth) / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the 10-day moving average of the close price
    ma_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the trend factor as the difference between the current close and the 10-day moving average
    trend_factor = df['close'] - ma_close_10
    
    # Combine the factors into a final heuristic score
    factor = diff_close_open * norm_diff_vwap * trend_factor
    
    return factor
```
