def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) over 7 days
    vwap = (df['amount'] / df['volume']).rolling(window=7).mean()

    # Calculate the exponential moving average (EMA) of the close price with a window of 7 days
    ema_close = df['close'].ewm(span=7, adjust=False).mean()

    # Calculate the relative strength index (RSI) with a window of 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the logarithmic returns over a 5-day period
    log_returns = np.log(df['close'] / df['close'].shift(5))

    # Calculate the 5-day rolling volatility of the close price
    rolling_volatility = df['close'].rolling(window=5).std()

    # Calculate the 5-day momentum of the close price
    momentum = df['close'].pct_change(periods=5)

    # Calculate the stochastic oscillator with a window of 14 days
    lowest_14 = df['low'].rolling(window=14).min()
    highest_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - lowest_14) / (highest_14 - lowest_14)
    d_percent = k_percent.rolling(window=3).mean()

    # Calculate the factor as a combination of various technical indicators
    factor = (df['close'] - ema_close) * rsi * log_returns * rolling_volatility * momentum * k_percent * d_percent

    return factor
