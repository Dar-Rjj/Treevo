```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the average volume over a 5-day window
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the momentum factor as the percentage change in closing price over the last 10 days
    momentum_10d = df['close'].pct_change(periods=10)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_diff / df['close'].shift(1)) * (avg_volume_5d / df['volume']) * momentum_10d
    
    return alpha_factor
```
