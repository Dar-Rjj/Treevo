def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change over the day
    price_change = df['close'] - df['open']

    # Calculate the average transaction amount per unit of volume to find the average transaction size
    avg_transaction_size = df['amount'] / (df['volume'] + 1e-7)

    # Calculate the relative strength index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(window=14).mean()
    roll_down = down.abs().rolling(window=14).mean()
    RS = roll_up / roll_down
    RSI = 100.0 - (100.0 / (1.0 + RS))

    # Combine the price change with the average transaction size and RSI to identify strong price movements
    # backed by significant trading activity and trend strength.
    alpha_factor = price_change * avg_transaction_size * RSI

    return alpha_factor
