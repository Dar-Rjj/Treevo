```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day and 50-day volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    df['volatility_50'] = df['daily_return'].rolling(window=50).std()
    
    # Calculate 20-day and 50-day momentum
    df['momentum_20'] = df['close'].pct_change(periods=20)
    df['momentum_50'] = df['close'].pct_change(periods=50)
    
    # Calculate average volume over 20 days to use as a liquidity measure
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the longer-term trend (50-day moving average)
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from the long-term trend
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / df['long_term_trend']
    
    # Calculate the relative strength index (RSI) for 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Weighted factor combining multi-period momentum, inverse volatility, trend distance, RSI, and VWAP
    alpha_factor = (
        ((df['momentum_20'] + df['momentum_50']) / 2) / 
        (df['volatility_20'] + df['volatility_50'] + 1e-7) * 
        (df['avg_volume_20'] / (df['volume'] + 1e-7)) * 
        df['trend_distance'] * 
        (100 - df['rsi']) * 
        (df['close'] / df['vwap'])
    )
    return alpha_factor
```
