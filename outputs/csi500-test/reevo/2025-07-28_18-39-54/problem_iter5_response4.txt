```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Dynamic volatility windows
    df['volatility_20d'] = df['daily_return'].rolling(window=20).std()
    df['volatility_60d'] = df['daily_return'].rolling(window=60).std()
    df['volatility_120d'] = df['daily_return'].rolling(window=120).std()
    df['dynamic_volatility'] = (df['volatility_20d'] + df['volatility_60d'] + df['volatility_120d']) / 3
    
    # Dynamic momentum windows
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    df['momentum_120d'] = df['close'].pct_change(periods=120)
    df['dynamic_momentum'] = (df['momentum_20d'] + df['momentum_60d'] + df['momentum_120d']) / 3
    
    # Volume trend with adaptive windows
    df['volume_trend_20d'] = df['volume'].pct_change(periods=20)
    df['volume_trend_60d'] = df['volume'].pct_change(periods=60)
    df['volume_trend_120d'] = df['volume'].pct_change(periods=120)
    df['adaptive_volume_trend'] = (df['volume_trend_20d'] + df['volume_trend_60d'] + df['volume_trend_120d']) / 3
    
    # Risk-adjusted returns with dynamic volatility
    df['risk_adjusted_return_20d'] = df['momentum_20d'] / (df['volatility_20d'] + 1e-7)
    df['risk_adjusted_return_60d'] = df['momentum_60d'] / (df['volatility_60d'] + 1e-7)
    df['risk_adjusted_return_120d'] = df['momentum_120d'] / (df['volatility_120d'] + 1e-7)
    df['dynamic_risk_adjusted_return'] = (df['risk_adjusted_return_20d'] + df['risk_adjusted_return_60d'] + df['risk_adjusted_return_120d']) / 3
    
    # Liquidity measure
    df['liquidity'] = df['amount'] / df['volume']
    df['avg_liquidity_20d'] = df['liquidity'].rolling(window=20).mean()
    df['avg_liquidity_60d'] = df['liquidity'].rolling(window=60).mean()
    df['avg_liquidity_120d'] = df['liquidity'].rolling(window=120).mean()
    df['dynamic_liquidity'] = (df['avg_liquidity_20d'] + df['avg_liquidity_60d'] + df['avg_liquidity_120d']) / 3
    
    # Combine the factors to create a new alpha factor
    alpha_factor = (
        (df['dynamic_risk_adjusted_return'] * df['adaptive_volume_trend']) +
        df['dynamic_liquidity']
    ) / 2
    
    return alpha_factor
```
