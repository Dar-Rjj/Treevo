```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted by the difference between close and open prices
    vol_weighted_diff = df['volume'] * diff_close_open
    
    # Alpha factor: volume weighted price difference divided by the average of high and low prices
    alpha_factor = vol_weighted_diff / (avg_high_low + 1e-7)
    
    return alpha_factor
```
