def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day, 60-day, and 120-day volatility
    df['volatility_20d'] = df['daily_return'].rolling(window=20).std()
    df['volatility_60d'] = df['daily_return'].rolling(window=60).std()
    df['volatility_120d'] = df['daily_return'].rolling(window=120).std()
    
    # Calculate 20-day, 60-day, and 120-day momentum
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    df['momentum_120d'] = df['close'].pct_change(periods=120)
    
    # Calculate the volume trend over 20, 60, and 120 days
    df['volume_trend_20d'] = df['volume'].pct_change(periods=20)
    df['volume_trend_60d'] = df['volume'].pct_change(periods=60)
    df['volume_trend_120d'] = df['volume'].pct_change(periods=120)
    
    # Risk-adjusted returns
    df['risk_adjusted_return_20d'] = df['momentum_20d'] / (df['volatility_20d'] + 1e-7)
    df['risk_adjusted_return_60d'] = df['momentum_60d'] / (df['volatility_60d'] + 1e-7)
    df['risk_adjusted_return_120d'] = df['momentum_120d'] / (df['volatility_120d'] + 1e-7)
    
    # Adaptive weights based on the standard deviation of the risk-adjusted returns
    std_20d = df['risk_adjusted_return_20d'].rolling(window=20).std()
    std_60d = df['risk_adjusted_return_60d'].rolling(window=60).std()
    std_120d = df['risk_adjusted_return_120d'].rolling(window=120).std()
    
    total_std = std_20d + std_60d + std_120d
    w_20d = std_20d / total_std
    w_60d = std_60d / total_std
    w_120d = std_120d / total_std
    
    # Liquidity factor: average volume over 20, 60, and 120 days
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['avg_volume_60d'] = df['volume'].rolling(window=60).mean()
    df['avg_volume_120d'] = df['volume'].rolling(window=120).mean()
    
    # Sentiment factor: ratio of close price to open price
    df['sentiment'] = df['close'] / df['open']
    
    # Mean reversion factor: distance from 20-day moving average
    df['ma_20d'] = df['close'].rolling(window=20).mean()
    df['mean_reversion_20d'] = df['close'] - df['ma_20d']
    
    # Combine the factors to create a new alpha factor using adaptive weights
    alpha_factor = (
        (w_20d * df['risk_adjusted_return_20d'] * df['volume_trend_20d'] * df['avg_volume_20d'] * df['sentiment'] * df['mean_reversion_20d']) +
        (w_60d * df['risk_adjusted_return_60d'] * df['volume_trend_60d'] * df['avg_volume_60d'] * df['sentiment'] * df['mean_reversion_20d']) +
        (w_120d * df['risk_adjusted_return_120d'] * df['volume_trend_120d'] * df['avg_volume_120d'] * df['sentiment'] * df['mean_reversion_20d'])
    )
