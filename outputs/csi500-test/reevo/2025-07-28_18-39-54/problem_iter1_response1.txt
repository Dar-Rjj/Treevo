```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in close price from one day to the next
    df['close_shift'] = df['close'].shift(1)
    # Calculate the simple moving average (SMA) of the close prices over 5 days
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    # Calculate a factor as the difference between today's close and the 5-day SMA, 
    # then divide it by the previous day's close (to scale with the price level).
    alpha_factor = (df['close'] - df['SMA_5']) / df['close_shift']
    
    return alpha_factor
```
