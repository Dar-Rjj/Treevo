```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 30-day and 90-day volatility
    df['volatility_30d'] = df['daily_return'].rolling(window=30).std()
    df['volatility_90d'] = df['daily_return'].rolling(window=90).std()
    
    # Calculate 30-day and 90-day momentum
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    df['momentum_90d'] = df['close'].pct_change(periods=90)
    
    # Risk-adjusted returns
    df['risk_adjusted_return_30d'] = df['momentum_30d'] / (df['volatility_30d'] + 1e-7)
    df['risk_adjusted_return_90d'] = df['momentum_90d'] / (df['volatility_90d'] + 1e-7)
    
    # Adaptive weights based on the standard deviation of the risk-adjusted returns
    std_30d = df['risk_adjusted_return_30d'].rolling(window=30).std()
    std_90d = df['risk_adjusted_return_90d'].rolling(window=90).std()
    
    total_std = std_30d + std_90d
    w_30d = std_30d / total_std
    w_90d = std_90d / total_std
    
    # Combine the factors to create a new alpha factor using adaptive weights
    alpha_factor = (
        (w_30d * df['risk_adjusted_return_30d']) +
        (w_90d * df['risk_adjusted_return_90d'])
    )
    
    return alpha_factor
```
