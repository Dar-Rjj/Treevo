```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate 20-day and 60-day momentum
    df['momentum_20'] = df['close'].pct_change(periods=20)
    df['momentum_60'] = df['close'].pct_change(periods=60)
    
    # Calculate average volume over 20 days to use as a liquidity measure
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the longer-term trend (50-day moving average)
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from the long-term trend
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / (df['long_term_trend'] + 1e-7)
    
    # Calculate the ratio of current volume to average volume
    df['volume_ratio'] = df['volume'] / (df['avg_volume'] + 1e-7)
    
    # Weighted factor combining both momentums, inverse volatility, volume ratio, and trend distance
    alpha_factor = (df['momentum_20'] + df['momentum_60']) / (df['volatility'] + 1e-7) * df['volume_ratio'] * df['trend_distance']
    
    return alpha_factor
```
