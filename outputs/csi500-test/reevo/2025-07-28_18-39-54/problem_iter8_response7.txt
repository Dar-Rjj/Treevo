def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Dynamic volatility with adaptive windows
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    df['volatility_30d'] = df['daily_return'].rolling(window=30).std()
    df['volatility_60d'] = df['daily_return'].rolling(window=60).std()
    df['dynamic_volatility'] = (df['volatility_10d'] + df['volatility_30d'] + df['volatility_60d']) / 3
    
    # Dynamic momentum with adaptive windows
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    df['dynamic_momentum'] = (df['momentum_10d'] + df['momentum_30d'] + df['momentum_60d']) / 3
    
    # Volume trend with adaptive windows
    df['volume_trend_10d'] = df['volume'].pct_change(periods=10)
    df['volume_trend_30d'] = df['volume'].pct_change(periods=30)
    df['volume_trend_60d'] = df['volume'].pct_change(periods=60)
    df['adaptive_volume_trend'] = (df['volume_trend_10d'] + df['volume_trend_30d'] + df['volume_trend_60d']) / 3
    
    # Risk-adjusted returns with dynamic volatility
    df['risk_adjusted_return_10d'] = df['momentum_10d'] / (df['volatility_10d'] + 1e-7)
    df['risk_adjusted_return_30d'] = df['momentum_30d'] / (df['volatility_30d'] + 1e-7)
    df['risk_adjusted_return_60d'] = df['momentum_60d'] / (df['volatility_60d'] + 1e-7)
    df['dynamic_risk_adjusted_return'] = (df['risk_adjusted_return_10d'] + df['risk_adjusted_return_30d'] + df['risk_adjusted_return_60d']) / 3
    
    # Liquidity measure
    df['liquidity'] = df['amount'] / df['volume']

    df['avg_liquidity_30d'] = df['liquidity'].rolling(window=30).mean()
    df['avg_liquidity_60d'] = df['liquidity'].rolling(window=60).mean()
    df['dynamic_liquidity'] = (df['avg_liquidity_10d'] + df['avg_liquidity_30d'] + df['avg_liquidity_60d']) / 3
