```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate 20-day momentum
    df['momentum'] = df['close'].pct_change(periods=20)
    
    # Calculate average volume over 20 days to use as a liquidity measure
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the longer-term trend (50-day moving average)
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from the long-term trend
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / df['long_term_trend']
    
    # Calculate the ratio of current volume to the average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Calculate the relative strength index (RSI) over 14 periods
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Weighted factor combining momentum, inverse volatility, trend distance, and RSI
    alpha_factor = (df['momentum'] / (df['volatility'] + 1e-7)) * (df['volume_ratio'] + 1) * df['trend_distance'] * (100 - df['rsi'])
    return alpha_factor
```
