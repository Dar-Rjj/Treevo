```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, adjusted by volume
    volatility_adjusted_by_volume = (df['high'] - df['low']) / (df['volume'] + 1e-7)
    
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the weighted price change by normalizing it with the adjusted volatility
    weighted_price_change = price_change / (volatility_adjusted_by_volume + 1e-7)
    
    # Calculate the cumulative amount over the last 5 days
    df['amount_5_day_sum'] = df['amount'].rolling(window=5).sum()
    
    # Calculate the factor as the weighted price change adjusted by the 5-day cumulative amount
    alpha_factor = weighted_price_change * (df['amount_5_day_sum'] + 1e-7)
    
    return alpha_factor
```
