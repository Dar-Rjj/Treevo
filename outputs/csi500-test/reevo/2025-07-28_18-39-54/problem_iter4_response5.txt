```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day, 50-day, and 100-day volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    df['volatility_50'] = df['daily_return'].rolling(window=50).std()
    df['volatility_100'] = df['daily_return'].rolling(window=100).std()
    
    # Calculate 20-day, 50-day, and 100-day momentum
    df['momentum_20'] = df['close'].pct_change(periods=20)
    df['momentum_50'] = df['close'].pct_change(periods=50)
    df['momentum_100'] = df['close'].pct_change(periods=100)
    
    # Calculate average volume over 20 days to use as a liquidity measure
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the longer-term trend (50-day moving average)
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from the long-term trend
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / df['long_term_trend']
    
    # Calculate the relative strength index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(window=14).mean()
    roll_down = down.abs().rolling(window=14).mean()
    RS = roll_up / (roll_down + 1e-7)
    RSI = 100.0 - (100.0 / (1.0 + RS))
    
    # Weighted factor combining multi-period momentum, inverse volatility, and trend distance
    alpha_factor = (
        (df['momentum_20'] + df['momentum_50'] + df['momentum_100']) / 3
    ) / (
        (df['volatility_20'] + df['volatility_50'] + df['volatility_100']) / 3 + 1e-7
    ) * (
        (df['avg_volume_20'] / (df['volume'] + 1e-7)) * df['trend_distance'] * (RSI / 100)
    )
    return alpha_factor
```
