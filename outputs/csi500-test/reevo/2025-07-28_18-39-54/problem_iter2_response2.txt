```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices, representing the price change over the day
    price_change = df['close'] - df['open']
    
    # Calculate the average transaction amount per unit of volume to find the average transaction size
    avg_transaction_size = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the weighted moving average of the close prices over 5 days using volume as the weight
    df['WMA_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # A factor that combines the price change with the average transaction size and the deviation from the 5-day WMA
    alpha_factor = (price_change * avg_transaction_size) + (df['close'] - df['WMA_5'])
    
    return alpha_factor
```
