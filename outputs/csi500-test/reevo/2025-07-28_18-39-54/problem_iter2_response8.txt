```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open prices as a measure of intraday momentum
    intraday_momentum = df['close'] - df['open']
    
    # Calculate the weighted average price using amount and volume, this acts as a different way to value the closing price
    weighted_price = (df['amount'] * df['close']) / df['volume']
    
    # Calculate the deviation of the current close price from the 5-day moving average close price
    moving_avg_5 = df['close'].rolling(window=5).mean()
    dev_from_ma5 = df['close'] - moving_avg_5
    
    # Calculate the relative strength over the last 5 days compared to the previous 5 days
    prev_close_5_days = df['close'].shift(5)
    recent_strength = (df['close'] - prev_close_5_days) / prev_close_5_days
    
    # Integrate a trend factor by considering the 10-day moving average
    moving_avg_10 = df['close'].rolling(window=10).mean()
    trend_factor = df['close'] - moving_avg_10
    
    # Combine all the factors into a final score with appropriate weights
    factor_value = (intraday_momentum + 
                    (weighted_price * 0.5) + 
                    dev_from_ma5 + 
                    (recent_strength * 0.7) + 
                    (trend_factor * 0.3))
    
    return factor_value
```
