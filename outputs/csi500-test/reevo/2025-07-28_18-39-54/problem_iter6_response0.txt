```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day, 60-day, and 120-day volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    df['volatility_60'] = df['daily_return'].rolling(window=60).std()
    df['volatility_120'] = df['daily_return'].rolling(window=120).std()
    
    # Calculate 20-day, 60-day, and 120-day momentum
    df['momentum_20'] = df['close'].pct_change(periods=20)
    df['momentum_60'] = df['close'].pct_change(periods=60)
    df['momentum_120'] = df['close'].pct_change(periods=120)
    
    # Calculate average volume over 20 days, 60 days, and 120 days to use as a liquidity measure
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['avg_volume_60'] = df['volume'].rolling(window=60).mean()
    df['avg_volume_120'] = df['volume'].rolling(window=120).mean()
    
    # Calculate the longer-term trend (50-day, 100-day, and 200-day moving average)
    df['long_term_trend_50'] = df['close'].rolling(window=50).mean()
    df['long_term_trend_100'] = df['close'].rolling(window=100).mean()
    df['long_term_trend_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate the distance of the current close price from the long-term trends
    df['trend_distance_50'] = (df['close'] - df['long_term_trend_50']) / (df['long_term_trend_50'] + 1e-7)
    df['trend_distance_100'] = (df['close'] - df['long_term_trend_100']) / (df['long_term_trend_100'] + 1e-7)
    df['trend_distance_200'] = (df['close'] - df['long_term_trend_200']) / (df['long_term_trend_200'] + 1e-7)
    
    # Calculate the ratio of current volume to average volumes
    df['volume_ratio_20'] = df['volume'] / (df['avg_volume_20'] + 1e-7)
    df['volume_ratio_60'] = df['volume'] / (df['avg_volume_60'] + 1e-7)
    df['volume_ratio_120'] = df['volume'] / (df['avg_volume_120'] + 1e-7)
    
    # Weighted factor combining both momentums, inverse volatilities, volume ratios, and trend distances
    alpha_factor = (
        (df['momentum_20'] + df['momentum_60'] + df['momentum_120']) / 
        (df['volatility_20'] + df['volatility_60'] + df['volatility_120'] + 1e-7) * 
        (df['volume_ratio_20'] + df['volume_ratio_60'] + df['volume_ratio_120']) / 3 * 
        (df['trend_distance_50'] + df['trend_distance_100'] + df['trend_distance_200']) / 3
    )
    
    return alpha_factor
```
