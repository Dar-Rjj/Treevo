```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day and 60-day volatility
    df['volatility_20d'] = df['daily_return'].rolling(window=20).std()
    df['volatility_60d'] = df['daily_return'].rolling(window=60).std()
    
    # Calculate 20-day, 60-day, and 120-day momentum
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    df['momentum_120d'] = df['close'].pct_change(periods=120)
    
    # Calculate the volume trend over 20 days and 60 days
    df['volume_trend_20d'] = df['volume'].pct_change(periods=20)
    df['volume_trend_60d'] = df['volume'].pct_change(periods=60)
    
    # Calculate the average transaction amount per unit of volume to find the average transaction size
    df['avg_transaction_size'] = df['amount'] / (df['volume'] + 1e-7)
    
    # Combine the factors to create a new alpha factor
    alpha_factor = (
        (df['momentum_20d'] / (df['volatility_20d'] + 1e-7)) * 
        (df['momentum_60d'] / (df['volatility_60d'] + 1e-7)) * 
        (df['momentum_120d'] / (df['volatility_60d'] + 1e-7)) * 
        (df['volume_trend_20d'] + 1) * 
        (df['volume_trend_60d'] + 1) * 
        df['avg_transaction_size']
    )
    
    return alpha_factor
```
