```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted by the difference between close and open prices
    vol_weighted_diff = df['volume'] * diff_close_open
    
    # Calculate the momentum over the last 5 days
    momentum_5d = df['close'].pct_change(periods=5)
    
    # Calculate the momentum over the last 10 days to capture a longer-term trend
    momentum_10d = df['close'].pct_change(periods=10)
    
    # Calculate the volatility over the last 10 days
    volatility_10d = df['close'].rolling(window=10).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    liquidity = df['amount'] / df['volume']
    
    # Calculate the dynamic volatility measure using the difference between high and low prices
    dynamic_volatility = (df['high'] - df['low']).rolling(window=10).std()
    
    # Calculate the price action as the ratio of the current close price to the average of high and low prices
    price_action = df['close'] / avg_high_low
    
    # Adjust for volatility in the alpha factor calculation
    adjusted_volatility = 1 / (volatility_10d + 1e-7)
    
    # Alpha factor: integrate both short-term and long-term momentum, dynamic and historical volatility, liquidity, price action, and volume weighted price difference
    alpha_factor = (vol_weighted_diff / (avg_high_low + 1e-7)) * (momentum_5d + 1) * (momentum_10d + 1) * adjusted_volatility * (1 / (dynamic_volatility + 1e-7)) * liquidity * price_action
    
    return alpha_factor
```
