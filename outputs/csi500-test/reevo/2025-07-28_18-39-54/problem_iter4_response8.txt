```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate 20-day momentum
    df['momentum'] = df['close'].pct_change(periods=20)
    
    # Calculate average volume over 20 days to use as a liquidity measure
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the longer-term trend (50-day moving average)
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance of the current close price from the long-term trend
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / df['long_term_trend']
    
    # Calculate the shorter-term trend (10-day moving average)
    df['short_term_trend'] = df['close'].rolling(window=10).mean()
    
    # Calculate the distance of the current close price from the short-term trend
    df['short_trend_distance'] = (df['close'] - df['short_term_trend']) / df['short_term_trend']
    
    # Additional robust trend measures
    df['long_term_momentum'] = df['close'].pct_change(periods=50)
    df['short_term_momentum'] = df['close'].pct_change(periods=10)
    
    # Calculate the relative strength index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(window=14).mean()
    roll_down = down.abs().rolling(window=14).mean()
    RS = roll_up / roll_down
    RSI = 100.0 - (100.0 / (1.0 + RS))
    
    # Weighted factor combining momentum, inverse volatility, and trend distances
    alpha_factor = (
        (df['momentum'] + df['short_term_momentum'] + df['long_term_momentum']) / 
        (df['volatility'] + 1e-7) * 
        (df['avg_volume'] / (df['volume'] + 1e-7)) * 
        (df['trend_distance'] + df['short_trend_distance']) * 
        (1 + (RSI - 50) / 50)
    )
    return alpha_factor
```
