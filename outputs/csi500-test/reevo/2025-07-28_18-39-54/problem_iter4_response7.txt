```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices, representing the price change over the day
    price_change = df['close'] - df['open']
    
    # Calculate the average transaction amount per unit of volume to find the average transaction size
    avg_transaction_size = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the momentum factor using the price change over the last 3 days for a more recent trend
    momentum_3d = df['close'].pct_change(periods=3)
    
    # Calculate the liquidity factor using the ratio of volume to the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    liquidity_factor = df['volume'] / (avg_volume_5d + 1e-7)
    
    # Calculate the volume shock factor as the difference between current volume and the average volume over the last 5 days
    volume_shock = df['volume'] - avg_volume_5d
    
    # Introduce a price range factor to capture the intraday volatility
    price_range = df['high'] - df['low']
    
    # A factor that combines the price change, average transaction size, momentum, liquidity, volume shock, and price range
    # to identify strong price movements backed by significant trading activity and intraday volatility, which could indicate a potential trend continuation.
    alpha_factor = (price_change * avg_transaction_size * momentum_3d * liquidity_factor * volume_shock * price_range).fillna(0)
    
    return alpha_factor
```
