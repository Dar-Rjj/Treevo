```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Adaptive volatility with exponentially weighted moving average (EWMA)
    df['volatility_ewma'] = df['daily_return'].ewm(span=20, adjust=False).std()
    
    # Adaptive momentum with exponentially weighted moving average (EWMA)
    df['momentum_ewma'] = df['close'].ewm(span=20, adjust=False).mean().pct_change(periods=20)
    
    # Volume trend over 20, 60, and 120 days
    df['volume_trend_20d'] = df['volume'].pct_change(periods=20)
    df['volume_trend_60d'] = df['volume'].pct_change(periods=60)
    df['volume_trend_120d'] = df['volume'].pct_change(periods=120)
    
    # Risk-adjusted returns using EWMA volatility
    df['risk_adjusted_return_ewma'] = df['momentum_ewma'] / (df['volatility_ewma'] + 1e-7)
    
    # Calculate the distance of the current close price from the 50-day moving average
    df['long_term_trend'] = df['close'].rolling(window=50).mean()
    df['trend_distance'] = (df['close'] - df['long_term_trend']) / df['long_term_trend']
    
    # Calculate the relative strength index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.ewm(com=13, adjust=False).mean()
    roll_down = down.abs().ewm(com=13, adjust=False).mean()
    RS = roll_up / (roll_down + 1e-7)
    RSI = 100.0 - (100.0 / (1.0 + RS))
    
    # Combine the factors to create a new alpha factor
    alpha_factor = (
        (df['risk_adjusted_return_ewma'] * df['volume_trend_20d']) +
        (df['risk_adjusted_return_ewma'] * df['volume_trend_60d']) +
        (df['risk_adjusted_return_ewma'] * df['volume_trend_120d'])
    ) / 3 * df['trend_distance'] * (RSI / 100)
    
    return alpha_factor
```
