```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the price change over the last 5 days
    price_change_5d = df['close'].pct_change(periods=5)
    
    # Combine the above factors to create a new alpha factor
    alpha_factor = (close_open_diff / (df['high'] - df['low'] + 1e-7)) * (avg_volume_5d / df['volume']) * price_change_5d
    
    return alpha_factor
```
