```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price, which represents the day's trend
    daily_trend = df['close'] - df['open']
    
    # Calculate the amplitude of the price movement for each day as the difference between high and low
    price_amplitude = df['high'] - df['low']
    
    # Calculate the average true range over the last 14 days to use as a volatility measure
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x['high']-x['low'], abs(x['high']-x['close'].shift(1)), abs(x['low']-x['close'].shift(1))), axis=1)
    atr_14 = tr.rolling(window=14).mean()
    
    # Calculate the factor by normalizing the daily trend with respect to the average true range
    # This measures the strength of the day's trend in context of the stock's volatility
    factor = daily_trend / (atr_14 + 1e-7)
    
    return factor
```
