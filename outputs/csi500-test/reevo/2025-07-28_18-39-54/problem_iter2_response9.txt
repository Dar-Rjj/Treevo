```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices, representing the daily price change
    price_diff = df['close'] - df['open']
    
    # Calculate the 5-day average volume to smooth out daily fluctuations
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the 10-day momentum as the percentage change in closing price
    momentum_10d = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day moving average of the closing price for trend following
    ma_20d = df['close'].rolling(window=20).mean()
    
    # Calculate the 60-day volatility as the standard deviation of the daily returns
    volatility_60d = df['close'].pct_change().rolling(window=60).std()
    
    # Combine the factors into a single alpha factor
    # The factor incorporates price change, volume trends, momentum, trend following, and volatility
    alpha_factor = (price_diff / df['close'].shift(1)) * (avg_volume_5d / df['volume']) * momentum_10d * (df['close'] / ma_20d) * (1 / (volatility_60d + 1e-7))
    
    return alpha_factor
```
