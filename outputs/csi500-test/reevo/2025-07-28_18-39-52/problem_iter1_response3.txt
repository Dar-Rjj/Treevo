```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative volume index changes
    df['volume_change'] = df['volume'].diff()
    df_pos_vol = df[df['volume_change'] > 0]['volume_change']
    df_neg_vol = df[df['volume_change'] < 0]['volume_change'].abs()

    # Calculate adjusted typical price based on volume change
    df['typical_price'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    df['adj_typical_price'] = (df['typical_price'] * df['volume_change']).rolling(window=5).sum() / df['volume_change'].rolling(window=5).sum()

    # Determine the weighted difference between typical prices on positive vs. negative volume days, smoothed over a 5 day period to reduce noise.
    pos_tp = (df['adj_typical_price'] * df_pos_vol).rolling(window=5).sum() / df_pos_vol.rolling(window=5).sum()
    neg_tp = (df['adj_typical_price'] * df_neg_vol).rolling(window=5).sum() / df_neg_vol.rolling(window=5).sum()
    
    # Create the final factor by finding the difference and dividing by a normalizing term (absolute value of negative tp)
    factor_value = (pos_tp - neg_tp) / (neg_tp.abs() + 1e-7)
    
    return factor_value
```
