```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and opening price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the normalized volume
    norm_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Alpha factor: (Close - Open) * (High + Low)/2 * Volume_normalized
    alpha_factor = close_open_diff * avg_high_low * norm_volume
    
    return alpha_factor
```
