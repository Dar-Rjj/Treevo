```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices (price range)
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the momentum as the percentage change in close price over the last 5 periods
    momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Calculate the liquidity as the ratio of volume to the price range
    liquidity = df['volume'] / (price_range + 1e-7)
    
    # Calculate the volatility as the standard deviation of the close price over the last 14 periods
    volatility = df['close'].rolling(window=14).std().fillna(0)
    
    # Calculate the dynamic ratio as the ratio of the smoothed volume-weighted difference to the smoothed amount-weighted difference
    dynamic_ratio = (smooth_vol_weighted_diff + 1e-7) / (smooth_amount_weighted_diff + 1e-7)
    
    # Final alpha factor: the ratio of the smoothed volume-weighted difference to the price range, 
    # plus the ratio of the smoothed amount-weighted difference to the price range, 
    # multiplied by the momentum, adjusted by the liquidity and volatility
    alpha_factor = ((smooth_vol_weighted_diff + smooth_amount_weighted_diff) / (price_range + 1e-7)) * momentum * liquidity * (1 / (volatility + 1e-7))
    
    return alpha_factor
```
