def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the momentum of the typical price over a 5-day period
    momentum_typical_price = typical_price - typical_price.shift(5)
    
    # Calculate the volume-weighted price
    volume_weighted_price = (df['close'] * df['volume']).rolling(window=5).mean() / df['volume'].rolling(window=5).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = (df['close'] - df['close'].shift(1)) * volume_weighted_price * df['volume']
    
    # Smooth the money flow ratio with a 10-day rolling mean
    smoothed_money_flow = money_flow_ratio.rolling(window=10).mean()
    
    # Combine the smoothed money flow and the momentum of the typical price
    alpha_factor = smoothed_money_flow * momentum_typical_price
    
    return alpha_factor
