```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2) / df['volume'].mean()
    
    # Calculate a smoothed version of the volume-weighted difference
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2) / df['amount'].mean()
    
    # Calculate a smoothed version of the amount-weighted difference
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Final alpha factor: the ratio of the smoothed volume-weighted difference to the price range, 
    # plus the ratio of the smoothed amount-weighted difference to the price range
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff) / (price_range + 1e-7)
    
    return alpha_factor
```
