```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and the previous day's close
    df['close_diff'] = df['close'].diff().fillna(0)
    
    # Calculate the average true range (ATR) over a 14-day period
    df['tr'] = df[['high' - 'low', 'high' - df['close'].shift(), df['close'].shift() - 'low']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean().fillna(method='bfill')
    
    # Calculate the ratio of close difference to ATR
    df['close_diff_to_atr'] = df['close_diff'] / df['atr_14']
    
    # Generate the alpha factor
    alpha_factor = df['close_diff_to_atr'] * (df['volume'] / df['volume'].rolling(window=5).mean().fillna(method='bfill'))
    
    return alpha_factor
```
