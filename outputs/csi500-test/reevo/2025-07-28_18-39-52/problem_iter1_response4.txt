```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the average price of the day
    df['avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    # Calculate the moving average of the volume over a 5-day window
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    # Calculate the change in volume compared to the 5-day moving average
    df['volume_change'] = df['volume'] - df['volume_ma_5']
    # Calculate the alpha factor as the product of the close difference, average price, and volume change
    alpha_factor = df['close_diff'] * df['avg_price'] * df['volume_change']
    return alpha_factor
```
