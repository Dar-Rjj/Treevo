```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price range as the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference using exponential smoothing
    smooth_vol_weighted_diff = vol_weighted_diff.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference using exponential smoothing
    smooth_amount_weighted_diff = amount_weighted_diff.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the market depth as the ratio of amount to volume
    market_depth = df['amount'] / (df['volume'] + 1e-7)
    
    # Smooth the market depth using exponential smoothing
    smooth_market_depth = market_depth.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the relative strength index (RSI) as a momentum indicator
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Incorporate RSI as a momentum factor
    momentum_factor = rsi.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the 20-day moving average of the closing price to consider longer-term trends
    long_term_trend = df['close'].ewm(span=20, adjust=False).mean().fillna(0)
    
    # Incorporate the longer-term trend into the alpha factor
    trend_factor = (df['close'] - long_term_trend) / (long_term_trend + 1e-7)
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # multiply by the smoothed market depth, and incorporate the momentum and trend factors
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff) / (price_range + 1e-7) * smooth_market_depth * momentum_factor * trend_factor
    
    return alpha_factor
```
