def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3

    # Calculate the money flow ratio
    money_flow_ratio = (df['close'] - df['close'].shift(1)) * typical_price * df['volume']

    # Calculate the smoothed money flow ratio by taking the rolling mean over a 20-day period
    smoothed_money_flow = money_flow_ratio.rolling(window=20).mean()

    # Calculate the normalized volume
    avg_volume = df['volume'].rolling(window=20).mean()
    normalized_volume = df['volume'] / avg_volume

    # Calculate the smoothed return
    daily_return = df['close'].pct_change()
    smoothed_return = daily_return.rolling(window=20).mean()

    # Calculate the volatility
    volatility = daily_return.rolling(window=20).std()

    # Generate the alpha factor by combining smoothed money flow, normalized volume, smoothed return, and inverse volatility
    alpha_factor = smoothed_money_flow * normalized_volume * smoothed_return * (1 / (volatility + 1e-7))

    return alpha_factor
