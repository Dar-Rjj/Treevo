```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference using exponential smoothing
    smooth_vol_weighted_diff = vol_weighted_diff.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference using exponential smoothing
    smooth_amount_weighted_diff = amount_weighted_diff.ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the relative strength index (RSI) as a measure of momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the volatility factor as the standard deviation of the last 10 days' returns
    volatility_factor = df['close'].pct_change().rolling(window=10).std().fillna(0)
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    vol_weighted_typical_price = (df['volume'] * typical_price).ewm(span=5, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted typical price
    amount_weighted_typical_price = (df['amount'] * typical_price).ewm(span=5, adjust=False).mean().fillna(0)
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # and the volume-weighted and amount-weighted typical prices, adjusted by the RSI and volatility factors
    alpha_factor = (
        (smooth_vol_weighted_diff + smooth_amount_weighted_diff + 
         vol_weighted_typical_price + amount_weighted_typical_price) / (price_range + 1e-7) * 
         (rsi / 50 + 1) * (1 / (volatility_factor + 1e-7))
    )
    
    return alpha_factor
```
