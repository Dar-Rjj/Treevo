```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between today's close and the close 5 days ago
    df['momentum_5'] = df['close'] - df['close'].shift(5)
    
    # Calculate the volatility as the standard deviation of the daily returns over a 5-day window
    df['return'] = df['close'].pct_change()
    df['volatility_5'] = df['return'].rolling(window=5).std()
    
    # Calculate the volume trend as the difference between today's volume and the average volume over the last 10 days
    df['volume_trend_10'] = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Calculate the weighted typical price, which is a mix of high, low, and close prices
    df['weighted_typical_price'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    
    # Calculate the alpha factor as the product of the momentum, inverse of volatility, and volume trend, 
    # multiplied by the weighted typical price
    df['alpha_factor'] = df['momentum_5'] * (1 / (df['volatility_5'] + 1e-7)) * df['volume_trend_10'] * df['weighted_typical_price']
    
    return df['alpha_factor']
```
