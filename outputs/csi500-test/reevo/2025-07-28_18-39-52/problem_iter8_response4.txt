```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference with an exponential moving average
    smooth_vol_weighted_diff = vol_weighted_diff.ewm(span=7, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference with an exponential moving average
    smooth_amount_weighted_diff = amount_weighted_diff.ewm(span=7, adjust=False).mean().fillna(0)
    
    # Calculate the volume-weighted typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vol_weighted_typical_price = (df['volume'] * typical_price).ewm(span=7, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted typical price
    amount_weighted_typical_price = (df['amount'] * typical_price).ewm(span=7, adjust=False).mean().fillna(0)
    
    # Calculate the smoothed return
    daily_return = df['close'].pct_change()
    smoothed_return = daily_return.ewm(span=7, adjust=False).mean().fillna(0)
    
    # Calculate the volatility of the close price
    volatility = df['close'].rolling(window=7).std().fillna(0)
    
    # Calculate the liquidity measure (volume divided by price range)
    liquidity = df['volume'] / (price_range + 1e-7)
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # the volume-weighted and amount-weighted typical prices, the smoothed return, the volatility, and the liquidity
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff + 
                    vol_weighted_typical_price + amount_weighted_typical_price + 
                    smoothed_return + volatility + liquidity) / (price_range + 1e-7)
    
    return alpha_factor
```
