```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference with a 5-day window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference with a 5-day window
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    vol_weighted_typical_price = (df['volume'] * typical_price).rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted typical price
    amount_weighted_typical_price = (df['amount'] * typical_price).rolling(window=5).mean().fillna(0)
    
    # Additional price/volume ratios
    close_to_open_ratio = df['close'] / df['open']
    volume_to_amount_ratio = df['volume'] / df['amount']
    
    # Momentum factor: 5-day return
    momentum_5d = (df['close'] / df['close'].shift(5)).fillna(1)
    
    # Volatility factor: 5-day standard deviation of returns
    daily_returns = df['close'].pct_change().fillna(0)
    volatility_5d = daily_returns.rolling(window=5).std().fillna(0)
    
    # Long-term trend factor: 20-day moving average
    long_term_trend = df['close'].rolling(window=20).mean().fillna(0)
    
    # Relative Strength Index (RSI) for 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean().fillna(0)
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean().fillna(0)
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # the volume-weighted and amount-weighted typical prices, additional price/volume ratios,
    # momentum, volatility, long-term trend, and RSI
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff +
                    vol_weighted_typical_price + amount_weighted_typical_price +
                    close_to_open_ratio + volume_to_amount_ratio +
                    momentum_5d + volatility_5d + long_term_trend + rsi) / (price_range + 1e-7)
    
    return alpha_factor
```
