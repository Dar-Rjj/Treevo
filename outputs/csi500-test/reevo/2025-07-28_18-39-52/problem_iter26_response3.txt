```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Smooth the volume-weighted difference with a moderate window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=7).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Smooth the amount-weighted difference with a moderate window
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=7).mean().fillna(0)
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price with a moderate window
    vol_weighted_typical_price = (df['volume'] * typical_price).rolling(window=7).mean().fillna(0)
    
    # Calculate the amount-weighted typical price with a moderate window
    amount_weighted_typical_price = (df['amount'] * typical_price).rolling(window=7).mean().fillna(0)
    
    # Additional price/volume ratios
    close_to_open_ratio = df['close'] / df['open']
    
    # Momentum and trend indicators
    momentum = df['close'].pct_change(periods=10).fillna(0)
    trend = df['close'].ewm(span=20, adjust=False).mean().fillna(0)
    
    # Combine the smoothed volume-weighted and amount-weighted differences,
    # the volume-weighted and amount-weighted typical prices, the close-to-open ratio, and momentum/trend indicators
    alpha_factor = (smooth_vol_weighted_diff * 0.25 + 
                    smooth_amount_weighted_diff * 0.25 + 
                    vol_weighted_typical_price * 0.1 + 
                    amount_weighted_typical_price * 0.1 + 
                    close_to_open_ratio * 0.1 +
                    momentum * 0.15 + 
                    trend * 0.05) / (price_range + 1e-7)
    
    return alpha_factor
```
