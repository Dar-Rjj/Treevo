```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference with exponential smoothing
    smooth_vol_weighted_diff = vol_weighted_diff.ewm(span=3, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference with exponential smoothing
    smooth_amount_weighted_diff = amount_weighted_diff.ewm(span=3, adjust=False).mean().fillna(0)
    
    # Calculate the volume-weighted typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vol_weighted_typical_price = (df['volume'] * typical_price).ewm(span=3, adjust=False).mean().fillna(0)
    
    # Calculate the amount-weighted typical price
    amount_weighted_typical_price = (df['amount'] * typical_price).ewm(span=3, adjust=False).mean().fillna(0)
    
    # Calculate the RSI of the typical price
    delta = typical_price.diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/3, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/3, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # the volume-weighted and amount-weighted typical prices, and the RSI of the typical price
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff + 
                    vol_weighted_typical_price + amount_weighted_typical_price + rsi) / (price_range + 1e-7)
    
    return alpha_factor
```
