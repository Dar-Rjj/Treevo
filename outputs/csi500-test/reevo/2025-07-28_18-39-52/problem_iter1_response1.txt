```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price momentum as the difference between the current close and the 5-day moving average of the close
    df['5_day_moving_avg'] = df['close'].rolling(window=5).mean()
    price_momentum = df['close'] - df['5_day_moving_avg']
    
    # Calculate the money flow ratio by using the amount (considered as the typical price multiplied by volume, which is similar to 'money flow' but with our available data)
    # Positive money flow days are days where the typical price increases, and negative money flow days are where it decreases. 
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['volume'] * df['typical_price']
    positive_money_flow = df.query('typical_price > typical_price.shift(1)').groupby(level=0)['money_flow'].sum().fillna(0)
    negative_money_flow = df.query('typical_price < typical_price.shift(1)').groupby(level=0)['money_flow'].sum().fillna(0)
    money_flow_ratio = (positive_money_flow / (negative_money_flow.replace(0, 1e-7))) - 1
    
    # Combine the two calculated factors to create a final score
    alpha_factor = price_momentum * money_flow_ratio
    return alpha_factor
```
