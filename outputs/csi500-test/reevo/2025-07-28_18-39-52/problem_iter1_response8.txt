```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and previous close
    df['price_change'] = df['close'].diff()
    
    # Calculate the relative change in volume compared to the mean of last 5 days
    df['volume_change_rel'] = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the volatility using the standard deviation of returns over the past 5 days
    df['volatility'] = df['price_change'].rolling(window=5).std()

    # Factor: Relative price change adjusted by relative volume change and volatility
    factor = df['price_change'] * df['volume_change_rel'] / (df['volatility'] + 1e-7)
    
    return factor
```
