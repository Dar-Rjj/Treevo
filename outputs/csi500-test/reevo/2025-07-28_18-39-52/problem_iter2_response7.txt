```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the moving average of the difference over a 10-day window
    moving_avg_diff = diff_close_open.rolling(window=10).mean()
    
    # Calculate the volatility (standard deviation of log returns)
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=10).std()
    
    # Calculate the volume trend by taking the 10-day moving average of volume
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # The alpha factor is the ratio of the 10-day moving average of the difference
    # between close and open prices to the 10-day volatility of log returns, 
    # multiplied by the volume trend and RSI
    factor = (moving_avg_diff / (volatility + 1e-7)) * volume_trend * rsi
    
    return factor
```
