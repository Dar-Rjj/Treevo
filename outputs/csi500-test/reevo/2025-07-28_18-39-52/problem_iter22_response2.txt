```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df.close) + (df.close - df.low)) / 2)
    
    # Smooth the volume-weighted difference with a short window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=3).mean().fillna(0)
    
    # Amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df.amount * ((df.high - df.open) + (df.open - df.low)) / 2)
    
    # Smooth the amount-weighted difference with a short window
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=3).mean().fillna(0)
    
    # Typical price
    typical_price = (df.high + df.low + df.close) / 3
    
    # Volume-weighted typical price
    vol_weighted_typical_price = (df.volume * typical_price).rolling(window=3).mean().fillna(0)
    
    # Amount-weighted typical price
    amount_weighted_typical_price = (df.amount * typical_price).rolling(window=3).mean().fillna(0)
    
    # Price/volume ratios
    close_to_open_ratio = df.close / df.open
    volume_to_amount_ratio = df.volume / df.amount
    
    # Momentum factor: 5-day return
    momentum_5d = (df.close / df.close.shift(5)).fillna(1)
    
    # Volatility factor: 5-day standard deviation of returns
    daily_returns = df.close.pct_change().fillna(0)
    volatility_5d = daily_returns.rolling(window=5).std().fillna(0)
    
    # Mean price
    mean_price = (df.open + df.close + df.high + df.low) / 4
    
    # Price variance
    price_variance = (df.high - df.low).rolling(window=5).var().fillna(0)
    
    # Additional price-derived features
    price_momentum = df.close.pct_change(periods=5).fillna(0)
    price_volatility = df.close.rolling(window=5).std().fillna(0)
    
    # Short-term volatility: 3-day standard deviation of returns
    short_term_volatility = daily_returns.rolling(window=3).std().fillna(0)
    
    # Dynamic factors
    dynamic_factor_1 = (vol_weighted_diff - amount_weighted_diff) / (price_range + 1e-7)
    dynamic_factor_2 = (vol_weighted_typical_price - amount_weighted_typical_price) / (price_range + 1e-7)
    
    # Final alpha factor: balance the contributions of each feature, including dynamic factors and short-term volatility
    alpha_factor = (smooth_vol_weighted_diff * 0.2 + smooth_amount_weighted_diff * 0.2 + 
                    vol_weighted_typical_price * 0.1 + amount_weighted_typical_price * 0.1 +
                    close_to_open_ratio * 0.1 + volume_to_amount_ratio * 0.05 +
                    momentum_5d * 0.1 + volatility_5d * 0.05 + 
                    mean_price * 0.05 + price_variance * 0.05 +
                    price_momentum * 0.05 + price_volatility * 0.05 +
                    dynamic_factor_1 * 0.05 + dynamic_factor_2 * 0.05 + 
                    short_term_volatility * 0.05) / (price_range + 1e-7)
    
    return alpha_factor
```
