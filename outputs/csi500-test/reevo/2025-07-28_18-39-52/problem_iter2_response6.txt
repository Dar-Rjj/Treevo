```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2) / df['volume'].mean()
    
    # Smooth the volume-weighted difference over a 5-day window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the average price and the close price
    avg_close_diff = avg_price - df['close']
    
    # Smooth the average-close difference over a 5-day window
    smooth_avg_close_diff = avg_close_diff.rolling(window=5).mean().fillna(0)
    
    # Final alpha factor: combination of smoothed volume-weighted difference and smoothed average-close difference, adjusted by the price range
    alpha_factor = (smooth_vol_weighted_diff + smooth_avg_close_diff) / (price_range + 1e-7)
    
    return alpha_factor
```
