```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Relative Strength Index (RSI) for 14 periods
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the liquidity-adjusted volume
    liquidity_adjusted_volume = df['volume'] / (typical_price * df['amount'])
    
    # Calculate the exponential moving average of the typical price with a span of 7
    ema_typical_price = typical_price.ewm(span=7, adjust=False).mean()
    
    # Calculate the momentum factor as the difference between the typical price and its EMA
    momentum_factor = typical_price - ema_typical_price
    
    # Calculate the volatility factor as the standard deviation of the typical price over a 5-day window
    volatility_factor = typical_price.rolling(window=5).std()
    
    # Final alpha factor: combine RSI, momentum, and volatility factors, and the liquidity-adjusted volume
    alpha_factor = (rsi + momentum_factor + volatility_factor + liquidity_adjusted_volume) / 4
    
    return alpha_factor
```
