```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices (price range)
    price_range = df['high'] - df['low']

    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)

    # Calculate a smoothed version of the volume-weighted difference with an exponential moving average
    smooth_vol_weighted_diff = vol_weighted_diff.ewm(span=3, adjust=False).mean().fillna(0)

    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)

    # Calculate a smoothed version of the amount-weighted difference with an exponential moving average
    smooth_amount_weighted_diff = amount_weighted_diff.ewm(span=3, adjust=False).mean().fillna(0)

    # Calculate the volume-weighted typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vol_weighted_typical_price = (df['volume'] * typical_price).ewm(span=3, adjust=False).mean().fillna(0)

    # Calculate the amount-weighted typical price
    amount_weighted_typical_price = (df['amount'] * typical_price).ewm(span=3, adjust=False).mean().fillna(0)

    # Calculate the Average True Range (ATR) over a 14-day period for volatility
    high_low = df['high'] - df['low']
    high_close = (df['high'] - df['close'].shift()).abs()
    low_close = (df['low'] - df['close'].shift()).abs()
    ranges = pd.concat([high_low, high_close, low_close], axis=1)
    atr_14 = ranges.max(axis=1).rolling(window=14).mean().fillna(0)

    # Calculate the trend strength using the Relative Strength Index (RSI)
    gain = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    loss = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    avg_gain_14 = gain.rolling(window=14).mean().fillna(0)
    avg_loss_14 = loss.rolling(window=14).mean().fillna(0)
    rs = avg_gain_14 / avg_loss_14
    rsi_14 = 100 - (100 / (1 + rs))

    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # and the volume-weighted and amount-weighted typical prices, with ATR and RSI
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff + 
                    vol_weighted_typical_price + amount_weighted_typical_price) / (price_range + 1e-7) * atr_14 * rsi_14

    return alpha_factor
```
