```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow ratio
    money_flow_ratio = (df['close'] - df['close'].shift(1)) * typical_price * df['volume']
    
    # Calculate the smoothed money flow ratio by taking the rolling mean
    smoothed_money_flow = money_flow_ratio.rolling(window=10).mean()
    
    return smoothed_money_flow
```
