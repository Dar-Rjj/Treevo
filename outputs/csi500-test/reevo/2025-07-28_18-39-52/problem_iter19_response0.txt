```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the volume-weighted difference with a short window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted average of (high - open) and (open - low)
    amount_weighted_diff = (df['amount'] * ((df['high'] - df['open']) + (df['open'] - df['low'])) / 2)
    
    # Calculate a smoothed version of the amount-weighted difference with a short window
    smooth_amount_weighted_diff = amount_weighted_diff.rolling(window=5).mean().fillna(0)
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price with a short window
    vol_weighted_typical_price = (df['volume'] * typical_price).rolling(window=5).mean().fillna(0)
    
    # Calculate the amount-weighted typical price with a short window
    amount_weighted_typical_price = (df['amount'] * typical_price).rolling(window=5).mean().fillna(0)
    
    # Additional price/volume ratios
    close_to_open_ratio = df['close'] / df['open']
    
    # Momentum factor: 1-day return
    momentum = df['close'].pct_change(periods=1).fillna(0)
    
    # Volatility factor: 5-day standard deviation of returns
    volatility = df['close'].pct_change().rolling(window=5).std().fillna(0)
    
    # Final alpha factor: combine the smoothed volume-weighted and amount-weighted differences,
    # the volume-weighted and amount-weighted typical prices, the close-to-open ratio,
    # the momentum, and the inverse of volatility
    alpha_factor = (smooth_vol_weighted_diff + smooth_amount_weighted_diff + 
                    vol_weighted_typical_price + amount_weighted_typical_price +
                    close_to_open_ratio + momentum - volatility) / (price_range + 1e-7)
    
    return alpha_factor
```
