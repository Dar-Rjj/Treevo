```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the volume-weighted average of (high - close) and (close - low)
    vol_weighted_diff = (df['volume'] * ((df['high'] - df['close']) + (df['close'] - df['low'])) / 2) / df['volume'].rolling(window=5).mean()
    
    # Smooth the volume-weighted difference using a 10-day rolling window
    smooth_vol_weighted_diff = vol_weighted_diff.rolling(window=10).mean().fillna(0)
    
    # Calculate the smoothed price range using a 10-day rolling window
    smooth_price_range = price_range.rolling(window=10).mean().fillna(0)
    
    # Final alpha factor: the ratio of the smoothed volume-weighted difference to the smoothed price range
    alpha_factor = smooth_vol_weighted_diff / (smooth_price_range + 1e-7)
    
    return alpha_factor
```
