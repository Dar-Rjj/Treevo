import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=25).std()
    weight_close = 1 / (volatility * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * 2 * weight_close) + df['amount']) / 6
    heuristics_matrix = smoothed_factor.rolling(window=20).mean()
    return heuristics_matrix
