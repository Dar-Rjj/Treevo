import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change().rolling(window=30).median()
    volatility = df['close'].rolling(window=50).std()
    weight_close = 1 / (volatility * df['volume'].apply(lambda x: max(1, x)).log())
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5 + momentum
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean().ewm(span=10, adjust=False).mean()
    return heuristics_matrix
