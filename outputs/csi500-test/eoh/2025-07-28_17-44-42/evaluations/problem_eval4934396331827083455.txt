def heuristics_v2(df):
    ma_close_20 = df['close'].rolling(window=20).mean()
    ma_close_80 = df['close'].rolling(window=80).mean()
    pct_change_40 = df['close'].pct_change().rolling(window=40).mean()
    heuristics_matrix = (ma_close_20 / ma_close_80) - pct_change_40
    return heuristics_matrix
