def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=5)
    volatility = df['close'].rolling(window=20).std()
    vol_sma = df['volume'].rolling(window=10).mean()
    weight_open = (momentum / (volatility + 0.0001)) * 1.6 * (1 + vol_sma.pct_change())
    weight_high = (momentum / (volatility + 0.0001)) * 1.8 * (1 + vol_sma.pct_change())
    weight_low = (momentum / (volatility + 0.0001)) * 2.0 * (1 + vol_sma.pct_change())
    weight_close = (momentum / (volatility + 0.0001)) * 2.2 * (1 + vol_sma.pct_change())
    smoothed_factor = (df['open'] * weight_open + df['high'] * weight_high + df['low'] * weight_low + df['close'] * weight_close + df['volume'] * df['amount']) / 7
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
