def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=25).std()
    avg_volume = df['volume'].rolling(window=20).mean()
    factor_ratio = df['close'] / avg_volume
    adjusted_factor = factor_ratio * (1 + std_log_returns)
    heuristics_matrix = adjusted_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
