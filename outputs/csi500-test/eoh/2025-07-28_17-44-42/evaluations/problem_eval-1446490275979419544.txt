def heuristics_v2(df):
    # Calculate the average of the daily high and low over a 30-day window
    avg_high_low = (df['high'].rolling(window=30).mean() + df['low'].rolling(window=30).mean()) / 2
    
    # Price movement strength: difference between the current close and the 30-day average of high-low average
    price_strength = df['close'] - avg_high_low
    
    # Price stability: standard deviation of the 30-day closing prices
    price_stability = df['close'].rolling(window=30).std()
    
    # Volume change: percentage change in volume compared to the 30-day average
    volume_change = df['volume'].pct_change(30)
    
    # Composite heuristic score
    heuristics_matrix = 0.5 * (price_strength / price_stability) + 0.5 * volume_change
    
    return heuristics_matrix
