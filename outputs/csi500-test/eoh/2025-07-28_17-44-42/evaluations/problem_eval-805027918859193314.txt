import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 20-day simple moving average of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day standard deviation of closing prices as a measure of volatility
    df['volatility'] = df['close'].rolling(window=5).std()
    
    # Calculate a heuristic based on the volume trend
    df['volume_trend'] = df['volume'].pct_change(periods=1)
    
    # Combine the factors to form the alpha factor
    df['alpha_factor'] = (df['SMA_5'] - df['SMA_20']) / df['volatility'] * df['volume_trend']
    
    # Extract the alpha factor as a Series
    heuristics_matrix = df['alpha_factor']
    
    return heuristics_matrix
