def heuristics_v2(df):
    momentum_60 = df['close'].pct_change(periods=60)
    close_to_min_50 = df['close'] / df['close'].rolling(window=50).min()
    volume_roc_20 = df['volume'].pct_change(periods=20)

    heuristics_matrix = (momentum_60 + close_to_min_50 + volume_roc_20) / 3
    return heuristics_matrix
