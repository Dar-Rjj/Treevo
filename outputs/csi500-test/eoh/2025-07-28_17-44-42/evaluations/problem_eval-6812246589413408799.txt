import pandas as pd

def heuristics_v2(df):
    ma_close_14 = df['close'].rolling(window=14).mean()
    ma_close_200 = df['close'].rolling(window=200).mean()
    ema_price_range_30 = ((df['high'] - df['low']).ewm(span=30, adjust=False).mean())
    volume_change_5 = df['volume'].pct_change(periods=5)
    
    heuristics_matrix = (ma_close_14 / ma_close_200) + ema_price_range_30 + volume_change_5
    return heuristics_matrix
