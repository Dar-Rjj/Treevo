import pandas as pd

def heuristics_v2(df):
    # Calculate momentum and volatility
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=25).std()
    
    # Calculate weight based on momentum and volatility
    weight = 1.0 * (momentum / volatility)
    
    # Calculate the weighted sum of market features
    weighted_sum = (df['open'] * 0.2 + 
                    df['high'] * 0.2 + 
                    df['low'] * 0.2 + 
                    df['close'] * 0.2 + 
                    (df['volume'] * df['amount']) * 0.2) * weight
    
    # Apply exponential moving average for smoothing
    heuristics_matrix = weighted_sum.ewm(span=10, adjust=False).mean()
    
    return heuristics_matrix
