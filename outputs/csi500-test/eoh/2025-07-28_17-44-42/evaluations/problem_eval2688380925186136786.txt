import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=50).std()
    weight_close = momentum / volatility
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 6
    heuristics_matrix = smoothed_factor.ewm(span=30, adjust=False).mean()
    return heuristics_matrix
