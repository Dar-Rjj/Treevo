def heuristics_v2(df):
    momentum = df['close'].pct_change().apply(lambda x: x**0.5)
    volatility = df['close'].rolling(window=35).std()
    weight_close = 4 * (momentum / volatility)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'].apply(np.log) * 2 * df['amount'])) / 8
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
