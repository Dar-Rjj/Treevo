def heuristics_v2(df):
    price_change = df['close'].pct_change()
    volatility = df['close'].rolling(window=50).std()
    dynamic_weight = (price_change / volatility).ewm(span=30, adjust=False).mean()
    weighted_volume = df['volume'] * dynamic_weight
    heuristics_matrix = (df['open'] + df['high'] + df['low'] + df['close'] + weighted_volume) / 5
    heuristics_matrix = heuristics_matrix.ewm(span=25, adjust=False).mean()
    return heuristics_matrix
