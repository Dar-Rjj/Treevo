def heuristics_v2(df):
    price_change = df['close'].pct_change()
    ema_price_change = price_change.ewm(span=14, adjust=False).mean()
    volume_momentum = df['volume'].pct_change().rolling(window=5).sum()
    heuristics_matrix = (ema_price_change + volume_momentum).rolling(window=7).mean()
    return heuristics_matrix
