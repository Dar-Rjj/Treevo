def heuristics_v2(df):
    price_momentum = df['close'] - df['open']
    ma_price_momentum_20 = price_momentum.rolling(window=20).mean()
    ema_volume_50 = df['volume'].ewm(span=50, adjust=False).mean()
    close_roc_14 = df['close'].pct_change(periods=14)
    
    heuristics_matrix = ma_price_momentum_20 + ema_volume_50 + close_roc_14
    return heuristics_matrix
