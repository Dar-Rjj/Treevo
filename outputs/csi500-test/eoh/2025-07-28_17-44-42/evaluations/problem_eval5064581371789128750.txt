import pandas as pd

def heuristics_v2(df):
    high_low_diff = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    avg_close = df['close'].rolling(window=30).mean()
    mov_avg_volume = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (high_low_diff / avg_close) - mov_avg_volume
    return heuristics_matrix
