def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_close = 3 * (momentum * df['volume'])
    adjustment_factor = df['high'] - df['low']
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount'] + adjustment_factor) / 6
    heuristics_matrix = smoothed_factor.ewm(span=30, adjust=False).mean()
    return heuristics_matrix
