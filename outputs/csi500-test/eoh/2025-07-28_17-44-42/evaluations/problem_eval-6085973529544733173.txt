import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=20).std()
    weight_close = 2 * (momentum / volatility)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'] * df['amount'])) / 5
    heuristics_matrix = smoothed_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
