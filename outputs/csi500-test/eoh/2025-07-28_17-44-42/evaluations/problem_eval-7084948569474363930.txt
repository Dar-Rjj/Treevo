import pandas as pd

def heuristics_v2(df):
    price_range = df['high'] - df['low']
    avg_price_range_10 = price_range.rolling(window=10).mean()
    avg_amount_10 = df['amount'].rolling(window=10).mean()
    heuristics_matrix = (avg_price_range_10 * avg_amount_10).rolling(window=5).mean()
    return heuristics_matrix
