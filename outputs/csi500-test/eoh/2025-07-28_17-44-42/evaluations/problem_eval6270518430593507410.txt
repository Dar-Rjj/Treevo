import pandas as pd
import numpy as np

def heuristics_v2(df):
    price_change = df['close'] - df['open']
    avg_price_change_20 = price_change.rolling(window=20).mean()
    log_volume = np.log(df['volume'])
    avg_log_volume_20 = log_volume.rolling(window=20).mean()
    heuristics_matrix = (avg_price_change_20 * avg_log_volume_20).rolling(window=7).mean()
    return heuristics_matrix
