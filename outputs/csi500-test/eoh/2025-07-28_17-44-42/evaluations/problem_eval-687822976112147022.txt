import pandas as pd

def heuristics_v2(df):
    def calculate_momentum(df, window):
        return df['close'].pct_change(window)

    def calculate_volatility(df, window):
        return df['close'].pct_change().rolling(window=window).std()

    def calculate_volume_change(df, window):
        return df['volume'].pct_change(window)
    
    momentum_20 = calculate_momentum(df, 20)
    momentum_60 = calculate_momentum(df, 60)
    volatility_30 = calculate_volatility(df, 30)
    volume_change_10 = calculate_volume_change(df, 10)
    
    heuristics_matrix = 0.5 * momentum_20 + 0.4 * momentum_60 - 0.1 * volatility_30 + 0.05 * volume_change_10
    return heuristics_matrix
