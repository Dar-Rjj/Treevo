import pandas as pd

def heuristics_v2(df):
    std_close_30 = df['close'].rolling(window=30).std()
    momentum_high_60 = df['high'] - df['high'].shift(60)
    rsi_volume_14 = 100 - (100 / (1 + (df['volume'].diff().clip(lower=0).rolling(window=14).mean() / df['volume'].diff().clip(upper=0).rolling(window=14).mean().abs())))
    
    heuristics_matrix = std_close_30 + momentum_high_60 - rsi_volume_14
    return heuristics_matrix
