def heuristics_v2(df):
    momentum = df['close'].pct_change().rolling(window=30).median()
    volatility = df['close'].rolling(window=30).std()
    weight_close = 1 / (volatility * np.log(df['volume'] + 1))
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5 + momentum
    heuristics_matrix = smoothed_factor.rolling(window=20).mean()
    return heuristics_matrix
