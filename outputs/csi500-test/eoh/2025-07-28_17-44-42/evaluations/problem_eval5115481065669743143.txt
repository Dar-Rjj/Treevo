def heuristics_v2(df):
    ma_close_50 = df['close'].rolling(window=50).mean()
    ma_close_200 = df['close'].rolling(window=200).mean()
    ema_price_range_100 = ((df['high'] - df['low']).ewm(span=100, adjust=False).mean())
    volume_change_20 = df['volume'].pct_change(periods=20)
    
    heuristics_matrix = (ma_close_50 / ma_close_200) + ema_price_range_100 + volume_change_20
    return heuristics_matrix
