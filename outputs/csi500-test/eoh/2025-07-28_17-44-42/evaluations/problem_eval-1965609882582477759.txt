import pandas as pd

def heuristics_v2(df):
    relative_strength = df['close'] / df['close'].rolling(window=10).mean()
    volatility = df['close'].rolling(window=20).std()
    weight_close = 3 * (relative_strength / volatility)
    factor = (df['open'] * weight_close + df['high'] + df['low'] + df['close'] + (df['volume'] * df['amount'])) / 5
    heuristics_matrix = factor.rolling(window=10).mean()
    return heuristics_matrix
