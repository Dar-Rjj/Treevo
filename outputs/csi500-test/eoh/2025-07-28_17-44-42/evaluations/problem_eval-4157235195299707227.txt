def heuristics_v2(df):
    ma_close_10 = df['close'].rolling(window=10).mean()
    ma_close_50 = df['close'].rolling(window=50).mean()
    std_daily_returns_60 = df['close'].pct_change().rolling(window=60).std()
    log_diff_close_min_10 = np.log(df['close'] / df['close'].rolling(window=10).min())
    
    heuristics_matrix = (ma_close_10 / ma_close_50) + std_daily_returns_60 + log_diff_close_min_10
    return heuristics_matrix
