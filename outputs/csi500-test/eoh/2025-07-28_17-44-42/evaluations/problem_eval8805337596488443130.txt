import pandas as pd

def heuristics_v2(df):
    positive = df['close'].diff().apply(lambda x: max(x, 0)).rolling(window=14).sum()
    negative = df['close'].diff().apply(lambda x: abs(min(x, 0))).rolling(window=14).sum()
    p_to_n_ratio = positive / negative
    weight_close = 1 / (df['close'].rolling(window=20).std() * p_to_n_ratio)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(alpha=0.1).mean()
    return heuristics_matrix
