def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=25).std()
    weight_close = 2 * (momentum / volatility)
    smoothed_factor = (df['open'] * 0.4 + df['high'] * 0.15 + df['low'] * 0.15 + (df['close'] * weight_close) * 0.1 + (df['volume'] * df['amount']) * 0.2)
    heuristics_matrix = smoothed_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
