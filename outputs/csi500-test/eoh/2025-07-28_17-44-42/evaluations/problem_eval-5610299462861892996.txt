import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=10)
    close_quantile = df['close'].rolling(window=60).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=False)
    adjusted_strength = (momentum + close_quantile) / 2
    heuristics_matrix = adjusted_strength.rolling(window=20).mean()
    return heuristics_matrix
