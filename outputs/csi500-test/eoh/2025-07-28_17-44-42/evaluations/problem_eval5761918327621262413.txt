def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_close = 4 * (momentum * df['volume'])
    adjustment_factor = df['close'].pct_change().rolling(window=14).std()
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount'] + adjustment_factor) / 6
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
