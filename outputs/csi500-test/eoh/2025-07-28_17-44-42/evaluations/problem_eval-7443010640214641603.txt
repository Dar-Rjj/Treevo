import pandas as pd

def heuristics_v2(df):
    avg_volume_to_close = df['volume'].rolling(window=20).mean() / df['close']
    roc_high_low = (df['high'] - df['low']).pct_change()
    combined_factor = avg_volume_to_close + roc_high_low
    heuristics_matrix = combined_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
