import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    rsi = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / df['close'].diff(1).clip(upper=0).rolling(window=14).mean().abs())))
    weight_close = 2 * momentum
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (rsi * df['volume'])) / 5
    heuristics_matrix = smoothed_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
