import pandas as pd

def heuristics_v2(df):
    volatility = df['close'].pct_change().rolling(window=10).std()
    weight_close = 2 * (volatility * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
