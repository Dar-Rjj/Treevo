def heuristics_v2(df):
    price_momentum = df['close'] - df['open']
    ma_price_momentum_20 = price_momentum.rolling(window=20).mean()
    ema_volume_90 = df['volume'].ewm(span=90, adjust=False).mean()
    close_roc_5 = df['close'].pct_change(periods=5)
    
    heuristics_matrix = ma_price_momentum_20 + ema_volume_90 + close_roc_5
    return heuristics_matrix
