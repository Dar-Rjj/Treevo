def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=30)
    volatility = df['close'].rolling(window=50).std()
    weight_close = 1.2 * (momentum / volatility)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'] * df['amount'])) / 6
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
