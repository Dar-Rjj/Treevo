import pandas as pd

def heuristics_v2(df):
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=9).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=9).mean()
    rsi = 100 - (100 / (1 + (gain / loss)))
    weight_close = 1 / (rsi * df['volume'])
    vwp = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    smoothed_factor = (vwp * weight_close + df['amount']) / 2
    heuristics_matrix = smoothed_factor.rolling(window=50).mean()
    return heuristics_matrix
