def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    std_dev_daily_return = daily_return.rolling(window=20).std()
    avg_amount = df['amount'].rolling(window=20).mean()
    adjustment_factor = df['close'] / avg_amount * std_dev_daily_return
    heuristics_matrix = adjustment_factor.ewm(span=15, adjust=False).mean()
    return heuristics_matrix
