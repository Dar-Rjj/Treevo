import pandas as pd

def heuristics_v2(df):
    pct_change_volume = df['volume'].pct_change()
    avg_true_range = (df['high'] - df['low']).rolling(window=14).mean()
    weight_volume = pct_change_volume / avg_true_range
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_volume) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(span=20).mean()
    return heuristics_matrix
