import pandas as pd

def heuristics_v2(df):
    close_change = df['close'].pct_change()
    rsi = 100 - (100 / (1 + (close_change.rolling(window=14).mean() / -close_change.rolling(window=14).std(ddof=0))))
    atr = df['high'].combine(df['low'], max) - df['low'].combine(df['close'].shift(1), min)
    atr = atr.rolling(window=14).mean()
    weight_close = rsi * atr
    smoothed_factor = (df['open'] * weight_close + df['high'] * weight_close + df['low'] * weight_close + df['close'] * weight_close) / 4
    heuristics_matrix = smoothed_factor.rolling(window=10).mean()
    return heuristics_matrix
