def heuristics_v2(df):
    price_ratio = df['high'] / df['low']
    avg_price_ratio_14 = price_ratio.rolling(window=14).mean()
    std_close_30 = df['close'].rolling(window=30).std()
    volume_momentum_5 = df['volume'].diff(periods=5)
    
    heuristics_matrix = (avg_price_ratio_14 + std_close_30) - volume_momentum_5
    return heuristics_matrix
