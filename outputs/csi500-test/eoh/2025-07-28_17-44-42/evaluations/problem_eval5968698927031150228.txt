def heuristics_v2(df):
    momentum = df['close'].pct_change().rolling(window=25).mean() * df['volume']
    volatility = df['close'].rolling(window=25).std()
    weight_close = 1 / (volatility)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + momentum) / 5
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
