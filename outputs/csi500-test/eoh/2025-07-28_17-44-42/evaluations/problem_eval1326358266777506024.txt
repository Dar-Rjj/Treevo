def heuristics_v2(df):
    price_momentum = df['close'] - df['open']
    ma_price_momentum_30 = price_momentum.rolling(window=30).mean()
    ema_amount_60 = df['amount'].ewm(span=60, adjust=False).mean()
    close_roc_10 = df['close'].pct_change(periods=10)
    
    heuristics_matrix = ma_price_momentum_30 + ema_amount_60 + close_roc_10
    return heuristics_matrix
