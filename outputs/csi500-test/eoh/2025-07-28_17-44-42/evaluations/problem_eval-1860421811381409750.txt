import pandas as pd

def heuristics_v2(df):
    rsi = 100 - (100 / (1 + (df['close'].diff().clip(lower=0).rolling(window=14).mean() / df['close'].diff().clip(upper=0).rolling(window=14).mean().abs())))
    atr = df['high'].combine(df['low'], max) - df['low'].combine(df['high'], min)
    weight_close = 1 / (rsi * atr)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=20).mean()
    return heuristics_matrix
