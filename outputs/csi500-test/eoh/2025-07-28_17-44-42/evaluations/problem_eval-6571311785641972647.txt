import pandas as pd

def heuristics_v2(df):
    price_diff = df['high'] - df['low']
    avg_close = df['close'].rolling(window=20).mean()
    adjusted_price = (price_diff + avg_close) / 2
    weighted_sum = (df['volume'] * 0.5 + df['amount'] * 0.5)
    heuristics_matrix = ((adjusted_price + weighted_sum) / 2).rolling(window=20, win_type='linear').mean()
    return heuristics_matrix
