def heuristics_v2(df):
    momentum = df['close'].pct_change()
    inverse_volatility = 1 / (df['close'].rolling(window=50).std() + 1e-6)  # Using inverse of volatility with a small constant to avoid division by zero
    weight_close = 5 * (momentum * inverse_volatility)  # Adjusted weight and incorporated inverse_volatility
    smoothed_factor = (df['open'] * 1.2 + df['high'] * 1.3 + df['low'] * 1.4 + (df['close'] * weight_close) + (df['volume'] * 2 * df['amount'])) / 7  # Different weights and divisor
    heuristics_matrix = smoothed_factor.ewm(span=25, adjust=False).mean()  # Changed span for exponential smoothing
    return heuristics_matrix
