def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_close = (momentum / df['volume']).fillna(0)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=20).mean()
    return heuristics_matrix
