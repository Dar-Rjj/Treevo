def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate volatility (standard deviation of daily returns)
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate a simple volume-to-price ratio
    df['volume_to_price'] = df['volume'] / df['close']
    
    # Weights for the linear combination (example values)
    weights = {'daily_return': 0.4, 'volatility': -0.3, 'volume_to_price': 0.6}
    
    # Generate the heuristics factor
    heuristics_series = df.apply(lambda row: sum(row[col] * weights[col] for col in weights.keys()), axis=1)
    
    # Apply a simple moving average to smooth out short-term fluctuations
    heuristics_matrix = heuristics_series.rolling(window=10).mean().to_frame(name='heuristics_value')
    
    return heuristics_matrix
