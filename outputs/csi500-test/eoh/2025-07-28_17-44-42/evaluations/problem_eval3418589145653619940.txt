def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    std_daily_returns_90 = daily_returns.rolling(window=90).std()
    vol_roc_14 = df['volume'].pct_change(periods=14)
    sma_close_21 = df['close'].rolling(window=21).mean()
    
    heuristics_matrix = std_daily_returns_90 + vol_roc_14 + sma_close_21
    return heuristics_matrix
