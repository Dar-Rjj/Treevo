def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=20)
    volatility = df['close'].rolling(window=100).std()
    weight_close = (momentum / volatility)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=10).mean()
    return heuristics_matrix
