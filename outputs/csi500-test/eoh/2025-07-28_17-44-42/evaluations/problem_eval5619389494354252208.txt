def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=5)
    weight_close = 2 * (momentum * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 6
    heuristics_matrix = smoothed_factor.ewm(span=50, adjust=False).mean()
    return heuristics_matrix
