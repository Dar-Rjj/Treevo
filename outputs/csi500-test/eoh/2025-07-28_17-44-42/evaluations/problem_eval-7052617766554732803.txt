import pandas as pd

def heuristics_v2(df):
    def calculate_momentum(df, window):
        return df['close'].pct_change(window)

    def calculate_volatility(df, window):
        return df['close'].pct_change().rolling(window=window).std()

    def calculate_volume_change(df, window):
        return df['volume'].pct_change(window)
    
    momentum_10 = calculate_momentum(df, 10)
    momentum_30 = calculate_momentum(df, 30)
    volatility_10 = calculate_volatility(df, 10)
    volume_change_5 = calculate_volume_change(df, 5)
    
    heuristics_matrix = 0.4 * momentum_10 + 0.3 * momentum_30 - 0.2 * volatility_10 + 0.1 * volume_change_5
    return heuristics_matrix
