def heuristics_v2(df):
    momentum = df['close'].pct_change().rolling(window=50).mean()
    volatility = df['close'].rolling(window=50).std()
    weight_close = 1 / (volatility * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5 + momentum
    heuristics_matrix = smoothed_factor.ewm(span=30, adjust=False).mean()
    return heuristics_matrix
