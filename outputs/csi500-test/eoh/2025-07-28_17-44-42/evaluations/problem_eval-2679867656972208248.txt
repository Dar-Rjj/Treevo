def heuristics_v2(df):
    ma_close_30 = df['close'].rolling(window=30).mean()
    ma_close_120 = df['close'].rolling(window=120).mean()
    ema_price_range_60 = ((df['high'] - df['low']).ewm(span=60, adjust=False).mean())
    volume_change_15 = df['volume'].pct_change(periods=15)
    
    heuristics_matrix = (ma_close_30 / ma_close_120) + ema_price_range_60 + volume_change_15
    return heuristics_matrix
