import pandas as pd

def heuristics_v2(df):
    std_close = df['close'].rolling(window=30).std()
    avg_close = df['close'].rolling(window=30).mean()
    ema_volume = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (std_close / avg_close) - ema_volume
    return heuristics_matrix
