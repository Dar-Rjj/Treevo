def heuristics_v2(df):
    price_range = df['high'] - df['low']
    avg_price_range_20 = price_range.rolling(window=20).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    heuristics_matrix = (avg_price_range_20 * avg_volume_10).rolling(window=7).mean()
    return heuristics_matrix
