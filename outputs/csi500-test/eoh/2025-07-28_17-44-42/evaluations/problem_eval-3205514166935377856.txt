import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change for close prices over a 5-day period
    roc_close = df['close'].pct_change(5)
    
    # Calculate the RSI for volumes over a 14-day period
    delta = df['volume'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Composite factor: Sum of ROC of close and RSI of volume
    heuristics_matrix = roc_close + rsi_volume
    
    return heuristics_matrix
