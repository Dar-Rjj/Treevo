def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=30).std()  # Changed window from 20 to 30
    weight_close = 3 * (momentum / volatility)  # Increased weight from 2 to 3
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'] * 2 * df['amount'])) / 7  # Adjusted weights and divisor
    heuristics_matrix = smoothed_factor.ewm(span=15, adjust=False).mean()  # Changed span from 10 to 15
    return heuristics_matrix
