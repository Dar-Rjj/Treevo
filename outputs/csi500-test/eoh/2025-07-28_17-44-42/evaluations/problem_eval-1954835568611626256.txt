def heuristics_v2(df):
    price_range = df['high'] - df['low']
    sma_price_range_20 = price_range.rolling(window=20).mean()
    ema_volume_50 = df['volume'].ewm(span=50, adjust=False).mean()
    close_roc_15 = df['close'].pct_change(periods=15)
    
    heuristics_matrix = sma_price_range_20 + ema_volume_50 * (1 + close_roc_15)
    return heuristics_matrix
