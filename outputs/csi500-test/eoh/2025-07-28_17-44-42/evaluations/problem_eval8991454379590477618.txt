import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    # Calculate short and long moving averages for close prices
    short_mavg = df['close'].rolling(window=short_window).mean()
    long_mavg = df['close'].rolling(window=long_window).mean()
    # Momentum: difference between today's closing price and the average of the last 20 days' closing prices
    momentum = df['close'] - long_mavg
    # Composite score calculation (simplified example)
    heuristics_matrix = (short_mavg - long_mavg) + momentum
    return heuristics_matrix
