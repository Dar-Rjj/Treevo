import pandas as pd

def heuristics_v2(df):
    roc_close_10 = df['close'].pct_change(periods=10)
    volume_activity_20 = df['volume'].rolling(window=20).sum()
    rs_price_5_10 = (df['close'].rolling(window=5).mean() / df['close'].rolling(window=10).mean())
    
    heuristics_matrix = roc_close_10 + (volume_activity_20 * rs_price_5_10)
    return heuristics_matrix
