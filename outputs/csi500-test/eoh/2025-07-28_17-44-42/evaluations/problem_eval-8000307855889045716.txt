def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=20)
    daily_volatility = df['close'].pct_change().rolling(window=10).std()
    volume_trend = df['volume'].pct_change(periods=30)
    custom_indicator = (df['close'] - df['open']) / df['open']
    
    heuristics_matrix = momentum + daily_volatility + volume_trend + custom_indicator
    return heuristics_matrix
