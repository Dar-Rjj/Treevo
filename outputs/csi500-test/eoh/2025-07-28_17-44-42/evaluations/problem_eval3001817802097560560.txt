import pandas as pd

def heuristics_v2(df):
    volatility = df['close'].pct_change().rolling(window=15).std()
    weight_close = 3 * (volatility * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 6
    heuristics_matrix = smoothed_factor.rolling(window=10).mean()
    return heuristics_matrix
