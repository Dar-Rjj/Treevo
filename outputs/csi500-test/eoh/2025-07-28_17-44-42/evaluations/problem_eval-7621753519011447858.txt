def heuristics_v2(df):
    momentum = df['close'].pct_change().rolling(window=df['volume'].rolling(window=25).mean().astype(int)).mean()
    weight_close = 1 / (df['volume'] * df['close'].rolling(window=25).std())
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5 + momentum
    heuristics_matrix = smoothed_factor.rolling(window=20).mean()
    return heuristics_matrix
