def heuristics_v2(df):
    trading_range = df['high'] - df['low']
    sma_trading_range_50 = trading_range.rolling(window=50).mean()
    normalized_trading_range = sma_trading_range_50 / df['close']
    
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    rsi_period = 14
    avg_gain = up.rolling(window=rsi_period).mean()
    avg_loss = abs(down.rolling(window=rsi_period).mean())
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    ema_rsi_10 = rsi.ewm(span=10, adjust=False).mean()
    
    roc_volume_20 = df['volume'].pct_change(periods=20)
    
    heuristics_matrix = (normalized_trading_range * ema_rsi_10) + roc_volume_20
    return heuristics_matrix
