def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=25).std()
    atr = (df['high'] - df['low']).rolling(window=14).mean()
    weight_close = 1 / (volatility * df['volume']) * atr
    composite_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = composite_factor.rolling(window=20).mean()
    return heuristics_matrix
