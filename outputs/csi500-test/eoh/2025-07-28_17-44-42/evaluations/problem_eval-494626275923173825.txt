def heuristics_v2(df):
    price_change = df['close'] - df['open']
    rel_strength = price_change / df['volume']
    weight_close = 1 + (rel_strength / rel_strength.rolling(window=30).mean())
    smoothed_factor = (df['open'] * 0.5 + df['high'] * 0.2 + df['low'] * 0.1 + df['close'] * weight_close + df['amount'] * 0.2) / 5
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
