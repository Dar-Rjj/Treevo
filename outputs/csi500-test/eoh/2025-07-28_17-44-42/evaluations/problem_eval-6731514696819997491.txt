def heuristics_v2(df):
    ema_returns = df['close'].pct_change().ewm(span=10, adjust=False).mean()
    std_deviation = df['close'].pct_change().rolling(window=10).std()
    weight_close = ema_returns / std_deviation
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=15).mean()
    return heuristics_matrix
