def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_close = 2 * (momentum * df['volume'])
    lagged_close = df['close'].shift(1)
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (lagged_close * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
