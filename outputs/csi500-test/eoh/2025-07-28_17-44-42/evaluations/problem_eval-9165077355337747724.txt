def heuristics_v2(df):
    def calculate_rsi(series, window):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def calculate_sma(series, window):
        return series.rolling(window=window).mean()
    
    def calculate_roc(series, window):
        return (series / series.shift(window) - 1) * 100
    
    # Calculate RSI for close prices over 14 days
    df['rsi_14d'] = calculate_rsi(df['close'], window=14)
    
    # Calculate 30-day SMA of the daily price range
    df['sma_range_30d'] = calculate_sma(df['high'] - df['low'], window=30)
    
    # Calculate 20-day ROC for close prices
    df['roc_20d'] = calculate_roc(df['close'], window=20)
    
    # Generate the heuristics factor
    heuristics_series = df['rsi_14d'] + df['sma_range_30d'] + df['roc_20d']
    
    heuristics_matrix = heuristics_series.to_frame(name='heuristics_value')
    
    return heuristics_matrix
