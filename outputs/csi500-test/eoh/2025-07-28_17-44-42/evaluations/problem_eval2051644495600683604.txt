import pandas as pd

def heuristics_v2(df):
    sma_close_30 = df['close'].rolling(window=30).mean()
    sma_high_10 = df['high'].rolling(window=10).mean()
    sma_low_20 = df['low'].rolling(window=20).mean()
    heuristics_matrix = (sma_close_30 - sma_high_10) / sma_low_20
    return heuristics_matrix
