import pandas as pd
import ta

def heuristics_v2(df):
    rsi = ta.momentum.RSIIndicator(df['close'], window=14).rsi()
    atr = ta.volatility.AverageTrueRange(df['high'], df['low'], df['close'], window=14).average_true_range()
    weight_close = 3 * (rsi / 100 * df['volume'])
    adjustment_factor = atr
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount'] + adjustment_factor) / 6
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
