import pandas as pd

def heuristics_v2(df):
    volume_change = df['volume'].pct_change()
    price_change = df['close'].pct_change()
    weight_close = (volume_change / (price_change + 1e-8)).fillna(0)  # Avoid division by zero
    heuristics_matrix = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = heuristics_matrix.expanding().mean()
    return heuristics_matrix
