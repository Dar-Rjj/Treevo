import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=10)
    volatility = df['close'].rolling(window=30).std()
    weight_close = (momentum / (volatility + 1e-6)) * 3
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'] / df['amount'])) / 5
    heuristics_matrix = smoothed_factor.ewm(span=15, adjust=False).mean()
    return heuristics_matrix
