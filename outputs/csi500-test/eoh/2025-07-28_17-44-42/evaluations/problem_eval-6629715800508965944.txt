def heuristics_v2(df):
    atr = df['high'].combine(df['low'], max).combine(df['close'], max).subtract(df['low'].combine(df['close'], min)).rolling(window=20).mean()
    rsi = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / df['close'].diff(1).clip(upper=0).rolling(window=14).mean().abs())))
    sma_volume_60 = df['volume'].rolling(window=60).mean()
    
    heuristics_matrix = (atr / df['close']) + rsi - sma_volume_60
    return heuristics_matrix
