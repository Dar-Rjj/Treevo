def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_close = 3 * (momentum * df['volume']) ** -1  # Inverse weighting
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    adjustment = df['close'].rolling(window=20).std()  # Standard deviation of recent prices
    heuristics_matrix = (smoothed_factor * adjustment).ewm(span=20, adjust=False).mean()
    return heuristics_matrix
