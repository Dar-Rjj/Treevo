def heuristics_v2(df):
    # Calculate the difference between open and close prices
    df['price_diff'] = df['close'] - df['open']
    
    # Calculate the relative change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Generate a time-lagged version of the price difference
    df['lagged_price_diff'] = df['price_diff'].shift(1)
    
    # Weights for the linear combination (example values)
    weights = {'price_diff': 0.5, 'volume_change': 0.3, 'lagged_price_diff': 0.2}
    
    # Generate the heuristics factor
    heuristics_series = df.apply(lambda row: sum(row[col] * weights[col] for col in weights.keys()), axis=1)
    
    heuristics_matrix = heuristics_series.to_frame(name='heuristics_value')
    
    return heuristics_matrix
