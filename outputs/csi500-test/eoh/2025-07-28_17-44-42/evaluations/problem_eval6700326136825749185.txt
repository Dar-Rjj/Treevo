def heuristics_v2(df):
    volatility = df['close'].rolling(window=20).std()
    weight_close = 1 / volatility
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean().ewm(span=20, adjust=False).mean() - smoothed_factor.ewm(span=20, adjust=False).mean().ewm(span=40, adjust=False).mean()
    return heuristics_matrix
