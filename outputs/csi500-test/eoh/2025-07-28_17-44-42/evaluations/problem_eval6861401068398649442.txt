def heuristics_v2(df):
    price_to_volume = df['close'] / df['volume']
    avg_price_to_volume_10 = price_to_volume.rolling(window=10).mean()
    daily_return = df['close'].pct_change()
    avg_daily_return_5 = daily_return.rolling(window=5).mean()
    heuristics_matrix = (avg_price_to_volume_10 * avg_daily_return_5).ewm(span=5, adjust=False).mean()
    return heuristics_matrix
