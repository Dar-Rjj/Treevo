def heuristics_v2(df):
    short_momentum = df['close'].pct_change(periods=10)
    long_momentum = df['close'].pct_change(periods=60)
    relative_strength = (short_momentum / long_momentum).fillna(0)
    weight_close = 2 * relative_strength
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + (df['volume'] * df['amount'])) / 5
    heuristics_matrix = smoothed_factor.rolling(window=25).mean()
    return heuristics_matrix
