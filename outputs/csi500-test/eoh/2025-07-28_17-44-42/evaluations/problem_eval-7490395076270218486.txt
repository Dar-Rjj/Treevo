def heuristics_v2(df):
    volatility = df['close'].pct_change().rolling(window=10).std()
    momentum = df['close'].pct_change().rolling(window=10).mean()
    weight_close = 2 * (volatility + momentum) * df['volume']
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=10).mean()
    return heuristics_matrix
