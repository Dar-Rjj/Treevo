import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    atr = df[['high', 'low']].rolling(window=14).std().mean(axis=1)
    weight_close = 1 / (atr * df['volume'])
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.rolling(window=10).mean()
    return heuristics_matrix
