def heuristics_v2(df):
    log_momentum = df['close'].pct_change().apply(lambda x: math.log(1 + x) if x > -1 else 0)
    ewm_volatility = df['close'].rolling(window=40).std().ewm(span=20, adjust=False).mean()
    weight_close = 5 * (log_momentum / ewm_volatility)
    smoothed_factor = (df['open'] * 1.2 + df['high'] * 1.3 + df['low'] * 1.4 + (df['close'] * weight_close) + (df['volume'] * 2 * df['amount'])) / 7
    heuristics_matrix = smoothed_factor.ewm(span=25, adjust=False).mean()
    return heuristics_matrix
