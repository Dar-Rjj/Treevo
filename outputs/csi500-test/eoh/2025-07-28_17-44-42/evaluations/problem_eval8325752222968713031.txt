def heuristics_v2(df):
    momentum = df['close'].pct_change()
    weight_momentum = (momentum / df['volume']).rolling(window=10).mean()
    weighted_sum = (df['open'] * 0.2 + df['high'] * 0.2 + df['low'] * 0.2 + df['close'] * 0.2 + df['amount'] * 0.2) * weight_momentum
    heuristics_matrix = weighted_sum.rolling(window=5).mean()
    return heuristics_matrix
