def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].rolling(window=40).std()  # Changed window from 30 to 40
    weight_close = 4 * (momentum / volatility)  # Increased weight from 3 to 4
    smoothed_factor = (df['open'] * 1.5 + df['high'] * 1.5 + df['low'] * 1.5 + (df['close'] * weight_close) + (df['volume'] * 2.5 * df['amount'])) / 9  # Adjusted weights and divisor
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()  # Changed span from 15 to 20
    return heuristics_matrix
