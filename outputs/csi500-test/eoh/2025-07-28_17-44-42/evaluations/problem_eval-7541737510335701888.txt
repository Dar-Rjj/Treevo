import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate rolling mean and standard deviation of daily returns over 20 days
    df['rolling_mean'] = df['daily_return'].rolling(window=20).mean()
    df['rolling_std'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Heuristic factor: Combine rolling mean, inverse of rolling std, and volume change
    df['heuristic_factor'] = (df['rolling_mean'] + (1 / df['rolling_std'])) * df['volume_change']
    
    # Prepare the output series
    heuristics_matrix = df['heuristic_factor'].dropna()
    
    return heuristics_matrix
