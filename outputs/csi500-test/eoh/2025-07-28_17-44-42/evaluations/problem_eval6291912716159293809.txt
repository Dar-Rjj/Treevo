def heuristics_v2(df):
    avg_ohl = (df['open'] + df['high'] + df['low']) / 3
    close_to_avg_ratio = df['close'] / avg_ohl.rolling(window=10).mean()
    volume_adjusted_factor = close_to_avg_ratio * df['volume'].rolling(window=30).mean()
    heuristics_matrix = volume_adjusted_factor.ewm(span=14, adjust=False).mean()
    return heuristics_matrix
