def heuristics_v2(df):
    atr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1).rolling(window=14).mean()
    weight_close = df['volume'] / atr
    smoothed_factor = (df['open'] + df['high'] + df['low'] + (df['close'] * weight_close) + df['amount']) / 5
    heuristics_matrix = smoothed_factor.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
