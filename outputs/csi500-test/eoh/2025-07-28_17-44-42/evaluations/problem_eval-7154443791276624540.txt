def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=10)
    volatility = df['close'].rolling(window=30).std()
    weight_open = (momentum / volatility) * 1.5
    weight_high = (momentum / volatility) * 1.8
    weight_low = (momentum / volatility) * 2.0
    weight_close = (momentum / volatility) * 2.2
    smoothed_factor = (df['open'] * weight_open + df['high'] * weight_high + df['low'] * weight_low + df['close'] * weight_close + df['volume'] * df['amount']) / 7
    heuristics_matrix = smoothed_factor.ewm(span=25, adjust=False).mean()
    return heuristics_matrix
