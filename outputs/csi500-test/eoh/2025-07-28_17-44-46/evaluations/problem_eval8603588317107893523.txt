import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 100-day cumulative volume
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    cum_vol_100 = df['volume'].rolling(window=100).sum()
    
    # Generate the new factor by dividing the 30-day cumulative volume by the 100-day cumulative volume
    factor_volume_ratio = cum_vol_30 / cum_vol_100
    
    # Calculate the 7-day Average True Range (ATR)
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift(1))
    low_close_prev = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    atr = tr.rolling(window=7).mean()
    
    # Combine the volume ratio with the ATR
    heuristics_matrix = factor_volume_ratio * atr
    
    return heuristics_matrix
