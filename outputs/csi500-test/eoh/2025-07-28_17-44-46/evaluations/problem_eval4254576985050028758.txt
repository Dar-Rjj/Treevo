def heuristics_v2(df):
    # Calculate the 30-day and 120-day moving averages
    ma_30 = df['amount'].rolling(window=30).mean()
    ma_120 = df['amount'].rolling(window=120).mean()
    
    # Generate the new factor by dividing the 30-day moving average by the 120-day moving average
    heuristics_matrix = ma_30 / ma_120
    
    return heuristics_matrix
