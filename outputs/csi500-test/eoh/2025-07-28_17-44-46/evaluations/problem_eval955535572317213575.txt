import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day EMA of the closing price
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day SMA of the highest high
    max_high_50 = df['high'].rolling(window=50).max()
    
    # Generate the ratio of the 20-day EMA close to the 50-day SMA high
    price_ratio = ema_close_20 / max_high_50
    
    # Calculate the 60-day EMA of the negative daily returns
    daily_returns = df['close'].pct_change()
    ema_neg_ret_60 = (-daily_returns).ewm(span=60, adjust=False).mean()
    
    # Combine the price ratio with the 60-day EMA of negative returns
    heuristics_matrix = price_ratio * ema_neg_ret_60
    
    return heuristics_matrix
