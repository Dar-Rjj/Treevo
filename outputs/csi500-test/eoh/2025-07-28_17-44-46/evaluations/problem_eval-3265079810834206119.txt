import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 60-day cumulative amount
    cum_amount_20 = df['amount'].rolling(window=20).sum()
    cum_amount_60 = df['amount'].rolling(window=60).sum()
    
    # Generate the new factor by dividing the 20-day cumulative amount by the 60-day cumulative amount
    heuristics_matrix = cum_amount_20 / cum_amount_60
    
    return heuristics_matrix
