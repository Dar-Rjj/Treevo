def heuristics_v2(df):
    close_pct_change = df['close'].pct_change()
    volume_pct_change = df['volume'].pct_change()
    close_mavg = close_pct_change.rolling(window=5).mean()
    volume_mavg = volume_pct_change.rolling(window=5).mean()
    heuristics_matrix = close_mavg - volume_mavg
    return heuristics_matrix
