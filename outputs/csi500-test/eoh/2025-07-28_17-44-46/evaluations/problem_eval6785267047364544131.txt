def heuristics_v2(df):
    df['price_change'] = df['close'].diff()
    df['volume_change'] = df['volume'].diff()
    df['volatility'] = df['close'].rolling(window=5).std()
    df['weighted_price_change'] = df['price_change'] / (df['volatility'] + 1e-6)
    df['weighted_volume_change'] = df['volume_change'] / (df['volatility'] + 1e-6)
    heuristics_matrix = (df['weighted_price_change'] * 0.6) + (df['weighted_volume_change'] * 0.4)
    return heuristics_matrix
