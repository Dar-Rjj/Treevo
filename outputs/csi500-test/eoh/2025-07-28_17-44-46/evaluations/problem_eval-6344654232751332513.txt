def heuristics_v2(df):
    # Calculate the ratio between the daily high price and the close price
    high_close_ratio = df['high'] / df['close']
    
    # Calculate the 5-day moving average of the high-close ratio
    high_close_ratio_mavg = high_close_ratio.rolling(window=5).mean()
    
    # Calculate the 5-day average volume
    avg_volume_5day = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio between the daily trading volume and the 5-day average volume
    volume_avg_ratio = df['volume'] / avg_volume_5day
    
    # Combine the two factors into one
    heuristics_matrix = high_close_ratio_mavg + volume_avg_ratio
    
    return heuristics_matrix
