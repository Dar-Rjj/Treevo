import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    tr = pd.DataFrame({'hl': df['high'] - df['low'], 'hc': abs(df['high'] - df['close'].shift(1)), 'lc': abs(df['low'] - df['close'].shift(1))})
    tr['tr'] = tr.max(axis=1)
    
    # Calculate the 14-day Exponential Moving Average of True Range (EMATR)
    ematr = tr['tr'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 30-day cumulative sum of volume
    cum_vol = df['volume'].rolling(window=30).sum()
    
    # Generate the new factor by dividing EMATR by the 30-day cumulative sum of volume
    heuristics_matrix = ematr / cum_vol
    
    return heuristics_matrix
