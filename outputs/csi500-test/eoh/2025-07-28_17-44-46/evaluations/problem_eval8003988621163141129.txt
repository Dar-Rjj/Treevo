import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 150-day SMA of closing prices
    sma_close_30 = df['close'].rolling(window=30).mean()
    sma_close_150 = df['close'].rolling(window=150).mean()
    
    # Generate the new factor by dividing the 30-day SMA of closing prices by the 150-day SMA of closing prices
    heuristics_matrix = sma_close_30 / sma_close_150
    
    return heuristics_matrix
