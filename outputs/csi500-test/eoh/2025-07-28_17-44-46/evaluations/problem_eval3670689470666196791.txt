def heuristics_v2(df):
    # Calculate the daily price change percentage
    df['price_change_pct'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the exponential moving average of the daily price change percentage over a 5-day window
    df['pct_ema'] = df['price_change_pct'].ewm(span=5).mean()
    
    # Calculate the exponential moving average of the daily trading volume over a 5-day window
    df['volume_ema'] = df['volume'].ewm(span=5).mean()
    
    # Combine the two factors into one using a simple weighted sum, weights are set arbitrarily for demonstration
    heuristics_matrix = 0.6 * df['pct_ema'] + 0.4 * df['volume_ema']
    
    return heuristics_matrix
