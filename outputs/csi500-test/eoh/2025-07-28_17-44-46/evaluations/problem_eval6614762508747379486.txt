import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 50-day EMA of the close price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Generate the new factor by subtracting the 50-day EMA from the 10-day EMA
    heuristics_matrix = ema_10 - ema_50
    
    return heuristics_matrix
