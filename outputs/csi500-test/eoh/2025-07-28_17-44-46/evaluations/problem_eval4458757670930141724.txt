import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 100-day Exponential Moving Average (EMA) of volume
    ema_vol_20 = df['volume'].ewm(span=20, adjust=False).mean()
    ema_vol_100 = df['volume'].ewm(span=100, adjust=False).mean()
    
    # Generate the new factor by dividing the 20-day EMA volume by the 100-day EMA volume
    factor_volume_ratio = ema_vol_20 / ema_vol_100
    
    # Calculate the 30-day Standard Deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_30 = daily_returns.rolling(window=30).std()
    
    # Combine the volume ratio with the standard deviation of returns
    heuristics_matrix = factor_volume_ratio * std_dev_30
    
    return heuristics_matrix
