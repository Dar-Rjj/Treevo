def heuristics_v2(df):
    # Calculate the 30-day and 120-day EWMA of the product of amount and volume
    ewma_amt_vol_30 = (df['amount'] * df['volume']).ewm(span=30).mean()
    ewma_amt_vol_120 = (df['amount'] * df['volume']).ewm(span=120).mean()
    
    # Generate the new factor by dividing the 30-day EWMA by the 120-day EWMA
    heuristics_matrix = ewma_amt_vol_30 / ewma_amt_vol_120
    
    return heuristics_matrix
