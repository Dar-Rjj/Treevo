import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 100-day Simple Moving Average (SMA) of the close price
    sma_close_20 = df['close'].rolling(window=20).mean()
    sma_close_100 = df['close'].rolling(window=100).mean()
    
    # Calculate the 50-day Simple Moving Average (SMA) of the volume
    sma_vol_50 = df['volume'].rolling(window=50).mean()
    
    # Generate the new factor by dividing the difference of the 20-day SMA and 100-day SMA of the close price
    # by the 50-day SMA of the volume
    heuristics_matrix = (sma_close_20 - sma_close_100) / sma_vol_50
    
    return heuristics_matrix
