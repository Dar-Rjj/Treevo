def heuristics_v2(df):
    df['ema20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema60'] = df['close'].ewm(span=60, adjust=False).mean()
    
    df['momentum_factor'] = (df['close'] / df[['ema20', 'ema60']].min(axis=1)) * (df['volume'] / df['volume'].shift(1))
    
    df['smoothed_momentum'] = df['momentum_factor'].ewm(span=10, adjust=False).mean()
    
    heuristics_matrix = df['smoothed_momentum']
    
    return heuristics_matrix
