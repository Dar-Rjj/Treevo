def heuristics_v2(df):
    # Calculate the 30-day and 100-day EMA of volume
    ema_vol_30 = df['volume'].ewm(span=30).mean()
    ema_vol_100 = df['volume'].ewm(span=100).mean()
    
    # Generate the new factor by dividing the 30-day EMA volume by the 100-day EMA volume
    heuristics_matrix = ema_vol_30 / ema_vol_100
    
    return heuristics_matrix
