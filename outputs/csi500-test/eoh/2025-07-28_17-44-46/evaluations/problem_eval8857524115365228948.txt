import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    tr = pd.DataFrame({'hl': df['high'] - df['low'], 'hc': abs(df['high'] - df['close'].shift(1)), 'lc': abs(df['low'] - df['close'].shift(1))})
    tr['tr'] = tr.max(axis=1)
    
    # Calculate Exponentially Weighted Moving Average (EWMA) of True Range over the last 14 days
    atr_ewma = tr['tr'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 5-day Simple Moving Average (SMA) of the Close Price
    ma_close = df['close'].rolling(window=5).mean()
    
    # Generate the new factor by dividing EWMA of TR by the SMA of the Close Price
    heuristics_matrix = atr_ewma / ma_close
    
    return heuristics_matrix
