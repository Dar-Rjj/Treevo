def heuristics_v2(df):
    ema_vol_30 = df['volume'].ewm(span=30, adjust=False).mean()
    ema_vol_120 = df['volume'].ewm(span=120, adjust=False).mean()
    factor_volume_ratio = ema_vol_30 / ema_vol_120
    daily_returns = df['close'].pct_change()
    std_dev_40 = daily_returns.rolling(window=40).std()
    mean_return_40 = daily_returns.rolling(window=40).mean()
    cv_40 = std_dev_40 / mean_return_40
    heuristics_matrix = factor_volume_ratio * cv_40
    return heuristics_matrix
