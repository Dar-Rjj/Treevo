import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day and 60-day EWMA of the 'close' price
    ewma_14 = df['close'].ewm(span=14).mean()
    ewma_60 = df['close'].ewm(span=60).mean()
    
    # Generate the new factor by dividing the 14-day EWMA by the 60-day EWMA
    heuristics_matrix = ewma_14 / ewma_60
    
    return heuristics_matrix
