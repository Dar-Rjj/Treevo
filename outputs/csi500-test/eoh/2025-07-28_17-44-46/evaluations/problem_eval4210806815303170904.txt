def heuristics_v2(df):
    # Calculate 10-day momentum
    momentum_10 = df['close'].pct_change(periods=10)
    
    # Calculate the ratio of current volume to the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Calculate the 5-day volatility of the closing price
    daily_returns = df['close'].pct_change()
    volatility_5 = daily_returns.rolling(window=5).std()
    
    # Generate heuristics matrix
    heuristics_matrix = (momentum_10 * volume_ratio) * volatility_5
    
    return heuristics_matrix
