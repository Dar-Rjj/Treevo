import pandas as pd

def heuristics_v2(df):
    # Calculate the 75-day and 150-day cumulative volume
    cum_vol_75 = df['volume'].rolling(window=75).sum()
    cum_vol_150 = df['volume'].rolling(window=150).sum()
    
    # Generate the new factor by dividing the 75-day cumulative volume by the 150-day cumulative volume
    heuristics_matrix = cum_vol_75 / cum_vol_150
    
    return heuristics_matrix
