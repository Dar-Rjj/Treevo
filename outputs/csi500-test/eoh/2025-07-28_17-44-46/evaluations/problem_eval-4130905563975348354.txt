def heuristics_v2(df):
    # Calculate the 10-day and 60-day moving average of volume
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    avg_vol_60 = df['volume'].rolling(window=60).mean()
    
    # Generate the new factor by dividing the 10-day average volume by the 60-day average volume
    heuristics_matrix = avg_vol_10 / avg_vol_60
    
    return heuristics_matrix
