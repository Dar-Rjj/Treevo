import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 60-day cumulative volume
    cum_volume_20 = df['volume'].rolling(window=20).sum()
    cum_volume_60 = df['volume'].rolling(window=60).sum()
    
    # Generate the new factor by dividing the 20-day cumulative volume by the 60-day cumulative volume
    heuristics_matrix = cum_volume_20 / cum_volume_60
    
    return heuristics_matrix
