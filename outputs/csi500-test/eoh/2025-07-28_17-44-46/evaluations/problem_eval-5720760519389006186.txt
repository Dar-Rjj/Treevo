def heuristics_v2(df):
    # Calculate daily returns
    returns = df['close'].pct_change()
    
    # Calculate the 30-day and 120-day cumulative returns
    cum_return_30 = (1 + returns).rolling(window=30).apply(lambda x: x.prod()) - 1
    cum_return_120 = (1 + returns).rolling(window=120).apply(lambda x: x.prod()) - 1
    
    # Generate the new factor by dividing the 30-day cumulative return by the 120-day cumulative return
    heuristics_matrix = cum_return_30 / cum_return_120
    
    return heuristics_matrix
