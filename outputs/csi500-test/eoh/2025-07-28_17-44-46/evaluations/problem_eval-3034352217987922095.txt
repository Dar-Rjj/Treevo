import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day cumulative amount
    cum_amount_20 = df['amount'].rolling(window=20).sum()
    
    # Calculate the 50-day EMA of volume
    ema_vol_50 = df['volume'].ewm(span=50).mean()
    
    # Generate the new factor by dividing the 20-day cumulative amount by the 50-day EMA volume
    heuristics_matrix = cum_amount_20 / ema_vol_50
    
    return heuristics_matrix
