import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'] - df['close'].shift(1)
    
    # Calculate the 20-day EMA of the price change
    ema_price_change = price_change.ewm(span=20, adjust=False).mean()
    
    # Calculate the daily trading range
    daily_range = df['high'] - df['low']
    
    # Calculate the 60-day SMA of the daily trading range
    sma_daily_range = daily_range.rolling(window=60).mean()
    
    # Generate the new factor by dividing the 20-day EMA of the price change by the 60-day SMA of the daily trading range
    heuristics_matrix = ema_price_change / sma_daily_range
    
    return heuristics_matrix
