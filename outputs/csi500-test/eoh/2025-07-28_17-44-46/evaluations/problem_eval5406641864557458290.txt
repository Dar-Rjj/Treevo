def heuristics_v2(df):
    # Calculate the daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the moving average of the daily price range over a 5-day window
    df['range_ma'] = df['price_range'].rolling(window=5).mean()
    
    # Calculate the moving average of the daily trading volume over a 5-day window
    df['volume_ma'] = df['volume'].rolling(window=5).mean()
    
    # Combine the two factors into one using a simple weighted sum, weights are set arbitrarily for demonstration
    heuristics_matrix = 0.7 * df['range_ma'] + 0.3 * df['volume_ma']
    
    return heuristics_matrix
