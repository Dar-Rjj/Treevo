import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) over a 14-day period to measure the speed and change of price movements
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the percentage change in volume over the past 5 days
    pct_change_volume = df['volume'].pct_change(periods=5)
    
    # Generate the new factor by multiplying the RSI with the percentage change in volume
    heuristics_matrix = rsi * pct_change_volume
    
    return heuristics_matrix
