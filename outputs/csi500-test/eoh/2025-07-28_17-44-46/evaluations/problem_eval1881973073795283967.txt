import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 150-day cumulative volume
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    cum_vol_150 = df['volume'].rolling(window=150).sum()
    
    # Generate the new factor by dividing the 30-day cumulative volume by the 150-day cumulative volume
    factor_volume_ratio = cum_vol_30 / cum_vol_150
    
    # Calculate the 10-day Exponentially Weighted Moving Average (EWMA) of the True Range
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift(1))
    low_close_prev = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    ewma_tr = tr.ewm(span=10).mean()
    
    # Combine the volume ratio with the EWMA of the True Range
    heuristics_matrix = factor_volume_ratio * ewma_tr
    
    return heuristics_matrix
