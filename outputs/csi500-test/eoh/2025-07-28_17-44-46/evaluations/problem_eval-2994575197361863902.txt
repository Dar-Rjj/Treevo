import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 150-day Exponential Moving Average (EMA) of close price
    ema_close_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_close_150 = df['close'].ewm(span=150, adjust=False).mean()
    
    # Generate the new factor by subtracting the 150-day EMA close from the 30-day EMA close
    factor_price_difference = ema_close_30 - ema_close_150
    
    # Calculate the 20-day Average of daily returns
    daily_returns = df['close'].pct_change()
    avg_ret_20 = daily_returns.rolling(window=20).mean()
    
    # Combine the price difference with the average of returns
    heuristics_matrix = factor_price_difference * avg_ret_20
    
    return heuristics_matrix
