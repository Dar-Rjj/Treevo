def heuristics_v2(df):
    # Calculate daily returns
    df['Return'] = df['close'].pct_change()
    
    # Calculate momentum as the percentage change in price over the last 20 days
    df['Momentum'] = df['close'].pct_change(periods=20)
    
    # Compute the rolling standard deviation of daily returns over the last 20 days
    df['Volatility'] = df['Return'].rolling(window=20).std()
    
    # Volatility-adjusted Momentum
    df['HeuristicFactor'] = df['Momentum'] / df['Volatility']
    
    heuristics_matrix = df['HeuristicFactor'].dropna()
    return heuristics_matrix
