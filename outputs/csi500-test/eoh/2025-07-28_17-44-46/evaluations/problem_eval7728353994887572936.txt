def heuristics_v2(df):
    # Calculate the 30-day and 150-day average volume
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    avg_vol_150 = df['volume'].rolling(window=150).mean()
    
    # Generate the new factor by dividing the 30-day average volume by the 150-day average volume
    heuristics_matrix = avg_vol_30 / avg_vol_150
    
    return heuristics_matrix
