def heuristics_v2(df):
    # Calculate the 14-day Exponential Moving Average (EMA) of the volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 20-day Simple Moving Average (SMA) of the high price
    sma_high = df['high'].rolling(window=20).mean()
    
    # Generate the new factor by dividing the EMA of volume by the SMA of high price
    heuristics_matrix = ema_volume / sma_high
    
    return heuristics_matrix
