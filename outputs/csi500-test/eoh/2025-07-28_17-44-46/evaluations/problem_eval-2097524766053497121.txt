def heuristics_v2(df):
    # Calculate the 50-day and 200-day EMA of volume
    ema_vol_50 = df['volume'].ewm(span=50).mean()
    ema_vol_200 = df['volume'].ewm(span=200).mean()
    
    # Generate the new factor by calculating the difference between the 50-day and 200-day EMA volumes,
    # then dividing this difference by the 200-day EMA volume
    heuristics_matrix = (ema_vol_50 - ema_vol_200) / ema_vol_200
    
    return heuristics_matrix
