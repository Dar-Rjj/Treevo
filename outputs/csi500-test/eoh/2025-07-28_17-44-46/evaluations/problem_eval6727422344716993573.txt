def heuristics_v2(df):
    # Calculate 20-day and 50-day exponential moving averages of the close price
    df['ema20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute the trend-following factor
    df['trend_factor'] = (df['close'] / df[['ema20', 'ema50']].min(axis=1)) * (df['amount'] / df['amount'].shift(1))
    
    # Apply an Exponential Moving Average (EMA) for smoothing
    df['smoothed_trend'] = df['trend_factor'].ewm(span=7, adjust=False).mean()
    
    # Prepare the output
    heuristics_matrix = df['smoothed_trend']
    
    return heuristics_matrix
