def heuristics_v2(df):
    # Calculate the 60-day and 200-day simple moving averages of closing prices
    sma_60 = df['close'].rolling(window=60).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Generate the difference between the 60-day and 200-day SMAs
    sma_diff = sma_60 - sma_200
    
    # Calculate the 14-day relative strength index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the SMA difference with the RSI
    heuristics_matrix = sma_diff * rsi
    
    return heuristics_matrix
