def heuristics_v2(df):
    df['sma5'] = df['close'].rolling(window=5).mean()
    df['sma10'] = df['close'].rolling(window=10).mean()
    
    df['sentiment_factor'] = (df['close'] / df[['sma5', 'sma10']].min(axis=1)) * (df['volume'] / df['volume'].shift(1))
    
    df['smoothed_sentiment'] = df['sentiment_factor'].rolling(window=5).mean()
    
    heuristics_matrix = df['smoothed_sentiment']
    
    return heuristics_matrix
