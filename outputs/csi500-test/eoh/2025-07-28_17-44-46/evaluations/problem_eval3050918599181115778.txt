def heuristics_v2(df):
    # Calculate the 60-day and 180-day cumulative volume
    cum_vol_60 = df['volume'].rolling(window=60).sum()
    cum_vol_180 = df['volume'].rolling(window=180).sum()
    
    # Generate the new factor by dividing the 60-day cumulative volume by the 180-day cumulative volume
    heuristics_matrix = cum_vol_60 / cum_vol_180
    
    return heuristics_matrix
