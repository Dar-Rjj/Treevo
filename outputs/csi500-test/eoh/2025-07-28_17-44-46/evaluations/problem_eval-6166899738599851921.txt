def heuristics_v2(df):
    # Calculate the daily difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute the 5-day moving average of the high-low difference
    df['high_low_mavg_5'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Compute the 10-day moving average of the volume
    df['volume_mavg_10'] = df['volume'].rolling(window=10).mean()
    
    # Generate the new factor by dividing the 5-day moving average of the high-low difference by the 10-day moving average of volume
    df['new_factor'] = df['high_low_mavg_5'] / df['volume_mavg_10']
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
