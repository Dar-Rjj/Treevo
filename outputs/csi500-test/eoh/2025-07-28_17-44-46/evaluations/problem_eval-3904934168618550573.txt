import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 500-day cumulative amount
    cum_amount_20 = df['amount'].rolling(window=20).sum()
    cum_amount_500 = df['amount'].rolling(window=500).sum()
    
    # Calculate the 20-day moving average price
    ma_price_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the percentage change in the 20-day moving average price
    pct_change_ma_price_20 = ma_price_20.pct_change().fillna(0)
    
    # Generate the new factor by dividing the (20-day cumulative amount * (1 + percentage change in 20-day MA price)) by the 500-day cumulative amount
    heuristics_matrix = (cum_amount_20 * (1 + pct_change_ma_price_20)) / cum_amount_500
    
    return heuristics_matrix
