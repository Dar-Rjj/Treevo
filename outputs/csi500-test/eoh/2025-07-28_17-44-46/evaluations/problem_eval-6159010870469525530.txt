def heuristics_v2(df):
    # Calculate the 20-day moving average of the daily range (high - low)
    df['avg_range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Find the minimum volume over the same 20-day window
    df['min_volume_20'] = df['volume'].rolling(window=20).min()
    
    # Generate the new factor by adjusting the average daily range with the ratio of today's volume to the minimum volume
    df['new_factor'] = df['avg_range_20'] * (df['volume'] / df['min_volume_20'])
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
