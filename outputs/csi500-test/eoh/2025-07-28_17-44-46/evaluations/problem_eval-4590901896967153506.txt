import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 50-day Exponential Moving Average (EMA) of volume
    ema_vol_10 = df['volume'].ewm(span=10, adjust=False).mean()
    ema_vol_50 = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Generate the new factor by dividing the 10-day EMA volume by the 50-day EMA volume
    factor_volume_ratio = ema_vol_10 / ema_vol_50
    
    # Calculate the 20-day Average True Range (ATR)
    tr = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    atr_20 = tr.rolling(window=20).mean()
    
    # Combine the volume ratio with the ATR
    heuristics_matrix = factor_volume_ratio * atr_20
    
    return heuristics_matrix
