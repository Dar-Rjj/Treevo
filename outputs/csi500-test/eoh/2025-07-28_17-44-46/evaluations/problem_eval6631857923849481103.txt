def heuristics_v2(df):
    # Calculate the 20-day and 100-day cumulative volume
    cum_vol_20 = df['volume'].rolling(window=20).sum()
    cum_vol_100 = df['volume'].rolling(window=100).sum()
    
    # Generate the new factor by dividing the 20-day cumulative volume by the 100-day cumulative volume
    heuristics_matrix = cum_vol_20 / cum_vol_100
    
    return heuristics_matrix
