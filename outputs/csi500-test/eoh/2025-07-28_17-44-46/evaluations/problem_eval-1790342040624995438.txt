import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 150-day Exponential Moving Average (EMA) of volume
    ema_vol_30 = df['volume'].ewm(span=30, adjust=False).mean()
    ema_vol_150 = df['volume'].ewm(span=150, adjust=False).mean()
    
    # Generate the new factor by dividing the 30-day EMA volume by the 150-day EMA volume
    factor_volume_ratio = ema_vol_30 / ema_vol_150
    
    # Calculate the 20-day Standard Deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    
    # Combine the volume ratio with the standard deviation of returns
    heuristics_matrix = factor_volume_ratio * std_dev_20
    
    return heuristics_matrix
