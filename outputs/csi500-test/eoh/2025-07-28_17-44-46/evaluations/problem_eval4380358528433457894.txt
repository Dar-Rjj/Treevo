def heuristics_v2(df):
    # Calculate the 20-day Exponential Moving Average (EMA) of the close price
    df['ema20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the momentum factor as the difference between the close price and its 20-day EMA
    df['momentum_factor'] = df['close'] - df['ema20']
    
    # Calculate the True Range
    df['true_range'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].max(axis=1)
    
    # Calculate the 20-day Average True Range (ATR)
    df['atr_20'] = df['true_range'].rolling(window=20).mean()
    
    # Calculate the average volume over the 20-day period
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Generate the new factor by scaling the momentum factor with ATR and the ratio of today's volume to the average volume
    df['new_factor'] = (df['momentum_factor'] / df['atr_20']) * (df['volume'] / df['avg_volume_20'])
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
