import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 200-day cumulative amount
    cum_amount_5 = df['amount'].rolling(window=5).sum()
    cum_amount_200 = df['amount'].rolling(window=200).sum()
    
    # Generate the new factor by dividing the 5-day cumulative amount by the 200-day cumulative amount
    heuristics_matrix = cum_amount_5 / cum_amount_200
    
    return heuristics_matrix
