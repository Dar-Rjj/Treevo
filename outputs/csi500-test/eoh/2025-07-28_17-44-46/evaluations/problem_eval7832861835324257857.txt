def heuristics_v2(df):
    # Calculate the 30-day and 100-day EMA of closing prices
    ema_close_30 = df['close'].ewm(span=30).mean()
    ema_close_100 = df['close'].ewm(span=100).mean()
    
    # Generate the new factor by dividing the 30-day EMA of closing prices by the 100-day EMA of closing prices
    heuristics_matrix = ema_close_30 / ema_close_100
    
    return heuristics_matrix
