import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 5-day average daily return
    avg_5day_return = df['daily_return'].rolling(window=5).mean()
    
    # Calculate 30-day standard deviation of daily returns
    std_30day_return = df['daily_return'].rolling(window=30).std()
    
    # Generate the new factor by dividing the 5-day average daily return by the 30-day standard deviation of daily returns
    heuristics_matrix = avg_5day_return / std_30day_return
    
    return heuristics_matrix
