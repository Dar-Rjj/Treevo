def heuristics_v2(df):
    # Calculate the 20-day exponential moving average of the daily range (high - low)
    df['ema_range_20'] = (df['high'] - df['low']).ewm(span=20, adjust=False).mean()
    
    # Find the maximum volume over the same 20-day window
    df['max_volume_20'] = df['volume'].rolling(window=20).max()
    
    # Generate the new factor by adjusting the EMA daily range with the ratio of today's volume to the maximum volume
    df['new_factor'] = df['ema_range_20'] * (df['volume'] / df['max_volume_20'])
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
