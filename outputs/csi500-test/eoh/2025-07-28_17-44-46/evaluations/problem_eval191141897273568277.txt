def heuristics_v2(df):
    # Calculate the 10-day and 60-day cumulative volume
    cum_vol_10 = df['volume'].rolling(window=10).sum()
    cum_vol_60 = df['volume'].rolling(window=60).sum()
    
    # Generate the new factor by dividing the 10-day cumulative volume by the 60-day cumulative volume
    heuristics_matrix = cum_vol_10 / cum_vol_60
    
    return heuristics_matrix
