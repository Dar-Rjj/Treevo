import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    tr = pd.DataFrame({'hl': df['high'] - df['low'], 'hc': abs(df['high'] - df['close'].shift(1)), 'lc': abs(df['low'] - df['close'].shift(1))})
    tr['tr'] = tr.max(axis=1)
    
    # Calculate Average True Range (ATR) over the last 14 days
    atr = tr['tr'].rolling(window=14).mean()
    
    # Calculate the 5-day moving average of the close price
    ma_close = df['close'].rolling(window=5).mean()
    
    # Generate the new factor by dividing ATR by the Moving Average of the Close Price
    heuristics_matrix = atr / ma_close
    
    return heuristics_matrix
