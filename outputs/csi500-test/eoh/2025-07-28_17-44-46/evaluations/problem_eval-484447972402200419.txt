import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 250-day cumulative amount
    cum_amount_10 = df['amount'].rolling(window=10).sum()
    cum_amount_250 = df['amount'].rolling(window=250).sum()
    
    # Generate the new factor by dividing the 10-day cumulative amount by the 250-day cumulative amount
    heuristics_matrix = cum_amount_10 / cum_amount_250
    
    return heuristics_matrix
