import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 250-day average amount
    avg_amount_10 = df['amount'].rolling(window=10).mean()
    avg_amount_250 = df['amount'].rolling(window=250).mean()
    
    # Generate the new factor by dividing the 10-day average amount by the 250-day average amount
    heuristics_matrix = avg_amount_10 / avg_amount_250
    
    return heuristics_matrix
