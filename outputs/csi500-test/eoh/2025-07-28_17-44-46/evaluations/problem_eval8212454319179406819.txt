import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 90-day cumulative amount
    cum_amount_30 = df['amount'].rolling(window=30).sum()
    cum_amount_90 = df['amount'].rolling(window=90).sum()
    
    # Generate the new factor by dividing the 30-day cumulative amount by the 90-day cumulative amount
    heuristics_matrix = cum_amount_30 / cum_amount_90
    
    return heuristics_matrix
