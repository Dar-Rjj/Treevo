import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day simple moving average of the high price
    sma_high_50 = df['high'].rolling(window=50).mean()
    
    # Calculate the 100-day simple moving average of the low price
    sma_low_100 = df['low'].rolling(window=100).mean()
    
    # Generate the new factor by dividing the 50-day SMA of the high price by the 100-day SMA of the low price
    heuristics_matrix = sma_high_50 / sma_low_100
    
    return heuristics_matrix
