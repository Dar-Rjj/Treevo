import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day cumulative volume
    cum_vol_50 = df['volume'].rolling(window=50).sum()
    cum_vol_200 = df['volume'].rolling(window=200).sum()
    
    # Generate the new factor by dividing the 50-day cumulative volume by the 200-day cumulative volume
    factor_volume_ratio = cum_vol_50 / cum_vol_200
    
    # Calculate the 14-day Average True Range (ATR)
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift(1))
    low_close_prev = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine the volume ratio with the ATR
    heuristics_matrix = factor_volume_ratio * atr
    
    return heuristics_matrix
