import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 60-day Exponential Moving Average (EMA) of close price
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_close_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Generate the new factor by dividing the 20-day EMA close price by the 60-day EMA close price
    factor_price_ratio = ema_close_20 / ema_close_60
    
    # Calculate the 10-day Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_10 = tr.rolling(window=10).mean()
    
    # Combine the price ratio with the ATR
    heuristics_matrix = factor_price_ratio * atr_10
    
    return heuristics_matrix
