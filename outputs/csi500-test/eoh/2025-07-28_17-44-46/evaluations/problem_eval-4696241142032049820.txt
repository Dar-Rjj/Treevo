    df['momentum'] = df['return'].rolling(window=10).mean()
    
    # Volatility: Standard deviation of returns over the last 10 days

    df['liquidity'] = df['volume'] / df['amount']
    
    # Aggregate the heuristics
    heuristics_matrix = pd.concat([df['momentum'], df['volatility'], df['liquidity']], axis=1)
    
    return heuristics_matrix
