def heuristics_v2(df):
    # Calculate the 60-day and 180-day cumulative product of amount and volume
    cum_amt_vol_60 = (df['amount'] * df['volume']).rolling(window=60).sum()
    cum_amt_vol_180 = (df['amount'] * df['volume']).rolling(window=180).sum()
    
    # Generate the new factor by dividing the 60-day cumulative product by the 180-day cumulative product
    heuristics_matrix = cum_amt_vol_60 / cum_amt_vol_180
    
    return heuristics_matrix
