def heuristics_v2(df):
    # Calculate the 30-day moving average of the daily range (high - low)
    df['avg_range_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Find the maximum volume over the same 30-day window
    df['max_volume_30'] = df['volume'].rolling(window=30).max()
    
    # Generate the new factor by adjusting the average daily range with the ratio of today's volume to the maximum volume
    df['new_factor'] = df['avg_range_30'] * (df['volume'] / df['max_volume_30'])
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
