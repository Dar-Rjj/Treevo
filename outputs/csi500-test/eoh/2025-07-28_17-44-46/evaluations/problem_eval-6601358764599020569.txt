def heuristics_v2(df):
    # Calculate the 15-day and 75-day Exponential Moving Average (EMA) of volume
    ema_vol_15 = df['volume'].ewm(span=15, adjust=False).mean()
    ema_vol_75 = df['volume'].ewm(span=75, adjust=False).mean()
    
    # Generate the new factor by dividing the 15-day EMA volume by the 75-day EMA volume
    factor_volume_ratio = ema_vol_15 / ema_vol_75
    
    # Calculate the 20-day Standard Deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    
    # Combine the volume ratio with the standard deviation of returns
    heuristics_matrix = factor_volume_ratio * std_dev_20
    
    return heuristics_matrix
