import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day exponential moving average (EMA) of the volume
    ema_vol_30 = df['volume'].ewm(span=30).mean()
    # Calculate the 30-day simple moving average (SMA) of the close price
    sma_close_30 = df['close'].rolling(window=30).mean()
    
    # Generate the new factor by dividing the EMA of the volume by the SMA of the close price
    heuristics_matrix = ema_vol_30 / sma_close_30
    
    return heuristics_matrix
