import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 500-day exponential moving average of the volume
    ema_20 = df['volume'].ewm(span=20, adjust=False).mean()
    ema_500 = df['volume'].ewm(span=500, adjust=False).mean()
    
    # Generate the new factor by subtracting the 20-day EMA from the 500-day EMA
    heuristics_matrix = ema_500 - ema_20
    
    return heuristics_matrix
