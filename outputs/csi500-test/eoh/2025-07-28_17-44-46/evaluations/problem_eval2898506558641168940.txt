def heuristics_v2(df):
    # Calculate 5-day and 20-day exponential moving averages of the close price
    df['ema5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the momentum-based factor
    df['momentum_factor'] = (df['close'] / df[['ema5', 'ema20']].min(axis=1)) * (df['close'] / df['close'].shift(1))
    
    # Apply a 7-day Exponential Moving Average (EMA) for smoothing
    df['smoothed_momentum'] = df['momentum_factor'].ewm(span=7, adjust=False).mean()
    
    # Prepare the output
    heuristics_matrix = df['smoothed_momentum']
    
    return heuristics_matrix
