import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 100-day cumulative amount
    cum_amount_20 = df['amount'].rolling(window=20).sum()
    cum_amount_100 = df['amount'].rolling(window=100).sum()
    
    # Generate the new factor by dividing the 20-day cumulative amount by the 100-day cumulative amount
    factor_amount_ratio = cum_amount_20 / cum_amount_100
    
    # Calculate the 20-day Standard Deviation of the closing price
    std_close = df['close'].rolling(window=20).std()
    
    # Combine the amount ratio with the Standard Deviation of the closing price
    heuristics_matrix = factor_amount_ratio * std_close
    
    return heuristics_matrix
