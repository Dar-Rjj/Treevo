import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day cumulative amount
    cum_amt_50 = df['amount'].rolling(window=50).sum()
    cum_amt_200 = df['amount'].rolling(window=200).sum()
    
    # Calculate the 50-day and 200-day EMA of the cumulative amount
    ema_50 = cum_amt_50.ewm(span=50, adjust=False).mean()
    ema_200 = cum_amt_200.ewm(span=200, adjust=False).mean()
    
    # Generate the new factor by dividing the 50-day EMA by the 200-day EMA
    factor_amount_ratio = ema_50 / ema_200
    
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the amount ratio with the RSI
    heuristics_matrix = factor_amount_ratio * rsi
    
    return heuristics_matrix
