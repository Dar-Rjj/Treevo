import pandas as pd

def heuristics_v2(df):
    # Calculate the volume-weighted moving average of close prices over the last 10 days
    vwma_close = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day rolling standard deviation of high prices
    std_high = df['high'].rolling(window=10).std()
    
    # Generate the new factor by dividing the volume-weighted moving average of close prices by the standard deviation of high prices
    heuristics_matrix = vwma_close / std_high
    
    return heuristics_matrix
