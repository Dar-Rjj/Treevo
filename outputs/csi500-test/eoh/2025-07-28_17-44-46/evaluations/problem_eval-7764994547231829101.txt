def heuristics_v2(df):
    # Calculate EMAs
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the difference between the EMAs
    ema_diff = ema_short - ema_long

    # Calculate the 30-day mean absolute deviation of daily returns
    daily_returns = df['close'].pct_change()
    mad = daily_returns.rolling(window=30).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)

    # Generate heuristics matrix
    heuristics_matrix = ema_diff * mad

    return heuristics_matrix
