def heuristics_v2(df):
    # Calculate the 60-day and 180-day SMA of the closing price
    sma_close_60 = df['close'].rolling(window=60).mean()
    sma_close_180 = df['close'].rolling(window=180).mean()
    
    # Generate the new factor by dividing the 60-day SMA of the closing price by the 180-day SMA of the closing price
    heuristics_matrix = sma_close_60 / sma_close_180
    
    return heuristics_matrix
