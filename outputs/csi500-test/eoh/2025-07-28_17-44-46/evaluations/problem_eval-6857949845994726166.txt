import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Moving average of the daily return
    df['moving_avg'] = df['daily_return'].rolling(window=10).mean()
    
    # Generate the heuristic factor by taking the difference
    heuristics_matrix = df['daily_return'] - df['moving_avg']
    
    # Return only the series of heuristic factors, dropping NaNs
    return heuristics_matrix
