def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    # Calculate log difference of volume
    df['log_volume_diff'] = np.log(df['volume']).diff()
    # Use standard deviation of close price as weight
    df['std_close'] = df['close'].rolling(window=20).std()
    # Compute the weighted sum
    heuristics_matrix = (df['daily_return'] + df['log_volume_diff']) * df['std_close']
    return heuristics_matrix
