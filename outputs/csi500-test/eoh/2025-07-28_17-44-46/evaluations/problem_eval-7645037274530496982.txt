import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 500-day moving average of the amount
    ma_amount_20 = df['amount'].rolling(window=20).mean()
    ma_amount_500 = df['amount'].rolling(window=500).mean()
    
    # Generate the new factor by dividing the 20-day moving average by the 500-day moving average
    heuristics_matrix = ma_amount_20 / ma_amount_500
    
    return heuristics_matrix
