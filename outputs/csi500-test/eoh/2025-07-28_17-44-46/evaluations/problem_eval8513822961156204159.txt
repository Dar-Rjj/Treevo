def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the 60-day and 250-day standard deviations of daily returns
    std_60 = daily_returns.rolling(window=60).std()
    std_250 = daily_returns.rolling(window=250).std()
    
    # Generate the new factor by dividing the 60-day standard deviation by the 250-day standard deviation
    heuristics_matrix = std_60 / std_250
    
    return heuristics_matrix
