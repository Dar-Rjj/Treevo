def heuristics_v2(df):
    # Calculate the 5-day momentum of the closing price
    df['momentum_5'] = df['close'].pct_change(periods=5)
    
    # Calculate the 5-day average trading volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Compute the relative strength of trading volume
    df['volume_strength'] = df['volume'] / df['avg_volume_5']
    
    # Compute the composite factor
    df['composite_factor'] = df['momentum_5'] * df['volume_strength']
    
    # Prepare the output
    heuristics_matrix = df['composite_factor']
    
    return heuristics_matrix
