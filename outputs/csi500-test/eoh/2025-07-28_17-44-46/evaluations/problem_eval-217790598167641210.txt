def heuristics_v2(df):
    # Calculate the 30-day and 150-day moving average of the volume
    ma_vol_30 = df['volume'].rolling(window=30).mean()
    ma_vol_150 = df['volume'].rolling(window=150).mean()
    
    # Generate the new factor by dividing the 30-day moving average volume by the 150-day moving average volume
    heuristics_matrix = ma_vol_30 / ma_vol_150
    
    return heuristics_matrix
