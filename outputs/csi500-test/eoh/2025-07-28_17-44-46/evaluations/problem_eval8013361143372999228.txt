def heuristics_v2(df):
    # Calculate the cumulative return over the past 20 days
    df['cumulative_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate the 20-day average trading volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Compute the factor as the ratio of the cumulative return to the average volume
    df['strength_factor'] = df['cumulative_return'] / df['avg_volume_20']
    
    # Prepare the output
    heuristics_matrix = df['strength_factor']
    
    return heuristics_matrix
