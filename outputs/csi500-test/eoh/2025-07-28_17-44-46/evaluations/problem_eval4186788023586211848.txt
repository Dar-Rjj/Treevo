def heuristics_v2(df):
    # Calculate EMAs
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    ema_very_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the difference between the 10-day and 20-day EMAs
    ema_diff = ema_short - ema_long
    
    # Calculate the 30-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev = daily_returns.rolling(window=30).std()
    
    # Generate heuristics matrix
    heuristics_matrix = (ema_diff * std_dev) / ema_very_long
    
    return heuristics_matrix
