import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day moving average of the daily high-low range
    ma_high_low_50 = (df['high'] - df['low']).rolling(window=50).mean()
    
    # Calculate the 200-day moving average of the closing price
    ma_close_200 = df['close'].rolling(window=200).mean()
    
    # Generate the new factor by dividing the 50-day MA of high-low range by the 200-day MA of close price
    heuristics_matrix = ma_high_low_50 / ma_close_200
    
    return heuristics_matrix
