def heuristics_v2(df):
    # Calculate the 30-day and 150-day cumulative price change
    cum_price_change_30 = (df['close'] - df['open']).rolling(window=30).sum()
    cum_price_change_150 = (df['close'] - df['open']).rolling(window=150).sum()
    
    # Generate the new factor by dividing the 30-day cumulative price change by the 150-day cumulative price change
    heuristics_matrix = cum_price_change_30 / cum_price_change_150
    
    return heuristics_matrix
