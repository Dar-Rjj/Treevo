def heuristics_v2(df):
    # Calculate the 10-day rolling standard deviation of the close price as a measure of volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Compute the percentage change in daily trading volume
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Generate the new factor by multiplying the volatility with the absolute value of the volume change percentage
    df['new_factor'] = df['volatility'] * abs(df['volume_change_pct'])
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
