import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 90-day cumulative volume
    cum_volume_30 = df['volume'].rolling(window=30).sum()
    cum_volume_90 = df['volume'].rolling(window=90).sum()
    
    # Generate the volume ratio factor
    volume_ratio = cum_volume_30 / cum_volume_90
    
    # Calculate the 50-day and 200-day Simple Moving Average (SMA) of the closing price
    sma_close_50 = df['close'].rolling(window=50).mean()
    sma_close_200 = df['close'].rolling(window=200).mean()
    
    # Generate the SMA difference factor
    sma_diff = sma_close_50 - sma_close_200
    
    # Combine the volume ratio with the SMA difference
    heuristics_matrix = volume_ratio * sma_diff
    
    return heuristics_matrix
