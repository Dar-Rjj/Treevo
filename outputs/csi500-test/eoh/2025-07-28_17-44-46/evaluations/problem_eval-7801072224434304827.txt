import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day moving average of the volume
    ma_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 10-day moving average of the closing price
    ma_close = df['close'].rolling(window=10).mean()
    
    # Generate the new factor by dividing the 20-day moving average of the volume by the 10-day moving average of the closing price
    heuristics_matrix = ma_volume / ma_close
    
    return heuristics_matrix
