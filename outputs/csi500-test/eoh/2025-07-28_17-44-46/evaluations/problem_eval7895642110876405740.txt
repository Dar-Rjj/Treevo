import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Compute the exponentially weighted moving average of daily returns
    ewma_returns = daily_returns.ewm(span=20).mean()
    
    # Calculate the difference between the highest and lowest prices over a 20-day period
    price_range = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Combine the EWMA of returns with the price range
    heuristics_matrix = ewma_returns * price_range
    
    return heuristics_matrix
