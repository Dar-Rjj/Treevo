def heuristics_v2(df):
    # Calculate the 50-day and 200-day EMA of the adjusted closing price
    ema_adj_close_50 = df['close'].ewm(span=50).mean()
    ema_adj_close_200 = df['close'].ewm(span=200).mean()
    
    # Generate the new factor by subtracting the 200-day EMA from the 50-day EMA
    heuristics_matrix = ema_adj_close_50 - ema_adj_close_200
    
    return heuristics_matrix
