import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and close prices
    diff_high_close = df['high'] - df['close']
    
    # Calculate the 10-day exponential moving average of the difference
    ema_diff_high_close = diff_high_close.ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day simple moving average of the volume
    sma_volume = df['volume'].rolling(window=30).mean()
    
    # Generate the new factor by dividing the EMA of the difference by the SMA of the volume
    heuristics_matrix = ema_diff_high_close / sma_volume
    
    return heuristics_matrix
