import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day Simple Moving Average (SMA) of the closing price
    sma_close_50 = df['close'].rolling(window=50).mean()
    sma_close_200 = df['close'].rolling(window=200).mean()
    
    # Generate the new factor by dividing the 50-day SMA of the closing price by the 200-day SMA of the closing price
    heuristics_matrix = sma_close_50 / sma_close_200
    
    return heuristics_matrix
