def heuristics_v2(df):
    # Calculate 10-day and 60-day price momentum
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_60 = df['close'].pct_change(periods=60)

    # Calculate 50-day and 200-day moving averages of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Determine the weight by the ratio of 50-day MA to 200-day MA
    weight = ma_50 / ma_200

    # Combine the momentums with the calculated weight
    heuristics_matrix = (momentum_10 * weight) + (momentum_60 * (1 - weight))

    return heuristics_matrix
