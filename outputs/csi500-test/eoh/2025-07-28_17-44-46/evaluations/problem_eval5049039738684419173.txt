import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low price ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the 20-day simple moving average of the high-low ratio
    df['sma_high_low_ratio'] = df['high_low_ratio'].rolling(window=20).mean()
    
    # Generate the heuristic factor by taking the difference
    heuristics_matrix = df['high_low_ratio'] - df['sma_high_low_ratio']
    
    # Return only the series of heuristic factors, dropping NaNs
    return heuristics_matrix
