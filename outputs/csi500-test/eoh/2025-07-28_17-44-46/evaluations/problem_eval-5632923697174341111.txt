def heuristics_v2(df):
    # Calculate the 30-day and 150-day cumulative range (high - low)
    cum_range_30 = (df['high'] - df['low']).rolling(window=30).sum()
    cum_range_150 = (df['high'] - df['low']).rolling(window=150).sum()
    
    # Generate the new factor by dividing the 30-day cumulative range by the 150-day cumulative range
    heuristics_matrix = cum_range_30 / cum_range_150
    
    return heuristics_matrix
