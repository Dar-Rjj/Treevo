def heuristics_v2(df):
    high_low_diff = df['high'] - df['low']
    close_std_5d = df['close'].rolling(window=5).std()
    volume_sma_5d = df['volume'].rolling(window=5).mean()
    factor = (high_low_diff / close_std_5d) * volume_sma_5d
    heuristics_matrix = pd.Series(factor, index=df.index)
    return heuristics_matrix
