import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day cumulative volume
    cum_vol_50 = df['volume'].rolling(window=50).sum()
    cum_vol_200 = df['volume'].rolling(window=200).sum()
    
    # Calculate the 50-day average closing price
    avg_close_50 = df['close'].rolling(window=50).mean()
    
    # Generate the new factor by dividing the 50-day cumulative volume weighted by the 50-day average close by the 200-day cumulative volume
    heuristics_matrix = (cum_vol_50 * avg_close_50) / cum_vol_200
    
    return heuristics_matrix
