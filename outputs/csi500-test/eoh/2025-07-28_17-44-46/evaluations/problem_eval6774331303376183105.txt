def heuristics_v2(df):
    # Calculate EMAs
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the ratio of the EMAs
    ema_ratio = ema_short / ema_long
    
    # Calculate the 30-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev = daily_returns.rolling(window=30).std()
    
    # Generate heuristics matrix
    heuristics_matrix = ema_ratio * std_dev
    
    return heuristics_matrix
