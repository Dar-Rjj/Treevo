def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 50-day moving average of the daily return
    df['avg_daily_return_50'] = df['daily_return'].rolling(window=50).mean()
    
    # Calculate the 50-day sum of the absolute daily returns
    df['sum_abs_daily_return_50'] = df['daily_return'].abs().rolling(window=50).sum()
    
    # Generate the new factor by dividing the 50-day average daily return by the 50-day sum of the absolute daily returns
    df['new_factor'] = df['avg_daily_return_50'] / df['sum_abs_daily_return_50']
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
