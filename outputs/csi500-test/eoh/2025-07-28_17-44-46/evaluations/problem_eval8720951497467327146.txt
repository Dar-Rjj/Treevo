import pandas as pd
import numpy as np

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    ema_diff = close_ema - volume_ema
    heuristics_matrix = np.log(ema_diff + 1)
    return heuristics_matrix
