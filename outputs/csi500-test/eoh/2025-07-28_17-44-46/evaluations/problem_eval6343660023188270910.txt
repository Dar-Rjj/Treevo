import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 100-day cumulative dollar volume
    cum_dollar_vol_20 = (df['close'] * df['volume']).rolling(window=20).sum()
    cum_dollar_vol_100 = (df['close'] * df['volume']).rolling(window=100).sum()
    
    # Generate the new factor by dividing the 20-day cumulative dollar volume by the 100-day cumulative dollar volume
    heuristics_matrix = cum_dollar_vol_20 / cum_dollar_vol_100
    
    return heuristics_matrix
