def heuristics_v2(df):
    # Calculate the 20-day rolling standard deviation of the close price as a measure of volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Compute the percentage change in daily trading volume
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Square the percentage change in volume to amplify the effect of large changes
    df['squared_volume_change_pct'] = df['volume_change_pct'] ** 2
    
    # Generate the new factor by multiplying the volatility with the squared volume change percentage
    df['new_factor'] = df['volatility'] * df['squared_volume_change_pct']
    
    # Prepare the output
    heuristics_matrix = df['new_factor']
    
    return heuristics_matrix
