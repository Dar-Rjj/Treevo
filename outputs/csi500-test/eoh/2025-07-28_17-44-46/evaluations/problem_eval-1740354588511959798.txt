import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 60-day cumulative high and low prices
    cum_high_10 = df['high'].rolling(window=10).sum()
    cum_low_60 = df['low'].rolling(window=60).sum()
    
    # Generate the new factor by dividing the 10-day cumulative high price by the 60-day cumulative low price
    heuristics_matrix = cum_high_10 / cum_low_60
    
    return heuristics_matrix
