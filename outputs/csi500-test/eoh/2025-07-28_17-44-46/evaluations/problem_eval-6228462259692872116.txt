def heuristics_v2(df):
    # Calculate the 30-day and 120-day cumulative volume
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    cum_amt_120 = df['amount'].rolling(window=120).sum()
    
    # Generate the new factor by dividing the 30-day cumulative volume by the 120-day cumulative amount
    heuristics_matrix = cum_vol_30 / cum_amt_120
    
    return heuristics_matrix
