def heuristics_v2(df):
    # Calculate the 20-day and 100-day EMA of volume
    ema_vol_20 = df['volume'].ewm(span=20).mean()
    ema_vol_100 = df['volume'].ewm(span=100).mean()
    
    # Generate the new factor by dividing the 20-day EMA volume by the 100-day EMA volume
    heuristics_matrix = ema_vol_20 / ema_vol_100
    
    return heuristics_matrix
