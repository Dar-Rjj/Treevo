import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 200-day simple moving averages
    sma_30 = df['close'].rolling(window=30).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Generate the new factor by subtracting the 200-day SMA from the 30-day SMA
    heuristics_matrix = sma_30 - sma_200

    return heuristics_matrix
