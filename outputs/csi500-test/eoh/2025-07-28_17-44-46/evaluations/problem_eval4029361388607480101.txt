import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'].diff()
    
    # Calculate the 20-day EMA of the daily price change
    ema_price_change = price_change.ewm(span=20).mean()
    
    # Calculate the 20-day standard deviation of volume
    std_volume = df['volume'].rolling(window=20).std()
    
    # Generate the new factor by combining the EMA of the price change and the standard deviation of volume
    heuristics_matrix = ema_price_change * std_volume
    
    return heuristics_matrix
