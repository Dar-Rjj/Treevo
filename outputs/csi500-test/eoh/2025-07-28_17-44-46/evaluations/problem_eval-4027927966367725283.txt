import pandas as pd

def heuristics_v2(df):
    price_momentum = df['close'].pct_change(periods=5)
    volume_momentum = df['volume'].pct_change(periods=5)
    avg_close = df['close'].rolling(window=10).mean()
    factor = (price_momentum - volume_momentum) * (df['close'] / avg_close)
    heuristics_matrix = pd.Series(factor, index=df.index)
    return heuristics_matrix
