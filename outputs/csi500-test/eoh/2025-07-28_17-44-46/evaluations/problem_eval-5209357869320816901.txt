import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day and 90-day cumulative volume
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    cum_vol_90 = df['volume'].rolling(window=90).sum()
    
    # Generate the new factor by dividing the 30-day cumulative volume by the 90-day cumulative volume
    heuristics_matrix = cum_vol_30 / cum_vol_90
    
    return heuristics_matrix
