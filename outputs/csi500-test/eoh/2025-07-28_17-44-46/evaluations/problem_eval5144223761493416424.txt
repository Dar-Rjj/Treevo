def heuristics_v2(df):
    # Calculate the 30-day and 120-day cumulative sum of the product of 'amount' and 'volume'
    cum_amt_vol_30 = (df['amount'] * df['volume']).rolling(window=30).sum()
    cum_amt_vol_120 = (df['amount'] * df['volume']).rolling(window=120).sum()
    
    # Generate the new factor by dividing the 30-day cumulative sum by the 120-day cumulative sum
    heuristics_matrix = cum_amt_vol_30 / cum_amt_vol_120
    
    return heuristics_matrix
