def heuristics_v2(df):
    # Calculate the difference between high and low
    hl_diff = df['high'] - df['low']
    # Calculate the ratio of close to volume
    close_volume_ratio = df['close'] / df['volume']
    # Combine the two for a new heuristic
    heuristics_matrix = (hl_diff * close_volume_ratio).rename('heuristic')
    return heuristics_matrix
