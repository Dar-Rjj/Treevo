import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day cumulative volume
    cum_vol_50 = df['volume'].rolling(window=50).sum()
    cum_vol_200 = df['volume'].rolling(window=200).sum()
    
    # Apply exponential moving average to smooth the cumulative volumes
    ema_cum_vol_50 = cum_vol_50.ewm(span=50, adjust=False).mean()
    ema_cum_vol_200 = cum_vol_200.ewm(span=200, adjust=False).mean()
    
    # Generate the new factor by dividing the smoothed 50-day cumulative volume by the smoothed 200-day cumulative volume
    heuristics_matrix = ema_cum_vol_50 / ema_cum_vol_200
    
    return heuristics_matrix
