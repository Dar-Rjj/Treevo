import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day momentum of the close price
    momentum_10 = df['close'] - df['close'].shift(10)
    
    # Calculate the 30-day standard deviation of the close price
    std_30 = df['close'].rolling(window=30).std()
    
    # Generate the new factor by dividing the 10-day momentum by the 30-day standard deviation
    heuristics_matrix = momentum_10 / std_30
    
    return heuristics_matrix
