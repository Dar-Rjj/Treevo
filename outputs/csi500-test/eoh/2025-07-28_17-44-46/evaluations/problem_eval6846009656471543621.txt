import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 30-day cumulative volume
    cum_vol_10 = df['volume'].rolling(window=10).sum()
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    
    # Generate the volume ratio
    vol_ratio = cum_vol_10 / cum_vol_30
    
    # Calculate the 20-day Average True Range (ATR)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=20).mean()
    
    # Generate the heuristics matrix
    heuristics_matrix = vol_ratio * atr
    
    return heuristics_matrix
