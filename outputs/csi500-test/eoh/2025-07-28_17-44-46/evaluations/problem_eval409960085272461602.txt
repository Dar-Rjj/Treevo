def heuristics_v2(df):
    # Calculate 20-day and 60-day moving averages of the close price
    df['ma20'] = df['close'].rolling(window=20).mean()
    df['ma60'] = df['close'].rolling(window=60).mean()
    
    # Compute the momentum-based factor
    df['momentum_factor'] = (df['close'] / df[['ma20', 'ma60']].min(axis=1)) * (df['volume'] / df['volume'].shift(1))
    
    # Apply exponential smoothing to the momentum-based factor
    df['smoothed_momentum'] = df['momentum_factor'].ewm(span=5, adjust=False).mean()
    
    # Prepare the output
    heuristics_matrix = df['smoothed_momentum']
    
    return heuristics_matrix
