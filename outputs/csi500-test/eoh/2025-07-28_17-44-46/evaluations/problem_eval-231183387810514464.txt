import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 20-day EMA of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Generate the new factor by dividing the 5-day EMA by the 20-day EMA
    heuristics_matrix = ema_5 / ema_20
    
    return heuristics_matrix
