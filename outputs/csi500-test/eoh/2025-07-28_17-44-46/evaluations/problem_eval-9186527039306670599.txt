def heuristics_v2(df):
    # Calculate the 10-day and 30-day SMA of the closing price
    sma_close_10 = df['close'].rolling(window=10).mean()
    sma_close_30 = df['close'].rolling(window=30).mean()
    
    # Generate the new factor by subtracting the 30-day SMA from the 10-day SMA
    heuristics_matrix = sma_close_10 - sma_close_30
    
    return heuristics_matrix
