def heuristics_v2(df):
    # Calculate 10-day and 20-day simple moving averages of the close price
    df['sma10'] = df['close'].rolling(window=10).mean()
    df['sma20'] = df['close'].rolling(window=20).mean()
    
    # Compute the relative strength factor
    df['relative_strength_factor'] = (df['close'] / df[['sma10', 'sma20']].min(axis=1)) * (df['amount'] / df['volume'])
    
    # Apply a weighted moving average for smoothing
    weights = np.arange(1, 6)
    df['smoothed_relative_strength'] = df['relative_strength_factor'].rolling(window=5).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Prepare the output
    heuristics_matrix = df['smoothed_relative_strength']
    
    return heuristics_matrix
