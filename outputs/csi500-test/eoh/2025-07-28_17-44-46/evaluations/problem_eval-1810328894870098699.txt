def heuristics_v2(df):
    def calculate_momentum(df, period):
        return df['close'].pct_change(period)
    
    def calculate_volume_trend(df, period):
        return df['volume'].rolling(window=period).mean().pct_change()
    
    short_lookback = 10
    medium_lookback = 30
    long_lookback = 90
    
    short_momentum = calculate_momentum(df, short_lookback)
    medium_momentum = calculate_momentum(df, medium_lookback)
    long_momentum = calculate_momentum(df, long_lookback)
    
    short_volume_trend = calculate_volume_trend(df, short_lookback)
    medium_volume_trend = calculate_volume_trend(df, medium_lookback)
    long_volume_trend = calculate_volume_trend(df, long_lookback)
    
    heuristics_matrix = (short_momentum + medium_momentum + long_momentum + 
                         short_volume_trend + medium_volume_trend + long_volume_trend) / 6

    return heuristics_matrix
