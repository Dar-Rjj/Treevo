import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day Simple Moving Average (SMA) of the closing price
    sma_close_50 = df['close'].rolling(window=50).mean()
    sma_close_200 = df['close'].rolling(window=200).mean()
    
    # Generate the difference between the 50-day SMA and 200-day SMA
    diff_sma = sma_close_50 - sma_close_200
    
    # Calculate the 10-day and 50-day cumulative volume
    cum_volume_10 = df['volume'].rolling(window=10).sum()
    cum_volume_50 = df['volume'].rolling(window=50).sum()
    
    # Generate the ratio of the 10-day cumulative volume to the 50-day cumulative volume
    volume_ratio = cum_volume_10 / cum_volume_50
    
    # Combine the difference in SMA with the volume ratio
    heuristics_matrix = diff_sma * volume_ratio
    
    return heuristics_matrix
