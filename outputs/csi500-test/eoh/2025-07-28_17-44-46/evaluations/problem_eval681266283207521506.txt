def heuristics_v2(df):
    # Calculate the 10-day standard deviation of the closing prices to capture volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate the 20-day cumulative sum of the difference between closing and opening prices, to gauge the directional trend
    df['directional_movement'] = (df['close'] - df['open']).rolling(window=20).sum()
    
    # Adjust the calculated volatility by the relative change in trading volume
    df['adjusted_volatility'] = df['volatility'] * (df['volume'] / df['volume'].shift(5))
    
    # Integrate the adjusted volatility with the directional movement
    heuristics_matrix = df['adjusted_volatility'] + df['directional_movement']
    
    return heuristics_matrix
