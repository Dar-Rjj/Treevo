def heuristics_v2(df):
    # Calculate 5-day momentum
    momentum_5 = df['close'].pct_change(periods=5)
    
    # Calculate the ratio of current volume to the 10-day average volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10
    
    # Calculate the 20-day volatility of the closing price
    daily_returns = df['close'].pct_change()
    volatility_20 = daily_returns.rolling(window=20).std()
    
    # Generate heuristics matrix
    heuristics_matrix = (momentum_5 * volume_ratio) / volatility_20
    
    return heuristics_matrix
