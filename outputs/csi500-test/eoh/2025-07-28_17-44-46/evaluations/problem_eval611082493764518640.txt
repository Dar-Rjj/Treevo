def heuristics_v2(df):
    # Calculate the 10-day and 200-day cumulative high and low prices
    cum_high_10 = df['high'].rolling(window=10).sum()
    cum_low_200 = df['low'].rolling(window=200).sum()
    
    # Generate the new factor by dividing the 10-day cumulative high by the 200-day cumulative low
    heuristics_matrix = cum_high_10 / cum_low_200
    
    return heuristics_matrix
