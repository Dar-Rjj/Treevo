import pandas as pd

def heuristics_v2(df):
    ema_close = df['close'].ewm(span=5).mean()
    ema_volume = df['volume'].ewm(span=5).mean()
    avg_amount = df['amount'].rolling(window=5).mean()
    factor = (ema_close - ema_volume) * (df['amount'] / avg_amount)
    heuristics_matrix = pd.Series(factor, index=df.index)
    return heuristics_matrix
