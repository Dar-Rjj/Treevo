import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day EMA of volume
    ema_vol_50 = df['volume'].ewm(span=50).mean()
    ema_vol_200 = df['volume'].ewm(span=200).mean()
    
    # Generate the new factor by dividing the 50-day EMA volume by the 200-day EMA volume
    heuristics_matrix = ema_vol_50 / ema_vol_200
    
    return heuristics_matrix
