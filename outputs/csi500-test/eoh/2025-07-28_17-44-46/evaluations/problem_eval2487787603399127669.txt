import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of amount to volume
    amount_volume_ratio = df['amount'] / df['volume']
    # Calculate the 5-day Exponential Moving Average of the ratio
    ema_amount_volume_ratio = amount_volume_ratio.ewm(span=5, adjust=False).mean()
    # Calculate the 5-day Simple Moving Average of the closing price
    sma_close = df['close'].rolling(window=5).mean()
    # Combine the two for a new heuristic
    heuristics_matrix = (ema_amount_volume_ratio / sma_close).rename('heuristic')
    return heuristics_matrix
