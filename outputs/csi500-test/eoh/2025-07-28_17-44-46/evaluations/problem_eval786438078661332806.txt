def heuristics_v2(df):
    # Calculate the 50-day and 200-day simple moving averages of the closing price
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the difference between the 50-day and 200-day SMAs
    sma_diff = sma_50 - sma_200
    
    # Calculate the 60-day standard deviation of the daily log returns
    daily_log_returns = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    std_dev_log = daily_log_returns.rolling(window=60).std()
    
    # Generate the heuristics matrix
    heuristics_matrix = sma_diff / std_dev_log
    
    return heuristics_matrix
