def heuristics_v2(df):
    # Calculate the 60-day and 240-day cumulative amount
    cum_amt_60 = df['amount'].rolling(window=60).sum()
    cum_amt_240 = df['amount'].rolling(window=240).sum()
    
    # Generate the new factor by dividing the 60-day cumulative amount by the 240-day cumulative amount
    heuristics_matrix = cum_amt_60 / cum_amt_240
    
    return heuristics_matrix
