def heuristics_v2(df):
    # Calculate 10-day and 30-day moving averages of the close price
    df['ma10'] = df['close'].rolling(window=10).mean()
    df['ma30'] = df['close'].rolling(window=30).mean()
    
    # Compute the momentum-based factor
    df['momentum_factor'] = (df['close'] / df[['ma10', 'ma30']].min(axis=1)) * (df['volume'] / df['volume'].shift(1))
    
    # Apply a simple smoothing (SMA) to the momentum-based factor
    df['smoothed_momentum'] = df['momentum_factor'].rolling(window=5).mean()
    
    # Prepare the output
    heuristics_matrix = df['smoothed_momentum']
    
    return heuristics_matrix
