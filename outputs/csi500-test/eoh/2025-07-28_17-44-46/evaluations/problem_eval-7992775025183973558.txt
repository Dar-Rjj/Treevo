import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day and 200-day standard deviation of closing prices
    std_close_50 = df['close'].rolling(window=50).std()
    std_close_200 = df['close'].rolling(window=200).std()
    
    # Generate the new factor by dividing the 50-day standard deviation of closing prices by the 200-day standard deviation of closing prices
    heuristics_matrix = std_close_50 / std_close_200
    
    return heuristics_matrix
