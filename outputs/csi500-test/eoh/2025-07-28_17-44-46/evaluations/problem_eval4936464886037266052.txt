import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low price ratio
    hl_ratio = df['high'] / df['low']
    
    # Calculate the change in trading volume
    volume_change = df['volume'] / df['volume'].shift(1)
    
    # Generate the new factor by combining the high-low price ratio and the change in trading volume
    heuristics_matrix = (hl_ratio + volume_change) / 2
    
    return heuristics_matrix
