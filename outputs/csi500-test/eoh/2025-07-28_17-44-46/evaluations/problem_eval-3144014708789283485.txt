def heuristics_v2(df):
    # Calculate the 30-day and 600-day cumulative amount
    cum_amount_30 = df['amount'].rolling(window=30).sum()
    cum_amount_600 = df['amount'].rolling(window=600).sum()
    
    # Calculate the 30-day moving average price
    ma_price_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the percentage change in the 30-day moving average price
    pct_change_ma_price_30 = ma_price_30.pct_change().fillna(0)
    
    # Generate the new factor by dividing the (30-day cumulative amount * (1 + percentage change in 30-day MA price)) by the 600-day cumulative amount
    heuristics_matrix = (cum_amount_30 * (1 + pct_change_ma_price_30)) / cum_amount_600
    
    return heuristics_matrix
