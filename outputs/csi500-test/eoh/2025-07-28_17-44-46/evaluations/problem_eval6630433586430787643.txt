import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 120-day cumulative volume
    cum_vol_20 = df['volume'].rolling(window=20).sum()
    cum_vol_120 = df['volume'].rolling(window=120).sum()
    
    # Generate the new factor by dividing the 20-day cumulative volume by the 120-day cumulative volume
    factor_volume_ratio = cum_vol_20 / cum_vol_120
    
    # Calculate the 5-day Exponentially Weighted Moving Average (EWMA) of the True Range
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift(1))
    low_close_prev = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    ewma_tr = tr.ewm(span=5).mean()
    
    # Combine the volume ratio with the EWMA of the True Range
    heuristics_matrix = factor_volume_ratio * ewma_tr
    
    return heuristics_matrix
