import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 60-day Exponential Moving Average (EMA) of the close price
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_close_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Generate the new factor by dividing the 20-day EMA close by the 60-day EMA close
    factor_price_ratio = ema_close_20 / ema_close_60
    
    # Calculate the 10-day Mean Absolute Deviation (MAD) of daily returns
    daily_returns = df['close'].pct_change()
    mad_10 = daily_returns.rolling(window=10).apply(lambda x: (x - x.mean()).abs().mean(), raw=True)
    
    # Combine the price ratio with the MAD of returns
    heuristics_matrix = factor_price_ratio * mad_10
    
    return heuristics_matrix
