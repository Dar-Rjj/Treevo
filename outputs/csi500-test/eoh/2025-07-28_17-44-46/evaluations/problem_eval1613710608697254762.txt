def heuristics_v2(df):
    # Calculate the 60-day moving average of the high and low prices
    ma_high_60 = df['high'].rolling(window=60).mean()
    ma_low_60 = df['low'].rolling(window=60).mean()

    # Generate the new factor by subtracting the 60-day moving average of the low price from the 60-day moving average of the high price
    heuristics_matrix = ma_high_60 - ma_low_60

    return heuristics_matrix
