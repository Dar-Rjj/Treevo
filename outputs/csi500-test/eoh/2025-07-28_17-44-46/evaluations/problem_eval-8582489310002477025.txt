import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change of the closing price
    roc = df['close'].pct_change(periods=5)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 10-day standard deviation of the daily return
    std_daily_return = daily_return.rolling(window=10).std()
    
    # Generate the new factor by dividing the 5-day ROC by the 10-day standard deviation of the daily return
    heuristics_matrix = roc / std_daily_return
    
    return heuristics_matrix
