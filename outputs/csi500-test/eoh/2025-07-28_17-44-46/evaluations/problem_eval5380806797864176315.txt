def heuristics_v2(df):
    # Calculate the 30-day and 120-day moving average volume
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    avg_vol_120 = df['volume'].rolling(window=120).mean()
    
    # Generate the new factor by dividing the 30-day moving average volume by the 120-day moving average volume
    heuristics_matrix = avg_vol_30 / avg_vol_120
    
    return heuristics_matrix
