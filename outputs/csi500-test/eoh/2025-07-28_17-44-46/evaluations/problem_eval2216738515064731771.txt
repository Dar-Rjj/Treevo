def heuristics_v2(df):
    # Calculate the 30-day and 120-day cumulative volume
    cum_vol_30 = df['volume'].rolling(window=30).sum()
    cum_vol_120 = df['volume'].rolling(window=120).sum()
    
    # Generate the new factor by subtracting the 120-day cumulative volume from the 30-day cumulative volume
    heuristics_matrix = cum_vol_30 - cum_vol_120
    
    return heuristics_matrix
