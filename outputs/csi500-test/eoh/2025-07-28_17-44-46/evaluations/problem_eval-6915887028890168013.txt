def heuristics_v2(df):
    close_std = df['close'].rolling(window=5).std()
    volume_std = df['volume'].rolling(window=5).std()
    daily_return = df['close'].pct_change()
    heuristics_matrix = (close_std / volume_std) * daily_return
    heuristics_matrix = heuristics_matrix.dropna()
    return heuristics_matrix
