import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    tr = pd.DataFrame({'hl': df['high'] - df['low'], 'hc': abs(df['high'] - df['close'].shift(1)), 'lc': abs(df['low'] - df['close'].shift(1))})
    tr['tr'] = tr.max(axis=1)
    
    # Calculate Exponential Moving Average of True Range (EMATR) over the last 14 days
    ematr = tr['tr'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 5-day moving average of the volume
    ma_volume = df['volume'].rolling(window=5).mean()
    
    # Generate the new factor by dividing EMATR by the Moving Average of the Volume
    heuristics_matrix = ematr / ma_volume
    
    return heuristics_matrix
