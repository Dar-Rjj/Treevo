import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the cube of the open-to-close price ratio
    cube_open_close_ratio = (df['open'] / df['close']) ** 3
    
    # Calculate the natural logarithm of the sum of high and low prices
    log_high_low_sum = np.log(df['high'] + df['low'])
    
    # Calculate the square of the volume over trading amount
    sq_volume_amount = (df['volume'] / df['amount']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.25, 0.15
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cube_open_close_ratio + w2 * log_high_low_sum + w3 * sq_volume_amount
    
    return heuristics_matrix
