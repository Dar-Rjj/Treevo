import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the square root of the absolute difference between the high and low prices
    sqrt_high_low_diff = np.sqrt(np.abs(df['high'] - df['low']))
    
    # Calculate twice the trading volume-to-amount ratio
    two_volume_amount_ratio = 2 * (df['volume'] / df['amount'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * sqrt_high_low_diff + w3 * two_volume_amount_ratio
    
    return heuristics_matrix
