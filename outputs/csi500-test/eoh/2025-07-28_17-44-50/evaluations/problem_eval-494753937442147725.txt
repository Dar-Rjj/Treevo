import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 5-day EMA of closing prices
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the standard deviation of daily returns over a 10-day period
    daily_returns = df['close'].pct_change()
    std_daily_returns = daily_returns.rolling(window=10).std()
    
    # Calculate the ratio of the current day's trading volume to the average volume over the past 20 days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_close + w2 * std_daily_returns + w3 * volume_ratio
    
    return heuristics_matrix
