import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the open-to-close price ratio
    sqrt_open_close_ratio = np.sqrt(df['open'] / df['close'])
    
    # Calculate the cube of the absolute difference between the natural logarithms of the closing and opening prices
    cube_abs_log_close_open_diff = (np.abs(np.log(df['close']) - np.log(df['open']))) ** 3
    
    # Calculate the arctangent of the trading volume over amount
    atan_volume_amount = np.arctan(df['volume'] / df['amount'])
    
    # Define weights for each component
    w1, w2, w3 = 0.3, 0.5, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_open_close_ratio + w2 * cube_abs_log_close_open_diff + w3 * atan_volume_amount
    
    return heuristics_matrix
