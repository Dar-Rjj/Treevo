import pandas as pd

def heuristics_v2(df):
    sma_10_close = df['close'].rolling(window=10).mean()
    sma_50_volume = df['volume'].rolling(window=50).mean()
    ratio_sma = sma_10_close / sma_50_volume
    high_low_diff_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    heuristics_matrix = ratio_sma + high_low_diff_20
    return heuristics_matrix
