import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the sine of the difference between high and low prices
    sin_high_low_diff = np.sin(df['high'] - df['low'])
    
    # Calculate the cube of the trading amount divided by the volume
    cube_amount_volume = (df['amount'] / df['volume']) ** 3
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * sin_high_low_diff + w3 * cube_amount_volume
    
    return heuristics_matrix
