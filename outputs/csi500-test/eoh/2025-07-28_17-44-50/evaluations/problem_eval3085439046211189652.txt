import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price difference
    exp_close_open_diff = np.exp(df['close'] - df['open'])
    
    # Calculate the arctangent of the high-to-low price ratio
    atan_high_low_ratio = np.arctan(df['high'] / df['low'])
    
    # Calculate the fourth root of the trading volume
    fourth_rt_volume = np.power(df['volume'], 1/4)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_diff + w2 * atan_high_low_ratio + w3 * fourth_rt_volume
    
    return heuristics_matrix
