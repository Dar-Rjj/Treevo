import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the cube of the absolute difference between the high and low prices
    cube_high_low_diff = np.cbrt(np.abs(df['high'] - df['low']))
    
    # Calculate the trading volume-to-amount ratio
    volume_amount_ratio = df['volume'] / df['amount']
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * cube_high_low_diff + w3 * volume_amount_ratio
    
    return heuristics_matrix
