import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Apply arctangent to the close-to-open price ratio
    atan_close_open_ratio = np.arctan(df['close'] / df['open'])
    
    # Raise the absolute difference between high and low prices to the power of 1.5
    pow_high_low_diff = np.abs(df['high'] - df['low']) ** 1.5
    
    # Calculate the natural logarithm of the trading amount-to-volume ratio
    log_amount_volume = np.log(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * atan_close_open_ratio + w2 * pow_high_low_diff + w3 * log_amount_volume
    
    return heuristics_matrix
