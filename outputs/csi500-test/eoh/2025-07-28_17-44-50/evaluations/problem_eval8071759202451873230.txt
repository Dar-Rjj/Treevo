def heuristics_v2(df):
    # Calculate the 14-day Rate of Change (ROC) for the close price
    roc_14d = df['close'].pct_change(14)
    
    # Calculate the 30-day standard deviation of daily returns as a measure of volatility
    daily_returns = df['close'].pct_change()
    vol_30d = daily_returns.rolling(window=30).std()
    
    # Calculate the 5-day and 20-day moving averages of the volume
    ma_volume_5d = df['volume'].rolling(window=5).mean()
    ma_volume_20d = df['volume'].rolling(window=20).mean()
    
    # Calculate the ratio of the 5-day MA volume to the 20-day MA volume
    vol_ratio = ma_volume_5d / ma_volume_20d
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * roc_14d + w2 * vol_30d + w3 * vol_ratio
    
    return heuristics_matrix
