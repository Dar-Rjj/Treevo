import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the logarithmic return
    exp_log_return = np.exp(np.log(df['close'] / df['open']))
    
    # Calculate the cube root of the ratio between the highest and lowest prices
    cbrt_high_low_ratio = np.cbrt(df['high'] / df['low'])
    
    # Calculate the arithmetic mean of the trading amount and volume
    mean_amount_volume = (df['amount'] + df['volume']) / 2
    
    # Define weights for each component
    w1, w2, w3 = 0.3, 0.4, 0.3
    
    # Combine the components into a single alpha factor using a weighted harmonic mean
    heuristics_matrix = 1 / ((w1 / exp_log_return) + (w2 / cbrt_high_low_ratio) + (w3 / mean_amount_volume))
    
    return heuristics_matrix
