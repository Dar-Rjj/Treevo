def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the high-low price spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define new weights for each component
    w1, w2, w3 = 0.5, 0.2, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * close_open_ratio + w2 * high_low_spread + w3 * sqrt_amount_volume
    
    return heuristics_matrix
