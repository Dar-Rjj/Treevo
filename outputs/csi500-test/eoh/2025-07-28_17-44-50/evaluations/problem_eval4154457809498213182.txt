import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the natural logarithm of the high-low spread
    log_high_low_spread = np.log(df['high'] - df['low'])
    
    # Calculate the trading amount to volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor using the geometric mean formula
    heuristics_matrix = (sqrt_close_open_ratio ** w1) * (np.exp(log_high_low_spread) ** w2) * (amount_volume_ratio ** w3)
    
    return heuristics_matrix
