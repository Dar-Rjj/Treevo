import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the ratio between close and open prices
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the cubic root of the absolute difference between high and low prices
    cbrt_high_low_diff = np.cbrt(np.abs(df['high'] - df['low']))
    
    # Calculate the exponential of the trading amount divided by the square root of volume
    exp_amount_sqrt_volume = np.exp(df['amount'] / np.sqrt(df['volume']))
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * cbrt_high_low_diff + w3 * exp_amount_sqrt_volume
    
    return heuristics_matrix
