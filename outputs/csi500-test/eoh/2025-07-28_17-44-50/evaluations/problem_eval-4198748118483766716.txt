def heuristics_v2(df):
    # Calculate the average of open, high, and low prices
    price_avg = df[['open', 'high', 'low']].mean(axis=1)
    
    # Calculate the difference between the close price and the average of open, high, low prices
    price_diff = df['close'] - price_avg
    
    # Calculate the 20-day moving average of the volume
    vol_20ma = df['volume'].rolling(window=20).mean()
    
    # Calculate the percentage change in the volume relative to its 20-day moving average
    vol_change_to_ma = df['volume'] / vol_20ma
    
    # Composite heuristic score: weighted sum of the price difference and the volume change to ma
    heuristics_matrix = 0.5 * price_diff + 0.5 * vol_change_to_ma
    
    return heuristics_matrix
