import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the high-to-low price ratio
    square_high_low_ratio = (df['high'] / df['low']) ** 2
    
    # Calculate the absolute value of the natural logarithm difference between the close and open prices
    abs_log_close_open_diff = np.abs(np.log(df['close']) - np.log(df['open']))
    
    # Calculate the square root of the trading volume
    sqrt_volume = np.sqrt(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_high_low_ratio + w2 * abs_log_close_open_diff + w3 * sqrt_volume
    
    return heuristics_matrix
