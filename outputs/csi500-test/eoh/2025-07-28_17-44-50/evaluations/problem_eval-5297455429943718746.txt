import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the close-to-high price ratio
    log_close_high_ratio = np.log(df['close'] / df['high'])
    
    # Calculate the square of the difference between the open and low prices
    square_open_low_diff = (df['open'] - df['low']) ** 2
    
    # Calculate the natural logarithm of the trading volume
    ln_volume = np.log(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.33, 0.33, 0.34
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_high_ratio + w2 * square_open_low_diff + w3 * ln_volume
    
    return heuristics_matrix
