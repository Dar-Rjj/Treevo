def heuristics_v2(df):
    # Calculate 20-day Simple Moving Average
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Distance between closing price and 20-day SMA
    dist_to_sma = df['close'] - sma_20
    
    # Positive volume days over the last 20 days
    pos_vol_days = df['volume'].gt(0).rolling(window=20).sum()
    
    # Total volume days over the last 20 days
    total_vol_days = df['volume'].notnull().rolling(window=20).sum()
    
    # Ratio of positive volume days to total volume days
    vol_ratio = pos_vol_days / total_vol_days
    
    # Combine the calculated features into a single factor
    heuristics_matrix = dist_to_sma * vol_ratio
    
    return heuristics_matrix
