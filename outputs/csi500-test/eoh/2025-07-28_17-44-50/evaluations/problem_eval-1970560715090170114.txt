import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between the close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the ratio of the high to low prices
    ratio_high_low = df['high'] / df['low']
    
    # Calculate the square root of the logarithm of the trading volume
    sqrt_log_volume = np.sqrt(np.log(df['volume']))
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * diff_close_open + w2 * ratio_high_low + w3 * sqrt_log_volume
    
    return heuristics_matrix
