import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the exponential moving average (EMA) of the daily return
    ema_daily_return = daily_return.ewm(span=10, adjust=False).mean()
    
    # Calculate the standard deviation of the last 10 days' closing prices
    std_close_prices = df['close'].rolling(window=10).std()
    
    # Calculate the ratio of today's volume to the average volume over the past 30 days
    avg_volume_30d = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30d
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_daily_return + w2 * std_close_prices + w3 * volume_ratio
    
    return heuristics_matrix
