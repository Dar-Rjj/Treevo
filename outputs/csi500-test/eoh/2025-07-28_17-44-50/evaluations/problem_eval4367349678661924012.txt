import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of closing price to the 50-day EMA of the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    close_ema_50_ratio = df['close'] / ema_50
    
    # Calculate the coefficient of variation of daily returns over the last 20 days
    daily_returns = df['close'].pct_change()
    cv_daily_returns = daily_returns.rolling(window=20).std() / daily_returns.rolling(window=20).mean()
    
    # Calculate the ratio of the 5-day EMA of volume to the 100-day EMA of volume
    ema_vol_5 = df['volume'].ewm(span=5, adjust=False).mean()
    ema_vol_100 = df['volume'].ewm(span=100, adjust=False).mean()
    ema_vol_ratio = ema_vol_5 / ema_vol_100
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * close_ema_50_ratio + w2 * cv_daily_returns + w3 * ema_vol_ratio
    
    return heuristics_matrix
