import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the close-to-open price ratio
    square_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the cube of the absolute difference between high and low prices
    cube_high_low_diff = np.abs(df['high'] - df['low']) ** 3
    
    # Calculate the exponential of the trading amount divided by the square root of volume
    exp_amount_sqrt_volume = np.exp(df['amount'] / np.sqrt(df['volume']))
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.2, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_close_open_ratio + w2 * cube_high_low_diff + w3 * exp_amount_sqrt_volume
    
    return heuristics_matrix
