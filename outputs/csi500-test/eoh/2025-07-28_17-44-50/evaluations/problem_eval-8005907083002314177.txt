import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Apply sine transformation to the cube root of the close-to-open price ratio
    sin_cbrt_close_open_ratio = np.sin(np.cbrt(df['close'] / df['open']))
    
    # Apply a logarithmic scale adjustment to the absolute difference between high and low prices
    log_adj_high_low_diff = np.log1p(np.abs(df['high'] - df['low']))
    
    # Calculate the cubic root of the trading amount-to-volume ratio
    cbrt_amount_volume = np.cbrt(df['amount'] / df['volume'])
    
    # Define new weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sin_cbrt_close_open_ratio + w2 * log_adj_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
