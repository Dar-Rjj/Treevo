import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 10-day exponential moving average of the closing price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the standard deviation of the daily returns over the last 10 days
    std_daily_returns = daily_returns.rolling(window=10).std()
    
    # Calculate the ratio between the highest and lowest prices over the last 10 days
    high_low_ratio = df['high'].rolling(window=10).max() / df['low'].rolling(window=10).min()
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_close + w2 * std_daily_returns + w3 * high_low_ratio
    
    return heuristics_matrix
