import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the close-to-open price ratio
    square_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the mean absolute difference between the natural logarithms of the high and low prices over a 5-day rolling window
    abs_log_high_low_diff = np.abs(np.log(df['high']) - np.log(df['low']))
    mean_abs_log_high_low_diff = abs_log_high_low_diff.rolling(window=5).mean()
    
    # Calculate the cube root of the trading amount-to-volume ratio
    cbrt_amount_volume = np.cbrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_close_open_ratio + w2 * mean_abs_log_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
