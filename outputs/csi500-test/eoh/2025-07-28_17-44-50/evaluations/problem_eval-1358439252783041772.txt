import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the square of the absolute difference between the natural logarithms of the high and low prices
    sq_abs_log_high_low_diff = (np.abs(np.log(df['high']) - np.log(df['low']))) ** 2
    
    # Calculate the fourth root of the trading amount-to-volume ratio
    fourth_rt_amount_volume = (df['amount'] / df['volume']) ** (1/4)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * sq_abs_log_high_low_diff + w3 * fourth_rt_amount_volume
    
    return heuristics_matrix
