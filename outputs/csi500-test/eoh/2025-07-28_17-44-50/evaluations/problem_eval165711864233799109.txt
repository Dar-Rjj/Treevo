import pandas as pd

def heuristics_v2(df):
    # Calculate the EMA of the adjusted close
    ema_adj_close = df['close'].ewm(span=15).mean()
    
    # EMA of the ratio between the opening and closing prices
    open_close_ratio = df['open'] / df['close']
    ema_open_close_ratio = open_close_ratio.ewm(span=15).mean()
    
    # EMA of the trading volume
    ema_volume = df['volume'].ewm(span=15).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_adj_close + w2 * ema_open_close_ratio + w3 * ema_volume
    
    return heuristics_matrix
