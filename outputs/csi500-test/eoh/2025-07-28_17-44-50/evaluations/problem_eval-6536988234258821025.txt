def heuristics_v2(df):
    # Calculate the logarithm of the closing price
    log_close = np.log(df['close'])
    
    # Calculate the square of the difference between the high and low prices
    square_high_low_diff = (df['high'] - df['low']) ** 2
    
    # Calculate the trading volume-to-amount ratio
    volume_amount_ratio = df['volume'] / df['amount']
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close + w2 * square_high_low_diff + w3 * volume_amount_ratio
    
    return heuristics_matrix
