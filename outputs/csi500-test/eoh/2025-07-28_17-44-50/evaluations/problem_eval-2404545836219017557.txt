import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Apply tangent transformation to the close-to-open price ratio
    tan_close_open_ratio = np.tan(df['close'] / df['open'])
    
    # Apply a quadratic scale adjustment to the absolute difference between high and low prices
    quad_adj_high_low_diff = (np.abs(df['high'] - df['low'])) ** 2
    
    # Calculate the square of the trading amount-to-volume ratio
    sq_amount_volume = (df['amount'] / df['volume']) ** 2
    
    # Define new weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * tan_close_open_ratio + w2 * quad_adj_high_low_diff + w3 * sq_amount_volume
    
    return heuristics_matrix
