import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 10-day change in closing price
    momentum = df['close'].pct_change(10)
    
    # Volatility: Standard deviation of daily return over the last 30 days
    volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Volume-activity ratio: Average trading volume for the last 5 days divided by the 60-day average volume
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    vol_activity_ratio = avg_vol_5d / avg_vol_60d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - volatility) * vol_activity_ratio
    
    return heuristics_matrix
