import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the sine of the logarithm of the close-to-low price ratio
    sin_log_close_low_ratio = np.sin(np.log(df['close'] / df['low']))
    
    # Calculate the square of the absolute difference between open and high prices
    square_open_high_diff = (np.abs(df['open'] - df['high'])) ** 2
    
    # Calculate the arctangent of the amount divided by the square root of volume
    atan_amount_sqrt_volume = np.arctan(df['amount'] / np.sqrt(df['volume']))
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sin_log_close_low_ratio + w2 * square_open_high_diff + w3 * atan_amount_sqrt_volume
    
    return heuristics_matrix
