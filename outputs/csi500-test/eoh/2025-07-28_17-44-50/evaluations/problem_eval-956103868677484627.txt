import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Apply cosine transformation to the close-to-open price ratio
    cos_close_open_ratio = np.cos(df['close'] / df['open'])
    
    # Apply a square root scale adjustment to the absolute difference between high and low prices
    sqrt_adj_high_low_diff = np.sqrt(np.abs(df['high'] - df['low']))
    
    # Calculate the square of the trading amount-to-volume ratio
    sq_amount_volume = (df['amount'] / df['volume']) ** 2
    
    # Define new weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cos_close_open_ratio + w2 * sqrt_adj_high_low_diff + w3 * sq_amount_volume
    
    return heuristics_matrix
