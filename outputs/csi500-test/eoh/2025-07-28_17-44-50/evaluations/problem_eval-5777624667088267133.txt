def heuristics_v2(df):
    # Calculate the 20-day moving average of high and low
    avg_high = df['high'].rolling(window=20).mean()
    avg_low = df['low'].rolling(window=20).mean()
    
    # Calculate the difference between the 20-day moving average of (high + low)/2 and the close price
    diff = ((avg_high + avg_low) / 2 - df['close'])
    
    # Adjust the difference by the 5-day moving average of volume, capturing trend strength
    volume_trend = df['volume'].rolling(window=5).mean()
    adjusted_diff = diff * (volume_trend / df['volume'])
    
    heuristics_matrix = adjusted_diff
    return heuristics_matrix
