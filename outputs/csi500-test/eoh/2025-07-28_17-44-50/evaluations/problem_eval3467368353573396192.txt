import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = (df['close'] / df['open']) ** (1/3)
    
    # Calculate the natural logarithm difference between the high and low prices, then divide by the open price
    log_high_low_diff_over_open = (np.log(df['high']) - np.log(df['low'])) / df['open']
    
    # Calculate the cube root of the product of trading amount and volume
    cbrt_amount_volume = (df['amount'] * df['volume']) ** (1/3)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * log_high_low_diff_over_open + w3 * cbrt_amount_volume
    
    return heuristics_matrix
