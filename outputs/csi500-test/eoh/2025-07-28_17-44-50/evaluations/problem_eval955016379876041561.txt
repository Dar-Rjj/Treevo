import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the rolling variance of the return
    returns = df['close'].pct_change()
    var_returns = returns.rolling(window=20).var()
    
    # Calculate the cube of the ratio between close and open prices
    cube_close_open_ratio = (df['close'] / df['open']) ** 3
    
    # Calculate the difference between the highest and lowest trading volumes over a 20-day window
    max_volume = df['volume'].rolling(window=20).max()
    min_volume = df['volume'].rolling(window=20).min()
    diff_volume = max_volume - min_volume
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * var_returns + w2 * cube_close_open_ratio + w3 * diff_volume
    
    return heuristics_matrix
