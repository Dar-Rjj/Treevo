import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the reciprocal of the absolute difference between the log of high and low prices
    inv_abs_log_high_low_diff = 1 / np.abs(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the natural logarithm of the trading amount adjusted by the cube of the volume
    ln_amount_adj_volume = np.log(df['amount'] * (df['volume'] ** 3))
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * inv_abs_log_high_low_diff + w3 * ln_amount_adj_volume
    
    return heuristics_matrix
