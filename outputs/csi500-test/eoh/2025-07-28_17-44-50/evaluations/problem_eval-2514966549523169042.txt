def heuristics_v2(df):
    # Calculate the cube of the close-to-open price ratio
    cube_close_open_ratio = (df['close'] / df['open']) ** 3
    
    # Calculate the square of the absolute difference between the natural logarithms of the high and low prices
    square_log_high_low_diff = (np.abs(np.log(df['high']) - np.log(df['low']))) ** 2
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cube_close_open_ratio + w2 * square_log_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
