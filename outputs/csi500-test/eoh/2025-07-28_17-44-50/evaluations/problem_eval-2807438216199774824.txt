import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 20-day change in closing price
    momentum = df['close'].pct_change(20)
    
    # Volatility: Average True Range (ATR) over the last 30 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=30).mean()
    
    # Volume-activity ratio: 10-day average volume divided by the 90-day average volume
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    avg_vol_90d = df['volume'].rolling(window=90).mean()
    vol_activity_ratio = avg_vol_10d / avg_vol_90d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - atr) * vol_activity_ratio
    
    return heuristics_matrix
