import pandas as pd

def heuristics_v2(df):
    # Relative Strength Index (RSI) calculation
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume Surge Indicator: Volume change over the last 5 days divided by the 30-day average volume
    vol_change_5d = df['volume'].pct_change(5)
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_surge_indicator = vol_change_5d / avg_vol_30d
    
    # Combine RSI and Volume Surge Indicator
    heuristics_matrix = rsi * vol_surge_indicator
    
    return heuristics_matrix
