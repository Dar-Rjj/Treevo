def heuristics_v2(df):
    # Calculate the weighted closing price using volume as weights
    weighted_close = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # The factor is the difference between the current close and the weighted close
    heuristics_matrix = df['close'] - weighted_close
    return heuristics_matrix
