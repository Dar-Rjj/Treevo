def heuristics_v2(df):
    df['close_20ma'] = df['close'].rolling(window=20).mean()
    df['volume_100ma'] = df['volume'].rolling(window=100).mean()
    df['momentum'] = (df['close'] - df['close_20ma']) * (df['volume'] / df['volume_100ma'])
    heuristics_matrix = df['momentum'].dropna()
    return heuristics_matrix
