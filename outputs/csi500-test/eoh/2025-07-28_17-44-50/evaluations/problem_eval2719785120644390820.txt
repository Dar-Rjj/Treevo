import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 50-day change in closing price
    momentum = df['close'].pct_change(50)
    
    # Volatility: Average True Range (ATR) over the last 10 and 60 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_10d = tr.rolling(window=10).mean()
    atr_60d = tr.rolling(window=60).mean()
    atr_ratio = atr_10d / atr_60d
    
    # Volume signal: Percentage change in trading volume over the last 10 days
    pct_change_volume_10 = df['volume'].pct_change(10)
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum + atr_ratio) * pct_change_volume_10
    
    return heuristics_matrix
