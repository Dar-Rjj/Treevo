import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the volume-to-amount ratio
    sqrt_volume_amount_ratio = np.sqrt(df['volume'] / df['amount'])
    
    # Calculate the absolute difference between the natural logarithms of the highest and lowest prices
    abs_log_high_low_diff = np.abs(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the square of the close-to-open price ratio
    square_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_volume_amount_ratio + w2 * abs_log_high_low_diff + w3 * square_close_open_ratio
    
    return heuristics_matrix
