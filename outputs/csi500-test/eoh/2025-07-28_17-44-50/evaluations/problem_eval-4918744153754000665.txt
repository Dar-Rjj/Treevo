import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the reciprocal of the high-low spread
    recip_high_low_spread = 1 / (df['high'] - df['low'])
    
    # Calculate the trading amount to volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Define weights for each component
    w1, w2, w3 = 0.3, 0.4, 0.3
    
    # Combine the components into a single alpha factor using the harmonic mean formula
    numerator = w1 + w2 + w3
    denominator = (w1 / exp_close_open_ratio) + (w2 / recip_high_low_spread) + (w3 / amount_volume_ratio)
    heuristics_matrix = numerator / denominator
    
    return heuristics_matrix
