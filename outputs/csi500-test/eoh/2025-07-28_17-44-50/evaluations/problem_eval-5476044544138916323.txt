import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = (df['close'] / df['open']) ** 0.5
    
    # Calculate the natural logarithm difference between the high and low prices
    log_high_low_diff = np.log(df['high']) - np.log(df['low'])
    
    # Calculate the square root of the sum of trading amount and volume
    sqrt_amount_volume = (df['amount'] + df['volume']) ** 0.5
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * log_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
