import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily return
    returns = df['close'].pct_change()
    
    # Exponential Moving Average (EMA) of the return
    ema_returns = returns.ewm(span=10).mean()
    
    # EMA of the log ratio between the highest and lowest prices
    high_low_log_ratio = np.log(df['high'] / df['low'])
    ema_high_low_log_ratio = high_low_log_ratio.ewm(span=15).mean()
    
    # EMA of the trading volume
    ema_volume = df['volume'].ewm(span=30).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_returns + w2 * ema_high_low_log_ratio + w3 * ema_volume
    
    return heuristics_matrix
