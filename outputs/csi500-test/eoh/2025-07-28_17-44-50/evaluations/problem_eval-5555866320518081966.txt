import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the absolute difference between the natural logarithms of the high and low prices
    abs_log_high_low_diff = np.abs(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the cube root of the trading amount-to-volume ratio
    cbrt_amount_volume = np.cbrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * abs_log_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
