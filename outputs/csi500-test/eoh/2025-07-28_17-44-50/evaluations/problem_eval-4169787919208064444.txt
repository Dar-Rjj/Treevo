import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the difference between high and low prices
    square_high_low_diff = (df['high'] - df['low']) ** 2
    
    # Calculate the exponential of the closing to opening price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the square root of the trading volume
    sqrt_volume = np.sqrt(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_high_low_diff + w2 * exp_close_open_ratio + w3 * sqrt_volume
    
    return heuristics_matrix
