import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the reciprocal of the cube of the natural logarithms difference between the high and low prices
    inv_log_high_low_diff_cube = 1 / (np.abs(np.log(df['high']) - np.log(df['low'])) ** 3)
    
    # Calculate the square of the product of trading amount and volume
    sq_amount_volume = (df['amount'] * df['volume']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * inv_log_high_low_diff_cube + w3 * sq_amount_volume
    
    return heuristics_matrix
