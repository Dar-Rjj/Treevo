import pandas as pd

def heuristics_v2(df):
    close_10 = df['close'].rolling(window=10).mean()
    close_30 = df['close'].rolling(window=30).mean()
    volume_5 = df['volume'].rolling(window=5).mean()
    volume_20 = df['volume'].rolling(window=20).mean()
    
    price_diff = close_10 - close_30
    volume_ratio = volume_5 / volume_20
    
    heuristics_matrix = price_diff * volume_ratio
    return heuristics_matrix
