def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the square of the absolute difference between the high and low prices
    square_high_low_diff = (df['high'] - df['low']).abs() ** 2
    
    # Calculate the natural logarithm of the trading volume
    log_volume = np.log(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * square_high_low_diff + w3 * log_volume
    
    return heuristics_matrix
