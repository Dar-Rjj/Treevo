import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Apply cosine transformation to the square of the close-to-open price ratio
    cos_sq_close_open_ratio = np.cos((df['close'] / df['open'])**2)
    
    # Apply a quadratic scale adjustment to the absolute difference between high and low prices
    quad_adj_high_low_diff = (np.abs(df['high'] - df['low']))**2
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define new weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cos_sq_close_open_ratio + w2 * quad_adj_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
