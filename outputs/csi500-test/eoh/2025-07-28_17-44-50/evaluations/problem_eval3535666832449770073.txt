import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the close-to-open price ratio
    sq_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the square root of the absolute difference between the high and low prices
    sqrt_high_low_diff = np.sqrt(np.abs(df['high'] - df['low']))
    
    # Calculate the logarithm of the trading volume
    log_volume = np.log(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sq_close_open_ratio + w2 * sqrt_high_low_diff + w3 * log_volume
    
    return heuristics_matrix
