def heuristics_v2(df):
    cbrt_close_open_ratio = np.cbrt(df['close'] / df['open'])
    exp_abs_log_high_low_diff = np.exp(np.abs(np.log(df['high']) - np.log(df['low'])))
    sq_amount_volume = np.square(df['amount'] / df['volume'])
    
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * exp_abs_log_high_low_diff + w3 * sq_amount_volume
    
    return heuristics_matrix
