import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the exponential of the negative natural logarithms difference between the high and low prices
    exp_neg_log_high_low_diff = np.exp(-(np.abs(np.log(df['high']) - np.log(df['low']))))
    
    # Calculate the square root of the product of trading amount and volume
    sqrt_amount_volume = (df['amount'] * df['volume']) ** (1/2)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * exp_neg_log_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
