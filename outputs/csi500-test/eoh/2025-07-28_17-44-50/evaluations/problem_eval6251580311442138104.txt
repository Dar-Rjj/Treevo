import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the cosine of the difference between the high and low prices
    cos_high_low_diff = np.cos(df['high'] - df['low'])
    
    # Calculate the natural logarithm of one plus the volume
    log_one_plus_volume = np.log(1 + df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * cos_high_low_diff + w3 * log_one_plus_volume
    
    return heuristics_matrix
