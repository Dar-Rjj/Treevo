def heuristics_v2(df):
    # Calculate the daily return
    returns = df['close'].pct_change()
    
    # Exponential Moving Average (EMA) of the return
    ema_returns = returns.ewm(span=10).mean()
    
    # Adjusted ratio: taking the natural logarithm of the high to low price ratio
    adj_high_low_ratio = (df['high'] / df['low']).apply(lambda x: np.log(x))
    ema_adj_high_low_ratio = adj_high_low_ratio.ewm(span=25).mean()
    
    # EMA of the volume
    ema_volume = df['volume'].ewm(span=30).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_returns + w2 * ema_adj_high_low_ratio + w3 * ema_volume
    
    return heuristics_matrix
