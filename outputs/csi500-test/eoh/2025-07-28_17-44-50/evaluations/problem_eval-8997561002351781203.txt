def heuristics_v2(df):
    def sma(series, window=5):
        return series.rolling(window=window).mean()

    log_returns = (df['close'] / df['close'].shift(1)).apply(np.log)
    volume_change = df['volume'] - df['volume'].shift(1)
    sma_log_returns_5 = sma(log_returns, 5)
    sma_volume_change_5 = sma(volume_change, 5)
    heuristics_matrix = (sma_log_returns_5 * sma_volume_change_5).dropna()
    
    return heuristics_matrix
