import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the high-to-low price ratio
    square_high_low_ratio = (df['high'] / df['low']) ** 2
    
    # Calculate the exponential of the difference between the close and open prices
    exp_close_open_diff = np.exp(df['close'] - df['open'])
    
    # Calculate the square root of the product of trading amount and volume
    sqrt_amount_volume_product = np.sqrt(df['amount'] * df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_high_low_ratio + w2 * exp_close_open_diff + w3 * sqrt_amount_volume_product
    
    return heuristics_matrix
