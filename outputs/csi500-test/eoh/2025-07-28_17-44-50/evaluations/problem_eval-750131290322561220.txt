import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the inverse of the close-to-open price ratio
    inv_close_open_ratio = 1 / (df['close'] / df['open'])
    
    # Calculate the square of the absolute difference between high and low prices
    sq_high_low_diff = (np.abs(df['high'] - df['low'])) ** 2
    
    # Calculate the base 2 logarithm of the trading amount divided by the volume
    log2_amount_volume = np.log2(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * inv_close_open_ratio + w2 * sq_high_low_diff + w3 * log2_amount_volume
    
    return heuristics_matrix
