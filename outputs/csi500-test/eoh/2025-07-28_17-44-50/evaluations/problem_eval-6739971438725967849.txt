import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the inverse of the close-to-open price ratio
    inv_close_open_ratio = 1 / (df['close'] / df['open'])
    
    # Calculate the inverse of the high-low spread normalized by the highest price
    inv_high_low_spread_normalized = 1 / ((df['high'] - df['low']) / df['high'])
    
    # Calculate the square of the volume over amount
    square_volume_amount = (df['volume'] / df['amount']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor using a weighted harmonic mean
    heuristics_matrix = 3 / (w1 / inv_close_open_ratio + w2 / inv_high_low_spread_normalized + w3 / square_volume_amount)
    
    return heuristics_matrix
