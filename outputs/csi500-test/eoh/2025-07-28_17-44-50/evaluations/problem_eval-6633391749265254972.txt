def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = np.cbrt(df['close'] / df['open'])
    
    # Calculate the absolute difference between the logarithms (base 10) of the high and low prices
    abs_log_high_low_diff = np.abs(np.log10(df['high']) - np.log10(df['low']))
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * abs_log_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
