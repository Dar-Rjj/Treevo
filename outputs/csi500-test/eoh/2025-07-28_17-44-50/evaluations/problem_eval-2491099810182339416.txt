def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the cube of the natural logarithm difference between the high and low prices
    cube_log_high_low_diff = (np.log(df['high']) - np.log(df['low'])) ** 3
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = (df['amount'] / df['volume']) ** (1/2)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor using a weighted product
    heuristics_matrix = (w1 * close_open_ratio) * (w2 * cube_log_high_low_diff) * (w3 * sqrt_amount_volume)
    
    return heuristics_matrix
