import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the high-to-low price ratio
    sqrt_high_low_ratio = np.sqrt(df['high'] / df['low'])
    
    # Calculate the natural logarithm of the close-to-average price ratio
    log_close_avg_diff = np.log(df['close'] / ((df['open'] + df['close']) / 2))
    
    # Calculate the square of the trading amount
    squared_amount = df['amount'] ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_high_low_ratio + w2 * log_close_avg_diff + w3 * squared_amount
    
    return heuristics_matrix
