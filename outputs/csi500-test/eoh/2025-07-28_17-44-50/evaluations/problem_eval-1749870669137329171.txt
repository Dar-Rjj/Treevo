import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the negative close-to-open price ratio
    exp_neg_close_open_ratio = np.exp(-(df['close'] / df['open']))
    
    # Calculate the square of the natural logarithm difference between the high and low prices
    log_high_low_diff_sq = (np.log(df['high']) - np.log(df['low'])) ** 2
    
    # Calculate the fourth root of the product of trading amount and the square of volume
    fourth_rt_amount_volume_sq = (df['amount'] * df['volume']**2) ** (1/4)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_neg_close_open_ratio + w2 * log_high_low_diff_sq + w3 * fourth_rt_amount_volume_sq
    
    return heuristics_matrix
