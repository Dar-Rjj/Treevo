def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = (df['close'] / df['open']) ** 0.5
    
    # Calculate the natural logarithm of the ratio between high and low prices
    log_high_low_ratio = np.log(df['high'] / df['low'])
    
    # Calculate the square of the product of trading amount and volume
    square_amount_volume = (df['amount'] * df['volume']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.3, 0.5, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * log_high_low_ratio + w3 * square_amount_volume
    
    return heuristics_matrix
