import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential moving average of the close-to-open price ratio with a span of 5
    ema_close_open_ratio = df['close'] / df['open'].ewm(span=5, adjust=False).mean()
    
    # Calculate the square of the natural logarithms difference between the high and low prices
    sqr_log_high_low_diff = (np.log(df['high']) - np.log(df['low'])) ** 2
    
    # Calculate the square root of the product of trading amount and volume
    sqrt_amount_volume = np.sqrt(df['amount'] * df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_close_open_ratio + w2 * sqr_log_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
