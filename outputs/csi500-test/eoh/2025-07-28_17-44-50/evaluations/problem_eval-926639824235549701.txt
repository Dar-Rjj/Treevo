import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between the natural logarithms of the close and open prices
    log_diff_close_open = np.log(df['close']) - np.log(df['open'])
    
    # Calculate the percentage change in volume
    pct_change_volume = df['volume'].pct_change()
    
    # Calculate the ratio of the current close to the 10-day average close
    avg_close_10 = df['close'].rolling(window=10).mean()
    ratio_close_avg = df['close'] / avg_close_10
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_diff_close_open + w2 * pct_change_volume + w3 * ratio_close_avg
    
    return heuristics_matrix
