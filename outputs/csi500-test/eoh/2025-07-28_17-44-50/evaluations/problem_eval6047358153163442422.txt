import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=10)
    volatility = df['close'].rolling(window=30).std()
    volume_sma_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    heuristics_matrix = (momentum + 1/volatility + volume_sma_ratio) / 3
    return heuristics_matrix
