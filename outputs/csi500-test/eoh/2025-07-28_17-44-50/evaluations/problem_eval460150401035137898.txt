def heuristics_v2(df):
    # Calculate the natural logarithm of the cube root of the close-to-open price ratio
    log_cbrt_close_open_ratio = np.log(np.cbrt(df['close'] / df['open']))
    
    # Calculate the square of the absolute difference between the natural logarithms of the high and low prices
    sq_abs_log_high_low_diff = (np.abs(np.log(df['high']) - np.log(df['low']))) ** 2
    
    # Calculate the cubic root of the trading amount-to-volume ratio
    cbrt_amount_volume = np.cbrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_cbrt_close_open_ratio + w2 * sq_abs_log_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
