def heuristics_v2(df):
    # Calculate the natural logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the cube of the difference between high and low prices
    cube_high_low_diff = (df['high'] - df['low']) ** 3
    
    # Calculate the square root of the trading amount divided by the volume
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.25, 0.25
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * cube_high_low_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
