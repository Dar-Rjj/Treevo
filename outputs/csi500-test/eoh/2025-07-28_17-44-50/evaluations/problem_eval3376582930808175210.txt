import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the square of the high-low spread
    sq_high_low_spread = (df['high'] - df['low']) ** 2
    
    # Calculate the cube root of the trading amount to volume ratio
    cbrt_amount_volume_ratio = np.cbrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor using a weighted arithmetic mean
    heuristics_matrix = w1 * log_close_open_ratio + w2 * sq_high_low_spread + w3 * cbrt_amount_volume_ratio
    
    return heuristics_matrix
