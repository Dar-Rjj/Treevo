import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the open-to-close price ratio
    exp_open_close_ratio = np.exp(df['open'] / df['close'])
    
    # Calculate the cubic root of the difference between the high and low prices
    cbrt_high_low_diff = (df['high'] - df['low']) ** (1/3)
    
    # Calculate the square of the trading volume
    sq_volume = df['volume'] ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_open_close_ratio + w2 * cbrt_high_low_diff + w3 * sq_volume
    
    return heuristics_matrix
