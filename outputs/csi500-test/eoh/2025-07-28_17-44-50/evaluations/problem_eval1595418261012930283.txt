def heuristics_v2(df):
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    inv_log_high_low_diff_squared = 1 / (np.abs(np.log(df['high']) - np.log(df['low'])) ** 2)
    log_amount_volume = np.log(df['amount'] * df['volume'])
    
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * inv_log_high_low_diff_squared + w3 * log_amount_volume
    
    return heuristics_matrix
