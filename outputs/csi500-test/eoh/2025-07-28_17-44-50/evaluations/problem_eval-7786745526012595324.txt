import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the square root of the absolute difference between the log of high and low prices
    sqrt_log_high_low_diff = np.sqrt(np.abs(np.log(df['high']) - np.log(df['low'])))
    
    # Calculate the product of the trading amount and the inverse of the volume
    amount_inv_volume = df['amount'] * (1 / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * sqrt_log_high_low_diff + w3 * amount_inv_volume
    
    return heuristics_matrix
