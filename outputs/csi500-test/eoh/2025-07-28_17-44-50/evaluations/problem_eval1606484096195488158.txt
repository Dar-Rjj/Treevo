def heuristics_v2(df):
    # Calculate the 50-day and 100-day close price differences for momentum
    df['momentum_50'] = df['close'].diff(50)
    df['momentum_100'] = df['close'].diff(100)
    
    # Combine the momentum components
    df['momentum_combined'] = df['momentum_50'] + df['momentum_100']
    
    # Calculate RSI over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the 30-day rolling correlation between volume and close prices
    df['corr_vol_close'] = df['volume'].rolling(window=30).corr(df['close'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * df['momentum_combined'] + w2 * df['rsi_14'] + w3 * df['corr_vol_close']
    
    return heuristics_matrix
