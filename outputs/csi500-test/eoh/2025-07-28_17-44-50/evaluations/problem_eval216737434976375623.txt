import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithmic return
    log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the square of the close-to-open price ratio
    square_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the square root of the trading volume
    sqrt_volume = np.sqrt(df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_returns + w2 * square_close_open_ratio + w3 * sqrt_volume
    
    return heuristics_matrix
