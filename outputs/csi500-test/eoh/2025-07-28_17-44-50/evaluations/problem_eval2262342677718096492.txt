import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the cube of the high-to-low price ratio
    cube_high_low_ratio = (df['high'] / df['low']) ** 3
    
    # Calculate the natural logarithm difference between the close and open prices
    log_close_open_diff = np.log(df['close']) - np.log(df['open'])
    
    # Calculate the cubic root of the trading amount
    cbrt_amount = np.cbrt(df['amount'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cube_high_low_ratio + w2 * log_close_open_diff + w3 * cbrt_amount
    
    return heuristics_matrix
