def heuristics_v2(df):
    # Calculate the square of the close-to-open price ratio
    sq_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the logarithmic difference between the high and low prices
    log_high_low_diff = np.log(df['high']) - np.log(df['low'])
    
    # Calculate the cube root of the trading amount-to-volume ratio
    cbrt_amount_volume = np.cbrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sq_close_open_ratio + w2 * log_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
