import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 30-day change in closing price
    momentum = df['close'].pct_change(30)
    
    # Volatility: Standard Deviation of True Range (ATR) over the last 20 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_std = tr.rolling(window=20).std()
    
    # Volume-activity ratio: 15-day EMA volume divided by the 120-day EMA volume
    ema_vol_15d = df['volume'].ewm(span=15, adjust=False).mean()
    ema_vol_120d = df['volume'].ewm(span=120, adjust=False).mean()
    vol_activity_ratio = ema_vol_15d / ema_vol_120d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - atr_std) * vol_activity_ratio
    
    return heuristics_matrix
