import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum signal: 30-day rate of change in closing price
    momentum = df['close'].pct_change(30)
    
    # Volatility: Coefficient of Variation (CV) of the True Range over the last 20 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    cv_tr = tr.rolling(window=20).std() / tr.rolling(window=20).mean()
    
    # Volume-activity log-ratio: Logarithm of the 15-day average volume divided by the 60-day average volume
    avg_vol_15d = df['volume'].rolling(window=15).mean()
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    vol_activity_log_ratio = np.log(avg_vol_15d / avg_vol_60d)
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - cv_tr) * vol_activity_log_ratio
    
    return heuristics_matrix
