def heuristics_v2(df):
    # Calculate the average closing price over the last 5 days
    avg_close_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the standard deviation of the closing price over the last 5 days
    std_close_5d = df['close'].rolling(window=5).std()
    
    # Calculate the ratio of the amount traded to the volume
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Compute the momentum-based component
    momentum_component = (df['close'] - avg_close_5d) / std_close_5d
    
    # Compute the liquidity-based component
    liquidity_component = np.log(amount_volume_ratio)
    
    # Combine the components into a single alpha factor
    heuristics_matrix = momentum_component + liquidity_component
    
    return heuristics_matrix
