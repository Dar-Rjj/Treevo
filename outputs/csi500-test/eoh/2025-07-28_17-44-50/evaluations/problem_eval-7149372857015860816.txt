def heuristics_v2(df):
    # Calculate the square root of the open-to-close price ratio
    sqrt_open_close_ratio = np.sqrt(df['open'] / df['close'])
    
    # Apply an exponential scale adjustment to the negative absolute high-low price difference
    exp_neg_adj_high_low_diff = np.exp(-np.abs(df['high'] - df['low']))
    
    # Calculate the fourth root of the trading volume
    quart_root_volume = np.power(df['volume'], 1/4)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.5, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_open_close_ratio + w2 * exp_neg_adj_high_low_diff + w3 * quart_root_volume
    
    return heuristics_matrix
