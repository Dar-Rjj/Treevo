def heuristics_v2(df):
    # Calculate the natural logarithm of the adjusted close-to-open price ratio
    log_adj_close_open_ratio = np.log((df['close'] - df['open']) / df['open'])
    
    # Calculate the cube of the difference between the high and low prices
    cubed_high_low_diff = (df['high'] - df['low']) ** 3
    
    # Calculate the exponential of the trading amount divided by the square of the volume
    exp_amount_volume_sq = np.exp(df['amount'] / (df['volume'] ** 2))
    
    # Define weights for each component
    w1, w2, w3 = 0.3, 0.4, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_adj_close_open_ratio + w2 * cubed_high_low_diff + w3 * exp_amount_volume_sq
    
    return heuristics_matrix
