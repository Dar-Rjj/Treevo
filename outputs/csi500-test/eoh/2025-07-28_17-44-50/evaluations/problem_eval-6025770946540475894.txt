import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the square root of the absolute difference between high and low prices
    sqrt_high_low_diff = np.sqrt(np.abs(df['high'] - df['low']))
    
    # Calculate the natural logarithm of the trading amount divided by the cube root of volume
    log_amount_cbrt_volume = np.log(df['amount'] / (df['volume'] ** (1/3)))
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * sqrt_high_low_diff + w3 * log_amount_cbrt_volume
    
    return heuristics_matrix
