def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the cube of the absolute difference between the natural logarithms of the high and low prices
    cube_abs_log_high_low_diff = (np.abs(np.log(df['high']) - np.log(df['low']))) ** 3
    
    # Calculate the square of the trading amount-to-volume ratio
    sq_amount_volume = (df['amount'] / df['volume']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * cube_abs_log_high_low_diff + w3 * sq_amount_volume
    
    return heuristics_matrix
