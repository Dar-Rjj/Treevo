def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = (df['close'] / df['open']) ** (1/3)
    
    # Calculate the absolute difference between the high and low prices
    abs_high_low_diff = abs(df['high'] - df['low'])
    
    # Calculate the square of the trading amount divided by volume
    sq_amount_per_volume = (df['amount'] / df['volume']) ** 2
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * abs_high_low_diff + w3 * sq_amount_per_volume
    
    return heuristics_matrix
