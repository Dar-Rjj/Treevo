import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square of the close-to-open price ratio
    square_close_open_ratio = (df['close'] / df['open']) ** 2
    
    # Calculate the cube of the absolute difference between high and low prices
    cube_high_low_diff = np.abs(df['high'] - df['low']) ** 3
    
    # Calculate the inverse of the natural logarithm of the trading amount divided by volume
    inv_log_amount_volume = 1 / np.log(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.25, 0.25
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_close_open_ratio + w2 * cube_high_low_diff + w3 * inv_log_amount_volume
    
    return heuristics_matrix
