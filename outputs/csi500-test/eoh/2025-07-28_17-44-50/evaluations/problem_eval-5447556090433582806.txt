import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute the hyperbolic tangent of the close-to-open price ratio
    tanh_close_open_ratio = np.tanh(df['close'] / df['open'])
    
    # Calculate the exponential moving average (EMA) of the high-low price difference
    ema_high_low_diff = df['high'].ewm(span=5, adjust=False).mean() - df['low'].ewm(span=5, adjust=False).mean()
    
    # Compute the square root of the product of trading amount and volume
    sqrt_amount_volume_prod = np.sqrt(df['amount'] * df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * tanh_close_open_ratio + w2 * ema_high_low_diff + w3 * sqrt_amount_volume_prod
    
    return heuristics_matrix
