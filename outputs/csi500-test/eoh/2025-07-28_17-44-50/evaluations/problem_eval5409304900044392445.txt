def heuristics_v2(df):
    # Calculate the square root of the logarithm of the close-to-open price ratio
    sqrt_log_close_open_ratio = np.sqrt(np.log(df['close'] / df['open']))
    
    # Calculate the natural logarithm of the inverse of the high-low price difference
    log_inv_high_low_diff = np.log(1 / (df['high'] - df['low']))
    
    # Calculate the logarithm of the product of trading amount and volume
    log_amount_volume = np.log(df['amount'] * df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_log_close_open_ratio + w2 * log_inv_high_low_diff + w3 * log_amount_volume
    
    return heuristics_matrix
