import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    returns = df['close'].pct_change()
    
    # Exponential Moving Average (EMA) of the return
    ema_returns = returns.ewm(span=20).mean()
    
    # EMA of the volatility (standard deviation of returns)
    ema_volatility = returns.rolling(window=20).std().ewm(span=20).mean()
    
    # EMA of the trading volume
    ema_volume = df['volume'].ewm(span=20).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_returns + w2 * ema_volatility + w3 * ema_volume
    
    return heuristics_matrix
