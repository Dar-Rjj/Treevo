def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the exponential of the difference between the logarithms of the high and low prices
    exp_log_high_low_diff = np.exp(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the natural logarithm of the product of trading amount and volume
    log_amount_volume = np.log(df['amount'] * df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * exp_log_high_low_diff + w3 * log_amount_volume
    
    return heuristics_matrix
