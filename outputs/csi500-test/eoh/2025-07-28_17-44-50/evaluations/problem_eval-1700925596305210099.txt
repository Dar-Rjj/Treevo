def heuristics_v2(df):
    def calculate_momentum(series, short_window, long_window):
        short_roc = series.pct_change(short_window)
        long_roc = series.pct_change(long_window)
        combined_roc = (short_roc + long_roc) / 2
        smoothed_roc = combined_roc.rolling(window=5).mean()
        return smoothed_roc

    heuristics_matrix = (calculate_momentum(df['close'], 5, 20) 
                        + calculate_momentum(df['volume'], 5, 20)) / 2
    return heuristics_matrix
