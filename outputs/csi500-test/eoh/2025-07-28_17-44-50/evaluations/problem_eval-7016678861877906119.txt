import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 20-day change in closing price
    momentum = df['close'].pct_change(20)
    
    # Volatility: Standard Deviation of True Range (ATR) over the last 10 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_std = tr.rolling(window=10).std()
    
    # Volume-activity ratio: 10-day EMA volume divided by the 90-day EMA volume
    ema_vol_10d = df['volume'].ewm(span=10, adjust=False).mean()
    ema_vol_90d = df['volume'].ewm(span=90, adjust=False).mean()
    vol_activity_ratio = ema_vol_10d / ema_vol_90d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - atr_std) * vol_activity_ratio
    
    return heuristics_matrix
