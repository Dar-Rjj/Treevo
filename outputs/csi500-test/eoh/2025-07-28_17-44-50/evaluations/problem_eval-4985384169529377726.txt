def heuristics_v2(df):
    # Calculate the square of the open-to-close price ratio
    sqr_open_close_ratio = (df['open'] / df['close']) ** 2
    
    # Apply hyperbolic tangent to the high-low difference
    tanh_high_low_diff = np.tanh(df['high'] - df['low'])
    
    # Compute the natural logarithm of the volume-over-amount
    ln_volume_amount = np.log(df['volume'] / df['amount'])
    
    # Define the weights for each component
    w1, w2, w3 = 0.6, 0.2, 0.2
    
    # Combine components into a single alpha factor
    heuristics_matrix = w1 * sqr_open_close_ratio + w2 * tanh_high_low_diff + w3 * ln_volume_amount
    
    return heuristics_matrix
