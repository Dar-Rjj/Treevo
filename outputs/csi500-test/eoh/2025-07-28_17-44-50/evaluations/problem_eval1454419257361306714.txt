import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the high-low price spread normalized by the opening price
    high_low_spread_normalized = (df['high'] - df['low']) / df['open']
    
    # Calculate the natural logarithm of the trading amount-to-volume ratio
    log_amount_volume = np.log(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor using a weighted geometric mean
    heuristics_matrix = np.power(close_open_ratio, w1) * np.power(high_low_spread_normalized, w2) * np.power(np.exp(log_amount_volume), w3)
    
    return heuristics_matrix
