import pandas as pd

def heuristics_v2(df):
    df['close_10_avg'] = df['close'].rolling(window=10).mean()
    df['momentum'] = (df['close'] - df['close_10_avg']) / df['close_10_avg']
    df['smoothed_momentum'] = df['momentum'].ewm(span=5, adjust=False).mean()
    heuristics_matrix = df['smoothed_momentum'].dropna()
    return heuristics_matrix
