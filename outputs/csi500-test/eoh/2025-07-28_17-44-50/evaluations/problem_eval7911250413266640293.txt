def heuristics_v2(df):
    # Momentum signal: 10-day change in closing price
    momentum = df['close'].pct_change(10)
    
    # Volatility: Standard deviation of the True Range (TR) over the last 20 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    tr_std = tr.rolling(window=20).std()
    
    # Volume-activity difference: 5-day average volume minus the 60-day average volume
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    vol_activity_diff = avg_vol_5d - avg_vol_60d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - tr_std) * vol_activity_diff
    
    return heuristics_matrix
