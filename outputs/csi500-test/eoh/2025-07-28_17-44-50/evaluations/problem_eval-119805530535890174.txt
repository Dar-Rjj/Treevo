def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = (df['close'] / df['open']) ** (1/3)
    
    # Calculate the difference between the natural logarithm of high and low, then multiply by their average
    log_high_low_diff = (np.log(df['high']) - np.log(df['low'])) * ((np.log(df['high']) + np.log(df['low'])) / 2)
    
    # Calculate the cube root of the trading amount
    cbrt_amount = df['amount'] ** (1/3)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * log_high_low_diff + w3 * cbrt_amount
    
    return heuristics_matrix
