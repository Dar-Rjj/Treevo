import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the difference between the average of high and low prices and the closing price
    avg_high_low_close_diff = (df['high'] + df['low'])/2 - df['close']
    
    # Calculate the square root of the trading amount-to-volume ratio
    sqrt_amount_volume = np.sqrt(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.25, 0.25
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * close_open_ratio + w2 * avg_high_low_close_diff + w3 * sqrt_amount_volume
    
    return heuristics_matrix
