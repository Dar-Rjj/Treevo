    ema_high_low_ratio = high_low_ratio.ewm(span=30).mean()
    
    # EMA of the trading volume
    ema_volume = df['volume'].ewm(span=30).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2

    heuristics_matrix = w1 * ema_log_returns + w2 * ema_high_low_ratio + w3 * ema_volume
