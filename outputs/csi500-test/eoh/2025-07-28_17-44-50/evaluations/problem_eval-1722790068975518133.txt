import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the sine of the close-to-high price ratio
    sin_close_high_ratio = np.sin(df['close'] / df['high'])
    
    # Calculate the square of the natural logarithm difference between the open and low prices
    sqr_log_open_low_diff = (np.log(df['open']) - np.log(df['low'])) ** 2
    
    # Calculate the trading amount-to-volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sin_close_high_ratio + w2 * sqr_log_open_low_diff + w3 * amount_volume_ratio
    
    return heuristics_matrix
