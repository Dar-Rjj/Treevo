import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential moving average of closing prices with a span of 5
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the natural logarithm of the volume-to-amount ratio
    log_volume_amount_ratio = np.log(df['volume'] / df['amount'])
    
    # Calculate the absolute difference between the high and low prices
    abs_high_low_diff = np.abs(df['high'] - df['low'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_close + w2 * log_volume_amount_ratio + w3 * abs_high_low_diff
    
    return heuristics_matrix
