import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily return
    returns = df['close'].pct_change()
    
    # Calculate the square root of the daily return
    sqrt_returns = np.sqrt(returns.abs())
    
    # Calculate the logarithmic difference between high and low prices
    log_high_low_diff = np.log(df['high']) - np.log(df['low'])
    
    # Calculate the ratio of volume to amount
    vol_to_amount_ratio = df['volume'] / df['amount']
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_returns + w2 * log_high_low_diff + w3 * vol_to_amount_ratio
    
    return heuristics_matrix
