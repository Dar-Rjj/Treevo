def heuristics_v2(df):
    def calculate_volatility_adjusted_momentum(series, short_window, long_window, vol_window):
        short_roc = series.pct_change(short_window)
        long_roc = series.pct_change(long_window)
        combined_roc = (short_roc + long_roc) / 2
        volatility = series.rolling(window=vol_window).std()
        adjusted_momentum = combined_roc / volatility
        return adjusted_momentum

    heuristics_matrix = (calculate_volatility_adjusted_momentum(df['close'], 5, 20, 20) 
                        + calculate_volatility_adjusted_momentum(df['volume'], 5, 20, 20)) / 2
    return heuristics_matrix
