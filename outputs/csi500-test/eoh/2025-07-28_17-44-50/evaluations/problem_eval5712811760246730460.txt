import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the cube of the high-minus-low price range
    cube_high_low_diff = (df['high'] - df['low']) ** 3
    
    # Calculate the natural logarithm of the trading amount-to-volume ratio
    ln_amount_volume = np.log(df['amount'] / df['volume'])
    
    # Combine the components into a single alpha factor using the geometric mean formula
    heuristics_matrix = (sqrt_close_open_ratio * cube_high_low_diff * ln_amount_volume) ** (1/3)
    
    return heuristics_matrix
