import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = np.cbrt(df['close'] / df['open'])
    
    # Calculate the square of the absolute difference between the natural logarithms of the high and low prices
    sq_abs_log_high_low_diff = np.square(np.abs(np.log(df['high']) - np.log(df['low'])))
    
    # Calculate the natural logarithm of the trading amount-to-volume ratio
    log_amount_volume = np.log(df['amount'] / df['volume'])
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * sq_abs_log_high_low_diff + w3 * log_amount_volume
    
    return heuristics_matrix
