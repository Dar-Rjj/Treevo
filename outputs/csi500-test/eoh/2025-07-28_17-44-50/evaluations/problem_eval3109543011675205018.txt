def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the square root of the natural logarithms difference between the high and low prices
    sqrt_log_high_low_diff = np.sqrt(np.abs(np.log(df['high']) - np.log(df['low'])))
    
    # Calculate the fourth root of the product of trading amount and volume
    fourth_rt_amount_volume = (df['amount'] * df['volume']) ** (1/4)
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * sqrt_log_high_low_diff + w3 * fourth_rt_amount_volume
    
    return heuristics_matrix
