import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the exponential moving average of the log returns with a span of 5 days
    ema_log_returns = log_returns.ewm(span=5, adjust=False).mean()
    
    # Calculate the price change normalized by volume
    normalized_price_change = (df['close'] - df['open']) / df['volume']
    
    # Calculate the coefficient of variation of the amount over a rolling window of 5 days
    cv_amount = df['amount'].rolling(window=5).std() / df['amount'].rolling(window=5).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.35, 0.25
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_log_returns + w2 * normalized_price_change + w3 * cv_amount
    
    return heuristics_matrix
