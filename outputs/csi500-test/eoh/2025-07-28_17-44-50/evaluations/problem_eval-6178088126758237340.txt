import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the cube root of the close-to-open price ratio
    cbrt_close_open_ratio = (df['close'] / df['open']) ** (1/3)
    
    # Calculate the absolute value of the natural logarithm difference between the high and low prices
    abs_log_high_low_diff = np.abs(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the product of trading amount and volume
    prod_amount_volume = df['amount'] * df['volume']
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * cbrt_close_open_ratio + w2 * abs_log_high_low_diff + w3 * prod_amount_volume
    
    return heuristics_matrix
