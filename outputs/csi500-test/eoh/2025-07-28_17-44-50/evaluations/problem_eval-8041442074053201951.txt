import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    # Compute the rolling mean of the price difference
    rolling_mean_price_diff = price_diff.rolling(window=10).mean()
    # Compute the rolling standard deviation of the volume
    rolling_std_volume = df['volume'].rolling(window=10).std()
    # Avoid division by zero
    rolling_std_volume[rolling_std_volume == 0] = 1e-6
    # Generate the heuristic score
    heuristics_matrix = rolling_mean_price_diff / rolling_std_volume
    return heuristics_matrix
