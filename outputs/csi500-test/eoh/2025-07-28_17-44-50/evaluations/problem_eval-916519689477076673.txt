import pandas as pd

def heuristics_v2(df):
    # Calculate the MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    
    # Calculate the MACD signal line
    signal_line = macd.ewm(span=9, adjust=False).mean()
    
    # Compute the MACD * signal line
    macd_signal_product = macd * signal_line
    
    # Calculate the RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 50-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    
    # Calculate the volume ratio
    volume_ratio = df['volume'] / avg_volume_50
    
    # Combine the components into a single alpha factor
    heuristics_matrix = (macd_signal_product * 0.4 + 
                         rsi * 0.3 + 
                         volume_ratio * 0.3)
    
    return heuristics_matrix
