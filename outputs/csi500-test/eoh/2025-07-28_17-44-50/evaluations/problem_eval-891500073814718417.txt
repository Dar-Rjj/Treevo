def heuristics_v2(df):
    # Calculate the averages
    avg_close_5d = df['close'].rolling(window=5).mean()
    avg_close_10d_prior = df['close'].shift(5).rolling(window=10).mean()
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    
    # Compute the momentum indicator
    momentum_indicator = (avg_close_5d - avg_close_10d_prior) / (df['amount'] / avg_volume_20d)
    
    heuristics_matrix = momentum_indicator
    return heuristics_matrix
