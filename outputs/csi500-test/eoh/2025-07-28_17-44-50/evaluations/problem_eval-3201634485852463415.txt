import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the natural logarithm of the high-low price spread
    log_high_low_spread = np.log(df['high'] - df['low'])
    
    # Calculate the cube of the square root of the trading amount-to-volume ratio
    cube_sqrt_amount_volume = np.power(np.sqrt(df['amount'] / df['volume']), 3)
    
    # Combine the components into a single alpha factor using the harmonic mean
    heuristics_matrix = 3 / (1/exp_close_open_ratio + 1/log_high_low_spread + 1/cube_sqrt_amount_volume)
    
    return heuristics_matrix
