def heuristics_v2(df):
    # Calculate the square of the log of the close-to-open price ratio
    square_log_close_open_ratio = np.square(np.log(df['close'] / df['open']))
    
    # Calculate the squared difference between the high and low prices
    square_high_low_diff = np.square(df['high'] - df['low'])
    
    # Calculate the natural log of the trading volume-to-amount ratio
    log_volume_amount_ratio = np.log(df['volume'] / df['amount'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * square_log_close_open_ratio + w2 * square_high_low_diff + w3 * log_volume_amount_ratio
    
    return heuristics_matrix
