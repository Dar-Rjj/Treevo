def heuristics_v2(df):
    df['close_50ma'] = df['close'].rolling(window=50).mean()
    df['amount_200ma'] = df['amount'].rolling(window=200).mean()
    df['momentum'] = (df['close'] - df['close_50ma']) * (df['amount'] / df['amount_200ma'])
    heuristics_matrix = df['momentum'].dropna()
    return heuristics_matrix
