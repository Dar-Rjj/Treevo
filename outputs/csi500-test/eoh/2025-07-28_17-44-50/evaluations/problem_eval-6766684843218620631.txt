import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential of the close-to-open price ratio
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    
    # Calculate the cubic root of the natural logarithm difference between the high and low prices
    cbrt_log_high_low_diff = (np.log(df['high']) - np.log(df['low'])) ** (1/3)
    
    # Calculate the inverse of the square root of the product of trading amount and volume
    inv_sqrt_amount_volume = 1 / ((df['amount'] * df['volume']) ** 0.5)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * cbrt_log_high_low_diff + w3 * inv_sqrt_amount_volume
    
    return heuristics_matrix
