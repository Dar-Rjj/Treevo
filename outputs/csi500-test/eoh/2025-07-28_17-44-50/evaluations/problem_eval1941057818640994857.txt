import pandas as pd
import numpy as np

def heuristics_v2(df):
    exp_close_open_ratio = np.exp(df['close'] / df['open'])
    cube_abs_log_high_low_diff = (np.abs(np.log(df['high']) - np.log(df['low']))) ** 3
    sqrt_amount_volume = (df['amount'] / df['volume']) ** (1/2)
    w1, w2, w3 = 0.4, 0.4, 0.2
    heuristics_matrix = w1 * exp_close_open_ratio + w2 * cube_abs_log_high_low_diff + w3 * sqrt_amount_volume
    return heuristics_matrix
