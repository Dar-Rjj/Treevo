def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the closing price
    df['roc_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day standard deviation of the volume
    df['volatility_20'] = df['volume'].rolling(window=20).std()
    
    # Composite factor: weighted sum of ROC and volume volatility
    df['composite_factor'] = (df['roc_10'] * 0.5 + 
                              df['volatility_20'] * 0.5)
    
    heuristics_matrix = df['composite_factor'].copy()
    
    return heuristics_matrix
