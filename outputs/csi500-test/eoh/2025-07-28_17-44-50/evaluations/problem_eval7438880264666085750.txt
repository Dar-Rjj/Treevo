import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the simple moving average of the close-to-open price ratio with a window of 5
    sma_close_open_ratio = (df['close'] / df['open']).rolling(window=5).mean()
    
    # Calculate the absolute value of the natural logarithm difference between the high and low prices
    abs_log_high_low_diff = np.abs(np.log(df['high']) - np.log(df['low']))
    
    # Calculate the cube root of the product of trading amount and volume
    cbrt_amount_volume = (df['amount'] * df['volume']) ** (1/3)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sma_close_open_ratio + w2 * abs_log_high_low_diff + w3 * cbrt_amount_volume
    
    return heuristics_matrix
