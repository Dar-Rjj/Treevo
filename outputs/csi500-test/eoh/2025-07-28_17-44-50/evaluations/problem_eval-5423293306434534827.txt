import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    returns = df['close'].pct_change()
    
    # Exponential Moving Average (EMA) of the return
    ema_returns = returns.ewm(span=20).mean()
    
    # EMA of the ratio between the highest and lowest prices
    high_low_ratio = df['high'] / df['low']
    ema_high_low_ratio = high_low_ratio.ewm(span=20).mean()
    
    # EMA of the trading amount
    ema_amount = df['amount'].ewm(span=20).mean()
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * ema_returns + w2 * ema_high_low_ratio + w3 * ema_amount
    
    return heuristics_matrix
