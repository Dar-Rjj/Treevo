def heuristics_v2(df):
    # Calculate the square root of the close-to-open price ratio
    sqrt_close_open_ratio = np.sqrt(df['close'] / df['open'])
    
    # Calculate the natural logarithm of the absolute difference between the high and low prices
    log_abs_high_low_diff = np.log(np.abs(df['high'] - df['low']))
    
    # Calculate the cube of the trading amount-to-volume ratio
    cube_amount_volume = (df['amount'] / df['volume']) ** 3
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.4, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sqrt_close_open_ratio + w2 * log_abs_high_low_diff + w3 * cube_amount_volume
    
    return heuristics_matrix
