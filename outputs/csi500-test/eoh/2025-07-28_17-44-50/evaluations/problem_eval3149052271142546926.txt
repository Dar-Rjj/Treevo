import pandas as pd

def heuristics_v2(df):
    # Momentum signal: 30-day change in closing price
    momentum = df['close'].pct_change(30)
    
    # Volatility: Average True Range (ATR) over the last 20 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=20).mean()
    
    # Volume-activity ratio: 15-day average volume divided by the 120-day average volume
    avg_vol_15d = df['volume'].rolling(window=15).mean()
    avg_vol_120d = df['volume'].rolling(window=120).mean()
    vol_activity_ratio = avg_vol_15d / avg_vol_120d
    
    # Combine the calculated features into a single factor
    heuristics_matrix = (momentum - atr) * vol_activity_ratio
    
    return heuristics_matrix
