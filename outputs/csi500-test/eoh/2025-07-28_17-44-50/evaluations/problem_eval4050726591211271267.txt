import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithm of the close-to-open price ratio
    log_close_open_ratio = np.log(df['close'] / df['open'])
    
    # Calculate the square root of the absolute difference between the high and low prices
    sqrt_high_low_diff = np.sqrt(np.abs(df['high'] - df['low']))
    
    # Calculate the trading volume scaled by the amount
    scaled_volume_amount = df['volume'] * (1 / df['amount'])
    
    # Define weights for each component
    w1, w2, w3 = 0.5, 0.3, 0.2
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * log_close_open_ratio + w2 * sqrt_high_low_diff + w3 * scaled_volume_amount
    
    return heuristics_matrix
