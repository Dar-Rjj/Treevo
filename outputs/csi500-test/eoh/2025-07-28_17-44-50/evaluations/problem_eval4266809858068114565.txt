import pandas as pd
import ta

def heuristics_v2(df):
    # Calculate 20-day EMA of the closing price
    ema_20 = ta.trend.ema_indicator(df['close'], window=20)
    
    # Calculate 14-day RSI
    rsi_14 = ta.momentum.rsi(df['close'], window=14)
    
    # Current closing price minus 20-day EMA
    close_minus_ema = df['close'] - ema_20
    
    # Combine the calculated features into a single factor
    heuristics_matrix = close_minus_ema * rsi_14
    
    return heuristics_matrix
