def heuristics_v2(df):
    # Calculate the 5-day simple moving average of the close price
    sma_close_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the logarithmic returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the 14-day standard deviation of the logarithmic returns
    std_log_returns_14d = log_returns.rolling(window=14).std()
    
    # Calculate the difference between the current day's high and low prices
    high_low_diff = df['high'] - df['low']
    
    # Define weights for each component
    w1, w2, w3 = 0.6, 0.3, 0.1
    
    # Combine the components into a single alpha factor
    heuristics_matrix = w1 * sma_close_5d + w2 * std_log_returns_14d + w3 * high_low_diff
    
    return heuristics_matrix
