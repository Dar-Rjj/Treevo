def heuristics_v2(df):
    # Calculate the close-to-open price ratio as a proxy for momentum
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the high-low price spread normalized by the opening price to capture intraday volatility
    high_low_spread_normalized = (df['high'] - df['low']) / df['open']
    
    # Calculate the signed log change in trading volume to identify significant changes in trading activity
    log_volume_change = np.log(df['volume']).diff().fillna(0)
    
    # Define weights for each component
    w1, w2, w3 = 0.4, 0.3, 0.3
    
    # Combine the components into a single alpha factor using a weighted sum
    heuristics_matrix = w1 * close_open_ratio + w2 * high_low_spread_normalized + w3 * log_volume_change
    
    return heuristics_matrix
