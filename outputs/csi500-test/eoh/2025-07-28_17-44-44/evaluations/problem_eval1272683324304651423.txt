def heuristics_v2(df):
    # Momentum signal: Smoothed Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(span=14).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14).mean()
    rs = gain / loss
    rsi_signal = 100 - (100 / (1 + rs))
    
    # Trend-following signal: Close price divided by 50-day Exponential Moving Average (EMA)
    ema_50 = df['close'].ewm(span=50).mean()
    trend_signal = df['close'] / ema_50
    
    # Liquidity signal: Normalized difference between today's volume and the 10-day mean volume
    mean_vol_10 = df['volume'].rolling(window=10).mean()
    std_vol_10 = df['volume'].rolling(window=10).std()
    liquidity_signal = (df['volume'] - mean_vol_10) / std_vol_10
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (rsi_signal + trend_signal + liquidity_signal) / 3
    
    return heuristics_matrix
