def heuristics_v2(df):
    # Calculate 20-day RSI of close prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi_20_close = 100 - (100 / (1 + rs))

    # Calculate 14-day EMA of the Stochastic Oscillator's %K
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_line = ((df['close'] - lowest_low) / (highest_high - lowest_low)) * 100
    ema_k_line = k_line.ewm(span=14, adjust=False).mean()

    # Calculate 7-day EMA of the ratio of high to low
    ratio_high_low = df['high'] / df['low']
    ema_ratio_high_low = ratio_high_low.ewm(span=7, adjust=False).mean()

    # Heuristic score calculation
    heuristics_matrix = (rsi_20_close / ema_k_line) - ema_ratio_high_low

    return heuristics_matrix
