def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_15 = (delta_close.where(delta_close > 0, 0)).rolling(window=15).mean()
    loss_15 = (-delta_close.where(delta_close < 0, 0)).rolling(window=15).mean()
    rsi_15 = 100 - (100 / (1 + gain_15 / loss_15))
    
    gain_30 = (delta_close.where(delta_close > 0, 0)).rolling(window=30).mean()
    loss_30 = (-delta_close.where(delta_close < 0, 0)).rolling(window=30).mean()
    rsi_30 = 100 - (100 / (1 + gain_30 / loss_30))
    
    rsi_ratio = rsi_15 / rsi_30
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    mean_5_close = df['close'].rolling(window=5).mean()
    
    heuristics_matrix = rsi_ratio - (ema_5_volume * mean_5_close)
    
    return heuristics_matrix
