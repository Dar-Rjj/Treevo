import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df):
    # Calculate 20-day and 50-day EMAs of close prices
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_50_close = df['close'].ewm(span=50, adjust=False).mean()

    # Prepare data for calculating the slope of the linear regression
    x = np.array([ema_20_close.values[-15:], ema_50_close.values[-15:]]).T
    y = np.arange(15)
    model = LinearRegression().fit(x, y)
    slope = model.coef_[0]

    # Calculate 30-day EMA of volume
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()

    # Calculate 30-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=30).mean()

    # Heuristic score calculation
    heuristics_matrix = pd.Series(slope - (ema_30_volume * atr ** (1/2)), index=df.index)

    return heuristics_matrix
