def heuristics_v2(df):
    df['momentum'] = df['close'].pct_change(20)
    tr = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x) - min(x), axis=1)
    df['volatility'] = tr.rolling(window=20).mean()
    df['liquidity'] = df['volume'] / (df['amount'] + 1e-6)
    heuristics_matrix = (df['momentum'] + df['volatility'] + df['liquidity']) / 3
    return heuristics_matrix
