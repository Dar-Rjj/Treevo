def heuristics_v2(df):
    std_5_close = df['close'].rolling(window=5).std()
    std_15_close = df['close'].rolling(window=15).std()
    
    std_ratio = std_5_close / std_15_close
    
    ema_7_volume = df['volume'].ewm(span=7, adjust=False).mean()
    mean_7_close = df['close'].rolling(window=7).mean()
    
    heuristics_matrix = std_ratio - (ema_7_volume * mean_7_close)
    
    return heuristics_matrix
