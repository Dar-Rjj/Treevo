import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 14-day and 30-day EMAs of close prices
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()

    # Compute the difference between the two EMAs
    ema_diff = ema_14_close - ema_30_close

    # Calculate 21-day EMA of amount
    ema_21_amount = df['amount'].ewm(span=21, adjust=False).mean()

    # Calculate 21-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()

    # Heuristic score calculation
    heuristics_matrix = pd.Series(ema_diff - (ema_21_amount * atr ** (1/2)), index=df.index)

    return heuristics_matrix
