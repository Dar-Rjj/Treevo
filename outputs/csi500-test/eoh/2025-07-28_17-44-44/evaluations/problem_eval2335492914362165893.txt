def heuristics_v2(df):
    std_20_close = df['close'].rolling(window=20).std()
    std_60_close = df['close'].rolling(window=60).std()
    std_ratio = std_20_close / std_60_close
    
    daily_returns = df['close'].pct_change()
    sma_30_returns = daily_returns.rolling(window=30).mean()
    
    heuristics_matrix = std_ratio + (sma_30_returns / df['volume'])
    
    return heuristics_matrix
