def heuristics_v2(df):
    # Calculate 50-day momentum (current close - close 50 days ago)
    momentum_50 = df['close'] - df['close'].shift(50)
    
    # Calculate 7-day minimum of the high prices
    min_high_7 = df['high'].rolling(window=7).min()
    
    # Calculate 14-day EMA of the volume
    ema_14_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Heuristic score calculation
    heuristics_matrix = momentum_50 + (min_high_7 / ema_14_volume)
    
    return heuristics_matrix
