import pandas as pd

def heuristics_v2(df):
    momentum_10 = df['close'] - df['close'].shift(10)
    amount_change = (df['amount'] - df['amount'].shift(5)) / df['amount'].shift(5)
    combined_factors = momentum_10 + amount_change
    heuristics_matrix = combined_factors.rolling(window=20).mean()
    return heuristics_matrix
