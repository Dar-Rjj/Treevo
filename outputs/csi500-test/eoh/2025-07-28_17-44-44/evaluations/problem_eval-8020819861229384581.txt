def heuristics_v2(df):
    # Calculate 14-day and 28-day RSIs of close prices
    delta_close = df['close'].diff()
    gain = (delta_close.where(delta_close > 0, 0)).rolling(window=14).mean()
    loss = (-delta_close.where(delta_close < 0, 0)).rolling(window=14).mean()
    rsi_14 = 100 - (100 / (1 + gain / loss))

    gain = (delta_close.where(delta_close > 0, 0)).rolling(window=28).mean()
    loss = (-delta_close.where(delta_close < 0, 0)).rolling(window=28).mean()
    rsi_28 = 100 - (100 / (1 + gain / loss))

    # Ratio of 14-day to 28-day RSIs
    rsi_ratio = rsi_14 / rsi_28

    # Calculate 5-day EMA of volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate 5-day standard deviation of close prices
    std_5_close = df['close'].rolling(window=5).std()

    # Heuristic score calculation
    heuristics_matrix = rsi_ratio - (ema_5_volume * std_5_close)

    return heuristics_matrix
