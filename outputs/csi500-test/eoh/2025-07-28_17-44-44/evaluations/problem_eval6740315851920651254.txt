import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Exponential Moving Average (EMA) Crossover: EMA(10) - EMA(30)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_crossover = ema_10 - ema_30
    
    # Ratio between the highest and lowest volume in the last 40 days
    vol_ratio = df['volume'].rolling(window=40).max() / df['volume'].rolling(window=40).min()
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Average True Range (ATR) over 20 days
    atr_20 = true_range.rolling(window=20).mean()
    
    # Volatility-adjusted momentum: difference between high and low adjusted by ATR
    vol_adj_momentum = (df['high'] - df['low']) / atr_20
    
    # Composite alpha factor
    heuristics_matrix = ema_crossover + vol_ratio + vol_adj_momentum
    
    return heuristics_matrix
