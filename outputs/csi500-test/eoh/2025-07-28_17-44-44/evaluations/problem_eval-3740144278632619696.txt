def heuristics_v2(df):
    df['momentum'] = df['close'].pct_change(10)
    df['volatility'] = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x) - min(x), axis=1).rolling(window=10).mean()
    df['liquidity'] = df['volume'] / (df['amount'] + 1e-6)
    heuristics_matrix = (2 * df['momentum'] + df['volatility'] + 3 * df['liquidity']) / 6
    return heuristics_matrix
