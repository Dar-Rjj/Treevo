def heuristics_v2(df):
    # Measure of price trend strength (average of the last 7 closing prices divided by the first)
    trend_strength = df['close'].rolling(window=7).mean() / df['close'].shift(6)
    
    # Average trading volume over the past 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Percentage change in the lowest price over the last 20 days
    low_change = (df['low'] - df['low'].shift(20)) / df['low'].shift(20)
    
    # Composite alpha factor
    heuristics_matrix = (trend_strength + avg_volume + low_change) / 3
    
    return heuristics_matrix
