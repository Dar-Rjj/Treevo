def heuristics_v2(df):
    # Calculate 30-day and 60-day EMAs of close prices
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()
    ema_60_close = df['close'].ewm(span=60, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_30_close - ema_60_close

    # Calculate 21-day EMA of volume
    ema_21_volume = df['volume'].ewm(span=21, adjust=False).mean()

    # Calculate 21-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close - (ema_21_volume * atr**0.5)

    return heuristics_matrix
