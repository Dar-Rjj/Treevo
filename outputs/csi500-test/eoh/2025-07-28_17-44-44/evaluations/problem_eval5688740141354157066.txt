def heuristics_v2(df):
    close_std_20 = df['close'].rolling(window=20).std()
    close_std_60 = df['close'].rolling(window=60).std()
    volume_ratio = df['volume'] / df['volume'].rolling(window=30).mean()
    weighted_sum_std = (0.6 * close_std_20 + 0.4 * close_std_60)
    heuristics_matrix = weighted_sum_std * volume_ratio
    return heuristics_matrix
