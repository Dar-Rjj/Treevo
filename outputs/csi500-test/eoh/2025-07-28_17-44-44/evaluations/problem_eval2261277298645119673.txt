import pandas as pd

def heuristics_v2(df):
    weights = list(range(1, 61))
    wma_60_close = df['close'].rolling(window=60).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
    log_volume_sum_15 = (df['volume'].rolling(window=15).sum()).apply(lambda x: pd.np.log(x) if x > 0 else 0)
    heuristics_matrix = wma_60_close - log_volume_sum_15
    return heuristics_matrix
