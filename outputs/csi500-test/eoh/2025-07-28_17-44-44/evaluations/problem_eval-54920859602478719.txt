def heuristics_v2(df):
    # Calculate 10-day and 30-day SMAs of close prices
    sma_10_close = df['close'].rolling(window=10).mean()
    sma_30_close = df['close'].rolling(window=30).mean()

    # Calculate the difference between 10-day and 30-day SMAs
    sma_diff = sma_10_close - sma_30_close

    # Calculate 14-day EMA of volume
    ema_14_volume = df['volume'].ewm(span=14, adjust=False).mean()

    # Calculate 21-day ATR
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()

    # Heuristic score calculation
    heuristics_matrix = pd.Series(sma_diff + (ema_14_volume * np.sqrt(atr)), index=df.index)

    return heuristics_matrix
