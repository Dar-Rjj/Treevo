import pandas as pd

def heuristics_v2(df):
    # Momentum of closing price over 50 days
    momentum_50 = df['close'] - df['close'].shift(50)
    
    # Calculate 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Ratio of current volume to 20-day moving average of volume
    vol_ratio_20 = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Composite alpha factor
    heuristics_matrix = momentum_50 + rsi + vol_ratio_20
    
    return heuristics_matrix
