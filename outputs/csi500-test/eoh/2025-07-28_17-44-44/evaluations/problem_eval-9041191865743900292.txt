def heuristics_v2(df):
    # Momentum signal: 10-day rate of change between high and low prices
    momentum_signal = (df['high'] - df['low']) / df['low'].shift(10)
    
    # Oscillator signal: difference between the closing price and its 20-day SMA
    sma_20_close = df['close'].rolling(window=20).mean()
    oscillator_signal = df['close'] - sma_20_close
    
    # Activity measure: difference between current and 30-day SMA of the trading amount
    sma_30_amount = df['amount'].rolling(window=30).mean()
    activity_measure = df['amount'] - sma_30_amount
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (momentum_signal + oscillator_signal + activity_measure) / 3

    return heuristics_matrix
