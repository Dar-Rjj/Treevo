import pandas as pd

def heuristics_v2(df):
    # Rate of change in closing prices over 14 days
    roc_close = df['close'].pct_change(periods=14)
    
    # Ratio between the highest and lowest volume in the last 30 days
    vol_ratio = df['volume'].rolling(window=30).max() / df['volume'].rolling(window=30).min()
    
    # Volatility-adjusted momentum: difference between high and low adjusted by standard deviation of returns over 20 days
    daily_returns = df['close'].pct_change()
    std_dev_returns = daily_returns.rolling(window=20).std()
    vol_adj_momentum = (df['high'] - df['low']) / std_dev_returns
    
    # Composite alpha factor
    heuristics_matrix = roc_close + vol_ratio + vol_adj_momentum
    
    return heuristics_matrix
