def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_50 = (delta_close.where(delta_close > 0, 0)).rolling(window=50).mean()
    loss_50 = (-delta_close.where(delta_close < 0, 0)).rolling(window=50).mean()
    rsi_50 = 100 - (100 / (1 + gain_50 / loss_50))
    
    gain_100 = (delta_close.where(delta_close > 0, 0)).rolling(window=100).mean()
    loss_100 = (-delta_close.where(delta_close < 0, 0)).rolling(window=100).mean()
    rsi_100 = 100 - (100 / (1 + gain_100 / loss_100))
    
    rsi_ratio = rsi_50 / rsi_100
    
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    mean_20_close = df['close'].rolling(window=20).mean()
    
    heuristics_matrix = rsi_ratio - (ema_20_volume * mean_20_close)
    
    return heuristics_matrix
