def heuristics_v2(df):
    # Calculate 14-day and 28-day EMAs of close prices
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_28_close = df['close'].ewm(span=28, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_14_close - ema_28_close

    # Calculate 5-day EMA of volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate 5-day standard deviation of close prices
    std_5_close = df['close'].rolling(window=5).std()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close - (ema_5_volume * std_5_close)

    return heuristics_matrix
