def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_14 = (delta_close.where(delta_close > 0, 0)).rolling(window=14).mean()
    loss_14 = (-delta_close.where(delta_close < 0, 0)).rolling(window=14).mean()
    rsi_14 = 100 - (100 / (1 + gain_14 / loss_14))
    
    gain_28 = (delta_close.where(delta_close > 0, 0)).rolling(window=28).mean()
    loss_28 = (-delta_close.where(delta_close < 0, 0)).rolling(window=28).mean()
    rsi_28 = 100 - (100 / (1 + gain_28 / loss_28))
    
    rsi_diff = rsi_14 - rsi_28
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    std_5_close = df['close'].rolling(window=5).std()
    
    heuristics_matrix = rsi_diff - (ema_5_volume * std_5_close)
    
    return heuristics_matrix
