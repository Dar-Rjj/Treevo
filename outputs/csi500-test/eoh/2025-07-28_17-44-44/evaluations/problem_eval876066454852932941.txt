import pandas as pd

def heuristics_v2(df):
    price_ratio = df['close'] / ((df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2)
    volume_change = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    combined_factors = price_ratio * volume_change
    heuristics_matrix = combined_factors.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
