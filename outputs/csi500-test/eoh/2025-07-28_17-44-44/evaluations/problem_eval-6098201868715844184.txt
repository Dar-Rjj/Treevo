def heuristics_v2(df):
    # Calculate 20-day and 50-day EMAs of close prices
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_50_close = df['close'].ewm(span=50, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_20_close - ema_50_close

    # Calculate 14-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_returns = daily_returns.rolling(window=14).std()

    # Calculate the volatility-adjusted volume
    adjusted_volume = df['volume'] * std_dev_returns

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close + adjusted_volume

    return heuristics_matrix
