import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate 14-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Calculate the ratio of daily returns to 14-day ATR
    ratio = daily_returns / atr
    
    # Calculate the 30-day moving average of the ratio
    heuristics_matrix = ratio.rolling(window=30).mean()
    
    return heuristics_matrix
