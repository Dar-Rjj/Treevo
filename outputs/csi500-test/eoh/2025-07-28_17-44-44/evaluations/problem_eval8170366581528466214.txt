def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative strength compared to the previous day
    df['rel_strength'] = (df['close'] - df[['open', 'high', 'low']].min(axis=1)) / (df[['open', 'high', 'low']].max(axis=1) - df[['open', 'high', 'low']].min(axis=1))
    
    # Calculate the volatility
    df['volatility'] = df['high'] - df['low']
    
    # Combine the above factors into a heuristic score
    df['heuristic_score'] = df['daily_return'] + df['rel_strength'] * df['volatility'] / df['volume']
    
    # Drop rows with NaN values generated by pct_change
    df.dropna(inplace=True)
    
    # Return the factor values as a Series
    heuristics_matrix = df['heuristic_score']
    return heuristics_matrix
