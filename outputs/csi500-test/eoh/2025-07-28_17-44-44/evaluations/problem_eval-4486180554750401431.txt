def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_20 = (delta_close.where(delta_close > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta_close.where(delta_close < 0, 0)).rolling(window=20).mean()
    rsi_20 = 100 - (100 / (1 + gain_20 / loss_20))
    
    gain_50 = (delta_close.where(delta_close > 0, 0)).rolling(window=50).mean()
    loss_50 = (-delta_close.where(delta_close < 0, 0)).rolling(window=50).mean()
    rsi_50 = 100 - (100 / (1 + gain_50 / loss_50))
    
    rsi_ratio = rsi_20 / rsi_50
    
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    mean_10_close = df['close'].rolling(window=10).mean()
    
    heuristics_matrix = rsi_ratio - (ema_10_volume * mean_10_close)
    
    return heuristics_matrix
