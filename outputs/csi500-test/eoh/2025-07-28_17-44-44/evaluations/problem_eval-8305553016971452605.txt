import pandas as pd

def heuristics_v2(df):
    avg_price = (df['high'].rolling(window=10).mean() + df['low'].rolling(window=10).mean()) / 2
    close_change = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    heuristics_matrix = avg_price * close_change * volume_ratio
    return heuristics_matrix
