def heuristics_v2(df):
    # Measure of price trend strength (average of the last 5 closing prices divided by the first)
    trend_strength = df['close'].rolling(window=5).mean() / df['close'].shift(4)
    
    # Average trading volume over the past 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Percentage change in the highest price over the last 15 days
    high_change = (df['high'] - df['high'].shift(15)) / df['high'].shift(15)
    
    # Composite alpha factor
    heuristics_matrix = (trend_strength + avg_volume + high_change) / 3
    
    return heuristics_matrix
