def heuristics_v2(df):
    # Calculate 20-day price momentum
    momentum_20 = df['close'] - df['close'].shift(20)
    
    # Calculate 14-day stochastic oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_14 = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Calculate 10-day EMA of volume
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate 30-day standard deviation of closing prices
    std_30_close = df['close'].rolling(window=30).std()
    
    # Heuristic score calculation
    heuristics_matrix = momentum_20 * k_14 - (ema_10_volume * std_30_close)
    
    return heuristics_matrix
