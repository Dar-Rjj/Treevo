def heuristics_v2(df):
    # Trend-following signal: difference between 5-day and 20-day moving average of close prices
    trend_signal = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Mean reversion signal: 10-day rate of change
    mean_reversion_signal = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volume surprise signal: difference between today's volume and 30-day average volume
    volume_surprise_signal = df['volume'] - df['volume'].rolling(window=30).mean()
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (trend_signal + mean_reversion_signal + volume_surprise_signal) / 3
    
    return heuristics_matrix
