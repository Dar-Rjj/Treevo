def heuristics_v2(df):
    # Momentum signal: Relative Strength Index (RSI) with a 20-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi_signal = 100 - (100 / (1 + rs))
    
    # Trend-following signal: Close price divided by 100-day SMA
    sma_100 = df['close'].rolling(window=100).mean()
    trend_signal = df['close'] / sma_100
    
    # Liquidity signal: Today's volume minus the 30-day average volume
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    liquidity_signal = df['volume'] - avg_vol_30
    
    # Composite alpha factor as a weighted sum of the above signals
    heuristics_matrix = 0.4 * rsi_signal + 0.4 * trend_signal + 0.2 * liquidity_signal
    
    return heuristics_matrix
