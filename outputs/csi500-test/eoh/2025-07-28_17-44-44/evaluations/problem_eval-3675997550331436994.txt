def heuristics_v2(df):
    # Calculate 10-day EMA of close prices
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 30-day EMA of volume
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Sum of EMAs
    sum_emas = ema_10_close + ema_30_volume
    
    # Calculate 14-day standard deviation of high prices
    std_high = df['high'].rolling(window=14).std()
    
    # Heuristic score calculation
    heuristics_matrix = sum_emas / std_high

    return heuristics_matrix
