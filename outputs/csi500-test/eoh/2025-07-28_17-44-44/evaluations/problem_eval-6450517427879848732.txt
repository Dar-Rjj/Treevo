def heuristics_v2(df):
    # Momentum signal: Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_signal = 100 - (100 / (1 + rs))
    
    # Trend-following signal: Close price divided by 50-day SMA
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = df['close'] / sma_50
    
    # Liquidity signal: Today's volume minus the 10-day minimum volume
    min_vol_10 = df['volume'].rolling(window=10).min()
    liquidity_signal = df['volume'] - min_vol_10
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (rsi_signal + trend_signal + liquidity_signal) / 3
    
    return heuristics_matrix
