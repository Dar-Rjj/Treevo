import pandas as pd

def heuristics_v2(df):
    close_roc_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    close_roc_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    close_roc_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    heuristics_matrix = 0.5 * close_roc_5 + 0.3 * close_roc_10 + 0.2 * close_roc_20
    return heuristics_matrix
