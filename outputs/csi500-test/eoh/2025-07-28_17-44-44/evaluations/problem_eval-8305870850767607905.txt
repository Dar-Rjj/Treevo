def heuristics_v2(df):
    # Calculate the daily closing price changes
    close_diff = df['close'].diff()
    
    # Calculate the 10-day weighted average of closing price changes with volume as weights
    weighted_avg_close_change = (close_diff * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the high-low price range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Adjust the weighted average closing change by the 5-day average high-low range
    heuristics_matrix = weighted_avg_close_change * avg_high_low_range
    
    return heuristics_matrix
