def heuristics_v2(df):
    close_prices = df['close']
    momentum_10 = (close_prices / close_prices.shift(10)) - 1
    momentum_30 = (close_prices / close_prices.shift(30)) - 1
    momentum_ratio = momentum_10 / momentum_30
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    mean_5_high_low = df[['high', 'low']].rolling(window=5).mean().mean(axis=1)
    
    heuristics_matrix = momentum_ratio - (ema_5_volume * mean_5_high_low)
    
    return heuristics_matrix
