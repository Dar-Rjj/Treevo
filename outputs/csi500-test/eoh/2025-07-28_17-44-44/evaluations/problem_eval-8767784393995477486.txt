def heuristics_v2(df):
    # Calculate 10-day momentum
    momentum_10 = df['close'].pct_change(periods=10)
    
    # Calculate 5-day ATR
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_5 = tr.rolling(window=5).mean()
    
    # Calculate 3-day EMA of the volume
    ema_3_volume = df['volume'].ewm(span=3, adjust=False).mean()
    
    # Combine the terms to form the heuristics matrix
    heuristics_matrix = momentum_10 - (atr_5 * ema_3_volume)
    
    return heuristics_matrix
