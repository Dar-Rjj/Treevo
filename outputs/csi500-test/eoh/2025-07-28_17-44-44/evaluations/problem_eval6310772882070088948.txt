def heuristics_v2(df):
    # Calculate 14-day Rate of Change (ROC) of close prices
    roc_14_close = df['close'].pct_change(periods=14)
    
    # Calculate 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(com=13, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(com=13, adjust=False).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Adjust ROC by RSI
    adjusted_roc = roc_14_close * (rsi_14 / 50)
    
    # Calculate 7-day EMA of volume
    ema_7_volume = df['volume'].ewm(span=7, adjust=False).mean()
    
    # Calculate 14-day standard deviation of close prices
    std_14_close = df['close'].rolling(window=14).std()
    
    # Heuristic score calculation
    heuristics_matrix = adjusted_roc - (ema_7_volume * std_14_close)

    return heuristics_matrix
