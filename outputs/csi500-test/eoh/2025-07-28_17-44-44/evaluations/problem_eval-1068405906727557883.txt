def heuristics_v2(df):
    # Momentum signal: Rate of Change (ROC) with a 14-day window
    roc_signal = df['close'].pct_change(periods=14)
    
    # Trend-following signal: Close price divided by 50-day EMA
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    trend_signal = df['close'] / ema_50
    
    # Liquidity signal: Today's volume minus the 10-day average volume
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    liquidity_signal = df['volume'] - avg_vol_10
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (roc_signal + trend_signal + liquidity_signal) / 3
    
    return heuristics_matrix
