def heuristics_v2(df):
    ma_30 = df['close'].rolling(window=30).mean()
    ma_60 = df['close'].rolling(window=60).mean()
    ma_ratio = ma_30 / ma_60

    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    std_20_close = df['close'].rolling(window=20).std()

    heuristics_matrix = ma_ratio - (ema_10_volume * std_20_close)
    
    return heuristics_matrix
