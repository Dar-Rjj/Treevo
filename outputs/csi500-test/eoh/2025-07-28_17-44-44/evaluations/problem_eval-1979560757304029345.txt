def heuristics_v2(df):
    delta_close = df['close'].diff()
    
    gain_5 = (delta_close.where(delta_close > 0, 0)).rolling(window=5).mean()
    loss_5 = (-delta_close.where(delta_close < 0, 0)).rolling(window=5).mean()
    rsi_5 = 100 - (100 / (1 + gain_5 / loss_5))
    
    gain_15 = (delta_close.where(delta_close > 0, 0)).rolling(window=15).mean()
    loss_15 = (-delta_close.where(delta_close < 0, 0)).rolling(window=15).mean()
    rsi_15 = 100 - (100 / (1 + gain_15 / loss_15))
    
    rsi_ratio = rsi_5 / rsi_15
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    std_5_close = df['close'].rolling(window=5).std()
    
    heuristics_matrix = rsi_ratio - (ema_5_volume * std_5_close)
    
    return heuristics_matrix
