def heuristics_v2(df):
    avg_close_10 = df['close'].rolling(window=10).mean()
    high_low_ratio = df['high'].rolling(window=30).max() / df['low'].rolling(window=30).min()
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    vol_diff = df['volume'] - avg_vol_20
    heuristics_matrix = (avg_close_10 + high_low_ratio + vol_diff) / 3
    return heuristics_matrix
