def heuristics_v2(df):
    # Calculate 50-day and 200-day moving averages of close prices
    ma_50_close = df['close'].rolling(window=50).mean()
    ma_200_close = df['close'].rolling(window=200).mean()

    # Difference between moving averages of close prices
    diff_ma_close = ma_50_close - ma_200_close

    # Calculate 30-day standard deviation of volume
    std_30_volume = df['volume'].rolling(window=30).std()

    # Heuristic score calculation
    heuristics_matrix = diff_ma_close / std_30_volume

    return heuristics_matrix
