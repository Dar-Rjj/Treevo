def heuristics_v2(df):
    # Momentum signal: Rate of Change (ROC) over 20 days
    roc_signal = df['close'].pct_change(periods=20)
    
    # Volatility signal: Standard deviation of daily returns over 30 days
    daily_returns = df['close'].pct_change()
    volatility_signal = daily_returns.rolling(window=30).std()
    
    # Liquidity signal: Today's volume divided by the 10-day average volume
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    liquidity_signal = df['volume'] / avg_vol_10
    
    # Composite alpha factor as the simple average of the above signals
    heuristics_matrix = (roc_signal + volatility_signal + liquidity_signal) / 3
    
    return heuristics_matrix
