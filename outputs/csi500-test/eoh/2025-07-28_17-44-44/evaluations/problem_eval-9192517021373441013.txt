def heuristics_v2(df):
    std_10_close = df['close'].rolling(window=10).std()
    std_20_close = df['close'].rolling(window=20).std()
    
    std_ratio = std_10_close / std_20_close
    
    ema_14_volume = df['volume'].ewm(span=14, adjust=False).mean()
    mean_14_close = df['close'].rolling(window=14).mean()
    
    heuristics_matrix = std_ratio - (ema_14_volume * mean_14_close)
    
    return heuristics_matrix
