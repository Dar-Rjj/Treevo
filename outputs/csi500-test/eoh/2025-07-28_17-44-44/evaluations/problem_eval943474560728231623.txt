def heuristics_v2(df):
    # Calculate 10-day and 25-day EMAs of close prices
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_25_close = df['close'].ewm(span=25, adjust=False).mean()

    # Difference of 10-day and 25-day EMAs of close prices
    diff_ema_close = ema_10_close - ema_25_close

    # Calculate 28-day EMA of volume
    ema_28_volume = df['volume'].ewm(span=28, adjust=False).mean()

    # Calculate 28-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=28).std()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close + (ema_28_volume * atr)

    return heuristics_matrix
