def heuristics_v2(df):
    # 5-day moving average of closing prices
    ma_close_5 = df['close'].rolling(window=5).mean()
    
    # Logarithmic difference between the highest and lowest prices over a 20-day period
    log_diff_high_low = (df['high'].rolling(window=20).max() / df['low'].rolling(window=20).min()).apply(np.log)
    
    # Accumulation Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl_ma_10 = adl.rolling(window=10).mean()
    adl_adjusted = adl - adl_ma_10
    
    # Composite alpha factor
    heuristics_matrix = ma_close_5 + log_diff_high_low + adl_adjusted
    
    return heuristics_matrix
