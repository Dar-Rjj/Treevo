def heuristics_v2(df):
    # Calculate 14-day and 28-day EMAs of close prices
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_28_close = df['close'].ewm(span=28, adjust=False).mean()

    # Difference of 14-day and 28-day EMAs of close prices
    diff_ema_close = ema_14_close - ema_28_close

    # Calculate 21-day EMA of volume
    ema_21_volume = df['volume'].ewm(span=21, adjust=False).mean()

    # Calculate 21-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()  # Use mean instead of std here

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close + (ema_21_volume * atr)

    return heuristics_matrix
