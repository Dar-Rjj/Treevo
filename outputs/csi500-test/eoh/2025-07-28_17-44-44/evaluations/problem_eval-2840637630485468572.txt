import pandas as pd

def heuristics_v2(df):
    close_roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    avg_volume = df['volume'].rolling(window=10).mean()
    combined_factors = close_roc * avg_volume
    heuristics_matrix = combined_factors.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
