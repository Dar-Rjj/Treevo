def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_20 = (delta_close.where(delta_close > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta_close.where(delta_close < 0, 0)).rolling(window=20).mean()
    rsi_20 = 100 - (100 / (1 + gain_20 / loss_20))
    
    gain_40 = (delta_close.where(delta_close > 0, 0)).rolling(window=40).mean()
    loss_40 = (-delta_close.where(delta_close < 0, 0)).rolling(window=40).mean()
    rsi_40 = 100 - (100 / (1 + gain_40 / loss_40))
    
    rsi_ratio = rsi_20 / rsi_40
    
    ema_15_volume = df['volume'].ewm(span=15, adjust=False).mean()
    median_15_close = df['close'].rolling(window=15).median()
    
    heuristics_matrix = rsi_ratio - (ema_15_volume * median_15_close)
    
    return heuristics_matrix
