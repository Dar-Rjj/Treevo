import pandas as pd

def heuristics_v2(df):
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    atr_10 = df['close'].rolling(window=10).apply(lambda x: (x.max() - x.min()) / 10, raw=True)
    heuristics_matrix = (sma_20 - sma_50) / atr_10
    return heuristics_matrix
