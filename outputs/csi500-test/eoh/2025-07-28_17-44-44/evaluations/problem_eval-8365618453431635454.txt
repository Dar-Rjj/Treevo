import pandas as pd

def heuristics_v2(df):
    price_ratio = df['close'] / ((df['high'].rolling(window=15).median() + df['low'].rolling(window=15).median()) / 2)
    volume_change_log = (df['volume'] - df['volume'].shift(7)) / df['volume'].shift(7)
    combined_factors = price_ratio * volume_change_log.apply(lambda x: pd.np.log(1 + x))
    heuristics_matrix = combined_factors.ewm(span=12, adjust=False).mean()
    return heuristics_matrix
