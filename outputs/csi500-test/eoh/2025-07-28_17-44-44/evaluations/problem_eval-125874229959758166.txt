def heuristics_v2(df):
    # Calculate 20-day return
    df['return_20'] = df['close'].pct_change(20)
    
    # Calculate z-score of 20-day return
    z_score_20 = (df['return_20'] - df['return_20'].mean()) / df['return_20'].std()
    
    # Calculate 14-day RSI
    delta_close = df['close'].diff()
    gain_14 = (delta_close.where(delta_close > 0, 0)).rolling(window=14).mean()
    loss_14 = (-delta_close.where(delta_close < 0, 0)).rolling(window=14).mean()
    rsi_14 = 100 - (100 / (1 + gain_14 / loss_14))
    
    # Calculate standard deviation of 14-day RSI
    std_rsi_14 = rsi_14.rolling(window=14).std()
    
    # Calculate 60-day EMA of volume
    ema_60_volume = df['volume'].ewm(span=60, adjust=False).mean()
    
    # Heuristic score calculation
    heuristics_matrix = z_score_20 + (std_rsi_14 * ema_60_volume)
    
    return heuristics_matrix
