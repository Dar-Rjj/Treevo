import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df):
    # Calculate 14-day and 30-day EMAs of close prices
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()

    # Prepare data for calculating the slope of the linear regression
    x = np.array([ema_14_close.values[-7:], ema_30_close.values[-7:]]).T
    y = np.arange(7)
    model = LinearRegression().fit(x, y)
    slope = model.coef_[0]

    # Calculate 21-day EMA of volume
    ema_21_volume = df['volume'].ewm(span=21, adjust=False).mean()

    # Calculate 21-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()

    # Heuristic score calculation
    heuristics_matrix = pd.Series(slope - (ema_21_volume * np.sqrt(atr)), index=df.index)

    return heuristics_matrix
