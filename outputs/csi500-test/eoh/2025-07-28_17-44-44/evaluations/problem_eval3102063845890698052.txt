def heuristics_v2(df):
    # Calculate 10-day and 30-day standard deviations of close prices
    std_10_close = df['close'].rolling(window=10).std()
    std_30_close = df['close'].rolling(window=30).std()

    # Difference between standard deviations of close prices
    diff_std_close = std_10_close - std_30_close

    # Calculate 5-day EMA of volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate 5-day median of close prices
    median_5_close = df['close'].rolling(window=5).median()

    # Heuristic score calculation
    heuristics_matrix = diff_std_close - (ema_5_volume * median_5_close)

    return heuristics_matrix
