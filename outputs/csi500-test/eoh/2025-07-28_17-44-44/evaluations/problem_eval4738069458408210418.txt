def heuristics_v2(df):
    # Calculate 50-day and 100-day EMAs of close prices
    ema_50_close = df['close'].ewm(span=50, adjust=False).mean()
    ema_100_close = df['close'].ewm(span=100, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_100_close - ema_50_close

    # Calculate 30-day EMA of volume
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()

    # Calculate 30-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=30).mean()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close + (ema_30_volume * atr)

    return heuristics_matrix
