import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Exponential Moving Average (EMA) Crossover: EMA(12) - EMA(26)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ema_crossover = ema_12 - ema_26
    
    # Ratio between the highest and lowest volume in the last 30 days
    vol_ratio = df['volume'].rolling(window=30).max() / df['volume'].rolling(window=30).min()
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Average True Range (ATR) over 14 days
    atr_14 = true_range.rolling(window=14).mean()
    
    # Volatility-adjusted momentum: difference between high and low adjusted by ATR
    vol_adj_momentum = (df['high'] - df['low']) / atr_14
    
    # Composite alpha factor
    heuristics_matrix = ema_crossover + vol_ratio + vol_adj_momentum
    
    return heuristics_matrix
