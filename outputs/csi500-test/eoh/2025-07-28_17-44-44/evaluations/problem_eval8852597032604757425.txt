import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day simple moving averages for closing prices
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the 5-day and 20-day moving averages
    ma_diff = sma_5 - sma_20
    
    # Compute the daily price change
    price_change = df['close'] - df['open']
    
    # Calculate the price-to-volume ratio
    price_to_volume = df['close'] / df['volume']
    
    # Calculate the 10-day standard deviation of closing prices for volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Combine heuristics into a single interpretable factor
    heuristics_matrix = (ma_diff + price_change + price_to_volume) / volatility
    
    return heuristics_matrix
