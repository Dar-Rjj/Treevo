def heuristics_v2(df):
    # Calculate 13-day and 26-day EMAs of close prices
    ema_13_close = df['close'].ewm(span=13, adjust=False).mean()
    ema_26_close = df['close'].ewm(span=26, adjust=False).mean()

    # Momentum component: difference between 13-day and 26-day EMAs of close prices
    momentum = ema_13_close - ema_26_close

    # Volatility component: 10-day standard deviation of close prices
    std_10_close = df['close'].rolling(window=10).std()

    # Volume adjustment: 5-day EMA of volume divided by 20-day EMA of volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    volume_adjustment = ema_5_volume / ema_20_volume

    # Heuristic score calculation
    heuristics_matrix = (momentum * std_10_close) - volume_adjustment

    return heuristics_matrix
