def heuristics_v2(df):
    # Calculate 10-day and 20-day ATRs
    tr = df['high'] - df['low']
    atr_10 = tr.rolling(window=10).mean()
    atr_20 = tr.rolling(window=20).mean()

    # Ratio of 10-day to 20-day ATRs
    atr_ratio = atr_10 / atr_20

    # Calculate 5-day SMA of close prices
    sma_5_close = df['close'].rolling(window=5).mean()

    # Calculate 5-day standard deviation of volume
    std_5_volume = df['volume'].rolling(window=5).std()

    # Heuristic score calculation
    heuristics_matrix = atr_ratio - (sma_5_close * std_5_volume)

    return heuristics_matrix
