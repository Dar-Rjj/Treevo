def heuristics_v2(df):
    momentum_10 = df['close'].pct_change(periods=10)
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    std_10_close = df['close'].rolling(window=10).std()
    
    heuristics_matrix = momentum_10 - (ema_5_volume * std_10_close)
    
    return heuristics_matrix
