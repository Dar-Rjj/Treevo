import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day and 30-day SMAs of close prices
    sma_10_close = df['close'].rolling(window=10).mean()
    sma_30_close = df['close'].rolling(window=30).mean()

    # Ratio between 10-day and 30-day SMAs of close prices
    ratio_sma_close = sma_10_close / sma_30_close

    # Calculate 5-day Rate of Change (ROC) of volume
    roc_5_volume = df['volume'].pct_change(periods=5)

    # Heuristic score calculation
    heuristics_matrix = ratio_sma_close * roc_5_volume

    return heuristics_matrix
