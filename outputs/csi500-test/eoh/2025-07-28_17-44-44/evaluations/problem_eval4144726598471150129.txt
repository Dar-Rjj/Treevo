def heuristics_v2(df):
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_60_close = df['close'].ewm(span=60, adjust=False).mean()
    
    macd_20_60 = ema_20_close - ema_60_close
    
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    median_10_high_low = (df['high'] + df['low']).rolling(window=10).median() / 2
    
    heuristics_matrix = macd_20_60 - (ema_10_volume * median_10_high_low)
    
    return heuristics_matrix
