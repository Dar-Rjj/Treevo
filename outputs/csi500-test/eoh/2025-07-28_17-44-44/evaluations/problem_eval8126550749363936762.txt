def heuristics_v2(df):
    # Calculate 20-day and 60-day EMAs of close prices
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_60_close = df['close'].ewm(span=60, adjust=False).mean()

    # Ratio between EMAs of close prices
    ratio_ema_close = ema_20_close / ema_60_close

    # Calculate 30-day EMA of amount
    ema_30_amount = df['amount'].ewm(span=30, adjust=False).mean()

    # Calculate 10-day sum of volume
    sum_10_volume = df['volume'].rolling(window=10).sum()

    # Heuristic score calculation
    heuristics_matrix = ratio_ema_close + (ema_30_amount - sum_10_volume)

    return heuristics_matrix
