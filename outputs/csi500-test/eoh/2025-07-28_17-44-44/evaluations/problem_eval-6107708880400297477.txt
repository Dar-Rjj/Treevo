def heuristics_v2(df):
    # Calculate 7-day and 21-day EMAs of close prices
    ema_7_close = df['close'].ewm(span=7, adjust=False).mean()
    ema_21_close = df['close'].ewm(span=21, adjust=False).mean()

    # Difference between 7-day and 21-day EMAs of close prices
    diff_ema_close = ema_7_close - ema_21_close

    # Calculate 14-day EMA of amount
    ema_14_amount = df['amount'].ewm(span=14, adjust=False).mean()

    # Calculate 14-day standard deviation of close prices
    std_14_close = df['close'].rolling(window=14).std()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close - (ema_14_amount * std_14_close)

    return heuristics_matrix
