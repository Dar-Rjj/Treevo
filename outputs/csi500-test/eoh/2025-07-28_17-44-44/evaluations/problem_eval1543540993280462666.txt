    df['SMA_long'] = df['close'].rolling(window=long_window).mean()

    # Compute momentum
    df['Momentum'] = df['close'] - df['close'].shift(1)

    # Combine factors with weights
    df['Combined_Score'] = (df['SMA_short'] * 0.6 + df['SMA_long'] * 0.4) + df['Momentum']

    # Apply a simple smoothing technique
    df['Smoothed_Score'] = df['Combined_Score'].rolling(window=3).mean()

    # Create the heuristics matrix
    heuristics_matrix = df['Smoothed_Score']
    
    return heuristics_matrix
