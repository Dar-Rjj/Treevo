def heuristics_v2(df):
    # Calculate 12-day momentum of close prices
    momentum_12 = df['close'] - df['close'].shift(12)

    # Calculate 20-day sum of log (base 10) of volume
    log_volume = np.log10(df['volume'])
    sum_log_volume = log_volume.rolling(window=20).sum()

    # Calculate 10-day standard deviation (volatility) of daily returns
    daily_return = df['close'].pct_change()
    std_10_daily_return = daily_return.rolling(window=10).std()

    # Heuristic score calculation
    heuristics_matrix = (momentum_12 / sum_log_volume) - std_10_daily_return

    return heuristics_matrix
