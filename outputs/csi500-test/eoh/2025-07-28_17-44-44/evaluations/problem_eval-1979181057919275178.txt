def heuristics_v2(df):
    # Calculate 20-day and 40-day EMAs of close prices
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_40_close = df['close'].ewm(span=40, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_20_close - ema_40_close

    # Calculate 10-day EMA of volume
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate 10-day standard deviation of close prices
    std_10_close = df['close'].rolling(window=10).std()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close / (ema_10_volume * std_10_close)

    return heuristics_matrix
