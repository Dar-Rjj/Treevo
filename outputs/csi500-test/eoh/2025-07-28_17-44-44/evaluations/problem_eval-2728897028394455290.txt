def heuristics_v2(df):
    sma_10_close = df['close'].rolling(window=10).mean()
    sma_20_close = df['close'].rolling(window=20).mean()
    
    sma_diff = sma_10_close - sma_20_close
    
    ema_14_volume = df['volume'].ewm(span=14, adjust=False).mean()
    max_14_close = df['close'].rolling(window=14).max()
    
    adjustment_term = ema_14_volume / max_14_close
    
    heuristics_matrix = sma_diff * adjustment_term
    
    return heuristics_matrix
