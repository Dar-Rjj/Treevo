def heuristics_v2(df):
    # Calculate 10-day and 20-day ATRs
    tr = df['high'] - df['low']
    atr_10 = tr.rolling(window=10).mean()
    atr_20 = tr.rolling(window=20).mean()

    # Ratio of 10-day to 20-day ATRs
    atr_ratio = atr_10 / atr_20

    # Calculate 5-day ATR
    atr_5 = tr.rolling(window=5).mean()

    # Calculate 5-day RSI of close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=5).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=5).mean()
    rs = gain / loss
    rsi_5 = 100 - (100 / (1 + rs))

    # Heuristic score calculation
    heuristics_matrix = atr_ratio + (rsi_5 * atr_5)

    return heuristics_matrix
