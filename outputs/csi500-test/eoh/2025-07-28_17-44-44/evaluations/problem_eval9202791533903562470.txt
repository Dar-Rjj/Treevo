def heuristics_v2(df):
    # Calculate 5-day and 10-day ATRs
    tr = df['high'] - df['low']
    atr_5 = tr.rolling(window=5).mean()
    atr_10 = tr.rolling(window=10).mean()

    # Ratio of 5-day to 10-day ATRs
    atr_ratio = atr_5 / atr_10

    # Calculate 10-day SMA of close prices
    sma_10_close = df['close'].rolling(window=10).mean()

    # Calculate 10-day standard deviation of volume
    std_10_volume = df['volume'].rolling(window=10).std()

    # Heuristic score calculation
    heuristics_matrix = atr_ratio + (sma_10_close * std_10_volume)

    return heuristics_matrix
