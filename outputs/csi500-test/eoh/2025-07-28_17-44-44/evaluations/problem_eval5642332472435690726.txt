def heuristics_v2(df):
    # Heuristic 1: Price Momentum - measures the momentum of the closing price
    price_momentum = df['close'].pct_change()
    
    # Heuristic 2: Volume Spike - identifies days with unusually high trading volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_spike = (df['volume'] - avg_volume) / avg_volume
    
    # Heuristic 3: Price Volatility - calculates the standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    price_volatility = daily_returns.rolling(window=20).std()
    
    # Combine heuristics into a single factor value per day
    heuristics_matrix = 0.5 * price_momentum + 0.3 * volume_spike - 0.2 * price_volatility

    return heuristics_matrix
