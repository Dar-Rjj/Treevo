import pandas as pd

def heuristics_v2(df):
    # Calculate 60-day momentum of close prices
    momentum_60_close = df['close'].pct_change(periods=60)

    # Calculate percentage change in volume from the previous day
    pct_change_volume = df['volume'].pct_change(periods=1)

    # Heuristic score calculation
    heuristics_matrix = momentum_60_close - pct_change_volume

    return heuristics_matrix
