import pandas as pd

def heuristics_v2(df):
    # Ratio of close to open prices
    close_open_ratio = df['close'] / df['open']
    
    # Standard deviation of daily returns over 10 days
    daily_returns = df['close'].pct_change()
    std_dev_returns = daily_returns.rolling(window=10).std()
    
    # Difference between the highest and lowest volume in the last 20 days
    vol_diff = df['volume'].rolling(window=20).max() - df['volume'].rolling(window=20).min()
    
    # Composite alpha factor
    heuristics_matrix = (close_open_ratio + std_dev_returns + vol_diff) / 3
    
    return heuristics_matrix
