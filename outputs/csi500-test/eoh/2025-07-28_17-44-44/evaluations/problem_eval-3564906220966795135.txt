def heuristics_v2(df):
    # Calculate the average of the last 5 days' close prices
    avg_close = df['close'].rolling(window=5).mean()
    # Calculate the standard deviation of the close prices and volume over the last 5 days
    std_close = df['close'].rolling(window=5).std()
    std_volume = df['volume'].rolling(window=5).std()
    # Compute the heuristic score as a weighted difference
    heuristics_matrix = (avg_close / std_close) - (df['volume'] / std_volume)
    return heuristics_matrix
