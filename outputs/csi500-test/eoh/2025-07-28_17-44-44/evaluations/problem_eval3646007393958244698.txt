import pandas as pd

def heuristics_v2(df):
    price_ratio = df['close'] / ((df['high'].rolling(window=15).mean() + df['low'].rolling(window=15).mean()) / 2)
    volume_change = (df['volume'] - df['volume'].shift(7)) / df['volume'].shift(7)
    combined_factors = price_ratio * volume_change
    heuristics_matrix = combined_factors.ewm(span=14, adjust=False).mean()
    return heuristics_matrix
