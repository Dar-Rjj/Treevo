def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_10 = (delta_close.where(delta_close > 0, 0)).rolling(window=10).mean()
    loss_10 = (-delta_close.where(delta_close < 0, 0)).rolling(window=10).mean()
    rsi_10 = 100 - (100 / (1 + gain_10 / loss_10))
    
    gain_20 = (delta_close.where(delta_close > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta_close.where(delta_close < 0, 0)).rolling(window=20).mean()
    rsi_20 = 100 - (100 / (1 + gain_20 / loss_20))
    
    rsi_ratio = rsi_10 / rsi_20
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    max_5_close = df['close'].rolling(window=5).max()
    
    heuristics_matrix = rsi_ratio - (ema_5_volume * max_5_close)
    
    return heuristics_matrix
