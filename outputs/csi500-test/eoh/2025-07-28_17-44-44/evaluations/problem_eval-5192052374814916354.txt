def heuristics_v2(df):
    momentum_5 = df['close'].pct_change(periods=5)
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_ratio = momentum_5 / momentum_10
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    sma_5_high = df['high'].rolling(window=5).mean()
    
    adjustment_term = ema_5_volume - sma_5_high
    
    heuristics_matrix = momentum_ratio * adjustment_term
    
    return heuristics_matrix
