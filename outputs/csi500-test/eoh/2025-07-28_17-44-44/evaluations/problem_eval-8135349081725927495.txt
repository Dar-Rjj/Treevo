def heuristics_v2(df):
    # Calculate 5-day and 20-day EMAs of close prices
    ema_5_close = df['close'].ewm(span=5, adjust=False).mean()
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_5_close - ema_20_close

    # Calculate 5-day EMA of volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate 5-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=5).mean()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close - (ema_5_volume * atr)

    return heuristics_matrix
