import pandas as pd

def heuristics_v2(df):
    # Compute MACD for close prices
    ema_12_close = df['close'].ewm(span=12, adjust=False).mean()
    ema_26_close = df['close'].ewm(span=26, adjust=False).mean()
    macd_close = ema_12_close - ema_26_close
    signal_close = macd_close.ewm(span=9, adjust=False).mean()

    # Compute MACD for volumes
    ema_12_volume = df['volume'].ewm(span=12, adjust=False).mean()
    ema_26_volume = df['volume'].ewm(span=26, adjust=False).mean()
    macd_volume = ema_12_volume - ema_26_volume
    signal_volume = macd_volume.ewm(span=9, adjust=False).mean()

    # Momentum factor
    momentum = (df['high'] - df['low']).rolling(window=10).mean()

    # Combining factors
    combined_factor = 2 * macd_close - signal_close + 1.5 * (macd_volume - signal_volume) + momentum
    
    heuristics_matrix = pd.Series(combined_factor, index=df.index)
    return heuristics_matrix
