def heuristics_v2(df):
    def stochastic_oscillator(df, n_days):
        low_n = df['low'].rolling(window=n_days).min()
        high_n = df['high'].rolling(window=n_days).max()
        k = (df['close'] - low_n) / (high_n - low_n) * 100
        return k
    
    stoch_10 = stochastic_oscillator(df, 10)
    stoch_30 = stochastic_oscillator(df, 30)
    
    stoch_ratio = stoch_10 / stoch_30
    
    ema_20_amount = df['amount'].ewm(span=20, adjust=False).mean()
    mean_20_high_low = (df['high'] + df['low']).rolling(window=20).mean() / 2
    
    heuristics_matrix = stoch_ratio - (ema_20_amount * mean_20_high_low)
    
    return heuristics_matrix
