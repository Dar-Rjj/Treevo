def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_5 = (delta_close.where(delta_close > 0, 0)).rolling(window=5).mean()
    loss_5 = (-delta_close.where(delta_close < 0, 0)).rolling(window=5).mean()
    rsi_5 = 100 - (100 / (1 + gain_5 / loss_5))
    
    gain_10 = (delta_close.where(delta_close > 0, 0)).rolling(window=10).mean()
    loss_10 = (-delta_close.where(delta_close < 0, 0)).rolling(window=10).mean()
    rsi_10 = 100 - (100 / (1 + gain_10 / loss_10))
    
    rsi_ratio = rsi_5 / rsi_10
    
    ema_10_amount = df['amount'].ewm(span=10, adjust=False).mean()
    std_30_close = df['close'].rolling(window=30).std()
    
    heuristics_matrix = rsi_ratio - (ema_10_amount * std_30_close)
    
    return heuristics_matrix
