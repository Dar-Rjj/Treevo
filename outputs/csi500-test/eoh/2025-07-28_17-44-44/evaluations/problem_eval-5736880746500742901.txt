def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_30 = (delta_close.where(delta_close > 0, 0)).rolling(window=30).mean()
    loss_30 = (-delta_close.where(delta_close < 0, 0)).rolling(window=30).mean()
    rsi_30 = 100 - (100 / (1 + gain_30 / loss_30))
    
    gain_60 = (delta_close.where(delta_close > 0, 0)).rolling(window=60).mean()
    loss_60 = (-delta_close.where(delta_close < 0, 0)).rolling(window=60).mean()
    rsi_60 = 100 - (100 / (1 + gain_60 / loss_60))
    
    rsi_ratio = rsi_30 / rsi_60
    
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    std_20_close = df['close'].rolling(window=20).std()
    
    heuristics_matrix = rsi_ratio - (ema_20_volume * std_20_close)
    
    return heuristics_matrix
