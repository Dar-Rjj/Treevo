def heuristics_v2(df):
    # Calculate 14-day and 28-day EMAs of close prices
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_28_close = df['close'].ewm(span=28, adjust=False).mean()

    # Logarithmic difference between 14-day and 28-day EMAs of close prices
    log_diff_ema_close = (ema_14_close - ema_28_close) / ema_28_close

    # Calculate 14-day EMA of volume
    ema_14_volume = df['volume'].ewm(span=14, adjust=False).mean()

    # Calculate 14-day Average True Range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()

    # Heuristic score calculation
    heuristics_matrix = log_diff_ema_close - (ema_14_volume * atr)

    return heuristics_matrix
