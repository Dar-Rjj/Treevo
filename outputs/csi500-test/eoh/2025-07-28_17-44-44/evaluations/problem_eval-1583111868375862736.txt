import pandas as pd

def heuristics_v2(df):
    sma_30_close = df['close'].rolling(window=30).mean()
    sma_15_high = df['high'].rolling(window=15).max().rolling(window=15).mean()
    sma_5_low = df['low'].rolling(window=5).min().rolling(window=5).mean()
    heuristics_matrix = (sma_30_close / sma_15_high) - sma_5_low
    return heuristics_matrix
