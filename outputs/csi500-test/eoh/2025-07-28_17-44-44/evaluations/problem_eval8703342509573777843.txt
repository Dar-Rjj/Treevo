def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_50 = (delta_close.where(delta_close > 0, 0)).rolling(window=50).mean()
    loss_50 = (-delta_close.where(delta_close < 0, 0)).rolling(window=50).mean()
    rsi_50 = 100 - (100 / (1 + gain_50 / loss_50))
    
    gain_150 = (delta_close.where(delta_close > 0, 0)).rolling(window=150).mean()
    loss_150 = (-delta_close.where(delta_close < 0, 0)).rolling(window=150).mean()
    rsi_150 = 100 - (100 / (1 + gain_150 / loss_150))
    
    rsi_ratio = rsi_50 / rsi_150
    
    ema_30_volume = df['volume'].ewm(span=30, adjust=False).mean()
    mean_30_close = df['close'].rolling(window=30).mean()
    
    heuristics_matrix = rsi_ratio - (ema_30_volume * mean_30_close)
    
    return heuristics_matrix
