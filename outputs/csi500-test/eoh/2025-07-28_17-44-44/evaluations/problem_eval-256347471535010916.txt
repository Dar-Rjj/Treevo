def heuristics_v2(df):
    # Calculate 15-day and 30-day EMAs of close prices
    ema_15_close = df['close'].ewm(span=15, adjust=False).mean()
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()

    # Difference between EMAs of close prices
    diff_ema_close = ema_15_close - ema_30_close

    # Calculate 7-day EMA of volume
    ema_7_volume = df['volume'].ewm(span=7, adjust=False).mean()

    # Calculate 7-day mean of close prices
    mean_7_close = df['close'].rolling(window=7).mean()

    # Heuristic score calculation
    heuristics_matrix = diff_ema_close - (ema_7_volume * mean_7_close)

    return heuristics_matrix
