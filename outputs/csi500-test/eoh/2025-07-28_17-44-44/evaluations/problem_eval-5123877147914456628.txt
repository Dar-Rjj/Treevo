    log_returns = np.log(df['close'] / df['close'].shift(1))

    # 30-day standard deviation of log returns
    std_30_log_returns = log_returns.rolling(window=30).std()

    # 10-day SMA of volume
    sma_10_volume = df['volume'].rolling(window=10).mean()

    # Heuristic score calculation
    heuristics_matrix = ratio_ema_close - (std_30_log_returns * sma_10_volume)

    return heuristics_matrix
