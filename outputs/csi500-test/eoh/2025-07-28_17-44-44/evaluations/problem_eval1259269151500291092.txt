def heuristics_v2(df):
    # Historical Volatility signal: standard deviation of daily returns over the past 20 days
    daily_returns = df['close'].pct_change()
    volatility_signal = daily_returns.rolling(window=20).std()
    
    # Relative Strength signal: ratio of today's close to the 25-day high
    high_25 = df['high'].rolling(window=25).max()
    relative_strength_signal = df['close'] / high_25
    
    # Composite alpha factor as the average of the above two signals
    heuristics_matrix = (volatility_signal + relative_strength_signal) / 2
    
    return heuristics_matrix
