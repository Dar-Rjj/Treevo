def heuristics_v2(df):
    ema_7_close = df['close'].ewm(span=7, adjust=False).mean()
    ema_14_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = ema_7_close / ema_14_close

    vol_std_7 = df['volume'].rolling(window=7).std()

    sma_5_high_low = (df['high'].rolling(window=5).mean() - df['low'].rolling(window=5).mean())
    sma_10_high_low = (df['high'].rolling(window=10).mean() - df['low'].rolling(window=10).mean())

    adjustment_term = sma_5_high_low - sma_10_high_low

    heuristics_matrix = ema_ratio / vol_std_7 + adjustment_term

    return heuristics_matrix
