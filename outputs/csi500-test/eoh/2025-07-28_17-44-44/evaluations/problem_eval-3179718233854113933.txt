def heuristics_v2(df):
    ema_21_close = df['close'].ewm(span=21, adjust=False).mean()
    ema_42_close = df['close'].ewm(span=42, adjust=False).mean()
    ema_ratio = ema_21_close / ema_42_close
    
    sma_10_volume = df['volume'].rolling(window=10).mean()
    max_10_close = df['close'].rolling(window=10).max()
    
    adjustment_term = sma_10_volume / max_10_close
    heuristics_matrix = ema_ratio - adjustment_term
    
    return heuristics_matrix
