import pandas as pd

def heuristics_v2(df):
    close_roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_roc_inv = 1 / ((df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10))
    weight = df['close'].rolling(window=10).mean()
    heuristics_matrix = (close_roc * weight) + (volume_roc_inv * (1 - weight))
    return heuristics_matrix
