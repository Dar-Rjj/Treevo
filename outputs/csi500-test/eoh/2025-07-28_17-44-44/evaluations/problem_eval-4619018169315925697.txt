import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change (ROC) of close prices
    roc_5_close = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Relative Strength Index (RSI) calculation with a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Adjusted liquidity: volume-to-amount ratio
    adjusted_liquidity = df['volume'] / (df['amount'] + 1e-6)  # Adding a small constant to avoid division by zero

    # Heuristic score calculation
    heuristics_matrix = (roc_5_close + rsi + adjusted_liquidity) / 3

    return heuristics_matrix
