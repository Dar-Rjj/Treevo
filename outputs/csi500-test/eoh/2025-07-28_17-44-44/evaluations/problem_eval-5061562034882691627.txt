def heuristics_v2(df):
    delta_close = df['close'].diff()
    gain_20 = (delta_close.where(delta_close > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta_close.where(delta_close < 0, 0)).rolling(window=20).mean()
    rsi_20 = 100 - (100 / (1 + gain_20 / loss_20))
    
    gain_30 = (delta_close.where(delta_close > 0, 0)).rolling(window=30).mean()
    loss_30 = (-delta_close.where(delta_close < 0, 0)).rolling(window=30).mean()
    rsi_30 = 100 - (100 / (1 + gain_30 / loss_30))
    
    rsi_ratio = rsi_20 / rsi_30
    
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    std_5_close = df['close'].rolling(window=5).std()
    
    heuristics_matrix = rsi_ratio - (ema_5_volume * std_5_close)
    
    return heuristics_matrix
