import pandas as pd

def heuristics_v2(df):
    # Generating heuristics based on simple moving average crossovers and volume changes
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    crossover_signal = (sma_10 - sma_30) / sma_30
    
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Simple heuristic combining the signals
    heuristics_matrix = 0.5 * crossover_signal + 0.5 * volume_change
    return heuristics_matrix
