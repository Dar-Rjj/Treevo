import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the short and long moving averages
    ma_diff = short_ma - long_ma
    
    # Calculate the volume-weighted price change
    vw_price_change = (df['close'] - df['open']) * df['volume']
    
    # Combine the moving average difference and the volume-weighted price change
    heuristics_matrix = ma_diff + vw_price_change
    
    return heuristics_matrix
