import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        prev_high = df['high'].shift(1).loc[row.name]
        price_change_pct = (row['close'] - row['open']) / row['open']
        high_ratio = row['high'] / prev_high
        weight = np.log(row['volume'])
        return (price_change_pct + high_ratio) * weight
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
