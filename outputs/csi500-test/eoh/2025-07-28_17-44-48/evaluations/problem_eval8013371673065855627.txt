def heuristics_v2(df):
    # Calculate the rate of change in volume
    roc_volume = df['volume'].pct_change().ewm(span=5, adjust=False).mean()
    
    # Calculate the true range
    tr = df['high'] - df['low']
    
    # Calculate the weighted average true range
    atr = tr.ewm(span=20, adjust=False).mean()
    
    # Combine the smoothed rate of change in volume and the weighted average true range
    heuristics_matrix = roc_volume + atr
    
    return heuristics_matrix
