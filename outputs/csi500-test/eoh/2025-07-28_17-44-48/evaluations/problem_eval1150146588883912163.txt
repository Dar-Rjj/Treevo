def heuristics_v2(df):
    # Calculate the 10-period momentum of closing prices
    momentum = df['close'].diff(periods=10)
    
    # Calculate the 10-period average of the high-low price difference
    avg_high_low_diff = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Combine the momentum with the average high-low difference
    heuristics_matrix = momentum + avg_high_low_diff
    
    return heuristics_matrix
