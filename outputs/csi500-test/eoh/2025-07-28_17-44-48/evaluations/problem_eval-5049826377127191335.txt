import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        avg_price = (row['open'] + row['close']) / 2
        price_range = np.sqrt(row['high'] - row['low'])
        return (avg_price * np.log(row['volume'])) / price_range
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
