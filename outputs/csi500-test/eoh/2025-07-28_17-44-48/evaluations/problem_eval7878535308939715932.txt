import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        sum_price = row['open'] + row['close']
        return (row['high'] - row['low']) / sum_price * np.sqrt(row['volume'] / df['volume'].mean())
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
