import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of high to low prices as a measure of relative volatility
    price_volatility_ratio = df['high'] / df['low']
    
    # Calculate the squared difference between closing and opening prices to gauge daily momentum intensity
    daily_momentum_squared = (df['close'] - df['open'])**2
    
    # Integrate the components into a single factor and scale by the logarithm of volume
    heuristics_matrix = (price_volatility_ratio + daily_momentum_squared) * np.log(df['volume'])
    
    return heuristics_matrix
