import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        hl_ratio = row['high'] / row['low']
        co_log = np.log(row['close'] / row['open'])
        return hl_ratio * co_log * (row['volume'] ** (1/3))
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
