import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        sum_diff = (abs(row['open'] - row['close']) + abs(row['high'] - row['low']))
        return sum_diff / row['close'] * np.sqrt(row['volume'])
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
