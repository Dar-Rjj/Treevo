def heuristics_v2(df):
    # Calculate the difference between high and low prices as a measure of volatility
    price_volatility = df['high'] - df['low']
    
    # Calculate the difference between closing and opening prices to gauge daily momentum
    daily_momentum = df['close'] - df['open']
    
    # Adjust the components with their respective weights
    weighted_volatility = 0.6 * price_volatility
    weighted_momentum = 0.4 * daily_momentum
    
    # Combine the components and adjust by the logarithm of trading volume
    heuristics_matrix = (weighted_volatility + weighted_momentum) * np.log1p(df['volume'])
    
    return heuristics_matrix
