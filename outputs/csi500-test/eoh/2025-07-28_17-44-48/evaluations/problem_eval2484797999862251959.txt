def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the rolling mean of the volume over a 5-day window
    df['volume_rolling_mean'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the difference between current volume and its rolling mean
    df['volume_diff'] = df['volume'] - df['volume_rolling_mean']
    
    # Compute a basic interaction term between the price change and the volume differential
    heuristics_matrix = (df['price_change'] * df['volume_diff']).fillna(0)
    
    return heuristics_matrix
