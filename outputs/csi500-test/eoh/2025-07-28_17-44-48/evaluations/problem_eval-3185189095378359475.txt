def heuristics_v2(df):
    # Rate of change in trading volume: (current volume - previous volume) / previous volume
    volume_roc = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weighted moving average of the last 5 days' close prices
    weights = [0.1, 0.2, 0.3, 0.2, 0.1]
    wma = df['close'].rolling(window=5).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Exponentially smoothed daily returns for volatility
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    smoothed_volatility = daily_returns.ewm(span=10, adjust=False).std()
    
    # Combine the factors into a single score
    heuristics_matrix = (volume_roc + wma + 1/smoothed_volatility)
    
    return heuristics_matrix
