import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        avg_price = (row['open'] + row['close']) / 2
        log_sum_prices = np.log(row['high'] + row['low'])
        return avg_price * log_sum_prices * np.cbrt(row['volume'])
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
