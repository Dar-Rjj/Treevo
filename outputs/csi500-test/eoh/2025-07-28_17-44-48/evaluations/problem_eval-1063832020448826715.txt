def heuristics_v2(df):
    def calc_volume_change(data, window):
        return data.pct_change(window)
    
    def calc_high_price_volatility(data, window):
        return data.rolling(window).std()

    volume_change = calc_volume_change(df['volume'], 20)
    high_price_volatility = calc_high_price_volatility(df['high'], 20)
    heuristics_matrix = (volume_change / high_price_volatility).dropna()
    
    return heuristics_matrix
