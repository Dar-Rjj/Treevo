def heuristics_v2(df):
    sma_window_trading_range = 10
    ema_window_close = 30
    trading_range = df['high'] - df['low']
    sma_trading_range = trading_range.rolling(window=sma_window_trading_range).mean()
    ema_close = df['close'].ewm(span=ema_window_close, adjust=False).mean()
    heuristics_matrix = sma_trading_range / ema_close
    return heuristics_matrix
