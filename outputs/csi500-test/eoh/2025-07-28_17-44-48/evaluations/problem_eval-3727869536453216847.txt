import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        price_ratio = row['open'] / row['close']
        log_volume = np.log10(row['volume'])
        hl_diff = row['high'] - row['low']
        return price_ratio * log_volume + hl_diff
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
