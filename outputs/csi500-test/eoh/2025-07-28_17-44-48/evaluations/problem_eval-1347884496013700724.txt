def heuristics_v2(df):
    # Calculate the 10-period exponential moving average of the close price
    ema_price = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the rate of change of the exponential moving average
    roc_ema = ema_price.pct_change()
    
    # Take the natural logarithm of the volume
    log_volume = np.log(df['volume'])
    
    # Combine the rate of change of the EMA and the logarithm of volume
    heuristics_matrix = roc_ema + log_volume
    
    return heuristics_matrix
