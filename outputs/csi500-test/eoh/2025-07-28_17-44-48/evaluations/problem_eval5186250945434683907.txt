def heuristics_v2(df):
    # Calculate the price momentum over a 10-day period
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate the average volatility over a 10-day period
    average_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Compute the momentum to volatility ratio
    heuristics_matrix = momentum / average_volatility
    
    return heuristics_matrix
