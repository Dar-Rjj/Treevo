def heuristics_v2(df):
    # Calculate the difference between high and low prices as a measure of volatility
    price_volatility = df['high'] - df['low']

    # Calculate the Relative Strength Index (RSI) based on the close prices for a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the average volume over a 14-day window
    avg_volume = df['volume'].rolling(window=14).mean()
    
    # Calculate the volume-to-average-volume ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Integrate the components into a single factor
    heuristics_matrix = (rsi * volume_ratio) / (price_volatility + 1e-6)

    return heuristics_matrix
