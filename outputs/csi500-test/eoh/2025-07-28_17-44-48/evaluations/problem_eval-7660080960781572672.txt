def heuristics_v2(df):
    ema_window_trading_range = 10
    sma_window_close = 30
    trading_range = df['high'] - df['low']
    ema_trading_range = trading_range.ewm(span=ema_window_trading_range, adjust=False).mean()
    sma_close = df['close'].rolling(window=sma_window_close).mean()
    heuristics_matrix = ema_trading_range / sma_close
    return heuristics_matrix
