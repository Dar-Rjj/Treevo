import pandas as pd

def heuristics_v2(df):
    # Momentum: Simple moving average of the last 5 days' close prices
    momentum = df['close'].rolling(window=5).mean()
    
    # Volatility: Standard deviation of daily returns over the last 10 days
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    volatility = daily_returns.rolling(window=10).std()
    
    # Volume Trend: Moving average of the last 5 days' volume
    volume_trend = df['volume'].rolling(window=5).mean()
    
    # Combine factors into a single score; for simplicity, we use a linear combination
    heuristics_matrix = (momentum + 1/volatility + volume_trend)
    
    return heuristics_matrix
