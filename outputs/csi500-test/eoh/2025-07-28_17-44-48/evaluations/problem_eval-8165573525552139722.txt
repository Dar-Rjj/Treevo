def heuristics_v2(df):
    # Calculate the 20-day simple moving average of closing prices
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the percentage change in the 20-day SMA
    sma_change = sma.pct_change()
    
    # Calculate the RSI over a 14-period span
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the RSI with the percentage change in the 20-day SMA
    heuristics_matrix = rsi + sma_change
    
    return heuristics_matrix
