import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        weight = np.sqrt(row['volume'])
        return (weight * row['high'] + weight * row['low'] + 2 * weight * row['close']) / (4 * weight)
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
