import pandas as pd
import numpy as np

def heuristics_v2(df):
    def compute_factor(row, ema_close, sma_open, log_volume):
        return (ema_close - sma_open) / log_volume
    
    # Calculate EMA and SMA
    df['ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['sma_open'] = df['open'].rolling(window=10).mean()
    df['log_volume'] = np.log(df['volume'])
    
    # Compute the factor
    heuristics_matrix = df.apply(lambda row: compute_factor(row, row['ema_close'], row['sma_open'], row['log_volume']), axis=1)
    
    return heuristics_matrix
