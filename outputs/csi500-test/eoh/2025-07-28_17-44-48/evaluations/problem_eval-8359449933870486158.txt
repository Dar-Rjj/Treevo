def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) of closing prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Exponential Moving Average (EMA) of trading volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Integrate the components into a single factor
    heuristics_matrix = rsi * ema_volume
    
    return heuristics_matrix
