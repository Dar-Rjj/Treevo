import pandas as pd

def heuristics_v2(df):
    # Calculate the standard deviation of the closing price over a 10-day window
    close_std = df['close'].rolling(window=10).std()
    
    # Calculate the daily price range
    price_range = df['high'] - df['low']
    
    # Adjust the volume by the daily price range
    adjusted_volume = df['volume'] / price_range
    
    # Combine the standard deviation of the closing price and the adjusted volume
    heuristics_matrix = close_std + adjusted_volume
    
    return heuristics_matrix
