def heuristics_v2(df):
    # Calculate the RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR
    tr = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the ROC of trading volume
    roc_volume = df['volume'].pct_change(periods=14)
    
    # Integrate the components into a single factor
    heuristics_matrix = (rsi + atr + roc_volume) / 3
    
    return heuristics_matrix
