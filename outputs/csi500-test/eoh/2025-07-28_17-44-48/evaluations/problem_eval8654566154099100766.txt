def heuristics_v2(df):
    ema_window_trading_range = 10
    ema_window_close = 30
    trading_range = df['high'] - df['low']
    ema_trading_range = trading_range.ewm(span=ema_window_trading_range, adjust=False).mean()
    ema_close = df['close'].ewm(span=ema_window_close, adjust=False).mean()
    heuristics_matrix = ema_trading_range / ema_close
    return heuristics_matrix
