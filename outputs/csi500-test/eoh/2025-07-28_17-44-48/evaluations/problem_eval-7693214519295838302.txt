import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Feature engineering to generate potential alphas
    alpha1 = df['close'] / df['open']
    alpha2 = (df['high'] - df['low']) / df['volume']
    alpha3 = np.log(df['amount'])
    alpha4 = np.exp((df['close'] - df['open']) / 10)
    
    # Combining alphas into a DataFrame
    alphas_df = pd.DataFrame({
        'alpha1': alpha1,
        'alpha2': alpha2,
        'alpha3': alpha3,
        'alpha4': alpha4
    })
    
    # Selecting the alpha with the highest absolute correlation to next day's return
    tomorrow_return = (df['close'].shift(-1) - df['close']) / df['close']
    correlations = alphas_df.corrwith(tomorrow_return).abs()
    best_alpha_name = correlations.idxmax()
    
    # Constructing the final heuristics matrix
    heuristics_matrix = alphas_df[best_alpha_name]
    
    return heuristics_matrix
