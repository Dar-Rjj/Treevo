import numpy as np
import pandas as pd

def heuristics_v2(df):
    def compute_factor(row):
        avg_price = (row['open'] + row['high'] + row['low']) / 3
        return row['close'] / avg_price * np.log(row['volume'] + 1)
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
