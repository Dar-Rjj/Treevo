import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in closing price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of amount to volume to infer investor sentiment
    amount_vol_ratio = df['amount'] / df['volume']
    
    # Combine the calculated features into a single heuristic factor
    heuristics_matrix = (price_diff + close_pct_change + amount_vol_ratio) / 3
    
    return heuristics_matrix
