import numpy as np
import pandas as pd

def heuristics_v2(df):
    def compute_factor(row):
        avg_price = (row['high'] + row['low']) / 2
        geom_mean = np.sqrt(row['open'] * row['close'])
        return avg_price / geom_mean * np.sqrt(row['volume'])
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
