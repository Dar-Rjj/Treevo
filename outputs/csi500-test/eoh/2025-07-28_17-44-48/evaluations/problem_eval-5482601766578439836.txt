def heuristics_v2(df):
    # Calculate the rate of change in volume
    roc_volume = df['volume'].pct_change().abs()
    
    # Calculate the true range
    tr = df['high'] - df['low']
    
    # Calculate the weighted average true range
    atr = tr.ewm(span=21, adjust=False).mean()
    
    # Combine the modified rate of change in volume and the weighted average true range
    heuristics_matrix = roc_volume * atr
    
    return heuristics_matrix
