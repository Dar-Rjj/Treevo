import pandas as pd

def heuristics_v2(df):
    # Rate of change in daily high prices
    high_roc = df['high'].pct_change()
    
    # Rate of change in daily low prices
    low_roc = df['low'].pct_change()
    
    # Cumulative sum of trading volume
    cum_volume = df['volume'].cumsum()
    
    # Ratio of closing price to opening price
    close_open_ratio = df['close'] / df['open']
    
    # Combine the calculated features into a single heuristic factor
    heuristics_matrix = (high_roc + low_roc + cum_volume + close_open_ratio) / 4
    
    return heuristics_matrix
