import pandas as pd

def heuristics_v2(df):
    # Calculate the simple heuristic factor
    df['heuristic_factor'] = (2*df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    # Smooth the heuristic factor over a 10-day window
    heuristics_matrix = df['heuristic_factor'].rolling(window=10).mean()
    return heuristics_matrix
