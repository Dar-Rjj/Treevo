import numpy as np
import pandas as pd

def heuristics_v2(df):
    def compute_factor(row):
        ema_close = df['close'].ewm(span=5, adjust=False).mean()[row.name]
        price_sum = row['open'] + row['high']
        return (ema_close / price_sum) * np.log(row['volume'])
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
