def heuristics_v2(df):
    # Calculate the rate of change in closing price
    roc_close = df['close'].pct_change()
    
    # Calculate the cumulative sum of trading volume
    cum_volume = df['volume'].cumsum()
    
    # Integrate the components into a single factor
    heuristics_matrix = roc_close * cum_volume
    
    return heuristics_matrix
