def heuristics_v2(df):
    # Calculate the logarithmic return as a measure of daily price change
    log_return = np.log(df['close'] / df['open'])
    
    # Calculate the average volume over the period
    avg_volume = df['volume'].mean()
    
    # Compute the ratio of volume to average volume
    volume_ratio = df['volume'] / avg_volume
    
    # Integrate the components into a single factor
    heuristics_matrix = log_return * volume_ratio
    
    return heuristics_matrix
