def heuristics_v2(df):
    # Calculate the difference between high and low prices as a measure of volatility
    price_volatility = df['high'] - df['low']
    
    # Calculate the difference between closing and opening prices to gauge daily momentum
    daily_momentum = df['close'] - df['open']
    
    # Integrate the components into a single factor
    heuristics_matrix = (price_volatility + daily_momentum) * df['volume']
    
    return heuristics_matrix
