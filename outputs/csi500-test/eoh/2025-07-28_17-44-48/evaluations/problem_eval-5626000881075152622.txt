import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in closing price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the closing price
    close_ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of amount to volume to infer investor sentiment
    amount_vol_ratio = df['amount'] / df['volume']
    
    # Combine the calculated features into a single heuristic factor with adjusted weights
    heuristics_matrix = (price_diff * 0.3 + close_pct_change * 0.4 + amount_vol_ratio * 0.2 + close_ma_5) / 1.9
    
    return heuristics_matrix
