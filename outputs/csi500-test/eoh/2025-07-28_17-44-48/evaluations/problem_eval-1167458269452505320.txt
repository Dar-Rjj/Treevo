def heuristics_v2(df):
    def calc_momentum(data, window):
        return data.pct_change(window)

    def calc_volatility(data, window):
        return data.rolling(window).std()

    momentum = calc_momentum(df['close'], 20)
    volatility = calc_volatility(df['close'], 20)
    heuristics_matrix = (momentum / volatility).dropna()
    
    return heuristics_matrix
