import numpy as np

def heuristics_v2(df):
    # Calculate the difference between high and low prices as a measure of volatility
    price_volatility = df['high'] - df['low']
    
    # Calculate the difference between closing and opening prices to gauge daily momentum
    daily_momentum = df['close'] - df['open']
    
    # Compute the ratio of daily momentum to price volatility
    momentum_to_volatility_ratio = daily_momentum / (price_volatility + 1e-6)
    
    # Use the logarithm of trading volume to account for liquidity
    log_volume = np.log(df['volume'] + 1e-6)
    
    # Integrate the components into a single factor
    heuristics_matrix = momentum_to_volatility_ratio * log_volume
    
    return heuristics_matrix
