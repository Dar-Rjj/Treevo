def heuristics_v2(df):
    # Calculate the rate of change in volume
    roc_volume = df['volume'].pct_change().apply(lambda x: np.log(1 + x) if x > 0 else -np.log(1 - x))
    
    # Calculate the true range
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift()), abs(x['low'] - x['close'].shift())), axis=1)
    
    # Calculate the weighted average true range
    atr = tr.ewm(span=14, adjust=False).mean()
    
    # Combine the log of the rate of change in volume and the weighted average true range
    heuristics_matrix = roc_volume + atr
    
    return heuristics_matrix
