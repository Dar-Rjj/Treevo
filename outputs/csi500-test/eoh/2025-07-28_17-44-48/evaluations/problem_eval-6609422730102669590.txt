def heuristics_v2(df):
    short_window = 5
    long_window = 20
    high_low_avg = (df['high'].rolling(window=short_window).mean() + df['low'].rolling(window=short_window).mean()) / 2
    close_long_avg = df['close'].rolling(window=long_window).mean()
    heuristics_matrix = high_low_avg - close_long_avg
    return heuristics_matrix
