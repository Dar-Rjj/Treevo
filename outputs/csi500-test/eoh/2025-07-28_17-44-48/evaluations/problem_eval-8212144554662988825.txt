import pandas as pd

def heuristics_v2(df):
    def compute_factor(row):
        avg_price = (row['high'] + row['low']) / 2
        ema_volume = df['volume'].ewm(span=5, adjust=False).mean()[row.name]
        return (row['close'] / avg_price) * ema_volume
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
