def heuristics_v2(df):
    daily_momentum = df['close'] - df['open']
    price_volatility = df['high'] - df['low']
    sma_volatility = price_volatility.rolling(window=30).mean()
    cumulative_momentum = daily_momentum.cumsum()
    heuristics_matrix = cumulative_momentum / sma_volatility
    return heuristics_matrix
