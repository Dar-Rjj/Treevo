import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices, then apply an exponential moving average
    price_diff_ema = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()
    
    # Calculate the percentage change in closing price, with a focus on recent changes using an EMA
    close_pct_change_ema = df['close'].pct_change().ewm(span=5, adjust=False).mean()
    
    # Calculate the ratio of amount to volume to infer investor sentiment, applying an EMA
    amount_vol_ratio_ema = (df['amount'] / df['volume']).ewm(span=7, adjust=False).mean()
    
    # Combine the calculated features into a single heuristic factor
    heuristics_matrix = 0.5 * price_diff_ema + 0.3 * close_pct_change_ema + 0.2 * amount_vol_ratio_ema
    
    return heuristics_matrix
