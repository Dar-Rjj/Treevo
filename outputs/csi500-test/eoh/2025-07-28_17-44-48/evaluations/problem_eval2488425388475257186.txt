def heuristics_v2(df):
    # Calculate the percentage change in closing prices
    close_change = df['close'].pct_change()
    
    # Calculate the 10-day moving average of trading volume
    volume_ma = df['volume'].rolling(window=10).mean()
    
    # Integrate the components into a single factor
    heuristics_matrix = close_change * volume_ma
    
    return heuristics_matrix
