import numpy as np

def heuristics_v2(df):
    def compute_factor(row):
        price_diff = row['high'] - row['low']
        price_sum = row['open'] + row['close']
        return (price_diff / price_sum) * np.sqrt(row['volume'])
    heuristics_matrix = df.apply(compute_factor, axis=1)
    return heuristics_matrix
