def heuristics_v2(df):
    # Calculate the percentage change in closing prices to gauge momentum
    close_momentum = df['close'].pct_change().fillna(0)
    
    # Calculate the relative trading volume pressure as the ratio of current volume to the average volume over a rolling window
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_pressure = df['volume'] / avg_volume
    
    # Integrate the components into a single factor
    heuristics_matrix = close_momentum * volume_pressure
    
    return heuristics_matrix
