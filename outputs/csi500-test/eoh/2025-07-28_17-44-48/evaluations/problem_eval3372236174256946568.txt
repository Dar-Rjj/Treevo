def heuristics_v2(df):
    # Calculate the 5-day percentage change in closing prices
    pct_change = df['close'].pct_change(5)
    
    # Calculate the 5-day average volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of current volume to the 5-day average volume
    volume_ratio = df['volume'] / avg_volume
    
    # Combine the two components into a single factor
    heuristics_matrix = pct_change * volume_ratio
    
    return heuristics_matrix
