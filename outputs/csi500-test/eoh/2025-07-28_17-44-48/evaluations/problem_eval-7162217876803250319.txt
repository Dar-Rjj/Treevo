def heuristics_v2(df):
    ema_window = 10
    sma_window = 30
    trading_range = df['high'] - df['low']
    ema_trading_range = trading_range.ewm(span=ema_window, adjust=False).mean()
    sma_close = df['close'].rolling(window=sma_window).mean()
    heuristics_matrix = ema_trading_range / sma_close
    return heuristics_matrix
