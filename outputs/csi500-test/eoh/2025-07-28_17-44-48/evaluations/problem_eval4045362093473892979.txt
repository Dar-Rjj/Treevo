import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of closing to opening prices
    price_ratio = df['close'] / df['open']
    
    # Use the natural logarithm of the volume to scale the price ratio
    log_volume = np.log(df['volume'])
    
    # Compute the final heuristics factor
    heuristics_matrix = price_ratio / log_volume
    
    return heuristics_matrix
