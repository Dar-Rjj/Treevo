def heuristics_v2(df):
    # Calculate the 20-day EWMA of the closing price
    ewma_close_20 = df['close'].ewm(span=20).mean()
    
    # Calculate the 20-day standard deviation of the daily returns
    std_return_20 = df['close'].pct_change().ewm(span=20).std()
    
    # Compute the heuristic factor
    heuristics_matrix = ewma_close_20 / std_return_20
    
    return heuristics_matrix
