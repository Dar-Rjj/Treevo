def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate 20-day average return
    avg_return_20 = df['return'].rolling(window=20).mean()
    
    # Calculate 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate the volume ratio (today's volume / 20-day average volume)
    volume_ratio = df['volume'] / avg_volume_20
    
    # Compute the heuristic factor
    heuristics_matrix = avg_return_20 * volume_ratio
    
    return heuristics_matrix
