```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    high_low_momentum = high_low_range.rolling(window=20).mean()
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    high_low_std = high_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (high_low_momentum * short_term_trend) - high_low_std
    
    # Adjust Combined Factor by Volume
    volume_adjusted_factor = combined_factor * df['volume']
    
    return volume_adjusted_factor
```
