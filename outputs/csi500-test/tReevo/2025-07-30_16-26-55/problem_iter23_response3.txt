```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame, range_window: int = 20, trend_window: int = 5) -> pd.Series:
    # Calculate High-to-Low Range
    high_to_low_range = data['high'] - data['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=range_window).std()
    rolling_mean = high_to_low_range.rolling(window=range_window).mean()
    range_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_term_trend = data['close'].rolling(window=trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * data['volume']
    
    return adjusted_factor
```
