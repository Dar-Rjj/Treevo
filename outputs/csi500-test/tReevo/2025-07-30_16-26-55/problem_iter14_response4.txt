```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    window_size = 30  # Define the window size for standard deviation
    rolling_std = high_to_low_range.rolling(window=window_size).std()
    
    # Normalize High-to-Low Range
    normalized_range = high_to_low_range / rolling_std
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_window = 10  # Define the short window for moving average
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Normalized High-to-Low Range and Short-Term Trend
    combined_factor = normalized_range * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
