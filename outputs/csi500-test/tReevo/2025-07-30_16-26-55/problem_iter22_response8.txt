```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=50, rolling_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Statistics of Range
    std_high_to_low_range = high_to_low_range.rolling(window=rolling_window).std()
    mean_high_to_low_range = high_to_low_range.rolling(window=rolling_window).mean()
    
    # Calculate High-to-Low Range Momentum
    range_momentum = std_high_to_low_range / mean_high_to_low_range
    
    # Calculate Short-Term and Long-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()
    long_term_trend = df['close'].rolling(window=long_window).mean()
    
    # Combine Momentum, Short-Term, and Long-Term Trends
    combined_factor = (range_momentum * short_term_trend) - long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
