```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std_range = high_to_low_range.rolling(window=20).std()
    rolling_mean_range = high_to_low_range.rolling(window=20).mean()
    normalized_rolling_std = rolling_std_range / rolling_mean_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (normalized_rolling_std * short_term_trend) - rolling_std_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
