```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 20-day Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=20).mean()
    
    # Calculate Upper and Lower Price Envelopes
    df['Upper_Envelope'] = df['SMA'] + df['close'].rolling(window=20).std()
    df['Lower_Envelope'] = df['SMA'] - df['close'].rolling(window=20).std()
    
    # Identify Breakouts
    df['Upper_Breakout'] = df['close'] > df['Upper_Envelope']
    df['Lower_Breakout'] = df['close'] < df['Lower_Envelope']
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Adjust Volume-Weighted Momentum with Volatility
    df['Volume_Adjusted_Momentum'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['ATR']
    
    # Generate Alpha Factor Signal
    df['Alpha_Factor'] = np.where(
        df['Upper_Breakout'], 
        df['Volume_Adjusted_Momentum'],
        np.where(df['Lower_Breakout'], -df['Volume_Adjusted_Momentum'], 0)
    )
    
    # Assign No-Signal for Intraday Traps
    df['Alpha_Factor'] = np.where((df['Upper_Breakout'] | df['Lower_Breakout']), df['Alpha_Factor'], 0)
    
    return df['Alpha_Factor']
```
