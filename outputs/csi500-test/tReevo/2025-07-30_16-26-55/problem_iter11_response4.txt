```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    range_rolling_std = high_to_low_range.rolling(window=20).std()
    
    # Normalize by Rolling Mean of High-to-Low Range
    range_rolling_mean = high_to_low_range.rolling(window=20).mean()
    range_momentum = (range_rolling_std / range_rolling_mean).fillna(0)
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean().fillna(0)
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - range_rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
