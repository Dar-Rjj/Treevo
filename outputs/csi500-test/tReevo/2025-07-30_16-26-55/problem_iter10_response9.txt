import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['hl_range'] = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    std_window = 30
    df['range_std'] = df['hl_range'].rolling(window=std_window).std()
    
    # Normalize High-to-Low Range Momentum by its rolling mean
    mean_window = 30
    df['normalized_momentum'] = (df['hl_range'] - df['hl_range'].rolling(window=mean_window).mean()) / df['range_std']
    
    # Calculate Short-Term and Long-Term Trend
    short_trend_window = 10
    long_trend_window = 50
    df['short_term_trend'] = df['close'].rolling(window=short_trend_window).mean()
    df['long_term_trend'] = df['close'].rolling(window=long_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    df['combined_factor'] = df['normalized_momentum'] * df['short_term_trend'] - df['range_std']
    
    # Integrate Long-Term Trend
    df['integrated_factor'] = df['combined_factor'] + df['long_term_trend']
    
    # Adjust Integrated Factor by Volume
    df['factor'] = df['integrated_factor'] * df['volume']
    
    return df['factor']
