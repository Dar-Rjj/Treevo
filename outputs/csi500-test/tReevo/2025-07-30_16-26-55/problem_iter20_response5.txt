```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    window = 30  # Define the rolling window size
    range_rolling_std = high_to_low_range.rolling(window=window).std()
    range_rolling_mean = high_to_low_range.rolling(window=window).mean()
    high_to_low_momentum = range_rolling_std / range_rolling_mean
    
    # Calculate Short-Term Trend
    short_term_window = 5  # Define the short-term moving average window
    short_term_trend = df['close'].rolling(window=short_term_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend - range_rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
