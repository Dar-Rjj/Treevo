```python
import pandas as pd

def heuristics_v2(df, short_term_window=20, long_term_window=252, momentum_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    high_to_low_momentum = high_to_low_range.rolling(window=momentum_window).std()
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = high_to_low_momentum * df['volume']
    
    # Integrate with Longer-Term Trends
    long_term_trend = df['close'].rolling(window=long_term_window).mean()
    short_term_trend = df['close'].rolling(window=short_term_window).mean()
    
    # Combine Short-Term and Long-Term Trends
    combined_trend = (short_term_trend - long_term_trend) / long_term_trend
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_momentum * combined_trend
    
    return alpha_factor
```
