```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Price Momentum
    df['10_SMA'] = df['close'].rolling(window=10).mean()
    df['short_mom'] = df['close'] - df['10_SMA']
    df['pos_short_mom_5d'] = df['short_mom'].where(df['short_mom'] > 0, 0).rolling(window=5).sum()

    # Medium-Term Price Momentum
    df['20_SMA'] = df['close'].rolling(window=20).mean()
    df['medium_mom'] = df['close'] - df['20_SMA']
    df['pos_medium_mom_10d'] = df['medium_mom'].where(df['medium_mom'] > 0, 0).rolling(window=10).sum()

    # Long-Term Price Momentum
    df['30_SMA'] = df['close'].rolling(window=30).mean()
    df['long_mom'] = df['close'] - df['30_SMA']
    df['pos_long_mom_20d'] = df['long_mom'].where(df['long_mom'] > 0, 0).rolling(window=20).sum()

    # Price Breakout Indicator
    df['20_high'] = df['high'].rolling(window=20).max()
    df['20_low'] = df['low'].rolling(window=20).min()
    df['dist_from_high'] = (df['close'] - df['20_high']) / df['20_high']
    df['dist_from_low'] = (df['close'] - df['20_low']) / df['20_low']
    df['breakout_indicator'] = df['dist_from_high'] + df['dist_from_low']

    # Volume Adjust Factor
    df['20_avg_vol'] = df['volume'].rolling(window=20).mean()
    df['vol_adjust_factor'] = df['volume'] / df['20_avg_vol']

    # Volatility Adjust Factor
    df['20_std_close'] = df['close'].rolling(window=20).std()
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], x['low']) - min(x['high'], x['low']), axis=1)
    df['20_atr'] = df['true_range'].rolling(window=20).mean()
    df['volatility_adjust_factor'] = (df['20_std_close'] + df['20_atr']) / 2

    # Combine Factors
    df['short_adj_mom'] = df['pos_short_mom_5d'] * df['vol_adjust_factor']
    df['medium_adj_mom'] = df['pos_medium_mom_10d'] * df['vol_adjust_factor']
    df['long_adj_mom'] = df['pos_long_mom_20d'] * df['vol_adjust_factor']
    df['combined_mom'] = df['short_adj_mom'] + df['medium_adj_mom'] + df['long_adj_mom']
    df['final_factor'] = df['combined_mom'] * df['volatility_adjust_factor'] + df['breakout_indicator']

    return df['final_factor'].dropna()
```
