```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily returns
    df['daily_return'] = df['close'].pct_change()

    # Compute 5-day momentum
    df['5_day_momentum'] = df['daily_return'].rolling(window=5).sum()

    # Compute 20-day mean reversion
    df['20_day_mean_reversion'] = df['daily_return'].rolling(window=20).mean()

    # Compute open-to-close return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']

    # Compute high-to-low range
    df['high_to_low_range'] = (df['high'] - df['low']) / df['low']

    # Adjust by volume
    df['volume_adjusted_open_to_close_return'] = df['open_to_close_return'] * df['volume']

    # Combine intraday components
    df['intraday_component'] = df['volume_adjusted_open_to_close_return'] - df['high_to_low_range']

    # Incorporate short-term historical information
    df['average_5_day_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_5_day_momentum'] = df['5_day_momentum'] * (df['average_5_day_volume'] / df['volume'])

    # Final alpha factor
    df['alpha_factor'] = (
        df['intraday_component'] +
        df['volume_adjusted_5_day_momentum'] +
        df['20_day_mean_reversion']
    )

    return df['alpha_factor'].dropna()
```
