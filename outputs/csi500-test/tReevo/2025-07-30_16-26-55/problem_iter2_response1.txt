```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Price Momentum
    df['10_SMA'] = df['close'].rolling(window=10).mean()
    df['short_momentum'] = df['close'] - df['10_SMA']
    df['positive_short_momentum'] = df['short_momentum'].where(df['short_momentum'] > 0, 0)
    df['short_term_mom_sum'] = df['positive_short_momentum'].rolling(window=5).sum()

    # Medium-Term Price Momentum
    df['20_SMA'] = df['close'].rolling(window=20).mean()
    df['medium_momentum'] = df['close'] - df['20_SMA']
    df['positive_medium_momentum'] = df['medium_momentum'].where(df['medium_momentum'] > 0, 0)
    df['medium_term_mom_sum'] = df['positive_medium_momentum'].rolling(window=10).sum()

    # Long-Term Price Momentum
    df['30_SMA'] = df['close'].rolling(window=30).mean()
    df['long_momentum'] = df['close'] - df['30_SMA']
    df['positive_long_momentum'] = df['long_momentum'].where(df['long_momentum'] > 0, 0)
    df['long_term_mom_sum'] = df['positive_long_momentum'].rolling(window=20).sum()

    # Price Breakout Indicator
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['distance_from_20_day_high'] = (df['close'] / df['20_day_high']) - 1
    df['distance_from_20_day_low'] = (df['close'] / df['20_day_low']) - 1
    df['breakout_indicator'] = df['distance_from_20_day_high'] + df['distance_from_20_day_low']

    # Volume Adjust Factor
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjust_factor'] = df['volume'] / df['20_day_avg_volume']

    # Combine Factors
    df['short_term_vol_adj_mom'] = df['short_term_mom_sum'] * df['volume_adjust_factor']
    df['medium_term_vol_adj_mom'] = df['medium_term_mom_sum'] * df['volume_adjust_factor']
    df['long_term_vol_adj_mom'] = df['long_term_mom_sum'] * df['volume_adjust_factor']
    
    df['combined_momentum'] = df['short_term_vol_adj_mom'] + df['medium_term_vol_adj_mom'] + df['long_term_vol_adj_mom']
    df['alpha_factor'] = df['combined_momentum'] + df['breakout_indicator']
    
    return df['alpha_factor']
```
