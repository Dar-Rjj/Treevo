```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    short_window = 10
    ema_high_low_range = high_low_range.ewm(span=short_window, adjust=False).mean()
    high_low_range_momentum = ema_high_low_range / df['volume']
    
    # Calculate Short-Term Trend
    short_trend_window = 5
    short_term_trend = df['close'].rolling(window=short_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_low_range_momentum * short_term_trend
    
    # Subtract Rolling Standard Deviation of High-to-Low Range
    std_high_low_range = high_low_range.rolling(window=short_window).std()
    combined_factor -= std_high_low_range
    
    # Adjust Combined Factor by Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    rolling_std_log_returns = log_returns.rolling(window=short_window).std()
    inverse_volatility = 1 / rolling_std_log_returns
    final_alpha_factor = combined_factor * inverse_volatility
    
    return final_alpha_factor
```
