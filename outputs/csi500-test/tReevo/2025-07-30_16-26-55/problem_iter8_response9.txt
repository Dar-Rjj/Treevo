import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation and Mean of High-to-Low Range
    window_size = 30  # Example window size, can be adjusted
    rolling_std = high_to_low_range.rolling(window=window_size).std()
    rolling_mean = high_to_low_range.rolling(window=window_size).mean()
    
    # Normalize by Rolling Mean of High-to-Low Range
    normalized_std = rolling_std / rolling_mean
    
    # Calculate High-to-Low Range Momentum
    range_momentum = (high_to_low_range - rolling_mean) / rolling_std
    
    # Calculate Short-Term Trend using Close prices
    short_window = 10  # Example short window size, can be adjusted
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_term_trend - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
