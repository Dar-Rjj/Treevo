```python
import pandas as pd

def heuristics_v2(df, momentum_window=20, volume_lookback=20):
    """
    Generate a novel alpha factor based on high-to-low range momentum adjusted by volume.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        momentum_window (int): Window size for calculating the rolling mean and standard deviation of the range.
        volume_lookback (int): Lookback period for normalizing the volume.
        
    Returns:
        pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Mean and Standard Deviation of the Range
    rolling_mean_range = high_to_low_range.rolling(window=momentum_window).mean()
    rolling_std_range = high_to_low_range.rolling(window=momentum_window).std()
    
    # Compute High-to-Low Range Momentum
    range_momentum = (high_to_low_range - rolling_mean_range) / rolling_std_range
    
    # Normalize Volume (Calculate Volume Ratio)
    average_volume = df['volume'].rolling(window=volume_lookback).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * volume_ratio
    
    return adjusted_momentum
```
