```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['daily_range'] = df['high'] - df['low']
    df['avg_5d_range'] = df['daily_range'].rolling(window=5).mean()

    # Incorporate Volume Trend
    df['volume_5d_ma'] = df['volume'].rolling(window=5).mean()

    # Combine Price and Volume Information
    df['price_volume_factor'] = (df['daily_range'] * df['volume']) / df['volume_5d_ma']

    # Smooth the Combined Factor
    df['final_alpha_factor'] = df['price_volume_factor'].rolling(window=10).mean()
    
    return df['final_alpha_factor'].dropna()
```
