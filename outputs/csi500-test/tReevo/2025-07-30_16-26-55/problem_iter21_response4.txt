```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Exponential Moving Average (EMA) of High-to-Low Range over a short window
    ema_short_window = 10  # Short window for EMA
    high_to_low_range_ema = high_to_low_range.ewm(span=ema_short_window, adjust=False).mean()
    
    # Combine Momentum and Volume
    factor_values = high_to_low_range_ema * df['volume']
    
    return factor_values
```
