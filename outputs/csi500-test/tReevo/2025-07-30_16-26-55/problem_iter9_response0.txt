```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    window_size = 10  # Example window size, can be adjusted
    high_to_low_range_std = high_to_low_range.rolling(window=window_size).std()
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend_window = 5  # Example window size for short-term trend, can be adjusted
    short_term_trend = df['close'].rolling(window=short_term_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (high_to_low_range / high_to_low_range_std) * short_term_trend - high_to_low_range_std
    
    # Adjust Combined Factor by Volume
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
