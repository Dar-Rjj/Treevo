```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_sma'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['close'] - df['10_sma']
    df['positive_short_term_momentum'] = df['short_term_momentum'].apply(lambda x: max(x, 0))
    df['5_day_positive_momentum'] = df['positive_short_term_momentum'].rolling(window=5).sum()

    # Calculate Long-Term Price Momentum
    df['30_sma'] = df['close'].rolling(window=30).mean()
    df['long_term_momentum'] = df['close'] - df['30_sma']
    df['positive_long_term_momentum'] = df['long_term_momentum'].apply(lambda x: max(x, 0))
    df['20_day_positive_momentum'] = df['positive_long_term_momentum'].rolling(window=20).sum()

    # Compute Price Breakout Indicator
    df['20_high'] = df['high'].rolling(window=20).max()
    df['20_low'] = df['low'].rolling(window=20).min()
    df['20_range'] = df['20_high'] - df['20_low']
    df['price_breakout'] = (df['close'] - df['20_low']) / df['20_range']

    # Volume Adjust Factor
    df['20_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjust_factor'] = df['volume'] / df['20_avg_volume']

    # Combine Factors
    df['short_term_adj_momentum'] = df['5_day_positive_momentum'] * df['volume_adjust_factor']
    df['long_term_adj_momentum'] = df['20_day_positive_momentum'] * df['volume_adjust_factor']
    df['combined_momentum'] = df['short_term_adj_momentum'] + df['long_term_adj_momentum']
    
    # Final factor
    df['factor'] = df['combined_momentum'] + df['price_breakout']
    
    return df['factor']
```
