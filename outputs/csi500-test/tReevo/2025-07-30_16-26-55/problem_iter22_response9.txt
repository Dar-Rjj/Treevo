```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    range_hl = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std_range = range_hl.rolling(window=20).std()
    rolling_mean_range = range_hl.rolling(window=20).mean()
    momentum = rolling_std_range / rolling_mean_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum * short_term_trend - rolling_std_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
