```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Open and Close-to-Low Ranges
    high_to_open_range = df['high'] - df['open']
    close_to_low_range = df['close'] - df['low']
    
    # Calculate Exponential Moving Average (EMA) of High-to-Low Range
    ema_high_to_low_range = high_to_low_range.ewm(span=10, adjust=False).mean()
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Combine with High-to-Open and Close-to-Low Ranges
    combined_momentum = (high_to_low_momentum + (high_to_open_range / high_to_low_range) + (close_to_low_range / high_to_low_range)) / 3
    
    # Calculate Short-Term and Long-Term Trends
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    long_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine Momentum and Trends
    combined_factor = combined_momentum * (short_term_trend / long_term_trend)
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    vol = log_returns.rolling(window=20).std().fillna(0)
    inverse_volatility = 1 / vol
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_factor = combined_factor * inverse_volatility
    
    # Incorporate Market Depth
    dollar_volume = df['amount'] / df['volume']
    market_depth_indicator = dollar_volume / df['close']
    
    # Adjust Final Factor by Market Depth
    final_factor = adjusted_factor * market_depth_indicator
    
    # Adjust Final Factor by Volume
    final_factor = final_factor * df['volume']
    
    return final_factor
```
