```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_window = 10  # define the window for rolling average, can be adjusted
    range_momentum = high_to_low_range.rolling(window=rolling_window).mean()
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    return adjusted_momentum
```
