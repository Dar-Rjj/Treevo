```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Price Momentum
    df['10_period_SMA_Close'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = (df['close'] - df['10_period_SMA_Close']).apply(lambda x: max(0, x))
    df['short_term_momentum_5_days'] = df['short_term_momentum'].rolling(window=5).sum()

    # Long-Term Price Momentum
    df['30_period_SMA_Close'] = df['close'].rolling(window=30).mean()
    df['long_term_momentum'] = (df['close'] - df['30_period_SMA_Close']).apply(lambda x: max(0, x))
    df['long_term_momentum_20_days'] = df['long_term_momentum'].rolling(window=20).sum()

    # Price Breakout Indicator
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['20_day_range'] = df['20_day_high'] - df['20_day_low']
    df['price_breakout_indicator'] = (df['close'] - df['20_day_low']) / df['20_day_range']

    # Volume Adjust Factor
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjust_factor'] = df['volume'] / df['20_day_avg_volume']

    # Historical Volatility
    df['20_day_std_close'] = df['close'].rolling(window=20).std()
    df['20_day_avg_close'] = df['close'].rolling(window=20).mean()
    df['historical_volatility'] = df['20_day_std_close'] / df['20_day_avg_close']

    # Market Sentiment
    df['20_day_high_low_diff_sum'] = (df['high'] - df['low']).rolling(window=20).sum()
    df['market_sentiment'] = df['20_day_high_low_diff_sum'] / df['20_day_avg_close']

    # Weighted Momentum
    weight_short_term = 0.6
    weight_long_term = 0.4
    df['weighted_momentum'] = (weight_short_term * df['short_term_momentum_5_days']) + (weight_long_term * df['long_term_momentum_20_days'])

    # Combined Price and Volume Factors
    df['combined_price_volume_factors'] = (df['weighted_momentum'] * df['volume_adjust_factor']) + df['price_breakout_indicator']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_price_volume_factors'] 
                                + df['historical_volatility'] 
                                + df['market_sentiment'])
    
    return df['final_alpha_factor'].dropna()
```
