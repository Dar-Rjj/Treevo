```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum using EMA
    range_momentum = high_to_low_range.ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term Trend using EMA of Close prices over a short window
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_term_trend
    
    # Calculate Volatility as the Rolling Standard Deviation of High-to-Low Range
    volatility = high_to_low_range.rolling(window=20).std()
    
    # Adjust Combined Factor by Volume and Inverse of Volatility
    adjusted_factor = (combined_factor * df['volume']) / volatility
    
    return adjusted_factor
```
