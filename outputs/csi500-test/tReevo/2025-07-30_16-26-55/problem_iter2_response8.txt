```python
import pandas as pd

def heuristics_v2(df, short_window=5, long_window=20):
    # Calculate Volume-Weighted High-to-Low Ratio
    df['high_low_ratio'] = (df['high'] - df['low']) / df['low']
    df['volume_weighted_high_low_ratio'] = df['high_low_ratio'] * df['volume']
    
    # Sum the volume-weighted high-low ratio over the specified period
    vw_hl_sum = df['volume_weighted_high_low_ratio'].rolling(window=short_window).sum()
    volume_sum = df['volume'].rolling(window=short_window).sum()
    
    # Calculate the Volume-Weighted High-to-Low Ratio
    df['vw_hl_ratio'] = vw_hl_sum / volume_sum
    
    # Calculate Short-term and Long-term Moving Averages
    df['short_ma'] = df['close'].rolling(window=short_window).mean()
    df['long_ma'] = df['close'].rolling(window=long_window).mean()
    
    # Determine Momentum
    df['momentum'] = df['short_ma'] - df['long_ma']
    df['momentum_direction'] = pd.cut(df['momentum'], bins=[-float('inf'), 0, float('inf')], labels=['Bearish', 'Bullish'])
    
    # Combine Factors
    df['alpha_factor'] = df['vw_hl_ratio']
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * 1.5 if row['momentum_direction'] == 'Bullish' else row['alpha_factor'] * 0.5 if row['momentum_direction'] == 'Bearish' else row['alpha_factor'], axis=1)
    
    return df['alpha_factor']
```
