```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (Rolling Standard Deviation)
    range_momentum = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend (Moving Average of Close Prices)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - high_to_low_range.rolling(window=20).std()
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
