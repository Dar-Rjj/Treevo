```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (rolling standard deviation)
    rolling_window = 30  # Set the rolling window size
    range_momentum = high_to_low_range.rolling(window=rolling_window).std()
    
    # Simplify with Short-Term Trends
    short_term_window = 5  # Set the short-term moving average window size
    short_term_trend = df['close'].rolling(window=short_term_window).mean()
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    # Combine the components into a final factor
    factor_values = adjusted_momentum / short_term_trend
    
    return factor_values
```
