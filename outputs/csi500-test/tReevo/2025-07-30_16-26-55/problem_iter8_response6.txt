```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate 10-day Rolling Standard Deviation and Mean of High-to-Low Range
    std_high_to_low_range = high_to_low_range.rolling(window=10).std()
    mean_high_to_low_range = high_to_low_range.rolling(window=10).mean()
    
    # Compute High-to-Low Range Momentum
    hl_range_momentum = std_high_to_low_range / mean_high_to_low_range
    
    # Adjust Momentum by Volume
    adjusted_momentum = hl_range_momentum * df['volume']
    
    return adjusted_momentum
```
