```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    ema_high_to_low_range = high_to_low_range.ewm(span=10, adjust=False).mean()
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Calculate Inverse Long-Term Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    long_term_volatility = log_returns.rolling(window=60).std(ddof=0)
    inverse_long_term_volatility = 1 / long_term_volatility
    
    # Adjust Combined Factor by Inverse Long-Term Volatility
    adjusted_combined_factor = combined_factor * inverse_long_term_volatility
    
    # Calculate Trade Volume Trend
    trade_volume_trend = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Adjust Final Factor by Trade Volume Trend
    final_factor = adjusted_combined_factor * (trade_volume_trend / df['volume'])
    
    return final_factor
```
