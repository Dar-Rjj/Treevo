```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation and Normalize by Rolling Mean of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=20).std()
    rolling_mean = high_to_low_range.rolling(window=20).mean()
    normalized_rolling_std = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Trend (Moving Average over a long window)
    long_term_trend = df['close'].rolling(window=60).mean()
    
    # Combine Momentum, Short-Term Trend, and Long-Term Trend
    combined_factor = (high_to_low_range * short_term_trend - normalized_rolling_std) / long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
