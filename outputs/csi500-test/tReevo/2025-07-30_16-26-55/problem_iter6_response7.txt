```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_sma'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['close'] - df['10_day_sma']
    df['positive_short_momentum'] = df['short_term_momentum'].where(df['short_term_momentum'] > 0, 0)
    df['sum_positive_short_momentum'] = df['positive_short_momentum'].rolling(window=5).sum()

    # Calculate Medium-Term Price Momentum
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['medium_term_momentum'] = df['close'] - df['20_day_sma']
    df['positive_medium_momentum'] = df['medium_term_momentum'].where(df['medium_term_momentum'] > 0, 0)
    df['sum_positive_medium_momentum'] = df['positive_medium_momentum'].rolling(window=10).sum()

    # Calculate Long-Term Price Momentum
    df['30_day_sma'] = df['close'].rolling(window=30).mean()
    df['long_term_momentum'] = df['close'] - df['30_day_sma']
    df['positive_long_momentum'] = df['long_term_momentum'].where(df['long_term_momentum'] > 0, 0)
    df['sum_positive_long_momentum'] = df['positive_long_momentum'].rolling(window=20).sum()

    # Refine Price Breakout Indicator
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['distance_from_high'] = (df['close'] / df['20_day_high']) * 100
    df['distance_from_low'] = (df['close'] / df['20_day_low']) * 100
    df['combined_distance'] = df['distance_from_high'] + (100 - df['distance_from_low'])

    # Integrate Trend Strength
    df['20_day_std'] = df['close'].rolling(window=20).std()
    df['ATR'] = df[['high', 'low', 'close']].diff(axis=1).iloc[:, [1, 2]].abs().max(axis=1)
    df['20_day_ATR'] = df['ATR'].rolling(window=20).mean()
    df['trend_strength'] = df['20_day_std'] + df['20_day_ATR']

    # Calculate Volume Adjust Factor
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjust_factor'] = df['volume'] / df['20_day_avg_volume']

    # Combine Factors
    df['short_term_vol_adj_momentum'] = df['sum_positive_short_momentum'] * df['volume_adjust_factor']
    df['medium_term_vol_adj_momentum'] = df['sum_positive_medium_momentum'] * df['volume_adjust_factor']
    df['long_term_vol_adj_momentum'] = df['sum_positive_long_momentum'] * df['volume_adjust_factor']
    df['combined_momentum'] = df['short_term_vol_adj_momentum'] + df['medium_term_vol_adj_momentum'] + df['long_term_vol_adj_momentum']
    df['final_momentum'] = df['combined_momentum'] * df['trend_strength']
    df['alpha_factor'] = df['final_momentum'] + df['combined_distance']

    # Apply a linear weighting factor
    df['final_alpha_factor'] = df['alpha_factor'] * 0.5  # Example linear weighting factor

    return df['final_alpha_factor']
```
