```python
import pandas as pd

def heuristics_v2(df, momentum_window=20, short_trend_window=5):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=momentum_window).std()
    rolling_mean = high_to_low_range.rolling(window=momentum_window).mean()
    range_momentum = (high_to_low_range - rolling_mean) / rolling_std
    
    # Calculate Short-Term Trend
    short_trend = df['close'].rolling(window=short_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_trend - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
