```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Exponential Moving Average (EMA) of High-to-Low Range
    ema_high_to_low_range = high_to_low_range.ewm(span=5, adjust=False).mean()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    rolling_std = log_returns.rolling(window=5).std()
    inverse_volatility = 1 / rolling_std
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_factor * df['volume']
    
    return final_factor
```
