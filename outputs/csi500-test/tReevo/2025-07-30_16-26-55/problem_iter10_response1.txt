```python
import pandas as pd

def heuristics_v2(data):
    # Calculate High-to-Low Range
    high_to_low_range = data['high'] - data['low']
    
    # Calculate 10-day Rolling Standard Deviation of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=10).std()
    
    # Calculate 10-day Rolling Mean of High-to-Low Range
    rolling_mean = high_to_low_range.rolling(window=10).mean()
    
    # Compute High-to-Low Range Momentum
    hl_momentum = rolling_std / rolling_mean
    
    # Adjust Momentum by Volume
    adjusted_momentum = hl_momentum * data['volume']
    
    return adjusted_momentum
```
