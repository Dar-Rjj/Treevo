```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    hl_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_mean = hl_range.rolling(window=10).mean()
    range_std = hl_range.rolling(window=10).std()
    range_momentum = (hl_range - range_mean) / range_std
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = range_momentum * df['volume']
    
    # Integrate Short-Term Trend Analysis
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_strength = short_term_ma - long_term_ma
    
    # Final alpha factor
    alpha_factor = volume_adjusted_momentum + trend_strength
    
    return alpha_factor
```
