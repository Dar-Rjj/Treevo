```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    rolling_std_high_to_low_range = high_to_low_range.rolling(window=20).std()
    
    # Compute Rolling Mean of High-to-Low Range
    rolling_mean_high_to_low_range = high_to_low_range.rolling(window=20).mean()
    
    # Calculate High-to-Low Range Momentum
    range_momentum = high_to_low_range / rolling_mean_high_to_low_range
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - rolling_std_high_to_low_range
    
    # Adjust Combined Factor by Volume
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
