```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Volume
    adjusted_range = high_to_low_range / df['volume']
    
    # Calculate Adjusted High-to-Low Range Momentum
    rolling_std = adjusted_range.rolling(window=20).std()
    rolling_mean = adjusted_range.rolling(window=20).mean()
    momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Trend
    long_term_trend = df['close'].rolling(window=50).mean()
    
    # Combine Momentum, Short-Term, and Long-Term Trends
    combined_factor = (momentum * short_term_trend) - long_term_trend
    
    # Final Adjustment
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
