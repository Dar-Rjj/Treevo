import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    range_series = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation and Mean of the Range
    window_size = 20  # You can adjust this window size
    rolling_std = range_series.rolling(window=window_size).std()
    rolling_mean = range_series.rolling(window=window_size).mean()
    
    # Compute High-to-Low Range Momentum
    momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend using a moving average over a short window
    trend_window = 5  # You can adjust this trend window size
    short_term_trend = df['close'].rolling(window=trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum * short_term_trend - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
