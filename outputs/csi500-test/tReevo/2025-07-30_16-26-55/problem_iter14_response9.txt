```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_window = 20
    high_to_low_momentum = high_to_low_range.rolling(window=rolling_window).std() / high_to_low_range.rolling(window=rolling_window).mean()
    
    # Calculate Short-Term Trend
    short_term_window = 5
    short_term_trend = df['close'].rolling(window=short_term_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Adjust by Volatility
    volatility = high_to_low_range.rolling(window=rolling_window).std()
    combined_factor_adjusted = combined_factor / volatility
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor_adjusted * df['volume']
    
    return adjusted_factor
```
