```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std_range = high_to_low_range.rolling(window=20).std()
    rolling_mean_range = high_to_low_range.rolling(window=20).mean()
    high_to_low_range_momentum = rolling_std_range / rolling_mean_range
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Trend (Moving Average over a long window)
    long_term_trend = df['close'].rolling(window=50).mean()
    
    # Combine Momentum, Short-Term, and Long-Term Trends
    combined_factor = (high_to_low_range_momentum * short_term_trend) - rolling_std_range + long_term_trend
    
    # Adjust for Volume Shocks
    daily_volume_change = df['volume'].pct_change().abs()
    volume_shock_threshold = 3  # Define a threshold to identify significant volume changes
    volume_shock_indicator = (daily_volume_change > volume_shock_threshold).astype(int)
    
    # Final Adjustment
    final_factor = combined_factor * (1 + volume_shock_indicator * (df['volume'] / df['volume'].rolling(window=20).mean()))
    
    return final_factor
```
