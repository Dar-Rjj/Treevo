```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=lookback_period).std()
    
    # Compute Rolling Mean of High-to-Low Range
    rolling_mean = high_to_low_range.rolling(window=lookback_period).mean()
    
    # Compute High-to-Low Range Momentum
    high_to_low_momentum = rolling_std / rolling_mean
    
    # Adjust Momentum by Volume
    adjusted_momentum = high_to_low_momentum * df['volume']
    
    return adjusted_momentum
```
