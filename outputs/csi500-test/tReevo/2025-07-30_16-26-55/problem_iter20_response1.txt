```python
import pandas as pd

def heuristics_v2(df, window=10, span=5):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (EMA of the range)
    ema_high_to_low_range = high_to_low_range.ewm(span=span, adjust=False).mean()
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    rolling_std_high_to_low_range = high_to_low_range.rolling(window=window).std()
    
    # Calculate Volatility-Adjusted Momentum
    volatility_adjusted_momentum = ema_high_to_low_range / rolling_std_high_to_low_range
    
    # Combine Momentum and Volatility-Adjusted Momentum
    combined_factor = ema_high_to_low_range * volatility_adjusted_momentum
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
