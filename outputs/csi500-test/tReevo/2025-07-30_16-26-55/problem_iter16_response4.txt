```python
import pandas as pd

def heuristics_v2(df, high_low_window=30, std_window=10, short_trend_window=5):
    # Calculate High-to-Low Range
    range_series = df['high'] - df['low']
    
    # Normalize High-to-Low Range
    rolling_mean_range = range_series.rolling(window=high_low_window).mean()
    normalized_range = range_series / rolling_mean_range
    
    # Calculate High-to-Low Range Momentum
    momentum = normalized_range.rolling(window=std_window).std()
    
    # Calculate Short-Term Trend
    short_trend = df['close'].rolling(window=short_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum * short_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
