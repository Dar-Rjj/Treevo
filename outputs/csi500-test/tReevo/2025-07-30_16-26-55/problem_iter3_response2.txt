```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Moving Average of High-to-Low Range
    short_term_window = 10  # Example: 10-day window
    sma_high_to_low_range = high_to_low_range.rolling(window=short_term_window).mean()
    
    # Calculate High-to-Low Range Momentum
    range_momentum = high_to_low_range - sma_high_to_low_range
    
    # Compute Rolling Standard Deviation of the Difference
    rolling_std = range_momentum.rolling(window=short_term_window).std()
    normalized_momentum = range_momentum / rolling_std
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = normalized_momentum * df['volume']
    
    return volume_adjusted_momentum
```
