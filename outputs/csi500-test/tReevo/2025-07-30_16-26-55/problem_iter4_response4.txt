```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate 20-day standard deviation of the High-to-Low Range
    std_high_to_low_range = high_to_low_range.rolling(window=20).std()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_range_momentum = high_to_low_range * std_high_to_low_range
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = high_to_low_range_momentum * df['volume']
    
    # Compute 20-day standard deviation of the Close prices
    close_price_std = df['close'].rolling(window=20).std()
    
    # Incorporate Volatility
    final_alpha_factor = volume_adjusted_momentum + close_price_std
    
    return final_alpha_factor
```
