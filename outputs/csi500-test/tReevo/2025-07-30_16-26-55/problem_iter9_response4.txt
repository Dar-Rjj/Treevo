```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=20).std()
    high_to_low_range_momentum = high_to_low_range / rolling_std
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_range_momentum * short_term_trend - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
