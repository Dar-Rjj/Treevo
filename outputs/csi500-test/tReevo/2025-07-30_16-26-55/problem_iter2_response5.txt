```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-period SMA of High and Low
    sma_high = df['high'].rolling(window=5).mean()
    sma_low = df['low'].rolling(window=5).mean()
    
    # Compute the difference between current High and 5-period SMA of High
    high_momentum = df['high'] - sma_high
    
    # Compute the difference between current Low and 5-period SMA of Low
    low_momentum = df['low'] - sma_low
    
    # Sum up the positive momentum over the last 5 days
    high_momentum_sum = high_momentum.rolling(window=5).apply(lambda x: x[x > 0].sum(), raw=True)
    low_momentum_sum = low_momentum.rolling(window=5).apply(lambda x: x[x > 0].sum(), raw=True)
    
    # Combine the High and Low momentum
    high_low_momentum = high_momentum_sum + low_momentum_sum
    
    # Calculate the average volume over the last 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Compute the ratio of today's volume to the 5-day average volume
    volume_adjust_factor = df['volume'] / avg_volume
    
    # Combine Factors
    factor_values = high_low_momentum * volume_adjust_factor
    
    return factor_values
```
