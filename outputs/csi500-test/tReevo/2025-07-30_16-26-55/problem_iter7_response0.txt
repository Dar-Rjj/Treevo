```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of Range
    range_volatility = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Combine Short-Term Momentum with Range Volatility
    combined_momentum = short_term_momentum * range_volatility
    
    # Adjust Combined Momentum by Volume
    adjusted_momentum = combined_momentum / df['volume']
    
    return adjusted_momentum
```
