import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation and Moving Average of the High-to-Low Range
    rolling_std_range = high_to_low_range.rolling(window=20).std()
    moving_avg_range = high_to_low_range.rolling(window=20).mean()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_momentum = (high_to_low_range - moving_avg_range) / rolling_std_range
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = high_to_low_momentum * df['volume']
    
    # Integrate Short-Term Trend
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_momentum * (short_term_ma - long_term_ma)
    
    return alpha_factor
