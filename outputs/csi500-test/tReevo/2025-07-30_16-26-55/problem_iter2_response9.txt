```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    short_term_momentum = df['close'] - short_term_sma
    short_term_momentum_pos = short_term_momentum.where(short_term_momentum > 0, 0)
    short_term_momentum_sum = short_term_momentum_pos.rolling(window=5).sum()

    # Calculate Long-Term Price Momentum
    long_term_sma = df['close'].rolling(window=30).mean()
    long_term_momentum = df['close'] - long_term_sma
    long_term_momentum_pos = long_term_momentum.where(long_term_momentum > 0, 0)
    long_term_momentum_sum = long_term_momentum_pos.rolling(window=20).sum()

    # Compute Price Breakout Indicator
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    range_20d = high_20d - low_20d
    price_breakout = (df['close'] - low_20d) / range_20d

    # Volume Adjust Factor
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_adjust = df['volume'] / avg_volume_20d

    # Incorporate Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    avg_close_20d = df['close'].rolling(window=20).mean()
    volatility = close_20d_std / avg_close_20d

    # Weighted Momentum
    weight_short_term = 0.6
    weight_long_term = 0.4
    weighted_momentum = (short_term_momentum_sum * weight_short_term) + (long_term_momentum_sum * weight_long_term)

    # Combine All Factors
    alpha_factor = (weighted_momentum * volume_adjust) + price_breakout + volatility

    return alpha_factor
```
