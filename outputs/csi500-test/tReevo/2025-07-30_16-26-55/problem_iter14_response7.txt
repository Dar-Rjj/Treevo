```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Volatility
    range_volatility = high_to_low_range.rolling(window=5).std()
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Trend (Moving Average over a long window)
    long_term_trend = df['close'].rolling(window=60).mean()
    
    # Combine Volatility, Short-Term Trend, and Long-Term Trend
    combined_factor = (range_volatility * short_term_trend) - long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor / df['volume']
    
    return adjusted_factor
```
