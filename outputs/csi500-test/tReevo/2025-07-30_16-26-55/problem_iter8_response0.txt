```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    roll_std = high_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Momentum
    short_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate Medium-Term Momentum
    medium_momentum = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Momentum
    long_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Combine Momentum Indicators
    combined_momentum = short_momentum + medium_momentum + long_momentum - roll_std
    
    # Calculate Volume Adjust Factor
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjust_factor = df['volume'] / avg_volume
    
    # Adjust Combined Factor by Volume
    adjusted_combined_factor = combined_momentum * volume_adjust_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_combined_factor
    
    return alpha_factor
```
