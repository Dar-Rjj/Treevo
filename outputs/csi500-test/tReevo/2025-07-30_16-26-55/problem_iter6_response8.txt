```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling 20-Day High-to-Low Range Momentum
    rolling_20_day_sum = high_to_low_range.rolling(window=20).sum()
    range_momentum = high_to_low_range / rolling_20_day_sum
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    return adjusted_momentum
```
