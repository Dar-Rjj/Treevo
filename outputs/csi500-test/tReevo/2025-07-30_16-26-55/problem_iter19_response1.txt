```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Exponential Moving Average (EMA) of High-to-Low Range
    ema_high_to_low_range = high_to_low_range.ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = ema_high_to_low_range * short_term_trend
    
    # Subtract Standard Deviation of High-to-Low Range
    std_high_to_low_range = high_to_low_range.rolling(window=5).std()
    combined_factor -= std_high_to_low_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
