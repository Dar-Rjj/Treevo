import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Volatility (20-day Rolling Standard Deviation of Close Prices)
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate High-to-Low Range Momentum (14-day Rolling Standard Deviation of High-to-Low Range)
    hlr_momentum = high_to_low_range.rolling(window=14).std()
    
    # Adjust High-to-Low Range Momentum by Volatility
    volatility_adjusted_hlr_momentum = hlr_momentum / volatility
    
    # Adjust Momentum by Volume
    adjusted_hlr_momentum = volatility_adjusted_hlr_momentum * df['volume']
    
    # Generate Alpha Factor Signal based on the adjusted momentum
    alpha_factor_signal = adjusted_hlr_momentum
    
    return alpha_factor_signal
