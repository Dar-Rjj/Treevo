```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Volatility (Rolling Standard Deviation of Range)
    rolling_window = 10
    range_volatility = high_to_low_range.rolling(window=rolling_window).std()
    
    # Adjust Volatility by Volume
    adjusted_volatility = range_volatility * df['volume']
    
    return adjusted_volatility
```
