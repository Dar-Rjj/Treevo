```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (EMA of the range)
    ema_window = 10  # Example window, can be adjusted
    high_to_low_range_momentum = high_to_low_range.ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Short-Term Trend (EMA of Close prices over a short window)
    short_trend_window = 5  # Example short window, can be adjusted
    short_term_trend = df['close'].ewm(span=short_trend_window, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_range_momentum * short_term_trend
    
    # Calculate Volatility (Rolling Standard Deviation of High-to-Low Range)
    volatility_window = 20  # Example window for volatility, can be adjusted
    volatility = high_to_low_range.rolling(window=volatility_window).std()
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume'] / volatility
    
    return adjusted_factor
```
