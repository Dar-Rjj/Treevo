```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (Rolling Standard Deviation)
    short_term_window = 5
    range_momentum = high_to_low_range.rolling(window=short_term_window).std()
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    return adjusted_momentum
```
