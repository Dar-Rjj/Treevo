```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    short_term_momentum = (df['close'] - short_term_sma).clip(lower=0)
    short_term_momentum_sum = short_term_momentum.rolling(window=5).sum()

    # Medium-Term Price Momentum
    medium_term_sma = df['close'].rolling(window=20).mean()
    medium_term_momentum = (df['close'] - medium_term_sma).clip(lower=0)
    medium_term_momentum_sum = medium_term_momentum.rolling(window=10).sum()

    # Long-Term Price Momentum
    long_term_sma = df['close'].rolling(window=30).mean()
    long_term_momentum = (df['close'] - long_term_sma).clip(lower=0)
    long_term_momentum_sum = long_term_momentum.rolling(window=20).sum()

    # Refine Price Breakout Indicator
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    breakout_high = (df['close'] / high_20 - 1) * 100
    breakout_low = (df['close'] / low_20 - 1) * 100
    breakout_indicator = breakout_high + breakout_low

    # Integrate Trend Strength
    rolling_std = df['close'].rolling(window=20).std()
    tr = df['high'] - df['low']
    atr = tr.rolling(window=20).mean()
    trend_strength = rolling_std + atr

    # Calculate Volume Adjust Factor
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjust_factor = df['volume'] / avg_volume_20

    # Combine Factors
    combined_momentum = (short_term_momentum_sum * volume_adjust_factor +
                         medium_term_momentum_sum * volume_adjust_factor +
                         long_term_momentum_sum * volume_adjust_factor)

    final_alpha = (combined_momentum * trend_strength + breakout_indicator)

    # Apply a Linear Weighting Factor
    linear_weighting_factor = 0.1
    final_alpha = final_alpha * linear_weighting_factor

    return final_alpha
```
