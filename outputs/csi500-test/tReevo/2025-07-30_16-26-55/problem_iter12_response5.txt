```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    window = 20  # Define the rolling window size
    range_std = high_to_low_range.rolling(window=window).std()
    range_mean = high_to_low_range.rolling(window=window).mean()
    range_momentum = (range_std / range_mean).fillna(0)
    
    # Calculate Short-Term Trend using Close prices
    short_term_window = 5  # Define the short-term moving average window size
    short_term_trend = df['close'].rolling(window=short_term_window).mean().fillna(0)
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - range_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
