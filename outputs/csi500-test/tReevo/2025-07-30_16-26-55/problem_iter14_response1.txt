```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=50):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=short_window).std()
    rolling_mean = high_to_low_range.rolling(window=short_window).mean()
    range_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term and Long-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()
    long_term_trend = df['close'].rolling(window=long_window).mean()
    
    # Combine Momentum, Short-Term Trend, and Long-Term Trend
    combined_factor = (range_momentum * short_term_trend - rolling_std) / long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
