```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    short_momentum = (df['close'] - short_term_sma).apply(lambda x: x if x > 0 else 0)
    short_momentum_sum = short_momentum.rolling(window=5).sum()

    # Calculate Medium-Term Price Momentum
    medium_term_sma = df['close'].rolling(window=20).mean()
    medium_momentum = (df['close'] - medium_term_sma).apply(lambda x: x if x > 0 else 0)
    medium_momentum_sum = medium_momentum.rolling(window=10).sum()

    # Calculate Long-Term Price Momentum
    long_term_sma = df['close'].rolling(window=30).mean()
    long_momentum = (df['close'] - long_term_sma).apply(lambda x: x if x > 0 else 0)
    long_momentum_sum = long_momentum.rolling(window=20).sum()

    # Compute Price Breakout Indicator
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    breakout_high = (df['close'] - high_20) / high_20
    breakout_low = (df['close'] - low_20) / low_20
    breakout_indicator = breakout_high + breakout_low

    # Calculate Volume Adjust Factor
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjust_factor = df['volume'] / avg_volume_20

    # Combine Factors
    short_momentum_adj = short_momentum_sum * volume_adjust_factor
    medium_momentum_adj = medium_momentum_sum * volume_adjust_factor
    long_momentum_adj = long_momentum_sum * volume_adjust_factor
    combined_momentum = short_momentum_adj + medium_momentum_adj + long_momentum_adj
    combined_factors = combined_momentum + breakout_indicator

    # Apply Linear Weighting
    weight_short = 0.3
    weight_medium = 0.2
    weight_long = 0.2
    weight_breakout = 0.3
    weighted_factors = (short_momentum_adj * weight_short +
                        medium_momentum_adj * weight_medium +
                        long_momentum_adj * weight_long +
                        breakout_indicator * weight_breakout)

    # Final Alpha Factor
    final_alpha_factor = weighted_factors

    return final_alpha_factor
```
