```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    short_window = 10  # Short rolling window
    range_momentum = high_to_low_range.rolling(window=short_window).std()
    
    # Integrate Volume into Momentum
    integrated_momentum = range_momentum * df['volume']
    
    # Calculate Short-Term Trend
    short_trend_window = 20  # Short moving average window
    short_term_trend = df['close'].rolling(window=short_trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = integrated_momentum * short_term_trend
    
    # Calculate Long-Term Trend
    long_trend_window = 60  # Long moving average window
    long_term_trend = df['close'].rolling(window=long_trend_window).mean()
    
    # Adjust Combined Factor by Long-Term Trend
    adjusted_factor = combined_factor / long_term_trend
    
    return adjusted_factor
```
