```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=30):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=long_window).std()
    
    # Compute Rolling Mean of High-to-Low Range
    rolling_mean = high_to_low_range.rolling(window=long_window).mean()
    
    # Normalize by Rolling Mean of High-to-Low Range
    high_to_low_momentum = (high_to_low_range - rolling_mean) / rolling_std
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend - rolling_std
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
