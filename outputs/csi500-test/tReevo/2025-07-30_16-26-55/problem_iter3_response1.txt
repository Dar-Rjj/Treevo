```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Define rolling window
    window_size = 5

    # Compute Rolling Standard Deviation and Mean of High-to-Low Range
    std_high_low_range = high_low_range.rolling(window=window_size).std()
    mean_high_low_range = high_low_range.rolling(window=window_size).mean()

    # Adjust Momentum by Volume
    momentum_adjusted_by_volume = (std_high_low_range / mean_high_low_range) * df['volume']

    return momentum_adjusted_by_volume
```
