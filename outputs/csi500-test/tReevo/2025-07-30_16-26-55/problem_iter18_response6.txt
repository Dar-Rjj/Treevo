```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=20, momentum_weight=0.5, trend_weight=0.3, open_close_weight=0.1, high_low_weight=0.1):
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation and Mean of High-to-Low Range
    std_high_to_low_range = df['high_to_low_range'].rolling(window=window).std()
    mean_high_to_low_range = df['high_to_low_range'].rolling(window=window).mean()
    
    # Calculate High-to-Low Range Momentum
    df['range_momentum'] = std_high_to_low_range / mean_high_to_low_range
    
    # Calculate Short-Term Trend (Exponential Moving Average over a short window)
    short_window = 5
    df['short_term_trend'] = df['close'].ewm(span=short_window, adjust=False).mean()
    
    # Integrate Open and Close Price Changes
    df['open_close_change'] = (df['close'] - df['open']) / df['open']
    
    # Integrate High and Low Price Changes
    df['high_low_change'] = (df['high'] - df['low']) / df['low']
    
    # Combine Momentum, Short-Term Trend, and Price Changes
    combined_factor = (
        df['range_momentum'] * momentum_weight +
        df['short_term_trend'] * trend_weight +
        df['open_close_change'] * open_close_weight +
        df['high_low_change'] * high_low_weight
    )
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * np.sqrt(df['volume'])
    
    return adjusted_factor
```
