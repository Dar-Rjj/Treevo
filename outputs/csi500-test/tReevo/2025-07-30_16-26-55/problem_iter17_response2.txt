```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range (Momentum)
    window_momentum = 10  # You can adjust the window size
    high_to_low_range_momentum = high_to_low_range.rolling(window=window_momentum).std()
    
    # Calculate Short-Term Trend using Close prices
    short_window = 5  # You can adjust the short-term window size
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_range_momentum * short_term_trend - high_to_low_range_momentum
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
