```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate EMA of High-to-Low Range
    ema_high_to_low_range = high_to_low_range.ewm(span=10, adjust=False).mean()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend (EMA over a short window)
    short_term_trend = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Trend (EMA over a long window)
    long_term_trend = df['close'].ewm(span=50, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_momentum_trend = high_to_low_momentum * short_term_trend
    
    # Combine Short-Term and Long-Term Trends
    combined_trends = short_term_trend - long_term_trend
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    rolling_std = log_returns.rolling(window=30).std()
    inverse_volatility = 1 / rolling_std
    
    # Adjust Combined Momentum and Trend by Inverse Volatility
    adjusted_combined_factor = combined_momentum_trend * inverse_volatility
    
    # Final Adjustment by Volume
    final_adjusted_factor = adjusted_combined_factor * df['volume']
    
    # Final Factor
    final_factor = final_adjusted_factor - combined_trends
    
    return final_factor
```
