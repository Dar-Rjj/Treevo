```python
import pandas as pd

def heuristics_v2(df, window_range=10, window_momentum=5, window_trend=5):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Mean of High-to-Low Range
    rolling_mean_range = high_to_low_range.rolling(window=window_range).mean()
    
    # Calculate Dynamic High-to-Low Range
    dynamic_high_to_low_range = high_to_low_range / rolling_mean_range
    
    # Calculate Rolling Standard Deviation of Dynamic High-to-Low Range
    rolling_std_dynamic_range = dynamic_high_to_low_range.rolling(window=window_momentum).std()
    
    # Calculate Short-Term Trend (Moving Average of Close Prices)
    short_term_trend = df['close'].rolling(window=window_trend).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (dynamic_high_to_low_range * short_term_trend) - rolling_std_dynamic_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
