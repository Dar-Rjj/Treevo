```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate High-to-Low Range
    high_to_low_range = data['high'] - data['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=10).std()
    rolling_mean = high_to_low_range.rolling(window=10).mean()
    high_to_low_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_window = 5
    short_term_trend = data['close'].rolling(window=short_window).mean()
    
    # Calculate Long-Term Trend
    long_window = 20
    long_term_trend = data['close'].rolling(window=long_window).mean()
    
    # Combine Momentum and Trends
    combined_factor = (high_to_low_momentum * short_term_trend - rolling_std) * long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * data['volume']
    
    return adjusted_factor
```
