```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range (20-period)
    high_low_range_std = high_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Momentum (difference between current Close and 5-period SMA of Close)
    close_sma_5 = df['close'].rolling(window=5).mean()
    short_term_momentum = df['close'] - close_sma_5
    
    # Combine Volatility and Momentum
    combined_vol_mom = high_low_range_std - short_term_momentum
    
    # Calculate Volume Adjust Factor (ratio of today's volume to the 20-day average volume)
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjust_factor = df['volume'] / avg_volume_20
    
    # Final Alpha Factor (Multiply the Combined Volatility and Momentum by the Volume Adjust Factor)
    final_alpha_factor = combined_vol_mom * volume_adjust_factor
    
    return final_alpha_factor
```
