```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    roll_std = high_low_range.rolling(window=20).std()
    roll_mean = high_low_range.rolling(window=20).mean()
    range_momentum = roll_std / roll_mean
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Trend (Moving Average over a long window)
    long_term_trend = df['close'].rolling(window=50).mean()
    
    # Combine Momentum, Short-Term, and Long-Term Trends
    combined_factor = (range_momentum * short_term_trend) - long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
