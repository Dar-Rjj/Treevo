import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['5Day_Avg_Range'] = df['High_Low_Range'].rolling(window=5).mean()

    # Incorporate Volume Trend
    df['Volume_5Day_Mean'] = df['volume'].rolling(window=5).mean()

    # Integrate Volatility
    df['Close_20Day_Std'] = df['close'].rolling(window=20).std()
    df['Close_20Day_Std_5Day_Mean'] = df['Close_20Day_Std'].rolling(window=5).mean()

    # Combine Price, Volume, and Volatility Information
    df['Combined_Factor'] = (df['High_Low_Range'] * df['volume']) / (df['Volume_5Day_Mean'] * df['Close_20Day_Std_5Day_Mean'])

    # Smooth the Combined Factor
    df['Smoothed_Combined_Factor'] = df['Combined_Factor'].rolling(window=10).mean()

    # Return the final alpha factor
    return df['Smoothed_Combined_Factor']
