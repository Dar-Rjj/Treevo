```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Volatility
    range_volatility = high_to_low_range.rolling(window=10).std() / high_to_low_range.rolling(window=10).mean()
    
    # Calculate Short-Term Trend (5-day Moving Average of Close prices)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_volatility * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
