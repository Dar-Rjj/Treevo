```python
import pandas as pd

def heuristics_v2(df, window=20):
    # Compute Rolling Average of Close Prices
    close_rolling_avg = df['close'].rolling(window=window).mean()
    
    # Compute Price Momentum
    price_momentum = df['close'] - close_rolling_avg
    
    # Compute Rolling Average of Volumes
    volume_rolling_avg = df['volume'].rolling(window=window).mean()
    
    # Compute Volume Momentum
    volume_momentum = df['volume'] - volume_rolling_avg
    
    # Weight by Volume: Multiply Price Momentum by Volume
    weighted_momentum = price_momentum * volume_momentum
    
    # Normalize by Current Volume: Avoid Division By Zero
    current_volume = df['volume'].replace(0, 1e-6)  # Small value to avoid division by zero
    normalized_momentum = weighted_momentum / current_volume
    
    return normalized_momentum
```
