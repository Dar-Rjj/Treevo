```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, decay_factor=0.95, volatility_scaling=1.0):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Weighted Moving Average of Log Returns
    df['volume_scaled'] = df['volume'] / df['volume'].rolling(window=lookback_period).max()
    df['weighted_log_return'] = df['log_return'] * df['volume_scaled']
    
    weighted_log_return_ma = df['weighted_log_return'].rolling(window=lookback_period).mean()
    
    # Calculate Exponential Moving Average of Weights
    df['exponential_weight'] = df['volume_scaled'].ewm(alpha=1 - decay_factor, adjust=False).mean()
    
    # Compute Final Alpha Factor
    alpha_factor = weighted_log_return_ma * df['exponential_weight']
    
    # Adjust for Volatility
    log_return_volatility = df['log_return'].rolling(window=lookback_period).std()
    alpha_factor_adjusted = alpha_factor / (log_return_volatility + 1e-8) * volatility_scaling
    
    return alpha_factor_adjusted
```
