```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum with Weighted EMA
    ema_range = high_to_low_range.ewm(span=20, adjust=False).mean()
    weighted_ema_range = ema_range * (df['amount'] / df['amount'].rolling(window=20).mean())
    momentum = weighted_ema_range / high_to_low_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].ewm(span=10, adjust=False).mean() / df['close']
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum * short_term_trend
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close']).diff()
    volatility = log_returns.rolling(window=20).std()
    inverse_volatility = 1 / volatility
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_factor * df['volume']
    
    # Integrate Cross-Asset Signals
    # Assuming a basket of related assets is available in the DataFrame 'df_related'
    # For simplicity, we will use a placeholder for the cross-asset correlation
    # In practice, this would be calculated using the close prices of related assets
    cross_asset_correlation = 0.5  # Placeholder value
    final_factor *= cross_asset_correlation
    
    return final_factor
```
