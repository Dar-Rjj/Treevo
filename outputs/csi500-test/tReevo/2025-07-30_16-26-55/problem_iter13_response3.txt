```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate High-to-Low Range Momentum
    short_window = 5  # Define a short window for the rolling sum and mean
    rolling_sum_high_low_range = high_low_range.rolling(window=short_window).sum()
    rolling_mean_high_low_range = high_low_range.rolling(window=short_window).mean()
    high_low_momentum = rolling_sum_high_low_range / rolling_mean_high_low_range

    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()

    # Combine Momentum and Short-Term Trend
    combined_factor = (high_low_momentum * short_term_trend) - rolling_sum_high_low_range

    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']

    return adjusted_factor
```
