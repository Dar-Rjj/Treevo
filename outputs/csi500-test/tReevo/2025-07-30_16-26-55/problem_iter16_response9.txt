```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    window_size = 20  # You can adjust the window size based on your needs
    high_to_low_momentum = high_to_low_range.rolling(window=window_size).std()
    
    # Calculate Short-Term Trend
    short_window = 5  # Short-term moving average window
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend - high_to_low_momentum
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
