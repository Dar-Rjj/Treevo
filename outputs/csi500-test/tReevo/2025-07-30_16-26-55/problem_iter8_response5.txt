```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (Rolling Standard Deviation)
    range_momentum = high_to_low_range.rolling(window=20).std()
    
    # Smooth Volume using Exponential Moving Average (EMA)
    smoothed_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Adjust Momentum by Smoothed Volume
    adjusted_momentum = range_momentum * smoothed_volume
    
    return adjusted_momentum
```
