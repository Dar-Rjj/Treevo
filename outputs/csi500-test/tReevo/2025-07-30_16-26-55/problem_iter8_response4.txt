```python
import pandas as pd

def heuristics_v2(df, window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Volatility
    range_volatility = high_to_low_range.rolling(window=window).std()
    
    # Calculate High-to-Low Range Momentum
    range_momentum = high_to_low_range / range_volatility
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    return adjusted_momentum
```
