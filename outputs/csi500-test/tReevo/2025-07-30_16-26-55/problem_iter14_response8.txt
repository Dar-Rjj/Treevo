```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=30):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Normalize High-to-Low Range by Rolling Mean of High-to-Low Range
    rolling_mean_range = high_to_low_range.rolling(window=long_window).mean()
    normalized_range = high_to_low_range / rolling_mean_range
    
    # Calculate Rolling Standard Deviation of Normalized High-to-Low Range
    rolling_std_normalized_range = normalized_range.rolling(window=short_window).std()
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (normalized_range * short_term_trend) - rolling_std_normalized_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
