```python
import pandas as pd

def heuristics_v2(df, short_window=10, rolling_std_window=30):
    """
    Generate a high-to-low range momentum adjusted by volume factor.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    - short_window: int, window size for the short-term moving average
    - rolling_std_window: int, window size for the rolling standard deviation
    
    Returns:
    - pandas Series indexed by date representing the factor values
    """
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Calculate High-to-Low Range Momentum
    hl_momentum = high_to_low_range.rolling(window=rolling_std_window).std()

    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()

    # Combine Momentum and Short-Term Trend
    combined_factor = hl_momentum * short_term_trend

    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']

    return adjusted_factor
```
