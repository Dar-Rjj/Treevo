```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    rolling_std = high_to_low_range.rolling(window=30).std()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_range_momentum = high_to_low_range / rolling_std
    
    # Calculate Short-Term Trend using a 10-day Moving Average of Close prices
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_range_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
