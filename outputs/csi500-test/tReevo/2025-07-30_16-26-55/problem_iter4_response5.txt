```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['range'] = df['high'] - df['low']
    
    # Calculate the rolling standard deviation of the range
    window = 10
    df['range_momentum'] = df['range'].rolling(window=window).std()
    
    # Adjust momentum by volume
    df['adjusted_momentum'] = df['range_momentum'] * df['volume']
    
    return df['adjusted_momentum']
```
