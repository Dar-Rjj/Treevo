```python
import pandas as pd
import numpy as np

def heuristics_v2(df, ema_span=10, trend_span=5, vol_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Exponential Moving Average (EMA) of High-to-Low Range
    ema_high_to_low_range = high_to_low_range.ewm(span=ema_span).mean()
    
    # Calculate High-to-Low Range Momentum
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].ewm(span=trend_span).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_dev = log_returns.rolling(window=vol_window).std()
    inverse_volatility = 1 / std_dev
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_factor * df['volume']
    
    return final_factor
```
