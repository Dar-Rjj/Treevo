```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window_range=10, window_ema=5, weight_momentum=0.4, weight_trend=0.3, weight_open_close=0.3):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = df['high_low_range'].rolling(window=window_range).std()
    rolling_mean = df['high_low_range'].rolling(window=window_range).mean()
    df['range_momentum'] = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend (Exponential Moving Average of Close Prices)
    df['short_term_trend'] = df['close'].ewm(span=window_ema, adjust=False).mean()
    
    # Integrate Open and Close Price Changes
    df['open_close_change'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum, Short-Term Trend, and Open-Close Change
    combined_factor = (
        weight_momentum * df['range_momentum'] +
        weight_trend * df['short_term_trend'] +
        weight_open_close * df['open_close_change']
    )
    
    # Adjust Combined Factor by Volume
    df['factor'] = combined_factor * np.sqrt(df['volume'])
    
    return df['factor']
```
