```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=5, ema_span=10, vol_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (EMA of High-to-Low Range divided by High-to-Low Range)
    ema_high_to_low_range = high_to_low_range.ewm(span=ema_span, adjust=False).mean()
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend (EMA of Close prices)
    short_term_trend = df['close'].ewm(span=short_window, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Calculate Inverse Volatility (Inverse of the Rolling Standard Deviation of logarithmic returns)
    log_returns = np.log(df['close']).diff()
    rolling_std = log_returns.rolling(window=vol_window).std()
    inverse_volatility = 1 / rolling_std
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_combined_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_combined_factor * df['volume']
    
    return final_factor
```
