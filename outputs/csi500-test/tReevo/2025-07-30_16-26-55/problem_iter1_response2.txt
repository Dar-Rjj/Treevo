```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Ratio
    high_to_low_ratio = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Weight
    volume_weighted_ratio = high_to_low_ratio * df['volume']
    
    # Incorporate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    momentum = long_term_ma - short_term_ma
    
    # Categorize Momentum
    def categorize_momentum(m):
        if m > 0:
            return 'Bullish'
        elif m < 0:
            return 'Bearish'
        else:
            return 'Neutral'
    momentum_category = momentum.apply(categorize_momentum)
    
    # Combine Factors
    factor = volume_weighted_ratio.copy()
    factor[momentum_category == 'Bullish'] *= 1.1
    factor[momentum_category == 'Bearish'] *= 0.9
    
    return factor
```
