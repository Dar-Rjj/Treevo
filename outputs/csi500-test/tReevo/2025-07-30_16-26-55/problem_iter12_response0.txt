```python
import pandas as pd

def heuristics_v2(df, range_window=20, trend_window=5):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std_range = high_to_low_range.rolling(window=range_window).std()
    rolling_mean_range = high_to_low_range.rolling(window=range_window).mean()
    high_to_low_momentum = rolling_std_range / rolling_mean_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=trend_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend - rolling_std_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
