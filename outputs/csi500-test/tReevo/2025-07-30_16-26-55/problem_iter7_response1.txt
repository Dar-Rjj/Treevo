```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=30):
    """
    Calculate a high-to-low range momentum factor adjusted by volume.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as dates.
    - short_window: Short-term rolling window for calculating momentum.
    - long_window: Long-term rolling window for calculating momentum.
    
    Returns:
    - pandas Series indexed by date representing the factor values.
    """
    # Calculate High-to-Low Range
    df['range'] = df['high'] - df['low']

    # Calculate Short-Term High-to-Low Range Momentum
    short_std = df['range'].rolling(window=short_window).std()
    short_mean = df['range'].rolling(window=short_window).mean()
    df['short_momentum'] = (df['range'] - short_mean) / short_std

    # Calculate Long-Term High-to-Low Range Momentum
    long_std = df['range'].rolling(window=long_window).std()
    long_mean = df['range'].rolling(window=long_window).mean()
    df['long_momentum'] = (df['range'] - long_mean) / long_std

    # Combine Short-Term and Long-Term Momentum
    df['combined_momentum'] = 0.5 * df['short_momentum'] + 0.5 * df['long_momentum']

    # Adjust Momentum by Volume
    df['factor'] = df['combined_momentum'] * df['volume']

    return df['factor']
```
