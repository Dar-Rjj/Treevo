```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Mean and Standard Deviation of High-to-Low Range
    window = 10
    rolling_mean = high_to_low_range.rolling(window=window).mean()
    rolling_std = high_to_low_range.rolling(window=window).std()
    
    # Compute Z-Score of High-to-Low Range
    z_score = (high_to_low_range - rolling_mean) / rolling_std
    
    # Adjust Momentum by Volume
    adjusted_momentum = z_score * df['volume']
    
    return adjusted_momentum
```
