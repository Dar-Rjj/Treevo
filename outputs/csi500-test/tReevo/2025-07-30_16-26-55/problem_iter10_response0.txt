```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_momentum = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Trend
    long_term_trend = df['close'].rolling(window=60).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - range_momentum
    
    # Integrate Long-Term Trend
    integrated_factor = combined_factor - long_term_trend
    
    # Adjust for Volume Changes
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_factor = integrated_factor * volume_change
    
    # Final Adjustment by Volume
    final_factor = adjusted_factor * df['volume']
    
    return final_factor
```
