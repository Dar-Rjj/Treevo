```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=10, momentum_window=30):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    std_high_to_low_range = high_to_low_range.rolling(window=momentum_window).std()
    
    # Normalize by Rolling Mean of High-to-Low Range
    mean_high_to_low_range = high_to_low_range.rolling(window=momentum_window).mean()
    normalized_std_high_to_low_range = std_high_to_low_range / mean_high_to_low_range
    
    # Calculate Short-Term Trend using Close prices
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = normalized_std_high_to_low_range * short_term_trend - std_high_to_low_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
