```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=20).std()
    rolling_mean = high_to_low_range.rolling(window=20).mean()
    high_to_low_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Trend
    long_term_trend = df['close'].rolling(window=20).mean()
    
    # Combine Momentum, Short-Term, and Long-Term Trends
    combined_factor = (high_to_low_momentum * short_term_trend) - long_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
