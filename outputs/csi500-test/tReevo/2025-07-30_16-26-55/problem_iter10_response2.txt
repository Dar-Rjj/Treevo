```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Integrate Volume Early
    volume_adjusted_range = high_to_low_range * df['volume']
    
    # Calculate High-to-Low Range Momentum
    window = 5  # Define the rolling window size
    rolling_std = volume_adjusted_range.rolling(window=window).std()
    rolling_mean = volume_adjusted_range.rolling(window=window).mean()
    range_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_window = 5  # Define the short moving average window
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_term_trend - rolling_std
    
    return combined_factor
```
