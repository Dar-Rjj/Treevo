```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate 5-day High-to-Low Range Momentum
    rolling_std = high_low_range.rolling(window=5).std()
    rolling_mean = high_low_range.rolling(window=5).mean()
    high_low_range_momentum = rolling_std / rolling_mean

    # Calculate Short-Term Price Momentum
    sma_5_close = df['close'].rolling(window=5).mean()
    short_term_price_momentum = df['close'] - sma_5_close

    # Adjust Momentum by Volume
    volume_adjusted_momentum = high_low_range_momentum * df['volume']

    # Combine Factors
    final_alpha_factor = short_term_price_momentum + volume_adjusted_momentum

    return final_alpha_factor
```
