```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    window = 20  # Define the rolling window size
    high_to_low_momentum = high_to_low_range.rolling(window=window).mean()
    
    # Calculate Volatility-Adjusted Momentum
    high_to_low_volatility = high_to_low_range.rolling(window=window).std()
    volatility_adjusted_momentum = high_to_low_momentum / high_to_low_volatility
    
    # Combine Momentum and Volatility-Adjusted Momentum
    combined_factor = high_to_low_momentum * volatility_adjusted_momentum
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
