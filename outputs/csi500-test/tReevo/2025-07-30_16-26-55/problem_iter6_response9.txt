```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate 20-day Rolling Standard Deviation of Range
    std_20 = high_to_low_range.rolling(window=20).std()
    
    # Calculate 20-day Moving Average of Range
    ma_20 = high_to_low_range.rolling(window=20).mean()
    
    # Adjust Momentum by Volume
    adjusted_momentum = (std_20 * df['volume']) / ma_20
    
    # Integrate Short-Term Trend
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Determine the trend
    trend_signal = (sma_5 > sma_20).astype(int) - (sma_5 < sma_20).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum * trend_signal
    
    return final_alpha_factor
```
