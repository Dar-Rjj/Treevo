```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    range_hl = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum using a short rolling window (e.g., 5 days)
    momentum_hl = range_hl.rolling(window=5).sum()
    
    # Calculate Short-Term Trend using Close prices with a short moving average window (e.g., 5 days)
    trend_short_term = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum_hl * trend_short_term
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor / df['volume']
    
    return adjusted_factor
```
