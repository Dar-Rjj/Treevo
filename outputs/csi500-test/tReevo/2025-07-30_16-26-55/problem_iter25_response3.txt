```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    ema_high_low_range = high_low_range.ewm(span=20, adjust=False).mean()
    high_low_momentum = high_low_range / ema_high_low_range
    
    # Calculate Short-Term Trend
    short_term_ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_low_momentum * (df['close'] / short_term_ema_close)
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    rolling_std = log_returns.rolling(window=20).std()
    inverse_volatility = 1 / rolling_std
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_factor * df['volume']
    
    return final_factor
```
