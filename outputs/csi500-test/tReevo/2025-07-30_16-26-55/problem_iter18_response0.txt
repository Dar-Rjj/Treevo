```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=50, range_momentum_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = high_to_low_range.rolling(window=range_momentum_window).std()
    rolling_mean = high_to_low_range.rolling(window=range_momentum_window).mean()
    high_to_low_range_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Calculate Long-Term Trend
    long_term_trend = df['close'].rolling(window=long_window).mean()
    
    # Combine Momentum and Trends
    combined_factor = (high_to_low_range_momentum * short_term_trend * long_term_trend) - rolling_std
    
    # Adjust Combined Factor by Volume
    average_volume = df['volume'].rolling(window=range_momentum_window).mean()
    adjusted_factor = combined_factor / average_volume
    
    return adjusted_factor
```
