```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    short_term_momentum = (df['close'] - short_term_sma).clip(lower=0)
    short_term_momentum = short_term_momentum.rolling(window=5).sum()

    # Medium-Term Price Momentum
    medium_term_sma = df['close'].rolling(window=20).mean()
    medium_term_momentum = (df['close'] - medium_term_sma).clip(lower=0)
    medium_term_momentum = medium_term_momentum.rolling(window=10).sum()

    # Long-Term Price Momentum
    long_term_sma = df['close'].rolling(window=30).mean()
    long_term_momentum = (df['close'] - long_term_sma).clip(lower=0)
    long_term_momentum = long_term_momentum.rolling(window=20).sum()

    # Price Breakout Indicator
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    distance_from_high = (df['close'] / high_20) - 1
    distance_from_low = 1 - (df['close'] / low_20)
    breakout_indicator = (distance_from_high + distance_from_low).abs()

    # Volume Adjust Factor
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjust_factor = df['volume'] / avg_volume_20

    # Combine Factors
    short_term_mom_vol_adj = short_term_momentum * volume_adjust_factor
    medium_term_mom_vol_adj = medium_term_momentum * volume_adjust_factor
    long_term_mom_vol_adj = long_term_momentum * volume_adjust_factor
    combined_momentum = short_term_mom_vol_adj + medium_term_mom_vol_adj + long_term_mom_vol_adj
    combined_factors = combined_momentum + breakout_indicator

    # Final Alpha Factor with Linear Weighting
    final_alpha = combined_factors * 0.8  # Applying a linear weighting factor of 0.8

    return final_alpha
```
