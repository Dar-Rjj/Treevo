```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    high_to_low_momentum = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend using Exponential Moving Average (EMA) over a short window
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
