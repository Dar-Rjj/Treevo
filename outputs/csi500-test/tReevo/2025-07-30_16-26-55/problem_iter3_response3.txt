import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation and Simple Moving Average of the High-to-Low Range
    rolling_window = 10
    df['range_std'] = df['high_to_low_range'].rolling(window=rolling_window).std()
    df['range_sma'] = df['high_to_low_range'].rolling(window=rolling_window).mean()
    
    # Compute High-to-Low Range Momentum
    df['range_momentum'] = df['range_std'] / df['range_sma']
    
    # Adjust Momentum by Volume
    df['adjusted_momentum'] = df['range_momentum'] * df['volume']
    
    return df['adjusted_momentum'].dropna()
