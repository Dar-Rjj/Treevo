import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    std_dev_range = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term High-to-Low Range Momentum
    short_term_momentum = high_to_low_range.rolling(window=5).mean()
    
    # Calculate Long-Term High-to-Low Range Momentum
    long_term_momentum = high_to_low_range.rolling(window=20).mean()
    
    # Adjust Momentum by Volume
    adjusted_short_term_momentum = short_term_momentum * df['volume']
    adjusted_long_term_momentum = long_term_momentum * df['volume']
    
    # Generate Final Alpha Factor
    alpha_factor = adjusted_short_term_momentum - adjusted_long_term_momentum
    
    return alpha_factor
