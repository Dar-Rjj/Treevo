```python
import pandas as pd

def heuristics_v2(df, window_range=20, window_momentum=10, window_trend=5):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Normalize High-to-Low Range
    rolling_mean_range = high_to_low_range.rolling(window=window_range).mean()
    normalized_range = high_to_low_range / rolling_mean_range
    
    # Calculate High-to-Low Range Momentum
    rolling_std_normalized_range = normalized_range.rolling(window=window_momentum).std()
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=window_trend).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (normalized_range * short_term_trend) - rolling_std_normalized_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
