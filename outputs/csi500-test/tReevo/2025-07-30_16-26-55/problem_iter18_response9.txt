```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_std = df['range'].rolling(window=10).std()
    range_mean = df['range'].rolling(window=10).mean()
    df['range_momentum'] = range_std / range_mean
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    df['short_term_trend'] = short_term_trend
    
    # Combine Momentum and Short-Term Trend
    df['combined_factor'] = (df['range_momentum'] * df['short_term_trend']) - range_std
    
    # Adjust Combined Factor by Volume
    volume_mean = df['volume'].rolling(window=10).mean()
    df['normalized_volume'] = df['volume'] / volume_mean
    df['adjusted_factor'] = df['combined_factor'] * df['normalized_volume']
    
    return df['adjusted_factor']
```
