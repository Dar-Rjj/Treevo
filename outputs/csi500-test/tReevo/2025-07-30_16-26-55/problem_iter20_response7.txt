```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (EMA of range over a short window, e.g., 5 days)
    short_window = 5
    high_to_low_range_momentum = high_to_low_range.ewm(span=short_window, adjust=False).mean()
    
    # Calculate Short-Term Trend (EMA of close prices over a short window, e.g., 5 days)
    short_term_trend = df['close'].ewm(span=short_window, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_range_momentum * short_term_trend
    
    # Calculate Volatility (Rolling Standard Deviation of High-to-Low Range over a short window, e.g., 5 days)
    volatility = high_to_low_range.rolling(window=short_window).std()
    
    # Adjust Combined Factor by Volatility
    adjusted_combined_factor = combined_factor / volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_combined_factor * df['volume']
    
    return final_factor
```
