```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=5, long_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    ema_high_to_low_range = high_to_low_range.ewm(span=long_window).mean()
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend
    short_trend = df['close'].ewm(span=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * (df['close'] / short_trend)
    
    # Calculate Inverse Volatility
    log_returns = np.log(df['close']).diff()
    vol = log_returns.rolling(window=long_window).std()
    inverse_volatility = 1 / vol
    
    # Adjust Combined Factor by Inverse Volatility
    adjusted_combined_factor = combined_factor * inverse_volatility
    
    # Adjust Final Factor by Volume
    final_factor = adjusted_combined_factor * df['volume']
    
    return final_factor
```
