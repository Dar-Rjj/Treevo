```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Integrate Volume Early
    volume_adjusted_range = high_to_low_range * df['volume']
    
    # Calculate High-to-Low Range Momentum
    rolling_std = volume_adjusted_range.rolling(window=5).std()
    rolling_mean = volume_adjusted_range.rolling(window=5).mean()
    range_momentum = rolling_std / rolling_mean
    
    # Calculate Short-Term Trend
    sma_5 = df['close'].rolling(window=5).mean()
    short_term_trend = df['close'] - sma_5
    
    # Combine Momentum and Short-Term Trend
    combined_factor = short_term_trend * range_momentum
    
    return combined_factor
```
