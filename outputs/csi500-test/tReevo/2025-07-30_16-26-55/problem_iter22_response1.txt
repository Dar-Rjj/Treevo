```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=5, ema_span=10, vol_window=20):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (EMA of range)
    htl_momentum = high_to_low_range.ewm(span=ema_span, adjust=False).mean()
    
    # Calculate Short-Term Trend (EMA of close prices over a short window)
    short_term_trend = df['close'].ewm(span=short_window, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = htl_momentum * short_term_trend
    
    # Calculate Volatility (Rolling Standard Deviation of High-to-Low Range)
    volatility = high_to_low_range.rolling(window=vol_window).std()
    
    # Adjust Combined Factor by Volume and Inverse Volatility
    adjusted_factor = combined_factor * df['volume'] / volatility
    
    return adjusted_factor
```
