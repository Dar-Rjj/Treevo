```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (rolling standard deviation)
    momentum_window = 10  # Example window, can be adjusted
    high_to_low_momentum = high_to_low_range.rolling(window=momentum_window).std()
    
    # Calculate Short-Term Trend (Simple Moving Average)
    sma_window = 5  # Example short-term span, can be adjusted
    short_term_trend = df['close'].rolling(window=sma_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
