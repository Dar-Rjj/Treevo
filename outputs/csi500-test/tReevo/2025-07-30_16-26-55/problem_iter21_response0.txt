```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Rolling Standard Deviation of High-to-Low Range
    rolling_std_window = 20  # Example window, can be adjusted
    range_momentum = high_to_low_range.rolling(window=rolling_std_window).std()
    
    # Calculate Short-Term Trend (Exponential Moving Average)
    short_window = 5  # Example short window, can be adjusted
    short_term_trend = df['close'].ewm(span=short_window, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
