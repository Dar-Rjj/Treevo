```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=10, long_window=30):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_momentum = high_to_low_range.rolling(window=long_window).std() / high_to_low_range.rolling(window=long_window).mean()
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - high_to_low_range.rolling(window=long_window).std()
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
