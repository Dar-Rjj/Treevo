```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum with a short rolling window (e.g., 5 days)
    high_to_low_momentum = high_to_low_range.rolling(window=5).sum()
    
    # Calculate Short-Term Trend using Close prices with a short moving average window (e.g., 5 days)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor / df['volume']
    
    return adjusted_factor
```
