```python
import pandas as pd

def heuristics_v2(df, short_window=10, range_window=20, vol_window=30):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_momentum = (high_to_low_range.rolling(window=range_window).std() 
                      / high_to_low_range.rolling(window=range_window).mean())
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].rolling(window=short_window).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = range_momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    volume_adjusted_factor = combined_factor * df['volume']
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=vol_window).std()
    
    # Final Adjustment by Volatility
    final_factor = volume_adjusted_factor / volatility
    
    return final_factor
```
