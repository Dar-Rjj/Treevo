```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    ema_high_to_low_range = high_to_low_range.ewm(span=10, adjust=False).mean()
    high_to_low_momentum = ema_high_to_low_range / high_to_low_range
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = high_to_low_momentum * short_term_trend
    
    # Incorporate Trading Volume
    volume_adjusted_factor = combined_factor * df['volume']
    
    # Calculate Logarithmic Returns
    log_returns = np.log(df['close']).diff()
    
    # Calculate Rolling Standard Deviation of Logarithmic Returns
    rolling_std = log_returns.rolling(window=20).std()
    
    # Calculate Inverse Volatility
    inverse_volatility = 1 / rolling_std
    
    # Adjust Final Factor by Inverse Volatility
    final_factor = volume_adjusted_factor * inverse_volatility
    
    return final_factor
```
