```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Average True Price Momentum.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    
    Returns:
    - A pandas Series with the factor values indexed by date.
    """
    # Calculate True Price
    true_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Momentum
    momentum = true_price - true_price.shift(5)
    
    # Smooth Momentum using Exponential Weighted Moving Average
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    return smoothed_momentum
```
