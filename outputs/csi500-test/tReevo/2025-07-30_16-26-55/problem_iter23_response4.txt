```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (Rolling Standard Deviation of Range)
    range_momentum = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend (Exponential Moving Average of Close prices)
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = (range_momentum * short_term_trend) - range_momentum
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
