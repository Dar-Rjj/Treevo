```python
import pandas as pd

def heuristics_v2(df):
    # Compute Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-to-Low Range and normalize it
    high_low_range = (df['high'] - df['low']) / df['low']
    normalized_high_low_range = high_low_range / high_low_range.mean()
    
    # Weight by Volume and adjust by High-Low Range
    weighted_return = open_to_close_return * df['volume']
    adjusted_weighted_return = weighted_return - normalized_high_low_range
    
    # Compute Short-Term Momentum
    daily_returns = df['close'].pct_change()
    short_term_momentum = daily_returns.rolling(window=5).sum()
    
    # Adjustment Factor using Average Daily Volume
    avg_daily_volume = df['volume'].rolling(window=5).mean()
    adjustment_factor = avg_daily_volume / df['volume']
    adjusted_short_term_momentum = short_term_momentum * adjustment_factor
    
    # Compute Long-Term Mean Reversion
    long_term_mean_reversion = daily_returns.rolling(window=20).mean()
    mean_reversion_adjustment = (daily_returns - long_term_mean_reversion)
    
    # Final Alpha Factor
    alpha_factor = adjusted_weighted_return + adjusted_short_term_momentum + mean_reversion_adjustment
    
    return alpha_factor
```
