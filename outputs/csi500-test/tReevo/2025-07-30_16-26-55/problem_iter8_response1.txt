```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum (20-day Rolling Standard Deviation)
    hl_range_momentum = high_to_low_range.rolling(window=20).std()
    
    # Calculate Short-Term Price Trend (10-period SMA of Close)
    short_term_trend = df['close'].rolling(window=10).mean()
    
    # Combine High-to-Low Range Momentum and Short-Term Trend
    combined_factor = hl_range_momentum * short_term_trend - high_to_low_range.rolling(window=20).std()
    
    # Calculate Volume Adjust Factor (Today's Volume / 20-day Average Volume)
    volume_adjust_factor = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * volume_adjust_factor
    
    return final_alpha_factor
```
