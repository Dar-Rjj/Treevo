```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum using rolling standard deviation
    momentum = high_to_low_range.rolling(window=20).std()
    
    # Adjust Momentum by Volume
    adjusted_momentum = momentum * df['volume']
    
    return adjusted_momentum
```
