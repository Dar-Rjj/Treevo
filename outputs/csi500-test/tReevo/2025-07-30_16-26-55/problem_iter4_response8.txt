```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    daily_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Momentum
    range_std = daily_range.rolling(window=10).std()
    range_sma = daily_range.rolling(window=10).mean()
    range_momentum = range_sma / range_std
    
    # Integrate Recent Volatility
    recent_volatility = df['close'].rolling(window=5).std().rolling(window=5).mean()
    
    # Adjust Momentum by Volume
    adjusted_momentum = range_momentum * df['volume']
    
    # Normalize by Recent Volatility
    final_factor = adjusted_momentum / recent_volatility
    
    return final_factor
```
