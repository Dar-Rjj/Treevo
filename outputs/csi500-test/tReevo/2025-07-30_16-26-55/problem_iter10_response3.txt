```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Standard Deviation of High-to-Low Range
    rolling_std_range = high_to_low_range.rolling(window=10).std()
    
    # Compute High-to-Low Range Momentum
    range_momentum = high_to_low_range / rolling_std_range
    
    # Calculate Short-Term Trend (Moving Average over a short window)
    short_term_trend = df['close'].rolling(window=5).mean()
    
    # Combine Momentum with Short-Term Trend
    combined_factor = range_momentum * short_term_trend - rolling_std_range
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    # Final Adjustment and Signal Generation (Applying a simple threshold)
    final_signal = adjusted_factor.apply(lambda x: 1 if x > 0 else -1)
    
    return final_signal
```
