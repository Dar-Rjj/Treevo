```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-to-Low Range
    volume_weighted_range = high_to_low_range * df['volume']
    
    # Calculate High-to-Low Range Momentum
    momentum = volume_weighted_range.rolling(window=20).std()
    
    # Calculate Short-Term Trend (Exponential Moving Average of Close prices)
    short_term_trend = df['close'].ewm(span=5, adjust=False).mean()
    
    # Combine Momentum and Short-Term Trend
    combined_factor = momentum * short_term_trend
    
    # Adjust Combined Factor by Volume
    adjusted_factor = combined_factor * df['volume']
    
    return adjusted_factor
```
