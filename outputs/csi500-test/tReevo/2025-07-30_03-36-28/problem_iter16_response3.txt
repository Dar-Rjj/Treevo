import pandas as pd

def heuristics_v2(df, short_sma_period=50, long_sma_period=200, vol_lookback=30, momentum_lookback=10, pct_change_lookback=5, avg_turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_period).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Rolling_Avg'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['Close_Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']

    # Consider Market Trend Alignment
    df['Market_Trend'] = df.apply(lambda row: 1 if row['SMA'] > row['Long_SMA'] else -1, axis=1)

    # Incorporate Enhanced Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Avg_Turnover'] = df['Daily_Turnover'].rolling(window=avg_turnover_lookback).mean()
    df['Liquidity_Ratio'] = df['Daily_Turnover'] / df['Avg_Turnover']
    df['Liquidity_Weight'] = df['Liquidity_Ratio'].apply(lambda x: 1.0 if x >= 1.0 else x)

    # Define Weights for Each Component
    momentum_weight = 0.3
    volatility_weight = 0.2
    pct_change_weight = 0.1
    high_low_range_weight = 0.1
    trend_weight = 0.15
    liquidity_weight = 0.15

    # Combine Components into the Alpha Factor
    df['Alpha_Factor'] = (
        momentum_weight * df['Close_Price_Momentum'] +
        volatility_weight * df['Vol_Rolling_Avg'] +
        pct_change_weight * df['Pct_Change_Close'] +
        high_low_range_weight * df['High_Low_Range'] +
        trend_weight * df['Market_Trend'] +
        liquidity_weight * df['Liquidity_Weight']
    )

    # Dynamically Adjust Weights Based on Market Trend
    df['Alpha_Factor'] = df.apply(
        lambda row: row['Alpha_Factor'] * (1 + 0.1 * row['Market_Trend']) if row['Market_Trend'] > 0 else row['Alpha_Factor'], 
        axis=1
    )

    return df['Alpha_Factor']
