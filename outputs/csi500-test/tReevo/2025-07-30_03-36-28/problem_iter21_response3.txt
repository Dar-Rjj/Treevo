```python
import pandas as pd

def heuristics_v2(df, short_sma_period=50, long_sma_period=200, vol_lookback=20, momentum_n=10, pct_change_lookback=10, liquidity_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_diff'] = df['high_low_diff'] * df['volume']
    df['volume_adjusted_volatility'] = df['vol_weighted_diff'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_n).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['market_trend'] = (df['SMA_short'] > df['SMA_long']).astype(int)

    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_turnover'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['liquidity_factor'] = df['avg_turnover'] / df['avg_turnover'].rolling(window=liquidity_lookback).max()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'volume_adjusted_volatility': -0.2,
        'pct_change_close': 0.2,
        'high_low_range': 0.1,
        'liquidity_factor': 0.1
    }
    
    df['alpha_factor'] = (
        df['price_momentum'] * weights['price_momentum'] +
        df['volume_adjusted_volatility'] * weights['volume_adjusted_volatility'] +
        df['pct_change_close'] * weights['pct_change_close'] +
        df['high_low_range'] * weights['high_low_range'] +
        df['liquidity_factor'] * weights['liquidity_factor']
    )

    # Adjust Weights Dynamically
    df['alpha_factor'] = df['alpha_factor'] * (1 + 0.1 * df['market_trend'])

    return df['alpha_factor']
```
