```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    sma_lookback = 20
    vol_adj_vol_lookback = 10
    momentum_avg_lookback = 20
    pct_change_lookback = 5
    weights = {'price_momentum': 0.4, 'vol_adj_volatility': 0.3, 'price_change_metrics': 0.3}

    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adj_volatility'] = df['volume_weighted_high_low'].rolling(window=vol_adj_vol_lookback).mean()
    
    # Compute Price Momentum
    df['avg_close'] = df['close'].rolling(window=momentum_avg_lookback).mean()
    df['price_momentum'] = (df['close'] - df['SMA']) / df['avg_close']

    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']
    df['price_change_metrics'] = (df['pct_change'] + df['high_low_range']) / 2  # Simplified combination for demonstration

    # Final Alpha Factor
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adj_volatility'] * df['vol_adj_volatility'] +
        weights['price_change_metrics'] * df['price_change_metrics']
    )

    return df['alpha_factor']
```
