```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_lookback = 20
    vol_adj_volatility_lookback = 10
    momentum_lookback = 20
    price_change_lookback = 5
    high_low_range_lookback = 5
    open_close_diff_lookback = 5
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_adj_volatility_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Price_Percentage_Change'] = df['close'].pct_change(price_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    df['Open_Close_Difference'] = (df['open'] - df['close']) / df['close']
    df['Close_Price_Absolute_Change'] = df['close'].diff(open_close_diff_lookback) / df['close']
    
    # Final Alpha Factor
    weights = {
        'Price_Momentum': 0.4,
        'Volume_Adjusted_Volatility': 0.3,
        'Close_Price_Percentage_Change': 0.1,
        'High_Low_Range': 0.1,
        'Open_Close_Difference': 0.1
    }
    
    df['Alpha_Factor'] = (df['Price_Momentum'] * weights['Price_Momentum'] +
                          df['Volume_Adjusted_Volatility'] * weights['Volume_Adjusted_Volatility'] +
                          df['Close_Price_Percentage_Change'] * weights['Close_Price_Percentage_Change'] +
                          df['High_Low_Range'] * weights['High_Low_Range'] +
                          df['Open_Close_Difference'] * weights['Open_Close_Difference'])
    
    return df['Alpha_Factor']
```
