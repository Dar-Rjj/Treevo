```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_adj_lookback=10, momentum_lookback=10, pct_change_lookback=5, short_sma_lookback=10, long_sma_lookback=50, turnover_lookback=10, atr_lookback=14):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volume_Adjusted_Volatility_SMA'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_adj_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Granular Trend Analysis
    df['Short_Term_SMA'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['Long_Term_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Trend_Indicator'] = 0
    df.loc[df['Short_Term_SMA'] > df['Long_Term_SMA'], 'Trend_Indicator'] = 1  # Bullish
    df.loc[df['Short_Term_SMA'] < df['Long_Term_SMA'], 'Trend_Indicator'] = -1  # Bearish
    
    # Incorporate Enhanced Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Daily_Turnover_SMA'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    
    # Calculate Average True Range (ATR)
    high_low = df['high'] - df['low']
    high_close = abs(df['high'] - df['close'].shift())
    low_close = abs(df['low'] - df['close'].shift())
    tr = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
    df['ATR'] = tr.rolling(window=atr_lookback).mean()
    
    # Combine Daily Turnover and ATR
    liquidity_weight = 0.5
    df['Combined_Liquidity'] = liquidity_weight * df['Daily_Turnover_SMA'] + (1 - liquidity_weight) * df['ATR']
    
    # Final Alpha Factor
    momentum_weight = 0.3
    volatility_weight = 0.2
    trend_weight = 0.2
    liquidity_weight = 0.3
    
    df['Alpha_Factor'] = (
        momentum_weight * df['Momentum'] +
        volatility_weight * df['Volume_Adjusted_Volatility_SMA'] +
        trend_weight * df['Trend_Indicator'] +
        liquidity_weight * df['Combined_Liquidity']
    )
    
    return df['Alpha_Factor']
```
