```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame, sma_window: int = 20, vol_adj_volatility_window: int = 20) -> pd.Series:
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = data['close'].rolling(window=sma_window).mean()
    
    # Compute High-Low Price Difference
    high_low_diff = data['high'] - data['low']
    
    # Apply Volume Weighting to High-Low Differences
    volume_weighted_high_low = high_low_diff * data['volume']
    
    # Calculate the Rolling Average of the Volume-Weighted High-Low Differences
    vol_adj_volatility = volume_weighted_high_low.rolling(window=vol_adj_volatility_window).mean()
    
    # Compute Price Momentum
    price_momentum = data['close'] - sma_close
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adj_volatility
    
    return alpha_factor
```
