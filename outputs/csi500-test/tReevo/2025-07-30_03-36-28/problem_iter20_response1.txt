```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_lookback=5, long_sma_lookback=50, z_score_lookback=20, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adjusted_volatility'] = df['vol_weighted_high_low'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['avg_close'] = df['close'].rolling(window=momentum_lookback).mean()
    df['momentum'] = (df['close'] - df['SMA']) / df['avg_close']
    
    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(periods=momentum_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['trend_indicator'] = np.where(df['SMA'] > df['long_SMA'], 1, -1)
    
    # Incorporate Mean Reversion
    df['z_score'] = (df['close'] - df['close'].rolling(window=z_score_lookback).mean()) / df['close'].rolling(window=z_score_lookback).std()
    df['mean_reversion'] = -df['z_score']
    
    # Incorporate Dynamic Liquidity Measures
    df['turnover'] = df['volume'] * df['close']
    df['avg_turnover'] = df['turnover'].rolling(window=turnover_lookback).mean()
    df['liquidity_weight'] = df['avg_turnover'] / df['avg_turnover'].max()
    
    # Final Alpha Factor
    weights = {
        'momentum': 0.3,
        'vol_adjusted_volatility': 0.2,
        'pct_change_close': 0.1,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'mean_reversion': 0.1,
        'liquidity_weight': 0.1
    }
    
    df['alpha_factor'] = (
        weights['momentum'] * df['momentum'] +
        weights['vol_adjusted_volatility'] * df['vol_adjusted_volatility'] +
        weights['pct_change_close'] * df['pct_change_close'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['trend_indicator'] * df['trend_indicator'] +
        weights['mean_reversion'] * df['mean_reversion'] +
        weights['liquidity_weight'] * df['liquidity_weight']
    )
    
    return df['alpha_factor']
```
