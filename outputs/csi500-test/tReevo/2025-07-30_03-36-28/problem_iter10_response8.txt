import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_sma=20, lookback_volatility=10, lookback_momentum_avg=20, 
                  lookback_percentage_change=5, lookback_sentiment=20, short_lookback_sma=5):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=lookback_sma).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_weighted_high_low'].rolling(window=lookback_volatility).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=lookback_momentum_avg).mean()
    
    # Incorporate Additional Price Change Metrics
    df['percentage_change'] = df['close'].pct_change(lookback_percentage_change)
    df['high_low_range'] = df['high'] - df['low']
    
    # Integrate Dynamic Market Sentiment
    df['net_returns'] = df['close'].pct_change()
    df['positive_returns'] = df['net_returns'].apply(lambda x: x if x > 0 else 0)
    df['negative_returns'] = df['net_returns'].apply(lambda x: x if x < 0 else 0)
    df['sentiment_score'] = df['positive_returns'].rolling(window=lookback_sentiment).sum() + \
                            df['negative_returns'].rolling(window=lookback_sentiment).sum()
    
    # Adjust Sentiment Score Based on Recent Market Trends
    df['short_term_sma'] = df['close'].rolling(window=short_lookback_sma).mean()
    df['trend_indicator'] = np.where(df['short_term_sma'] > df['SMA'], 'Bullish', 'Bearish')
    df['adjusted_sentiment_score'] = np.where(df['trend_indicator'] == 'Bullish', df['sentiment_score'] * 1.1, df['sentiment_score'] * 0.9)
    
    # Combine Components into the Final Alpha Factor
    weight_price_momentum = 0.4
    weight_volatility = 0.2
    weight_percentage_change = 0.2
    weight_sentiment = 0.2
    
    df['alpha_factor'] = (weight_price_momentum * df['price_momentum'] +
                          weight_volatility * (1 / df['volatility']) +
                          weight_percentage_change * df['percentage_change'] +
                          weight_sentiment * df['adjusted_sentiment_score'])
    
    return df['alpha_factor']
