```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    sma_lookback = 20
    vol_adj_vol_lookback = 20
    price_mom_lookback = 20
    pct_change_lookback = 5
    sentiment_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Vol'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adjusted_Vol_Rolling_Avg'] = df['Volume_Adjusted_Vol'].rolling(window=vol_adj_vol_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=price_mom_lookback).mean()
    
    # Calculate the Percentage Change in Close Price
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    
    # Calculate the High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Integrate Market Sentiment
    df['Net_Returns'] = df['close'].pct_change()
    df['Positive_Returns'] = df['Net_Returns'].apply(lambda x: x if x > 0 else 0)
    df['Negative_Returns'] = df['Net_Returns'].apply(lambda x: x if x < 0 else 0)
    df['Sentiment_Score'] = df['Positive_Returns'].rolling(window=sentiment_lookback).sum() + df['Negative_Returns'].rolling(window=sentiment_lookback).sum()
    
    # Combine the components into the final alpha factor
    df['Alpha_Factor'] = (
        0.3 * df['Price_Momentum'] +
        0.2 * df['Vol_Adjusted_Vol_Rolling_Avg'] +
        0.2 * df['Pct_Change'] +
        0.3 * df['Sentiment_Score']
    )
    
    return df['Alpha_Factor']
```
