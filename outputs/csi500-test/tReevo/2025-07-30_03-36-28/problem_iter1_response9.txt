import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, smooth_period=7):
    # Calculate True Range
    df['High-Low'] = df['high'] - df['low']
    df['High-Close1'] = abs(df['high'] - df['close'].shift(1))
    df['Low-Close1'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['High-Low', 'High-Close1', 'Low-Close1']].max(axis=1)
    
    # Sum N-period True Ranges
    df['SumTrueRange'] = df['TrueRange'].rolling(window=n).sum()
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Average N-period High-Low Range
    df['AvgHighLowRange'] = df['HighLowRange'].rolling(window=n).mean()
    
    # Factor Construction: Subtract Sum N-period True Ranges from Average N-period High-Low Range
    df['Factor'] = df['AvgHighLowRange'] - df['SumTrueRange']
    
    # Smooth Factor
    df['SmoothedFactor'] = df['Factor'].rolling(window=smooth_period).mean()
    
    return df['SmoothedFactor']
