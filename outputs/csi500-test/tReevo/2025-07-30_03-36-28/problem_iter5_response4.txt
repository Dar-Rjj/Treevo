```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, voladj_lookback=10, momentum_avg_lookback=5, pct_change_lookback=5, weights=(0.4, 0.3, 0.3)):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolumeWeightedHL'] = df['HighLowDiff'] * df['volume']
    df['VolAdjVolatility'] = df['VolumeWeightedHL'].rolling(window=voladj_lookback).mean()
    
    # Compute Price Momentum
    df['CloseAvg'] = df['close'].rolling(window=momentum_avg_lookback).mean()
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['CloseAvg']
    
    # Incorporate Additional Price Change Metrics
    df['PctChangeClose'] = df['close'].pct_change(periods=pct_change_lookback)
    df['HighLowRange'] = df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = (weights[0] * df['PriceMomentum'] + 
                    weights[1] * df['VolAdjVolatility'] + 
                    weights[2] * df['PctChangeClose'])
    
    return alpha_factor
```
