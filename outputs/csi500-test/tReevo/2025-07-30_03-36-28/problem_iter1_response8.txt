```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Compute 5-day and 20-day average volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Multiply returns by corresponding volume
    df['5_day_momentum'] = df['5_day_return'] * df['5_day_avg_volume']
    df['20_day_momentum'] = df['20_day_return'] * df['20_day_avg_volume']
    
    # Subtract 20-day momentum from 5-day momentum to get the final alpha factor
    df['alpha_factor'] = df['5_day_momentum'] - df['20_day_momentum']
    
    return df['alpha_factor']
```
