import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    sma_short_period = 20
    sma_long_period = 60
    vol_rolling_period = 30
    momentum_lookback = 10
    pct_change_period = 5
    liquidity_rolling_period = 30
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_rolling_period).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Pct_Change'] = df['close'].pct_change(periods=pct_change_period)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Liquidity_Rolling_Avg'] = df['Daily_Turnover'].rolling(window=liquidity_rolling_period).mean()
    df['Liquidity_Factor'] = df['Daily_Turnover'] / df['Liquidity_Rolling_Avg']
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': -0.2,
        'Close_Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Liquidity_Factor': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Close_Pct_Change'] * df['Close_Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Liquidity_Factor'] * df['Liquidity_Factor']
    )
    
    return df['Alpha_Factor']
