import pandas as pd
import numpy as np

def heuristics_v2(df, sma_lookback=20, vola_lookback=20, momentum_lookback=20, pct_change_lookback=5, long_sma_lookback=100, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Vola'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vola_lookback).mean()
    
    # Compute Price Momentum
    avg_close = df['close'].rolling(window=momentum_lookback).mean()
    df['Momentum'] = (df['close'] - df['SMA']) / avg_close
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    df['Long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Trend_Indicator'] = np.where(df['SMA'] > df['Long_SMA'], 1, -1)  # 1 for Bullish, -1 for Bearish
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Average_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Adjustment'] = np.where(df['Average_Turnover'] > df['Average_Turnover'].shift(1), 1.2, 0.8)
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Vola': -0.3,
        'Pct_Change': 0.2,
        'High_Low_Range': -0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Vola'] * df['Volume_Adjusted_Vola'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range']
    )
    
    # Adjust Weights Dynamically Based on Market Trend
    df['Alpha_Factor'] = df['Alpha_Factor'] * df['Trend_Indicator'] * df['Liquidity_Adjustment']
    
    return df['Alpha_Factor']
