```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    sma_short_period = 20
    sma_long_period = 60
    vol_adj_volatility_period = 20
    price_momentum_period = 20
    percentage_change_period = 5
    daily_turnover_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Weighted_HL'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adjusted_Volatility'] = df['Volume_Weighted_HL'].rolling(window=vol_adj_volatility_period).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=price_momentum_period).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Percentage_Change'] = df['close'].pct_change(percentage_change_period)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = np.where(df['SMA_Short'] > df['SMA_Long'], 'Bullish', 'Bearish')
    
    # Incorporate Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=daily_turnover_period).mean()
    
    # Dynamically Adjust Component Weights Based on Market Trend
    bull_weights = {'Price_Momentum': 0.4, 'Vol_Adjusted_Volatility': 0.3, 'Percentage_Change': 0.2, 'High_Low_Range': 0.1}
    bear_weights = {'Price_Momentum': 0.2, 'Vol_Adjusted_Volatility': 0.1, 'Percentage_Change': 0.3, 'High_Low_Range': 0.4}
    
    # Adjust Weights Based on Liquidity
    high_liquidity_weights = {'Price_Momentum': 0.5, 'Vol_Adjusted_Volatility': 0.3, 'Percentage_Change': 0.15, 'High_Low_Range': 0.05}
    low_liquidity_weights = {'Price_Momentum': 0.3, 'Vol_Adjusted_Volatility': 0.2, 'Percentage_Change': 0.25, 'High_Low_Range': 0.25}
    
    # Combine components
    def combine_factors(row):
        if row['Trend_Indicator'] == 'Bullish':
            weights = bull_weights
        else:
            weights = bear_weights
        
        if row['Rolling_Daily_Turnover'] > df['Rolling_Daily_Turnover'].median():
            weights = high_liquidity_weights
        else:
            weights = low_liquidity_weights
        
        factor = (weights['Price_Momentum'] * row['Price_Momentum'] +
                  weights['Vol_Adjusted_Volatility'] * row['Vol_Adjusted_Volatility'] +
                  weights['Percentage_Change'] * row['Percentage_Change'] +
                  weights['High_Low_Range'] * row['High_Low_Range'])
        
        return factor
    
    df['Alpha_Factor'] = df.apply(combine_factors, axis=1)
    
    return df['Alpha_Factor']
```
