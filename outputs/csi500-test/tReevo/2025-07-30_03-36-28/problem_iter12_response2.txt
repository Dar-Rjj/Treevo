```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_high_low = high_low_diff * df['volume']
    vol_adj_volatility = volume_weighted_high_low.rolling(window=10).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma_20) / df['close'].rolling(window=20).mean()
    
    # Incorporate Additional Price Change Metrics
    pct_change_close = df['close'].pct_change(periods=5)
    high_low_range = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    sma_200 = df['close'].rolling(window=200).mean()
    trend_indicator = (sma_20 > sma_200).astype(int)
    
    # Incorporate Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_daily_turnover = daily_turnover.rolling(window=30).mean()
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': 0.3,
        'pct_change_close': 0.1,
        'trend_indicator': 0.1,
        'avg_daily_turnover': 0.1
    }
    
    # Adjust Weights Based on Market Trend
    adjusted_weights = {key: val + (0.1 if key == 'price_momentum' else 0) for key, val in weights.items()}
    if trend_indicator.iloc[-1] == 0:  # Bearish trend
        adjusted_weights['price_momentum'] -= 0.1
    
    # Adjust Weights Based on Liquidity
    if avg_daily_turnover.iloc[-1] < avg_daily_turnover.mean():
        adjusted_weights['vol_adj_volatility'] -= 0.1
    else:
        adjusted_weights['vol_adj_volatility'] += 0.1
    
    # Ensure the Weights Sum to 1
    sum_weights = sum(adjusted_weights.values())
    adjusted_weights = {key: val / sum_weights for key, val in adjusted_weights.items()}
    
    # Combine Components
    alpha_factor = (
        adjusted_weights['price_momentum'] * price_momentum +
        adjusted_weights['vol_adj_volatility'] * vol_adj_volatility +
        adjusted_weights['pct_change_close'] * pct_change_close +
        adjusted_weights['trend_indicator'] * trend_indicator +
        adjusted_weights['avg_daily_turnover'] * (avg_daily_turnover / avg_daily_turnover.mean() - 1)
    )
    
    return alpha_factor
```
