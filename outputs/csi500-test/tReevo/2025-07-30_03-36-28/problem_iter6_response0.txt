```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    short_sma_lookback = 50
    long_sma_lookback = 200
    vol_lookback = 20
    price_change_lookback = 10
    momentum_avg_n = 10
    bullish_weight = 0.6
    bearish_weight = 0.4
    
    # Calculate Simple Moving Averages
    df['SMA_short'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_adjusted_vol'] = (df['high_low_diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_avg_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['price_change_pct'] = df['close'].pct_change(periods=price_change_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Determine Market Trend
    df['trend'] = (df['SMA_short'] > df['SMA_long']).astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Define Weights for the Components
    weights = {
        'price_momentum': 0.4,
        'vol_adjusted_vol': -0.3,
        'price_change_pct': 0.2,
        'high_low_range': -0.1
    }
    
    # Adjust Weights Based on Market Trend
    df['adjusted_price_momentum'] = df['price_momentum'] * (df['trend'] * bullish_weight + (1 - df['trend']) * bearish_weight)
    df['adjusted_vol_adjusted_vol'] = df['vol_adjusted_vol'] * (df['trend'] * bullish_weight + (1 - df['trend']) * bearish_weight)
    df['adjusted_price_change_pct'] = df['price_change_pct'] * (df['trend'] * bullish_weight + (1 - df['trend']) * bearish_weight)
    df['adjusted_high_low_range'] = df['high_low_range'] * (df['trend'] * bullish_weight + (1 - df['trend']) * bearish_weight)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['adjusted_price_momentum'] * weights['price_momentum'] +
        df['adjusted_vol_adjusted_vol'] * weights['vol_adjusted_vol'] +
        df['adjusted_price_change_pct'] * weights['price_change_pct'] +
        df['adjusted_high_low_range'] * weights['high_low_range']
    )
    
    return df['alpha_factor']
```
