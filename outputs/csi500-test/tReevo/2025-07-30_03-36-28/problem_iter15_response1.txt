```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_short_period = 20
    sma_long_period = 60
    vol_adj_vol_period = 20
    momentum_lookback = 20
    pct_change_period = 5
    daily_turnover_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Vol'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adj_Vol_Rolling_Avg'] = df['Volume_Adjusted_Vol'].rolling(window=vol_adj_vol_period).mean()
    
    # Compute Price Momentum
    df['Close_Avg'] = df['close'].rolling(window=momentum_lookback).mean()
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['Close_Avg']
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_period)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    df['Market_Trend'] = df['SMA_Short'] > df['SMA_Long']
    df['Market_Trend'] = df['Market_Trend'].astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=daily_turnover_period).mean()
    df['Liquidity_Adjusted_Weight'] = df['Rolling_Avg_Turnover'] / df['Rolling_Avg_Turnover'].max()
    
    # Define weights for each component
    weights = {
        'Price_Momentum': 0.3,
        'Vol_Adj_Vol_Rolling_Avg': -0.2,
        'Pct_Change': 0.2,
        'High_Low_Range': 0.1,
        'Liquidity_Adjusted_Weight': 0.2
    }
    
    # Dynamically adjust weights based on market trend
    df['Total_Weight'] = df.apply(lambda row: sum([weights[key] * row[key] for key in weights]), axis=1)
    df['Trend_Adjusted_Weight'] = df['Total_Weight'] * (1 + 0.1 * df['Market_Trend'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Trend_Adjusted_Weight']
    
    return df['Alpha_Factor'].dropna()
```
