```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_adj_volatility = (high_low_diff * df['volume']).rolling(window=10).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma_20) / df['close'].rolling(window=20).mean()
    
    # Incorporate Additional Price Change Metrics
    pct_change_5 = df['close'].pct_change(5)
    high_low_range = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    sma_200 = df['close'].rolling(window=200).mean()
    trend_indicator = (sma_20 > sma_200).astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Incorporate Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_daily_turnover = daily_turnover.rolling(window=30).mean()
    
    # Define the Weights for Each Component
    weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': 0.3,
        'pct_change_5': 0.1,
        'trend_indicator': 0.1,
        'avg_daily_turnover': 0.1
    }
    
    # Adjust Weights Based on Market Trend
    price_momentum_weight = weights['price_momentum'] + (0.1 if trend_indicator.iloc[-1] == 1 else -0.1)
    vol_adj_volatility_weight = weights['vol_adj_volatility'] + (0.1 if avg_daily_turnover.iloc[-1] > avg_daily_turnover.mean() else -0.1)
    
    # Normalize the Weights to Sum to 1
    total_weight = price_momentum_weight + vol_adj_volatility_weight + weights['pct_change_5'] + weights['trend_indicator'] + weights['avg_daily_turnover']
    normalized_weights = {
        'price_momentum': price_momentum_weight / total_weight,
        'vol_adj_volatility': vol_adj_volatility_weight / total_weight,
        'pct_change_5': weights['pct_change_5'] / total_weight,
        'trend_indicator': weights['trend_indicator'] / total_weight,
        'avg_daily_turnover': weights['avg_daily_turnover'] / total_weight
    }
    
    # Combine the Components into the Alpha Factor
    alpha_factor = (
        price_momentum * normalized_weights['price_momentum'] +
        vol_adj_volatility * normalized_weights['vol_adj_volatility'] +
        pct_change_5 * normalized_weights['pct_change_5'] +
        trend_indicator * normalized_weights['trend_indicator'] +
        avg_daily_turnover * normalized_weights['avg_daily_turnover']
    )
    
    return alpha_factor
```
