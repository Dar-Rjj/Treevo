import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_lookback=5, pct_change_lookback=5, long_sma_lookback=60, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_diff'] = df['high_low_diff'] * df['volume']
    df['vol_adj_volatility'] = df['vol_weighted_diff'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['avg_close'] = df['close'].rolling(window=momentum_lookback).mean()
    df['price_momentum'] = (df['close'] - df['SMA']) / df['avg_close']
    
    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['trend_indicator'] = (df['SMA'] > df['long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_turnover'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    df['liquidity_adjustment'] = (df['daily_turnover'] / df['avg_turnover']).clip(lower=0.5, upper=2.0)
    
    # Define the Weights for Each Component
    weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': 0.3,
        'pct_change': 0.2,
        'high_low_range': 0.1
    }
    
    # Ensure the Weights Sum to 1
    assert sum(weights.values()) == 1.0
    
    # Combine components into a weighted sum
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adj_volatility'] * df['vol_adj_volatility'] +
        weights['pct_change'] * df['pct_change'] +
        weights['high_low_range'] * df['high_low_range']
    )
    
    # Dynamically Adjust Weights Based on Market Trend
    df['alpha_factor'] = df['alpha_factor'] * (1 + 0.1 * (df['trend_indicator'] - 0.5))
    
    # Apply Liquidity Adjustment
    df['alpha_factor'] = df['alpha_factor'] * df['liquidity_adjustment']
    
    return df['alpha_factor']
