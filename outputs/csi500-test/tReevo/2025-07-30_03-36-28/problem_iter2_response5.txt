```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, n_close_prices=20, smoothing_factor=0.1):
    # Calculate Simple Moving Average of Close Prices
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_adjusted_volatility'] = df['high_low_diff'] * df['volume']
    df['volatility_sma'] = df['volume_adjusted_volatility'].rolling(window=lookback_period).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['sma_close']) / df['close'].rolling(window=n_close_prices).mean()
    
    # Integrate Bid-Ask Spread
    df['bid_ask_spread'] = (df['open'] - df['close']).abs()
    df['bid_ask_spread_pct_change'] = df['bid_ask_spread'].pct_change().fillna(0)
    df['adjusted_bid_ask_spread'] = df['bid_ask_spread_pct_change'].ewm(alpha=smoothing_factor).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['price_momentum'] / df['volatility_sma']) + df['adjusted_bid_ask_spread']
    
    return df['alpha_factor']
```
