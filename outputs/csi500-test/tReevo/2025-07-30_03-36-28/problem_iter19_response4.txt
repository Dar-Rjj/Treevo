```python
import pandas as pd

def heuristics_v2(df, short_sma_period=50, long_sma_period=200, volatility_lookback=20, momentum_lookback=10, percentage_change_lookback=10, liquidity_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_adj_volatility'] = (df['high_low_diff'] * df['volume']).rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['percentage_change'] = df['close'].pct_change(percentage_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = df.apply(lambda row: 1 if row['SMA_short'] > row['SMA_long'] else -1, axis=1)

    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover_rolling_avg'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['liquidity_adjusted_weight'] = df['turnover_rolling_avg'] / df['turnover_rolling_avg'].max()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': -0.2,
        'percentage_change': 0.2,
        'high_low_range': 0.1,
        'liquidity_adjusted_weight': 0.1
    }
    
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adj_volatility'] * df['vol_adj_volatility'] +
        weights['percentage_change'] * df['percentage_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['liquidity_adjusted_weight'] * df['liquidity_adjusted_weight']
    )

    # Adjust Weights Dynamically Based on Market Trend
    df['alpha_factor'] = df['alpha_factor'] * df['trend_indicator']

    return df['alpha_factor']
```
