import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Return
    df['daily_log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Combine to Form Intraday Momentum
    df['intraday_momentum'] = df['daily_log_return'] * df['high_low_range']
    
    # Compute Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Determine Volume Intensity Ratio
    df['avg_volume'] = (df['volume'] + df['volume'].shift(1)) / 2
    df['volume_intensity_ratio'] = df['volume_change'] / df['avg_volume']
    
    # Weight Intraday Momentum by Volume Intensity Ratio
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * df['volume_intensity_ratio']
    
    # Generate Final Alpha Factor
    alpha_factor = df['adjusted_intraday_momentum']
    
    return alpha_factor
