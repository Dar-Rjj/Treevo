```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_lookback=20, voladj_lookback=20, momentum_lookback=10, pct_change_lookback=10, long_sma_lookback=100, liquidity_lookback=20):
    # Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_weighted_high_low = high_low_diff * df['volume']
    vol_adj_volatility = vol_weighted_high_low.rolling(window=voladj_lookback).mean()
    
    # Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Additional Price Change Metrics
    pct_change = df['close'].pct_change(periods=pct_change_lookback)
    high_low_range = df['high'] - df['low']
    
    # Market Trend Alignment
    long_sma = df['close'].rolling(window=long_sma_lookback).mean()
    trend_indicator = np.where(sma > long_sma, 1, -1)
    trend_slope = np.gradient(long_sma)
    trend_strength = np.where(trend_slope > 0, 1, -1)
    trend_indicator = trend_indicator * trend_strength
    
    # Dynamic Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_daily_turnover = daily_turnover.rolling(window=liquidity_lookback).mean()
    liquidity_weight = np.where(avg_daily_turnover > 0, avg_daily_turnover / avg_daily_turnover.max(), 0)
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'vol_adj_volatility': -0.2,
        'pct_change': 0.2,
        'high_low_range': 0.1,
        'trend_indicator': 0.15,
        'liquidity_weight': 0.05
    }
    
    alpha_factor = (
        weights['price_momentum'] * price_momentum +
        weights['vol_adj_volatility'] * vol_adj_volatility +
        weights['pct_change'] * pct_change +
        weights['high_low_range'] * high_low_range +
        weights['trend_indicator'] * trend_indicator +
        weights['liquidity_weight'] * liquidity_weight
    )
    
    return alpha_factor
```
