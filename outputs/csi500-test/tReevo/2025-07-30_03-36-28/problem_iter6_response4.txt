```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_lookback = 20
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    hl_diff = df['high'] - df['low']
    vol_weighted_diff = hl_diff * df['volume']
    volatility_lookback = 10
    df['VolAdjVol'] = vol_weighted_diff.rolling(window=volatility_lookback).mean()
    
    # Compute Price Momentum
    price_momentum_lookback = 5
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=price_momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    # Percentage Change in Close Price
    pct_change_lookback = 1
    df['PctChange'] = df['close'].pct_change(pct_change_lookback)
    # High-Low Range
    df['HLRange'] = df['high'] - df['low']

    # Final Alpha Factor
    # Weights for each component
    momentum_weight = 0.5
    volatility_weight = 0.3
    change_metrics_weight = 0.2
    df['AlphaFactor'] = (momentum_weight * df['PriceMomentum'] +
                         volatility_weight * df['VolAdjVol'] +
                         change_metrics_weight * (df['PctChange'] + df['HLRange']))

    return df['AlphaFactor']
```
