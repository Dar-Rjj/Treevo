import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=200, vol_lookback=10, momentum_lookback=10, pct_change_lookback=10, turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_period).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volatility_Rolling_Avg'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Close_Avg'] = df['close'].rolling(window=momentum_lookback).mean()
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['Close_Avg']
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA'] > df['Long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_Rolling_Avg'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Weight'] = df['Turnover_Rolling_Avg'] / df['Turnover_Rolling_Avg'].max()
    
    # Final Alpha Factor
    weights = {
        'Price_Momentum': 0.4,
        'Volatility_Rolling_Avg': 0.2,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Liquidity_Weight': 0.2
    }
    
    df['Alpha_Factor'] = (
        df['Price_Momentum'] * weights['Price_Momentum'] +
        (1 / df['Volatility_Rolling_Avg']) * weights['Volatility_Rolling_Avg'] +
        df['Pct_Change'] * weights['Pct_Change'] +
        df['High_Low_Range'] * weights['High_Low_Range'] +
        df['Liquidity_Weight'] * weights['Liquidity_Weight']
    )
    
    # Adjust Weights Dynamically Based on Market Trend
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + df['Trend_Indicator'] * 0.1)
    
    return df['Alpha_Factor'].dropna()
