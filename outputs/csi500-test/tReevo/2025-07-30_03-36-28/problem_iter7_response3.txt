import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=20, price_change_lookback=10, momentum_weight=0.5, vol_weight=0.3, price_change_weight=0.2):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolumeWeightedHL'] = df['HighLowDiff'] * df['volume']
    df['VolumeAdjustedVolatility'] = df['VolumeWeightedHL'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=sma_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['ClosePctChange'] = df['close'].pct_change(periods=price_change_lookback)
    df['HighLowRange'] = df['high'] - df['low']
    
    # Final Alpha Factor
    df['AlphaFactor'] = (
        momentum_weight * df['PriceMomentum'] -
        vol_weight * df['VolumeAdjustedVolatility'] +
        price_change_weight * df['ClosePctChange']
    )
    
    return df['AlphaFactor']
