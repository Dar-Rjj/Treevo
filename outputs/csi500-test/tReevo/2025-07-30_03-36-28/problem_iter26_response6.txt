import pandas as pd
import numpy as np

def heuristics_v2(df, short_sma_window=20, long_sma_window=200, vol_adj_window=30, momentum_lookback=60, price_change_lookback=10, turnover_lookback=50):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_adj_vol'] = df['high_low_diff'] * df['volume']
    df['vol_adj_vol_rolling'] = df['vol_adj_vol'].rolling(window=vol_adj_window).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(periods=price_change_lookback)
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    
    # Determine Market Trend
    df['long_term_SMA'] = df['close'].rolling(window=long_sma_window).mean()
    df['trend_indicator'] = np.where(df['SMA'] > df['long_term_SMA'], 1, -1)
    
    # Calculate Daily Turnover
    df['daily_turnover'] = df['volume'] * df['close']
    
    # Calculate the Rolling Average of Daily Turnover
    df['turnover_rolling'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    
    # Adjust Weights Based on Liquidity
    df['liquidity_weight'] = df['turnover_rolling'] / df['turnover_rolling'].max()
    
    # Final Alpha Factor
    # Define the Weights for Each Component
    weight_price_momentum = 0.4
    weight_vol_adj_vol = 0.2
    weight_pct_change_close = 0.1
    weight_high_low_range = 0.1
    weight_trend_indicator = 0.2
    
    # Combine components
    df['alpha_factor'] = (
        weight_price_momentum * df['price_momentum'] +
        weight_vol_adj_vol * (1 / df['vol_adj_vol_rolling']) +  # Inverse of volume-adjusted volatility
        weight_pct_change_close * df['pct_change_close'] +
        weight_high_low_range * df['high_low_range'] +
        weight_trend_indicator * df['trend_indicator']
    )
    
    # Adjust Weights Dynamically
    df['alpha_factor'] = df['alpha_factor'] * df['liquidity_weight']
    
    return df['alpha_factor']
