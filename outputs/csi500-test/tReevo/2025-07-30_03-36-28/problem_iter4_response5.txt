```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    sma = df['close'].rolling(window=lookback_period).mean()
    
    # Compute Volume-Adjusted Volatility
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_close'] = (df['high'] - df['previous_close']).abs()
    df['low_close'] = (df['low'] - df['previous_close']).abs()
    df['true_range'] = df[['high_low', 'high_close', 'low_close']].max(axis=1)
    
    # Apply Volume Weighting
    df['volume_adjusted_tr'] = df['volume'] * df['true_range']
    volume_adjusted_volatility = df['volume_adjusted_tr'].rolling(window=lookback_period).std()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=lookback_period).std()
    
    # Final Alpha Factor
    alpha_factor = price_momentum / volume_adjusted_volatility
    return alpha_factor
```
