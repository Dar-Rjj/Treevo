def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['close'].diff()
    
    # Calculate Cumulative Volume over the Last 5 Days
    df['cumulative_volume'] = df['volume'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Price Change
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    df['volume_weighted_price_change'] = (df['daily_price_movement'] * df['cumulative_volume']) / average_volume_5_days
    
    # Calculate Momentum Component
    df['momentum_component'] = (df['high'] - df['low']) * df['volume_weighted_price_change']
    
    # The final alpha factor is the Momentum Component
    return df['momentum_component']
