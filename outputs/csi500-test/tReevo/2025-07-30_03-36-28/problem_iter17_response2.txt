import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_lookback=5, pct_change_lookback=10, long_sma_lookback=50, turnover_lookback=20):
    # Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_diff = high_low_diff * df['volume']
    vol_adjusted_volatility = volume_weighted_diff.rolling(window=vol_lookback).mean()
    
    # Price Momentum
    close_n_average = df['close'].rolling(window=momentum_lookback).mean()
    price_momentum = (df['close'] - sma) / close_n_average
    
    # Additional Price Change Metrics
    pct_change = df['close'].pct_change(pct_change_lookback)
    high_low_range = df['high'] - df['low']
    
    # Market Trend Alignment
    long_sma = df['close'].rolling(window=long_sma_lookback).mean()
    trend_indicator = (sma > long_sma).astype(int)  # Bullish (1) or Bearish (0)
    
    # Dynamic Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_daily_turnover = daily_turnover.rolling(window=turnover_lookback).mean()
    liquidity_weight = (avg_daily_turnover / avg_daily_turnover.mean()).clip(0.5, 2)  # Adjust weights based on liquidity
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'vol_adjusted_volatility': -0.2,  # Negative weight to penalize high volatility
        'pct_change': 0.2,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'liquidity_weight': 0.1
    }
    
    alpha_factor = (
        price_momentum * weights['price_momentum'] +
        vol_adjusted_volatility * weights['vol_adjusted_volatility'] +
        pct_change * weights['pct_change'] +
        high_low_range * weights['high_low_range'] +
        trend_indicator * weights['trend_indicator'] +
        liquidity_weight * weights['liquidity_weight']
    )
    
    # Adjust Weights Dynamically Based on Market Trend
    alpha_factor = alpha_factor * (1 + 0.1 * trend_indicator)  # Increase weight for bullish trends
    
    return alpha_factor
