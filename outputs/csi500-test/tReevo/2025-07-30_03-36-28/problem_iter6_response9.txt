```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_lookback = 20
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolWeightedHL'] = df['HighLowDiff'] * df['volume']
    volatility_lookback = 10
    df['VolumeAdjVolatility'] = df['VolWeightedHL'].rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    momentum_n = 5
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()

    # Incorporate Market Trends
    long_term_trend_lookback = 60
    short_term_trend_lookback = 10
    df['LongTermTrend'] = df['close'].rolling(window=long_term_trend_lookback).mean()
    df['ShortTermTrend'] = df['close'].rolling(window=short_term_trend_lookback).mean()
    long_weight = 0.7
    short_weight = 0.3
    df['MarketTrends'] = long_weight * df['LongTermTrend'] + short_weight * df['ShortTermTrend']

    # Incorporate Additional Price Change Metrics
    price_change_lookback = 1
    df['PriceChangePct'] = df['close'].pct_change(periods=price_change_lookback)
    df['HighLowRange'] = df['high'] - df['low']

    # Final Alpha Factor
    alpha_momentum_weight = 0.4
    alpha_volatility_weight = 0.3
    alpha_trends_weight = 0.2
    alpha_change_metrics_weight = 0.1
    df['AlphaFactor'] = (
        alpha_momentum_weight * df['PriceMomentum'] +
        alpha_volatility_weight * df['VolumeAdjVolatility'] +
        alpha_trends_weight * df['MarketTrends'] +
        alpha_change_metrics_weight * df['PriceChangePct']
    )

    return df['AlphaFactor']
```
