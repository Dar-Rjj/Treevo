```python
import pandas as pd

def heuristics_v2(df, short_lookback=20, long_lookback=100, price_change_lookback=10, turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_short = df['close'].rolling(window=short_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_high_low_diff = high_low_diff * df['volume']
    vol_adj_volatility = volume_weighted_high_low_diff.rolling(window=short_lookback).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma_short) / df['close'].rolling(window=price_change_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    pct_change = df['close'].pct_change(periods=price_change_lookback)
    high_low_range = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    sma_long = df['close'].rolling(window=long_lookback).mean()
    trend_indicator = (sma_short > sma_long).astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Incorporate Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_turnover = daily_turnover.rolling(window=turnover_lookback).mean()
    
    # Define Weights for Each Component
    base_weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': -0.2,
        'pct_change': 0.1,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'avg_turnover': 0.1
    }
    
    # Adjust Weights Based on Market Trend and Liquidity
    weights = base_weights.copy()
    if trend_indicator.iloc[-1] == 1:  # Bullish
        weights['price_momentum'] += 0.1
        weights['vol_adj_volatility'] -= 0.1
    else:  # Bearish
        weights['price_momentum'] -= 0.1
        weights['vol_adj_volatility'] += 0.1
    
    if avg_turnover.iloc[-1] > avg_turnover.mean():
        weights['price_momentum'] += 0.05
        weights['vol_adj_volatility'] -= 0.05
    else:
        weights['price_momentum'] -= 0.05
        weights['vol_adj_volatility'] += 0.05
    
    # Normalize Weights
    total_weight = sum(weights.values())
    weights = {k: v / total_weight for k, v in weights.items()}
    
    # Final Alpha Factor
    alpha_factor = (
        price_momentum * weights['price_momentum'] +
        vol_adj_volatility * weights['vol_adj_volatility'] +
        pct_change * weights['pct_change'] +
        high_low_range * weights['high_low_range'] +
        trend_indicator * weights['trend_indicator'] +
        avg_turnover * weights['avg_turnover']
    )
    
    return alpha_factor
```
