```python
import pandas as pd

def heuristics_v2(df, short_sma_lookback=20, long_sma_lookback=200, vol_lookback=20, momentum_lookback=60, pct_change_lookback=10, turnover_lookback=30, dollar_volume_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    n_close_prices = df['close'].rolling(window=momentum_lookback).mean()
    df['Momentum'] = (df['close'] - df['SMA']) / n_close_prices

    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['Trend_Indicator'] = 0
    df.loc[df['SMA'] > df['Long_SMA'], 'Trend_Indicator'] = 1  # Bullish
    df.loc[df['SMA'] < df['Long_SMA'], 'Trend_Indicator'] = -1  # Bearish

    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()

    df['Average_Dollar_Volume'] = (df['volume'] * df['close']).rolling(window=dollar_volume_lookback).mean()

    # Adjust Weights Based on Liquidity and Average Dollar Volume
    liquidity_weight = (df['Rolling_Avg_Turnover'] / df['Rolling_Avg_Turnover'].max()).fillna(0)
    avg_dollar_volume_weight = (df['Average_Dollar_Volume'] / df['Average_Dollar_Volume'].max()).fillna(0)

    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': 0.2,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Liquidity_Weight': 0.1,
        'Avg_Dollar_Volume_Weight': 0.1
    }

    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Liquidity_Weight'] * liquidity_weight +
        weights['Avg_Dollar_Volume_Weight'] * avg_dollar_volume_weight
    )

    # Adjust Weights Dynamically
    df['Alpha_Factor'] *= (1 + df['Trend_Indicator'] * 0.1)  # Increase/Decrease for Bullish/Bearish Trends
    df['Alpha_Factor'] *= (1 + 0.1 * (liquidity_weight - 0.5))  # Increase/Decrease for Higher/Lower Liquidity
    df['Alpha_Factor'] *= (1 + 0.1 * (avg_dollar_volume_weight - 0.5))  # Increase/Decrease for Higher/Lower Avg Dollar Volume

    return df['Alpha_Factor']
```
