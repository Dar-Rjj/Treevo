import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_period = 20
    rolling_avg_period = 20
    avg_n_period = 10
    pct_change_period = 5
    rolling_std_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolAdjVolatility'] = df['HighLowDiff'] * df['volume']
    df['VolAdjVolatilityRollingAvg'] = df['VolAdjVolatility'].rolling(window=rolling_avg_period).mean()
    
    # Compute Price Momentum
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=avg_n_period).mean()
    
    # Incorporate Additional Price Change Metrics
    df['PctChangeClose'] = df['close'].pct_change(pct_change_period)
    df['HighLowRange'] = df['high'] - df['low']
    df['RollingStdHighLowRange'] = df['HighLowRange'].rolling(window=rolling_std_period).std()
    
    # Define Weights for Each Component
    weight_price_momentum = 0.4
    weight_vol_adj_volatility = 0.3
    weight_pct_change_close = 0.2
    weight_rolling_std_high_low_range = 0.1
    
    # Final Alpha Factor
    df['AlphaFactor'] = (weight_price_momentum * df['PriceMomentum'] + 
                         weight_vol_adj_volatility * df['VolAdjVolatilityRollingAvg'] + 
                         weight_pct_change_close * df['PctChangeClose'] + 
                         weight_rolling_std_high_low_range * df['RollingStdHighLowRange'])
    
    return df['AlphaFactor']
