import pandas as pd

def heuristics_v2(df, short_sma_window=20, long_sma_window=100, vol_adj_vol_window=20, momentum_n=5, pct_change_window=5, turnover_window=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_window).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Vol_Adjusted_Vol'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_adj_vol_window).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(pct_change_window)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Market_Trend'] = (df['SMA'] > df['Long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_window).mean()
    
    # Dynamically Adjust Component Weights Based on Market Trend
    bullish_weights = {'Price_Momentum': 0.4, 'Vol_Adjusted_Vol': 0.3, 'Pct_Change_Close': 0.1, 'High_Low_Range': 0.1, 'Dynamic_Liquidity': 0.1}
    bearish_weights = {'Price_Momentum': 0.2, 'Vol_Adjusted_Vol': 0.2, 'Pct_Change_Close': 0.2, 'High_Low_Range': 0.2, 'Dynamic_Liquidity': 0.2}
    
    df['Weight_Price_Momentum'] = df['Market_Trend'].apply(lambda x: bullish_weights['Price_Momentum'] if x == 1 else bearish_weights['Price_Momentum'])
    df['Weight_Vol_Adjusted_Vol'] = df['Market_Trend'].apply(lambda x: bullish_weights['Vol_Adjusted_Vol'] if x == 1 else bearish_weights['Vol_Adjusted_Vol'])
    df['Weight_Pct_Change_Close'] = df['Market_Trend'].apply(lambda x: bullish_weights['Pct_Change_Close'] if x == 1 else bearish_weights['Pct_Change_Close'])
    df['Weight_High_Low_Range'] = df['Market_Trend'].apply(lambda x: bullish_weights['High_Low_Range'] if x == 1 else bearish_weights['High_Low_Range'])
    df['Weight_Dynamic_Liquidity'] = df['Market_Trend'].apply(lambda x: bullish_weights['Dynamic_Liquidity'] if x == 1 else bearish_weights['Dynamic_Liquidity'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Weight_Price_Momentum'] * df['Price_Momentum'] +
        df['Weight_Vol_Adjusted_Vol'] * df['Vol_Adjusted_Vol'] +
        df['Weight_Pct_Change_Close'] * df['Pct_Change_Close'] +
        df['Weight_High_Low_Range'] * df['High_Low_Range'] +
        df['Weight_Dynamic_Liquidity'] * (df['Daily_Turnover'] / df['Rolling_Avg_Turnover'])
    )
    
    return df['Alpha_Factor']
