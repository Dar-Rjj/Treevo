import pandas as pd

def heuristics_v2(df, N=20):
    # Compute daily return using close price
    df['daily_return'] = df['close'].pct_change()
    
    # Compute N-day moving average of daily return
    df[f'{N}_day_momentum'] = df['daily_return'].rolling(window=N).mean()
    
    # Compute daily volume change
    df['daily_volume_change'] = df['volume'].pct_change()
    
    # Compute N-day moving average of daily volume change
    df[f'{N}_day_volume_trend'] = df['daily_volume_change'].rolling(window=N).mean()
    
    # Multiply momentum and volume trends
    df['convergence_factor'] = df[f'{N}_day_momentum'] * df[f'{N}_day_volume_trend']
    
    # Return the factor values as a pandas Series
    return df['convergence_factor']
