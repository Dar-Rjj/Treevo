import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, pct_change_lookback=5, momentum_weight=0.5, volatility_weight=0.3, price_change_weight=0.2):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=sma_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Final Alpha Factor
    alpha_factor = (momentum_weight * df['Price_Momentum'] 
                    + volatility_weight * df['Volume_Adjusted_Volatility'] 
                    + price_change_weight * df['Close_Pct_Change'])
    
    return alpha_factor.dropna()
