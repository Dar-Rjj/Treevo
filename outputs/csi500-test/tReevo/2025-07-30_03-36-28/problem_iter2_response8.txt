```python
import pandas as pd

def heuristics_v2(df, lookback_sma=20, lookback_volatility=20, lookback_momentum=10):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=lookback_sma).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_diff = high_low_diff * df['volume']
    volume_adjusted_volatility = volume_weighted_diff.rolling(window=lookback_volatility).sum() / df['volume'].rolling(window=lookback_volatility).sum()
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma.shift(lookback_momentum)
    
    # Final Alpha Factor
    alpha_factor = price_momentum / volume_adjusted_volatility
    
    return alpha_factor
```
