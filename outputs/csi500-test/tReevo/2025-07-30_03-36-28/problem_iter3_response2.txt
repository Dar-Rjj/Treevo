```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_lookback=20, vol_lookback=10):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute True Range
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_close'] = (df['high'] - df['prev_close']).abs()
    df['low_close'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['high_low', 'high_close', 'low_close']].max(axis=1)
    
    # Apply Volume Weighting
    df['volume_weighted_tr'] = df['volume'] * df['true_range']
    
    # Calculate Rolling Average of Volume-Weighted True Range
    df['volatility'] = df['volume_weighted_tr'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = df['close'] - df['SMA']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] / df['volatility']
    
    return df['alpha_factor'].dropna()
```
