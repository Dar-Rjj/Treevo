import pandas as pd

def heuristics_v2(df, short_window=20, long_window=200, momentum_lookback=10, vol_lookback=30, liquidity_lookback=30, trend_weight=0.5):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_adjusted_volatility'] = df['high_low_diff'] * df['volume']
    df['vol_rolling_avg'] = df['volume_adjusted_volatility'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['close_pct_change'] = df['close'].pct_change().rolling(window=momentum_lookback).mean()
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['long_SMA'] = df['close'].rolling(window=long_window).mean()
    df['trend_indicator'] = (df['SMA'] > df['long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['liquidity_rolling_avg'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['liquidity_weight'] = df['liquidity_rolling_avg'] / df['liquidity_rolling_avg'].max()
    
    # Final Alpha Factor
    alpha_factor = (
        trend_weight * df['trend_indicator'] * df['price_momentum'] +
        (1 - trend_weight) * (df['vol_rolling_avg'] + df['close_pct_change'] + df['high_low_range'])
    )
    
    # Adjust Weights Dynamically Based on Liquidity
    alpha_factor *= df['liquidity_weight']
    
    return alpha_factor.dropna()
