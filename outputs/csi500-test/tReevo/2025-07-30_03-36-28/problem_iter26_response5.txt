import pandas as pd

def heuristics_v2(df, sma_lookback=20, vola_lookback=20, momentum_lookback=10, long_sma_lookback=50, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High-Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High-Low_Diff'] * df['volume']
    df['VAV_Rolling_Avg'] = df['Volume_Adjusted_Volatility'].rolling(window=vola_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Determine Market Trend
    df['Long_Term_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_Rolling_Avg'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    
    # Define Weights
    bullish_weight = 0.6
    bearish_weight = 0.4
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (bullish_weight * df['Price_Momentum'] + 
                          (1 - bullish_weight) * (1 / df['VAV_Rolling_Avg'])) * df['Trend_Indicator'] + \
                         (bearish_weight * df['Price_Momentum'] + 
                          (1 - bearish_weight) * (1 / df['VAV_Rolling_Avg'])) * (1 - df['Trend_Indicator'])
    
    return df['Alpha_Factor']
