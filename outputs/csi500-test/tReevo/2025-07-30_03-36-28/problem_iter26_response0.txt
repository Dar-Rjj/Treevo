import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=20, momentum_n=10, trend_lookback=60, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Pct_Change'] = df['close'].pct_change(periods=momentum_n)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=trend_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Turnover_Weight'] = (df['Daily_Turnover'] / df['Rolling_Avg_Turnover']).clip(lower=0.5, upper=2.0)
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': 0.3,
        'Close_Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1
    }
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Close_Pct_Change'] * df['Close_Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator']
    ) * df['Turnover_Weight']
    
    return df['Alpha_Factor']
