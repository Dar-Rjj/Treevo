```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    sma = df['close'].rolling(window=lookback_period).mean()
    
    # Compute True Range
    tr = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    tr[0] = df['high'][0] - df['low'][0]  # For the first row, use the high-low range
    
    # Apply Volume Weighting to True Range
    vol_weighted_tr = tr * df['volume']
    vol_adjusted_volatility = vol_weighted_tr.rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=lookback_period).std()
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_volatility
    
    return alpha_factor
```
