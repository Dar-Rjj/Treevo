import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=100, vol_lookback=20, momentum_lookback=20, pct_change_lookback=10, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_period).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Market_Trend'] = (df['SMA'] > df['Long_SMA']).astype(int)  # 1 for Bullish, 0 for Bearish
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Weight'] = df['Rolling_Avg_Turnover'] / df['Rolling_Avg_Turnover'].max()
    
    # Define Weights for Each Component
    price_momentum_weight = 0.3
    volatility_weight = 0.2
    pct_change_weight = 0.1
    high_low_range_weight = 0.1
    liquidity_weight = 0.3
    
    # Dynamically Adjust Weights Based on Market Trend
    if df['Market_Trend'].iloc[-1] == 1:  # Bullish
        price_momentum_weight *= 1.1
        volatility_weight *= 0.9
        pct_change_weight *= 1.1
        high_low_range_weight *= 0.9
        liquidity_weight *= 1.1
    else:  # Bearish
        price_momentum_weight *= 0.9
        volatility_weight *= 1.1
        pct_change_weight *= 0.9
        high_low_range_weight *= 1.1
        liquidity_weight *= 0.9
    
    # Normalize Weights to Sum to 1
    total_weight = price_momentum_weight + volatility_weight + pct_change_weight + high_low_range_weight + liquidity_weight
    price_momentum_weight /= total_weight
    volatility_weight /= total_weight
    pct_change_weight /= total_weight
    high_low_range_weight /= total_weight
    liquidity_weight /= total_weight
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (price_momentum_weight * df['Price_Momentum'] +
                           volatility_weight * df['Volume_Adjusted_Volatility'] +
                           pct_change_weight * df['Pct_Change_Close'] +
                           high_low_range_weight * df['High_Low_Range'] +
                           liquidity_weight * df['Liquidity_Weight'])
    
    return df['Alpha_Factor']
