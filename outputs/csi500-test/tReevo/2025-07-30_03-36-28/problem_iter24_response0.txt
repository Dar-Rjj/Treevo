import pandas as pd

def heuristics_v2(df, short_sma_window=10, voladj_window=20, momentum_n=30, pct_change_window=5, long_sma_window=60, liquidity_window=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolAdjVol'] = (df['HighLowDiff'] * df['volume']).rolling(window=voladj_window).mean()
    
    # Compute Price Momentum
    df['PriceMomentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['PctChange'] = df['close'].pct_change(periods=pct_change_window)
    df['HighLowRange'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['LongSMA'] = df['close'].rolling(window=long_sma_window).mean()
    df['TrendIndicator'] = df.apply(lambda row: 1 if row['SMA'] > row['LongSMA'] else 0, axis=1)
    
    # Incorporate Dynamic Liquidity Measures
    df['DailyTurnover'] = df['volume'] * df['close']
    df['RollingDailyTurnover'] = df['DailyTurnover'].rolling(window=liquidity_window).mean()
    
    df['AvgDollarVolume'] = df['volume'] * df['close']
    df['RollingAvgDollarVolume'] = df['AvgDollarVolume'].rolling(window=liquidity_window).mean()
    
    # Final Alpha Factor
    weights = {
        'PriceMomentum': 0.4,
        'PctChange': 0.2,
        'HighLowRange': 0.1,
        'VolAdjVol': -0.1,
        'TrendIndicator': 0.1,
        'RollingDailyTurnover': 0.05,
        'RollingAvgDollarVolume': 0.05
    }
    
    df['AlphaFactor'] = (weights['PriceMomentum'] * df['PriceMomentum'] +
                         weights['PctChange'] * df['PctChange'] +
                         weights['HighLowRange'] * df['HighLowRange'] / df['close'] +
                         weights['VolAdjVol'] * df['VolAdjVol'] +
                         weights['TrendIndicator'] * df['TrendIndicator'] +
                         weights['RollingDailyTurnover'] * (df['RollingDailyTurnover'] / df['DailyTurnover']) +
                         weights['RollingAvgDollarVolume'] * (df['RollingAvgDollarVolume'] / df['AvgDollarVolume']))
    
    return df['AlphaFactor']
