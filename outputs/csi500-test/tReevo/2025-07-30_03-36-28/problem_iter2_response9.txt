```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    df['High_Low_Momentum'] = (df['High'] - df['High'].shift(1) + df['Low'] - df['Low'].shift(1)) / 2
    
    # Calculate Price Range Expansion
    df['Range_Expansion'] = (df['High'] - df['Low']) - (df['High'].shift(1) - df['Low'].shift(1))
    
    # Incorporate Volatility
    df['Volatility'] = df['Close'].rolling(window=20).std()
    df['Adjusted_High_Low_Momentum'] = df['High_Low_Momentum'] * df['Volatility']
    
    # Calculate 20-day Moving Average
    df['20MA_Close'] = df['Close'].rolling(window=20).mean()
    
    # Combine with Volume Activity
    df['20MA_Volume'] = df['Volume'].rolling(window=20).mean()
    df['Volume_Activity'] = df['Volume'] > df['20MA_Volume']
    df['Combined_Factor'] = df['Adjusted_High_Low_Momentum'] * df['Volume_Activity'].astype(int)
    
    # Incorporate Relative Strength
    df['Relative_Strength'] = df['Close'] / df['Close'].shift(10)
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Relative_Strength']
    
    return df['Final_Alpha_Factor']
```
