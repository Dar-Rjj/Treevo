```python
import pandas as pd

def heuristics_v2(df, short_sma_lookback=20, long_sma_lookback=100, vol_lookback=20, momentum_lookback=10, pct_change_lookback=5, bullish_weight=0.6, bearish_weight=0.4):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adjusted_volatility'] = df['vol_weighted_high_low'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = 0
    df.loc[df['SMA_short'] > df['SMA_long'], 'trend_indicator'] = 1  # Bullish
    df.loc[df['SMA_short'] < df['SMA_long'], 'trend_indicator'] = -1  # Bearish

    # Final Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] * 0.3 +
        df['vol_adjusted_volatility'] * 0.2 +
        df['pct_change'] * 0.2 +
        df['high_low_range'] * 0.1
    )

    # Adjust Weights Dynamically
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * (bullish_weight if row['trend_indicator'] == 1 else bearish_weight), axis=1)

    return df['alpha_factor']
```
