```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    df['SMA'] = df['close'].rolling(window=lookback_period).mean()
    
    # Compute True Range
    high_low_diff = df['high'] - df['low']
    high_close_diff = (df['high'] - df['close']).abs()
    low_close_diff = (df['low'] - df['close']).abs()
    true_range = pd.concat([high_low_diff, high_close_diff, low_close_diff], axis=1).max(axis=1)
    
    # Compute Volume-Adjusted True Range
    volume_weighted_tr = true_range * df['volume']
    rolling_sum_volume_weighted_tr = volume_weighted_tr.rolling(window=lookback_period).sum()
    rolling_sum_volume = df['volume'].rolling(window=lookback_period).sum()
    vol_adjusted_tr = rolling_sum_volume_weighted_tr / rolling_sum_volume
    
    # Compute Price Momentum
    price_momentum = df['close'] - df['SMA']
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_tr
    
    return alpha_factor
```
