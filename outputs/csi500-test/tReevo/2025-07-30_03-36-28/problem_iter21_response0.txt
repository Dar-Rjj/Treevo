```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    sma_short_window = 10
    sma_long_window = 50
    vol_adj_volatility_window = 20
    momentum_window = 30
    percentage_change_window = 5
    dynamic_liquidity_window = 20
    
    # Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_window).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_window).mean()
    
    # Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Weighted_HL'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adj_Volatility'] = df['Volume_Weighted_HL'].rolling(window=vol_adj_volatility_window).mean()
    
    # Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=momentum_window).mean()
    
    # Additional Price Change Metrics
    df['Close_Percentage_Change'] = df['close'].pct_change(periods=percentage_change_window)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Market Trend Alignment
    df['Trend_Indicator'] = np.where(df['SMA_Short'] > df['SMA_Long'], 1, -1)
    
    # Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Avg_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=dynamic_liquidity_window).mean()
    
    # Final Alpha Factor
    bullish_weights = {
        'Price_Momentum': 0.4,
        'Vol_Adj_Volatility': -0.2,
        'Close_Percentage_Change': 0.2,
        'High_Low_Range': 0.1,
        'Avg_Daily_Turnover': 0.1
    }
    
    bearish_weights = {
        'Price_Momentum': 0.3,
        'Vol_Adj_Volatility': -0.1,
        'Close_Percentage_Change': 0.2,
        'High_Low_Range': 0.1,
        'Avg_Daily_Turnover': 0.3
    }
    
    df['Alpha_Factor'] = 0
    for component in bullish_weights.keys():
        df['Alpha_Factor'] += np.where(df['Trend_Indicator'] == 1, bullish_weights[component], bearish_weights[component]) * df[component]
    
    return df['Alpha_Factor']
```
