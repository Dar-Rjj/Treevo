```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()

    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_weighted_high_low = high_low_diff * df['volume']
    vol_adjusted_volatility = vol_weighted_high_low.rolling(window=10).mean()

    # Compute Price Momentum
    price_momentum = (df['close'] - sma_20) / df['close'].rolling(window=5).mean()

    # Incorporate Additional Price Change Metrics
    pct_change_close = df['close'].pct_change(periods=1)
    high_low_range = df['high'] - df['low']

    # Consider Market Trend Alignment
    sma_100 = df['close'].rolling(window=100).mean()
    market_trend = (sma_20 > sma_100).astype(int)  # 1 if Bullish, 0 if Bearish

    # Define Weights for the Components
    weights = {
        'price_momentum': 0.4,
        'vol_adjusted_volatility': 0.3,
        'pct_change_close': 0.2,
        'market_trend': 0.1
    }

    # Adjust Weights Based on Market Trend
    adjusted_weights = {key: (val + 0.1 if market_trend[-1] == 1 else val - 0.1) for key, val in weights.items()}
    adjusted_weights = {key: max(0, min(1, val)) for key, val in adjusted_weights.items()}  # Ensure weights are between 0 and 1
    total_weight = sum(adjusted_weights.values())
    adjusted_weights = {key: val / total_weight for key, val in adjusted_weights.items()}  # Normalize weights

    # Final Alpha Factor
    alpha_factor = (
        adjusted_weights['price_momentum'] * price_momentum +
        adjusted_weights['vol_adjusted_volatility'] * vol_adjusted_volatility +
        adjusted_weights['pct_change_close'] * pct_change_close +
        adjusted_weights['market_trend'] * market_trend
    )

    return alpha_factor
```
