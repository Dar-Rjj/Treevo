```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Close Price Momentum
    close_t = df['close']
    close_t_minus_10 = df['close'].shift(10)
    price_momentum_10d = close_t - close_t_minus_10
    
    # Measure 10-Day Volume Growth
    avg_volume_10d = df['volume'].rolling(window=10).mean().shift(1)
    volume_growth_10d = df['volume'] - avg_volume_10d
    
    # Combine Momentum and Volume
    improved_price_momentum_with_volume_influence = price_momentum_10d * volume_growth_10d
    
    return improved_price_momentum_with_volume_influence
```
