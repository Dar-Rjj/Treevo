import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    sma_lookback = 20
    vol_avg_lookback = 20
    momentum_n = 10
    pct_change_lookback = 5
    short_sma_lookback = 10
    long_sma_lookback = 50
    liquidity_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Refined Volatility Measure
    df['TrueRange'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['VolumeWeightedTR'] = df['volume'] * df['TrueRange']
    df['RefinedVolatility'] = df['VolumeWeightedTR'].rolling(window=vol_avg_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['PctChange'] = df['close'].pct_change(periods=pct_change_lookback)
    df['HighLowRange'] = df['high'] - df['low']
    
    # Consider Adaptive Trend-Following
    df['ShortSMA'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['LongSMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['AdaptiveTrend'] = np.where(df['ShortSMA'] > df['LongSMA'], 1, -1)
    
    # Incorporate Dynamic Liquidity Measures
    df['DailyTurnover'] = df['volume'] * df['close']
    df['Liquidity'] = df['DailyTurnover'].rolling(window=liquidity_lookback).mean()
    df['LiquidityWeight'] = df['Liquidity'] / df['Liquidity'].rolling(window=liquidity_lookback).max()
    
    # Final Alpha Factor
    # Define weights for each component
    w_momentum = 0.3
    w_volatility = 0.2
    w_pct_change = 0.2
    w_high_low_range = 0.1
    w_adaptive_trend = 0.1
    w_liquidity = 0.1
    
    # Adjust weights based on adaptive trend
    df['AdaptiveTrendWeight'] = np.where(df['AdaptiveTrend'] == 1, 1.5, 0.5)
    df['MomentumWeighted'] = df['Momentum'] * w_momentum * df['AdaptiveTrendWeight']
    df['VolatilityWeighted'] = df['RefinedVolatility'] * w_volatility * df['AdaptiveTrendWeight']
    df['PctChangeWeighted'] = df['PctChange'] * w_pct_change * df['AdaptiveTrendWeight']
    df['HighLowRangeWeighted'] = df['HighLowRange'] * w_high_low_range * df['AdaptiveTrendWeight']
    df['LiquidityWeighted'] = df['LiquidityWeight'] * w_liquidity
    
    # Combine all components to form the alpha factor
    df['AlphaFactor'] = (df['MomentumWeighted'] + 
                         df['VolatilityWeighted'] + 
                         df['PctChangeWeighted'] + 
                         df['HighLowRangeWeighted'] + 
                         df['LiquidityWeighted'])
    
    return df['AlphaFactor']
