```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    sma_short_window = 20
    vol_adj_volatility_window = 10
    momentum_lookback = 5
    pct_change_window = 1
    long_sma_window = 100
    turnover_rolling_avg_window = 30
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_window).mean()
    
    # Compute Refined Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Weighted_HL'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adjusted_Volatility'] = df['Volume_Weighted_HL'].rolling(window=vol_adj_volatility_window).std()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_window)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['SMA_Long'] = df['close'].rolling(window=long_sma_window).mean()
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_Avg'] = df['Daily_Turnover'].rolling(window=turnover_rolling_avg_window).mean()
    
    # Define the weights for the components
    bullish_weights = [0.5, 0.25, 0.15, 0.05, 0.05]
    bearish_weights = [0.3, 0.4, 0.15, 0.05, 0.05]
    
    # Adjust weights based on market trend and liquidity
    df['Alpha_Factor'] = 0
    for i, row in df.iterrows():
        if row['Trend_Indicator'] == 1:
            weights = bullish_weights
        else:
            weights = bearish_weights
        
        liquidity_weight_factor = 1 + (row['Turnover_Avg'] / df['Turnover_Avg'].mean() - 1) * 0.5
        adjusted_weights = [w * liquidity_weight_factor for w in weights]
        
        alpha_factor = (
            adjusted_weights[0] * row['Momentum'] +
            adjusted_weights[1] * row['Vol_Adjusted_Volatility'] +
            adjusted_weights[2] * (row['Pct_Change'] + row['High_Low_Range']) +
            adjusted_weights[3] * row['Trend_Indicator'] +
            adjusted_weights[4] * row['Turnover_Avg']
        )
        
        df.at[i, 'Alpha_Factor'] = alpha_factor
    
    return df['Alpha_Factor']
```
