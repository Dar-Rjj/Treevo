```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_short_period = 10
    sma_long_period = 50
    atr_period = 14
    pct_change_period = 5
    daily_turnover_lookback = 20
    
    # Calculate Simple Moving Averages
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Calculate Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=sma_short_period).mean()
    
    # Calculate True Range and Average True Range (ATR)
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x[0], x[1]), axis=1) - df['close'].shift(1)
    df['ATR'] = df['True_Range'].rolling(window=atr_period).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=sma_short_period).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_period)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Incorporate Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Avg_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=daily_turnover_lookback).mean()
    
    # Weights for the final alpha factor
    weights = {
        'Price_Momentum': 0.3,
        'ATR': 0.2,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Avg_Daily_Turnover': 0.2
    }
    
    # Adjust weights based on market trend and liquidity
    df['Weight_Adjustment'] = 1.0
    df.loc[df['Trend_Indicator'] == 1, 'Weight_Adjustment'] = 1.1  # Increase weight for bullish trends
    df.loc[df['Trend_Indicator'] == 0, 'Weight_Adjustment'] = 0.9  # Decrease weight for bearish trends
    df.loc[df['Avg_Daily_Turnover'] > df['Avg_Daily_Turnover'].mean(), 'Weight_Adjustment'] = 1.1  # Increase weight for higher liquidity
    df.loc[df['Avg_Daily_Turnover'] <= df['Avg_Daily_Turnover'].mean(), 'Weight_Adjustment'] = 0.9  # Decrease weight for lower liquidity
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        weights['Price_Momentum'] * df['Price_Momentum'] +
        weights['ATR'] * df['ATR'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Avg_Daily_Turnover'] * df['Avg_Daily_Turnover']
    ) * df['Weight_Adjustment']
    
    return df['Alpha_Factor']
```
