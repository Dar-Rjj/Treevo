import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=200, vol_lookback=30, momentum_n=10, pct_change_period=10, turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=long_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['volume'] * df['High_Low_Diff']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_period)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Adjustment'] = 1 + (df['Rolling_Avg_Turnover'] - df['Daily_Turnover'].mean()) / df['Daily_Turnover'].std()
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': -0.3,
        'Pct_Change': 0.2,
        'High_Low_Range': 0.1,
        'Liquidity_Adjustment': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Liquidity_Adjustment'] * df['Liquidity_Adjustment']
    )
    
    # Adjust Weights Dynamically
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + 0.2 * (df['Trend_Indicator'] * 2 - 1))
    
    return df['Alpha_Factor']
