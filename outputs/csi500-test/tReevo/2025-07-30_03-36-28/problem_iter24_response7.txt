```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    short_sma_period = 10
    long_sma_period = 50
    vol_lookback = 20
    momentum_lookback = 30
    pct_change_lookback = 5
    liquidity_lookback = 30

    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low_diff'] = df['high_low_diff'] * df['volume']
    df['volume_adjusted_volatility'] = df['vol_weighted_high_low_diff'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = (df['SMA_short'] > df['SMA_long']).astype(int)

    # Incorporate Enhanced Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['rolling_daily_turnover'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['average_dollar_volume'] = df['volume'] * df['close']
    df['rolling_average_dollar_volume'] = df['average_dollar_volume'].rolling(window=liquidity_lookback).mean()
    df['liquidity_ratio'] = df['volume'] / df['close']
    df['rolling_liquidity_ratio'] = df['liquidity_ratio'].rolling(window=liquidity_lookback).mean()

    # Adjust Weights Based on Daily Turnover, Average Dollar Volume, and Liquidity Ratio
    df['weight_daily_turnover'] = df['rolling_daily_turnover'] / df['rolling_daily_turnover'].max()
    df['weight_average_dollar_volume'] = df['rolling_average_dollar_volume'] / df['rolling_average_dollar_volume'].max()
    df['weight_liquidity_ratio'] = df['rolling_liquidity_ratio'] / df['rolling_liquidity_ratio'].max()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'volume_adjusted_volatility': 0.2,
        'pct_change_close': 0.1,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'weight_daily_turnover': 0.1,
        'weight_average_dollar_volume': 0.1,
        'weight_liquidity_ratio': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['volume_adjusted_volatility'] * df['volume_adjusted_volatility'] +
        weights['pct_change_close'] * df['pct_change_close'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['trend_indicator'] * df['trend_indicator'] +
        weights['weight_daily_turnover'] * df['weight_daily_turnover'] +
        weights['weight_average_dollar_volume'] * df['weight_average_dollar_volume'] +
        weights['weight_liquidity_ratio'] * df['weight_liquidity_ratio']
    )

    return df['alpha_factor']
```
