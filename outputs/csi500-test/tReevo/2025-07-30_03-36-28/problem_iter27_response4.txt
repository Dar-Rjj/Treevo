```python
import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=252, vol_lookback=20, momentum_lookback=120, pct_change_lookback=20, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adjusted_volatility'] = df['vol_weighted_high_low'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = 0
    df.loc[df['SMA_short'] > df['SMA_long'], 'trend_indicator'] = 1  # Bullish
    df.loc[df['SMA_short'] < df['SMA_long'], 'trend_indicator'] = -1  # Bearish

    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['rolling_turnover'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    df['liquidity_weight'] = df['rolling_turnover'] / df['rolling_turnover'].max()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'vol_adjusted_volatility': -0.3,
        'pct_change': 0.2,
        'high_low_range': -0.1,
        'trend_indicator': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adjusted_volatility'] * df['vol_adjusted_volatility'] +
        weights['pct_change'] * df['pct_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['trend_indicator'] * df['trend_indicator']
    )

    # Adjust Weights Dynamically Based on Trend
    df['alpha_factor'] = df['alpha_factor'] * (1 + df['trend_indicator'] * 0.1)

    # Apply liquidity weight
    df['alpha_factor'] = df['alpha_factor'] * df['liquidity_weight']

    return df['alpha_factor']
```
