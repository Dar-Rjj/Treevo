import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=100, vol_lookback=20, momentum_n=20, pct_change_period=10, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_diff'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_weighted_diff'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_n).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_period)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()
    df['trend'] = df.apply(lambda row: 1 if row['SMA_short'] > row['SMA_long'] else -1, axis=1)

    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover_rolling_avg'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    df['liquidity_factor'] = df['turnover_rolling_avg'] / df['daily_turnover']

    # Final Alpha Factor
    weights = {
        'momentum': 0.4,
        'volatility': -0.3,
        'pct_change': 0.2,
        'high_low_range': 0.1,
        'liquidity_factor': 0.1
    }
    
    df['alpha_factor'] = (weights['momentum'] * df['momentum'] +
                          weights['volatility'] * df['volatility'] +
                          weights['pct_change'] * df['pct_change'] +
                          weights['high_low_range'] * df['high_low_range'] +
                          weights['liquidity_factor'] * df['liquidity_factor'])
    
    # Adjust Weights Dynamically Based on Market Trend
    df['alpha_factor'] *= df['trend']

    return df['alpha_factor']
