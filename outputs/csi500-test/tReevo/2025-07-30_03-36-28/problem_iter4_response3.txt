import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    # Calculate True Range
    tr = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    # Apply Volume Weighting
    vw_tr = tr * df['volume']
    # Calculate Rolling Average of Volume-Weighted True Range
    vol_adj_volatility = vw_tr.rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adj_volatility
    return alpha_factor
