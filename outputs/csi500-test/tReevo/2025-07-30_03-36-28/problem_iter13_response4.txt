```python
import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=200, volatility_lookback=20, price_change_lookback=10, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_weighted_high_low'].rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    df['avg_close'] = df['close'].rolling(window=short_sma_period).mean()
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['avg_close']

    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=price_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = (df['SMA_short'] > df['SMA_long']).astype(int)

    # Incorporate Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover_rolling_avg'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'volatility': -0.2,
        'pct_change': 0.2,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'turnover_rolling_avg': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['volatility'] * df['volatility'] +
        weights['pct_change'] * df['pct_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['trend_indicator'] * df['trend_indicator'] +
        weights['turnover_rolling_avg'] * df['turnover_rolling_avg']
    )

    # Adjust Weights Based on Market Trend
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * 1.1 if row['trend_indicator'] == 1 else row['alpha_factor'], axis=1)
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * 0.9 if row['trend_indicator'] == 0 else row['alpha_factor'], axis=1)

    # Adjust Weights Based on Liquidity
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * 1.05 if row['turnover_rolling_avg'] > df['turnover_rolling_avg'].median() else row['alpha_factor'], axis=1)
    df['alpha_factor'] = df.apply(lambda row: row['alpha_factor'] * 0.95 if row['turnover_rolling_avg'] <= df['turnover_rolling_avg'].median() else row['alpha_factor'], axis=1)

    return df['alpha_factor']
```
