```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=50, vol_lookback=20, momentum_lookback=30, trend_lookback=20, liquidity_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_vol'] = df['high_low_diff'] * df['volume']
    df['volume_adjusted_vol'] = df['volume_weighted_vol'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Recent Price Action
    df['pct_change'] = df['close'].pct_change()
    df['high_low_range'] = df['high'] - df['low']
    
    # Refine Volatility Adjustments
    df['std_dev'] = df['close'].rolling(window=vol_lookback).std()
    df['refined_vol'] = df['volume_adjusted_vol'] * df['std_dev']
    
    # Consider Market Trend Alignment
    df['long_term_SMA'] = df['close'].rolling(window=long_window).mean()
    df['trend_indicator'] = (df['SMA'] > df['long_term_SMA']).astype(int)
    
    # Enhance Trend and Liquidity Factor Integration
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['liquidity_factor'] = (df['volume'] / df['avg_daily_turnover']).clip(lower=0.5, upper=1.5)
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'refined_vol': -0.2,
        'pct_change': 0.1,
        'high_low_range': -0.1,
        'trend_indicator': 0.1,
        'liquidity_factor': 0.1
    }
    
    # Adjust Weights Dynamically Based on Market Trend
    df['adjusted_trend_weight'] = df['trend_indicator'] * 0.2 + 0.8
    for key in weights:
        weights[key] *= df['adjusted_trend_weight']
    
    # Ensure the Weights Sum to 1
    total_weight = sum(weights.values())
    for key in weights:
        weights[key] /= total_weight
    
    # Combine Components
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['refined_vol'] * df['refined_vol'] +
        weights['pct_change'] * df['pct_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['trend_indicator'] * df['trend_indicator'] +
        weights['liquidity_factor'] * df['liquidity_factor']
    )
    
    return df['alpha_factor']
```
