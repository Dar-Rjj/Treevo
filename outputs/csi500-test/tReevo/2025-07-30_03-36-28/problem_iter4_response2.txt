```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute True Range
    high_low_diff = df['high'] - df['low']
    high_close_diff = (df['high'] - df['close']).abs()
    low_close_diff = (df['low'] - df['close']).abs()
    true_range = pd.concat([high_low_diff, high_close_diff, low_close_diff], axis=1).max(axis=1)
    
    # Compute Volume-Weighted True Range
    volume_weighted_true_range = true_range * df['volume']
    
    # Compute Volume-Adjusted Volatility
    rolling_sum_vol_weighted_tr = volume_weighted_true_range.rolling(window=lookback_period).sum()
    rolling_sum_volume = df['volume'].rolling(window=lookback_period).sum()
    vol_adjusted_volatility = rolling_sum_vol_weighted_tr / rolling_sum_volume
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma_close
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_volatility
    
    return alpha_factor
```
