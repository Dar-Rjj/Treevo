```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['log_return'].rolling(window=5).sum()

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['log_return'].rolling(window=20).sum()

    # Subtract Short-Term Momentum from Long-Term Momentum
    df['momentum_diff'] = df['long_term_momentum'] - df['short_term_momentum']

    # Weight by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['weighted_momentum'] = df['momentum_diff'] * df['avg_volume']

    # Incorporate Price Volatility
    df['price_volatility'] = df['log_return'].rolling(window=20).std()
    df['weighted_momentum_vol'] = df['weighted_momentum'] / df['price_volatility']

    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Calculate Intraday Close-Open Range
    df['intraday_close_open_range'] = df['close'] - df['open']

    # Compute Intraday Volatility Skew
    df['intraday_vol_skew'] = (df['intraday_high_low_range'] - df['intraday_close_open_range']) / df['intraday_high_low_range']

    # Adjust for Volume Impact
    df['intraday_vol_skew_vol_adj'] = df['intraday_vol_skew'] * df['volume'] ** (1/3)

    # Detect Volume Shocks
    df['daily_vol_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['avg_daily_vol_change'] = df['daily_vol_change'].rolling(window=20).mean()
    df['volume_shock'] = (df['daily_vol_change'] > 3 * df['avg_daily_vol_change']).astype(int)

    # Integrate Factors
    df['alpha_factor'] = (df['weighted_momentum_vol'] + df['intraday_vol_skew_vol_adj']) * df['volume_shock']

    return df['alpha_factor']
```
