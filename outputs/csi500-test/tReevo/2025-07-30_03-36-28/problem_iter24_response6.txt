import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_n=10, pct_change_lookback=5, long_term_sma_lookback=50, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volume_Adjusted_Volatility_SMA'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=long_term_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_SMA'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Factor'] = df['Turnover_SMA'] / df['Daily_Turnover'].rolling(window=turnover_lookback).std()
    
    # Final Alpha Factor
    # Define the Weights for Each Component
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility_SMA': 0.2,
        'Pct_Change_Close': 0.1,
        'High_Low_Range': 0.1,
        'Liquidity_Factor': 0.2
    }
    
    # Adjust Weights Dynamically Based on Market Trend
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility_SMA'] * df['Volume_Adjusted_Volatility_SMA'] +
        weights['Pct_Change_Close'] * df['Pct_Change_Close'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Liquidity_Factor'] * df['Liquidity_Factor']
    )
    
    df['Alpha_Factor'] = df['Alpha_Factor'] * (df['Trend_Indicator'] * 0.2 + 0.8)
    
    return df['Alpha_Factor']
