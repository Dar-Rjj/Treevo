```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    sma_lookback = 20
    vol_adj_volatility_lookback = 10
    momentum_lookback = 5
    percentage_change_lookback = 1
    previous_close_lookback = 1
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volume_Adjusted_Volatility'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_adj_volatility_lookback).mean()
    
    # Compute Price Momentum
    df['Close_Price_Mean'] = df['close'].rolling(window=momentum_lookback).mean()
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['Close_Price_Mean']
    
    # Incorporate Additional Price Change Metrics
    df['Percentage_Change'] = df['close'].pct_change(percentage_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Prev_Close_Price'] = df['close'].shift(previous_close_lookback)
    df['Price_Change_Prev_Close'] = df['close'] - df['Prev_Close_Price']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        0.4 * df['Price_Momentum'] +
        0.3 * df['Volume_Adjusted_Volatility'] +
        0.2 * df['Percentage_Change'] +
        0.05 * df['High_Low_Range'] / df['close'] +  # Normalize High-Low Range
        0.05 * df['Open_Close_Diff'] / df['close']   # Normalize Open-Close Difference
    )
    
    return df['Alpha_Factor']
```
