```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_weighted_high_low_diff = high_low_diff * df['volume']
    sum_vol_weighted_high_low_diff = vol_weighted_high_low_diff.rolling(window=lookback_period).sum()
    sum_volume = df['volume'].rolling(window=lookback_period).sum()
    volume_adjusted_volatility = sum_vol_weighted_high_low_diff / sum_volume
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma_close
    avg_close = df['close'].rolling(window=lookback_period).mean()
    normalized_price_momentum = price_momentum / avg_close
    
    # Final Alpha Factor
    alpha_factor = normalized_price_momentum / volume_adjusted_volatility
    
    return alpha_factor
```
