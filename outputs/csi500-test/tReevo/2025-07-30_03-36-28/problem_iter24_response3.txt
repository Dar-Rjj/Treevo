```python
import pandas as pd

def heuristics_v2(df, short_sma_lookback=20, long_sma_lookback=250, vol_lookback=30, momentum_lookback=60, pct_change_lookback=10, turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['SMA_Long'] = df['close'].rolling(window=long_sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Vol'] = df['High_Low_Diff'] * df['volume']
    df['Volatility'] = df['Volume_Adjusted_Vol'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']

    # Consider Market Trend Alignment
    df['Trend_Indicator'] = df['SMA_Short'] > df['SMA_Long']
    df['Trend_Indicator'] = df['Trend_Indicator'].astype(int)

    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_Avg'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()

    # Final Alpha Factor
    bullish_weight = 0.7  # Example weight for bullish trend
    bearish_weight = 0.3  # Example weight for bearish trend
    default_weights = {
        'Momentum': 0.4,
        'Volatility': 0.3,
        'Pct_Change': 0.2,
        'High_Low_Range': 0.1
    }

    def adjust_weights(trend, weights):
        if trend == 1:
            return {k: v * bullish_weight for k, v in weights.items()}
        else:
            return {k: v * bearish_weight for k, v in weights.items()}

    df['Alpha_Factor'] = 0
    for date in df.index:
        trend = df.loc[date, 'Trend_Indicator']
        adjusted_weights = adjust_weights(trend, default_weights)
        alpha_factor = (df.loc[date, 'Momentum'] * adjusted_weights['Momentum'] +
                        df.loc[date, 'Volatility'] * adjusted_weights['Volatility'] +
                        df.loc[date, 'Pct_Change'] * adjusted_weights['Pct_Change'] +
                        df.loc[date, 'High_Low_Range'] * adjusted_weights['High_Low_Range'])
        df.loc[date, 'Alpha_Factor'] = alpha_factor

    return df['Alpha_Factor']
```
