```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_volatility = (high_low_diff * df['volume']).rolling(window=lookback_period).mean()
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma_close
    normalized_price_momentum = price_momentum / df['close'].rolling(window=lookback_period).mean()
    
    # Final Alpha Factor
    alpha_factor = normalized_price_momentum / volume_weighted_volatility
    
    return alpha_factor
```
