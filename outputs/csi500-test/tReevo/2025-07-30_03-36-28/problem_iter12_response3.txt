import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_adj_vol_lookback=10, pct_change_lookback=5, long_sma_lookback=200, daily_turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['sma'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adj_vol'] = df['vol_weighted_high_low'].rolling(window=vol_adj_vol_lookback).mean()
    
    # Compute Price Momentum
    df['avg_close'] = df['close'].rolling(window=sma_lookback).mean()
    df['price_momentum'] = (df['close'] - df['sma']) / df['avg_close']
    
    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['long_sma'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['trend'] = df['sma'] > df['long_sma']
    df['trend_indicator'] = df['trend'].astype(int)
    
    # Incorporate Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=daily_turnover_lookback).mean()
    
    # Final Alpha Factor
    base_weights = {
        'price_momentum': 0.4,
        'vol_adj_vol': 0.3,
        'pct_change': 0.1,
        'trend_indicator': 0.1,
        'avg_daily_turnover': 0.1
    }
    
    df['alpha_factor'] = (
        base_weights['price_momentum'] * df['price_momentum'] +
        base_weights['vol_adj_vol'] * df['vol_adj_vol'] +
        base_weights['pct_change'] * df['pct_change'] +
        base_weights['trend_indicator'] * df['trend_indicator'] +
        base_weights['avg_daily_turnover'] * df['avg_daily_turnover']
    )
    
    # Adjust Weights Based on Market Trend
    df.loc[df['trend'], 'alpha_factor'] = (
        (base_weights['price_momentum'] + 0.1) * df['price_momentum'] +
        base_weights['vol_adj_vol'] * df['vol_adj_vol'] +
        base_weights['pct_change'] * df['pct_change'] +
        (base_weights['trend_indicator'] - 0.1) * df['trend_indicator'] +
        base_weights['avg_daily_turnover'] * df['avg_daily_turnover']
    )
    
    df.loc[~df['trend'], 'alpha_factor'] = (
        (base_weights['price_momentum'] - 0.1) * df['price_momentum'] +
        base_weights['vol_adj_vol'] * df['vol_adj_vol'] +
        base_weights['pct_change'] * df['pct_change'] +
        (base_weights['trend_indicator'] + 0.1) * df['trend_indicator'] +
        base_weights['avg_daily_turnover'] * df['avg_daily_turnover']
    )
    
    # Adjust Weights Based on Liquidity
    high_liquidity_mask = df['avg_daily_turnover'] > df['avg_daily_turnover'].quantile(0.75)
    low_liquidity_mask = df['avg_daily_turnover'] < df['avg_daily_turnover'].quantile(0.25)
    
    df.loc[high_liquidity_mask, 'alpha_factor'] = (
        base_weights['price_momentum'] * df['price_momentum'] +
        (base_weights['vol_adj_vol'] + 0.05) * df['vol_adj_vol'] +
        base_weights['pct_change'] * df['pct_change'] +
        base_weights['trend_indicator'] * df['trend_indicator'] +
