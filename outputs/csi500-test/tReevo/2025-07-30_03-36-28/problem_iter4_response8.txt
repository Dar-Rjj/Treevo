import pandas as pd

def heuristics_v2(df, base_lookback=20):
    # Calculate the rolling standard deviation of close prices
    rolling_std = df['close'].rolling(window=base_lookback).std()
    
    # Define the dynamic lookback period for SMA based on market volatility
    def adjust_lookback(rolling_std, base_lookback):
        if rolling_std.iloc[-1] > rolling_std.median():
            return max(5, int(base_lookback * 0.8))  # Shorter lookback in high volatility
        else:
            return min(40, int(base_lookback * 1.2))  # Longer lookback in low volatility
    
    lookback = adjust_lookback(rolling_std, base_lookback)
    
    # Calculate the Simple Moving Average (SMA) of close prices
    sma = df['close'].rolling(window=lookback).mean()
    
    # Compute High-Low price difference
    high_low_diff = df['high'] - df['low']
    
    # Apply volume weighting to the High-Low price difference
    vol_weighted_high_low = high_low_diff * df['volume']
    
    # Calculate the rolling average of the volume-weighted High-Low differences
    vol_adjusted_volatility = vol_weighted_high_low.rolling(window=lookback).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=lookback).sum()
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_volatility
    
    return alpha_factor
