```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=20, momentum_lookback=20, pct_change_lookback=5, long_sma_lookback=60, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    df['Long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    df['Liquidity_Adjustment'] = df['Rolling_Turnover'] / df['Daily_Turnover'].rolling(window=turnover_lookback).std()
    
    # Define Weights for Each Component
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': 0.1,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Liquidity_Adjustment': 0.2
    }
    
    # Ensure the Weights Sum to 1
    assert sum(weights.values()) == 1
    
    # Adjust Weights Dynamically Based on Market Trend
    df['Dynamic_Weight_Momentum'] = df['Trend_Indicator'] * weights['Momentum'] + (1 - df['Trend_Indicator']) * weights['Momentum'] * 0.8
    df['Dynamic_Weight_VAV'] = df['Trend_Indicator'] * weights['Volume_Adjusted_Volatility'] + (1 - df['Trend_Indicator']) * weights['Volume_Adjusted_Volatility'] * 0.8
    df['Dynamic_Weight_Pct_Change'] = df['Trend_Indicator'] * weights['Pct_Change'] + (1 - df['Trend_Indicator']) * weights['Pct_Change'] * 0.8
    df['Dynamic_Weight_HL_Range'] = df['Trend_Indicator'] * weights['High_Low_Range'] + (1 - df['Trend_Indicator']) * weights['High_Low_Range'] * 0.8
    df['Dynamic_Weight_Liquidity'] = df['Trend_Indicator'] * weights['Liquidity_Adjustment'] + (1 - df['Trend_Indicator']) * weights['Liquidity_Adjustment'] * 0.8
    
    # Combine the Components
    df['Alpha_Factor'] = (
        df['Dynamic_Weight_Momentum'] * df['Momentum'] +
        df['Dynamic_Weight_VAV'] * df['Volume_Adjusted_Volatility'] +
        df['Dynamic_Weight_Pct_Change'] * df['Pct_Change'] +
        df['Dynamic_Weight_HL_Range'] * df['High_Low_Range'] +
        df['Dynamic_Weight_Liquidity'] * df['Liquidity_Adjustment']
    )
    
    return df['Alpha_Factor']
```
