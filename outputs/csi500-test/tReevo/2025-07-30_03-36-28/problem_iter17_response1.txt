import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, n_close_prices=5, pct_change_lookback=5, long_sma_lookback=200, liquidity_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High-Low'] = df['high'] - df['low']
    df['Volume-Adjusted_Volatility'] = (df['High-Low'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=n_close_prices).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High-Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Long_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=liquidity_lookback).mean()
    df['Liquidity_Adjustment'] = (df['Rolling_Avg_Turnover'] / df['Daily_Turnover']).fillna(1)
    
    # Final Alpha Factor
    weights = {
        'Price_Momentum': 0.4,
        'Volume-Adjusted_Volatility': -0.3,
        'Pct_Change': 0.2,
        'High-Low_Range': 0.1,
        'Trend_Indicator': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Price_Momentum'] * df['Price_Momentum'] +
        weights['Volume-Adjusted_Volatility'] * df['Volume-Adjusted_Volatility'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High-Low_Range'] * df['High-Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator']
    ) * df['Liquidity_Adjustment']
    
    return df['Alpha_Factor']
