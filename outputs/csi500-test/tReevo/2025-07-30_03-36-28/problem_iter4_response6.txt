```python
import pandas as pd

def heuristics_v2(df, lookback_sma=50, lookback_volatility=20, lookback_percentage_change=10, weight_momentum=0.5, weight_volatility=0.3, weight_additional=0.2):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=lookback_sma).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_high_low = high_low_diff * df['volume']
    volume_adjusted_volatility = volume_weighted_high_low.rolling(window=lookback_volatility).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=lookback_sma).mean()
    
    # Incorporate Additional Price Change Metrics
    percentage_change_close = df['close'].pct_change(periods=lookback_percentage_change)
    high_low_range = df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = (
        weight_momentum * price_momentum + 
        weight_volatility * volume_adjusted_volatility + 
        weight_additional * (percentage_change_close + high_low_range)
    )
    
    return alpha_factor
```
