```python
import pandas as pd

def heuristics_v2(df, lookback_sma=20, lookback_volatility=20):
    """
    Calculate the Price Momentum with Volume-Adjusted Volatility.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    - lookback_sma: int, the lookback period for the Simple Moving Average.
    - lookback_volatility: int, the lookback period for the rolling average of the volume-weighted combined differences.
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=lookback_sma).mean()
    
    # Compute Volume-Adjusted Volatility
    # Calculate High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate High-Close and Low-Close Price Differences
    df['high_close_diff'] = df['high'] - df['close'].shift(1)
    df['low_close_diff'] = df['close'].shift(1) - df['low']
    
    # Combine Price Differences
    df['max_diff'] = df[['high_low_diff', 'high_close_diff', 'low_close_diff']].max(axis=1)
    
    # Apply Volume Weighting
    df['vol_weighted_max_diff'] = df['volume'] * df['max_diff']
    
    # Calculate the Rolling Average of the Volume-Weighted Combined Differences
    df['vol_adjusted_volatility'] = df['vol_weighted_max_diff'].rolling(window=lookback_volatility).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=lookback_sma).mean()
    
    # Final Alpha Factor
    alpha_factor = df['price_momentum'] / df['vol_adjusted_volatility']
    
    return alpha_factor
```
