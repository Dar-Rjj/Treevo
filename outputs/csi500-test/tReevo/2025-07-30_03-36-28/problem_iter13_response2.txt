```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    short_sma_period = 10
    long_sma_period = 50
    price_change_period = 5
    volatility_lookback = 20
    turnover_lookback = 20

    # Calculate Simple Moving Average (SMA) of Close Prices
    df['short_sma'] = df['close'].rolling(window=short_sma_period).mean()
    df['long_sma'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_volatility'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_weighted_volatility'].rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['short_sma']) / df['close'].rolling(window=short_sma_period).mean()

    # Incorporate Additional Price Change Metrics
    df['close_pct_change'] = df['close'].pct_change(periods=price_change_period)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = np.where(df['short_sma'] > df['long_sma'], 'Bullish', 'Bearish')

    # Incorporate Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'volatility': -0.3,
        'close_pct_change': 0.2,
        'high_low_range': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['volatility'] * df['volatility'] +
        weights['close_pct_change'] * df['close_pct_change'] +
        weights['high_low_range'] * df['high_low_range']
    )

    # Adjust Weights Based on Market Trend
    df['alpha_factor'] = np.where(df['trend_indicator'] == 'Bullish', df['alpha_factor'] * 1.1, df['alpha_factor'] * 0.9)

    # Adjust Weights Based on Liquidity
    df['alpha_factor'] = np.where(df['turnover'] > df['turnover'].mean(), df['alpha_factor'] * 1.1, df['alpha_factor'] * 0.9)

    return df['alpha_factor']
```
