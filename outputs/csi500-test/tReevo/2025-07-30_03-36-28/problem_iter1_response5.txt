import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    df['HighDiff'] = df['high'] - df['high'].shift(1)
    df['LowDiff'] = df['low'] - df['low'].shift(1)
    df['HighLowMomentum'] = (df['HighDiff'] + df['LowDiff']) / 2
    
    # Calculate Price Range Expansion
    df['TodayRange'] = df['high'] - df['low']
    df['PrevDayRange'] = df['TodayRange'].shift(1)
    df['RangeExpansion'] = df['TodayRange'] - df['PrevDayRange']
    
    # Volume Activity Indicator
    df['Volume20DMA'] = df['volume'].rolling(window=20).mean()
    df['VolumeActivity'] = df.apply(lambda row: row['HighLowMomentum'] * row['volume'] if row['volume'] > row['Volume20DMA'] else 0, axis=1)
    
    # Final Alpha Factor
    alpha_factor = df['VolumeActivity']
    
    return alpha_factor
