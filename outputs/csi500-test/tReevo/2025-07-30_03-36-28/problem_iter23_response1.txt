```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_adj_vol_lookback=10, momentum_n=5, pct_change_lookback=5, long_term_sma_lookback=60, liquidity_lookback=20):
    # Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_high_low = high_low_diff * df['volume']
    vol_adj_vol = volume_weighted_high_low.rolling(window=vol_adj_vol_lookback).mean()
    
    # Price Momentum
    avg_close = df['close'].rolling(window=momentum_n).mean()
    price_momentum = (df['close'] - sma) / avg_close
    
    # Percentage Change in Close Price
    pct_change = df['close'].pct_change(periods=pct_change_lookback)
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Long-Term SMA
    long_term_sma = df['close'].rolling(window=long_term_sma_lookback).mean()
    
    # Market Trend Indicator
    trend_indicator = (sma > long_term_sma).astype(int)
    
    # Dynamic Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    rolling_avg_turnover = daily_turnover.rolling(window=liquidity_lookback).mean()
    liquidity_adjustment = (rolling_avg_turnover / rolling_avg_turnover.max()).fillna(0)
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'vol_adj_vol': -0.2,
        'pct_change': 0.1,
        'high_low_range': 0.1,
        'trend_indicator': 0.2,
        'liquidity_adjustment': 0.1
    }
    
    alpha_factor = (
        weights['price_momentum'] * price_momentum +
        weights['vol_adj_vol'] * vol_adj_vol +
        weights['pct_change'] * pct_change +
        weights['high_low_range'] * high_low_range +
        weights['trend_indicator'] * trend_indicator +
        weights['liquidity_adjustment'] * liquidity_adjustment
    )
    
    return alpha_factor
```
