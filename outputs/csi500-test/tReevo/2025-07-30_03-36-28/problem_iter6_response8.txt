```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Apply Volume Weighting to High-Low Price Difference
    vol_weighted_high_low = high_low_diff * df['volume']
    
    # Calculate Rolling Sum of Volume-Weighted High-Low Differences
    rolling_sum_vol_weighted_high_low = vol_weighted_high_low.rolling(window=lookback_period).sum()
    
    # Calculate Rolling Sum of Volume
    rolling_sum_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Adjusted Volatility
    vol_adjusted_volatility = rolling_sum_vol_weighted_high_low / rolling_sum_volume
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma_close
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_volatility
    
    return alpha_factor
```
