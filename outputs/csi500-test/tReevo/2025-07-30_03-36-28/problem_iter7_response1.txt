import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Lookback Simple Moving Average (SMA) of Close Prices
    def dynamic_lookback_sma(close, high, low):
        lookback = 20  # Base lookback period
        trend_alignment = (high - close) / (high - low)
        adjusted_lookback = lookback * (1 + trend_alignment.mean())
        adjusted_lookback = int(max(5, min(100, adjusted_lookback)))  # Ensure lookback is within a reasonable range
        return close.rolling(window=adjusted_lookback).mean()

    # Compute Volume-Adjusted Volatility
    def volume_adjusted_volatility(high, low, volume):
        high_low_diff = high - low
        vol_weighted_diff = high_low_diff * volume
        lookback = 20  # Base lookback period
        recent_volatility = high_low_diff.rolling(window=lookback).std()
        adjusted_lookback = lookback * (1 + recent_volatility.mean())
        adjusted_lookback = int(max(5, min(100, adjusted_lookback)))  # Ensure lookback is within a reasonable range
        return vol_weighted_diff.rolling(window=adjusted_lookback).mean()

    # Compute Price Momentum
    def price_momentum(close, dynamic_sma):
        n = 20  # Lookback period for the average of close prices
        momentum = (close - dynamic_sma) / close.rolling(window=n).mean()
        return momentum

    # Incorporate Additional Price Change Metrics
    def additional_price_change_metrics(close, high, low):
        pct_change_period = 10  # Lookback period for percentage change
        pct_change = close.pct_change(periods=pct_change_period)
        high_low_range = (high - low) / close
        return pct_change, high_low_range

    # Final Alpha Factor
    def final_alpha_factor(momentum, vol_adj_volatility, pct_change, high_low_range):
        weights = {
            'momentum': 0.4,
            'vol_adj_volatility': 0.3,
            'pct_change': 0.2,
            'high_low_range': 0.1
        }
        alpha_factor = (weights['momentum'] * momentum +
                       weights['vol_adj_volatility'] * vol_adj_volatility +
                       weights['pct_change'] * pct_change +
                       weights['high_low_range'] * high_low_range)
        return alpha_factor

    # Calculate the components
    dynamic_sma = dynamic_lookback_sma(df['close'], df['high'], df['low'])
    vol_adj_volatility = volume_adjusted_volatility(df['high'], df['low'], df['volume'])
    momentum = price_momentum(df['close'], dynamic_sma)
    pct_change, high_low_range = additional_price_change_metrics(df['close'], df['high'], df['low'])

    # Combine the components to get the final alpha factor
    alpha_factor = final_alpha_factor(momentum, vol_adj_volatility, pct_change, high_low_range)

    return alpha_factor
