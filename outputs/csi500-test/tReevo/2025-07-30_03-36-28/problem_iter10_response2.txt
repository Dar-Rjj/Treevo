```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    short_sma_period = 10
    long_sma_period = 50
    vol_adj_volatility_period = 20
    pct_change_period = 10
    daily_turnover_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adj_volatility'] = df['vol_weighted_high_low'].rolling(window=vol_adj_volatility_period).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=short_sma_period).mean()
    
    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(periods=pct_change_period)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['trend_indicator'] = df.apply(lambda row: 'Bullish' if row['SMA_short'] > row['SMA_long'] else 'Bearish', axis=1)
    
    # Incorporate Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=daily_turnover_period).mean()
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'vol_adj_volatility': 0.2,
        'pct_change_close': 0.1,
        'high_low_range': 0.1,
        'avg_daily_turnover': 0.2
    }
    
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adj_volatility'] * df['vol_adj_volatility'] +
        weights['pct_change_close'] * df['pct_change_close'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['avg_daily_turnover'] * df['avg_daily_turnover']
    )
    
    # Adjust Weights Based on Market Trend
    df['alpha_factor'] = df.apply(
        lambda row: row['alpha_factor'] * 1.2 if row['trend_indicator'] == 'Bullish' else row['alpha_factor'] * 0.8, 
        axis=1
    )
    
    # Adjust Weights Based on Liquidity
    df['alpha_factor'] = df.apply(
        lambda row: row['alpha_factor'] * 1.2 if row['avg_daily_turnover'] > df['avg_daily_turnover'].median() else row['alpha_factor'] * 0.8, 
        axis=1
    )
    
    return df['alpha_factor']
```
