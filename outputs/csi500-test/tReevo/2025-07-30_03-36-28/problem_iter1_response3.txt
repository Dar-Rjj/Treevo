import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    close_t1 = df['close'].shift(1)  # Close price at t-1
    close_t10 = df['close'].shift(10)  # Close price at t-10
    price_momentum = close_t1 - close_t10  # Difference to calculate momentum
    
    # Measure Volume Growth
    avg_volume_10days = df['volume'].rolling(window=10).mean().shift(1)  # 10-day average volume until t-1
    today_volume_growth = df['volume'] - avg_volume_10days  # Today's volume growth compared to its average
    
    # Combine Momentum and Volume
    price_momentum_with_volume_influence = price_momentum * today_volume_growth
    
    return price_momentum_with_volume_influence.rename('PriceMomentumWithVolumeInfluence')
