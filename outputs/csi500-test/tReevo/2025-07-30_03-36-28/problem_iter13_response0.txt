import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_n=5, pct_change_lookback=5, long_term_sma_lookback=60, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = (df['high'] - df['low']) / df['close']
    
    # Consider Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=long_term_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int)
    
    # Incorporate Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_lookback).mean()
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Volatility': 0.2,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Rolling_Avg_Turnover': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Rolling_Avg_Turnover'] * df['Rolling_Avg_Turnover']
    )
    
    # Adjust Weights Based on Market Trend
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + 0.1 * (df['Trend_Indicator'] * 2 - 1))
    
    # Adjust Weights Based on Liquidity
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + 0.1 * (df['Rolling_Avg_Turnover'] / df['Rolling_Avg_Turnover'].mean() - 1))
    
    return df['Alpha_Factor']
