```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_adj_vol_lookback=20, pct_change_lookback=10, weight_momentum=0.5, weight_volatility=0.3, weight_pct_change=0.1, weight_range=0.1):
    """
    Generate a novel alpha factor using price momentum, volume-adjusted volatility, and additional metrics.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as (date).
    - sma_lookback: Lookback period for the Simple Moving Average.
    - vol_adj_vol_lookback: Lookback period for the Volume-Adjusted Volatility.
    - pct_change_lookback: Lookback period for the Percentage Change in Close Price.
    - weight_momentum: Weight for the Price Momentum component.
    - weight_volatility: Weight for the Volume-Adjusted Volatility component.
    - weight_pct_change: Weight for the Percentage Change in Close Price component.
    - weight_range: Weight for the High-Low Range component.
    
    Returns:
    - pandas Series representing the alpha factor.
    """
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_weighted_diff = high_low_diff * df['volume']
    vol_adj_vol = vol_weighted_diff.rolling(window=vol_adj_vol_lookback).mean()
    
    # Compute Price Momentum
    avg_n_close = df['close'].rolling(window=sma_lookback).mean()
    price_momentum = (df['close'] - sma) / avg_n_close
    
    # Incorporate Additional Price Change Metrics
    close_pct_change = df['close'].pct_change(periods=pct_change_lookback)
    high_low_range = df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = (weight_momentum * price_momentum 
                    - weight_volatility * vol_adj_vol 
                    + weight_pct_change * close_pct_change 
                    + weight_range * high_low_range)
    
    return alpha_factor
```
