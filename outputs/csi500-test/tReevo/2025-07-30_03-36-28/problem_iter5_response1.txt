```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_n=5, pct_change_lookback=1, weights=(0.5, 0.3, 0.2)):
    """
    Generate an alpha factor based on price momentum with volume-adjusted volatility and additional price change metrics.
    
    Parameters:
    - df: DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    - sma_lookback: Lookback period for Simple Moving Average (SMA) of close prices
    - vol_lookback: Lookback period for the rolling average of volume-weighted high-low differences
    - momentum_n: Lookback period for the average of the last N close prices
    - pct_change_lookback: Lookback period for the percentage change in close price
    - weights: Weights for combining the components (Price Momentum, Volume-Adjusted Volatility, Additional Price Change Metrics)
    
    Returns:
    - Series indexed by (date) representing the factor values
    """
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    vol_adjusted_volatility = (high_low_diff * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    close_avg_n = df['close'].rolling(window=momentum_n).mean()
    price_momentum = (df['close'] - sma) / close_avg_n
    
    # Incorporate Additional Price Change Metrics
    pct_change = df['close'].pct_change(pct_change_lookback)
    high_low_range = df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = (weights[0] * price_momentum +
                    weights[1] * vol_adjusted_volatility +
                    weights[2] * (pct_change + high_low_range))
    
    return alpha_factor
```
