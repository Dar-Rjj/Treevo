```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_lookback = 20
    volatility_lookback = 10
    momentum_lookback = 5
    price_change_lookback = 5
    open_close_lookback = 5
    long_sma_lookback = 100
    short_sma_lookback = 50
    liquidity_lookback = 30

    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Enhanced Price Change Metrics
    df['Close_Percent_Change'] = df['close'].pct_change(periods=price_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Diff'] = df['open'] - df['close']
    df['Open_Close_Diff_Rolling_Avg'] = df['Open_Close_Diff'].rolling(window=open_close_lookback).mean()

    # Consider Refined Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=long_sma_lookback).mean()
    df['Short_Term_SMA'] = df['close'].rolling(window=short_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int) + (df['Short_Term_SMA'] > df['Long_Term_SMA']).astype(int)
    df['Trend_Indicator'] = df['Trend_Indicator'].rolling(window=3).mean()  # Smoothing the trend indicator

    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Liquidity_Rolling_Avg'] = df['Daily_Turnover'].rolling(window=liquidity_lookback).mean()
    df['Liquidity_Weight'] = (df['Liquidity_Rolling_Avg'] / df['Liquidity_Rolling_Avg'].max()).clip(lower=0.1, upper=1)

    # Final Alpha Factor
    weights = {
        'Momentum': 0.3,
        'Volume_Adjusted_Volatility': -0.2,
        'Close_Percent_Change': 0.2,
        'Open_Close_Diff_Rolling_Avg': -0.1,
        'Trend_Indicator': 0.2,
        'Liquidity_Weight': 0.1
    }

    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Volatility'] * df['Volume_Adjusted_Volatility'] +
        weights['Close_Percent_Change'] * df['Close_Percent_Change'] +
        weights['Open_Close_Diff_Rolling_Avg'] * df['Open_Close_Diff_Rolling_Avg'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Liquidity_Weight'] * df['Liquidity_Weight']
    )

    return df['Alpha_Factor']
```
