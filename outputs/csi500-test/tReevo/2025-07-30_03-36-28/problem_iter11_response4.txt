```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    sma_short_period = 20
    sma_long_period = 100
    vol_adj_volatility_period = 20
    percentage_change_period = 5
    turnover_rolling_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_adj_volatility_period).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=sma_short_period).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Percentage_Change'] = df['close'].pct_change(periods=percentage_change_period)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Market_Trend'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)  # Bullish: 1, Bearish: 0
    
    # Incorporate Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Turnover_Rolling_Avg'] = df['Daily_Turnover'].rolling(window=turnover_rolling_period).mean()
    
    # Define weights for the components
    weight_price_momentum = 0.4
    weight_vol_adj_volatility = -0.2
    weight_close_percentage_change = 0.2
    weight_high_low_range = 0.1
    weight_turnover = 0.1
    
    # Adjust weights based on market trend and liquidity
    df['Weight_Adj_Price_Momentum'] = weight_price_momentum * (1 + df['Market_Trend'] * 0.2)
    df['Weight_Adj_Vol_Adj_Volatility'] = weight_vol_adj_volatility * (1 - df['Market_Trend'] * 0.2)
    df['Weight_Adj_Close_Percentage_Change'] = weight_close_percentage_change * (1 + df['Market_Trend'] * 0.2)
    df['Weight_Adj_High_Low_Range'] = weight_high_low_range * (1 + df['Market_Trend'] * 0.2)
    df['Weight_Adj_Turnover'] = weight_turnover * (1 + (df['Turnover_Rolling_Avg'] / df['Daily_Turnover'].median()))
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Weight_Adj_Price_Momentum'] * df['Price_Momentum'] +
        df['Weight_Adj_Vol_Adj_Volatility'] * df['Volume_Adjusted_Volatility'] +
        df['Weight_Adj_Close_Percentage_Change'] * df['Close_Percentage_Change'] +
        df['Weight_Adj_High_Low_Range'] * df['High_Low_Range'] +
        df['Weight_Adj_Turnover'] * df['Turnover_Rolling_Avg']
    )
    
    return df['Alpha_Factor']
```
