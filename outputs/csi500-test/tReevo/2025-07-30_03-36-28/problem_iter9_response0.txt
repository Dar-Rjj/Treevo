```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    sma_short_window = 20
    sma_long_window = 60
    vol_adjusted_volatility_window = 20
    price_change_lookback = 5
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_window).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_adjusted_volatility_window).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=sma_short_window).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Close_Price_Change'] = df['close'].pct_change(periods=price_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Define the weights for each component
    weight_momentum = 0.4
    weight_volatility = 0.3
    weight_price_change = 0.2
    weight_high_low_range = 0.1
    
    # Adjust weights based on market trend
    df['Weight_Momentum'] = weight_momentum + (df['Trend_Indicator'] * 0.1)
    df['Weight_Volatility'] = weight_volatility - (df['Trend_Indicator'] * 0.1)
    df['Weight_Price_Change'] = weight_price_change
    df['Weight_High_Low_Range'] = weight_high_low_range
    
    # Ensure the weights sum to 1
    df['Total_Weight'] = df['Weight_Momentum'] + df['Weight_Volatility'] + df['Weight_Price_Change'] + df['Weight_High_Low_Range']
    df['Weight_Momentum'] = df['Weight_Momentum'] / df['Total_Weight']
    df['Weight_Volatility'] = df['Weight_Volatility'] / df['Total_Weight']
    df['Weight_Price_Change'] = df['Weight_Price_Change'] / df['Total_Weight']
    df['Weight_High_Low_Range'] = df['Weight_High_Low_Range'] / df['Total_Weight']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Weight_Momentum'] * df['Price_Momentum'] +
        df['Weight_Volatility'] * df['Volume_Adjusted_Volatility'] +
        df['Weight_Price_Change'] * df['Close_Price_Change'] +
        df['Weight_High_Low_Range'] * df['High_Low_Range']
    )
    
    return df['Alpha_Factor']
```
