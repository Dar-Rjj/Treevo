import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define the initial lookback period for SMA
    initial_lookback = 20
    
    # Calculate the high-low price difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate the rolling average of the high-low differences to evaluate recent market volatility
    rolling_volatility = high_low_diff.rolling(window=initial_lookback).mean()
    
    # Adjust the lookback period based on recent market volatility
    def adjust_lookback_period(volatility, threshold=0.5):
        return np.where(volatility > threshold, 10, 20)
    
    adaptive_lookback = adjust_lookback_period(rolling_volatility)
    
    # Calculate the adaptive lookback simple moving average (SMA) of close prices
    def adaptive_sma(close, lookback):
        sma_values = []
        for i in range(len(close)):
            if i < initial_lookback - 1:
                sma_values.append(np.nan)
            else:
                sma_values.append(np.mean(close[i-lookback[i]+1:i+1]))
        return pd.Series(sma_values, index=close.index)
    
    close_sma = adaptive_sma(df['close'], adaptive_lookback)
    
    # Compute volume-adjusted volatility
    volume_adjusted_volatility = (df['volume'] * high_low_diff).rolling(window=adaptive_lookback).mean()
    
    # Compute price momentum
    price_momentum = (df['close'] - close_sma) / df['close'].rolling(window=initial_lookback).mean()
    
    # Incorporate additional price change metrics
    percentage_change_close = df['close'].pct_change(periods=initial_lookback)
    high_low_range = df['high'] - df['low']
    
    # Combine components into the final alpha factor using a weighted sum
    weights = {
        'price_momentum': 0.5,
        'volume_adjusted_volatility': -0.3,
        'percentage_change_close': 0.2
    }
    
    alpha_factor = (
        weights['price_momentum'] * price_momentum +
        weights['volume_adjusted_volatility'] * volume_adjusted_volatility +
        weights['percentage_change_close'] * percentage_change_close
    )
    
    return alpha_factor
