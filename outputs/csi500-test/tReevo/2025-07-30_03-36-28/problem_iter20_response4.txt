```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    short_lookback = 10
    medium_lookback = 50
    long_lookback = 200
    vol_lookback = 20
    turnover_lookback = 30
    atr_lookback = 14

    # Calculate Simple Moving Averages (SMA)
    df['SMA_short'] = df['close'].rolling(window=short_lookback).mean()
    df['SMA_medium'] = df['close'].rolling(window=medium_lookback).mean()
    df['SMA_long'] = df['close'].rolling(window=long_lookback).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_high_low'] = df['high_low_diff'] * df['volume']
    df['vol_adjusted_volatility'] = df['vol_weighted_high_low'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=short_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['pct_change_close'] = df['close'].pct_change(short_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = 'Neutral'
    df.loc[df['SMA_short'] > df['SMA_medium'], 'trend_indicator'] = 'Bullish'
    df.loc[df['SMA_short'] < df['SMA_medium'], 'trend_indicator'] = 'Bearish'

    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    df['liquidity_weight'] = (df['avg_daily_turnover'] / df['avg_daily_turnover'].max()).fillna(0)

    # Cross-Verify with Additional Volatility Measures
    df['std_dev_close'] = df['close'].rolling(window=short_lookback).std()
    df['true_range'] = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    df['atr'] = df['true_range'].rolling(window=atr_lookback).mean()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'vol_adjusted_volatility': -0.2,
        'pct_change_close': 0.2,
        'high_low_range': 0.1,
        'liquidity_weight': 0.1,
        'std_dev_close': -0.05,
        'atr': -0.05
    }

    df['alpha_factor'] = (
        df['price_momentum'] * weights['price_momentum'] +
        df['vol_adjusted_volatility'] * weights['vol_adjusted_volatility'] +
        df['pct_change_close'] * weights['pct_change_close'] +
        df['high_low_range'] * weights['high_low_range'] +
        df['liquidity_weight'] * weights['liquidity_weight'] +
        df['std_dev_close'] * weights['std_dev_close'] +
        df['atr'] * weights['atr']
    )

    # Adjust Weights Dynamically based on Market Trend
    trend_weights = {
        'Bullish': 1.1,
        'Neutral': 1.0,
        'Bearish': 0.9
    }
    df['alpha_factor'] *= df['trend_indicator'].map(trend_weights)

    return df['alpha_factor']
```
