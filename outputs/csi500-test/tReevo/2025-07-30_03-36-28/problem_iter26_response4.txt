```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_lookback=20, vol_adj_vol_lookback=20, momentum_n=20, pct_change_lookback=10, long_term_sma_lookback=50, roc_lookback=10, turnover_roll_avg_lookback=20, turnover_roll_std_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Weighted_High_Low_Diff'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adjusted_Volatility'] = df['Volume_Weighted_High_Low_Diff'].rolling(window=vol_adj_vol_lookback).std()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Enhanced Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=long_term_sma_lookback).mean()
    df['Trend_Indicator'] = np.where(df['SMA'] > df['Long_Term_SMA'], 1, -1)  # 1 for Bullish, -1 for Bearish
    df['ROC_Long_Term_SMA'] = df['Long_Term_SMA'].pct_change(periods=roc_lookback)
    df['Trend_Indicator_Adjusted'] = np.where(df['ROC_Long_Term_SMA'] > 0, df['Trend_Indicator'], -df['Trend_Indicator'])
    
    # Incorporate Optimized Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_roll_avg_lookback).mean()
    df['Rolling_Std_Turnover'] = df['Daily_Turnover'].rolling(window=turnover_roll_std_lookback).std()
    df['Liquidity_Adjusted_Weight'] = (df['Rolling_Avg_Turnover'] / df['Rolling_Std_Turnover']).rank(pct=True)
    
    # Final Alpha Factor
    weights = {
        'Price_Momentum': 0.4,
        'Vol_Adjusted_Volatility': -0.2,
        'Pct_Change_Close': 0.1,
        'Trend_Indicator_Adjusted': 0.2,
        'Liquidity_Adjusted_Weight': 0.3
    }
    
    df['Alpha_Factor'] = (weights['Price_Momentum'] * df['Price_Momentum'] +
                          weights['Vol_Adjusted_Volatility'] * df['Vol_Adjusted_Volatility'] +
                          weights['Pct_Change_Close'] * df['Pct_Change_Close'] +
                          weights['Trend_Indicator_Adjusted'] * df['Trend_Indicator_Adjusted'] +
                          weights['Liquidity_Adjusted_Weight'] * df['Liquidity_Adjusted_Weight'])
    
    return df['Alpha_Factor']
```
