```python
import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=200, volatility_lookback=60, momentum_lookback=10, 
                  percentage_change_lookback=5, daily_turnover_lookback=30, avg_dollar_volume_lookback=30, bid_ask_spread_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_adj_volatility'] = df['high_low_diff'] * df['volume']
    df['vol_adj_volatility'] = df['vol_adj_volatility'].rolling(window=volatility_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_lookback).mean()

    # Incorporate Additional Price Change Metrics
    df['percentage_change'] = df['close'].pct_change(percentage_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = (df['SMA_short'] > df['SMA_long']).astype(int)

    # Incorporate Advanced Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=daily_turnover_lookback).mean()
    
    df['avg_dollar_volume'] = df['volume'] * df['close']
    df['avg_dollar_volume'] = df['avg_dollar_volume'].rolling(window=avg_dollar_volume_lookback).mean()
    
    df['bid_ask_spread'] = df['open'] - df['close']
    df['avg_bid_ask_spread'] = df['bid_ask_spread'].rolling(window=bid_ask_spread_lookback).mean()

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'vol_adj_volatility': -0.2,
        'percentage_change': 0.2,
        'high_low_range': 0.1,
        'trend_indicator': 0.1,
        'avg_daily_turnover': 0.05,
        'avg_dollar_volume': 0.05,
        'avg_bid_ask_spread': -0.05
    }
    
    df['alpha_factor'] = (weights['price_momentum'] * df['price_momentum'] +
                          weights['vol_adj_volatility'] * df['vol_adj_volatility'] +
                          weights['percentage_change'] * df['percentage_change'] +
                          weights['high_low_range'] * df['high_low_range'] +
                          weights['trend_indicator'] * df['trend_indicator'] +
                          weights['avg_daily_turnover'] * df['avg_daily_turnover'] / df['avg_daily_turnover'].max() +
                          weights['avg_dollar_volume'] * df['avg_dollar_volume'] / df['avg_dollar_volume'].max() -
                          weights['avg_bid_ask_spread'] * df['avg_bid_ask_spread'] / df['avg_bid_ask_spread'].max())
    
    return df['alpha_factor']
```
