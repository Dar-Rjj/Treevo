```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily log return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute high-low range
    df['range'] = df['high'] - df['low']
    
    # Combine to form intraday momentum
    df['intraday_momentum'] = df['log_return'] * df['range']
    
    # Compute volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Determine volume intensity ratio
    df['volume_intensity_ratio'] = df['volume_change'] / ((df['volume'] + df['volume'].shift(1)) / 2)
    
    # Weight intraday momentum by volume intensity ratio
    df['weighted_intraday_momentum'] = df['intraday_momentum'] * df['volume_intensity_ratio']
    
    # Calculate price volatility
    df['log_returns_20'] = np.log(df['close'] / df['close'].shift(1)).rolling(window=20).std()
    
    # Normalize intraday momentum by price volatility
    df['alpha_factor'] = df['weighted_intraday_momentum'] / df['log_returns_20']
    
    return df['alpha_factor'].dropna()
```
