import pandas as pd

def heuristics_v2(df, short_sma_window=20, long_sma_window=60, vol_adj_lookback=20, momentum_lookback=10, pct_change_lookback=5, turnover_lookback=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_window).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_adj_volatility'] = df['high_low_diff'] * df['volume']
    df['vol_adj_volatility'] = df['vol_adj_volatility'].rolling(window=vol_adj_lookback).mean()
    
    # Compute Price Momentum
    avg_close = df['close'].rolling(window=momentum_lookback).mean()
    df['price_momentum'] = (df['close'] - df['SMA']) / avg_close
    
    # Incorporate Additional Price Change Metrics
    df['pct_change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['long_term_SMA'] = df['close'].rolling(window=long_sma_window).mean()
    df['trend_indicator'] = (df['SMA'] > df['long_term_SMA']).astype(int)  # 1 if bullish, 0 if bearish
    
    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover_ma'] = df['daily_turnover'].rolling(window=turnover_lookback).mean()
    
    # Final Alpha Factor
    # Define the Weights for Each Component
    w_price_momentum = 0.3
    w_vol_adj_volatility = -0.2
    w_pct_change = 0.2
    w_high_low_range = 0.1
    w_turnover = 0.1
    w_trend = 0.1
    
    # Adjust Weights Dynamically Based on Trend
    df['alpha_factor'] = (
        (w_price_momentum + df['trend_indicator'] * 0.1) * df['price_momentum'] +
        (w_vol_adj_volatility - df['trend_indicator'] * 0.1) * df['vol_adj_volatility'] +
        (w_pct_change + df['trend_indicator'] * 0.1) * df['pct_change'] +
        (w_high_low_range + df['trend_indicator'] * 0.1) * df['high_low_range'] +
        (w_turnover + df['trend_indicator'] * 0.1) * df['turnover_ma'] +
        w_trend * df['trend_indicator']
    )
    
    return df['alpha_factor']
