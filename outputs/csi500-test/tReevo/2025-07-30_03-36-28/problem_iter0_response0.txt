def heuristics_v2(df):
    """
    Calculate a volume-weighted close-to-open return.
    
    :param df: pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    :return: pandas Series with factor values
    """
    # Calculate close-to-open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume'].shift(1)
    
    return volume_weighted_return
