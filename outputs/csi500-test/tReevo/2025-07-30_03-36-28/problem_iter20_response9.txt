```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    short_sma_period = 20
    long_sma_period = 50
    vol_lookback = 20
    momentum_n = 10
    trend_lookback = 50
    liquidity_lookback = 20
    vwap_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_vol'] = (df['high_low_diff'] * df['volume']).rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['close_pct_change'] = df['close'].pct_change()
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['trend_indicator'] = (df['SMA_short'] > df['SMA_long']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['avg_daily_turnover'] = df['daily_turnover'].rolling(window=liquidity_lookback).mean()
    df['avg_dollar_volume'] = (df['volume'] * df['close']).rolling(window=liquidity_lookback).mean()
    df['vwap'] = (df['volume'] * df['close']).rolling(window=vwap_lookback).sum() / df['volume'].rolling(window=vwap_lookback).sum()
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'volume_weighted_vol': -0.2,
        'close_pct_change': 0.1,
        'high_low_range': 0.1,
        'avg_daily_turnover': 0.1,
        'avg_dollar_volume': 0.1,
        'vwap': 0.1
    }
    
    # Adjust Weights Dynamically
    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['volume_weighted_vol'] * df['volume_weighted_vol'] +
        weights['close_pct_change'] * df['close_pct_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['avg_daily_turnover'] * df['avg_daily_turnover'] +
        weights['avg_dollar_volume'] * df['avg_dollar_volume'] +
        weights['vwap'] * df['vwap']
    )
    
    # Adjust for market trend
    df['alpha_factor'] = df['alpha_factor'] * (1 + df['trend_indicator'] * 0.1)
    
    return df['alpha_factor'].dropna()
```
