import pandas as pd

def heuristics_v2(df, short_lookback=20, long_lookback=200, volatility_lookback=30, trend_weight=0.5):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_lookback).mean()
    df['SMA_long'] = df['close'].rolling(window=long_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_adjusted_volatility'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_adjusted_volatility'].rolling(window=volatility_lookback).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=short_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['percentage_change'] = df['close'].pct_change(periods=short_lookback)
    df['high_low_range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['trend_indicator'] = (df['SMA_short'] > df['SMA_long']).astype(int)
    
    # Final Alpha Factor
    weights = {
        'price_momentum': 0.3,
        'volatility': 0.2,
        'percentage_change': 0.2,
        'high_low_range': 0.1,
        'trend_indicator': trend_weight
    }
    
    df['alpha_factor'] = (
        df['price_momentum'] * weights['price_momentum'] +
        df['volatility'] * weights['volatility'] +
        df['percentage_change'] * weights['percentage_change'] +
        df['high_low_range'] * weights['high_low_range'] +
        df['trend_indicator'] * weights['trend_indicator']
    )
    
    return df['alpha_factor']
