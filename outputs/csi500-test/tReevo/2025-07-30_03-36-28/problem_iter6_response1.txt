import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define the base lookback period for calculating rolling standard deviation
    base_lookback = 20
    
    # Calculate the rolling standard deviation of close prices
    rolling_std = df['close'].rolling(window=base_lookback).std()
    
    # Determine the dynamic lookback period based on market volatility
    def adjust_lookback(rolling_std):
        if rolling_std > rolling_std.median():
            return 10  # Shorter lookback in high volatility
        else:
            return 30  # Longer lookback in low volatility
    
    # Apply the adjustment function to create a dynamic lookback period
    dynamic_lookback = rolling_std.apply(adjust_lookback)
    
    # Calculate Simple Moving Average (SMA) of Close Prices using the dynamic lookback period
    sma = df['close'].rolling(window=dynamic_lookback, min_periods=1).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_diff = high_low_diff * df['volume']
    vol_adjusted_volatility = volume_weighted_diff.rolling(window=dynamic_lookback, min_periods=1).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=dynamic_lookback, min_periods=1).sum()
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_adjusted_volatility
    
    return alpha_factor
