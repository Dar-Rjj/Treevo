```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Short-Term Momentum (5-day sum of log returns)
    df['short_term_momentum'] = df['log_return'].rolling(window=5).sum()
    
    # Calculate Long-Term Momentum (20-day sum of log returns)
    df['long_term_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Subtract Short-Term Momentum from Long-Term Momentum
    df['momentum_diff'] = df['long_term_momentum'] - df['short_term_momentum']
    
    # Calculate Average Volume over the past 20 days
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Multiply Long-Term - Short-Term Momentum by Average Volume
    df['weighted_momentum'] = df['momentum_diff'] * df['avg_volume']
    
    # Calculate Price Volatility (20-day standard deviation of log returns)
    df['price_volatility'] = df['log_return'].rolling(window=20).std()
    
    # Divide the Weighted Momentum by the Price Volatility
    df['alpha_factor'] = df['weighted_momentum'] / df['price_volatility']
    
    return df['alpha_factor']
```
