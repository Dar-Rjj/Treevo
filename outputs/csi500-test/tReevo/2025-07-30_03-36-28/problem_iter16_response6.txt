import pandas as pd

def heuristics_v2(df, short_sma_window=20, long_sma_window=50, vol_avg_window=20, price_mom_window=20, pct_change_window=10, liquidity_window=20):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_window).mean()
    df['Long_SMA'] = df['close'].rolling(window=long_sma_window).mean()

    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Vol_Adjusted_Volatility'] = (df['volume'] * df['High_Low_Diff']).rolling(window=vol_avg_window).mean()

    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=price_mom_window).mean()

    # Incorporate Additional Price Change Metrics
    df['Pct_Change_Close'] = df['close'].pct_change(periods=pct_change_window)
    df['High_Low_Range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA'] > df['Long_SMA']).astype(int)

    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Liquidity'] = df['Daily_Turnover'].rolling(window=liquidity_window).mean()
    df['Liquidity_Weight'] = (df['Liquidity'] / df['Liquidity'].rolling(window=liquidity_window).max()).fillna(1)

    # Final Alpha Factor
    # Define the Weights for Each Component
    price_mom_weight = 0.3
    vol_adj_vol_weight = 0.2
    pct_change_weight = 0.2
    high_low_range_weight = 0.1
    trend_weight = 0.1
    liquidity_weight = 0.1

    # Combine Components
    df['Alpha_Factor'] = (
        price_mom_weight * df['Price_Momentum'] +
        vol_adj_vol_weight * df['Vol_Adjusted_Volatility'] +
        pct_change_weight * df['Pct_Change_Close'] +
        high_low_range_weight * df['High_Low_Range'] +
        trend_weight * df['Trend_Indicator'] +
        liquidity_weight * df['Liquidity_Weight']
    )

    # Dynamically Adjust Weights Based on Recent Performance
    # Evaluate the Recent Performance of Each Component
    recent_performance_window = 10
    recent_performance = df[['Price_Momentum', 'Vol_Adjusted_Volatility', 'Pct_Change_Close', 'High_Low_Range', 'Trend_Indicator', 'Liquidity_Weight']].rolling(window=recent_performance_window).mean().dropna()

    # Normalize the recent performance to sum to 1
    recent_performance_sum = recent_performance.sum(axis=1)
    recent_performance = recent_performance.divide(recent_performance_sum, axis=0)

    # Adjust Weights to Favor Components with Stronger Recent Performance
    adjusted_weights = recent_performance.iloc[-1].values
    df['Alpha_Factor_Adjusted'] = (
        adjusted_weights[0] * df['Price_Momentum'] +
        adjusted_weights[1] * df['Vol_Adjusted_Volatility'] +
        adjusted_weights[2] * df['Pct_Change_Close'] +
        adjusted_weights[3] * df['High_Low_Range'] +
        adjusted_weights[4] * df['Trend_Indicator'] +
        adjusted_weights[5] * df['Liquidity_Weight']
    )

    return df['Alpha_Factor_Adjusted'].dropna()
