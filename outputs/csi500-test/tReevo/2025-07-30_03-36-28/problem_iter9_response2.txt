```python
import pandas as pd

def heuristics_v2(df, short_sma_period=20, long_sma_period=200, vol_lookback=20, price_change_lookback=10):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=short_sma_period).mean()
    df['LTSMA'] = df['close'].rolling(window=long_sma_period).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['vol_weighted_diff'] = df['high_low_diff'] * df['volume']
    df['vol_adjusted_volatility'] = df['vol_weighted_diff'].rolling(window=vol_lookback).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=short_sma_period).mean()

    # Incorporate Additional Price Change Metrics
    df['close_pct_change'] = df['close'].pct_change(periods=price_change_lookback)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend_indicator'] = df['SMA'] > df['LTSMA']

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'vol_adjusted_volatility': -0.3,
        'close_pct_change': 0.2,
        'high_low_range': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['vol_adjusted_volatility'] * df['vol_adjusted_volatility'] +
        weights['close_pct_change'] * df['close_pct_change'] +
        weights['high_low_range'] * df['high_low_range']
    )

    # Adjust Weights Based on Market Trend
    df.loc[df['trend_indicator'], 'alpha_factor'] *= 1.2
    df.loc[~df['trend_indicator'], 'alpha_factor'] *= 0.8

    return df['alpha_factor'].dropna()
```
