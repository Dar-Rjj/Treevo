```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Simple Moving Average of Close Prices
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Apply Volume Weighting to High-Low Price Difference
    volume_weighted_diff = high_low_diff * df['volume']
    
    # Compute Rolling Sum of Volume-Weighted High-Low Differences
    rolling_sum_volume_weighted_diff = volume_weighted_diff.rolling(window=lookback_period).sum()
    
    # Compute Price Momentum
    price_momentum = df['close'] - sma_close
    
    # Final Alpha Factor: Divide the Price Momentum by the Rolling Sum of Volume-Weighted High-Low Differences
    alpha_factor = price_momentum / rolling_sum_volume_weighted_diff
    
    return alpha_factor
```
