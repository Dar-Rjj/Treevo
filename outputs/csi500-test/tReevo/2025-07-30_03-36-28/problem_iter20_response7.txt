```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_lookback = 50
    vol_adj_volatility_lookback = 21
    momentum_n = 20
    pct_change_lookback = 5
    long_term_sma_lookback = 200
    daily_turnover_lookback = 20
    avg_dollar_volume_lookback = 20
    bid_ask_spread_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA'] = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Vol_Adj_Vol_Rolling_Avg'] = df['Volume_Adjusted_Volatility'].rolling(window=vol_adj_volatility_lookback).mean()
    
    # Compute Price Momentum
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Long_Term_SMA'] = df['close'].rolling(window=long_term_sma_lookback).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_Term_SMA']).astype(int)
    
    # Incorporate Dynamic Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Avg_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=daily_turnover_lookback).mean()
    df['Avg_Dollar_Volume'] = (df['volume'] * df['close']).rolling(window=avg_dollar_volume_lookback).mean()
    df['Bid_Ask_Spread'] = df['open'] - df['close']
    df['Rolling_Avg_Bid_Ask_Spread'] = df['Bid_Ask_Spread'].rolling(window=bid_ask_spread_lookback).mean()
    
    # Adjust Weights Based on Liquidity, Average Dollar Volume, and Bid-Ask Spread
    liquidity_weight = df['Rolling_Avg_Daily_Turnover'].rank(pct=True)
    avg_dollar_volume_weight = df['Avg_Dollar_Volume'].rank(pct=True)
    bid_ask_spread_weight = (1 - df['Rolling_Avg_Bid_Ask_Spread'].rank(pct=True))
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.3,
        'Vol_Adj_Vol_Rolling_Avg': 0.2,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Indicator': 0.1,
        'Liquidity_Weight': 0.1,
        'Avg_Dollar_Volume_Weight': 0.1,
        'Bid_Ask_Spread_Weight': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Vol_Adj_Vol_Rolling_Avg'] * df['Vol_Adj_Vol_Rolling_Avg'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] +
        weights['Trend_Indicator'] * df['Trend_Indicator'] +
        weights['Liquidity_Weight'] * liquidity_weight +
        weights['Avg_Dollar_Volume_Weight'] * avg_dollar_volume_weight +
        weights['Bid_Ask_Spread_Weight'] * bid_ask_spread_weight
    )
    
    return df['Alpha_Factor']
```
