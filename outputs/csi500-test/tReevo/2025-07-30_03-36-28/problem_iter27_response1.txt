```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_lookback=10, momentum_n=5, pct_change_lookback=10, long_sma_lookback=200, turnover_lookback=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_diff = high_low_diff * df['volume']
    vol_adj_volatility = volume_weighted_diff.rolling(window=vol_lookback).mean()
    
    # Compute Price Momentum
    avg_close = df['close'].rolling(window=momentum_n).mean()
    price_momentum = (df['close'] - sma) / avg_close
    
    # Incorporate Additional Price Change Metrics
    pct_change = df['close'].pct_change(periods=pct_change_lookback)
    high_low_range = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    long_sma = df['close'].rolling(window=long_sma_lookback).mean()
    trend_indicator = (sma > long_sma).astype(int)  # 1 for bullish, 0 for bearish
    
    # Incorporate Dynamic Liquidity Measures
    daily_turnover = df['volume'] * df['close']
    avg_turnover = daily_turnover.rolling(window=turnover_lookback).mean()
    liquidity_weight = (avg_turnover / avg_turnover.max()).clip(0.1, 1)  # Ensure weights are between 0.1 and 1
    
    # Final Alpha Factor
    alpha_factor = (
        0.4 * price_momentum + 
        0.3 * vol_adj_volatility + 
        0.1 * pct_change + 
        0.1 * high_low_range + 
        0.1 * trend_indicator
    )
    
    # Adjust Weights Dynamically Based on Market Trend
    alpha_factor = alpha_factor * (1 + 0.1 * trend_indicator)  # Increase weight for bullish trends
    
    # Apply Liquidity Weight
    alpha_factor = alpha_factor * liquidity_weight
    
    return alpha_factor
```
