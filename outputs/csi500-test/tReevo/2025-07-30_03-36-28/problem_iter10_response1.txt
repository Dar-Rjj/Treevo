```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_short_lookback = 20
    volatility_lookback = 10
    average_lookback = 20
    pct_change_lookback = 5
    sma_long_lookback = 50
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volatility_Rolling_Avg'] = df['Volume_Adjusted_Volatility'].rolling(window=volatility_lookback).mean()
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=average_lookback).mean()
    
    # Incorporate Robust Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Diff'] = df['open'] - df['close']
    
    # Consider Market Trend Alignment
    df['SMA_Long'] = df['close'].rolling(window=sma_long_lookback).mean()
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Define weights for the components
    base_weights = {
        'Price_Momentum': 0.4,
        'Volatility_Rolling_Avg': 0.3,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Open_Close_Diff': 0.1
    }
    
    # Adjust weights based on market trend
    for component, weight in base_weights.items():
        df[f'{component}_Weight'] = weight * (1 + 0.1 * (df['Trend_Indicator'] * 2 - 1))
    
    # Final Alpha Factor
    alpha_factor = (
        df['Price_Momentum'] * df['Price_Momentum_Weight'] +
        df['Volatility_Rolling_Avg'] * df['Volatility_Rolling_Avg_Weight'] +
        df['Pct_Change'] * df['Pct_Change_Weight'] +
        df['High_Low_Range'] * df['High_Low_Range_Weight'] +
        df['Open_Close_Diff'] * df['Open_Close_Diff_Weight']
    )
    
    return alpha_factor
```
