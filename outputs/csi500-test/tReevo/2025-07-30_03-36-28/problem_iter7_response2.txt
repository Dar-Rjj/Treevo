```python
import pandas as pd

def heuristics_v2(df, sma_lookback=20, vol_adj_lookback=10, momentum_n=5, pct_change_lookback=10, high_low_std_lookback=10, weights=(0.5, 0.3, 0.2)):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma = df['close'].rolling(window=sma_lookback).mean()
    
    # Compute Volume-Adjusted Volatility
    high_low_diff = df['high'] - df['low']
    volume_weighted_high_low = high_low_diff * df['volume']
    vol_adj_volatility = volume_weighted_high_low.rolling(window=vol_adj_lookback).mean()
    
    # Compute Price Momentum
    price_momentum = (df['close'] - sma) / df['close'].rolling(window=momentum_n).mean()
    
    # Incorporate Additional Price Change Metrics
    pct_change = df['close'].pct_change(periods=pct_change_lookback)
    high_low_range = df['high'] - df['low']
    high_low_std = high_low_range.rolling(window=high_low_std_lookback).std()
    
    # Final Alpha Factor
    alpha_factor = (weights[0] * price_momentum + 
                    weights[1] * vol_adj_volatility + 
                    weights[2] * pct_change / high_low_std)
    
    return alpha_factor
```
