```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    lookback_sma = 20
    df['SMA'] = df['close'].rolling(window=lookback_sma).mean()
    
    # Compute Volume-Adjusted Volatility
    lookback_volatility = 20
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Volume_Adjusted_Volatility'] = df['Volume_Adjusted_Volatility'].rolling(window=lookback_volatility).mean()
    
    # Compute Price Momentum
    lookback_momentum = 10
    df['Momentum'] = (df['close'] - df['SMA']) / df['close'].rolling(window=lookback_momentum).mean()
    
    # Incorporate Additional Price Change Metrics
    lookback_percentage_change = 5
    df['Percentage_Change'] = df['close'].pct_change(periods=lookback_percentage_change)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    long_lookback_sma = 60
    df['Long_SMA'] = df['close'].rolling(window=long_lookback_sma).mean()
    df['Trend_Indicator'] = (df['SMA'] > df['Long_SMA']).astype(int)
    
    # Incorporate Liquidity Measures
    lookback_turnover = 20
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Average_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=lookback_turnover).mean()
    
    # Refine Volatility Adjustments
    lookback_historical_volatility = 20
    df['Historical_Volatility'] = df['close'].rolling(window=lookback_historical_volatility).std()
    df['Adjusted_Volume_Adjusted_Volatility'] = df['Volume_Adjusted_Volatility'] * (1 + df['Historical_Volatility'])
    
    # Dynamic Weight Adjustments Based on Market Conditions and Trends
    bullish_weight = 0.7
    bearish_weight = 0.3
    high_liquidity_weight = 0.8
    low_liquidity_weight = 0.2
    df['Weighted_Momentum'] = df['Momentum'] * (bullish_weight * df['Trend_Indicator'] + bearish_weight * (1 - df['Trend_Indicator']))
    df['Weighted_Volatility'] = df['Adjusted_Volume_Adjusted_Volatility'] * (high_liquidity_weight * (df['Average_Daily_Turnover'] > df['Average_Daily_Turnover'].mean()) + low_liquidity_weight * (df['Average_Daily_Turnover'] <= df['Average_Daily_Turnover'].mean()))
    df['Weighted_Percentage_Change'] = df['Percentage_Change'] * (bullish_weight * df['Trend_Indicator'] + bearish_weight * (1 - df['Trend_Indicator']))
    df['Weighted_High_Low_Range'] = df['High_Low_Range'] * (bullish_weight * df['Trend_Indicator'] + bearish_weight * (1 - df['Trend_Indicator']))
    
    # Final Alpha Factor
    weights = {
        'Weighted_Momentum': 0.4,
        'Weighted_Volatility': 0.3,
        'Weighted_Percentage_Change': 0.2,
        'Weighted_High_Low_Range': 0.1
    }
    df['Alpha_Factor'] = (df['Weighted_Momentum'] * weights['Weighted_Momentum'] +
                          df['Weighted_Volatility'] * weights['Weighted_Volatility'] +
                          df['Weighted_Percentage_Change'] * weights['Weighted_Percentage_Change'] +
                          df['Weighted_High_Low_Range'] * weights['Weighted_High_Low_Range'])
    
    return df['Alpha_Factor'].dropna()
```
