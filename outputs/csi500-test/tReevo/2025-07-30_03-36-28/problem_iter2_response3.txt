import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    df['Close_t-50'] = df['close'].shift(50)
    df['LongTermMomentum'] = df['close'] - df['Close_t-50']
    
    # Measure Short-Term Price Momentum
    df['Close_t-10'] = df['close'].shift(10)
    df['ShortTermMomentum'] = df['close'] - df['Close_t-10']
    
    # Compute Exponential Moving Average (EMA) of Volume
    df['Volume_EMA_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Determine Volume Trend
    df['VolumeTrend'] = df['volume'] - df['Volume_EMA_20']
    
    # Combine Long-Term and Short-Term Price Momentum
    df['CombinedMomentum'] = df['LongTermMomentum'] + df['ShortTermMomentum']
    
    # Incorporate Volume Influence
    df['PriceMomentumWithVolumeInfluence'] = df['CombinedMomentum'] * df['VolumeTrend']
    
    # Return the factor values
    return df['PriceMomentumWithVolumeInfluence']
