```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate True Range
    df['PrevClose'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['PrevClose'])
    df['TR3'] = abs(df['low'] - df['PrevClose'])
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Sum N-period True Ranges
    df['SumNTrueRange'] = df['TrueRange'].rolling(window=n).sum()

    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']

    # Average N-period High-Low Range
    df['AvgNHighLowRange'] = df['HighLowRange'].rolling(window=n).mean()

    # Construct Volatility Difference
    df['VolatilityDifference'] = df['AvgNHighLowRange'] - df['SumNTrueRange']

    # Smooth Volatility Difference
    df['SmoothedVolDiff'] = df['VolatilityDifference'].rolling(window=n).mean()

    # Calculate Price Momentum
    df['5DayReturn'] = df['close'].pct_change(periods=5)
    df['20DayReturn'] = df['close'].pct_change(periods=20)

    # Incorporate Volume Momentum
    df['5DayAvgVolume'] = df['volume'].rolling(window=5).mean()
    df['20DayAvgVolume'] = df['volume'].rolling(window=20).mean()

    # Combine Momentum Indicators
    df['5DayMom'] = df['5DayReturn'] * df['5DayAvgVolume']
    df['20DayMom'] = df['20DayReturn'] * df['20DayAvgVolume']
    df['CombinedMomentum'] = df['5DayMom'] - df['20DayMom']

    # Final Alpha Factor Construction
    df['FinalAlphaFactor'] = df['SmoothedVolDiff'] + df['CombinedMomentum']

    # Apply Weighted Moving Average to the final alpha factor
    df['WeightedMA'] = df['FinalAlphaFactor'].ewm(span=n, adjust=False).mean()

    return df['WeightedMA']
```
