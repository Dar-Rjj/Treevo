```python
import pandas as pd

def heuristics_v2(df):
    # Define lookback periods
    sma_short_period = 20
    sma_long_period = 50
    vol_adj_volatility_period = 10
    price_change_period = 5
    avg_close_period = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = (df['High_Low_Diff'] * df['volume']).rolling(window=vol_adj_volatility_period).mean()
    
    # Compute Price Momentum
    df['Average_Close'] = df['close'].rolling(window=avg_close_period).mean()
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['Average_Close']
    
    # Incorporate Additional Price Change Metrics
    df['Percentage_Change'] = df['close'].pct_change(periods=price_change_period)
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Diff'] = df['open'] - df['close']
    df['Absolute_Change'] = df['close'].diff(periods=price_change_period).abs()
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = 0
    df.loc[df['SMA_Short'] > df['SMA_Long'], 'Trend_Indicator'] = 1  # Bullish
    df.loc[df['SMA_Short'] < df['SMA_Long'], 'Trend_Indicator'] = -1  # Bearish
    
    # Define initial weights for each component
    weights_bullish = {'Price_Momentum': 0.5, 'Volume_Adjusted_Volatility': 0.25, 
                       'Percentage_Change': 0.15, 'High_Low_Range': 0.05, 'Open_Close_Diff': 0.05}
    weights_bearish = {'Price_Momentum': 0.3, 'Volume_Adjusted_Volatility': 0.4, 
                       'Percentage_Change': 0.1, 'High_Low_Range': 0.1, 'Open_Close_Diff': 0.1}
    
    # Final Alpha Factor
    def compute_alpha_factor(row):
        if row['Trend_Indicator'] == 1:
            weights = weights_bullish
        else:
            weights = weights_bearish
        
        alpha_factor = (
            weights['Price_Momentum'] * row['Price_Momentum'] +
            weights['Volume_Adjusted_Volatility'] * row['Volume_Adjusted_Volatility'] +
            weights['Percentage_Change'] * row['Percentage_Change'] +
            weights['High_Low_Range'] * row['High_Low_Range'] +
            weights['Open_Close_Diff'] * row['Open_Close_Diff']
        )
        return alpha_factor
    
    df['Alpha_Factor'] = df.apply(compute_alpha_factor, axis=1)
    
    return df['Alpha_Factor']
```
