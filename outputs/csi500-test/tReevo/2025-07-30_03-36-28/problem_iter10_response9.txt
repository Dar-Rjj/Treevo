```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback periods
    sma_short_period = 20
    volatility_lookback = 10
    momentum_lookback = 20
    pct_change_lookback = 5
    sma_long_period = 50
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Vol'] = (df['High_Low_Diff'] * df['volume']).rolling(window=volatility_lookback).mean()
    
    # Compute Price Momentum
    avg_close = df['close'].rolling(window=momentum_lookback).mean()
    df['Momentum'] = (df['close'] - df['SMA_Short']) / avg_close
    
    # Incorporate Additional Price Change Metrics
    df['Pct_Change'] = df['close'].pct_change(periods=pct_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Dynamic Market Trend Alignment
    df['Trend_Indicator'] = (df['SMA_Short'] > df['SMA_Long']).astype(int)
    
    # Adjust Weights Based on Market Trend
    df['Trend_Weight'] = df['Trend_Indicator'] * 1.5 + (1 - df['Trend_Indicator']) * 0.5
    
    # Consider Sector-Specific Volatility (Assuming sector is a column in the DataFrame)
    # For simplicity, we will use the average volume-adjusted volatility for the sector
    sector_volatility = df.groupby('sector')['Volume_Adjusted_Vol'].transform('mean')
    df['Sector_Volatility'] = sector_volatility
    df['Sector_Volatility_Weight'] = (df['Sector_Volatility'] - df['Sector_Volatility'].min()) / (df['Sector_Volatility'].max() - df['Sector_Volatility'].min())
    
    # Final Alpha Factor
    weights = {
        'Momentum': 0.4,
        'Volume_Adjusted_Vol': 0.3,
        'Pct_Change': 0.1,
        'High_Low_Range': 0.1,
        'Trend_Weight': 0.1
    }
    
    df['Alpha_Factor'] = (
        weights['Momentum'] * df['Momentum'] +
        weights['Volume_Adjusted_Vol'] * df['Volume_Adjusted_Vol'] +
        weights['Pct_Change'] * df['Pct_Change'] +
        weights['High_Low_Range'] * df['High_Low_Range'] *
        df['Trend_Weight'] * df['Sector_Volatility_Weight']
    )
    
    return df['Alpha_Factor'].dropna()
```
