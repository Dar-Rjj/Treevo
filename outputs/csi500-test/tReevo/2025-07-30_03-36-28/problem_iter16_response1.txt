```python
import pandas as pd

def heuristics_v2(df, short_window=20, long_window=100, atr_window=14, turnover_window=30):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_window).mean()

    # Compute Volume-Adjusted Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_high_low_diff'] = df['high_low_diff'] * df['volume']
    df['volatility'] = df['volume_weighted_high_low_diff'].rolling(window=short_window).mean()

    # Enhance Volatility Measures
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].max(axis=1) - df[['high', 'low', 'prev_close']].min(axis=1)
    df['ATR'] = df['true_range'].rolling(window=atr_window).mean()

    # Compute Price Momentum
    df['price_momentum'] = (df['close'] - df['SMA_short']) / df['close'].rolling(window=short_window).mean()

    # Incorporate Additional Price Change Metrics
    df['percentage_change'] = df['close'].pct_change(periods=short_window)
    df['high_low_range'] = df['high'] - df['low']

    # Consider Market Trend Alignment
    df['trend'] = (df['SMA_short'] > df['SMA_long']).astype(int)

    # Incorporate Liquidity Measures
    df['daily_turnover'] = df['volume'] * df['close']
    df['turnover_rolling_avg'] = df['daily_turnover'].rolling(window=turnover_window).mean()
    df['liquidity_factor'] = df['turnover_rolling_avg'] / df['close']

    # Final Alpha Factor
    weights = {
        'price_momentum': 0.4,
        'volatility': -0.2,
        'percentage_change': 0.2,
        'high_low_range': 0.1,
        'liquidity_factor': 0.1
    }

    df['alpha_factor'] = (
        weights['price_momentum'] * df['price_momentum'] +
        weights['volatility'] * df['volatility'] +
        weights['percentage_change'] * df['percentage_change'] +
        weights['high_low_range'] * df['high_low_range'] +
        weights['liquidity_factor'] * df['liquidity_factor']
    )

    # Adjust Weights Based on Market Trend
    df.loc[df['trend'] == 1, 'alpha_factor'] *= 1.2  # Increase weight for bullish trends
    df.loc[df['trend'] == 0, 'alpha_factor'] *= 0.8  # Decrease weight for bearish trends

    # Adjust Weights Based on Liquidity
    df['alpha_factor'] *= df['liquidity_factor'].rank(pct=True)

    return df['alpha_factor']
```
