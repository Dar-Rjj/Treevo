```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    sma_short = 20
    sma_long = 50
    atr_lookback = 14
    historical_volatility_lookback = 20
    price_momentum_lookback = 10
    percentage_change_lookback = 5
    daily_turnover_lookback = 20
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Short'] = df['close'].rolling(window=sma_short).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long).mean()
    
    # Compute Volume-Adjusted Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Volume_Adjusted_Volatility'] = df['High_Low_Diff'] * df['volume']
    df['Rolling_Volume_Adjusted_Volatility'] = df['Volume_Adjusted_Volatility'].rolling(window=atr_lookback).mean()
    
    # Enhance Volatility Measures
    df['True_Range'] = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=atr_lookback).mean()
    
    # Refine Volatility Adjustments
    df['Historical_Volatility'] = df['close'].pct_change().rolling(window=historical_volatility_lookback).std() * np.sqrt(historical_volatility_lookback)
    df['Refined_Volatility'] = df['Rolling_Volume_Adjusted_Volatility'] * (1 + df['Historical_Volatility'])
    
    # Compute Price Momentum
    df['Price_Momentum'] = (df['close'] - df['SMA_Short']) / df['close'].rolling(window=price_momentum_lookback).mean()
    
    # Incorporate Additional Price Change Metrics
    df['Percentage_Change'] = df['close'].pct_change(periods=percentage_change_lookback)
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Consider Market Trend Alignment
    df['Trend_Indicator'] = np.where(df['SMA_Short'] > df['SMA_Long'], 1, -1)
    
    # Incorporate Liquidity Measures
    df['Daily_Turnover'] = df['volume'] * df['close']
    df['Rolling_Daily_Turnover'] = df['Daily_Turnover'].rolling(window=daily_turnover_lookback).mean()
    
    # Dynamic Weight Adjustments Based on Market Conditions and Trends
    df['Weight_Trend'] = np.where(df['Trend_Indicator'] == 1, 1.5, 0.5)
    df['Weight_Liquidity'] = df['Rolling_Daily_Turnover'] / df['Rolling_Daily_Turnover'].max()
    
    # Final Alpha Factor
    weights_sum = 1
    weight_price_momentum = 0.3
    weight_refined_volatility = 0.2
    weight_percentage_change = 0.2
    weight_high_low_range = 0.1
    weight_trend = 0.1
    weight_liquidity = 0.1
    
    df['Alpha_Factor'] = (
        weight_price_momentum * df['Price_Momentum'] +
        weight_refined_volatility * df['Refined_Volatility'] +
        weight_percentage_change * df['Percentage_Change'] +
        weight_high_low_range * df['High_Low_Range'] +
        weight_trend * df['Trend_Indicator'] * df['Weight_Trend'] +
        weight_liquidity * df['Weight_Liquidity']
    )
    
    return df['Alpha_Factor']
```
