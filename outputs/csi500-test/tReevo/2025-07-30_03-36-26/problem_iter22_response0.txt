```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume

    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()

    # Integrate Dynamic Short-Term Momentum (5-10 days)
    short_term_period = 7  # Example: 7 days
    short_term_momentum = vwap_deviation.rolling(window=short_term_period).sum()
    combined_short_term = cumulative_vwap_deviation + short_term_momentum

    # Integrate Dynamic Medium-Term Momentum (10-20 days)
    medium_term_period = 15  # Example: 15 days
    medium_term_momentum = vwap_deviation.rolling(window=medium_term_period).sum()
    combined_medium_term = combined_short_term + medium_term_momentum

    # Integrate Dynamic Long-Term Momentum (20-40 days)
    long_term_period = 30  # Example: 30 days
    long_term_momentum = vwap_deviation.rolling(window=long_term_period).sum()
    combined_long_term = combined_medium_term + long_term_momentum

    # Calculate VWAP Deviation Volatility
    vwap_deviation_volatility = vwap_deviation.rolling(window=long_term_period).std()

    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    absolute_vwap_deviation = (df['close'] - vwap).abs()
    intraday_volatility = high_low_range + absolute_vwap_deviation

    # Final Alpha Factor
    final_alpha_factor = combined_long_term + vwap_deviation_volatility + intraday_volatility

    return final_alpha_factor
```
