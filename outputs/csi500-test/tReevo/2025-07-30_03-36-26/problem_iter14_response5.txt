import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5d
    
    # Incorporate Close-to-Close Momentum
    close_return_5d = df['close'].pct_change(periods=5)
    weighted_high_low_momentum = weighted_high_low_spread + close_return_5d
    
    # Incorporate Volume Momentum
    volume_change_5d = df['volume'].diff(periods=5)
    weighted_high_low_vol_momentum = weighted_high_low_momentum * volume_change_5d
    
    # Incorporate 20-Day Price Trend
    moving_avg_20d = df['close'].rolling(window=20).mean()
    price_trend_20d = df['close'] - moving_avg_20d
    alpha_factor = weighted_high_low_vol_momentum + price_trend_20d
    
    # Incorporate Multi-Period Momentum
    close_return_10d = df['close'].pct_change(periods=10)
    alpha_factor += close_return_10d
    
    # Incorporate Volume Volatility
    volume_std_10d = df['volume'].rolling(window=10).std()
    alpha_factor += volume_std_10d
    
    # Incorporate Price Volatility
    close_std_10d = df['close'].rolling(window=10).std()
    alpha_factor += close_std_10d
    
    # Dynamic Weighting Based on Market Conditions
    # Calculate 30-Day Historical Performance
    cumulative_returns_30d = (df['close'] / df['close'].shift(30)) - 1
    
    # Determine Weights Based on Performance
    weights = (cumulative_returns_30d.rank() / len(cumulative_returns_30d)).fillna(0)
    
    # Adjust Weights Based on Market Volatility
    market_volatility_30d = df['close'].rolling(window=30).std()
    weights = weights * (1 - (market_volatility_30d / market_volatility_30d.max()))
    
    # Integrate Seasonal and Calendar Effects
    # For simplicity, let's assume a fixed seasonal adjustment factor
    seasonal_adjustment = 1.0
    # In practice, you would use historical data to identify seasonal patterns and adjust weights accordingly
    
    # Apply Weights to Final Alpha Factor
    final_alpha_factor = alpha_factor * weights * seasonal_adjustment
    
    return final_alpha_factor
