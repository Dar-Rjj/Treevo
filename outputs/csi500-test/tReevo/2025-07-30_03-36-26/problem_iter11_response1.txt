```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    daily_vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - daily_vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Short-Term Momentum
    short_term_momentum_period = 5
    short_term_momentum = (df['close'] - df['close'].shift(short_term_momentum_period)) / df['close'].shift(short_term_momentum_period)
    short_term_factor = cumulative_vwap_deviation + short_term_momentum
    
    # Integrate Medium-Term Momentum
    medium_term_momentum_period = 10
    medium_term_momentum = (df['close'] - df['close'].shift(medium_term_momentum_period)) / df['close'].shift(medium_term_momentum_period)
    medium_term_factor = cumulative_vwap_deviation + medium_term_momentum
    
    # Adaptive Weighting based on Volatility
    volatility = df['close'].pct_change().rolling(window=20).std()
    high_volatility_threshold = volatility.quantile(0.75)
    low_volatility_threshold = volatility.quantile(0.25)
    
    weights = pd.DataFrame(index=df.index)
    weights['short_term_weight'] = (volatility > high_volatility_threshold).astype(int) * 0.6 + (volatility <= high_volatility_threshold).astype(int) * 0.4
    weights['medium_term_weight'] = (volatility < low_volatility_threshold).astype(int) * 0.6 + (volatility >= low_volatility_threshold).astype(int) * 0.4
    weights['vwap_deviation_weight'] = 1 - (weights['short_term_weight'] + weights['medium_term_weight'])
    
    # Combine Weighted Factors
    final_alpha_factor = (
        weights['vwap_deviation_weight'] * cumulative_vwap_deviation +
        weights['short_term_weight'] * short_term_momentum +
        weights['medium_term_weight'] * medium_term_momentum
    )
    
    return final_alpha_factor
```
