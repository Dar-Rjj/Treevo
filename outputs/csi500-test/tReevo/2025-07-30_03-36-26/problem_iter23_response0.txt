```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['TotalVolume'] = df['volume']
    df['TotalDollarValue'] = df['close'] * df['volume']
    df['VWAP'] = df.groupby(df.index.date)['TotalDollarValue'].cumsum() / df.groupby(df.index.date)['TotalVolume'].cumsum()

    # Calculate VWAP Deviation
    df['VWAPDeviation'] = df['close'] - df['VWAP']

    # Calculate Cumulative VWAP Deviation
    df['CumulativeVWAPDeviation'] = df.groupby(df.index.date)['VWAPDeviation'].cumsum()

    # Integrate Short-Term Momentum (5 days)
    short_term_days = 5
    df['LogReturnsShort'] = np.log(df['close'] / df['close'].shift(1))
    df['SumLogReturnsShort'] = df['LogReturnsShort'].rolling(window=short_term_days).sum()
    df['CumulativeVWAPDeviationShort'] = df['CumulativeVWAPDeviation'] + df['SumLogReturnsShort']

    # Integrate Medium-Term Momentum (10 days)
    medium_term_days = 10
    df['LogReturnsMedium'] = np.log(df['close'] / df['close'].shift(1))
    df['SumLogReturnsMedium'] = df['LogReturnsMedium'].rolling(window=medium_term_days).sum()
    df['CumulativeVWAPDeviationMedium'] = df['CumulativeVWAPDeviation'] + df['SumLogReturnsMedium']

    # Integrate Long-Term Momentum (20 days)
    long_term_days = 20
    df['LogReturnsLong'] = np.log(df['close'] / df['close'].shift(1))
    df['SumLogReturnsLong'] = df['LogReturnsLong'].rolling(window=long_term_days).sum()
    df['CumulativeVWAPDeviationLong'] = df['CumulativeVWAPDeviation'] + df['SumLogReturnsLong']

    # Calculate Intraday Volatility
    df['HighLowRange'] = df['high'] - df['low']
    df['AbsVWAPDeviation'] = np.abs(df['close'] - df['VWAP'])
    df['IntradayVolatility'] = df['HighLowRange'] + df['AbsVWAPDeviation']

    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['CumulativeVWAPDeviation'] 
        + df['SumLogReturnsShort'] 
        + df['SumLogReturnsMedium'] 
        + df['SumLogReturnsLong'] 
        + df['IntradayVolatility']
    )

    return df['FinalAlphaFactor']
```
