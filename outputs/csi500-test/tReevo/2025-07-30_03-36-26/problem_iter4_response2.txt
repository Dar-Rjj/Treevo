```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['VWAP_Deviation'] = df['close'] - df['Daily_VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['Cumulative_VWAP_Deviation'] = df['VWAP_Deviation'].cumsum()
    
    # Integrate Volatility
    df['Daily_Volatility'] = (df['high'] - df['low']) / df['close']
    df['Volatility_Adjusted_VWAP_Deviation'] = df['Cumulative_VWAP_Deviation'] * df['Daily_Volatility']
    
    # Incorporate Market Sentiment
    df['Intraday_Momentum'] = df['close'] - df['open']
    df['Volume_Adjusted_Intraday_Momentum'] = df['Intraday_Momentum'] * df['volume']
    df['Combined_Factor_1'] = df['Volatility_Adjusted_VWAP_Deviation'] + df['Volume_Adjusted_Intraday_Momentum']
    
    # Consider Relative Strength
    df['Returns'] = df['close'].pct_change()
    gain = df['Returns'].apply(lambda x: max(x, 0))
    loss = df['Returns'].apply(lambda x: -min(x, 0))
    
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    df['RSI'] = 100 - (100 / (1 + (avg_gain / avg_loss)))
    df['Final_Alpha_Factor'] = df['Combined_Factor_1'] + df['RSI']
    
    return df['Final_Alpha_Factor']
```
