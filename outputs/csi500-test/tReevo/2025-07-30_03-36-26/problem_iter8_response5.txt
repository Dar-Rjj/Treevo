import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average Volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5
    
    # Incorporate Close-to-Close Momentum
    close_return_5 = df['close'].pct_change(periods=5)
    factor_1 = weighted_high_low_spread + close_return_5
    
    # Incorporate Volume Momentum
    volume_change_5 = df['volume'].diff(periods=5)
    factor_2 = factor_1 * volume_change_5
    
    # Incorporate 20-Day Price Trend
    moving_avg_20 = df['close'].rolling(window=20).mean()
    price_trend_20 = df['close'] - moving_avg_20
    factor_3 = factor_2 + price_trend_20
    
    # Incorporate Multi-Period Momentum
    close_return_10 = df['close'].pct_change(periods=10)
    factor_4 = factor_3 + close_return_10
    
    # Incorporate Volume Volatility
    volume_volatility_10 = df['volume'].rolling(window=10).std()
    factor_5 = factor_4 + volume_volatility_10
    
    # Incorporate Price Volatility
    price_volatility_10 = df['close'].rolling(window=10).std()
    factor_6 = factor_5 + price_volatility_10
    
    # Dynamic Weighting Based on Market Conditions
    historical_performance_30 = (df['close'] / df['close'].shift(30)) - 1
    market_volatility_30 = df['close'].rolling(window=30).std()
    
    # Determine Weights Based on Performance and Market Volatility
    performance_weight = (historical_performance_30 - historical_performance_30.min()) / (historical_performance_30.max() - historical_performance_30.min())
    volatility_weight = 1 - (market_volatility_30 / market_volatility_30.max())
    final_weights = performance_weight * volatility_weight
    
    # Apply Weights to Final Alpha Factor
    final_alpha_factor = final_weights * factor_6
    
    return final_alpha_factor
