import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['TotalVolume'] = df['volume']
    df['DollarValue'] = df['close'] * df['volume']
    df['VWAP'] = (df.groupby(df.index)['DollarValue'].transform('sum') /
                  df.groupby(df.index)['TotalVolume'].transform('sum'))
    
    # Calculate VWAP Deviation
    df['VWAPDeviation'] = df['close'] - df['VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['CumulativeVWAPDeviation'] = df['VWAPDeviation'].rolling(window=len(df), min_periods=1).sum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum = df['close'].pct_change(periods=10)
    
    # Final Alpha Factor
    df['AlphaFactor'] = (df['CumulativeVWAPDeviation'] 
                         + short_term_momentum 
                         + medium_term_momentum)
    
    return df['AlphaFactor'].dropna()
