import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume

    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()

    # Integrate Adaptive Momentum
    def adaptive_momentum_period(volatility):
        if volatility < 1.0:
            return 5
        elif volatility < 2.0:
            return 10
        else:
            return 20
    
    recent_volatility = df['close'].rolling(window=20).std()
    momentum_period = recent_volatility.apply(adaptive_momentum_period)
    momentum = vwap_deviation.rolling(window=momentum_period, min_periods=1).sum()

    # Combine with Cumulative VWAP Deviation
    combined_factor = cumulative_vwap_deviation + momentum

    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    absolute_vwap_deviation = np.abs(df['close'] - vwap)
    intraday_volatility = high_low_range + absolute_vwap_deviation

    # Final Alpha Factor
    final_alpha_factor = combined_factor + intraday_volatility

    return final_alpha_factor
