```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Short-Term Momentum
    short_term_momentum_period = 5
    short_term_momentum = vwap_deviation.rolling(window=short_term_momentum_period).sum()
    weighted_short_term_momentum = short_term_momentum * 0.3
    cumulative_vwap_deviation += weighted_short_term_momentum
    
    # Integrate Medium-Term Momentum
    medium_term_momentum_period = 10
    medium_term_momentum = vwap_deviation.rolling(window=medium_term_momentum_period).sum()
    weighted_medium_term_momentum = medium_term_momentum * 0.2
    cumulative_vwap_deviation += weighted_medium_term_momentum
    
    # Integrate Long-Term Momentum
    long_term_momentum_period = 20
    long_term_momentum = vwap_deviation.rolling(window=long_term_momentum_period).sum()
    weighted_long_term_momentum = long_term_momentum * 0.1
    cumulative_vwap_deviation += weighted_long_term_momentum
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    abs_vwap_deviation = (df['close'] - vwap).abs()
    intraday_volatility = high_low_range + abs_vwap_deviation
    dynamic_adjustment_factor = 1.0  # Example: This can be dynamically adjusted based on market conditions
    dynamically_adjusted_intraday_volatility = intraday_volatility * dynamic_adjustment_factor
    
    # Final Alpha Factor
    final_alpha_factor = cumulative_vwap_deviation + dynamically_adjusted_intraday_volatility
    
    return final_alpha_factor
```
