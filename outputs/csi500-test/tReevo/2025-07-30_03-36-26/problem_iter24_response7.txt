```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume']
    total_dollar_value = df['close'] * df['volume']
    vwap = (total_dollar_value / total_volume).fillna(0)
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum_period = 5
    short_term_vwap_deviations = vwap_deviation.rolling(window=short_term_momentum_period).sum().fillna(0)
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum_period = 10
    medium_term_vwap_deviations = vwap_deviation.rolling(window=medium_term_momentum_period).sum().fillna(0)
    
    # Integrate Long-Term Momentum (20 days)
    long_term_momentum_period = 20
    long_term_vwap_deviations = vwap_deviation.rolling(window=long_term_momentum_period).sum().fillna(0)
    
    # Combine with Cumulative VWAP Deviation
    combined_factor = cumulative_vwap_deviation + short_term_vwap_deviations + medium_term_vwap_deviations + long_term_vwap_deviations
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    absolute_vwap_deviation = (df['close'] - vwap).abs()
    intraday_volatility = 0.7 * high_low_range + 0.3 * absolute_vwap_deviation
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + intraday_volatility
    
    return final_alpha_factor
```
