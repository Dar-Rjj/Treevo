import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Close-to-Current Day Open Return
    df['prev_close_to_current_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Average of Intraday and Previous Day to Current Day Open Returns
    df['average_return'] = (df['intraday_return'] + df['prev_close_to_current_open_return']) / 2
    
    # Calculate Volume Difference
    df['volume_difference'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust by Volume Difference
    df['adjusted_average_return'] = df['average_return'] * (1 + df['volume_difference'])
    
    # Drop the first row which will have NaN values due to shift operations
    df = df.dropna()
    
    # Return the factor values
    return df['adjusted_average_return']
