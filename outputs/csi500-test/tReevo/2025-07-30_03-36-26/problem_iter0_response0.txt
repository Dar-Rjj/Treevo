```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    -----------
    df : pandas.DataFrame
        DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    --------
    pandas.Series
        Series with the factor values indexed by (date).
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
