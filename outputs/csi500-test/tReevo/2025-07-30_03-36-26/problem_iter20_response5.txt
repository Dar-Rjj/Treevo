import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_dollar_value'] = df['volume'] * df['close']
    total_volume = df['volume'].sum()
    total_dollar_value = df['total_dollar_value'].sum()
    vwap = total_dollar_value / total_volume
    df['vwap'] = (df['total_dollar_value'].cumsum() / df['volume'].cumsum()).fillna(0)
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum_period = 5
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term_momentum_period).sum().fillna(0)
    df['cumulative_vwap_deviation'] += df['short_term_momentum']
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum_period = 10
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term_momentum_period).sum().fillna(0)
    df['cumulative_vwap_deviation'] += df['medium_term_momentum']
    
    # Integrate Long-Term Momentum (20 days)
    long_term_momentum_period = 20
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term_momentum_period).sum().fillna(0)
    df['cumulative_vwap_deviation'] += df['long_term_momentum']
    
    # Calculate Dynamic Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['abs_vwap_deviation'] = abs(df['close'] - df['vwap'])
    df['dynamic_intraday_volatility'] = (df['high_low_range'] + df['abs_vwap_deviation']) / 2
    
    # Final Alpha Factor
    df['alpha_factor'] = df['cumulative_vwap_deviation'] + df['dynamic_intraday_volatility']
    
    return df['alpha_factor']
