```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 100 Day Moving Average of Close Prices
    df['100_day_ma'] = df['close'].rolling(window=100).mean()
    # Subtract 100 Day Moving Average from Current Close Price
    long_term_trend = df['close'] - df['100_day_ma']

    # Calculate 20 Day Moving Average of Close Prices
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    # Subtract 20 Day Moving Average from Current Close Price
    medium_term_momentum = df['close'] - df['20_day_ma_close']

    # Calculate 20 Day Moving Average of Volumes
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    # Divide Current Volume by its 20 Day Moving Average
    volume_trend = df['volume'] / df['20_day_ma_volume']

    # Calculate 20 Day Rolling Correlation between Close Prices and Volumes
    df['volume_price_corr'] = df['close'].rolling(window=20).corr(df['volume'])

    # Synchronized Change
    synchronized_change = medium_term_momentum * volume_trend
    # Adjust by Volume-Price Correlation
    synchronized_change = np.where(
        df['volume_price_corr'] > 0,
        synchronized_change * (1 + df['volume_price_corr']),
        synchronized_change * (1 - np.abs(df['volume_price_corr']))
    )

    # Apply Threshold Filtering
    threshold = 0.5
    synchronized_change = np.where(
        np.abs(synchronized_change) > threshold,
        synchronized_change,
        0
    )

    # Combine Synchronized Change with Long-Term Price Trend
    # Weighted Sum of Synchronized Change and Long-Term Trend
    factor_value = 0.7 * synchronized_change + 0.3 * long_term_trend

    return pd.Series(factor_value, index=df.index)
```
