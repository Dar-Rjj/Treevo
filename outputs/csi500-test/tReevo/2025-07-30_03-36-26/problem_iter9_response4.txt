```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['daily_vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['daily_vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum_period = 5
    df['short_term_momentum'] = df['close'].pct_change(periods=short_term_momentum_period)
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum_period = 10
    df['medium_term_momentum'] = df['close'].pct_change(periods=medium_term_momentum_period)
    
    # Incorporate Exponential Moving Average (EMA) on Close Price
    ema_window = 10
    df['ema_close'] = df['close'].ewm(span=ema_window, adjust=False).mean()
    
    # Combine Short-Term and Medium-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] + df['medium_term_momentum']
    
    # Adjust Weights Based on Recent Performance
    recent_performance_weight = 0.7  # Example weight
    df['adjusted_momentum'] = (recent_performance_weight * df['short_term_momentum']) + \
                              ((1 - recent_performance_weight) * df['medium_term_momentum'])
    
    # Incorporate Volume Trend
    volume_ma_window = 10
    df['volume_trend'] = df['volume'].rolling(window=volume_ma_window).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['cumulative_vwap_deviation'] +
                          df['adjusted_momentum'] +
                          (df['ema_close'] - df['close']) +
                          (df['volume'] - df['volume_trend']))
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
