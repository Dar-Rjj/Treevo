```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['daily_dollar_value'] = df['volume'] * df['close']
    df['total_volume'] = df['volume'].rolling(window=1).sum()
    df['total_dollar_value'] = df['daily_dollar_value'].rolling(window=1).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Short-Term Momentum (5 days)
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=5).sum()
    df['cumulative_vwap_deviation'] += df['short_term_momentum']

    # Integrate Medium-Term Momentum (10 days)
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=10).sum()
    df['cumulative_vwap_deviation'] += df['medium_term_momentum']

    # Integrate Long-Term Momentum (20 days)
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=20).sum()
    df['cumulative_vwap_deviation'] += df['long_term_momentum']

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['abs_vwap_deviation'] = abs(df['close'] - df['vwap'])
    df['intraday_volatility'] = df['high_low_range'] + df['abs_vwap_deviation']

    # Incorporate Short-Term Trend Strength
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['short_term_trend_strength'] = (df['close'] - df['sma_5']) * df['short_term_momentum']

    # Incorporate Medium-Term Trend Strength
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['medium_term_trend_strength'] = (df['close'] - df['sma_10']) * df['medium_term_momentum']

    # Incorporate Long-Term Trend Strength
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['long_term_trend_strength'] = (df['close'] - df['sma_20']) * df['long_term_momentum']

    # Final Alpha Factor
    df['alpha_factor'] = (df['cumulative_vwap_deviation'] 
                          + df['short_term_momentum'] 
                          + df['medium_term_momentum'] 
                          + df['long_term_momentum'] 
                          + df['intraday_volatility'] 
                          + df['short_term_trend_strength'] 
                          + df['medium_term_trend_strength'] 
                          + df['long_term_trend_strength'])

    return df['alpha_factor']
```
