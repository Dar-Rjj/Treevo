```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Total Volume and Total Dollar Value for Intraday VWAP
    df['Total_Volume'] = df['volume']
    df['Dollar_Value'] = df['volume'] * (df['open'] + df['high'] + df['low'] + df['close']) / 4
    total_dollar_value = df.groupby(level='date')['Dollar_Value'].sum()
    total_volume = df.groupby(level='date')['Total_Volume'].sum()

    # Calculate Intraday VWAP
    vwap = total_dollar_value / total_volume
    df['VWAP'] = df.index.get_level_values('date').map(vwap)

    # Calculate Average Price
    df['Average_Price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4

    # Calculate VWAP Deviation
    df['VWAP_Deviation'] = df['Average_Price'] - df['VWAP']

    # Incorporate Liquidity
    df['Daily_Turnover'] = ((df['volume'].shift(1) + df['volume'].shift(2)) / 2) * df['close']
    df['Adjusted_VWAP_Deviation'] = df['VWAP_Deviation'] * df['Daily_Turnover']

    # Calculate Close-to-Close Momentum
    df['Close_to_Close_Momentum'] = df['close'].pct_change(1)

    # Calculate Cumulative Adjusted VWAP Deviation
    df['Cumulative_Adjusted_VWAP_Deviation'] = df.groupby(level='date')['Adjusted_VWAP_Deviation'].cumsum()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Cumulative_Adjusted_VWAP_Deviation'] + df['Close_to_Close_Momentum']

    return df['Alpha_Factor']
```
