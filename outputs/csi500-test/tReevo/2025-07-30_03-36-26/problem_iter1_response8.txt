import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Daily_Momentum'] = df['close'].diff()
    
    # Calculate Volume Growth
    df['Volume_Growth'] = df['volume'].diff()
    
    # Combine Momentum and Volume
    constant = 1e-6  # Small constant to avoid division by zero
    df['Combined_MV'] = (df['Daily_Momentum'] * df['Volume_Growth']) + constant
    
    # Weight by Volume
    df['Weighted_MV'] = df['Combined_MV'] / df['volume']
    
    # Smoothen the Factor
    short_term_window = 5
    df['Smoothed_Factor'] = df['Weighted_MV'].ewm(span=short_term_window, adjust=False).mean()
    
    # Normalize the Result by Its Own Maximum within the Short Term Window
    df['Max_Short_Term'] = df['Smoothed_Factor'].rolling(window=short_term_window).max()
    df['Normalized_Factor'] = df['Smoothed_Factor'] / df['Max_Short_Term']
    
    # Drop rows with NaN values created by the rolling and ewm functions
    df.dropna(inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['Normalized_Factor']
