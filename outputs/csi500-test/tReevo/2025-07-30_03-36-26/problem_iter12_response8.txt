```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Short-Term and Medium-Term Momentum
    volatility = df['close'].rolling(window=20).std()
    high_volatility = volatility > volatility.mean()
    
    # Dynamic Short-Term Momentum Period
    short_term_period = 3 + 2 * (volatility < volatility.mean())
    short_term_momentum = (df['close'] - df['close'].shift(short_term_period)) / df['close'].shift(short_term_period)
    
    # Dynamic Medium-Term Momentum Period
    medium_term_period = 10 + 4 * (volatility < volatility.mean())
    medium_term_momentum = (df['close'] - df['close'].shift(medium_term_period)) / df['close'].shift(medium_term_period)
    
    # Adaptive Weighting
    w1 = 1 / (1 + volatility)
    w2 = 0.5 + 0.5 * (volatility - volatility.min()) / (volatility.max() - volatility.min())
    w3 = 1 - w1 - w2
    
    # Combine Weighted Factors
    final_alpha_factor = (w1 * cumulative_vwap_deviation) + (w2 * short_term_momentum) + (w3 * medium_term_momentum)
    
    return final_alpha_factor
```
