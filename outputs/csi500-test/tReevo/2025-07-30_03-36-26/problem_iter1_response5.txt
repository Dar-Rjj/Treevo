import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread to Average True Range Ratio
    high = df['high']
    low = df['low']
    close = df['close']
    volume = df['volume']
    
    # High - Low
    high_low_spread = high - low
    
    # Calculate True Range
    prev_close = close.shift(1)
    true_range = pd.DataFrame({
        'high_prev_close': np.abs(high - prev_close),
        'prev_close_low': np.abs(prev_close - low),
        'high_low': high - low
    }).max(axis=1)
    
    # 20-Day Moving Average of True Range
    atr_20 = true_range.rolling(window=20).mean()
    
    # High-Low/ATR Ratio
    hl_atr_ratio = high_low_spread / atr_20
    
    # Derive Momentum in Volume-Adjusted Close Prices
    adj_close = close * volume
    log_adj_close = np.log(adj_close)
    five_day_log_return = log_adj_close - log_adj_close.shift(5)
    
    # Normalize 5-Day Log Return (not using global stats)
    norm_five_day_log_return = (five_day_log_return - five_day_log_return.rolling(window=20).mean()) / five_day_log_return.rolling(window=20).std()
    
    # Incorporate Open-to-High Gap as a Reversal Indicator
    open_price = df['open']
    open_high_gap = (high - open_price) / open_price
    
    # Combine All Factors into a Composite Alpha Signal
    # Assign Weights to Each Factor
    weight_hl_atr = 0.4
    weight_vol_adj_mom = 0.3
    weight_open_high_gap = 0.3
    
    # Sum Weighted Factors for Final Alpha Score
    alpha_score = (weight_hl_atr * hl_atr_ratio + 
                   weight_vol_adj_mom * norm_five_day_log_return + 
                   weight_open_high_gap * open_high_gap)
    
    return alpha_score.dropna()
