```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return using High and Low prices
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume
    df['volume_adjusted_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Cumulative Volume-Adjusted Return over 5 days
    df['cumulative_volume_adjusted_return'] = df['volume_adjusted_return'].rolling(window=5).sum()
    
    # Return the alpha factor as a pandas Series
    return df['cumulative_volume_adjusted_return']
```
