```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['Total_Dollar_Value'] = df['volume'] * df['close']
    df['VWAP'] = df['Total_Dollar_Value'].rolling(window=1, min_periods=1).sum() / df['volume'].rolling(window=1, min_periods=1).sum()

    # Calculate VWAP Deviation
    df['VWAP_Deviation'] = df['close'] - df['VWAP']

    # Calculate Cumulative VWAP Deviation
    df['Cumulative_VWAP_Deviation'] = df['VWAP_Deviation'].cumsum()

    # Integrate Short-Term Momentum (5 days)
    df['Short_Term_Momentum'] = df['VWAP_Deviation'].rolling(window=5, min_periods=1).sum()
    df['Cumulative_VWAP_Deviation'] += df['Short_Term_Momentum']

    # Integrate Medium-Term Momentum (10 days)
    df['Medium_Term_Momentum'] = df['VWAP_Deviation'].rolling(window=10, min_periods=1).sum()
    df['Cumulative_VWAP_Deviation'] += df['Medium_Term_Momentum']

    # Integrate Long-Term Momentum (20 days)
    df['Long_Term_Momentum'] = df['VWAP_Deviation'].rolling(window=20, min_periods=1).sum()
    df['Cumulative_VWAP_Deviation'] += df['Long_Term_Momentum']

    # Calculate Intraday Volatility
    df['High_Low_Range'] = df['high'] - df['low']
    df['Absolute_VWAP_Deviation'] = abs(df['close'] - df['VWAP'])
    df['Intraday_Volatility'] = df['High_Low_Range'] + df['Absolute_VWAP_Deviation']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Cumulative_VWAP_Deviation'] + df['Intraday_Volatility']

    return df['Final_Alpha_Factor']
```
