import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['close'] * df['volume']
    df['vwap'] = df.groupby(df.index.date)['total_dollar_value'].transform('sum') / df.groupby(df.index.date)['total_volume'].transform('sum')
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Weighted Cumulative VWAP Deviation
    weight = 0.9  # decay factor
    df['weighted_cumulative_vwap_deviation'] = 0
    for i in range(1, len(df)):
        df.loc[df.index[i], 'weighted_cumulative_vwap_deviation'] = weight * df.loc[df.index[i-1], 'weighted_cumulative_vwap_deviation'] + df.loc[df.index[i], 'vwap_deviation']
    
    # Integrate Short-Term Momentum
    short_term = 5
    short_weights = [weight**(short_term-i) for i in range(1, short_term+1)]
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term).apply(lambda x: sum(x[-i] * short_weights[i-1] for i in range(1, short_term+1)), raw=True)
    
    # Integrate Medium-Term Momentum
    medium_term = 10
    medium_weights = [weight**(medium_term-i) for i in range(1, medium_term+1)]
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term).apply(lambda x: sum(x[-i] * medium_weights[i-1] for i in range(1, medium_term+1)), raw=True)
    
    # Integrate Long-Term Momentum
    long_term = 20
    long_weights = [weight**(long_term-i) for i in range(1, long_term+1)]
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term).apply(lambda x: sum(x[-i] * long_weights[i-1] for i in range(1, long_term+1)), raw=True)
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['absolute_vwap_deviation'] = df['vwap_deviation'].abs()
    df['open_close_volatility'] = (df['close'] - df['open']).abs()
    df['intraday_volatility'] = df['high_low_range'] + df['absolute_vwap_deviation'] + df['open_close_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_cumulative_vwap_deviation'] 
                          + df['short_term_momentum'] 
                          + df['medium_term_momentum'] 
                          + df['long_term_momentum'] 
                          + df['intraday_volatility'])
    
    return df['alpha_factor']
