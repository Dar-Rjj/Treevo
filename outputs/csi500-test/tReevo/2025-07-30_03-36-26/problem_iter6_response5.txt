import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['dollar_value'] = df['volume'] * (df['high'] + df['low'] + df['close']) / 3
    df['vwap'] = df['dollar_value'] / df['total_volume']

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Intraday Momentum Intensity
    df['intraday_range_percentage'] = (df['high'] - df['low']) / df['low']
    df['intraday_momentum_intensity'] = df['volume'] * df['intraday_range_percentage']
    df['cumulative_vwap_deviation_with_momentum'] = df['cumulative_vwap_deviation'] * df['intraday_momentum_intensity']

    # Apply Dynamic Weighting
    lookback_period = 20  # Define the lookback period for historical standard deviation
    df['vwap_deviation_std'] = df['vwap_deviation'].rolling(window=lookback_period).std()
    df['current_vwap_deviation_z_score'] = df['vwap_deviation'] / df['vwap_deviation_std']
    df['final_alpha_factor'] = df['cumulative_vwap_deviation_with_momentum'] * df['current_vwap_deviation_z_score']

    # Return the final alpha factor
    return df['final_alpha_factor'].dropna()
