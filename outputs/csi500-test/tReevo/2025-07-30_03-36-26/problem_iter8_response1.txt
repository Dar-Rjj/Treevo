```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].sum()
    df['total_dollar_value'] = (df['close'] * df['volume']).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum
    short_term_period = 5
    df['short_term_momentum'] = df['close'].pct_change(short_term_period)
    
    # Integrate Medium-Term Momentum
    medium_term_period = 10
    df['medium_term_momentum'] = df['close'].pct_change(medium_term_period)
    
    # Incorporate Volume Trend
    df['volume_change'] = df['volume'].diff()
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['cumulative_vwap_deviation'] +
        df['short_term_momentum'] +
        df['medium_term_momentum'] +
        df['volume_change']
    )
    
    return df['alpha_factor']
```
