```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['dollar_value'] = df['volume'] * (df['high'] + df['low'] + df['close']) / 3
    df['vwap'] = df['dollar_value'] / df['total_volume']

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Intraday Momentum Intensity
    df['intraday_range_percentage'] = (df['high'] - df['low']) / df['low']
    df['intraday_momentum_intensity'] = df['intraday_range_percentage'] * df['volume']
    df['adjusted_cumulative_vwap_deviation'] = df['cumulative_vwap_deviation'] * df['intraday_momentum_intensity']

    # Incorporate Multi-Period Momentum
    df['5_day_return'] = df['close'].pct_change(periods=5)
    df['20_day_return'] = df['close'].pct_change(periods=20)

    # Weighted Sum of 5-Day and 20-Day Close Price Returns
    df['weighted_5_day_return'] = 0.6 * df['5_day_return']
    df['weighted_20_day_return'] = 0.4 * df['20_day_return']

    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_cumulative_vwap_deviation'] * (df['weighted_5_day_return'] + df['weighted_20_day_return'])

    return df['alpha_factor'].dropna()
```
