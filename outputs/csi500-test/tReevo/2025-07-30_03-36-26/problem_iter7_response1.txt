```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['close'] * df['volume']
    df['vwap'] = df.groupby(df.index.date)['total_dollar_value'].cumsum() / df.groupby(df.index.date)['total_volume'].cumsum()

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Adaptive Exponential Moving Average (EMA)
    short_term_ema_period = 5
    medium_term_ema_period = 10
    df['short_ema'] = df['close'].ewm(span=short_term_ema_period, adjust=False).mean()
    df['medium_ema'] = df['close'].ewm(span=medium_term_ema_period, adjust=False).mean()
    df['adaptive_ema'] = (df['short_ema'] + df['medium_ema']) / 2

    # Integrate Multi-Period Momentum
    short_momentum_period = 5
    medium_momentum_period = 10
    df['short_momentum'] = df['close'].pct_change(periods=short_momentum_period)
    df['medium_momentum'] = df['close'].pct_change(periods=medium_momentum_period)
    df['multi_period_momentum'] = (df['short_momentum'] + df['medium_momentum']) / 2

    # Integrate Volatility
    short_volatility_period = 5
    medium_volatility_period = 10
    df['short_volatility'] = df['close'].rolling(window=short_volatility_period).std()
    df['medium_volatility'] = df['close'].rolling(window=medium_volatility_period).std()
    df['multi_period_volatility'] = (df['short_volatility'] + df['medium_volatility']) / 2

    # Incorporate Intraday Trends
    df['intraday_high_low_range'] = df['high'] - df['low']
    df['intraday_open_close_range'] = df['close'] - df['open']
    df['intraday_trend'] = (df['intraday_high_low_range'] + df['intraday_open_close_range']) / 2

    # Integrate Volume-Adjusted Returns
    df['daily_return'] = df['close'].pct_change()
    df['volume_adjusted_return'] = df['daily_return'] * df['volume']
    df['volume_adjusted_returns'] = df['volume_adjusted_return'].rolling(window=10).mean()

    # Final Alpha Factor
    df['alpha_factor'] = (
        df['cumulative_vwap_deviation'] +
        df['adaptive_ema'] +
        df['multi_period_momentum'] -
        df['multi_period_volatility'] +
        df['intraday_trend'] +
        df['volume_adjusted_returns']
    )

    return df['alpha_factor']
```
