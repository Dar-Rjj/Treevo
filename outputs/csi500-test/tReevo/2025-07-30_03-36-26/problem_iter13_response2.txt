```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=5).sum()
    df['cum_vwap_deviation_with_short_term'] = df['cum_vwap_deviation'] + df['short_term_momentum']
    
    # Integrate Medium-Term Momentum (10 days)
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=10).sum()
    df['cum_vwap_deviation_with_medium_term'] = df['cum_vwap_deviation_with_short_term'] + df['medium_term_momentum']
    
    # Integrate Long-Term Momentum (20 days)
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=20).sum()
    df['cum_vwap_deviation_with_long_term'] = df['cum_vwap_deviation_with_medium_term'] + df['long_term_momentum']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) + abs(df['close'] - df['vwap'])
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['cum_vwap_deviation_with_long_term'] 
                          + df['intraday_volatility'])
    
    return df['alpha_factor']
```
