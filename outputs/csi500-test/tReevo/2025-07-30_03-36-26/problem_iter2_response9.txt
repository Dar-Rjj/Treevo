```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5d
    
    # Incorporate Close-to-Close Momentum
    close_return_5d = (df['close'] / df['close'].shift(4)) - 1
    factor_with_momentum = weighted_high_low_spread + close_return_5d
    
    # Incorporate Volume Momentum
    volume_change_5d = (df['volume'] / df['volume'].shift(4)) - 1
    factor_with_vol_momentum = factor_with_momentum * volume_change_5d
    
    # Incorporate 20-Day Price Trend
    moving_avg_20d = df['close'].rolling(window=20).mean()
    price_trend_20d = df['close'] - moving_avg_20d
    final_factor = factor_with_vol_momentum + price_trend_20d
    
    return final_factor
```
