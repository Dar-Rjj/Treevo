```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by 5-Day Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5d
    
    # Incorporate 5-Day Close Price Return (Momentum)
    close_return_5d = (df['close'] / df['close'].shift(4)) - 1
    alpha_factor = weighted_high_low_spread + close_return_5d
    
    # Incorporate 5-Day Volume Change
    volume_change_5d = (df['volume'] / df['volume'].shift(4)) - 1
    alpha_factor *= volume_change_5d
    
    # Incorporate 20-Day Price Trend
    moving_avg_20d = df['close'].rolling(window=20).mean()
    price_trend_20d = df['close'] - moving_avg_20d
    alpha_factor += price_trend_20d
    
    # Incorporate 10-Day Close Price Return (Multi-Period Momentum)
    close_return_10d = (df['close'] / df['close'].shift(9)) - 1
    alpha_factor += close_return_10d
    
    # Incorporate 10-Day Volume Standard Deviation (Volume Volatility)
    vol_volatility_10d = df['volume'].rolling(window=10).std()
    alpha_factor += vol_volatility_10d
    
    # Incorporate 10-Day Close Price Standard Deviation (Price Volatility)
    price_volatility_10d = df['close'].rolling(window=10).std()
    alpha_factor += price_volatility_10d
    
    return alpha_factor
```
