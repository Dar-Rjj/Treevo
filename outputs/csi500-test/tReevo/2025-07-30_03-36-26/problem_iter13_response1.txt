```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    df['short_momentum'] = df['vwap_deviation'].rolling(window=5).sum()
    df['cum_vwap_deviation'] += df['short_momentum']
    
    # Integrate Medium-Term Momentum (10 days)
    df['medium_momentum'] = df['vwap_deviation'].rolling(window=10).sum()
    df['cum_vwap_deviation'] += df['medium_momentum']
    
    # Integrate Long-Term Momentum (20 days)
    df['long_momentum'] = df['vwap_deviation'].rolling(window=20).sum()
    df['cum_vwap_deviation'] += df['long_momentum']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).abs() + (df['close'] - df['vwap']).abs()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['cum_vwap_deviation'] 
                          + df['short_momentum'] 
                          + df['medium_momentum'] 
                          + df['long_momentum'] 
                          + df['intraday_volatility'])
    
    return df['alpha_factor']
```
