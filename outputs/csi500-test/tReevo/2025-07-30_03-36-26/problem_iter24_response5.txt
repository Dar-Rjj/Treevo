```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].sum()
    df['total_dollar_value'] = (df['close'] * df['volume']).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum
    short_term_period = 5  # Example period, can be adaptive
    decay_rate = 0.01  # Example decay rate, can be tuned
    weights = np.array([np.exp(-decay_rate * t) for t in range(short_term_period)])
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term_period).apply(lambda x: np.sum(x * weights), raw=True)
    df['cum_vwap_deviation'] += df['short_term_momentum']
    
    # Integrate Adaptive Medium-Term Momentum
    medium_term_period = 10  # Example period, can be adaptive
    weights = np.array([np.exp(-decay_rate * t) for t in range(medium_term_period)])
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term_period).apply(lambda x: np.sum(x * weights), raw=True)
    df['cum_vwap_deviation'] += df['medium_term_momentum']
    
    # Integrate Adaptive Long-Term Momentum
    long_term_period = 20  # Example period, can be adaptive
    weights = np.array([np.exp(-decay_rate * t) for t in range(long_term_period)])
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term_period).apply(lambda x: np.sum(x * weights), raw=True)
    df['cum_vwap_deviation'] += df['long_term_momentum']
    
    # Calculate VWAP Deviation Volatility
    recent_period = 20  # Example recent period, can be tuned
    df['vwap_dev_volatility'] = df['vwap_deviation'].rolling(window=recent_period).std()
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['abs_vwap_deviation'] = df['vwap_deviation'].abs()
    df['intraday_volatility'] = df['high_low_range'] + df['abs_vwap_deviation']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['cum_vwap_deviation'] + df['vwap_dev_volatility'] + df['intraday_volatility']
    
    return df['alpha_factor']
```
