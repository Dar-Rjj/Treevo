```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Trend
    long_term_trend = df['close'].rolling(window=100).mean()
    long_term_diff = df['close'] - long_term_trend
    
    # Calculate Medium-Term Price Momentum
    medium_term_momentum = df['close'].rolling(window=20).mean()
    medium_term_diff = df['close'] - medium_term_momentum
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    short_term_diff = df['close'] - short_term_momentum
    
    # Assess Volume Trend
    volume_trend = df['volume'].rolling(window=20).mean()
    volume_adj = df['volume'] / volume_trend
    
    # Evaluate Volume-Price Correlation
    vol_price_corr = df['close'].rolling(window=20).corr(df['volume'])
    
    # Calculate Synchronized Change
    synchronized_change_med = medium_term_diff * volume_adj
    synchronized_change_short = short_term_diff * volume_adj
    
    # Adjust by Volume-Price Correlation
    synchronized_change = (synchronized_change_med + synchronized_change_short) * \
                          (vol_price_corr > 0).apply(lambda x: 1.5 if x else 0.5)
    
    # Filter for Strong Signals
    threshold = 0.05
    synchronized_change = synchronized_change.abs().where(synchronized_change.abs() > threshold, 0)
    
    # Incorporate Long-Term Trend
    final_signal = 0.6 * synchronized_change + 0.4 * long_term_diff
    
    return final_signal
```
