```python
import pandas as pd

def heuristics_v2(df, threshold=0.5):
    # Calculate 20 Day Moving Average of Close Prices
    df['close_20_day_ma'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20 Day Moving Average from Current Close Price to get Price Momentum
    df['price_momentum'] = df['close'] - df['close_20_day_ma']
    
    # Calculate 20 Day Moving Average of Volumes
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    
    # Divide Current Volume by its 20 Day Moving Average to get Volume Trend Adjustment
    df['volume_trend_adjustment'] = df['volume'] / df['volume_20_day_ma']
    
    # Multiply Price Momentum by the Volume Trend Adjustment to get Synchronized Change
    df['synchronized_change'] = df['price_momentum'] * df['volume_trend_adjustment']
    
    # Apply Threshold Filtering to Filter for Strong Signals
    df['filtered_signal'] = df['synchronized_change'].where(df['synchronized_change'].abs() > threshold, other=0)
    
    # Return the alpha factor as a pandas Series
    return df['filtered_signal']
```
