```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Trend
    long_term_trend = df['close'] - df['close'].rolling(window=100).mean()
    
    # Calculate Medium-Term Price Momentum
    medium_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Assess Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Evaluate Volume Trends
    long_term_volume_trend = df['volume'] / df['volume'].rolling(window=100).mean()
    medium_term_volume_trend = df['volume'] / df['volume'].rolling(window=20).mean()
    short_term_volume_trend = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Volume-Price Correlation Analysis
    long_term_corr = df['close'].rolling(window=100).corr(df['volume'])
    medium_term_corr = df['close'].rolling(window=20).corr(df['volume'])
    short_term_corr = df['close'].rolling(window=5).corr(df['volume'])
    
    # Synchronized Change Calculation
    synchronized_change_medium = medium_term_momentum * medium_term_volume_trend
    synchronized_change_short = short_term_momentum * short_term_volume_trend
    
    # Adjust by Volume-Price Correlation
    synchronized_change_medium = synchronized_change_medium * (medium_term_corr > 0) * (medium_term_corr + 1)
    synchronized_change_short = synchronized_change_short * (short_term_corr > 0) * (short_term_corr + 1)
    
    # Filter for Strong Signals
    threshold = 0.5
    synchronized_change_medium = synchronized_change_medium.where(synchronized_change_medium.abs() > threshold, 0)
    synchronized_change_short = synchronized_change_short.where(synchronized_change_short.abs() > threshold, 0)
    
    # Incorporate Long-Term Trend
    alpha_factor = 0.5 * (synchronized_change_medium + synchronized_change_short) + 0.5 * long_term_trend
    
    return alpha_factor
```
