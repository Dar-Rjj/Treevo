import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum (5 days)
    df['short_term_momentum'] = df['close'].pct_change(periods=5)
    
    # Integrate Adaptive Medium-Term Momentum (10 days)
    df['medium_term_momentum'] = df['close'].pct_change(periods=10)
    
    # Incorporate Volume Momentum (5-day volume change)
    df['volume_momentum'] = df['volume'].diff(periods=5)
    
    # Incorporate Price Volatility (10-day close price standard deviation)
    df['price_volatility'] = df['close'].rolling(window=10).std()
    
    # Dynamic Weighting Based on Market Conditions
    # Calculate 30-Day Historical Performance
    df['30_day_perf'] = df['close'].pct_change(periods=30)
    
    # Calculate 30-Day Market Volatility
    df['30_day_volatility'] = df['close'].rolling(window=30).std()
    
    # Determine Weights Based on Performance and Volatility
    df['performance_weight'] = df['30_day_perf'] / df['30_day_volatility']
    
    # Normalize Weights
    performance_weight = (df['performance_weight'] - df['performance_weight'].mean()) / df['performance_weight'].std()
    
    # Combine all factors with dynamic weights
    df['alpha_factor'] = (
        df['cum_vwap_deviation'] * performance_weight +
        df['short_term_momentum'] * performance_weight +
        df['medium_term_momentum'] * performance_weight +
        df['volume_momentum'] * performance_weight +
        df['price_volatility'] * performance_weight
    )
    
    # Return the alpha factor
    return df['alpha_factor']
