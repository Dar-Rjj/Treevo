```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['Total Dollar Value'] = df['close'] * df['volume']
    daily_dollar_value = df['Total Dollar Value'].sum()
    daily_volume = df['volume'].sum()
    vwap = daily_dollar_value / daily_volume

    # Calculate VWAP Deviation
    df['VWAP Deviation'] = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    df['Cumulative VWAP Deviation'] = df['VWAP Deviation'].cumsum()

    # Calculate Price Momentum (5-day window)
    df['Price SMA 5'] = df['close'].rolling(window=5).mean()
    df['Price Momentum'] = df['close'] - df['Price SMA 5']

    # Calculate Volume Trend (5-day EMA)
    df['Volume EMA 5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['Volume Trend'] = df['volume'] - df['Volume EMA 5']

    # Integrate Factors
    # Weighted Sum of the Above Factors
    df['Alpha Factor'] = (
        0.3 * df['VWAP Deviation'] +
        0.2 * df['Cumulative VWAP Deviation'] +
        0.3 * df['Price Momentum'] +
        0.2 * df['Volume Trend']
    )

    return df['Alpha Factor']
```
