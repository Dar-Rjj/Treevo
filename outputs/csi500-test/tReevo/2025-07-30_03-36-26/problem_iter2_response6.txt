```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap_numerator = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap_numerator.sum() / df['volume'].sum()
    
    # Adjust Daily Price Change with VWAP
    adjusted_daily_price_change = daily_price_change / vwap
    
    # Incorporate Volume Impact
    volume_adjusted_price_trend = adjusted_daily_price_change * df['volume']
    
    # Final Volume-Adjusted Price Trend (VAPT)
    vapt = volume_adjusted_price_trend
    
    return vapt
```
