```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Moving Average
    short_term_days = 5
    short_term_ma = df['close'].rolling(window=short_term_days).mean()
    
    # Long-Term Moving Average
    long_term_days = 20
    long_term_ma = df['close'].rolling(window=long_term_days).mean()

    # Price Oscillator
    price_oscillator = short_term_ma - long_term_ma
    
    # Volume-Weighted Adjustment
    recent_volume_avg = df['volume'].rolling(window=short_term_days).mean()
    historical_volume_avg = df['volume'].rolling(window=long_term_days).mean()
    volume_ratio = recent_volume_avg / historical_volume_avg
    
    # Adjust Price Oscillator
    alpha_factor = price_oscillator * volume_ratio
    
    return alpha_factor
```
