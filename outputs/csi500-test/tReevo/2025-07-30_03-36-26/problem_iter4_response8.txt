```python
import pandas as pd

def heuristics_v2(df, window=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust for Volume
    volume_adjusted_return = intraday_return * df['volume']

    # Calculate Cumulative Volume-Adjusted Return
    cumulative_volume_adjusted_return = volume_adjusted_return.rolling(window=window).sum()

    # Incorporate Price Trend Intensity
    high_low_range = df['high'] - df['low']
    price_trend_intensity = high_low_range * cumulative_volume_adjusted_return

    # Final Volume-Adjusted Cumulative Momentum (VACM)
    vacm = cumulative_volume_adjusted_return * price_trend_intensity

    return vacm
```
