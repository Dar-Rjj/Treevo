```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].sum()
    df['total_dollar_value'] = df['close'] * df['volume']
    df['vwap'] = df['total_dollar_value'].cumsum() / df['volume'].cumsum()

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Short-Term Momentum (5 days)
    short_term_period = 5
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term_period).sum()
    df['cumulative_vwap_deviation'] += df['short_term_momentum']

    # Integrate Medium-Term Momentum (10 days)
    medium_term_period = 10
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term_period).sum()
    df['cumulative_vwap_deviation'] += df['medium_term_momentum']

    # Integrate Long-Term Momentum (20 days)
    long_term_period = 20
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term_period).sum()
    df['cumulative_vwap_deviation'] += df['long_term_momentum']

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['absolute_vwap_deviation'] = abs(df['close'] - df['vwap'])
    df['intraday_volatility'] = df['high_low_range'] + df['absolute_vwap_deviation']

    # Incorporate Short-Term Trend Strength (5-day SMA)
    df['5_day_sma'] = df['close'].rolling(window=short_term_period).mean()
    df['short_term_trend_strength'] = (df['close'] - df['5_day_sma']) * df['short_term_momentum']

    # Incorporate Medium-Term Trend Strength (10-day SMA)
    df['10_day_sma'] = df['close'].rolling(window=medium_term_period).mean()
    df['medium_term_trend_strength'] = (df['close'] - df['10_day_sma']) * df['medium_term_momentum']

    # Incorporate Long-Term Trend Strength (20-day SMA)
    df['20_day_sma'] = df['close'].rolling(window=long_term_period).mean()
    df['long_term_trend_strength'] = (df['close'] - df['20_day_sma']) * df['long_term_momentum']

    # Final Alpha Factor
    df['alpha_factor'] = (
        df['cumulative_vwap_deviation'] +
        df['intraday_volatility'] +
        df['short_term_trend_strength'] +
        df['medium_term_trend_strength'] +
        df['long_term_trend_strength']
    )

    return df['alpha_factor']
```
