import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily VWAP Calculation
    df['total_volume'] = df['volume'].rolling(window=1, min_periods=1).sum()
    df['total_dollar_value'] = (df['close'] * df['volume']).rolling(window=1, min_periods=1).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # VWAP Deviation from Close
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Price Momentum
    df['previous_close'] = df['close'].shift(1)
    df['momentum'] = df['close'] - df['previous_close']
    
    # Volatility
    df['volatility'] = df['close'].rolling(window=20, min_periods=1).std()
    
    # Volume-Weighted Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Cumulative Adjusted Return
    df['cumulative_adjusted_return'] = df['volume_weighted_return'].cumsum()
    
    # Alpha Factor: Combination of VWAP Deviation, Momentum, and Cumulative Adjusted Return
    df['alpha_factor'] = (df['vwap_deviation'] + df['momentum'] - df['cumulative_adjusted_return']) / df['volatility']
    
    return df['alpha_factor']
