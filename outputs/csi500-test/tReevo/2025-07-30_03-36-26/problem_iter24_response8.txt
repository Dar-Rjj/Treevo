```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume']
    total_dollar_value = df['volume'] * df['close']
    vwap = (total_dollar_value / total_volume).rolling(window=1, min_periods=1).mean()
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.rolling(window=1, min_periods=1).sum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum = vwap_deviation.rolling(window=5, min_periods=1).sum()
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum = vwap_deviation.rolling(window=10, min_periods=1).sum()
    
    # Integrate Long-Term Momentum (20 days)
    long_term_momentum = vwap_deviation.rolling(window=20, min_periods=1).sum()
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    absolute_vwap_deviation = abs(df['close'] - vwap)
    intraday_volatility = high_low_range + absolute_vwap_deviation
    
    # Final Alpha Factor
    alpha_factor = (
        cumulative_vwap_deviation
        + short_term_momentum
        + medium_term_momentum
        + long_term_momentum
        + intraday_volatility
    )
    
    return alpha_factor
```
