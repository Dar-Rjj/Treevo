```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['DollarValue'] = df['close'] * df['volume']
    df['TotalVolume'] = df['volume']
    df['VWAP'] = df['DollarValue'].rolling(window=1, min_periods=1).sum() / df['TotalVolume'].rolling(window=1, min_periods=1).sum()
    
    # Calculate VWAP Deviation
    df['VWAPDeviation'] = df['close'] - df['VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['CumulativeVWAPDeviation'] = df['VWAPDeviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum (5 days)
    short_term_momentum = df['close'].pct_change(5)
    df['ShortTermMomentumFactor'] = 0.6 * df['CumulativeVWAPDeviation'] + 0.4 * short_term_momentum
    
    # Integrate Adaptive Medium-Term Momentum (10 days)
    medium_term_momentum = df['close'].pct_change(10)
    df['MediumTermMomentumFactor'] = 0.7 * df['CumulativeVWAPDeviation'] + 0.3 * medium_term_momentum
    
    # Final Alpha Factor
    df['AlphaFactor'] = 0.5 * (df['ShortTermMomentumFactor'] + df['MediumTermMomentumFactor']) + 0.5 * df['CumulativeVWAPDeviation']
    
    return df['AlphaFactor']
```
