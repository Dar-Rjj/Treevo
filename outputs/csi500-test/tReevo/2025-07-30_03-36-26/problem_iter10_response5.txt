```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['dollar_value'] = df['close'] * df['volume']
    daily_vwap = df.groupby(level=0).apply(lambda x: x['dollar_value'].sum() / x['total_volume'].sum())
    df['vwap'] = daily_vwap

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df.groupby(level=0)['vwap_deviation'].cumsum()

    # Integrate Short-Term Momentum (5-day return)
    df['5_day_return'] = df['close'].pct_change(periods=5)
    df['short_term_momentum'] = df['5_day_return'] * df['cumulative_vwap_deviation']

    # Integrate Medium-Term Momentum (10-day return)
    df['10_day_return'] = df['close'].pct_change(periods=10)
    df['medium_term_momentum'] = df['10_day_return'] * df['cumulative_vwap_deviation']

    # Final Alpha Factor
    df['alpha_factor'] = df['cumulative_vwap_deviation'] + df['short_term_momentum'] + df['medium_term_momentum']

    return df['alpha_factor']
```
