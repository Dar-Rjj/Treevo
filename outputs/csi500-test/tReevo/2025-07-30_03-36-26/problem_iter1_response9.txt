```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Average Volume
    avg_volume = (df['volume'].shift(1) + df['volume'].shift(2)) / 2
    
    # Weight High-Low Spread by Average Volume
    weighted_high_low_spread = high_low_spread * avg_volume
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Add Close-to-Close Momentum to Weighted High-Low Spread
    factor_values = weighted_high_low_spread + close_to_close_return
    
    return factor_values
```
