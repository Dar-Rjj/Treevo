```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].rolling(window=1, min_periods=1).sum()
    df['total_dollar_value'] = (df['close'] * df['volume']).rolling(window=1, min_periods=1).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum_period = 5
    df['short_term_return'] = df['close'].pct_change(periods=short_term_momentum_period)
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum_period = 10
    df['medium_term_return'] = df['close'].pct_change(periods=medium_term_momentum_period)
    
    # Final Alpha Factor
    # Weighted Sum of Cumulative VWAP Deviation, Short-Term Momentum, and Medium-Term Momentum
    df['alpha_factor'] = (
        0.5 * df['cumulative_vwap_deviation'] + 
        0.3 * df['short_term_return'].fillna(0) + 
        0.2 * df['medium_term_return'].fillna(0)
    )
    
    return df['alpha_factor']
```
