```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Intraday Momentum Intensity
    df['intraday_range_percentage'] = (df['high'] - df['low']) / df['low']
    df['intraday_momentum_intensity'] = df['intraday_range_percentage'] * df['volume']

    # Combine Price Change with VWAP and Intraday Momentum
    df['vampt'] = df['daily_change'] * df['vwap'] * df['intraday_momentum_intensity']
    
    return df['vampt']
```
