```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Daily VWAP
    data['total_volume'] = data['volume'].sum()
    data['total_dollar_value'] = (data['close'] * data['volume']).sum()
    data['vwap'] = data['total_dollar_value'] / data['total_volume']
    
    # Calculate VWAP Deviation
    data['vwap_deviation'] = data['close'] - data['vwap']
    
    # Calculate Cumulative VWAP Deviation
    data['cum_vwap_deviation'] = data['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Weighted Momentum
    short_term_period = 5
    short_term_weights = np.arange(1, short_term_period + 1)
    data['short_term_momentum'] = data['vwap_deviation'].rolling(window=short_term_period).apply(
        lambda x: (x * short_term_weights).sum(), raw=True
    )
    data['cum_vwap_deviation'] += data['short_term_momentum']
    
    # Integrate Medium-Term Weighted Momentum
    medium_term_period = 10
    medium_term_weights = np.arange(1, medium_term_period + 1)
    data['medium_term_momentum'] = data['vwap_deviation'].rolling(window=medium_term_period).apply(
        lambda x: (x * medium_term_weights).sum(), raw=True
    )
    data['cum_vwap_deviation'] += data['medium_term_momentum']
    
    # Integrate Long-Term Weighted Momentum
    long_term_period = 20
    long_term_weights = np.arange(1, long_term_period + 1)
    data['long_term_momentum'] = data['vwap_deviation'].rolling(window=long_term_period).apply(
        lambda x: (x * long_term_weights).sum(), raw=True
    )
    data['cum_vwap_deviation'] += data['long_term_momentum']
    
    # Calculate Intraday Volatility
    data['high_low_range'] = data['high'] - data['low']
    data['abs_vwap_deviation'] = abs(data['close'] - data['vwap'])
    data['intraday_volatility'] = data['high_low_range'] + data['abs_vwap_deviation']
    
    # Final Alpha Factor
    data['alpha_factor'] = data['cum_vwap_deviation'] + data['intraday_volatility']
    
    return data['alpha_factor']
```
