import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].sum()
    df['total_dollar_value'] = (df['volume'] * df['close']).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum
    short_term_period = 5  # Example: 5-10 days based on market conditions
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term_period).sum()
    df['combined_cumulative_vwap_deviation'] = df['cumulative_vwap_deviation'] + df['short_term_momentum']
    
    # Integrate Adaptive Medium-Term Momentum
    medium_term_period = 10  # Example: 10-20 days based on market conditions
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term_period).sum()
    df['combined_cumulative_vwap_deviation'] += df['medium_term_momentum']
    
    # Integrate Adaptive Long-Term Momentum
    long_term_period = 20  # Example: 20-40 days based on market conditions
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term_period).sum()
    df['combined_cumulative_vwap_deviation'] += df['long_term_momentum']
    
    # Calculate VWAP Deviation Volatility
    vwap_volatility_period = 10  # Example: recent period
    df['vwap_deviation_volatility'] = df['vwap_deviation'].rolling(window=vwap_volatility_period).std()
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['absolute_vwap_deviation'] = df['vwap_deviation'].abs()
    df['intraday_volatility'] = df['high_low_range'] + df['absolute_vwap_deviation']
    
    # Calculate Adaptive Weighting Factors
    # Example: Using simple moving averages to determine market conditions
    sma_short = df['close'].rolling(window=5).mean()
    sma_long = df['close'].rolling(window=20).mean()
    market_trend = np.where(sma_short > sma_long, 1, -1)
    
    # Adjust Weights for Different Components
    weight_short_term = np.where(market_trend == 1, 0.4, 0.2)
    weight_medium_term = np.where(market_trend == 1, 0.3, 0.1)
    weight_long_term = np.where(market_trend == 1, 0.2, 0.4)
    weight_vwap_volatility = np.where(market_trend == 1, 0.05, 0.1)
    weight_intraday_volatility = np.where(market_trend == 1, 0.05, 0.1)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        weight_short_term * df['short_term_momentum'] +
        weight_medium_term * df['medium_term_momentum'] +
        weight_long_term * df['long_term_momentum'] +
        weight_vwap_volatility * df['vwap_deviation_volatility'] +
        weight_intraday_volatility * df['intraday_volatility']
    )
    
    return df['alpha_factor']
