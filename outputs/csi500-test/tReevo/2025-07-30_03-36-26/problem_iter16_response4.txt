```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['TotalVolume'] = df['volume']
    df['TotalDollarValue'] = df['volume'] * df['close']
    df['VWAP'] = df['TotalDollarValue'].rolling(window=1, min_periods=1).sum() / df['TotalVolume'].rolling(window=1, min_periods=1).sum()
    
    # Calculate VWAP Deviation
    df['VWAPDeviation'] = df['close'] - df['VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['CumulativeVWAPDeviation'] = df['VWAPDeviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum
    short_term_momentum_period = 5
    df['ShortTermMomentum'] = df['VWAPDeviation'].rolling(window=short_term_momentum_period, min_periods=1).sum()
    df['CombinedFactor'] = df['CumulativeVWAPDeviation'] + df['ShortTermMomentum']
    
    # Integrate Adaptive Medium-Term Momentum
    medium_term_momentum_period = 10
    df['MediumTermMomentum'] = df['VWAPDeviation'].rolling(window=medium_term_momentum_period, min_periods=1).sum()
    df['CombinedFactor'] += df['MediumTermMomentum']
    
    # Integrate Adaptive Long-Term Momentum
    long_term_momentum_period = 20
    df['LongTermMomentum'] = df['VWAPDeviation'].rolling(window=long_term_momentum_period, min_periods=1).sum()
    df['CombinedFactor'] += df['LongTermMomentum']
    
    # Calculate Intraday Volatility
    df['HighLowRange'] = df['high'] - df['low']
    df['AbsoluteVWAPDeviation'] = abs(df['close'] - df['VWAP'])
    df['IntradayVolatility'] = df['HighLowRange'] + df['AbsoluteVWAPDeviation']
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['CombinedFactor'] + df['IntradayVolatility']
    
    return df['AlphaFactor']
```
