import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['Total Volume'] = df['volume'].sum()
    df['Total Dollar Value'] = (df['close'] * df['volume']).sum()
    df['VWAP'] = df['Total Dollar Value'] / df['Total Volume']
    
    # Calculate VWAP Deviation
    df['VWAP Deviation'] = df['close'] - df['VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['Cumulative VWAP Deviation'] = df['VWAP Deviation'].cumsum()
    
    # Integrate Adaptive Short-Term Momentum
    short_term_momentum_period = 5  # Example: 5 days
    df['Short-Term Momentum'] = df['VWAP Deviation'].rolling(window=short_term_momentum_period).sum()
    df['Cumulative VWAP Deviation with Short-Term Momentum'] = df['Cumulative VWAP Deviation'] + df['Short-Term Momentum']
    
    # Integrate Adaptive Medium-Term Momentum
    medium_term_momentum_period = 10  # Example: 10 days
    df['Medium-Term Momentum'] = df['VWAP Deviation'].rolling(window=medium_term_momentum_period).sum()
    df['Cumulative VWAP Deviation with Medium-Term Momentum'] = df['Cumulative VWAP Deviation with Short-Term Momentum'] + df['Medium-Term Momentum']
    
    # Integrate Adaptive Long-Term Momentum
    long_term_momentum_period = 20  # Example: 20 days
    df['Long-Term Momentum'] = df['VWAP Deviation'].rolling(window=long_term_momentum_period).sum()
    df['Cumulative VWAP Deviation with Long-Term Momentum'] = df['Cumulative VWAP Deviation with Medium-Term Momentum'] + df['Long-Term Momentum']
    
    # Calculate Intraday Volatility
    df['High-Low Range'] = df['high'] - df['low']
    df['Absolute VWAP Deviation'] = (df['close'] - df['VWAP']).abs()
    df['Intraday Volatility'] = df['High-Low Range'] + df['Absolute VWAP Deviation']
    
    # Final Alpha Factor
    df['Final Alpha Factor'] = df['Cumulative VWAP Deviation with Long-Term Momentum'] + df['Intraday Volatility']
    
    return df['Final Alpha Factor']
