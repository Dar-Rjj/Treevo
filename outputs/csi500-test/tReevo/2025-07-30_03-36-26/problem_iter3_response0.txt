```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['DV_sum'] = df['volume'] * df['close']
    df['V_sum'] = df['volume']
    df['VWAP'] = df.groupby(df.index.date)['DV_sum'].transform('sum') / df.groupby(df.index.date)['V_sum'].transform('sum')
    
    # Calculate VWAP Deviation
    df['VWAP_Dev'] = df['close'] - df['VWAP']
    
    # Construct Cumulative VWAP Deviation
    df['Cum_VWAP_Dev'] = df.groupby(df.index.date)['VWAP_Dev'].cumsum()
    
    # Compute Price Momentum over 20 days
    df['Price_Momentum'] = df['close'].pct_change(20)
    
    # Evaluate Volatility over 20 days
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Volatility'] = df['Log_Returns'].rolling(window=20).std()
    
    # Assess Volume Trend over 20 days
    df['Volume_MA'] = df['volume'].rolling(window=20).mean()
    df['Rel_Volume'] = df['volume'] / df['Volume_MA']
    
    # Synthesize Alpha Factor
    df['Alpha_Signal'] = (df['Cum_VWAP_Dev'] + df['Price_Momentum']) / (df['Volatility'] + 1) * df['Rel_Volume']
    
    return df['Alpha_Signal']
```
