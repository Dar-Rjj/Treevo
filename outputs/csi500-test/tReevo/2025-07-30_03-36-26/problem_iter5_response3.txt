import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['Total Volume'] = df['volume']
    df['Dollar Value'] = df['close'] * df['volume']
    df['VWAP'] = df['Dollar Value'] / df['Total Volume']
    
    # Calculate VWAP Deviation
    df['VWAP Deviation'] = df['close'] - df['VWAP']
    
    # Calculate EMA of VWAP
    ema_vwap_period = 14
    df['EMA_VWAP'] = df['VWAP'].ewm(span=ema_vwap_period, adjust=False).mean()
    
    # Calculate Multi-Period Momentum
    short_ema_period = 5
    long_ema_period = 20
    df['Short_EMA'] = df['close'].ewm(span=short_ema_period, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=long_ema_period, adjust=False).mean()
    df['Momentum'] = df['Short_EMA'] - df['Long_EMA']
    
    # Integrate Volume Trend
    df['Volume Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Price Volatility
    volatility_period = 20
    df['Volatility'] = df['close'].rolling(window=volatility_period).std()
    
    # Combine All Factors into a Single Alpha Factor
    alpha_factor = (df['VWAP Deviation'] +
                    df['Momentum'] +
                    df['Volume Ratio'] +
                    df['Volatility'])
    
    return alpha_factor
