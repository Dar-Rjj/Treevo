```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Daily VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Short-Term Momentum (5 days)
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Integrate Medium-Term Momentum (10 days)
    df['medium_term_momentum'] = df['close'] - df['close'].shift(10)
    
    # Incorporate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['cum_vwap_deviation'] +
        df['short_term_momentum'] +
        df['medium_term_momentum'] +
        df['volume_change']
    )
    
    return df['alpha_factor']
```
