```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['close'] * df['volume']
    df['vwap'] = df['total_dollar_value'] / df['total_volume']

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Short-Term Weighted Momentum
    short_term_period = 5
    df['short_term_momentum'] = df['vwap_deviation'].rolling(window=short_term_period).mean()
    recent_volatility_short = df['close'].pct_change().rolling(window=10).std()
    adaptive_weights_short = 1 + (recent_volatility_short / recent_volatility_short.rolling(window=10).mean())
    df['weighted_short_term_momentum'] = df['short_term_momentum'] * adaptive_weights_short

    # Integrate Medium-Term Weighted Momentum
    medium_term_period = 10
    df['medium_term_momentum'] = df['vwap_deviation'].rolling(window=medium_term_period).mean()
    recent_volatility_medium = df['close'].pct_change().rolling(window=20).std()
    adaptive_weights_medium = 1 + (recent_volatility_medium / recent_volatility_medium.rolling(window=20).mean())
    df['weighted_medium_term_momentum'] = df['medium_term_momentum'] * adaptive_weights_medium

    # Integrate Long-Term Weighted Momentum
    long_term_period = 20
    df['long_term_momentum'] = df['vwap_deviation'].rolling(window=long_term_period).mean()
    recent_volatility_long = df['close'].pct_change().rolling(window=30).std()
    adaptive_weights_long = 1 + (recent_volatility_long / recent_volatility_long.rolling(window=30).mean())
    df['weighted_long_term_momentum'] = df['long_term_momentum'] * adaptive_weights_long

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['absolute_vwap_deviation'] = np.abs(df['close'] - df['vwap'])
    df['intraday_volatility'] = df['high_low_range'] + df['absolute_vwap_deviation']

    # Adjust for Market Regime
    df['ma_short'] = df['close'].rolling(window=50).mean()
    df['ma_long'] = df['close'].rolling(window=200).mean()
    df['market_regime'] = np.where(df['ma_short'] > df['ma_long'], 'Bull', 'Bear')

    # Adjust Weights Based on Market Regime
    df['combined_factor'] = df['cumulative_vwap_deviation'] + df['weighted_short_term_momentum'] + \
                            df['weighted_medium_term_momentum'] + df['weighted_long_term_momentum'] + \
                            df['intraday_volatility']
    
    bull_weight = 1.5
    bear_weight = 0.5
    df['adjusted_combined_factor'] = np.where(df['market_regime'] == 'Bull', df['combined_factor'] * bull_weight, 
                                              df['combined_factor'] * bear_weight)

    return df['adjusted_combined_factor']
```
