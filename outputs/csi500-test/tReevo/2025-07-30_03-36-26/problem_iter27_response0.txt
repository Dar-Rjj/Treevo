```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['DollarValue'] = df['volume'] * df['close']
    df['CumulativeVolume'] = df['volume'].rolling(window=1, min_periods=1).sum()
    df['CumulativeDollarValue'] = df['DollarValue'].rolling(window=1, min_periods=1).sum()
    df['VWAP'] = df['CumulativeDollarValue'] / df['CumulativeVolume']

    # Calculate VWAP Deviation
    df['VWAPDeviation'] = df['close'] - df['VWAP']

    # Calculate Cumulative VWAP Deviation
    df['CumulativeVWAPDeviation'] = df['VWAPDeviation'].cumsum()

    # Integrate Short-Term Momentum (5 days)
    df['ShortTermMomentum'] = df['VWAPDeviation'].rolling(window=5, min_periods=1).sum()

    # Integrate Medium-Term Momentum (10 days)
    df['MediumTermMomentum'] = df['VWAPDeviation'].rolling(window=10, min_periods=1).sum()

    # Integrate Long-Term Momentum (20 days)
    df['LongTermMomentum'] = df['VWAPDeviation'].rolling(window=20, min_periods=1).sum()

    # Calculate Intraday Volatility
    df['HighLowRange'] = df['high'] - df['low']
    df['AbsVWAPDeviation'] = df['VWAPDeviation'].abs()
    df['IntradayVolatility'] = df['HighLowRange'] + df['AbsVWAPDeviation']

    # Final Alpha Factor
    df['AlphaFactor'] = (
        df['CumulativeVWAPDeviation'] +
        df['ShortTermMomentum'] +
        df['MediumTermMomentum'] +
        df['LongTermMomentum'] +
        df['IntradayVolatility']
    )

    return df['AlphaFactor']
```
