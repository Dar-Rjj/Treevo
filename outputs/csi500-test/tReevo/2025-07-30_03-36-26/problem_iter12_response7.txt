```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['close'] * df['volume']
    df['vwap'] = df['total_dollar_value'].cumsum() / df['total_volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Integrate Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_range'] / df['open']
    
    # Integrate Short-Term Momentum (5 days)
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Integrate Medium-Term Momentum (10 days)
    df['medium_term_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Integrate Long-Term Momentum (20 days)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['cumulative_vwap_deviation'] +
        df['intraday_volatility'] +
        df['short_term_momentum'] +
        df['medium_term_momentum'] +
        df['long_term_momentum']
    )
    
    return df['alpha_factor']
```
