```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Intraday Volatility
    intraday_high_low_range = df['high'] - df['low']
    intraday_volatility = intraday_high_low_range / df['open']
    
    # Integrate Short-Term Momentum (5 days)
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Integrate Medium-Term Momentum (10 days)
    medium_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Final Alpha Factor
    alpha_factor = (
        cumulative_vwap_deviation 
        + intraday_volatility 
        + short_term_momentum 
        + medium_term_momentum
    )
    
    return alpha_factor
```
