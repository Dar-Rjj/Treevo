```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_value'] = df['close'] * df['volume']
    daily_vwap = df.groupby(level=0).apply(lambda x: x['total_value'].sum() / x['volume'].sum())
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - daily_vwap.reindex(df.index, level=0)

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df.groupby(level=0)['vwap_deviation'].cumsum()

    # Integrate Adaptive Short-Term Momentum (5 days)
    short_term_return = df['close'].pct_change(periods=5)
    short_term_momentum = df['cumulative_vwap_deviation'] * short_term_return

    # Integrate Adaptive Medium-Term Momentum (10 days)
    medium_term_return = df['close'].pct_change(periods=10)
    medium_term_momentum = df['cumulative_vwap_deviation'] * medium_term_return

    # Final Alpha Factor
    alpha_factor = (
        0.7 * df['cumulative_vwap_deviation'] +
        0.2 * short_term_momentum +
        0.1 * medium_term_momentum
    )
    
    return alpha_factor
```
