```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5d
    
    # Incorporate Close-to-Close Momentum
    close_return_5d = df['close'].pct_change(periods=5)
    weighted_high_low_close_momentum = weighted_high_low_spread + close_return_5d
    
    # Incorporate Volume Momentum
    volume_change_5d = df['volume'].pct_change(periods=5)
    weighted_high_low_close_vol_momentum = weighted_high_low_close_momentum * volume_change_5d
    
    # Incorporate 20-Day Price Trend
    moving_avg_20d = df['close'].rolling(window=20).mean()
    price_trend_20d = df['close'] - moving_avg_20d
    factor_with_price_trend = weighted_high_low_close_vol_momentum + price_trend_20d
    
    # Incorporate Multi-Period Momentum
    close_return_10d = df['close'].pct_change(periods=10)
    factor_with_multi_period_momentum = factor_with_price_trend + close_return_10d
    
    # Incorporate Volume Volatility
    volume_std_10d = df['volume'].rolling(window=10).std()
    factor_with_vol_volatility = factor_with_multi_period_momentum + volume_std_10d
    
    # Incorporate Price Volatility
    close_std_10d = df['close'].rolling(window=10).std()
    final_factor_without_weights = factor_with_vol_volatility + close_std_10d
    
    # Dynamic Weighting
    cum_returns_30d = (df['close'] / df['close'].shift(30)).apply(np.log)
    performance_weights = cum_returns_30d / cum_returns_30d.abs().sum()
    
    # Apply Weights to Final Alpha Factor
    final_alpha_factor = final_factor_without_weights * performance_weights
    
    return final_alpha_factor
```
