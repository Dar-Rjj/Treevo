import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).sum()
    vwap = total_dollar_value / total_volume

    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()

    # Integrate Intraday Momentum Intensity
    intraday_range_percentage = (df['high'] - df['low']) / df['low']
    volume_impact = df['volume'] * intraday_range_percentage
    intraday_momentum_intensity = volume_impact.sum() / total_volume
    adjusted_cumulative_vwap_deviation = cumulative_vwap_deviation * intraday_momentum_intensity

    # Incorporate Multi-Period Momentum
    five_day_return = df['close'].pct_change(5)
    ten_day_return = df['close'].pct_change(10)
    combined_returns = 0.6 * five_day_return + 0.4 * ten_day_return
    final_alpha_factor = adjusted_cumulative_vwap_deviation * combined_returns

    # Explore Alternative Volume-Weighted Metrics
    vw_open = (df['volume'] * df['open']).sum() / total_volume
    vw_high = (df['volume'] * df['high']).sum() / total_volume
    vw_low = (df['volume'] * df['low']).sum() / total_volume
    vw_close = (df['volume'] * df['close']).sum() / total_volume

    # Final Alpha Factor
    return final_alpha_factor
