```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['vwap_dev'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_dev'] = df['vwap_dev'].cumsum()
    
    # Integrate Short-Term Momentum
    short_term_momentum_period = 5
    df['short_term_mom'] = df['vwap_dev'].rolling(window=short_term_momentum_period).sum()
    df['cum_vwap_dev_with_short_mom'] = df['cum_vwap_dev'] + df['short_term_mom']
    
    # Integrate Medium-Term Momentum
    medium_term_momentum_period = 10
    df['medium_term_mom'] = df['vwap_dev'].rolling(window=medium_term_momentum_period).sum()
    df['cum_vwap_dev_with_medium_mom'] = df['cum_vwap_dev_with_short_mom'] + df['medium_term_mom']
    
    # Integrate Long-Term Momentum
    long_term_momentum_period = 20
    df['long_term_mom'] = df['vwap_dev'].rolling(window=long_term_momentum_period).sum()
    df['cum_vwap_dev_with_long_mom'] = df['cum_vwap_dev_with_medium_mom'] + df['long_term_mom']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['abs_vwap_dev'] = abs(df['vwap_dev'])
    df['intraday_volatility'] = df['high_low_range'] + df['abs_vwap_dev']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['cum_vwap_dev_with_long_mom'] + df['intraday_volatility']
    
    return df['alpha_factor']
```
