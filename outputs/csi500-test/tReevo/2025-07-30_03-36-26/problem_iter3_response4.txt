```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume'].sum()
    df['total_dollar_value'] = (df['close'] * df['volume']).sum()
    df['vwap'] = df['total_dollar_value'] / df['total_volume']
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].diff()
    
    # Calculate Volatility
    df['volatility'] = df['high'] - df['low']
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'].diff()
    
    # Combine Factors into Alpha Signal
    df['alpha_signal'] = (
        df['vwap_deviation'] + 
        df['cumulative_vwap_deviation'] + 
        df['price_momentum'] + 
        df['volatility'] + 
        df['volume_trend']
    )
    
    # Return the alpha signal as a Series
    return df['alpha_signal']
```
