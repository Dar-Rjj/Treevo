```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['VWAP'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP Deviation
    df['VWAP_Deviation'] = df['close'] - df['VWAP']
    
    # Calculate Cumulative VWAP Deviation
    df['Cumulative_VWAP_Deviation'] = df['VWAP_Deviation'].cumsum()
    
    # Integrate Dynamic Short-Term Momentum
    short_term_momentum_period = 5
    df['Short_Term_Momentum'] = np.log(df['close']).diff(short_term_momentum_period)
    
    # Integrate Dynamic Medium-Term Momentum
    medium_term_momentum_period = 10
    df['Medium_Term_Momentum'] = np.log(df['close']).diff(medium_term_momentum_period)
    
    # Final Alpha Factor
    # Weighted Sum of Cumulative VWAP Deviation, Short-Term Momentum, and Medium-Term Momentum
    weight_cumulative_vwap_deviation = 0.4
    weight_short_term_momentum = 0.3
    weight_medium_term_momentum = 0.3
    
    df['Alpha_Factor'] = (
        weight_cumulative_vwap_deviation * df['Cumulative_VWAP_Deviation'] +
        weight_short_term_momentum * df['Short_Term_Momentum'] +
        weight_medium_term_momentum * df['Medium_Term_Momentum']
    )
    
    return df['Alpha_Factor']
```
