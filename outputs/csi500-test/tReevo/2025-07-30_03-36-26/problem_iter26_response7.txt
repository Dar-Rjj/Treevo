```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume']
    total_dollar_value = df['volume'] * df['close']
    vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    vwap_deviation.name = 'vwap_deviation'
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    cumulative_vwap_deviation.name = 'cumulative_vwap_deviation'
    
    # Integrate Simplified Momentum
    short_term_momentum = vwap_deviation.rolling(window=5).mean()
    medium_term_momentum = vwap_deviation.rolling(window=10).mean()
    long_term_momentum = vwap_deviation.rolling(window=20).mean()
    
    # Combine Simplified Momenta with weights
    combined_momentum = (short_term_momentum * 0.4) + (medium_term_momentum * 0.3) + (long_term_momentum * 0.3)
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    abs_vwap_deviation = np.abs(df['close'] - vwap)
    intraday_volatility = np.sqrt((high_low_range ** 2) + (abs_vwap_deviation ** 2))
    
    # Final Alpha Factor
    final_alpha_factor = cumulative_vwap_deviation + combined_momentum + intraday_volatility
    final_alpha_factor.name = 'alpha_factor'
    
    return final_alpha_factor
```
