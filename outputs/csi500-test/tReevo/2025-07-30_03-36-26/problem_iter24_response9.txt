```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume

    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    df['cum_vwap_dev'] = vwap_deviation.cumsum()

    # Integrate Adaptive Short-Term Momentum
    recent_volatility = df['close'].rolling(window=20).std().iloc[-1]
    short_term_period = 5 if recent_volatility > recent_volatility.mean() else 10
    short_term_momentum = vwap_deviation.rolling(window=short_term_period).sum()
    df['cum_vwap_dev'] += short_term_momentum

    # Integrate Adaptive Medium-Term Momentum
    medium_term_period = 10 if recent_volatility > recent_volatility.mean() else 20
    medium_term_momentum = vwap_deviation.rolling(window=medium_term_period).sum()
    df['cum_vwap_dev'] += medium_term_momentum

    # Integrate Adaptive Long-Term Momentum
    long_term_period = 20 if recent_volatility > recent_volatility.mean() else 30
    long_term_momentum = vwap_deviation.rolling(window=long_term_period).sum()
    df['cum_vwap_dev'] += long_term_momentum

    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    abs_vwap_dev = (df['close'] - vwap).abs()
    intraday_volatility = high_low_range + abs_vwap_dev

    # Final Alpha Factor
    alpha_factor = df['cum_vwap_dev'] + intraday_volatility

    return alpha_factor
```
