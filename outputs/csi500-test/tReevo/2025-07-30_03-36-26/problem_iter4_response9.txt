```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Trend
    long_term_trend = df['close'] - df['close'].rolling(window=100).mean()
    
    # Calculate Medium-Term Price Momentum
    medium_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Assess Volume Trend
    volume_trend = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Evaluate Volume-Price Correlation
    volume_price_corr = df['close'].rolling(window=20).corr(df['volume'])
    
    # Calculate Synchronized Change
    combined_momentum = 0.7 * medium_term_momentum + 0.3 * short_term_momentum
    synchronized_change = combined_momentum * volume_trend
    
    # Adjust by Volume-Price Correlation
    synchronized_change = synchronized_change * (1 + volume_price_corr)
    
    # Filter for Strong Signals
    threshold = 0.05
    synchronized_change_filtered = synchronized_change.abs() > threshold
    synchronized_change = synchronized_change[synchronized_change_filtered]
    
    # Incorporate Long-Term Trend
    alpha_factor = 0.6 * synchronized_change + 0.4 * long_term_trend
    
    return alpha_factor
```
