import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['TP_Volume'] = df['Typical_Price'] * df['volume']
    cumulative_TP_Volume = df['TP_Volume'].cumsum()
    cumulative_Volume = df['volume'].cumsum()
    df['VWAP'] = cumulative_TP_Volume / cumulative_Volume
    
    # Combine Price Change with VWAP
    df['Price_Change_VWAP_Normalized'] = df['Daily_Price_Change'] / df['VWAP']
    df['VAPT'] = df['Price_Change_VWAP_Normalized'] * df['volume']
    
    return df['VAPT']
