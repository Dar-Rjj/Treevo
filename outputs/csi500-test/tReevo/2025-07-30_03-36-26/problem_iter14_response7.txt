import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume'].sum()
    total_dollar_value = (df['close'] * df['volume']).sum()
    vwap = total_dollar_value / total_volume

    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap

    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()

    # Integrate Dynamic Short-Term Momentum (5 days)
    short_term_momentum_period = 5
    short_term_momentum = vwap_deviation.rolling(window=short_term_momentum_period).sum()
    cumulative_vwap_deviation += short_term_momentum

    # Integrate Dynamic Medium-Term Momentum (10 days)
    medium_term_momentum_period = 10
    medium_term_momentum = vwap_deviation.rolling(window=medium_term_momentum_period).sum()
    cumulative_vwap_deviation += medium_term_momentum

    # Integrate Dynamic Long-Term Momentum (20 days)
    long_term_momentum_period = 20
    long_term_momentum = vwap_deviation.rolling(window=long_term_momentum_period).sum()
    cumulative_vwap_deviation += long_term_momentum

    # Calculate Refined Intraday Volatility
    high_low_range = df['high'] - df['low']
    absolute_vwap_deviation = abs(df['close'] - vwap)
    refined_intraday_volatility = high_low_range + absolute_vwap_deviation

    # Final Alpha Factor
    final_alpha_factor = cumulative_vwap_deviation + refined_intraday_volatility

    return final_alpha_factor
