```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['volume'] * df['close']
    df['vwap'] = df['total_dollar_value'].rolling(window=1).sum() / df['total_volume'].rolling(window=1).sum()
    
    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()
    
    # Calculate Exponential Moving Average (EMA) of VWAP
    ema_period = 12
    df['ema_vwap'] = df['vwap'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate Multi-Period Momentum
    momentum_periods = [5, 10, 20]
    for period in momentum_periods:
        df[f'momentum_{period}'] = df['close'].pct_change(period)
    df['multi_period_momentum'] = df[[f'momentum_{p}' for p in momentum_periods]].mean(axis=1)
    
    # Integrate Price Volatility
    volatility_period = 20
    df['price_volatility'] = df['close'].rolling(window=volatility_period).std()
    df['adjusted_momentum'] = df['multi_period_momentum'] / df['price_volatility']
    
    # Analyze Volume Trends
    df['volume_change'] = df['volume'].diff()
    volume_momentum_period = 10
    df['volume_momentum'] = df['volume'].pct_change(volume_momentum_period)
    df['combined_momentum'] = df['adjusted_momentum'] + df['volume_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['cumulative_vwap_deviation'] + 
                          (df['vwap'] - df['ema_vwap']) + 
                          df['multi_period_momentum'] + 
                          df['adjusted_momentum'] + 
                          df['combined_momentum'])
    
    return df['alpha_factor']
```
