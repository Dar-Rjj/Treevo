```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust for Volume
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Cumulative Volume-Adjusted Return over the last 5 days
    df['cumulative_volume_adjusted_return'] = df['volume_adjusted_intraday_return'].rolling(window=5).sum()
    
    return df['cumulative_volume_adjusted_return']
```
