```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    total_volume = df['volume']
    total_dollar_value = df['close'] * df['volume']
    vwap = total_dollar_value / total_volume
    
    # Calculate VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Calculate Cumulative VWAP Deviation
    cumulative_vwap_deviation = vwap_deviation.cumsum()
    
    # Integrate Short-Term Weighted Momentum (5 days)
    short_term_momentum_period = 5
    log_returns_short = np.log(df['close'] / df['close'].shift(1))
    exp_decay_factors_short = np.exp(-np.arange(short_term_momentum_period))
    weighted_momentum_short = log_returns_short.rolling(window=short_term_momentum_period).apply(lambda x: np.sum(x * exp_decay_factors_short), raw=True)
    
    # Integrate Medium-Term Weighted Momentum (10 days)
    medium_term_momentum_period = 10
    log_returns_medium = np.log(df['close'] / df['close'].shift(1))
    exp_decay_factors_medium = np.exp(-np.arange(medium_term_momentum_period))
    weighted_momentum_medium = log_returns_medium.rolling(window=medium_term_momentum_period).apply(lambda x: np.sum(x * exp_decay_factors_medium), raw=True)
    
    # Integrate Long-Term Weighted Momentum (20 days)
    long_term_momentum_period = 20
    log_returns_long = np.log(df['close'] / df['close'].shift(1))
    exp_decay_factors_long = np.exp(-np.arange(long_term_momentum_period))
    weighted_momentum_long = log_returns_long.rolling(window=long_term_momentum_period).apply(lambda x: np.sum(x * exp_decay_factors_long), raw=True)
    
    # Combine with Cumulative VWAP Deviation
    combined_factor = cumulative_vwap_deviation + weighted_momentum_short + weighted_momentum_medium + weighted_momentum_long
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    abs_vwap_deviation = np.abs(df['close'] - vwap)
    intraday_volatility = high_low_range + abs_vwap_deviation
    
    # Final Alpha Factor
    alpha_factor = combined_factor + intraday_volatility
    
    return alpha_factor
```
