```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Normalize Price Change by VWAP and multiply by Volume
    df['vapt'] = (df['daily_price_change'] / df['vwap']) * df['volume']
    
    # Incorporate Intraday Momentum
    df['intraday_range_pct'] = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume Impact
    df['volume_impact'] = df['intraday_range_pct'] * df['volume']
    
    # Incorporate Close Price Direction
    df['directional_bias'] = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Final VAPT Factor
    df['final_vapt'] = df['vapt'] + df['volume_impact'] + df['directional_bias']
    
    return df['final_vapt']
```
