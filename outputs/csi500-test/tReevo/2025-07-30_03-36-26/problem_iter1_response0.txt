```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume Impact
    volume_adjusted_range = intraday_range * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Incorporate Close Price Direction
    close_price_direction = (df['close'] - df['open']) / df['open']
    
    # Determine Directional Bias
    directional_bias = close_price_direction * volume_adjusted_range
    
    return directional_bias
```
