```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['VWAP'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()

    # Calculate VWAP Deviation
    df['VWAP_Deviation'] = df['close'] - df['VWAP']

    # Calculate Cumulative VWAP Deviation
    df['Cumulative_VWAP_Deviation'] = df['VWAP_Deviation'].cumsum()

    # Integrate Volatility
    df['True_Range'] = df['high'] - df['low']
    df['Volatility_20D_MA'] = df['True_Range'].rolling(window=20).mean()
    df['Normalized_VWAP_Deviation'] = df['VWAP_Deviation'] / df['Volatility_20D_MA']

    # Incorporate More Price Levels
    df['Open_Close_Spread'] = df['close'] - df['open']
    df['Open_Close_Spread_20D_MA'] = df['Open_Close_Spread'].rolling(window=20).mean()
    df['Normalized_Open_Close_Spread'] = df['Open_Close_Spread'] / df['Open_Close_Spread_20D_MA']

    # Combine Factors
    weight_normalized_vwap_deviation = 0.7
    weight_normalized_open_close_spread = 0.3
    df['Combined_Factor'] = (weight_normalized_vwap_deviation * df['Normalized_VWAP_Deviation'] +
                             weight_normalized_open_close_spread * df['Normalized_Open_Close_Spread'])

    # Apply Dynamic Thresholds
    combined_factor_20D_std = df['Combined_Factor'].rolling(window=20).std()
    combined_factor_20D_mean = df['Combined_Factor'].rolling(window=20).mean()
    upper_threshold = combined_factor_20D_mean + 1.5 * combined_factor_20D_std
    lower_threshold = combined_factor_20D_mean - 1.5 * combined_factor_20D_std

    df['Filtered_Combined_Factor'] = df['Combined_Factor']
    df.loc[df['Combined_Factor'] < lower_threshold, 'Filtered_Combined_Factor'] = 0
    df.loc[df['Combined_Factor'] > upper_threshold, 'Filtered_Combined_Factor'] = 0

    # Final Alpha Factor
    alpha_factor = df['Filtered_Combined_Factor']

    return alpha_factor
```
