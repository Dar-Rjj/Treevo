```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily VWAP
    df['total_volume'] = df['volume']
    df['total_dollar_value'] = df['volume'] * df['close']
    df['vwap'] = df['total_dollar_value'].rolling(window=1, min_periods=1).sum() / df['total_volume'].rolling(window=1, min_periods=1).sum()

    # Calculate VWAP Deviation
    df['vwap_deviation'] = df['close'] - df['vwap']

    # Calculate Cumulative VWAP Deviation
    df['cumulative_vwap_deviation'] = df['vwap_deviation'].cumsum()

    # Integrate Short-Term Momentum
    short_term_period = 5
    df['short_term_momentum'] = np.log(df['close']).diff(short_term_period)
    df['cumulative_vwap_deviation_with_short_momentum'] = df['cumulative_vwap_deviation'] + df['short_term_momentum']

    # Integrate Medium-Term Momentum
    medium_term_period = 10
    df['medium_term_momentum'] = np.log(df['close']).diff(medium_term_period)
    df['cumulative_vwap_deviation_with_medium_momentum'] = df['cumulative_vwap_deviation_with_short_momentum'] + df['medium_term_momentum']

    # Integrate Long-Term Momentum
    long_term_period = 20
    df['long_term_momentum'] = np.log(df['close']).diff(long_term_period)
    df['final_alpha_factor'] = df['cumulative_vwap_deviation_with_medium_momentum'] + df['long_term_momentum']

    return df['final_alpha_factor']
```
