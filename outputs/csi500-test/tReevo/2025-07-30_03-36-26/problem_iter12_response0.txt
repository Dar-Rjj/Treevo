```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * avg_volume_5d
    
    # Incorporate Close-to-Close Momentum
    close_return_5d = df['close'].pct_change(periods=5)
    close_momentum_factor = weighted_high_low_spread + close_return_5d
    
    # Incorporate Volume Momentum
    volume_change_5d = df['volume'].pct_change(periods=5)
    volume_momentum_factor = close_momentum_factor * volume_change_5d
    
    # Incorporate 20-Day Price Trend
    moving_avg_20d = df['close'].rolling(window=20).mean()
    price_trend_factor = df['close'] - moving_avg_20d
    final_alpha_factor = volume_momentum_factor + price_trend_factor
    
    # Incorporate Multi-Period Momentum
    close_return_10d = df['close'].pct_change(periods=10)
    multi_period_momentum_factor = final_alpha_factor + close_return_10d
    
    # Incorporate Volume Volatility
    volume_std_10d = df['volume'].rolling(window=10).std()
    volume_volatility_factor = multi_period_momentum_factor + volume_std_10d
    
    # Incorporate Price Volatility
    close_std_10d = df['close'].rolling(window=10).std()
    price_volatility_factor = volume_volatility_factor + close_std_10d
    final_alpha_factor = price_volatility_factor
    
    # Dynamic Weighting Based on Market Conditions
    historical_performance_30d = df['close'].pct_change(periods=30).cumsum()
    market_volatility_30d = df['close'].rolling(window=30).std()
    
    # Determine Weights Based on Performance
    performance_weight = (historical_performance_30d - historical_performance_30d.min()) / (historical_performance_30d.max() - historical_performance_30d.min())
    
    # Adjust Weights Based on Market Volatility
    volatility_weight = 1 - (market_volatility_30d - market_volatility_30d.min()) / (market_volatility_30d.max() - market_volatility_30d.min())
    
    # Apply Weights to Final Alpha Factor
    final_alpha_factor = final_alpha_factor * performance_weight * volatility_weight
    
    return final_alpha_factor
```
