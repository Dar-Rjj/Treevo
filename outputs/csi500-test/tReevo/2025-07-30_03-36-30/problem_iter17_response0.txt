import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10_day_return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Combine 5-Day and 10-Day Returns
    df['combined_return'] = (df['5_day_return'] * 0.7) + (df['10_day_return'] * 0.3)
    
    # Volume-Weighted Multi-Day Price Momentum
    df['vol_weighted_momentum'] = df['combined_return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_day_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20_day_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_in_amount'] = df['amount'] - df['amount'].shift(1)
    df['adjusted_momentum'] = df['vol_weighted_momentum'] + df['daily_change_in_amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['enhanced_volatility_index']
    
    return df['final_factor']
