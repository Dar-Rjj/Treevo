```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Combine 3-Day and 5-Day Returns using a weighted average
    df['weighted_return'] = df['3d_return'] * 0.6 + df['5d_return'] * 0.4
    
    # Multiply by Volume of day t to get Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']

    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average the Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation: Subtract Open-High-Low Volatility Index from Volume-Adjusted Multi-Day Price Momentum
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['ohl_volatility_index']
    
    return df['alpha_factor']
```
