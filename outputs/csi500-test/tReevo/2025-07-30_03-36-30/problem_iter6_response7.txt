```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['log_return'] * df['volume']
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Long-Term Log Return
    df['long_term_log_return'] = np.log(df['close']) - np.log(df['close'].shift(20))
    
    # Integrate Long-Term Trend
    df['integrated_momentum'] = df['volume_adjusted_momentum'] + df['long_term_log_return']
    
    # True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df['true_range'].combine_first(abs(df['high'] - df['close'].shift(1)))
    df['true_range'] = df['true_range'].combine_first(abs(df['low'] - df['close'].shift(1)))
    
    # 14-Period Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Normalize with ATR
    df['normalized_momentum'] = df['integrated_momentum'] / df['atr_14']
    
    # Final Factor Calculation
    alpha_factor = df['normalized_momentum']
    
    return alpha_factor
```
