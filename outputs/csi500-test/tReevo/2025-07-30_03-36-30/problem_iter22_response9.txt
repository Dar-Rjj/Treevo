```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10d_return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Apply Decay Factor to Returns
    df['5d_return_decayed'] = df['5d_return'] * 0.8
    df['10d_return_decayed'] = df['10d_return'] * 0.7
    
    # Combine 5-Day and 10-Day Returns
    df['combined_return'] = (df['5d_return_decayed'] * 0.6) + (df['10d_return_decayed'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['20d_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20d_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20d_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20d_high_low_range'] + df['20d_open_close_range']) / df['20d_mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20d_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20d_volume_ema']
    
    # Adjust for Market Capitalization
    df['market_cap'] = df['close'] * df['amount']
    df['20d_avg_market_cap'] = df['market_cap'].rolling(window=20).mean()
    
    # Final Factor Calculation
    df['final_factor'] = ((df['adjusted_momentum_with_sentiment'] - df['enhanced_volatility_index']) * 
                          df['normalized_volume'] / df['20d_avg_market_cap'])
    
    return df['final_factor']
```
