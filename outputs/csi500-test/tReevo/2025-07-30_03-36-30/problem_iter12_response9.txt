```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Smooth Returns with EMA
    df['3d_ema'] = df['3d_return'].ewm(span=2).mean()
    df['5d_ema'] = df['5d_return'].ewm(span=4).mean()
    
    # Combine 3-Day and 5-Day Smoothed Returns
    df['combined_returns'] = 0.6 * df['3d_ema'] + 0.4 * df['5d_ema']
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_returns'] * df['volume']

    # Calculate 5-Day High-Low Range
    df['5d_high_low_range'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate 5-Day Open-Close Range
    df['5d_open_close_range'] = df['open'].rolling(window=5).max() - df['close'].rolling(window=5).min()
    
    # 5-Day Mean Price
    df['5d_mean_price'] = (df['open'].rolling(window=5).sum() + df['close'].rolling(window=5).sum()) / 10
    
    # Enhanced Open-High-Low Volatility Index
    df['volatility_index'] = (df['5d_high_low_range'] + df['5d_open_close_range']) / df['5d_mean_price']
    
    # Calculate 10-Day Volume EMA
    df['10d_volume_ema'] = df['volume'].ewm(span=10).mean()
    
    # Normalize Volume of day t
    df['volume_trend'] = df['volume'] / df['10d_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['volatility_index'] * df['volume_trend']
    
    return df['final_factor'].dropna()
```
