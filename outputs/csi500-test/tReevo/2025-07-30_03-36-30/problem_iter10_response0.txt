```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5d_return'] = (df['close'].shift(-4) - df['close']) / df['close']

    # Dynamic Weighting of 3-Day and 5-Day Returns
    df['3d_weight'] = df['volume'].shift(-1) / df['volume'].shift(-2)
    df['5d_weight'] = df['volume'].shift(-1) / df['volume'].shift(-4)

    df['3d_vol_adj_return'] = df['3d_return'] * df['3d_weight']
    df['5d_vol_adj_return'] = df['5d_return'] * df['5d_weight']

    # Volume-Adjusted Multi-Day Price Momentum
    df['vol_adj_momentum'] = (df['3d_vol_adj_return'] + df['5d_vol_adj_return']) * df['volume']

    # Calculate Enhanced Volatility Index
    df['high_low_spread'] = (df['high'] - df['low']) / ((df['open'] + df['close']) / 2)
    df['open_close_spread'] = (df['open'] - df['close']).abs() / ((df['open'] + df['close']) / 2)
    df['combined_spreads'] = (df['high_low_spread'] + df['open_close_spread']) / 2

    # 10-Day Historical Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['10d_volatility'] = df['log_returns'].rolling(window=10).std() * np.sqrt(252)

    # Combine Volatility Measures
    df['enhanced_volatility_index'] = (df['combined_spreads'] + df['10d_volatility']) / 2

    # Final Factor Calculation
    df['final_factor'] = df['vol_adj_momentum'] - df['enhanced_volatility_index']

    return df['final_factor'].dropna()
```
