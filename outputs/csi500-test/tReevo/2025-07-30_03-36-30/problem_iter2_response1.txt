```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['open'] - df['close']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Price Momentum
    momentum = df['close'].pct_change()
    
    # Combine Volatility and Momentum
    combined_factor = momentum - volatility_indicator
    
    # Volume-Weight the Combined Factor
    volume_weighted_factor = combined_factor * df['volume']
    
    # Incorporate Absolute Price Differences
    # Calculate Open to Previous Close Absolute Difference
    open_to_prev_close_diff = (df['open'] - df['close'].shift(1)).abs()
    
    # Calculate Close to Previous Close Absolute Difference
    close_to_prev_close_diff = (df['close'] - df['close'].shift(1)).abs()
    
    # Sum Absolute Differences for Enhanced Volatility
    enhanced_volatility = open_to_prev_close_diff + close_to_prev_close_diff
    
    # Adjust Final Factor with Enhanced Volatility
    final_factor = volume_weighted_factor - enhanced_volatility
    
    return final_factor
```
