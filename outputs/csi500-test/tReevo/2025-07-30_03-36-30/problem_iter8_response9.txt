```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'] / df['close'].shift(2)) - 1
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Combine 3-Day and 5-Day Returns
    df['weighted_return'] = df['3_day_return'] * 0.6 + df['5_day_return'] * 0.4
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Integrate Long-Term Trend
    df['20_day_log_return'] = np.log(df['close'] / df['close'].shift(20))
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] + df['20_day_log_return'] - df['volatility_index']
    
    return df['final_factor'].dropna()
```
