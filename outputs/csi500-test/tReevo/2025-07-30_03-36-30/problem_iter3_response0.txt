```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Volume-Adjusted Price Momentum
    df['Volume_Adjusted_Return'] = df['Daily_Return'] * df['volume']
    df['Momentum_21EMA'] = df['Volume_Adjusted_Return'].ewm(span=21, adjust=False).mean()
    
    # Calculate High-Low Spread and Open-Close Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = np.abs(df['open'] - df['close'])
    
    # Combine Spreads to get Volatility Index
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    df['Volatility_21EMA'] = df['Volatility_Index'].ewm(span=21, adjust=False).mean()
    
    # Integrate Long-Term Trend
    df['50SMA'] = df['close'].rolling(window=50).mean()
    df['Long_Term_Trend'] = df['close'] - df['50SMA']
    
    # Normalize the components
    df['Normalized_Momentum'] = df['Volume_Adjusted_Return'] / df['Momentum_21EMA']
    df['Normalized_Volatility'] = df['Volatility_Index'] / df['Volatility_21EMA']
    df['Normalized_Trend'] = df['Long_Term_Trend'] / df['50SMA']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Normalized_Momentum'] + df['Normalized_Volatility'] + df['Normalized_Trend']
    
    return df['Alpha_Factor']
```
