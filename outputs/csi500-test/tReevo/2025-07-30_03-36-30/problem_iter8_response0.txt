```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Exponentially Smooth 3-Day and 5-Day Returns
    df['smoothed_3d_return'] = df['3d_return'].ewm(alpha=0.6, adjust=False).mean()
    df['smoothed_5d_return'] = df['5d_return'].ewm(alpha=0.4, adjust=False).mean()
    
    # Combine Exponentially Smoothed 3-Day and 5-Day Returns
    df['combined_smoothed_return'] = (df['smoothed_3d_return'] * 0.6) + (df['smoothed_5d_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_smoothed_return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['average_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = df['average_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['factor_value'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['factor_value']
```
