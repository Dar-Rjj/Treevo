```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, and 10-Day Rolling Returns
    df['3D_Return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Weighted Average of 3-Day, 5-Day, and 10-Day Returns
    df['Combined_Return'] = 0.3 * df['3D_Return'] + 0.4 * df['5D_Return'] + 0.3 * df['10D_Return']
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['Open_Close_Spread'] = df['open'] - df['close']
    
    # Calculate Mean Price
    df['Mean_Price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    
    # Calculate High-Mean and Low-Mean Spreads
    df['High_Mean_Spread'] = df['high'] - df['Mean_Price']
    df['Low_Mean_Spread'] = df['Mean_Price'] - df['low']
    
    # Combine Spreads
    df['Enhanced_Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread'] + 
                                       df['High_Mean_Spread'] + df['Low_Mean_Spread']) / df['Mean_Price']
    
    # Calculate Daily Change in Amount and Volume
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Normalize Daily Change in Amount and Volume
    df['Normalized_Amount_Change'] = df['Amount_Change'] / df['amount']
    df['Normalized_Volume_Change'] = df['Volume_Change'] / df['volume']
    
    # Add Normalized Changes to Volume-Adjusted Multi-Day Price Momentum
    df['Market_Sentiment_Adjusted_Momentum'] = (df['Volume_Adjusted_Momentum'] + 
                                                df['Normalized_Amount_Change'] + 
                                                df['Normalized_Volume_Change'])
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Market_Sentiment_Adjusted_Momentum'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor']
```
