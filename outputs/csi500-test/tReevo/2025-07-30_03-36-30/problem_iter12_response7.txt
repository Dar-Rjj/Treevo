```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) / df['close']) - 1
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Dynamic Weighting of 3-Day and 5-Day Returns
    df['3_day_weight'] = df['volume'].shift(-1) / (df['volume'].shift(-1) + df['volume'].shift(-2))
    df['5_day_weight'] = df['volume'].shift(-1) / (df['volume'].shift(-1) + df['volume'].shift(-4))
    
    df['weighted_3_day_return'] = df['3_day_return'] * df['3_day_weight']
    df['weighted_5_day_return'] = df['5_day_return'] * df['5_day_weight']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = (df['weighted_3_day_return'] + df['weighted_5_day_return']) * df['volume']
    
    # Calculate Enhanced Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = np.abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['combined_spread'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # 10-Day Historical Volatility
    df['log_returns'] = np.log(df['close']).diff()
    df['10_day_volatility'] = df['log_returns'].rolling(window=10).std() * np.sqrt(252)
    
    # Average 10-Day Historical Volatility and Combined Spreads
    df['enhanced_volatility_index'] = (df['combined_spread'] + df['10_day_volatility']) / 2
    
    # Calculate Liquidity Measure
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['recent_volume_change'] = (df['volume'] - df['10_day_avg_volume']) / df['10_day_avg_volume']
    
    # Final Factor Calculation
    df['alpha_factor'] = (df['volume_adjusted_momentum'] - df['enhanced_volatility_index']) * df['recent_volume_change']
    
    return df['alpha_factor']
```
