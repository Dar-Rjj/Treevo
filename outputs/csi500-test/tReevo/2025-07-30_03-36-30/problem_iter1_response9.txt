```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum (5-Day Moving Average of Close price)
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Momentum (20-Day Moving Average of Close price)
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Difference
    momentum_difference = long_term_momentum - short_term_momentum
    
    # Calculate Volume Trend (10-Day Moving Average of Volume)
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Multiply Momentum Difference by Volume Trend to get the final factor
    factor_values = momentum_difference * volume_trend
    
    return factor_values
```
