```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Apply decay to returns
    df['5D_Decayed_Return'] = df['5D_Return'] * 0.8
    df['10D_Decayed_Return'] = df['10D_Return'] * 0.7
    
    # Weighted average of 5-day and 10-day returns
    df['Weighted_Returns'] = (df['5D_Decayed_Return'] * 0.6) + (df['10D_Decayed_Return'] * 0.4)
    
    # Multiply by volume
    df['Volume_Adjusted_Momentum'] = df['Weighted_Returns'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['Mean_Price_20D'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['Mean_Price_20D']

    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']

    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']

    # Adjust for Market Capitalization
    df['20D_Market_Cap'] = df['close'] * df['amount']
    df['Final_Factor'] = (df['Adjusted_Momentum'] - df['Enhanced_Volatility_Index']) * df['Normalized_Volume'] / df['20D_Market_Cap']

    return df['Final_Factor']
```
