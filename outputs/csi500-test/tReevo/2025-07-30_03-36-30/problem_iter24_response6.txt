```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 10-Day and 20-Day Rolling Returns
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    df['20D_Return'] = (df['close'].shift(-19) / df['close']) - 1
    
    # Combine 10-Day and 20-Day Returns using weighted average
    df['Combined_Return'] = 0.6 * df['10D_Return'] + 0.4 * df['20D_Return']
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).mean() - df['close'].rolling(window=20).mean()
    
    # Calculate 20-Day Mean Price
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    
    # Combine 20-Day Ranges and normalize by 20-Day Mean Price
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Calculate Daily Change in Amount
    df['Daily_Change_Amount'] = df['amount'].diff().fillna(0)
    
    # Normalize Daily Change in Amount
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    
    # Add to Volume-Adjusted Long-Term Price Momentum
    df['Momentum_With_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']
    
    # Calculate 20-Day Volume EMA
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Momentum_With_Sentiment'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor']
```
