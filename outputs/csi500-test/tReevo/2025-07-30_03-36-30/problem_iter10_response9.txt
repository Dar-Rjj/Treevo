import pandas as pd

def heuristics_v2(df):
    # Compute 2-Day, 3-Day, and 5-Day Rolling Returns
    df['2_day_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    df['3_day_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']

    # Weighted Average of 2-Day, 3-Day, and 5-Day Returns
    df['weighted_average_return'] = (0.3 * df['2_day_return'] + 0.4 * df['3_day_return'] + 0.3 * df['5_day_return'])

    # Volume Trend
    df['volume_trend'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_average_return'] * df['volume_trend']

    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']

    # Average High-Low and Open-Close Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2

    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2

    # Enhanced Open-High-Low Volatility Index
    df['volatility_index'] = df['avg_spread'] / df['mean_price']

    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']

    # Return the final factor as a Series
    return df['final_factor'].dropna()
