```python
import pandas as pd

def heuristics_v2(df, lookback_window=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['close'].shift(1)) / intraday_range
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=lookback_window).mean()
    volume_ratio = df['volume'] / avg_volume
    adjusted_momentum = intraday_momentum * volume_ratio
    
    return adjusted_momentum
```
