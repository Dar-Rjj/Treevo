```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Weighted Average of 5-Day and 10-Day Returns
    df['Weighted_Return'] = df['5D_Return'] * 0.7 + df['10D_Return'] * 0.3
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()).abs()
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).mean() / 2
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum'] + df['Daily_Change_in_Amount']) - df['Enhanced_Volatility_Index'] * df['Normalized_Volume']
    
    return df['Final_Factor']
```
