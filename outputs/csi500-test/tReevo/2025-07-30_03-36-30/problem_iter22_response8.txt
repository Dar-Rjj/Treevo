```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute rolling returns
    df['3d_return'] = df['close'].pct_change(3)
    df['5d_return'] = df['close'].pct_change(5)
    df['10d_return'] = df['close'].pct_change(10)
    df['20d_return'] = df['close'].pct_change(20)
    df['60d_return'] = df['close'].pct_change(60)
    
    # Weighted average of the returns
    df['weighted_return'] = (df['3d_return'] * 0.1 + 
                             df['5d_return'] * 0.2 + 
                             df['10d_return'] * 0.2 + 
                             df['20d_return'] * 0.3 + 
                             df['60d_return'] * 0.2)
    
    # Volume-Adjusted Multi-Timeframe Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Advanced Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = np.abs(df['open'] - df['close'])
    df['mean_price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    df['high_mean_spread'] = df['high'] - df['mean_price']
    df['low_mean_spread'] = df['mean_price'] - df['low']
    
    # Rolling standard deviations
    df['5d_std'] = df['close'].rolling(window=5).std()
    df['10d_std'] = df['close'].rolling(window=10).std()
    
    # Combine spreads and standard deviations
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread'] + 
                              df['high_mean_spread'] + df['low_mean_spread'] + 
                              df['5d_std'] + df['10d_std']) / df['mean_price']
    
    # Comprehensive Sentiment Indicators
    df['daily_change_amount'] = df['amount'].diff()
    df['daily_change_volume'] = df['volume'].diff()
    df['daily_change_close'] = df['close'].diff()
    
    # Normalize daily changes
    df['normalized_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    df['normalized_daily_change_volume'] = df['daily_change_volume'] / df['volume']
    df['normalized_daily_change_close'] = df['daily_change_close'] / df['close']
    
    # Add normalized changes to volume-adjusted momentum
    df['adjusted_momentum'] = (df['volume_adjusted_momentum'] + 
                               df['normalized_daily_change_amount'] + 
                               df['normalized_daily_change_volume'] + 
                               df['normalized_daily_change_close'])
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['volatility_index']
    
    return df['final_factor'].dropna()
```
