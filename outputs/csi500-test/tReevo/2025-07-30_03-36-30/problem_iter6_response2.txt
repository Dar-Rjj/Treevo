```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = df['close'].pct_change(periods=3)
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = df['close'].pct_change(periods=5)
    
    # Combine 3-Day and 5-Day Returns (Weighted Average)
    df['combined_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Combine Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Calculate Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Compute 20-Day Moving Average of Volume
    df['20_day_vol_ma'] = df['volume'].rolling(window=20).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['volume'] / df['20_day_vol_ma']
    
    # Final Factor Calculation
    df['final_factor'] = (df['volume_adjusted_momentum'] - df['volatility_index']) * df['volume_ratio']
    
    # Drop intermediate columns if needed
    df.drop(columns=['3_day_return', '5_day_return', 'combined_return', 'volume_adjusted_momentum',
                     'high_low_spread', 'open_close_spread', 'avg_spread', 'mean_price', 
                     'volatility_index', '20_day_vol_ma', 'volume_ratio'], inplace=True)
    
    return df['final_factor']
```
