import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    df['Combined_Return'] = (df['5D_Return'] * 0.6) + (df['10D_Return'] * 0.4)
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High'] = df['high'].rolling(window=20).max()
    df['20D_Low'] = df['low'].rolling(window=20).min()
    df['20D_Open'] = df['open'].rolling(window=20).mean()
    df['20D_Close'] = df['close'].rolling(window=20).mean()
    df['20D_Range'] = (df['20D_High'] - df['20D_Low']) + (df['20D_Open'] - df['20D_Close'])
    df['20D_Mean_Price'] = (df['20D_Open'] + df['20D_Close']) / 2
    df['Enhanced_Volatility_Index'] = df['20D_Range'] / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum'] - df['Enhanced_Volatility_Index'] 
                          + df['Daily_Change_in_Amount'] + df['Normalized_Volume'])
    
    return df['Final_Factor']
