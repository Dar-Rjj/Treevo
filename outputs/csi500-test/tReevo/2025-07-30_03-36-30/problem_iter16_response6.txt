```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-Day and 20-Day Rolling Returns
    df['10_day_return'] = df['close'].pct_change(10)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Combine 10-Day and 20-Day Returns
    df['combined_return'] = (df['10_day_return'] * 0.6) + (df['20_day_return'] * 0.4)
    
    # Volume-Adjusted Long-Term Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df['true_range'].fillna(method='ffill')
    
    # Combine Spreads and True Range
    df['mean_price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    df['refined_volatility'] = (df['high_low_spread'] + df['open_close_spread'] + df['true_range']) / df['mean_price']
    
    # Integrate Dynamic Market Sentiment
    df['5_day_change_amount'] = (df['amount'] - df['amount'].shift(5)) / df['amount']
    df['5_day_change_volume'] = (df['volume'] - df['volume'].shift(5)) / df['volume']
    df['dynamic_market_sentiment'] = (df['5_day_change_amount'] + df['5_day_change_volume']) / 2
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['refined_volatility'] + df['dynamic_market_sentiment']
    
    return df['final_factor']
```
