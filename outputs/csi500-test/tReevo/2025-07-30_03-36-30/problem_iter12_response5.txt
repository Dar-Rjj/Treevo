```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    
    # Calculate 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Combine 3-Day and 5-Day Returns
    df['weighted_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Final Factor Calculation
    df['factor_value'] = df['volume_adjusted_momentum'] - df['intraday_volatility']
    
    return df['factor_value']
```
