```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Level Momentum
    df['3D_Return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Dynamic Weighted Average of 3-Day, 5-Day, and 10-Day Returns
    df['3D_Weight'] = df['volume'] / (df['volume'] + df['volume'].shift(1) + df['volume'].shift(2))
    df['5D_Weight'] = df['volume'] / (df['volume'] + df['volume'].shift(1) + df['volume'].shift(2) + df['volume'].shift(3) + df['volume'].shift(4))
    df['10D_Weight'] = df['volume'] / (df['volume'] + df['volume'].shift(1) + df['volume'].shift(2) + df['volume'].shift(3) + df['volume'].shift(4) + df['volume'].shift(5) + df['volume'].shift(6) + df['volume'].shift(7) + df['volume'].shift(8) + df['volume'].shift(9))
    
    df['Volume_Adjusted_Momentum'] = (df['3D_Return'] * df['3D_Weight'] + 
                                      df['5D_Return'] * df['5D_Weight'] + 
                                      df['10D_Return'] * df['10D_Weight']) * df['volume']

    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']

    # Calculate Liquidity Measure
    df['10D_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    df['Liquidity_Measure'] = df['volume'] / df['10D_Avg_Volume']

    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum'] - df['Volatility_Index']) * df['Liquidity_Measure']

    return df['Final_Factor']
```
