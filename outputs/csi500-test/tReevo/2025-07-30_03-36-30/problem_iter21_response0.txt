```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5DayReturn'] = (df['close'].shift(-4) / df['close']) - 1
    df['10DayReturn'] = (df['close'].shift(-9) / df['close']) - 1
    df['WeightedReturns'] = 0.7 * df['5DayReturn'] + 0.3 * df['10DayReturn']
    df['VolumeAdjustedMomentum'] = df['WeightedReturns'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20DayHigh'] = df['high'].rolling(window=20).max()
    df['20DayLow'] = df['low'].rolling(window=20).min()
    df['20DayHighLowRange'] = df['20DayHigh'] - df['20DayLow']
    
    df['20DayOpenCloseRange'] = (df['open'] - df['close']).rolling(window=20).sum()
    df['20DayMeanPrice'] = (df['open'] + df['close']).rolling(window=20).mean() / 2
    
    df['EnhancedVolatilityIndex'] = (df['20DayHighLowRange'] + abs(df['20DayOpenCloseRange'])) / df['20DayMeanPrice']

    # Integrate Market Sentiment
    df['DailyChangeInAmount'] = df['amount'] - df['amount'].shift(1)
    df['NormalizedDailyChangeInAmount'] = df['DailyChangeInAmount'] / df['amount']
    df['VolumeAdjustedMomentumWithSentiment'] = df['VolumeAdjustedMomentum'] + df['NormalizedDailyChangeInAmount']

    # Incorporate Liquidity and Trend Strength
    df['20DayVolumeEMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['NormalizedVolume'] = df['volume'] / df['20DayVolumeEMA']
    df['FinalFactor'] = df['VolumeAdjustedMomentumWithSentiment'] - df['EnhancedVolatilityIndex'] * df['NormalizedVolume']

    return df['FinalFactor'].dropna()
```
