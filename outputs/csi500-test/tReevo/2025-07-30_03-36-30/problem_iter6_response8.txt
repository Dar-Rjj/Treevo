```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['open'] - df['close']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Price Momentum
    lookback_period = 20
    price_momentum = df['close'] - df['close'].shift(lookback_period)
    
    # Adjust Momentum by Volatility
    adjusted_momentum = price_momentum - volatility_indicator
    
    # Calculate Relative Strength
    avg_close = df['close'].rolling(window=lookback_period).mean()
    relative_strength = df['close'] / avg_close
    
    # Volume-Weight the Adjusted Momentum
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_ratio = df['volume'] / avg_volume
    volume_weighted_momentum = adjusted_momentum * volume_ratio
    
    # Adjust Final Alpha Factor by Relative Strength
    final_alpha_factor = volume_weighted_momentum * relative_strength
    
    return final_alpha_factor
```
