```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3_day_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns with Weights
    df['weighted_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average High-Low and Open-Close Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['final_factor']
```
