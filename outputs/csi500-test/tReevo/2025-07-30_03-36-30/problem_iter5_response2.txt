```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3D_Return'] = df['close'].pct_change(3)
    df['5D_Return'] = df['close'].pct_change(5)
    
    # Weighted Average of 3-Day and 5-Day Returns
    df['Weighted_Avg_Return'] = (df['3D_Return'] * 0.6) + (df['5D_Return'] * 0.4)
    
    # Adjust by Volume
    df['Volume_Adjusted_Momentum'] = df['Weighted_Avg_Return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    
    # Average High-Low and Open-Close Spreads
    df['Avg_Spread'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / 2
    
    # Divide by Mean Price
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = df['Avg_Spread'] / df['Mean_Price']
    
    # Final Factor Calculation
    df['Alpha_Factor'] = df['Volume_Adjusted_Momentum'] - df['Volatility_Index']
    
    return df['Alpha_Factor']
```
