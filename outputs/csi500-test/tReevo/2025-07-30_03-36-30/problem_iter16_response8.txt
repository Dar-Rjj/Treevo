import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Rolling Return
    df['5_day_return'] = df['close'].pct_change(5)
    
    # Calculate 10-Day Rolling Return
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Combine 5-Day and 10-Day Returns
    df['weighted_return'] = (df['5_day_return'] * 0.6) + (df['10_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Calculate Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Combine Spreads to get Volatility Index
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Calculate Daily Change in Amount
    df['daily_change_amount'] = df['amount'].diff()
    
    # Adjust Daily Change in Amount
    df['adjusted_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['volatility_index'] + df['adjusted_daily_change_amount']
    
    return df['alpha_factor']
