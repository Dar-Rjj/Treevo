```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3_day_return'] = df['close'].pct_change(periods=3)
    df['5_day_return'] = df['close'].pct_change(periods=5)
    
    # Combine 3-Day and 5-Day Returns with weights
    df['combined_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Compute Intraday Volume-Weighted Spreads
    df['volume_weighted_high_low'] = df['high_low_spread'] * df['volume']
    df['volume_weighted_open_close'] = df['open_close_spread'] * df['volume']
    
    # Combine Spreads
    df['average_spread'] = (df['volume_weighted_high_low'] + df['volume_weighted_open_close']) / 2
    
    # Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['enhanced_intraday_volatility'] = df['average_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['enhanced_intraday_volatility']
    
    return df['final_factor']
```
