```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'].shift(1) * df['volume']
    
    # Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / (2 * df['mean_price'])
    
    # Adaptive Moving Average (EMA with a span of 20 days)
    df['adaptive_moving_avg'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Long-Term Momentum
    df['long_term_momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Combine Factors
    weights = {
        'volume_adjusted_momentum': 0.5,
        'ohl_volatility_index': 0.3,
        'adaptive_moving_avg': 0.1,
        'long_term_momentum': 0.1
    }
    
    df['combined_factor'] = (df['volume_adjusted_momentum'] * weights['volume_adjusted_momentum'] +
                             df['ohl_volatility_index'] * weights['ohl_volatility_index'] +
                             df['adaptive_moving_avg'] * weights['adaptive_moving_avg'] +
                             df['long_term_momentum'] * weights['long_term_momentum'])
    
    return df['combined_factor'].dropna()
```
