```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Exponential Weighted Moving Average (EWMA) of Returns
    decay_factor = 0.94
    df['ewma_returns'] = df['daily_return'].ewm(alpha=1-decay_factor, adjust=False).mean()
    
    # Calculate Price-Averaged Volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['price_weighted_volume'] = (df['volume'] * df['typical_price']).rolling(window=50).mean() / df['typical_price'].rolling(window=50).sum()

    # Calculate Long-Term Trend
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['long_term_trend'] = df['50_day_ma'] - df['200_day_ma']
    
    # Final Factor Calculation
    df['factor_value'] = (df['ewma_returns'] * df['price_weighted_volume']) + df['long_term_trend']

    return df['factor_value'].dropna()
```
