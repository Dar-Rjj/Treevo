```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'].shift(-2) / df['close'].shift(-1)) - 1
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) / df['close'].shift(-1)) - 1
    
    # Combine 3-Day and 5-Day Returns with weights
    df['combined_return'] = 0.6 * df['3d_return'] + 0.4 * df['5d_return']
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume'].shift(-1)
    
    # Calculate Intra-Day Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intra_day_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / (2 * df['mean_price'])
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['intra_day_volatility_index']

    return df['final_factor'].shift(1).dropna()
```
