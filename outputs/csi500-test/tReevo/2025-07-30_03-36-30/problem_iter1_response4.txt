import pandas as pd

def heuristics_v2(df):
    # Compute Daily High Price Change
    df['high_change'] = df['high'] - df['high'].shift(1)
    
    # Compute Daily Low Price Change
    df['low_change'] = df['low'] - df['low'].shift(1)
    
    # Calculate High-to-Low Price Momentum
    df['high_to_low_momentum'] = df['high_change'] - df['low_change']
    
    # Calculate Average Volume over a fixed number of days (e.g., 20)
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Normalize by Current Volume
    df['volume_intensity'] = df['volume'] / df['avg_volume']
    
    # Adjust Momentum by Volume Intensity
    df['adjusted_momentum'] = df['high_to_low_momentum'] * df['volume_intensity']
    
    # Return the factor values as a pandas Series
    return df['adjusted_momentum']
