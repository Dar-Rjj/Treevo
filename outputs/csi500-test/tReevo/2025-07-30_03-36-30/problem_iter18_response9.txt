import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, 10-Day, and 20-Day Rolling Returns
    df['3D_Return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    df['20D_Return'] = (df['close'].shift(-19) / df['close']) - 1
    
    # Weighted Average of 3-Day, 5-Day, 10-Day, and 20-Day Returns
    df['Weighted_Returns'] = (
        df['3D_Return'] * 0.2 +
        df['5D_Return'] * 0.3 +
        df['10D_Return'] * 0.25 +
        df['20D_Return'] * 0.25
    )
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Weighted_Returns'] * df['volume']
    
    # Calculate Comprehensive Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    df['High_Mean_Spread'] = df['high'] - df['Mean_Price']
    df['Low_Mean_Spread'] = df['Mean_Price'] - df['low']
    
    # Combine Spreads
    df['Comprehensive_Volatility'] = (df['High_Low_Spread'] + df['Open_Close_Spread'] + 
                                      df['High_Mean_Spread'] + df['Low_Mean_Spread']) / df['Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Daily_Change_Volume'] = df['volume'] - df['volume'].shift(1)
    df['Normalized_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Normalized_Change_Volume'] = df['Daily_Change_Volume'] / df['volume']
    
    # Add Normalized Changes to Volume-Adjusted Multi-Timeframe Price Momentum
    df['Market_Sentiment_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Change_Amount'] + df['Normalized_Change_Volume']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Market_Sentiment_Adjusted_Momentum'] - df['Comprehensive_Volatility']
    
    return df['Final_Factor']
