```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1

    # Apply Decay Factor to Returns
    df['5D_Decayed_Return'] = df['5D_Return'] * 0.8
    df['10D_Decayed_Return'] = df['10D_Return'] * 0.7

    # Combine 5-Day and 10-Day Returns
    df['Combined_Return'] = (df['5D_Decayed_Return'] * 0.6) + (df['10D_Decayed_Return'] * 0.4)

    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']

    # Calculate Enhanced 20-Day Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = abs(df['open'].rolling(window=20).sum() - df['close'].rolling(window=20).sum())
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment and Liquidity
    df['Daily_Change_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    df['Market_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']

    # Adjust for Market Capitalization
    df['20D_Market_Cap'] = df['close'].rolling(window=20).mean() * df['amount'].rolling(window=20).mean()
    df['Final_Factor'] = (df['Market_Sentiment'] * df['Normalized_Volume']) / df['20D_Market_Cap']

    # Final Factor Calculation
    df['Final_Factor'] = df['Final_Factor'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor']
```
