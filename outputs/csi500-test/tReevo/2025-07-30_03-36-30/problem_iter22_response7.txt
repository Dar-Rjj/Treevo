import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5_Day_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10_Day_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Dynamic Weighted Average of 5-Day and 10-Day Returns
    df['5_Day_EMA'] = df['5_Day_Return'].ewm(span=10, adjust=False).mean()
    df['Combined_Return'] = (df['5_Day_Return'] * (1 - df['5_Day_EMA'])) + (df['10_Day_Return'] * df['5_Day_EMA'])
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20_High_Low_Range'] + df['20_Open_Close_Range']) / df['20_Mean_Price']
    
    # Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    
    # Liquidity and Trend Strength
    df['20_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20_Volume_EMA']
    
    # Adaptive Volatility Measures
    df['Volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['20_Volatility_EMA'] = df['Volatility'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volatility'] = df['Volatility'] / df['20_Volatility_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum'] - df['Enhanced_Volatility_Index'] 
                          + df['Normalized_Daily_Change_Amount'] + df['Normalized_Volume'] 
                          + df['Normalized_Volatility'])
    
    return df['Final_Factor'].dropna()
