```python
import pandas as pd

def heuristics_v2(df, lookback_window=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['open'] - df['close']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Price Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum - volatility_indicator
    
    # Volume-Weight the Adjusted Momentum
    # Calculate Average Volume over Lookback Period
    avg_volume = df['volume'].rolling(window=lookback_window).mean()
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Volume-Weight the Adjusted Momentum
    volume_weighted_adjusted_momentum = adjusted_momentum * volume_ratio
    
    # Final Alpha Factor
    alpha_factor = volume_weighted_adjusted_momentum
    
    return alpha_factor
```
