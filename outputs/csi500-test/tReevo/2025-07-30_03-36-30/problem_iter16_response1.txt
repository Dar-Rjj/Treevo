```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5, 10, and 20-Day Rolling Returns
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10_day_return'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['20_day_return'] = (df['close'].shift(-19) - df['close']) / df['close']

    # Weighted Average of 5, 10, and 20-Day Returns
    df['weighted_return'] = 0.3 * df['5_day_return'] + 0.4 * df['10_day_return'] + 0.3 * df['20_day_return']

    # Volume-Weighted Multi-Day Price Momentum
    df['volume_weighted_momentum'] = df['weighted_return'] * df['volume']

    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']

    # Smooth Volatility Index with Exponential Smoothing
    df['smoothed_volatility'] = df['volatility_index'].ewm(alpha=0.2, adjust=False).mean()

    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['smoothed_daily_change_amount'] = df['daily_change_amount'].ewm(alpha=0.2, adjust=False).mean()
    df['market_sentiment'] = df['volume_weighted_momentum'] + df['smoothed_daily_change_amount']

    # Adjust for Cumulative Market Impact
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_deviation'] = (df['volume'] - df['10_day_avg_volume']) / df['10_day_avg_volume']
    df['adjusted_momentum'] = df['market_sentiment'] * df['volume_deviation']

    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['smoothed_volatility']

    return df['final_factor']
```
