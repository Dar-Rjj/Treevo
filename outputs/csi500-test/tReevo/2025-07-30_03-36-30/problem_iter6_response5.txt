```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, and 10-Day Rolling Returns
    df['3d_return'] = (df['close'] / df['close'].shift(2)) - 1
    df['5d_return'] = (df['close'] / df['close'].shift(4)) - 1
    df['10d_return'] = (df['close'] / df['close'].shift(9)) - 1
    
    # Combine 3-Day, 5-Day, and 10-Day Returns with weights
    df['weighted_returns'] = (df['3d_return'] * 0.3 + 
                              df['5d_return'] * 0.5 + 
                              df['10d_return'] * 0.2)
    
    # Multiply by Volume of day t
    df['vol_adj_momentum'] = df['weighted_returns'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['factor_value'] = df['vol_adj_momentum'] - df['ohl_volatility_index']
    
    return df['factor_value']
```
