```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Weighted Price Momentum
    df['volume_weighted_momentum'] = df['daily_return'] * df['volume']
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average True Range (ATR) over 7 days
    def true_range(row):
        return max(row['high'] - row['low'], abs(row['high'] - row['close'].shift(1)), abs(row['low'] - row['close'].shift(1)))
    
    df['true_range'] = df.apply(true_range, axis=1)
    df['atr_7'] = df['true_range'].rolling(window=7).mean()
    
    # Combine Spreads and ATR
    # Assigning weights based on historical performance (example: 0.5 for high-low, 0.5 for open-close)
    df['composite_volatility'] = 0.5 * df['high_low_spread'] + 0.5 * df['open_close_spread'] + df['atr_7']
    
    # Relative Price Change (20 day)
    df['relative_price_change'] = df['close'] / df['close'].shift(20) - 1
    
    # Adjust Combined Factor
    df['adjusted_factor'] = (df['volume_weighted_momentum'] / df['composite_volatility']) * df['relative_price_change']
    
    # Return the final alpha factor
    return df['adjusted_factor'].dropna()
```
