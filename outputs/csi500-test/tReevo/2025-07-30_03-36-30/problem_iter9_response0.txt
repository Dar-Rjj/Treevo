```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Combine 3-Day and 5-Day Returns using a weighted average
    df['combined_return'] = (0.6 * df['3d_return'] + 0.4 * df['5d_return'])
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Combine Spreads
    df['average_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['volatility_index'] = df['average_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['alpha_factor']
```
