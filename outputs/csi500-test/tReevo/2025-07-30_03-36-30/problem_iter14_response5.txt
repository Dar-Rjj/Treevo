```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Short-Term Price Momentum
    df['2D_Return'] = df['close'].pct_change(periods=1).shift(1)
    df['4D_Return'] = df['close'].pct_change(periods=3)
    df['Combined_Return'] = 0.7 * df['2D_Return'] + 0.3 * df['4D_Return']
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']

    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']

    # Integrate Dynamic Market Sentiment
    df['5D_Change_in_Amount'] = (df['amount'] - df['amount'].shift(4)) / df['amount']
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['5D_Change_in_Amount']

    # Incorporate Volume Trends
    df['5D_Volume_MA'] = df['volume'].rolling(window=5).mean()
    df['Volume_Trend'] = (df['volume'] - df['5D_Volume_MA']) / df['5D_Volume_MA']
    df['Final_Factor'] = df['Adjusted_Momentum'] * (1 + df['Volume_Trend'])

    # Final Factor Calculation
    df['Final_Factor'] -= df['Volatility_Index']

    return df['Final_Factor']
```
