```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, and 10-Day Rolling Returns
    df['3_day_return'] = df['close'].pct_change(periods=3)
    df['5_day_return'] = df['close'].pct_change(periods=5)
    df['10_day_return'] = df['close'].pct_change(periods=10)
    
    # Combine 3-Day, 5-Day, and 10-Day Returns with weights
    df['weighted_returns'] = (df['3_day_return'] * 0.4) + (df['5_day_return'] * 0.3) + (df['10_day_return'] * 0.3)
    
    # Volume-Weighted Multi-Period Price Momentum
    df['volume_weighted_momentum'] = df['weighted_returns'] * df['volume']
    
    # Calculate Enhanced Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['log_returns'] = np.log(df['close']).diff()
    df['3_day_std_dev'] = df['log_returns'].rolling(window=3).std()
    
    # Combine Spreads and Standard Deviation
    df['average_spreads'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['enhanced_volatility_index'] = df['average_spreads'] + df['3_day_std_dev']
    
    # Normalize by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['normalized_volatility_index'] = df['enhanced_volatility_index'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_momentum'] - df['normalized_volatility_index']
    
    return df['final_factor']
```
