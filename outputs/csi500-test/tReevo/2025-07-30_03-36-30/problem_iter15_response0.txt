```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    five_day_return = (df['close'] / df['close'].shift(4)) - 1
    ten_day_return = (df['close'] / df['close'].shift(9)) - 1

    # Combine 5-Day and 10-Day Returns
    weighted_return = (five_day_return * 0.7) + (ten_day_return * 0.3)

    # Volume-Adjusted Multi-Day Price Momentum
    volume_adjusted_momentum = weighted_return * df['volume']

    # Calculate Open-High-Low Volatility Index
    high_low_spread = df['high'] - df['low']
    open_close_spread = abs(df['open'] - df['close'])
    mean_price = (df['open'] + df['close']) / 2
    volatility_index = (high_low_spread + open_close_spread) / mean_price

    # Integrate Market Sentiment
    daily_change_in_amount = df['amount'] - df['amount'].shift(1)
    sentiment_adjusted_momentum = volume_adjusted_momentum + daily_change_in_amount

    # Final Factor Calculation
    final_factor = sentiment_adjusted_momentum - volatility_index

    return final_factor
```
