```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'] / df['close'].shift(2)) - 1
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Combine 3-Day and 5-Day Returns
    df['combined_return'] = (df['3d_return'] * 0.6) + (df['5d_return'] * 0.4)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Combine Spreads
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intra_day_volatility_index'] = (df[['high_low_spread', 'open_close_spread']].mean(axis=1)) / df['mean_price']
    
    # Intra-Day Volatility Trend
    df['5d_intra_day_volatility_ma'] = df['intra_day_volatility_index'].rolling(window=5).mean()
    df['trend_adjusted_volatility'] = (df['intra_day_volatility_index'] - df['5d_intra_day_volatility_ma']) / df['5d_intra_day_volatility_ma']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['trend_adjusted_volatility']
    
    return df['final_factor']
```
