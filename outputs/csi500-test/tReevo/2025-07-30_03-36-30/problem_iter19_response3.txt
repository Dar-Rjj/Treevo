```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Compute 10-Day Rolling Return
    df['10d_return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Combine 5-Day and 10-Day Returns
    df['weighted_return'] = (df['5d_return'] * 0.7) + (df['10d_return'] * 0.3)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['20d_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20d_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20d_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility'] = (df['20d_high_low_range'] + df['20d_open_close_range']) / df['20d_mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_in_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['market_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_in_amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20d_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20d_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['market_sentiment'] * df['normalized_volume'] - df['enhanced_volatility']
    
    return df['final_factor'].dropna()
```
