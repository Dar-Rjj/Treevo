import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Combine 3-Day and 5-Day Returns
    df['combined_return'] = 0.6 * df['3_day_return'] + 0.4 * df['5_day_return']
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']

    # Calculate Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    df['intraday_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']

    # Integrate Market Sentiment
    df['daily_change_in_amount'] = df['amount'].diff()
    df['normalized_daily_change_in_amount'] = df['daily_change_in_amount'] / df['amount']
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_in_amount']

    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['intraday_volatility_index']

    return df['final_factor'].dropna()
