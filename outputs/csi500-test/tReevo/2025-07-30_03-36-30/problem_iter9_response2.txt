```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day Rolling Return
    df['3d_return'] = (df['close'] / df['close'].shift(2)) - 1
    
    # Calculate 5-Day Rolling Return
    df['5d_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Weighted Average of 3-Day and 5-Day Returns
    df['weighted_return'] = 0.6 * df['3d_return'] + 0.4 * df['5d_return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average High-Low and Open-Close Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['ohl_volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['factor'] = df['volume_adjusted_momentum'] - df['ohl_volatility_index']
    
    return df['factor']
```
