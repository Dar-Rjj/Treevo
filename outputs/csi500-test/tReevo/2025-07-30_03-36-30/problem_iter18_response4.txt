```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Apply Decay Factor to Returns
    df['Decayed_5D_Return'] = df['5D_Return'] * 0.8
    df['Decayed_10D_Return'] = df['10D_Return'] * 0.7
    
    # Combine 5-Day and 10-Day Returns with Weights
    df['Weighted_Return'] = (df['Decayed_5D_Return'] * 0.6) + (df['Decayed_10D_Return'] * 0.4)
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()).abs()
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['Volume_Adjusted_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    df['Liquidity_Adjusted_Momentum'] = df['Volume_Adjusted_Sentiment'] * df['Normalized_Volume']
    
    # Adjust for Market Capitalization
    df['20D_Market_Cap'] = (df['close'] * df['amount']).rolling(window=20).mean()
    df['Final_Factor'] = (df['Liquidity_Adjusted_Momentum'] - df['Volatility_Index']) / df['20D_Market_Cap']
    
    return df['Final_Factor']
```
