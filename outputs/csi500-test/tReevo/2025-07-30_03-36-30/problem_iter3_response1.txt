```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Volume-Adjusted Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Daily_Return'].shift(1) * df['volume']
    
    # Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = (df['open'] - df['close']).abs()
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / (2 * df['Mean_Price'])
    
    # Integrate Longer-Term Trends
    df['20_Day_MA'] = df['close'].rolling(window=20).mean()
    df['50_Day_MA'] = df['close'].rolling(window=50).mean()
    
    # Adaptive Moving Averages
    df['10_Day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['30_Day_EMA'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Leverage Intra-Day Price Movements
    df['Intra_Day_Range'] = df['high'] - df['low']
    df['Intra_Day_Volatility'] = df['Intra_Day_Range'] / df['open']
    
    # Liquidity Metrics
    df['Turnover_Ratio'] = df['volume'] / df['close']
    df['Dollar_Volume'] = df['volume'] * df['close']
    
    # Combine all the factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] + 
        df['Volatility_Index'] + 
        (df['20_Day_MA'] - df['50_Day_MA']) +
        (df['10_Day_EMA'] - df['30_Day_EMA']) +
        df['Intra_Day_Volatility'] +
        df['Turnover_Ratio'] +
        np.log(df['Dollar_Volume'])
    )
    
    # Normalize the alpha factor
    df['Alpha_Factor'] = (df['Alpha_Factor'] - df['Alpha_Factor'].mean()) / df['Alpha_Factor'].std()

    return df['Alpha_Factor']
```
