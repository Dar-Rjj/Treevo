import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Combine 5-Day and 10-Day Returns
    df['Combined_Return'] = 0.6 * df['5D_Return'] + 0.4 * df['10D_Return']
    
    # Volume-Adjusted Short-Term Price Momentum
    df['Volume_Adjusted_Momentum'] = df['volume'] * df['Combined_Return']
    
    # Calculate 20-Day High-Low Range
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    
    # 20-Day Mean Price
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['Momentum_With_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']
    
    # Calculate 10-Day Volume EMA
    df['10D_Volume_EMA'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Normalize Volume of day t
    df['Normalized_Volume'] = df['volume'] / df['10D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Momentum_With_Sentiment'] * df['Normalized_Volume'] - df['Volatility_Index']
    
    return df['Final_Factor']
