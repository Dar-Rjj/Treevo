```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns with weights
    df['combined_return'] = (0.6 * df['3d_return'] + 0.4 * df['5d_return'])
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads and divide by Mean Price
    mean_price = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / mean_price
    
    # Calculate Daily Change in Amount
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    
    # Normalize Daily Change in Amount
    df['normalized_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_amount']
    
    # Final Factor Calculation: Subtract Open-High-Low Volatility Index from Adjusted Momentum
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['ohl_volatility_index']
    
    return df['final_factor']
```
