```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5DayReturn'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10DayReturn'] = (df['close'].shift(-9) - df['close']) / df['close']

    # Apply Decay Factor to Returns
    df['Decayed5DayReturn'] = df['5DayReturn'] * 0.8
    df['Decayed10DayReturn'] = df['10DayReturn'] * 0.7

    # Combine 5-Day and 10-Day Returns
    df['CombinedReturn'] = (df['Decayed5DayReturn'] * 0.6 + df['Decayed10DayReturn'] * 0.4)

    # Volume-Adjusted Multi-Day Price Momentum
    df['VolumeAdjustedMomentum'] = df['CombinedReturn'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20DayHighLowRange'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20DayOpenCloseRange'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20DayMeanPrice'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['EnhancedVolatilityIndex'] = (df['20DayHighLowRange'] + df['20DayOpenCloseRange']) / df['20DayMeanPrice']

    # Integrate Market Sentiment
    df['DailyChangeAmount'] = df['amount'].diff()
    df['NormalizedDailyChangeAmount'] = df['DailyChangeAmount'] / df['amount']
    df['VolumeAdjustedMomentumWithSentiment'] = df['VolumeAdjustedMomentum'] + df['NormalizedDailyChangeAmount']

    # Incorporate Liquidity and Trend Strength
    df['20DayVolumeEMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['NormalizedVolume'] = df['volume'] / df['20DayVolumeEMA']

    # Adjust for Market Capitalization
    df['MarketCap'] = df['close'] * df['amount']
    df['20DayAverageMarketCap'] = df['MarketCap'].rolling(window=20).mean()
    df['FinalFactor'] = (df['VolumeAdjustedMomentumWithSentiment'] * df['NormalizedVolume']) / df['20DayAverageMarketCap'] - df['EnhancedVolatilityIndex']

    return df['FinalFactor']
```
