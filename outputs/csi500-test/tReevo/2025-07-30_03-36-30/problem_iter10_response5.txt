```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    three_day_return = (df['close'].shift(-2) / df['close']) - 1
    five_day_return = (df['close'].shift(-4) / df['close']) - 1
    weighted_return = (three_day_return * 0.6) + (five_day_return * 0.4)
    volume_adjusted_momentum = weighted_return * df['volume']

    # Calculate Enhanced Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_weighted_high_low = high_low_spread * df['volume']
    volume_weighted_open_close = open_close_spread * df['volume']
    
    mean_price = (df['open'] + df['close']) / 2
    intraday_volatility = ((volume_weighted_high_low + volume_weighted_open_close) / 2) / mean_price
    
    # Final Factor Calculation
    final_factor = volume_adjusted_momentum - intraday_volatility
    
    return final_factor
```
