```python
import pandas as pd

def heuristics_v2(df, lookback=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['close'] - df['open']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Price Momentum
    momentum = df['close'].pct_change()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum - (volatility_indicator / df['close'])
    
    # Volume-Weight the Adjusted Momentum
    average_volume = df['volume'].rolling(window=lookback).mean()
    volume_ratio = df['volume'] / average_volume
    volume_weighted_adjusted_momentum = adjusted_momentum * volume_ratio
    
    return volume_weighted_adjusted_momentum
```
