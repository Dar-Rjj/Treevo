```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, and 10-Day Rolling Returns
    df['3_day_return'] = (df['close'] / df['close'].shift(2)) - 1
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    df['10_day_return'] = (df['close'] / df['close'].shift(9)) - 1
    
    # Combine 3-Day, 5-Day, and 10-Day Returns using weights
    df['weighted_return'] = (df['3_day_return'] * 0.4) + (df['5_day_return'] * 0.3) + (df['10_day_return'] * 0.3)
    
    # Volume-Weighted Multi-Period Price Momentum
    df['volume_weighted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Compute 3-Day Standard Deviation of Log Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['3_day_std'] = df['log_return'].rolling(window=3).std()
    
    # Combine Spreads and Standard Deviation
    df['spread_sum'] = df['high_low_spread'] + df['open_close_spread']
    df['dynamic_volatility_index'] = df['spread_sum'] + df['3_day_std']
    
    # Adjust by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['adjusted_dvi'] = df['dynamic_volatility_index'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_momentum'] - df['adjusted_dvi']
    
    return df['final_factor']
```
