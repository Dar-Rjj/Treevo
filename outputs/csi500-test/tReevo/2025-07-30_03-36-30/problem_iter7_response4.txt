```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Combine 3-Day and 5-Day Returns with weighted average
    df['weighted_return'] = df['3_day_return'] * 0.6 + df['5_day_return'] * 0.4
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average High-Low and Open-Close Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2.0
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2.0
    
    # Open-High-Low Volatility Index
    df['volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['final_factor']
```
