```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute Simple Moving Average of Log Returns (SMA_L)
    window = 20
    df['SMA_L'] = df['log_return'].rolling(window=window).mean()
    
    # Estimate Volatility by Standard Deviation of Log Returns (SD_L)
    df['SD_L'] = df['log_return'].rolling(window=window).std()
    
    # Adjust Momentum by Volume
    df['volume_momentum'] = df['volume'].diff()
    df['adjusted_momentum'] = df['SMA_L'] * df['volume_momentum']
    
    # Combine Momentum and Volatility Components
    df['factor_value'] = df['adjusted_momentum'] - df['SD_L']
    
    return df['factor_value'].dropna()
```
