import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].pct_change(5))
    df['10D_Return'] = (df['close'].pct_change(10))
    
    # Dynamic Weighted Average of 5-Day and 10-Day Returns
    df['5D_EMA'] = df['5D_Return'].ewm(span=10).mean()
    df['Combined_Return'] = (df['5D_Return'] * (1 - df['5D_EMA'])) + (df['10D_Return'] * df['5D_EMA'])
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'].pct_change()
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Volume_Adjusted_Momentum_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Incorporate Adaptive Volatility Measures
    df['20D_Volatility_EMA'] = df['Enhanced_Volatility_Index'].ewm(span=20).mean()
    df['Normalized_Volatility'] = df['Enhanced_Volatility_Index'] / df['20D_Volatility_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum_Sentiment'] - df['Enhanced_Volatility_Index']) + df['Normalized_Volume'] + df['Normalized_Volatility']
    
    return df['Final_Factor']
