```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Momentum (5-Day MA)
    short_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Medium-Term Momentum (10-Day MA)
    medium_momentum = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Momentum (20-Day MA)
    long_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Short-Medium Momentum Difference
    short_medium_diff = short_momentum - medium_momentum
    
    # Calculate Medium-Long Momentum Difference
    medium_long_diff = medium_momentum - long_momentum
    
    # Calculate Volume Trend (10-Day Moving Average of Volume)
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Factor in Volume Trend
    volume_adjusted_short_medium_diff = short_medium_diff * volume_trend
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate True Range
    true_range = np.maximum(np.maximum(df['high'] - df['low'], 
                                       np.abs(df['high'] - df['close'].shift(1))), 
                            np.abs(df['low'] - df['close'].shift(1)))
    
    # Calculate Average True Range (ATR) (14-Day)
    atr = true_range.rolling(window=14).mean()
    
    # Adjust Momentum Differences by ATR
    adjusted_short_medium_diff = volume_adjusted_short_medium_diff / atr
    adjusted_medium_long_diff = medium_long_diff / atr
    
    # Final Alpha Factor
    alpha_factor = adjusted_short_medium_diff + adjusted_medium_long_diff
    
    return alpha_factor
```
