import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10_day_return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Apply Decay Factor to Returns
    df['5_day_return_weighted'] = df['5_day_return'] * 0.8
    df['10_day_return_weighted'] = df['10_day_return'] * 0.7
    
    # Combine 5-Day and 10-Day Returns
    df['combined_return'] = (df['5_day_return_weighted'] * 0.6) + (df['10_day_return_weighted'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate 20-Day High-Low and Open-Close Ranges
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_day_open_close_range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min())
    
    # Combine 20-Day Ranges
    df['20_day_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_in_amount'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_change_in_amount'] = df['daily_change_in_amount'] / df['amount']
    df['market_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_in_amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']
    df['liquidity_and_trend_strength'] = df['market_sentiment'] * df['normalized_volume']
    
    # Adjust for Market Capitalization
    df['20_day_market_capitalization'] = df['close'].rolling(window=20).mean() * df['amount'].rolling(window=20).mean()
    df['final_factor'] = (df['liquidity_and_trend_strength'] - df['enhanced_volatility_index']) / df['20_day_market_capitalization']
    
    return df['final_factor'].dropna()
