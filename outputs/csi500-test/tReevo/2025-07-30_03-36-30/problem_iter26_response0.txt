```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 10-Day Rolling Return
    df['10_Day_Rolling_Return'] = (df['close'] / df['close'].shift(9)) - 1
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['10_Day_Rolling_Return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20_Day_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20_Day_Open_Close_Range'] = abs(df['open'] - df['close']).rolling(window=20).sum()
    
    # Combine 20-Day Ranges
    df['20_Day_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20_Day_High_Low_Range'] + df['20_Day_Open_Close_Range']) / df['20_Day_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_in_Amount'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_in_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20_Day_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20_Day_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Momentum_with_Sentiment'] - df['Enhanced_Volatility_Index'] * df['Normalized_Volume']
    
    return df['Final_Factor'].dropna()
```
