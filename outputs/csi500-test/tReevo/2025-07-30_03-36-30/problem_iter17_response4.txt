```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10_day_return'] = (df['close'].shift(-9) / df['close']) - 1
    df['weighted_return'] = (df['5_day_return'] * 0.7) + (df['10_day_return'] * 0.3)
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['20_day_range_high_low'] = df['20_day_high'] - df['20_day_low']
    
    df['20_day_open'] = df['open'].rolling(window=20).mean()
    df['20_day_close'] = df['close'].rolling(window=20).mean()
    df['20_day_range_open_close'] = (df['20_day_open'] - df['20_day_close']).abs()
    
    df['20_day_mean_price'] = (df['20_day_open'] + df['20_day_close']) / 2
    df['enhanced_volatility_index'] = (df['20_day_range_high_low'] + df['20_day_range_open_close']) / df['20_day_mean_price']

    # Integrate Market Sentiment
    df['daily_change_in_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['market_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_in_amount']

    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_volume'] = df['volume'] / df['20_day_volume_ema']

    # Final Factor Calculation
    df['final_factor'] = (df['market_sentiment'] * df['adjusted_volume']) - df['enhanced_volatility_index']

    return df['final_factor']
```
