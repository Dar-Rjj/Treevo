```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns using a weighted average
    df['combined_return'] = 0.6 * df['3d_return'] + 0.4 * df['5d_return']
    
    # Multiply by Volume of day t to get Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low and Open-Close Spreads
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average of the two spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['ohl_volatility_index'] = df['avg_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['ohl_volatility_index']
    
    return df['final_factor'].dropna()
```
