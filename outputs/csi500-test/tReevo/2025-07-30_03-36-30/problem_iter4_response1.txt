```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['open'] - df['close']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(1)
    
    # Calculate Medium-Term Price Momentum
    medium_term_momentum = df['close'].pct_change(5)
    
    # Combine Short-Term and Medium-Term Momentum
    combined_momentum = short_term_momentum - medium_term_momentum
    
    # Incorporate Volatility Adjustment
    adjusted_factor = combined_momentum - volatility_indicator
    
    # Volume-Weight the Adjusted Factor
    final_factor = adjusted_factor * df['volume']
    
    return final_factor
```
