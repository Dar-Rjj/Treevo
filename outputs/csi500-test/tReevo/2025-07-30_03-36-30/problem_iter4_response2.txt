```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'] * df['volume']
    
    # Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Longer-Term Trends
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['ma_200'] = df['close'].rolling(window=200).mean()
    
    # Adaptive Moving Averages
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Intra-Day Price Movements
    df['intra_day_range'] = df['high'] - df['low']
    df['intra_day_volatility'] = df['intra_day_range'] / df['open']
    
    # Liquidity Metrics
    df['turnover_ratio'] = df['volume'] / df['close']
    df['dollar_volume'] = df['volume'] * df['close']
    
    # Combine all factors to create a final alpha factor
    df['alpha_factor'] = (
        df['volume_adjusted_momentum'] +
        df['volatility_index'] +
        df['ma_20'] - df['ma_50'] +
        df['ema_10'] - df['ema_30'] +
        df['intra_day_volatility'] +
        df['turnover_ratio'] +
        df['dollar_volume']
    )
    
    return df['alpha_factor']
```
