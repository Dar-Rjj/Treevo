import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day and 5-Day Rolling Returns
    df['3D_Return'] = (df['close'] / df['close'].shift(2)) - 1
    df['5D_Return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Combine 3-Day and 5-Day Returns using a weighted average
    df['Combined_Returns'] = 0.6 * df['3D_Return'] + 0.4 * df['5D_Return']
    
    # Compute 10-Day Moving Average of Volume
    df['10D_Volume_MA'] = df['volume'].rolling(window=10).mean()
    
    # Compute Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['10D_Volume_MA']
    
    # Multiply Combined Returns by Volume Ratio
    df['Volume_Adjusted_Momentum'] = df['Combined_Returns'] * df['Volume_Ratio']
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    
    # Calculate Mean Price
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    
    # Combine Spreads to get Volatility Index
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    
    # Calculate Daily Change in Amount
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    
    # Compute 5-Day Moving Average of Amount
    df['5D_Amount_MA'] = df['amount'].rolling(window=5).mean()
    
    # Normalize Daily Change in Amount
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['5D_Amount_MA']
    
    # Add Refined Daily Change in Amount to Volume-Adjusted Multi-Day Price Momentum
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']
    
    # Subtract Open-High-Low Volatility Index from Adjusted Momentum
    df['Final_Factor'] = df['Momentum_with_Sentiment'] - df['Volatility_Index']
    
    # Drop rows with NaN values created by rolling windows
    df.dropna(inplace=True)
    
    return df['Final_Factor']
