```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5_Day_Return'] = (df['close'] / df['close'].shift(4)) - 1
    df['10_Day_Return'] = (df['close'] / df['close'].shift(9)) - 1
    
    # Apply Decay Factor to Returns
    df['5_Day_Return_Decayed'] = df['5_Day_Return'] * 0.8
    df['10_Day_Return_Decayed'] = df['10_Day_Return'] * 0.7
    
    # Combine 5-Day and 10-Day Returns (Weighted Average)
    df['Combined_Return'] = (df['5_Day_Return_Decayed'] * 0.6) + (df['10_Day_Return_Decayed'] * 0.4)
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['20_Day_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_Day_Open_Close_Range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()).abs()
    df['20_Day_Range_Sum'] = df['20_Day_High_Low_Range'] + df['20_Day_Open_Close_Range']
    df['20_Day_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['Enhanced_Volatility_Index'] = df['20_Day_Range_Sum'] / df['20_Day_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_in_Amount'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_in_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20_Day_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20_Day_Volume_EMA']
    df['Adjusted_Momentum'] = df['Momentum_with_Sentiment'] * df['Normalized_Volume']
    
    # Adjust for Market Capitalization
    df['Market_Cap'] = df['close'] * df['amount']
    df['20_Day_Avg_Market_Cap'] = df['Market_Cap'].rolling(window=20).mean()
    df['Final_Factor'] = df['Adjusted_Momentum'] - df['Enhanced_Volatility_Index']
    df['Final_Factor'] = df['Final_Factor'] / df['20_Day_Avg_Market_Cap']
    
    return df['Final_Factor'].dropna()
```
