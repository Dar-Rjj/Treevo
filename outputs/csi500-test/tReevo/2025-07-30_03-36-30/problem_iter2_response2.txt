```python
import pandas as pd

def heuristics_v2(df, lookback_window=20):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    
    # Adjust Close-to-Open Return by Volume
    avg_volume = df['volume'].rolling(window=lookback_window).mean()
    volume_ratio = df['volume'] / avg_volume
    volume_adjusted_momentum = close_to_open_return * volume_ratio
    
    # Combine High-Low Spread and Volume-Adjusted Momentum
    combined_factor = high_low_spread + volume_adjusted_momentum
    
    return combined_factor
```
