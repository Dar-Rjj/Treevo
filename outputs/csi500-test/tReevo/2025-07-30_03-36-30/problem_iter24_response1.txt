import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10_day_return'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['weighted_return'] = 0.7 * df['5_day_return'] + 0.3 * df['10_day_return']
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_day_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20_day_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']

    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'].diff()
    df['normalized_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    df['sentiment_adjusted_momentum'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_amount']

    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']

    # Final Factor Calculation
    df['final_factor'] = df['sentiment_adjusted_momentum'] * df['normalized_volume'] - df['enhanced_volatility_index']
    
    return df['final_factor'].dropna()
