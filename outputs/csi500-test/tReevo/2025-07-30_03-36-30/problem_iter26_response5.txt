```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    df['Weighted_Return'] = (df['5D_Return'] * 0.7) + (df['10D_Return'] * 0.3)
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High'] = df['high'].rolling(window=20).max()
    df['20D_Low'] = df['low'].rolling(window=20).min()
    df['20D_Range'] = df['20D_High'] - df['20D_Low']
    
    df['20D_Open'] = df['open'].rolling(window=20).mean()
    df['20D_Close'] = df['close'].rolling(window=20).mean()
    df['20D_OpenClose_Range'] = (df['20D_Open'] - df['20D_Close']).abs()
    
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility'] = (df['20D_Range'] + df['20D_OpenClose_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'].diff()
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['EMA_Daily_Change_Amount'] = df['Normalized_Daily_Change_Amount'].ewm(span=2, adjust=False).mean()
    
    df['Volume_Adjusted_Momentum_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['EMA_Daily_Change_Amount']

    # Incorporate Liquidity and Trend Strength
    df['10D_Volume_EMA'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Liquidity_Adjusted_Volume'] = df['volume'] / df['10D_Volume_EMA']

    # Final Factor Calculation
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum_Sentiment'] * df['Liquidity_Adjusted_Volume']) - df['Enhanced_Volatility']

    return df['Final_Factor']
```
