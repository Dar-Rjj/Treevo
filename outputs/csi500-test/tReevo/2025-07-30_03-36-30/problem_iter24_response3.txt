```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Timeframe Price Momentum
    df['3d_return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5d_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10d_return'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['20d_return'] = (df['close'].shift(-19) - df['close']) / df['close']
    
    df['weighted_return'] = (0.2 * df['3d_return'] + 
                             0.3 * df['5d_return'] + 
                             0.25 * df['10d_return'] + 
                             0.25 * df['20d_return'])
    
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']

    # Calculate Enhanced Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['open'] - df['close']
    df['mean_price'] = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    df['high_mean_spread'] = df['high'] - df['mean_price']
    df['low_mean_spread'] = df['mean_price'] - df['low']
    
    df['volatility_index'] = (df['high_low_spread'] + 
                              df['open_close_spread'] + 
                              df['high_mean_spread'] + 
                              df['low_mean_spread']) / df['mean_price']

    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['daily_change_volume'] = df['volume'] - df['volume'].shift(1)
    
    df['normalized_change_amount'] = df['daily_change_amount'] / df['amount']
    df['normalized_change_volume'] = df['daily_change_volume'] / df['volume']
    
    df['sentiment_adjusted_momentum'] = (df['volume_adjusted_momentum'] + 
                                         df['normalized_change_amount'] + 
                                         df['normalized_change_volume'])

    # Incorporate Liquidity
    df['20d_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['relative_volume'] = df['volume'] / df['20d_volume_ema']

    # Final Factor Calculation
    df['final_factor'] = (df['sentiment_adjusted_momentum'] * df['relative_volume']) - df['volatility_index']
    
    return df['final_factor']
```
