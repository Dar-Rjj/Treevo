```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'] / df['close'].shift(2)) - 1
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Weighted Average of 3-Day and 5-Day Returns
    df['weighted_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_amount_change'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_amount_change'] = df['daily_amount_change'] / df['amount']
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_amount_change']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['volatility_index']
    
    return df['final_factor']
```
