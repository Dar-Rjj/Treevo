```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = (df['close'].shift(-4) / df['close'].shift(-5)) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close'].shift(-10)) - 1
    df['Weighted_Return'] = 0.6 * df['5D_Return'] + 0.4 * df['10D_Return']
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_HighLow_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_OpenClose_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['Enhanced_Volatility_Index'] = (df['20D_HighLow_Range'] + df['20D_OpenClose_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Volume_Adjusted_Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']

    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Adjusted_Volume'] = df['volume'] / df['20D_Volume_EMA']
    df['Final_Factor'] = (df['Volume_Adjusted_Momentum_with_Sentiment'] * df['Adjusted_Volume']) - df['Enhanced_Volatility_Index']

    return df['Final_Factor']
```
