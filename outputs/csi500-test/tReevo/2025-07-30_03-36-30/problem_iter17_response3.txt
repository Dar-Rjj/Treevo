```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Combine 5-Day and 10-Day Returns (Weighted Average)
    df['combined_return'] = 0.7 * df['5_day_return'] + 0.3 * df['10_day_return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20_day_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    
    # Combine 20-Day Ranges
    df['20_day_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']
    
    # Integrate Market Sentiment
    df['daily_amount_change'] = df['amount'].diff()
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_amount_change']
    
    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['enhanced_volatility_index'] * df['normalized_volume']
    
    return df['final_factor']
```
