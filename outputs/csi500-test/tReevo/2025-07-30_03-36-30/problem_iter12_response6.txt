import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    
    # Compute 5-Day Rolling Return
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Compute 3-Day and 5-Day Return Variances
    df['3d_var'] = df['close'].rolling(window=3).var()
    df['5d_var'] = df['close'].rolling(window=5).var()
    
    # Weight 3-Day and 5-Day Returns
    df['3d_weighted_return'] = df['3d_return'] * (1 / df['3d_var'])
    df['5d_weighted_return'] = df['5d_return'] * (1 / df['5d_var'])
    
    # Normalize Weights
    total_weight = df['3d_weighted_return'] + df['5d_weighted_return']
    df['3d_normalized_weight'] = df['3d_weighted_return'] / total_weight
    df['5d_normalized_weight'] = df['5d_weighted_return'] / total_weight
    
    # Volume Trend
    df['5d_vol_ma'] = df['volume'].rolling(window=5).mean()
    df['vol_trend'] = df['volume'] / df['5d_vol_ma']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = (
        (df['3d_normalized_weight'] * df['3d_return']) + 
        (df['5d_normalized_weight'] * df['5d_return'])
    ) * df['vol_trend']
    
    # Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['average_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_vol_index'] = df['average_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['ohl_vol_index']
    
    return df['final_factor'].dropna()
