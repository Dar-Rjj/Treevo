```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day, 5-Day, and 10-Day Rolling Returns
    df['3d_return'] = df['close'].pct_change(periods=3)
    df['5d_return'] = df['close'].pct_change(periods=5)
    df['10d_return'] = df['close'].pct_change(periods=10)
    
    # Combine 3-Day, 5-Day, and 10-Day Returns with weights
    df['weighted_return'] = (df['3d_return'] * 0.3 + 
                             df['5d_return'] * 0.3 + 
                             df['10d_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_weighted_return'] = df['weighted_return'] * df['volume']
    
    # Calculate Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (np.sqrt(df['high_low_spread'] * df['open_close_spread']) / df['mean_price']).fillna(0)
    
    # Integrate Normalized Market Sentiment
    df['daily_change_amount'] = df['amount'].diff()
    df['normalized_daily_change'] = df['daily_change_amount'] / df['amount']
    df['adjusted_momentum'] = df['volume_weighted_return'] + df['normalized_daily_change']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['intraday_volatility']
    
    return df['final_factor'].dropna()
```
