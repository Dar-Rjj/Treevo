import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Multi-Day Price Momentum
    df['3d_return'] = (df['close'].pct_change(3) * 0.6)
    df['5d_return'] = (df['close'].pct_change(5) * 0.4)
    df['multi_day_momentum'] = (df['3d_return'] + df['5d_return']) * df['volume']
    
    # Calculate Refined Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Normalized Market Sentiment
    df['daily_change_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['momentum_with_sentiment'] = df['multi_day_momentum'] + df['daily_change_amount']
    
    # Adjust for Cumulative Market Impact
    df['10d_vol_avg'] = df['volume'].rolling(window=10).mean()
    df['vol_deviation'] = (df['volume'] - df['10d_vol_avg']) / df['10d_vol_avg']
    df['adjusted_momentum'] = df['momentum_with_sentiment'] * df['vol_deviation']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['intraday_volatility']
    
    return df['final_factor'].dropna()
