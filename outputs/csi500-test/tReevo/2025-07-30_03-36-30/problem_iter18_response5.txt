```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Combine 5-Day and 10-Day Returns using a weighted average
    df['combined_return'] = df['5_day_return'] * 0.7 + df['10_day_return'] * 0.3
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20_day_open_close_range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()).abs()
    
    # 20-Day Mean Price
    df['20_day_mean_price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']
    
    # Daily Change in Amount
    df['daily_change_in_amount'] = df['amount'].diff()
    
    # Normalize Daily Change in Amount
    df['normalized_daily_change_in_amount'] = df['daily_change_in_amount'] / df['amount']
    
    # Integrate Market Sentiment
    df['momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change_in_amount']
    
    # 20-Day Volume EMA
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Volume of day t
    df['adjusted_volume'] = df['volume'] / df['20_day_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['momentum_with_sentiment'] * df['adjusted_volume'] - df['enhanced_volatility_index']
    
    return df['final_factor']
```
