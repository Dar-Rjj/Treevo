```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    # Compute 10-Day Rolling Return
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Combine 5-Day and 10-Day Returns
    df['Combined_Return'] = 0.6 * df['5D_Return'] + 0.4 * df['10D_Return']
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).sum() - df['close'].rolling(window=20).sum()
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['Volume_Adjusted_Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Adjusted_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum_with_Sentiment'] - df['Enhanced_Volatility_Index'] * df['Adjusted_Volume']
    
    return df['Final_Factor']
```
