```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5DayReturn'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10DayReturn'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Combine 5-Day and 10-Day Returns with weights
    df['CombinedReturn'] = 0.7 * df['5DayReturn'] + 0.3 * df['10DayReturn']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['VolumeAdjustedMomentum'] = df['CombinedReturn'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['HighLowSpread'] = df['high'] - df['low']
    df['OpenCloseSpread'] = (df['open'] - df['close']).abs()
    df['MeanPrice'] = (df['open'] + df['close']) / 2
    df['VolatilityIndex'] = (df['HighLowSpread'] + df['OpenCloseSpread']) / df['MeanPrice']
    
    # Integrate Market Sentiment
    df['DailyChangeAmount'] = df['amount'] - df['amount'].shift(1)
    df['NormalizedChangeAmount'] = df['DailyChangeAmount'] / df['amount']
    df['VolumeAdjustedMomentumWithSentiment'] = df['VolumeAdjustedMomentum'] + df['NormalizedChangeAmount']
    
    # Final Factor Calculation
    df['FinalFactor'] = df['VolumeAdjustedMomentumWithSentiment'] - df['VolatilityIndex']
    
    return df['FinalFactor']
```
