```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'] * df['volume']
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads
    df['combined_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Calculate Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Calculate Open-High-Low Volatility Index
    df['ohl_vol_index'] = df['combined_spread'] / df['mean_price']
    
    # Final factor: Volume-Adjusted Price Momentum with Open-High-Low Volatility Index
    df['factor_value'] = df['volume_adjusted_momentum'] * df['ohl_vol_index']
    
    return df['factor_value'].dropna()
```
