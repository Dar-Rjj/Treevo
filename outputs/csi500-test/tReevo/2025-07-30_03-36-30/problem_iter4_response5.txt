```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'] * df['volume']
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Intra-Day Volatility
    df['intra_day_volatility'] = ((df['high_low_spread'] + df['open_close_spread']) / 2) / df['mean_price']
    
    # 5-Day Moving Average of Intra-Day Volatility
    df['intra_day_volatility_ma_5'] = df['intra_day_volatility'].rolling(window=5).mean()
    
    # Adjust Combined Spread by Trend
    df['trend_adjusted_intra_day_volatility'] = (df['intra_day_volatility'] - df['intra_day_volatility_ma_5']) / df['intra_day_volatility_ma_5']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['trend_adjusted_intra_day_volatility']
    
    return df['alpha_factor'].dropna()
```
