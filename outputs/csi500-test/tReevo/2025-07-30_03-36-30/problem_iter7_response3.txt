```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = df['close'].pct_change(periods=3)
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = df['close'].pct_change(periods=5)
    
    # Weighted Average of 3-Day and 5-Day Returns
    df['weighted_return'] = df['3_day_return'] * 0.6 + df['5_day_return'] * 0.4
    
    # Compute Exponential Weighted Moving Average of Volume
    ewm_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Adjust by Exponentially Weighted Volume
    df['volume_adjusted_return'] = df['weighted_return'] * ewm_volume
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Combine Spreads and Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']).abs() / df['mean_price']
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_return'] - df['ohl_volatility_index']
    
    # Drop intermediate columns to return only the final factor
    final_factor = df['final_factor'].copy()
    return final_factor
```
