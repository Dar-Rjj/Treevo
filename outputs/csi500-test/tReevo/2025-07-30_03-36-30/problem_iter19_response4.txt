```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5d_return'] = df['close'].pct_change(periods=5)
    
    # Compute 10-Day Rolling Return
    df['10d_return'] = df['close'].pct_change(periods=10)
    
    # Combine 5-Day and 10-Day Returns with weights
    df['combined_return'] = 0.7 * df['5d_return'] + 0.3 * df['10d_return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Compute 20-Day High-Low Range
    df['20d_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Compute 20-Day Open-Close Range
    df['20d_open_close_range'] = (df['open'] - df['close']).rolling(window=20).sum()
    
    # 20-Day Mean Price
    df['20d_mean_price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['enhanced_volatility_index'] = (df['20d_high_low_range'] + df['20d_open_close_range']) / df['20d_mean_price']
    
    # Daily Change in Amount
    df['daily_amount_change'] = df['amount'].pct_change(periods=1)
    
    # Adjusted Momentum with Market Sentiment
    df['adjusted_momentum_sentiment'] = df['volume_adjusted_momentum'] + df['daily_amount_change']
    
    # Calculate 20-Day Volume EMA
    df['20d_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['normalized_volume'] = df['volume'] / df['20d_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_sentiment'] - df['enhanced_volatility_index'] * df['normalized_volume']
    
    return df['final_factor']
```
