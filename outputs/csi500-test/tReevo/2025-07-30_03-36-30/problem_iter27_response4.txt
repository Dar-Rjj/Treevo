import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Combine 5-Day and 10-Day Returns (Weighted Average)
    df['combined_return'] = 0.6 * df['5_day_return'] + 0.4 * df['10_day_return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['20_day_high_low_range'] = df['20_day_high'] - df['20_day_low']
    
    # Calculate 20-Day Open-Close Range
    df['20_day_open_close_range'] = (df['open'] - df['close']).abs().rolling(window=20).sum()
    
    # 20-Day Mean Price
    df['20_day_mean_price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']
    
    # Integrate Market Sentiment: Daily Change in Amount
    df['daily_change_in_amount'] = df['amount'].diff()
    
    # 20-Day Volume Exponential Moving Average (EMA)
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = (
        df['volume_adjusted_momentum'] 
        - df['enhanced_volatility_index'] 
        + df['daily_change_in_amount'] 
        + df['normalized_volume']
    )
    
    return df['final_factor'].dropna()
