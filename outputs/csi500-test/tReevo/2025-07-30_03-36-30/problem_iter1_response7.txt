```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Exponential Weighted Moving Average (EWMA) of Returns
    decay_factor = 0.94
    df['ewma_returns'] = df['daily_return'].ewm(alpha=(1 - decay_factor), adjust=False).mean()

    # Calculate Typical Price
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3

    # Calculate Price-Averaged Volume
    df['price_averaged_volume'] = df['volume'] * df['typical_price']

    # Final Factor Calculation
    df['final_factor'] = df['ewma_returns'] * df['price_averaged_volume']

    return df['final_factor']
```
