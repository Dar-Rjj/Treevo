```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Weighted Average of 5-Day and 10-Day Returns
    df['Combined_Return'] = (df['5D_Return'] * 0.6) + (df['10D_Return'] * 0.4)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Enhanced Open-High-Low Volatility Index
    df['20D_High'] = df['high'].rolling(window=20).max()
    df['20D_Low'] = df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['20D_Volatility_Index'] = (df['20D_High'] - df['20D_Low'] + df['20D_Open_Close_Range']) / ((df['open'] + df['close']).rolling(window=20).sum() / 40)
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Sentiment_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] * df['Normalized_Daily_Change_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Sentiment_Adjusted_Momentum'] - df['20D_Volatility_Index']
    
    return df['Final_Factor']
```
