import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Combine 5-Day and 10-Day Returns using a weighted average
    df['Combined_Return'] = (df['5D_Return'] * 0.7) + (df['10D_Return'] * 0.3)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # 20-Day High-Low Range
    df['20D_High'] = df['high'].rolling(window=20).max()
    df['20D_Low'] = df['low'].rolling(window=20).min()
    df['20D_High_Low_Range'] = df['20D_High'] - df['20D_Low']
    
    # 20-Day Open-Close Range
    df['20D_Open_Close_Range'] = abs(df['open'] - df['close']).rolling(window=20).sum()
    
    # 20-Day Mean Price
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Daily Change in Amount
    df['Daily_Change_Amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Integrate Market Sentiment
    df['Market_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']
    
    # 20-Day Volume EMA
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Market_Sentiment'] - df['Enhanced_Volatility_Index'] * df['Normalized_Volume']
    
    return df['Final_Factor'].dropna()
