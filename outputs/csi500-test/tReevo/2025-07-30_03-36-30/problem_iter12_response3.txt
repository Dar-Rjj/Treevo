import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['2D_Return'] = (df['close'].shift(1) / df['close'].shift(2)) - 1
    df['3D_Return'] = (df['close'].shift(1) / df['close'].shift(3)) - 1
    df['5D_Return'] = (df['close'].shift(1) / df['close'].shift(5)) - 1
    df['Weighted_Return'] = (0.3 * df['2D_Return']) + (0.4 * df['3D_Return']) + (0.3 * df['5D_Return'])
    df['Volume_Trend'] = df['volume'] / df['volume'].shift(1)
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['Volume_Trend']

    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['OHL_Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = (df['amount'] - df['amount'].shift(1))
    df['Normalized_Daily_Change_in_Amount'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_in_Amount']

    # Incorporate Liquidity Measures
    df['Average_Volume_10D'] = df['volume'].rolling(window=10).mean()
    df['Turnover_Ratio'] = df['volume'] / df['Average_Volume_10D']

    # Final Factor Calculation
    df['Final_Factor'] = df['Momentum_with_Sentiment'] - df['OHL_Volatility_Index'] + df['Turnover_Ratio']

    return df['Final_Factor'].dropna()
