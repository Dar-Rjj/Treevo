```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(4) / df['close']) - 1
    
    # Compute 10-Day Rolling Return
    df['10_day_return'] = (df['close'].shift(9) / df['close']) - 1
    
    # Weighted Average of 5-Day and 10-Day Returns
    df['weighted_returns'] = df['5_day_return'] * 0.6 + df['10_day_return'] * 0.4
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_returns'] * df['volume']

    # Compute 20-Day High-Low Range
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Compute 20-Day Open-Close Range
    df['20_day_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    
    # Combine 20-Day Ranges
    df['combined_ranges'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / 2
    
    # Divide by 20-Day Mean Price
    df['20_day_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = df['combined_ranges'] / df['20_day_mean_price']

    # Calculate Daily Change in Amount
    df['daily_amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Add Daily Change in Amount to Volume-Adjusted Multi-Day Price Momentum
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] + df['daily_amount_change']
    
    # Calculate 20-Day Volume EMA
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['normalized_volume'] = df['volume'] / df['20_day_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['enhanced_volatility_index'] * df['normalized_volume']
    
    return df['final_factor']
```
