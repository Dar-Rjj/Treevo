```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = df['close'].pct_change(periods=3)
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = df['close'].pct_change(periods=5)
    
    # Combine 3-Day and 5-Day Returns
    df['combined_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads and Divide by Mean Price
    mean_price = (df['open'] + df['close']) / 2
    df['ohlv_index'] = (df['high_low_spread'] + df['open_close_spread']) / (2 * mean_price)
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] - df['ohlv_index']
    
    return df['final_factor'].dropna()
```
