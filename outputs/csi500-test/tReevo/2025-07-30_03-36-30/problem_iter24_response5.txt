```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day and 50-Day Rolling Returns
    df['20_Day_Return'] = df['close'].pct_change(periods=20)
    df['50_Day_Return'] = df['close'].pct_change(periods=50)
    
    # Combine 20-Day and 50-Day Returns
    df['Weighted_Return'] = (df['20_Day_Return'] * 0.7) + (df['50_Day_Return'] * 0.3)
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20_Day_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20_Day_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    
    # Calculate 20-Day Mean Price
    df['20_Day_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Combine 20-Day Ranges
    df['Enhanced_Volatility_Index'] = (df['20_Day_High_Low_Range'] + df['20_Day_Open_Close_Range']) / df['20_Day_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'].diff()
    df['Normalized_Daily_Change_in_Amount'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Volume_Adjusted_Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_in_Amount']
    
    # Incorporate Liquidity and Trend Strength
    df['50_Day_Volume_EMA'] = df['volume'].ewm(span=50, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['50_Day_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum_with_Sentiment'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor']
```
