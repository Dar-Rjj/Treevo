```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    
    # Adjust by Volume
    volume_adjusted_momentum = close_to_open_return * df['volume']
    
    # Combine Spread and Momentum
    factor_value = high_low_spread + volume_adjusted_momentum
    
    return factor_value
```
