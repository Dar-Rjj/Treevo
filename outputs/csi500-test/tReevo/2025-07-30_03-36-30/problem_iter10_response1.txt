```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 3-Day and 5-Day Rolling Returns
    df['3D_Return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Smooth Returns with EMA
    df['3D_Smoothed_Return'] = df['3D_Return'].ewm(span=2).mean()
    df['5D_Smoothed_Return'] = df['5D_Return'].ewm(span=4).mean()
    
    # Combine 3-Day and 5-Day Smoothed Returns
    df['Combined_Return'] = 0.6 * df['3D_Smoothed_Return'] + 0.4 * df['5D_Smoothed_Return']
    
    # Multiply by Volume of day t
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['OHLC_Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    
    # Incorporate Volume Trend
    df['10D_Volume_EMA'] = df['volume'].ewm(span=10).mean()
    df['Normalized_Volume'] = df['volume'] / df['10D_Volume_EMA']
    
    # Adjust Volatility with More Price Points
    df['5D_High_Low_Range'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    df['5D_Open_Close_Range'] = abs(df['open'] - df['close']).rolling(window=5).sum()
    df['5D_Mean_Price'] = (df['open'] + df['close']).rolling(window=5).sum() / 10
    df['Adjusted_OHLC_Volatility'] = (df['5D_High_Low_Range'] + df['5D_Open_Close_Range']) / df['5D_Mean_Price']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum'] - df['Adjusted_OHLC_Volatility'] * df['Normalized_Volume']
    
    return df['Final_Factor']
```
