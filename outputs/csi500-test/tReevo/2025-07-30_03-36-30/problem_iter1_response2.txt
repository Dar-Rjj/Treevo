import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'].shift(-1) * df['volume']
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads: Average High-Low and Open-Close Spreads
    df['average_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    
    # Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = df['average_spread'] / df['mean_price']
    
    # Final Alpha Factor: Multiply Volume-Adjusted Price Momentum with Open-High-Low Volatility Index
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['ohl_volatility_index']
    
    return df['alpha_factor']
