```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Multiply by Volume of day t to get Volume-Adjusted Short-Term Price Momentum
    df['volume_adjusted_momentum'] = df['5_day_return'] * df['volume']
    
    # Calculate Simplified Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['high'] + df['low']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_in_amount'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_change'] = df['daily_change_in_amount'] / df['amount']
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['normalized_daily_change']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['volatility_index']
    
    return df['final_factor'].dropna()
```
