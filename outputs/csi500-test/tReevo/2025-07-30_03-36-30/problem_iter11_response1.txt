```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = df['close'].pct_change(periods=3)
    df['5d_return'] = df['close'].pct_change(periods=5)
    
    # Combine 3-Day and 5-Day Returns with weights
    df['combined_return'] = df['3d_return'] * 0.6 + df['5d_return'] * 0.4
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = (df['open'] - df['close']).abs()
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Market Sentiment: Daily Change in Amount
    df['daily_change_amount'] = df['amount'].diff()
    
    # Integrate Market Sentiment with Volume-Adjusted Multi-Day Price Momentum
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_amount']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['volatility_index']
    
    return df['final_factor']
```
