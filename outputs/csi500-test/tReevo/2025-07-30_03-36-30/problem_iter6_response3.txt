```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day Exponentially-Weighted Return
    df['ewm_3_day_return'] = (df['close'].shift(2) * 0.1 + df['close'].shift(1) * 0.2 + df['close'] * 0.7) / df['close'].shift(2) - 1
    
    # Compute 5-Day Exponentially-Weighted Return
    df['ewm_5_day_return'] = (df['close'].shift(4) * 0.1 + df['close'].shift(3) * 0.2 + df['close'].shift(2) * 0.2 + df['close'].shift(1) * 0.2 + df['close'] * 0.3) / df['close'].shift(4) - 1
    
    # Combine 3-Day and 5-Day Returns
    df['weighted_avg_return'] = df['ewm_3_day_return'] * 0.6 + df['ewm_5_day_return'] * 0.4
    
    # Adjust by Volume
    df['volume_adjusted_return'] = df['weighted_avg_return'] * df['volume']
    
    # Calculate Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) / (2 * df['mean_price'])
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_return'] - df['intraday_volatility']

    return df['final_factor'].dropna()
```
