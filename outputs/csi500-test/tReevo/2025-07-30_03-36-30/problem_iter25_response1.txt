import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10D_Return'] = (df['close'].shift(-9) / df['close']) - 1
    
    # Weighted Average of 5-Day and 10-Day Returns
    df['Combined_Return'] = 0.6 * df['5D_Return'] + 0.4 * df['10D_Return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20D_Open_Close_Range'] = (df['open'] - df['close']).rolling(window=20).sum()
    
    # 20-Day Mean Price
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    
    # Enhanced Open-High-Low Volatility Index
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change']
    
    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Adjusted_Volume'] = df['volume'] / df['20D_Volume_EMA']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Momentum_with_Sentiment'] * df['Adjusted_Volume'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor'].dropna()
