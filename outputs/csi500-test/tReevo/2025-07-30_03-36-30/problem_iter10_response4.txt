import pandas as pd

def heuristics_v2(df):
    # Compute 3-day rolling return
    df['3D_Return'] = df['close'].pct_change(periods=3)
    
    # Compute 5-day rolling return
    df['5D_Return'] = df['close'].pct_change(periods=5)
    
    # Compute 3-day and 5-day return variances
    df['3D_Return_Var'] = df['close'].rolling(window=3).var()
    df['5D_Return_Var'] = df['close'].rolling(window=5).var()
    
    # Weight 3-day and 5-day returns
    df['Weighted_3D_Return'] = df['3D_Return'] * (1 / df['3D_Return_Var'])
    df['Weighted_5D_Return'] = df['5D_Return'] * (1 / df['5D_Return_Var'])
    
    # Normalize weights
    total_weight = df['Weighted_3D_Return'] + df['Weighted_5D_Return']
    df['Normalized_3D_Return'] = df['Weighted_3D_Return'] / total_weight
    df['Normalized_5D_Return'] = df['Weighted_5D_Return'] / total_weight
    
    # Compute 5-day volume moving average
    df['5D_Volume_MA'] = df['volume'].rolling(window=5).mean()
    
    # Compute Volume Trend
    df['Volume_Trend'] = df['volume'] / df['5D_Volume_MA']
    
    # Combine normalized returns with volume trend
    df['Volume_Adjusted_Momentum'] = (df['Normalized_3D_Return'] + df['Normalized_5D_Return']) * df['Volume_Trend']

    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    mean_price = (df['open'] + df['close']) / 2
    df['Combined_Spread'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / 2
    df['OHL_Volatility_Index'] = df['Combined_Spread'] / mean_price
    
    # Final factor calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum'] - df['OHL_Volatility_Index']
    
    return df['Final_Factor'].dropna()
