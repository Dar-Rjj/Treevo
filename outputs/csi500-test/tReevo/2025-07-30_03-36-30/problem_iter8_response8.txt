```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) / df['close']) - 1
    
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns with weights
    df['combined_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['alpha_factor']
```
