```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Log Returns
    df['log_return_3d'] = np.log(df['close'] / df['close'].shift(2))
    df['log_return_5d'] = np.log(df['close'] / df['close'].shift(4))
    
    # Combine 3-Day and 5-Day Log Returns with weights
    df['combined_log_return'] = (df['log_return_3d'] * 0.6) + (df['log_return_5d'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_log_return'] * df['volume']
    
    # Compute High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Combine Spreads
    df['average_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = df['average_spread'] / df['mean_price']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['alpha_factor']
```
