```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) - df['close']) / df['close']

    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) - df['close']) / df['close']

    # Combine 3-Day and 5-Day Returns with weights
    df['weighted_return'] = (0.6 * df['3_day_return'] + 0.4 * df['5_day_return']) * df['volume']

    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])

    # Combine Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = df['avg_spread'] / df['mean_price']

    # Final Factor Calculation
    df['factor_value'] = df['weighted_return'] - df['volatility_index']

    return df['factor_value']
```
