```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['10_day_return'] = (df['close'].shift(-9) / df['close']) - 1
    df['weighted_return'] = (df['5_day_return'] * 0.6) + (df['10_day_return'] * 0.4)
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20_day_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20_day_open_close_range'] = (df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min())
    df['20_day_mean_price'] = (df['open'] + df['close']).rolling(window=20).mean() / 2
    df['enhanced_volatility_index'] = (df['20_day_high_low_range'] + df['20_day_open_close_range']) / df['20_day_mean_price']

    # Integrate Market Sentiment
    df['daily_change_in_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['sentiment_adjusted_momentum'] = df['volume_adjusted_momentum'] + df['daily_change_in_amount']

    # Incorporate Liquidity and Trend Strength
    df['20_day_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_volume'] = df['volume'] / df['20_day_volume_ema']

    # Final Factor Calculation
    df['final_factor'] = df['sentiment_adjusted_momentum'] - df['enhanced_volatility_index'] * df['adjusted_volume']
    
    return df['final_factor']
```
