```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Compute 20-Day Rolling Return
    df['20_day_return'] = (df['close'] / df['close'].shift(19)) - 1
    
    # Combine 5-Day and 20-Day Returns
    df['combined_return'] = (df['5_day_return'] * 0.6) + (df['20_day_return'] * 0.4)
    
    # Volume-Adjusted Long-Term Price Momentum
    df['adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate Simplified Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['high'] + df['low']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_change'] = df['daily_change_amount'] / df['amount']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['adjusted_momentum'] - df['volatility_index'] + df['normalized_daily_change']
    
    return df['alpha_factor']
```
