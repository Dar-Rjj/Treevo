import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Combine 5-Day and 10-Day Returns with weights
    df['combined_return'] = (df['5_day_return'] * 0.7) + (df['10_day_return'] * 0.3)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Simplified Market Sentiment
    df['daily_change_amount'] = df['amount'].diff()
    df['adjusted_daily_change_amount'] = df['daily_change_amount'] / df['amount']
    
    # Integrate Simplified Market Sentiment into Volume-Adjusted Momentum
    df['momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['adjusted_daily_change_amount']
    
    # Final Factor Calculation
    df['final_factor'] = df['momentum_with_sentiment'] - df['volatility_index']
    
    # Return the final factor as a Series
    return df['final_factor'].dropna()
