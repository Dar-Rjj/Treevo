```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-Day and 20-Day Rolling Returns
    df['10DayReturn'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['20DayReturn'] = (df['close'].shift(-19) - df['close']) / df['close']
    
    # Combine 10-Day and 20-Day Returns using a weighted average
    df['CombinedReturn'] = 0.6 * df['10DayReturn'] + 0.4 * df['20DayReturn']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['VolumeAdjustedMomentum'] = df['CombinedReturn'] * df['volume']
    
    # Calculate High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['OpenCloseSpread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads and divide by Mean Price
    df['MeanPrice'] = (df['open'] + df['close']) / 2
    df['OpenHighLowVolatility'] = (df['HighLowSpread'] + df['OpenCloseSpread']) / df['MeanPrice']
    
    # Integrate Simplified Market Sentiment
    df['DailyChangeInAmount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    df['AdjustedMomentum'] = df['VolumeAdjustedMomentum'] + df['DailyChangeInAmount']
    
    # Final Factor Calculation
    df['FinalFactor'] = df['AdjustedMomentum'] - df['OpenHighLowVolatility']
    
    return df['FinalFactor'].dropna()
```
