```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Adjusted Price Momentum
    df['volume_adjusted_momentum'] = df['daily_return'].shift(-1) * df['volume']
    
    # Compute High-Low Spread and Open-Close Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Calculate the Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Combine Spreads
    df['combined_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    df['intra_day_volatility'] = df['combined_spread'] / df['mean_price']
    
    # Calculate 5-Day Moving Average of Intra-Day Volatility
    df['intra_day_volatility_5d_ma'] = df['intra_day_volatility'].rolling(window=5).mean()
    
    # Factor in Intra-Day Volatility Trend
    df['volatility_trend'] = (df['intra_day_volatility'] - df['intra_day_volatility_5d_ma']) / df['intra_day_volatility_5d_ma']
    
    # Final Alpha Factor: Volume-Adjusted Momentum adjusted by Intra-Day Volatility Trend
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['volatility_trend']
    
    return df['alpha_factor']
```
