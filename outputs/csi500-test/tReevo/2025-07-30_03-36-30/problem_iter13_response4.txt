```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day Rolling Return
    df['3_day_return'] = df['close'].pct_change(periods=3)
    # Compute 5-Day Rolling Return
    df['5_day_return'] = df['close'].pct_change(periods=5)
    
    # Combine 3-Day and 5-Day Returns using a weighted average
    df['weighted_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Combine Spreads
    df['combined_spread'] = df['high_low_spread'] + df['open_close_spread']
    # Divide by Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = df['combined_spread'] / df['mean_price']
    
    # Integrate Market Sentiment
    # Calculate Daily Change in Amount
    df['daily_amount_change'] = df['amount'].diff()
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    df['momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_amount_change']
    
    # Final Factor Calculation
    # Subtract Open-High-Low Volatility Index from Adjusted Momentum
    df['final_factor'] = df['momentum_with_sentiment'] - df['ohl_volatility_index']
    
    return df['final_factor']
```
