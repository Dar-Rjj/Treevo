```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['5D_EMA'] = df['5D_Return'].ewm(span=10, adjust=False).mean()
    df['Weighted_Return'] = (df['5D_Return'] * (1 - df['5D_EMA'])) + (df['10D_Return'] * df['5D_EMA'])
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = abs(df['open'] - df['close']).rolling(window=20).sum()
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Daily_Change_Amount'] = df['Daily_Change_Amount'] / df['amount']
    df['Volume_Adjusted_Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_Amount']

    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']
    df['Final_Factor'] = df['Volume_Adjusted_Momentum_with_Sentiment'] * df['Normalized_Volume']

    # Incorporate Adaptive Volatility Measures
    df['20D_Volatility_EMA'] = df['Enhanced_Volatility_Index'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volatility'] = df['Enhanced_Volatility_Index'] / df['20D_Volatility_EMA']
    df['Final_Factor'] -= df['Normalized_Volatility']

    # Drop rows with NaN values created by the rolling window calculations
    df.dropna(inplace=True)

    return df['Final_Factor']
```
