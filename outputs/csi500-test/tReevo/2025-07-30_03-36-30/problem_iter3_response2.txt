import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Price Momentum
    df['Daily_Return'] = df['close'].pct_change()
    df['Volume_Adjusted_Momentum'] = df['Daily_Return'].shift(1) * df['volume']
    
    # Calculate Enhanced Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = np.abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Average_Spread'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / 2
    df['Volatility_Index'] = df['Average_Spread'] / df['Mean_Price']
    
    # Integrate Longer-Term Trends
    df['20_Day_EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['50_Day_EMA_Close'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Use Adaptive Moving Averages
    df['20_Day_EMA_Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['50_Day_EMA_Volume'] = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Leverage Intra-Day Price Movements
    df['Intra_Day_Momentum'] = df['high'] - df['low']
    df['Intra_Day_Reversal'] = df['high'] - df['low'].shift(1)
    
    # Incorporate Liquidity Metrics (Assuming free_float_shares, ask, bid are available in the DataFrame)
    df['Turnover_Ratio'] = df['volume'] / df['free_float_shares']
    df['Bid_Ask_Spread'] = df['ask'] - df['bid']
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] +
        df['Volatility_Index'] +
        (df['20_Day_EMA_Close'] - df['50_Day_EMA_Close']) +
        (df['20_Day_EMA_Volume'] - df['50_Day_EMA_Volume']) +
        df['Intra_Day_Momentum'] +
        df['Intra_Day_Reversal'] +
        df['Turnover_Ratio'] +
        df['Bid_Ask_Spread']
    )
    
    return df['Alpha_Factor']
