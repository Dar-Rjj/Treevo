```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['3D_Return'] = (df['close'].shift(-2) - df['close']) / df['close']
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    
    df['3D_Weight'] = df['volume'].shift(-1) / (df['volume'].shift(-2) + df['volume'].shift(-1))
    df['5D_Weight'] = df['volume'].shift(-1) / (df['volume'].shift(-4) + df['volume'].shift(-1))
    
    df['Volume_Adjusted_Momentum'] = (df['3D_Return'] * df['3D_Weight'] + 
                                      df['5D_Return'] * df['5D_Weight']) * df['volume']
    
    # Calculate Enhanced Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Combined_Spreads'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    
    df['10D_Historical_Volatility'] = df['close'].rolling(window=10).std()
    df['Enhanced_Volatility_Index'] = (df['Combined_Spreads'] + df['10D_Historical_Volatility']) / 2
    
    # Calculate Liquidity Measure
    df['10D_Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Liquidity_Measure'] = (df['volume'] - df['10D_Average_Volume']) / df['10D_Average_Volume']
    
    # Final Factor Calculation
    df['Alpha_Factor'] = (df['Volume_Adjusted_Momentum'] - df['Enhanced_Volatility_Index']) * df['Liquidity_Measure']
    
    return df['Alpha_Factor']
```
