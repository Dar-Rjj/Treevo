```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume']) / df['volume']
    
    # Short-Term Trend
    df['7_day_ma'] = df['volume_weighted_price'].rolling(window=7).mean()
    df['20_day_ma'] = df['volume_weighted_price'].rolling(window=20).mean()
    df['short_term_trend'] = df['7_day_ma'] - df['20_day_ma']
    
    # Price Reversal Indicator
    df['price_reversal'] = ((df['high'] - df['low']) > 2 * (df['close'] - df['open'])).astype(int)
    
    # Combine Momentum and Reversal
    df['alpha_factor'] = df['short_term_trend'] * df['price_reversal']
    
    return df['alpha_factor']
```
