```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day and 5-Day Rolling Returns
    df['3D_Return'] = (df['close'].pct_change(3)).shift(-2)
    df['5D_Return'] = (df['close'].pct_change(5)).shift(-4)

    # Combine 3-Day and 5-Day Returns with Weights
    df['Combined_Return'] = df['3D_Return'] * 0.6 + df['5D_Return'] * 0.4

    # Volume-Adjusted Short-Term Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']

    # Enhanced Open-High-Low Volatility Index
    df['10D_High'] = df['high'].rolling(window=10).max()
    df['10D_Low'] = df['low'].rolling(window=10).min()
    df['10D_Open'] = df['open'].rolling(window=10).mean()
    df['10D_Close'] = df['close'].rolling(window=10).mean()

    df['10D_High_Low_Range'] = df['10D_High'] - df['10D_Low']
    df['10D_Open_Close_Range'] = df['10D_Open'] - df['10D_Close']

    df['10D_Mean_Price'] = (df['10D_Open'] + df['10D_Close']) / 2
    df['Enhanced_Volatility_Index'] = (df['10D_High_Low_Range'] + df['10D_Open_Close_Range']) / df['10D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'].diff()
    df['Normalized_Daily_Change_in_Amount'] = df['Daily_Change_in_Amount'] / df['amount']

    df['Market_Sentiment_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change_in_Amount']

    # Incorporate Liquidity and Trend Strength
    df['10D_Volume_EMA'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Adjusted_Volume'] = df['volume'] / df['10D_Volume_EMA']

    # Final Factor Calculation
    df['Final_Factor'] = df['Market_Sentiment_Adjusted_Momentum'] * df['Adjusted_Volume'] - df['Enhanced_Volatility_Index']

    return df['Final_Factor']
```
