import pandas as pd

def heuristics_v2(data):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    data['5D_Return'] = (data['close'].shift(-4) - data['close']) / data['close']
    data['10D_Return'] = (data['close'].shift(-9) - data['close']) / data['close']
    data['Weighted_Return'] = 0.6 * data['5D_Return'] + 0.4 * data['10D_Return']
    data['Volume_Adjusted_Momentum'] = data['Weighted_Return'] * data['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    data['20D_High_Low_Range'] = data['high'].rolling(window=20).max() - data['low'].rolling(window=20).min()
    data['20D_Open_Close_Range'] = (data['open'].rolling(window=20).sum() + data['close'].rolling(window=20).sum()) / 20
    data['20D_Mean_Price'] = (data['open'] + data['close']).rolling(window=20).mean() / 2
    data['Enhanced_Volatility_Index'] = (data['20D_High_Low_Range'] + data['20D_Open_Close_Range']) / data['20D_Mean_Price']

    # Integrate Market Sentiment
    data['Daily_Change_in_Amount'] = data['amount'] - data['amount'].shift(1)
    data['Normalized_Daily_Change_in_Amount'] = data['Daily_Change_in_Amount'] / data['amount']
    data['Momentum_with_Sentiment'] = data['Volume_Adjusted_Momentum'] + data['Normalized_Daily_Change_in_Amount']

    # Incorporate Liquidity and Trend Strength
    data['20D_Volume_EMA'] = data['volume'].ewm(span=20, adjust=False).mean()
    data['Normalized_Volume'] = data['volume'] / data['20D_Volume_EMA']

    # Final Factor Calculation
    data['Final_Factor'] = (data['Momentum_with_Sentiment'] + data['Normalized_Volume']) - data['Enhanced_Volatility_Index']
    
    return data['Final_Factor'].dropna()
