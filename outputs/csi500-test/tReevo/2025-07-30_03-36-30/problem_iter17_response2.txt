```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10D_Return'] = (df['close'].shift(-9) - df['close']) / df['close']
    df['Weighted_Return'] = 0.7 * df['5D_Return'] + 0.3 * df['10D_Return']
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    df['20D_Mean_Price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'].diff()
    df['Normalized_Daily_Change'] = df['Daily_Change_in_Amount'] / df['amount']
    df['Momentum_with_Sentiment'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Daily_Change']

    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Liquidity_Adjusted_Volume'] = df['volume'] / df['20D_Volume_EMA']
    df['Final_Factor'] = df['Momentum_with_Sentiment'] * df['Liquidity_Adjusted_Volume'] - df['Enhanced_Volatility_Index']

    return df['Final_Factor']
```
