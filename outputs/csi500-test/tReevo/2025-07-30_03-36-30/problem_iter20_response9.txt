```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    five_day_return = (df['close'].shift(-4) / df['close']).shift(4)
    ten_day_return = (df['close'].shift(-9) / df['close']).shift(9)

    # Combine 5-Day and 10-Day Returns using a weighted average
    combined_return = (five_day_return * 0.7 + ten_day_return * 0.3)

    # Multiply by volume to get Volume-Adjusted Multi-Day Price Momentum
    volume_adjusted_momentum = combined_return * df['volume']

    # Calculate Open-High-Low Volatility Index
    high_low_spread = df['high'] - df['low']
    open_close_spread = abs(df['open'] - df['close'])
    mean_price = (df['open'] + df['close']) / 2
    ohl_volatility = (high_low_spread + open_close_spread) / mean_price

    # Integrate Simplified Market Sentiment
    daily_change_in_amount = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    adjusted_momentum = volume_adjusted_momentum + daily_change_in_amount

    # Final Factor Calculation
    final_factor = adjusted_momentum - ohl_volatility

    return final_factor
```
