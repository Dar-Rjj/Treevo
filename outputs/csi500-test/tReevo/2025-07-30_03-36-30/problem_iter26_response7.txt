```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day and 10-Day Rolling Returns
    df['5d_return'] = (df['close'].shift(-4) - df['close']) / df['close']
    df['10d_return'] = (df['close'].shift(-9) - df['close']) / df['close']
    
    # Combine 5-Day and 10-Day Returns with weighted average
    df['weighted_return'] = 0.6 * df['5d_return'] + 0.4 * df['10d_return']
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate 20-Day High-Low Range
    df['20d_high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 20-Day Open-Close Range
    df['20d_open_close_range'] = df['open'].rolling(window=20).max() - df['close'].rolling(window=20).min()
    
    # Combine 20-Day Ranges
    df['20d_mean_price'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 40
    df['enhanced_volatility_index'] = (df['20d_high_low_range'] + df['20d_open_close_range']) / df['20d_mean_price']
    
    # Calculate Daily Change in Amount
    df['daily_change_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    
    # Add to Volume-Adjusted Multi-Day Price Momentum
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_amount']
    
    # Calculate 20-Day Volume EMA
    df['20d_volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Normalize Volume of day t
    df['normalized_volume'] = df['volume'] / df['20d_volume_ema']
    
    # Final Factor Calculation
    df['final_factor'] = (df['adjusted_momentum_with_sentiment'] * df['normalized_volume']) - df['enhanced_volatility_index']
    
    return df['final_factor'].dropna()
```
