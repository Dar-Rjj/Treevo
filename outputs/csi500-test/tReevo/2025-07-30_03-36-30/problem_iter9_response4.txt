```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Multi-Day Price Momentum
    # Compute 3-Day Rolling Return
    df['3_day_return'] = (df['close'].shift(-2) / df['close']) - 1
    # Compute 5-Day Rolling Return
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Weighted Average of 3-Day and 5-Day Returns
    df['weighted_returns'] = 0.6 * df['3_day_return'] + 0.4 * df['5_day_return']
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['weighted_returns'] * df['volume']

    # Open-High-Low Volatility Index
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    # Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Average High-Low and Open-Close Spreads
    df['avg_spread'] = (df['high_low_spread'] + df['open_close_spread']) / 2
    # Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Divide by Mean Price to get the volatility index
    df['volatility_index'] = df['avg_spread'] / df['mean_price']

    # Final Factor Calculation
    # Subtract Open-High-Low Volatility Index from Volume-Adjusted Multi-Day Price Momentum
    df['factor'] = df['volume_adjusted_momentum'] - df['volatility_index']
    
    return df['factor']
```
