```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns
    df['weighted_return'] = (df['3d_return'] * 0.6) + (df['5d_return'] * 0.4)
    
    # Multiply by Volume of day t
    df['volume_weighted_momentum'] = df['weighted_return'] * df['volume']
    
    # Calculate Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['momentum_with_sentiment'] = df['volume_weighted_momentum'] + df['daily_change_amount']
    
    # Adjust for Cumulative Market Impact
    df['10d_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_deviation'] = (df['volume'] - df['10d_avg_volume']) / df['10d_avg_volume']
    df['adjusted_momentum'] = df['momentum_with_sentiment'] * (1 + df['volume_deviation'])
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['intraday_volatility']
    
    return df['final_factor']
```
