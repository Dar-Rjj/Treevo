```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 3-Day and 5-Day Rolling Returns
    df['3_day_return'] = (df['close'] / df['close'].shift(2)) - 1
    df['5_day_return'] = (df['close'] / df['close'].shift(4)) - 1
    
    # Combine 3-Day and 5-Day Returns into a weighted average
    df['combined_return'] = (df['3_day_return'] * 0.6) + (df['5_day_return'] * 0.4)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = abs(df['open'] - df['close'])
    
    # Calculate Mean Price
    df['mean_price'] = (df['open'] + df['close']) / 2
    
    # Open-High-Low Volatility Index
    df['ohlv_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Daily Change in Amount
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    
    # Integrate Market Sentiment
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_amount']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['ohlv_index']
    
    return df['final_factor']
```
