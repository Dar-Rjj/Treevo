```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns with Weights
    df['combined_return'] = 0.6 * df['3d_return'] + 0.4 * df['5d_return']
    
    # Multiply by Volume of day t
    df['volume_weighted_momentum'] = df['combined_return'] * df['volume']

    # Calculate Refined Intraday Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Normalized Market Sentiment
    df['daily_change_amount'] = (df['amount'] - df['amount'].shift(1)) / df['amount']
    df['adjusted_momentum'] = df['volume_weighted_momentum'] + df['daily_change_amount']
    
    # Final Factor Calculation
    df['alpha_factor'] = df['adjusted_momentum'] - df['intraday_volatility']
    
    return df['alpha_factor']
```
