```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Multi-Day Price Momentum
    df['5D_Return'] = df['close'].pct_change(5)
    df['10D_Return'] = df['close'].pct_change(10)
    df['Weighted_Return'] = df['5D_Return'] * 0.6 + df['10D_Return'] * 0.4
    df['Volume_Adjusted_Momentum'] = df['Weighted_Return'] * df['volume']

    # Calculate Enhanced Open-High-Low Volatility Index
    df['20D_High_Low_Range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['20D_Open_Close_Range'] = (df['open'].rolling(window=20).sum() + df['close'].rolling(window=20).sum()) / 20
    df['20D_Mean_Price'] = (df['open'] + df['close']).rolling(window=20).sum() / 40
    df['Enhanced_Volatility_Index'] = (df['20D_High_Low_Range'] + df['20D_Open_Close_Range']) / df['20D_Mean_Price']

    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'].diff()
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Daily_Change_Amount']

    # Incorporate Liquidity and Trend Strength
    df['20D_Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Normalized_Volume'] = df['volume'] / df['20D_Volume_EMA']

    # Final Factor Calculation
    df['Final_Factor'] = df['Adjusted_Momentum'] - df['Enhanced_Volatility_Index']
    
    return df['Final_Factor']
```
