```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 20-Day Rolling Returns
    df['5_day_return'] = (df['close'].shift(-4) / df['close']) - 1
    df['20_day_return'] = (df['close'].shift(-19) / df['close']) - 1
    
    # Combine 5-Day and 20-Day Returns with Weights
    df['combined_return'] = 0.6 * df['5_day_return'] + 0.4 * df['20_day_return']
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['high'] + df['low']) / 2
    df['volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['normalized_daily_change'] = df['daily_change_amount'] / df['amount']
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] + df['normalized_daily_change']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum'] - df['volatility_index']
    
    return df['final_factor'].dropna()
```
