```python
import pandas as pd

def heuristics_v2(df):
    # Compute 3-Day and 5-Day Rolling Returns
    df['3d_return'] = (df['close'].shift(-2) / df['close']) - 1
    df['5d_return'] = (df['close'].shift(-4) / df['close']) - 1
    
    # Combine 3-Day and 5-Day Returns
    df['combined_return'] = (0.6 * df['3d_return'] + 0.4 * df['5d_return'])
    
    # Multiply by Volume of day t
    df['volume_adjusted_momentum'] = df['combined_return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = abs(df['open'] - df['close'])
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['ohl_volatility_index'] = (df['high_low_spread'] + df['open_close_spread']) / df['mean_price']
    
    # Integrate Market Sentiment
    df['daily_change_in_amount'] = df['amount'] - df['amount'].shift(1)
    df['adjusted_momentum_with_sentiment'] = df['volume_adjusted_momentum'] + df['daily_change_in_amount']
    
    # Final Factor Calculation
    df['final_factor'] = df['adjusted_momentum_with_sentiment'] - df['ohl_volatility_index']
    
    return df['final_factor']
```
