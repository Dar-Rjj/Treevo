```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 20-Day Rolling Returns
    df['5D_Return'] = (df['close'].shift(-4) / df['close']) - 1
    df['20D_Return'] = (df['close'].shift(-19) / df['close']) - 1
    
    # Combine 5-Day and 20-Day Returns with Weights
    df['Combined_Return'] = (df['5D_Return'] * 0.6) + (df['20D_Return'] * 0.4)
    
    # Volume-Adjusted Long-Term Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate Simplified Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Mean_Price'] = (df['high'] + df['low']) / 2
    df['Simplified_Volatility'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Normalized_Sentiment'] = df['Daily_Change_Amount'] / df['amount']
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum'] + df['Normalized_Sentiment'] - df['Simplified_Volatility']
    
    return df['Final_Factor'].dropna()
```
