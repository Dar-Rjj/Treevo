```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Absolute Difference
    open_close_diff = (df['open'] - df['close']).abs()
    
    # Sum Range and Difference for Volatility Indicator
    volatility_indicator = high_low_range + open_close_diff
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change()
    
    # Combine Volatility and Momentum
    combined_factor = price_momentum - volatility_indicator / df['close']
    
    # Volume-Weight the Combined Factor
    volume_weighted_factor = combined_factor * df['volume']
    
    # Integrate Intra-Day Price Movements
    intra_day_range = df['high'] - df['low']
    intra_day_volatility = intra_day_range / df['open']
    
    # Adjust for Intra-Day Volatility
    adjusted_factor = volume_weighted_factor - intra_day_volatility
    
    # Calculate Volume-Weighted Daily Return
    daily_return = df['close'].pct_change()
    volume_weighted_daily_return = daily_return * df['volume']
    
    # Integrate Volume-Weighted Daily Return
    final_factor = adjusted_factor + volume_weighted_daily_return
    
    return final_factor
```
