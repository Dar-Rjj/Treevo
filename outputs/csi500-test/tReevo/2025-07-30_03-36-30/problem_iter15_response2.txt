import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day and 10-Day Rolling Returns
    df['5D_Return'] = (df['close'] / df['close'].shift(4)) - 1
    df['10D_Return'] = (df['close'] / df['close'].shift(9)) - 1
    
    # Combine 5-Day and 10-Day Returns using a weighted average
    df['Combined_Return'] = (df['5D_Return'] * 0.7) + (df['10D_Return'] * 0.3)
    
    # Volume-Adjusted Multi-Day Price Momentum
    df['Volume_Adjusted_Momentum'] = df['Combined_Return'] * df['volume']
    
    # Calculate Open-High-Low Volatility Index
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Spread'] = df['open'] - df['close']
    df['Mean_Price'] = (df['open'] + df['close']) / 2
    df['Volatility_Index'] = (df['High_Low_Spread'] + df['Open_Close_Spread']) / df['Mean_Price']
    
    # Integrate Market Sentiment
    df['Daily_Change_in_Amount'] = df['amount'] - df['amount'].shift(1)
    df['Adjusted_Market_Sentiment'] = df['Daily_Change_in_Amount'] / df['amount']
    
    # Final Factor Calculation
    df['Alpha_Factor'] = df['Volume_Adjusted_Momentum'] - df['Volatility_Index'] + df['Adjusted_Market_Sentiment']
    
    return df['Alpha_Factor']
