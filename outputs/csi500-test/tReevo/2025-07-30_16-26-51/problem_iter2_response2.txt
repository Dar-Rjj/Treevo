```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate 20-Day Simple Moving Average (SMA) of Daily Returns
    df['sma_20'] = df['daily_returns'].rolling(window=20).mean()
    
    # Calculate 20-Day Exponential Moving Average (EMA) of Daily Returns
    df['ema_20'] = df['daily_returns'].ewm(span=20, adjust=False).mean()
    
    # Calculate Daily Logarithmic Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 5-Day Realized Volatility
    df['realized_volatility'] = df['log_returns'].rolling(window=5).std() * np.sqrt(252)
    
    # Calculate Volume Weighted Return
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate 5-Day Moving Sum of Volume
    df['volume_sum_5'] = df['volume'].rolling(window=5).sum()
    
    # Combine SMA and EMA
    df['sma_ema_combined'] = (df['sma_20'] * df['ema_20']) / df['realized_volatility']
    
    # Adjust for Volume Dynamics
    df['sma_ema_vol_adjusted'] = df['sma_ema_combined'] * df['volume_weighted_return'] / df['volume_sum_5']
    
    # Calculate 5-Day Moving Sum of Price Range
    df['price_range'] = df['high'] - df['low']
    df['price_range_sum_5'] = df['price_range'].rolling(window=5).sum()
    
    # Normalize by 5-Day Moving Sum of Volume
    df['normalized_price_range'] = df['price_range_sum_5'] / df['volume_sum_5']
    
    # Final Factor
    df['final_factor'] = df['sma_ema_vol_adjusted'] * df['normalized_price_range'] * df['volume_weighted_return']
    
    return df['final_factor'].dropna()
```
