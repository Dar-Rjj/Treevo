import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] / df['open'].shift(1) - 1) * df['volume']

    # Short-Term Lookback (5 days)
    short_ma = df['Volume_Weighted_Return'].rolling(window=5).mean()
    short_std = df['Volume_Weighted_Return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    mid_ma = df['Volume_Weighted_Return'].rolling(window=10).mean()
    mid_std = df['Volume_Weighted_Return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    long_ma = df['Volume_Weighted_Return'].rolling(window=20).mean()
    long_std = df['Volume_Weighted_Return'].rolling(window=20).std()

    # Combine Multi-Period Volatilities
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    combined_volatility = (short_std * short_weight + mid_std * mid_weight + long_std * long_weight)

    # Incorporate Dynamic Market Trend Adjustments
    close_5_ma = df['close'].rolling(window=5).mean()
    close_20_ma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (close_5_ma > close_20_ma).astype(int)

    # Adjust Combined Factor by Price Trend Indicator
    adjusted_factor = (short_ma + mid_ma + long_ma) / combined_volatility * price_trend_indicator

    # Final Alpha Factor
    final_alpha_factor = adjusted_factor
    return final_alpha_factor
