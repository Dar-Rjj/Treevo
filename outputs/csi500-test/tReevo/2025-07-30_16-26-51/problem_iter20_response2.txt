```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).mean()
    df['5_day_STD'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).mean()
    df['10_day_STD'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).mean()
    df['20_day_STD'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['5_day_STD'] + 
                                 mid_term_weight * df['10_day_STD'] + 
                                 long_term_weight * df['20_day_STD'])
    
    # Incorporate Momentum
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    df['Price_Trend_Indicator'] = np.where(df['5_day_EMA'] > df['20_day_EMA'], 1, -1)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'] / df['Combined_Volatility']) * df['Price_Trend_Indicator']
    
    return df['Alpha_Factor']
```
