```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Volume-Weighted Close-to-Open Return
    data['Close_to_Open_Return'] = (data['close'].shift(-1) - data['open']) / data['open']
    data['Volume_Weighted_Return'] = data['Close_to_Open_Return'] * data['volume']

    # Adaptive Lookback Periods
    short_lookback_days = 5
    mid_lookback_days = 10
    long_lookback_days = 20

    # Short-Term Lookback
    data['Short_SMA'] = data['Volume_Weighted_Return'].rolling(window=short_lookback_days).mean()
    data['Short_STD'] = data['Volume_Weighted_Return'].rolling(window=short_lookback_days).std()

    # Mid-Term Lookback
    data['Mid_SMA'] = data['Volume_Weighted_Return'].rolling(window=mid_lookback_days).mean()
    data['Mid_STD'] = data['Volume_Weighted_Return'].rolling(window=mid_lookback_days).std()

    # Long-Term Lookback
    data['Long_SMA'] = data['Volume_Weighted_Return'].rolling(window=long_lookback_days).mean()
    data['Long_STD'] = data['Volume_Weighted_Return'].rolling(window=long_lookback_days).std()

    # Combine Multi-Period Volatilities with Weights
    short_volatility_weight = 0.4
    mid_volatility_weight = 0.3
    long_volatility_weight = 0.3

    data['Combined_Volatility'] = (short_volatility_weight * data['Short_STD'] +
                                  mid_volatility_weight * data['Mid_STD'] +
                                  long_volatility_weight * data['Long_STD'])

    # Incorporate Dynamic Market Trend Adjustments
    data['5-Day_EMA'] = data['close'].ewm(span=5, adjust=False).mean()
    data['20-Day_EMA'] = data['close'].ewm(span=20, adjust=False).mean()
    data['Price_Trend_Indicator'] = data['5-Day_EMA'] - data['20-Day_EMA']

    # Adjust Combined Factor by Price Trend Indicator
    data['Adjusted_Combined_Volatility'] = data['Combined_Volatility'] + data['Price_Trend_Indicator']

    # Final Alpha Factor
    data['Alpha_Factor'] = (data['Volume_Weighted_Return'] / data['Adjusted_Combined_Volatility']).shift(1)

    return data['Alpha_Factor'].dropna()
```
