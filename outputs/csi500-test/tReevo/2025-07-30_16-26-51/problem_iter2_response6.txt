```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum Component
    df['daily_diff'] = df['close'] - df['close'].shift(1)
    decay_rate = 0.9
    df['momentum'] = df['daily_diff'].ewm(alpha=(1 - decay_rate), adjust=False).mean()

    # Calculate Overbought or Oversold Pressure
    sma_20 = df['close'].rolling(window=20).mean()
    df['above_sma'] = (df['close'] > sma_20).rolling(window=20).sum()
    df['below_sma'] = (df['close'] < sma_20).rolling(window=20).sum()
    df['overbought'] = df['above_sma'] / 20
    df['oversold'] = df['below_sma'] / 20
    df['overbought'] = np.clip(df['overbought'], 0.1, 0.9)  # Avoid extreme values
    df['oversold'] = np.clip(df['oversold'], 0.1, 0.9)  # Avoid extreme values

    # Incorporate Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    df['momentum_vol_adj'] = df['momentum'] / volatility
    df['overbought_vol_adj'] = df['overbought'] / volatility
    df['oversold_vol_adj'] = df['oversold'] / volatility

    # Incorporate Trend Strength
    sma_50 = df['close'].rolling(window=50).mean()
    df['trend_strength'] = sma_50 - sma_20

    # Incorporate Liquidity
    avg_volume = df['volume'].rolling(window=20).mean()
    df['combined_factor'] = (df['momentum_vol_adj'] * 0.7 + 
                             (df['oversold_vol_adj'] - df['overbought_vol_adj']) * 0.1 + 
                             df['trend_strength'] * 0.2)
    df['final_factor'] = df['combined_factor'] * avg_volume

    return df['final_factor']
```
