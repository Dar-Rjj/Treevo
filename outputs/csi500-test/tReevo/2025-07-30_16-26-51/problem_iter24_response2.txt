```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    lookbacks = [5, 10, 20]
    smas = {}
    sds = {}
    
    for lookback in lookbacks:
        smas[lookback] = df['volume_weighted_return'].rolling(window=lookback).mean()
        sds[lookback] = df['volume_weighted_return'].rolling(window=lookback).std()
    
    # Combine Multi-Period Volatilities
    combined_volatility = 0.4 * sds[5] + 0.3 * sds[10] + 0.3 * sds[20]
    
    # Incorporate Dynamic Market Trend Adjustments
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    price_trend_indicator = (ma_5 > ma_20).astype(int) - (ma_5 < ma_20).astype(int)
    
    bullish_adjustment = 1.2
    neutral_adjustment = 1.0
    bearish_adjustment = 0.8
    
    trend_adjustment = (
        (price_trend_indicator == 1) * bullish_adjustment +
        (price_trend_indicator == 0) * neutral_adjustment +
        (price_trend_indicator == -1) * bearish_adjustment
    )
    
    # Combine Momentum Signals
    combined_momentum_signal = 0.5 * smas[5] + 0.3 * smas[10] + 0.2 * smas[20]
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum_signal * trend_adjustment) / combined_volatility
    
    return alpha_factor
```
