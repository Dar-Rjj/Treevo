```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate High-Low Momentum (sum of last 5 days' High-Low Price Ratios)
    df['High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=5).sum()
    
    # Calculate Logarithmic Returns
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Price Volatility (Standard Deviation of Log Returns over the last 20 days)
    df['Price_Volatility'] = df['Log_Return'].rolling(window=20).std()
    
    # Smooth Volume Changes (Exponential Moving Average with a span of 20 days)
    df['Smoothed_Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Volume Correction Factor (Volume of day t / Average Volume of past 20 days)
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Correction_Factor'] = df['volume'] / df['Avg_Volume_20']
    
    # Adjust High-Low Momentum by Volatility and Volume
    df['Adjusted_High_Low_Momentum'] = (df['High_Low_Momentum'] / df['Price_Volatility']) * df['Volume_Correction_Factor']
    
    # Incorporate Smoothed Volume Change
    df['Combined_Factor'] = df['Adjusted_High_Low_Momentum'] * (df['Smoothed_Volume'] / df['volume'])
    
    # Ensure Bounded and Interpretable
    df['Final_Alpha_Factor'] = df['Combined_Factor'].clip(lower=-3, upper=3)
    
    return df['Final_Alpha_Factor']
```
