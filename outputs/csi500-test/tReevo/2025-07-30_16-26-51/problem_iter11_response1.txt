```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['SMA_5_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['STD_5_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['SMA_10_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['STD_10_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['SMA_20_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['STD_20_Volume_Weighted_Return'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3
    
    df['Combined_Volatility'] = (
        short_term_volatility_weight * df['STD_5_Volume_Weighted_Return'] +
        mid_term_volatility_weight * df['STD_10_Volume_Weighted_Return'] +
        long_term_volatility_weight * df['STD_20_Volume_Weighted_Return']
    )
    
    # Incorporate Dynamic Market Trend Adjustments
    df['SMA_5_Close'] = df['close'].rolling(window=5).mean()
    df['SMA_20_Close'] = df['close'].rolling(window=20).mean()
    
    df['Price_Trend_Indicator'] = np.where(
        df['SMA_5_Close'] > df['SMA_20_Close'], 1.2, 0.8
    )
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        (df['SMA_5_Volume_Weighted_Return'] + df['SMA_10_Volume_Weighted_Return'] + df['SMA_20_Volume_Weighted_Return']) / 
        df['Combined_Volatility'] * df['Price_Trend_Indicator']
    )
    
    return df['Final_Alpha_Factor']
```
