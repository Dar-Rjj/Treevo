```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close']) / df['close'].shift(1)
    
    # Calculate Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(1)
    
    # Weight by Volume
    weighted_close_to_open_return = close_to_open_return * df['volume']
    weighted_daily_momentum = daily_momentum * df['volume']
    
    # Combine the two factors
    factor = (weighted_close_to_open_return + weighted_daily_momentum) / 2
    
    return factor
```
