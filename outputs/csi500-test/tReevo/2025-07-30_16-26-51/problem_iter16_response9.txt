```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Assign Weights to Each Volatility
    short_term_volatility_weight = 0.3
    mid_term_volatility_weight = 0.4
    long_term_volatility_weight = 0.3
    
    df['combined_volatility'] = (
        df['short_term_std'] * short_term_volatility_weight +
        df['mid_term_std'] * mid_term_volatility_weight +
        df['long_term_std'] * long_term_volatility_weight
    )
    
    # Incorporate Momentum and Mean Reversion
    df['momentum_10'] = df['close'] - df['close'].shift(10)
    df['mean_reversion_30'] = df['close'] - df['close'].rolling(window=30).mean()
    
    # Integrate Liquidity and Volatility
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = df['sma_5'] - df['sma_20']
    
    # Adjust Combined Factor by Price Trend Indicator
    combined_factor = df['volume_weighted_return'] + df['momentum_10'] + df['mean_reversion_30']
    adjusted_combined_factor = combined_factor * (1 + df['price_trend_indicator'])
    
    # Final Alpha Factor
    df['alpha_factor'] = adjusted_combined_factor / df['combined_volatility']
    
    return df['alpha_factor'].dropna()
```
