```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_window, mid_term_window, long_term_window = 5, 10, 20
    df['short_term_sma'] = df['vw_return'].rolling(window=short_term_window).mean()
    df['mid_term_sma'] = df['vw_return'].rolling(window=mid_term_window).mean()
    df['long_term_sma'] = df['vw_return'].rolling(window=long_term_window).mean()
    
    df['short_term_std'] = df['vw_return'].rolling(window=short_term_window).std()
    df['mid_term_std'] = df['vw_return'].rolling(window=mid_term_window).std()
    df['long_term_std'] = df['vw_return'].rolling(window=long_term_window).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight, mid_term_weight, long_term_weight = 0.4, 0.3, 0.3
    df['combined_volatility'] = (short_term_weight * df['short_term_std'] +
                                 mid_term_weight * df['mid_term_std'] +
                                 long_term_weight * df['long_term_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5d_ma_close'] = df['close'].rolling(window=short_term_window).mean()
    df['20d_ma_close'] = df['close'].rolling(window=long_term_window).mean()
    df['price_trend'] = (df['5d_ma_close'] > df['20d_ma_close']).astype(int) * 2 - 1
    
    # Final Alpha Factor
    df['alpha_factor'] = df['vw_return'] / df['combined_volatility'] * df['price_trend']
    
    return df['alpha_factor']
```
