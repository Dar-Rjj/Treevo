```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_sma = df['Volume_Weighted_Return'].rolling(window=5).mean()
    mid_term_sma = df['Volume_Weighted_Return'].rolling(window=10).mean()
    long_term_sma = df['Volume_Weighted_Return'].rolling(window=20).mean()
    
    short_term_std = df['Volume_Weighted_Return'].rolling(window=5).std()
    mid_term_std = df['Volume_Weighted_Return'].rolling(window=10).std()
    long_term_std = df['Volume_Weighted_Return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    combined_volatility = (0.4 * short_term_std) + (0.3 * mid_term_std) + (0.3 * long_term_std)
    
    # Incorporate Dynamic Market Trend Adjustments
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    price_trend_indicator = (sma_5 > sma_20).astype(int) * 2 - 1  # 1 if Bullish, -1 if Bearish
    
    combined_return = (0.4 * short_term_sma) + (0.3 * mid_term_sma) + (0.3 * long_term_sma)
    adjusted_combined_return = combined_return * (1.2 * (price_trend_indicator == 1) + 0.8 * (price_trend_indicator == -1))
    
    # Final Alpha Factor
    alpha_factor = adjusted_combined_return / combined_volatility
    
    return alpha_factor
```
