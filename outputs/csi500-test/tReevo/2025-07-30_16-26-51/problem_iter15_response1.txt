```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['5d_sma_vw_return'] = df['vw_return'].rolling(window=short_term).mean()
    df['5d_std_vw_return'] = df['vw_return'].rolling(window=short_term).std()
    
    # Mid-Term Lookback
    df['10d_sma_vw_return'] = df['vw_return'].rolling(window=mid_term).mean()
    df['10d_std_vw_return'] = df['vw_return'].rolling(window=mid_term).std()
    
    # Long-Term Lookback
    df['20d_sma_vw_return'] = df['vw_return'].rolling(window=long_term).mean()
    df['20d_std_vw_return'] = df['vw_return'].rolling(window=long_term).std()
    
    # Combine Multi-Period Volatilities
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    
    df['combined_volatility'] = (short_weight * df['5d_std_vw_return'] +
                                 mid_weight * df['10d_std_vw_return'] +
                                 long_weight * df['20d_std_vw_return'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5d_ma_close'] = df['close'].rolling(window=short_term).mean()
    df['20d_ma_close'] = df['close'].rolling(window=long_term).mean()
    
    df['trend'] = (df['5d_ma_close'] > df['20d_ma_close']).astype(int) * 2 - 1  # 1 if 5d > 20d, -1 otherwise
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vw_return'].fillna(0) / df['combined_volatility']) * df['trend']
    
    return df['alpha_factor']
```
