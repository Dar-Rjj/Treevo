import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    df['5D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).std()

    df['10D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).std()

    df['20D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.3
    mid_term_vol_weight = 0.4
    long_term_vol_weight = 0.3

    df['Combined_Volatility'] = (short_term_vol_weight * df['5D_STD_Volume_Weighted_Return']) + \
                                (mid_term_vol_weight * df['10D_STD_Volume_Weighted_Return']) + \
                                (long_term_vol_weight * df['20D_STD_Volume_Weighted_Return'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5D_SMA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20D_SMA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    df['Price_Trend_Indicator'] = np.where(df['5D_SMA_Close'] > df['20D_SMA_Close'], 1, -1)

    # Adjust Combined Factor by Price Trend Indicator
    df['Adjusted_Combined_Volatility'] = df['Combined_Volatility'] * df['Price_Trend_Indicator']

    # Integrate Momentum and Volatility
    df['5D_Price_Momentum'] = df['close'].pct_change(periods=short_term_lookback)
    df['20D_Price_Momentum'] = df['close'].pct_change(periods=long_term_lookback)

    # Assign Weights to Each Momentum
    short_term_mom_weight = 0.6
    long_term_mom_weight = 0.4

    df['Combined_Momentum'] = (short_term_mom_weight * df['5D_Price_Momentum']) + \
                              (long_term_mom_weight * df['20D_Price_Momentum'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Volume_Weighted_Return'] / df['Adjusted_Combined_Volatility']) + df['Combined_Momentum']

    # Apply Nonlinear Transformation
    df['Final_Alpha_Factor_Transformed'] = np.log(np.sqrt(np.abs(df['Final_Alpha_Factor'])) + 1) * np.sign(df['Final_Alpha_Factor'])

    return df['Final_Alpha_Factor_Transformed']
