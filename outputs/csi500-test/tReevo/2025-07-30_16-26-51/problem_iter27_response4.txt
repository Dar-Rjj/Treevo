```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Short-Term Lookback (5 days)
    short_term_sma = df['Close_to_Open_Return'].rolling(window=5).mean()
    short_term_std = df['Close_to_Open_Return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    mid_term_sma = df['Close_to_Open_Return'].rolling(window=10).mean()
    mid_term_std = df['Close_to_Open_Return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    long_term_sma = df['Close_to_Open_Return'].rolling(window=20).mean()
    long_term_std = df['Close_to_Open_Return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3
    combined_volatility = (short_term_vol_weight * short_term_std +
                           mid_term_vol_weight * mid_term_std +
                           long_term_vol_weight * long_term_std)
    
    # Incorporate Dynamic Market Trend Adjustments
    close_5d_ma = df['close'].rolling(window=5).mean()
    close_20d_ma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (close_5d_ma > close_20d_ma).astype(int)
    
    # Final Alpha Factor
    alpha_factor = df['Close_to_Open_Return'] / (combined_volatility * price_trend_indicator + 1)
    
    return alpha_factor
```
