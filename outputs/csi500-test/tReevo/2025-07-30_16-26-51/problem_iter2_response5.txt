```python
import pandas as pd

def heuristics_v2(df, lookback_momentum=10, lookback_volume=20):
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']

    # Calculate High-Low Momentum
    high_low_momentum = (high_low_ratio / high_low_ratio.shift(lookback_momentum)) - 1

    # Calculate Open-Close Price Movement
    open_close_movement = df['close'] - df['open']

    # Integrate Open-Close Price Movement
    open_close_momentum = (open_close_movement / open_close_movement.shift(lookback_momentum)) - 1

    # Combine High-Low Momentum and Open-Close Momentum
    combined_momentum = high_low_momentum + open_close_momentum

    # Volume Correction Factor
    volume_correction_factor = df['volume'].rolling(window=lookback_volume).sum()

    # Dynamic Volume Adjustment
    avg_volume = df['volume'].rolling(window=lookback_volume).mean()
    relative_volume_increase = df['volume'] / avg_volume

    # Final Alpha Factor
    alpha_factor = combined_momentum * volume_correction_factor * relative_volume_increase

    return alpha_factor
```
