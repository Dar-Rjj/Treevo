import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume'].shift(-1)
    
    # Dynamic Lookback Periods
    lookback_periods = [5, 10, 20]
    for period in lookback_periods:
        df[f'sma_{period}'] = df['volume_weighted_return'].rolling(window=period).mean()
        df[f'std_{period}'] = df['volume_weighted_return'].rolling(window=period).std()
    
    # Combine Multi-Period Volatility
    df['combined_volatility'] = (df['std_5'] + df['std_10'] + df['std_20']) / 3
    
    # Integrate Market Signals
    df['high_low_range'] = df['high'] - df['low']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Final Factor
    df['final_factor'] = df['volume_weighted_return'] / df['combined_volatility']
    df['final_factor'] = np.log(df['final_factor']).apply(np.sqrt)
    
    return df['final_factor'].dropna()
