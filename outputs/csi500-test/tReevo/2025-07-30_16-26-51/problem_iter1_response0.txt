import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Components
    open_to_high_return = (df['high'] - df['open']) / df['open']
    open_to_low_return = (df['low'] - df['open']) / df['open']
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Return Components
    abs_open_to_high_return = open_to_high_return.abs()
    abs_open_to_low_return = open_to_low_return.abs()
    abs_open_to_close_return = open_to_close_return.abs()
    
    avg_abs_returns = (abs_open_to_high_return + abs_open_to_low_return + abs_open_to_close_return) / 3
    sum_returns = open_to_high_return + open_to_low_return + open_to_close_return
    
    # Weight by Volume and Amount
    volume_weight = df['volume']
    amount_weight = df['amount']
    
    final_factor = (sum_returns * (volume_weight + amount_weight)) / avg_abs_returns
    
    return final_factor
