```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume-Weight the Return
    df['volume_weighted_return'] = df['close_to_open_return'] * (df['volume'] / df['volume'].mean())
    
    # Dynamic Lookback Periods for Momentum
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['5_day_ema'] = df['volume_weighted_return'].ewm(span=short_term_lookback, adjust=False).mean()
    
    # Mid-Term Lookback
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['10_day_ema'] = df['volume_weighted_return'].ewm(span=mid_term_lookback, adjust=False).mean()
    
    # Long-Term Lookback
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['20_day_ema'] = df['volume_weighted_return'].ewm(span=long_term_lookback, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['short_term_vol'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    df['long_term_vol'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    df['adaptive_vol'] = 0.6 * df['short_term_vol'] + 0.4 * df['long_term_vol']
    
    # Integrated Momentum and Adaptive Volatility
    df['integrated_momentum'] = (df['5_day_sma'] + df['10_day_ema'] + df['20_day_ema']) / df['adaptive_vol']
    
    # Incorporate High-Low Price Ratio
    df['high_low_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Integrate High-Low Momentum
    df['sum_high_low_5'] = df['high_low_ratio'].rolling(window=short_term_lookback).sum()
    df['sum_high_low_10'] = df['high_low_ratio'].rolling(window=mid_term_lookback).sum()
    df['sum_high_low_20'] = df['high_low_ratio'].rolling(window=long_term_lookback).sum()
    
    # Adaptive High-Low Momentum
    df['adaptive_high_low_momentum'] = 0.3 * df['sum_high_low_5'] + 0.4 * df['sum_high_low_10'] + 0.3 * df['sum_high_low_20']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['integrated_momentum'] + df['adaptive_high_low_momentum'] + df['volume_weighted_return']
    
    return df['alpha_factor']
```
