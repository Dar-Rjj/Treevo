```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_returns = close_to_open_return * df['volume']
    
    # Short-Term (5 days) Lookback
    short_term_sma = volume_weighted_returns.rolling(window=5).mean()
    short_term_std = volume_weighted_returns.rolling(window=5).std()
    
    # Long-Term (20 days) Lookback
    long_term_sma = volume_weighted_returns.rolling(window=20).mean()
    long_term_std = volume_weighted_returns.rolling(window=20).std()
    
    # Trading Volume Trends
    short_term_volume_trend = df['volume'].rolling(window=5).mean()
    long_term_volume_trend = df['volume'].rolling(window=20).mean()
    
    # Adjust with Volatility
    adjusted_short_term_sma = short_term_sma / (short_term_std + 1e-6)
    adjusted_long_term_sma = long_term_sma / (long_term_std + 1e-6)
    
    adjusted_short_term_volume_trend = short_term_volume_trend / (short_term_std + 1e-6)
    adjusted_long_term_volume_trend = long_term_volume_trend / (long_term_std + 1e-6)
    
    # Combine the factors
    factor = (adjusted_short_term_sma - adjusted_long_term_sma) * (adjusted_short_term_volume_trend - adjusted_long_term_volume_trend)
    
    return factor
```
