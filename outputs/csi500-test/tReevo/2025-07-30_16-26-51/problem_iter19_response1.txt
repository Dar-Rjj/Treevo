```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_sma = df['vw_return'].rolling(window=5).mean()
    short_term_std = df['vw_return'].rolling(window=5).std()
    mid_term_sma = df['vw_return'].rolling(window=10).mean()
    mid_term_std = df['vw_return'].rolling(window=10).std()
    long_term_sma = df['vw_return'].rolling(window=20).mean()
    long_term_std = df['vw_return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    combined_volatility = (short_term_std * short_term_weight) + \
                         (mid_term_std * mid_term_weight) + \
                         (long_term_std * long_term_weight)
    
    # Incorporate Dynamic Market Trend Adjustments
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_indicator = (ma_5 > ma_20).astype(float) * 0.2 + 1  # Bullish: 1.1, Bearish: 0.9
    
    # Final Alpha Factor
    combined_factor = (short_term_sma + mid_term_sma + long_term_sma) / combined_volatility
    final_alpha = combined_factor * trend_indicator
    
    return final_alpha
```
