```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    long_term_lookback = 20
    
    # Short-Term Lookback
    short_term_sma = volume_weighted_return.rolling(window=short_term_lookback).mean()
    short_term_std = volume_weighted_return.rolling(window=short_term_lookback).std()
    
    # Long-Term Lookback
    long_term_sma = volume_weighted_return.rolling(window=long_term_lookback).mean()
    long_term_std = volume_weighted_return.rolling(window=long_term_lookback).std()
    
    # Stabilize with Volatility
    short_term_stabilized = short_term_sma / short_term_std
    long_term_stabilized = long_term_sma / long_term_std
    
    # Final Factor
    final_factor = short_term_stabilized - long_term_stabilized
    
    return final_factor
```
