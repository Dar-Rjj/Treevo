```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    short_sma = df['volume_weighted_return'].rolling(window=5).mean()
    short_std = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    mid_sma = df['volume_weighted_return'].rolling(window=10).mean()
    mid_std = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    long_sma = df['volume_weighted_return'].rolling(window=20).mean()
    long_std = df['volume_weighted_return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    short_volatility_weight = 0.4
    mid_volatility_weight = 0.3
    long_volatility_weight = 0.3
    
    combined_volatility = (short_volatility_weight * short_std +
                           mid_volatility_weight * mid_std +
                           long_volatility_weight * long_std)
    
    # Incorporate Refined Market Trend Adjustments
    df['5_day_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_day_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    price_trend_indicator = np.where(df['5_day_ema'] > df['20_day_ema'], 1, -1)
    
    # Final Alpha Factor
    combined_factor = (df['volume_weighted_return'] / combined_volatility) * price_trend_indicator
    
    return combined_factor
```
