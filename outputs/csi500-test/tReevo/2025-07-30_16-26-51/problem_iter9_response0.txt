```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Adaptive Volatility: Combine Short-Term, Mid-Term, and Long-Term Volatilities
    weights = [0.5, 0.3, 0.2]  # Example weights for short, mid, long term
    df['adaptive_volatility'] = (weights[0] * df['short_term_std'] +
                                 weights[1] * df['mid_term_std'] +
                                 weights[2] * df['long_term_std'])
    
    # Final Factor: Use Combined Volume-Weighted Close-to-Open Return divided by Adaptive Volatility
    df['final_factor'] = (df['short_term_sma'] + df['mid_term_sma'] + df['long_term_sma']) / df['adaptive_volatility']
    
    return df['final_factor']
```
