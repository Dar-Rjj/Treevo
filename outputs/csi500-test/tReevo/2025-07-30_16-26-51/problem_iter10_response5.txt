```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Incorporate Market Sentiment (assuming a 'sentiment' column exists)
    if 'sentiment' in df.columns:
        df['sentiment_adjusted_return'] = df['volume_weighted_return'] * df['sentiment']
    else:
        df['sentiment_adjusted_return'] = df['volume_weighted_return']
    
    # Dynamic Lookback Periods
    short_term_ema = df['sentiment_adjusted_return'].ewm(span=5, adjust=False).mean()
    mid_term_ema = df['sentiment_adjusted_return'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['sentiment_adjusted_return'].ewm(span=20, adjust=False).mean()
    extra_long_term_ema = df['sentiment_adjusted_return'].ewm(span=40, adjust=False).mean()
    
    short_term_std = df['sentiment_adjusted_return'].rolling(window=5).std()
    mid_term_std = df['sentiment_adjusted_return'].rolling(window=10).std()
    long_term_std = df['sentiment_adjusted_return'].rolling(window=20).std()
    extra_long_term_std = df['sentiment_adjusted_return'].rolling(window=40).std()
    
    # Combine Multi-Period Volatility
    combined_volatility = (short_term_std + mid_term_std + long_term_std + extra_long_term_std) / 4
    
    # Final Factor
    final_factor = df['sentiment_adjusted_return'] / combined_volatility
    
    return final_factor
```
