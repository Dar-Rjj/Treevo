```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_window = 5
    mid_term_window = 10
    long_term_window = 20
    
    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=short_term_window).mean()
    df['Short_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=short_term_window).std()
    
    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=mid_term_window).mean()
    df['Mid_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=mid_term_window).std()
    
    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=long_term_window).mean()
    df['Long_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=long_term_window).std()
    
    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.3
    mid_term_vol_weight = 0.4
    long_term_vol_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_vol_weight * df['Short_Term_STD'] +
                                 mid_term_vol_weight * df['Mid_Term_STD'] +
                                 long_term_vol_weight * df['Long_Term_STD'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Price_Trend'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'] / df['Combined_Volatility']) * df['Price_Trend']
    
    return df['Alpha_Factor']
```
