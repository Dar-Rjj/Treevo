import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    volume_weighted_return = close_to_open_return * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    short_sma = volume_weighted_return.rolling(window=short_term_lookback).mean()
    short_std = volume_weighted_return.rolling(window=short_term_lookback).std()

    mid_sma = volume_weighted_return.rolling(window=mid_term_lookback).mean()
    mid_std = volume_weighted_return.rolling(window=mid_term_lookback).std()

    long_sma = volume_weighted_return.rolling(window=long_term_lookback).mean()
    long_std = volume_weighted_return.rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_volatility_weight = 0.4
    mid_volatility_weight = 0.3
    long_volatility_weight = 0.3

    combined_volatility = (short_volatility_weight * short_std +
                          mid_volatility_weight * mid_std +
                          long_volatility_weight * long_std)

    # Incorporate Dynamic Market Trend Adjustments
    close_5d_ma = df['close'].rolling(window=short_term_lookback).mean()
    close_20d_ma = df['close'].rolling(window=long_term_lookback).mean()

    price_trend_indicator = (close_5d_ma > close_20d_ma).astype(int) * 2 - 1

    # Final Alpha Factor
    combined_factor = (volume_weighted_return.rolling(window=short_term_lookback).sum() /
                       combined_volatility) * price_trend_indicator

    return combined_factor.dropna()
