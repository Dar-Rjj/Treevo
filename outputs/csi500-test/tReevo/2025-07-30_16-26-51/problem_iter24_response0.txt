```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_sma'] = df['vw_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['vw_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_sma'] = df['vw_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['vw_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_sma'] = df['vw_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['vw_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    df['combined_volatility'] = (short_weight * df['short_std'] + 
                                 mid_weight * df['mid_std'] + 
                                 long_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_indicator'] = np.where(df['ema_5_close'] > df['ema_20_close'], 1, -1)
    
    # Integrate Volume Trends
    df['ema_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['ema_20_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_trend_indicator'] = np.where(df['ema_5_volume'] > df['ema_20_volume'], 1, -1)
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vw_return'] / df['combined_volatility']) * \
                         (df['price_trend_indicator'] * df['volume_trend_indicator'])
    
    return df['alpha_factor']
```
