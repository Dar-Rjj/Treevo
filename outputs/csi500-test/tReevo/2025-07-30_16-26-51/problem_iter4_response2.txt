```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    """
    Generate a novel alpha factor using High-Low Price Momentum with Dynamic Volume Adjustment.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date)
    - lookback_period: int, the number of periods to look back for momentum and volume calculations
    
    Returns:
    - pandas Series indexed by (date) representing the factor values
    """
    
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    sum_high_low_ratio = high_low_ratio.rolling(window=lookback_period).sum()
    avg_high_low_ratio = high_low_ratio.rolling(window=lookback_period).mean()
    high_low_momentum = sum_high_low_ratio / avg_high_low_ratio
    
    # Dynamic Volume Adjustment
    sum_volume = df['volume'].rolling(window=lookback_period).sum()
    current_volume = df['volume']
    volume_momentum = current_volume / sum_volume
    
    # Combine High-Low Momentum and Volume Momentum
    combined_momentum = high_low_momentum * volume_momentum
    
    # Final Factor
    final_factor = combined_momentum
    
    return final_factor
```
