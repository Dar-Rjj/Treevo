```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    df['high_low_momentum'] = df['high_low_ratio'].rolling(window=5).sum()
    
    # Calculate Volume Anomaly
    df['volume_anomaly'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate 5-Day Realized Volatility
    df['log_returns'] = np.log(df['close']).diff()
    df['realized_volatility'] = df['log_returns'].rolling(window=5).std()
    
    # Smooth Volume Changes
    df['smoothed_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    df['combined_factor'] = df['high_low_momentum'] + df['volume_anomaly'] * df['smoothed_volume']
    
    # Adjust for Volatility
    df['volatility_adjusted_factor'] = df['combined_factor'] / df['realized_volatility']
    
    # Incorporate Close-to-Open Return Weighting
    df['final_alpha_factor'] = df['volatility_adjusted_factor'] + df['weighted_close_to_open_return']
    
    return df['final_alpha_factor']
```
