```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) / df['close']) - 1
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20

    # Short-Term Lookback
    df['short_term_sma'] = df['volume_weighted_close_to_open_return'].rolling(window=short_term).mean()
    df['short_term_std'] = df['volume_weighted_close_to_open_return'].rolling(window=short_term).std()

    # Mid-Term Lookback
    df['mid_term_sma'] = df['volume_weighted_close_to_open_return'].rolling(window=mid_term).mean()
    df['mid_term_std'] = df['volume_weighted_close_to_open_return'].rolling(window=mid_term).std()

    # Long-Term Lookback
    df['long_term_sma'] = df['volume_weighted_close_to_open_return'].rolling(window=long_term).mean()
    df['long_term_std'] = df['volume_weighted_close_to_open_return'].rolling(window=long_term).std()

    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3

    df['combined_volatility'] = (short_term_volatility_weight * df['short_term_std'] +
                                 mid_term_volatility_weight * df['mid_term_std'] +
                                 long_term_volatility_weight * df['long_term_std'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    df['price_trend_indicator'] = df['5_day_ma_close'] - df['20_day_ma_close']
    positive_trend_adjustment = 1.1
    negative_trend_adjustment = 0.9
    df['trend_adjusted_factor'] = df['volume_weighted_close_to_open_return'] * (
        positive_trend_adjustment if df['price_trend_indicator'] > 0 else negative_trend_adjustment)

    # Add Momentum Interaction Term
    df['5_day_momentum'] = df['close'].pct_change(periods=5)
    df['20_day_momentum'] = df['close'].pct_change(periods=20)
    df['momentum_interaction_term'] = (df['5_day_momentum'] + df['20_day_momentum']) / 2

    # Add Liquidity Interaction Term
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['liquidity_interaction_term'] = (df['5_day_avg_volume'] + df['20_day_avg_volume']) / 2

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['trend_adjusted_factor'] * df['momentum_interaction_term'] * df['liquidity_interaction_term']) / df['combined_volatility']

    return df['final_alpha_factor'].dropna()
```
