import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'].shift(-1) / df['open'] - 1
    
    # Volume-Weight the Return
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume'].shift(-1)
    
    # Dynamic Lookback Periods for Momentum
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback (e.g., 5 days)
    df['5d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term).mean()
    df['5d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=short_term, adjust=False).mean()
    
    # Mid-Term Lookback (e.g., 10 days)
    df['10d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term).mean()
    df['10d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=mid_term, adjust=False).mean()
    
    # Long-Term Lookback (e.g., 20 days)
    df['20d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term).mean()
    df['20d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=long_term, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['5d_Vol'] = df['Volume_Weighted_Return'].rolling(window=short_term).std()
    df['20d_Vol'] = df['Volume_Weighted_Return'].rolling(window=long_term).std()
    df['Adaptive_Vol'] = 0.6 * df['5d_Vol'] + 0.4 * df['20d_Vol']
    
    # Final Factor with Momentum and Adaptive Volatility
    df['Combined_VW_Return'] = (df['5d_SMA_VW_Return'] + df['10d_EMA_VW_Return'] + df['20d_EMA_VW_Return']) / 3
    df['Final_Factor'] = df['Combined_VW_Return'] / df['Adaptive_Vol']
    
    return df['Final_Factor'].dropna()
