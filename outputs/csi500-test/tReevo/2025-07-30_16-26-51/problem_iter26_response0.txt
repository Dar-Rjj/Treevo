```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_sma = df['Volume_Weighted_Return'].rolling(window=5).mean()
    short_term_std = df['Volume_Weighted_Return'].rolling(window=5).std()
    mid_term_ema = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    mid_term_std = df['Volume_Weighted_Return'].rolling(window=10).std()
    long_term_ema = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()
    long_term_std = df['Volume_Weighted_Return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    recent_volatility_trend = df['Volume_Weighted_Return'].rolling(window=5).std()
    short_term_weight = recent_volatility_trend / (short_term_std + mid_term_std + long_term_std)
    mid_term_weight = recent_volatility_trend / (short_term_std + mid_term_std + long_term_std)
    long_term_weight = 1 - (short_term_weight + mid_term_weight)
    combined_volatility = (short_term_weight * short_term_std) + (mid_term_weight * mid_term_std) + (long_term_weight * long_term_std)
    
    # Incorporate Dynamic Market Trend Adjustments
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_trend_indicator = np.where(ma_5 > ma_20, 1.2, 0.8)
    
    # Integrate High-Low Price Ratio
    high_low_price_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Incorporate Additional Market Indicators
    daily_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    average_daily_price_change = daily_price_change.rolling(window=5).mean()
    price_change_adjustment = np.where(average_daily_price_change > 0, 1.1, 0.9)
    
    # Final Alpha Factor
    combined_factor = (short_term_sma + mid_term_ema + long_term_ema) / combined_volatility * high_low_price_ratio
    trend_adjusted_factor = combined_factor * price_trend_indicator * price_change_adjustment
    
    return trend_adjusted_factor
```
