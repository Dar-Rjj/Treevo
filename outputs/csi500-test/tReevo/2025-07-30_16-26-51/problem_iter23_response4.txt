```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Short-Term Lookback (5 days)
    short_term_sma = df['Close_to_Open_Return'].rolling(window=5).mean()
    short_term_std = df['Close_to_Open_Return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    mid_term_sma = df['Close_to_Open_Return'].rolling(window=10).mean()
    mid_term_std = df['Close_to_Open_Return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    long_term_sma = df['Close_to_Open_Return'].rolling(window=20).mean()
    long_term_std = df['Close_to_Open_Return'].rolling(window=20).std()

    # Combine Multi-Period Volatilities with Weights
    combined_volatility = (0.3 * short_term_std + 0.4 * mid_term_std + 0.3 * long_term_std)

    # Incorporate Dynamic Market Trend Adjustments
    df['5D_MA_Close'] = df['close'].rolling(window=5).mean()
    df['20D_MA_Close'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = (df['5D_MA_Close'] > df['20D_MA_Close']).astype(int) * 1.2 + (df['5D_MA_Close'] <= df['20D_MA_Close']).astype(int) * 0.8

    # Final Alpha Factor
    df['Alpha_Factor'] = (short_term_sma + mid_term_sma + long_term_sma) / combined_volatility * df['Price_Trend_Indicator']

    return df['Alpha_Factor']
```
