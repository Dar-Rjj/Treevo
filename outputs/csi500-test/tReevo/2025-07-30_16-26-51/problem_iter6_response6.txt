```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Calculate 5-Day Realized Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Combine Factors
    combined_factor = volume_weighted_close_to_open / realized_volatility
    
    # Calculate 5-Day Momentum
    five_day_momentum = df['close'] - df['close'].shift(5)
    
    # Weight 5-Day Momentum by Volume
    volume_weighted_momentum = five_day_momentum * df['volume']
    
    # Final Factor
    final_factor = combined_factor + volume_weighted_momentum
    
    return final_factor
```
