```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_mavg'] = df['vw_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['vw_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_mavg'] = df['vw_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['vw_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_mavg'] = df['vw_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['vw_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    df['combined_volatility'] = (short_weight * df['short_std'] + 
                                 mid_weight * df['mid_std'] + 
                                 long_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['ma_5'] = df['close'].rolling(window=short_term_lookback).mean()
    df['ma_20'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['trend_indicator'] = 0
    df.loc[df['ma_5'] > df['ma_20'], 'trend_indicator'] = 1
    df.loc[df['ma_5'] < df['ma_20'], 'trend_indicator'] = -1
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vw_return'] / df['combined_volatility']) * df['trend_indicator']
    
    return df['alpha_factor']
```
