```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    df['Short_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['Short_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    df['Mid_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['Mid_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    df['Long_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['Long_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.3
    mid_term_weight = 0.4
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['Short_Term_STD'] + 
                                 mid_term_weight * df['Mid_Term_STD'] + 
                                 long_term_weight * df['Long_Term_STD'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_MA'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20_Day_MA'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = (df['5_Day_MA'] > df['20_Day_MA']).astype(int) * 2 - 1
    
    df['Adjusted_Combined_Volatility'] = df['Combined_Volatility'] * df['Price_Trend_Indicator']
    
    # Integrate Momentum and Volatility
    df['5_Day_Momentum'] = (df['close'] - df['close'].shift(short_term_lookback)) / df['close'].shift(short_term_lookback)
    df['20_Day_Momentum'] = (df['close'] - df['close'].shift(long_term_lookback)) / df['close'].shift(long_term_lookback)
    
    # Assign Weights to Each Momentum
    five_day_momentum_weight = 0.6
    twenty_day_momentum_weight = 0.4
    
    df['Combined_Momentum'] = (five_day_momentum_weight * df['5_Day_Momentum'] + 
                               twenty_day_momentum_weight * df['20_Day_Momentum'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Weighted_Close_to_Open_Return'] / (df['Combined_Volatility'] + df['Combined_Momentum'])
    
    return df['Final_Alpha_Factor']
```
