```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Short-Term Lookback (5 days)
    short_term_sma = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=5).mean()
    short_term_std = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    mid_term_sma = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=10).mean()
    mid_term_std = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    long_term_sma = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=20).mean()
    long_term_std = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=20).std()

    # Combine Multi-Period Volatilities with Weights
    combined_volatility = (short_term_std * 0.4 + 
                           mid_term_std * 0.3 + 
                           long_term_std * 0.3)

    # Incorporate Dynamic Market Trend Adjustments
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend_indicator = (ema_5 - ema_20) / df['close']
    
    # Combined Factor
    combined_factor = (df['Volume_Weighted_Close_to_Open_Return'] /
                       combined_volatility)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * price_trend_indicator

    return final_alpha_factor
```
