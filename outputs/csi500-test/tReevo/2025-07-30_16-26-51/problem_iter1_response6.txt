import pandas as pd

def heuristics_v2(df):
    # Compute Logarithmic Return
    df['log_return'] = df['close'].apply(lambda x: np.log(x)) - df['close'].shift(1).apply(lambda x: np.log(x))
    
    # Sum the last 5 days' Log Returns for Daily Price Momentum
    df['daily_price_momentum'] = df['log_return'].rolling(window=5).sum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume of past 20 days
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    
    # Compute Relative Volume Increase
    df['relative_volume_increase'] = df['volume'] / df['avg_volume_20_days']
    
    # Determine if Spike Occurred
    volume_spike_threshold = 2.0  # Example threshold
    relative_volume_spike_threshold = 1.5  # Example threshold
    df['volume_spike'] = ((df['volume_change'] > volume_spike_threshold) & 
                          (df['relative_volume_increase'] > relative_volume_spike_threshold)).astype(int)
    
    # Combine Momentum and Volume Spike
    momentum_constant = 1.0
    volume_spike_constant = 10.0  # This constant can be adjusted to weigh the importance of a volume spike
    
    df['alpha_factor'] = (df['daily_price_momentum'] * momentum_constant + 
                          df['volume_spike'] * volume_spike_constant)
    
    return df['alpha_factor']
