```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Stabilize with Combined Multi-Period Volatility
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    df['combined_volatility'] = (short_term_weight * df['short_term_std'] +
                                 mid_term_weight * df['mid_term_std'] +
                                 long_term_weight * df['long_term_std'])
    
    # Integrate Market Signals
    df['high_low_range'] = df['high'] - df['low']
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Final Factor
    df['final_factor'] = df['volume_weighted_return'] / df['combined_volatility']
    df['nonlinear_transformed_factor'] = np.log(np.sqrt(np.abs(df['final_factor'])))
    
    # Enhance with Additional Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Weight by Volume
    df['5_day_volume_weighted_momentum'] = df['5_day_momentum'] * df['volume']
    df['10_day_volume_weighted_momentum'] = df['10_day_momentum'] * df['volume']
    
    # Combine Different Time Frame Momentums
    df['combined_momentum'] = df['5_day_volume_weighted_momentum'] + df['10_day_volume_weighted_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5).mean()
    
    # Adjust for Short-Term Volatility
    df['adjusted_smoothed_momentum'] = df['smoothed_combined_momentum'] / df['short_term_std']
    
    # Final Integrated Factor
    df['final_integrated_factor'] = df['nonlinear_transformed_factor'] + df['adjusted_smoothed_momentum']
    
    return df['final_integrated_factor'].dropna()
```
