```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] / df['open'].shift(1) - 1) * df['volume']

    # Short-Term Lookback (5 days)
    short_term_sma = df['close_to_open_return'].rolling(window=5).mean()
    short_term_std = df['close_to_open_return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    mid_term_sma = df['close_to_open_return'].rolling(window=10).mean()
    mid_term_std = df['close_to_open_return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    long_term_sma = df['close_to_open_return'].rolling(window=20).mean()
    long_term_std = df['close_to_open_return'].rolling(window=20).std()

    # Adaptive Volatility Stabilization
    short_weight = 0.7
    long_weight = 0.3
    adaptive_volatility = (short_weight * short_term_std) + (long_weight * long_term_std)

    # Final Factor Calculation
    df['factor'] = df['close_to_open_return'] / adaptive_volatility

    return df['factor']
```
