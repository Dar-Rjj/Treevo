```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Volume-Weighted Close-to-Open Return
    data['Volume_Weighted_Return'] = (data['close'] - data['open'].shift(1)) * data['volume']
    
    # Adaptive Lookback Periods
    short_term_sma = data['Volume_Weighted_Return'].rolling(window=5).mean()
    short_term_std = data['Volume_Weighted_Return'].rolling(window=5).std()
    
    mid_term_sma = data['Volume_Weighted_Return'].rolling(window=10).mean()
    mid_term_std = data['Volume_Weighted_Return'].rolling(window=10).std()
    
    long_term_sma = data['Volume_Weighted_Return'].rolling(window=20).mean()
    long_term_std = data['Volume_Weighted_Return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities with Dynamic Weights
    recent_volatility_trend = short_term_std / long_term_std
    short_term_vol_weight = 1 / (1 + np.exp(-recent_volatility_trend))
    mid_term_vol_weight = 0.5 * (1 - short_term_vol_weight)
    long_term_vol_weight = 0.5 * (1 - short_term_vol_weight)
    
    combined_volatility = (short_term_vol_weight * short_term_std +
                           mid_term_vol_weight * mid_term_std +
                           long_term_vol_weight * long_term_std)
    
    # Incorporate Dynamic Market Trend Adjustments
    data['5D_MA_Close'] = data['close'].rolling(window=5).mean()
    data['20D_MA_Close'] = data['close'].rolling(window=20).mean()
    price_trend = (data['5D_MA_Close'] > data['20D_MA_Close']).astype(int)
    trend_adjustment = price_trend.map({1: 1.2, 0: 0.8})
    
    # Integrate High-Low Price Ratio
    high_low_ratio = (data['high'] - data['low']) / (data['high'] + data['low'])
    
    # Incorporate Additional Market Indicators
    daily_price_change = (data['close'] - data['close'].shift(1)) / data['close'].shift(1)
    avg_daily_price_change = daily_price_change.rolling(window=5).mean()
    avg_daily_price_change_adjustment = (avg_daily_price_change > 0).astype(float) * 1.2 + (avg_daily_price_change <= 0).astype(float) * 0.8
    
    # Final Alpha Factor
    combined_factor = (0.3 * short_term_sma + 
                       0.3 * mid_term_sma.ewm(span=10, adjust=False).mean() + 
                       0.4 * long_term_sma.ewm(span=20, adjust=False).mean())
    
    final_alpha_factor = (combined_factor * trend_adjustment * high_low_ratio * avg_daily_price_change_adjustment) / combined_volatility
    
    return final_alpha_factor
```
