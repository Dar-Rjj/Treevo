```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['HighLowRatio'] = df['high'] / df['low']

    # Calculate High-Low Momentum
    df['HighLowMomentum'] = df['HighLowRatio'].rolling(window=5).sum()

    # Calculate Volume Anomaly
    df['AvgVolume_20D'] = df['volume'].rolling(window=20).mean()
    df['VolumeAnomaly'] = df['volume'] - df['AvgVolume_20D']

    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open']) / df['open']

    # Weight by Volume
    df['VolumeWeightedReturn'] = df['CloseToOpenReturn'] * df['volume']

    # Calculate 5-Day and 20-Day Realized Volatility
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    df['RealizedVol_5D'] = df['LogReturn'].rolling(window=5).std() * np.sqrt(252)
    df['RealizedVol_20D'] = df['LogReturn'].rolling(window=20).std() * np.sqrt(252)

    # Smooth Volume Changes
    df['SmoothedVolume'] = df['volume'].ewm(span=20, adjust=False).mean()

    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    df['CombinedFactor'] = (df['HighLowMomentum'] + df['VolumeAnomaly']) * df['SmoothedVolume']

    # Adjust for Volatility
    df['AdjustedFactor'] = df['CombinedFactor'] / df['RealizedVol_5D']

    # Incorporate Price Trend Analysis
    df['MA_5D'] = df['close'].rolling(window=5).mean()
    df['MA_20D'] = df['close'].rolling(window=20).mean()
    df['PriceTrendIndicator'] = df['MA_5D'] - df['MA_20D']
    df['FinalAlphaFactor'] = df['AdjustedFactor'] * df['PriceTrendIndicator']

    return df['FinalAlphaFactor'].dropna()
```
