import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Period
    short_term_lookback = 5
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_sma'] = df['close_to_open_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['close_to_open_return'].rolling(window=short_term_lookback).std()
    
    # Long-Term Lookback
    df['long_sma'] = df['close_to_open_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['close_to_open_return'].rolling(window=long_term_lookback).std()
    
    # Stabilize with Volatility
    df['short_vol_adjusted'] = df['close_to_open_return'] / df['short_std']
    df['long_vol_adjusted'] = df['close_to_open_return'] / df['long_std']
    
    # Combine Adjusted Factors
    w1 = 0.7  # weight for short-term
    w2 = 0.3  # weight for long-term
    df['alpha_factor'] = w1 * df['short_vol_adjusted'] + w2 * df['long_vol_adjusted']
    
    return df['alpha_factor'].dropna()
