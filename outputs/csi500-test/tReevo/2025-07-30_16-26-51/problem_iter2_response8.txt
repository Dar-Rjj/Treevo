```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Exponential Moving Average (EMA) of Close Prices (Last 20 Days)
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['ema_close']
    
    # Compute Standard Deviation of Daily Returns (Last 20 Days)
    df['daily_returns'] = df['close'].pct_change()
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    
    # Normalize Price Momentum by Volatility
    df['normalized_momentum'] = df['price_momentum'] / df['volatility']
    
    # Compute Exponential Moving Average (EMA) of Volume (Last 20 Days)
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Identify Volume Spike
    df['volume_spike'] = np.where(df['volume'] > 2 * df['ema_volume'], 1, 0)
    
    # Adjust Price Momentum based on Volume Spike
    df['adjusted_momentum'] = np.where(df['volume_spike'] == 1, df['normalized_momentum'] * 1.5, df['normalized_momentum'])
    
    # Factor Value
    factor_value = df['adjusted_momentum']
    
    return factor_value
```
