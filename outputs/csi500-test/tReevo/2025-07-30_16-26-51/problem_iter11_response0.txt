```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'].shift(-1) - df['open']) * df['volume']
    
    # Adaptive Lookback Periods
    lookbacks = [5, 10, 20]
    for lookback in lookbacks:
        df[f'vw_{lookback}_sma'] = df['vw_return'].rolling(window=lookback).mean()
        df[f'vw_{lookback}_std'] = df['vw_return'].rolling(window=lookback).std()
    
    # Weights for each volatility
    weights = [0.5, 0.3, 0.2]  # Short, Mid, Long term weights
    combined_volatility = sum([weights[i] * df[f'vw_{lookbacks[i]}_std'] for i in range(len(lookbacks))])
    
    # Dynamic Market Trend Adjustments
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['trend_indicator'] = (df['5_day_ma'] > df['20_day_ma']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vw_return'].rolling(window=5).mean() / combined_volatility) * df['trend_indicator']
    
    return df['alpha_factor']
```
