import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Volume-Weight the Return
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Periods for Momentum
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback (5 days)
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['5_day_ema'] = df['volume_weighted_return'].ewm(span=short_term_lookback, adjust=False).mean()
    
    # Mid-Term Lookback (10 days)
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['10_day_ema'] = df['volume_weighted_return'].ewm(span=mid_term_lookback, adjust=False).mean()
    
    # Long-Term Lookback (20 days)
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['20_day_ema'] = df['volume_weighted_return'].ewm(span=long_term_lookback, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    short_term_volatility = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    long_term_volatility = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    df['adaptive_volatility'] = 0.6 * short_term_volatility + 0.4 * long_term_volatility
    
    # Final Factor with Integrated Momentum and Adaptive Volatility
    df['final_factor'] = (df['5_day_sma'] + df['10_day_ema'] + df['20_day_ema']) / df['adaptive_volatility']
    
    return df['final_factor'].dropna()
