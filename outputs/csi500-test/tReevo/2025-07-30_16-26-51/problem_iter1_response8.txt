import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Intraday Range Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Features
    combined_features = range_to_open_ratio * intraday_return
    
    # Calculate Volume Change
    volume_change = df['volume'].pct_change()
    
    # Weight by Volume Change
    weighted_feature = combined_features * volume_change
    
    return weighted_feature.dropna()
