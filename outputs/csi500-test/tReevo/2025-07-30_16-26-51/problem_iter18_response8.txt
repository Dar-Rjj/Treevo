```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_lookback = 5
    mid_lookback = 10
    long_lookback = 20
    
    # Short-Term Lookback
    df['SMA_5'] = df['Close_to_Open_Return'].rolling(window=short_lookback).mean()
    df['STD_5'] = df['Close_to_Open_Return'].rolling(window=short_lookback).std()
    
    # Mid-Term Lookback
    df['SMA_10'] = df['Close_to_Open_Return'].rolling(window=mid_lookback).mean()
    df['STD_10'] = df['Close_to_Open_Return'].rolling(window=mid_lookback).std()
    
    # Long-Term Lookback
    df['SMA_20'] = df['Close_to_Open_Return'].rolling(window=long_lookback).mean()
    df['STD_20'] = df['Close_to_Open_Return'].rolling(window=long_lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_vol_weight = 0.4
    mid_vol_weight = 0.3
    long_vol_weight = 0.3
    
    df['Combined_Volatility'] = (
        short_vol_weight * df['STD_5'] +
        mid_vol_weight * df['STD_10'] +
        long_vol_weight * df['STD_20']
    )
    
    # Incorporate Dynamic Market Trend Adjustments
    df['EMA_5'] = df['close'].ewm(span=short_lookback, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=long_lookback, adjust=False).mean()
    df['Price_Trend_Indicator'] = df['EMA_5'] - df['EMA_20']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        (df['SMA_5'] + df['SMA_10'] + df['SMA_20']) / 
        df['Combined_Volatility'] * 
        df['Price_Trend_Indicator']
    )
    
    return df['Alpha_Factor']
```
