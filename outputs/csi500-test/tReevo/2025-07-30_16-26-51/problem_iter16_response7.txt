import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Volume-Weight the Return
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5_day_ema'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    # Mid-Term Lookback (10 days)
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['10_day_ema'] = df['volume_weighted_return'].ewm(span=10, adjust=False).mean()
    
    # Long-Term Lookback (20 days)
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20_day_ema'] = df['volume_weighted_return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    short_term_volatility = df['volume_weighted_return'].rolling(window=5).std()
    long_term_volatility = df['volume_weighted_return'].rolling(window=20).std()
    adaptive_volatility = 0.6 * short_term_volatility + 0.4 * long_term_volatility
    
    # Integrated Momentum and Adaptive Volatility
    df['integrated_momentum'] = (df['5_day_sma'] + df['10_day_ema'] + df['20_day_ema']) / adaptive_volatility
    
    # High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # High-Low Momentum
    df['high_low_momentum'] = df['high_low_ratio'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['integrated_momentum'] + df['high_low_momentum'] + df['volume_weighted_return']
    
    return df['alpha_factor']

# Example usage:
# alpha_factor = heuristics_v2(your_dataframe)
