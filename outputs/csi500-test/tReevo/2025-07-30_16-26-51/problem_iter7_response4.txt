```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Weight by Volume
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume'].shift(-1)
    
    # Dynamic Lookback Periods
    short_term_days = 5
    mid_term_days = 10
    long_term_days = 20
    
    # Short-Term Lookback
    df['Short_MA'] = df['Volume_Weighted_Return'].rolling(window=short_term_days).mean()
    df['Short_STD'] = df['Volume_Weighted_Return'].rolling(window=short_term_days).std()
    
    # Mid-Term Lookback
    df['Mid_MA'] = df['Volume_Weighted_Return'].rolling(window=mid_term_days).mean()
    df['Mid_STD'] = df['Volume_Weighted_Return'].rolling(window=mid_term_days).std()
    
    # Long-Term Lookback
    df['Long_MA'] = df['Volume_Weighted_Return'].rolling(window=long_term_days).mean()
    df['Long_STD'] = df['Volume_Weighted_Return'].rolling(window=long_term_days).std()
    
    # Stabilize with Multi-Period Volatility
    # Combine Short-Term and Long-Term Volatilities
    df['Combined_Volatility'] = (df['Short_STD'] * 0.5 + df['Long_STD'] * 0.5)
    
    # Final Factor
    df['Final_Factor'] = df['Volume_Weighted_Return'] / df['Combined_Volatility']
    
    return df['Final_Factor'].shift(1)  # Shift to align with the next day's return
```
