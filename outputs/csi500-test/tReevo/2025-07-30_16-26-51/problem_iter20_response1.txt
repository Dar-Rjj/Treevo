```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['EMA_5'] = df['Volume_Weighted_Return'].ewm(span=short_term, adjust=False).mean()
    df['STD_5'] = df['Volume_Weighted_Return'].ewm(span=short_term, adjust=False).std()
    
    # Mid-Term Lookback
    df['EMA_10'] = df['Volume_Weighted_Return'].ewm(span=mid_term, adjust=False).mean()
    df['STD_10'] = df['Volume_Weighted_Return'].ewm(span=mid_term, adjust=False).std()
    
    # Long-Term Lookback
    df['EMA_20'] = df['Volume_Weighted_Return'].ewm(span=long_term, adjust=False).mean()
    df['STD_20'] = df['Volume_Weighted_Return'].ewm(span=long_term, adjust=False).std()
    
    # Combine Multi-Period Volatilities
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    df['Combined_Volatility'] = (short_weight * df['STD_5'] + 
                                 mid_weight * df['STD_10'] + 
                                 long_weight * df['STD_20'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['EMA_Close_5'] = df['close'].ewm(span=short_term, adjust=False).mean()
    df['EMA_Close_20'] = df['close'].ewm(span=long_term, adjust=False).mean()
    df['Price_Trend_Indicator'] = np.where(df['EMA_Close_5'] > df['EMA_Close_20'], 1, -1)
    
    # Integrate Momentum
    df['ROC_5'] = df['close'] / df['close'].shift(short_term) - 1
    df['ROC_20'] = df['close'] / df['close'].shift(long_term) - 1
    df['Momentum_Indicator'] = np.where((df['ROC_5'] > 0) & (df['ROC_20'] > 0), 1, -1)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'] / df['Combined_Volatility'] * 
                          df['Price_Trend_Indicator'] * df['Momentum_Indicator'])
    
    return df['Alpha_Factor']
```
