```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_ret'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['short_sma'] = df['vw_ret'].rolling(window=short_term).mean()
    df['short_std'] = df['vw_ret'].rolling(window=short_term).std()
    
    # Mid-Term Lookback
    df['mid_sma'] = df['vw_ret'].rolling(window=mid_term).mean()
    df['mid_std'] = df['vw_ret'].rolling(window=mid_term).std()
    
    # Long-Term Lookback
    df['long_sma'] = df['vw_ret'].rolling(window=long_term).mean()
    df['long_std'] = df['vw_ret'].rolling(window=long_term).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    df['combined_volatility'] = (short_weight * df['short_std'] + 
                                 mid_weight * df['mid_std'] + 
                                 long_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma'] = df['close'].rolling(window=short_term).mean()
    df['20_day_ma'] = df['close'].rolling(window=long_term).mean()
    
    # Price Trend Indicator
    df['price_trend'] = np.where(df['5_day_ma'] > df['20_day_ma'], 1, -1)
    
    # Integrate Momentum
    df['momentum'] = (df['close'] - df['close'].shift(mid_term)) / df['close'].shift(mid_term)
    
    # Final Alpha Factor
    positive_trend_multiplier = 1.2
    negative_trend_multiplier = 0.8
    momentum_multiplier = 1.1
    inverse_momentum_multiplier = 0.9
    
    df['alpha_factor'] = (
        df['vw_ret'] / df['combined_volatility'] * 
        np.where(df['price_trend'] > 0, positive_trend_multiplier, negative_trend_multiplier) * 
        np.where(df['momentum'] > 0, momentum_multiplier, inverse_momentum_multiplier)
    )
    
    return df['alpha_factor']
```
