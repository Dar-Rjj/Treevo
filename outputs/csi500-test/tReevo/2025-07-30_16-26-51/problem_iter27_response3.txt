```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5D_SMA_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10D_SMA_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20D_SMA_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_Close_to_Open_Return'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3
    
    df['Combined_Volatility'] = (df['5D_STD_Close_to_Open_Return'] * short_term_volatility_weight +
                                 df['10D_STD_Close_to_Open_Return'] * mid_term_volatility_weight +
                                 df['20D_STD_Close_to_Open_Return'] * long_term_volatility_weight)
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_MA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20D_MA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = 0
    df.loc[df['5D_MA_Close'] > df['20D_MA_Close'], 'Price_Trend_Indicator'] = 1.1
    df.loc[df['5D_MA_Close'] < df['20D_MA_Close'], 'Price_Trend_Indicator'] = 0.9
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['5D_SMA_Close_to_Open_Return'] / df['Combined_Volatility']
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Price_Trend_Indicator']
    
    return df['Final_Alpha_Factor']
```
