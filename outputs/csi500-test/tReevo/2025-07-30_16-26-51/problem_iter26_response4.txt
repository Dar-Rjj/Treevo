```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    # Short-Term Lookback
    df['short_term_ema'] = df['volume_weighted_return'].ewm(span=short_term_lookback, adjust=False).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()

    # Mid-Term Lookback
    df['mid_term_ema'] = df['volume_weighted_return'].ewm(span=mid_term_lookback, adjust=False).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()

    # Long-Term Lookback
    df['long_term_ema'] = df['volume_weighted_return'].ewm(span=long_term_lookback, adjust=False).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities with Dynamic Weights
    recent_volatility = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    short_term_volatility_weight = 1 / (1 + np.exp(-recent_volatility))
    mid_term_volatility_weight = 1 / (1 + np.exp(recent_volatility))
    long_term_volatility_weight = 1 - short_term_volatility_weight - mid_term_volatility_weight

    df['combined_volatility'] = (short_term_volatility_weight * df['short_term_std'] +
                                 mid_term_volatility_weight * df['mid_term_std'] +
                                 long_term_volatility_weight * df['long_term_std'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ema_close'] = df['close'].ewm(span=short_term_lookback, adjust=False).mean()
    df['20_day_ema_close'] = df['close'].ewm(span=long_term_lookback, adjust=False).mean()
    df['price_trend_indicator'] = (df['5_day_ema_close'] > df['20_day_ema_close']).astype(int)

    df['adjusted_factor'] = df['combined_volatility'] * df['price_trend_indicator']

    # Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['adjusted_factor']

    return df['alpha_factor']
```
