```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Short-Term Lookback (5 days)
    df['SMA_5_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['STD_5_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    df['SMA_10_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=10).mean()
    df['STD_10_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    df['SMA_20_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['STD_20_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=20).std()

    # Assign Weights to Each Volatility
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3

    # Combine Multi-Period Volatilities
    df['Combined_Volatility'] = (short_term_weight * df['STD_5_Volume_Weighted_Return'] +
                                 mid_term_weight * df['STD_10_Volume_Weighted_Return'] +
                                 long_term_weight * df['STD_20_Volume_Weighted_Return'])

    # Calculate 5-Day and 20-Day Moving Averages of Close Prices
    df['SMA_5_Close'] = df['close'].rolling(window=5).mean()
    df['SMA_20_Close'] = df['close'].rolling(window=20).mean()

    # Price Trend Indicator
    df['Price_Trend_Indicator'] = (df['SMA_5_Close'] > df['SMA_20_Close']).astype(int) * 2 - 1

    # Combined Volume-Weighted Close-to-Open Return
    combined_return = df['SMA_5_Volume_Weighted_Return']

    # Final Alpha Factor
    df['Alpha_Factor'] = (combined_return / df['Combined_Volatility']) * df['Price_Trend_Indicator']

    return df['Alpha_Factor']
```
