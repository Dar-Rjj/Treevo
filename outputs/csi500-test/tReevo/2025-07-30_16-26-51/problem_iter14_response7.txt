```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'].shift(-1) / df['open'] - 1
    
    # Volume-Weight the Return
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume'].shift(-1)
    
    # Short-Term Lookback (5 days)
    df['5_day_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['5_day_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=5, adjust=False).mean()
    
    # Mid-Term Lookback (10 days)
    df['10_day_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=10).mean()
    df['10_day_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    
    # Long-Term Lookback (20 days)
    df['20_day_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['20_day_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    short_term_vol = df['Volume_Weighted_Return'].rolling(window=5).std()
    long_term_vol = df['Volume_Weighted_Return'].rolling(window=20).std()
    adaptive_vol = 0.6 * short_term_vol + 0.4 * long_term_vol
    
    # Integrated Momentum and Adaptive Volatility
    integrated_momentum = (df['5_day_SMA_VW_Return'] + df['10_day_EMA_VW_Return'] + df['20_day_EMA_VW_Return']) / adaptive_vol
    
    # Incorporate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Integrate High-Low Momentum
    high_low_momentum = df['High_Low_Ratio'].rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = integrated_momentum + high_low_momentum + df['Volume_Weighted_Return']
    
    return alpha_factor
```
