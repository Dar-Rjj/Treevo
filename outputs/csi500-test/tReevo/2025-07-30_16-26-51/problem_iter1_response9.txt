import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Compute Average True Range (ATR) over N days
    df['atr'] = df['true_range'].rolling(window=N).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(N)
    
    # Adjust Momentum by ATR
    df['adjusted_momentum'] = df['price_momentum'] / df['atr']
    
    # Sign of the Adjusted Momentum
    def sign_adjusted_momentum(x):
        if x > 0:
            return 1
        elif x < 0:
            return -1
        else:
            return 0
    
    df['factor'] = df['adjusted_momentum'].apply(sign_adjusted_momentum)
    
    return df['factor']
