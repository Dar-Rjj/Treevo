```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['5_day_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['10_day_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['20_day_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatility
    df['combined_volatility'] = (df['5_day_std'] + df['10_day_std'] + df['20_day_std']) / 3
    
    # Integrate Market Signals
    df['high_low_range'] = df['high'] - df['low']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Final Factor
    df['final_factor'] = (df['volume_weighted_return'] / df['combined_volatility'])
    df['final_factor'] = np.log(df['final_factor'])  # Logarithmic Transformation
    df['final_factor'] = np.sqrt(df['final_factor'])  # Square Root Transformation
    
    return df['final_factor']
```
