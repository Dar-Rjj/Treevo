```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_c2o = close_to_open_return * df['volume']
    
    # Calculate 5-Day Realized Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Combine Factors: Volume-Weighted Close-to-Open Return divided by 5-Day Realized Volatility
    combined_factor = volume_weighted_c2o / realized_volatility
    
    # Calculate 5-Day Momentum
    five_day_momentum = df['close'] - df['close'].shift(5)
    volume_weighted_5dm = five_day_momentum * df['volume']
    
    # Calculate 10-Day Momentum
    ten_day_momentum = df['close'] - df['close'].shift(10)
    volume_weighted_10dm = ten_day_momentum * df['volume']
    
    # Combine Different Time Frame Momentums
    combined_momentum = volume_weighted_5dm + volume_weighted_10dm
    
    # Smooth Combined Momentum using EMA with a span of 5 days
    smoothed_momentum = combined_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Factor
    final_factor = combined_factor + smoothed_momentum
    
    return final_factor
```
