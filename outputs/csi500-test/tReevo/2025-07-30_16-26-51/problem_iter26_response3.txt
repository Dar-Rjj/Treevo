```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] / df['open'].shift(1) - 1) * df['volume']

    # Short-Term Lookback (5 days)
    short_ma = df['vw_return'].rolling(window=5).mean()
    short_std = df['vw_return'].rolling(window=5).std()

    # Mid-Term Lookback (10 days)
    mid_ma = df['vw_return'].rolling(window=10).mean()
    mid_std = df['vw_return'].rolling(window=10).std()

    # Long-Term Lookback (20 days)
    long_ma = df['vw_return'].rolling(window=20).mean()
    long_std = df['vw_return'].rolling(window=20).std()

    # Combine Multi-Period Volatilities with Weights
    short_vol_weight = 0.4
    mid_vol_weight = 0.3
    long_vol_weight = 0.3
    combined_volatility = (short_vol_weight * short_std + 
                           mid_vol_weight * mid_std + 
                           long_vol_weight * long_std)

    # Incorporate Dynamic Market Trend Adjustments
    short_close_ma = df['close'].rolling(window=5).mean()
    long_close_ma = df['close'].rolling(window=20).mean()
    trend_indicator = np.where(short_close_ma > long_close_ma, 1, -1)

    # Final Alpha Factor
    alpha_factor = (short_ma / combined_volatility) * trend_indicator

    return alpha_factor
```
