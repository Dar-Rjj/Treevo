```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate 5-Day Realized Volatility
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['realized_volatility'] = df['log_return'].rolling(window=5).std()
    
    # Combine Factors
    df['combined_factor'] = df['volume_weighted_close_to_open_return'] / df['realized_volatility']
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    df['5_day_volume_weighted_momentum'] = df['5_day_momentum'] * df['volume']
    
    # Calculate 10-Day Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    df['10_day_volume_weighted_momentum'] = df['10_day_momentum'] * df['volume']
    
    # Combine Different Time Frame Momentums
    df['combined_momentum'] = df['5_day_volume_weighted_momentum'] + df['10_day_volume_weighted_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=5, min_periods=5).mean()
    
    # Adjust for Short-Term Volatility
    df['adjusted_smoothed_combined_momentum'] = df['smoothed_combined_momentum'] / df['realized_volatility']
    
    # Final Factor
    df['final_factor'] = df['combined_factor'] + df['adjusted_smoothed_combined_momentum']
    
    return df['final_factor']
```
