import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Multi-Period Momentum Indicators
    short_term_sma = df['vw_return'].rolling(window=5).mean()
    short_term_ema = df['vw_return'].ewm(span=5, adjust=False).mean()
    mid_term_sma = df['vw_return'].rolling(window=10).mean()
    mid_term_ema = df['vw_return'].ewm(span=10, adjust=False).mean()
    long_term_sma = df['vw_return'].rolling(window=20).mean()
    long_term_ema = df['vw_return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_vol = log_returns.rolling(window=5).std()
    long_term_vol = log_returns.rolling(window=20).std()
    adaptive_vol = 0.6 * short_term_vol + 0.4 * long_term_vol
    
    # Market Regime Detection
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_trend = np.where(ma_5 > ma_20, 1, -1)
    
    # Combine Momentum Signals
    combined_momentum = 0.5 * (short_term_sma + short_term_ema) + \
                        0.3 * (mid_term_sma + mid_term_ema) + \
                        0.2 * (long_term_sma + long_term_ema)
    
    # Market Regime Adjustment
    regime_adjusted_momentum = np.where(price_trend == 1, 1.2 * combined_momentum, 0.8 * combined_momentum)
    
    # Final Alpha Factor
    alpha_factor = (short_term_sma + mid_term_ema + long_term_ema) / adaptive_vol
    alpha_factor = np.where(price_trend == 1, 1.2 * alpha_factor, 0.8 * alpha_factor)
    
    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
