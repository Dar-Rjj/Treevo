import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum Component
    df['daily_diff'] = df['close'].diff()
    ewa_decay = 0.9  # Adaptive decay rate for EWA (e.g., based on recent volatility)
    df['momentum_ewa'] = df['daily_diff'].ewm(alpha=ewa_decay).mean()

    # Calculate Overbought or Oversold Pressure
    sma_window = 20  # Adaptive window for SMA
    df['sma_20'] = df['close'].rolling(window=sma_window).mean()
    df['days_above_sma'] = (df['close'] > df['sma_20']).astype(int).rolling(window=sma_window).sum()
    df['days_below_sma'] = (df['close'] < df['sma_20']).astype(int).rolling(window=sma_window).sum()
    df['overbought_signal'] = df['days_above_sma'] / sma_window
    df['oversold_signal'] = df['days_below_sma'] / sma_window
    overbought_bound = 0.8  # Bound to avoid extreme values
    oversold_bound = 0.2
    df['overbought_signal'] = df['overbought_signal'].clip(upper=overbought_bound)
    df['oversold_signal'] = df['oversold_signal'].clip(lower=oversold_bound)

    # Incorporate Adaptive Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_20'] = df['daily_return'].rolling(window=sma_window).std()
    df['inv_volatility_20'] = 1 / df['volatility_20']
    df['scaled_momentum'] = df['momentum_ewa'] * df['inv_volatility_20']
    df['scaled_overbought'] = df['overbought_signal'] * df['inv_volatility_20']
    df['scaled_oversold'] = df['oversold_signal'] * df['inv_volatility_20']

    # Incorporate Trend Strength
    long_sma_window = 50  # Adaptive window for longer SMA
    df['sma_50'] = df['close'].rolling(window=long_sma_window).mean()
    df['trend_strength'] = df['sma_50'] - df['sma_20']
    df['scaled_trend_strength'] = df['trend_strength'] * df['inv_volatility_20']

    # Incorporate Adaptive Liquidity
    df['avg_volume_20'] = df['volume'].rolling(window=sma_window).mean()
    df['scaled_combined_factor'] = (df['scaled_momentum'] + 
                                    df['scaled_overbought'] - 
                                    df['scaled_oversold'] + 
                                    df['scaled_trend_strength']) * df['avg_volume_20']

    # Combine All Components
    momentum_weight = 0.6  # High weight to daily momentum
    overbought_weight = 0.1  # Low weight to overbought signal
    trend_weight = 0.3  # Moderate weight to trend strength

    df['final_alpha_factor'] = (momentum_weight * df['scaled_momentum'] +
                                overbought_weight * df['scaled_overbought'] * (df['overbought_signal'] > 0.5) -
                                overbought_weight * df['scaled_oversold'] * (df['oversold_signal'] > 0.5) +
                                trend_weight * df['scaled_trend_strength'])

    return df['final_alpha_factor']
