import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_sma = df['Volume_Weighted_Return'].rolling(window=5).mean()
    short_term_std = df['Volume_Weighted_Return'].rolling(window=5).std()

    mid_term_sma = df['Volume_Weighted_Return'].rolling(window=10).mean()
    mid_term_std = df['Volume_Weighted_Return'].rolling(window=10).std()

    long_term_sma = df['Volume_Weighted_Return'].rolling(window=20).mean()
    long_term_std = df['Volume_Weighted_Return'].rolling(window=20).std()

    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3

    combined_volatility = (short_term_volatility_weight * short_term_std +
                           mid_term_volatility_weight * mid_term_std +
                           long_term_volatility_weight * long_term_std)

    # Incorporate Dynamic Market Trend Adjustments
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    price_trend_indicator = (ema_5 - ema_20) / ema_20

    # Final Alpha Factor
    combined_factor = df['Volume_Weighted_Return'].rolling(window=20).mean()
    final_alpha_factor = combined_factor / combined_volatility * price_trend_indicator

    return final_alpha_factor.dropna()
