```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Weight by Volume
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume'].shift(-1)
    
    # Short-Term Lookback (5 days)
    df['5D_SMA'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['5D_STD'] = df['Volume_Weighted_Return'].rolling(window=5).std()
    
    # Long-Term Lookback (20 days)
    df['20D_SMA'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['20D_STD'] = df['Volume_Weighted_Return'].rolling(window=20).std()
    
    # Stabilize with Volatility
    df['Stable_Factor'] = (df['5D_SMA'] / df['5D_STD']) - (df['20D_SMA'] / df['20D_STD'])
    
    # Drop NaN values and return the factor
    factor = df['Stable_Factor'].dropna()
    
    return factor
```
