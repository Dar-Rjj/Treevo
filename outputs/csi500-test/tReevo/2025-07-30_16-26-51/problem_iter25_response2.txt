import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_window = 5
    mid_term_window = 10
    long_term_window = 20
    
    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=short_term_window).mean()
    df['Short_Term_STD'] = df['Close_to_Open_Return'].rolling(window=short_term_window).std()
    
    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=mid_term_window).mean()
    df['Mid_Term_STD'] = df['Close_to_Open_Return'].rolling(window=mid_term_window).std()
    
    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=long_term_window).mean()
    df['Long_Term_STD'] = df['Close_to_Open_Return'].rolling(window=long_term_window).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (df['Short_Term_STD'] * short_term_weight +
                                 df['Mid_Term_STD'] * mid_term_weight +
                                 df['Long_Term_STD'] * long_term_weight)
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_MA'] = df['close'].rolling(window=short_term_window).mean()
    df['20_Day_MA'] = df['close'].rolling(window=long_term_window).mean()
    
    df['Price_Trend_Indicator'] = (df['5_Day_MA'] > df['20_Day_MA']).astype(int)
    
    # Adjust Combined Factor by Price Trend Indicator
    df['Combined_Factor_Adjusted'] = df['Combined_Volatility']
    df.loc[df['Price_Trend_Indicator'] == 1, 'Combined_Factor_Adjusted'] *= 1.1
    df.loc[df['Price_Trend_Indicator'] == 0, 'Combined_Factor_Adjusted'] *= 0.9
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Close_to_Open_Return'] / df['Combined_Factor_Adjusted']
    
    return df['Final_Alpha_Factor']
