import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the relative volume compared to the 30-day average
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume_30']
    
    # Calculate the volatility of the price change over the last 10 days
    df['volatility_10'] = df['price_change'].rolling(window=10).std()
    
    # Calculate the ratio of the amount traded to the volume
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Calculate the RSI (Relative Strength Index) for the closing prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI'] = 100 - (100 / (1 + RS))
    
    # Combine the factors into a single score
    df['factor'] = (df['price_change'] * df['relative_volume'] 
                    / df['volatility_10'] * df['amount_per_volume'] 
                    * (1 - df['RSI'] / 100))
    
    return df['factor'].dropna()
