```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] / df['open'].shift(1) - 1
    
    # Volume-Weight the Return
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Dynamic Lookback Periods for Momentum
    df['5d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['5d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=5, adjust=False).mean()
    df['10d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=10).mean()
    df['10d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    df['20d_SMA_VW_Return'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['20d_EMA_VW_Return'] = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Short_Term_Volatility'] = df['Log_Return'].rolling(window=5).std()
    df['Long_Term_Volatility'] = df['Log_Return'].rolling(window=20).std()
    df['Adaptive_Volatility'] = 0.6 * df['Short_Term_Volatility'] + 0.4 * df['Long_Term_Volatility']
    
    # Combine Momentum Signals
    df['Momentum_Signal'] = (0.5 * (df['5d_SMA_VW_Return'] + df['5d_EMA_VW_Return'])) + \
                            (0.3 * (df['10d_SMA_VW_Return'] + df['10d_EMA_VW_Return'])) + \
                            (0.2 * (df['20d_SMA_VW_Return'] + df['20d_EMA_VW_Return']))
    
    # Final Factor with Momentum and Adaptive Volatility
    df['Alpha_Factor'] = (df['5d_SMA_VW_Return'] + df['10d_EMA_VW_Return'] + df['20d_EMA_VW_Return']) / df['Adaptive_Volatility']
    
    return df['Alpha_Factor']
```
