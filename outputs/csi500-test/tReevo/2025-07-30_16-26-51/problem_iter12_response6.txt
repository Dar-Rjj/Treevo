```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'].shift(-1) / df['open'] - 1
    
    # Volume-Weight the Return
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Dynamic Lookback Periods for Momentum
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['SMA_5_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term).mean()
    df['EMA_5_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=short_term, adjust=False).mean()
    
    # Mid-Term Lookback
    df['SMA_10_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term).mean()
    df['EMA_10_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=mid_term, adjust=False).mean()
    
    # Long-Term Lookback
    df['SMA_20_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term).mean()
    df['EMA_20_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=long_term, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['Short_Term_Volatility'] = df['Volume_Weighted_Return'].rolling(window=short_term).std()
    df['Long_Term_Volatility'] = df['Volume_Weighted_Return'].rolling(window=long_term).std()
    df['Adaptive_Volatility'] = 0.6 * df['Short_Term_Volatility'] + 0.4 * df['Long_Term_Volatility']
    
    # Integrated Momentum and Adaptive Volatility
    df['Integrated_Momentum'] = (df['SMA_5_Volume_Weighted_Return'] + 
                                 df['EMA_10_Volume_Weighted_Return'] + 
                                 df['EMA_20_Volume_Weighted_Return']) / 3
    df['Momentum_Adaptive_Volatility'] = df['Integrated_Momentum'] / df['Adaptive_Volatility']
    
    # Incorporate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Integrate High-Low Momentum
    df['High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=short_term).sum()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Momentum_Adaptive_Volatility'] + 
                          df['High_Low_Momentum'] + 
                          df['Volume_Weighted_Return'])
    
    return df['Alpha_Factor']
```
