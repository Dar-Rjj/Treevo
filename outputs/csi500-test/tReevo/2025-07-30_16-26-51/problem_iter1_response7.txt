```python
import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']

    # Calculate High-Low Momentum
    high_low_momentum = high_low_ratio.pct_change(periods=lookback_period)

    # Volume Correction Factor
    volume_sum = df['volume'].rolling(window=lookback_period).sum()
    volume_correction_factor = df['volume'] / volume_sum

    # Multiply by Volume Correction Factor
    factor_values = high_low_momentum * volume_correction_factor
    
    return factor_values
```
