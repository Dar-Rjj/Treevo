```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vwco_return'] = (df['close'].shift(-1) - df['open']) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback (e.g., 5 days)
    df['short_sma'] = df['vwco_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['vwco_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback (e.g., 10 days)
    df['mid_sma'] = df['vwco_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['vwco_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback (e.g., 20 days)
    df['long_sma'] = df['vwco_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['vwco_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_vol_weight = 0.4
    mid_vol_weight = 0.3
    long_vol_weight = 0.3
    
    df['combined_volatility'] = (short_vol_weight * df['short_std'] +
                                 mid_vol_weight * df['mid_std'] +
                                 long_vol_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    df['trend_indicator'] = (df['5_day_ma_close'] > df['20_day_ma_close']).astype(int)
    df['trend_indicator'] = df['trend_indicator'].replace({1: 1, 0: -1})
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vwco_return'] / df['combined_volatility']) * df['trend_indicator']
    
    return df['alpha_factor']
```
