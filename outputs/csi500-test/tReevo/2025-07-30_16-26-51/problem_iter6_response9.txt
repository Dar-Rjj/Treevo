```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']

    # Calculate Weighted High-Low Momentum
    weights = [5, 4, 3, 2, 1]
    high_low_momentum = (df['High_Low_Ratio'].rolling(window=5).apply(
        lambda x: np.sum(x * weights), raw=True)).shift(1)

    # Volume Correction Factor
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    volume_correction_factor = df['volume'] / df['Avg_Volume_20']

    # Calculate Price Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_volatility = log_returns.rolling(window=20).std()

    # Smooth Volume Changes with EMA
    smoothed_volume_changes = df['volume'].ewm(span=20, adjust=False).mean()

    # Market Sentiment Indicator
    daily_range = df['high'] - df['low']
    gap = df['open'] - df['close'].shift(1)
    directional_movement = df['close'] - df['open']
    market_sentiment = (daily_range + gap + directional_movement) / 3

    # Combine all factors
    alpha_factor = (high_low_momentum * volume_correction_factor /
                    price_volatility * smoothed_volume_changes * market_sentiment)

    return alpha_factor
```
