```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Compute Average Close Price (Last 10 Days)
    avg_close_10 = data['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum = data['close'] - avg_close_10
    
    # Compute Average Volume (Last 10 Days)
    avg_volume_10 = data['volume'].rolling(window=10).mean()
    
    # Detect Volume Spike
    volume_spike = (data['volume'] - avg_volume_10) > (2 * avg_volume_10)
    volume_spike = volume_spike.astype(int)
    
    # Combine Price Momentum and Volume Spike to form the final factor
    alpha_factor = price_momentum * volume_spike
    
    return alpha_factor
```
