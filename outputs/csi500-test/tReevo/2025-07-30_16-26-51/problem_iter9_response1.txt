```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5_day_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['10_day_std'] = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20_day_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Adaptive Volatility
    # Combine Short-Term, Mid-Term, and Long-Term Volatilities
    # Dynamic Weighting Based on Market Conditions
    df['combined_volatility'] = (df['5_day_std'] + df['10_day_std'] + df['20_day_std']) / 3
    
    # Final Factor
    df['factor'] = df['volume_weighted_return'] / df['combined_volatility']
    
    return df['factor']
```
