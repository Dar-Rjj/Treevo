import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume'].shift(-1)
    
    # Adaptive Lookback Periods
    lookbacks = [5, 10, 20]
    for lookback in lookbacks:
        df[f'{lookback}D_SMA_VWCTOR'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=lookback).mean()
        df[f'{lookback}D_STD_VWCTOR'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['5D_STD_VWCTOR'] +
                                 mid_term_weight * df['10D_STD_VWCTOR'] +
                                 long_term_weight * df['20D_STD_VWCTOR'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_SMA_Close'] = df['close'].rolling(window=5).mean()
    df['20D_SMA_Close'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = df['5D_SMA_Close'] - df['20D_SMA_Close']
    
    # Final Alpha Factor
    df['Combined_Factor'] = (df['Volume_Weighted_Close_to_Open_Return'] /
                             df['Combined_Volatility'] *
                             df['Price_Trend_Indicator'])
    
    return df['Combined_Factor'].dropna()
