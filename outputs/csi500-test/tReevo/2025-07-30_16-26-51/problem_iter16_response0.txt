```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume-Weight the Return
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Dynamic Lookback Periods for Momentum
    df['5_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['10_day_EMA'] = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    df['20_day_EMA'] = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    short_term_vol = df['Volume_Weighted_Return'].rolling(window=5).std()
    long_term_vol = df['Volume_Weighted_Return'].rolling(window=20).std()
    df['Adaptive_Volatility'] = 0.6 * short_term_vol + 0.4 * long_term_vol
    
    # Integrated Momentum and Adaptive Volatility
    df['Integrated_Momentum'] = (df['5_day_SMA'] + df['10_day_EMA'] + df['20_day_EMA']) / 3
    df['Integrated_Momentum_Adaptive_Vol'] = df['Integrated_Momentum'] / df['Adaptive_Volatility']
    
    # Incorporate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Integrate High-Low Momentum
    df['High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Integrated_Momentum_Adaptive_Vol'] + df['High_Low_Momentum'] + df['Volume_Weighted_Return']
    
    return df['Alpha_Factor']
```
