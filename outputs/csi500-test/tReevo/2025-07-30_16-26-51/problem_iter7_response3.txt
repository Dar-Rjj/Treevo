import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Volume-Weight the Return
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Periods
    lookback_periods = [5, 10, 20]
    for period in lookback_periods:
        df[f'{period}_day_sma'] = df['volume_weighted_return'].rolling(window=period).mean()
        df[f'{period}_day_std'] = df['volume_weighted_return'].rolling(window=period).std()
    
    # Adaptive Volatility Stabilization
    df['short_term_volatility'] = df['5_day_std']
    df['long_term_volatility'] = df['20_day_std']
    df['adaptive_volatility'] = (df['short_term_volatility'] + df['long_term_volatility']) / 2
    
    # Final Factor
    df['factor'] = df['volume_weighted_return'] / df['adaptive_volatility']
    
    return df['factor']
