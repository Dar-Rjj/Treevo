```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_term_ema'] = df['volume_weighted_return'].ewm(span=short_term_lookback, adjust=False).mean()
    df['short_term_std'] = df['volume_weighted_return'].ewm(span=short_term_lookback, adjust=False).std()
    
    # Mid-Term Lookback
    df['mid_term_ema'] = df['volume_weighted_return'].ewm(span=mid_term_lookback, adjust=False).mean()
    df['mid_term_std'] = df['volume_weighted_return'].ewm(span=mid_term_lookback, adjust=False).std()
    
    # Long-Term Lookback
    df['long_term_ema'] = df['volume_weighted_return'].ewm(span=long_term_lookback, adjust=False).mean()
    df['long_term_std'] = df['volume_weighted_return'].ewm(span=long_term_lookback, adjust=False).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.3
    mid_term_volatility_weight = 0.4
    long_term_volatility_weight = 0.3
    
    df['combined_volatility'] = (short_term_volatility_weight * df['short_term_std'] +
                                 mid_term_volatility_weight * df['mid_term_std'] +
                                 long_term_volatility_weight * df['long_term_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['ema_5'] = df['close'].ewm(span=short_term_lookback, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=long_term_lookback, adjust=False).mean()
    
    df['price_trend_indicator'] = (df['ema_5'] - df['ema_20']) / df['ema_20']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_return'] / df['combined_volatility']) * df['price_trend_indicator']
    
    return df['alpha_factor']
```
