```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    df['5D_SMA_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    df['10D_SMA_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    df['20D_SMA_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_VWCTO'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['5D_STD_VWCTO'] + 
                                 mid_term_weight * df['10D_STD_VWCTO'] + 
                                 long_term_weight * df['20D_STD_VWCTO'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_MA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20D_MA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = (df['5D_MA_Close'] - df['20D_MA_Close']) / df['20D_MA_Close']
    
    # High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Integrate High-Low Momentum
    df['5D_High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=short_term_lookback).sum()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Volume_Weighted_Close_to_Open_Return'] / df['Combined_Volatility'] + 
                                df['5D_High_Low_Momentum'] * df['Price_Trend_Indicator'])
    
    return df['Final_Alpha_Factor']
```
