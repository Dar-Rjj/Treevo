import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Identify Directional Movement
    df['up_movement'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['down_movement'] = -df['daily_return'].apply(lambda x: x if x < 0 else 0)
    
    # Cumulative Sum of Positive and Negative Movements
    df['cum_up'] = df['up_movement'].rolling(window=N, min_periods=1).sum()
    df['cum_down'] = df['down_movement'].rolling(window=N, min_periods=1).sum()
    
    # Compute Moving Averages of Up and Down Movements
    df['ma_up'] = df['up_movement'].rolling(window=N, min_periods=1).mean()
    df['ma_down'] = df['down_movement'].rolling(window=N, min_periods=1).mean()
    
    # Momentum Score Difference
    df['momentum_score'] = df['ma_up'] - df['ma_down']
    
    # Volume Weighting
    df['total_volume'] = df['volume'].rolling(window=N, min_periods=1).sum()
    df['factor_value'] = df['momentum_score'] / df['total_volume']
    
    return df['factor_value']
