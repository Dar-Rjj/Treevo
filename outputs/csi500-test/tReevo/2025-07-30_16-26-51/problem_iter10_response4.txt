```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['HighLowRatio'] = df['high'] / df['low']

    # Calculate High-Low Momentum
    df['HighLowMomentum'] = df['HighLowRatio'].rolling(window=5).sum()

    # Calculate Volume Anomaly
    df['VolumeAnomaly'] = df['volume'] - df['volume'].rolling(window=20).mean()

    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open']) / df['open']

    # Weight by Volume
    df['WeightedCloseToOpenReturn'] = df['CloseToOpenReturn'] * df['volume']

    # Calculate 5-Day Realized Volatility
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    df['5DayRealizedVolatility'] = df['LogReturn'].rolling(window=5).std()

    # Smooth Volume Changes
    df['SmoothedVolume'] = df['volume'].ewm(span=20, adjust=False).mean()

    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    df['CombinedFactor'] = (df['HighLowMomentum'] + df['VolumeAnomaly']) * df['SmoothedVolume']

    # Adjust for Volatility
    df['AdjustedFactor'] = df['CombinedFactor'] / df['5DayRealizedVolatility']

    # Incorporate Weighted Close-to-Open Return
    df['AdjustedFactorWithReturn'] = df['AdjustedFactor'] + df['WeightedCloseToOpenReturn']

    # Integrate Additional Price Momentum
    df['5DayPriceMomentum'] = df['close'] - df['close'].shift(5)
    df['FinalAlphaFactor'] = df['AdjustedFactorWithReturn'] + df['5DayPriceMomentum']

    # Drop rows with NaN values
    df.dropna(inplace=True)

    return df['FinalAlphaFactor']
```
