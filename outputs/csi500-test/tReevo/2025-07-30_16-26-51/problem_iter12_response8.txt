```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    df['Short_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['Mid_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['Long_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    
    df['Short_Term_STD'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    df['Mid_Term_STD'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    df['Long_Term_STD'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_volatility_weight * df['Short_Term_STD'] +
                                 mid_term_volatility_weight * df['Mid_Term_STD'] +
                                 long_term_volatility_weight * df['Long_Term_STD'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['20_Day_MA_Close'] = df['close'].rolling(window=20).mean()
    
    df['Price_Trend_Indicator'] = (df['5_Day_MA_Close'] > df['20_Day_MA_Close']).astype(int)
    
    # Optimize Volatility Weights Based on Trend
    def optimize_weights(row):
        if row['Price_Trend_Indicator'] == 1:
            return 0.5, 0.25, 0.25  # Positive trend: increase short-term weight
        else:
            return 0.25, 0.375, 0.375  # Negative trend: decrease short-term weight, increase mid and long-term weights
    
    df['Short_Term_Weight'], df['Mid_Term_Weight'], df['Long_Term_Weight'] = zip(*df.apply(optimize_weights, axis=1))
    
    df['Optimized_Combined_Volatility'] = (df['Short_Term_Weight'] * df['Short_Term_STD'] +
                                           df['Mid_Term_Weight'] * df['Mid_Term_STD'] +
                                           df['Long_Term_Weight'] * df['Long_Term_STD'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Close_to_Open_Return'] / df['Optimized_Combined_Volatility']
    
    return df['Alpha_Factor'].dropna()
```
