```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_ma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_ma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_ma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_volatility_weight = 0.4
    mid_volatility_weight = 0.3
    long_volatility_weight = 0.3
    
    df['combined_volatility'] = (short_volatility_weight * df['short_std'] +
                                 mid_volatility_weight * df['mid_std'] +
                                 long_volatility_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    df['price_trend'] = (df['5_day_ma_close'] > df['20_day_ma_close']).astype(int)
    
    # Apply Positive or Negative Adjustment
    trend_adjustment_factor = 1.5
    df['adjusted_combined_volatility'] = df['combined_volatility'] * (1 + (df['price_trend'] * trend_adjustment_factor - 0.5 * trend_adjustment_factor))
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['adjusted_combined_volatility']
    
    return df['alpha_factor']
```
