```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Multi-Period High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']

    # Calculate Short-Term High-Low Momentum
    short_lookback = 5
    df['short_momentum_sum'] = df['high_low_ratio'].rolling(window=short_lookback).sum()
    df['short_momentum_avg'] = df['high_low_ratio'].rolling(window=short_lookback).mean()

    # Calculate Long-Term High-Low Momentum
    long_lookback = 20
    df['long_momentum_sum'] = df['high_low_ratio'].rolling(window=long_lookback).sum()
    df['long_momentum_avg'] = df['high_low_ratio'].rolling(window=long_lookback).mean()

    # Dynamic Volume Adjustment
    df['volume_sum_short'] = df['volume'].rolling(window=short_lookback).sum()
    df['volume_sum_long'] = df['volume'].rolling(window=long_lookback).sum()
    df['volume_momentum'] = df['volume'] / (df['volume_sum_short'] + df['volume_sum_long'])

    # Market Sentiment Indicator
    df['close_to_close_return'] = df['close'].pct_change()
    df['sentiment_5day'] = df['close_to_close_return'].rolling(window=short_lookback).mean()
    df['sentiment_20day'] = df['close_to_close_return'].rolling(window=long_lookback).mean()

    # Combine Momentum, Volume, and Market Sentiment
    df['combined_momentum'] = (
        df['short_momentum_avg'] * df['volume_momentum'] + 
        df['long_momentum_avg'] * df['volume_momentum'] + 
        (df['sentiment_5day'] + df['sentiment_20day'])
    )

    # Final Factor
    alpha_factor = df['combined_momentum']

    return alpha_factor
```
