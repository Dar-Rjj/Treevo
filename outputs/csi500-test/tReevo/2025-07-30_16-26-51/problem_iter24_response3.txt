```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['5d_sma_vw_return'] = df['vw_return'].rolling(window=short_term).mean()
    df['5d_std_vw_return'] = df['vw_return'].rolling(window=short_term).std()
    
    # Mid-Term Lookback
    df['10d_ema_vw_return'] = df['vw_return'].ewm(span=mid_term, adjust=False).mean()
    df['10d_std_vw_return'] = df['vw_return'].rolling(window=mid_term).std()
    
    # Long-Term Lookback
    df['20d_ema_vw_return'] = df['vw_return'].ewm(span=long_term, adjust=False).mean()
    df['20d_std_vw_return'] = df['vw_return'].rolling(window=long_term).std()
    
    # Combine Multi-Period Volatilities with Dynamic Weights
    recent_volatility_trend = df['vw_return'].rolling(window=short_term).std().pct_change()
    df['short_term_vol_weight'] = 1 + recent_volatility_trend
    df['mid_term_vol_weight'] = 1 + recent_volatility_trend.rolling(window=mid_term).mean()
    df['long_term_vol_weight'] = 1 + recent_volatility_trend.rolling(window=long_term).mean()
    
    df['combined_volatility'] = (
        df['short_term_vol_weight'] * df['5d_std_vw_return'] +
        df['mid_term_vol_weight'] * df['10d_std_vw_return'] +
        df['long_term_vol_weight'] * df['20d_std_vw_return']
    )
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5d_ma_close'] = df['close'].rolling(window=short_term).mean()
    df['20d_ma_close'] = df['close'].rolling(window=long_term).mean()
    
    df['price_trend'] = np.where(df['5d_ma_close'] > df['20d_ma_close'], 1, -1)
    
    df['adjusted_factor'] = np.where(
        df['price_trend'] == 1,
        1.2 * df[['5d_sma_vw_return', '10d_ema_vw_return', '20d_ema_vw_return']].mean(axis=1),
        0.8 * df[['5d_sma_vw_return', '10d_ema_vw_return', '20d_ema_vw_return']].mean(axis=1)
    )
    
    # Integrate High-Low Price Ratio
    df['high_low_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Incorporate Additional Market Indicators
    df['daily_price_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5d_avg_daily_price_change'] = df['daily_price_change'].rolling(window=short_term).mean()
    
    df['final_adjusted_factor'] = np.where(
        df['5d_avg_daily_price_change'] > 0,
        1.2 * df['adjusted_factor'],
        0.8 * df['adjusted_factor']
    )
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['final_adjusted_factor'] / df['combined_volatility'] * df['high_low_ratio']
    )
    
    return df['alpha_factor'].dropna()
```
