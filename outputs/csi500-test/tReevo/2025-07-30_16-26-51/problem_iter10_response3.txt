```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Define lookback periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Calculate Simple Moving Averages and Standard Deviations for each lookback period
    short_ma = volume_weighted_return.rolling(window=short_term_lookback).mean()
    short_std = volume_weighted_return.rolling(window=short_term_lookback).std()
    
    mid_ma = volume_weighted_return.rolling(window=mid_term_lookback).mean()
    mid_std = volume_weighted_return.rolling(window=mid_term_lookback).std()
    
    long_ma = volume_weighted_return.rolling(window=long_term_lookback).mean()
    long_std = volume_weighted_return.rolling(window=long_term_lookback).std()
    
    # Assign weights to each volatility
    short_weight = 0.4
    mid_weight = 0.3
    long_weight = 0.3
    
    # Combine Multi-Period Volatilities
    combined_volatility = (short_weight * short_std + 
                           mid_weight * mid_std + 
                           long_weight * long_std)
    
    # Final Factor: Combined Volume-Weighted Close-to-Open Return divided by Combined Volatility
    final_factor = (short_ma + mid_ma + long_ma) / combined_volatility
    
    return final_factor
```
