```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5D_SMA_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10D_SMA_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20D_SMA_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_VW_Return'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_vol_weight * df['5D_STD_VW_Return'] +
                                 mid_term_vol_weight * df['10D_STD_VW_Return'] +
                                 long_term_vol_weight * df['20D_STD_VW_Return'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_MA_Close'] = df['close'].rolling(window=5).mean()
    df['20D_MA_Close'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = df['5D_MA_Close'] - df['20D_MA_Close']
    
    # Adjust Combined Factor by Price Trend Indicator
    df['Adjusted_Combined_Factor'] = df['Combined_Volatility'] + df['Price_Trend_Indicator']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Weighted_Close_to_Open_Return'] / df['Adjusted_Combined_Factor']
    
    return df['Final_Alpha_Factor'].dropna()
```
