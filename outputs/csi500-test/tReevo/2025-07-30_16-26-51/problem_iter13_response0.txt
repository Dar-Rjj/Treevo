import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5D_SMA_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10D_SMA_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20D_SMA_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_Close_to_Open'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_vol_weight * df['5D_STD_Close_to_Open'] +
                                 mid_term_vol_weight * df['10D_STD_Close_to_Open'] +
                                 long_term_vol_weight * df['20D_STD_Close_to_Open'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_SMA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20D_SMA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = 0
    df.loc[df['5D_SMA_Close'] > df['20D_SMA_Close'], 'Price_Trend_Indicator'] = 1
    df.loc[df['5D_SMA_Close'] < df['20D_SMA_Close'], 'Price_Trend_Indicator'] = -1
    
    positive_trend_adjustment = 1.1
    negative_trend_adjustment = 0.9
    
    df['Combined_Factor'] = df['5D_SMA_Close_to_Open'] / df['Combined_Volatility']
    df['Trend_Adjusted_Combined_Factor'] = (
        df['Combined_Factor'] * 
        (df['Price_Trend_Indicator'].apply(lambda x: positive_trend_adjustment if x == 1 else negative_trend_adjustment))
    )
    
    return df['Trend_Adjusted_Combined_Factor']

# Example usage:
# factor_values = heuristics_v2(your_dataframe)
