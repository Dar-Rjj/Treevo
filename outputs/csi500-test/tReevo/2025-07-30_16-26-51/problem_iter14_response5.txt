```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) / df['open'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_mavg'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_mavg'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_mavg'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatility
    df['combined_volatility'] = (df['short_std'] + df['mid_std'] + df['long_std']) / 3
    
    # Integrate Market Signals
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Final Factor
    df['final_factor'] = df['volume_weighted_return'] / df['combined_volatility']
    
    # Apply Nonlinear Transformation
    df['log_transformed_factor'] = np.log(np.abs(df['final_factor'])) * np.sign(df['final_factor'])
    df['sqrt_transformed_factor'] = np.sqrt(np.abs(df['log_transformed_factor'])) * np.sign(df['log_transformed_factor'])
    
    # Output Final Alpha Factor
    return df['sqrt_transformed_factor']
```
