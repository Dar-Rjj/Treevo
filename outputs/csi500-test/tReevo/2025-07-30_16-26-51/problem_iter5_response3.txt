import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['vol_weighted_c2o_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['short_mavg'] = df['vol_weighted_c2o_return'].rolling(window=5).mean()
    df['short_std'] = df['vol_weighted_c2o_return'].rolling(window=5).std()
    
    # Long-Term Lookback (20 days)
    df['long_mavg'] = df['vol_weighted_c2o_return'].rolling(window=20).mean()
    df['long_std'] = df['vol_weighted_c2o_return'].rolling(window=20).std()
    
    # Combine Short-Term and Long-Term Volatility
    def select_volatility(row):
        if row['short_std'] > row['long_std']:
            return 'short'
        else:
            return 'long'

    df['selected_volatility'] = df.apply(select_volatility, axis=1)
    
    # Adjust Volume-Weighted Close-to-Open Return by Selected Volatility
    def adjust_by_volatility(row):
        if row['selected_volatility'] == 'short':
            return row['vol_weighted_c2o_return'] / row['short_std'] if row['short_std'] != 0 else 0
        else:
            return row['vol_weighted_c2o_return'] / row['long_std'] if row['long_std'] != 0 else 0

    df['alpha_factor'] = df.apply(adjust_by_volatility, axis=1)
    
    return df['alpha_factor']
