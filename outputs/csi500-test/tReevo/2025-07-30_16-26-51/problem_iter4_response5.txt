```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['High-Low_Ratio'] = df['high'] / df['low']
    
    # Calculate High-Low Momentum (sum of the last 5 days' High-Low Price Ratios)
    df['High-Low_Momentum'] = df['High-Low_Ratio'].rolling(window=5).sum()
    
    # Calculate Dynamic Volume Factor
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Dynamic_Volume_Factor'] = df['volume'] / df['Average_Volume_20']
    
    # Calculate Price Volatility
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Volatility'] = df['Log_Return'].rolling(window=20).std()
    
    # Smooth Volume Changes using EMA with a span of 20 days
    df['Smoothed_Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Integrate Dynamic Volume and Smoothed Volume
    df['Integrated_Volume_Factor'] = df['Dynamic_Volume_Factor'] * (df['volume'] / df['Smoothed_Volume'])
    
    # Adjust High-Low Momentum with Volume and Volatility
    df['Adjusted_High-Low_Momentum'] = (df['High-Low_Momentum'] * df['Integrated_Volume_Factor']) / df['Price_Volatility']
    
    # Final Factor Value
    factor_values = df['Adjusted_High-Low_Momentum']
    
    return factor_values
```
