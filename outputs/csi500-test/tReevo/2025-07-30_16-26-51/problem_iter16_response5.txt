import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods for Momentum
    short_term_sma = df['volume_weighted_return'].rolling(window=5).mean()
    mid_term_ema = df['volume_weighted_return'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['volume_weighted_return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    short_term_volatility = df['volume_weighted_return'].rolling(window=5).std()
    long_term_volatility = df['volume_weighted_return'].rolling(window=20).std()
    adaptive_volatility = 0.6 * short_term_volatility + 0.4 * long_term_volatility
    
    # Integrated Momentum and Adaptive Volatility
    integrated_momentum = (short_term_sma + mid_term_ema + long_term_ema) / adaptive_volatility
    
    # High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # High-Low Momentum
    high_low_momentum = high_low_ratio.rolling(window=5).sum()
    
    # Intraday High-Low Range
    intraday_range = (df['high'] - df['low']) / df['low']
    
    # Intraday Volume Spike
    volume_spike = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Market Sentiment
    market_sentiment = (intraday_range + volume_spike) / 2
    
    # Final Alpha Factor
    alpha_factor = (integrated_momentum + high_low_momentum + df['volume_weighted_return']) * market_sentiment
    
    return alpha_factor
