```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate 5-Day High-Low Momentum
    df['high_low_momentum_5d'] = df['high_low_ratio'].rolling(window=5).sum()
    
    # Calculate Volume Anomaly
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_anomaly'] = df['volume'] - df['avg_volume_20d']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate 5-Day Realized Volatility
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['realized_volatility_5d'] = df['log_return'].rolling(window=5).std()
    
    # Smooth Volume Changes
    df['smoothed_volume_ema_20d'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    df['combined_factor'] = (df['high_low_momentum_5d'] + df['volume_anomaly']) * df['smoothed_volume_ema_20d']
    
    # Adjust for Volatility
    df['adjusted_factor'] = df['combined_factor'] / df['realized_volatility_5d']
    
    # Incorporate Weighted Close-to-Open Return
    df['final_alpha_factor'] = df['adjusted_factor'] + df['weighted_close_to_open_return']
    
    return df['final_alpha_factor'].dropna()
```
