```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_sma = df['Close_to_Open_Return'].rolling(window=5).mean()
    short_term_std = df['Close_to_Open_Return'].rolling(window=5).std()
    
    mid_term_sma = df['Close_to_Open_Return'].rolling(window=10).mean()
    mid_term_std = df['Close_to_Open_Return'].rolling(window=10).std()
    
    long_term_sma = df['Close_to_Open_Return'].rolling(window=20).mean()
    long_term_std = df['Close_to_Open_Return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    combined_volatility = (0.4 * short_term_std) + (0.3 * mid_term_std) + (0.3 * long_term_std)
    
    # Incorporate Dynamic Market Trend Adjustments
    five_day_ma_close = df['close'].rolling(window=5).mean()
    twenty_day_ma_close = df['close'].rolling(window=20).mean()
    
    price_trend_indicator = np.where(five_day_ma_close > twenty_day_ma_close, 1, -1)
    
    adjusted_factor = np.where(price_trend_indicator == 1, 1.2 * short_term_sma, 0.8 * short_term_sma)
    
    # Integrate High-Low Price Ratio
    high_low_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_factor / combined_volatility) * high_low_ratio
    
    return final_alpha_factor
```
