```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    df['High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=5).sum()
    
    # Volume Correction Factor
    df['Volume_Correction'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Price Volatility
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Volatility'] = df['Log_Return'].rolling(window=20).std()
    
    # Smooth Volume Changes
    df['Smoothed_Volume_Change'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine Momentum, Volume Correction, and Volatility
    df['Alpha_Factor'] = (df['High_Low_Momentum'] * df['Volume_Correction']) / df['Price_Volatility'] * df['Smoothed_Volume_Change']
    
    return df['Alpha_Factor']
```
