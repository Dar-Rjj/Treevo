import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.3
    mid_term_volatility_weight = 0.4
    long_term_volatility_weight = 0.3
    
    df['combined_volatility'] = (df['short_term_std'] * short_term_volatility_weight +
                                 df['mid_term_std'] * mid_term_volatility_weight +
                                 df['long_term_std'] * long_term_volatility_weight)

    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_sma_close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20_day_sma_close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['price_trend_indicator'] = df['5_day_sma_close'] - df['20_day_sma_close']
    df['price_trend_indicator'] = df['price_trend_indicator'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['volume_weighted_return'] / df['combined_volatility']) * df['price_trend_indicator']
    
    return df['final_alpha_factor']
