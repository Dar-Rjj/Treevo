```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    short_term_ema = df['volume_weighted_return'].ewm(span=5).mean()
    short_term_std = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    mid_term_ema = df['volume_weighted_return'].ewm(span=10).mean()
    mid_term_std = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    long_term_ema = df['volume_weighted_return'].ewm(span=20).mean()
    long_term_std = df['volume_weighted_return'].rolling(window=20).std()
    
    # Adaptive Volatility Stabilization
    adaptive_volatility = 0.6 * short_term_std + 0.4 * long_term_std
    
    # Integrated Momentum and Adaptive Volatility
    integrated_momentum = (short_term_ema + mid_term_ema + long_term_ema) / adaptive_volatility
    
    # High-Low Price Ratio
    df['high_low_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Sum the last 5, 10, and 20 days' High-Low Price Ratios
    short_term_high_low = df['high_low_ratio'].rolling(window=5).sum()
    mid_term_high_low = df['high_low_ratio'].rolling(window=10).sum()
    long_term_high_low = df['high_low_ratio'].rolling(window=20).sum()
    
    # Adaptive High-Low Momentum
    adaptive_high_low_momentum = 0.3 * short_term_high_low + 0.4 * mid_term_high_low + 0.3 * long_term_high_low
    
    # Final Alpha Factor
    final_alpha_factor = integrated_momentum + adaptive_high_low_momentum + df['volume_weighted_return']
    
    return final_alpha_factor
```
