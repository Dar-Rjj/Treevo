```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Apply Dynamic Lookback Period
    ma_20 = close_to_open_return.rolling(window=20).mean()
    ma_60 = close_to_open_return.rolling(window=60).mean()
    
    # Adjust for Volatility
    volatility_20 = close_to_open_return.rolling(window=20).std()
    adjusted_return = (volume_weighted_return - ma_60 + ma_20) / volatility_20
    
    return adjusted_return
```
