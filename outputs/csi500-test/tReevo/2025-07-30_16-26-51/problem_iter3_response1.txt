```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'] / df['close'].shift(1) - 1
    
    # Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_CtoO_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Calculate Simple Moving Average (SMA)
    lookback_period = 20
    df['SMA'] = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['SMA']
    
    # Composite Alpha Factor
    df['Alpha_Factor'] = df['Volume_Weighted_CtoO_Return'] + df['Price_Momentum']
    
    return df['Alpha_Factor']
```
