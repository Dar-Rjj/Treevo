```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Dynamic Lookback Period
    # Short-Term Lookback (5 days)
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5_day_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Long-Term Lookback (20 days)
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20_day_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Stabilize with Volatility
    # Short-Term Stabilization
    df['short_term_stable_factor'] = df['5_day_sma'] / df['5_day_std']
    
    # Long-Term Stabilization
    df['long_term_stable_factor'] = df['20_day_sma'] / df['20_day_std']
    
    # Final factor: average of short-term and long-term stabilized factors
    df['factor'] = 0.5 * df['short_term_stable_factor'] + 0.5 * df['long_term_stable_factor']
    
    return df['factor']
```
