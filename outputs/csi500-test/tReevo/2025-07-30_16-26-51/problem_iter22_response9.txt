```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['Short_Term_Std'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()

    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['Mid_Term_Std'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()

    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['Long_Term_Std'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    df['Combined_Volatility'] = (0.4 * df['Short_Term_Std']) + (0.3 * df['Mid_Term_Std']) + (0.3 * df['Long_Term_Std'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_SMA_Close'] = df['close'].rolling(window=5).mean()
    df['20_Day_SMA_Close'] = df['close'].rolling(window=20).mean()
    
    df['Price_Trend_Indicator'] = (df['5_Day_SMA_Close'] > df['20_Day_SMA_Close']).astype(int) * 2 - 1  # 1 for Bullish, -1 for Bearish

    df['Adjusted_Combined_Factor'] = df['Combined_Volatility'] * df['Price_Trend_Indicator'].apply(lambda x: 1.2 if x == 1 else 0.8)

    # Final Alpha Factor
    df['High_Low_Price_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    df['Final_Alpha_Factor'] = (df['Close_to_Open_Return'] / df['Combined_Volatility']) * df['Adjusted_Combined_Factor'] * df['High_Low_Price_Ratio']

    return df['Final_Alpha_Factor']
```
