```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Short-Term Lookback (5 days)
    df['sma_5'] = df['vw_return'].rolling(window=5).mean()
    df['std_5'] = df['vw_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['sma_10'] = df['vw_return'].rolling(window=10).mean()
    df['std_10'] = df['vw_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['sma_20'] = df['vw_return'].rolling(window=20).mean()
    df['std_20'] = df['vw_return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatilities
    df['weighted_volatility'] = 0.3 * df['std_5'] + 0.4 * df['std_10'] + 0.3 * df['std_20']
    
    # Incorporate Dynamic Market Trend Adjustments
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['ma_5'] > df['ma_20']).astype(int) * 2 - 1
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vw_return'] / df['weighted_volatility']) * df['price_trend_indicator']
    
    return df['alpha_factor']
```
