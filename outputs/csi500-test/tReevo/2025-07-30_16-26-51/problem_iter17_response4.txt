```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['Short_Term_STD'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['Mid_Term_STD'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['Long_Term_STD'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities with Weights
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['Short_Term_STD'] + 
                                 mid_term_weight * df['Mid_Term_STD'] + 
                                 long_term_weight * df['Long_Term_STD'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['20_Day_MA_Close'] = df['close'].rolling(window=20).mean()
    
    df['Price_Trend_Indicator'] = np.where(df['5_Day_MA_Close'] > df['20_Day_MA_Close'], 1, -1)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Close_to_Open_Return'] / (df['Combined_Volatility'] * df['Price_Trend_Indicator'])
    
    return df['Final_Alpha_Factor']
```
