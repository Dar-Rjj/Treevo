import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_CtoO_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_SMA_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=short_term_lookback).mean()
    df['5_day_STD_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10_day_SMA_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=mid_term_lookback).mean()
    df['10_day_STD_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20_day_SMA_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=long_term_lookback).mean()
    df['20_day_STD_VW_CtoO_Return'] = df['Volume_Weighted_CtoO_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.3
    mid_term_volatility_weight = 0.4
    long_term_volatility_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_volatility_weight * df['5_day_STD_VW_CtoO_Return'] +
                                 mid_term_volatility_weight * df['10_day_STD_VW_CtoO_Return'] +
                                 long_term_volatility_weight * df['20_day_STD_VW_CtoO_Return'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_SMA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20_day_SMA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = df['5_day_SMA_Close'] / df['20_day_SMA_Close']
    
    df['Adjusted_Combined_Factor'] = (df['5_day_SMA_VW_CtoO_Return'] + 
                                      df['10_day_SMA_VW_CtoO_Return'] + 
                                      df['20_day_SMA_VW_CtoO_Return']) * df['Price_Trend_Indicator']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Combined_Factor'] / df['Combined_Volatility']
    
    return df['Alpha_Factor']
