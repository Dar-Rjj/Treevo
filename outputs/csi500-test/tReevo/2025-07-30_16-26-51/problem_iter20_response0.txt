```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['Short_Term_EMA'] = df['Close_to_Open_Return'].ewm(span=short_term, adjust=False).mean()
    df['Short_Term_Std'] = df['Close_to_Open_Return'].rolling(window=short_term).std()
    
    # Mid-Term Lookback
    df['Mid_Term_EMA'] = df['Close_to_Open_Return'].ewm(span=mid_term, adjust=False).mean()
    df['Mid_Term_Std'] = df['Close_to_Open_Return'].rolling(window=mid_term).std()
    
    # Long-Term Lookback
    df['Long_Term_EMA'] = df['Close_to_Open_Return'].ewm(span=long_term, adjust=False).mean()
    df['Long_Term_Std'] = df['Close_to_Open_Return'].rolling(window=long_term).std()
    
    # Calculate Volatility Trend Strength
    df['Short_Term_Vol_Trend'] = df['Short_Term_Std'].pct_change().fillna(0)
    df['Mid_Term_Vol_Trend'] = df['Mid_Term_Std'].pct_change().fillna(0)
    df['Long_Term_Vol_Trend'] = df['Long_Term_Std'].pct_change().fillna(0)
    
    # Assign Weights Based on Trend Strength
    total_trend_strength = df['Short_Term_Vol_Trend'] + df['Mid_Term_Vol_Trend'] + df['Long_Term_Vol_Trend']
    df['Short_Term_Vol_Weight'] = df['Short_Term_Vol_Trend'] / total_trend_strength
    df['Mid_Term_Vol_Weight'] = df['Mid_Term_Vol_Trend'] / total_trend_strength
    df['Long_Term_Vol_Weight'] = df['Long_Term_Vol_Trend'] / total_trend_strength
    
    # Sum of Weighted Volatilities
    df['Combined_Vol'] = (df['Short_Term_Vol_Weight'] * df['Short_Term_Std'] +
                          df['Mid_Term_Vol_Weight'] * df['Mid_Term_Std'] +
                          df['Long_Term_Vol_Weight'] * df['Long_Term_Std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_EMA_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_Day_EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend_Indicator'] = np.sign(df['5_Day_EMA_Close'] - df['20_Day_EMA_Close'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Close_to_Open_Return'] / df['Combined_Vol']) * df['Price_Trend_Indicator']
    
    return df['Alpha_Factor']
```
