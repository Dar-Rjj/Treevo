```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    df['Short_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).mean()
    df['Mid_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).mean()
    df['Long_Term_SMA'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).mean()

    df['Short_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).std()
    df['Mid_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).std()
    df['Long_Term_STD'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.3
    mid_term_vol_weight = 0.4
    long_term_vol_weight = 0.3

    df['Combined_Volatility'] = (short_term_vol_weight * df['Short_Term_STD'] +
                                 mid_term_vol_weight * df['Mid_Term_STD'] +
                                 long_term_vol_weight * df['Long_Term_STD'])

    # Incorporate Dynamic Market Trend Adjustments
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    df['Price_Trend_Indicator'] = np.where(df['EMA_5'] > df['EMA_20'], 1, -1)

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'].rolling(window=long_term_lookback).sum() /
                          df['Combined_Volatility']) * df['Price_Trend_Indicator']

    return df['Alpha_Factor']
```
