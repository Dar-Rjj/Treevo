```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_t'] = df['close']
    df['open_t_1'] = df['open'].shift(1)
    df['volume_weighted_return'] = (df['close_t'] - df['open_t_1']) * df['volume']
    
    # Multi-Period Momentum Indicators
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5_day_ema'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['10_day_ema'] = df['volume_weighted_return'].ewm(span=10, adjust=False).mean()
    
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20_day_ema'] = df['volume_weighted_return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['short_term_vol'] = df['log_returns'].rolling(window=5).std()
    df['long_term_vol'] = df['log_returns'].rolling(window=20).std()
    df['adaptive_vol'] = 0.6 * df['short_term_vol'] + 0.4 * df['long_term_vol']
    
    # Advanced Market Regime Detection
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['10_day_ma'] = df['close'].rolling(window=10).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    def price_trend_indicator(row):
        if row['5_day_ma'] > row['20_day_ma']:
            return 1
        elif row['10_day_ma'] > row['20_day_ma']:
            return 0.5
        else:
            return -1
    
    df['price_trend'] = df.apply(price_trend_indicator, axis=1)
    
    # Combined Momentum Signal
    df['combined_momentum_signal'] = 0.5 * (df['5_day_sma'] + df['5_day_ema']) + \
                                     0.3 * (df['10_day_sma'] + df['10_day_ema']) + \
                                     0.2 * (df['20_day_sma'] + df['20_day_ema'])
    
    # Market Regime Adjustment
    def regime_adjustment_factor(row):
        if row['price_trend'] == 1:
            return 1.2 * row['combined_momentum_signal']
        elif row['price_trend'] == 0.5:
            return 1.0 * row['combined_momentum_signal']
        else:
            return 0.8 * row['combined_momentum_signal']
    
    df['regime_adjusted_factor'] = df.apply(regime_adjustment_factor, axis=1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['regime_adjusted_factor'] / df['adaptive_vol']
    
    return df['final_alpha_factor'].dropna()
```
