```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) / df['open'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume'].shift(-1)
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5d_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['5d_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10d_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['10d_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20d_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['20d_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatility
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3
    
    df['combined_volatility'] = (short_term_vol_weight * df['5d_std'] +
                                 mid_term_vol_weight * df['10d_std'] +
                                 long_term_vol_weight * df['20d_std'])
    
    # Integrate Market Signals
    df['high_low_range'] = df['high'] - df['low']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Final Factor
    df['combined_volume_weighted_return'] = (df['5d_sma'] + df['10d_sma'] + df['20d_sma']) / 3
    df['final_factor'] = df['combined_volume_weighted_return'] / df['combined_volatility']
    
    # Nonlinear Transformation
    df['final_factor'] = np.log(df['final_factor'] + 1)  # Logarithmic Transformation
    df['final_factor'] = np.sqrt(df['final_factor'])  # Square Root Transformation
    
    return df['final_factor']
```
