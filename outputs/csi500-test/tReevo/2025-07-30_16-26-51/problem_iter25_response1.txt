```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['Short_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=short_term_lookback).std()

    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['Mid_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=mid_term_lookback).std()

    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['Long_Term_STD'] = df['Volume_Weighted_Close_to_Open_Return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3

    df['Combined_Volatility'] = (short_term_volatility_weight * df['Short_Term_STD'] +
                                 mid_term_volatility_weight * df['Mid_Term_STD'] +
                                 long_term_volatility_weight * df['Long_Term_STD'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_Day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_Day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()

    df['Price_Trend_Indicator'] = df['5_Day_EMA'] - df['20_Day_EMA']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Short_Term_SMA'] + df['Mid_Term_SMA'] + df['Long_Term_SMA']) / df['Combined_Volatility']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * np.sign(df['Price_Trend_Indicator'])

    return df['Final_Alpha_Factor'].dropna()
```
