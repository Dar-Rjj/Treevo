```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_SMA'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).mean()
    df['5_day_STD'] = df['Close_to_Open_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10_day_SMA'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).mean()
    df['10_day_STD'] = df['Close_to_Open_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20_day_SMA'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).mean()
    df['20_day_STD'] = df['Close_to_Open_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['5_day_STD'] + 
                                 mid_term_weight * df['10_day_STD'] + 
                                 long_term_weight * df['20_day_STD'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_MA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20_day_MA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = np.where(df['5_day_MA_Close'] > df['20_day_MA_Close'], 1, -1)
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['Close_to_Open_Return'] / df['Combined_Volatility']
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Price_Trend_Indicator']
    
    return df['Final_Alpha_Factor']
```
