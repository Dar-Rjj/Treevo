```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate Logarithmic Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 5-Day Realized Volatility
    df['5_day_volatility'] = df['log_return'].rolling(window=5).std()
    
    # Combine Factors
    df['combined_factor'] = df['volume_weighted_return'] / df['5_day_volatility']
    
    # Final Factor
    final_factor = df['combined_factor'].dropna()
    
    return final_factor
```
