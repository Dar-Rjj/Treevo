import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    high_low_momentum = high_low_ratio.rolling(window=5).sum()
    
    # Calculate Volume Anomaly
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_anomaly = df['volume'] - avg_volume_20
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume
    weighted_close_to_open = close_to_open_return * df['volume']
    
    # Calculate 5-Day Realized Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Smooth Volume Changes
    smoothed_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    combined_factor = high_low_momentum + volume_anomaly * smoothed_volume
    
    # Adjust for Volatility
    adjusted_combined_factor = combined_factor / realized_volatility
    
    # Incorporate Price Trend Analysis
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_trend_indicator = ma_5 - ma_20
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_factor * price_trend_indicator
    
    return final_alpha_factor
