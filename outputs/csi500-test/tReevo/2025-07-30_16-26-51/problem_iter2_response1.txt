```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    high_low_momentum = high_low_ratio.rolling(window=lookback_period).sum() / high_low_ratio.rolling(window=lookback_period).mean()
    
    # Dynamic Volume Adjustment
    sum_volume = df['volume'].rolling(window=lookback_period).sum()
    volume_momentum = df['volume'] / sum_volume
    
    # Combine Momentum and Volume Adjustment
    combined_factor = high_low_momentum * volume_momentum
    
    return combined_factor
```
