```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_period=20, ema_amount_period=10):
    # Calculate Simple Moving Average of Close Price (SMA)
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Close Price Momentum
    df['Momentum'] = df['close'] - df['SMA']
    
    # Integrate Volume and Amount Interaction Factor
    # Calculate Logarithm of Volume
    df['Log_Volume'] = np.log(df['volume'])
    
    # Calculate Exponential Moving Average of Amount
    df['EMA_Amount'] = df['amount'].ewm(span=ema_amount_period, adjust=False).mean()
    
    # Combine Close Price Momentum with Volume and Amount Interactions
    df['Weighted_Momentum'] = df['Momentum'] * df['Log_Volume'] * df['EMA_Amount']
    
    # Final Output: The Weighted Price Momentum with Volume and Amount Interaction
    return df['Weighted_Momentum']
```
