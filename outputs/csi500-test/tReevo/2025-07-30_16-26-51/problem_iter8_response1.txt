```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate High-Low Momentum
    df['High_Low_Momentum'] = df['High_Low_Ratio'].rolling(window=5).sum()
    
    # Calculate Volume Anomaly
    df['Volume_Anomaly'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Calculate 5-Day Realized Volatility
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['5D_Realized_Volatility'] = df['Log_Return'].rolling(window=5).std()
    
    # Smooth Volume Changes
    df['Smoothed_Volume_Change'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine High-Low Momentum, Volume Anomaly, and Smoothed Volume
    df['Combined_Factor'] = (df['High_Low_Momentum'] + df['Volume_Anomaly']) * df['Smoothed_Volume_Change']
    
    # Adjust for Volatility
    df['Adjusted_Factor'] = df['Combined_Factor'] / df['5D_Realized_Volatility']
    
    # Incorporate Weighted Close-to-Open Return
    df['Adjusted_Weighted_Factor'] = df['Adjusted_Factor'] + df['Weighted_Close_to_Open_Return']
    
    # Integrate Additional Price Momentum
    df['5D_Price_Momentum'] = df['close'] - df['close'].shift(5)
    df['Final_Alpha_Factor'] = df['Adjusted_Weighted_Factor'] + df['5D_Price_Momentum']
    
    return df['Final_Alpha_Factor']
```
