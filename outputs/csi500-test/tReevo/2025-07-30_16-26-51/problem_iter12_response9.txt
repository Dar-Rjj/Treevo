```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['5d_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5d_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['10d_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['10d_std'] = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['20d_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20d_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Extended Lookback (30 days)
    df['30d_sma'] = df['volume_weighted_return'].rolling(window=30).mean()
    df['30d_std'] = df['volume_weighted_return'].rolling(window=30).std()
    
    # Adaptive Volatility
    df['60d_std'] = df['volume_weighted_return'].rolling(window=60).std()
    
    # Categorize Market as High, Medium, or Low Volatility
    def categorize_volatility(std):
        if std > std.quantile(0.67):
            return 'high'
        elif std < std.quantile(0.33):
            return 'low'
        else:
            return 'medium'
    
    df['volatility_regime'] = df['60d_std'].apply(categorize_volatility)
    
    # Assign Weights to Short-Term, Mid-Term, Long-Term, and Extended Volatilities
    def assign_weights(regime):
        if regime == 'high':
            return [0.4, 0.3, 0.2, 0.1]
        elif regime == 'medium':
            return [0.25, 0.25, 0.25, 0.25]
        else:  # low
            return [0.1, 0.2, 0.3, 0.4]
    
    df['weights'] = df['volatility_regime'].apply(assign_weights)
    
    # Calculate Combined Volatility
    df['combined_volatility'] = (df['5d_std'] * df['weights'].apply(lambda x: x[0]) +
                                 df['10d_std'] * df['weights'].apply(lambda x: x[1]) +
                                 df['20d_std'] * df['weights'].apply(lambda x: x[2]) +
                                 df['30d_std'] * df['weights'].apply(lambda x: x[3]))
    
    # Final Factor
    df['final_factor'] = df['volume_weighted_return'] / df['combined_volatility']
    
    return df['final_factor']
```
