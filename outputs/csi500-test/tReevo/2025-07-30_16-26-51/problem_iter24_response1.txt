```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series indexed by date representing the alpha factor values.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the normalized volume
    normalized_volume = (df['volume'] - df['volume'].rolling(window=10).mean()) / df['volume'].rolling(window=10).std()
    
    # Calculate the volatility
    volatility = df['close'].rolling(window=10).std() / df['close']
    
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Heuristic: Combine the factors
    alpha_factor = (daily_return + normalized_volume - volatility + relative_range + momentum) / 5
    
    return alpha_factor
```
