```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Multi-Period Momentum Indicators
    df['5_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['5_day_EMA'] = df['Volume_Weighted_Return'].ewm(span=5, adjust=False).mean()
    df['10_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=10).mean()
    df['10_day_EMA'] = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    df['20_day_SMA'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['20_day_EMA'] = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()

    # Adaptive Volatility Stabilization
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Short_Term_Volatility'] = df['Log_Return'].rolling(window=5).std()
    df['Long_Term_Volatility'] = df['Log_Return'].rolling(window=20).std()
    df['Adaptive_Volatility'] = 0.6 * df['Short_Term_Volatility'] + 0.4 * df['Long_Term_Volatility']

    # Combine Momentum Signals
    df['Combined_Momentum'] = 0.5 * (df['5_day_SMA'] + df['5_day_EMA']) + \
                              0.3 * (df['10_day_SMA'] + df['10_day_EMA']) + \
                              0.2 * (df['20_day_SMA'] + df['20_day_EMA'])

    # Market Regime Detection
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = np.where(df['5_day_MA'] > df['20_day_MA'], 1, -1)

    # Market Regime Adjustment
    df['Bullish_Factor'] = 1.2 * df['Combined_Momentum']
    df['Bearish_Factor'] = 0.8 * df['Combined_Momentum']
    df['Adjusted_Factor'] = np.where(df['Price_Trend_Indicator'] == 1, df['Bullish_Factor'], df['Bearish_Factor'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['5_day_SMA'] + df['10_day_EMA'] + df['20_day_EMA']) / df['Adaptive_Volatility']
    df['Final_Alpha_Factor'] = np.where(df['Price_Trend_Indicator'] == 1, df['Final_Alpha_Factor'] * 1.2, df['Final_Alpha_Factor'] * 0.8)
    
    return df['Final_Alpha_Factor']
```
