```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Exponential Moving Averages
    df['short_ema'] = df['volume_weighted_return'].ewm(span=short_term, adjust=False).mean()
    df['mid_ema'] = df['volume_weighted_return'].ewm(span=mid_term, adjust=False).mean()
    df['long_ema'] = df['volume_weighted_return'].ewm(span=long_term, adjust=False).mean()

    # Standard Deviations
    df['short_std'] = df['volume_weighted_return'].ewm(span=short_term, adjust=False).std()
    df['mid_std'] = df['volume_weighted_return'].ewm(span=mid_term, adjust=False).std()
    df['long_std'] = df['volume_weighted_return'].ewm(span=long_term, adjust=False).std()

    # Combine Multi-Period Volatilities
    short_vol_weight = 0.4
    mid_vol_weight = 0.3
    long_vol_weight = 0.3
    
    df['combined_volatility'] = (short_vol_weight * df['short_std'] +
                                 mid_vol_weight * df['mid_std'] +
                                 long_vol_weight * df['long_std'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ema_close'] = df['close'].ewm(span=short_term, adjust=False).mean()
    df['20_day_ema_close'] = df['close'].ewm(span=long_term, adjust=False).mean()
    
    df['price_trend_indicator'] = np.where(df['5_day_ema_close'] > df['20_day_ema_close'], 1, -1)

    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_return'] / 
                           df['combined_volatility']) * df['price_trend_indicator']

    return df['alpha_factor']
```
