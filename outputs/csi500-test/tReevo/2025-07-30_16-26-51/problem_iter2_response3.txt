```python
import pandas as pd

def heuristics_v2(df, lookback=10):
    """
    Calculate the Volume Weighted Close-to-Open Return with Price Change Momentum.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    lookback (int): The lookback period for momentum calculation.
    
    Returns:
    pd.Series: A series of alpha factor values.
    """
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Apply Momentum Calculation
    df['price_momentum'] = df['price_change'].rolling(window=lookback).sum()
    
    # Combine Volume Weighted Return and Price Momentum
    df['alpha_factor'] = df['volume_weighted_return'] + df['price_momentum']
    
    return df['alpha_factor']
```
