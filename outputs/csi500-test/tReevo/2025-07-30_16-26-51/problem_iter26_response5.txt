```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_volatility_weight = 0.4
    mid_volatility_weight = 0.3
    long_volatility_weight = 0.3
    
    df['combined_volatility'] = (short_volatility_weight * df['short_std'] +
                                 mid_volatility_weight * df['mid_std'] +
                                 long_volatility_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    df['price_trend_indicator'] = np.where(df['5_day_ma'] > df['20_day_ma'], 1, -1)
    
    # Adjust Combined Factor by Price Trend Indicator
    df['adjusted_combined_volatility'] = np.where(
        df['price_trend_indicator'] == 1,
        (short_volatility_weight * 1.2 + mid_volatility_weight * 0.8 + long_volatility_weight * 0.8) * df['combined_volatility'],
        (short_volatility_weight * 0.8 + mid_volatility_weight * 0.8 + long_volatility_weight * 1.2) * df['combined_volatility']
    )
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['adjusted_combined_volatility']
    
    return df['alpha_factor']
```
