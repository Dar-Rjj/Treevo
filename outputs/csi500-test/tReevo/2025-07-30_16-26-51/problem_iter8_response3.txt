```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) / df['close']) - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['volume'] * df['close_to_open_return']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5_day_sma'] = df['volume_weighted_return'].rolling(window=short_term_lookback).mean()
    df['5_day_std'] = df['volume_weighted_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10_day_sma'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).mean()
    df['10_day_std'] = df['volume_weighted_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20_day_sma'] = df['volume_weighted_return'].rolling(window=long_term_lookback).mean()
    df['20_day_std'] = df['volume_weighted_return'].rolling(window=long_term_lookback).std()
    
    # Stabilize with Multi-Period Volatility
    # Combine Short-Term, Mid-Term, and Long-Term Volatilities
    df['combined_volatility'] = (df['5_day_std'] * 0.3) + (df['10_day_std'] * 0.4) + (df['20_day_std'] * 0.3)
    
    # Final Factor
    df['final_factor'] = (df['volume_weighted_return'].rolling(window=short_term_lookback).mean() +
                          df['volume_weighted_return'].rolling(window=mid_term_lookback).mean() +
                          df['volume_weighted_return'].rolling(window=long_term_lookback).mean()) / df['combined_volatility']
    
    return df['final_factor']
```
