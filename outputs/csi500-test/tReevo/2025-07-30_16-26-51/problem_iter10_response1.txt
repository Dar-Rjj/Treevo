```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Adaptive Volatility
    volatility_regime = df['volume_weighted_return'].rolling(window=20).std().apply(lambda x: 'high' if x > 0.03 else 'low')
    df['combined_volatility'] = (
        volatility_regime.apply(lambda x: 0.6 if x == 'high' else 0.4) * df['short_term_std'] +
        volatility_regime.apply(lambda x: 0.3 if x == 'high' else 0.4) * df['mid_term_std'] +
        volatility_regime.apply(lambda x: 0.1 if x == 'high' else 0.2) * df['long_term_std']
    )
    
    # Additional Price-Based Factors
    df['high_low_ratio'] = df['high'] / df['low']
    df['high_low_momentum'] = df['high_low_ratio'].rolling(window=5).sum() + df['high_low_ratio']
    
    # Incorporate Smoothed Volume
    df['smoothed_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine Multi-Period Volume-Weighted Close-to-Open Return
    df['combined_factor'] = df['short_term_sma'] + df['mid_term_sma'] + df['long_term_sma']
    
    # Adjust for Volatility
    df['adjusted_factor'] = df['combined_factor'] / df['combined_volatility']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_factor'] + df['high_low_momentum'] + df['smoothed_volume']
    
    return df['final_alpha_factor'].dropna()
```
