```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['open']
    df['vol_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20

    # Short-Term Lookback
    df['short_term_sma'] = df['vol_weighted_close_to_open_return'].rolling(window=short_term_lookback).mean()
    df['short_term_std'] = df['vol_weighted_close_to_open_return'].rolling(window=short_term_lookback).std()

    # Mid-Term Lookback
    df['mid_term_sma'] = df['vol_weighted_close_to_open_return'].rolling(window=mid_term_lookback).mean()
    df['mid_term_std'] = df['vol_weighted_close_to_open_return'].rolling(window=mid_term_lookback).std()

    # Long-Term Lookback
    df['long_term_sma'] = df['vol_weighted_close_to_open_return'].rolling(window=long_term_lookback).mean()
    df['long_term_std'] = df['vol_weighted_close_to_open_return'].rolling(window=long_term_lookback).std()

    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3

    df['combined_volatility'] = (short_term_volatility_weight * df['short_term_std'] +
                                 mid_term_volatility_weight * df['mid_term_std'] +
                                 long_term_volatility_weight * df['long_term_std'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_sma_close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20_day_sma_close'] = df['close'].rolling(window=long_term_lookback).mean()

    # Price Trend Indicator
    df['price_trend_indicator'] = df['5_day_sma_close'] - df['20_day_sma_close']

    # Momentum Indicator
    df['momentum_indicator'] = df['5_day_sma_close'] / df['20_day_sma_close']

    # Adjust Combined Factor by Price Trend and Momentum Indicators
    df['adjusted_combined_volatility'] = df['combined_volatility'] * df['price_trend_indicator'] * df['momentum_indicator']

    # Final Alpha Factor
    df['alpha_factor'] = df['vol_weighted_close_to_open_return'] / df['adjusted_combined_volatility']

    return df['alpha_factor']
```
