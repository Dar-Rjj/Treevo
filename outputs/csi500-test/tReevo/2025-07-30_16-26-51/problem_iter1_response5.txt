```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-Day Simple Moving Average (SMA) of Daily Returns
    df['sma_20'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate 20-Day Exponential Moving Average (EMA) of Daily Returns
    df['ema_20'] = df['daily_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Daily Logarithmic Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 5-Day Realized Volatility
    df['realized_volatility_5'] = df['log_return'].rolling(window=5).std()
    
    # Calculate Volume Weighted Return
    df['volume_weighted_return'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    
    # Calculate 5-Day Moving Sum of Volume
    df['sum_volume_5'] = df['volume'].rolling(window=5).sum()
    
    # Combine Factors
    df['combined_factor'] = (df['sma_20'] * df['ema_20']) / df['realized_volatility_5']
    
    # Final Factor
    df['final_factor'] = (df['combined_factor'] * df['volume_weighted_return']) / df['sum_volume_5']
    
    return df['final_factor'].dropna()
```
