```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['vol_weighted_c_to_o_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1) * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['short_ma'] = df['vol_weighted_c_to_o_return'].rolling(window=short_term_lookback).mean()
    df['short_std'] = df['vol_weighted_c_to_o_return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['mid_ma'] = df['vol_weighted_c_to_o_return'].rolling(window=mid_term_lookback).mean()
    df['mid_std'] = df['vol_weighted_c_to_o_return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['long_ma'] = df['vol_weighted_c_to_o_return'].rolling(window=long_term_lookback).mean()
    df['long_std'] = df['vol_weighted_c_to_o_return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3
    
    df['combined_volatility'] = (short_term_vol_weight * df['short_std'] +
                                 mid_term_vol_weight * df['mid_std'] +
                                 long_term_vol_weight * df['long_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5d_ma_close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20d_ma_close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['price_trend_indicator'] = (df['5d_ma_close'] > df['20d_ma_close']).astype(int) * 2 - 1
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vol_weighted_c_to_o_return'] /
                          df['combined_volatility']) * df['price_trend_indicator']
    
    return df['alpha_factor']
```
