import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Volume-Weighted Close-to-Open Return
    data['volume_weighted_return'] = (data['close'] - data['open'].shift(1)) * data['volume']
    
    # Multi-Period Momentum Indicators
    for period in [5, 10, 20]:
        data[f'{period}_day_SMA_vol_weighted_return'] = data['volume_weighted_return'].rolling(window=period).mean()
        data[f'{period}_day_EMA_vol_weighted_return'] = data['volume_weighted_return'].ewm(span=period, adjust=False).mean()

    # Adaptive Volatility Stabilization
    data['log_returns'] = np.log(data['close']) - np.log(data['close'].shift(1))
    data['short_term_volatility'] = data['log_returns'].rolling(window=5).std()
    data['long_term_volatility'] = data['log_returns'].rolling(window=20).std()
    data['adaptive_volatility'] = 0.6 * data['short_term_volatility'] + 0.4 * data['long_term_volatility']

    # Market Regime Detection
    data['5_day_MA_close'] = data['close'].rolling(window=5).mean()
    data['20_day_MA_close'] = data['close'].rolling(window=20).mean()
    data['price_trend_indicator'] = np.where(data['5_day_MA_close'] > data['20_day_MA_close'], 1, -1)

    # Combine Momentum Signals
    data['combined_momentum_signal'] = (
        0.5 * (data['5_day_SMA_vol_weighted_return'] + data['5_day_EMA_vol_weighted_return']) +
        0.3 * (data['10_day_SMA_vol_weighted_return'] + data['10_day_EMA_vol_weighted_return']) +
        0.2 * (data['20_day_SMA_vol_weighted_return'] + data['20_day_EMA_vol_weighted_return'])
    )

    # Market Regime Adjustment
    data['bullish_factor'] = 1.2 * data['combined_momentum_signal']
    data['bearish_factor'] = 0.8 * data['combined_momentum_signal']
    data['regime_adjusted_factor'] = np.where(
        data['price_trend_indicator'] == 1,
        data['bullish_factor'],
        data['bearish_factor']
    )

    # Final Alpha Factor
    data['alpha_factor'] = data['regime_adjusted_factor'] / data['adaptive_volatility']

    return data['alpha_factor'].dropna()
