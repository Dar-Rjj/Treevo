```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Short-Term Lookback (5 days)
    df['5D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=5).mean()
    df['5D_EMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=5, adjust=False).mean()

    # Mid-Term Lookback (10 days)
    df['10D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=10).mean()
    df['10D_EMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()

    # Long-Term Lookback (20 days)
    df['20D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    df['20D_EMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()

    # Combine Multi-Period Volatilities
    df['5D_Volatility'] = df['Volume_Weighted_Return'].rolling(window=5).std()
    df['10D_Volatility'] = df['Volume_Weighted_Return'].rolling(window=10).std()
    df['20D_Volatility'] = df['Volume_Weighted_Return'].rolling(window=20).std()

    # Assign Weights to Each Volatility
    short_term_vol_weight = 0.4
    mid_term_vol_weight = 0.3
    long_term_vol_weight = 0.3

    df['Combined_Volatility'] = (short_term_vol_weight * df['5D_Volatility'] +
                                 mid_term_vol_weight * df['10D_Volatility'] +
                                 long_term_vol_weight * df['20D_Volatility'])

    # Incorporate Dynamic Market Trend Adjustments
    df['5D_SMA_Close'] = df['close'].rolling(window=5).mean()
    df['20D_SMA_Close'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = df['5D_SMA_Close'] - df['20D_SMA_Close']

    # Final Alpha Factor
    df['Combined_Volume_Weighted_Return'] = (df['5D_SMA_Volume_Weighted_Return'] + 
                                              df['10D_SMA_Volume_Weighted_Return'] + 
                                              df['20D_SMA_Volume_Weighted_Return'])

    df['Final_Alpha_Factor'] = (df['Combined_Volume_Weighted_Return'] / df['Combined_Volatility']) * df['Price_Trend_Indicator']

    return df['Final_Alpha_Factor']
```
