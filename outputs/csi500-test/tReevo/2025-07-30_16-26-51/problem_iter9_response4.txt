```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['short_ma'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['short_std'] = df['volume_weighted_return'].rolling(window=5).std()
    
    # Mid-Term Lookback (10 days)
    df['mid_ma'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['mid_std'] = df['volume_weighted_return'].rolling(window=10).std()
    
    # Long-Term Lookback (20 days)
    df['long_ma'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['long_std'] = df['volume_weighted_return'].rolling(window=20).std()
    
    # Combine Multi-Period Volatility
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['combined_volatility'] = (short_term_weight * df['short_std'] +
                                 mid_term_weight * df['mid_std'] +
                                 long_term_weight * df['long_std'])
    
    # Final Factor
    df['final_factor'] = (df['short_ma'] + df['mid_ma'] + df['long_ma']) / df['combined_volatility']
    
    return df['final_factor']
```
