```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Short-Term Lookback (5 days)
    short_term_window = 5
    short_term_sma = df['volume_weighted_return'].rolling(window=short_term_window).mean()
    short_term_std = df['volume_weighted_return'].rolling(window=short_term_window).std()

    # Long-Term Lookback (20 days)
    long_term_window = 20
    long_term_sma = df['volume_weighted_return'].rolling(window=long_term_window).mean()
    long_term_std = df['volume_weighted_return'].rolling(window=long_term_window).std()

    # Stabilize with Volatility
    df['short_term_normalized'] = short_term_sma / short_term_std
    df['long_term_normalized'] = long_term_sma / long_term_std

    # Final factor: Combine both normalized terms
    df['factor'] = df['short_term_normalized'] + df['long_term_normalized']

    return df['factor']
```
