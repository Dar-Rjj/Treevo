import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume-Weight the Return
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Short-Term Lookback (5 days)
    df['5d_sma_vw_return'] = df['volume_weighted_return'].rolling(window=5).mean()
    df['5d_ema_vw_return'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    # Mid-Term Lookback (10 days)
    df['10d_sma_vw_return'] = df['volume_weighted_return'].rolling(window=10).mean()
    df['10d_ema_vw_return'] = df['volume_weighted_return'].ewm(span=10, adjust=False).mean()
    
    # Long-Term Lookback (20 days)
    df['20d_sma_vw_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    df['20d_ema_vw_return'] = df['volume_weighted_return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    df['short_term_vol'] = df['volume_weighted_return'].rolling(window=5).std()
    df['long_term_vol'] = df['volume_weighted_return'].rolling(window=20).std()
    df['adaptive_vol'] = 0.6 * df['short_term_vol'] + 0.4 * df['long_term_vol']
    
    # Integrated Momentum and Adaptive Volatility
    df['integrated_momentum'] = (df['5d_sma_vw_return'] + df['10d_ema_vw_return'] + df['20d_ema_vw_return']) / 3
    df['stabilized_momentum'] = df['integrated_momentum'] / df['adaptive_vol']
    
    # Incorporate High-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Integrate High-Low Momentum
    df['high_low_momentum'] = df['high_low_ratio'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['stabilized_momentum'] + df['high_low_momentum'] + df['volume_weighted_return']
    
    return df['alpha_factor'].dropna()
