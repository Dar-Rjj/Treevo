import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['5D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).mean()
    df['5D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['10D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).mean()
    df['10D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['20D_SMA_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).mean()
    df['20D_STD_Volume_Weighted_Return'] = df['Volume_Weighted_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_volatility_weight = 0.4
    mid_term_volatility_weight = 0.3
    long_term_volatility_weight = 0.3
    
    df['Combined_Volatility'] = (
        short_term_volatility_weight * df['5D_STD_Volume_Weighted_Return'] +
        mid_term_volatility_weight * df['10D_STD_Volume_Weighted_Return'] +
        long_term_volatility_weight * df['20D_STD_Volume_Weighted_Return']
    )
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5D_SMA_Close'] = df['close'].rolling(window=short_term_lookback).mean()
    df['20D_SMA_Close'] = df['close'].rolling(window=long_term_lookback).mean()
    
    df['Price_Trend_Indicator'] = df['5D_SMA_Close'] - df['20D_SMA_Close']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'] / df['Combined_Volatility']) * df['Price_Trend_Indicator']
    
    return df['Alpha_Factor'].dropna()
