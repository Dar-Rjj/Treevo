```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Adaptive Lookback Periods
    short_term = 5
    mid_term = 10
    long_term = 20
    
    # Short-Term Lookback
    df['short_term_sma'] = df['volume_weighted_return'].rolling(window=short_term).mean()
    df['short_term_std'] = df['volume_weighted_return'].rolling(window=short_term).std()
    
    # Mid-Term Lookback
    df['mid_term_sma'] = df['volume_weighted_return'].rolling(window=mid_term).mean()
    df['mid_term_std'] = df['volume_weighted_return'].rolling(window=mid_term).std()
    
    # Long-Term Lookback
    df['long_term_sma'] = df['volume_weighted_return'].rolling(window=long_term).mean()
    df['long_term_std'] = df['volume_weighted_return'].rolling(window=long_term).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['combined_volatility'] = (short_term_weight * df['short_term_std'] +
                                 mid_term_weight * df['mid_term_std'] +
                                 long_term_weight * df['long_term_std'])
    
    # Incorporate Dynamic Market Trend Adjustments
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    df['price_trend_indicator'] = np.where(df['5_day_ma_close'] > df['20_day_ma_close'], 1, -1)
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_return'] / df['combined_volatility']) * df['price_trend_indicator']
    
    return df['alpha_factor']
```
