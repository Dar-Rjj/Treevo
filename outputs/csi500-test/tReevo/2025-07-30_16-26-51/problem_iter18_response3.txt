```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Return'] = (df['close'] - df['open'].shift(1)) * df['volume']
    
    # Multi-Period Momentum Indicators
    short_term_sma = df['Volume_Weighted_Return'].rolling(window=5).mean()
    short_term_ema = df['Volume_Weighted_Return'].ewm(span=5, adjust=False).mean()
    mid_term_sma = df['Volume_Weighted_Return'].rolling(window=10).mean()
    mid_term_ema = df['Volume_Weighted_Return'].ewm(span=10, adjust=False).mean()
    long_term_sma = df['Volume_Weighted_Return'].rolling(window=20).mean()
    long_term_ema = df['Volume_Weighted_Return'].ewm(span=20, adjust=False).mean()
    
    # Adaptive Volatility Stabilization
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_vol = log_returns.rolling(window=5).std()
    mid_vol = log_returns.rolling(window=10).std()
    long_vol = log_returns.rolling(window=20).std()
    adaptive_volatility = 0.4 * short_vol + 0.3 * mid_vol + 0.3 * long_vol
    
    # Combine Momentum Signals
    combined_momentum = 0.5 * (short_term_sma + short_term_ema) + \
                        0.3 * (mid_term_sma + mid_term_ema) + \
                        0.2 * (long_term_sma + long_term_ema)
    
    # Market Regime Detection
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_trend = np.where(ma_5 > ma_20, 1, -1)
    
    # Market Regime Adjustment
    bullish_factor = 1.2 * combined_momentum
    bearish_factor = 0.8 * combined_momentum
    regime_adjusted_factor = np.where(price_trend == 1, bullish_factor, bearish_factor)
    
    # Integrate High-Low Price Ratio
    high_low_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = (short_term_sma + mid_term_ema + long_term_ema) * \
                         regime_adjusted_factor / adaptive_volatility * high_low_ratio
    
    return final_alpha_factor
```
