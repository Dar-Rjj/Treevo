```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    df['Close-to-Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['Volume-Weighted_Return'] = df['Close-to-Open_Return'] * df['volume']
    
    # Dynamic Lookback Periods
    short_term_lookback = 5
    mid_term_lookback = 10
    long_term_lookback = 20
    
    # Short-Term Lookback
    df['Short_Term_SMA'] = df['Volume-Weighted_Return'].rolling(window=short_term_lookback).mean()
    df['Short_Term_STD'] = df['Volume-Weighted_Return'].rolling(window=short_term_lookback).std()
    
    # Mid-Term Lookback
    df['Mid_Term_SMA'] = df['Volume-Weighted_Return'].rolling(window=mid_term_lookback).mean()
    df['Mid_Term_STD'] = df['Volume-Weighted_Return'].rolling(window=mid_term_lookback).std()
    
    # Long-Term Lookback
    df['Long_Term_SMA'] = df['Volume-Weighted_Return'].rolling(window=long_term_lookback).mean()
    df['Long_Term_STD'] = df['Volume-Weighted_Return'].rolling(window=long_term_lookback).std()
    
    # Combine Multi-Period Volatilities
    short_term_weight = 0.4
    mid_term_weight = 0.3
    long_term_weight = 0.3
    
    df['Combined_Volatility'] = (short_term_weight * df['Short_Term_STD']) + \
                                (mid_term_weight * df['Mid_Term_STD']) + \
                                (long_term_weight * df['Long_Term_STD'])
    
    # Final Factor
    df['Final_Factor'] = (df['Short_Term_SMA'] + df['Mid_Term_SMA'] + df['Long_Term_SMA']) / 3
    df['Final_Factor'] = df['Final_Factor'] / df['Combined_Volatility']
    
    return df['Final_Factor']
```
