import pandas as pd

def heuristics_v2(df, N=10):
    """
    Calculate the High-Low Range Momentum Indicator.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    - N: Number of periods to calculate the momentum over.
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate N-period rolling window of High-Low Range
    df['range_momentum'] = df['high_low_range'].rolling(window=N).apply(
        lambda x: x[-1] - x[0], raw=True
    )
    
    return df['range_momentum']
