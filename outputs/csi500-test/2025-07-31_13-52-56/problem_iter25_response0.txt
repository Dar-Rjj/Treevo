```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Evaluate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Intraday Return with Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility'].replace(0, np.nan)
    
    # Compute Enhanced High-to-Low Range
    df['enhanced_high_low_range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['adjusted_high_low_range'] = df['enhanced_high_low_range'] * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    df['volume_10_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = np.where(df['volume'] > df['volume_10_ma'], 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    df['integrated_adjusted_range'] = df['adjusted_high_low_range'] * df['volume_trend']
    
    # Calculate Momentum Signal
    df['momentum_signal'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Trend
    df['combined_factors'] = df['integrated_adjusted_range'] - df['momentum_signal']
    
    # Identify Volume Shock
    df['volume_30_ma'] = df['volume'].rolling(window=30).mean()
    df['volume_shock'] = np.where(df['volume'] > 1.5 * df['volume_30_ma'], 1, 0)
    
    # Integrate Momentum and Volume Shock
    df['integrated_momentum_volume_shock'] = df['combined_factors'] * df['volume_shock']
    
    # Calculate 20-day Price Momentum
    df['price_momentum_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Combine High-Low Range and Close Price Momentum
    df['avg_high_low_momentum'] = (df['enhanced_high_low_range'] + df['price_momentum_20'].rolling(window=30).mean()) / 2
    
    # Final Factor Combination
    df['final_factor'] = (
        df['adjusted_intraday_return'] * df['integrated_momentum_volume_shock'] +
        df['price_momentum_20'] +
        (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10) +
        (df['high'] - df['low']).rolling(window=20).mean()
    )
    
    return df['final_factor']
```
