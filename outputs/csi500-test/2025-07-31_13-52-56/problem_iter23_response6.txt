```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volatility_adjusted_intraday_return'] = df['intraday_return'] / (df['high'] - df['low'])
    
    # Compute 30-day Price Momentum
    df['30_day_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    avg_amount_30 = df['amount'].rolling(window=30).mean()
    df['volume_shock'] = (df['volume'] > 2 * avg_volume_30).astype(int)
    df['amount_shock'] = (df['amount'] > 2 * avg_amount_30).astype(int)
    
    # Enhanced Volume and Price Volatility Components
    df['high_low_range_vol_adj'] = (df['high'] - df['low']) * df['volume']
    df['cum_high_low_range_vol_adj'] = df['high_low_range_vol_adj'].rolling(window=30).sum()
    
    df['open_close_change'] = df['close'] - df['open']
    df['cum_open_close_change'] = df['open_close_change'].rolling(window=30).sum()
    
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_moving_avg']).astype(int) * 2 - 1
    df['volume_momentum'] = df['volume'].rolling(window=30).sum() - df['volume'].shift(30).rolling(window=30).sum()
    
    # Composite Alpha Factors Synthesis
    df['adjusted_volume_momentum'] = df['volume_momentum'] * (df['high'] / df['low'])
    df['inverted_volume_momentum'] = -1 * df['volume_momentum']
    df['composite_alpha'] = (df['inverted_volume_momentum'] + 
                             df['adjusted_volume_momentum'] + 
                             (df['high'] - df['low']) * df['volume'] * df['volume_momentum'])
    df['volume_trend_high_low_ratio'] = (df['high'] / df['low']) * df['volume_trend']
    
    # Weighted and Integrated Alpha Factor
    df['weighted_intraday_return'] = 0.2 * df['volatility_adjusted_intraday_return']
    df['weighted_30_day_price_momentum'] = 0.3 * df['30_day_price_momentum']
    df['weighted_volume_amount_shocks'] = 0.1 * df['volume_shock'] * df['amount_shock']
    df['weighted_adjusted_volume_momentum'] = 0.2 * df['adjusted_volume_momentum']
    df['weighted_inverted_volume_momentum'] = 0.15 * df['inverted_volume_momentum']
    df['weighted_composite_alpha'] = 0.15 * df['composite_alpha']
    
    df['final_alpha'] = (df['weighted_intraday_return'] + 
                         df['weighted_30_day_price_momentum'] + 
                         df['weighted_volume_amount_shocks'] + 
                         df['weighted_adjusted_volume_momentum'] + 
                         df['weighted_inverted_volume_momentum'] + 
                         df['weighted_composite_alpha'])
    
    # Incorporate Volume Trend
    df['30_day_volume_momentum'] = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    df['final_alpha_with_volume_trend'] = df['final_alpha'] * df['30_day_volume_momentum']
    
    # Refinement through Contextual Analysis
    df['trend_direction_multiplier'] = ((df['close'] > df['open'].shift(1)).astype(int) * 2 - 1).fillna(1)
    df['final_alpha_refined'] = df['final_alpha_with_volume_trend'] * df['trend_direction_multiplier']
    
    return df['final_alpha_refined']
```
