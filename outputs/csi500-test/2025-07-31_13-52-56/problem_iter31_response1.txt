```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Evaluate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Return with Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Compute Enhanced High-to-Low Range
    enhanced_high_to_low_range = (df['high'] - df['low']) * (df['high'] + df['low']) / 2
    
    # Adjust for Volume
    adjusted_high_to_low_range = enhanced_high_to_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean()
    volume_trend_signal = np.where(df['volume'] > volume_trend, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_volume_trend = adjusted_high_to_low_range * volume_trend_signal
    
    # Calculate Momentum Signal
    momentum_signal = df['close'].ewm(span=10, adjust=False).mean()
    
    # Identify Volume Shock
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_shock = np.where(df['volume'] > 1.5 * avg_volume_30, 1, 0)
    
    # Integrate Momentum and Volume Shock
    integrated_momentum_volume_shock = momentum_signal * volume_shock
    
    # Combine High-Low Range and Momentum with Volume Trends
    combined_factor = integrated_volume_trend - integrated_momentum_volume_shock
    
    # Calculate 20-day Price Momentum
    price_momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Final Factor Combination
    final_factor = combined_factor + price_momentum_20
    
    return final_factor
```
