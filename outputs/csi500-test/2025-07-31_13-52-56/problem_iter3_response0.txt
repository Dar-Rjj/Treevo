```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_to_low_range = high_to_low_range * df['volume']
    
    # Detect Volume Spike
    volume_5_day_moving_average = df['volume'].rolling(window=5).mean()
    volume_spike_flag = (df['volume'] > 1.5 * volume_5_day_moving_average).astype(int)
    
    # Calculate Momentum Signal
    momentum_signal = df['close'].shift(0) - df['close'].shift(10)
    
    # Combine Adjusted Range, Momentum, and Volume Spike
    factor = adjusted_high_to_low_range + momentum_signal
    factor = factor * (volume_spike_flag * 2 + 1 - volume_spike_flag)
    
    return factor
```
