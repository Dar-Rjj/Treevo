import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum over N periods
    n_periods = 30
    df['high_low_range_momentum'] = df['high_low_range'].rolling(window=n_periods).apply(
        lambda x: (x[-1] - x[0]) / n_periods, raw=True)
    
    # Calculate 30-day Close Price Momentum
    df['close_momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Combine High-Low Range and Close Price Momentum
    df['integrated_momentum'] = (df['high_low_range_momentum'] + df['close_momentum_30']) / 2
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    
    # Integrate Momentum and Volume Shock
    df['integrated_momentum_volume_shock'] = df['integrated_momentum'] * df['volume_shock']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df[['high', 'low', 'close', 'open']].max(axis=1) - 
                                 df[['high', 'low', 'close', 'open']].min(axis=1))
    
    # Adjust Intraday Return by Intraday Volatility
    df['intraday_volatility_adjusted'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate Volume Displacement
    df['volume_displacement'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['intraday_volatility_adjusted'] * df['volume_displacement']
    
    # Integrate Intraday and Historical Factors
    df['final_factor'] = df['integrated_momentum_volume_shock'] * df['intraday_factors']
    
    # Add the result to Price Momentum
    look_back_period = 50
    df['price_momentum'] = (df['close'] - df['close'].shift(look_back_period)) / df['close'].shift(look_back_period)
    
    # Add the result to Volume Acceleration
    volume_look_back_period = 20
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(volume_look_back_period)) / df['volume'].shift(volume_look_back_period)
    
    # Add the result to Trading Range Expansion
    trading_range_look_back_period = 30
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], abs(row['high'] - row['close'].shift(1)), abs(row['low'] - row['close'].shift(1))), axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=trading_range_look_back_period).mean()
    
    # Use the Sum as the Final Factor
    df['final_factor'] += df['price_momentum'] + df['volume_acceleration'] + df['average_true_range']
    
    return df['final_factor'].dropna()
