```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_moving_avg).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    
    # Measure Volume Momentum Indicator
    volume_aggregated = df['volume'].rolling(window=10).sum()
    volume_momentum = volume_aggregated.diff().fillna(0)
    
    # Quantify High-Low Range and Volatility
    high_low_range_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Average True Range
    tr = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    atr = tr.rolling(window=10).mean()
    
    # Integrate Components for Alpha Factor
    high_to_low_adjusted = high_to_low_ratio * volume_trend
    range_expansion_factor = high_low_range_ratio * volume_momentum
    alpha_factor = high_to_low_adjusted + range_expansion_factor / atr
    
    return alpha_factor
```
