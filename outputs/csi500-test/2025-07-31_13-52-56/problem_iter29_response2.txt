import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_daily_range = daily_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_10_day_ma, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_factor = adjusted_daily_range * volume_trend
    
    # Analyze Close Price Position within Daily Range
    close_position = (df['close'] - df['low']) / daily_range
    avg_close_position = close_position.rolling(window=20).mean()
    close_position_factor = close_position - avg_close_position
    
    # Investigate Open-Close Return
    open_close_return = (df['close'] - df['open']) / df['open']
    avg_open_close_return = open_close_return.rolling(window=20).mean()
    open_close_return_factor = open_close_return - avg_open_close_return
    
    # Analyze Amount Traded
    amount_traded = df['volume'] * df['close']
    avg_amount_traded = amount_traded.rolling(window=20).mean()
    amount_traded_factor = amount_traded / avg_amount_traded
    
    # Calculate 30-day Price Momentum
    price_momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    avg_volume_30 = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    volume_shock = np.where(df['volume'] > 2 * avg_volume_30, 1, 0)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    momentum_with_shock = price_momentum_30 * volume_shock
    
    # Detect Volume Spike
    volume_5_day_ma = df['volume'].rolling(window=5).mean()
    volume_spike = np.where(df['volume'] > 2.0 * volume_5_day_ma, 1, 0)
    
    # Calculate 10-day Momentum
    price_momentum_10 = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted Daily Range, 10-day Momentum, and Volume Spike
    combined_factor = (integrated_factor + price_momentum_10) * (volume_spike * 3.0 + 1.0 - volume_spike)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = momentum_with_shock + combined_factor
    
    return final_alpha_factor
