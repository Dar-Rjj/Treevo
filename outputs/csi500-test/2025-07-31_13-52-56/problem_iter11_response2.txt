```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df[['high', 'low', 'close', 'open']].std(axis=1)

    # Adjust Intraday Return
    df['intraday_volatility_adjusted'] = df['intraday_return'] / df['intraday_volatility']

    # Calculate 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30, min_periods=30).mean()).astype(int)

    # Identify Amount Shock
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30, min_periods=30).mean()).astype(int)

    # Multiply 30-day Momentum by Volume Shock indicator
    df['30_day_momentum_volume_shock'] = df['30_day_momentum'] * df['volume_shock']

    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']

    # Adjust for Volume
    df['adjusted_high_to_low_range'] = df['high_to_low_range'] * df['volume']

    # Detect Volume Spike
    df['5_day_vol_ma'] = df['volume'].rolling(window=5, min_periods=5).mean()
    df['volume_spike'] = (df['volume'] > 1.7 * df['5_day_vol_ma']).astype(int)

    # Calculate 10-day Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)

    # Combine Adjusted Range, 10-day Momentum, and Volume Spike
    df['combined_adjusted_range'] = (df['adjusted_high_to_low_range'] + df['10_day_momentum']) * (df['volume_spike'] * 2.5 + 1)

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['intraday_volatility_adjusted'] +
        df['30_day_momentum_volume_shock'] +
        df['combined_adjusted_range']
    )

    # Drop intermediate columns if not needed
    df.drop(columns=[
        'intraday_return', 'intraday_volatility', 'intraday_volatility_adjusted',
        '30_day_momentum', 'volume_shock', 'amount_shock',
        '30_day_momentum_volume_shock', 'high_to_low_range', 'adjusted_high_to_low_range',
        '5_day_vol_ma', 'volume_spike', '10_day_momentum', 'combined_adjusted_range'
    ], inplace=True)

    return df['final_alpha_factor']
```
