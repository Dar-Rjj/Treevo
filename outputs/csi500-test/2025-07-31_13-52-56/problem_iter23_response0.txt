import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_10day_ma, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_range = adjusted_range * volume_trend
    
    # Calculate 30-day Price Momentum
    momentum_30day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    volume_30day_ma = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    volume_shock = np.where(df['volume'] > 2 * volume_30day_ma, 1, 0)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    momentum_with_shock = momentum_30day * volume_shock
    
    # Detect Volume Spike
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_spike = np.where(df['volume'] > 2.0 * volume_5day_ma, 1, 0)
    
    # Calculate 10-day Momentum
    momentum_10day = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted High-to-Low Range, 10-day Momentum, and Volume Spike
    combined_signal = integrated_range + momentum_10day
    combined_signal = combined_signal * (volume_spike * 3.0 + 1)
    
    # Synthesize Final Alpha Factor
    alpha_factor = momentum_with_shock + combined_signal
    
    return alpha_factor
