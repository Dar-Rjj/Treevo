```python
import pandas as pd

def heuristics_v2(df):
    # 20-Day Price Momentum
    df['Ct_20'] = df['close'].shift(20)
    df['20_day_momentum'] = (df['close'] - df['Ct_20']) / df['Ct_20']
    
    # Volume Surge Indicator
    df['Med_Vt_20'] = df['volume'].rolling(window=20, min_periods=1).median().shift(1)  # Exclude day t
    df['volume_surge'] = (df['volume'] > 2 * df['Med_Vt_20']).astype(int)
    
    # High-Low Range Volatility
    df['range'] = df['high'] - df['low']
    df['high_low_volatility'] = df['range'].rolling(window=10, min_periods=1).std()
    
    # Combine Factors
    df['alpha_factor'] = df['20_day_momentum'] + df['volume_surge'] + df['high_low_volatility']
    
    return df['alpha_factor']
```
