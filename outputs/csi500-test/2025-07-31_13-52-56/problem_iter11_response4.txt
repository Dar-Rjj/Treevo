```python
import pandas as pd

def heuristics_v2(df, N=10, n=30):
    # Calculate High-Low Range and Its Volume-Adjusted Version
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * df['volume']

    # Evaluate Cumulative and Dynamic High-Low Measures
    df['Cumulative_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=N).sum()
    df['High_to_Low_Price_Ratio'] = df['high'] / df['low']

    # Analyze Volume Trends and Momentum
    df['Volume_Moving_Average'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_Moving_Average']).astype(int) * 2 - 1
    df['Aggregated_Volume'] = df['volume'].rolling(window=n).sum()
    df['Volume_Momentum'] = df['Aggregated_Volume'] - df['Aggregated_Volume'].shift(n)

    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['Cumulative_Open_Close_Price_Change'] = df['Open_Close_Price_Change'].rolling(window=N).sum()
    df['Average_True_Range'] = df['High_Low_Range'].rolling(window=n).mean()

    # Integrate 30-Day Price Momentum
    df['Ct'] = df['close']
    df['Ct_30'] = df['close'].shift(30)
    df['30_Day_Price_Momentum'] = (df['Ct'] - df['Ct_30']) / df['Ct_30']

    # Volume Shock
    df['Vt'] = df['volume']
    df['Avg_Vt_30'] = df['volume'].rolling(window=30).mean().shift(1)
    df['Volume_Shock'] = (df['Vt'] > 2 * df['Avg_Vt_30']).astype(int)

    # High-Low Range Momentum
    df['High_Low_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(N)

    # Synthesize Components into Alpha Factor
    alpha_factor = (
        df['High_to_Low_Price_Ratio'] * df['Volume_Trend'] +
        df['Volume_Adjusted_High_Low_Range'] * df['Volume_Momentum'] +
        df['High_Low_Range'] * (df['Aggregated_Volume'].shift(1) - df['Aggregated_Volume'].shift(n)) +
        df['30_Day_Price_Momentum'] * df['Volume_Shock'] +
        df['High_Low_Range_Momentum']
    )

    return alpha_factor
```
