import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Adjusted Volume Momentum
    df['V_sum'] = df['volume'].rolling(window=n).sum()
    df['V_change'] = df['V_sum'] - df['V_sum'].shift(n)
    
    # Measure and Adjust High-Low Price Ratio
    df['H/L'] = df['high'] / df['low']
    df['Volume_MA'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    df['Adjusted_HL'] = df['H/L'] * df['Volume_Trend']
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=n).std()
    
    # Weighted Combined Alpha
    df['Alpha'] = (0.5 * df['V_change'] + 0.5 * df['Price_Volatility'])
    
    return df['Alpha']
