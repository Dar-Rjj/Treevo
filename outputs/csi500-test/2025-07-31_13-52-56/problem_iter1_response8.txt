```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Cumulative High-Low Range over N days
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=N).sum()
    
    # Calculate Open-to-Close Price Change
    open_close_change = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days
    cumulative_open_close_change = open_close_change.rolling(window=N).sum()
    
    # Final Alpha Factor: High-Low Range Momentum Indicator
    alpha_factor = cumulative_high_low_range / cumulative_open_close_change
    
    return alpha_factor
```
