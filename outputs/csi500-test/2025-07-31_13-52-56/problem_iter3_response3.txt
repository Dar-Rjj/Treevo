```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    def price_momentum(close, lookback):
        return close - close.shift(lookback)
    
    # Calculate Volume Acceleration
    def volume_acceleration(volume, lookback):
        return volume - volume.shift(lookback)
    
    # Calculate True Range
    def true_range(high, low, close_prev):
        tr1 = high - low
        tr2 = (high - close_prev).abs()
        tr3 = (low - close_prev).abs()
        return pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range
    def average_true_range(high, low, close, lookback):
        tr = true_range(high, low, close.shift(1))
        atr = tr.rolling(window=lookback).mean()
        return atr
    
    # Determine Look-Back Periods
    momentum_lookback = 50
    volume_lookback = 20
    atr_lookback = 20
    
    # Calculate the components
    momentum = price_momentum(df['close'], momentum_lookback)
    volume_acc = volume_acceleration(df['volume'], volume_lookback)
    atr = average_true_range(df['high'], df['low'], df['close'], atr_lookback)
    
    # Combine the factors
    factor = (momentum * volume_acc) + atr
    
    return factor
```
