```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return and Adjusted Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['adjusted_intraday_return'] = df['intraday_return'] / (df['high'] - df['low'])
    
    # Compute 20/30-day Price Momentum
    df['20_day_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    df['volume_shock'] = (df['volume'] > 1.5 * df['volume'].rolling(window=30).mean()).astype(int)
    df['amount_shock'] = (df['amount'] > 1.5 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['high_low_range'] = df['high'] - df['low']
    df['volume_adjusted_high_low'] = (df['high'] - df['low']) * np.sqrt(df['volume'])
    df['high_to_low_ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['volume_moving_average'] = df['volume'].rolling(window=7).mean()
    df['volume_trend'] = np.where(df['volume'] > df['volume_moving_average'], 1, -1)
    df['volume_momentum'] = df['volume'].rolling(window=7).sum() - df['volume'].rolling(window=14).sum().shift(7)
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['open_close_change'] = df['close'] - df['open']
    df['ema_close_7'] = df['close'].ewm(span=8, adjust=False).mean()
    
    # Composite Alpha Factors Synthesis
    df['composite_alpha'] = (df['volume_momentum'] * df['high_to_low_ratio']) + \
                           (-1 * df['volume_momentum']) + \
                           (df['volume_momentum'] * df['high_low_range'] * df['volume']) + \
                           (df['high_to_low_ratio'] * df['volume_trend'])
    
    # Weighted and Integrated Alpha Factor
    df['final_alpha'] = (0.2 * df['adjusted_intraday_return']) + \
                        (0.3 * df['30_day_momentum']) + \
                        (0.4 * df['20_day_momentum']) + \
                        (0.1 * df['volume_shock'] * df['amount_shock']) + \
                        (0.15 * df['volume_momentum']) + \
                        (0.05 * -1 * df['volume_momentum']) + \
                        (0.1 * df['composite_alpha'])
    
    # Incorporate Volume Trend
    df['volume_trend_factor'] = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    df['final_alpha'] *= df['volume_trend_factor']
    
    return df['final_alpha']
```
