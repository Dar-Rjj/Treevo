```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Intraday Return
    df['intraday_vol_adj_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate 30-day Price Momentum
    df['30d_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    
    # Identify Amount Shock
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Combine Factors
    df['combined_factor'] = df['intraday_vol_adj_return'] * df['30d_momentum'] * df['volume_shock'] * df['amount_shock']
    
    # Final Weighting
    alpha_factor = df['combined_factor']
    
    return alpha_factor
```
