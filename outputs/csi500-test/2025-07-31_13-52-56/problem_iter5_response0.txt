import pandas as pd

def heuristics_v2(df):
    # Calculate 30-Day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['30_day_avg_volume'] = df['volume'].rolling(window=30, min_periods=1).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['30_day_avg_volume']).astype(int)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['volume'] * df['high_low_range']
    
    # Calculate Cumulative High-Low Range over N days (e.g., 30 days)
    N = 30
    df['cumulative_high_low_range'] = df['volume_adjusted_high_low_range'].rolling(window=N, min_periods=1).sum()
    
    # Calculate Open-to-Close Price Change
    df['open_close_change'] = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days (e.g., 30 days)
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=N, min_periods=1).sum()
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['cumulative_high_low_range'] / df['cumulative_open_close_change']
    
    # Calculate Difference in Volume Adjusted High-Low Range over N days
    df['volume_adjusted_high_low_range_diff'] = df['volume_adjusted_high_low_range'] - df['volume_adjusted_high_low_range'].shift(N-1).fillna(0)
    
    # Combine Factors
    df['alpha_factor'] = (df['30_day_momentum'] * df['volume_shock']) + df['high_low_range_momentum'] * df['volume_adjusted_high_low_range_diff']
    
    return df['alpha_factor']
