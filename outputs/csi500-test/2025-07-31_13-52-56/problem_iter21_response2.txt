```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range = high_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_range = adjusted_range * volume_trend
    
    # Calculate Momentum Signal
    smoothing_factor = 2 / (15 + 1)
    momentum_ema = df['close'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Trend
    alpha_factor = integrated_adjusted_range - momentum_ema
    
    return alpha_factor
```
