import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range and Volume-Adjusted Version
    high_low_range = df['high'] - df['low']
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])

    # Calculate Cumulative High-Low Range over N days (e.g., 10 days)
    N = 10
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=N).sum()
    
    # Calculate High-to-Low Price Ratio
    high_low_ratio = df['high'] / df['low']

    # Compute Volume Moving Average with a 10-day window
    volume_moving_avg = df['volume'].rolling(window=10).mean()

    # Determine Volume Trend
    volume_trend = np.where(df['volume'] > volume_moving_avg, 1, -1)

    # Measure Volume Momentum
    volume_momentum = df['volume'].rolling(window=10).sum().pct_change()

    # Integrate Open-Close Momentum with Volatility Indicators
    open_close_price_change = df['close'] - df['open']
    close_ema = df['close'].ewm(span=10, adjust=False).mean()

    # Incorporate Volume Trend with High-Low Price Ratio
    high_low_ratio_volume_trend = high_low_ratio * volume_trend

    # Combine Volume Momentum and Adjusted High-Low Range
    combined_volume_momentum = (df['high'] - df['low']) * np.sqrt(df['volume']) * volume_momentum

    # Integrate Open-Close Momentum with Volume-Adjusted High-Low Range
    integrated_open_close = (df['close'] - df['open']) * (df['high'] - df['low']) * np.sqrt(df['volume'])

    # Subtract Momentum Signal from Integrated Adjusted High-Low Range
    momentum_subtracted = integrated_open_close - combined_volume_momentum

    # Calculate Relative Strength
    relative_strength = df['close'] / df['close'].rolling(window=50).mean()

    # Incorporate Relative Strength into Alpha Factor
    alpha_factor = momentum_subtracted * relative_strength

    # Calculate Intraday Return and Adjustments
    intraday_return = (df['high'] - df['low']) / df['low']
    adjusted_intraday_return = intraday_return / (df['high'] - df['low'])

    # Compute 30-day Price Momentum
    price_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Analyze Volume and Amount Shocks
    volume_shock = np.where(df['volume'] > 2 * df['volume'].rolling(window=30).mean(), 1, 0)
    amount_shock = np.where(df['amount'] > 2 * df['amount'].rolling(window=30).mean(), 1, 0)

    # Composite Alpha Factors Synthesis
    composite_alpha = (volume_momentum * (df['high'] / df['low'])) + (-1 * volume_momentum) + \
                     (combined_volume_momentum + (df['high'] / df['low']) * volume_trend)

    # Weighted and Integrated Alpha Factor
    weighted_alpha = (0.2 * adjusted_intraday_return) + (0.3 * price_momentum) + \
                    (0.1 * volume_shock * amount_shock) + (0.2 * combined_volume_momentum) + \
                    (0.15 * -volume_momentum) + (0.1 * composite_alpha)

    # Incorporate Volume Trend
    volume_trend_30 = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    final_alpha = weighted_alpha * volume_trend_30

    return final_alpha
