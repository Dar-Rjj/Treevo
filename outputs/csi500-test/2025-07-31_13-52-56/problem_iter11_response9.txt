import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    :param df: A pandas DataFrame with (open, high, low, close, amount, volume) columns and (date) as index.
    :return: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate rolling mean of daily return over 10 days
    df['rolling_mean_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate rolling standard deviation of daily return over 10 days
    df['rolling_std_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the ratio of intraday volatility to rolling standard deviation
    df['volatility_ratio'] = df['intraday_volatility'] / df['rolling_std_return']
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] / df['close'].shift(10)
    
    # Combine factors into a composite factor
    df['composite_factor'] = (df['volatility_ratio'] * df['momentum']).rank(pct=True)
    
    return df['composite_factor']
