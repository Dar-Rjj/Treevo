```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['avg_volume_30_days'] = df['volume'].rolling(window=30, min_periods=1).mean()
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_30_days']).astype(int)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    df['momentum_with_shock'] = df['30_day_momentum'] * df['volume_shock']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Detect Volume Spike
    df['5_day_avg_volume'] = df['volume'].rolling(window=5, min_periods=1).mean()
    df['volume_spike'] = (df['volume'] > 1.7 * df['5_day_avg_volume']).astype(int)
    
    # Calculate 10-day Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted Range, 10-day Momentum, and Volume Spike
    df['combined_adjusted_range'] = (df['volume_adjusted_high_low_range'] + df['10_day_momentum']) * (df['volume_spike'] * 2.5 + 1)
    
    # Calculate Cumulative High-Low Range
    N = 30
    df['cumulative_high_low_range'] = df['volume_adjusted_high_low_range'].rolling(window=N, min_periods=1).sum()
    
    # Calculate Open-to-Close Price Change
    df['open_close_change'] = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=N, min_periods=1).sum()
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['cumulative_high_low_range'] / df['cumulative_open_close_change']
    
    # Calculate Difference in Volume Adjusted High-Low Range
    df['diff_volume_adjusted_high_low_range'] = df['volume_adjusted_high_low_range'] - df['volume_adjusted_high_low_range'].shift(N-1)
    
    # Calculate Volume Momentum
    n = 30
    df['V_sum'] = df['volume'].rolling(window=n, min_periods=1).sum()
    df['volume_momentum'] = df['V_sum'] - df['V_sum'].shift(n)
    
    # Measure High-Low Range
    df['atr_n_days'] = df['high_low_range'].rolling(window=n, min_periods=1).mean()
    df['high_low_range_ratio'] = df['high_low_range'] / df['low']
    
    # Combined Alpha Factor
    df['adjusted_volume_momentum'] = df['volume_momentum'] * df['high_low_range_ratio']
    df['inverted_volume_momentum'] = -1 * df['volume_momentum']
    df['composite_alpha'] = df['inverted_volume_momentum'] + df['high_low_range_momentum'] + df['adjusted_volume_momentum']
    df['final_factor'] = (df['momentum_with_shock'] + df['combined_adjusted_range']) * df['diff_volume_adjusted_high_low_range']
    
    return df['final_factor']
```
