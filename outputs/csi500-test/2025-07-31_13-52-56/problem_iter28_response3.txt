```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range: High - Low
    high_low_range = df['high'] - df['low']
    
    # Multiply High-to-Low Range by sqrt(Volume)
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    
    # Calculate Cumulative High-Low Range: Sum of Volume Adjusted High-Low Range over N days
    N = 10
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=N).sum()
    
    # Calculate High-to-Low Price Ratio: High / Low
    high_low_price_ratio = df['high'] / df['low']
    
    # Compute Volume Moving Average: Use Volume with a 10-day window
    volume_moving_average = df['volume'].rolling(window=10).mean()
    
    # Determine Volume Trend: +1 if Current Volume > Moving Average, 0 otherwise
    volume_trend = (df['volume'] > volume_moving_average).astype(int)
    
    # Measure Volume Momentum: Change in Aggregated Volume (Sum of past n days' volume)
    n = 5
    aggregated_volume_t_minus_1 = df['volume'].shift(1).rolling(window=n).sum()
    aggregated_volume_t_minus_n = df['volume'].shift(n).rolling(window=n).sum()
    volume_momentum = (aggregated_volume_t_minus_1 - aggregated_volume_t_minus_n) / aggregated_volume_t_minus_n
    
    # Calculate Open-to-Close Price Change: Close - Open
    open_close_price_change = df['close'] - df['open']
    
    # Compute 10-day Exponential Moving Average (EMA) of Close price
    smoothing_factor = 2 / (10 + 1)
    close_ema = df['close'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Combine Volume Trend with High-Low Price Ratio: (High / Low) * Volume Trend
    combined_volume_trend_high_low_ratio = high_low_price_ratio * volume_trend
    
    # Incorporate Volume Momentum and Adjusted High-Low Range: (High - Low) * sqrt(Volume) * Volume Momentum
    integrated_volume_momentum_high_low_range = volume_adjusted_high_low_range * volume_momentum
    
    # Integrate Open-Close Momentum with Volume-Adjusted High-Low Range: (Close - Open) * (High - Low) * sqrt(Volume)
    integrated_open_close_momentum = open_close_price_change * volume_adjusted_high_low_range
    
    # Subtract Momentum Signal from Integrated Adjusted High-Low Range
    final_alpha_factor = (
        combined_volume_trend_high_low_ratio +
        integrated_volume_momentum_high_low_range +
        integrated_open_close_momentum -
        close_ema
    )
    
    return final_alpha_factor
```
