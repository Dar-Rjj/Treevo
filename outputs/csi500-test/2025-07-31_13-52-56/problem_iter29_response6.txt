```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_volatility = (df['high'] - df['low'])
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Compute 20-day and 30-day Price Momentum
    momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    volume_shock = (df['volume'] > 1.5 * df['volume'].rolling(window=30).mean()).astype(int)
    amount_shock = (df['amount'] > 1.5 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Enhanced Volume and Price Volatility Components
    high_low_range = (df['high'] - df['low']) * np.sqrt(df['volume'])
    cumulative_high_low_range = high_low_range.rolling(window=20).sum()
    
    open_close_change = df['close'] - df['open']
    cumulative_open_close_change = open_close_change.rolling(window=20).sum()
    
    volume_moving_avg = df['volume'].rolling(window=7).mean()
    volume_trend = ((df['volume'] > volume_moving_avg) * 2 - 1).astype(int)
    volume_momentum = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    
    # Composite Alpha Factors Synthesis
    adjusted_volume_momentum = volume_momentum * (df['high'] / df['low'])
    inverted_volume_momentum = -1 * volume_momentum
    composite_alpha = (inverted_volume_momentum + adjusted_volume_momentum + 
                       (df['high'] - df['low']) * np.sqrt(df['volume']) * volume_momentum)
    combined_volume_trend_price_ratio = (df['high'] / df['low']) * volume_trend
    
    # Weighted and Integrated Alpha Factor
    weighted_intraday_adjustment = 0.2 * adjusted_intraday_return
    weighted_momentum_20 = 0.4 * momentum_20
    weighted_momentum_30 = 0.3 * momentum_30
    weighted_volume_amount_shocks = 0.1 * volume_shock * amount_shock
    weighted_adjusted_volume_momentum = 0.2 * adjusted_volume_momentum
    weighted_inverted_volume_momentum = 0.05 * inverted_volume_momentum
    weighted_composite_alpha = 0.05 * composite_alpha
    
    # Incorporate Volume Trend
    thirty_day_volume_momentum = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    
    # Combine all components
    alpha_factor = (weighted_intraday_adjustment + weighted_momentum_20 + 
                    weighted_momentum_30 + weighted_volume_amount_shocks + 
                    weighted_adjusted_volume_momentum + weighted_inverted_volume_momentum + 
                    weighted_composite_alpha)
    
    # Refinement through Contextual Analysis
    recent_trend_direction = (df['close'] > df['open'].shift(1)).astype(int) * 2 - 1
    final_alpha = alpha_factor * recent_trend_direction * thirty_day_volume_momentum
    
    return final_alpha
```
