```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    momentum_lookback = 50  # Experiment with 20, 50, and 100 days
    df['price_momentum'] = df['close'] - df['close'].shift(momentum_lookback)
    
    # Calculate Volume Acceleration
    volume_lookback = 20  # Experiment with 10, 20, and 30 days
    df['volume_acceleration'] = df['volume'] - df['volume'].shift(volume_lookback)
    
    # Combine Momentum and Volume
    df['intermediate_factor'] = df['price_momentum'] * df['volume_acceleration']
    
    # Calculate High-to-Low Price Ratio
    df['high_to_low_ratio'] = df['high'] / df['low']
    
    # Calculate Volume Trend
    volume_trend_window = 10  # Choose a window (e.g., 10 days)
    df['volume_ma'] = df['volume'].rolling(window=volume_trend_window).mean()
    df['volume_trend'] = (df['volume'] > df['volume_ma']).astype(int) * 2 - 1
    
    # Final Combination
    df['final_factor'] = df['high_to_low_ratio'] * df['volume_trend'] + df['intermediate_factor']
    
    return df['final_factor']
```
