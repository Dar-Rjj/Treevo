```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range and Adjusted Intraday Return
    df['High_Low_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['High_Low_Range']
    
    # Evaluate Volume Dynamics and Trends
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend'] = np.where(df['volume'] > df['Volume_MA_20'], 1, -1)
    df['Volume_Shock'] = np.where(df['volume'] > 2 * df['volume'].rolling(window=30).mean(), 1, 0)
    df['Volume_Momentum'] = df['volume'].diff(30)
    
    # Integrate Volume-Adjusted High-Low Range
    df['Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(df['volume'])
    df['Integrated_Adjusted_Range'] = df['Adjusted_High_Low_Range'] * df['Volume_Trend']
    
    # Calculate Momentum Signals
    df['EMA_15'] = df['close'].ewm(span=15, adjust=False).mean()
    df['Price_Momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    df['Momentum_Signal'] = df['close'] - df['EMA_15']
    
    # Analyze Open-Close Momentum with Volatility Indicators
    df['Open_Close_Change'] = df['close'] - df['open']
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Synthesize Components into Alpha Factor
    df['Combined_Volume_Trend'] = (df['high'] / df['low']) * df['Volume_Trend']
    df['Integrated_Adjusted_Range_Momentum'] = df['Adjusted_High_Low_Range'] * df['Volume_Momentum']
    df['Open_Close_Momentum'] = df['Open_Close_Change'] * df['Adjusted_High_Low_Range']
    df['Adjusted_Volume_Momentum'] = df['Volume_Momentum'] * (df['high'] / df['low'])
    df['Inverted_Volume_Momentum'] = -1 * df['Volume_Momentum']
    df['Composite_Alpha'] = df['Inverted_Volume_Momentum'] + df['Adjusted_Volume_Momentum'] + (df['high'] - df['low']) * df['volume'] * df['Volume_Momentum']
    
    # Weighted and Integrated Alpha Factor
    df['Weighted_Adjusted_Intraday_Return'] = 0.2 * df['Adjusted_Intraday_Return']
    df['Weighted_Price_Momentum_30'] = 0.3 * df['Price_Momentum_30']
    df['Weighted_Volume_Shock'] = 0.1 * df['Volume_Shock']
    df['Weighted_Adjusted_Volume_Momentum'] = 0.2 * df['Adjusted_Volume_Momentum']
    df['Weighted_Inverted_Volume_Momentum'] = 0.15 * df['Inverted_Volume_Momentum']
    df['Weighted_Composite_Alpha'] = 0.15 * df['Composite_Alpha']
    
    # Final Alpha
    df['Final_Alpha'] = (
        df['Weighted_Adjusted_Intraday_Return'] +
        df['Weighted_Price_Momentum_30'] +
        df['Weighted_Volume_Shock'] +
        df['Weighted_Adjusted_Volume_Momentum'] +
        df['Weighted_Inverted_Volume_Momentum'] +
        df['Weighted_Composite_Alpha']
    )
    
    return df['Final_Alpha']
```
