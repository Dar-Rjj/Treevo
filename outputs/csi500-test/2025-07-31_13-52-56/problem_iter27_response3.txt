```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Evaluate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Return with Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Compute Enhanced High-to-Low Range
    enhanced_high_low_range = (df['high'] - df['low']) * (df['high'] + df['low']) / 2
    
    # Adjust for Volume
    adjusted_high_low_range = enhanced_high_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma_10, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_high_low_range = adjusted_high_low_range * volume_trend
    
    # Calculate Momentum Signal
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Trend
    combined_high_low_momentum = integrated_adjusted_high_low_range - ema_close_10
    
    # Identify Volume Shock
    volume_shock = np.where(df['volume'] > 1.5 * df['volume'].rolling(window=30).mean(), 1, 0)
    
    # Integrate Momentum and Volume Shock
    integrated_momentum_volume_shock = combined_high_low_momentum * volume_shock
    
    # Calculate 20-day Price Momentum
    price_momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Combine High-Low Range and Close Price Momentum
    weighted_average_momentum = 0.5 * combined_high_low_momentum + 0.5 * price_momentum_20
    
    # Add 10-day Volume Acceleration
    volume_acceleration_10 = df['volume'] - df['volume'].shift(10)
    
    # Add 20-day Average True Range
    atr_20 = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Final Factor Combination
    final_factor = (adjusted_intraday_return + weighted_average_momentum + volume_acceleration_10 + atr_20) * integrated_momentum_volume_shock
    
    return final_factor
```
