import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, alpha=0.9):
    # Calculate Volume Momentum Indicator
    df['Volume_Aggregated'] = df['volume'].rolling(window=n).sum()
    df['Volume_Momentum'] = (df['Volume_Aggregated'] - df['Volume_Aggregated'].shift(n)) / df['Volume_Aggregated'].shift(n)

    # Measure High-Low Range Dynamics
    df['True_Range'] = df[['high', 'low', 'close']].join(df[['high', 'low', 'close']].shift(1)).apply(
        lambda x: max(x['high'] - x['low'], x['high'] - x['close_1'], x['close_1'] - x['low']), axis=1)
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['high']

    # Combine Volume Momentum and Range Dynamics
    df['Adjusted_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'] * df['Volume_Momentum'].apply(lambda x: 1 if x > 0 else -1)

    # Calculate High-Low Range Momentum with Exponential Smoothing
    df['High_Low_Range_Smoothed'] = df['High_Low_Range'].ewm(alpha=alpha, adjust=False).mean()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_High_Low_Range_Ratio'] * df['High_Low_Range_Smoothed']

    return df['Alpha_Factor']
