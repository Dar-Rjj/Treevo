```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    df['High_to_Low_Ratio'] = df['high'] / df['low']
    
    # Evaluate Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_MA_10']).astype(int) * 2 - 1
    
    # Integrate Components for Adjusted High-to-Low Price Ratio
    df['Adjusted_High_to_Low_Ratio'] = df['High_to_Low_Ratio'] * df['Volume_Trend']
    
    # Calculate 30-day Price Momentum
    df['Price_Momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['Volume_MA_30'] = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    df['Volume_Shock'] = (df['volume'] > 2 * df['Volume_MA_30']).astype(int)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    df['Momentum_with_Volume_Shock'] = df['Price_Momentum_30'] * df['Volume_Shock']
    
    # Calculate Adjusted High-to-Low Range
    df['High_to_Low_Range'] = df['high'] - df['low']
    df['Adjusted_High_to_Low_Range'] = df['High_to_Low_Range'] * df['volume']
    
    # Detect Volume Spike
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_Spike'] = (df['volume'] > 2.0 * df['Volume_MA_5']).astype(int)
    
    # Calculate 10-day Momentum
    df['Momentum_10'] = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted High-to-Low Range, 10-day Momentum, and Volume Spike
    df['Combined_Adjusted_Range'] = (df['Adjusted_High_to_Low_Range'] + df['Momentum_10']) * (df['Volume_Spike'] * 3.0 + 1 - df['Volume_Spike'])
    
    # Synthesize Final Alpha Factor
    df['Alpha_Factor'] = df['Momentum_with_Volume_Shock'] + df['Combined_Adjusted_Range']
    
    return df['Alpha_Factor']
```
