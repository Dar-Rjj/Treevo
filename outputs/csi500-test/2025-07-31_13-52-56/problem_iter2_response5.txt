import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust Intraday Return
    df['intraday_vol_adj_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['avg_volume_30_day'] = df['volume'].rolling(window=30).mean()
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_30_day']).astype(int)
    
    # Weight by Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Combine Factors
    df['alpha_factor'] = (df['intraday_vol_adj_return'] * 
                          df['30_day_momentum'] * 
                          df['volume_shock'] * 
                          df['intraday_range'])
    
    return df['alpha_factor']
