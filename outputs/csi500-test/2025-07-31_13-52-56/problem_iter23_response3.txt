```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate Enhanced High-Low Range and Its Volume-Adjusted Version
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(df['volume'])
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['Cumulative_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=N).sum()
    df['High_Low_Price_Ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['Volume_Moving_Average'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_Moving_Average']).astype(int) * 2 - 1
    df['Volume_Momentum'] = df['volume'].rolling(window=N).sum() - df['volume'].shift(N).rolling(window=N).sum()
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['EMA_Close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Synthesize Components into Alpha Factor
    df['Component_1'] = df['High_Low_Price_Ratio'] * df['Volume_Trend']
    df['Component_2'] = df['High_Low_Range'] * np.sqrt(df['volume']) * df['Volume_Momentum']
    df['Component_3'] = df['Open_Close_Price_Change'] * df['Volume_Adjusted_High_Low_Range']
    
    df['Alpha_Factor'] = (df['Component_1'] + df['Component_2'] + df['Component_3']) - df['Open_Close_Price_Change']
    
    return df['Alpha_Factor']
```
