import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Advanced High-to-Low Price Ratio
    high_to_low_ratio = (df['high'] - df['low']) / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    V_sum = df['volume'].rolling(window=10).sum()
    V_change = V_sum - V_sum.shift(10)
    
    # Quantify High-Low Range and Volatility
    high_low_range = df['high'] - df['low']
    adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-c': abs(df['high'] - df['close'].shift()),
        'l-c': abs(df['low'] - df['close'].shift())
    }).max(axis=1)
    atr = tr.rolling(window=10).mean()
    price_volatility = df['close'].rolling(window=10).std()
    
    # Integrated and Weighted Alpha Factor
    adjusted_high_low_price_ratio = high_to_low_ratio * volume_trend
    adjusted_high_low_range_ratio = adjusted_high_low_range * volume_trend
    adjusted_volume_momentum = V_change * high_to_low_ratio
    inverted_volume_momentum = -1 * V_change
    composite_alpha = inverted_volume_momentum + price_volatility + adjusted_volume_momentum
    
    weighted_adjusted_volume_momentum = 0.4 * adjusted_volume_momentum
    weighted_inverted_volume_momentum = 0.3 * inverted_volume_momentum
    final_alpha = weighted_adjusted_volume_momentum + weighted_inverted_volume_momentum + 0.3 * price_volatility
    
    # Combine Adjusted Range, Momentum, and Volume Signals
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    momentum_signal = df['close'] - ema_close
    volume_signal = df['volume'] - ema_volume
    adjusted_high_low_range_signal = adjusted_high_low_range - momentum_signal
    final_alpha *= volume_signal
    
    return final_alpha
