```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['intraday_volatility'] = df['high'] - df['low']
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']

    # Compute 30-day Price Momentum
    df['30_day_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Analyze Volume and Amount Shocks
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)

    # Enhanced Volume and Price Volatility Components
    df['vol_adj_high_low_range'] = (df['high'] - df['low']) * np.sqrt(df['volume'])
    df['cumulative_vol_adj_high_low_range'] = df['vol_adj_high_low_range'].rolling(window=10).sum()
    df['open_close_price_change'] = df['close'] - df['open']
    df['cumulative_open_close_price_change'] = df['open_close_price_change'].rolling(window=10).sum()

    # Volume Trends and Momentum
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_moving_avg']).astype(int)
    df['volume_momentum'] = df['volume'].rolling(window=10).sum() - df['volume'].shift(10).rolling(window=10).sum()

    # Composite Alpha Factors Synthesis
    df['adjusted_volume_momentum'] = df['volume_momentum'] * (df['high'] / df['low'])
    df['inverted_volume_momentum'] = -1 * df['volume_momentum']
    df['composite_alpha'] = df['inverted_volume_momentum'] + df['adjusted_volume_momentum'] + (df['high'] - df['low']) * np.sqrt(df['volume']) * df['volume_momentum']
    df['volume_trend_high_low_ratio'] = (df['high'] / df['low']) * df['volume_trend']

    # Weighted and Integrated Alpha Factor
    df['weighted_intraday_return'] = 0.2 * df['adjusted_intraday_return']
    df['weighted_30_day_price_momentum'] = 0.3 * df['30_day_price_momentum']
    df['weighted_volume_amount_shocks'] = 0.1 * df['volume_shock'] * df['amount_shock']
    df['weighted_adjusted_volume_momentum'] = 0.2 * df['adjusted_volume_momentum']
    df['weighted_inverted_volume_momentum'] = 0.15 * df['inverted_volume_momentum']
    df['weighted_composite_alpha'] = 0.1 * df['composite_alpha']
    
    df['final_alpha'] = (
        df['weighted_intraday_return'] +
        df['weighted_30_day_price_momentum'] +
        df['weighted_volume_amount_shocks'] +
        df['weighted_adjusted_volume_momentum'] +
        df['weighted_inverted_volume_momentum'] +
        df['weighted_composite_alpha']
    )

    # Incorporate Volume Trend
    df['30_day_volume_momentum'] = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    df['final_alpha'] = df['final_alpha'] * df['30_day_volume_momentum']

    # Integrate Additional Crossover Components
    df['adjusted_high_to_low_range'] = (df['high'] - df['low']) * np.sqrt(df['volume'])
    df['volume_trend'] = (df['volume'] > df['volume_moving_avg']).astype(int)
    df['integrated_vol_adj_high_low_range'] = df['adjusted_high_to_low_range'] * df['volume_trend']
    df['momentum_signal'] = df['close'].ewm(span=10, adjust=False).mean()
    df['combined_adj_range_momentum'] = df['integrated_vol_adj_high_low_range'] - df['momentum_signal']

    # Update Final Alpha
    df['final_alpha'] += df['combined_adj_range_momentum']

    return df['final_alpha']
```
