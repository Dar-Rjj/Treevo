import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price-Volume Interaction Metrics
    high_low_range = df['high'] - df['low']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Analyze Cumulative and Dynamic Price-Volume Measures
    cumulative_volume_adjusted_high_low_range = volume_adjusted_high_low_range.rolling(window=n).sum()
    high_to_low_price_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trends and Momentum
    volume_moving_average = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_moving_average).astype(int) * 2 - 1
    aggregated_volume = df['volume'].rolling(window=n).sum()
    volume_momentum = aggregated_volume - aggregated_volume.shift(n)
    
    # Integrate Open-Close Momentum and Volatility Indicators
    open_close_price_change = df['close'] - df['open']
    average_true_range = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Synthesize Components into Integrated Alpha Factor
    component_1 = high_to_low_price_ratio * volume_trend
    component_2 = high_low_range * df['volume'] * volume_momentum
    component_3 = open_close_price_change * high_low_range * df['volume']
    price_reversal = (df['high'] - df['low']) / (df['high'] + df['low'])
    adjusted_price_reversal = price_reversal * df['volume']
    
    # Final Alpha Factor
    alpha_factor = (
        0.25 * component_1 +
        0.25 * component_2 +
        0.25 * component_3 +
        0.25 * adjusted_price_reversal
    )
    
    return alpha_factor
