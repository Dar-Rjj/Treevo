import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    
    # Measure Volume Momentum Indicator
    n = 10
    aggregated_volume = df['volume'].rolling(window=n).sum()
    volume_momentum = aggregated_volume.diff().fillna(0)
    
    # Quantify High-Low Range and Volatility
    high_low_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Average True Range
    true_range = np.maximum(np.maximum(df['high'] - df['low'], 
                                       abs(df['high'] - df['close'].shift(1))), 
                            abs(df['low'] - df['close'].shift(1)))
    atr = true_range.rolling(window=n).mean()
    
    # Combine Volume Momentum and Price Volatility
    rolling_volume_sum = df['volume'].rolling(window=n).sum()
    log_return_volume = np.log(rolling_volume_sum / rolling_volume_sum.shift(1))
    tr_standard_deviation = true_range.rolling(window=n).std()
    
    # Integrate Components for Alpha Factor
    adjusted_high_low_ratio = high_low_ratio * volume_trend
    adjusted_high_low_range_ratio = high_low_range_ratio * np.sign(volume_momentum)
    
    inverted_volume_momentum = -volume_momentum
    alpha_factor = adjusted_high_low_ratio + adjusted_high_low_range_ratio + inverted_volume_momentum + tr_standard_deviation
    
    return alpha_factor
