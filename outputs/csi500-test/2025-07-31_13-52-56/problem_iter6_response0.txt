```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Momentum
    df['Rolling_Volume_Sum'] = df['volume'].rolling(window=20).sum()
    df['Log_Return_Volume'] = np.log(df['Rolling_Volume_Sum']) - np.log(df['Rolling_Volume_Sum'].shift(1))
    
    # Calculate Price Volatility
    df['True_Range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['Std_True_Range'] = df['True_Range'].rolling(window=20).std()
    
    # Calculate Volume Spike
    df['Rolling_Volume_Avg'] = df['volume'].rolling(window=20).mean()
    df['Volume_Spike'] = (df['volume'] - df['Rolling_Volume_Avg']) / df['Rolling_Volume_Avg']
    
    # Combine Volume Momentum, Price Volatility, and Volume Spike
    df['Inverted_Vol_Mom'] = -df['Log_Return_Volume']
    df['Composite_Alpha'] = df['Inverted_Vol_Mom'] + df['Std_True_Range'] * df['Volume_Spike']
    
    return df['Composite_Alpha'].dropna()
```
