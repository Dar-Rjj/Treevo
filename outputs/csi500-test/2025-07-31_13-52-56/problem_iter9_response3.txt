```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Volume Momentum
    df['V_sum'] = df['volume'].rolling(window=n).sum()
    df['V_change'] = df['V_sum'] - df['V_sum'].shift(n)
    
    # Measure High-Low Range
    df['ATR'] = df[['high', 'low']].apply(lambda x: (x['high'] - x['low']), axis=1).rolling(window=n).mean()
    df['H/L'] = df['high'] / df['low']
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=n).std()
    
    # Combined Alpha Factor
    df['Adjusted_Volume_Momentum'] = df['V_change'] * df['H/L']
    df['Inverted_Volume_Momentum'] = -1 * df['V_change']
    df['Composite_Alpha'] = (df['Inverted_Volume_Momentum'] + 
                             df['Price_Volatility'] + 
                             df['Adjusted_Volume_Momentum'])
    
    return df['Composite_Alpha']
```
