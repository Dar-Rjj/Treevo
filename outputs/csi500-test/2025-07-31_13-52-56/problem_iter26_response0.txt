import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day moving average of the volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume
    df['vol_ma_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day moving average to the 30-day moving average for both price and volume
    df['price_ma_ratio'] = df['ma_10'] / df['ma_30']
    df['vol_ma_ratio'] = df['vol_ma_10'] / df['vol_ma_30']
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['price_ma_ratio'] * df['vol_ma_ratio']) / df['rsi_14']
    
    return df['alpha_factor']
