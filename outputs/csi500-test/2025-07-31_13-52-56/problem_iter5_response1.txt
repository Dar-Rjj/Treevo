```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Intraday Volatility
    intraday_volatility = df[['high', 'low', 'close', 'open']].std(axis=1)

    # Adjust Intraday Return
    intraday_volatility_adjusted = intraday_return / intraday_volatility

    # Calculate Volume Displacement
    volume_displacement = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Combine Intraday Factors
    combined_intraday_factors = intraday_volatility_adjusted * volume_displacement

    # Calculate Price Momentum
    lookback_period_momentum = 20  # Experiment with 20, 50, and 100 days
    price_momentum = (df['close'] - df['close'].shift(lookback_period_momentum)) / df['close'].shift(lookback_period_momentum)

    # Calculate Volume Acceleration
    lookback_period_volume = 10  # Experiment with 10, 20, and 30 days
    volume_acceleration = (df['volume'] - df['volume'].shift(lookback_period_volume)) / df['volume'].shift(lookback_period_volume)

    # Calculate Trading Range Expansion
    lookback_period_trading_range = 10  # Experiment with 10, 20, and 30 days
    true_range = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    average_true_range = true_range.rolling(window=lookback_period_trading_range).mean()

    # Combine All Factors
    final_factor = (combined_intraday_factors * price_momentum) + volume_acceleration + average_true_range

    return final_factor
```
