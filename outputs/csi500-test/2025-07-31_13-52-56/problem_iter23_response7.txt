import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength of the 10-day to 30-day moving average
    df['rs_ma'] = df['ma_10'] / df['ma_30']
    
    # Calculate the 10-day standard deviation of the daily returns
    df['std_10'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the daily returns
    df['std_30'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the volatility ratio
    df['vol_ratio'] = df['std_10'] / df['std_30']
    
    # Calculate the 10-day cumulative sum of volume
    df['cum_vol_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 30-day cumulative sum of volume
    df['cum_vol_30'] = df['volume'].rolling(window=30).sum()
    
    # Calculate the volume ratio
    df['vol_amount_ratio'] = df['cum_vol_10'] / df['cum_vol_30']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['rs_ma'] * df['vol_ratio']) + df['vol_amount_ratio']
    
    return df['alpha_factor'].dropna()
