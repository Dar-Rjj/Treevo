```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Evaluate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Intraday Return with Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility'].replace(0, np.nan)
    
    # Compute Enhanced High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['adjusted_high_low_range'] = df['high_low_range'] * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = np.where(df['volume'] > df['volume_ma_10'], 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    df['integrated_adjusted_range'] = df['adjusted_high_low_range'] * df['volume_trend']
    
    # Calculate 30-day Price Momentum
    df['momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['volume_shock'] = np.where(df['volume'] > 2 * df['volume'].rolling(window=30).mean(), 1, 0)
    
    # Combine 30-day Momentum and Volume Shock
    df['combined_momentum_shock'] = df['momentum_30'] * df['volume_shock']
    
    # Calculate 10-day Exponential Moving Average (EMA) of Close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, 10-day EMA, and Volume Trend
    df['combined_adjusted_ema'] = df['integrated_adjusted_range'] - df['ema_10']
    
    # Detect Volume Spike
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = np.where(df['volume'] > 1.7 * df['volume_ma_5'], 1, 0)
    
    # Calculate 10-day Price Momentum
    df['momentum_10'] = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted High-to-Low Range, 10-day Momentum, and Volume Spike
    df['combined_signal'] = (df['integrated_adjusted_range'] + df['momentum_10']) * (2.5 if df['volume_spike'] == 1 else 1)
    
    # Calculate 5-day Price Momentum
    df['momentum_5'] = df['close'] - df['close'].shift(5)
    
    # Integrate 5-day Momentum into Combined Signal
    df['enhanced_signal'] = df['combined_signal'] + df['momentum_5']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['combined_momentum_shock'] + df['enhanced_signal']
    
    return df['final_alpha_factor']
```
