```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_to_low_range = high_to_low_range * df['volume']
    
    # Calculate 30-day Price Momentum
    price_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    volume_30_day_avg = df['volume'].rolling(window=30).mean()
    volume_shock = (df['volume'] > 2 * volume_30_day_avg).astype(int)
    
    # Enhance Price Momentum with High-Low Range
    high_30_day = df['high'].rolling(window=30).max()
    low_30_day = df['low'].rolling(window=30).min()
    high_low_range_30_day = high_30_day - low_30_day
    enhanced_price_momentum = price_momentum / high_low_range_30_day
    
    # Combine Adjusted Range with Enhanced Momentum
    combined_factor = adjusted_high_to_low_range + enhanced_price_momentum
    
    # Combine Enhanced Momentum and Volume Shock
    final_factor = combined_factor * volume_shock
    
    return final_factor
```
