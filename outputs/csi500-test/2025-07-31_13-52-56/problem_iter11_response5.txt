```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Ratio with Volume Trend
    high_low_ratio = df['high'] / df['low']
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    high_low_with_volume_trend = high_low_ratio * volume_trend

    # Evaluate Volume and Price Momentum
    volume_momentum = df['volume'].rolling(window=10).sum().pct_change()
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    price_volatility = true_range.rolling(window=10).std()
    combined_volume_price = -volume_momentum + price_volatility

    # Integrate Intraday and Long-term Momentum with Enhanced Shocks
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).std()
    adjusted_intraday_return = intraday_return / intraday_volatility

    price_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    volume_shock = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    amount_shock = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)

    combined_intraday_long_term = adjusted_intraday_return + price_momentum_30d + volume_shock + amount_shock

    # Integrate Components for Final Alpha Factor
    adjusted_high_low = high_low_with_volume_trend * (volume_momentum > 0).astype(int) * 2 - 1
    final_alpha = adjusted_high_low + combined_intraday_long_term

    return final_alpha
```
