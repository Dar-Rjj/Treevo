import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    adjusted_high_to_low_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_moving_average = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_moving_average, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_range = adjusted_high_to_low_range * volume_trend
    
    # Calculate Adjusted Volume Momentum
    n_days = 10
    v_sum = df['volume'].rolling(window=n_days).sum()
    v_change = v_sum.diff().fillna(0)
    
    # Measure and Adjust High-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    adjusted_high_low_ratio = high_low_ratio * volume_trend
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=n_days).std()
    
    # Calculate Momentum Signal
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Integrated Adjusted High-to-Low Range and Momentum
    combined_signal = integrated_adjusted_range - ema_close
    
    # Weighted Combined Alpha
    final_alpha = (0.5 * v_change + 0.5 * price_volatility)
    
    return final_alpha
