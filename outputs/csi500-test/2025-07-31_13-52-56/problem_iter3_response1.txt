import pandas as pd

def heuristics_v2(df, n=10, m=20):
    # Calculate Volume Momentum Indicator
    df['volume_momentum'] = df['volume'].rolling(window=n).sum() - df['volume'].shift(n).rolling(window=n).sum()
    
    # Measure Price Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close']), abs(x['low'] - x['close'])), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=n).mean()
    
    # Analyze Price Trend
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_position'] = df['close'] - df['sma_close']
    
    # Adjusted Volume Momentum
    df['adjusted_volume_momentum'] = df['volume_momentum'] * df['price_position'].apply(lambda x: 1 if x > 0 else -1)
    
    # Adjusted Average True Range
    df['adjusted_avg_true_range'] = df['avg_true_range'] * df['volume_momentum'].apply(lambda x: 1 if x > 0 else -1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_volume_momentum'] + df['adjusted_avg_true_range']
    
    return df['alpha_factor']
