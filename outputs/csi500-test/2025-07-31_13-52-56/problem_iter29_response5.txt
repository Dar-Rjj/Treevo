```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    df['high_low_ratio'] = df['high'] - df['low']
    
    # Evaluate Volume Trend
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = np.where(df['volume'] > df['volume_ma_10'], 1, -1)
    
    # Integrate Components for Adjusted High-to-Low Price Ratio
    df['adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume_trend']
    
    # Adjust for Volume
    df['adjusted_high_low_volume'] = df['adjusted_high_low_ratio'] * np.sqrt(df['volume'])
    
    # Calculate 30-day Price Momentum
    df['momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['volume_shock'] = np.where(df['volume'] > 2 * df['volume'].rolling(window=30).mean(), 1, 0)
    
    # Combine 30-day Momentum and Volume Shock
    df['momentum_30_volume_shock'] = df['momentum_30'] * df['volume_shock']
    
    # Detect Volume Spike
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = np.where(df['volume'] > 1.7 * df['volume_ma_5'], 1, 0)
    
    # Combine Adjusted High-to-Low Range, 30-day Momentum, and Volume Spike
    df['combined_signal'] = (df['adjusted_high_low_volume'] + df['momentum_30_volume_shock']) * (2.5 if df['volume_spike'] == 1 else 1)
    
    # Calculate 10-day Price Momentum
    df['momentum_10'] = df['close'] - df['close'].shift(10)
    
    # Integrate 10-day Momentum into Combined Signal
    df['combined_signal_with_10_day_momentum'] = df['combined_signal'] + df['momentum_10']
    
    # Calculate 5-day Price Momentum
    df['momentum_5'] = df['close'] - df['close'].shift(5)
    
    # Integrate 5-day Momentum into Combined Signal
    df['combined_signal_with_5_day_momentum'] = df['combined_signal_with_10_day_momentum'] + df['momentum_5']
    
    # Calculate Relative Strength
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['relative_strength'] = df['close'] / df['sma_50']
    
    # Incorporate Relative Strength into Alpha Factor
    df['alpha_factor'] = df['combined_signal_with_5_day_momentum'] * df['relative_strength']
    
    return df['alpha_factor']
```
