import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    aggregated_volume = df['volume'].rolling(window=5).sum()
    volume_momentum = aggregated_volume.diff().fillna(0)
    
    # Quantify High-Low Range and Volatility
    high_low_range = df['high'] - df['low']
    atr = (df['high'] - df['low']).rolling(window=14).mean()  # Average True Range
    
    # Compute Volume Adjusted High-Low Range
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Calculate Cumulative High-Low Range
    cumulative_high_low = volume_adjusted_high_low.rolling(window=30).sum()
    
    # Calculate 30-Day Price Momentum
    close_30_days_ago = df['close'].shift(30)
    price_momentum = (df['close'] - close_30_days_ago) / close_30_days_ago
    
    # Identify Volume Shock
    volume_30_day_avg = df['volume'].rolling(window=30).mean().shift(1)  # Exclude current day
    volume_shock = (df['volume'] > 2 * volume_30_day_avg).astype(int)
    
    # Calculate Open-to-Close Price Change
    open_close_change = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change
    cumulative_open_close = open_close_change.rolling(window=30).sum()
    
    # Combine Momentum and Volume Shock
    combined_momentum_shock = price_momentum * volume_shock
    
    # Generate Composite Alpha Factor
    adjusted_high_low_ratio = high_low_ratio * volume_trend
    adjusted_high_low_range = high_low_range * volume_momentum
    composite_alpha_factor = (
        adjusted_high_low_ratio +
        adjusted_high_low_range +
        combined_momentum_shock +
        cumulative_open_close
    )
    
    return composite_alpha_factor
