import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_moving_avg).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    
    # Measure Volume Momentum Indicator
    daily_aggregated_volume = df['volume'].rolling(window=10).sum()
    volume_momentum = daily_aggregated_volume - daily_aggregated_volume.shift(1)
    
    # Quantify High-Low Range and Volatility
    high_low_range = df['high'] - df['low']
    average_true_range = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Adjust for Volume
    adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Momentum Signal
    momentum_signal = df['close'] - df['close'].shift(10)
    
    # Integrate Components for Alpha Factor
    alpha_factor = (high_to_low_ratio * volume_trend) + adjusted_high_low_range + momentum_signal
    
    return alpha_factor
