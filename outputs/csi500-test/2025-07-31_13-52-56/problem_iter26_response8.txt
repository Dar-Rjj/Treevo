```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Evaluate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Return with Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Compute Enhanced High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma_10, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_range = adjusted_high_low_range * volume_trend
    
    # Calculate Momentum Signal
    momentum_signal = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Trend
    combined_factors = integrated_adjusted_range - momentum_signal
    
    # Identify Volume Shock
    volume_shock = np.where(df['volume'] > 1.5 * df['volume'].rolling(window=30).mean(), 1, 0)
    
    # Integrate Momentum and Volume Shock
    integrated_momentum_volume_shock = combined_factors * volume_shock
    
    # Calculate 20-day Price Momentum
    price_momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Relative Strength
    relative_strength = df['close'] / df['close'].rolling(window=50).mean()
    
    # Calculate 10-day Volume Acceleration
    volume_acceleration = df['volume'] / df['volume'].shift(10) - 1
    
    # Calculate 20-day Average True Range
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr_20 = tr.rolling(window=20).mean()
    
    # Final Factor Combination
    final_factor = (integrated_momentum_volume_shock + price_momentum_20) * relative_strength + volume_acceleration + atr_20
    
    return final_factor
```
