import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average volume over the past 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the ratio of today's volume to the average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_10d']
    
    # Calculate the volatility of the last 5 days' returns
    df['volatility_5d'] = df['daily_return'].rolling(window=5).std()
    
    # Calculate the momentum factor as the 30-day return
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    
    # Calculate the RSI (Relative Strength Index) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Calculate the alpha factor as a combination of the above factors
    df['alpha_factor'] = (df['volume_ratio'] * df['momentum_30d']) / (df['volatility_5d'] + 1e-6)
    
    return df['alpha_factor']
