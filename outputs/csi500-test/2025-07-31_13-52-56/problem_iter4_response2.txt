```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Sum Volume Adjusted High-Low Range over N days
    cum_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=N).sum()
    
    # Calculate Open-to-Close Price Change
    open_to_close_change = df['close'] - df['open']
    
    # Sum Open-to-Close Price Change over N days
    cum_open_to_close_change = open_to_close_change.rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = cum_volume_adjusted_high_low / cum_open_to_close_change
    
    return alpha_factor
```
