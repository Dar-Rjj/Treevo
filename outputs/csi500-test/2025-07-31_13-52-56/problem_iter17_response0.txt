import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_volatility = df['high'] - df['low']
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Compute 30-day Price Momentum
    price_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    volume_shock = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    amount_shock = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Enhanced Volume and Price Volatility Components
    high_low_range = (df['high'] - df['low']) * df['volume']
    cumulative_high_low_range = high_low_range.rolling(window=10).sum()
    
    open_close_change = df['close'] - df['open']
    cumulative_open_close_change = open_close_change.rolling(window=10).sum()
    
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_moving_avg).astype(int) * 2 - 1
    volume_momentum = (df['volume'].rolling(window=10).sum() - df['volume'].rolling(window=20).sum().shift(10))
    
    # Composite Alpha Factors Synthesis
    adjusted_volume_momentum = volume_momentum * (df['high'] / df['low'])
    inverted_volume_momentum = -1 * volume_momentum
    composite_alpha = inverted_volume_momentum + adjusted_volume_momentum + high_low_range * volume_momentum
    combined_high_low_volume_trend = (df['high'] / df['low']) * volume_trend
    
    # Weighted and Integrated Alpha Factor
    weighted_intraday_return = 0.2 * adjusted_intraday_return
    weighted_price_momentum = 0.3 * price_momentum_30d
    weighted_volume_amount_shocks = 0.1 * volume_shock * amount_shock
    weighted_adjusted_volume_momentum = 0.2 * adjusted_volume_momentum
    weighted_inverted_volume_momentum = 0.15 * inverted_volume_momentum
    weighted_composite_alpha = 0.15 * composite_alpha
    
    final_alpha = (weighted_intraday_return + weighted_price_momentum + weighted_volume_amount_shocks +
                   weighted_adjusted_volume_momentum + weighted_inverted_volume_momentum + weighted_composite_alpha)
    
    return final_alpha
