```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Calculate Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    
    # Calculate Amount Trend
    amount_ma = df['amount'].rolling(window=10).mean()
    amount_trend = (df['amount'] > amount_ma).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    
    # Combine High-to-Low Price Ratio with Volume and Amount Trends
    factor = high_to_low_ratio * volume_trend * amount_trend
    
    return factor
```
