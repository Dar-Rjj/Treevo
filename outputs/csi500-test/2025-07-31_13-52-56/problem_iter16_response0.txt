import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate High-Low Range and Its Volume-Adjusted Version
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['Cumulative_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=n).sum()
    df['High_Low_Price_Ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['Volume_Moving_Average'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_Moving_Average']).astype(int) * 2 - 1
    df['Aggregated_Volume_t_1'] = df['volume'].shift(1)
    df['Aggregated_Volume_t_n'] = df['volume'].shift(n)
    df['Volume_Momentum'] = df['Aggregated_Volume_t_1'] - df['Aggregated_Volume_t_n']
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['Average_True_Range'] = df['High_Low_Range'].rolling(window=n).mean()
    
    # Synthesize Components into Alpha Factor
    df['Combined_Volume_Trend_High_Low_Ratio'] = df['High_Low_Price_Ratio'] * df['Volume_Trend']
    df['Incorporated_Volume_Momentum_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['volume'] * df['Volume_Momentum']
    df['Integrated_Open_Close_Momentum_Adjusted_High_Low_Range'] = (df['close'] - df['open']) * (df['high'] - df['low']) * df['volume']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Combined_Volume_Trend_High_Low_Ratio'] +
        df['Incorporated_Volume_Momentum_Adjusted_High_Low_Range'] +
        df['Integrated_Open_Close_Momentum_Adjusted_High_Low_Range']
    )
    
    return df['Alpha_Factor'].dropna()
