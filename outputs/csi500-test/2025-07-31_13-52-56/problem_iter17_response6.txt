import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_10_day_ma).astype(int) * 2 - 1
    
    # Integrate Components for Adjusted High-to-Low Price Ratio
    adjusted_high_to_low_ratio = high_to_low_ratio * volume_trend
    
    # Calculate 30-day Price Momentum
    price_momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    volume_shock = (df['volume'] > 2 * df['volume'].rolling(window=30, min_periods=1).mean().shift(1)).astype(int)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    momentum_volume_shock = price_momentum_30 * volume_shock
    
    # Calculate Adjusted High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    adjusted_high_to_low_range = high_to_low_range * df['volume']
    
    # Detect Volume Spike
    volume_5_day_ma = df['volume'].rolling(window=5).mean()
    volume_spike_flag = (df['volume'] > 1.7 * volume_5_day_ma).astype(int)
    
    # Calculate 10-day Momentum
    price_momentum_10 = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted High-to-Low Range, 10-day Momentum, and Volume Spike
    combined_factor = (adjusted_high_to_low_range + price_momentum_10) * (1 + 1.5 * volume_spike_flag)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = momentum_volume_shock + combined_factor
    
    return final_alpha_factor
