```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_to_low_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma_10, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_range = adjusted_high_to_low_range * volume_trend
    
    # Calculate Momentum Signal
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Trend
    combined_factor = integrated_adjusted_range - ema_close_10
    
    # Calculate Relative Strength
    sma_close_50 = df['close'].rolling(window=50).mean()
    relative_strength = df['close'] / sma_close_50
    
    # Incorporate Relative Strength into Alpha Factor
    alpha_factor = combined_factor * relative_strength
    
    return alpha_factor
```
