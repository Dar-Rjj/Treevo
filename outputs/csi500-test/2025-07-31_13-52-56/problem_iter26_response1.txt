```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A pandas Series with the calculated alpha factor values, indexed by date.
    """
    # Calculate the relative strength index (RSI) based on the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the average true range (ATR) using the high, low, and close prices
    tr = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the money flow index (MFI) using the typical price and volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum())))
    
    # Combine the RSI, ATR, and MFI into a single alpha factor
    alpha_factor = (rsi - 50) / 50 + (atr / df['close']).pct_change().fillna(0) + (mfi - 50) / 50
    
    return alpha_factor
```
