```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Enhance Price Momentum with High-Low Range
    high_low_range = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    df['enhanced_momentum'] = df['30_day_momentum'] / high_low_range
    
    # Identify Volume Shock
    avg_volume_30 = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    df['volume_shock'] = (df['volume'] > 2 * avg_volume_30).astype(int)
    
    # Combine Enhanced Momentum and Volume Shock
    df['combined_momentum_vol_shock'] = df['enhanced_momentum'] * df['volume_shock']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    df['vol_adjusted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Cumulative High-Low Range over N days
    N = 30
    df['cumulative_high_low_range'] = df['vol_adjusted_high_low_range'].rolling(window=N).sum()
    
    # Calculate Open-to-Close Price Change
    df['open_close_change'] = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=N).sum()
    
    # Final Combined Alpha Factor
    df['high_low_range_momentum'] = df['cumulative_high_low_range'] / df['cumulative_open_close_change']
    df['final_alpha_factor'] = df['high_low_range_momentum'] * df['combined_momentum_vol_shock'] * (df['vol_adjusted_high_low_range'] - df['vol_adjusted_high_low_range'].shift(N-1))
    
    return df['final_alpha_factor']
```
