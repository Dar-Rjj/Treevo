```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Volume Momentum Indicator
    df['Volume_Sum'] = df['volume'].rolling(window=n).sum()
    df['Volume_Change'] = df['Volume_Sum'] - df['Volume_Sum'].shift(n)

    # Measure High-Low Range Expansion
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['Avg_True_Range'] = df['True_Range'].rolling(window=n).mean()
    df['High_Low_Range_Ratio'] = df['high'] / df['low']

    # Calculate Price Volatility
    df['True_Range_Volatility'] = df['True_Range'].rolling(window=n).std()

    # Adjusted Volume Momentum
    df['Adj_Volume_Momentum'] = df['Volume_Change'] * df['High_Low_Range_Ratio']
    df['Adj_Volume_Momentum'] = np.where(df['Volume_Change'] > 0, df['Adj_Volume_Momentum'], -df['Adj_Volume_Momentum'])

    # Composite Alpha
    df['Inverted_Volume_Momentum'] = -df['Adj_Volume_Momentum']
    df['Composite_Alpha'] = df['Inverted_Volume_Momentum'] + df['True_Range_Volatility']

    return df['Composite_Alpha'].dropna()
```
