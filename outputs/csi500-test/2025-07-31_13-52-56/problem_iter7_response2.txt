```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['avg_volume_30'] = df['volume'].rolling(window=30, min_periods=30).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_30']).astype(int)
    
    # Multiply 30-day Momentum by Volume Shock indicator
    df['factor'] = df['price_momentum'] * df['volume_shock']
    
    return df['factor']
```
