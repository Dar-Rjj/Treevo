```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Calculate 10-day Momentum
    momentum_signal = df['close'] - df['close'].shift(10)
    
    # Calculate 10-day Moving Average of Volume
    volume_signal = df['volume'].rolling(window=10).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Signals
    factor = (adjusted_range + momentum_signal) * volume_signal
    
    return factor
```
