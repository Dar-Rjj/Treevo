```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-Day Exponential Moving Average (EMA) Price Momentum
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    close_price = df['close']
    ema_momentum = (close_price - ema_30) / ema_30
    
    # Identify Sustained Volume Surge
    avg_volume_50 = df['volume'].rolling(window=50).mean().shift(1)
    volume_surge = (df['volume'] > 1.5 * avg_volume_50).rolling(window=3).sum() == 3
    volume_surge_indicator = volume_surge.astype(int)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range 60-Day Momentum
    high_low_range_60 = high_low_range.rolling(window=60).apply(lambda x: x[-1] - x[0], raw=True)
    
    # Combine Factors
    combined_factor = (ema_momentum * volume_surge_indicator + high_low_range_60)
    
    return combined_factor
```
