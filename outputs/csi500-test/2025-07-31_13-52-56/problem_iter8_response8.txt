```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Ratio with Volume Trend
    high_low_ratio = df['high'] / df['low']
    
    # Compute Volume Moving Average
    volume_ma = df['volume'].rolling(window=10).mean()
    
    # Determine if Volume is Increasing or Decreasing
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1
    
    # Combine High-to-Low Price Ratio with Volume Trend
    high_low_volume_trend = high_low_ratio * volume_trend

    # Evaluate Volume and Price Momentum
    # Measure Volume Momentum Indicator
    volume_momentum = df['volume'].rolling(window=10).sum().pct_change()

    # Calculate Price Volatility
    true_range = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    price_volatility = true_range.rolling(window=10).std()

    # Combine Volume Momentum and Price Volatility
    inverted_volume_momentum = -volume_momentum
    volume_price_factor = inverted_volume_momentum + price_volatility

    # Integrate Intraday and Long-term Momentum with Volume Shocks
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Intraday Volatility Adjusted
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close'] + df['open']) / 4)
    adjusted_intraday_return = intraday_return / intraday_volatility

    # Calculate 30-day Price Momentum
    price_momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Identify Volume Shock
    volume_shock = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)

    # Combine Intraday and Long-term Factors
    combined_intraday_long_term = adjusted_intraday_return + price_momentum_30 + volume_shock

    # Integrate Components for Final Alpha Factor
    # Adjusted High-Low Range Ratio
    adjusted_high_low_ratio = high_low_volume_trend * (volume_momentum > 0).astype(int)

    # Final Alpha Factor
    final_alpha = adjusted_high_low_ratio + combined_intraday_long_term

    return final_alpha
```
