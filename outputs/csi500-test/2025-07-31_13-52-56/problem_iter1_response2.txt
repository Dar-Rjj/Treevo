import pandas as pd
from sklearn.linear_model import LinearRegression

def heuristics_v2(df):
    # Momentum Component
    short_term_momentum = df['close'] - df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Support Component
    def fit_linear_regression(series, window):
        model = LinearRegression()
        X = pd.DataFrame(range(1, window + 1), index=series.index[-window:])
        y = series.iloc[-window:].values.reshape(-1, 1)
        model.fit(X, y)
        return model.coef_[0][0]
    
    low_price_trendline_slope = df['low'].rolling(window=10).apply(fit_linear_regression, args=(10,), raw=False)
    high_price_resistance_slope = df['high'].rolling(window=10).apply(fit_linear_regression, args=(10,), raw=False)
    
    # Integrate Momentum and Support
    combined_momentum = 0.6 * short_term_momentum + 0.4 * long_term_momentum
    support_component = (low_price_trendline_slope > high_price_resistance_slope).astype(int) - 0.5
    
    # Volume Influence
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_influence = daily_volume_change / average_volume_10_days
    
    # Final Integration
    alpha_factor = (combined_momentum * volume_influence + support_component).dropna()
    
    return alpha_factor
