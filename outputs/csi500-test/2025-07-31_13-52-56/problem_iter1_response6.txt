```python
import pandas as pd

def heuristics_v2(df):
    # Determine the look-back periods for momentum and volume
    momentum_lookback = 50  # Experiment with 20, 50, and 100 days
    volume_lookback = 20    # Experiment with 10, 20, and 30 days

    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(momentum_lookback)

    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'].pct_change(volume_lookback)

    # Combine Momentum and Volume
    df['factor'] = df['price_momentum'] * df['volume_acceleration']

    # Return the factor as a pandas Series indexed by date
    return df['factor']
```
