```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum over N periods
    N = 10
    df['high_low_range_momentum'] = df['high_low_range'].rolling(window=N).apply(lambda x: x[-1] - x[0], raw=True)
    
    # Calculate Close Price Momentum over 30 days
    df['close_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Combine High-Low Range and Close Price Momentum
    df['integrated_momentum'] = (df['high_low_range_momentum'] + df['close_price_momentum']) / 2
    
    # Identify Volume Shock
    df['avg_volume_30d'] = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_30d']).astype(int)
    
    # Integrate Momentum and Volume Shock
    df['alpha_factor'] = df['integrated_momentum'] * df['volume_shock']
    
    return df['alpha_factor'].dropna()
```
