import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Volume Momentum
    V_sum = df['volume'].rolling(window=n).sum()
    V_change = V_sum - V_sum.shift(n)

    # Measure High-Low Range
    ATR = df[['high', 'low']].apply(lambda x: (x['high'] - x['low']), axis=1).rolling(window=n).mean()
    H_L = (df['high'] - df['low']) / df['close']

    # Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=n).std()

    # Combined Alpha Factor
    adjusted_volume_momentum = V_change * H_L
    inverted_volume_momentum = -1 * V_change
    composite_alpha = inverted_volume_momentum + price_volatility + adjusted_volume_momentum

    return composite_alpha
