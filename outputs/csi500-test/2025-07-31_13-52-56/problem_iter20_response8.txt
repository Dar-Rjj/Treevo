```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return and Adjusted Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['adjusted_intraday_return'] = df['intraday_return'] / (df['high'] - df['low'])
    
    # Compute 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['high_low_range'] = df['high'] - df['low']
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * np.sqrt(df['volume'])
    df['cumulative_high_low_range'] = df['volume_adjusted_high_low_range'].rolling(window=30).sum()
    df['high_to_low_ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['volume_moving_average'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_moving_average']).astype(int) * 2 - 1
    df['volume_momentum'] = df['volume'].rolling(window=10).sum() - df['volume'].rolling(window=20).sum().shift(10)
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['open_close_price_change'] = df['close'] - df['open']
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Synthesize Components into Alpha Factor
    df['volume_trend_high_low_ratio'] = df['high_to_low_ratio'] * df['volume_trend']
    df['volume_momentum_high_low_range'] = df['high_low_range'] * np.sqrt(df['volume']) * df['volume_momentum']
    df['open_close_momentum_high_low_range'] = (df['close'] - df['open']) * (df['high'] - df['low']) * np.sqrt(df['volume'])
    df['adjusted_volume_momentum'] = df['volume_momentum'] * (df['high'] / df['low'])
    df['inverted_volume_momentum'] = -1 * df['volume_momentum']
    df['composite_alpha'] = (df['inverted_volume_momentum'] + df['adjusted_volume_momentum'] + 
                             (df['high'] - df['low']) * df['volume'] * df['volume_momentum'])
    
    # Weighted and Integrated Alpha Factor
    df['weighted_intraday_return'] = 0.2 * df['adjusted_intraday_return']
    df['weighted_30_day_momentum'] = 0.3 * df['30_day_momentum']
    df['weighted_volume_amount_shocks'] = 0.1 * df['volume_shock'] * df['amount_shock']
    df['weighted_adjusted_volume_momentum'] = 0.2 * df['adjusted_volume_momentum']
    df['weighted_inverted_volume_momentum'] = 0.15 * df['inverted_volume_momentum']
    df['weighted_composite_alpha'] = 0.15 * df['composite_alpha']
    
    # Final Alpha
    df['final_alpha'] = (df['weighted_intraday_return'] + df['weighted_30_day_momentum'] + 
                         df['weighted_volume_amount_shocks'] + df['weighted_adjusted_volume_momentum'] + 
                         df['weighted_inverted_volume_momentum'] + df['weighted_composite_alpha'])
    
    return df['final_alpha']
```
