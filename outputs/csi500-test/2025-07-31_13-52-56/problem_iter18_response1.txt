import pandas as pd

def heuristics_v2(df):
    # Compute the daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the daily price range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute the daily volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute the daily amount change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Detect Inside Day pattern
    df['inside_day'] = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1))
    
    # Detect Outside Day pattern
    df['outside_day'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    
    # Compute the 5-day moving average of daily returns
    df['ma_5'] = df['daily_return'].rolling(window=5).mean()
    
    # Compute the 10-day moving average of daily returns
    df['ma_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Determine the momentum
    df['momentum'] = df['ma_5'] - df['ma_10']
    
    # Compute the daily gap
    df['daily_gap'] = df['open'] - df['close'].shift(1)
    
    # Compute the daily closing direction
    df['closing_direction'] = df['close'] - df['open']
    
    # Create a composite factor by summing up selected individual factors
    df['composite_factor'] = (
        df['daily_return'] + 
        df['daily_range'] * 0.5 +  # Weighted by 0.5
        df['volume_change'] * 0.3 +  # Weighted by 0.3
        df['amount_change'] * 0.2 +  # Weighted by 0.2
        df['inside_day'].astype(int) * 0.8 +  # Weighted by 0.8
        df['outside_day'].astype(int) * 0.7 +  # Weighted by 0.7
        df['momentum'] * 1.0 +  # Weighted by 1.0
        df['daily_gap'] * 0.6 +  # Weighted by 0.6
        df['closing_direction'] * 0.4  # Weighted by 0.4
    )
    
    # Return the composite factor
    return df['composite_factor'].dropna()
