```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    adjusted_high_to_low_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Calculate 10-day EMA of Close Price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 10-day EMA of Volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Combine Signals
    combined_signal = (adjusted_high_to_low_range - ema_close) * ema_volume
    
    return combined_signal
```
