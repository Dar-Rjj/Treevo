```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['30_day_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['30_day_avg_volume'] = df['volume'].rolling(window=30, min_periods=30).mean()
    df['volume_shock'] = (df['volume'] > 2 * df['30_day_avg_volume']).astype(int)
    
    # Enhance Price Momentum with High-Low Range
    df['30_day_high'] = df['high'].rolling(window=30, min_periods=30).max()
    df['30_day_low'] = df['low'].rolling(window=30, min_periods=30).min()
    df['30_day_high_low_range'] = df['30_day_high'] - df['30_day_low']
    df['enhanced_momentum'] = df['30_day_price_momentum'] / df['30_day_high_low_range']
    
    # Combine Enhanced Momentum and Volume Shock
    df['alpha_factor'] = df['enhanced_momentum'] * df['volume_shock']
    
    return df['alpha_factor']
```
