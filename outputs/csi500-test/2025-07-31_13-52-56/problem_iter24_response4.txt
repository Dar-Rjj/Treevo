```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return and Adjusted Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Volatility'] = df['high'] - df['low']
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['Volatility'].replace(0, np.nan)
    
    # Compute 30-day Price Momentum
    df['30d_Price_Momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    df['Volume_Shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    df['Amount_Shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low'] = df['High_Low_Range'] * np.sqrt(df['volume'])
    df['Cumulative_High_Low_Range'] = df['Volume_Adjusted_High_Low'].rolling(window=10).sum()
    df['High_to_Low_Ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['Volume_MA_10d'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_MA_10d']).astype(int) * 2 - 1
    df['Volume_Momentum'] = df['volume'].rolling(window=10).sum() - df['volume'].shift(10).rolling(window=10).sum()
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Momentum'] = df['close'] - df['open']
    df['EMA_Close_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Synthesize Components into Alpha Factor
    df['Volume_Trend_High_Low_Ratio'] = df['High_to_Low_Ratio'] * df['Volume_Trend']
    df['Volume_Momentum_Adjusted_High_Low'] = df['High_Low_Range'] * np.sqrt(df['volume']) * df['Volume_Momentum']
    df['Open_Close_Volume_Adjusted_High_Low'] = df['Open_Close_Momentum'] * df['High_Low_Range'] * np.sqrt(df['volume'])
    df['Adjusted_Volume_Momentum'] = df['Volume_Momentum'] * (df['high'] / df['low'])
    df['Inverted_Volume_Momentum'] = -1 * df['Volume_Momentum']
    df['Composite_Alpha'] = df['Inverted_Volume_Momentum'] + df['Adjusted_Volume_Momentum'] + (df['High_Low_Range'] * df['volume'] * df['Volume_Momentum'])
    
    # Weighted and Integrated Alpha Factor
    df['Weighted_Intraday_Return'] = 0.2 * df['Adjusted_Intraday_Return']
    df['Weighted_30d_Price_Momentum'] = 0.3 * df['30d_Price_Momentum']
    df['Weighted_Volume_Amount_Shocks'] = 0.1 * df['Volume_Shock'] * df['Amount_Shock']
    df['Weighted_Adjusted_Volume_Momentum'] = 0.2 * df['Adjusted_Volume_Momentum']
    df['Weighted_Inverted_Volume_Momentum'] = 0.15 * df['Inverted_Volume_Momentum']
    df['Weighted_Composite_Alpha'] = 0.15 * df['Composite_Alpha']
    
    # Final Alpha
    df['Final_Alpha'] = (
        df['Weighted_Intraday_Return'] +
        df['Weighted_30d_Price_Momentum'] +
        df['Weighted_Volume_Amount_Shocks'] +
        df['Weighted_Adjusted_Volume_Momentum'] +
        df['Weighted_Inverted_Volume_Momentum'] +
        df['Weighted_Composite_Alpha']
    )
    
    return df['Final_Alpha']
```
