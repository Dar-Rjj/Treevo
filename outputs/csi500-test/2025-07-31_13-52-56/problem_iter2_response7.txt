```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Cumulative High-Low Range
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=N).sum()
    
    # Calculate Momentum
    momentum = cumulative_high_low_range.diff(periods=N-1)
    
    # Calculate Open-to-Close Price Change
    open_to_close_change = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change
    cumulative_open_to_close_change = open_to_close_change.rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = momentum / cumulative_open_to_close_change
    
    return alpha_factor
```
