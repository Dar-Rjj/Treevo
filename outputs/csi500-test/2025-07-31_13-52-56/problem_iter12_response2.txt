```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Volume Momentum
    V_sum = df['volume'].rolling(window=n).sum()
    V_change = V_sum - V_sum.shift(n)
    
    # Measure High-Low Range
    ATR = (df['high'] - df['low']).rolling(window=n).mean()
    H_L = (df['high'] / df['low']).rolling(window=n).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=n).std()
    
    # Combined Alpha Factor
    adjusted_volume_momentum = V_change * H_L
    inverted_volume_momentum = -1 * V_change
    composite_alpha = inverted_volume_momentum + price_volatility + adjusted_volume_momentum
    
    # Weighted Composite Alpha
    weighted_adjusted_volume_momentum = 0.4 * adjusted_volume_momentum
    weighted_inverted_volume_momentum = 0.3 * inverted_volume_momentum
    final_alpha = (weighted_adjusted_volume_momentum 
                   + weighted_inverted_volume_momentum 
                   + 0.3 * price_volatility)
    
    return final_alpha
```
