```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Momentum
    rolling_volume_sum = df['volume'].rolling(window=20).sum()
    log_return_volume = np.log(rolling_volume_sum) - np.log(rolling_volume_sum.shift(1))
    
    # Calculate Price Volatility
    high = df['high']
    low = df['low']
    close = df['close']
    true_range = np.maximum(high - low, np.abs(high - close.shift(1)), np.abs(low - close.shift(1)))
    price_volatility = true_range.rolling(window=20).std()
    
    # Combine Volume Momentum and Price Volatility
    volume_momentum = -log_return_volume
    composite_alpha = volume_momentum + price_volatility
    
    return composite_alpha
```
