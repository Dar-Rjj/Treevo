```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_to_low_range = high_to_low_range * np.sqrt(df['volume'])
    
    # Calculate 10-day Exponential Moving Average (EMA) of Close price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 10-day Exponential Moving Average (EMA) of Volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range, Momentum, and Volume Signals
    momentum_signal = ema_close
    volume_signal = ema_volume
    factor = (adjusted_high_to_low_range - momentum_signal) * volume_signal
    
    return factor
```
