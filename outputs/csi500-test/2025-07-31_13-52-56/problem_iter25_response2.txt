import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Range
    adjusted_range = (df['high'] - df['low']) * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_range = adjusted_range * volume_trend
    
    # Calculate Momentum Signal
    alpha = 2 / (11 + 1)  # Smoothing factor for EMA
    momentum_signal = df['close'].ewm(alpha=alpha, adjust=False).mean()
    
    # Combine Adjusted Range and Momentum
    alpha_factor = integrated_range - momentum_signal
    
    return alpha_factor
