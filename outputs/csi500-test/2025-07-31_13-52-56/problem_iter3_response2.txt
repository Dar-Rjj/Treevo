```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['avg_vol_30'] = df['volume'].rolling(window=30, min_periods=1).mean()
    df['volume_shock'] = (df['volume'] > 2 * df['avg_vol_30']).astype(int)
    
    # Calculate 30-day Price Volatility
    df['volatility_30'] = df['close'].rolling(window=30, min_periods=1).std()
    
    # Combine Momentum, Volume Shock, and Price Volatility
    df['factor'] = (df['momentum'] * df['volume_shock']) / df['volatility_30']
    
    # Return the factor as a pandas Series
    return df['factor']
```
