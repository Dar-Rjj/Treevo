```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Volume Momentum Indicator
    df['Volume_Momentum'] = df['volume'].rolling(window=n).sum() / df['volume'].shift(n).rolling(window=n).sum()

    # Measure High-Low Range Expansion
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=n).mean()
    df['High_Low_Range_Ratio'] = (df['high'] - df['low']) / df['Average_True_Range']

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']

    # Intraday Volatility Adjusted
    df['Intraday_Volatility'] = df[['high', 'low', 'close', 'open']].std(axis=1)
    df['Intraday_Volatility_Adjusted_Return'] = df['Intraday_Return'] / df['Intraday_Volatility']

    # Calculate Volume Displacement
    df['Volume_Displacement'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Combine Volume Momentum and Range Expansion
    df['Adjusted_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'] * df['Volume_Momentum']
    df['Combined_Factor_1'] = df['Adjusted_High_Low_Range_Ratio'].where(df['Volume_Momentum'] > 0, -df['Adjusted_High_Low_Range_Ratio'])

    # Combine Factors
    df['Final_Factor'] = (
        df['Intraday_Volatility_Adjusted_Return'] +
        df['Volume_Displacement'] +
        df['Combined_Factor_1']
    ) / 3

    return df['Final_Factor'].dropna()
```
