```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Volume Momentum Indicator
    df['volume_momentum'] = df['volume'].rolling(window=n).sum() / df['volume'].shift(n).rolling(window=n).sum()
    
    # Measure High-Low Range Expansion
    df['true_range'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=n).mean()
    df['high_low_range_ratio'] = (df['high'] - df['low']) / df['avg_true_range']
    
    # Combine Volume Momentum and Range Expansion
    df['adjusted_high_low_range_ratio'] = df['high_low_range_ratio'] * df['volume_momentum']
    
    # Calculate High-Low Range Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(n-1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_high_low_range_ratio'] * df['high_low_range_momentum']
    
    return df['alpha_factor'].dropna()
```
