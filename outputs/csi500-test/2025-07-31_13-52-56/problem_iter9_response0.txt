```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day Price Momentum
    df['20_day_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Identify Volume Shock
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_20']).astype(int)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    df['vol_adj_high_low_range'] = df['volume'] * df['high_low_range']
    
    # Calculate Cumulative High-Low Range over N days (e.g., 20 days)
    N = 20
    df['cumulative_vol_adj_high_low_range'] = df['vol_adj_high_low_range'].rolling(window=N).sum()
    
    # Calculate Open-to-Close Price Change
    df['open_close_change'] = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=N).sum()
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['cumulative_vol_adj_high_low_range'] / df['cumulative_open_close_change']
    
    # Calculate Difference in Volume Adjusted High-Low Range
    df['diff_vol_adj_high_low_range'] = df['vol_adj_high_low_range'] - df['vol_adj_high_low_range'].shift(N-1)
    
    # Combine Factors
    df['alpha_factor'] = (df['20_day_momentum'] * df['volume_shock'] + 
                          df['high_low_range_momentum'] * df['diff_vol_adj_high_low_range'])
    
    return df['alpha_factor']
```
