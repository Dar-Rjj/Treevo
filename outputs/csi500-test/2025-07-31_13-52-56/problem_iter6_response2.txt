```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range = high_to_low_range * df['volume']
    
    # Detect Volume Spike
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_spike_flag = (df['volume'] > 1.7 * volume_5day_ma).astype(int)
    
    # Calculate Momentum Signal
    momentum_signal = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted Range, Momentum, and Volume Spike
    factor_value = adjusted_range + momentum_signal
    factor_value = factor_value * (1 + 1.5 * volume_spike_flag)
    
    return factor_value
```
