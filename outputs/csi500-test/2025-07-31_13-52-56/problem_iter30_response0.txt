```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages for different time periods
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between short-term and long-term SMAs
    df['SMA_5_20_diff'] = df['SMA_5'] - df['SMA_20']
    df['SMA_20_50_diff'] = df['SMA_20'] - df['SMA_50']
    
    # Calculate the Average True Range (ATR)
    df['TrueRange'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['ATR_14'] = df['TrueRange'].rolling(window=14).mean()
    
    # Calculate price change over a certain period
    df['Price_Change_1D'] = df['close'].diff()
    df['Price_Change_5D'] = df['close'] - df['close'].shift(5)
    
    # Analyze volume trends
    df['Volume_SMA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_SMA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_OSC'] = (df['Volume_SMA_5'] - df['Volume_SMA_20']) / df['Volume_SMA_20']
    
    # Calculate the Accumulation/Distribution Line (ADL)
    df['ADL'] = (df['close'] - df['low']) - (df['high'] - df['close'])
    df['ADL'] = df['ADL'].cumsum()
    
    # Investigate the relationship between price and volume
    df['Price_Volume_Ratio'] = df['close'] / df['volume']
    df['Price_Change_Volume_Ratio'] = df['Price_Change_1D'] / df['volume']
    
    # Create a composite factor combining multiple indicators
    df['Composite_Factor'] = (df['SMA_5_20_diff'] + 
                              df['ATR_14'] * 0.5 + 
                              df['Price_Change_5D'] / df['Price_Change_1D'] + 
                              df['Volume_OSC'] * 0.3)
    
    # Return the composite factor
    return df['Composite_Factor']
```
