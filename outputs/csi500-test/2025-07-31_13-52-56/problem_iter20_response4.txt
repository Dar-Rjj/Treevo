import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Volume Momentum
    V_sum = df['volume'].rolling(window=n).sum()
    V_change = V_sum - V_sum.shift(n)

    # Measure High-Low Range
    H_L_ratio = df['high'] / df['low']

    # Adjusted High-Low Price Ratio with Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1
    adjusted_H_L = H_L_ratio * volume_trend

    # Price Volatility
    price_volatility = df['close'].rolling(window=n).std()

    # Evaluate Volume Trend
    volume_trend_factor = (df['volume'] > volume_ma).astype(int) * 2 - 1

    # Weighted Combined Alpha
    adjusted_volume_momentum = V_change * H_L_ratio
    inverted_volume_momentum = -1 * V_change
    final_alpha = (0.4 * adjusted_volume_momentum + 
                   0.3 * inverted_volume_momentum + 
                   0.3 * price_volatility)

    return final_alpha
