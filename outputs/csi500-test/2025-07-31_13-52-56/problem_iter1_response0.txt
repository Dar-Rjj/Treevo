```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Calculate Volume Moving Average (e.g., 10 days)
    window = 10
    volume_ma = df['volume'].rolling(window=window).mean()
    
    # Determine if Volume is Increasing or Decreasing
    volume_trend = (df['volume'] > volume_ma).astype(int) - (df['volume'] < volume_ma).astype(int)
    
    # Combine High-to-Low Price Ratio and Volume Trend
    factor = high_to_low_ratio * volume_trend
    
    return factor
```
