```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate High-Low Range and its Volume-Adjusted Version
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(df['volume'])

    # Evaluate Cumulative and Dynamic High-Low Measures
    df['Cumulative_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=n).sum()
    df['High_to_Low_Ratio'] = df['high'] / df['low']

    # Analyze Volume Trends and Momentum
    df['Volume_MA'] = df['volume'].rolling(window=n).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_MA']).astype(int)
    df['Aggregated_Volume_t_1'] = df['volume'].shift(1).fillna(0).rolling(window=n).sum()
    df['Aggregated_Volume_t_n'] = df['volume'].rolling(window=n).sum()
    df['Volume_Momentum'] = df['Aggregated_Volume_t_n'] - df['Aggregated_Volume_t_1']

    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['Close_EMA'] = df['close'].ewm(span=n, adjust=False).mean()

    # Synthesize Components into Alpha Factor
    df['Component_1'] = df['High_to_Low_Ratio'] * df['Volume_Trend']
    df['Component_2'] = df['High_Low_Range'] * np.sqrt(df['volume']) * df['Volume_Momentum']
    df['Component_3'] = df['Open_Close_Price_Change'] * df['High_Low_Range'] * np.sqrt(df['volume'])

    df['Alpha_Factor'] = (df['Component_1'] + df['Component_2'] + df['Component_3']) - df['Open_Close_Price_Change']

    return df['Alpha_Factor']
```
