```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Price Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Determine if Volume is Increasing or Decreasing
    volume_ma_window = 10
    df['Volume_MA'] = df['volume'].rolling(window=volume_ma_window).mean()
    df['Volume_Trend'] = np.where(df['volume'] > df['Volume_MA'], 1, -1)
    
    # Combine High-to-Low Price Ratio with Volume Trend
    df['High_Low_Volume_Trend'] = df['High_Low_Ratio'] * df['Volume_Trend']
    
    # Measure Volume Momentum Indicator
    volume_momentum_window = 5
    df['Volume_Momentum'] = df['volume'].rolling(window=volume_momentum_window).sum() - df['volume'].shift(volume_momentum_window).rolling(window=volume_momentum_window).sum()
    
    # Calculate Price Volatility
    df['True_Range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    volatility_window = 20
    df['Price_Volatility'] = df['True_Range'].rolling(window=volatility_window).std()
    
    # Combine Volume Momentum and Price Volatility
    df['Inverted_Volume_Momentum'] = -df['Volume_Momentum']
    df['Combined_Volume_Price'] = df['Inverted_Volume_Momentum'] + df['Price_Volatility']
    
    # Adjusted High-Low Range Ratio
    df['Adjusted_High_Low_Range'] = df['High_Low_Volume_Trend'] * np.sign(df['Volume_Momentum'])
    
    # Combine Adjusted High-Low Range Ratio with Combined Volume and Price Momentum
    df['Final_Alpha_Factor'] = df['Adjusted_High_Low_Range'] + df['Combined_Volume_Price']
    
    return df['Final_Alpha_Factor']
```
