```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Range
    high_low_range = df['high'] - df['low']
    adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    
    # Evaluate Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = np.where(df['volume'] > volume_10d_ma, 1, -1)
    
    # Integrate Volume Trend with Adjusted Range
    integrated_adjusted_range = adjusted_high_low_range * volume_trend
    
    # Calculate Momentum Signal
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine Adjusted Range and Momentum
    combined_alpha_factor = integrated_adjusted_range - close_10d_ema
    
    # Calculate Relative Strength
    close_50d_sma = df['close'].rolling(window=50).mean()
    relative_strength = df['close'] / close_50d_sma
    
    # Incorporate Relative Strength into Alpha Factor
    final_alpha_factor = combined_alpha_factor * relative_strength
    
    return final_alpha_factor
```
