```python
import pandas as pd

def heuristics_v2(df, N=30):
    # 30-Day Price Momentum
    close_prices = df['close']
    price_momentum = (close_prices - close_prices.shift(N)) / close_prices.shift(N)
    
    # Volume Shock
    volume = df['volume']
    avg_volume_30 = volume.rolling(window=N).mean().shift(1)  # exclude day t
    volume_shock = (volume > 2 * avg_volume_30).astype(int)
    
    # High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    range_momentum = (high_low_range - high_low_range.shift(N-1)) / high_low_range.shift(N-1)
    
    # Combine Factors
    combined_factor = (price_momentum * volume_shock) + range_momentum
    
    return combined_factor
```
