```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df[['high', 'low', 'close', 'open']].std(axis=1))
    
    # Adjust Intraday Return
    intraday_volatility_adjusted = intraday_return / intraday_volatility
    
    # Calculate 30-day Price Momentum
    price_momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    avg_volume_30 = df['volume'].rolling(window=30, min_periods=1).mean().shift(1)
    volume_shock = (df['volume'] > 2 * avg_volume_30).astype(int)
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_high_low_range = high_low_range * df['volume']
    
    # Detect Volume Spike
    moving_avg_volume_5 = df['volume'].rolling(window=5, min_periods=1).mean()
    volume_spike = (df['volume'] > 1.7 * moving_avg_volume_5).astype(int)
    
    # Combine Factors
    combined_factor = (intraday_volatility_adjusted + 
                       price_momentum_30 + 
                       volume_shock + 
                       adjusted_high_low_range) * (2.5 if volume_spike == 1 else 1)
    
    return combined_factor
```
