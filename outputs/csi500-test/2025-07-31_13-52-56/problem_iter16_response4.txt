```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > volume_ma_10).astype(int) * 2 - 1
    
    # Integrate Components for Alpha Factor
    alpha_factor = high_to_low_ratio * volume_trend
    
    return alpha_factor
```
