```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Advanced High-to-Low Price Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Evaluate Volume Trend and Momentum
    volume_ma_window = 10
    volume_ma = df['volume'].rolling(window=volume_ma_window).mean()
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    
    n_days = 5
    V_sum = df['volume'].rolling(window=n_days).sum()
    V_change = V_sum.diff().fillna(0)
    
    # Quantify High-Low Range and Volatility
    high_low_range_ratio = (df['high'] - df['low']) / df['low']
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    avg_true_range = true_range.rolling(window=n_days).mean()
    price_volatility = df['close'].rolling(window=n_days).std()
    
    # Integrated and Weighted Alpha Factor
    adjusted_high_low_ratio = high_low_ratio * volume_trend
    
    adjusted_high_low_range_ratio = high_low_range_ratio * volume_trend
    adjusted_volume_momentum = V_change * high_low_ratio
    inverted_volume_momentum = -1 * V_change
    
    composite_alpha = (inverted_volume_momentum + price_volatility + adjusted_volume_momentum)
    
    weighted_adjusted_volume_momentum = 0.4 * adjusted_volume_momentum
    weighted_inverted_volume_momentum = 0.3 * inverted_volume_momentum
    final_alpha = (weighted_adjusted_volume_momentum + weighted_inverted_volume_momentum + 0.3 * price_volatility)
    
    return final_alpha
```
