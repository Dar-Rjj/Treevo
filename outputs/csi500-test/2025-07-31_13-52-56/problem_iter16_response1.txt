```python
import pandas as pd

def heuristics_v2(df):
    # 30-Day Price Momentum
    df['Ct_30'] = df['close'].shift(30)
    df['30_day_momentum'] = (df['close'] - df['Ct_30']) / df['Ct_30']
    
    # Volume Shock
    df['Avg_Vt_30'] = df['volume'].rolling(window=30).mean()
    df['volume_shock'] = (df['volume'] > 2 * df['Avg_Vt_30']).astype(int)
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # High-Low Range Momentum
    df['range_momentum'] = df['high_low_range'].diff(periods=30)
    
    # Combine Components for High-Low Range Momentum with Volume Shock
    df['combined_high_low_volume'] = (df['30_day_momentum'] * df['volume_shock']) + df['range_momentum']
    
    # High-to-Low Price Ratio with Volume Trend
    df['high_low_ratio'] = df['high'] / df['low']
    df['vol_ma'] = df['volume'].rolling(window=30).mean()
    df['volume_trend'] = (df['volume'] > df['vol_ma']).astype(int) * 2 - 1  # 1 if volume is above MA, -1 if below
    df['high_low_vol_trend'] = df['high_low_ratio'] * df['volume_trend']
    
    # Volume and Price Momentum
    n_days = 30
    df['volume_momentum'] = df['volume'].rolling(window=n_days).sum()
    
    # True Range
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['close'] - x['high']), abs(x['close'] - x['low'])), axis=1)
    df['price_volatility'] = df['tr'].rolling(window=n_days).std()
    
    # Combine Volume and Price Momentum
    df['combined_vol_price_momentum'] = df['price_volatility'] / df['volume_momentum']
    
    # Integrate Components for Final Alpha Factor
    df['final_alpha_factor'] = df['high_low_vol_trend'] + df['combined_vol_price_momentum']
    
    return df['final_alpha_factor'].dropna()
```
