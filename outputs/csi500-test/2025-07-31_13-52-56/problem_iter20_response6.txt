```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    df['High_Low_Ratio'] = (df['high'] - df['low']) / df['low']
    
    # Evaluate Volume Trend and Momentum
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = np.where(df['volume'] > df['Volume_MA_10'], 1, -1)
    
    # Quantify High-Low Range and Volatility
    df['High_Low_Range'] = (df['high'] - df['low'])
    df['High_Low_Range_Adjusted'] = df['High_Low_Range'] * np.sqrt(df['volume'])
    
    # Calculate Average True Range
    df['True_Range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_10'] = df['True_Range'].rolling(window=10).mean()
    
    # Measure Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=10).std()
    
    # Integrated and Weighted Alpha Factor
    df['Adjusted_High_Low_Ratio'] = df['High_Low_Ratio'] * df['Volume_Trend']
    df['Adjusted_High_Low_Range'] = df['High_Low_Range_Adjusted'] * df['Volume_Trend']
    
    df['V_change'] = df['volume'].pct_change()
    df['Inverted_Volume_Momentum'] = -1 * df['V_change']
    df['Composite_Alpha'] = df['Inverted_Volume_Momentum'] + df['Price_Volatility'] + df['Adjusted_High_Low_Range']
    
    df['Weighted_Adjusted_Volume_Momentum'] = 0.4 * df['Adjusted_High_Low_Range']
    df['Weighted_Inverted_Volume_Momentum'] = 0.3 * df['Inverted_Volume_Momentum']
    df['Final_Alpha'] = (df['Weighted_Adjusted_Volume_Momentum'] + 
                         df['Weighted_Inverted_Volume_Momentum'] + 
                         0.3 * df['Price_Volatility'])
    
    # Combine Adjusted Range, Momentum, and Volume Signals
    df['Momentum_Signal'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Volume_Signal'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Adjusted_High_Low_Range_Signal'] = df['High_Low_Range_Adjusted'] - df['Momentum_Signal']
    df['Final_Signal'] = df['Adjusted_High_Low_Range_Signal'] * df['Volume_Signal']
    
    return df['Final_Alpha']
```
