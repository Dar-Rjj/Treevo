```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Evaluate Volume Trend
    df['volume_10_day_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_10_day_ma']).astype(int) * 2 - 1
    
    # Integrate Components for Adjusted High-to-Low Price Ratio
    df['adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume_trend']
    
    # Calculate 30-day Price Momentum
    df['close_30_days_ago'] = df['close'].shift(30)
    df['30_day_momentum'] = (df['close'] - df['close_30_days_ago']) / df['close_30_days_ago']
    
    # Identify Volume Shock
    df['volume_30_day_ma'] = df['volume'].rolling(window=30).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['volume_30_day_ma']).astype(int)
    
    # Combine 30-day Momentum and Volume Shock
    df['30_day_momentum_with_shock'] = df['30_day_momentum'] * df['volume_shock']
    
    # Detect Volume Spike
    df['volume_5_day_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 1.7 * df['volume_5_day_ma']).astype(int)
    
    # Calculate 10-day Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted High-to-Low Range, 10-day Momentum, and Volume Spike
    df['combined_signal'] = df['adjusted_high_low_ratio'] + df['10_day_momentum']
    df['combined_signal'] = df['combined_signal'] * (1 + 1.5 * df['volume_spike'])
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['30_day_momentum_with_shock'] + df['combined_signal']
    
    return df['alpha_factor']
```
