```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day Price Momentum
    df['close_t'] = df['close']
    df['close_t-30'] = df['close'].shift(30)
    df['momentum_30'] = (df['close_t'] - df['close_t-30']) / df['close_t-30']
    
    # Identify Volume Shock
    df['volume_t'] = df['volume']
    df['avg_volume_30'] = df['volume'].rolling(window=30, min_periods=30).mean().shift(1)
    df['volume_shock'] = (df['volume_t'] > 2 * df['avg_volume_30']).astype(int)
    
    # Combine Momentum and Volume Shock
    df['alpha_factor'] = df['momentum_30'] * df['volume_shock']
    
    return df['alpha_factor'].dropna()
```
