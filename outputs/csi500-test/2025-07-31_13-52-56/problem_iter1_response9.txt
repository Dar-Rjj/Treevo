import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df[['high', 'low', 'close', 'open']].std(axis=1)
    
    # Adjust Intraday Return by Intraday Volatility
    df['intraday_volatility_adjusted'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate Volume Displacement
    df['volume_displacement'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weight by Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    df['weight_by_intraday_range'] = df['intraday_range'] / df['intraday_range'].rolling(window=20).mean()
    
    # Combine Factors
    df['factor'] = df['intraday_volatility_adjusted'] * df['volume_displacement'] * df['weight_by_intraday_range']
    
    # Final Weighting
    df['final_factor'] = df['factor'] / df['factor'].rolling(window=20).mean()
    
    return df['final_factor']
