```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['adjusted_intraday_return'] = df['intraday_return'] / (df['high'] - df['low'])

    # Compute 30-day Price Momentum
    df['30_day_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Analyze Volume and Amount Shocks
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30).mean()).astype(int)
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30).mean()).astype(int)

    # Enhanced Volume and Price Volatility Components
    df['high_low_volume'] = (df['high'] - df['low']) * df['volume']
    df['open_close_change'] = df['close'] - df['open']
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_moving_avg']).astype(int) * 2 - 1

    # Composite Alpha Factors Synthesis
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    df['composite_alpha_1'] = df['volume_momentum'] * (df['high'] / df['low'])
    df['composite_alpha_2'] = -1 * df['volume_momentum']
    df['composite_alpha_3'] = (-1 * df['volume_momentum'] + df['high_low_volume'] * df['volume_momentum'])
    df['composite_alpha_4'] = (df['high'] / df['low']) * df['volume_trend']

    # Weighted and Integrated Alpha Factor
    df['weighted_intraday_adjustment'] = 0.2 * df['adjusted_intraday_return']
    df['weighted_30_day_momentum'] = 0.3 * df['30_day_price_momentum']
    df['weighted_volume_amount_shock'] = 0.1 * df['volume_shock'] * df['amount_shock']
    df['weighted_volume_momentum'] = 0.2 * df['volume_trend']
    df['weighted_inverted_volume_momentum'] = 0.15 * df['composite_alpha_2']
    df['weighted_composite_alpha'] = 0.1 * (df['composite_alpha_1'] + df['composite_alpha_3'] + df['composite_alpha_4'])

    df['alpha_factor'] = (
        df['weighted_intraday_adjustment'] +
        df['weighted_30_day_momentum'] +
        df['weighted_volume_amount_shock'] +
        df['weighted_volume_momentum'] +
        df['weighted_inverted_volume_momentum'] +
        df['weighted_composite_alpha']
    )

    # Incorporate Volume Trend
    df['final_alpha'] = df['alpha_factor'] * df['volume_momentum']

    return df['final_alpha']
```
