import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Return
    intraday_volatility_adjusted = intraday_return / intraday_volatility
    
    # Calculate 30-day Price Momentum
    price_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    avg_volume_30d = df['volume'].rolling(window=30).mean()
    volume_shock = (df['volume'] > 2 * avg_volume_30d).astype(int)
    
    # Identify Amount Shock
    avg_amount_30d = df['amount'].rolling(window=30).mean()
    amount_shock = (df['amount'] > 2 * avg_amount_30d).astype(int)
    
    # Combine Factors
    combined_factor = (intraday_volatility_adjusted * 
                       price_momentum_30d * 
                       volume_shock * 
                       amount_shock)
    
    # Final Factor
    return combined_factor
