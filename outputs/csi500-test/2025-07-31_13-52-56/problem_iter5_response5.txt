```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    momentum_lookback = 50
    df['PriceMomentum'] = df['close'] - df['close'].shift(momentum_lookback)
    
    # Calculate Volume Acceleration
    volume_lookback = 20
    df['VolumeAcceleration'] = df['volume'] - df['volume'].shift(volume_lookback)
    
    # Combine Momentum and Volume
    df['IntermediateFactor'] = df['PriceMomentum'] * df['VolumeAcceleration']
    
    # Calculate High-to-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['AdjustedHighLowRange'] = df['HighLowRange'] * df['volume']
    
    # Detect Volume Spike
    df['Volume5DayMA'] = df['volume'].rolling(window=5).mean()
    df['VolumeSpike'] = (df['volume'] > 1.5 * df['Volume5DayMA']).astype(int)
    
    # Calculate Volume Trend
    volume_trend_window = 10
    df['Volume10DayMA'] = df['volume'].rolling(window=volume_trend_window).mean()
    df['VolumeTrend'] = (df['volume'] > df['Volume10DayMA']).astype(int) * 2 - 1
    
    # Final Combination
    df['FinalFactor'] = (df['AdjustedHighLowRange'] * df['VolumeTrend'] + df['IntermediateFactor']) * (1 + df['VolumeSpike'])
    
    return df['FinalFactor']
```
