```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range = high_to_low_range * df['volume']
    
    # Calculate 10-day Momentum
    momentum_signal = df['close'].diff(10)
    
    # Combine Adjusted Range with Momentum
    factor = adjusted_range + momentum_signal
    
    return factor
```
