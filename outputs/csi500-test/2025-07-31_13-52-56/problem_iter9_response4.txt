import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df[['high', 'low', 'close', 'open']].max(axis=1) - 
                                 df[['high', 'low', 'close', 'open']].min(axis=1)) / df['low']
    
    # Adjust Intraday Return
    df['intraday_vol_adjusted'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate 30-day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30, min_periods=30).mean()).astype(int)
    
    # Identify Amount Shock
    df['amount_shock'] = (df['amount'] > 2 * df['amount'].rolling(window=30, min_periods=30).mean()).astype(int)
    
    # Combine Factors
    df['factor'] = df['intraday_vol_adjusted'] * df['30_day_momentum'] * df['volume_shock'] * df['amount_shock']
    
    # Final Factor
    return df['factor']
