```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    """
    Calculate a novel alpha factor based on volume momentum and price volatility.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    n (int): Number of days to look back for volume momentum and ATR calculation
    
    Returns:
    pd.Series: Alpha factor values indexed by date
    """
    
    # Volume Momentum Indicator
    df['volume_sum_t1'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_sum_tn'] = df['volume'].rolling(window=n+1).sum().shift(1)
    df['volume_momentum'] = (df['volume_sum_tn'] - df['volume_sum_t1']) / df['volume_sum_t1']
    
    # Average True Range Calculation
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], x.shift(1)['close']) - min(x['low'], x.shift(1)['close']), axis=1)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # High-Low Range Ratio
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    
    # Adjusted High-Low Range Ratio
    df['adjusted_high_low_ratio'] = df['high_low_range'] / df['avg_high_low_range']
    df['adjusted_high_low_ratio'] *= np.sign(df['volume_momentum'])
    
    # Final Alpha Factor
    alpha_factor = df['adjusted_high_low_ratio']
    
    return alpha_factor
```
