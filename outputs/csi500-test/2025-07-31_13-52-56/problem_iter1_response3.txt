```python
import pandas as pd

def heuristics_v2(df, n=5):
    """
    Calculate the Volume Momentum and Price Volatility Alpha factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    n (int): Number of days to look back for volume aggregation and ATR calculation.
    
    Returns:
    pd.Series: Series with the calculated alpha factor indexed by date.
    """
    # Calculate Volume Momentum Indicator
    df['agg_volume'] = df['volume'].rolling(window=n).sum()
    df['volume_momentum'] = df['agg_volume'] - df['agg_volume'].shift(n)
    
    # Measure High-Low Range Expansion
    # Calculate Average True Range (ATR)
    df['tr'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['atr'] = df['tr'].rolling(window=n).mean()
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Combine Volume Momentum and Range Expansion
    df['adjusted_high_low_range_ratio'] = df['high_low_range_ratio'] * df['volume_momentum']
    
    # Consider Sign of Volume Momentum
    df['alpha_factor'] = df.apply(lambda row: row['adjusted_high_low_range_ratio'] if row['volume_momentum'] > 0 else -row['adjusted_high_low_range_ratio'], axis=1)
    
    return df['alpha_factor'].dropna()
```
