```python
import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Adjusted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_Range'] = df['High_Low_Range'] * df['volume']
    
    # Calculate 5-period Rolling Average of the Adjusted Range
    df['Rolling_Avg_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=5).mean()
    
    # Compute High-Low Range Momentum
    df['Range_Momentum'] = df['Rolling_Avg_Adjusted_Range'].diff(periods=5)
    
    # Compute Close Price Momentum
    df['Close_Price_Momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Combine Adjusted High-Low Range and Close Price Momentum
    df['Combined_Momentum'] = (df['Range_Momentum'] + df['Close_Price_Momentum']) / 2
    
    # Detect Volume Spike
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Volume_MA_5']).astype(int)
    
    # Detect Volume Shock
    df['Volume_MA_30'] = df['volume'].rolling(window=30).mean()
    df['Volume_Shock'] = (df['volume'] > 2 * df['Volume_MA_30']).astype(int)
    
    # Integrate Momentum, Volume Spike, and Volume Shock
    df['Factor'] = df['Combined_Momentum'] * (1 + df['Volume_Spike'] + df['Volume_Shock'])
    
    return df['Factor']
```
