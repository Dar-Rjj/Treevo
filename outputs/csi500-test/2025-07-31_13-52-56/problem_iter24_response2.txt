```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return and Adjustments
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    intraday_volatility = df['high'] - df['low']
    df['adjusted_intraday_return'] = df['intraday_return'] / intraday_volatility
    
    # Compute 30-day Price Momentum
    df['price_momentum_30d'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Analyze Volume and Amount Shocks
    avg_volume_30d = df['volume'].rolling(window=30).mean()
    avg_amount_30d = df['amount'].rolling(window=30).mean()
    df['volume_shock'] = (df['volume'] > 2 * avg_volume_30d).astype(int)
    df['amount_shock'] = (df['amount'] > 2 * avg_amount_30d).astype(int)
    
    # Enhanced Volume and Price Volatility Components
    df['hl_range_vol_adjusted'] = (df['high'] - df['low']) * df['volume']
    df['cumulative_hl_range_vol_adjusted'] = df['hl_range_vol_adjusted'].rolling(window=30).sum()
    
    df['open_close_change'] = df['close'] - df['open']
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=30).sum()
    
    df['volume_ma_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = (df['volume'] > df['volume_ma_10d']).astype(int) - ((df['volume'] < df['volume_ma_10d']).astype(int))
    
    df['volume_momentum'] = (df['volume'].rolling(window=30).sum() - df['volume'].shift(30).rolling(window=30).sum())
    
    # Composite Alpha Factors Synthesis
    df['adjusted_volume_momentum'] = df['volume_momentum'] * (df['high'] / df['low'])
    df['inverted_volume_momentum'] = -1 * df['volume_momentum']
    df['composite_alpha'] = (df['inverted_volume_momentum'] + df['adjusted_volume_momentum'] + df['hl_range_vol_adjusted'] * df['volume_momentum'])
    df['volume_trend_high_low_ratio'] = (df['high'] / df['low']) * df['volume_trend']
    
    # Weighted and Integrated Alpha Factor
    df['weighted_intraday_return'] = 0.2 * df['adjusted_intraday_return']
    df['weighted_price_momentum_30d'] = 0.3 * df['price_momentum_30d']
    df['weighted_volume_amount_shocks'] = 0.1 * df['volume_shock'] * df['amount_shock']
    df['weighted_adjusted_volume_momentum'] = 0.2 * df['adjusted_volume_momentum']
    df['weighted_inverted_volume_momentum'] = 0.15 * df['inverted_volume_momentum']
    df['weighted_composite_alpha'] = 0.1 * df['composite_alpha']
    
    df['final_alpha'] = (
        df['weighted_intraday_return'] +
        df['weighted_price_momentum_30d'] +
        df['weighted_volume_amount_shocks'] +
        df['weighted_adjusted_volume_momentum'] +
        df['weighted_inverted_volume_momentum'] +
        df['weighted_composite_alpha']
    )
    
    # Incorporate Volume Trend
    df['volume_momentum_30d'] = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    df['final_alpha'] *= df['volume_momentum_30d']
    
    return df['final_alpha']
```
