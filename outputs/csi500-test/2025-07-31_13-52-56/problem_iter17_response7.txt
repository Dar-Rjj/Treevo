```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Range
    adjusted_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_range *= np.sqrt(df['volume'])
    
    # Calculate 30-day Price Momentum
    momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Enhance Price Momentum with High-Low Range
    high_30 = df['high'].rolling(window=30).max()
    low_30 = df['low'].rolling(window=30).min()
    range_30 = high_30 - low_30
    enhanced_momentum = momentum_30 / range_30
    
    # Identify Volume Shock
    avg_volume_30 = df['volume'].rolling(window=30).mean().shift(1)  # exclude current day
    volume_shock = (df['volume'] > 2 * avg_volume_30).astype(int)
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = df[['high', 'low', 'close', 'open']].std(axis=1)
    
    # Adjust Intraday Return by Intraday Volatility
    intraday_adjusted = intraday_return / intraday_volatility
    
    # Combine Adjusted Range, 30-day Momentum, Volume Shock, and Intraday Adjusted
    combined_30 = (adjusted_range + momentum_30) * volume_shock * intraday_adjusted
    
    # Detect Volume Spike
    avg_volume_5 = df['volume'].rolling(window=5).mean().shift(1)  # exclude current day
    volume_spike = (df['volume'] > 1.7 * avg_volume_5).astype(int)
    
    # Calculate 10-day Momentum
    momentum_10 = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted Range, 10-day Momentum, and Volume Spike
    combined_10 = (adjusted_range + momentum_10) * (1 + 1.5 * volume_spike)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = combined_30 + combined_10
    
    return final_alpha_factor
```
