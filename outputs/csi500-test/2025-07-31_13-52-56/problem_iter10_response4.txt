```python
import pandas as pd

def heuristics_v2(df, N=10, n=5):
    # Calculate High-Low Range and Its Volume-Adjusted Version
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Evaluate Cumulative and Dynamic High-Low Measures
    df['Cumulative_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=N).sum()
    df['High_Low_Price_Ratio'] = df['high'] / df['low']
    
    # Analyze Volume Trends and Momentum
    df['Volume_MA'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_MA']).astype(int) * 2 - 1
    df['Aggregated_Volume_t_minus_1'] = df['volume'].rolling(window=n).sum().shift(1)
    df['Aggregated_Volume_t_minus_n'] = df['volume'].rolling(window=n).sum().shift(n)
    df['Volume_Momentum'] = df['Aggregated_Volume_t_minus_1'] - df['Aggregated_Volume_t_minus_n']
    
    # Integrate Open-Close Momentum with Volatility Indicators
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['Cumulative_Open_Close_Price_Change'] = df['Open_Close_Price_Change'].rolling(window=N).sum()
    df['Average_True_Range'] = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Synthesize Components into Alpha Factor
    df['Volume_Trend_High_Low_Ratio'] = df['High_Low_Price_Ratio'] * df['Volume_Trend']
    df['Volume_Momentum_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['volume'] * df['Volume_Momentum']
    df['Momentum_Range_Expansion'] = (df['high'] - df['low']) * (df['Aggregated_Volume_t_minus_1'] - df['Aggregated_Volume_t_minus_n'])
    
    df['Alpha_Factor'] = (df['Volume_Trend_High_Low_Ratio'] + 
                          df['Volume_Momentum_Adjusted_High_Low_Range'] + 
                          df['Momentum_Range_Expansion'])
    
    return df['Alpha_Factor']
```
