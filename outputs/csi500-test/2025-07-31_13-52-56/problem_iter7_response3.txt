```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-Day Price Momentum
    df['30_day_price_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['30_day_avg_volume'] = df['volume'].rolling(window=30).mean().shift(1)  # Exclude current day
    df['volume_shock'] = (df['volume'] > 2 * df['30_day_avg_volume']).astype(int)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum over the last 30 days
    df['high_low_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(30)
    
    # Combine Factors
    df['combined_factor'] = (df['30_day_price_momentum'] * df['volume_shock']) + df['high_low_range_momentum']
    
    return df['combined_factor']
```
