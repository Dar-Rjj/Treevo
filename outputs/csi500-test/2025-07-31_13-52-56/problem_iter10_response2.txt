```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Compute Volume Moving Average (10 days)
    volume_ma = df['volume'].rolling(window=10).mean()
    
    # Determine if Volume is Increasing or Decreasing
    volume_trend = np.where(df['volume'] > volume_ma, 1, -1)
    
    # Combine High-to-Low Price Ratio with Volume Trend
    combined_high_low_volume_trend = high_to_low_ratio * volume_trend
    
    # Measure Volume Momentum Indicator
    volume_momentum = df['volume'].rolling(window=10).sum()
    
    # Compute True Range
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Compute Standard Deviation of True Range for Price Volatility
    price_volatility = true_range.rolling(window=10).std()
    
    # Invert Volume Momentum
    inverted_volume_momentum = 1 / volume_momentum
    
    # Combine Inverted Volume Momentum and Price Volatility
    combined_vol_price_momentum = inverted_volume_momentum + price_volatility
    
    # Adjusted High-Low Range Ratio
    adjusted_high_low_ratio = combined_high_low_volume_trend * np.sign(volume_momentum)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_high_low_ratio + combined_vol_price_momentum
    
    return final_alpha_factor
```
