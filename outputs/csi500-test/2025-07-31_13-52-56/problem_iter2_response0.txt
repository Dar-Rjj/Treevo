```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-Day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['30_day_avg_volume'] = df['volume'].rolling(window=30).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['30_day_avg_volume']).astype(int)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum over a 30-day window
    df['high_low_range_30_day'] = df['high_low_range'].rolling(window=30).apply(
        lambda x: x[-1] - x[0], raw=True
    )
    
    # Combine Factors
    df['alpha_factor'] = (df['30_day_momentum'] * df['volume_shock']) + df['high_low_range_30_day']
    
    return df['alpha_factor']
```
