```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Momentum
    window = 20  # Define the rolling window size
    
    # Compute Rolling Volume Sum
    df['rolling_volume_sum'] = df['volume'].rolling(window=window).sum()
    
    # Compute Log Return of Volume
    df['volume_momentum'] = np.log(df['rolling_volume_sum']) - np.log(df['rolling_volume_sum'].shift(1))
    
    # Calculate Price Volatility
    # Compute True Range
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    
    # Compute Standard Deviation of True Range
    df['price_volatility'] = df['true_range'].rolling(window=window).std()
    
    # Combine Volume Momentum and Price Volatility
    # Invert Volume Momentum
    df['inverted_volume_momentum'] = -df['volume_momentum']
    
    # Add Inverted Volume Momentum to Price Volatility
    df['composite_alpha'] = df['inverted_volume_momentum'] + df['price_volatility']
    
    return df['composite_alpha'].dropna()
```
