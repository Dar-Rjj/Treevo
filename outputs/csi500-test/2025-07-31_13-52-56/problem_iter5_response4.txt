```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df[['high', 'low', 'close', 'open']].std(axis=1))
    
    # Adjust Intraday Return by Intraday Volatility
    df['intraday_volatility_adjusted'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate 30-day Price Momentum
    df['price_momentum_30d'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 2 * df['volume'].rolling(window=30, min_periods=1).mean().shift(1)).astype(int)
    
    # Combine Factors
    df['factor'] = df['intraday_volatility_adjusted'] * df['price_momentum_30d'] * df['volume_shock']
    
    return df['factor']
```
