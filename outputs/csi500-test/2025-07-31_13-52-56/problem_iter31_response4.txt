```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return and Adjusted Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / (df['high'] - df['low'])

    # 20-day Price Momentum
    df['20_day_Price_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Volume and Amount Shocks
    df['30_day_Avg_Volume'] = df['volume'].rolling(window=30).mean()
    df['Volume_Shock'] = (df['volume'] > 1.5 * df['30_day_Avg_Volume']).astype(int)
    
    df['30_day_Avg_Amount'] = df['amount'].rolling(window=30).mean()
    df['Amount_Shock'] = (df['amount'] > 1.5 * df['30_day_Avg_Amount']).astype(int)

    # High-Low Range and Volume-Adjusted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(df['volume'])

    # Volume Trends and Momentum
    df['7_day_Volume_MA'] = df['volume'].rolling(window=7).mean()
    df['Volume_Trend'] = np.where(df['volume'] > df['7_day_Volume_MA'], 1, -1)
    df['Volume_Momentum'] = df['volume'].rolling(window=7).sum() - df['volume'].shift(7).rolling(window=7).sum()

    # Open-Close Momentum with EMA
    df['Open_Close_Price_Change'] = df['close'] - df['open']
    df['7_day_EMA_Close'] = df['close'].ewm(span=7, adjust=False).mean()

    # Combined Alpha Factors
    df['Vol_Adj_High_Low_Range_Vol_Mom'] = df['Volume_Adjusted_High_Low_Range'] * df['Volume_Momentum']
    df['Open_Close_Mom_Vol_Adj_High_Low_Range'] = df['Open_Close_Price_Change'] * df['Volume_Adjusted_High_Low_Range']
    df['Adjusted_Volume_Momentum'] = df['Volume_Momentum'] * (df['high'] / df['low'])
    df['Inverted_Volume_Momentum'] = -1 * df['Volume_Momentum']

    # Weighted and Integrated Alpha
    df['Weighted_Adjusted_Intraday_Return'] = 0.25 * df['Adjusted_Intraday_Return']
    df['Weighted_20_day_Price_Momentum'] = 0.4 * df['20_day_Price_Momentum']
    df['Weighted_Volume_Shock_Amount_Shock'] = 0.1 * df['Volume_Shock'] * df['Amount_Shock']
    df['Weighted_Adjusted_Volume_Momentum'] = 0.15 * df['Adjusted_Volume_Momentum']
    df['Weighted_Inverted_Volume_Momentum'] = 0.05 * df['Inverted_Volume_Momentum']
    df['Weighted_Vol_Adj_High_Low_Range_Vol_Mom'] = 0.05 * df['Vol_Adj_High_Low_Range_Vol_Mom']

    # Final Alpha
    df['Final_Alpha'] = (
        df['Weighted_Adjusted_Intraday_Return'] +
        df['Weighted_20_day_Price_Momentum'] +
        df['Weighted_Volume_Shock_Amount_Shock'] +
        df['Weighted_Adjusted_Volume_Momentum'] +
        df['Weighted_Inverted_Volume_Momentum'] +
        df['Weighted_Vol_Adj_High_Low_Range_Vol_Mom']
    )

    return df['Final_Alpha'].dropna()
```
