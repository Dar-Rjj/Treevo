import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Volume Momentum
    V_sum = df['volume'].rolling(window=n).sum()
    V_change = V_sum - V_sum.shift(n)
    
    # Measure High-Low Range Ratio (H/L)
    H_L_ratio = df['high'] / df['low']
    
    # Adjusted High-Low Price Ratio with Volume Trend
    vol_moving_avg = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] > vol_moving_avg).astype(int) * 2 - 1  # +1 for increasing, -1 for decreasing
    adjusted_H_L_ratio = H_L_ratio * volume_trend
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=n).std()
    
    # Weighted Combined Alpha
    adjusted_volume_momentum = V_change * H_L_ratio
    inverted_volume_momentum = -1 * V_change
    final_alpha = 0.4 * adjusted_volume_momentum + 0.3 * inverted_volume_momentum + 0.3 * price_volatility
    
    return final_alpha
