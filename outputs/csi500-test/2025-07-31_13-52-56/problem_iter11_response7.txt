```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-to-Low Price Ratio
    high_to_low_ratio = df['high'] / df['low']
    
    # Compute Volume Moving Average (10-day)
    volume_ma = df['volume'].rolling(window=10).mean()
    
    # Determine if Volume is Increasing or Decreasing
    volume_trend = (df['volume'] > volume_ma).astype(int) * 2 - 1
    
    # Combine High-to-Low Price Ratio with Volume Trend
    combined_ratio_trend = high_to_low_ratio * volume_trend
    
    # Measure Volume Momentum Indicator
    aggregated_volume = df['volume'].rolling(window=10).sum()
    volume_momentum = aggregated_volume.pct_change()
    
    # Calculate True Range and Standard Deviation of True Range
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    tr_std = true_range.rolling(window=10).std()
    
    # Combine Volume Momentum and Price Volatility
    inverted_vol_momentum = -volume_momentum
    combined_vol_price_vol = inverted_vol_momentum + tr_std
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df[['high', 'low', 'close', 'open']].max(axis=1) - 
                           df[['high', 'low', 'close', 'open']].min(axis=1))
    
    # Adjust Intraday Return by Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Calculate 30-day Price Momentum
    price_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    avg_volume_30d = df['volume'].rolling(window=30).mean().shift(1)
    volume_shock = (df['volume'] > 2 * avg_volume_30d).astype(int)
    
    # Combine Intraday and Long-term Factors
    combined_intraday_longterm = adjusted_intraday_return + price_momentum_30d + volume_shock
    
    # Adjusted High-Low Range Ratio
    adjusted_high_low_ratio = combined_ratio_trend * (volume_momentum > 0).astype(int)
    
    # Combine Adjusted High-Low Range Ratio with Integrated Intraday and Long-term Factors
    final_factor = adjusted_high_low_ratio + combined_intraday_longterm
    
    return final_factor
```
