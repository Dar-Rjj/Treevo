import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10'] = df['return'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the daily return
    df['ma_30'] = df['return'].rolling(window=30).mean()
    
    # Calculate the momentum factor as the difference between the 10-day and 30-day moving averages
    df['momentum'] = df['ma_10'] - df['ma_30']
    
    # Calculate the price range as the difference between the high and low prices
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the 10-day moving average of the price range
    df['price_range_ma_10'] = df['price_range'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the price range
    df['price_range_ma_30'] = df['price_range'].rolling(window=30).mean()
    
    # Calculate the volatility factor as the difference between the 10-day and 30-day moving averages of the price range
    df['volatility'] = df['price_range_ma_10'] - df['price_range_ma_30']
    
    # Calculate the trading intensity as the ratio of volume to the average volume over the past 30 days
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['trading_intensity'] = df['volume'] / df['avg_volume_30']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['momentum'] + df['volatility'] + df['trading_intensity']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
