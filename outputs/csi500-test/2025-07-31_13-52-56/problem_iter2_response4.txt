```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Cumulative High-Low Range over N days
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=n_days).sum()
    
    # Calculate Open-to-Close Price Change
    open_close_price_change = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days
    cumulative_open_close_price_change = open_close_price_change.rolling(window=n_days).sum()
    
    # Calculate High-Low Range Momentum
    high_low_range_momentum = cumulative_high_low_range / cumulative_open_close_price_change
    
    # Calculate the difference between the latest and earliest Volume Adjusted High-Low Range in the window
    diff_volume_adjusted_high_low_range = volume_adjusted_high_low_range - volume_adjusted_high_low_range.shift(n_days-1)
    
    # Final Alpha Factor
    alpha_factor = high_low_range_momentum * diff_volume_adjusted_high_low_range
    
    return alpha_factor
```
