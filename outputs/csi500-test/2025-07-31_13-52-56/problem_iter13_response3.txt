import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on high-low price dynamics and volume trends.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate Enhanced High-to-Low Price Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Evaluate Volume Trend
    window = 10
    df['volume_ma'] = df['volume'].rolling(window=window).mean()
    df['volume_trend'] = (df['volume'] > df['volume_ma']).astype(int) - (df['volume'] < df['volume_ma']).astype(int)
    
    # Quantify High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    n_days = 14
    df['atr'] = df[['high', 'low', 'close']].apply(lambda x: (x['high'] - x['low']), axis=1).rolling(window=n_days).mean()
    df['high_low_range_ratio'] = df['high_low_range'] / df['atr']
    
    # Integrate Components for Alpha Factor
    df['adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume_trend']
    df['volume_momentum_range_expansion'] = df['high_low_range_ratio'] * df['volume_trend']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_high_low_ratio'] + df['volume_momentum_range_expansion']
    
    return df['alpha_factor']
