```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-Day Price Momentum
    df['30_day_momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean().shift(1)
    df['volume_shock'] = (df['volume'] > 2 * df['avg_volume_30']).astype(int)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['volume'] * df['high_low_range']
    
    # Calculate Cumulative High-Low Range over N days (N=30)
    df['cumulative_high_low_range'] = df['volume_adjusted_high_low_range'].rolling(window=30).sum()
    
    # Calculate Open-to-Close Price Change
    df['open_close_change'] = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days (N=30)
    df['cumulative_open_close_change'] = df['open_close_change'].rolling(window=30).sum()
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = df['cumulative_high_low_range'] / df['cumulative_open_close_change']
    
    # Combine Momentum and Volume Shock
    df['combined_momentum_volume_shock'] = df['30_day_momentum'] * df['volume_shock']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (
        df['combined_momentum_volume_shock'] * 
        df['high_low_range_momentum'] * 
        (df['volume_adjusted_high_low_range'] - df['volume_adjusted_high_low_range'].shift(29))
    )
    
    return df['final_alpha_factor']
```
