```python
import pandas as pd

def heuristics_v2(df):
    # 20-Day Price Momentum
    df['Ct'] = df['close']
    df['Ct_20'] = df['close'].shift(20)
    df['20_day_momentum'] = (df['Ct'] - df['Ct_20']) / df['Ct_20']
    
    # Volume Surge Indicator
    df['Vt'] = df['volume']
    df['Med_Vt_20'] = df['volume'].rolling(window=20, min_periods=1).median().shift(1)
    df['volume_surge_indicator'] = (df['Vt'] > 2 * df['Med_Vt_20']).astype(int)
    
    # High-Low Range Volatility
    df['Ht'] = df['high']
    df['Lt'] = df['low']
    df['range'] = df['Ht'] - df['Lt']
    df['range_volatility'] = df['range'].rolling(window=10, min_periods=1).std()
    
    # Combine Factors
    df['alpha_factor'] = (df['20_day_momentum'] * df['volume_surge_indicator']) + df['range_volatility']
    
    return df['alpha_factor']
```
