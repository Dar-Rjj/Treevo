```python
import pandas as pd

def heuristics_v2(df, N=30):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Cumulative High-Low Range over N days
    cumulative_high_low_range = volume_adjusted_high_low_range.rolling(window=N).sum()
    
    # Calculate Open-to-Close Price Change
    open_close_change = df['close'] - df['open']
    
    # Calculate Cumulative Open-to-Close Price Change over N days
    cumulative_open_close_change = open_close_change.rolling(window=N).sum()
    
    # Calculate 30-day Price Momentum
    price_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Identify Volume Shock
    avg_volume_30_days = df['volume'].rolling(window=30).mean().shift(1)  # Exclude day t
    volume_shock = (df['volume'] > 2 * avg_volume_30_days).astype(int)
    
    # Combine Momentum and Volume Shock
    combined_momentum_and_volume_shock = price_momentum * volume_shock
    
    # Final Combined Alpha Factor
    high_low_range_momentum = cumulative_high_low_range / cumulative_open_close_change
    final_factor = high_low_range_momentum * combined_momentum_and_volume_shock
    final_factor *= (volume_adjusted_high_low_range - volume_adjusted_high_low_range.shift(N-1))
    
    return final_factor
```
