```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Adaptive Volatility
    rolling_window = 30
    df['volatility'] = df['log_returns'].rolling(window=rolling_window).std()
    df['volatility'] = df['volatility'].fillna(df['volatility'].mean())  # Fill initial NaN values with the mean
    
    # Moving average of recent returns for adaptive weights
    ma_window = 10
    df['recent_ma'] = df['log_returns'].rolling(window=ma_window).mean()
    df['adaptive_volatility'] = df['volatility'] * (df['recent_ma'] + 1)
    
    # Dynamic Trend Following
    short_term_window = 10
    long_term_window = 50
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()
    df['trend_strength'] = df['long_term_ma'] - df['short_term_ma']
    
    # Momentum-based filter
    roc_period = 10
    df['roc'] = df['close'].pct_change(periods=roc_period) * 100
    df['dynamic_trend'] = df['trend_strength'] * df['roc']
    
    # Volume-Adjusted Momentum
    volume_window = 20
    df['avg_volume'] = df['volume'].rolling(window=volume_window).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    df['volume_adjusted_momentum'] = df['roc'] * df['volume_ratio']
    
    # Relative Strength Index (RSI)
    rsi_period = 14
    def compute_rsi(series, period):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi'] = compute_rsi(df['close'], rsi_period)
    
    # Sentiment Analysis (Assume we have a 'sentiment' column in the DataFrame)
    df['sentiment_factor'] = df['rsi'] * df.get('sentiment', 1.0)  # Using RSI to adjust sentiment
    
    # Final Alpha Factor Calculation
    alpha_factor = (
        df['adaptive_volatility'] +
        df['dynamic_trend'] +
        df['volume_adjusted_momentum'] +
        df['sentiment_factor']
    )
    
    return alpha_factor
```
