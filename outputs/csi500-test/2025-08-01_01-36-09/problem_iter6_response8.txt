```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term EMA (10 days)
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA (50 days)
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate Volume-Weighted Average Return
    daily_returns = df['close'].pct_change()
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    combined_momentum = (short_ema - df['close']) + (long_ema - df['close'])
    adjusted_momentum = combined_momentum * volume_weighted_returns
    
    # Incorporate Volatility
    historical_volatility = df['close'].pct_change().rolling(window=20).std()
    volatility_adjusted_momentum = adjusted_momentum / historical_volatility
    
    # Assess Trend Following Potential
    trend_following = df['close'].rolling(window=50).sum()
    final_factor_value = volatility_adjusted_momentum + 0.1 * trend_following
    
    return final_factor_value
```
