```python
def heuristics_v2(df):
    # Calculate Daily High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute 20-Day Sum of High-Low Differences
    df['volatility_measure'] = df['high_low_diff'].rolling(window=20).sum()
    
    # Calculate 60-Day Close Price Momentum
    df['momentum_60_day'] = df['close'] - df['close'].shift(60)
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['momentum_60_day'] / df['volatility_measure']
    
    return df['adjusted_momentum']
```
