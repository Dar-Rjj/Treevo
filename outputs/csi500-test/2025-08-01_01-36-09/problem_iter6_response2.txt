```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['10_day_avg'] - df['close']
    
    # Calculate Long-Term Price Momentum
    df['50_day_avg'] = df['close'].rolling(window=50).mean()
    df['long_term_momentum'] = df['50_day_avg'] - df['close']
    
    # Adjust by Volume-Weighted Average Return
    df['daily_return'] = (df['close'] / df['open']) - 1
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    
    # Integrate Short-Term and Long-Term Momentum
    df['combined_momentum'] = (df['short_term_momentum'] + df['long_term_momentum']) * df['volume_weighted_return']
    
    # Incorporate Volatility
    df['20_day_returns'] = df['close'].pct_change()
    df['historical_volatility'] = df['20_day_returns'].rolling(window=20).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['historical_volatility']
    
    # Assess Trend Following Potential
    df['50_day_sum_close'] = df['close'].rolling(window=50).sum()
    df['trend_direction'] = np.where(df['50_day_sum_close'] > df['close'], 1, 0)
    df['trend_weight'] = df['trend_direction'].apply(lambda x: 1.5 if x == 1 else 0.5)
    df['final_factor_value'] = df['adjusted_combined_momentum'] * df['trend_weight']
    
    return df['final_factor_value'].dropna()
```
