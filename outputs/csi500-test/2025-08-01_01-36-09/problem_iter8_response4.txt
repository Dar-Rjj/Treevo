import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Price Momentum
    df['10_day_close'] = df['close'].shift(10)
    df['price_momentum'] = df['close'] - df['10_day_close']
    df['smoothed_momentum'] = df['price_momentum'].ewm(span=10, adjust=False).mean()

    # Incorporate Volume Dynamics
    df['volume_variability'] = df['volume'].rolling(window=10).std()
    
    # Evaluate Short-Term Trend
    df['5_day_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['short_term_trend'] = (df['close'] > df['5_day_ema']).astype(int)

    # Combine Momentum and Volume
    df['momentum_volume_factor'] = df['smoothed_momentum'] * df['volume_variability']

    # Integrate Short-Term Price Oscillations
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    average_gain = up.rolling(window=14).mean()
    average_loss = abs(down.rolling(window=14).mean())
    rs = average_gain / average_loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Incorporate Market Microstructure
    df['spread'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['mid_price'] = (df['high'] + df['low']) / 2
    df['order_book_imbalance'] = df['mid_price'] - df['close']
    df['microstructure_factor'] = df['spread'] * df['order_book_imbalance']

    # Determine Final Factor Value
    df['final_factor_value'] = (df['momentum_volume_factor'] * df['short_term_trend'] 
                                + df['rsi'] 
                                + df['microstructure_factor'] 
                                + 0.3)

    return df['final_factor_value'].dropna()
