import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the percentage change in volume
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Calculate the average of high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the relative strength index (RSI) for the closing price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Factor: Price change * Volume change % * RSI
    df['factor'] = df['price_change'] * df['volume_change_pct'] * df['rsi']
    
    return df['factor']
