import pandas as pd

def heuristics_v2(df):
    # Calculate Day-to-Day Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Identify Volatile Price Movements
    df['high_low_spread'] = df['high'] - df['low']
    df['relative_daily_move'] = df['close'] / ((df['high'] + df['low']) / 2)
    
    # Combine Volume and Price Movements
    df['vol_price_interaction'] = df['volume_change'] * df['relative_daily_move']
    
    # Assess Trend Following Potential
    df['adj_close'] = df['close']  # Assuming no dividends, you can adjust for dividends if needed
    df['5d_moving_sum'] = df['adj_close'].rolling(window=5).sum()
    df['trend_following'] = df['5d_moving_sum'] / df['adj_close'].rolling(window=5).mean()
    
    # Determine Factor Value
    weight_trend = 0.5  # You can adjust this weight based on your preference
    df['factor_value'] = (weight_trend * df['trend_following']) + df['vol_price_interaction']
    
    return df['factor_value']
