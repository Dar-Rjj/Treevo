```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Close Price
    df['weighted_close'] = df['close'] * df['volume']
    
    # Calculate Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Volume and Amount Interaction
    df['volume_amount_interaction'] = df['volume'] * df['amount']
    
    # Combine Factors
    df['factor'] = (df['momentum'] * df['volume_amount_interaction']) / df['weighted_close']
    
    return df['factor']
```
