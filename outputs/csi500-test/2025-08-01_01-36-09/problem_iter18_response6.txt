import pandas as pd

def heuristics_v2(df):
    # Calculate Granular Short-Term Price Momentum
    df['5_day_avg'] = df['close'].rolling(window=5).mean()
    df['granular_short_momentum'] = df['close'] - df['5_day_avg']
    
    # Calculate Short-Term Price Momentum
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    df['short_momentum'] = df['close'] - df['10_day_avg']
    
    # Calculate Medium-Term Price Momentum
    df['30_day_avg'] = df['close'].rolling(window=30).mean()
    df['medium_momentum'] = df['close'] - df['30_day_avg']
    
    # Calculate Long-Term Price Momentum
    df['50_day_avg'] = df['close'].rolling(window=50).mean()
    df['long_momentum'] = df['close'] - df['50_day_avg']
    
    # Combine Multi-Period Momenta
    df['combined_momentum'] = df['granular_short_momentum'] + df['short_momentum'] + df['medium_momentum'] + df['long_momentum']
    
    # Calculate Volume-Weighted Average Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).sum() / df['volume'].sum()
    
    # Adjust Combined Momentum by Volume-Weighted Average Return
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_weighted_return']
    
    # Assess Trend Following Potential
    df['trend_following'] = 1.0 * (df['50_day_avg'] > df['close'])
    df['trend_following'] = df['trend_following'].replace(0, 0.5)
    
    # Determine Final Factor Value
    df['final_factor_value'] = df['adjusted_combined_momentum'] + df['trend_following']
    
    # Additional Enhancements
    # Calculate Short-Term Volatility
    df['10_day_volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Medium-Term Volatility
    df['30_day_volatility'] = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Calculate Long-Term Volatility
    df['50_day_volatility'] = (df['high'] - df['low']).rolling(window=50).mean()
    
    # Combine Multi-Period Volatilities
    df['combined_volatility'] = df['10_day_volatility'] + df['30_day_volatility'] + df['50_day_volatility']
    
    # Adjust Combined Momentum by Combined Volatility
    df['new_adjusted_combined_momentum'] = df['adjusted_combined_momentum'] / df['combined_volatility']
    
    # Re-evaluate Final Factor Value
    df['revalued_final_factor_value'] = df['new_adjusted_combined_momentum'] + df['trend_following']
    
    # Consider Volume Trends
    df['10_day_volume_avg'] = df['volume'].rolling(window=10).mean()
    df['30_day_volume_avg'] = df['volume'].rolling(window=30).mean()
    df['50_day_volume_avg'] = df['volume'].rolling(window=50).mean()
    
    # Combine Volume Trends
    df['combined_volume_trends'] = df['10_day_volume_avg'] + df['30_day_volume_avg'] + df['50_day_volume_avg']
    
    # Adjust Final Factor Value by Volume Trends
    df['final_factor_value'] = df['revalued_final_factor_value'] * df['combined_volume_trends']
    
    return df['final_factor_value']
