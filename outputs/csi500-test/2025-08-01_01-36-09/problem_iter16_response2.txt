```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_mom = df['close'].rolling(window=10).mean() - df['close']
    
    # Calculate Medium-Term Price Momentum
    med_mom = df['close'].rolling(window=30).mean() - df['close']
    
    # Calculate Long-Term Price Momentum
    long_mom = df['close'].rolling(window=50).mean() - df['close']
    
    # Combine Multi-Period Momenta with dynamic weights
    combined_mom = 0.4 * short_mom + 0.3 * med_mom + 0.3 * long_mom
    
    # Calculate Volume-Weighted Average Return
    daily_returns = (df['close'] / df['open']) - 1
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Adjust Combined Momentum by Volume-Weighted Average Return
    adjusted_mom = combined_mom * volume_weighted_returns
    
    # Assess Trend Following Potential
    ma_50 = df['close'].rolling(window=50).mean()
    trend_following = (ma_50 > df['close']).astype(float) * 1 + (ma_50 <= df['close']).astype(float) * 0.5
    
    # Integrate Trend Following into Momentum
    integrated_mom = adjusted_mom * trend_following
    
    # Calculate Short-Term Volatility
    short_vol = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Medium-Term Volatility
    med_vol = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Calculate Long-Term Volatility
    long_vol = (df['high'] - df['low']).rolling(window=50).mean()
    
    # Combine Multi-Period Volatilities
    combined_vol = short_vol + med_vol + long_vol
    
    # Adjust Final Factor by Combined Volatility
    final_factor = integrated_mom / combined_vol
    
    return final_factor
```
