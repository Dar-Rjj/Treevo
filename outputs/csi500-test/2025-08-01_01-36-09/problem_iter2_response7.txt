```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_diff'] = df['close'] - df['close'].shift(14)
    
    # Smooth the Difference using a 14-day Simple Moving Average
    df['smoothed_momentum'] = df['close_diff'].rolling(window=14).mean()
    
    # Incorporate Volume Dynamics
    df['volume_variability'] = df['volume'].rolling(window=14).std()
    
    # Combine Momentum and Volume
    df['VAMO'] = df['smoothed_momentum'] * df['volume_variability']
    
    return df['VAMO']
```
