```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adaptive Volatility
    window = 30
    df['volatility'] = df['log_returns'].rolling(window=window).std()
    recent_returns = df['log_returns'].rolling(window=10).mean()
    df['adaptive_weights'] = recent_returns.rolling(window=5).mean() / df['volatility']
    
    # Dynamic Trend Following
    short_term_window = 10
    long_term_window = 50
    df['short_mavg'] = df['close'].rolling(window=short_term_window).mean()
    df['long_mavg'] = df['close'].rolling(window=long_term_window).mean()
    df['trend_strength'] = (df['long_mavg'] - df['short_mavg']) / df['volatility']
    
    # Momentum-based filter (Rate of Change)
    roc_window = 10
    df['momentum_roc'] = df['close'].pct_change(periods=roc_window)
    df['volume_adjusted_momentum'] = df['momentum_roc'] * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Relative Strength Index (RSI)
    def rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi'] = rsi(df['close'])
    df['rsi_signal'] = np.where(df['rsi'] > 70, -1, np.where(df['rsi'] < 30, 1, 0))
    
    # Combine factors
    df['alpha_factor'] = (
        df['adaptive_weights'] +
        df['trend_strength'] +
        df['volume_adjusted_momentum'] +
        df['rsi_signal']
    )
    
    # Normalize the alpha factor
    df['alpha_factor'] = (df['alpha_factor'] - df['alpha_factor'].mean()) / df['alpha_factor'].std()
    
    return df['alpha_factor']
```
