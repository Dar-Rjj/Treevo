```python
def heuristics_v2(data):
    # Calculate Daily High-Low Price Difference
    data['high_low_diff'] = data['high'] - data['low']
    
    # Compute 20-Day Sum of High-Low Differences as a volatility measure
    data['volatility_measure'] = data['high_low_diff'].rolling(window=20).sum()
    
    # Calculate 20-Day Close Price Momentum
    data['momentum_20_day'] = data['close'] - data['close'].shift(20)
    
    # Adjust Momentum by Volatility
    data['adjusted_momentum'] = data['momentum_20_day'] / data['volatility_measure']
    
    return data['adjusted_momentum']
```
