```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    short_term_momentum = df['close'] - df['10_day_avg']
    
    # Calculate Long-Term Price Momentum
    df['50_day_avg'] = df['close'].rolling(window=50).mean()
    long_term_momentum = df['close'] - df['50_day_avg']
    
    # Adjust by Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    combined_momentum = (short_term_momentum + long_term_momentum) * volume_weighted_returns
    
    # Assess Trend Following Potential
    df['50_day_sum_close'] = df['close'].rolling(window=50).sum()
    trend_direction = df['50_day_sum_close'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Determine Final Factor Value
    final_factor = combined_momentum * trend_direction
    
    return final_factor
```
