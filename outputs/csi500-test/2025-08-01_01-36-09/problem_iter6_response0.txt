import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Price Momentum
    df['Close_10_days_ago'] = df['close'].shift(10)
    df['Price_Diff'] = df['close'] - df['Close_10_days_ago']
    df['Smoothed_Momentum'] = df['Price_Diff'].ewm(span=10).mean()

    # Incorporate Volume Dynamics
    df['Volume_Variability'] = df['volume'].rolling(window=10).std()

    # Evaluate Short-Term Trend
    df['5_day_EMA'] = df['close'].ewm(span=5).mean()
    df['Short_Term_Trend'] = (df['close'] > df['5_day_EMA']).astype(int)

    # Combine Momentum and Volume
    df['Combined_Factor'] = df['Smoothed_Momentum'] * df['Volume_Variability']

    # Integrate Short-Term Price Oscillations (14-day RSI)
    def rsi(series, period):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=period, min_periods=1).mean()
        avg_loss = loss.rolling(window=period, min_periods=1).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    df['RSI_14'] = rsi(df['close'], 14)

    # Determine Final Factor Value
    df['Final_Factor'] = (df['Combined_Factor'] * df['Short_Term_Trend'] + df['RSI_14']) + 0.3

    # Return the final factor values
    return df['Final_Factor'].dropna()
