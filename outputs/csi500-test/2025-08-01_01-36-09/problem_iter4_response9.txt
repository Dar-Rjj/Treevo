```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Price Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Compute 20-Day Sum of High-Low Differences as a volatility measure
    df['Volatility_20'] = df['High_Low_Diff'].rolling(window=20).sum()
    
    # Calculate 120-Day Close Price Momentum
    df['Momentum_120'] = df['close'] - df['close'].shift(120)
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Momentum_120'] / df['Volatility_20']
    
    # Compute daily returns using Close and Open prices
    df['Daily_Return'] = (df['close'] / df['open']) - 1
    
    # Multiply each daily return by the respective volume
    df['Volume_Weighted_Return'] = df['Daily_Return'] * df['volume']
    
    # Compute the sum of volume-weighted returns over the period
    total_volume = df['volume'].rolling(window=120).sum()
    df['Volume_Weighted_Average_Return'] = df['Volume_Weighted_Return'].rolling(window=120).sum() / total_volume
    
    # Evaluate Long-Term Direction
    df['Trend_Following'] = df['close'].rolling(window=120).sum()
    
    # Determine Final Factor Value
    # Apply Weight to Trend Following Component (weight = 0.5 for example)
    weight_trend_following = 0.5
    df['Factor_Value'] = (df['Trend_Following'] * weight_trend_following + 
                          df['Adjusted_Momentum'] * (1 - weight_trend_following)) * df['Volume_Weighted_Average_Return']
    
    # Return the final factor values
    return df['Factor_Value']
```
