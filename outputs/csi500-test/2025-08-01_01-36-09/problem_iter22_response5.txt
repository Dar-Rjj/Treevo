```python
import pandas as pd
import numpy as np
from arch import arch_model
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.tsa.stattools import acf, pacf
from sklearn.preprocessing import MinMaxScaler

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Estimate time-varying volatility using GARCH(1,1) model
    garch_model = arch_model(df['log_returns'].dropna(), vol='Garch', p=1, q=1)
    garch_results = garch_model.fit(disp='off')
    df['garch_volatility'] = garch_results.conditional_volatility
    
    # Fit an ARIMA model to the log returns
    log_returns = df['log_returns'].dropna()
    p = 1 if len(log_returns) > 1 and acf(log_returns, nlags=1)[1] > 0.5 else 0
    q = 1 if len(log_returns) > 1 and pacf(log_returns, nlags=1)[1] > 0.5 else 0
    d = 1 if len(log_returns) > 1 and np.abs(np.diff(log_returns)).mean() > 0.01 else 0
    arima_model = ARIMA(log_returns, order=(p, d, q))
    arima_results = arima_model.fit()
    df['arima_forecast'] = arima_results.predict(start=len(log_returns), end=len(log_returns))
    
    # Calculate short-term and long-term EMAs
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['ema_trend'] = (df['ema_12'] > df['ema_26']).astype(int) - (df['ema_12'] < df['ema_26']).astype(int)
    
    # Calculate TEMA for smoother trend identification
    df['tema'] = df['close'].ewm(span=30, adjust=False).mean().ewm(span=30, adjust=False).mean().ewm(span=30, adjust=False).mean()
    
    # Calculate 10-day ROC
    df['roc_10'] = df['close'].pct_change(10)
    # Adjust ROC by multiplying with volume ratio
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=50).mean()
    df['roc_adjusted'] = df['roc_10'] * df['volume_ratio']
    
    # Normalize sentiment scores
    scaler = MinMaxScaler(feature_range=(-1, 1))
    df['sentiment_score_normalized'] = scaler.fit_transform(df[['sentiment_score']])
    
    # Add sentiment score to the overall alpha factor
    df['alpha_factor'] = df['garch_volatility'] + df['arima_forecast'] + df['ema_trend'] + df['tema'] + df['roc_adjusted'] + df['sentiment_score_normalized']
    
    # Adjust Alpha Factors Based on Macroeconomic Conditions
    df['alpha_factor'] = df['alpha_factor'] * (1 + df['gdp_growth_rate'] * 0.5 - df['unemployment_rate'] * 0.3 + df['inflation_rate'] * 0.2)
    
    # Adjust Alpha Factor Based on Sector-Specific Indicators
    df.loc[df['sector'] == 'Technology', 'alpha_factor'] += df['rd_expenditure'] * 0.1
    df.loc[df['sector'] == 'Financials', 'alpha_factor'] += (df['dividend_yield'] + df['pe_ratio']) * 0.1
    df.loc[df['sector'] == 'Energy', 'alpha_factor'] += df['crude_oil_price'] * 0.1
    
    return df['alpha_factor']
```
