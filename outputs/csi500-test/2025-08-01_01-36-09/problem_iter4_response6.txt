```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_diff'] = df['close'].diff(periods=14)
    
    # Smooth the Difference
    df['smoothed_momentum'] = df['close_diff'].rolling(window=14).mean()
    
    # Incorporate Volume Dynamics
    df['volume_variability'] = df['volume'].rolling(window=14).std()
    
    # Adjust for Volume Trend
    df['volume_sma'] = df['volume'].rolling(window=14).mean()
    df['volume_adjusted_variability'] = df['volume_variability'] / df['volume_sma']
    
    # Combine Momentum and Volume
    df['VASM'] = df['smoothed_momentum'] * df['volume_adjusted_variability']
    
    return df['VASM']
```
