import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Historical Volatility (Standard Deviation of log returns)
    window = 30  # Rolling window size
    df['historical_volatility'] = df['log_returns'].rolling(window=window).std()
    
    # Dynamic Weights (SMA of past volatilities)
    df['volatility_sma'] = df['historical_volatility'].rolling(window=window).mean()
    df['dynamic_weights'] = df['historical_volatility'] / df['volatility_sma']
    
    # Momentum (Difference between current and past prices)
    momentum_window = 10  # Momentum window size
    df['momentum'] = df['close'].pct_change(periods=momentum_window)
    
    # Volume-Adjusted Momentum
    average_volume = df['volume'].rolling(window=momentum_window).mean()
    df['volume_adjusted_momentum'] = df['momentum'] * (df['volume'] / average_volume)
    
    # Sentiment (Placeholder for sentiment scores, assume it's a column in the DataFrame)
    # df['sentiment_score'] = ...  # Placeholder for actual sentiment score
    # df['volume_adjusted_momentum'] *= df['sentiment_score']
    
    # Dynamic Trend Following
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=50).mean()
    df['trend_signal'] = (short_term_ma > long_term_ma).astype(int) - (short_term_ma < long_term_ma).astype(int)
    
    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Adjust trend signal based on RSI
    df['trend_signal'] = df['trend_signal'].where((df['rsi'] < 70) & (df['rsi'] > 30), 0)
    
    # Integrate Macroeconomic Data (Placeholder for macroeconomic indicators)
    # df['gdp_growth'] = ...  # Placeholder for GDP growth
    # df['inflation_rate'] = ...  # Placeholder for inflation rate
    # df['interest_rate'] = ...  # Placeholder for interest rate
    
    # Adjust alpha factors based on macroeconomic indicators
    # df['adjusted_trend_signal'] = df['trend_signal'] * (1 + df['gdp_growth'] - df['inflation_rate'] + df['interest_rate'])
    
    # Combine all alpha factors
    df['alpha_factor'] = (df['dynamic_weights'] + df['volume_adjusted_momentum'] + df['trend_signal']) / 3
    
    return df['alpha_factor']
