import pandas as pd

def heuristics_v2(df):
    # Calculate the mid-price
    df['mid_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the money flow ratio
    df['money_flow'] = df['close'] * df['volume']
    df['money_flow_ratio'] = df['money_flow'] / df['amount']
    
    # Calculate the relative strength index (RSI) based on the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the weighted average of the RSI and money flow ratio
    df['alpha_factor'] = 0.6 * df['rsi'] + 0.4 * df['money_flow_ratio']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
