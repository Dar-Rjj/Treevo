```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Price Momentum
    df['Close_10_days_ago'] = df['close'].shift(10)
    df['Price_Momentum'] = df['close'] - df['Close_10_days_ago']
    df['Smoothed_Momentum'] = df['Price_Momentum'].ewm(span=10).mean()
    
    # Incorporate Volume Dynamics
    df['Volume_Variability'] = df['volume'].rolling(window=10).std()
    
    # Evaluate Short-Term Trend
    df['5_day_EMA'] = df['close'].ewm(span=5).mean()
    df['Short_Term_Trend'] = (df['close'] > df['5_day_EMA']).astype(int)
    
    # Combine Momentum and Volume
    df['Combined_Factor'] = df['Smoothed_Momentum'] * df['Volume_Variability']
    
    # Integrate Short-Term Price Oscillations
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['14_day_RSI'] = 100 - (100 / (1 + rs))
    
    # Determine Final Factor Value
    df['Final_Factor'] = (df['Combined_Factor'] * df['Short_Term_Trend']) + df['14_day_RSI'] + 0.3
    
    # Enhance Responsiveness
    df['2_day_ROC'] = (df['close'] / df['close'].shift(2)) - 1
    df['Enhanced_Final_Factor'] = df['Final_Factor'] * df['2_day_ROC']
    
    return df['Enhanced_Final_Factor']
```
