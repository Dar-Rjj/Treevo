```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Incorporate Volume Weighting
    volume_weighted_return = close_open_return * df['volume']
    
    # Merge with High-Low Range Effect
    momentum_effect = volume_weighted_return + high_low_range
    
    # Aggregate over Short-term Window (e.g., 5-day moving average)
    short_term_momentum = momentum_effect.rolling(window=5).mean()
    
    # Subtract from Long-term Trend (e.g., 20-day moving average)
    long_term_trend = momentum_effect.rolling(window=20).mean()
    final_momentum_indicator = short_term_momentum - long_term_trend
    
    return final_momentum_indicator
```
