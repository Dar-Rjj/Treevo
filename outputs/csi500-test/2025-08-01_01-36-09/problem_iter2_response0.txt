```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute 20-Day Sum of High-Low Differences
    df['volatility_measure'] = df['high_low_diff'].rolling(window=20).sum()
    
    # Calculate 20-Day Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['close_momentum'] / df['volatility_measure']
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Volume-Weighted Average Return
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Final Factor Calculation
    df['factor'] = df['adjusted_momentum'] * df['volume_weighted_return']
    
    return df['factor'].dropna()
```
