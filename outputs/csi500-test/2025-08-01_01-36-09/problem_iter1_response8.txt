```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff(1)
    
    # Compute Volume Adjusted Factor
    volume_adjusted_factor = df['volume'] * price_momentum
    
    # Normalize by High-Low Range
    factor_values = volume_adjusted_factor / high_low_range
    
    return factor_values
```
