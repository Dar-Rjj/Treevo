import pandas as pd

def heuristics_v2(df):
    # Short-Term Price Momentum
    short_term_days = 5
    short_term_avg = df['close'].rolling(window=short_term_days).mean()
    short_term_momentum = short_term_avg - df['close']
    
    # Long-Term Price Momentum
    long_term_days = 30
    long_term_avg = df['close'].rolling(window=long_term_days).mean()
    long_term_momentum = long_term_avg - df['close']
    
    # Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Combine Short-Term and Long-Term Momentum
    combined_momentum = short_term_momentum + long_term_momentum
    
    # Adjust by Volume-Weighted Average Return
    adjusted_momentum = combined_momentum * volume_weighted_returns
    
    # Short-Term Volatility
    short_term_volatility = df['close'].rolling(window=short_term_days).std()
    
    # Long-Term Volatility
    long_term_volatility = df['close'].rolling(window=long_term_days).std()
    
    # Assess Trend Following Potential
    trend_50_day_sum = df['close'].rolling(window=50).sum()
    trend_component = trend_50_day_sum.diff().fillna(0)
    
    # Final Factor Value
    final_factor = adjusted_momentum + trend_component
    
    # Market Sentiment Score
    # Simple sentiment score based on the difference between high and low prices
    sentiment_score = (df['high'] - df['low']) / df['close']
    final_factor = final_factor * sentiment_score
    
    return final_factor
