```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term, Medium-Term, and Long-Term Price Momentum
    short_momentum = df['close'].rolling(window=10).mean() - df['close']
    medium_momentum = df['close'].rolling(window=30).mean() - df['close']
    long_momentum = df['close'].rolling(window=50).mean() - df['close']

    # Combine Multi-Period Momenta
    combined_momentum = short_momentum + medium_momentum + long_momentum

    # Calculate Volume-Weighted Average Return
    daily_return = (df['close'] / df['open']) - 1
    volume_weighted_return = (daily_return * df['volume']).sum() / df['volume'].sum()

    # Adjust Combined Momentum by Volume-Weighted Average Return
    adjusted_momentum = combined_momentum * volume_weighted_return

    # Assess Trend Following Potential
    trend_weight = 1.0 if df['close'].rolling(window=50).mean().iloc[-1] > df['close'].iloc[-1] else 0.5
    trend_component = adjusted_momentum * trend_weight

    # Calculate Short-Term, Medium-Term, and Long-Term Volatility
    short_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    medium_volatility = (df['high'] - df['low']).rolling(window=30).mean()
    long_volatility = (df['high'] - df['low']).rolling(window=50).mean()

    # Combine Multi-Period Volatilities
    combined_volatility = short_volatility + medium_volatility + long_volatility

    # Adjust Combined Momentum by Combined Volatility
    volatility_adjusted_momentum = adjusted_momentum / combined_volatility

    # Re-evaluate Final Factor Value with Refined Trend Weights
    refined_trend_weight = 1.2 if df['close'].rolling(window=50).mean().iloc[-1] > df['close'].iloc[-1] else 0.8
    final_factor = volatility_adjusted_momentum + (trend_component * refined_trend_weight)

    return final_factor
```
