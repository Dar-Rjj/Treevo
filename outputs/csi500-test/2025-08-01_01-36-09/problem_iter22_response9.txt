import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Calculate Historical Volatility
    rolling_window = 20
    df['historical_volatility'] = df['log_returns'].rolling(window=rolling_window).std() * np.sqrt(252)
    
    # Incorporate Dynamic Weights
    dynamic_weights_window = 10
    df['sma_volatility'] = df['historical_volatility'].rolling(window=dynamic_weights_window).mean()
    df['dynamic_weights'] = (df['sma_volatility'] / df['historical_volatility']).shift(1)
    
    # Combine with Momentum
    momentum_window = 10
    df['momentum'] = df['close'].pct_change(momentum_window)
    df['adjusted_momentum'] = df['momentum'] * df['dynamic_weights']
    
    # Create a Volume-Adjusted Momentum Alpha Factor
    average_volume_window = 20
    df['average_volume'] = df['volume'].rolling(window=average_volume_window).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    df['volume_adjusted_momentum'] = df['momentum'] * df['volume_ratio']
    
    # Construct a Dynamic Trend Following Alpha Factor
    short_term_ma_window = 50
    long_term_ma_window = 200
    df['short_term_ma'] = df['close'].rolling(window=short_term_ma_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_ma_window).mean()
    df['trend_signal'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Enhance with RSI
    rsi_window = 14
    df['delta'] = df['close'].diff()
    gain = df['delta'].mask(df['delta'] < 0, 0)
    loss = -df['delta'].mask(df['delta'] > 0, 0)
    avg_gain = gain.rolling(window=rsi_window).mean()
    avg_loss = loss.rolling(window=rsi_window).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    df['rsi_adjusted_trend'] = df['trend_signal'] * (1 - df['rsi'] / 100)
    
    # Integrate Macroeconomic Data
    # Example: Assume we have macroeconomic data in a separate DataFrame
    # macro_df = pd.read_csv('macro_data.csv')
    # df = df.join(macro_df, on='date')
    # df['macro_adjusted_alpha'] = df['alpha'] * (1 + df['gdp_growth_rate'])
    
    # Apply Machine Learning for Weight Adjustments
    # Example: Assume we have a trained model
    # from sklearn.ensemble import RandomForestRegressor
    # model = RandomForestRegressor()
    # X_train, y_train = prepare_data(df)
    # model.fit(X_train, y_train)
    # df['predicted_returns'] = model.predict(df[['alpha']])
    # df['final_alpha'] = df['alpha'] * df['predicted_returns']
    
    # For simplicity, we combine the factors without machine learning
    df['final_alpha'] = (df['adjusted_momentum'] + df['volume_adjusted_momentum'] + df['rsi_adjusted_trend']) / 3
    
    return df['final_alpha']
