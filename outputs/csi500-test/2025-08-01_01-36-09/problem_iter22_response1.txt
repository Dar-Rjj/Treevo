import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average true range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the price momentum
    df['momentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate the accumulation distribution line
    df['mfm'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['mfv'] = df['mfm'] * df['volume']
    df['adl'] = df['mfv'].cumsum()
    
    # Alpha factor: weighted combination of the above features
    df['alpha_factor'] = (df['momentum'] + 
                          (1 / df['avg_true_range']) +
                          df['adl'].pct_change()) / 3
    
    return df['alpha_factor']
