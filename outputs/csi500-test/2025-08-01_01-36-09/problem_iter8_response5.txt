```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute daily returns using Close and Open prices
    daily_returns = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Average Return
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    short_momentum = short_ema - df['close']
    long_momentum = long_ema - df['close']
    combined_momentum = short_momentum + long_momentum
    
    # Adjust Combined Momentum by Volume-Weighted Average Return
    adjusted_momentum = combined_momentum * volume_weighted_returns
    
    # Historical Volatility
    historical_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Normalize the adjusted momentum by historical volatility
    normalized_momentum = adjusted_momentum / historical_volatility
    
    # Evaluate Long-Term Trend
    moving_sum = df['close'].rolling(window=50).sum()
    trend = np.where(moving_sum > moving_sum.shift(1), 1, -1)
    
    # Determine Final Factor Value with dynamic weighting based on the long-term trend
    final_factor = normalized_momentum * trend
    
    return final_factor
```
