import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return using Close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Smoothed Momentum
    window = 20
    df['sum_daily_return'] = df['daily_return'].rolling(window=window).sum()
    df['min_daily_return'] = df['daily_return'].rolling(window=window).min()
    df['smoothed_momentum'] = (df['sum_daily_return'] - df['min_daily_return']).ewm(span=window, adjust=False).mean()
    
    # Calculate Daily Volatility using High and Low prices
    df['true_range'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    
    # Aggregate Daily Volatility Over a Window
    df['aggregated_volatility'] = df['true_range'].rolling(window=window).sum() / window
    
    # Combine Momentum and Volatility
    df['combined_factor'] = df['smoothed_momentum'] / df['aggregated_volatility']
    df['long_term_avg'] = df['combined_factor'].expanding(min_periods=window).mean()
    df['alpha_factor'] = df['combined_factor'] - df['long_term_avg']
    
    return df['alpha_factor']
