```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['close'] - df['10_day_avg_close']
    
    # Calculate Long-Term Price Momentum
    df['50_day_avg_close'] = df['close'].rolling(window=50).mean()
    df['long_term_momentum'] = df['close'] - df['50_day_avg_close']
    
    # Adjust by Volume-Weighted Average Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    
    # Integrate Short-Term and Long-Term Momentum
    df['combined_momentum'] = (df['short_term_momentum'] + df['long_term_momentum']) * df['volume_weighted_return']
    
    # Assess Trend Following Potential
    df['50_day_sum_close'] = df['close'].rolling(window=50).sum()
    trend_following_weight = 0.5  # Can be adjusted or optimized
    df['trend_following_component'] = df['50_day_sum_close'] * trend_following_weight
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] + df['trend_following_component']
    
    return df['alpha_factor']
```
