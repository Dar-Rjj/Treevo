```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the average of the open and close prices
    df['avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate the ratio of amount to volume
    df['amount_to_vol'] = df['amount'] / df['volume']
    
    # Compute a weighted score based on various metrics
    # This is a simple example; actual weights may vary based on backtesting
    df['score'] = 0.4 * df['pct_change_close'] + 0.3 * (df['close'] - df['open']) / df['open'] + 0.3 * df['amount_to_vol']
    
    # Normalize the score
    df['score'] = (df['score'] - df['score'].mean()) / df['score'].std()
    
    return df['score']
```
