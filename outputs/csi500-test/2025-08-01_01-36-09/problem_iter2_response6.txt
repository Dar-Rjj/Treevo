import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Smoothed Momentum
    window = 20
    df['adj_sum_returns'] = df['daily_return'].rolling(window=window).sum() - df['daily_return'].rolling(window=window).min()
    df['smoothed_momentum'] = df['adj_sum_returns'].ewm(span=window, adjust=False).mean()
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Aggregate True Range Over a Window
    df['avg_true_range'] = df['true_range'].rolling(window=window).sum() / window
    
    # Combine Momentum and Volatility
    df['mom_vol_ratio'] = df['smoothed_momentum'] / df['avg_true_range']
    long_term_window = 60
    df['long_term_ema'] = df['mom_vol_ratio'].ewm(span=long_term_window, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['mom_vol_ratio'] - df['long_term_ema']
    
    return df['alpha_factor'].dropna()
