```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['close'] - df['10_day_avg_close']
    
    # Calculate Long-Term Price Momentum
    df['50_day_avg_close'] = df['close'].rolling(window=50).mean()
    df['long_term_momentum'] = df['close'] - df['50_day_avg_close']
    
    # Adjust by Volume-Weighted Average Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    total_volume = df['volume'].sum()
    volume_weighted_avg_return = df['volume_weighted_return'].sum() / total_volume
    
    # Integrate Short-Term and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['adjusted_combined_momentum'] = df['combined_momentum'] * volume_weighted_avg_return
    
    # Assess Trend Following Potential
    df['50_day_sum_close'] = df['close'].rolling(window=50).sum()
    trend_following_component = df['50_day_sum_close'] / 50
    weight_trend_following = 0.5  # This can be adjusted based on the desired weight
    final_factor_value = df['adjusted_combined_momentum'] + (weight_trend_following * trend_following_component)
    
    return final_factor_value.dropna()
```
