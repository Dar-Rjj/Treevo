```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df, sector_df, interest_rates, inflation_rates):
    # Calculate Rolling Standard Deviation (Volatility)
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)

    # Calculate Exponential Moving Averages (EMA)
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()

    # Define Trend Strength
    df['trend_strength'] = df['ema_10'] - df['ema_50']

    # Normalize Trend Strength by Volatility
    df['composite_trend_volatility'] = df['trend_strength'] / df['volatility']

    # Volume-Adjusted Momentum
    df['momentum'] = df['close'].pct_change(20)
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_momentum'] = df['momentum'] * df['average_volume']

    # Sentiment Analysis using RSI
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Define Sentiment
    df['sentiment_factor'] = np.where(df['rsi'] > 70, 1, np.where(df['rsi'] < 30, -1, 0))

    # Macroeconomic Data
    economic_data = pd.DataFrame({
        'interest_rate': interest_rates,
        'inflation_rate': inflation_rates
    }).resample('M').last().pct_change()

    economic_data['economic_indicator'] = (economic_data['interest_rate'] - economic_data['inflation_rate']).fillna(0)
    df = df.merge(economic_data, left_index=True, right_index=True, how='left')
    df['macroeconomic_factor'] = np.where(df['economic_indicator'] > 0, 1, np.where(df['economic_indicator'] < 0, -1, 0))

    # Sector-Specific Indicators
    sector_performance = sector_df.groupby('sector')['close'].pct_change(20).groupby(sector_df['sector']).mean()
    df['sector_momentum'] = sector_performance[df['sector']].reindex(df.index, method='ffill')

    # Combine All Factors
    X = df[['composite_trend_volatility', 'volume_adjusted_momentum', 'sentiment_factor', 'macroeconomic_factor', 'sector_momentum']].dropna()
    y = df['close'].pct_change(20).shift(-20).dropna()

    # Train a Model to Predict Future Returns
    model = LinearRegression()
    model.fit(X, y)

    # Use Model Predictions as Weights
    weights = model.coef_
    df['final_alpha_factor'] = (df[['composite_trend_volatility', 'volume_adjusted_momentum', 'sentiment_factor', 'macroeconomic_factor', 'sector_momentum']] * weights).sum(axis=1)

    return df['final_alpha_factor']
```
