import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_days = 10
    short_term_momentum = df['close'].rolling(window=short_term_days).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_days = 50
    long_term_momentum = df['close'].rolling(window=long_term_days).mean()
    
    # Adjust by Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    combined_momentum = (short_term_momentum - df['close']) + (long_term_momentum - df['close'])
    adjusted_combined_momentum = combined_momentum * volume_weighted_returns
    
    # Assess Trend Following Potential
    trend_following_potential = df['close'].rolling(window=long_term_days).sum().pct_change()
    
    # Determine Final Factor Value
    final_factor_value = adjusted_combined_momentum + (trend_following_potential * 0.5)  # Apply weight to trend following component
    
    return final_factor_value.dropna()
