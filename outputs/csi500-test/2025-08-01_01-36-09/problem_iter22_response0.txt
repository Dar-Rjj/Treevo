```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adaptive Volatility
    window = 30
    df['volatility'] = df['log_returns'].rolling(window=window).std()
    recent_returns_ema = df['log_returns'].ewm(span=10, adjust=False).mean()
    df['adaptive_volatility'] = df['volatility'] * (1 + recent_returns_ema)
    
    # Dynamic Trend Following
    short_window = 10
    long_window = 50
    df['short_ma'] = df['close'].rolling(window=short_window).mean()
    df['long_ma'] = df['close'].rolling(window=long_window).mean()
    df['trend_strength'] = (df['short_ma'] - df['long_ma']) / df['close']
    df['roc'] = df['close'].pct_change(periods=short_window)
    df['trend_confirmed'] = (df['trend_strength'] > 0) & (df['roc'] > 0)
    
    # Volume-Adjusted Momentum
    roc_window = 10
    avg_volume = df['volume'].rolling(window=roc_window).mean()
    df['volume_adjusted_momentum'] = (df['close'] / df['close'].shift(roc_window) - 1) * (df['volume'] / avg_volume)
    
    # Relative Strength Index (RSI)
    rsi_window = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_window).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    df['rsi_filter'] = ((df['rsi'] < 30) | (df['rsi'] > 70))
    
    # Combine all components
    df['alpha_factor'] = (
        df['adaptive_volatility'] +
        df['trend_strength'] * df['trend_confirmed'].astype(int) +
        df['volume_adjusted_momentum'] +
        df['rsi_filter'].astype(int)
    )
    
    # Final Alpha Factor
    df['alpha_factor'] = df['alpha_factor'].fillna(0)
    
    return df['alpha_factor']
```
