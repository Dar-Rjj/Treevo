```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_ewm = df['close'].ewm(span=10).mean()
    short_momentum = df['close'] - short_term_ewm
    
    # Calculate Medium-Term Price Momentum
    medium_term_ewm = df['close'].ewm(span=30).mean()
    medium_momentum = df['close'] - medium_term_ewm
    
    # Calculate Long-Term Price Momentum
    long_term_ewm = df['close'].ewm(span=50).mean()
    long_momentum = df['close'] - long_term_ewm
    
    # Adjust by Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Multi-Period Momentum
    combined_momentum = (short_momentum + medium_momentum + long_momentum) * volume_weighted_returns
    
    # Assess Trend Following Potential
    trend_ewma = df['close'].ewm(span=50).mean()
    trend_weight = 1 if trend_ewma[-1] > trend_ewma[0] else -1
    trend_factor = trend_weight * combined_momentum
    
    # Incorporate Price Volatility
    daily_returns_vol = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns_vol.ewm(span=20).std()
    adjusted_momentum = trend_factor / volatility
    
    # Consider Market Sentiment
    sentiment_score = (df['high'] - df['low']) / df['close']
    final_factor = adjusted_momentum * sentiment_score
    
    return final_factor
```
