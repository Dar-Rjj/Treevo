```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    df['short_term_momentum'] = df['close'] - df['10_day_avg']

    # Calculate Long-Term Price Momentum
    df['50_day_avg'] = df['close'].rolling(window=50).mean()
    df['long_term_momentum'] = df['close'] - df['50_day_avg']

    # Adjust by Volume-Weighted Average Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()

    # Integrate Short-Term and Long-Term Momentum
    df['combined_momentum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_weighted_return']

    # Assess Trend Following Potential
    df['50_day_trend'] = df['close'].rolling(window=50).sum()
    df['trend_following_potential'] = df['adjusted_combined_momentum'] + df['50_day_trend']

    return df['trend_following_potential'].dropna()
```
