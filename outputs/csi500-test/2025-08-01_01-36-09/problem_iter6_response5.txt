```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_period = 10
    short_term_avg = df['close'].rolling(window=short_term_period).mean()
    short_term_momentum = short_term_avg - df['close']
    
    # Calculate Long-Term Price Momentum
    long_term_period = 50
    long_term_avg = df['close'].rolling(window=long_term_period).mean()
    long_term_momentum = long_term_avg - df['close']
    
    # Adjust by Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    combined_momentum = short_term_momentum + long_term_momentum
    adjusted_combined_momentum = combined_momentum * volume_weighted_returns
    
    # Assess Trend Following Potential
    trend_following = df['close'].rolling(window=long_term_period).sum()
    
    # Determine Final Factor Value
    final_factor = adjusted_combined_momentum + 0.5 * trend_following
    
    return final_factor
```
