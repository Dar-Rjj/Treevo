```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_avg = df['close'].rolling(window=10).mean()
    short_term_momentum = df['close'] - short_term_avg
    
    # Calculate Long-Term Price Momentum
    long_term_avg = df['close'].rolling(window=50).mean()
    long_term_momentum = df['close'] - long_term_avg
    
    # Adjust by Volume-Weighted Average Return
    daily_returns = (df['close'] - df['open']) / df['open']
    volume_weighted_returns = (daily_returns * df['volume']).sum() / df['volume'].sum()
    
    # Integrate Short-Term and Long-Term Momentum
    combined_momentum = short_term_momentum + long_term_momentum
    integrated_momentum = combined_momentum * volume_weighted_returns
    
    # Assess Trend Following Potential
    moving_sum_50 = df['close'].rolling(window=50).sum()
    trend_weight = (moving_sum_50.diff() > 0).astype(int) * 2 - 1  # 1 for increasing, -1 for decreasing
    
    # Apply Trend Weight to Integrated Momentum
    final_factor = trend_weight * integrated_momentum
    
    return final_factor
```
