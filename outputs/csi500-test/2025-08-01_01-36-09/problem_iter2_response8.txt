```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['5_day_price_momentum'] = df['5_day_avg_close'] - df['close']
    
    # Calculate daily returns using Close and Open prices
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute volume-weighted average return
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    total_volume = df['volume'].rolling(window=5).sum()
    df['volume_weighted_avg_return'] = df['volume_weighted_return'].rolling(window=5).sum() / total_volume
    
    # Adjust 5-Day Price Momentum by Volume-Weighted Average Return
    df['adjusted_momentum'] = df['5_day_price_momentum'] * df['volume_weighted_avg_return']
    
    # Drop rows with NaN values resulting from the rolling window calculations
    df.dropna(inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['adjusted_momentum']
```
