```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['Close_14_days_ago'] = df['close'].shift(14)
    df['Price_Momentum'] = df['close'] - df['Close_14_days_ago']
    
    # Smooth the Difference
    df['Smoothed_Momentum'] = df['Price_Momentum'].ewm(span=14, adjust=False).mean()
    
    # Incorporate Volume Dynamics
    df['Volume_Variability'] = df['volume'].rolling(window=14).std()
    
    # Integrate Volatility
    df['Daily_Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Historical_Volatility'] = df['Daily_Log_Returns'].rolling(window=20).std()
    df['Normalized_Volatility'] = df['Historical_Volatility'] / df['close'].rolling(window=20).mean()
    
    # Combine Momentum, Volume, and Volatility
    df['EVAMO'] = df['Smoothed_Momentum'] * df['Volume_Variability'] * df['Normalized_Volatility']
    
    return df['EVAMO'].dropna()
```
