```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=14, O=20):
    # Calculate Price Momentum using Close price
    df['Return'] = df['close'].pct_change(N)
    
    # Calculate Average True Range (ATR) using High and Low prices
    df['H-L'] = df['high'] - df['low']
    df['H-Cp'] = abs(df['high'] - df['close'].shift(1))
    df['L-Cp'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-Cp', 'L-Cp']].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=M).mean()
    
    # Calculate the average volume over the past O days
    df['Avg_Volume'] = df['volume'].rolling(window=O).mean()
    
    # Adjust Price Momentum for Volume
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume']
    df['Momentum_Adjusted'] = df['Return'] * df['Volume_Ratio']
    
    # Drop unnecessary columns
    alpha_factor = df['Momentum_Adjusted'].dropna()
    
    return alpha_factor
```
