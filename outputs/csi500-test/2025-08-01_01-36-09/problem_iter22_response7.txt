```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Adaptive Volatility
    rolling_window = 30
    df['volatility'] = df['log_returns'].rolling(window=rolling_window).std()
    df['volatility_weight'] = df['log_returns'].rolling(window=5).mean().abs() * 10
    
    # Dynamic Trend Following
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=50).mean()
    df['trend_strength'] = long_term_ma - short_term_ma
    df['momentum'] = df['close'].pct_change(10)
    df['trend_filter'] = (df['trend_strength'] > 0) & (df['momentum'] > 0)
    
    # Volume-Adjusted Momentum
    average_volume = df['volume'].rolling(window=rolling_window).mean()
    current_volume = df['volume']
    df['volume_adjusted_momentum'] = (df['close'].pct_change(10)) * (current_volume / average_volume)
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    df['rsi_signal'] = ((df['rsi'] > 70) | (df['rsi'] < 30)).astype(int)
    
    # Composite Alpha Factor
    alpha_factor = (
        df['volatility_weight'] * df['volatility'] +
        df['trend_strength'] * df['trend_filter'] +
        df['volume_adjusted_momentum'] -
        (df['rsi'] - 50) * df['rsi_signal']
    )
    
    return alpha_factor
```
