```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Price Momentum
    df['close_diff_10_days'] = df['close'].diff(10)
    df['smoothed_momentum'] = df['close_diff_10_days'].rolling(window=10).mean()
    
    # Incorporate Volume Dynamics
    df['volume_variability'] = df['volume'].rolling(window=10).std()
    
    # Evaluate Short-Term Trend
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['short_term_trend'] = (df['close'] > df['5_day_ma']).astype(int)
    
    # Combine Momentum and Volume
    df['combined_momentum_volume'] = df['smoothed_momentum'] * df['volume_variability']
    
    # Determine Final Factor Value
    bias_term = 0.3
    df['final_factor'] = (df['combined_momentum_volume'] * df['short_term_trend']) + bias_term
    
    return df['final_factor']
```
