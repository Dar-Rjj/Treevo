```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute 20-Day Sum of High-Low Differences
    df['volatility_measure'] = df['high_low_diff'].rolling(window=20).sum()
    
    # Calculate 20-Day Close Price Momentum
    df['close_momentum'] = df['close'] - df['close'].shift(20)
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['close_momentum'] / df['volatility_measure']
    
    # Evaluate Long-Term Direction
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['long_term_direction'] = (df['close'] > df['50_day_ma']).astype(int)
    
    # Determine Final Factor Value
    df['final_factor_value'] = (df['adjusted_momentum'] * df['long_term_direction']) + 0.5
    
    return df['final_factor_value'].dropna()
```
