```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['price_momentum'] = df['close'] - df['10_day_avg_close']
    
    # Calculate Volume-Weighted Average Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = (df['daily_return'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust Price Momentum by Volume-Weighted Average Return
    df['adjusted_price_momentum'] = df['price_momentum'] * df['volume_weighted_return']
    
    # Calculate Volatility (Average True Range)
    high_low = df['high'] - df['low']
    high_close = np.abs(df['high'] - df['close'].shift())
    low_close = np.abs(df['low'] - df['close'].shift())
    tr = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
    df['atr_10'] = tr.rolling(window=10).mean()
    
    # Final Alpha Factor: Adjust for Volatility
    df['alpha_factor'] = df['adjusted_price_momentum'] / df['atr_10']
    
    return df['alpha_factor'].dropna()
```
