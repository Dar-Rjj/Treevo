```python
import pandas as pd

def heuristics_v2(data):
    # Calculate Short-Term Price Momentum
    short_mom = data['close'].rolling(window=10).mean() - data['close']
    
    # Calculate Medium-Term Price Momentum
    medium_mom = data['close'].rolling(window=30).mean() - data['close']
    
    # Calculate Long-Term Price Momentum
    long_mom = data['close'].rolling(window=50).mean() - data['close']
    
    # Combine Multi-Period Momenta
    combined_mom = short_mom + medium_mom + long_mom
    
    # Calculate Volume-Weighted Average Return
    daily_return = (data['close'] - data['open']) / data['open']
    volume_weighted_return = (daily_return * data['volume']).sum() / data['volume'].sum()
    
    # Adjust Combined Momentum by Volume-Weighted Average Return
    adjusted_mom = combined_mom * volume_weighted_return
    
    # Assess Trend Following Potential
    trend_50_ma = data['close'].rolling(window=50).mean()
    trend_following_weight = (trend_50_ma > data['close']).astype(float)
    trend_following_weight[trend_following_weight == 0] = 0.5
    
    # Add the trend following component
    adjusted_mom += (combined_mom * trend_following_weight)
    
    # Calculate Short-Term Volatility
    short_vol = (data['high'] - data['low']).rolling(window=10).mean()
    
    # Calculate Medium-Term Volatility
    medium_vol = (data['high'] - data['low']).rolling(window=30).mean()
    
    # Calculate Long-Term Volatility
    long_vol = (data['high'] - data['low']).rolling(window=50).mean()
    
    # Combine Multi-Period Volatilities
    combined_vol = short_vol + medium_vol + long_vol
    
    # Adjust Combined Momentum by Combined Volatility
    final_factor = adjusted_mom / combined_vol
    
    return final_factor
```
