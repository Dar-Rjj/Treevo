```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine and Smooth
    combined_factor = intraday_return * volume_change
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
