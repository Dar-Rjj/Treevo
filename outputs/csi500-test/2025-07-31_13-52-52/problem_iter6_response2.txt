```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int)
    volume_change_indicator = volume_change_indicator.apply(lambda x: 1 if x == 1 else -1)
    
    # Compute Amount Change Indicator
    amount_change_indicator = (df['amount'] > df['amount'].shift(1)).astype(int)
    amount_change_indicator = amount_change_indicator.apply(lambda x: 1 if x == 1 else -1)
    
    # Combine and Smooth
    combined_factor = intraday_return * volume_change_indicator * amount_change_indicator
    
    # Apply Exponential Moving Average
    smoothing_factor = 0.5  # You can adjust this factor
    ema_factor = combined_factor.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return ema_factor
```
