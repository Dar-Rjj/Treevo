```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['Price_Movement_Ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    df['5D_Price_Movement_Ratio'] = df['Price_Movement_Ratio'].rolling(window=5).sum()
    df['5D_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5D_Rolling_Momentum'] = df['5D_Price_Movement_Ratio'] / df['5D_High_Low_Range']
    
    # Adjust for Volume
    df['Volume_Change_Indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    df['5D_Rolling_Momentum_Adjusted'] = df['5D_Rolling_Momentum'] * df['Volume_Change_Indicator']
    
    # Calculate 10-Day Price and Intraday Momentum
    df['10D_Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    df['Combined_Momentum'] = df['10D_Price_Momentum'] + df['Intraday_Return']
    df['Combined_Momentum_Adjusted'] = np.where(df['Volume_Change_Indicator'] > 0, df['Combined_Momentum'], 0) * df['Volume_Change_Indicator']
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = abs(df['high'] - df['low'])
    
    # Calculate Price Trend Indicator
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend_Indicator'] = np.where(df['Short_EMA'] > df['Long_EMA'], 1, -1)
    
    # Incorporate Volatility into Momentum
    df['Normalized_10D_Momentum'] = df['10D_Price_Momentum'] / df['Daily_Volatility']
    df['Adjusted_10D_Momentum'] = df['Normalized_10D_Momentum'] * df['Price_Trend_Indicator']
    
    # Combine Factors
    df['Combined_Factor'] = df['Adjusted_10D_Momentum'] * df['Price_Trend_Indicator']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['5D_Rolling_Momentum_Adjusted'] + df['Combined_Factor']
    
    # Smooth the Result
    df['Smoothed_Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(span=10, adjust=False).mean()
    df['Final_Smoothed_Alpha_Factor'] = df['Smoothed_Final_Alpha_Factor'].rolling(window=10).mean()
    
    # Incorporate Volume Momentum
    df['Volume_Momentum'] = df['volume'] - df['volume'].shift(1)
    df['Final_Alpha_Factor_Adjusted'] = df['Final_Smoothed_Alpha_Factor'] * df['Volume_Momentum']
    
    # Final Adjustment for Volume
    df['Log_Volume'] = np.log(df['volume'] + 1)
    df['Final_Alpha_Factor_Final'] = df['Final_Alpha_Factor_Adjusted'] * df['Log_Volume']
    
    return df['Final_Alpha_Factor_Final']
```
