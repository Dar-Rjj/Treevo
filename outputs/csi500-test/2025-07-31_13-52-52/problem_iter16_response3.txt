import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['close_prev'] = df['close'].shift(1)
    opening_gap = df['open'] - df['close_prev']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_range = high_low_range * volume_change
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_avg = adjusted_range.rolling(window=lookback_period).mean()
    ts_momentum = adjusted_range - moving_avg
    
    # Calculate 5-Day Rolling Momentum
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    rolling_momentum = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust Momentum for Volume
    volume_adjusted_momentum = rolling_momentum * df['volume']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    close_open_return = (df['close'] - df['open']) / df['open']
    daily_volatility = (df['high'] - df['low']).abs()
    final_factor = (volume_adjusted_momentum * close_open_return) / daily_volatility
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    final_alpha_factor = final_factor * avg_5_day_volume
    
    return final_alpha_factor
