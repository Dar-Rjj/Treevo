```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['prev_close'] = df['close'].shift(1)
    df['open_gap'] = df['open'] - df['prev_close']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['vol_wt_range_ma'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['ts_momentum'] = df['volume_weighted_range'] - df['vol_wt_range_ma']
    
    # Calculate 5-Day Rolling Momentum
    df['5_day_sum_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['5_day_sum_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['5_day_rolling_momentum'] = df['5_day_sum_price_movement_ratio'] / df['5_day_sum_high_low_range']
    
    # Adjust Momentum for Volume
    df['volume_adjusted_momentum'] = df['5_day_rolling_momentum'] * df['volume']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    df['final_factor'] = (df['close_open_return'] * df['volume_adjusted_momentum']) / df['daily_volatility']
    
    # Combine Close-to-Open Return with Volume
    df['combined_close_open_return'] = (df['close_open_return'] * df['volume']) / df['daily_volatility']
    
    # Volume Adjusted Momentum
    df['close_diff_5d'] = df['close'].diff().rolling(window=5).sum()
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_adj_momentum'] = df['close_diff_5d'] * df['avg_volume_5d']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['ts_momentum'] + df['volume_adj_momentum'] + df['combined_close_open_return']
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['avg_volume_5d']
    
    return df['final_alpha_factor']
```
