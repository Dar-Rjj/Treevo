    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)

    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']

    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_range'] = df['high_low_range'] * df['volume_momentum']
    
    # Apply Time Series Momentum
    lookback_period = 10

    df['momentum'] = df['adjusted_range'] - df['moving_average']

    df['5_day_rolling_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['5_day_rolling_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['5_day_rolling_momentum'] = df['5_day_rolling_price_movement_ratio'] / df['5_day_rolling_high_low_range']

    df['volume_adjusted_momentum'] = df['volume'] * df['5_day_rolling_momentum']
