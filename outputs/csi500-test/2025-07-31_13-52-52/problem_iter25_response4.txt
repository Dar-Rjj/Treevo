```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_combined_factor'] = df['combined_factor'] - (df['opening_gap'] / df['daily_volatility'])
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    df['5d_price_movement_ratio_sum'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['combined_momentum'] = df['close_to_open_return'] + df['5d_price_movement_ratio_sum']
    
    # Incorporate Volume and Amount Momentum
    df['avg_volume_10_to_2'] = df['volume'].rolling(window=9, min_periods=1).mean().shift(1)
    df['avg_amount_10_to_2'] = df['amount'].rolling(window=9, min_periods=1).mean().shift(1)
    
    df['volume_change'] = df['volume'] - df['avg_volume_10_to_2']
    df['amount_change'] = df['amount'] - df['avg_amount_10_to_2']
    
    df['combined_volume_amount_momentum'] = df['volume_change'] * df['amount_change']
    df['momentum_with_volume_amount'] = df['combined_momentum'] * df['combined_volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['alpha_factor_ma'] = df['momentum_with_volume_amount'].rolling(window=20).mean()
    df['alpha_factor_diff'] = df['momentum_with_volume_amount'] - df['alpha_factor_ma']
    
    # Compute 10-Day Price Momentum
    df['10d_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = df['volume_change'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Apply Volume Confirmation
    df['confirmed_momentum'] = df['momentum_with_volume_amount'] * df['volume_change_indicator']
    df['confirmed_momentum'] = df['confirmed_momentum'].where(df['volume_change'] > 0, 0)
    
    # Integrate Both Metrics
    df['averaged_high_low_momentum'] = df['high_low_range'].rolling(window=20).mean()
    df['smoothed_intraday_return'] = df['close_to_open_return'].rolling(window=20).mean()
    
    # Use Unweighted Average
    df['integrated_metrics'] = (df['averaged_high_low_momentum'] + df['smoothed_intraday_return']) / 2
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['confirmed_momentum'] + df['integrated_metrics']) / 2
    
    # Smooth the Result
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].rolling(window=20).mean()
    
    return df['final_alpha_factor_smoothed']
```
