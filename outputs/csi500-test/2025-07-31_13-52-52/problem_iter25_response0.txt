```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['open_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['moving_average_adjusted_range'] = df['adjusted_high_low_range'].rolling(window=lookback_period).mean()
    df['ts_momentum_difference'] = df['adjusted_high_low_range'] - df['moving_average_adjusted_range']
    
    # Calculate 5-Day Rolling Momentum
    df['5_day_rolling_sum_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['5_day_rolling_sum_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['5_day_rolling_momentum'] = df['5_day_rolling_sum_price_movement_ratio'] / df['5_day_rolling_sum_high_low_range']
    
    # Adjust Momentum for Volume
    df['volume_adjusted_momentum'] = df['volume'] * df['5_day_rolling_momentum']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    df['integrated_factor'] = (df['close_open_return'] * df['volume_adjusted_momentum']) / df['daily_volatility']
    
    # Combine Close-to-Open Return with Volume
    df['combined_close_open_return'] = (df['close_open_return'] * df['volume']) / df['daily_volatility']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['ts_momentum_difference'] + df['volume_adjusted_momentum'] + df['combined_close_open_return']
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['5_day_avg_volume']
    
    return df['final_alpha_factor']
```
