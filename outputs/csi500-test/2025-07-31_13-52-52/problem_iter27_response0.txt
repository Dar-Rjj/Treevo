import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Compute 5-day Rolling Sum of Price Movement Ratio
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    
    # Compute 5-day Rolling Sum of High-Low Range
    rolling_sum_high_low = high_low_range.rolling(window=5).sum()
    
    # Compute 5-Day Rolling Momentum
    rolling_momentum = rolling_sum_price_movement / rolling_sum_high_low
    
    # Adjust for Volume
    adjusted_rolling_momentum = rolling_momentum * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_volume = intraday_return * volume_change_indicator
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    ema_combined_intraday_volume = combined_intraday_volume.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (intraday_return * df['volume']) / daily_volatility
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Integrate Both Momentum Indicators
    integrated_momentum = rolling_sum_price_movement + close_to_open_return
    
    # Incorporate Volume and Amount Momentum
    avg_volume_10_days = df['volume'].rolling(window=10).mean().shift(1)
    avg_amount_10_days = df['amount'].rolling(window=10).mean().shift(1)
    volume_change = df['volume'] - avg_volume_10_days
    amount_change = df['amount'] - avg_amount_10_days
    combined_volume_amount_momentum = (volume_change + amount_change) * high_low_range
    
    # Calculate 10-Day Price Momentum
    ten_day_price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine and Adjust Price and Intraday Momentum
    combined_price_intraday = ten_day_price_momentum + intraday_return
    adjusted_by_volume = combined_price_intraday * (volume_change > 0).astype(int) * volume_change_indicator
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Final Momentum Indicators
    final_momentum_indicators = (adjusted_rolling_momentum + ten_day_price_momentum) * price_trend_indicator
    
    # Smooth the Result
    smoothed_result = final_momentum_indicators.rolling(window=10).mean().ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_rolling_momentum + smoothed_result + combined_volume_amount_momentum * volume_change_indicator
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
