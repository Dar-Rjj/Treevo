```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = np.where(df['short_ema'] > df['long_ema'], 1, -1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = (df['price_movement_ratio'].rolling(window=5).sum() 
                                / df['high_low_range'].rolling(window=5).sum())
    
    # Combine Momentum and Intraday Indicators
    df['combined_indicator'] = (df['high_low_range'] * df['volume_change'] * df['price_trend'])
    df['smoothed_combined_indicator'] = df['combined_indicator'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['final_factor'] = df['smoothed_combined_indicator'] * np.log(df['volume'] + 1)
    
    return df['final_factor']
```
