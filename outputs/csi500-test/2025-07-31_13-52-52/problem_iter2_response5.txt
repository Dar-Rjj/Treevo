```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday'] = df['intraday_return'] * df['volume_change']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['sma_combined'] = df['combined_intraday'].rolling(window=short_period).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['ema_combined'] = df['combined_intraday'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate High-to-Low Price Percentage Change
    df['high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Average Over a Lookback Period
    lookback_period = 10
    df['avg_high_low_pct_change'] = df['high_low_pct_change'].rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Weighted Average
    weight_ema = 0.7
    weight_avg_high_low = 0.3
    df['alpha_factor'] = (weight_ema * df['ema_combined'] + 
                          weight_avg_high_low * df['avg_high_low_pct_change'])
    
    return df['alpha_factor']
```
