```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday'] = df['intraday_return'] * df['volume_change']
    
    # Smooth the Result
    short_period = 5
    df['smoothed_intraday'] = df['combined_intraday'].rolling(window=short_period).mean()
    df['smoothed_intraday'] = df['smoothed_intraday'].ewm(span=short_period, adjust=False).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Calculate Averaged High-to-Low Momentum
    lookback_period = 20
    df['high_low_momentum'] = (df['high'] - df['low']).pct_change().rolling(window=lookback_period).sum() / lookback_period
    
    # Combine and Adjust Factors
    df['adjusted_close_to_open'] = df['close_to_open_return'] * df['volume'] / df['daily_volatility']
    df['adjusted_factor'] = df['adjusted_close_to_open'] - df['5_day_momentum']
    
    # Integrate Both Metrics
    weights = {'smoothed_intraday': 0.4, 'adjusted_factor': 0.3, 'high_low_momentum': 0.3}
    df['alpha_factor'] = (
        df['smoothed_intraday'] * weights['smoothed_intraday'] +
        df['adjusted_factor'] * weights['adjusted_factor'] +
        df['high_low_momentum'] * weights['high_low_momentum']
    )
    
    return df['alpha_factor']
```
