```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Cumulative Price Change
    df['Cumulative_Price_Change'] = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust Cumulative Price Change by Volume Trend
    df['Adjusted_Momentum'] = df['Cumulative_Price_Change'] * df['Volume_Trend']
    
    # Smooth the Adjusted Momentum
    df['Smoothed_Adjusted_Momentum'] = df['Adjusted_Momentum'].rolling(window=5).mean()
    
    return df['Smoothed_Adjusted_Momentum']
```
