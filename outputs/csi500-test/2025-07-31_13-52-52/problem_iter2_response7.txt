```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].diff(10)
    
    # Confirm with Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(1).rolling(window=9).mean()
    df['combined_trend'] = df['price_momentum'] * df['volume_change']
    df['combined_trend'] = df.apply(lambda row: row['combined_trend'] if row['volume_change'] > 0 else 0, axis=1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Adjust for Volume
    df['alpha_factor'] = df['momentum_indicator'] * df['combined_trend']
    
    return df['alpha_factor']
```
