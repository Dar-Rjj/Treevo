import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicator
    momentum_indicator = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    adjusted_momentum = momentum_indicator * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_open_return * df['volume']) / daily_volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_momentum + combined_factor
    
    return alpha_factor
