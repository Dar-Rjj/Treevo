```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Compute Amount Change Indicator
    df['amount_change_indicator'] = np.where(df['amount'] > df['amount'].shift(1), 1, -1)
    
    # Combine Intraday and Volume-Amount Indicators
    df['combined_indicators'] = df['intraday_return'] * df['volume_change_indicator'] * df['amount_change_indicator']
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.5
    df['smoothed_combined_indicators'] = df['combined_indicators'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['previous_close'] = df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] - (df['open'] - df['previous_close'])
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['adjusted_high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['alpha_factor_moving_avg'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['time_series_momentum'] = df['volume_weighted_range'] - df['alpha_factor_moving_avg']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Close-to-Open Return with Volume and Daily Volatility
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_adjusted_return'] = df['close_to_open_return'] * df['volume'] / df['daily_volatility']
    
    # Volume Adjusted Momentum
    df['close_5_days_sum'] = df['close'].rolling(window=5).sum().diff()
    df['volume_average_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = df['close_5_days_sum'] * df['volume_average_5_days']
    
    # Momentum and Volume-Adjusted Close-to-Open Return
    df['momentum_volume_adjusted_return'] = df['volume_adjusted_return'] - df['volume_adjusted_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['time_series_momentum'] + df['volume_adjusted_momentum'] + df['momentum_volume_adjusted_return']
    
    return df['final_alpha_factor'].dropna()
```
