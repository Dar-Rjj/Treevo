```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_high_low_range'] = (df['high'] - df['low']) - df['opening_gap']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() + df['close_to_open_return']
    
    # Incorporate Volume and Amount Momentum
    df['volume_change'] = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    df['amount_change'] = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    df['volume_amount_momentum'] = df['volume_change'] + df['amount_change']
    df['combined_momentum'] = df['momentum_indicator'] * df['volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['alpha_factor_ma'] = df['combined_momentum'].rolling(window=20).mean()
    df['time_series_momentum'] = df['combined_momentum'] - df['alpha_factor_ma']
    
    # Calculate High-Low Spread Change
    df['high_low_spread_change'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    df['high_low_spread_momentum'] = df['high_low_spread_change'].rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['open'] + df['high'] + df['low'] + df['close']) / 4) * df['volume'] / df['volume']
    
    # Introduce Price Trend Component
    df['price_trend'] = df['close'] - df['close'].ewm(span=30, adjust=False).mean()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    df['combined_factor_vwap'] = df['vwap'] * df['high_low_spread_momentum'] * df['adjusted_high_low_range']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['time_series_momentum'] + df['combined_factor_vwap'] + df['price_trend']
    
    # Final Adjustment for Volume
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    
    return df['final_alpha_factor'].dropna()
```
