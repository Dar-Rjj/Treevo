```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1

    # Combine Intraday Return and Volume Change
    df['intraday_return_volume'] = df['intraday_return'] * df['volume_change_indicator']

    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['sma_intraday_return_volume'] = df['intraday_return_volume'].rolling(window=short_period).mean()

    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2  # Define Smoothing Factor
    df['ema_intraday_return_volume'] = df['intraday_return_volume'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Averaged High-to-Low Momentum
    lookback_period = 20  # Define Lookback Period
    df['hl_momentum'] = (df['high'] - df['low']) / df['low']
    df['avg_hl_momentum'] = df['hl_momentum'].rolling(window=lookback_period).sum() / lookback_period

    # Integrate Both Metrics using an Unweighted Average
    df['alpha_factor'] = (df['ema_intraday_return_volume'] + df['avg_hl_momentum']) / 2

    return df['alpha_factor']
```
