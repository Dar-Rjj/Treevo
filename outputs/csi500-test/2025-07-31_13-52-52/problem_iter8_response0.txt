import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    rolling_momentum = (price_movement_ratio.rolling(window=5).sum() 
                        / high_low_range.rolling(window=5).sum())
    
    # Adjust Momentum for Volume
    volume_adjusted_momentum = rolling_momentum * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    final_factor = (volume_adjusted_momentum * close_open_return) / daily_volatility
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    average_volume = df['volume'].rolling(window=5).mean()
    final_alpha_factor = final_factor * average_volume
    
    return final_alpha_factor
