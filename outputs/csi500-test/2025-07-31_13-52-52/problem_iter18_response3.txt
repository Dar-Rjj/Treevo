import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']

    # Calculate 5-day Rolling Sum of Price Movement Ratio and High-Low Range
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()

    # Compute Momentum Indicator
    momentum_indicator = rolling_sum_price_movement / rolling_sum_high_low_range

    # Adjust for Volume
    adjusted_momentum = momentum_indicator * df['volume']

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Compute Volume Change Indicator
    volume_change_indicator = np.where(df['volume'] > df['volume'].shift(1), 1, -1)

    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_volume = intraday_return * volume_change_indicator

    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    smoothed_combined = combined_intraday_volume.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Daily Volatility
    daily_volatility = abs(df['high'] - df['low'])

    # Combine Factors
    combined_factor = (intraday_return * df['volume']) / daily_volatility

    # Calculate 5-day Moving Average of Close Prices
    moving_avg_5_day_close = df['close'].rolling(window=5).mean()

    # Calculate Relative Strength
    relative_strength = df['close'] / moving_avg_5_day_close

    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum + combined_factor) * relative_strength

    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    high_low_percentage_changes = (df['high'] - df['low']) / df['low']
    averaged_high_low_volatility = high_low_percentage_changes.rolling(window=lookback_period).mean()

    # Integrate Both Metrics
    weight_adjusted_momentum = 0.6
    weight_averaged_volatility = 0.4
    integrated_factor = (weight_adjusted_momentum * adjusted_momentum) + (weight_averaged_volatility * averaged_high_low_volatility)

    # Calculate 20-day Moving Average of Close Prices
    moving_avg_20_day_close = df['close'].rolling(window=20).mean()

    # Calculate Long-Term Trend Indicator
    long_term_trend = df['close'] / moving_avg_20_day_close

    # Enhance Final Alpha Factor
    enhanced_final_alpha_factor = final_alpha_factor * long_term_trend

    return integrated_factor * enhanced_final_alpha_factor
