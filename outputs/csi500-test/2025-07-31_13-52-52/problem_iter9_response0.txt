```python
import pandas as pd

def heuristics_v2(df, window_size=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Compute Amount Change Indicator
    amount_change = (df['amount'] > df['amount'].shift(1)).astype(int) * 2 - 1
    
    # Combine and Smooth
    combined_factor = intraday_return * volume_change * amount_change
    smoothed_factor = combined_factor.rolling(window=window_size).mean()
    
    return smoothed_factor
```
