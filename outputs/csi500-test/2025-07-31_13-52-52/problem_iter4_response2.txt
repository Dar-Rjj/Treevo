```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Calculate 5-Day Momentum
    momentum_5_day = df['close'].diff().rolling(window=5).sum()
    
    # Combine and Adjust Factors
    factor = (close_to_open_return * df['volume']) / daily_volatility - momentum_5_day
    
    return factor
```
