```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_range
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    momentum_adjusted = (momentum / avg_volume_5_days).fillna(0)
    combined_factor = combined_factor + momentum_adjusted
    
    # Introduce Trend Sensitivity
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    trend_sensitivity = (ma_20 > ma_50) * (1 + 0.05 * (ma_20 - ma_50)) + (ma_20 < ma_50) * (1 - 0.05 * (ma_50 - ma_20))
    combined_factor = combined_factor * trend_sensitivity
    
    return combined_factor
```
