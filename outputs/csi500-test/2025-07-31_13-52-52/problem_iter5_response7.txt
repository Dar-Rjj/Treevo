import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday'] = df['intraday_return'] * df['volume_change']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    df['sma_intraday'] = df['combined_intraday'].rolling(window=5).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['ema_intraday'] = df['combined_intraday'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Combine and Adjust Factors
    df['adjusted_close_to_open'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    df['adjusted_factors'] = df['adjusted_close_to_open'] - df['5_day_momentum']
    
    # Integrate Both Metrics
    weight_intraday = 0.6
    weight_close_to_open = 0.4
    df['final_alpha_factor'] = (weight_intraday * df['ema_intraday']) + (weight_close_to_open * df['adjusted_factors'])
    
    return df['final_alpha_factor']
