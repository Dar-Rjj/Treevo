```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    factor_1 = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] / df['low'] - 1
    
    # Compute 10-Day Price Momentum
    ten_day_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    combined_momentum = intraday_return + ten_day_momentum
    
    # Apply Volume Confirmation
    confirmed_momentum = combined_momentum * volume_change_indicator
    confirmed_momentum[volume_change_indicator <= 0] = 0
    
    # Integrate Both Metrics
    integrated_metric = (factor_1 + confirmed_momentum) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (momentum + integrated_metric) / 2
    
    # Smooth the Result
    smoothed_alpha_factor = final_alpha_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_alpha_factor
```
