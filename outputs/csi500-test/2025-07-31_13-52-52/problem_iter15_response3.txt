```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_range
    
    # Adjust for Momentum
    close_diff_sum = df['close'].diff().rolling(window=5).sum()
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    momentum_adjustment = close_diff_sum / avg_volume_5_days
    
    # Final Alpha Factor
    alpha_factor = combined_factor * momentum_adjustment
    
    return alpha_factor
```
