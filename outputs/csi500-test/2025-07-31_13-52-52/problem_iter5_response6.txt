```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday_vol'] = df['intraday_return'] * df['volume_change']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['smoothed_intraday_vol_sma'] = df['combined_intraday_vol'].rolling(window=short_period).mean()
    
    # Smooth the Result using Exponential Moving Average
    smoothing_factor = 0.2
    df['smoothed_intraday_vol_ema'] = df['combined_intraday_vol'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate High-to-Low Price Percentage Change
    df['high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Average Over a Lookback Period
    lookback_period = 10
    df['avg_high_low_pct_change'] = df['high_low_pct_change'].rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Unweighted Average
    df['alpha_factor'] = (df['smoothed_intraday_vol_ema'] + df['avg_high_low_pct_change']) / 2
    
    return df['alpha_factor']
```
