```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factors'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    df['adjusted_momentum'] = momentum
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = np.where(df['short_sma'] > df['long_sma'], 1, -1)
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['10_day_momentum'] + df['intraday_return']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['combined_momentum'] * df['volume_change_indicator']
    df['volume_confirmed_momentum'] = np.where(df['volume_change_indicator'] > 0, df['volume_confirmed_momentum'], 0)
    
    # Integrate Both Metrics
    weights = [0.5, 0.5]  # Define weights for each metric
    df['integrated_metrics'] = (weights[0] * df['daily_volatility'] + weights[1] * df['intraday_return'])
    
    # Synthesize Final Alpha Factor
    final_weights = [0.4, 0.6]  # Define weights for each component
    df['final_alpha_factor'] = (final_weights[0] * df['adjusted_momentum'] + final_weights[1] * df['integrated_metrics'])
    
    # Incorporate Volume and Amount Momentum
    df['volume_change'] = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    df['amount_change'] = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    df['combined_volume_amount_momentum'] = (df['volume_change'] + df['amount_change']) / (df['high'] - df['low'])
    
    df['final_alpha_factor'] *= df['combined_volume_amount_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = np.where(df['volume_change_indicator'] > 0, df['final_alpha_factor'], 0)
    df['final_alpha_factor'] = np.where(df['intraday_return'] > 0, df['final_alpha_factor'], 0)
    
    # Smooth the Result
    df['smoothed_final_alpha_factor'] = df['final_alpha_factor'].ewm(span=10).mean()
    
    return df['smoothed_final_alpha_factor']
```
