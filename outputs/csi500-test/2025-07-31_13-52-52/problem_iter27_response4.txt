import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_momentum_5d'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Adjust for Volume
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    df['adjusted_rolling_momentum_5d'] = df['rolling_momentum_5d'] * df['volume_change_indicator']
    
    # Calculate 10-Day Price Momentum
    df['price_momentum_10d'] = df['close'] - df['close'].shift(10)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    df['combined_momentum'] = df['price_momentum_10d'] + df['intraday_return']
    df['combined_momentum'] = np.where(df['volume_change_indicator'] > 0, df['combined_momentum'], 0) * df['volume_change_indicator']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_indicator'] = np.where(df['short_ema'] > df['long_ema'], 1, -1)
    
    # Incorporate Volatility into Momentum
    df['normalized_momentum_10d'] = df['price_momentum_10d'] / df['daily_volatility']
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['price_trend_indicator']
    
    # Combine Factors
    df['combined_factor'] = df['adjusted_rolling_momentum_5d'] + df['adjusted_combined_momentum']
    
    # Smooth the Result
    df['smoothed_factor'] = df['combined_factor'].ewm(span=10, adjust=False).mean() + df['combined_factor'].rolling(window=10).mean()
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['final_alpha_factor'] = df['smoothed_factor'] * (df['volume'] + 1)
    
    return df['final_alpha_factor'].dropna()
