import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate 5-Day Rolling Momentum
    rolling_momentum = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    volume_adjusted_momentum_5d = rolling_momentum * df['volume']
    
    # Calculate 10-Day Price Momentum
    price_momentum_10d = df['close'].diff(periods=10)
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    combined_momentum = price_momentum_10d + intraday_return
    volume_change = df['volume'].diff()
    volume_change_indicator = (volume_change > 0).astype(int) - (volume_change <= 0).astype(int)
    adjusted_combined_momentum = combined_momentum * volume_change_indicator
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Factors
    combined_factor = adjusted_combined_momentum * price_trend_indicator
    
    # Final Alpha Factor
    final_alpha_factor = volume_adjusted_momentum_5d + combined_factor
    final_alpha_factor_smoothed = final_alpha_factor.rolling(window=10).mean().ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume Momentum
    volume_change = df['volume'].diff()
    final_alpha_factor_with_volume = final_alpha_factor_smoothed * volume_change
    
    # Final Adjustment for Volume
    final_alpha_factor_adjusted = final_alpha_factor_with_volume * df['volume']
    
    return final_alpha_factor_adjusted
