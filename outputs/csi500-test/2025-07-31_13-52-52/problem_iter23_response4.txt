```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow ratio
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Calculate the price momentum
    price_momentum = df['close'] / df['close'].shift(12) - 1
    
    # Calculate the volume change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Combine the factors
    factor = money_flow_ratio * price_momentum * volume_change
    
    return factor
```
