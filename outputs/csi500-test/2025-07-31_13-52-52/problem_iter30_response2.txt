def heuristics_v2(df):
    # Momentum Indicators
    df['WMA'] = (df['close'] * 2 + df['high'] + df['low']) / 4
    df['WMA_10'] = df['WMA'].rolling(window=10).mean()
    
    # RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Volatility Indicators
    df['TR'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs().combine(
        df['high'].combine(df['close'].shift(1), lambda x1, x2: abs(x1 - x2)),
        max
    ).combine(
        df['low'].combine(df['close'].shift(1), lambda x1, x2: abs(x1 - x2)),
        max
    )
    
    sma_close_20 = df['close'].rolling(window=20).mean()
    std_close_20 = df['close'].rolling(window=20).std()
    df['Bollinger_Upper'] = sma_close_20 + 2 * std_close_20
    df['Bollinger_Lower'] = sma_close_20 - 2 * std_close_20
    
    # Volume Based Factors
    df['OBV'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'OBV'] = df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'OBV'] = -df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Money Flow Index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_avg = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_avg = negative_money_flow.rolling(window=14).sum()
    df['MFI'] = 100 - (100 / (1 + (positive_money_flow_avg / negative_money_flow_avg)))
    
    # Price-Volume Combined Factors
    df['PVT'] = (df['close'].diff() / df['close'].shift(1) * df['volume']).cumsum()
    
    df['A/D'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['A/D'] = df['A/D'].cumsum()
    
    # Gap and Opening Factors
    df['Open-Close_Ratio'] = df['open'] / df['close'].shift(1)
    df['High-Low_Gap'] = df['high'] - df[['close', 'open']].max(axis=1)
    df['Low-High_Reversal'] = df[['open', 'close']].min(axis=1) - df['low']
    
    # Market Sentiment Factors
    df['Bearish_Bullish'] = (df['close'] < df['open']).astype(int)
    df['Hammer_Pattern'] = (df['close'] - df['low']) > 2 * (df['open'] - df['close']) & (df['close'] - df['open']) < 0.3 * (df['high'] - df['low'])
    
    # Transaction Value Considerations
    df['Average_Value_Per_Transaction'] = (df['high'] + df['low']) / 2 * df['volume']
    df['Value_Growth'] = df['Average_Value_Per_Transaction'].pct_change(periods=10)
    
    # Statistical Measures on Historical Data
    df['Return_Skewness'] = df['close'].pct_change().rolling(window=20).apply(lambda x: x.skew())
    df['Return_Kurtosis'] = df['close'].pct_change().rolling(window=20).apply(lambda x: x.kurtosis())
