import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratio
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    
    # Calculate 5-day Rolling Sum of High-Low Range
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    
    # Compute Momentum Indicator
    momentum_indicator = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust for Volume
    adjusted_momentum_indicator = momentum_indicator * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Compute Amount Change Indicator
    amount_change_indicator = df['amount'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_indicator = intraday_return * volume_change_indicator * amount_change_indicator
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    smoothed_combined_indicator = combined_intraday_indicator.ewm(alpha=smoothing_factor).mean()
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (intraday_return * df['volume']) / daily_volatility
    
    # Calculate 5-day Moving Average of Close Prices
    moving_avg_close = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    relative_strength = df['close'] / moving_avg_close
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum_indicator + combined_factor) * relative_strength
    
    return final_alpha_factor
