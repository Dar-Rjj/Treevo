```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['open_gap'] = (df['open'] - df['close'].shift(1)).fillna(0)
    df['adjusted_range'] = df['high_low_range'] - df['open_gap']
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1).fillna(0)
    df['volume_weighted_range'] = df['adjusted_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['vol_weighted_range_ma'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['momentum'] = df['volume_weighted_range'] - df['vol_weighted_range_ma']
    
    # Calculate High-Low Spread Change
    df['high_low_spread_change'] = df['high_low_range'] - df['high_low_range'].shift(1).fillna(0)
    
    # Calculate Momentum of High-Low Spread Change
    spread_momentum_lookback = 5
    df['spread_momentum'] = df['high_low_spread_change'].rolling(window=spread_momentum_lookback).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine VWAP and High-Low Spread Momentum with Adjusted Range
    df['alpha_factor'] = df['vwap'] * df['spread_momentum'] * df['momentum']
    
    return df['alpha_factor']
```
