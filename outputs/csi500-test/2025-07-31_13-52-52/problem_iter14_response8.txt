```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Compute Amount Change Indicator
    amount_change_indicator = (df['amount'] > df['amount'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return, Volume Change, and Amount Change
    combined_intraday_return = intraday_return * volume_change_indicator * amount_change_indicator
    
    # Smooth the Result using Simple Moving Average over a Short Period
    short_period = 5
    sma_smoothed = combined_intraday_return.rolling(window=short_period).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    ema_smoothed = sma_smoothed.ewm(alpha=smoothing_factor).mean()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    high_low_volatility = (df['high'] - df['low']) / df['low']
    avg_high_low_volatility = high_low_volatility.rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Unweighted Average
    final_alpha_factor = (ema_smoothed + avg_high_low_volatility) / 2
    
    return final_alpha_factor
```
