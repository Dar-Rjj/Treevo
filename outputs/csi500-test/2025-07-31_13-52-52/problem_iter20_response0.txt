```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low and Close-Open Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Price_Movement_Ratio'] = (df['close'] - df['open']) / df['open']
    df['5_Day_Rolling_Momentum'] = df['Price_Movement_Ratio'].rolling(window=5).sum() / df['High_Low_Range'].rolling(window=5).mean()
    
    # Adjust for Volume
    df['Volume_Change_Indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    df['5_Day_Adjusted_Momentum'] = df['5_Day_Rolling_Momentum'] * df['Volume_Change_Indicator']
    
    # Calculate 10-Day Price and Intraday Momentum
    df['10_Day_Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Combined_Momentum'] = df['10_Day_Price_Momentum'] + df['Intraday_Return']
    
    # Adjust by Volume
    df['Combined_Adjusted_Momentum'] = np.where(df['Volume_Change_Indicator'] > 0, df['Combined_Momentum'], 0)
    df['Combined_Adjusted_Momentum'] *= df['Volume_Change_Indicator']
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Price Trend Indicator
    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend_Indicator'] = np.where(df['Short_EMA'] > df['Long_EMA'], 1, -1)
    
    # Combine Factors
    df['Combined_Factor'] = df['Combined_Adjusted_Momentum'] * df['Price_Trend_Indicator']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['5_Day_Adjusted_Momentum'] + df['Combined_Factor']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(span=10, adjust=False).mean().rolling(window=10).mean()
    
    # Incorporate Volume Momentum
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Final_Alpha_Factor'] *= df['Volume_Change']
    
    # Final Adjustment for Volume
    df['Final_Alpha_Factor'] *= np.log(df['volume'] + 1)
    
    return df['Final_Alpha_Factor']
```
