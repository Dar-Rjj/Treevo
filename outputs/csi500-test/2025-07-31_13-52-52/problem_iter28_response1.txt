```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    df['momentum_adjustment'] = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Apply Volume Confirmation
    df['volume_confirmed_combined_factor'] = df['combined_factor'] * df['volume_change']
    
    # Synthesize Final Alpha Factor
    # Define Weights for Each Component
    weights = {
        'momentum_adjustment': 0.4,
        '10_day_momentum': 0.3,
        'volume_confirmed_combined_factor': 0.3
    }
    
    df['final_alpha_factor'] = (weights['momentum_adjustment'] * df['momentum_adjustment'] +
                                weights['10_day_momentum'] * df['10_day_momentum'] +
                                weights['volume_confirmed_combined_factor'] * df['volume_confirmed_combined_factor'])
    
    # Smooth the Result
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    return df['final_alpha_factor_smoothed']
```
