```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Compute Volume Change Indicator
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Short SMA (5 days) and Long SMA (20 days) of Close prices
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    
    # Determine Price Trend Indicator
    price_trend = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratios
    rolling_sum_price_ratio = price_movement_ratio.rolling(window=5).sum()
    
    # Calculate 5-day Rolling Sum of High-Low Ranges
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    
    # Calculate Momentum Indicator
    momentum_indicator = rolling_sum_price_ratio / rolling_sum_high_low_range
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    combined_indicator = (intraday_return * volume_change * price_trend)
    
    # Smooth the Result with Simple Moving Average (10 days)
    smoothed_combined_indicator = combined_indicator.rolling(window=10).mean()
    
    # Exponential Moving Average with Smoothing Factor
    ema_smoothed_combined_indicator = smoothed_combined_indicator.ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    adjusted_momentum = ema_smoothed_combined_indicator * df['volume']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (close_to_open_return * df['volume']) / daily_volatility
    
    # Calculate 5-day Moving Average of Close Prices
    moving_avg_5d_close = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    relative_strength = df['close'] / moving_avg_5d_close
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum + combined_factors * relative_strength
    
    return final_alpha_factor
```
