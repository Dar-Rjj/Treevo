```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int)
    volume_change_indicator = volume_change * 2 - 1  # 1 if increased, -1 if decreased
    
    # Combine Intraday Return and Volume Change
    combined_return = intraday_return * volume_change_indicator
    
    # Smooth the Result using Simple Moving Average over a Short Period
    short_period = 7
    sma_smoothed_return = combined_return.rolling(window=short_period).mean()
    
    # Further smooth using Exponential Moving Average
    smoothing_factor = 0.3
    ema_smoothed_return = sma_smoothed_return.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 21
    high_to_low_volatility = (df['high'] - df['low']) / df['low']
    averaged_high_to_low_volatility = high_to_low_volatility.rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Unweighted Average
    final_alpha_factor = (ema_smoothed_return + averaged_high_to_low_volatility) / 2
    
    return final_alpha_factor
```
