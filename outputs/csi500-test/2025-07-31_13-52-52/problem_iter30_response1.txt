import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['Price_Movement_Ratio'] = (df['close'] - df['open']) / df['open']
    
    # Compute 5-day Rolling Sum of Price Movement Ratio
    df['Rolling_Sum_Price_Movement_Ratio'] = df['Price_Movement_Ratio'].rolling(window=5).sum()
    
    # Compute 5-day Rolling Sum of High-Low Range
    df['Rolling_Sum_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).sum()
    
    # Compute 5-Day Rolling Momentum
    df['5_Day_Rolling_Momentum'] = df['Rolling_Sum_Price_Movement_Ratio'] / df['Rolling_Sum_High_Low_Range']
    
    # Adjust for Volume
    df['Volume_Adjusted_5_Day_Rolling_Momentum'] = df['5_Day_Rolling_Momentum'] * df['volume']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    df['Volume_Change_Indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Combine Intraday and Volume-Amount Indicators
    df['Combined_Intraday_Volume'] = df['Intraday_Return'] * df['Volume_Change_Indicator']
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['Smoothed_Combined_Intraday_Volume'] = df['Combined_Intraday_Volume'].ewm(alpha=smoothing_factor).mean()
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['Combined_Factors'] = (df['Intraday_Return'] * df['volume']) / df['Daily_Volatility']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    df['Close_to_Open_Return'] = df['Close_to_Open_Return'].shift(1)  # Shift to align with the current day
    
    # Integrate Both Momentum Indicators
    df['Integrated_Momentum'] = df['Rolling_Sum_Price_Movement_Ratio'] + df['Close_to_Open_Return']
    
    # Incorporate Volume and Amount Momentum
    df['Volume_Change'] = df['volume'] - df['volume'].rolling(window=10).mean().shift(1)
    df['Amount_Change'] = df['amount'] - df['amount'].rolling(window=10).mean().shift(1)
    df['Weighted_Volume_Amount_Momentum'] = (df['Volume_Change'] + df['Amount_Change']) * df['High_Low_Range']
    
    # Calculate 10-Day Price Momentum
    df['10_Day_Price_Momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine and Adjust Price and Intraday Momentum
    df['Combined_Price_Intraday_Momentum'] = df['10_Day_Price_Momentum'] + df['Intraday_Return']
    df['Adjusted_Price_Intraday_Momentum'] = np.where(df['Volume_Change'] > 0, df['Combined_Price_Intraday_Momentum'], 0) * df['Volume_Change_Indicator']
    
    # Calculate Price Trend Indicator
    df['Short_SMA'] = df['close'].rolling(window=5).mean()
    df['Long_SMA'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Indicator'] = np.where(df['Short_SMA'] > df['Long_SMA'], 1, -1)
    
    # Combine Final Momentum Indicators
    df['Final_Momentum'] = (df['Volume_Adjusted_5_Day_Rolling_Momentum'] + df['10_Day_Price_Momentum']) * df['Price_Trend_Indicator']
    
    # Smooth the Result
    df['SMA_Final_Momentum'] = df['Final_Momentum'].rolling(window=10).mean()
    df['EMA_Final_Momentum'] = df['SMA_Final_Momentum'].ewm(alpha=smoothing_factor).mean()
