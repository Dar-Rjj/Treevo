```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = np.where(df['short_ema'] > df['long_ema'], 1, -1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_momentum'] = (df['high_low_range'] * df['volume_change'] * df['price_trend'])
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['adjusted_momentum'] = df['smoothed_combined_momentum'] * np.log(df['volume'] + 1)
    
    # Calculate 5-day Moving Average of Close Prices
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['ma_5_close']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] * df['relative_strength']
    
    return df['alpha_factor']
```
