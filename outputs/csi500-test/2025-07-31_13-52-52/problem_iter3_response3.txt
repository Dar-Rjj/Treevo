```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].diff(periods=10)
    
    # Calculate 10-Day Volume Change
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean().shift(1)
    df['volume_change'] = df['volume'] - df['avg_volume_10']
    
    # Calculate 10-Day Amount Change
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean().shift(1)
    df['amount_change'] = df['amount'] - df['avg_amount_10']
    
    # Combine Momentum, Volume, and Amount Trends
    df['factor'] = df['price_momentum'] * df['volume_change'] * df['amount_change']
    
    # Assign Weight based on Volume and Amount changes
    df['factor'] = df.apply(lambda row: row['factor'] if (row['volume_change'] > 0 and row['amount_change'] > 0) else 0, axis=1)
    
    return df['factor']
```
