```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['open_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['moving_avg_vol_wt_range'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['ts_momentum'] = df['volume_weighted_range'] - df['moving_avg_vol_wt_range']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_sum_price_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['rolling_sum_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['5_day_rolling_momentum'] = df['rolling_sum_price_ratio'] / df['rolling_sum_high_low_range']
    
    # Adjust Momentum for Volume
    df['volume_adjusted_momentum'] = df['volume'] * df['5_day_rolling_momentum']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['volatility'] = (df['high'] - df['low']).abs()
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_momentum_indicator'] = df['volume_adjusted_momentum'] * df['close_open_return'] / df['volatility']
    
    # Combine Close-to-Open Return with Volume
    df['combined_close_open_volume'] = (df['close_open_return'] * df['volume']) / df['volatility']
    
    # Calculate Averaged High-to-Low Volatility
    volatility_lookback_period = 20
    df['high_to_low_volatility'] = (df['high'] - df['low']).pct_change().abs().rolling(window=volatility_lookback_period).sum() / volatility_lookback_period
    
    # Integrate Both Metrics
    weight1, weight2 = 0.6, 0.4
    df['interim_alpha_factor'] = weight1 * df['adjusted_momentum_indicator'] + weight2 * df['combined_close_open_volume']
    
    # Calculate 5-day Moving Average of Close Prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['5_day_ma_close']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['interim_alpha_factor'] * df['relative_strength']
    
    # Calculate 20-day Moving Average of Close Prices
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate Long-Term Trend Indicator
    df['long_term_trend'] = df['close'] / df['20_day_ma_close']
    
    # Enhance Final Alpha Factor
    df['enhanced_final_alpha_factor'] = df['final_alpha_factor'] * df['long_term_trend']
    
    # Incorporate Long-Term Volatility
    df['20_day_volatility'] = df['close'].rolling(window=20).std()
    df['final_alpha_factor_with_volatility'] = df['enhanced_final_alpha_factor'] / (df['20_day_volatility'] + 1e-8)
    
    # Integrate Recent Volume Trend
    df['5_day_ema_volume'] = df['volume'].ewm(span=5).mean()
    df['final_alpha_factor'] = df['final_alpha_factor_with_volatility'] * df['5_day_ema_volume']
    
    return df['final_alpha_factor']
```
