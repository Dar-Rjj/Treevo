```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return (High - Low)
    intraday_return = df['high'] - df['low']
    
    # Compute Volume Change Indicator
    volume_change = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend = np.where(short_sma > long_sma, 1, -1)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio (Close / Open)
    price_movement_ratio = df['close'] / df['open']
    
    # Calculate Momentum Indicator
    momentum_indicator = (price_movement_ratio.rolling(window=5).sum() / 
                          high_low_range.rolling(window=5).sum())
    
    # Combine Momentum and Intraday Indicators
    combined_momentum_intraday = (intraday_return * volume_change * price_trend)
    
    # Smooth the Result with Simple Moving Average (10 days) and Exponential Moving Average
    sma_10 = combined_momentum_intraday.rolling(window=10).mean()
    ema_smoothed = sma_10.ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    final_factor = ema_smoothed * df['volume']
    
    return final_factor
```
