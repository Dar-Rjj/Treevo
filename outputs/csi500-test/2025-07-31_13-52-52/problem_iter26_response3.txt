```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['open_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_range'] = df['high_low_range'] * df['volume_change_indicator']
    
    # Apply Time Series Momentum
    df['moving_average_adj_range'] = df['adjusted_range'].rolling(window=10).mean()
    df['momentum_diff'] = df['adjusted_range'] - df['moving_average_adj_range']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['rolling_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['rolling_momentum'] = df['rolling_price_movement_ratio'] / df['rolling_high_low_range']
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_momentum'] = (df['high'] - df['low']) * df['volume_change_indicator'] * df['momentum_diff']
    df['smoothed_combined_momentum'] = df['combined_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['log_volume'] = np.log(df['volume'] + 1)
    df['adjusted_combined_momentum'] = df['smoothed_combined_momentum'] * df['log_volume']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['daily_volatility'] = np.abs(df['high'] - df['low'])
    df['vol_adjusted_momentum'] = df['adjusted_combined_momentum'] * (df['close'] / df['open']) / df['daily_volatility']
    
    # Calculate Averaged High-to-Low Volatility
    df['high_low_volatility'] = df['high'] - df['low']
    df['avg_high_low_volatility'] = df['high_low_volatility'].rolling(window=20).mean()
    
    # Calculate 5-day Moving Average of Close Prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['close'] - df['5_day_ma_close']) / df['5_day_ma_close']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['vol_adjusted_momentum'] + df['rolling_momentum'] * df['relative_strength']
    
    # Calculate 20-day Moving Average of Close Prices
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate Long-Term Trend Indicator
    df['long_term_trend'] = df['close'] / df['20_day_ma_close']
    
    # Enhance Final Alpha Factor
    df['enhanced_final_alpha_factor'] = df['final_alpha_factor'] * df['long_term_trend']
    
    # Incorporate Long-Term Volatility
    df['20_day_volatility'] = df['close'].rolling(window=20).std()
    df['inverse_20_day_volatility'] = 1 / df['20_day_volatility']
    df['final_alpha_factor_with_volatility'] = df['enhanced_final_alpha_factor'] * df['inverse_20_day_volatility']
    
    # Integrate Recent Volume Trend
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['final_alpha_factor_with_volume_trend'] = df['final_alpha_factor_with_volatility'] * df['5_day_ema_volume']
    
    return df['final_alpha_factor_with_volume_trend']
```
