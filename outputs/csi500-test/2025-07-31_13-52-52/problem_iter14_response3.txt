```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratio
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    
    # Calculate 5-day Rolling Sum of High-Low Range
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    
    # Compute Momentum Indicator
    momentum_indicator = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust for Volume
    adjusted_momentum = momentum_indicator * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_volume = intraday_return * volume_change
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.5
    smoothed_combined = combined_intraday_volume.ewm(alpha=smoothing_factor).mean()
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (intraday_return * df['volume']) / daily_volatility
    
    # Calculate 5-day Moving Average of Close Prices
    moving_avg_close = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    relative_strength = df['close'] / moving_avg_close
    
    # Final Alpha Factor
    final_alpha = (adjusted_momentum + combined_factors) * relative_strength
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 30
    high_low_volatility = ((df['high'] - df['low']) / df['low']).rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics
    weight_adjusted_momentum = 0.6
    weight_high_low_volatility = 0.4
    final_alpha_factor = (weight_adjusted_momentum * final_alpha) + (weight_high_low_volatility * high_low_volatility)
    
    return final_alpha_factor
```
