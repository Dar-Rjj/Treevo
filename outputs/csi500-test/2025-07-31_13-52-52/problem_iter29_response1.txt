    # Calculate High-Low and Close-Open Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Price_Movement_Ratio'] = (df['close'] - df['open']) / df['open']
    df['5D_Price_Movement_Ratio_Sum'] = df['Price_Movement_Ratio'].rolling(window=5).sum()
    df['5D_High_Low_Range_Sum'] = df['High_Low_Range'].rolling(window=5).sum()
    df['5D_Rolling_Momentum'] = df['5D_Price_Movement_Ratio_Sum'] / df['5D_High_Low_Range_Sum']

    df['Volume_Change_Indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1

    df['10D_Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    df['Combined_Momentum'] = df['10D_Price_Momentum'] + df['Intraday_Return']

    df['Combined_Momentum_Adjusted'] = df['Combined_Momentum'] * (df['Volume_Change_Indicator'] == 1)

    df['Daily_Volatility'] = abs(df['high'] - df['low'])

    df['Short_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Trend_Indicator'] = (df['Short_EMA'] > df['Long_EMA']).astype(int) * 2 - 1
