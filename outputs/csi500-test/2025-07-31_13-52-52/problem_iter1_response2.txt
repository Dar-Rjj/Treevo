```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Smooth High-Low Difference with a moving average
    window_size = 10  # Example window size, can be adjusted
    smoothed_high_low_diff = high_low_diff.rolling(window=window_size).mean()
    
    # Incorporate Volume
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Multiply Volume Delta with Smoothed High-Low Difference
    volume_weighted_smoothed_diff = smoothed_high_low_diff * volume_delta
    
    # Incorporate Price Change
    # Calculate Close-to-Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Final Alpha Factor
    alpha_factor = close_to_close_return + volume_weighted_smoothed_diff
    
    return alpha_factor
```
