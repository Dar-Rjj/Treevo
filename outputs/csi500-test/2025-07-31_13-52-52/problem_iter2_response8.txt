```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    open_gap = df['open'] - df['close'].shift(1)
    adjusted_high_low_range = high_low_range - open_gap
    
    # Incorporate Volume Momentum
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_weighted_range = adjusted_high_low_range * volume_change
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_average = volume_weighted_range.rolling(window=lookback_period).mean()
    alpha_factor_momentum = volume_weighted_range - moving_average
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_range - high_low_range.shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    spread_momentum_lookback = 5
    high_low_spread_momentum = high_low_spread_change.rolling(window=spread_momentum_lookback).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume'] / df['volume'].expanding().sum()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    alpha_factor = vwap * high_low_spread_momentum * alpha_factor_momentum
    
    return alpha_factor
```
