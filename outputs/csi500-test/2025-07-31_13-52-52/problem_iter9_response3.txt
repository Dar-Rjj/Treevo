import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday_volume'] = df['intraday_return'] * df['volume_change_indicator']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['sma_combined'] = df['combined_intraday_volume'].rolling(window=short_period).mean()
    
    # Smooth the Result using Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['ema_combined'] = df['combined_intraday_volume'].ewm(alpha=smoothing_factor).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate 7-Day Momentum
    df['7_day_momentum'] = df['close'].diff().rolling(window=7).sum()
    
    # Combine and Adjust Factors
    df['adjusted_close_to_open'] = df['close_to_open_return'] * df['volume'] / df['daily_volatility']
    
    # Integrate Both Metrics
    weight_intraday = 0.6
    weight_close_to_open = 0.4
    df['final_alpha_factor'] = (weight_intraday * df['ema_combined'] + 
                                weight_close_to_open * df['adjusted_close_to_open'] - 
                                df['7_day_momentum'])
    
    return df['final_alpha_factor'].dropna()
