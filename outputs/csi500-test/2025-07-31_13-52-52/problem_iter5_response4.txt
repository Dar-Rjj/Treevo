```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine and Adjust Factors
    combined_factor = (intraday_return * volume_change_indicator * price_trend_indicator + 
                       close_to_open_return * df['volume'] / daily_volatility)
    
    # Smooth the Combined Result
    medium_sma = combined_factor.rolling(window=10).mean()
    smooth_factor = medium_sma.ewm(alpha=0.2).mean()
    
    return smooth_factor
```
