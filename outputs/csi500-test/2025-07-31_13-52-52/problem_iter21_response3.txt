```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()

    # Momentum signals
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).mean()
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).mean()
    df['momentum_crossover'] = df['short_term_momentum'] - df['long_term_momentum']

    # Volume-based indicators
    df['daily_volume_change'] = df['volume'].diff()
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_increase_ratio'] = df['volume'] / df['volume_ma_10']

    # Price volatility
    df['high_low_range'] = df['high'] - df['low']
    df['atr_14'] = df['high_low_range'].rolling(window=14).mean()
    df['volatility_ratio'] = df['high_low_range'] / df['atr_14']

    # Money flow
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['raw_money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['raw_money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    df['positive_money_flow_14'] = df['positive_money_flow'].rolling(window=14).sum()
    df['negative_money_flow_14'] = df['negative_money_flow'].rolling(window=14).sum()
    df['money_flow_ratio'] = df['positive_money_flow_14'] / df['negative_money_flow_14']

    # Price patterns
    df['engulfing_pattern'] = (df['open'] < df['close'].shift(1)) & (df['close'] > df['open'].shift(1))
    df['doji_pattern'] = (df['close'] - df['open']).abs() <= 0.1 * (df['high'] - df['low'])
    df['hammer_or_inverted_hammer'] = (df['low'] < df['close'].shift(1)) & (df['close'] > df['open']) & (df['close'] - df['open']) > 2 * (df['open'] - df['low'])

    # Trend indicators
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_above_sma_50'] = df['close'] > df['sma_50']
    df['ema_20_cross_sma_50'] = df['ema_20'] > df['sma_50']

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_crossover'] +
        df['volume_increase_ratio'] +
        df['volatility_ratio'] +
        df['money_flow_ratio'] +
        df['engulfing_pattern'].astype(int) +
        df['doji_pattern'].astype(int) +
        df['hammer_or_inverted_hammer'].astype(int) +
        df['price_above_sma_50'].astype(int) +
        df['ema_20_cross_sma_50'].astype(int)
    )

    return df['alpha_factor']
```
