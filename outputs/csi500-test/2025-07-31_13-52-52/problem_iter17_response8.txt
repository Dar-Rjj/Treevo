import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day standard deviation of the daily return
    df['std_30_daily_return'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the 10-day exponential moving average of the close price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the close price
    df['ema_30_close'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the difference between the 10-day and 30-day EMA of the close price
    df['ema_diff'] = df['ema_10_close'] - df['ema_30_close']
    
    # Calculate the 10-day rolling sum of the volume
    df['rolling_sum_10_volume'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day rolling mean of the amount
    df['rolling_mean_10_amount'] = df['amount'].rolling(window=10).mean()
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = (df['ma_10_daily_return'] / df['std_30_daily_return']) * df['ema_diff'] * (df['rolling_sum_10_volume'] / df['rolling_mean_10_amount'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
