import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Short SMA (5 days) and Long SMA (20 days) of Close prices
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    
    # Determine Price Trend Indicator
    df['price_trend_indicator'] = (df['short_sma'] > df['long_sma']).astype(int) * 2 - 1
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratios and High-Low Ranges
    df['rolling_sum_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['rolling_sum_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['rolling_sum_price_movement_ratio'] / df['rolling_sum_high_low_range']
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_indicator'] = (df['high'] - df['low']) * df['volume_change_indicator'] * df['price_trend_indicator']
    
    # Smooth the Result with Simple Moving Average (10 days)
    df['smoothed_combined_indicator'] = df['combined_indicator'].rolling(window=10).mean()
    
    # Adjust for Volume
    df['adjusted_momentum'] = df['smoothed_combined_indicator'] * df['volume']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factors'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Calculate 5-day Moving Average of Close Prices
    df['5_day_close_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['5_day_close_ma']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_momentum'] + df['combined_factors']) * df['relative_strength']
    
    return df['alpha_factor']
