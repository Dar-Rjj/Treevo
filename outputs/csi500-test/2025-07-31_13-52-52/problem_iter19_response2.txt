```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = df['volume'].diff()
    df['volume_change_indicator'] = np.where(df['volume_change'] > 0, 1, -1)
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday_vol'] = df['intraday_return'] * df['volume_change_indicator']
    df['smoothed_intraday_vol'] = df['combined_intraday_vol'].rolling(window=7).mean()
    
    # Calculate High-Low and Close-Open Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    df['5_day_rolling_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    df['adjusted_5_day_momentum'] = df['5_day_rolling_momentum'] * df['volume']
    
    # Calculate 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine and Adjust Price and Intraday Momentum
    df['combined_momentum'] = df['10_day_price_momentum'] + df['intraday_return']
    df['adjusted_combined_momentum'] = np.where(df['volume_change'] > 0, df['combined_momentum'], 0) * df['volume_change_indicator']
    
    # Calculate Averaged High-to-Low Volatility
    df['high_low_volatility'] = (df['high'] - df['low']) / df['open']
    df['avg_high_low_volatility'] = df['high_low_volatility'].rolling(window=20).sum() / 20
    
    # Integrate Both Metrics
    df['integrated_metric'] = (df['smoothed_intraday_vol'] + df['avg_high_low_volatility']) / 2
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_5_day_momentum'] + df['adjusted_combined_momentum']) * df['volume_change_indicator']
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=20, adjust=False).mean()
    
    return df['final_alpha_factor']
```
