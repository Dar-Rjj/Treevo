```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = (df['short_ema'] - df['long_ema']) / df['long_ema']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_momentum'] = (df['high_low_range'] * df['volume_change_indicator'] * df['price_trend']).ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['adjusted_combined_momentum'] = df['combined_momentum'] * np.log(df['volume'] + 1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Additional Factors
    df['additional_factors'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Calculate 5-day Moving Average of Close Prices
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['close'] - df['5_day_ma']) / df['5_day_ma']
    
    # Combine and Adjust Price and Intraday Momentum
    df['10_day_momentum'] = df['close'].pct_change(periods=10)
    df['momentum_intraday'] = df['10_day_momentum'] + df['intraday_return']
    df['adjusted_momentum_intraday'] = df['momentum_intraday'] * (df['volume_change_indicator'] > 0).astype(int)
    
    # Calculate Averaged High-to-Low Volatility
    df['high_low_volatility'] = (df['high'] - df['low']) / df['open']
    df['averaged_high_low_volatility'] = df['high_low_volatility'].rolling(window=20).mean()
    
    # Integrate Both Metrics
    df['integrated_metrics'] = (df['intraday_return'].ewm(span=10, adjust=False).mean() + df['averaged_high_low_volatility']) / 2
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_combined_momentum'] +
        df['additional_factors'] +
        (df['rolling_momentum'] * df['volume_change_indicator'] * df['price_trend']).ewm(span=10, adjust=False).mean()
    ) * df['relative_strength']
    
    return df['final_alpha_factor']
```
