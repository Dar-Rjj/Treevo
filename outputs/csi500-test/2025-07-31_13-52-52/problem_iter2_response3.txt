```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Percentage Change
    high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Define Lookback Period
    lookback_period = 20
    
    # Average Over a Lookback Period
    avg_high_low_pct_change = high_low_pct_change.rolling(window=lookback_period).mean()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_spread - high_low_spread.shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    momentum_high_low_spread_change = high_low_spread_change.rolling(window=5).sum()
    
    # Combine Averaged High-to-Low Momentum and High-Low Spread Momentum
    combined_momentum = avg_high_low_pct_change + momentum_high_low_spread_change
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * vwap
    
    return alpha_factor
```
