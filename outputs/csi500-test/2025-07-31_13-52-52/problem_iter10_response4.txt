```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors: Multiply Close-to-Open Return by Volume and divide by Daily Volatility
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum: Calculate the sum of the differences of Close prices of the last 5 days
    momentum_adjustment = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() * df['volume'].rolling(window=5).mean()
    
    # Final factor: Combine the combined factor with the momentum adjustment
    final_factor = combined_factor + momentum_adjustment
    
    return final_factor
```
