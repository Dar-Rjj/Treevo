```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_range
    
    # Adjust for Momentum
    momentum = (df['close'].diff().rolling(window=5).sum()) / df['volume'].rolling(window=5).mean()
    combined_factor_with_momentum = combined_factor * (1 + momentum)
    
    # Introduce Trend Sensitivity
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    trend_sensitivity = (ma_20 - ma_50) * 0.05
    trend_adjusted_factor = combined_factor_with_momentum * (1 + trend_sensitivity)
    
    return trend_adjusted_factor
```
