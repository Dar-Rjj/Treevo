import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Adjust for Opening Gap
    previous_close = df['close'].shift(1)
    gap_adjusted_range = high_low_range - (df['open'] - previous_close)

    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']

    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)

    # Combine Momentum Indicators
    combined_momentum = price_movement_ratio.rolling(window=5).sum() + price_momentum

    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    amount_change = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)

    # Multiply Combined Momentum by Combined Volume-Amount Momentum
    combined_volume_amount_momentum = volume_change * amount_change
    adjusted_momentum = combined_momentum * combined_volume_amount_momentum

    # Apply Time Series Momentum
    alpha_factor_ma = adjusted_momentum.rolling(window=10).mean()
    alpha_factor = adjusted_momentum - alpha_factor_ma

    # Final Alpha Factor
    alpha_factor = alpha_factor.where(volume_change > 0, 0)

    return alpha_factor
