```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = (df['short_ema'] - df['long_ema']) / df['long_ema']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_momentum_intraday_price_trend'] = (df['high'] - df['low']) * df['volume_change'] * df['price_trend']
    df['smoothed_combined_momentum_intraday_price_trend'] = df['combined_momentum_intraday_price_trend'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['adjusted_combined_momentum_intraday_price_trend'] = df['smoothed_combined_momentum_intraday_price_trend'] * np.log(df['volume'] + 1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = np.abs(df['high'] - df['low'])
    
    # Combine Additional Factors
    df['combined_additional_factors'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Calculate 5-day Moving Average of Close Prices
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['close'] - df['ma_5_close']) / df['ma_5_close']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_combined_momentum_intraday_price_trend'] + df['combined_additional_factors']) * df['relative_strength']
    
    return df['alpha_factor']
```
