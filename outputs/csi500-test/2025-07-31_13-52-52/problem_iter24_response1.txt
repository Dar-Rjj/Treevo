import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate 5-Day Rolling Momentum
    rolling_momentum = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    volume_adjusted_momentum = rolling_momentum * df['volume']
    
    # Calculate 10-Day Price Momentum
    ten_day_price_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    combined_momentum = ten_day_price_momentum + intraday_return
    volume_change_indicator = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    volume_adjusted_combined_momentum = combined_momentum * np.where(volume_change_indicator > 0, 1, 0)
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Price Trend Indicator
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_trend_indicator = np.where(short_ema > long_ema, 1, -1)
    
    # Combine Factors
    combined_factor = volume_adjusted_combined_momentum * price_trend_indicator
    
    # Final Alpha Factor
    final_alpha_factor = volume_adjusted_momentum + combined_factor
    final_alpha_factor = final_alpha_factor.rolling(window=10).mean() * 0.6 + final_alpha_factor.ewm(span=10, adjust=False).mean() * 0.4
    
    # Final Adjustment for Volume
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
