```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    combined_signal = intraday_return * volume_change
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    sma_smoothed = combined_signal.rolling(window=short_period).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    ema_smoothed = combined_signal.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    high_to_low_volatility = (df['high'] - df['low']) / df['low']
    avg_high_to_low_volatility = high_to_low_volatility.rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Weighted Average
    weight_intraday_momentum = 0.6
    weight_volatility = 0.4
    final_alpha_factor = (weight_intraday_momentum * ema_smoothed) + (weight_volatility * avg_high_to_low_volatility)
    
    return final_alpha_factor
```
