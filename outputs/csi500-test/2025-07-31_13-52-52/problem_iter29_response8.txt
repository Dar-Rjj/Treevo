import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['vol_adjusted_momentum'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_high_low_range'] = (df['high'] - df['low']) - df['opening_gap']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Combine Momentum Indicators
    df['combined_momentum'] = df['close_to_open_return'] + df['price_movement_ratio'].rolling(window=5).sum()
    
    # Incorporate Volume and Amount Momentum
    df['volume_change'] = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    df['amount_change'] = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    df['combined_volume_amount_momentum'] = (df['volume_change'] + df['amount_change'])
    df['volume_adjusted_momentum'] = df['combined_momentum'] * df['combined_volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['ts_momentum_ma'] = df['volume_adjusted_momentum'].rolling(window=20).mean()
    df['ts_momentum_diff'] = df['volume_adjusted_momentum'] - df['ts_momentum_ma']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] / df['low'] - 1
    
    # Compute 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) - (df['volume'] < df['volume'].shift(1)).astype(int)
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['short_sma'] > df['long_sma']).astype(int) - (df['short_sma'] < df['long_sma']).astype(int)
    
    # Combine Intraday and 10-Day Momentum
    df['intraday_10_day_momentum'] = df['intraday_return'] + df['10_day_price_momentum']
    
    # Adjust for Momentum
    df['momentum_adjustment'] = (df['close'].diff().rolling(window=5).sum()) * df['volume'].rolling(window=5).mean()
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['volume_adjusted_momentum'] * df['volume_change_indicator']
    
    # Integrate Both Metrics
    df['integrated_metrics'] = (df['adjusted_high_low_range'] + df['intraday_return']) / 2
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['ts_momentum_diff'] + df['integrated_metrics'] + df['momentum_adjustment']) / 3
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
