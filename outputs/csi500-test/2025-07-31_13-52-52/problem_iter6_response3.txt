```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']
    
    # Calculate 10-Day Volume Change
    df['volume_change'] = df['volume'] - df['volume'].rolling(window=10).apply(lambda x: x[:-1].mean(), raw=True)
    
    # Calculate 10-Day Amount Change
    df['amount_change'] = df['amount'] - df['amount'].rolling(window=10).apply(lambda x: x[:-1].mean(), raw=True)
    
    # Combine Price Momentums
    df['combined_momentum'] = df['price_momentum'] * df['open_to_close_momentum']
    
    # Adjust by Volume and Amount Trends
    df['alpha_factor'] = df.apply(
        lambda row: row['combined_momentum'] if row['volume_change'] > 0 and row['amount_change'] > 0 else 0, axis=1
    )
    
    return df['alpha_factor']
```
