```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    five_day_momentum = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust for Volume
    volume_adjusted_momentum = five_day_momentum * df['volume']
    
    # Compute 10-Day Price Momentum
    ten_day_price_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust 10-Day Price and Intraday Momentum
    combined_momentum = ten_day_price_momentum + intraday_return
    volume_change_indicator = (df['volume'] > df['volume'].shift()).astype(int) * 2 - 1
    combined_momentum_adjusted = combined_momentum * volume_change_indicator
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Factors
    close_open_return_volume = (df['close'] - df['open']) / df['open'] * df['volume']
    combined_factors = close_open_return_volume / daily_volatility
    
    # Final Alpha Factor
    final_alpha_factor = (volume_adjusted_momentum + combined_momentum_adjusted * price_trend_indicator).rolling(window=10).mean().ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift()
    volume_weighted_range = (df['high'] - df['low']) * volume_change
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_range - high_low_range.shift()
    
    # Calculate Momentum of High-Low Spread Change
    high_low_spread_momentum = high_low_spread_change.rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    combined_vwap_high_low = vwap * high_low_spread_momentum * volume_weighted_range
    
    # Final Alpha Factor
    final_alpha_factor += combined_vwap_high_low
    
    # Final Adjustment for Volume
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
```
