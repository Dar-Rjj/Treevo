```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low and Close-Open Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    df['rolling_5_day_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).mean()
    df['adjusted_5_day_momentum'] = df['rolling_5_day_momentum'] * np.log(df['volume'] + 1)

    # Compute 10-Day Price and Intraday Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['combined_momentum'] = df['10_day_price_momentum'] + df['intraday_return']

    # Adjust by Volume
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    df['adjusted_combined_momentum'] = df['combined_momentum'] * np.where(df['volume_change_indicator'] > 0, 1, 0)

    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])

    # Calculate Price Trend Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['trend_indicator'] = np.where(df['short_ema'] > df['long_ema'], 1, -1)

    # Combine Factors
    df['combined_factor'] = df['adjusted_combined_momentum'] * df['trend_indicator']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_5_day_momentum'] + df['combined_factor']
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()

    # Adjust for Volume
    df['final_alpha_factor'] = df['final_alpha_factor'] * np.log(df['volume'] + 1)

    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['final_alpha_factor'] = (df['final_alpha_factor'] * df['price_movement_ratio']) / df['daily_volatility']

    # Calculate Averaged High-to-Low Volatility
    df['high_to_low_volatility'] = (df['high'] - df['low']) / df['low']
    df['averaged_high_to_low_volatility'] = df['high_to_low_volatility'].rolling(window=20).mean()

    # Calculate 5-day Moving Average of Close Prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()

    # Calculate Relative Strength
    df['relative_strength'] = (df['close'] - df['5_day_ma_close']) / df['5_day_ma_close']

    # Enhance Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['trend_indicator']

    # Incorporate Long-Term Volatility
    df['20_day_volatility'] = df['close'].rolling(window=20).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / (df['20_day_volatility'] + 1e-6)

    # Integrate Recent Volume Trend
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['vwap'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['5_day_ema_volume'] * df['vwap']

    return df['final_alpha_factor']
```
