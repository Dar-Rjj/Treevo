```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    open_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    adjusted_high_low_range = high_low_range * volume_change
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_avg = adjusted_high_low_range.rolling(window=lookback_period).mean()
    ts_momentum_diff = adjusted_high_low_range - moving_avg
    
    # Calculate 5-Day Rolling Momentum
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    five_day_rolling_momentum = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust Momentum for Volume
    volume_adjusted_momentum = five_day_rolling_momentum * df['volume']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    close_open_return = (df['close'] - df['open']) / df['open']
    daily_volatility = (df['high'] - df['low']).abs()
    final_factor = (volume_adjusted_momentum * close_open_return) / daily_volatility
    
    # Combine Close-to-Open Return with Volume
    combined_close_open_return = (close_open_return * df['volume']) / daily_volatility
    
    # Volume Adjusted Momentum
    close_prices_last_5_days = df['close'].diff().rolling(window=5).sum()
    avg_volume_last_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum_final = close_prices_last_5_days * avg_volume_last_5_days
    
    # Final Alpha Factor
    final_alpha_factor = (ts_momentum_diff + volume_adjusted_momentum_final + combined_close_open_return) * df['volume'].rolling(window=5).mean()
    
    # Incorporate Long-Term Volatility
    long_term_volatility = df['close'].rolling(window=20).std()
    final_alpha_factor *= 1 / long_term_volatility
    
    # Integrate Recent Volume Trend
    ema_volume_5_days = df['volume'].ewm(span=5).mean()
    final_alpha_factor *= ema_volume_5_days
    
    # Incorporate Short-Term Price Trend
    short_term_trend = df['close'] - df['close'].rolling(window=5).mean()
    final_alpha_factor *= short_term_trend
    
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    final_alpha_factor *= rsi
    
    # Enhance with Accumulation/Distribution Line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adl = money_flow_volume.cumsum()
    final_alpha_factor *= adl
    
    return final_alpha_factor
```
