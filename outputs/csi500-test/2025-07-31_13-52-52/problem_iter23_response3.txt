import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is a combination of momentum and volatility, using historical price and volume data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor, indexed by (date).
    """
    # Calculate the 20-day simple moving average of the closing price
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['std_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the 5-day simple moving average of the volume
    df['sma_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Momentum factor: 10-day rate of change in the closing price
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Volatility factor: 20-day standard deviation of the closing price
    df['volatility_20'] = df['std_20']
    
    # Volume factor: 5-day simple moving average of the volume
    df['volume_factor_5'] = df['sma_volume_5']
    
    # Alpha factor: combination of momentum and volatility, weighted by volume
    df['alpha_factor'] = (df['momentum_10'] * df['volume_factor_5']) / (df['volatility_20'] + 1e-6)
    
    return df['alpha_factor']
