import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Adjust for Opening Gap
    opening_gap = df['open'].shift(-1) - df['close']

    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']

    # Incorporate Volume Momentum
    avg_volume = df['volume'].rolling(window=10).mean().shift(1)
    volume_change = df['volume'] / avg_volume
    adjusted_range = high_low_range * volume_change

    # Apply Time Series Momentum
    lookback_period = 10
    moving_avg = adjusted_range.rolling(window=lookback_period).mean()
    ts_momentum = adjusted_range - moving_avg

    # Calculate 5-Day Rolling Momentum
    rolling_price_movement_ratio = price_movement_ratio.rolling(window=5).sum()
    rolling_high_low_range = high_low_range.rolling(window=5).sum()
    rolling_momentum = rolling_price_movement_ratio / rolling_high_low_range

    # Adjust Momentum for Volume
    volume_adjusted_momentum = rolling_momentum * df['volume']

    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    close_to_open_return = (df['close'] - df['open']) / df['open']
    daily_volatility = (df['high'] - df['low']).abs()
    vol_adj_factor = (volume_adjusted_momentum * close_to_open_return) / daily_volatility

    # Combine Close-to-Open Return with Volume
    combined_close_open_vol = (close_to_open_return * df['volume']) / daily_volatility

    # Calculate Averaged High-to-Low Volatility
    avg_high_low_volatility = (df['high'] - df['low']).abs().rolling(window=lookback_period).mean()

    # Integrate Both Metrics
    weights = [0.5, 0.5]
    interim_alpha_factor = (weights[0] * vol_adj_factor + weights[1] * combined_close_open_vol) / avg_high_low_volatility

    # Calculate 5-day Moving Average of Close Prices
    ma_5_close = df['close'].rolling(window=5).mean()
    relative_strength = df['close'] / ma_5_close

    # Final Alpha Factor
    final_alpha_factor = interim_alpha_factor * relative_strength

    # Calculate 20-day Moving Average of Close Prices
    ma_20_close = df['close'].rolling(window=20).mean()
    long_term_trend = df['close'] / ma_20_close

    # Enhance Final Alpha Factor
    final_alpha_factor *= long_term_trend

    # Incorporate Long-Term Volatility
    long_term_volatility = df['close'].rolling(window=20).std()
    final_alpha_factor *= 1 / long_term_volatility

    # Integrate Recent Volume Trend
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    final_alpha_factor *= ema_5_volume

    return final_alpha_factor
