```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() * df['volume']
    
    # Final Alpha Factor
    alpha_factor = combined_factor + momentum
    
    return alpha_factor
```
