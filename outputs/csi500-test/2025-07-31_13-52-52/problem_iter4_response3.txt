```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate 10-Day Volume Change
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean().shift(1)
    df['volume_change'] = df['volume'] - df['avg_volume_10']
    
    # Combine Momentum and Volume Trends
    df['factor'] = df.apply(lambda row: row['price_momentum'] if row['volume_change'] > 0 else 0, axis=1)
    
    return df['factor']
```
