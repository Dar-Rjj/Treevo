import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate the high-to-low price percentage change
    df['high_to_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the rolling average of the high-to-low percentage change over the lookback period
    df['avg_momentum'] = df['high_to_low_pct_change'].rolling(window=lookback_period).mean()
    
    # Return the factor values as a pandas Series
    return df['avg_momentum']
