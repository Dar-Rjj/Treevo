```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    # Use Open price and Close price (previous day)
    df['prev_close'] = df['close'].shift(1)
    opening_gap = df['open'] - df['prev_close']
    adjusted_range = high_low_range - opening_gap
    
    # Incorporate Volume Momentum
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_weighted_range = adjusted_range * volume_change
    
    # Apply Time Series Momentum
    # Calculate Alpha Factor's Moving Average over a specified lookback period (e.g., 10 days)
    lookback_period = 10
    moving_average = volume_weighted_range.rolling(window=lookback_period).mean()
    
    # Calculate Difference between Current Value and Moving Average
    alpha_factor = volume_weighted_range - moving_average
    
    return alpha_factor
```
