import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided market data.
    The factor is a combination of momentum, volatility, and volume-based indicators.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as 'date'.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor for each date.
    """
    # Calculate the 10-day simple moving average of the close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day simple moving average of the close price
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day standard deviation of the close price
    df['std_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the close price
    df['std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the 10-day average daily return
    df['avg_ret_10'] = df['close'].pct_change().rolling(window=10).mean()
    
    # Calculate the 30-day average daily return
    df['avg_ret_30'] = df['close'].pct_change().rolling(window=30).mean()
    
    # Calculate the 10-day average volume
    df['avg_vol_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day average volume
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] / df['sma_30'] - 1
    
    # Calculate the volatility factor
    df['volatility'] = (df['std_10'] / df['std_30']).apply(lambda x: 1 if x > 1 else 0)
    
    # Calculate the volume factor
    df['volume_factor'] = (df['avg_vol_10'] / df['avg_vol_30']).apply(lambda x: 1 if x > 1.5 else 0)
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['momentum'] * df['volatility'] * df['volume_factor'])
    
    return df['alpha_factor']
