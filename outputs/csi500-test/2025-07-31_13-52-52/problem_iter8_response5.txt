import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    opening_gap_adjusted = (df['open'] - df['close'].shift(1)) / (df['high'] - df['low'])
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    momentum_indicators = close_to_open_return + price_movement_ratio.rolling(window=5).sum()
    
    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    amount_change = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    combined_momentum = (volume_change + amount_change) / 2
    
    # Multiply Combined Momentum by Combined Volume-Amount Momentum
    adjusted_momentum = momentum_indicators * combined_momentum
    
    # Apply Time Series Momentum
    alpha_factor_moving_avg = adjusted_momentum.rolling(window=10).mean()
    alpha_factor = adjusted_momentum - alpha_factor_moving_avg
    
    # Final Alpha Factor
    final_alpha_factor = alpha_factor.where(volume_change > 0, 0)
    
    return final_alpha_factor
