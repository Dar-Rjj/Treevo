import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    ten_day_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    combined_momentum = ten_day_momentum + intraday_return
    
    # Apply Volume Confirmation
    combined_momentum_with_volume = combined_momentum * (volume_change_indicator > 0)
    
    # Calculate Averaged High-to-Low Volatility
    high_low_volatility = (df['high'] - df['low']).pct_change().rolling(window=10).mean()
    
    # Integrate Both Metrics
    integrated_metrics = (high_low_volatility + intraday_return.rolling(window=10).mean()) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (combined_momentum_with_volume + integrated_metrics) / 2
    
    # Smooth the Result
    smoothed_final_alpha_factor = final_alpha_factor.ewm(span=10, adjust=False).mean()
    
    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].shift(1).rolling(window=9).mean()
    amount_change = df['amount'] - df['amount'].shift(1).rolling(window=9).mean()
    combined_volume_amount_momentum = (volume_change + amount_change) * (df['high'] - df['low']) / df['close']
    
    # Apply Time Series Momentum
    alpha_factor_moving_avg = smoothed_final_alpha_factor.rolling(window=10).mean()
    time_series_momentum = smoothed_final_alpha_factor - alpha_factor_moving_avg
    
    # Final Alpha Factor
    final_alpha_factor = time_series_momentum * (volume_change_indicator > 0)
    
    return final_alpha_factor
