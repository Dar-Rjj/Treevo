```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['prev_close'] = df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] - (df['open'] - df['prev_close'])
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['adjusted_high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['vwr_moving_avg'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['ts_momentum'] = df['volume_weighted_range'] - df['vwr_moving_avg']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate High-to-Low Price Percentage Change
    df['high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Average Over a Lookback Period
    df['avg_high_low_pct_change'] = df['high_low_pct_change'].rolling(window=lookback_period).mean()
    
    # Combine Averages and Momentum
    df['combined_averages_momentum'] = (df['price_movement_ratio'].rolling(window=5).sum() 
                                        / df['high_low_range'].rolling(window=5).sum()) * df['avg_high_low_pct_change']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(lookback_period)
    
    # Combine Momentum Indicators
    df['combined_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() + df['price_momentum']
    
    # Adjust for Volume
    df['avg_volume'] = df['volume'].rolling(window=9).mean().shift(1)
    df['volume_change_10d'] = df['volume'] - df['avg_volume']
    
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_change_10d']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Final Alpha Factor Combination
    conditions = [
        (df['volume_change_10d'] > 0),
        (df['volume_change_10d'] <= 0)
    ]
    choices = [
        df['vwap'] * df['ts_momentum'] * df['adjusted_combined_momentum'] * df['avg_high_low_pct_change'],
        0
    ]
    df['alpha_factor'] = pd.np.select(conditions, choices, default=0)
    
    return df['alpha_factor']
```
