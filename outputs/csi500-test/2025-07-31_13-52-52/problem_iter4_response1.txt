```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    volume_change = df['volume'].diff()
    volume_indicator = volume_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine and Smooth
    combined_factor = intraday_return * volume_indicator
    
    # Define Smoothing Factor
    smoothing_factor = 0.5
    
    # Apply Exponential Moving Average
    ema_factor = combined_factor.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return ema_factor
```
