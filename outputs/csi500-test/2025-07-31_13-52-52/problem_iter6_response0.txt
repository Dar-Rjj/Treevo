```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Cumulative Price Change
    df['cumulative_price_change'] = df['close'] - df['close'].shift(20)
    
    # Calculate Weighted Volume Trend
    df['weighted_volume_trend'] = df['volume'] * (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust Cumulative Price Change by Weighted Volume Trend
    df['adjusted_momentum'] = df['cumulative_price_change'] * df['weighted_volume_trend']
    
    # Smooth the Adjusted Momentum with 5-day Exponential Moving Average
    df['smoothed_adjusted_momentum'] = df['adjusted_momentum'].ewm(span=5, min_periods=5).mean()
    
    return df['smoothed_adjusted_momentum']
```
