import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day EMA of the money flow
    ema_money_flow = money_flow.ewm(span=14).mean()
    
    # Calculate the 14-day EMA of the volume
    ema_volume = df['volume'].ewm(span=14).mean()
    
    # Calculate the normalized money flow
    normalized_money_flow = ema_money_flow / ema_volume
    
    # Calculate the 10-day rolling sum of the normalized money flow
    rolling_sum_normalized_money_flow = normalized_money_flow.rolling(window=10).sum()
    
    # Calculate the final factor
    factor = rolling_sum_normalized_money_flow / df['close']
    
    return factor
