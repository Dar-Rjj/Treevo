```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate Rate of Change (ROC)
    df['ROC_1'] = df['close'].pct_change(periods=1)
    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_10'] = df['close'].pct_change(periods=10)

    # Combine SMAs and ROCs for momentum factor
    df['Momentum_Factor'] = (df['SMA_5'] + df['SMA_10']) / 2 * (df['ROC_1'] + df['ROC_5'])

    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['TR'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1) \
        .combine(df['high'] - df['prev_close'], max) \
        .combine(df['prev_close'] - df['low'], max)

    # Calculate Average True Range (ATR)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    df['Volatility_Factor'] = df['ATR_14'] / df['ATR_14']

    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))

    # RSI conditions for overbought and oversold
    df['Overbought'] = df['RSI_14'] > 70
    df['Oversold'] = df['RSI_14'] < 30

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0.0
    df.loc[df['close'] > df['close'].shift(1), 'OBV'] += df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'OBV'] -= df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Calculate Volume Factor
    df['SMA_VWAP_20'] = df['VWAP'].rolling(window=20).mean()
    df['Volume_Factor'] = df['VWAP'] / df['SMA_VWAP_20']

    # Combine momentum, volatility, and volume factors into a comprehensive alpha factor
    df['Alpha_Factor'] = df['Momentum_Factor'] * df['Volatility_Factor'] * df['Volume_Factor']

    return df['Alpha_Factor']
```
