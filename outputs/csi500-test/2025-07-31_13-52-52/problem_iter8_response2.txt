```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low and Close-Open Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Price_Movement_Ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # 5-Day Rolling Momentum
    df['5_Day_Rolling_Momentum'] = df['Price_Movement_Ratio'].rolling(window=5).sum() / df['High_Low_Range'].rolling(window=5).sum()
    df['5_Day_Adjusted_Momentum'] = df['5_Day_Rolling_Momentum'] * df['volume']
    
    # 10-Day Price Momentum
    df['10_Day_Price_Momentum'] = df['close'] - df['close'].shift(10)
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    df['Combined_Momentum'] = df['10_Day_Price_Momentum'] + df['Intraday_Return']
    df['Volume_Change_Indicator'] = (df['volume'] > df['volume'].shift()).astype(int) * 2 - 1
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] * (df['Volume_Change_Indicator'] * (df['volume'] > df['volume'].shift()))
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['Close_to_Open_Return_by_Volume'] = (df['close'] - df['open']) * df['volume']
    df['Adjusted_Factor'] = df['Close_to_Open_Return_by_Volume'] / df['Daily_Volatility']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['5_Day_Adjusted_Momentum'] + df['Adjusted_Combined_Momentum'] + df['Adjusted_Factor']
    
    return df['Final_Alpha_Factor']
```
