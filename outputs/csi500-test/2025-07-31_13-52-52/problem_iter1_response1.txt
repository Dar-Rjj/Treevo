```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_factor'] = df['intraday_return'] * df['volume_change_indicator']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['sma_smoothed'] = df['combined_factor'].rolling(window=short_period, min_periods=1).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['ema_smoothed'] = df['combined_factor'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Return the final smoothed factor
    return df['ema_smoothed']
```
