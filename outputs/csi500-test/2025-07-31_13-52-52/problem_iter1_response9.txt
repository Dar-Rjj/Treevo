```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate High-Low Spread Change
    df['high_low_spread_change'] = df['high_low_spread'] - df['high_low_spread'].shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    df['high_low_spread_momentum'] = df['high_low_spread_change'].rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine VWAP and High-Low Spread Momentum
    df['alpha_factor'] = df['vwap'] * df['high_low_spread_momentum']
    
    return df['alpha_factor']
```
