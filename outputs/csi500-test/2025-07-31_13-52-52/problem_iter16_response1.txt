```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Compute Volume Change Indicator
    volume_change = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Combine Intraday Return and Volume Change
    combined_intraday = intraday_return * volume_change
    
    # Define Smoothing Factor
    smoothing_factor = 0.5
    
    # Apply Exponential Moving Average
    smoothed_intraday = combined_intraday.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Volatility
    high_low_volatility = (df['high'] / df['low'] - 1).rolling(window=20).mean()
    
    # Define Weights for Each Metric
    weight_smoothed_intraday = 0.6
    weight_high_low_volatility = 0.4
    
    # Calculate Weighted Sum
    final_alpha_factor = (weight_smoothed_intraday * smoothed_intraday +
                          weight_high_low_volatility * high_low_volatility)
    
    return final_alpha_factor
```
