```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low and Close-Open Momentum
    high_low_range = df['high'] - df['low']
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    rolling_momentum = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    adjusted_rolling_momentum = rolling_momentum * df['volume']

    # Calculate 10-Day Price and Intraday Momentum
    ten_day_price_momentum = df['close'].diff(periods=10)
    intraday_return = (df['high'] - df['low']) / df['low']
    combined_momentum = ten_day_price_momentum + intraday_return

    # Adjust by Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    adjusted_combined_momentum = combined_momentum * volume_change_indicator

    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()

    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1

    # Combine Factors
    combined_factor = adjusted_combined_momentum * price_trend_indicator

    # Final Alpha Factor
    final_alpha_factor = adjusted_rolling_momentum + combined_factor
    final_alpha_factor = final_alpha_factor.rolling(window=10).mean().ewm(span=10, adjust=False).mean()

    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].rolling(window=9).mean()
    amount_change = df['amount'] - df['amount'].rolling(window=9).mean()
    combined_volume_amount_momentum = (volume_change + amount_change) * high_low_range
    final_alpha_factor = final_alpha_factor * combined_volume_amount_momentum

    # Final Adjustment for Volume
    final_alpha_factor = final_alpha_factor * df['volume']

    return final_alpha_factor
```
