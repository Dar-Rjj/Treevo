```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Momentum Indicator
    momentum_indicator = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    adjusted_momentum_indicator = momentum_indicator * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] / df['open'] - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    df['prev_close'] = df['close'].shift(1)
    adjusted_high_low_range = high_low_range - (df['open'] - df['prev_close'])
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_range_with_volume = adjusted_high_low_range * volume_change
    
    # Apply Time Series Momentum
    alpha_factor_moving_avg = adjusted_range_with_volume.rolling(window=10).mean()
    time_series_momentum = adjusted_range_with_volume - alpha_factor_moving_avg
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_range - high_low_range.shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    spread_change_momentum = high_low_spread_change.rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['open'] + df['high'] + df['low'] + df['close']) / 4) * df['volume']
    vwap = vwap.cumsum() / df['volume'].cumsum()
    
    # Introduce Price Trend Component
    price_trend = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    combined_vwap_spread = vwap * spread_change_momentum * adjusted_momentum_indicator
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum_indicator + combined_factor + combined_vwap_spread + price_trend)
    
    # Final Adjustment for Volume
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
```
