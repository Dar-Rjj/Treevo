```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'].diff(10)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate 10-Day Volume Change
    df['10_day_volume_change'] = df['volume'] - df['volume'].rolling(window=10).mean().shift(1)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Price and Intraday Momentum
    df['combined_momentum'] = df['10_day_price_momentum'] + df['intraday_return']
    
    # Apply Volume Confirmation
    df['result'] = df['combined_momentum'] * df['volume_change_indicator']
    df['result'] = df.apply(lambda row: row['result'] if row['10_day_volume_change'] > 0 else 0, axis=1)
    
    return df['result']
```
