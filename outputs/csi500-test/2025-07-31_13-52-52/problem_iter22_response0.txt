import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Compute 5-day Rolling Sum of Price Movement Ratio and High-Low Range
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low = high_low_range.rolling(window=5).sum()
    
    # Compute Momentum Indicator
    momentum_indicator = rolling_sum_price_movement / rolling_sum_high_low
    
    # Adjust for Volume
    adjusted_momentum = momentum_indicator * df['volume']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_volume = intraday_return * volume_change
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    smoothed_combined = combined_intraday_volume.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (intraday_return * df['volume']) / daily_volatility
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Integrate Both Momentum Indicators
    integrated_momentum = rolling_sum_price_movement + close_to_open_return
    
    # Incorporate Volume and Amount Momentum
    avg_volume_9_days = df['volume'].rolling(window=10).mean().shift(1)
    avg_amount_9_days = df['amount'].rolling(window=10).mean().shift(1)
    volume_change_abs = df['volume'] - avg_volume_9_days
    amount_change = df['amount'] - avg_amount_9_days
    volume_amount_momentum = (volume_change_abs + amount_change) * high_low_range
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_avg_alpha = volume_amount_momentum.rolling(window=lookback_period).mean()
    ts_momentum = volume_amount_momentum - moving_avg_alpha
    
    # Final Alpha Factor
    final_alpha_factor = ts_momentum.where(volume_change > 0, 0)
    
    return final_alpha_factor
