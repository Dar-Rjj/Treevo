```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['volume_adjusted_clo'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    df['momentum_5d'] = (df['close'] - df['close'].shift(5)).rolling(window=5).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    df['price_momentum_10d'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['sma_5'] > df['sma_20']).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['intraday_return'] + df['price_momentum_10d']
    
    # Apply Volume Confirmation
    df['combined_momentum_volume_confirmed'] = df['combined_momentum'] * df['volume_change_indicator']
    df['combined_momentum_volume_confirmed'] = df['combined_momentum_volume_confirmed'].where(df['volume_change_indicator'] > 0, 0)
    
    # Integrate Both Metrics
    df['integrated_metrics'] = 0.5 * (df['momentum_5d'] + df['combined_momentum_volume_confirmed'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = 0.5 * (df['integrated_metrics'] + df['volume_adjusted_clo'])
    
    # Smooth the Result
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['final_alpha_factor_smoothed']
```
