```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Generate Momentum Indicator
    momentum_indicator = (price_movement_ratio.rolling(window=5).sum() * 
                          high_low_range.rolling(window=5).mean())
    
    # Integrate Volume into Momentum
    volume_weighted_momentum = momentum_indicator * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    adjusted_combined_factor = combined_factor - opening_gap
    
    # Apply Time Series Momentum
    moving_average = adjusted_combined_factor.rolling(window=10).mean()
    alpha_factor = adjusted_combined_factor - moving_average
    
    return alpha_factor
```
