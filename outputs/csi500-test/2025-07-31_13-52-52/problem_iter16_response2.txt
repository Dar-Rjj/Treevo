```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    volume_change = df['volume'].diff()
    volume_change_indicator = volume_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday Return and Volume Change
    combined_factor = intraday_return * volume_change_indicator
    smoothed_factor = combined_factor.rolling(window=5).mean()
    
    # Calculate Averaged High-to-Low Volatility
    high_low_volatility = (df['high'] - df['low']) / df['low']
    lookback_period = 20
    averaged_high_low_volatility = high_low_volatility.rolling(window=lookback_period).sum() / lookback_period
    
    # Integrate Both Metrics using Weighted Average
    weight_smoothed_factor = 0.6
    weight_averaged_high_low_volatility = 0.4
    final_alpha_factor = (weight_smoothed_factor * smoothed_factor) + (weight_averaged_high_low_volatility * averaged_high_low_volatility)
    
    return final_alpha_factor
```
