```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate 10-Day Volume Change
    df['avg_volume_10d'] = df['volume'].rolling(window=10, min_periods=9).mean().shift(1)
    df['volume_change'] = df['volume'] - df['avg_volume_10d']
    
    # Combine Momentum and Volume Trends
    df['combined_factor'] = df['price_momentum'] * df['volume_change']
    df['final_factor'] = df.apply(lambda row: row['combined_factor'] if row['volume_change'] > 0 else 0, axis=1)
    
    return df['final_factor']
```
