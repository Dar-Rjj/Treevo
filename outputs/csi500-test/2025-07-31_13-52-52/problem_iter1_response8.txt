```python
import pandas as pd

def heuristics_v2(df):
    n = 20
    m = 5
    
    # Calculate Cumulative Price Change
    cumulative_price_change = df['close'] - df['close'].shift(n)
    
    # Calculate Volume Trend
    average_volume = df['volume'].rolling(window=n).mean()
    volume_trend = df['volume'] - average_volume
    
    # Adjust Cumulative Price Change by Volume Trend
    adjusted_momentum = cumulative_price_change * volume_trend
    
    # Smooth the Adjusted Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=m).mean()
    
    return smoothed_momentum
```
