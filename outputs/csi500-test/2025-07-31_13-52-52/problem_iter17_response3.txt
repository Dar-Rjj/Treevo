```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Compute 10-Day Price Momentum
    price_momentum = df['close'].shift(1) - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    combined_momentum = price_momentum + intraday_return
    
    # Apply Volume Confirmation
    combined_momentum_with_volume = combined_momentum * volume_change_indicator
    combined_momentum_with_volume[(df['volume'] <= df['volume'].shift(1))] = 0
    
    # Calculate Daily Volatility
    daily_volatility = np.abs(df['high'] - df['low'])
    
    # Adjust Factors
    adjusted_combined_momentum = combined_momentum_with_volume / daily_volatility
    
    # Integrate Both Metrics
    intermediate_alpha_factor = (intraday_return + adjusted_combined_momentum) / 2
    
    # Smooth the Result
    smoothed_momentum = intermediate_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Momentum
    close_price_diff_sum = df['close'].diff().rolling(window=5).sum()
    average_volume_last_5_days = df['volume'].rolling(window=5).mean()
    momentum_adjustment = close_price_diff_sum / average_volume_last_5_days
    
    # Combine Close-to-Open Factors
    close_to_open_factor = (close_to_open_return * df['volume']) / daily_range
    
    # Final Integration
    final_alpha_factor = (smoothed_momentum + close_to_open_factor) / 2
    
    return final_alpha_factor
```
