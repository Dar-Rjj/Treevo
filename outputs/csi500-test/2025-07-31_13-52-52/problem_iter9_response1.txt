```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_return'] = df['intraday_return'] * df['volume_change_indicator']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    short_period = 5
    df['sma_combined_return'] = df['combined_return'].rolling(window=short_period).mean()
    
    # Further smooth using Exponential Moving Average
    smoothing_factor = 0.2
    df['ema_combined_return'] = df['sma_combined_return'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    df['high_low_volatility'] = (df['high'] - df['low']) / df['low']
    df['averaged_high_low_volatility'] = df['high_low_volatility'].rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Unweighted Average
    df['alpha_factor'] = (df['ema_combined_return'] + df['averaged_high_low_volatility']) / 2
    
    return df['alpha_factor']
```
