```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    factor_1 = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum
    momentum_adjustment = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] / df['low']) - 1
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    intraday_combined = intraday_return * volume_change_indicator
    smoothed_intraday = intraday_combined.ewm(span=5, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Momentum
    lookback_period = 5
    high_to_low_momentum = (df['high'] / df['low'] - 1).rolling(window=lookback_period).sum() / lookback_period
    
    # Integrate Both Metrics
    integrated_metrics = (smoothed_intraday + high_to_low_momentum) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (factor_1 + momentum_adjustment + integrated_metrics) / 3
    
    return final_alpha_factor
```
