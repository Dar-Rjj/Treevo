import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    open_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Incorporate Volume Momentum
    average_volume = df['volume'].rolling(window=10).mean()
    volume_momentum = (df['volume'] - average_volume) / average_volume
    
    # Apply Time Series Momentum
    adjusted_range = high_low_range * volume_change_indicator
    ma_adjusted_range = adjusted_range.rolling(window=10).mean()
    time_series_momentum = adjusted_range - ma_adjusted_range
    
    # Calculate 5-Day Rolling Momentum
    rolling_price_movement_ratio = price_movement_ratio.rolling(window=5).sum()
    rolling_high_low_range = high_low_range.rolling(window=5).sum()
    five_day_rolling_momentum = rolling_price_movement_ratio / rolling_high_low_range
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    combined_momentum = (df['high'] - df['low']) * volume_change_indicator * price_movement_ratio
    smoothed_combined_momentum = combined_momentum.ewm(span=10).mean()
    
    # Adjust for Volume
    log_volume = np.log(df['volume'] + 1)
    volume_adjusted_momentum = smoothed_combined_momentum * log_volume
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    daily_volatility = (df['high'] - df['low']).abs()
    close_open_return = (df['close'] - df['open']) / df['open']
    adjusted_final_factor = (volume_adjusted_momentum * close_open_return) / daily_volatility
    
    # Calculate Averaged High-to-Low Volatility
    high_low_volatility = (df['high'] - df['low']).abs()
    avg_high_low_volatility = high_low_volatility.rolling(window=20).mean()
    
    # Calculate 5-day Moving Average of Close Prices
    close_5d_ma = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    relative_strength = (df['close'] - close_5d_ma) / close_5d_ma
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_final_factor * relative_strength
    
    # Calculate 20-day Moving Average of Close Prices
    close_20d_ma = df['close'].rolling(window=20).mean()
    
    # Calculate Long-Term Trend Indicator
    long_term_trend = df['close'] / close_20d_ma
    
    # Enhance Final Alpha Factor
    enhanced_final_alpha_factor = final_alpha_factor * long_term_trend
    
    # Incorporate Long-Term Volatility
    long_term_volatility = df['close'].rolling(window=20).std()
    inverse_volatility = 1 / long_term_volatility
    final_alpha_with_volatility = enhanced_final_alpha_factor * inverse_volatility
    
    # Integrate Recent Volume Trend
    ema_volume = df['volume'].ewm(span=5).mean()
    final_alpha = final_alpha_with_volatility * ema_volume
    
    return final_alpha
