```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['short_sma'] > df['long_sma']).astype(int) * 2 - 1
    
    # Combine Price and Intraday Momentum
    df['combined_momentum'] = df['10_day_momentum'] + df['intraday_return']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['combined_momentum'] * df['volume_change_indicator']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['volume_confirmed_momentum'] * df['price_trend_indicator']
    
    # Smooth the Result
    df['smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    return df['smoothed_alpha_factor']
```
