```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['momentum_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] / df['close'].shift(1) - 1
    df['adjusted_momentum_factor'] = df['momentum_factor'] - df['opening_gap']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['adjusted_momentum_factor'] * df['volume_change_indicator']
    df['volume_confirmed_momentum'] = df['volume_confirmed_momentum'] * (df['volume_change_indicator'] > 0)
    
    # Adjust for Momentum
    df['5_day_momentum'] = (df['close'].diff().rolling(window=5).sum()) * df['volume'].rolling(window=5).mean()
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['10_day_momentum'] + df['intraday_return']
    
    # Incorporate Volume and Amount Momentum
    avg_volume = df['volume'].rolling(window=9).mean().shift(1)
    avg_amount = df['amount'].rolling(window=9).mean().shift(1)
    
    df['volume_change'] = df['volume'] - avg_volume
    df['amount_change'] = df['amount'] - avg_amount
    
    df['volume_amount_momentum'] = (df['volume_change'] + df['amount_change']) / df['high_low_range']
    
    # Integrate Both Metrics
    df['integrated_momentum'] = 0.5 * df['combined_momentum'] + 0.5 * df['volume_amount_momentum']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = 0.5 * df['volume_confirmed_momentum'] + 0.5 * df['integrated_momentum']
    
    # Apply Time Series Momentum
    final_alpha_factor_ma = final_alpha_factor.rolling(window=20).mean()
    final_alpha_factor_ts = final_alpha_factor - final_alpha_factor_ma
    
    # Smooth the Result
    final_alpha_factor_smoothed = final_alpha_factor_ts.rolling(window=20).mean()
    
    # Final Alpha Factor
    final_alpha_factor_smoothed = final_alpha_factor_smoothed * (df['volume_change_indicator'] > 0)
    
    return final_alpha_factor_smoothed
```
