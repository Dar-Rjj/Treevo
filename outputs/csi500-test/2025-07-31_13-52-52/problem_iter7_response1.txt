```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate 5-day rolling sum of Price Movement Ratios
    df['rolling_price_movement_sum'] = df['price_movement_ratio'].rolling(window=5).sum()
    
    # Calculate 5-day rolling sum of High-Low Ranges
    df['rolling_high_low_range_sum'] = df['high_low_range'].rolling(window=5).sum()
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['rolling_price_movement_sum'] / df['rolling_high_low_range_sum']
    
    # Adjust for Volume
    df['adjusted_momentum'] = df['momentum_indicator'] * df['volume']
    
    # Return the factor values as a pandas Series
    return df['adjusted_momentum']
```
