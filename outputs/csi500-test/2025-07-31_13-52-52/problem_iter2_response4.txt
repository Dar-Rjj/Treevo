import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] / df['open']) - 1
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Indicators
    combined_momentum = price_movement_ratio.rolling(window=5).sum() + price_momentum
    
    # Adjust for Volume
    volume_change = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    
    # Multiply Combined Momentum by Volume Change
    adjusted_momentum = combined_momentum * volume_change
    
    # Final Alpha Factor
    alpha_factor = adjusted_momentum.where(volume_change > 0, 0)
    
    return alpha_factor
