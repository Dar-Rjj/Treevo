```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate 5-Day Rolling Momentum
    rolling_momentum = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    volume_adjusted_momentum = rolling_momentum * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    adjusted_range = high_low_range - opening_gap
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_weighted_range = adjusted_range * volume_change
    
    # Apply Time Series Momentum
    alpha_factor_ma = volume_weighted_range.rolling(window=10).mean()
    alpha_factor_diff = volume_weighted_range - alpha_factor_ma
    
    # Calculate 10-Day Price and Intraday Momentum
    ten_day_price_momentum = df['close'] - df['close'].shift(10)
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    combined_momentum = ten_day_price_momentum + intraday_return
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int)
    adjusted_combined_momentum = combined_momentum * volume_change_indicator
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_range - high_low_range.shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    high_low_spread_momentum = high_low_spread_change.rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    combined_factors = vwap * high_low_spread_momentum * volume_weighted_range
    
    # Final Alpha Factor
    final_alpha_factor = alpha_factor_diff + adjusted_combined_momentum + combined_factors
    
    # Final Adjustment for Volume
    final_alpha_factor_volume_adjusted = final_alpha_factor * df['volume']
    
    return final_alpha_factor_volume_adjusted
```
