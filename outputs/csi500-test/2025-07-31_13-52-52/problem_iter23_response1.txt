```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    adjusted_combined_factors = combined_factors - opening_gap / (df['high'] - df['low'])
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open'] - 1
    
    # Combine Momentum Indicators
    momentum_indicators = close_to_open_return + price_movement_ratio.rolling(window=5).sum()
    
    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    amount_change = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    combined_volume_amount_momentum = volume_change + amount_change
    momentum_with_volume_amount = momentum_indicators * combined_volume_amount_momentum
    
    # Apply Time Series Momentum
    moving_average = momentum_with_volume_amount.rolling(window=20).mean()
    time_series_momentum = momentum_with_volume_amount - moving_average
    
    # Calculate Intraday Return
    intraday_return = df['high'] / df['low'] - 1
    
    # Compute 10-Day Price Momentum
    ten_day_price_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    volume_change_indicator = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_trend_indicator = np.where(short_ema > long_ema, 1, -1)
    
    # Combine Intraday and 10-Day Momentum
    combined_momentum = intraday_return + ten_day_price_momentum
    
    # Apply Volume Confirmation
    confirmed_momentum = combined_momentum * volume_change_indicator
    confirmed_momentum = np.where(volume_change > 0, confirmed_momentum, 0)
    
    # Integrate Both Metrics
    integrated_metrics = (confirmed_momentum + ten_day_price_momentum) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (time_series_momentum + integrated_metrics) / 2
    
    # Smooth the Result
    smoothed_final_alpha_factor = final_alpha_factor.rolling(window=20).mean()
    
    # Final Adjustment for Volume
    final_alpha_factor = smoothed_final_alpha_factor * np.log(df['volume'] + 1)
    
    return final_alpha_factor
```
