import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factors'] = df['close_to_open_return'] * df['volume'] / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap_adjusted'] = df['daily_volatility'] - (df['open'] - df['close'].shift(1))
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    df['5_day_price_momentum'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['combined_momentum'] = df['close_to_open_return'] + df['5_day_price_momentum']
    
    # Incorporate Volume and Amount Momentum
    df['avg_volume_10_to_2_days'] = df['volume'].rolling(window=9).mean().shift(1)
    df['volume_change'] = df['volume'] - df['avg_volume_10_to_2_days']
    
    df['avg_amount_10_to_2_days'] = df['amount'].rolling(window=9).mean().shift(1)
    df['amount_change'] = df['amount'] - df['avg_amount_10_to_2_days']
    
    df['combined_volume_amount_momentum'] = df['volume_change'] + df['amount_change']
    df['momentum_weighted_by_volume_amount'] = df['combined_momentum'] * df['combined_volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['moving_average_alpha'] = df['momentum_weighted_by_volume_amount'].rolling(window=20).mean()
    df['alpha_diff_from_ma'] = df['momentum_weighted_by_volume_amount'] - df['moving_average_alpha']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = np.where(df['short_sma'] > df['long_sma'], 1, -1)
    
    # Combine Intraday and 10-Day Momentum
    df['combined_intraday_10_day_momentum'] = df['intraday_return'] + df['10_day_price_momentum']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = np.where(df['volume_change_indicator'] > 0, df['combined_intraday_10_day_momentum'], 0)
    
    # Integrate Both Metrics
    df['integrated_metrics'] = (df['alpha_diff_from_ma'] + df['volume_confirmed_momentum']) / 2
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_metrics']
    
    # Smooth the Result
    df['smoothed_final_alpha_factor'] = df['final_alpha_factor'].rolling(window=10).mean()
    
    # Final Adjustment for Volume
    df['final_alpha_factor'] = df['smoothed_final_alpha_factor'] * np.log(df['volume'] + 1)
    
    return df['final_alpha_factor'].dropna()
