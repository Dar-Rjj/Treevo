import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement
    df['Price_Movement'] = df['close'] - df['open']
    
    # Identify Enhanced Price Breakaways
    df['High_Breakaway'] = (df['high'] > df['high'].shift(1)) & ((df['high'] - df['high'].shift(1)) / df['high'].shift(1) > 0.05)
    df['Low_Breakaway'] = (df['low'] < df['low'].shift(1)) & ((df['low'].shift(1) - df['low']) / df['low'].shift(1) > 0.05)
    
    # Volume-Adjusted Score with Volume Momentum
    conditions = [
        (df['High_Breakaway'] & ~df['Low_Breakaway']),
        (df['Low_Breakaway'] & ~df['High_Breakaway']),
        (~df['High_Breakaway'] & ~df['Low_Breakaway'])
    ]
    choices = [
        (df['Price_Movement'] * df['volume'] * (df['volume'] - df['volume'].shift(1))),
        -(df['Price_Movement'] * df['volume'] * (df['volume'] - df['volume'].shift(1))),
        0
    ]
    df['Volume_Adjusted_Score'] = pd.np.select(conditions, choices, default=0)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Score'].rolling(window=n).sum()
    
    return df['Final_Alpha_Factor']
