import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    open_gap = df['open'] - df['close'].shift(1)
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_range = high_low_range * volume_change
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_average = adjusted_range.rolling(window=lookback_period).mean()
    time_series_momentum_diff = adjusted_range - moving_average
    
    # Calculate 5-Day Rolling Momentum
    rolling_sum_price_movement_ratio = price_movement_ratio.rolling(window=5).sum()
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    rolling_momentum = rolling_sum_price_movement_ratio / rolling_sum_high_low_range
    
    # Adjust Momentum for Volume
    adjusted_momentum = rolling_momentum * df['volume']
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    daily_volatility = (df['high'] - df['low']).abs()
    close_to_open_return = (df['close'] - df['open']) / df['open']
    final_factor_1 = (adjusted_momentum * close_to_open_return) / daily_volatility
    
    # Combine Close-to-Open Return with Volume
    combined_close_to_open_return = (close_to_open_return * df['volume']) / daily_volatility
    
    # Volume Adjusted Momentum
    close_prices_last_5_days = df['close'].diff().rolling(window=5).sum()
    avg_volume_last_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = close_prices_last_5_days * avg_volume_last_5_days
    
    # Final Alpha Factor
    final_alpha_factor = (time_series_momentum_diff + volume_adjusted_momentum + combined_close_to_open_return)
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    final_alpha_factor *= five_day_avg_volume
    
    # Incorporate Long-Term Volatility
    long_term_volatility = df['close'].rolling(window=20).std()
    final_alpha_factor /= long_term_volatility
    
    # Integrate Recent Volume Trend
    ema_volume_5_days = df['volume'].ewm(span=5).mean()
    final_alpha_factor *= ema_volume_5_days
    
    return final_alpha_factor
