import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Short SMA (5 days) of Close prices
    df['short_sma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long SMA (20 days) of Close prices
    df['long_sma'] = df['close'].rolling(window=20).mean()
    
    # Determine Price Trend Indicator
    df['price_trend'] = np.where(df['short_sma'] > df['long_sma'], 1, -1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratios
    df['rolling_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    
    # Calculate 5-day Rolling Sum of High-Low Ranges
    df['rolling_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['rolling_price_movement_ratio'] / df['rolling_high_low_range']
    
    # Adjust for Volume
    df['adjusted_momentum'] = df['momentum_indicator'] * df['volume']
    
    # Combine Intraday and Volume-Amount Indicators
    df['combined_intraday_volume'] = df['intraday_return'] * df['volume_change']
    
    # Apply Exponential Moving Average for Smoothing
    df['smoothed_combined_intraday_volume'] = df['combined_intraday_volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factor'] = df['intraday_return'] * df['volume'] / df['daily_volatility']
    
    # Calculate 5-day Moving Average of Close Prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['5_day_ma_close']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_momentum'] + df['combined_factor']) * df['relative_strength']
    
    return df['final_alpha_factor']
