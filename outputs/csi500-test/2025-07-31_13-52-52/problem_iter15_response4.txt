import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors: Multiply Close-to-Open Return by Volume and Divide by Daily Volatility
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Combine Momentum Indicators: 5-day rolling sum of Price Movement Ratio
    momentum_indicator = price_movement_ratio.rolling(window=5).sum() + close_to_open_return
    
    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].rolling(window=10).mean().shift(1)
    amount_change = df['amount'] - df['amount'].rolling(window=10).mean().shift(1)
    
    # Combine Volume Change and Amount Change, Weight by the High-Low Range
    combined_momentum = (volume_change + amount_change) * high_low_range
    
    # Multiply Combined Momentum by Combined Volume-Amount Momentum
    adjusted_momentum = momentum_indicator * combined_momentum
    
    # Apply Time Series Momentum: Calculate Alpha Factor's Moving Average over a specified lookback period (e.g., 10 days)
    moving_average = adjusted_momentum.rolling(window=10).mean()
    
    # Calculate Difference between Current Value and Moving Average
    alpha_factor = adjusted_momentum - moving_average
    
    # Final Alpha Factor: If Volume Change > 0, keep result; If Volume Change <= 0, set result to 0
    final_alpha_factor = alpha_factor.where(volume_change > 0, 0)
    
    return final_alpha_factor
