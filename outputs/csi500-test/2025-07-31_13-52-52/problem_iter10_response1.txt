```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Generate Momentum Indicator
    rolling_price_movement = price_movement_ratio.rolling(window=5).sum()
    rolling_high_low_range = high_low_range.rolling(window=5).sum()
    momentum_indicator = rolling_price_movement * rolling_high_low_range
    
    # Integrate Volume
    volume_adjusted_momentum = momentum_indicator * df['volume']
    
    return volume_adjusted_momentum
```
