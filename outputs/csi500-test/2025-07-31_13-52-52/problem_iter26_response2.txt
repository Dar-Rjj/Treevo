```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    df['5_day_momentum'] = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['short_sma'] > df['long_sma']).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['intraday_return'] + df['10_day_momentum']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['combined_momentum'] * df['volume_change_indicator']
    df['volume_confirmed_momentum'] = df['volume_confirmed_momentum'].where(df['volume_change_indicator'] > 0, 0)
    
    # Integrate Both Metrics
    weights = [0.5, 0.5]  # Define Weights for Each Metric
    df['integrated_metrics'] = (weights[0] * df['combined_momentum'] + weights[1] * df['5_day_momentum'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (weights[0] * df['volume_confirmed_momentum'] + weights[1] * df['integrated_metrics'])
    
    # Smooth the Result
    df['smoothed_alpha_factor'] = df['final_alpha_factor'].rolling(window=10).mean()
    
    return df['smoothed_alpha_factor']
```
