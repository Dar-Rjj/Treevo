```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Momentum Adjusted
    momentum_adjusted_factor = combined_factor - momentum
    
    return momentum_adjusted_factor
```
