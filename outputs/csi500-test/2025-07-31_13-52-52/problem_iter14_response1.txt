import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['opening_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] * (1 + df['opening_gap'])
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_high_low'] = df['adjusted_high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['volume_weighted_high_low_ma'] = df['volume_weighted_high_low'].rolling(window=lookback_period).mean()
    df['ts_momentum'] = df['volume_weighted_high_low'] - df['volume_weighted_high_low_ma']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Combine Intraday Return and Volume Change
    df['intraday_vol_adj_return'] = df['intraday_return'] * (df['volume'] > df['volume'].shift(1))
    short_period = 5
    df['smoothed_intraday_vol_adj_return'] = df['intraday_vol_adj_return'].ewm(span=short_period, adjust=False).mean()
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Adjust for Momentum
    momentum_days = 5
    df['momentum_sum_diff'] = (df['close'] - df['close'].shift(momentum_days)).rolling(window=momentum_days).sum() * df['volume'].rolling(window=momentum_days).mean()
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = (df['close'] - df['close'].shift(momentum_days)).rolling(window=momentum_days).sum()
    
    # Calculate Averaged High-to-Low Momentum
    lookback_period = 10
    df['high_to_low_momentum'] = (df['high'] - df['low']) / df['low']
    df['averaged_high_to_low_momentum'] = df['high_to_low_momentum'].rolling(window=lookback_period).mean()
    
    # Combine and Adjust Factors
    df['adjusted_close_to_open_return'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    df['adjusted_factor'] = df['adjusted_close_to_open_return'] - df['5_day_momentum']
    
    # Integrate Both Metrics
    weights = [0.4, 0.3, 0.3]
    df['integrated_metrics'] = (
        weights[0] * df['smoothed_intraday_vol_adj_return'] +
        weights[1] * df['adjusted_close_to_open_return'] +
        weights[2] * df['averaged_high_to_low_momentum']
    )
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['adjusted_factor'] + 0.4 * df['integrated_metrics']
    
    # Smooth the Result
    df['final_smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    return df['final_smoothed_alpha_factor'].dropna()
