import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['volume_adjusted_return'] = df['close_to_open_return'] * df['volume']
    df['factor'] = df['volume_adjusted_return'] / df['daily_volatility']
    
    # Adjust for Momentum
    df['close_diff_sum'] = df['close'].diff().rolling(window=5).sum()
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['momentum_adjustment'] = df['close_diff_sum'] * df['avg_volume_5_days']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['factor'] + df['momentum_adjustment']
    
    return df['alpha_factor']
