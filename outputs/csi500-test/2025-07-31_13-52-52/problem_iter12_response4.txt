```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() + df['close_to_open_return']
    
    # Incorporate Volume and Amount Momentum
    df['avg_volume_9d'] = df['volume'].shift(1).rolling(window=9).mean()
    df['avg_amount_9d'] = df['amount'].shift(1).rolling(window=9).mean()
    
    df['volume_change'] = df['volume'] - df['avg_volume_9d']
    df['amount_change'] = df['amount'] - df['avg_amount_9d']
    
    df['volume_amount_momentum'] = (df['volume_change'] + df['amount_change']) / 2
    
    # Multiply Combined Momentum by Combined Volume-Amount Momentum
    df['adjusted_momentum'] = df['momentum_indicator'] * df['volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['alpha_factor_ma'] = df['adjusted_momentum'].rolling(window=10).mean()
    df['alpha_factor_diff'] = df['adjusted_momentum'] - df['alpha_factor_ma']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df.apply(lambda row: row['alpha_factor_diff'] if row['volume_change'] > 0 else 0, axis=1)
    
    return df['final_alpha_factor']
```
