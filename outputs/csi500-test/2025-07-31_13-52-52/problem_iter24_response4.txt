```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    df['5_day_momentum'] = (df['close'] - df['close'].shift(5)).cumsum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = np.where(df['short_sma'] > df['long_sma'], 1, -1)
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['intraday_return'] + df['10_day_price_momentum']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['combined_momentum'] * df['volume_change_indicator']
    df['volume_confirmed_momentum'] = np.where(df['volume_change_indicator'] > 0, df['volume_confirmed_momentum'], 0)
    
    # Integrate Both Metrics
    df['integrated_metrics'] = (df['5_day_momentum'] + df['volume_confirmed_momentum']) / 2
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_factor'] + df['integrated_metrics']) / 2
    
    # Smooth the Result
    df['smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    return df['smoothed_alpha_factor']
```
