```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['prev_close'] = df['close'].shift(1)
    df['open_gap'] = df['open'] - df['prev_close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_range'] = df['high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    df['moving_avg_adjusted_range'] = df['adjusted_range'].rolling(window=10).mean()
    df['momentum_diff'] = df['adjusted_range'] - df['moving_avg_adjusted_range']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_price_movement_ratio'] = df['price_movement_ratio'].rolling(window=5).sum()
    df['rolling_high_low_range'] = df['high_low_range'].rolling(window=5).sum()
    df['rolling_momentum'] = df['rolling_price_movement_ratio'] / df['rolling_high_low_range']
    
    # Calculate Price Trend Indicator
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_indicator'] = np.where(short_ema > long_ema, 1, -1)
    
    # Combine Momentum, Intraday, and Price Trend Indicators
    df['combined_indicator'] = (df['high'] - df['low']) * df['volume_change'] * df['price_trend_indicator']
    df['smoothed_combined_indicator'] = df['combined_indicator'].ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume
    df['log_volume'] = np.log(df['volume'] + 1)
    df['adjusted_combined_indicator'] = df['smoothed_combined_indicator'] * df['log_volume']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = np.abs(df['high'] - df['low'])
    
    # Integrate Volume-Adjusted Momentum and Volatility into Final Factor
    df['integrated_factor'] = (df['adjusted_combined_indicator'] * df['close_to_open_return']) / df['daily_volatility']
    
    # Apply a 5-Day Average Volume Adjustment to the Final Alpha Factor
    df['avg_5d_volume'] = df['volume'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['integrated_factor'] * df['avg_5d_volume']
    
    # Calculate 5-day Moving Average of Close Prices
    df['5d_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['5d_ma_close']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['relative_strength']
    
    return df['final_alpha_factor']
```
