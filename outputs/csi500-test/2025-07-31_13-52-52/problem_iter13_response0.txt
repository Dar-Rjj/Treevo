```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_period = 5
    long_period = 20
    df['sma_short'] = df['close'].rolling(window=short_period, min_periods=1).mean()
    df['sma_long'] = df['close'].rolling(window=long_period, min_periods=1).mean()
    df['price_trend'] = (df['sma_short'] > df['sma_long']).astype(int) * 2 - 1
    
    # Combine and Adjust Factors
    df['combined_factor'] = df['intraday_return'] * df['volume_change'] * df['price_trend']
    
    # Smooth the Combined Result
    smoothing_period = 10
    df['smoothed_factor'] = df['combined_factor'].ewm(span=smoothing_period, min_periods=1).mean()
    
    return df['smoothed_factor']
```
