```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate the difference between today's close and yesterday's close (momentum factor)
    df['close_diff'] = df['close'].diff()
    
    # Use the high-low spread to indicate market volatility
    df['high_low_spread'] = df['high'] - df['low']
    
    # Formulate a factor by dividing today's volume by the average volume over the past N days
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=n_days).mean()
    
    # Create a factor that measures the correlation between volume changes and price changes over a specified time window
    df['volume_change'] = df['volume'].pct_change()
    df['price_change'] = df['close'].pct_change()
    df['vol_price_corr'] = df[['volume_change', 'price_change']].rolling(window=n_days).corr().unstack().iloc[::2, 1].reset_index(drop=True)
    
    # Design a factor based on the relationship between the change in closing price and the change in volume
    df['price_vol_change'] = df['close_diff'] * df['volume_change']
    
    # Investigate the interaction between the open-close spread and the volume traded at the opening session
    df['open_close_spread'] = df['close'] - df['open']
    df['open_session_sentiment'] = df['open_close_spread'] * (df['volume'] / df['volume'].shift(1))
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_diff'] + 
        df['high_low_spread'] + 
        df['volume_ratio'] + 
        df['vol_price_corr'] + 
        df['price_vol_change'] + 
        df['open_session_sentiment']
    )
    
    return df['alpha_factor']
```
