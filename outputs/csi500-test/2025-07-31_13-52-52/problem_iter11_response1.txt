```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Generate Momentum Indicator
    momentum_indicator = price_movement_ratio.rolling(window=5).sum() * high_low_range.rolling(window=5).mean()
    
    # Integrate Volume into Momentum
    volume_momentum = momentum_indicator * df['volume']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factors = (close_open_return * df['volume']) / daily_volatility
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    adjusted_high_low_range = high_low_range - opening_gap.fillna(0)
    
    # Incorporate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_weighted_adjusted_range = adjusted_high_low_range * volume_change
    
    # Apply Time Series Momentum
    alpha_factor_ma = volume_weighted_adjusted_range.rolling(window=10).mean()
    ts_momentum = volume_weighted_adjusted_range - alpha_factor_ma
    
    # Calculate High-Low Spread Change
    high_low_spread_change = high_low_range - high_low_range.shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    high_low_spread_momentum = high_low_spread_change.rolling(window=5).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    combined_vwap_high_low_momentum = vwap * high_low_spread_momentum * volume_weighted_adjusted_range
    
    # Final Alpha Factor
    final_alpha_factor = ts_momentum + combined_factors + combined_vwap_high_low_momentum
    
    # Final Adjustment for Volume
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
```
