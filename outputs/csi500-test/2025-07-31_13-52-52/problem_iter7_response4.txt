```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend'] = (df['short_sma'] > df['long_sma']).astype(int) * 2 - 1
    
    # Combine Indicators
    df['indicator'] = df['intraday_return'] * df['volume_change'] * df['price_trend']
    
    # Smooth the Result
    df['smoothed_indicator'] = df['indicator'].rolling(window=10).mean()
    df['ema_smoothed_indicator'] = df['smoothed_indicator'].ewm(span=10, adjust=False).mean()
    
    return df['ema_smoothed_indicator']
```
