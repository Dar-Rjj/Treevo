```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = abs(df['high'] - df['low'])
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high'] - df['low'] - df['opening_gap']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Combine Momentum Indicators
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=5).sum() + df['close_to_open_return']
    
    # Incorporate Volume and Amount Momentum
    df['volume_change'] = df['volume'] - df['volume'].rolling(window=9).mean().shift(1)
    df['amount_change'] = df['amount'] - df['amount'].rolling(window=9).mean().shift(1)
    df['volume_amount_momentum'] = df['volume_change'] + df['amount_change']
    df['combined_momentum'] = df['momentum_indicator'] * df['volume_amount_momentum']
    
    # Apply Time Series Momentum
    df['alpha_factor_ma'] = df['combined_factor'].rolling(window=10).mean()
    df['alpha_factor_diff'] = df['combined_factor'] - df['alpha_factor_ma']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Compute 10-Day Price Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = (df['short_sma'] > df['long_sma']).astype(int) * 2 - 1
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['intraday_return'] + df['10_day_price_momentum']
    
    # Apply Volume Confirmation
    df['volume_confirmed_momentum'] = df['combined_momentum'] * df['volume_change_indicator']
    df['volume_confirmed_momentum'] = df['volume_confirmed_momentum'] * (df['volume_change_indicator'] > 0)
    
    # Integrate Both Metrics
    weights = [0.5, 0.5]  # Define weights for each metric
    df['integrated_metrics'] = (df['combined_momentum'] * weights[0]) + (df['volume_confirmed_momentum'] * weights[1])
    
    # Synthesize Final Alpha Factor
    final_weights = [0.4, 0.6]  # Define weights for each component
    df['final_alpha_factor'] = (df['alpha_factor_diff'] * final_weights[0]) + (df['integrated_metrics'] * final_weights[1])
    
    # Smooth the Result
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(span=10, adjust=False).mean()
    
    return df['final_alpha_factor_smoothed']
```
