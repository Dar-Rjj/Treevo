```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_range
    
    # Adjust for Momentum
    momentum = (df['close'].diff().rolling(window=5).sum()) / df['volume'].rolling(window=5).mean()
    adjusted_combined_factor = combined_factor * momentum
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'].diff() > 0).astype(int) * 2 - 1
    
    # Calculate Short SMA (5 days) of Close prices
    short_sma = df['close'].rolling(window=5).mean()
    
    # Calculate Long SMA (20 days) of Close prices
    long_sma = df['close'].rolling(window=20).mean()
    
    # Determine Price Trend Indicator
    price_trend_indicator = (short_sma > long_sma).astype(int) * 2 - 1
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Calculate 5-day Rolling Sum of Price Movement Ratios
    rolling_sum_price_movement = price_movement_ratio.rolling(window=5).sum()
    
    # Calculate 5-day Rolling Sum of High-Low Ranges
    rolling_sum_high_low_range = high_low_range.rolling(window=5).sum()
    
    # Calculate Momentum Indicator
    momentum_indicator = rolling_sum_price_movement / rolling_sum_high_low_range
    
    # Adjust for Volume
    volume_adjusted_momentum = df['volume'] * momentum_indicator
    
    # Combine Intraday and Volume-Amount Indicators
    combined_intraday_volume = intraday_return * volume_change_indicator * volume_adjusted_momentum
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    smoothed_factor = combined_intraday_volume.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return smoothed_factor
```
