```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['intraday_momentum'] = df['intraday_return'] * df['volume_change_indicator']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 5 days)
    df['intraday_momentum_sma'] = df['intraday_momentum'].rolling(window=5).mean()
    
    # Use Exponential Moving Average for Smoothing
    df['intraday_momentum_ema'] = df['intraday_momentum_sma'].ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Smooth High-Low Difference using Moving Average
    df['smoothed_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Incorporate Volume into High-Low Difference
    df['volume_delta'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_high_low'] = df['smoothed_high_low_diff'] * df['volume_delta']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['intraday_momentum_ema'] + df['volume_weighted_high_low']) * df['close_to_close_return']
    
    return df['final_alpha_factor']
```
