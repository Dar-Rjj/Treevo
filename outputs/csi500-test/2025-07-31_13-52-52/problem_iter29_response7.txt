```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on (open, high, low, close, amount, volume) prices.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3.0
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day exponential moving average of the typical price
    ema_typical_price = typical_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the ratio of the two EMAs
    factor = ema_money_flow / ema_typical_price
    
    return factor
```
