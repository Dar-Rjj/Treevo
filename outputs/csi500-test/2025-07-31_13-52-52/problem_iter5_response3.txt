```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['open_gap'] = df['open'] - df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] - df['open_gap'].abs()
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['adjusted_high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['volume_weighted_ma'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['ts_momentum'] = df['volume_weighted_range'] - df['volume_weighted_ma']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(lookback_period)
    
    # Combine Momentum Indicators
    df['combined_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() + df['price_momentum']
    
    # Adjust for Volume
    df['avg_volume_10_days'] = df['volume'].shift(1).rolling(window=lookback_period-1).mean()
    df['volume_change_10_days'] = df['volume'] - df['avg_volume_10_days']
    df['combined_momentum_adj'] = df['combined_momentum'] * df['volume_change_10_days']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor Combination
    df['alpha_factor'] = 0
    df.loc[df['volume_change_10_days'] > 0, 'alpha_factor'] = df['vwap'] * df['ts_momentum'] * df['combined_momentum_adj']
    
    return df['alpha_factor']
```
