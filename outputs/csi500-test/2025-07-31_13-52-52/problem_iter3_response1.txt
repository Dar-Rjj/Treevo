```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['close_prev'] = df['close'].shift(1)
    open_gap = (df['open'] - df['close_prev']).abs()
    adjusted_high_low_range = high_low_range - open_gap
    
    # Incorporate Volume and Amount Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    combined_momentum = (df['volume_change'] + df['amount_change']) / 2
    weighted_range = adjusted_high_low_range * combined_momentum
    
    # Apply Time Series Momentum
    lookback_period = 10
    moving_avg = weighted_range.rolling(window=lookback_period).mean()
    alpha_factor = weighted_range - moving_avg
    
    return alpha_factor
```
