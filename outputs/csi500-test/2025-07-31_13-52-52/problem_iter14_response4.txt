```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return and Volume Change
    df['combined_intraday_vol'] = df['intraday_return'] * df['volume_change']
    
    # Smooth the Result using Simple Moving Average over a Short Period (e.g., 7 days)
    short_period = 7
    df['sma_smoothed'] = df['combined_intraday_vol'].rolling(window=short_period).mean()
    
    # Use Exponential Moving Average for Smoothing
    smoothing_factor = 0.5
    df['ema_smoothed'] = df['sma_smoothed'].ewm(alpha=smoothing_factor).mean()
    
    # Calculate High-Low and Close-Open Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['price_movement_ratio'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Rolling Momentum
    df['rolling_momentum'] = df['price_movement_ratio'].rolling(window=5).sum() / df['high_low_range'].rolling(window=5).sum()
    
    # Adjust for Volume
    df['adjusted_momentum'] = df['rolling_momentum'] * df['volume']
    
    # Calculate 10-Day Price and Intraday Momentum
    df['10_day_price_momentum'] = df['close'] - df['close'].shift(10)
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Combine and Adjust Price and Intraday Momentum
    df['combined_momentum'] = df['10_day_price_momentum'] + df['intraday_momentum']
    df['combined_momentum'] = df['combined_momentum'] * (df['volume_change'] > 0)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 30
    df['high_low_volatility'] = df['daily_volatility'].rolling(window=lookback_period).sum() / lookback_period
    
    # Integrate Both Metrics using Unweighted Average
    df['final_alpha_factor'] = (df['ema_smoothed'] + df['adjusted_momentum'] + df['combined_momentum'] + df['high_low_volatility']) / 4.0
    
    # Final Alpha Factor with Price Trend Adjustment
    df['final_alpha_factor'] = (df['final_alpha_factor'] * df['volume_change']).ewm(span=10).mean()
    
    return df['final_alpha_factor']
```
