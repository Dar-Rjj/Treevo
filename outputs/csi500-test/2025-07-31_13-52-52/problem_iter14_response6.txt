```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute 10-Day Price Momentum
    df['10_day_momentum'] = df['close'].diff(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = np.where(df['volume'].diff() > 0, 1, -1)
    
    # Compute Amount Change Indicator
    df['amount_change_indicator'] = np.where(df['amount'].diff() > 0, 1, -1)
    
    # Combine Intraday Return, Volume Change, and Amount Change
    df['combined_intraday'] = df['intraday_return'] * df['volume_change_indicator'] * df['amount_change_indicator']
    
    # Smooth the Result using Simple Moving Average over a Short Period (5 days)
    df['smoothed_combined_intraday_sma'] = df['combined_intraday'].rolling(window=5).mean()
    
    # Apply Exponential Moving Average for Smoothing
    alpha = 0.2  # Define Smoothing Factor
    df['smoothed_combined_intraday_ema'] = df['combined_intraday'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Price Trend Indicator
    df['short_sma'] = df['close'].rolling(window=5).mean()
    df['long_sma'] = df['close'].rolling(window=20).mean()
    df['price_trend_indicator'] = np.where(df['short_sma'] > df['long_sma'], 1, -1)
    
    # Combine Intraday and 10-Day Momentum
    df['combined_momentum'] = df['intraday_return'] + df['10_day_momentum']
    
    # Apply Volume Confirmation
    df['momentum_with_volume_confirmation'] = df['combined_momentum'] * df['volume_change_indicator']
    df['momentum_with_volume_confirmation'] = np.where(df['volume_change_indicator'] > 0, df['momentum_with_volume_confirmation'], 0)
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20  # Define Lookback Period
    df['high_low_volatility'] = (df['high'] - df['low']) / df['low']
    df['averaged_high_low_volatility'] = df['high_low_volatility'].rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics using Unweighted Average
    df['final_factor'] = (df['smoothed_combined_intraday_ema'] + df['averaged_high_low_volatility']) / 2
    
    # Smooth the Final Alpha Factor using Simple Moving Average (10 days) and Exponential Moving Average
    df['final_factor_sma'] = df['final_factor'].rolling(window=10).mean()
    df['final_factor_ema'] = df['final_factor'].ewm(span=10, adjust=False).mean()
    
    return df['final_factor_ema']
```
