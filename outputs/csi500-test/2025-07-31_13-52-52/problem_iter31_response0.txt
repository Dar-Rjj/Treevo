```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Adjust for Momentum
    df['momentum_adjustment'] = (df['close'] - df['close'].shift(5)).rolling(window=5).sum() * df['volume'].rolling(window=5).mean()
    
    # Compute 10-Day Price Momentum
    df['price_momentum_10d'] = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    df['volume_change_indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Apply Volume Confirmation
    df['volume_confirmed_factor'] = df['combined_factor'] * df['volume_change_indicator']
    
    # Integrate Both Metrics
    df['integrated_metrics'] = (df['momentum_adjustment'] + df['volume_confirmed_factor']) / 2
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['integrated_metrics'] + df['price_momentum_10d']) / 2
    
    return df['final_alpha_factor']
```
