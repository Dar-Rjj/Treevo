import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    volume_adjusted_return = close_to_open_return * df['volume']
    volatility_adjusted_return = volume_adjusted_return / daily_volatility
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Combine Momentum Indicators
    rolling_price_movement = price_movement_ratio.rolling(window=5).sum()
    combined_momentum = close_to_open_return + rolling_price_movement
    
    # Incorporate Volume and Amount Momentum
    volume_change = df['volume'] - df['volume'].shift(1).rolling(window=9).mean()
    amount_change = df['amount'] - df['amount'].shift(1).rolling(window=9).mean()
    combined_volume_amount_momentum = (volume_change + amount_change) / 2
    
    momentum_adjusted = combined_momentum * combined_volume_amount_momentum
    
    # Apply Time Series Momentum
    alpha_factor_ma = momentum_adjusted.rolling(window=10).mean()
    alpha_factor_diff = momentum_adjusted - alpha_factor_ma
    
    # Final Alpha Factor
    final_alpha_factor = alpha_factor_diff.where(volume_change > 0, 0)
    
    return final_alpha_factor
