```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Compute Amount Change Indicator
    df['amount_change'] = (df['amount'] > df['amount'].shift(1)).astype(int) * 2 - 1
    
    # Combine Intraday Return, Volume Change, and Amount Change
    df['combined_signal'] = df['intraday_return'] * df['volume_change'] * df['amount_change']
    
    # Smooth the Result
    short_period = 5
    smoothing_factor = 0.3
    df['smoothed_signal'] = df['combined_signal'].rolling(window=short_period).mean()
    df['smoothed_signal_ema'] = df['combined_signal'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    df['high_low_volatility'] = (df['high'] - df['low']) / df['low']
    df['averaged_high_low_volatility'] = df['high_low_volatility'].rolling(window=lookback_period).mean()
    
    # Integrate Both Metrics
    weight_combined_signal = 0.6
    weight_averaged_high_low_volatility = 0.4
    df['alpha_factor'] = (weight_combined_signal * df['smoothed_signal_ema'] + 
                          weight_averaged_high_low_volatility * df['averaged_high_low_volatility'])
    
    return df['alpha_factor']
```
