import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = (df['close'] - df['open']) / df['open']
    
    # Compute Momentum Indicator
    momentum_indicator = price_movement_ratio.rolling(window=5).sum() / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    adjusted_momentum_indicator = momentum_indicator * df['volume']
    
    # Combine Intraday and Volume-Amount Indicators
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_change = df['volume'] - df['volume'].shift(1)
    combined_factor = intraday_return * volume_change
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor_volatility = (intraday_return * df['volume']) / daily_volatility
    
    # Calculate 5-day Moving Average of Close Prices
    moving_avg_5_close = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    relative_strength = df['close'] / moving_avg_5_close
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum_indicator + combined_factor_volatility) * relative_strength
    
    # Calculate Averaged High-to-Low Volatility
    lookback_period = 20
    high_low_pct_change = (df['high'] - df['low']) / df['low']
    avg_high_low_volatility = high_low_pct_change.rolling(window=lookback_period).sum() / lookback_period
    
    # Integrate Both Metrics using Weighted Average
    weight_adjusted_momentum = 0.6
    weight_combined_factor = 0.4
    weighted_sum = (weight_adjusted_momentum * adjusted_momentum_indicator) + (weight_combined_factor * combined_factor_volatility)
    
    # Calculate 20-day Moving Average of Close Prices
    moving_avg_20_close = df['close'].rolling(window=20).mean()
    
    # Calculate Long-Term Trend Indicator
    long_term_trend_indicator = df['close'] / moving_avg_20_close
    
    # Enhance Final Alpha Factor
    enhanced_final_alpha_factor = final_alpha_factor * long_term_trend_indicator
    
    return enhanced_final_alpha_factor
