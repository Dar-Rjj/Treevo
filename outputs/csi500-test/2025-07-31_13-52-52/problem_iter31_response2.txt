```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    
    # Combine Factors
    combined_factor = (close_to_open_return * df['volume']) / daily_volatility
    
    # Adjust for Momentum
    momentum = (df['close'] - df['close'].shift(5)).sum() * df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['high'] / df['low'] - 1
    
    # Compute 10-Day Price Momentum
    ten_day_momentum = df['close'] - df['close'].shift(10)
    
    # Compute Volume Change Indicator
    volume_change_indicator = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Apply Volume Confirmation
    volume_confirmed_result = combined_factor * volume_change_indicator
    volume_confirmed_result[volume_change_indicator <= 0] = 0
    
    # Integrate Both Metrics
    integrated_metrics = (momentum + intraday_return) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (volume_confirmed_result + integrated_metrics + ten_day_momentum) / 3
    
    # Smooth the Result
    final_alpha_factor = final_alpha_factor.ewm(span=10, adjust=False).mean()
    
    return final_alpha_factor
```
