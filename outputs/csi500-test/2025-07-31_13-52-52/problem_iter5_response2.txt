```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust for Opening Gap
    df['prev_close'] = df['close'].shift(1)
    df['adjusted_high_low_range'] = df['high_low_range'] - (df['open'] - df['prev_close'])
    
    # Incorporate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_weighted_range'] = df['adjusted_high_low_range'] * df['volume_change']
    
    # Apply Time Series Momentum
    lookback_period = 10
    df['moving_average'] = df['volume_weighted_range'].rolling(window=lookback_period).mean()
    df['time_series_momentum'] = df['volume_weighted_range'] - df['moving_average']
    
    # Calculate High-Low Spread Change
    df['high_low_spread_change'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Momentum of High-Low Spread Change
    momentum_lookback = 5
    df['high_low_spread_momentum'] = df['high_low_spread_change'].rolling(window=momentum_lookback).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine VWAP, High-Low Spread Momentum, and Adjusted Range
    df['combined_factor'] = df['vwap'] * df['high_low_spread_momentum'] * df['volume_weighted_range']
    
    # Calculate Price Movement Ratio
    df['price_movement_ratio'] = df['close'] / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['price_movement_ratio'].rolling(window=momentum_lookback).sum() / df['high_low_range'].rolling(window=momentum_lookback).sum()
    
    # Final Adjustment for Volume
    df['final_alpha_factor'] = df['combined_factor'] * df['momentum_indicator']
    
    return df['final_alpha_factor']
```
