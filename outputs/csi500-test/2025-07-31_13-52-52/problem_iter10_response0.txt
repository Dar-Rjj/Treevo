import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio
    price_movement_ratio = df['close'] / df['open']
    
    # Calculate High-to-Low Price Percentage Change
    high_to_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Average Over a Lookback Period
    avg_high_to_low_pct_change = high_to_low_pct_change.rolling(window=lookback_period).mean()
    
    # 5-day rolling sum of Price Movement Ratio
    rolling_sum_price_movement_ratio = price_movement_ratio.rolling(window=5).sum()
    
    # Divide by 5-day rolling sum of High-Low Range
    combined_indicator = rolling_sum_price_movement_ratio / high_low_range.rolling(window=5).sum()
    
    # Adjust for Volume
    adjusted_indicator = combined_indicator * df['volume']
    
    return adjusted_indicator
