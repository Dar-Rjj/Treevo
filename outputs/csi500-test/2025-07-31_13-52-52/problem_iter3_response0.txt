```python
import pandas as pd

def heuristics(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Change Indicator
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Price Trend Indicator
    short_period = 5
    long_period = 20
    df['sma_short'] = df['close'].rolling(window=short_period).mean()
    df['sma_long'] = df['close'].rolling(window=long_period).mean()
    df['price_trend'] = (df['sma_short'] > df['sma_long']).astype(int) * 2 - 1
    
    # Combine Intraday Return, Volume Change, and Price Trend
    df['combined_factor'] = df['intraday_return'] * df['volume_change'] * df['price_trend']
    
    # Smooth the Result using Simple Moving Average over a Medium Period
    medium_period = 10
    df['smoothed_factor'] = df['combined_factor'].rolling(window=medium_period).mean()
    
    # Apply Exponential Moving Average for Smoothing
    smoothing_factor = 0.2
    df['final_factor'] = df['smoothed_factor'].ewm(alpha=smoothing_factor).mean()
    
    return df['final_factor']
```
