```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    
    # Calculate Momentum
    df['momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['momentum'] * df['volume'].rolling(window=5).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] * df['volume']) / df['daily_volatility']
    
    # Momentum and Volume Adjusted
    df['alpha_factor'] = df['combined_factor'] - df['volume_adjusted_momentum']
    
    return df['alpha_factor']
```
