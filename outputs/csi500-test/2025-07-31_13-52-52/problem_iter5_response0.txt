```python
import pandas as pd

def heuristics_v2(df, lookback_period=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Movement Ratio (Close/Open)
    price_movement_ratio = df['close'] / df['open']
    
    # Calculate High-to-Low Price Percentage Change
    high_to_low_percentage_change = (df['high'] - df['low']) / df['low']
    
    # Average High-to-Low Percentage Change over a Lookback Period
    avg_high_to_low_percentage_change = high_to_low_percentage_change.rolling(window=lookback_period).mean()
    
    # 5-day rolling sum of Price Movement Ratio
    rolling_sum_price_movement_ratio = price_movement_ratio.rolling(window=lookback_period).sum()
    
    # 5-day rolling sum of High-Low Range
    rolling_sum_high_low_range = high_low_range.rolling(window=lookback_period).sum()
    
    # Combine Averages and Momentum
    combined_indicator = (rolling_sum_price_movement_ratio / rolling_sum_high_low_range) * avg_high_to_low_percentage_change
    
    # Adjust for Volume
    adjusted_indicator = combined_indicator * df['volume']
    
    return adjusted_indicator
```
