```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum_ema'] = df['high_low_diff'].ewm(span=10, min_periods=5).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10, min_periods=5).mad()
    df['intraday_adjusted_momentum'] = df['intraday_momentum_ema'] / (df['volume_volatility'] + 1e-6)

    # Calculate Volume-Weighted Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['price_change'] * np.log(df['volume'])

    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['intraday_adjusted_momentum']

    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['open'].shift(1)
    df['long_term_price_trend'] = df['close'].rolling(window=200, min_periods=100).mean()
    df['intraday_price_trend_adjusted'] = df['intraday_adjusted_volume_weighted_momentum'] * df['price_trend'] * df['long_term_price_trend']

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['intraday_price_trend_adjusted'] - df['intraday_price_trend_adjusted'].rolling(window=50, min_periods=25).mean()

    # Analyze Simple Moving Averages (SMA)
    df['sma_5'] = df['close'].rolling(window=5, min_periods=3).mean()
    df['sma_20'] = df['close'].rolling(window=20, min_periods=10).mean()

    # Examine Rate of Change (ROC)
    df['roc_5'] = df['close'] / df['close'].shift(5) - 1
    df['roc_20'] = df['close'] / df['close'].shift(20) - 1

    # Examine Trading Volume
    df['volume_sma_5'] = df['volume'].rolling(window=5, min_periods=3).mean()
    df['volume_sma_20'] = df['volume'].rolling(window=20, min_periods=10).mean()

    # Investigate Historical Volatility
    df['historical_volatility'] = df['close'].pct_change().rolling(window=20, min_periods=10).std() * np.sqrt(252)

    # Combine Signals for Composite Factor
    composite_factor = (
        0.3 * df['sma_5'] + 
        0.7 * df['sma_20'] + 
        0.3 * df['roc_5'] + 
        0.7 * df['roc_20'] + 
        0.5 * df['volume_sma_5'] + 
        0.5 * df['volume_sma_20']
    )

    # Enhance with Price-Oriented Indicators
    df['price_volume_ratio'] = df['close'] / df['volume']
    df['short_term_price_trend'] = df['close'].rolling(window=50, min_periods=25).mean()
    
    # Final Composite Factor
    final_composite_factor = (
        0.5 * composite_factor + 
        0.2 * df['price_volume_ratio'] + 
        0.3 * df['short_term_price_trend']
    )

    return final_composite_factor.dropna()
```
