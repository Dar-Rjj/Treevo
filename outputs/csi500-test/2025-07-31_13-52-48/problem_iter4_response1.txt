```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    
    # Determine Threshold for High Volume Change
    volume_change_threshold = df['volume_change'].quantile(0.75)
    
    # Filter High Volume Shocks
    df['high_volume_shock'] = df['volume_change'] < volume_change_threshold
    df['factor'] = df['price_momentum'] * df['high_volume_shock']
    
    return df['factor'].dropna()
```
