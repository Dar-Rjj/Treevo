import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by (date).
    
    Returns:
    pd.Series: A Series of alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['10_day_ma_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['10_day_std_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day momentum
    df['10_day_momentum'] = df['close'].pct_change(10)
    
    # Calculate the 10-day volume weighted price
    df['10_day_volume_weighted_price'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day volume weighted price relative to the current close price
    df['10_day_volume_weighted_price_rel'] = df['10_day_volume_weighted_price'] / df['close']
    
    # Calculate the 10-day average range (high - low) divided by the close price
    df['10_day_avg_range_rel'] = (df['high'] - df['low']).rolling(window=10).mean() / df['close']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        df['10_day_ma_return'] +
        df['10_day_momentum'] +
        df['10_day_volume_weighted_price_rel'] -
        df['10_day_avg_range_rel']
    ) / df['10_day_std_return']
    
    return df['alpha_factor'].dropna()
