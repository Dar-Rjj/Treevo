```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume'].shift(1)
    
    # Rolling Sum of Volume-Weighted Momentum over 20 Days
    df['rolling_sum_20'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Moving Average of Rolling Sum over 60 Days
    df['moving_avg_60'] = df['rolling_sum_20'].rolling(window=60).mean()
    
    # Compute Dynamic Volume-Weighted Momentum Oscillator
    df['dynamic_vol_weighted_mom_osc'] = df['rolling_sum_20'] - df['moving_avg_60']
    
    # Return the factor values
    return df['dynamic_vol_weighted_mom_osc'].dropna()
```
