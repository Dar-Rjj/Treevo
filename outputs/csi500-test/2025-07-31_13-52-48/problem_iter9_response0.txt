import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, alpha=0.1, beta=0.2):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    df['ema_intraday_return'] = df['intraday_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Compute Standard Deviation of Intraday Price Ranges
    df['std_intraday_price_range'] = df['intraday_price_range'].rolling(window=n).std()
    
    # Incorporate Volatility Component
    df['volatility_adjusted_ema'] = df['ema_intraday_return'] * df['std_intraday_price_range']
    
    # Calculate Volume Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * df['volume'] / df['open']
    
    # Adjust Exponential Moving Average with Volume Weighted Intraday Return
    df['adjusted_ema'] = df['volume_weighted_intraday_return'].ewm(alpha=beta, adjust=False).mean()
    
    # Compute Volume Rate of Change
    df['volume_roc'] = df['volume'].pct_change(periods=m)
    
    # Integrate Volume Trend
    df['final_alpha_factor'] = df['adjusted_ema'] + df['volume_roc']
    
    # Drop any rows with NaN values
    df.dropna(inplace=True)
    
    return df['final_alpha_factor']
