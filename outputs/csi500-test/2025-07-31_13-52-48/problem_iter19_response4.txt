```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    
    volume_threshold = df['volume_change'].quantile(0.75)
    amount_threshold = df['amount_change'].quantile(0.75)
    
    df = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['vol_weighted_ma_intraday_volatility'] = df['vol_weighted_intraday_volatility'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['reversal_score'] = df['close_to_open_return'] / df['vol_weighted_ma_intraday_volatility']

    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = np.log(2) / half_life
    df['exp_decay_reversal_score'] = df['reversal_score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['vol_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['vol_weighted_ma_intraday_range'] = df['vol_weighted_intraday_range'].rolling(window=10).mean()

    # Combine Reversal Score and Intraday Range
    df['combined_score'] = df['exp_decay_reversal_score'] + df['intraday_range']
    df['exp_decay_combined_score'] = df['combined_score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Price Movement
    df['price_movement_high_low'] = (df['high'] - df['low']) / df['open']
    df['price_movement_close_open'] = (df['close'] - df['open']) / df['open']
    df['avg_price_movement'] = (df['price_movement_high_low'] + df['price_movement_close_open']) / 2

    # Smooth the Average using a 3-day Exponential Moving Average
    df['smoothed_price_movement'] = df['avg_price_movement'].ewm(span=3, adjust=False).mean()

    # Confirm with Volume
    df['volume_change_percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Weighted Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_price_movement'] * df['volume_change_percentage']

    return df['final_alpha_factor']
```
