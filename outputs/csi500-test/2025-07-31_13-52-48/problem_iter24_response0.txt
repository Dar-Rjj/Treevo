import numpy as np

    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)).abs()
    df['amount_change'] = (df['amount'] - df['amount'].shift(1)).abs()
    
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    
    df = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_vol_weighted_mom'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['daily_price_change'].rolling(window=20).std()
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_vol_weighted_mom'] = df['rolling_vol_weighted_mom'] / df['price_volatility']
    
    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['ma_60_adjusted_vol_weighted_mom'] = df['adjusted_vol_weighted_mom'].rolling(window=60).mean()
    df['final_momentum_factor'] = df['adjusted_vol_weighted_mom'] - df['ma_60_adjusted_vol_weighted_mom']

    df['intraday_volatility'] = df['high'] - df['low']

    df['vol_weighted_intraday_vol'] = df['intraday_volatility'] * df['volume']
    df['moving_avg_vol_weighted_intraday_vol'] = df['vol_weighted_intraday_vol'].rolling(window=10).mean()

    df['intraday_reversal_score'] = df['close_to_open_return'] / df['moving_avg_vol_weighted_intraday_vol']
