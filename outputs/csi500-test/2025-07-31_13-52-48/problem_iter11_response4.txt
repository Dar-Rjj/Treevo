```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    
    volume_threshold = df['volume_change'].quantile(0.75)
    amount_threshold = df['amount_change'].quantile(0.75)
    
    df_filtered = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    # Calculate Intraday Volatility
    df_filtered['intraday_volatility'] = df_filtered['high'] - df_filtered['low']

    # Calculate Close-to-Open Return
    df_filtered['close_to_open_return'] = df_filtered['close'] - df_filtered['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df_filtered['vol_weighted_intraday_volatility'] = df_filtered['intraday_volatility'] * df_filtered['volume']
    df_filtered['vol_weighted_moving_avg_intraday_volatility'] = df_filtered['vol_weighted_intraday_volatility'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df_filtered['reversal_score'] = df_filtered['close_to_open_return'] / df_filtered['vol_weighted_moving_avg_intraday_volatility']

    # Apply Exponential Decay to Reversal Score
    half_life = 5
    df_filtered['decayed_reversal_score'] = df_filtered['reversal_score'].ewm(halflife=half_life, adjust=False).mean()

    # Calculate Intraday Range
    df_filtered['intraday_range'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['open']

    # Calculate Volume-Weighted Moving Average of Intraday Range
    df_filtered['vol_weighted_intraday_range'] = df_filtered['intraday_range'] * df_filtered['volume']
    df_filtered['vol_weighted_moving_avg_intraday_range'] = df_filtered['vol_weighted_intraday_range'].rolling(window=10).mean()

    # Combine Reversal Score and Intraday Range
    df_filtered['combined_score'] = df_filtered['decayed_reversal_score'] + df_filtered['vol_weighted_moving_avg_intraday_range']
    df_filtered['decayed_combined_score'] = df_filtered['combined_score'].ewm(halflife=half_life, adjust=False).mean()

    # Calculate Intraday Price Movement
    df_filtered['price_movement_1'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['open']
    df_filtered['price_movement_2'] = (df_filtered['close'] - df_filtered['open']) / df_filtered['open']
    df_filtered['avg_price_movement'] = (df_filtered['price_movement_1'] + df_filtered['price_movement_2']) / 2

    # Smooth the Average using a 3-day Exponential Moving Average
    df_filtered['smoothed_price_movement'] = df_filtered['avg_price_movement'].ewm(span=3, adjust=False).mean()

    # Confirm with Volume
    df_filtered['volume_change'] = df_filtered['volume'] - df_filtered['volume'].shift(1)
    df_filtered['volume_change_pct'] = df_filtered['volume_change'] / df_filtered['volume'].shift(1)

    # Weighted Final Alpha Factor
    df_filtered['final_alpha_factor'] = df_filtered['smoothed_price_movement'] * df_filtered['volume_change_pct']

    return df_filtered['final_alpha_factor']
```
