import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.ewm(span=14, min_periods=14).mean()

    # Adjust by Volume Volatility
    daily_volume_change = df['volume'].diff()
    volume_volatility = daily_volume_change.rolling(window=20).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    final_intraday_adjustment = intraday_momentum / volume_volatility

    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()

    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = daily_price_change * np.log(df['volume'])

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    rolling_sum_volume_weighted = volume_weighted_momentum.rolling(window=20).sum()

    # Adjust by Intraday High-Low Momentum
    adjusted_volume_weighted_momentum = rolling_sum_volume_weighted * final_intraday_adjustment

    # Incorporate Price Trend
    price_change = df['close'] - df['open'].shift(1)
    long_term_price_trend = df['close'].rolling(window=200).mean()
    trend_adjusted_momentum = adjusted_volume_weighted_momentum * price_change * long_term_price_trend

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = trend_adjusted_momentum - trend_adjusted_momentum.rolling(window=50).mean()

    return dynamic_momentum_oscillator
