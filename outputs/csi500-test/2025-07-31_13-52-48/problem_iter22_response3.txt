```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['momentum_5'] = df['log_return'].rolling(window=5).sum()
    df['vol_weighted_mom'] = df['log_return'] * df['volume']
    df['vol_weighted_mom_20'] = df['vol_weighted_mom'].rolling(window=20).sum()
    df['combined_momentum'] = df['momentum_5'] * df['vol_weighted_mom_20']

    # Apply Combined Volume and Amount Shock Filter
    df['abs_vol_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['abs_amt_change'] = abs(df['amount'] - df['amount'].shift(1))
    vol_75th = df['abs_vol_change'].quantile(0.75)
    amt_75th = df['abs_amt_change'].quantile(0.75)
    df['shock_filter'] = (df['abs_vol_change'] < vol_75th) & (df['abs_amt_change'] < amt_75th)

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_vol'] = df['intraday_volatility'] * df['volume']
    df['vol_weighted_intraday_vol_10'] = df['vol_weighted_intraday_vol'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['vol_weighted_intraday_vol_10']
    df['intraday_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=5).mean()

    # Adjust Combined Momentum by Daily Price Volatility
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['daily_price_volatility'] = df['daily_price_change'].rolling(window=20).std()
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['daily_price_volatility']

    # Calculate Momentum Difference
    df['adjusted_combined_momentum_20'] = df['adjusted_combined_momentum'].rolling(window=20).mean()
    df['adjusted_combined_momentum_60'] = df['adjusted_combined_momentum'].rolling(window=60).mean()
    df['momentum_difference'] = df['adjusted_combined_momentum_20'] - df['adjusted_combined_momentum_60']

    # Apply Shock Filter to the final factor
    df['final_factor'] = df['momentum_difference'] * df['shock_filter']

    return df['final_factor']
```
