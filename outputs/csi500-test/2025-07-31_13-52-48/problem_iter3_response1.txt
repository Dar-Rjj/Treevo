```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Apply Weighted Moving Average to High-Low Difference using Volume
    window_period = 10
    weighted_moving_avg = (high_low_diff * df['volume']).rolling(window=window_period).sum() / df['volume'].rolling(window=window_period).sum()
    
    # Calculate Price Momentum over the Same Window Period
    close_price_momentum = df['close'].pct_change(periods=window_period)
    
    # Integrate Volume into Price Momentum
    avg_volume = df['volume'].rolling(window=window_period).mean()
    volume_adjusted_momentum = close_price_momentum * avg_volume
    
    # Combine High-Low Difference and Price Momentum
    combined_signal = weighted_moving_avg + volume_adjusted_momentum
    
    return combined_signal
```
