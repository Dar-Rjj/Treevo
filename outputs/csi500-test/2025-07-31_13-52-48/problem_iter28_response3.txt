```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Simple Moving Average (SMA) of close prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate 20-day Rate of Change (ROC) of close prices
    roc_20 = df['close'].pct_change(periods=20)
    
    # Calculate 20-day Volume Moving Average
    vol_sma_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate 20-day Historical Volatility of close prices
    volatility_20 = df['close'].pct_change().rolling(window=20).std() * (252 ** 0.5)
    
    # Construct a composite factor using SMA, ROC, and volume
    # Weighted sum: 0.4 * SMA, 0.4 * ROC, 0.2 * Volume
    composite_factor = 0.4 * sma_20 + 0.4 * roc_20 + 0.2 * vol_sma_20
    
    return composite_factor
```
