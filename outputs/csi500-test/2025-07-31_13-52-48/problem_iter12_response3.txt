```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_vwm'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['price_change'].rolling(window=20).std()
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_vwm'] = df['rolling_sum_vwm'] / df['price_volatility']
    
    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['factor'] = df['adjusted_vwm'] - df['adjusted_vwm'].rolling(window=60).mean()
    
    return df['factor']
```
