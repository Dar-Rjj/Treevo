import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by (date).
    """
    # Calculate the 5-day moving average of the closing price
    df['close_5d_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day sum of the volume
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the 5-day sum of the amount
    df['amount_5d_sum'] = df['amount'].rolling(window=5).sum()
    
    # Calculate the daily range (high - low)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the 5-day average of the daily range
    df['range_5d_avg'] = df['daily_range'].rolling(window=5).mean()
    
    # Calculate the 5-day sum of the daily range
    df['range_5d_sum'] = df['daily_range'].rolling(window=5).sum()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['close_5d_ma'] * df['volume_5d_sum']) / (df['amount_5d_sum'] + df['range_5d_sum'])
    
    return df['alpha_factor']
