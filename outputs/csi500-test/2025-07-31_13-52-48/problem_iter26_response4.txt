```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum_ema'] = df['high_low_diff'].ewm(span=10, min_periods=10).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'].diff()
    df['volume_volatility'] = df['volume_change'].rolling(window=30, min_periods=1).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    df['intraday_momentum_adjusted'] = df['intraday_momentum_ema'] / df['volume_volatility']

    # Calculate Volume-Weighted Momentum
    df['price_change'] = df['close'].pct_change()
    df['log_volume'] = np.log(df['volume'])
    df['volume_weighted_momentum'] = df['price_change'] * df['log_volume']

    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_volume_weighted_momentum'] = df['volume_weighted_momentum'] * df['intraday_momentum_adjusted']

    # Incorporate Price Trend
    df['long_term_price_trend'] = df['close'].rolling(window=200, min_periods=1).mean()
    df['price_change_open_close'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['trend_adjusted_intraday_momentum'] = df['intraday_adjusted_volume_weighted_momentum'] * df['price_change_open_close'] * df['long_term_price_trend']

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_intraday_momentum'] - df['trend_adjusted_intraday_momentum'].rolling(window=50, min_periods=1).mean()

    return df['dynamic_momentum_oscillator']
```
