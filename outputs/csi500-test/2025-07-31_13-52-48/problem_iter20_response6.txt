import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the index (date) representing the factor values.
    """
    # Calculate the intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily range
    df['daily_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the cumulative intraday return over the last 5 days
    df['cumulative_intraday_return'] = df['intraday_return'].rolling(window=5).sum()
    
    # Calculate the factor
    df['factor'] = df['intraday_return'] * df['daily_range'] * df['relative_volume'] * df['cumulative_intraday_return']
    
    return df['factor']
