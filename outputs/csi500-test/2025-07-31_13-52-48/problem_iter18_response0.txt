```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    df['CloseHighRatio'] = df['close'] / df['high']
    df['CloseLowRatio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['CombinedRatio'] = 0.7 * df['CloseHighRatio'] + 0.3 * df['CloseLowRatio']
    
    # Calculate Intraday Price Movements
    df['HighLowDiff'] = df['high'] - df['low']
    df['HighLowPct'] = df['HighLowDiff'] / df['open']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['CloseOpenPct'] = df['CloseOpenDiff'] / df['open']
    
    # Average Intraday Price Movements
    df['AvgPctMove'] = (df['HighLowPct'] + df['CloseOpenPct']) / 2
    df['SmoothedPctMove'] = df['AvgPctMove'].ewm(span=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    df['VolumeAdjMomentum'] = df['PriceChange'] * np.sqrt(df['volume'])
    
    # Confirm with Volume
    df['VolChange'] = df['volume'] - df['volume'].shift(1)
    df['VolChangePct'] = df['VolChange'] / df['volume'].shift(1)
    df['WeightedFinalFactor'] = df['SmoothedPctMove'] * df['VolChangePct']
    
    # Apply Volume Shock Filter
    abs_vol_change = np.abs(df['volume'] - df['volume'].shift(1))
    volume_shock_threshold = abs_vol_change.quantile(0.85)
    df['WeightedFinalFactor'] = df['WeightedFinalFactor'].where(abs_vol_change < volume_shock_threshold, 0)
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    df['EMAIntradayReturn'] = df['IntradayReturn'].ewm(span=5).mean()
    
    # Incorporate Volatility Component
    df['IntradayRange'] = df['high'] - df['low']
    df['VolatilityAdjustedMomentum'] = df['VolumeAdjMomentum'] / df['IntradayRange'].rolling(window=5).std()
    
    # Calculate Price Momentum
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    df['PriceMomentum'] = df['LogReturn'].rolling(window=4).sum()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['WeightedFinalFactor'] + df['VolatilityAdjustedMomentum'] + df['PriceMomentum']
    
    return df['FinalAlphaFactor']
```
