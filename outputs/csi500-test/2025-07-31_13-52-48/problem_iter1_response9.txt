```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    alpha = 0.1  # Exponential Smoothing Factor
    df['intraday_momentum_ema'] = df['intraday_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Standard Deviation of Intraday Price Ranges
    n_days = 20  # Consider Last N Days
    df['intraday_volatility'] = df['intraday_range'].rolling(window=n_days).std()
    
    # Incorporate Volatility Component
    df['volatility_inverse'] = 1 / df['intraday_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['intraday_momentum_ema'] * df['volatility_inverse']
    
    return df['alpha_factor']
```
