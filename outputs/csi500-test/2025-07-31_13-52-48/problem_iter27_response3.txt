import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.ewm(span=10, adjust=False).mean()

    # Adjust by Volume Volatility
    volume_change = df['volume'].diff()
    volume_volatility = volume_change.rolling(window=10).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    adjusted_intraday_momentum = intraday_momentum / volume_volatility

    # Calculate Daily Price Change
    price_change = df['close'].diff()

    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = price_change * (df['volume'] ** (1/3))

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    rolling_volume_weighted_momentum = volume_weighted_momentum.rolling(window=20).sum()

    # Adjust by Intraday High-Low Momentum
    adjusted_volume_weighted_momentum = rolling_volume_weighted_momentum * adjusted_intraday_momentum

    # Incorporate Price Trend
    long_term_price_trend = df['close'].ewm(span=200, adjust=False).mean()
    price_trend = (df['close'] - df['open'].shift(1)) * (long_term_price_trend - df['close'])

    final_adjusted_momentum = adjusted_volume_weighted_momentum * price_trend

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = final_adjusted_momentum - final_adjusted_momentum.rolling(window=50).mean()

    return dynamic_momentum_oscillator
