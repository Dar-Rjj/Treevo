```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Intraday Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Intraday_Momentum'] = df['Log_Return'].rolling(window=5).mean()
    
    # Adjust by Volume Volatility
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Volatility'] = df['Volume_Change'].rolling(window=20).std()
    df['Adjusted_Momentum'] = df['Intraday_Momentum'] / df['Volume_Volatility']
    
    # Apply Combined Volume and Amount Shock Filter
    df['Abs_Volume_Change'] = abs(df['Volume_Change'])
    df['Abs_Amount_Change'] = abs(df['amount'] - df['amount'].shift(1))
    vol_75_percentile = df['Abs_Volume_Change'].quantile(0.75)
    amount_75_percentile = df['Abs_Amount_Change'].quantile(0.75)
    df_filtered = df[(df['Abs_Volume_Change'] < vol_75_percentile) & (df['Abs_Amount_Change'] < amount_75_percentile)]
    
    # Calculate Intraday Reversal Score
    df_filtered['Intraday_Reversal'] = (df_filtered['close'] - df_filtered['open']) * df_filtered['volume']
    df_filtered['Vol_Weighted_MA'] = df_filtered['Intraday_Reversal'].rolling(window=10).mean()
    df_filtered['Reversal_Score'] = df_filtered['Intraday_Reversal'] / df_filtered['Vol_Weighted_MA']
    
    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = 0.5 ** (1/half_life)
    df_filtered['Decayed_Reversal_Score'] = df_filtered['Reversal_Score'].ewm(alpha=(1 - decay_factor), adjust=False).mean()
    
    # Calculate and Interact Intraday Volatility with Volume
    df_filtered['Intraday_Volatility'] = (df_filtered['high'] - df_filtered['low']) * df_filtered['volume']
    
    # Calculate Intraday Price Movement
    df_filtered['Price_Movement1'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['open']
    df_filtered['Price_Movement2'] = (df_filtered['close'] - df_filtered['open']) / df_filtered['open']
    df_filtered['Average_Price_Movement'] = (df_filtered['Price_Movement1'] + df_filtered['Price_Movement2']) / 2
    
    # Smooth the Average using a 3-day Exponential Moving Average
    df_filtered['Smoothed_Price_Movement'] = df_filtered['Average_Price_Movement'].ewm(span=3, adjust=False).mean()
    
    # Confirm with Volume
    df_filtered['Volume_Change_Percentage'] = (df_filtered['volume'] - df_filtered['volume'].shift(1)) / df_filtered['volume'].shift(1)
    
    # Weighted Final Alpha Factor
    df_filtered['Final_Alpha_Factor'] = df_filtered['Smoothed_Price_Movement'] * df_filtered['Volume_Change_Percentage']
    
    # Calculate Close to High and Low Ratios
    df_filtered['CloseHighRatio'] = df_filtered['close'] / df_filtered['high']
    df_filtered['CloseLowRatio'] = df_filtered['close'] / df_filtered['low']
    
    # Combine Close to High and Low Ratios
    df_filtered['CombinedRatio'] = 0.7 * df_filtered['CloseHighRatio'] + 0.3 * df_filtered['CloseLowRatio']
    
    # Final Alpha Factor
    df_filtered['Final_Alpha_Factor'] = df_filtered['Final_Alpha_Factor'] * df_filtered['CombinedRatio']
    
    return df_filtered['Final_Alpha_Factor']
```
