```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    intraday_ema = df['intraday_return'].ewm(span=5, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    intraday_volatility = df['intraday_range'].rolling(window=20).std()
    intraday_momentum = intraday_ema * intraday_volatility
    
    # Introduce Enhanced Volume Impact
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_ema = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    df['volume_roc'] = df['volume'].pct_change(periods=5)
    final_alpha_factor = volume_adjusted_ema + df['volume_roc']
    
    # Calculate Daily Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    daily_momentum = df['log_return'].rolling(window=5).sum()
    
    # Apply Combined Volume and Amount Shock Filter
    df['abs_volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['abs_amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_quantile = df['abs_volume_change'].quantile(0.75)
    amount_quantile = df['abs_amount_change'].quantile(0.75)
    df_filtered = df[(df['abs_volume_change'] < volume_quantile) & (df['abs_amount_change'] < amount_quantile)]
    
    # Calculate Intraday Reversal Score
    df['close_to_open'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_volatility'] = (df['high'] - df['low']).ewm(span=5, adjust=False).mean() * df['volume'].ewm(span=5, adjust=False).mean()
    df['reversal_score'] = df['close_to_open'] / df['volume_weighted_volatility']
    reversal_ema = df['reversal_score'].ewm(halflife=5, adjust=False).mean()
    
    # Calculate Daily Price Volatility
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    daily_volatility = df['daily_price_change'].rolling(window=20).std()
    momentum_adjusted = (intraday_momentum + daily_momentum) / daily_volatility
    
    # Calculate Price Breakout Indicator
    high_ma = df['high'].rolling(window=20).mean()
    low_ma = df['low'].rolling(window=20).mean()
    price_breakout_score = (high_ma - low_ma) / high_ma
    
    # Combine Ratios
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    combined_ratios = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    final_alpha_factor += momentum_adjusted + combined_ratios + price_breakout_score + reversal_ema
    
    # Compute Dynamic Price Momentum Oscillator
    rolling_sum = final_alpha_factor.rolling(window=60).sum()
    moving_average = final_alpha_factor.rolling(window=60).mean()
    dynamic_oscillator = rolling_sum - moving_average
    
    return dynamic_oscillator
```
