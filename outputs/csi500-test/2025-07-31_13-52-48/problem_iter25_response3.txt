```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum_ema'] = df['high_low_diff'].ewm(span=10, adjust=False).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    df['intraday_momentum_adjusted'] = df['intraday_momentum_ema'] / df['volume_volatility']

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * np.log(df['volume'])

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()

    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_momentum'] = df['rolling_volume_weighted_momentum'] * df['intraday_momentum_adjusted']

    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['open'].shift(1)
    df['final_factor'] = df['intraday_adjusted_momentum'] * df['price_trend']

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['final_factor'] - df['final_factor'].rolling(window=50).mean()

    # Analyze Closing Prices
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['cross_above_sma'] = (df['close'] > df['sma_20']) & (df['close'].shift(1) <= df['sma_20'])
    df['cross_below_sma'] = (df['close'] < df['sma_20']) & (df['close'].shift(1) >= df['sma_20'])

    # Examine Volume Trends
    df['significant_volume_increase'] = (df['volume'] > df['volume'].shift(1) * 1.5)

    # Consider Open-High-Low-Close (OHLC) Relationships
    def detect_candlestick_pattern(row):
        open, high, low, close = row['open'], row['high'], row['low'], row['close']
        if open == close:
            return 'Doji'
        if close > open and (high - close) < (close - open) and (close - open) > (open - low):
            return 'Bullish_Hammer'
        if close < open and (high - open) > (open - close) and (open - close) > (close - low):
            return 'Bearish_Hammer'
        if close > open and (close - open) > (open - low) and (high - close) < (close - open):
            return 'Bullish_Engulfing'
        if close < open and (open - close) > (close - low) and (high - open) < (open - close):
            return 'Bearish_Engulfing'
        return 'None'

    df['candlestick_pattern'] = df.apply(detect_candlestick_pattern, axis=1)

    # Final factor combining all components
    final_factor = df['dynamic_momentum_oscillator'] * df['cross_above_sma'] * df['significant_volume_increase'] * (df['candlestick_pattern'] != 'None')

    return final_factor
```
