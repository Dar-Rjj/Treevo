```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.rolling(window=5).mean()

    # Adjust by Volume Volatility
    volume_change = df['volume'].diff()
    volume_volatility = volume_change.rolling(window=10).std()
    intraday_momentum_adjusted = intraday_momentum / volume_volatility

    # Calculate Daily Momentum
    daily_momentum = df['close'].diff()

    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = daily_momentum * intraday_momentum_adjusted

    # Rolling Sum of Volume-Adjusted Momentum
    rolling_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = rolling_volume_adjusted_momentum - rolling_volume_adjusted_momentum.rolling(window=60).mean()

    # Introduce Open-Close Price Difference
    open_close_diff = df['open'] - df['close']

    # Calculate Intraday Price Range Ratio
    price_range_ratio = open_close_diff / high_low_diff

    # Adjust Intraday Momentum by Price Range Ratio
    adjusted_intraday_momentum = intraday_momentum * price_range_ratio

    # Calculate Final Momentum Factor
    final_momentum_factor = adjusted_intraday_momentum + volume_adjusted_momentum

    # Compute Rolling Sum of Final Momentum
    rolling_final_momentum = final_momentum_factor.rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    final_dynamic_momentum_oscillator = rolling_final_momentum - rolling_final_momentum.rolling(window=60).mean()

    # Introduce Weighted Open-Close Price Difference
    weighted_open_close_diff = (df['open'] - df['close']) * df['volume']

    # Calculate Weighted Intraday Price Range Ratio
    weighted_price_range_ratio = weighted_open_close_diff / high_low_diff

    # Adjust Intraday Momentum by Weighted Price Range Ratio
    weighted_adjusted_intraday_momentum = intraday_momentum * weighted_price_range_ratio

    # Calculate Final Adjusted Momentum Factor
    final_adjusted_momentum_factor = weighted_adjusted_intraday_momentum + volume_adjusted_momentum

    # Compute Rolling Sum of Final Adjusted Momentum
    rolling_final_adjusted_momentum = final_adjusted_momentum_factor.rolling(window=20).sum()

    # Compute Final Dynamic Momentum Oscillator
    final_dynamic_momentum_oscillator = rolling_final_adjusted_momentum - rolling_final_adjusted_momentum.rolling(window=60).mean()

    return final_dynamic_momentum_oscillator
```
