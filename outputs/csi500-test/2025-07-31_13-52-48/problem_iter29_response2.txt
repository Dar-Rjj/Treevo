import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    
    df['daily_percentage_change'] = df['close'].pct_change()
    df['5_day_sum_pct_change'] = df['daily_percentage_change'].rolling(window=5).sum()
    
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['20_day_avg_volume']
    
    df['20_day_total_amount'] = df['amount'].rolling(window=20).sum()
    df['avg_20_day_amount'] = df['amount'].rolling(window=20).mean()
    df['amount_ratio'] = df['amount'] / df['avg_20_day_amount']
    
    df['20_day_std_daily_returns'] = df['daily_percentage_change'].rolling(window=20).std()
    df['14_day_avg_true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs().rolling(window=14).mean()
    
    df['20_day_avg_high_low'] = (df['high'] + df['low']) / 2
    df['20_day_diff_high_low'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Intraday and Volume Interaction
    df['intraday_price_range'] = df['high'] - df['low']
    df['close_to_open_ratio'] = df['close'] / df['open']
    df['intraday_volatility'] = df['close_to_open_ratio'] * df['intraday_price_range']
    df['intraday_volatility_with_volume'] = df['intraday_volatility'] * df['volume']
    
    # Alpha Factor: Synthesis of all components
    alpha_factor = (
        df['5_day_SMA'] +
        df['20_day_SMA'] +
        df['50_day_SMA'] +
        df['5_day_sum_pct_change'] +
        df['volume_ratio'] +
        df['amount_ratio'] +
        df['20_day_std_daily_returns'] +
        df['14_day_avg_true_range'] +
        df['20_day_avg_high_low'] +
        df['20_day_diff_high_low'] +
        df['intraday_price_range'] +
        df['close_to_open_ratio'] +
        df['intraday_volatility_with_volume']
    )
    
    return alpha_factor
