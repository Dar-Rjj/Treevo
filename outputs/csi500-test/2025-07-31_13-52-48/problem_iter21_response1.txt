import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided
    (open, high, low, close) prices, amount, and volume. The factor is designed to help predict
    future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the daily returns
    df['ret_ma_5'] = df['returns'].rolling(window=5).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['std_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the 10-day cumulative sum of the volume
    df['cum_vol_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 5-day exponential moving average of the amount
    df['ema_amount_5'] = df['amount'].ewm(span=5, adjust=False).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close'] - df['ma_10']) / df['std_20'] + \
                         df['ret_ma_5'] * (df['volume'] / df['cum_vol_10']) + \
                         (df['amount'] - df['ema_amount_5']) / df['ema_amount_5']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
