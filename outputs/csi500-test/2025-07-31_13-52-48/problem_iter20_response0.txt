```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the cumulative money flow over a 10-day rolling window
    cmf_10 = money_flow.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day moving average of the close price
    ma_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of the close price
    std_close_10 = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day volatility
    volatility_10 = std_close_10 / ma_close_10
    
    # Calculate the 10-day momentum of the close price
    momentum_10 = df['close'] / df['close'].shift(10) - 1
    
    # Combine the factors to create the final alpha factor
    alpha_factor = (cmf_10 + momentum_10) / (volatility_10 + 1e-6)
    
    return alpha_factor
```
