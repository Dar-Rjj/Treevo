import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['lagged_close'] = df['close'].shift(1)
    df['daily_momentum'] = df['close'] - df['lagged_close']
    
    # Calculate Volume-Adjusted Momentum
    df['volume_normalized'] = df['volume'] / df['volume'].rolling(window=30).mean()
    df['volume_adjusted_momentum'] = df['daily_momentum'] * df['volume_normalized']
    
    # Rolling Sum of Volume-Adjusted Momentum over 20 Days
    df['rolling_sum_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['moving_average_60_days'] = df['rolling_sum_volume_adjusted_momentum'].rolling(window=60).mean()
    df['dynamic_momentum_oscillator'] = df['rolling_sum_volume_adjusted_momentum'] - df['moving_average_60_days']
    
    return df['dynamic_momentum_oscillator']
