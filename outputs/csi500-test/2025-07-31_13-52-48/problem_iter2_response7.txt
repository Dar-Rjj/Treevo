```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_movement = (df['high'] - df['low']) / df['open']
    close_open_movement = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Movements
    avg_movement = (high_low_movement + close_open_movement) / 2
    smoothed_movement = avg_movement.ewm(span=3).mean()
    
    # Confirm with Volume
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    weighted_price_movement = smoothed_movement * volume_change
    
    # Incorporate Volatility Component
    intraday_price_range = df['high'] - df['low']
    volatility = intraday_price_range.rolling(window=5).std()
    inverse_volatility = 1 / volatility
    
    # Final Alpha Factor
    alpha_factor = weighted_price_movement * inverse_volatility
    
    return alpha_factor
```
