import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = np.abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = np.abs(df['amount'] - df['amount'].shift(1))

    volume_threshold = df['volume_change'].quantile(0.75)
    amount_threshold = df['amount_change'].quantile(0.75)

    df = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['vol_weighted_intraday_volatility_ma'] = df['vol_weighted_intraday_volatility'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['reversal_score'] = df['close_to_open_return'] / df['vol_weighted_intraday_volatility_ma']

    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = np.exp(np.log(0.5) / half_life)
    df['decayed_reversal_score'] = df['reversal_score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['vol_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['vol_weighted_intraday_range_ma'] = df['vol_weighted_intraday_range'].rolling(window=10).mean()

    # Combine Reversal Score and Intraday Range
    df['combined_score'] = df['decayed_reversal_score'] + df['intraday_range']
    df['decayed_combined_score'] = df['combined_score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Price Movement
    df['price_movement_high_low'] = (df['high'] - df['low']) / df['open']
    df['price_movement_close_open'] = (df['close'] - df['open']) / df['open']
    df['average_price_movement'] = (df['price_movement_high_low'] + df['price_movement_close_open']) / 2

    # Smooth the Average using a 3-day Simple Moving Average
    df['smoothed_price_movement'] = df['average_price_movement'].rolling(window=3).mean()

    # Confirm with Volume
    df['volume_change_percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Calculate Trend Signal
    df['sma'] = df['close'].rolling(window=10).mean()
    df['deviation_from_sma'] = df['close'] - df['sma']

    # Measure Overall Volatility
    df['daily_range'] = df['high'] - df['low']
    df['overall_volatility'] = df['daily_range'].rolling(window=10).sum()

    # Capture Market Sentiment
    df['net_change'] = df['close'] - df['open']
    df['positive_net_changes'] = df['net_change'].apply(lambda x: 1 if x > 0 else 0)
    df['negative_net_changes'] = df['net_change'].apply(lambda x: 1 if x < 0 else 0)
    df['positive_net_changes_rolling'] = df['positive_net_changes'].rolling(window=10).sum()
    df['negative_net_changes_rolling'] = df['negative_net_changes'].rolling(window=10).sum()
    df['market_sentiment'] = df['positive_net_changes_rolling'] - df['negative_net_changes_rolling']

    # Analyze Trading Volume
    df['volume_with_price_correlation'] = df['volume'].rolling(window=10).corr(df['close'])
