```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    lookback_period_momentum = 5
    df['intraday_momentum'] = df['high_low_diff'].rolling(window=lookback_period_momentum).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    lookback_period_volatility = 10
    df['volume_volatility'] = df['volume_change'].rolling(window=lookback_period_volatility).std()
    df['volume_adjusted_momentum'] = df['intraday_momentum'] / df['volume_volatility']

    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['volume'] * df['volume_adjusted_momentum']

    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_trend_adjusted_momentum'] = df['volume_adjusted_momentum'] * df['price_change']

    # Compute Dynamic Momentum Oscillator
    rolling_sum_days = 60
    df['dynamic_momentum_oscillator'] = df['price_trend_adjusted_momentum'].rolling(window=rolling_sum_days).sum() - df['price_trend_adjusted_momentum'].rolling(window=rolling_sum_days).mean()

    return df['dynamic_momentum_oscillator']
```
