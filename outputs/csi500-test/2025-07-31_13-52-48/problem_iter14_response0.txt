```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['vol_weighted_mom_20d_sum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = df['price_change'].rolling(window=20).std()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['intraday_momentum_adjusted'] = df['avg_high_low_diff'] / df['volume_volatility']
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adj_vol_weighted_mom'] = df['volume_weighted_momentum'] * df['intraday_momentum_adjusted']
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adj_vol_weighted_mom'] = df['vol_weighted_mom_20d_sum'] / df['daily_price_volatility']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['close'].shift(1)
    df['trend_adj_intraday_adj_vol_weighted_mom'] = df['intraday_adj_vol_weighted_mom'] * df['price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['adj_vol_weighted_mom'] - df['adj_vol_weighted_mom'].rolling(window=60).mean()
    
    return df['dynamic_momentum_oscillator']
```
