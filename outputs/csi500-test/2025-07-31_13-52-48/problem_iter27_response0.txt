```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, short_term_window=10, long_term_window=50, short_ema_span=12, long_ema_span=26, signal_span=9):
    # Calculate Price Momentum
    close_price = df['close']
    price_momentum = close_price / close_price.shift(lookback_period) - 1
    
    # Calculate Volume Momentum
    volume = df['volume']
    volume_momentum = volume / volume.shift(lookback_period) - 1
    
    # Calculate High-Low Range Momentum
    high_price = df['high']
    low_price = df['low']
    high_low_range_momentum = (high_price - low_price) / (high_price.shift(lookback_period) - low_price.shift(lookback_period)) - 1
    
    # Calculate Historical Volatility
    log_returns = np.log(close_price / close_price.shift(1))
    historical_volatility = np.sqrt(np.mean(log_returns.rolling(window=lookback_period).var()) * 252)
    
    # Calculate Money Flow Index (MFI)
    typical_price = (high_price + low_price + close_price) / 3
    raw_money_flow = typical_price * volume
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=lookback_period).sum()
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=lookback_period).sum()
    money_flow_index = positive_money_flow / (positive_money_flow + negative_money_flow)
    
    # Calculate Accumulation/Distribution Line (A/D Line)
    money_flow_multiplier = ((close_price - low_price) - (high_price - close_price)) / (high_price - low_price)
    ad_line = (money_flow_multiplier * volume).cumsum()
    
    # Calculate Moving Averages
    short_term_ma = close_price.rolling(window=short_term_window).mean()
    long_term_ma = close_price.rolling(window=long_term_window).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    ema_short = close_price.ewm(span=short_ema_span, adjust=False).mean()
    ema_long = close_price.ewm(span=long_ema_span, adjust=False).mean()
    macd_line = ema_short - ema_long
    signal_line = macd_line.ewm(span=signal_span, adjust=False).mean()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        price_momentum +
        volume_momentum +
        high_low_range_momentum +
        historical_volatility +
        money_flow_index +
        ad_line.pct_change() +
        (short_term_ma - long_term_ma) / long_term_ma +
        (macd_line - signal_line) / signal_line
    )
    
    return alpha_factor
```
