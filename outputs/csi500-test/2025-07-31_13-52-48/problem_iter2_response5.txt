import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    lookback_period_intraday = 5
    df['intraday_momentum'] = df['high_low_diff'].rolling(window=lookback_period_intraday).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    lookback_period_volatility = 10
    df['volume_volatility'] = df['volume_change'].rolling(window=lookback_period_volatility).std()
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] / df['volume_volatility']

    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)

    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['daily_momentum'] * df['adjusted_intraday_momentum']

    # Rolling Sum of Volume-Adjusted Momentum
    rolling_sum_period = 20
    df['rolling_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=rolling_sum_period).sum()

    # Compute Dynamic Momentum Oscillator
    moving_average_period = 60
    df['dynamic_momentum_oscillator'] = df['rolling_volume_adjusted_momentum'] - df['rolling_volume_adjusted_momentum'].rolling(window=moving_average_period).mean()

    return df['dynamic_momentum_oscillator'].dropna()
