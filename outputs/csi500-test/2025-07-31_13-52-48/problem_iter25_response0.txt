```python
import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(df, N=10):
    # Calculate the difference between today's close and open prices (Close-Open)
    df['Close_Open'] = df['close'] - df['open']
    
    # Calculate the difference between today's high and low prices (High-Low)
    df['High_Low'] = df['high'] - df['low']
    
    # Examine the ratio of today’s volume to the average of past N days’ volumes (Volume/AVG(Volume[t-N:t-1]))
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=N).mean().shift(1)
    
    # Consider the product of today’s Close-Open and Volume (Close-Open * Volume)
    df['Close_Open_Volume'] = df['Close_Open'] * df['volume']
    
    # Compute the sum of the last N days' Close-Open differences (SUM(Close-Open[t-N:t-1]))
    df['Sum_Close_Open'] = df['Close_Open'].rolling(window=N).sum().shift(1)
    
    # Measure the maximum High minus minimum Low over the last N days (MAX(High)-MIN(Low)[t-N:t-1])
    df['Max_High_Min_Low'] = df['high'].rolling(window=N).max() - df['low'].rolling(window=N).min()
    
    # Calculate the rate of change of Close price over the last N days (Close[t]/Close[t-N]-1)
    df['ROC_Close'] = (df['close'] / df['close'].shift(N) - 1) * 100
    
    # Determine the slope of the linear regression of Close price over the last N days
    df['Slope_Close'] = df['close'].rolling(window=N).apply(lambda x: linregress(np.arange(N), x).slope, raw=False)
    
    # Combine multiple indicators for a composite factor
    # Add the Close-Open difference, Volume, and Rate of Change (Close-Open + Volume + ROC(Close))
    df['Composite_Factor_1'] = df['Close_Open'] + df['volume'] + df['ROC_Close']
    
    # Use a weighted sum of selected sub-thoughts to form a final alpha factor
    df['Final_Alpha_Factor'] = (
        0.4 * df['Close_Open'] +
        0.3 * df['Volume_Ratio'] +
        0.2 * df['Sum_Close_Open'] +
        0.1 * df['Slope_Close']
    )
    
    return df['Final_Alpha_Factor']
```
