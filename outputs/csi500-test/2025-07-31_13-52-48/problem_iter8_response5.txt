```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor that could help predict future stock returns.
    
    The function takes a DataFrame with columns: open, high, low, close, amount, volume,
    and returns a Series of the calculated alpha factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with market data, indexed by date.
    
    Returns:
    pd.Series: Series of the calculated alpha factor, indexed by date.
    """
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the EMA of money flow with a window of 14 days
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the EMA of volume with a window of 14 days
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the normalized money flow
    norm_money_flow = ema_money_flow / ema_volume
    
    # Calculate the price momentum as the ratio of current close to the 14-day EMA of close
    close_ema = df['close'].ewm(span=14, adjust=False).mean()
    price_momentum = df['close'] / close_ema - 1
    
    # Calculate the final alpha factor as the product of normalized money flow and price momentum
    alpha_factor = norm_money_flow * price_momentum
    
    return alpha_factor
```
