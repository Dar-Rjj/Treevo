```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume'].shift(1)
    
    # Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_vol_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Compute Dynamic Volume-Weighted Momentum Oscillator
    df['vol_weighted_momentum_oscillator'] = df['rolling_sum_vol_weighted_momentum'] - df['rolling_sum_vol_weighted_momentum'].rolling(window=60).mean()
    
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=10, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_price_range'] = (df['high'] - df['low']) / df['close']
    df['std_intraday_price_range'] = df['intraday_price_range'].rolling(window=20).std()
    df['adjusted_intraday_momentum'] = df['ema_intraday_return'] / df['std_intraday_price_range']
    
    # Introduce Volume Impact
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['final_ema_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['vol_weighted_momentum_oscillator'] + df['adjusted_intraday_momentum'] + df['final_ema_intraday_return']
    
    return df['alpha_factor']
```
