```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum_ema'] = df['high_low_diff'].ewm(span=10, adjust=False).mean()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'].diff()
    df['volume_volatility'] = df['volume_change'].rolling(window=10).apply(lambda x: x.abs().mean(), raw=True)
    df['final_adjustment'] = df['intraday_momentum_ema'] / df['volume_volatility']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * np.log(df['volume'])
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_momentum'] = df['rolling_volume_weighted_momentum'] * df['final_adjustment']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['open'].shift(1)
    df['trend_adjusted_momentum'] = df['intraday_adjusted_momentum'] * df['price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_momentum'] - df['trend_adjusted_momentum'].rolling(window=50).mean()
    
    # Analyze Closing Prices
    df['moving_avg_20'] = df['close'].rolling(window=20).mean()
    df['price_cross_above'] = (df['close'] > df['moving_avg_20']) & (df['close'].shift(1) <= df['moving_avg_20'])
    df['price_cross_below'] = (df['close'] < df['moving_avg_20']) & (df['close'].shift(1) >= df['moving_avg_20'])
    
    # Examine Volume Trends
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1) * 1.5)
    
    # Consider Open-High-Low-Close (OHLC) Relationships
    df['doji'] = (df['high'] - df['low']) > 0.01 * (df['high'] + df['low']) / 2 and abs(df['close'] - df['open']) / (df['high'] - df['low']) < 0.1
    df['hammer'] = (df['high'] - df['low']) > 0.01 * (df['high'] + df['low']) / 2 and (df['close'] - df['low']) / (df['high'] - df['low']) > 0.6 and (df['open'] - df['low']) / (df['high'] - df['low']) > 0.6
    df['engulfing'] = (df['close'] > df['open']) & (df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1))
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['dynamic_momentum_oscillator'] * (df['price_cross_above'].astype(int) - df['price_cross_below'].astype(int)) * df['volume_increase'].astype(int) * (df['doji'].astype(int) + df['hammer'].astype(int) + df['engulfing'].astype(int))
    
    return df['alpha_factor']
```
