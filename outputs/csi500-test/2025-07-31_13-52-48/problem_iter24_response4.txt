import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.rolling(window=20).mean()

    # Adjust by Volume Volatility
    daily_volume_change = df['volume'].diff()
    volume_volatility = daily_volume_change.rolling(window=20).std()
    adjusted_intraday_momentum = intraday_momentum / volume_volatility

    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)

    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = daily_price_change * np.sqrt(df['volume'])

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    rolling_sum_volume_weighted_momentum = volume_weighted_momentum.rolling(window=20).sum()

    # Adjust by Intraday High-Low Momentum
    intraday_adjusted_momentum = rolling_sum_volume_weighted_momentum * adjusted_intraday_momentum

    # Incorporate Price Trend
    price_change = df['close'] - df['open'].shift(1)
    long_term_trend = df['close'].ewm(span=200, adjust=False).mean()
    trend_adjusted_momentum = intraday_adjusted_momentum * price_change * long_term_trend

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = trend_adjusted_momentum - trend_adjusted_momentum.rolling(window=50).mean()

    return dynamic_momentum_oscillator
