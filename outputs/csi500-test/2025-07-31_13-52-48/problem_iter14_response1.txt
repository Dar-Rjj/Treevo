```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    money_flow_10d_avg = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of volume
    volume_10d_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate the normalized money flow
    normalized_money_flow = money_flow / volume_10d_avg
    
    # Calculate the 10-day moving average of the normalized money flow
    normalized_money_flow_10d_avg = normalized_money_flow.rolling(window=10).mean()
    
    # Calculate the final factor
    factor = normalized_money_flow / normalized_money_flow_10d_avg - 1
    
    return factor
```
