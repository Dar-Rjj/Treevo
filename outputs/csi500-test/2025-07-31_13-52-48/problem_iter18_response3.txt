import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum_ema'] = df['high_low_diff'].ewm(span=10, adjust=False).mean()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    df['intraday_momentum_adjusted'] = df['intraday_momentum_ema'] / df['volume_volatility']
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * np.log(df['volume'])
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_momentum'] = df['rolling_volume_weighted_momentum'] * df['intraday_momentum_adjusted']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['open'].shift(1)
    df['trend_adjusted_momentum'] = df['intraday_adjusted_momentum'] * df['price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_momentum'] - df['trend_adjusted_momentum'].rolling(window=50).mean()
    
    return df['dynamic_momentum_oscillator'].dropna()
