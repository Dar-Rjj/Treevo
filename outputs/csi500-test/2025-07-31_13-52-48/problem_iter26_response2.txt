```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Apply Enhanced Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    df['filter'] = (df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)
    
    # Average Intraday High-Low Difference Over Period
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=7).mean()
    
    # Adjust by Volume Volatility
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['daily_volume_change'].rolling(window=15).std()
    df['adjusted_high_low_diff'] = df['avg_high_low_diff'] / df['volume_volatility']
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['log_return'] * np.sqrt(df['volume'])
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_vol_weighted_mom'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['price_volatility'] = df['log_return'].rolling(window=30).std()
    df['adjusted_rolling_vol_weighted_mom'] = df['rolling_sum_vol_weighted_mom'] / df['price_volatility']
    
    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['60_day_ma_adjusted_vol_weighted_mom'] = df['adjusted_rolling_vol_weighted_mom'].rolling(window=60).mean()
    df['final_vol_weighted_mom'] = df['adjusted_rolling_vol_weighted_mom'] - df['60_day_ma_adjusted_vol_weighted_mom']
    
    # Combine Intraday and Open-Close Price Movements
    df['intraday_price_movement'] = (df['high'] - df['low']) / df['open']
    df['open_close_price_movement'] = (df['close'] - df['open']) / df['open']
    df['average_price_movement'] = (df['intraday_price_movement'] + df['open_close_price_movement']) / 2
    df['smoothed_average_price_movement'] = df['average_price_movement'].ewm(span=5).mean()
    
    # Calculate Intraday Price Range Ratio
    df['open_close_diff'] = df['close'] - df['open']
    df['high_low_range'] = df['high'] - df['low']
    df['price_range_ratio'] = df['open_close_diff'] / df['high_low_range']
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_intraday_high_low_diff'] = df['avg_high_low_diff'] * df['price_range_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_intraday_high_low_diff'] + df['final_vol_weighted_mom']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_sum_final_momentum'] = df['final_momentum_factor'].rolling(window=25).sum()
    
    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['open'].shift(1)
    df['long_term_price_trend'] = df['close'].ewm(span=200).mean()
    df['trend_adjusted_final_momentum'] = df['rolling_sum_final_momentum'] * df['price_change'] * df['long_term_price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_final_momentum'] - df['trend_adjusted_final_momentum'].rolling(window=60).mean()
    
    return df['dynamic_momentum_oscillator']
```
