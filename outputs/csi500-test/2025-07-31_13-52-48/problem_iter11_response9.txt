import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as 'date'.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor for each date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['range'] = df['high'] - df['low']
    
    # Calculate the average price
    df['avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()
    
    # Calculate the relative strength of the close price to the VWAP
    df['close_to_vwap'] = df['close'] / df['vwap']
    
    # Calculate the relative strength of the close price to the average price
    df['close_to_avg'] = df['close'] / df['avg_price']
    
    # Calculate the 10-day moving average of the daily returns
    df['ma_return_10'] = df['return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily returns
    df['std_return_10'] = df['return'].rolling(window=10).std()
    
    # Calculate the 10-day moving average of the intraday range
    df['ma_range_10'] = df['range'].rolling(window=10).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_to_vwap'] * 
        df['close_to_avg'] * 
        df['ma_return_10'] / 
        df['std_return_10'] * 
        df['ma_range_10']
    )
    
    # Drop any rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
