```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_vol_weighted_mom'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['price_change'].rolling(window=20).std()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['intraday_adj_factor'] = df['avg_high_low_diff'] / df['volume_volatility']
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adj_vol_weighted_mom'] = df['volume_weighted_momentum'] * df['intraday_adj_factor']
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_rolling_vol_weighted_mom'] = df['rolling_vol_weighted_mom'] / df['price_volatility']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['close'].shift(1)
    df['momentum_adjusted_by_price_trend'] = df['intraday_adj_vol_weighted_mom'] * df['price_trend']
    
    # Combine Intraday and Open-Close Price Movements
    df['perc_price_movement_intraday'] = (df['high'] - df['low']) / df['open']
    df['perc_price_movement_open_close'] = (df['close'] - df['open']) / df['open']
    df['avg_perc_movement'] = (df['perc_price_movement_intraday'] + df['perc_price_movement_open_close']) / 2
    df['smoothed_avg_perc_movement'] = df['avg_perc_movement'].ewm(span=3).mean()
    
    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Range Ratio
    df['open_close_range'] = df['close'] - df['open']
    df['high_low_range'] = df['high'] - df['low']
    df['price_range_ratio'] = df['open_close_range'] / df['high_low_range']
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_high_low_diff'] = df['high_low_diff'] * df['price_range_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_high_low_diff'] + df['momentum_adjusted_by_price_trend']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_final_momentum'] - df['rolling_final_momentum'].rolling(window=60).mean()
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['close_to_open_ratio'] * df['intraday_price_range']
    
    # Interact Intraday Volatility with Volume
    df['intraday_volatility_interacted_with_volume'] = df['intraday_volatility'] * df['volume']
    
    return df['intraday_volatility_interacted_with_volume']
```
