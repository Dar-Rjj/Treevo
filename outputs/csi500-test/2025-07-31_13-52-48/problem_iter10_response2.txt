```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Compute Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Volume-Adjusted Momentum
    return_series = df['close'].pct_change()
    volume_adjusted_momentum = return_series * df['volume']
    
    # Calculate Price Volatility
    prices = df[['open', 'high', 'low', 'close']]
    price_volatility = prices.std(axis=1)
    
    # Combine Factors
    factor = combined_ratio + volume_adjusted_momentum + price_volatility
    
    return factor
```
