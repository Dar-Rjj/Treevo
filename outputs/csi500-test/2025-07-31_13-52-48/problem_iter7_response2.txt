import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['20_day_rolling_sum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Compute 60-Day Moving Average of Volume-Weighted Momentum
    df['60_day_moving_avg'] = df['volume_weighted_momentum'].rolling(window=60).mean()
    
    # Compute Dynamic Volume-Weighted Momentum Oscillator
    df['dynamic_volume_weighted_momentum_oscillator'] = df['20_day_rolling_sum'] - df['60_day_moving_avg']
    
    # Return the factor values as a pandas Series indexed by (date)
    return df['dynamic_volume_weighted_momentum_oscillator'].dropna()
