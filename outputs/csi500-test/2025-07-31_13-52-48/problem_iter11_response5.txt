```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_vol_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_high_low_momentum'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Adjust by Volume Volatility
    df['daily_volume_change'] = df['volume'].diff()
    df['volume_volatility'] = df['daily_volume_change'].rolling(window=10).std()
    
    df['final_adjustment'] = df['intraday_high_low_momentum'] / df['volume_volatility']
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adjusted_vol_weighted_momentum'] = df['volume_weighted_momentum'] * df['final_adjustment']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'].diff()
    df['trend_adjusted_momentum'] = df['intraday_adjusted_vol_weighted_momentum'] * df['price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['rolling_sum_adj_vol_weighted_momentum'] = df['trend_adjusted_momentum'].rolling(window=20).sum()
    df['moving_avg_60_days'] = df['trend_adjusted_momentum'].rolling(window=60).mean()
    df['dynamic_momentum_oscillator'] = df['rolling_sum_adj_vol_weighted_momentum'] - df['moving_avg_60_days']
    
    return df['dynamic_momentum_oscillator']
```
