```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    combined_momentum = log_returns.rolling(window=5).sum() * df['volume'].rolling(window=20).sum()

    # Apply Combined Volume and Amount Shock Filter
    abs_volume_change = (df['volume'] - df['volume'].shift(1)).abs()
    abs_amount_change = (df['amount'] - df['amount'].shift(1)).abs()
    
    volume_75th = abs_volume_change.quantile(0.75)
    amount_75th = abs_amount_change.quantile(0.75)
    
    shock_filter = (abs_volume_change < volume_75th) & (abs_amount_change < amount_75th)
    combined_momentum_filtered = combined_momentum.where(shock_filter, 0)

    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    vol_weighted_intraday_volatility = (intraday_volatility * df['volume']).rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    reversal_score = close_to_open_return / vol_weighted_intraday_volatility
    reversal_score_exp_decay = reversal_score.ewm(halflife=5).mean()
    
    # Adjust Combined Momentum by Daily Price Volatility
    daily_price_change = df['close'] - df['close'].shift(1)
    daily_price_volatility = daily_price_change.rolling(window=20).std()
    adjusted_combined_momentum = combined_momentum_filtered / daily_price_volatility
    
    # Subtract 60-Day Moving Average of Adjusted Combined Momentum from 20-Day Adjusted Combined Momentum
    adjusted_combined_momentum_20 = adjusted_combined_momentum.rolling(window=20).mean()
    adjusted_combined_momentum_60 = adjusted_combined_momentum.rolling(window=60).mean()
    final_factor = adjusted_combined_momentum_20 - adjusted_combined_momentum_60
    
    return final_factor
```
