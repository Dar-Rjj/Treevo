import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range: (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative amount: amount / volume
    df['relative_amount'] = df['amount'] / df['volume']
    
    # Calculate the 10-day moving average of the relative range
    df['ma_relative_range_10d'] = df['relative_range'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the relative amount
    df['ma_relative_amount_10d'] = df['relative_amount'].rolling(window=10).mean()
    
    # Calculate the alpha factor: (daily_return * ma_relative_range_10d) + ma_relative_amount_10d
    df['alpha_factor'] = (df['daily_return'] * df['ma_relative_range_10d']) + df['ma_relative_amount_10d']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
