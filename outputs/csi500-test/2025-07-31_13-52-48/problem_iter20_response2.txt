import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_10d_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the typical price
    typical_price_10d_ma = typical_price.rolling(window=10).mean()
    
    # Calculate the Money Flow Index (MFI) ratio
    mfi_ratio = money_flow_10d_ma / (typical_price_10d_ma * volume_10d_ma)
    
    # Calculate the 10-day rolling standard deviation of the typical price
    typical_price_10d_std = typical_price.rolling(window=10).std()
    
    # Calculate the Z-score of the typical price
    z_score = (typical_price - typical_price_10d_ma) / typical_price_10d_std
    
    # Combine the MFI ratio and Z-score to create the alpha factor
    alpha_factor = mfi_ratio * z_score
    
    return alpha_factor
