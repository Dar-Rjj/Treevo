```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    volume_weighted_volatility = (intraday_volatility * df['volume']).rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    reversal_score = close_to_open_return / volume_weighted_volatility
    
    # Apply Exponential Decay to Reversal Score with a half-life of 5 days
    decay_factor = np.exp(np.log(0.5) / 5)
    exponential_decay = reversal_score.ewm(alpha=decay_factor, adjust=False).mean()
    
    return exponential_decay
```
