def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['HighLowDiff'] = df['high'] - df['low']
    df['IntradayMomentum'] = df['HighLowDiff'].rolling(window=20).mean()
    
    # Adjust by Volume Volatility
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['VolumeVolatility'] = df['VolumeChange'].rolling(window=20).std()
    df['IntradayMomentumAdjusted'] = df['IntradayMomentum'] / df['VolumeVolatility']
    
    # Calculate Daily Price Change
    df['DailyPriceChange'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['VolumeWeightedMomentum'] = df['DailyPriceChange'] * np.sqrt(df['volume'])
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['RollingSumVWM'] = df['VolumeWeightedMomentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['IntradayAdjustedVWM'] = df['RollingSumVWM'] * df['IntradayMomentumAdjusted']
    
    # Incorporate Price Trend
    df['PriceChange'] = df['close'] - df['open'].shift(1)
    df['LongTermPriceTrend'] = df['close'].ewm(span=200, adjust=False).mean()

    df['DynamicMomentumOscillator'] = df['IntradayAdjustedVWM_Trend'] - df['IntradayAdjustedVWM_Trend'].rolling(window=50).mean()
