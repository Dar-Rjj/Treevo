```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    intraday_momentum = df['high_low_diff'].rolling(window=20).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_volatility = df['volume_change'].rolling(window=20).std()
    final_adjustment = intraday_momentum / volume_volatility

    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = intraday_momentum * final_adjustment

    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['close'].shift(1)
    trend_adjusted_momentum = volume_adjusted_momentum * df['price_change']

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = trend_adjusted_momentum - trend_adjusted_momentum.rolling(window=60).mean()

    return dynamic_momentum_oscillator
```
