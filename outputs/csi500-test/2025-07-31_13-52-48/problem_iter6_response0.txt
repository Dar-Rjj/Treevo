```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    ema_alpha = 0.1  # Exponential Smoothing Factor for Intraday Momentum
    ema_alpha_volume = 0.2  # Exponential Smoothing Factor for Volume Weighted Intraday Return
    n_days_volatility = 5  # Number of days to consider for Volatility
    m_days_volume_roc = 3  # Number of days to consider for Volume Rate of Change
    
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['ema_intraday_return'] = df['intraday_return'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_price_range'] = (df['high'] - df['low']) / df['open']
    df['std_intraday_price_range'] = df['intraday_price_range'].rolling(window=n_days_volatility).std()
    df['volatility_adjusted_ema'] = df['ema_intraday_return'] / (df['std_intraday_price_range'] + 1e-8)
    
    # Introduce Enhanced Volume Impact
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * df['volume'] / df['open']
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(alpha=ema_alpha_volume, adjust=False).mean()
    
    # Integrate Volume Trend
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=m_days_volume_roc)
    
    # Combine all components to form the final alpha factor
    df['alpha_factor'] = df['ema_volume_weighted_intraday_return'] * (1 + df['volume_rate_of_change'])
    
    return df['alpha_factor']
```
