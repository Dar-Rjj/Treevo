```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Average Over Period (5 days)
    lookback_period_intraday = 5
    df['intraday_momentum'] = df['high_low_diff'].rolling(window=lookback_period_intraday).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Volatility (10 days)
    lookback_period_volatility = 10
    df['volume_volatility'] = df['volume_change'].rolling(window=lookback_period_volatility).std()
    
    # Final Adjustment: Divide Intraday Momentum by Volume Volatility
    df['adjusted_momentum'] = df['intraday_momentum'] / (df['volume_volatility'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['volume']
    
    # Compute Dynamic Momentum Oscillator
    rolling_sum_window = 20  # Example window for rolling sum, can be adjusted
    moving_average_window = 60  # 60 days moving average
    df['dynamic_momentum_oscillator'] = df['volume_adjusted_momentum'].rolling(window=rolling_sum_window).sum() - df['volume_adjusted_momentum'].rolling(window=moving_average_window).mean()
    
    # Return the final factor as a pandas Series
    return df['dynamic_momentum_oscillator']
```
