```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Difference
    df['intraday_high_low_diff'] = df['high'] - df['low']
    
    # Average Intraday High-Low Difference over 5 days
    df['avg_intraday_high_low_diff'] = df['intraday_high_low_diff'].rolling(window=5).mean()
    
    # Calculate Weighted Intraday Volatility
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Volume-Weighted Moving Average of Intraday Volatility over 10 days
    df['vwma_intraday_volatility'] = df['weighted_intraday_volatility'].rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['vwma_intraday_volatility']
    
    # Apply Exponential Decay with Half-life of 5 Days
    df['smoothed_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=5, ignore_na=True).mean()
    
    # Calculate Price Momentum
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['price_momentum'] = df['log_returns'].rolling(window=5).sum()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Volatility over 10 days
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    
    # Adjust Intraday Momentum by Volume Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['volume_volatility']
    
    # Calculate Close to High Ratio
    df['close_to_high_ratio'] = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['combined_close_ratios'] = 0.5 * df['close_to_high_ratio'] + 0.5 * df['close_to_low_ratio']
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Apply Volume Shock Filter
    df['abs_volume_change'] = (df['volume'] - df['volume'].shift(1)).abs()
    volume_shock_threshold = df['abs_volume_change'].quantile(0.75)
    df['filtered_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].where(df['abs_volume_change'] < volume_shock_threshold)
    
    # Calculate Intraday Price Movement
    df['percent_price_movement_hl'] = (df['high'] - df['low']) / df['open']
    df['percent_price_movement_co'] = (df['close'] - df['open']) / df['open']
    df['avg_percent_price_movement'] = 0.5 * df['percent_price_movement_hl'] + 0.5 * df['percent_price_movement_co']
    
    # Smooth the Average using a 3-day Exponential Moving Average
    df['smoothed_avg_percent_price_movement'] = df['avg_percent_price_movement'].ewm(span=3, ignore_na=True).mean()
    
    # Synthesize Combined Factors
    df['alpha_factor'] = (
        df['smoothed_reversal_score'] +
        df['adjusted_price_momentum'] +
        df['filtered_volume_adjusted_momentum'].fillna(0) +
        df['combined_close_ratios'] +
        df['smoothed_avg_percent_price_movement']
    )
    
    return df['alpha_factor']
```
