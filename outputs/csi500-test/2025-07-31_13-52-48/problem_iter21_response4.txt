import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)).abs()
    df['amount_change'] = (df['amount'] - df['amount'].shift(1)).abs()
    
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    
    df = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()

    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = df['daily_price_change'].rolling(window=20).std()

    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_volume_weighted_momentum'] = df['rolling_sum_volume_weighted_momentum'] / df['daily_price_volatility']

    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['adjusted_momentum_60_ma'] = df['adjusted_volume_weighted_momentum'].rolling(window=60).mean()
    df['momentum_factor'] = df['adjusted_volume_weighted_momentum'] - df['adjusted_momentum_60_ma']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['volume_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['volatility_moving_avg'] = df['volume_weighted_intraday_volatility'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['reversal_score'] = df['close_to_open_return'] / df['volatility_moving_avg']

    # Apply Exponential Decay to Reversal Score
    half_life = 4
    decay_factor = 0.5 ** (1 / half_life)
    df['decayed_reversal_score'] = df['reversal_score'] * decay_factor ** (np.arange(len(df)))

    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['range_moving_avg'] = df['volume_weighted_intraday_range'].rolling(window=10).mean()

    # Combine Reversal Score and Intraday Range
    df['combined_score'] = df['decayed_reversal_score'] + df['intraday_range']
    df['decayed_combined_score'] = df['combined_score'] * decay_factor ** (np.arange(len(df)))

    # Calculate Intraday Price Movement
    df['pct_high_low'] = (df['high'] - df['low']) / df['open']
    df['pct_close_open'] = (df['close'] - df['open']) / df['open']
    df['avg_pct_movement'] = (df['pct_high_low'] + df['pct_close_open']) / 2

    # Smooth the Average using a 3-day Exponential Moving Average
    df['smoothed_price_movement'] = df['avg_pct_movement'].ewm(span=3, adjust=False).mean()

    # Confirm with Volume
    df['volume_change_percent'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Weighted Final Alpha Factor
    df['alpha_factor'] = df['smoothed_price_movement'] * df['volume_change_percent']

    return df['alpha_factor
