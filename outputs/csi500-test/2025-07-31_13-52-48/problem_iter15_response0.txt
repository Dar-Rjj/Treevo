```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=6).sum().shift(1)
    
    # Apply Combined Volume and Amount Shock Filter
    df['abs_vol_change'] = np.abs(df['volume'] - df['volume'].shift(1))
    df['abs_amount_change'] = np.abs(df['amount'] - df['amount'].shift(1))
    
    vol_threshold = df['abs_vol_change'].quantile(0.8)
    amount_threshold = df['abs_amount_change'].quantile(0.8)
    
    df['filtered_days'] = (df['abs_vol_change'] < vol_threshold) & (df['abs_amount_change'] < amount_threshold)
    df['filtered_price_momentum'] = df['price_momentum'] * df['filtered_days']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['vol_weighted_moving_avg'] = df['vol_weighted_intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['vol_weighted_moving_avg']
    
    # Apply Exponential Decay with Half-life of 3 Days
    half_life = 3
    decay_factor = np.exp(-np.log(2) / half_life)
    df['intraday_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=half_life, adjust=False).mean()
    
    # Combine the filtered price momentum and intraday reversal score
    df['alpha_factor'] = df['filtered_price_momentum'] + df['intraday_reversal_score']
    
    return df['alpha_factor']
```
