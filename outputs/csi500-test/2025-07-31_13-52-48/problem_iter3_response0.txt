```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    alpha = 0.1  # Exponential Smoothing Factor
    intraday_momentum = intraday_return.ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Compute Standard Deviation of Intraday Price Ranges
    n_days = 30  # Consider Last N Days
    std_intraday_price_range = intraday_price_range.rolling(window=n_days).std()
    volatility_component = 1 / std_intraday_price_range
    
    # Adjust Intraday Momentum with Volatility Component
    adjusted_intraday_momentum = intraday_momentum * volatility_component
    
    # Calculate Volume Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] * df['volume']) / (df['open'] * df['volume'])
    
    # Adjust Exponential Moving Average with Volume Weighted Intraday Return
    beta = 0.2  # New Exponential Smoothing Factor
    final_factor = (adjusted_intraday_momentum + volume_weighted_intraday_return).ewm(alpha=beta, adjust=False).mean()
    
    return final_factor
```
