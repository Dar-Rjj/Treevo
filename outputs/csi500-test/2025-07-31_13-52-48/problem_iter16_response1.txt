```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    intraday_momentum = df['high_low_diff'].rolling(window=10).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_volatility = df['volume_change'].rolling(window=10).std()
    intraday_momentum_adjusted = intraday_momentum / volume_volatility

    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['close'].shift(1)
    intraday_momentum_with_trend = intraday_momentum_adjusted * df['price_change']

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = intraday_momentum_with_trend - intraday_momentum_with_trend.rolling(window=60).mean()

    return dynamic_momentum_oscillator
```
