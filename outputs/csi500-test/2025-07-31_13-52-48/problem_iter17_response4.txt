import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Interact Intraday Volatility with Volume
    intraday_volatility_volume = intraday_range * df['volume']
    
    # Apply Enhanced Combined Volume and Amount Shock Filter
    df['volume_change'] = df['volume'].diff().abs()
    df['amount_change'] = df['amount'].diff().abs()
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    filter_mask = (df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)
    filtered_df = df[filter_mask]
    
    # Calculate Price Momentum
    price_momentum = (np.log(filtered_df['close']).diff().rolling(window=5).sum()).shift(1)
    
    # Calculate Daily Price Change
    daily_price_change = filtered_df['close'].diff()
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = daily_price_change * filtered_df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    rolling_sum_volume_weighted_momentum = volume_weighted_momentum.rolling(window=20).sum()
    
    # Calculate Daily Price Volatility
    daily_price_volatility = daily_price_change.rolling(window=20).std()
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    adjusted_volume_weighted_momentum = rolling_sum_volume_weighted_momentum / daily_price_volatility
    
    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    adjusted_volume_weighted_momentum_60day_ma = adjusted_volume_weighted_momentum.rolling(window=60).mean()
    adjusted_momentum_diff = adjusted_volume_weighted_momentum - adjusted_volume_weighted_momentum_60day_ma
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = (filtered_df['close'] / filtered_df['open'] - 1) / (intraday_volatility_volume.rolling(window=20).mean())
    
    # Apply Exponential Decay to Reversal Score
    decay_factor = 0.5 ** (1 / 3)
    intraday_reversal_score_exp_decay = intraday_reversal_score.ewm(span=3, adjust=False).mean()
    
    # Calculate Intraday Range
    intraday_range_normalized = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Range
    volume_weighted_intraday_range = (intraday_range_normalized * df['volume']).rolling(window=10).mean()
    
    # Combine Reversal Score and Intraday Range
    combined_score = intraday_reversal_score_exp_decay + volume_weighted_intraday_range
    combined_score_exp_decay = combined_score.ewm(span=3, adjust=False).mean()
    
    return combined_score_exp_decay
