import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as (date).
                       
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / df['close'].shift(1)
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Calculate the money flow index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (df['daily_return'] 
                          + df['relative_range'] 
                          + (df['close'] - df['vwap']) / df['close'] 
                          + (50 - df['mfi']) / 50)
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
