```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff_5d'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['intraday_high_low_adjustment'] = df['avg_high_low_diff_5d'] / (df['volume_volatility'] + 1e-6)  # Avoid division by zero
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adjusted_volume_weighted_momentum'] = df['rolling_volume_weighted_momentum'] * df['intraday_high_low_adjustment']
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['price_volatility'] = df['price_change'].rolling(window=20).std()
    df['adjusted_volume_weighted_momentum'] = df['rolling_volume_weighted_momentum'] / (df['price_volatility'] + 1e-6)  # Avoid division by zero
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['close'].shift(1)
    df['trend_adjusted_momentum'] = df['intraday_adjusted_volume_weighted_momentum'] * df['price_trend']
    
    # Combine Intraday and Open-Close Price Movements
    df['percent_price_movement_high_low'] = (df['high'] - df['low']) / df['open']
    df['percent_price_movement_close_open'] = (df['close'] - df['open']) / df['open']
    df['average_percent_movement'] = (df['percent_price_movement_high_low'] + df['percent_price_movement_close_open']) / 2
    df['smoothed_average_percent_movement'] = df['average_percent_movement'].ewm(span=3).mean()
    
    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Range Ratio
    df['high_low_diff'] = df['high'] - df['low']
    df['open_close_diff'] = df['close'] - df['open']
    df['price_range_ratio'] = df['open_close_diff'] / (df['high_low_diff'] + 1e-6)  # Avoid division by zero
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_intraday_high_low_diff'] = df['high_low_diff'] * df['price_range_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_volume_weighted_momentum'] + df['trend_adjusted_momentum']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_sum_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['moving_avg_60d'] = df['final_momentum_factor'].rolling(window=60).mean()
    df['dynamic_momentum_oscillator'] = df['rolling_sum_final_momentum'] - df['moving_avg_60d']
    
    # Interact Intraday Volatility with Volume
    df['intraday_volatility'] = df['close'] / df['open'] + df['high_low_diff']
    df['intraday_volatility_with_volume'] = df['intraday_volatility'] * df['volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['dynamic_momentum_oscillator'] * df['intraday_volatility_with_volume']
    
    return df['alpha_factor']
```
