```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Apply Enhanced Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    df['shock_filter'] = (df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)
    
    # Average Intraday High-Low Difference Over Period
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=7).mean()
    
    # Adjust by Volume Volatility
    df['volume_volatility'] = df['volume_change'].rolling(window=15).std()
    df['adjusted_high_low_diff'] = df['avg_high_low_diff'] / df['volume_volatility']
    
    # Calculate Volume-Weighted Momentum
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['price_volatility'] = df['daily_price_change'].rolling(window=30).std()
    df['adjusted_volume_weighted_momentum'] = df['rolling_sum_volume_weighted_momentum'] / df['price_volatility']
    
    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['ma_60_adjusted_volume_weighted_momentum'] = df['adjusted_volume_weighted_momentum'].rolling(window=60).mean()
    df['momentum_factor'] = df['adjusted_volume_weighted_momentum'] - df['ma_60_adjusted_volume_weighted_momentum']
    
    # Combine Intraday and Open-Close Price Movements
    df['percent_movement_high_low'] = (df['high'] - df['low']) / df['open']
    df['percent_movement_close_open'] = (df['close'] - df['open']) / df['open']
    df['avg_percent_movement'] = (df['percent_movement_high_low'] + df['percent_movement_close_open']) / 2
    df['smoothed_avg_percent_movement'] = df['avg_percent_movement'].ewm(span=5).mean()
    
    # Calculate Intraday Price Range Ratio
    df['open_close_diff'] = df['close'] - df['open']
    df['intraday_range_ratio'] = df['open_close_diff'] / df['high_low_diff']
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_intraday_diff'] = df['avg_high_low_diff'] * df['intraday_range_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_intraday_diff'] + df['adjusted_volume_weighted_momentum']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_sum_final_momentum'] = df['final_momentum_factor'].rolling(window=25).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_sum_final_momentum'] - df['rolling_sum_final_momentum'].rolling(window=60).mean()
    
    return df['dynamic_momentum_oscillator'].dropna()
```
