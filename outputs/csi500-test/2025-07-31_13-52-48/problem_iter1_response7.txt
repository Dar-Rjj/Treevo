```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_movement'] = (df['high'] - df['low']) / df['open']
    df['close_open_movement'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Price Movements
    df['avg_movement'] = (df['high_low_movement'] + df['close_open_movement']) / 2
    df['smoothed_movement'] = df['avg_movement'].ewm(span=3, adjust=False).mean()
    
    # Confirm with Volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weighted Final Alpha Factor
    df['alpha_factor'] = df['smoothed_movement'] * df['volume_change']
    
    return df['alpha_factor']
```
