```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    intraday_momentum = df['log_return'].rolling(window=5).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_volatility = df['volume_change'].rolling(window=5).std()
    adjusted_momentum = intraday_momentum / volume_volatility

    # Apply Combined Volume and Amount Shock Filter
    abs_volume_change = (df['volume'] - df['volume'].shift(1)).abs()
    abs_amount_change = (df['amount'] - df['amount'].shift(1)).abs()
    percentile_75_volume = abs_volume_change.rolling(window=20).quantile(0.75)
    percentile_75_amount = abs_amount_change.rolling(window=20).quantile(0.75)
    shock_filter = (abs_volume_change < percentile_75_volume) & (abs_amount_change < percentile_75_amount)
    adjusted_momentum_filtered = adjusted_momentum.where(shock_filter, other=np.nan)

    # Calculate Intraday Reversal Score
    close_to_open_return = df['close'] - df['open']
    intraday_volatility = (df['high'] - df['low'])
    volume_weighted_intraday_volatility = (intraday_volatility * df['volume']).rolling(window=10).mean()
    reversal_score = close_to_open_return / volume_weighted_intraday_volatility

    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = np.exp(-np.log(2) / half_life)
    decayed_reversal_score = reversal_score.ewm(alpha=(1 - decay_factor)).mean()

    # Calculate and Interact Intraday Volatility with Volume
    close_to_open_ratio = df['close'] / df['open']
    high_low_diff = df['high'] - df['low']
    intraday_volatility_interacted = (close_to_open_ratio * high_low_diff) * df['volume']

    # Final Alpha Factor
    alpha_factor = adjusted_momentum_filtered * decayed_reversal_score * intraday_volatility_interacted

    return alpha_factor
```
