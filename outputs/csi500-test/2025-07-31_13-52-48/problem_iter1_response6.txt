```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Calculate Daily Volume Change
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))

    # Determine Threshold for High Volume Change
    volume_threshold = df['volume_change'].quantile(0.75)

    # Filter out days with high volume shocks
    df['filtered_price_momentum'] = df['price_momentum'].where(df['volume_change'] < volume_threshold, np.nan)

    return df['filtered_price_momentum'].dropna()
```
