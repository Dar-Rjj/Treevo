```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Volume-Adjusted Momentum
    df['previous_close'] = df['close'].shift(1)
    momentum = (df['close'] - df['previous_close']) * df['volume']
    
    # Calculate Weighted Intraday Volatility
    weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    vwma_intraday_volatility = weighted_intraday_volatility.rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    close_to_open_return = df['close'] / df['open'] - 1
    intraday_reversal_score = close_to_open_return / vwma_intraday_volatility
    
    # Apply Exponential Decay to Smooth the Reversal Score
    half_life = 5
    decay_factor = np.exp(-np.log(2) / half_life)
    smoothed_reversal_score = intraday_reversal_score.ewm(alpha=decay_factor, adjust=False).mean()
    
    # Combine Adjusted Momentum and Combined Ratio
    factor = combined_ratio + momentum + smoothed_reversal_score
    
    return factor
```
