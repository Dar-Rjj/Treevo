```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    intraday_momentum = df['high_low_diff'].rolling(window=5).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_volatility = df['volume_change'].rolling(window=10).std()
    intraday_momentum_vol_adj = intraday_momentum / volume_volatility

    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)

    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['volume'] * intraday_momentum_vol_adj

    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']

    # Calculate Intraday Price Range Ratio
    df['high_low_range'] = df['high'] - df['low']
    df['price_range_ratio'] = df['open_close_diff'] / df['high_low_range']

    # Adjust Intraday Momentum by Price Range Ratio
    df['intraday_momentum_adj'] = intraday_momentum * df['price_range_ratio']

    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['intraday_momentum_adj'] + df['volume_adjusted_momentum']

    # Compute Rolling Sum of Final Momentum
    df['rolling_sum_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_sum_final_momentum'] - df['rolling_sum_final_momentum'].rolling(window=60).mean()

    return df['dynamic_momentum_oscillator']
```
