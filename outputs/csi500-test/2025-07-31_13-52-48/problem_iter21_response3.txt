```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Volume-Weighted Momentum
    df['vol_weighted_momentum'] = df['price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_vol_weighted_momentum'] = df['vol_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=5).mean()
    
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['price_change'].rolling(window=20).std()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'].diff()
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['final_adjustment'] = df['avg_high_low_diff'] / df['volume_volatility']
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adjusted_momentum'] = df['vol_weighted_momentum'] * df['final_adjustment']
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_vol_weighted_momentum'] = df['rolling_vol_weighted_momentum'] / df['price_volatility']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'].diff()
    df['trend_adjusted_momentum'] = df['intraday_adjusted_momentum'] * df['price_trend']
    
    # Combine Intraday and Open-Close Price Movements
    df['high_low_percentage'] = (df['high'] - df['low']) / df['open']
    df['open_close_percentage'] = (df['close'] - df['open']) / df['open']
    df['avg_percentage_movement'] = (df['high_low_percentage'] + df['open_close_percentage']) / 2
    df['smoothed_avg_movement'] = df['avg_percentage_movement'].ewm(span=3).mean()
    
    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Range Ratio
    df['open_close_range'] = df['close'] - df['open']
    df['high_low_range'] = df['high'] - df['low']
    df['price_range_ratio'] = df['open_close_range'] / df['high_low_range']
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_high_low_diff'] = df['high_low_diff'] * df['price_range_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum'] = df['adjusted_high_low_diff'] + df['trend_adjusted_momentum']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_final_momentum'] = df['final_momentum'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['momentum_oscillator'] = df['rolling_final_momentum'] - df['rolling_final_momentum'].rolling(window=60).mean()
    
    return df['momentum_oscillator']
```
