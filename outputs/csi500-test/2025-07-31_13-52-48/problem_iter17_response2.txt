import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.rolling(window=20).mean()

    # Adjust by Volume Volatility
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_volatility = volume_change.rolling(window=20).std()
    adjusted_intraday_momentum = intraday_momentum / volume_volatility

    # Incorporate Price Trend
    price_change = df['close'] - df['close'].shift(1)
    adjusted_intraday_momentum *= price_change

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = adjusted_intraday_momentum - adjusted_intraday_momentum.rolling(window=60).mean()

    # Apply Volume-Adjusted Momentum Filter
    volume_adjusted_momentum = (df['volume'] - df['volume'].shift(1)) * (df['close'] - df['close'].shift(1))
    rolling_sum_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=20).sum()
    threshold = rolling_sum_volume_adjusted_momentum.abs().quantile(0.8)
    filtered_volume_adjusted_momentum = rolling_sum_volume_adjusted_momentum[rolling_sum_volume_adjusted_momentum.abs() < threshold]

    # Calculate Intraday Price Movement
    percent_price_movement_high_low = (df['high'] - df['low']) / df['open']
    percent_price_movement_close_open = (df['close'] - df['open']) / df['open']
    average_percent_movement = (percent_price_movement_high_low + percent_price_movement_close_open) / 2
    smoothed_average = average_percent_movement.ewm(span=5).mean()

    # Combine Ratios
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    combined_ratios = 0.6 * close_to_high_ratio + 0.4 * close_to_low_ratio

    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    combined_factor = smoothed_average + filtered_volume_adjusted_momentum + combined_ratios

    # Compute Dynamic Price Momentum Oscillator
    dynamic_price_momentum_oscillator = rolling_sum_volume_adjusted_momentum - rolling_sum_volume_adjusted_momentum.rolling(window=30).mean()

    # Apply Volume Shock Filter
    abs_volume_change = (df['volume'] - df['volume'].shift(1)).abs()
    volume_shock_threshold = abs_volume_change.quantile(0.8)
    filtered_dynamic_price_momentum_oscillator = dynamic_price_momentum_oscillator[abs_volume_change < volume_shock_threshold]

    # Confirm with Volume
    volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    weighted_final_alpha_factor = smoothed_average * volume_change_percentage

    return weighted_final_alpha_factor
