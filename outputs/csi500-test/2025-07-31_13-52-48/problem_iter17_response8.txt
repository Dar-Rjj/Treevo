def heuristics_v2(df):

    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    
    volume_threshold = df['volume_change'].quantile(0.8)
    amount_threshold = df['amount_change'].quantile(0.8)
    
    df = df[(df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)]

    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_sum_vol_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()

    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = df['daily_price_change'].rolling(window=20).std()

    # Adjust Volume-Weighted Momentum by Price Volatility
    df['adjusted_vol_weighted_momentum'] = df['rolling_sum_vol_weighted_momentum'] / df['daily_price_volatility']

    # Subtract 60-Day Moving Average of Adjusted Volume-Weighted Momentum from 20-Day Adjusted Rolling Sum
    df['ma_adjusted_vol_weighted_momentum_60'] = df['adjusted_vol_weighted_momentum'].rolling(window=60).mean()
    df['momentum_factor'] = df['adjusted_vol_weighted_momentum'] - df['ma_adjusted_vol_weighted_momentum_60']

    df['intraday_volatility'] = df['high'] - df['low']
