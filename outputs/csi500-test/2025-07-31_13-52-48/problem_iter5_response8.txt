```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Volume-Adjusted Momentum
    price_return = df['close'] - df['close'].shift(1)
    volume_adjusted_momentum = price_return * df['volume']
    
    # Calculate Price Volatility
    price_volatility = df[['open', 'high', 'low', 'close']].std(axis=1)
    
    # Apply Volume Shock Filter
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    abs_daily_volume_change = abs(daily_volume_change)
    volume_shock_threshold = abs_daily_volume_change.quantile(0.75)
    filtered_days = abs_daily_volume_change < volume_shock_threshold
    
    # Combine Adjusted Momentum, Combined Ratio, and Price Volatility
    alpha_factor = (combined_ratio
                    + volume_adjusted_momentum
                    + price_volatility) * filtered_days
    
    return alpha_factor
```
