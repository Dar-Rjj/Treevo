```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Intraday Volatility
    intraday_volatility = (intraday_price_range * close_to_open_ratio).rolling(window=10, min_periods=1).std()
    
    # Interact Intraday Volatility with Volume
    factor = intraday_volatility * df['volume']
    
    return factor
```
