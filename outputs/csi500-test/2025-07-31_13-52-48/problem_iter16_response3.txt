```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['volume_adj_momentum'] = (df['volume'] - df['volume'].shift(1)) * df['price_momentum']
    df['volume_adj_momentum_rolling_sum'] = df['volume_adj_momentum'].rolling(window=20).sum()
    volume_adj_momentum_75th = df['volume_adj_momentum_rolling_sum'].quantile(0.75)
    df['volume_adj_momentum_filtered'] = df['volume_adj_momentum_rolling_sum'].where(abs(df['volume_adj_momentum_rolling_sum']) < volume_adj_momentum_75th, 0)

    # Calculate Intraday High-Low Difference
    df['intraday_high_low_diff'] = df['high'] - df['low']
    df['avg_intraday_high_low_diff'] = df['intraday_high_low_diff'].rolling(window=5).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['intraday_high_low_adj'] = df['avg_intraday_high_low_diff'] / df['volume_volatility']

    # Combine Intraday and Open-Close Price Movements
    df['intraday_pct_move'] = (df['high'] - df['low']) / df['open']
    df['open_close_pct_move'] = (df['close'] - df['open']) / df['open']
    df['avg_pct_move'] = (df['intraday_pct_move'] + df['open_close_pct_move']) / 2
    df['smoothed_avg_pct_move'] = df['avg_pct_move'].ewm(span=3).mean()

    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']

    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = df['open_close_diff'] / (df['high'] - df['low'])

    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_intraday_high_low_diff'] = df['intraday_high_low_diff'] * df['intraday_price_range_ratio']

    # Calculate Final Momentum Factor
    df['final_momentum'] = df['adjusted_intraday_high_low_diff'] + df['volume_adj_momentum_filtered']

    # Compute Rolling Sum of Final Momentum
    df['final_momentum_rolling_sum'] = df['final_momentum'].rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['final_momentum_rolling_sum'] - df['final_momentum_rolling_sum'].rolling(window=60).mean()

    return df['dynamic_momentum_oscillator']
```
