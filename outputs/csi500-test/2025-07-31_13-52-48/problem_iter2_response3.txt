```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range * df['volume']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    vol_weighted_moving_avg = intraday_volatility.rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    intraday_reversal_score = close_to_open_return / vol_weighted_moving_avg
    
    # Apply Decay Function to Smooth the Reversal Score
    decay_factor = 0.5 ** (1/5)
    smoothed_reversal_score = intraday_reversal_score.ewm(halflife=5, adjust=False).mean()
    
    # Aggregate Daily Volume
    daily_volume = df['volume']
    
    # Interact Smoothed Reversal Score with Volume
    factor = smoothed_reversal_score * daily_volume
    
    return factor
```
