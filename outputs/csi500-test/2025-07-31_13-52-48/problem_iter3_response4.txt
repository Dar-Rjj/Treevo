```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Volume-Adjusted Momentum
    previous_close = df['close'].shift(1)
    momentum = df['close'] - previous_close
    volume_adjusted_momentum = momentum * df['volume']
    
    # Calculate Price Volatility
    price_volatility = df[['open', 'high', 'low', 'close']].std(axis=1)
    
    # Combine Adjusted Momentum, Combined Ratio, and Price Volatility
    factor = combined_ratio + volume_adjusted_momentum + price_volatility
    
    return factor
```
