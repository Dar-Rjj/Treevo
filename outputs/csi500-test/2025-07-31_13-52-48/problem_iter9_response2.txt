```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High-Low Spread
    high_low_spread = df['high'] - df['low']
    close_to_spread = df['close'] / high_low_spread
    
    # Calculate Volume-Adjusted Momentum
    prev_close = df['close'].shift(1)
    return_momentum = df['close'] - prev_close
    volume_adjusted_momentum = return_momentum * np.sqrt(df['volume'])
    
    # Calculate Price Range
    price_range = high_low_spread
    
    # Calculate Price Range Adjusted Volatility
    ohlc = df[['open', 'high', 'low', 'close']]
    volatility = ohlc.std(axis=1)
    adjusted_volatility = volatility / price_range
    
    # Calculate Trend Indicator
    positive_returns = (df['close'] > prev_close).astype(int)
    trend_indicator = positive_returns.rolling(window=5, min_periods=1).sum()
    
    # Combine Momentum, Spread, Adjusted Volatility, and Trend
    factor = (
        close_to_spread +
        volume_adjusted_momentum +
        adjusted_volatility +
        trend_indicator
    )
    
    return factor
```
