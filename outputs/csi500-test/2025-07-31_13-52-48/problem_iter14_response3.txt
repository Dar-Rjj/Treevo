```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['sum_log_returns_5d'] = df['log_return'].rolling(window=5).sum()
    df['volume_weighted_momentum'] = (df['close'] / df['close'].shift(20) - 1) * df['volume']
    df['combined_momentum'] = df['sum_log_returns_5d'] * df['volume_weighted_momentum'].rolling(window=20).sum()

    # Apply Combined Volume and Amount Shock Filter
    df['abs_volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['abs_amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_75th_percentile = df['abs_volume_change'].quantile(0.75)
    amount_75th_percentile = df['abs_amount_change'].quantile(0.75)
    df['shock_filter'] = (df['abs_volume_change'] < volume_75th_percentile) & (df['abs_amount_change'] < amount_75th_percentile)
    df['filtered_combined_momentum'] = df['combined_momentum'] * df['shock_filter']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['vol_weighted_avg_intraday_volatility'] = df['vol_weighted_intraday_volatility'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['vol_weighted_avg_intraday_volatility']
    df['exp_decay_intraday_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=5).mean()

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Calculate Daily Price Volatility
    df['daily_price_volatility'] = df['daily_price_change'].rolling(window=20).std()

    # Adjust Combined Momentum by Price Volatility
    df['adjusted_combined_momentum'] = df['filtered_combined_momentum'] / df['daily_price_volatility']

    # Subtract 60-Day Moving Average of Adjusted Combined Momentum from 20-Day Adjusted Combined Momentum
    df['final_factor'] = df['adjusted_combined_momentum'].rolling(window=20).mean() - df['adjusted_combined_momentum'].rolling(window=60).mean()

    return df['final_factor'].dropna()
```
