```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Average Over Period (5 days)
    lookback_period = 5
    df['intraday_momentum'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Volatility (10 days)
    volume_volatility_lookback = 10
    df['volume_volatility'] = df['volume_change'].rolling(window=volume_volatility_lookback).std()
    
    # Final Adjustment
    df['volume_adjusted_momentum'] = df['intraday_momentum'] / df['volume_volatility']
    
    # Calculate Close to High Ratio
    df['close_to_high_ratio'] = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Combine Ratios
    df['combined_ratio'] = 0.5 * df['close_to_high_ratio'] + 0.5 * df['close_to_low_ratio']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Adjusted Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['volume_adjusted_momentum']
    
    # Integrate Intraday Reversal Score
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['volume_weighted_momentum'].rolling(window=lookback_period).mean()
    df['smoothed_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=5, adjust=False).mean()
    
    # Calculate Price Momentum
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_returns'].rolling(window=5).sum()
    
    # Apply Volume Shock Filter
    df['abs_volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    volume_shock_threshold = df['abs_volume_change'].quantile(0.75)
    df = df[df['abs_volume_change'] < volume_shock_threshold]
    
    # Adjust Intraday Momentum by Price Range Ratio
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * df['combined_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_intraday_momentum'] + df['volume_weighted_momentum']
    
    # Compute Rolling Sum of Final Momentum (20 days)
    df['rolling_sum_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['momentum_oscillator'] = df['rolling_sum_final_momentum'] - df['rolling_sum_final_momentum'].rolling(window=60).mean()
    
    return df['momentum_oscillator'].dropna()
```
