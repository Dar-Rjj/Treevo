```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Apply Combined Volume and Amount Shock Filter
    df['volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_threshold = df['volume_change'].quantile(0.75)
    amount_threshold = df['amount_change'].quantile(0.75)
    df['shock_filter'] = (df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_intraday_volatility'] = (df['intraday_volatility'] * df['volume']).rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    df['reversal_score'] = df['close_to_open_return'] / df['vol_weighted_intraday_volatility']
    
    # Apply Exponential Decay to Reversal Score
    df['exp_decay_reversal_score'] = df['reversal_score'].ewm(halflife=5, min_periods=5).mean()
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['vol_weighted_intraday_range'] = (df['intraday_range'] * df['volume']).rolling(window=10).mean()
    
    # Combine Reversal Score and Intraday Range
    df['combined_score'] = df['exp_decay_reversal_score'] + df['vol_weighted_intraday_range']
    df['exp_decay_combined_score'] = df['combined_score'].ewm(halflife=5, min_periods=5).mean()
    
    # Calculate Intraday Price Movement
    df['high_low_movement'] = (df['high'] - df['low']) / df['open']
    df['close_open_movement'] = (df['close'] - df['open']) / df['open']
    df['avg_movement'] = (df['high_low_movement'] + df['close_open_movement']) / 2
    df['smoothed_avg_movement'] = df['avg_movement'].rolling(window=3).mean()
    
    # Confirm with Volume
    df['volume_change_percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Weighted Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_avg_movement'] * df['volume_change_percentage']
    
    return df['final_alpha_factor'].fillna(0)
```
