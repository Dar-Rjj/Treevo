```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.ewm(span=10, adjust=False).mean()
    
    # Adjust by Volume Volatility
    daily_volume_change = df['volume'].diff()
    volume_volatility = daily_volume_change.rolling(window=10).std()
    intraday_adjustment = intraday_momentum / volume_volatility
    
    # Calculate Volume-Weighted Momentum
    price_change = df['close'].pct_change()
    volume_weighted_momentum = price_change * np.log(df['volume'])
    
    # Adjust by Intraday High-Low Momentum
    volume_weighted_adjusted = volume_weighted_momentum * intraday_adjustment
    
    # Incorporate Price Trend
    price_trend_change = df['close'] - df['open'].shift(1)
    long_term_price_trend = df['close'].ewm(span=200, adjust=False).mean()
    trend_adjusted_momentum = volume_weighted_adjusted * price_trend_change * (df['close'] - long_term_price_trend)
    
    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = trend_adjusted_momentum - trend_adjusted_momentum.ewm(span=50, adjust=False).mean()
    
    return dynamic_momentum_oscillator
```
