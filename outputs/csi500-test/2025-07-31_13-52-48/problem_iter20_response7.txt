```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    money_flow_10d_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = money_flow_10d_ma / volume_10d_ma
    
    # Calculate the 5-day moving average of the close price
    close_5d_ma = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price
    close_20d_ma = df['close'].rolling(window=20).mean()
    
    # Calculate the trend strength
    trend_strength = (close_5d_ma - close_20d_ma) / close_20d_ma
    
    # Combine the money flow ratio and trend strength to form the final factor
    factor = money_flow_ratio * trend_strength
    
    return factor
```
