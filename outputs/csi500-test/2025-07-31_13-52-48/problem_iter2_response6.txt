```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Volume-Adjusted Momentum
    close_shifted = df['close'].shift(1)
    momentum = df['close'] - close_shifted
    volume_adjusted_momentum = momentum * df['volume']
    
    # Combine Adjusted Momentum and Combined Ratio
    combined_factor = combined_ratio + volume_adjusted_momentum
    
    # Calculate Price Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_momentum = log_returns.rolling(window=5).sum()
    
    # Apply Volume Shock Filter
    volume_change = abs(df['volume'] - df['volume'].shift(1))
    volume_change_threshold = volume_change.quantile(0.75)
    filter_condition = (volume_change < volume_change_threshold)
    
    # Apply the filter condition to the combined factor
    combined_factor_filtered = combined_factor[filter_condition]
    
    return combined_factor_filtered
```
