```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * df['volume']
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['20d_rolling_sum_vwm'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Compute 60-Day Moving Average of Volume-Weighted Momentum
    df['60d_moving_avg_vwm'] = df['volume_weighted_momentum'].rolling(window=60).mean()
    
    # Subtract 60-Day Moving Average from 20-Day Rolling Sum
    df['factor'] = df['20d_rolling_sum_vwm'] - df['60d_moving_avg_vwm']
    
    return df['factor']
```
