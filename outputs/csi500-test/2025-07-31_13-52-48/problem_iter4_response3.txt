import pandas as pd
import numpy as np

def heuristics_v2(df, span=10, n_days=20):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    smoothing_factor = 2 / (span + 1)
    df['ema_intraday_return'] = df['intraday_return'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Standard Deviation of Intraday Range over the last N days
    df['std_intraday_range'] = df['intraday_range'].rolling(window=n_days).std()
    
    # Multiply Intraday Momentum by Standard Deviation Inverse
    df['alpha_factor'] = df['ema_intraday_return'] / df['std_intraday_range']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
