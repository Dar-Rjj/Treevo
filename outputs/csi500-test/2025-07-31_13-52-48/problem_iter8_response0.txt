```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = df['close'].pct_change()
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume Shock Filter
    df['volume_change'] = df['volume'].diff().abs()
    volume_threshold = df['volume_change'].quantile(0.75)
    df_filtered = df[df['volume_change'] < volume_threshold]

    # Calculate Intraday Price Movement
    df_filtered['intraday_high_low'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['open']
    df_filtered['intraday_close_open'] = (df_filtered['close'] - df_filtered['open']) / df_filtered['open']

    # Combine Intraday Price Movements
    df_filtered['avg_intraday_movement'] = (df_filtered['intraday_high_low'] + df_filtered['intraday_close_open']) / 2
    df_filtered['smoothed_avg_intraday_movement'] = df_filtered['avg_intraday_movement'].rolling(window=3).mean()

    # Confirm with Volume
    df_filtered['volume_change_percentage'] = (df_filtered['volume'] - df_filtered['volume'].shift(1)) / df_filtered['volume'].shift(1)

    # Calculate Close to High and Low Ratios
    df_filtered['close_to_high_ratio'] = df_filtered['close'] / df_filtered['high']
    df_filtered['close_to_low_ratio'] = df_filtered['close'] / df_filtered['low']
    df_filtered['combined_ratios'] = 0.5 * df_filtered['close_to_high_ratio'] + 0.5 * df_filtered['close_to_low_ratio']

    # Calculate Volume-Adjusted Momentum
    df_filtered['volume_adjusted_momentum'] = (df_filtered['close'] - df_filtered['close'].shift(1)) * df_filtered['volume']

    # Calculate Price Volatility
    df_filtered['price_volatility'] = df_filtered[['open', 'high', 'low', 'close']].std(axis=1)

    # Weighted Final Alpha Factor
    df_filtered['alpha_factor'] = (
        df_filtered['smoothed_avg_intraday_movement'] * df_filtered['volume_change_percentage'] +
        df_filtered['combined_ratios'] +
        df_filtered['volume_adjusted_momentum'] +
        df_filtered['price_volatility']
    )

    return df_filtered['alpha_factor']
```
