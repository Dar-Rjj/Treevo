```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum'] = df['high_low_diff'].ewm(span=10, adjust=False).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'].diff()
    df['volume_volatility'] = df['volume_change'].abs().rolling(window=20).mean()
    df['final_adjustment'] = df['intraday_momentum'] / df['volume_volatility']

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * (df['volume'] ** (1/3))

    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()

    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_momentum'] = df['rolling_volume_weighted_momentum'] * df['final_adjustment']

    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['open'].shift(1)
    df['long_term_price_trend'] = df['close'].ewm(span=200, adjust=False).mean()
    df['trend_adjusted_momentum'] = df['intraday_adjusted_momentum'] * df['price_change'] * df['long_term_price_trend']

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_momentum'] - df['trend_adjusted_momentum'].rolling(window=50).mean()

    return df['dynamic_momentum_oscillator']
```
