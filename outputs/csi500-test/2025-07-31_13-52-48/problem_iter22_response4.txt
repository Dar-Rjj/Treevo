```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['ema_intraday_return'] = df['intraday_return'].ewm(alpha=0.1, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    df['std_intraday_return'] = df['intraday_return'].rolling(window=20).std()
    df['volatility_adjusted_momentum'] = df['ema_intraday_return'] * df['std_intraday_return']
    
    # Introduce Enhanced Volume Impact
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * df['volume']
    df['adjusted_ema_intraday_return'] = df['volume_weighted_intraday_return'].ewm(alpha=0.05, adjust=False).mean()
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=20)
    df['enhanced_volume_factor'] = df['adjusted_ema_intraday_return'] + df['volume_rate_of_change']
    
    # Calculate Daily Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['daily_price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Apply Volume-Adjusted Momentum Filter
    df['volume_adjusted_momentum'] = df['volume'] * df['daily_price_momentum']
    df['rolling_vol_adj_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    threshold = df['rolling_vol_adj_momentum'].abs().quantile(0.75)
    df['filtered_vol_adj_momentum'] = df['rolling_vol_adj_momentum'].where(df['rolling_vol_adj_momentum'].abs() < threshold, 0)
    
    # Calculate Intraday Price Movement
    df['price_movement_high_low'] = (df['high'] - df['low']) / df['open']
    df['price_movement_close_open'] = (df['close'] - df['open']) / df['open']
    df['avg_price_movement'] = (df['price_movement_high_low'] + df['price_movement_close_open']) / 2
    df['smoothed_avg_price_movement'] = df['avg_price_movement'].ewm(span=3, adjust=False).mean()
    
    # Calculate Price Breakout Indicator
    df['ma_high_20'] = df['high'].rolling(window=20).mean()
    df['ma_low_20'] = df['low'].rolling(window=20).mean()
    df['price_breakout_indicator'] = (df['ma_high_20'] - df['ma_low_20']) / df['ma_high_20']
    
    # Combine Ratios
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    df['combined_ratios'] = df['close_to_high_ratio'] + df['close_to_low_ratio']
    
    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    df['dynamic_price_momentum_oscillator'] = df['rolling_vol_adj_momentum'] - df['rolling_vol_adj_momentum'].rolling(window=60).mean()
    df['final_alpha_factor'] = (
        df['filtered_vol_adj_momentum'] +
        df['combined_ratios'] +
        df['price_breakout_indicator'] +
        df['dynamic_price_momentum_oscillator']
    )
    
    return df['final_alpha_factor']
```
