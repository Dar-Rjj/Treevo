import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['intraday_momentum'] = df['log_return'].rolling(window=5).sum() / 5
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['adjusted_momentum'] = df['intraday_momentum'] / df['volume_volatility']
    
    # Apply Combined Volume and Amount Shock Filter
    volume_threshold = df['volume_change'].abs().quantile(0.75)
    amount_threshold = (df['amount'] - df['amount'].shift(1)).abs().quantile(0.75)
    df['shock_filter'] = (df['volume_change'].abs() < volume_threshold) & ((df['amount'] - df['amount'].shift(1)).abs() < amount_threshold)
    df['filtered_momentum'] = df['adjusted_momentum'] * df['shock_filter']
    
    # Calculate Intraday Reversal Score
    df['close_to_open'] = df['close'] - df['open']
    df['intraday_volatility'] = (df['high'] - df['low']) * df['volume']
    df['volatility_ma'] = df['intraday_volatility'].rolling(window=10).mean()
    df['reversal_score'] = df['close_to_open'] / df['volatility_ma']
    
    # Apply Exponential Decay to Reversal Score
    half_life = 5
    df['exponential_decay'] = df['reversal_score'].ewm(halflife=half_life, adjust=False).mean()
    
    # Interact Intraday Volatility with Volume
    df['intraday_volatility_interacted'] = df['intraday_volatility'] * df['volume']
    
    # Final Factor
    df['alpha_factor'] = df['filtered_momentum'] * df['exponential_decay'] * df['intraday_volatility_interacted']
    
    return df['alpha_factor'].dropna()
