```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Intraday Volatility (weighted average of Close-to-Open Ratio and Intraday Price Range)
    # We can use a simple equal weight for both components
    intraday_volatility = (close_to_open_ratio + intraday_price_range) / 2
    
    # Interact Intraday Volatility with Volume and Amount
    factor = intraday_volatility * df['volume'] * df['amount']
    
    return factor
```
