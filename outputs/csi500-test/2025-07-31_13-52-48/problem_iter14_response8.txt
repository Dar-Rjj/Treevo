```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_ema'] = df['intraday_return'].ewm(alpha=0.1, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_range'] = df['high'] - df['low']
    df['volatility'] = df['intraday_range'].rolling(window=20).std()
    df['intraday_momentum_vol'] = df['intraday_ema'] * df['volatility']
    
    # Introduce Enhanced Volume Impact
    df['volume_weighted_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['volume_weighted_ema'] = df['volume_weighted_return'].ewm(alpha=0.15, adjust=False).mean()
    df['volume_roc'] = df['volume'] / df['volume'].shift(20) - 1
    df['final_alpha_factor'] = df['volume_weighted_ema'] + df['volume_roc']
    
    # Calculate Daily Price Momentum
    df['daily_momentum'] = np.log(df['close']).diff().rolling(window=5).sum()
    
    # Apply Volume-Adjusted Momentum Filter
    df['volume_adjusted_momentum'] = (df['volume'] - df['volume'].shift(1)) * df['daily_momentum']
    df['rolling_sum_volume_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    threshold = df['volume_adjusted_momentum'].abs().quantile(0.75)
    df = df[df['volume_adjusted_momentum'].abs() < threshold]
    
    # Calculate Intraday Price Movement
    df['price_movement1'] = (df['high'] - df['low']) / df['open']
    df['price_movement2'] = (df['close'] - df['open']) / df['open']
    df['average_price_movement'] = (df['price_movement1'] + df['price_movement2']) / 2
    df['smoothed_price_movement'] = df['average_price_movement'].ewm(span=3, adjust=False).mean()
    
    # Combine Ratios
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    df['combined_ratios'] = 0.5 * df['close_to_high_ratio'] + 0.5 * df['close_to_low_ratio']
    
    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    df['final_alpha'] = df['smoothed_price_movement'] + df['rolling_sum_volume_momentum'] + df['combined_ratios']
    
    # Compute Dynamic Price Momentum Oscillator
    df['dynamic_oscillator'] = df['rolling_sum_volume_momentum'] - df['rolling_sum_volume_momentum'].rolling(window=60).mean()
    
    return df['final_alpha']
```
