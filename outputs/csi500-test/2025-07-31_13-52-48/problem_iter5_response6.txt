```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Volume-Adjusted Momentum
    prev_close = df['close'].shift(1)
    return_adj_by_volume = (df['close'] - prev_close) * df['volume']
    
    # Calculate Weighted Intraday Volatility
    weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    vol_weighted_moving_avg = weighted_intraday_volatility.rolling(window=10).mean()
    
    # Integrate Intraday Reversal Score
    close_to_open_return = (df['close'] - df['open']) / (df['open'])
    intraday_reversal_score = close_to_open_return / vol_weighted_moving_avg
    smoothed_reversal_score = intraday_reversal_score.ewm(halflife=5).mean()
    
    # Calculate Price Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_momentum = log_returns.rolling(window=5).sum().shift(1)
    
    # Apply Volume Shock Filter
    daily_volume_change = np.abs(df['volume'] - df['volume'].shift(1))
    volume_shock_threshold = daily_volume_change.quantile(0.75)
    filtered_days = (daily_volume_change < volume_shock_threshold)
    
    # Final Alpha Factor
    alpha_factor = (
        combined_ratio +
        return_adj_by_volume +
        smoothed_reversal_score +
        price_momentum
    ) * filtered_days
    
    return alpha_factor
```
