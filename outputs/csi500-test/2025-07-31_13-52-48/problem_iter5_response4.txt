```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    df['CloseHighRatio'] = df['close'] / df['high']
    df['CloseLowRatio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['CombinedRatio'] = 0.5 * df['CloseHighRatio'] + 0.5 * df['CloseLowRatio']
    
    # Calculate Intraday Price Movements
    df['HighLowDiff'] = df['high'] - df['low']
    df['HighLowPct'] = (df['high'] - df['low']) / df['open']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['CloseOpenPct'] = (df['close'] - df['open']) / df['open']
    
    # Average Intraday Price Movements
    df['AvgPctMove'] = (df['HighLowPct'] + df['CloseOpenPct']) / 2
    df['SmoothedPctMove'] = df['AvgPctMove'].ewm(span=3, min_periods=3).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    df['VolumeAdjMomentum'] = df['PriceChange'] * df['volume']
    
    # Confirm with Volume
    df['VolChange'] = df['volume'] - df['volume'].shift(1)
    df['VolChangePct'] = df['VolChange'] / df['volume'].shift(1)
    df['WeightedFinalFactor'] = df['SmoothedPctMove'] * df['VolChangePct']
    
    # Calculate Intraday Volatility
    df['IntradayVolatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseOpenReturn'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['VolWtdIntradayVolatility'] = df['IntradayVolatility'] * df['volume']
    df['VolWtdIntradayVolatilityMA'] = df['VolWtdIntradayVolatility'].rolling(window=10, min_periods=10).mean()
    
    # Calculate Intraday Reversal Score
    df['ReversalScore'] = df['CloseOpenReturn'] / df['VolWtdIntradayVolatilityMA']
    
    # Apply Exponential Decay to Reversal Score
    df['DecayedReversalScore'] = df['ReversalScore'].ewm(halflife=5, min_periods=5).mean()
    
    # Combine All Factors
    df['AlphaFactor'] = (
        df['CombinedRatio'] +
        df['VolumeAdjMomentum'] +
        df['WeightedFinalFactor'] +
        df['DecayedReversalScore']
    )
    
    return df['AlphaFactor']
```
