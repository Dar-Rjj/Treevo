```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['daily_price_change'] * np.log(df['volume'])
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['rolling_vol_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=20).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['ema_high_low_diff'] = df['high_low_diff'].ewm(span=5, adjust=False).mean()
    
    # Calculate Daily Price Volatility
    df['price_volatility'] = df['daily_price_change'].rolling(window=20).std()
    
    # Adjust by Volume Volatility
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['daily_volume_change'].rolling(window=10).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    df['intraday_adjustment'] = df['ema_high_low_diff'] / df['volume_volatility']
    
    # Calculate Intraday-Adjusted Volume-Weighted Momentum
    df['intraday_adj_vol_weighted_momentum'] = df['volume_weighted_momentum'] * df['intraday_adjustment']
    
    # Adjust Volume-Weighted Momentum by Price Volatility
    df['vol_weighted_momentum_adj_price_vol'] = df['rolling_vol_weighted_momentum'] / df['price_volatility']
    
    # Incorporate Price Trend
    df['price_change_trend'] = df['close'] - df['open'].shift(1)
    df['momentum_adj_price_trend'] = df['intraday_adj_vol_weighted_momentum'] * df['price_change_trend']
    
    # Combine Intraday and Open-Close Price Movements
    df['percent_intraday_move'] = (df['high'] - df['low']) / df['open']
    df['percent_open_close_move'] = (df['close'] - df['open']) / df['open']
    df['avg_percent_move'] = (df['percent_intraday_move'] + df['percent_open_close_move']) / 2
    df['smoothed_avg_percent_move'] = df['avg_percent_move'].ewm(span=3, adjust=False).mean()
    
    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Calculate Intraday Price Range Ratio
    df['open_close_ratio'] = df['open_close_diff'] / (df['high'] - df['low'])
    
    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adj_intraday_high_low_diff'] = df['high_low_diff'] * df['open_close_ratio']
    
    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adj_intraday_high_low_diff'] + df['momentum_adj_price_trend']
    
    # Compute Rolling Sum of Final Momentum
    df['rolling_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_final_momentum'] - df['rolling_final_momentum'].rolling(window=60).mean()
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['close_to_open_ratio'] * df['intraday_price_range']
    
    # Interact Intraday Volatility with Volume
    df['intraday_volatility_volume'] = df['intraday_volatility'] * df['volume']
    
    # Return the final factor
    return df['intraday_volatility_volume']
```
