```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Return'].rolling(window=5).sum()

    # Calculate Daily Volume-Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = (df['volume'] - df['volume'].shift(1)) * df['Price_Momentum']

    # Compute Rolling Sum of Volume-Adjusted Momentum over 20 Days
    df['Rolling_Sum_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=20).sum()

    # Determine Threshold for High Volume-Adjusted Momentum
    threshold = df['Rolling_Sum_Volume_Adjusted_Momentum'].abs().quantile(0.75)

    # Filter High Volume-Adjusted Momentum
    df['Filtered_Volume_Adjusted_Momentum'] = df['Rolling_Sum_Volume_Adjusted_Momentum'].where(
        df['Rolling_Sum_Volume_Adjusted_Momentum'].abs() < threshold, np.nan
    )

    # Compute Moving Average over 60 Days
    df['MA_60'] = df['Filtered_Volume_Adjusted_Momentum'].rolling(window=60).mean()

    # Compute Dynamic Price Momentum Oscillator
    df['Dynamic_Price_Momentum_Oscillator'] = df['Filtered_Volume_Adjusted_Momentum'] - df['MA_60']

    return df['Dynamic_Price_Momentum_Oscillator']
```
