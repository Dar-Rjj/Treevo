```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Apply Combined Volume and Amount Shock Filter
    df['abs_volume_change'] = np.abs(df['volume'] - df['volume'].shift(1))
    df['abs_amount_change'] = np.abs(df['amount'] - df['amount'].shift(1))
    
    volume_75th = df['abs_volume_change'].quantile(0.75)
    amount_75th = df['abs_amount_change'].quantile(0.75)
    
    df['filtered'] = (df['abs_volume_change'] < volume_75th) & (df['abs_amount_change'] < amount_75th)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Intraday Reversal Score
    df['reversal_score'] = df['close_to_open_return'] / df['intraday_volatility']
    
    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = np.log(2) / half_life
    df['reversal_score_decay'] = df['reversal_score'] * np.exp(-decay_factor * (df.index - df.index.min()).days)
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open']
    
    # Apply Exponential Decay to Intraday Momentum
    df['intraday_momentum_decay'] = df['intraday_momentum'] * np.exp(-decay_factor * (df.index - df.index.min()).days)
    
    # Combine Reversal Score, Intraday Range, and Intraday Momentum
    df['combined_factor'] = df['reversal_score_decay'] + df['intraday_range'] + df['intraday_momentum_decay']
    
    # Apply Exponential Decay to the Combined Factor
    df['combined_factor_decay'] = df['combined_factor'] * np.exp(-decay_factor * (df.index - df.index.min()).days)
    
    # Apply the filter
    df['alpha_factor'] = df['combined_factor_decay'] * df['filtered']
    
    return df['alpha_factor']
```
