```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Intraday Volatility
    intraday_volatility = 0.5 * (intraday_price_range + close_to_open_ratio)
    
    # Interact Intraday Volatility with Volume
    intraday_volatility_with_volume = intraday_volatility * df['volume']
    
    # Calculate Simple Moving Averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    sma_diff_5_20 = sma_5 - sma_20
    
    # Examine Price Changes
    price_change_pct = df['close'].pct_change()
    price_change_sum_5d = price_change_pct.rolling(window=5).sum()
    
    # Incorporate Volume
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20d
    
    # Integrate Amount
    avg_amount_20d = df['amount'].rolling(window=20).mean()
    amount_ratio = df['amount'] / avg_amount_20d
    
    # Investigate Volatility
    daily_returns = df['close'].pct_change()
    std_dev_20d = daily_returns.rolling(window=20).std()
    atr_14d = df[['high', 'low']].rolling(window=14).apply(lambda x: max(x) - min(x), raw=True).mean(axis=1)
    
    # Analyze High and Low Prices
    avg_high_20d = df['high'].rolling(window=20).mean()
    avg_low_20d = df['low'].rolling(window=20).mean()
    high_low_diff_20d = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_volatility_with_volume +
        sma_diff_5_20 +
        price_change_sum_5d +
        volume_ratio +
        amount_ratio +
        std_dev_20d +
        atr_14d +
        high_low_diff_20d
    )
    
    return alpha_factor
```
