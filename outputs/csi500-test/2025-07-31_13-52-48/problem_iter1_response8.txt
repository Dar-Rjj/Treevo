```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a high-low momentum indicator as an alpha factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as dates.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Apply Moving Average to High-Low Difference
    window_period = 10
    ma_high_low_diff = high_low_diff.rolling(window=window_period).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change(periods=window_period)
    
    # Combine High-Low Difference and Price Momentum
    combined_signal = ma_high_low_diff + price_momentum
    
    return combined_signal
```
