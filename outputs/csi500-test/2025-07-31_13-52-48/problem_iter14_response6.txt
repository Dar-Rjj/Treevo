import pandas as pd

def heuristics_v2(df):
    # Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Interact Intraday Price Range with Volume
    intraday_price_volume = intraday_price_range * df['volume']
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate 20-Day Rolling Sum of Daily Price Changes
    rolling_sum_20 = daily_price_change.rolling(window=20).sum()
    
    # Calculate 60-Day Moving Average of Daily Price Changes
    moving_avg_60 = daily_price_change.rolling(window=60).mean()
    
    # Compute Momentum Difference
    momentum_diff = rolling_sum_20 - moving_avg_60
    
    # Apply Volume Shock Filter
    daily_volume_change = df['volume'].diff().abs()
    volume_shock_threshold = daily_volume_change.quantile(0.75)
    volume_shock_filter = daily_volume_change < volume_shock_threshold
    filtered_momentum_diff = momentum_diff.where(volume_shock_filter, other=0)
    
    # Calculate Intraday Price Movement
    intraday_high_low_movement = (df['high'] - df['low']) / df['open']
    intraday_close_open_movement = (df['close'] - df['open']) / df['open']
    intraday_price_movement = (intraday_high_low_movement + intraday_close_open_movement) / 2
    smoothed_intraday_price_movement = intraday_price_movement.rolling(window=3).mean()
    
    # Confirm with Volume
    volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Close to High Ratio
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    close_to_low_ratio = df['close'] / df['low']
    
    # Combine Ratios
    combined_ratio = 0.5 * close_to_high_ratio + 0.5 * close_to_low_ratio
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Price Volatility
    price_volatility = df[['open', 'high', 'low', 'close']].std(axis=1)
    
    # Weighted Final Alpha Factor
    final_alpha_factor = (
        smoothed_intraday_price_movement * volume_change_percentage +
        combined_ratio +
        volume_adjusted_momentum +
        price_volatility
    )
    
    return final_alpha_factor
