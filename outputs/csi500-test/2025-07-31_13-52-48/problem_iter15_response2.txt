```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['price_momentum'] * df['volume_change']
    df['rolling_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    threshold = df['rolling_volume_adjusted_momentum'].abs().quantile(0.8)
    df = df[df['rolling_volume_adjusted_momentum'].abs() < threshold]

    # Calculate Intraday Price Movement
    df['intraday_high_low'] = (df['high'] - df['low']) / df['open']
    df['intraday_close_open'] = (df['close'] - df['open']) / df['open']
    df['avg_intraday_movement'] = (df['intraday_high_low'] + df['intraday_close_open']) / 2
    df['smoothed_intraday_movement'] = df['avg_intraday_movement'].ewm(span=5, adjust=False).mean()

    # Combine Ratios
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    df['combined_ratios'] = 0.6 * df['close_to_high_ratio'] + 0.4 * df['close_to_low_ratio']

    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    df['combined_factor'] = df['smoothed_intraday_movement'] + df['rolling_volume_adjusted_momentum'] + df['combined_ratios']

    # Compute Dynamic Price Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_volume_adjusted_momentum'] - df['rolling_volume_adjusted_momentum'].rolling(window=30).mean()

    # Apply Volume Shock Filter
    df['volume_shock'] = (df['volume'] - df['volume'].shift(1)).abs()
    volume_shock_threshold = df['volume_shock'].quantile(0.8)
    df = df[df['volume_shock'] < volume_shock_threshold]

    # Confirm with Volume
    df['volume_change_percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['final_alpha_factor'] = df['smoothed_intraday_movement'] * df['volume_change_percentage']

    return df['final_alpha_factor']
```
