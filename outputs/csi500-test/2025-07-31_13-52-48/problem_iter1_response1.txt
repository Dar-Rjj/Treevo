```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum (Average over 5 days)
    lookback_period_momentum = 5
    df['intraday_momentum'] = df['high_low_diff'].rolling(window=lookback_period_momentum).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Volatility (Standard Deviation of Volume Changes over 10 days)
    lookback_period_volatility = 10
    df['volume_volatility'] = df['volume_change'].rolling(window=lookback_period_volatility).std()
    
    # Adjust Intraday Momentum by Volume Volatility
    df['adjusted_momentum'] = df['intraday_momentum'] / (df['volume_volatility'] + 1e-6)  # Add small constant to avoid division by zero
    
    # Return the final factor as a pandas Series
    return df['adjusted_momentum']
```
