```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(1)
    
    # Adjust by Volume
    adjusted_momentum = daily_momentum * df['volume']
    
    # Compute Dynamic Indicator
    sum_20_days = adjusted_momentum.rolling(window=20).sum()
    moving_avg_60_days = sum_20_days.rolling(window=60).mean()
    
    dynamic_indicator = sum_20_days - moving_avg_60_days
    
    return dynamic_indicator
```
