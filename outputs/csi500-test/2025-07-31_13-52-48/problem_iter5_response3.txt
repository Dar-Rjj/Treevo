```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Difference
    df['intraday_high_low_diff'] = df['high'] - df['low']
    
    # Average Intraday High-Low Difference over 5 days
    df['avg_intraday_high_low_diff'] = df['intraday_high_low_diff'].rolling(window=5).mean()
    
    # Calculate Weighted Intraday Volatility
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vol_weighted_moving_avg_intraday_volatility'] = df['weighted_intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] / df['open']) - 1
    
    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['vol_weighted_moving_avg_intraday_volatility']
    
    # Apply Exponential Decay to Smooth the Reversal Score
    df['smoothed_reversal_score'] = df['intraday_reversal_score'].ewm(halflife=5).mean()
    
    # Calculate Price Momentum
    df['log_returns'] = np.log(df['close']).diff()
    df['price_momentum'] = df['log_returns'].rolling(window=5).sum().shift(1)
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    
    # Compute Volume Volatility over 10 days
    df['volume_volatility'] = df['daily_volume_change'].rolling(window=10).std()
    
    # Adjust Intraday Momentum by Volume Volatility
    df['adjusted_price_momentum'] = df['price_momentum'] / df['volume_volatility']
    
    # Calculate Close to High Ratio
    df['close_to_high_ratio'] = df['close'] / df['high']
    
    # Calculate Close to Low Ratio
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['combined_close_ratios'] = 0.5 * df['close_to_high_ratio'] + 0.5 * df['close_to_low_ratio']
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Apply Volume Shock Filter
    df['abs_volume_change'] = df['volume'].diff().abs()
    volume_shock_threshold = df['abs_volume_change'].quantile(0.75)
    df = df[df['abs_volume_change'] < volume_shock_threshold]
    
    # Synthesize Combined Factors
    df['synthesized_factor'] = (
        df['smoothed_reversal_score'] +
        df['adjusted_price_momentum'] +
        df['volume_adjusted_momentum'] +
        df['combined_close_ratios']
    )
    
    return df['synthesized_factor']
```
