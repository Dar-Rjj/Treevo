```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['price_momentum'] * df['volume_change']
    df['rolling_vol_adj_mom'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    threshold = df['rolling_vol_adj_mom'].abs().quantile(0.75)
    df = df[df['rolling_vol_adj_mom'].abs() < threshold]

    # Calculate Intraday Price Movement
    df['high_low_movement'] = (df['high'] - df['low']) / df['open']
    df['open_close_movement'] = (df['close'] - df['open']) / df['open']

    # Combine Intraday Price Movements
    df['avg_movement'] = (df['high_low_movement'] + df['open_close_movement']) / 2
    df['smoothed_avg_movement'] = df['avg_movement'].ewm(span=3).mean()

    # Confirm with Volume
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Weighted Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_avg_movement'] * df['volume_change_pct']

    return df['final_alpha_factor']
```
