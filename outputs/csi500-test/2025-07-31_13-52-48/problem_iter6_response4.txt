```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff.rolling(window=5).mean()

    # Adjust by Volume Volatility
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_volatility = daily_volume_change.rolling(window=10).std()
    final_adjustment = intraday_momentum / volume_volatility

    # Calculate Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(1)

    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = daily_momentum * final_adjustment

    # Rolling Sum of Volume-Adjusted Momentum
    rolling_sum_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    dynamic_momentum_oscillator = rolling_sum_volume_adjusted_momentum - rolling_sum_volume_adjusted_momentum.rolling(window=60).mean()

    # Introduce Open-Close Price Difference
    open_close_diff = df['close'] - df['open']

    # Calculate Intraday Price Range Ratio
    intraday_price_range_ratio = open_close_diff / high_low_diff

    # Adjust Intraday Momentum by Price Range Ratio
    adjusted_intraday_momentum = intraday_momentum * intraday_price_range_ratio

    # Calculate Final Momentum Factor
    final_momentum_factor = adjusted_intraday_momentum + volume_adjusted_momentum

    # Compute Rolling Sum of Final Momentum
    rolling_sum_final_momentum = final_momentum_factor.rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    final_dynamic_momentum_oscillator = rolling_sum_final_momentum - rolling_sum_final_momentum.rolling(window=60).mean()

    return final_dynamic_momentum_oscillator
```
