```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum'] = df['high_low_diff'].ewm(span=10, min_periods=5).mean()
    
    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=20, min_periods=10).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    df['intraday_adjusted_momentum'] = df['intraday_momentum'] / (df['volume_volatility'] + 1e-8)
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['price_change'] * df['volume'] ** (1/3)
    
    # Compute 20-Day Rolling Sum of Volume-Weighted Momentum
    df['volume_weighted_momentum_20d'] = df['volume_weighted_momentum'].rolling(window=20, min_periods=10).sum()
    
    # Adjust by Intraday High-Low Momentum
    df['intraday_adjusted_volume_momentum'] = df['volume_weighted_momentum_20d'] * df['intraday_adjusted_momentum']
    
    # Incorporate Price Trend
    df['price_trend'] = df['close'] - df['open'].shift(1)
    df['long_term_price_trend'] = df['close'].ewm(span=200, min_periods=100).mean()
    df['trend_adjusted_momentum'] = df['intraday_adjusted_volume_momentum'] * df['price_trend'] * df['long_term_price_trend']
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['trend_adjusted_momentum'] - df['trend_adjusted_momentum'].rolling(window=50, min_periods=25).mean()
    
    return df['dynamic_momentum_oscillator'].dropna()
```
