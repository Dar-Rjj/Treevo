```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Change']
    df['Rolling_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=20).sum()
    
    volume_adj_momentum_threshold = df['Volume_Adjusted_Momentum'].abs().quantile(0.75)
    df['High_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].abs() < volume_adj_momentum_threshold

    # Calculate Intraday Price Movement
    df['Intraday_High_Low'] = (df['high'] - df['low']) / df['open']
    df['Intraday_Close_Open'] = (df['close'] - df['open']) / df['open']
    df['Average_Intraday_Movement'] = (df['Intraday_High_Low'] + df['Intraday_Close_Open']) / 2
    df['Smoothed_Average_Intraday_Movement'] = df['Average_Intraday_Movement'].ewm(span=3).mean()

    # Combine Ratios
    df['Close_to_High_Ratio'] = df['close'] / df['high']
    df['Close_to_Low_Ratio'] = df['close'] / df['low']
    df['Combined_Ratios'] = 0.5 * df['Close_to_High_Ratio'] + 0.5 * df['Close_to_Low_Ratio']

    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    df['Alpha_Factor'] = (df['Smoothed_Average_Intraday_Movement'] 
                          + df['Rolling_Volume_Adjusted_Momentum']
                          + df['Combined_Ratios'])

    # Compute Dynamic Price Momentum Oscillator
    df['Dynamic_Price_Momentum_Oscillator'] = (df['Rolling_Volume_Adjusted_Momentum'] 
                                              - df['Rolling_Volume_Adjusted_Momentum'].rolling(window=60).mean())

    # Apply Volume Shock Filter
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)).abs()
    volume_shock_threshold = df['Daily_Volume_Change'].quantile(0.75)
    df['High_Volume_Shock'] = df['Daily_Volume_Change'] < volume_shock_threshold

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] * df['High_Volume_Adjusted_Momentum'] * df['High_Volume_Shock']

    return df['Final_Alpha_Factor'].dropna()
```
