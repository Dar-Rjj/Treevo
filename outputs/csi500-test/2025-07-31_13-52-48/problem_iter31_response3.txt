```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['High_Low_Momentum'] = df['High_Low_Diff'].ewm(span=10, adjust=False).mean()

    # Adjust by Volume Volatility
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Volatility'] = df['Volume_Change'].rolling(window=20).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    df['Intraday_Adjusted_Momentum'] = df['High_Low_Momentum'] / df['Volume_Volatility']

    # Calculate Volume-Weighted Momentum
    df['Price_Change'] = df['close'] - df['open'].shift(1)
    df['Volume_Weighted_Momentum'] = df['Price_Change'] * np.log(df['volume'])

    # Incorporate Price Trend
    df['Price_Trend'] = (df['close'] - df['open'].shift(1)) * df['close'].rolling(window=200).mean()
    df['Adjusted_Volume_Weighted_Momentum'] = df['Volume_Weighted_Momentum'] * df['Price_Trend']

    # Compute Dynamic Momentum Oscillator
    df['Dynamic_Momentum_Oscillator'] = df['Adjusted_Volume_Weighted_Momentum'] - df['close'].rolling(window=50).mean()

    return df['Dynamic_Momentum_Oscillator'].dropna()
```
