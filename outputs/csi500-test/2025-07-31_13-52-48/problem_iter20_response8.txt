```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    df['CloseHighRatio'] = df['close'] / df['high']
    df['CloseLowRatio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['CombinedRatio'] = 0.7 * df['CloseHighRatio'] + 0.3 * df['CloseLowRatio']
    
    # Calculate Intraday Price Movements
    df['HighLowDiff'] = df['high'] - df['low']
    df['HighLowPct'] = df['HighLowDiff'] / df['open']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['CloseOpenPct'] = df['CloseOpenDiff'] / df['open']
    
    # Average Intraday Price Movements
    df['AvgPctMove'] = (df['HighLowPct'] + df['CloseOpenPct']) / 2
    df['SmoothedPctMove'] = df['AvgPctMove'].ewm(span=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    df['VolumeAdjMomentum'] = df['PriceChange'] * np.sqrt(df['volume'])
    
    # Confirm with Volume
    df['VolChange'] = df['volume'] - df['volume'].shift(1)
    df['VolChangePct'] = df['VolChange'] / df['volume'].shift(1)
    
    # Weighted Final Alpha Factor
    df['WeightedFinalFactor'] = df['SmoothedPctMove'] * df['VolChangePct']
    
    # Apply Volume Shock Filter
    df['AbsVolChange'] = df['volume'] - df['volume'].shift(1)
    vol_change_85th = df['AbsVolChange'].quantile(0.85)
    df = df[df['AbsVolChange'] < vol_change_85th]
    
    # Calculate Intraday Reversal Score
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    intraday_volatility = (df['high'] - df['low']) / df['open']
    df['VolumeWeightedIntradayVolatility'] = (intraday_volatility * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    df['ReversalScore'] = df['IntradayReturn'] / df['VolumeWeightedIntradayVolatility']
    df['DecayedReversalScore'] = df['ReversalScore'].ewm(halflife=5).mean()
    
    # Calculate Intraday Range
    df['IntradayRange'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['VolumeWeightedIntradayRange'] = (df['IntradayRange'] * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Combine Reversal Score and Intraday Range
    df['CombinedReversalRange'] = df['DecayedReversalScore'] + df['IntradayRange']
    df['DecayedCombinedReversalRange'] = df['CombinedReversalRange'].ewm(halflife=5).mean()
    
    # Smooth the Average using a 3-day Exponential Moving Average
    df['SmoothedIntradayPriceMovement'] = df['AvgPctMove'].ewm(span=3).mean()
    
    # Calculate Price Momentum
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    df['PriceMomentum'] = df['LogReturn'].rolling(window=5).sum()
    
    # Weighted Final Alpha Factor
    df['FinalAlphaFactor'] = df['SmoothedPctMove'] * df['VolChangePct']
    
    return df['FinalAlphaFactor'].dropna()
```
