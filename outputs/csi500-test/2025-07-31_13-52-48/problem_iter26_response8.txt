```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Intraday Volatility (using a weighted average of Close-to-Open Ratio and Intraday Price Range)
    intraday_volatility = 0.5 * close_to_open_ratio + 0.5 * intraday_price_range
    
    # Interact Intraday Volatility with Volume
    alpha_factor = intraday_volatility * df['volume']
    
    return alpha_factor
```
