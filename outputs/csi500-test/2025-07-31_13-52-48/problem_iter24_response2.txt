import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns for open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    ma_money_flow = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of money flow
    std_money_flow = money_flow.rolling(window=10).std()
    
    # Calculate the z-score of money flow
    z_score_money_flow = (money_flow - ma_money_flow) / std_money_flow
    
    # Calculate the 20-day exponential moving average of close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the percentage change in close price relative to the EMA
    pct_change_close_to_ema = (df['close'] - ema_close) / ema_close
    
    # Combine the z-score of money flow and the percentage change in close price relative to the EMA
    factor = z_score_money_flow * pct_change_close_to_ema
    
    return factor
