import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)
    
    # Adjust Momentum by Volume
    df['adjusted_momentum'] = df['daily_momentum'] * df['volume']
    
    # Compute Rolling Sum of Adjusted Momentum over 20 Days
    df['rolling_sum_adjusted_momentum'] = df['adjusted_momentum'].rolling(window=20).sum()
    
    # Compute Moving Average over 60 Days
    df['moving_average_60'] = df['rolling_sum_adjusted_momentum'].rolling(window=60).mean()
    
    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_sum_adjusted_momentum'] - df['moving_average_60']
    
    # Return the factor values
    return df['dynamic_momentum_oscillator']
