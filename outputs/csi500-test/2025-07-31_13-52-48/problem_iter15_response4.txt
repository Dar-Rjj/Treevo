```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    close_high_ratio = df['close'] / df['high']
    close_low_ratio = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    combined_ratio = 0.6 * close_high_ratio + 0.4 * close_low_ratio
    
    # Calculate Intraday Price Movements
    high_low_diff = df['high'] - df['low']
    high_low_pct = high_low_diff / df['open']
    close_open_diff = df['close'] - df['open']
    close_open_pct = close_open_diff / df['open']
    
    # Average Intraday Price Movements
    avg_pct_move = (high_low_pct + close_open_pct) / 2
    smoothed_pct_move = avg_pct_move.ewm(span=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    price_change = df['close'].diff()
    volume_adj_momentum = price_change * np.sqrt(df['volume'])
    
    # Confirm with Volume
    vol_change = df['volume'].diff()
    vol_change_pct = vol_change / df['volume'].shift(1)
    
    # Weighted Final Alpha Factor
    weighted_final_factor = smoothed_pct_move * vol_change_pct
    
    # Apply Volume Shock Filter
    abs_vol_change = abs(df['volume'] - df['volume'].shift(1))
    volume_shock_threshold = abs_vol_change.quantile(0.9)
    filtered_factor = weighted_final_factor[abs_vol_change < volume_shock_threshold]
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    ema_intraday_return = intraday_return.ewm(span=5).mean()
    
    # Incorporate Volatility Component
    intraday_range = df['high'] - df['low']
    volatility_adjusted_momentum = volume_adj_momentum / intraday_range.rolling(window=5).std()
    
    # Calculate Price Momentum
    price_momentum = np.log(df['close']).diff().rolling(window=5).sum().shift(1)
    
    # Combine all components into the final alpha factor
    final_alpha_factor = (filtered_factor + ema_intraday_return + volatility_adjusted_momentum + price_momentum).dropna()
    
    return final_alpha_factor
```
