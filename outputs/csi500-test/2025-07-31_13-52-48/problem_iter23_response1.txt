```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor, indexed by 'date'.
    """
    # Calculate the relative strength of the close price compared to the open price
    relative_strength = (df['close'] - df['open']) / df['open']
    
    # Calculate the volatility as the range between high and low prices
    volatility = (df['high'] - df['low']) / df['low']
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor as the difference between the 10-day and 30-day moving averages
    momentum = ma_10 - ma_30
    
    # Combine the factors into a single alpha factor
    alpha_factor = (relative_strength + momentum) / (volatility + 1e-6) * vwap
    
    return alpha_factor
```
