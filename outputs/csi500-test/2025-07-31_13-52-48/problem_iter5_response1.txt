```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Exponential Moving Average of Intraday Return
    alpha = 0.2  # Exponential Smoothing Factor
    df['ema_intraday_return'] = df['intraday_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_range'] = df['high'] - df['low']
    df['std_intraday_range'] = df['intraday_range'].rolling(window=10).std()
    df['intraday_momentum_volatility'] = df['ema_intraday_return'] * (1 / df['std_intraday_range'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Weighted Intraday Volatility
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['vwma_intraday_volatility'] = df['weighted_intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['reversal_score'] = df['close_to_open_return'] / df['vwma_intraday_volatility']
    
    # Apply Decay Function to Smooth the Reversal Score
    half_life = 5
    df['smoothed_reversal_score'] = df['reversal_score'].ewm(halflife=half_life, adjust=False).mean()
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()
    
    # Apply Volume Shock Filter
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)).abs()
    volume_threshold = df['volume_change'].quantile(0.75)
    df['volume_shock_filter'] = (df['volume_change'] < volume_threshold)
    
    # Combine Intraday Reversal, Price Momentum, and Intraday Momentum with Volatility
    df['combined_factor'] = (df['smoothed_reversal_score'] + 
                             df['price_momentum'] + 
                             df['intraday_momentum_volatility'])
    
    # Apply the volume shock filter
    df['final_factor'] = df['combined_factor'] * df['volume_shock_filter']
    
    return df['final_factor']
```
