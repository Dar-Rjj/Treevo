```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['momentum_5d'] = df['log_return'].rolling(window=5).sum()
    df['volume_weighted_momentum'] = df['momentum_5d'] * df['volume'].rolling(window=20).sum()
    
    # Apply Combined Volume and Amount Shock Filter
    df['abs_volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    df['abs_amount_change'] = abs(df['amount'] - df['amount'].shift(1))
    volume_75th = df['abs_volume_change'].quantile(0.75)
    amount_75th = df['abs_amount_change'].quantile(0.75)
    df['shock_filtered'] = (df['abs_volume_change'] < volume_75th) & (df['abs_amount_change'] < amount_75th)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['volatility_weighted'] = df['intraday_volatility'] * df['volume']
    df['volatility_weighted_moving_avg'] = df['volatility_weighted'].rolling(window=10).mean()
    
    # Calculate Intraday Reversal Score
    df['intraday_reversal_score'] = df['close_to_open_return'] / df['volatility_weighted_moving_avg']
    df['intraday_reversal_exp_decay'] = df['intraday_reversal_score'].ewm(halflife=3).mean()
    
    # Adjust Combined Momentum by Daily Price Volatility
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['daily_price_volatility'] = df['daily_price_change'].rolling(window=20).std()
    df['adjusted_combined_momentum'] = df['volume_weighted_momentum'] / df['daily_price_volatility']
    
    # Calculate Momentum Difference
    df['momentum_20d'] = df['adjusted_combined_momentum'].rolling(window=20).mean()
    df['momentum_60d'] = df['adjusted_combined_momentum'].rolling(window=60).mean()
    df['momentum_difference'] = df['momentum_20d'] - df['momentum_60d']
    
    # Introduce Price Trend Indicator
    df['sma_50d'] = df['close'].rolling(window=50).mean()
    df['price_trend_indicator'] = np.where(df['close'] > df['sma_50d'], 1, -1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_difference'] * df['price_trend_indicator']
    
    return df['alpha_factor']
```
