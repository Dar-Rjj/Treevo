```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High-Low Spread
    high_low_spread = df['high'] - df['low']
    close_to_spread = df['close'] / high_low_spread
    
    # Calculate Volume-Adjusted Momentum
    prev_close = df['close'].shift(1)
    price_return = df['close'] - prev_close
    volume_adjusted_momentum = price_return * np.sqrt(df['volume'])
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Calculate Price Range Adjusted Volatility
    prices = df[['open', 'high', 'low', 'close']]
    volatility = prices.std(axis=1)
    price_range_adjusted_volatility = volatility / price_range
    
    # Combine Momentum, Spread, and Adjusted Volatility
    alpha_factor = close_to_spread + volume_adjusted_momentum + price_range_adjusted_volatility
    
    return alpha_factor
```
