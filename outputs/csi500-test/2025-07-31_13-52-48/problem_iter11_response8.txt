```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    alpha = 0.1  # Exponential smoothing factor
    n_days_volatility = 20  # Number of days to consider for volatility
    m_days_volume_trend = 10  # Number of days to consider for volume trend
    
    # Calculate Intraday Momentum
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum_ema'] = df['intraday_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Incorporate Volatility Component
    df['intraday_price_range'] = (df['high'] - df['low']) / df['open']
    df['volatility'] = df['intraday_price_range'].rolling(window=n_days_volatility).std()
    df['intraday_momentum_adjusted'] = df['intraday_momentum_ema'] * df['volatility']
    
    # Introduce Enhanced Volume Impact
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * df['volume'] / df['open']
    df['volume_weighted_intraday_return_ema'] = df['volume_weighted_intraday_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Integrate Volume Trend
    df['volume_roc'] = df['volume'].pct_change(periods=m_days_volume_trend)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_intraday_return_ema'] + df['volume_roc']
    
    return df['alpha_factor']
```
