```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['momentum'] = df['log_return'].rolling(window=5).sum().shift(1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = np.abs(df['volume'] - df['volume'].shift(1))
    
    # Calculate Daily Amount Change
    df['amount_change'] = np.abs(df['amount'] - df['amount'].shift(1))
    
    # Determine Thresholds for High Volume and Amount Changes
    volume_threshold = df['volume_change'].quantile(0.75)
    amount_threshold = df['amount_change'].quantile(0.75)
    
    # Apply Combined Volume and Amount Shock Filter
    df['shock_filter'] = (df['volume_change'] < volume_threshold) & (df['amount_change'] < amount_threshold)
    
    # Filter the momentum by the shock filter
    df['filtered_momentum'] = df['momentum'] * df['shock_filter']
    
    return df['filtered_momentum'].dropna()
```
