```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = df['close'].pct_change()
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume Shock Filter
    df['volume_change'] = df['volume'].diff().abs()
    volume_threshold = df['volume_change'].quantile(0.75)
    df_filtered = df[df['volume_change'] < volume_threshold]

    # Calculate Intraday Price Movement
    df_filtered['high_low_movement'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['open']
    df_filtered['close_open_movement'] = (df_filtered['close'] - df_filtered['open']) / df_filtered['open']
    
    # Combine Intraday Price Movements
    df_filtered['intraday_movement'] = (df_filtered['high_low_movement'] + df_filtered['close_open_movement']) / 2
    df_filtered['smoothed_intraday_movement'] = df_filtered['intraday_movement'].rolling(window=3).mean()

    # Confirm with Volume
    df_filtered['volume_change_pct'] = (df_filtered['volume'] - df_filtered['volume'].shift(1)) / df_filtered['volume'].shift(1)

    # Weighted Final Alpha Factor
    df_filtered['alpha_factor'] = df_filtered['smoothed_intraday_movement'] * df_filtered['volume_change_pct']

    return df_filtered['alpha_factor']
```
