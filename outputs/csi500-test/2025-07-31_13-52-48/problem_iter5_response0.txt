import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Change']
    df['Rolling_Sum_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=20).sum()
    
    # Determine Threshold for High Volume-Adjusted Momentum
    volume_adjusted_momentum_75th = df['Volume_Adjusted_Momentum'].abs().quantile(0.75)
    df = df[df['Volume_Adjusted_Momentum'].abs() < volume_adjusted_momentum_75th]

    # Calculate Close to High Ratio and Close to Low Ratio
    df['Close_High_Ratio'] = df['close'] / df['high']
    df['Close_Low_Ratio'] = df['close'] / df['low']

    # Combine Ratios
    df['Combined_Ratio'] = 0.5 * df['Close_High_Ratio'] + 0.5 * df['Close_Low_Ratio']

    # Combine Adjusted Momentum and Combined Ratio
    df['Combined_Factor'] = df['Combined_Ratio'] + df['Rolling_Sum_Volume_Adjusted_Momentum']

    # Compute Dynamic Price Momentum Oscillator
    df['Dynamic_Price_Momentum_Oscillator'] = df['Rolling_Sum_Volume_Adjusted_Momentum'] - df['Rolling_Sum_Volume_Adjusted_Momentum'].rolling(window=60).mean()

    # Apply Volume Shock Filter
    df['Abs_Volume_Change'] = (df['volume'] - df['volume'].shift(1)).abs()
    volume_change_75th = df['Abs_Volume_Change'].quantile(0.75)
    df = df[df['Abs_Volume_Change'] < volume_change_75th]

    # Return the final factor
    return df['Dynamic_Price_Momentum_Oscillator'].dropna()
