```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['Volume_Adjusted_Momentum'] = (df['volume'] - df['volume'].shift(1)) * df['Price_Momentum']
    df['Rolling_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=20).sum()
    threshold_vol_adj_mom = df['Rolling_Volume_Adjusted_Momentum'].abs().quantile(0.75)
    df = df[df['Rolling_Volume_Adjusted_Momentum'].abs() < threshold_vol_adj_mom]

    # Calculate Intraday Price Movement
    df['Intraday_High_Low'] = (df['high'] - df['low']) / df['open']
    df['Intraday_Close_Open'] = (df['close'] - df['open']) / df['open']
    df['Average_Intraday_Movement'] = (df['Intraday_High_Low'] + df['Intraday_Close_Open']) / 2
    df['Smoothed_Average_Intraday_Movement'] = df['Average_Intraday_Movement'].ewm(span=3, adjust=False).mean()

    # Combine Ratios
    df['Close_to_High_Ratio'] = df['close'] / df['high']
    df['Close_to_Low_Ratio'] = df['close'] / df['low']
    df['Combined_Ratios'] = 0.5 * df['Close_to_High_Ratio'] + 0.5 * df['Close_to_Low_Ratio']

    # Combine Adjusted Momentum, Intraday Movement, and Ratios
    df['Combined_Factor'] = df['Rolling_Volume_Adjusted_Momentum'] + df['Smoothed_Average_Intraday_Movement'] + df['Combined_Ratios']

    # Compute Dynamic Price Momentum Oscillator
    df['Dynamic_Price_Oscillator'] = df['Rolling_Volume_Adjusted_Momentum'] - df['Rolling_Volume_Adjusted_Momentum'].rolling(window=60).mean()

    # Apply Volume Shock Filter
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    threshold_vol_change = df['Volume_Change'].abs().quantile(0.75)
    df = df[df['Volume_Change'].abs() < threshold_vol_change]

    # Confirm with Volume
    df['Volume_Change_Percentage'] = df['Volume_Change'] / df['volume'].shift(1)
    df['Final_Alpha_Factor'] = df['Smoothed_Average_Intraday_Movement'] * df['Volume_Change_Percentage']

    return df['Final_Alpha_Factor']
```
