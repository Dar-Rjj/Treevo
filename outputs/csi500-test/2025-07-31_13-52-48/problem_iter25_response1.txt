```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Combined_Price_Momentum'] = df['Log_Return'].rolling(window=5).sum()

    # Apply Combined Volume Shock Filter
    df['Volume_Change'] = abs(df['volume'] - df['volume'].shift(1))
    volume_75th_percentile = df['Volume_Change'].quantile(0.75)
    df = df[df['Volume_Change'] < volume_75th_percentile]

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Volatility
    df['Vol_Weighted_Volatility'] = df['Intraday_Volatility'] * df['volume']
    df['Vol_Weighted_MA_Volatility'] = df['Vol_Weighted_Volatility'].rolling(window=10).mean() / df['volume'].rolling(window=10).mean()

    # Calculate Intraday Reversal Score
    df['Reversal_Score'] = df['Close_to_Open_Return'] / df['Vol_Weighted_MA_Volatility']

    # Apply Exponential Decay to Reversal Score
    half_life = 5
    decay_factor = 0.5 ** (1/half_life)
    df['Decayed_Reversal_Score'] = df['Reversal_Score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['open']

    # Calculate Volume-Weighted Moving Average of Intraday Range
    df['Vol_Weighted_Range'] = df['Intraday_Range'] * df['volume']
    df['Vol_Weighted_MA_Range'] = df['Vol_Weighted_Range'].rolling(window=10).mean() / df['volume'].rolling(window=10).mean()

    # Combine Reversal Score and Intraday Range
    df['Combined_Score'] = df['Decayed_Reversal_Score'] + df['Vol_Weighted_MA_Range']
    df['Decayed_Combined_Score'] = df['Combined_Score'].ewm(alpha=decay_factor, adjust=False).mean()

    # Calculate Intraday Price Movement
    df['Price_Movement_HL'] = (df['high'] - df['low']) / df['open']
    df['Price_Movement_CO'] = (df['close'] - df['open']) / df['open']
    df['Average_Price_Movement'] = (df['Price_Movement_HL'] + df['Price_Movement_CO']) / 2

    # Smooth the Average using a 3-day Exponential Moving Average
    df['Smoothed_Price_Movement'] = df['Average_Price_Movement'].ewm(span=3, adjust=False).mean()

    # Confirm with Volume
    df['Volume_Change_Percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Weighted Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Smoothed_Price_Movement'] * df['Volume_Change_Percentage']

    # Calculate Close to High and Low Ratios
    df['CloseHighRatio'] = df['close'] / df['high']
    df['CloseLowRatio'] = df['close'] / df['low']

    # Combine Close to High and Low Ratios
    df['CombinedRatio'] = 0.7 * df['CloseHighRatio'] + 0.3 * df['CloseLowRatio']

    # Apply Volume Shock Filter
    volume_85th_percentile = df['Volume_Change'].quantile(0.85)
    df = df[df['Volume_Change'] < volume_85th_percentile]

    return df['Final_Alpha_Factor']
```
