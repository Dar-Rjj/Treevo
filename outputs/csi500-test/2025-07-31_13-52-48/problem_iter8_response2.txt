```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 10-day moving average of the typical price
    ma_typical_price_10 = typical_price.rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the typical price
    std_typical_price_10 = typical_price.rolling(window=10).std()
    
    # Calculate the 10-day z-score of the typical price
    z_score_typical_price_10 = (typical_price - ma_typical_price_10) / std_typical_price_10
    
    # Calculate the 10-day moving average of the volume
    ma_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day z-score of the volume
    z_score_volume_10 = (df['volume'] - ma_volume_10) / df['volume'].rolling(window=10).std()
    
    # Combine the z-scores to create the alpha factor
    alpha_factor = z_score_typical_price_10 * z_score_volume_10
    
    return alpha_factor
```
