def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Normalize the volume weighted return by the mean volume
    mean_volume = df['volume'].mean()
    df['normalized_volume_weighted_return'] = df['volume_weighted_return'] / mean_volume
    
    return df['normalized_volume_weighted_return']
