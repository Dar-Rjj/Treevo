```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['50_day_sma'] = df['close'].rolling(window=50).mean()
    
    # Compute the difference between shorter and longer period moving averages
    df['5_20_day_sma_diff'] = df['5_day_sma'] - df['20_day_sma']
    
    # Examine price changes
    df['close_pct_change'] = df['close'].pct_change()
    df['5_day_close_pct_sum'] = df['close_pct_change'].rolling(window=5).sum()
    
    # Incorporate volume
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['20_day_avg_volume']
    
    # Integrate amount
    df['20_day_total_amount'] = df['amount'].rolling(window=20).sum()
    df['avg_amount'] = df['20_day_total_amount'] / 20
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Investigate volatility
    df['20_day_return_std'] = df['close'].pct_change().rolling(window=20).std()
    df['14_day_avg_true_range'] = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Analyze high and low prices
    df['20_day_avg_high_low'] = (df['high'] + df['low']) / 2
    df['20_day_avg_high_low_mean'] = df['20_day_avg_high_low'].rolling(window=20).mean()
    df['20_day_high_low_diff'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['5_20_day_sma_diff'] +
        df['5_day_close_pct_sum'] +
        df['volume_ratio'] +
        df['amount_ratio'] -
        df['20_day_return_std'] -
        df['14_day_avg_true_range'] +
        df['20_day_avg_high_low_mean'] -
        df['20_day_high_low_diff']
    )
    
    return df['alpha_factor']
```
