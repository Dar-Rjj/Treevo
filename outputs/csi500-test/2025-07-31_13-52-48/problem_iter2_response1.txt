```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    close_high_ratio = df['close'] / df['high']
    close_low_ratio = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    combined_ratio = 0.5 * close_high_ratio + 0.5 * close_low_ratio
    
    # Calculate Intraday Price Movements
    high_low_diff = df['high'] - df['low']
    high_low_pct = high_low_diff / df['open']
    close_open_diff = df['close'] - df['open']
    close_open_pct = close_open_diff / df['open']
    
    # Average Intraday Price Movements
    avg_pct_move = (high_low_pct + close_open_pct) / 2
    smoothed_pct_move = avg_pct_move.ewm(span=3, adjust=False).mean()
    
    # Calculate Volume-Adjusted Momentum
    price_change = df['close'].diff()
    volume_adj_momentum = price_change * df['volume']
    
    # Confirm with Volume
    vol_change = df['volume'].diff()
    vol_change_pct = vol_change / df['volume'].shift(1)
    weighted_final_factor = smoothed_pct_move * vol_change_pct
    
    # Combine All Factors
    final_alpha_factor = combined_ratio + volume_adj_momentum + weighted_final_factor
    
    return final_alpha_factor
```
