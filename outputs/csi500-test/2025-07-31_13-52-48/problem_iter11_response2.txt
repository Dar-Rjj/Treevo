```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close to High and Low Ratios
    df['CloseHighRatio'] = df['close'] / df['high']
    df['CloseLowRatio'] = df['close'] / df['low']
    
    # Combine Close to High and Low Ratios
    df['CombinedRatio'] = 0.5 * df['CloseHighRatio'] + 0.5 * df['CloseLowRatio']
    
    # Calculate Intraday Price Movements
    df['HighLowDiff'] = df['high'] - df['low']
    df['HighLowPct'] = df['HighLowDiff'] / df['open']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['CloseOpenPct'] = df['CloseOpenDiff'] / df['open']
    
    # Average Intraday Price Movements
    df['AvgPctMove'] = (df['HighLowPct'] + df['CloseOpenPct']) / 2
    df['SmoothedPctMove'] = df['AvgPctMove'].ewm(span=3).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    df['VolumeAdjMomentum'] = df['PriceChange'] * df['volume']
    
    # Confirm with Volume
    df['VolChange'] = df['volume'] - df['volume'].shift(1)
    df['VolChangePct'] = df['VolChange'] / df['volume'].shift(1)
    df['WeightedFinalFactor'] = df['SmoothedPctMove'] * df['VolChangePct']
    
    # Apply Volume Shock Filter
    df['AbsVolChange'] = df['VolChange'].abs()
    vol_shock_threshold = df['AbsVolChange'].quantile(0.75)
    df['FilteredFactor'] = df.apply(lambda row: row['WeightedFinalFactor'] if row['AbsVolChange'] < vol_shock_threshold else np.nan, axis=1)
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] / df['open']) - 1
    
    # Compute Exponential Moving Average of Intraday Return
    df['EMAIntradayReturn'] = df['IntradayReturn'].ewm(span=3).mean()
    
    # Incorporate Volatility Component
    df['IntradayRange'] = df['high'] - df['low']
    df['VolatilityAdjMomentum'] = df['VolumeAdjMomentum'] / df['IntradayRange'].rolling(window=5).std()
    
    # Calculate Price Momentum
    df['LogReturns'] = np.log(df['close'] / df['close'].shift(1))
    df['PriceMomentum'] = df['LogReturns'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['FilteredFactor'] * df['EMAIntradayReturn'] * df['VolatilityAdjMomentum'] * df['PriceMomentum']
    
    return df['AlphaFactor']
```
