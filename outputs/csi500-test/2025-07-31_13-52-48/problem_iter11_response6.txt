```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=5).sum()

    # Apply Volume-Adjusted Momentum Filter
    df['volume_adjusted_momentum'] = (df['volume'] - df['volume'].shift(1)) * df['price_momentum']
    df['rolling_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    df['abs_rolling_volume_adjusted_momentum'] = df['rolling_volume_adjusted_momentum'].abs()
    volume_momentum_threshold = df['abs_rolling_volume_adjusted_momentum'].quantile(0.75)
    df = df[df['abs_rolling_volume_adjusted_momentum'] < volume_momentum_threshold]

    # Calculate Intraday High-Low Difference
    df['intraday_high_low_diff'] = df['high'] - df['low']

    # Average Intraday High-Low Difference Over Period
    df['avg_intraday_high_low_diff'] = df['intraday_high_low_diff'].rolling(window=5).mean()

    # Adjust by Volume Volatility
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_volatility'] = df['volume_change'].rolling(window=10).std()
    df['final_adjustment'] = df['intraday_high_low_diff'] / df['volume_volatility']

    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)

    # Calculate Volume-Adjusted Daily Momentum
    df['volume_adjusted_daily_momentum'] = df['volume'] * df['final_adjustment']

    # Combine Intraday and Open-Close Price Movements
    df['high_low_percentage'] = (df['high'] - df['low']) / df['open']
    df['open_close_percentage'] = (df['close'] - df['open']) / df['open']
    df['average_percentage_movement'] = (df['high_low_percentage'] + df['open_close_percentage']) / 2
    df['smoothed_average_percentage_movement'] = df['average_percentage_movement'].ewm(span=3, adjust=False).mean()

    # Confirm with Volume
    df['volume_change_percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Introduce Open-Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']

    # Calculate Intraday Price Range Ratio
    df['open_close_range'] = df['close'] - df['open']
    df['high_low_range'] = df['high'] - df['low']
    df['price_range_ratio'] = df['open_close_range'] / df['high_low_range']

    # Adjust Intraday High-Low Difference by Price Range Ratio
    df['adjusted_intraday_high_low_diff'] = df['intraday_high_low_diff'] * df['price_range_ratio']

    # Calculate Final Momentum Factor
    df['final_momentum_factor'] = df['adjusted_intraday_high_low_diff'] + df['volume_adjusted_daily_momentum']

    # Compute Rolling Sum of Final Momentum
    df['rolling_final_momentum'] = df['final_momentum_factor'].rolling(window=20).sum()

    # Compute Dynamic Momentum Oscillator
    df['dynamic_momentum_oscillator'] = df['rolling_final_momentum'] - df['rolling_final_momentum'].rolling(window=60).mean()

    return df['dynamic_momentum_oscillator']
```
