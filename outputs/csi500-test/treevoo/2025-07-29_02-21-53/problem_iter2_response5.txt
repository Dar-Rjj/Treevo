```python
import pandas as pd

def heuristics_v2(df, N=5, M=10):
    # Calculate Signed Transaction Volume
    df['high_low_diff'] = df['high'] - df['low']
    df['signed_volume'] = df['volume'] * df['high_low_diff'].apply(lambda x: 1 if x > 0 else -1)

    # Calculate Accumulated Buy-Sell Imbalance
    df['positive_signed_volume'] = df['signed_volume'].apply(lambda x: x if x > 0 else 0)
    df['negative_signed_volume'] = df['signed_volume'].apply(lambda x: x if x < 0 else 0)
    df['accumulated_buy'] = df['positive_signed_volume'].rolling(window=N).sum()
    df['accumulated_sell'] = abs(df['negative_signed_volume']).rolling(window=N).sum()

    # Compute Order Imbalance Score
    df['total_absolute_volumes'] = (df['positive_signed_volume'] + abs(df['negative_signed_volume'])).rolling(window=N).sum()
    df['order_imbalance_score'] = (df['accumulated_buy'] / df['total_absolute_volumes']) - (df['accumulated_sell'] / df['total_absolute_volumes'])

    # Calculate Cumulative Return over N Days
    df['cumulative_return'] = (df['close'] / df['close'].shift(N) - 1) * 100

    # Measure Volume Trend
    df['volume_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['volume_trend'] = df['volume'] / df['volume_moving_avg']

    # Calculate Adjusted High-Low Spread with Order Imbalance
    df['adjusted_high_low_spread'] = (df['high'] - df['low']) * (1 + df['order_imbalance_score'])

    # Combine Adjusted High-Low Spread and Cumulative Return
    df['final_alpha_factor'] = 0.5 * df['adjusted_high_low_spread'] + 0.5 * df['cumulative_return']

    return df['final_alpha_factor'].dropna()
```
