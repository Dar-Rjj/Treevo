```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['price_momentum_5d'] = df['close'] - df['close'].shift(5)

    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']

    # Smooth High-Low Range with 20-day Moving Average
    df['smoothed_high_low_range'] = df['daily_high_low_range'].rolling(window=20).mean()

    # Calculate Long-Term High-Low Range (50-day)
    df['long_term_high_low_range'] = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()

    # Compute Momentum of High-Low Ranges
    df['range_momentum'] = df['long_term_high_low_range'] - df['smoothed_high_low_range']

    # Weight by Volume
    df['weighted_range_momentum'] = df['range_momentum'] * df['volume']

    # Combine All Factors
    df['combined_factor'] = df['price_momentum_5d'] + df['weighted_range_momentum']

    return df['combined_factor']
```
