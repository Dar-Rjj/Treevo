import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price over a 10-day window
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the ratio of the current close price to the 10-day mean
    df['close_to_10d_mean_ratio'] = df['close'] / df['close_10d_mean']
    
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the average of the amount traded over a 5-day window
    df['amount_5d_mean'] = df['amount'].rolling(window=5).mean()
    
    # Calculate the ratio of the current amount to the 5-day mean
    df['amount_to_5d_mean_ratio'] = df['amount'] / df['amount_5d_mean']
    
    # Calculate the alpha factor as a weighted combination of the ratios
    df['alpha_factor'] = (df['close_to_10d_mean_ratio'] + df['amount_to_5d_mean_ratio']) / 2
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor'].dropna()
