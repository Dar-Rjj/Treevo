```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Exponential Moving Average of Returns
    df['short_ema'] = df['daily_return'].ewm(span=12, adjust=False).mean()
    df['long_ema'] = df['daily_return'].ewm(span=26, adjust=False).mean()
    
    # Calculate MACD
    df['macd'] = df['short_ema'] - df['long_ema']
    
    # Calculate Volume Adjusted Momentum
    df['raw_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['volume_adjusted_momentum'] = df['raw_momentum'] * df['volume']
    
    # Calculate Intraday Price Movement
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    df['sum_high_low'] = df['high_low_diff'].rolling(window=50).sum()
    df['sum_close_open'] = df['close_open_diff'].rolling(window=50).sum()
    
    # Calculate Price Momentum Score
    df['price_volatility'] = df['close'].rolling(window=50).std()
    df['price_momentum_score'] = (df['sum_high_low'] + df['sum_close_open']) * df['price_volatility']
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=50).mean()
    df['high_volume'] = df['volume'] > 1.5 * df['avg_volume']
    
    # Normalize MACD (Do not use global statistics, local normalization)
    df['normalized_macd'] = df['macd'] / df['macd'].rolling(window=50).max()
    
    # Combine MACD and Volume Adjusted Momentum
    df['combined_macd_volume_momentum'] = df['normalized_macd'] + df['volume_adjusted_momentum']
    
    # Integrate Price Momentum and Volume Activity
    df['alpha_factor'] = df['price_momentum_score'] * df['high_volume']
    
    return df['alpha_factor']
```
