```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price and High-Low Range Movements
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Aggregate and Smooth Price Movements
    period = 20
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=period).sum()
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=period).sum()
    df['smoothed_high_low_diff'] = df['high_low_diff'].rolling(window=period).mean()
    
    # Combine Price and High-Low Range Momentum
    df['momentum_score'] = (df['sum_close_open_diff'] - df['sum_high_low_diff']) / df['close'].rolling(window=period).std()
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=period).mean()
    df['high_volume'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Identify Sudden Volume Spikes
    df['volume_spike'] = (df['volume'] > 2 * avg_volume).astype(int)
    
    # Integrate Price, High-Low Range, and Volume
    df['alpha_factor'] = df['momentum_score'] * df['high_volume'] + df['momentum_score'] * df['volume_spike']
    
    return df['alpha_factor']
```
