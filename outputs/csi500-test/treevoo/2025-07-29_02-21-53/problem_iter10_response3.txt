```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Aggregate Price Movements
    lookback_period = 20
    df['Sum_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=lookback_period).sum()
    df['Sum_Close_Open_Diff'] = df['Close_Open_Diff'].rolling(window=lookback_period).sum()

    # Calculate Integrated Momentum Score
    df['Integrated_Momentum'] = df['Sum_High_Low_Diff'] + df['Sum_Close_Open_Diff']

    # Weight by Recent Price Volatility
    df['Volatility'] = df['close'].rolling(window=lookback_period).std()
    df['Weighted_Momentum'] = df['Integrated_Momentum'] * (df['Volatility'] / df['Volatility'].max())

    # Evaluate Volume Activity and True Range
    avg_volume = df['volume'].mean()
    high_volume_threshold = 1.5  # example threshold
    df['High_Volume'] = (df['volume'] > avg_volume * high_volume_threshold).astype(int)

    df['True_Range'] = df[['high', 'low']].subtract(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=lookback_period).mean()

    # Integrate Price, Volume, and Order Imbalance
    df['Volume_Ratio'] = df['volume'] / avg_volume
    df['Price_Momentum_ATR'] = df['Integrated_Momentum'] / df['ATR']

    # Integrate Order Imbalance Score
    # This is a placeholder, as the exact calculation of order imbalance is not provided
    df['Order_Imbalance_Score'] = 0  # Placeholder for order imbalance score

    # Combine Price, Volume, and Order Imbalance Effects
    df['Alpha_Factor'] = df['Price_Momentum_ATR'] * df['Volume_Ratio'] * df['Order_Imbalance_Score']

    return df['Alpha_Factor'].dropna()
```
