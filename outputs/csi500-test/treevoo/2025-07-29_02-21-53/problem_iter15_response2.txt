```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['short_term_smoothed_range'] = df['high_low_range'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range
    df['long_term_high'] = df['high'].rolling(window=50).max()
    df['long_term_low'] = df['low'].rolling(window=50).min()
    df['long_term_high_low_range'] = df['long_term_high'] - df['long_term_low']
    
    # Compute Momentum of High-Low Ranges
    df['range_momentum'] = df['long_term_high_low_range'] - df['short_term_smoothed_range']
    
    # Weight by Volume
    df['volume_weighted_momentum'] = df['range_momentum'] * df['volume']
    
    # Integrate Price Change
    df['daily_price_change'] = df['close'] - df['open']
    df['smoothed_price_change'] = df['daily_price_change'].ewm(span=20, adjust=False).mean()
    
    # Combine with High-Low Range Momentum
    df['combined_factor'] = df['smoothed_price_change'] + df['volume_weighted_momentum']
    
    # Integrate Price Trend
    df['long_term_close'] = df['close'].rolling(window=100).mean()
    df['smoothed_long_term_close'] = df['long_term_close'].ewm(span=100, adjust=False).mean()
    df['price_trend_momentum'] = df['smoothed_long_term_close'] - df['smoothed_price_change']
    
    # Combine with High-Low Range and Volume Momentum
    df['final_factor'] = df['combined_factor'] + df['price_trend_momentum']
    
    return df['final_factor']
```
