```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference and Close-Open Change
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']

    # Aggregate and Smooth Price Movements
    short_term_periods = 20
    long_term_periods = 50
    
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=short_term_periods).sum()
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=short_term_periods).sum()

    df['smooth_high_low_range'] = df['high_low_diff'].ewm(span=short_term_periods, adjust=False).mean()
    df['long_term_high_low_range'] = (df['high'].rolling(window=long_term_periods).max() - 
                                      df['low'].rolling(window=long_term_periods).min())

    # Compute Momentum of High-Low Ranges
    df['momentum_high_low_range'] = df['long_term_high_low_range'] - df['smooth_high_low_range']

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['prev_close']), abs(x['low'] - x['prev_close'])), axis=1)

    # Calculate Rolling Average True Range (ATR)
    n_days = 14
    df['atr'] = df['true_range'].rolling(window=n_days).mean()

    # Adjust Price Momentum by ATR
    df['adjusted_momentum'] = (df['sum_high_low_diff'] + df['sum_close_open_change']) / df['atr']

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=short_term_periods).mean()
    df['is_high_volume'] = df['volume'] > 1.5 * df['avg_volume']

    # Detect Volume Spike
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['is_volume_spike'] = df['volume'] > 2 * df['5_day_avg_volume']

    # Adjust Momentum Based on Volume
    df['adjusted_momentum'] = np.where(df['is_volume_spike'], df['adjusted_momentum'] * 0.75, df['adjusted_momentum'])

    # Integrate Combined Factors
    df['interim_alpha_factor'] = df['adjusted_momentum'] * df['is_high_volume']

    # Combine ATR and Price Momentum
    df['combined_alpha_factor'] = df['interim_alpha_factor'] + df['momentum_high_low_range']

    # Incorporate Recent Price Trend
    df['recent_price_trend'] = df['close'].rolling(window=short_term_periods).mean()
    trend_multiplier = np.where(df['recent_price_trend'] > df['close'].shift(short_term_periods), 1, -1)
    df['final_alpha_factor'] = df['combined_alpha_factor'] * trend_multiplier

    # Introduce New Factor: Volume-Weighted ATR
    high_volume_true_range = df[df['is_high_volume']]['true_range']
    df['volume_weighted_atr'] = high_volume_true_range.rolling(window=n_days).mean()

    # Adjust Interim Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] / df['volume_weighted_atr']

    return df['final_alpha_factor']
```
