import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate Close-Open Change
    df['close_open_change'] = df['close'] - df['open']
    
    # Aggregate and Smooth Price Movements
    short_term = 20
    long_term = 50
    
    # Sum of High-Low Differences over a Period
    df['agg_high_low_diff'] = df['high_low_diff'].rolling(window=short_term).sum()
    
    # Sum of Close-Open Changes over a Period
    df['agg_close_open_change'] = df['close_open_change'].rolling(window=short_term).sum()
    
    # Smooth High-Low Range
    df['smooth_high_low_range'] = df['high_low_diff'].ewm(span=short_term).mean()
    
    # Long-Term High-Low Range
    df['long_term_high_low_range'] = df['high'].rolling(window=long_term).max() - df['low'].rolling(window=long_term).min()
    
    # Compute Momentum of High-Low Ranges
    df['high_low_momentum'] = df['long_term_high_low_range'] - df['smooth_high_low_range']
    
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Rolling Average True Range (ATR)
    n_days_atr = 14
    df['atr'] = df['true_range'].rolling(window=n_days_atr).mean()
    
    # Adjust Price Momentum by ATR
    df['weighted_high_low_range'] = df['agg_high_low_diff'] / df['atr']
    df['weighted_close_open_change'] = df['agg_close_open_change'] / df['atr']
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=short_term).mean()
    df['high_volume_indicator'] = (df['volume'] > 1.5 * df['avg_volume']).astype(int)
    
    # Detect Volume Spike
    df['five_day_volume_avg'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 2 * df['five_day_volume_avg']).astype(int)
    
    # Adjust Momentum Based on Volume
    df['adjusted_momentum'] = (df['weighted_high_low_range'] + df['weighted_close_open_change']) * (0.8 ** df['volume_spike'])
    
    # Integrate Combined Factors
    df['interim_alpha_factor'] = df['adjusted_momentum'] * df['high_volume_indicator']
    
    # Combine ATR and Price Momentum
    df['interpretable_factor'] = df['interim_alpha_factor'] + df['high_low_momentum']
    
    # Incorporate Recent Price Trend
    df['moving_avg_close'] = df['close'].rolling(window=short_term).mean()
    recent_trend_multiplier = np.sign(df['close'] - df['moving_avg_close'])
    df['interpretable_factor'] *= recent_trend_multiplier
    
    # Introduce New Factor: Volume-Weighted ATR
    df['high_volume_true_range'] = df['true_range'] * df['high_volume_indicator']
    df['volume_weighted_atr'] = df['high_volume_true_range'].rolling(window=n_days_atr).mean()
    df['final_alpha_factor'] = df['interpretable_factor'] / df['volume_weighted_atr']
    
    return df['final_alpha_factor']
