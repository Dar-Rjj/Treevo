```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the percentage change from open to close
    pct_change_open_close = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the relative distance of the close price from the daily high and low
    rel_dist_from_high = (df['high'] - df['close']) / daily_range
    rel_dist_from_low = (df['close'] - df['low']) / daily_range
    
    # Combine the factors into a single alpha factor
    alpha_factor = (pct_change_open_close * 0.4 
                    + (vwap - df['close']) * 0.3
                    + (rel_dist_from_high - rel_dist_from_low) * 0.3)
    
    return alpha_factor
```
