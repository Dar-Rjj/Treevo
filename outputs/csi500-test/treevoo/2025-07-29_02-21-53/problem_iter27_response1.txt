import pandas as pd

def heuristics_v2(df, N=10, M=14, K=20, L=50, P=20, Q=10, R=30):
    # Compute daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Sum of returns over a lookback window of N days
    df['N_day_return_sum'] = df['daily_return'].rolling(window=N).sum()
    
    # Weighted sum of returns over the lookback window
    weights = pd.Series(range(1, N+1), index=df.index[-N:])
    df['weighted_N_day_return_sum'] = df['daily_return'].rolling(window=N).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False)
    
    # True range
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    
    # Average true range (ATR) over M days
    df['ATR'] = df['true_range'].rolling(window=M).mean()
    
    # Relative ATR
    df['relative_ATR'] = df['ATR'] / df['ATR'].rolling(window=2*M).mean()
    
    # Volatility shift
    df['volatility_shift'] = df['ATR'] - df['ATR'].rolling(window=K).mean()
    
    # Volume intensity
    avg_volume = df['volume'].rolling(window=L).mean()
    df['volume_intensity'] = df['volume'] / avg_volume
    
    # Money flow
    df['money_flow'] = df['volume'] * (2 * df['close'] - df['high'] - df['low'])
    df['sum_money_flow'] = df['money_flow'].rolling(window=P).sum()
    
    # Normalized money flow
    df['normalized_money_flow'] = df['sum_money_flow'] / df['volume'].rolling(window=P).sum()
    
    # Composite factor
    df['composite_factor'] = 0.4 * df['N_day_return_sum'] + 0.3 * df['relative_ATR'] + 0.3 * df['volume_intensity']
    
    # Momentum factor for composite factor
    df['composite_momentum'] = df['composite_factor'].pct_change(Q)
    
    # Cumulative volume delta
    df['cumulative_volume_delta'] = (df['volume'] * (df['close'] - df['close'].shift(1))).rolling(window=R).sum()
    
    # Combined factor
    df['combined_factor'] = df['N_day_return_sum'] + df['cumulative_volume_delta']
    
    # Return the final alpha factor
    return df['combined_factor']
