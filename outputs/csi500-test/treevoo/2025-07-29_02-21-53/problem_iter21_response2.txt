import pandas as pd

def heuristics_v2(df):
    # Compute True Range
    df['True_Range'] = df[['High', 'Low']].sub(df['Close'].shift(1), axis=0).abs().max(axis=1)
    
    # Compute Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['High'] - df['Low']) + (df['Close'] - df['Open'])
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_TR'] = df['True_Range'] * df['Volume']
    df['Volume_Adjusted_IPM'] = df['Intraday_Price_Movement'] * df['Volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['EMA_High_Low_20'] = df['High'] - df['Low']
    df['EMA_High_Low_20'] = df['EMA_High_Low_20'].ewm(span=20, adjust=False).mean()
    
    df['EMA_Close_Open_10'] = df['Close'] - df['Open']
    df['EMA_Close_Open_10'] = df['EMA_Close_Open_10'].ewm(span=10, adjust=False).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Range_Momentum'] = df['EMA_Close_Open_10'] - df['EMA_High_Low_20']
    df['Volatility'] = df['Close'].pct_change().rolling(window=20).std() * (252 ** 0.5)
    df['Weighted_Range_Momentum'] = df['Range_Momentum'] * df['Volatility']
    
    # Evaluate Volume Activity
    df['EMA_Volume_20'] = df['Volume'].ewm(span=20, adjust=False).mean()
    df['High_Volume_Day'] = (df['Volume'] > df['EMA_Volume_20']).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Combined_Measure'] = df['Volume_Adjusted_TR'] + df['Volume_Adjusted_IPM']
    
    # Generate Final Alpha Factor
    df['Alpha_Factor'] = df['Combined_Measure'] * df['High_Volume_Day'] + \
                         (df['Range_Momentum'] * df['EMA_Close_Open_10'] * df['Volume'])
    
    return df['Alpha_Factor']
