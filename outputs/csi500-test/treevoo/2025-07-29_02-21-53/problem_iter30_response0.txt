```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range
    df['TrueRange'] = np.maximum(df['high'] - df['low'], 
                                 np.abs(df['high'] - df['close'].shift(1)), 
                                 np.abs(df['low'] - df['close'].shift(1)))
    
    # Adjust True Range by Square Root of Volume
    df['VolumeAdjustedTrueRange'] = df['TrueRange'] * np.sqrt(df['volume'])
    
    # Smooth High-Low Ranges with 20-day EMA
    df['HighLowDiff'] = df['high'] - df['low']
    df['SmoothedHighLow'] = df['HighLowDiff'].ewm(span=20, adjust=False).mean()
    
    # Integrate Volume-Adjusted True Range and Smoothed High-Low Ranges
    df['InteractionTerm'] = df['SmoothedHighLow'] * df['HighLowDiff']
    df['CombinedFactor'] = df['VolumeAdjustedTrueRange'] + df['InteractionTerm']
    
    # Incorporate Price Momentum
    df['SMA5'] = df['close'].rolling(window=5).mean()
    df['SMA20'] = df['close'].rolling(window=20).mean()
    df['Momentum'] = df['SMA5'] - df['SMA20']
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['CombinedFactor'] + df['Momentum']
    
    return df['AlphaFactor']
```
