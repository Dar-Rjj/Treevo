```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the amount-weighted average of close and open prices
    avg_price = (df['close'] + df['open']) / 2
    
    # Calculate the volume-to-amount ratio
    vol_to_amt_ratio = df['volume'] / df['amount']
    
    # Generate the alpha factor
    alpha_factor = (price_diff / avg_price) * vol_to_amt_ratio
    
    return alpha_factor
```
