```python
import pandas as pd

def heuristics_v2(df):
    # Compute True Range
    df['TrueRange'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['close'].shift(1) - x['low'])), 
        axis=1
    )
    
    # Compute Intraday Price Movement
    df['HighLowDiff'] = df['high'] - df['low']
    df['CloseOpenDiff'] = df['close'] - df['open']
    
    # Adjust for Volume Impact on Price
    df['VolAdjTrueRange'] = df['TrueRange'] * df['volume']
    df['VolAdjCloseOpenDiff'] = df['CloseOpenDiff'] * df['volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['EMA_HighLowDiff_20'] = df['HighLowDiff'].ewm(span=20, adjust=False).mean()
    df['EMA_CloseOpenDiff_20'] = df['CloseOpenDiff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['CombinedMomentum'] = (df['EMA_HighLowDiff_20'] - df['EMA_CloseOpenDiff_20']) / df['close'].std()
    
    # Evaluate Volume Activity
    df['EMA_Volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['HighVolumeDay'] = (df['volume'] > df['EMA_Volume_20']).astype(int)
    
    # Generate Final Alpha Factor
    df['WeightedAvg'] = (df['VolAdjTrueRange'] + df['VolAdjCloseOpenDiff']) / 2
    df['VolumeTrend'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['CombinedMeasure'] = (df['CombinedMomentum'] * df['VolumeTrend'])
    
    # Dynamic Momentum Indicator
    df['AlphaFactor_yesterday'] = df['CombinedMeasure'].shift(1)
    df['AlphaFactor_trend'] = (df['CombinedMeasure'] - df['AlphaFactor_yesterday']) / df['close'].std()
    df['AlphaFactor_smooth'] = df['AlphaFactor_trend'].ewm(span=20, adjust=False).mean()
    
    # Adjust for High Volume Days
    df['FinalAlphaFactor'] = df['AlphaFactor_smooth'] + (df['CombinedMomentum'] * df['CloseOpenDiff']) * df['HighVolumeDay']
    
    return df['FinalAlphaFactor']
```
