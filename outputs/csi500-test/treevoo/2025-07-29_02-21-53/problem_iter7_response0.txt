```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Combine Measures with a weighted average
    # Assign Weights (example: 0.5 for each, can be adjusted)
    weight_high_low = 0.5
    weight_close_open = 0.5
    combined_measures = (weight_high_low * high_low_spread + 
                         weight_close_open * close_open_change)
    
    # Multiply by Volume
    weighted_combined_measures = combined_measures * df['volume']
    
    # Apply 14-day Exponential Moving Average (EMA)
    ema_14 = weighted_combined_measures.ewm(span=14, adjust=False).mean()
    
    # Compute Integrated Momentum Indicator
    integrated_momentum = ema_14 - ema_14.shift(14)
    
    return integrated_momentum
```
