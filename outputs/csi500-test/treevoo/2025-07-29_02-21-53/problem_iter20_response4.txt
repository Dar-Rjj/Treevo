```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Close-Open Change
    df['Close_Open_Change'] = df['close'] - df['open']
    
    # Sum of High-Low Ranges over a Period (e.g., 20 days)
    df['Sum_High_Low_Range'] = df['High_Low_Range'].rolling(window=20).sum()
    
    # Sum of Close-Open Changes over a Period (e.g., 20 days)
    df['Sum_Close_Open_Change'] = df['Close_Open_Change'].rolling(window=20).sum()
    
    # Smooth High-Low Range with Exponential Moving Average (e.g., 20-day)
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum of High-Low Ranges (e.g., 20-day EMA - 50-day EMA)
    df['Momentum_High_Low_Range'] = df['High_Low_Range'].ewm(span=20, adjust=False).mean() - df['High_Low_Range'].ewm(span=50, adjust=False).mean()
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'close']].shift(1).max(axis=1) - df[['high', 'low', 'close']].shift(1).min(axis=1)
    
    # Calculate Rolling Average True Range (e.g., 20-day sum)
    df['Average_True_Range'] = df['True_Range'].rolling(window=20).sum()
    
    # Calculate Integrated Momentum Score
    df['Price_Volatility'] = df['close'].rolling(window=20).std()
    df['Weighted_Price_Volatility'] = df['Price_Volatility'] / df['Price_Volatility'].rolling(window=20).max()
    df['Integrated_Momentum_Score'] = (df['Sum_High_Low_Range'] + df['Sum_Close_Open_Change']) * df['Weighted_Price_Volatility']
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    
    # Combine Volume-Adjusted High-Low with Open-Close
    df['Combined_Measure'] = (df['Volume_Adjusted_High_Low_Range'] * 0.6 + df['Close_Open_Change'] * 0.4) / df['volume']
    
    # Dynamic Momentum Indicator
    df['Yesterday_Combined_Measure'] = df['Combined_Measure'].shift(1)
    df['Momentum_Change'] = df['Combined_Measure'] - df['Yesterday_Combined_Measure']
    df['Dynamic_Momentum_Indicator'] = df['Momentum_Change'] / df['Price_Volatility']
    
    # Evaluate Volume Activity
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['High_Volume_Day'] = df['volume'] > df['Average_Volume'] * 1.5
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Integrate Combined Factors
    df['Integrated_Factor'] = df['Integrated_Momentum_Score'] * df['Volume_Ratio']
    
    # Generate Final Alpha Factor
    df['Alpha_Factor'] = df['Integrated_Factor'] + df['Dynamic_Momentum_Indicator']
    
    return df['Alpha_Factor']
```
