```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])),
        axis=1
    )
    
    # Calculate Rolling Average True Range (ATR) over 14 days
    n_days = 14
    df['atr'] = df['true_range'].rolling(window=n_days).mean()
    
    # Calculate Short-Term and Long-Term Price Averages
    df['short_term_avg'] = df['close'].rolling(window=10).mean()
    df['long_term_avg'] = df['close'].rolling(window=30).mean()
    
    # Determine Price Momentum
    df['price_momentum'] = df['short_term_avg'] - df['long_term_avg']
    
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    period = 5
    df['agg_high_low'] = df['high_low_range'].rolling(window=period).sum()
    df['agg_close_open'] = df['close_open_change'].rolling(window=period).sum()
    
    # Combine ATR and Price Momentum
    df['normalized_momentum'] = df['price_momentum'] / df['atr']
    
    # Adjust Momentum Based on Intraday Movements
    df['weighted_momentum'] = df['normalized_momentum'] * 0.7 + \
                              (df['high_low_range'] / df['close']) * 0.2 + \
                              (df['close_open_change'] / df['close']) * 0.1
    
    # Evaluate Volume Activity
    avg_volume_period = 20
    df['avg_volume'] = df['volume'].rolling(window=avg_volume_period).mean()
    df['high_volume'] = df['volume'] > df['avg_volume'] * 1.5
    
    # Detect Volume Spike
    df['volume_spike'] = df['volume'] > 2 * df['volume'].rolling(window=5).mean()
    
    # Adjust Momentum Based on Volume
    df['final_momentum'] = df['weighted_momentum'].where(~df['volume_spike'], df['weighted_momentum'] * 0.8)
    
    # Integrate Combined Factors
    df['alpha_factor'] = df['final_momentum'] * df['high_volume'].astype(int)
    
    # Incorporate Recent Price Trend
    trend_period = 10
    df['recent_trend'] = df['close'].rolling(window=trend_period).mean() - df['close'].shift(trend_period)
    df['trend_multiplier'] = df['recent_trend'].apply(lambda x: 1.2 if x > 0 else 0.8)
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['alpha_factor'] * df['trend_multiplier']
    
    return df['final_alpha_factor'].dropna()
```
