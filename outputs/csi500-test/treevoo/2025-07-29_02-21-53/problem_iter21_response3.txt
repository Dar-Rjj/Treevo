```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average of open and close price
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the money flow by multiplying the average price with volume
    money_flow = avg_price * df['volume']
    
    # Calculate the weighted money flow by dividing the money flow by the daily range
    # This gives a measure of money flow adjusted for the volatility of the day
    weighted_money_flow = money_flow / daily_range
    
    # Compute the factor value as the moving average of the weighted money flow over the last 5 days
    factor_value = weighted_money_flow.rolling(window=5).mean()
    
    return factor_value
```
