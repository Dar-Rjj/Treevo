```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['TrueRange'] = df[['High', 'Low']].sub(df['Close'].shift(1), axis=0).abs().max(axis=1)
    
    # Compute Intraday Price Movement using High, Low, Open, and Close
    df['IntradayPriceMovement'] = (df['High'] - df['Low']) + (df['Close'] - df['Open'])
    
    # Adjust for Volume Impact on Price
    df['VolumeAdjustedTrueRange'] = df['TrueRange'] * df['Volume']
    df['VolumeAdjustedIntradayPriceMovement'] = df['IntradayPriceMovement'] * df['Volume']
    
    # Apply Exponential Moving Average (20-day) to Sum of Daily High-Low Differences
    df['EMA_HighLowRange'] = df['High'].sub(df['Low']).ewm(span=20, min_periods=20).mean()
    
    # Apply Exponential Moving Average (20-day) to Sum of Daily Close-Open Differences
    df['EMA_CloseOpenDifference'] = (df['Close'] - df['Open']).ewm(span=20, min_periods=20).mean()
    
    # Subtract Short-Term from Long-Term Smoothed Ranges
    short_term_ema_highlow = df['High'].sub(df['Low']).ewm(span=5, min_periods=5).mean()
    df['Momentum_HighLowRange'] = df['EMA_HighLowRange'] - short_term_ema_highlow
    
    # Weight by Recent Price Volatility (ATR over 14 days)
    df['ATR'] = df['TrueRange'].rolling(window=14, min_periods=14).mean()
    df['WeightedMomentum_HighLowRange'] = df['Momentum_HighLowRange'] / df['ATR']
    
    # Calculate Exponential Moving Average of Volume
    df['EMA_Volume'] = df['Volume'].ewm(span=20, min_periods=20).mean()
    
    # Identify High Volume Days
    df['HighVolumeIndicator'] = (df['Volume'] > df['EMA_Volume']).astype(int)
    
    # Combine Volume-Adjusted Measures
    df['CombinedMeasure'] = (df['VolumeAdjustedTrueRange'] + df['VolumeAdjustedIntradayPriceMovement']) / 2
    
    # Assess Trend Continuity and Reversal Potential
    df['TrendContinuity'] = df['CombinedMeasure'] * df['WeightedMomentum_HighLowRange']
    
    # Multiply Combined Measures by High Volume Indicator
    df['FinalFactor'] = df['TrendContinuity'] * df['HighVolumeIndicator']
    
    # Add Interaction Term (High-Low Range Momentum * Close-Open Difference)
    df['InteractionTerm'] = df['Momentum_HighLowRange'] * df['EMA_CloseOpenDifference']
    df['FinalFactor'] += df['InteractionTerm']
    
    return df['FinalFactor']
```
