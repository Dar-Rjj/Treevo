```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price in the range of high and low
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the normalized money flow by dividing the amount by volume
    df['money_flow'] = df['amount'] / df['volume']
    
    # Create a factor that combines the two signals
    # The idea is to favor stocks that are closing near the high and have strong money flow
    df['factor'] = df['close_position'] * df['money_flow']
    
    # Return the factor as a pandas Series
    return df['factor']
```
