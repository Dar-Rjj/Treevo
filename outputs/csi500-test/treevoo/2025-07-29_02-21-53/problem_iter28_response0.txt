import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day, 10-Day, and 20-Day Simple Moving Averages (SMA) of Close Prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Compute the difference between 10-Day SMA and 20-Day SMA as a Momentum Indicator
    df['Momentum_SMA'] = df['SMA_10'] - df['SMA_20']
    
    # Identify Days Where Close Price Exceeds Highest High or Lowest Low over Last 30 Days
    df['High_30'] = df['high'].rolling(window=30).max()
    df['Low_30'] = df['low'].rolling(window=30).min()
    df['Breakout_Up'] = (df['close'] > df['High_30']).astype(int)
    df['Breakout_Down'] = (df['close'] < df['Low_30']).astype(int)
    
    # Assign a Score Based on Number of Consecutive Days of Breakout
    df['Consecutive_Breakout_Up'] = df['Breakout_Up'].groupby((df['Breakout_Up'] != df['Breakout_Up'].shift()).cumsum()).cumsum()
    df['Consecutive_Breakout_Down'] = df['Breakout_Down'].groupby((df['Breakout_Down'] != df['Breakout_Down'].shift()).cumsum()).cumsum()
    
    # Calculate the Percentage Change in Volume from Previous Day
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Mark Days with Large Volume Increase (>10%) and Associated Price Movement (>2%)
    df['Large_Volume_Price_Movement'] = ((df['Volume_Change'] > 0.1) & (df['close'].pct_change() > 0.02)).astype(int)
    
    # Determine if the Stock Opened Higher but Closed Lower (Bearish Signal)
    df['Bearish_Signal'] = ((df['open'] > df['close']) & (df['open'].shift() < df['close'].shift())).astype(int)
    
    # Determine if the Stock Opened Lower but Closed Higher (Bullish Signal)
    df['Bullish_Signal'] = ((df['open'] < df['close']) & (df['open'].shift() > df['close'].shift())).astype(int)
    
    # Calculate the Ratio of Amount to Volume (Price Weighted Volume)
    df['Amount_to_Volume_Ratio'] = df['amount'] / df['volume']
    
    # Measure the Difference Between Today’s Open and Yesterday’s Close
    df['Gap_Open'] = df['open'] - df['close'].shift()
    
    # Categorize Gaps as Up-Gap (Open > Yesterday's Close) or Down-Gap (Open < Yesterday's Close)
    df['Up_Gap'] = (df['Gap_Open'] > 0).astype(int)
    df['Down_Gap'] = (df['Gap_Open'] < 0).astype(int)
    
    # Compute the Range (High - Low) for Each Day
    df['Daily_Range'] = df['high'] - df['low']
    
    # Track Moving Average of Daily Ranges to Assess Volatility Trends
    df['MA_Range_20'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Compute the Volume Weighted Price (VWP) for Each Day
    df['VWP'] = (df['amount'] / df['volume'])
    
    # Compare VWP with the Simple Moving Average (SMA) to Identify Divergences
    df['VWP_vs_SMA_20'] = df['VWP'] - df['SMA_20']
    
    # Calculate the Standard Deviation of Daily Returns Over a 20-Day Period
    df['Daily_Returns'] = df['close'].pct_change()
    df['Std_Dev_20'] = df['Daily_Returns'].rolling(window=20).std()
