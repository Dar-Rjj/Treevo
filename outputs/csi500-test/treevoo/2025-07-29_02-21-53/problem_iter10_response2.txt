```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Multiply by Sign of High-Low Spread
    sign_high_low = (df['high'] - df['low']).apply(lambda x: 1 if x > 0 else -1)
    df['weighted_high_low_diff'] = df['high_low_diff'] * sign_high_low
    df['weighted_close_open_diff'] = df['close_open_diff'] * sign_high_low
    
    # Sum of High-Low and Close-Open Differences over a Period (e.g., 5 days)
    high_low_sum = df['weighted_high_low_diff'].rolling(window=5).sum()
    close_open_sum = df['weighted_close_open_diff'].rolling(window=5).sum()
    
    # Sum Positive and Negative Signed Volumes
    df['positive_volume'] = df['volume'].where(df['volume'] > 0, 0)
    df['negative_volume'] = df['volume'].where(df['volume'] < 0, 0)
    pos_vol_sum = df['positive_volume'].rolling(window=5).sum()
    neg_vol_sum = df['negative_volume'].rolling(window=5).sum()
    
    # Calculate Standard Deviation of Close Prices
    std_close = df['close'].rolling(window=5).std()
    
    # Weight by Recent Price Volatility
    volatility_weight = std_close / std_close.rolling(window=5).mean()
    
    # Divide Positive by Total Absolute Volumes
    total_abs_volume = df['volume'].abs().rolling(window=5).sum()
    pos_ratio = pos_vol_sum / total_abs_volume
    neg_ratio = neg_vol_sum / total_abs_volume
    
    # Integrated Momentum Score
    momentum_score = (high_low_sum + close_open_sum) * volatility_weight * (pos_ratio - neg_ratio)
    
    # Average Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Identify Days with High Volume
    high_volume = df['volume'] > avg_volume * 1.5
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].diff(axis=1).max(axis=1).fillna(0)
    df['true_range'] = df.apply(lambda row: max(row['true_range'], abs(row['high'] - row['prev_close']), abs(row['low'] - row['prev_close'])), axis=1)
    
    # Rolling Average True Range
    atr = df['true_range'].rolling(window=5).mean()
    
    # Volume Ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Normalize Price Momentum by ATR
    normalized_momentum = momentum_score / atr
    
    # Integrate Order Imbalance Score
    order_imbalance_score = pos_ratio - neg_ratio
    
    # Combine Price, Volume, and Order Imbalance Effects
    integrated_factor = normalized_momentum * volume_ratio * order_imbalance_score
    
    return integrated_factor
```
