import pandas as pd

def heuristics_v2(df):
    # Calculate Signed Transaction Volume
    signed_volume = df['volume'] * ((df['high'] - df['low']) > 0).astype(int) * 2 - 1
    
    # Calculate Accumulated Buy-Sell Imbalance
    positive_signed_volume = signed_volume[signed_volume > 0].sum()
    negative_signed_volume = signed_volume[signed_volume < 0].sum()
    total_absolute_volume = abs(signed_volume).sum()
    
    # Compute Order Imbalance Score
    positive_ratio = positive_signed_volume / total_absolute_volume
    negative_ratio = abs(negative_signed_volume) / total_absolute_volume
    order_imbalance_score = positive_ratio - negative_ratio
    
    return pd.Series(order_imbalance_score, index=df.index, name='order_imbalance_factor')
