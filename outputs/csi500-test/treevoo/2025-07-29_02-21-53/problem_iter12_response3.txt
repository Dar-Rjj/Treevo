```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].max(axis=1) - df[['high', 'low', 'prev_close']].min(axis=1)
    
    # Adjust for Volume Impact on Price
    df['vol_adjusted_tr'] = df['true_range'] * np.sqrt(df['volume'])
    
    # Calculate Open-Close Momentum
    df['oc_momentum'] = (df['close'] - df['open']) / df['open']
    
    # Aggregate Price Movements
    window = 5  # Example rolling window size
    df['sum_high_low'] = df['high'] - df['low']
    df['sum_high_low_rolling'] = df['sum_high_low'].rolling(window=window).sum()
    df['sum_close_open'] = df['close'] - df['open']
    df['sum_close_open_rolling'] = df['sum_close_open'].rolling(window=window).sum()
    
    # Combine Volume-Adjusted High-Low with Open-Close
    df['historical_volatility'] = df['close'].pct_change().rolling(window=window).std() * np.sqrt(252)
    combined = (df['sum_high_low_rolling'] + df['sum_close_open_rolling']) / 2
    df['weighted_avg'] = combined * df['historical_volatility']
    
    # Analyze Volume Trend Over Time
    df['volume_change_rate'] = df['volume'].pct_change()
    df['adjusted_combined'] = df['weighted_avg'] * df['volume_change_rate']
    
    # Dynamic Momentum Indicator
    df['yesterday_alpha'] = df['adjusted_combined'].shift(1)
    df['today_alpha'] = df['adjusted_combined']
    df['dynamic_momentum'] = df['yesterday_alpha'] + df['today_alpha']
    df['smoothed_momentum'] = df['dynamic_momentum'].ewm(span=window, adjust=False).mean()
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['high_volume'] = df['volume'] > df['avg_volume']
    
    # Generate Final Alpha Factor
    df['final_alpha'] = df['smoothed_momentum'] * df['high_volume']
    
    return df['final_alpha']
```
