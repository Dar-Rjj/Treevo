```python
import pandas as pd

def heuristics_v2(df, price_momentum_window=10, volume_ma_window=30, high_volume_threshold=2.0, atr_window=14):
    # Calculate Intraday Price Movement
    df['HighLowDiff'] = df['high'] - df['low']
    df['CloseOpenDiff'] = df['close'] - df['open']

    # Aggregate Price Movements
    df['SumHighLow'] = df['HighLowDiff'].rolling(window=price_momentum_window).sum()
    df['SumCloseOpen'] = df['CloseOpenDiff'].rolling(window=price_momentum_window).sum()

    # Calculate Price Momentum Score
    df['PriceMomentumScore'] = df['SumHighLow'] + df['SumCloseOpen']
    df['CloseStdDev'] = df['close'].rolling(window=price_momentum_window).std()
    df['PriceMomentumScore'] *= df['CloseStdDev']

    # Evaluate Volume Activity
    df['AvgVolume'] = df['volume'].rolling(window=volume_ma_window).mean()
    df['HighVolume'] = (df['volume'] > df['AvgVolume'] * high_volume_threshold).astype(int)

    # Integrate Price Momentum and Volume Activity
    df['IntermediateAlphaFactor'] = df['PriceMomentumScore'] * df['HighVolume']

    # Calculate True Range
    df['TrueRange'] = df[['high', 'low', 'close']].join(df['close'].shift(1), rsuffix='_prev').apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close_prev']), abs(x['low'] - x['close_prev'])), axis=1)

    # Calculate Rolling Average True Range
    df['ATR'] = df['TrueRange'].rolling(window=atr_window).mean()

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(price_momentum_window)

    # Combine ATR, Price, and Volume Effects
    df['FinalAlphaFactor'] = (df['PriceMomentum'] / df['ATR']) * df['IntermediateAlphaFactor']

    return df['FinalAlphaFactor'].dropna()
```
