```python
import pandas as pd

def heuristics_v2(df, n_days=10, vol_threshold=1.5):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    df['agg_high_low_range'] = df['high_low_range'].rolling(window=n_days).sum()
    df['agg_close_open_change'] = df['close_open_change'].rolling(window=n_days).sum()
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['prev_close']).abs()
    df['tr3'] = (df['prev_close'] - df['low']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Rolling Average True Range
    df['avg_true_range'] = df['true_range'].rolling(window=n_days).mean()
    
    # Calculate Price Momentum Score
    price_momentum_score = 0.6 * df['agg_high_low_range'] + 0.4 * df['agg_close_open_change']
    close_std = df['close'].rolling(window=n_days).std()
    price_momentum_score = price_momentum_score * (close_std + 1)  # Adjust by recent price volatility
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    df['high_vol_indicator'] = (df['volume'] > vol_threshold * df['avg_volume']).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    final_factor = price_momentum_score * df['high_vol_indicator']
    
    # Combine ATR and Price Momentum
    final_factor = (final_factor / df['avg_true_range']).fillna(0)
    
    return final_factor
```
