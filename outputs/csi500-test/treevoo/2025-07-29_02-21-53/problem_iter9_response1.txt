import pandas as pd

def heuristics_v2(df, short_window=20, long_window=50, volume_threshold=1.5):
    # Compute High-Low Difference
    df['High-Low'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['Close-Open'] = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period (short and long)
    df['Sum_High-Low_Short'] = df['High-Low'].rolling(window=short_window).sum()
    df['Sum_High-Low_Long'] = df['High-Low'].rolling(window=long_window).sum()
    
    # Sum of Close-Open Differences over a Period (short and long)
    df['Sum_Close-Open_Short'] = df['Close-Open'].rolling(window=short_window).sum()
    df['Sum_Close-Open_Long'] = df['Close-Open'].rolling(window=long_window).sum()
    
    # Smooth High-Low Range
    df['EMA_High-Low_Short'] = df['High-Low'].ewm(span=short_window, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range
    df['Max_High_Long'] = df['high'].rolling(window=long_window).max()
    df['Min_Low_Long'] = df['low'].rolling(window=long_window).min()
    df['Long_Term_High-Low_Range'] = df['Max_High_Long'] - df['Min_Low_Long']
    
    # Combined Price and High-Low Range Momentum
    df['Momentum'] = (df['Sum_High-Low_Short'] + df['Sum_Close-Open_Short']) - (df['EMA_High-Low_Short'] - df['Long_Term_High-Low_Range'])
    
    # Weight by Recent Price Volatility
    df['Volatility'] = df['close'].rolling(window=short_window).std()
    df['Weighted_Momentum'] = df['Momentum'] * df['Volatility']
    
    # Evaluate Volume Activity
    df['Average_Volume'] = df['volume'].ewm(span=short_window, adjust=False).mean()
    df['Volume_Indicator'] = (df['volume'] > df['Average_Volume'] * volume_threshold).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Integrated_Momentum'] = df['Weighted_Momentum'] * df['Volume_Indicator']
    
    # Include Interaction Term
    df['Interaction_Term'] = df['Long_Term_High-Low_Range'] * df['Sum_Close-Open_Short']
    df['Final_Alpha_Factor'] = df['Integrated_Momentum'] + df['Interaction_Term']
    
    return df['Final_Alpha_Factor']
