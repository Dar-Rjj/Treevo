```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Volume Impact on Price
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Calculate Open-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Assign Weights Based on Historical Volatility
    # For simplicity, we will use a fixed weight. In practice, this could be based on historical volatility.
    weight_volume_adjusted = 0.6
    weight_open_close = 0.4
    
    # Calculate Averages
    combined_measure = (weight_volume_adjusted * volume_adjusted_high_low) + (weight_open_close * open_close_momentum)
    
    # Analyze Volume Change Over Time
    volume_change = df['volume'].pct_change()
    
    # Adjust Combined Measure Based on Volume Trend
    combined_measure = combined_measure * (1 + volume_change)
    
    # Incorporate Previous Day's Indicator
    df['previous_indicator'] = combined_measure.shift(1)
    
    # Use Today's Intraday and Volume Synchronized Oscillator
    today_indicator = combined_measure
    
    # Determine Direction of Momentum Change
    momentum_change_direction = today_indicator - df['previous_indicator']
    
    # Factor in Volatility for Smoothing
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    smoothed_momentum_change_direction = momentum_change_direction / volatility
    
    # Final Dynamic Momentum Indicator
    dynamic_momentum_indicator = smoothed_momentum_change_direction
    
    return dynamic_momentum_indicator
```
