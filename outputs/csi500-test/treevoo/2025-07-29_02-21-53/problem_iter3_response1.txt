```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = df.abs(df['high'] - df['previous_close'])
    df['prev_close_low'] = df.abs(df['previous_close'] - df['low'])
    df['true_range'] = df[['high_low', 'high_prev_close', 'prev_close_low']].max(axis=1)
    
    # Define a rolling window
    n_days = 14
    
    # Apply weights to each true range (e.g., Exponential Decay)
    weights = [0.5**i for i in range(n_days)]
    df['weighted_atr'] = df['true_range'].rolling(window=n_days).apply(lambda x: (x * weights[:len(x)]).sum(), raw=True)
    
    # Calculate Price Momentum
    n_momentum_days = 10
    df['price_momentum'] = df['close'] - df['close'].shift(n_momentum_days)
    
    # Combine Weighted ATR and Price Momentum
    df['inverse_weighted_atr'] = 1 / df['weighted_atr']
    df['factor'] = df['price_momentum'] * df['inverse_weighted_atr']
    
    return df['factor']
```
