```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period (e.g., 20 days)
    df['HL_Sum_20'] = df['High_Low_Diff'].rolling(window=20).sum()
    
    # Sum of Close-Open Differences over a Period (e.g., 20 days)
    df['CO_Sum_20'] = df['Close_Open_Diff'].rolling(window=20).sum()
    
    # Smooth High-Low Range (e.g., 20-day Moving Average)
    df['HL_Smooth_20'] = df['High_Low_Diff'].rolling(window=20).mean()
    
    # Long-Term High-Low Range (e.g., 50-day)
    df['HL_Long_50'] = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Combined_Momentum'] = df['CO_Sum_20'] + (df['HL_Long_50'] - df['HL_Smooth_20'])
    
    # Weight by Recent Price Volatility (e.g., 20-day Standard Deviation)
    df['Close_Volatility_20'] = df['close'].rolling(window=20).std()
    df['Combined_Momentum_Weighted'] = df['Combined_Momentum'] * (df['Close_Volatility_20'] / df['Close_Volatility_20'].mean())
    
    # Compute Average Volume (e.g., 20-day Moving Average)
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Identify Days with High Volume
    high_volume_threshold = 1.5
    df['High_Volume'] = (df['volume'] > high_volume_threshold * df['Avg_Volume_20']).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Alpha_Factor'] = df['Combined_Momentum_Weighted'] * df['High_Volume']
    
    # Incorporate Volume Spike Impact
    volume_spike_threshold = 2.0
    df['Volume_Spike'] = (df['volume'] > volume_spike_threshold * df['Avg_Volume_20']).astype(int)
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + df['Volume_Spike'])
    
    # Volume Weighted Average Price
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['Simple_Avg_Price_20'] = df['close'].rolling(window=20).mean()
    
    # Adjust Alpha Factor Based on VWAP Comparison
    df['Alpha_Factor'] = df['Alpha_Factor'] * ((df['VWAP'] - df['Simple_Avg_Price_20']) / df['Simple_Avg_Price_20'])
    
    return df['Alpha_Factor']
```
