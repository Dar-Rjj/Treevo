def heuristics_v2(data):
    # Calculate the percentage change in close price
    data['close_change'] = data['close'].pct_change()
    
    # Calculate the average of high and low prices
    data['avg_high_low'] = (data['high'] + data['low']) / 2
    
    # Calculate the relative strength index (RSI) based on close price
    delta = data['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    data['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the ratio of volume to amount
    data['vol_to_amt_ratio'] = data['volume'] / data['amount']
    
    # Combine the factors into a single alpha factor
    data['alpha_factor'] = (data['close_change'] + data['avg_high_low'] + data['rsi'] + data['vol_to_amt_ratio']) / 4
    
    return data['alpha_factor']
