```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['prev_close']).abs()
    df['tr3'] = (df['prev_close'] - df['low']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Adjust for Volume Impact on Price
    df['volume_adjusted_tr'] = df['true_range'] * df['volume']

    # Calculate Open-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']

    # Aggregate Price Movements
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['price_movement_sum'] = df['high_low_diff'] + df['close_open_diff']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['combined_measure'] = 0.5 * df['volume_adjusted_tr'] + 0.5 * df['open_close_momentum']

    # Dynamic Momentum Indicator
    df['yesterday_alpha'] = df['combined_measure'].shift(1)
    df['today_alpha'] = df['combined_measure']
    df['dynamic_momentum'] = (df['today_alpha'] + df['yesterday_alpha']) / 2
    df['trend_continuity'] = df['dynamic_momentum'].rolling(window=5).std()

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['high_volume_days'] = (df['volume'] > df['avg_volume'] * 1.5).astype(int)

    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['combined_measure'] * df['high_volume_days']

    return df['final_alpha_factor']
```
