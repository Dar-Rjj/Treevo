```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the ratio of price change to the average true range
    df['avg_true_range'] = df[['high', 'low', 'close']].rolling(window=14).apply(lambda x: max(x) - min(x), raw=True)
    df['price_change_to_atr_ratio'] = df['price_change'] / df['avg_true_range']
    
    # Calculate the ratio of volume change to the average volume
    df['avg_volume'] = df['volume'].rolling(window=14).mean()
    df['volume_change_to_avg_volume_ratio'] = df['volume_change'] / df['avg_volume']
    
    # Heuristic factor: price momentum with volume confirmation
    df['heuristic_factor'] = (df['price_change_to_atr_ratio'] * df['volume_change_to_avg_volume_ratio'])
    
    return df['heuristic_factor'].dropna()
```
