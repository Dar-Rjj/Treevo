```python
import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['True Range'] = df[['high', 'low', 'Previous Close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                                 x['high'] - x['Previous Close'], 
                                                                                 x['Previous Close'] - x['low']), axis=1)
    
    # Adjust for Volume Impact on Price
    df['Volume Adjusted True Range'] = df['True Range'] * np.sqrt(df['volume'])
    
    # Smooth True Range
    df['Short-Term Smoothed True Range'] = df['Volume Adjusted True Range'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term True Range
    df['Long-Term Smoothed True Range'] = df['Volume Adjusted True Range'].ewm(span=50, adjust=False).mean()
    
    # Compute Momentum of True Ranges
    df['Momentum of True Ranges'] = df['Long-Term Smoothed True Range'] - df['Short-Term Smoothed True Range']
    
    # Weight by Volume
    df['Volume-Weighted Momentum of True Ranges'] = df['Momentum of True Ranges'] * df['volume']
    
    # Integrate Price Change
    df['Open-Close Momentum'] = (df['close'] - df['open']) / df['open']
    df['Smoothed Open-Close Momentum'] = df['Open-Close Momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine with Momentum of True Ranges
    df['Combined Factor 1'] = df['Volume-Weighted Momentum of True Ranges'] + df['Smoothed Open-Close Momentum']
    
    # Integrate Price Trend
    df['Long-Term Price Trend'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Price Trend Momentum'] = df['Long-Term Price Trend'] - df['Smoothed Open-Close Momentum']
    
    # Combine with True Range and Volume Momentum
    df['Combined Factor 2'] = df['Combined Factor 1'] + df['Price Trend Momentum']
    
    # Analyze Volume Trend Over Time
    df['Volume Change Rate'] = df['volume'].pct_change()
    df['Adjusted Combined Factor'] = df['Combined Factor 2'] * (1 + df['Volume Change Rate'])
    
    # Generate Final Alpha Factor
    df['Yesterday Alpha Factor'] = df['Adjusted Combined Factor'].shift(1)
    df['Trend Continuity'] = df[['Adjusted Combined Factor', 'Yesterday Alpha Factor']].apply(lambda x: linregress([1, 2], [x[0], x[1]])[0], axis=1)
    
    # Multiply Combined True Range, Open-Close Momentum, and Volume-Adjusted True Range with High Volume Indicator
    df['Final Alpha Factor'] = df['Trend Continuity'] * (df['volume'] > df['volume'].quantile(0.8))
    
    return df['Final Alpha Factor']
```
