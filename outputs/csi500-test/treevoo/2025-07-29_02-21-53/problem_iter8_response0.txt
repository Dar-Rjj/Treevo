```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Rolling Average True Range (ATR)
    df['PrevClose'] = df['close'].shift(1)
    df['TrueRange'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['HighLowDiff'] = df['high'] - df['low']
    df['HighPCDiff'] = df['high'] - df['PrevClose']
    df['PCLowDiff'] = df['PrevClose'] - df['low']
    df['TrueRange'] = df[['HighLowDiff', 'HighPCDiff', 'PCLowDiff']].max(axis=1)
    n = 14
    df['ATR'] = df['TrueRange'].rolling(window=n).mean()

    # Calculate Intraday Price Movement and Signed Transaction Volume
    df['HighLowDiff'] = df['high'] - df['low']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['IntradayMovement'] = df['HighLowDiff'] * (df['high'] - df['low']).apply(lambda x: 1 if x > 0 else -1)

    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    period = 20
    df['SumHighLow'] = df['HighLowDiff'].rolling(window=period).sum()
    df['SumCloseOpen'] = df['CloseOpenDiff'].rolling(window=period).sum()
    df['PositiveVolume'] = df['volume'] * (df['CloseOpenDiff'] > 0).astype(int)
    df['NegativeVolume'] = df['volume'] * (df['CloseOpenDiff'] < 0).astype(int)
    df['SumPositiveVolume'] = df['PositiveVolume'].rolling(window=period).sum()
    df['SumNegativeVolume'] = df['NegativeVolume'].rolling(window=period).sum()

    # Calculate Integrated Momentum Score
    df['StdDevClose'] = df['close'].rolling(window=period).std()
    df['IntegratedMomentum'] = df['SumHighLow'] * df['StdDevClose'] + df['SumCloseOpen'] * df['StdDevClose']
    df['TotalVolume'] = df['volume'].rolling(window=period).sum()
    df['IntegratedMomentum'] += (df['SumPositiveVolume'] / df['TotalVolume']) - (df['SumNegativeVolume'] / df['TotalVolume'])

    # Adjust for Volume Impact on Price
    df['VolumeAdjustedHighLow'] = df['HighLowDiff'] * df['volume']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['CombinedMeasure'] = (df['VolumeAdjustedHighLow'] + df['CloseOpenDiff']) / 2

    # Dynamic Momentum Indicator
    df['PrevDayIndicator'] = df['IntegratedMomentum'].shift(1)
    df['TodaysIndicator'] = df['IntegratedMomentum']
    df['MomentumChange'] = df['TodaysIndicator'] - df['PrevDayIndicator']
    df['MomentumChangeSmoothed'] = df['MomentumChange'] / df['StdDevClose']

    # Evaluate Volume Activity
    df['AverageVolume'] = df['volume'].rolling(window=period).mean()
    df['HighVolume'] = (df['volume'] > df['AverageVolume']).astype(int)

    # Integrate Combined Factors
    df['VolumeRatio'] = df['volume'] / df['AverageVolume']
    df['NormalizedPriceMomentum'] = df['IntegratedMomentum'] / df['ATR']
    df['AlphaFactor'] = df['NormalizedPriceMomentum'] * df['VolumeRatio'] * df['MomentumChangeSmoothed']

    return df['AlphaFactor']
```
