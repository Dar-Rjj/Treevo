```python
def heuristics_v2(data):
    # Calculate Intraday Price Movement
    high_low_diff = data['high'] - data['low']
    close_open_diff = data['close'] - data['open']

    # Aggregate Price Movements
    sum_high_low = high_low_diff.rolling(window=5).sum()
    sum_close_open = close_open_diff.rolling(window=5).sum()

    # Calculate Price Momentum Score
    momentum_score = (sum_high_low + sum_close_open) / 2
    recent_volatility = data['close'].rolling(window=5).std()
    weighted_momentum = momentum_score * recent_volatility

    # Evaluate Volume Activity
    avg_volume = data['volume'].rolling(window=20).mean()
    high_volume = (data['volume'] > avg_volume * 1.5).astype(int)

    # Integrate Price Momentum and Volume Activity
    alpha_factor = weighted_momentum * high_volume

    return alpha_factor
```
