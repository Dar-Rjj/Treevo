import pandas as pd

def heuristics_v2(data):
    # Momentum Indicator Development
    def sma(data, window):
        return data.rolling(window=window).mean()

    def ema(data, span):
        return data.ewm(span=span, adjust=False).mean()

    short_sma = sma(data['close'], 5)
    long_sma = sma(data['close'], 20)
    short_ema = ema(data['close'], 9)
    long_ema = ema(data['close'], 18)

    sma_crossover = short_sma - long_sma
    ema_crossover = short_ema - long_ema

    # Volume Trend Analysis
    data['volume_change'] = data['volume'].diff()
    positive_volume_change = (data['volume_change'] > 0) * data['volume_change']
    negative_volume_change = (data['volume_change'] < 0) * data['volume_change']

    positive_vol_sum = positive_volume_change.rolling(window=20).sum()
    negative_vol_sum = negative_volume_change.rolling(window=20).sum()
    volume_accumulation_oscillator = positive_vol_sum - negative_vol_sum.abs()

    # Price Action Strength
    data['true_range'] = data[['high', 'low']].apply(lambda x: max(x['high'], x['close'].shift(1)) - min(x['low'], x['close'].shift(1)), axis=1)
    avg_true_range = data['true_range'].rolling(window=14).mean()
    price_strength = data['true_range'] / avg_true_range

    # Market Sentiment Factors
    open_close_spread = data['open'] - data['close']
    daily_high_low_range_percentage = (data['high'] - data['low']) / data['low'] * 100
    combined_sentiment_score = open_close_spread + daily_high_low_range_percentage

    # Hybrid Factor Creation
    sma_adjusted_by_volume = (short_sma - long_sma) * data['volume']
    ema_adjusted_by_volume = (short_ema - long_ema) * data['volume']

    # Synthetic Alpha Factor
    alpha_factor = 0.4 * sma_crossover + 0.3 * volume_accumulation_oscillator + 0.3 * combined_sentiment_score

    return alpha_factor
