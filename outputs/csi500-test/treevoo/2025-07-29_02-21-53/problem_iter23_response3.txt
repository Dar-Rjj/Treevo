```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['true_range'] = df['true_range'].fillna(0)
    
    # Multiply True Range by Volume
    df['volume_adjusted_true_range'] = df['true_range'] * df['volume']
    
    # Apply Exponential Moving Average (20-day) to Daily High-Low Differences
    df['smoothed_high_low_range'] = df['high'] - df['low']
    df['smoothed_high_low_range'] = df['smoothed_high_low_range'].ewm(span=20).mean()
    
    # Combine Volume-Adjusted True Range and Smoothed High-Low Ranges
    df['combined_measures'] = df['volume_adjusted_true_range'] + df['smoothed_high_low_range']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['combined_measures']
    
    # Calculate 5-day and 20-day Simple Moving Averages of Close Prices
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Compute the Difference between 5-day and 20-day Moving Averages
    df['price_momentum_diff'] = df['sma_5'] - df['sma_20']
    
    # Multiply the Price Momentum Difference by the Volume
    df['volume_adjusted_price_momentum'] = df['price_momentum_diff'] * df['volume']
    
    # Add the Price Momentum Measure to the Final Alpha Factor
    df['final_alpha_factor'] += df['volume_adjusted_price_momentum']
    
    # Add Interaction Term (Smoothed High-Low Range * High-Low Difference)
    df['interaction_term'] = df['smoothed_high_low_range'] * (df['high'] - df['low'])
    df['final_alpha_factor'] += df['interaction_term']
    
    # Ensure the Final Factor is Coherent and Interpretable
    final_alpha_factor = df['final_alpha_factor']
    
    return final_alpha_factor
```
