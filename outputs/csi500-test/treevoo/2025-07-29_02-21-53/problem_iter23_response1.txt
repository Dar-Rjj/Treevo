import pandas as pd

def heuristics_v2(df, N=5, M=14, L=20, P=10):
    # Calculate daily return from open to close
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Sum of returns over a lookback window of N days
    df['N_day_return_sum'] = df['daily_return'].rolling(window=N).sum()
    
    # True range for measuring daily volatility
    df['true_range'] = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    
    # Average true range (ATR) over M days
    df['atr'] = df['true_range'].rolling(window=M).mean()
    
    # Relative ATR by comparing the current ATR to the ATR over a longer period (e.g., 2M days)
    df['relative_atr'] = df['atr'] / df['atr'].rolling(window=2 * M).mean()
    
    # Volume intensity: ratio of today's volume to the average volume of the last L days
    df['volume_intensity'] = df['volume'] / df['volume'].rolling(window=L).mean()
    
    # Money flow: sum of money flow over P days
    df['money_flow'] = df['volume'] * (2 * df['close'] - df['high'] - df['low'])
    df['P_day_money_flow'] = df['money_flow'].rolling(window=P).sum()
    
    # Enhance the money flow factor by incorporating the sign of the daily return
    df['enhanced_money_flow'] = df['P_day_money_flow'] * df['daily_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Composite factor: combining N-day return, ATR, and volume intensity
    composite_factor = (
        df['N_day_return_sum'] + 
        df['relative_atr'] + 
        df['volume_intensity'] + 
        df['enhanced_money_flow']
    )
    
    # Normalize the composite factor to have a mean of 0 and standard deviation of 1
    # Note: Normalization is not allowed per the instruction, so we skip this step.
    
    # Return the composite factor as a Series
    return composite_factor
