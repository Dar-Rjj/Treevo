```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['True Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], x[0] - x['Previous Close'], x['Previous Close'] - x[1]), axis=1)
    
    # Calculate Rolling Average True Range (ATR)
    n_days = 14
    df['ATR'] = df['True Range'].rolling(window=n_days).mean()
    
    # Calculate Short-Term and Long-Term Price Averages
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    df['30_day_avg'] = df['close'].rolling(window=30).mean()
    
    # Determine Price Momentum
    df['Price Momentum'] = df['10_day_avg'] - df['30_day_avg']
    
    # Detect Volume Spike
    df['5_day_vol_avg'] = df['volume'].rolling(window=5).mean()
    df['Volume Spike'] = df['volume'] > 2 * df['5_day_vol_avg']
    
    # Adjust Momentum Based on Volume
    df['Adjusted Momentum'] = df.apply(lambda x: x['Price Momentum'] * 0.8 if x['Volume Spike'] else x['Price Momentum'], axis=1)
    
    # Combine ATR and Price Momentum
    df['Normalized Momentum'] = df['Adjusted Momentum'] / df['ATR']
    
    # Evaluate Volume Activity
    df['Average Volume'] = df['volume'].rolling(window=30).mean()
    df['High Volume'] = df['volume'] > 1.5 * df['Average Volume']
    
    # Integrate Combined Factors
    df['Combined Factor'] = df['Normalized Momentum'] * df['High Volume'].astype(int)
    
    # Calculate Intraday Price Movement
    df['High-Low Range'] = df['high'] - df['low']
    df['Close-Open Change'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    df['Sum High-Low Range'] = df['High-Low Range'].rolling(window=10).sum()
    df['Sum Close-Open Change'] = df['Close-Open Change'].rolling(window=10).sum()
    
    # Calculate Price Momentum Score
    df['Standard Deviation Close'] = df['close'].rolling(window=10).std()
    df['Weighted High-Low Range'] = df['Sum High-Low Range'] / df['Standard Deviation Close']
    df['Weighted Close-Open Change'] = df['Sum Close-Open Change'] / df['Standard Deviation Close']
    df['Price Momentum Score'] = df['Weighted High-Low Range'] + df['Weighted Close-Open Change']
    
    # Combine ATR, Price Momentum, and Intraday Effects
    df['Final Alpha Factor'] = df['Combined Factor'] + df['Price Momentum Score']
    
    return df['Final Alpha Factor']
```
