```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement and High-Low Range
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']

    # Aggregate Price Movements and Smooth High-Low Ranges
    window_short = 10
    window_long = 50
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=window_short).sum()
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=window_short).sum()
    df['smoothed_high_low_range'] = df['high_low_diff'].rolling(window=20).mean()
    df['long_term_high_low_range'] = df['high_low_diff'].rolling(window=window_long).max()

    # Calculate Combined Price and High-Low Range Momentum
    df['combined_momentum'] = (df['sum_high_low_diff'] + df['sum_close_open_diff']) / 2
    df['range_momentum'] = df['long_term_high_low_range'] - df['smoothed_high_low_range']

    # Weight by Recent Price Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['weighted_momentum'] = df['combined_momentum'] * (1 + df['volatility'])

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['high_volume_indicator'] = (df['volume'] > df['avg_volume']).astype(int)

    # Integrate Price, High-Low Range, and Volume Activity
    df['alpha_factor'] = df['weighted_momentum'] * df['high_volume_indicator']

    return df['alpha_factor']
```
