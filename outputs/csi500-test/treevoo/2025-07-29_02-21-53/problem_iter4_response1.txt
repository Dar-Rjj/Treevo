import pandas as pd

def heuristics_v2(df):
    # Calculate the signed transaction volume
    signed_volume = df['volume'] * ((df['high'] - df['low']) > 0).astype(int) * 2 - 1
    
    # Separate positive and negative signed volumes
    positive_signed_volume = signed_volume[signed_volume > 0]
    negative_signed_volume = signed_volume[signed_volume < 0]
    
    # Calculate accumulated buy-sell imbalance
    positive_sum = positive_signed_volume.sum()
    negative_sum = negative_signed_volume.sum()
    
    # Compute order imbalance score
    total_absolute_volumes = abs(signed_volume).sum()
    order_imbalance_score = (positive_sum / total_absolute_volumes) - (negative_sum / total_absolute_volumes)
    
    return pd.Series(order_imbalance_score, index=df.index)
