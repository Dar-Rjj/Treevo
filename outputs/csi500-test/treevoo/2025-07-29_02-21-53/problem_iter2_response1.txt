```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Oscillation
    HL_oscillation = df['high'] - df['low']
    
    # Smooth Oscillation with Exponential Moving Average
    HL_ema = HL_oscillation.ewm(span=14, adjust=False).mean()
    
    # Compute Momentum Indicator from EMA
    momentum_indicator = HL_ema - HL_ema.shift(14)
    
    # Calculate Intraday Price Movement
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Aggregate Price Movements over a Period
    sum_high_low_diff = high_low_diff.rolling(window=14).sum()
    sum_close_open_diff = close_open_diff.rolling(window=14).sum()
    
    # Calculate Price Momentum Score
    price_momentum_score = (sum_high_low_diff + sum_close_open_diff) * df['close'].rolling(window=14).std()
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=14).mean()
    high_volume_indicator = (df['volume'] > avg_volume).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    final_alpha_factor = price_momentum_score * high_volume_indicator
    
    return final_alpha_factor
```
