```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-day and 30-day Close Price Averages
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['30_day_avg_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['10_day_avg_close'] - df['30_day_avg_close']
    
    # Calculate 5-day Volume Average
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Identify Volume Spike
    df['volume_spike'] = (df['volume'] > 2 * df['5_day_avg_volume']).astype(int)
    
    # Adjust Momentum with Volume Spike
    df['adjusted_momentum'] = df.apply(
        lambda row: row['price_momentum'] * 0.8 if row['volume_spike'] else row['price_momentum'], 
        axis=1
    )
    
    return df['adjusted_momentum']
```
