```python
import pandas as pd

def heuristics_v2(df, N=10, M=30):
    # Calculate Cumulative Return over N Days
    df['Cumulative_Return'] = df['close'].pct_change(periods=N)
    
    # Measure Volume Trend
    df['Volume_Moving_Avg'] = df['volume'].rolling(window=M).mean()
    df['Volume_Trend'] = (df['volume'] > df['Volume_Moving_Avg']).astype(int)
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Adjust High-Low Spread by Volume Trend
    # If Volume > Moving Average, Increase Weight
    # If Volume <= Moving Average, Decrease Weight
    df['Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * (df['Volume_Trend'] * 2 - 1)
    
    # Combine Cumulative Return and Adjusted High-Low Spread
    alpha_factor = df['Cumulative_Return'] + df['Adjusted_High_Low_Spread']
    
    return alpha_factor
```
