```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['high_low_diff'] = df['high'] - df['low']
    df['high_prev_close_diff'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close_diff'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['high_low_diff', 'high_prev_close_diff', 'low_prev_close_diff']].max(axis=1)
    
    # Calculate Intraday Price Movement
    df['high_low_spread'] = df['high'] - df['low']
    df['sign_high_low_spread'] = df['high_low_spread'].apply(lambda x: 1 if x > 0 else -1)
    df['high_low_movement'] = (df['high'] - df['low']) * df['sign_high_low_spread']
    df['close_open_movement'] = (df['close'] - df['open']) * df['sign_high_low_spread']

    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    period = 14
    df['sum_high_low_movement'] = df['high_low_movement'].rolling(window=period).sum()
    df['sum_close_open_movement'] = df['close_open_movement'].rolling(window=period).sum()
    df['positive_volume'] = df['volume'] * (df['close'] > df['open']).astype(int)
    df['negative_volume'] = df['volume'] * (df['close'] < df['open']).astype(int)
    df['sum_positive_volume'] = df['positive_volume'].rolling(window=period).sum()
    df['sum_negative_volume'] = df['negative_volume'].rolling(window=period).sum()

    # Calculate Integrated Momentum Score
    df['volatility'] = df['close'].rolling(window=period).std()
    df['integrated_momentum'] = (df['sum_high_low_movement'] + df['sum_close_open_movement']) / df['volatility']
    df['total_volume'] = df['volume'].rolling(window=period).sum().abs()
    df['positive_volume_ratio'] = df['sum_positive_volume'] / df['total_volume']
    df['negative_volume_ratio'] = df['sum_negative_volume'] / df['total_volume']
    df['integrated_momentum_score'] = df['integrated_momentum'] * (df['positive_volume_ratio'] - df['negative_volume_ratio'])

    # Adjust for Volume Impact on Price
    df['volume_adjusted_high_low'] = (df['high'] - df['low']) * df['volume']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['combined_measure'] = (df['volume_adjusted_high_low'] + df['close_open_movement']) / 2

    # Dynamic Momentum Indicator
    df['yesterday_indicator'] = df['integrated_momentum_score'].shift(1)
    df['momentum_change'] = df['integrated_momentum_score'] - df['yesterday_indicator']
    df['dynamic_momentum'] = df['momentum_change'] / df['volatility']

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=period).mean()
    df['high_volume_day'] = (df['volume'] > df['avg_volume'] * 1.5).astype(int)

    # Integrate Combined Factors
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    df['integrated_factor'] = df['integrated_momentum_score'] * df['volume_ratio'] * df['high_volume_day']
    df['order_imbalance_score'] = (df['sum_positive_volume'] - df['sum_negative_volume']) / df['total_volume']
    df['final_alpha_factor'] = df['integrated_factor'] + df['order_imbalance_score']

    # Generate Final Alpha Factor
    return df['final_alpha_factor']
```
