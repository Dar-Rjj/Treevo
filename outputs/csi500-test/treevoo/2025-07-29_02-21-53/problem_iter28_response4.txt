```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    df['true_range_volume_adj'] = df['true_range'] * df['volume']

    # Apply Exponential Moving Average (20-day) to Daily High-Low Differences
    df['high_low_diff'] = df['high'] - df['low']
    df['smoothed_high_low_range'] = df['high_low_diff'].ewm(span=20).mean()

    # Combine Volume-Adjusted True Range and Smoothed High-Low Ranges
    df['integrated_vol_true_range_smoothed_high_low'] = df['true_range_volume_adj'] + df['smoothed_high_low_range']

    # Add Interaction Term (Smoothed High-Low Range * High-Low Difference)
    df['interaction_term'] = df['smoothed_high_low_range'] * df['high_low_diff']

    # Calculate 5-day and 20-day Simple Moving Averages of Close Prices
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()

    # Compute the Difference between 5-day and 20-day Moving Averages
    df['price_momentum'] = df['sma_5'] - df['sma_20']

    # Integrate into Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_vol_true_range_smoothed_high_low'] + 
        df['interaction_term'] + 
        df['price_momentum']
    )

    return df['final_alpha_factor']
```
