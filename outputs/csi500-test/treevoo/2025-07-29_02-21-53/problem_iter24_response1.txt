```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Change
    close_open_change = df['close'] - df['open']
    
    # Sum of High-Low Ranges over a Period (e.g., 20 days)
    sum_high_low_range = high_low_range.rolling(window=20).sum()
    
    # Sum of Close-Open Changes over a Period (e.g., 20 days)
    sum_close_open_change = close_open_change.rolling(window=20).sum()
    
    # Smooth High-Low Range with EMA (e.g., 20-day)
    smooth_high_low_range = high_low_range.ewm(span=20, adjust=False).mean()
    
    # Long-Term High-Low Range (e.g., 50-day)
    long_term_high_low_range = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min())
    
    # Compute Momentum of High-Low Ranges
    momentum_high_low = long_term_high_low_range - smooth_high_low_range
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Rolling Average True Range (ATR) (e.g., 14 days)
    atr = true_range.rolling(window=14).mean()
    
    # Adjust Price Momentum by ATR
    adjusted_momentum = (sum_high_low_range + sum_close_open_change) / atr
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    high_volume_days = (df['volume'] > avg_volume).astype(int)
    
    # Detect Volume Spike
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > 2 * five_day_avg_volume).astype(int)
    
    # Adjust Momentum Based on Volume
    adjusted_momentum = adjusted_momentum * (1 - 0.2 * volume_spike)
    
    # Integrate Combined Factors
    combined_factor = adjusted_momentum * high_volume_days
    
    # Combine ATR and Price Momentum
    interim_alpha = combined_factor + momentum_high_low
    
    # Incorporate Recent Price Trend
    recent_price_trend = df['close'].rolling(window=20).mean()
    trend_multiplier = (df['close'] > recent_price_trend).astype(int) * 1.1 - (df['close'] < recent_price_trend).astype(int) * 0.9
    alpha_with_trend = interim_alpha * trend_multiplier
    
    # Incorporate Volume Trend
    volume_trend = df['volume'].rolling(window=20).mean()
    volume_trend_multiplier = (df['volume'] > volume_trend).astype(int) * 1.1 - (df['volume'] < volume_trend).astype(int) * 0.9
    final_alpha = alpha_with_trend * volume_trend_multiplier
    
    return final_alpha
```
