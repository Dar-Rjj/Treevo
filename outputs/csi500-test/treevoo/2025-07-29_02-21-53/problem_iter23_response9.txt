```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A series indexed by date representing the factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume-weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the factor
    factor = (buying_pressure - selling_pressure) * (df['close'] - vwap)
    
    return factor
```
