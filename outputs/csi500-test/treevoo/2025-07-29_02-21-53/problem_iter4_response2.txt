import pandas as pd

def heuristics_v2(df, atr_window=14, momentum_window=10):
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_pc'] = abs(df['high'] - df['previous_close'])
    df['pc_low'] = abs(df['previous_close'] - df['low'])
    df['true_range'] = df[['high_low', 'high_pc', 'pc_low']].max(axis=1)
    
    # Calculate Rolling Average True Range
    df['atr'] = df['true_range'].rolling(window=atr_window).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(momentum_window)
    
    # Combine ATR and Price Momentum
    df['factor'] = df['price_momentum'] / df['atr']
    
    return df['factor']
