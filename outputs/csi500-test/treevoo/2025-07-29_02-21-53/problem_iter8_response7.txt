```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = abs(df['high'] - df['Prev_Close'])
    df['Low_PrevClose'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    
    # Calculate Rolling Average True Range (ATR) with 14 days window
    n_days = 14
    df['ATR'] = df['True_Range'].rolling(window=n_days).sum() / n_days
    
    # Calculate Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Exponential Moving Average of Returns
    short_ema = 12
    long_ema = 26
    df['Short_EMA_Return'] = df['Daily_Return'].ewm(span=short_ema, adjust=False).mean()
    df['Long_EMA_Return'] = df['Daily_Return'].ewm(span=long_ema, adjust=False).mean()
    
    # Calculate MACD (Moving Average Convergence Divergence)
    df['MACD'] = df['Short_EMA_Return'] - df['Long_EMA_Return']
    
    # Calculate Intraday Price Movement
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements over a period (e.g., 10 days)
    aggregate_period = 10
    df['Sum_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=aggregate_period).sum()
    df['Sum_Close_Open_Diff'] = df['Close_Open_Diff'].rolling(window=aggregate_period).sum()
    
    # Calculate Price Momentum Score
    df['Price_Momentum_Score'] = df['Sum_High_Low_Diff'] + df['Sum_Close_Open_Diff']
    df['Recent_Volatility'] = df['close'].rolling(window=aggregate_period).std()
    df['Adjusted_Price_Momentum'] = df['Price_Momentum_Score'] * (1 + df['Recent_Volatility'])
    
    # Evaluate Volume Activity
    avg_volume_window = 10
    df['Average_Volume'] = df['volume'].rolling(window=avg_volume_window).mean()
    df['High_Volume_Indicator'] = (df['volume'] > df['Average_Volume']).astype(int)
    
    # Integrate Combined Factors
    max_macd_window = 50
    df['Normalized_MACD'] = df['MACD'] / df['MACD'].rolling(window=max_macd_window).max()
    df['Volume_Adjusted_Momentum'] = df['Normalized_MACD'] + df['Adjusted_Price_Momentum']
    df['Final_Alpha_Factor'] = df['Volume_Adjusted_Momentum'] * df['High_Volume_Indicator']
    
    return df['Final_Alpha_Factor']
```
