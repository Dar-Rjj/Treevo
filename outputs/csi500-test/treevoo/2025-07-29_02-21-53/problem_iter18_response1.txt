```python
import pandas as pd

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['True_Range'] = df[['High', 'Low']].sub(df['Close'].shift(), axis=0).abs().max(axis=1)
    
    # Compute Intraday Price Movement using High, Low, Open, and Close
    df['Intraday_Price_Movement'] = (df['Close'] - df['Open'])
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_TR'] = df['True_Range'] * df['Volume']
    df['Volume_Adjusted_IPM'] = df['Intraday_Price_Movement'] * df['Volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['Smoothed_High_Low_Range'] = df['High'] - df['Low']
    df['Smoothed_High_Low_Range'] = df['Smoothed_High_Low_Range'].ewm(span=10).mean()
    
    df['Smoothed_Price_Movement'] = df['Intraday_Price_Movement'].ewm(span=10).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Range_Momentum'] = df['Smoothed_High_Low_Range'].ewm(span=30).mean() - df['Smoothed_High_Low_Range'].ewm(span=10).mean()
    df['Price_Volatility'] = df['Intraday_Price_Movement'].rolling(window=10).std()
    df['Combined_Momentum'] = df['Range_Momentum'] / df['Price_Volatility']
    
    # Evaluate Volume Activity
    df['EMA_Volume'] = df['Volume'].ewm(span=10).mean()
    df['High_Volume_Day'] = df['Volume'] > df['EMA_Volume']
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Combined_Measures'] = df['Volume_Adjusted_TR'] + df['Volume_Adjusted_IPM']
    df['Trend_Continuity'] = df['Combined_Momentum'] * df['High_Volume_Day']
    
    # Generate Final Alpha Factor
    df['Interaction_Term'] = df['Range_Momentum'] * df['Intraday_Price_Movement']
    df['Final_Alpha_Factor'] = df['Trend_Continuity'] + df['Interaction_Term']
    
    return df['Final_Alpha_Factor']
```
