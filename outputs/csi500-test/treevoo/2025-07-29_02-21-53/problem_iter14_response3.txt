```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the money flow ratio
    df['money_flow'] = (df['close'] - df['low']) / (df['high'] - df['low']) * df['volume']
    
    # Calculate the normalized money flow over the last 5 days
    df['money_flow_5d'] = df['money_flow'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the volatility of the close price
    df['volatility'] = df['close'].rolling(window=5).std()
    
    # Calculate the factor as a combination of the above metrics
    df['factor'] = df['close_change'] * df['money_flow_5d'] / (df['volatility'] + 1e-6)
    
    # Return the factor as a Series
    return df['factor'].dropna()
```
