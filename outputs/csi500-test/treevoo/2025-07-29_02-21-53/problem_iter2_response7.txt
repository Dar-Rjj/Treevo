```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Oscillation
    HL_oscillation = df['high'] - df['low']
    
    # Smooth Oscillation with Exponential Moving Average over 14 days
    EMA_oscillation = HL_oscillation.ewm(span=14, adjust=False).mean()
    
    # Compute Momentum Indicator from EMA
    momentum_indicator = EMA_oscillation - EMA_oscillation.shift(14)
    
    # Calculate Signed Transaction Volume
    signed_volume = df['volume'] * np.sign(df['high'] - df['low'])
    
    # Calculate Accumulated Buy-Sell Imbalance
    positive_volumes = signed_volume[signed_volume > 0].sum()
    negative_volumes = signed_volume[signed_volume < 0].sum().abs()
    
    # Compute Order Imbalance Score
    total_absolute_volumes = signed_volume.abs().sum()
    order_imbalance_score = (positive_volumes / total_absolute_volumes) - (negative_volumes / total_absolute_volumes)
    
    # Synthesize Final Alpha Factor
    alpha_factor = momentum_indicator * order_imbalance_score
    
    return alpha_factor
```
