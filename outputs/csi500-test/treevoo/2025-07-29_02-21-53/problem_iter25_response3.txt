```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['True_Range'] = df[['High', 'Low']].sub(df['Close'].shift(1), axis=0).abs().max(axis=1)
    
    # Compute Intraday Price Movement using High, Low, Open, and Close
    df['Intraday_Price_Movement'] = (df['High'] - df['Low']) + (df['Close'] - df['Open'])

    # Adjust for Volume Impact on Price
    df['True_Range_Volume_Adjusted'] = df['True_Range'] * df['Volume']
    df['Intraday_Price_Movement_Volume_Adjusted'] = df['Intraday_Price_Movement'] * df['Volume']

    # Compute High-Low Difference and Close-Open Difference
    df['High_Low_Diff'] = df['High'] - df['Low']
    df['Close_Open_Diff'] = df['Close'] - df['Open']

    # Smooth High-Low Range and Aggregate Price Movements
    df['EMA_High_Low_20'] = df['High_Low_Diff'].ewm(span=20, adjust=False).mean()
    df['Sum_High_Low_Diff_20'] = df['High_Low_Diff'].rolling(window=20).sum()
    df['Sum_Close_Open_Diff_20'] = df['Close_Open_Diff'].rolling(window=20).sum()

    # Calculate Long-Term High-Low Range
    df['Long_Term_High_Low_Range_50'] = df['High'].rolling(window=50).max() - df['Low'].rolling(window=50).min()

    # Compute Combined Price and High-Low Range Momentum
    df['Combined_Momentum'] = df['Sum_High_Low_Diff_20'] + df['Sum_Close_Open_Diff_20'] - (df['Long_Term_High_Low_Range_50'] - df['EMA_High_Low_20'])

    # Weight by Recent Price Volatility
    df['Std_Close_20'] = df['Close'].rolling(window=20).std()
    df['Weighted_Combined_Momentum'] = df['Combined_Momentum'] * df['Std_Close_20']

    # Evaluate Volume Activity
    df['EMA_Volume_20'] = df['Volume'].ewm(span=20, adjust=False).mean()
    df['High_Volume'] = df['Volume'] > 1.5 * df['EMA_Volume_20']

    # Integrate Price, High-Low Range, and Volume Factors
    df['Volume_Ratio'] = df['Volume'] / df['EMA_Volume_20']
    df['Volume_Adjusted_Measure'] = df['Weighted_Combined_Momentum'] * df['Volume_Ratio']

    # Assess Trend Continuity and Reversal Potential
    df['Market_Return'] = df['Close'].pct_change()
    df['Adjusted_Alpha_Factor'] = df['Volume_Adjusted_Measure'] + df['Market_Return']

    # Consider Volume Weighted Price Change
    df['VWAP'] = (df['Close'] * df['Volume']).cumsum() / df['Volume'].cumsum()
    df['Final_Alpha_Factor'] = df['Adjusted_Alpha_Factor'] * (df['VWAP'] - df['Close'])

    # Adjust Alpha Factor for Volume Spike
    df['Volume_Spike'] = df['Volume'] > 2 * df['EMA_Volume_20']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * (1 + 3 * df['Volume_Spike'].astype(int))

    return df['Final_Alpha_Factor'].dropna()
```
