```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=5).sum()
    
    # Sum of Close-Open Differences over a Period
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=5).sum()
    
    # Calculate Price Momentum Score
    price_momentum = df['sum_high_low_diff'] + df['sum_close_open_diff']
    
    # Calculate Standard Deviation of Close Prices
    std_close = df['close'].rolling(window=5).std()
    price_momentum_weighted = price_momentum * (std_close / std_close.mean())
    
    # Compute Average Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Identify Days with High Volume
    df['high_volume'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Compute Volume Ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Multiply Price Momentum Score by Volume Ratio
    alpha_factor = price_momentum_weighted * volume_ratio * df['high_volume']
    
    # Incorporate Market Index Returns (assuming 'market_return' is available in the DataFrame)
    if 'market_return' in df.columns:
        market_return = df['market_return']
        alpha_factor = alpha_factor * (1 + market_return)
    
    return alpha_factor
```
