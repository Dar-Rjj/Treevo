```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Rolling Average True Range
    n_days_atr = 14  # You can adjust the number of days for ATR
    df['atr'] = df['true_range'].rolling(window=n_days_atr).mean()
    
    # Calculate Price Momentum
    n_days_momentum = 10  # You can adjust the number of days for momentum
    df['price_momentum'] = df['close'] - df['close'].shift(n_days_momentum)
    
    # Combine ATR and Price Momentum
    df['normalized_momentum'] = df['price_momentum'] / df['atr']
    
    return df['normalized_momentum']
```
