```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range and normalized close price
    price_range = df['high'] - df['low']
    close_normalized = (df['close'] - df['low']) / price_range
    
    # Calculate the volume-weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the factor as a combination of the above metrics
    factor = (close_normalized + volume_weighted_close / df['amount'] + avg_price) / 3
    
    return factor
```
