```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['High-Low'] = df['high'] - df['low']
    df['High-PrevClose'] = abs(df['high'] - df['Previous Close'])
    df['Low-PrevClose'] = abs(df['low'] - df['Previous Close'])
    df['True Range'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)

    # Calculate Rolling Average True Range (ATR) with 14 days window
    n_days = 14
    df['ATR'] = df['True Range'].rolling(window=n_days).mean()

    # Calculate Intraday High-Low Range
    df['Intraday Range'] = df['high'] - df['low']

    # Adjust for Volume Impact on Price
    df['Volume Adjusted Intraday Range'] = df['Intraday Range'] * df['volume']

    # Calculate Open-Close Momentum
    df['Open-Close Momentum'] = df['close'] - df['open']

    # Combine Volume-Adjusted High-Low with Open-Close
    weight_intraday = 0.6
    weight_open_close = 0.4
    df['Combined Measure'] = (weight_intraday * df['Volume Adjusted Intraday Range']) + (weight_open_close * df['Open-Close Momentum'])

    # Introduce Volume Trend Analysis
    df['Volume Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume Trend'] = np.sign(df['Volume Change'])

    # Dynamic Momentum Indicator
    df['Yesterday Combined Measure'] = df['Combined Measure'].shift(1)
    df['Momentum Change'] = df['Combined Measure'] - df['Yesterday Combined Measure']
    df['Volatility'] = df['close'].pct_change().rolling(window=30).std()
    df['Smoothed Momentum Change'] = df['Momentum Change'] / df['Volatility']

    # Aggregate Price Movements and Smooth High-Low Ranges
    df['Sum High-Low'] = df['Intraday Range'].rolling(window=20).sum()
    df['Sum Close-Open'] = (df['close'] - df['open']).rolling(window=20).sum()
    df['Smoothed High-Low Range'] = df['Intraday Range'].ewm(span=20, adjust=False).mean()
    df['Long-Term High-Low Range'] = df['Intraday Range'].rolling(window=50).mean()

    # Calculate Combined Price and High-Low Range Momentum
    df['Combined Momentum'] = df['Sum High-Low'] + df['Sum Close-Open'] + (df['Long-Term High-Low Range'] - df['Smoothed High-Low Range'])
    df['Price Volatility'] = df['close'].pct_change().rolling(window=10).std()
    df['Weighted Combined Momentum'] = df['Combined Momentum'] * df['Price Volatility']

    # Evaluate Volume Activity
    df['Average Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['High Volume'] = df['volume'] > 1.5 * df['Average Volume']

    # Integrate Combined Factors
    df['Final Alpha Factor'] = df['ATR'] * df['Smoothed Momentum Change'] * df['Weighted Combined Momentum'] * df['High Volume']

    return df['Final Alpha Factor'].dropna()
```
