import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['PrevClose'] = df['close'].shift(1)
    df['TrueRange'] = df[['high', 'low', 'PrevClose']].max(axis=1) - df[['high', 'low', 'PrevClose']].min(axis=1)

    # Calculate Intraday Price Movement
    df['HighLowDiff'] = df['high'] - df['low']
    df['CloseOpenDiff'] = df['close'] - df['open']
    df['IntradayMomentum'] = df['HighLowDiff'] * (df['high'] - df['low']).apply(lambda x: 1 if x > 0 else -1)

    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    window = 5  # Define the period for aggregation
    df['SumHighLowDiff'] = df['HighLowDiff'].rolling(window=window).sum()
    df['SumCloseOpenDiff'] = df['CloseOpenDiff'].rolling(window=window).sum()
    df['PosVolume'] = df['volume'] * (df['close'] > df['open']).astype(int)
    df['NegVolume'] = df['volume'] * (df['close'] < df['open']).astype(int)
    df['AccumPosVolume'] = df['PosVolume'].rolling(window=window).sum()
    df['AccumNegVolume'] = df['NegVolume'].rolling(window=window).sum()

    # Calculate Integrated Momentum Score
    df['Volatility'] = df['close'].rolling(window=window).std()
    df['IntegratedMomentumScore'] = (df['SumHighLowDiff'] + df['SumCloseOpenDiff']) / df['Volatility']
    df['TotalAbsVolume'] = df['volume'].rolling(window=window).sum().abs()
    df['PosVolumeRatio'] = df['AccumPosVolume'] / df['TotalAbsVolume']
    df['NegVolumeRatio'] = df['AccumNegVolume'] / df['TotalAbsVolume']
    df['IntegratedMomentumScore'] += df['PosVolumeRatio'] - df['NegVolumeRatio']

    # Adjust for Volume Impact on Price
    df['VolumeAdjustedHighLow'] = df['HighLowDiff'] * df['volume']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['CombinedMeasure'] = 0.5 * df['VolumeAdjustedHighLow'] + 0.5 * df['CloseOpenDiff']

    # Introduce Volume Trend Analysis
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['CombinedMeasure'] *= (1 + df['VolumeChange'].rolling(window=window).mean())

    # Dynamic Momentum Indicator
    df['PrevDayIndicator'] = df['IntegratedMomentumScore'].shift(1)
    df['TrendContinuity'] = df['IntegratedMomentumScore'] - df['PrevDayIndicator']
    df['DynamicMomentumIndicator'] = df['TrendContinuity'] / df['Volatility']

    # Evaluate Volume Activity
    df['AvgVolume'] = df['volume'].rolling(window=window).mean()
    df['HighVolume'] = (df['volume'] > df['AvgVolume']).astype(int)

    # Integrate Combined Factors
    df['VolumeRatio'] = df['volume'] / df['AvgVolume']
    df['FinalAlphaFactor'] = df['IntegratedMomentumScore'] * df['VolumeRatio'] * df['DynamicMomentumIndicator']

    return df['FinalAlphaFactor']
