```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday and Long-Term Price Movements
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Smooth High-Low Range
    high_low_range_ema = high_low_diff.ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range
    long_term_high = df['high'].rolling(window=50).max()
    long_term_low = df['low'].rolling(window=50).min()
    long_term_high_low_range = long_term_high - long_term_low
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    sum_close_open_diff = close_open_diff.rolling(window=20).sum()
    smoothed_high_low_range = high_low_diff.rolling(window=20).mean()
    
    # Compute Combined Price and High-Low Range Momentum
    combined_momentum = (sum_high_low_diff + sum_close_open_diff) / smoothed_high_low_range
    
    # Weight by Recent Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    weighted_momentum = combined_momentum * price_volatility
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    high_volume_days = df['volume'] > 1.5 * avg_volume
    
    # Integrate Price, High-Low Range, and Volume Factors
    volume_ratio = df['volume'] / avg_volume
    integrated_factor = weighted_momentum * volume_ratio
    
    # Final Alpha Factor
    # Adjust for Market Trends (assuming market index returns are in a column named 'market_return')
    market_trend_adjustment = df['market_return'].rolling(window=20).mean()
    final_alpha_factor = integrated_factor + market_trend_adjustment
    
    # Adjust Alpha Factor for Volume Spike
    volume_spike = df['volume'] / avg_volume
    final_alpha_factor[volume_spike > 3] *= volume_spike[volume_spike > 3]
    
    return final_alpha_factor
```
