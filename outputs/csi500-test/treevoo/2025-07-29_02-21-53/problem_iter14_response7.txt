```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday and Long-Term Price Movements
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Smooth High-Low Range
    df['hl_ema_20'] = df['high_low_diff'].ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range
    df['long_term_high'] = df['high'].rolling(window=50).max()
    df['long_term_low'] = df['low'].rolling(window=50).min()
    df['long_term_range'] = df['long_term_high'] - df['long_term_low']
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    df['sum_high_low_20'] = df['high_low_diff'].rolling(window=20).sum()
    df['sum_close_open_20'] = df['close_open_diff'].rolling(window=20).sum()
    df['smoothed_hl_20'] = df['hl_ema_20'].rolling(window=20).mean()
    
    # Compute Combined Price and High-Low Range Momentum
    df['combined_momentum'] = (df['sum_high_low_20'] + df['sum_close_open_20']) - (df['long_term_range'] - df['smoothed_hl_20'])
    
    # Weight by Recent Price Volatility
    df['price_volatility'] = df['close'].rolling(window=20).std()
    df['weighted_momentum'] = df['combined_momentum'] * (df['price_volatility'] / df['price_volatility'].mean())
    
    # Evaluate Volume Activity
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = df['volume'] > df['avg_volume_20'] * 1.5
    
    # Integrate Price, High-Low Range, and Volume Factors
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['momentum_with_volume'] = df['weighted_momentum'] * df['volume_ratio']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_with_volume']
    
    # Adjust for Market Trends and Volume Spike Impact
    df['market_return'] = df['close'].pct_change()
    df['alpha_factor'] += df['market_return'] * 0.1  # Adjust by 10% of market return
    df['volume_spike'] = df['volume'] / df['avg_volume_20'] > 2  # Volume more than 2x average
    df['alpha_factor'] = df['alpha_factor'] * (1 + 0.5 * df['volume_spike'])  # Increase factor on high volume spike days
    
    # Consider Volume Weighted Price Change
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['simple_avg'] = df['close'].rolling(window=20).mean()
    df['alpha_factor'] += (df['vwap'] - df['simple_avg']) * 0.1  # Adjust by 10% of the difference
    
    return df['alpha_factor']
```
