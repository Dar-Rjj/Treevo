```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by (date).
    """
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of volume to amount
    df['volume_to_amount'] = df['volume'] / df['amount']
    
    # Calculate the average of the open, high, low, and close prices
    df['price_avg'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['close_change'] * df['volume_to_amount']) * df['price_avg']
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
```
