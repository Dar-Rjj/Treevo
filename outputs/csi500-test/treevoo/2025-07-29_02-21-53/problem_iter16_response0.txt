```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Close-Open Change
    df['close_open_change'] = df['close'] - df['open']

    # Sum of High-Low Ranges over a Period
    df['sum_high_low_range'] = df['high_low_range'].rolling(window=20).sum()
    
    # Sum of Close-Open Changes over a Period
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=20).sum()

    # Smooth High-Low Range with Exponential Moving Average (EMA)
    df['smooth_high_low_range'] = df['high_low_range'].ewm(span=20, adjust=False).mean()

    # Compute Momentum of High-Low Ranges
    df['momentum_high_low_range'] = df['smooth_high_low_range'] - df['smooth_high_low_range'].shift(10)

    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)

    # Calculate Rolling Average True Range
    df['rolling_atr'] = df['true_range'].rolling(window=20).sum()

    # Calculate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=20).std()

    # Adjust by Recent Price Volatility
    df['price_momentum_score'] = (df['sum_high_low_range'] + df['sum_close_open_change']) / df['price_volatility']

    # Evaluate Volume Activity
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = df['volume'] > 1.5 * df['average_volume']

    # Integrate Price Momentum and Volume Activity
    df['interim_alpha_factor'] = df['price_momentum_score'] * df['high_volume']

    # Combine ATR and Price Momentum
    df['interpretable_factor'] = df['momentum_high_low_range'] * df['interim_alpha_factor']

    # Final Alpha Factor
    df['volume_ratio'] = df['volume'] / df['average_volume']
    df['final_alpha_factor'] = df['interpretable_factor'] * df['volume_ratio']

    return df['final_alpha_factor']
```
