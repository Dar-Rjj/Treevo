```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Close-Open Change
    close_open_change = df['close'] - df['open']
    
    # Sum of High-Low Ranges over a Period (e.g., 10 days)
    sum_high_low_range = high_low_range.rolling(window=10).sum()
    
    # Sum of Close-Open Changes over a Period (e.g., 10 days)
    sum_close_open_change = close_open_change.rolling(window=10).sum()
    
    # Smooth High-Low Range with EMA (e.g., 20-day)
    smoothed_high_low_range = high_low_range.ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (e.g., 50-day)
    long_term_high_low_range = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min())
    
    # Compute Momentum of High-Low Ranges
    momentum_high_low_range = long_term_high_low_range - smoothed_high_low_range
    
    # Calculate True Range
    prev_close = df['close'].shift(1)
    true_range = np.maximum.reduce([df['high'] - df['low'], abs(df['high'] - prev_close), abs(df['low'] - prev_close)])
    
    # Calculate Rolling Average True Range (ATR) (e.g., 14 days)
    atr = true_range.rolling(window=14).mean()
    
    # Adjust Price Momentum by ATR
    adjusted_momentum = (sum_high_low_range + sum_close_open_change) / atr
    
    # Evaluate Volume Activity: Compute Average Volume (e.g., 30 days)
    avg_volume = df['volume'].rolling(window=30).mean()
    
    # Identify High Volume Days
    high_volume_indicator = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Detect Volume Spike
    vol_5_day_avg = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > 2 * vol_5_day_avg).astype(int)
    
    # Adjust Momentum Based on Volume
    adjusted_momentum = adjusted_momentum * (1 - 0.2 * volume_spike)
    
    # Integrate Combined Factors
    combined_factor = adjusted_momentum * high_volume_indicator
    
    # Combine ATR and Price Momentum
    interim_alpha_factor = combined_factor + momentum_high_low_range
    
    # Incorporate Recent Price Trend
    recent_price_trend = df['close'].rolling(window=10).mean().pct_change()
    trend_multiplier = np.where(recent_price_trend > 0, 1.1, 0.9)
    
    # Final Alpha Factor
    final_alpha_factor = interim_alpha_factor * trend_multiplier
    
    return final_alpha_factor
```
