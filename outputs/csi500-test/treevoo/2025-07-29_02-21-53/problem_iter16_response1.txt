```python
import pandas as pd

def heuristics_v2(df):
    # Compute True Range
    df['True_Range'] = df['high'] - df['low']
    
    # Compute Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] + df['close']) / 2 - (df['open'] + df['low']) / 2
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_TR'] = df['True_Range'] * df['volume']
    df['Volume_Adjusted_IPM'] = df['Intraday_Price_Movement'] * df['volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['Smoothed_High_Low_Range'] = df['high'] - df['low']
    df['Smoothed_High_Low_Range'] = df['Smoothed_High_Low_Range'].ewm(span=20).mean()
    df['Smoothed_Price_Movement'] = df['close'] - df['open']
    df['Smoothed_Price_Movement'] = df['Smoothed_Price_Movement'].ewm(span=20).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Combined_Momentum'] = df['Smoothed_Price_Movement'] - df['Smoothed_High_Low_Range']
    df['Price_Volatility'] = df['close'].rolling(window=20).std()
    df['Weighted_Combined_Momentum'] = df['Combined_Momentum'] * df['Price_Volatility']
    
    # Evaluate Volume Activity
    df['EMA_Volume'] = df['volume'].ewm(span=20).mean()
    df['High_Volume_Indicator'] = (df['volume'] > df['EMA_Volume']).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Integrated_Measure'] = (df['Volume_Adjusted_TR'] + df['Volume_Adjusted_IPM']) * df['High_Volume_Indicator']
    
    # Generate Final Alpha Factor
    df['Alpha_Factor'] = df['Integrated_Measure'] * (df['Weighted_Combined_Momentum'] * (df['close'] - df['open']))
    
    return df['Alpha_Factor']
```
