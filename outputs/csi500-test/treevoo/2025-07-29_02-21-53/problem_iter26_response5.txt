```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator Development
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Crossover_Signal'] = df['SMA_5'] - df['SMA_20']

    # Volume Trend Analysis
    df['Volume_Change'] = df['volume'].diff()
    df['Positive_Volume_Sum'] = (df['Volume_Change'] > 0) * df['Volume_Change'].rolling(window=20).sum()
    df['Negative_Volume_Sum'] = (df['Volume_Change'] < 0) * abs(df['Volume_Change']).rolling(window=20).sum()
    df['Volume_Accumulation_Oscillator'] = df['Positive_Volume_Sum'] - df['Negative_Volume_Sum']

    # Price Action Strength
    df['True_Range'] = pd.concat([df['high'] - df['low'], 
                                  abs(df['high'] - df['close'].shift(1)), 
                                  abs(df['low'] - df['close'].shift(1))], axis=1).max(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Price_Strength_Indicator'] = df['True_Range'] / df['ATR_14']

    # Market Sentiment Factors
    df['Open_Close_Spread'] = df['open'] - df['close']
    df['Daily_High_Low_Range_Percentage'] = (df['high'] - df['low']) / df['low'] * 100
    df['Combined_Sentiment_Score'] = df['Open_Close_Spread'] + df['Daily_High_Low_Range_Percentage']

    # Synthetic Alpha Factor
    df['Synthetic_Alpha_Factor'] = (df['Crossover_Signal'] * 0.4 +
                                    df['Volume_Accumulation_Oscillator'] * 0.3 +
                                    df['Combined_Sentiment_Score'] * 0.3)

    return df['Synthetic_Alpha_Factor']
```
