```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Smooth High-Low Range with 20-day EMA
    high_low_range_ema = high_low_diff.ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range over 50 days
    long_term_high_low_range = (df['high'].rolling(window=50, min_periods=1).max() 
                                - df['low'].rolling(window=50, min_periods=1).min())
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    sum_high_low_diff = high_low_diff.rolling(window=20, min_periods=1).sum()
    sum_close_open_diff = close_open_diff.rolling(window=20, min_periods=1).sum()
    
    # Combine Price and High-Low Range Momentum
    combined_momentum = (sum_high_low_diff + sum_close_open_diff 
                         - (high_low_range_ema - long_term_high_low_range))
    
    # Weight by Recent Price Volatility
    price_volatility = df['close'].rolling(window=20, min_periods=1).std()
    weighted_momentum = combined_momentum * price_volatility
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].ewm(span=20, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume
    high_volume_days = volume_ratio > 1.5  # Threshold for high volume days
    
    # Integrate Price, High-Low Range, and Volume Factors
    momentum_with_volume = weighted_momentum * volume_ratio
    final_alpha_factor = (momentum_with_volume + high_low_range_ema)
    
    # Smooth Oscillation with Exponential Moving Average
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=14, adjust=False).mean()
    
    return final_alpha_factor_smoothed
```
