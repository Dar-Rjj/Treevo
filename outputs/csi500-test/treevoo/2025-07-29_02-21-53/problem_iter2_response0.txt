```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Assign Weights (example: 0.5 for high-low spread, 0.5 for close-to-open)
    weight_high_low = 0.5
    weight_close_open = 0.5
    
    # Calculate Averages
    weighted_average = (weight_high_low * intraday_high_low_spread) + (weight_close_open * close_to_open_change)
    
    # Adjust for Volume
    adjusted_intraday_measures = weighted_average * df['volume']
    
    # Apply Exponential Moving Average (EMA) over 14 days
    ema_14_days = adjusted_intraday_measures.ewm(span=14, adjust=False).mean()
    
    # Compute Integrated Momentum Indicator
    integrated_momentum_indicator = ema_14_days - ema_14_days.shift(14)
    
    return integrated_momentum_indicator
```
