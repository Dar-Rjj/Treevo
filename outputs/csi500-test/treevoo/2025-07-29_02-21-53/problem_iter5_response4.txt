```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Exponential Moving Average of Returns
    df['short_ema'] = df['daily_return'].ewm(span=12, adjust=False).mean()
    df['long_ema'] = df['daily_return'].ewm(span=26, adjust=False).mean()
    
    # Calculate MACD
    df['macd'] = df['short_ema'] - df['long_ema']
    
    # Calculate Volume Adjusted Momentum
    raw_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['volume_adjusted_momentum'] = raw_momentum * df['volume']
    
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    sum_high_low_range = df['high_low_range'].rolling(window=10).sum()
    sum_close_open_change = df['close_open_change'].rolling(window=10).sum()
    
    # Calculate Price Momentum Score
    close_volatility = df['close'].rolling(window=10).std()
    price_momentum_score = (sum_high_low_range + sum_close_open_change) / close_volatility
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['high_volume_indicator'] = (df['volume'] > df['avg_volume']).astype(int)
    
    # Normalize MACD (local normalization)
    max_macd = df['macd'].rolling(window=50).max()
    normalized_macd = df['macd'] / max_macd
    
    # Combine MACD and Volume Adjusted Momentum
    combined_macd_momentum = normalized_macd + df['volume_adjusted_momentum']
    
    # Integrate Price Momentum and Volume Activity
    final_alpha_factor = price_momentum_score * df['high_volume_indicator'] + combined_macd_momentum
    
    return final_alpha_factor
```
