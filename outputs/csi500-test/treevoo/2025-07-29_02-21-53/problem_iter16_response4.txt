```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'previous_close']].min(axis=1)

    # Calculate Intraday Price Movement
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_movement'] = df['close_open_diff'] * (df['high'] - df['low']).apply(lambda x: 1 if x > 0 else -1)

    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    period = 5
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=period).sum()
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=period).sum()
    df['pos_vol'] = (df['volume'] * (df['close'] > df['open'])).rolling(window=period).sum()
    df['neg_vol'] = (df['volume'] * (df['close'] <= df['open'])).rolling(window=period).sum()

    # Calculate Integrated Momentum Score
    df['close_volatility'] = df['close'].rolling(window=period).std()
    df['integrated_momentum_score'] = (
        (df['sum_high_low_diff'] + df['sum_close_open_diff']) / df['close_volatility'] 
        + (df['pos_vol'] - df['neg_vol']) / (df['pos_vol'] + df['neg_vol'])
    )

    # Adjust for Volume Impact on Price
    df['volume_adjusted_high_low'] = df['high_low_diff'] * df['volume']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['combined_price_movement'] = (df['volume_adjusted_high_low'] + df['close_open_diff']) / 2

    # Dynamic Momentum Indicator
    df['yesterday_indicator'] = df['intraday_movement'].shift(1)
    df['momentum_continuity'] = (df['intraday_movement'] - df['yesterday_indicator']) / df['close_volatility']
    df['dynamic_momentum'] = df['intraday_movement'] + df['momentum_continuity']

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=period).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    df['high_volume'] = df['volume'] > df['avg_volume']

    # Integrate Combined Factors
    df['order_imbalance_score'] = (df['pos_vol'] - df['neg_vol']) / (df['pos_vol'] + df['neg_vol'])
    df['alpha_factor'] = df['integrated_momentum_score'] * df['volume_ratio'] + df['order_imbalance_score']

    return df['alpha_factor']
```
