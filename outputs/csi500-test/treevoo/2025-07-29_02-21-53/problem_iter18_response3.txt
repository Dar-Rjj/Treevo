import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()

    # Price Change Over Time
    df['PCT_change_1'] = df['close'].pct_change(periods=1)
    df['PCT_change_5'] = df['close'].pct_change(periods=5)
    df['PCT_change_20'] = df['close'].pct_change(periods=20)

    # Volume Rate of Change (VROC)
    df['VROC_5'] = df['volume'].pct_change(periods=5)
    df['VROC_20'] = df['volume'].pct_change(periods=20)

    # Volume Weighted Average Price (VWAP)
    df['VWAP_daily'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP_5'] = df['VWAP_daily'].rolling(window=5).mean()
    df['VWAP_20'] = df['VWAP_daily'].rolling(window=20).mean()

    # Historical Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['HistVol_20'] = log_returns.rolling(window=20).std() * np.sqrt(20)
    df['HistVol_50'] = log_returns.rolling(window=50).std() * np.sqrt(50)

    # True Range and Average True Range (ATR)
    df['TrueRange'] = df[['high', 'low']].diff(axis=1).max(axis=1)
    df['ATR_10'] = df['TrueRange'].rolling(window=10).mean()

    # Inside Bar
    df['InsideBar'] = ((df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1))).astype(int)
    df['ConsecutiveInsideBars'] = df['InsideBar'].groupby((df['InsideBar'] != df['InsideBar'].shift()).cumsum()).cumsum()

    # Outside Bar
    df['OutsideBar'] = ((df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))).astype(int)
    df['ConsecutiveOutsideBars'] = df['OutsideBar'].groupby((df['OutsideBar'] != df['OutsideBar'].shift()).cumsum()).cumsum()

    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()

    df['Signal_Line'] = df['MACD_Line'].ewm(span=9, adjust=False).mean()
    df['MACD_Histogram'] = df['MACD_Line'] - df['Signal_Line']
