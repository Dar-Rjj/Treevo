```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday and Long-Term Price Movements
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Smooth High-Low Range with 20-day EMA
    df['high_low_ema_20'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    df['long_term_high'] = df['high'].rolling(window=50).max()
    df['long_term_low'] = df['low'].rolling(window=50).min()
    df['long_term_high_low_range'] = df['long_term_high'] - df['long_term_low']
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    df['sum_high_low_diff_20'] = df['high_low_diff'].rolling(window=20).sum()
    df['sum_close_open_diff_20'] = df['close_open_diff'].rolling(window=20).sum()
    
    # Compute Combined Price and High-Low Range Momentum
    df['combined_momentum'] = (
        df['sum_high_low_diff_20'] + 
        df['sum_close_open_diff_20'] + 
        (df['long_term_high_low_range'] - df['high_low_ema_20'])
    )
    
    # Weight by Recent Price Volatility
    df['close_volatility'] = df['close'].rolling(window=20).std()
    df['vol_weighted_momentum'] = df['combined_momentum'] * df['close_volatility']
    
    # Evaluate Volume Activity
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['high_volume'] = df['volume_ratio'] > 1.5
    
    # Integrate Price, High-Low Range, and Volume Factors
    df['final_alpha'] = df['vol_weighted_momentum'] * df['volume_ratio']
    
    # Adjust for Market Trends
    if 'market_return' in df.columns:
        df['final_alpha'] += df['market_return']
    
    # Identify Sudden Volume Spikes
    df['volume_spike'] = df['volume_ratio'] > 2.0
    df.loc[df['volume_spike'], 'final_alpha'] *= 1.1  # Increase factor value on high volume spike days
    
    return df['final_alpha']
```
