```python
import pandas as pd

def heuristics_v2(df):
    # Compute True Range
    df['TrueRange'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['TrueRange'] = df[['TrueRange', 'close']].apply(lambda x: max(x[0], abs(x[0] - x[1])), axis=1)
    
    # Adjust True Range by Square Root of Volume
    df['VolumeAdjustedTrueRange'] = df['TrueRange'] * (df['volume'] ** 0.5)
    
    # Apply Exponential Moving Average (20-day) to Daily High-Low Differences
    df['SmoothedHighLowRange'] = df['high'] - df['low']
    df['SmoothedHighLowRange'] = df['SmoothedHighLowRange'].ewm(span=20, min_periods=20).mean()
    
    # Combine Volume-Adjusted True Range and Smoothed High-Low Ranges
    df['CombinedFactor'] = df['VolumeAdjustedTrueRange'] + df['SmoothedHighLowRange']
    
    # Add Interaction Term (Smoothed High-Low Range * High-Low Difference)
    df['InteractionTerm'] = df['SmoothedHighLowRange'] * (df['high'] - df['low'])
    df['FinalAlphaFactor'] = df['CombinedFactor'] + df['InteractionTerm']
    
    # Calculate 5-day and 20-day Simple Moving Averages of Close Prices
    df['SMA5'] = df['close'].rolling(window=5, min_periods=5).mean()
    df['SMA20'] = df['close'].rolling(window=20, min_periods=20).mean()
    
    # Compute the Difference between 5-day and 20-day Moving Averages
    df['PriceMomentum'] = df['SMA5'] - df['SMA20']
    
    # Integrate Price Momentum into Final Alpha Factor
    df['FinalAlphaFactor'] += df['PriceMomentum']
    
    return df['FinalAlphaFactor']
```
