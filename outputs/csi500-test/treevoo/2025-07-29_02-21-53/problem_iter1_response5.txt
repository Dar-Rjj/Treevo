```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements over a 5-day period
    df['agg_high_low_diff'] = df['high_low_diff'].rolling(window=5).sum()
    df['agg_close_open_diff'] = df['close_open_diff'].rolling(window=5).sum()
    
    # Calculate Price Momentum Score
    price_momentum_score = df['agg_high_low_diff'] + df['agg_close_open_diff']
    
    # Weight by Recent Price Volatility
    df['volatility'] = df['close'].rolling(window=5).std()
    weighted_price_momentum = price_momentum_score * (df['volatility'] / df['volatility'].max())
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['high_volume'] = (df['volume'] > df['avg_volume']).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    alpha_factor = weighted_price_momentum * df['high_volume']
    
    return alpha_factor
```
