```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Smooth High-Low Range (20-day moving average)
    short_term_range = daily_range.rolling(window=20).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    long_term_high = df['high'].rolling(window=50).max()
    long_term_low = df['low'].rolling(window=50).min()
    long_term_range = long_term_high - long_term_low
    
    # Compute Momentum of High-Low Ranges
    range_momentum = long_term_range - short_term_range
    
    # Weight by Volume
    factor = range_momentum * df['volume']
    
    return factor
```
