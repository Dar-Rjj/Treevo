```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['High-Low_Range'] = df['high'] - df['low']
    df['Close-Open_Change'] = df['close'] - df['open']

    # Aggregate Price Movements
    window = 20
    df['Sum_High-Low_Ranges'] = df['High-Low_Range'].rolling(window).sum()
    df['Sum_Close-Open_Changes'] = df['Close-Open_Change'].rolling(window).sum()

    # Smooth High-Low Range
    df['EMA_High-Low_Range'] = df['High-Low_Range'].ewm(span=window).mean()

    # Compute Momentum of High-Low Ranges
    short_window = 10
    long_window = 50
    df['Short_Term_Smoothed_Range'] = df['High-Low_Range'].ewm(span=short_window).mean()
    df['Long_Term_Smoothed_Range'] = df['High-Low_Range'].ewm(span=long_window).mean()
    df['Momentum_High-Low_Ranges'] = df['Long_Term_Smoothed_Range'] - df['Short_Term_Smoothed_Range']

    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['True_Range'] = df[['high', 'low', 'Prev_Close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['Prev_Close']), abs(x['low'] - x['Prev_Close'])), axis=1
    )

    # Calculate Rolling Average True Range
    df['ATR'] = df['True_Range'].rolling(window).sum()

    # Calculate Integrated Momentum Score
    df['Volatility'] = df['close'].rolling(window).std()
    df['Positive_Volumes'] = df['volume'].where(df['Close-Open_Change'] > 0, 0)
    df['Negative_Volumes'] = df['volume'].where(df['Close-Open_Change'] < 0, 0)
    df['Total_Abs_Volumes'] = df['volume'].abs().rolling(window).sum()
    df['Pos_Volume_Ratio'] = (df['Positive_Volumes'] / df['Total_Abs_Volumes']).rolling(window).sum()
    df['Neg_Volume_Ratio'] = (df['Negative_Volumes'] / df['Total_Abs_Volumes']).rolling(window).sum()
    df['Integrated_Momentum_Score'] = (df['Pos_Volume_Ratio'] - df['Neg_Volume_Ratio']) * df['Volatility']

    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_High-Low'] = df['High-Low_Range'] * df['volume']

    # Combine Volume-Adjusted High-Low with Open-Close
    df['Weighted_Average'] = (df['Volume_Adjusted_High-Low'] + df['Close-Open_Change']) / 2

    # Dynamic Momentum Indicator
    df['Previous_DMI'] = df['Weighted_Average'].shift(1)
    df['Current_DMI'] = df['Weighted_Average']
    df['Trend_Continuity'] = (df['Current_DMI'] - df['Previous_DMI']) / df['Volatility']

    # Evaluate Volume Activity
    df['Average_Volume'] = df['volume'].rolling(window).mean()
    high_volume_threshold = 1.5
    df['High_Volume'] = (df['volume'] > high_volume_threshold * df['Average_Volume']).astype(int)

    # Integrate Combined Factors
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    df['Integrated_Factor'] = df['Integrated_Momentum_Score'] * df['Volume_Ratio']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Trend_Continuity'] + df['Integrated_Factor'] + df['ATR']) / 3

    return df['Final_Alpha_Factor'].dropna()
```
