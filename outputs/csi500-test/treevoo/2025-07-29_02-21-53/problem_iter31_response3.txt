import pandas as pd

def heuristics_v2(df):
    # Compute the Simple Moving Average (SMA) of closing prices over a 14-day window
    df['SMA'] = df['close'].rolling(window=14).mean()
    
    # Compute the Exponential Moving Average (EMA) of closing prices over a 14-day window
    df['EMA'] = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the difference between EMA and SMA to capture short-term momentum
    df['Momentum'] = df['EMA'] - df['SMA']
    
    # Compute the Relative Strength Index (RSI) using the closing prices with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate the Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    df['%K'] = 100 * (df['close'] - low_14) / (high_14 - low_14)
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Compute the On-Balance Volume (OBV)
    df['OBV'] = ((df['close'] > df['close'].shift(1)) * df['volume']) - ((df['close'] < df['close'].shift(1)) * df['volume'])
    df['OBV'] = df['OBV'].cumsum()
    
    # Calculate the Chaikin Money Flow (CMF) over a 14-day window
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF'] = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Create a composite factor by summing individual factors
    df['Composite_Factor'] = df['Momentum'] + df['RSI'] + df['%K'] + df['%D'] + df['OBV'] + df['CMF']
    
    # Return the Composite Factor
    return df['Composite_Factor']
