import pandas as pd

def heuristics_v2(df, lookback_period=20, volume_threshold=1.5):
    # Compute High-Low Range
    df['High-Low_Range'] = df['high'] - df['low']
    
    # Compute Close-Open Change
    df['Close-Open_Change'] = df['close'] - df['open']
    
    # Aggregate Price Movements
    df['Sum_High-Low_Range'] = df['High-Low_Range'].rolling(window=lookback_period).sum()
    df['Sum_Close-Open_Change'] = df['Close-Open_Change'].rolling(window=lookback_period).sum()
    
    # Calculate Standard Deviation of Close Prices
    df['Close_STD'] = df['close'].rolling(window=lookback_period).std()
    
    # Adjust by Recent Price Volatility (Higher Weight to More Volatile Days)
    df['Weighted_High-Low_Range'] = df['Sum_High-Low_Range'] / df['Close_STD']
    df['Weighted_Close-Open_Change'] = df['Sum_Close-Open_Change'] / df['Close_STD']
    
    # Calculate Price Momentum Score
    df['Price_Momentum_Score'] = df['Weighted_High-Low_Range'] + df['Weighted_Close-Open_Change']
    
    # Evaluate Volume Activity
    df['Average_Volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['Is_High_Volume'] = (df['volume'] > df['Average_Volume'] * volume_threshold).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    df['Alpha_Factor'] = df['Price_Momentum_Score'] * df['Is_High_Volume']
    
    return df['Alpha_Factor'].dropna()
