```python
import pandas as pd

def heuristics_v2(df, period=10):
    # Calculate Intraday Price Movement
    high_low_range = df['high'] - df['low']
    close_open_change = df['close'] - df['open']

    # Aggregate Price Movements
    sum_high_low = high_low_range.rolling(window=period).sum()
    sum_close_open = close_open_change.rolling(window=period).sum()

    # Calculate Price Momentum Score
    weighted_sum = 0.5 * sum_high_low + 0.5 * sum_close_open
    recent_volatility = df['close'].rolling(window=period).std()
    adjusted_momentum = weighted_sum / (recent_volatility + 1e-6)  # Adding a small constant to avoid division by zero

    # Evaluate Volume Activity
    moving_avg_volume = df['volume'].rolling(window=period).mean()
    high_volume_days = (df['volume'] > moving_avg_volume).astype(int)

    # Integrate Price Momentum and Volume Activity
    price_momentum_and_volume = adjusted_momentum * high_volume_days

    # Adjust Alpha Factor by Recent Price Trend
    moving_avg_close = df['close'].rolling(window=period).mean()
    trend_multiplier = (df['close'] > moving_avg_close).astype(int)  # Positive or Negative
    alpha_factor = price_momentum_and_volume * trend_multiplier

    return alpha_factor
```
