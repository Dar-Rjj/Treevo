```python
import pandas as pd

def heuristics_v2(df, short_window=20, long_window=50, volume_spike_threshold=2.0):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period
    sum_high_low_diff = high_low_diff.rolling(window=short_window).sum()
    
    # Sum of Close-Open Differences over a Period
    sum_close_open_diff = close_open_diff.rolling(window=short_window).sum()
    
    # Smooth High-Low Range using Moving Average
    smoothed_high_low_range = high_low_diff.rolling(window=short_window).mean()
    
    # Calculate Long-Term High-Low Range
    long_term_high_low_range = (df['high'].rolling(window=long_window).max() - 
                                df['low'].rolling(window=long_window).min())
    
    # Subtract Short-Term Smoothed Range from Long-Term Range
    range_momentum = long_term_high_low_range - smoothed_high_low_range
    
    # Weight by Recent Price Volatility
    price_volatility = df['close'].rolling(window=short_window).std()
    weighted_momentum = (sum_high_low_diff + sum_close_open_diff) * price_volatility
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=short_window).mean()
    high_volume_indicator = (df['volume'] > volume_spike_threshold * avg_volume).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    combined_momentum = weighted_momentum + range_momentum
    final_alpha_factor = combined_momentum * high_volume_indicator
    
    return final_alpha_factor
```
