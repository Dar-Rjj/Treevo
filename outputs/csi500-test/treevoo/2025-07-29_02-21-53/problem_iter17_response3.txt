```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday and Long-Term Price Movements
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Smooth High-Low Range
    high_low_range_ema = high_low_diff.ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range
    long_term_high_low_range = (df['high'].rolling(window=50).max() - 
                                 df['low'].rolling(window=50).min())
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    sum_close_open_diff = close_open_diff.rolling(window=20).sum()
    high_low_range_ma = high_low_range_ema.rolling(window=20).mean()
    
    # Compute Combined Price and High-Low Range Momentum
    combined_momentum = (sum_high_low_diff + sum_close_open_diff +
                         (long_term_high_low_range - high_low_range_ema))
    recent_volatility = df['close'].rolling(window=20).std()
    weighted_momentum = combined_momentum * recent_volatility
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    high_volume_days = (volume_ratio > 1.5).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Factors
    momentum_with_volume = weighted_momentum * volume_ratio
    final_alpha_factor = (momentum_with_volume + 
                          (high_term_high_low_range - high_low_range_ema) * high_volume_days)
    
    # Consider Volume Weighted Price Change
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_vs_simple_avg = vwap - df['close'].rolling(window=20).mean()
    final_alpha_factor += vwap_vs_simple_avg
    
    return final_alpha_factor
```
