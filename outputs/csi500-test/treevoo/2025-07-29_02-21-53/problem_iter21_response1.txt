```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Smooth High-Low Range (Exponential Moving Average)
    df['ema_high_low_20'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    df['long_term_high_low_range'] = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    df['sum_high_low_diff_20'] = df['high_low_diff'].rolling(window=20).sum()
    df['sum_close_open_diff_20'] = df['close_open_diff'].rolling(window=20).sum()
    df['sma_high_low_20'] = df['high_low_diff'].rolling(window=20).mean()
    
    # Compute Combined Price and High-Low Range Momentum
    df['combined_momentum'] = df['sum_high_low_diff_20'] + df['sum_close_open_diff_20'] - (df['sma_high_low_20'] - df['ema_high_low_20'])
    
    # Weight by Recent Price Volatility (Standard Deviation of Close Prices)
    df['close_volatility'] = df['close'].rolling(window=20).std()
    df['weighted_momentum'] = df['combined_momentum'] * (df['close_volatility'] / df['close_volatility'].rolling(window=20).mean())
    
    # Evaluate Volume Activity
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['high_volume'] = df['volume_ratio'] > 1.5  # Threshold for high volume
    
    # Integrate Price, High-Low Range, and Volume Factors
    df['momentum_score'] = df['weighted_momentum'] * df['volume_ratio']
    
    # Adjust for Market Trends (Market Index Returns)
    market_index_return = df['close'].pct_change().rolling(window=20).mean()
    df['adjusted_alpha'] = df['momentum_score'] * (1 + market_index_return)
    
    # Consider Volume Weighted Price Change
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['final_alpha'] = df['adjusted_alpha'] * (df['vwap'] / df['close'])
    
    # Adjust Alpha Factor for Volume Spike
    df['volume_spike'] = df['volume'] / df['avg_volume_20']
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=10).mean()
    df['final_alpha'] = np.maximum(df['final_alpha'] * df['volume_momentum'], 0)
    
    return df['final_alpha']
```
