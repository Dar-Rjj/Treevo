```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    
    # Calculate Rolling Average True Range (ATR) with 14 days window
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Exponential Moving Average of Returns
    df['ema_12'] = df['daily_return'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['daily_return'].ewm(span=26, adjust=False).mean()
    
    # Calculate MACD
    df['macd'] = df['ema_12'] - df['ema_26']
    
    # Calculate Intraday Price Movement
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements over 10 days
    df['agg_high_low_diff'] = df['high_low_diff'].rolling(window=10).sum()
    df['agg_close_open_diff'] = df['close_open_diff'].rolling(window=10).sum()
    
    # Calculate Price Momentum Score
    close_volatility = df['close'].rolling(window=10).std()
    df['price_momentum_score'] = (df['agg_high_low_diff'] + df['agg_close_open_diff']) / close_volatility
    
    # Evaluate Volume Activity
    df['vol_avg'] = df['volume'].rolling(window=10).mean()
    df['high_volume_days'] = (df['volume'] > df['vol_avg']).astype(int)
    
    # Integrate Combined Factors
    df['integrated_factors'] = df['atr_14'] + df['price_momentum_score'] * df['high_volume_days'] + df['macd']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['integrated_factors']
    
    return df['alpha_factor']
```
