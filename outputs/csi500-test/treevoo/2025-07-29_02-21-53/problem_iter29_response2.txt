```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Sum of returns over a lookback window of N days
    N = 5
    df['N_day_return'] = df['daily_return'].rolling(window=N).sum()
    
    # True range
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    
    # Average true range (ATR) over M days
    M = 14
    df['ATR'] = df['true_range'].rolling(window=M).mean()
    
    # Relative ATR
    df['relative_ATR'] = df['ATR'] / df['ATR'].rolling(window=2 * M).mean()
    
    # Directional ATR
    df['directional_ATR'] = df['ATR'] * df['daily_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Volume intensity
    L = 20
    df['volume_intensity'] = df['volume'] / df['volume'].rolling(window=L).mean()
    
    # Money flow
    P = 10
    df['money_flow'] = df['volume'] * (2 * df['close'] - df['high'] - df['low'])
    df['money_flow_sum'] = df['money_flow'].rolling(window=P).sum()
    
    # Composite factor
    df['composite_factor'] = (
        0.4 * df['N_day_return'] + 
        0.3 * df['relative_ATR'] + 
        0.3 * df['volume_intensity']
    )
    
    # Volume-Adjusted True Range
    df['VATR'] = df['true_range'] * df['volume']
    
    # Smoothed High-Low Ranges
    df['smoothed_HL'] = df['high'] - df['low']
    df['smoothed_HL'] = df['smoothed_HL'].ewm(span=20, min_periods=20).mean()
    
    # Interaction term
    df['interaction_term'] = df['smoothed_HL'] * (df['high'] - df['low'])
    
    # Price Momentum
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['price_momentum'] = df['5_day_MA'] - df['20_day_MA']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['composite_factor'] + 
        df['VATR'] + 
        df['interaction_term'] + 
        df['price_momentum']
    )
    
    return df['final_alpha_factor']
```
