```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-day and 30-day close price averages
    df['10_day_avg'] = df['close'].rolling(window=10).mean()
    df['30_day_avg'] = df['close'].rolling(window=30).mean()
    
    # Determine price momentum
    df['momentum'] = df['10_day_avg'] - df['30_day_avg']
    
    # Detect volume spike
    df['5_day_vol_avg'] = df['volume'].rolling(window=5).mean()
    df['spike_detected'] = df['volume'] > 2 * df['5_day_vol_avg']
    
    # Adjust momentum based on spike
    df['adjusted_momentum'] = df.apply(lambda row: row['momentum'] * 0.8 if row['spike_detected'] else row['momentum'], axis=1)
    
    return df['adjusted_momentum']
```
