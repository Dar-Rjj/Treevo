```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Signed Transaction Volume
    df['high_low_spread'] = df['high'] - df['low']
    df['signed_volume'] = df['volume'] * df['high_low_spread'].apply(lambda x: 1 if x > 0 else -1)

    # Calculate Accumulated Buy-Sell Imbalance
    positive_volumes = df[df['signed_volume'] > 0]['signed_volume']
    negative_volumes = df[df['signed_volume'] < 0]['signed_volume'].abs()
    df['accumulated_positive_vol'] = positive_volumes.rolling(window=10, min_periods=1).sum()
    df['accumulated_negative_vol'] = negative_volumes.rolling(window=10, min_periods=1).sum()

    # Compute Order Imbalance Score
    df['total_absolute_vol'] = df['accumulated_positive_vol'] + df['accumulated_negative_vol']
    df['order_imbalance_score'] = (df['accumulated_positive_vol'] / df['total_absolute_vol']) - (df['accumulated_negative_vol'] / df['total_absolute_vol'])

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range_high_low'] = df['high'] - df['low']
    df['true_range_high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['true_range_low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['true_range_high_low', 'true_range_high_prev_close', 'true_range_low_prev_close']].max(axis=1)

    # Calculate Rolling Average True Range
    df['avg_true_range'] = df['true_range'].rolling(window=14, min_periods=1).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(periods=14)

    # Combine ATR and Price Momentum with Order Imbalance
    df['normalized_price_momentum'] = df['price_momentum'] / df['avg_true_range']
    df['alpha_factor'] = df['normalized_price_momentum'] * df['order_imbalance_score']

    return df['alpha_factor']
```
