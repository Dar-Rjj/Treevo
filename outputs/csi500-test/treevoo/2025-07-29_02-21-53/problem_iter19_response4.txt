```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Smooth High-Low Range with 20-day EMA
    df['high_low_ema_20'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    df['long_high_low_range'] = df['high_low_diff'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    df['sum_high_low_diff_20'] = df['high_low_diff'].rolling(window=20).sum()
    df['sum_close_open_diff_20'] = df['close_open_diff'].rolling(window=20).sum()
    
    # Compute Combined Price and High-Low Range Momentum
    df['combined_momentum'] = (df['sum_high_low_diff_20'] + df['sum_close_open_diff_20']) - (df['long_high_low_range'] - df['high_low_ema_20'])
    
    # Weight by Recent Price Volatility (Standard Deviation of Close Prices over 20 days)
    df['close_std_20'] = df['close'].rolling(window=20).std()
    df['weighted_momentum'] = df['combined_momentum'] * df['close_std_20']
    
    # Evaluate Volume Activity
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = df['volume'] > 1.5 * df['avg_volume_20']
    
    # Integrate Price, High-Low Range, and Volume Factors
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['weighted_momentum_vol'] = df['weighted_momentum'] * df['volume_ratio']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['weighted_momentum_vol']
    
    # Adjust for Market Trends
    df['market_return'] = df['close'].pct_change()
    df['final_alpha_factor'] = df['final_alpha_factor'] * (1 + df['market_return'])
    
    # Consider Volume Weighted Price Change
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['final_alpha_factor'] = df['final_alpha_factor'] * (df['close'] / df['vwap'])
    
    # Adjust Alpha Factor for Volume Spike
    df['volume_spike'] = df['volume'] / df['avg_volume_20']
    df['final_alpha_factor'] = df['final_alpha_factor'] * (1 + df['volume_spike'])
    
    return df['final_alpha_factor']
```
