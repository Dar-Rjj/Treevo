import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with 20-day EMA
    ema_20_high_low = high_low_range.ewm(span=20, adjust=False).mean()
    
    # Compute Momentum of High-Low Ranges
    short_term_ema = ema_20_high_low.ewm(span=10, adjust=False).mean()
    momentum_high_low = ema_20_high_low - short_term_ema
    
    # Weight by Volume
    volume_weighted_momentum = momentum_high_low * df['volume']
    
    # Integrate True Range
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Analyze Intraday Price Movements
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    sign_high_low = (df['high'] - df['low']).apply(lambda x: 1 if x > 0 else -1)
    intraday_price_movement = (high_low_diff + close_open_diff) * sign_high_low
    
    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    sum_close_open_diff = close_open_diff.rolling(window=20).sum()
    positive_volume = df['volume'].where(df['close'] > df['open'], other=0)
    negative_volume = df['volume'].where(df['close'] <= df['open'], other=0)
    sum_positive_volume = positive_volume.rolling(window=20).sum()
    sum_negative_volume = negative_volume.rolling(window=20).sum()
    
    # Calculate Integrated Momentum Score
    price_volatility = df['close'].rolling(window=20).std()
    weighted_high_low_diff = sum_high_low_diff / price_volatility
    weighted_close_open_diff = sum_close_open_diff / price_volatility
    total_absolute_volume = (positive_volume.abs() + negative_volume.abs()).rolling(window=20).sum()
    ratio_positive_volume = sum_positive_volume / total_absolute_volume
    ratio_negative_volume = sum_negative_volume / total_absolute_volume
    integrated_momentum_score = (weighted_high_low_diff + weighted_close_open_diff) + ratio_positive_volume - ratio_negative_volume
    
    # Adjust for Volume Impact on Price
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Combine Volume-Adjusted High-Low with Open-Close
    combined_measure = (volume_adjusted_high_low + close_open_diff) / 2
    
    # Dynamic Momentum Indicator
    yesterday_indicator = integrated_momentum_score.shift(1)
    today_indicator = integrated_momentum_score
    direction_change = (today_indicator - yesterday_indicator).apply(lambda x: 1 if x > 0 else -1)
    smoothed_indicator = direction_change * integrated_momentum_score.ewm(span=20, adjust=False).mean()
    
    # Generate Final Alpha Factor
    final_alpha_factor = (true_range + volume_weighted_momentum + smoothed_indicator) * combined_measure
    
    return final_alpha_factor
