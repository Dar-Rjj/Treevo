```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Smooth High-Low Range with 20-day EMA
    smoothed_high_low_range = high_low_diff.ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    long_term_high_low_range = (df['high'].rolling(window=50).max() - 
                                df['low'].rolling(window=50).min())
    
    # Aggregate Price Movements over a period
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    sum_close_open_diff = close_open_diff.rolling(window=20).sum()
    
    # Combined Price and High-Low Range Momentum
    combined_momentum = (sum_high_low_diff + sum_close_open_diff) - \
                        (smoothed_high_low_range - long_term_high_low_range)
    
    # Weight by Recent Price Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    weighted_momentum = combined_momentum * (recent_volatility / recent_volatility.mean())
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    high_volume_days = (df['volume'] > avg_volume * 1.5).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Factors
    volume_ratio = df['volume'] / avg_volume
    integrated_factor = weighted_momentum * volume_ratio + smoothed_high_low_range
    
    # Adjust for Market Trends and Volume Spike Impact
    market_index_return = df['close'].pct_change().rolling(window=20).mean()
    adjusted_factor = integrated_factor + market_index_return
    
    # Consider Volume Weighted Price Change
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    simple_avg = df['close'].rolling(window=20).mean()
    final_alpha_factor = adjusted_factor + (vwap - simple_avg)
    
    return final_alpha_factor
```
