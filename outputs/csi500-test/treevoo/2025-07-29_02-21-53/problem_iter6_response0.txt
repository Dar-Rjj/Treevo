```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period (e.g., 20 days)
    df['sum_high_low_diff_20'] = df['high_low_diff'].rolling(window=20).sum()
    
    # Sum of Close-Open Differences over a Period (e.g., 20 days)
    df['sum_close_open_diff_20'] = df['close_open_diff'].rolling(window=20).sum()
    
    # Smooth High-Low Range with Exponential Moving Average (e.g., 20-day)
    df['smooth_high_low_range'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (e.g., 50-day)
    df['long_term_high_low_range'] = df['high_low_diff'].rolling(window=50).max()
    
    # Compare Aggregate Price Movements and High-Low Ranges
    df['combined_momentum'] = df['sum_high_low_diff_20'] + df['sum_close_open_diff_20'] - (df['smooth_high_low_range'] - df['long_term_high_low_range'])
    
    # Weight by Recent Price Volatility (e.g., 20-day standard deviation)
    df['price_volatility'] = df['close'].rolling(window=20).std()
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['price_volatility']
    
    # Compute Average Volume with Exponential Moving Average (e.g., 20-day)
    df['avg_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Identify Days with High Volume
    df['high_volume_indicator'] = (df['volume'] > df['avg_volume']).astype(int)
    
    # Multiply Combined Momentum Score by High Volume Indicator
    df['alpha_factor'] = df['weighted_combined_momentum'] * df['high_volume_indicator']
    
    return df['alpha_factor']
```
