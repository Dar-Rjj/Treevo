```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with 20-day Exponential Moving Average
    smoothed_high_low_range = daily_high_low_range.ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    long_term_high_low_range = (df['high'].rolling(window=50).max() - 
                                df['low'].rolling(window=50).min())
    
    # Compute Momentum of High-Low Ranges
    high_low_momentum = long_term_high_low_range - smoothed_high_low_range
    
    # Weight by Volume
    weighted_momentum = high_low_momentum * df['volume']
    
    # Integrate Price Change
    daily_price_change = df['close'] - df['open']
    
    # Smooth Price Change with 20-day Exponential Moving Average
    smoothed_price_change = daily_price_change.ewm(span=20).mean()
    
    # Combine with High-Low Range Momentum
    final_factor = smoothed_price_change + weighted_momentum
    
    return final_factor
```
