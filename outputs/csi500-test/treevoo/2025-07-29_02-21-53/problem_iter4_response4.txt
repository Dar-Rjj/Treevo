```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements over a Period (e.g., 10 days)
    period = 10
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=period).sum()
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=period).sum()
    
    # Calculate Price Momentum Score
    df['price_momentum_score'] = (df['sum_high_low_diff'] + df['sum_close_open_diff']) / 2
    
    # Weight by Recent Price Volatility
    df['close_std'] = df['close'].rolling(window=period).std()
    df['price_momentum_score_weighted'] = df['price_momentum_score'] * df['close_std']
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=period).mean()
    df['high_volume'] = (df['volume'] > df['avg_volume']).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    df['alpha_factor'] = df['price_momentum_score_weighted'] * df['high_volume']
    
    return df['alpha_factor']
```
