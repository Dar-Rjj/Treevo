def heuristics_v2(df):
    # Calculate 5-Day, 10-Day, and 20-Day Simple Moving Averages (SMA) of Close Prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Compute the difference between 10-Day SMA and 20-Day SMA as a Momentum Indicator
    df['Momentum'] = df['SMA_10'] - df['SMA_20']

    # Identify Days Where Close Price Exceeds Highest High or Lowest Low over Last 30 Days
    df['HH_30'] = df['high'].rolling(window=30).max()
    df['LL_30'] = df['low'].rolling(window=30).min()
    df['Breakout_Score'] = 0
    df.loc[df['close'] > df['HH_30'], 'Breakout_Score'] = 1
    df.loc[df['close'] < df['LL_30'], 'Breakout_Score'] = -1

    df['Volume_Change'] = df['volume'].pct_change()

    df['Large_Volume_Price_Move'] = ((df['Volume_Change'] > 0.1) & (df['close'].pct_change() > 0.02)).astype(int)

    df.loc[(df['open'] < df['close']) & (df['open'] < df.shift(1)['close']), 'Price_Behavior'] = 1

    df['Amount_to_Volume_Ratio'] = df['amount'] / df['volume']

    df['Ratio_MA_5'] = df['Amount_to_Volume_Ratio'].rolling(window=5).mean()
    df['Ratio_MA_10'] = df['Amount_to_Volume_Ratio'].rolling(window=10).mean()

    df['Ratio_Crossover'] = (df['Ratio_MA_5'] > df['Ratio_MA_10']).astype(int)
