```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 10-Day Volume Surge
    df['10_day_volume_surge'] = df['volume'] - df['volume'].shift(10)
    
    # Combine Price Momentum and Volume Surge
    df['factor'] = df['5_day_momentum'] * df['10_day_volume_surge']
    
    return df['factor']
```
