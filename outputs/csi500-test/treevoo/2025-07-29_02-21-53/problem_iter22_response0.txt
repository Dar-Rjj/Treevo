import pandas as pd

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df[['true_range', 'prev_close', 'high', 'low']].apply(
        lambda x: max(x[0], abs(x[1] - x[2]), abs(x[1] - x[3])), axis=1)
    
    # Multiply True Range by Volume
    df['volume_adjusted_true_range'] = df['true_range'] * df['volume']
    
    # Apply Exponential Moving Average (20-day) to Daily High-Low Differences
    df['high_low_diff'] = df['high'] - df['low']
    df['smoothed_high_low_range'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Combine Volume-Adjusted True Range and Smoothed High-Low Ranges
    df['combined_measures'] = df['volume_adjusted_true_range'] + df['smoothed_high_low_range']
    
    # High Volume Indicator (1 if volume is above its 20-day moving average, else 0)
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['high_volume_indicator'] = (df['volume'] > df['volume_ma']).astype(int)
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_measures'] * df['high_volume_indicator'] + \
                         (df['smoothed_high_low_range'] * df['high_low_diff'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
