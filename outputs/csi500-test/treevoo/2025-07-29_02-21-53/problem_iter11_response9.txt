```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low price (price range)
    price_range = df['high'] - df['low']
    
    # Calculate the ratio of close to open price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the transaction amount per volume
    amount_per_volume = df['amount'] / df['volume']
    
    # Generate the alpha factor by combining the above three components
    # The idea is that large price ranges with a closing price higher than the opening,
    # combined with high transaction amounts relative to volume, may indicate positive sentiment.
    alpha_factor = (price_range * close_open_ratio * amount_per_volume).rank(pct=True)
    
    return alpha_factor
```
