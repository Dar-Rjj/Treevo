import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range (20-day EMA)
    smoothed_high_low_range = daily_high_low_range.ewm(span=20).mean()
    
    # Compute Momentum of High-Low Ranges
    long_term_range = daily_high_low_range.rolling(window=20).mean()
    high_low_momentum = long_term_range - smoothed_high_low_range
    
    # Weight by Volume
    weighted_momentum = high_low_momentum * df['volume']
    
    # Integrate Price Change
    daily_price_change = df['close'] - df['open']
    smoothed_price_change = daily_price_change.ewm(span=20).mean()
    combined_factor = weighted_momentum + smoothed_price_change
    
    # Integrate Price Trend
    long_term_price_trend = df['close'].rolling(window=100).mean()
    smoothed_long_term_price_trend = long_term_price_trend.ewm(span=100).mean()
    price_trend_momentum = long_term_price_trend - smoothed_price_change
    
    # Combine with High-Low Range and Volume Momentum
    final_factor = combined_factor + price_trend_momentum
    
    return final_factor
