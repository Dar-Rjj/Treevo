```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Smooth High-Low Range with 20-day EMA
    high_low_smoothed = high_low_diff.ewm(span=20).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    long_term_high = df['high'].rolling(window=50).max()
    long_term_low = df['low'].rolling(window=50).min()
    long_term_range = long_term_high - long_term_low
    
    # Momentum of High-Low Ranges
    high_low_momentum = long_term_range - high_low_smoothed
    
    # Sum of High-Low Differences over a Period (e.g., 20 days)
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    
    # Sum of Close-Open Differences over a Period (e.g., 20 days)
    sum_close_open_diff = close_open_diff.rolling(window=20).sum()
    
    # Price Momentum Score
    price_momentum_score = sum_high_low_diff + sum_close_open_diff
    recent_volatility = df['close'].rolling(window=20).std()
    price_momentum_score_weighted = price_momentum_score / recent_volatility
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    high_volume_days = df['volume'] > avg_volume * 1.5  # Threshold for high volume
    
    # Volume Ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Integrate Price and Volume Factors
    price_volume_factor = price_momentum_score_weighted * volume_ratio
    final_alpha_factor = price_volume_factor + high_low_momentum
    
    return final_alpha_factor
```
