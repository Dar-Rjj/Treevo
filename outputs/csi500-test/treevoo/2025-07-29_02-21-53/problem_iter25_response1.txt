```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range using High, Low, and Previous Close
    df['TrueRange'] = df[['High', 'Low', 'Close']].join(df['Close'].shift(1), rsuffix='_Prev').apply(
        lambda x: max(x['High'] - x['Low'], abs(x['High'] - x['Close_Prev']), abs(x['Low'] - x['Close_Prev'])), axis=1)
    
    # Compute Intraday Price Movement using High, Low, Open, and Close
    df['IntradayMovement'] = (df['Close'] - df['Open'])
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_TrueRange'] = df['TrueRange'] * df['Volume']
    df['Volume_Adjusted_IntradayMovement'] = df['IntradayMovement'] * df['Volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['High_Low_Range'] = df['High'] - df['Low']
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=20).mean()
    df['Smoothed_Price_Movement'] = df['IntradayMovement'].ewm(span=20).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Combined_Momentum'] = df['Smoothed_High_Low_Range'] - df['Smoothed_Price_Movement']
    df['Combined_Momentum'] *= df['TrueRange'].rolling(window=20).std()
    
    # Evaluate Volume Activity
    df['Smoothed_Volume'] = df['Volume'].ewm(span=20).mean()
    df['High_Volume_Indicator'] = (df['Volume'] > df['Smoothed_Volume']).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['Integrated_Measure'] = (df['Volume_Adjusted_TrueRange'] + df['Volume_Adjusted_IntradayMovement']) * df['High_Volume_Indicator']
    df['Trend_Continuity'] = df['Combined_Momentum'] * df['High_Low_Range']
    
    # Generate Final Alpha Factor
    df['Alpha_Factor'] = df['Integrated_Measure'] + df['Trend_Continuity']
    
    return df['Alpha_Factor']
```
