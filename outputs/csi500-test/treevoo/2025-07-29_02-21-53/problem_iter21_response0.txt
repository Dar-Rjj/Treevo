```python
import pandas as pd

def heuristics_v2(df):
    # Momentum indicators
    short_mavg = df['close'].rolling(window=5).mean()
    long_mavg = df['close'].rolling(window=20).mean()
    momentum = short_mavg - long_mavg
    
    # Volatility measures
    daily_returns = df['close'].pct_change()
    volatility_std = daily_returns.rolling(window=10).std()
    daily_range = df['high'] - df['low']
    avg_range = daily_range.rolling(window=10).mean()
    
    # Volume-based alpha factors
    volume_10dma = df['volume'].rolling(window=10).mean()
    vol_ratio = df['volume'] / volume_10dma
    
    # Price-volume interactions
    price_changes = df['close'].diff()
    vol_changes = df['volume'].diff()
    corr_price_vol = df['close'].rolling(window=10).corr(df['volume'])
    abnormal_vol_days = (df['volume'] > 2 * volume_10dma).rolling(window=30).sum()
    
    # Intra-day price patterns
    intra_day_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    positive_close_open = ((df['close'] - df['open']) > 0).astype(int)
    sum_positive_c_o = (df['close'] - df['open']).apply(lambda x: max(0, x)).rolling(window=10).sum()
    sum_negative_c_o = (df['close'] - df['open']).apply(lambda x: min(0, x)).rolling(window=10).sum()
    up_days = (df['close'] > df['open']).rolling(window=10).sum()
    down_days = (df['close'] < df['open']).rolling(window=10).sum()
    
    # Combine features into a single alpha factor
    alpha_factor = (
        0.3 * momentum +
        0.2 * (volatility_std + avg_range) +
        0.1 * vol_ratio +
        0.1 * corr_price_vol +
        0.1 * (abnormal_vol_days + up_days - down_days) +
        0.2 * (sum_positive_c_o - sum_negative_c_o)
    )
    
    return alpha_factor
```
