```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    
    # Aggregate Price Movements over a period (e.g., 10 days)
    high_low_sum = high_low_diff.rolling(window=10).sum()
    close_open_sum = close_open_diff.rolling(window=10).sum()
    
    # Calculate Price Momentum Score
    price_momentum = (high_low_sum + close_open_sum) / 2
    std_close = df['close'].rolling(window=10).std()
    price_momentum *= std_close
    
    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=10).mean()
    high_volume_indicator = (df['volume'] > avg_volume * 1.5).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    intermediate_alpha = price_momentum * high_volume_indicator
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    true_range = pd.concat(
        [df['high'] - df['low'], 
         abs(df['high'] - df['prev_close']), 
         abs(df['low'] - df['prev_close'])], 
        axis=1).max(axis=1)
    
    # Calculate Rolling Average True Range
    atr = true_range.rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_change = df['close'] - df['close'].shift(10)
    
    # Combine ATR, Price, and Volume Effects
    final_alpha = (price_change / atr) * intermediate_alpha
    
    return final_alpha
```
