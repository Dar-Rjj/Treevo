```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Range and Close-Open Change
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']

    # Aggregate Price Movements over a 20-day period
    df['sum_high_low_range'] = df['high_low_range'].rolling(window=20).sum()
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=20).sum()

    # Smooth High-Low Range with 20-day EMA
    df['smoothed_high_low_range'] = df['high_low_range'].ewm(span=20, adjust=False).mean()

    # Calculate Long-Term High-Low Range (50-day)
    df['long_term_high'] = df['high'].rolling(window=50).max()
    df['long_term_low'] = df['low'].rolling(window=50).min()
    df['long_term_high_low_range'] = df['long_term_high'] - df['long_term_low']

    # Compute Momentum of High-Low Ranges
    df['range_momentum'] = df['long_term_high_low_range'] - df['smoothed_high_low_range']

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['prev_close']).abs()
    df['tr3'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Rolling Average True Range (ATR) over 20 days
    df['atr_20'] = df['true_range'].rolling(window=20).mean()

    # Adjust Price Momentum by ATR
    df['adjusted_momentum'] = (df['sum_high_low_range'] + df['sum_close_open_change']) / df['atr_20']

    # Evaluate Volume Activity
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = (df['volume'] > 1.5 * df['avg_volume_20']).astype(int)

    # Detect Volume Spike
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 2 * df['avg_volume_5']).astype(int)

    # Adjust Momentum Based on Volume
    df['dampened_momentum'] = df['adjusted_momentum'] * (1 - 0.2 * df['volume_spike'])

    # Integrate Combined Factors
    df['interim_alpha_factor'] = df['dampened_momentum'] * df['high_volume']

    # Combine ATR and Price Momentum
    df['final_alpha_factor'] = df['interim_alpha_factor'] + df['range_momentum']

    # Incorporate Recent Price Trend
    df['moving_avg_close_20'] = df['close'].rolling(window=20).mean()
    df['recent_trend'] = np.where(df['close'] > df['moving_avg_close_20'], 1, 0.8)

    # Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['recent_trend']

    return df['final_alpha_factor']
```
