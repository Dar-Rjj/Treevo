```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference and Close-Open Change
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']
    
    # Sum of High-Low Differences over a Period (e.g., 20 days)
    df['sum_high_low_diff'] = df['high_low_diff'].rolling(window=20).sum()
    
    # Sum of Close-Open Changes over a Period (e.g., 20 days)
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=20).sum()
    
    # Smooth High-Low Range with EMA (e.g., 20-day)
    df['smoothed_high_low_range'] = df['high_low_diff'].ewm(span=20, adjust=False).mean()
    
    # Long-Term High-Low Range (e.g., 50-day)
    df['long_term_high_low_range'] = df['high_low_diff'].rolling(window=50).max() - df['high_low_diff'].rolling(window=50).min()
    
    # Momentum of High-Low Ranges
    df['high_low_momentum'] = df['long_term_high_low_range'] - df['smoothed_high_low_range']
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high' - 'low', 'high' - 'prev_close', 'prev_close' - 'low']].max(axis=1)
    
    # Calculate Rolling Average True Range (ATR) (e.g., 14-day)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Adjust Price Momentum by ATR
    df['adjusted_price_momentum'] = (df['sum_high_low_diff'] + df['sum_close_open_change']) / df['atr']
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = df['volume'] > df['avg_volume'] * 1.5
    
    # Detect Volume Spike
    df['5_day_vol_avg'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = df['volume'] > 2 * df['5_day_vol_avg']
    
    # Adjust Momentum Based on Volume
    df['dampened_momentum'] = df['adjusted_price_momentum'] * df['high_volume']
    df['dampened_momentum'] = df.apply(lambda x: x['dampened_momentum'] * 0.75 if x['volume_spike'] else x['dampened_momentum'], axis=1)
    
    # Integrate Combined Factors
    df['interim_alpha_factor'] = df['dampened_momentum'] * df['high_volume']
    
    # Combine ATR and Price Momentum
    df['interpretable_factor'] = df['interim_alpha_factor'] + df['high_low_momentum']
    
    # Incorporate Recent Price Trend
    df['recent_price_trend'] = df['close'].rolling(window=10).mean()
    df['price_trend_multiplier'] = df['recent_price_trend'].apply(lambda x: 1.2 if x > 0 else 0.8)
    
    # Introduce New Factor: Volume-Weighted ATR
    high_volume_days = df[df['high_volume']]
    high_volume_atr = high_volume_days['true_range'].rolling(window=14).mean()
    df['volume_weighted_atr'] = df['atr'] * df['high_volume']
    df['volume_weighted_atr'].fillna(method='ffill', inplace=True)
    
    # Adjust Interim Alpha Factor
    df['adjusted_alpha_factor'] = df['interpretable_factor'] / df['volume_weighted_atr']
    
    # Introduce New Factor: Volume-Weighted Price Momentum
    high_volume_price_momentum = high_volume_days['close_open_change'].rolling(window=20).sum()
    df['volume_weighted_price_momentum'] = df['close_open_change'] * df['high_volume']
    df['volume_weighted_price_momentum'].fillna(method='ffill', inplace=True)
    
    # Adjust Final Alpha Factor
    df['final_adjusted_alpha_factor'] = df['adjusted_alpha_factor'] * df['volume_weighted_price_momentum']
    
    return df['final_adjusted_alpha_factor']
```
