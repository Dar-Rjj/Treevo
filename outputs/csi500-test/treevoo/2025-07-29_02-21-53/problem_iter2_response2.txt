```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['True_Range'] = df[['high', 'low', 'Prev_Close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['Prev_Close']), abs(x['low'] - x['Prev_Close'])), axis=1)

    # Calculate Rolling Average True Range (ATR) over 14 days
    N_ATR = 14
    df['ATR'] = df['True_Range'].rolling(window=N_ATR).mean()

    # Calculate Intraday Price Movement
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Aggregate Price Movements over a period of 10 days
    N_Aggregate = 10
    df['Sum_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=N_Aggregate).sum()
    df['Sum_Close_Open_Diff'] = df['Close_Open_Diff'].rolling(window=N_Aggregate).sum()

    # Calculate Price Momentum over 10 days
    N_Momentum = 10
    df['Price_Momentum'] = df['close'] - df['close'].shift(N_Momentum)

    # Combine ATR and Price Momentum
    df['Normalized_Price_Momentum'] = df['Price_Momentum'] / df['ATR']

    # Evaluate Volume Activity
    df['Average_Volume'] = df['volume'].rolling(window=N_ATR).mean()
    df['High_Volume_Indicator'] = (df['volume'] > df['Average_Volume']).astype(int)

    # Integrate Combined Factors
    df['Alpha_Factor'] = df['Normalized_Price_Momentum'] * df['High_Volume_Indicator']

    return df['Alpha_Factor']
```
