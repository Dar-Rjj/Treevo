```python
import pandas as pd

def heuristics_v2(df, period=10, atr_period=14):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Aggregate Price Movements
    sum_high_low_diff = high_low_diff.rolling(window=period).sum()
    sum_close_open_diff = close_open_diff.rolling(window=period).sum()
    
    # Calculate Integrated Momentum Score
    momentum_score = sum_high_low_diff + sum_close_open_diff
    std_close = df['close'].rolling(window=period).std()
    weighted_momentum_score = momentum_score * (std_close / std_close.rolling(window=period).mean())
    
    # Identify Days with High Volume
    avg_volume = df['volume'].rolling(window=period).mean()
    high_volume_days = (df['volume'] > avg_volume).astype(int)
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Rolling Average True Range
    atr = true_range.rolling(window=atr_period).mean()
    
    # Integrate Price, Volume, and ATR
    volume_ratio = df['volume'] / avg_volume
    normalized_momentum = weighted_momentum_score / atr
    integrated_factor = normalized_momentum * volume_ratio
    
    return integrated_factor
```
