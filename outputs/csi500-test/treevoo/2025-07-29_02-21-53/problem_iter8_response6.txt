```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']

    # Aggregate Price Movements over a 10-day period
    df['sum_high_low_range'] = df['high_low_range'].rolling(window=10).sum()
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=10).sum()

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['prev_close']).abs()
    df['tr3'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Rolling Average True Range (ATR) over 14 days
    df['atr'] = df['true_range'].rolling(window=14).mean()

    # Adjust Price Momentum by ATR
    df['weighted_high_low'] = df['sum_high_low_range'] / df['atr']
    df['weighted_close_open'] = df['sum_close_open_change'] / df['atr']
    df['adjusted_momentum'] = (df['weighted_high_low'] + df['weighted_close_open']) / 2

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['high_volume'] = (df['volume'] > df['avg_volume'] * 1.5).astype(int)

    # Detect Volume Spike
    df['5d_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 2 * df['5d_avg_volume']).astype(int)

    # Adjust Momentum Based on Volume
    df['adjusted_momentum'] = df.apply(lambda row: row['adjusted_momentum'] * 0.8 if row['volume_spike'] == 1 else row['adjusted_momentum'], axis=1)

    # Integrate Combined Factors
    df['final_alpha_factor'] = df['adjusted_momentum'] * df['high_volume']

    # Incorporate Recent Price Trend
    df['ma_close'] = df['close'].rolling(window=10).mean()
    df['recent_trend_multiplier'] = (df['close'] - df['ma_close']).apply(lambda x: 1.1 if x > 0 else 0.9)
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['recent_trend_multiplier']

    return df['final_alpha_factor']
```
