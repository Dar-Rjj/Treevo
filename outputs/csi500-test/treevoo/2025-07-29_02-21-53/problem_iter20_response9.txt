import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range
    df['TrueRange'] = df[['High', 'Low']].apply(lambda x: x['High'] - x['Low'], axis=1)
    df['TrueRange'] = df.apply(lambda x: max(x['TrueRange'], abs(x['High'] - x['Close'].shift(1)), abs(x['Low'] - x['Close'].shift(1))), axis=1)
    
    # Compute Intraday Price Movement
    df['IntradayPriceMovement'] = (df['High'] - df['Low']) + (df['Close'] - df['Open'])
    
    # Adjust for Volume Impact on Price
    df['VolumeAdjustedTrueRange'] = df['TrueRange'] * df['Volume']
    df['VolumeAdjustedIntradayPriceMovement'] = df['IntradayPriceMovement'] * df['Volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    df['SmoothedHighLowRange'] = df['High'] - df['Low']
    df['SmoothedPriceMovement'] = df['Close'] - df['Open']
    df['SmoothedHighLowRangeEMA'] = df['SmoothedHighLowRange'].ewm(span=20).mean()
    df['SmoothedPriceMovementEMA'] = df['SmoothedPriceMovement'].ewm(span=20).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['CombinedMomentum'] = df['SmoothedPriceMovementEMA'] - df['SmoothedHighLowRangeEMA']
    df['PriceVolatility'] = df['Close'].pct_change().rolling(window=20).std()
    df['WeightedCombinedMomentum'] = df['CombinedMomentum'] / df['PriceVolatility']
    
    # Evaluate Volume Activity
    df['VolumeEMA'] = df['Volume'].ewm(span=20).mean()
    df['HighVolumeIndicator'] = (df['Volume'] > df['VolumeEMA']).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    df['CombinedMeasure'] = (df['VolumeAdjustedTrueRange'] + df['VolumeAdjustedIntradayPriceMovement']) * df['HighVolumeIndicator']
    df['TrendContinuityReversal'] = df['CombinedMomentum'] * (df['High'] - df['Low'])
    
    # Generate Final Alpha Factor
    df['AlphaFactor'] = df['CombinedMeasure'] + df['TrendContinuityReversal']
    
    return df['AlphaFactor']
