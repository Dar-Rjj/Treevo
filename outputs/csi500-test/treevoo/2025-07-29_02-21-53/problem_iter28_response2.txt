```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Differences
    df['open_to_close'] = df['close'] - df['open']
    df['high_to_low'] = df['high'] - df['low']
    df['day_to_day_close'] = df['close'].diff()

    # Identify Trend Indicators
    short_window = 5
    long_window = 20
    df['short_moving_avg'] = df['close'].rolling(window=short_window).mean()
    df['long_moving_avg'] = df['close'].rolling(window=long_window).mean()
    df['price_momentum'] = df['close'].pct_change()
    df['rate_of_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # RSI without normalization
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Volume Trends
    df['avg_volume'] = df['volume'].rolling(window=14).mean()
    df['volume_change'] = df['volume'].diff()

    # Amount Analysis
    df['total_amount_traded'] = df['amount']
    df['amount_change'] = df['amount'].diff()

    # Combine Price and Volume
    df['obv'] = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0) * df['volume']
    df['obv'] = df['obv'].cumsum()

    # Money Flow Index (MFI) without normalization
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_flow_14 = positive_flow.rolling(window=14).sum()
    negative_flow_14 = negative_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (positive_flow_14 / negative_flow_14)))

    # Detect Market Sentiment
    df['high_low_spread'] = df['high'] - df['low']
    df['bollinger_bands_width'] = df['close'].rolling(window=20).std() * 2
    df['atr'] = (df['high'] - df['low']).rolling(window=14).mean()

    # Evaluate Technical Indicators
    df['stochastic_oscillator_k'] = (df['close'] - df['low'].rolling(window=14).min()) / (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())
    df['macd_line'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()

    # Consider Order Book Imbalance
    df['bid_ask_spread'] = df['high'] - df['low']

    # Generate the alpha factor
    alpha_factor = (
        df['open_to_close'] 
        + df['high_to_low'] 
        + df['day_to_day_close'] 
        + (df['short_moving_avg'] - df['long_moving_avg'])
        + df['price_momentum']
        + df['rate_of_change']
        + df['rsi']
        + df['avg_volume']
        + df['volume_change']
        + df['total_amount_traded']
        + df['amount_change']
        + df['obv']
        + df['mfi']
        + df['high_low_spread']
        + df['bollinger_bands_width']
        + df['atr']
        + df['stochastic_oscillator_k']
        + df['macd_line']
        + df['signal_line']
        + df['bid_ask_spread']
    )

    return alpha_factor
```
