```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, high_volume_threshold=1.5, trend_lookback=10):
    # Calculate Intraday Price Movement
    df['high_low_range'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']
    
    # Aggregate Price Movements over a Period
    df['sum_high_low_range'] = df['high_low_range'].rolling(window=lookback_period).sum()
    df['sum_close_open_change'] = df['close_open_change'].rolling(window=lookback_period).sum()
    
    # Calculate Price Momentum Score
    df['std_close'] = df['close'].rolling(window=lookback_period).std()
    df['weighted_high_low_range'] = df['high_low_range'] / (df['std_close'] + 1e-6)
    df['weighted_close_open_change'] = df['close_open_change'] / (df['std_close'] + 1e-6)
    df['price_momentum_score'] = (df['weighted_high_low_range'] + df['weighted_close_open_change']) / 2
    
    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['high_volume'] = (df['volume'] > high_volume_threshold * df['avg_volume']).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    df['price_momentum_x_volume'] = df['price_momentum_score'] * df['high_volume']
    
    # Incorporate Recent Price Trend
    df['recent_trend'] = df['close'].rolling(window=trend_lookback).mean()
    df['trend_multiplier'] = (df['close'] - df['recent_trend']) / (df['close'].shift(trend_lookback) - df['recent_trend'].shift(trend_lookback))
    df['alpha_factor'] = df['price_momentum_x_volume'] * df['trend_multiplier']
    
    return df['alpha_factor']
```
