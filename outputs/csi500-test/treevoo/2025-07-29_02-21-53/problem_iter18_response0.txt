```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily True Range
    df['prev_close'] = df['close'].shift(1)
    df['high_low_diff'] = df['high'] - df['low']
    df['high_prev_close_diff'] = df['high'] - df['prev_close']
    df['prev_close_low_diff'] = df['prev_close'] - df['low']
    df['true_range'] = df[['high_low_diff', 'high_prev_close_diff', 'prev_close_low_diff']].max(axis=1)

    # Smooth True Range (20-day EMA)
    df['smoothed_true_range'] = df['true_range'].ewm(span=20, adjust=False).mean()

    # Calculate Long-Term True Range (50-day High and Low)
    df['long_term_high'] = df['high'].rolling(window=50).max()
    df['long_term_low'] = df['low'].rolling(window=50).min()
    df['long_term_true_range'] = df['long_term_high'] - df['long_term_low']

    # Compute Momentum of True Ranges
    df['true_range_momentum'] = df['long_term_true_range'] - df['smoothed_true_range']

    # Weight by Volume
    df['weighted_true_range_momentum'] = df['true_range_momentum'] * df['volume']

    # Integrate Intraday Price Movements
    df['high_low_diff_daily'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']

    # Aggregate and Smooth Intraday Price Movements (20-day EMA)
    df['sum_high_low_diff'] = df['high_low_diff_daily'].rolling(window=20).sum()
    df['sum_close_open_diff'] = df['close_open_diff'].rolling(window=20).sum()
    df['smoothed_high_low_diff'] = df['sum_high_low_diff'].ewm(span=20, adjust=False).mean()
    df['smoothed_close_open_diff'] = df['sum_close_open_diff'].ewm(span=20, adjust=False).mean()

    # Combine True Range and Intraday Price Movements
    df['combined_factor'] = df['weighted_true_range_momentum'] + df['smoothed_high_low_diff'] + df['smoothed_close_open_diff']

    # Evaluate Volume Activity
    df['avg_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['high_volume'] = df['volume'] > 1.5 * df['avg_volume']

    # Integrate Combined Factors
    df['final_combined_factor'] = df['combined_factor'] * df['high_volume']

    # Final Alpha Factor with Exponential Moving Average (14-day EMA)
    df['alpha_factor'] = df['final_combined_factor'].ewm(span=14, adjust=False).mean()

    return df['alpha_factor']
```
