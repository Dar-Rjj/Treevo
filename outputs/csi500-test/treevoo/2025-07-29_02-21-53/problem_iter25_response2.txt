```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for different time periods
    df['Daily_ROC'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Weekly_ROC'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['Monthly_ROC'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Identify Reversal Points
    df['Intraday_Range'] = df['high'] - df['low']
    df['Weekly_Range'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()

    # Volume Analysis
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Spike_Indicator'] = (df['volume'] > 2 * df['volume'].shift(1)).astype(int)

    # Use Open and Close Prices to Determine Trend Strength
    df['Bullish_Days_Ratio'] = (df['close'] > df['open']).rolling(window=20).mean()
    df['Bearish_Days_Ratio'] = (df['open'] > df['close']).rolling(window=20).mean()

    # Incorporate Amount Data
    df['Average_Amount'] = df['amount'].rolling(window=20).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']

    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['Daily_ROC'] + 
        df['Weekly_ROC'] + 
        df['Monthly_ROC'] + 
        df['Intraday_Range'] + 
        df['Weekly_Range'] + 
        df['Volume_Change'] + 
        df['Volume_Spike_Indicator'] + 
        df['Bullish_Days_Ratio'] - 
        df['Bearish_Days_Ratio'] + 
        df['Amount_Ratio']
    )

    return alpha_factor
```
