```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Combine Intraday Measures with weighted average
    # Assign weights (e.g., 0.6 for high-low spread and 0.4 for close-open change)
    combined_measure = 0.6 * high_low_spread + 0.4 * close_open_change
    
    # Adjust for Volume
    volume_adjusted_measure = combined_measure * df['volume']
    
    # Calculate Daily Momentum Score
    momentum_score = volume_adjusted_measure.diff()
    
    # Determine Trend Direction
    trend_indicator = momentum_score.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    return trend_indicator
```
