```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Adjust for Volume Impact
    df['Previous_Close'] = df['close'].shift(1)
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                        abs(x['high'] - df.loc[x.name, 'Previous_Close']), 
                                                                        abs(x['low'] - df.loc[x.name, 'Previous_Close'])), axis=1)
    df['Adjusted_True_Range'] = df['True_Range'] * (df['volume'] ** 0.5)

    # Calculate Intraday Price Movement and Signed Transaction Volume
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Sign_High_Low_Spread'] = df['High_Low_Diff'].apply(lambda x: 1 if x > 0 else -1)
    df['Signed_Volume'] = df['volume'] * df['Sign_High_Low_Spread']

    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    period = 10
    df['Sum_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=period).sum()
    df['Sum_Close_Open_Diff'] = df['Close_Open_Diff'].rolling(window=period).sum()
    df['Pos_Signed_Volume'] = df['Signed_Volume'].apply(lambda x: x if x > 0 else 0)
    df['Neg_Signed_Volume'] = df['Signed_Volume'].apply(lambda x: x if x < 0 else 0)
    df['Sum_Pos_Signed_Volume'] = df['Pos_Signed_Volume'].rolling(window=period).sum()
    df['Sum_Neg_Signed_Volume'] = df['Neg_Signed_Volume'].rolling(window=period).sum()

    # Calculate Integrated Momentum Score
    df['Close_Price_Std'] = df['close'].rolling(window=period).std()
    df['Integrated_Momentum_Score'] = (df['Sum_High_Low_Diff'] + df['Sum_Close_Open_Diff']) * df['Close_Price_Std']
    df['Total_Abs_Volume'] = df['volume'].abs().rolling(window=period).sum()
    df['Pos_Volume_Ratio'] = df['Sum_Pos_Signed_Volume'] / df['Total_Abs_Volume']
    df['Neg_Volume_Ratio'] = df['Sum_Neg_Signed_Volume'] / df['Total_Abs_Volume']
    df['Integrated_Momentum_Score'] = df['Integrated_Momentum_Score'] * (df['Pos_Volume_Ratio'] - df['Neg_Volume_Ratio'])

    # Combine Volume-Adjusted High-Low with Open-Close
    weight_hl = 0.6
    weight_oc = 0.4
    df['Combined_Measure'] = weight_hl * df['Adjusted_True_Range'] + weight_oc * df['Close_Open_Diff']

    # Dynamic Momentum Indicator
    df['Yesterday_Oscillator'] = df['Integrated_Momentum_Score'].shift(1)
    df['Trend_Continuity'] = df['Integrated_Momentum_Score'] - df['Yesterday_Oscillator']
    df['Smoothed_Trend'] = df['Trend_Continuity'].ewm(span=period, adjust=False).mean()

    # Evaluate Volume Activity
    df['Rolling_Avg_Volume'] = df['volume'].rolling(window=period).mean()
    df['High_Volume_Indicator'] = (df['volume'] > df['Rolling_Avg_Volume']).astype(int)

    # Integrate Combined Factors
    df['Volume_Ratio'] = df['volume'] / df['Rolling_Avg_Volume']
    df['Final_Factor'] = df['Integrated_Momentum_Score'] * df['Volume_Ratio'] * df['High_Volume_Indicator']

    return df['Final_Factor']
```
