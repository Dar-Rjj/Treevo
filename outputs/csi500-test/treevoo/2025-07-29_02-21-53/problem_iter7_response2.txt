```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Combine Intraday Measures with equal weights
    combined_intraday = (high_low_spread + close_open_change) / 2
    
    # Adjust for Volume
    volume_adjusted_intraday = combined_intraday * df['volume']
    
    # Apply Exponential Moving Average (EMA) over 14 days
    ema_14 = volume_adjusted_intraday.ewm(span=14).mean()
    
    # Compute Integrated Momentum Indicator
    integrated_momentum = ema_14 - ema_14.shift(14)
    
    return integrated_momentum
```
