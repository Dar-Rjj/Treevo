```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Exponential Moving Average of Returns
    short_term_ema = daily_return.ewm(span=12, adjust=False).mean()
    long_term_ema = daily_return.ewm(span=26, adjust=False).mean()

    # Calculate MACD
    macd = short_term_ema - long_term_ema

    # Calculate Volume Adjusted Momentum
    raw_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_adjusted_momentum = raw_momentum * df['volume']

    # Normalize MACD
    max_macd_50 = macd.rolling(window=50).max()
    normalized_macd = macd / max_macd_50

    # Combine MACD and Volume Adjusted Momentum
    alpha_factor = normalized_macd + volume_adjusted_momentum

    return alpha_factor
```
