```python
def heuristics_v2(df):
    # Calculate Price Range: High-Low Range
    price_range = df['high'] - df['low']
    
    # Evaluate Closing Price Dynamics: Close to Open Change
    close_open_change = df['close'] - df['open']
    
    # Compute Average Transaction Size
    avg_transaction_size = df['amount'] / df['volume']
    
    # Examine Volume Relative to Price Movement: Volume Weighted Return
    volume_weighted_return = (df['close'] - df['open']) * (df['volume'] / df['amount'])
    
    # Short-Term Momentum: 2-Period Rate of Change
    short_term_momentum = (df['close'] / df['close'].shift(1)) - 1
    
    # Long-Term Momentum: 5-Period Rate of Change
    long_term_momentum = (df['close'] / df['close'].shift(4)) - 1
    
    # Measure Price Reversals: Bearish/Bullish Reversal Indicator
    reversal_indicator = (df['high'] - df['close']) - (df['open'] - df['low'])
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        price_range + 
        close_open_change + 
        avg_transaction_size + 
        volume_weighted_return + 
        short_term_momentum + 
        long_term_momentum + 
        reversal_indicator
    )
    
    return alpha_factor
```
