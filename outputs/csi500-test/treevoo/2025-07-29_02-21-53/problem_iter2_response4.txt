```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Signed Transaction Volume
    df['high_low_spread'] = df['high'] - df['low']
    df['sign_high_low_spread'] = df['high_low_spread'].apply(lambda x: 1 if x > 0 else -1)
    df['signed_volume'] = df['volume'] * df['sign_high_low_spread']
    
    # Calculate Accumulated Buy-Sell Imbalance
    df['positive_volume'] = df['signed_volume'].apply(lambda x: x if x > 0 else 0)
    df['negative_volume'] = df['signed_volume'].apply(lambda x: abs(x) if x < 0 else 0)
    df['acc_buy'] = df['positive_volume'].rolling(window=5).sum()
    df['acc_sell'] = df['negative_volume'].rolling(window=5).sum()
    
    # Compute Order Imbalance Score
    df['total_abs_volume'] = df['positive_volume'] + df['negative_volume']
    df['order_imbalance_score'] = (df['acc_buy'] / df['total_abs_volume']) - (df['acc_sell'] / df['total_abs_volume'])
    
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].subtract(df['previous_close'], axis=0).abs().max(axis=1)
    df['true_range'] = df[['high', 'low', 'previous_close']].max(axis=1) - df[['high', 'low', 'previous_close']].min(axis=1)
    
    # Calculate Rolling Average True Range
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Price Momentum
    n_days = 10
    df['price_momentum'] = df['close'] - df['close'].shift(n_days)
    
    # Combine ATR and Price Momentum with Order Imbalance
    df['normalized_price_momentum'] = df['price_momentum'] / df['avg_true_range']
    df['alpha_factor'] = df['order_imbalance_score'] + df['normalized_price_momentum']
    
    return df['alpha_factor']
```
