import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average true range over 14 days
    df['true_range'] = df[['high' - 'low', 
                           ('high' - 'close').abs(), 
                           ('low' - 'close').abs()]].max(axis=1)
    df['avg_true_range_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the money flow multiplier
    df['money_flow_multiplier'] = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    
    # Calculate the 14-day money flow ratio
    df['money_flow_ratio'] = df['money_flow_volume'].rolling(window=14).sum() / (df['volume'].rolling(window=14).sum() * df['avg_true_range_14'])
    
    # Generate the alpha factor
    alpha_factor = df['price_change'] * df['money_flow_ratio']
    
    return alpha_factor.dropna()
