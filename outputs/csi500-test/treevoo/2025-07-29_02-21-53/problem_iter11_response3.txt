```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range and Intraday Price Movement
    df['True_Range'] = np.maximum(
        np.maximum(df['High'] - df['Low'], abs(df['High'] - df['Close'].shift(1))),
        abs(df['Low'] - df['Close'].shift(1))
    )
    
    df['Intraday_Price_Movement'] = df['High'] - df['Low']
    df['Close_Open_Diff'] = df['Close'] - df['Open']
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_TR'] = df['True_Range'] * df['Volume']
    df['Volume_Adjusted_Price_Movement'] = df['Close_Open_Diff'] * df['Volume']
    
    # Aggregate and Smooth High-Low Ranges and Price Movements
    high_low_ema = df['Intraday_Price_Movement'].ewm(span=20, adjust=False).mean()
    close_open_ema = df['Close_Open_Diff'].ewm(span=20, adjust=False).mean()
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Price_High_Low_Momentum'] = high_low_ema - close_open_ema
    df['Price_Volatility'] = df['Close'].rolling(window=20).std()
    df['Weighted_Momentum'] = df['Price_High_Low_Momentum'] * df['Price_Volatility']
    
    # Evaluate Volume Activity
    avg_volume = df['Volume'].ewm(span=20, adjust=False).mean()
    df['High_Volume'] = (df['Volume'] > 1.5 * avg_volume).astype(int)
    
    # Integrate Price, High-Low Range, and Volume Activity
    combined_measure = (df['Volume_Adjusted_TR'] + df['Volume_Adjusted_Price_Movement']) / 2
    combined_measure_ema = combined_measure.ewm(span=20, adjust=False).mean()
    
    # Dynamic Momentum Indicator
    df['Prev_Factor'] = combined_measure_ema.shift(1)
    df['Trend_Continuity'] = (combined_measure_ema - df['Prev_Factor']).rolling(window=20).std()
    
    # Generate Final Alpha Factor
    interaction_term = df['Price_High_Low_Momentum'] * df['Close_Open_Diff']
    df['Final_Alpha_Factor'] = (combined_measure_ema + interaction_term) * df['High_Volume']
    
    return df['Final_Alpha_Factor']
```
