```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_change'] = df['close'] - df['open']

    # Aggregate and Smooth Price Movements
    window_short = 20
    window_long = 50
    df['sum_high_low'] = df['high_low_diff'].rolling(window=window_short).sum()
    df['sum_close_open'] = df['close_open_change'].rolling(window=window_short).sum()
    df['smoothed_high_low'] = df['high_low_diff'].ewm(span=window_short, adjust=False).mean()
    df['long_term_high_low_range'] = df['high_low_diff'].rolling(window=window_long).max()

    # Compute Momentum of High-Low Ranges
    df['momentum_volatility'] = df['long_term_high_low_range'] - df['smoothed_high_low']

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].apply(
        lambda x: max(x['high'] - x['low'], x['high'] - x['prev_close'], x['prev_close'] - x['low']), axis=1)

    # Calculate Rolling Average True Range (ATR)
    atr_window = 14
    df['atr'] = df['true_range'].rolling(window=atr_window).mean()

    # Adjust Price Momentum by ATR
    df['weighted_high_low'] = df['sum_high_low'] * 0.6
    df['weighted_close_open'] = df['sum_close_open'] * 0.4
    df['adjusted_momentum_score'] = (df['weighted_high_low'] + df['weighted_close_open']) / df['atr']

    # Evaluate Volume Activity
    volume_window = 20
    df['avg_volume'] = df['volume'].rolling(window=volume_window).mean()
    df['high_volume'] = df['volume'] > df['avg_volume'] * 1.5
    df['volume_spike'] = df['volume'] > df['volume'].rolling(window=5).mean() * 2

    # Adjust Momentum Based on Volume
    df['adjusted_momentum'] = df['adjusted_momentum_score']
    df.loc[df['volume_spike'], 'adjusted_momentum'] = df['adjusted_momentum'] * 0.7

    # Combine Volume-Adjusted High-Low with Open-Close
    df['combined_measure'] = df['weighted_high_low'] * 0.6 + df['weighted_close_open'] * 0.4

    # Dynamic Momentum Indicator
    df['intraday_oscillator_yesterday'] = df['adjusted_momentum'].shift(1)
    df['trend_continuity'] = df['adjusted_momentum'] - df['intraday_oscillator_yesterday']
    df['std_returns'] = df['close'].pct_change().rolling(window=20).std()
    df['dynamic_momentum'] = df['trend_continuity'] / df['std_returns']

    # Integrate Combined Factors
    df['final_alpha'] = df['adjusted_momentum'] * (1 + df['high_volume'].astype(int)) * df['dynamic_momentum']

    return df['final_alpha']
```
