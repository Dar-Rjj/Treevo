```python
import pandas as pd

def heuristics_v2(df):
    # Compute Open-High Difference
    open_high_diff = df['high'] - df['open']
    
    # Compute Close-Low Difference
    close_low_diff = df['close'] - df['low']
    
    # Aggregate Price Changes
    period = 5  # Example period, can be adjusted
    open_high_sum = open_high_diff.rolling(window=period).sum()
    close_low_sum = close_low_diff.rolling(window=period).sum()
    
    # Calculate Price Momentum Score
    price_momentum_score = (open_high_sum + close_low_sum) / 2
    
    # Weight by Recent Price Volatility
    close_std = df['close'].rolling(window=period).std()
    price_momentum_score_weighted = price_momentum_score * close_std
    
    # Evaluate Volume Activity
    volume_ema = df['volume'].ewm(span=period, adjust=False).mean()
    above_average_volume = (df['volume'] > volume_ema).astype(int)
    
    # Integrate Price Momentum and Volume Activity
    alpha_factor = price_momentum_score_weighted * above_average_volume
    
    return alpha_factor
```
