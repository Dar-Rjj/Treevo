```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Price Movement
    high_low_diff = df['high'] - df['low']
    close_open_change = df['close'] - df['open']

    # Aggregate and Smooth Price Movements
    sum_high_low_diff = high_low_diff.rolling(window=20).sum()
    sum_close_open_change = close_open_change.rolling(window=20).sum()
    smoothed_high_low_range = sum_high_low_diff.ewm(span=20, adjust=False).mean()
    long_term_high_low_range = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min())

    # Compute Momentum of High-Low Ranges
    high_low_momentum = long_term_high_low_range - smoothed_high_low_range

    # Calculate True Range
    true_range = pd.DataFrame({
        'HL': df['high'] - df['low'],
        'HC': df['high'] - df['close'].shift(1),
        'LC': df['low'] - df['close'].shift(1)
    }).abs().max(axis=1)

    # Calculate Rolling Average True Range (ATR)
    atr = true_range.rolling(window=20).mean()

    # Adjust Price Momentum by ATR
    weighted_high_low_range = sum_high_low_diff / atr
    weighted_close_open_change = sum_close_open_change / atr

    # Evaluate Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    high_volume_indicator = (df['volume'] > 1.5 * avg_volume).astype(int)
    volume_spike = (df['volume'] > 2 * df['volume'].rolling(window=5).mean()).astype(int)

    # Adjust Momentum Based on Volume
    adjusted_momentum = (weighted_high_low_range + weighted_close_open_change) * (1 - 0.25 * volume_spike)

    # Integrate Combined Factors
    interim_alpha_factor = adjusted_momentum * high_volume_indicator

    # Combine ATR and Price Momentum
    interpretable_factor = interim_alpha_factor + high_low_momentum

    # Final Alpha Factor
    recent_price_trend = df['close'].rolling(window=20).mean()
    trend_multiplier = np.where(recent_price_trend > 0, 1.1, 0.9)
    final_alpha_factor = interpretable_factor * trend_multiplier

    # Introduce New Factor: Volume-Weighted ATR
    high_volume_days = df.loc[high_volume_indicator == 1]
    if not high_volume_days.empty:
        volume_weighted_atr = true_range.loc[high_volume_days.index].rolling(window=20).mean()
        final_alpha_factor = final_alpha_factor / volume_weighted_atr

    return final_alpha_factor
```
