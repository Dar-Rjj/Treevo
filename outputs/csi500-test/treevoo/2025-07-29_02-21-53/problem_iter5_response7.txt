```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement and Signed Transaction Volume
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    sign_high_low = np.sign(high_low_diff)
    signed_price_movement = (high_low_diff + close_open_diff) * sign_high_low
    
    # Aggregate Price Movements and Accumulated Buy-Sell Imbalance
    period = 5  # Define the period for aggregation
    sum_high_low_diff = high_low_diff.rolling(window=period).sum()
    sum_close_open_diff = close_open_diff.rolling(window=period).sum()
    signed_volume = df['volume'] * np.sign(signed_price_movement)
    positive_volumes = signed_volume.where(signed_volume > 0, 0).rolling(window=period).sum()
    negative_volumes = signed_volume.where(signed_volume < 0, 0).rolling(window=period).sum().abs()
    
    # Calculate Integrated Momentum Score
    close_std = df['close'].rolling(window=period).std()
    momentum_score = (sum_high_low_diff + sum_close_open_diff) * close_std
    total_volumes = positive_volumes + negative_volumes
    momentum_score += positive_volumes / total_volumes
    momentum_score -= negative_volumes / total_volumes
    
    # Evaluate Volume Activity and True Range
    avg_volume = df['volume'].rolling(window=period).mean()
    high_volume_days = df['volume'] > avg_volume
    true_range = pd.concat([df['high'] - df['low'], 
                            (df['high'] - df['close'].shift(1)).abs(), 
                            (df['close'].shift(1) - df['low']).abs()], axis=1).max(axis=1)
    atr = true_range.rolling(window=period).mean()
    
    # Integrate Price, Volume, and Order Imbalance
    volume_ratio = df['volume'] / avg_volume
    normalized_momentum = momentum_score / atr
    final_factor = normalized_momentum * volume_ratio
    final_factor = final_factor.fillna(0)
    
    return final_factor
```
