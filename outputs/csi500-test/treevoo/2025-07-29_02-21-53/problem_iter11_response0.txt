```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['True_Range'] = df.apply(lambda x: max(x['high'] - x['low'], x['high'] - x['Prev_Close'], x['Prev_Close'] - x['low']), axis=1)
    
    # Calculate Rolling Average True Range (ATR) over 14 days
    n_days = 14
    df['ATR'] = df['True_Range'].rolling(window=n_days).mean()
    
    # Calculate Intraday High-Low Spread and Close-Open Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Close_Open_Diff'] = df['close'] - df['open']
    
    # Aggregate Price Movements and Smooth High-Low Ranges
    period = 50
    df['Sum_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=period).sum()
    df['Sum_Close_Open_Diff'] = df['Close_Open_Diff'].rolling(window=period).sum()
    df['Smooth_High_Low_Range'] = df['High_Low_Diff'].ewm(span=20, adjust=False).mean()
    df['Long_Term_High_Low_Range'] = df['High_Low_Diff'].rolling(window=50).mean()
    
    # Adjust for Volume Impact on Price
    df['Volume_Adjusted_High_Low'] = df['High_Low_Diff'] * df['volume']
    
    # Combine Volume-Adjusted High-Low with Open-Close
    volatility = df['close'].pct_change().std()
    df['Weighted_Average'] = (df['Volume_Adjusted_High_Low'] * 0.6 + df['Close_Open_Diff'] * 0.4) / (volatility + 1e-6)
    
    # Introduce Volume Trend Analysis
    df['Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Volume_Trend'] = df['volume'] > df['Volume_EMA']
    
    # Dynamic Momentum Indicator
    df['Prev_Indicator'] = df['Weighted_Average'].shift(1)
    df['Momentum_Change'] = df['Weighted_Average'] - df['Prev_Indicator']
    df['Volatility_Smoothing'] = df['close'].pct_change().rolling(window=10).std()
    df['Trend_Continuity'] = df['Momentum_Change'] / (df['Volatility_Smoothing'] + 1e-6)
    
    # Calculate Combined Price and High-Low Range Momentum
    df['Combined_Momentum'] = df['High_Low_Diff'] + df['Close_Open_Diff'] - (df['Smooth_High_Low_Range'] - df['Long_Term_High_Low_Range'])
    df['Price_Volatility'] = df['close'].rolling(window=20).std()
    df['Weighted_Combined_Momentum'] = df['Combined_Momentum'] * df['Price_Volatility']
    
    # Evaluate Volume Activity
    df['Volume_Activity'] = df['volume'] > df['Volume_EMA']
    
    # Integrate Combined Factors
    df['Final_Alpha_Factor'] = df['Weighted_Combined_Momentum'] * df['Volume_Activity']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(span=14, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
