```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for different time periods
    df['Daily_ROC'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Weekly_ROC'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['Monthly_ROC'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Identify Reversal Points
    df['Intraday_Range'] = df['high'] - df['low']
    df['Weekly_Range'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Volume Analysis
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Spike'] = (df['volume'] > 2 * df['volume'].shift(1)).astype(int)
    
    # Use Open and Close Prices to Determine Trend Strength
    df['Bullish_Days_Ratio'] = (df['close'] > df['open']).rolling(window=20).mean()
    df['Bearish_Days_Ratio'] = (df['open'] > df['close']).rolling(window=20).mean()
    
    # Incorporate Amount Data
    df['Average_Amount'] = df['amount'].rolling(window=20).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Integrate Price and Volume Factors
    df['ROC_Volume_Spike_Combined'] = df['Daily_ROC'] * df['Volume_Spike']
    df['ROC_Intraday_Range_Combined'] = df['Daily_ROC'] * df['Intraday_Range']
    
    # Analyze Price and Amount Correlation
    df['Close_Amount_Corr'] = df[['close', 'amount']].rolling(window=20).corr().iloc[::2, :]['close'].reset_index(level=0, drop=True)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['Daily_ROC'] + 
        df['Weekly_ROC'] + 
        df['Monthly_ROC'] + 
        df['Intraday_Range'] + 
        df['Weekly_Range'] + 
        df['Volume_Change'] + 
        df['Volume_Spike'] + 
        df['Bullish_Days_Ratio'] + 
        df['Bearish_Days_Ratio'] + 
        df['Amount_Ratio'] + 
        df['ROC_Volume_Spike_Combined'] + 
        df['ROC_Intraday_Range_Combined'] + 
        df['Close_Amount_Corr']
    )
    
    return alpha_factor
```
