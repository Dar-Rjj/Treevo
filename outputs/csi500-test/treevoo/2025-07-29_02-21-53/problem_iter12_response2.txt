```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with 20-day EMA
    short_term_range_ema = daily_high_low_range.ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term High-Low Range (50-day)
    long_term_high_low_range = df[['high', 'low']].rolling(window=50).max().iloc[:, 0] - df[['high', 'low']].rolling(window=50).min().iloc[:, 1]
    
    # Compute Momentum of High-Low Ranges
    range_momentum = long_term_high_low_range - short_term_range_ema
    
    # Weight by Volume
    volume_weighted_momentum = range_momentum * df['volume']
    
    # Integrate Price Change
    daily_price_change = df['close'] - df['open']
    smoothed_daily_price_change = daily_price_change.ewm(span=20, adjust=False).mean()
    
    # Combine with High-Low Range Momentum
    combined_factor = volume_weighted_momentum + smoothed_daily_price_change
    
    # Integrate Price Trend
    long_term_price_trend = df['close'].rolling(window=50).mean()
    smoothed_long_term_price_trend = long_term_price_trend.ewm(span=50, adjust=False).mean()
    price_trend_momentum = smoothed_long_term_price_trend - smoothed_daily_price_change
    
    # Final Factor
    final_factor = combined_factor + price_trend_momentum
    
    return final_factor
```
