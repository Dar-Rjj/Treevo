```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range with 20-day EMA
    df['high_low_range_ema_20'] = df['high_low_range'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum of High-Low Ranges
    df['high_low_range_momentum'] = df['high_low_range_ema_20'].diff(20)
    
    # Weight by Volume
    df['high_low_range_vol_weighted'] = df['high_low_range_momentum'] * df['volume']
    
    # Integrate Price Change
    df['price_change'] = df['close'] - df['open']
    df['price_change_ema_20'] = df['price_change'].ewm(span=20, adjust=False).mean()
    df['combined_factor'] = df['high_low_range_vol_weighted'] + df['price_change_ema_20']
    
    # Integrate Price Trend
    df['long_term_trend'] = df['close'].rolling(window=100).mean()
    df['long_term_trend_ema_100'] = df['long_term_trend'].ewm(span=100, adjust=False).mean()
    df['price_trend_momentum'] = df['long_term_trend_ema_100'].diff(100)
    
    # Combine all factors
    df['final_factor'] = df['combined_factor'] + df['price_trend_momentum']
    
    return df['final_factor']
```
