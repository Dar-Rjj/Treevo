```python
import pandas as pd

def heuristics_v2(df):
    """
    The function computes a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the amount to volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Heuristic alpha factor
    factor = (price_diff / avg_price) * amount_volume_ratio
    
    return factor
```
