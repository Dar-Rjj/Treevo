```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()

    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return

    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']

    # Calculate Intraday Volatility (Average of (High - Low) over 10 days)
    intraday_volatility = high_low_range.rolling(window=10).mean()

    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)

    # Calculate Close Price Momentum (Sum of ((Close at day t / Close at day t-1) - 1) over the last 20 days)
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()

    # Calculate Intraday Range Momentum
    intraday_range_momentum = (high_low_range / high_low_range.shift(1)) - 1

    # Calculate Close-Open Spread Ratio
    close_open_spread_ratio = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))

    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = volume_adjusted_intraday_range + combined_value_momentum - close_open_spread_ratio

    # Calculate Previous Day's Intraday Range
    prev_day_high_low_range = df['high'].shift(1) - df['low'].shift(1)

    # Calculate Price Change Velocity
    price_change_velocity = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) * (df['volume'].shift(1) - df['volume'].shift(2))

    # Adjust Volatility by Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    adjusted_volatility_by_volume = intraday_volatility * avg_volume

    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    combined_intraday_range_momentum = intraday_range_momentum * price_change_velocity * df['volume']

    # Combine Price, Volume, and Volatility Indicators
    combined_price_volume_volatility = price_change_velocity * volume_acceleration * adjusted_volatility_by_volume

    # Synthesize Final Alpha Factor
    final_alpha_factor = (intraday_volatility * (intermediate_alpha_factor + close_price_momentum + intraday_range_momentum) +
                          combined_intraday_range_momentum + combined_price_volume_volatility)

    return final_alpha_factor
```
