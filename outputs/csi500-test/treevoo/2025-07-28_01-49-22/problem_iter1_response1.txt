```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    high_low_volatility = high_low_range.rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_volatility = high_low_volatility * avg_volume
    
    return adjusted_volatility
```
