import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with EMA (10-day window)
    smooth_high_low_range = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = np.where(smooth_high_low_range > 1.0, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_high_low_range = volume_weighted_high_low_range / volatility
    
    # Calculate Absolute Close-to-Close Return
    absolute_close_to_close_return = df['close'].diff().abs()
    
    # Combine Adjusted High-Low Range and Absolute Close-to-Close Return
    combined_value = adjusted_high_low_range - absolute_close_to_close_return
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = combined_value + close_price_momentum
    
    # Generate Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * df['volume']
    
    return final_alpha_factor
