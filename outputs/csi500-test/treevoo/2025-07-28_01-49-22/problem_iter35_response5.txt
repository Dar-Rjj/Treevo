```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / df['high'].shift(1) - df['low'].shift(1)
    
    # Smooth High-Low Range Ratio
    smooth_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smooth_high_low_range_ratio > 1.3).astype(int) * 1.8 + (smooth_high_low_range_ratio < 1.3).astype(int) * 0.2
    
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low_range = high_low_range * df['volume']
    smooth_vol_weighted_high_low_range = vol_weighted_high_low_range.ewm(span=10).mean()
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adj_vol_weighted_high_low_range = smooth_vol_weighted_high_low_range / volatility
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = np.abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    combined_value_momentum = combined_value.diff()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (adj_vol_weighted_high_low_range
                                 + combined_value_momentum
                                 + intraday_range_ratio
                                 - (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * intraday_signal
    
    # Generate Final Alpha Factor
    final_alpha_factor = adjusted_intermediate_alpha_factor + close_price_momentum
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    smoothed_volume_weighted_final_alpha_factor = volume_weighted_final_alpha_factor.ewm(span=5).mean()
    
    return smoothed_volume_weighted_final_alpha_factor
```
