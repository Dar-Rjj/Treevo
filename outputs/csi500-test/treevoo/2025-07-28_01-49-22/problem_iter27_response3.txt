```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value_momentum = high_low_range - close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = volume_adjusted_intraday_range + combined_value_momentum + intraday_range_ratio - (df['open'] - df['close'].shift(1))
    
    # Adjust Intermediate Alpha Factor
    adjustment = intraday_range_ratio.apply(lambda x: 1.2 if x > 0.5 else 0.8)
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * adjustment
    
    # Introduce Volume Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Weighted Intraday Volatility
    weighted_intraday_volatility = intraday_volatility * (df['volume'] / df['volume'].shift(1))
    
    # Generate Final Alpha Factor
    final_alpha_factor = weighted_intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum + volume_momentum)
    
    return final_alpha_factor
```
