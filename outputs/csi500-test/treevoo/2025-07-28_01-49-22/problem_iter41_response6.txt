```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Smoothed High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['smoothed_high_low_range'] = df['high_low_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['close'] / df['open'] > 1.3, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = (df['high_low_range'] * df['volume']).ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = df['volume_weighted_high_low_range'] + df['combined_value_momentum'] + (df['high_low_range'] / df['smoothed_high_low_range']) - (df['open'] - df['close'].shift(1))
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] / df['volatility']
    
    # Introduce Dynamic Threshold Adjustment
    df['dynamic_threshold'] = (df['high_low_range'] / (df['high'] + df['low'])).rolling(window=10).mean()
    df['signal'] = np.where((df['high_low_range'] / (df['high'] + df['low'])) > df['dynamic_threshold'], 1, -1)
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] * np.where(df['intraday_signal'] == 1, 1.8, 0.2) * np.where(df['signal'] == 1, 1.8, 0.2)
    
    # Introduce Additional Momentum Component
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = np.where(df['ma_50'] > df['ma_200'], 1.2, 0.8)
    
    # Adjust Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']) * df['long_term_momentum']
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = (df['final_alpha_factor'] * df['volume']).ewm(span=5).mean()
    
    # Introduce Volume Momentum
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_momentum'] = np.where(df['volume'] / df['avg_volume_10d'] > 1, 1.2, 0.8)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['volume_weighted_final_alpha_factor'] * df['volume_momentum']
    
    return df['final_alpha_factor']
```
