```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range with 5-day EMA
    df['ema_high_low_range'] = df['high_low_range'].ewm(span=5).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = np.where(df['ema_high_low_range'] > 1.2, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = (df['high_low_range'] * df['volume']).ewm(span=5).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'].diff(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'].diff(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['volume_weighted_high_low_range'] +
        df['combined_value_momentum'] +
        df['intraday_volatility'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['volatility_adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor based on Intraday Momentum Signal
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['intraday_momentum_signal'] == 1,
        df['volatility_adjusted_intermediate_alpha_factor'] * 1.6,
        df['volatility_adjusted_intermediate_alpha_factor'] * 0.4
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (
        (df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']) *
        df['intraday_volatility']
    )
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = (df['final_alpha_factor'] * df['volume']).ewm(span=10).mean()
    
    # Introduce Additional Momentum Component
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = np.where(df['ma_50'] > df['ma_200'], 1.3, 0.7)
    
    # Adjust Final Alpha Factor based on Long-Term Momentum
    df['final_alpha_factor_adjusted'] = df['volume_weighted_final_alpha_factor'] * df['long_term_momentum']
    
    return df['final_alpha_factor_adjusted']
```
