import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio_ma = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    dynamic_threshold_signal = (high_low_range / (df['high'] + df['low'])) > high_low_ratio_ma
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.copy()
    adjusted_intermediate_alpha_factor[(high_low_range / (df['high'] + df['low'])) > 1.0] *= 1.5
    adjusted_intermediate_alpha_factor[(high_low_range / (df['high'] + df['low'])) < 1.0] *= 0.5
    
    # Introduce Volume Momentum
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10d
    volume_momentum_adjustment = volume_ratio.apply(lambda x: 0.15 if x > 1 else -0.15)
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * (1 + volume_momentum_adjustment)
    
    # Adjust Final Alpha Factor by Volume
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    final_alpha_factor *= n_period_avg_volume
    
    return final_alpha_factor
