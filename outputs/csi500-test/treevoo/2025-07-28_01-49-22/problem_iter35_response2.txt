```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Smooth Volume-Weighted High-Low Range with EMA
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=15).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        smoothed_volume_weighted_high_low_range +
        combined_value_momentum +
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1)).abs()
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio_10_day_ma = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    dynamic_threshold_signal = (high_low_range / (df['high'] + df['low'])) > high_low_ratio_10_day_ma
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjustment_factor = np.where((high_low_range / (df['high'] + df['low'])) > 1.0, 1.5, 0.5)
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * adjustment_factor
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_intermediate_alpha_factor /= volatility
    
    # Compare with Baseline (Intraday Range Ratio's Median)
    median_high_low_ratio = (high_low_range / (df['high'] + df['low'])).median()
    baseline_signal = (high_low_range / (df['high'] + df['low'])) > median_high_low_ratio
    
    # Adjust Intermediate Alpha Factor
    final_adjustment_factor = np.where(baseline_signal, 1.7, 0.3)
    adjusted_intermediate_alpha_factor *= final_adjustment_factor
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * intraday_volatility
    
    # Adjust Final Alpha Factor by Volume
    ema_volume = df['volume'].ewm(span=20).mean()
    final_alpha_factor *= ema_volume
    
    return final_alpha_factor
```
