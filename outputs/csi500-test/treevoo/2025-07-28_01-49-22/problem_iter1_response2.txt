```python
def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Previous Day's Close-to-Open Gap
    previous_close = df['close'].shift(1)
    close_to_open_gap = (df['open'] - previous_close) / previous_close
    
    # Combine for Final Alpha Factor
    alpha_factor = volume_adjusted_intraday_return - close_to_open_gap
    
    return alpha_factor
```
