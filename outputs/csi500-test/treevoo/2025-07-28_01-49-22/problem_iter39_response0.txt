```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smoothed_high_low_range'] = df['high_low_range'].ewm(span=5).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['close'] / df['open'] > 1.2, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['smoothed_volume_weighted_high_low'] = df['volume_weighted_high_low'].ewm(span=5).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['smoothed_volume_weighted_high_low'] 
                                + df['combined_value_momentum'] 
                                + df['smoothed_high_low_range'] 
                                - (df['open'] - df['close'].shift(1)))
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha'] = df['intermediate_alpha'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    df['adjusted_intermediate_alpha'] = df['adjusted_intermediate_alpha'] * (1.6 * (df['intraday_signal'] == 1) + 0.4 * (df['intraday_signal'] == -1))
    
    # Generate Final Alpha Factor
    df['final_alpha'] = (df['adjusted_intermediate_alpha'] + df['close_price_momentum']) * df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_final_alpha'] = df['final_alpha'] * df['volume']
    df['ema_volume_weighted_final_alpha'] = df['volume_weighted_final_alpha'].ewm(span=10).mean()
    
    # Introduce Additional Momentum Component
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = np.where(df['50_day_ma'] > df['200_day_ma'], 1, -1)
    
    # Adjust Final Alpha Factor based on Long-Term Momentum
    df['final_alpha'] = df['ema_volume_weighted_final_alpha'] * (1.3 * (df['long_term_momentum'] == 1) + 0.7 * (df['long_term_momentum'] == -1))
    
    # Introduce Short-Term Reversal Component
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Adjust Final Alpha Factor based on 5-Day Return
    df['final_alpha'] = df['final_alpha'] - df['5_day_return'] * (df['5_day_return'] > 0) + df['5_day_return'] * (df['5_day_return'] < 0)
    
    return df['final_alpha']
```
