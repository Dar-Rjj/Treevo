import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()

    # Calculate Exponential Moving Average (EMA)
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()

    # Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # On-Balance Volume (OBV)
    df['OBV'] = 0
    df.loc[df['close'] > df['close'].shift(), 'OBV'] = df['volume']
    df.loc[df['close'] < df['close'].shift(), 'OBV'] = -df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Volume Oscillator
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Oscillator'] = df['Volume_MA_5'] - df['Volume_MA_10']

    # Rate of Change (ROC)
    df['ROC_5'] = df['close'].pct_change(periods=5)

    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI'] = 100 - (100 / (1 + RS))

    # Moving Average Convergence Divergence (MACD)
    df['MACD'] = df['EMA_12'] - df['EMA_26']
    df['MACD_Signal'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['MACD_Histogram'] = df['MACD'] - df['MACD_Signal']

    # Price-Volume Trend (PVT)
    df['PVT'] = (df['close'] / df['close'].shift(1) - 1) * df['volume']
    df['PVT'] = df['PVT'].cumsum()

    # Negative Volume Index (NVI)
    df['NVI'] = 0
    mask = df['volume'] < df['volume'].shift()
    df.loc[mask, 'NVI'] = (df['close'] / df['close'].shift() - 1) * 100
    df['NVI'] = df['NVI'].cumsum()

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    MFI_ratio = positive_money_flow_sum / negative_money_flow_sum
    df['MFI'] = 100 - (100 / (1 + MFI_ratio))

    # Chaikin Oscillator
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['Chaikin_Oscillator'] = adl.ewm(span=3, adjust=False).mean() - adl.ewm(span=10, adjust=False).mean()
