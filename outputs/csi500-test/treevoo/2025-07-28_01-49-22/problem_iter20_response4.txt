```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Compute Previous Day's Close-to-Open Change
    close_open_change = df['open'] - df['close'].shift(1)
    
    # Calculate Daily High-Low Range
    daily_high_low_range = high_low_range
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = daily_high_low_range.rolling(window=15).sum() / 15
    
    # Smooth High-Low Range Volatility
    smoothed_high_low_range_volatility = high_low_range_volatility.ewm(span=15, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    close_open_difference = df['close'] - df['open']
    weighted_close_open_difference = 0.1 * close_open_difference
    combined_smoothed_volatility = smoothed_high_low_range_volatility + weighted_close_open_difference
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        close_open_change
    )
    
    # Adjust Intermediate Alpha Factor
    adjustment = 1.5 if intraday_range_ratio > 1.0 else 0.5
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * adjustment
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    dynamic_threshold = smoothed_high_low_range_volatility.ewm(span=15, adjust=False).mean()
    signal = (smoothed_high_low_range_volatility > dynamic_threshold).astype(int)
    adjusted_combined_alpha_factor = adjusted_intermediate_alpha_factor * (signal * 1.6 + (1 - signal) * 0.4)
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_combined_alpha_factor + close_price_momentum) * df['volume']
    
    return final_alpha_factor
```
