```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day's close to the current day's open
    df['price_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate the ratio of the volume to the amount, which gives a sense of the average transaction size
    df['avg_transaction_size'] = df['amount'] / df['volume']
    
    # Compute the intraday volatility as the difference between high and low prices
    df['intraday_volatility'] = df['high'] - df['low']
    
    # The alpha factor is the product of the price change, average transaction size, and intraday volatility
    alpha_factor = (df['price_change'] * df['avg_transaction_size'] * df['intraday_volatility'])
    
    return alpha_factor
```
