```python
import pandas as pd

def heuristics_v2(df, n_days):
    # Compute daily price change using close price
    df['daily_price_change'] = df['close'].pct_change()
    
    # Determine median volume over the last N days
    df['median_volume'] = df['volume'].rolling(window=n_days, min_periods=1).median()
    
    # Filter out days with volume below the median
    df_filtered = df[df['volume'] >= df['median_volume']]
    
    # Calculate cumulative price change over the filtered days
    df_filtered['cumulative_momentum'] = df_filtered['daily_price_change'].rolling(window=n_days, min_periods=1).sum()
    
    # Return the factor values
    return df_filtered['cumulative_momentum']
```
