```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smooth_high_low'] = df['high_low_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = (df['smooth_high_low'] > 1.0).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['smooth_volume_weighted_high_low'] = df['volume_weighted_high_low'].ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['smooth_volume_weighted_high_low'] 
                                + df['combined_value_momentum'] 
                                + df['intraday_volatility'] 
                                - (df['open'] - df['close'].shift(1)))
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha'] = df['intermediate_alpha'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha'] = df['adjusted_intermediate_alpha'] * (1.5 * df['intraday_signal'] + 0.5 * (1 - df['intraday_signal']))
    
    # Generate Final Alpha Factor
    df['final_alpha'] = (df['adjusted_intermediate_alpha'] + df['close_momentum']) * df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_final_alpha'] = df['final_alpha'] * df['volume']
    df['smooth_final_alpha'] = df['volume_weighted_final_alpha'].ewm(span=5).mean()
    
    return df['smooth_final_alpha']
```
