```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change in closing prices
    df['roc_close'] = df['close'].pct_change(periods=5)
    
    # Calculate the deviation of volume from its 20-day moving average
    df['vol_deviation'] = (df['volume'] - df['volume'].rolling(window=20).mean()) / df['volume'].rolling(window=20).std()
    
    # Create a momentum-based indicator by multiplying the rate of change in closing prices by the volume deviation
    df['momentum_indicator'] = df['roc_close'] * df['vol_deviation']
    
    # Construct an intraday volatility factor based on the difference between high and low prices
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Integrate the intraday volatility factor with the momentum-based indicator
    df['composite_factor_1'] = df['momentum_indicator'] + df['intraday_volatility']
    
    # Create a sentiment factor by comparing opening and closing prices
    df['sentiment_factor'] = (df['close'] > df['open']).astype(int).rolling(window=5).mean() - (df['close'] < df['open']).astype(int).rolling(window=5).mean()
    
    # Incorporate the sentiment factor into the composite factor
    df['composite_factor_2'] = df['composite_factor_1'] + df['sentiment_factor']
    
    # Evaluate the consistency of large transaction amounts with significant price changes
    df['amount_deviation'] = (df['amount'] - df['amount'].rolling(window=20).mean()) / df['amount'].rolling(window=20).std()
    
    # Generate a factor that captures the frequency and magnitude of large transactions
    df['large_transaction_factor'] = df['amount_deviation'] * df['close'].pct_change().abs()
    
    # Combine the large transaction factor with the composite factor to enhance predictive power
    df['final_alpha_factor'] = df['composite_factor_2'] + df['large_transaction_factor']
    
    return df['final_alpha_factor']
```
