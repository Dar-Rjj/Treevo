```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)

    # Smooth High-Low Range Ratio
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10).mean()

    # Generate Intraday Momentum Signal
    df['Intraday_Signal'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > 1.3, 1, -1)

    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=10).mean()

    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Adjusted_Volume_Weighted_High_Low_Range'] = df['Smoothed_Volume_Weighted_High_Low_Range'] / df['Volatility']

    # Calculate Absolute Close-to-Close Return
    df['Absolute_Close_to_Close_Return'] = np.abs(df['close'] - df['close'].shift(1))

    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] + df['Absolute_Close_to_Close_Return']

    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)

    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()

    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = df['Adjusted_Volume_Weighted_High_Low_Range'] + df['Combined_Value_Momentum'] + df['Intraday_Range_Ratio'] - (df['open'] - df['close'].shift(1))

    # Introduce Dynamic Threshold Adjustment
    df['10_Day_MA_Intraday_Range_Ratio'] = df['Intraday_Range_Ratio'].rolling(window=10).mean()
    df['Dynamic_Signal'] = np.where(df['Intraday_Range_Ratio'] > df['10_Day_MA_Intraday_Range_Ratio'], 1.8, 0.2)

    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] * df['Dynamic_Signal']

    # Introduce Volume Momentum
    df['10_Day_Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Momentum'] = df['volume'] / df['10_Day_Average_Volume']
    df['Volume_Adjustment'] = np.where(df['Volume_Momentum'] > 1, 1.2, 0.8)
    df['Final_Alpha_Factor'] = (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']) * df['Volume_Adjustment']

    # Weight by Volume
    df['Volume_Weighted_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['volume']
    df['EMA_Volume_Weighted_Final_Alpha_Factor'] = df['Volume_Weighted_Final_Alpha_Factor'].ewm(span=5).mean()

    return df['EMA_Volume_Weighted_Final_Alpha_Factor']
```
