```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility (Average of High-Low Range over 10 days)
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Calculate Previous Day's Close-to-Open Change
    close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Smooth High-Low Range Ratio
    ema_window_size = 10
    smoothed_high_low_range_ratio = intraday_range_ratio.ewm(span=ema_window_size, adjust=False).mean()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range + 
        combined_value_momentum - 
        close_to_open_change + 
        intraday_range_ratio
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    positive_signal = (smoothed_high_low_range_ratio > 1.0)
    negative_signal = (smoothed_high_low_range_ratio < 1.0)
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.where(negative_signal, intermediate_alpha_factor * 1.5).where(positive_signal, intermediate_alpha_factor * 0.5)
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum + intraday_range_momentum)
    
    return final_alpha_factor
```
