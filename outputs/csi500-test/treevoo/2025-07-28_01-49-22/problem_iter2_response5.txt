```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 10-Day Simple Moving Average (SMA)
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate Momentum
    df['momentum'] = df['close'] - df['sma_10']
    
    # Adjust Momentum by Intraday Volatility
    df['avg_daily_range'] = df['daily_range'].rolling(window=10).sum() / 10
    df['adjusted_momentum'] = df['momentum'] / df['avg_daily_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Adjusted Intraday Return
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Gap
    df['prev_close_to_open_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine for Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] + df['volume_adjusted_intraday_return'] - df['prev_close_to_open_gap']
    
    return df['alpha_factor']
```
