```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Calculate Short-Term and Long-Term Moving Averages of Daily Returns
    short_term_ma = df['daily_return'].rolling(window=5).mean()
    long_term_ma = df['daily_return'].rolling(window=30).mean()
    momentum = short_term_ma - long_term_ma
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_intraday_range = high_low_range + abs_close_to_close
    
    # Volume-Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Generate Alpha Factor from Momentum
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    baseline = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    
    alpha_factor = np.where(smoothed_high_low_range_ratio > baseline, 1, -1) * df['volume']
    
    # Adjust Alpha Factor Based on Intraday Range Ratio
    intraday_range_ratio = (high_low_range / (df['high'] + df['low']))
    alpha_factor = np.where(intraday_range_ratio > 0.7, alpha_factor * 1.4,
                            np.where(intraday_range_ratio < 0.4, alpha_factor * 0.6, alpha_factor))
    
    # Introduce Volume Momentum
    volume_momentum = (df['volume'].diff() / df['volume'].shift(1))
    
    # Volume-Adjusted Close Price Momentum
    volume_adjusted_close_momentum = (df['close'].diff()) * (df['volume'] / df['volume'].shift(1))
    
    # Adjust Alpha Factor by Close-Price Change
    alpha_factor = alpha_factor * df['close'].diff()
    
    # Incorporate Close-Open Difference
    close_open_diff_weight = 0.5  # Fixed Weight
    alpha_factor = alpha_factor + (df['close'] - df['open']) * close_open_diff_weight
    
    # Incorporate 10-Day Average Volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_adjustment = df['volume'] / avg_volume_10d
    
    # Generate Final Alpha Factor
    final_alpha_factor = (momentum * (alpha_factor + volume_adjusted_close_momentum + volume_momentum) * volume_adjustment)
    
    return final_alpha_factor
```
