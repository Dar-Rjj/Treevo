```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range + abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Compute Previous Day's Close-to-Open Change
    close_open_change = df['open'] - df['close'].shift(1)
    
    # Calculate Daily High-Low Range
    daily_high_low_range = high_low_range
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = daily_high_low_range.rolling(window=10).sum() / 10
    
    # Smooth High-Low Range Volatility
    ema_window_size = 15
    smoothed_high_low_range_volatility = high_low_range_volatility.ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    close_open_difference = df['close'] - df['open']
    weighted_close_open_difference = 0.1 * close_open_difference
    combined_smoothed_high_low_range_volatility = smoothed_high_low_range_volatility + weighted_close_open_difference
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        close_open_change
    )
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha_factor = np.where(
        intraday_range_ratio > 0.7,
        intermediate_alpha_factor * 1.5,
        intermediate_alpha_factor * 0.5
    )
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    dynamic_threshold = smoothed_high_low_range_volatility.ewm(span=ema_window_size, adjust=False).mean()
    signal = np.where(
        smoothed_high_low_range_volatility > dynamic_threshold,
        1.5,
        0.5
    )
    
    # Adjust Combined Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * signal
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum) * df['volume']
    
    return final_alpha_factor
```
