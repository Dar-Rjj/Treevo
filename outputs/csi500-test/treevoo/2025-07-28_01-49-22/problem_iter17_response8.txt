```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average true range (ATR) over 14 days
    df['h_l'] = df['high'] - df['low']
    df['h_cy'] = (df['high'] - df['close'].shift(1)).abs()
    df['l_cy'] = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = df[['h_l', 'h_cy', 'l_cy']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the money flow ratio
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    df['money_ratio'] = df['positive_money_flow'].rolling(window=14).sum() / df['negative_money_flow'].rolling(window=14).sum()
    
    # Combine ATR and money flow ratio to create the alpha factor
    df['alpha_factor'] = (df['atr_14'] / df['close']) * df['money_ratio']
    
    # Return the alpha factor as a pandas Series indexed by date
    return df['alpha_factor']
```
