```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    high_low_range_ratio_ema = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = np.where(high_low_range_ratio_ema > 1.3, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    volume_weighted_high_low_range_ema = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_volume_weighted_high_low_range = volume_weighted_high_low_range_ema / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    
    # Combine Value
    combined_value = high_low_range - np.abs(close_to_close_return)
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        adjusted_volume_weighted_high_low_range +
        combined_value +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = np.where(
        intraday_signal == 1,
        intermediate_alpha_factor * 1.8,
        intermediate_alpha_factor * 0.2
    )
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        close_price_momentum * 
        adjusted_intermediate_alpha_factor
    )
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    final_alpha_factor_ema = volume_weighted_final_alpha_factor.ewm(span=5).mean()
    
    return final_alpha_factor_ema
```
