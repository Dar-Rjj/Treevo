def heuristics_v2(df, N=10, M=20):
    # Compute Daily Price Change using Close price
    df['daily_price_change'] = df['close'].pct_change()
    
    # Sum Over N Days for Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Determine Median Volume over Last N Days
    median_volume = df['volume'].rolling(window=N).median()
    
    # Exclude Days Below Median Volume
    df_filtered = df[df['volume'] >= median_volume]
    
    # Calculate High-Low Range
    df_filtered['high_low_range'] = df_filtered['high'] - df_filtered['low']
    
    # Calculate Absolute Close-to-Close Return
    df_filtered['abs_close_to_close_return'] = abs(df_filtered['close'].pct_change())
    
    # Calculate Combined Value
    df_filtered['combined_value'] = df_filtered['high_low_range'] - df_filtered['abs_close_to_close_return']
    
    # Calculate Combined Value Momentum
    df_filtered['combined_value_momentum'] = df_filtered['combined_value'].pct_change()
    
    # Calculate Intraday Volatility
    df_filtered['intraday_volatility'] = df_filtered['high_low_range'].rolling(window=10).mean()
    
    # Calculate Momentum
    df_filtered['momentum'] = df_filtered['close'].pct_change().rolling(window=M).sum()
    
    # Generate Final Alpha Factor
    df_filtered['alpha_factor'] = (df_filtered['intraday_volatility'] * df_filtered['momentum']) + df_filtered['combined_value_momentum']
    
    return df_filtered['alpha_factor']
