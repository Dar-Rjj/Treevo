import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Smooth High-Low Range
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=15).mean()

    # Generate Intraday Momentum Signal
    df['Intraday_Momentum_Signal'] = (df['close'] / df['open']).apply(lambda x: 1.6 if x > 1.2 else 0.4)

    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=15).mean()

    # Calculate Absolute Close-to-Close Return
    df['Absolute_Close_To_Close_Return'] = abs(df['close'] - df['close'].shift(1))

    # Combine High-Low and Close-to-Close Returns
    df['Combined_Value'] = df['High_Low_Range'] + df['Absolute_Close_To_Close_Return']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=15).mean()

    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)

    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()

    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Smoothed_Volume_Weighted_High_Low_Range'] 
        + df['Combined_Value_Momentum'] 
        + df['Intraday_Volatility'] 
        - (df['open'] - df['close'].shift(1))
    )

    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Intermediate_Alpha_Factor_Adjusted'] = df['Intermediate_Alpha_Factor'] / df['Volatility']

    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = df['Intermediate_Alpha_Factor_Adjusted'] * df['Intraday_Momentum_Signal']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']) * df['Intraday_Volatility']

    # Weight by Volume
    df['Volume_Weighted_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['volume']
    df['Final_Alpha_Factor_Smoothed'] = df['Volume_Weighted_Final_Alpha_Factor'].ewm(span=5).mean()

    # Introduce Additional Momentum Component
    df['50_MA'] = df['close'].rolling(window=50).mean()
    df['200_MA'] = df['close'].rolling(window=200).mean()
    df['Long_Term_Momentum_Adjustment'] = np.where(df['50_MA'] > df['200_MA'], 1.3, 0.7)
    df['Final_Alpha_Factor_Adjusted_Long_Term'] = df['Final_Alpha_Factor_Smoothed'] * df['Long_Term_Momentum_Adjustment']

    def rsi(series, period):
        delta = series.diff().dropna()
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=period).mean()
        avg_loss = loss.rolling(window=period).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
