import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] + df['abs_close_return']
    
    # Calculate Volume Adjusted Intraday Range
    df['vol_adj_intraday_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=15).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = df['high_low_range'] / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['vol_adj_intraday_range'] + 
                                df['combined_value_momentum'] + 
                                df['intraday_range_ratio'] - 
                                abs(df['open'] - df['close'].shift(1)))
    
    # Compare with Baseline (Dynamic Threshold, e.g., Intraday Range Ratio's Median)
    median_intraday_range_ratio = df['intraday_range_ratio'].median()
    df['signal'] = np.where(df['intraday_range_ratio'] > median_intraday_range_ratio, 1.7, 0.3)
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha'] = df['intermediate_alpha'] * df['signal']
    
    # Generate Final Alpha Factor
    df['final_alpha'] = df['intraday_volatility'] * (df['adjusted_intermediate_alpha'] + df['close_price_momentum'])
    
    # Adjust Final Alpha Factor by Volume
    df['ema_volume'] = df['volume'].ewm(span=10).mean()
    df['final_alpha_adjusted'] = df['final_alpha'] * df['ema_volume']
    
    return df['final_alpha_adjusted']
