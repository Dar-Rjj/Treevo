import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change Velocity
    df['price_change_velocity'] = df['close'] - df['close'].shift(5)

    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))

    # Calculate Volume Moving Average for the last 7 days
    df['volume_moving_average'] = df['volume'].rolling(window=7).mean()

    # Combine Price Change Velocity, Volume Acceleration, and Volume Moving Average
    df['alpha_factor'] = (df['price_change_velocity'] * df['volume_acceleration']) / df['volume_moving_average']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
