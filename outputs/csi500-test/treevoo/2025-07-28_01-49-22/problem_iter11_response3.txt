```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    df['prev_intraday_range'] = df['intraday_range'].shift(1)
    
    # Calculate Intraday Range Momentum
    df['intraday_momentum'] = (df['intraday_range'] / df['prev_intraday_range']) - 1
    
    # Calculate Close-Open Spread Ratio
    df['close_open_spread'] = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate Price Change Velocity
    df['price_velocity'] = df['close'] / df['close'].shift(5)
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'] / df['volume'].shift(1) / df['volume'].shift(2)
    
    # Calculate High-Low Range Volatility
    n = 10
    df['hl_volatility'] = df['intraday_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['hl_volatility'] * df['volume'].rolling(window=n).mean()
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    df['momentum_volume'] = (df['intraday_momentum'] * df['price_velocity'] * df['volume'])
    
    # Smooth High-Low Range Ratio
    df['smoothed_hl_ratio'] = df['intraday_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = df['intraday_momentum'].apply(lambda x: 1 if x > 1.0 else -1 if x < 1.0 else 0)
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['intraday_range'] - df['abs_close_return']
    
    # Calculate Combined Value Momentum
    df['combined_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Synthesize Alpha Factor
    df['alpha_factor'] = df['adjusted_volatility'] * (df['combined_momentum'] + df['close_momentum'])
    
    # Final Alpha Factor
    df['final_alpha'] = df['alpha_factor'] * df['volume']
    
    return df['final_alpha']
```
