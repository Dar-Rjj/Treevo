```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Calculate Short-Term and Long-Term Moving Averages of Daily Returns
    short_term_window = 5
    long_term_window = 30
    
    df['short_term_ma'] = df['daily_return'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['daily_return'].rolling(window=long_term_window).mean()
    
    # Subtract Short-Term Moving Average from Long-Term Moving Average
    df['ma_diff'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close'] = np.abs(df['daily_return'])
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_range'] = df['high_low_range'] - df['abs_close_to_close']
    
    # Calculate Volume-Adjusted Intraday Range
    df['volume_adjusted_range'] = df['high_low_range'] * df['volume']
    
    # Generate Alpha Factor from Momentum
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    ema_window = 10
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=ema_window, adjust=False).mean()
    
    dynamic_threshold = df['smoothed_high_low_range_ratio'].rolling(window=ema_window).mean()
    
    df['momentum_signal'] = np.where(df['smoothed_high_low_range_ratio'] > dynamic_threshold, 1, -1)
    df['alpha_factor_momentum'] = df['momentum_signal'] * df['volume']
    
    # Adjust Alpha Factor Based on Intraday Range Ratio
    intraday_range_ratio = df['high_low_range'] / (df['high'] + df['low'])
    df['alpha_factor_momentum'] = np.where(intraday_range_ratio > 0.6, df['alpha_factor_momentum'] * 1.3,
                                           np.where(intraday_range_ratio < 0.4, df['alpha_factor_momentum'] * 0.7,
                                                    df['alpha_factor_momentum']))
    
    # Introduce Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Weighted Intraday Volatility
    df['intraday_volatility_weighted'] = df['high_low_range'] * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume-Adjusted Close Price Momentum
    df['volume_adjusted_close_momentum'] = df['daily_return'] * (df['volume'] / df['volume'].shift(1))
    
    # Adjust Alpha Factor by Close-Price Change
    df['alpha_factor_momentum'] = df['alpha_factor_momentum'] * df['daily_return']
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.1
    df['alpha_factor_momentum'] = df['alpha_factor_momentum'] + fixed_weight * df['close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    n_period = 10
    df['average_volume'] = df['volume'].rolling(window=n_period).mean()
    df['volume_adjusted_high_low'] = df['high_low_range'] * df['average_volume']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['short_term_ma'] - df['long_term_ma']) * (
        df['alpha_factor_momentum'] + 
        df['volume_adjusted_close_momentum'] + 
        df['volume_momentum'] + 
        df['volume_adjusted_high_low']
    )
    
    return df['final_alpha_factor']
```
