```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    n = 15  # n-period
    high_low_range_volatility = daily_high_low_range.rolling(window=n).sum() / n
    
    # Smooth High-Low Range Volatility
    ema_window_size = 15
    smoothed_high_low_range_volatility = high_low_range_volatility.ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    close_open_difference = df['close'] - df['open']
    weighted_close_open_difference = close_open_difference * 0.5  # Use a fixed weight
    combined_volatility = smoothed_high_low_range_volatility + weighted_close_open_difference
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = volume_adjusted_intraday_range_ratio - close_to_open_change
    
    # Compare with Baseline and Adjust Combined Alpha Factor
    positive_signal = (combined_volatility > 0.8)
    negative_signal = (combined_volatility < 0.8)
    adjusted_combined_alpha_factor = combined_alpha_factor.where(positive_signal, combined_alpha_factor * 0.4) * 1.6
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_alpha_factor * df['volume']
    
    return final_alpha_factor
```
