```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    smooth_high_low_ratio = high_low_ratio.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_momentum_signal = (smooth_high_low_ratio > 1.0).astype(int) - (smooth_high_low_ratio < 1.0).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Generate Alpha Factor
    alpha_factor = (smoothed_volume_weighted_high_low_range > 1.0).astype(int) - (smoothed_volume_weighted_high_low_range < 1.0).astype(int)
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    alpha_factor_adjusted = alpha_factor / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = np.abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    final_alpha_factor = intraday_volatility * (combined_value_momentum + close_price_momentum)
    
    # Weight by Volume
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
```
