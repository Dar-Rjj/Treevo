```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute rolling averages
    df['short_mavg'] = df['close'].rolling(window=5).mean()
    df['long_mavg'] = df['close'].rolling(window=20).mean()
    df['mavg_diff'] = df['short_mavg'] - df['long_mavg']
    
    # Compute rolling volatility
    df['short_vol'] = df['daily_returns'].rolling(window=5).std()
    df['long_vol'] = df['daily_returns'].rolling(window=20).std()
    
    # Calculate daily range and its percentage of close price
    df['daily_range'] = df['high'] - df['low']
    df['range_percent_close'] = df['daily_range'] / df['close']
    
    # Compute moving averages of daily range
    df['range_5d_avg'] = df['range_percent_close'].rolling(window=5).mean()
    df['range_20d_avg'] = df['range_percent_close'].rolling(window=20).mean()
    
    # Calculate volume relative to average
    df['vol_5d_avg'] = df['volume'].rolling(window=5).mean()
    df['vol_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['vol_ratio_5d'] = df['volume'] / df['vol_5d_avg']
    df['vol_ratio_20d'] = df['volume'] / df['vol_20d_avg']
    
    # Calculate amount relative to average
    df['amount_5d_avg'] = df['amount'].rolling(window=5).mean()
    df['amount_20d_avg'] = df['amount'].rolling(window=20).mean()
    df['amount_ratio_5d'] = df['amount'] / df['amount_5d_avg']
    df['amount_ratio_20d'] = df['amount'] / df['amount_20d_avg']
    
    # Calculate momentum indicators
    df['roc_10d'] = df['close'].pct_change(10)
    df['roc_30d'] = df['close'].pct_change(30)
    
    # Calculate sentiment indicators
    df['ema_5d'] = df['daily_returns'].ewm(span=5, adjust=False).mean()
    df['ema_20d'] = df['daily_returns'].ewm(span=20, adjust=False).mean()
    
    # Combine indicators with weights
    df['composite_factor'] = (
        0.2 * df['mavg_diff'] + 
        0.1 * df['short_vol'] + 
        0.1 * df['long_vol'] + 
        0.1 * df['range_5d_avg'] + 
        0.1 * df['range_20d_avg'] + 
        0.1 * df['vol_ratio_5d'] + 
        0.1 * df['vol_ratio_20d'] + 
        0.05 * df['amount_ratio_5d'] + 
        0.05 * df['amount_ratio_20d'] + 
        0.05 * df['roc_10d'] + 
        0.05 * df['roc_30d'] + 
        0.05 * df['ema_5d'] + 
        0.05 * df['ema_20d']
    )
    
    return df['composite_factor']
```
