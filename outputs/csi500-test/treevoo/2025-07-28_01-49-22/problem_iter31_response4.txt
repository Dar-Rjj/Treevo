    df['A/D'] = 0  # Initialize the A/D line
    df['A/D'] = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    df['A/D'] = df['A/D'].cumsum()  # Cumulative sum to get the A/D Line
    
    # Open Interest Analysis
    df['open_interest_analysis'] = (df['close'] - df['open']) / df['open']
    
    # Combine all factors into a single alpha factor by averaging them
    df['alpha_factor'] = df[['momentum', 'volatility', 'volume_momentum', 'price_reversal', 'A/D', 'open_interest_analysis']].mean(axis=1)
