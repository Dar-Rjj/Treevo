import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10 days)
    smooth_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    # Set Dynamic Threshold (20-day moving average of the smoothed ratio)
    dynamic_threshold = smooth_high_low_range_ratio.rolling(window=20).mean()
    
    # Compare with Dynamic Threshold
    momentum_signal = np.where(smooth_high_low_range_ratio > dynamic_threshold, 1, 
                               np.where(smooth_high_low_range_ratio < dynamic_threshold, -1, 0))
    
    # Apply EMA on Volume (5 days)
    ema_volume = df['volume'].ewm(span=5).mean()
    enhanced_momentum = momentum_signal * ema_volume
    
    # Calculate High-Low Range Volatility (n-period sum of daily high-low ranges divided by n, using 20 days)
    volatility = (high_low_range.rolling(window=20).sum()) / 20
    
    # Adjust Volatility by EMA of Volume (5 days)
    adjusted_volatility = volatility * ema_volume
    
    # Incorporate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Fixed Weight for the Close-Open Difference
    fixed_weight = 0.5  # Example fixed weight
    weighted_close_open_diff = fixed_weight * close_open_diff
    
    # Combine with the Adjusted Volatility
    alpha_factor = adjusted_volatility + weighted_close_open_diff
    
    return alpha_factor
