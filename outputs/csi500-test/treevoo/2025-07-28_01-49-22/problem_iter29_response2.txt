import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range
    smooth_high_low_range = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smooth_high_low_range > 1.0).astype(int) - (smooth_high_low_range < 1.0).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    alpha_factor = vol_weighted_high_low_range / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low and Close-to-Close Returns
    combined_value = smooth_high_low_range - close_to_close_return
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        vol_weighted_high_low_range +
        combined_value +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha_factor = np.where(
        alpha_factor > 1.0,
        intermediate_alpha_factor * 1.5,
        np.where(alpha_factor < 1.0, intermediate_alpha_factor * 0.5, intermediate_alpha_factor)
    )
    
    # Incorporate Dynamic Threshold Adjustment
    ten_day_ma_intraday_range = intraday_range_ratio.rolling(window=10).mean()
    dynamic_adjustment = np.where(
        intraday_range_ratio > ten_day_ma_intraday_range,
        1.5,
        0.5
    )
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * dynamic_adjustment
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_to_close_return) * volatility
    
    # Introduce Volume Momentum
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    current_vol_mom = df['volume'] / ten_day_avg_volume
    final_alpha_factor = np.where(
        current_vol_mom > 1,
        final_alpha_factor * 1.15,
        np.where(current_vol_mom < 1, final_alpha_factor * 0.85, final_alpha_factor)
    )
    
    # Weight by Volume
    volume_weighted_final_alpha = (final_alpha_factor * df['volume']).ewm(span=5).mean()
    
    return pd.Series(volume_weighted_final_alpha, index=df.index, name='alpha_factor')
