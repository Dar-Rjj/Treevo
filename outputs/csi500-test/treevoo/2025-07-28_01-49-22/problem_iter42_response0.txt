```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=20).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['smoothed_high_low_range_ratio'] > 1.3, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = (df['high_low_range'] * df['volume']).ewm(span=20).mean()
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=30).std()
    df['volume_weighted_high_low_range_vol_adj'] = df['volume_weighted_high_low_range'] / df['volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'].diff()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=30).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['volume_weighted_high_low_range_vol_adj'] +
        df['combined_value_momentum'] +
        df['intraday_range_ratio'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['intraday_signal'] == 1, 
        df['intermediate_alpha_factor'] * 2.0, 
        df['intermediate_alpha_factor'] * 0.05
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = (df['final_alpha_factor'] * df['volume']).ewm(span=10).mean()
    
    return df['volume_weighted_final_alpha_factor']
```
