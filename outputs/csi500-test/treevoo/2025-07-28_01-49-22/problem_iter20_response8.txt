import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range
    smooth_high_low_range = high_low_range.ewm(span=12).mean()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=12).mean()
    
    # Generate Alpha Factor from High-Low Range
    alpha_high_low = (smooth_high_low_range > 1.5).astype(int) - (smooth_high_low_range < 1.5).astype(int)
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_high_low_close_to_close = high_low_range - abs_close_to_close_return
    
    # Smooth Combined High-Low and Close-to-Close
    smooth_combined_high_low_close_to_close = combined_high_low_close_to_close.ewm(span=12).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = (smooth_combined_high_low_close_to_close > 1.5).astype(int) - (smooth_combined_high_low_close_to_close < 1.5).astype(int)
    combined_alpha_factor = combined_alpha_factor * (1.6 if combined_alpha_factor > 0 else 0.4)
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_high_low_close_to_close - combined_high_low_close_to_close.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=25).sum()
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1)) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread_ratio = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = volume_adjusted_intraday_range + combined_value_momentum - close_open_spread_ratio
    
    # Calculate Price Change Velocity
    price_change_velocity = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Volatility by Volume
    avg_volume = df['volume'].rolling(window=12).mean()
    adjusted_volatility = (df['high'] - df['low']) * avg_volume
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    combined_intraday_range_momentum_price_velocity = (intraday_range_momentum * price_change_velocity) * df['volume']
    
    # Combine Price, Volume, and Volatility Indicators
    combined_price_volume_volatility = (price_change_velocity * volume_acceleration) * adjusted_volatility
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_factor + close_price_momentum + intraday_range_momentum 
                          + combined_intraday_range_momentum_price_velocity + combined_price_volume_volatility)
    
    return final_alpha_factor
