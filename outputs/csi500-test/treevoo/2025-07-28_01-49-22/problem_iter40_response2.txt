import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = high_low_range * df['volume']
    ema_volume_weighted_high_low = volume_weighted_high_low.ewm(span=10).mean()
    
    # Calculate Intraday Momentum Signal
    intraday_momentum_signal = (ema_volume_weighted_high_low > 1.35).astype(int) * 2 - 1
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_volume_weighted_high_low = ema_volume_weighted_high_low / volatility
    
    # Calculate Absolute Close-to-Close Return
    absolute_close_to_close_return = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value_momentum = high_low_range + absolute_close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        adjusted_volume_weighted_high_low +
        combined_value_momentum +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    adjustment_factor = np.where(intraday_range_ratio > 0.6, 1.2, 0.8)
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * adjustment_factor
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum)
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    ema_volume_weighted_final_alpha_factor = volume_weighted_final_alpha_factor.ewm(span=7).mean()
    
    return ema_volume_weighted_final_alpha_factor
