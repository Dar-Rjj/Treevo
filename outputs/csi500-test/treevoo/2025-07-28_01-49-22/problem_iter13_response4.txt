```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    prev_close = df['close'].shift(1)
    today_open = df['open']
    close_to_open_change = today_open - prev_close
    
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    n = 10  # n-period
    high_low_range_volatility = daily_high_low_range.rolling(window=n).mean()
    
    # Adjust Volatility by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_volatility = high_low_range_volatility * avg_volume
    
    # Smooth High-Low Range Volatility with EMA
    ema_window_size = 10
    smoothed_volatility = adjusted_volatility.ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    weight = 0.5  # Fixed Weight for the Close-Open Difference
    close_open_difference = df['close'] - df['open']
    combined_volatility = smoothed_volatility + weight * close_open_difference
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = volume_adjusted_intraday_range_ratio - close_to_open_change
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    signal = np.where(smoothed_volatility > 1.0, 1.5, 0.5)
    adjusted_combined_alpha_factor = combined_alpha_factor * signal
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_alpha_factor * df['volume']
    
    return final_alpha_factor
```
