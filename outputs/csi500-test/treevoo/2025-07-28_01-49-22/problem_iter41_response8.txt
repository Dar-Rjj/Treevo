import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smoothed_high_low_range'] = df['daily_high_low_range'].ewm(span=10, adjust=False).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = df['close'] / df['open']
    df['positive_intraday_signal'] = (df['intraday_momentum_signal'] > 1.3).astype(int)
    df['negative_intraday_signal'] = (df['intraday_momentum_signal'] < 1.3).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['daily_high_low_range'] * df['volume']
    df['smoothed_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10, adjust=False).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['daily_high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['daily_high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['smoothed_volume_weighted_high_low_range'] +
        df['combined_value_momentum'] +
        df['intraday_volatility'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['positive_intraday_signal'] == 1,
        df['adjusted_intermediate_alpha_factor'] * 1.8,
        np.where(
            df['negative_intraday_signal'] == 1,
            df['adjusted_intermediate_alpha_factor'] * 0.2,
            df['adjusted_intermediate_alpha_factor']
        )
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']) * df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    df['smoothed_volume_weighted_final_alpha_factor'] = df['volume_weighted_final_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    # Introduce Additional Momentum Component
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = np.where(
        df['50_day_ma'] > df['200_day_ma'], 1, 0
    )
    df['negative_long_term_momentum'] = np.where(
        df['50_day_ma'] < df['200_day_ma'], 1, 0
    )
    
    # Adjust Final Alpha Factor based on Long-Term Momentum
    df['final_alpha_factor'] = np.where(
        df['long_term_momentum'] == 1,
        df['smoothed_volume_weighted_final_alpha_factor'] * 1.2,
        np.where(
            df['negative_long_term_momentum'] == 1,
            df['smoothed_volume_weighted_final_alpha_factor'] * 0.8,
            df['smoothed_volume_weighted_final_alpha_factor']
        )
    )
    
    # Integrate High-Low and Close-to-Close Returns
    df['integrated_returns'] = df['volume_weighted_high_low_range'] + df['abs_close_to_close_return']
    df
