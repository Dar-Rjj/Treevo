```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (
        volume_adjusted_intraday_range +
        combined_value - combined_value.shift(1) +
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    hl_ratio = high_low_range / (df['high'] + df['low'])
    hl_ratio_10_day_ma = hl_ratio.rolling(window=10).mean()
    dynamic_threshold = hl_ratio.apply(lambda x: 1.5 if x > hl_ratio_10_day_ma else 0.5)
    intermediate_alpha *= dynamic_threshold
    
    # Introduce Volume Momentum
    avg_volume_10_day = df['volume'].rolling(window=10).mean()
    volume_momentum = (df['volume'] / avg_volume_10_day).apply(lambda x: 1.1 if x > 1 else 0.9)
    intermediate_alpha *= volume_momentum
    
    # Generate Final Alpha Factor
    close_to_close_return_20_day_sum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    avg_high_low_10_day = high_low_range.rolling(window=10).mean()
    final_alpha = (intermediate_alpha + close_to_close_return_20_day_sum) * avg_high_low_10_day
    
    return final_alpha
```
