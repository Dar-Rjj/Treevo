```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with EMA (10-day window)
    smoothed_high_low_range = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smoothed_high_low_range > 1.0).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Adjust for Volatility (20-day window)
    volatility = df['close'].rolling(window=20).std()
    adjusted_smoothed_high_low_range = smoothed_high_low_range / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low and Close-to-Close Returns
    combined_value = adjusted_smoothed_high_low_range - close_to_close_return
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (smoothed_volume_weighted_high_low_range 
                                 + combined_value 
                                 + intraday_range_ratio 
                                 - (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * (intraday_signal * 0.5 + 1)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + df['close'].diff()) * volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    final_alpha_factor = volume_weighted_final_alpha_factor.ewm(span=5).mean()
    
    return final_alpha_factor
```
