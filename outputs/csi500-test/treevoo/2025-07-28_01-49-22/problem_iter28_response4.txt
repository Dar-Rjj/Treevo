```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Generate Intraday Momentum Signal
    intraday_signal = (high_low_range > 1.0).astype(int) - (high_low_range < 1.0).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = high_low_range * df['volume']
    volume_weighted_ema = volume_weighted_high_low.ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor
    alpha_factor = (volume_weighted_ema > 1.0).astype(int) - (volume_weighted_ema < 1.0).astype(int)
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    alpha_factor /= volatility
    
    # Calculate Absolute Close-to-Close Return
    close_return = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_return
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = volume_weighted_ema + combined_value + intraday_range_ratio - (df['open'] - df['close'].shift(1))
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha = np.where(intraday_range_ratio > 1.0, intermediate_alpha * 1.5, intermediate_alpha * 0.5)
    
    # Generate Final Alpha Factor
    final_alpha = volatility * (intermediate_alpha + df['close'].pct_change())
    
    # Weight by Volume
    final_alpha *= df['volume']
    
    return final_alpha
```
