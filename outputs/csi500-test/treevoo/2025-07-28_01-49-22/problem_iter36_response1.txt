```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Calculate Short-Term and Long-Term Moving Averages of Daily Returns
    short_term_window = 5
    long_term_window = 30
    df['short_term_ma'] = df['daily_return'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['daily_return'].rolling(window=long_term_window).mean()
    df['ma_diff'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close'] = df['daily_return'].abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_range'] = df['high_low_range'] - df['abs_close_to_close']
    
    # Calculate Volume Adjusted Intraday Range
    df['volume_adjusted_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['ma_diff'] + df['combined_range'] - (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor Based on Intraday Range Ratio
    intraday_range_ratio = df['high_low_range'] / (df['high'] + df['low'])
    df['adjusted_intermediate_alpha'] = df['intermediate_alpha'].where(intraday_range_ratio <= 0.6, df['intermediate_alpha'] * 1.4)
    df['adjusted_intermediate_alpha'] = df['adjusted_intermediate_alpha'].where(intraday_range_ratio >= 0.4, df['adjusted_intermediate_alpha'] * 0.6)
    
    # Introduce Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Weighted Intraday Volatility
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume-Adjusted Close Price Momentum
    df['volume_adjusted_close_momentum'] = df['close_price_momentum'] * (df['volume'] / df['volume'].shift(1))
    
    # Incorporate Volume-Adjusted Intraday Range
    df['volume_adjusted_intraday_range'] = df['high_low_range'] * (df['volume'] / df['volume'].shift(1))
    
    # Generate Final Alpha Factor
    df['final_alpha'] = df['ma_diff'] * (df['adjusted_intermediate_alpha'] + df['volume_adjusted_close_momentum'] + df['volume_momentum'] + df['volume_adjusted_intraday_range'])
    
    return df['final_alpha']
```
