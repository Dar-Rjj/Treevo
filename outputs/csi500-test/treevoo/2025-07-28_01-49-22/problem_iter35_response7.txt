```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range
    smoothed_high_low_range = daily_high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (df['high'] - df['low']) / df['close'].shift(1)
    positive_intraday_signal = intraday_signal > 1.3
    negative_intraday_signal = intraday_signal < 1.3
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (daily_high_low_range * df['volume']).ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    combined_value = daily_high_low_range + abs_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = daily_high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (volume_weighted_high_low_range + 
                                 combined_value_momentum + 
                                 intraday_signal - 
                                 (df['open'] - df['close'].shift(1)))
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor / volatility
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    adjusted_intermediate_alpha_factor[positive_intraday_signal] *= 1.8
    adjusted_intermediate_alpha_factor[negative_intraday_signal] *= 0.2
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * intraday_volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = (final_alpha_factor * df['volume']).ewm(span=5).mean()
    
    # Introduce Additional Momentum Component
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    positive_long_term_momentum = ma_50 > ma_200
    negative_long_term_momentum = ma_50 < ma_200
    
    # Adjust Final Alpha Factor based on Long-Term Momentum
    final_alpha_factor[positive_long_term_momentum] *= 1.2
    final_alpha_factor[negative_long_term_momentum] *= 0.8
    
    return final_alpha_factor
```
