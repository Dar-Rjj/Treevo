```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['HL_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10-day window)
    df['Smoothed_HL_Range_Ratio'] = df['HL_Range_Ratio'].ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_HL_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_HL_Range'] = df['Volume_Weighted_HL_Range'].ewm(span=10).mean()
    
    # Generate Alpha Factor from High-Low Range
    df['Alpha_Factor_HL'] = np.where(df['Smoothed_HL_Range_Ratio'] > 1.0, 1, -1)
    
    # Calculate Volatility (20-day standard deviation of close price)
    df['Volatility'] = df['close'].rolling(window=20).std()
    
    # Normalize Alpha Factor by Volatility
    df['Normalized_Alpha_Factor'] = df['Alpha_Fctor_HL'] / df['Volatility']
    
    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['Volume_Adjusted_Intraday_Range_Ratio'] = df['Intraday_Range_Ratio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['Prev_Close_to_Open_Change'] = df['open'] - df['close'].shift(1)
    
    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_to_Close_Return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_HL_and_CtoC'] = df['High_Low_Range'] - df['Abs_Close_to_Close_Return']
    
    # Smooth Combined High-Low and Close-to-Close
    df['Smoothed_Combined_HL_CtoC'] = df['Combined_HL_and_CtoC'].ewm(span=10).mean()
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Generate Combined Alpha Factor
    df['Combined_Alpha_Factor'] = (df['Volume_Adjusted_Intraday_Range_Ratio'] - df['Prev_Close_to_Open_Change'])
    df['Combined_Alpha_Factor_Signal'] = np.where(df['Smoothed_Combined_HL_CtoC'] > 1.0, 1.5, 0.5)
    df['Adjusted_Combined_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Combined_Alpha_Factor_Signal']
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_HL_and_CtoC'] - df['Combined_HL_and_CtoC'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Combined_Alpha_Factor'] * (df['Intraday_Volatility'] * (df['Combined_Value_Momentum'] + df['Close_Price_Momentum']))
    
    return df['Final_Alpha_Factor']
```
