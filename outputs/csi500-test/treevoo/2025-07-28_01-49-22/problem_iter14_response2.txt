```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)

    # Smooth High-Low Range Ratio with EMA (10-day window)
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()

    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['smoothed_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()

    # Generate Alpha Factor from High-Low Range
    df['alpha_factor_high_low'] = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1, -1)

    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['alpha_factor_high_low_normalized'] = df['alpha_factor_high_low'] / df['volatility']

    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Compute Volume Adjusted Intraday Range Ratio
    df['volume_adjusted_intraday_range_ratio'] = df['intraday_range_ratio'] * df['volume']

    # Calculate Previous Day's Close-to-Open Change
    df['prev_day_close_to_open_change'] = df['open'] - df['close'].shift(1)

    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))

    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_high_low_and_close_to_close'] = df['high_low_range'] - df['abs_close_to_close_return']

    # Smooth Combined High-Low and Close-to-Close
    df['smoothed_combined_high_low_and_close_to_close'] = df['combined_high_low_and_close_to_close'].ewm(span=10).mean()

    # Generate Combined Alpha Factor
    combined_alpha_factor = (df['volume_adjusted_intraday_range_ratio'] - df['prev_day_close_to_open_change'])
    df['combined_alpha_factor'] = np.where(df['smoothed_combined_high_low_and_close_to_close'] > 1.0,
                                           1.5 * combined_alpha_factor,
                                           0.5 * combined_alpha_factor)

    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_high_low_and_close_to_close'] - df['combined_high_low_and_close_to_close'].shift(1)

    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_alpha_factor'] * 
                                (df['combined_value_momentum'] + df['close_price_momentum'])) * df['volume']

    return df['final_alpha_factor']
```
