```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['vol_adjusted_intraday_range_ratio'] = df['intraday_range_ratio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['prev_close_to_open_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    n = 20  # n-period
    df['high_low_range_volatility'] = df['daily_high_low_range'].rolling(window=n).mean()
    
    # Smooth High-Low Range Volatility using EMA
    ema_window_size = 20
    df['smoothed_high_low_range_volatility'] = df['high_low_range_volatility'].ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.5  # You can adjust this weight
    df['weighted_close_open_diff'] = fixed_weight * df['close_open_diff']
    df['incorporated_smoothed_volatility'] = df['smoothed_high_low_range_volatility'] + df['weighted_close_open_diff']
    
    # Generate Combined Alpha Factor
    df['combined_alpha_factor'] = df['vol_adjusted_intraday_range_ratio'] - df['prev_close_to_open_change']
    
    # Compare with Baseline (Dynamic Threshold: EMA of Smoothed High-Low Range Volatility)
    df['signal'] = df['smoothed_high_low_range_volatility'].apply(lambda x: 1 if x > df['smoothed_high_low_range_volatility'].ewm(span=ema_window_size, adjust=False).mean().iloc[-1] else -1)
    df['adjusted_combined_alpha_factor'] = df['combined_alpha_factor'] * df['signal'].apply(lambda x: 1.7 if x > 0 else 0.3)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_combined_alpha_factor'] * df['volume']
    
    return df['final_alpha_factor']
```
