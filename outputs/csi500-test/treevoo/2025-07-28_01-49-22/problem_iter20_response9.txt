```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['Intraday_Momentum_Signal'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > 1.2, 1, -1)
    
    # Calculate Absolute Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] - df['Close_to_Close_Return']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=5).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum
    df['N_day_Returns'] = (df['close'] / df['close'].shift(1)) - 1
    df['Close_Price_Momentum'] = df['N_day_Returns'].rolling(window=30).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=30).median()
    df = df[df['volume'] >= median_volume]
    
    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=30).std()
    df['Adjusted_Alpha'] = df['Intraday_Volatility'] * (df['Combined_Value_Momentum'] + df['Close_Price_Momentum']) / df['Volatility']
    
    # Incorporate Open and Close Prices
    df['Open_to_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Open_to_Close_Momentum'] = df['Open_to_Close_Return'].rolling(window=30).sum()
    
    # Final Alpha Factor
    df['Final_Alpha'] = (df['Intraday_Volatility'] * 
                         (df['Combined_Value_Momentum'] + 
                          df['Close_Price_Momentum'] + 
                          df['Open_to_Close_Momentum'])) * df['volume']
    
    return df['Final_Alpha']
```
