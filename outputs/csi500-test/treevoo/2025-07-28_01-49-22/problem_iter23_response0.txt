```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'].shift(1) - df['close']).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    hl_ratio_10_day_ma = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    dynamic_threshold_signal = (high_low_range / (df['high'] + df['low'])) - hl_ratio_10_day_ma
    adjusted_intermediate_alpha = intermediate_alpha * (dynamic_threshold_signal > 0).astype(int)
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    high_low_range_ratio = high_low_range / (df['high'] + df['low'])
    adjustment_factor = high_low_range_ratio.apply(lambda x: 1.5 if x > 1.0 else 0.5)
    adjusted_intermediate_alpha = adjusted_intermediate_alpha * adjustment_factor
    
    # Introduce Volume Momentum
    avg_volume_10_day = df['volume'].rolling(window=10).mean()
    volume_momentum = (df['volume'] / avg_volume_10_day)
    final_alpha_factor = adjusted_intermediate_alpha * (1 + 0.15 * (volume_momentum > 1).astype(int) - 0.15 * (volume_momentum < 1).astype(int))
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (final_alpha_factor + close_price_momentum)
    
    # Adjust Final Alpha Factor by Volume
    avg_volume_n_period = df['volume'].rolling(window=30).mean()
    final_alpha_factor = final_alpha_factor * avg_volume_n_period
    
    return final_alpha_factor
```
