import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range + abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=20).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=30).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    hl_ratio_10_day_ma = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    dynamic_threshold = intermediate_alpha_factor.where(
        (high_low_range / (df['high'] + df['low'])) > hl_ratio_10_day_ma, 
        -intermediate_alpha_factor
    )
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjusted_intermediate_alpha_factor = dynamic_threshold.where(
        (high_low_range / (df['high'] + df['low'])) > 0.7, 
        dynamic_threshold * 0.6
    ).where(
        (high_low_range / (df['high'] + df['low'])) <= 0.7, 
        dynamic_threshold * 1.4
    )
    
    # Introduce Volume Momentum
    avg_volume_10_day = df['volume'].rolling(window=10).mean()
    volume_momentum = (df['volume'] / avg_volume_10_day).apply(lambda x: 1.2 if x > 1 else 0.8)
    final_alpha_factor = adjusted_intermediate_alpha_factor * volume_momentum + close_price_momentum
    
    # Adjust Final Alpha Factor by Volume
    avg_volume_n_period = df['volume'].expanding().mean()
    final_alpha_factor_adjusted = final_alpha_factor * avg_volume_n_period
    
    return final_alpha_factor_adjusted
