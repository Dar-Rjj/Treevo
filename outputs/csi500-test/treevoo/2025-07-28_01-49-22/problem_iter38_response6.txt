```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['Intraday_Momentum_Signal'] = (df['close'] / df['open']).shift(1)
    df['Intraday_Signal'] = df['Intraday_Momentum_Signal'] > 1.3
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_to_Close_Return'] = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low and Close-to-Close Returns
    df['Combined_Value'] = df['High_Low_Range'] + df['Abs_Close_to_Close_Return']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Smoothed_Volume_Weighted_High_Low_Range'] 
        + df['Combined_Value_Momentum'] 
        + df['Intraday_Volatility'] 
        - (df['open'] - df['close'].shift(1))
    )
    
    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Intermediate_Alpha_Factor_Adjusted'] = df['Intermediate_Alpha_Factor'] / df['Volatility']
    
    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = df.apply(
        lambda row: row['Intermediate_Alpha_Factor_Adjusted'] * 1.8 if row['Intraday_Signal'] else row['Intermediate_Alpha_Factor_Adjusted'] * 0.2,
        axis=1
    )
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']) * df['Intraday_Volatility']
    
    # Weight by Volume
    df['Volume_Weighted_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['volume']
    df['Final_Alpha_Factor_EMA'] = df['Volume_Weighted_Final_Alpha_Factor'].ewm(span=5).mean()
    
    # Introduce Additional Momentum Component
    df['50_MA'] = df['close'].rolling(window=50).mean()
    df['200_MA'] = df['close'].rolling(window=200).mean()
    df['Long_Term_Momentum'] = df['50_MA'] > df['200_MA']
    
    # Adjust Final Alpha Factor
    df['Final_Alpha_Factor_Adjusted'] = df.apply(
        lambda row: row['Final_Alpha_Factor_EMA'] * 1.2 if row['Long_Term_Momentum'] else row['Final_Alpha_Factor_EMA'] * 0.8,
        axis=1
    )
    
    # Integrate High-Low and Close-to-Close Returns
    df['Integrated_Value'] = df['Volume_Weighted_High_Low_Range'] + df['Abs_Close_to_Close_Return']
    df['Integrated_Value_EMA'] = df['Integrated_Value'].ewm(span=10).mean()
    
    return df['Final_Alpha_Factor_Adjusted']
```
