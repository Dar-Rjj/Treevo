```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA
    ema_window = 10
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=ema_window).mean()
    
    # Generate Alpha Factor
    dynamic_threshold = df['smoothed_high_low_range_ratio'].mean()
    df['alpha_factor'] = (df['smoothed_high_low_range_ratio'] > dynamic_threshold).astype(int) - (df['smoothed_high_low_range_ratio'] < dynamic_threshold).astype(int)
    df['alpha_factor'] *= df['volume']
    
    # Calculate Daily Close-Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Adjust Alpha Factor by Close-Price Change
    df['alpha_factor'] *= df['close_price_change']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'] - df['close'].shift(1)
    
    # Adjust Close-to-Close Return by Volume
    df['volume_adjusted_close_to_close_return'] = df['close_to_close_return'] * df['volume']
    
    # Combine Intraday Range and Volume-Adjusted Close-to-Close Return
    df['combined_value'] = df['intraday_range'] + df['volume_adjusted_close_to_close_return']
    
    # Calculate 10-day SMA of Combined Values
    sma_window = 10
    df['sma_combined_value'] = df['combined_value'].rolling(window=sma_window).mean()
    
    # Final Output
    df['final_output'] = df['combined_value'] - df['sma_combined_value']
    
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    n_period = 10
    df['high_low_range_volatility'] = df['daily_high_low_range'].rolling(window=n_period).sum() / n_period
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=n_period).mean()
    df['volatility_adjusted_by_volume'] = df['high_low_range_volatility'] * df['avg_volume']
    
    return df['final_output'], df['volatility_adjusted_by_volume']
```
