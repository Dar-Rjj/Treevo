```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High-Low Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['High-Low Range Ratio'] = df['High-Low Range'] / df['High-Low Range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA with a window size of 10 days
    ema_window_size = 10
    df['Smoothed High-Low Range Ratio'] = df['High-Low Range Ratio'].ewm(span=ema_window_size, min_periods=ema_window_size).mean()
    
    # Generate Alpha Factor based on comparison with dynamic threshold (median of EMA)
    dynamic_threshold = df['Smoothed High-Low Range Ratio'].rolling(window=ema_window_size, min_periods=ema_window_size).median()
    df['Alpha Factor'] = np.where(df['Smoothed High-Low Range Ratio'] > dynamic_threshold, 1, -1)
    df['Alpha Factor'] *= np.log(df['volume'])
    
    # Calculate Daily Close-Price Change
    df['Close-Price Change'] = df['close'] - df['close'].shift(1)
    
    # Adjust Alpha Factor by Close-Price Change
    df['Adjusted Alpha Factor'] = df['Alpha Factor'] * df['Close-Price Change']
    
    # Calculate n-period High-Low Range Volatility
    n = 10
    df['n-period High-Low Range'] = df['High-Low Range'].rolling(window=n, min_periods=n).sum() / n
    
    # Calculate Standard Deviation of Volume (n-period)
    df['Volume Std'] = df['volume'].rolling(window=n, min_periods=n).std()
    
    # Adjust Volatility by Volume
    df['Volatility by Volume'] = df['n-period High-Low Range'] * df['Volume Std']
    
    # Final Alpha Factor
    df['Final Alpha Factor'] = df['Adjusted Alpha Factor'] / df['Volatility by Volume']
    
    return df['Final Alpha Factor'].dropna()
```
