```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Smoothed High-Low Range Ratio
    smoothed_high_low_range = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smoothed_high_low_range > 1.0).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Generate Alpha Factor based on Volume-Weighted High-Low Range
    alpha_factor = (volume_weighted_high_low_range > 1.0).astype(int) * 2 - 1
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    alpha_factor /= volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value.diff()
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = volume_weighted_high_low_range + combined_value_momentum + intraday_range_ratio - (df['open'] - df['close'].shift(1))
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha_factor = intermediate_alpha_factor.where(intraday_range_ratio > 1.0, intermediate_alpha_factor * 0.5)
    intermediate_alpha_factor = intermediate_alpha_factor.where(intraday_range_ratio <= 1.0, intermediate_alpha_factor * 1.5)
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (intermediate_alpha_factor + close_price_momentum)
    
    # Weight by Volume
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
```
