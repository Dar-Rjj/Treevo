```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range + abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value.diff()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Compute Previous Day's Close-to-Open Change
    close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Calculate Daily Close-Open Difference
    close_open_difference = df['close'] - df['open']
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = high_low_range.rolling(window=20).sum() / 20
    
    # Smooth High-Low Range Volatility with EMA
    ema_window_size = 15
    smoothed_high_low_range_volatility = high_low_range_volatility.ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    weighted_close_open_difference = 0.5 * close_open_difference
    combined_volatility = smoothed_high_low_range_volatility + weighted_close_open_difference
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        close_to_open_change
    )
    
    # Adjust Intermediate Alpha Factor
    adjustment_factor = np.where(intraday_range_ratio > 0.5, 1.2, 0.8)
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * adjustment_factor
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    dynamic_threshold = combined_volatility.ewm(span=ema_window_size, adjust=False).mean()
    signal = np.where(combined_volatility > dynamic_threshold, 1, -1)
    
    # Adjust Combined Alpha Factor
    combined_alpha_factor = np.where(signal == 1, 1.4 * adjusted_intermediate_alpha_factor, 0.6 * adjusted_intermediate_alpha_factor)
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (combined_alpha_factor + close_price_momentum) * df['volume']
    
    return final_alpha_factor
```
