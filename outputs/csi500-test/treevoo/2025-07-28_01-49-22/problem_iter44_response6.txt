```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low price
    price_diff = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the weighted volume by amount
    weighted_volume = df['volume'] * (df['amount'] / df['close'])
    
    # Create an alpha factor that is the ratio of price difference to average price, scaled by the weighted volume
    alpha_factor = (price_diff / avg_price) * weighted_volume
    
    return alpha_factor
```
