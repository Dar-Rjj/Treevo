```python
def heuristics_v2(df, n=10):
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = daily_high_low_range.rolling(window=n).sum() - daily_high_low_range
    
    # Adjust Momentum by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_momentum = high_low_momentum / avg_volume
    
    return adjusted_momentum
```
