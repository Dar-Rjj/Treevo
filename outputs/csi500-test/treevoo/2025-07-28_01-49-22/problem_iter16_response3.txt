```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    df['prev_intraday_range'] = df['intraday_range'].shift(1)
    
    # Calculate Intraday Range Momentum
    df['intraday_range_momentum'] = (df['intraday_range'] / df['prev_intraday_range']) - 1
    
    # Calculate Close-Open Spread Ratio
    df['close_open_spread'] = df['close'] - df['open']
    df['prev_close_open_spread'] = df['close_open_spread'].shift(1)
    df['close_open_spread_ratio'] = df['close_open_spread'] * df['prev_close_open_spread']
    
    # Calculate Price Change Velocity
    df['price_change_velocity'] = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate High-Low Range Volatility
    n = 10
    df['hl_range_volatility'] = df['intraday_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_volatility'] = df['hl_range_volatility'] * df['avg_volume']
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    df['combined_momentum'] = df['intraday_range_momentum'] * df['price_change_velocity'] * df['volume']
    
    # Smooth High-Low Range Ratio
    df['smoothed_hl_range'] = df['intraday_range'].ewm(span=10, adjust=False).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = df['intraday_range_momentum'].apply(lambda x: 1 if x > 1.0 else -1 if x < 1.0 else 0)
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['intraday_range'] - df['abs_close_to_close_return']
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['n_day_returns'] = (df['close'] / df['close'].shift(1)) - 1
    df['close_price_momentum'] = df['n_day_returns'].rolling(window=20).sum()
    
    # Synthesize Alpha Factor
    df['alpha_factor'] = df['adjusted_volatility'] * (df['combined_value_momentum'] + df['close_price_momentum'])
    
    # Weight by Volume
    df['final_alpha_factor'] = df['alpha_factor'] * df['volume']
    
    return df['final_alpha_factor']
```
