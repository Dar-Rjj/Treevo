import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Range
    prev_intraday_range = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Range Momentum
    range_momentum = (intraday_range / prev_intraday_range) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread = (df['close'] - df['open']) * ((df['close'].shift(1) - df['open'].shift(1)))
    
    # Calculate Price Change Velocity
    price_change_velocity = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - df['volume'].shift(1) - df['volume'].shift(2)
    
    # Calculate High-Low Range Volatility
    n = 20  # Number of periods
    high_low_volatility = (intraday_range.rolling(window=n).sum()) / n
    
    # Adjust Volatility by Volume
    avg_volume_n = df['volume'].rolling(window=n).mean()
    adjusted_volatility = high_low_volatility * avg_volume_n
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    combined_momentum = range_momentum * price_change_velocity * df['volume']
    
    # Generate Intraday Momentum Signal
    intraday_signal = (combined_momentum > 1.0).astype(int) - (combined_momentum < 1.0).astype(int)
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = intraday_range - abs(close_to_close_return)
    
    # Calculate Combined Value Momentum
    value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    n_days = 20
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=n_days).sum()
    
    # Synthesize Alpha Factor
    alpha_factor = adjusted_volatility * (value_momentum + close_price_momentum)
    
    # Final Alpha Factor, Weighted by Volume
    final_alpha_factor = alpha_factor * df['volume']
    
    return final_alpha_factor
