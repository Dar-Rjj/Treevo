```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['ema_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Smooth High-Low Range Ratio
    df['smoothed_high_low_range_ratio'] = df['intraday_range_ratio'].ewm(span=10, adjust=False).mean()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['ema_volume_weighted_high_low_range'] +
        df['combined_value_momentum'] -
        (df['close'] - df['open']) / df['open'] +
        df['intraday_range_ratio'] -
        (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    )
    
    # Adjust Intermediate Alpha Factor
    adjustment = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1.5, 0.5)
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] * adjustment
    
    # Calculate Intraday Range Momentum
    df['intraday_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['final_intermediate_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] / df['volatility']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['intraday_volatility'] * (
        df['final_intermediate_alpha_factor'] +
        df['close_price_momentum'] +
        df['intraday_range_momentum']
    )
    
    # Weight by Volume
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    
    return df['final_alpha_factor']
```
