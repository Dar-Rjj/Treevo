```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    
    # Compute Volume Adjusted Intraday Return
    df['VolumeAdjustedIntradayReturn'] = df['IntradayReturn'] * df['volume']

    # Calculate Previous Day's Close-to-Open Gap
    df['PrevCloseToOpenGap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine for Preliminary Alpha Factor
    df['PreliminaryAlphaFactor'] = df['VolumeAdjustedIntradayReturn'] - df['PrevCloseToOpenGap']

    # Calculate Price Change Velocity
    df['PriceChangeVelocity'] = (df['close'] - df['close'].shift(5))

    # Calculate Volume Acceleration
    df['VolumeAcceleration'] = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))

    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['PriceChangeVelocity'] * df['VolumeAcceleration']) * df['PreliminaryAlphaFactor']

    return df['FinalAlphaFactor']
```
