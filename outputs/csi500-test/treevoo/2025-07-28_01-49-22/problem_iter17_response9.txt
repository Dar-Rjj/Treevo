```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10-day window)
    df['Smooth_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['Intraday_Signal'] = np.where(df['Smooth_High_Low_Range_Ratio'] > 1.0, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smooth_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=10).mean()
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = np.where(df['Smooth_Volume_Weighted_High_Low_Range'] > 1.0, 1, -1)
    
    # Adjust for Volatility (20-day window)
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Alpha_Factor_Adjusted'] = df['Alpha_Factor'] / df['Volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['Close_Return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] - df['Close_Return']
    
    # Calculate Intraday Volatility (Average of (High - Low) over 10 days)
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum (last 20 days)
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Smooth_Volume_Weighted_High_Low_Range'] +
        df['Combined_Value_Momentum'] +
        df['Intraday_Range_Ratio'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    df['Baseline_Comparison'] = np.where(df['Intraday_Range_Ratio'] > 1.0, 1, -1)
    
    # Adjust Intermediate Alpha Factor based on Baseline Comparison
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(
        df['Baseline_Comparison'] == 1, 
        df['Intermediate_Alpha_Factor'] * 1.5, 
        df['Intermediate_Alpha_Factor'] * 0.5
    )
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Intraday_Volatility'] * (
        df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']
    )
    
    # Weight by Volume
    df['Final_Alpha_Factor'] *= df['volume']
    
    return df['Final_Alpha_Factor']
```
