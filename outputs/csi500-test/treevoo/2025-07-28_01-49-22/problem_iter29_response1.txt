import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA (10 days)
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10, adjust=False).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_to_Close_Return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] + df['Abs_Close_to_Close_Return']
    
    # Calculate Volume Adjusted Intraday Range
    df['Volume_Adjusted_Intraday_Range'] = df['High_Low_Range'] * df['volume']
    
    # Calculate Intraday Volatility (average of (High - Low) over 10 days)
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Alpha Factor from Enhanced Momentum
    df['Dynamic_Threshold'] = df['Smoothed_High_Low_Range_Ratio'].rolling(window=20).mean()
    df['Enhanced_Momentum_Signal'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > df['Dynamic_Threshold'], 1, -1)
    
    # Calculate EMA of Volume (5 days)
    df['EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Adjusted_Volatility'] = df['Intraday_Volatility'] + 0.3 * df['Close_Open_Diff']
    
    # Calculate High-Low Range Volatility
    df['High_Low_Range_Volatility'] = df['High_Low_Range'].rolling(window=10).sum() / 10
    
    # Smooth High-Low Range Volatility using EMA (15 days)
    df['Smoothed_High_Low_Range_Volatility'] = df['High_Low_Range_Volatility'].ewm(span=15, adjust=False).mean()
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Volume_Adjusted_Intraday_Range'] +
        df['Combined_Value_Momentum'] +
        df['High_Low_Range_Ratio'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor based on High-Low Range Ratio
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(
        (df['High_Low_Range'] / (df['high'] + df['low'])) > 0.5,
        df['Intermediate_Alpha_Factor'] * 1.2,
        df['Intermediate_Alpha_Factor'] * 0.8
    )
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    df['Baseline_Signal'] = np.where(
        df['Smoothed_High_Low_Range_Volatility'] > df['Dynamic_Threshold'],
        1, -1
    )
    
    # Adjust Combined Alpha Factor
    df['Adjusted_Combined_Alpha_Factor'] = np.where(
        df['Baseline_Signal'] == 1,
        df['Adjusted_Intermediate_Alpha_Factor'] * 1.4,
        df['Adjusted_Intermediate_Alpha_Factor'] * 0.6
    )
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Intraday_Volatility'] *
        (df['Adjusted_Combined_Alpha_Factor'] + df['Close_Price_Momentum']) *
        df['EMA_Volume']
    )
    
    return df['Final_Alpha
