```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Alpha Factor from Momentum
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    df['ema_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    df['dynamic_threshold'] = df['ema_high_low_range_ratio'].rolling(window=10).mean()
    df['momentum_signal'] = np.where(df['ema_high_low_range_ratio'] > df['dynamic_threshold'], 1, -1)
    df['momentum_factor'] = df['momentum_signal'] * df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['adjusted_momentum_factor'] = df['momentum_factor'] * df['close_price_change']
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.5  # Fixed weight for the close-open difference
    df['combined_alpha'] = df['adjusted_momentum_factor'] + (fixed_weight * df['close_open_diff'])
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    n_period = 10
    df['avg_volume'] = df['volume'].rolling(window=n_period).mean()
    df['enhanced_alpha'] = df['high_low_range'] * df['avg_volume']
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    df['combined_enhanced_alpha'] = df['enhanced_alpha'] - df['abs_close_to_close_return']
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=30).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=n_period).median()
    df['filtered_alpha'] = np.where(df['volume'] >= median_volume, df['combined_enhanced_alpha'], 0)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=30).std()
    df['final_alpha'] = df['filtered_alpha'] / df['volatility']
    
    return df['final_alpha']
```
