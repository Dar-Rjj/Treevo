```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-based Factor
    df['momentum_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility-based Factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_20'] = df['log_returns'].rolling(window=20).std()
    
    # Volume-related Momentum
    df['volume_momentum_20'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    df['volume_momentum_20'] = df['volume_momentum_20'] * (df['close'] > df['open'])
    
    # Price Reversal Indicator
    df['price_reversal'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Accumulation/Distribution Line
    df['ad_line'] = 0
    for i in range(1, len(df)):
        prev_ad = df.loc[df.index[i-1], 'ad_line']
        df.loc[df.index[i], 'ad_line'] = prev_ad + (df.loc[df.index[i], 'close'] - df.loc[df.index[i], 'open']) / (df.loc[df.index[i], 'high'] - df.loc[df.index[i], 'low']) * df.loc[df.index[i], 'volume']
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smooth_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Intraday Momentum Signal
    df['intraday_signal'] = df['smooth_high_low_range_ratio'].apply(lambda x: 1 if x > 1.3 else (-1 if x < 1.3 else 0))
    
    # Volume-Weighted High-Low Range
    df['vol_weighted_high_low'] = df['high_low_range'] * df['volume']
    
    # Adjust for Volatility
    df['alpha_factor'] = df['vol_weighted_high_low'] / df['volatility_20']
    df['alpha_factor'] = df['alpha_factor'] * df['intraday_signal'].apply(lambda x: 1.8 if x > 0 else (0.2 if x < 0 else 1))
    
    # Absolute Close-to-Close Return
    df['abs_close_return'] = df['close'] - df['close'].shift(1)
    
    # Combined High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_return']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Close Price Momentum
    df['close_price_momentum'] = df['close'].pct_change().rolling(window=20).sum()
    
    # Intraday Range Ratio
    df['intraday_range_ratio'] = df['high_low_range'] / (df['high'] + df['low'])
    
    # Intermediate Alpha Factor
    df['intermediate_alpha'] = df['vol_weighted_high_low'] + df['combined_value'] + df['intraday_range_ratio'] - (df['open'] - df['close'].shift(1))
    df['intermediate_alpha'] = df['intermediate_alpha'] * df['intraday_signal'].apply(lambda x: 1.8 if x > 0 else (0.2 if x < 0 else 1))
    
    # Final Alpha Factor
    df['final_alpha'] = df['intraday_volatility'] * (df['intermediate_alpha'] + df['close_price_momentum'])
    
    # Weight by Volume
    df['final_alpha'] = (df['final_alpha'] * df['volume']).ewm(span=5).mean()
    
    return df['final_alpha']
```
