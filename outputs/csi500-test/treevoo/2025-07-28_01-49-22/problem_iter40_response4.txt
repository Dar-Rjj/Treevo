```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Smoothed High-Low Range
    high_low_range = df['high'] - df['low']
    ema_high_low_range = high_low_range.ewm(span=10).mean()

    # Generate Intraday Momentum Signal
    intraday_signal = (df['close'] / df['open']).shift(1)
    positive_intraday_signal = (intraday_signal > 1.3).astype(int)
    negative_intraday_signal = (intraday_signal < 1.3).astype(int)

    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = (high_low_range * df['volume']).ewm(span=10).mean()

    # Calculate Absolute Close-to-Close Return
    close_to_close_return = abs(df['close'] - df['close'].shift(1))

    # Combine High-Low and Close-to-Close Returns
    combined_value = high_low_range + close_to_close_return

    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()

    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)

    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()

    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_weighted_high_low +
        combined_value_momentum +
        (high_low_range / ema_high_low_range) -
        (df['open'] - df['close'].shift(1))
    )

    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor / volatility

    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = (
        adjusted_intermediate_alpha_factor * (1.8 * positive_intraday_signal + 0.2 * negative_intraday_signal)
    )

    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * intraday_volatility

    # Weight by Volume
    final_alpha_factor_volume_weighted = (final_alpha_factor * df['volume']).ewm(span=5).mean()

    # Introduce Additional Momentum Component
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    long_term_momentum = (ma_50 > ma_200).astype(float) * 1.2 + (ma_50 < ma_200).astype(float) * 0.8

    # Adjust Final Alpha Factor
    final_alpha_factor_adjusted = final_alpha_factor_volume_weighted * long_term_momentum

    return final_alpha_factor_adjusted
```
