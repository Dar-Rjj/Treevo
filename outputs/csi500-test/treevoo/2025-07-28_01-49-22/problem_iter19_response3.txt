```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = volume_adjusted_intraday_range + combined_value_momentum - (df['open'] - df['close'].shift(1))
    
    # Compare Intraday Range Ratio to Baseline
    high_low_range_ratio = high_low_range / (df['high'] + df['low'])
    signal = (high_low_range_ratio > 1.0).astype(int) - (high_low_range_ratio < 1.0).astype(int)
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * (1.5 if signal.sum() > 0 else 0.5)
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum)
    
    return final_alpha_factor
```
