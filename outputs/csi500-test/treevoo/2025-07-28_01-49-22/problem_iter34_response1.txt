```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['ema_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()
    
    # Generate Alpha Factor and Baseline Comparison
    df['alpha_signal'] = np.where(df['ema_volume_weighted_high_low_range'] > 1.3, 1, -1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_alpha_factor'] = df['ema_volume_weighted_high_low_range'] / df['volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['close_to_close_return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['close_to_close_return']
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['ema_volume_weighted_high_low_range'] +
        df['combined_value_momentum'] +
        df['intraday_range_ratio'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['alpha_signal'] == 1,
        df['intermediate_alpha_factor'] * 1.8,
        df['intermediate_alpha_factor'] * 0.2
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    df['ema_volume_weighted_final_alpha_factor'] = df['volume_weighted_final_alpha_factor'].ewm(span=5).mean()
    
    return df['ema_volume_weighted_final_alpha_factor']
```
