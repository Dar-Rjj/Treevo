import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value_momentum = high_low_range - close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = (df['high'] - df['low']) * df['volume']
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range 
        + combined_value_momentum 
        + (df['high'] - df['low']) / (df['high'] + df['low'])
        - (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    high_low_ratio_10_day_ma = high_low_ratio.rolling(window=10).mean()
    dynamic_threshold = high_low_ratio > high_low_ratio_10_day_ma
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjustment_factor = 1.5 * (high_low_ratio > 1.0) + 0.5 * (high_low_ratio < 1.0)
    intermediate_alpha_factor = intermediate_alpha_factor * adjustment_factor
    
    # Introduce Volume Momentum
    volume_10_day_avg = df['volume'].rolling(window=10).mean()
    volume_momentum = df['volume'] / volume_10_day_avg
    volume_momentum_adjustment = 1.2 * (volume_momentum > 1) + 0.8 * (volume_momentum < 1)
    final_alpha_factor = intermediate_alpha_factor * volume_momentum_adjustment
    
    # Generate Final Alpha Factor
    final_alpha_factor += (df['close'] - df['close'].shift(1))
    
    return final_alpha_factor
