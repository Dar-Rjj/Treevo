```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility (Average of (High - Low) over 10 days)
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio (EMA with 10-day Window)
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Alpha Factor from High-Low Range
    alpha_factor_high_low = (smoothed_high_low_range_ratio > 1.0).astype(int) - (smoothed_high_low_range_ratio < 1.0).astype(int)
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=20).std()
    
    # Normalize Alpha Factor by Volatility
    normalized_alpha_factor = alpha_factor_high_low / volatility
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    prev_day_close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Smooth Combined High-Low and Close-to-Close (EMA with 10-day Window)
    smoothed_combined = (high_low_range - abs_close_to_close).ewm(span=10).mean()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range + 
        combined_value_momentum - 
        prev_day_close_to_open_change +
        intraday_range_ratio
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    positive_signal = (smoothed_high_low_range_ratio > 1.0).astype(int)
    negative_signal = (smoothed_high_low_range_ratio < 1.0).astype(int)
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * (positive_signal * 1.5 + negative_signal * 0.5)
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = (
        volume_adjusted_intraday_range_ratio - 
        prev_day_close_to_open_change
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    positive_signal_combined = (smoothed_combined > 1.0).astype(int)
    negative_signal_combined = (smoothed_combined < 1.0).astype(int)
    
    # Adjust Combined Alpha Factor
    adjusted_combined_alpha_factor = combined_alpha_factor * (positive_signal_combined * 1.5 + negative_signal_combined * 0.5)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        intraday_volatility * 
        (adjusted_intermediate_alpha_factor + close_price_momentum + intraday_range_ratio) * 
        (adjusted_combined_alpha_factor * (intraday_volatility * (combined_value_momentum + close_price_momentum)))
    )
    
    return final_alpha_factor
```
