```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'].shift(1) - df['close']).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value_momentum = high_low_range + abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility (Average of (High - Low) over 15 days)
    intraday_volatility = high_low_range.rolling(window=15).mean()
    
    # Calculate Close Price Momentum (Sum of ((Close at day t / Close at day t-1) - 1) over the last 20 days)
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        (df['high'] - df['low']) / (df['high'] + df['low']) -
        (df['open'] - df['close'].shift(1)).abs()
    )
    
    # Compare with Baseline (Intraday Range Ratio's Median)
    baseline_median = (df['high'] - df['low']) / (df['high'] + df['low']).median()
    positive_signal = (df['high'] - df['low']) / (df['high'] + df['low']) > baseline_median
    negative_signal = (df['high'] - df['low']) / (df['high'] + df['low']) < baseline_median
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.copy()
    adjusted_intermediate_alpha_factor[positive_signal] *= 1.7
    adjusted_intermediate_alpha_factor[negative_signal] *= 0.3
    
    # Generate Final Alpha Factor
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum)
    
    # Adjust Final Alpha Factor by Volume
    n_period = 20  # Example period for EMA
    ema_volume = df['volume'].ewm(span=n_period, adjust=False).mean()
    final_alpha_factor *= ema_volume
    
    return final_alpha_factor
```
