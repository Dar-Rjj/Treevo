```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['hl_ratio_ema'] = df['hl_ratio'].ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = df['hl_ratio_ema'].mean()
    df['momentum_signal'] = np.where(df['hl_ratio_ema'] > dynamic_threshold, 1, -1)
    df['alpha_factor_momentum'] = df['momentum_signal'] * df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['alpha_factor_momentum'] *= df['close_change']
    
    # Calculate High-Low Range Volatility
    df['hl_range_volatility'] = df['high_low_range'].rolling(window=15).sum() / 15
    
    # Smooth High-Low Range Volatility
    df['hl_range_volatility_ema'] = df['hl_range_volatility'].ewm(span=15, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    df['weighted_close_open_diff'] = 0.5 * df['close_open_diff']
    df['smoothed_hl_volatility_with_open_close'] = df['hl_range_volatility_ema'] + df['weighted_close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_hl_range'] = df['high_low_range'] * df['avg_volume_10']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Previous Day's Close-to-Open Change
    df['prev_day_close_open_change'] = df['open'] - df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (df['volume_adjusted_hl_range'] + df['combined_value_momentum'] 
                                      - df['prev_day_close_open_change'] 
                                      + df['intraday_range_ratio'])
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(df['hl_ratio_ema'] > dynamic_threshold, 
                                                        1.5 * df['intermediate_alpha_factor'], 
                                                        0.5 * df['intermediate_alpha_factor'])
    
    # Calculate Intraday Range Momentum
    df['intraday_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['intraday_volatility'] 
                                * (df['adjusted_intermediate_alpha_factor'] 
                                   + df['close_price_momentum'] 
                                   + df['intraday_range_momentum']))
    
    # Adjust Final Alpha Factor
    df['final_alpha_factor'] = np.where(df['hl_range_volatility_ema'] > 0.8, 
                                        1.6 * df['final_alpha_factor'], 
                                        0.4 * df['final_alpha_factor'])
    
    return df['final_alpha_factor']
```
