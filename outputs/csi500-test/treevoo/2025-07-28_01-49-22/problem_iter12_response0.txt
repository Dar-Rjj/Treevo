```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume over n-period
    avg_volume_n = df['volume'].rolling(window=n).mean()
    
    # Adjust by Relative Closing Price Change
    close_change_ratio = df['close'] / df['close'].shift(1)
    
    # Combine all the components
    factor = high_low_range * avg_volume_n * close_change_ratio
    
    return factor
```
