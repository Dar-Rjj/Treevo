```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smoothed_hl_ratio'] = df['hl_ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['smoothed_hl_ratio'] > 1.3, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_hl_range'] = df['high_low_range'] * df['volume']
    df['vol_weighted_hl_range_ema'] = df['vol_weighted_hl_range'].ewm(span=10).mean()
    
    # Generate Alpha Factor
    df['alpha_factor'] = np.where(df['vol_weighted_hl_range_ema'] > 1.3, 1, -1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adj_alpha_factor'] = df['alpha_factor'] / df['volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['vol_weighted_hl_range_ema'] 
        + df['combined_value_momentum'] 
        + df['intraday_range_ratio'] 
        - (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['alpha_factor'] == 1, 
        df['intermediate_alpha_factor'] * 1.8, 
        df['intermediate_alpha_factor'] * 0.2
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (
        (df['adjusted_intermediate_alpha_factor'] + df['close_momentum']) 
        * df['intraday_volatility']
    )
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    df['volume_weighted_final_alpha_factor_ema'] = df['volume_weighted_final_alpha_factor'].ewm(span=5).mean()
    
    return df['volume_weighted_final_alpha_factor_ema']
```
