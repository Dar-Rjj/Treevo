```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1, -1)
    
    # Calculate Absolute Close-to-Close Return
    df['close_to_close_return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - np.abs(df['close_to_close_return'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum (Sum of N-day Returns over the last 20 days)
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['intraday_volatility'] * (df['combined_value_momentum'] + df['close_price_momentum']) * df['volume']
    
    return df['alpha_factor']
```
