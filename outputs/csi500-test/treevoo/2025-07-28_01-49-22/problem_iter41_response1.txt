```python
def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the range of the day (high - low)
    df['range'] = df['high'] - df['low']
    
    # Calculate the volume weighted average price (VWAP) for the day
    df['vwap'] = (df['amount'] / df['volume']).shift(1)
    
    # Calculate the relative strength index (RSI) based on the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['close_change'] / df['range']) * (df['vwap'] / df['close']) * (df['rsi'] / 50)
    
    return df['alpha_factor']
```
