```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    previous_close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_high_low_close = high_low_range - abs_close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Smooth Combined High-Low and Close-to-Close
    smoothed_combined_high_low_close = combined_high_low_close.ewm(span=10, adjust=False).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = volume_adjusted_intraday_range_ratio - previous_close_to_open_change
    
    # Adjust Combined Alpha Factor based on Baseline
    combined_alpha_factor = combined_alpha_factor * np.where(smoothed_combined_high_low_close > 1.0, 1.5, 0.5)
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10, adjust=False).mean()
    
    # Generate High-Low Range Momentum Alpha Factor
    high_low_range_momentum_alpha = np.where(smoothed_high_low_range_ratio > 1.0, 1.2, 0.8)
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_alpha_factor - combined_alpha_factor.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (combined_alpha_factor * 
                          (intraday_volatility * 
                           (combined_value_momentum + close_price_momentum)))
    
    # Incorporate High-Low Range Momentum Alpha Factor
    final_alpha_factor = final_alpha_factor * high_low_range_momentum_alpha
    
    return final_alpha_factor
```
