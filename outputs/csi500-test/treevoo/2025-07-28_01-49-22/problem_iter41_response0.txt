```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Smoothed High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=10).mean()

    # Generate Intraday Momentum Signal
    intraday_signal = np.where(df['High_Low_Range'] > 1.3, 1, -1)
    df['Intraday_Signal'] = intraday_signal

    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low'] = (df['High_Low_Range'] * df['volume']).ewm(span=10).mean()

    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_Return'] = abs(df['close'] - df['close'].shift(1))

    # Combine High-Low and Close-to-Close Returns
    df['Combined_Value'] = df['High_Low_Range'] + df['Abs_Close_Return']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()

    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)

    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()

    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Volume_Weighted_High_Low'] +
        df['Combined_Value_Momentum'] +
        (df['High_Low_Range'] / df['Intraday_Volatility']) -
        (df['open'] - df['close'].shift(1))
    )

    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Adjusted_Intermediate_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] / df['Volatility']

    # Adjust Intermediate Alpha Factor based on Intraday Signal
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(
        df['Intraday_Signal'] == 1,
        df['Adjusted_Intermediate_Alpha_Factor'] * 1.8,
        df['Adjusted_Intermediate_Alpha_Factor'] * 0.2
    )

    # Introduce Additional Momentum Component
    df['50_MA'] = df['close'].rolling(window=50).mean()
    df['200_MA'] = df['close'].rolling(window=200).mean()
    long_term_momentum = np.where(df['50_MA'] > df['200_MA'], 1, -1)
    df['Long_Term_Momentum'] = long_term_momentum

    # Adjust Final Alpha Factor based on Long-Term Momentum
    df['Final_Alpha_Factor'] = (
        (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']) * df['Intraday_Volatility']
    )
    df['Final_Alpha_Factor'] = np.where(
        df['Long_Term_Momentum'] == 1,
        df['Final_Alpha_Factor'] * 1.2,
        df['Final_Alpha_Factor'] * 0.8
    )

    # Weight by Volume
    df['Volume_Weighted_Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['volume']
    df['Volume_Adjusted_Final_Alpha_Factor'] = df['Volume_Weighted_Final_Alpha_Factor'].ewm(span=7).mean()

    return df['Volume_Adjusted_Final_Alpha_Factor']
```
