```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Smooth High-Low Range with 10-day EMA
    smooth_high_low_range = high_low_range.ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    vol_weighted_high_low_range = (high_low_range * data['volume']).ewm(span=10).mean()
    
    # Generate Alpha Factor from High-Low Range
    alpha_factor_high_low = np.where(smooth_high_low_range > 1.0, 1, -1)
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (data['high'] - data['low']) / (data['high'] + data['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    vol_adjusted_intraday_range_ratio = intraday_range_ratio * data['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    close_to_open_change = data['open'] - data['close'].shift(1)
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = np.abs(data['close'] - data['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_high_low_close = high_low_range - abs_close_to_close_return
    
    # Smooth Combined High-Low and Close-to-Close with 10-day EMA
    smooth_combined_high_low_close = combined_high_low_close.ewm(span=10).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = vol_adjusted_intraday_range_ratio - close_to_open_change
    combined_alpha_signal = np.where(smooth_combined_high_low_close > 1.0, 1.5, 0.5)
    adjusted_combined_alpha_factor = combined_alpha_factor * combined_alpha_signal
    
    # Calculate Combined Value Momentum
    combined_value_momentum = (combined_alpha_factor - combined_alpha_factor.shift(1))
    
    # Calculate Close Price Momentum over the last 20 days
    close_price_momentum = ((data['close'] / data['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_alpha_factor * (combined_value_momentum + close_price_momentum)
    
    # Introduce a New Factor: Close-Open Gap
    close_open_gap = data['close'] - data['open']
    
    # Smooth Close-Open Gap with 5-day EMA
    smooth_close_open_gap = close_open_gap.ewm(span=5).mean()
    
    # Generate Alpha Factor from Close-Open Gap
    alpha_factor_close_open = np.where(smooth_close_open_gap > 0.0, 1, -1)
    
    # Integrate Close-Open Gap into Final Alpha Factor
    final_alpha_factor += smooth_close_open_gap * alpha_factor_close_open
    
    return final_alpha_factor
```
