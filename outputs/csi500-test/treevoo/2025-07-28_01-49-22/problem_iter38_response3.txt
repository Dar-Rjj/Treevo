```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the range (open, high, low)
    df['close_range_position'] = (df['close'] - df[['open', 'high', 'low']].min(axis=1)) / (df[['open', 'high', 'low']].max(axis=1) - df[['open', 'high', 'low']].min(axis=1))
    
    # Calculate the ratio of amount to volume
    df['amount_to_volume'] = df['amount'] / df['volume']
    
    # Calculate the difference between the high and low prices
    df['price_spread'] = df['high'] - df['low']

    # Calculate a weighted sum of the above features to form the alpha factor
    # We assume that the close price being higher in the range, a higher amount-to-volume ratio, and a wider price spread are all positive signals
    df['alpha_factor'] = 0.4 * df['close_range_position'] + 0.3 * df['amount_to_volume'] + 0.3 * df['price_spread']

    return df['alpha_factor']
```
