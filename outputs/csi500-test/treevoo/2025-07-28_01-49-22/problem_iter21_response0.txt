```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Analyze volume and amount
    df['daily_turnover_rate'] = df['volume'] / df['volume'].shift(1)
    df['daily_amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Incorporate high and low prices
    df['daily_range'] = df['high'] - df['low']
    df['daily_range_percentage'] = df['daily_range'] / df['close'].shift(1)
    
    # Composite factor using daily return, daily range, and daily turnover rate
    df['composite_factor_1'] = df['daily_return'] * df['daily_range'] * df['daily_turnover_rate']
    
    # Composite factor using daily return, daily range percentage, and daily amount change
    df['composite_factor_2'] = df['daily_return'] * df['daily_range_percentage'] * df['daily_amount_change']
    
    # Explore additional price patterns
    df['momentum_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['momentum_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Incorporate price and volume changes
    df['price_volume_trend'] = (df['close'] - df['close'].shift(1)) / (df['volume'] - df['volume'].shift(1))
    df['price_amount_trend'] = (df['close'] - df['close'].shift(1)) / (df['amount'] - df['amount'].shift(1))
    
    # Explore relative strength
    # Assuming there is an 'index_close' column in the dataframe
    if 'index_close' in df.columns:
        df['relative_strength'] = df['close'] / df['index_close']
    
    # Return a pandas Series with the composite factor 2
    return df['composite_factor_2']
```
