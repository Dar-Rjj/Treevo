```python
def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the difference between the VWAP and the close price
    df['vwap_close_diff'] = df['vwap'] - df['close']
    
    # Calculate the ratio of the high to low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Generate a signal based on the combination of the above features
    df['signal'] = df['close_change'] * df['vwap_close_diff'] * df['high_low_ratio']
    
    # Return the signal as the alpha factor
    return df['signal']
```
