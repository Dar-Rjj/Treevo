```python
import pandas as pd

def heuristics_v2(df, N=10, M=20):
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'].pct_change()
    
    # Sum Over N Days (Cumulative Price Change)
    df['cumulative_price_change'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Determine Median Volume
    df['median_volume'] = df['volume'].rolling(window=N).median()
    
    # Apply Volume Filter
    df['volume_filter'] = (df['volume'] > df['median_volume']).astype(int)
    
    # Calculate Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_range'].rolling(window=10).mean()
    
    # Calculate Momentum
    df['N_day_returns'] = df['close'].pct_change()
    df['momentum'] = df['N_day_returns'].rolling(window=M).sum()
    
    # Combine Factors
    df['final_alpha_factor'] = (df['cumulative_price_change'] * 
                                 df['intraday_volatility'] * 
                                 df['momentum'] * 
                                 df['volume_filter'])
    
    return df['final_alpha_factor'].dropna()
```
