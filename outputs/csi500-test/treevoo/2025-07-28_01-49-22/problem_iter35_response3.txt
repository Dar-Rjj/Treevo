import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smoothed_high_low_range_ratio > 1.35).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Generate Alpha Factor
    alpha_factor = (volume_weighted_high_low_range > 1.35).astype(int) * 2 - 1
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_alpha_factor = alpha_factor / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_return = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range + close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value.diff()
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_weighted_high_low_range +
        combined_value_momentum +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha_factor = np.where(
        intraday_range_ratio > 0.6,
        intermediate_alpha_factor * 1.2,
        intermediate_alpha_factor * 0.8
    )
    
    # Compare with Dynamic Threshold
    dynamic_threshold = smoothed_high_low_range_ratio.rolling(window=20).mean()
    signal = (smoothed_high_low_range_ratio > dynamic_threshold).astype(int) * 2 - 1
    
    # Adjust Combined Alpha Factor
    adjusted_intermediate_alpha_factor = np.where(
        signal > 0,
        intermediate_alpha_factor * 1.4,
        intermediate_alpha_factor * 0.6
    )
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        intraday_volatility *
        (adjusted_intermediate_alpha_factor + close_price_momentum) *
        df['volume']
    )
    
    # Weight by Volume
    final_alpha_factor = (final_alpha_factor * df['volume']).ewm(span=7).mean()
    
    return final_alpha_factor
