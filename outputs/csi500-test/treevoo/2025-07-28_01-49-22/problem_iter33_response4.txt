```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (
        volume_adjusted_intraday_range + 
        combined_value_momentum + 
        (high_low_range / (df['high'] + df['low'])) - 
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio = high_low_range / (df['high'] + df['low'])
    high_low_ratio_5d_ma = high_low_ratio.rolling(window=5).mean()
    dynamic_signal = high_low_ratio > high_low_ratio_5d_ma
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjusted_intermediate_alpha = intermediate_alpha.copy()
    adjusted_intermediate_alpha[high_low_ratio > 0.8] *= 1.3
    adjusted_intermediate_alpha[high_low_ratio < 0.8] *= 0.7
    
    # Introduce Volume Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    current_volume_vs_avg = df['volume'] / avg_volume_5d
    volume_momentum = current_volume_vs_avg > 1
    adjusted_intermediate_alpha[volume_momentum] *= 1.15
    adjusted_intermediate_alpha[~volume_momentum] *= 0.85
    
    # Generate Final Alpha Factor
    final_alpha = adjusted_intermediate_alpha + close_price_momentum
    
    # Adjust Final Alpha Factor by Volume
    n_period_avg_volume = df['volume'].rolling(window=20).mean()
    final_alpha *= n_period_avg_volume
    
    return final_alpha
```
