```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Calculate Short-Term and Long-Term Moving Averages of Daily Returns
    short_term_window = 5
    long_term_window = 30
    df['short_term_ma'] = df['daily_returns'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['daily_returns'].rolling(window=long_term_window).mean()
    df['momentum_signal'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close'] = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_range'] = df['high_low_range'] + df['abs_close_to_close']
    
    # Calculate Volume-Adjusted Intraday Range
    df['volume_adjusted_intraday_range'] = df['high_low_range'] * df['volume']
    
    # Generate Alpha Factor from Momentum
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    ema_window = 10
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=ema_window, adjust=False).mean()
    df['dynamic_threshold'] = (df['high_low_range'] / (df['high'] + df['low'])).rolling(window=ema_window).mean()
    df['alpha_factor_momentum'] = (df['smoothed_high_low_range_ratio'] > df['dynamic_threshold']).astype(int) - (df['smoothed_high_low_range_ratio'] < df['dynamic_threshold']).astype(int)
    df['alpha_factor_momentum'] *= df['volume']
    
    # Adjust Alpha Factor Based on Intraday Range Ratio
    intraday_range_ratio = df['high_low_range'] / (df['high'] + df['low'])
    df['alpha_factor_momentum'] = df['alpha_factor_momentum'] * intraday_range_ratio.apply(lambda x: 1.4 if x > 0.7 else (0.6 if x < 0.4 else 1))
    
    # Introduce Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Weighted Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'] * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume-Adjusted Close Price Momentum
    df['volume_adjusted_close_price_momentum'] = df['daily_returns'] * (df['volume'] / df['volume'].shift(1))
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['alpha_factor_momentum'] *= df['close_price_change']
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    df['weighted_close_open_diff'] = 0.5 * df['close_open_diff']  # Fixed weight of 0.5
    df['alpha_factor_momentum'] += df['weighted_close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    n_period = 10
    df['average_volume'] = df['volume'].rolling(window=n_period).mean()
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * df['average_volume']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['short_term_ma'] - df['long_term_ma']) * (df['alpha_factor_momentum'] + df['volume_adjusted_close_price_momentum'] + df['volume_momentum']) * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    return df['final_alpha_factor'].dropna()
```
