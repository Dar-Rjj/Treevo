```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10 days)
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    df['Dynamic_Threshold'] = df['Smoothed_High_Low_Range_Ratio'].rolling(window=20).mean()
    df['Momentum_Signal'] = df['Smoothed_High_Low_Range_Ratio'] - df['Dynamic_Threshold']
    df['Momentum_Signal'] = df['Momentum_Signal'].apply(lambda x: 1 if x > 0 else -1)
    
    # EMA of Volume (5 days)
    df['EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Incorporate Volume into Momentum Signal
    df['Volume_Adjusted_Momentum_Signal'] = df['Momentum_Signal'] * df['EMA_Volume']
    
    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['Volume_Adjusted_Intraday_Range_Ratio'] = df['Intraday_Range_Ratio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['Prev_Close_to_Open_Change'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily Close-Open Difference
    df['Close_Open_Difference'] = df['close'] - df['open']
    
    # Calculate High-Low Range Volatility
    df['High_Low_Range_Volatility'] = df['High_Low_Range'].rolling(window=15).sum() / 15
    
    # Smooth High-Low Range Volatility with EMA (15 days)
    df['Smoothed_High_Low_Range_Volatility'] = df['High_Low_Range_Volatility'].ewm(span=15, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    fixed_weight = 0.5  # Fixed Weight for the Close-Open Difference
    df['Adjusted_High_Low_Range_Volatility'] = df['Smoothed_High_Low_Range_Volatility'] + fixed_weight * df['Close_Open_Difference']
    
    # Generate Combined Alpha Factor
    df['Combined_Alpha_Factor'] = df['Volume_Adjusted_Intraday_Range_Ratio'] - df['Prev_Close_to_Open_Change']
    
    # Compare with Baseline (Fixed Threshold, e.g., 0.8)
    df['Signal'] = df['Smoothed_High_Low_Range_Volatility'].apply(lambda x: 1.6 if x > 0.8 else 0.4)
    df['Adjusted_Combined_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Signal']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Combined_Alpha_Factor'] * df['volume']
    
    return df['Final_Alpha_Factor']
```
