import pandas as pd

def heuristics_v2(df, N=10, M=5, Weight1=0.3, Weight2=0.2, Weight3=0.4, Weight4=0.1, Weight5=0.6, Weight6=0.4):
    # Calculate daily return
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    
    # Determine average volume over the past N days
    df['Average_Volume'] = df['volume'].rolling(window=N).mean()
    
    # Create a volume ratio factor
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Generate a weighted score combining daily return and volume ratio
    df['Weighted_Score'] = (df['Daily_Return'] * Weight1) + (df['Volume_Ratio'] * Weight2)
    
    # Calculate the intra-day price range
    df['Intra_Day_Range'] = df['high'] - df['low']
    
    # Standardize the intra-day movement
    df['Standardized_Intra_Day_Movement'] = df['Intra_Day_Range'] / df['close']
    
    # Update the weighted score to include intra-day movement
    df['Updated_Weighted_Score'] = (df['Weighted_Score'] * Weight3) + (df['Standardized_Intra_Day_Movement'] * Weight4)
    
    # Compute the momentum over M days
    df['Momentum'] = (df['close'] - df['close'].shift(M)) / df['close'].shift(M)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Updated_Weighted_Score'] * Weight5) + (df['Momentum'] * Weight6)
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor'].dropna()
