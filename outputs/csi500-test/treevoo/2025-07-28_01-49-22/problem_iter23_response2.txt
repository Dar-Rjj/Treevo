```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative spread
    df['relative_spread'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted price
    df['volume_weighted_price'] = (df['open'] + 2 * df['close']) * df['volume'] / (4 * df['volume'])
    
    # Heuristic: Relative Spread Momentum
    # This factor is based on the idea that stocks with high relative spread have a higher volatility and may exhibit momentum.
    df['relative_spread_momentum'] = df['relative_spread'].rolling(window=10).mean() * df['daily_return'].rolling(window=5).mean()
    
    # Heuristic: Volume Weighted Price Change
    # This factor captures the change in the volume-weighted price, which can be an indicator of future price movements.
    df['volume_weighted_price_change'] = df['volume_weighted_price'].diff()
    
    # Combine the two heuristics to create the final alpha factor
    df['alpha_factor'] = df['relative_spread_momentum'] + df['volume_weighted_price_change']
    
    return df['alpha_factor'].dropna()
```
