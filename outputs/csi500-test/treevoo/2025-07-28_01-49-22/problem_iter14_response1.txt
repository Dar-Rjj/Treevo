```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_window=10, close_open_weight=0.5):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['Prev_High_Low_Range']
    
    # Smooth High-Low Range Ratio using EMA
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=ema_window, min_periods=ema_window).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = df['Smoothed_High_Low_Range_Ratio'].rolling(window=ema_window).mean()
    df['Alpha_Factor'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > dynamic_threshold, 1, -1) * df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['Close_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Alpha_Factor'] *= df['Close_Price_Change']
    
    # Calculate High-Low Range Volatility
    df['High_Low_Range_Volatility'] = df['High_Low_Range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    average_volume = df['volume'].rolling(window=n).mean()
    df['Volume_Adjusted_Volatility'] = df['High_Low_Range_Volatility'] * average_volume
    
    # Incorporate Close-Open Difference
    df['Close_Open_Difference'] = df['close'] - df['open']
    df['Enhanced_Volatility'] = df['Volume_Adjusted_Volatility'] + (df['Close_Open_Difference'] * close_open_weight)
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    df['Avg_Volume'] = average_volume
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * df['Avg_Volume']
    
    # Combine all factors
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] + df['Enhanced_Volatility'] + df['Volume_Adjusted_High_Low_Range']
    
    return df['Final_Alpha_Factor']
```
