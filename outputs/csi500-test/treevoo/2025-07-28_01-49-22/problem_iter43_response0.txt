```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smooth_high_low'] = df['high_low_range'].ewm(span=5).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = (df['close'] / df['close'].shift(1)).apply(lambda x: 1 if x > 1.2 else -1 if x < 1.2 else 0)
    
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['smooth_vol_weighted_high_low'] = df['vol_weighted_high_low'].ewm(span=5).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['smooth_vol_weighted_high_low'] +
                                df['combined_value_momentum'] +
                                (df['high_low_range'] / df['intraday_volatility']) -
                                (df['open'] - df['close'].shift(1)))
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adj_intermediate_alpha'] = df['intermediate_alpha'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    df['adj_intermediate_alpha'] = df.apply(
        lambda row: row['adj_intermediate_alpha'] * 1.6 if row['intraday_signal'] == 1 else 
                    row['adj_intermediate_alpha'] * 0.4 if row['intraday_signal'] == -1 else 
                    row['adj_intermediate_alpha'], axis=1)
    
    # Generate Final Alpha Factor
    df['final_alpha'] = (df['adj_intermediate_alpha'] + df['close_price_momentum']) * df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_final_alpha'] = df['final_alpha'] * df['volume']
    df['smooth_volume_weighted_final_alpha'] = df['volume_weighted_final_alpha'].ewm(span=10).mean()
    
    # Introduce Additional Momentum Component
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['ma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = df.apply(
        lambda row: 1.3 if row['ma_50'] > row['ma_200'] else 0.7, axis=1)
    
    # Adjust Final Alpha Factor
    df['final_alpha'] = df['smooth_volume_weighted_final_alpha'] * df['long_term_momentum']
    
    return df['final_alpha']
```
