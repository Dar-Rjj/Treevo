import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change Velocity
    price_change_velocity = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - 2 * df['volume'].shift(1) + df['volume'].shift(2)
    
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility (n-period = 5 days)
    high_low_range_volatility = daily_high_low_range.rolling(window=5).sum() / 5
    
    # Adjust Volatility by Volume (average volume over 5 days)
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    adjusted_volatility_by_volume = high_low_range_volatility * avg_volume_5_days
    
    # Combine Price, Volume, and Volatility Indicators
    alpha_factor = (price_change_velocity * volume_acceleration * adjusted_volatility_by_volume)
    
    return alpha_factor
