```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate High-Low Range Ratio
    df['high_low_ratio'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Smooth High-Low Range Ratio
    df['smoothed_high_low_ratio'] = df['high_low_ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['smoothed_high_low_ratio'] > 1.0, 1, -1)
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'].diff()
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=20).median()
    df['volume_filtered'] = df['volume'] > median_volume
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['intraday_volatility'] * 
        (df['combined_value_momentum'] + df['close_price_momentum']) * 
        df['volume']
    )
    
    df['alpha_factor'] = df['alpha_factor'].where(df['volume_filtered'], 0)
    
    return df['alpha_factor']
```
