def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Range
    previous_intraday_range = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Range Momentum
    range_momentum = (intraday_range / previous_intraday_range) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread = df['close'] - df['open']
    previous_close_open_spread = df['close'].shift(1) - df['open'].shift(1)
    close_open_spread_ratio = close_open_spread * previous_close_open_spread
    
    # Combine both factors to create the final alpha factor
    alpha_factor = range_momentum + close_open_spread_ratio
    
    return alpha_factor
