```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with EMA (10-day window)
    smoothed_high_low = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smoothed_high_low > 1.0).astype(int) * 1.5 + (smoothed_high_low < 1.0).astype(int) * 0.5
    
    # Volume-Weighted High-Low Range
    volume_weighted_high_low = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_volume_weighted_high_low = volume_weighted_high_low / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low and Close-to-Close Returns
    combined_value = smoothed_high_low - close_to_close_return
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (adjusted_volume_weighted_high_low 
                                 + combined_value 
                                 + intraday_range_ratio 
                                 - (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor based on Intraday Signal
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * intraday_signal
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_to_close_return) * volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = (final_alpha_factor * df['volume']).ewm(span=5).mean()
    
    return volume_weighted_final_alpha_factor
```
