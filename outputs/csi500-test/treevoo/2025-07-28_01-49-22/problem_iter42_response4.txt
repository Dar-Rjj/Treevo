import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio with EMA (15-day window)
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=15).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = np.where(smoothed_high_low_range_ratio > 1.4, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=15).mean()
    
    # Generate Alpha Factor
    alpha_factor = np.where(smoothed_volume_weighted_high_low_range > 1.4, 1, -1)
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=25).std()
    adjusted_alpha_factor = smoothed_volume_weighted_high_low_range / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=15).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=25).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        smoothed_volume_weighted_high_low_range +
        combined_value_momentum +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = np.where(alpha_factor > 0, intermediate_alpha_factor * 2.0, intermediate_alpha_factor * 0.05)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * intraday_volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    final_alpha_factor = volume_weighted_final_alpha_factor.ewm(span=7).mean()
    
    return final_alpha_factor
