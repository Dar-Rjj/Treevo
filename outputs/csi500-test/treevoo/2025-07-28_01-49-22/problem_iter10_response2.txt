```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Range
    df['prev_intraday_range'] = df['intraday_range'].shift(1)
    
    # Calculate Intraday Range Momentum
    df['intraday_range_momentum'] = (df['intraday_range'] / df['prev_intraday_range']) - 1
    
    # Calculate Close-Open Spread Ratio
    df['close_open_spread_ratio'] = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate Price Change Velocity
    df['price_change_velocity'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(1)) / (df['volume'].shift(1) - df['volume'].shift(2))
    
    # Calculate High-Low Range Volatility (n-period sum divided by n, n=5)
    n = 5
    df['high_low_volatility'] = df['intraday_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume (multiply by average volume over the same n-period)
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_high_low_volatility'] = df['high_low_volatility'] * df['avg_volume']
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    df['intraday_price_volume_factor'] = df['intraday_range_momentum'] * df['price_change_velocity'] * df['volume']
    
    # Combine Price, Volume, and Volatility Indicators
    df['price_volume_volatility_factor'] = df['price_change_velocity'] * df['volume_acceleration'] * df['adjusted_high_low_volatility']
    
    # Synthesize Alpha Factor
    df['alpha_factor'] = df['intraday_price_volume_factor'] + df['price_volume_volatility_factor']
    
    return df['alpha_factor']
```
