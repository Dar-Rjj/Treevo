```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smoothed_hl_ratio'] = df['hl_ratio'].ewm(span=10).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    df['dynamic_threshold'] = df['smoothed_hl_ratio'].rolling(window=20).mean()
    df['momentum_signal'] = df['smoothed_hl_ratio'].gt(df['dynamic_threshold']).astype(int) - df['smoothed_hl_ratio'].lt(df['dynamic_threshold']).astype(int)
    
    # EMA of Volume
    df['ema_volume'] = df['volume'].ewm(span=5).mean()
    df['momentum_factor'] = df['momentum_signal'] * df['ema_volume']
    
    # Calculate High-Low Range Volatility
    n = 20  # n-period
    df['volatility'] = df['high_low_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['adjusted_volatility'] = df['volatility'] * df['ema_volume']
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.1  # Fixed weight for the close-open difference
    df['weighted_close_open_diff'] = df['close_open_diff'] * fixed_weight
    df['adjusted_volatility_with_co'] = df['adjusted_volatility'] + df['weighted_close_open_diff']
    
    # Calculate High-Low Range Momentum
    df['hl_momentum'] = df['smoothed_hl_ratio'].ewm(span=10).mean()
    
    # Adjust Momentum by Volume
    df['adjusted_momentum'] = df['hl_momentum'] * df['ema_volume']
    
    # Combine Momentum and Volatility
    df['alpha_factor'] = df['adjusted_momentum'] + df['adjusted_volatility_with_co']
    
    return df['alpha_factor']
```
