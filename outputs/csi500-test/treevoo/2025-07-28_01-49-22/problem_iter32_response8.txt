import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-based Factor
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility-based Factor
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['Volatility'] = log_returns.rolling(window=20).std()
    
    # Volume-related Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    df['Volume_Momentum'] = np.where(df['close'] > df['open'], volume_momentum, 0)
    
    # Price Reversal Indicator
    df['Price_Reversal'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Accumulation/Distribution Line
    df['A/D'] = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    df['A/D'] = df['A/D'].cumsum()
    
    # Open Interest Analysis
    df['Open_Interest'] = (df['close'] - df['open']) / df['open']
    
    # High-Low Range and Intraday Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    df['HL_Range_Ratio'] = df['High_Low_Range'] / df['Prev_High_Low_Range']
    df['Smoothed_HL_Range_Ratio'] = df['HL_Range_Ratio'].ewm(span=10).mean()
    
    # Intraday Momentum Signal
    df['Intraday_Signal'] = np.where(df['Smoothed_HL_Range_Ratio'] > 1.3, 1, -1)
    
    # Volume-Weighted High-Low Range
    df['VW_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_VW_High_Low_Range'] = df['VW_High_Low_Range'].ewm(span=10).mean()
    
    # Adjust for Volatility
    df['Alpha_Factor'] = df['Smoothed_VW_High_Low_Range'] / df['Volatility']
    
    # Absolute Close-to-Close Return
    df['Abs_Close_Return'] = df['close'] - df['close'].shift(1)
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] - df['Abs_Close_Return']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (df['Smoothed_VW_High_Low_Range'] + 
                                       df['Combined_Value_Momentum'] + 
                                       df['Intraday_Range_Ratio'] - 
                                       (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(df['Intraday_Signal'] == 1, 
                                                        df['Intermediate_Alpha_Factor'] * 1.8, 
                                                        df['Intermediate_Alpha_Factor'] * 0.2)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Intraday_Volatility'] * (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum'])
