```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['Smoothed_HL_Range'] = df['High_Low_Range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['Intraday_Signal'] = np.where(df['Smoothed_HL_Range'] > 1.25, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_HL_Range'] = df['High_Low_Range'] * df['volume']
    df['Smoothed_Volume_Weighted_HL_Range'] = df['Volume_Weighted_HL_Range'].ewm(span=10).mean()
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = np.where(df['Smoothed_Volume_Weighted_HL_Range'] > 1.25, 1, -1)
    
    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Alpha_Factor'] /= df['Volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'].diff().abs()
    
    # Combine High-Low and Close-to-Close Returns
    df['Combined_Value'] = df['Smoothed_HL_Range'] - df['Close_to_Close_Return']
    
    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'].diff()
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha'] = (df['Smoothed_Volume_Weighted_HL_Range'] 
                                + df['Combined_Value_Momentum'] 
                                + df['Intraday_Range_Ratio'] 
                                - (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor
    df['Intermediate_Alpha'] = np.where(df['Alpha_Factor'] > 0, df['Intermediate_Alpha'] * 1.75, df['Intermediate_Alpha'] * 0.25)
    
    # Incorporate Dynamic Threshold Adjustment
    df['10_Day_MA_Intraday_Range_Ratio'] = df['Intraday_Range_Ratio'].rolling(window=10).mean()
    df['Intermediate_Alpha'] = np.where(df['Intraday_Range_Ratio'] > df['10_Day_MA_Intraday_Range_Ratio'], 
                                        df['Intermediate_Alpha'] * 1.75, 
                                        df['Intermediate_Alpha'] * 0.25)
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Final Alpha Factor
    df['Final_Alpha'] = (df['Intermediate_Alpha'] 
                         + df['Close_to_Close_Return'] 
                         + df['Close_Price_Momentum']) * df['Intraday_Volatility']
    
    # Introduce Volume Momentum
    df['10_Day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Momentum'] = df['volume'] / df['10_Day_Avg_Volume']
    df['Final_Alpha'] = np.where(df['Volume_Momentum'] > 1, df['Final_Alpha'] * 1.15, df['Final_Alpha'] * 0.85)
    
    # Weight by Volume
    df['Volume_Weighted_Final_Alpha'] = df['Final_Alpha'] * df['volume']
    df['Final_Alpha'] = df['Volume_Weighted_Final_Alpha'].ewm(span=6).mean()
    
    return df['Final_Alpha']
```
