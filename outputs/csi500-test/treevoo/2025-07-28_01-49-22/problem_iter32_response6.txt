```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smoothed_high_low_range'] = df['high_low_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = (df['high_low_range'] > 1.0).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['smoothed_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=15).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['smoothed_volume_weighted_high_low_range'] +
        df['combined_value_momentum'] +
        (df['high'] - df['low']) / (df['high'] + df['low']) -
        abs(df['open'] - df['close'].shift(1))
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] / df['volatility']
    
    # Compare with Baseline (Intraday Range Ratio's Median)
    intraday_range_ratio_median = (df['high'] - df['low']) / (df['high'] + df['low']).median()
    df['signal'] = (df['high'] - df['low']) / (df['high'] + df['low']) > intraday_range_ratio_median
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['signal'],
        df['adjusted_intermediate_alpha_factor'] * 1.7,
        df['adjusted_intermediate_alpha_factor'] * 0.3
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']) * df['intraday_volatility']
    
    # Adjust Final Alpha Factor by Volume
    df['ema_volume'] = df['volume'].ewm(span=10).mean()
    df['final_alpha_factor'] *= df['ema_volume']
    
    return df['final_alpha_factor']
```
