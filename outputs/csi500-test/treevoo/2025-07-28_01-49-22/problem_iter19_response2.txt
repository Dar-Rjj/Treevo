```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_Return'] = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] - df['Abs_Close_Return']
    
    # Calculate Volume Adjusted Intraday Range
    df['Volume_Adjusted_Intraday_Range'] = df['High_Low_Range'] * df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Momentum
    df['Intraday_Range_Momentum'] = (df['High_Low_Range'] / df['High_Low_Range'].shift(1)) - 1
    
    # Calculate Close-Open Spread Ratio
    df['Close_Open_Spread_Ratio'] = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (df['Volume_Adjusted_Intraday_Range'] + df['Combined_Value_Momentum']) - df['Close_Open_Spread_Ratio']
    
    # Calculate Price Change Velocity
    df['Price_Change_Velocity'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Volatility by Volume
    df['Adjusted_Volatility_by_Volume'] = df['Intraday_Volatility'] * df['volume'].rolling(window=10).mean()
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    df['Intraday_Range_Momentum_Price_Velocity_Volume'] = (df['Intraday_Range_Momentum'] * df['Price_Change_Velocity']) * df['volume']
    
    # Combine Price, Volume, and Volatility Indicators
    df['Price_Volume_Volatility_Indicator'] = (df['Price_Change_Velocity'] * df['Volume_Acceleration']) * df['Adjusted_Volatility_by_Volume']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Intermediate_Alpha_Factor'] + 
                                df['Close_Price_Momentum'] + 
                                df['Intraday_Range_Momentum'] + 
                                df['Intraday_Range_Momentum_Price_Velocity_Volume'] + 
                                df['Price_Volume_Volatility_Indicator'])
    
    return df['Final_Alpha_Factor']
```
