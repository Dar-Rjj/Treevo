```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA
    df['ema_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = df['ema_high_low_range_ratio'].mean()
    df['alpha_factor'] = df['ema_high_low_range_ratio'].apply(lambda x: 1 if x > dynamic_threshold else -1)
    df['alpha_factor'] *= df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['alpha_factor'] *= df['close_change']
    
    # Calculate High-Low Range Volatility
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=5).sum() / 5
    
    # Smooth High-Low Range Volatility using EMA
    df['ema_high_low_range_volatility'] = df['high_low_range_volatility'].ewm(span=15).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.1  # Example weight, can be tuned
    df['combined_volatility'] = df['ema_high_low_range_volatility'] + fixed_weight * df['close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    avg_volume = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * avg_volume
    df['alpha_factor'] += df['volume_adjusted_high_low_range']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    n_days = 30
    df['n_day_returns'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=n_days).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=5).median()
    df = df[df['volume'] >= median_volume]
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=30).std()
    df['alpha_factor'] /= df['volatility']
    
    # Incorporate Trading Volume Spike
    avg_volume_n_period = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = df['volume'] / avg_volume_n_period
    df['alpha_factor'] *= df['volume_spike']
    
    # Final Alpha Factor
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()
    df['final_alpha_factor'] = (df['combined_value_momentum'] + df['n_day_returns'] + df['close_open_diff']) * df['intraday_volatility'] * df['volume']
    
    return df['final_alpha_factor']
```
