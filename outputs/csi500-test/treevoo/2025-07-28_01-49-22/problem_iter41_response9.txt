```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smoothed_hl_range_ratio'] = df['hl_range_ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['smoothed_hl_range_ratio'] > 1.35, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_hl_range'] = df['high_low_range'] * df['volume']
    df['smoothed_volume_weighted_hl_range'] = df['volume_weighted_hl_range'].ewm(span=10).mean()
    
    # Generate Alpha Factor
    df['alpha_factor'] = np.where(df['smoothed_volume_weighted_hl_range'] > 1.35, 1, -1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_alpha_factor'] = df['smoothed_volume_weighted_hl_range'] / df['volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'] - df['close'].shift(1)
    
    # Calculate Short-Term and Long-Term Moving Averages of Daily Returns
    df['short_term_ma'] = df['daily_returns'].rolling(window=5).mean()
    df['long_term_ma'] = df['daily_returns'].rolling(window=30).mean()
    df['moving_average_diff'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['smoothed_volume_weighted_hl_range'] + 
        df['combined_value_momentum'] + 
        df['intraday_range_ratio'] - 
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['intraday_signal'] == 1, 
        df['intermediate_alpha_factor'] * 1.9, 
        df['intermediate_alpha_factor'] * 0.1
    )
    
    # Introduce Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Volume-Adjusted Close Price Momentum
    df['volume_adjusted_close_price_momentum'] = df['close_price_momentum'] * (df['volume'] / df['volume'].shift(1))
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (
        df['moving_average_diff'] * (
            df['adjusted_intermediate_alpha_factor'] + 
            df['volume_adjusted_close_price_momentum'] + 
            df['volume_momentum']
        )
    )
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    df['ema_volume_weighted_final_alpha_factor'] = df['volume_weighted_final_alpha_factor'].ewm(span=5).mean()
    
    return df['ema_volume_weighted_final_alpha_factor']
```
