```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10-day window)
    df['smoothed_high_low_ratio'] = df['high_low_ratio'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = np.where(df['smoothed_high_low_ratio'] > 1.0, 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['ema_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()
    
    # Generate Alpha Factor
    df['alpha_factor'] = np.where(df['ema_volume_weighted_high_low_range'] > 1.0, 1, -1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['alpha_factor_vol_adjusted'] = df['alpha_factor'] / df['volatility']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['ema_volume_weighted_high_low_range'] + 
        df['combined_value_momentum'] + 
        df['intraday_range_ratio'] - 
        (df['open'] - df['close'].shift(1))
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    df['signal'] = np.where(df['intraday_range_ratio'] > 1.0, 1, -1)
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = np.where(
        df['signal'] == 1, 
        df['intermediate_alpha_factor'] * 1.5,
        df['intermediate_alpha_factor'] * 0.5
    )
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['intraday_volatility'] * (
        df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']
    )
    
    # Weight by Volume
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    
    return df['final_alpha_factor']
```
