import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Range
    prev_intraday_range = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = (intraday_range / prev_intraday_range) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread = (df['close'] - df['open']) * ((df['close'].shift(1) - df['open'].shift(1)))
    
    # Calculate Price Change Velocity
    price_change_velocity = (df['close'] / df['close'].shift(5)) - 1
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    alpha_factor = intraday_range_momentum * price_change_velocity * df['volume']

    return alpha_factor
