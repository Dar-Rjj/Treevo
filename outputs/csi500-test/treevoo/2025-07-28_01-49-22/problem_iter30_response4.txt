```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.copy()
    adjusted_intermediate_alpha_factor[intraday_range_ratio > 0.6] *= 1.3
    adjusted_intermediate_alpha_factor[intraday_range_ratio < 0.4] *= 0.7
    
    # Introduce Volume Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Weighted Intraday Volatility
    weighted_intraday_volatility = intraday_volatility * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume-Adjusted Close Price Momentum
    volume_adjusted_close_price_momentum = close_price_momentum * (df['volume'] / df['volume'].shift(1))
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        weighted_intraday_volatility *
        (adjusted_intermediate_alpha_factor + volume_adjusted_close_price_momentum + volume_momentum)
    )
    
    return final_alpha_factor
```
