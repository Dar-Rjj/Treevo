import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Generate Alpha Factor
    alpha_factor = np.where(smoothed_high_low_range_ratio > 1.0, 1, -1) * smoothed_volume_weighted_high_low_range
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=20).std()
    
    # Adjust Alpha Factor for Volatility
    adjusted_alpha_factor = alpha_factor / volatility
    
    return adjusted_alpha_factor
