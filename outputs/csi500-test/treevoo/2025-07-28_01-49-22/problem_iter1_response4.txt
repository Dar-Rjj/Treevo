import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, alpha=0.1):
    # Calculate Momentum Component
    # Compute Logarithmic Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(n))
    
    # Smoothing Momentum
    df['momentum_ema'] = df['log_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Multiply Momentum by Volume
    df['weighted_momentum'] = df['momentum_ema'] * df['volume']
    
    # Calculate Volatility Component
    # Compute Absolute Daily Return
    df['abs_daily_return'] = abs(np.log(df['close']) - np.log(df['close'].shift(1)))
    
    # Sum of Absolute Returns over n Days
    df['sum_abs_returns'] = df['abs_daily_return'].rolling(window=n).sum()
    
    # Average by Counting Periods
    df['volatility_component'] = df['sum_abs_returns'] / n
    
    # Divide Volatility Component by Volume
    df['weighted_volatility'] = df['volatility_component'] / df['volume']
    
    # Final Alpha Factor Computation
    df['alpha_factor'] = df['weighted_momentum'] - df['weighted_volatility']
    
    return df['alpha_factor']
