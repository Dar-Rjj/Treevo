```python
def heuristics_v2(df):
    # Compute Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Calculate Momentum Factor
    momentum_window_size = 20
    df['momentum_factor'] = df['daily_returns'].rolling(window=momentum_window_size).mean()
    
    # Calculate Reversal Factor
    short_term_window_size = 5
    long_term_window_size = 30
    
    df['short_term_moving_avg'] = df['daily_returns'].rolling(window=short_term_window_size).mean()
    df['long_term_moving_avg'] = df['daily_returns'].rolling(window=long_term_window_size).mean()
    df['reversal_factor'] = df['long_term_moving_avg'] - df['short_term_moving_avg']
    
    # Combine Momentum and Reversal Factors
    df['alpha_factor'] = df['momentum_factor'] + df['reversal_factor']
    
    return df['alpha_factor']
```
