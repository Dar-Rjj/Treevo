```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    ema_window = 10
    df['ema_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=ema_window, adjust=False).mean()
    
    # Generate Alpha Factor based on Dynamic Threshold
    dynamic_threshold = df['ema_high_low_range_ratio'].expanding().mean()
    df['alpha_factor'] = np.where(df['ema_high_low_range_ratio'] > dynamic_threshold, 1, -1) * df['volume']
    
    # Calculate Daily Close-Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Adjust Alpha Factor by Close-Price Change
    df['alpha_factor_close_adjusted'] = df['alpha_factor'] * df['close_price_change']
    
    # Calculate High-Low Range Volatility
    volatility_window = 10
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=volatility_window).sum() / volatility_window
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=volatility_window).mean()
    df['volatility_adjusted_alpha_factor'] = df['alpha_factor_close_adjusted'] * (df['high_low_range_volatility'] * df['avg_volume'])
    
    # Calculate Open-Close Price Difference
    df['open_close_diff'] = df['open'] - df['close']
    
    # Adjust Alpha Factor by Open-Close Price Difference
    df['alpha_factor_open_close_adjusted'] = df['alpha_factor_close_adjusted'] * df['open_close_diff']
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * df['volatility_adjusted_alpha_factor'] + 0.5 * df['alpha_factor_open_close_adjusted']
    
    return final_alpha_factor
```
