import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    smoothed_high_low_ratio = high_low_ratio.ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = high_low_range * df['volume']
    smoothed_volume_weighted_high_low = volume_weighted_high_low.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_momentum_signal = np.where(smoothed_high_low_ratio > 1.2, 1, 
                                        np.where(smoothed_high_low_ratio < 0.8, -1, 0))
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=20).median()
    volume_filtered_df = df[df['volume'] >= median_volume]
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = (high_low_range / high_low_range.shift(1)) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread_ratio = (df['close'] - df['open']) * (df['close'].shift(1) - df['open'].shift(1))
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = smoothed_volume_weighted_high_low + combined_value_momentum - close_open_spread_ratio
    
    # Calculate Previous Day's Intraday Range
    prev_intraday_range = df['high'].shift(1) - df['low'].shift(1)
    
    # Calculate Price Change Velocity
    price_change_velocity = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    
    # Adjust Volatility by Volume
    adjusted_volatility_by_volume = intraday_volatility * df['volume'].rolling(window=10).mean()
    
    # Combine Intraday Range Momentum and Price Change Velocity with Volume
    combined_intraday_momentum_velocity_volume = (intraday_range_momentum * price_change_velocity) * df['volume']
    
    # Combine Price, Volume, and Volatility Indicators
    combined_price_volume_volatility = (price_change_velocity * volume_acceleration) * adjusted_volatility_by_volume
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        intraday_volatility * (intermediate_alpha_factor + close_price_momentum + intraday_range_momentum) +
        combined_intraday_momentum_velocity_volume +
        combined_price_volume_volatility
    )
    
    return final_alpha_factor
