```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (Window Size: 10 days)
    df['smooth_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    df['dynamic_threshold'] = df['smooth_high_low_range_ratio'].rolling(window=20).mean()
    df['momentum_signal'] = np.where(df['smooth_high_low_range_ratio'] > df['dynamic_threshold'], 1, -1)
    
    # EMA of Volume (Window Size: 5 days)
    df['ema_volume'] = df['volume'].ewm(span=5).mean()
    df['enhanced_momentum_factor'] = df['momentum_signal'] * df['ema_volume']
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['volume_adjusted_intraday_range_ratio'] = df['intraday_range_ratio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['prev_day_close_to_open_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily Close-Open Difference
    df['daily_close_open_diff'] = df['close'] - df['open']
    
    # Calculate High-Low Range Volatility
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=10).sum() / 10
    
    # Smooth High-Low Range Volatility with EMA (Window Size: 15 days)
    df['smooth_high_low_range_volatility'] = df['high_low_range_volatility'].ewm(span=15).mean()
    
    # Incorporate Close-Open Difference
    fixed_weight = 0.5
    df['incorporated_volatility'] = df['smooth_high_low_range_volatility'] + fixed_weight * df['daily_close_open_diff']
    
    # Generate Combined Alpha Factor
    df['combined_alpha_factor'] = df['volume_adjusted_intraday_range_ratio'] - df['prev_day_close_to_open_change']
    
    # Compare with Baseline (Fixed Threshold: 0.8)
    df['baseline_signal'] = np.where(df['smooth_high_low_range_volatility'] > 0.8, 1, -1)
    df['adjusted_combined_alpha_factor'] = df['combined_alpha_factor'] * df['baseline_signal'].apply(lambda x: 1.6 if x == 1 else 0.4)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_combined_alpha_factor'] * df['volume']
    
    return df['final_alpha_factor']
```
