def heuristics_v2(df):
    # Calculate Price Change Velocity
    price_change_velocity = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    
    # Combine Price Change Velocity and Volume Acceleration
    alpha_factor = price_change_velocity * volume_acceleration
    
    return alpha_factor
