```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'].shift(1) - df['close']).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range * abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * np.sqrt(df['volume'])
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=20).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = (combined_value / combined_value.shift(1)) - 1
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=30).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range + 
        combined_value_momentum + 
        (high_low_range / (df['high'] + df['low'])) -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio = high_low_range / (df['high'] + df['low'])
    high_low_ratio_20_day_ma = high_low_ratio.rolling(window=20).mean()
    positive_signal = (high_low_ratio > high_low_ratio_20_day_ma)
    negative_signal = (high_low_ratio < high_low_ratio_20_day_ma)
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.copy()
    adjusted_intermediate_alpha_factor[high_low_ratio > 1.2] *= 2.0
    adjusted_intermediate_alpha_factor[high_low_ratio < 0.8] *= 0.5
    
    # Introduce Volume Momentum
    volume_20_day_avg = df['volume'].rolling(window=20).mean()
    current_volume_ratio = df['volume'] / volume_20_day_avg
    adjusted_intermediate_alpha_factor[current_volume_ratio > 1.2] *= 1.25
    adjusted_intermediate_alpha_factor[current_volume_ratio < 0.8] *= 0.75
    
    # Generate Final Alpha Factor
    final_alpha_factor = adjusted_intermediate_alpha_factor + close_price_momentum
    
    # Adjust Final Alpha Factor by Volume
    n_period_average_volume = df['volume'].rolling(window=20).mean()
    final_alpha_factor = final_alpha_factor * n_period_average_volume
    
    return final_alpha_factor
```
