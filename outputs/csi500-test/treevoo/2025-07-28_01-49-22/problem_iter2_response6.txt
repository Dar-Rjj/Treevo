```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine Intraday Range and Absolute Close-to-Close Return
    combined_values = intraday_range + abs_close_return
    
    # Calculate 10-day simple moving average (SMA) of combined values
    sma_10_day = combined_values.rolling(window=10).mean()
    
    # Calculate Momentum from Combined Values
    momentum_combined = combined_values - sma_10_day
    
    return momentum_combined
```
