```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    prev_close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_high_low_and_close_to_close = high_low_range - abs_close_to_close_return
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    smooth_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = smooth_high_low_range_ratio.mean()
    alpha_factor_momentum = np.where(smooth_high_low_range_ratio > dynamic_threshold, 1, -1)
    alpha_factor_momentum = alpha_factor_momentum * df['volume']
    
    # Calculate Daily Close-Price Change
    daily_close_price_change = df['close'] - df['close'].shift(1)
    
    # Adjust Alpha Factor by Close-Price Change
    adjusted_alpha_factor_momentum = alpha_factor_momentum * daily_close_price_change
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Smooth Combined High-Low and Close-to-Close
    smooth_combined_high_low_and_close_to_close = combined_high_low_and_close_to_close.ewm(span=10).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = volume_adjusted_intraday_range_ratio - prev_close_to_open_change
    combined_alpha_factor_signal = np.where(smooth_combined_high_low_and_close_to_close > 1.0, 1.5, 0.5)
    adjusted_combined_alpha_factor = combined_alpha_factor * combined_alpha_factor_signal
    
    # Calculate Combined Value Momentum
    combined_value_momentum = (combined_alpha_factor - combined_alpha_factor.shift(1))
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_alpha_factor * (intraday_volatility * (combined_value_momentum + close_price_momentum))
    final_alpha_factor *= np.where(alpha_factor_momentum > 0, 1.2, 0.8)
    
    return final_alpha_factor
```
