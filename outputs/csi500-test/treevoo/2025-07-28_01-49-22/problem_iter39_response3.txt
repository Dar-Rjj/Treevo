import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-based Factor
    df['momentum_factor'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility-based Factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_factor'] = df['log_returns'].rolling(window=20).std()
    
    # Volume-related Momentum
    df['volume_momentum'] = 0
    df.loc[df['close'] > df['open'], 'volume_momentum'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    
    # Price Reversal Indicator
    df['price_reversal'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Accumulation/Distribution Line
    df['A/D'] = 0
    df['A/D'] = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    df['A/D'] = df['A/D'].cumsum()
    
    # High-Low Range and Intraday Momentum
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    df['smoothed_high_low_ratio'] = df['high_low_ratio'].ewm(span=10).mean()
    df['intraday_signal'] = np.where(df['smoothed_high_low_ratio'] > 1.3, 1, -1)
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['alpha_signal'] = np.where(df['volume_weighted_high_low_range'] > 1.3, 1, -1)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_alpha_factor'] = df['volume_weighted_high_low_range'] / df['volatility']

    df['abs_close_return'] = df['close'] - df['close'].shift(1)
    df['combined_high_low_abs_return'] = df['high_low_range'] - df['abs_close_return']

    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()

    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()

    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])

    df['intermediate_alpha_factor'] = df['volume_weighted_high_low_range'] + \
                                      df['combined_high_low_abs_return'] + \
                                      df['intraday_range_ratio'] - \
                                      (df['open'] - df['close'].shift(1))

    df['intermediate_alpha_factor'] = df['intermediate_alpha_factor'] * np.where(df['alpha_signal'] == 1, 1.8, 0.2)
