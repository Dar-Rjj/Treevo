```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Generate Alpha Factor from Momentum
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    df['dynamic_threshold'] = df['smoothed_high_low_range_ratio'].rolling(window=10).mean()
    df['signal'] = np.where(df['smoothed_high_low_range_ratio'] > df['dynamic_threshold'], 1, -1)
    df['alpha_factor'] = df['signal'] * df['volume']

    # Adjust Alpha Factor by Close-Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['alpha_factor'] *= df['close_price_change']

    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    weighted_close_open_diff = 0.5 * df['close_open_diff']
    df['alpha_factor'] += weighted_close_open_diff

    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * df['avg_volume']
    df['alpha_factor'] *= df['volume_adjusted_high_low_range']

    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = (df['close'] - df['close'].shift(1)).abs()

    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_factor'] = df['high_low_range'] - df['abs_close_to_close_return']
    df['alpha_factor'] += df['combined_factor']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()

    # Calculate Close Price Momentum
    df['n_day_returns'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=30).sum()
    df['alpha_factor'] *= df['n_day_returns']

    # Apply Volume Filter
    df['median_volume'] = df['volume'].rolling(window=30).median()
    df['alpha_factor'] = np.where(df['volume'] < df['median_volume'], 0, df['alpha_factor'])

    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=30).std()
    df['alpha_factor'] /= df['volatility']

    # Incorporate Trading Volume Spike
    df['volume_spike'] = df['volume'] / df['avg_volume']
    df['alpha_factor'] *= df['volume_spike']

    return df['alpha_factor']
```
