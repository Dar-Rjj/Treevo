```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate the daily high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the n-period average volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Multiply the high-low range by the average volume
    alpha_factor = df['high_low_range'] * df['avg_volume']
    
    return alpha_factor
```
