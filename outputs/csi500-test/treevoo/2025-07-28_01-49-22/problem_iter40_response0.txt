```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].shift(1)
    
    # Apply EMA on High-Low Range Ratio (e.g., 10 days)
    df['Smoothed_High_Low_Range_Ratio'] = df['High_Low_Range_Ratio'].ewm(span=10).mean()
    
    # Calculate Combined Value
    df['Combined_Value'] = (df['high'] - df['low']) + abs(df['close'] - df['close'].shift(1))
    
    # Calculate Volume Adjusted Intraday Range
    df['Volume_Adjusted_Intraday_Range'] = df['High_Low_Range'] * df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'] - df['Combined_Value'].shift(1)
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = df['close'].pct_change().rolling(window=20).sum()
    
    # Generate Enhanced Momentum Alpha Factor
    df['Dynamic_Threshold'] = df['Smoothed_High_Low_Range_Ratio'].rolling(window=20).mean()
    df['Enhanced_Momentum_Alpha_Factor'] = np.where(
        df['Smoothed_High_Low_Range_Ratio'] > df['Dynamic_Threshold'], 
        1, 
        -1
    )
    df['EMA_Volume'] = df['volume'].ewm(span=5).mean()
    df['Enhanced_Momentum_Alpha_Factor'] *= df['EMA_Volume']
    
    # Incorporate Close-Open Difference
    df['Close_Open_Difference'] = df['close'] - df['open']
    fixed_weight = 0.5  # Fixed weight for the Close-Open Difference
    df['Adjusted_Volatility'] = df['Intraday_Volatility'] + (fixed_weight * df['Close_Open_Difference'])
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Volume_Adjusted_Intraday_Range'] + 
        df['Combined_Value_Momentum'] + 
        df['Smoothed_High_Low_Range_Ratio'] - 
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(
        (df['high'] - df['low']) / (df['high'] + df['low']) > 0.6, 
        1.2 * df['Intermediate_Alpha_Factor'], 
        0.8 * df['Intermediate_Alpha_Factor']
    )
    
    # Compare with Baseline and adjust combined alpha factor
    df['Signal'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > df['Dynamic_Threshold'], 1, -1)
    df['Adjusted_Combined_Alpha_Factor'] = np.where(
        df['Signal'] > 0, 
        1.4 * df['Adjusted_Intermediate_Alpha_Factor'], 
        0.6 * df['Adjusted_Intermediate_Alpha_Factor']
    )
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Intraday_Volatility'] * 
        (df['Adjusted_Combined_Alpha_Factor'] + df['Close_Price_Momentum']) * 
        df['volume']
    )
    
    return df['Final_Alpha_Factor']
```
