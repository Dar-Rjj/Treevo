```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA
    df['smoothed_hl_range_ratio'] = df['hl_range_ratio'].ewm(span=10).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = df['smoothed_hl_range_ratio'].mean()
    df['alpha_signal'] = np.where(df['smoothed_hl_range_ratio'] > dynamic_threshold, 1, -1)
    df['momentum_alpha'] = df['alpha_signal'] * df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['adjusted_alpha'] = df['momentum_alpha'] * df['close_change']
    
    # Calculate High-Low Range Volatility
    n = 10
    df['range_volatility'] = df['high_low_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['vol_adjusted_volatility'] = df['range_volatility'] * df['avg_volume']
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.5
    df['combined_volatility'] = df['vol_adjusted_volatility'] + fixed_weight * df['close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    df['vol_adj_high_low_range'] = df['high_low_range'] * df['avg_volume']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Previous Day's Close-to-Open Change
    df['prev_close_to_open'] = df['open'] - df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['vol_adj_high_low_range'] + 
                                df['combined_value_momentum'] - 
                                df['prev_close_to_open'] + 
                                df['intraday_range_ratio'])
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha'] = np.where(df['smoothed_hl_range_ratio'] > dynamic_threshold, 
                                                 df['intermediate_alpha'] * 1.5, 
                                                 df['intermediate_alpha'] * 0.5)
    
    # Calculate Intraday Range Momentum
    df['intraday_range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Generate Final Alpha Factor
    df['final_alpha'] = (df['intraday_volatility'] * 
                         (df['adjusted_intermediate_alpha'] + 
                          df['close_price_momentum'] + 
                          df['intraday_range_momentum']))
    
    return df['final_alpha']
```
