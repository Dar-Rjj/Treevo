```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor
    df['alpha_factor'] = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1, -1)
    
    return df['alpha_factor']
```
