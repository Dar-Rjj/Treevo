```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute rolling averages for short-term (5-day) and long-term (20-day)
    df['short_term_sma'] = df['close'].rolling(window=5).mean()
    df['long_term_sma'] = df['close'].rolling(window=20).mean()
    
    # Difference between short and long-term averages
    df['sma_diff'] = df['short_term_sma'] - df['long_term_sma']
    
    # Compute rolling volatility for short-term (5-day) and long-term (20-day)
    df['short_term_volatility'] = df['daily_returns'].rolling(window=5).std()
    df['long_term_volatility'] = df['daily_returns'].rolling(window=20).std()
    
    # Calculate daily range
    df['daily_range'] = df['high'] - df['low']
    df['daily_range_percent'] = df['daily_range'] / df['close']
    
    # Compute moving averages of daily range
    df['range_5d_ma'] = df['daily_range'].rolling(window=5).mean()
    df['range_20d_ma'] = df['daily_range'].rolling(window=20).mean()
    
    # Calculate volume relative to average
    df['volume_5d_avg'] = df['volume'].rolling(window=5).mean()
    df['volume_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_5d'] = df['volume'] / df['volume_5d_avg']
    df['volume_ratio_20d'] = df['volume'] / df['volume_20d_avg']
    
    # Calculate amount relative to average
    df['amount_5d_avg'] = df['amount'].rolling(window=5).mean()
    df['amount_20d_avg'] = df['amount'].rolling(window=20).mean()
    df['amount_ratio_5d'] = df['amount'] / df['amount_5d_avg']
    df['amount_ratio_20d'] = df['amount'] / df['amount_20d_avg']
    
    # Create a composite factor with weighted sum
    df['composite_factor'] = (
        0.3 * df['sma_diff'] + 
        0.2 * (df['short_term_volatility'] - df['long_term_volatility']) + 
        0.1 * (df['range_5d_ma'] - df['range_20d_ma']) + 
        0.1 * (df['volume_ratio_5d'] - df['volume_ratio_20d']) + 
        0.1 * (df['amount_ratio_5d'] - df['amount_ratio_20d'])
    )
    
    return df['composite_factor']
```
