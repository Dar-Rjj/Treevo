```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the difference between close and open prices
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the money flow by multiplying the price change with the volume
    df['money_flow'] = df['price_change'] * df['volume']
    
    # Calculate the factor: Money Flow / Price Range
    # This factor could be interpreted as a measure of the momentum in the context of the trading range.
    # A higher value would indicate strong buying or selling pressure within the day's range.
    factor = df['money_flow'] / df['price_range']
    
    return factor
```
