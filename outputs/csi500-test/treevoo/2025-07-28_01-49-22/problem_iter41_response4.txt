import pandas as pd

def heuristics_v2(df):
    # Simple Moving Average (SMA) of Close Prices over 20 days
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) of Close Prices over 9 days
    df['EMA_9'] = df['close'].ewm(span=9, adjust=False).mean()
    
    # Relative Strength Index (RSI) of Close Prices over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Rate of Change (ROC) of Close Prices with a 1-day lag
    df['ROC_1'] = df['close'].pct_change(periods=1)
    
    # True Range (TR) calculated from High, Low, and Close prices
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    
    # Average True Range (ATR) over 14 days
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Volume Simple Moving Average (VSMA) over 20 days
    df['VSMA_20'] = df['volume'].rolling(window=20).mean()
    
    # Volume Exponential Moving Average (VEMA) over 9 days
    df['VEMA_9'] = df['volume'].ewm(span=9, adjust=False).mean()
    
    # On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Chaikin Money Flow (CMF) over 20 days
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Price-Volume Trend (PVT)
    df['PVT'] = (df['close'].diff(1) / df['close'].shift(1)) * df['volume']
    df['PVT'] = df['PVT'].cumsum()
    
    # Inside Day pattern detection
    df['Inside_Day'] = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1))
    
    # Outside Day pattern detection
    df['Outside_Day'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    
    # Dynamic Support and Resistance using SMA(20) and EMA(9)
    df['Dynamic_Support'] = df[['SMA_20', 'EMA_9']].min(axis=1)
    df['Dynamic_Resistance'] = df[['SMA_20', 'EMA_9']].max(axis=1)

    df['Divergence_RSI_ROC'] = (df['RSI_14'] - df['RSI_14'].shift(1)) * (df['ROC_1'] - df['ROC_1'].shift(1)) < 0
