```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of high and low with volume
    weight_avg = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate the difference between close and open, then multiply by amount
    diff_close_open = (df['close'] - df['open']) * df['amount']
    
    # Generate the alpha factor by combining the two components
    alpha_factor = weight_avg + diff_close_open
    
    return alpha_factor
```
