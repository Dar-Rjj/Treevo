```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = np.abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value.diff()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio
    ema_high_low_range_ratio = high_low_range_ratio.ewm(span=10, adjust=False).mean()
    
    # Compare with Baseline (Dynamic Threshold)
    dynamic_threshold = ema_high_low_range_ratio.mean()
    signal = np.where(ema_high_low_range_ratio > dynamic_threshold, 1.5, 0.5)
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * signal
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = high_low_range.rolling(window=10).sum() / 10
    
    # Adjust Volatility by Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    adjusted_volatility = high_low_range_volatility * avg_volume
    
    # Incorporate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    weighted_close_open_diff = close_open_diff * 0.1  # Fixed weight
    volatility_with_close_open = adjusted_volatility + weighted_close_open_diff
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * avg_volume
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = high_low_range - high_low_range.shift(1)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        intraday_volatility *
        (adjusted_intermediate_alpha_factor + close_price_momentum + intraday_range_momentum)
    )
    
    # Adjust Final Alpha Factor by Volume
    daily_high_low_range = high_low_range
    final_alpha_factor *= daily_high_low_range * avg_volume
    
    return final_alpha_factor
```
