```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['volume_adjusted_intraday_range_ratio'] = df['intraday_range_ratio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['prev_close_to_open_change'] = df['open'].shift(-1) - df['close']
    
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility (n-period sum)
    n = 10
    df['high_low_volatility'] = df['daily_high_low_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_high_low_volatility'] = df['high_low_volatility'] * df['avg_volume']
    
    # Smooth High-Low Range Volatility with EMA
    ema_window = 10
    df['smoothed_high_low_volatility'] = df['adjusted_high_low_volatility'].ewm(span=ema_window, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.5
    df['combined_volatility'] = df['smoothed_high_low_volatility'] + fixed_weight * df['close_open_diff']
    
    # Generate Combined Alpha Factor
    df['combined_alpha_factor'] = df['volume_adjusted_intraday_range_ratio'] - df['prev_close_to_open_change']
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    df['signal'] = np.where(df['smoothed_high_low_volatility'] > 1.0, 1.5, 0.5)
    df['adjusted_combined_alpha_factor'] = df['combined_alpha_factor'] * df['signal']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['daily_high_low_range'] / df['daily_high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=ema_window, adjust=False).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1, -1)
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['daily_high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['daily_high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['intraday_volatility'] * (df['combined_value_momentum'] + df['close_price_momentum']) * df['volume']
    
    return df['final_alpha_factor']
```
