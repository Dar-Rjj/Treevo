```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_to_close_return
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    n_day_returns = (df['close'] / df['close'].shift(1)) - 1
    close_price_momentum = n_day_returns.rolling(window=20).sum()
    
    # Generate Interim Alpha Factor
    interim_alpha_factor = (intraday_volatility * close_price_momentum) + combined_value_momentum
    
    # Calculate Previous Day's Intraday Range
    prev_high_low_range = (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = (high_low_range / prev_high_low_range) - 1
    
    # Calculate Close-Open Spread Ratio
    close_open_spread_ratio = (df['close'] - df['open']) * ((df['close'].shift(1) - df['open'].shift(1)))
    
    # Calculate Price Change Velocity
    price_change_velocity = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - df['volume'].shift(1) - df['volume'].shift(2)
    
    # Calculate Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = daily_high_low_range.rolling(window=10).sum() / 10
    
    # Adjust Volatility by Volume
    adjusted_volatility_by_volume = high_low_range_volatility * df['volume'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Range Momentum
    volume_weighted_intraday_range_momentum = intraday_range_momentum * df['volume'].rolling(window=10).mean()
    
    # Combine Volume-Weighted Intraday Range Momentum and Price Change Velocity
    vol_weighted_intraday_range_and_price_velocity = volume_weighted_intraday_range_momentum * price_change_velocity * df['volume']
    
    # Combine Price, Volume, and Volatility Indicators
    price_vol_volatility_indicator = price_change_velocity * volume_acceleration * adjusted_volatility_by_volume
    
    # Generate Adjusted Alpha Factor
    adjusted_alpha_factor = vol_weighted_intraday_range_and_price_velocity + price_vol_volatility_indicator
    
    # Generate Final Alpha Factor
    final_alpha_factor = interim_alpha_factor + adjusted_alpha_factor
    
    # Calculate Open Price Momentum
    open_price_momentum = df['open'] - df['open'].shift(1)
    
    # Calculate Volume-Weighted Open Price Momentum
    volume_weighted_open_price_momentum = open_price_momentum * df['volume'].rolling(window=10).mean()
    
    # Incorporate Open Price Momentum into Final Alpha Factor
    final_alpha_factor += volume_weighted_open_price_momentum
    
    return final_alpha_factor
```
