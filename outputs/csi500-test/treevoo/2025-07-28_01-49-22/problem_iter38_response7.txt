```python
import pandas as pd

def heuristics_v2(df):
    """
    This function takes a pandas DataFrame with columns (open, high, low, close, amount, volume)
    and calculates a novel alpha factor based on the thought tree.
    
    :param df: pandas DataFrame with index (date) and columns (open, high, low, close, amount, volume)
    :return: pandas Series representing the factor values
    """
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure by considering the close price minus the low price
    buying_pressure = df['close'] - df['low']
    
    # Calculate the selling pressure by considering the high price minus the open price
    selling_pressure = df['high'] - df['open']
    
    # Calculate the net pressure by subtracting selling pressure from buying pressure
    net_pressure = buying_pressure - selling_pressure
    
    # Calculate the money flow by multiplying the net pressure with the volume
    money_flow = net_pressure * df['volume']
    
    # Calculate the weighted money flow by dividing the money flow by the price range
    # Avoid division by zero by adding a small constant to the price range
    weighted_money_flow = money_flow / (price_range + 1e-6)
    
    # The final factor is the rolling sum of the weighted money flow over the last 5 days
    factor = weighted_money_flow.rolling(window=5).sum()
    
    return factor
```
