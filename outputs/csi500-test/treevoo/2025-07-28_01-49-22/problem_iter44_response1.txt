import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Smoothed High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Smoothed_High_Low_Range'] = df['High_Low_Range'].ewm(span=10, adjust=False).mean()

    # Generate Intraday Momentum Signal
    df['Intraday_Momentum_Signal'] = np.where(df['close'] > df['open'] * 1.3, 1, -1)

    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = (df['High_Low_Range'] * df['volume']).ewm(span=10, adjust=False).mean()

    # Calculate Absolute Close-to-Close Return
    df['Abs_Close_to_Close_Return'] = abs(df['close'].diff(1))

    # Combine High-Low and Close-to-Close Returns
    df['Combined_Value'] = df['High_Low_Range'] + df['Abs_Close_to_Close_Return']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()

    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'].diff(1)

    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()

    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (df['Volume_Weighted_High_Low_Range'] 
                                       + df['Combined_Value_Momentum'] 
                                       - (df['open'] - df['close'].shift(1)))

    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Adjusted_Intermediate_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] / df['Volatility']

    # Adjust Intermediate Alpha Factor based on Intraday Momentum Signal
    df['Adjusted_Intermediate_Alpha_Factor'] = np.where(
        df['Intraday_Momentum_Signal'] == 1,
        df['Adjusted_Intermediate_Alpha_Factor'] * 1.8,
        df['Adjusted_Intermediate_Alpha_Factor'] * 0.2
    )

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Intermediate_Alpha_Factor'] + df['Close_Price_Momentum']) * df['Intraday_Volatility']

    # Weight by Volume
    df['Volume_Weighted_Final_Alpha_Factor'] = (df['Final_Alpha_Factor'] * df['volume']).ewm(span=5, adjust=False).mean()

    # Introduce Additional Momentum Component
    df['MA_50'] = df['close'].rolling(window=50).mean()
    df['MA_200'] = df['close'].rolling(window=200).mean()
    df['Long_Term_Momentum'] = np.where(df['MA_50'] > df['MA_200'], 1, -1)

    # Adjust Final Alpha Factor based on Long-Term Momentum
    df['Final_Alpha_Factor_Adjusted'] = np.where(
        df['Long_Term_Momentum'] == 1,
        df['Volume_Weighted_Final_Alpha_Factor'] * 1.2,
        df['Volume_Weighted_Final_Alpha_Factor'] * 0.8
    )

    # Calculate True Range (TR)
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)

    # Calculate Average True Range (ATR)
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = np.where(df['close'] > df['close'].shift(1), df['volume'], -df['volume'])
    df['OBV'] = df['OBV'].cumsum()

    # Calculate OBV trend over 30 days
    df['OBV_Trend'] = df['OBV'].rolling(window=30).mean()
