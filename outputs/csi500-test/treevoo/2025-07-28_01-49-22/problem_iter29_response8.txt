```python
def heuristics_v2(df):
    # Calculate the price change from open to close
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the volume-weighted price change
    df['vwap_change'] = (df['amount'] * df['price_change']) / df['volume']
    
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / (df['high'] - df['low'])
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['vwap_change'] / df['intraday_range']) * df['norm_volume']
    
    return df['alpha_factor']
```
