```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range with EMA
    ema_high_low_range = high_low_range.ewm(span=10, adjust=False).mean()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (high_low_range * df['volume']).ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor from High-Low Range
    alpha_factor_high_low = (ema_high_low_range > 1.0).astype(int) * 2 - 1
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_high_low_close = high_low_range - abs_close_to_close_return
    
    # Smooth Combined High-Low and Close-to-Close
    ema_combined_high_low_close = combined_high_low_close.ewm(span=10, adjust=False).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = volume_adjusted_intraday_range_ratio - close_to_open_change
    combined_alpha_signal = (ema_combined_high_low_close > 1.0).astype(int) * 2 - 1
    adjusted_combined_alpha_factor = combined_alpha_factor * np.where(combined_alpha_signal > 0, 1.5, 0.5)
    
    # Calculate Combined Value Momentum
    combined_value_momentum = (adjusted_combined_alpha_factor - adjusted_combined_alpha_factor.shift(1)).fillna(0)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum().fillna(0)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_alpha_factor * (combined_value_momentum + close_price_momentum)
    
    return final_alpha_factor
```
