```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_return']
    
    # Calculate Volume Adjusted Intraday Range
    df['vol_adj_intraday_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Intraday Volatility (Average of (High - Low) over 10 days)
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'].diff()
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = df['high_low_range'] / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha'] = (df['vol_adj_intraday_range'] + 
                                df['combined_value_momentum'] + 
                                df['intraday_range_ratio'] - 
                                (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor based on Intraday Range Ratio
    df['adjusted_intermediate_alpha'] = df.apply(
        lambda row: row['intermediate_alpha'] * 1.2 if row['intraday_range_ratio'] > 0.5 else row['intermediate_alpha'] * 0.8,
        axis=1
    )
    
    # Introduce Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Generate Final Alpha Factor
    df['final_alpha'] = df['intraday_volatility'] * (df['adjusted_intermediate_alpha'] + 
                                                     df['close_price_momentum'] + 
                                                     df['volume_momentum'])
    
    return df['final_alpha'].dropna()
```
