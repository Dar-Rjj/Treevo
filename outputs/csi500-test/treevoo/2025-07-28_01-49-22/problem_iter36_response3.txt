import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio with EMA
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_signal = (smoothed_high_low_range_ratio > 1.35).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    smoothed_volume_weighted_high_low_range = volume_weighted_high_low_range.ewm(span=10).mean()
    
    # Generate Alpha Factor
    alpha_factor = (smoothed_volume_weighted_high_low_range > 1.35 * smoothed_volume_weighted_high_low_range.median()).astype(int) * 2 - 1
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_alpha_factor = alpha_factor / volatility
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (smoothed_volume_weighted_high_low_range +
                                 combined_value_momentum +
                                 intraday_range_ratio -
                                 (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor based on signal
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * (alpha_factor > 0).map({True: 1.9, False: 0.1})
    
    # Generate Final Alpha Factor
    final_alpha_factor = (adjusted_intermediate_alpha_factor + close_price_momentum) * intraday_volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = final_alpha_factor * df['volume']
    final_alpha_factor = volume_weighted_final_alpha_factor.ewm(span=5).mean()
    
    return final_alpha_factor
