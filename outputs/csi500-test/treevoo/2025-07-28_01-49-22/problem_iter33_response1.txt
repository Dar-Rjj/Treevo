```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio with EMA (10 days)
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Volume Adjusted Intraday Range
    df['volume_adjusted_intraday_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum over the last 20 days
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Dynamic Threshold for Smoothed High-Low Range Ratio
    df['dynamic_threshold'] = df['smoothed_high_low_range_ratio'].rolling(window=20).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    df['signal'] = np.where(df['smoothed_high_low_range_ratio'] > df['dynamic_threshold'], 1, -1)
    
    # Apply EMA on Volume (5 days)
    df['ema_volume'] = df['volume'].ewm(span=5).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Add Weighted Close-Open Difference to Adjusted Volatility
    fixed_weight = 0.3
    df['adjusted_volatility'] = df['intraday_volatility'] + (fixed_weight * df['close_open_diff'])
    
    # Calculate High-Low Range Volatility
    n = 10
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=n).sum() / n
    
    # Smooth High-Low Range Volatility with EMA (15 days)
    df['smoothed_high_low_range_volatility'] = df['high_low_range_volatility'].ewm(span=15).mean()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['volume_adjusted_intraday_range'] 
        + df['combined_value_momentum'] 
        + df['high_low_range_ratio'] 
        - (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    df['adjustment_factor'] = np.where(df['high_low_range'] / (df['high'] + df['low']) > 0.6, 1.2, 0.8)
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] * df['adjustment_factor']
    
    # Compare with Baseline (Dynamic Threshold)
    df['baseline_signal'] = np.where(df['smoothed_high_low_range_volatility'] > df['dynamic_threshold'], 1, -1)
    
    # Adjust Combined Alpha Factor
    df['final_adjustment'] = np.where(df['baseline_signal'] > 0, 1.4, 0.6)
    df['adjusted_combined_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] * df['final_adjustment']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['intraday_volatility'] * (df['adjusted_combined_alpha_factor'] + df['close_price_momentum']) * df['ema_volume']
    
    return df['final_alpha_factor']
```
