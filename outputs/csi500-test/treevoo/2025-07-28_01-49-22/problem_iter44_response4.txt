```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of close to open price
    df['close_to_open'] = df['close'] / df['open']
    
    # Calculate the ratio of high to low price
    df['high_to_low'] = df['high'] / df['low']
    
    # Calculate the ratio of amount to volume
    df['amount_to_volume'] = df['amount'] / df['volume']
    
    # Combine the factors into a single factor
    df['factor'] = (df['close_change'] + df['close_to_open'] + df['high_to_low'] + df['amount_to_volume']) / 4
    
    return df['factor'].dropna()
```
