import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the open-close range
    df['open_close_range'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted price
    df['volume_weighted_price'] = (df['amount'] / df['volume'])
    
    # Calculate the relative position of the close price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Construct the heuristic factor
    df['heuristic_factor'] = (df['open_close_range'] 
                              + 2 * df['close_position'] 
                              - 0.5 * df['high_low_range'] 
                              + 0.3 * df['volume_weighted_price'].pct_change())
    
    # Return the factor values, excluding the first row due to NaN values from pct_change
    return df['heuristic_factor'].iloc[1:]
