```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    ten_day_ma_intraday_range_ratio = intraday_range_ratio.rolling(window=10).mean()
    dynamic_threshold = (intraday_range_ratio > ten_day_ma_intraday_range_ratio).astype(int) * 1.5 + (intraday_range_ratio < ten_day_ma_intraday_range_ratio).astype(int) * 0.5
    
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor * dynamic_threshold
    
    # Introduce Volume Momentum
    ten_day_average_volume = df['volume'].rolling(window=10).mean()
    current_volume_ratio = df['volume'] / ten_day_average_volume
    volume_momentum_adjustment = (current_volume_ratio > 1).astype(int) * 1.15 + (current_volume_ratio < 1).astype(int) * 0.85
    final_alpha_factor = intraday_volatility * (adjusted_intermediate_alpha_factor + close_price_momentum) * volume_momentum_adjustment
    
    return final_alpha_factor
```
