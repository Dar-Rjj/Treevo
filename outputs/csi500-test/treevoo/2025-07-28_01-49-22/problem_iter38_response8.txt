```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the volume-weighted price change
    volume_weighted_price_change = price_change * df['volume']
    
    # Calculate the amount-weighted price change
    amount_weighted_price_change = price_change * df['amount']
    
    # Combine the high-low ratio, volume-weighted, and amount-weighted price change
    factor = (high_low_ratio + volume_weighted_price_change + amount_weighted_price_change) / 3
    
    return factor
```
