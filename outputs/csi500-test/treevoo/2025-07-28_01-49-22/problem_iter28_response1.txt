import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range
        + combined_value_momentum
        + (high_low_range / (df['high'] + df['low']))
        - (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    ten_day_ma_high_low_ratio = (high_low_range / (df['high'] + df['low'])).rolling(window=10).mean()
    dynamic_threshold = (high_low_range / (df['high'] + df['low'])) > ten_day_ma_high_low_ratio
    intermediate_alpha_factor = intermediate_alpha_factor * (1.5 * dynamic_threshold + 0.5 * (1 - dynamic_threshold))
    
    # Introduce Volume Momentum
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    current_volume_ratio = df['volume'] / ten_day_avg_volume
    volume_momentum_multiplier = 1.15 * (current_volume_ratio > 1) + 0.85 * (current_volume_ratio < 1)
    intermediate_alpha_factor = intermediate_alpha_factor * volume_momentum_multiplier
    
    # Generate Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor + close_price_momentum
    
    # Adjust Final Alpha Factor by Volume
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    final_alpha_factor = final_alpha_factor * n_period_avg_volume
    
    return final_alpha_factor
