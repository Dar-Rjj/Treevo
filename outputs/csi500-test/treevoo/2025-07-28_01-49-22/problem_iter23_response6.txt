```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate the absolute value of close-open difference
    abs_close_open_diff = (df['close'] - df['open']).abs()
    
    # Calculate the weighted average price
    weighted_avg_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    
    # Calculate the amount-volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Calculate the alpha factor
    alpha_factor = (high_low_diff / abs_close_open_diff) * (weighted_avg_price / amount_volume_ratio)
    
    return alpha_factor
```
