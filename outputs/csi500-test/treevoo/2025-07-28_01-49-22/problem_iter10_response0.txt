import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    volume_adjusted_intraday_range_ratio = intraday_range_ratio * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    prev_day_close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio using EMA with a window size of 10 days
    ema_window_size = 10
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=ema_window_size, adjust=False).mean()
    
    # Generate Combined Alpha Factor
    combined_alpha_factor = (volume_adjusted_intraday_range_ratio - prev_day_close_to_open_change)
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0) and Adjust Combined Alpha Factor
    baseline_threshold = 1.0
    adjusted_combined_alpha_factor = np.where(
        smoothed_high_low_range_ratio > baseline_threshold,
        combined_alpha_factor * 1.5,
        combined_alpha_factor * 0.5
    )
    
    # Final Alpha Factor: Adjusted Combined Alpha Factor multiplied by Volume
    final_alpha_factor = adjusted_combined_alpha_factor * df['volume']
    
    return pd.Series(final_alpha_factor, index=df.index)
