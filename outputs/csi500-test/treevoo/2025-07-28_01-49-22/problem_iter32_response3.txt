```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_to_close_return = (df['close'].diff()).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_to_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_adjusted_intraday_range +
        combined_value +
        intraday_range_ratio -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha_factor = intermediate_alpha_factor.where(
        intraday_range_ratio <= 0.6, 
        intermediate_alpha_factor * 1.3
    ).where(
        intraday_range_ratio >= 0.4, 
        intermediate_alpha_factor * 0.7
    )
    
    # Introduce Volume Momentum
    volume_momentum = (df['volume'].diff() / df['volume'].shift(1)).fillna(0)
    
    # Calculate Weighted Intraday Volatility
    weighted_intraday_volatility = intraday_volatility * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Volume-Adjusted Close Price Momentum
    volume_adjusted_close_price_momentum = close_price_momentum * (df['volume'] / df['volume'].shift(1))
    
    # Calculate Daily Returns for Momentum and Reversal Factors
    daily_returns = df['close'].diff()
    
    # Calculate Short-Term Moving Average of Daily Returns
    short_term_moving_avg = daily_returns.rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average of Daily Returns
    long_term_moving_avg = daily_returns.rolling(window=30).mean()
    
    # Calculate Reversal Factor
    reversal_factor = long_term_moving_avg - short_term_moving_avg
    
    # Combine Momentum and Reversal Factors
    combined_momentum_reversal_factor = volume_adjusted_close_price_momentum + reversal_factor
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        weighted_intraday_volatility *
        (adjusted_intermediate_alpha_factor + combined_momentum_reversal_factor + volume_momentum)
    )
    
    return final_alpha_factor
```
