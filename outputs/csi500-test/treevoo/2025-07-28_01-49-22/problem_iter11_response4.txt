```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['volume_adjusted_intraday_range_ratio'] = df['intraday_range_ratio'] * df['volume']
    
    # Calculate 10-day Volume-Weighted Average Price (VWAP)
    df['vwap_10_day'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Momentum
    df['momentum'] = df['close'] - df['vwap_10_day']
    
    # Calculate Daily High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Volatility
    df['high_low_range_volatility'] = df['daily_high_low_range'].rolling(window=10).mean()
    
    # Adjust Volatility by Volume
    average_volume_10_day = df['volume'].rolling(window=10).mean()
    df['adjusted_volatility'] = df['high_low_range_volatility'] * average_volume_10_day
    
    # Smooth High-Low Range Volatility using EMA
    df['smoothed_volatility'] = df['adjusted_volatility'].ewm(span=10).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    fixed_weight = 0.1  # Fixed weight for the close-open difference
    df['smoothed_volatility_with_close_open'] = df['smoothed_volatility'] + fixed_weight * df['close_open_diff']
    
    # Generate Combined Alpha Factor
    df['combined_alpha_factor'] = (
        df['volume_adjusted_intraday_range_ratio'] 
        - df['close_open_diff'].shift(1)  # Previous Day's Close-to-Open Change
        + df['momentum']
    )
    
    # Apply Baseline Comparison
    df['signal'] = np.where(df['smoothed_volatility'] > 1.0, 1.5, 0.5)
    df['combined_alpha_factor'] = df['combined_alpha_factor'] * df['signal']
    
    # Adjust by Volume
    df['final_alpha_factor'] = df['combined_alpha_factor'] * average_volume_10_day
    
    return df['final_alpha_factor']
```
