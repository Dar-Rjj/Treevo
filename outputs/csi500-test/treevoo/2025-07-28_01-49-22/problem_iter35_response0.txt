```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Smooth High-Low Range
    df['smooth_high_low_range'] = df['high_low_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_signal'] = (df['high_low_range'] > 1.3).astype(int) * 2 - 1  # 1 for positive, -1 for negative
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['smooth_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low and Close-to-Close Returns
    df['combined_value'] = df['high_low_range'] + df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (
        df['smooth_volume_weighted_high_low_range'] +
        df['combined_value_momentum'] +
        df['high_low_range'] / df['intraday_volatility'] -
        (df['open'] - df['close'].shift(1))
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_intermediate_alpha_factor'] = df['intermediate_alpha_factor'] / df['volatility']
    
    # Adjust Intermediate Alpha Factor
    df['adjusted_intermediate_alpha_factor'] = df['adjusted_intermediate_alpha_factor'] * (df['intraday_signal'] * 0.8 + 1.5)
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_intermediate_alpha_factor'] + df['close_price_momentum']) * df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_final_alpha_factor'] = df['final_alpha_factor'] * df['volume']
    df['smooth_volume_weighted_final_alpha_factor'] = df['volume_weighted_final_alpha_factor'].ewm(span=5).mean()
    
    # Introduce Additional Momentum Component
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['long_term_momentum'] = (df['50_day_ma'] > df['200_day_ma']).astype(int) * 2 - 1  # 1 for positive, -1 for negative
    
    # Adjust Final Alpha Factor
    df['final_alpha_factor'] = df['smooth_volume_weighted_final_alpha_factor'] * (df['long_term_momentum'] * 0.4 + 1.2)
    
    return df['final_alpha_factor']
```
