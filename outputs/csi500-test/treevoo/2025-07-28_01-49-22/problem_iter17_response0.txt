```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['Close_to_Close_Return'] = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['Combined_Value'] = df['High_Low_Range'] - df['Close_to_Close_Return']
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * df['volume']
    df['EMA_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'].ewm(span=10).mean()
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'].rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['Combined_Value_Momentum'] = df['Combined_Value'].diff()
    
    # Calculate Close Price Momentum
    df['Close_Price_Momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['Intraday_Range_Ratio'] = df['High_Low_Range'] / (df['high'] + df['low'])
    
    # Smooth High-Low Range Ratio
    df['Smoothed_High_Low_Range_Ratio'] = df['Intraday_Range_Ratio'].ewm(span=10).mean()
    
    # Generate Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['EMA_Volume_Weighted_High_Low_Range'] +
        df['Combined_Value_Momentum'] -
        (df['close'] - df['open']) / (df['high'] + df['low']) +
        df['Intraday_Range_Ratio'] -
        df['close'].diff().shift(1)
    )
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    df['Signal'] = np.where(df['Smoothed_High_Low_Range_Ratio'] > 1.0, 1.5, 0.5)
    
    # Adjust Intermediate Alpha Factor
    df['Adjusted_Intermediate_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] * df['Signal']
    
    # Calculate Intraday Range Momentum
    df['Intraday_Range_Momentum'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Normalized_Intermediate_Alpha_Factor'] = df['Adjusted_Intermediate_Alpha_Factor'] / df['Volatility']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Intraday_Volatility'] * (
            df['Normalized_Intermediate_Alpha_Factor'] + 
            df['Close_Price_Momentum'] + 
            df['Intraday_Range_Momentum']
        )
    )
    
    # Weight by Volume
    df['Final_Alpha_Factor_Weighted'] = df['Final_Alpha_Factor'] * df['volume']
    
    return df['Final_Alpha_Factor_Weighted']
```
