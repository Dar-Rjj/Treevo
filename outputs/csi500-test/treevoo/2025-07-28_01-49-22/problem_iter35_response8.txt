```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = (df['close'].shift(1) - df['close']).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - close_to_close_return
    
    # Calculate Volume-Weighted Intraday Range
    volume_weighted_intraday_range = high_low_range * df['volume']
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_price_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Previous Day's Close-to-Open Change
    close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Smooth High-Low Range Volatility
    ema_window_size = 15
    smoothed_high_low_range_volatility = high_low_range.ewm(span=ema_window_size, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    close_open_difference = df['close'] - df['open']
    fixed_weight = 0.5
    adjusted_smoothed_high_low_range_volatility = smoothed_high_low_range_volatility + fixed_weight * close_open_difference
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (
        volume_weighted_intraday_range +
        combined_value_momentum +
        intraday_range_ratio -
        close_to_open_change
    )
    
    # Adjust Intermediate Alpha Factor
    intermediate_alpha_factor[intraday_range_ratio > 1.0] *= 1.8
    intermediate_alpha_factor[intraday_range_ratio < 1.0] *= 0.4
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    dynamic_threshold = adjusted_smoothed_high_low_range_volatility.ewm(span=ema_window_size, adjust=False).mean()
    positive_signal = adjusted_smoothed_high_low_range_volatility > dynamic_threshold
    negative_signal = adjusted_smoothed_high_low_range_volatility < dynamic_threshold
    
    # Adjust Combined Alpha Factor
    intermediate_alpha_factor[positive_signal] *= 1.7
    intermediate_alpha_factor[negative_signal] *= 0.3
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    intermediate_alpha_factor /= volatility
    
    # Generate Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_factor + close_price_momentum) * df['volume']
    
    # Weight by Volume
    ema_volume_window = 5
    final_alpha_factor = final_alpha_factor.ewm(span=ema_volume_window, adjust=False).mean()
    
    return final_alpha_factor
```
