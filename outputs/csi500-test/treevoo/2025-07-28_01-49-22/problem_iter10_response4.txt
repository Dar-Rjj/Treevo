```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range.rolling(window=n).sum() - high_low_range
    
    # Adjust High-Low Range Momentum by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_high_low_momentum = high_low_momentum / avg_volume
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = adjusted_high_low_momentum - close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Final Alpha Factor
    alpha_factor = intraday_volatility * (adjusted_high_low_momentum + combined_value_momentum)
    
    return alpha_factor
```
