```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Range
    prev_high_low_range = high_low_range.shift(1)
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / prev_high_low_range
    
    # Smooth High-Low Range Ratio using EMA with a 10-day window
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=10, adjust=False).mean()
    
    # Generate Alpha Factor
    alpha_factor = pd.Series(index=df.index, dtype='float64')
    alpha_factor[smoothed_high_low_range_ratio > 1.0] = 1
    alpha_factor[smoothed_high_low_range_ratio < 1.0] = -1
    
    # Multiply by Volume Weight
    recent_volume_average = df['volume'].rolling(window=5).mean()
    volume_weight = df['volume'] / recent_volume_average
    alpha_factor *= volume_weight
    
    return alpha_factor
```
