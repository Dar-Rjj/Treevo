def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Range
    prev_intraday_range = intraday_range.shift(1)
    
    # Calculate Intraday Range Momentum
    range_momentum = (intraday_range / prev_intraday_range) - 1
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = range_momentum - abs(close_to_close_return)
    
    # Calculate Combined Value Momentum
    combined_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close-Open Spread Ratio
    close_open_spread = df['close'] - df['open']
    prev_close_open_spread = close_open_spread.shift(1)
    close_open_spread_ratio = close_open_spread * prev_close_open_spread
    
    # Final alpha factor
    alpha_factor = combined_momentum + close_open_spread_ratio

    return alpha_factor
