import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = high_low_range / high_low_range.shift(1)
    
    # Smooth High-Low Range Ratio using EMA
    ema_window_size = 10
    smoothed_high_low_range_ratio = high_low_range_ratio.ewm(span=ema_window_size, adjust=False).mean()
    
    # Generate Alpha Factor from Enhanced Momentum
    dynamic_threshold = smoothed_high_low_range_ratio.rolling(window=20).mean()
    momentum_signal = (smoothed_high_low_range_ratio > dynamic_threshold).astype(int) - (smoothed_high_low_range_ratio < dynamic_threshold).astype(int)
    
    # Volume Weighted Moving Average
    vwma_window_size = 5
    volume_weighted_average = (df['close'] * df['volume']).rolling(window=vwma_window_size).sum() / df['volume'].rolling(window=vwma_window_size).sum()
    alpha_factor_momentum = momentum_signal * volume_weighted_average
    
    # Calculate High-Low Range Volatility
    n_period = 20
    volatility = high_low_range.rolling(window=n_period).sum() / n_period
    
    # Adjust Volatility by Exponential Moving Average of Volume
    ema_volume_window_size = 5
    ema_volume = df['volume'].ewm(span=ema_volume_window_size, adjust=False).mean()
    adjusted_volatility = volatility * ema_volume
    
    # Incorporate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    fixed_weight = 0.1  # You can adjust this weight as needed
    combined_alpha_factor = adjusted_volatility + fixed_weight * close_open_diff
    
    # Final Alpha Factor
    final_alpha_factor = combined_alpha_factor * alpha_factor_momentum
    
    return final_alpha_factor
