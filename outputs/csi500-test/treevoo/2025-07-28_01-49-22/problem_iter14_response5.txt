```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['high_low_range_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio
    df['smoothed_high_low_range_ratio'] = df['high_low_range_ratio'].ewm(span=10).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['smoothed_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'].ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    df['intraday_momentum_signal'] = np.where(df['smoothed_high_low_range_ratio'] > 1.0, 1, -1)
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum
    df['close_price_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Apply Volume Filter
    median_volume = df['volume'].rolling(window=20).median()
    df['volume_filtered'] = np.where(df['volume'] > median_volume, 1, 0)
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_alpha_factor'] = (df['intraday_volatility'] * (df['combined_value_momentum'] + df['close_price_momentum'])) / df['volatility']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_alpha_factor'] * df['volume_filtered'] * df['volume']
    
    return df['final_alpha_factor']
```
