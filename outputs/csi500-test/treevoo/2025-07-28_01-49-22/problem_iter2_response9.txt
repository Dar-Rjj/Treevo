```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate n-period High-Low Range Volatility
    df['high_low_volatility'] = df['high_low_range'].rolling(window=n).sum() / n
    
    # Adjust Volatility by Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_volatility'] = df['high_low_volatility'] * df['avg_volume']
    
    # Calculate Previous Day's Intraday Range
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate Intraday Range Momentum
    df['intraday_momentum'] = (df['high_low_range'] / df['prev_high_low_range']) - 1
    
    # Integrate Volatility and Momentum
    df['factor'] = df['adjusted_volatility'] * df['intraday_momentum']
    
    return df['factor']
```
