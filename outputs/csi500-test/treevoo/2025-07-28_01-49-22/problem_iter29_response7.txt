import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_intraday_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility (Average of High-Low over 10 days)
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_momentum = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (
        volume_adjusted_intraday_range + 
        combined_value_momentum + 
        (high_low_range / (df['high'] + df['low'])) - 
        (df['open'] - df['close'].shift(1))
    )
    
    # Introduce Dynamic Threshold Adjustment
    high_low_ratio = high_low_range / (df['high'] + df['low'])
    high_low_10_day_ma = high_low_ratio.rolling(window=10).mean()
    signal = (high_low_ratio > high_low_10_day_ma).astype(int) - (high_low_ratio < high_low_10_day_ma).astype(int)
    
    # Adjust Intermediate Alpha Factor by High-Low Range Ratio
    adjustment_factor = (high_low_ratio > 1.0).astype(float) * 1.5 + (high_low_ratio < 1.0).astype(float) * 0.5
    adjusted_intermediate_alpha = intermediate_alpha * adjustment_factor
    
    # Introduce Volume Momentum
    avg_volume_10_day = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10_day
    volume_adjustment = (volume_ratio > 1).astype(float) * 0.15 - (volume_ratio < 1).astype(float) * 0.15
    final_alpha = (adjusted_intermediate_alpha + close_momentum) * (1 + volume_adjustment)
    
    # Adjust Final Alpha Factor by Average Volume
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    final_alpha = final_alpha * n_period_avg_volume
    
    return final_alpha
