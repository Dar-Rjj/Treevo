```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Ratio
    df['IntradayRangeRatio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Compute Volume Adjusted Intraday Range Ratio
    df['VolumeAdjustedIntradayRange'] = df['IntradayRangeRatio'] * df['volume']
    
    # Calculate Previous Day's Close-to-Open Change
    df['PrevDayCloseToOpenChange'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Smooth High-Low Range with EMA (12-day window)
    df['SmoothedHighLowRange'] = df['HighLowRange'].ewm(span=12).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['VolumeWeightedHighLowRange'] = df['HighLowRange'] * df['volume']
    df['SmoothedVolumeWeightedHighLowRange'] = df['VolumeWeightedHighLowRange'].ewm(span=12).mean()
    
    # Generate Alpha Factor from High-Low Range
    df['AlphaFactorHL'] = df['SmoothedVolumeWeightedHighLowRange'].apply(lambda x: 1 if x > 1.5 else -1)
    
    # Calculate Absolute Close-to-Close Return
    df['AbsCloseToCloseReturn'] = abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['CombinedHighLowAndCloseToClose'] = df['HighLowRange'] - df['AbsCloseToCloseReturn']
    df['SmoothedCombinedHighLowAndCloseToClose'] = df['CombinedHighLowAndCloseToClose'].ewm(span=12).mean()
    
    # Generate Combined Alpha Factor
    df['CombinedAlphaFactor'] = (df['VolumeAdjustedIntradayRange'] - df['PrevDayCloseToOpenChange'])
    dynamic_threshold = df['SmoothedHighLowRange'].ewm(span=12).mean()
    df['CombinedAlphaFactor'] = df['CombinedAlphaFactor'].apply(
        lambda x, y: x * 1.6 if y > dynamic_threshold else x * 0.4, args=(dynamic_threshold,)
    )
    
    # Calculate High-Low Range Volatility
    df['HighLowRangeVolatility'] = df['HighLowRange'].rolling(window=15).sum() / 15
    
    # Smooth High-Low Range Volatility
    df['SmoothedHighLowRangeVolatility'] = df['HighLowRangeVolatility'].ewm(span=15).mean()
    
    # Incorporate Close-Open Difference
    df['CloseOpenDifference'] = df['close'] - df['open']
    df['CombinedVolatility'] = df['SmoothedHighLowRangeVolatility'] + 0.1 * df['CloseOpenDifference']
    
    # Calculate Combined Value Momentum
    df['CombinedValueMomentum'] = df['SmoothedCombinedHighLowAndCloseToClose'] - df['SmoothedCombinedHighLowAndCloseToClose'].shift(1)
    
    # Calculate Close Price Momentum
    df['ClosePriceMomentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=25).sum()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['CombinedAlphaFactor'] * (df['CombinedValueMomentum'] + df['ClosePriceMomentum']) * df['volume']
    
    return df['FinalAlphaFactor']
```
