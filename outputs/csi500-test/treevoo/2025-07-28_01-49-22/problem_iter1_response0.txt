```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Momentum
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    df['momentum'] = df['close'] - df['10_day_SMA']
    
    # Sum of daily ranges over the 10-day window
    df['sum_daily_range_10'] = df['daily_range'].rolling(window=10).sum()
    
    # Divide sum by 10 to find average daily range
    df['avg_daily_range'] = df['sum_daily_range_10'] / 10
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['momentum'] / df['avg_daily_range']
    
    return df['adjusted_momentum']
```
