```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Ratio
    df['hl_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Smooth High-Low Range Ratio using EMA (10 days)
    df['ema_hl_ratio'] = df['hl_ratio'].ewm(span=10).mean()
    
    # Generate Alpha Factor from Momentum
    dynamic_threshold = df['ema_hl_ratio'].mean()
    df['alpha_factor'] = np.where(df['ema_hl_ratio'] > dynamic_threshold, 1, -1) * df['volume']
    
    # Adjust Alpha Factor by Close-Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['alpha_factor'] *= df['close_change']
    
    # Calculate High-Low Range Volatility (n-period sum divided by n)
    n = 5
    df['hl_volatility'] = df['high_low_range'].rolling(window=n).sum() / n
    
    # Smooth High-Low Range Volatility using EMA (15 days)
    df['ema_hl_volatility'] = df['hl_volatility'].ewm(span=15).mean()
    
    # Incorporate Close-Open Difference
    fixed_weight = 0.5
    df['close_open_diff'] = df['close'] - df['open']
    df['combined_volatility'] = df['ema_hl_volatility'] + fixed_weight * df['close_open_diff']
    
    # Enhance Alpha Factor with Volume-Adjusted High-Low Range
    avg_volume = df['volume'].rolling(window=n).mean()
    df['vol_adj_hl_range'] = df['high_low_range'] * avg_volume
    df['alpha_factor'] += df['vol_adj_hl_range']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_return'] = np.abs(df['close'] - df['close'].shift(1))
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_return']
    
    # Calculate Intraday Volatility (Average of (High - Low) over 5 days)
    df['intraday_volatility'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate Combined Value Momentum
    df['combined_momentum'] = df['combined_value'] - df['combined_value'].shift(1)
    
    # Calculate Close Price Momentum (Sum of N-day Returns over the last 30 days)
    df['close_momentum'] = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=30).sum()
    
    # Apply Volume Filter (Exclude days below median volume)
    median_volume = df['volume'].rolling(window=30).median()
    df['alpha_factor'] = np.where(df['volume'] >= median_volume, df['alpha_factor'], 0)
    
    # Adjust for Volatility (Calculate Standard Deviation with 30-day Window)
    df['volatility'] = df['close'].rolling(window=30).std()
    df['alpha_factor'] /= df['volatility']
    
    # Calculate Open-to-Close Return
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Open-to-Close Momentum (Sum of N-day Open-to-Close Returns over the last 30 days)
    df['open_close_momentum'] = df['open_close_return'].rolling(window=30).sum()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['intraday_volatility'] * 
                                (df['combined_momentum'] + df['close_momentum'] + df['open_close_momentum']) * 
                                df['volume'])
    
    return df['final_alpha_factor']
```
