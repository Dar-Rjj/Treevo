```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Smooth High-Low Range using EMA with 10-day window
    smoothed_high_low = high_low_range.ewm(span=10).mean()
    
    # Generate Intraday Momentum Signal
    intraday_momentum_signal = (smoothed_high_low > 1.0).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low = (high_low_range * df['volume']).ewm(span=10).mean()
    
    # Generate Alpha Factor and compare to baseline
    alpha_factor = (volume_weighted_high_low > 1.0).astype(int) * 2 - 1
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=20).std()
    adjusted_alpha_factor = alpha_factor / volatility
    
    # Calculate Absolute Close-to-Close Return
    close_to_close_return = df['close'].diff().abs()
    
    # Combine High-Low and Close-to-Close Returns
    combined_value = smoothed_high_low - close_to_close_return
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (volume_weighted_high_low +
                                 combined_value +
                                 intraday_range_ratio -
                                 (df['open'] - df['close'].shift(1)))
    
    # Adjust Intermediate Alpha Factor based on signal
    intermediate_alpha_factor = np.where(alpha_factor > 0, intermediate_alpha_factor * 1.5, intermediate_alpha_factor * 0.5)
    
    # Generate Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_factor + close_to_close_return) * volatility
    
    # Weight by Volume
    volume_weighted_final_alpha_factor = (final_alpha_factor * df['volume']).ewm(span=5).mean()
    
    return volume_weighted_final_alpha_factor
```
