```python
def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by date, representing the factor values.
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of close to open price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the difference between close and average of high and low prices
    diff_close_avg = df['close'] - avg_high_low
    
    # Calculate the ratio of volume to amount
    vol_amount_ratio = df['volume'] / df['amount']
    
    # Combine the features into a single factor
    factor = (close_change * close_open_ratio + diff_close_avg) * vol_amount_ratio
    
    return factor
```
