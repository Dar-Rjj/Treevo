```python
import pandas as pd

def heuristics(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    abs_close_return = (df['close'] - df['close'].shift(1)).abs()
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    combined_value = high_low_range - abs_close_return
    
    # Calculate Volume Adjusted Intraday Range
    volume_adjusted_range = high_low_range * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=10).mean()
    
    # Calculate Combined Value Momentum
    combined_value_momentum = combined_value - combined_value.shift(1)
    
    # Calculate Close Price Momentum
    close_momentum = ((df['close'] / df['close'].shift(1)) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    intraday_range_ratio = high_low_range / (df['high'] + df['low'])
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (volume_adjusted_range 
                          + combined_value_momentum 
                          + intraday_range_ratio 
                          - (df['open'] - df['close'].shift(1)))
    
    # Compare with Baseline (Fixed Threshold, e.g., 1.0)
    positive_signal = (intraday_range_ratio > 1.0)
    negative_signal = (intraday_range_ratio < 1.0)
    
    # Adjust Intermediate Alpha Factor
    adjusted_intermediate_alpha = intermediate_alpha.copy()
    adjusted_intermediate_alpha[positive_signal] *= 1.5
    adjusted_intermediate_alpha[negative_signal] *= 0.5
    
    # Generate Final Alpha Factor
    final_alpha = intraday_volatility * (adjusted_intermediate_alpha + close_momentum)
    
    # Adjust Final Alpha Factor by Volume
    daily_high_low_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=10).mean()
    final_alpha = final_alpha * (daily_high_low_range * avg_volume)
    
    return final_alpha
```
