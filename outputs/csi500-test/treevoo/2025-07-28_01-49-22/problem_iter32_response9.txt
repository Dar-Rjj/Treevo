```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Absolute Close-to-Close Return
    df['abs_close_to_close'] = abs(df['close'].diff())
    
    # Combine High-Low Range and Absolute Close-to-Close Return
    df['combined_value'] = df['high_low_range'] - df['abs_close_to_close']
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Combined Value Momentum
    df['combined_value_momentum'] = df['combined_value'].diff()
    
    # Calculate Close Price Momentum (Sum of ((Close at day t / Close at day t-1) - 1) over the last 20 days)
    df['close_price_momentum'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).sum()
    
    # Calculate Intraday Range Ratio
    df['intraday_range_ratio'] = df['high_low_range'] / (df['high'] + df['low'])
    
    # Compute Previous Day's Close-to-Open Change
    df['close_to_open_change'] = df['open'] - df['close'].shift(1)
    
    # Smooth High-Low Range Volatility with EMA (15 days)
    df['smoothed_high_low_range_volatility'] = df['high_low_range'].ewm(span=15, adjust=False).mean()
    
    # Incorporate Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    df['weighted_close_open_diff'] = 0.5 * df['close_open_diff']  # Fixed weight for close-open difference
    df['adjusted_smoothed_high_low_range_volatility'] = df['smoothed_high_low_range_volatility'] + df['weighted_close_open_diff']
    
    # Generate Intermediate Alpha Factor
    df['intermediate_alpha_factor'] = (df['volume_weighted_intraday_range'] +
                                       df['combined_value_momentum'] +
                                       df['intraday_range_ratio'] -
                                       df['close_to_open_change'])
    
    # Adjust Intermediate Alpha Factor based on intraday range ratio
    df['intermediate_alpha_factor'] = np.where(df['intraday_range_ratio'] > 1.0,
                                               df['intermediate_alpha_factor'] * 1.8,
                                               df['intermediate_alpha_factor'] * 0.4)
    
    # Compare with Baseline (Dynamic Threshold, e.g., EMA of Smoothed High-Low Range Volatility)
    dynamic_threshold = df['smoothed_high_low_range_volatility'].ewm(span=15, adjust=False).mean()
    df['signal'] = np.where(df['smoothed_high_low_range_volatility'] > dynamic_threshold, 1, -1)
    
    # Adjust Combined Alpha Factor based on signal
    df['adjusted_intermediate_alpha_factor'] = np.where(df['signal'] == 1,
                                                        df['intermediate_alpha_factor'] * 1.7,
                                                        df['intermediate_alpha_factor'] * 0.3)
    
    # Adjust for Volatility (Standard Deviation of Close Price with 20-day Window)
    df['volatility'] = df['close'].rolling(window=20).std()
    df['alpha_factor_adjusted_for_volatility'] = df['adjusted_intermediate_alpha_factor'] / df['volatility']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['alpha_factor_adjusted_for_volatility'] + df['close_price_momentum']) * df['volume']
    
    # Weight by Volume (Apply EMA to Volume-Weighted Final Alpha Factor with 5-day Window)
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
```
