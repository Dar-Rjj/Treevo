```python
import pandas as pd

def heuristics_v2(df, N=10, M=10):
    # Calculate the daily return based on close prices
    df['daily_return'] = df['close'].diff()
    
    # Calculate the average daily trading volume over the past N days
    df['avg_volume'] = df['volume'].rolling(window=N).mean()
    
    # Compare today's volume with the average volume to identify unusual trading activity
    df['volume_change'] = df['volume'] - df['avg_volume']
    
    # Calculate the daily price range (high - low)
    df['price_range'] = df['high'] - df['low']
    
    # Measure the volatility by analyzing the standard deviation of daily price ranges over the past M days
    df['volatility'] = df['price_range'].rolling(window=M).std()
    
    # Compute the difference between today's open price and yesterday's close price
    df['open_close_diff'] = df['open'] - df['close'].shift(1)
    
    # Create a weighted sum of the daily return, normalized volume change, and daily price range
    df['composite_factor'] = (df['daily_return'] + 
                              0.5 * df['volume_change'] / df['avg_volume'] +
                              0.3 * df['price_range'] / df['volatility'] +
                              0.2 * df['open_close_diff'])
    
    # Return the composite factor as a pandas Series
    return df['composite_factor']
```
