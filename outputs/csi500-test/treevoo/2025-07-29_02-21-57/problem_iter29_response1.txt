```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    df['AvgDailyRange'] = df['DailyRange'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1))
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1))
    
    # Calculate Breakout Strength
    df['BreakoutStrength'] = 0.0
    df.loc[df['UpwardBreakout'], 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['AvgDailyRange']
    df.loc[df['DownwardBreakout'], 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['AvgDailyRange']
    
    # Cumulate Breakout Strength over the past 10 days
    df['PositiveBS'] = df['BreakoutStrength'].where(df['BreakoutStrength'] > 0, 0).rolling(window=10).sum()
    df['NegativeBS'] = df['BreakoutStrength'].where(df['BreakoutStrength'] < 0, 0).rolling(window=10).sum()
    df['CumulatedBS'] = df['PositiveBS'] - (-df['NegativeBS'])
    
    # Calculate Intraday High-Low Spread
    df['IntradayHLSpread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 10-Day Volume Adjusted Momentum
    df['10DayClosePriceReturn'] = (df['close'] - df['close'].shift(9)) / df['close'].shift(9)
    df['10DayVolumeRatio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    df['VolumeAdjustedMomentum'] = df['10DayClosePriceReturn'] * df['10DayVolumeRatio']
    
    # Adjust Momentum by ATR
    df['WeightedMomentum'] = (df['close'] - df['close'].shift(14)) * df['volume']
    df['ATRAdjustedMomentum'] = df['WeightedMomentum'] / df['ATR']
    
    # Generate Alpha Factor
    df['AlphaFactor'] = (df['IntradayHLSpread'] + df['CloseToOpenReturn'] + df['CumulatedBS'] + 
                         df['VolumeAdjustedMomentum'] + df['ATRAdjustedMomentum'])
    
    # Incorporate Volume Trend
    df['10DayMAVolume'] = df['volume'].rolling(window=10).mean()
    df['VolumeTrendAdjustment'] = 1.0
    df.loc[df['volume'] > df['10DayMAVolume'], 'VolumeTrendAdjustment'] = 1.5
    df.loc[df['volume'] < df['10DayMAVolume'], 'VolumeTrendAdjustment'] = 0.5
    
    df['AlphaFactor'] *= df['VolumeTrendAdjustment']
    
    return df['AlphaFactor']
```
