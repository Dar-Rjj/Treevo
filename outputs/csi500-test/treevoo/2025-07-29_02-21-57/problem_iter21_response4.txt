```python
def heuristics_v2(df):
    # Calculate the difference between the high and low price, capturing intraday volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate the percentage change in closing price from the previous day
    price_momentum = df['close'].pct_change()
    
    # Calculate the ratio of closing price to the average of high and low prices
    price_ratio = df['close'] / ((df['high'] + df['low']) / 2)
    
    # Calculate the amount per volume as a liquidity indicator
    liquidity = df['amount'] / df['volume']
    
    # Combine all the factors into one alpha factor
    alpha_factor = (intraday_volatility * price_momentum * price_ratio) / liquidity
    
    return alpha_factor
```
