import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the given thought tree.
    The alpha factor is a combination of price and volume features, capturing
    the momentum and volume spread in the market. The idea is to identify stocks
    that are experiencing strong price movements along with high trading volumes,
    which could indicate a sustained trend.

    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, amount, volume.

    Returns:
    pd.Series: A Series representing the computed alpha factor indexed by date.
    """

    # Calculate the price change from the previous day's close to the current day's close
    df['price_change'] = df['close'].pct_change()

    # Calculate the volume change from the previous day's volume to the current day's volume
    df['volume_change'] = df['volume'].pct_change()

    # Compute the alpha factor as a combination of price and volume changes
    df['alpha_factor'] = df['price_change'] * df['volume_change']

    # Return the alpha factor as a Series
    return df['alpha_factor']
