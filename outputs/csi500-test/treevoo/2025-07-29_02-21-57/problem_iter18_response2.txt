```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    
    # Calculate 10-Day Volume Adjusted Momentum
    df['10DayCloseReturn'] = df['close'] / df['close'].shift(10) - 1
    df['10DayVolumeRatio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    df['10DayVolAdjMomentum'] = df['10DayCloseReturn'] * df['10DayVolumeRatio']
    
    # Weight by Volume
    df['WeightedMomentum'] = df['volume'] * df['10DayVolAdjMomentum']
    
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20DayAvgRange'] = df['DailyRange'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['BreakoutStrength'] = 0
    df.loc[df['UpwardBreakout'] == 1, 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgRange']
    df.loc[df['DownwardBreakout'] == 1, 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgRange']
    
    # Cumulate Breakout Strength
    df['CumulatedPosBS'] = df['BreakoutStrength'].clip(lower=0).rolling(window=15).sum()
    df['CumulatedNegBS'] = df['BreakoutStrength'].clip(upper=0).rolling(window=15).sum()
    df['CumulatedBreakoutStrength'] = df['CumulatedPosBS'] - df['CumulatedNegBS']
    
    # Calculate Intraday High-Low Spread
    df['IntradayHLSpread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['OpenToCloseReturn'] = df['close'] / df['open'] - 1
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Combine Indicators
    df['CombinedIndicators'] = (df['ATR'] + 
                                df['10DayVolAdjMomentum'] + 
                                df['IntradayHLSpread'] + 
                                df['OpenToCloseReturn'] + 
                                df['VWAP'] + 
                                df['CumulatedBreakoutStrength']) / 6
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = df['WeightedMomentum'] / df['ATR']
    
    # Generate Final Alpha Factor
    df['FinalAlphaFactor'] = df['ATRAdjustedMomentum'] + df['CombinedIndicators'] + df['VWAP']
    
    return df['FinalAlphaFactor']
```
