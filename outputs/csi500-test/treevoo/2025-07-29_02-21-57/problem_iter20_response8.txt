```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the difference between high and low, then divide by the average volume
    hl_diff_over_volume = (df['high'] - df['low']) / df['volume'].mean()
    
    # Calculate the ratio of amount to volume
    amount_to_volume = df['amount'] / df['volume']
    
    # Generate the heuristic factor using a weighted sum
    factor = 0.4 * close_to_open + 0.3 * hl_diff_over_volume + 0.3 * amount_to_volume
    
    return factor
```
