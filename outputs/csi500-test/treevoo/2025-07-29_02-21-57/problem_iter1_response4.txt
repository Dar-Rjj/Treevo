```python
import pandas as pd

def heuristics_v2(df):
    # Get High, Low, and Previous Close prices
    high = df['high']
    low = df['low']
    close = df['close']
    prev_close = close.shift(1).fillna(close)

    # Calculate True Range
    tr1 = high - low
    tr2 = (high - prev_close).abs()
    tr3 = (low - prev_close).abs()
    true_range = pd.DataFrame({'tr1': tr1, 'tr2': tr2, 'tr3': tr3}).max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()

    # Calculate Price Momentum
    momentum = close - close.shift(14)

    # Weight by Volume
    volume_weighted_momentum = momentum * df['volume']

    # Normalize by ATR
    normalized_momentum = volume_weighted_momentum / atr

    # Generate Alpha Factor
    alpha_factor = atr + normalized_momentum

    return alpha_factor
```
