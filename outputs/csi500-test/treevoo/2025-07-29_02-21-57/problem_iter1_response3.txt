import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Smooth High-Low Difference using EMA with span 10
    ema_high_low_diff = high_low_diff.ewm(span=10).mean()
    
    # Subtract EMA of High-Low Difference from Previous EMA
    diff_ema_high_low_diff = ema_high_low_diff.diff()
    
    # Calculate Close-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Integrate with Close-Open Return
    factor_value = close_open_return + (diff_ema_high_low_diff * df['volume'])
    
    return factor_value
