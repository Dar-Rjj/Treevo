import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and High-Low Spread
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    
    # Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] - df['open']
    
    # Short-Term Volume Weighted Moving Average (VWMA) over 5 days
    df['ShortTermVWMA'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Long-Term Volume Weighted Moving Average (VWMA) over 20 days
    df['LongTermVWMA'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # 5-Day Volume Adjusted Momentum
    df['5DayClosePriceReturn'] = df['close'].pct_change(periods=5)
    df['5DayVolumeRatio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['VolumeAdjustedMomentum'] = df['5DayClosePriceReturn'] * df['5DayVolumeRatio']
    
    # 10-Day Price Oscillation
    df['10DayHighPrice'] = df['high'].rolling(window=10).max()
    df['10DayLowPrice'] = df['low'].rolling(window=10).min()
    df['10DayPriceRange'] = df['10DayHighPrice'] - df['10DayLowPrice']
    
    # 10-Day Volume-Adjusted Range
    df['10DayVolumeSum'] = df['volume'].rolling(window=10).sum()
    df['VolumeAdjustedRange'] = df['10DayPriceRange'] * df['10DayVolumeSum']
    
    # Combined Crossover, Momentum, and Enhanced Price Oscillation Signal
    df['CrossoverSignal'] = (df['ShortTermVWMA'] > df['LongTermVWMA']).astype(int) - (df['ShortTermVWMA'] < df['LongTermVWMA']).astype(int)
    df['MomentumSignal'] = (df['VolumeAdjustedMomentum'] > 0).astype(int) - (df['VolumeAdjustedMomentum'] < 0).astype(int)
    df['OscillationSignal'] = (df['10DayPriceRange'] > df['10DayPriceRange'].mean()).astype(int) - (df['10DayPriceRange'] < df['10DayPriceRange'].mean()).astype(int)
    df['VolumeAdjustedRangeSignal'] = (df['VolumeAdjustedRange'] > df['VolumeAdjustedRange'].mean()).astype(int) - (df['VolumeAdjustedRange'] < df['VolumeAdjustedRange'].mean()).astype(int)
    
    # Combine Signals
    df['CombinedSignal'] = df['CrossoverSignal'] + df['MomentumSignal'] + df['OscillationSignal'] + df['VolumeAdjustedRangeSignal']
    
    # Weight by Volume
    df['WeightedMomentum'] = df['VolumeAdjustedMomentum'] * df['volume']
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = df['WeightedMomentum'] / df['ATR']
    
    # Generate Alpha Factor
    df['AlphaFactor'] = df['CloseToOpenReturn'] + df['ATRAdjustedMomentum']
    df['AverageHighLowSpread'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['AlphaFactor'] += (df['high'] - df['low']) / df['AverageHighLowSpread']
    
    # 21-Day Exponential Moving Average (EMA)
    df['21DayEMA'] = df['close'].ewm(span=21, adjust=False).mean()
    df['5DayEMA'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # EMA-based Momentum
    df['EMAMomentum'] = df['5DayEMA'] - df['21DayEMA']
    
    # Integrate EMA-based Momentum into Alpha Factor
    df['AlphaFactor']
