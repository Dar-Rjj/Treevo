```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5 days) and Long-Term (20 days) Volume Weighted Moving Average (VWMA)
    def calculate_vwma(price, volume, window):
        return (price * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()

    short_vwma = calculate_vwma(df['close'], df['volume'], 5)
    long_vwma = calculate_vwma(df['close'], df['volume'], 20)

    # Calculate 5-Day Volume Adjusted Momentum
    five_day_return = df['close'].pct_change(periods=5)
    average_5d_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'].rolling(window=5).sum() / average_5d_volume
    momentum = five_day_return * volume_ratio

    # Calculate Intraday Return Series
    intraday_return = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages of Intraday Returns
    short_intraday_ma = intraday_return.rolling(window=5).mean()
    long_intraday_ma = intraday_return.rolling(window=20).mean()

    # Compute Intraday Return Difference
    intraday_diff = short_intraday_ma - long_intraday_ma

    # Filter by Volume Change
    volume_change = df['volume'].diff()
    filtered_intraday_diff = intraday_diff.where(volume_change > 0, 0)

    # Calculate High-Low Price Oscillation
    high_low_range = df['high'] - df['low']
    std_high_low = high_low_range.rolling(window=5).std()
    mean_high_low = high_low_range.rolling(window=5).mean()
    price_oscillation = (high_low_range - mean_high_low) / std_high_low

    # Combined Crossover and Momentum Signal
    vwma_signal = (short_vwma > long_vwma).astype(int) - (short_vwma < long_vwma).astype(int)
    momentum_signal = (momentum > 0).astype(int) - (momentum < 0).astype(int)
    intraday_sentiment_signal = (filtered_intraday_diff > 0).astype(int) - (filtered_intraday_diff < 0).astype(int)

    # Combine Signals
    combined_signal = vwma_signal + momentum_signal + intraday_sentiment_signal
    final_signal = combined_signal.map({
        3: 'Strong Buy',
        2: 'Moderate Buy',
        1: 'Hold (Buy)',
        -1: 'Hold (Sell)',
        -2: 'Moderate Sell',
        -3: 'Strong Sell'
    })

    # Convert to numerical values for ease of use in further analysis
    final_signal = final_signal.map({
        'Strong Buy': 1,
        'Moderate Buy': 0.5,
        'Hold (Buy)': 0,
        'Hold (Sell)': 0,
        'Moderate Sell': -0.5,
        'Strong Sell': -1
    })

    return final_signal
```
