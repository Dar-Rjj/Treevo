    # Calculate Short-Term (5-day) and Long-Term (20-day) Volume Weighted Moving Average (VWMA)
    data['VWMA_5'] = (data['close'] * data['volume']).rolling(window=5).sum() / data['volume'].rolling(window=5).sum()
    data['VWMA_20'] = (data['close'] * data['volume']).rolling(window=20).sum() / data['volume'].rolling(window=20).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    data['5D_Return'] = data['close'].pct_change(periods=5)
    avg_5D_volume = data['volume'].rolling(window=5).mean()
    data['5D_Volume_Ratio'] = (data['volume'].rolling(window=5).sum() / avg_5D_volume)
    data['Volume_Adjusted_Momentum'] = data['5D_Return'] * data['5D_Volume_Ratio']

    # Combined Crossover and Momentum Signal
    data['Crossover_Signal'] = 0
    data.loc[data['VWMA_5'] > data['VWMA_20'], 'Crossover_Signal'] = 1
    data.loc[data['VWMA_5'] < data['VWMA_20'], 'Crossover_Signal'] = -1
    
    data['Momentum_Signal'] = 0
    data.loc[data['Volume_Adjusted_Momentum'] > 0, 'Momentum_Signal'] = 1
    data.loc[data['Volume_Adjusted_Momentum'] < 0, 'Momentum_Signal'] = -1

    # Combine Signals
    data['Combined_Signal'] = 0
    data.loc[(data['Crossover_Signal'] == 1) & (data['Momentum_Signal'] == 1), 'Combined_Signal'] = 2
    data.loc[(data['Crossover_Signal'] == -1) & (data['Momentum_Signal'] == -1), 'Combined_Signal'] = -2
    data.loc[(data['Crossover_Signal'] == 1) & (data['Momentum_Signal'] == 0), 'Combined_Signal'] = 1

    data['TR1'] = data['high'] - data['low']

    data['TR3'] = abs(data['low'] - data['close'].shift(1))
