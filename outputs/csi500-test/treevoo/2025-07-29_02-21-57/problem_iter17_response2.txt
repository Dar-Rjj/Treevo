import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of close prices over the last 10 days
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the momentum of close prices over the last 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the volatility (standard deviation) of close prices over the last 10 days
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Calculate the volume-weighted close price
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine the factors to create the alpha factor
    df['alpha_factor'] = (df['close_10d_mean'] - df['close']) / df['volatility_10d'] + df['momentum_5d'] * df['vwap']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
