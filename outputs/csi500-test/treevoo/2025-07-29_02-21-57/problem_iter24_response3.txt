```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Price Movement Analysis
    data['Day_Range'] = data['close'] - data['open']
    data['Pct_Change'] = (data['close'] - data['open']) / data['open']

    # Volatility Measure
    data['High_Low'] = data['high'] - data['low']
    data['High_PrevClose'] = abs(data['high'] - data['close'].shift(1))
    data['Low_PrevClose'] = abs(data['low'] - data['close'].shift(1))
    data['True_Range'] = data[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    data['ATR_10'] = data['True_Range'].rolling(window=10).mean()

    # Volume Trend
    data['Vol_Sum_10'] = data['volume'].rolling(window=10).sum()
    data['Vol_ROC'] = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)

    # Volume-Price Relationship
    data['Bearish_Divergence'] = ((data['close'] < data['close'].shift(1)) & (data['volume'] > data['volume'].shift(1))).astype(int)
    data['Bullish_Divergence'] = ((data['close'] > data['close'].shift(1)) & (data['volume'] < data['volume'].shift(1))).astype(int)

    # Momentum Indicators
    data['SMA_5'] = data['close'].rolling(window=5).mean()
    data['SMA_20'] = data['close'].rolling(window=20).mean()
    data['SMA_Crossover'] = (data['SMA_5'] > data['SMA_20']).astype(int) - (data['SMA_5'] < data['SMA_20']).astype(int)
    data['Price_Oscillator'] = data['SMA_5'] - data['SMA_20']

    # Advanced Alpha Concepts
    data['Typical_Price'] = (data['high'] + data['low'] + data['close']) / 3
    data['Money_Flow'] = data['Typical_Price'] * data['volume']
    data['Positive_Money_Flow'] = data['Money_Flow'].where(data['Typical_Price'] > data['Typical_Price'].shift(1), 0)
    data['Negative_Money_Flow'] = data['Money_Flow'].where(data['Typical_Price'] < data['Typical_Price'].shift(1), 0)
    data['MFI_10'] = 100 - (100 / (1 + data['Positive_Money_Flow'].rolling(window=10).sum() / data['Negative_Money_Flow'].rolling(window=10).sum()))

    data['Multiplier'] = ((data['close'] - data['low']) - (data['high'] - data['close'])) / (data['high'] - data['low'])
    data['A_D_Line'] = (data['Multiplier'] * data['volume']).cumsum()

    # Market Sentiment
    data['Close_High_Ratio'] = data['close'] / data['high']
    data['Close_Low_Ratio'] = data['close'] / data['low']
    data['Opening_Gap'] = data['open'] - data['close'].shift(1)

    # Combine all factors into a single alpha factor
    data['Alpha_Factor'] = (
        data['Day_Range'] +
        data['Pct_Change'] +
        data['ATR_10'] +
        data['Vol_Sum_10'] +
        data['Vol_ROC'] +
        data['Bearish_Divergence'] +
        data['Bullish_Divergence'] +
        data['SMA_Crossover'] +
        data['Price_Oscillator'] +
        data['MFI_10'] +
        data['A_D_Line'] +
        data['Close_High_Ratio'] +
        data['Close_Low_Ratio'] +
        data['Opening_Gap']
    )

    return data['Alpha_Factor']
```
