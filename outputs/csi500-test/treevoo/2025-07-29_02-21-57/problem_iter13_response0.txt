def heuristics_v2(df):
    # Calculate True Range
    df['Previous_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Previous_Close'])
    df['TR3'] = abs(df['low'] - df['Previous_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Volume_Weighted_Momentum'] = df['Momentum'] * df['volume']

    # Calculate Volume-Weighted Average Price (VWAP)
    price_sum = (df['open'] + df['high'] + df['low'] + df['close']) * df['volume']
    total_volume = df['volume']
    df['VWAP'] = price_sum / total_volume

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Volume_Weighted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['VWAP']

    return df['Alpha_Factor']
