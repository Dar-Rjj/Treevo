```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5D_Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    df['Short_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    df['Long_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Compare Short-Term VWMA and Long-Term VWMA
    df['VWMA_Crossover'] = np.where(df['Short_Term_VWMA'] > df['Long_Term_VWMA'], 1, 
                                    np.where(df['Short_Term_VWMA'] < df['Long_Term_VWMA'], -1, 0))

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5D_High_Low_Std'] = df['High_Low_Range'].rolling(window=5).std()
    df['5D_High_Low_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['High_Low_Oscillation'] = df['High_Low_Range'] - df['5D_High_Low_Mean']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['14D_ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Normalize by ATR
    df['Normalized_Momentum'] = df['Weighted_Momentum'] / df['14D_ATR']

    # Adjust Signal with Price Momentum
    df['Adjusted_Signal'] = np.where((df['Price_Momentum'] > 0) & (df['VWMA_Crossover'] == 1), 2, 
                                     np.where((df['Price_Momentum'] < 0) & (df['VWMA_Crossover'] == -1), -2, 
                                              np.where((df['Price_Momentum'] > 0) & (df['VWMA_Crossover'] == 0), 1, 
                                                       np.where((df['Price_Momentum'] < 0) & (df['VWMA_Crossover'] == 0), -1, 0))))

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['Normalized_Momentum'] + df['5D_Volume_Adjusted_Momentum'] + df['High_Low_Oscillation'] + df['VWMA_Crossover']

    return df['Alpha_Factor']
```
