```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Series
    intraday_return = df['close'] / df['open'] - 1
    
    # Compute Moving Averages
    short_term_ma = intraday_return.rolling(window=5).mean()
    long_term_ma = intraday_return.rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    intraday_return_diff = long_term_ma - short_term_ma
    
    # Apply Volume Filter
    intraday_return_diff[df['volume'] <= df['volume'].shift(1)] = 0
    
    return intraday_return_diff
```
