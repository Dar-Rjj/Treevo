```python
import pandas as pd

def heuristics_v2(df):
    # Simple Moving Averages
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Exponential Moving Averages
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # True Range and Average True Range
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()

    # New Highs or Lows
    df['New_High'] = (df['high'] > df['high'].rolling(window=5).max()).astype(int)
    df['New_Low'] = (df['low'] < df['low'].rolling(window=5).min()).astype(int)

    # Rate of Change (ROC)
    df['ROC_12'] = (df['close'] / df['close'].shift(12)) - 1

    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + RS))

    # On-Balance Volume (OBV)
    df['OBV'] = (df['volume'] * (df['close'] > df['close'].shift(1))).cumsum() - (df['volume'] * (df['close'] < df['close'].shift(1))).cumsum()

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).abs().rolling(window=14).sum()
    money_ratio = positive_flow / negative_flow
    df['MFI_14'] = 100 - (100 / (1 + money_ratio))

    # Accumulation/Distribution Line (ADL)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (money_flow_multiplier * df['volume']).cumsum()
    df['ADL'] = adl

    # Combine all factors into a single alpha factor
    alpha_factor = (df['SMA_10'] - df['SMA_20']) + (df['EMA_10'] - df['EMA_20']) + df['ATR_14'] + df['New_High'] - df['New_Low'] + df['ROC_12'] + df['RSI_14'] + df['OBV'] + df['MFI_14'] + df['ADL']

    return alpha_factor
```
