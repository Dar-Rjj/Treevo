```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_Day_Close_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5_Day_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5_Day_Volume_Adjusted_Momentum'] = df['5_Day_Close_Return'] * df['5_Day_Volume_Ratio']
    
    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5_Day_High_Low_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['High_Low_Price_Oscillation'] = df['High_Low_Range'] - df['5_Day_High_Low_Mean']
    
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['14_Period_ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Price and Range Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    df['Daily_Range'] = df['high'] - df['low']
    df['Combined_Momentum'] = df['Price_Momentum'] + df['Daily_Range']
    
    # Apply Exponential Moving Average (EMA) Smoothing
    alpha = 0.3
    df['EMA_Smoothed_Combined_Momentum'] = df['Combined_Momentum'].ewm(alpha=alpha, adjust=False).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = df['close'] > df['high'].shift(1)
    df['Downward_Breakout'] = df['close'] < df['low'].shift(1)
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = 0
    df.loc[df['Upward_Breakout'], 'Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['Daily_Range'].rolling(window=20).mean()
    df.loc[df['Downward_Breakout'], 'Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['Daily_Range'].rolling(window=20).mean()
    
    # Cumulate Breakout Strength
    df['Cumulative_Breakout_Strength'] = df['Breakout_Strength'].rolling(window=10).apply(lambda x: x[x > 0].sum() - x[x < 0].sum(), raw=True)
    
    # Weight by Volume
    df['Volume_Adjusted_Combined_Momentum'] = df['Combined_Momentum'] * df['volume']
    
    # Combine Volume Adjusted Momentum, ATR, and EMA Smoothed Combined Momentum
    df['Combined_Factor'] = df['5_Day_Volume_Adjusted_Momentum'] + df['14_Period_ATR'] + df['EMA_Smoothed_Combined_Momentum']
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = (df['Combined_Momentum'] + df['High_Low_Price_Oscillation']) * df['Cumulative_Breakout_Strength']
    
    return df['Alpha_Factor']
```
