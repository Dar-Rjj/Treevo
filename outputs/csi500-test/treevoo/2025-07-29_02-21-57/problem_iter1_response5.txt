```python
import pandas as pd

def heuristics_v2(df, short_window=10, long_window=30):
    # Calculate the Short-Term Volume Weighted Moving Average (VWMA)
    short_vwma = (df['close'] * df['volume']).rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()

    # Calculate the Long-Term Volume Weighted Moving Average (VWMA)
    long_vwma = (df['close'] * df['volume']).rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()

    # Generate the Crossover Signal
    crossover_signal = (short_vwma > long_vwma).astype(int) - (short_vwma < long_vwma).astype(int)

    return crossover_signal
```
