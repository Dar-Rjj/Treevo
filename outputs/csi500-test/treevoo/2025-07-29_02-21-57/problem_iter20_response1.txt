```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff(1)
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].diff(1)
    
    # Calculate the amount change from the previous day
    amount_change = df['amount'].diff(1)
    
    # Heuristic: If the price change is positive and the volume increase is greater than the amount increase,
    # it could indicate a bullish trend where the stock is being accumulated.
    alpha_factor = (price_change > 0) & (volume_change > amount_change)
    
    # Convert boolean to integer for easier interpretation
    alpha_factor = alpha_factor.astype(int)
    
    return alpha_factor
```
