```python
def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Averages
    short_term_ma = df['intraday_return'].rolling(window=5).mean()
    long_term_ma = df['intraday_return'].rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    df['intraday_return_diff'] = long_term_ma - short_term_ma
    
    # Compute Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Apply Condition
    df['factor'] = df.apply(lambda row: row['intraday_return_diff'] if row['volume_change'] > 0 else 0, axis=1)
    
    # Return the factor as a pandas Series
    return df['factor']
```
