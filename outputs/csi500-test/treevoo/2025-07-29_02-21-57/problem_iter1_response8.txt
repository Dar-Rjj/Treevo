    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    average_daily_range = daily_range.rolling(window=20).mean()
    
    # Identify Breakouts
    upward_breakout = (df['close'] > df['high'].shift(1)).astype(int)
    downward_breakout = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    breakout_strength = pd.Series(index=df.index, dtype=float)
    breakout_strength[upward_breakout == 1] = (df['close'] - df['high'].shift(1)) / average_daily_range
    breakout_strength[downward_breakout == 1] = (df['low'].shift(1) - df['close']) / average_daily_range
    
    # Cumulate Breakout Strength over the past 10 days
    positive_strength = breakout_strength.where(breakout_strength > 0, 0).rolling(window=10).sum()
    negative_strength = breakout_strength.where(breakout_strength < 0, 0).rolling(window=10).sum()
    cumulative_breakout_strength = positive_strength - (-negative_strength)
    
    return cumulative_breakout_strength
