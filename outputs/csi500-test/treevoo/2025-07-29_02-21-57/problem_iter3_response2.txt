```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Close Price Return
    df['close_t-5'] = df['close'].shift(5)
    close_price_return = (df['close'] / df['close_t-5']) - 1
    
    # Calculate 5-Day Volume Ratio
    df['volume_5_day_sum'] = df['volume'].rolling(window=5).sum()
    average_volume_5_day = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume_5_day_sum'] / average_volume_5_day
    
    # Multiply Close Price Return by Volume Ratio
    volume_adjusted_momentum = close_price_return * volume_ratio
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-Day Standard Deviation of High-Low Range
    std_high_low_range = high_low_range.rolling(window=5).std()
    
    # Adjust Price Oscillation
    mean_high_low_range = high_low_range.rolling(window=5).mean()
    adjusted_high_low_range = (high_low_range - mean_high_low_range) / std_high_low_range
    moving_avg_adjusted_high_low_range = adjusted_high_low_range.rolling(window=5).mean()
    
    # Combine both factors
    factor = volume_adjusted_momentum + moving_avg_adjusted_high_low_range
    
    return factor
```
