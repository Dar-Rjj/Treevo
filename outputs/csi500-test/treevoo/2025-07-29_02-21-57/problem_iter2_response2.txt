```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).sum() / 14
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] - df['open']
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    df['VolumeWeightedMomentum'] = df['PriceMomentum'] * df['volume']
    
    # Calculate Volume Weighted Price
    df['VolumeWeightedPrice'] = (df['open'] * df['volume'] + 
                                 df['high'] * df['volume'] + 
                                 df['low'] * df['volume'] + 
                                 df['close'] * df['volume']) / (4 * df['volume'])
    
    # Combine Indicators
    df['CombinedIndicators'] = (df['IntradayHighLowSpread'] + 
                                df['CloseToOpenReturn'] + 
                                df['VolumeWeightedPrice']) / 3
    df['CombinedIndicatorAdjustment'] = df['VolumeWeightedPrice'] - df['CombinedIndicators']
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = df['VolumeWeightedMomentum'] / df['ATR']
    
    # Generate Alpha Factor
    df['AlphaFactor'] = df['ATRAdjustedMomentum'] + df['CombinedIndicatorAdjustment']
    
    return df['AlphaFactor']
```
