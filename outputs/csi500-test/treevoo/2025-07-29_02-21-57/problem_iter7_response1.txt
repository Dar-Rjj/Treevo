import pandas as pd

def heuristics_v2(df, N=20, alpha=0.3):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(N)
    
    # Initialize EMA with the first raw momentum value
    df['ema_smoothed'] = 0
    df.loc[df.index[N], 'ema_smoothed'] = df.loc[df.index[N], 'raw_momentum']
    
    # EMA Smoothing
    for i in range(N+1, len(df)):
        df.loc[df.index[i], 'ema_smoothed'] = (df.loc[df.index[i], 'raw_momentum'] * alpha) + (df.loc[df.index[i-1], 'ema_smoothed'] * (1 - alpha))
        
    return df['ema_smoothed']
