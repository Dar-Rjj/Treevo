```python
import pandas as pd

def heuristics_v2(df):
    # Price Momentum Factors
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Signal'] = (df['EMA_5'] > df['EMA_20']).astype(int) * 2 - 1
    
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Return_Momentum'] = df['Daily_Return'].rolling(window=7).sum()
    
    # Volume and Amount Factors
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWAP_Signal'] = (df['close'] > df['VWAP']).astype(int) * 2 - 1
    
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Change_Avg'] = df['Volume_Change'].rolling(window=5).mean()
    
    # Volatility Factors
    df['True_Range'] = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    df['ATR_Signal'] = (df['ATR'] > df['ATR'].expanding().mean()).astype(int) * 2 - 1
    
    df['Return_Std'] = df['Daily_Return'].rolling(window=20).std()
    df['Volatility_Signal'] = (df['Return_Std'] > df['Return_Std'].expanding().mean()).astype(int) * 2 - 1
    
    # Pattern Recognition
    df['Bullish_Engulfing'] = ((df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & 
                               (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))).astype(int) * 2 - 1
    df['Bearish_Engulfing'] = ((df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1)) & 
                               (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))).astype(int) * 2 - 1
    
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Trend_Following_Signal'] = (df['close'] > df['SMA_50']).astype(int) * 2 - 1
    
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['Channel_Breakout_Signal'] = ((df['close'] > df['20_day_high']) * 1 + (df['close'] < df['20_day_low']) * -1).fillna(0)
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (df['Momentum_Signal'] + 
                          df['Return_Momentum'] + 
                          df['VWAP_Signal'] + 
                          df['Volume_Change_Avg'] + 
                          df['ATR_Signal'] + 
                          df['Volatility_Signal'] + 
                          df['Bullish_Engulfing'] + 
                          df['Bearish_Engulfing'] + 
                          df['Trend_Following_Signal'] + 
                          df['Channel_Breakout_Signal'])
    
    return df['Alpha_Factor']
```
