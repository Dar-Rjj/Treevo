```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20DayAvgRange'] = df['DailyRange'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['BreakoutStrength'] = 0.0
    df.loc[df['UpwardBreakout'] == 1, 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgRange']
    df.loc[df['DownwardBreakout'] == 1, 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgRange']
    
    # Cumulate Breakout Strength over the past 10 days
    df['PositiveBreakoutStrength'] = df['BreakoutStrength'].clip(lower=0).rolling(window=10).sum()
    df['NegativeBreakoutStrength'] = df['BreakoutStrength'].clip(upper=0).rolling(window=10).sum()
    df['CumulatedBreakoutStrength'] = df['PositiveBreakoutStrength'] - df['NegativeBreakoutStrength']
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Price
    df['VolumeWeightedPrice'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume'])
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayReturn'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5DayVolumeAdjustedMomentum'] = df['5DayReturn'] * df['5DayVolumeRatio']
    
    # Calculate High-Low Price Oscillation
    df['HighLowOscillation'] = df['high'] - df['low']
    df['10DayMeanHLRange'] = df['HighLowOscillation'].rolling(window=10).mean()
    df['10DayStdHLRange'] = df['HighLowOscillation'].rolling(window=10).std()
    df['AdjustedPriceOscillation'] = (df['HighLowOscillation'] - df['10DayMeanHLRange']) / df['10DayStdHLRange']
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Averages of Intraday Returns
    df['ShortTermMA'] = df['IntradayReturn'].rolling(window=5).mean()
    df['LongTermMA'] = df['IntradayReturn'].rolling(window=20).mean()
    df['IntradayReturnDifference'] = df['ShortTermMA'] - df['LongTermMA']
    
    # Filter by Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['FilteredIntradayReturnDifference'] = df['IntradayReturnDifference'].where(df['VolumeChange'] > 0, 0)
    
    # Combine Indicators
    df['AlphaFactor'] = (df['IntradayHighLowSpread'] + 
                         df['CloseToOpenReturn'] + 
                         df['VolumeWeightedPrice'] + 
                         df['CumulatedBreakoutStrength'] + 
                         df['5DayVolumeAdjustedMomentum'] + 
                         df['AdjustedPriceOscillation'] + 
                         df['FilteredIntradayReturnDifference'])
    
    return df['AlphaFactor']
```
