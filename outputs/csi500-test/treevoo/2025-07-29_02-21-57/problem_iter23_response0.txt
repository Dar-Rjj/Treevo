```python
def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the daily trading intensity
    df['trading_intensity'] = df['volume'] / (df['high'] - df['low'])
    
    # Calculate the close-to-open ratio
    df['close_to_open'] = df['close'] / df['open']
    
    # Calculate the volume weighted price
    df['vwap'] = (df['amount'] / df['volume']).fillna(0)
    
    # Heuristic alpha factor
    df['alpha_factor'] = (df['daily_return'] * df['trading_intensity'] * 
                          df['close_to_open'] + (df['close'] - df['vwap']) / df['daily_range'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
