```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the log difference between close and open price
    log_diff = (df['close'] - df['open']).apply(lambda x: x if x > 0 else -x)
    
    # Calculate the ratio of the amount to volume
    amount_to_volume = df['amount'] / df['volume']
    
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Compute a weighted sum of these features
    factor = (log_diff * 0.4) + (amount_to_volume * 0.3) + (high_low_range * 0.3)
    
    return factor
```
