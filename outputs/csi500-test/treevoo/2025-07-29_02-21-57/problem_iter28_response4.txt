```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_window = 5
    df['Short_VWMA'] = (df['close'] * df['volume']).rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_window = 20
    df['Long_VWMA'] = (df['close'] * df['volume']).rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()
    
    # Combined Crossover Signal
    df['Crossover_Signal'] = 0
    df.loc[df['Short_VWMA'] > df['Long_VWMA'], 'Crossover_Signal'] = 1  # Buy Signal
    df.loc[df['Short_VWMA'] < df['Long_VWMA'], 'Crossover_Signal'] = -1  # Sell Signal
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'].pct_change(periods=5)
    df['5D_Volume_Sum'] = df['volume'].rolling(window=5).sum()
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Sum']
    
    # Evaluate Volume Adjusted Momentum
    df['Momentum_Signal'] = 0
    df.loc[df['Volume_Adjusted_Momentum'] > 0, 'Momentum_Signal'] = 1  # Buy Signal
    df.loc[df['Volume_Adjusted_Momentum'] < 0, 'Momentum_Signal'] = -1  # Sell Signal
    
    # Combine Crossover and Momentum Signals
    df['Combined_Signal'] = 0
    df.loc[(df['Crossover_Signal'] == 1) & (df['Momentum_Signal'] == 1), 'Combined_Signal'] = 2  # Strong Buy
    df.loc[(df['Crossover_Signal'] == -1) & (df['Momentum_Signal'] == -1), 'Combined_Signal'] = -2  # Strong Sell
    df.loc[(df['Crossover_Signal'] == 1) & (df['Momentum_Signal'] != 1), 'Combined_Signal'] = 1  # Moderate Buy
    df.loc[(df['Crossover_Signal'] == -1) & (df['Momentum_Signal'] != -1), 'Combined_Signal'] = -1  # Moderate Sell
    
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    atr_window = 14
    df['ATR'] = df['True_Range'].rolling(window=atr_window).mean()
    
    # Adjust Momentum by ATR
    df['Adjusted_Momentum'] = (df['close'] - df['close'].shift(atr_window)) * df['Volume_Adjusted_Momentum'] / df['ATR']
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Momentum'] + (df['open'] - df['close'].shift(1))
    
    # Combine with Crossover and Momentum Signal
    df['Final_Signal'] = 0
    df.loc[(df['Alpha_Factor'] > 0) & (df['Combined_Signal'] > 0), 'Final_Signal'] = 2  # Strong Buy
    df.loc[(df['Alpha_Factor'] < 0) & (df['Combined_Signal'] < 0), 'Final_Signal'] = -2  # Strong Sell
    df.loc[(df['Alpha_Factor'] > 0) & (df['Combined_Signal'] <= 0), 'Final_Signal'] = 1  # Moderate Buy
    df.loc[(df['Alpha_Factor'] < 0) & (df['Combined_Signal'] >= 0), 'Final_Signal'] = -1  # Moderate Sell
    
    return df['Final_Signal'].fillna(0)
```
