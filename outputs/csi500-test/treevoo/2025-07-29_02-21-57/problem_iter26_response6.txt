```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Volume Weighted Moving Average (VWMA)
    short_window = 10
    long_window = 30
    df['price_volume'] = df['close'] * df['volume']
    df['short_term_vwma'] = df['price_volume'].rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()
    df['long_term_vwma'] = df['price_volume'].rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()

    # Combined Crossover Signal
    df['crossover_signal'] = 0
    df.loc[df['short_term_vwma'] > df['long_term_vwma'], 'crossover_signal'] = 1
    df.loc[df['short_term_vwma'] < df['long_term_vwma'], 'crossover_signal'] = -1

    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_close_return'] = df['close'] / df['close'].shift(5) - 1
    df['5_day_vol_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=50).mean()
    df['volume_adjusted_momentum'] = df['5_day_close_return'] * df['5_day_vol_ratio']

    # Evaluate Volume Adjusted Momentum
    df['momentum_signal'] = 0
    df.loc[df['volume_adjusted_momentum'] > 0, 'momentum_signal'] = 1
    df.loc[df['volume_adjusted_momentum'] < 0, 'momentum_signal'] = -1

    # Combine Crossover and Momentum Signals
    df['combined_signal'] = 0
    df.loc[(df['crossover_signal'] == 1) & (df['momentum_signal'] == 1), 'combined_signal'] = 1
    df.loc[(df['crossover_signal'] == -1) & (df['momentum_signal'] == -1), 'combined_signal'] = -1
    df.loc[(df['crossover_signal'] == 1) & (df['momentum_signal'] != 1), 'combined_signal'] = 0.5
    df.loc[(df['crossover_signal'] == -1) & (df['momentum_signal'] != -1), 'combined_signal'] = -0.5
    df.loc[(df['crossover_signal'] == 0) & (df['momentum_signal'] != 0), 'combined_signal'] = 0

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['10_day_mean_high_low'] = df['high_low_range'].rolling(window=10).mean()
    df['10_day_std_high_low'] = df['high_low_range'].rolling(window=10).std()
    df['adjusted_price_oscillation'] = (df['high_low_range'] - df['10_day_mean_high_low']) / df['10_day_std_high_low']

    # Incorporate Price Oscillation
    df['oscillation_signal'] = 0
    df.loc[df['adjusted_price_oscillation'] > 0, 'oscillation_signal'] = 1
    df.loc[df['adjusted_price_oscillation'] < 0, 'oscillation_signal'] = -1

    # Generate Combined Signal
    df['final_signal'] = 0
    df.loc[(df['combined_signal'] == 1) & (df['oscillation_signal'] == 1), 'final_signal'] = 1
    df.loc[(df['combined_signal'] == -1) & (df['oscillation_signal'] == -1), 'final_signal'] = -1
    df.loc[(df['combined_signal'] == 0.5) & (df['oscillation_signal'] != -1), 'final_signal'] = 0.5
    df.loc[(df['combined_signal'] == -0.5) & (df['oscillation_signal'] != 1), 'final_signal'] = -0.5
    df.loc[(df['combined_signal'] == 0) & (df['oscillation_signal'] != 0), 'final_signal'] = 0

    return df['final_signal']
```
