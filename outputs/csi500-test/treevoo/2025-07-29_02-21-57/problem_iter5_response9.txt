```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Series
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Averages of Intraday Returns
    short_term_ma = df['intraday_return'].rolling(window=5).mean()
    long_term_ma = df['intraday_return'].rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    intraday_return_diff = long_term_ma - short_term_ma
    
    # Filter by Volume Change
    volume_change = df['volume'].pct_change()
    filtered_intraday_return_diff = intraday_return_diff.where(volume_change > 0, 0)
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_price_return = (df['close'] / df['close'].shift(5)) - 1
    five_day_volume = df['volume'].rolling(window=5).sum()
    avg_five_day_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = five_day_volume / avg_five_day_volume
    volume_adjusted_momentum = close_price_return * volume_ratio
    
    # Calculate High-Low Price Oscillation
    high_low_range = df['high'] - df['low']
    five_day_high_low_std = high_low_range.rolling(window=5).std()
    five_day_high_low_mean = high_low_range.rolling(window=5).mean()
    normalized_price_oscillation = (high_low_range - five_day_high_low_mean) / five_day_high_low_std
    
    # Calculate True Range
    previous_close = df['close'].shift(1)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - previous_close).abs()
    tr3 = (df['low'] - previous_close).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    weighted_momentum = price_momentum * df['volume']
    
    # Normalize by ATR
    normalized_weighted_momentum = weighted_momentum / atr
    
    # Generate Alpha Factor
    alpha_factor = (
        normalized_weighted_momentum +
        volume_adjusted_momentum +
        normalized_price_oscillation +
        filtered_intraday_return_diff
    )
    
    return alpha_factor
```
