import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Close Price Return
    close_price_return = df['close'].pct_change(5)
    
    # Calculate 5-Day Volume Ratio
    sum_5d_volume = df['volume'].rolling(window=5).sum()
    avg_5d_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = sum_5d_volume / avg_5d_volume
    
    # Multiply Close Price Return by Volume Ratio
    volume_adjusted_momentum = close_price_return * volume_ratio
    
    # Calculate High-Low Price Oscillation
    high_low_range = df['high'] - df['low']
    std_high_low_range = high_low_range.rolling(window=10).std()
    mean_high_low_range = high_low_range.rolling(window=10).mean()
    price_oscillation = (high_low_range - mean_high_low_range) / std_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['close'].pct_change() - df['open'].pct_change()
    
    # Compute Moving Averages
    short_term_ma = intraday_return.rolling(window=5).mean()
    long_term_ma = intraday_return.rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    intraday_return_difference = short_term_ma - long_term_ma
    
    # Filter by Volume Change
    volume_change = df['volume'].diff()
    filtered_intraday_return_diff = intraday_return_difference.where(volume_change > 0, 0)
    
    # Calculate Short-Term and Long-Term Volume Weighted Moving Average (VWMA)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Generate Combined Signal
    vwma_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    oscillation_signal = (price_oscillation > 0).astype(int) - (price_oscillation < 0).astype(int)
    intraday_signal = (filtered_intraday_return_diff > 0).astype(int) - (filtered_intraday_return_diff < 0).astype(int)
    
    # Combine signals
    combined_signal = vwma_signal + momentum_signal + oscillation_signal + intraday_signal
    
    # Final Alpha Factor
    alpha_factor = pd.Series(index=df.index, dtype=float)
    alpha_factor[combined_signal == 4] = 1
    alpha_factor[combined_signal == -4] = -1
    alpha_factor[(combined_signal >= 2) & (combined_signal < 4)] = 0.5
    alpha_factor[(combined_signal <= -2) & (combined_signal > -4)] = -0.5
    alpha_factor[(combined_signal == 0) | ((combined_signal > 0) & (combined_signal < 2)) | ((combined_signal < 0) & (combined_signal > -2))] = 0
    
    # Detect Volume Spikes
    avg_20d_volume = df['volume'].rolling(window=20).mean()
    volume_spike = (df['volume'] > 2 * avg_20d_volume).astype(float)
    
    # Amplify Signals
    alpha_factor = alpha_factor * (1 + volume_spike)
    
    return alpha_factor
