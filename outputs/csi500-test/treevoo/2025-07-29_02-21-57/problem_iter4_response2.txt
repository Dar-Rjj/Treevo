```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Series
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Averages of Intraday Returns
    short_term_window = 5
    long_term_window = 20
    
    df['short_term_ma'] = df['intraday_return'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['intraday_return'].rolling(window=long_term_window).mean()
    
    # Compute Intraday Return Difference
    df['intraday_diff'] = df['long_term_ma'] - df['short_term_ma']
    
    # Filter by Volume Change
    df['volume_change'] = df['volume'].diff()
    df['factor'] = df.apply(lambda row: row['intraday_diff'] if row['volume_change'] > 0 else 0, axis=1)
    
    return df['factor']
```
