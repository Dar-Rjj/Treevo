```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Previous Close']).abs()
    df['TR3'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True Range'].rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday High-Low'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    df['Weighted Momentum'] = df['Momentum'] * df['volume']
    
    # Combine Indicators
    df['Combined Indicators'] = (df['Intraday High-Low'] + df['Close-to-Open'] + df['Weighted Momentum']) / 3
    df['Adjusted Momentum'] = df['Weighted Momentum'] - df['Combined Indicators']
    
    # Generate Alpha Factor
    df['Alpha Factor'] = (df['Weighted Momentum'] / df['ATR']) + (df['Adjusted Momentum'] / df['ATR'])
    
    return df['Alpha Factor']
```
