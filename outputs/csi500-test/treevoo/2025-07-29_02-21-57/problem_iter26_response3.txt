```python
def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    - data: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    - pandas Series representing the factor values
    """
    # Calculate the price range
    price_range = data['high'] - data['low']
    
    # Calculate the typical price
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    
    # Calculate the volume weighted price
    vwp = (data['amount'] / data['volume'])
    
    # Calculate the factor as a combination of normalized price range and the difference between typical price and volume weighted price
    factor = (price_range / typical_price) * (typical_price - vwp)
    
    return factor
```
