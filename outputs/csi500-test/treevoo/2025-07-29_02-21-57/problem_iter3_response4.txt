```python
import pandas as pd

def heuristics_v2(df, short_window=20, long_window=50, momentum_lookback=10):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    df['Short_VWMA'] = (df['close'] * df['volume']).rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    df['Long_VWMA'] = (df['close'] * df['volume']).rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()
    
    # Calculate Price Momentum
    df['Momentum'] = df['close'].diff().rolling(window=momentum_lookback).sum()
    
    # Generate Crossover Signals
    df['Crossover_Signal'] = 0
    df.loc[df['Short_VWMA'] > df['Long_VWMA'], 'Crossover_Signal'] = 1
    df.loc[df['Short_VWMA'] < df['Long_VWMA'], 'Crossover_Signal'] = -1
    
    # Adjust Signal with Price Momentum
    df['Final_Signal'] = 0
    df.loc[(df['Crossover_Signal'] == 1) & (df['Momentum'] > 0), 'Final_Signal'] = 2  # Strengthen Buy Signal
    df.loc[(df['Crossover_Signal'] == -1) & (df['Momentum'] < 0), 'Final_Signal'] = -2  # Strengthen Sell Signal
    df.loc[(df['Crossover_Signal'] == 0) & (df['Momentum'] > 0), 'Final_Signal'] = 1  # Mild Buy Signal
    df.loc[(df['Crossover_Signal'] == 0) & (df['Momentum'] < 0), 'Final_Signal'] = -1  # Mild Sell Signal
    
    return df['Final_Signal']
```
