```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['Previous_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Previous_Close'])
    df['TR3'] = abs(df['low'] - df['Previous_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['Daily_Range'] = df['high'] - df['low']

    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()

    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['Breakout_Strength_Up'] = (df['close'] - df['high'].shift(1)) / df['Avg_Daily_Range_20']
    df['Breakout_Strength_Down'] = (df['low'].shift(1) - df['close']) / df['Avg_Daily_Range_20']

    # Cumulate Breakout Strength
    df['Pos_Breakout_Sum_10'] = df['Breakout_Strength_Up'].rolling(window=10).sum()
    df['Neg_Breakout_Sum_10'] = df['Breakout_Strength_Down'].rolling(window=10).sum()
    df['Cumulated_Breakout_Strength'] = df['Pos_Breakout_Sum_10'] - df['Neg_Breakout_Sum_10']

    # Calculate Price Momentum
    df['Price_Momentum_14'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['volume'] * df['Price_Momentum_14']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Alpha_Factor'] = (
        df['Intraday_High_Low_Spread'] + 
        df['Close_to_Open_Return'] + 
        df['Cumulated_Breakout_Strength'] + 
        df['ATR_Adjusted_Momentum']
    )

    return df['Alpha_Factor']
```
