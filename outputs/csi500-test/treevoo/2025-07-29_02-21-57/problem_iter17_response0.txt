```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'].pct_change(periods=5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['10D_High_Low_Std'] = df['High_Low_Range'].rolling(window=10).std()
    df['10D_High_Low_Mean'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['10D_High_Low_Mean']) / df['10D_High_Low_Std']

    # Compute Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages of Intraday Returns
    df['Short_Term_MA'] = df['Intraday_Return'].rolling(window=5).mean()
    df['Long_Term_MA'] = df['Intraday_Return'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['Intraday_Return_Diff'] = df['Short_Term_MA'] - df['Long_Term_MA']

    # Filter by Volume Change
    df['Volume_Change'] = df['volume'].diff()
    df['Filtered_Intraday_Return_Diff'] = df.apply(lambda row: row['Intraday_Return_Diff'] if row['Volume_Change'] > 0 else 0, axis=1)

    # Calculate Short-Term and Long-Term VWMA
    df['VWMA_Short_Term'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWMA_Long_Term'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Generate Combined Signal
    def generate_signal(row):
        vwma_signal = 1 if row['VWMA_Short_Term'] > row['VWMA_Long_Term'] else -1 if row['VWMA_Short_Term'] < row['VWMA_Long_Term'] else 0
        momentum_signal = 1 if row['Volume_Adjusted_Momentum'] > 0 else -1 if row['Volume_Adjusted_Momentum'] < 0 else 0
        oscillation_signal = 1 if row['Price_Oscillation'] > 0 else -1 if row['Price_Oscillation'] < 0 else 0

        if vwma_signal == 1 and momentum_signal == 1 and oscillation_signal == 1:
            return 1
        elif vwma_signal == -1 and momentum_signal == -1 and oscillation_signal == -1:
            return -1
        elif (vwma_signal == 1 and momentum_signal == 1) or (vwma_signal == 1 and oscillation_signal == 1) or (momentum_signal == 1 and oscillation_signal == 1):
            return 0.5
        elif (vwma_signal == -1 and momentum_signal == -1) or (vwma_signal == -1 and oscillation_signal == -1) or (momentum_signal == -1 and oscillation_signal == -1):
            return -0.5
        else:
            return 0

    df['Signal'] = df.apply(generate_signal, axis=1)

    return df['Signal']
```
