```python
import pandas as pd

def heuristics_v2(df, N=20, alpha=0.3):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(N)
    
    # EMA Smoothing
    ema = [df['raw_momentum'][N]]  # Initialize the EMA list with the first raw momentum
    for i in range(N+1, len(df)):
        ema_today = (df['raw_momentum'][i] * alpha) + (ema[-1] * (1 - alpha))
        ema.append(ema_today)
    
    # Create a Series for the EMA values and align it with the original DataFrame index
    ema_series = pd.Series(ema, index=df.index[N+1:])
    
    return ema_series
```
