import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term VWMA (e.g., 5-day)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term VWMA (e.g., 20-day)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_t_minus_5 = df['close'].shift(5)
    close_t = df['close']
    price_return = (close_t - close_t_minus_5) / close_t_minus_5
    volume_ratio = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = price_return * volume_ratio
    
    # Combined Crossover and Momentum Signal
    crossover_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    
    # Combine Signals
    combined_signal = crossover_signal + momentum_signal
    strong_buy = (combined_signal == 2).astype(int)
    strong_sell = (combined_signal == -2).astype(int)
    moderate_buy = ((crossover_signal == 1) & (momentum_signal != 1)).astype(int)
    moderate_sell = ((crossover_signal == -1) & (momentum_signal != -1)).astype(int)
    hold = (combined_signal == 0).astype(int)
    
    # Calculate True Range
    high = df['high']
    low = df['low']
    prev_close = df['close'].shift(1).fillna(df['close'])
    tr1 = high - low
    tr2 = (high - prev_close).abs()
    tr3 = (low - prev_close).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-Period ATR
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    intraday_spread = high - low
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    weighted_momentum = price_momentum * df['volume']
    
    # Calculate VWAP
    vwap = (df[['open', 'high', 'low', 'close']] * df['volume']).sum(axis=1) / (4 * df['volume'])
    
    # Adjust Momentum by ATR
    atr_adjusted_momentum = weighted_momentum / atr
    
    # Generate Alpha Factor
    alpha_factor = vwap + atr_adjusted_momentum
    alpha_factor += strong_buy * 2 + moderate_buy
    alpha_factor -= strong_sell * 2 + moderate_sell
    alpha_factor = alpha_factor.fillna(0)
    
    return alpha_factor
