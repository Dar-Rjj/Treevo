```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5_day_volume_sum'] = df['volume'].rolling(window=5).sum()
    avg_5_day_volume = df['volume'].rolling(window=5 * 20).mean()  # 20-periods to smooth out the volume
    df['5_day_volume_ratio'] = df['5_day_volume_sum'] / avg_5_day_volume
    df['volume_adjusted_momentum'] = df['5_day_return'] * df['5_day_volume_ratio']

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    std_10_day_high_low = df['high_low_range'].rolling(window=10).std()
    mean_10_day_high_low = df['high_low_range'].rolling(window=10).mean()
    df['price_oscillation'] = (df['high_low_range'] - mean_10_day_high_low) / std_10_day_high_low

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1

    # Compute Moving Averages of Intraday Returns
    df['short_term_ma'] = df['intraday_return'].rolling(window=5).mean()
    df['long_term_ma'] = df['intraday_return'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['intraday_return_diff'] = df['short_term_ma'] - df['long_term_ma']

    # Filter by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['filtered_intraday_return_diff'] = df['intraday_return_diff'].where(df['volume_change'] > 0, 0)

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['volume_adjusted_momentum'] + 
                          df['price_oscillation'] + 
                          df['filtered_intraday_return_diff'])
    
    return df['alpha_factor']
```
