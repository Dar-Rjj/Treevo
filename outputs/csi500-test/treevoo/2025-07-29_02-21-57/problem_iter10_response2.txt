```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['MA_20_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = 0.0
    df.loc[df['Upward_Breakout'] == 1, 'Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['MA_20_Daily_Range']
    df.loc[df['Downward_Breakout'] == 1, 'Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['MA_20_Daily_Range']
    
    # Cumulate Breakout Strength over the past 10 days
    df['Cumulated_Positive_Breakout_Strength'] = df['Breakout_Strength'].apply(lambda x: max(x, 0)).rolling(window=10).sum()
    df['Cumulated_Negative_Breakout_Strength'] = df['Breakout_Strength'].apply(lambda x: min(x, 0)).rolling(window=10).sum()
    df['Cumulated_Breakout_Strength'] = df['Cumulated_Positive_Breakout_Strength'] + df['Cumulated_Negative_Breakout_Strength']
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Price
    df['Volume_Weighted_Price'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume'])
    
    # Combine Indicators
    df['Alpha_Factor'] = (
        df['Intraday_High_Low_Spread'] +
        df['Close_to_Open_Return'] +
        df['Volume_Weighted_Price'] +
        df['Cumulated_Breakout_Strength']
    )
    
    return df['Alpha_Factor']
```
