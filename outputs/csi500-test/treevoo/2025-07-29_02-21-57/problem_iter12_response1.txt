```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 20-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    df['IntradaySpread'] = df['high'] - df['low']
    
    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Weight by Volume
    df['WeightedMomentum'] = df['Momentum'] * df['volume']
    
    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Combine Indicators
    df['CombinedIndicator'] = (df['IntradaySpread'] + (df['close'] - df['open']) + df['VWAP']) / 3
    df['AdjustedVWAP'] = df['VWAP'] - df['CombinedIndicator']
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = df['WeightedMomentum'] / df['ATR']
    
    # Generate Alpha Factor
    df['AlphaFactor'] = df['ATRAdjustedMomentum'] + df['AdjustedVWAP']
    
    return df['AlphaFactor']
```
