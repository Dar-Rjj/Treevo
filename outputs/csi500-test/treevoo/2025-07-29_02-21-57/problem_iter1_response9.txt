def heuristics_v2(data):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (data['close'] - data['open']) / data['open']
    
    # Calculate Volume Weighted Price
    volume_sum = data['volume'].sum()
    volume_weighted_price = (
        (data['open'] * data['volume']).sum() + 
        (data['high'] * data['volume']).sum() + 
        (data['low'] * data['volume']).sum() + 
        (data['close'] * data['volume']).sum()
    ) / volume_sum
    
    # Combine Indicators
    combined_indicator = (intraday_high_low_spread + close_to_open_return + volume_weighted_price) / 3
    alpha_factor = volume_weighted_price - combined_indicator
    
    return alpha_factor
