```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['breakout_strength'] = 0.0
    df.loc[df['upward_breakout'] == 1, 'breakout_strength'] = (df['close'] - df['high'].shift(1)) / df['avg_daily_range_20']
    df.loc[df['downward_breakout'] == 1, 'breakout_strength'] = (df['low'].shift(1) - df['close']) / df['avg_daily_range_20']
    
    # Cumulate Breakout Strength over the past 10 days
    df['cumulated_breakout_strength'] = df['breakout_strength'].rolling(window=10).sum()
    
    # Calculate Price Momentum Components
    df['short_term_momentum'] = df['close'].pct_change(periods=5)
    df['long_term_momentum'] = df['close'].pct_change(periods=20)
    
    # Integrate Momentum
    momentum_weight_short = 0.7
    momentum_weight_long = 0.3
    df['integrated_momentum'] = (momentum_weight_short * df['short_term_momentum']) + (momentum_weight_long * df['long_term_momentum'])
    
    # Calculate Liquidity Components
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['average_true_range_14'] = df['true_range'].rolling(window=14).mean()
    
    # Integrate Liquidity
    liquidity_weight_vpr = 0.6
    liquidity_weight_atr = 0.4
    df['integrated_liquidity'] = (liquidity_weight_vpr * df['volume_to_price_ratio']) + (liquidity_weight_atr * df['average_true_range_14'])
    
    # Combine Integrated Momentum and Liquidity with Breakout Strength
    momentum_liquidity_weight = 0.5
    final_alpha_factor = (momentum_liquidity_weight * df['integrated_momentum']) + (momentum_liquidity_weight * df['integrated_liquidity']) + df['cumulated_breakout_strength']
    
    return final_alpha_factor
```
