```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_close_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5_day_volume_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = df['5_day_close_return'] * df['5_day_volume_ratio']
    
    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['5_day_high_low_std'] = df['high_low_range'].rolling(window=5).std()
    df['5_day_high_low_mean'] = df['high_low_range'].rolling(window=5).mean()
    df['price_oscillation'] = df['high_low_range'] - df['5_day_high_low_mean']

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['prev_close']).abs()
    df['TR3'] = (df['low'] - df['prev_close']).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['weighted_momentum'] = df['volume'] * df['price_momentum']

    # Normalize by ATR
    df['normalized_momentum'] = df['weighted_momentum'] / df['ATR']

    # Generate Alpha Factor
    df['alpha_factor'] = df['normalized_momentum'] + df['volume_adjusted_momentum'] + df['price_oscillation']

    return df['alpha_factor']
```
