```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'].pct_change(periods=5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5D_Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['10D_HL_Std'] = df['High_Low_Range'].rolling(window=10).std()
    df['10D_HL_Mean'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['10D_HL_Mean']) / df['10D_HL_Std']

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages
    df['5D_Intraday_MA'] = df['Intraday_Return'].rolling(window=5).mean()
    df['20D_Intraday_MA'] = df['Intraday_Return'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['Intraday_Return_Diff'] = df['5D_Intraday_MA'] - df['20D_Intraday_MA']

    # Filter by Volume Change
    df['Volume_Change'] = df['volume'].diff()
    df['Filtered_Intraday_Return_Diff'] = np.where(df['Volume_Change'] > 0, df['Intraday_Return_Diff'], 0)

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['14D_ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])

    # Combine Indicators
    combined_indicator = (df['Intraday_High_Low_Spread'] + df['Close_to_Open_Return'] + df['Weighted_Momentum'] + df['VWAP']) / 4
    df['Combined_Indicator_Adjustment'] = df['Weighted_Momentum'] - combined_indicator

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['14D_ATR']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Combined_Indicator_Adjustment'] + df['VWAP']

    return df['Alpha_Factor']
```
