```python
import pandas as pd

def heuristics_v2(df, N=20, alpha=0.3):
    # Calculate Price Momentum
    df['momentum'] = df['close'] - df['close'].shift(N)
    
    # Initialize EMA with the raw momentum value
    ema_series = df['momentum'].copy()
    ema_series.iloc[:N] = 0  # Set initial values to 0 or any other appropriate initialization
    
    # Apply Exponential Moving Average (EMA) Smoothing
    for i in range(N, len(ema_series)):
        if not pd.isna(ema_series.iloc[i]):
            ema_series.iloc[i] = (df['momentum'].iloc[i] * alpha) + (ema_series.iloc[i-1] * (1 - alpha))
    
    return ema_series
```
