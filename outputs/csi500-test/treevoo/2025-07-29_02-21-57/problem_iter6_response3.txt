```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5-day) Volume Weighted Moving Average (VWMA)
    short_term_days = 5
    df['Close_Vol'] = df['close'] * df['volume']
    df['Short_Term_VWMA'] = df['Close_Vol'].rolling(window=short_term_days).sum() / df['volume'].rolling(window=short_term_days).sum()
    
    # Calculate Long-Term (20-day) Volume Weighted Moving Average (VWMA)
    long_term_days = 20
    df['Long_Term_VWMA'] = df['Close_Vol'].rolling(window=long_term_days).sum() / df['volume'].rolling(window=long_term_days).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_Day_Close_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    average_5_day_volume = df['volume'].rolling(window=5).mean()
    df['5_Day_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / average_5_day_volume
    df['Volume_Adjusted_Momentum'] = df['5_Day_Close_Return'] * df['5_Day_Volume_Ratio']
    
    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5_Day_High_Low_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Price_Oscillation'] = df['High_Low_Range'] - df['5_Day_High_Low_Mean']
    
    # Generate signals
    df['VWMA_Signal'] = (df['Short_Term_VWMA'] > df['Long_Term_VWMA']).astype(int) - (df['Short_Term_VWMA'] < df['Long_Term_VWMA']).astype(int)
    df['Momentum_Signal'] = (df['Volume_Adjusted_Momentum'] > 0).astype(int) - (df['Volume_Adjusted_Momentum'] < 0).astype(int)
    df['Oscillation_Signal'] = (df['Price_Oscillation'] > 0).astype(int) - (df['Price_Oscillation'] < 0).astype(int)
    
    # Combine signals
    df['Combined_Signal'] = df['VWMA_Signal'] + df['Momentum_Signal'] + df['Oscillation_Signal']
    
    # Map combined signal to final output
    conditions = [
        (df['Combined_Signal'] == 3),
        (df['Combined_Signal'] == -3),
        (df['Combined_Signal'] >= 2),
        (df['Combined_Signal'] <= -2),
        (df['Combined_Signal'] == 1),
        (df['Combined_Signal'] == -1),
        (df['Combined_Signal'] == 0)
    ]
    choices = [1, -1, 0.5, -0.5, 0.25, -0.25, 0]
    df['Final_Signal'] = pd.np.select(conditions, choices, default=0)
    
    return df['Final_Signal']
```
