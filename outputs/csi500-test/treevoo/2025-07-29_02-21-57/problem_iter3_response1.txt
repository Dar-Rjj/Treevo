```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Close Price Return
    df['close_t-5'] = df['close'].shift(5)
    df['5_day_return'] = (df['close'] - df['close_t-5']) / df['close_t-5']
    
    # Calculate 5-Day Volume Ratio
    df['5_day_volume'] = df['volume'].rolling(window=5).sum()
    avg_5_day_volume = df['volume'].rolling(window=5*40).mean()  # Assuming 40 trading days in a month for average
    df['volume_ratio'] = df['5_day_volume'] / avg_5_day_volume
    
    # Multiply Close Price Return by Volume Ratio
    df['volume_adjusted_momentum'] = df['5_day_return'] * df['volume_ratio']
    
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 10-Day Standard Deviation and Mean of High-Low Range
    std_10_day_high_low = df['high_low_range'].rolling(window=10).std()
    mean_10_day_high_low = df['high_low_range'].rolling(window=10).mean()
    
    # Adjust Price Oscillation
    df['price_oscillation_adj'] = (df['high_low_range'] - mean_10_day_high_low) / std_10_day_high_low
    
    # Combine both factors into one
    df['factor_value'] = (df['volume_adjusted_momentum'] + df['price_oscillation_adj']) / 2
    return df['factor_value'].dropna()
```
