```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=14):
    # Calculate True Range and High-Low Spread
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['prev_close'])
    df['tr3'] = abs(df['low'] - df['prev_close'])
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['avg_high_low_spread'] = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    
    # Calculate 14-Period Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=lookback_period).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(lookback_period)
    
    # Weight by Volume
    df['weighted_momentum'] = df['price_momentum'] * df['volume']
    
    # Adjust Momentum by ATR
    df['adjusted_momentum'] = df['weighted_momentum'] / df['atr_14']
    
    # Generate Alpha Factor
    df['alpha_factor'] = (df['adjusted_momentum'] + 
                          df['close_to_open_return'] + 
                          (df['high'] - df['low']) / df['avg_high_low_spread'])
    
    # Explore Trend and Volatility Characteristics
    df['momentum'] = df['close'] - df['close'].shift(1)
    df['trend_direction'] = np.where(df['close'] > df['close'].shift(1), 1, -1)
    df['volatility'] = df['close'].pct_change().rolling(window=lookback_period).std()
    df['volume_accumulation'] = df['volume'].rolling(window=lookback_period).sum()
    
    # Examine Opening and Closing Prices
    df['up_day'] = np.where(df['close'] > df['open'], 1, 0)
    df['down_day'] = np.where(df['close'] < df['open'], 1, 0)
    df['up_day_count'] = df['up_day'].rolling(window=lookback_period).sum()
    df['down_day_count'] = df['down_day'].rolling(window=lookback_period).sum()
    
    # Use Candlestick Patterns for Predictive Insights
    df['doji'] = np.where((df['high'] - df['low']) > 0.001 * (df['high'] + df['low']) / 2, 0, 1)
    df['marubozu'] = np.where((df['open'] == df['low']) & (df['close'] == df['high']), 1, 0)
    df['candlestick_pattern_frequency'] = df['doji'] + df['marubozu']
    
    # Determine the strength of these patterns
    df['pattern_strength'] = (abs(df['close'] - df['open']) / (df['high'] - df['low']))
    
    return df['alpha_factor']
```
