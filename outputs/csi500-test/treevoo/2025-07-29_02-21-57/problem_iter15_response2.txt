```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the amount weighted price
    weighted_price = df['amount'] / df['volume']
    
    # Calculate the difference between the close and open prices
    price_diff = df['close'] - df['open']
    
    # Alpha factor: (average price - weighted price) * price difference
    alpha_factor = (avg_price - weighted_price) * price_diff
    
    return alpha_factor
```
