import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayCloseReturn'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5DayVolAdjMomentum'] = df['5DayCloseReturn'] * df['5DayVolumeRatio']

    # Calculate High-Low Price Oscillation
    df['HLRange'] = df['high'] - df['low']
    df['HLStdDev'] = df['HLRange'].rolling(window=5).std()
    df['HLMean'] = df['HLRange'].rolling(window=5).mean()
    df['HLOscillation'] = df['HLRange'] - df['HLMean']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['WeightedMomentum'] = df['PriceMomentum'] * df['volume']

    # Normalize by ATR
    df['NormMomentum'] = df['WeightedMomentum'] / df['ATR']

    # Generate Alpha Factor
    df['AlphaFactor'] = df['NormMomentum'] + df['5DayVolAdjMomentum'] + df['HLOscillation']

    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    df['ShortTermVWMA'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    df['LongTermVWMA'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate Intraday Return Series
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages of Intraday Returns
    df['IntradayReturnShortMA'] = df['IntradayReturn'].rolling(window=5).mean()
    df['IntradayReturnLongMA'] = df['IntradayReturn'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['IntradayReturnDiff'] = df['IntradayReturnLongMA'] - df['IntradayReturnShortMA']

    # Filter by Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['FilteredIntradayReturnDiff'] = df.apply(lambda row: row['IntradayReturnDiff'] if row['VolumeChange'] > 0 else 0, axis=1)

    # Combined Crossover and Momentum Signal
    def combine_signals(row):
        signals = []
        if row['ShortTermVWMA'] > row['LongTermVWMA']:
            signals.append('Buy')
        elif row['ShortTermVWMA'] < row['LongTermVWMA']:
            signals.append('Sell')

        if row['5DayVolAdjMomentum'] > 0:
            signals.append('Buy')
        elif row['5DayVolAdjMomentum'] < 0:
            signals.append('Sell')

        if row['FilteredIntradayReturnDiff'] > 0:
            signals.append('Buy')
        elif row['FilteredIntradayReturnDiff'] < 0:
            signals.append('Sell')

        buy_count = signals.count('Buy')
        sell_count = signals.count('Sell')

        if buy_count == 3:
            return 'Strong Buy'
        elif sell_count == 3:
            return 'Strong Sell'
        elif buy_count >= 2:
            return 'Moderate Buy'
        elif sell_count >= 2:
            return 'Moderate Sell'
        else:
            return 'Hold'

    df['CombinedSignal'] = df.apply(combine_signals, axis=1)

    return df['AlphaFactor']
