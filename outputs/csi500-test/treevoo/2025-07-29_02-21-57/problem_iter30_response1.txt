```python
import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['Price_Change_Over_Period'] = df['close'].pct_change()
    truerange = lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1)))
    df['True_Range'] = df.apply(truerange, axis=1)
    df['ADX_Pos_DM'] = (df['high'].diff() > 0).astype(int) * df['high'].diff()
    df['ADX_Neg_DM'] = (df['low'].diff() < 0).astype(int) * -df['low'].diff()
    df['ADX_Pos_DM'] = df['ADX_Pos_DM'].ewm(span=14, adjust=False).mean()
    df['ADX_Neg_DM'] = df['ADX_Neg_DM'].ewm(span=14, adjust=False).mean()
    df['ADX_TR'] = df['True_Range'].ewm(span=14, adjust=False).mean()
    df['ADX_Pos_DI'] = 100 * (df['ADX_Pos_DM'] / df['ADX_TR'])
    df['ADX_Neg_DI'] = 100 * (df['ADX_Neg_DM'] / df['ADX_TR'])
    df['ADX'] = 100 * (abs(df['ADX_Pos_DI'] - df['ADX_Neg_DI']) / (df['ADX_Pos_DI'] + df['ADX_Neg_DI']))

    # Volume-Related Indicators
    df['Volume_Difference'] = df['volume'] - df['volume'].shift()
    df['Positive_Volume_Index'] = (df['close'] > df['close'].shift()).astype(int) * df['volume']
    df['Negative_Volume_Index'] = (df['close'] < df['close'].shift()).astype(int) * df['volume']

    # Oscillator-Based Alphas
    gain = lambda x: x if x > 0 else 0
    loss = lambda x: -x if x < 0 else 0
    df['Gain'] = df['close'].diff().apply(gain)
    df['Loss'] = df['close'].diff().apply(loss)
    df['Avg_Gain'] = df['Gain'].rolling(window=14).mean()
    df['Avg_Loss'] = df['Loss'].rolling(window=14).mean()
    df['RSI'] = 100 - (100 / (1 + (df['Avg_Gain'] / df['Avg_Loss'])))

    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD_Line'] = df['EMA_12'] - df['EMA_26']

    # Analyzing Gap Patterns at Market Open
    df['Overnight_Return'] = (df['open'] / df['close'].shift()) - 1
    def gap_fill_ratio(row):
        if row['low'] < row['open']:
            return (row['low'] - df.loc[row.name - pd.Timedelta(days=1), 'high']) / (row['open'] - df.loc[row.name - pd.Timedelta(days=1), 'close'])
        else:
            return (row['high'] - df.loc[row.name - pd.Timedelta(days=1), 'low']) / (row['open'] - df.loc[row.name - pd.Timedelta(days=1), 'close'])

    df['Gap_Fill_Ratio'] = df.apply(gap_fill_ratio, axis=1)

    # Range and Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']

    # Trading Activity Metrics
    df['Turnover_Rate'] = df['volume'] / df['amount']
    large_trade_threshold = df['volume'].quantile(0.8)  # Example threshold, can be adjusted
    df['Large_Trade_Proportion'] = (df['volume'] > large_trade_threshold).astype(int) * df['volume'] / df['volume'].sum()

    # Selecting the final factor for output
    # This is just an example; you could combine or select any of the factors created above
    factor = df['ADX'] + df['MACD_Line'] + df['RSI'] + df['Turnover_Rate'] + df['Gap_Fill_Ratio']

    return factor.dropna()
```
