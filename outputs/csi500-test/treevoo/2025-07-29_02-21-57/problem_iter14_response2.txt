```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term VWMA (5 days)
    df['PriceVolumeShort'] = df['close'] * df['volume']
    df['VWMA_Short'] = df['PriceVolumeShort'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Long-Term VWMA (20 days)
    df['PriceVolumeLong'] = df['close'] * df['volume']
    df['VWMA_Long'] = df['PriceVolumeLong'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayReturn'] = df['close'].pct_change(periods=5)
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()
    df['VolumeAdjustedMomentum'] = df['5DayReturn'] * df['5DayVolumeRatio']

    # Calculate High-Low Price Oscillation
    df['HighLowRange'] = df['high'] - df['low']
    df['5DayMeanHLRange'] = df['HighLowRange'].rolling(window=5).mean()
    df['HighLowOscillation'] = df['HighLowRange'] - df['5DayMeanHLRange']

    # Combined Crossover and Momentum Signal
    df['VWMACrossoverSignal'] = (df['VWMA_Short'] > df['VWMA_Long']).astype(int) - (df['VWMA_Short'] < df['VWMA_Long']).astype(int)
    df['VolumeAdjustedMomentumSignal'] = (df['VolumeAdjustedMomentum'] > 0).astype(int) - (df['VolumeAdjustedMomentum'] < 0).astype(int)
    df['HighLowOscillationSignal'] = (df['HighLowOscillation'] > 0).astype(int) - (df['HighLowOscillation'] < 0).astype(int)

    # Combine Signals
    df['CombinedSignal'] = df['VWMACrossoverSignal'] + df['VolumeAdjustedMomentumSignal'] + df['HighLowOscillationSignal']

    # Calculate 10-Day Price Volatility
    df['10DayStd'] = df['close'].rolling(window=10).std()
    df['10DayAvgClose'] = df['close'].rolling(window=10).mean()
    df['10DayVolatility'] = df['10DayStd'] / df['10DayAvgClose']

    # Evaluate 10-Day Price Volatility
    volatility_threshold = 0.1
    df['VolatilitySignal'] = (df['10DayVolatility'] > volatility_threshold).astype(int) - (df['10DayVolatility'] <= volatility_threshold).astype(int)

    # Combine All Signals
    df['TotalSignal'] = df['CombinedSignal'] + df['VolatilitySignal']

    # Interpret the final signal
    conditions = [
        (df['TotalSignal'] == 4),
        (df['TotalSignal'] == 3),
        (df['TotalSignal'] == 2),
        (df['TotalSignal'] == -2),
        (df['TotalSignal'] == -3),
        (df['TotalSignal'] == -4)
    ]
    choices = [1, 1, 0.5, -0.5, -1, -1]
    df['FinalSignal'] = pd.np.select(conditions, choices, default=0)

    return df['FinalSignal']
```
