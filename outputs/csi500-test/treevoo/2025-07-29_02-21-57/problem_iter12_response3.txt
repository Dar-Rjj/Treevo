```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=50, M=20, P=20, Q=20, R=10, S=20, T=30, U=20, V=10, W=10):
    # Momentum-based Alpha Factor
    momentum = df['close'] - df['close'].rolling(window=N).mean()
    
    # Reversal-based Alpha Factor
    reversal = df['close'] - df['high'].rolling(window=M).max()
    
    # Volatility-based Alpha Factor
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=P).std()
    
    # Volume-based Alpha Factor
    volume_ratio = df['volume'] / df['volume'].rolling(window=Q).mean()
    
    # Price Movement-based Alpha Factor
    intraday_range = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Accumulation/Distribution-like Alpha Factor
    ad = ((df['close'] - df['open']) * df['volume']).rolling(window=R).sum()
    
    # Breakout-based Alpha Factor
    breakout = (df['close'] > df['high'].rolling(window=S).max()).astype(int) - \
               (df['close'] < df['low'].rolling(window=S).min()).astype(int)
    
    # Trend Strength Alpha Factor
    trend_strength = df['close'].rolling(window=T).apply(lambda x: np.polyfit(range(T), x, 1)[0])
    
    # Liquidity-based Alpha Factor
    liquidity = (df['volume'] / (df['high'] - df['low'])).rolling(window=U).mean()
    
    # Order Imbalance Alpha Factor
    up_ticks = (df['close'] > df['close'].shift(1)).astype(int)
    down_ticks = (df['close'] < df['close'].shift(1)).astype(int)
    order_imbalance = up_ticks.rolling(window=V).sum() - down_ticks.rolling(window=V).sum()
    
    # Market Sentiment Alpha Factor
    sentiment = (df['close'] > df['open']).astype(int).rolling(window=W).sum() - \
                (df['close'] < df['open']).astype(int).rolling(window=W).sum()
    
    # Combined Alpha Score
    alpha_score = (momentum + reversal + volatility + volume_ratio + intraday_range + ad +
                   breakout + trend_strength + liquidity + order_imbalance + sentiment) / 11
    
    return alpha_score
```
