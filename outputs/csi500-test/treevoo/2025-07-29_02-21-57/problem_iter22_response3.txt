```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term VWMA
    short_term_window = 10
    long_term_window = 30
    df['VWMA_short'] = (df['close'] * df['volume']).rolling(window=short_term_window).sum() / df['volume'].rolling(window=short_term_window).sum()
    df['VWMA_long'] = (df['close'] * df['volume']).rolling(window=long_term_window).sum() / df['volume'].rolling(window=long_term_window).sum()

    # Combined Crossover Signal
    df['crossover_signal'] = 0
    df.loc[df['VWMA_short'] > df['VWMA_long'], 'crossover_signal'] = 1
    df.loc[df['VWMA_short'] < df['VWMA_long'], 'crossover_signal'] = -1

    # Calculate 5-Day Volume Adjusted Momentum
    df['close_5d_return'] = df['close'].pct_change(periods=5)
    df['vol_5d_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = df['close_5d_return'] * df['vol_5d_ratio']
    df['momentum_signal'] = 0
    df.loc[df['volume_adjusted_momentum'] > 0, 'momentum_signal'] = 1
    df.loc[df['volume_adjusted_momentum'] < 0, 'momentum_signal'] = -1

    # Combine Crossover and Momentum Signals
    df['combined_signal'] = 0
    df.loc[(df['crossover_signal'] == 1) & (df['momentum_signal'] == 1), 'combined_signal'] = 1
    df.loc[(df['crossover_signal'] == -1) & (df['momentum_signal'] == -1), 'combined_signal'] = -1
    df.loc[(df['crossover_signal'] == 1) & (df['momentum_signal'] != 1), 'combined_signal'] = 0.5
    df.loc[(df['crossover_signal'] == -1) & (df['momentum_signal'] != -1), 'combined_signal'] = -0.5

    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['prev_close']).abs()
    df['tr3'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate 14-Period ATR
    atr_window = 14
    df['atr'] = df['true_range'].rolling(window=atr_window).mean()

    # Calculate Price Momentum and Weight by Volume
    df['price_momentum'] = df['close'] - df['close'].shift(atr_window)
    df['weighted_momentum'] = df['price_momentum'] * df['volume']

    # Adjust Momentum by ATR
    df['atr_adjusted_momentum'] = df['weighted_momentum'] / df['atr']

    # Generate Alpha Factor
    df['close_to_open_return'] = df['open'] - df['close'].shift(1)
    df['alpha_factor'] = df['close_to_open_return'] + df['atr_adjusted_momentum']

    # Combine with Crossover and Momentum Signal
    df['final_signal'] = 0
    df.loc[(df['alpha_factor'] > 0) & (df['combined_signal'] == 1), 'final_signal'] = 1
    df.loc[(df['alpha_factor'] < 0) & (df['combined_signal'] == -1), 'final_signal'] = -1
    df.loc[(df['alpha_factor'] > 0) & (df['combined_signal'] != 1), 'final_signal'] = 0.5
    df.loc[(df['alpha_factor'] < 0) & (df['combined_signal'] != -1), 'final_signal'] = -0.5

    return df['final_signal']
```
