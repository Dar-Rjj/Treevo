def heuristics_v2(df):
    """
    Generate an alpha factor based on a heuristic approach.
    The function takes a pandas DataFrame with columns: open, high, low, close, amount, volume.
    It returns a pandas Series representing the factor values.
    
    :param df: pandas DataFrame with columns: open, high, low, close, amount, volume
    :return: pandas Series indexed by date with the factor values
    """
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the difference between the close and open prices
    price_change = df['close'] - df['open']
    
    # Calculate the ratio of the price change to the price range
    price_change_ratio = price_change / price_range
    
    # Calculate the ratio of the closing price to the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the ratio of the volume to the amount
    vol_to_amount_ratio = df['volume'] / df['amount']
    
    # Combine the factors into a single alpha factor
    alpha_factor = price_change_ratio * close_to_avg_ratio * vol_to_amount_ratio
    
    return alpha_factor
