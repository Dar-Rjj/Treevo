import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']

    # Smooth High-Low Difference
    ema_high_low = high_low_diff.ewm(span=10, adjust=False).mean()
    smoothed_high_low_diff = ema_high_low - ema_high_low.shift(1)

    # Integrate with Close-Open Return
    close_open_return = (df['close'] / df['open']) - 1
    integrated_value = close_open_return + (smoothed_high_low_diff * df['volume'])

    # Incorporate Price Momentum
    n_days = 20
    price_momentum = df['close'] - df['close'].shift(n_days)
    
    # EMA Smoothing for Price Momentum
    alpha = 0.3
    momentum_ema = price_momentum.ewm(alpha=alpha, adjust=False).mean()

    # Combine all components
    final_factor = integrated_value + momentum_ema
    
    return final_factor
