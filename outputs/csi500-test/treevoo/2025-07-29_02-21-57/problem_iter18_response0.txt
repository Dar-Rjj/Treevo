```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term VWMA (e.g., 5-day)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term VWMA (e.g., 20-day)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_return_5d = df['close'] / df['close'].shift(5) - 1
    volume_ratio_5d = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = close_return_5d * volume_ratio_5d
    
    # Combined Crossover and Momentum Signal
    crossover_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    combined_signal = crossover_signal + momentum_signal
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Momentum (14 days)
    price_momentum = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    weighted_momentum = price_momentum * df['volume']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['open'] * df['volume']) + (df['high'] * df['volume']) + (df['low'] * df['volume']) + (df['close'] * df['volume'])) / (df['volume'] * 4)
    
    # Adjust Momentum by ATR
    atr_adjusted_momentum = weighted_momentum / atr
    
    # Generate Alpha Factor
    alpha_factor = vwap + atr_adjusted_momentum
    alpha_factor += (intraday_high_low_spread - intraday_high_low_spread.mean()) / intraday_high_low_spread.std()
    alpha_factor += combined_signal * 1.5  # Increase/decrease based on the combined signal strength
    
    return alpha_factor
```
