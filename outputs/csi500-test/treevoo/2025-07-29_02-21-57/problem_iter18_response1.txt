```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 21-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=21).mean()
    
    # Calculate 7-Day Volume Adjusted Momentum
    df['7DayCloseReturn'] = df['close'] / df['close'].shift(7) - 1
    df['7DayVolumeRatio'] = df['volume'].rolling(window=7).sum() / df['volume'].rolling(window=7 * 21).mean()
    df['7DayVolAdjMomentum'] = df['7DayCloseReturn'] * df['7DayVolumeRatio']
    
    # Weight by Volume
    df['WeightedMomentum'] = df['volume'] * df['7DayVolAdjMomentum']
    
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20DayAvgDailyRange'] = df['DailyRange'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['BreakoutStrength'] = 0
    df.loc[df['UpwardBreakout'] == 1, 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgDailyRange']
    df.loc[df['DownwardBreakout'] == 1, 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgDailyRange']
    
    # Cumulate Breakout Strength
    df['CumulatedBreakoutStrength'] = df['BreakoutStrength'].rolling(window=15).apply(lambda x: x[x > 0].sum() - x[x < 0].sum(), raw=False)
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] / df['open'] - 1
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(21)
    
    # Calculate Volume-Weighted Average Price
    prices = [df['open'], df['high'], df['low'], df['close']]
    volumes = [df['volume']] * 4
    weighted_prices = sum(p * v for p, v in zip(prices, volumes))
    total_volume = sum(volumes)
    df['VWAP'] = weighted_prices / total_volume
    
    # Combine Indicators
    df['CombinedIndicator'] = (df['7DayVolAdjMomentum'] + df['IntradayHighLowSpread'] + 
                               df['CloseToOpenReturn'] + df['VWAP'] + 
                               df['CumulatedBreakoutStrength'] + df['PriceMomentum']) / 6
    df['AdjustedMomentum'] = df['WeightedMomentum'] - df['CombinedIndicator']
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = df['AdjustedMomentum'] / df['ATR']
    
    # Generate Final Alpha Factor
    df['AlphaFactor'] = df['ATRAdjustedMomentum'] + df['VWAP']
    
    return df['AlphaFactor']
```
