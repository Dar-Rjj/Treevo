import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Previous Close']).abs()
    df['TR3'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True Range'].rolling(window=14).mean()
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5-Day Close Price Return'] = df['close'].pct_change(periods=5)
    df['5-Day Volume Sum'] = df['volume'].rolling(window=5).sum()
    df['Average 5-Day Volume'] = df['volume'].rolling(window=5).mean()
    df['5-Day Volume Ratio'] = df['5-Day Volume Sum'] / df['Average 5-Day Volume']
    df['Volume Adjusted Momentum'] = df['5-Day Close Price Return'] * df['5-Day Volume Ratio']
    
    # Weight by Volume
    df['Weighted Volume Momentum'] = df['volume'] * df['Volume Adjusted Momentum']
    
    # Calculate Daily Range
    df['Daily Range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20-Day Avg Daily Range'] = df['Daily Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Upward Breakout'] = df['close'] > df['high'].shift(1)
    df['Downward Breakout'] = df['close'] < df['low'].shift(1)
    
    # Calculate Breakout Strength
    df['Breakout Strength'] = 0
    df.loc[df['Upward Breakout'], 'Breakout Strength'] = (df['close'] - df['high'].shift(1)) / df['20-Day Avg Daily Range']
    df.loc[df['Downward Breakout'], 'Breakout Strength'] = (df['low'].shift(1) - df['close']) / df['20-Day Avg Daily Range']
    
    # Cumulate Breakout Strength
    df['Positive Breakout Strength'] = df['Breakout Strength'].clip(lower=0).rolling(window=10).sum()
    df['Negative Breakout Strength'] = df['Breakout Strength'].clip(upper=0).rolling(window=10).sum()
    df['Cumulated Breakout Strength'] = df['Positive Breakout Strength'] + df['Negative Breakout Strength']
    
    # Calculate Intraday High-Low Spread
    df['Intraday High-Low Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open Return'] = df['close'] / df['open'] - 1
    
    # Calculate Price Momentum
    df['Price Momentum'] = df['close'] - df['close'].shift(14)
    
    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Combine Indicators
    df['Weighted Momentum'] = df['Weighted Volume Momentum'] + df['Price Momentum']
    df['Combined Indicator'] = (df['Intraday High-Low Spread'] + 
                                df['Close-to-Open Return'] + 
                                df['VWAP'] + 
                                df['Cumulated Breakout Strength'] + 
                                df['Price Momentum'] + 
                                (df['high'] - df['low'])) / 6
    df['Adjusted Momentum'] = df['Weighted Momentum'] - df['Combined Indicator']
    
    # Adjust Momentum by ATR
    df['ATR Adjusted Momentum'] = df['Weighted Momentum'] / df['ATR']
    
    # Generate Final Alpha Factor
    df['Final Alpha Factor'] = df['ATR Adjusted Momentum'] + df['Combined Indicator'] + df['VWAP']
    
    return df['Final Alpha Factor']
