```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_close_return'] = df['close'].pct_change(periods=5)
    df['5_day_volume_sum'] = df['volume'].rolling(window=5).sum()
    df['avg_5_day_volume'] = df['volume'].rolling(window=5 * 21).mean()  # 21 trading days in a month
    df['5_day_volume_ratio'] = df['5_day_volume_sum'] / df['avg_5_day_volume']
    df['volume_adjusted_momentum'] = df['5_day_close_return'] * df['5_day_volume_ratio']

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['10_day_high_low_std'] = df['high_low_range'].rolling(window=10).std()
    df['10_day_high_low_mean'] = df['high_low_range'].rolling(window=10).mean()
    df['adjusted_price_oscillation'] = (df['high_low_range'] - df['10_day_high_low_mean']) / df['10_day_high_low_std']

    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages of Intraday Returns
    df['short_term_intraday_ma'] = df['intraday_return'].rolling(window=5).mean()
    df['long_term_intraday_ma'] = df['intraday_return'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['intraday_return_diff'] = df['short_term_intraday_ma'] - df['long_term_intraday_ma']

    # Filter by Volume Change
    df['volume_change'] = df['volume'].diff()
    df['filtered_intraday_return_diff'] = df['intraday_return_diff'].where(df['volume_change'] > 0, 0)

    # Calculate Short-Term and Long-Term VWMA
    def vwma(prices, volumes, window):
        return (prices * volumes).rolling(window=window).sum() / volumes.rolling(window=window).sum()
    
    df['short_term_vwma'] = vwma(df['close'], df['volume'], 5)
    df['long_term_vwma'] = vwma(df['close'], df['volume'], 20)

    # Generate Combined Signal
    df['vwma_signal'] = (df['short_term_vwma'] > df['long_term_vwma']).astype(int) - (df['short_term_vwma'] < df['long_term_vwma']).astype(int)
    df['momentum_signal'] = (df['volume_adjusted_momentum'] > 0).astype(int) - (df['volume_adjusted_momentum'] < 0).astype(int)
    df['price_oscillation_signal'] = (df['adjusted_price_oscillation'] > 0).astype(int) - (df['adjusted_price_oscillation'] < 0).astype(int)

    conditions = [
        (df['vwma_signal'] == 1) & (df['momentum_signal'] == 1) & (df['price_oscillation_signal'] == 1),
        (df['vwma_signal'] == -1) & (df['momentum_signal'] == -1) & (df['price_oscillation_signal'] == -1),
        (df['vwma_signal'] == 1) & (df['momentum_signal'] == 1) & (df['price_oscillation_signal'] == 0),
        (df['vwma_signal'] == -1) & (df['momentum_signal'] == -1) & (df['price_oscillation_signal'] == 0),
        (df['vwma_signal'] == 1) & (df['momentum_signal'] == 0) & (df['price_oscillation_signal'] == 1),
        (df['vwma_signal'] == -1) & (df['momentum_signal'] == 0) & (df['price_oscillation_signal'] == -1),
    ]
    choices = [1, -1, 0.5, -0.5, 0.5, -0.5]
    df['alpha_factor'] = pd.np.select(conditions, choices, default=0)

    return df['alpha_factor']
```
