```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(periods=20)
    
    # Integrate Momentum
    integrated_momentum = 0.8 * short_term_momentum + 0.2 * long_term_momentum
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    prev_close = df['close'].shift(1).fillna(df['close'])
    tr2 = (df['high'] - prev_close).abs()
    tr3 = (df['low'] - prev_close).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_t5 = df['close'].shift(5)
    close_t = df['close']
    five_day_return = (close_t - close_t5) / close_t5
    five_day_volume_ratio = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean().rolling(window=5).sum()
    volume_adjusted_momentum = five_day_return * five_day_volume_ratio
    
    # Combine Indicators
    atr_momentum = integrated_momentum + atr
    vwma_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    volume_momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Generate Alpha Factor
    combined_signals = atr_momentum + vwma_signal + volume_momentum_signal + intraday_high_low_spread + close_to_open_return
    alpha_factor = volume_weighted_price - combined_signals
    alpha_factor = 0.6 * integrated_momentum + 0.4 * combined_signals
    alpha_factor_sma = alpha_factor.rolling(window=5).mean()
    
    return alpha_factor_sma
```
