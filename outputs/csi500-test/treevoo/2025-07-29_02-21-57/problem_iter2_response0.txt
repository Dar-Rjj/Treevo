```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Prev_Close']).abs()
    df['TR3'] = (df['low'] - df['Prev_Close']).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Momentum'] * df['volume']

    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Volume Weighted Price
    df['Volume_Weighted_Price'] = (df['open'] * df['volume'] + 
                                   df['high'] * df['volume'] + 
                                   df['low'] * df['volume'] + 
                                   df['close'] * df['volume']) / (4 * df['volume'])

    # Combine Indicators
    df['Alpha_Factor'] = (df['ATR'] * df['Momentum'] + 
                          df['Weighted_Momentum'] + 
                          df['Intraday_Spread'] + 
                          df['Close_to_Open_Return'] + 
                          df['Volume_Weighted_Price'])
    
    return df['Alpha_Factor']
```
