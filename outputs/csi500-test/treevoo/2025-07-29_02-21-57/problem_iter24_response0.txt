import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term VWMA (10-day)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Long-Term VWMA (50-day)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_return_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    volume_ratio_5d = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=50).mean()
    volume_adjusted_momentum = close_return_5d * volume_ratio_5d
    
    # Detect Volume Spikes
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_spike = (df['volume'] > 2 * avg_volume_20d).astype(int)
    
    # Combined Crossover, Momentum, and Volume Spike Signal
    vwma_crossover = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    
    # Amplify the current signal based on volume spike
    combined_signal = vwma_crossover + momentum_signal
    combined_signal[volume_spike == 1] *= 2
    
    return combined_signal
