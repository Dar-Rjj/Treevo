```python
import pandas as pd

def heuristics_v2(df, N=20, breakout_lookback=10, alpha=0.3):
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(N)
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Combine Momentum and Range
    df['Combined_Momentum'] = df['Price_Momentum'] + df['Daily_Range']
    
    # Apply Exponential Moving Average (EMA) Smoothing
    df['EMA_Combined_Momentum'] = df['Combined_Momentum'].ewm(alpha=alpha, adjust=False).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    average_daily_range = df['Daily_Range'].rolling(window=N).mean()
    df['Upward_Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / average_daily_range
    df['Downward_Breakout_Strength'] = (df['low'].shift(1) - df['close']) / average_daily_range
    
    # Cumulate Breakout Strength
    df['Cumulative_Positive_Breakout_Strength'] = df['Upward_Breakout_Strength'].rolling(window=breakout_lookback).sum()
    df['Cumulative_Negative_Breakout_Strength'] = df['Downward_Breakout_Strength'].rolling(window=breakout_lookback).sum()
    df['Net_Cumulative_Breakout_Strength'] = df['Cumulative_Positive_Breakout_Strength'] - df['Cumulative_Negative_Breakout_Strength']
    
    # Combine EMA and Breakout Strength
    df['Crossover_Momentum_and_Breakout_Indicator'] = df['EMA_Combined_Momentum'] * df['Net_Cumulative_Breakout_Strength']
    
    return df['Crossover_Momentum_and_Breakout_Indicator']
```
