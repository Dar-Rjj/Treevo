```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Prev_Close'])
    df['TR3'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] / df['open'] - 1

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open']*df['volume'] + df['high']*df['volume'] + df['low']*df['volume'] + df['close']*df['volume']) / (4 * df['volume'])

    # Combine Indicators
    df['Combined_Indicators'] = (df['Intraday_HL_Spread'] + df['Close_to_Open_Return'] + df['Weighted_Momentum'] + df['VWAP']) / 4
    df['Adjusted_Momentum'] = df['Weighted_Momentum'] - df['Combined_Indicators']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Adjusted_Momentum'] + df['VWAP']

    return df['Alpha_Factor'].dropna()
```
