```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the price movement relative to the volume
    price_movement = (df['close'] - df['open']) / df['volume']
    
    # Calculate the factor value using a combination of the above metrics
    factor_values = (daily_range * avg_open_close) + price_movement
    
    return factor_values
```
