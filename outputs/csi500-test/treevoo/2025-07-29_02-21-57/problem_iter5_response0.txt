```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Close Price Return
    df['5D_Return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate 5-Day Volume Ratio
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    
    # Multiply Close Price Return by Volume Ratio
    df['Volume_Adjusted_Momentum'] = df['5D_Return'] * df['5D_Volume_Ratio']
    
    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5D_HL_Std'] = df['High_Low_Range'].rolling(window=5).std()
    df['5D_HL_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['5D_HL_Mean']) / df['5D_HL_Std']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] / df['open']) - 1
    
    # Compute Moving Averages of Intraday Returns
    df['5D_Intraday_MA'] = df['Intraday_Return'].rolling(window=5).mean()
    df['20D_Intraday_MA'] = df['Intraday_Return'].rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    df['Intraday_Return_Diff'] = df['5D_Intraday_MA'] - df['20D_Intraday_MA']
    
    # Compute Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Apply Condition on Volume Change
    df['Filtered_Intraday_Return_Diff'] = df.apply(lambda row: row['Intraday_Return_Diff'] if row['Volume_Change'] > 0 else 0, axis=1)
    
    # Combine all the components to form the final factor
    df['Alpha_Factor'] = (df['Volume_Adjusted_Momentum'] + 
                          df['Price_Oscillation'] + 
                          df['Filtered_Intraday_Return_Diff'])
    
    return df['Alpha_Factor']
```
