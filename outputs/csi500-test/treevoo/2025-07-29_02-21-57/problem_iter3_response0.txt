```python
import pandas as pd

def heuristics_v2(df, N=20, alpha=0.3):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(N)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(N)
    
    # Adjust Raw Momentum with Volume Ratio
    df['adjusted_momentum'] = df['raw_momentum'] * df['volume_ratio']
    
    # Initialize EMA with the first adjusted momentum value
    df['ema_adjusted_momentum'] = 0
    df.loc[df.index[N], 'ema_adjusted_momentum'] = df.loc[df.index[N], 'adjusted_momentum']
    
    # Update EMA values
    for i in range(N+1, len(df)):
        df.loc[df.index[i], 'ema_adjusted_momentum'] = (df.loc[df.index[i], 'adjusted_momentum'] * alpha) + (df.loc[df.index[i-1], 'ema_adjusted_momentum'] * (1 - alpha))
    
    # Return the factor values as a Series
    return df['ema_adjusted_momentum'].dropna()
```
