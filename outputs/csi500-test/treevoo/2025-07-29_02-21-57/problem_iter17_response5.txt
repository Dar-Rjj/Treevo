import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Volume Weighted Moving Averages
    short_term = 10
    long_term = 30
    
    df['Short_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=short_term).sum() / df['volume'].rolling(window=short_term).sum()
    df['Long_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=long_term).sum() / df['volume'].rolling(window=long_term).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['Close_Return_5D'] = df['close'].pct_change(5)
    df['Volume_Ratio_5D'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum_5D'] = df['Close_Return_5D'] * df['Volume_Ratio_5D']

    # Generate Crossover and Momentum Signals
    df['VWMA_Crossover_Signal'] = np.where(df['Short_Term_VWMA'] > df['Long_Term_VWMA'], 1, -1)
    df['Momentum_Signal'] = np.where(df['Volume_Adjusted_Momentum_5D'] > 0, 1, -1)

    # Combine Signals
    df['Combined_Signal'] = np.where((df['VWMA_Crossover_Signal'] == 1) & (df['Momentum_Signal'] == 1), 2, 
                                     np.where((df['VWMA_Crossover_Signal'] == -1) & (df['Momentum_Signal'] == -1), -2, 
                                              np.where((df['VWMA_Crossover_Signal'] == 1) & (df['Momentum_Signal'] != 1), 1, 
                                                       np.where((df['VWMA_Crossover_Signal'] == -1) & (df['Momentum_Signal'] != -1), -1, 0))))

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    atr_period = 14
    df['ATR'] = df['True_Range'].rolling(window=atr_period).mean()

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Close_to_Open_Return']

    # Combine with Crossover and Momentum Signal
    df['Final_Signal'] = np.where((df['Alpha_Factor'] > 0) & (df['Combined_Signal'] == 2), 2,
                                  np.where((df['Alpha_Factor'] < 0) & (df['Combined_Signal'] == -2), -2,
                                           np.where((df['Alpha_Factor'] > 0) & (df['Combined_Signal'] == 1), 1,
                                                    np.where((df['Alpha_Factor'] < 0) & (df['Combined_Signal'] == -1), -1,
                                                             np.where((df['Alpha_Factor'] > 0) & (df['Combined_Signal'] != 2), 1,
                                                                      np.where((df['Alpha_Factor'] < 0) & (df['Combined_Signal'] != -2), -1, 0))))))

    return df['Final_Signal'].fillna(0)
