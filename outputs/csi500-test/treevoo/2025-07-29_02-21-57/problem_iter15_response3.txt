```python
import pandas as pd

def heuristics_v2(df):
    # Simple Moving Averages
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Averages
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # True Range and Average True Range
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # New Highs or Lows
    df['New_High'] = (df['high'] > df['high'].rolling(window=5).max()).astype(int)
    df['New_Low'] = (df['low'] < df['low'].rolling(window=5).min()).astype(int)
    
    # Rate of Change
    df['ROC_12'] = (df['close'] - df['close'].shift(12)) / df['close'].shift(12)
    
    # Relative Strength Index
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # On-Balance Volume
    df['OBV'] = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int) - df['volume'] * (df['close'] < df['close'].shift(1)).astype(int)).cumsum()
    
    # Volume Weighted Average Price
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    
    # Composite Alpha Factor
    df['Alpha_Factor'] = (df['SMA_10'] - df['SMA_20']) + (df['EMA_10'] - df['EMA_20']) + (df['ROC_12'] - df['RSI_14']) + df['OBV'] + df['VWAP']
    
    return df['Alpha_Factor']
```
