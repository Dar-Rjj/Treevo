```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term VWMA
    short_term = 10
    long_term = 30
    df['VolumeClose'] = df['close'] * df['volume']
    
    df['ShortTermVWMA'] = (df['VolumeClose'].rolling(window=short_term).sum() / 
                           df['volume'].rolling(window=short_term).sum())
    df['LongTermVWMA'] = (df['VolumeClose'].rolling(window=long_term).sum() / 
                          df['volume'].rolling(window=long_term).sum())
    
    # Combined Crossover Signal
    df['CrossoverSignal'] = 0
    df.loc[df['ShortTermVWMA'] > df['LongTermVWMA'], 'CrossoverSignal'] = 1
    df.loc[df['ShortTermVWMA'] < df['LongTermVWMA'], 'CrossoverSignal'] = -1
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayMomentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['5DayVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeAdjustedMomentum'] = df['5DayMomentum'] * df['5DayVolume']
    
    # Evaluate Volume Adjusted Momentum
    df['MomentumSignal'] = 0
    df.loc[df['VolumeAdjustedMomentum'] > 0, 'MomentumSignal'] = 1
    df.loc[df['VolumeAdjustedMomentum'] < 0, 'MomentumSignal'] = -1
    
    # Combine Crossover and Momentum Signals
    df['CombinedSignal'] = 0
    df.loc[(df['CrossoverSignal'] == 1) & (df['MomentumSignal'] == 1), 'CombinedSignal'] = 1
    df.loc[(df['CrossoverSignal'] == -1) & (df['MomentumSignal'] == -1), 'CombinedSignal'] = -1
    df.loc[(df['CrossoverSignal'] == 1) & (df['MomentumSignal'] != 1), 'CombinedSignal'] = 0.5
    df.loc[(df['CrossoverSignal'] == -1) & (df['MomentumSignal'] != -1), 'CombinedSignal'] = -0.5
    
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period ATR
    atr_period = 14
    df['ATR'] = df['TrueRange'].rolling(window=atr_period).mean()
    
    # Adjust Momentum by ATR
    df['ATRAdjustedMomentum'] = (df['close'] - df['close'].shift(atr_period)) * df['volume'] / df['ATR']
    
    # Introduce Short-Term Reversal Factor
    df['1DayReversal'] = df['open'] - df['close'].shift(1)
    df['ReversalSignal'] = 0
    df.loc[df['1DayReversal'] > 0, 'ReversalSignal'] = 1
    df.loc[df['1DayReversal'] < 0, 'ReversalSignal'] = -1
    
    # Generate Alpha Factor
    df['AlphaFactor'] = df['ATRAdjustedMomentum'] + df['1DayReversal']
    
    # Final Signal
    df['FinalSignal'] = 0
    df.loc[(df['AlphaFactor'] > 0) & (df['CombinedSignal'] == 1), 'FinalSignal'] = 1
    df.loc[(df['AlphaFactor'] < 0) & (df['CombinedSignal'] == -1), 'FinalSignal'] = -1
    df.loc[(df['AlphaFactor'] > 0) & (df['CombinedSignal'] != 1), 'FinalSignal'] = 0.5
    df.loc[(df['AlphaFactor'] < 0) & (df['CombinedSignal'] != -1), 'FinalSignal'] = -0.5
    
    return df['FinalSignal']
```
