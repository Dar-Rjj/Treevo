```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a novel heuristic approach.
    
    :param df: A pandas DataFrame with (open, high, low, close, amount, volume) columns and date as index.
    :return: A pandas Series of the generated factor values indexed by date.
    """
    # Calculate the intraday return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the turnover ratio
    turnover_ratio = df['volume'] / (df['amount'] / df['close'])
    
    # Calculate the price momentum
    price_momentum = df['close'].pct_change()
    
    # Combine the factors to create the alpha factor
    factor = (intraday_return + turnover_ratio * 0.5 + price_momentum * 0.3)
    
    return factor
```
