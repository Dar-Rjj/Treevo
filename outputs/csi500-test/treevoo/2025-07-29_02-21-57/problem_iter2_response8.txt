```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 5-Day Volume Adjusted Momentum
    df['5_Day_Close_Return'] = df['close'].pct_change(periods=5)
    df['5_Day_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['5_Day_Close_Return'] * df['5_Day_Volume_Ratio']

    # High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['High_Low_Range'].rolling(window=5).mean()) / df['High_Low_Range'].rolling(window=5).std()

    # True Range
    df['Previous_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Previous_Close'])
    df['TR3'] = abs(df['low'] - df['Previous_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # 14-Period Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Normalize by ATR
    df['Normalized_Momentum'] = df['Weighted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['Alpha_Factor'] = (df['Normalized_Momentum'] 
                          + df['Volume_Adjusted_Momentum'] 
                          + df['Price_Oscillation'])

    return df['Alpha_Factor']
```
