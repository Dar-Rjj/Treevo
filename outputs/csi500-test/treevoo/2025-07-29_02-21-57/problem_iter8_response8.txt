```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous close to the current open
    df['overnight_move'] = df['open'] - df['close'].shift(1)
    
    # Calculate the buying pressure during the day (volume weighted by the difference between close and low)
    df['buying_pressure'] = (df['close'] - df['low']) * df['volume']
    
    # Calculate the selling pressure during the day (volume weighted by the difference between high and close)
    df['selling_pressure'] = (df['high'] - df['close']) * df['volume']
    
    # Calculate the net buying pressure (buying pressure minus selling pressure)
    df['net_buying_pressure'] = df['buying_pressure'] - df['selling_pressure']
    
    # Calculate the factor value as a combination of overnight move and net buying pressure
    df['factor_value'] = (df['overnight_move'] + df['net_buying_pressure']) / df['amount']
    
    return df['factor_value']
```
