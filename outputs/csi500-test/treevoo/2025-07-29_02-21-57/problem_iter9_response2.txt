```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].pct_change(5)
    long_term_momentum = df['close'].pct_change(20)
    
    # Integrate Momentum
    integrated_momentum = 0.7 * short_term_momentum + 0.3 * long_term_momentum
    
    # Calculate True Range
    df['previous_close'] = df['close'].shift(1)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['previous_close']).abs()
    tr3 = (df['low'] - df['previous_close']).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(14)) / 14
    
    # Weight by Volume
    volume_weighted_momentum = price_momentum * df['volume']
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (
        df['open'] * df['volume'] + 
        df['high'] * df['volume'] + 
        df['low'] * df['volume'] + 
        df['close'] * df['volume']
    ) / (4 * df['volume'])
    
    # Combine Indicators
    combined_indicators = (
        atr + 
        intraday_high_low_spread + 
        close_to_open_return + 
        volume_weighted_price
    )
    
    # Generate Alpha Factor
    alpha_factor = volume_weighted_momentum - combined_indicators
    
    # Combine Momentum and Integrated Indicators
    final_alpha_factor = 0.6 * integrated_momentum + 0.4 * combined_indicators
    
    # Apply Smoothing
    final_alpha_factor = final_alpha_factor.rolling(window=5).mean()
    
    return final_alpha_factor
```
