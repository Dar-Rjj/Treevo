```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Close Price Return
    close_t = df['close']
    close_t_minus_5 = df['close'].shift(5)
    close_price_return = (close_t - close_t_minus_5) / close_t_minus_5
    
    # Calculate 5-Day Volume Ratio
    volume_sum_5_day = df['volume'].rolling(window=5).sum()
    average_volume_5_day = df['volume'].rolling(window=20).mean()  # 20 days for a more stable average
    volume_ratio = volume_sum_5_day / average_volume_5_day
    
    # Multiply Close Price Return by Volume Ratio to get Volume Adjusted Momentum
    volume_adjusted_momentum = close_price_return * volume_ratio
    
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-Day Standard Deviation of High-Low Range
    high_low_range_5_day = high_low_range.rolling(window=5).std()
    mean_high_low_range_5_day = high_low_range.rolling(window=5).mean()
    
    # Adjust Price Oscillation
    adjusted_price_oscillation = (high_low_range - mean_high_low_range_5_day) / high_low_range_5_day
    
    # Combine Volume Adjusted Momentum and Adjusted Price Oscillation
    alpha_factor = volume_adjusted_momentum + adjusted_price_oscillation

    return alpha_factor
```
