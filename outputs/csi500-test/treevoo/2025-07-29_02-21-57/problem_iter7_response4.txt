```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_term_window = 10
    df['Short_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=short_term_window).sum() / df['volume'].rolling(window=short_term_window).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_term_window = 30
    df['Long_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=long_term_window).sum() / df['volume'].rolling(window=long_term_window).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = (df['close'].shift(-5) - df['close']) / df['close']
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # VWMA Crossover Signal
    df['VWMA_Crossover_Signal'] = 0
    df.loc[df['Short_Term_VWMA'] > df['Long_Term_VWMA'], 'VWMA_Crossover_Signal'] = 1
    df.loc[df['Short_Term_VWMA'] < df['Long_Term_VWMA'], 'VWMA_Crossover_Signal'] = -1

    # Volume Adjusted Momentum Signal
    df['Momentum_Signal'] = 0
    df.loc[df['Volume_Adjusted_Momentum'] > 0, 'Momentum_Signal'] = 1
    df.loc[df['Volume_Adjusted_Momentum'] < 0, 'Momentum_Signal'] = -1

    # Combine Signals
    df['Combined_Signal'] = 0
    df.loc[(df['VWMA_Crossover_Signal'] == 1) & (df['Momentum_Signal'] == 1), 'Combined_Signal'] = 2  # Strong Buy
    df.loc[(df['VWMA_Crossover_Signal'] == -1) & (df['Momentum_Signal'] == -1), 'Combined_Signal'] = -2  # Strong Sell
    df.loc[(df['VWMA_Crossover_Signal'] == 1) & (df['Momentum_Signal'] != 1), 'Combined_Signal'] = 1  # Moderate Buy
    df.loc[(df['VWMA_Crossover_Signal'] == -1) & (df['Momentum_Signal'] != -1), 'Combined_Signal'] = -1  # Moderate Sell

    return df['Combined_Signal']
```
