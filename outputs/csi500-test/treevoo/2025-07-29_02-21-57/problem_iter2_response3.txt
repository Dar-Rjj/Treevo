```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_window = 10
    long_window = 30
    df['Volume_Close'] = df['volume'] * df['close']
    df['Short_VWMA'] = df['Volume_Close'].rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    df['Long_VWMA'] = df['Volume_Close'].rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['Close_Return_5D'] = df['close'].pct_change(periods=5)
    df['Volume_Ratio_5D'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Vol_Adjusted_Mom'] = df['Close_Return_5D'] * df['Volume_Ratio_5D']
    
    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['HL_Range_5D_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['HL_Range_5D_Std'] = df['High_Low_Range'].rolling(window=5).std()
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['HL_Range_5D_Mean']) / df['HL_Range_5D_Std']
    
    # Generate Signals
    df['VWMA_Signal'] = 0
    df.loc[df['Short_VWMA'] > df['Long_VWMA'], 'VWMA_Signal'] = 1
    df.loc[df['Short_VWMA'] < df['Long_VWMA'], 'VWMA_Signal'] = -1
    
    df['Momentum_Signal'] = 0
    df.loc[df['Vol_Adjusted_Mom'] > 0, 'Momentum_Signal'] = 1
    df.loc[df['Vol_Adjusted_Mom'] < 0, 'Momentum_Signal'] = -1
    
    # Combine Signals
    df['Combined_Signal'] = 0
    df.loc[(df['VWMA_Signal'] == 1) & (df['Momentum_Signal'] == 1), 'Combined_Signal'] = 2
    df.loc[(df['VWMA_Signal'] == -1) & (df['Momentum_Signal'] == -1), 'Combined_Signal'] = -2
    df.loc[(df['VWMA_Signal'] == 1) & (df['Momentum_Signal'] != 1), 'Combined_Signal'] = 1
    df.loc[(df['VWMA_Signal'] == -1) & (df['Momentum_Signal'] != -1), 'Combined_Signal'] = -1
    
    return df['Combined_Signal']
```
