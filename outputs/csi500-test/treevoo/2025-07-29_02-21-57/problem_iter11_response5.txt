```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the percentage of the close price relative to the high and low
    df['close_to_high'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the momentum of the price change over a 5-day period
    df['momentum_5_day'] = df['close'].pct_change(5)
    
    # Calculate the ratio of the amount to the volume
    df['amount_to_volume'] = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['price_change'] * df['close_to_high'] * df['momentum_5_day'] * df['amount_to_volume']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
