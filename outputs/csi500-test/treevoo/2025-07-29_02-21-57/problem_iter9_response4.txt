```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['5D_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5D_Std_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).std()
    df['5D_Mean_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Price_Oscillation'] = df['High_Low_Range'] - df['5D_Mean_High_Low_Range']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Price and Range Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    df['Daily_Range'] = df['high'] - df['low']
    df['Combined_Momentum'] = df['Price_Momentum'] + df['Daily_Range']

    # Apply Exponential Moving Average (EMA) Smoothing
    alpha = 0.3
    df['EMA_Smoothed_Combined_Momentum'] = df['Combined_Momentum'].ewm(alpha=alpha, adjust=False).mean()

    # Identify Breakouts
    df['Breakout'] = 0
    df.loc[df['close'] > df['high'].shift(1), 'Breakout'] = 1  # Upward Breakout
    df.loc[df['close'] < df['low'].shift(1), 'Breakout'] = -1  # Downward Breakout

    # Calculate Breakout Strength
    df['Breakout_Strength'] = 0
    df.loc[df['Breakout'] == 1, 'Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['Daily_Range'].rolling(window=20).mean()
    df.loc[df['Breakout'] == -1, 'Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['Daily_Range'].rolling(window=20).mean()

    # Cumulate Breakout Strength
    df['Cumulative_Breakout_Strength'] = df['Breakout_Strength'].rolling(window=10).sum()

    # Weight by Volume
    df['Volume_Adjusted_Combined_Momentum'] = df['Combined_Momentum'] * df['volume']

    # Combine Volume Adjusted Momentum, ATR, and EMA Smoothed Combined Momentum
    df['Combined_Factor'] = df['Volume_Adjusted_Momentum'] + df['ATR'] + df['EMA_Smoothed_Combined_Momentum']

    # Generate Alpha Factor
    df['Alpha_Factor'] = (df['Combined_Momentum'] - df['Price_Oscillation']) * df['Cumulative_Breakout_Strength']

    return df['Alpha_Factor']
```
