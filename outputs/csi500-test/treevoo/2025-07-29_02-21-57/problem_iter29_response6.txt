```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term VWMA (5 days)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term VWMA (14 days)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Generate Crossover Signal
    crossover_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_price_return_5d = (df['close'] / df['close'].shift(5)) - 1
    volume_sum_5d = df['volume'].rolling(window=5).sum()
    volume_adjusted_momentum_5d = close_price_return_5d * volume_sum_5d
    
    # Evaluate Volume Adjusted Momentum
    momentum_signal = (volume_adjusted_momentum_5d > 0).astype(int) - (volume_adjusted_momentum_5d < 0).astype(int)
    
    # Combine Crossover and Momentum Signals
    combined_signal = pd.Series(index=df.index, dtype=float)
    combined_signal[(crossover_signal == 1) & (momentum_signal == 1)] = 1
    combined_signal[(crossover_signal == -1) & (momentum_signal == -1)] = -1
    combined_signal[(crossover_signal == 1) & (momentum_signal != -1)] = 0.5
    combined_signal[(crossover_signal == -1) & (momentum_signal != 1)] = -0.5
    combined_signal[~((crossover_signal == 1) | (crossover_signal == -1))] = 0
    
    # Calculate True Range
    true_range = pd.DataFrame({
        'high_low': df['high'] - df['low'],
        'high_prev_close': abs(df['high'] - df['close'].shift()),
        'low_prev_close': abs(df['low'] - df['close'].shift())
    }).max(axis=1)
    
    # Calculate 14-Period ATR
    atr_14 = true_range.rolling(window=14).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(14)
    
    # Adjust Momentum by ATR
    atr_adjusted_momentum = price_momentum / atr_14
    
    # Generate Alpha Factor
    alpha_factor = atr_adjusted_momentum + (df['close'] - df['open'])
    final_alpha_factor = pd.Series(index=df.index, dtype=float)

    final_alpha_factor[(alpha_factor > 0) & (combined_signal == 1)] = 1
    final_alpha_factor[(alpha_factor < 0) & (combined_signal == -1)] = -1
    final_alpha_factor[(alpha_factor > 0) & (combined_signal != -1)] = 0.5
    final_alpha_factor[(alpha_factor < 0) & (combined_signal != 1)] = -0.5
    final_alpha_factor[~((alpha_factor > 0) | (alpha_factor < 0))] = 0

    return final_alpha_factor
```
