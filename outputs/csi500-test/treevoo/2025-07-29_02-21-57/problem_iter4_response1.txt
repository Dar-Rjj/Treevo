```python
import pandas as pd

def heuristics_v2(df, N=20, alpha=0.3):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(N)
    
    # Initialize EMA with the first raw momentum value
    df['ema_smoothed_momentum'] = 0
    df.iloc[N, df.columns.get_loc('ema_smoothed_momentum')] = df.iloc[N]['raw_momentum']
    
    # Apply EMA Smoothing
    for i in range(N+1, len(df)):
        ema_yesterday = df.iloc[i-1]['ema_smoothed_momentum']
        raw_momentum_today = df.iloc[i]['raw_momentum']
        df.at[df.index[i], 'ema_smoothed_momentum'] = (raw_momentum_today * alpha) + (ema_yesterday * (1 - alpha))
    
    return df['ema_smoothed_momentum']
```
