```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayCloseReturn'] = (df['close'] / df['close'].shift(5)) - 1
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5DayVolumeAdjustedMomentum'] = df['5DayCloseReturn'] * df['5DayVolumeRatio']

    # Calculate High-Low Price Oscillation
    df['HighLowRange'] = df['high'] - df['low']
    df['5DayStdHL'] = df['HighLowRange'].rolling(window=5).std()
    df['5DayMeanHL'] = df['HighLowRange'].rolling(window=5).mean()
    df['PriceOscillation'] = df['HighLowRange'] - df['5DayMeanHL']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR14'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['VolumeWeightedMomentum'] = df['PriceMomentum'] * df['volume']

    # Combine Volume Adjusted Momentum and ATR Momentum
    df['CombinedMomentum'] = df['5DayVolumeAdjustedMomentum'] + df['VolumeWeightedMomentum'] / df['ATR14']

    # Generate Alpha Factor
    df['AlphaFactor'] = df['CombinedMomentum'] + df['PriceOscillation']

    return df['AlphaFactor'].dropna()
```
