```python
def heuristics_v2(data):
    """
    Generate an alpha factor based on the given market data.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
        pd.Series: A Series with index as date and values as the generated alpha factor.
    """
    # Calculate the price range
    price_range = data['high'] - data['low']
    
    # Calculate the buying pressure
    buying_pressure = (data['close'] - data['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (data['high'] - data['close']) / price_range
    
    # Calculate the volume weighted average price
    vwap = (data['amount'] / data['volume']).rolling(window=5).mean()
    
    # Calculate the return of the closing price
    close_return = data['close'].pct_change()
    
    # Alpha factor: (buying pressure - selling pressure) * VWAP * return
    alpha_factor = (buying_pressure - selling_pressure) * vwap * close_return
    
    return alpha_factor
```
