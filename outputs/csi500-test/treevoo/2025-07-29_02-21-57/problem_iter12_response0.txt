```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5DayCloseReturn'] = (df['close'] / df['close'].shift(5)) - 1
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5DayVolAdjMomentum'] = df['5DayCloseReturn'] * df['5DayVolumeRatio']

    # Calculate High-Low Price Oscillation
    df['HighLowRange'] = df['high'] - df['low']
    df['5DayHLStd'] = df['HighLowRange'].rolling(window=5).std()
    df['PriceOscillation'] = df['HighLowRange'] - df['HighLowRange'].rolling(window=5).mean()

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR14'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate Price and Range Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(20)
    df['DailyRange'] = df['high'] - df['low']
    df['CombinedMomentum'] = df['PriceMomentum'] + df['DailyRange']

    # Apply Exponential Moving Average (EMA) Smoothing
    alpha = 0.3
    df['EMA_Smoothed'] = df['CombinedMomentum'].ewm(alpha=alpha, adjust=False).mean()

    # Identify Breakouts
    df['BreakoutType'] = 0
    df.loc[df['close'] > df['high'].shift(1), 'BreakoutType'] = 1  # Upward Breakout
    df.loc[df['close'] < df['low'].shift(1), 'BreakoutType'] = -1  # Downward Breakout

    # Calculate Breakout Strength
    df['20DayAvgRange'] = df['DailyRange'].rolling(window=20).mean()
    df['BreakoutStrength'] = 0
    df.loc[df['BreakoutType'] == 1, 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgRange']
    df.loc[df['BreakoutType'] == -1, 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgRange']

    # Cumulate Breakout Strength
    df['CumulativeBreakoutStrength'] = df['BreakoutStrength'].rolling(window=10).apply(lambda x: x[x > 0].sum() - x[x < 0].sum(), raw=True)

    # Weight by Volume
    df['WeightedMomentum'] = df['CombinedMomentum'] * df['volume']

    # Combine Volume Adjusted Momentum, ATR, and EMA Smoothed Combined Momentum
    df['CombinedFactor'] = df['5DayVolAdjMomentum'] + df['ATR14'] + df['EMA_Smoothed']

    # Generate Alpha Factor
    df['AlphaFactor'] = (df['CombinedMomentum'] - df['PriceOscillation']) * df['CumulativeBreakoutStrength'] * df['WeightedMomentum'] + df['CombinedFactor']

    return df['AlphaFactor']
```
