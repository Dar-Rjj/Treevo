```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Series
    intraday_returns = (df['close'] - df['open']) / df['open']
    
    # Compute Short-Term Moving Average of Intraday Returns (5 days)
    short_term_ma = intraday_returns.rolling(window=5).mean()
    
    # Compute Long-Term Moving Average of Intraday Returns (20 days)
    long_term_ma = intraday_returns.rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    intraday_return_diff = long_term_ma - short_term_ma
    
    # Compute Volume Change
    volume_change = df['volume'].diff()
    
    # Apply Condition: If Volume Change is positive, use Intraday Return Difference; if negative, set to 0
    factor_values = intraday_return_diff.where(volume_change > 0, 0)
    
    return factor_values
```
