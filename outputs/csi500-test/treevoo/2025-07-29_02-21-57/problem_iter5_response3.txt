```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_close_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5_day_volume_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5_day_vol_adj_momentum'] = df['5_day_close_return'] * df['5_day_volume_ratio']

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['5_day_high_low_mean'] = df['high_low_range'].rolling(window=5).mean()
    df['5_day_high_low_std'] = df['high_low_range'].rolling(window=5).std()
    df['price_oscillation'] = (df['high_low_range'] - df['5_day_high_low_mean']) / df['5_day_high_low_std']

    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()

    # Calculate Price and Range Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    df['daily_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['price_momentum'] + df['daily_range']

    # Apply Exponential Moving Average (EMA) Smoothing
    alpha = 0.3
    df['ema_combined_momentum'] = df['combined_momentum'].ewm(alpha=alpha, adjust=False).mean()

    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    df['breakout_strength_up'] = (df['close'] - df['high'].shift(1)) / df['avg_daily_range_20']
    df['breakout_strength_down'] = (df['low'].shift(1) - df['close']) / df['avg_daily_range_20']
    df['cumulative_breakout_strength'] = df['breakout_strength_up'].rolling(window=10).sum() - df['breakout_strength_down'].rolling(window=10).sum()

    # Weight by Volume
    df['weighted_combined_momentum'] = df['volume'] * df['combined_momentum']

    # Combine Volume Adjusted Momentum, ATR, and EMA Smoothed Combined Momentum
    df['combined_factor'] = df['5_day_vol_adj_momentum'] + df['atr_14'] + df['ema_combined_momentum']

    # Generate Alpha Factor
    df['alpha_factor'] = (df['combined_momentum'] + df['price_oscillation']) * df['cumulative_breakout_strength']

    return df['alpha_factor']
```
