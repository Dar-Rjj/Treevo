```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['PreviousClose'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['PreviousClose']).abs()
    df['TR3'] = (df['low'] - df['PreviousClose']).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['DailyRange'] = df['high'] - df['low']

    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['14DayATR'] = df['TrueRange'].rolling(window=14).mean()
    df['20DayAvgDailyRange'] = df['DailyRange'].rolling(window=20).mean()

    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['UpwardStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgDailyRange']
    df['DownwardStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgDailyRange']
    df['BreakoutStrength'] = df['UpwardStrength'] * df['UpwardBreakout'] + df['DownwardStrength'] * df['DownwardBreakout']

    # Cumulate Breakout Strength
    df['PositiveCumulatedStrength'] = df['BreakoutStrength'].where(df['BreakoutStrength'] > 0, 0).rolling(window=10).sum()
    df['NegativeCumulatedStrength'] = df['BreakoutStrength'].where(df['BreakoutStrength'] < 0, 0).rolling(window=10).sum()
    df['CumulatedBreakoutStrength'] = df['PositiveCumulatedStrength'] - (-df['NegativeCumulatedStrength'])

    # Calculate 10-Day Volume Adjusted Momentum
    df['10DayClosePriceReturn'] = (df['close'] / df['close'].shift(9)) - 1
    df['10DayVolumeRatio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10 * 21).mean()
    df['10DayVolumeAdjustedMomentum'] = df['10DayClosePriceReturn'] * df['10DayVolumeRatio']

    # Calculate High-Low Price Oscillation
    df['HighLowRange'] = df['high'] - df['low']
    df['20DayMeanHighLowRange'] = df['HighLowRange'].rolling(window=20).mean()
    df['20DayStdHighLowRange'] = df['HighLowRange'].rolling(window=20).std()
    df['AdjustedPriceOscillation'] = (df['HighLowRange'] - df['20DayMeanHighLowRange']) / df['20DayStdHighLowRange']

    # Calculate Intraday Volatility
    df['IntradayVolatility'] = (df['high'] - df['low']) / df['open']
    df['10DayMovingAverageIntradayVolatility'] = df['IntradayVolatility'].rolling(window=10).mean()

    # Combine Indicators
    df['AlphaFactor'] = (
        df['CumulatedBreakoutStrength'] +
        df['10DayVolumeAdjustedMomentum'] +
        df['AdjustedPriceOscillation'] +
        df['10DayMovingAverageIntradayVolatility']
    )

    return df['AlphaFactor']
```
