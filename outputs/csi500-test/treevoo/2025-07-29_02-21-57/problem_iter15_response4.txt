```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_5_day_return = df['close'].pct_change(periods=5)
    volume_ratio = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = close_5_day_return * volume_ratio
    
    # Calculate High-Low Price Oscillation
    high_low_range = df['high'] - df['low']
    high_low_mean = high_low_range.rolling(window=5).mean()
    high_low_std = high_low_range.rolling(window=5).std()
    high_low_oscillation = high_low_range - high_low_mean
    
    # Calculate True Range
    prev_close = df['close'].shift(1)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - prev_close).abs()
    tr3 = (df['low'] - prev_close).abs()
    true_range = np.maximum.reduce([tr1, tr2, tr3])
    
    # Calculate 14-Period Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    weighted_momentum = price_momentum * df['volume']
    
    # Normalize by ATR
    normalized_momentum = weighted_momentum / atr
    
    # Generate Alpha Factor
    alpha_factor = normalized_momentum + volume_adjusted_momentum + high_low_oscillation
    
    # Calculate Intraday Return Series
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Averages of Intraday Returns
    short_term_intraday_ma = intraday_return.rolling(window=5).mean()
    long_term_intraday_ma = intraday_return.rolling(window=20).mean()
    
    # Compute Intraday Return Difference
    intraday_return_diff = short_term_intraday_ma - long_term_intraday_ma
    
    # Filter by Volume Change
    volume_change = df['volume'].pct_change()
    intraday_sentiment = intraday_return_diff.where(volume_change > 0, 0)
    
    # Combined Crossover and Momentum Signal
    vwma_crossover = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    intraday_signal = (intraday_sentiment > 0).astype(int) - (intraday_sentiment < 0).astype(int)
    
    combined_signal = vwma_crossover + momentum_signal + intraday_signal
    combined_signal = combined_signal.map({3: 'Strong Buy', -3: 'Strong Sell', 2: 'Moderate Buy', -2: 'Moderate Sell', 1: 'Hold', -1: 'Hold', 0: 'Hold'})
    
    return alpha_factor, combined_signal
```
