```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume-Weighted Close Price Return
    df['5DayClosePriceReturn'] = (df['close'] / df['close'].shift(5)) - 1
    df['5DayVolumeRatio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5DayVolumeWeightedCloseReturn'] = df['5DayClosePriceReturn'] * df['5DayVolumeRatio']

    # Calculate High-Low Price Oscillation with Exponential Smoothing
    df['HighLowRange'] = df['high'] - df['low']
    df['ExpSmoothHighLowRange'] = df['HighLowRange'].ewm(alpha=0.2, adjust=False).mean()
    df['HighLowOscillation'] = df['HighLowRange'] - df['ExpSmoothHighLowRange']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['WeightedMomentum'] = df['PriceMomentum'] * df['volume']

    # Generate Alpha Factor
    df['AlphaFactor'] = df['WeightedMomentum'] + df['5DayVolumeWeightedCloseReturn'] + df['HighLowOscillation']

    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']

    # Calculate 20-Day Moving Average of Daily Range
    df['20DayAvgDailyRange'] = df['DailyRange'].rolling(window=20).mean()

    # Identify Breakouts
    df['UpwardBreakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['DownwardBreakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['BreakoutStrength'] = 0
    df.loc[df['UpwardBreakout'] == 1, 'BreakoutStrength'] = (df['close'] - df['high'].shift(1)) / df['20DayAvgDailyRange']
    df.loc[df['DownwardBreakout'] == 1, 'BreakoutStrength'] = (df['low'].shift(1) - df['close']) / df['20DayAvgDailyRange']

    # Cumulate Breakout Strength
    df['CumulatedBreakoutStrength'] = df['BreakoutStrength'].rolling(window=10).apply(lambda x: x[x > 0].sum() - x[x < 0].sum(), raw=True)

    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open']) / df['open']

    # Combine Indicators
    df['CombinedIndicators'] = df['IntradayHighLowSpread'] + df['CloseToOpenReturn'] + df['CumulatedBreakoutStrength']

    # Add to Alpha Factor
    df['AlphaFactor'] += df['CombinedIndicators']

    return df['AlphaFactor']
```
