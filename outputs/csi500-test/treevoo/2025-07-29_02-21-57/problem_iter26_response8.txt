```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Prev_Close']).abs()
    df['TR3'] = (df['low'] - df['Prev_Close']).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['Daily_Range'] = df['high'] - df['low']

    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()

    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['Breakout_Strength_Up'] = ((df['close'] - df['high'].shift(1)) / df['Avg_Daily_Range_20']).where(df['Upward_Breakout'] == 1, 0)
    df['Breakout_Strength_Down'] = ((df['low'].shift(1) - df['close']) / df['Avg_Daily_Range_20']).where(df['Downward_Breakout'] == 1, 0)
    df['Cumulative_Breakout_Strength'] = df['Breakout_Strength_Up'].rolling(window=10).sum() - df['Breakout_Strength_Down'].rolling(window=10).sum()

    # Calculate 10-Day Volume Adjusted Momentum
    df['Close_Return_10'] = (df['close'] - df['close'].shift(9)) / df['close'].shift(9)
    df['Volume_Ratio_10'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Momentum'] = df['Close_Return_10'] * df['Volume_Ratio_10']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    df['Avg_Intraday_Volatility_10'] = df['Intraday_Volatility'].rolling(window=10).mean()

    # Adjust Momentum by ATR
    df['Weighted_Momentum'] = (df['close'] - df['close'].shift(14)) * df['volume']
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR']

    # Generate Alpha Factor
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Close_to_Open_Return']
    df['Alpha_Factor'] += df['Cumulative_Breakout_Strength'] + df['Volume_Adjusted_Momentum'] + df['Avg_Intraday_Volatility_10']

    # Incorporate Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Surge'] = (df['volume'] > df['Volume_MA_10']).astype(float)
    df['Volume_Decline'] = (df['volume'] < df['Volume_MA_10']).astype(float)
    df['Volume_Adjustment'] = df['Volume_Surge'] * 1.5 + df['Volume_Decline'] * 0.5
    df['Alpha_Factor'] *= df['Volume_Adjustment']

    return df['Alpha_Factor']
```
