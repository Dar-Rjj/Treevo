```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5-day) Volume Weighted Moving Average (VWMA)
    df['short_term_vwma'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Long-Term (20-day) Volume Weighted Moving Average (VWMA)
    df['long_term_vwma'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_close_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5_day_volume_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()
    df['volume_adjusted_momentum'] = df['5_day_close_return'] * df['5_day_volume_ratio']

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['5_day_high_low_mean'] = df['high_low_range'].rolling(window=5).mean()
    df['high_low_oscillation'] = df['high_low_range'] - df['5_day_high_low_mean']

    # Generate signals
    df['vwma_crossover_signal'] = (df['short_term_vwma'] > df['long_term_vwma']).astype(int) - (df['short_term_vwma'] < df['long_term_vwma']).astype(int)
    df['volume_adjusted_momentum_signal'] = (df['volume_adjusted_momentum'] > 0).astype(int) - (df['volume_adjusted_momentum'] < 0).astype(int)
    df['high_low_oscillation_signal'] = (df['high_low_oscillation'] > 0).astype(int) - (df['high_low_oscillation'] < 0).astype(int)

    # Combine signals
    df['combined_signal'] = df['vwma_crossover_signal'] + df['volume_adjusted_momentum_signal'] + df['high_low_oscillation_signal']

    return df['combined_signal']
```
