```python
def heuristics_v2(df):
    # Calculate the difference between the close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume, indicating average price
    avg_price = df['amount'] / df['volume']
    
    # Determine the relative position of the close price within the day's range
    relative_close_position = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine the factors with a weighted sum
    factor = 0.5 * price_diff + 0.3 * avg_price + 0.2 * relative_close_position
    
    return factor
```
