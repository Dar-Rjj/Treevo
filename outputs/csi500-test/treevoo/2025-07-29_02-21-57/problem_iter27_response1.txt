```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_window = 10
    df['Short_VWMA'] = (df['close'] * df['volume']).rolling(window=short_window).sum() / df['volume'].rolling(window=short_window).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_window = 30
    df['Long_VWMA'] = (df['close'] * df['volume']).rolling(window=long_window).sum() / df['volume'].rolling(window=long_window).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'].pct_change(periods=5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate 10-Day Price Range
    df['10D_High'] = df['high'].rolling(window=10).max()
    df['10D_Low'] = df['low'].rolling(window=10).min()

    # Generate VWMA Crossover Signal
    df['VWMA_Crossover_Signal'] = 0
    df.loc[df['Short_VWMA'] > df['Long_VWMA'], 'VWMA_Crossover_Signal'] = 1
    df.loc[df['Short_VWMA'] < df['Long_VWMA'], 'VWMA_Crossover_Signal'] = -1

    # Generate Volume Adjusted Momentum Signal
    df['Momentum_Signal'] = 0
    df.loc[df['Volume_Adjusted_Momentum'] > 0, 'Momentum_Signal'] = 1
    df.loc[df['Volume_Adjusted_Momentum'] < 0, 'Momentum_Signal'] = -1

    # Generate 10-Day Price Range Signal
    df['Price_Range_Signal'] = 0
    close_within_20pct_high = (df['close'] > df['10D_High'] * 0.8) & (df['close'] <= df['10D_High'])
    close_within_20pct_low = (df['close'] >= df['10D_Low']) & (df['close'] < df['10D_Low'] * 1.2)
    df.loc[close_within_20pct_high, 'Price_Range_Signal'] = -1
    df.loc[close_within_20pct_low, 'Price_Range_Signal'] = 1

    # Combine Signals
    df['Combined_Signal'] = 0
    buy_signals = (df['VWMA_Crossover_Signal'] == 1) & (df['Momentum_Signal'] == 1) & (df['Price_Range_Signal'] == 1)
    sell_signals = (df['VWMA_Crossover_Signal'] == -1) & (df['Momentum_Signal'] == -1) & (df['Price_Range_Signal'] == -1)
    moderate_buy_signals = (df[['VWMA_Crossover_Signal', 'Momentum_Signal', 'Price_Range_Signal']].apply(lambda x: sum(x == 1), axis=1) == 2)
    moderate_sell_signals = (df[['VWMA_Crossover_Signal', 'Momentum_Signal', 'Price_Range_Signal']].apply(lambda x: sum(x == -1), axis=1) == 2)

    df.loc[buy_signals, 'Combined_Signal'] = 2
    df.loc[sell_signals, 'Combined_Signal'] = -2
    df.loc[moderate_buy_signals, 'Combined_Signal'] = 1
    df.loc[moderate_sell_signals, 'Combined_Signal'] = -1

    return df['Combined_Signal']
```
