```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (10-day) Volume Weighted Moving Average (VWMA)
    short_term = 10
    df['Short_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=short_term).sum() / df['volume'].rolling(window=short_term).sum()

    # Calculate Long-Term (50-day) Volume Weighted Moving Average (VWMA)
    long_term = 50
    df['Long_Term_VWMA'] = (df['close'] * df['volume']).rolling(window=long_term).sum() / df['volume'].rolling(window=long_term).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5_Day_Close_Price_Return'] = df['close'].pct_change(5)
    df['5_Day_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=50).mean()
    df['Volume_Adjusted_Momentum'] = df['5_Day_Close_Price_Return'] * df['5_Day_Volume_Ratio']

    # Detect Volume Spikes
    df['20_Day_Avg_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Spike'] = (df['volume'] > 2 * df['20_Day_Avg_Volume']).astype(int)

    # Calculate 10-Day Price Oscillation
    df['10_Day_High_Low_Range'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    df['10_Day_Mean_Close'] = df['close'].rolling(window=10).mean()
    df['Price_Oscillation'] = (df['10_Day_High_Low_Range'] / df['10_Day_Mean_Close']) > 0.1

    # Combined Crossover, Momentum, Price Oscillation, and Volume Spike Signal
    def generate_signal(row):
        vwma_crossover = 1 if row['Short_Term_VWMA'] > row['Long_Term_VWMA'] else -1 if row['Short_Term_VWMA'] < row['Long_Term_VWMA'] else 0
        momentum = 1 if row['Volume_Adjusted_Momentum'] > 0 else -1 if row['Volume_Adjusted_Momentum'] < 0 else 0
        oscillation = 1 if row['Price_Oscillation'] else 0
        volume_spike = row['Volume_Spike']
        
        if vwma_crossover == 1 and momentum == 1 and oscillation == 1:
            return 3  # strong Buy Signal
        elif vwma_crossover == -1 and momentum == -1 and oscillation == 1:
            return -3  # strong Sell Signal
        elif vwma_crossover == 1 and momentum == 1:
            return 2 + volume_spike  # moderate Buy Signal
        elif vwma_crossover == -1 and momentum == -1:
            return -2 - volume_spike  # moderate Sell Signal
        elif vwma_crossover == 1 and momentum == 0:
            return 1 + volume_spike  # mild Buy Signal
        elif vwma_crossover == -1 and momentum == 0:
            return -1 - volume_spike  # mild Sell Signal
        else:
            return 0  # hold position

    df['Signal'] = df.apply(generate_signal, axis=1)

    return df['Signal']
```
