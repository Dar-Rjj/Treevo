```python
import pandas as pd

def heuristics_v2(df):
    # Calculate open to close price change
    open_to_close = df['close'] - df['open']

    # Calculate high to low price difference
    high_to_low = df['high'] - df['low']

    # Calculate moving average crossover (short-term 5 days, long-term 20 days)
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    ma_crossover = short_term_ma - long_term_ma

    # Calculate momentum (n=10 days)
    momentum = (df['close'] / df['close'].shift(10)) - 1

    # Calculate true range
    prev_close = df['close'].shift(1)
    true_range = (df['high'] - df['low']).combine((df['high'] - prev_close).abs(), max).combine((df['low'] - prev_close).abs(), max)

    # Calculate On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[0] = 0
    for i in range(1, len(obv)):
        if df['close'][i] > df['close'][i-1]:
            obv[i] = obv[i-1] + df['volume'][i]
        elif df['close'][i] < df['close'][i-1]:
            obv[i] = obv[i-1] - df['volume'][i]
        else:
            obv[i] = obv[i-1]

    # Calculate volume change (n=10 days)
    volume_change = (df['volume'] / df['volume'].shift(10)) - 1

    # Calculate average transaction size
    avg_transaction_size = df['amount'] / df['volume']

    # Combine all alpha factors into a single DataFrame
    alpha_factors = pd.DataFrame({
        'OpenToClose': open_to_close,
        'HighToLow': high_to_low,
        'MACrossover': ma_crossover,
        'Momentum': momentum,
        'TrueRange': true_range,
        'OBV': obv,
        'VolumeChange': volume_change,
        'AvgTransactionSize': avg_transaction_size
    })

    # For simplicity, let's assume we sum all the factors to create a final factor
    final_factor = alpha_factors.sum(axis=1)

    return final_factor
```
