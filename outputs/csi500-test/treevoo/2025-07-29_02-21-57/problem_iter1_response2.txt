```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily return using close price
    df['daily_return'] = df['close'].pct_change()

    # Define lookback window for momentum
    lookback_window_momentum = 20

    # Aggregate returns over the period to compute momentum
    df['momentum'] = df['daily_return'].rolling(window=lookback_window_momentum).sum()

    # Compute average volume
    lookback_window_volume = 60
    df['avg_volume'] = df['volume'].rolling(window=lookback_window_volume).mean()

    # Set threshold for high volume
    high_volume_threshold = 1.5 * df['avg_volume']

    # Identify high volume days
    df['high_volume_day'] = (df['volume'] > high_volume_threshold).astype(int)

    # Filter momentum by high volume days
    df['adjusted_daily_return'] = df['daily_return'] * df['high_volume_day']
    
    # Define lookback window for adjusted momentum
    lookback_window_adjusted_momentum = 20

    # Aggregate adjusted returns over the period to compute adjusted momentum
    df['adjusted_momentum'] = df['adjusted_daily_return'].rolling(window=lookback_window_adjusted_momentum).sum()

    # Return the factor values
    return df['adjusted_momentum']
```
