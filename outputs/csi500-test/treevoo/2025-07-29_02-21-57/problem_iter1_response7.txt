```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].pct_change(5)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].pct_change(20)
    
    # Integrate Momentum
    momentum = 0.7 * short_term_momentum + 0.3 * long_term_momentum
    
    # Calculate Volume-to-Price Ratio
    volume_to_price_ratio = df['volume'] / df['close']
    
    # Calculate Average True Range
    high_low = df['high'] - df['low']
    high_close = (df['high'] - df['close'].shift(1)).abs()
    low_close = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
    average_true_range = true_range.rolling(window=14).mean()
    
    # Integrate Liquidity
    liquidity = 0.6 * volume_to_price_ratio + 0.4 * average_true_range
    
    # Combine Momentum and Liquidity
    final_alpha_factor = 0.5 * momentum + 0.5 * liquidity
    
    return final_alpha_factor
```
