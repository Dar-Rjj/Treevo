def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Previous Close']).abs()
    df['TR3'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True Range'].rolling(window=14).sum() / 14

    # Calculate Close-to-Open Return
    df['Close to Open Return'] = df['close'] - df['open']

    # Calculate Price Momentum
    df['Price Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Volume Weighted Momentum'] = df['volume'] * df['Price Momentum']

    # Adjust Momentum by ATR
    df['ATR Adjusted Momentum'] = df['Volume Weighted Momentum'] / df['ATR']

    # Generate Alpha Factor
    df['Alpha Factor'] = df['ATR Adjusted Momentum'] + df['Close to Open Return']

    return df['Alpha Factor'].dropna()
