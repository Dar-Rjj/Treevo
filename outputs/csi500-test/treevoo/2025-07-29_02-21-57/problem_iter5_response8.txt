```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['avg_daily_range'] = df['daily_range'].rolling(window=22).mean()
    
    # Identify Breakouts
    df['up_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['down_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['up_breakout_strength'] = (df['close'] - df['high'].shift(1)) / df['avg_daily_range']
    df['down_breakout_strength'] = (df['low'].shift(1) - df['close']) / df['avg_daily_range']
    
    # Cumulate Breakout Strength over the past 10 days
    df['cum_up_breakout_strength'] = df['up_breakout_strength'].rolling(window=10).sum()
    df['cum_down_breakout_strength'] = df['down_breakout_strength'].rolling(window=10).sum()
    df['cum_breakout_strength'] = df['cum_up_breakout_strength'] - df['cum_down_breakout_strength']
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['close_t_minus_5'] = df['close'].shift(5)
    df['5_day_return'] = (df['close'] - df['close_t_minus_5']) / df['close_t_minus_5']
    df['5_day_volume'] = df['volume'].rolling(window=5).sum()
    df['avg_5_day_volume'] = df['5_day_volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['5_day_volume'] / df['avg_5_day_volume']
    df['vam'] = df['5_day_return'] * df['volume_ratio']
    
    # Calculate High-Low Price Oscillation
    df['hl_range'] = df['high'] - df['low']
    df['5_day_std_hl_range'] = df['hl_range'].rolling(window=5).std()
    df['5_day_mean_hl_range'] = df['hl_range'].rolling(window=5).mean()
    df['adjusted_hl_range'] = (df['hl_range'] - df['5_day_mean_hl_range']) / df['5_day_std_hl_range']
    df['po'] = df['adjusted_hl_range'].rolling(window=5).mean()
    
    # Integrate Volume-Adjusted Momentum and Price Oscillation
    vam_weight = 0.7
    po_weight = 0.3
    df['integrated_momentum_oscillation'] = vam_weight * df['vam'] + po_weight * df['po']
    
    # Combine Integrated Components with Breakout Strength
    integrated_weight = 0.6
    breakout_strength_weight = 0.4
    df['alpha_factor'] = (integrated_weight * df['integrated_momentum_oscillation'] 
                          + breakout_strength_weight * df['cum_breakout_strength'])
    
    return df['alpha_factor']
```
