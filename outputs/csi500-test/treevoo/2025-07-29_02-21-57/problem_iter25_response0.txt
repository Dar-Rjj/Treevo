```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['breakout_strength'] = 0
    df.loc[df['upward_breakout'] == 1, 'breakout_strength'] = (df['close'] - df['high'].shift(1)) / df['daily_range']
    df.loc[df['downward_breakout'] == 1, 'breakout_strength'] = (df['low'].shift(1) - df['close']) / df['daily_range']
    
    # Cumulate Breakout Strength
    df['positive_bs_sum'] = df['breakout_strength'].apply(lambda x: max(x, 0)).rolling(window=5).sum()
    df['negative_bs_sum'] = df['breakout_strength'].apply(lambda x: max(x, 0) * -1).rolling(window=5).sum()
    df['cumulated_breakout_strength'] = df['positive_bs_sum'] - df['negative_bs_sum']
    
    # Calculate 10-Day Volume Adjusted Momentum
    df['10_day_return'] = (df['close'] - df['close'].shift(9)) / df['close'].shift(9)
    df['10_day_volume_ratio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    df['10_day_vol_adj_momentum'] = df['10_day_return'] * df['10_day_volume_ratio']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    df['10_day_intraday_volatility_ma'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Combine Indicators
    df['alpha_factor'] = df['cumulated_breakout_strength'] + df['10_day_vol_adj_momentum'] + df['10_day_intraday_volatility_ma']
    
    return df['alpha_factor']
```
