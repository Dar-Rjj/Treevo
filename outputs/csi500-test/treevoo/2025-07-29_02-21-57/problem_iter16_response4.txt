```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'].pct_change(5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['10D_High_Low_Std'] = df['High_Low_Range'].rolling(window=10).std()
    df['10D_High_Low_Mean'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Price_Oscillation_Adjusted'] = (df['High_Low_Range'] - df['10D_High_Low_Mean']) / df['10D_High_Low_Std']

    # Compute Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Compute Moving Averages
    df['Short_Term_MA'] = df['Intraday_Return'].rolling(window=5).mean()
    df['Long_Term_MA'] = df['Intraday_Return'].rolling(window=20).mean()

    # Compute Intraday Return Difference
    df['Intraday_Return_Diff'] = df['Short_Term_MA'] - df['Long_Term_MA']

    # Filter by Volume Change
    df['Volume_Change'] = df['volume'].diff()
    df['Filtered_Intraday_Return_Diff'] = df.apply(lambda row: row['Intraday_Return_Diff'] if row['Volume_Change'] > 0 else 0, axis=1)

    # Calculate Short-Term and Long-Term VWMA
    def vwma(prices, volumes, window):
        return (prices * volumes).rolling(window=window).sum() / volumes.rolling(window=window).sum()

    df['Short_Term_VWMA'] = vwma(df['close'], df['volume'], 5)
    df['Long_Term_VWMA'] = vwma(df['close'], df['volume'], 20)

    # Generate Combined Signal
    def generate_signal(row):
        vwma_signal = 0
        if row['Short_Term_VWMA'] > row['Long_Term_VWMA']:
            vwma_signal = 1
        elif row['Short_Term_VWMA'] < row['Long_Term_VWMA']:
            vwma_signal = -1

        momentum_signal = 0
        if row['Volume_Adjusted_Momentum'] > 0:
            momentum_signal = 1
        elif row['Volume_Adjusted_Momentum'] < 0:
            momentum_signal = -1

        if vwma_signal == 1 and momentum_signal == 1:
            return 1
        elif vwma_signal == -1 and momentum_signal == -1:
            return -1
        elif vwma_signal == 1 and momentum_signal == 0:
            return 0.5
        elif vwma_signal == -1 and momentum_signal == 0:
            return -0.5
        else:
            return 0

    df['Alpha_Factor'] = df.apply(generate_signal, axis=1)
    return df['Alpha_Factor']
```
