```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5D_Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['5D_High_Low_STD'] = df['High_Low_Range'].rolling(window=5).std()
    df['5D_High_Low_Mean'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Price_Oscillation'] = (df['High_Low_Range'] - df['5D_High_Low_Mean']) / df['5D_High_Low_STD']

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['14D_ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['14D_Price_Momentum'] = (df['close'] - df['close'].shift(14))

    # Weight by Volume
    df['Volume_Weighted_Momentum'] = df['volume'] * df['14D_Price_Momentum']

    # Combine Volume Adjusted Momentum and ATR Momentum
    df['Combined_Momentum'] = df['5D_Volume_Adjusted_Momentum'] + df['14D_ATR']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['Combined_Momentum'] + df['Price_Oscillation']

    return df['Alpha_Factor']
```
