```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5-day) VWMA
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term (20-day) VWMA
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combined Crossover Signal
    crossover_signal = (short_term_vwma > long_term_vwma).astype(int) - (short_term_vwma < long_term_vwma).astype(int)
    
    # Calculate 5-Day Volume Adjusted Momentum
    close_price_return = df['close'].pct_change(periods=5)
    volume_ratio = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=50).mean()
    volume_adjusted_momentum = close_price_return * volume_ratio
    
    # Evaluate Volume Adjusted Momentum
    momentum_signal = (volume_adjusted_momentum > 0).astype(int) - (volume_adjusted_momentum < 0).astype(int)
    
    # Combine Crossover and Momentum Signals
    combined_signal_1 = (crossover_signal == 1) & (momentum_signal == 1)
    combined_signal_2 = (crossover_signal == -1) & (momentum_signal == -1)
    combined_signal_3 = (crossover_signal == 1) & (momentum_signal != 1)
    combined_signal_4 = (crossover_signal == -1) & (momentum_signal != -1)
    
    # Calculate High-Low Price Oscillation
    high_low_range = df['high'] - df['low']
    std_high_low = high_low_range.rolling(window=10).std()
    mean_high_low = high_low_range.rolling(window=10).mean()
    price_oscillation = (high_low_range - mean_high_low) / std_high_low
    
    # Incorporate Price Oscillation
    oscillation_signal = (price_oscillation > 0).astype(int) - (price_oscillation < 0).astype(int)
    
    # Generate Combined Signal
    final_signal = pd.Series(0, index=df.index)
    final_signal[combined_signal_1 & (oscillation_signal == 1)] = 1
    final_signal[combined_signal_2 & (oscillation_signal == -1)] = -1
    final_signal[(combined_signal_3 | ((crossover_signal == 1) & (oscillation_signal == 1)))] = 0.5
    final_signal[(combined_signal_4 | ((crossover_signal == -1) & (oscillation_signal == -1)))] = -0.5
    
    return final_signal
```
