```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Prev_Close'])
    df['TR3'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=50).mean()
    df['5D_Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']
    
    # Weight by Volume
    df['Weighted_5D_Volume_Adjusted_Momentum'] = df['volume'] * df['5D_Volume_Adjusted_Momentum']
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20D_Avg_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = 0
    df.loc[df['Upward_Breakout'] == 1, 'Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['20D_Avg_Daily_Range']
    df.loc[df['Downward_Breakout'] == 1, 'Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['20D_Avg_Daily_Range']
    
    # Cumulate Breakout Strength
    df['Positive_BS_Sum'] = df['Breakout_Strength'].where(df['Breakout_Strength'] > 0, 0).rolling(window=10).sum()
    df['Negative_BS_Sum'] = df['Breakout_Strength'].where(df['Breakout_Strength'] < 0, 0).rolling(window=10).sum()
    df['Cumulated_Breakout_Strength'] = df['Positive_BS_Sum'] - df['Negative_BS_Sum']
    
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(14))
    
    # Calculate Volume-Weighted Average Price
    df['VWAP'] = (df['open']*df['volume'] + df['high']*df['volume'] + df['low']*df['volume'] + df['close']*df['volume']) / (4 * df['volume'])
    
    # Combine Indicators
    df['Combined_Indicators'] = (df['Weighted_5D_Volume_Adjusted_Momentum'] + 
                                 df['Intraday_HL_Spread'] + 
                                 df['Close_to_Open_Return'] + 
                                 df['VWAP'] + 
                                 df['Cumulated_Breakout_Strength'] + 
                                 df['Price_Momentum']) / 6
    
    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_5D_Volume_Adjusted_Momentum'] / df['ATR']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['ATR_Adjusted_Momentum'] + 
                                df['Combined_Indicators'] + 
                                df['VWAP'])
    
    return df['Final_Alpha_Factor']
```
