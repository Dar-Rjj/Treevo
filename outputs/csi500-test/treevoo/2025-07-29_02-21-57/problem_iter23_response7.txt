def heuristics_v2(data):
    # Calculate the typical price as the average of high, low, and close
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    
    # Calculate the money flow by multiplying the typical price with the volume
    money_flow = typical_price * data['volume']
    
    # Calculate the 14-period exponential moving average of the typical price
    ema_typical_price = typical_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-period exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # The alpha factor is the ratio of the EMA of the money flow to the EMA of the typical price
    alpha_factor = ema_money_flow / ema_typical_price
    
    return alpha_factor
