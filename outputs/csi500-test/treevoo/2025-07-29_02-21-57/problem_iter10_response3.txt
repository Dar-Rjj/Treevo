import pandas as pd

def heuristics(data):
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    short_term_days = 5
    data['short_vwma'] = (data['close'] * data['volume']).rolling(window=short_term_days).sum() / data['volume'].rolling(window=short_term_days).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    long_term_days = 20
    data['long_vwma'] = (data['close'] * data['volume']).rolling(window=long_term_days).sum() / data['volume'].rolling(window=long_term_days).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    data['close_t_minus_5'] = data['close'].shift(5)
    data['price_return'] = (data['close'] - data['close_t_minus_5']) / data['close_t_minus_5']
    data['volume_sum'] = data['volume'].rolling(window=5).sum()
    data['average_5_day_volume'] = data['volume'].rolling(window=20).mean()
    data['volume_ratio'] = data['volume_sum'] / (data['average_5_day_volume'] * 5)
    data['volume_adjusted_momentum'] = data['price_return'] * data['volume_ratio']

    # Calculate High-Low Price Oscillation
    data['high_low_range'] = data['high'] - data['low']
    data['5_day_mean_high_low_range'] = data['high_low_range'].rolling(window=5).mean()
    data['price_oscillation'] = data['high_low_range'] - data['5_day_mean_high_low_range']

    # Generate signals based on VWMA Crossover, Volume Adjusted Momentum, and High-Low Price Oscillation
    data['vwma_crossover_signal'] = (data['short_vwma'] > data['long_vwma']).astype(int) - (data['short_vwma'] < data['long_vwma']).astype(int)
    data['volume_adjusted_momentum_signal'] = (data['volume_adjusted_momentum'] > 0).astype(int) - (data['volume_adjusted_momentum'] < 0).astype(int)
    data['price_oscillation_signal'] = (data['price_oscillation'] > 0).astype(int) - (data['price_oscillation'] < 0).astype(int)

    # Combine signals
    data['combined_signal'] = (data['vwma_crossover_signal'] + data['volume_adjusted_momentum_signal'] + data['price_oscillation_signal'])
    data['heuristic_factor'] = data['combined_signal'].apply(lambda x: 1 if x == 3 else (-1 if x == -3 else (0.5 if x == 2 else (-0.5 if x == -2 else 0))))

    return data['heuristic_factor']
