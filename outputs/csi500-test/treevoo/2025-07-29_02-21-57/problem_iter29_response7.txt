def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day average money flow
    avg_money_flow_14 = money_flow.rolling(window=14).mean()
    
    # Calculate the 14-day average volume
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = avg_money_flow_14 / avg_volume_14
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the alpha factor
    alpha_factor = money_flow_ratio * price_change
    
    return alpha_factor
