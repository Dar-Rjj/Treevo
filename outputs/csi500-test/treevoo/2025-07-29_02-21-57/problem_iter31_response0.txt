```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change from Close to Open price
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range and ATR
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs()
    df['true_range'] = df[['true_range', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    atr_14 = df['true_range'].rolling(window=14).mean()
    
    # Volume-Price Relationship: Bearish and Bullish Divergence
    df['price_change'] = df['close'].diff()
    df['volume_change'] = df['volume'].diff()
    bearish_divergence = (df['price_change'] < 0) & (df['volume_change'] > 0)
    bullish_divergence = (df['price_change'] > 0) & (df['volume_change'] < 0)
    
    # Simple Moving Average Crossover
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    crossover_signal = sma_5 - sma_20
    
    # Price Oscillator
    oscillator = sma_5 - sma_20
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = positive_flow.rolling(window=14).sum() / (positive_flow.rolling(window=14).sum() + negative_flow.rolling(window=14).sum())
    
    # High vs. Low Position
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    
    # Breakout Analysis
    upward_breakout = (df['close'] > df['high'].shift(1)) & (df['close'] > df['close'].shift(1))
    downward_breakout = (df['close'] < df['low'].shift(1)) & (df['close'] < df['close'].shift(1))
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    upward_breakout_strength = (df['close'] - df['high'].shift(1)) / avg_daily_range
    downward_breakout_strength = (df['low'].shift(1) - df['close']) / avg_daily_range
    upward_breakout_strength = upward_breakout_strength.where(upward_breakout, 0)
    downward_breakout_strength = downward_breakout_strength.where(downward_breakout, 0)
    cumulated_breakout_strength = upward_breakout_strength.rolling(window=10).sum() - downward_breakout_strength.rolling(window=10).sum()
    
    # Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    adjusted_breakout_strength = cumulated_breakout_strength.where(volume_ratio > 1.5, cumulated_breakout_strength * 1.5)
    
    # Price Momentum
    momentum = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    weighted_momentum = momentum * df['volume']
    
    # Adjust Momentum by ATR
    atr_adjusted_momentum = weighted_momentum / atr_14
    
    # Generate Alpha Factor
    alpha_factor = close_to_open_return + adjusted_breakout_strength + atr_adjusted_momentum
    
    return alpha_factor
```
