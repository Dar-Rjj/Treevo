```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Prev_Close'])
    df['TR3'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['Upward_Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['Avg_Daily_Range_20']
    df['Downward_Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['Avg_Daily_Range_20']
    
    # Cumulate Breakout Strength
    df['Cumulative_Up_Breakout_Strength'] = df['Upward_Breakout_Strength'].rolling(window=10).sum().fillna(0)
    df['Cumulative_Down_Breakout_Strength'] = df['Downward_Breakout_Strength'].rolling(window=10).sum().fillna(0)
    df['Net_Cumulative_Breakout_Strength'] = df['Cumulative_Up_Breakout_Strength'] - df['Cumulative_Down_Breakout_Strength']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(14)) * df['volume']
    
    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Price_Momentum'] / df['ATR_14']
    
    # Generate Alpha Factor
    df['Intraday_Spread'] = df['high'] - df['low']
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Alpha_Factor'] = df['Intraday_Spread'] + df['Close_to_Open_Return'] + df['Net_Cumulative_Breakout_Strength'] + df['ATR_Adjusted_Momentum']
    
    return df['Alpha_Factor']
```
