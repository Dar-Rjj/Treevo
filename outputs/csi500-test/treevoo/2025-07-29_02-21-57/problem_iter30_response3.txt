```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Prev_Close']).abs()
    df['TR3'] = (df['low'] - df['Prev_Close']).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['Daily_Range'] = df['high'] - df['low']

    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()

    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['Breakout_Strength_Up'] = ((df['close'] - df['high'].shift(1)) / df['Avg_Daily_Range_20']).fillna(0)
    df['Breakout_Strength_Down'] = ((df['low'].shift(1) - df['close']) / df['Avg_Daily_Range_20']).fillna(0)

    # Cumulate Breakout Strength
    df['Cumulative_Breakout_Strength'] = (df['Breakout_Strength_Up'].rolling(window=10).sum() -
                                          df['Breakout_Strength_Down'].rolling(window=10).sum())

    # Calculate 10-Day Volume Adjusted Momentum
    df['10_Day_Close_Return'] = (df['close'] / df['close'].shift(9)) - 1
    df['10_Day_Volume_Ratio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Momentum'] = df['10_Day_Close_Return'] * df['10_Day_Volume_Ratio']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    df['10_Day_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=10).mean()

    # Adjust Momentum by ATR
    df['Weighted_Momentum'] = (df['close'] - df['close'].shift(14)) * df['volume']
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Alpha_Factor'] = (df['ATR_Adjusted_Momentum'] + df['Close_to_Open_Return'] +
                          df['Cumulative_Breakout_Strength'] + df['Volume_Adjusted_Momentum'] +
                          df['10_Day_Intraday_Volatility'])

    # Incorporate Volume Trend
    df['10_Day_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Surge'] = (df['volume'] > df['10_Day_MA_Volume']).astype(int)
    df['Volume_Decline'] = (df['volume'] < df['10_Day_MA_Volume']).astype(int)
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1.5 * df['Volume_Surge'] + 0.5 * df['Volume_Decline'])

    # Calculate 5-Day Exponential Moving Average (EMA) of Close Price
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_5_Trend'] = (df['EMA_5'] > df['EMA_5'].shift(1)).astype(int)
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1.2 * df['EMA_5_Trend'] + 0.8 * (1 - df['EMA_5_Trend']))

    return df['Alpha_Factor']
```
