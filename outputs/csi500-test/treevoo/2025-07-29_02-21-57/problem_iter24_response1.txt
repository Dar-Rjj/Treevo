```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term (5 days) and Long-Term (20 days) VWMA
    short_term = 5
    long_term = 20
    
    df['VWMA_Short'] = (df['close'] * df['volume']).rolling(window=short_term).sum() / df['volume'].rolling(window=short_term).sum()
    df['VWMA_Long'] = (df['close'] * df['volume']).rolling(window=long_term).sum() / df['volume'].rolling(window=long_term).sum()

    # Generate Crossover Signal
    df['Crossover'] = np.where(df['VWMA_Short'] > df['VWMA_Long'], 1, -1)

    # Calculate 5-Day Volume Adjusted Momentum
    df['Price_Return'] = df['close'].pct_change(periods=5)
    df['Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['Vol_Adjusted_Momentum'] = df['Price_Return'] * df['Volume_Ratio']

    # Evaluate Volume Adjusted Momentum
    df['Momentum_Signal'] = np.where(df['Vol_Adjusted_Momentum'] > 0, 1, -1)

    # Combine Crossover and Momentum Signals
    df['Combined_Signal'] = np.where((df['Crossover'] == 1) & (df['Momentum_Signal'] == 1), 2,
                                     np.where((df['Crossover'] == -1) & (df['Momentum_Signal'] == -1), -2,
                                     np.where(df['Crossover'] == 1, 1,
                                     np.where(df['Crossover'] == -1, -1, 0))))

    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period ATR
    df['ATR'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['close'].pct_change() + df['ATR_Adjusted_Momentum'] + df['Combined_Signal']

    # Calculate Volume-Price Correlation
    df['Volume_Price_Correlation'] = df['close'].rolling(window=20).corr(df['volume'])

    # Integrate Volume-Price Correlation into Alpha Factor
    df['Alpha_Factor'] = df['Alpha_Factor'] * (1 + np.sign(df['Volume_Price_Correlation']))

    return df['Alpha_Factor']
```
