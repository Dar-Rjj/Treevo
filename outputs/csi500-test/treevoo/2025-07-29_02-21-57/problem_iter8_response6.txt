```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']

    # Calculate 20-Day Moving Average of Daily Range
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=22).mean()

    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)

    # Calculate Breakout Strength
    df['breakout_strength'] = 0
    df.loc[df['upward_breakout'] == 1, 'breakout_strength'] = (df['close'] - df['high'].shift(1)) / df['avg_daily_range_20']
    df.loc[df['downward_breakout'] == 1, 'breakout_strength'] = (df['low'].shift(1) - df['close']) / df['avg_daily_range_20']

    # Cumulate Breakout Strength over the past 10 days
    df['cum_breakout_strength'] = df['breakout_strength'].rolling(window=10).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5_day_return'] = df['close'] / df['close'].shift(5) - 1
    df['5_day_vol_ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5*21).mean()
    df['vol_adj_momentum'] = df['5_day_return'] * df['5_day_vol_ratio']

    # Calculate High-Low Price Oscillation
    df['high_low_range'] = df['high'] - df['low']
    df['std_high_low_5'] = df['high_low_range'].rolling(window=5).std()
    df['mean_high_low_5'] = df['high_low_range'].rolling(window=5).mean()
    df['price_oscillation'] = (df['high_low_range'] - df['mean_high_low_5']) / df['std_high_low_5']
    df['price_oscillation_avg_5'] = df['price_oscillation'].rolling(window=5).mean()

    # Integrate Volume-Adjusted Momentum and Price Oscillation
    df['integrated_factor'] = 0.7 * df['vol_adj_momentum'] + 0.3 * df['price_oscillation_avg_5']

    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['weighted_momentum'] = df['price_momentum'] * df['volume']

    # Normalize by ATR
    df['normalized_momentum'] = df['weighted_momentum'] / df['atr_14']

    # Combine Integrated Components with Breakout Strength
    df['final_alpha_factor'] = 0.6 * df['integrated_factor'] + 0.4 * df['cum_breakout_strength'] + df['normalized_momentum']

    return df['final_alpha_factor']
```
