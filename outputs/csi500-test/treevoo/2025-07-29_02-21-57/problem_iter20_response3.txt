def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['breakout_strength'] = 0.0
    df.loc[df['upward_breakout'] == 1, 'breakout_strength'] = (df['close'] - df['high'].shift(1)) / df['daily_range']
    df.loc[df['downward_breakout'] == 1, 'breakout_strength'] = (df['low'].shift(1) - df['close']) / df['daily_range']

    df['positive_breakout_strength'] = df['breakout_strength'].apply(lambda x: max(x, 0))
    df['negative_breakout_strength'] = df['breakout_strength'].apply(lambda x: min(x, 0))
    df['cumulated_positive_breakout_strength'] = df['positive_breakout_strength'].rolling(window=10).sum()
    df['cumulated_negative_breakout_strength'] = df['negative_breakout_strength'].rolling(window=10).sum()
    df['cumulated_breakout_strength'] = df['cumulated_positive_breakout_strength'] - (-df['cumulated_negative_breakout_strength'])
    
    # Calculate Short-Term Volume Weighted Moving Average (VWMA)
    df['short_term_vwma'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term Volume Weighted Moving Average (VWMA)
    df['long_term_vwma'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
