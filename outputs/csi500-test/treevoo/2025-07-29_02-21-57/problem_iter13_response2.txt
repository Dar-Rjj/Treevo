```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Volume-Weighted Close Price Return
    df['5D_Close_Return'] = df['close'].pct_change(periods=5)
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5D_VW_Close_Return'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']
    
    # Calculate High-Low Price Oscillation with Exponential Smoothing
    df['High_Low_Range'] = df['high'] - df['low']
    df['Exp_Smooth_High_Low_Range'] = df['High_Low_Range'].ewm(alpha=0.2, min_periods=1).mean()
    df['Price_Oscillation'] = df['High_Low_Range'] - df['Exp_Smooth_High_Low_Range']
    
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    df['Weighted_Momentum'] = df['Momentum'] * df['volume']
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = df['Weighted_Momentum'] + df['5D_VW_Close_Return'] + df['Price_Oscillation']
    
    return df['Alpha_Factor']
```
