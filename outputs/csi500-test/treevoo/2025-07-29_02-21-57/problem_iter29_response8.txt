```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return based on close prices
    df['daily_return'] = df['close'].diff()
    
    # Calculate the average daily trading volume over the past 10 days
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Compare today's volume with the average volume to identify unusual trading activity
    df['volume_change'] = df['volume'] / df['avg_volume_10']
    
    # Calculate True Range components
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate 14-Period Average True Range (ATR)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate 20-Day Moving Average of Daily Range
    df['avg_daily_range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Identify Breakouts
    df['upward_breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['downward_breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['breakout_strength_up'] = (df['close'] - df['high'].shift(1)) / df['avg_daily_range_20']
    df['breakout_strength_down'] = (df['low'].shift(1) - df['close']) / df['avg_daily_range_20']
    
    # Cumulate Breakout Strength over the past 10 days
    df['cum_breakout_strength_up'] = df['breakout_strength_up'].rolling(window=10).sum()
    df['cum_breakout_strength_down'] = df['breakout_strength_down'].rolling(window=10).sum()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Impact
    df['volume_impact'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df.loc[df['volume_impact'] > 1.5, 'cum_breakout_strength_up'] *= 1.5
    df.loc[df['volume_impact'] > 1.5, 'cum_breakout_strength_down'] *= 1.5
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(14)
    
    # Adjust Momentum by ATR
    df['atr_adjusted_momentum'] = df['price_momentum'] / df['atr_14']
    
    # Combine Indicators to generate Alpha Factor
    df['alpha_factor'] = (
        df['intraday_high_low_spread'] +
        df['close_to_open_return'] +
        (df['cum_breakout_strength_up'] - df['cum_breakout_strength_down']) +
        df['atr_adjusted_momentum']
    )
    
    return df['alpha_factor']
```
