import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=14):
    # Calculate True Range and High-Low Spread
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Prev_Close'])
    df['TR3'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=lookback_period).mean()

    # Calculate 14-Period Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Short-Term Volume Weighted Moving Average (VWMA) over 5 days
    df['Volume_Weighted_Price'] = df['close'] * df['volume']
    df['Short_Term_VWMA'] = df['Volume_Weighted_Price'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Long-Term Volume Weighted Moving Average (VWMA) over 20 days
    df['Long_Term_VWMA'] = df['Volume_Weighted_Price'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5Day_Close_Return'] = df['close'].pct_change(periods=5)
    df['5Day_Volume_Ratio'] = df['volume'] / df['volume'].shift(5)
    df['5Day_Momentum'] = df['5Day_Close_Return'] * df['5Day_Volume_Ratio']

    # Calculate 10-Day Price Oscillation
    df['10Day_High'] = df['high'].rolling(window=10).max()
    df['10Day_Low'] = df['low'].rolling(window=10).min()
    df['10Day_Price_Range'] = df['10Day_High'] - df['10Day_Low']

    # Calculate 10-Day Volume-Adjusted Range
    df['10Day_Total_Volume'] = df['volume'].rolling(window=10).sum()
    df['10Day_Volume_Adjusted_Range'] = df['10Day_Price_Range'] * df['10Day_Total_Volume']

    # Combined Crossover, Momentum, and Enhanced Price Oscillation Signal
    df['VWMA_Signal'] = np.where(df['Short_Term_VWMA'] > df['Long_Term_VWMA'], 1, -1)
    df['Momentum_Signal'] = np.where(df['5Day_Momentum'] > 0, 1, -1)
    df['Price_Oscillation_Signal'] = np.where(df['10Day_Price_Range'] > df['10Day_Price_Range'].rolling(window=lookback_period).mean(), 1, -1)
    df['Volume_Adjusted_Range_Signal'] = np.where(df['10Day_Volume_Adjusted_Range'] > df['10Day_Volume_Adjusted_Range'].rolling(window=lookback_period).mean(), 1, -1)

    # Combine Signals
    df['Combined_Signal'] = df['VWMA_Signal'] + df['Momentum_Signal'] + df['Price_Oscillation_Signal'] + df['Volume_Adjusted_Range_Signal']
    df['Combined_Signal'] = np.where(df['Combined_Signal'] == 4, 2, np.where(df['Combined_Signal'] == -4, -2, df['Combined_Signal']))
    df['Combined_Signal'] = np.where((df['Combined_Signal'] >= 3) & (df['Combined_Signal'] < 4), 1, np.where((df['Combined_Signal'] <= -3) & (df['Combined_Signal'] > -4), -1, df['Combined_Signal']))

    # Weight by Volume and Adjust Momentum by ATR
    df['Weighted_Momentum'] = df['5Day_Momentum'] * df['volume']
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR_14']
