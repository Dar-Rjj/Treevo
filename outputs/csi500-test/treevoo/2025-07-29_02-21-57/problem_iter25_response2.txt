import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and High-Low Spread
    df['Previous_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Previous_Close']).abs()
    df['TR3'] = (df['low'] - df['Previous_Close']).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(14)

    # Weight by Volume
    df['Volume_Weighted_Momentum'] = df['Price_Momentum'] * df['volume']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Volume_Weighted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Average_High_Low_Spread_14'] = df['High_Low_Spread'].rolling(window=14).mean()
    df['Spread_Relative_Strength'] = df['High_Low_Spread'] / df['Average_High_Low_Spread_14']
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Close_to_Open_Return'] + df['Spread_Relative_Strength']

    # Explore Trend and Volatility Characteristics
    df['Trend_Direction'] = (df['close'] > df['close'].shift(1)).rolling(window=14).sum()
    df['Volatility'] = df['close'].pct_change().rolling(window=30).std()
    df['Volume_Accumulation'] = df['volume'].rolling(window=14).sum()

    # Examine Opening and Closing Prices
    df['Up_Day'] = (df['close'] > df['open']).astype(int)
    df['Down_Day'] = (df['close'] < df['open']).astype(int)
    df['Up_Day_Count'] = df['Up_Day'].rolling(window=14).sum()
    df['Down_Day_Count'] = df['Down_Day'].rolling(window=14).sum()

    df['Pattern_Strength'] = (df['close'] - df['open']).abs() / (df['high'] - df['low'])
