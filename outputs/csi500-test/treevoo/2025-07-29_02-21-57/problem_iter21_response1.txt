```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['Prev_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Prev_Close'])
    df['TR3'] = abs(df['low'] - df['Prev_Close'])
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate 14-Period Average True Range (ATR) and 20-Day Moving Average of Daily Range
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Breakout_Up'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Breakout_Down'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['Breakout_Strength_Up'] = (df['close'] - df['high'].shift(1)) / df['Avg_Daily_Range_20']
    df['Breakout_Strength_Down'] = (df['low'].shift(1) - df['close']) / df['Avg_Daily_Range_20']
    df['Breakout_Strength'] = df['Breakout_Strength_Up'] * df['Breakout_Up'] - df['Breakout_Strength_Down'] * df['Breakout_Down']
    
    # Cumulate Breakout Strength
    df['Cum_Breakout_Strength'] = df['Breakout_Strength'].rolling(window=10).sum()
    
    # Calculate Price Momentum
    df['Momentum'] = df['close'] - df['close'].shift(14)
    
    # Weight by Volume
    df['Weighted_Momentum'] = df['Momentum'] * df['volume']
    
    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['ATR_14']
    
    # Generate Alpha Factor
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Alpha_Factor'] = df['Daily_Range'] + df['Close_to_Open_Return'] + df['Cum_Breakout_Strength'] + df['ATR_Adjusted_Momentum']
    
    # Introduce Relative Strength
    df['EMA_Stock_14'] = df['close'].ewm(span=14, adjust=False).mean()
    df['EMA_SP500_14'] = df['SP500_close'].ewm(span=14, adjust=False).mean()  # Assuming SP500_close is in the DataFrame
    df['Relative_Strength'] = df['EMA_Stock_14'] / df['EMA_SP500_14']
    
    # Combine with Alpha Factor
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] * df['Relative_Strength']
    
    return df['Final_Alpha_Factor']
```
