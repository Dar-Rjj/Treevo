import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Close Price Return
    df['5D_Close_Return'] = df['close'].pct_change(5)
    
    # Calculate 5-Day Volume Ratio
    df['5D_Volume_Sum'] = df['volume'].rolling(window=5).sum()
    avg_5D_volume = df['volume'].rolling(window=5).mean().shift(1)
    df['5D_Volume_Ratio'] = df['5D_Volume_Sum'] / avg_5D_volume
    
    # Volume Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']
    
    # High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    mean_high_low_range = df['High_Low_Range'].rolling(window=10).mean()
    std_high_low_range = df['High_Low_Range'].rolling(window=10).std()
    df['Adjusted_Price_Oscillation'] = (df['High_Low_Range'] - mean_high_low_range) / std_high_low_range
    
    # Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Moving Averages of Intraday Returns
    df['ST_Intraday_MA'] = df['Intraday_Return'].rolling(window=5).mean()
    df['LT_Intraday_MA'] = df['Intraday_Return'].rolling(window=20).mean()
    
    # Intraday Return Difference
    df['Intraday_Return_Diff'] = df['LT_Intraday_MA'] - df['ST_Intraday_MA']
    
    # Volume Change
    df['Volume_Change'] = df['volume'].diff()
    df['Filtered_Intraday_Return_Diff'] = df['Intraday_Return_Diff'] * (df['Volume_Change'] > 0)
    
    # Volume Weighted Moving Average (VWMA)
    def vwma(price, volume, window):
        return (price * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()
    
    df['Short_Term_VWMA'] = vwma(df['close'], df['volume'], 5)
    df['Long_Term_VWMA'] = vwma(df['close'], df['volume'], 20)
    
    # Generate Combined Signal
    signals = []
    for index, row in df.iterrows():
        vwma_signal = 1 if row['Short_Term_VWMA'] > row['Long_Term_VWMA'] else -1
        momentum_signal = 1 if row['Volume_Adjusted_Momentum'] > 0 else -1
        price_oscillation_signal = 1 if row['Adjusted_Price_Oscillation'] > 0 else -1
        
        if vwma_signal == 1 and momentum_signal == 1 and price_oscillation_signal == 1:
            signal = 1
        elif vwma_signal == -1 and momentum_signal == -1 and price_oscillation_signal == -1:
            signal = -1
        elif (vwma_signal == 1 and momentum_signal == 1) or (vwma_signal == -1 and momentum_signal == -1):
            signal = 0.5 if vwma_signal == 1 else -0.5
        elif (vwma_signal == 1 and price_oscillation_signal == 1) or (vwma_signal == -1 and price_oscillation_signal == -1):
            signal = 0.5 if vwma_signal == 1 else -0.5
        else:
            signal = 0
        signals.append(signal)
    
    df['Alpha_Factor'] = signals
    return df['Alpha_Factor']
