```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the difference between high and low prices
    high_low_diff = df['high'] - df['low']
    
    # Calculate the weighted volume by amount
    weighted_volume = df['volume'] * df['amount']
    
    # Generate the alpha factor
    alpha_factor = (close_to_avg_ratio * high_low_diff) / (weighted_volume + 1e-6)
    
    return alpha_factor
```
