```python
def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (df['close'] - df['low']) / df['daily_range']
    
    # Calculate the selling pressure
    df['selling_pressure'] = (df['high'] - df['close']) / df['daily_range']
    
    # Calculate the volume-weighted buying pressure
    df['volume_weighted_bp'] = df['buying_pressure'] * df['volume']
    
    # Calculate the volume-weighted selling pressure
    df['volume_weighted_sp'] = df['selling_pressure'] * df['volume']
    
    # Calculate the net volume-weighted pressure
    df['net_volume_weighted_pressure'] = df['volume_weighted_bp'] - df['volume_weighted_sp']
    
    # Calculate the factor as the rolling mean of the net volume-weighted pressure
    factor = df['net_volume_weighted_pressure'].rolling(window=5).mean()
    
    return factor
```
