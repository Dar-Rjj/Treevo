import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Components
    short_term_momentum = df['close'].pct_change(5)
    long_term_momentum = df['close'].pct_change(20)

    # Integrate Momentum
    momentum = 0.8 * short_term_momentum + 0.2 * long_term_momentum

    # Calculate Liquidity Components
    volume_to_price_ratio = df['volume'] / df['close']
    true_range = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    average_true_range = true_range.rolling(window=14).mean()

    # Integrate Liquidity
    liquidity = 0.7 * volume_to_price_ratio + 0.3 * average_true_range

    # Combine Momentum and Liquidity
    final_alpha_factor = 0.6 * momentum + 0.4 * liquidity

    # Apply Smoothing
    final_alpha_factor_sma = final_alpha_factor.rolling(window=5).mean()

    return final_alpha_factor_sma.dropna()
