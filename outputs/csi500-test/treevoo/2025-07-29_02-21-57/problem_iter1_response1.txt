```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-Day Close Price Return
    close_t = df['close']
    close_t_minus_5 = df['close'].shift(5)
    close_return = (close_t - close_t_minus_5) / close_t_minus_5
    
    # Calculate 5-Day Volume Ratio
    volume_5_days_sum = df['volume'].rolling(window=5).sum()
    average_5_day_volume = df['volume'].rolling(window=5 * 20).mean()  # Assuming a 20-day period for average
    volume_ratio = volume_5_days_sum / average_5_day_volume
    
    # Multiply Close Price Return by Volume Ratio
    volume_adjusted_momentum = close_return * volume_ratio
    
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-Day Standard Deviation of High-Low Range
    high_low_std = high_low_range.rolling(window=5).std()
    high_low_mean = high_low_range.rolling(window=5).mean()
    
    # Normalize Price Oscillation
    price_oscillation_normalized = (high_low_range - high_low_mean) / high_low_std
    
    # Combine Volume Adjusted Momentum with Price Oscillation
    factor_value = volume_adjusted_momentum + price_oscillation_normalized
    
    return factor_value
```
