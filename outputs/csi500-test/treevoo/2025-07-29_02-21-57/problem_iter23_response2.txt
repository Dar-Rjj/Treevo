```python
def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the average of high and low prices
    avg_price = (data['high'] + data['low']) / 2
    
    # Calculate the difference between close and open prices
    price_diff = data['close'] - data['open']
    
    # Calculate the volume weighted amount
    vol_weighted_amount = data['amount'] / data['volume']
    
    # Combine the above metrics to form the alpha factor
    factor = (avg_price * price_diff) / vol_weighted_amount
    
    return factor
```
