```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5-day) Volume Weighted Moving Average (VWMA)
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Long-Term (20-day) Volume Weighted Moving Average (VWMA)
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate 5-Day Volume Adjusted Momentum
    price_return = df['close'] / df['close'].shift(5) - 1
    volume_ratio = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = price_return * volume_ratio
    
    # Generate VWMA Crossover Signal
    vwma_signal = pd.Series(index=df.index, dtype='float64')
    vwma_signal[short_term_vwma > long_term_vwma] = 1  # Buy Signal
    vwma_signal[short_term_vwma < long_term_vwma] = -1  # Sell Signal
    
    # Generate Momentum Signal
    momentum_signal = pd.Series(index=df.index, dtype='float64')
    momentum_signal[volume_adjusted_momentum > 0] = 1  # Buy Signal
    momentum_signal[volume_adjusted_momentum < 0] = -1  # Sell Signal
    
    # Combine Signals
    combined_signal = pd.Series(index=df.index, dtype='float64')
    combined_signal[(vwma_signal == 1) & (momentum_signal == 1)] = 2  # Strong Buy
    combined_signal[(vwma_signal == -1) & (momentum_signal == -1)] = -2  # Strong Sell
    combined_signal[(vwma_signal == 1) & (momentum_signal != -1)] = 1  # Moderate Buy
    combined_signal[(vwma_signal == -1) & (momentum_signal != 1)] = -1  # Moderate Sell
    combined_signal[(vwma_signal != 1) & (vwma_signal != -1)] = 0  # Hold
    
    return combined_signal
```
