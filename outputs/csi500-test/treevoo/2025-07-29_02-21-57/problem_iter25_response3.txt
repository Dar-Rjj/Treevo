```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) for 10 periods
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate Exponential Moving Average (EMA) for 10 periods
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate True Range (TR)
    df['TR'] = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    
    # Calculate Average True Range (ATR) for 14 periods
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Identify new 5-period high and low
    df['New_High'] = df['high'].rolling(window=6).max() == df['high']
    df['New_Low'] = df['low'].rolling(window=6).min() == df['low']
    
    # Calculate Rate of Change (ROC) for 12 periods
    df['ROC_12'] = (df['close'] - df['close'].shift(12)) / df['close'].shift(12)
    
    # Calculate Relative Strength Index (RSI) for 14 periods
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['volume'] * (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))).cumsum()
    
    # Calculate Money Flow Index (MFI) for 14 periods
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum().abs()
    money_ratio = positive_money_flow / negative_money_flow
    df['MFI_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate Accumulation/Distribution Line (ADL)
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (mfm * df['volume']).cumsum()
    df['ADL'] = adl
    
    # Generate the alpha factor
    alpha_factor = (
        df['SMA_10'] + 
        df['EMA_10'] + 
        df['ATR_14'] + 
        df['ROC_12'] + 
        df['RSI_14'] + 
        df['OBV'] + 
        df['MFI_14'] + 
        df['ADL']
    )
    
    return alpha_factor
```
