```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high to low price
    high_to_low = df['high'] / df['low']
    
    # Calculate the average of the two ratios
    avg_ratio = (close_to_open + high_to_low) / 2
    
    # Calculate the difference between the high and low prices
    high_low_diff = df['high'] - df['low']
    
    # Calculate the volume weighted by the amount
    vol_weighted = df['volume'] * df['amount']
    
    # Create a factor that combines the average ratio, high-low difference, and volume-weighted amount
    factor = (avg_ratio * high_low_diff) + vol_weighted
    
    return factor
```
