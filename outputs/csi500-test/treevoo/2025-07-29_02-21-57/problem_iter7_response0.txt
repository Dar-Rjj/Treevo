import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Daily Range
    df['20D_Avg_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Identify Breakouts
    df['Upward_Breakout'] = (df['close'] > df['high'].shift(1)).astype(int)
    df['Downward_Breakout'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate Breakout Strength
    df['Breakout_Strength'] = 0
    df.loc[df['Upward_Breakout'] == 1, 'Breakout_Strength'] = (df['close'] - df['high'].shift(1)) / df['20D_Avg_Daily_Range']
    df.loc[df['Downward_Breakout'] == 1, 'Breakout_Strength'] = (df['low'].shift(1) - df['close']) / df['20D_Avg_Daily_Range']
    
    # Cumulate Breakout Strength over the past 10 days
    df['Cumulative_Breakout_Strength'] = df['Breakout_Strength'].rolling(window=10).sum()
    
    # Calculate Price Momentum Components
    df['5D_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20D_Return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Integrate Momentum
    df['Integrated_Momentum'] = 0.7 * df['5D_Return'] + 0.3 * df['20D_Return']
    
    # Calculate Liquidity Components
    df['Volume_to_Price_Ratio'] = df['volume'] / df['close']
    df['14D_ATR'] = df['high'] - df['low']
    
    # Integrate Liquidity
    df['Integrated_Liquidity'] = 0.6 * df['Volume_to_Price_Ratio'] + 0.4 * df['14D_ATR']
    
    # Combine Integrated Momentum and Liquidity with Breakout Strength
    df['Alpha_Factor'] = 0.5 * df['Integrated_Momentum'] + 0.5 * df['Integrated_Liquidity'] + df['Cumulative_Breakout_Strength']
    
    return df['Alpha_Factor']
