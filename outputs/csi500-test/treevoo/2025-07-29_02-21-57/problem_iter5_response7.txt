```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TR3'] = (df['low'] - df['close'].shift(1)).abs()
    df['TrueRange'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)

    # Calculate 14-Period Average True Range (ATR)
    df['ATR_14'] = df['TrueRange'].rolling(window=14).mean()

    # Calculate 5-Day Volume Adjusted Momentum
    df['5D_Close_Return'] = df['close'] / df['close'].shift(5) - 1
    df['5D_Volume_Ratio'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=5).mean()
    df['5D_Volume_Adjusted_Momentum'] = df['5D_Close_Return'] * df['5D_Volume_Ratio']

    # Calculate High-Low Price Oscillation
    df['High_Low_Range'] = df['high'] - df['low']
    df['10D_High_Low_SD'] = df['High_Low_Range'].rolling(window=10).std()
    df['10D_High_Low_Mean'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Adjusted_Price_Oscillation'] = (df['High_Low_Range'] - df['10D_High_Low_Mean']) / df['10D_High_Low_SD']

    # Combine Indicators
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Close_To_Open_Return'] = df['close'] / df['open'] - 1
    df['Volume_Weighted_Price'] = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    df['Combined_Indicators'] = (df['5D_Volume_Adjusted_Momentum'] + 
                                df['Adjusted_Price_Oscillation'] + 
                                df['Intraday_High_Low_Spread'] + 
                                df['Close_To_Open_Return']).mean(axis=1)
    
    df['Combined_Adjustment'] = df['Volume_Weighted_Price'] - df['Combined_Indicators']

    # Adjust Momentum by ATR
    df['ATR_Adjusted_Momentum'] = df['5D_Volume_Adjusted_Momentum'] / df['ATR_14']

    # Generate Alpha Factor
    df['Alpha_Factor'] = df['ATR_Adjusted_Momentum'] + df['Combined_Adjustment']

    return df['Alpha_Factor']
```
