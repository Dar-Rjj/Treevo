```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * (df['amount'] ** 0.5)
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * (df['amount'] ** 0.5)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_momentum = momentum / volume_amount_adjusted_range
    
    # Combine Volume-Amount-Adjusted Range with Weighted Price Change
    final_alpha_factor = combined_momentum + volume_amount_adjusted_range + weighted_price_change
    
    return final_alpha_factor
```
