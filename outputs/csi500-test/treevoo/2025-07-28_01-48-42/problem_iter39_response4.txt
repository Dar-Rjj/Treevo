```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    intraday_reversal = 1 / intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Integrate Momentum and Reversal
    momentum = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = momentum * intraday_reversal
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Multiply Price Change by Volume and Square Root of Amount
    price_change_adjusted = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Add to Combined Momentum and Intraday Reversal
    enhanced_signal = combined_momentum_reversal + price_change_adjusted + adjusted_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = abs(df['open'] - df['close'])
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = enhanced_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_open_diff = df['close'] - df['open'].shift(-1)
    confirmed_signal = smoothed_signal * close_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_return = (df['close'] / df['close'].shift(1)) - 1
    smoothed_price_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] / df['volume'].shift(1)
    volume_spike = (volume_change > 1.8 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Adjust Intraday Reversal by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Integrate Adjusted Intraday Reversal into Final Factor
    final_factor = confirmed_signal + adjusted_intraday_reversal
    
    # Generate New Factor Value
    new_factor = five_day_momentum / df['close']
    
    # Final Alpha Factor
    weighted_sum = (intraday_volatility * 0.35) + (enhanced_signal * 0.25) + (new_factor * 0.4) + (price_change / intraday_volatility * 0.1)
    final_alpha_factor = weighted_sum * volume_spike
    
    # Multiply by Trend Strength
    long_term_momentum = df['close'].pct_change().ewm(span=25, adjust=False).mean()
    final_alpha_factor = final_alpha_factor * long_term_momentum
    
    return final_alpha_factor
```
