```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum (Sum over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Calculate Daily Open-Close Price Spread
    daily_spread = df['open'] - df['close']
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * daily_spread.abs()
    
    # Weight Adjusted Momentum by Volume
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    
    # Combine High-Low Range with Weighted Adjusted Momentum
    combined_factor = weighted_adjusted_momentum / high_low_range
    
    return combined_factor
```
