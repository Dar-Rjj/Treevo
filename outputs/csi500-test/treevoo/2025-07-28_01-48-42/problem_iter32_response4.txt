import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    open_close_spread = df['close'] - df['open']

    price_change = df['close'].diff()

    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range

    momentum_adjusted_by_spread = five_day_momentum * np.abs(open_close_spread)

    combined_signal = momentum_adjusted_by_spread / adjusted_range

    enhanced_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = enhanced_combined_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = 1 / (1 + intraday_return)

    close_to_open_diff = df['open'].shift(-1) - df['close']

    weighted_momentum_reversal = momentum_adjusted_by_spread * np.abs(open_close_spread)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    combined_momentum_reversal = combined_factor + weighted_momentum_reversal

    daily_close_return = df['close'].pct_change()

    day_to_day_volume_change = df['volume'].diff()
    vol_spike_factor = (day_to_day_volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)

    long_term_momentum = df['close'].pct_change().ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = df[['high', 'low']].apply(lambda x: (x['high'] - x['low']) / x['open'], axis=1).rolling(window=5).std()

    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Adjust Momentum by Intraday Volatility
    volatility_adjusted_momentum = five_day_momentum * intraday_volatility

    ten_day_high_low_momentum = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / df['close']
    ten_day_volume_ma = df['volume'].rolling(window=10).mean()
    factor_value = ten_day_high_low_momentum * ten_day_volume_ma / df['close']

    enhanced_combined_signal_ema = combined_signal.ewm(span=5, adjust=False).mean()

    combined_factors = (
        0.3 * intraday_volatility +
        0.4 * factor_value +
        0.2 * enhanced_combined_signal_ema +
        0.1 * generated_factor_value
    )

    final_alpha_factor = (
        combined_factors * smoothed_momentum * 
        vol_spike_factor * long_term_momentum * 
        intraday_volatility + close_to_open_diff
    )
    
    # Apply Volume Weighting
    final_alpha_factor = final_alpha_factor * df['volume']
