import pandas as pd

def heuristics_v2(df):
    # Compute Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period_momentum = 10
    raw_momentum = df['close'] - df['close'].rolling(window=lookback_period_momentum).mean()
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * open_close_spread.abs()
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Incorporate Enhanced Price Trend Sensitivity
    trend_direction = (df['close'] > df['open']).astype(int)
    high_low_range = df['high'] - df['low']
    momentum_with_trend = volume_weighted_momentum * (1 + (trend_direction * 2 - 1) * (high_low_range / df['open']))
    
    # Incorporate Opening Price Volatility
    lookback_period_volatility = 5
    opening_price_volatility = df['open'].rolling(window=lookback_period_volatility).std()
    momentum_with_open_volatility = momentum_with_trend * (1 + (trend_direction * 2 - 1) * (opening_price_volatility / df['open']))
    
    # Incorporate Volume Trend Sensitivity
    lookback_period_volume_trend = 10
    average_volume = df['volume'].rolling(window=lookback_period_volume_trend).mean()
    volume_trend_direction = (df['volume'] > average_volume).astype(int)
    volume_trend_factor = (df['volume'] - average_volume) / average_volume
    momentum_with_volume_trend = momentum_with_open_volatility * (1 + (volume_trend_direction * 2 - 1) * volume_trend_factor)
    
    return momentum_with_volume_trend
