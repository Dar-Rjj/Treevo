```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (intraday_return + 1)
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Momentum and Intraday Reversal
    momentum_reversal_combined = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = volume_adjusted_range + weighted_price_change + momentum_reversal_combined
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    
    # Multiply Volume-Adjusted Reversal with C-O Diff
    co_volume_reversal = intraday_reversal_score * close_open_diff * df['volume']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Compute Weighted Momentum
    weighted_momentum = (price_change * df['volume']).rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = weighted_momentum * abs(open_close_spread)
    
    # Volume-Adjust High-Low Range
    volume_adjusted_high_low = df['volume'] * high_low_range
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_factor + co_volume_reversal + adjusted_momentum) / volume_adjusted_high_low
    
    return final_alpha_factor
```
