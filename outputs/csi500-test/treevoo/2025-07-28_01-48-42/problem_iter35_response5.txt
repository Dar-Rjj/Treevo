```python
def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Momentum
    lookback_period = 10
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * open_close_spread.abs()
    
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust Momentum by High-Low Range
    adjusted_momentum_by_range = adjusted_momentum / high_low_range
    
    # Weight Adjusted Momentum by Volume
    final_adjusted_momentum = adjusted_momentum_by_range * df['volume']
    
    return final_adjusted_momentum
```
