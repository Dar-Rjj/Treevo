```python
def heuristics_v2(data):
    import pandas as pd

    # Calculate the percentage change in close prices
    data['close_change'] = data['close'].pct_change()
    
    # Calculate the ratio of close to high, which indicates how the stock closed relative to its daily high
    data['close_to_high'] = data['close'] / data['high']
    
    # Calculate the buying pressure: (close - low) / (high - low), indicating where the closing price stands within the day's range
    data['buying_pressure'] = (data['close'] - data['low']) / (data['high'] - data['low'])
    
    # Calculate the ratio of volume to amount, a simplistic measure of liquidity or activity per unit of value
    data['volume_to_amount'] = data['volume'] / data['amount']

    # Alpha factor combining the ratios and percentages
    alpha_factor = (
        data['close_change'] 
        + data['close_to_high'] 
        + data['buying_pressure'] 
        + data['volume_to_amount']
    ) / 4

    # Return the alpha factor as a pandas Series
    return alpha_factor
```
