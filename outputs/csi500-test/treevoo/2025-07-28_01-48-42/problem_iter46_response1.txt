```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    intraday_reversal = -intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    combined_adjusted = volume_adjusted + amount_adjusted
    
    # Integrate Momentum and Reversal
    momentum_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal
    
    # Multiply Price Change by Volume and Square Root of Amount
    price_change_volume_amount = (df['close'] - df['open']) * df['volume'] * np.sqrt(df['amount'])
    
    # Add to Combined Momentum and Intraday Reversal
    combined_signal = momentum_reversal + combined_adjusted + price_change_volume_amount
    
    # Compute 5-Day Momentum
    five_day_momentum = (df['close'] - df['close'].shift(5)).rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = abs(df['open'] - df['close'])
    adjusted_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    final_factor = smoothed_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_return = (df['close'] / df['close'].shift(1)) - 1
    smoothed_price_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'] / df['volume'].shift(1)
    volume_spike = (day_to_day_volume_change > 1.8 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Adjust Intraday Reversal by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Integrate Adjusted Intraday Reversal into Final Factor
    final_factor += adjusted_intraday_reversal * volume_spike
    
    # Generate New Factor Value
    generated_factor = five_day_momentum / df['close']
    
    # Incorporate Long-Term Trend Strength
    long_term_momentum = (df['close'] / df['close'].shift(25)) - 1
    smoothed_long_term_momentum = long_term_momentum.ewm(span=25, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.35 * intraday_volatility +
        0.25 * final_factor +
        0.4 * generated_factor +
        0.1 * (five_day_momentum / df['close'].rolling(window=5).std())
    )
    
    return final_alpha_factor
```
