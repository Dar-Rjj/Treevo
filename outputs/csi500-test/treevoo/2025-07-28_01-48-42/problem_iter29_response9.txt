```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the money flow ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(close_pct_change > 0, 0)
    negative_flow = money_flow.where(close_pct_change < 0, 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    
    # Calculate the weighted average of the last 5 days' close prices using volume as weights
    weighted_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Alpha factor: combination of money flow ratio and weighted close price
    alpha_factor = (money_flow_ratio + weighted_close) / 2
    
    return alpha_factor
```
