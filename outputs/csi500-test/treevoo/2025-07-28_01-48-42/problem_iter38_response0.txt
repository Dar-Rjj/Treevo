```python
def heuristics_v2(df):
    # Compute Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].shift(lookback_period)
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * open_close_spread.abs()
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Determine Trend Direction
    trend_direction = (df['close'] > df['open']).astype(int)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Modify Momentum based on Trend
    trend_modifier = (1 + (high_low_range / df['open']) * trend_direction) * (2 * trend_direction - 1)
    momentum_with_trend = volume_weighted_momentum * trend_modifier
    
    # Incorporate Opening Price Volatility
    volatility_lookback = 5
    opening_price_volatility = df['open'].rolling(window=volatility_lookback).std()
    
    # Further Adjust Momentum
    volatility_ratio = opening_price_volatility / df['open']
    final_momentum = momentum_with_trend * (1 + (volatility_ratio * (2 * trend_direction - 1)))
    
    return final_momentum
```
