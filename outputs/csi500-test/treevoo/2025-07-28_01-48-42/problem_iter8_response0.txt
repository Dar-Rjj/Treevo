```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Calculate High-Low Range Momentum
    momentum = high_low_range.diff()
    momentum_sum = momentum.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    avg_volume = df['volume'].rolling(window=5).mean()
    avg_amount = df['amount'].rolling(window=5).mean()
    normalized_momentum = momentum_sum / (avg_volume * avg_amount)
    
    # Integrate Momentum, Reversal, and Volume-Amount Adjustment
    adjusted_reversal = normalized_momentum * intraday_reversal_score
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    final_factor = adjusted_reversal * close_open_diff
    
    # Apply Moving Average Smoothing
    smoothed_factor = final_factor.rolling(window=10).mean()
    
    return smoothed_factor
```
