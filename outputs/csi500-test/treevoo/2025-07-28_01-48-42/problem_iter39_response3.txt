```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    
    # Calculate the relative strength compared to the previous day
    df['prev_close'] = df['close'].shift(1)
    df['rel_strength'] = df['close'] / df['prev_close']
    
    # Calculate the accumulation distribution line
    df['money_flow_multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    df['adl'] = df['money_flow_volume'].cumsum()
    
    # Combine the two signals into one alpha factor
    df['alpha_factor'] = df['rel_strength'] * df['adl']
    
    return df['alpha_factor']
```
