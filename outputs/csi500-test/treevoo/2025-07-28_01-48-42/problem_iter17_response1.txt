import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (intraday_return + 1e-8)  # Add a small epsilon to avoid division by zero
    
    # Adjust for Volume and Amount
    volume_adjusted = df['volume'] * high_low_range
    amount_adjusted = np.sqrt(df['amount']) * high_low_range
    combined_adjusted = volume_adjusted + amount_adjusted
    
    # Integrate Momentum and Reversal
    momentum_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'].pct_change()
    weighted_price_change = price_change * (df['volume'] + np.sqrt(df['amount']))
    integrated_momentum_reversal = momentum_reversal + weighted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_five_day_momentum = five_day_mission * open_close_spread
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_five_day_momentum / combined_adjusted
    
    # Enhance Combined Signal with Smoothing
    smoothened_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_open_diff = df['close'] - df['open']
    final_factor = smoothened_signal * (combined_adjusted * close_open_diff)
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].pct_change()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    alpha_factor = smoothened_signal * smoothed_momentum * volume_spike
    
    return alpha_factor
