```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum = momentum / volume_adjusted_range
    
    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].diff()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike_factor = (day_to_day_volume_change > 2 * std_volume).astype(int)
    
    # Confirm with Volume Relative Strength
    daily_volume_rank = df['volume'].rank(pct=True)
    volume_relative_strength = intraday_reversal * daily_volume_rank
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'].shift(-1) - df['open']
    
    # Final Alpha Factor
    final_alpha = (combined_momentum * smoothed_momentum 
                   * volume_spike_factor 
                   * volume_relative_strength 
                   * close_to_open_diff)
    
    return final_alpha
```
