```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-Day High-Low Range Momentum
    range_momentum = high_low_range.diff().rolling(window=5).sum()
    
    # Adjust for Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    adjusted_momentum = range_momentum * avg_volume_5d
    
    # Apply Exponential Moving Average Smoothing
    smoothed_adjusted_momentum = adjusted_momentum.ewm(span=10).mean()
    interim_alpha_factor_1 = smoothed_adjusted_momentum
    
    # Calculate Logarithmic Daily Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    ema_21 = log_returns.ewm(span=21).mean()
    ema_60 = log_returns.ewm(span=60).mean()
    momentum_component = ema_21 - ema_60
    
    # Volatility Component
    abs_log_returns = np.abs(log_returns)
    ema_21_abs = abs_log_returns.ewm(span=21).mean()
    ema_60_abs = abs_log_returns.ewm(span=60).mean()
    volatility_component = ema_21_abs - ema_60_abs
    
    # Volume Component
    log_volume_changes = np.log(df['volume'] / df['volume'].shift(1))
    ema_21_vol = log_volume_changes.ewm(span=21).mean()
    ema_60_vol = log_volume_changes.ewm(span=60).mean()
    volume_component = ema_21_vol - ema_60_vol
    
    # Final Alpha Factor 2
    final_alpha_factor_2 = momentum_component + volatility_component + volume_component
    
    # Final Alpha Factor
    final_alpha_factor = interim_alpha_factor_1 + final_alpha_factor_2
    
    return final_alpha_factor
```
