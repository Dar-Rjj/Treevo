```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_volatility = (intraday_returns_high - intraday_returns_low).rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] + df['low']) / 2 - df['open']
    intraday_reversal_score = -intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['amount'] * df['volume']
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = combined_momentum_reversal + volume_adjusted_range
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    integrated_price_volume = price_change_momentum * volume_delta
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    integrated_final = integrated_signal * close_open_diff
    
    # Generate Factor Value
    ten_day_high_low_momentum = high_low_range.rolling(window=10).mean()
    ten_day_volume_ma = df['volume'].rolling(window=10).mean()
    factor_value = (ten_day_high_low_momentum * ten_day_volume_ma) / df['close']
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothened_signal = integrated_final.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    final_factor = smoothened_signal * daily_volatility
    
    # Apply a Second Smoothing Layer
    second_smoothened_signal = final_factor.rolling(window=10).mean()
    
    # Combine All Factors
    combined_factors = (intraday_volatility * 0.3) + (factor_value * 0.4) + (second_smoothened_signal * 0.2) + (integrated_price_volume * 0.1)
    
    # Final Alpha Factor
    volume_spike_factor = (volume_delta > 2 * df['volume'].rolling(window=20).std()).astype(int)
    final_alpha_factor = combined_factors * volume_spike_factor
    
    return final_alpha_factor
```
