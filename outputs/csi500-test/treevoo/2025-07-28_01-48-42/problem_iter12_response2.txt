```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_returns = (intraday_returns_high + intraday_returns_low) / 2
    intraday_volatility = intraday_returns.rolling(window=20).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Momentum
    momentum = price_change.rolling(window=15).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum_and_volume = momentum / volume_adjusted_range
    
    # Calculate High-Low Price Difference
    high_low_price_diff = df['high'] - df['low']
    
    # Calculate 20-Day High-Low Price Momentum
    high_low_momentum = high_low_price_diff.rolling(window=20).sum()
    
    # Calculate 20-Day Volume Moving Average
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    
    # Combine 20-Day High-Low Price Momentum and 20-Day Volume Moving Average
    combined_high_low_momentum_and_volume = high_low_momentum * volume_20_day_ma
    
    # Generate Factor Value
    generated_factor_value = combined_high_low_momentum_and_volume / df['close']
    
    # Combine All Factors
    weighted_intraday_volatility = 0.3 * intraday_volatility
    weighted_combined_high_low_momentum_and_volume = 0.45 * combined_high_low_momentum_and_volume
    weighted_combined_momentum_and_volume = 0.15 * combined_momentum_and_volume
    weighted_generated_factor_value = 0.1 * generated_factor_value
    
    factor = (weighted_intraday_volatility 
              + weighted_combined_high_low_momentum_and_volume 
              + weighted_combined_momentum_and_volume 
              + weighted_generated_factor_value)
    
    return factor
```
