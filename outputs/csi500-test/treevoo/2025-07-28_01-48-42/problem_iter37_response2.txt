```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / intraday_return

    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_range = volume_adjusted_range + amount_adjusted_range

    # Combine Momentum and Volume-Amount-Adjusted Range
    price_change = df['close'].diff()
    five_day_momentum = price_change.rolling(window=5).sum()
    open_close_spread = np.abs(df['open'] - df['close'])
    adjusted_momentum = five_day_momentum * open_close_spread
    momentum_adjusted_signal = adjusted_momentum / combined_range

    # Enhance Combined Signal with Smoothing
    smoothed_signal = momentum_adjusted_signal.ewm(span=5, adjust=False).mean()

    # Integrate Intraday Reversal into Final Factor
    close_open_diff = df['close'] - df['open']
    integrated_reversal = close_open_diff * intraday_reversal

    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_factor = smoothed_signal * daily_volatility

    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)
    trend_confirmation = volatility_factor * long_term_momentum

    # Incorporate Cumulative Return
    daily_returns = df['close'].pct_change()
    cumulative_return = daily_returns.rolling(window=20).sum()
    trend_return = trend_confirmation * cumulative_return

    # Incorporate Volume-Based Factors
    volume_change = df['volume'].diff()
    cumulative_volume = df['volume'].rolling(window=20).sum()
    final_factor = trend_return * cumulative_volume

    return final_factor
```
