```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_returns = (intraday_returns_high + intraday_returns_low) / 2
    intraday_volatility = intraday_returns.rolling(window=10).std()
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Compute Price Change
    price_change = df['close'].diff(1)
    
    # Adjust for Volume
    volume_adjusted_range = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * open_close_spread.abs()
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_signal = adjusted_momentum / volume_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, min_periods=5).mean()
    
    # Calculate 10-Day High-Low Price Momentum
    high_low_momentum = high_low_range.rolling(window=10).sum()
    
    # Calculate 10-Day Volume Moving Average
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    
    # Combine 10-Day High-Low Price Momentum and 10-Day Volume Moving Average
    combined_high_low_vol = high_low_momentum * volume_moving_avg
    
    # Generate Factor Value
    factor_value = combined_high_low_vol / df['close']
    
    # Combine All Factors
    combined_factors = (intraday_volatility * 0.3 +
                        combined_high_low_vol * 0.4 +
                        smoothed_signal * 0.2 +
                        factor_value * 0.1)
    
    # Calculate Day-to-Day Volume Change
    volume_change = df['volume'].diff(1)
    
    # Indicate Significant Volume Increase
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    final_factor = combined_factors * smoothed_signal * volume_spike
    
    return final_factor
```
