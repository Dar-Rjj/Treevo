```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Adjust for Volume
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range / volume_5d_avg
    
    # Integrate Momentum and Reversal
    momentum = adjusted_high_low_range - adjusted_high_low_range.shift(1)
    combined_factor = momentum * intraday_reversal_score
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open'].shift(-1)
    
    # Multiply Volume-Adjusted Reversal with C-O Diff
    final_factor = combined_factor * close_open_diff
    
    # Apply Moving Average Smoothing
    smoothed_factor = final_factor.rolling(window=10).mean()
    
    return smoothed_factor
```
