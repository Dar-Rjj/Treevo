```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Intraday Return Volatility
    intraday_returns_high = (df['high'] - df['open']) / df['open']
    intraday_returns_low = (df['low'] - df['open']) / df['open']
    intraday_return_volatility = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1).rolling(window=10).std()

    # Compute Price Change
    price_change = df['close'].diff()

    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']

    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()

    # Combine Momentum and Volume-Adjusted Range
    combined_signal = momentum / volume_adjusted_range

    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = combined_signal.ewm(span=5).mean()

    # Calculate 10-Day High-Low Price Momentum
    high_low_price_difference = df['high'] - df['low']
    high_low_momentum = high_low_price_difference.rolling(window=10).sum()

    # Calculate 10-Day Volume Moving Average
    volume_ma = df['volume'].rolling(window=10).mean()

    # Combine 10-Day High-Low Price Momentum and 10-Day Volume Moving Average
    combined_high_low_volume = high_low_momentum * volume_ma

    # Generate Factor Value
    factor_value = combined_high_low_volume / df['close']

    # Combine All Factors
    final_factor = (
        intraday_return_volatility * 0.3 +
        combined_high_low_volume * 0.4 +
        enhanced_combined_signal * 0.2 +
        factor_value * 0.1
    )

    return final_factor
```
