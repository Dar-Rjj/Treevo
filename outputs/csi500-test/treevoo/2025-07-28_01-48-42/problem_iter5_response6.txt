```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Multiply Price Change Momentum by Volume Delta
    price_volume_momentum = price_change_momentum * volume_delta
    
    # Apply Exponential Moving Average (EMA) with 5-day span
    smoothed_price_volume_momentum = price_volume_momentum.ewm(span=5, adjust=False).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['open'] - df['low']
    intraday_return_volatility = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1, ddof=0).rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum_vol_range = (momentum / volume_adjusted_range).fillna(0)
    
    # Calculate Average Volume Spike
    volume_difference = df['volume'] - df['volume'].shift(1)
    avg_volume_spike = volume_difference.rolling(window=5).sum()
    
    # Calculate Exponential Moving Average of Price Change
    ema_price_change = price_change.ewm(span=14, adjust=False).mean()
    
    # Normalize Price Change by EMA
    normalized_price_change = price_change / ema_price_change
    
    # Combine All Factors
    combined_factor = (0.3 * intraday_return_volatility +
                       0.2 * avg_volume_spike +
                       0.1 * normalized_price_change +
                       0.4 * combined_momentum_vol_range)
    
    return combined_factor
```
