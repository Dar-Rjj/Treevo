```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['open'] - df['low']
    intraday_returns = (intraday_returns_high + intraday_returns_low) / 2
    intraday_return_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal_score = -intraday_return
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_range_diff * intraday_reversal_score
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    weighted_price_change = df['close'] - df['open'].shift(1)
    integrated_factor = combined_momentum_reversal + volume_adjusted_range + weighted_price_change
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open']
    final_integrated_factor = integrated_factor * close_to_open_diff
    
    # Generate Factor Value
    factor_value = (high_low_range - high_low_range.shift(10)) * df['volume'].rolling(window=10).mean() / df['close']
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = final_integrated_factor.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2 / 4) + ((df['close'] - df['open']) ** 2 / 4))
    volatility_integrated_factor = smoothed_signal * daily_volatility
    
    # Apply a Second Smoothing Layer
    second_smoothed_signal = volatility_integrated_factor.rolling(window=10).mean()
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = second_smoothed_signal.ewm(span=5, adjust=False).mean()
    
    # Combine All Factors
    weighted_sum = (intraday_return_volatility * 0.3) + (factor_value * 0.4) + (enhanced_combined_signal * 0.2) + (volatility_integrated_factor * 0.1)
    
    # Final Alpha Factor
    volume_spike = (df['volume'] - df['volume'].shift(1)) > 2 * df['volume'].rolling(window=20).std()
    volume_spike_factor = volume_spike.astype(int)
    final_alpha_factor = weighted_sum * enhanced_combined_signal * volume_spike_factor
    
    return final_alpha_factor
```
