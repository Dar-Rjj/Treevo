```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    volume_amount_adjusted = volume_adjusted + amount_adjusted
    
    # Adjust 5-Day Momentum by Open-Close Spread
    momentum_adjusted = five_day_momentum * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted / volume_amount_adjusted
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = smoothed_combined_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    
    # Integrate into Final Factor
    integrated_reversal = intraday_reversal * close_open_diff
    
    # Adjust 5-Day Momentum by Open-Close Spread (Intraday Reversal)
    momentum_reversal_adjusted = five_day_momentum * np.abs(open_close_spread) * integrated_reversal
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_combined_signal = combined_factor + momentum_reversal_adjusted
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    final_factor = (final_combined_signal * smoothed_momentum * volume_spike) + close_open_diff
    
    return final_factor
```
