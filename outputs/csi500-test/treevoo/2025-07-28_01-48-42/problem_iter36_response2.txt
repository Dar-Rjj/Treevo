```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.exp(-intraday_return))
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    high_low_volume_adjusted = high_low_range * avg_volume_5d
    high_low_amount_adjusted = high_low_range * np.sqrt(avg_amount_5d)
    volume_amount_adjusted = high_low_volume_adjusted + high_low_amount_adjusted
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range.diff()
    combined_momentum_reversal = high_low_range_diff * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = combined_momentum_reversal + volume_amount_adjusted
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    momentum_5d = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['close'] - df['open']).abs()
    adjusted_momentum_5d = momentum_5d * open_close_spread
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum_5d / volume_amount_adjusted
    
    # Enhance Combined Signal with Smoothing
    ema_span = 5
    smoothed_combined_signal = combined_signal.ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Apply a Second Smoothing Layer
    sma_window = 10
    smoothed_signal = smoothed_combined_signal.rolling(window=sma_window, min_periods=sma_window).mean()
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, min_periods=12).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    std_volume_20d = df['volume'].rolling(window=20).std()
    volume_spike_factor = (volume_change > 2 * std_volume_20d).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().ewm(span=20, min_periods=20).mean()
    
    # Calculate Intraday Volatility
    intraday_returns = (df['high'] - df['low']) / df['low']
    intraday_volatility = intraday_returns.rolling(window=5).std()
    
    # Adjust Intraday Reversal Score by Intraday Volatility
    adjusted_intraday_reversal_score = intraday_reversal_score * intraday_volatility
    
    # Adjust 5-Day Momentum by Intraday Volatility
    adjusted_momentum_5d_volatility = adjusted_momentum_5d * intraday_volatility
    
    # Generate Factor Value
    high_low_momentum_10d = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    volume_moving_avg_10d = df['volume'].rolling(window=10).mean()
    factor_value = (high_low_momentum_10d * volume_moving_avg_10d) / df['close']
    
    # Combine All Factors
    combined_factors = (
        0.3 * intraday_volatility +
        0.4 * factor_value +
        0.2 * smoothed_signal +
        0.1 * adjusted_momentum_5d_volatility
    )
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_factors * smoothed_momentum * 
        volume_spike_factor * long_term_momentum * 
        intraday_volatility + open_close_spread
    )
    
    return final_alpha_factor
```
