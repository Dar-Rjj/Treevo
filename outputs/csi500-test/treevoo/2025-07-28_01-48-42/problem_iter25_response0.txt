```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Intraday Return Volatility
    df['Intraday_High_Returns'] = df['high'] - df['open']
    df['Intraday_Low_Returns'] = df['low'] - df['open']
    df['Intraday_Return_Volatility'] = (df[['Intraday_High_Returns', 'Intraday_Low_Returns']]
                                        .rolling(window=10)
                                        .std()
                                        .max(axis=1))

    # Compute Price Change
    df['Price_Change'] = df['close'] - df['close'].shift(1)

    # Adjust for Volume
    df['Volume_Adjusted_Range'] = df['volume'] * df['High_Low_Range']

    # Compute Momentum
    df['Momentum'] = df['Price_Change'].rolling(window=10).sum()

    # Adjust Momentum by Price Spread
    df['Daily_Price_Spread'] = np.abs(df['close'] - df['open'])
    df['Adjusted_Momentum'] = df['Momentum'] * df['Daily_Price_Spread']

    # Combine Adjusted Momentum and Volume
    df['Combined_Factor'] = df['Adjusted_Momentum'] / df['Volume_Adjusted_Range']

    # Enhance Signal with Smoothing
    df['Enhanced_Signal'] = df['Combined_Factor'].ewm(span=5, adjust=False).mean()

    # Generate Factor Value
    df['Generated_Factor'] = df['Momentum'] / df['close']

    # New Factor: Price-Volatility Ratio
    df['Price_Volatility_Ratio'] = df['Price_Change'] / df['Intraday_Return_Volatility']

    # Weighted Sum
    df['Weighted_Sum'] = (0.3 * df['Intraday_Return_Volatility'] +
                          0.2 * df['Enhanced_Signal'] +
                          0.4 * df['Generated_Factor'] +
                          0.1 * df['Price_Volatility_Ratio'])

    # Multiply by Volume Spike Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Std'] = df['volume'].rolling(window=20).std()
    df['Volume_Spike'] = (df['Volume_Change'] > 2 * df['Volume_Std']).astype(int)

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Weighted_Sum'] * df['Volume_Spike']

    return df['Alpha_Factor']
```
