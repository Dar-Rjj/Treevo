```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal_score = 1 / (intraday_return + 1e-6)  # Avoid division by zero
    
    # Adjust for Volume and Amount
    volume_amount_adjusted = high_low_range * df['volume'] * df['amount']
    
    # Integrate Momentum and Reversal
    high_low_diff = high_low_range.diff().fillna(0)
    combined_momentum_reversal = high_low_diff * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = combined_momentum_reversal + volume_amount_adjusted
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1).fillna(0)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1).fillna(0)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1).fillna(0)
    
    # Integrate Price Momentum and Volume Delta
    price_volume_momentum = price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open']
    final_factor = integrated_signal * close_to_open_diff
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_final_factor = final_factor.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2 / 4) + ((df['close'] - df['open'])**2 / 4))
    volatility_adjusted_factor = smoothed_final_factor * daily_volatility
    
    # Apply a Second Smoothing Layer
    final_smoothed_factor = volatility_adjusted_factor.rolling(window=10, min_periods=1).mean()
    
    return final_smoothed_factor
```
