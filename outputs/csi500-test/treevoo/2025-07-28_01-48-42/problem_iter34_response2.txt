```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return_high = df['high'] - df['open']
    intraday_return_low = df['open'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_reversal_score = 1 / (intraday_return_high + intraday_return_low)
    
    # Integrate Momentum and Reversal
    momentum_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume'] * df['amount']
    
    # Integrate Volume-Adjusted Reversal with C-O Diff
    close_open_diff = df['close'] - df['open']
    integrated_signal = volume_adjusted * close_open_diff
    
    # Generate Factor Value
    factor_value = (df['high'].rolling(window=10).mean() - df['low'].rolling(window=10).mean()) * df['volume'].rolling(window=10).mean()
    factor_value /= df['close']
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = integrated_signal.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2) / 4 + ((df['close'] - df['open'])**2) / 4)
    volatility_adjusted = smoothed_signal * daily_volatility
    
    # Apply a Second Smoothing Layer
    final_smoothed_signal = volatility_adjusted.rolling(window=10).mean()
    
    # Combine All Factors
    combined_factors = (intraday_return_high * 0.3) + (factor_value * 0.4) + (smoothed_signal * 0.2) + (final_smoothed_signal * 0.1)
    
    # Final Alpha Factor
    # Calculate Day-to-Day Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    final_alpha_factor = combined_factors * final_smoothed_signal * volume_spike
    
    return final_alpha_factor
```
