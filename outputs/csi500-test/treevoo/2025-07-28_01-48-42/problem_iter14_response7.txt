import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_high_return = df['high'] - df['open']
    intraday_low_return = df['low'] - df['open']
    intraday_volatility = (intraday_high_return + intraday_low_return).rolling(window=10).std()
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_signal = momentum / volume_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Generate Factor Value
    factor_value = enhanced_signal / df['close']
    
    # Compute Intraday Reversal Score
    intraday_return = df['high'] - df['low']
    intraday_reversal_score = 1 / intraday_return
    
    # Integrate Volume-Adjusted Range with Weighted Price Change
    weighted_price_change = (price_change * intraday_reversal_score) + volume_adjusted_range
    
    # Combine All Factors
    final_factor = (intraday_volatility * 0.3) + (factor_value * 0.7)
    
    return final_factor
