import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Momentum
    lookback_period = 10
    avg_close = df['close'].rolling(window=lookback_period).mean()
    momentum = df['close'] - avg_close
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * (1 + open_close_spread.abs())
    
    # Weight Adjusted Momentum by Volume
    ma_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_momentum = adjusted_momentum / ma_volume
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Integrate Momentum, Reversal, and Volume-Amount Adjustment
    high_low_momentum = high_low_range.rolling(window=lookback_period).sum()
    combined_momentum_reversal = volume_weighted_momentum * intraday_reversal * high_low_momentum
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Compute High-Low Range Adjusted by Volume
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Integrate Price Momentum and Volume Delta
    integrated_signal = price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    daily_volume_rank = df['volume'].rank(pct=True)
    daily_amount_rank = df['amount'].rank(pct=True)
    close_to_open_diff = df['close'] - df['open'].shift(1)
    
    # Final Factor
    final_factor = (combined_momentum_reversal * daily_volume_rank * daily_amount_rank * close_to_open_diff) + integrated_signal
    
    # Apply a Smoothing Filter to the Combined Signal
    ema_span = 10
    smoothed_final_factor = final_factor.ewm(span=ema_span).mean()
    
    return smoothed_final_factor
