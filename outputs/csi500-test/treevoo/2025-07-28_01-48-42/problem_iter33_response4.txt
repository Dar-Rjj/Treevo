import pandas as pd

def heuristics_v2(df, N=10, M=10, K=10, L=10, P=10):
    # Calculate the difference between open and close prices (OC_diff)
    df['OC_diff'] = df['close'] - df['open']
    
    # Compute the moving average of OC_diff over N days (MA_OC_diff)
    df['MA_OC_diff'] = df['OC_diff'].rolling(window=N).mean()
    
    # Calculate the range of daily price movement (High - Low)
    df['range'] = df['high'] - df['low']
    
    # Compute the moving average of the range over M days (MA_range)
    df['MA_range'] = df['range'].rolling(window=M).mean()
    
    # Calculate the ratio of volume to price change (Volume / (Close - Open))
    df['vol_price_ratio'] = df['volume'] / (df['close'] - df['open']).replace(0, 1)  # Avoid division by zero
    
    # Compute the moving average of this ratio over K days (MA_vol_price_ratio)
    df['MA_vol_price_ratio'] = df['vol_price_ratio'].rolling(window=K).mean()
    
    # Determine the percentage increase in volume compared to the previous day
    df['volume_increase'] = df['volume'].pct_change()
    
    # Create a binary indicator for significant volume spikes (1 if spike, 0 otherwise)
    df['vol_spike'] = (df['volume_increase'] > 2.0).astype(int)
    
    # Calculate the ratio of amount to price change (Amount / (Close - Open))
    df['amt_price_ratio'] = df['amount'] / (df['close'] - df['open']).replace(0, 1)  # Avoid division by zero
    
    # Compute the moving average of this ratio over L days (MA_amt_price_ratio)
    df['MA_amt_price_ratio'] = df['amt_price_ratio'].rolling(window=L).mean()
    
    # Compute the standard deviation of the amount over P days (Std_Amount)
    df['Std_Amount'] = df['amount'].rolling(window=P).std()
    
    # Combine all the sub-factors into a single alpha factor
    df['alpha_factor'] = (
        df['MA_OC_diff'] + 
        df['MA_range'] + 
        df['MA_vol_price_ratio'] + 
        df['vol_spike'] + 
        df['MA_amt_price_ratio'] + 
        df['Std_Amount']
    )
    
    return df['alpha_factor']
