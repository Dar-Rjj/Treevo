```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Adjust for Volume: Compute 5-day average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    high_low_range_vol_adj = high_low_range / avg_volume_5d
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range - high_low_range.shift(1)
    momentum_sum_5d = high_low_range_diff.rolling(window=5).sum()
    integrated_momentum = momentum_sum_5d * intraday_reversal_score
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    
    # Integrate into Final Factor
    final_factor = integrated_momentum * close_open_diff
    
    # Apply Moving Average Smoothing
    alpha_factor = final_factor.rolling(window=10).mean()
    
    return alpha_factor
```
