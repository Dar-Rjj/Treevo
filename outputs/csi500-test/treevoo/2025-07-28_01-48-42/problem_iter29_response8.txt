```python
import pandas as pd

def heuristics_v2(df):
    """
    Given a DataFrame with (open, high, low, close, amount, volume) prices,
    this function computes a novel alpha factor based on the relationship between
    the closing price and the volume traded, adjusted by the high-low price range.
    
    Parameters:
    df (pd.DataFrame): A DataFrame indexed by (date) with columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the high-low price range
    high_low_range = df['high'] - df['low']
    
    # Calculate the volume-adjusted closing price
    volume_adjusted_close = df['close'] * df['volume'] / high_low_range
    
    # The alpha factor is the difference between the volume-adjusted closing price and the opening price
    factor = volume_adjusted_close - df['open']
    
    return factor
```
