```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Returns
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']

    # Calculate Intraday Return Volatility
    intraday_return_volatility = (intraday_returns_high - intraday_returns_low).rolling(window=10).std()

    # Calculate Average Volume Spike
    volume_difference = df['volume'] - df['volume'].shift(1)
    average_volume_spike = volume_difference.rolling(window=5).sum()

    # Calculate Close Price Momentum
    price_change_20_days = df['close'] - df['close'].shift(20)
    close_price_momentum_ema = df['close'].ewm(span=14, adjust=False).mean()
    close_price_momentum = price_change_20_days / close_price_momentum_ema

    # Calculate Trading Range Intensity
    true_range_1 = df['high'] - df['low']
    true_range_2 = (df['high'] - df['close'].shift(1)).abs()
    true_range_3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([true_range_1, true_range_2, true_range_3], axis=1).max(axis=1)
    trading_range_intensity = true_range.rolling(window=10).mean()

    # Combine Factors
    factor_values = (
        0.5 * intraday_return_volatility +
        0.3 * average_volume_spike +
        0.1 * close_price_momentum +
        0.1 * trading_range_intensity
    )

    return factor_values
```
