```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['low'] - df['open']
    intraday_return_volatility = (intraday_high_returns.rolling(window=10).std() + 
                                  intraday_low_returns.rolling(window=10).std()) / 2
    
    # Compute Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_five_day_momentum = five_day_momentum * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_five_day_momentum / combined_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = enhanced_combined_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open'].shift(1)
    integrated_reversal = intraday_reversal * close_open_diff
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Generate New Factor Value
    generated_factor = five_day_momentum / df['close']
    
    # Integrate with Intraday Reversal
    integrated_factor = generated_factor + integrated_reversal
    
    # Final Alpha Factor
    trend_strength = df['close'].pct_change().ewm(span=20).mean()
    final_alpha_factor = (intraday_return_volatility * 0.3 + 
                          enhanced_combined_signal * 0.2 + 
                          integrated_factor * 0.4 + 
                          (price_change / intraday_return_volatility) * 0.1) * volume_spike * trend_strength
    
    return final_alpha_factor
```
