```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range.diff().rolling(window=5).sum()

    # Adjust for Volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    adjusted_momentum = high_low_momentum / avg_volume_10d

    # Apply Exponential Moving Average Smoothing
    smoothed_momentum = adjusted_momentum.ewm(span=10).mean()
    interim_alpha_factor_1 = smoothed_momentum

    # Calculate Logarithmic Daily Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))

    # Enhanced Momentum Component
    ema_21d = log_returns.ewm(span=21).mean()
    ema_60d = log_returns.ewm(span=60).mean()
    enhanced_momentum = ema_21d - ema_60d

    # Enhanced Volatility Component
    squared_returns = log_returns ** 2
    sum_squared_21d = squared_returns.rolling(window=21).sum()
    sum_squared_60d = squared_returns.rolling(window=60).sum()
    enhanced_volatility = sum_squared_21d - sum_squared_60d

    # Enhanced Volume Component
    log_volume_changes = np.log(df['volume'] / df['volume'].shift(1))
    ema_volume_21d = log_volume_changes.ewm(span=21).mean()
    ema_volume_60d = log_volume_changes.ewm(span=60).mean()
    enhanced_volume = ema_volume_21d - ema_volume_60d

    # Combine Components
    final_alpha_factor_2 = enhanced_momentum + enhanced_volatility + enhanced_volume

    # Final Alpha Factor
    final_alpha_factor = interim_alpha_factor_1 + final_alpha_factor_2

    return final_alpha_factor
```
