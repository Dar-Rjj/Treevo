```python
def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the money flow by multiplying the average price of the day by the volume
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['average_price'] * df['volume']
    
    # Calculate the relative strength by dividing the amount by the money flow
    df['relative_strength'] = df['amount'] / df['money_flow']
    
    # Calculate the momentum by taking the difference between the current close and the close 5 days ago
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # The alpha factor is a combination of the price change, relative strength, and momentum
    df['alpha_factor'] = df['price_change'] * df['relative_strength'] * df['momentum']
    
    return df['alpha_factor']
```
