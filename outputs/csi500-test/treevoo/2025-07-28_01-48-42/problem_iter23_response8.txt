```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.exp(-intraday_return))
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_range = volume_adjusted_range + amount_adjusted_range
    
    # Integrate Momentum and Reversal
    momentum = high_low_range - high_low_range.shift(1)
    integrated_momentum_reversal = momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'] - df['close'].shift(1)
    weighted_price_change = (price_change * df['volume']) + (price_change * np.sqrt(df['amount']))
    combined_signal = integrated_momentum_reversal + combined_range + weighted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_open_diff = df['close'] - df['open'].shift(-1)
    final_factor = smoothed_signal * close_open_diff * intraday_reversal_score
    
    # Calculate Smoothed Price Momentum
    daily_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = final_factor * smoothed_momentum * volume_spike + close_open_diff
    
    return final_alpha_factor
```
