```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    intraday_reversal_score = 1 - intraday_return
    
    # Compute Daily Volume Rank
    daily_volume_rank = df['volume'].rolling(window=21).apply(lambda x: (x[-1] > x[:-1]).mean(), raw=False)
    
    # Compute Daily Amount Rank
    daily_amount_rank = df['amount'].rolling(window=21).apply(lambda x: (x[-1] > x[:-1]).mean(), raw=False)
    
    # Adjust by Volume and Amount Relative Strength
    adjusted_reversal = intraday_reversal_score * daily_volume_rank * daily_amount_rank
    
    # Calculate Close-to-Open Difference
    close_to_open_diff = df['close'] - df['open']
    
    # Integrate into Final Factor
    final_factor = adjusted_reversal * close_to_open_diff
    
    return final_factor
```
