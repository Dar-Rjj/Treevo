```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.abs(intraday_return))
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    adjusted_high_low_range = volume_adjusted + amount_adjusted
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range.diff()
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'].diff()
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    integrated_signal = combined_momentum_reversal + weighted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = np.abs(df['open'] - df['close'])
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = integrated_signal.ewm(span=7, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'].shift(1) - df['open']
    final_factor = smoothed_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_price_momentum = daily_close_return.ewm(span=15, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].pct_change()
    volume_spike = (day_to_day_volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Close-to-Open Price Change
    close_to_open_price_change = df['close'].shift(1) - df['open']
    
    # Calculate 3-Day Price Momentum
    three_day_momentum = price_change.rolling(window=3).sum()
    
    # Adjust 3-Day Momentum by Volume Spike
    adjusted_three_day_momentum = three_day_momentum * volume_spike
    
    # Final Alpha Factor
    final_alpha_factor = (final_factor * smoothed_price_momentum * adjusted_three_day_momentum) + close_to_open_price_change
    
    return final_alpha_factor
```
