```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['low'] - df['open']
    intraday_volatility = (intraday_high_returns + intraday_low_returns).rolling(window=10).std()
    
    # Calculate Daily Price Spread
    daily_price_spread = df['close'] - df['open']
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Price Spread
    adjusted_momentum = momentum * abs(daily_price_spread)
    
    # Combine Adjusted Momentum and Volume
    combined_factor = adjusted_momentum / volume_adjusted
    
    # Enhance Signal with Smoothing
    enhanced_signal = combined_factor.ewm(span=5).mean()
    
    # Generate Factor Value
    generated_factor = momentum / df['close']
    
    # New Factor: Price-Volatility Ratio
    price_volatility_ratio = price_change / intraday_volatility
    
    # Final Alpha Factor
    final_alpha = 0.3 * intraday_volatility + 0.2 * enhanced_signal + 0.4 * generated_factor + 0.1 * price_volatility_ratio
    
    # Multiply by Volume Spike Indicator
    volume_change = df['volume'].diff()
    volume_spike_indicator = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    final_alpha *= volume_spike_indicator
    
    return final_alpha
```
