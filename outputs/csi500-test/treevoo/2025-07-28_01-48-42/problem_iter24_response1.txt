```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range * (avg_volume_5d * avg_amount_5d)
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    integrated_momentum_reversal = high_low_momentum * intraday_reversal
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    daily_volume_rank = df['volume'].rank(pct=True)
    daily_amount_rank = df['amount'].rank(pct=True)
    close_to_open_diff = df['close'] - df['open']
    
    # Combine Integrated Momentum, Intraday Reversal, and Volume-Amount-Adjusted Range
    combined_factor = (integrated_momentum_reversal + 
                       adjusted_high_low_range * 
                       daily_volume_rank * 
                       daily_amount_rank * 
                       close_to_open_diff)
    
    # Apply a Smoothing Filter
    ema_span = 5
    smoothed_factor = combined_factor.ewm(span=ema_span).mean()
    
    # New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + 
                               ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted_factor = smoothed_factor * daily_volatility
    
    # Apply a Second Smoothing Layer
    sma_window = 10
    final_smoothed_factor = volatility_adjusted_factor.rolling(window=sma_window).mean()
    
    # Incorporate Price-Volatility Relationship
    price_volatility_ratio = df['close'] / daily_volatility
    final_factor = final_smoothed_factor * price_volatility_ratio
    
    # On-Balance Volume (OBV)
    obv = (np.sign(price_change) * df['volume']).cumsum()
    final_factor_with_obv = final_factor + obv
    
    return final_factor_with_obv
```
