```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    df['reversal_score'] = 1 - df['intraday_return']
    
    # Compute Daily Volume Rank
    df['volume_rank'] = df['volume'].rolling(window=20).rank(pct=True)
    
    # Adjust by Volume Relative Strength
    df['volume_adjusted_reversal'] = df['reversal_score'] * df['volume_rank']
    
    # Consider Close-to-Open Price Change
    df['close_to_open_diff'] = df['close'] - df['open'].shift(1)
    
    # Integrate into Final Factor
    df['factor_part_1'] = df['volume_adjusted_reversal'] * df['close_to_open_diff']
    
    # Calculate Price Change Momentum
    df['price_momentum'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Delta
    df['combined_signal'] = df['price_momentum'] * df['volume_delta']
    
    # Apply a Smoothing Filter using Exponential Moving Average (EMA)
    df['smoothed_signal'] = df['combined_signal'].ewm(span=5, adjust=False).mean()
    
    # Combine with Dynamic Volume-Momentum
    df['final_factor'] = df['factor_part_1'] + df['smoothed_signal']
    
    return df['final_factor']
```
