```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate 10-Day High-Low Price Momentum
    high_low_momentum = high_low_diff.rolling(window=10).sum()
    
    # Calculate 10-Day Volume Moving Average
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    
    # Combine Momentum and Volume
    combined_momentum_volume = high_low_momentum * volume_moving_avg
    
    # Generate Factor Value
    factor_value = combined_momentum_volume / df['close']
    
    return factor_value
```
