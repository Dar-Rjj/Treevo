```python
import pandas as pd

def heuristics_v2(df):
    # Constants
    lookback_period = 10
    
    # Calculate Daily Open-Close Price Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    df['abs_open_close_spread'] = df['open_close_spread'].abs()
    df['adjusted_momentum'] = df['momentum'] / df['abs_open_close_spread']
    
    # Weight Adjusted Momentum by Volume
    df['weighted_adjusted_momentum'] = df['adjusted_momentum'] * df['volume']
    df['sum_weighted_momentum'] = df['weighted_adjusted_momentum'].rolling(window=lookback_period).sum()
    df['total_volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['final_alpha_factor_1'] = df['sum_weighted_momentum'] / df['total_volume']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['true_range'] = df[['high_low_diff', 'high' - df['close'].shift(1), df['close'].shift(1) - df['low']]].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=lookback_period).mean()
    
    # Weight Intraday Volatility by Volume
    df['weighted_intraday_volatility'] = df['atr'] * df['volume']
    df['sum_weighted_intraday_volatility'] = df['weighted_intraday_volatility'].rolling(window=lookback_period).sum()
    df['final_alpha_factor_2'] = df['sum_weighted_intraday_volatility'] / df['total_volume']
    
    # Combine Final Alpha Factors
    df['final_alpha_factor'] = df['final_alpha_factor_1'] + df['final_alpha_factor_2']
    
    # Return the final alpha factor
    return df['final_alpha_factor']
```
