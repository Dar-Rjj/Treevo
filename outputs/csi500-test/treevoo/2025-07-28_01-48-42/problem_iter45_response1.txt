```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_momentum=10, lookback_volatility=10, lookback_consistency=20, lookback_vw_return=5):
    # Calculate Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].rolling(window=lookback_momentum).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * spread.abs()
    
    # Calculate Daily High-Low Range
    range_ = df['high'] - df['low']
    
    # Further Adjust Momentum by High-Low Range
    final_adjusted_momentum = adjusted_momentum / range_
    
    # Weight Final Adjusted Momentum by Volume
    weighted_momentum = final_adjusted_momentum * df['volume']
    
    # Final Alpha Factor Component 1
    total_volume = df['volume'].rolling(window=lookback_momentum).sum()
    component_1 = weighted_momentum.rolling(window=lookback_momentum).sum() / total_volume
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    true_range = high_low_diff
    atr = true_range.rolling(window=lookback_volatility).mean()
    weighted_intraday_volatility = atr * df['volume']
    
    # Final Alpha Factor Component 2
    component_2 = weighted_intraday_volatility.rolling(window=lookback_volatility).sum() / total_volume
    
    # Calculate Price Movement Consistency
    ema_close = df['close'].ewm(span=lookback_consistency, adjust=False).mean()
    daily_returns = df['close'].pct_change()
    std_dev_returns = daily_returns.rolling(window=lookback_consistency).std()
    component_3 = std_dev_returns / ema_close
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = (df['close'].pct_change() * df['volume']).rolling(window=lookback_vw_return).mean() / df['volume'].rolling(window=lookback_vw_return).mean()
    
    # Combine Final Alpha Factors
    alpha_factor = component_1 + component_2 + component_3 + volume_weighted_return
    
    return alpha_factor
```
