```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range.diff()
    high_low_momentum_sum = high_low_momentum.rolling(window=15).sum()
    
    # Adjust for Volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    adjusted_momentum = high_low_momentum_sum / avg_volume_5
    
    # Apply Exponential Moving Average Smoothing
    smoothed_momentum = adjusted_momentum.ewm(span=20, adjust=False).mean()
    interim_alpha_factor_1 = smoothed_momentum
    
    # Calculate Logarithmic Daily Returns
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    ema_21_returns = daily_returns.ewm(span=21, adjust=False).mean()
    ema_63_returns = daily_returns.ewm(span=63, adjust=False).mean()
    momentum_component = ema_21_returns - ema_63_returns
    
    # Volatility Component
    abs_daily_returns = daily_returns.abs()
    ema_21_abs_returns = abs_daily_returns.ewm(span=21, adjust=False).mean()
    ema_63_abs_returns = abs_daily_returns.ewm(span=63, adjust=False).mean()
    volatility_component = ema_21_abs_returns - ema_63_abs_returns
    
    # Volume Component
    log_volume_changes = np.log(df['volume'] / df['volume'].shift(1))
    ema_21_volume_changes = log_volume_changes.ewm(span=21, adjust=False).mean()
    ema_63_volume_changes = log_volume_changes.ewm(span=63, adjust=False).mean()
    volume_momentum = ema_21_volume_changes - ema_63_volume_changes
    
    # Open-Close Spread
    open_close_diff = df['open'] - df['close']
    open_close_spread = open_close_diff.ewm(span=21, adjust=False).mean()
    
    # Combine Components
    final_momentum = momentum_component
    final_volatility = volatility_component
    final_volume_momentum = volume_momentum
    final_open_close_spread = open_close_spread
    final_alpha_factor_2 = final_momentum + final_volatility + final_volume_momentum + final_open_close_spread
    
    # Final Alpha Factor
    final_alpha_factor = interim_alpha_factor_1 * final_alpha_factor_2
    
    return final_alpha_factor
```
