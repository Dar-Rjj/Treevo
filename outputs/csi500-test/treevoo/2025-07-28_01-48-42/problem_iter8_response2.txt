```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']

    # Calculate Price Change
    price_change = df['close'].diff()

    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()

    # Adjust for Volume
    volume_adjusted_momentum = momentum * (df['volume'] * high_low_range)

    # Adjust Momentum by Open-Close Spread
    spread_adjusted_momentum = volume_adjusted_momentum * abs(open_close_spread)

    # Combine Volume-Adjusted Momentum and Spread
    combined_momentum_spread = spread_adjusted_momentum / (df['volume'] * high_low_range)

    # Calculate Intraday Return Volatility
    intraday_high_return = df['high'] - df['open']
    intraday_low_return = df['low'] - df['open']
    intraday_returns = pd.concat([intraday_high_return, intraday_low_return], axis=1).max(axis=1)
    intraday_volatility = intraday_returns.rolling(window=10).std()

    # Calculate Average Volume Spike
    volume_difference = df['volume'].diff()
    average_volume_spike = volume_difference.rolling(window=5).sum()

    # Calculate Trading Range Intensity
    true_range_1 = df['high'] - df['low']
    true_range_2 = (df['high'] - df['close'].shift(1)).abs()
    true_range_3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([true_range_1, true_range_2, true_range_3], axis=1).max(axis=1)
    trading_range_intensity = true_range.rolling(window=10).mean()

    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()

    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].diff()
    volume_spike_factor = (day_to_day_volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)

    # Final Alpha Factor
    final_alpha_factor = (combined_momentum_spread * smoothed_momentum) * volume_spike_factor
    weighted_intraday_volatility = (intraday_volatility * 0.3) + (average_volume_spike * 0.2) + (trading_range_intensity * 0.1)
    combined_factors = (final_alpha_factor * 0.5) + (weighted_intraday_volatility * 0.5)

    return combined_factors
```
