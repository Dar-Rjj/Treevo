```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate High-Low Price Momentum (sum over the last 10 days)
    high_low_momentum = high_low_diff.rolling(window=10).sum()
    
    # Calculate Volume Trend (10-day Moving Average of Volume)
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Combine High-Low Price Momentum and Volume Trend
    combined_momentum_volume = high_low_momentum * volume_trend
    
    # Calculate Factor Value (divide by Close price)
    factor_value = combined_momentum_volume / df['close']
    
    return factor_value
```
