import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate 5-Day High-Low Range Momentum
    high_low_momentum = high_low_range.diff().rolling(window=5).sum()

    # Adjust for Volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    adjusted_high_low_momentum = high_low_momentum / avg_volume_10d

    # Apply Exponential Moving Average Smoothing
    smoothed_adjusted_momentum = adjusted_high_low_momentum.ewm(span=10, min_periods=10).mean()
    interim_alpha_factor_1 = smoothed_adjusted_momentum

    # Calculate Logarithmic Daily Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))

    # Enhanced Momentum Component
    ema_21d = log_returns.ewm(span=21, min_periods=21).mean()
    ema_60d = log_returns.ewm(span=60, min_periods=60).mean()
    enhanced_momentum = ema_21d - ema_60d

    # Enhanced Volatility Component
    squared_returns = log_returns ** 2
    sum_squared_21d = squared_returns.rolling(window=21).sum()
    sum_squared_60d = squared_returns.rolling(window=60).sum()
    enhanced_volatility = sum_squared_21d - sum_squared_60d

    # Enhanced Volume Component
    log_volume_changes = np.log(df['volume'] / df['volume'].shift(1))
    ema_volume_21d = log_volume_changes.ewm(span=21, min_periods=21).mean()
    ema_volume_60d = log_volume_changes.ewm(span=60, min_periods=60).mean()
    enhanced_volume_momentum = ema_volume_21d - ema_volume_60d

    # Combine Components
    final_alpha_factor_2 = enhanced_momentum + enhanced_volatility + enhanced_volume_momentum

    # Integrate Open and Close Prices
    open_close_range = df['open'] - df['close']
    open_close_to_high_low_ratio = open_close_range / high_low_range

    # Apply Additional Momentum on Open-Close Range
    open_close_momentum = (open_close_range - open_close_range.shift(5)).rolling(window=5).sum()

    # Adjust for Volume Again
    adjusted_open_close_momentum = open_close_momentum / avg_volume_10d

    # Apply Exponential Moving Average Smoothing Again
    smoothed_adjusted_open_close_momentum = adjusted_open_close_momentum.ewm(span=10, min_periods=10).mean()
    interim_alpha_factor_3 = smoothed_adjusted_open_close_momentum

    # Combine All Components
    final_alpha_factor = interim_alpha_factor_1 + final_alpha_factor_2 + interim_alpha_factor_3

    return final_alpha_factor
