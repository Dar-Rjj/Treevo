import pandas as pd

def heuristics_v2(df):
    # Calculate intraday movement
    df['Intraday_Movement'] = df['close'] - df['open']
    
    # Calculate daily return
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-day, 10-day, and 20-day average daily returns
    df['5Day_Avg_Return'] = df['Daily_Return'].rolling(window=5).mean()
    df['10Day_Avg_Return'] = df['Daily_Return'].rolling(window=10).mean()
    df['20Day_Avg_Return'] = df['Daily_Return'].rolling(window=20).mean()
    
    # Calculate the range (High - Low) as a measure of daily volatility
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate True Range
    df['True_Range'] = df.apply(
        lambda row: max(row['high'] - row['low'], 
                        abs(row['high'] - df['close'].shift(1).fillna(0)), 
                        abs(row['low'] - df['close'].shift(1).fillna(0))), axis=1)
    
    # Calculate 10-day and 20-day Average True Range (ATR)
    df['10Day_ATR'] = df['True_Range'].rolling(window=10).mean()
    df['20Day_ATR'] = df['True_Range'].rolling(window=20).mean()
    
    # Calculate the Volume to Price Ratio
    df['Volume_to_Price_Ratio'] = df['volume'] / (df['close'] - df['open'])
    
    # Calculate volume change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate 5-day and 10-day moving sum of volume change
    df['5Day_Volume_Change_Sum'] = df['Volume_Change'].rolling(window=5).sum()
    df['10Day_Volume_Change_Sum'] = df['Volume_Change'].rolling(window=10).sum()
    
    # Calculate the Amount to Price Ratio
    df['Amount_to_Price_Ratio'] = df['amount'] / df['close']
    
    # Calculate Daily VWAP
    df['Daily_VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3) / df['volume']
    
    # Calculate the percentage change in closing price over a fixed period (e.g., 5 days)
    df['5Day_Price_Change'] = df['close'].pct_change(periods=5)
    
    # Calculate 14-day and 7-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain_14 = gain.rolling(window=14).mean()
    avg_loss_14 = loss.rolling(window=14).mean()
    rs_14 = avg_gain_14 / avg_loss_14
    df['14Day_RSI'] = 100 - (100 / (1 + rs_14))
    
    avg_gain_7 = gain.rolling(window=7).mean()
    avg_loss_7 = loss.rolling(window=7).mean()
    rs_7 = avg_gain_7 / avg_loss_7
    df['7Day_RSI'] = 100 - (100 / (1 + rs_7))
    
    # Calculate the moving average of the closing price over a short and long period
    df['10Day_MA'] = df['close'].rolling(window=10).mean()
    df['50Day_MA'] = df['close'].rolling(window=50).mean()

    df['Trend_Factor'] = df['10Day_MA'] - df['50Day_MA']
