import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return Volatility
    intraday_return_high = df['high'] - df['open']
    intraday_return_low = df['low'] - df['open']
    intraday_returns = (intraday_return_high + intraday_return_low) / 2
    intraday_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] + df['low']) / 2 - df['open']
    intraday_reversal_score = -intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['amount'] * df['volume']
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range.diff(1)
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_momentum_reversal = combined_momentum_reversal + volume_adjusted_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    price_volume_momentum = price_change_momentum * volume_delta
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open']
    
    # Generate Factor Value
    factor_value = (high_low_range.rolling(window=10).mean() * df['volume'].rolling(window=10).mean()) / df['close']
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = integrated_momentum_reversal.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt((df['high'] - df['low'])**2 / 4 + (df['close'] - df['open'])**2 / 4)
    final_factor_with_volatility = factor_value * daily_volatility
    
    # Apply a Second Smoothing Layer
    smoothened_momentum = smoothed_signal.rolling(window=10).mean().ewm(span=5, adjust=False).mean()
    
    # Combine All Factors
    combined_factors = (intraday_volatility * 0.3 +
                        factor_value * 0.4 +
                        smoothed_signal * 0.2 +
                        final_factor_with_volatility * 0.1)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_factor = combined_factors + integrated_momentum_reversal
    
    # Final Alpha Factor
    volume_spike = (volume_delta > 2 * df['volume'].rolling(window=20).std()).astype(int)
    final_alpha_factor = final_factor * smoothened_momentum * volume_spike
    
    return final_alpha_factor
