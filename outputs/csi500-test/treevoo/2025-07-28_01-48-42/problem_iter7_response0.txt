```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    factor_values = momentum / volume_amount_adjusted_range
    
    return factor_values
```
