import pandas as pd

def heuristics_v2(df):
    # Basic Alpha Factors
    df['price_momentum_1'] = df['close'].pct_change()
    df['price_momentum_5'] = df['close'].pct_change(5)
    df['price_momentum_20'] = df['close'].pct_change(20)
    
    df['volume_momentum_1'] = df['volume'].pct_change()
    df['volume_momentum_5'] = df['volume'].pct_change(5)
    df['volume_momentum_20'] = df['volume'].pct_change(20)
    
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range_5'] = df['high_low_range'].rolling(window=5).mean()
    df['avg_high_low_range_20'] = df['high_low_range'].rolling(window=20).mean()
    
    # Volatility-Based Alpha Factors
    df['price_volatility_5'] = df['close'].pct_change().rolling(window=5).std()
    df['price_volatility_20'] = df['close'].pct_change().rolling(window=20).std()
    
    df['volume_volatility_5'] = df['volume'].pct_change().rolling(window=5).std()
    df['volume_volatility_20'] = df['volume'].pct_change().rolling(window=20).std()
    
    # Technical Indicators
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['ma_diff'] = df['ma_5'] - df['ma_20']
    
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_diff'] = df['ema_5'] - df['ema_20']
    
    # Relative Strength Indicators
    # Assuming 'benchmark' is a column in the DataFrame representing the benchmark index
    df['rel_price_strength_5'] = df['close'].pct_change() - df['benchmark'].pct_change()
    df['rel_price_strength_20'] = df['close'].pct_change(20) - df['benchmark'].pct_change(20)
    
    # Assuming 'market_volume' is a column in the DataFrame representing the average market volume
    df['rel_volume_strength_5'] = df['volume'].pct_change() - df['market_volume'].pct_change()
    df['rel_volume_strength_20'] = df['volume'].pct_change(20) - df['market_volume'].pct_change(20)
    
    # Trend and Reversal Indicators
    df['trend_continuation_up'] = (df['close'].diff() > 0).cumsum()
    df['trend_continuation_down'] = (df['close'].diff() < 0).cumsum()
    
    # Reversal Patterns (Simplified for this example)
    df['hammer'] = (df['close'] > df['open']) & (df['low'] < df['close']) & (df['close'] - df['open'] < df['open'] - df['low'])
    df['hanging_man'] = (df['close'] < df['open']) & (df['low'] < df['close']) & (df['open'] - df['close'] < df['open'] - df['low'])
    df['bullish_engulfing'] = (df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & (df['close'] > df['open'].shift(1))
    df['bearish_engulfing'] = (df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1)) & (df['close'] < df['open'].shift(1))
    
    # Market Sentiment Indicators
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df[df['typical_price'] > df['typical_price'].shift(1)]['money_flow']
    df['negative_money_flow'] = df[df['typical_price'] < df['typical_price'].shift(1)]['money_flow
