```python
import pandas as pd

def heuristics_v2(df):
    # Compute Open-Close Spread
    spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].shift(lookback_period)
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * spread.abs()
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Incorporate Enhanced Price Trend Sensitivity
    trend_direction = (df['close'] > df['open']).astype(int)  # 1 if positive, 0 if negative
    high_low_range = df['high'] - df['low']
    price_trend_factor = (1 + (high_low_range / df['open']) * (2 * trend_direction - 1))
    trend_adjusted_momentum = volume_weighted_momentum * price_trend_factor
    
    # Incorporate Opening Price Volatility
    open_vol_lookback = 5
    opening_price_volatility = df['open'].rolling(window=open_vol_lookback).std()
    open_vol_factor = (1 + (opening_price_volatility / df['open']) * (2 * trend_direction - 1))
    open_vol_adjusted_momentum = trend_adjusted_momentum * open_vol_factor
    
    # Incorporate Trading Volume Volatility
    vol_vol_lookback = 5
    volume_volatility = df['volume'].rolling(window=vol_vol_lookback).std()
    avg_volume = df['volume'].rolling(window=vol_vol_lookback).mean()
    volume_vol_factor = (1 + (volume_volatility / avg_volume) * (2 * trend_direction - 1))
    volume_vol_adjusted_momentum = open_vol_adjusted_momentum * volume_vol_factor
    
    # Incorporate Amount-Driven Volatility
    amount_vol_lookback = 5
    amount_volatility = df['amount'].rolling(window=amount_vol_lookback).std()
    avg_amount = df['amount'].rolling(window=amount_vol_lookback).mean()
    amount_vol_factor = (1 + (amount_volatility / avg_amount) * (2 * trend_direction - 1))
    final_momentum = volume_vol_adjusted_momentum * amount_vol_factor
    
    return final_momentum
```
