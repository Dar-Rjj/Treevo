import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['close']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (intraday_return + 1e-6)  # Avoid division by zero
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['amount'] * df['volume']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Momentum and Intraday Reversal
    momentum_intraday_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_momentum_reversal = momentum_intraday_reversal + volume_amount_adjusted_range + weighted_price_change
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    volume_adjusted_reversal_close_open = intraday_reversal_score * close_open_diff * df['volume']
    
    # Final Alpha Factor
    final_factor = integrated_momentum_reversal + volume_adjusted_reversal_close_open
    
    return final_factor
