```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Calculate Price Change
    price_change = data['close'].diff()
    
    # Compute Intraday Reversal Score
    intraday_return = (data['high'] - data['low']) / data['open']
    intraday_reversal_score = 1 / intraday_return
    
    # Adjust for Volume and Amount
    volume_amount_adjusted = data['volume'] * high_low_range * data['amount']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * data['volume']
    
    # Combine Momentum and Intraday Reversal
    high_low_momentum = high_low_range.diff()
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = combined_momentum_reversal + volume_amount_adjusted + weighted_price_change
    
    # Consider Close-to-Open Price Change
    close_open_diff = data['close'] - data['open']
    integrated_with_close_open = integrated_factor + (intraday_reversal_score * close_open_diff * data['volume'])
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = data['close'] - data['open']
    
    # Compute Weighted Momentum
    weighted_momentum = (price_change * data['volume']).rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = weighted_momentum * open_close_spread.abs()
    
    # Volume-Adjust High-Low Range
    volume_adjusted_high_low = data['volume'] * high_low_range
    
    # Introduce Open-Price Based Momentum
    open_price_change = data['open'].diff()
    open_price_based_momentum = open_price_change * data['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_with_close_open + 
                          (adjusted_momentum / volume_adjusted_high_low) + 
                          open_price_based_momentum)
    
    return final_alpha_factor
```
