```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(0) - df['close'].shift(1)
    
    # Compute Momentum (sum over the last 5 days)
    momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum = momentum / volume_adjusted_range
    
    # Adjust for Amount
    amount_adjusted = combined_momentum * df['amount']
    
    # Final Alpha Factor
    alpha_factor = amount_adjusted + weighted_price_change
    
    return alpha_factor
```
