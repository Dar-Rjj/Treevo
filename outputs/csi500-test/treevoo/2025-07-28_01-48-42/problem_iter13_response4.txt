```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Adjust for Volume and Amount
    volume_amount_adjustment = high_low_range * df['volume'] * df['amount']
    
    # Combine Momentum and Intraday Reversal
    high_low_momentum = high_low_range.diff()
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = combined_momentum_reversal + volume_amount_adjustment + price_change
    
    # Final Alpha Factor
    final_alpha_factor = integrated_factor
    
    return final_alpha_factor
```
