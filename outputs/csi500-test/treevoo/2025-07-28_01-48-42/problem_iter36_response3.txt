```python
import pandas as pd

def heuristics_v2(df, N=10, Q=5, M=10, R=5, K=10, S=5, T=30, L=10, U=5, P=10, V=5):
    # Calculate the difference between open and close prices (OC_diff)
    OC_diff = df['close'] - df['open']
    
    # Compute the moving average of OC_diff over N days (MA_OC_diff)
    MA_OC_diff = OC_diff.rolling(window=N).mean()
    
    # Calculate the rate of change of MA_OC_diff over Q days (ROC_MA_OC_diff)
    ROC_MA_OC_diff = MA_OC_diff.pct_change(periods=Q)
    
    # Calculate the range of daily price movement (High - Low)
    price_range = df['high'] - df['low']
    
    # Compute the moving average of the range over M days (MA_range)
    MA_range = price_range.rolling(window=M).mean()
    
    # Calculate the rate of change of MA_range over R days (ROC_MA_range)
    ROC_MA_range = MA_range.pct_change(periods=R)
    
    # Calculate the ratio of volume to price change (Volume / (Close - Open))
    vol_price_ratio = df['volume'] / (df['close'] - df['open'])
    
    # Compute the moving average of this ratio over K days (MA_vol_price_ratio)
    MA_vol_price_ratio = vol_price_ratio.rolling(window=K).mean()
    
    # Calculate the rate of change of MA_vol_price_ratio over S days (ROC_MA_vol_price_ratio)
    ROC_MA_vol_price_ratio = MA_vol_price_ratio.pct_change(periods=S)
    
    # Determine the percentage increase in volume compared to the previous day
    vol_increase = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Create a binary indicator for significant volume spikes (1 if spike, 0 otherwise)
    vol_spike = (vol_increase > 0.5).astype(int)
    
    # Calculate the frequency of significant volume spikes over T days (Freq_Vol_Spikes)
    Freq_Vol_Spikes = vol_spike.rolling(window=T).sum() / T
    
    # Calculate the ratio of amount to price change (Amount / (Close - Open))
    amt_price_ratio = df['amount'] / (df['close'] - df['open'])
    
    # Compute the moving average of this ratio over L days (MA_amt_price_ratio)
    MA_amt_price_ratio = amt_price_ratio.rolling(window=L).mean()
    
    # Calculate the rate of change of MA_amt_price_ratio over U days (ROC_MA_amt_price_ratio)
    ROC_MA_amt_price_ratio = MA_amt_price_ratio.pct_change(periods=U)
    
    # Compute the standard deviation of the amount over P days (Std_Amount)
    Std_Amount = df['amount'].rolling(window=P).std()
    
    # Calculate the rate of change of Std_Amount over V days (ROC_Std_Amount)
    ROC_Std_Amount = Std_Amount.pct_change(periods=V)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (ROC_MA_OC_diff + ROC_MA_range + ROC_MA_vol_price_ratio + 
                    Freq_Vol_Spikes + ROC_MA_amt_price_ratio + ROC_Std_Amount)
    
    return alpha_factor
```
