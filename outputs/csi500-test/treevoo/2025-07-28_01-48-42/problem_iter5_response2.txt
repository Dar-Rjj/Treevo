import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change Momentum
    price_change_momentum = df['close'].shift(-1) - df['open']
    
    # Calculate Volume Delta
    volume_delta = df['volume'].diff()
    
    # Compute High-Low Range Adjusted by Volume
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Combine Momentum and Volume-Adjusted High-Low Range
    momentum_sum_10_days = price_change_momentum.rolling(window=10).sum()
    combined_signal = momentum_sum_10_days / volume_adjusted_high_low
    
    # Integrate Price Momentum and Volume Delta
    integrated_signal = price_change_momentum * volume_delta
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    return smoothed_signal
