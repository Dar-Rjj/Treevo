```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['low'] - df['open']
    intraday_volatility = (intraday_high_returns.rolling(window=10).std() + 
                           intraday_low_returns.rolling(window=10).std()) / 2
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * df['amount']**0.5
    combined_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust Momentum by Open-Close Spread
    momentum_adjusted = five_day_momentum * abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted / combined_adjusted_range
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * df['amount']**0.5
    
    # Combine Adjusted Range with Weighted Price Change
    factor_component_1 = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open'].shift(1)
    reversed_close_open = intraday_reversal * close_open_diff
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Generate New Factor Value
    new_factor = (five_day_momentum / df['close']) + reversed_close_open
    
    # Calculate Price-Volatility Ratio
    price_volatility_ratio = price_change / intraday_volatility
    
    # Calculate Long-Term Trend Strength
    long_term_momentum = df['close'].pct_change().rolling(window=20).mean()
    trend_strength = long_term_momentum.ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = (intraday_volatility * 0.3 + 
                   enhanced_signal * 0.2 + 
                   new_factor * 0.4 + 
                   price_volatility_ratio * 0.1)
    
    final_alpha = final_alpha * volume_spike * trend_strength
    
    # Integrate with Market Sentiment
    market_sentiment = (df['close'] - df['open']).rolling(window=10).mean()
    final_alpha = final_alpha * market_sentiment
    
    return final_alpha
```
