```python
import pandas as pd

def heuristics_v2(df, N=10, M=10, K=10, L=10, P=10, Q=10, R=10, S=10, T=10, U=50, V=50):
    # Sub-thought: Analyzing the relationship between open and close prices
    df['OC_diff'] = df['close'] - df['open']
    df['MA_OC_diff'] = df['OC_diff'].rolling(window=N).mean()

    # Sub-thought: Investigate the trend of high and low prices
    df['range'] = df['high'] - df['low']
    df['MA_range'] = df['range'].rolling(window=M).mean()

    # Sub-thought: Examine the relationship between volume and price changes
    df['vol_price_ratio'] = df['volume'] / (df['close'] - df['open'])
    df['MA_vol_price_ratio'] = df['vol_price_ratio'].rolling(window=K).mean()

    # Sub-thought: Identify significant volume spikes
    df['volume_spike'] = (df['volume'] / df['volume'].shift(1) - 1) > 0.5
    df['volume_spike'] = df['volume_spike'].astype(int)

    # Sub-thought: Explore the relationship between amount and price changes
    df['amt_price_ratio'] = df['amount'] / (df['close'] - df['open'])
    df['MA_amt_price_ratio'] = df['amt_price_ratio'].rolling(window=L).mean()

    # Sub-thought: Analyze the consistency of amount
    df['Std_Amount'] = df['amount'].rolling(window=P).std()

    # Sub-thought: Calculate the rate of change of closing prices over Q days
    df['ROC_close'] = df['close'].pct_change(periods=Q)
    df['MA_ROC_close'] = df['ROC_close'].rolling(window=R).mean()

    # Sub-thought: Investigate Volatility
    df['daily_return'] = df['close'].pct_change()
    df['Std_Daily_Returns'] = df['daily_return'].rolling(window=S).std()
    df['MA_Std_Daily_Returns'] = df['Std_Daily_Returns'].rolling(window=T).mean()

    # Sub-thought: Interpreting Moving Averages
    df['SMA_close_U'] = df['close'].rolling(window=U).mean()
    df['EMA_close_V'] = df['close'].ewm(span=V, adjust=False).mean()
    df['EMA_SMA_diff'] = df['EMA_close_V'] - df['SMA_close_U']

    # Sub-thought: Combining Indicators
    weights = {
        'MA_OC_diff': 0.2,
        'MA_range': 0.1,
        'MA_vol_price_ratio': 0.15,
        'volume_spike': 0.1,
        'MA_amt_price_ratio': 0.15,
        'Std_Amount': -0.1,
        'MA_ROC_close': 0.1,
        'MA_Std_Daily_Returns': -0.1,
        'EMA_SMA_diff': 0.1
    }
    
    df['Composite_Score'] = sum(weight * df[feature].fillna(0) for feature, weight in weights.items())

    return df['Composite_Score']
```
