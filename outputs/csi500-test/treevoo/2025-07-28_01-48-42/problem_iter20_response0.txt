```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the intraday range
    close_to_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume'])
    
    # Calculate the distance between the closing price and VWAP
    close_to_vwap = (df['close'] - vwap) / vwap
    
    # Combine the factors: close_to_range and close_to_vwap
    factor = 0.5 * close_to_range + 0.5 * close_to_vwap
    
    return factor
```
