```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_high_low_range = volume_adjusted_range + amount_adjusted_range
    
    # Integrate Momentum and Reversal
    momentum = high_low_range - high_low_range.shift(1)
    integrated_momentum_reversal = momentum * intraday_reversal
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    volume_amount_adjusted_price_change = price_change * (df['volume'] + np.sqrt(df['amount']))
    
    combined_signal = integrated_momentum_reversal + volume_amount_adjusted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    confirmed_signal = smoothed_combined_signal + close_to_open_diff * (df['volume'] + np.sqrt(df['amount']))
    
    # Calculate Smoothed Price Momentum
    daily_close_return = df['close'] / df['close'].shift(1) - 1
    smoothed_price_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_factor = (volume_change > 1.5 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Adjust Intraday Reversal by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Integrate Adjusted Intraday Reversal into Final Factor
    final_factor = confirmed_signal * smoothed_price_momentum * volume_spike_factor + close_to_open_diff + adjusted_intraday_reversal
    
    return final_factor
```
