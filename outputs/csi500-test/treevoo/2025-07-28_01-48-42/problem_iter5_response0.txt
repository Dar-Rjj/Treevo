```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_returns = (intraday_returns_high + intraday_returns_low) / 2
    intraday_return_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum_volume = momentum / volume_adjusted_range
    
    # Calculate High-Low Price Difference
    high_low_price_diff = df['high'] - df['low']
    
    # Calculate 10-Day High-Low Price Momentum
    high_low_momentum = high_low_price_diff.rolling(window=10).sum()
    
    # Calculate 10-Day Volume Moving Average
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    
    # Combine 10-Day High-Low Price Momentum and 10-Day Volume Moving Average
    combined_high_low_vol = high_low_momentum * volume_ma_10
    
    # Generate Factor Value
    factor_value = combined_high_low_vol / df['close']
    
    # Combine All Factors
    weighted_intraday_volatility = 0.3 * intraday_return_volatility
    weighted_combined_high_low_vol = 0.4 * combined_high_low_vol
    weighted_combined_momentum_volume = 0.2 * combined_momentum_volume
    weighted_factor_value = 0.1 * factor_value
    
    final_factor = (weighted_intraday_volatility + 
                    weighted_combined_high_low_vol + 
                    weighted_combined_momentum_volume + 
                    weighted_factor_value)
    
    return final_factor
```
