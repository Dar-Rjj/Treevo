```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / intraday_return

    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['amount'] * df['volume']

    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range.diff()
    combined_momentum_reversal = high_low_range_diff * intraday_reversal_score

    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    weighted_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_signal = combined_momentum_reversal + volume_adjusted_range * weighted_price_change

    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open']
    integrated_signal_with_c_o = integrated_signal + close_to_open_diff * df['volume']

    # Adjust Momentum by Open-Close Spread
    open_close_spread = df['close'] - df['open']
    final_factor = integrated_signal_with_c_o * open_close_spread.abs()

    # Apply a Smoothing Filter to the Combined Signal
    smoothed_factor = final_factor.ewm(span=5).mean()

    return smoothed_factor
```
