```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    total_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = five_day_momentum * open_close_spread.abs()
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum / total_adjusted_range
    
    # Enhance Combined Signal with Smoothing (EMA)
    enhanced_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = enhanced_combined_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (intraday_return + 1e-6)
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'].shift(1) - df['open']
    volume_adjusted_reversal = intraday_reversal * df['volume']
    
    # Integrate Volume-Adjusted Reversal with C-O Diff
    integrated_reversal = volume_adjusted_reversal * close_to_open_diff.abs()
    
    # Adjust Momentum by Open-Close Spread (Intraday Reversal)
    weighted_momentum = five_day_momentum * open_close_spread.abs()
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = combined_factor + weighted_momentum + integrated_reversal
    
    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_return.rolling(window=5).std()
    
    # Adjust Intraday Reversal Score by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Adjust Momentum by Intraday Volatility
    volatility_adjusted_momentum = five_day_momentum * intraday_volatility
    
    # Final Alpha Factor
    final_alpha_factor = (
        enhanced_combined_signal * smoothed_momentum * 
        volume_spike * long_term_momentum * intraday_volatility + 
        close_to_open_diff
    )
    
    return final_alpha_factor
```
