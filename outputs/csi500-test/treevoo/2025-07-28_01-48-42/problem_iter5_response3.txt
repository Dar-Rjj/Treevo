```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum (sum over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_high_low = high_low_range * df['volume']
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * abs(open_close_spread)
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_factor = adjusted_momentum / volume_adjusted_high_low
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_factor * smoothed_momentum * volume_spike
    
    return alpha_factor
```
