```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low for each day
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the weighted amount by volume
    weighted_amount = df['amount'] * df['volume']
    
    # Calculate the factor as the difference between the close price and the average of high and low
    # then divide it by the weighted amount
    factor = (df['close'] - avg_high_low) / weighted_amount
    
    return factor
```
