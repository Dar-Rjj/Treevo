```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Daily Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    df['21_day_momentum'] = df['log_return'].rolling(window=21).mean()
    df['60_day_momentum'] = df['log_return'].rolling(window=60).mean()
    df['momentum'] = df['21_day_momentum'] - df['60_day_momentum']
    
    # Volatility Component
    df['abs_log_return'] = df['log_return'].abs()
    df['21_day_volatility'] = df['abs_log_return'].rolling(window=21).sum()
    df['60_day_volatility'] = df['abs_log_return'].rolling(window=60).sum()
    df['volatility'] = df['21_day_volatility'] - df['60_day_volatility']
    
    # Volume Component
    df['log_volume_change'] = np.log(df['volume'] / df['volume'].shift(1))
    df['21_day_volume_momentum'] = df['log_volume_change'].rolling(window=21).mean()
    df['60_day_volume_momentum'] = df['log_volume_change'].rolling(window=60).mean()
    df['volume_momentum'] = df['21_day_volume_momentum'] - df['60_day_volume_momentum']
    
    # Final Factor: Momentum + Volatility + Volume Momentum
    df['alpha_factor'] = df['momentum'] + df['volatility'] + df['volume_momentum']
    
    return df['alpha_factor']
```
