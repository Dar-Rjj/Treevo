```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume information.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index being the date.
                       
    Returns:
    pd.Series: A Series with the computed alpha factor values, indexed by date.
    """
    # Calculate the relative difference between the close and open prices
    relative_price_change = (df['close'] - df['open']) / df['open']
    
    # Measure the liquidity or turnover by using the ratio of amount to volume
    liquidity = df['amount'] / df['volume']
    
    # Combine the two signals in a way that emphasizes stocks with positive price change
    # and high liquidity. This is a simple heuristic that could indicate momentum with
    # good liquidity, which might be a positive signal for future returns.
    alpha_factor = relative_price_change * liquidity
    
    return alpha_factor
```
