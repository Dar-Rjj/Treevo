```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low price
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the trading intensity as volume divided by price range
    trading_intensity = df['volume'] / price_range
    
    # Calculate the dollar value traded
    dollar_value_traded = df['amount'] * df['close']
    
    # Generate the alpha factor
    alpha_factor = (close_change * trading_intensity) + (dollar_value_traded / df['volume'])
    
    return alpha_factor
```
