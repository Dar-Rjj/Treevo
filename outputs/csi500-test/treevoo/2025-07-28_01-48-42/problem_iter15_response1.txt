import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    scaled_high_low_range = high_low_range * (avg_volume_5d / avg_amount_5d)
    
    # Integrate Momentum, Reversal, and Volume-Amount Adjustment
    high_low_range_diff = high_low_range.diff().fillna(0)
    momentum_5d = high_low_range_diff.rolling(window=5).sum()
    combined_momentum_reversal = momentum_5d * intraday_reversal_score
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1).fillna(0)
    
    # Compute Price Change Momentum
    price_momentum = df['close'] - df['open'].shift(1).fillna(0)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1).fillna(0)
    
    # Compute High-Low Range Adjusted by Volume
    high_low_range_vol_adj = high_low_range * df['volume']
    
    # Integrate Price Momentum and Volume Delta
    integrated_momentum_volume = price_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    volume_rank = df['volume'].rank(pct=True)
    amount_rank = df['amount'].rank(pct=True)
    close_open_diff = df['close'] - df['open']
    combined_signal = (combined_momentum_reversal * volume_rank * amount_rank) + (integrated_momentum_volume * close_open_diff)
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = combined_signal.ewm(span=5).mean()
    
    return smoothed_signal
