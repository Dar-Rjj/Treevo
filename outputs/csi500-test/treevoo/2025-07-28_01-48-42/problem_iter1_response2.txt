```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['high_low_momentum'] = df['high_low_range'].diff()
    df['high_low_momentum_5d'] = df['high_low_momentum'].rolling(window=5).sum()
    
    # Adjust for Volume
    df['volume_5d_avg'] = df['volume'].rolling(window=5).mean()
    df['adjusted_momentum'] = df['high_low_momentum_5d'] / df['volume_5d_avg']
    
    # Apply Moving Average Smoothing
    df['final_alpha_factor'] = df['adjusted_momentum'].rolling(window=10).mean()
    
    return df['final_alpha_factor']
```
