```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted = df['volume'] * high_low_range
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * np.abs(open_close_spread)
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_signal = adjusted_momentum / volume_adjusted
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum
    daily_return = df['close'].pct_change()
    smoothed_momentum = daily_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike_factor = (volume_change > 2 * std_volume).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_signal * smoothed_momentum * volume_spike_factor
    
    return final_alpha_factor
```
