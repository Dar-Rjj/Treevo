```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5, 10, and 20-day average daily return
    df['5d_avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['10d_avg_daily_return'] = df['daily_return'].rolling(window=10).mean()
    df['20d_avg_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['14d_RSI'] = 100 - (100 / (1 + rs))
    
    avg_gain_7 = gain.rolling(window=7).mean()
    avg_loss_7 = loss.rolling(window=7).mean()
    rs_7 = avg_gain_7 / avg_loss_7
    df['7d_RSI'] = 100 - (100 / (1 + rs_7))
    
    # Calculate True Range and ATR
    df['high_low'] = df['high'] - df['low']
    df['high_close'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low', 'high_close', 'low_close']].max(axis=1)
    
    df['10d_ATR'] = df['true_range'].rolling(window=10).mean()
    df['20d_ATR'] = df['true_range'].rolling(window=20).mean()
    
    # Calculate volume change and its moving sum
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['5d_vol_change_sum'] = df['volume_change'].rolling(window=5).sum()
    df['10d_vol_change_sum'] = df['volume_change'].rolling(window=10).sum()
    
    # Calculate VWAP
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate MFI
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_14 = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_14 = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_14 / negative_money_flow_14
    df['14d_MFI'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate high-low difference and its moving average
    df['high_low_diff'] = df['high'] - df['low']
    df['5d_high_low_diff_ma'] = df['high_low_diff'].rolling(window=5).mean()
    df['10d_high_low_diff_ma'] = df['high_low_diff'].rolling(window=10).mean()
    
    # Composite alpha factor
    df['composite_alpha'] = 0.5 * df['20d_avg_daily_return'] + 0.3 * df['10d_ATR'] + 0.2 * df['14d_RSI']
    
    return df['composite_alpha'].dropna()
```
