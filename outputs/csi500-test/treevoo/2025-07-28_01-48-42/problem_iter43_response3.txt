```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['low'] - df['open']
    intraday_volatility = (intraday_high_returns - intraday_low_returns).rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * df['amount'] ** 0.5
    adjusted_range = volume_adjusted + amount_adjusted
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust Momentum by Open-Close Spread
    momentum_adjusted = five_day_momentum * abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted / adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=5).mean()
    
    # Combine Adjusted Range with Weighted Price Change
    weighted_price_change = (enhanced_signal + price_change).ewm(span=5).mean()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'].shift(-1) - df['open']
    integrated_factor = intraday_reversal * close_to_open_diff * df['volume']
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Generate New Factor Value
    new_factor_value = (five_day_momentum / df['close']) + integrated_factor
    
    # Calculate Price-Volatility Ratio
    price_volatility_ratio = price_change / intraday_volatility
    
    # Calculate Long-Term Trend Strength
    long_term_momentum = df['close'].pct_change().rolling(window=20).mean()
    trend_strength = long_term_momentum.ewm(span=20).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (intraday_volatility * 0.3 + 
                          enhanced_signal * 0.2 + 
                          new_factor_value * 0.4 + 
                          price_volatility_ratio * 0.1)
    final_alpha_factor = final_alpha_factor * volume_spike * trend_strength
    
    return final_alpha_factor
```
