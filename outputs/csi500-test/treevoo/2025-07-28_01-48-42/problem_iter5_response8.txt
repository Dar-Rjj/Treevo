```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Combine High-Low Range and Intraday Reversal Score
    combined_score = high_low_range * intraday_reversal
    
    # Calculate Price Change
    price_change = df['close'].pct_change()
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Compute High-Low Range Adjusted by Volume
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Integrate Price Momentum and Volume Delta
    integrated_signal = price_change_momentum * volume_delta
    
    # Adjust by Volume and Amount Relative Strength
    daily_volume_rank = df['volume'].rolling(window=20).rank(pct=True)
    daily_amount_rank = df['amount'].rolling(window=20).rank(pct=True)
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open']
    
    # Aggregate for Interpretability
    aggregated_factor = (combined_score * close_to_open_diff * daily_volume_rank * daily_amount_rank * integrated_signal)
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_factor = aggregated_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
