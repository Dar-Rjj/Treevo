```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = (df['high'] - df['open']) / df['open']
    intraday_returns_low = (df['low'] - df['open']) / df['open']
    intraday_returns = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1)
    intraday_return_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum = momentum / volume_adjusted_range
    
    # Calculate Average Volume Spike
    volume_difference = df['volume'].diff()
    average_volume_spike = volume_difference.rolling(window=5).sum()
    
    # Calculate Exponential Moving Average of Price Change
    ema_price_change = price_change.ewm(span=14, adjust=False).mean()
    
    # Normalize Price Change by EMA
    normalized_price_change = price_change / ema_price_change
    
    # Combine All Factors
    factor = (0.3 * intraday_return_volatility + 
              0.2 * average_volume_spike + 
              0.1 * normalized_price_change + 
              0.4 * combined_momentum)
    
    return factor
```
