```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return

    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range

    # Combine Momentum and Volume-Amount-Adjusted Range
    price_change = df['close'].diff()
    five_day_momentum = price_change.rolling(window=5).sum()
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_momentum = five_day_momentum * open_close_spread
    combined_signal = adjusted_momentum / adjusted_range

    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()

    # Integrate Intraday Reversal into Final Factor
    close_to_open_diff = df['close'] - df['open'].shift(1)
    integrated_reversal = close_to_open_diff * intraday_reversal

    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2 / 4) + ((df['close'] - df['open'])**2 / 4))
    
    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Add a New Momentum Adjustment
    short_term_momentum = df['close'] - df['close'].shift(10)

    # Final Alpha Factor
    alpha_factor = (
        smoothed_signal * 
        integrated_reversal * 
        daily_volatility * 
        long_term_momentum * 
        short_term_momentum
    )

    return alpha_factor
```
