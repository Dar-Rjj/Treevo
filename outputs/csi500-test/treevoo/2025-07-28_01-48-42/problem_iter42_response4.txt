```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily and weekly price changes
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['weekly_price_change'] = df['close'] - df['close'].shift(7)
    
    # Determine trend direction using moving averages
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    df['200_day_MA'] = df['close'].rolling(window=200).mean()
    df['uptrend'] = (df['50_day_MA'] > df['200_day_MA']).astype(int)
    
    # Calculate daily and weekly volume changes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['weekly_volume_change'] = df['volume'] - df['volume'].shift(7)
    
    # Identify volume spikes
    df['avg_30_day_volume'] = df['volume'].rolling(window=30).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['avg_30_day_volume']).astype(int)
    
    # Create a price-volume momentum factor
    df['price_volume_momentum'] = (df['daily_price_change'] * df['daily_volume_change']).rolling(window=5).sum()
    
    # Develop a price-volume breakout signal
    price_threshold = 0.02 * df['close'].shift(1)
    volume_threshold = 1.5 * df['avg_30_day_volume']
    df['price_breakout'] = (df['close'] - df['close'].shift(1) > price_threshold).astype(int)
    df['volume_breakout'] = (df['volume'] > volume_threshold).astype(int)
    df['price_volume_breakout'] = (df['price_breakout'] & df['volume_breakout']).astype(int)
    
    # Calculate range-related factors
    df['daily_range'] = df['high'] - df['low']
    df['average_true_range'] = df['daily_range'].rolling(window=14).mean()
    
    # Analyze gaps
    df['up_gap'] = (df['open'] > df['close'].shift(1)).astype(int)
    df['down_gap'] = (df['open'] < df['close'].shift(1)).astype(int)
    
    # Analyze transaction amount changes
    df['daily_amount_change'] = df['amount'] - df['amount'].shift(1)
    df['weekly_amount_change'] = df['amount'] - df['amount'].shift(7)
    
    # Develop a transaction amount-based factor
    df['amount_volume_ratio'] = df['daily_amount_change'] / df['daily_volume_change']
    
    # Evaluate market sentiment
    df['close_open_ratio'] = df['close'] / df['open']
    df['high_low_ratio'] = df['high'] / df['low']
    df['sentiment_score'] = (df['close_open_ratio'] > 1).astype(int) + (df['high_low_ratio'] > 1).astype(int) - (df['close_open_ratio'] < 1).astype(int) - (df['high_low_ratio'] < 1).astype(int)
    
    # Integrate factors into a composite alpha
    df['composite_alpha'] = (
        0.3 * df['price_volume_momentum'] +
        0.2 * df['price_volume_breakout'] +
        0.1 * df['amount_volume_ratio'] +
        0.1 * df['sentiment_score'] +
        0.1 * df['uptrend'] +
        0.1 * df['up_gap'] -
        0.1 * df['down_gap']
    )
    
    return df['composite_alpha'].dropna()
```
