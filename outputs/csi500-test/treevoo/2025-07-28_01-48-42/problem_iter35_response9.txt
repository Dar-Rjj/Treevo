```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Adjust Momentum by Open-Close Spread
    momentum_adjusted = five_day_momentum * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted / adjusted_range
    
    # Enhance Combined Signal with Smoothing (EMA, span=5)
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_with_weighted_change = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'].shift(1) - df['open']
    
    # Integrate into Final Factor
    volume_adjusted_reversal = intraday_reversal * df['volume']
    integrated_reversal = volume_adjusted_reversal * close_open_diff
    
    # Adjust Momentum by Open-Close Spread (Intraday Reversal)
    weighted_momentum = five_day_momentum * np.abs(open_close_spread)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_combined_signal = combined_with_weighted_change + weighted_momentum + integrated_reversal
    
    # Calculate Daily Volatility
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2 / 4) + ((df['close'] - df['open']) ** 2 / 4))
    
    # Adjust Intraday Reversal Score by Daily Volatility
    volatility_adjusted_reversal = intraday_reversal * daily_volatility
    
    # Adjust Momentum by Daily Volatility
    volatility_adjusted_momentum = five_day_momentum * daily_volatility
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = (
        final_combined_signal * 
        smoothed_momentum * 
        volume_spike * 
        long_term_momentum * 
        daily_volatility + 
        close_open_diff
    )
    
    return alpha_factor
```
