```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_returns = (intraday_returns_high + intraday_returns_low) / 2
    intraday_volatility = intraday_returns.rolling(window=10).std()
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    volume_adjusted_range = high_low_range / (avg_volume_5d * avg_amount_5d)
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_signal = momentum / volume_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Generate Factor Value
    factor_value = enhanced_signal / df['close']
    
    # Compute Intraday Reversal Score
    intraday_return = df['high'] - df['low']
    reversal_score = 1 / (1 + intraday_return)
    
    # Integrate Momentum, Reversal, and Volume-Amount Adjustment
    range_momentum = (high_low_range - high_low_range.shift(1)).rolling(window=5).sum()
    combined_momentum_reversal = (range_momentum * reversal_score).ewm(span=5, adjust=False).mean()
    
    # Calculate Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Compute High-Low Range Adjusted by Volume
    hl_range_volume_adjusted = high_low_range * df['volume']
    
    # Integrate Price Momentum and Volume Delta
    integrated_momentum = price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    volume_rank = df['volume'].rank(axis=1, pct=True).rolling(window=20).mean()
    amount_rank = df['amount'].rank(axis=1, pct=True).rolling(window=20).mean()
    close_to_open_diff = df['close'] - df['open']
    confirmed_signal = combined_momentum_reversal * (volume_rank * amount_rank) * close_to_open_diff
    
    # Combine All Factors
    combined_factor = (0.3 * intraday_volatility) + (0.7 * factor_value) + confirmed_signal
    
    # Apply a Smoothing Filter to the Combined Signal
    final_signal = combined_factor.ewm(span=5, adjust=False).mean()
    
    return final_signal
```
