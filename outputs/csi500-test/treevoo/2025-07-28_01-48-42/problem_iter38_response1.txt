```python
import pandas as pd

def heuristics_v2(df):
    lookback_period = 10
    long_lookback_period = 20
    
    # Calculate Daily Open-Close Price Spread
    daily_spread = df['open'] - df['close']
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum / daily_spread.abs()
    
    # Weight Adjusted Momentum by Volume
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    
    # Sum the weighted adjusted momentum over the lookback period
    sum_weighted_momentum = weighted_adjusted_momentum.rolling(window=lookback_period).sum()
    
    # Total volume over the lookback period
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Final Alpha Factor Component 1
    alpha_factor_1 = sum_weighted_momentum / total_volume
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    true_range = df[['high' - 'close'].shift(1), ('low' - 'close').shift(1), high_low_diff]].max(axis=1)
    atr = true_range.rolling(window=lookback_period).mean()
    
    # Weight Intraday Volatility by Volume
    weighted_intraday_volatility = atr * df['volume']
    
    # Sum the weighted intraday volatility over the lookback period
    sum_weighted_intraday_volatility = weighted_intraday_volatility.rolling(window=lookback_period).sum()
    
    # Final Alpha Factor Component 2
    alpha_factor_2 = sum_weighted_intraday_volatility / total_volume
    
    # Calculate Price Movement Consistency
    moving_avg_close = df['close'].rolling(window=long_lookback_period).mean()
    daily_returns = df['close'].pct_change()
    std_daily_returns = daily_returns.rolling(window=long_lookback_period).std()
    
    # Final Alpha Factor Component 3
    alpha_factor_3 = std_daily_returns / moving_avg_close
    
    # Combine Final Alpha Factors
    final_alpha_factor = alpha_factor_1 + alpha_factor_2 + alpha_factor_3
    
    return final_alpha_factor
```
