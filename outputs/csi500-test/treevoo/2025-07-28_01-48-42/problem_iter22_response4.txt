```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.exp(-intraday_return))
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * np.sqrt(df['amount'])
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'] - df['close'].shift(1)
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    integrated_signal = combined_momentum_reversal + volume_amount_adjusted_range + weighted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = np.abs(df['open'] - df['close'])
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = integrated_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    integrated_with_c_o_diff = smoothed_combined_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_price_return = (df['close'] / df['close'].shift(1)) - 1
    smoothed_price_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    std_volume_20_days = df['volume'].rolling(window=20).std()
    volume_spike_factor = (volume_change > 2 * std_volume_20_days).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = integrated_with_c_o_diff * smoothed_price_momentum * volume_spike_factor
    
    return final_alpha_factor
```
