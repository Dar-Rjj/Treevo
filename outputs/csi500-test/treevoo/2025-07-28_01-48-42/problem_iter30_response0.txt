```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Adjust Momentum by Open-Close Spread
    momentum_adjusted_by_spread = five_day_momentum * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted_by_spread / adjusted_range
    
    # Enhance Combined Signal with Smoothing (EMA)
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['open'].shift(-1) - df['close']
    intraday_reversal_adjusted = intraday_reversal_score * close_to_open_diff
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = combined_factor + intraday_reversal_adjusted
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = price_change.ewm(span=20, adjust=False).mean()
    trend_strength = long_term_momentum.ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = ((df['high'] - df['low']) / df['low']).rolling(window=5).std()
    
    # Adjust Intraday Reversal Score by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal_score * intraday_volatility
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = five_day_momentum * intraday_volatility
    
    # Final Alpha Factor
    final_factor = (
        enhanced_signal * smoothed_momentum * 
        volume_spike * trend_strength * 
        intraday_volatility + close_to_open_diff
    )
    
    return final_factor
```
