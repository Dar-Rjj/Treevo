```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Invert Intraday Return for Reversal
    intraday_reversal = 1 / intraday_return

    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    combined_adjusted = volume_adjusted + amount_adjusted

    # Integrate Momentum and Reversal
    momentum_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal

    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)

    # Multiply Price Change by Volume and Square Root of Amount
    price_change_adjusted = price_change * df['volume'] * np.sqrt(df['amount'])

    # Add to Combined Momentum and Intraday Reversal
    combined_signal = momentum_reversal + price_change_adjusted + combined_adjusted

    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()

    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_momentum = five_day_momentum * open_close_spread

    # Enhance Combined Signal with Smoothing
    smoothed_signal = combined_signal.ewm(span=5, adjust=False).mean()

    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    integrated_signal = smoothed_signal + (combined_adjusted * close_to_open_diff)

    # Calculate Smoothed Price Momentum
    daily_return = df['close'] / df['close'].shift(1)
    smoothed_momentum = daily_return.ewm(span=12, adjust=False).mean()

    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 1.8 * df['volume'].rolling(window=20).std()).astype(int)

    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)

    # Adjust Intraday Reversal by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility

    # Integrate Adjusted Intraday Reversal into Final Factor
    final_factor = integrated_signal + (adjusted_intraday_reversal * combined_adjusted)

    # Generate New Factor Value
    new_factor = five_day_momentum / df['close'] + adjusted_intraday_reversal

    # Integrate Price-Volatility Ratio
    price_volatility_ratio = (df['close'] - df['close'].shift(1)) / intraday_volatility

    # Adjust by Volume Spike Indicator
    adjusted_new_factor = new_factor * volume_spike

    # Incorporate Long-Term Trend Strength
    long_term_momentum = df['close'] / df['close'].shift(25)
    long_term_trend_strength = long_term_momentum.ewm(span=25, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = (
        0.35 * intraday_volatility +
        0.25 * smoothed_signal +
        0.4 * new_factor +
        0.1 * price_volatility_ratio
    ) * long_term_trend_strength

    return final_alpha_factor
```
