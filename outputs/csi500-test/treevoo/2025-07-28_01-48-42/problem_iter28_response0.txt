```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute 15-day Momentum
    momentum_15d = price_change.rolling(window=15).sum()
    
    # Adjust for Volatility
    daily_volatility = (df['high'] - df['low']) / df['open']
    adjusted_momentum_15d = momentum_15d / daily_volatility
    
    # Momentum Component
    returns = price_change / df['close'].shift(1)
    ema_28_returns = returns.ewm(span=28, adjust=False).mean()
    ema_70_returns = returns.ewm(span=70, adjust=False).mean()
    momentum_component = ema_28_returns - ema_70_returns
    
    # Volatility Component
    abs_returns = np.abs(returns)
    ema_28_abs_returns = abs_returns.ewm(span=28, adjust=False).mean()
    ema_70_abs_returns = abs_returns.ewm(span=70, adjust=False).mean()
    volatility_component = ema_28_abs_returns - ema_70_abs_returns
    
    # Volume Component
    log_volume_changes = np.log(df['volume']) - np.log(df['volume'].shift(1))
    ema_28_volume_changes = log_volume_changes.ewm(span=28, adjust=False).mean()
    ema_70_volume_changes = log_volume_changes.ewm(span=70, adjust=False).mean()
    volume_momentum = ema_28_volume_changes - ema_70_volume_changes
    
    # Open-Close Spread
    open_close_diff = df['close'] - df['open']
    open_close_spread = open_close_diff.ewm(span=28, adjust=False).mean()
    
    # Combine Components
    combined_alpha_factor = momentum_component + volatility_component + volume_momentum + open_close_spread
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum_15d + combined_alpha_factor
    
    return final_alpha_factor
```
