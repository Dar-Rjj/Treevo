```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum over the last 10 days
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Adjusted High-Low Range and Momentum
    combined_momentum = momentum / (volume_adjusted_high_low_range + 1e-6)  # Add small constant to avoid division by zero
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + weighted_price_change
    
    return alpha_factor
```
