```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Returns
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_volatility = (intraday_returns_high.rolling(window=10).std() + 
                           intraday_returns_low.rolling(window=10).std()) / 2
    
    # Calculate Average Volume Spike
    volume_diff = df['volume'].diff()
    avg_volume_spike = volume_diff.rolling(window=5).sum()

    # Calculate Close Price Momentum
    price_change = df['close'].pct_change().fillna(0)
    ema_price_change = price_change.ewm(span=14, adjust=False).mean()
    normalized_price_change = price_change / ema_price_change

    # Combine Factors
    factor = (0.5 * intraday_volatility + 
              0.3 * avg_volume_spike + 
              0.2 * normalized_price_change)
    
    return factor
```
