```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from open to close
    df['Intraday_Movement'] = df['close'] - df['open']
    
    # Calculate the range (High - Low) as a measure of daily volatility
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to price change (Volume to Price Ratio)
    df['Volume_to_Price_Ratio'] = df['volume'] / df['Intraday_Movement'].abs()
    
    # Calculate the ratio of amount to price (Amount to Price Ratio)
    df['Amount_to_Price_Ratio'] = df['amount'] / df['close']
    
    # Calculate the percentage change in closing price over a fixed period (e.g., 5 days)
    df['Momentum_5D'] = df['close'].pct_change(periods=5)
    
    # Calculate the moving average of the closing price over a short period (e.g., 10 days)
    df['Short_MA'] = df['close'].rolling(window=10).mean()
    
    # Calculate the moving average of the closing price over a long period (e.g., 50 days)
    df['Long_MA'] = df['close'].rolling(window=50).mean()
    
    # Determine the difference between the short-term and long-term moving averages (Trend Factor)
    df['Trend_Factor'] = df['Short_MA'] - df['Long_MA']
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        df['Intraday_Movement'] +
        df['Daily_Range'] +
        df['Volume_to_Price_Ratio'] +
        df['Amount_to_Price_Ratio'] +
        df['Momentum_5D'] +
        df['Trend_Factor']
    )
    
    return alpha_factor
```
