```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Open-Close Spread
    open_close_spread = df['open'] - df['close']

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Daily Open-Close Price Spread
    daily_open_close_spread = df['close'] - df['open']

    # Calculate 5-Day Price Change
    df['5_day_price_change'] = df['close'].diff().rolling(window=5).sum()

    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    adjusted_high_low_range = volume_adjusted + amount_adjusted

    # Adjust 5-Day Price Change by Absolute Open-Close Spread
    absolute_daily_spread = np.abs(daily_open_close_spread)
    adjusted_5_day_price_change = df['5_day_price_change'] / absolute_daily_spread

    # Weight Adjusted 5-Day Price Change by Volume
    weighted_adjusted_5_day_price_change = adjusted_5_day_price_change * df['volume']
    total_volume = df['volume'].rolling(window=5).sum()
    final_alpha_factor_1 = weighted_adjusted_5_day_price_change.rolling(window=5).sum() / total_volume

    # Calculate Intraday Volatility
    true_range = df[['high', 'low', 'close']].shift(1).fillna(0).apply(lambda x: max(x) - min(x), axis=1)
    atr = true_range.rolling(window=20).mean()
    intraday_volatility = (df['high'] - df['low']) / atr

    # Weight Intraday Volatility by Volume
    weighted_intraday_volatility = intraday_volatility * df['volume']
    final_alpha_factor_2 = weighted_intraday_volatility.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate Price Movement Consistency
    close_prices = df['close']
    moving_average = close_prices.rolling(window=20).mean()
    daily_returns = close_prices.pct_change()
    std_dev = daily_returns.rolling(window=20).std()
    final_alpha_factor_3 = std_dev / moving_average

    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return

    # Consider Close-to-Open Price Change
    close_to_open_diff = df['open'].shift(-1) - df['close']
    volume_adjusted_reversal = intraday_reversal * df['volume'] * close_to_open_diff

    # Adjust Intraday Reversal Score by Daily Volatility
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2 / 4) + ((df['close'] - df['open']) ** 2 / 4))
    adjusted_intraday_reversal = intraday_reversal * daily_volatility

    # Adjust 5-Day Momentum by Daily Volatility
    adjusted_5_day_momentum = df['5_day_price_change'] * daily_volatility

    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()

    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)

    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().rolling(window=20).mean()
    trend_strength = long_term_momentum.ewm(span=20, adjust=False).mean()

    # Combine Final Alpha Factors
    enhanced_combined_signal = final_alpha_factor_1 + final_alpha_factor_2 + final_alpha_factor_3
    combined_signal = (enhanced_combined_signal * smoothed_momentum * volume_spike * trend_strength * daily_volatility) + close_to_open_diff

    return combined_signal
```
