```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (price_change * df['volume']).rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = weighted_momentum * open_close_spread.abs()
    
    # Volume-Adjust High-Low Range
    volume_adjusted_high_low_range = df['volume'] * high_low_range
    
    # Combine Volume-Adjusted Components
    factor_values = adjusted_momentum / volume_adjusted_high_low_range

    return factor_values
```
