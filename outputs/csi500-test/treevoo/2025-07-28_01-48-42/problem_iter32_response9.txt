```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous close to the current open
    df['overnight_move'] = df['open'] - df['close'].shift(1)
    
    # Calculate the intraday move from the open to the close
    df['intraday_move'] = df['close'] - df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the ratio of overnight move to the daily range
    df['overnight_to_range_ratio'] = df['overnight_move'] / df['daily_range']
    
    # Calculate the ratio of intraday move to the daily range
    df['intraday_to_range_ratio'] = df['intraday_move'] / df['daily_range']
    
    # Combine the two ratios to create a new factor
    df['combined_factor'] = df['overnight_to_range_ratio'] + df['intraday_to_range_ratio']
    
    # Return the combined factor as a Series
    return df['combined_factor']
```
