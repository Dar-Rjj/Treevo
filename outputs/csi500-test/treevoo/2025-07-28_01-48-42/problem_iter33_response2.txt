```python
def heuristics_v2(df):
    # Compute Open-Close Spread
    spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * spread.abs()
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Incorporate Price Trend Sensitivity
    high_low_range = df['high'] - df['low']
    trend_direction = (df['close'] > df['open']).astype(int)
    price_trend_sensitivity = (1 + (trend_direction * 2 - 1) * (high_low_range / df['open']))
    
    # Modify Momentum based on Trend
    final_factor = volume_weighted_momentum * price_trend_sensitivity
    
    return final_factor
```
