```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the ratio of high to low
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the weighted close: (open + 2 * close + high + low) / 5
    weighted_close = (df['open'] + 2 * df['close'] + df['high'] + df['low']) / 5
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the factor as a combination of the above metrics
    factor = (pct_change_close * high_low_ratio + weighted_close) / avg_volume_5d
    
    return factor
```
