```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_return_high = df['high'] - df['open']
    intraday_return_low = df['low'] - df['open']
    intraday_returns = pd.concat([intraday_return_high, intraday_return_low], axis=1).std(axis=1)
    intraday_return_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_momentum_volume = momentum / volume_adjusted_range
    
    # Calculate High-Low Price Difference
    high_low_price_diff = df['high'] - df['low']
    
    # Calculate 10-Day High-Low Price Momentum
    high_low_price_momentum = high_low_price_diff.rolling(window=10).sum()
    
    # Calculate 10-Day Volume Moving Average
    volume_moving_average = df['volume'].rolling(window=10).mean()
    
    # Combine 10-Day High-Low Price Momentum and 10-Day Volume Moving Average
    combined_high_low_momentum_volume = high_low_price_momentum * volume_moving_average
    
    # Generate Factor Value
    factor_value = combined_high_low_momentum_volume / df['close']
    
    # Combine All Factors
    weighted_intraday_return_volatility = intraday_return_volatility * 0.3
    weighted_combined_high_low_momentum_volume = combined_high_low_momentum_volume * 0.4
    weighted_combined_momentum_volume = combined_momentum_volume * 0.2
    weighted_factor_value = factor_value * 0.1
    
    # Sum of Weighted Factors
    alpha_factor = (
        weighted_intraday_return_volatility +
        weighted_combined_high_low_momentum_volume +
        weighted_combined_momentum_volume +
        weighted_factor_value
    )
    
    return alpha_factor
```
