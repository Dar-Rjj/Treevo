```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum (Sum over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Combine Momentum and Volume-Adjusted Range
    combined_signal = momentum / volume_adjusted_range
    
    # Calculate Volume Delta
    volume_delta = df['volume'].diff()
    
    # Enhance Combined Signal with Volume Delta
    enhanced_signal = combined_signal * volume_delta
    
    # Apply a Smoothing Filter (Exponential Moving Average with span 5)
    smoothed_signal = enhanced_signal.ewm(span=5, adjust=False).mean()
    
    return smoothed_signal
```
