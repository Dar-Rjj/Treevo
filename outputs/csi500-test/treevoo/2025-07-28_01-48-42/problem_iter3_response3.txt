```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate High-Low Price Momentum (sum over the last 10 days)
    high_low_momentum = high_low_diff.rolling(window=10).sum()
    
    # Calculate Volume Trend (10-day Moving Average)
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Calculate Amount Trend (10-day Moving Average)
    amount_trend = df['amount'].rolling(window=10).mean()
    
    # Combine High-Low Price Momentum, Volume, and Amount Trends
    combined_trends = high_low_momentum * volume_trend * amount_trend
    
    # Calculate Factor Value (divide by Close price)
    factor_value = combined_trends / df['close']
    
    return factor_value
```
