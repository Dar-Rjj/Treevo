import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + intraday_return)
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range - high_low_range.shift(1)
    integrated_mr = high_low_range * intraday_reversal_score
    
    # Combine High-Low Range and Intraday Reversal Score
    combined_signal = high_low_range * intraday_reversal_score
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    integrated_pv = price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_open_diff = df['close'] - df['open']
    final_factor = (integrated_mr + integrated_pv) * close_open_diff
    
    # Apply a Smoothing Filter to the Combined Signal
    final_factor_smoothed = final_factor.ewm(span=5, min_periods=1).mean()
    
    return final_factor_smoothed
