```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff(1)
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Calculate Volume Delta
    volume_delta = df['volume'].diff(1)
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * volume_delta
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volatility
    volatility_adjusted_momentum = (momentum + volume_adjusted_range) / daily_volatility
    
    # Combine Momentum, Volume-Adjusted Range, and Volatility Adjustment
    combined_factor = volatility_adjusted_momentum.rolling(window=20).sum() / volume_adjusted_range
    
    return combined_factor
```
