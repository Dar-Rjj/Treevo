```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['low'] - df['open']
    intraday_volatility = (intraday_high_returns - intraday_low_returns).rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = five_day_momentum * abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum / combined_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=5).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    final_combined_signal = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['open'].shift(-1) - df['close']
    
    # Integrate into Final Factor
    volume_adjusted_reversal = intraday_reversal * df['volume']
    integrated_factor = volume_adjusted_reversal * close_to_open_diff
    
    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].diff()
    volume_spike_indicator = (day_to_day_volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Generate New Factor Value
    new_factor_value = five_day_momentum / df['close']
    new_factor_value_integrated = new_factor_value + integrated_factor
    
    # Final Alpha Factor
    price_volatility_ratio = price_change / intraday_volatility
    long_term_momentum = df['close'].pct_change().ewm(span=20).mean()
    
    final_alpha_factor = (0.3 * intraday_volatility + 
                          0.2 * enhanced_signal + 
                          0.4 * new_factor_value_integrated + 
                          0.1 * price_volatility_ratio)
    
    final_alpha_factor *= volume_spike_indicator
    final_alpha_factor *= long_term_momentum
    
    return final_alpha_factor
```
