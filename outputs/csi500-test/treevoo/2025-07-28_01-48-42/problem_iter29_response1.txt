```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Calculate Momentum (Lookback Period: 10 days)
    momentum = df['close'] - df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum / spread.abs()
    
    # Weight Adjusted Momentum by Volume
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Calculate daily return and log return
    daily_return = (df['close'] - df['open']) / df['open']
    log_return = np.log(df['close'] / df['open'])
    
    # Introduce a concept of price momentum
    moving_avg_5d = daily_return.rolling(window=5).mean()
    price_momentum = daily_return - moving_avg_5d
    
    # Explore the volatility of the stock
    volatility = daily_return.rolling(window=10).std()
    
    # Evaluate the relationship between the closing price and the volume
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = (df['volume'] / avg_volume_10d) * daily_return
    
    # Incorporate the high-low spread into the factor creation
    percent_range = (df['high'] - df['low']) / df['close']
    high_low_factor = percent_range * daily_return
    
    # Investigate the effect of large trades on stock movement
    volume_90th_percentile = df['volume'].rolling(window=30).quantile(0.9)
    large_trade_flag = (df['volume'] > volume_90th_percentile).astype(int)
    
    # Analyze the pattern of the opening and closing gaps
    prev_close = df['close'].shift(1)
    open_gap = df['open'] - prev_close
    gap_factor = open_gap * df['return'].shift(-1)  # Assuming next day's return is available in 'return' column, or compute it if not
    
    # Combine all the factors to form the final alpha factor
    alpha_factor = (
        volume_weighted_momentum +
        price_momentum +
        (volatility * 0.5) +  # Adjust the weight of volatility
        (volume_ratio * 0.3) +  # Adjust the weight of volume ratio
        (high_low_factor * 0.2) +  # Adjust the weight of high-low factor
        (large_trade_flag * 0.1) +  # Adjust the weight of large trade flag
        gap_factor
    )
    
    return alpha_factor
```
