```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    momentum_5_day = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    adjusted_range = high_low_range * (df['volume'] * df['amount'] ** 0.5)
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * df['amount'] ** 0.5
    
    # Combine Momentum and Adjusted Range
    combined_momentum = momentum_5_day / adjusted_range
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + weighted_price_change
    
    return alpha_factor
```
