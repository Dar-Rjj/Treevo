```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_range = volume_adjusted_range + amount_adjusted_range
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    price_change = df['close'].diff()
    five_day_momentum = price_change.rolling(window=5).sum()
    open_close_spread = df['open'] - df['close']
    adjusted_five_day_momentum = five_day_momentum * np.abs(open_close_spread)
    momentum_ratio = adjusted_five_day_momentum / combined_range
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = momentum_ratio.ewm(span=5, adjust=False).mean()
    
    # Integrate Intraday Reversal into Final Factor
    close_to_open_diff = df['close'] - df['open'].shift(1)
    intraday_reversal_factor = intraday_reversal * close_to_open_diff
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2 / 4) + ((df['close'] - df['open'])**2 / 4))
    volatility_adjusted_signal = smoothed_signal * daily_volatility
    
    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)
    trend_confirmed_signal = volatility_adjusted_signal * long_term_momentum
    
    # Incorporate Price-Based Momentum Indicators
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    cumulative_return = daily_returns.rolling(window=5).sum()
    price_momentum_adjusted_signal = trend_confirmed_signal * cumulative_return
    
    # Incorporate Volume-Based Factors
    volume_change = df['volume'].diff()
    cumulative_volume = df['volume'].rolling(window=5).sum()
    volume_adjusted_signal = price_momentum_adjusted_signal * cumulative_volume
    
    # Final Alpha Factor
    final_alpha_factor = (
        smoothed_signal * 
        intraday_reversal_factor * 
        daily_volatility * 
        long_term_momentum * 
        cumulative_return * 
        cumulative_volume + 
        close_to_open_diff
    )
    
    return final_alpha_factor
```
