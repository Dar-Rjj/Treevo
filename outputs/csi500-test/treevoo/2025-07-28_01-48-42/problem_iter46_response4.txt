```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)

    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    combined_adjustment = volume_adjusted + amount_adjusted

    # Combine Momentum and Volume-Amount-Adjusted Range
    price_change = df['close'].diff()
    five_day_momentum = price_change.rolling(window=5).sum()
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_momentum = five_day_momentum * open_close_spread
    momentum_adjusted_range = adjusted_momentum / combined_adjustment

    # Enhance Combined Signal with Smoothing
    smoothed_signal = momentum_adjusted_range.ewm(span=5, adjust=False).mean()

    # Integrate Intraday Reversal into Final Factor
    close_to_open_diff = df['close'] - df['open']
    integrated_reversal = intraday_reversal * close_to_open_diff

    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted_factor = smoothed_signal * daily_volatility

    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)
    final_factor = volatility_adjusted_factor * long_term_momentum * integrated_reversal

    return final_factor
```
