```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_momentum_and_adjusted_range = momentum / volume_amount_adjusted_range
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_and_adjusted_range + (volume_amount_adjusted_range * weighted_price_change)
    
    return final_alpha_factor
```
