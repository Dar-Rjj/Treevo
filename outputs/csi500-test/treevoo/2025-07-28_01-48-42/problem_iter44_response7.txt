```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (2 * df['close'] - df['low'] - df['high']) / daily_range
    
    # Calculate the money flow
    money_flow = buying_pressure * df['volume'] * df['close']
    
    # Calculate the heuristic factor, which is the 10-day moving average of the money flow
    heuristic_factor = money_flow.rolling(window=10).mean()
    
    return heuristic_factor
```
