```python
def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Determine Lookback Period (e.g., 10 days)
    lookback_period = 10
    
    # Compute Momentum
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    abs_spread = spread.abs()
    adjusted_momentum = momentum / abs_spread
    
    # Weight Adjusted Momentum by Volume
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    
    # Sum the weighted adjusted momentum over the lookback period
    sum_weighted_adjusted_momentum = weighted_adjusted_momentum.rolling(window=lookback_period).sum()
    
    # Total volume over the lookback period
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Final Alpha Factor
    alpha_factor = sum_weighted_adjusted_momentum / total_volume
    
    return alpha_factor
```
