```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, close) and (high, low)
    avg_open_close = (df['open'] + df['close']) / 2
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the two averages
    diff_avg = avg_open_close - avg_high_low
    
    # Calculate the weighted sum of the difference and volume
    weighted_sum = diff_avg * df['volume']
    
    # Calculate the ratio of the weighted sum to the amount
    factor_values = weighted_sum / df['amount']
    
    return factor_values
```
