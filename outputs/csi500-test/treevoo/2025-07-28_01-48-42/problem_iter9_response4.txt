```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    df['range_momentum'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['sum_range_momentum_5d'] = df['range_momentum'].rolling(window=5).sum()
    
    # Adjust for Volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['adjusted_momentum'] = df['sum_range_momentum_5d'] * (df['volume'] / df['avg_volume_5d'])
    
    # Apply Exponential Moving Average Smoothing
    df['smoothed_adjusted_momentum'] = df['adjusted_momentum'].ewm(span=10, adjust=False).mean()
    
    # Calculate Logarithmic Daily Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    df['ema_returns_21d'] = df['log_returns'].ewm(span=21, adjust=False).mean()
    df['ema_returns_60d'] = df['log_returns'].ewm(span=60, adjust=False).mean()
    df['momentum'] = df['ema_returns_21d'] - df['ema_returns_60d']
    
    # Volatility Component
    df['abs_log_returns'] = df['log_returns'].abs()
    df['ema_abs_returns_21d'] = df['abs_log_returns'].ewm(span=21, adjust=False).mean()
    df['ema_abs_returns_60d'] = df['abs_log_returns'].ewm(span=60, adjust=False).mean()
    df['volatility'] = df['ema_abs_returns_21d'] - df['ema_abs_returns_60d']
    
    # Volume Component
    df['log_volume_changes'] = np.log(df['volume'] / df['volume'].shift(1))
    df['ema_volume_changes_21d'] = df['log_volume_changes'].ewm(span=21, adjust=False).mean()
    df['ema_volume_changes_60d'] = df['log_volume_changes'].ewm(span=60, adjust=False).mean()
    df['volume_momentum'] = df['ema_volume_changes_21d'] - df['ema_volume_changes_60d']
    
    # Combine Components
    df['final_momentum'] = df['momentum']
    df['final_volatility'] = df['volatility']
    df['final_volume_momentum'] = df['volume_momentum']
    df['final_alpha_factor_2'] = df['final_momentum'] + df['final_volatility'] + df['final_volume_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_adjusted_momentum'] + df['final_alpha_factor_2']
    
    return df['final_alpha_factor']
```
