```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Calculate Momentum
    lookback_period = 10
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum with Absolute Daily Open-Close Spread
    abs_spread = spread.abs()
    adjusted_momentum = momentum / abs_spread
    
    # Volume-Weight the Adjusted Momentum
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    aggregated_weighted_adjusted_momentum = weighted_adjusted_momentum.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Determine Final Alpha Factor
    alpha_factor = aggregated_weighted_adjusted_momentum / total_volume
    
    return alpha_factor
```
