```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute 5-Day Momentum
    momentum_5d = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    combined_adjustment = volume_adjusted + amount_adjusted
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum_5d * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum / combined_adjustment
    
    # Enhance Combined Signal with Smoothing (EMA)
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_weighted_signal = enhanced_signal + weighted_price_change
    
    # Compute Intraday Return
    intraday_return_high = df['high'] - df['open']
    intraday_return_low = df['low'] - df['open']
    intraday_return = (intraday_return_high - intraday_return_low) / df['open']
    
    # Compute Intraday Reversal Score
    intraday_reversal_score = -intraday_return
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range.diff()
    integrated_signal = high_low_range_diff * intraday_reversal_score
    
    # Generate Factor Value
    factor_value = (df['high'] - df['low']).rolling(window=10).mean() * df['volume'].rolling(window=10).mean() / df['close']
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_factor = factor_value * daily_volatility
    
    # Combine All Factors
    combined_factors = (intraday_return * 0.3 + 
                        factor_value * 0.4 + 
                        enhanced_signal * 0.2 + 
                        volatility_factor * 0.1)
    
    # Final Alpha Factor
    smoothened_momentum = momentum_5d.ewm(span=20, adjust=False).mean()
    final_alpha = combined_factors * smoothened_momentum
    
    # Volume Spike Factor
    volume_spike = (df['volume'].diff() > 0).astype(int)
    final_alpha *= volume_spike
    
    # Trend Strength
    long_term_momentum = momentum_5d.ewm(span=20, adjust=False).mean()
    final_alpha *= long_term_momentum
    
    # Multiply by Intraday Volatility
    final_alpha *= daily_volatility
    
    # Add to Close-to-Open Price Change
    final_alpha += (df['close'] - df['open'].shift(1))
    
    # Apply Volume Weighting
    final_alpha *= df['volume']
    
    return final_alpha
```
