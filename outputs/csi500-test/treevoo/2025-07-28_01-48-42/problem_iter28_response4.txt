```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / intraday_return  # Invert for reversal
    
    # Adjust for Volume and Amount
    avg_5d_volume = df['volume'].rolling(window=5).mean()
    avg_5d_amount = df['amount'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range * (avg_5d_volume * avg_5d_amount)
    
    # Integrate Momentum and Reversal
    prev_high_low_range = high_low_range.shift(1)
    momentum_and_reversal = (high_low_range - prev_high_low_range) * intraday_reversal_score
    
    # Add to Integrated Momentum and Intraday Reversal
    integrated_signal = momentum_and_reversal + adjusted_high_low_range
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Price Change Momentum
    close_open_momentum = df['close'] - df['open'].shift(1)
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    volume_rank = df['volume'].rank(pct=True, method='dense')
    amount_rank = df['amount'].rank(pct=True, method='dense')
    close_to_open_diff = df['close'] - df['open']
    
    # Integrate into Final Factor
    final_factor = (volume_rank * amount_rank * close_to_open_diff) * integrated_signal
    
    # Apply a Smoothing Filter (EMA)
    ema_span = 5
    smoothed_factor = final_factor.ewm(span=ema_span, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted_factor = smoothed_factor * daily_volatility
    
    # Apply a Second Smoothing Layer (SMA)
    sma_window = 10
    final_smoothed_factor = volatility_adjusted_factor.rolling(window=sma_window).mean()
    
    return final_smoothed_factor
```
