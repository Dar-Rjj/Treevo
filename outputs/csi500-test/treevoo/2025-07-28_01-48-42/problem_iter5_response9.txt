```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Daily Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    df['momentum_21'] = df['log_returns'].rolling(window=21).mean()
    df['momentum_60'] = df['log_returns'].rolling(window=60).mean()
    df['momentum'] = df['momentum_21'] - df['momentum_60']
    
    # Volatility Component
    df['abs_log_returns'] = np.abs(df['log_returns'])
    df['volatility_21'] = df['abs_log_returns'].rolling(window=21).sum()
    df['volatility_60'] = df['abs_log_returns'].rolling(window=60).sum()
    df['volatility'] = df['volatility_21'] - df['volatility_60']
    
    # Volume Component
    df['log_volume_changes'] = np.log(df['volume'] / df['volume'].shift(1))
    df['volume_momentum_21'] = df['log_volume_changes'].rolling(window=21).mean()
    df['volume_momentum_60'] = df['log_volume_changes'].rolling(window=60).mean()
    df['volume_momentum'] = df['volume_momentum_21'] - df['volume_momentum_60']
    
    # Intraday Return Volatility
    df['intraday_high'] = df['high'] - df['open']
    df['intraday_low'] = df['low'] - df['open']
    df['intraday_volatility'] = (df['intraday_high'] + df['intraday_low']).rolling(window=10).std()
    
    # Average Volume Spike
    df['volume_spike'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_spike'] = df['volume_spike'].rolling(window=5).sum()
    
    # Close Price Momentum
    df['price_change_20'] = df['close'] - df['close'].shift(20)
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    df['close_momentum'] = df['price_change_20'] / df['ema_close']
    
    # Final Factor
    df['alpha_factor'] = (
        0.4 * df['momentum'] +
        0.3 * df['volatility'] +
        0.2 * df['volume_momentum'] +
        0.05 * df['intraday_volatility'] +
        0.03 * df['avg_volume_spike'] +
        0.02 * df['close_momentum']
    )
    
    return df['alpha_factor']
```
