```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from open to close
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the percentage of the high price relative to the low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the amount to volume ratio
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Heuristic: Weighted combination of price change, high-low ratio, and amount-volume ratio
    df['alpha_factor'] = (df['price_change'] * 0.4) + \
                         (df['high_low_ratio'] * 0.3) + \
                         (df['amount_volume_ratio'] * 0.3)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
