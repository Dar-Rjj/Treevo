```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (intraday_return + 1e-6)  # Avoid division by zero
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'] - df['close'].shift(1)
    price_change_momentum = df['close'] - df['open'].shift(1)
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    integrated_factor = combined_momentum_reversal + volume_adjusted_range + price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open']
    integrated_factor += close_to_open_diff * (df['volume'] * df['amount'])
    
    # Adjust Momentum by Open-Close Spread
    open_close_spread = df['close'] - df['open']
    adjusted_momentum = integrated_factor * np.abs(open_close_spread)
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_factor = adjusted_momentum.ewm(span=5).mean()
    
    return smoothed_factor
```
