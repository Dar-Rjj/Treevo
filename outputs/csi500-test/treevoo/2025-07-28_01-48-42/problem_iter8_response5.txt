```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_momentum = momentum * (df['volume'] * high_low_range)
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = volume_adjusted_momentum * abs(open_close_spread)
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_factor = adjusted_momentum / (df['volume'] * high_low_range)
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change()
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = (df['high'] - df['open']).pct_change()
    intraday_returns_low = (df['low'] - df['open']).pct_change()
    intraday_volatility = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1, ddof=0).rolling(window=10).std()
    
    # Calculate Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'].diff()
    
    # Multiply Price Change Momentum by Volume Delta
    price_volume_momentum = price_change_momentum * volume_delta
    
    # Apply Exponential Moving Average (EMA) with 5-day span
    smoothed_price_volume_momentum = price_volume_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = (
        0.4 * combined_factor +
        0.3 * intraday_volatility +
        0.2 * smoothed_price_volume_momentum +
        0.1 * volume_spike
    )
    
    return alpha_factor
```
