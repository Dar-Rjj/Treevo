```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Returns
    df['intraday_return_high'] = df['high'] - df['open']
    df['intraday_return_low'] = df['low'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_returns = (df['intraday_return_high'] + df['intraday_return_low']) / 2
    df['intraday_return_volatility'] = intraday_returns.rolling(window=10).std()
    
    # Calculate Average Volume Spike
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_spike'] = df['volume_diff'].rolling(window=5).sum()
    
    # Calculate Close Price Momentum
    df['price_change_20d'] = df['close'] - df['close'].shift(20)
    df['close_price_momentum_ema'] = df['price_change_20d'].ewm(span=14, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Momentum
    df['momentum'] = df['price_change'].rolling(window=10).sum()
    
    # Adjust for Volume
    df['volume_adjusted_range'] = df['volume'] * df['high_low_range']
    
    # Combine Momentum and Volume-Adjusted Range
    df['combined_momentum'] = df['momentum'].rolling(window=10).sum() / df['volume_adjusted_range']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['combined_momentum'] / df['daily_volatility']
    
    # Smoothed Momentum
    df['smoothed_momentum'] = df['close'].pct_change().ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_std = df['volume_change'].rolling(window=20).std()
    df['volume_spike_factor'] = (df['volume_change'] > 2 * volume_std).astype(int)
    
    # Weighted Factors
    df['weighted_intraday_volatility'] = df['intraday_return_volatility'] * 0.5
    df['weighted_avg_volume_spike'] = df['avg_volume_spike'] * 0.3
    df['weighted_close_price_momentum'] = (df['price_change_20d'] / df['close_price_momentum_ema']) * 0.2
    df['weighted_factors'] = df['weighted_intraday_volatility'] + df['weighted_avg_volume_spike'] + df['weighted_close_price_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['volatility_adjusted_momentum'] * df['smoothed_momentum'] * df['volume_spike_factor']
    
    return df['final_alpha_factor']
```
