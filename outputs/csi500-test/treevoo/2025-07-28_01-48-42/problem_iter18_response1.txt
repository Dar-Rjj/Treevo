import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.exp(-intraday_return))  # Sigmoid function to invert the return
    
    # Adjust for Volume and Amount
    volume_amount_adjusted = high_low_range * df['amount'] * df['volume']
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = momentum_reversal + volume_amount_adjusted
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    close_open_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    price_volume_momentum = close_open_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_open_diff = df['close'] - df['open']
    final_factor = integrated_signal * close_open_diff
    
    # Apply a Smoothing Filter to the Combined Signal
    final_factor_smoothed = final_factor.ewm(span=5).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    final_factor_with_volatility = final_factor_smoothed * daily_volatility
    
    # Apply a Second Smoothing Layer
    final_factor_sma = final_factor_with_volatility.rolling(window=10).mean()
    
    return final_factor_sma
