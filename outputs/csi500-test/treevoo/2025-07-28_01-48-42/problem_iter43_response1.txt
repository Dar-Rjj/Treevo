```python
import pandas as pd

def heuristics_v2(df):
    # Compute Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * abs(open_close_spread)
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Determine Trend Direction
    trend_direction = (df['close'] > df['open']).astype(int)  # Positive if close > open, else negative
    
    # Modify Momentum based on Trend
    high_low_ratio = (df['high'] - df['low']) / df['open']
    momentum_trend_adjusted = volume_weighted_momentum * ((1 + trend_direction * high_low_ratio) + (1 - (1 - trend_direction) * high_low_ratio))
    
    # Calculate Opening Price Volatility
    volatility_lookback = 5
    opening_price_volatility = df['open'].rolling(window=volatility_lookback).std()
    open_volatility_ratio = opening_price_volatility / df['open']
    
    # Further Adjust Momentum
    final_momentum = momentum_trend_adjusted * ((1 + trend_direction * open_volatility_ratio) + (1 - (1 - trend_direction) * open_volatility_ratio))
    
    return final_momentum
```
