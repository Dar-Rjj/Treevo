```python
import pandas as pd

def heuristics_v2(df):
    # Calculate momentum indicators
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    sma_ratio = sma_short / sma_long
    
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    ema_ratio = ema_short / ema_long
    
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate trend indicators
    macd_line = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    plus_di = (df['high'] - df['high'].shift(1)).ewm(span=14, adjust=False).mean()
    minus_di = (df['low'].shift(1) - df['low']).ewm(span=14, adjust=False).mean()
    dx = 100 * (plus_di - minus_di) / (plus_di + minus_di)
    adx = dx.ewm(span=14, adjust=False).mean()
    
    # Calculate volatility indicator
    tr = pd.DataFrame({
        'hl': df['high'] - df['low'],
        'hc': (df['high'] - df['close'].shift()).abs(),
        'lc': (df['low'] - df['close'].shift()).abs()
    }).max(axis=1)
    atr = tr.ewm(span=14, adjust=False).mean()
    
    # Calculate volume-based indicators
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv = obv.cumsum()
    
    cmf_numerator = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    cmf_denominator = df['volume'].rolling(window=20).sum()
    cmf = cmf_numerator.rolling(window=20).sum() / cmf_denominator
    
    # Integrate multiple indicators
    cmi = (sma_ratio + ema_ratio + roc) / 3
    cti = (macd_histogram + adx) / 2
    cvi = atr
    cvoi = (obv + cmf) / 2
    
    # Generate final alpha factor
    faf = 0.4 * cmi + 0.3 * cti + 0.15 * cvi + 0.15 * cvoi
    
    # Additional alpha factors
    pvr = df['close'] / df['volume']
    hls = df['high'] - df['low']
    ocs = df['close'] - df['open']
    
    # Combine and refine composite indicators
    ecm = (cmi + pvr) / 2
    ect = (cti + hls) / 2
    ecv = (cvi + ocs) / 2
    ecvo = (cvoi + pvr) / 2
    
    # Generate refined final alpha factor
    r_faf = 0.4 * ecm + 0.3 * ect + 0.15 * ecv + 0.15 * ecvo
    
    return r_faf
```
