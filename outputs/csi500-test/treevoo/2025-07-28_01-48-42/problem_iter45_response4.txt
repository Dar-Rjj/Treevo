def heuristics_v2(df):
    # Compute Open-Close Spread
    spread = df['open'] - df['close']

    # Calculate Raw Momentum with a lookback period of 10 days
    raw_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * spread.abs()

    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']

    # Incorporate Enhanced Price Trend Sensitivity
    trend_direction = (df['close'] > df['open']).astype(int)
    high_low_range = df['high'] - df['low']
    
    # Modify Momentum based on Trend
    price_trend_adjustment = (1 + (high_low_range / df['open']) * (1 if trend_direction == 1 else -1))
    momentum_with_price_trend = volume_weighted_momentum * price_trend_adjustment

    # Incorporate Opening Price Volatility
    opening_volatility = df['open'].rolling(window=5).std()
    opening_volatility_ratio = opening_volatility / df['open']
    final_momentum_with_price_trend = momentum_with_price_trend * (1 + (opening_volatility_ratio if trend_direction == 1 else -opening_volatility_ratio))

    # Incorporate Volume Trend Sensitivity
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_trend_direction = (df['volume'] > avg_volume).astype(int)
    volume_trend_ratio = (df['volume'] - avg_volume) / avg_volume
    volume_trend_adjustment = (1 + volume_trend_ratio * (1 if volume_trend_direction == 1 else -1))
    final_momentum_with_volume_trend = final_momentum_with_price_trend * volume_trend_adjustment

    # Introduce Intraday Momentum Factor
    intraday_return = (df['close'] - df['open']) / df['open']
    avg_intraday_return = intraday_return.rolling(window=5).mean()
    intraday_momentum = intraday_return - avg_intraday_return

    # Adjust Intraday Momentum by Spread
    adjusted_intraday_momentum = intraday_momentum * spread.abs()

    # Apply Volume Weighting to Intraday Momentum
    volume_weighted_intraday_momentum = adjusted_intraday_momentum * df['volume']

    # Determine Intraday Trend Direction
    intraday_trend_direction = ((df['close'] - df['low']) > (df['high'] - df['open'])).astype(int)
    intraday_high_low_range = df['high'] - df['low']
    
    # Modify Intraday Momentum based on Trend
    intraday_price_trend_adjustment = (1 + (intraday_high_low_range / df['open']) * (1 if intraday_trend_direction == 1 else -1))
    intraday_momentum_with_price_trend = volume_weighted_intraday_momentum * intraday_price_trend_adjustment

    # Incorporate Intraday Opening Price Volatility
    intraday_opening_volatility = df['open'].rolling(window=5).std()
    intraday_opening_volatility_ratio = intraday_opening_volatility / df['open']
    final_intraday_momentum_with_price_trend = intraday_momentum_with_price_trend * (1 + (intraday_opening_volatility_ratio if intraday_trend_direction == 1 else -intraday_opening_volatility_ratio))

    # Final Intraday Momentum Adjustment
    final_factor = final_momentum_with_volume_trend * final_intraday_momentum_with_price_trend

    return final_factor
