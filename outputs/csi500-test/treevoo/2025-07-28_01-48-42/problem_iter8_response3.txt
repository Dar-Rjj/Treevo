```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = (high_low_range * df['volume'] * df['amount'])
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_momentum_volume_amount = momentum / volume_amount_adjusted_range
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum_volume_amount + 
                    (volume_amount_adjusted_range + weighted_price_change))
    
    return alpha_factor
```
