```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_return_volatility = (intraday_returns_high - intraday_returns_low).rolling(window=10).std()
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_high_low = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * abs(open_close_spread)
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_signal = adjusted_momentum / volume_adjusted_high_low
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Generate Factor Value
    factor_value = (df['high'].rolling(window=10).mean() - df['low'].rolling(window=10).mean()) * df['volume'].rolling(window=10).mean() / df['close']
    
    # Introduce New Factor: Price-Volatility Ratio
    price_volatility_ratio = price_change / intraday_return_volatility
    
    # Combine All Factors
    combined_factors = (intraday_return_volatility * 0.3) + (factor_value * 0.4) + (enhanced_combined_signal * 0.2) + (price_volatility_ratio * 0.1)
    
    # Final Alpha Factor
    volume_spike_factor = (df['volume'] - df['volume'].shift(1)) > 2 * df['volume'].rolling(window=20).std()
    final_alpha_factor = combined_factors * enhanced_combined_signal * volume_spike_factor.astype(int)
    
    return final_alpha_factor
```
