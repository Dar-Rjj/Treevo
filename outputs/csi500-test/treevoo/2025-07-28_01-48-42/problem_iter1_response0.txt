```python
def heuristics_v2(data):
    # Calculate Intraday Momentum
    intraday_momentum = data['high'] - data['low']
    
    # Determine Reversal
    reversal = data['open'] - data['close']
    
    # Adjust by Volume
    volume_adjusted_reversal = reversal / data['volume']
    
    # Combine Intraday Momentum with Volume-Adjusted Reversal
    alpha_factor = intraday_momentum + volume_adjusted_reversal
    
    return alpha_factor
```
