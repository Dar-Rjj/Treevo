import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted_component = df['volume'] * high_low_range * df['amount']
    
    # Combine Momentum and Intraday Reversal
    momentum_difference = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = momentum_difference * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = combined_momentum_reversal + volume_adjusted_component
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    integrated_with_close_open = integrated_factor * close_open_diff
    
    # Adjust Momentum by Open-Close Spread
    open_close_spread = df['close'] - df['open']
    momentum_adjusted_by_spread = combined_momentum_reversal * abs(open_close_spread)
    
    # Final Alpha Factor
    final_alpha = integrated_with_close_open + momentum_adjusted_by_spread
    
    return final_alpha
