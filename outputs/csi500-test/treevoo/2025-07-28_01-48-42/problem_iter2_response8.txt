```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Momentum
    df['momentum'] = df['price_change'].rolling(window=10).sum()
    
    # Adjust for Volume
    df['volume_adjusted_range'] = df['volume'] * df['high_low_range']
    
    # Combine Momentum and Volume-Adjusted Range
    df['combined_momentum'] = df['momentum'] / df['volume_adjusted_range']
    
    # Calculate Price Momentum
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    df['smoothed_momentum'] = df['daily_return'].ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['smoothed_momentum'] * df['volume_spike']
    
    return df['alpha_factor']
```
