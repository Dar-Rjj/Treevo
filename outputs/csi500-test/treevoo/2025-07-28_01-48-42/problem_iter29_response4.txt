```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff().fillna(0)
    
    # Compute 5-Day Momentum
    momentum_5d = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum_5d * abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum / combined_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_momentum_reversal = enhanced_combined_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open'].shift(1).fillna(0)
    volume_adjusted_reversal = intraday_reversal * df['volume']
    
    # Adjust Momentum by Open-Close Spread (Intraday Reversal)
    weighted_momentum_reversal = momentum_5d * abs(open_close_spread) + volume_adjusted_reversal * close_to_open_diff
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_signal = combined_momentum_reversal + weighted_momentum_reversal
    
    # Calculate Price Momentum
    daily_close_return = df['close'].pct_change().fillna(0)
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff().fillna(0)
    std_volume = df['volume'].rolling(window=20).std().fillna(0)
    volume_spike = (volume_change > 2 * std_volume).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().fillna(0)
    trend_strength = long_term_momentum.ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    intraday_returns = (df['high'] - df['low']) / df['low']
    intraday_volatility = intraday_returns.rolling(window=5).std().fillna(0)
    
    # Adjust Intraday Reversal Score by Intraday Volatility
    adjusted_intraday_reversal = intraday_reversal * intraday_volatility
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum_volatility = momentum_5d * intraday_volatility
    
    # Generate Factor Value
    high_low_momentum = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / df['close']
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    factor_value = high_low_momentum * volume_moving_avg / df['close']
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = enhanced_combined_signal.ewm(span=5, adjust=False).mean()
    
    # Combine All Factors
    combined_factors = 0.3 * intraday_volatility + 0.4 * factor_value + 0.2 * enhanced_combined_signal + 0.1 * smoothed_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (combined_factors * smoothed_momentum * volume_spike * trend_strength * intraday_volatility) + close_to_open_diff
    
    return final_alpha_factor
```
