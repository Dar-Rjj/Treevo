import pandas as pd

def heuristics_v2(df):
    # Calculate Logarithmic Daily Returns
    df['log_returns'] = df['close'].apply(lambda x: x / df['close'].shift(1)).apply(lambda x: np.log(x))

    # Momentum Component
    df['21_day_moving_avg_returns'] = df['log_returns'].rolling(window=21).mean()
    df['60_day_moving_avg_returns'] = df['log_returns'].rolling(window=60).mean()
    df['momentum'] = df['21_day_moving_avg_returns'] - df['60_day_moving_avg_returns']

    # Volatility Component
    df['abs_log_returns'] = df['log_returns'].abs()
    df['21_day_sum_abs_returns'] = df['abs_log_returns'].rolling(window=21).sum()
    df['60_day_sum_abs_returns'] = df['abs_log_returns'].rolling(window=60).sum()
    df['volatility'] = df['21_day_sum_abs_returns'] - df['60_day_sum_abs_returns']

    # Final Factor
    df['factor_value'] = df['momentum'] + df['volatility']
    
    return df['factor_value']
