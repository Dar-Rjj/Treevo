```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Determine Lookback Period for Momentum and ATR
    lookback_period_momentum = 10
    lookback_period_volatility = 10
    lookback_period_movement = 20
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].rolling(window=lookback_period_momentum).mean()
    
    # Adjust Momentum by Absolute Open-Close Spread
    adjusted_momentum = momentum / open_close_spread.abs()
    
    # Weight Adjusted Momentum by Volume
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    sum_weighted_adjusted_momentum = weighted_adjusted_momentum.rolling(window=lookback_period_momentum).sum()
    total_volume = df['volume'].rolling(window=lookback_period_momentum).sum()
    
    # Final Alpha Factor Component 1
    alpha_factor_1 = sum_weighted_adjusted_momentum / total_volume
    
    # Calculate Intraday Volatility (High-Low Difference)
    intraday_volatility = df['high'] - df['low']
    
    # Calculate True Range for each day
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Average True Range (ATR) over a lookback period
    atr = true_range.rolling(window=lookback_period_volatility).mean()
    
    # Weight Intraday Volatility by Volume
    weighted_intraday_volatility = intraday_volatility * df['volume']
    sum_weighted_intraday_volatility = weighted_intraday_volatility.rolling(window=lookback_period_volatility).sum()
    
    # Final Alpha Factor Component 2
    alpha_factor_2 = sum_weighted_intraday_volatility / total_volume
    
    # Calculate Price Movement Consistency
    moving_average_close = df['close'].rolling(window=lookback_period_movement).mean()
    daily_returns = df['close'].pct_change()
    std_daily_returns = daily_returns.rolling(window=lookback_period_movement).std()
    
    # Final Alpha Factor Component 3
    alpha_factor_3 = std_daily_returns / moving_average_close
    
    # Combine Final Alpha Factors
    final_alpha_factor = alpha_factor_1 + alpha_factor_2 + alpha_factor_3
    
    return final_alpha_factor
```
