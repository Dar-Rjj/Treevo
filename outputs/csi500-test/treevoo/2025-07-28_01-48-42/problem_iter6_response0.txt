```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    momentum = high_low_range.diff()
    momentum_5day_sum = momentum.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    avg_volume_5day = df['volume'].rolling(window=5).mean()
    avg_amount_5day = df['amount'].rolling(window=5).mean()
    
    adjusted_momentum = momentum_5day_sum / avg_volume_5day / avg_amount_5day
    
    # Apply Moving Average Smoothing
    smoothed_factor = adjusted_momentum.rolling(window=10).mean()
    
    return smoothed_factor
```
