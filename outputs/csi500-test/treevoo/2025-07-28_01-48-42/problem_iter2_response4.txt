```python
def heuristics_v2(data):
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = data['close'] - data['open']
    
    # Calculate Price Change
    price_change = data['close'].diff()
    
    # Compute Momentum over the last 10 days
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * open_close_spread.abs()
    
    # Volume-Adjust High-Low Range
    volume_adjusted_high_low = data['volume'] * high_low_range
    
    # Combine Volume-Adjusted Components
    factor = adjusted_momentum / volume_adjusted_high_low
    
    return factor
```
