```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate the trend strength based on the difference between Close and Open
    df['TrendStrength'] = (df['Close'] - df['Open']).apply(lambda x: x if x > 0 else -x)
    
    # Identify the presence of a long wick (shadow)
    df['WickIndicator'] = ((df['High'] - df['Low']) / (df['Close'] - df['Open']).abs()).apply(lambda x: 1 if x > 2 else 0)
    
    # Compute a factor that measures the ratio of volume change to price change
    df['VolumePriceMomentum'] = 0.0
    for i in range(1, len(df)):
        if df['Volume'].iloc[i] > df['Volume'].iloc[i-1]:
            if (df['Close'].iloc[i] - df['Close'].iloc[i-1]) > 0:
                df.at[df.index[i], 'VolumePriceMomentum'] = (df['Volume'].iloc[i] / df['Volume'].iloc[i-1]) * (df['Close'].iloc[i] - df['Close'].iloc[i-1])
        elif df['Volume'].iloc[i] < df['Volume'].iloc[i-1]:
            if (df['Close'].iloc[i] - df['Close'].iloc[i-1]) < 0:
                df.at[df.index[i], 'VolumePriceMomentum'] = (df['Volume'].iloc[i] / df['Volume'].iloc[i-1]) * (df['Close'].iloc[i] - df['Close'].iloc[i-1])
    
    # Calculate the Average True Range (ATR) over a specified period without normalization
    df['ATR'] = (df['High'] - df['Low']).rolling(window=n).sum() / n
    
    # Use ATR to create a factor that adjusts the impact of price changes based on recent volatility
    df['AdjustedPriceChangeFactor'] = (df['Close'] - df['Close'].shift(1)) / df['ATR']
    
    # Develop a factor that compares the current day's volume to the moving average of past volumes
    df['MAV'] = df['Volume'].rolling(window=n).mean()
    df['VolumeMomentumFactor'] = (df['Volume'] - df['MAV']) / df['MAV']
    
    # Calculate the average amount traded over a period
    df['AAT'] = df['Amount'].rolling(window=n).mean()
    
    # Create a factor that measures the deviation of the current day's amount from the average
    df['AmountDeviationFactor'] = (df['Amount'] - df['AAT']) / df['AAT']
    
    # Combine all factors into a single alpha factor
    df['AlphaFactor'] = (
        df['TrendStrength'] + 
        df['WickIndicator'] + 
        df['VolumePriceMomentum'] + 
        df['AdjustedPriceChangeFactor'] + 
        df['VolumeMomentumFactor'] + 
        df['AmountDeviationFactor']
    )
    
    return df['AlphaFactor']
```
