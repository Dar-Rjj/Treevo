```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff(1)
    
    # Compute Momentum: Sum of price change over the last 10 days
    momentum = price_change.rolling(window=10).sum()
    
    # Calculate Volume Delta: Volume (t) minus Volume (t-1)
    volume_delta = df['volume'].diff(1)
    
    # Adjust for Volume: Multiply High-Low Range by Volume Delta
    volume_adjusted_range = high_low_range * volume_delta
    
    # Combine Momentum and Volume-Adjusted Range
    factor = (momentum / volume_adjusted_range).rolling(window=10).sum()
    
    return factor
```
