```python
def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Calculate Momentum
    lookback_period = 10
    df['avg_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum'] = df['close'] - df['avg_close']
    
    # Adjust Momentum by Open-Close Spread
    df['adjusted_momentum'] = df['momentum'] * df['open_close_spread'].abs()
    
    # Weight by Volume
    df['volume_weighted_adjusted_momentum'] = df['adjusted_momentum'] * df['volume']
    
    return df['volume_weighted_adjusted_momentum']
```
