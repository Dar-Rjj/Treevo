```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily open-close spread
    spread = df['open'] - df['close']
    
    # Define the lookback period for momentum calculation
    lookback_period = 10
    
    # Calculate the simple moving average of the close prices over the lookback period
    sma_close = df['close'].rolling(window=lookback_period).mean()
    
    # Compute the momentum
    momentum = df['close'] - sma_close
    
    # Adjust the momentum by the absolute value of the spread
    adjusted_momentum = momentum / spread.abs()
    
    # Calculate the volume-weighted adjusted momentum
    weighted_momentum = adjusted_momentum * df['volume']
    
    return weighted_momentum
```
