```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.exp(-intraday_return))
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range * (avg_volume_5d * avg_amount_5d)
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_momentum * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'] - df['open'].shift(1)
    integrated_factor = combined_momentum_reversal + adjusted_high_low_range + price_change
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    daily_volume_rank = df['volume'].rank(pct=True, method='dense')
    daily_amount_rank = df['amount'].rank(pct=True, method='dense')
    close_to_open_diff = df['close'] - df['open'].shift(1)
    confirmed_factor = integrated_factor * close_to_open_diff * daily_volume_rank * daily_amount_rank
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_factor = confirmed_factor.ewm(span=5, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted_factor = smoothed_factor * daily_volatility
    
    # Introduce a New Liquidity Component
    daily_liquidity = (df['amount'] / df['volume']) * 1000
    final_factor = volatility_adjusted_factor * daily_liquidity
    
    return final_factor
```
