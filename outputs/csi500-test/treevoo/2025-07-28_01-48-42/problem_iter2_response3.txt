```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['daily_return'] = df['close'].pct_change()
    df['smoothed_momentum'] = df['daily_return'].ewm(span=12).mean()

    # Calculate Intraday Reversal Score
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['intraday_reversal'] = -df['intraday_return']

    # Calculate Volume Spike Factor
    df['volume_change'] = df['volume'].diff()
    df['vol_spike'] = (df['volume_change'] > 2 * df['volume'].rolling(window=20).std()).astype(int)

    # Confirm with Volume Relative Strength
    df['vol_rank'] = df['volume'].rolling(window=20).rank(pct=True)
    df['vol_adjusted_reversal'] = df['intraday_reversal'] * df['vol_rank']

    # Consider Close-to-Open Price Change
    df['close_to_open_diff'] = df['open'].shift(-1) - df['close']
    df['final_vol_adjusted_reversal'] = df['vol_adjusted_reversal'] * df['close_to_open_diff']

    # Combine Factors
    df['alpha_factor'] = df['smoothed_momentum'] * df['vol_spike'] * df['final_vol_adjusted_reversal']
    return df['alpha_factor']
```
