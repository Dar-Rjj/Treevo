```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['close']
    
    # Compute Momentum (Sum over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * volume_delta
    
    # Calculate Daily Volatility
    daily_volatility = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust for Volatility
    combined_factor = (momentum + volume_adjusted_range) / daily_volatility
    
    # Combine Momentum, Volume-Adjusted Range, and Volatility Adjustment
    final_factor = combined_factor.rolling(window=10).sum() / volume_adjusted_range
    
    return final_factor
```
