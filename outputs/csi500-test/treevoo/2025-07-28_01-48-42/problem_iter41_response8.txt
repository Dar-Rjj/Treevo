```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].shift(lookback_period)
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * np.abs(open_close_spread)
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Incorporate Enhanced Price Trend Sensitivity
    trend_direction = (df['close'] > df['open']).astype(int) * 2 - 1
    high_low_range = df['high'] - df['low']
    ratio_high_low_to_open = high_low_range / df['open']
    modified_momentum = volume_weighted_momentum * (1 + trend_direction * ratio_high_low_to_open)
    
    # Incorporate Opening Price Volatility
    opening_price_volatility_lookback = 5
    opening_price_volatility = df['open'].rolling(window=opening_price_volatility_lookback).std()
    ratio_opening_volatility_to_open = opening_price_volatility / df['open']
    modified_momentum *= (1 + trend_direction * ratio_opening_volatility_to_open)
    
    # Incorporate Trading Volume Volatility
    volume_volatility_lookback = 5
    volume_volatility = df['volume'].rolling(window=volume_volatility_lookback).std()
    average_volume = df['volume'].rolling(window=volume_volatility_lookback).mean()
    ratio_volume_volatility_to_average_volume = volume_volatility / average_volume
    final_momentum = modified_momentum * (1 + trend_direction * ratio_volume_volatility_to_average_volume)
    
    return final_momentum
```
