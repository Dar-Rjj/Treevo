```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the closing price over a 10-day window
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of the closing price to the amount
    df['close_to_amount_ratio'] = df['close'] / df['amount']
    
    # Calculate the momentum of the closing price
    df['close_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate the weighted average of the above factors
    df['factor_value'] = (
        0.4 * (df['close'] - df['close_10d_mean']) +
        0.3 * df['high_low_diff'] +
        0.2 * df['close_to_amount_ratio'] +
        0.1 * df['close_momentum']
    )
    
    return df['factor_value']
```
