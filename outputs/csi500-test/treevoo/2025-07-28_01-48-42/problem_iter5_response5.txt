```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate High-Low Price Momentum
    df['high_low_momentum'] = df['high_low_diff'].rolling(window=10).sum()
    
    # Calculate Intraday Reversal Score
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    df['intraday_reversal_score'] = 1 / df['intraday_return']
    
    # Adjust by Volume and Amount Relative Strength
    df['volume_rank'] = df['volume'].rolling(window=21).apply(lambda x: pd.Series(x).rank().iloc[-1], raw=True)
    df['volume_trend'] = df['volume'].rolling(window=10).mean()
    df['amount_trend'] = df['amount'].rolling(window=10).mean()
    df['volume_amount_trend'] = df['volume_trend'] * df['amount_trend']
    df['volume_adjusted_reversal_score'] = df['intraday_reversal_score'] * df['volume_rank']
    
    # Combine High-Low Price Momentum, Volume, and Amount Trends
    df['combined_momentum'] = df['high_low_momentum'] * df['volume_amount_trend'] * df['volume_adjusted_reversal_score']
    
    # Integrate Intraday Reversal Effect
    df['integrated_momentum'] = df['combined_momentum'] + df['intraday_reversal_score']
    
    # Calculate Close-to-Open Price Change
    df['close_to_open_change'] = df['close'] - df['open'].shift(1)
    
    # Apply Dynamic Volume-Momentum
    df['price_momentum'] = df['close'] - df['open'].shift(1)
    df['volume_delta'] = df['volume'] - df['volume'].shift(1)
    df['dynamic_volume_momentum'] = df['price_momentum'] * df['volume_delta']
    
    # Apply a Smoothing Filter
    df['smoothed_signal'] = df['integrated_momentum'].ewm(span=5, adjust=False).mean() + df['dynamic_volume_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['smoothed_signal']
```
