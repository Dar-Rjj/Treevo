```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the midpoint of the high and low prices
    mid_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the average of open and close, and the midpoint of high and low
    price_diff = avg_open_close - mid_high_low
    
    # Calculate the ratio of amount to volume
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Combine the price difference and the amount-volume ratio
    factor = price_diff * amount_volume_ratio
    
    return factor
```
