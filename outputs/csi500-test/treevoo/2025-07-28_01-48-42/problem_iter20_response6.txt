```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_high_return = df['high'] - df['open']
    intraday_low_return = df['low'] - df['open']
    intraday_returns = (intraday_high_return + intraday_low_return) / 2
    intraday_volatility = intraday_returns.rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    reversal_score = -intraday_return  # Invert for Reversal
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Price Change
    price_change = df['close'].diff().fillna(0)
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    open_close_spread = df['close'] - df['open']
    weighted_momentum = momentum * abs(open_close_spread)
    
    # Combine Momentum, Intraday Reversal, and Volume-Adjusted Range
    high_low_diff = high_low_range.diff().fillna(0)
    combined_signal = (weighted_momentum + (high_low_diff * reversal_score))
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    adjusted_reversal = (volume_adjusted_range * reversal_score) * close_to_open_diff
    
    # Generate Factor Value
    factor_value = (momentum * df['volume'].rolling(window=10).mean()) / df['close']
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Combine All Factors
    final_factor = (intraday_volatility * 0.3) + (factor_value * 0.4) + (smoothed_combined_signal * 0.2) + (adjusted_reversal * 0.1)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_factor += combined_signal
    
    # Final Alpha Factor
    # Multiply Combined Factors by Smoothened Momentum
    final_factor *= smoothed_combined_signal
    
    # Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_std = df['volume'].rolling(window=20).std()
    volume_spike = (volume_change > 2 * volume_std).astype(int)
    final_factor *= volume_spike
    
    return final_factor
```
