```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / (1 + intraday_return)
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    adjusted_high_low_range = volume_adjusted + amount_adjusted
    
    # Integrate Momentum and Reversal
    momentum_and_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'] - df['close'].shift(1)
    weighted_price_change = price_change * (df['volume'] + np.sqrt(df['amount']))
    integrated_signal = momentum_and_reversal + weighted_price_change + adjusted_high_low_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = np.abs(df['open'] - df['close'])
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_signal = integrated_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open'].shift(-1)
    final_factor = smoothed_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_return = df['close'] / df['close'].shift(1) - 1
    smoothed_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'] / df['volume'].shift(1) - 1
    volume_spike = (day_to_day_volume_change > 2 * day_to_day_volume_change.rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    final_alpha = final_factor * smoothed_momentum * volume_spike + close_to_open_diff
    
    return final_alpha
```
