```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range - high_low_range.shift(1)
    high_low_momentum_sum = high_low_momentum.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    adjusted_momentum = high_low_momentum_sum / volume_5d_avg
    
    # Apply Exponential Moving Average Smoothing
    smoothed_adjusted_momentum = adjusted_momentum.ewm(span=20, min_periods=20).mean()
    
    # Calculate Logarithmic Daily Returns
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    momentum_21d = daily_returns.ewm(span=21, min_periods=21).mean()
    momentum_63d = daily_returns.ewm(span=63, min_periods=63).mean()
    final_momentum = momentum_21d - momentum_63d
    
    # Volatility Component
    abs_daily_returns = daily_returns.abs()
    volatility_21d = abs_daily_returns.ewm(span=21, min_periods=21).mean()
    volatility_63d = abs_daily_returns.ewm(span=63, min_periods=63).mean()
    final_volatility = volatility_21d - volatility_63d
    
    # Volume Component
    log_volume_changes = np.log(df['volume'] / df['volume'].shift(1))
    volume_21d = log_volume_changes.ewm(span=21, min_periods=21).mean()
    volume_63d = log_volume_changes.ewm(span=63, min_periods=63).mean()
    final_volume_momentum = volume_21d - volume_63d
    
    # Combine Components
    final_alpha_factor_2 = final_momentum + final_volatility + final_volume_momentum
    final_alpha_factor = smoothed_adjusted_momentum + final_alpha_factor_2
    
    return final_alpha_factor
```
