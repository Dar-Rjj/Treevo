```python
def heuristics_v2(df):
    # Calculate the relative change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the amount change
    df['amount_change'] = df['amount'].pct_change()
    
    # Combine the factors to form a composite alpha factor
    df['alpha_factor'] = (df['close_change'] * 0.4 + 
                          df['range'] * 0.3 + 
                          df['volume_change'] * 0.15 + 
                          df['amount_change'] * 0.15)
    
    return df['alpha_factor'].dropna()
```
