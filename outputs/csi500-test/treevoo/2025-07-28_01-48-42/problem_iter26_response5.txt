```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_high_returns = df['high'] - df['open']
    intraday_low_returns = df['open'] - df['low']
    intraday_volatility = (intraday_high_returns + intraday_low_returns).rolling(window=10).std()
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_range = volume_adjusted_range + amount_adjusted_range
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust Momentum by Open-Close Spread
    momentum_adjusted = five_day_momentum * np.abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = momentum_adjusted / combined_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_factor = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open'].shift(1)
    integrated_reversal = intraday_reversal * close_open_diff * df['volume']
    
    # Adjust Momentum by Open-Close Spread (Intraday Reversal)
    weighted_momentum = momentum_adjusted * np.abs(open_close_spread)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_combined_factor = combined_factor + weighted_momentum + integrated_reversal
    
    # Calculate Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].diff()
    volume_spike = (day_to_day_volume_change > 2 * day_to_day_volume_change.rolling(window=20).std()).astype(int)
    
    # Compute Trend Strength
    long_term_momentum = df['close'].pct_change().ewm(span=20, adjust=False).mean()
    
    # Generate Factor Value
    generated_factor = five_day_momentum / df['close']
    
    # New Factor: Price-Volatility Ratio
    price_volatility_ratio = price_change / intraday_volatility
    
    # Final Alpha Factor
    alpha_factor = (0.3 * intraday_volatility + 
                    0.2 * enhanced_signal + 
                    0.4 * generated_factor + 
                    0.1 * price_volatility_ratio) * volume_spike * long_term_momentum
    
    return alpha_factor
```
