```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between closing price and opening price
    df['close_open_diff'] = df['close'] - df['open']
    
    # Experiment with different values of N (e.g., 5, 10, 20) for rolling sum
    N_values = [5, 10, 20]
    for N in N_values:
        df[f'close_open_diff_sum_{N}'] = df['close_open_diff'].rolling(window=N).sum()
    
    # Calculate the percentage change in closing price from the previous day
    df['daily_return'] = df['close'].pct_change()
    
    # Experiment with different values of M (e.g., 10, 30, 60) for rolling std
    M_values = [10, 30, 60]
    for M in M_values:
        df[f'daily_return_std_{M}'] = df['daily_return'].rolling(window=M).std()
    
    # Compute the ratio of volume to price change (volume / (close - open))
    df['volume_price_ratio'] = df['volume'] / df['close_open_diff']
    
    # Vary K (e.g., 5, 14, 28) to explore short, medium, and long-term trends
    K_values = [5, 14, 28]
    for K in K_values:
        df[f'volume_price_ratio_avg_{K}'] = df['volume_price_ratio'].rolling(window=K).mean()
    
    # Calculate the average intraday range (high - low) over L days
    L_values = [7, 14, 30]
    df['intraday_range'] = df['high'] - df['low']
    for L in L_values:
        df[f'intraday_range_avg_{L}'] = df['intraday_range'].rolling(window=L).mean()
    
    # Formulate a factor that combines the close-open difference, daily volume, and their interaction
    df['close_open_volume_interaction'] = df['close_open_diff'] * df['volume']
    
    # Count the number of positive and negative returns over the last P days
    P_values = [5, 10, 21]
    for P in P_values:
        df[f'positive_returns_count_{P}'] = df['daily_return'].gt(0).rolling(window=P).sum()
        df[f'negative_returns_count_{P}'] = df['daily_return'].lt(0).rolling(window=P).sum()
    
    # Select one of the factors as the final alpha factor
    # For example, let's use the 10-day sum of close-open difference
    alpha_factor = df['close_open_diff_sum_10']
    
    return alpha_factor
```
