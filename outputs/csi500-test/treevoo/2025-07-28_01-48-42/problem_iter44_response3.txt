```python
import pandas as pd

def heuristics_v2(df):
    # Compute Open-Close Spread
    spread = df['open'] - df['close']
    
    # Calculate Raw Momentum
    lookback_period = 10
    raw_momentum = df['close'] - df['close'].shift(lookback_period)
    
    # Adjust Momentum by Spread
    adjusted_momentum = raw_momentum * spread.abs()
    
    # Apply Volume Weighting
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Incorporate Enhanced Price Trend Sensitivity
    trend = (df['close'] > df['open']).astype(int)  # 1 for positive, 0 for negative
    high_low_range = df['high'] - df['low']
    price_trend_factor = (1 + (high_low_range / df['open']) * (2 * trend - 1))
    momentum_with_price_trend = volume_weighted_momentum * price_trend_factor
    
    # Incorporate Opening Price Volatility
    opening_volatility_lookback = 5
    opening_price_volatility = df['open'].rolling(window=opening_volatility_lookback).std()
    opening_price_volatility_factor = (1 + (opening_price_volatility / df['open']) * (2 * trend - 1))
    momentum_with_opening_volatility = momentum_with_price_trend * opening_price_volatility_factor
    
    # Incorporate Volume Trend Sensitivity
    volume_trend_lookback = 10
    average_volume = df['volume'].rolling(window=volume_trend_lookback).mean()
    volume_trend = (df['volume'] > average_volume).astype(int)  # 1 for positive, 0 for negative
    volume_trend_factor = 1 + ((df['volume'] - average_volume) / average_volume) * (2 * volume_trend - 1)
    momentum_with_volume_trend = momentum_with_opening_volatility * volume_trend_factor
    
    # Incorporate Trading Volume Volatility
    volume_volatility_lookback = 5
    volume_volatility = df['volume'].rolling(window=volume_volatility_lookback).std()
    volume_volatility_factor = (1 + (volume_volatility / average_volume) * (2 * trend - 1))
    final_momentum = momentum_with_volume_trend * volume_volatility_factor
    
    return final_momentum
```
