```python
import pandas as pd

def heuristics_v2(df, period=10):
    # Calculate simple moving average of close prices over the specified period
    df['SMA'] = df['close'].rolling(window=period).mean()
    
    # Determine the difference between the current close price and the moving average
    df['Close_SMA_Diff'] = df['close'] - df['SMA']
    
    # Measure the change in volume from the previous day
    df['Volume_Change'] = df['volume'].diff()
    
    # Calculate the ratio of price change to volume change
    df['Price_Volume_Ratio'] = (df['close'].pct_change()) / df['Volume_Change']
    
    # Assess the range (high - low) as a measure of intraday volatility
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Consider the ratio of (close - open) to (high - low) to identify closing strength or weakness
    df['Close_Open_Ratio'] = (df['close'] - df['open']) / df['Intraday_Range']
    
    # Calculate the total traded amount over a specific period to gauge overall market interest
    df['Total_Amount'] = df['amount'].rolling(window=period).sum()
    
    # Use the ratio of the current amount to the moving average of the amount over a given period to detect significant deviations
    df['Amount_Moving_Avg'] = df['amount'].rolling(window=period).mean()
    df['Amount_Ratio'] = df['amount'] / df['Amount_Moving_Avg']
    
    # Combine the different factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Close_SMA_Diff'] + 
        df['Price_Volume_Ratio'] + 
        df['Close_Open_Ratio'] + 
        df['Amount_Ratio']
    )
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
```
