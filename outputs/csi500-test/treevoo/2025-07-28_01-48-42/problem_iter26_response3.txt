```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_return_volatility = (intraday_returns_high - intraday_returns_low).rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] + df['low']) / 2 - df['open']
    intraday_reversal_score = -intraday_return  # Invert for reversal
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    open_close_spread = df['close'] - df['open']
    weighted_momentum = momentum * open_close_spread.abs()
    
    # Combine Momentum, Intraday Reversal, and Volume-Adjusted Range
    high_low_range_diff = high_low_range.diff()
    combined_signal = high_low_range_diff * intraday_reversal_score + weighted_momentum
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'] - df['open'].shift(1)
    adjusted_reversal = intraday_reversal_score * close_to_open_diff
    combined_signal += adjusted_reversal
    
    # Generate Factor Value
    factor_value = (high_low_range.rolling(window=10).sum() * df['volume'].rolling(window=10).mean()) / df['close']
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Combine All Factors
    combined_factors = (intraday_return_volatility * 0.3) + (factor_value * 0.4) + (smoothed_combined_signal * 0.2) + (factor_value * 0.1)
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    combined_factors += volume_adjusted_range * price_change
    
    # Final Alpha Factor
    volume_change = df['volume'].diff()
    volume_spike_factor = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    final_alpha_factor = combined_factors * smoothed_combined_signal * volume_spike_factor
    
    return final_alpha_factor
```
