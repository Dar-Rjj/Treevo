import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Momentum
    lookback_period = 10
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * open_close_spread.abs()
    
    # Weight Adjusted Momentum by Volume
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Weighted Price Change
    weighted_price_change = (price_change * df['volume']).rolling(window=lookback_period).sum()
    
    # Adjust Weighted Price Change by Open-Close Spread
    adjusted_weighted_price_change = weighted_price_change * open_close_spread.abs()
    
    # Volume-Adjust High-Low Range
    volume_adjusted_high_low_range = df['volume'] * high_low_range
    
    # Combine Volume-Adjusted Components
    combined_volume_adjusted_components = volume_weighted_momentum / volume_adjusted_high_low_range
    
    # Final Alpha Factor
    final_alpha_factor = volume_weighted_momentum + combined_volume_adjusted_components
    
    return final_alpha_factor
