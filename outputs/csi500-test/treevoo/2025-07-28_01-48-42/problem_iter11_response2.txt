```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = df['high'] - df['low']
    intraday_reversal = -intraday_return  # Invert for reversal
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate High-Low Range Momentum
    high_low_momentum = high_low_range.diff()
    high_low_momentum_5d = high_low_momentum.rolling(window=5).sum()
    
    # Adjust for Volume
    weighted_momentum = (df['close'].diff() * df['volume']).rolling(window=10).sum()
    volume_adjusted_momentum = weighted_momentum * open_close_spread.abs()
    volume_adjusted_high_low_range = df['volume'] * high_low_range
    normalized_momentum = high_low_momentum_5d / volume_adjusted_high_low_range
    
    # Integrate Momentum, Reversal, and Volume-Adjusted Components
    momentum_reversal = normalized_momentum * intraday_reversal
    close_open_diff = df['close'] - df['open'].shift(1)
    integrated_factor = momentum_reversal * close_open_diff
    
    # Apply Moving Average Smoothing
    final_alpha_factor = integrated_factor.rolling(window=10).mean()
    
    return final_alpha_factor
```
