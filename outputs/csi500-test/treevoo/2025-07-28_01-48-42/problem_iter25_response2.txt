```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Compute 15-day Momentum
    df['15_day_momentum'] = df['price_change'].rolling(window=15).sum()
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volatility
    df['adjusted_15_day_momentum'] = df['15_day_momentum'] / df['daily_volatility']
    
    # Calculate Logarithmic Daily Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    df['ema_28_returns'] = df['log_returns'].ewm(span=28, adjust=False).mean()
    df['ema_70_returns'] = df['log_returns'].ewm(span=70, adjust=False).mean()
    df['momentum_component'] = df['ema_28_returns'] - df['ema_70_returns']
    
    # Volatility Component
    df['abs_log_returns'] = np.abs(df['log_returns'])
    df['ema_28_abs_returns'] = df['abs_log_returns'].ewm(span=28, adjust=False).mean()
    df['ema_70_abs_returns'] = df['abs_log_returns'].ewm(span=70, adjust=False).mean()
    df['volatility_component'] = df['ema_28_abs_returns'] - df['ema_70_abs_returns']
    
    # Volume Component
    df['log_volume_changes'] = np.log(df['volume'] / df['volume'].shift(1))
    df['ema_28_volume_changes'] = df['log_volume_changes'].ewm(span=28, adjust=False).mean()
    df['ema_70_volume_changes'] = df['log_volume_changes'].ewm(span=70, adjust=False).mean()
    df['volume_momentum'] = df['ema_28_volume_changes'] - df['ema_70_volume_changes']
    
    # Open-Close Spread
    df['open_close_diff'] = df['close'] - df['open']
    df['ema_28_open_close_diff'] = df['open_close_diff'].ewm(span=28, adjust=False).mean()
    
    # Combine Components
    df['final_momentum'] = df['momentum_component']
    df['final_volatility'] = df['volatility_component']
    df['final_volume_momentum'] = df['volume_momentum']
    df['final_open_close_spread'] = df['ema_28_open_close_diff']
    df['combined_alpha_factor'] = (df['final_momentum'] + 
                                   df['final_volatility'] + 
                                   df['final_volume_momentum'] + 
                                   df['final_open_close_spread'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_15_day_momentum'] + df['combined_alpha_factor']
    
    return df['final_alpha_factor']
```
