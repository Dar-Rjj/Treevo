import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Price Change
    df['Price_Change'] = df['close'].diff()
    
    # Compute Momentum
    df['Momentum'] = df['Price_Change'].rolling(window=15).sum()
    
    # Calculate Volume Delta
    df['Volume_Delta'] = df['volume'].diff()
    
    # Adjust for Volume
    df['Volume_Adjusted_Range'] = df['High_Low_Range'] * df['Volume_Delta']
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = df[['high', 'low', 'open']].std(axis=1)
    
    # Adjust for Volatility
    df['Volatility_Adjusted_Momentum'] = (df['Momentum'] + df['Volume_Adjusted_Range']) / df['Daily_Volatility']
    
    # Combine Momentum, Volume-Adjusted Range, and Volatility Adjustment
    df['Combined_Factor'] = df['Volatility_Adjusted_Momentum'].rolling(window=25).sum() / df['Volume_Adjusted_Range']
    
    # Calculate High-Low Range Momentum
    df['High_Low_Range_Momentum'] = df['High_Low_Range'].diff().rolling(window=15).sum()
    
    # Adjust for Volume
    df['7Day_Avg_Volume'] = df['volume'].rolling(window=7).mean()
    df['Volume_Adjusted_High_Low_Range_Momentum'] = df['High_Low_Range_Momentum'] * df['7Day_Avg_Volume']
    
    # Apply Exponential Moving Average Smoothing
    df['Smoothed_Adjusted_Momentum'] = df['Volume_Adjusted_High_Low_Range_Momentum'].ewm(span=25).mean()
    
    # Calculate Logarithmic Daily Returns
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Momentum Component
    df['28Day_EMA_Returns'] = df['Log_Returns'].ewm(span=28).mean()
    df['70Day_EMA_Returns'] = df['Log_Returns'].ewm(span=70).mean()
    df['Momentum_Component'] = df['28Day_EMA_Returns'] - df['70Day_EMA_Returns']
    
    # Volatility Component
    df['Abs_Log_Returns'] = df['Log_Returns'].abs()
    df['28Day_EMA_Abs_Returns'] = df['Abs_Log_Returns'].ewm(span=28).mean()
    df['70Day_EMA_Abs_Returns'] = df['Abs_Log_Returns'].ewm(span=70).mean()
    df['Volatility_Component'] = df['28Day_EMA_Abs_Returns'] - df['70Day_EMA_Abs_Returns']
    
    # Volume Component
    df['Log_Volume_Changes'] = np.log(df['volume'] / df['volume'].shift(1))
    df['28Day_EMA_Volume_Changes'] = df['Log_Volume_Changes'].ewm(span=28).mean()
    df['70Day_EMA_Volume_Changes'] = df['Log_Volume_Changes'].ewm(span=70).mean()
    df['Volume_Component'] = df['28Day_EMA_Volume_Changes'] - df['70Day_EMA_Volume_Changes']
    
    # Open-Close Spread
    df['Open_Close_Diff'] = df['open'] - df['close']
    df['28Day_EMA_Open_Close_Diff'] = df['Open_Close_Diff'].ewm(span=28).mean()
    df['Open_Close_Spread'] = df['28Day_EMA_Open_Close_Diff']
    
    # Combine Components
    df['Final_Momentum'] = df['Momentum_Component']
    df['Final_Volatility'] = df['Volatility_Component']
    df['Final_Volume_Momentum'] = df['Volume_Component']
    df['Final_Open_Close_Spread'] = df['Open_Close_Spread']
    df['Final_Alpha_Factor_2'] = (df['Final_Momentum'] + df['Final_Volatility'] + 
                                  df['Final_Volume_Momentum'] + df['Final_Open_Close_Spread'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Smoothed_Adjusted_Momentum'] + df['Final_Alpha_Factor_2']
