```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change Momentum: Close price (t) - Open price (t-1)
    df['PriceChangeMomentum'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta: Volume (t) - Volume (t-1)
    df['VolumeDelta'] = df['volume'] - df['volume'].shift(1)
    
    # Multiply Price Change Momentum by Volume Delta
    df['CombinedFactor'] = df['PriceChangeMomentum'] * df['VolumeDelta']
    
    # Apply Exponential Moving Average (EMA) with 5-day span
    df['SmoothedFactor'] = df['CombinedFactor'].ewm(span=5, adjust=False).mean()
    
    return df['SmoothedFactor']
```
