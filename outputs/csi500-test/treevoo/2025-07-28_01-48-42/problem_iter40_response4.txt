```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over 20 days
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) over 10 days
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily price change (Close_t - Close_(t-1))
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate High-Low spread (High_t - Low_t)
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close/Open ratio (Close_t / Open_t)
    df['Close_Open_Ratio'] = df['close'] / df['open']
    
    # Calculate Close/Previous Close ratio (Close_t / Close_(t-1))
    df['Close_Prev_Close_Ratio'] = df['close'] / df['close'].shift(1)
    
    # Calculate Volume change (Volume_t - Volume_(t-1))
    df['Volume_Change'] = df['volume'].diff()
    
    # Calculate Volume-weighted close
    df['Volume_Weighted_Close'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Price-Volume Trend (PVT)
    df['PVT'] = (df['close'].diff() / df['close'].shift(1)) * df['volume']
    df['PVT'] = df['PVT'].cumsum()
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0
    df['OBV'] = df['OBV'].mask(df['close'] > df['close'].shift(1), df['volume'])
    df['OBV'] = df['OBV'].mask(df['close'] < df['close'].shift(1), -df['volume'])
    df['OBV'] = df['OBV'].cumsum()
    
    # Calculate Three-day price pattern
    df['Three_Day_Price_Pattern'] = (df['close'] - df['open']) + (df['close'].shift(1) - df['open'].shift(1)) + (df['close'].shift(2) - df['open'].shift(2))
    
    # Calculate Inside bar pattern
    df['Inside_Bar_Pattern'] = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1)).astype(int)
    
    # Combine multiple factors to create a final alpha factor
    df['Alpha_Factor'] = (
        df['SMA_20'] + 
        df['EMA_10'] + 
        df['Daily_Price_Change'] + 
        df['High_Low_Spread'] + 
        df['Close_Open_Ratio'] + 
        df['Close_Prev_Close_Ratio'] + 
        df['Volume_Change'] + 
        df['Volume_Weighted_Close'] + 
        df['PVT'] + 
        df['OBV'] + 
        df['Three_Day_Price_Pattern'] + 
        df['Inside_Bar_Pattern']
    )
    
    return df['Alpha_Factor']
```
