```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Momentum (Sum of Price Change over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    factor = momentum / volume_amount_adjusted_range
    
    return factor
```
