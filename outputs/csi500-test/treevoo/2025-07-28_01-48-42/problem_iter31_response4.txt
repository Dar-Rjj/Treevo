import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.abs(intraday_return))
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Integrate Momentum and Reversal
    momentum_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Multiply Price Change by Volume and Square Root of Amount
    price_change_volume_adjusted = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Add to Combined Momentum and Intraday Reversal
    combined_signal = momentum_reversal + price_change_volume_adjusted
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = np.abs(df['open'] - df['close'])
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'].shift(-1) - df['open']
    integrated_final_factor = smoothed_combined_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_return = (df['close'] / df['close'].shift(1)) - 1
    smoothed_price_momentum = daily_close_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_factor = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = integrated_final_factor * smoothed_price_momentum * volume_spike_factor + close_to_open_diff
    
    return final_alpha_factor
