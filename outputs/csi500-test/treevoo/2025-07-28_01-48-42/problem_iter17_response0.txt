```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price from the previous day
    price_change_pct = avg_price.pct_change()
    
    # Calculate the ratio of volume to amount
    vol_to_amt_ratio = df['volume'] / df['amount']
    
    # Calculate the percentage change in volume to amount ratio from the previous day
    vol_to_amt_ratio_change_pct = vol_to_amt_ratio.pct_change()
    
    # The alpha factor is the product of the percentage change in average price and the percentage change in volume to amount ratio
    alpha_factor = price_change_pct * vol_to_amt_ratio_change_pct
    
    return alpha_factor
```
