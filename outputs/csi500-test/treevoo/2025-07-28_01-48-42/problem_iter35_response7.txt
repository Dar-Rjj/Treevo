```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of the close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of the close price to the open price
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate the trading range (high - low)
    trading_range = df['high'] - df['low']
    
    # Calculate the ratio of the close price to the trading range
    close_to_range_ratio = df['close'] / trading_range
    
    # Calculate the volume weighted by the close price
    vol_weighted_close = df['volume'] * df['close']
    
    # Combine the above factors into a single alpha factor
    alpha_factor = (close_change + close_to_open_ratio - close_to_range_ratio) * vol_weighted_close
    
    return alpha_factor
```
