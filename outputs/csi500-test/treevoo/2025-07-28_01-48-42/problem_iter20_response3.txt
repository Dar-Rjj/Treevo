import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 - intraday_return
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range * (avg_volume_5d / avg_amount_5d)
    
    # Integrate Momentum and Reversal
    high_low_diff = high_low_range - high_low_range.shift(1)
    combined_momentum_reversal = high_low_diff * intraday_reversal
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = combined_momentum_reversal + adjusted_high_low_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    daily_volume_rank = df['volume'].rolling(window=20).rank(pct=True)
    daily_amount_rank = df['amount'].rolling(window=20).rank(pct=True)
    close_to_open_diff = df['close'] - df['open']
    
    # Multiply by Combined High-Low and Reversal
    confirmed_factor = integrated_factor * daily_volume_rank * daily_amount_rank * close_to_open_diff
    
    # Apply a Smoothing Filter to the Combined Signal
    ema_span = 5
    smoothed_factor = confirmed_factor.ewm(span=ema_span, adjust=False).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2 / 4) + ((df['close'] - df['open'])**2 / 4))
    volatility_integrated_factor = smoothed_factor * daily_volatility
    
    # Apply a Second Smoothing Layer
    sma_window = 10
    final_factor = volatility_integrated_factor.rolling(window=sma_window).mean()
    
    return final_factor
