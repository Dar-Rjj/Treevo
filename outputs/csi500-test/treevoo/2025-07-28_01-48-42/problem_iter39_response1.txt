```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = -intraday_return
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['volume']
    amount_adjusted = high_low_range * np.sqrt(df['amount'])
    volume_amount_adjusted = volume_adjusted + amount_adjusted
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    price_change = df['close'].diff()
    five_day_momentum = price_change.rolling(window=5).sum()
    open_close_spread = (df['close'] - df['open']).abs()
    adjusted_momentum = five_day_momentum * open_close_spread
    combined_signal = adjusted_momentum / volume_amount_adjusted
    
    # Enhance Combined Signal with Smoothing
    ema_span = 5
    smoothed_signal = combined_signal.ewm(span=ema_span, adjust=False).mean()
    
    # Integrate Intraday Reversal into Final Factor
    close_open_diff = df['close'] - df['open'].shift(1)
    integrated_reversal = intraday_reversal * close_open_diff
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted = daily_volatility * smoothed_signal * integrated_reversal
    
    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)
    trend_confirmed = volatility_adjusted * long_term_momentum
    
    # Add a New Momentum Adjustment
    short_term_momentum = df['close'] - df['close'].shift(10)
    momentum_adjusted = trend_confirmed * short_term_momentum
    
    # Introduce a New Intraday Volatility Component
    intraday_volatility = (df['high'] - df['low']) / df['close']
    final_factor = momentum_adjusted * intraday_volatility
    
    return final_factor
```
