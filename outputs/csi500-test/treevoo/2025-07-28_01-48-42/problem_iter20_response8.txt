```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    lookback_period = 10
    avg_close = df['close'].rolling(window=lookback_period).mean()
    adjusted_momentum = (df['close'] - avg_close) * abs(open_close_spread)
    
    # Weight Adjusted Momentum by Volume
    weighted_adjusted_momentum = adjusted_momentum * df['volume']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_return_volatility = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1).rolling(window=lookback_period).std()
    
    # Compute Intraday Reversal Score
    intraday_return = df['high'] - df['low']
    intraday_reversal_score = -intraday_return
    
    # Adjust for Volume
    volume_adjusted_reversal = intraday_reversal_score * high_low_range * df['volume']
    
    # Combine Momentum, Intraday Reversal, and Volume-Adjusted Range
    combined_signal = (df['high'] - df['low']).diff() * intraday_reversal_score + volume_adjusted_reversal
    
    # Consider Close-to-Open Price Change
    close_to_open_diff = df['close'].shift(1) - df['open']
    combined_signal += close_to_open_diff * volume_adjusted_reversal
    
    # Generate Factor Value
    ten_day_high_low_momentum = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    ten_day_volume_ma = df['volume'].rolling(window=lookback_period).mean()
    factor_value = (ten_day_high_low_momentum * ten_day_volume_ma) / df['close']
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Combine All Factors
    combined_factors = (intraday_return_volatility * 0.3 +
                        factor_value * 0.4 +
                        smoothed_combined_signal * 0.2 +
                        combined_signal * 0.1)
    
    # Final Alpha Factor
    volume_change = df['volume'].diff()
    significant_volume_increase = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    final_alpha_factor = combined_factors * smoothed_combined_signal * significant_volume_increase
    
    return final_alpha_factor
```
