```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Price Momentum
    high_low_momentum = high_low_diff.rolling(window=10).sum()
    
    # Calculate High-Low Range Momentum
    high_low_range_diff = high_low_range - high_low_range.shift(1)
    high_low_range_momentum = high_low_range_diff.rolling(window=5).sum()
    
    # Adjust for Volume
    volume_5_day_avg = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = high_low_range_momentum / volume_5_day_avg
    
    # Combine High-Low Price Momentum and Volume-Adjusted Range Momentum
    combined_momentum = high_low_momentum * volume_adjusted_momentum
    
    # Apply Moving Average Smoothing
    smoothed_momentum = combined_momentum.rolling(window=10).mean()
    
    # Calculate Factor Value
    factor_value = smoothed_momentum / df['close']
    
    return factor_value
```
