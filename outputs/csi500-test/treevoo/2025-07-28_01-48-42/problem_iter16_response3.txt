```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['open'] - df['low']
    intraday_return_volatility = (intraday_returns_high + intraday_returns_low).rolling(window=10).std()
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Compute Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted = df['volume'] * high_low_range
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * np.abs(open_close_spread)
    
    # Combine Volume-Adjusted Momentum and Spread
    combined_signal = adjusted_momentum / volume_adjusted
    
    # Enhance Combined Signal with Smoothing
    enhanced_combined_signal = combined_signal.ewm(span=5).mean()
    
    # Generate Factor Value
    ten_day_high_low_momentum = high_low_range.rolling(window=10).mean()
    ten_day_volume_ma = df['volume'].rolling(window=10).mean()
    factor_value = (ten_day_high_low_momentum * ten_day_volume_ma) / df['close']
    
    # Combine All Factors
    weighted_sum = 0.3 * intraday_return_volatility + 0.4 * factor_value + 0.2 * enhanced_combined_signal + 0.1 * factor_value
    
    # Final Alpha Factor
    final_alpha_factor = weighted_sum * enhanced_combined_signal
    
    # Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike_factor = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    final_alpha_factor = final_alpha_factor * volume_spike_factor
    
    return final_alpha_factor
```
