```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Logarithmic Daily Returns
    df['returns'] = df['close'].apply(lambda x: (x / df.shift(1)['close']).log())
    
    # Momentum Component
    df['21_day_mean_returns'] = df['returns'].rolling(window=21).mean()
    df['60_day_mean_returns'] = df['returns'].rolling(window=60).mean()
    df['momentum'] = df['21_day_mean_returns'] - df['60_day_mean_returns']
    
    # Volatility Component
    df['21_day_sum_abs_returns'] = df['returns'].abs().rolling(window=21).sum()
    df['60_day_sum_abs_returns'] = df['returns'].abs().rolling(window=60).sum()
    df['volatility'] = df['21_day_sum_abs_returns'] - df['60_day_sum_abs_returns']
    
    # Final factor
    df['factor'] = df['momentum'] + df['volatility']
    
    return df['factor'].dropna()
```
