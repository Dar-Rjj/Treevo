```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Compute Momentum (Sum over the last 10 days)
    momentum = price_change.rolling(window=10).sum()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Combine Momentum and Volume-Adjusted Range
    combined_factor = momentum / volume_adjusted_range
    
    return combined_factor
```
