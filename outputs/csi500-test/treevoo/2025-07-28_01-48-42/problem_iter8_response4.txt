```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_returns = pd.concat([intraday_returns_high, intraday_returns_low], axis=1).std(axis=1)
    intraday_return_volatility = intraday_returns.rolling(window=10).std()
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Adjust for Volume
    volume_adjusted_range = high_low_range * df['volume']
    
    # Compute Momentum
    momentum = price_change.rolling(window=10).sum()
    
    # Combine Momentum and Volume-Adjusted Range
    combined_signal = momentum / volume_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    combined_signal_smoothed = combined_signal.ewm(span=5).mean()
    
    # Calculate High-Low Price Difference
    high_low_price_difference = df['high'] - df['low']
    
    # Calculate 10-Day High-Low Price Momentum
    high_low_momentum = high_low_price_difference.rolling(window=10).sum()
    
    # Calculate 10-Day Volume Moving Average
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    
    # Combine 10-Day High-Low Price Momentum and 10-Day Volume Moving Average
    combined_momentum_volume = high_low_momentum * volume_10_day_ma
    
    # Generate Factor Value
    factor_value = combined_momentum_volume / df['close']
    
    # Combine All Factors
    final_factor = (0.3 * intraday_return_volatility + 
                    0.4 * combined_momentum_volume + 
                    0.2 * combined_signal_smoothed + 
                    0.1 * factor_value)
    
    return final_factor
```
