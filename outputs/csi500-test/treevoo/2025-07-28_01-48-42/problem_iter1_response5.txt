```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Invert Intraday Return for Reversal
    intraday_reversal_score = -intraday_return
    
    # Compute Daily Volume Rank
    daily_volume_rank = df['volume'].rolling(window=21).apply(lambda x: (x[-1] > x[:-1]).sum(), raw=True)
    
    # Adjust by Volume Relative Strength
    volume_adjusted_reversal = intraday_reversal_score * (daily_volume_rank / 20)
    
    # Calculate Close-to-Open Difference
    close_to_open_diff = (df['close'] - df['open']) / df['open']
    
    # Integrate into Final Factor
    final_factor = volume_adjusted_reversal * close_to_open_diff
    
    return final_factor
```
