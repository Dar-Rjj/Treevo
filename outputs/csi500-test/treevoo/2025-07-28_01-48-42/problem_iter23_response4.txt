```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal = 1 / intraday_return  # Invert for reversal
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = high_low_range * df['volume'] * df['amount']
    
    # Integrate Momentum and Reversal
    high_low_momentum = high_low_range - high_low_range.shift(1)
    momentum_reversal = high_low_momentum * intraday_reversal
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    combined_signal = momentum_reversal + volume_amount_adjusted_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_change_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    price_vol_momentum = price_change_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open']
    final_factor = combined_signal + price_vol_momentum + close_to_open_diff * df['volume']
    
    # Apply a Smoothing Filter to the Combined Signal
    ema_span = 5
    smoothed_final_factor = final_factor.ewm(span=ema_span, min_periods=ema_span).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_integrated_factor = smoothed_final_factor * daily_volatility
    
    # Apply a Second Smoothing Layer
    sma_window = 10
    final_smoothed_factor = volatility_integrated_factor.rolling(window=sma_window, min_periods=sma_window).mean()
    
    return final_smoothed_factor
```
