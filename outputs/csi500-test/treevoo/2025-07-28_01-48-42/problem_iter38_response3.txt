import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate SMAs
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_ratio'] = df['SMA_50'] / df['SMA_200']

    # Calculate EMAs
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['EMA_ratio'] = df['EMA_12'] / df['EMA_26']

    # Rate of Change (ROC)
    df['ROC_14_close'] = (df['close'].pct_change(periods=14) * 100)
    df['ROC_14_high'] = (df['high'].pct_change(periods=14) * 100)
    df['ROC_14_low'] = (df['low'].pct_change(periods=14) * 100)

    # MACD
    df['MACD_line'] = df['EMA_12'] - df['EMA_26']
    df['Signal_line'] = df['MACD_line'].ewm(span=9, adjust=False).mean()
    df['MACD_histogram'] = df['MACD_line'] - df['Signal_line']

    # DMI
    df['+DM'] = df['high'].diff().apply(lambda x: x if x > 0 else 0)
    df['-DM'] = -df['low'].diff().apply(lambda x: x if x < 0 else 0)
    df['+DI'] = df['+DM'].rolling(window=14).mean() / df[['high', 'low']].diff(axis=1).abs().max(axis=1).rolling(window=14).mean()
    df['-DI'] = df['-DM'].rolling(window=14).mean() / df[['high', 'low']].diff(axis=1).abs().max(axis=1).rolling(window=14).mean()
    df['DX'] = 100 * (df['+DI'] - df['-DI']).abs() / (df['+DI'] + df['-DI'])
    df['ADX'] = df['DX'].ewm(span=14, adjust=False).mean()

    # ATR
    df['TR'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['ATR'] = df['TR'].ewm(span=14, adjust=False).mean()

    # Bollinger Bands
    df['BB_middle'] = df['close'].rolling(window=20).mean()
    df['BB_std'] = df['close'].rolling(window=20).std()
    df['BB_upper'] = df['BB_middle'] + 2 * df['BB_std']
    df['BB_lower'] = df['BB_middle'] - 2 * df['BB_std']

    # OBV
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # CMF
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Composite Indicators
    df['CMI'] = (df['SMA_ratio'] + df['EMA_ratio'] + df['ROC_14_close']) / 3
    df['CTI'] = (df['MACD_histogram'] + df['ADX'] + (df['+DI'] - df['-DI'])) / 3
    df['CVI'] = (df['ATR'] + (df['BB_upper'] - df['BB_lower'])) / 2
    df['CVoI'] = (df['OBV'] + df['CMF']) / 2

    # Novel Indicators
    df['HLR'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    df['HLR'] = df['HLR'].ewm(span=14, adjust=False).mean()
