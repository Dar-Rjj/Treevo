import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (intraday_return + 1e-6)  # Add small epsilon to avoid division by zero
    
    # Adjust for Volume and Amount
    volume_amount_adjusted = df['volume'] * high_low_range * df['amount']
    
    # Integrate Momentum and Reversal
    momentum_and_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    combined_signal = momentum_and_reversal + volume_amount_adjusted
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price Momentum and Volume Delta
    price_volume_integrated = price_momentum * volume_delta
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    close_to_open_diff = df['close'] - df['open']
    final_confirmation = volume_amount_adjusted * close_to_open_diff
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = combined_signal.ewm(span=5).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_integrated = smoothed_signal * daily_volatility
    
    # Apply a Second Smoothing Layer
    second_smoothed_signal = volatility_integrated.rolling(window=10).mean()
    
    # Enhance with Trend Confirmation
    long_term_momentum = df['close'] - df['close'].shift(20)
    trend_confirmed_factor = second_smoothed_signal * long_term_momentum
    
    # Final Smoothing for Robustness
    final_factor = trend_confirmed_factor.rolling(window=15, center=True).mean().fillna(method='bfill')
    
    return final_factor
