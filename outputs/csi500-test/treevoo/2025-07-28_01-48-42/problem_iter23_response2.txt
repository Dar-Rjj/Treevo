```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices, and close and open prices
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate the ratio of the amount to volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Calculate a weighted sum of these features
    df['weighted_sum'] = (0.4 * df['high_low_diff']) + (0.3 * df['close_open_diff']) + (0.3 * df['amount_volume_ratio'])
    
    # Generate the alpha factor by taking the rolling mean of the weighted sum over 10 days
    alpha_factor = df['weighted_sum'].rolling(window=10).mean()
    
    return alpha_factor
```
