import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Identify Short-Term Reversal
    df['Short_Term_Reversal'] = (df['close'] - df['open']) / df['open']
    
    # Detect Long-Term Mean Reversion
    sma_200 = df['close'].rolling(window=200).mean()
    df['Long_Term_Mean_Reversion'] = (df['close'] - sma_200) / sma_200
    
    # Measure Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Amount Flow
    df['Amount_Flow'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Price to Volume Ratio
    df['Price_to_Volume_Ratio'] = (df['close'] - df['close'].shift(1)) / (df['volume'] - df['volume'].shift(1))
    
    # Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP'] = vwap
    
    # Daily Range as a Proportion of Price
    df['Daily_Range_Proportion'] = (df['high'] - df['low']) / df['close']
    
    # Standard Deviation of Daily Returns
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Std_Dev_Daily_Returns'] = daily_returns.rolling(window=20).std()
    
    # Number of Days with Positive Return
    positive_returns = (df['close'] - df['open']) > 0
    df['Positive_Returns_Count'] = positive_returns.rolling(window=20).sum()
    
    # Number of Days with Increasing Volume
    increasing_volume = df['volume'] > df['volume'].shift(1)
    df['Increasing_Volume_Count'] = increasing_volume.rolling(window=20).sum()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['Price_Momentum'] +
        df['Short_Term_Reversal'] +
        df['Long_Term_Mean_Reversion'] +
        df['Volume_Change'] +
        df['Amount_Flow'] +
        df['Price_to_Volume_Ratio'] +
        df['VWAP'] +
        df['Daily_Range_Proportion'] +
        df['Std_Dev_Daily_Returns'] +
        df['Positive_Returns_Count'] +
        df['Increasing_Volume_Count']
    )
    
    return alpha_factor
