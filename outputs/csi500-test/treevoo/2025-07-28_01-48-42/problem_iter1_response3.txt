```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Change Momentum
    price_change_momentum = df['close'].shift(-1) - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Delta
    combined_signal = price_change_momentum * volume_delta
    
    # Apply a Smoothing Filter (Exponential Moving Average with span 5)
    ema_span = 5
    smoothed_signal = combined_signal.ewm(span=ema_span, adjust=False).mean()
    
    return smoothed_signal
```
