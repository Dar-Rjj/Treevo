```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pandas.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pandas.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the price change from the previous day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average price of the day
    df['avg_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume'])
    
    # Calculate the relative strength indicator (RSI) like metric
    gain = df['price_change'].apply(lambda x: max(x, 0))
    loss = df['price_change'].apply(lambda x: -min(x, 0))
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (rsi + df['momentum']) * (df['avg_price'] - df['vwap'])
    
    return df['alpha_factor']
```
