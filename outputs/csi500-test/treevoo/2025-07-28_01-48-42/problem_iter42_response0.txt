```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Calculate Momentum
    lookback_period = 10
    avg_close = df['close'].rolling(window=lookback_period).mean()
    momentum = df['close'] - avg_close
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    mean_spread = spread.rolling(window=lookback_period).mean()
    adjusted_momentum = momentum * (spread.abs() / mean_spread)
    
    # Calculate Volume Volatility
    vol_lookback_period = 30
    volume_volatility = df['volume'].rolling(window=vol_lookback_period).std()
    
    # Adjust Adjusted Momentum by Volume Volatility
    final_adjusted_momentum = adjusted_momentum / volume_volatility
    
    # Weight Final Adjusted Momentum by Volume
    weighted_momentum = final_adjusted_momentum * df['volume']
    
    return weighted_momentum
```
