```python
import pandas as pd

def heuristics_v2(df, lookback_period=10, consistency_lookback=20):
    # Calculate Daily Open-Close Price Spread
    spread = df['open'] - df['close']
    
    # Calculate Momentum
    avg_close = df['close'].rolling(window=lookback_period).mean()
    momentum = df['close'] - avg_close
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum * spread.abs()
    
    # Calculate Daily High-Low Range
    range_high_low = df['high'] - df['low']
    
    # Further Adjust Momentum by High-Low Range
    further_adjusted_momentum = adjusted_momentum / range_high_low
    
    # Weight Final Adjusted Momentum by Volume
    volume_weighted_momentum = further_adjusted_momentum * df['volume']
    
    # Final Alpha Factor Component 1
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    component_1 = (volume_weighted_momentum.rolling(window=lookback_period).sum() / total_volume)
    
    # Calculate Intraday Volatility
    true_range = df[['high', 'low', 'close']].diff(axis=1).max(axis=1)
    atr = true_range.rolling(window=lookback_period).mean()
    intraday_volatility = atr * df['volume']
    
    # Final Alpha Factor Component 2
    weighted_intraday_volatility = intraday_volatility.rolling(window=lookback_period).sum()
    component_2 = weighted_intraday_volatility / total_volume
    
    # Calculate Price Movement Consistency
    ema_close = df['close'].ewm(span=consistency_lookback, adjust=False).mean()
    daily_returns = df['close'].pct_change()
    std_dev_returns = daily_returns.rolling(window=consistency_lookback).std()
    
    # Final Alpha Factor Component 3
    component_3 = std_dev_returns / ema_close
    
    # Combine Final Alpha Factors
    alpha_factor = component_1 + component_2 + component_3
    
    return alpha_factor
```
