import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['open'] - df['close']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Momentum
    lookback_period = 10
    momentum = df['close'] - df['close'].rolling(window=lookback_period).mean()
    
    # Adjust Momentum by Absolute Daily Open-Close Spread
    adjusted_momentum = momentum / open_close_spread.abs()
    
    # Weight Adjusted Momentum by Volume
    volume_weighted_momentum = adjusted_momentum * df['volume']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Momentum with Price Change
    momentum_with_price_change = price_change.rolling(window=lookback_period).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_range = df['volume'] * high_low_range * df['amount']
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume']
    
    # Enhance Price Change Factor
    average_price_change = price_change.rolling(window=lookback_period).mean()
    enhanced_price_change = price_change - average_price_change
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_momentum = momentum / (volume_amount_adjusted_range + 1e-8)  # avoid division by zero
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum + weighted_price_change + enhanced_price_change
    
    return final_alpha_factor
