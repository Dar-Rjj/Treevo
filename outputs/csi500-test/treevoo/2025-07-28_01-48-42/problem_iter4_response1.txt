```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Returns
    intraday_high_return = df['high'] - df['open']
    intraday_low_return = df['low'] - df['open']
    
    # Calculate Intraday Return Volatility
    intraday_volatility = (intraday_high_return - intraday_low_return).rolling(window=10).std()
    
    # Determine Volume Difference
    volume_difference = df['volume'].diff(periods=1)
    
    # Calculate Average Volume Spike
    average_volume_spike = volume_difference.rolling(window=5).sum()
    
    # Calculate Close Price Momentum
    price_change_20d = df['close'].pct_change(periods=20)
    ema_14d = df['close'].ewm(span=14, adjust=False).mean()
    
    # Combine Factors
    factor = (0.5 * intraday_volatility) + (0.3 * average_volume_spike) + (0.2 * (price_change_20d / ema_14d))
    
    return factor
```
