```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return Volatility
    intraday_returns_high = df['high'] - df['open']
    intraday_returns_low = df['low'] - df['open']
    intraday_volatility = (intraday_returns_high - intraday_returns_low).rolling(window=10).std()
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['open']
    intraday_reversal = -intraday_return
    
    # Integrate Momentum and Reversal
    momentum_reversal = high_low_range.diff().multiply(intraday_reversal)
    
    # Adjust for Volume and Amount
    volume_adjusted = high_low_range * df['amount'] * df['volume']
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'].diff()
    integrated_signal = momentum_reversal + volume_adjusted + price_change
    
    # Compute Price Change Momentum
    close_open_diff = df['close'] - df['open'].shift(1)
    integrated_signal += close_open_diff * intraday_reversal
    
    # Generate Factor Value
    factor_value = (high_low_range.diff(periods=10) * df['volume'].rolling(window=10).mean()) / df['close']
    
    # Apply a Smoothing Filter to the Combined Signal
    smoothed_signal = integrated_signal.ewm(span=5).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low'])**2 / 4) + ((df['close'] - df['open'])**2 / 4))
    volatility_adjusted_factor = factor_value * daily_volatility
    
    # Apply a Second Smoothing Layer
    final_smoothed_signal = smoothed_signal.rolling(window=10).mean().ewm(span=5).mean()
    
    # Combine All Factors
    combined_factors = (intraday_volatility * 0.3) + (volatility_adjusted_factor * 0.4) + (final_smoothed_signal * 0.2) + (factor_value * 0.1)
    
    # Final Alpha Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    final_alpha_factor = combined_factors * final_smoothed_signal * volume_spike
    
    return final_alpha_factor
```
