```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Momentum
    lookback_period = 10
    avg_close = df['close'].rolling(window=lookback_period).mean()
    momentum = df['close'] - avg_close
    
    # Adjust Momentum by Open-Close Spread
    adjusted_momentum = momentum * open_close_spread.abs()
    
    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    range_adjusted_momentum = adjusted_momentum / high_low_range
    
    # Weight by Volume
    volume_weighted_momentum = range_adjusted_momentum * df['volume']
    
    return volume_weighted_momentum
```
