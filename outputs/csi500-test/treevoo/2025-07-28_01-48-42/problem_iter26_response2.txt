```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) based on the close price
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the average true range (ATR) for the last 14 days
    tr = pd.DataFrame(index=df.index)
    tr['h-l'] = df['high'] - df['low']
    tr['h-pc'] = abs(df['high'] - df['close'].shift(1))
    tr['l-pc'] = abs(df['low'] - df['close'].shift(1))
    tr['tr'] = tr.max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()
    
    # Calculate a momentum factor
    momentum = df['close'] - df['close'].shift(14)
    
    # Combine the RSI, ATR, and Momentum to create the alpha factor
    alpha_factor = (rsi + atr + momentum) / 3
    
    return alpha_factor
```
