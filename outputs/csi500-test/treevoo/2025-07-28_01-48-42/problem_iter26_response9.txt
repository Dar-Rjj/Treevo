import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / (1 + np.abs(intraday_return))
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    volume_amount_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Integrate Momentum and Reversal
    high_low_range_diff = high_low_range.diff()
    momentum_reversal_combined = high_low_range_diff * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    price_change = df['close'].diff()
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    combined_signal = momentum_reversal_combined + volume_amount_adjusted_range + weighted_price_change
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust 5-Day Momentum by Open-Close Spread
    open_close_spread = (df['open'] - df['close']).abs()
    adjusted_five_day_momentum = five_day_momentum * open_close_spread
    
    # Enhance Combined Signal with Smoothing
    smoothed_combined_signal = combined_signal.ewm(span=5, adjust=False).mean()
    
    # Confirm with Close-to-Open Movement
    close_to_open_diff = df['close'].shift(-1) - df['open']
    integrated_signal = smoothed_combined_signal * close_to_open_diff
    
    # Calculate Smoothed Price Momentum
    daily_close_price_return = df['close'].pct_change()
    smoothed_price_momentum = daily_close_price_return.ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    volume_change = df['volume'].diff()
    volume_spike_factor = (volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Calculate Close-to-Open Price Change
    close_to_open_price_change = df['close'] - df['open'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (
        integrated_signal * smoothed_price_momentum * volume_spike_factor + 
        close_to_open_price_change
    )
    
    return final_alpha_factor
