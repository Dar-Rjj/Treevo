```python
import pandas as pd

def heuristics_v2(df):
    # Compute Price Change
    df['price_change'] = df['close'] - df['open'].shift(1)

    # Compute Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(1)

    # Multiply Price Change by Volume Delta
    df['price_volume_momentum'] = df['price_change'] * df['volume_delta']

    # Smoothed Price-Volume Momentum
    df['smoothed_price_volume_momentum'] = df['price_volume_momentum'].ewm(span=5, adjust=False).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Intraday Returns
    df['intraday_return_high'] = df['high'] - df['open']
    df['intraday_return_low'] = df['open'] - df['low']

    # Compute Standard Deviation over past 10 days
    df['intraday_return_volatility'] = (df[['intraday_return_high', 'intraday_return_low']]
                                        .std(axis=1)
                                        .rolling(window=10, min_periods=1)
                                        .mean())

    # Adjust for Volume
    df['volume_adjusted_range'] = df['volume'] * df['high_low_range']

    # Compute Momentum
    df['momentum'] = df['price_change'].rolling(window=10, min_periods=1).sum()

    # Combine Momentum and Volume-Adjusted Range
    df['combined_momentum_range'] = df['momentum'] / df['volume_adjusted_range']

    # Calculate Average Volume Spike
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['average_volume_spike'] = df['volume_difference'].rolling(window=5, min_periods=1).sum()

    # Combine All Factors
    df['alpha_factor'] = (0.3 * df['intraday_return_volatility'] +
                          0.2 * df['average_volume_spike'] +
                          0.4 * df['combined_momentum_range'])

    return df['alpha_factor']
```
