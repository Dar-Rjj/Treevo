import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Reversal Score
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_reversal_score = 1 / intraday_return
    
    # Adjust for Volume and Amount
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    volume_adjusted_high_low_range = high_low_range * (avg_volume_5d / avg_amount_5d)
    
    # Integrate Momentum and Reversal
    momentum_and_reversal = (high_low_range - high_low_range.shift(1)) * intraday_reversal_score
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    integrated_factor = momentum_and_reversal + volume_adjusted_high_low_range
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Compute Price Change Momentum
    price_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Delta
    volume_delta = df['volume'] - df['volume'].shift(1)
    
    # Confirm with Volume, Amount, and Close-to-Open Movement
    daily_volume_rank = df['volume'].rank(pct=True)
    daily_amount_rank = df['amount'].rank(pct=True)
    close_to_open_diff = df['close'] - df['open']
    
    # Multiply by Combined High-Low and Reversal
    combined_signal = (momentum_and_reversal * close_to_open_diff) * (daily_volume_rank * daily_amount_rank)
    
    # Apply a Smoothing Filter to the Combined Signal
    ema_5d = combined_signal.ewm(span=5, min_periods=5).mean()
    
    # Introduce a New Volatility Component
    daily_volatility = np.sqrt(((df['high'] - df['low']) ** 2) / 4 + ((df['close'] - df['open']) ** 2) / 4)
    volatility_adjusted_signal = ema_5d * daily_volatility
    
    # Apply a Second Smoothing Layer
    sma_10d = volatility_adjusted_signal.rolling(window=10, min_periods=10).mean()
    
    # Introduce a New Liquidity Component
    daily_liquidity = (df['amount'] / df['volume']) * 1000
    final_factor = sma_10d * daily_liquidity
    
    return final_factor
