```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Daily Open-Close Price Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff().fillna(0)
    
    # Compute 5-Day Momentum
    five_day_momentum = price_change.rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_adjusted_range = high_low_range * df['volume']
    amount_adjusted_range = high_low_range * np.sqrt(df['amount'])
    combined_adjusted_range = volume_adjusted_range + amount_adjusted_range
    
    # Adjust 5-Day Momentum by Open-Close Spread
    adjusted_momentum = five_day_momentum * abs(open_close_spread)
    
    # Combine Momentum and Volume-Amount-Adjusted Range
    combined_signal = adjusted_momentum / combined_adjusted_range
    
    # Enhance Combined Signal with Smoothing
    enhanced_signal = combined_signal.ewm(span=12, adjust=False).mean()
    
    # Calculate Weighted Price Change
    weighted_price_change = price_change * df['volume'] * np.sqrt(df['amount'])
    
    # Combine Adjusted Range with Weighted Price Change
    combined_signal_with_weighted_change = enhanced_signal + weighted_price_change
    
    # Compute Intraday Reversal Score
    intraday_return = df['high'] - df['low']
    intraday_reversal = -intraday_return
    
    # Consider Close-to-Open Price Change
    close_open_diff = df['close'] - df['open']
    
    # Integrate Volume-Adjusted Reversal with C-O Diff
    integrated_reversal = intraday_reversal * close_open_diff
    
    # Integrate Volume-Amount-Adjusted Range with Weighted Price Change
    final_combined_signal = combined_signal_with_weighted_change + integrated_reversal
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change().ewm(span=12, adjust=False).mean()
    
    # Calculate Volume Spike Factor
    day_to_day_volume_change = df['volume'].diff().fillna(0)
    std_dev_volume = df['volume'].rolling(window=20).std()
    volume_spike_factor = (day_to_day_volume_change > 2 * std_dev_volume).astype(int)
    
    # Final Alpha Factor
    alpha_factor = final_combined_signal * price_momentum * volume_spike_factor + close_open_diff
    
    return alpha_factor
```
