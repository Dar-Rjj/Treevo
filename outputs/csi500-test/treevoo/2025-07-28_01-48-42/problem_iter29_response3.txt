```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the body range of the candle
    body_range = df['close'] - df['open']
    
    # Calculate the upper and lower shadow
    upper_shadow = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow = df[['open', 'close']].min(axis=1) - df['low']
    
    # Calculate the total range
    total_range = df['high'] - df['low']
    
    # Calculate the body to total range ratio
    body_to_total_range = body_range / total_range
    
    # Calculate the shadow dominance factor
    shadow_dominance = (upper_shadow + lower_shadow) / total_range
    
    # Calculate the volume to amount ratio
    volume_to_amount = df['volume'] / df['amount']
    
    # Combine the factors
    factor_values = (body_to_total_range * shadow_dominance * volume_to_amount)
    
    return pd.Series(factor_values, index=df.index, name='heuristics_v2')
```
