```python
def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average of (high - low) over the last 5 days
    df['avg_hl_5d'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the ratio of (close - open) to (high - low)
    df['c_o_to_h_l_ratio'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the factor: (daily return * c_o_to_h_l_ratio * avg_hl_5d) / vwap
    df['factor'] = (df['daily_return'] * df['c_o_to_h_l_ratio'] * df['avg_hl_5d']) / df['vwap']
    
    # Return the factor as a pandas Series
    return df['factor'].dropna()
```
