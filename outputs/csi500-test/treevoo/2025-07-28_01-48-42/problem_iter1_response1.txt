```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Smoothed Momentum using EMA over 12 days
    df['smoothed_momentum'] = df['daily_return'].ewm(span=12, adjust=False).mean()
    
    # Calculate Day-to-Day Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Indicate Significant Volume Increase
    std_vol = df['volume_change'].rolling(window=20).std()
    df['volume_spike'] = (df['volume_change'] > 2 * std_vol).astype(int)
    
    # Combine Factors: Multiply Smoothed Momentum by Volume Spike Factor
    df['combined_factor'] = df['smoothed_momentum'] * df['volume_spike']
    
    return df['combined_factor']
```
