```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5):
    # Compute Close Price Difference
    close_diff = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    smoothed_momentum = close_diff.rolling(window=m).mean()
    
    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=n).mean()
    daily_volume_ratio = df['volume'] / mean_volume
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Logarithmic Return
    log_return = np.log(intraday_return + 1e-8)  # Small constant to avoid log(0)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    weighted_intraday_return = intraday_return * (df['volume'] / df['volume'].sum())
    weighted_intraday_range = intraday_range * (df['volume'] / df['volume'].sum())
    combined_factor = weighted_intraday_return + weighted_intraday_range
    
    # Integrate Volume Adjusted Momentum
    final_combined_factor = combined_factor + volume_adjusted_momentum
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = np.where(short_term_momentum > long_term_momentum, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_trend_direction = trend_direction * volume_change
    
    # Calculate Relative Strength
    relative_strength = (short_term_momentum - long_term_momentum) / (df[['high', 'low', 'close']].mul(df['volume'], axis=0).sum(axis=1) / df['volume'])
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    
    # Combine Factors
    price_range_ratio = (df['high'] - df['low']) / df['close']
    combined_factors = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Synthesize Final Alpha Factor
    alpha_factor = relative_strength + combined_factors + final_combined_factor + final_adjusted_cumulative_return
    
    return alpha_factor
```
