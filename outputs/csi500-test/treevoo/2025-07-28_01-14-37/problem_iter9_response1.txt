```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Calculate Volume Impact
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_impact = df['volume'] / avg_volume_5d
    
    # Combine Factors
    combined_sum = intraday_movement + close_to_open_return
    weighted_sum = combined_sum * price_range_ratio
    final_factor = weighted_sum * volume_impact
    
    return final_factor
```
