```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change and High-to-Low Range
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['high_low_range'] = df['high'] - df['low']
    
    # Separate Positive and Negative Gains
    df['positive_gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_gain'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    
    # Sum of Positive and Negative Gains over a period (e.g., 14 days)
    positive_sum = df['positive_gain'].rolling(window=14).sum()
    negative_sum = df['negative_gain'].rolling(window=14).sum()
    
    # Adjust for Volume and Amount
    weighted_positive_gain = (df['positive_gain'] * df['volume']) + (df['positive_gain'] * df['amount'])
    weighted_negative_gain = (df['negative_gain'] * df['volume']) + (df['negative_gain'] * df['amount'])
    
    # Calculate RSI
    rs = positive_sum / negative_sum
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Momentum
    weighted_momentum = df['price_change'] / vwp
    
    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / df['high_low_range']
    
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) - (short_term_ma < long_term_ma).astype(int)
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = trend_direction * volume_change
    
    # Combine Adjusted Volatility Momentum and Adjusted Momentum
    combined_momentum = adjusted_weighted_momentum + adjusted_momentum
    
    # Final Factor
    factor = rsi + combined_momentum
    
    return factor
```
