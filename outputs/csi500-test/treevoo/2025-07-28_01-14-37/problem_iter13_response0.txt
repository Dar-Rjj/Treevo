```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    weighted_close_change = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_close_change / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    
    # Final Factor
    final_factor = (volatility_adjusted_momentum * volume_change * trend_direction)
    
    return final_factor
```
