```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    momentum = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_momentum = momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Weighted Momentum
    log_returns = np.log(df['close'] / df['close'].shift(5))
    weighted_log_returns = log_returns.ewm(span=5).mean()
    adjusted_weighted_momentum = weighted_log_returns / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    ema_difference = short_term_ma - long_term_ma
    relative_strength = (adjusted_weighted_momentum - ema_difference) / weighted_momentum
    combined_factor = relative_strength + final_adjusted_cumulative_return
    integrated_factor = combined_factor / ema_difference
    
    # Enhance with Intraday Price Behavior
    intraday_price_range = df['high'] - df['low']
    intraday_price_mean = (df['high'] + df['low'] + df['close']) / 3
    intraday_price_variability = intraday_price_range / intraday_price_mean
    enhanced_momentum = integrated_factor * intraday_price_variability
    
    # Incorporate Open Price Momentum
    open_price_change = df['open'].diff()
    open_price_momentum = open_price_change / (df['open'] * df['volume'])
    final_momentum = enhanced_momentum + open_price_momentum
    
    # Final Factor
    final_factor = (final_momentum * volume_change * trend_direction) + (final_adjusted_cumulative_return * trend_direction)
    
    return final_factor
```
