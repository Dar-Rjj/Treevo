```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, short_term=5, long_term=20):
    # Compute Close Price Difference
    close_diff = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    smoothed_momentum = close_diff.rolling(window=m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_momentum = combined_factor + volume_adjusted_momentum
    
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=short_term).mean()
    long_term_ma = df['close'].rolling(window=long_term).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_trend = trend_direction * volume_change
    
    # Calculate Relative Strength
    price_momentum = short_term_ma - long_term_ma
    weighted_price = (df['high'] + df['low'] + df['close']) * df['volume']
    volume_weighted_price_movement = weighted_price.rolling(window=short_term).sum() / df['volume'].rolling(window=short_term).sum()
    relative_strength = (price_momentum - long_term_ma) / volume_weighted_price_movement
    
    # Momentum Signal
    momentum_signal = (integrated_momentum > 0).astype(int) * 2 - 1
    
    # Final Alpha Factor
    alpha_factor = momentum_signal * relative_strength * confirmed_trend
    
    return alpha_factor
```
