```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Close Price Change
    df['close_change'] = df['close'].diff()
    
    # Compute Weighted Momentum
    df['weighted_momentum'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['weighted_momentum_factor'] = df['close_change'] / df['weighted_momentum']
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['weighted_momentum_factor'] / df['high_low_range']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'].diff()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a 5-day window
    df['cumulative_return'] = df['daily_return'].rolling(window=5).sum()
    
    # Adjust by Price Volatility
    df['price_volatility'] = df['close'].rolling(window=5).std()
    df['adjusted_cumulative_return'] = df['cumulative_return'] / df['price_volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['volatility_adjusted_momentum'] - df['weighted_momentum'].ewm(span=5, adjust=False).mean()
    df['relative_strength_adjusted'] = df['relative_strength'] / (df['close'] * df['volume'])
    
    # Enhance with Intraday Price Behavior
    df['intraday_price_variability'] = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)
    df['enhanced_volatility_adjusted_momentum'] = df['volatility_adjusted_momentum'] * df['intraday_price_variability']
    
    # Incorporate Open Price Momentum
    df['open_price_change'] = df['open'].diff()
    df['open_price_momentum'] = df['open_price_change'] / (df['open'] * df['volume'])
    df['final_adjusted_momentum'] = df['enhanced_volatility_adjusted_momentum'] + df['open_price_momentum']
    
    # Final Factor
    df['alpha_factor'] = (df['final_adjusted_momentum'] * df['volume_change'] * df['trend_direction']) + (df['adjusted_cumulative_return'] * df['trend_direction'])
    
    return df['alpha_factor']
```
