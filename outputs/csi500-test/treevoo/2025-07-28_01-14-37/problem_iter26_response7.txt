```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price momentum by comparing the current close to the previous close
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average transaction size (amount/volume) and interpret it as buying power
    df['buying_power'] = df['amount'] / df['volume']
    
    # Calculate the price volatility using the range (high - low)
    df['price_volatility'] = df['high'] - df['low']
    
    # Construct a factor that combines price momentum, buying power, and inverse of price volatility
    df['factor'] = (df['price_momentum'] * df['buying_power']) / df['price_volatility']
    
    return df['factor']
```
