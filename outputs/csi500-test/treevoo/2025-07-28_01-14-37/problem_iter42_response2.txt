import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Moving Average Crossover
    df['SMA_Crossover'] = df['SMA_5'] - df['SMA_10']
    
    # Identify Price Trends
    df['Trend_Direction'] = np.where(df['close'] > df['open'], 1, -1)
    df['Trend_Strength'] = df['close'] - df['open']
    
    # Analyze Volatility
    df['High_Low_Range'] = df['high'] - df['low']
    df['True_Range'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Evaluate Volume Changes
    df['Volume_Delta'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulation/Distribution Line (A/D Line)
    df['Money_Flow_Multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['Money_Flow_Volume'] = df['Money_Flow_Multiplier'] * df['volume']
    df['A/D_Line'] = df['Money_Flow_Volume'].cumsum()
    
    # On-Balance Volume (OBV)
    df['OBV'] = np.where(df['close'] > df['close'].shift(1), df['volume'], 
                        np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    
    # Chaikin Oscillator
    df['3d_EMA_A/D'] = df['A/D_Line'].ewm(span=3, adjust=False).mean()
    df['10d_EMA_A/D'] = df['A/D_Line'].ewm(span=10, adjust=False).mean()
    df['Chaikin_Oscillator'] = df['3d_EMA_A/D'] - df['10d_EMA_A/D']
    
    # Relative Strength Index (RSI)
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Up_Move'] = np.where(df['Price_Change'] > 0, df['Price_Change'], 0)
    df['Down_Move'] = np.where(df['Price_Change'] < 0, -df['Price_Change'], 0)
    df['Average_Up'] = df['Up_Move'].rolling(window=14).mean()
    df['Average_Down'] = df['Down_Move'].rolling(window=14).mean()
    df['RSI'] = 100 - (100 / (1 + (df['Average_Up'] / df['Average_Down'])))
    
    # Stochastic Oscillator
    df['%K'] = (100 * (df['close'] - df['low'].rolling(window=14).min()) / 
                (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()))
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Rate of Change (ROC)
    df['ROC'] = 100 * (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Cum_Vol_Price'] = (df['Typical_Price'] * df['volume']).cumsum()
