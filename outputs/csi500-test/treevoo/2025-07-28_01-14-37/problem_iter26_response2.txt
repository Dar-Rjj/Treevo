```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=cumulative_window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=cumulative_window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Adjust for Skewness and Kurtosis
    skewness = df['combined_intraday_factor'].skew()
    kurtosis = df['combined_intraday_factor'].kurtosis()
    df['rescaled_combined_intraday_factor'] = (df['combined_intraday_factor'] - df['combined_intraday_factor'].mean()) / df['combined_intraday_factor'].std()
    df['rescaled_combined_intraday_factor'] = df['rescaled_combined_intraday_factor'] / (1 + skewness) / (1 + kurtosis)
    
    # Calculate Price Momentum
    short_term_window = 5
    long_term_window = 20
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_trend_direction'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_ma'] - df['long_term_ma']
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['volume_weighted_price_movement'] = df['weighted_price'].rolling(window=short_term_window).sum() / df['volume'].rolling(window=short_term_window).sum()
    df['relative_strength_adjusted'] = df['relative_strength'] / df['volume_weighted_price_movement']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength_adjusted'] + df['adjusted_cumulative_return']
    df['final_integrated_factor'] = df['integrated_factor'] + df['rescaled_combined_intraday_factor']
    final_skewness = df['final_integrated_factor'].skew()
    final_kurtosis = df['final_integrated_factor'].kurtosis()
    df['final_rescaled_integrated_factor'] = (df['final_integrated_factor'] - df['final_integrated_factor'].mean()) / df['final_integrated_factor'].std()
    df['final_rescaled_integrated_factor'] = df['final_rescaled_integrated_factor'] / (1 + final_skewness) / (1 + final_kurtosis)
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = df['intraday_movement'] + df['close_to_open_return']
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    df['weighted_combined_factors'] = df['combined_factors'] * df['price_range_ratio']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['final_rescaled_integrated_factor'] + df['weighted_combined_factors']
    
    return df['final_alpha_factor']
```
