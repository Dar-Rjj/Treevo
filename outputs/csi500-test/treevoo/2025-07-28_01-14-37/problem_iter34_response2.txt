```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_diff = short_term_ma - long_term_ma
    trend_direction = (trend_diff > 0).astype(int) * 2 - 1  # +1 if positive, -1 otherwise
    trend_direction = trend_direction * (trend_direction.shift(1) == trend_direction).astype(int)  # +1 if same, 0 if mixed
    
    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    trend_volume_adjusted = trend_direction * volume_change
    trend_volume_adjusted = trend_volume_adjusted.where(volume_change.abs() > 1000, 0)  # Exclude small changes
    
    # Calculate Relative Strength
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    volume_weighted_price = (weighted_price * df['volume']).sum() / df['volume'].sum()
    relative_strength = (short_term_ma - long_term_ma) / volume_weighted_price
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4 * np.sqrt(df['volume'])
    weighted_momentum = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_volatility = weighted_momentum / high_low_range
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume_5d
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = (momentum - df['close'].rolling(window=50).mean()) + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (high_low_range / (df['high'] - df['low']))
    
    # Confirm with Volume
    adjusted_trend = trend_volume_adjusted * (df['volume'] - df['volume'].shift(1))
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_volatility + adjusted_cumulative_return + combined_factors) * relative_strength * adjusted_trend
    
    return final_alpha_factor
```
