```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_sum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_direction = np.where(ma_5 > ma_20, 1, -1)
    
    # Calculate Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_factor = relative_strength + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    combined_factors_weighted = combined_factors * high_to_low_range
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    volume_weighted_price_movement = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    relative_strength = (momentum - df['close'].rolling(window=50).mean()) / volume_weighted_price_movement
    
    # Calculate Intraday Reversal Indicator
    intraday_reversal = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])
    
    # Integrate Intraday Reversal Indicator
    intraday_reversal_adjusted = intraday_reversal * intraday_volatility
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_weighted_momentum + integrated_factor + combined_factors_weighted) * relative_strength + intraday_reversal_adjusted
    
    return final_alpha_factor
```
