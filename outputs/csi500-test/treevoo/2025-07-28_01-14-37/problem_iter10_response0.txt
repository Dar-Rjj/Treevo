```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Logarithmic Return
    log_return = np.log(df['close'] / df['open'])
    
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    weighted_intraday_return = intraday_return * df['volume']
    weighted_intraday_range = intraday_range * df['volume']
    
    combined_factor = (weighted_intraday_return + weighted_intraday_range) / (df['volume'] + 1e-6)
    
    # Adjust for Skewness
    skewness = skew(combined_factor, nan_policy='omit')
    rescaled_combined_factor_skew = combined_factor - skewness * (combined_factor - combined_factor.mean())
    
    # Adjust for Kurtosis
    kurt = kurtosis(rescaled_combined_factor_skew, nan_policy='omit')
    rescaled_combined_factor_kurt = rescaled_combined_factor_skew - kurt * (rescaled_combined_factor_skew - rescaled_combined_factor_skew.mean())**3
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * np.sqrt(df['volume'])
    volume_weighted_price_movement = weighted_price.sum() / df['volume'].sum()
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Exponential Moving Average (EMA)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate Final Alpha Factor
    alpha_factor = momentum - ema_50 + volume_weighted_price_movement
    alpha_factor = 0.5 * rescaled_combined_factor_kurt + 0.5 * alpha_factor
    
    # Final Rescaling
    final_skew = skew(alpha_factor, nan_policy='omit')
    final_alpha_factor_skew = alpha_factor - final_skew * (alpha_factor - alpha_factor.mean())
    
    final_kurt = kurtosis(final_alpha_factor_skew, nan_policy='omit')
    final_alpha_factor_kurt = final_alpha_factor_skew - final_kurt * (final_alpha_factor_skew - final_alpha_factor_skew.mean())**3
    
    return final_alpha_factor_kurt
```
