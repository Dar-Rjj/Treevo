import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_threshold = 0.1 * df['volume']
    adjusted_trend = np.where(volume_change > volume_threshold, trend_direction * volume_change, trend_direction)

    # Validate Trend with Price Action
    price_range = df['high'] - df['low']
    within_range = (price_range < (short_term_ma - long_term_ma))
    adjusted_trend = np.where(within_range, adjusted_trend * 1.5, adjusted_trend * 0.5)

    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume

    # Adjust by Price Volatility
    std_dev_close = df['close'].rolling(window=5).std()
    adjusted_cumulative_return = (adjusted_cumulative_return / std_dev_close) + 1

    # Calculate Momentum
    lookback_period = 5
    log_returns = np.log(df['close'] / df['close'].shift(lookback_period))
    momentum = log_returns.rolling(window=lookback_period).sum()

    # Adjust by Intraday Volatility and Volume
    weighted_momentum = momentum * df['volume'] / intraday_volatility

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_factor = relative_strength + adjusted_cumulative_return

    # Calculate Intraday Movement and Close-to-Open Return
    intraday_movement = df['high'] - df['low']
    close_to_open_return = df['open'] / df['close'].shift(1) - 1

    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (price_range / df['close'])

    # Introduce New Factor: Intraday Momentum
    intraday_log_return = np.log(df['high'] / df['low'])
    intraday_momentum = intraday_log_return * df['volume']

    # Integrate Intraday Momentum with Other Factors
    final_alpha_factor = integrated_factor + combined_factors + intraday_momentum

    return final_alpha_factor
