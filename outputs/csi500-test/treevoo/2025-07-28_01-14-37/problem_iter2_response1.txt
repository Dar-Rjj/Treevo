import pandas as pd

def heuristics_v2(df, window=5, momentum_window=10, sma_window=50):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Cumulative Return over the specified window
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Adjust by Price Volatility
    df['price_volatility'] = df['close'].rolling(window=window).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['price_volatility']) + 1
    
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(momentum_window)
    
    # Calculate 50-day Simple Moving Average (SMA)
    df['sma_50'] = df['close'].rolling(window=sma_window).mean()
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_volume = df['volume'].rolling(window=window).sum()
    df['volume_weighted_price_movement'] = df['weighted_price'].rolling(window=window).sum() / total_volume
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = (df['momentum'] - df['sma_50']) / df['volume_weighted_price_movement']
    df['alpha_factor'] = df['relative_strength'] + df['final_adjusted_cumulative_return']
    
    return df['alpha_factor'].dropna()
