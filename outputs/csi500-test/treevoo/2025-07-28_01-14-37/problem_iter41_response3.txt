```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Subtract today's SMA from yesterday's SMA to identify momentum shifts
    df['Momentum_Shift'] = df['SMA_5'] - df['SMA_5'].shift(1)
    
    # Calculate daily volume change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the ratio of amount to volume
    df['Amount_Volume_Ratio'] = df['amount'] / df['volume']
    
    # Determine Trend Direction
    df['Trend_Direction'] = (df['SMA_5'] - df['SMA_20']).apply(lambda x: 1 if x > 0 else -1)
    df['Trend_Confirmation'] = (df['Trend_Direction'] == df['Trend_Direction'].shift(1)).astype(int)
    
    # Enhance with Volume Confirmation
    threshold = df['Volume_Change'].quantile(0.5)
    df['Volume_Adjusted_Trend'] = df['Trend_Confirmation'] * (df['Volume_Change'] > threshold).astype(int)
    
    # Calculate Relative Strength
    df['Volume_Weighted_Price'] = (df['high'] + df['low'] + df['close']) * df['volume'] / df['volume'].sum()
    df['Relative_Strength'] = (df['SMA_5'] - df['SMA_20']) / df['Volume_Weighted_Price']
    
    # Identify Volatility Patterns
    df['Intraday_Volatility'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Explore Price Oscillators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['Price_Oscillator'] = df['SMA_50'] - df['SMA_200']
    
    # Construct Reversal Signals
    df['Days_Since_Highest_High'] = (df['high'].cummax() == df['high']).astype(int).groupby((df['high'].cummax() != df['high']).cumsum()).cumcount()
    df['Days_Since_Lowest_Low'] = (df['low'].cummin() == df['low']).astype(int).groupby((df['low'].cummin() != df['low']).cumsum()).cumcount()
    
    # Evaluate Price Action Relative to Moving Averages
    df['Distance_to_Moving_Average'] = df['close'] - df['SMA_20']
    
    # Calculate Intraday Movement
    df['Intraday_Movement'] = df['high'] - df['low']
    
    # Combine Factors
    df['Combined_Factor'] = (df['Intraday_Movement'] + df['close'] - df['open']) * (df['high'] - df['low']) / df['close']
    
    # Confirm with Volume
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Trend'] * df['Volume_Change']
    
    # Calculate Intraday Reversal Indicator
    df['Intraday_Reversal'] = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])
    
    # Integrate Intraday Reversal Indicator
    df['Integrated_Reversal'] = df['Intraday_Reversal'] * df['Intraday_Volatility']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Momentum'] + df['Relative_Strength'] + df['Combined_Factor'] + df['Integrated_Reversal'])
    
    return df['Final_Alpha_Factor']
```
