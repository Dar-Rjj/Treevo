```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Compute Close Price Difference
    close_diff = df['close'].diff(n)
    
    # Smoothing the Momentum
    smooth_momentum = close_diff.rolling(window=m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = smooth_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_momentum = combined_factor + volume_adjusted_momentum
    
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_confirmed_trend = trend_direction * volume_change
    
    # Calculate Relative Strength
    relative_strength = short_term_ma - long_term_ma
    
    # Momentum Signal
    momentum_signal = integrated_momentum + volume_confirmed_trend + relative_strength
    
    return momentum_signal
```
