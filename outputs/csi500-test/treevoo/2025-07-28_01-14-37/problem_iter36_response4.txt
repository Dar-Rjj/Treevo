```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, lookback_period=20, short_term_ma=5, long_term_ma=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['volume_adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']) / window

    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()

    # Calculate Short-Term and Long-Term Moving Averages
    df['short_term_ma'] = df['close'].rolling(window=short_term_ma).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_ma).mean()

    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend_direction'] = df['trend_direction'] * df['volume_change']

    # Adjust by Intraday Volatility and Volume
    df['weighted_price_momentum'] = df['price_momentum'] * df['volume'] / df['intraday_volatility']

    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - df['long_term_ma']) / df['volume_adjusted_cumulative_return']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['volume_adjusted_cumulative_return'] + df['combined_intraday_factor']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['adjusted_trend_direction'] + df['weighted_price_momentum']

    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) / df['open']

    # Incorporate Intraday Reversal into Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] + df['intraday_reversal']

    return df['final_alpha_factor']
```
