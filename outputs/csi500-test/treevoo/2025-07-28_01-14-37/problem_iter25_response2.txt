```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = df['close'].pct_change() / (weighted_price * df['volume'])
    
    # Adjust for Volatility
    adjusted_volatility_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    average_daily_range = high_to_low_range.rolling(window=5).mean()
    confirmed_momentum = adjusted_volatility_momentum * volume_change * trend_direction * average_daily_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility).apply(lambda x: max(x, 0)) + 1
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = (ema_short - ema_long) / (weighted_price * df['volume'])
    combined_factor = relative_strength + final_adjusted_cumulative_return
    combined_factor = combined_factor / (ema_short - ema_long)
    
    # Incorporate Intraday Price Oscillation
    intraday_oscillation = ((df['high'] - df['close']) / (df['high'] - df['low'])) + ((df['close'] - df['low']) / (df['high'] - df['low']))
    average_intraday_oscillation = intraday_oscillation.rolling(window=5).mean()
    
    # Final Factor
    final_factor = (confirmed_momentum + (final_adjusted_cumulative_return * trend_direction)) * average_intraday_oscillation
    
    return final_factor
```
