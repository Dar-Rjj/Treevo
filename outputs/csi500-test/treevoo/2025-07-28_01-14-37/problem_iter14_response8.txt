```python
def heuristics_v2(data):
    # Calculate the percentage change in close price
    data['close_change'] = data['close'].pct_change()
    
    # Calculate the percentage change in volume
    data['volume_change'] = data['volume'].pct_change()
    
    # Calculate the high-low spread
    data['high_low_spread'] = (data['high'] - data['low']) / data['close']
    
    # Calculate the open-close spread
    data['open_close_spread'] = (data['close'] - data['open']) / data['close']
    
    # Calculate the amount to volume ratio
    data['amount_volume_ratio'] = data['amount'] / data['volume']
    
    # Generate the alpha factor
    data['alpha_factor'] = (data['close_change'] * data['volume_change'] 
                            + data['high_low_spread'] 
                            + data['open_close_spread'] 
                            + data['amount_volume_ratio'])
    
    return data['alpha_factor'].dropna()
```
