import pandas as pd

def heuristics_v2(df, window=5, lookback_period=10):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_range']) / df['average_volume']
    
    # Calculate Price Momentum
    df['log_return'] = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Calculate Short-Term Moving Average (e.g., 5-day)
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average (e.g., 20-day)
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'].diff()
    df['adjusted_trend'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - df['long_term_ma']) / df['close']
    df['volume_weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['total_volume'] = df['volume'].rolling(window=window).sum()
    df['volume_weighted_price_movement'] = df['volume_weighted_price'].rolling(window=window).sum() / df['total_volume']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_alpha'] = df['relative_strength'] + df['adjusted_cumulative_return']
    df['final_combined_factor'] = (df['combined_intraday_factor'] + df['volume_weighted_price_movement']) / 2
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) / df['open']
    
    # Incorporate Intraday Reversal into Final Alpha Factor
    df['final_alpha_factor'] = (df['integrated_alpha'] + df['final_combined_factor'] + df['intraday_reversal']) / 3
    
    return df['final_alpha_factor'].dropna()
