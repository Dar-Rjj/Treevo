```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Adjust for Skewness and Kurtosis
    skewness = df['combined_intraday'].skew()
    kurtosis = df['combined_intraday'].kurt()
    df['rescaled_combined_intraday'] = (df['combined_intraday'] - df['combined_intraday'].mean()) / df['combined_intraday'].std()
    df['final_intraday_factor'] = (df['rescaled_combined_intraday'] - skewness) / kurtosis
    
    # Calculate Price Momentum
    short_window = 5
    long_window = 20
    df['short_moving_avg'] = df['close'].rolling(window=short_window).mean()
    df['long_moving_avg'] = df['close'].rolling(window=long_window).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_moving_avg'] > df['long_moving_avg'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend_direction'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_moving_avg'] - df['long_moving_avg']) / df['long_moving_avg']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return']
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    total_weighted_price = df['weighted_price'].rolling(window=window).sum()
    total_volume = df['volume'].rolling(window=window).sum()
    df['volume_weighted_price_movement'] = total_weighted_price / total_volume
    
    # Combine Factors
    df['final_rescaling'] = (df['integrated_factor'] + df['volume_weighted_price_movement']) / 2
    df['final_alpha_factor'] = df['final_rescaling'] + df['final_intraday_factor']
    
    return df['final_alpha_factor']
```
