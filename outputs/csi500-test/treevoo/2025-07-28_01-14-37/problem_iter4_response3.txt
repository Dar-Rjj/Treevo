```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20):
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    intraday_volatility = df['high_low_range'].rolling(window=n).mean()
    
    # Calculate Momentum
    log_returns = np.log(df['close']) - np.log(df['close'].shift(m))
    sum_log_returns = log_returns.rolling(window=m).sum()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = sum_log_returns / intraday_volatility
    
    return adjusted_momentum
```
