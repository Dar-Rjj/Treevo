```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) * df['volume']
    total_volume_weighted_price = weighted_price.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Relative Strength
    relative_strength = (momentum - sma_50) / total_volume_weighted_price
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = ((df['high'] - df['low']) / df['high']) * 100
    
    # Calculate Average Daily Return
    average_daily_return = df['close'].pct_change().rolling(window=10).mean()
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    log_return = np.log(df['close'] / df['close'].shift(10))
    adjusted_momentum = log_return / intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Logarithmic Return
    log_intraday_return = np.log(intraday_return + 1e-8)  # Add small constant to avoid log(0)
    log_intraday_return = np.clip(log_intraday_return, np.percentile(log_intraday_return, 1), np.percentile(log_intraday_return, 99))  # Winsorize
    
    # Combine Adjusted Momentum and Intraday Return
    volume_smoothed = df['volume'].ewm(span=10, adjust=False).mean()  # Exponential smoothing
    weighted_sum = (adjusted_momentum * volume_smoothed) + (log_intraday_return * volume_smoothed)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (relative_strength + combined_factors + average_daily_return + weighted_sum)
    
    return final_alpha_factor
```
