```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_momentum = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    momentum = (df['close'].diff() / weighted_momentum).fillna(0)
    
    # Adjust for Volatility
    adjusted_momentum = momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff().fillna(0)
    
    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    adjusted_cumulative_return = (cumulative_return + 1) / price_volatility
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Weighted Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    exp_weights = np.exp(np.linspace(-1, 0, 5))
    weighted_log_returns = (log_return.rolling(window=5).apply(lambda x: (x * exp_weights).sum(), raw=False)).fillna(0)
    adjusted_weighted_momentum = weighted_log_returns / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = adjusted_weighted_momentum - (ema_short - ema_long)
    volume_weighted_price_movement = (df['high'] - df['low']) * df['volume']
    integrated_factor = (relative_strength / volume_weighted_price_movement) + adjusted_cumulative_return
    integrated_factor *= (ema_short - ema_long)
    
    # Enhance with Intraday Price Behavior
    intraday_price_range = df['high'] - df['low']
    intraday_price_mean = (df['high'] + df['low'] + df['close']) / 3
    intraday_price_variability = intraday_price_range / intraday_price_mean
    enhanced_adjusted_momentum = adjusted_momentum * intraday_price_variability
    
    # Incorporate Open Price Momentum
    open_price_momentum = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    open_price_momentum_adj = open_price_momentum / intraday_volatility
    
    # Final Factor
    final_factor = (enhanced_adjusted_momentum * volume_change * trend_direction * open_price_momentum_adj) + (adjusted_cumulative_return * trend_direction)
    
    return final_factor
```
