```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Momentum
    weighted_momentum = close_price_change / vwp
    
    # Adjust for Volatility
    adjusted_volatility_momentum = weighted_momentum / high_to_low_range
    
    # Calculate Price Momentum
    short_term_mavg = df['close'].rolling(window=5).mean()
    long_term_mavg = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_mavg > long_term_mavg).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    
    # Adjust Momentum by Volume Change
    adjusted_momentum = trend_direction * volume_change
    
    # Combine with Adjusted Volatility Momentum
    combined_momentum = adjusted_volatility_momentum + adjusted_momentum
    
    # Factor
    factor = combined_momentum
    
    return factor
```
