```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_change / (weighted_price * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_factor = relative_strength + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = high_low_range / df['close']
    weighted_combined_factors = combined_factors * price_range_ratio
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    confirmed_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    weighted_price = (df['high'] + df['low'] + df['close']) * df['volume']
    total_weighted_price = weighted_price.rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    relative_strength_final = (momentum - long_term_ma) / volume_weighted_price_movement
    
    # Incorporate Price Oscillator
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_oscillator = short_term_ema - long_term_ema
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (integrated_factor * relative_strength_final * confirmed_momentum) / intraday_volatility
    
    return final_alpha_factor
```
