```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term Moving Average (5-day)
    short_term_ma = df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Moving Average (20-day)
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Set a volume threshold percentage
    volume_threshold = 0.1 * df['volume']
    
    # Adjust Momentum by Volume Change if Volume Change exceeds threshold
    adjusted_trend = trend_direction * (volume_change.where(volume_change.abs() > volume_threshold, 0))
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Validate Trend with Price Action
    within_short_long = (df['high'] < short_term_ma) & (df['low'] > long_term_ma)
    confidence_adjustment = within_short_long.astype(int) * 2 - 1
    final_trend = adjusted_trend * confidence_adjustment
    
    # Calculate Relative Strength
    price_momentum = short_term_ma - long_term_ma
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).sum() / df['volume'].sum()
    relative_strength = (price_momentum - long_term_ma) / weighted_price
    
    # Final Alpha Factor
    alpha_factor = final_trend * relative_strength
    
    return alpha_factor
```
