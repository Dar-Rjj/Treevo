```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the close price and the open price
    daily_momentum = df['close'] - df['open']
    
    # Compute the 10-day moving average of the daily momentum
    momentum_10_day = daily_momentum.rolling(window=10).mean()
    
    # Calculate the 50-day moving average of the daily momentum
    momentum_50_day = daily_momentum.rolling(window=50).mean()
    
    # Compute the difference between the 10-day and 50-day moving averages
    momentum_diff = momentum_10_day - momentum_50_day
    
    # Use the sign of the difference to determine the trend change
    momentum_sign = momentum_diff.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Multiply this sign by the absolute value of the 10-day moving average to create a scaled factor
    momentum_scaled = momentum_sign * abs(momentum_10_day)
    
    # Calculate the ratio of the high price to the low price to capture the intraday volatility
    intraday_volatility = df['high'] / df['low']
    
    # Compute the 20-day moving average of the intraday volatility
    intraday_volatility_20_day = intraday_volatility.rolling(window=20).mean()
    
    # Identify days where the intraday volatility is above its 20-day moving average
    high_volatility_days = (intraday_volatility > intraday_volatility_20_day).astype(int)
    
    # Calculate the number of such days in the past 10 days
    high_volatility_count = high_volatility_days.rolling(window=10).sum()
    
    # Calculate the percentage change in volume from the previous day
    volume_change = df['volume'].pct_change()
    
    # Compute the 10-day moving average of the volume change
    volume_change_10_day = volume_change.rolling(window=10).mean()
    
    # Identify days where the volume change is significantly above its 10-day moving average
    high_volume_activity_days = (volume_change > volume_change_10_day).astype(int)
    
    # Calculate the number of such days in the past 5 days
    high_volume_activity_count = high_volume_activity_days.rolling(window=5).sum()
    
    # Combine the factors
    alpha_factor = (momentum_scaled + high_volatility_count + high_volume_activity_count) / 3
    
    return alpha_factor
```
