```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=14):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Calculate Momentum
    log_returns = np.log(df['close'] / df['close'].shift(m))
    momentum = log_returns.rolling(window=m).sum()
    adjusted_momentum = momentum / intraday_volatility
    
    # Calculate Price Changes
    price_changes = df['close'].diff()
    
    # Separate Positive and Negative Gains
    positive_gains = price_changes.clip(lower=0)
    negative_gains = -price_changes.clip(upper=0)
    
    # Adjust for Volume
    volume_weighted_positive_gains = (positive_gains * df['volume']).rolling(window=m).sum()
    volume_weighted_negative_gains = (negative_gains * df['volume']).rolling(window=m).sum()
    
    # Calculate RSI
    rs = volume_weighted_positive_gains / volume_weighted_negative_gains
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Momentum and RSI
    combined_factor = adjusted_momentum * rsi
    
    return combined_factor
```
