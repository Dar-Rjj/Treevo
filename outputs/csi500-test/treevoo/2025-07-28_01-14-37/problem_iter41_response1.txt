```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, lookback=30):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['combined_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']) / df['volume']
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(lookback))
    df['sum_log_returns'] = df['log_return'].rolling(window=lookback).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['volume_adjusted_momentum'] = df['sum_log_returns'] * df['volume']
    df['final_factor'] = df['volume_adjusted_momentum'] / df['avg_intraday_volatility']
    
    return df['final_factor']
```
