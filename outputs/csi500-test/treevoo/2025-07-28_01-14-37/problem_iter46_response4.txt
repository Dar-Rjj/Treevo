import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma - long_term_ma).apply(lambda x: 1 if x > 0 else -1)
    
    # Enhance with Volume Confirmation
    volume_change = df['volume'].diff()
    volume_adjusted_trend = trend_direction * volume_change
    volume_threshold = volume_change.quantile(0.3)  # Set a threshold for volume change
    volume_adjusted_trend = volume_adjusted_trend.apply(lambda x: x if abs(x) > volume_threshold else 0)
    
    # Calculate Relative Strength
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = weighted_price.rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    relative_strength = (short_term_ma - long_term_ma) / volume_weighted_price_movement
    
    # Evaluate Intraday Dynamics
    intraday_return = df['close'] - df['open']
    intraday_volatility = df['high'] - df['low']
    cumulative_return = intraday_return.rolling(window=5).sum()
    prev_close = df['close'].shift(1)
    intraday_reversal = (df['high'] - prev_close) - (prev_close - df['low'])
    
    # Integrate Volume and Transaction Value
    avg_volume = df['volume'].rolling(window=20).mean()
    transaction_value = df['close'] * df['volume']
    avg_transaction_value = transaction_value.rolling(window=20).mean()
    volume_diff = df['volume'] - avg_volume
    value_diff = transaction_value - avg_transaction_value
    
    # Adjust for Volatility
    volatility_adjusted_momentum = (short_term_ma - long_term_ma) / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    adjusted_cumulative_return = cumulative_return - (short_term_ma - long_term_ma)
    combined_factor = relative_strength + adjusted_cumulative_return
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (volatility_adjusted_momentum + adjusted_cumulative_return + combined_factor) * relative_strength + intraday_reversal
    
    return final_alpha_factor
