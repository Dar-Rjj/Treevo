```python
def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Momentum
    weighted_momentum = close_price_change / vwp
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum_by_volume = trend_direction * volume_change
    
    # Final Factor
    final_factor = (adjusted_momentum + adjusted_momentum_by_volume) / 2
    
    return final_factor
```
