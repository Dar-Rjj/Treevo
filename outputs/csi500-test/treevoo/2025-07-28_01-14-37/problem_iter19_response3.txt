```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_momentum = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    price_change = df['close'].diff()
    momentum = price_change / weighted_momentum
    
    # Adjust for Volatility
    volatility_adjusted_momentum = momentum / high_low_range
    
    # Determine Trend Direction
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_direction = (ma_5 > ma_20).astype(int) - (ma_5 < ma_20).astype(int)
    
    # Incorporate Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_oscillator = ema_5 - ema_20
    
    # Confirm with Volume
    final_factor = (volatility_adjusted_momentum * trend_direction * price_oscillator)
    
    return final_factor
```
