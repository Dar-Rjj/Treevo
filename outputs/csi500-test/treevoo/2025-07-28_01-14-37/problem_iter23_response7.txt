```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term (5-day) and Long-Term (20-day) Moving Averages using Close price
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int)
    df['trend_direction'] = df['trend_direction'].apply(lambda x: 1 if x == 1 else -1)

    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Adjust Momentum by Volume Change
    df['adjusted_momentum'] = df['trend_direction'] * df['volume_change']

    # Calculate Relative Strength
    df['price_momentum'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = df['weighted_price'].sum()
    total_volume = df['volume'].sum()
    volume_weighted_price_movement = total_weighted_price / total_volume

    # Calculate the final factor
    df['relative_strength'] = df['price_momentum'] / volume_weighted_price_movement

    return df['relative_strength']
```
