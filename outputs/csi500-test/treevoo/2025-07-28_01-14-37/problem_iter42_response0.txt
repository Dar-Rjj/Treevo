```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, p=10):
    # Compute the difference between the close and open prices
    close_open_diff = df['close'] - df['open']
    
    # Measure the volatility using the high and low prices
    price_range = df['high'] - df['low']
    
    # Evaluate the momentum by looking at recent close prices
    weights = np.arange(1, n+1)
    weighted_close_prices = df['close'].rolling(window=n).apply(
        lambda x: np.sum(x * weights) / np.sum(weights), raw=True
    )
    momentum = df['close'] - weighted_close_prices
    
    # Analyze trading volume for potential buying or selling pressure
    median_volume = df['volume'].rolling(window=m).median()
    volume_signal = (df['volume'] > median_volume).astype(int)
    
    # Use the amount traded to estimate the value of transactions
    transaction_value = df['close'] * df['volume']
    avg_transaction_value = transaction_value.rolling(window=p).mean()
    transaction_signal = (transaction_value - avg_transaction_value) / avg_transaction_value
    
    # Combine all the signals into a single factor
    alpha_factor = (close_open_diff / price_range) + momentum + volume_signal + transaction_signal
    return alpha_factor
```
