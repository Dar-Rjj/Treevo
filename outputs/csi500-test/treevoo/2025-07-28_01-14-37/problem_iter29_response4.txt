```python
def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the difference between close and open prices
    price_change = df['close'] - df['open']
    
    # Calculate the ratio of price change to the price range
    price_change_ratio = price_change / price_range
    
    # Calculate the ratio of volume to amount
    vol_to_amt_ratio = df['volume'] / df['amount']
    
    # Combine the two ratios to form the heuristic factor
    factor = price_change_ratio * vol_to_amt_ratio
    
    return factor
```
