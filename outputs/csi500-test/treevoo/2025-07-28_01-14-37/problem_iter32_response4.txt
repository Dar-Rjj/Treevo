def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the weighted average of the open, high, low, and close prices
    # The weights are based on the volume traded
    w_open = df['open'] * df['volume']
    w_high = df['high'] * df['volume']
    w_low = df['low'] * df['volume']
    w_close = df['close'] * df['volume']
    total_volume = df['volume'].sum()
    weighted_average_price = (w_open + w_high + w_low + w_close) / (4 * total_volume)
    
    # Calculate the price change between the weighted average price and the close price
    price_change = weighted_average_price - df['close']
    
    # Calculate the factor value
    factor_value = price_change / df['close']
    
    return factor_value
