```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = trend_direction * volume_change
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    volume_adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (volume_adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Relative Strength
    price_momentum = short_term_ma - long_term_ma
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    relative_strength = (price_momentum - long_term_ma) / weighted_price
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = relative_strength + final_adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (df['high'] - df['low']) / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = integrated_factor + combined_factors
    
    return final_alpha_factor
```
