```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Momentum_Shift'] = df['SMA_5'] - df['SMA_5'].shift(1)
    
    # Analyze Volume and Amount Trends
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_Volume_Ratio'] = df['amount'] / df['volume']
    
    # Determine Trend Direction
    df['Short_Long_Trend'] = df['SMA_5'] - df['SMA_20']
    df['Trend_Direction'] = df['Short_Long_Trend'].apply(lambda x: 1 if x > 0 else -1)
    df['Prev_Trend_Direction'] = df['Trend_Direction'].shift(1)
    df['Confirmed_Trend'] = (df['Trend_Direction'] == df['Prev_Trend_Direction']).astype(int)
    
    # Enhance with Volume Confirmation
    df['Volume_Confirmed_Trend'] = df['Confirmed_Trend'] * df['Volume_Change']
    df['Volume_Confirmed_Trend'] = df['Volume_Confirmed_Trend'].apply(lambda x: 0 if abs(x) < 1e6 else x)  # Threshold
    
    # Calculate Relative Strength
    df['Relative_Strength'] = (df['SMA_5'] - df['SMA_20']) / df['close'].ewm(span=20, adjust=False).mean()
    
    # Identify Volatility Patterns
    df['Daily_Range'] = df['high'] - df['low']
    df['Volatility'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Explore Price Oscillators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['Price_Oscillator'] = df['SMA_50'] - df['SMA_200']
    
    # Construct Reversal Signals
    df['Days_Since_Highest_High'] = (df['high'].expanding().max() == df['high']).cumsum()
    df['Days_Since_Lowest_Low'] = (df['low'].expanding().min() == df['low']).cumsum()
    
    # Evaluate Price Action Relative to Moving Averages
    df['Distance_to_Moving_Avg'] = df['close'] - df['SMA_20']
    
    # Calculate Intraday Movement
    df['Intraday_Movement'] = df['high'] - df['low']
    
    # Combine Factors
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['Combined_Factors'] = df['Intraday_Movement'] + df['Close_to_Open_Return']
    df['Price_Range_Ratio'] = df['high'] / df['low']
    df['Weighted_Combined_Factors'] = df['Combined_Factors'] * df['Price_Range_Ratio']
    
    # Confirm with Volume
    df['Adjusted_Momentum'] = df['Volume_Confirmed_Trend'] * df['Momentum_Shift']
    
    # Calculate Intraday Reversal Indicator
    df['Intraday_Reversal'] = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])
    
    # Integrate Intraday Reversal Indicator
    df['Volatility_Adjusted_Reversal'] = df['Intraday_Reversal'] * df['Volatility']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Adjusted_Momentum'] + 
        df['Relative_Strength'] + 
        df['Volatility_Adjusted_Reversal'] + 
        df['Weighted_Combined_Factors']
    )
    
    return df['Final_Alpha_Factor']
```
