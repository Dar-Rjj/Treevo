```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_change / (weighted_sum * df['volume'])
    
    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Smoothing the Momentum
    smoothed_momentum = volatility_adjusted_momentum.rolling(window=5).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_factor = volume_adjusted_momentum + combined_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Momentum: Log Return
    log_return = np.log(df['close'] / df['close'].shift(5))
    intraday_volatility_adjusted_momentum = log_return.sum() / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    log_intraday_return = np.log(df['close'] / df['open'])
    volume_weighted_return = log_intraday_return.ewm(span=5, adjust=False).mean()
    volume_weighted_return = np.clip(volume_weighted_return, volume_weighted_return.quantile(0.05), volume_weighted_return.quantile(0.95))
    
    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    average_daily_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    cumulative_vwap = vwap.cumsum()
    cumulative_volume = df['volume'].cumsum()
    vwap = cumulative_vwap / cumulative_volume
    
    # Final Alpha Factor
    final_alpha_factor = (integrated_factor + intraday_volatility_adjusted_momentum + volume_weighted_return) * (volume_change * trend_direction * average_daily_range) + vwap
    
    return final_alpha_factor
```
