```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    df['trend_direction'] = (df['5_day_MA'] > df['20_day_MA']).astype(int) * 2 - 1

    # Enhance with Volume Confirmation
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_momentum'] = df['trend_direction'] * df['volume_change']

    # Calculate Relative Strength
    df['relative_strength'] = (df['5_day_MA'] - df['20_day_MA']) / df['close'].rolling(window=5).std()

    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return Over 5 Days
    df['cumulative_return_5d'] = df['daily_return'].rolling(window=5).sum()

    # Adjust by Volume
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return_5d'] * df['average_volume_5d']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_range'].rolling(window=5).mean()

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['momentum_adjusted'] = df['trend_direction'] - df['20_day_MA']
    df['integrated_relative_strength'] = df['relative_strength'] + df['adjusted_cumulative_return']

    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Open-to-Low Return
    df['open_to_low_return'] = df['open'] - df['low']

    # Combine Factors
    df['combined_sum'] = df['intraday_movement'] + df['close_to_open_return'] + df['open_to_low_return']
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    df['weighted_combined_factors'] = df['combined_sum'] * df['price_range_ratio']

    # Confirm with Volume
    df['volume_change_confirm'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_momentum_by_volume'] = df['trend_direction'] * df['volume_change_confirm']

    # Calculate Intraday Reversal Indicator
    df['intraday_reversal_indicator'] = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])

    # Integrate Intraday Reversal Indicator
    df['integrated_intraday_reversal'] = df['intraday_reversal_indicator'] * df['intraday_volatility']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_adjusted'] + 
                                df['integrated_relative_strength'] + 
                                df['weighted_combined_factors']) * df['relative_strength'] + df['integrated_intraday_reversal']

    return df['final_alpha_factor']
```
