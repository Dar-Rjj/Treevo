import pandas as pd

def heuristics_v2(df, n=20, m=60, k=30, window=5):
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    intraday_volatility = df['high_low_range'].rolling(window=n).mean()
    
    # Calculate Momentum
    log_returns = df['close'].apply(lambda x: np.log(x) - np.log(df['close'].shift(m)))
    sum_log_returns = log_returns.rolling(window=m).sum()
    adjusted_momentum = sum_log_returns / intraday_volatility
    
    # Calculate Volume Trend
    volume_change = (df['volume'] - df['volume'].shift(k))
    volume_trend = volume_change.rolling(window=k).sum()
    momentum_with_volume_trend = adjusted_momentum * volume_trend
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=window).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=window).mean()
    cumulative_return_adjusted = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    close_std = df['close'].rolling(window=window).std()
    final_cumulative_return = (cumulative_return_adjusted / close_std) + 1
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume'])
    total_weighted_price = weighted_price.rolling(window=window).sum()
    total_volume = df['volume'].rolling(window=window).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = (adjusted_momentum - sma_50) / volume_weighted_price_movement
    integrated_return = relative_strength + final_cumulative_return
    
    # Final Alpha Factor
    alpha_factor = momentum_with_volume_trend + integrated_return
    
    return alpha_factor
