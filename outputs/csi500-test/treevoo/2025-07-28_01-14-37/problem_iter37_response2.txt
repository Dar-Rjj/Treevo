```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average (SMA) of close prices over a specific period
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    
    # Subtract today's SMA from yesterday's SMA to identify momentum shifts
    sma_50_shift = sma_50 - sma_50.shift(1)
    sma_200_shift = sma_200 - sma_200.shift(1)
    
    # Calculate the rate of change (ROC) in closing prices
    roc = df['close'].pct_change(periods=1)
    
    # Compare ROC with a fixed threshold to gauge upward or downward pressure
    roc_threshold = 0.01
    roc_signal = (roc > roc_threshold).astype(int) - (roc < -roc_threshold).astype(int)
    
    # Calculate the daily volume change
    volume_change = df['volume'].pct_change(periods=1)
    
    # Calculate the ratio of amount to volume
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Compute the range (high - low) for each day
    daily_range = df['high'] - df['low']
    
    # Average the ranges over a given period to measure volatility
    avg_range = daily_range.rolling(window=20).mean()
    
    # Calculate the difference between SMA of different periods (e.g., 50-day and 200-day)
    sma_difference = sma_50 - sma_200
    
    # Determine the number of days since the last highest high
    days_since_high = (df['high'].cummax() == df['high']).astype(int).diff().fillna(0).cumsum()
    
    # Track the number of days since the last lowest low
    days_since_low = (df['low'].cummin() == df['low']).astype(int).diff().fillna(0).cumsum()
    
    # Calculate the distance between the current close and a key moving average
    distance_to_sma_50 = df['close'] - sma_50
    
    # Combine all the signals into a single alpha factor
    alpha_factor = (
        sma_50_shift + 
        sma_200_shift + 
        roc_signal + 
        volume_change + 
        amount_volume_ratio + 
        avg_range + 
        sma_difference + 
        days_since_high + 
        days_since_low + 
        distance_to_sma_50
    )
    
    return alpha_factor
```
