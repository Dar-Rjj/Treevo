```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=60, k=30):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=n).mean()

    # Calculate Momentum
    log_returns = np.log(df['close']) - np.log(df['close'].shift(m))
    sum_log_returns = log_returns.rolling(window=m).sum()
    adjusted_momentum = sum_log_returns / intraday_volatility

    # Calculate Volume Trend
    volume_change = df['volume'] - df['volume'].shift(k)
    sum_volume_changes = volume_change.rolling(window=k).sum()
    volume_trend = (sum_volume_changes + 1) / (df['volume'].rolling(window=k).std() + 1)

    # Combine Adjusted Momentum with Volume Trend
    factor = adjusted_momentum * volume_trend

    return factor
```
