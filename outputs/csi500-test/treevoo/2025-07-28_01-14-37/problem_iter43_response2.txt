```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return Over 5 Days
    df['cumulative_return'] = df['daily_return'].rolling(window=5).sum()
    
    # Adjust by Volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume_5d']
    
    # Calculate Intraday Volatility (Average High-Low Range over n days)
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range_5d'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Combine Intraday Momentum and Volatility (Weighted Sum)
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_high_low_range']) / (df['volume'] * 2)
    
    # Calculate Price Momentum
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.sign(df['short_term_ma'] - df['long_term_ma'])
    df['confirmed_trend_direction'] = np.where(df['trend_direction'] == df['trend_direction'].shift(1), 1, 0)
    
    # Enhance with Volume Confirmation
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['enhanced_trend'] = df['trend_direction'] * df['volume_change']
    df['enhanced_trend'] = np.where(np.abs(df['volume_change']) < 1e-6, 0, df['enhanced_trend'])
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma']) / (df['high'] + df['low'] + 2 * df['close']).mean()
    
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Weighted Momentum
    df['weighted_price'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    df['weighted_momentum'] = (df['close'] - df['close'].shift(1)) / (df['weighted_price'] * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['weighted_momentum'] / df['high_low_range']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return'] - df['close'].rolling(window=50).mean()
    
    # Combine Factors
    df['combined_factors'] = df['intraday_return'] + (df['close'] - df['open'])
    df['combined_factors'] = df['combined_factors'] * (df['high'] - df['low'])
    
    # Confirm with Volume
    df['combined_factors'] = df['combined_factors'] * df['volume_change']
    
    # Calculate Intraday Reversal Indicator
    df['intraday_reversal'] = df['high'] - df['close'].shift(1) - (df['close'].shift(1) - df['low'])
    
    # Integrate Intraday Reversal Indicator
    df['integrated_reversal'] = df['intraday_reversal'] * df['avg_high_low_range_5d']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['volatility_adjusted_momentum'] + 
        df['integrated_factor'] + 
        df['combined_factors'] + 
        df['integrated_reversal']
    )
    
    return df['final_alpha_factor'].dropna()
```
