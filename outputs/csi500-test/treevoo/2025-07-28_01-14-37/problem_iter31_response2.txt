import pandas as pd

def heuristics_v2(df, window=5, lookback_period=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(lookback_period)).apply(np.log)
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['weighted_price_momentum'] = df['price_momentum'] * df['volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    df['trend_direction'] = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_trend'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - long_term_ma) / df['weighted_price_momentum']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['final_combined_factor'] = df['relative_strength'] + df['adjusted_cumulative_return']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['final_combined_factor'] + df['confirmed_trend']
    
    return df['final_alpha_factor']
