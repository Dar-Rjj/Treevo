import pandas as pd

def heuristics_v2(df, window=5, lookback_period=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return Over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = (df['high_low_range'] * df['volume']).rolling(window=window).mean()
    df['combined_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(lookback_period)).apply(np.log)
    df['sum_log_returns'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['weighted_momentum'] = df['sum_log_returns'] * df['volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=10).mean()
    df['long_term_ma'] = df['close'].rolling(window=50).mean()
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['weighted_momentum'] + df['adjusted_cumulative_return']
    df['volume_weighted_price_movement'] = df['combined_factor']
    df['final_combined_factor'] = (df['relative_strength'] + df['volume_weighted_price_movement'])
    
    # Synthesize Final Alpha Factor
    df['volume_weighted_momentum'] = df['log_return'] * df['volume']
    df['final_alpha_factor'] = df['final_combined_factor'] + df['volume_weighted_momentum']
    
    return df['final_alpha_factor']
