```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term and Long-Term Moving Averages
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma'])
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['volume_weighted_price_movement'] = df['weighted_price'].sum() / df['volume'].sum()
    
    df['relative_strength'] = df['relative_strength'] / df['volume_weighted_price_movement']
    
    # Enhance Relative Strength
    df['volatility'] = df['close'].rolling(window=20).std()
    df['enhanced_relative_strength'] = df['relative_strength'] * df['volatility']
    
    return df['enhanced_relative_strength'].dropna()
```
