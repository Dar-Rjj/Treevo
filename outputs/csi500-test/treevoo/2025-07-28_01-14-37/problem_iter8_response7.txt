```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns as 
                           open, high, low, close, amount, volume.
                           
    Returns:
        pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) / price_range
    
    # Calculate the relative volume
    relative_volume = df['volume'] / df['amount']
    
    # Combine the factors to create the final alpha factor
    factor = buying_pressure * relative_volume
    
    return factor
```
