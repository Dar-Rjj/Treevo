import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_adjusted_cumulative_return = cumulative_return * avg_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (volume_adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Momentum
    lookback_period = 5
    log_return = np.log(df['close'] / df['close'].shift(lookback_period))
    momentum = log_return.rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    weighted_momentum = momentum * avg_volume / avg_high_low_range
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    sma_50 = df['close'].rolling(window=50).mean()
    relative_strength = momentum - sma_50
    integrated_factor = relative_strength + final_adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = (df['high'] - df['low']) / df['close']
    weighted_combined_factors = combined_factors * price_range_ratio
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = integrated_factor + weighted_combined_factors
    
    return final_alpha_factor
