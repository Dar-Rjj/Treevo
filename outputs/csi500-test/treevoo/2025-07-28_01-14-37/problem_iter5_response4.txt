```python
import pandas as pd

def heuristics_v2(data):
    # Calculate Daily Return
    data['DailyReturn'] = data['close'] / data['close'].shift(1) - 1
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    data['CumulativeReturn'] = data['DailyReturn'].rolling(window=window).sum()
    
    # Adjust by Volume
    data['AverageVolume'] = data['volume'].rolling(window=window).mean()
    data['AdjustedCumulativeReturn'] = data['CumulativeReturn'] * data['AverageVolume']
    
    # Adjust by Price Volatility
    data['CloseStdDev'] = data['close'].rolling(window=window).std()
    data['FinalAdjustedCumulativeReturn'] = (data['AdjustedCumulativeReturn'] / data['CloseStdDev']).add(1)
    
    # Calculate Momentum
    momentum_window = 10
    data['Momentum'] = data['close'] - data['close'].shift(momentum_window)
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma_window = 50
    data['SMA'] = data['close'].rolling(window=sma_window).mean()
    
    # Calculate Volume-Weighted Price Movement
    data['WeightedPrice'] = (data['high'] + data['low'] + data['close']) / 3 * data['volume']
    data['TotalVolume'] = data['volume'].rolling(window=window).sum()
    data['VolumeWeightedPriceMovement'] = data['WeightedPrice'].rolling(window=window).sum() / data['TotalVolume']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    data['RelativeStrength'] = (data['Momentum'] - data['SMA']) / data['VolumeWeightedPriceMovement']
    data['IntegratedFactor'] = data['RelativeStrength'] + data['FinalAdjustedCumulativeReturn']
    
    # Calculate Intraday Movement
    data['IntradayMovement'] = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    data['CloseToOpenReturn'] = data['close'] / data['open'] - 1
    
    # Calculate Price Range Ratio
    data['PriceRangeRatio'] = ((data['high'] - data['low']) / data['high']) * 100
    
    # Combine Factors
    data['CombinedFactors'] = (data['IntradayMovement'] + data['CloseToOpenReturn']) * data['PriceRangeRatio']
    
    # Synthesize Final Alpha Factor
    data['FinalAlphaFactor'] = data['IntegratedFactor'] + data['CombinedFactors']
    
    return data['FinalAlphaFactor'].dropna()
```
