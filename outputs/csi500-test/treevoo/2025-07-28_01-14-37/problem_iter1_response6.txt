```python
import pandas as pd

def heuristics_v2(df, n=14, m=7):
    # Compute Close Price Difference
    df['close_diff'] = df['close'].diff(n)
    
    # Smoothing the Momentum
    df['raw_momentum'] = df['close_diff'].rolling(window=m).mean()
    
    # Compute Daily Volume Ratio
    mean_volume = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / mean_volume
    
    # Adjust the Momentum with Volume
    df['adjusted_momentum'] = df['raw_momentum'] * df['volume_ratio']
    
    # Momentum Signal
    df['momentum_signal'] = df['adjusted_momentum'].apply(lambda x: 1 if x > 0 else (0 if x == 0 else -1))
    
    return df['momentum_signal']
```
