```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_sum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume and Range
    volume_change = df['volume'].diff()
    
    # Calculate Average Daily Range
    average_daily_range = high_to_low_range.rolling(window=5).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    cumulative_vwap = vwap.cumsum()
    cumulative_volume = df['volume'].cumsum()
    vwap = cumulative_vwap / cumulative_volume
    
    # Final Factor
    final_factor = (
        adjusted_momentum * 
        volume_change * 
        trend_direction * 
        average_daily_range * 
        vwap
    )
    
    return final_factor
```
