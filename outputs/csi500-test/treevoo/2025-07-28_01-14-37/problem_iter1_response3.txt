```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = weighted_price.rolling(window=50).sum()
    total_volume = df['volume'].rolling(window=50).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    
    # Calculate Relative Strength
    relative_strength = (momentum - sma_50) / volume_weighted_price_movement
    
    return relative_strength
```
