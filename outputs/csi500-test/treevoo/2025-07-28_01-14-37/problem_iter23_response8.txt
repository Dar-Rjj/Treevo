```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    df['Trend_Direction'] = (df['SMA_5'] > df['SMA_20']).astype(int) * 2 - 1

    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].diff().fillna(0)

    # Adjust Momentum by Volume Change
    df['Adjusted_Momentum'] = df['Trend_Direction'] * df['Volume_Change']

    # Calculate Volume-Weighted Price Movement
    df['Weighted_Price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['Total_Weighted_Price'] = df['Weighted_Price'].rolling(window=20).sum()
    df['Total_Volume'] = df['volume'].rolling(window=20).sum()
    df['Volume_Weighted_Price_Movement'] = df['Total_Weighted_Price'] / df['Total_Volume']

    # Calculate Relative Strength
    df['Relative_Strength'] = (df['SMA_5'] - df['SMA_20']) / df['Volume_Weighted_Price_Movement']

    return df['Relative_Strength']
```
