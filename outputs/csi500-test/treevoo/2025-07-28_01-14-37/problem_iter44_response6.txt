import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term (5-day) and Long-Term (20-day) Moving Averages
    df['short_mavg'] = df['close'].rolling(window=5).mean()
    df['long_mavg'] = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_mavg'] > df['long_mavg'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'].pct_change()
    volume_threshold = 0.05  # Example threshold
    df['adjusted_trend'] = df['trend_direction'] * np.where(df['volume_change'] > volume_threshold, df['volume_change'], 1)

    # Validate Trend with Price Action
    df['price_range'] = df['high'] - df['low']
    df['trend_validation'] = np.where((df['price_range'] < df['short_mavg']) & (df['price_range'] > df['long_mavg']), 1.5, 1)
    df['final_trend'] = df['adjusted_trend'] * df['trend_validation']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=cumulative_window).sum()

    # Adjust Cumulative Return by Average Volume
    df['average_volume'] = df['volume'].rolling(window=cumulative_window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Adjust by Price Volatility
    df['price_volatility'] = df['close'].rolling(window=cumulative_window).std()
    df['final_adjusted_return'] = (df['adjusted_cumulative_return'] / df['price_volatility']) + 1

    # Simple Price Momentum
    n_days = 5
    df['simple_price_momentum'] = df['close'] - df['close'].shift(n_days)

    # RSI Momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n_days).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n_days).mean()
    rs = gain / loss
    df['rsi_momentum'] = 100 - (100 / (1 + rs))

    # Combine Momentum and RSI
    df['combined_momentum'] = df['simple_price_momentum'] + df['rsi_momentum']

    # Volume Trend
    df['volume_moving_avg'] = df['volume'].rolling(window=n_days).mean()
    df['volume_trend'] = df['volume'] - df['volume_moving_avg']

    # Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=n_days).std()

    # True Range
    df['true_range'] = np.maximum(df['high'] - df['low'], 
                                  np.abs(df['high'] - df['close'].shift(1)),
                                  np.abs(df['low'] - df['close'].shift(1))).max(axis=1)

    df['atr'] = df['true_range'].rolling(window=n_days).mean()

    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = df['ema_12'] - df['ema_26']
