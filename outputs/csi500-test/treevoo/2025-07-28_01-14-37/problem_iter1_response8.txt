```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Separate Positive and Negative Gains
    positive_gains = df['price_change'].apply(lambda x: max(x, 0))
    negative_gains = df['price_change'].apply(lambda x: abs(min(x, 0)))
    
    # Sum of Positive and Negative Gains over a Period (e.g., 14 days)
    period = 14
    positive_sum = positive_gains.rolling(window=period).sum()
    negative_sum = negative_gains.rolling(window=period).sum()
    
    # Adjust for Volume
    volume_weighted_positive_sum = (positive_gains * df['volume']).rolling(window=period).sum()
    volume_weighted_negative_sum = (negative_gains * df['volume']).rolling(window=period).sum()
    
    # Calculate RS (Relative Strength)
    rs = volume_weighted_positive_sum / volume_weighted_negative_sum
    
    # Apply RSI Formula
    rsi = 100 - (100 / (1 + rs))
    
    # Return the factor as a pandas Series
    return rsi
```
