```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between the close and open price
    rel_diff_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative difference between the high and low price
    rel_diff_high_low = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative trading volume
    rel_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the relative amount traded
    rel_amount = df['amount'] / df['amount'].rolling(window=20).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = rel_diff_close_open + rel_diff_high_low * rel_volume + rel_amount
    
    return alpha_factor
```
