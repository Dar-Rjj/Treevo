```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change (close - open)
    df['price_change'] = df['close'] - df['open']
    
    # Calculate High-Low Range (high - low)
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Open-Previous Close Difference (open - previous_close)
    df['open_prev_close_diff'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Price (volume * close)
    df['volume_weighted_price'] = df['volume'] * df['close']
    
    # Calculate Volume-Weighted Average Price (volume * (open + high + low + close) / 4)
    df['volume_weighted_avg_price'] = df['volume'] * (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate Simple Moving Average (SMA) of Closing Prices
    df['sma_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate Weighted Moving Average (WMA) of Closing Prices
    weights = pd.Series(range(1, 6))
    df['wma_5'] = df['close'].rolling(window=5).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
    
    # Calculate Rate of Change (ROC) (current_close / previous_n_period_close - 1)
    df['roc_5'] = df['close'] / df['close'].shift(5) - 1
    
    # Calculate True Range (max(high - low, abs(high - previous_close), abs(low - previous_close)))
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    
    # Calculate Volume Change (current_volume - previous_volume)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate On-Balance Volume (OBV) (cumulative sum of volume adjusted for price direction)
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Calculate Cumulative Amount (cumulative sum of amount)
    df['cumulative_amount'] = df['amount'].cumsum()
    
    # Calculate Amount Change (current_amount - previous_amount)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Calculate Price-Volume Trend (PVT) (previous_PVT + (current_close - previous_close) / previous_close * current_volume)
    df['pvt'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    df['pvt'] = df['pvt'].cumsum()
    
    # Calculate Negative Volume Index (NVI) (if volume < previous_volume, NVI = previous_NVI + (current_close - previous_close) / previous_close * 100, else NVI = previous_NVI)
    df['nvi'] = 0
    nvi = 0
    for i in range(1, len(df)):
        if df['volume'].iloc[i] < df['volume'].iloc[i-1]:
            nvi += (df['close'].iloc[i] - df['close'].iloc[i-1]) / df['close'].iloc[i-1] * 100
        df['nvi'].iloc[i] = nvi
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['price_change'] + 
        df['high_low_range'] + 
        df['open_prev_close_diff'] + 
        df['volume_weighted_price'] + 
        df['volume_weighted_avg_price'] + 
        df['sma_5'] + 
        df['wma_5'] + 
        df['roc_5'] + 
        df['true_range'] + 
        df['volume_change'] + 
        df['obv'] + 
        df['cumulative_amount'] + 
        df['amount_change'] + 
        df['pvt'] + 
        df['nvi']
    )
    
    return alpha_factor
```
