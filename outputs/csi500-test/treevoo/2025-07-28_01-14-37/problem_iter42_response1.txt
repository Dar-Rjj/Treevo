```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the body of the candle
    df['body'] = df['close'] - df['open']
    
    # Calculate the upper and lower shadows
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    
    # Calculate a simple moving average of the close price
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the factor as a weighted sum of the components
    factor = (df['body'] * 0.3) + (df['upper_shadow'] * 0.2) + (df['lower_shadow'] * 0.2) + (df['sma_close_10'] * 0.1) + (df['volume_change'] * 0.2)
    
    return factor
```
