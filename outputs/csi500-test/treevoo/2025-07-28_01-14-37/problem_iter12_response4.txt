```python
import pandas as pd

def heuristics_v2(df, window=5, momentum_window=10, sma_window=50):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over the specified window
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Adjust by Price Volatility
    df['std_close'] = df['close'].rolling(window=window).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['std_close']) + 1
    
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(momentum_window)
    
    # Calculate 50-day Simple Moving Average (SMA)
    df['sma_50'] = df['close'].rolling(window=sma_window).mean()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['momentum'] - df['sma_50']
    df['integrated_factor'] = df['final_adjusted_cumulative_return'] + df['relative_strength']
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    df['combined_factors'] = df['intraday_movement'] + df['close_to_open_return']
    df['price_range_ratio'] = df['high'] / df['low']
    df['weighted_combined_factors'] = df['combined_factors'] * df['price_range_ratio']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['integrated_factor'] + df['weighted_combined_factors']
    
    return df['alpha_factor']
```
