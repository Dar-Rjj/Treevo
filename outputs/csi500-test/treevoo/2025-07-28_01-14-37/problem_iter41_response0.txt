import pandas as pd

def heuristics_v2(df, n=5, m=30):
    # Daily Return (Thought 2-4)
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Lagged Close Price for Momentum (Thought 5)
    df['lagged_close'] = df['close'].shift(n)
    df['momentum'] = (df['close'] / df['lagged_close']) - 1
    
    # Cumulative Return (Thought 6-8)
    df['cumulative_return'] = df['daily_return'].rolling(window=n).sum()
    
    # Adjust by Volume (Thought 9-14)
    df['avg_volume'] = df['volume'].rolling(window=n).sum() / n
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Intraday Volatility (Thought 15-17)
    df['intraday_volatility'] = (df['high'] - df['low'])
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=n).mean()
    
    # Intraday Return (Thought 18-19)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility (Thought 20-26)
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['combined_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']).rolling(window=n).mean()

    df['final_factor'] = df['weighted_price_momentum'] + df['combined_factor']
    
    return df['final_factor'].dropna()
