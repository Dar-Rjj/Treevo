```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=5, long_term_ma=20, m_days=30):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=lookback_period).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=lookback_period).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_high_low_range']
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(m_days))
    df['sum_log_returns'] = df['log_return'].rolling(window=m_days).sum()
    
    # Calculate Short-Term Moving Average
    df['short_term_ma'] = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate Long-Term Moving Average
    df['long_term_ma'] = df['close'].rolling(window=long_term_ma).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Adjust by Intraday Volatility and Volume
    df['weighted_sum_log_returns'] = df['sum_log_returns'] * df['volume'] / df['intraday_volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['weighted_sum_log_returns'] + df['adjusted_cumulative_return']
    df['volume_weighted_price_movement'] = (df['close'] - df['close'].shift(lookback_period)) * df['volume']
    df['integrated_factor'] = df['relative_strength'] + df['volume_weighted_price_movement']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['volume_adjusted_momentum']
    
    # Incorporate Intraday Reversal into Final Alpha Factor
    df['intraday_reversal'] = df['open'] - df['close']
    df['final_alpha_factor'] = df['final_alpha_factor'] + df['intraday_reversal']
    
    return df['final_alpha_factor']
```
