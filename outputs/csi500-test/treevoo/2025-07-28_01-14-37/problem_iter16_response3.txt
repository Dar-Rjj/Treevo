```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5):
    # Compute Close Price Difference
    close_diff = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    smooth_momentum = close_diff.rolling(window=m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = smooth_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_factor = combined_factor + volume_adjusted_momentum
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Momentum: Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(m))
    momentum = log_return.rolling(window=m).sum() / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    log_intraday_return = np.log(df['close']) - np.log(df['open'])
    winsorized_log_intraday_return = log_intraday_return.clip(lower=log_intraday_return.quantile(0.05), upper=log_intraday_return.quantile(0.95))
    volume_weighted_intraday_return = winsorized_log_intraday_return * daily_volume_ratio
    
    # Final Alpha Factor
    alpha_factor = integrated_factor + momentum + volume_weighted_intraday_return
    
    return alpha_factor
```
