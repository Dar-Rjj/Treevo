```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the price movement from the previous close to the current open
    df['prev_close_to_open'] = df['open'] - df['close'].shift(1)
    
    # Calculate the difference between high and low for the day
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the percentage change in volume from the previous day
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate the ratio of amount to volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Heuristic factor: Combine the calculated features
    factor = (df['prev_close_to_open'] / df['daily_range']) * df['volume_change'] * df['amount_volume_ratio']
    
    return factor
```
