```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']

    # Adjust for Skewness and Kurtosis
    skewness = df['combined_intraday_factor'].skew()
    kurtosis = df['combined_intraday_factor'].kurtosis()
    df['rescaled_combined_intraday_factor'] = (df['combined_intraday_factor'] - df['combined_intraday_factor'].mean()) / df['combined_intraday_factor'].std()
    df['rescaled_combined_intraday_factor'] = (df['rescaled_combined_intraday_factor'] - skewness) / kurtosis

    # Calculate Price Momentum
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['trend_direction'] * df['volume_change']

    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma']) / df['average_volume']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_momentum'] = df['adjusted_cumulative_return'] + df['relative_strength']

    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['total_weighted_price'] = df['weighted_price'].rolling(window=window).sum()
    df['total_volume'] = df['volume'].rolling(window=window).sum()
    df['volume_weighted_price_movement'] = df['total_weighted_price'] / df['total_volume']

    # Final Rescaling
    df['final_factor'] = df['integrated_momentum'] + df['volume_weighted_price_movement']
    df['final_factor'] = (df['final_factor'] - df['final_factor'].mean()) / df['final_factor'].std()
    df['final_factor'] = (df['final_factor'] - df['final_factor'].skew()) / df['final_factor'].kurtosis()

    return df['final_factor'].dropna()
```
