```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5, decay_factor=0.9):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Return
    df['Log_Return'] = np.log(df['Intraday_Return'] + 1e-6)
    
    # Calculate High-to-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Close Price Difference
    df['Close_Diff'] = df['close'].diff()
    
    # Smoothing the Momentum
    df['Smoothed_Momentum'] = df['Close_Diff'].rolling(window=m).mean()
    
    # Volume Weighing Process
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['Volume_Adjusted_Momentum'] = df['Smoothed_Momentum'] * df['Volume_Ratio']
    
    # Calculate Intraday Movement
    df['Intraday_Movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    df['Price_Range_Ratio'] = ((df['high'] - df['low']) / df['high']) * 100
    
    # Combine Factors
    df['Combined_Factor'] = (df['Intraday_Movement'] + df['Close_to_Open_Return']) * df['Price_Range_Ratio']
    
    # Integrate Volume Adjusted Momentum
    df['Integrated_Volume_Adjusted_Momentum'] = df['Combined_Factor'] + df['Volume_Adjusted_Momentum']
    
    # Calculate Weighted Momentum
    df['VWP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    df['Weighted_Momentum'] = df['Close_Diff'] / df['VWP']
    
    # Adjust for Volatility
    df['Volatility_Adjusted_Momentum'] = df['Weighted_Momentum'] / df['High_Low_Range']
    
    # Combine Intraday and Volatility-Adjusted Momentum
    df['Log_Return_Smoothed'] = df['Log_Return'].ewm(alpha=decay_factor).mean() * df['Volume_Ratio']
    df['Vol_Adjusted_Mom_Smoothed'] = df['Volatility_Adjusted_Momentum'].ewm(alpha=decay_factor).mean() * df['Volume_Ratio']
    df['Combined_Momentum'] = (df['Log_Return_Smoothed'] + df['Vol_Adjusted_Mom_Smoothed']) / 2
    
    # Determine Trend Direction
    df['Short_Term_MA'] = df['close'].rolling(window=5).mean()
    df['Long_Term_MA'] = df['close'].rolling(window=20).mean()
    df['Trend_Direction'] = np.where(df['Short_Term_MA'] > df['Long_Term_MA'], 1, -1)
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'].diff()
    df['Adjusted_Momentum'] = df['Trend_Direction'] * df['Volume_Change']
    df['Final_Factor'] = (df['Integrated_Volume_Adjusted_Momentum'] + df['Adjusted_Momentum']) / 2
    
    # Calculate Relative Strength
    df['Price_Momentum'] = (df['close'] - df['Long_Term_MA'])
    df['Volume_Weighted_Price_Movement'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    df['Relative_Strength'] = df['Price_Momentum'] / df['Volume_Weighted_Price_Movement']
    
    return df['Final_Factor'] + df['Relative_Strength']
```
