```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Log Return over m days lookback period
    log_return = np.log(df['close']) - np.log(df['close'].shift(m))
    
    # Calculate Open-to-Close Log Return
    open_to_close_log_return = np.log(df['close']) - np.log(df['open'])
    
    # Multiply Log Returns by Volume
    volume_weighted_log_return = log_return * df['volume']
    volume_weighted_open_to_close_log_return = open_to_close_log_return * df['volume']
    
    # Sum of Volume-Weighted Log Returns and Open-to-Close Log Returns
    total_volume_weighted_return = (volume_weighted_log_return + 
                                    volume_weighted_open_to_close_log_return).rolling(window=m).sum()
    
    # Adjust by Intraday Volatility
    alpha_factor = total_volume_weighted_return / intraday_volatility
    
    return alpha_factor
```
