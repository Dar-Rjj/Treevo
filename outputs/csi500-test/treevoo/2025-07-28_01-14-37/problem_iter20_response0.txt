import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume

    # Adjust by Price Volatility
    price_std = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_std) + 1

    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()

    # Adjust by Intraday Volatility and Volume
    weighted_momentum = (momentum * average_volume) / intraday_volatility

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - momentum.rolling(window=50).mean()
    integrated_factor = relative_strength + final_adjusted_cumulative_return

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close']) / df['open']

    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = (df['high'] - df['low']) / df['close']
    weighted_combined_factors = combined_factors * price_range_ratio

    # Synthesize Final Alpha Factor
    final_alpha_factor = integrated_factor + weighted_combined_factors

    return final_alpha_factor
