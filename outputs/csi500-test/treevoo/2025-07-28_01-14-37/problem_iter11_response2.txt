```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100

    # Combine Intraday and Close-to-Open Factors
    combined_factor = intraday_movement + close_to_open_return
    weighted_combined_factor = combined_factor * price_range_ratio

    # Incorporate Volume Impact
    volume_impact = df['volume']
    combined_with_volume = weighted_combined_factor + volume_impact

    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume

    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1

    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'].diff()

    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Compute Weighted Momentum
    weighted_momentum = close_price_change / vwp

    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_momentum / high_to_low_range

    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1

    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum = trend_direction * volume_change
    final_momentum = (volatility_adjusted_momentum + adjusted_momentum) / 2

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = short_term_ma - long_term_ma
    integrated_factor = (relative_strength / vwp) + final_adjusted_cumulative_return

    return integrated_factor
```
