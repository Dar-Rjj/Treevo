```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10):
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=n).mean()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=n).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Adjust by Price Volatility
    df['std_close'] = df['close'].rolling(window=n).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['std_close']) + 1
    
    # Calculate Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(m))
    df['momentum'] = df['log_return'].rolling(window=m).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['weighted_momentum'] = df['momentum'] * df['average_volume'] / df['intraday_volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['rsi'] = df['momentum'] - df['close'].rolling(window=50).mean()
    df['integrated_factor'] = df['rsi'] + df['final_adjusted_cumulative_return']
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    df['combined_factors'] = df['intraday_movement'] + df['close_to_open_return']
    df['price_range_ratio'] = df['high'] - df['low']
    df['weighted_combined_factors'] = df['combined_factors'] * df['price_range_ratio']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['integrated_factor'] + df['weighted_combined_factors']
    
    return df['alpha_factor']
```
