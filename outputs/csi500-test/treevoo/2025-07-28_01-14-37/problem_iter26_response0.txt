import pandas as pd

def heuristics_v2(df, window=5, lookback_period=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(lookback_period)).apply(np.log)
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['weighted_momentum'] = df['price_momentum'] * df['volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    df['trend_direction'] = np.where(df['short_ma'] > df['long_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - df['long_ma']) / df['volume_weighted_price_movement']
    df['volume_weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['total_volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['volume_weighted_price_movement'] = df['volume_weighted_price'].rolling(window=lookback_period).sum() / df['total_volume']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return'] + df['volume_weighted_price_movement']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['combined_intraday_factor']
    
    # Incorporate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['log_return'] * df['volume']
    df['final_alpha_factor'] += df['volume_weighted_momentum']
    
    return df['final_alpha_factor']
