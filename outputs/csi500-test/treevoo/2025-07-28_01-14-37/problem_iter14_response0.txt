```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Logarithmic Return
    log_return = np.log(1 + intraday_return + 1e-8)
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Difference
    close_diff = df['close'].diff()
    
    # Smoothing the Momentum
    momentum = close_diff.rolling(window=5).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_momentum = volume_adjusted_momentum + combined_factor
    
    # Calculate Weighted Momentum
    vwp = df['amount'] / df['volume']
    weighted_momentum = close_diff / vwp
    
    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / high_low_range
    
    # Combine Intraday and Volatility-Adjusted Momentum
    decay_factor = 0.9
    exp_smoothed_log_return = log_return.ewm(alpha=decay_factor).mean() * daily_volume_ratio
    exp_smoothed_adjusted_momentum = adjusted_weighted_momentum.ewm(alpha=decay_factor).mean() * daily_volume_ratio
    combined_momentum = exp_smoothed_log_return + exp_smoothed_adjusted_momentum
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_trend = trend_direction * volume_change
    final_factor = (integrated_momentum + adjusted_trend) / 2
    
    return final_factor
```
