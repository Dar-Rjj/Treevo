```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day Moving Averages (Close price)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['5_day_MA'] > df['20_day_MA']).astype(int) * 2 - 1
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Momentum by Volume Change
    df['adjusted_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['5_day_MA'] - df['20_day_MA'])
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    total_weighted_price = df['weighted_price'].sum()
    total_volume = df['volume'].sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    
    df['relative_strength'] = df['relative_strength'] / volume_weighted_price_movement
    
    return df['relative_strength']
```
