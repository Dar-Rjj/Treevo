import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = (df['close'].diff() / (weighted_sum * np.sqrt(df['volume'])))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    combined_intraday_factor = (intraday_return * df['volume']) / high_to_low_range
    
    # Adjust by Intraday Volatility and Volume
    log_returns = np.log(df['close'] / df['close'].shift(1))
    adjusted_cumulative_return = (log_returns * df['volume']) / high_to_low_range
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    momentum_by_volume_change = trend_direction * volume_change
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = combined_intraday_factor + adjusted_cumulative_return
    final_combined_factor = (combined_intraday_factor + momentum_by_volume_change) / 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = relative_strength + final_combined_factor
    
    return final_alpha_factor
