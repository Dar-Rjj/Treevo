import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    trend_confirmed = trend_direction * (trend_direction.shift(1) == trend_direction).astype(int)
    
    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    trend_adjusted = trend_confirmed * volume_change
    trend_adjusted[volume_change.abs() < 1e-6] = 0
    
    # Calculate Relative Strength
    relative_strength = (short_term_ma - long_term_ma) / ((df['high'] + df['low'] + 2*df['close']) * df['volume']).sum() / df['volume'].sum()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2*df['close']) / 4 * np.sqrt(df['volume'])
    weighted_momentum = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Combine Intraday Momentum and Volatility
    combined_intraday_factor = (intraday_return * df['volume']) + (intraday_volatility * df['volume'])
    
    # Calculate Price Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    price_momentum = log_return.rolling(window=5).sum()
    
    # Adjust by Intraday Volatility and Volume
    adjusted_price_momentum = price_momentum * df['volume'] / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = adjusted_cumulative_return + relative_strength
    final_integrated_factor = (combined_intraday_factor + (df['high'] + df['low'] + 2*df['close']) * df['volume']).sum() / df['volume'].sum()
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (df['high'] - df['low']) / (df['high'] - df['low']).max()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_price_momentum + integrated_factor + combined_factors) * relative_strength
    
    return final_alpha_factor
