```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=5).sum()
    
    # Adjust by Volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume_5d']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open'])
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = (df['high'] - df['low']) * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Calculate Price Momentum
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.sign(df['short_term_ma'] - df['long_term_ma'])
    df['confirmed_trend_direction'] = np.where(df['trend_direction'] == df['trend_direction'].shift(1), 1, 0)
    
    # Enhance with Volume Confirmation
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['confirmed_trend_direction'] * df['volume_change']
    df['volume_adjusted_trend'] = np.where(abs(df['volume_adjusted_trend']) < 1e-6, 0, df['volume_adjusted_trend'])
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma']) / ((df['high'] + df['low'] + 2 * df['close']) * df['volume']).div(df['volume'].cumsum())
    
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Close Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Weighted Momentum
    df['weighted_momentum'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    df['weighted_momentum'] = df['close_price_change'] / (df['weighted_momentum'] * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['weighted_momentum'] / df['high_to_low_range']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_relative_strength'] = df['relative_strength'] + df['adjusted_cumulative_return'] - df['short_term_ma'].rolling(window=50).mean()
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Combine Factors
    df['combined_factors'] = (df['intraday_movement'] + df['intraday_return']) * df['high_to_low_range']
    
    # Confirm with Volume
    df['volume_adjusted_combined_factors'] = df['combined_factors'] * df['volume_adjusted_trend']
    
    # Calculate Intraday Reversal Indicator
    df['intraday_reversal_indicator'] = (df['high'].shift(1) - df['close'].shift(1)) - (df['close'].shift(1) - df['low'].shift(1))
    
    # Integrate Intraday Reversal Indicator
    df['integrated_intraday_reversal'] = df['intraday_reversal_indicator'] * df['intraday_volatility']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['volatility_adjusted_momentum'] + df['integrated_relative_strength'] + df['volume_adjusted_combined_factors'] + df['integrated_intraday_reversal'])
    
    return df['final_alpha_factor'].dropna()
```
