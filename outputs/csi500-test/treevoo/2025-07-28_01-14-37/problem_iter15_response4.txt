```python
import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(data):
    # Calculate High-to-Low Range
    high_to_low_range = data['high'] - data['low']
    
    # Calculate Close Price Change
    close_price_change = data['close'].diff()
    
    # Compute Weighted Momentum
    sum_price = (data['high'] + data['low'] + 2 * data['close']) / 4
    weighted_momentum = close_price_change / (sum_price * np.sqrt(data['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = data['close'].rolling(window=5).mean()
    long_term_ma = data['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = data['volume'].diff()
    intermediate_factor = adjusted_momentum * volume_change * trend_direction
    
    # Confirm with Price
    def calculate_slope(series):
        x = np.arange(len(series))
        y = series.values
        slope, _, _, _, _ = linregress(x, y)
        return slope
    
    price_slope = data['close'].rolling(window=10).apply(calculate_slope, raw=False)
    final_factor = intermediate_factor * price_slope
    
    return final_factor
```
