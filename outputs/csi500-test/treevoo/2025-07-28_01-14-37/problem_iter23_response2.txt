import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day moving averages
    df['short_mavg'] = df['close'].rolling(window=5).mean()
    df['long_mavg'] = df['close'].rolling(window=20).mean()

    # Determine trend direction
    df['trend_dir'] = (df['short_mavg'] > df['long_mavg']).astype(int) * 2 - 1

    # Confirm with volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['trend_dir'] * df['volume_change']

    df['price_momentum'] = df['short_mavg']
    df['volume_weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']

    df['relative_strength'] = (df['price_momentum'] - df['long_mavg']) / df['total_volume_weighted_price']

    # Calculate intraday volatility
    df['intraday_volatility'] = df['high'] - df['low']
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=5).mean()

    # Calculate daily return
    df['daily_return'] = df['close'] / df['close'].shift(1)

    # Calculate cumulative return
    df['cumulative_return'] = df['daily_return'].rolling(window=5).apply(lambda x: x.prod())

    # Adjust by volume
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']

    # Adjust by price volatility
    df['price_volatility'] = df['close'].rolling(window=5).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['price_volatility']) + 1

    # Integrate relative strength and adjusted cumulative return
    df['integrated_factor'] = df['relative_strength'] + df['final_adjusted_cumulative_return']

    # Calculate intraday movement
    df['intraday_movement'] = df['high'] - df['low']

    # Calculate close-to-open return
    df['close_to_open_return'] = df['close'] / df['open']

    # Combine factors
    df['combined_factors'] = df['intraday_movement'] + df['close_to_open_return']
