import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term Moving Average (5-day)
    short_term_ma = df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Moving Average (20-day)
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust Momentum by Volume Change
    adjusted_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    relative_strength = (short_term_ma - long_term_ma) / calculate_volume_weighted_price_movement(df)
    
    # Calculate Volatility (Standard Deviation of Close prices, 20-day)
    volatility = df['close'].rolling(window=20).std()
    
    # Normalize Relative Strength by Volatility
    enhanced_relative_strength = relative_strength / volatility
    
    return enhanced_relative_strength

def calculate_volume_weighted_price_movement(df):
    # Calculate Weighted Price
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Sum the weighted price and divide by total volume
    volume_weighted_price_movement = weighted_price.sum() / df['volume'].sum()
    
    return volume_weighted_price_movement
