```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    close_price_change = df['close'] - df['close'].shift(1)
    momentum_ratio = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    volatility_adjusted_momentum = momentum_ratio / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1  # +1 if short > long, -1 otherwise
    
    # Incorporate Price Oscillator
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_oscillator = short_term_ema - long_term_ema
    
    # Include Volume Oscillator
    short_term_volume_ma = df['volume'].rolling(window=5).mean()
    long_term_volume_ma = df['volume'].rolling(window=20).mean()
    volume_oscillator = short_term_volume_ma - long_term_volume_ma
    
    # Combine Factors
    final_factor = (volatility_adjusted_momentum * trend_direction * price_oscillator * volume_oscillator)
    
    return final_factor
```
