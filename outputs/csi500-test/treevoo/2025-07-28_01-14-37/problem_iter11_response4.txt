```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=5, m_days=30):
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    intraday_volatility = df['high_low_range'].rolling(window=n_days).mean()

    # Calculate Momentum
    log_returns = np.log(df['close'] / df['close'].shift(m_days))
    momentum = log_returns.rolling(window=m_days).sum()
    
    # Adjust by Intraday Volatility and Volume
    weighted_momentum = (momentum * df['volume']) / intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Logarithmic Return
    log_intraday_return = np.log(1 + intraday_return)
    
    # Combine Intraday Momentum and Volatility
    combined_factor = (log_intraday_return * df['volume']) / intraday_volatility
    
    # Adjust for Skewness and Kurtosis
    skewness = combined_factor.skew()
    kurtosis = combined_factor.kurtosis()
    adjusted_combined_factor = (combined_factor - skewness) / (kurtosis + 1)
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df[['high', 'low', 'close']] * np.sqrt(df['volume'])).sum(axis=1) / df['volume']
    
    # Calculate Final Alpha Factor
    alpha_factor = weighted_momentum + adjusted_combined_factor + weighted_price
    alpha_factor = (alpha_factor - alpha_factor.ewm(span=m_days).mean()) / alpha_factor.std()
    
    return alpha_factor
```
