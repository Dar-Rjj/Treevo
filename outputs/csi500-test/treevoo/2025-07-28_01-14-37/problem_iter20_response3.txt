```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Weighted Momentum
    df['weighted_momentum'] = (df['high'] + df['low'] + 2 * df['close']) / 4 * df['volume']
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['momentum_ratio'] = df['close_change'] / df['weighted_momentum']
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['momentum_ratio'] / df['high_low_range']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume and Range
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['average_daily_range'] = df['high_low_range'].rolling(window=5).mean()
    df['volatility_adjusted_momentum_confirmed'] = (
        df['volatility_adjusted_momentum'] * 
        df['volume_change'] * 
        df['trend_direction'] * 
        df['average_daily_range']
    )
    
    # Incorporate Price Oscillator
    df['short_term_ema'] = df['close'].ewm(span=5).mean()
    df['long_term_ema'] = df['close'].ewm(span=20).mean()
    df['price_oscillator'] = df['short_term_ema'] - df['long_term_ema']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=5).sum()
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Adjust by Price Volatility
    df['price_volatility'] = df['close'].rolling(window=5).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['price_volatility']) + 1
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Weighted Momentum
    df['log_return'] = df['close'] / df['close'].shift(5) - 1
    weights = [0.2, 0.18, 0.16, 0.14, 0.1]
    df['weighted_log_return'] = df['log_return'].rolling(window=5).apply(lambda x: sum(x * weights), raw=True)
    df['weighted_momentum_intraday'] = df['weighted_log_return'] / df['avg_intraday_volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['weighted_momentum_intraday'] - df['price_oscillator']
    df['volume_weighted_price_movement'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    df['combined_factor'] = (df['relative_strength'] / df['volume_weighted_price_movement']) + df['final_adjusted_cumulative_return']
    df['ema_difference'] = df['short_term_ema'] - df['long_term_ema']
    df['final_combined_factor'] = df['combined_factor'] / df['ema_difference']
    
    # Final Factor
    df['alpha_factor'] = (
        df['volatility_adjusted_momentum_confirmed'] + 
        (df['final_adjusted_cumulative_return'] * df['trend_direction'] * df['price_oscillator'])
    )
    
    return df['alpha_factor']
```
