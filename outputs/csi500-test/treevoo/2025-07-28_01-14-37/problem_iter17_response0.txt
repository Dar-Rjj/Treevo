```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_price * df['volume'])
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume and Range
    volume_change = df['volume'].diff()
    avg_daily_range = high_to_low_range.rolling(window=5).mean()
    combined_factor = adjusted_momentum * volume_change * trend_direction * avg_daily_range
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Weighted Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    exponential_weights = np.exp(-np.arange(1, 6)[::-1])
    weighted_log_returns = (log_returns.rolling(window=5).apply(lambda x: np.sum(x * exponential_weights), raw=True))
    adjusted_weighted_momentum = weighted_log_returns / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = adjusted_weighted_momentum - df['close'].ewm(span=5, adjust=False).mean()
    volume_weighted_price_movement = (df['close'] - df['close'].shift(5)).abs() / df['volume']
    combined_factor_2 = relative_strength / volume_weighted_price_movement + final_adjusted_cumulative_return * trend_direction
    ema_difference = short_term_ma - long_term_ma
    final_combined_factor = combined_factor_2 / ema_difference
    
    # Final Factor
    final_factor = (combined_factor + final_combined_factor * trend_direction)
    
    return final_factor
```
