```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term = df['close'].rolling(window=5).mean()
    long_term = df['close'].rolling(window=20).mean()

    # Calculate Short-Term and Long-Term Difference
    trend_direction = np.where(short_term - long_term > 0, 1, -1)

    # Subtract the open price from the close price on day t
    close_open_diff = df['close'] - df['open']

    # Calculate the range as the difference between the high and low prices on day t
    price_range = df['high'] - df['low']

    # Compare the current day's volume to the average volume over the past m days (m < t)
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_trend = np.where(df['volume'] > avg_volume, 1, -1)

    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)

    # Adjust Trend by Volume Change
    volume_adjusted_trend = np.where(volume_change > 10000, trend_direction * volume_change, 0)

    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)

    # Calculate Intraday Reversal Indicator
    intraday_reversal = (df['high'] - df['close'].shift(1)) + (df['close'].shift(1) - df['low'])

    # Calculate Intraday Turnover
    intraday_turnover = df['volume'] / df['volume'].shift(1)

    # Calculate Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= np.sqrt(df['volume'])
    weighted_momentum = close_price_change / (weighted_momentum * np.sqrt(df['volume']))

    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_momentum / high_low_range

    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()

    # Combine Factors
    combined_factors = (intraday_reversal + close_open_diff) * price_range

    # Confirm with Volume
    adjusted_momentum = volatility_adjusted_momentum * volume_adjusted_trend

    # Integrate Intraday Reversal Indicator
    intraday_reversal_indicator = intraday_reversal * intraday_volatility

    # Final Alpha Factor Synthesis
    final_alpha = (adjusted_momentum + combined_factors + intraday_reversal_indicator) * trend_direction

    return final_alpha
```
