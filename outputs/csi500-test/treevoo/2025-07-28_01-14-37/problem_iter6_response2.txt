```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    volume_weighted_movement = weighted_price.sum() / df['volume'].sum()
    
    # Calculate Relative Strength
    relative_strength = (momentum - sma_50) / volume_weighted_movement
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = ((df['high'] - df['low']) / df['high']) * 100
    
    # Calculate Average Daily Return over the last 10 days
    average_daily_return = df['close'].pct_change().rolling(window=10).mean()
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = relative_strength + combined_factors + average_daily_return
    
    return final_alpha_factor
```
