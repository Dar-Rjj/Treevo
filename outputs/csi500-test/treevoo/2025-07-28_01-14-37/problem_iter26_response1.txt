```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_term_window=5, long_term_window=20, lookback_period=20, volume_threshold=0.1):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=short_term_window).sum()

    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=short_term_window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=short_term_window).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_range']).rolling(window=short_term_window).mean()

    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()

    # Adjust by Intraday Volatility and Volume
    df['volume_weighted_momentum'] = (df['price_momentum'] * df['volume']) / df['intraday_volatility']

    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['trend_direction'] * np.where(df['volume_change'] > (volume_threshold * df['volume']), df['volume_change'], 0)

    # Validate Trend with Price Action
    df['price_range'] = df['high'] - df['low']
    df['confidence'] = np.where((df['price_range'] < df['short_term_ma']) & (df['price_range'] > df['long_term_ma']), 1, -1)
    df['validated_trend'] = df['trend_direction'] * df['confidence']

    # Calculate Relative Strength
    df['volume_weighted_price_movement'] = (df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (3 * df['volume'])
    df['relative_strength'] = (df['price_momentum'] - df['long_term_ma']) / df['volume_weighted_price_movement']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return'] + df['combined_intraday_factor'] + df['volume_weighted_momentum']

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['volume_adjusted_trend']

    # Introduce New Factor: Intraday Momentum
    df['intraday_log_return'] = np.log(df['high'] / df['low'])
    df['volume_adjusted_intraday_momentum'] = df['intraday_log_return'] * df['volume']

    # Final Alpha Factor
    df['final_alpha_factor'] += df['volume_adjusted_intraday_momentum']

    return df['final_alpha_factor']
```
