```python
def heuristics_v2(data):
    # Calculate the difference between high and low, and close and open
    data['hl_diff'] = data['high'] - data['low']
    data['co_diff'] = data['close'] - data['open']

    # Calculate the ratio of amount to volume
    data['amount_to_volume_ratio'] = data['amount'] / data['volume']

    # Calculate a weighted sum of the differences and the ratio
    # The weights are chosen heuristically; you can experiment with different values
    factor = 0.4 * data['hl_diff'] + 0.3 * data['co_diff'] + 0.3 * data['amount_to_volume_ratio']

    return factor
```
