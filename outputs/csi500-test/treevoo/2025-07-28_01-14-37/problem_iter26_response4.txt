```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the turnover rate
    turnover_rate = df['amount'] / df['volume']
    
    # Calculate the momentum factor based on the ratio of close to average price and turnover rate
    momentum_factor = (close_to_avg_ratio * turnover_rate).rolling(window=5).mean()
    
    # The final heuristic alpha factor is the difference between the momentum factor and its rolling mean
    alpha_factor = momentum_factor - momentum_factor.rolling(window=20).mean()
    
    return alpha_factor
```
