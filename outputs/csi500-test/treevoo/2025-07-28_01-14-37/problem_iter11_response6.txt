import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5):
    # Compute Close Price Difference
    df['close_diff'] = df['close'].diff(n)
    
    # Smoothing the Momentum
    df['smoothed_momentum'] = df['close_diff'].rolling(window=m).mean()
    
    # Volume Weighing Process
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['volume_adjusted_momentum'] = df['smoothed_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = 100 * (df['high'] - df['low']) / df['high']
    
    # Combine Factors
    df['combined_factors'] = (df['intraday_movement'] + df['close_to_open_return']) * df['price_range_ratio']
    
    # Integrate Volume Adjusted Momentum
    df['integrated_factor'] = df['combined_factors'] + df['volume_adjusted_momentum']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(m))
    df['momentum'] = df['log_return'].rolling(window=m).sum() / df['intraday_volatility']
    
    # Volume-Adjusted Intraday Return
    df['intraday_log_return'] = np.log(df['close'] / df['open'])
    df['volume_weighted_intraday_return'] = df['intraday_log_return'].ewm(span=n, adjust=False).mean()
    df['winsorized_intraday_return'] = df['volume_weighted_intraday_return'].clip(lower=df['volume_weighted_intraday_return'].quantile(0.01), upper=df['volume_weighted_intraday_return'].quantile(0.99))
    
    # Calculate Price Momentum
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_momentum_by_volume'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma']) / df['volume_weighted_price_movement']
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['volume_weighted_price_movement'] = df['weighted_price'].sum() / df['volume'].sum()
    
    # Combine
    df['final_factor'] = 0.5 * df['momentum'] + 0.5 * df['winsorized_intraday_return']
    
    # Final Factor
    df['skewness_adjusted_factor'] = (df['final_factor'] - df['final_factor'].mean()) / df['final_factor'].std()
    df['kurtosis_adjusted_factor'] = (df['skewness_adjusted_factor'] - df['skewness_adjusted_factor'].kurt()) * df['final_factor'].std() + df['final_factor'].mean()
    
    return df['kurtosis_adjusted_factor']
