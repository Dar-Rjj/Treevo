```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1

    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Adjust Momentum by Volume Change
    adjusted_momentum = trend_direction * volume_change

    # Calculate Relative Strength
    relative_strength = (short_term_ma - long_term_ma)

    # Calculate Volume-Weighted Price Movement
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume'])
    total_weighted_price = weighted_price.sum()
    total_volume = df['volume'].sum()
    volume_weighted_price_movement = total_weighted_price / total_volume

    # Final Relative Strength
    final_relative_strength = relative_strength / volume_weighted_price_movement

    # Calculate Volatility
    volatility = df['close'].rolling(window=20).std()

    # Adjust Relative Strength by Volatility
    enhanced_relative_strength = final_relative_strength * volatility

    return enhanced_relative_strength
```
