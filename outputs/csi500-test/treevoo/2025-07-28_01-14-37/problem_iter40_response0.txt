```python
import pandas as pd

def heuristics_v2(df, n=5, m=5, p=5):
    # Calculate the difference between the close and open prices
    df['close_open_diff'] = df['close'] - df['open']
    
    # Measure the volatility using the high and low prices
    df['range'] = df['high'] - df['low']
    
    # Evaluate the momentum by looking at recent close prices
    df['avg_close_n_days'] = df['close'].rolling(window=n).mean()
    df['momentum'] = df['close'] - df['avg_close_n_days']
    
    # Analyze trading volume for potential buying or selling pressure
    df['avg_volume_m_days'] = df['volume'].rolling(window=m).mean()
    df['volume_pressure'] = (df['volume'] > df['avg_volume_m_days']).astype(int)
    
    # Use the amount traded to estimate the value of transactions
    df['transaction_value'] = df['close'] * df['volume']
    df['avg_transaction_value_p_days'] = df['transaction_value'].rolling(window=p).mean()
    df['transaction_value_diff'] = df['transaction_value'] - df['avg_transaction_value_p_days']
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        df['close_open_diff'] + 
        df['range'] + 
        df['momentum'] + 
        df['volume_pressure'] * 10 +  # Weighted to emphasize volume pressure
        df['transaction_value_diff']
    )
    
    return alpha_factor
```
