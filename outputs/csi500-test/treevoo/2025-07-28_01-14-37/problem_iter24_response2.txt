```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = (close_price_change / (weighted_sum * df['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Smoothing the Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=20).mean()
    daily_volume_ratio = df['volume'] / mean_volume
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_volume_adjusted_momentum = volume_adjusted_momentum + combined_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()
    
    # Log Return
    log_return = np.log(df['close'] / df['close'].shift(5))
    sum_log_returns = log_return.rolling(window=5).sum()
    momentum = sum_log_returns / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    log_intraday_return = np.log(df['close'] / df['open'])
    winsorized_return = np.clip(log_intraday_return, -3, 3)
    volume_weighted_return = winsorized_return.ewm(span=5, adjust=False).mean()
    
    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_daily_range = high_to_low_range.rolling(window=5).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    final_factor = (
        integrated_volume_adjusted_momentum + momentum + volume_weighted_return
    ) * (volume_change * trend_direction * avg_daily_range) + vwap
    
    return final_factor
```
