```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    df['Cumulative_Return'] = df['Daily_Return'].rolling(window=window).sum()

    # Adjust by Volume
    df['Average_Volume'] = df['volume'].rolling(window=window).mean()
    df['Volume_Adjusted_Cumulative_Return'] = df['Cumulative_Return'] * df['Average_Volume']

    # Adjust by Price Volatility
    df['Close_Std'] = df['close'].rolling(window=window).std()
    df['Volatility_Adjusted_Return'] = (df['Volume_Adjusted_Cumulative_Return'] / df['Close_Std']) + 1

    # Calculate Intraday Return
    df['Intraday_Return'] = df['close'] / df['open'] - 1

    # Calculate Logarithmic Return
    small_constant = 1e-8
    df['Log_Intraday_Return'] = np.log(df['Intraday_Return'] + 1 + small_constant)

    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    df['Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    df['Weighted_Intraday_Range'] = df['Intraday_Range'] * df['volume']
    df['Combined_Intraday_Momentum'] = df['Weighted_Intraday_Return'] + df['Weighted_Intraday_Range']
    
    # Rescale for Skewness and Kurtosis
    skewness = df['Combined_Intraday_Momentum'].skew()
    kurtosis = df['Combined_Intraday_Momentum'].kurtosis()
    df['Combined_Intraday_Momentum_Scaled'] = (df['Combined_Intraday_Momentum'] - df['Combined_Intraday_Momentum'].mean()) / df['Combined_Intraday_Momentum'].std()
    df['Combined_Intraday_Momentum_Scaled'] = df['Combined_Intraday_Momentum_Scaled'] * (1 / (1 + skewness)) * (1 / (1 + kurtosis))

    # Calculate Volume-Weighted Price Movement
    df['Weighted_Price'] = (df['high'] + df['low'] + df['close']) / 3 * np.sqrt(df['volume'])
    total_volume = df['volume'].rolling(window=window).sum()
    df['Volume_Weighted_Price_Movement'] = df['Weighted_Price'].rolling(window=window).sum() / total_volume

    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(10)
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Relative_Strength'] = df['Momentum'] - df['EMA_50']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['Integrated_Momentum'] = df['Relative_Strength'] + df['Volatility_Adjusted_Return']
    df['Final_Factor'] = df['Integrated_Momentum'] + df['Volume_Weighted_Price_Movement']

    # Final Rescaling for Skewness and Kurtosis
    skewness = df['Final_Factor'].skew()
    kurtosis = df['Final_Factor'].kurtosis()
    df['Final_Factor_Scaled'] = (df['Final_Factor'] - df['Final_Factor'].mean()) / df['Final_Factor'].std()
    df['Final_Factor_Scaled'] = df['Final_Factor_Scaled'] * (1 / (1 + skewness)) * (1 / (1 + kurtosis))

    return df['Final_Factor_Scaled']
```
