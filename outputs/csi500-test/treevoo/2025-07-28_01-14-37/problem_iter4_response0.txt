```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m=5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Momentum
    close_prices = df['close']
    log_returns = np.log(close_prices) - np.log(close_prices.shift(m))
    momentum = log_returns.rolling(window=m).sum()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum / intraday_volatility
    
    return adjusted_momentum
```
