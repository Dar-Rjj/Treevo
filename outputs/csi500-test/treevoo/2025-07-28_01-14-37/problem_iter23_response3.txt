```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'].diff()

    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_momentum * df['volume'])

    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_to_low_range

    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1

    # Smoothing the Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()

    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Price Range Ratio
    price_range_ratio = (intraday_movement / df['high']) * 100

    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio

    # Integrate Volume Adjusted Momentum
    integrated_momentum = volume_adjusted_momentum + combined_factor

    # Calculate Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()

    # Momentum
    log_return = np.log(df['close']) - np.log(df['close'].shift(5))
    sum_log_returns = log_return / intraday_volatility

    # Volume-Adjusted Intraday Return
    log_intraday_return = np.log(df['close']) - np.log(df['open'])
    volume_weighted_return = log_intraday_return * daily_volume_ratio

    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    average_daily_range = high_to_low_range.rolling(window=5).mean()
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Final Alpha Factor
    final_factor = (integrated_momentum + sum_log_returns + volume_weighted_return) * (volume_change * trend_direction * average_daily_range) + vwap

    return final_factor
```
