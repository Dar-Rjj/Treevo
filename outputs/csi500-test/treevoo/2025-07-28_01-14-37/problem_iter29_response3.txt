```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the weighted average price
    wap = (df['high'] + df['low'] + 2 * df['close']) / 4
    
    # Calculate the momentum of the weighted average price
    momentum_wap = wap.pct_change(periods=10)
    
    # Calculate the volume to amount ratio
    volume_to_amount_ratio = df['volume'] / df['amount']
    
    # Calculate the factor as a combination of the above signals
    factor = (pct_change_close + momentum_wap) * volume_to_amount_ratio
    return factor
```
