```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the selling pressure
    df['selling_pressure'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the money flow
    df['money_flow'] = df['price_change'] * df['volume']
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['buying_pressure'] - df['selling_pressure']) * df['money_flow'] / df['vwap']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
