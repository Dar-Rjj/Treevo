import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = ((df['high'] + df['low'] + 2 * df['close']) / 4) * df['volume']
    weighted_momentum = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    intermediate_factor = adjusted_momentum * volume_change * trend_direction
    
    # Confirm with Price
    price_slope = []
    for i in range(len(df)):
        if i < 9:
            slope = np.nan
        else:
            x = np.arange(10)
            y = df['close'].iloc[i-9:i+1].values
            slope, _, _, _, _ = linregress(x, y)
        price_slope.append(slope)
    
    final_factor = intermediate_factor * price_slope
    
    return pd.Series(final_factor, index=df.index, name='Hybrid_Enhanced_Volatility_Adjusted_Momentum')
