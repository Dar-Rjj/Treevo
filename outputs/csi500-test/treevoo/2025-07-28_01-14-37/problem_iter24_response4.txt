```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['vol_adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Adjust for Skewness and Kurtosis
    skewness = skew(df['combined_intraday'].dropna())
    kurt = kurtosis(df['combined_intraday'].dropna())
    df['combined_intraday'] = (df['combined_intraday'] - np.mean(df['combined_intraday'])) / np.std(df['combined_intraday'])
    df['combined_intraday'] = (df['combined_intraday'] - skewness) / (1 + kurt)
    
    # Calculate Price Momentum
    short_term_window = 5
    long_term_window = 20
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_ma'] - df['long_term_ma']) / df['vol_adjusted_cumulative_return']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['vol_adjusted_cumulative_return'] + df['combined_intraday']
    
    # Adjust for Skewness and Kurtosis
    skewness = skew(df['integrated_factor'].dropna())
    kurt = kurtosis(df['integrated_factor'].dropna())
    df['integrated_factor'] = (df['integrated_factor'] - np.mean(df['integrated_factor'])) / np.std(df['integrated_factor'])
    df['integrated_factor'] = (df['integrated_factor'] - skewness) / (1 + kurt)
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    df['combined_factors'] = df['intraday_movement'] + df['close_to_open_return']
    
    # Weight by Price Range Ratio
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    df['weighted_combined_factors'] = df['combined_factors'] * df['price_range_ratio']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['weighted_combined_factors']
    
    return df['final_alpha_factor']
```
