```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Weighted Momentum
    df['weighted_momentum'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['price_change'] = df['close'].diff()
    df['momentum_factor'] = df['price_change'] / df['weighted_momentum']
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['momentum_factor'] / df['high_low_range']
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    df['trend_direction'] = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Incorporate Price Oscillator
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    df['price_oscillator'] = short_term_ema - long_term_ema
    
    # Combine Factors
    df['final_factor'] = df['volatility_adjusted_momentum'] * df['trend_direction'] * df['price_oscillator']
    
    return df['final_factor']
```
