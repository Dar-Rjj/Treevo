```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_diff = short_term_ma - long_term_ma
    trend_dir = np.sign(trend_diff)
    trend_dir_prev = trend_dir.shift(1)
    confirmed_trend = (trend_dir == trend_dir_prev).astype(int) * trend_dir

    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_threshold = 100000  # Example threshold
    volume_adj_trend = confirmed_trend * volume_change
    volume_adj_trend[(volume_change < volume_threshold) | (volume_change > -volume_threshold)] = 0

    # Calculate Relative Strength
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    volume_weighted_price = (weighted_price * df['volume']).sum() / df['volume'].sum()
    relative_strength = (short_term_ma - long_term_ma) / volume_weighted_price

    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_change = df['close'] - df['close'].shift(1)

    # Compute Weighted Momentum
    weighted_momentum_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_change / (weighted_momentum_sum * np.sqrt(df['volume']))

    # Adjust for Volatility
    volatility_adj_momentum = weighted_momentum / high_low_range

    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    adj_cumulative_return = cumulative_return * avg_volume

    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(20))
    momentum = log_return.rolling(window=20).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = (momentum - long_term_ma) + adj_cumulative_return

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * high_low_range

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adj_momentum = integrated_factor * volume_change

    # Calculate Intraday Reversal Indicator
    intraday_reversal = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])

    # Integrate Intraday Reversal Indicator
    integrated_intraday_reversal = intraday_reversal * intraday_volatility

    # Calculate Intraday Turnover
    intraday_turnover = df['volume'] / df['volume'].shift(1)

    # Adjust for Intraday Turnover
    final_alpha_factor = (volatility_adj_momentum + integrated_factor + combined_factors) * relative_strength
    final_alpha_factor += integrated_intraday_reversal
    final_alpha_factor *= intraday_turnover

    return final_alpha_factor.dropna()
```
