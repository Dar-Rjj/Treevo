```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_momentum_sum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    trend_5_day_ma = df['close'].rolling(window=5).mean()
    trend_20_day_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(trend_5_day_ma > trend_20_day_ma, 1, -1)
    
    # Smoothing the Momentum
    smoothed_momentum = adjusted_weighted_momentum.rolling(window=5).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].diff()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume_5_days
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_relative_strength = relative_strength + adjusted_cumulative_return
    
    # Integrate Volume Adjusted Momentum
    integrated_volume_adjusted_momentum = volume_adjusted_momentum + combined_factor
    
    # Calculate Volume-Adjusted Intraday Return
    logarithmic_return = np.log(df['close'] / df['open'])
    volume_weighted_log_return = logarithmic_return * daily_volume_ratio
    winsorized_log_return = volume_weighted_log_return.clip(lower=volume_weighted_log_return.quantile(0.05), upper=volume_weighted_log_return.quantile(0.95))
    
    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    average_daily_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    open_to_close_price_movement = (df['close'] - df['open']).rolling(window=5).mean()
    final_alpha_factor = (integrated_volume_adjusted_momentum + integrated_relative_strength + winsorized_log_return) * (volume_change * trend_direction * average_daily_range) + vwap + open_to_close_price_movement
    
    return final_alpha_factor
```
