```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_diff = short_term_ma - long_term_ma
    trend_direction = np.sign(trend_diff)
    confirmed_trend = np.where((trend_direction == trend_direction.shift(1)), 1, 0)

    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_adjusted_trend = confirmed_trend * volume_change
    volume_threshold = 1000000  # Example threshold
    volume_adjusted_trend = np.where(volume_change < volume_threshold, 0, volume_adjusted_trend)

    # Calculate Relative Strength
    weighted_price = (df['high'] + df['low'] + df['close'] * 2) / 4 * df['volume']
    volume_weighted_price = weighted_price.sum() / df['volume'].sum()
    relative_strength = (short_term_ma - long_term_ma) / volume_weighted_price

    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)

    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + df['close'] * 2) / 4 * np.sqrt(df['volume'])
    close_change_to_momentum = close_price_change / weighted_momentum

    # Adjust for Volatility
    adjusted_momentum = close_change_to_momentum / high_low_range

    # Calculate Intraday Volatility
    high_low_range_avg = high_low_range.rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume_5_days

    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(1))
    momentum = log_return.rolling(window=5).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    adjusted_momentum = momentum - long_term_ma
    integrated_factor = adjusted_momentum + adjusted_cumulative_return

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = high_low_range
    combined_factors_weighted = combined_factors * price_range_ratio

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum_by_volume = adjusted_momentum * volume_change

    # Calculate Intraday Reversal Indicator
    intraday_reversal_indicator = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])

    # Integrate Intraday Reversal Indicator
    integrated_reversal = intraday_reversal_indicator * high_low_range_avg

    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_momentum_by_volume + adjusted_cumulative_return + combined_factors_weighted) * relative_strength + integrated_reversal

    return final_alpha_factor
```
