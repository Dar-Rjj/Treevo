```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Exponential Moving Average (EMA)
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate Volume-Weighted Price Movement
    df['Weighted_Price'] = (df['high'] + df['low'] + df['close']) / 3 * (df['volume'] ** 0.5)
    total_volume_weighted_price = df['Weighted_Price'].rolling(window=50).sum()
    total_volume = df['volume'].rolling(window=50).sum()
    df['Volume_Weighted_Price_Movement'] = total_volume_weighted_price / total_volume
    
    # Calculate Relative Strength
    df['Relative_Strength'] = (df['Momentum'] - df['EMA_50']) / df['Volume_Weighted_Price_Movement']
    
    return df['Relative_Strength']
```
