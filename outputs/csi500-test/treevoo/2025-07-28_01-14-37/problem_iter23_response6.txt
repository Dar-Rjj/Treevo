```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= np.sqrt(df['volume'])
    weighted_momentum = close_price_change / (weighted_momentum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return over 5 days
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Momentum
    lookback_period = 20
    log_return = np.log(df['close'] / df['close'].shift(lookback_period))
    momentum = log_return.rolling(window=lookback_period).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_factor = relative_strength + final_adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close']).shift(-1)
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = (df['high'] - df['low']) / df['close']
    weighted_combined_factors = combined_factors * price_range_ratio
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    confirmed_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    volume_weighted_price = (df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (3 * df['volume'])
    volume_weighted_price_movement = volume_weighted_price.rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    relative_strength_final = (relative_strength / volume_weighted_price_movement)
    
    # Introduce New Factor: Intraday Momentum
    intraday_log_return = np.log(df['high'] / df['low'])
    intraday_momentum = intraday_log_return * df['volume']
    
    # Integrate Intraday Momentum with Other Factors
    integrated_intraday_factor = final_adjusted_cumulative_return + intraday_momentum
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = relative_strength_final + weighted_combined_factors + integrated_intraday_factor
    
    return final_alpha_factor
```
