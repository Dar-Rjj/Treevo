import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_sum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    combined_factor = (intraday_return * df['volume']) + high_low_range
    final_combined_factor = combined_factor / high_low_range
    
    # Calculate Price Momentum
    m = 5  # Lookback period
    log_returns = np.log(df['close'] / df['close'].shift(m))
    price_momentum = log_returns.rolling(window=m).sum()
    
    # Adjust by Intraday Volatility and Volume
    adjusted_cumulative_return = price_momentum * df['volume'] / high_low_range
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    confirmed_momentum = trend_direction * volume_change
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = final_combined_factor + adjusted_cumulative_return
    integrated_factor = (relative_strength + confirmed_momentum) / 2
    
    # Synthesize Final Alpha Factor
    volume_weighted_momentum = log_returns * df['volume']
    final_alpha_factor = integrated_factor + volume_weighted_momentum
    
    return final_alpha_factor
