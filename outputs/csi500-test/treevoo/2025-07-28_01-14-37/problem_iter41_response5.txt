```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, lookback_period=20):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['Cumulative_Return'] = df['Daily_Return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['Average_Volume'] = df['volume'].rolling(window=window).mean()
    df['Volume_Adjusted_Cumulative_Return'] = df['Cumulative_Return'] * df['Average_Volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    df['Average_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    
    # Combine Intraday Momentum and Volatility
    df['Combined_Intraday_Factor'] = (df['Volume_Weighted_Intraday_Return'].rolling(window=window).mean() +
                                      df['Average_Intraday_Volatility'])
    
    # Calculate Price Momentum
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['Price_Momentum'] = df['Log_Return'].rolling(window=lookback_period).sum()
    
    # Calculate Short-Term Moving Average
    df['Short_Term_MA'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average
    df['Long_Term_MA'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['Trend_Direction'] = np.where(df['Short_Term_MA'] > df['Long_Term_MA'], 1, -1)
    
    # Confirm with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Adjusted_Trend'] = df['Trend_Direction'] * df['Volume_Change']
    
    # Adjust by Intraday Volatility and Volume
    df['Adjusted_Price_Momentum'] = (df['Price_Momentum'] * df['volume']) / df['Average_Intraday_Volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['Relative_Strength'] = df['Price_Momentum'] - df['Long_Term_MA']
    df['Volume_Weighted_Price_Movement'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).sum() / df['volume'].sum()
    df['Integrated_Factor'] = df['Relative_Strength'] + df['Volume_Adjusted_Cumulative_Return'] / df['Volume_Weighted_Price_Movement']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Integrated_Factor'] + 
                                df['Combined_Intraday_Factor'] + 
                                df['Adjusted_Price_Momentum'])
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) / df['open']
    
    # Incorporate Intraday Reversal into Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Final_Alpha_Factor'] * 0.8 + df['Intraday_Reversal'] * 0.2)
    
    return df['Final_Alpha_Factor']
```
