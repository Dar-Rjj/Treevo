import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term (5-day) and Long-Term (20-day) Moving Averages
    short_mavg = df['close'].rolling(window=5).mean()
    long_mavg = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_mavg > long_mavg).astype(int)
    trend_direction[trend_direction == 0] = -1
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    price_momentum = short_mavg - long_mavg
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).sum() / df['volume'].sum()
    relative_strength = price_momentum / weighted_price
    
    # Incorporate Volatility
    historical_volatility = df['close'].rolling(window=20).std()
    final_factor = relative_strength * historical_volatility
    
    return final_factor
