import pandas as pd

def heuristics_v2(df):
    """
    The function takes a pandas DataFrame with columns: open, high, low, close, amount, volume.
    It generates a novel and interpretable alpha factor based on the following thought tree:
    1. Calculate the buying pressure using the difference between close and low.
    2. Calculate the selling pressure using the difference between high and close.
    3. Determine the net pressure by subtracting the selling pressure from the buying pressure.
    4. Normalize the net pressure by the average of the day's high and low to get a relative measure.
    5. Finally, multiply this relative measure by the volume to scale the factor by the trading activity.
    
    :param df: pandas DataFrame with (date) as index and columns: open, high, low, close, amount, volume
    :return: pandas Series indexed by (date) representing the factor values
    """
    buying_pressure = df['close'] - df['low']
    selling_pressure = df['high'] - df['close']
    net_pressure = buying_pressure - selling_pressure
    relative_net_pressure = net_pressure / ((df['high'] + df['low']) / 2)
    factor_values = relative_net_pressure * df['volume']
    
    return factor_values
