```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=n).mean()

    # Calculate Weighted Momentum
    log_returns = np.log(df['close']).diff()
    weighted_log_returns = log_returns.rolling(window=m).apply(lambda x: (x * np.exp(np.linspace(0, 1, m))).sum(), raw=True)
    weighted_momentum = weighted_log_returns / intraday_volatility

    # Determine Trend Direction
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    trend_direction = np.where(short_term_ema > long_term_ema, 1, -1)

    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum = trend_direction * weighted_momentum * volume_change

    return adjusted_momentum
```
