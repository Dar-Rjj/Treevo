```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df, window=5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=window).mean()
    df['volume_adjusted_cumulative_return'] = df['cumulative_return'] * avg_volume

    # Adjust by Price Volatility
    close_std = df['close'].rolling(window=window).std()
    df['volatility_adjusted_cumulative_return'] = (df['volume_adjusted_cumulative_return'] / close_std) + 1

    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    df['avg_intraday_volatility'] = intraday_volatility.rolling(window=window).mean()

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1

    # Calculate Logarithmic Return
    df['log_intraday_return'] = np.log(df['intraday_return'] + 1e-8)

    # Combine Intraday Momentum and Volatility
    weighted_intraday_return = df['intraday_return'] * df['volume']
    weighted_intraday_volatility = df['avg_intraday_volatility'] * df['volume']
    combined_intraday = (weighted_intraday_return + weighted_intraday_volatility) / (2 * df['volume'])

    # Adjust for Skewness and Kurtosis
    skewness = skew(combined_intraday)
    kurt = kurtosis(combined_intraday)
    combined_intraday = (combined_intraday - combined_intraday.mean()) / combined_intraday.std()
    combined_intraday = (combined_intraday - skewness) / kurt

    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) * np.sqrt(df['volume'])
    total_weighted_price = weighted_price.rolling(window=window).sum()
    total_volume = df['volume'].rolling(window=window).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume

    # Calculate Trend Direction
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    trend_direction = np.where(short_ema > long_ema, 1, -1)

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = trend_direction * volume_change
    final_momentum = (adjusted_momentum + df['volatility_adjusted_cumulative_return']) / 2

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = short_ema - long_ema
    integrated_factor = relative_strength + df['volatility_adjusted_cumulative_return']

    # Combine with Volume-Weighted Price Movement
    final_factor = (integrated_factor + volume_weighted_price_movement) / 2

    # Final Rescaling
    final_factor = (final_factor - final_factor.mean()) / final_factor.std()
    final_factor = (final_factor - skew(final_factor)) / kurt(final_factor)

    return final_factor
```
