```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, window=5):
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=n).mean()

    # Calculate Daily Return
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1

    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Adjust by Price Volatility
    df['price_volatility'] = df['close'].rolling(window=window).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / df['price_volatility']) + 1

    # Calculate Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(m))
    df['momentum'] = df['log_return'].rolling(window=m).sum()

    # Adjust by Intraday Volatility and Volume
    df['weighted_momentum'] = (df['momentum'] * df['average_volume']) / df['intraday_volatility']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['momentum'] - df['close'].rolling(window=50).mean()
    df['combined_factor'] = df['relative_strength'] + df['final_adjusted_cumulative_return']

    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] / df['close'].shift(1)) - 1

    # Combine Factors
    df['combined_sum'] = df['intraday_movement'] + df['close_to_open_return']
    df['price_range_ratio'] = df['high_low_range'] / df['close']
    df['weighted_combined_sum'] = df['combined_sum'] * df['price_range_ratio']

    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['weighted_combined_sum']

    # Incorporate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['log_return'] * df['volume']
    df['enhanced_alpha_factor'] = df['alpha_factor'] + df['volume_weighted_momentum']

    return df['enhanced_alpha_factor'].dropna()
```
