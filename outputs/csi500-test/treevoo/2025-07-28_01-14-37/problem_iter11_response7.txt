```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()

    # Calculate Intraday Return
    intraday_return = (df['close'] / df['open']) - 1

    # Calculate Logarithmic Return
    log_return = np.log(1 + intraday_return)

    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=n).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume

    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=n).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1

    # Combine Intraday Momentum and Volatility
    weighted_intraday_return = intraday_return * df['volume']
    weighted_intraday_range = (df['high'] - df['low']) * df['volume']
    combined_factor = weighted_intraday_return + weighted_intraday_range

    # Adjust for Skewness and Kurtosis
    skewness = combined_factor.skew()
    kurtosis = combined_factor.kurtosis()
    rescaled_combined_factor = (combined_factor - combined_factor.mean()) / combined_factor.std()
    rescaled_combined_factor = (rescaled_combined_factor - skewness) / kurtosis

    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) * np.sqrt(df['volume'])
    total_volume = df['volume'].rolling(window=n).sum()
    volume_weighted_price_movement = weighted_price.rolling(window=n).sum() / total_volume

    # Calculate Weighted Momentum
    lookback_log_return = np.log(df['close'] / df['close'].shift(m))
    exp_weights = np.exp(np.linspace(0, 1, m))
    weighted_log_returns = lookback_log_return.rolling(window=m).apply(lambda x: (x * exp_weights).sum(), raw=True)
    weighted_momentum = weighted_log_returns / intraday_volatility

    # Determine Trend Direction
    short_ema = df['close'].ewm(span=5).mean()
    long_ema = df['close'].ewm(span=20).mean()
    trend_direction = np.where(short_ema > long_ema, 1, -1)

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = trend_direction * volume_change
    final_factor = (weighted_momentum + adjusted_momentum) / 2

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = short_ema - long_ema
    integrated_factor = relative_strength / volume_weighted_price_movement + final_adjusted_cumulative_return

    # Final Rescaling
    skewness = integrated_factor.skew()
    kurtosis = integrated_factor.kurtosis()
    final_rescaled_factor = (integrated_factor - integrated_factor.mean()) / integrated_factor.std()
    final_rescaled_factor = (final_rescaled_factor - skewness) / kurtosis

    return final_rescaled_factor
```
