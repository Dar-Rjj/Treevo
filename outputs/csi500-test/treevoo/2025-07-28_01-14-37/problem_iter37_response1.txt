```python
import pandas as pd

def heuristics_v2(df):
    # Simple Price Momentum
    df['momentum_10'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['momentum_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # VWAP Momentum
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume'].cumsum())
    df['vwap_momentum_10'] = (df['vwap'] - df['vwap'].shift(10)) / df['vwap'].shift(10)
    
    # True Range and ATR
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # High-Low Breakouts
    df['high_10'] = df['high'].rolling(window=10).max()
    df['low_10'] = df['low'].rolling(window=10).min()
    df['breakout_high'] = (df['close'] > df['high_10']).astype(int)
    df['breakout_low'] = (df['close'] < df['low_10']).astype(int)
    
    # Closing Price vs. Previous Day's Range
    df['outside_range'] = ((df['close'] < df['low'].shift(1)) | (df['close'] > df['high'].shift(1))).astype(int)
    
    # Money Flow Index (MFI)
    df['tp'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['tp'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['tp'] > df['tp'].shift(1), 0).rolling(window=14).sum()
    df['negative_money_flow'] = df['money_flow'].where(df['tp'] < df['tp'].shift(1), 0).rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (df['positive_money_flow'] / df['negative_money_flow'])))
    
    # Chaikin Money Flow (CMF)
    df['mfm'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['cmf'] = (df['mfm'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Pivot Points
    df['pivot'] = (df['high'].shift(1) + df['low'].shift(1) + df['close'].shift(1)) / 3
    df['support1'] = (df['pivot'] * 2) - df['high'].shift(1)
    df['resistance1'] = (df['pivot'] * 2) - df['low'].shift(1)
    
    # Moving Averages as Dynamic Support/Resistance
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Volume Spike Analysis
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1) * 1.5).astype(int)
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_20'] +
        df['vwap_momentum_10'] +
        df['atr_14'] +
        df['breakout_high'] -
        df['breakout_low'] +
        df['outside_range'] +
        df['mfi'] +
        df['cmf'] +
        (df['close'] - df['sma_50']) / df['sma_50'] +
        df['open_close_spread'] +
        df['volume_spike']
    )
    
    return df['alpha_factor']
```
