```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= np.sqrt(df['volume'])
    weighted_momentum = close_price_change / (weighted_momentum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    confirmed_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    price_momentum = adjusted_momentum
    volume_weighted_price = (df['high'] + df['low'] + df['close']) * df['volume']
    volume_weighted_price = volume_weighted_price / df['volume']
    relative_strength = (price_momentum - long_term_ma) / volume_weighted_price
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    final_adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()
    
    # Adjust by Intraday Volatility and Volume
    weighted_momentum = momentum * df['volume'] / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = (relative_strength - df['close'].rolling(window=50).mean()) + final_adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (high_low_range / df['close'])
    
    # Introduce New Factor: Intraday Momentum
    intraday_log_return = np.log(df['high'] / df['low'])
    intraday_momentum = intraday_log_return * df['volume']
    
    # Integrate Intraday Momentum with Other Factors
    final_integrated_factor = integrated_factor + combined_factors + intraday_momentum
    
    return final_integrated_factor
```
