```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, m=20, n=10):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['volume_adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=n).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['high_low_range'] * df['volume']
    df['combined_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(m))
    df['sum_log_returns'] = df['log_return'].rolling(window=m).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['volume_weighted_momentum'] = df['sum_log_returns'] * df['volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=10).mean()
    df['long_term_ma'] = df['close'].rolling(window=30).mean()
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend'] = df['trend_direction'] * df['volume_change']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['volume_weighted_momentum'] + df['volume_adjusted_cumulative_return']
    df['final_combined_factor'] = df['combined_factor'] + df['adjusted_trend']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['relative_strength'] + df['final_combined_factor']
    
    return df['alpha_factor']
```
