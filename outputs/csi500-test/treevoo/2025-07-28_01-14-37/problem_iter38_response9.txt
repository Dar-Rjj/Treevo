```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the relative change in average price
    avg_price_change = (avg_price - avg_price.shift(1)) / avg_price.shift(1)
    
    # Calculate the ratio of volume to amount, which can be seen as a price proxy
    vol_to_amount_ratio = df['volume'] / df['amount']
    
    # Calculate the relative change in this ratio
    vol_to_amount_change = (vol_to_amount_ratio - vol_to_amount_ratio.shift(1)) / vol_to_amount_ratio.shift(1)
    
    # Combine the two relative changes into a single factor
    factor = avg_price_change * vol_to_amount_change
    
    return factor
```
