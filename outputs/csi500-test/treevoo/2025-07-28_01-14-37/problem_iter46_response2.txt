import pandas as pd

def heuristics_v2(df, window=5, n_days=5, m_days=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['volume_adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=n_days).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['combined_intraday_momentum'] = df['weighted_intraday_return'] + df['weighted_intraday_volatility']
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(m_days)).apply(np.log)
    df['price_momentum'] = df['log_return'].rolling(window=m_days).sum()
    
    # Calculate Short-Term Moving Average
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Moving Average
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_momentum'] = df['trend_direction'] * df['volume_change']
    
    # Adjust by Intraday Volatility and Volume
    df['adjusted_price_momentum'] = df['price_momentum'] * df['volume'] / df['intraday_volatility']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['volume_adjusted_cumulative_return'] + df['price_momentum']
    
    # Combine with Volume-Weighted Price Movement
    df['final_combined_factor'] = df['combined_intraday_momentum'] + df['confirmed_momentum']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['relative_strength'] + df['adjusted_price_momentum'] + df['final_combined_factor']
    
    return df['alpha_factor'].dropna()
