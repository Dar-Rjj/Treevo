```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow ratio inspired by the accumulation/distribution line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    
    # Calculate a simple moving average of the money flow volume to smooth out short-term fluctuations
    sma_money_flow_volume = money_flow_volume.rolling(window=14).mean()
    
    # Calculate the alpha factor as the difference between the current money flow volume and its moving average
    alpha_factor = money_flow_volume - sma_money_flow_volume
    
    return alpha_factor
```
