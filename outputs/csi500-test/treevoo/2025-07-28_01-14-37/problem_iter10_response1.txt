```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Logarithmic Return and adjust for bias
    log_return = np.log(intraday_return + 1) + 1e-8
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Momentum
    weighted_momentum = close_price_change / vwp
    
    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / high_low_range
    
    # Exponential Smoothing with decay factor
    decay_factor = 0.9
    smoothed_log_return = log_return.ewm(alpha=decay_factor).mean() * df['volume']
    smoothed_adjusted_momentum = adjusted_weighted_momentum.ewm(alpha=decay_factor).mean() * df['volume']
    
    # Combine Intraday and Volatility-Adjusted Momentum
    combined_momentum = 0.5 * smoothed_log_return + 0.5 * smoothed_adjusted_momentum
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    momentum_with_volume = combined_momentum * (trend_direction * volume_change)
    
    # Adjust for Skewness and Kurtosis
    skewness = (df['close'] - df['close'].mean()) ** 3 / (len(df) * (df['close'].std(ddof=0) ** 3))
    kurtosis = (df['close'] - df['close'].mean()) ** 4 / (len(df) * (df['close'].std(ddof=0) ** 4)) - 3
    rescaled_skewness = (skewness - skewness.min()) / (skewness.max() - skewness.min())
    rescaled_kurtosis = (kurtosis - kurtosis.min()) / (kurtosis.max() - kurtosis.min())
    final_factor = momentum_with_volume / high_low_range * (rescaled_skewness + rescaled_kurtosis)
    
    return final_factor
```
