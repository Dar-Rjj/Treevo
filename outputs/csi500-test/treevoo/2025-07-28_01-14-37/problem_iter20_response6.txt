import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, lookback_period=30):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']).rolling(window=window).mean()
    
    # Adjust for Skewness and Kurtosis
    df['skewness'] = df['combined_intraday_factor'].rolling(window=window).apply(lambda x: pd.Series(x).skew(), raw=True)
    df['kurtosis'] = df['combined_intraday_factor'].rolling(window=window).apply(lambda x: pd.Series(x).kurtosis(), raw=True)
    df['adjusted_intraday_factor'] = (df['combined_intraday_factor'] - df['skewness']) / (1 + df['kurtosis'])
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['adjusted_price_momentum'] = df['price_momentum'] * df['average_volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=50).mean()
    df['trend_direction'] = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_trend'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['total_weighted_price'] = df['weighted_price'].rolling(window=lookback_period).sum()
    df['total_volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['volume_weighted_price_movement'] = df['total_weighted_price'] / df['total_volume']
    df['relative_strength'] = (df['price_momentum'] - long_term_ma) / df['volume_weighted_price_movement']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['ema_momentum'] = df['price_momentum'].ewm(span=lookback_period, adjust=False).mean()
    df['integrated_factor'] = (df['relative_strength'] - df['ema_momentum']) + df['adjusted_cumulative_return']
    
    # Final Rescaling
    df['final_skewness'] = df['integrated_factor'].rolling(window=window).apply(lambda x: pd.Series(x).skew(), raw=True)
    df['final_kurtosis'] = df['integrated_factor'].rolling(window=window).apply(lambda x: pd.Series(x).kurtosis(), raw=True)
    df['rescaled_factor'] = (df['integrated_factor'] - df['final_skewness']) / (1 + df['final_kurtosis'])
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['relative_strength'] + df['rescaled_factor']
