```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']

    # Adjust for Skewness and Kurtosis
    skewness = df['combined_intraday_factor'].skew()
    kurtosis = df['combined_intraday_factor'].kurtosis()
    df['rescaled_combined_intraday_factor'] = (df['combined_intraday_factor'] - skewness) / kurtosis

    # Calculate Price Momentum
    m = 20
    df['log_return'] = np.log(df['close'] / df['close'].shift(m))
    df['price_momentum'] = df['log_return'].rolling(window=m).sum()

    # Calculate Short-Term and Long-Term Moving Averages
    short_window = 5
    long_window = 20
    df['short_mavg'] = df['close'].rolling(window=short_window).mean()
    df['long_mavg'] = df['close'].rolling(window=long_window).mean()

    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_mavg'] > df['long_mavg'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend'] = df['trend_direction'] * df['volume_change']

    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - df['long_mavg']) / df['volume_weighted_price_movement']
    df['volume_weighted_price_movement'] = (df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return']
    df['final_integrated_factor'] = (df['integrated_factor'] + df['rescaled_combined_intraday_factor']) * df['volume_weighted_price_movement']

    # Adjust for Skewness and Kurtosis
    skewness_final = df['final_integrated_factor'].skew()
    kurtosis_final = df['final_integrated_factor'].kurtosis()
    df['rescaled_final_integrated_factor'] = (df['final_integrated_factor'] - skewness_final) / kurtosis_final

    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) / df['open']

    # Incorporate Intraday Reversal into Final Alpha Factor
    df['final_alpha_factor'] = df['rescaled_final_integrated_factor'] + df['intraday_reversal']

    return df['final_alpha_factor']
```
