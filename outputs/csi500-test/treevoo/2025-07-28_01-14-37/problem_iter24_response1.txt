```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust Momentum by Volume Change
    adjusted_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    price_momentum = short_term_ma - long_term_ma
    weighted_price = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).sum() / df['volume'].sum()
    relative_strength = (price_momentum / weighted_price)
    
    # Incorporate Price Volatility
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range = daily_price_range.rolling(window=20).mean()
    volatility_adjusted_relative_strength = relative_strength * (daily_price_range / avg_daily_price_range)
    
    return volatility_adjusted_relative_strength
```
