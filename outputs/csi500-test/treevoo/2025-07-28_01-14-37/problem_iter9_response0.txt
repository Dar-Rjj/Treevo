```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df, n_days=20, m_days=5, smoothing_factor=0.2, upper_bound=3, lower_bound=-3):
    # Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n_days).mean()
    
    # Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(m_days))
    df['sum_log_returns'] = df['log_return'].rolling(window=m_days).sum()
    df['momentum'] = df['sum_log_returns'] / df['avg_high_low_range']
    
    # Volume-Adjusted Intraday Return
    df['intraday_log_return'] = np.log(df['close']) - np.log(df['open'])
    
    # Winsorize the volume
    df['volume_winsorized'] = df['volume'].clip(lower=df['volume'].quantile(0.05), upper=df['volume'].quantile(0.95))
    
    # Exponential Smoothing
    df['volume_weighted_intraday_return'] = df['intraday_log_return'] * df['volume_winsorized']
    df['smoothed_intraday_return'] = df['volume_weighted_intraday_return'].ewm(alpha=smoothing_factor).mean()
    
    # Combine
    df['weighted_sum'] = 0.5 * df['momentum'] + 0.5 * df['smoothed_intraday_return']
    
    # Adjust for Skewness and Kurtosis
    weighted_sum = df['weighted_sum'].dropna()
    skew_adj = skew(weighted_sum)
    kurt_adj = kurtosis(weighted_sum, fisher=False)
    
    df['final_factor'] = (df['weighted_sum'] - skew_adj) / (kurt_adj - 3)
    
    return df['final_factor']
```
