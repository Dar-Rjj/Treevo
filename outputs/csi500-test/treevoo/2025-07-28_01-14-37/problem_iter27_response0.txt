import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term (5-day) and Long-Term (20-day) Moving Averages
    df['short_term_MA'] = df['close'].rolling(window=5).mean()
    df['long_term_MA'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_diff'] = df['short_term_MA'] - df['long_term_MA']
    df['trend_direction'] = df['trend_diff'].apply(lambda x: 1 if x > 0 else -1)
    
    # Confirm with Previous Day's Trend
    df['previous_trend'] = df['trend_direction'].shift(1)
    df['confirmed_trend'] = ((df['trend_direction'] == df['previous_trend']).astype(int) * df['trend_direction'])
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Trend by Volume Change
    df['adjusted_trend'] = df['confirmed_trend'] * df['volume_change']
    threshold = 0.01 * df['volume'].mean()  # Example threshold, 1% of average volume
    df['adjusted_trend'] = df['adjusted_trend'].apply(lambda x: x if abs(x) > threshold else 0)
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['total_weighted_price'] = df['weighted_price'].rolling(window=5).sum()
    df['total_volume'] = df['volume'].rolling(window=5).sum()
    df['volume_weighted_price_movement'] = df['total_weighted_price'] / df['total_volume']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_MA'] - df['long_term_MA']) / df['volume_weighted_price_movement']
    
    # Return the factor values
    return df['relative_strength'].dropna()
