```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term (5-day) and Long-Term (20-day) Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Subtract today's SMA from yesterday's SMA to identify momentum shifts
    df['momentum_shift'] = df['SMA_5'] - df['SMA_5'].shift(1)
    
    # Calculate daily volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the ratio of amount to volume
    df['avg_transaction_size'] = df['amount'] / df['volume']
    
    # Determine trend direction
    df['trend_direction'] = (df['SMA_5'] - df['SMA_20']).apply(lambda x: 1 if x > 0 else -1)
    df['trend_continuation'] = (df['trend_direction'] * df['trend_direction'].shift(1)).apply(lambda x: 1 if x == 1 else 0)
    
    # Adjust trend by volume change with a threshold
    df['adjusted_trend'] = df['trend_continuation'] * df['volume_change'].apply(lambda x: x if abs(x) > 0.01 * df['volume'].mean() else 0)
    
    # Calculate relative strength
    df['volume_weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['total_volume_weighted_price'] = df['volume_weighted_price'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['relative_strength'] = (df['SMA_5'] - df['SMA_20']) / df['total_volume_weighted_price']
    
    # Identify volatility patterns
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Explore price oscillators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['price_oscillator'] = df['SMA_50'] - df['SMA_200']
    
    # Construct reversal signals
    df['days_since_highest_high'] = (df['high'].cummax() != df['high']).astype(int).cumsum()
    df['days_since_lowest_low'] = (df['low'].cummin() == df['low']).astype(int).cumsum()
    
    # Evaluate price action relative to moving averages
    df['distance_to_SMA_5'] = df['close'] - df['SMA_5']
    
    # Calculate intraday movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Combine factors
    df['combined_factors'] = (df['intraday_movement'] + (df['close'] - df['open'])) * (df['high'] - df['low'])
    
    # Confirm with volume
    df['volume_adjusted_momentum'] = df['adjusted_trend'] * df['volume_change']
    
    # Calculate intraday reversal indicator
    df['intraday_reversal_indicator'] = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])
    
    # Integrate intraday reversal indicator
    df['smoothed_intraday_reversal'] = df['intraday_reversal_indicator'] * df['intraday_volatility']
    df['smoothed_intraday_reversal'] = df['smoothed_intraday_reversal'].rolling(window=3).mean()
    
    # Synthesize final alpha factor
    df['final_alpha_factor'] = (
        df['volume_adjusted_momentum'] 
        + df['relative_strength'] 
        + df['smoothed_intraday_reversal']
        + df['combined_factors']
    )
    
    return df['final_alpha_factor']
```
