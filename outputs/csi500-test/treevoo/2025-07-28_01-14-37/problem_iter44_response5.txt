import pandas as pd

def heuristics_v2(df, window=5, lookback=20, short_term=5, long_term=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['avg_volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['avg_volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_volatility']
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(lookback)).apply(lambda x: math.log(x))
    df['price_momentum'] = df['log_return'].rolling(window=lookback).sum()
    
    # Calculate Short-Term Moving Average
    df['short_term_ma'] = df['close'].rolling(window=short_term).mean()
    
    # Calculate Long-Term Moving Average
    df['long_term_ma'] = df['close'].rolling(window=long_term).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['trend_direction'] * df['volume_change']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['adjusted_cumulative_return'] + df['price_momentum']
    df['volume_weighted_price_movement'] = df['volume_adjusted_trend'] + df['combined_intraday_factor']
    df['final_factor'] = df['relative_strength'] + df['volume_weighted_price_movement']
    
    # Synthesize Final Alpha Factor
    df['volume_weighted_momentum'] = (df['close'] / df['close'].shift(lookback)).apply(lambda x: math.log(x)) * df['volume']
    df['final_alpha_factor'] = df['final_factor'] + df['volume_weighted_momentum']
    
    # Incorporate Intraday Reversal into Final Alpha Factor
    df['intraday_reversal'] = -df['intraday_return']
    df['final_alpha_factor'] = (df['final_alpha_factor'] + df['intraday_reversal']) / 2
    
    return df['final_alpha_factor']
