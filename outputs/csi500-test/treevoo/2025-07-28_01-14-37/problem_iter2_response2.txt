```python
import pandas as pd
import numpy as np
from scipy.stats import mstats, skew, kurtosis

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Logarithmic Return and Adjust for Bias
    df['log_intraday_return'] = np.log(df['intraday_return'] + 1e-8)
    # Winsorize to remove extreme values
    df['log_intraday_return'] = mstats.winsorize(df['log_intraday_return'], limits=[0.01, 0.01])
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Close Price Change
    df['close_price_change'] = df['close'].diff()
    
    # Exponential Smoothing for Volume
    decay_factor = 0.95
    df['volume_weight'] = df['volume'].ewm(alpha=decay_factor, adjust=False).mean()
    
    # Weighted Sum of Intraday and High-to-Low Factors
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_weight']
    df['weighted_high_to_low_range'] = df['high_to_low_range'] * df['volume_weight']
    df['combined_factor'] = (df['weighted_intraday_return'] + df['weighted_high_to_low_range']) / 2
    
    # Adjust for Skewness
    skewness = skew(df['combined_factor'])
    df['skewness_adjusted_factor'] = (df['combined_factor'] - np.mean(df['combined_factor'])) / np.std(df['combined_factor'])
    
    # Adjust for Kurtosis
    kurt = kurtosis(df['skewness_adjusted_factor'])
    df['kurtosis_adjusted_factor'] = df['skewness_adjusted_factor'] / np.sqrt(kurt)
    
    # Normalize by High-to-Low Range
    df['final_factor'] = df['kurtosis_adjusted_factor'] / df['high_to_low_range']
    
    return df['final_factor']
```
