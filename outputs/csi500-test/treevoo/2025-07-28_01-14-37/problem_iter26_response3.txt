import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)

    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'].diff()

    # Compute Weighted Momentum
    weighted_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_sum * np.sqrt(df['volume']))

    # Adjust for Volatility
    volatility_adjusted_momentum = weighted_momentum / high_to_low_range

    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum = volatility_adjusted_momentum * (trend_direction * volume_change)

    # Calculate Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume

    # Calculate Momentum
    lookback_period = 5
    log_return = np.log(df['close'] / df['close'].shift(lookback_period))
    momentum = log_return.rolling(window=lookback_period).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - long_term_ma
    combined_return = adjusted_cumulative_return + relative_strength

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * (high_to_low_range / df['close'])

    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_momentum + combined_return + combined_factors) * relative_strength

    return final_alpha_factor
