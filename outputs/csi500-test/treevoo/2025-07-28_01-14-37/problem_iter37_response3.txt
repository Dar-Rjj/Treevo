import pandas as pd

def heuristics_v2(df, window=5, n_days=5, m_days=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['vol_adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n_days).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']

    df['long_term_ma'] = df['close'].rolling(window=20).mean()

    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1

    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    df['adjusted_price_momentum'] = df['price_momentum'] * df['volume'] / df['avg_intraday_range']
