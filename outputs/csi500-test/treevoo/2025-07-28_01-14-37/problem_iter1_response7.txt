```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Calculate Exponential Weighted Average Returns
    decay_factor = 0.9  # Example decay factor, can be adjusted
    df['ewa_returns'] = df['daily_returns'].ewm(alpha=decay_factor, adjust=False).mean()
    
    # Volume Adjustment
    volume_window = 30  # Time period for calculating average volume, can be adjusted
    df['average_volume'] = df['volume'].rolling(window=volume_window).mean()
    df['volume_adjustment_ratio'] = df['volume'] / df['average_volume']
    df['adjusted_ema'] = df['ewa_returns'] * df['volume_adjustment_ratio']
    
    # Return the final alpha factor
    return df['adjusted_ema']
```
