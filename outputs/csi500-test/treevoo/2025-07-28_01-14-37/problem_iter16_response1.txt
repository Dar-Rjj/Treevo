```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close Price Difference
    n = 5  # lookback period for momentum
    df['close_diff'] = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    m = 10  # moving average window
    df['smoothed_momentum'] = df['close_diff'].rolling(window=m).mean()
    
    # Volume Weighing Process
    n_volume = 30  # lookback period for mean volume
    df['mean_volume'] = df['volume'].rolling(window=n_volume).mean()
    df['volume_ratio'] = df['volume'] / df['mean_volume']
    df['adjusted_momentum'] = df['smoothed_momentum'] * df['volume_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Logarithmic Return
    small_constant = 1e-8  # to avoid log(0)
    df['log_return'] = np.log(df['intraday_return'] + small_constant)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_range']) / df['volume']
    
    # Integrate Volume Adjusted Momentum
    df['integrated_momentum'] = df['combined_factor'] + df['adjusted_momentum']
    
    # Calculate Short-Term Momentum
    short_term_window = 5
    df['short_term_momentum'] = df['close'].rolling(window=short_term_window).mean()
    
    # Calculate Long-Term Momentum
    long_term_window = 20
    df['long_term_momentum'] = df['close'].rolling(window=long_term_window).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_momentum'] > df['long_term_momentum'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_trend'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['short_term_momentum'] - df['long_term_momentum'])
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    total_volume = df['volume'].sum()
    df['volume_weighted_movement'] = df['weighted_price'].sum() / total_volume
    
    # Normalize relative strength by volume-weighted price movement
    df['normalized_relative_strength'] = df['relative_strength'] / df['volume_weighted_movement']
    
    # Final Alpha Factor
    alpha_factor = 0.5 * df['integrated_momentum'] + 0.5 * df['normalized_relative_strength']
    
    return alpha_factor
```
