```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_change / (weighted_price * df['volume'])
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Smoothing the Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / mean_volume
    volume_adjusted_momentum = smoothed_momentum * volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = ((df['high'] - df['low']) / df['high']) * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_momentum = volume_adjusted_momentum + combined_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Momentum (Log Return)
    log_return = np.log(df['close']).diff(20)
    momentum = log_return.sum() / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    intraday_log_return = np.log(df['close'] / df['open'])
    winsorized_intraday_return = np.clip(intraday_log_return, a_min=-3, a_max=3) * volume_ratio
    
    # Confirm with Volume and Range
    volume_change = df['volume'].diff()
    avg_daily_range = high_low_range.rolling(window=5).mean()
    
    # Calculate VWAP
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Alpha Factor
    final_factor = (integrated_momentum + momentum + winsorized_intraday_return) * (volume_change * trend_direction * avg_daily_range) + vwap
    
    # Calculate Relative Strength
    price_momentum = df['close'].pct_change(20)
    volume_weighted_price_movement = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).sum() / df['volume'].sum()
    relative_strength = (price_momentum - long_term_ma) / volume_weighted_price_movement
    
    # Final Factor
    final_factor = final_factor * relative_strength
    
    return final_factor
```
