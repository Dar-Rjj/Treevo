```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term Moving Average (5-day)
    short_term_ma = df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Moving Average (20-day)
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust Momentum by Volume Change
    adjusted_momentum = trend_direction * volume_change
    
    return adjusted_momentum
```
