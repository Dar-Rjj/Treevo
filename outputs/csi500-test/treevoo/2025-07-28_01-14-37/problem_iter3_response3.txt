```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a modified RSI with volume and amount influence.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate Daily Price Change
    price_changes = df['close'].diff()
    
    # Separate Positive and Negative Gains
    positive_gains = price_changes.where(price_changes > 0, 0)
    negative_gains = -price_changes.where(price_changes < 0, 0)
    
    # Weighted Sums of Positive and Negative Gains
    weighted_positive_gains = (positive_gains * df['volume'] * df['amount']).rolling(window=14).sum()
    weighted_negative_gains = (negative_gains * df['volume'] * df['amount']).rolling(window=14).sum()
    
    # Compute RS (Relative Strength)
    rs = weighted_positive_gains / weighted_negative_gains
    
    # Apply RSI Formula
    rsi = 100 - (100 / (1 + rs))
    
    return rsi
```
