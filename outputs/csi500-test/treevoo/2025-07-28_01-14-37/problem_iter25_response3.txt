```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    close_price_change = df['close'].diff()
    weighted_momentum_value = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_volatility_momentum = weighted_momentum_value / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    
    # Calculate Cumulative Return
    daily_return = df['close'].pct_change()
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust Cumulative Return by Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return / intraday_volatility
    
    # Incorporate Open Price Momentum
    open_price_change = df['open'].diff()
    open_price_momentum = open_price_change / (df['open'] * df['volume'])
    
    # Enhance with Intraday Price Behavior
    intraday_price_range = df['high'] - df['low']
    intraday_price_mean = (df['high'] + df['low'] + df['close']) / 3
    intraday_price_variability = intraday_price_range / intraday_price_mean
    
    # Final Factor
    final_factor = (
        (adjusted_volatility_momentum * volume_change * trend_direction) +
        (adjusted_cumulative_return * trend_direction * intraday_price_variability) +
        open_price_momentum
    )
    
    return final_factor
```
