```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_momentum_sum * df['volume'])
    
    # Adjust for Volatility
    adjusted_volatility_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume and Range
    volume_change = df['volume'].diff()
    average_daily_range = high_to_low_range.rolling(window=5).mean()
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Factor
    final_factor = (adjusted_volatility_momentum * volume_change * trend_direction * average_daily_range) + vwap
    
    return final_factor
```
