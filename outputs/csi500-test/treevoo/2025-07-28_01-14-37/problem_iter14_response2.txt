```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series of alpha factor values indexed by (date).
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the transaction value per unit volume
    value_per_volume = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    alpha_factor = close_change * close_to_avg_ratio * value_per_volume
    
    return alpha_factor
```
