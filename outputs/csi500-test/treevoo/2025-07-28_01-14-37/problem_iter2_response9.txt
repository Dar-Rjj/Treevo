import pandas as pd

def heuristics_v2(df, n_days=20, m_days=10):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n_days).mean()

    # Calculate Raw Momentum
    close_difference = df['close'].diff(m_days)
    smoothed_momentum = close_difference.rolling(window=m_days).sum() / m_days

    # Adjust by Intraday Volatility
    adjusted_momentum = smoothed_momentum / intraday_volatility

    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=n_days).mean()
    daily_volume_ratio = df['volume'] / mean_volume
    volume_adjusted_momentum = adjusted_momentum * daily_volume_ratio

    # Momentum Signal
    momentum_signal = volume_adjusted_momentum.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))

    return momentum_signal
