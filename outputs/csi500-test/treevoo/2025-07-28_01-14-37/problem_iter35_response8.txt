```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Moving Averages
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    momentum_by_volume = trend_direction * volume_change
    
    # Calculate Relative Strength
    relative_strength = (short_term_ma - long_term_ma)
    
    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = weighted_price.rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    relative_strength /= volume_weighted_price_movement
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return Over 5 Days
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Combine Intraday Momentum and Volatility
    weighted_intraday_return = intraday_return * df['volume']
    weighted_intraday_range = intraday_range * df['volume']
    combined_factor = weighted_intraday_return + weighted_intraday_range
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    momentum_minus_50_sma = short_term_ma - long_term_ma
    integrated_relative_strength = relative_strength + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Open-to-Low Return
    open_to_low_return = df['open'] - df['low']
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return + open_to_low_return
    price_range_ratio = (df['high'] - df['low']) / df['close']
    combined_factors *= price_range_ratio
    
    # Confirm with Volume
    adjusted_momentum_by_volume = trend_direction * volume_change
    
    # Calculate Intraday Reversal Indicator
    intraday_reversal_indicator = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])
    
    # Integrate Intraday Reversal Indicator
    intraday_reversal_adjusted = intraday_reversal_indicator * intraday_volatility
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (integrated_relative_strength + combined_factors + intraday_reversal_adjusted) * relative_strength + intraday_reversal_adjusted
    
    return final_alpha_factor
```
