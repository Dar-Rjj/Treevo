```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df, n_days=10, rsi_period=14, decay_factor=0.9):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    avg_intraday_volatility = intraday_volatility.rolling(window=n_days).mean()
    
    # Calculate Price Changes
    price_changes = df['close'].diff()
    
    # Separate Positive and Negative Gains
    positive_gains = price_changes.where(price_changes > 0, 0)
    negative_gains = abs(price_changes.where(price_changes < 0, 0))
    
    # Sum of Positive and Negative Gains over a period
    sum_positive_gains = positive_gains.rolling(window=rsi_period).sum()
    sum_negative_gains = negative_gains.rolling(window=rsi_period).sum()
    
    # Adjust for Volume and Amount
    weighted_positive_gains = (positive_gains * df['volume'] * df['amount']).rolling(window=rsi_period).sum()
    weighted_negative_gains = (negative_gains * df['volume'] * df['amount']).rolling(window=rsi_period).sum()
    
    # Calculate RSI
    rs = weighted_positive_gains / weighted_negative_gains
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Return
    log_return = np.log(intraday_return + 1e-6)  # Add small constant to avoid log(0)
    log_return = np.clip(log_return, -3, 3)  # Winsorize
    
    # Exponential Smoothing
    adjusted_momentum = log_return.ewm(alpha=decay_factor, adjust=False).mean()
    smoothed_rsi = rsi.ewm(alpha=decay_factor, adjust=False).mean()
    
    # Combine Adjusted Momentum and RSI
    combined_factor = 0.5 * adjusted_momentum + 0.5 * smoothed_rsi
    
    # Adjust for Skewness
    factor_skewness = skew(combined_factor.dropna())
    rescaled_factor = combined_factor - factor_skewness
    
    # Adjust for Kurtosis
    factor_kurtosis = kurtosis(rescaled_factor.dropna())
    final_factor = rescaled_factor / (1 + factor_kurtosis)
    
    return final_factor
```
