```python
import pandas as pd
import numpy as np
from scipy.stats.mstats import winsorize

def heuristics_v2(df, n=30, m=10):
    # Compute Close Price Difference
    close_diff = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    smoothed_momentum = close_diff.rolling(m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(n).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_factor = combined_factors + volume_adjusted_momentum
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(n).mean()
    
    # Momentum
    log_return = np.log(df['close'] / df['close'].shift(m))
    sum_log_returns = log_return.rolling(m).sum()
    momentum = sum_log_returns / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    intraday_log_return = np.log(df['close'] / df['open'])
    winsorized_intraday_return = winsorize(intraday_log_return, limits=[0.05, 0.05])
    volume_weighted_intraday_return = winsorized_intraday_return * daily_volume_ratio
    exponential_smoothing = volume_weighted_intraday_return.ewm(span=n).mean()
    
    # Final Factor
    final_factor = integrated_factor + momentum + exponential_smoothing
    
    # Adjust for Skewness and Kurtosis
    skewness = final_factor.skew()
    kurtosis = final_factor.kurtosis()
    final_factor = (final_factor - final_factor.mean()) / final_factor.std()
    final_factor = final_factor * (1 - skewness / 3) * (1 - kurtosis / 6)
    
    return final_factor
```
