import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, lookback_period=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']) / df['volume']
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['price_momentum'] = df['log_return'].rolling(window=lookback_period).sum()
    
    # Calculate Short-Term and Long-Term Moving Averages
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_trend_direction'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['price_momentum'] - df['long_term_ma']) / df['volume_weighted_price_movement']
    
    # Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) * df['volume']
    df['total_weighted_price'] = df['weighted_price'].rolling(window=window).sum()
    df['total_volume'] = df['volume'].rolling(window=window).sum()
    df['volume_weighted_price_movement'] = df['total_weighted_price'] / df['total_volume']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['integrated_factor'] = df['adjusted_cumulative_return'] + df['relative_strength']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) / df['open']
    
    # Incorporate Intraday Reversal into Final Alpha Factor
    df['final_alpha_factor'] = 0.8 * df['integrated_factor'] + 0.2 * df['intraday_reversal']
    
    # Synthesize Final Alpha Factor
    df['log_return_long'] = np.log(df['close'] / df['close'].shift(lookback_period))
    df['volume_weighted_momentum'] = df['log_return_long'] * df['volume']
    df['final_alpha_factor'] += df['volume_weighted_momentum']
    
    return df['final_alpha_factor']
