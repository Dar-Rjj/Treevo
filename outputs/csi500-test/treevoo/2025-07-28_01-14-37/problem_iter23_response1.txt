def heuristics_v2(data):
    # Calculate the return from open to close
    data['intraday_return'] = (data['close'] - data['open']) / data['open']
    
    # Calculate the range of the day
    data['range'] = (data['high'] - data['low']) / data['low']
    
    # Calculate the volume weighted price
    data['vwp'] = (data['amount'] / data['volume'])
    
    # Calculate the momentum factor
    data['momentum'] = data['vwp'].pct_change()
    
    # Calculate the heuristic factor
    data['heuristic_factor'] = (data['intraday_return'] + data['range']) * data['momentum']
    
    # Return the heuristic factor as a pandas Series
    return data['heuristic_factor']
