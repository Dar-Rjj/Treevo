```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Compute Close Price Difference
    df['CloseDiff'] = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    df['MomentumMA'] = df['CloseDiff'].rolling(window=m).mean()
    
    # Volume Weighing Process
    df['MeanVolume'] = df['volume'].rolling(window=n).mean()
    df['VolumeRatio'] = df['volume'] / df['MeanVolume']
    df['AdjustedMomentum'] = df['MomentumMA'] * df['VolumeRatio']
    
    # Calculate Intraday Movement
    df['IntradayMovement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    df['PriceRangeRatio'] = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    df['CombinedFactor'] = (df['IntradayMovement'] + df['CloseToOpenReturn']) * df['PriceRangeRatio']
    
    # Integrate Volume Adjusted Momentum
    df['IntegratedMomentum'] = df['CombinedFactor'] + df['AdjustedMomentum']
    
    # Momentum Signal
    df['MomentumSignal'] = df['IntegratedMomentum'].apply(lambda x: 'Positive' if x > 0 else 'Negative or Neutral')
    
    # Interpreting the Alpha Factor
    alpha_factor = df['IntegratedMomentum']
    
    return alpha_factor
```
