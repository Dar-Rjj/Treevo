import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_threshold = 0.1 * df['volume'].shift(1)
    adjusted_trend = trend_direction * np.where(volume_change > volume_threshold, volume_change, 0)
    
    # Validate Trend with Price Action
    price_range = df['high'] - df['low']
    within_range = ((price_range < short_term_ma) & (price_range > long_term_ma)).astype(int)
    adjusted_trend = adjusted_trend * (1 + within_range)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=5).std()
    adjusted_cumulative_return = (adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Momentum
    lookback_period = 20
    log_returns = np.log(df['close'] / df['close'].shift(lookback_period))
    momentum = log_returns.rolling(window=lookback_period).sum()
    
    # Adjust by Intraday Volatility and Volume
    weighted_momentum = (momentum * df['volume']) / intraday_volatility
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    integrated_factor = relative_strength + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=5).mean()
    combined_factors = combined_factors * price_range_ratio
    
    # Introduce New Factor: Intraday Momentum
    intraday_log_return = np.log(df['high'] / df['low'])
    intraday_momentum = intraday_log_return * df['volume']
    
    # Integrate Intraday Momentum with Other Factors
    final_factor = integrated_factor + combined_factors + intraday_momentum
    
    return final_factor
