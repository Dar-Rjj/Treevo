```python
def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date, and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index as date, representing the alpha factor values.
    """
    
    # Calculate the percentage change in closing price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the average of the high and low prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the close price to the average of high and low prices
    df['close_to_avg_ratio'] = df['close'] / df['avg_high_low']
    
    # Calculate the difference between the high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the weighted sum of the high-low difference and the close-to-avg ratio
    df['weighted_sum'] = 0.6 * df['high_low_diff'] + 0.4 * df['close_to_avg_ratio']
    
    # Calculate the final alpha factor by multiplying the weighted sum with the percentage change in closing price
    alpha_factor = df['weighted_sum'] * df['close_pct_change']
    
    return alpha_factor
```
