```python
import pandas as pd
import numpy as np

def heuristics_v2(df, m):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(m))
    
    # Multiply Log Return by Volume
    volume_weighted_log_returns = log_return * df['volume']
    
    # Sum of Volume-Weighted Log Returns over the lookback period
    sum_volume_weighted_log_returns = volume_weighted_log_returns.rolling(window=m).sum()
    
    # Adjust by Intraday Volatility
    factor_values = sum_volume_weighted_log_returns / intraday_volatility
    
    return factor_values
```
