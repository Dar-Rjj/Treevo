```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5):
    # Compute Close Price Difference
    close_diff = df['close'].diff(n)
    
    # Smoothing the Momentum
    momentum_smoothed = close_diff.rolling(window=m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = momentum_smoothed * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_factor = volume_adjusted_momentum + combined_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Momentum
    log_return = np.log(df['close']).diff(m)
    sum_log_return = log_return.rolling(window=m).sum()
    volatility_adjusted_momentum = sum_log_return / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    log_return_intraday = np.log(df['close'] / df['open'])
    volume_weighted_intraday_return = log_return_intraday * daily_volume_ratio
    winsorized_return = volume_weighted_intraday_return.clip(lower=np.percentile(volume_weighted_intraday_return, 5), upper=np.percentile(volume_weighted_intraday_return, 95))
    
    # Final Alpha Factor
    final_alpha_factor = integrated_factor + volatility_adjusted_momentum + winsorized_return
    
    return final_alpha_factor
```
