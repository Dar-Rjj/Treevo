```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_price_change = df['close'].diff()

    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_momentum * np.sqrt(df['volume']))

    # Adjust for Volatility
    adjusted_weighted_momentum = weighted_momentum / high_low_range

    # Determine Trend Direction
    trend_5_day_ma = df['close'].rolling(window=5).mean()
    trend_20_day_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(trend_5_day_ma > trend_20_day_ma, 1, -1)

    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()

    # Calculate Daily Return
    daily_return = df['close'].diff()

    # Calculate Cumulative Return Over 5 Days
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume

    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.rolling(window=5).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    relative_strength_adjusted = relative_strength + adjusted_cumulative_return

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = high_low_range
    combined_factors = combined_factors * price_range_ratio

    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum_by_volume = trend_direction * volume_change

    # Calculate Volume-Weighted Price Movement
    volume_weighted_price_movement = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Relative Strength
    relative_strength = (momentum - df['close'].rolling(window=50).mean()) / volume_weighted_price_movement

    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_weighted_momentum + adjusted_cumulative_return + combined_factors) * relative_strength

    return final_alpha_factor
```
