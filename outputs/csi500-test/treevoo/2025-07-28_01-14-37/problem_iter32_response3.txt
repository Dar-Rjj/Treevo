```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative difference between high and low prices
    df['high_low_diff'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the money flow ratio
    df['money_flow_ratio'] = (df['close'] - df['open']) / df['amount']
    
    # Calculate the volume weighted close
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Combine the above features using a weighted sum
    df['alpha_factor'] = 0.4 * df['close_change'] + \
                         0.3 * df['high_low_diff'] + \
                         0.2 * df['money_flow_ratio'] + \
                         0.1 * df['volume_weighted_close']
    
    return df['alpha_factor'].dropna()
```
