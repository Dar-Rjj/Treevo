```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= df['volume']
    weighted_momentum = close_price_change / (weighted_momentum * df['volume'])
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Smoothing the Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=20).mean()
    daily_volume_ratio = df['volume'] / mean_volume
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement and Close-to-Open Return
    intraday_movement = df['high'] - df['low']
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = (high_low_range / df['high']) * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_momentum = volume_adjusted_momentum + combined_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.sum() / intraday_volatility
    
    # Volume-Adjusted Intraday Return
    log_intraday_return = np.log(df['close'] / df['open'])
    volume_weighted_return = log_intraday_return * df['volume']
    volume_weighted_return = volume_weighted_return.ewm(span=5, adjust=False).mean()
    volume_weighted_return = np.clip(volume_weighted_return, -3, 3)  # Winsorize
    
    # Confirm with Volume and Range
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_daily_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    cumulative_vwap = vwap.cumsum()
    cumulative_volume = df['volume'].cumsum()
    vwap = cumulative_vwap / cumulative_volume
    
    # Final Alpha Factor
    final_factor = (integrated_momentum + momentum + volume_weighted_return) * (volume_change * trend_direction * avg_daily_range) + vwap
    
    return final_factor
```
