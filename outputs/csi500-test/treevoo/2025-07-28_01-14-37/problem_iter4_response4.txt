```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Compute Close Price Difference
    df['close_diff'] = df['close'] - df['close'].shift(n)
    
    # Smoothing the Momentum
    df['momentum'] = df['close_diff'].rolling(window=m).mean()
    
    # Volume Weighing Process
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['adjusted_momentum'] = df['momentum'] * df['volume_ratio']
    
    # Momentum Signal
    df['signal'] = df['adjusted_momentum'].apply(lambda x: 1 if x > 0 else (0 if x == 0 else -1))
    
    return df['signal']
```
