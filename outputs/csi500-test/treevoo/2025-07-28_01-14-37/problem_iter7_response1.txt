```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum: Difference between current close and 10-day lagged close
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Exponential Moving Average (EMA)
    ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate Volume-Weighted Price
    weighted_price = (df['high'] * np.sqrt(df['volume']) + 
                      df['low'] * np.sqrt(df['volume']) + 
                      df['close'] * np.sqrt(df['volume']))
    
    # Calculate Total Weighted Volume
    total_weighted_volume = np.sqrt(df['volume']).sum()
    
    # Sum the weighted prices
    volume_weighted_price = weighted_price.sum()
    
    # Calculate Relative Strength
    relative_strength = (momentum - ema) / (volume_weighted_price / total_weighted_volume)
    
    return relative_strength
```
