```python
import pandas as pd
import numpy as np
from scipy.stats import skew, kurtosis

def heuristics_v2(df, n_days=5, m_days=20, decay_factor=0.9, upper_bound=3, lower_bound=-3):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=n_days).mean()
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(m_days))
    momentum = log_return.rolling(window=m_days).sum() / intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = np.log((df['close'] + 1e-8) / (df['open'] + 1e-8))
    intraday_return = np.clip(intraday_return, lower_bound, upper_bound)
    
    # Weighted Sum of Momentum and Intraday Return
    volume_smoothed = df['volume'].ewm(alpha=decay_factor).mean()
    weighted_momentum = momentum * volume_smoothed
    weighted_intraday_return = intraday_return * volume_smoothed
    combined_factor = (weighted_momentum + weighted_intraday_return) / 2
    
    # Adjust for Skewness
    skewness = skew(combined_factor.dropna())
    adjusted_skewness = combined_factor - skewness
    
    # Adjust for Kurtosis
    kurt = kurtosis(combined_factor.dropna())
    final_factor = adjusted_skewness / kurt
    
    return final_factor
```
