```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    trend_diff = short_term_ma - long_term_ma
    trend_direction = np.sign(trend_diff)
    confirmed_trend = np.where(trend_direction == trend_direction.shift(1), 1, 0)
    
    # Enhance with Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_threshold = 10000  # Example threshold
    adjusted_trend = np.where(np.abs(volume_change) > volume_threshold, confirmed_trend * volume_change, 0)
    
    # Calculate Relative Strength
    volume_weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_volume = df['volume'].sum()
    avg_volume_weighted_price = volume_weighted_price.sum() / total_volume
    relative_strength = (short_term_ma - long_term_ma) / avg_volume_weighted_price
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Weighted Momentum
    weighted_momentum_sum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = close_price_change / (weighted_momentum_sum * np.sqrt(df['volume']))
    
    # Adjust for Volatility
    adjusted_volatility = weighted_momentum / high_low_range
    
    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(20))
    momentum = log_return.rolling(window=20).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    integrated_factor = momentum - long_term_ma + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = (intraday_movement + close_open_return) * high_low_range
    
    # Confirm with Volume
    adjusted_momentum = adjusted_trend * volume_change
    
    # Calculate Intraday Reversal Indicator
    intraday_reversal_indicator = (df['close'].shift(1) - df['high']) + (df['low'] - df['close'].shift(1))
    
    # Integrate Intraday Reversal Indicator
    intraday_reversal_factor = intraday_reversal_indicator * intraday_volatility
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (integrated_factor + combined_factors) * relative_strength + intraday_reversal_factor
    
    return final_alpha_factor
```
