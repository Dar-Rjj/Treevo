```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close Price Change
    close_change = df['close'].diff()

    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= np.sqrt(df['volume'])
    weighted_momentum = close_change / weighted_momentum

    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range

    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)

    # Smoothing the Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()

    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = smoothed_momentum * daily_volume_ratio

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Price Range Ratio
    price_range_ratio = ((df['high'] - df['low']) / df['high']) * 100

    # Combine Factors
    combined_factors = (intraday_movement + close_to_open_return) * price_range_ratio

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = short_term_ma - df['close'].rolling(window=50).mean()
    adjusted_cumulative_return = df['close'].pct_change().cumsum()
    integrated_relative_strength = relative_strength + adjusted_cumulative_return

    # Integrate Volume Adjusted Momentum
    integrated_momentum = volume_adjusted_momentum + combined_factors

    # Calculate Intraday Volatility
    intraday_volatility = high_low_range.rolling(window=5).mean()

    # Momentum
    log_return = np.log(df['close'] / df['close'].shift(5))
    momentum = log_return.sum() / intraday_volatility

    # Volume-Adjusted Intraday Return
    intraday_log_return = np.log(df['close'] / df['open'])
    intraday_log_return = intraday_log_return.ewm(span=5).mean()
    upper_bound = intraday_log_return.quantile(0.95)
    lower_bound = intraday_log_return.quantile(0.05)
    winsorized_return = np.clip(intraday_log_return, lower_bound, upper_bound)
    volume_weighted_return = winsorized_return * daily_volume_ratio

    # Confirm with Volume and Range
    volume_change = df['volume'].diff()
    average_daily_range = high_low_range.rolling(window=5).mean()

    vwap = (df['high'] + df['low'] + df['close']) / 3
    vwap = (vwap * df['volume']).cumsum() / df['volume'].cumsum()

    # Synthesize Final Alpha Factor
    final_factor = (integrated_momentum + integrated_relative_strength + combined_factors) * relative_strength
    final_factor += (volume_adjusted_momentum + momentum + volume_weighted_return) * (volume_change * trend_direction * average_daily_range)
    final_factor += vwap

    return final_factor
```
