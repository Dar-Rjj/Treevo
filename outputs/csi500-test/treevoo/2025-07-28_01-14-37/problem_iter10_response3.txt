```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open']
    
    # Calculate Logarithmic Return
    log_return = np.log(intraday_return + 1e-8)
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Momentum
    weighted_momentum = close_price_change / vwp
    
    # Adjust for Volatility
    adjusted_momentum = weighted_momentum / high_low_range
    
    # Combine Intraday and Volatility-Adjusted Momentum
    volume_weighted_log_return = log_return * df['volume']
    smoothed_volume_weighted_log_return = volume_weighted_log_return.ewm(alpha=0.2).mean()
    smoothed_adjusted_momentum = adjusted_momentum * df['volume']
    combined_momentum = 0.5 * smoothed_volume_weighted_log_return + 0.5 * smoothed_adjusted_momentum.ewm(alpha=0.2).mean()
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum_by_volume = trend_direction * volume_change
    final_factor = (combined_momentum + adjusted_momentum_by_volume) / 2
    
    # Adjust for Skewness and Kurtosis
    skewness = final_factor.skew()
    kurtosis = final_factor.kurt()
    rescaled_skewness = (final_factor - final_factor.mean()) / final_factor.std()
    rescaled_kurtosis = (final_factor - final_factor.mean()) / final_factor.std()
    normalized_factor = (rescaled_skewness + rescaled_kurtosis) / high_low_range
    
    return normalized_factor
```
