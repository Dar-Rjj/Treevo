```python
import pandas as pd

def heuristics_v2(df, n=14, m=9):
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range.rolling(window=n).mean()

    # Calculate Raw Momentum
    close_diff = df['close'].diff(periods=m)
    smoothed_momentum = close_diff.rolling(window=m).mean()

    # Adjust by Intraday Volatility
    adjusted_momentum = smoothed_momentum / intraday_volatility

    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = adjusted_momentum * daily_volume_ratio

    # Momentum Signal
    momentum_signal = (volume_adjusted_momentum > 0).astype(int)

    # Calculate Price Changes
    price_change = df['close'].diff()

    # Separate Positive and Negative Gains
    positive_gains = price_change.where(price_change > 0, 0).rolling(window=n).sum()
    negative_gains = price_change.where(price_change < 0, 0).abs().rolling(window=n).sum()

    # Adjust for Volume
    weighted_positive_gains = positive_gains * df['volume'].rolling(window=n).sum()
    weighted_negative_gains = negative_gains * df['volume'].rolling(window=n).sum()

    # Calculate RSI
    rs = weighted_positive_gains / weighted_negative_gains
    rsi = 100 - (100 / (1 + rs))

    # Combine Momentum and RSI
    combined_indicator = volume_adjusted_momentum * rsi

    return combined_indicator
```
