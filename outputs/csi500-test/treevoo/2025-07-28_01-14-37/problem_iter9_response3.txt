def heuristics_v2(data, n_days=5, m_days=20):
    # Calculate Intraday Volatility
    intraday_volatility = (data['high'] - data['low']).rolling(window=n_days).mean()
    
    # Calculate Momentum using Log Returns
    log_return = (data['close'] / data['close'].shift(m_days)).apply(np.log)
    momentum = log_return.rolling(window=m_days).sum()
    
    # Adjust by Intraday Volatility and Volume
    weighted_momentum = (momentum * data['volume']) / intraday_volatility
    
    return weighted_momentum
