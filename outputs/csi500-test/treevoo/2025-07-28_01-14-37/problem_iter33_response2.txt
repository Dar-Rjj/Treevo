```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Compute Weighted Momentum
    weighted_momentum = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum *= np.sqrt(df['volume'])
    weighted_momentum = close_price_change / weighted_momentum
    
    # Adjust for Volatility
    adjusted_volatility_momentum = weighted_momentum / high_to_low_range
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    trend_direction = (short_term_ma > long_term_ma).astype(int) * 2 - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = high_to_low_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume
    
    # Calculate Momentum
    log_return = np.log(df['close'] / df['close'].shift(20))
    momentum = log_return.rolling(window=20).sum()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - df['close'].rolling(window=50).mean()
    combined_return = relative_strength + adjusted_cumulative_return
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Combine Factors
    combined_factors = intraday_movement + close_to_open_return
    price_range_ratio = (df['high'] - df['low']) / df['close']
    weighted_combined_factors = combined_factors * price_range_ratio
    
    # Confirm with Volume
    volume_change = df['volume'].diff()
    adjusted_momentum = trend_direction * volume_change
    
    # Calculate Relative Strength
    volume_weighted_price_movement = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    relative_strength = (momentum - long_term_ma) / volume_weighted_price_movement
    
    # Incorporate Price Oscillator
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_oscillator = short_term_ema - long_term_ema
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_volatility_momentum + combined_return + weighted_combined_factors) * relative_strength / intraday_volatility
    
    return final_alpha_factor
```
