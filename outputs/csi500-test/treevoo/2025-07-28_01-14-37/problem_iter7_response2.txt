```python
import pandas as pd

def heuristics_v2(df, window=5, momentum_window=10, sma_window=50):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window
    cumulative_return = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=window).mean()
    adjusted_cumulative_return = cumulative_return * avg_volume
    
    # Adjust by Price Volatility
    close_std = df['close'].rolling(window=window).std()
    volatility_adjusted_cumulative_return = (adjusted_cumulative_return / close_std) + 1
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(momentum_window)
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma = df['close'].rolling(window=sma_window).mean()
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = momentum - sma
    final_factor = relative_strength + volatility_adjusted_cumulative_return
    
    return final_factor
```
