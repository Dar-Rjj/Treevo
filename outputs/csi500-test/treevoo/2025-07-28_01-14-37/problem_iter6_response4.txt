```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a 5-day window
    window_size = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window_size).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window_size).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Adjust by Price Volatility
    df['std_dev_close'] = df['close'].rolling(window=window_size).std()
    df['final_adjusted_return'] = (df['adjusted_cumulative_return'] / df['std_dev_close']) + 1
    
    # Calculate Momentum
    momentum_window = 10
    df['momentum'] = df['close'] - df['close'].shift(momentum_window)
    
    # Calculate 50-day Simple Moving Average (SMA)
    sma_window = 50
    df['sma'] = df['close'].rolling(window=sma_window).mean()
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['total_weighted_price'] = df['weighted_price'].rolling(window=window_size).sum()
    df['total_volume'] = df['volume'].rolling(window=window_size).sum()
    df['volume_weighted_price_movement'] = df['total_weighted_price'] / df['total_volume']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['momentum'] - df['sma']
    df['integrated_factor'] = df['relative_strength'] / df['volume_weighted_price_movement'] + df['final_adjusted_return']
    
    # Introduce Volume-Weighted Momentum
    volume_weighted_momentum_window = 10
    df['volume_weighted_momentum'] = df['volume_weighted_price_movement'].pct_change(volume_weighted_momentum_window)
    df['final_integrated_factor'] = df['integrated_factor'] + df['volume_weighted_momentum']
    
    return df['final_integrated_factor'].dropna()
```
