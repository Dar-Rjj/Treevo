```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day Short-Term Moving Average (Close price)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    
    # Compute 20-day Long-Term Moving Average (Close price)
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = (df['5_day_MA'] > df['20_day_MA']).astype(int) * 2 - 1
    
    # Calculate Volume Change (Current Volume - Previous Volume)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Momentum by Volume Change
    df['momentum_adjusted'] = df['trend_direction'] * df['volume_change']
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = df['weighted_price'].sum()
    total_volume = df['volume'].sum()
    volume_weighted_price_movement = total_weighted_price / total_volume
    
    # Subtract Long-Term Moving Average from Price Momentum
    df['price_momentum'] = df['close'] - df['20_day_MA']
    
    # Divide the result by Volume-Weighted Price Movement to get the factor
    df['factor'] = df['price_momentum'] / volume_weighted_price_movement
    
    return df['factor']
```
