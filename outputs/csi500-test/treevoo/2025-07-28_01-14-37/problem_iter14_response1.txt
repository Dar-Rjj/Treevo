```python
def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2

    # Calculate the money flow, which is the product of the average price and volume
    money_flow = avg_price * df['volume']
    
    # Calculate the momentum of the money flow
    momentum = money_flow - money_flow.shift(1)
    
    # Calculate the high-low ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the final factor by multiplying the momentum with the high-low ratio
    factor = momentum * high_low_ratio
    
    return factor
```
