import pandas as pd

def heuristics_v2(df):
    # Compute Close Price Difference
    n = 5  # Number of days for the difference
    close_diff = df['close'] - df['close'].shift(n)

    # Smoothing the Momentum
    m = 10  # Number of days for the moving average
    smoothed_momentum = close_diff.rolling(window=m).mean()

    # Volume Weighing Process
    mean_volume = df['volume'].rolling(window=n).mean()
    daily_volume_ratio = df['volume'] / mean_volume
    adjusted_momentum = smoothed_momentum * daily_volume_ratio

    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['high'] * 100

    # Combine Intraday Factors
    combined_intraday_factors = (intraday_movement + close_to_open_return) * price_range_ratio

    # Integrate Volume Adjusted Momentum
    integrated_momentum = adjusted_momentum + combined_intraday_factors

    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    long_term_momentum = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = (short_term_momentum > long_term_momentum).astype(int) * 2 - 1

    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    confirmed_momentum = trend_direction * volume_change

    # Calculate Relative Strength
    relative_strength = (short_term_momentum - long_term_momentum) / calculate_volume_weighted_price_movement(df)

    # Final Alpha Factor
    final_alpha_factor = 0.6 * integrated_momentum + 0.4 * relative_strength

    return final_alpha_factor

def calculate_volume_weighted_price_movement(df):
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_weighted_price = weighted_price.rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    return total_weighted_price / total_volume
