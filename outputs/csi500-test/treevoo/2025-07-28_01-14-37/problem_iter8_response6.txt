```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * avg_volume

    # Adjust by Price Volatility
    std_close = df['close'].rolling(window=window).std()
    df['final_adjusted_cumulative_return'] = (df['adjusted_cumulative_return'] / std_close) + 1

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Logarithmic Return and adjust for bias
    df['log_return'] = np.log(df['intraday_return'] + 1e-6)

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    intraday_momentum_volatility = (df['intraday_return'] * df['volume'] + df['intraday_range'] * df['volume'])

    # Adjust for Skewness and Kurtosis
    skewness = df['intraday_return'].skew()
    kurtosis = df['intraday_return'].kurtosis()
    rescaled_intraday_momentum_volatility = (intraday_momentum_volatility - skewness) / (kurtosis + 1)

    # Calculate Volume-Weighted Price Movement
    weighted_price = (df['high'] + df['low'] + df['close']) * np.sqrt(df['volume'])
    total_volume = df['volume'].sum()
    volume_weighted_price_movement = weighted_price.sum() / total_volume

    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(10)

    # Calculate 50-day Exponential Moving Average (EMA)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()

    # Integrate Relative Strength and Adjusted Cumulative Return
    relative_strength = df['momentum'] - ema_50
    combined_factor = relative_strength + df['final_adjusted_cumulative_return'] + rescaled_intraday_momentum_volatility * volume_weighted_price_movement

    return combined_factor
```
