```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate 50-day Simple Moving Average (SMA)
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    total_volume = df['volume'].sum()
    volume_weighted_price = df['weighted_price'].sum() / total_volume
    
    # Calculate Relative Strength
    df['relative_strength'] = (df['momentum'] - df['sma_50']) / volume_weighted_price
    
    # Calculate Intraday Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = ((df['high'] - df['low']) / df['high']) * 100
    
    # Combine Factors
    combined_factor = df['intraday_movement'] + df['close_to_open_return']
    df['combined_factors'] = combined_factor * df['price_range_ratio']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['relative_strength'] + df['combined_factors']
    
    return df['alpha_factor']
```
