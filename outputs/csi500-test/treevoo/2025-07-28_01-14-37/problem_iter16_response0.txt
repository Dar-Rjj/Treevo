```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, momentum_window=10, ema_window=50):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume']
    df['combined_factor'] = df['weighted_intraday_return'] + df['weighted_intraday_range']

    # Adjust for Skewness and Kurtosis
    skewness = df['combined_factor'].skew()
    kurtosis = df['combined_factor'].kurtosis()
    df['rescaled_combined_factor'] = (df['combined_factor'] - df['combined_factor'].mean()) / df['combined_factor'].std()
    df['rescaled_combined_factor'] = (df['rescaled_combined_factor'] - skewness) / kurtosis

    # Calculate Volume-Weighted Price Movement
    df['weighted_price'] = (df['high'] + df['low'] + df['close']) / 3 * np.sqrt(df['volume'])
    df['volume_weighted_price_movement'] = df['weighted_price'].rolling(window=window).sum() / df['volume'].rolling(window=window).sum()

    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(momentum_window)

    # Calculate 50-day Exponential Moving Average (EMA)
    df['ema'] = df['close'].ewm(span=ema_window, adjust=False).mean()

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['momentum'] - df['ema']
    df['integrated_momentum'] = df['adjusted_cumulative_return'] + df['relative_strength']

    # Combine with Volume-Weighted Price Movement
    df['final_factor'] = df['integrated_momentum'] + df['volume_weighted_price_movement']

    # Final Rescaling
    final_skewness = df['final_factor'].skew()
    final_kurtosis = df['final_factor'].kurtosis()
    df['rescaled_final_factor'] = (df['final_factor'] - df['final_factor'].mean()) / df['final_factor'].std()
    df['rescaled_final_factor'] = (df['rescaled_final_factor'] - final_skewness) / final_kurtosis

    return df['rescaled_final_factor'].dropna()
```
