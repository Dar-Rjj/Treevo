```python
def heuristics_v2(df):
    # Calculate the relative strength index (RSI) like feature
    df['price_change'] = df['close'].diff()
    df['gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rsi_like'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))

    # Calculate the money flow index (MFI) like feature
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df[df['price_change'] > 0]['money_flow']
    df['negative_money_flow'] = df[df['price_change'] < 0]['money_flow']
    df['positive_money_flow_sum'] = df['positive_money_flow'].rolling(window=14).sum()
    df['negative_money_flow_sum'] = df['negative_money_flow'].rolling(window=14).sum()
    df['mfi_like'] = 100 - (100 / (1 + (df['positive_money_flow_sum'] / df['negative_money_flow_sum'])))
    
    # Combine the RSI-like and MFI-like features to create a novel alpha factor
    df['alpha_factor'] = (df['rsi_like'] + df['mfi_like']) / 2

    return df['alpha_factor']
```
