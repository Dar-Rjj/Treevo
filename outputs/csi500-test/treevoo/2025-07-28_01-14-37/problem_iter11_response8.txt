```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=20):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).mean()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    cumulative_return = daily_return.rolling(window=n).sum()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=n).mean()
    volume_adjusted_cumulative_return = cumulative_return * average_volume
    
    # Adjust by Price Volatility
    price_volatility = df['close'].rolling(window=n).std()
    volatility_adjusted_cumulative_return = (volume_adjusted_cumulative_return / price_volatility) + 1
    
    # Calculate Weighted Momentum
    log_returns = np.log(df['close'] / df['close'].shift(m))
    weights = np.exp(-np.arange(m)[::-1] / m)
    weighted_log_returns = (log_returns.rolling(window=m).apply(lambda x: (x * weights).sum(), raw=False))
    weighted_momentum = weighted_log_returns / intraday_volatility
    
    # Determine Trend Direction
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    trend_direction = np.where(short_term_ema > long_term_ema, 1, -1)
    
    # Confirm with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_trend_by_volume = trend_direction * volume_change
    final_factor = (volatility_adjusted_cumulative_return + adjusted_trend_by_volume) / 2
    
    # Smoothing the Momentum
    smoothed_momentum = final_factor.rolling(window=m).mean()
    
    # Volume Weighing Process
    daily_volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_weighted_momentum = smoothed_momentum * daily_volume_ratio
    
    # Calculate Intraday Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Range Ratio
    price_range_ratio = ((df['high'] - df['low']) / df['high']) * 100
    
    # Combine Factors
    combined_factor = (intraday_movement + close_to_open_return) * price_range_ratio
    
    # Integrate Volume Adjusted Momentum
    integrated_factor = combined_factor + volume_weighted_momentum
    
    # Calculate Relative Strength
    long_term_ma = df['close'].rolling(window=20).mean()
    relative_strength = (weighted_momentum - long_term_ma) / (df['volume'] * (df['high'] + df['low'] + df['close']) / 3)
    
    return integrated_factor
```
