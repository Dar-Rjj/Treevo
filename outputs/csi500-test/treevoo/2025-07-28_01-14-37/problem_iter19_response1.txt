```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['range'] = df['high'] - df['low']
    
    # Compute Weighted Momentum
    df['weighted_price'] = (df['high'] + df['low'] + 2 * df['close']) / 4
    df['price_change'] = df['close'].diff()
    df['weighted_momentum'] = df['price_change'] / (df['weighted_price'] * df['volume'])
    
    # Adjust for Volatility
    df['volatility_adjusted_momentum'] = df['weighted_momentum'] / df['range']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=5).mean()
    df['long_term_ma'] = df['close'].rolling(window=20).mean()
    df['trend_direction'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Confirm with Volume and Range
    df['volume_change'] = df['volume'].diff()
    df['average_daily_range'] = df['range'].rolling(window=5).mean()
    
    # Final Factor
    df['factor'] = (df['volatility_adjusted_momentum'] * 
                    df['volume_change'] * 
                    df['trend_direction'] * 
                    df['average_daily_range'])
    
    return df['factor']
```
