```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Cumulative Return
    window = 5
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()

    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']

    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    n_days = 5
    df['intraday_volatility'] = df['high_low_range'].rolling(window=n_days).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1

    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = df['weighted_intraday_return'] + df['weighted_high_low_range']

    # Calculate Price Momentum
    m_days = 20
    df['log_return'] = np.log(df['close'] / df['close'].shift(m_days))
    df['price_momentum'] = df['log_return'].rolling(window=m_days).sum()

    # Calculate Short-Term Moving Average
    short_term_window = 5
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()

    # Calculate Long-Term Moving Average
    long_term_window = 20
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()

    # Determine Trend Direction
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_trend_direction'] = df['trend_direction'] * df['volume_change']

    # Adjust by Intraday Volatility and Volume
    df['weighted_log_return'] = df['price_momentum'] * df['volume']
    df['weighted_momentum'] = df['weighted_log_return'] / df['intraday_volatility']

    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['price_momentum'] + df['adjusted_cumulative_return']
    df['volume_weighted_price_movement'] = df['combined_intraday_factor'] + df['weighted_momentum']
    df['integrated_factor'] = (df['relative_strength'] + df['volume_weighted_price_movement'])

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + (df['log_return'] * df['volume'])
    df['intraday_reversal'] = (df['open'] - df['close']) / df['open']
    df['final_rescaled_factor'] = df['final_alpha_factor'] + df['intraday_reversal']

    return df['final_rescaled_factor']
```
