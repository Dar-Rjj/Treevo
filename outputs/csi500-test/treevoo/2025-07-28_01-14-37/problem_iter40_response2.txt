```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()

    # Determine Trend Direction
    trend_direction = (short_term_ma - long_term_ma).apply(lambda x: 1 if x > 0 else -1)

    # Enhance with Volume Confirmation
    volume_change = df['volume'].diff()
    threshold = 0.01 * df['volume'].mean()
    adjusted_trend = trend_direction * volume_change
    adjusted_trend[abs(volume_change) < threshold] = 0

    # Calculate Relative Strength
    weighted_price_movement = (df[['high', 'low', 'close']].sum(axis=1) / 3 * df['volume']).sum() / df['volume'].sum()
    relative_strength = (short_term_ma - long_term_ma) / weighted_price_movement

    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Weighted Momentum
    sum_high_low_close = (df['high'] + df['low'] + 2 * df['close']) / 4
    weighted_momentum = (df['close'].diff() / (sum_high_low_close * np.sqrt(df['volume'])))

    # Adjust for Volatility
    adjusted_volatility = weighted_momentum / high_low_range

    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Cumulative Return
    cumulative_return = daily_return.rolling(window=5).sum()

    # Adjust by Volume
    average_volume = df['volume'].rolling(window=5).mean()
    adjusted_cumulative_return = cumulative_return * average_volume

    # Calculate Momentum
    momentum = np.log(df['close'] / df['close'].shift(5)).rolling(window=5).sum()

    # Integrate Relative Strength and Adjusted Cumulative Return
    combined_factors = (momentum - long_term_ma) + adjusted_cumulative_return

    # Calculate Intraday Reversal Indicator
    intraday_reversal_indicator = (df['high'] - df['close'].shift(1)) - (df['close'].shift(1) - df['low'])

    # Integrate Intraday Reversal Indicator
    intraday_volatility = high_low_range.rolling(window=5).std()
    intraday_reversal_factor = intraday_reversal_indicator * intraday_volatility

    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_volatility * relative_strength + adjusted_cumulative_return + combined_factors + intraday_reversal_factor)
    
    return final_alpha_factor
```
