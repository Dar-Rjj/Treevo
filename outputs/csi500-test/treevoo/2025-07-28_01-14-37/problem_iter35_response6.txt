import pandas as pd

def heuristics_v2(df, window=5, n_days=5, m_days=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['vol_adjusted_cumulative_return'] = df['cumulative_return'] * df['average_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=n_days).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['weighted_intraday_range'] = df['high_low_range'] * df['volume']
    df['combined_intraday_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_range']).rolling(window=n_days).mean()
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(m_days)).apply(np.log)
    df['price_momentum'] = df['log_return'].rolling(window=m_days).sum()
    
    # Adjust by Intraday Volatility and Volume
    df['volume_weighted_momentum'] = df['price_momentum'] * df['average_volume'] / df['intraday_volatility']
    
    # Determine Trend Direction
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=50).mean()
    df['trend_direction'] = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Confirm with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_trend'] = df['trend_direction'] * df['volume_change']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['confirmed_trend'] + df['vol_adjusted_cumulative_return']
    df['final_combined_factor'] = df['combined_intraday_factor'] + df['volume_weighted_momentum']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['relative_strength'] + df['final_combined_factor']
    
    return df['alpha_factor'].dropna()
