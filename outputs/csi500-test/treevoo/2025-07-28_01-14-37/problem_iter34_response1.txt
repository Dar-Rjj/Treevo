import pandas as pd

def heuristics_v2(df, window=5, lookback=30):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Cumulative Return over a specified window (e.g., 5 days)
    df['cumulative_return'] = df['daily_return'].rolling(window=window).sum()
    
    # Adjust by Volume
    df['avg_volume'] = df['volume'].rolling(window=window).mean()
    df['adjusted_cumulative_return'] = df['cumulative_return'] * df['avg_volume']
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Momentum and Volatility
    df['weighted_intraday_return'] = df['intraday_return'] * df['avg_volume']
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['avg_volume']
    df['combined_factor'] = (df['weighted_intraday_return'] + df['weighted_intraday_volatility']) / 2
    
    # Calculate Price Momentum
    df['log_return'] = (df['close'] / df['close'].shift(lookback)).apply(np.log)
    df['price_momentum'] = df['log_return'].rolling(window=lookback).sum()

    df['adjusted_price_momentum'] = (df['price_momentum'] * df['avg_volume']) / df['intraday_volatility']
    
    # Determine Trend Direction
    df['short_term_ma'] = df['close'].rolling(window=10).mean()
    df['long_term_ma'] = df['close'].rolling(window=50).mean()
    df['trend_direction'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['confirmed_trend'] = df['trend_direction'] * df['volume_change']
    
    # Integrate Relative Strength and Adjusted Cumulative Return
    df['relative_strength'] = df['price_momentum'] + df['confirmed_trend']
    df['integrated_factor'] = df['relative_strength'] + df['adjusted_cumulative_return']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['combined_factor']
    
    return df['final_alpha_factor']
