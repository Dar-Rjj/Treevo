```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']

    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']

    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = daily_range * df['volume'] + open_to_close_movement

    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    spread_momentum = high_low_spread.rolling(window=20).mean().diff()

    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()

    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['close'].diff()) * df['volume']

    # Incorporate Relative Strength
    relative_strength = df['close'] / df['close'].rolling(window=20).mean()

    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff + open_low_diff) * df['volume'] / df['volume'].sum()

    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    standard_deviation = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / standard_deviation

    # Combined Indicators
    combined_indicator = 0.5 * volume_adjusted_breakout_strength + spread_momentum + volume_adjusted_momentum
    combined_indicator_with_relative_strength = combined_indicator * relative_strength
    final_alpha_factor = combined_indicator_with_relative_strength * enhanced_price_reversion * weighted_volatility

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.mean() + 2 * final_alpha_factor.std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)

    return strong_moves
```
