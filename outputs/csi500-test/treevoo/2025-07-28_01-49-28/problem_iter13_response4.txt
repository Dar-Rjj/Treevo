```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Simple Momentum
    simple_momentum = df['close'] / df['close'].shift(n) - 1
    
    # Average Volume over the period
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Adjusted Momentum by Volume
    volume_adjusted_momentum = simple_momentum * (avg_volume / df['volume'])
    
    # High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # High-to-Low Return
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combined Momentum and High-Low Return
    combined_momentum_highlow = volume_adjusted_momentum + high_low_return
    
    # Open-High Price Difference
    open_high_diff = df['high'] - df['open']
    
    # Open-Low Price Difference
    open_low_diff = df['open'] - df['low']
    
    # Volatility Measure
    volatility = open_high_diff + open_low_diff
    
    # Volume Percentage Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Volatility using Volume Change
    adjusted_volatility = volatility * (1 + volume_change)
    
    # Amount per Unit Close Price Rate of Change
    amount_sum = df['amount'].rolling(window=n).sum() / df['close']
    prev_amount_sum = df['amount'].rolling(window=n).sum().shift(n) / df['close'].shift(n)
    amount_rate_of_change = (amount_sum / prev_amount_sum) - 1
    
    # Integrated Alpha
    integrated_alpha = combined_momentum_highlow + adjusted_volatility + amount_rate_of_change
    
    return integrated_alpha
```
