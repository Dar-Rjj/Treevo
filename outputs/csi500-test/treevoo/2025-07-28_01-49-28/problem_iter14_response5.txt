```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pandas.DataFrame): A DataFrame indexed by (date) with columns representing various market features (open, high, low, close, amount, volume).
    
    Returns:
    pandas.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the intraday price range
    intraday_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the day's range
    close_to_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate the volume-weighted close
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=1)
    
    # Combine the factors into a single alpha factor
    alpha_factor = close_to_range + momentum + (volume_weighted_close / df['amount'])
    
    return alpha_factor
```
