```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    high_low = df['high'] - df['low']
    close_prev_close = abs(df['close'] - df['Previous Close'])
    high_prev_close = abs(df['high'] - df['Previous Close'])
    low_prev_close = abs(df['low'] - df['Previous Close'])

    true_range = pd.DataFrame({
        'High-Low': high_low,
        'Close-PrevClose': close_prev_close,
        'High-PrevClose': high_prev_close,
        'Low-PrevClose': low_prev_close
    }).max(axis=1)

    # Calculate Moving Average of True Range
    atr_period = 14  # Example period, can be adjusted
    moving_average_tr = true_range.rolling(window=atr_period).mean()

    return moving_average_tr
```
