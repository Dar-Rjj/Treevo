```python
def heuristics_v2(df):
    # Calculate the simple moving average of close prices over 5 and 10 days
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the exponential moving average of close prices over 5 and 10 days
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative difference between the EMAs and SMAs
    rel_diff_sma = (sma_5 - sma_10) / sma_10
    rel_diff_ema = (ema_5 - ema_10) / ema_10
    
    # Calculate the weighted combination of the relative differences
    factor = 0.6 * rel_diff_ema + 0.4 * rel_diff_sma
    
    return factor
```
