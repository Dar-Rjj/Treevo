```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    weighted_momentum = momentum * df['volume']
    
    # Combine with Volume-Adjusted High-Low Spread
    combined_factor = volume_adjusted_high_low_spread + weighted_momentum
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (min_proximity + 1e-6)  # Avoid division by zero
    daily_returns = df['close'].pct_change()
    std_dev_20_day = daily_returns.rolling(window=20).std()
    enhanced_reversion = inverted_proximity / std_dev_20_day
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Combine All Factors
    final_alpha_factor = combined_factor * enhanced_reversion * open_high_low_volatility
    
    return final_alpha_factor
```
