```python
import pandas as pd

def heuristics_v2(df):
    # Short-term, medium-term, and long-term moving averages of close prices
    short_term_ma = df['close'].rolling(window=5).mean()
    medium_term_ma = df['close'].rolling(window=20).mean()
    long_term_ma = df['close'].rolling(window=60).mean()
    
    # Daily range and its average over 10 days
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    
    # Momentum indicators
    roc_10d = df['close'].pct_change(periods=10)
    relative_strength = (df['close'] - df['close'].rolling(window=10).min()) / df['close'].rolling(window=10).min()
    
    # Volume analysis
    high_volume_flag = (df['volume'] > df['volume'].rolling(window=30).mean() * 1.2).astype(int)
    recent_cum_vol = df['volume'].rolling(window=5).sum()
    past_cum_vol = df['volume'].rolling(window=20).sum()
    vol_trend_strength = recent_cum_vol / past_cum_vol
    
    # Open and Close price dynamics
    close_open_ratio = df['close'] / df['open']
    consecutive_close_above_open = (df['close'] > df['open']).groupby((df['close'] <= df['open']).cumsum()).cumcount()
    
    # Composite signals
    momentum_with_volume_surge = roc_10d * high_volume_flag
    multi_factor_score = (short_term_ma - long_term_ma) * relative_strength
    
    # Final Alpha Factor
    alpha_factor = (
        (short_term_ma - medium_term_ma) * 0.3 +
        (medium_term_ma - long_term_ma) * 0.3 +
        (relative_strength - 1) * 0.2 +
        (momentum_with_volume_surge + multi_factor_score) * 0.1 +
        (vol_trend_strength - 1) * 0.1
    )
    
    return alpha_factor
```
