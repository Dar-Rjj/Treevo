```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_avg_high_low_spread = high_low_spread.rolling(window=20).mean()
    spread_momentum = moving_avg_high_low_spread.diff()

    # Volume-Adjusted Momentum
    close_diff = df['close'].diff()
    volume_adjusted_momentum = close_diff * df['volume']

    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff * df['volume'] + open_low_diff * df['volume']) / (2 * df['volume'])

    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = -min_proximity
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / twenty_day_std

    # Long-Term Trend Indicator
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    long_term_trend = sma_50 - sma_200

    # Combined Indicators
    combined_indicator = 0.5 * volume_adjusted_breakout_strength + spread_momentum + volume_adjusted_momentum

    # Final Alpha Factor
    final_alpha_factor = combined_indicator * enhanced_price_reversion * open_high_low_volatility * long_term_trend

    return final_alpha_factor
```
