```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    adjusted_open_to_close = open_to_close_movement * df['volume']
    
    # Combine Indicators
    combined_indicator = adjusted_open_to_close + daily_range
    
    # Filter for Strong Moves
    threshold = 100000  # example threshold, this should be optimized
    strong_moves = (combined_indicator > threshold).astype(int)
    
    # Calculate Moving Sum of High-Low Price Spreads
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Calculate Volume Trend
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_trend = volume_ma_20 - volume_ma_20.shift(1)
    
    # Final Alpha Factor
    alpha_factor = (spread_momentum + volume_trend) * strong_moves
    
    return alpha_factor
```
