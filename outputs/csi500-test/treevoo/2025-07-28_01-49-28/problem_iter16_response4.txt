import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Weighted Momentum
    momentum = df['close'].diff()
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Exponential Moving Average of Momentum
    ema_momentum = volume_weighted_momentum.ewm(span=10, min_periods=10).mean()
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Calculate Relative Strength Index (RSI)
    daily_returns = df['close'].pct_change()
    average_gain = daily_returns.where(daily_returns > 0, 0).ewm(alpha=1/14, min_periods=14, adjust=False).mean()
    average_loss = daily_returns.where(daily_returns < 0, 0).abs().ewm(alpha=1/14, min_periods=14, adjust=False).mean()
    relative_strength = average_gain / average_loss
    rsi = 100 - (100 / (1 + relative_strength))
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    distance_to_high = df['close'] - ten_day_high
    distance_to_low = df['close'] - ten_day_low
    proximity = 1 / (1 + pd.concat([distance_to_high.abs(), distance_to_low.abs()], axis=1).min(axis=1))
    std_dev_daily_returns = daily_returns.rolling(window=20).std()
    adjusted_proximity = proximity / std_dev_daily_returns
    
    # Combine Factors
    factor = (
        (volume_adjusted_high_low_spread + volume_weighted_momentum) *
        price_change_ratio *
        ema_momentum *
        adjusted_proximity *
        rsi
    )
    
    return factor.rename('High-Frequency Trading Activity, Momentum with Volume Influence, and Relative Strength Index')
