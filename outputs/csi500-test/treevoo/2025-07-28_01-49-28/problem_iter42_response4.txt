```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume information.
    
    Parameters:
    df (pandas.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'],
                           indexed by date.
                           
    Returns:
    pandas.Series: A series of alpha factor values, indexed by date.
    """
    # Calculate the average price for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the money flow ratio as the ratio of amount to volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # The alpha factor is the product of the average price and the normalized money flow ratio
    alpha_factor = avg_price * money_flow_ratio
    
    return alpha_factor
```
