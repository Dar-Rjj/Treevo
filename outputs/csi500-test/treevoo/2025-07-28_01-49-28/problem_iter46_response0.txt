```python
import pandas as pd

def heuristics_v2(df):
    # Calculate raw volatility factor
    df['raw_volatility'] = df['high'] - df['low']
    
    # Calculate raw price movement factor
    df['raw_price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate raw volume change factor
    df['raw_volume_change'] = df['volume'].diff()
    
    # Calculate raw volume-price interaction factor
    df['raw_volume_price_interaction'] = (df['volume'] * (df['close'] - df['open']))
    
    # Summarize the (close - open) * volume over the last 5 days as a raw price-volume composite factor
    df['raw_price_volume_composite'] = (df['close'] - df['open']) * df['volume']
    df['raw_price_volume_composite_5d'] = df['raw_price_volume_composite'].rolling(window=5).sum()
    
    # Calculate the (open - previous close) as a raw gap factor
    df['raw_gap'] = df['open'] - df['close'].shift(1)
    
    # Define positive money flow as (close - open) * volume when close > open, otherwise 0
    df['positive_money_flow'] = (df['close'] - df['open']) * df['volume'] * (df['close'] > df['open']).astype(int)
    
    # Define total money flow as (close - open) * volume, regardless of price direction
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Use the ratio of positive to total money flow as a raw money flow index
    df['raw_money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']
    
    # Use the (amount / volume) as a raw transaction size factor
    df['raw_transaction_size'] = df['amount'] / df['volume']
    
    # Calculate the (close - open) * (volume / average volume) as a raw momentum factor
    avg_volume = df['volume'].rolling(window=20).mean()
    df['raw_momentum'] = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    
    # Calculate the (high - close) / (high - low) as a raw bearish pressure factor
    df['raw_bearish_pressure'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the (close - low) / (high - low) as a raw bullish pressure factor
    df['raw_bullish_pressure'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Use the rolling average transaction size over the last 5 days as a raw transaction size trend factor
    df['raw_transaction_size_trend'] = df['raw_transaction_size'].rolling(window=5).mean()
    
    # Calculate the sum of (close - open) over the last 10 days as a raw cumulative price change factor
    df['raw_cumulative_price_change'] = df['close'] - df['open']
    df['raw_cumulative_price_change_10d'] = df['raw_cumulative_price_change'].rolling(window=10).sum()
    
    # Calculate the sum of (today's volume - yesterday's volume) over the last 10 days as a raw cumulative volume change factor
    df['raw_cumulative_volume_change_10d'] = df['raw_volume_change'].rolling(window=10).sum()
    
    # Calculate the (close - open) * (volume / average volume) over the last 20 days as a long-term raw momentum factor
    df['long_term_raw_momentum'] = (df['close'] - df['open']) * (df['volume'] / avg_volume)
    df['long_term_raw_momentum_20d'] = df['long_term_raw_momentum'].rolling(window=20).sum()
    
    # Select the final alpha factor
    alpha_factor = df['raw_price_volume_composite_5d']
    
    return alpha_factor
```
