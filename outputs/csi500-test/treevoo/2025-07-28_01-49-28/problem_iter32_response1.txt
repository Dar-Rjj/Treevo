import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close
    
    # High-Low Spread Momentum
    high_low_spread_momentum = (df['high'] - df['low']).rolling(window=10).sum() - (df['high'] - df['low']).shift(1).rolling(window=10).sum()
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume
    
    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = ten_day_low - df['close']
    min_proximity = np.minimum(np.abs(proximity_to_high), np.abs(proximity_to_low))
    twenty_day_volatility = df['close'].rolling(window=20).std()
    adjusted_proximity = min_proximity / twenty_day_volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_average_volatility = (open_high_diff + open_low_diff) * df['volume'] / (df['volume'].rolling(window=10).mean())
    
    # Enhanced Momentum Factor
    n = 10
    smoothed_volume = df['volume'].ewm(span=n, adjust=False).mean()
    enhanced_momentum = (df['close'] - df['close'].shift(n)) * smoothed_volume
    
    # High-Low Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = enhanced_momentum + high_low_return
    
    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    weighted_avg_volume = df['volume'].rolling(window=10).sum() / 10
    volume_weighted_spread = spread * weighted_avg_volume
    
    # Amount per Unit Close Price Rate of Change
    sum_amounts = df['amount'].rolling(window=10).sum()
    current_ratio = sum_amounts / df['close']
    previous_ratio = sum_amounts.shift(10) / df['close'].shift(10)
    differential = current_ratio - previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength +
        volume_weighted_spread +
        volume_adjusted_momentum +
        combined_factors +
        weighted_average_volatility +
        differential
    ) * price_change_ratio
    
    # Filter for Strong Moves
    strong_move_threshold = 0.5
    strong_move = (integrated_alpha_factor > strong_move_threshold).astype(int)
    
    # Price Trend Indicator
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_trend_score = long_term_momentum - short_term_momentum
    
    # Enhanced Alpha Factor
    enhanced_alpha_factor = integrated_alpha_factor * price_trend_score
    
    return pd.Series(enhanced_alpha_factor, index=df.index, name='Enhanced Alpha Factor')
