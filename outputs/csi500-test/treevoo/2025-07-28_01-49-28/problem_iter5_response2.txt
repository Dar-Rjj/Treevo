import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_movement = open_to_close_movement * df['volume']
    
    # Combine Indicators
    combined_indicator = (daily_range * df['volume']) + volume_adjusted_movement
    strong_move = (combined_indicator > 0).astype(int)
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum
    momentum = df['close'].diff(1)
    volume_adjusted_momentum = volume_adjusted_high_low_spread + momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Overnight Return
    overnight_return = df['open'] / df['close'].shift(1) - 1
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Volume Weighted Overnight Return
    previous_day_volume = df['volume'].shift(1)
    volume_weighted_overnight_return = overnight_return * previous_day_volume
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Momentum Score
    momentum_score = intraday_return / intraday_volatility
    
    # Reversal Score
    reversal_score = overnight_return / intraday_range
    
    # Composite Alpha Factor
    composite_alpha_factor = (
        strong_move +
        volume_adjusted_momentum +
        momentum_score -
        reversal_score
    )
    
    return composite_alpha_factor
