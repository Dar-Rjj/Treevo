import pandas as pd

def heuristics_v2(df):
    # Raw volatility factor
    df['volatility'] = df['high'] - df['low']
    
    # Raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / df['volatility']
    
    # Simple Momentum for Close Prices
    df['momentum_close'] = df['close'] - df['close'].shift(1)
    
    # Long-Term Trend Indicator
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['long_term_trend'] = df['sma_50'] - df['sma_200']
    
    # Raw volume change factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Volume-Adjusted Breakout Strength
    df['volume_adjusted_breakout_strength'] = df['volatility'] * df['volume'] + (df['close'] - df['open'])
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Summarize the (close - open) * volume over the last 5 days as a raw price-volume composite factor
    df['price_volume_composite'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_composite_5d'] = df['price_volume_composite'].rolling(window=5).sum()
    
    # Raw gap factor
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Spread Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['avg_high_low_spread_20d'] = df['high_low_spread'].rolling(window=20).mean()
    df['spread_momentum'] = df['avg_high_low_spread_20d'] - df['avg_high_low_spread_20d'].shift(1)
    
    # Positive and total money flow
    df['positive_money_flow'] = (df['close'] - df['open']).where(df['close'] > df['open'], 0) * df['volume']
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']
    
    # Transaction size factor
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Price-amount interaction factor
    df['price_amount_interaction'] = (df['close'] - df['open']) * df['transaction_size']
    
    # Momentum factor
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['momentum_factor'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume'])
    
    # Price-volume-amount composite factor
    df['price_volume_amount_composite'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume']) * df['transaction_size']
    
    # Open-High-Low Volatility
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    df['weighted_avg_volatility'] = (df['open_high_diff'] + df['open_low_diff']) * df['volume'] / df['volume'].rolling(window=20).sum()
    
    # Enhanced Price Reversion
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    df['proximity_to_high'] = df['close'] - df['10d_high']
    df['proximity_to_low'] = df['close'] - df['10d_low']
    df['min_proximity'] = df[['proximity_to_high', 'proximity_to_low']].min(axis=1)
    df['inverted_proximity'] = 1 / df['min_proximity'].abs()
    df['std_dev_20d'] = df['close'].pct_change().rolling(window=20).std()
    df['enhanced_price_reversion'] = df
