```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Combine Indicators: Weighted Sum of Range and Movement
    combined_indicator = volume_adjusted_range + open_to_close_movement
    
    # Filter for Strong Moves
    threshold = 100  # This threshold may need to be tuned
    strong_moves = (combined_indicator > threshold).astype(int)
    
    # Calculate Daily High-Low Price Spread
    daily_high_low_spread = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    moving_sum_high_low = daily_high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low.diff()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Close Price
    moving_avg_volume_weighted_close = volume_weighted_close.rolling(window=20).mean()
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = volume_weighted_close - moving_avg_volume_weighted_close
    
    # Final Factor: Strong Moves and Volume-Weighted Momentum
    factor_values = strong_moves * volume_weighted_momentum
    
    return factor_values
```
