import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's close and yesterday's close (Momentum)
    df['momentum'] = df['close'].diff()
    
    # Consider the absolute value of the change (Volatility Insight)
    df['volatility_insight'] = df['momentum'].abs()
    
    # Determine the trend over a short window (e.g., 5 days) using close prices (Short-Term Trend)
    df['short_term_trend'] = df['close'].rolling(window=5).mean()
    
    # Evaluate the current day close relative to the moving average (Trend Strength)
    df['trend_strength'] = df['close'] / df['short_term_trend']
    
    # Calculate the difference between today's high and low (Intra-Day Range)
    df['intra_day_range'] = df['high'] - df['low']
    
    # Combine with the close price to assess closing position within the range (Close Position Index)
    df['close_position_index'] = (df['close'] - df['low']) / df['intra_day_range']
    
    # Compute the ratio of today's high to today's open (Bullish Sentiment)
    df['bullish_sentiment'] = df['high'] / df['open']
    
    # Compare this ratio with the ratio of today's close to today's open (Sentiment Confirmation)
    df['sentiment_confirmation'] = (df['close'] / df['open']) - df['bullish_sentiment']
    
    # Evaluate the relationship between today's volume and the average volume over a period (Volume Anomaly)
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_anomaly'] = df['volume'] / df['avg_volume_5d']
    
    # Check if today's volume is significantly higher or lower than the average (Volume Spike)
    df['volume_spike'] = (df['volume_anomaly'] > 1.5) | (df['volume_anomaly'] < 0.5)
    
    # Analyze the amount traded in relation to the price movement
    df['price_movement'] = df['close'] - df['open']
    df['price_volume_correlation'] = df['amount'] / df['price_movement']
    
    # Create a composite score by summing standardized values of momentum, trend, and volume anomaly (Composite Score)
    df['composite_score'] = df['momentum'] + df['trend_strength'] + df['volume_anomaly']
    
    # Return the composite score as the alpha factor
    return df['composite_score']
