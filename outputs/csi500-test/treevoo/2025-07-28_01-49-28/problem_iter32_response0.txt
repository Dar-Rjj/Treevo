```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average price for each day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the money flow, which is the product of average price and volume
    money_flow = avg_price * df['volume']
    
    # Calculate the 14-day sum of positive and negative money flow
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)].rolling(window=14).sum()
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)].rolling(window=14).sum()
    
    # The money flow ratio is the ratio of positive to negative money flow
    money_flow_ratio = positive_money_flow / negative_money_flow
    
    # The heuristic factor is a function of the money flow ratio that we believe captures momentum
    factor_values = money_flow_ratio - 1
    
    return factor_values
```
