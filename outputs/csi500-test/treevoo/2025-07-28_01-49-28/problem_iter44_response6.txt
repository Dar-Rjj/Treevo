```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Dynamic Price Reversion
    ten_day_moving_high = df['high'].rolling(window=10).max()
    ten_day_moving_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_moving_high
    proximity_to_low = df['close'] / ten_day_moving_low
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    dynamic_price_reversion = 1 / (min_proximity * df['close'].pct_change().rolling(window=20).std())
    
    # Weighted Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff + open_low_diff) * np.sqrt(df['volume'])
    
    # Sum of Exponential Weights on High-Low Spreads
    high_low_spread = df['high'] - df['low']
    ewm_high_low_spread = high_low_spread.ewm(span=20, adjust=False).mean()
    spread_momentum = ewm_high_low_spread.diff()
    
    # Volume-Adjusted Intraday Movement
    volume_adjusted_intraday_movement = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Simple Momentum
    simple_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_daily_range * dynamic_price_reversion * weighted_volatility * spread_momentum
    final_alpha_factor += volume_adjusted_intraday_movement
    final_alpha_factor += simple_momentum
    
    # Enhanced Volume-Adjusted Breakout Strength
    breakout_strength_high = (df['high'] - df['close']) * df['volume']
    breakout_strength_low = (df['close'] - df['low']) * df['volume']
    enhanced_breakout_strength = breakout_strength_high - breakout_strength_low
    
    # Integrated Volume-Adjusted Breakout Indicator
    integrated_breakout_indicator_pos = np.maximum(enhanced_breakout_strength, 0)
    integrated_breakout_indicator_neg = np.minimum(enhanced_breakout_strength, 0)
    
    # Combined Momentum Indicator
    combined_momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    combined_momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    combined_momentum = combined_momentum_5 + combined_momentum_20
    
    # Final Enhanced Alpha Factor
    final_enhanced_alpha_factor = final_alpha_factor * (integrated_breakout_indicator_pos - integrated_breakout_indicator_neg)
    final_enhanced_alpha_factor += combined_momentum
    
    # Filter for Strong Moves
    dynamic_threshold = final_enhanced_alpha_factor.rolling(window=20).quantile(0.9)
    strong_moves = (final_enhanced_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
