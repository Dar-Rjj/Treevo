```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread (Last 20 Days)
    moving_sum_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum_volume_adjusted_high_low_spread - moving_sum_volume_adjusted_high_low_spread.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted Close-Open Spread (Last 10 Days)
    moving_avg_volume_adjusted_close_open_spread = volume_adjusted_close_open_spread.rolling(window=10).mean()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    integrated_spread = moving_avg_volume_adjusted_close_open_spread - moving_avg_volume_adjusted_close_open_spread.shift(1)
    
    # Combine the two momentum factors
    final_factor = adjusted_spread_momentum + integrated_spread
    
    return final_factor
```
