import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 - min_proximity
    
    # Adjust for Volatility
    returns = df['close'].pct_change()
    std_dev_20 = returns.rolling(window=20).std()
    vol_adjusted_reversion = inverted_proximity / std_dev_20
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    high_low_spread = df['high'] - df['low']
    moving_sum_20 = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_20 - moving_sum_20.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_oc = (df['close'] - df['open']) * df['volume']
    
    # Price Action and Volume Factors
    price_volatility = df['high'] - df['low']
    price_strength = (df['close'] - df['open']) / price_volatility
    volume_change = df['volume'] - df['volume'].shift(1)
    price_volume_relationship = (df['volume'] * (df['close'] - df['open']))
    price_gap = df['open'] - df['close'].shift(1)
    positive_money_flow = (df['close'] - df['open']).where(df['close'] > df['open'], 0) * df['volume']
    total_money_flow = (df['close'] - df['open']) * df['volume']
    money_flow_ratio = positive_money_flow / total_money_flow
    amount_per_volume = df['amount'] / df['volume']
    momentum_factor = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Final Alpha Factor
    alpha_factor = (
        volume_adjusted_range * 
        vol_adjusted_reversion * 
        weighted_avg_volatility * 
        spread_momentum * 
        price_strength * 
        volume_change * 
        price_volume_relationship * 
        price_gap * 
        money_flow_ratio * 
        amount_per_volume * 
        momentum_factor * 
        volume_adjusted_oc
    )
    
    # Filter for Strong Moves
    dynamic_threshold = alpha_factor.rolling(window=20).quantile(0.75)
    strong_moves = (alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
