import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Adjusted_Breakout_Strength'] = (df['Daily_Range'] * df['volume']) + df['Open_to_Close_Movement']
    
    # Calculate Moving Average of High-Low Price Spread
    df['High_Low_Spread_MA'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['High_Low_Spread_MA'].diff()
    
    # Incorporate Volume into High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['Daily_Range'] * df['volume']
    
    # Calculate Volume-Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = (df['close'].diff()) * df['volume']
    
    # Integrate Open-High-Low Volatility
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Open_High_Low_Volatility'] = (df['Open_High_Diff'] + df['Open_Low_Diff']) * df['volume']
    
    # Incorporate Enhanced Price Reversion
    df['10_Day_High'] = df['high'].rolling(window=10).max()
    df['10_Day_Low'] = df['low'].rolling(window=10).min()
    df['Proximity_to_High'] = 1 / (df['close'] - df['10_Day_High'])
    df['Proximity_to_Low'] = 1 / (df['close'] - df['10_Day_Low'])
    df['Min_Proximity'] = df[['Proximity_to_High', 'Proximity_to_Low']].min(axis=1)
    df['Volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['Enhanced_Price_Reversion'] = df['Min_Proximity'] / df['Volatility']
    
    # Calculate Moving Sum of High-Low Price Spreads
    df['High_Low_Spread_Sum'] = df['Daily_Range'].rolling(window=20).sum()
    df['Sum_Spread_Momentum'] = df['High_Low_Spread_Sum'].diff()
    
    # Calculate Exponential Moving Average of Momentum
    df['EMA_Momentum'] = df['Volume_Adjusted_Momentum'].ewm(span=10, adjust=False).mean()
    
    # Calculate Relative Strength Index (RSI)
    df['Daily_Returns'] = df['close'].pct_change()
    df['Gains'] = df['Daily_Returns'].where(df['Daily_Returns'] > 0, 0)
    df['Losses'] = -df['Daily_Returns'].where(df['Daily_Returns'] < 0, 0)
    df['Avg_Gain'] = df['Gains'].rolling(window=14).mean()
    df['Avg_Loss'] = df['Losses'].rolling(window=14).mean()
    df['RS'] = df['Avg_Gain'] / df['Avg_Loss']
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Amount per Unit Close Price Rate of Change
    n = 5  # Example: 5 days
    df['Amount_Sum'] = df['amount'].rolling(window=n).sum()
    df['Amount_per_Unit_Close'] = df['Amount_Sum'] / df['close']
    df['Prev_Amount_per_Unit_Close'] = df['Amount_per_Unit_Close'].shift(n)
    df['ROC_Amount_per_Unit_Close'] = df['Amount_per_Unit_Close'] - df['Prev_Amount_per_Unit_Close']
    
    # Combine Indicators
    df['Combined_Indicator'] = 1.0 * df['Volume_Adjusted_Breakout_Strength'] + df['Sum_Spread_Momentum'] + df['Volume_Adjusted_Momentum']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Combined_Indicator'] 
        * df['EMA_Momentum'] 
        * df['RSI'] 
        * df['Enhanced_Price_Reversion'] 
        * df['Open_High_Low_Volatility'] 
        * df['Sum_Spread_Momentum'] 
        * df
