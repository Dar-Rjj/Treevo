```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Daily Returns
    daily_returns = df['close'].diff()
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Combine Returns and Volume Ratio
    adjusted_returns = (daily_returns * volume_ratio).rolling(window=lookback_period).sum()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Combine Adjusted Returns and Volume-Adjusted High-Low Spread
    combined_indicator = adjusted_returns + volume_adjusted_high_low_spread.rolling(window=lookback_period).sum()
    
    # Finalize VAMHLI
    overall_average_return = df['close'].pct_change().rolling(window=lookback_period).mean()
    vamhli = combined_indicator - overall_average_return
    
    return vamhli
```
