```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted High-Low Spread (last 20 days)
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=20).mean()
    
    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = volume_adjusted_high_low_spread - ma_volume_adjusted_high_low_spread.shift(1)
    volume_and_momentum_enhanced_high_low_spread = adjusted_spread_momentum
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted Close-Open Spread (last 10 days)
    moving_sum_volume_adjusted_close_open_spread = volume_adjusted_close_open_spread.rolling(window=10).sum()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    integrated_spread_momentum = volume_adjusted_close_open_spread - moving_sum_volume_adjusted_close_open_spread.shift(1)
    volume_momentum_close_open_integrated_spread = integrated_spread_momentum
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Smoothed Momentum
    daily_returns = df['close'].pct_change()
    smoothed_momentum = daily_returns.rolling(window=10).sum() / df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (min_proximity + 1e-6)  # avoid division by zero
    volatility = df['close'].pct_change().rolling(window=10).std()
    adjusted_proximity = inverted_proximity / volatility
    
    # Combine Factors
    combined_factor = (
        volume_and_momentum_enhanced_high_low_spread +
        volume_momentum_close_open_integrated_spread +
        volume_weighted_momentum
    ) * price_change_ratio * smoothed_momentum * adjusted_proximity
    
    return combined_factor
```
