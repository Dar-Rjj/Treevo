def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = df['volume'] / df['volume'].mean()
    volume_adjusted_breakout_strength = (daily_range * volume_factor) + open_to_close_movement
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Momentum
    momentum = df['close'].diff(1)
    volume_adjusted_high_low_spread_and_momentum = momentum + volume_adjusted_high_low_spread
    
    # Combine Indicators
    weight_breakout_strength = 0.6
    combined_indicator = (volume_adjusted_breakout_strength * weight_breakout_strength) + volume_adjusted_high_low_spread_and_momentum
    
    # Filter for Strong Moves
    threshold = combined_indicator.mean()
    strong_move_indicator = (combined_indicator > threshold).astype(int)
    
    return strong_move_indicator
