```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = daily_range * df['volume'] + open_to_close
    
    # Spread Momentum
    spread_momentum = (df['high'] - df['low']).rolling(window=20).mean().diff()
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = simple_momentum * df['volume']
    
    # Incorporate Relative Strength
    relative_strength = df['close'] / df['close'].rolling(window=20).mean()
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (min_proximity.abs() + 1e-6)
    price_reversion = inverted_proximity / df['close'].pct_change().rolling(window=20).std()
    
    # Combined Indicators
    combined_indicator = 0.5 * volume_adjusted_breakout_strength + spread_momentum + volume_adjusted_momentum
    combined_indicator *= relative_strength
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicator * price_reversion * open_high_low_volatility
    
    # Analyze close prices for predictive signals
    momentum = simple_momentum
    short_term_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Examine open, high, low, and close prices for intra-day patterns
    intra_day_range = df['high'] - df['low']
    close_position_index = (df['close'] - df['low']) / intra_day_range
    bullish_sentiment = df['high'] / df['open']
    sentiment_confirmation = df['close'] / df['open']
    
    # Investigate volume and amount data for additional insights
    volume_anomaly = df['volume'] / df['volume'].rolling(window=20).mean()
    price_volume_correlation = df['amount'] / (df['close'] - df['open'])
    
    # Combine multiple factors to form composite indicators
    composite_score = (momentum + short_term_trend + volume_anomaly) / 3
    
    # Filter for Strong Moves
    dynamic_threshold = 0.5 * combined_indicator.std()
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)
    
    return final_alpha_factor
```
