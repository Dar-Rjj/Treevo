```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low = high_low_spread * df['volume']
    
    # Moving Average of Volume-Adjusted High-Low Spread (Last 20 Days)
    ma_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=20).mean()
    
    # Volume and Momentum Enhanced High-Low Spread
    momentum_enhanced_high_low = volume_adjusted_high_low - ma_volume_adjusted_high_low.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Volume-Adjusted Close-Open Spread
    volume_adjusted_close_open = close_open_spread * df['volume']
    
    # Moving Sum of Volume-Adjusted Close-Open Spread (Last 10 Days)
    sum_volume_adjusted_close_open = volume_adjusted_close_open.rolling(window=10).sum()
    
    # Volume, Momentum, and Close-Open Integrated Spread
    integrated_close_open = volume_adjusted_close_open - sum_volume_adjusted_close_open.shift(1)
    
    # Volume-Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']
    
    # Smoothed Momentum
    daily_returns = df['close'].pct_change()
    smoothed_momentum = daily_returns.rolling(window=10).sum() / df['volume']
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = proximity_to_high.abs().where(proximity_to_high.abs() < proximity_to_low.abs(), proximity_to_low.abs())
    min_proximity_inverted = 1 / min_proximity
    
    # Adjust for Volatility
    std_dev_daily_returns = daily_returns.rolling(window=10).std()
    volatility_adjusted_proximity = min_proximity_inverted / std_dev_daily_returns
    
    # Enhanced Volatility Adjustment
    ema_daily_returns = daily_returns.ewm(span=10).mean()
    enhanced_volatility_adjusted_proximity = min_proximity_inverted / ema_daily_returns
    
    # Combine Factors
    combined_factors = (momentum_enhanced_high_low + volume_weighted_momentum) * price_change_ratio * smoothed_momentum * enhanced_volatility_adjusted_proximity + integrated_close_open
    
    # Calculate 5-Day High-Low Spread
    five_day_high = df['high'].rolling(window=5).max()
    five_day_low = df['low'].rolling(window=5).min()
    five_day_high_low_spread = five_day_high - five_day_low
    
    # Volume-Adjusted 5-Day High-Low Spread
    volume_adjusted_5_day_high_low = five_day_high_low_spread * df['volume']
    
    # Moving Average of Volume-Adjusted 5-Day High-Low Spread (Last 10 Days)
    ma_volume_adjusted_5_day_high_low = volume_adjusted_5_day_high_low.rolling(window=10).mean()
    
    # Volume and Momentum Enhanced 5-Day High-Low Spread
    momentum_enhanced_5_day_high_low = volume_adjusted_5_day_high_low - ma_volume_adjusted_5_day_high_low.shift(1)
    
    # Volume-Weighted 5-Day Momentum
    five_day_momentum = df['close'] - df['close'].shift(5)
    volume_weighted_5_day_momentum = five_day_momentum * df['volume']
    
    # Smoothed 5-Day Momentum
    five_day_daily_returns = df['close'].pct_change(periods=5)
    smoothed_5_day_momentum = five_day_daily_returns.rolling(window=10).sum() / df['volume']
    
    # Integrate 5-Day Momentum into Combined Factors
    integrated_5_day_factors = (momentum_enhanced_5_day_high_low + volume_weighted_5_day_momentum) * price_change_ratio * smoothed_5_day_momentum * enhanced_volatility_adjusted_proximity
    final_combined_factors = combined_factors + integrated_5_day_factors
    
    return final_combined_factors
```
