```python
def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Return
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Calculate Volume Percentage Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust for Volume Trend
    adjusted_high_low_return = high_low_return * volume_change
    
    # Integrate Price and Volume Signals over recent days (e.g., 5 days)
    integrated_signal = adjusted_high_low_return.rolling(window=5).sum()
    
    return integrated_signal
```
