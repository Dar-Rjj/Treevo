import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    weighted_momentum = momentum * df['volume']
    
    # Calculate Smoothed Momentum
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    smoothed_momentum = daily_returns.rolling(window=20).sum() / df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    proximity = 1 / pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    
    # Adjust for Volatility
    volatility = df['close'].pct_change().rolling(window=20).std()
    adjusted_proximity = proximity / volatility
    
    # Combine Factors
    factor = (
        (volume_adjusted_spread + weighted_momentum) 
        * price_change_ratio 
        * smoothed_momentum 
        * adjusted_proximity
    )
    
    return factor.rename("Volume-Adjusted High-Low Spread, Weighted Momentum, and Proximity to 10-Day Range")
