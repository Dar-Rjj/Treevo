```python
def heuristics_v2(df):
    # Calculate the return of the close price
    df['return'] = df['close'].pct_change()
    
    # Calculate the moving average of the return
    df['ma_return'] = df['return'].rolling(window=5).mean()
    
    # Calculate the difference between high and low, representing the daily volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to amount, which can be seen as a proxy for average price
    df['volume_to_amount'] = df['volume'] / df['amount']
    
    # Calculate the alpha factor by combining the moving average return, daily volatility and volume to amount ratio
    df['alpha_factor'] = (df['ma_return'] * df['daily_volatility']) + df['volume_to_amount']
    
    return df['alpha_factor'].dropna()
```
