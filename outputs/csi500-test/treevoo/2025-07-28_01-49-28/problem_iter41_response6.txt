```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * np.sqrt(df['volume'])) + open_to_close
    
    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_20 = high_low_spread.rolling(window=20).sum()
    spread_momentum = moving_sum_20 - moving_sum_20.shift(1)
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume = (df['volume'] + df['volume'].shift(1)) / 2
    adjusted_momentum_by_volume = simple_momentum * (df['volume'] / avg_volume)
    
    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = np.minimum(np.abs(proximity_to_high), np.abs(proximity_to_low))
    twenty_day_volatility = df['close'].rolling(window=20).std()
    inverted_min_proximity = 1 / (min_proximity + 1e-6)
    adjusted_proximity = inverted_min_proximity / (twenty_day_volatility + 1e-6)
    
    # Integrated Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    max_volatility = np.maximum(open_high_diff, open_low_diff)
    volume_pct_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = max_volatility * np.abs(volume_pct_change)
    
    # Enhanced Momentum Factor
    smoothed_volume = df['volume'].ewm(span=10).mean()
    adjusted_momentum = (df['close'] - df['close'].shift(10)) * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_amounts = df['amount'].rolling(window=10).sum()
    current_ratio = sum_amounts / df['close']
    previous_ratio = df['amount'].rolling(window=20).sum().shift(10) / df['close'].shift(10)
    differential = (current_ratio - previous_ratio) / (previous_ratio + 1e-6)
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength + 
        spread_momentum + 
        adjusted_momentum_by_volume + 
        combined_factors + 
        adjusted_volatility + 
        differential
    ) * price_change_ratio
    
    # Filter for Strong Moves
    strong_moves = (integrated_alpha_factor > 0).astype(int)
    
    # Price Trend Indicator
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_trend_score = (long_term_momentum - short_term_momentum) * integrated_alpha_factor
    
    # Price Oscillation
    short_term_high = df['high'].rolling(window=10).max()
    short_term_low = df['low'].rolling(window=10).min()
    price_oscillation = (short_term_high - short_term_low) / df['close'].shift(1)
    
    # Enhanced Alpha Factor
    enhanced_alpha_factor = (
        0.7 * integrated_alpha_factor + 
        0.3 * (price_trend_score + price_oscillation)
    )
    
    return enhanced_alpha_factor
```
