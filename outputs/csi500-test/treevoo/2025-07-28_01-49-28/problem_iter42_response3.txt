```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Daily Range
    df['Adjusted_Daily_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Price Reversion
    df['10D_High'] = df['high'].rolling(window=10).max()
    df['10D_Low'] = df['low'].rolling(window=10).min()
    df['Proximity_to_10D_High'] = df['close'] / df['10D_High']
    df['Proximity_to_10D_Low'] = df['close'] / df['10D_Low']
    df['Inverted_Proximity'] = 1 / ((df['Proximity_to_10D_High'] + df['Proximity_to_10D_Low']) / 2)
    df['20D_Std_Returns'] = df['close'].pct_change().rolling(window=20).std()
    df['Price_Reversion'] = df['Inverted_Proximity'] / df['20D_Std_Returns']
    
    # Open-High-Low Volatility
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Weighted_Average_Volatility'] = (df['Open_High_Diff'] * df['volume'] + df['Open_Low_Diff'] * df['volume']) / (2 * df['volume'])
    
    # Spread Momentum
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Current_Moving_Sum'] = df['High_Low_Spread'].rolling(window=5).sum()
    df['Previous_Moving_Sum'] = df['Current_Moving_Sum'].shift(1)
    df['Spread_Momentum'] = df['Current_Moving_Sum'] - df['Previous_Moving_Sum']
    
    # Volume-Adjusted Open-to-Close Movement
    df['Volume_Adjusted_Open_Close'] = (df['close'] - df['open']) * df['volume']
    
    # Simple Momentum
    df['Simple_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Trend Strength
    df['50D_MA'] = df['close'].rolling(window=50).mean()
    df['200D_MA'] = df['close'].rolling(window=200).mean()
    df['Trend_Strength'] = df['50D_MA'] - df['200D_MA']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Adjusted_Daily_Range'] * 
                          df['Price_Reversion'] * 
                          df['Weighted_Average_Volatility'] * 
                          df['Spread_Momentum'] + 
                          df['Volume_Adjusted_Open_Close'] + 
                          df['Simple_Momentum']) * df['Trend_Strength']
    
    # Filter for Strong Moves
    dynamic_threshold = df['Alpha_Factor'].quantile(0.75)  # Example threshold
    df['Filtered_Alpha_Factor'] = df['Alpha_Factor'].apply(lambda x: 1 if x > dynamic_threshold else 0)
    
    return df['Filtered_Alpha_Factor']
```
