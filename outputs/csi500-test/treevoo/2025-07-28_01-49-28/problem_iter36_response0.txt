```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Daily Range
    df['AdjDailyRange'] = (df['high'] - df['low']) * df['volume']
    
    # Dynamic Price Reversion
    df['10DayHigh'] = df['high'].rolling(window=10).max()
    df['10DayLow'] = df['low'].rolling(window=10).min()
    df['ProxTo10DayHigh'] = df['close'] / df['10DayHigh']
    df['ProxTo10DayLow'] = df['close'] / df['10DayLow']
    df['MinProximity'] = df[['ProxTo10DayHigh', 'ProxTo10DayLow']].min(axis=1)
    df['20DayStdRet'] = df['close'].pct_change().rolling(window=20).std()
    df['DynamicReversion'] = 1 / (df['MinProximity'] * df['20DayStdRet'])
    
    # Weighted Open-High-Low Volatility
    df['OH_Diff'] = df['high'] - df['open']
    df['OL_Diff'] = df['open'] - df['low']
    df['WeightedVolatility'] = (df['OH_Diff'] + df['OL_Diff']) / (2 * np.sqrt(df['volume']))
    
    # Sum of Exponential Weights on High-Low Spreads
    df['HL_Spreads'] = df['high'] - df['low']
    df['ExpWeightedSum_HL'] = df['HL_Spreads'].ewm(span=20, adjust=False).mean()
    
    # Spread Momentum
    df['SpreadMomentum'] = df['ExpWeightedSum_HL'].diff()
    
    # Volume-Adjusted Intraday Movement
    df['IntradayMovement'] = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Simple Momentum
    df['SimpleMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['AdjDailyRange'] *
        df['DynamicReversion'] *
        df['WeightedVolatility'] *
        df['SpreadMomentum']
    ) + df['IntradayMovement'] + df['SimpleMomentum']
    
    # Filter for Strong Moves
    dynamic_threshold = df['FinalAlphaFactor'].quantile(0.75)
    df['StrongMoveIndicator'] = (df['FinalAlphaFactor'] > dynamic_threshold).astype(int)
    
    return df['StrongMoveIndicator']
```
