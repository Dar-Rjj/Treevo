import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Simple Momentum
    simple_momentum = df['close'].diff()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Adjusted Momentum by Volume
    adjusted_momentum_by_volume = simple_momentum * (average_volume / df['volume'])
    
    # Combine Volume-Adjusted High-Low Spread and Volume-Adjusted Momentum
    combined_factor = volume_adjusted_high_low_spread + adjusted_momentum_by_volume
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Multiply the Sum by Price Change Ratio
    combined_factor *= price_change_ratio
    
    # Calculate Exponential Moving Average of Momentum
    ema_momentum = adjusted_momentum_by_volume.ewm(span=10, adjust=False).mean()
    
    # Multiply the Result by EMA-Momentum
    combined_factor *= ema_momentum
    
    # Calculate Relative Strength Index (RSI)
    daily_returns = df['close'].pct_change()
    positive_returns = daily_returns.clip(lower=0)
    negative_returns = -daily_returns.clip(upper=0)
    avg_gain = positive_returns.ewm(com=13, adjust=False).mean()
    avg_loss = negative_returns.ewm(com=13, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Multiply the Final Result by RSI
    combined_factor *= rsi
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    distance_to_high = ten_day_high - df['close']
    distance_to_low = df['close'] - ten_day_low
    min_proximity = np.minimum(distance_to_high, distance_to_low)
    inverted_proximity = 1 / (min_proximity + 1e-6)
    std_20_days = daily_returns.rolling(window=20).std()
    adjusted_proximity = inverted_proximity / std_20_days
    
    # Multiply the Final Result by Adjusted Proximity
    combined_factor *= adjusted_proximity
    
    # Amount per Unit Close Price Rate of Change
    n = 10
    amount_per_unit_close = df['amount'].rolling(window=n).sum() / df['close']
    previous_amount_per_unit_close = df['amount'].rolling(window=n).sum().shift(n) / df['close'].shift(n)
    amount_rate_of_change = amount_per_unit_close - previous_amount_per_unit_close
    
    # Multiply the Final Result by Amount per Unit Close Price Rate of Change
    combined_factor *= amount_rate_of_change
    
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    threshold = 5 * df['volume'].rolling(window=10).std()
    breakout_strength = (volume_adjusted_breakout_strength > threshold).astype(int)
    
    # Multiply the Final Result by Volume-Adjusted Breakout Strength
    combined_factor *= breakout_strength
    
    # Label as Integrated Alpha with Enhanced Momentum, Volume-Price Dynamics, and RSI
    return pd.Series(combined_factor, index=df.index, name='Integrated Alpha')
