```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Close Price Trend Indicator
    n = 10  # Lookback period
    ema_close = df['close'].ewm(span=n, adjust=False).mean()
    vol_rolling_sum = df['volume'].rolling(window=n).sum()
    volume_weighted_factor = df['volume'] / vol_rolling_sum
    volume_adjusted_ema = ema_close * volume_weighted_factor

    # Open-High-Low Volatility Index with Volume Adjustment
    high_low_diff = df['high'] - df['low']
    open_med_diff = (df['open'] - (df['high'] + df['low'])/2).abs()
    volatility_index = high_low_diff + open_med_diff
    volatility_index = volatility_index * (df['volume'] / df['volume'].rolling(window=5).mean())

    # Price-Amount Efficiency Ratio
    abs_change_close = df['close'].diff().abs().rolling(window=n).sum()
    amount_efficiency_ratio = df['amount'].rolling(window=n).sum() / abs_change_close
    previous_efficiency_ratio = df['amount'].shift(n).rolling(window=n).sum() / df['close'].shift(n).diff().abs().rolling(window=n).sum()
    efficiency_ratio_slope = (amount_efficiency_ratio - previous_efficiency_ratio) / n

    # Combine all alpha factors
    combined_factor = (volume_adjusted_ema.rank() + 
                       volatility_index.rank(ascending=False) + 
                       efficiency_ratio_slope.rank()) / 3
    
    return combined_factor
```
