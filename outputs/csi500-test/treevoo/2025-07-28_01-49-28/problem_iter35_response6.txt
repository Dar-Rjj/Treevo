import pandas as pd

def heuristics_v2(df):
    # Calculate raw volatility factor
    df['volatility'] = df['high'] - df['low']
    
    # Calculate raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / df['volatility']
    
    # Calculate raw volume change factor
    df['volume_change'] = df['volume'].diff()
    
    # Calculate raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Calculate raw price-volume composite factor
    df['price_volume_composite'] = (df['close'] - df['open']).mul(df['volume']).rolling(window=5).sum()
    
    # Calculate raw gap factor
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate positive and total money flow
    df['positive_money_flow'] = ((df['close'] > df['open']) * (df['close'] - df['open']) * df['volume']).fillna(0)
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate raw money flow index
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']
    
    # Calculate raw transaction size factor
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Calculate raw momentum factor
    df['momentum'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Calculate 20-day percentage change in closing prices
    df['20d_pct_change'] = df['close'].pct_change(periods=20)
    
    # Calculate 10-day rate of change (ROC) of closing prices
    df['10d_roc'] = df['close'].pct_change(periods=10)
    
    # Calculate standard deviation of log-returns over a 30-day window
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['30d_log_return_std'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the 14-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate 10-day high and low prices
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    
    # Calculate proximity to 10-day range
    df['prox_to_10d_high'] = df['close'] / df['10d_high']
    df['prox_to_10d_low'] = df['close'] / df['10d_low']
    
    # Invert minimum proximity
    df['min_prox_inv'] = 1 / df[['prox_to_10d_high', 'prox_to_10d_low']].min(axis=1)
    
    # Adjust for 20-day volatility
    df['20d_vol'] = df['close'].pct_change().rolling(window=20).std()
    df['adj_min_prox_inv'] = df['min_prox_inv'] / df['20d_vol']
    
    # Open-High Difference
    df['open_high_diff'] = df['high'] - df['open']
    
    # Open-Low Difference
    df['open_low_diff'] = df['open'] - df['low']
    
    # Weighted Average by Volume
    df['weighted_avg'] = (df['open_high_diff'] + df['open_low_diff']) * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    df['amount_per_close'] = df['amount'] / df['close']
    df['roc_amount_per_close'] = df['amount_per_close'].pct_change(periods=5)
    
    # Final Alpha Factor
    alpha_factor = (df['adj_min_prox_inv'] * df['momentum'] * df['roc_amount_per_close'])
    
    return alpha_factor.dropna()
