```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted High-Low Spread (20 days)
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=20).mean()
    
    # Calculate Adjusted Spread Momentum
    momentum_high_low_spread = volume_adjusted_high_low_spread - ma_volume_adjusted_high_low_spread.shift(1)
    momentum_high_low_spread.name = 'Volume and Momentum Enhanced High-Low Spread'
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted Close-Open Spread (10 days)
    sum_volume_adjusted_close_open_spread = volume_adjusted_close_open_spread.rolling(window=10).sum()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    momentum_close_open_spread = volume_adjusted_close_open_spread - sum_volume_adjusted_close_open_spread.shift(1)
    momentum_close_open_spread.name = 'Volume, Momentum, and Close-Open Integrated Spread'
    
    # Calculate Weighted Momentum
    weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    weighted_momentum.name = 'Volume-Weighted Momentum'
    
    # Calculate Smoothed Momentum
    smoothed_momentum = df['close'].pct_change().rolling(window=10).sum() / df['volume']
    smoothed_momentum.name = 'Smoothed Momentum'
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_high_inverted = 1 / proximity_to_high
    
    # Adjust for Volatility
    volatility = df['close'].pct_change().rolling(window=10).std()
    enhanced_proximity = proximity_to_high_inverted / volatility
    
    # Enhanced Volatility Adjustment
    ema_volatility = df['close'].pct_change().ewm(span=10).mean()
    enhanced_proximity_final = enhanced_proximity / ema_volatility
    
    # Combine Factors
    combined_factor = (
        (momentum_high_low_spread + weighted_momentum) * 
        price_change_ratio * 
        smoothed_momentum * 
        enhanced_proximity_final
    )
    
    return combined_factor
```
