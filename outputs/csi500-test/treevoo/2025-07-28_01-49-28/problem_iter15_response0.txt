import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    combined_indicator = volume_adjusted_range + open_to_close_movement
    
    # Filter for Strong Moves
    strong_move = (combined_indicator > 0).astype(int)
    
    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    moving_sum_high_low_spread = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread.diff(1)
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Close Price
    ma_volume_weighted_close = volume_weighted_close.rolling(window=20).mean()
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = volume_weighted_close - ma_volume_weighted_close
    
    # Calculate Average True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    average_true_range = true_range.rolling(window=20).mean()
    
    # Calculate True Range Volatility
    true_range_volatility = average_true_range / volume_weighted_close
    
    # Calculate Combined Momentum-Volatility Factor
    combined_momentum_volatility_factor = volume_weighted_momentum / true_range_volatility
    
    return combined_momentum_volatility_factor
