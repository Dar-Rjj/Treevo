```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Daily Range
    adj_daily_range = (df['high'] - df['low']) * df['volume']

    # Price Reversion
    high_15 = df['high'].rolling(window=15).max()
    low_15 = df['low'].rolling(window=15).min()
    proximity_to_high = df['close'] / high_15
    proximity_to_low = df['close'] / low_15
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    std_25 = df['close'].pct_change().rolling(window=25).std()
    price_reversion = 1 / (min_proximity * std_25)

    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    vol_weight = np.sqrt(df['volume'])
    ohl_volatility = (open_high_diff + open_low_diff) * vol_weight

    # Moving Sum of High-Low Spreads
    hl_spread = df['high'] - df['low']
    sum_hl_spreads = hl_spread.rolling(window=25).sum()

    # Spread Momentum
    spread_momentum = (sum_hl_spreads - sum_hl_spreads.shift(1)) / sum_hl_spreads.shift(1)

    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_oc = (df['close'] - df['open']) * np.sqrt(df['volume'])

    # Simple Momentum
    simple_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Final Alpha Factor
    final_alpha = (adj_daily_range * price_reversion * ohl_volatility * spread_momentum 
                   + volume_adjusted_oc + simple_momentum)

    # Filter for Strong Moves
    dynamic_threshold = final_alpha.quantile(0.75)
    final_alpha_filtered = (final_alpha > dynamic_threshold).astype(int)

    return final_alpha_filtered
```
