```python
import pandas as pd

def heuristics_v2(df):
    # Calculate (high - low) as a raw volatility factor
    df['volatility'] = df['high'] - df['low']
    
    # Calculate (close - open) / (high - low) as a raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate today's volume - yesterday's volume as a raw volume change factor
    df['volume_change'] = df['volume'].diff()
    
    # Calculate (volume * (close - open)) as a raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Summarize the (close - open) * volume over the last 5 days as a raw price-volume composite factor
    df['price_volume_5d'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_5d'] = df['price_volume_5d'].rolling(window=5).sum()
    
    # Calculate (open - previous close) as a raw gap factor
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Define positive money flow as (close - open) * volume when close > open, otherwise 0
    df['positive_money_flow'] = ((df['close'] > df['open']) * (df['close'] - df['open']) * df['volume'])
    
    # Define total money flow as (close - open) * volume, regardless of price direction
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Use the ratio of positive to total money flow as a raw money flow index
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow'].abs()
    
    # Use the (amount / volume) as a raw transaction size factor
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Calculate (close - open) * (amount / volume) as a raw price-amount interaction factor
    df['price_amount_interaction'] = (df['close'] - df['open']) * (df['amount'] / df['volume'])
    
    # Calculate the (close - open) * (volume / average volume) as a raw momentum factor
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['momentum'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume'])
    
    # Calculate the (close - open) * (volume / average volume) * (amount / volume) as a raw price-volume-amount composite factor
    df['price_volume_amount_composite'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume']) * (df['amount'] / df['volume'])
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['volatility'] + 
        df['price_movement'] + 
        df['volume_change'] + 
        df['volume_price_interaction'] + 
        df['price_volume_5d'] + 
        df['gap'] + 
        df['money_flow_index'] + 
        df['transaction_size'] + 
        df['price_amount_interaction'] + 
        df['momentum'] + 
        df['price_volume_amount_composite']
    )
    
    return alpha_factor
```
