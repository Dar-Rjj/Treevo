```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range + open_to_close_movement) * df['volume']

    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    current_day_moving_sum = high_low_spread.rolling(window=10).sum()
    previous_day_moving_sum = high_low_spread.shift(1).rolling(window=10).sum()
    high_low_spread_momentum = current_day_moving_sum - previous_day_moving_sum

    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume

    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = df['close'] - ten_day_high
    proximity_to_ten_day_low = ten_day_low - df['close']
    min_proximity = pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    inverted_min_proximity = -min_proximity
    twenty_day_volatility = df['close'].rolling(window=20).std()
    adjusted_inverted_min_proximity = inverted_min_proximity / twenty_day_volatility

    # Integrate Open-High-Low Volatility
    open_high_difference = df['high'] - df['open']
    open_low_difference = df['open'] - df['low']
    weighted_average_by_volume = (open_high_difference + open_low_difference) * df['volume'] / 2

    # Combine all factors into a single alpha factor
    alpha_factor = (volume_adjusted_breakout_strength + 
                    high_low_spread_momentum + 
                    volume_adjusted_momentum + 
                    adjusted_inverted_min_proximity + 
                    weighted_average_by_volume)

    return alpha_factor
```
