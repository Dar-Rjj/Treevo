import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 - min_proximity
    
    # Adjust for Volatility
    daily_returns = df['close'].pct_change()
    twenty_day_volatility = daily_returns.rolling(window=20).std()
    reversion_factor = inverted_proximity / twenty_day_volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_open_high_low = (open_high_diff + open_low_diff) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Weighted Average Spread
    volume_weighted_average_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_average_spread = (df['high'] - df['low']) * volume_weighted_average_volume
    
    # Enhanced Momentum Factor
    simple_momentum = df['close'] - df['close'].shift(20)
    smoothed_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_momentum = simple_momentum * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_low_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_of_amounts = df['amount'].rolling(window=20).sum()
    current_ratio = sum_of_amounts / df['close']
    previous_ratio = (df['amount'].rolling(window=20).sum().shift(20)) / df['close'].shift(20)
    differential = (current_ratio - previous_ratio) / previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Volume-Weighted Moving Average Crossover
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwma_crossover = (short_term_vwma > long_term_vwma).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_daily_range * reversion_factor * weighted_open_high_low * spread_momentum
        + volume_adjusted_open_close
        + volume_weighted_average_spread
        + adjusted_momentum
        + combined_factors
        + differential
    ) * price_change_ratio * vwma_crossover
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).quantile(0.75)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
