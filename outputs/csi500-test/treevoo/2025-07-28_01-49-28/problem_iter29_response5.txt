```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    proximity_to_low = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    volatility = df['close'].pct_change().rolling(window=20).std()
    adjusted_reversion = inverted_proximity / volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff + open_low_diff) * df['volume']
    open_high_low_volatility = weighted_avg.rolling(window=20).mean()
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread.diff()
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = volume_adjusted_range + volume_adjusted_open_close
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()
    
    # Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = (volume_adjusted_range * adjusted_reversion * open_high_low_volatility * spread_momentum +
                          volume_adjusted_breakout_strength + 
                          volume_adjusted_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.quantile(0.75)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
