def heuristics_v2(data):
    """
    Generate an alpha factor based on the given market data.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
        
    Returns:
        pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the relative strength index (RSI) for the close prices
    delta = data['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the percentage change in volume
    pct_change_volume = data['volume'].pct_change().fillna(0)
    
    # Calculate the money flow index (MFI)
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    raw_money_flow = typical_price * data['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine RSI and MFI to create a novel alpha factor
    factor = rsi * mfi
    
    return factor
