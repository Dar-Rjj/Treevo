```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Adjusted_Breakout_Strength'] = (df['Daily_Range'] * np.sqrt(df['volume'])) + df['Open_to_Close_Movement']
    
    # Calculate 5-day Momentum
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Momentum_Score'] = np.where(df['close'] > df['5_day_EMA'], 1, -1)
    
    # Combine Indicators
    df['Combined_Indicator'] = (df['Volume_Adjusted_Breakout_Strength'] * df['Momentum_Score']) + df['Open_to_Close_Movement']
    
    # Filter for Strong Moves
    threshold = 0.5 * df['Combined_Indicator'].std()
    df['Strong_Move'] = np.where(df['Combined_Indicator'] > threshold, 1, 0)
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Trend-Following Signal
    df['Trend_Following_Signal'] = df['SMA_5'] - df['SMA_20']
    
    # Analyze Past Returns
    df['daily_return'] = df['close'].pct_change()
    df['weekly_return'] = df['close'].pct_change(periods=5)
    df['monthly_return'] = df['close'].pct_change(periods=21)
    df['Weighted_Sum_of_Returns'] = 0.5 * df['daily_return'] + 0.3 * df['weekly_return'] + 0.2 * df['monthly_return']
    
    # Measure Overnight Sentiment
    df['Overnight_Sentiment'] = df['open'] - df['close'].shift(1)
    
    # Evaluate Volume-Based Factors
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    df['OBV_Factor'] = df['OBV'] * df['close'].pct_change()
    
    # Calculate Volume-Adjusted High-Low Spread
    df['Volume_Adjusted_HL_Spread'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    df['Volume_Weighted_Momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Price Change Ratio
    df['Price_Change_Ratio'] = df['close'] / df['open']
    
    # Incorporate Volatility Index
    df['Volatility_Index'] = df['daily_return'].rolling(window=30).std()
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Strong_Move'] +
        df['Trend_Following_Signal'] +
        df['Weighted_Sum_of_Returns'] +
        df['Overnight_Sentiment'] +
        df['OBV_Factor'] +
        df['Volume_Adjusted_HL_Spread'] +
        df['Volume_Weighted_Momentum'] -
        df['Volatility_Index']
    ) * df['Price_Change_Ratio']
    
    return df['Final_Alpha_Factor']
```
