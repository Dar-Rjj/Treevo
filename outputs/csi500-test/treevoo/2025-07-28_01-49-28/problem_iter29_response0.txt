```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range and Open-to-Close Movement
    daily_range = df['high'] - df['low']
    price_movement = (df['close'] - df['open']) / daily_range
    
    # Assess Volume Changes and Market Sentiment
    volume_change = df['volume'].diff()
    volume_price_interaction = df['volume'] * (df['close'] - df['open'])
    
    # Summarize (close - open) * volume over the last 5 days
    price_volume_composite = (df['close'] - df['open']) * df['volume']
    price_volume_composite = price_volume_composite.rolling(window=5).sum()
    
    # Investigate Price Gaps
    price_gap = df['open'] - df['close'].shift(1)
    
    # Incorporate Money Flow and Transaction Size
    positive_money_flow = ((df['close'] - df['open']) * df['volume']).apply(lambda x: x if x > 0 else 0)
    total_money_flow = (df['close'] - df['open']) * df['volume']
    money_flow_index = positive_money_flow / total_money_flow
    
    transaction_size = df['amount'] / df['volume']
    
    # Integrate High-Low Spread and Momentum
    high_low_spread_20 = daily_range.rolling(window=20).sum()
    spread_momentum = high_low_spread_20 - high_low_spread_20.shift(1)
    
    # Adjust for 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = -min_proximity
    twenty_day_volatility = df['close'].rolling(window=20).std()
    adjusted_proximity = inverted_proximity / twenty_day_volatility
    
    # Volume-Adjusted Momentum with Open-to-Close Movement
    simple_momentum = df['close'] - df['close'].shift(20)
    average_volume = df['volume'].rolling(window=20).mean()
    adjusted_momentum = simple_momentum * (df['volume'] / average_volume)
    open_close_movement = df['close'] - df['open']
    combined_indicator = 0.5 * adjusted_momentum + 0.5 * open_close_movement
    strong_move = (combined_indicator > 0).astype(int)
    
    # Enhance with Amount per Unit Close Price Rate of Change
    amount_sum_n_days = df['amount'].rolling(window=10).sum()
    amount_per_unit_close = amount_sum_n_days / df['close']
    amount_sum_previous_period = df['amount'].rolling(window=20).sum().shift(10)
    amount_per_unit_close_previous = amount_sum_previous_period / df['close'].shift(10)
    amount_rate_of_change = amount_per_unit_close - amount_per_unit_close_previous
    
    # Final Alpha Factor Composition
    alpha_factor = (daily_range * 
                    spread_momentum * 
                    adjusted_proximity * 
                    adjusted_momentum * 
                    open_close_movement * 
                    strong_move * 
                    amount_rate_of_change)
    
    return alpha_factor
```
