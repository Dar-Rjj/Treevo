```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = (df['volume'] / df['volume'].rolling(window=20).mean())
    volume_adjusted_breakout_strength = daily_range * volume_factor + open_close_movement
    
    # Calculate Moving Averages
    short_term_sma = df['close'].rolling(window=5).mean()
    long_term_sma = df['close'].rolling(window=20).mean()
    
    # Compute Trend-Following Signal
    sma_difference = short_term_sma - long_term_sma
    
    # Investigate Momentum
    daily_returns = df['close'].pct_change()
    weekly_returns = df['close'].pct_change(periods=5)
    monthly_returns = df['close'].pct_change(periods=20)
    weighted_sum_of_returns = 0.4 * daily_returns + 0.3 * weekly_returns + 0.3 * monthly_returns
    
    # Evaluate Overnight Sentiment
    gap_indicator = df['open'] - df['close'].shift(1)
    
    # Analyze Volume-Based Factors
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv_adjusted_momentum = obv * (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    combined_indicator = (
        0.2 * volume_adjusted_breakout_strength + 
        0.2 * sma_difference + 
        0.2 * weighted_sum_of_returns + 
        0.2 * gap_indicator + 
        0.2 * obv_adjusted_momentum
    )
    
    # Filter for Strong Moves
    threshold = combined_indicator.quantile(0.75)
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
```
