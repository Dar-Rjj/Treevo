import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    breakout_strength = (daily_range * df['volume']) + open_to_close
    
    # Calculate Spread Momentum
    spread_momentum = (df['high'] - df['low']).rolling(window=5).sum() - (df['high'].shift(1) - df['low'].shift(1)).rolling(window=5).sum()
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Volume-Adjusted Momentum
    avg_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * avg_volume
    
    # Incorporate Relative Strength
    relative_strength = df['close'] / df['close'].rolling(window=20).mean()
    
    # Combine Indicators
    combined_indicator = (breakout_strength * 0.5) + spread_momentum + volume_adjusted_momentum
    combined_indicator *= relative_strength
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.rolling(window=30).mean() + 2 * combined_indicator.rolling(window=30).std()
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)
    
    # Calculate Momentum Indicators
    sma_10 = df['close'].rolling(window=10).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    pct_change_20 = df['close'].pct_change(periods=20)
    
    # Explore Volatility Measures
    log_returns = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x))
    vol_30 = log_returns.rolling(window=30).std()
    high_low_range = df['high'] - df['low']
    high_low_avg_20 = high_low_range.rolling(window=20).mean()
    
    # Investigate Trading Activity
    volume_sma_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / volume_sma_20
    price_range = df['high'] - df['low']
    
    # Final Alpha Factor
    alpha_factor = (strong_moves * 0.4) + (sma_10 / sma_50) + (vol_30 * 0.2) + (high_low_avg_20 * 0.1) + (volume_ratio * 0.1) + (price_range * 0.1)
    
    return alpha_factor
