import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    breakout_strength = (daily_range * df['volume']) + open_close_movement
    
    # Moving Average of High-Low Price Spread (Last 20 Days)
    high_low_spread = daily_range.rolling(window=20).mean()
    
    # Spread Momentum
    spread_momentum = high_low_spread - high_low_spread.shift(1)
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_spread = daily_range * df['volume']
    
    # Volume-Adjusted Momentum
    adjusted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff * df['volume'] + open_low_diff * df['volume']) / (df['volume'] + df['volume'])
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.Series([min(x, y) for x, y in zip(proximity_to_high, proximity_to_low)])
    inverted_proximity = -min_proximity
    std_dev_20_days = df['close'].pct_change().rolling(window=20).std()
    enhanced_reversion = inverted_proximity / std_dev_20_days
    
    # Moving Sum of High-Low Price Spreads
    moving_sum_spreads = daily_range.rolling(window=20).sum()
    
    # Combined Indicators
    combined_indicators = (breakout_strength * 0.5) + spread_momentum + adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicators * enhanced_reversion * weighted_volatility
    
    return final_alpha_factor
