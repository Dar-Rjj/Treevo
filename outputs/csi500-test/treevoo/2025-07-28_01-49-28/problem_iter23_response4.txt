```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    high_10d = df['high'].rolling(window=10).max()
    low_10d = df['low'].rolling(window=10).min()
    
    proximity_high = (df['close'] - high_10d) / (high_10d - low_10d)
    proximity_low = (df['close'] - low_10d) / (high_10d - low_10d)
    
    min_proximity = pd.Series(0.0, index=df.index)
    min_proximity[proximity_high > 0] = proximity_high[proximity_high > 0]
    min_proximity[proximity_low < 0] = proximity_low[proximity_low < 0]
    
    inverted_proximity = 1 / (1 + min_proximity.abs())
    
    std_20d = df['close'].pct_change().rolling(window=20).std()
    volatility_adjusted_proximity = inverted_proximity / std_20d
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff + open_low_diff) * df['volume'] / 2
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = volume_adjusted_daily_range + volume_adjusted_open_close
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Volume-Adjusted Momentum
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_daily_range *
        volatility_adjusted_proximity *
        weighted_avg_volatility *
        spread_momentum
    ) + volume_adjusted_breakout_strength + volume_adjusted_momentum
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.quantile(0.75)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
