```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * np.sqrt(df['volume'])) + open_to_close_movement
    
    # High-Low Spread Momentum
    high_low_spread_20d = (df['high'] - df['low']).rolling(window=20).sum()
    spread_momentum = high_low_spread_20d - high_low_spread_20d.shift(1)
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'].diff(1)
    avg_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    adjusted_proximity = min_proximity / twenty_day_volatility
    
    # Integrated Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    max_volatility = pd.concat([open_high_diff, open_low_diff], axis=1).max(axis=1)
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = max_volatility * np.abs(volume_percentage_change)
    
    # Enhanced Momentum Factor
    n = 5
    simple_momentum_n = df['close'] - df['close'].shift(n)
    smoothed_volume = df['volume'].ewm(span=n, adjust=False).mean()
    adjusted_momentum = simple_momentum_n * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_to_low_range = df['high'] - df['low']
    high_low_return = high_to_low_range / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_amounts = df['amount'].rolling(window=n).sum()
    current_ratio = sum_amounts / df['close']
    previous_ratio = df['amount'].rolling(window=n*2).sum().shift(n) / df['close'].shift(n)
    differential = (current_ratio - previous_ratio) / previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength +
        spread_momentum +
        volume_adjusted_momentum +
        combined_factors +
        adjusted_volatility +
        differential
    ) * price_change_ratio
    
    # Filter for Strong Moves
    threshold = integrated_alpha_factor.quantile(0.75)
    strong_moves = (integrated_alpha_factor > threshold).astype(int)
    
    # Price Trend Indicator
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_trend_score = (long_term_momentum - short_term_momentum) * integrated_alpha_factor
    
    # Price Oscillation
    short_term_high = df['high'].rolling(window=10).max()
    short_term_low = df['low'].rolling(window=10).min()
    price_oscillation = (short_term_high - short_term_low) / df['close'].shift(1)
    
    # Enhanced Alpha Factor
    enhanced_alpha_factor = (
        0.7 * integrated_alpha_factor +
        0.3 * (price_trend_score + price_oscillation)
    )
    
    return enhanced_alpha_factor
```
