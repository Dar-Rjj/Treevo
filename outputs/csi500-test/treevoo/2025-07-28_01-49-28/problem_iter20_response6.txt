```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    vol_adjusted_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = pd.DataFrame({'proximity_to_high': proximity_to_high, 'proximity_to_low': proximity_to_low}).min(axis=1)
    inverted_proximity = 1 / min_proximity
    
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / twenty_day_std
    
    # Integrate Open-High-Low Volatility
    open_high_diff = (df['high'] - df['open']) * df['volume']
    open_low_diff = (df['open'] - df['low']) * df['volume']
    open_high_low_volatility = (open_high_diff + open_low_diff) / 2
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spreads = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spreads.diff()
    
    # Volume-Adjusted Open-to-Close Movement
    vol_adj_open_close_movement = (df['close'] - df['open']) * df['volume']
    
    # Volume-Adjusted Breakout Strength
    vol_adj_breakout_strength = (daily_range * df['volume']) + vol_adj_open_close_movement
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()
    
    # Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=2).mean()
    vol_adj_momentum = simple_momentum * avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = (vol_adjusted_daily_range * enhanced_price_reversion * open_high_low_volatility * spread_momentum 
                          + vol_adj_breakout_strength + vol_adj_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + final_alpha_factor.rolling(window=20).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
