```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Adjusted Daily Range
    df['adj_daily_range'] = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Price Reversion
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['proximity_high'] = df['close'] / df['10_day_high']
    df['proximity_low'] = df['close'] / df['10_day_low']
    df['min_proximity'] = df[['proximity_high', 'proximity_low']].min(axis=1)
    df['inverted_proximity'] = 1 / df['min_proximity']
    df['20_day_std'] = df['close'].pct_change().rolling(window=20).std()
    df['price_reversion'] = df['inverted_proximity'] / df['20_day_std']
    
    # Open-High-Low Volatility
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    df['weighted_avg_diff'] = (df['open_high_diff'] + df['open_low_diff']) * df['volume']
    df['ohlv_volatility'] = df['weighted_avg_diff'].rolling(window=20).mean()
    
    # Moving Sum of High-Low Spreads
    df['high_low_spread'] = df['high'] - df['low']
    df['moving_sum_spread'] = df['high_low_spread'].rolling(window=20).sum()
    
    # Spread Momentum
    df['spread_momentum'] = df['moving_sum_spread'] - df['moving_sum_spread'].shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    df['vol_adj_open_close'] = (df['close'] - df['open']) * df['volume']
    
    # Simple Momentum
    df['simple_momentum'] = df['close'] - df['close'].shift(1)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adj_daily_range'] * df['price_reversion'] * df['ohlv_volatility'] * 
                                df['spread_momentum'] + df['vol_adj_open_close'] + df['simple_momentum'])
    
    # Filter for Strong Moves
    dynamic_threshold = df['final_alpha_factor'].quantile(0.75)  # Example threshold
    df['final_alpha_factor'] = df['final_alpha_factor'].apply(lambda x: 1 if x > dynamic_threshold else 0)
    
    return df['final_alpha_factor']
```
