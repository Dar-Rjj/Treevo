import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = df['volume'] / df['volume'].mean()
    volume_adjusted_breakout_strength = (daily_range * volume_factor) + open_to_close_movement
    
    # Calculate 3-day Momentum
    three_day_moving_average = df['close'].rolling(window=3).mean()
    momentum = (df['close'] > three_day_moving_average).astype(int) * 2 - 1
    
    # Incorporate Momentum into Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength_with_momentum = volume_adjusted_breakout_strength * momentum
    
    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    moving_sum_high_low_spread = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Combine Indicators
    combined_indicator = (
        0.5 * volume_adjusted_breakout_strength_with_momentum +
        0.3 * spread_momentum +
        0.2 * volume_adjusted_high_low_spread
    )
    
    # Filter for Strong Moves
    strong_move_threshold = combined_indicator.quantile(0.75)
    factor_values = (combined_indicator > strong_move_threshold).astype(int)
    
    return factor_values
