import pandas as pd

def heuristics_v2(df):
    # Compute the 10-day and 50-day simple moving average (SMA) of closing prices
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()

    # Calculate the percentage change in closing prices over a 20-day period
    df['Pct_Change_20'] = df['close'].pct_change(20)

    # Compute the 10-day rate of change (ROC) of closing prices
    df['ROC_10'] = df['close'].pct_change(10)

    # Calculate the standard deviation of log-returns over a 30-day window
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Std_Dev_30'] = df['Log_Return'].rolling(window=30).std()

    # Determine the high-low range (High - Low) for each day and compute its 20-day moving average
    df['HL_Range'] = df['high'] - df['low']
    df['HL_Range_MA_20'] = df['HL_Range'].rolling(window=20).mean()

    # Calculate the 14-day relative strength index (RSI) based on the close prices
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))

    # Compute the 20-day moving average of volume
    df['Vol_MA_20'] = df['volume'].rolling(window=20).mean()

    # Calculate the ratio of today's volume to the 20-day moving average of volume
    df['Vol_Ratio'] = df['volume'] / df['Vol_MA_20']

    # Analyze the 10-day cumulative volume delta (difference between up-volume and down-volume)
    df['Volume_Delta'] = df['volume'] * (df['close'] > df['close'].shift(1)).astype(int) - df['volume'] * (df['close'] < df['close'].shift(1)).astype(int)
    df['Cum_Vol_Delta_10'] = df['Volume_Delta'].rolling(window=10).sum()

    # Analyze the daily price range (High - Low) as a factor
    df['Daily_Price_Range'] = df['high'] - df['low']

    # Compute the relationship between Open and Close prices; (Close - Open) / Open
    df['OC_Relationship'] = (df['close'] - df['open']) / df['open']

    # Calculate the 20-day moving average of the daily price range (High - Low)
    df['Daily_Price_Range_MA_20'] = df['HL_Range'].rolling(window=20).mean()

    # Combine all factors into a single alpha factor
    alpha_factor = (df['SMA_10'] - df['SMA_50']) + df['Pct_Change_20'] + df['ROC_10'] + df['Std_Dev_30'] + df['HL_Range_MA_20'] + df['RSI_14'] + df['Vol_Ratio'] + df['Cum_Vol_Delta_10'] + df['Daily_Price_Range'] + df['OC_Relationship'] + df['Daily_Price_Range_MA_20']

    return alpha_factor
