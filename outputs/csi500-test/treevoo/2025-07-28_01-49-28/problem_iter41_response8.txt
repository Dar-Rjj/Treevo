import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / min_proximity
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    adjusted_proximity = inverted_proximity / std_dev_20
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_diff = (open_high_diff + open_low_diff) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    weighted_avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_spread = spread * weighted_avg_volume
    
    # Enhanced Momentum Factor
    simple_momentum = df['close'] - df['close'].shift(20)
    smoothed_volume = df['volume'].ewm(span=20).mean()
    adjusted_momentum = simple_momentum * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_low_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_amounts = df['amount'].rolling(window=20).sum()
    current_ratio = sum_amounts / df['close']
    previous_ratio = df['amount'].shift(20).rolling(window=20).sum() / df['close'].shift(20)
    differential = (current_ratio - previous_ratio) / previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Volume-Weighted Moving Average Crossover
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwma_crossover = short_term_vwma - long_term_vwma
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_range * adjusted_proximity * weighted_avg_diff * spread_momentum
        + volume_adjusted_open_close
        + volume_weighted_spread
        + adjusted_momentum
        + combined_factors
        + differential
    ) * price_change_ratio + vwma_crossover
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + 2 * final_alpha_factor.rolling(window=20).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
