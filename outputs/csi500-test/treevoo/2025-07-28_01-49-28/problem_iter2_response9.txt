```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Daily Returns
    daily_returns = df['close'].diff()
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Combine Daily Returns and Volume Ratio
    combined_returns = (daily_returns * volume_ratio).rolling(window=lookback_period).sum()
    
    # Finalize VAMHI
    vamhi = volume_adjusted_high_low_spread + combined_returns
    
    # Calculate the Average of VAMHI
    average_vamhi = combined_returns.mean() - combined_returns.rolling(window=lookback_period).mean()
    
    return vamhi - average_vamhi
```
