```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Weighted_Breakout_Strength'] = (df['Daily_Range'] * df['volume']) + df['Open_to_Close_Movement']
    
    # Calculate Moving Standard Deviation of High-Low Price Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Moving_Std_20'] = df['High_Low_Spread'].rolling(window=20).std()
    
    # Calculate Spread Volatility
    df['Spread_Volatility'] = df['Moving_Std_20'] / df['Moving_Std_20'].shift(1)
    
    # Incorporate Volume into High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['Daily_Range'] * df['volume']
    
    # Calculate Volume-Weighted Momentum
    df['Volume_Weighted_Momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Open-High-Low Volatility
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Open_High_Low_Volatility'] = ((df['Open_High_Diff'] + df['Open_Low_Diff']) * df['volume']) / 2
    
    # Incorporate Enhanced Price Reversion with Smoothing
    df['10_Day_High'] = df['high'].rolling(window=10).max()
    df['10_Day_Low'] = df['low'].rolling(window=10).min()
    df['Proximity_High'] = df['10_Day_High'] - df['close']
    df['Proximity_Low'] = df['close'] - df['10_Day_Low']
    df['Min_Proximity'] = df[['Proximity_High', 'Proximity_Low']].min(axis=1)
    df['Inverted_Proximity'] = 1 / df['Min_Proximity']
    df['Exponential_Moving_Average_20'] = df['close'].pct_change().ewm(span=20).mean()
    df['Enhanced_Price_Reversion'] = df['Inverted_Proximity'] / df['Exponential_Moving_Average_20']
    
    # Calculate Moving Sum of High-Low Price Spreads
    df['Moving_Sum_20'] = df['High_Low_Spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['Moving_Sum_20'] - df['Moving_Sum_20'].shift(1)
    
    # Combine Indicators
    df['Combined_Indicator'] = (
        0.5 * df['Volume_Weighted_Breakout_Strength'] + 
        0.3 * df['Spread_Volatility'] + 
        0.2 * df['Volume_Weighted_Momentum']
    )
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Combined_Indicator'] * 
        df['Enhanced_Price_Reversion'] * 
        df['Open_High_Low_Volatility'] * 
        df['Spread_Momentum']
    )
    
    return df['Final_Alpha_Factor']
```
