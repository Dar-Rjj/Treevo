```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']

    # Price Reversion
    highest_high_10 = df['high'].rolling(window=10).max()
    lowest_low_10 = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / highest_high_10
    proximity_to_low = df['close'] / lowest_low_10
    price_reversion = 1 / (proximity_to_high + proximity_to_low) / df['close'].pct_change().rolling(window=20).std()

    # Open-High-Low Volatility
    open_high_low_volatility = (df['high'] - df['open']) * (df['low'] - df['open']) * df['volume']
    open_high_low_volatility = open_high_low_volatility.rolling(window=5).mean()

    # Spread Momentum
    spread_momentum = (df['high'] - df['low']).rolling(window=5).sum() - (df['high'] - df['low']).shift(1).rolling(window=5).sum()

    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']

    # Simple Momentum
    simple_momentum = df['close'].diff()

    # SMA_10 and SMA_30
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()

    # PriceChangeRatio_5 and PriceChangeRatio_20
    price_change_ratio_5 = (df['close'] / df['close'].shift(5)) - 1
    price_change_ratio_20 = (df['close'] / df['close'].shift(20)) - 1

    # DailyReturnStd_10 and DailyReturnStd_30
    daily_return_std_10 = df['close'].pct_change().rolling(window=10).std()
    daily_return_std_30 = df['close'].pct_change().rolling(window=30).std()

    # RangeVolatility_10 and RangeVolatility_30
    range_volatility_10 = (df['high'] - df['low']).rolling(window=10).mean()
    range_volatility_30 = (df['high'] - df['low']).rolling(window=30).mean()

    # VolumeTrend_10 and VolumeTrend_30
    volume_trend_10 = (df['volume'] / df['volume'].shift(10)) - 1
    volume_trend_30 = (df['volume'] / df['volume'].shift(30)) - 1

    # VolumePriceInteraction_10 and VolumePriceInteraction_30
    volume_price_interaction_10 = (df['close'] * df['volume']).rolling(window=10).mean()
    volume_price_interaction_30 = (df['close'] * df['volume']).rolling(window=30).mean()

    # DojiPattern
    doji_pattern = (abs(df['open'] - df['close']) / (df['high'] - df['low'])) < 0.05

    # BullishEngulfingPattern
    bullish_engulfing_pattern = (df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))

    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * price_reversion * open_high_low_volatility * spread_momentum +
                          volume_adjusted_open_close + simple_momentum +
                          sma_10 + sma_30 + price_change_ratio_5 + price_change_ratio_20 +
                          daily_return_std_10 + daily_return_std_30 + range_volatility_10 + range_volatility_30 +
                          volume_trend_10 + volume_trend_30 + volume_price_interaction_10 + volume_price_interaction_30 +
                          doji_pattern.astype(int) + bullish_engulfing_pattern.astype(int))

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=10).median()
    final_alpha_factor_filtered = (final_alpha_factor > dynamic_threshold).astype(int)

    return final_alpha_factor_filtered
```
