```python
def heuristics_v2(df):
    # Calculate the price change from the previous close to the current open
    df['price_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate the average of high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of volume to amount
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the factor as a combination of the above heuristics
    df['factor'] = (df['price_change'] / df['close'].shift(1)) * df['avg_price'] * df['volume_to_amount_ratio']
    
    return df['factor']
```
