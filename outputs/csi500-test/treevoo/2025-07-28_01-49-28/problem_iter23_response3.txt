```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / min_proximity
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    enhanced_reversion = inverted_proximity / volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_diff = (open_high_diff + open_low_diff) * df['volume']
    open_high_low_volatility = weighted_avg_diff.rolling(window=20).mean()
    
    # Calculate Moving Sum of High-Low Price Spreads
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low_spread = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread.diff()
    
    # Volume-Adjusted Open-to-Close Movement
    open_close_movement = (df['close'] - df['open']) * df['volume']
    
    # Volume-Adjusted Breakout Strength
    breakout_strength = volume_adjusted_range + open_close_movement
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()
    
    # Volume-Adjusted Momentum
    avg_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = (volume_adjusted_range * enhanced_reversion * open_high_low_volatility * spread_momentum 
                          + breakout_strength + volume_adjusted_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=10).mean() + 2 * final_alpha_factor.rolling(window=10).std()
    strong_moves = np.where(final_alpha_factor > dynamic_threshold, 1, 0)
    
    return pd.Series(strong_moves, index=df.index)
```
