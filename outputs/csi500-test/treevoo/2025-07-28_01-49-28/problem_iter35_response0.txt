```python
def heuristics_v2(data):
    # Calculate the percentage change in close price
    data['close_change'] = data['close'].pct_change()
    
    # Calculate the difference between high and low as a measure of volatility
    data['hl_diff'] = data['high'] - data['low']
    
    # Calculate the ratio of amount to volume, which can indicate average transaction size
    data['avg_tx_size'] = data['amount'] / data['volume']
    
    # Calculate the factor: (close price change) * (high-low difference) * (average transaction size)
    data['factor'] = data['close_change'] * data['hl_diff'] * data['avg_tx_size']
    
    return data['factor'].dropna()
```
