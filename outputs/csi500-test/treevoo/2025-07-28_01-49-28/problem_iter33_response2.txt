```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_20 = high_low_spread.rolling(window=20).sum()
    spread_momentum = moving_sum_20 - moving_sum_20.shift(1)
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume_3 = df['volume'].rolling(window=3).mean()
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume_3)
    
    # Relative Strength
    avg_close_20 = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] / avg_close_20
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - ten_day_high
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high.abs(), proximity_to_low.abs()], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    std_dev_20 = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / std_dev_20
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # High-Low Return with Spread Momentum
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    high_low_return_with_spread_momentum = high_low_return * spread_momentum
    
    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    n = 5
    amount_sum_n_days = df['amount'].rolling(window=n).sum()
    close_price_rate = amount_sum_n_days / df['close']
    previous_period_rate = (df['amount'].shift(n+1).rolling(window=n).sum()) / df['close'].shift(n+1)
    amount_per_unit_close_rate_change = close_price_rate - previous_period_rate
    
    # Combined Indicators
    combined_indicator = (volume_adjusted_breakout_strength * 0.4) + spread_momentum + volume_adjusted_momentum
    combined_indicator *= relative_strength
    final_alpha_factor = combined_indicator * enhanced_price_reversion * weighted_avg_volatility * high_low_return_with_spread_momentum * amount_per_unit_close_rate_change
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.quantile(0.75)
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)
    
    return final_alpha_factor * strong_moves
```
