import pandas as pd

def heuristics_v2(df):
    # Momentum-based factors
    df['ROC'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    df['PPO'] = (ema_12 - ema_26) / ema_26
    
    # Liquidity-based factors
    df['OBV'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'OBV'] = df['OBV'].shift(1) + df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'OBV'] = df['OBV'].shift(1) - df['volume']
    df['OBV'].fillna(method='ffill', inplace=True)
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    cumulative_volume_price = (df['volume'] * typical_price).cumsum()
    cumulative_volume = df['volume'].cumsum()
    df['VWAP'] = cumulative_volume_price / cumulative_volume

    # Integrated factors
    df['MomentumLiquidityFactor'] = df['ROC'] * df['OBV'] / df['VWAP']
    df['DirectionalMovementFactor'] = (df['close'] - df['open']) * df['volume']

    return df[['MomentumLiquidityFactor', 'DirectionalMovementFactor']]
