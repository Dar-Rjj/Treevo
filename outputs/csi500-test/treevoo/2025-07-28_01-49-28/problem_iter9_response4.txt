```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Simple Momentum for Close Prices
    close_momentum = df['close'] - df['close'].shift(n)
    
    # Adjust Momentum by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_factor = avg_volume / avg_volume.mean()
    adjusted_close_momentum = close_momentum * volume_factor
    
    # Calculate High-to-Low Range and Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combine Close Price Momentum and High-to-Low Return
    combined_momentum = adjusted_close_momentum + high_low_return
    
    # Open-High and Open-Low Price Differences
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    
    # Weighted Average by Volume
    weighted_avg_diff = (open_high_diff + open_low_diff) * df['volume'] / df['volume'].rolling(window=n).sum()
    
    # Adjust Volatility by Volume Trend
    volume_change = df['volume'].pct_change()
    adjusted_volatility = weighted_avg_diff * (1 + volume_change)
    
    # Amount per Unit Close Price Rate of Change
    amount_per_close = df['amount'].rolling(window=n).sum() / df['close']
    amount_per_close_prev = df['amount'].rolling(window=2*n).sum().shift(n) / df['close'].shift(n)
    amount_per_close_roc = amount_per_close - amount_per_close_prev
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + adjusted_volatility + amount_per_close_roc
    
    return alpha_factor
```
