```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Factor'] = (df['Daily_Range'] * df['volume']) + df['Open_to_Close_Movement']
    df['Volume_Adjusted_Breakout_Strength'] = df['Volume_Factor']
    
    # Calculate Daily High-Low Price Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    df['Moving_Sum_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['Moving_Sum_High_Low_Spread'] - df['Moving_Sum_High_Low_Spread'].shift(1)
    
    # Incorporate Volume into High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Calculate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(1)) + df['Volume_Adjusted_High_Low_Spread']
    
    # Combine Indicators
    weight_breakout_strength = 0.5
    df['Combined_Indicator'] = (weight_breakout_strength * df['Volume_Adjusted_Breakout_Strength'] +
                                df['Spread_Momentum'] + 
                                df['Momentum'])
    
    # Filter for Strong Moves
    threshold = 0.01
    df['Alpha_Factor'] = (df['Combined_Indicator'] > threshold).astype(int)
    
    return df['Alpha_Factor']
```
