```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate the open-close difference
    open_close_diff = df['open'] - df['close']
    
    # Calculate the volume-weighted price
    volume_weighted_price = (df['amount'] / df['volume']) * df['close']
    
    # Calculate a combined factor using the differences and volume-weighted price
    alpha_factor = (high_low_diff + open_close_diff) * volume_weighted_price
    
    return alpha_factor
```
