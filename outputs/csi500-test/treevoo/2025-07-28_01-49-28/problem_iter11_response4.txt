```python
def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Calculate Enhanced Momentum
    enhanced_momentum = (df['close'] - df['close'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    enhanced_momentum = enhanced_momentum.fillna(0)  # Handle the first day where there is no previous close
    
    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close_movement + enhanced_momentum
    
    # Define a threshold for strong moves
    threshold = 0.5  # This can be tuned based on backtesting results
    
    # Filter for Strong Moves
    factor = combined_indicator.apply(lambda x: 1 if x > threshold else 0)
    
    return factor
```
