```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Calculate Spread Momentum
    daily_high_low_spread = df['high'] - df['low']
    moving_sum_20_days = daily_high_low_spread.rolling(window=20).sum()
    spread_momentum = moving_sum_20_days.diff()
    
    # Calculate Volume-Adjusted Momentum
    simple_momentum = df['close'].diff()
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume
    
    # Combine Indicators
    combined_indicator = (
        volume_adjusted_breakout_strength * 0.5 +
        spread_momentum +
        volume_adjusted_momentum
    )
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.rolling(window=10).quantile(0.9)
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)
    
    return strong_moves
```
