```python
import pandas as pd

def heuristics_v2(df):
    n = 10  # Define the lookback period
    
    # Calculate Simple Trend for Close Prices
    simple_trend_close = df['close'].pct_change(periods=n)
    
    # Compute Variance in Volume (t to t-n)
    volume_variance = df['volume'].rolling(window=n).var()
    
    # Adjust Trend by Volume Variance
    adjusted_trend = simple_trend_close * (volume_variance + 1)  # Add 1 to avoid multiplication by zero
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Integrate Open Price Movement
    open_avg_high_low = abs(df['open'] - (df['high'] + df['low']) / 2)
    
    # Weighted Summation Based on Volume
    recent_volume_change = df['volume'].pct_change().fillna(0)
    weighted_sum = (high_low_diff + open_avg_high_low) * (recent_volume_change + 1)
    
    # Amount per Unit Change in Close Price
    abs_close_change = abs(df['close'].diff()).rolling(window=n).sum()
    sum_amount = df['amount'].rolling(window=n).sum()
    efficiency_ratio = sum_amount / abs_close_change
    
    # Measure the Directional Change in this Efficiency Ratio
    prev_efficiency_ratio = (df['amount'].shift(n).rolling(window=n).sum() 
                             / abs(df['close'].shift(n).diff()).rolling(window=n).sum())
    directional_change = (efficiency_ratio - prev_efficiency_ratio) / prev_efficiency_ratio * 100
    
    # Combine all alpha factors
    alpha_factor = (adjusted_trend + weighted_sum + directional_change) / 3
    
    return alpha_factor
```
