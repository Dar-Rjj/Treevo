```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Simple Momentum for Close Prices
    df['SimpleMomentum'] = df['close'].pct_change(periods=n)
    
    # Adjust Momentum by Average Volume
    df['AverageVolume'] = df['volume'].rolling(window=n).mean()
    df['VolumeFactor'] = df['volume'] / df['AverageVolume']
    df['VolumeAdjustedMomentum'] = df['SimpleMomentum'] * df['VolumeFactor']
    
    # Open-High Price Difference
    df['OpenHighDiff'] = df['high'] - df['open']
    
    # Open-Low Price Difference
    df['OpenLowDiff'] = df['low'] - df['open']
    
    # Weighted Average by Volume
    df['WeightedAvg'] = (df['OpenHighDiff'] + df['OpenLowDiff']) / 2 * df['volume']
    df['OHVolatilityFactor'] = df['WeightedAvg'].rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Amount per Unit Close Price Rate of Change
    df['AmountPerClose'] = df['amount'].rolling(window=n).sum() / df['close']
    df['PrevAmountPerClose'] = df['amount'].shift(n).rolling(window=n).sum() / df['close'].shift(n)
    df['AmountRateChange'] = (df['AmountPerClose'] - df['PrevAmountPerClose']) / df['PrevAmountPerClose']
    
    # Combined Alpha Factor
    alpha_factor = (
        df['VolumeAdjustedMomentum'] +
        df['OHVolatilityFactor'] +
        df['AmountRateChange']
    )
    
    return alpha_factor.dropna()
```
