```python
import pandas as pd

def heuristics_v2(df):
    # Analyze price action to identify trend and momentum factors
    df['raw_volatility'] = df['high'] - df['low']
    df['price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['simple_momentum_close'] = df['close'].diff()
    df['long_term_trend'] = df['close'].rolling(window=50).mean() - df['close'].rolling(window=200).mean()

    # Examine volume changes to infer market sentiment
    df['volume_change'] = df['volume'].diff()
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    df['volume_adjusted_breakout'] = (df['high'] - df['low']) * df['volume'] + (df['close'] - df['open'])
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']

    # Combine price and volume information to create more complex factors
    df['price_volume_composite'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_composite_5d'] = df['price_volume_composite'].rolling(window=5).sum()
    df['gap_factor'] = df['open'] - df['close'].shift(1)
    df['spread_momentum'] = df['high'] - df['low']
    df['spread_momentum_20d'] = df['spread_momentum'].rolling(window=20).mean()
    df['spread_momentum_diff'] = df['spread_momentum_20d'] - df['spread_momentum_20d'].shift(1)

    # Incorporate the concept of money flow by analyzing the amount traded
    df['positive_money_flow'] = ((df['close'] > df['open']) * (df['close'] - df['open'])) * df['volume']
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']

    # Analyze the relationship between price and amount
    df['transaction_size'] = df['amount'] / df['volume']
    df['price_amount_interaction'] = (df['close'] - df['open']) * (df['amount'] / df['volume'])

    # Investigate the momentum of price and volume together
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['momentum_factor'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume'])

    # Analyze the relationship between price, volume, and amount
    df['price_volume_amount_composite'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume']) * (df['amount'] / df['volume'])

    # Enhanced Price Reversion
    df['highest_high_10d'] = df['high'].rolling(window=10).max()
    df['lowest_low_10d'] = df['low'].rolling(window=10).min()
    df['proximity'] = df[['highest_high_10d', 'lowest_low_10d']].apply(lambda x: min(x), axis=1)
    df['inverted_proximity'] = 1 / df['proximity']
    df['std_daily_returns'] = df['close'].pct_change().rolling(window=20).std()
    df['enhanced_price_reversion'] = df['inverted_proximity'] / df['std_daily_returns']

    # Combined Indicators
    df['combined_indicator'] = (df['volume_adjusted_breakout'] + df['spread_momentum_diff'] + df['volume_adjusted_momentum']) / 3
    df['relative_strength'] = df['close'] / df['close'].rolling(window=20).mean()
    df['final_alpha_factor'] = df['combined_indicator'] * df['enhanced_price_reversion'] * df['raw_volatility'] * df['long_term_trend'] * df['relative_strength']

    # Filter for Strong Moves
    df['strong_move_indicator'] = (df['combined_indicator'] > df['combined_indicator'].quantile(0.75)).astype(int)

    return df['final_alpha_factor']
```
