```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = (df['high'] - df['low']) * df['volume']

    # Incorporate Dynamic Price Reversion
    five_day_high = df['high'].rolling(window=5).max()
    five_day_low = df['low'].rolling(window=5).min()
    proximity_to_high = df['close'] / five_day_high
    proximity_to_low = df['close'] / five_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    ten_day_volatility = df['close'].rolling(window=10).std()
    dynamic_reversion = inverted_proximity / ten_day_volatility

    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_open_high = open_high_diff * df['volume']
    weighted_open_low = open_low_diff * df['volume']
    open_high_low_volatility = (weighted_open_high + weighted_open_low) / (2 * df['volume'])

    # Moving Sum of 10-Day High-Low Spreads
    ten_day_spread_sum = (df['high'] - df['low']).rolling(window=10).sum()

    # Spread Momentum
    spread_momentum = ten_day_spread_sum.diff()

    # Trend Component
    fifty_day_sma = df['close'].rolling(window=50).mean()
    trend_component = df['close'] - fifty_day_sma

    # Final Alpha Factor
    alpha_factor = (
        volume_adjusted_range *
        dynamic_reversion *
        open_high_low_volatility *
        spread_momentum *
        trend_component
    )

    return alpha_factor
```
