```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Daily Range
    df['Volume_Adjusted_Daily_Range'] = (df['high'] - df['low']) * df['volume']
    
    # 15-Day High and Low
    df['15Day_High'] = df['high'].rolling(window=15).max()
    df['15Day_Low'] = df['low'].rolling(window=15).min()
    
    # Proximity to 15-Day Range
    df['Proximity_to_15Day_Range'] = df[['15Day_High', '15Day_Low']].apply(lambda x: min(x[0] - df['close'], df['close'] - x[1]), axis=1)
    
    # Adjust for 20-Day Volatility
    df['20Day_Volatility'] = df['close'].rolling(window=20).std()
    df['Enhanced_Reversion'] = 1 / df['Proximity_to_15Day_Range'] / df['20Day_Volatility']
    
    # Open-High-Low Volatility with Volume Weighting
    df['Weighted_Open_High_Difference'] = (df['high'] - df['open']) * df['volume']
    df['Weighted_Open_Low_Difference'] = (df['open'] - df['low']) * df['volume']
    df['Combined_Weighted_Volatility'] = (df['Weighted_Open_High_Difference'] + df['Weighted_Open_Low_Difference']) / 2
    
    # Spread Momentum with Volume Weighting
    df['Volume_Weighted_Spread'] = (df['high'] - df['low']) * df['volume']
    df['30Day_Spread_Momentum'] = df['Volume_Weighted_Spread'].rolling(window=30).sum()
    df['Spread_Momentum'] = df['30Day_Spread_Momentum'].diff()
    
    # Trend Strength with Short-Term and Long-Term Averages
    df['60Day_MA'] = df['close'].rolling(window=60).mean()
    df['20Day_MA'] = df['close'].rolling(window=20).mean()
    df['Trend_Strength'] = (df['60Day_MA'] - df['20Day_MA']) / df['20Day_MA']
    
    # Momentum Factor with Dual Time Frames
    df['10Day_Return'] = (df['close'] / df['close'].shift(10)) - 1
    df['50Day_Return'] = (df['close'] / df['close'].shift(50)) - 1
    df['Combined_Momentum'] = df['10Day_Return'] + df['50Day_Return']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Adjusted_Daily_Range'] * 
                          df['Enhanced_Reversion'] * 
                          df['Combined_Weighted_Volatility'] * 
                          df['Spread_Momentum'] * 
                          df['Trend_Strength'] * 
                          df['Combined_Momentum'])
    
    return df['Alpha_Factor']
```
