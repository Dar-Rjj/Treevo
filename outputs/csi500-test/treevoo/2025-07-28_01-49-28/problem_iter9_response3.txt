def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Calculate Volume Volatility
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_volatility = volume_change ** 2
    
    # Combine Factors
    combined_factor = (volume_adjusted_high_low_spread + volume_weighted_momentum) * price_change_ratio * volume_volatility
    
    return combined_factor
