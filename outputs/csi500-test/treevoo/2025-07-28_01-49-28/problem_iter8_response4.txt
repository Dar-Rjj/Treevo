```python
def heuristics_v2(df):
    # Volume-Adjusted Close Price Momentum
    momentum_21 = df['close'].pct_change(21)
    momentum_63 = df['close'].pct_change(63)
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_adjusted_momentum = (momentum_21 + momentum_63) * (avg_volume_21 + avg_volume_63) / 2

    # Combined Price Momentum
    price_momentum_21 = df['close'] - df['close'].shift(21)
    price_momentum_63 = df['close'] - df['close'].shift(63)
    combined_price_momentum = 0.5 * price_momentum_21 + 0.5 * price_momentum_63

    # Volume Reversal Indicator
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_difference = avg_volume_21 - avg_volume_63
    volume_reversal = volume_difference / avg_volume_63

    # Open-High-Low Volatility Factor
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['low'] - df['open']
    weighted_open_high = open_high_diff * df['volume']
    weighted_open_low = open_low_diff * df['volume']
    open_high_low_volatility = (weighted_open_high + weighted_open_low) / (df['volume'] * 2)

    # Price-Amount Interaction Effect
    amount_per_close = df['amount'].rolling(window=20).sum() / df['close']
    previous_amount_per_close = df['amount'].rolling(window=40).sum().shift(20) / df['close'].shift(20)
    price_amount_interaction = amount_per_close - previous_amount_per_close

    # Daily Return
    daily_return = df['close'] - df['close'].shift(1)

    # Smoothed Momentum
    smoothed_momentum = daily_return.rolling(window=20).sum() / df['volume'].rolling(window=20).mean()

    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_high = df['close'] - ten_day_high
    proximity_low = df['close'] - ten_day_low
    proximity = proximity_high.where(proximity_high < proximity_low, proximity_low)
    inverted_proximity = 1 / (proximity.abs() + 1e-6)
    std_dev_20 = daily_return.rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / std_dev_20

    # Volume-Adjusted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_adjusted_spread = high_low_spread * df['volume']

    # Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_spread = volume_adjusted_spread.rolling(window=20).sum()

    # Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum_spread - moving_sum_spread.shift(1)

    # Final Alpha Factor
    alpha_factor = (
        volume_adjusted_momentum +
        combined_price_momentum +
        volume_reversal +
        open_high_low_volatility +
        price_amount_interaction +
        smoothed_momentum +
        enhanced_price_reversion +
        adjusted_spread_momentum
    )

    return alpha_factor
```
