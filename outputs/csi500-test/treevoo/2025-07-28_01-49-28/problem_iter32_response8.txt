```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range + open_to_close_movement) * df['volume']

    # High-Low Spread Momentum
    high_low_spread_momentum = (df['high'] - df['low']).rolling(window=5).sum() - (df['high'] - df['low']).shift(1).rolling(window=5).sum()

    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume

    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_10day_high = df['close'] - ten_day_high
    proximity_to_10day_low = df['close'] - ten_day_low
    min_proximity = proximity_to_10day_high.abs().combine(proximity_to_10day_low.abs(), min)
    inverted_min_proximity = 1 / min_proximity
    twenty_day_volatility = df['close'].rolling(window=20).std()
    enhanced_proximity = inverted_min_proximity / twenty_day_volatility

    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume'] / (df['high'] - df['low'])

    # Enhanced Reversion Factor
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    deviation_from_short_term_ma = df['close'] - short_term_ma
    deviation_from_long_term_ma = df['close'] - long_term_ma
    combined_reversion_score = deviation_from_short_term_ma * 0.6 + deviation_from_long_term_ma * 0.4
    volume_adjusted_reversion_score = combined_reversion_score * df['volume']
    reversion_ema = volume_adjusted_reversion_score.ewm(span=10, adjust=False).mean()
    final_reversion_indicator = volume_adjusted_reversion_score / reversion_ema

    # Integrated Alpha Factor
    alpha_factor = (
        volume_adjusted_breakout_strength
        + high_low_spread_momentum
        + volume_adjusted_momentum
        + enhanced_proximity
        + weighted_open_high_low_volatility
        + final_reversion_indicator
    )

    return alpha_factor
```
