```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Volume Weighted Overnight Return
    volume_weighted_overnight_return = overnight_return * df['volume'].shift(1)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Momentum Score
    momentum_score = volume_weighted_intraday_return / intraday_volatility
    
    # Reversal Score
    reversal_score = volume_weighted_overnight_return / intraday_range
    
    # Combine Momentum and Reversal Scores into a single factor
    alpha_factor = momentum_score - reversal_score
    
    return alpha_factor
```
