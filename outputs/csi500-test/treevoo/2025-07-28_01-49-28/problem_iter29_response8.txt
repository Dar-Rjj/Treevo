```python
import pandas as pd

def heuristics_v2(df):
    """
    Given a DataFrame with (open, high, low, close, amount, volume) columns,
    this function computes an alpha factor based on the thought tree.
    The output is a pandas Series indexed by (date).
    
    :param df: A pandas DataFrame with (open, high, low, close, amount, volume) as columns.
    :return: A pandas Series representing the factor values.
    """
    # Thought Tree:
    # 1. Calculate the difference between high and low prices as price range
    # 2. Calculate the ratio of the close price to the open price as a momentum indicator
    # 3. Calculate the ratio of the amount to the volume to get a weighted average price
    # 4. Combine the above three factors into a single alpha factor
    
    price_range = df['high'] - df['low']
    momentum = df['close'] / df['open']
    weighted_price = df['amount'] / df['volume']
    
    # Combine the factors
    alpha_factor = price_range * momentum * weighted_price
    
    return alpha_factor
```
