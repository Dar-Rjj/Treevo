```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Enhanced Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = df['close'] / ten_day_high
    proximity_to_ten_day_low = df['close'] / ten_day_low
    inverted_proximity = 1 / pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    enhanced_reversion = inverted_proximity / twenty_day_volatility
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_by_volume = (open_high_diff + open_low_diff) / 2 * df['volume']
    
    # Spread Momentum
    thirty_day_moving_sum_spread = (df['high'] - df['low']).rolling(window=30).sum()
    spread_momentum = thirty_day_moving_sum_spread.diff()
    
    # Trend Strength
    fifty_day_ma = df['close'].rolling(window=50).mean()
    twenty_day_ma = df['close'].rolling(window=20).mean()
    trend_strength = (fifty_day_ma - twenty_day_ma) / twenty_day_ma
    
    # Combined Momentum
    ten_day_return = (df['close'] / df['close'].shift(10)) - 1
    thirty_day_return = (df['close'] / df['close'].shift(30)) - 1
    combined_momentum = ten_day_return + thirty_day_return
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'].diff()
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Volume-Weighted Price Change
    volume_weighted_price_change = (df['close'].diff()) * (df['volume'] / avg_volume)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * enhanced_reversion * weighted_avg_by_volume * 
                          spread_momentum * trend_strength * combined_momentum * 
                          volume_adjusted_momentum * volume_weighted_price_change)
    
    return final_alpha_factor
```
