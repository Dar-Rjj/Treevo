```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['open_close_movement'] = df['close'] - df['open']
    
    # Adjust for Volume: Volume-Adjusted Breakout Strength
    df['volume_adjusted_breakout_strength'] = df['daily_range'] * df['volume'] + df['open_close_movement']
    
    # Calculate Daily High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    df['moving_sum_high_low_spread'] = df['high_low_spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['spread_momentum'] = df['moving_sum_high_low_spread'] - df['moving_sum_high_low_spread'].shift(1)
    
    # Incorporate Volume into High-Low Spread
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Calculate Simple Momentum for Close Prices
    df['simple_momentum'] = df['close'] - df['close'].shift(1)
    
    # Adjust Momentum by Average Volume
    df['average_volume'] = (df['volume'] + df['volume'].shift(1)) / 2
    df['volume_adjusted_momentum'] = df['simple_momentum'] * df['average_volume']
    
    # Combine Indicators: Weighted Sum
    combined_indicator = (
        0.5 * df['volume_adjusted_breakout_strength'] +
        0.3 * df['spread_momentum'] +
        0.2 * df['volume_adjusted_momentum']
    )
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.rolling(window=20).mean() + combined_indicator.rolling(window=20).std()
    alpha_factor = (combined_indicator > dynamic_threshold).astype(int)
    
    return alpha_factor
```
