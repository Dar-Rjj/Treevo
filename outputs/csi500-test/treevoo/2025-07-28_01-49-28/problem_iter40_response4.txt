```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Price Reversion
    highest_high_10d = df['high'].rolling(window=10).max()
    lowest_low_10d = df['low'].rolling(window=10).min()
    proximity_to_10d_range = 1 - ((df['close'] - lowest_low_10d) / (highest_high_10d - lowest_low_10d))
    std_20d_returns = df['close'].pct_change().rolling(window=20).std()
    price_reversion = proximity_to_10d_range / std_20d_returns
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff * df['volume'] + open_low_diff * df['volume']) / (2 * df['volume'])
    
    # Spread Momentum
    spread_momentum = (df['high'] - df['low']).diff()
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Simple Momentum
    simple_momentum = df['close'].diff()
    
    # Final Alpha Factor
    final_alpha_factor = (
        adjusted_daily_range * 
        price_reversion * 
        weighted_avg * 
        spread_momentum + 
        volume_adjusted_open_close + 
        simple_momentum
    )
    final_alpha_factor = (final_alpha_factor > 0).astype(int)
    
    return final_alpha_factor
```
