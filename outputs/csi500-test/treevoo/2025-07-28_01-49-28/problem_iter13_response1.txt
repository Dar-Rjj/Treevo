```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_daily_range = daily_range * df['volume']
    
    # Combine Indicators
    combined_indicator = open_close_movement + volume_adjusted_daily_range
    
    # Filter for Strong Moves
    threshold = combined_indicator.quantile(0.75)  # Example threshold, 75th percentile
    strong_moves = (combined_indicator > threshold).astype(int)
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_spread - moving_sum_spread.shift(1)
    
    # Final Alpha Factor
    alpha_factor = strong_moves * spread_momentum
    
    return alpha_factor
```
