```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Volume-Adjusted Momentum with Close-to-Open Movement
    simple_momentum = df['close'] - df['close'].shift(n)
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_momentum = simple_momentum * (avg_volume / df['volume'])
    close_to_open_movement = df['open'] - df['close'].shift(1)
    
    combined_indicator = (adjusted_momentum + close_to_open_movement) * (df['volume'] / avg_volume)
    strong_move_indicator = (combined_indicator > 0).astype(int)
    
    # High-Low Return with Spread Momentum
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    daily_high_low_spread_sum = high_low_range.rolling(window=20).sum()
    spread_momentum = daily_high_low_spread_sum - daily_high_low_spread_sum.shift(1)
    
    # Open-High-Low Volatility with Volume-Weighted Adjustments
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff + open_low_diff) * df['volume'] / (2 * avg_volume)
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = weighted_avg * (1 + volume_change)
    
    volume_weighted_close = df['close'] * df['volume']
    moving_avg_volume_weighted_close = volume_weighted_close.rolling(window=n).mean()
    volume_weighted_momentum = volume_weighted_close - moving_avg_volume_weighted_close
    
    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    amount_per_close = df['amount'].rolling(window=n).sum() / df['close']
    prev_amount_per_close = df['amount'].rolling(window=n*2).sum().shift(n) / df['close'].shift(n)
    amount_per_close_rate_of_change = amount_per_close - prev_amount_per_close
    
    # Combined Alpha Factor
    alpha_factor = (strong_move_indicator + high_low_return + spread_momentum + 
                    adjusted_volatility + volume_weighted_momentum + amount_per_close_rate_of_change)
    
    return alpha_factor
```
