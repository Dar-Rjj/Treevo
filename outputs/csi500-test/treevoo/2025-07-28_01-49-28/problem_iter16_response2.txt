```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * np.sqrt(df['volume'])
    
    # Calculate 5-day Momentum
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    momentum_score = (df['close'] > df['5_day_ma']).astype(int) * 2 - 1
    
    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close_movement
    combined_indicator_with_momentum = combined_indicator * momentum_score
    
    # Filter for Strong Moves
    threshold = combined_indicator_with_momentum.mean() + 2 * combined_indicator_with_momentum.std()
    strong_moves = (combined_indicator_with_momentum > threshold).astype(int)
    
    return strong_moves
```
