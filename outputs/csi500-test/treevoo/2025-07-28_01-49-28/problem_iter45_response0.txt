```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Day-over-Day Returns
    day_over_day_returns = df['close'].diff()
    
    # Identify Reversal Patterns
    bearish_engulfing = (
        (df['high'] > df['high'].shift(1)) & 
        (df['low'] < df['low'].shift(1)) & 
        (df['close'] < df['open']) & 
        (df['close'].shift(1) > df['open'].shift(1))
    )
    
    bullish_engulfing = (
        (df['low'] < df['low'].shift(1)) & 
        (df['high'] > df['high'].shift(1)) & 
        (df['close'] > df['open']) & 
        (df['close'].shift(1) < df['open'].shift(1))
    )
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Detect Volume Spikes
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_spike = (df['volume'] > 2 * avg_volume)
    
    # Calculate Simple Moving Average (SMA)
    sma = df['close'].rolling(window=10).mean()
    
    # Calculate Exponential Moving Average (EMA)
    ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate Rate of Change (ROC)
    roc = ((df['close'] - df['close'].shift(10)) / df['close'].shift(10)) * 100
    
    # Calculate Accumulation/Distribution Line (ADL)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (money_flow_multiplier * df['volume']).cumsum()
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate Chaikin Money Flow (CMF)
    cmf_numerator = (money_flow_multiplier * df['volume']).rolling(window=10).sum()
    cmf_denominator = df['volume'].rolling(window=10).sum()
    cmf = cmf_numerator / cmf_denominator
    
    # Combine all factors into a single DataFrame
    factors = pd.DataFrame({
        'High-Low Range': high_low_range,
        'Open-Close Spread': open_close_spread,
        'Day-over-Day Returns': day_over_day_returns,
        'Bearish Engulfing': bearish_engulfing,
        'Bullish Engulfing': bullish_engulfing,
        'VWAP': vwap,
        'Volume Change': volume_change,
        'Volume Spike': volume_spike,
        'SMA': sma,
        'EMA': ema,
        'Momentum': momentum,
        'ROC': roc,
        'ADL': adl,
        'OBV': obv,
        'CMF': cmf
    })
    
    # Generate the final alpha factor
    alpha_factor = (factors['High-Low Range'] + factors['Open-Close Spread'] + factors['Day-over-Day Returns'] + 
                    factors['Bearish Engulfing'] - factors['Bullish Engulfing'] + 
                    factors['VWAP'] + factors['Volume Change'] + factors['Volume Spike'] + 
                    factors['SMA'] + factors['EMA'] + factors['Momentum'] + factors['ROC'] + 
                    factors['ADL'] + factors['OBV'] + factors['CMF'])
    
    return alpha_factor
```
