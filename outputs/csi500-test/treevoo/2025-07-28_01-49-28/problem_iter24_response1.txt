```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices as a measure of volatility
    df['volatility'] = df['high'] - df['low']
    
    # Calculate (close - open) / (high - low) as a raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Compute today's volume - yesterday's volume as a raw volume change factor
    df['volume_change'] = df['volume'].diff()
    
    # Calculate (volume * (close - open)) as a raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Summarize the (close - open) * volume over the last 5 days as a raw price-volume composite factor
    df['price_volume_composite'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_composite_5d'] = df['price_volume_composite'].rolling(window=5).sum()
    
    # Calculate the (open - previous close) as a raw gap factor
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate the positive money flow
    df['positive_money_flow'] = (df['close'] - df['open']).apply(lambda x: max(x, 0)) * df['volume']
    
    # Calculate the total money flow
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Use the ratio of positive to total money flow as a raw money flow index
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']
    
    # Calculate the (amount / volume) as a raw transaction size factor
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Calculate the (close - open) * (volume / average volume) as a raw momentum factor
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['momentum_factor'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume'])
    
    # Create the final alpha factor by combining the above factors
    df['alpha_factor'] = (
        df['volatility'] + 
        df['price_movement'] + 
        df['volume_change'] + 
        df['volume_price_interaction'] + 
        df['price_volume_composite_5d'] + 
        df['price_gap'] + 
        df['money_flow_index'] + 
        df['transaction_size'] + 
        df['momentum_factor']
    )
    
    return df['alpha_factor']
```
