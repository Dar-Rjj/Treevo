```python
def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = df['volume']
    
    # Calculate Momentum
    momentum = df['close'].pct_change()
    
    # Combine Indicators
    combined_indicator = (daily_range * volume_factor + 
                          open_to_close_movement + 
                          momentum)
    
    # Filter for Strong Moves
    threshold = 0.5  # This is a heuristic threshold, can be adjusted
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
```
