```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # High-Low Spread Momentum
    current_spread_sum = df['high'] - df['low']
    prev_spread_sum = df['high'].shift(1) - df['low'].shift(1)
    high_low_momentum = current_spread_sum - prev_spread_sum
    
    # Volume-Weighted Close-Open Spread
    close_open_spread = (df['close'] - df['open']) * df['volume']
    
    # Combined Volume-Adjusted Spreads
    combined_spreads = (df['high'] - df['low']) * df['volume'] + close_open_spread
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Smoothed Momentum
    daily_returns = df['close'].pct_change()
    smoothed_momentum = daily_returns.rolling(window=10).sum() / df['volume']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_high = df['close'] - ten_day_high
    proximity_low = df['close'] - ten_day_low
    min_proximity = np.minimum(np.abs(proximity_high), np.abs(proximity_low))
    min_proximity_inverted = 1 / (1 + min_proximity)
    ten_day_volatility = daily_returns.rolling(window=10).std()
    enhanced_proximity = min_proximity_inverted / ten_day_volatility
    
    # Enhanced Volatility Adjustment
    ten_day_ema_returns = daily_returns.ewm(span=10, adjust=False).mean()
    enhanced_vol_adj_proximity = enhanced_proximity / ten_day_ema_returns
    
    # Amount per Unit Close Price Rate of Change
    amount_rate_of_change = (df['amount'].rolling(window=10).sum() / df['close']) - (df['amount'].shift(10).rolling(window=10).sum() / df['close'].shift(10))
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        breakout_strength +
        combined_spreads +
        volume_weighted_momentum +
        smoothed_momentum +
        enhanced_vol_adj_proximity +
        amount_rate_of_change
    ) * price_change_ratio
    
    # Filter for Strong Moves
    strong_moves_threshold = 1.5
    strong_moves = (integrated_alpha_factor > strong_moves_threshold).astype(int)
    
    # Trend Following Indicator
    ten_day_sma = df['close'].rolling(window=10).mean()
    ten_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    trend_following_score = (ten_day_ema - ten_day_sma) / ten_day_ema
    
    # Combine Integrated Alpha Factor and Trend Following Score
    enhanced_alpha_factor = integrated_alpha_factor + trend_following_score
    
    return enhanced_alpha_factor
```
