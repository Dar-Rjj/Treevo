```python
import pandas as pd

def heuristics_v2(df, n=5, threshold=0.5):
    # Calculate Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    weighted_average_volume = df['volume'].rolling(window=n).sum() / n
    volume_weighted_average_spread = spread * weighted_average_volume
    
    # Integrated Momentum Factor
    simple_momentum = df['close'] / df['close'].shift(n) - 1
    adjusted_momentum_by_volume = simple_momentum * weighted_average_volume
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = ten_day_high - df['close']
    proximity_to_ten_day_low = df['close'] - ten_day_low
    minimum_proximity = pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    inverted_proximity = 1 / minimum_proximity
    std_dev_daily_returns = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / std_dev_daily_returns
    
    # Integrate Open-High-Low Volatility
    open_high_difference = df['high'] - df['open']
    open_low_difference = df['open'] - df['low']
    open_high_low_volatility = ((open_high_difference + open_low_difference) / 2) * df['volume']
    
    # Combined Momentum and High-Low Return
    high_to_low_range = df['high'] - df['low']
    high_to_low_return = high_to_low_range / df['close'].shift(1)
    combined_momentum_and_high_low_return = adjusted_momentum_by_volume + high_to_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_of_amounts = df['amount'].rolling(window=n).sum()
    amount_per_unit_close_price = sum_of_amounts / df['close']
    previous_ratio = (df['amount'].shift(n).rolling(window=n).sum()) / df['close'].shift(n)
    differential_between_ratios = amount_per_unit_close_price - previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_breakout_strength 
        + volume_weighted_average_spread 
        + adjusted_momentum_by_volume 
        + combined_momentum_and_high_low_return 
        + open_high_low_volatility 
        + differential_between_ratios
    ) * price_change_ratio * enhanced_price_reversion
    
    # Filter for Strong Moves
    strong_moves = (final_alpha_factor > threshold).astype(int)
    
    return final_alpha_factor * strong_moves
```
