```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_close_movement
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low = volume_adjusted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted Close-Open Spread
    moving_avg_close_open = volume_adjusted_close_open_spread.rolling(window=10).mean()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    integrated_spread = moving_avg_close_open - moving_avg_close_open.shift(1)
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Add Volume-Adjusted High-Low Spread to Momentum
    volume_adjusted_high_low_momentum = momentum + volume_adjusted_high_low_spread
    
    # Combine Indicators
    combined_indicator = (0.5 * volume_adjusted_breakout_strength) + adjusted_spread_momentum + volume_adjusted_high_low_momentum
    
    # Filter for Strong Moves
    threshold = combined_indicator.quantile(0.75)
    factor_values = combined_indicator.apply(lambda x: 1 if x > threshold else 0)
    
    return factor_values
```
