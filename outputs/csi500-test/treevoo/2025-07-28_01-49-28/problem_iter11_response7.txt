```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Adjusted_Breakout_Strength'] = (df['Daily_Range'] * df['volume']) + df['Open_to_Close_Movement']
    
    # Calculate Daily High-Low Price Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    df['Moving_Sum_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['Moving_Sum_High_Low_Spread'].diff()
    
    # Incorporate Volume into High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Calculate Simple Momentum for Close Prices
    df['Simple_Momentum'] = df['close'].pct_change(periods=1)
    
    # Adjust Momentum by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Momentum'] = df['Simple_Momentum'] * (df['volume'] / df['Average_Volume'])
    
    # Combine Indicators
    weight_breakout_strength = 0.5
    df['Combined_Indicator'] = (weight_breakout_strength * df['Volume_Adjusted_Breakout_Strength']) + df['Spread_Momentum'] + df['Volume_Adjusted_Momentum']
    
    # Filter for Strong Moves
    threshold = 0.05
    df['Strong_Move'] = df['Combined_Indicator'].apply(lambda x: 1 if x > threshold else 0)
    
    return df['Strong_Move']
```
