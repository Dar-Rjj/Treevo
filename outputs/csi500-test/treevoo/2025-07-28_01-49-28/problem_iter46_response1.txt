import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 20-day simple moving average (SMA) of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate 14-day Relative Strength Index (RSI) from closing prices
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Calculate 14-day Rate of Change (ROC) from closing prices
    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Calculate 5-day and 20-day sum of trading volumes
    df['Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Volume_20'] = df['volume'].rolling(window=20).sum()
    
    # Calculate 20-day and 60-day standard deviation of daily returns
    df['Return'] = df['close'].pct_change()
    df['Volatility_20'] = df['Return'].rolling(window=20).std()
    df['Volatility_60'] = df['Return'].rolling(window=60).std()
    
    # Calculate 5-day and 20-day average true range (ATR) from high, low, and close prices
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['ATR_5'] = tr.rolling(window=5).mean()
    df['ATR_20'] = tr.rolling(window=20).mean()
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['SMA_5'] - df['SMA_20']) + (df['RSI_14'] - 50) + \
                         df['ROC_14'] + (df['Volume_5'] - df['Volume_20']) + \
                         (df['Volatility_20'] - df['Volatility_60']) + (df['ATR_5'] - df['ATR_20'])
    
    return df['alpha_factor']
