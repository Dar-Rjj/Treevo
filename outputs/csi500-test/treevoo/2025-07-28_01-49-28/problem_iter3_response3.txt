```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Smoothed Momentum
    df['smoothed_momentum'] = df['daily_return'].rolling(window=20).sum() / df['volume']
    
    # Incorporate Enhanced Price Reversion
    # Calculate 10-Day High and Low Prices
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    
    # Compute the Proximity to Range
    df['proximity_high'] = df['close'] / df['10_day_high']
    df['proximity_low'] = df['close'] / df['10_day_low']
    
    # Combine Proximity Metrics
    df['min_proximity'] = df[['proximity_high', 'proximity_low']].min(axis=1)
    df['inverted_proximity'] = 1 / df['min_proximity']
    
    # Adjust for Volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    df['adjusted_proximity'] = df['inverted_proximity'] / df['volatility']
    
    # Weight Smoothed Momentum
    df['alpha_factor'] = df['smoothed_momentum'] * df['adjusted_proximity']
    
    return df['alpha_factor']
```
