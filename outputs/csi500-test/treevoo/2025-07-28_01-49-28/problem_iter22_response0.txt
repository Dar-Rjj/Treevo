import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * (df['volume'] ** 0.5)
    
    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_close_movement
    
    # Calculate 5-day Momentum
    close_5day_ma = df['close'].rolling(window=5).mean()
    momentum_score = (df['close'] > close_5day_ma).astype(int) * 2 - 1  # 1 if current close > 5-day MA, -1 otherwise
    adjusted_combined_indicator = combined_indicator * momentum_score
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'].diff() * df['volume']).fillna(0)
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Final Combination
    final_combined_indicator = (high_low_spread + volume_weighted_momentum) * price_change_ratio + adjusted_combined_indicator
    
    # Filter for Strong Moves
    threshold = final_combined_indicator.mean() + 2 * final_combined_indicator.std()  # Example threshold
    strong_moves = (final_combined_indicator > threshold).astype(int)
    
    return pd.Series(strong_moves, index=df.index, name='alpha_factor')
