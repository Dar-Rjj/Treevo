```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_breakout_strength = (daily_range * df['volume'].pow(0.5)) + open_to_close_movement
    
    # Calculate 5-day Momentum
    momentum_5d = df['close'].rolling(window=5).mean()
    momentum_score = (df['close'] > momentum_5d).astype(int) * 2 - 1  # 1 if True, -1 otherwise
    
    # Combine Indicators
    combined_indicator = (volume_adjusted_breakout_strength * momentum_score) + open_to_close_movement
    
    # Filter for Strong Moves
    strong_move_threshold = combined_indicator.quantile(0.8)
    strong_move = (combined_indicator > strong_move_threshold).astype(int)
    
    # Calculate Simple Moving Averages (SMA)
    sma_5d = df['close'].rolling(window=5).mean()
    sma_20d = df['close'].rolling(window=20).mean()
    
    # Trend-Following Signal
    trend_signal = sma_5d - sma_20d
    
    # Analyze Past Returns
    daily_returns = df['close'].pct_change()
    weekly_returns = daily_returns.rolling(window=5).sum()
    monthly_returns = daily_returns.rolling(window=21).sum()
    weighted_sum_of_returns = (daily_returns * 0.3 + weekly_returns * 0.4 + monthly_returns * 0.3)
    
    # Measure Overnight Sentiment
    overnight_sentiment = df['open'] - df['close'].shift(1)
    
    # Evaluate Volume-Based Factors
    obv = df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)
    obv_factor = obv * (df['close'] - df['close'].shift(1))
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Volume-Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Synthesize Final Alpha Factor
    alpha_factor = (strong_move + trend_signal + weighted_sum_of_returns + overnight_sentiment + 
                    obv_factor + volume_adjusted_high_low_spread + volume_weighted_momentum) * price_change_ratio
    
    return alpha_factor
```
