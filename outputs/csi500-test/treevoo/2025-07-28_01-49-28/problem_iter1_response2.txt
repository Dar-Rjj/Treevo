```python
def heuristics_v2(df, n=5):
    # Simple Momentum for Close Prices
    close_momentum = df['close'] / df['close'].shift(n) - 1
    
    # Compute Average Volume (t to t-n)
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Adjust Momentum by Volume
    volume_factor = avg_volume / df['volume'].mean()
    volume_adjusted_close_momentum = close_momentum * volume_factor
    
    # Open-High and Open-Low Price Differences
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['low'] - df['open']
    
    # Weighted Average of Open-High, Open-Low Differences
    weighted_avg_diff = (open_high_diff + open_low_diff) / 2 * (df['volume'] / df['volume'].mean())
    
    # Amount per Unit Close Price
    amount_per_unit_close = df['amount'].rolling(window=n).sum() / df['close']
    
    # Rate of Change in this Ratio
    prev_amount_per_unit_close = df['amount'].shift(n+1).rolling(window=n).sum() / df['close'].shift(n)
    rate_of_change = amount_per_unit_close / prev_amount_per_unit_close - 1
    
    # Final Alpha Factor
    alpha_factor = (volume_adjusted_close_momentum 
                    + weighted_avg_diff 
                    + rate_of_change)
    
    return alpha_factor
```
