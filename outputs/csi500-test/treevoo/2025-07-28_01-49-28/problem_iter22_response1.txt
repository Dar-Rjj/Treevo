import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Calculate Volume-Adjusted Momentum
    close_shifted = df['close'].shift(1)
    volume_adjusted_momentum = (df['close'] - close_shifted) * df['volume']
    
    # Calculate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = ten_day_high - df['close']
    proximity_to_ten_day_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    adjusted_reversion = inverted_proximity / twenty_day_std
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_average_by_volume = ((open_high_diff + open_low_diff) / 2) * df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (volume_adjusted_breakout_strength * 
                          volume_adjusted_momentum * 
                          adjusted_reversion * 
                          weighted_average_by_volume)
    
    return final_alpha_factor
