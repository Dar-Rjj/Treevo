```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['Volume_Adjusted_Breakout_Strength'] = (df['Daily_Range'] * np.sqrt(df['volume'])) + ((df['close'] - df['open']) * df['volume'])
    
    # Incorporate Enhanced Price Reversion
    df['10D_High'] = df['high'].rolling(window=10).max()
    df['10D_Low'] = df['low'].rolling(window=10).min()
    
    df['Proximity_to_10D_High'] = df['close'] / df['10D_High']
    df['Proximity_to_10D_Low'] = df['close'] / df['10D_Low']
    
    df['Min_Proximity'] = df[['Proximity_to_10D_High', 'Proximity_to_10D_Low']].min(axis=1)
    df['Inverted_Proximity'] = 1 / df['Min_Proximity']
    
    df['20D_Std_Dev'] = df['close'].pct_change().rolling(window=20).std()
    df['Enhanced_Price_Reversion'] = df['Inverted_Proximity'] / df['20D_Std_Dev']
    
    # Integrate Open-High-Low Volatility
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['OHLC_Volatility'] = (df['Open_High_Diff'] + df['Open_Low_Diff']) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    df['20D_HL_Spread'] = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['20D_HL_Spread'] - df['20D_HL_Spread'].shift(1)
    
    # Simple Momentum for Close Prices
    df['Simple_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Volume-Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Simple_Momentum'] * df['volume']
    
    # Combine Indicators
    alpha_factor = (df['Volume_Adjusted_Breakout_Strength'] * 0.5 +
                    df['Spread_Momentum'] * 0.3 +
                    df['Volume_Weighted_Momentum'] * 0.2)
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = alpha_factor * df['close'].pct_change() * df['Enhanced_Price_Reversion'] + df['OHLC_Volatility']
    
    return df['Final_Alpha_Factor']
```
