```python
import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel alpha factor based on the given market features.
    The factor is calculated as the weighted sum of the percentage change in closing price and
    the ratio of the current volume to the 5-day moving average volume.
    
    :param df: A pandas DataFrame with (open, high, low, close, amount, volume) columns and (date) index
    :return: A pandas Series with (date) index representing the factor values
    """
    # Calculate the percentage change in closing price
    close_change = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the volume
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume ratio as the current volume divided by its 5-day moving average
    volume_ratio = df['volume'] / volume_5d_avg
    
    # Generate the alpha factor as the weighted sum of close_change and volume_ratio
    # The weights 0.7 and 0.3 are chosen heuristically
    factor = 0.7 * close_change + 0.3 * volume_ratio
    
    return factor
```
