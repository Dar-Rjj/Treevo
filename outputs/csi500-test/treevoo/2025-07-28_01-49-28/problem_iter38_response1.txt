```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between high and low prices to capture intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    
    # Calculate the buying pressure by comparing close and open prices
    buying_pressure = (df['close'] - df['open']) / df['open']
    
    # Calculate the liquidity factor by dividing the amount by volume
    liquidity = df['amount'] / df['volume']
    
    # Combine the three factors using a weighted sum
    factor_values = 0.4 * intraday_volatility + 0.3 * buying_pressure + 0.3 * liquidity
    
    return factor_values
```
