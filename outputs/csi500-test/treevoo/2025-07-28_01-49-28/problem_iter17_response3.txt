```python
import pandas as pd

def heuristics_v2(df, n=5, threshold=0.1):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_breakout_strength = daily_range * df['volume'] + open_to_close_movement
    
    # Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Simple Momentum
    simple_momentum = df['close'].pct_change(n)
    
    # Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Adjusted Momentum by Volume
    adjusted_momentum = simple_momentum * avg_volume
    
    # High-Low Return
    high_low_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Momentum and High-Low Return
    combined_momentum_high_low = adjusted_momentum + high_low_return
    
    # Open-High Price Difference
    open_high_diff = df['high'] - df['open']
    
    # Open-Low Price Difference
    open_low_diff = df['open'] - df['low']
    
    # Volatility Calculation
    volatility_calculation = pd.concat([open_high_diff, open_low_diff], axis=1).max(axis=1)
    
    # Volume Percentage Change
    volume_pct_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Volatility by Volume Change
    adjusted_volatility = volatility_calculation * volume_pct_change
    
    # Amount per Unit Close Price Rate of Change
    amount_sum_t = df['amount'].rolling(window=n).sum() / df['close']
    amount_sum_t_n = df['amount'].rolling(window=n).sum().shift(n) / df['close'].shift(n)
    amount_per_unit_close_price_roc = (amount_sum_t - amount_sum_t_n) / amount_sum_t_n
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Combine Factors
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength + 
        high_low_spread + 
        combined_momentum_high_low + 
        adjusted_volatility + 
        amount_per_unit_close_price_roc
    ) * price_change_ratio
    
    # Filter for Strong Moves
    strong_moves = (integrated_alpha_factor > threshold).astype(int)
    
    return strong_moves
```
