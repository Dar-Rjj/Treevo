```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Create a factor based on the ratio of volume to price difference
    factor = df['volume'] / price_diff
    
    # Adjust the factor by the average price
    factor *= avg_price
    
    # Generate a momentum indicator by comparing the current close to the previous close
    momentum = df['close'].pct_change()
    
    # Combine the two components into a final alpha factor
    alpha_factor = factor * momentum
    
    return alpha_factor
```
