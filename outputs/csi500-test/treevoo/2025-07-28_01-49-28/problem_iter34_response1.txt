import pandas as pd

def heuristics_v2(df):
    # Calculate raw volatility factor
    df['volatility'] = df['high'] - df['low']
    
    # Calculate raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate raw volume change factor
    df['volume_change'] = df['volume'].diff()
    
    # Calculate raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Calculate raw price-volume composite factor
    df['price_volume_composite'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_composite_5d'] = df['price_volume_composite'].rolling(window=5).sum()
    
    # Calculate raw gap factor
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Define positive and total money flow
    df['positive_money_flow'] = (df['close'] - df['open']).apply(lambda x: x if x > 0 else 0) * df['volume']
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate raw money flow index
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow'].abs()
    
    # Calculate raw transaction size factor
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Calculate raw momentum factor
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['momentum'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume'])
    
    # Calculate percentage change in closing prices over a 20-day period
    df['close_20d_pct_change'] = df['close'].pct_change(periods=20)
    
    # Calculate 10-day rate of change (ROC) of closing prices
    df['roc_10d'] = df['close'].pct_change(periods=10)

    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x))
    df['volatility_30d'] = df['log_return'].rolling(window=30).std()

    df['loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rsi'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))
    
    # Combine all factors to create the alpha factor
    alpha_factor = (
        df['volatility'] +
        df['price_movement'] +
        df['volume_change'] +
        df['volume_price_interaction'] +
        df['price_volume_composite_5d'] +
        df['gap'] +
        df['money_flow_index'] +
        df['transaction_size'] +
        df['momentum'] +
        df['close_20d_pct_change'] +
        df['roc_10d'] +
        df['volatility_30d'] +
        df['rsi']
    )
    
    return alpha_factor
