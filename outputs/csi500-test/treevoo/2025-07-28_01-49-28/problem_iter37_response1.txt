```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range + open_to_close) * df['volume']

    # High-Low Spread Momentum
    current_moving_sum_high_low_spreads = (df['high'] - df['low']).rolling(window=10).sum()
    previous_moving_sum_high_low_spreads = (df['high'] - df['low']).shift(1).rolling(window=10).sum()
    high_low_spread_momentum = current_moving_sum_high_low_spreads - previous_moving_sum_high_low_spreads

    # Volume-Adjusted Momentum
    simple_momentum = df['close'].diff(1)
    average_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * average_volume

    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = df['close'] - ten_day_high
    proximity_to_ten_day_low = ten_day_low - df['close']
    min_proximity = pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    inverted_min_proximity = -min_proximity

    # Enhanced Reversion Factor
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    deviation_from_ma_5 = df['close'] - ma_5
    deviation_from_ma_20 = df['close'] - ma_20
    combined_reversion_score = 0.7 * deviation_from_ma_5 + 0.3 * deviation_from_ma_20
    volume_adjusted_reversion = combined_reversion_score * df['volume']

    # Combine all factors
    alpha_factor = (
        volume_adjusted_breakout_strength +
        high_low_spread_momentum +
        volume_adjusted_momentum +
        inverted_min_proximity +
        volume_adjusted_reversion
    )

    return alpha_factor
```
