```python
def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of high price to low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the ratio of close price to open price
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Calculate the volume weighted by the close price
    df['volume_weighted_close'] = df['volume'] * df['close']
    
    # Combine the features into a single alpha factor
    df['alpha_factor'] = (df['close_change'] + df['high_low_ratio'] - df['close_open_ratio']) * df['volume_weighted_close']
    
    return df['alpha_factor']
```
