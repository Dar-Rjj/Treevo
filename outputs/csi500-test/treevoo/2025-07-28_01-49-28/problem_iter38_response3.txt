import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (df['close'] - ten_day_high) / (ten_day_high - ten_day_low)
    proximity_to_low = (ten_day_low - df['close']) / (ten_day_high - ten_day_low)
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / (1 + min_proximity)
    volatility = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_diff = (open_high_diff + open_low_diff) * df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Moving Sum of High-Low Price Spreads
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    weighted_average_volume = df['volume'].rolling(window=20).sum() / 20
    volume_weighted_spread = spread * weighted_average_volume
    
    # Enhanced Momentum Factor
    simple_momentum = df['close'] - df['close'].shift(20)
    smoothed_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_momentum = simple_momentum * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_low_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    amount_sum = df['amount'].rolling(window=20).sum()
    current_ratio = amount_sum / df['close']
    previous_amount_sum = df['amount'].rolling(window=40).apply(lambda x: x[:20].sum(), raw=True)
    previous_ratio = previous_amount_sum / df['close'].shift(20)
    differential = (current_ratio - previous_ratio) / previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_range * enhanced_price_reversion * weighted_avg_diff * spread_momentum
        + volume_adjusted_open_close
        + volume_weighted_spread
        + adjusted_momentum
        + combined_factors
        + differential
    ) * price_change_ratio
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.quantile(0.95)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
