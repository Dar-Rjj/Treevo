```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Simple Momentum for Close Prices
    close_price_momentum = df['close'].pct_change(periods=n)
    
    # Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Volume-Adjusted Close Price Momentum
    volume_adjusted_momentum = close_price_momentum / avg_volume
    
    # Open-High Price Difference
    open_high_diff = df['high'] - df['open']
    
    # Open-Low Price Difference
    open_low_diff = df['open'] - df['low']
    
    # Aggregate Differences (simple sum in this case, can be modified to weighted sum etc.)
    ohl_volatility = open_high_diff + open_low_diff
    
    # Sum of Amounts over past n days
    amount_sum_n_days = df['amount'].rolling(window=n).sum()
    
    # Ratio of sum of amounts over close price
    amount_close_ratio = amount_sum_n_days / df['close']
    
    # Differential Between the Ratios
    amount_ratio_diff = amount_close_ratio.pct_change(periods=n)
    
    # Integrate all components into a single alpha factor
    alpha_factor = volume_adjusted_momentum * ohl_volatility * amount_ratio_diff
    
    return alpha_factor
```
