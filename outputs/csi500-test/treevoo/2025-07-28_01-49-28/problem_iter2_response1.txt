```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    mean_10_day_volume = df['volume'].rolling(window=10).mean()
    volume_factor = df['volume'] / mean_10_day_volume
    
    # Combine Indicators
    combined_indicator = (daily_range * volume_factor) + open_to_close_movement
    
    # Filter for Strong Moves
    strong_move_threshold = 0.5
    strong_moves = (combined_indicator > strong_move_threshold).astype(int)
    
    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread
    moving_sum_high_low_spread = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)
    
    # Combine the two indicators
    alpha_factor = (strong_moves + spread_momentum) / 2
    
    return alpha_factor
```
