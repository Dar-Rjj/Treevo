```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * np.sqrt(df['volume'])
    
    # Calculate 5-day Momentum
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    momentum = (df['close'] > df['5_day_ma']).astype(int) * 2 - 1
    
    # Combine Indicators
    combined_indicator = (volume_adjusted_range + open_close_movement) * momentum
    
    # Calculate Simple Moving Averages (SMA)
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    
    # Trend-Following Signal
    trend_signal = short_sma - long_sma
    
    # Analyze Past Returns
    daily_return = df['close'].pct_change()
    weekly_return = df['close'].pct_change(5)
    monthly_return = df['close'].pct_change(21)
    weighted_returns = (daily_return + weekly_return + monthly_return) / 3
    
    # Measure Overnight Sentiment
    overnight_sentiment = df['open'] - df['close'].shift(1)
    
    # Evaluate Volume-Based Factors
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    obv_factor = obv * (df['close'] - df['close'].shift(1))
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Calculate Volume Volatility
    volume_volatility = (df['volume'] - df['volume'].shift(1)) ** 2
    
    # Final Combination
    final_indicator = (
        (volume_adjusted_spread + volume_weighted_momentum) * 
        price_change_ratio * 
        volume_volatility + 
        combined_indicator + 
        trend_signal + 
        weighted_returns + 
        overnight_sentiment + 
        obv_factor
    )
    
    # Filter for Strong Moves
    threshold = final_indicator.quantile(0.75)
    strong_move = (final_indicator > threshold).astype(int)
    
    return strong_move
```
