```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the range of the day (high - low)
    df['day_range'] = df['high'] - df['low']
    
    # Calculate the average of open and close prices
    df['avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate the difference between high and the average price
    df['high_avg_diff'] = df['high'] - df['avg_price']
    
    # Calculate the difference between low and the average price
    df['low_avg_diff'] = df['low'] - df['avg_price']
    
    # Calculate the factor as a weighted sum of the above metrics
    df['factor'] = (df['close_change'] * 0.4) + (df['volume_change'] * 0.3) + (df['high_avg_diff'] * 0.15) + (df['low_avg_diff'] * 0.15)
    
    return df['factor']
```
