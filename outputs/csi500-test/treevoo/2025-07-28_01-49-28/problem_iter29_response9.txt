```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    vol_adj_breakout_strength = (daily_range * df['volume']) + open_to_close
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    ma_high_low_spread = high_low_spread.rolling(window=20).mean()
    spread_momentum = ma_high_low_spread - ma_high_low_spread.shift(1)
    
    # Volume-Adjusted Momentum
    close_diff = df['close'].diff()
    vol_adj_momentum = close_diff * df['volume']
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Incorporate Relative Strength
    avg_close_20 = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] / avg_close_20
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_vol = (open_high_diff + open_low_diff) * df['volume']
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_high = df['close'] - ten_day_high
    proximity_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_high, proximity_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity.abs()
    std_dev_20 = df['close'].pct_change().rolling(window=20).std()
    price_reversion = inverted_proximity / std_dev_20
    
    # Long-Term Trend Indicator
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    long_term_trend = sma_50 - sma_200
    
    # Combined Indicators
    combined_indicator = 0.5 * vol_adj_breakout_strength + spread_momentum + vol_adj_momentum
    combined_indicator_with_rel_strength = combined_indicator * relative_strength
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicator_with_rel_strength * price_reversion * weighted_avg_vol * long_term_trend
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.quantile(0.8)
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)
    
    return final_alpha_factor
```
