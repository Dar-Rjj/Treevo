```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Moving Sum of High-Low Price Spread for the last 20 days
    df['moving_sum'] = df['high_low_spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum by subtracting the moving sum of the previous day from the current day
    df['spread_momentum'] = df['moving_sum'] - df['moving_sum'].shift(1)
    
    # Drop rows with NaN values which are a result of the rolling window operation
    df.dropna(inplace=True)
    
    # Return the spread momentum as a pandas Series
    return df['spread_momentum']
```
