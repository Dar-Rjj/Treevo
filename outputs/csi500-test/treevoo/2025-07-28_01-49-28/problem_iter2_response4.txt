```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread for the last 20 days
    moving_sum = volume_adjusted_spread.rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum.diff()
    
    return adjusted_spread_momentum
```
