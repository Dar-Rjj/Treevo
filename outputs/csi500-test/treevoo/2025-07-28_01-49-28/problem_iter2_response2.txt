def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume for High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Combine with Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread_momentum = volume_adjusted_high_low_spread + momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Overnight Return
    overnight_return = df['open'] - df['close'].shift(1)
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Volume Weighted Overnight Return
    volume_weighted_overnight_return = overnight_return * df['volume'].shift(1)
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Momentum Score
    momentum_score = intraday_return / (intraday_volatility + 1e-8)  # Small epsilon to avoid division by zero
    
    # Reversal Score
    reversal_score = overnight_return / (intraday_range + 1e-8)  # Small epsilon to avoid division by zero
    
    # Composite Alpha Factor
    composite_alpha_factor = (volume_adjusted_high_low_spread_momentum 
                              + momentum_score 
                              - reversal_score)
    
    return composite_alpha_factor
