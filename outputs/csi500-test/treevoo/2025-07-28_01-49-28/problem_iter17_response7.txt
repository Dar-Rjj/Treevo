```python
import pandas as pd

def heuristics_v2(df):
    # Simple Momentum for Close Prices
    n = 5
    df['Simple_Momentum'] = df['close'] - df['close'].shift(n)
    
    # Average Volume (t to t-n)
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Adjusted Momentum by Volume
    df['Volume_Factor'] = df['Average_Volume'] / df['volume'].shift(n)
    df['Adjusted_Momentum'] = df['Simple_Momentum'] * df['Volume_Factor']
    
    # Open-to-Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * df['volume']
    
    # Combine Indicators
    momentum_weight = 0.7
    movement_weight = 0.3
    strong_move_threshold = 0.05
    df['Combined_Indicator'] = (df['Volume_Adjusted_Momentum'] * momentum_weight + 
                                df['Open_to_Close_Movement'] * movement_weight)
    df['Strong_Move_Indicator'] = (df['Combined_Indicator'] > strong_move_threshold).astype(int)
    
    # High-Low Return with Spread Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Return'] = df['High_Low_Range'] / df['close'].shift(1)
    df['Daily_High_Low_Spread'] = df['high'] - df['low']
    df['Moving_Sum_High_Low_Spread'] = df['Daily_High_Low_Spread'].rolling(window=20).sum()
    df['Spread_Momentum'] = df['Moving_Sum_High_Low_Spread'] - df['Moving_Sum_High_Low_Spread'].shift(1)
    
    # Combined Momentum, High-Low Return, and Strong Move Indicator
    df['Combined_Momentum'] = df['Volume_Adjusted_Momentum'] + df['High_Low_Return'] + df['Strong_Move_Indicator']
    
    # Open-High-Low Volatility with Volume-Weighted Adjustments
    df['Open_High_Difference'] = df['high'] - df['open']
    df['Open_Low_Difference'] = df['low'] - df['open']
    df['Weighted_Average_by_Volume'] = (df['Open_High_Difference'] * df['volume'] + 
                                        df['Open_Low_Difference'] * df['volume']) / (2 * df['volume'])
    df['Volume_Percentage_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Volatility'] = df['Weighted_Average_by_Volume'] * (1 + df['Volume_Percentage_Change'])
    
    # Volume-Weighted Close Price
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    df['Moving_Average_Volume_Weighted_Close'] = df['Volume_Weighted_Close'].rolling(window=20).mean()
    df['Volume_Weighted_Momentum'] = df['Volume_Weighted_Close'] - df['Moving_Average_Volume_Weighted_Close']
    
    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    n = 10
    df['Sum_Amount_Past_n_Days'] = df['amount'].rolling(window=n).sum() / df['close']
    df['Sum_Amount_Previous_Period'] = df['amount'].shift(n).rolling(window=n).sum() / df['close'].shift(n)
    df['Rate_of_Change'] = df['Sum_Amount_Past_n_Days'] - df['Sum_Amount_Previous_Period']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Combined_Momentum'] + 
                          df['Adjusted_Volatility'] + 
                          df['Volume_Weighted_Momentum'] + 
                          df['Rate_of_Change'])
    
    return df['Alpha_Factor']
```
