```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Momentum
    momentum = df['close'].diff()
    
    # Weight the Momentum by Volume
    weighted_momentum = momentum * df['volume']
    
    # Combine Volume-Adjusted High-Low Spread and Weighted Momentum
    combined_factor = (volume_adjusted_high_low_spread + weighted_momentum).dropna()
    
    return combined_factor
```
