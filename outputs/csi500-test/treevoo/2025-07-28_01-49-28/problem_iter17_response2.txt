```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']

    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']

    # Adjust for Volume
    volume_adjusted_daily_range = daily_range * np.sqrt(df['volume'])
    adjusted_open_close_movement = open_close_movement + volume_adjusted_daily_range

    # Calculate 5-day Momentum
    five_day_moving_avg = df['close'].rolling(window=5).mean()
    five_day_momentum = (df['close'] > five_day_moving_avg).astype(int) * 2 - 1

    # Incorporate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']

    # Calculate Exponential Moving Average of Momentum
    ema_momentum = volume_weighted_momentum.ewm(span=10, adjust=False).mean()

    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']

    # Calculate Relative Strength Index (RSI)
    daily_returns = df['close'].pct_change()
    positive_returns = daily_returns.where(daily_returns > 0, 0)
    negative_returns = -daily_returns.where(daily_returns < 0, 0)
    average_gains = positive_returns.ewm(alpha=1/14, adjust=False).mean()
    average_losses = negative_returns.ewm(alpha=1/14, adjust=False).mean()
    relative_strength = average_gains / average_losses
    rsi = 100 - (100 / (1 + relative_strength))

    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    distance_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    distance_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    proximity = 1 / (1 + np.minimum(distance_to_high, distance_to_low))
    std_dev = daily_returns.rolling(window=20).std()
    adjusted_proximity = proximity / std_dev

    # Combine Indicators
    combined_indicator = (adjusted_open_close_movement + high_low_spread) * five_day_momentum

    # Final Factor Combination
    final_factor = (high_low_spread + volume_weighted_momentum) * price_change_ratio * ema_momentum * adjusted_proximity * rsi * combined_indicator

    # Filter for Strong Moves
    threshold = final_factor.quantile(0.75)
    final_factor = (final_factor > threshold).astype(int)

    return final_factor
```
