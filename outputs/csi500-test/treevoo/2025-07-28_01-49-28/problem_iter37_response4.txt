```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    df['adjusted_daily_range'] = (df['high'] - df['low']) * df['volume']
    
    # Price Reversion
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['proximity_to_10_day_range'] = (df['10_day_high'] - df['close']) / (df['10_day_high'] - df['10_day_low'])
    df['std_dev_20_days'] = df['close'].pct_change().rolling(window=20).std()
    df['price_reversion'] = 1 - df['proximity_to_10_day_range'] / df['std_dev_20_days']

    # Open-High-Low Volatility
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    df['ohlv_weighted'] = (df['open_high_diff'] + df['open_low_diff']) * df['volume']
    df['ohlv_volatility'] = df['ohlv_weighted'] / df['volume']

    # Spread Momentum
    df['daily_spread'] = df['high'] - df['low']
    df['spread_momentum'] = df['daily_spread'] - df['daily_spread'].shift(1)

    # Volume-Adjusted Open-to-Close Movement
    df['volume_adjusted_open_close'] = (df['close'] - df['open']) * df['volume']

    # Simple Momentum
    df['simple_momentum'] = df['close'] - df['close'].shift(1)

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_daily_range'] *
        df['price_reversion'] *
        df['ohlv_volatility'] *
        df['spread_momentum'] +
        df['volume_adjusted_open_close'] +
        df['simple_momentum']
    )

    # Dynamic Threshold
    threshold = df['final_alpha_factor'].median()  # Example: using median as a dynamic threshold
    df['alpha_factor'] = (df['final_alpha_factor'] > threshold).astype(int)

    return df['alpha_factor']
```
