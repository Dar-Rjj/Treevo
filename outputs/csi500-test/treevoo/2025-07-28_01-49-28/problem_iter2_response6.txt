```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Close Price Momentum
    simple_momentum_21 = df['close'] / df['close'].shift(21) - 1
    simple_momentum_63 = df['close'] / df['close'].shift(63) - 1
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_factor = (avg_volume_21 + avg_volume_63) / 2
    vol_adjusted_momentum = (simple_momentum_21 + simple_momentum_63) * volume_factor
    
    # Combined Price Momentum
    price_momentum_21 = df['close'] - df['close'].shift(21)
    price_momentum_63 = df['close'] - df['close'].shift(63)
    combined_momentum = 0.5 * price_momentum_21 + 0.5 * price_momentum_63
    
    # Volume Reversal Indicator
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_diff = avg_volume_21 - avg_volume_63
    volume_reversal = volume_diff / avg_volume_63
    
    # Open-High-Low Volatility Factor
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff + open_low_diff) * df['volume'] / (avg_volume_21 + avg_volume_63)

    # Price-Amount Interaction Effect
    amount_per_unit_close = df['amount'].rolling(window=21).sum() / df['close']
    previous_amount_per_unit_close = df['amount'].rolling(window=42).apply(lambda x: x[:21].sum(), raw=False) / df['close'].shift(21)
    rate_of_change = amount_per_unit_close - previous_amount_per_unit_close
    
    # Combine all factors
    alpha_factor = vol_adjusted_momentum + combined_momentum + volume_reversal + weighted_volatility + rate_of_change
    return alpha_factor
```
