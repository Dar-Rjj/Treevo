```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the average price for each day
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the normalized volume
    norm_volume = df['volume'] / df['amount']
    
    # Calculate the factor value
    factor_value = (avg_price * norm_volume) / df['close']
    
    return factor_value
```
