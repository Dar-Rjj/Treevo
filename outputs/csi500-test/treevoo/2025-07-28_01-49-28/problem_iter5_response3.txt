```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Momentum for Close Prices
    df['momentum_21'] = df['close'].pct_change(21)
    df['momentum_63'] = df['close'].pct_change(63)
    
    # Adjust Momentum by Volume
    df['avg_volume_21'] = df['volume'].rolling(window=21).mean()
    df['avg_volume_63'] = df['volume'].rolling(window=63).mean()
    
    df['volume_adjusted_momentum_21'] = df['momentum_21'] * (df['avg_volume_21'] / df['avg_volume_21'].mean())
    df['volume_adjusted_momentum_63'] = df['momentum_63'] * (df['avg_volume_63'] / df['avg_volume_63'].mean())
    
    # Combine 21-day and 63-day Momentums
    df['combined_momentum'] = 0.5 * df['volume_adjusted_momentum_21'] + 0.5 * df['volume_adjusted_momentum_63']
    
    # Volume-Adjusted High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Volume Reversal Indicator
    df['avg_volume_21'] = df['volume'].rolling(window=21).mean()
    df['avg_volume_63'] = df['volume'].rolling(window=63).mean()
    df['volume_reversal'] = (df['avg_volume_21'] - df['avg_volume_63']) / df['avg_volume_63']
    
    # Open-High-Low Volatility Factor
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    
    df['weighted_open_high_low_volatility'] = (df['open_high_diff'] + df['open_low_diff']) * (df['volume'] / df['volume'].mean())
    
    # Price-Amount Interaction Effect
    n = 21
    df['sum_amount_n_days'] = df['amount'].rolling(window=n).sum()
    df['ratio_amount_close'] = df['sum_amount_n_days'] / df['close']
    
    df['prev_ratio_amount_close'] = df['ratio_amount_close'].shift(n)
    df['rate_of_change_ratio'] = df['ratio_amount_close'] - df['prev_ratio_amount_close']
    
    # Combine Adjusted Momentum, High-Low Spread, and Volume Reversal
    df['combined_indicator'] = (
        df['combined_momentum'] +
        df['volume_adjusted_high_low_spread'] +
        df['volume_reversal']
    )
    
    # Finalize VIMRHL
    lookback_period = 63
    df['overall_average_return'] = df['close'].pct_change().rolling(window=lookback_period).mean()
    df['vimrhl'] = df['combined_indicator'].rolling(window=lookback_period).mean() - df['overall_average_return']
    
    return df['vimrhl']
```
