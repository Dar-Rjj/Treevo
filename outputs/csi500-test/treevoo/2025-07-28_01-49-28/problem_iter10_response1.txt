```python
def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    df['Volume_Adjusted_Spread'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Moving Sum of the Volume-Adjusted High-Low Spread over the last 20 days
    df['Moving_Sum'] = df['Volume_Adjusted_Spread'].rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    df['Momentum'] = df['Moving_Sum'] - df['Moving_Sum'].shift(1)
    
    # The factor is labeled as 'Volume and Momentum Enhanced High-Low Spread'
    return df['Momentum'].rename('Volume and Momentum Enhanced High-Low Spread')
```
