```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range + open_to_close_movement) * df['volume']

    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_20d = high_low_spread.rolling(window=20).sum()
    previous_day_moving_sum = moving_sum_20d.shift(1)
    high_low_spread_momentum = moving_sum_20d - previous_day_moving_sum

    # 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()

    # Proximity to 10-Day Range
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_min_proximity = 1 / min_proximity

    # Adjust for 20-Day Volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()
    adjusted_proximity = inverted_min_proximity / twenty_day_volatility

    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume']

    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    amount_per_unit_close = df['amount'] / df['close']
    amount_per_unit_close_sum_5d = amount_per_unit_close.rolling(window=5).sum()
    amount_per_unit_close_sum_10d = amount_per_unit_close.rolling(window=10).sum()
    rate_of_change = (amount_per_unit_close_sum_5d - amount_per_unit_close_sum_10d) / amount_per_unit_close_sum_10d

    # Volume-Weighted Moving Average Crossover
    short_term_vwma = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    long_term_vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwma_crossover = short_term_vwma - long_term_vwma

    # High-Frequency Trading Impact
    avg_daily_volume = df['volume'].rolling(window=20).mean()
    large_volume_trades = (df['volume'] > 3 * avg_daily_volume).rolling(window=10).sum()
    hf_trend = (df['close'].pct_change() > 0).rolling(window=10).sum() - (df['close'].pct_change() < 0).rolling(window=10).sum()
    hf_impact = large_volume_trades * hf_trend

    # Final Alpha Factor
    strong_move_indicator = (df['close'] - df['open']) / (df['high'] - df['low'])
    final_alpha_factor = (
        volume_adjusted_breakout_strength * 
        high_low_spread_momentum * 
        adjusted_proximity * 
        open_high_low_volatility * 
        rate_of_change * 
        vwma_crossover * 
        hf_impact * 
        strong_move_indicator
    )

    return final_alpha_factor
```
