```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']

    # Calculate Open-to-Close Movement
    open_to_close = df['close'] - df['open']

    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']

    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close

    # Filter for Strong Moves
    threshold = combined_indicator.mean()  # Example threshold: mean of the combined indicator
    strong_moves = (combined_indicator > threshold).astype(int)

    # Calculate Moving Sum of High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    moving_sum_20_days = high_low_spread.rolling(window=20).sum()

    # Calculate Spread Momentum
    spread_momentum = moving_sum_20_days - moving_sum_20_days.shift(1)

    # Final Factor
    factor = strong_moves * spread_momentum

    return factor
```
