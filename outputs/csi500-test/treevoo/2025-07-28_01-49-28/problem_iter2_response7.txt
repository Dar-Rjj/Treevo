```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['True Range'] = 0
    df['True Range'] = df.apply(
        lambda x: max(x['High'] - x['Low'], 
                      abs(x['Close'] - df.loc[x.name - pd.Timedelta(days=1), 'Close']), 
                      x['High'] - df.loc[x.name - pd.Timedelta(days=1), 'Close'] if x['High'] > df.loc[x.name - pd.Timedelta(days=1), 'Close'] else df.loc[x.name - pd.Timedelta(days=1), 'Close'] - x['Low']),
        axis=1
    )

    # Calculate Moving Average of True Range
    moving_avg_true_range = df['True Range'].rolling(window=21).mean()

    # Calculate Price Momentum
    price_momentum_21 = df['Close'] - df['Close'].shift(21)
    price_momentum_63 = df['Close'] - df['Close'].shift(63)
    combined_price_momentum = 0.5 * price_momentum_21 + 0.5 * price_momentum_63

    # Calculate Volume Reversal
    volume_21_avg = df['Volume'].rolling(window=21).mean()
    volume_63_avg = df['Volume'].rolling(window=63).mean()
    volume_diff = volume_21_avg - volume_63_avg
    normalized_volume_reversal = volume_diff / volume_63_avg

    # Combine All Indicators
    alpha_factor = (moving_avg_true_range * combined_price_momentum * normalized_volume_reversal).dropna()

    return alpha_factor
```
