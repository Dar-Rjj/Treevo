```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    
    proximity_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    
    min_proximity = pd.DataFrame({'proximity_high': proximity_high, 'proximity_low': proximity_low}).min(axis=1)
    inverted_proximity = 1 - min_proximity
    
    twenty_day_volatility = df['close'].rolling(window=20).std()
    reversion_factor = inverted_proximity / twenty_day_volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    
    weighted_avg = (open_high_diff + open_low_diff) * df['volume']
    normalized_weighted_avg = weighted_avg / df['volume'].rolling(window=20).mean()
    
    # Moving Sum of 20-Day High-Low Spreads
    high_low_spread = df['high'] - df['low']
    moving_sum = high_low_spread.rolling(window=20).sum()
    
    # Spread Momentum
    spread_momentum = moving_sum - moving_sum.shift(1)
    
    # Final Alpha Factor
    alpha_factor = (volume_adjusted_range 
                    * reversion_factor 
                    * normalized_weighted_avg 
                    * spread_momentum)
    
    return alpha_factor
```
