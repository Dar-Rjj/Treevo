import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    adjusted_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    twenty_day_volatility = df['close'].pct_change().rolling(window=20).std()
    reversion_factor = inverted_proximity / twenty_day_volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_diff = (open_high_diff + open_low_diff) / 2 * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spreads = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spreads.diff()
    
    # Calculate 5-day Momentum
    five_day_ma = df['close'].rolling(window=5).mean()
    momentum_5d = np.where(df['close'] > five_day_ma, 1, -1)
    
    # Calculate 5-day Volatility
    volatility_5d = df['close'].rolling(window=5).std() * np.sqrt(df['volume'])
    
    # Combine Indicators
    combined_indicator = (adjusted_daily_range * np.sqrt(df['volume']) 
                          + weighted_avg_diff 
                          + volatility_5d) * momentum_5d
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * reversion_factor * weighted_avg_diff 
                          * spread_momentum * combined_indicator)
    return final_alpha_factor.rename('Final Alpha Factor')
