import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    spread = df['high'] - df['low']
    
    # Incorporate Volume into Daily Range
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume in Open-to-Close Movement
    volume_factor = df['volume'] / df['volume'].mean()
    volume_adjusted_breakout_strength = open_to_close_movement * volume_factor + volume_adjusted_spread
    
    # Calculate Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Exponential Moving Average of Momentum
    ema_momentum = volume_weighted_momentum.ewm(span=10, adjust=False).mean()
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Calculate Relative Strength Index (RSI)
    daily_returns = df['close'].pct_change()
    gains = daily_returns.where(daily_returns > 0, 0)
    losses = -daily_returns.where(daily_returns < 0, 0)
    avg_gain = gains.rolling(window=14).mean().ewm(alpha=1/14, adjust=False).mean()
    avg_loss = losses.rolling(window=14).mean().ewm(alpha=1/14, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (min_proximity + 1e-8)  # Adding a small constant to avoid division by zero
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    adjusted_proximity = inverted_proximity / twenty_day_std
    
    # Combine Factors
    integrated_factor = (volume_adjusted_spread + volume_weighted_momentum) * price_change_ratio
    integrated_factor *= ema_momentum
    integrated_factor *= adjusted_proximity
    integrated_factor *= rsi
    integrated_factor += volume_adjusted_breakout_strength
    return integrated_factor.rename("Integrated High-Frequency Trading and Volume-Adjusted Momentum with RSI")
