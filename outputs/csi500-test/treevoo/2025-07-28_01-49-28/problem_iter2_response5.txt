```python
def heuristics_v2(df):
    # Calculate Daily High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Incorporate Volume
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread
    df['moving_sum'] = df['volume_adjusted_high_low_spread'].rolling(window=20).sum()
    
    # Calculate Combined Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    df['combined_momentum'] = (df['price_momentum'] 
                               + df['moving_sum'] 
                               - df['moving_sum'].shift(1))
    
    return df['combined_momentum']
```
