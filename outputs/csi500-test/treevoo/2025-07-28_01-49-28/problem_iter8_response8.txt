import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume: Multiply Daily Range by Volume and Add Open-to-Close Movement
    volume_factor = 10000  # Example volume factor, can be adjusted based on the data
    volume_adjusted_breakout_strength = (daily_range * df['volume'] / volume_factor) + open_to_close_movement
    
    # Calculate Simple Momentum for Close Prices
    n = 5  # Example lookback period, can be adjusted
    simple_momentum = df['close'].pct_change(periods=n)
    
    # Adjust Momentum by Average Volume
    average_volume = df['volume'].rolling(window=n).mean()
    adjusted_momentum = simple_momentum * (df['volume'] / average_volume)
    
    # Calculate High-to-Low Return
    high_low_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combine Indicators
    breakout_weight = 0.7  # Weight for volume-adjusted breakout strength, can be adjusted
    combined_indicator = (volume_adjusted_breakout_strength * breakout_weight) + adjusted_momentum
    
    # Filter for Strong Moves
    threshold = 0.05  # Example threshold, can be adjusted
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
