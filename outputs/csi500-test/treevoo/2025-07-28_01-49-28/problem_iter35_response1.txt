```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    
    proximity_to_high = (df['close'] - ten_day_high).abs()
    proximity_to_low = (df['close'] - ten_day_low).abs()
    
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    
    # Adjust for Volatility
    daily_returns = df['close'].pct_change()
    twenty_day_std = daily_returns.rolling(window=20).std()
    volatility_adjusted_proximity = inverted_proximity / twenty_day_std
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    
    weighted_open_high = open_high_diff * df['volume']
    weighted_open_low = open_low_diff * df['volume']
    
    open_high_low_volatility = (weighted_open_high + weighted_open_low) / (2 * df['volume'])
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread.diff()
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_range * 
        volatility_adjusted_proximity * 
        open_high_low_volatility * 
        spread_momentum + 
        volume_adjusted_open_close
    )
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.quantile(0.75)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
