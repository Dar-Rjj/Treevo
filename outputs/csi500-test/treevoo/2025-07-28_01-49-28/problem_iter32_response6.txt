```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close
    
    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    spread_momentum = high_low_spread.rolling(window=20).sum() - high_low_spread.shift(1).rolling(window=20).sum()
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume = (df['volume'] + df['volume'].shift(1)) / 2
    adjusted_momentum_by_volume = simple_momentum * (df['volume'] / avg_volume)
    
    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = df['close'] - ten_day_high
    proximity_to_ten_day_low = df['close'] - ten_day_low
    inverted_proximity = 1 / pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).abs().min(axis=1)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    enhanced_with_ten_day = inverted_proximity / twenty_day_volatility
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    max_volatility = pd.concat([open_high_diff, open_low_diff], axis=1).max(axis=1)
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = max_volatility * volume_percentage_change
    
    # Enhanced Momentum Factor
    simple_momentum_n = df['close'] - df['close'].shift(20)
    smoothed_volume = df['volume'].ewm(span=20).mean()
    adjusted_momentum = simple_momentum_n * smoothed_volume
    
    # High-Low Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Amount per Unit Close Price Rate of Change
    sum_amounts = df['amount'].rolling(window=20).sum()
    current_ratio = sum_amounts / df['close']
    previous_ratio = df['amount'].shift(20).rolling(window=20).sum() / df['close'].shift(20)
    differential = current_ratio - previous_ratio
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength +
        spread_momentum +
        adjusted_momentum_by_volume +
        combined_factors +
        adjusted_volatility +
        differential
    ) * price_change_ratio
    
    # Enhanced Alpha Factor
    threshold = 1.5
    strong_moves = (integrated_alpha_factor > threshold).astype(int)
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_trend_score = long_term_momentum - short_term_momentum
    enhanced_alpha_factor = integrated_alpha_factor + price_trend_score
    
    return enhanced_alpha_factor
```
