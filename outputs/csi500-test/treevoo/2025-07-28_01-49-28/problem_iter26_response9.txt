```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices as a raw volatility factor
    df['volatility'] = df['high'] - df['low']
    
    # Calculate (close - open) / (high - low) as a raw price movement factor
    df['price_movement'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Compute today's volume - yesterday's volume as a raw volume change factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate (volume * (close - open)) as a raw volume-price interaction factor
    df['volume_price_interaction'] = df['volume'] * (df['close'] - df['open'])
    
    # Summarize the (close - open) * volume over the last 5 days as a raw price-volume composite factor
    df['price_volume_composite_5d'] = df['volume'] * (df['close'] - df['open']).rolling(window=5).sum()
    
    # Calculate the (open - previous close) as a raw gap factor
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Define positive money flow as (close - open) * volume when close > open, otherwise 0
    df['positive_money_flow'] = df.apply(lambda row: (row['close'] - row['open']) * row['volume'] if row['close'] > row['open'] else 0, axis=1)
    
    # Define total money flow as (close - open) * volume, regardless of price direction
    df['total_money_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Use the ratio of positive to total money flow as a raw money flow index
    df['money_flow_index'] = df['positive_money_flow'] / df['total_money_flow']
    
    # Calculate the (amount / volume) to find the average transaction size
    df['transaction_size'] = df['amount'] / df['volume']
    
    # Calculate (close - open) * (amount / volume) as a raw price-amount interaction factor
    df['price_amount_interaction'] = (df['close'] - df['open']) * df['transaction_size']
    
    # Calculate the (close - open) * (volume / average volume) as a raw momentum factor
    df['average_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['momentum'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume_20d'])
    
    # Calculate the (close - open) * (volume / average volume) * (amount / volume) as a raw price-volume-amount composite factor
    df['average_transaction_size_20d'] = df['transaction_size'].rolling(window=20).mean()
    df['price_volume_amount_composite'] = (df['close'] - df['open']) * (df['volume'] / df['average_volume_20d']) * (df['transaction_size'] / df['average_transaction_size_20d'])
    
    # Return the factors as a Series
    return df[['volatility', 'price_movement', 'volume_change', 'volume_price_interaction', 
               'price_volume_composite_5d', 'gap', 'money_flow_index', 'transaction_size', 
               'price_amount_interaction', 'momentum', 'price_volume_amount_composite']]
```
