```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    df['Open_Close_Movement'] = df['close'] - df['open']
    
    # Adjust for Volume
    df['Volume_Adjusted_Breakout_Strength'] = (df['Daily_Range'] * df['volume']) + df['Open_Close_Movement']
    
    # Calculate Moving Average of High-Low Price Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['High_Low_Spread_MA'] = df['High_Low_Spread'].rolling(window=20).mean()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['High_Low_Spread_MA'] - df['High_Low_Spread_MA'].shift(1)
    
    # Incorporate Volume into High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Calculate Volume-Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate Open-High-Low Volatility
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Open_High_Low_Volatility'] = (df['Open_High_Diff'] + df['Open_Low_Diff']) * df['volume']
    
    # Incorporate Enhanced Price Reversion
    df['10_Day_High'] = df['high'].rolling(window=10).max()
    df['10_Day_Low'] = df['low'].rolling(window=10).min()
    df['Proximity_to_High'] = df['10_Day_High'] - df['close']
    df['Proximity_to_Low'] = df['close'] - df['10_Day_Low']
    df['Min_Proximity'] = df[['Proximity_to_High', 'Proximity_to_Low']].min(axis=1)
    df['Inverted_Proximity'] = 1 / (df['Min_Proximity'] + 1e-6)  # Add a small constant to avoid division by zero
    df['20_Day_STD'] = df['close'].pct_change().rolling(window=20).std()
    df['Enhanced_Price_Reversion'] = df['Inverted_Proximity'] / (df['20_Day_STD'] + 1e-6)  # Add a small constant to avoid division by zero
    
    # Calculate Moving Sum of High-Low Price Spreads
    df['High_Low_Spread_Sum'] = df['High_Low_Spread'].rolling(window=20).sum()
    
    # Calculate Spread Momentum
    df['Spread_Momentum'] = df['High_Low_Spread_Sum'] - df['High_Low_Spread_Sum'].shift(1)
    
    # Combine Indicators
    df['Combined_Indicator'] = (df['Volume_Adjusted_Breakout_Strength'] * 0.5 + 
                                df['Spread_Momentum'] + 
                                df['Volume_Adjusted_Momentum'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Combined_Indicator'] * 
                                 df['Enhanced_Price_Reversion'] * 
                                 df['Open_High_Low_Volatility'] * 
                                 df['Spread_Momentum'])
    
    return df['Final_Alpha_Factor']
```
