import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (5 days) and long-term (20 days) simple moving averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Compute the difference between short-term and long-term SMAs as a trend signal
    df['SMA_diff'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate daily, weekly (5 days), and monthly (20 days) return series
    df['daily_return'] = df['close'].pct_change()
    df['weekly_return'] = df['close'].pct_change(periods=5)
    df['monthly_return'] = df['close'].pct_change(periods=20)
    
    # Generate a weighted sum of returns over different periods to capture varying momentum signals
    df['momentum_score'] = 0.5 * df['daily_return'] + 0.3 * df['weekly_return'] + 0.2 * df['monthly_return']
    
    # Measure the gap up or down (open - close of the previous day) as an indicator of overnight sentiment
    df['overnight_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Introduce a factor that combines OBV with closing price movement to gauge market interest
    df['market_interest'] = df['OBV'] * df['close'].pct_change()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['SMA_diff'] + df['momentum_score'] + df['overnight_gap'] + df['market_interest']
    
    return df['alpha_factor']
