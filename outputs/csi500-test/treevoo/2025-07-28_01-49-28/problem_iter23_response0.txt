```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the intraday return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume-to-amount ratio
    df['volume_to_amount'] = df['volume'] / df['amount']
    
    # Create a factor that combines these three components
    df['alpha_factor'] = df['intraday_return'] * df['daily_range'] / df['volume_to_amount']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
