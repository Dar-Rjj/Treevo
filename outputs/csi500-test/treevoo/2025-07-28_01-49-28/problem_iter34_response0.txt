import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']

    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']

    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (1 + min_proximity)
    
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    volatility_adjusted_reversion = inverted_proximity / twenty_day_std
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    
    open_high_volatility = open_high_diff * df['volume']
    open_low_volatility = open_low_diff * df['volume']
    
    weighted_average_volatility = (open_high_volatility + open_low_volatility) / (2 * df['volume'])

    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spread = (df['high'] - df['low']).rolling(window=20).sum()

    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)

    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']

    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_range * 
        volatility_adjusted_reversion * 
        weighted_average_volatility * 
        spread_momentum
    ) + volume_adjusted_open_close

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + 2 * final_alpha_factor.rolling(window=20).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)

    return strong_moves
