import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    breakout_strength = (daily_range + open_to_close_movement) * df['volume']

    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low = high_low_spread.rolling(window=20).sum()
    previous_day_moving_sum = moving_sum_high_low.shift(1)
    high_low_spread_momentum = moving_sum_high_low - previous_day_moving_sum

    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume

    # Combine Indicators
    combined_indicator = (
        0.4 * breakout_strength +
        0.3 * high_low_spread_momentum +
        0.3 * volume_adjusted_momentum
    )

    # Filter for Strong Moves
    dynamic_threshold = combined_indicator.rolling(window=20).quantile(0.75)
    strong_moves = (combined_indicator > dynamic_threshold).astype(int)

    return strong_moves
