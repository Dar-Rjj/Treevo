```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Daily Range
    adj_daily_range = (df['high'] - df['low']) * df['volume']

    # 10-Day High and Low
    highest_high_10d = df['high'].rolling(window=10).max()
    lowest_low_10d = df['low'].rolling(window=10).min()

    # Proximity to 10-Day Range
    close_to_10d_high = df['close'] / highest_high_10d
    close_to_10d_low = df['close'] / lowest_low_10d

    # Inverted Proximity / 20-Day Standard Deviation of Returns
    std_dev_20d = df['close'].pct_change().rolling(window=20).std()
    price_reversion = (1 / (close_to_10d_high - close_to_10d_low)) / std_dev_20d

    # Open-High-Low Volatility Weighted by Volume
    open_high_vol = (df['high'] - df['open']) * df['volume']
    open_low_vol = (df['open'] - df['low']) * df['volume']
    ohl_volatility = (open_high_vol + open_low_vol) / 2

    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    spread_momentum = high_low_spread.rolling(window=5).sum() - high_low_spread.shift(1).rolling(window=5).sum()

    # Volume-Adjusted Open-to-Close Movement
    vol_adj_open_close = (df['close'] - df['open']) * df['volume']

    # Simple Momentum
    simple_momentum = df['close'] - df['close'].shift(1)

    # Final Alpha Factor
    final_alpha_factor = adj_daily_range * price_reversion * ohl_volatility * spread_momentum
    final_alpha_factor += vol_adj_open_close
    final_alpha_factor += simple_momentum

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + 2 * final_alpha_factor.rolling(window=20).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)

    return strong_moves
```
