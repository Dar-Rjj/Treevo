```python
import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Volume-Adjusted Breakout Strength
    df['daily_range'] = df['high'] - df['low']
    df['open_to_close_move'] = df['close'] - df['open']
    df['volume_adjusted_breakout_strength'] = (df['daily_range'] + df['open_to_close_move']) / df['volume']

    # High-Low Spread Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['sum_high_low_spread_20d'] = df['high_low_spread'].rolling(window=20).sum()
    df['prev_day_sum_high_low_spread_20d'] = df['sum_high_low_spread_20d'].shift(1)
    df['high_low_spread_momentum'] = df['sum_high_low_spread_20d'] - df['prev_day_sum_high_low_spread_20d']

    # Enhance with 10-Day High and Low Prices
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    df['proximity_to_10d_high'] = (df['close'] - df['10d_high']) / df['10d_high']
    df['proximity_to_10d_low'] = (df['10d_low'] - df['close']) / df['10d_low']
    df['min_proximity_inverted'] = 1 / (df[['proximity_to_10d_high', 'proximity_to_10d_low']].abs().min(axis=1) + 1e-6)
    df['volatility_20d'] = df['close'].rolling(window=20).std()
    df['enhanced_10d'] = df['min_proximity_inverted'] / df['volatility_20d']

    # Integrate Open-High-Low Volatility
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    df['weighted_avg_volatility'] = (df['open_high_diff'] + df['open_low_diff']) * df['volume'] / df['volume'].rolling(window=20).mean()

    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    df['sum_amounts_n_days'] = df['amount'].rolling(window=n).sum() / df['close'].rolling(window=n).mean()
    df['sum_amounts_prev_period'] = df['amount'].shift(n+1).rolling(window=n).sum() / df['close'].shift(n+1).rolling(window=n).mean()
    df['amount_per_unit_close_change'] = df['sum_amounts_n_days'] - df['sum_amounts_prev_period']

    # Combined Indicator
    df['combined_indicator'] = (
        df['volume_adjusted_breakout_strength'] +
        df['high_low_spread_momentum'] +
        df['enhanced_10d'] +
        df['weighted_avg_volatility'] +
        df['amount_per_unit_close_change']
    )

    # Strong Move Indicator
    df['strong_move_indicator'] = (df['proximity_to_10d_high'] * (df['close'] > df['10d_high'])) + (df['proximity_to_10d_low'] * (df['close'] < df['10d_low']))

    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_indicator'] + df['strong_move_indicator']

    return df['final_alpha_factor']
```
