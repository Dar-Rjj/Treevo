```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adj_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Enhanced Reversion
    highest_high_10d = df['high'].rolling(window=10).max()
    lowest_low_10d = df['low'].rolling(window=10).min()
    proximity_to_10d_high = df['close'] / highest_high_10d
    proximity_to_10d_low = df['close'] / lowest_low_10d
    min_proximity = pd.DataFrame({'prox_high': proximity_to_10d_high, 'prox_low': proximity_to_10d_low}).min(axis=1)
    inverted_proximity = 1 / min_proximity
    std_20d = df['close'].rolling(window=20).std()
    enhanced_reversion = inverted_proximity / std_20d
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff + open_low_diff) * df['volume']
    
    # Spread Momentum
    spread_momentum = (df['high'] - df['low']).rolling(window=30).sum().diff()
    
    # Trend Strength
    trend_strength = (df['close'].rolling(window=50).mean() - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).mean()
    
    # Combined Momentum
    ten_day_return = (df['close'] / df['close'].shift(10)) - 1
    thirty_day_return = (df['close'] / df['close'].shift(30)) - 1
    combined_momentum = ten_day_return + thirty_day_return
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Final Alpha Factor
    final_alpha_factor = (
        adj_daily_range * 
        enhanced_reversion * 
        weighted_avg * 
        spread_momentum * 
        trend_strength * 
        combined_momentum * 
        volume_adjusted_momentum
    )
    
    return final_alpha_factor
```
