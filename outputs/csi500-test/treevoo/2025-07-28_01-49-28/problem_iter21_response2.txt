import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_close_movement
    
    # Calculate Moving Sum of High-Low Price Spread
    high_low_spread_moving_sum = df['high'] - df['low']
    high_low_spread_moving_sum = high_low_spread_moving_sum.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = high_low_spread_moving_sum.diff()
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'].diff()
    
    # Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * avg_volume
    
    # Incorporate Relative Strength
    avg_close_20_day = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] / avg_close_20_day
    
    # Combine Indicators
    combined_indicator = (volume_adjusted_breakout_strength * 0.5) + \
                         (spread_momentum * 0.3) + \
                         (volume_adjusted_momentum * 0.2)
    
    # Incorporate Relative Strength
    combined_indicator_with_relative_strength = combined_indicator * relative_strength
    
    # Filter for Strong Moves
    dynamic_threshold = combined_indicator_with_relative_strength.mean() + 2 * combined_indicator_with_relative_strength.std()
    strong_moves = combined_indicator_with_relative_strength > dynamic_threshold
    factor_values = strong_moves.astype(int)
    
    return pd.Series(factor_values, index=df.index)
