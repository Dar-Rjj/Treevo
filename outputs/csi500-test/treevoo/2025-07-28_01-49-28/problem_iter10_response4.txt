```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Calculate 3-day Momentum
    moving_avg_3d = df['close'].rolling(window=3).mean()
    momentum = (df['close'] > moving_avg_3d).astype(int) * 2 - 1  # 1 if true, -1 otherwise
    
    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close_movement
    combined_indicator_with_momentum = combined_indicator * momentum
    
    # Filter for Strong Moves
    threshold = combined_indicator.std()  # Define a threshold based on standard deviation of the combined indicator
    strong_moves = (combined_indicator_with_momentum > threshold).astype(int)
    
    return strong_moves
```
