```python
def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = df['volume'] / df['volume'].mean()
    adjusted_daily_range = daily_range * volume_factor
    
    # Calculate Momentum
    moving_average_3d = df['close'].rolling(window=3).mean()
    momentum = (df['close'] > moving_average_3d).astype(int) * 2 - 1  # 1 if True, -1 otherwise
    
    # Combine Indicators
    combined_indicator = adjusted_daily_range + open_to_close_movement
    combined_indicator *= momentum
    
    # Filter for Strong Moves
    threshold = combined_indicator.mean() + combined_indicator.std()
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
```
