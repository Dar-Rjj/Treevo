```python
def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Combine Indicators
    combined_indicator = (volume_adjusted_range + open_close_movement + momentum)
    
    # Define a threshold for strong moves
    threshold = 0.5 * (df['high'] - df['low']).mean()
    
    # Filter for Strong Moves
    factor_values = (combined_indicator > threshold).astype(int)
    
    return factor_values
```
