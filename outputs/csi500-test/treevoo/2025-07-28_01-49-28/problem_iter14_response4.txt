```python
import pandas as pd

def heuristics_v2(df, threshold=0.5):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Weighted Momentum
    momentum = df['close'].diff()
    volume_weighted_momentum = momentum * df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Combine Factors
    combined_factor = (volume_adjusted_breakout_strength + 
                       volume_adjusted_high_low_spread + 
                       volume_weighted_momentum) * price_change_ratio
    
    # Filter for Strong Moves
    strong_moves = (combined_factor > threshold).astype(int)
    
    return strong_moves
```
