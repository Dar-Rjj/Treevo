```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Overnight Return
    df['overnight_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Volume Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Volume Weighted Overnight Return
    df['volume_weighted_overnight_return'] = df['overnight_return'] * df['volume'].shift(1)
    
    # Calculate Intraday Average True Range
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Intraday Relative Strength Index (RSI)
    delta = df['intraday_return']
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Momentum Score
    df['momentum_score'] = df['volume_weighted_intraday_return'] / df['atr_14']
    
    # Reversal Score
    df['reversal_score'] = df['volume_weighted_overnight_return'] / (df['rsi_14'] + 1e-8)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_score'] - df['reversal_score']
    
    return df['alpha_factor']
```
