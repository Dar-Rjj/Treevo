```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    adj_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    high_10 = df['high'].rolling(window=10).max()
    low_10 = df['low'].rolling(window=10).min()
    proximity_high = df['close'] / high_10
    proximity_low = df['close'] / low_10
    min_proximity = pd.DataFrame({'proximity_high': proximity_high, 'proximity_low': proximity_low}).min(axis=1)
    inv_proximity = 1 / min_proximity
    
    # Adjust for Volatility
    std_dev_20 = df['close'].pct_change().rolling(window=20).std()
    adjusted_proximity = inv_proximity / std_dev_20
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_hl_spread = daily_range.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_hl_spread - moving_sum_hl_spread.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    vol_adj_open_close = (df['close'] - df['open']) * df['volume']
    
    # Analyze Price Action and Volume
    price_volatility = df['high'] - df['low']
    relative_strength = (df['close'] - df['open']) / price_volatility
    volume_change = df['volume'] - df['volume'].shift(1)
    vol_price_relationship = (df['volume'] * (df['close'] - df['open']))
    summary_vol_adj_open_close = vol_adj_open_close.rolling(window=5).sum()
    price_gap = df['open'] - df['close'].shift(1)
    positive_money_flow = ((df['close'] > df['open']) * (df['close'] - df['open']) * df['amount']).fillna(0)
    total_money_flow = (df['close'] - df['open']) * df['amount']
    money_flow_ratio = positive_money_flow / total_money_flow
    amount_per_volume = df['amount'] / df['volume']
    momentum_price_volume = (df['close'] - df['open']) * (amount_per_volume / amount_per_volume.rolling(window=20).mean())
    
    # Final Alpha Factor
    final_alpha_factor = (adj_daily_range * 
                          adjusted_proximity * 
                          weighted_avg_volatility * 
                          spread_momentum * 
                          relative_strength * 
                          vol_price_relationship * 
                          summary_vol_adj_open_close * 
                          price_gap * 
                          money_flow_ratio * 
                          amount_per_volume * 
                          momentum_price_volume +
                          vol_adj_open_close)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).quantile(0.8)
    final_alpha_factor = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return final_alpha_factor
```
