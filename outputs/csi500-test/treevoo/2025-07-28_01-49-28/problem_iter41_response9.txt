```python
import pandas as pd

def heuristics_v2(df):
    """
    The function calculates a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor, indexed by date.
    """
    # Calculate the difference between the high and low price
    price_range = df['high'] - df['low']
    
    # Calculate the average volume over the last 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the average amount over the last 10 days
    avg_amount_10d = df['amount'].rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = (df['close'] - df['open']) / (avg_amount_10d / avg_volume_10d)
    
    # Calculate the volatility as the standard deviation of the last 10 closing prices
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the alpha factor
    alpha_factor = (price_range * money_flow_ratio) / (volatility + 1e-6)
    
    return alpha_factor
```
