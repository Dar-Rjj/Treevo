```python
import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Calculate Simple Momentum for Close Prices
    simple_momentum = df['close'] / df['close'].shift(n) - 1
    
    # Adjust Momentum by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_adjusted_momentum = simple_momentum * avg_volume
    
    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Integrate Spread Momentum
    moving_sum_high_low_spreads = high_low_spread.rolling(window=20).sum()
    spread_momentum = moving_sum_high_low_spreads - moving_sum_high_low_spreads.shift(1)
    
    # Open-High-Low Volatility Factor
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff + open_low_diff) * df['volume'] / df['volume'].rolling(window=n).sum()
    
    # Price-Amount Interaction Effect
    amount_per_close = df['amount'].rolling(window=m).sum() / df['close']
    prev_amount_per_close = df['amount'].rolling(window=m).sum().shift(m) / df['close'].shift(m)
    rate_of_change = amount_per_close - prev_amount_per_close
    
    # Combine all the factors into one alpha factor
    alpha_factor = (
        volume_adjusted_momentum 
        + spread_momentum 
        + weighted_avg_volatility
        + rate_of_change
    )
    
    return alpha_factor
```
