import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = daily_range * df['volume'] + open_to_close_movement
    
    # Calculate Moving Sum of High-Low Price Spread (last 20 days)
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low_spread = high_low_spread.rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread.diff(1)
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Volume-Adjusted Momentum
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume
    
    # Combine Indicators with weights
    combined_indicator = (0.5 * volume_adjusted_breakout_strength +
                          0.3 * spread_momentum +
                          0.2 * volume_adjusted_momentum)
    
    # Filter for Strong Moves with a dynamic threshold
    threshold = combined_indicator.rolling(window=20).quantile(0.8)
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return pd.Series(strong_moves, index=df.index, name='alpha_factor')
