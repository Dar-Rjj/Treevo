import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_to_close_movement
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    ma_high_low_spread = high_low_spread.rolling(window=20).mean()
    spread_momentum = ma_high_low_spread - ma_high_low_spread.shift(1)
    
    # Volume-Adjusted Momentum
    price_change = df['close'] - df['close'].shift(1)
    volume_adjusted_momentum = price_change * df['volume']
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_diff = (open_high_diff * df['volume'] + open_low_diff * df['volume'])
    open_high_low_volatility = weighted_diff / df['volume']
    
    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / (min_proximity + 1e-6)
    std_dev_returns = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / (std_dev_returns + 1e-6)
    
    # Combined Indicators
    combined_indicator = (volume_adjusted_breakout_strength * 0.5) + spread_momentum + volume_adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicator * enhanced_price_reversion * open_high_low_volatility
    return final_alpha_factor.rename("Final Alpha Factor")
