```python
import pandas as pd

def heuristics(df):
    # Daily Range
    daily_range = df['high'] - df['low']
    
    # Volume-Adjusted Range
    vol_adj_range = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Extreme Prices
    high_10d = df['high'].rolling(window=10).max()
    low_10d = df['low'].rolling(window=10).min()
    
    # Proximity to 10-Day Range
    close_to_high_10d = df['close'] / high_10d
    close_to_low_10d = df['close'] / low_10d
    
    # Min Proximity
    min_proximity = pd.DataFrame({'close_to_high': close_to_high_10d, 'close_to_low': close_to_low_10d}).min(axis=1)
    
    # Inverted Proximity
    inv_proximity = 1 / min_proximity
    
    # 20-Day Standard Deviation
    std_20d = df['close'].rolling(window=20).std()
    
    # Volatility-Adjusted Reversion
    vol_adj_reversion = inv_proximity / std_20d
    
    # Open-High-Low Volatility
    oh_diff = df['high'] - df['open']
    ol_diff = df['open'] - df['low']
    weighted_avg_vol = (oh_diff + ol_diff) * df['volume']
    
    # Spread Momentum
    high_low_spread_20d = (df['high'] - df['low']).rolling(window=20).sum()
    momentum = high_low_spread_20d.diff()
    
    # 50-Day and 20-Day Moving Averages
    ma_50d = df['close'].rolling(window=50).mean()
    ma_20d = df['close'].rolling(window=20).mean()
    
    # Trend Strength
    trend = (ma_50d - ma_20d) / ma_20d
    
    # Alpha Factor
    alpha_factor = (vol_adj_range * vol_adj_reversion * weighted_avg_vol * momentum * trend)
    
    return alpha_factor
```
