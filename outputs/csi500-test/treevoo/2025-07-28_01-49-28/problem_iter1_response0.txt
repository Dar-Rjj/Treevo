```python
def heuristics_v2(data):
    # Calculate High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * data['volume']
    
    # Calculate Momentum
    momentum = data['close'].diff()
    
    # Combine with Volume-Adjusted High-Low Spread
    factor_value = volume_adjusted_high_low_spread + momentum
    
    return factor_value
```
