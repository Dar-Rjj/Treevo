```python
def heuristics_v2(df):
    import pandas as pd

    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']

    # Calculate Enhanced Open-to-Close Movement
    df['enhanced_oc_movement'] = df['close'] - df['open']
    
    # Adjust for Opening Gap
    df['previous_close'] = df['close'].shift(1)
    df['gap_up'] = (df['open'] > df['previous_close']).astype(int)
    df['gap_down'] = (df['open'] < df['previous_close']).astype(int)
    
    df['enhanced_oc_movement'] += (df['open'] - df['previous_close']) * (df['gap_up'] - df['gap_down'])

    # Adjust for Volume
    df['median_10_day_volume'] = df['volume'].rolling(window=10, min_periods=1).median()
    df['volume_factor'] = df['volume'] / df['median_10_day_volume']

    # Combine Indicators
    df['combined_indicator'] = (df['daily_range'] * df['volume_factor']) + df['enhanced_oc_movement']

    # Filter for Strong Moves
    threshold = df['combined_indicator'].quantile(0.75)  # Example threshold: 75th percentile
    df['alpha_factor'] = (df['combined_indicator'] > threshold).astype(int)

    return df['alpha_factor']
```
