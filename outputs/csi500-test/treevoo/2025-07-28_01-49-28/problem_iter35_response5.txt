```python
def heuristics_v2(df):
    # Calculate the average of high and low for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the ratio of volume to amount, which can be seen as a price-weighted volume
    volume_to_amount_ratio = df['volume'] / df['amount']
    
    # The factor is a combination of these two ratios, capturing the idea that
    # stocks with a higher close relative to the day's average price and higher volume per traded amount
    # may have more momentum.
    factor = close_to_avg_ratio * volume_to_amount_ratio
    
    return factor
```
