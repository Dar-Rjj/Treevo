```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    vol_adj_high_low_spread = high_low_spread * df['volume']
    
    # Moving Average of Volume-Adjusted High-Low Spread (20 days)
    ma_vol_adj_high_low_spread = vol_adj_high_low_spread.rolling(window=20).mean()
    
    # Volume and Momentum Enhanced High-Low Spread
    mom_enh_high_low_spread = vol_adj_high_low_spread - ma_vol_adj_high_low_spread.shift(1)

    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Volume-Adjusted Close-Open Spread
    vol_adj_close_open_spread = close_open_spread * df['volume']
    
    # Moving Sum of Volume-Adjusted Close-Open Spread (10 days)
    sum_vol_adj_close_open_spread = vol_adj_close_open_spread.rolling(window=10).sum()
    
    # Volume, Momentum, and Close-Open Integrated Spread
    int_close_open_spread = vol_adj_close_open_spread - sum_vol_adj_close_open_spread.shift(1)

    # Volume-Weighted Momentum
    momentum = df['close'] - df['close'].shift(1)
    vol_weighted_momentum = momentum * df['volume']

    # Smoothed Momentum (10-day daily returns divided by volume)
    smoothed_momentum = (df['close'] / df['close'].shift(10) - 1).rolling(window=10).sum() / df['volume']

    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']

    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = 1 / pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    
    # Adjust for Volatility (10-day standard deviation of daily returns)
    std_dev_10d = df['close'].pct_change().rolling(window=10).std()
    adjusted_proximity = min_proximity / std_dev_10d
    
    # Enhanced Volatility Adjustment (10-day exponential moving average of daily returns)
    ema_10d = df['close'].pct_change().ewm(span=10, adjust=False).mean()
    enhanced_vol_adj_proximity = min_proximity / ema_10d

    # Combine Factors
    factor = (mom_enh_high_low_spread + vol_weighted_momentum) * price_change_ratio * smoothed_momentum * enhanced_vol_adj_proximity + int_close_open_spread

    return factor
```
