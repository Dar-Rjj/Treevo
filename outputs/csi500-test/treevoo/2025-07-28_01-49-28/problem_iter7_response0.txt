```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_20_days = volume_adjusted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum_20_days - moving_sum_20_days.shift(1)
    
    # Label as Volume and Momentum Enhanced High-Low Spread
    return adjusted_spread_momentum.dropna()
```
