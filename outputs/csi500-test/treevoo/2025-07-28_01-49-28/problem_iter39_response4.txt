def heuristics_v2(df):
    # Calculate the relative strength of the close price compared to the average of open, high, and low
    avg_price = (df['open'] + df['high'] + df['low']) / 3
    rel_strength = (df['close'] - avg_price) / avg_price

    # Calculate the money flow by multiplying the typical price with the volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']

    # Calculate the money flow ratio as the difference between positive and negative money flow
    pos_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    neg_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    money_flow_ratio = pos_money_flow.rolling(window=14).sum() / neg_money_flow.rolling(window=14).sum()

    # Combine the relative strength and money flow ratio to create the alpha factor
    alpha_factor = rel_strength * money_flow_ratio

    return alpha_factor
