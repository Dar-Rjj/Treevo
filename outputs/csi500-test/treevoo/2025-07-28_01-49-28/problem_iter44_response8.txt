import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * np.sqrt(df['volume'])) + open_to_close_movement
    
    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    n = 5  # Number of periods for weighted average volume
    weighted_average_volume = df['volume'].rolling(window=n).mean()
    volume_weighted_average_spread = spread * weighted_average_volume
    
    # Enhanced Momentum Factor
    simple_momentum = df['close'] - df['close'].shift(n)
    smoothed_volume = df['volume'].ewm(span=n, adjust=False).mean()
    adjusted_momentum = simple_momentum * np.log(smoothed_volume + 1)
    
    # High-Low Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combined Factors
    combined_factors = adjusted_momentum + high_low_return
    
    # Open-High-Low Volatility
    open_high_price_diff = df['high'] - df['open']
    open_low_price_diff = df['open'] - df['low']
    max_volatility = pd.concat([open_high_price_diff, open_low_price_diff], axis=1).max(axis=1)
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = max_volatility * abs(volume_percentage_change)
    
    # Amount per Unit Close Price Rate of Change
    n = 10  # Number of periods for sum of amounts
    sum_of_amounts = df['amount'].rolling(window=n).sum()
    current_ratio = sum_of_amounts / df['close']
    previous_sum_of_amounts = df['amount'].rolling(window=n*2).sum().shift(n)
    previous_ratio = previous_sum_of_amounts / df['close'].shift(n)
    differential = (current_ratio - previous_ratio) / previous_ratio
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (
        volume_adjusted_breakout_strength + 
        volume_weighted_average_spread + 
        adjusted_momentum + 
        combined_factors + 
        adjusted_volatility + 
        differential
    )
    
    # Price Trend Indicator
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_trend_score = (long_term_momentum - short_term_momentum) * integrated_alpha_factor
    
    # Price Oscillation
    short_term_high = df['high'].rolling(window=10).max()
    short_term_low = df['low'].rolling(window=10).min()
    price_oscillation = (short_term_high - short_term_low) / df['close'].shift(1)
    
    # Enhanced Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = df['close'] / ten_day_high
    proximity_to_ten_day_low = df['close'] / ten_day_low
    min_proximity = pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    inverted_min_proximity = 1 / min_proximity
    twenty_day_std = df['close'].rolling(window=20).std()
    enhanced_reversion = inverted_min_proximity / twenty_day_std
    
    # Combined Momentum
    ten_day_return = (df['close'] / df['close'].shift(10)) - 1
    thirty_day_return = (df['close'] / df['close'].shift(30)) - 1
    combined_momentum = ten_day_return + thirty_day_return
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Final Alpha Factor
    final_alpha_factor = (
        integrated_alpha_factor + 
        price_trend_score + 
        price_oscillation + 
        enhanced_reversion + 
        combined_momentum + 
        volume_adjusted_momentum
    )
    
    return final_alpha_factor.rename
