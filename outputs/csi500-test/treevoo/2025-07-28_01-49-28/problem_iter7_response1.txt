import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Return
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Incorporate Volume
    volume_adjusted_spread = high_low_range * df['volume']
    
    # Adjust for Volume Trend
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_signal = volume_change * high_low_return
    
    # Integrate Signals over Recent Days (e.g., 5 days)
    integrated_signal = adjusted_signal.rolling(window=5).sum()
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Combine Integrated Signal with Momentum
    final_signal = integrated_signal + momentum
    final_signal.name = 'Volume-Adjusted High-Low Spread and Momentum'
    
    return final_signal
