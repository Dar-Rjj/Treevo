```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate 21-day Price Momentum
    df['momentum_21'] = df['close'] - df['close'].shift(21)
    
    # Calculate 63-day Price Momentum
    df['momentum_63'] = df['close'] - df['close'].shift(63)
    
    # Combine 21-day and 63-day Momentums
    df['combined_momentum'] = 0.5 * df['momentum_21'] + 0.5 * df['momentum_63']
    
    # Compute 21-day Average Volume
    df['avg_volume_21'] = df['volume'].rolling(window=21).mean()
    
    # Compute 63-day Average Volume
    df['avg_volume_63'] = df['volume'].rolling(window=63).mean()
    
    # Difference in 21-day and 63-day Average Volumes
    df['volume_diff'] = df['avg_volume_21'] - df['avg_volume_63']
    
    # Normalize Volume Reversal
    df['volume_reversal'] = df['volume_diff'] / df['avg_volume_63']
    
    # Combine Price Momentum and Volume Reversal
    df['alpha_factor'] = df['combined_momentum'] * df['volume_reversal']
    
    return df['alpha_factor']
```
