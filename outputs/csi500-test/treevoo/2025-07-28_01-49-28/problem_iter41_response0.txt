```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adj_daily_range = (df['high'] - df['low']) * df['volume']

    # Price Reversion
    highest_high_10 = df['high'].rolling(window=10).max()
    lowest_low_10 = df['low'].rolling(window=10).min()
    proximity_to_high = (highest_high_10 - df['close']) / (highest_high_10 - lowest_low_10)
    proximity_to_low = (df['close'] - lowest_low_10) / (highest_high_10 - lowest_low_10)
    inverted_proximity = 1 / (proximity_to_high + proximity_to_low)
    price_reversion = inverted_proximity / df['close'].pct_change().rolling(window=20).std()

    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    ohl_volatility = (open_high_diff + open_low_diff) / 2 * df['volume']

    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    spread_momentum = high_low_spread.rolling(window=5).sum() - high_low_spread.shift(1).rolling(window=5).sum()

    # Volume-Adjusted Open-to-Close Movement
    vol_adj_open_close = (df['close'] - df['open']) * df['volume']

    # Simple Momentum
    simple_momentum = df['close'] - df['close'].shift(1)

    # Momentum-Based Alpha Factors
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    price_change_ratio_5 = (df['close'] / df['close'].shift(5)) - 1
    price_change_ratio_20 = (df['close'] / df['close'].shift(20)) - 1

    # Volatility-Based Alpha Factors
    daily_return_std_10 = df['close'].pct_change().rolling(window=10).std()
    daily_return_std_30 = df['close'].pct_change().rolling(window=30).std()
    range_volatility_10 = (df['high'] - df['low']).rolling(window=10).mean()
    range_volatility_30 = (df['high'] - df['low']).rolling(window=30).mean()

    # Volume-Based Alpha Factors
    volume_trend_10 = (df['volume'] / df['volume'].shift(10)) - 1
    volume_trend_30 = (df['volume'] / df['volume'].shift(30)) - 1
    volume_price_interaction_10 = (df['close'] * df['volume']).rolling(window=10).mean()
    volume_price_interaction_30 = (df['close'] * df['volume']).rolling(window=30).mean()

    # Pattern-Based Alpha Factors
    doji_pattern = (df['open'] == df['close']) & (df['high'] - df['low'] > 0.001 * df['open'])
    bullish_engulfing_pattern = (df['open'] < df['close']) & (df['open'].shift(1) > df['close'].shift(1)) & \
                               (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))

    # Final Alpha Factor
    final_alpha_factor = (adj_daily_range * price_reversion * ohl_volatility * spread_momentum +
                          vol_adj_open_close + simple_momentum +
                          sma_10 + sma_30 + price_change_ratio_5 + price_change_ratio_20 +
                          daily_return_std_10 + daily_return_std_30 + range_volatility_10 + range_volatility_30 +
                          volume_trend_10 + volume_trend_30 + volume_price_interaction_10 + volume_price_interaction_30 +
                          doji_pattern + bullish_engulfing_pattern)

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).quantile(0.75)
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)

    return strong_moves
```
