```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Dynamic Price Reversion
    rolling_high = df['high'].rolling(window=10).max()
    rolling_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] - rolling_low
    proximity_to_low = rolling_high - df['close']
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    std_dev_returns = df['close'].pct_change().rolling(window=20).std()
    dynamic_price_reversion = min_proximity / std_dev_returns
    
    # Weighted Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff + open_low_diff) * df['volume'].apply(lambda x: x ** 0.5)
    
    # Sum of Exponential Weights on High-Low Spreads
    ewma_spread = (df['high'] - df['low']).ewm(span=20).mean()
    
    # Spread Momentum
    spread_momentum = ewma_spread - ewma_spread.shift(1)
    
    # Volume-Adjusted Intraday Movement
    volume_adjusted_intraday = (df['close'] - df['open']) * df['volume'].apply(lambda x: x ** 0.5)
    
    # Simple Momentum
    simple_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * dynamic_price_reversion * weighted_volatility * spread_momentum +
                          volume_adjusted_intraday + simple_momentum)
    
    # Enhanced Volume-Adjusted Breakout Strength
    high_close_adjusted = (df['high'] - df['close']) * df['volume']
    close_low_adjusted = (df['close'] - df['low']) * df['volume']
    breakout_strength = high_close_adjusted - close_low_adjusted
    
    # Integrated Volume-Adjusted Breakout Indicator
    positive_breakout = breakout_strength.apply(lambda x: max(x, 0))
    negative_breakout = breakout_strength.apply(lambda x: min(x, 0))
    
    # Combined Momentum Indicator
    momentum_5_days = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    momentum_20_days = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    combined_momentum = momentum_5_days + momentum_20_days
    
    # Final Enhanced Alpha Factor
    enhanced_alpha_factor = (final_alpha_factor * positive_breakout + final_alpha_factor * negative_breakout + 
                             combined_momentum)
    threshold = enhanced_alpha_factor.rolling(window=10).mean() + 2 * enhanced_alpha_factor.rolling(window=10).std()
    strong_moves = (enhanced_alpha_factor > threshold).astype(int)
    
    return strong_moves
```
