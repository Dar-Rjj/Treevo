```python
import pandas as pd

def heuristics_v2(df):
    """
    Compute a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series of alpha factor values indexed by date.
    """
    # Calculate the average price for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the money flow ratio
    money_flow_ratio = (df['close'] - df['open']) / (avg_price * df['volume'])
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    
    # Combine the factors to form the final alpha factor
    alpha_factor = (money_flow_ratio * price_change * volume_change).fillna(0)
    
    return alpha_factor
```
