```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']

    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    inverted_proximity = 1 / (proximity_to_high + proximity_to_low)

    # Adjust for Volatility
    returns = df['close'].pct_change()
    twenty_day_std = returns.rolling(window=20).std()
    volatility_adjusted_proximity = inverted_proximity / twenty_day_std

    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg_volatility = (open_high_diff + open_low_diff) * df['volume']

    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_spreads = (df['high'] - df['low']).rolling(window=20).sum()

    # Calculate Spread Momentum
    spread_momentum = moving_sum_spreads.diff()

    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_oc_movement = (df['close'] - df['open']) * df['volume']

    # Analyze Price Action and Volume
    price_action_volatility = df['high'] - df['low']
    price_strength = (df['close'] - df['open']) / price_action_volatility
    volume_change = df['volume'].diff()
    price_volume_relationship = (df['volume'] * (df['close'] - df['open']))
    five_day_summary = (df['close'] - df['open']) * df['volume'].rolling(window=5).sum()
    price_gap = df['open'] - df['close'].shift(1)
    positive_money_flow = ((df['close'] - df['open']) * df['volume']).where(df['close'] > df['open'], 0)
    total_money_flow = (df['close'] - df['open']) * df['volume']
    money_flow_ratio = positive_money_flow / total_money_flow
    amount_per_volume = df['amount'] / df['volume']
    momentum_factor = (df['close'] - df['open']) * (df['volume'] / df['volume'].rolling(window=20).mean())

    # Final Alpha Factor
    final_alpha_factor = (
        volume_adjusted_range * 
        volatility_adjusted_proximity * 
        weighted_avg_volatility * 
        spread_momentum * 
        price_strength * 
        volume_change * 
        five_day_summary * 
        price_gap * 
        money_flow_ratio * 
        amount_per_volume * 
        momentum_factor * 
        volume_adjusted_oc_movement
    )

    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + final_alpha_factor.rolling(window=20).std() * 2
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)

    return strong_moves
```
