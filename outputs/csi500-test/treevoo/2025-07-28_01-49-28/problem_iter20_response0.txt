```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price over a 10-day window
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()

    # Calculate the difference between the high and low prices, which can be seen as a measure of volatility
    df['high_low_diff'] = df['high'] - df['low']

    # Calculate the ratio of the closing price to the opening price, which captures the price change within a day
    df['close_open_ratio'] = df['close'] / df['open']

    # Calculate the weighted volume by amount to capture the liquidity and buying power
    df['volume_weighted_by_amount'] = df['volume'] * df['amount']

    # Alpha factor based on the relationship between the above calculated features:
    # It assumes that if the stock has a higher than usual intra-day volatility (high_low_diff), 
    # a significant positive daily return (close_open_ratio > 1.05) and an increasing trend in closing price (close > close_10d_mean),
    # then it is likely to attract more buyers, especially if there's also a high trading volume indicating strong liquidity.
    df['alpha_factor'] = ((df['high_low_diff'] > df['high_low_diff'].mean()) & 
                          (df['close_open_ratio'] > 1.05) &
                          (df['close'] > df['close_10d_mean']) *
                          (df['volume_weighted_by_amount']))

    return df['alpha_factor']
```
