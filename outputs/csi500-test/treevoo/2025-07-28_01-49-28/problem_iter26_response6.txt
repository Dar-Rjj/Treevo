import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range * df['volume']) + open_close_movement

    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_average_20 = high_low_spread.rolling(window=20).mean()
    spread_momentum = moving_average_20.diff()

    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['close'].diff() * df['volume'])

    # Enhanced Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    volatility_calculation = pd.concat([open_high_diff, open_low_diff], axis=1).max(axis=1)
    
    volume_percentage_change = df['volume'].pct_change().fillna(0)  # Avoid division by zero
    adjusted_volatility = volatility_calculation * volume_percentage_change

    # Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    std_deviation = df['close'].pct_change().rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / std_deviation

    # Combined Indicators
    combined_indicators = (0.5 * volume_adjusted_breakout_strength + 
                           0.3 * spread_momentum + 
                           0.2 * volume_adjusted_momentum)

    # Final Alpha Factor
    final_alpha_factor = combined_indicators * enhanced_price_reversion * adjusted_volatility
    
    return final_alpha_factor.rename('Final Alpha Factor')
