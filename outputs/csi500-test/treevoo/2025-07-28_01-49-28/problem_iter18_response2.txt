```python
import pandas as pd

def heuristics_v2(df, threshold=0.5):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_adjusted_range = daily_range * (df['volume'] ** 0.5)
    
    # Calculate 5-day Momentum
    moving_avg_5 = df['close'].rolling(window=5).mean()
    momentum = (df['close'] > moving_avg_5).astype(int) * 2 - 1  # 1 if current close > 5-day MA, -1 otherwise
    
    # Calculate 5-day Volatility
    std_dev_5 = df['close'].rolling(window=5).std()
    volatility_scaled = std_dev_5 * (df['volume'] ** 0.5)
    
    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close_movement + volatility_scaled
    combined_indicator = combined_indicator * momentum
    
    # Filter for Strong Moves
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
```
