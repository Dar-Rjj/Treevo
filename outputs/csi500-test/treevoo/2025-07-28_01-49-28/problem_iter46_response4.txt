```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term moving average (SMA) of closing prices over last 5 days
    sma = df['close'].rolling(window=5).mean()
    
    # Calculate long-term moving average (LMA) of closing prices over last 20 days
    lma = df['close'].rolling(window=20).mean()
    
    # Compute the difference between LMA and SMA to identify trend
    trend_factor = lma - sma
    
    return trend_factor
```
