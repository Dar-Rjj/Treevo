import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted High-Low Spread
    vol_adj_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Moving Average of Volume-Adjusted High-Low Spread
    ma_vol_adj_high_low_spread = vol_adj_high_low_spread.rolling(window=20).mean()
    
    # Volume and Momentum Enhanced High-Low Spread
    vol_mom_enhanced_high_low_spread = vol_adj_high_low_spread - ma_vol_adj_high_low_spread.shift(1)
    
    # Volume-Adjusted Close-Open Spread
    vol_adj_close_open_spread = (df['close'] - df['open']) * df['volume']
    
    # Moving Sum of Volume-Adjusted Close-Open Spread
    sum_vol_adj_close_open_spread = vol_adj_close_open_spread.rolling(window=10).sum()
    
    # Volume, Momentum, and Close-Open Integrated Spread
    vol_mom_close_open_integrated_spread = vol_adj_close_open_spread - sum_vol_adj_close_open_spread.shift(1)
    
    # Volume-Weighted Momentum
    vol_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Smoothed Momentum
    smoothed_momentum = (df['close'].pct_change().rolling(window=10).sum()) / df['volume']
    
    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = proximity_to_high.combine(proximity_to_low, min)
    inv_min_proximity = 1 / min_proximity
    ten_day_std = df['close'].pct_change().rolling(window=10).std()
    enhanced_volatility_adjusted_proximity = inv_min_proximity / ten_day_std
    
    # Combine Factors
    combined_factors = (vol_mom_enhanced_high_low_spread + vol_weighted_momentum) * price_change_ratio * smoothed_momentum * enhanced_volatility_adjusted_proximity + vol_mom_close_open_integrated_spread
    
    # Volume-Adjusted 5-Day High-Low Spread
    vol_adj_5day_high_low_spread = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()) * df['volume']
    
    # Moving Average of Volume-Adjusted 5-Day High-Low Spread
    ma_vol_adj_5day_high_low_spread = vol_adj_5day_high_low_spread.rolling(window=10).mean()
    
    # Volume and Momentum Enhanced 5-Day High-Low Spread
    vol_mom_enhanced_5day_high_low_spread = vol_adj_5day_high_low_spread - ma_vol_adj_5day_high_low_spread.shift(1)
    
    # Volume-Weighted 5-Day Momentum
    vol_weighted_5day_momentum = (df['close'] - df['close'].shift(5)) * df['volume']
    
    # Smoothed 5-Day Momentum
    smoothed_5day_momentum = (df['close'].pct_change().rolling(window=5).sum()) / df['volume']
    
    # Integrate 5-Day Momentum into Combined Factors
    integrated_5day_momentum_factors = (vol_mom_enhanced_5day_high_low_spread + vol_weighted_5day_momentum) * price_change_ratio * smoothed_5day_momentum * enhanced_volatility_adjusted_proximity
    combined_factors += integrated_5day_momentum_factors
    
    # 10-Day Simple Moving Average
    ten_day_sma = df['close'].rolling(window=10).mean()
    
    # 10-Day Exponential Moving Average
    ten_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Trend Following Indicator
    trend_following_indicator = (ten_day_sma - ten_day_ema) / ten_day_ema
    
    # Incorporate Trend Following Indicator into Combined Factors
    final_factor = combined_factors * trend_following_indicator
    
    return final_factor
