```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adj_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    close_to_high = df['close'] / ten_day_high
    close_to_low = df['close'] / ten_day_low
    min_proximity = pd.concat([close_to_high, close_to_low], axis=1).min(axis=1)
    price_reversion = (1 - min_proximity) / df['close'].pct_change().rolling(window=20).std()
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff + open_low_diff) * df['volume'] / 2
    
    # Moving Sum of High-Low Spreads
    moving_sum_spreads = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Spread Momentum
    spread_momentum = moving_sum_spreads - moving_sum_spreads.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    vol_adj_open_close = (df['close'] - df['open']) * df['volume']
    
    # Simple Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (adj_daily_range * price_reversion * weighted_avg * spread_momentum
                          + vol_adj_open_close + simple_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=20).mean() + 2 * final_alpha_factor.rolling(window=20).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
