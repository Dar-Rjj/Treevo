```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    volume_adjusted_breakout_strength = (daily_range + open_to_close_movement) * df['volume']

    # High-Low Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_20 = high_low_spread.rolling(window=20).sum()
    high_low_spread_momentum = moving_sum_20 - moving_sum_20.shift(1)

    # Enhance with 10-Day High and Low Prices
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_ten_day_high = (df['close'] / ten_day_high) - 1
    proximity_to_ten_day_low = 1 - (df['close'] / ten_day_low)
    min_proximity_inverted = 1 / pd.concat([proximity_to_ten_day_high, proximity_to_ten_day_low], axis=1).min(axis=1)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    enhanced_factor = min_proximity_inverted / twenty_day_volatility

    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff + open_low_diff) / 2
    weighted_average_by_volume = open_high_low_volatility * df['volume']
    integrated_volatility = weighted_average_by_volume.rolling(window=20).mean()

    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    n = 5
    amount_per_unit_close_current = (df['amount'] / df['close']).rolling(window=n).sum()
    amount_per_unit_close_previous = (df['amount'] / df['close']).shift(n).rolling(window=n).sum()
    amount_rate_of_change = amount_per_unit_close_current - amount_per_unit_close_previous

    # Volume-Weighted Moving Average Crossover
    short_term_vwma_5 = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    long_term_vwma_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwma_crossover = short_term_vwma_5 - long_term_vwma_20

    # Volume-Weighted Price Momentum
    vwp_5 = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    vwp_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwp_momentum = vwp_5 - vwp_20

    # Final Alpha Factor
    combined_indicator = (
        volume_adjusted_breakout_strength +
        high_low_spread_momentum +
        enhanced_factor +
        integrated_volatility
    )
    strong_move_indicator = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    final_alpha_factor = (
        combined_indicator *
        strong_move_indicator *
        amount_rate_of_change *
        vwma_crossover *
        vwp_momentum
    )

    return final_alpha_factor
```
