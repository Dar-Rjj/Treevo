```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    inverted_proximity = 1 / (proximity_to_high + proximity_to_low)
    twenty_day_std = df['close'].pct_change().rolling(window=20).std()
    price_reversion = inverted_proximity / twenty_day_std
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_high_low_spread = high_low_spread.rolling(window=5).sum()
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_movement = (df['close'] - df['open']) * df['volume']
    
    # Simple Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * price_reversion * weighted_open_high_low_volatility * spread_momentum
                          + volume_adjusted_movement + simple_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.rolling(window=10).mean() + 2 * final_alpha_factor.rolling(window=10).std()
    strong_moves = (final_alpha_factor > dynamic_threshold).astype(int)
    
    return strong_moves
```
