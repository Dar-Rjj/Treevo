def heuristics_v2(df):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['High-Low'] = df['high'] - df['low']
    df['Close-PrevClose'] = (df['close'] - df['Previous Close']).abs()
    df['High-PrevClose'] = (df['high'] - df['Previous Close']).abs()
    df['PrevClose-Low'] = (df['Previous Close'] - df['low']).abs()

    df['True Range'] = df[['High-Low', 'Close-PrevClose']].max(axis=1)
    high_greater_than_prev_close = df['high'] > df['Previous Close']
    df.loc[high_greater_than_prev_close, 'True Range'] = df.loc[high_greater_than_prev_close, 'High-PrevClose']
    df.loc[~high_greater_than_prev_close, 'True Range'] = df.loc[~high_greater_than_prev_close, 'PrevClose-Low']

    # Moving Average of True Range over a given period (e.g., 14 days)
    df['ATR Momentum'] = df['True Range'].rolling(window=14).mean()

    return df['ATR Momentum']
