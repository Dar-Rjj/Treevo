```python
import pandas as pd

def heuristics_v2(df, n=5, breakout_weight=0.6, momentum_weight=0.4, high_low_weight=0.3, threshold=0.1):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_to_close = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_adjusted_breakout_strength = daily_range * volume_factor + open_to_close
    
    # Calculate Simple Momentum for Close Prices
    simple_momentum = (df['close'] / df['close'].shift(n)) - 1
    
    # Adjust Momentum by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Return
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Combine Indicators
    combined_indicator = (breakout_weight * volume_adjusted_breakout_strength) + (momentum_weight * adjusted_momentum)
    combined_indicator += (high_low_weight * high_low_return)
    
    # Filter for Strong Moves
    strong_moves = combined_indicator.apply(lambda x: 1 if x > threshold else 0)
    
    return strong_moves
```
