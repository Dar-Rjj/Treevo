def heuristics_v2(data):
    # Calculate Daily Range
    data['daily_range'] = data['high'] - data['low']
    
    # Calculate Open-to-Close Movement
    data['open_close_movement'] = data['close'] - data['open']
    
    # Adjust for Volume
    data['volume_adjusted_breakout_strength'] = (data['daily_range'] * data['volume']) + data['open_close_movement']
    
    # Calculate Moving Average of High-Low Price Spread
    data['high_low_spread'] = data['high'] - data['low']
    data['high_low_spread_ma'] = data['high_low_spread'].rolling(window=20).mean()
    
    # Calculate Spread Momentum
    data['spread_momentum'] = data['high_low_spread_ma'] - data['high_low_spread_ma'].shift(1)
    
    # Incorporate Volume into High-Low Spread
    data['volume_adjusted_high_low_spread'] = data['high_low_spread'] * data['volume']
    
    # Calculate Volume-Adjusted Momentum
    data['volume_adjusted_momentum'] = (data['close'] - data['close'].shift(1)) * data['volume']
    
    # Combine Indicators
    weight = 0.5  # Example weight, can be adjusted
    data['combined_indicator'] = (weight * data['volume_adjusted_breakout_strength'] 
                                  + data['spread_momentum'] 
                                  + data['volume_adjusted_momentum'])
    
    # Filter for Strong Moves
    threshold = 1.0  # Example threshold, can be adjusted
    data['alpha_factor'] = (data['combined_indicator'] > threshold).astype(int)
    
    return data['alpha_factor']
