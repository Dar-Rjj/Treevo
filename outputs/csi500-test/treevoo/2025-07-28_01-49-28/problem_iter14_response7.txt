```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Momentum with Open-to-Close Movement
    n = 20  # Number of days to look back for momentum and volume
    df['Simple_Momentum'] = df['close'].pct_change(n)
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Factor'] = df['volume'] / df['Average_Volume']
    df['Adjusted_Momentum'] = df['Simple_Momentum'] * df['Volume_Factor']
    df['Open_to_Close_Movement'] = (df['close'] - df['open']) / df['open']
    df['Combined_Indicator'] = 0.7 * df['Adjusted_Momentum'] + 0.3 * df['Open_to_Close_Movement']
    df['Strong_Move'] = df['Combined_Indicator'].apply(lambda x: 1 if x > 0.05 else 0)

    # High-Low Return with Spread Momentum
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Return'] = df['High_Low_Range'] / df['close'].shift(1)
    df['Daily_High_Low_Spread'] = df['high'] - df['low']
    df['Moving_Sum_High_Low'] = df['Daily_High_Low_Spread'].rolling(window=20).sum()
    df['Spread_Momentum'] = df['Moving_Sum_High_Low'].diff()

    # Combined Momentum, High-Low Return, and Strong Move Indicator
    df['Integrated_Momentum'] = df['Adjusted_Momentum'] + df['High_Low_Return'] + df['Strong_Move']

    # Open-High-Low Volatility with Volume-Weighted Adjustments
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Volatility_Index'] = (df['Open_High_Diff'] + df['Open_Low_Diff']) / 2
    df['Volume_Change_Pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Volatility'] = df['Volatility_Index'] * (1 + df['Volume_Change_Pct'])
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    df['Moving_Avg_Volume_Weighted_Close'] = df['Volume_Weighted_Close'].rolling(window=20).mean()
    df['Volume_Weighted_Momentum'] = df['Volume_Weighted_Close'] - df['Moving_Avg_Volume_Weighted_Close']

    # Amount per Unit Close Price Rate of Change with Volume-Weighted Adjustments
    n_amount = 10  # Number of days to sum amount
    df['Amount_Sum'] = df['amount'].rolling(window=n_amount).sum()
    df['Prev_Amount_Sum'] = df['amount'].shift(n_amount).rolling(window=n_amount).sum()
    df['Ratio_Current'] = df['Amount_Sum'] / df['close']
    df['Ratio_Previous'] = df['Prev_Amount_Sum'] / df['close'].shift(n_amount)
    df['Differential_Ratio'] = df['Ratio_Current'] - df['Ratio_Previous']

    # Final Integrated Alpha
    df['Alpha_Factor'] = df['Integrated_Momentum'] + df['Adjusted_Volatility'] + df['Volume_Weighted_Momentum'] + df['Differential_Ratio']

    return df['Alpha_Factor']
```
