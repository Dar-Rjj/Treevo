```python
def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the average of the high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume'])
    
    # Calculate the difference between the VWAP and the average price
    df['price_diff'] = df['vwap'] - df['avg_price']
    
    # Calculate the factor as a combination of the price change and the price difference
    alpha_factor = df['close_change'] * df['price_diff']
    
    return alpha_factor
```
