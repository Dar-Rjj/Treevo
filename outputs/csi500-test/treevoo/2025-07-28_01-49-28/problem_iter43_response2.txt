```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Daily Range
    adjusted_daily_range = (df['high'] - df['low']) * df['volume']
    
    # Enhanced Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (df['close'] - ten_day_high) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    min_proximity = -pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    std_20 = df['close'].rolling(window=20).std()
    enhanced_reversion = min_proximity / std_20
    
    # Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_avg = (open_high_diff + open_low_diff) * df['volume']
    
    # Spread Momentum
    high_low_spread = df['high'] - df['low']
    moving_sum_30 = high_low_spread.rolling(window=30).sum()
    spread_momentum = moving_sum_30 - moving_sum_30.shift(1)
    
    # Trend Strength
    ma_50 = df['close'].rolling(window=50).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_strength = (ma_50 - ma_20) / ma_20
    
    # Combined Momentum
    ten_day_return = (df['close'] / df['close'].shift(10)) - 1
    thirty_day_return = (df['close'] / df['close'].shift(30)) - 1
    combined_momentum = ten_day_return + thirty_day_return
    
    # Volume-Adjusted Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    avg_volume = df['volume'].rolling(window=2).mean()
    volume_adjusted_momentum = simple_momentum * (df['volume'] / avg_volume)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_daily_range * enhanced_reversion * weighted_avg * 
                          spread_momentum * trend_strength * combined_momentum * 
                          volume_adjusted_momentum)
    
    return final_alpha_factor
```
