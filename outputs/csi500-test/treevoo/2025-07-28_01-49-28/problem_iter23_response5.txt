import pandas as pd

def heuristics_v2(df):
    # Enhanced Volume-Adjusted Breakout Strength
    daily_range = df['high'] - df['low']
    open_to_close_movement = df['close'] - df['open']
    enhanced_volume_adjusted_breakout_strength = daily_range * df['volume'] + open_to_close_movement

    # Volume-Weighted Average Spread
    spread = df['high'] - df['low']
    weighted_average_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_average_spread = spread * weighted_average_volume

    # Enhanced Integrated Momentum Factor
    simple_momentum = df['close'] / df['close'].shift(5) - 1
    adjusted_momentum_by_volume = simple_momentum * weighted_average_volume

    # High-Low Return
    high_low_range = df['high'] - df['low']
    high_low_return = high_low_range / df['close'].shift(1)

    # Combined Enhanced Momentum and High-Low Return
    combined_enhanced_momentum_and_high_low_return = adjusted_momentum_by_volume + high_low_return

    # Enhanced Open-High-Low Volatility
    open_high_price_diff = df['high'] - df['open']
    open_low_price_diff = df['open'] - df['low']
    volatility_calculation = pd.concat([open_high_price_diff, open_low_price_diff], axis=1).max(axis=1)
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_volatility = volatility_calculation * volume_percentage_change

    # Amount per Unit Close Price Rate of Change
    amount_sum = df['amount'].rolling(window=5).sum()
    current_ratio = amount_sum / df['close']
    previous_ratio = df['amount'].rolling(window=5).sum().shift(5) / df['close'].shift(5)
    amount_per_unit_close_price_roc = current_ratio - previous_ratio

    # Price Change Ratio
    price_change_ratio = df['close'] / df['open']

    # Combine Enhanced Factors
    integrated_enhanced_alpha_factor = (
        enhanced_volume_adjusted_breakout_strength +
        volume_weighted_average_spread +
        combined_enhanced_momentum_and_high_low_return +
        adjusted_volatility +
        amount_per_unit_close_price_roc
    ) * price_change_ratio

    # Filter for Strong Moves
    threshold = integrated_enhanced_alpha_factor.mean() + 2 * integrated_enhanced_alpha_factor.std()
    strong_moves = (integrated_enhanced_alpha_factor > threshold).astype(int)

    return strong_moves
