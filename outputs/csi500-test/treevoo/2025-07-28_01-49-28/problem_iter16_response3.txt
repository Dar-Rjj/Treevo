```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Compute the difference between short-term and long-term SMAs
    df['SMA_diff'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate daily, weekly, and monthly return series
    df['daily_return'] = df['close'].pct_change()
    df['weekly_return'] = df['close'].pct_change(periods=5)
    df['monthly_return'] = df['close'].pct_change(periods=20)
    
    # Generate a weighted sum of returns over different periods
    df['weighted_returns'] = 0.1 * df['daily_return'] + 0.3 * df['weekly_return'] + 0.6 * df['monthly_return']
    
    # Measure the gap up or down (open - close of the previous day) as an indicator of overnight sentiment
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    df['OBV'] = obv.cumsum()
    
    # Combine OBV with closing price movement to gauge market interest
    df['OBV_close_movement'] = df['OBV'] * df['close'].pct_change()
    
    # Create the alpha factor by combining all the sub-factors
    df['alpha_factor'] = df['SMA_diff'] + df['weighted_returns'] + df['gap'] + df['OBV_close_movement']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
