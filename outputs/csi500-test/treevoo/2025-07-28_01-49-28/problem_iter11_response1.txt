```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low = volume_adjusted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Adjusted Spread Momentum
    momentum_high_low = moving_sum_high_low - moving_sum_high_low.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted Close-Open Spread
    moving_avg_close_open = volume_adjusted_close_open_spread.rolling(window=10).mean()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    momentum_close_open = moving_avg_close_open - moving_avg_close_open.shift(1)
    
    # Calculate Weighted Momentum
    weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Smoothed Momentum
    smoothed_momentum = (df['close'] - df['close'].shift(20)).sum() / df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    
    # Current Close to 10-Day High and Low
    proximity_high = df['close'] - ten_day_high
    proximity_low = df['close'] - ten_day_low
    
    # Minimum Proximity
    min_proximity = pd.concat([proximity_high, proximity_low], axis=1).min(axis=1)
    
    # Adjust for Volatility
    volatility = df['close'].pct_change().rolling(window=20).std()
    adjusted_proximity = -min_proximity / volatility
    
    # Combine Factors
    factor = (momentum_high_low + weighted_momentum) * price_change_ratio * smoothed_momentum * adjusted_proximity
    
    return factor
```
