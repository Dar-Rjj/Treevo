```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted High-Low Spread
    moving_avg_high_low_spread = volume_adjusted_high_low_spread.rolling(window=20).mean()
    
    # Calculate Adjusted Spread Momentum
    momentum_high_low_spread = volume_adjusted_high_low_spread - moving_avg_high_low_spread.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted Close-Open Spread
    moving_sum_close_open_spread = volume_adjusted_close_open_spread.rolling(window=10).sum()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    integrated_momentum_close_open = volume_adjusted_close_open_spread - moving_sum_close_open_spread.shift(1)
    
    # Calculate Weighted Momentum
    weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Smoothed Momentum
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    smoothed_momentum = daily_returns.rolling(window=10).sum() / df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = ten_day_high - df['close']
    proximity_to_low = df['close'] - ten_day_low
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / (min_proximity + 1e-6)  # Add a small epsilon to avoid division by zero
    
    # Adjust for Volatility
    daily_returns_std = daily_returns.rolling(window=10).std()
    volatility_adjusted_proximity = inverted_proximity / daily_returns_std
    
    # Enhanced Volatility Adjustment
    exp_moving_avg_daily_returns = daily_returns.ewm(span=10, adjust=False).mean()
    enhanced_volatility_adjusted_proximity = inverted_proximity / exp_moving_avg_daily_returns
    
    # Combine Factors
    combined_factor = (momentum_high_low_spread + weighted_momentum) * price_change_ratio * smoothed_momentum * enhanced_volatility_adjusted_proximity + integrated_momentum_close_open
    
    return combined_factor
```
