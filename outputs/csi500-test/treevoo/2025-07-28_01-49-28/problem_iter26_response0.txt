```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_daily_range = daily_range * df['volume']
    
    # Incorporate Enhanced Price Reversion
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = (ten_day_high - df['close']) / (ten_day_high - ten_day_low)
    proximity_to_low = (df['close'] - ten_day_low) / (ten_day_high - ten_day_low)
    min_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / min_proximity
    
    # Adjust for Volatility
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    volatility_adjusted_reversion = inverted_proximity / std_dev_20
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_open_high_low = (open_high_diff + open_low_diff) * df['volume']
    
    # Calculate Moving Sum of High-Low Price Spreads
    moving_sum_high_low_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)
    
    # Volume-Adjusted Open-to-Close Movement
    volume_adjusted_open_close = (df['close'] - df['open']) * df['volume']
    
    # Volume-Adjusted Breakout Strength
    volume_adjusted_breakout_strength = volume_adjusted_daily_range + volume_adjusted_open_close
    
    # Simple Momentum for Close Prices
    simple_momentum = df['close'] - df['close'].shift(1)
    
    # Volume-Adjusted Momentum
    average_volume = (df['volume'] + df['volume'].shift(1)) / 2
    volume_adjusted_momentum = simple_momentum * average_volume
    
    # Final Alpha Factor
    final_alpha_factor = (volume_adjusted_daily_range * 
                          volatility_adjusted_reversion * 
                          weighted_open_high_low * 
                          spread_momentum + 
                          volume_adjusted_breakout_strength + 
                          volume_adjusted_momentum)
    
    # Filter for Strong Moves
    dynamic_threshold = final_alpha_factor.quantile(0.75)
    strong_moves = final_alpha_factor > dynamic_threshold
    final_alpha_factor_filtered = strong_moves.astype(int)
    
    return final_alpha_factor_filtered
```
