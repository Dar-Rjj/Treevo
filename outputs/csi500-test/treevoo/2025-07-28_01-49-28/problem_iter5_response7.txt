```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Close Price Momentum
    close_momentum_21 = df['close'] / df['close'].shift(21) - 1
    close_momentum_63 = df['close'] / df['close'].shift(63) - 1
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_factor_21 = avg_volume_21 / avg_volume_21.rolling(window=42).mean()
    volume_factor_63 = avg_volume_63 / avg_volume_63.rolling(window=126).mean()
    volume_adjusted_momentum_21 = close_momentum_21 * volume_factor_21
    volume_adjusted_momentum_63 = close_momentum_63 * volume_factor_63
    
    # Combined Price Momentum
    price_momentum_21 = df['close'] - df['close'].shift(21)
    price_momentum_63 = df['close'] - df['close'].shift(63)
    combined_price_momentum = 0.5 * price_momentum_21 + 0.5 * price_momentum_63
    
    # Volume Reversal Indicator
    avg_volume_21 = df['volume'].rolling(window=21).mean()
    avg_volume_63 = df['volume'].rolling(window=63).mean()
    volume_diff = avg_volume_21 - avg_volume_63
    volume_reversal = volume_diff / avg_volume_63
    
    # Open-High-Low Volatility Factor
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['low'] - df['open']
    weighted_avg_diff = (df['volume'] * open_high_diff + df['volume'] * open_low_diff.abs()) / (2 * df['volume'])
    
    # Price-Amount Interaction Effect
    amount_per_close = df['amount'].rolling(window=21).sum() / df['close']
    prev_amount_per_close = df['amount'].rolling(window=21).sum().shift(21) / df['close'].shift(21)
    interaction_effect = amount_per_close - prev_amount_per_close
    
    # Daily Return
    daily_return = df['close'].pct_change()
    
    # Smoothed Momentum
    smoothed_momentum = daily_return.rolling(window=20).sum() / df['volume'].rolling(window=20).mean()
    
    # Enhanced Price Reversion
    high_10 = df['high'].rolling(window=10).max()
    low_10 = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / high_10
    proximity_to_low = df['close'] / low_10
    combined_proximity = np.minimum(proximity_to_high, proximity_to_low)
    inverted_proximity = 1 / combined_proximity
    std_dev_20 = daily_return.rolling(window=20).std()
    enhanced_price_reversion = inverted_proximity / std_dev_20
    
    # Weighted Final Alpha Factor
    alpha_factor = (
        (volume_adjusted_momentum_21 + volume_adjusted_momentum_63) / 2 +
        combined_price_momentum +
        volume_reversal +
        weighted_avg_diff +
        interaction_effect
    ) * smoothed_momentum * enhanced_price_reversion
    
    return alpha_factor
```
