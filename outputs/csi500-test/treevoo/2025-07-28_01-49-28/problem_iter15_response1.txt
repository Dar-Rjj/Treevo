```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=20).mean()
    
    # Calculate Adjusted Spread Momentum
    momentum_high_low = volume_adjusted_high_low_spread - ma_volume_adjusted_high_low_spread.shift(1)
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Incorporate Volume into Close-Open Spread
    volume_adjusted_close_open_spread = close_open_spread * df['volume']
    
    # Calculate Moving Sum of Volume-Adjusted Close-Open Spread
    moving_sum_volume_adjusted_close_open_spread = volume_adjusted_close_open_spread.rolling(window=10).sum()
    
    # Integrate Momentum with Volume-Adjusted Close-Open Spread
    integrated_momentum_close_open = volume_adjusted_close_open_spread - moving_sum_volume_adjusted_close_open_spread.shift(1)
    
    # Calculate Weighted Momentum
    weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Smoothed Momentum
    smoothed_momentum = df['close'].pct_change().rolling(window=10).sum() / df['volume']
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['open']
    
    # Proximity to 10-Day Range
    ten_day_high = df['high'].rolling(window=10).max()
    ten_day_low = df['low'].rolling(window=10).min()
    proximity_to_high = df['close'] / ten_day_high
    proximity_to_low = df['close'] / ten_day_low
    min_proximity = pd.concat([proximity_to_high, proximity_to_low], axis=1).min(axis=1)
    adjusted_proximity = min_proximity / df['close'].pct_change().rolling(window=10).std()
    
    # Combine Factors
    factor = (momentum_high_low + weighted_momentum) * price_change_ratio * smoothed_momentum * adjusted_proximity
    factor.name = "Volume-Adjusted High-Low and Close-Open Spreads, Momentum, and Proximity to 10-Day Range"
    
    return factor
```
