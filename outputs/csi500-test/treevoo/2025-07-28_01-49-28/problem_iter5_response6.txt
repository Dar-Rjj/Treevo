```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume to get Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Weighted Momentum
    close_price_change = df['close'].diff(1)
    weighted_momentum = close_price_change * df['volume']
    
    # Combine with Volume-Adjusted High-Low Spread
    combined_factor = volume_adjusted_high_low_spread + weighted_momentum
    
    # Adjust for Volume Trend
    volume_percentage_change = df['volume'].pct_change(1)
    final_alpha_factor = combined_factor * (1 + volume_percentage_change)
    
    return final_alpha_factor
```
