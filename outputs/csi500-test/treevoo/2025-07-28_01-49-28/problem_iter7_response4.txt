```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']

    # Calculate Open-to-Close Movement
    open_to_close_movement = df['close'] - df['open']

    # Adjust for Volume
    volume_adjusted_range = daily_range * df['volume']

    # Combine Indicators
    combined_indicator = volume_adjusted_range + open_to_close_movement

    # Filter for Strong Moves
    strong_moves = (combined_indicator > 0).astype(int)

    # Calculate Daily High-Low Price Spread
    daily_high_low_spread = df['high'] - df['low']

    # Calculate Moving Sum of High-Low Price Spread
    moving_sum_high_low_spread = daily_high_low_spread.rolling(window=20).sum()

    # Calculate Spread Momentum
    spread_momentum = moving_sum_high_low_spread - moving_sum_high_low_spread.shift(1)

    # Combine Spread Momentum and Strong Moves
    alpha_factor = strong_moves * spread_momentum

    return alpha_factor
```
