```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).abs() / (df['high'] + df['low'])
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate Typical Price for MFI
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Positive and Negative Money Flow
    df['Money_Flow'] = df['Typical_Price'] * df['volume']
    df['Positive_Money_Flow'] = df[df['Typical_Price'] > df['Typical_Price'].shift(1)]['Money_Flow']
    df['Negative_Money_Flow'] = df[df['Typical_Price'] < df['Typical_Price'].shift(1)]['Money_Flow']

    # Calculate 14-day Sum of Positive and Negative Money Flow
    positive_money_flow_14 = df['Positive_Money_Flow'].rolling(window=14).sum()
    negative_money_flow_14 = df['Negative_Money_Flow'].rolling(window=14).sum()
    
    # Calculate Money Flow Index (MFI)
    df['MFI'] = 100 - (100 / (1 + (positive_money_flow_14 / negative_money_flow_14)))
    
    # Calculate Gain and Loss for RSI
    df['Gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    
    # Calculate 14-day Average Gain and Loss
    avg_gain = df['Gain'].rolling(window=14).mean()
    avg_loss = df['Loss'].rolling(window=14).mean()
    
    # Calculate Relative Strength (RS)
    df['RS'] = avg_gain / avg_loss
    
    # Calculate RSI
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0
    obv = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv += df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv -= df['volume'].iloc[i]
        df.at[df.index[i], 'OBV'] = obv
    
    # Calculate Accumulation/Distribution Line (A/D Line)
    df['AD_Line'] = (df['close'] - df['low'] - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['AD_Line'] = df['AD_Line'].cumsum()
    
    # Calculate Price Volume Trend (PVT)
    df['PVT'] = 0
    pvt = 0
    for i in range(1, len(df)):
        pvt += (df['close'].iloc[i] - df['close'].iloc[i-1]) / df['close'].iloc[i-1] * df['volume'].iloc[i]
        df.at[df.index[i], 'PVT'] = pvt
    
    # Combine Factors
    df['Composite_Score'] = df['Daily_Return'] + df['Intraday_Volatility'] + df['VWAP'] + df['MFI'] + df['RSI'] + df['OBV'] + df['AD_Line'] + df['PVT']
    
    return df['Composite_Score']
```
