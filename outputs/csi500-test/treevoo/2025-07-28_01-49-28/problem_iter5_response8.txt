```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Return
    high_low_return = high_low_range / df['close'].shift(1)
    
    # Incorporate Volume into High-Low Return
    volume_adjusted_high_low_spread = high_low_range * df['volume']
    
    # Adjust for Volume Trend
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_trend_adjusted_high_low_return = high_low_return * volume_change
    
    # Integrate Price, Volume Signals, and Simple Momentum
    recent_days = 5
    sum_volume_trend_adjusted_high_low_return = volume_trend_adjusted_high_low_return.rolling(window=recent_days).sum()
    simple_momentum = df['close'] - df['close'].shift(1)

    # Open-High-Low Volatility Factor
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    open_high_low_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Amount per Unit Close Price Rate of Change
    n_days = 5
    amount_per_unit_close_current = df['amount'].rolling(window=n_days).sum() / df['close']
    amount_per_unit_close_previous = df['amount'].shift(n_days+1).rolling(window=n_days).sum() / df['close'].shift(n_days+1)
    amount_per_unit_close_roc = amount_per_unit_close_current - amount_per_unit_close_previous
    
    # Combine Integrated Signals
    combined_alpha_factor = (
        sum_volume_trend_adjusted_high_low_return + 
        simple_momentum + 
        open_high_low_volatility + 
        amount_per_unit_close_roc
    )
    
    return combined_alpha_factor
```
