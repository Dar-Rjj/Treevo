import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Adjust for Volume
    volume_factor = (df['volume'] / df['volume'].mean())
    volume_adjusted_breakout_strength = (daily_range * volume_factor) + open_close_movement
    
    # Calculate Daily High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Moving Average of High-Low Price Spread
    moving_average_high_low_spread = high_low_spread.rolling(window=20).mean()
    
    # Calculate Spread Momentum
    spread_momentum = moving_average_high_low_spread.diff()
    
    # Incorporate Volume into High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(1)) * volume_factor
    volume_adjusted_momentum = momentum * df['volume']
    
    # Combine Indicators with weights
    combined_indicator = (0.5 * volume_adjusted_breakout_strength) + (0.3 * spread_momentum) + (0.2 * volume_adjusted_momentum)
    
    # Filter for Strong Moves
    threshold = combined_indicator.mean() + 2 * combined_indicator.std()
    strong_moves = (combined_indicator > threshold).astype(int)
    
    return strong_moves
