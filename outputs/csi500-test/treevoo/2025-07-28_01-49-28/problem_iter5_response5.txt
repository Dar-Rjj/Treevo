def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Incorporate Volume
    volume_adjusted_high_low_spread = high_low_spread * df['volume']

    # Calculate Moving Sum of Volume-Adjusted High-Low Spread for the last 20 days
    moving_sum_volume_adjusted_spread = volume_adjusted_high_low_spread.rolling(window=20).sum()

    # Calculate Adjusted Spread Momentum
    adjusted_spread_momentum = moving_sum_volume_adjusted_spread - moving_sum_volume_adjusted_spread.shift(1)

    # Incorporate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Combine Volume and Momentum Enhanced High-Low Spread with Price Momentum
    alpha_factor = adjusted_spread_momentum + price_momentum

    return alpha_factor
