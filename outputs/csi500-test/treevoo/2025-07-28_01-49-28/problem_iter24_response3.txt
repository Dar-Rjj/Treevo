```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjusted_range = (df['high'] - df['low']) * df['volume']
    
    # Incorporate Enhanced Price Reversion
    # 10-Day High and Low Prices
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    
    # Proximity to 10-Day Range
    proximity_to_10d_high = (df['close'] - df['10d_high']).abs()
    proximity_to_10d_low = (df['close'] - df['10d_low']).abs()
    
    # Combine Proximity Metrics
    min_proximity = pd.concat([proximity_to_10d_high, proximity_to_10d_low], axis=1).min(axis=1)
    inverted_proximity = 1 / min_proximity
    
    # Adjust for 20-Day Volatility
    volatility_20d = df['close'].rolling(window=20).std()
    reversion_factor = inverted_proximity / volatility_20d
    
    # Integrate Open-High-Low Volatility
    open_high_diff = df['high'] - df['open']
    open_low_diff = df['open'] - df['low']
    weighted_volatility = (open_high_diff + open_low_diff) * df['volume']
    
    # Moving Sum of 20-Day High-Low Spreads
    moving_sum_20d_spread = (df['high'] - df['low']).rolling(window=20).sum()
    
    # Spread Momentum
    spread_momentum = moving_sum_20d_spread.diff(1)
    
    # Trend Strength Indicator
    ma_50 = df['close'].rolling(window=50).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    trend_strength = (ma_50 - ma_20) / ma_20
    
    # Final Alpha Factor
    alpha_factor = (
        volume_adjusted_range * 
        reversion_factor * 
        weighted_volatility * 
        spread_momentum * 
        trend_strength
    )
    
    return alpha_factor
```
