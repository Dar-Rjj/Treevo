import pandas as pd

def heuristics_v2(df):
    # Compute the 10-day and 50-day simple moving average (SMA) of closing prices
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the percentage change in closing prices over a 20-day period
    df['Price_Change_20'] = df['close'].pct_change(periods=20)
    
    # Compute the difference between the 10-day SMA and the 50-day SMA
    df['SMA_Diff'] = df['SMA_10'] - df['SMA_50']
    
    # Calculate the rate of change (ROC) of closing prices over a 14-day period
    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Calculate the standard deviation of log-returns over a 30-day window
    df['Log_Returns'] = df['close'].apply(lambda x: np.log(x))
    df['Std_30'] = df['Log_Returns'].rolling(window=30).std()
    
    # Determine the high-low range (High - Low) for each day and compute its 20-day moving average
    df['High_Low_Range'] = df['high'] - df['low']
    df['HL_Range_MA_20'] = df['High_Low_Range'].rolling(window=20).mean()
    
    # Calculate the average true range (ATR) over a 14-day period
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Compute the coefficient of variation of daily returns over a 30-day window
    df['Daily_Returns'] = df['close'].pct_change()
    df['CV_30'] = df['Daily_Returns'].rolling(window=30).std() / df['Daily_Returns'].rolling(window=30).mean()
    
    # Compute the 20-day moving average of volume
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the ratio of today's volume to the 20-day moving average of volume
    df['Volume_Ratio'] = df['volume'] / df['Volume_MA_20']
    
    # Compute the on-balance volume (OBV)
    df['OBV'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Analyze the daily price range (High - Low) as a factor
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Compute the relationship between Open and Close prices; (Close - Open) / Open
    df['OC_Relationship'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative strength index (RSI) based on the OHLC data
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI'] = 100 - (100 / (1 + RS))
    
    # Combine the 20-day price momentum with the 20-day average volume
    df['Momentum_Volume_Composite'] = df['Price_Change_20'] * df['Volume_MA_20']
    
    # Create a composite score using the 10-day SMA, 50-day SMA, and ATR
    df['SMA_ATR_Composite'] = df['SMA_Diff'] / df['ATR_14']
    
    # Develop a combined factor using ROC, ATR, and the 20-day moving average of volume
    df['ROC_ATR_Volume_Composite'] = df['ROC_14'] * df['ATR_14'] * df['Volume_MA_20']
    
    # Return the final alpha factor
    return df['ROC_ATR_Volume_Composite']
