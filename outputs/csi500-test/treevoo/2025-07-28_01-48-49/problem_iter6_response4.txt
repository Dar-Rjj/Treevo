```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = current_volume - prev_volume
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_range * volume_change
    intraday_momentum_score = intraday_momentum_score.where(intraday_momentum_score >= 0, 0)
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()
    
    # Finalize Alpha Factor
    intraday_return = (df['close'] - df['open']) / df['open']
    close_5d_ago = df['close'].shift(5)
    five_day_momentum = (df['close'] - close_5d_ago) / close_5d_ago
    high_low_spread_momentum = avg_high_low_spread_5d * five_day_momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    
    return integrated_momentum_factor
```
