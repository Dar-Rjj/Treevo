```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Average Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    adjusted_range = intraday_range / avg_volume_20
    
    # Incorporate Close Price Momentum
    close_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_5 = high_low_spread.rolling(window=5).sum() / 5
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_10 = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply 5-Day Average High-Low Spread by Momentum
    adjusted_high_low = avg_high_low_5 * momentum
    
    # Add 10-Day EMA of High-Low Spread to 20-Day Price Return
    price_return_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    ema_and_price_return = ema_high_low_10 + price_return_20
    
    # Calculate Open Adjustment Factor
    open_adj_factor = df['open'] - df['open'].shift(1)
    
    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_low_open = (df['high'] - df['low']) / open_adj_factor
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_high_low_movement = adjusted_high_low_open * (df['volume'] * close_diff)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Calculate Open Price Impact
    relative_open_price_change = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Final Factor Calculation
    final_factor = (
        0.20 * adjusted_range +
        0.20 * close_momentum +
        0.20 * weighted_high_low_movement +
        0.15 * avg_daily_range +
        0.15 * volume_weighted_return +
        0.10 * relative_open_price_change
    )
    
    return final_factor
```
