import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum = high_low_spread * momentum_5d
    
    # Calculate Enhanced Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    intraday_weighted = intraday_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = intraday_weighted * abs(df['volume'] * close_diff)
    intraday_momentum_score = intraday_momentum_score.clip(lower=0)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_weight = df['volume']
    weighted_intraday_return_reversal = (intraday_return_weight * intraday_movement) + (reversal_weight * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factors = (intraday_momentum_score * volume_weighted_return) * intraday_volatility
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_combined_factors = combined_factors / avg_volume
    
    # Incorporate Close Price Momentum
    close_momentum = (df['close'] - df['close'].rolling(window=10).mean()) / df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = close_momentum * intraday_volatility
    
    # Final Factor Calculation
    final_factor = (0.3 * intraday_momentum_score + 0.3 * intraday_volatility + 0.4 * adjusted_momentum) * 0.5 + weighted_intraday_return_reversal * 0.5
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_alpha_factor = final_factor * price_level_ratio
    
    return final_alpha_factor
