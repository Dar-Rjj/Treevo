```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['amount']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=n_days).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Subtract N-day Moving Average of Adjusted Momentum to get the final factor
    final_factor = adjusted_momentum - adjusted_momentum.rolling(window=n_days).mean()
    
    return final_factor
```
