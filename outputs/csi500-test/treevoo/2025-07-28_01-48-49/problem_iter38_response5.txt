```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the relative position of the closing price within the range
    df['close_range_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow multiplier
    df['money_flow_multiplier'] = (2 * df['close'] - df['low'] - df['high']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    
    # Calculate the cumulative money flow volume
    df['cum_money_flow_volume'] = df['money_flow_volume'].rolling(window=14).sum()
    
    # Calculate the cumulative volume
    df['cum_volume'] = df['volume'].rolling(window=14).sum()
    
    # Calculate the money flow ratio
    df['money_flow_ratio'] = df['cum_money_flow_volume'] / df['cum_volume']
    
    # Calculate the money flow index
    df['money_flow_index'] = 100 - (100 / (1 + df['money_flow_ratio']))
    
    # Calculate the final factor
    df['factor'] = df['close_range_position'] * df['money_flow_index']
    
    return df['factor']
```
