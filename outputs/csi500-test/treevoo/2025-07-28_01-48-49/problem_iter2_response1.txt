```python
import pandas as pd

def heuristics_v2(df, n_days=5):
    # Calculate High-to-Low Movement and Adjustment
    high_to_low = df['high'] - df['low']
    
    open_adjustment = df['open'] - df['open'].shift(1)
    
    adjusted_high_to_low = high_to_low / open_adjustment
    
    # Incorporate Volume, Close Difference, and Intraday Volatility
    volume = df['volume']
    close_diff = df['close'] - df['close'].shift(1)
    intraday_price_range = df['high'] - df['low']
    intraday_volatility = intraday_price_range / df['amount']
    
    synthesized_value = (adjusted_high_to_low * (volume * close_diff)) + intraday_volatility
    
    # Final Synthesis with Momentum
    momentum = df['close'] - df['close'].rolling(window=n_days).mean()
    adjusted_momentum = momentum * synthesized_value
    final_alpha = adjusted_momentum - adjusted_momentum.rolling(window=n_days).mean()
    
    return final_alpha
```
