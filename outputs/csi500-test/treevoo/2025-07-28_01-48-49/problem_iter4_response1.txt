```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Combine Factors: Multiply Close-to-Close Return by Intraday Range
    combined_factor = close_to_close_return * intraday_range
    
    # Filter for Recent Trend: Calculate 5-Day Moving Average of Close Prices
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Subtract from Current Close Price
    recent_trend = df['close'] - ma_5
    
    # Final Heuristic: Multiply the combined factor by the recent trend
    final_factor = combined_factor * recent_trend
    
    return final_factor
```
