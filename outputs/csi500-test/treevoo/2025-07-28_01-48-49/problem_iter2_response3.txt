import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range (High - Low)
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Difference
    df['CloseToOpenDiff'] = df['close'] - df['open'].shift(-1)
    
    # Apply Weight by Volume
    df['VolumeWeightedDiff'] = df['CloseToOpenDiff'] * df['volume']
    
    # Calculate Sum of Volume-Weighted Differences over a 5-day rolling window
    df['SumVolumeWeightedDiff'] = df['VolumeWeightedDiff'].rolling(window=5).sum()
    
    # Calculate Simple Moving Average of Weighted Sums over 10 days
    df['SMA_WeightedDiff'] = df['SumVolumeWeightedDiff'].rolling(window=10).mean()
    
    # Calculate Average True Range (ATR) over 14 days
    df['TrueRange'] = df[['high' - 'low', 'high' - 'close'.shift(1), 'low' - 'close'.shift(1)]].abs().max(axis=1)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    
    # Adjust Alpha Factor for Volatility Impact
    df['AlphaFactor'] = df['SMA_WeightedDiff'] / df['ATR']
    
    # Calculate 5-Day Average High-Low Spread
    df['AvgHighLowSpread'] = df['DailyRange'].rolling(window=5).mean()
    
    # Calculate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Combine Alpha Factor with Momentum
    df['FinalAlphaFactor'] = df['AlphaFactor'] * df['Momentum']
    
    return df['FinalAlphaFactor'].dropna()
