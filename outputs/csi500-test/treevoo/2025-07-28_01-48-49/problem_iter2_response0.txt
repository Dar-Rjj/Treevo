```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_adj_factor = df['open'] - df['open'].shift(1)
    
    # Adjust High-to-Low Movement by Open Adjustment
    adj_high_to_low_movement = high_to_low_movement / open_adj_factor
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_adj_high_to_low = adj_high_to_low_movement * (df['volume'] * close_diff)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume'].sum()
    
    # Combine Factors
    combined_factors = (close_to_close_return * intraday_range) / vwap
    
    # Filter for Recent Trend
    recent_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = weighted_adj_high_to_low * combined_factors * recent_trend
    
    return alpha_factor
```
