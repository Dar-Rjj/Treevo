def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_difference = df['volume'].diff(1)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = intraday_range * volume_difference
    
    # Calculate 5-Day Average High-Low Spread
    five_day_avg_high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor
    five_day_momentum = (df['open'] - df['close'].shift(5)) / df['close'].shift(5)
    integrated_momentum_factor = (five_day_avg_high_low_spread * five_day_momentum) + intraday_momentum_score
    
    return integrated_momentum_factor
