```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum = high_low_spread * momentum_5d
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Intraday Volatility
    intraday_volatility = intraday_range.rolling(window=5).mean()
    
    # Calculate Close-to-Close Return
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Price Difference with Volume
    weighted_diff = df['volume'] * close_to_close_return
    sum_weighted_diff = weighted_diff.rolling(window=5).sum()
    
    # Combine Factors
    combined_factor = close_to_close_return * intraday_range
    adjusted_combined_factor = combined_factor / intraday_volatility
    
    # Introduce Smoothed Moving Average Filter
    sma_weighted_sums = sum_weighted_diff.rolling(window=10).mean()
    smoothed_factor = sma_weighted_sums * spread_momentum
    
    # Filter for Recent Price Trend
    ma_close_5d = df['close'].rolling(window=5).mean()
    recent_price_trend = df['close'] - ma_close_5d
    
    # Introduce Volume Trend
    volume_sum_5d = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5d
    
    # Final Factor Computation
    final_factor = smoothed_factor * volume_ratio * recent_price_trend * spread_momentum
    
    return final_factor
```
