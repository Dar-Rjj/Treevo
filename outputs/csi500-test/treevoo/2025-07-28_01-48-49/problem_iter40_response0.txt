```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Price over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) of Close Price over 20 days
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Rate of Change (ROC) of Close Price from 1 period ago
    roc_1 = df['close'].pct_change(periods=1)
    
    # Calculate Relative Strength Index (RSI) of Close Price over 14 days
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate True Range (TR)
    tr = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over 14 days
    atr_14 = tr.rolling(window=14).mean()
    
    # High-Low Range
    hl_range = df['high'] - df['low']
    
    # New Highs/New Lows over 10 days
    new_high = (df['high'].rolling(window=10).max() == df['high']).astype(int)
    new_low = (df['low'].rolling(window=10).min() == df['low']).astype(int)
    
    # Positive (+DI) and Negative (-DI) Directional Indicators over 14 days
    dm_plus = (df['high'].diff(1).fillna(0) > df['low'].diff(1).fillna(0)) * (df['high'].diff(1).fillna(0))
    dm_minus = (df['low'].diff(1).fillna(0) > df['high'].diff(1).fillna(0)) * (df['low'].diff(1).fillna(0))
    di_plus = 100 * (dm_plus.rolling(window=14).sum() / tr.rolling(window=14).sum())
    di_minus = 100 * (dm_minus.rolling(window=14).sum() / tr.rolling(window=14).sum())
    
    # Moving Average Convergence Divergence (MACD) and Signal Line
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Custom Indicators
    daily_returns = df['close'].pct_change()
    cum_sum_daily_returns = daily_returns.cumsum()
    volume_weighted_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine all indicators into a single alpha factor
    alpha_factor = (
        sma_20 + 
        ema_20 + 
        roc_1 + 
        rsi_14 + 
        obv + 
        atr_14 + 
        hl_range + 
        new_high + 
        new_low + 
        di_plus - di_minus + 
        macd_line - signal_line + 
        cum_sum_daily_returns + 
        volume_weighted_close
    )
    
    return alpha_factor
```
