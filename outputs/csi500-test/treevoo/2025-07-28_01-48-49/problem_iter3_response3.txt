import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['volume']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=n).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_momentum - adjusted_momentum.rolling(window=n).mean()
    
    return alpha_factor
