```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_intraday_price_movement = intraday_price_movement * (df['volume'] * close_difference)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount'] * intraday_price_movement
    reversal_signal_weight = df['volume'] * reversal_signal
    weighted_intraday_return_and_reversal = intraday_return_weight + reversal_signal_weight
    
    # Combine Enhanced Factors
    combined_factor = (weighted_intraday_price_movement * intraday_volatility) * df['volume'].rolling(window=30).mean()
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=30).mean()
    adjusted_combined_factor = combined_factor / average_volume
    
    # Incorporate Close Price Momentum
    momentum = df['close'].rolling(window=30).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Final Factor Calculation
    final_factor = (
        0.7 * (0.5 * adjusted_combined_factor + 0.3 * intraday_volatility + 0.2 * adjusted_momentum) +
        0.3 * weighted_intraday_return_and_reversal
    )
    
    return final_factor
```
