import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate 5-Day Average Intraday Price Range
    avg_intraday_range = intraday_range.rolling(window=5).mean()
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=n).mean()
    
    # Adjust Momentum
    adjusted_momentum = momentum * avg_intraday_range
    
    # Final Alpha Factor
    final_alpha = adjusted_momentum - adjusted_momentum.rolling(window=n).mean()
    
    return final_alpha
