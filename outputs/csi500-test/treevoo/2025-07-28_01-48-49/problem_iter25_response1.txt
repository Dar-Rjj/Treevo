```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Intraday Price Movement
    df['Prev_Open'] = df['open'].shift(1)
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['Prev_Open'])
    
    # Weight by Volume and Close Difference
    df['Volume_Close_Diff'] = df['volume'] * (df['close'] - df['close'].shift(1))
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * df['Volume_Close_Diff']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    df['Intraday_Volatility_20'] = df['Intraday_Volatility'].rolling(window=20).mean()
    
    # Close-to-Open Reversal Signal
    df['Close_to_Open_Reversal'] = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Close_to_Open_Reversal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return_and_Reversal'] = (df['amount'] * df['Intraday_Price_Movement']) + df['Volume_Weighted_Return']
    
    # Combine Enhanced Factors
    df['Enhanced_Factor'] = (df['Weighted_Intraday_Price_Movement'] * df['Volume_Weighted_Return']) + (df['Intraday_Volatility_20'] * df['Volume_Weighted_Return'])
    
    # Adjusted Momentum
    df['Momentum'] = (df['close'] - df['close'].rolling(window=10).mean()) * df['Intraday_Volatility_20']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (0.3 * df['Intraday_Price_Movement'] + 
                                 0.3 * df['Intraday_Volatility_20'] + 
                                 0.4 * df['Momentum']) * 0.5 + \
                                (df['Weighted_Intraday_Return_and_Reversal'] * 0.5)
    
    return df['Final_Alpha_Factor']
```
