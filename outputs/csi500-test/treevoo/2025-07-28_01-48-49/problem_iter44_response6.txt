```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between the close and open prices
    relative_diff = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Determine the direction of the day's trading
    direction = (df['close'] > df['open']).astype(int)
    
    # Calculate the factor value by combining the three components
    factor_values = relative_diff * vwap * direction
    
    return factor_values
```
