```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Movement
    df['adj_high_low'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['weighted_adj_high_low'] = df['adj_high_low'] * df['volume'] * df['close_diff']
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(-1) - 1
    
    # Calculate Volume-Weighted Return
    df['vol_weighted_return'] = df['volume'] * df['close_to_open_return']
    
    # Combine Enhanced Factors
    df['combined_factor'] = df['weighted_adj_high_low'] * df['intraday_volatility'] * df['vol_weighted_return']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_combined_factor'] = df['combined_factor'] / df['avg_volume']
    
    # Incorporate Close Price Momentum
    df['momentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Final Factor Calculation
    df['final_factor'] = 0.3 * df['adj_high_low'] + 0.3 * df['intraday_volatility'] + 0.4 * df['momentum']
    return df['final_factor']
```
