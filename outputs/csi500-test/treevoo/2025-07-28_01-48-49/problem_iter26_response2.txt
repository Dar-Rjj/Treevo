```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of close to open price
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of amount to volume
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Calculate the normalized difference between high and low
    high_low_diff = (df['high'] - df['low']) / df['low']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (close_pct_change + close_to_open_ratio + amount_to_volume_ratio + high_low_diff) / 4
    
    return alpha_factor
```
