```python
def heuristics_v2(df):
    # Calculate the weighted average of high and low prices, using volume as weights
    weighted_avg_price = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between the close price and the weighted average price
    close_to_weighted_avg_diff = df['close'] - weighted_avg_price
    
    # Calculate a smoothed version of the amount traded
    smoothed_amount = df['amount'].rolling(window=5).mean()
    
    # The final factor is the ratio of the close-to-weighted-avg difference to the smoothed amount
    factor_values = close_to_weighted_avg_diff / smoothed_amount
    
    return factor_values
```
