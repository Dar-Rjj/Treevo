```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    df['5_day_avg_intraday_range'] = df['intraday_range'].rolling(window=5).mean()
    
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    df['vol_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_price_range'] = (df['price_range'] * df['volume_change']) / df['vol_ma_20']
    
    # Apply Price Level Adjustment
    df['price_level_ratio'] = df['close'] / df['open']
    df['adjusted_price_range'] = df['adjusted_price_range'] * df['price_level_ratio']
    
    # Incorporate Close Price Momentum
    df['momentum'] = df['close'] / df['close'].shift(10)
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=5).mean()
    
    # Calculate 5-Day Return
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Determine Intraday Volatility Adjusted Momentum
    df['intraday_vol_adj_momentum'] = df['5_day_return'] * df['5_day_avg_intraday_range']
    
    # Compute Intraday Reversal Signal
    df['intraday_reversal'] = (df['close'] - df['open']) / df['5_day_avg_intraday_range']
    
    # Incorporate Volume Impact
    df['5_day_cum_vol'] = df['volume'].rolling(window=5).sum()
    df['volume_weight'] = df['volume'] / df['5_day_cum_vol']
    
    # Calculate Volume-Weighted Return
    df['vol_weighted_return'] = df['volume'] * df['intraday_reversal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['intraday_return_weight'] = df['amount']
    df['reversal_signal_weight'] = df['volume']
    df['weighted_intraday_return'] = (df['intraday_return_weight'] * df['intraday_reversal']) + (df['reversal_signal_weight'] * df['intraday_reversal'])
    
    # Combine 5-Day Average Intraday Range and Momentum
    df['combined_high_low_spread_momentum'] = df['5_day_avg_intraday_range'] * df['momentum']
    
    # Final Factor Calculation
    df['final_factor'] = (0.25 * df['adjusted_price_range']) + (0.25 * df['momentum']) + (0.25 * df['5_day_avg_intraday_range']) + (0.15 * df['avg_daily_range']) + (0.1 * df['vol_weighted_return'])
    
    return df['final_factor']
```
