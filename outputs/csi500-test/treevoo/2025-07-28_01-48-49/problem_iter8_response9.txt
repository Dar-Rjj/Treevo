import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted High-to-Low Movement
    adj_high_low_movement = (df['high'] - df['low']) / (df['open'] - df['close'])

    # Weight by Volume and Previous Day's Volume Change
    prev_vol_change = df['volume'].shift(1) - df['volume'].shift(2)
    weighted_adj_high_low = adj_high_low_movement * (df['volume'] * prev_vol_change)

    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']

    # Adjust by Average Volume
    avg_20d_volume = df['volume'].rolling(window=20).mean()
    intraday_vol_adjusted = intraday_price_range / avg_20d_volume

    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['amount']

    # Calculate Momentum
    momentum = df['close'].rolling(window=5).mean()

    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility

    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Close-to-Open Reversal Signal
    close_to_open_reversal = (df['close'] - df['open']) / df['open']

    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return = (df['amount'] * intraday_return) + (df['volume'] * close_to_open_reversal)

    # Intermediate Combined Factors
    intermediate_factor_1 = (weighted_adj_high_low * 0.5) + (weighted_intraday_return * 0.5)
    intermediate_factor_2 = (adjusted_momentum * 0.6) + (close_to_open_reversal * 0.4)

    # Final Alpha Factor
    final_alpha_factor = (intermediate_factor_1 * 0.5) + (intermediate_factor_2 * 0.5)

    return final_alpha_factor
