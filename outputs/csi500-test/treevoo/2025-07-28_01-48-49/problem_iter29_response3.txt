def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by (date).
    """
    # Calculate the true range
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df[['low', 'close']].min(axis=1)) / true_range
    
    # Calculate the selling pressure
    selling_pressure = (df[['high', 'close']].max(axis=1) - df['close']) / true_range
    
    # Calculate the net buying pressure
    net_buying_pressure = buying_pressure - selling_pressure
    
    # Calculate the weighted volume
    weighted_volume = df['volume'] * (df['close'] - df['open']) / df['close']
    
    # Combine the net buying pressure and weighted volume to generate the alpha factor
    alpha_factor = net_buying_pressure * weighted_volume
    
    return alpha_factor
