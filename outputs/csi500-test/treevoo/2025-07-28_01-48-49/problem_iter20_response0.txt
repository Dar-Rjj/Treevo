```python
def heuristics_v2(df):
    # Input: df - a pandas DataFrame with (open, high, low, close, amount, volume) as columns
    
    # Calculate the price movement from open to close
    price_movement = df['close'] - df['open']
    
    # Calculate the buying pressure using the price movement and volume
    buying_pressure = price_movement * df['volume']
    
    # Calculate the weighted average of high and low, using amount as weight
    weighted_price = (df['high']*df['amount'] + df['low']*df['amount']) / (2*df['amount'])
    
    # Calculate the difference between the weighted price and the closing price
    diff_weighted_close = weighted_price - df['close']
    
    # Generate the alpha factor by combining buying pressure and the difference between weighted price and close
    alpha_factor = (buying_pressure * diff_weighted_close).rolling(window=5).mean()
    
    return alpha_factor
```
