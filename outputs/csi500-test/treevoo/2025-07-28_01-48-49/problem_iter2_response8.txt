import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    
    # Calculate Dynamic Price Difference
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Aggregate Price Difference with Volume
    volume_weighted_return = df['volume'] * close_to_open_return
    sum_weighted_differences = volume_weighted_return.rolling(window=5).sum()
    
    # Enhance with Smoothed Moving Average Filter
    sma_weighted_sums = sum_weighted_differences.rolling(window=10).mean()
    
    # Finalize Alpha Factor
    alpha_factor = sma_weighted_sums / avg_daily_range
    
    # Combine Intraday Volatility and Volume-Weighted Return
    combined_alpha = alpha_factor * volume_weighted_return
    
    return combined_alpha
