import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_movement = intraday_movement * df['volume'] * close_diff
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    daily_range_20dma = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return = (df['volume'] * intraday_movement) + (df['amount'] * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factor = weighted_intraday_movement * volume_weighted_return
    
    # Adjust by Average Volume
    avg_volume_40dma = df['volume'].rolling(window=40).mean()
    adjusted_combined_factor = combined_factor / avg_volume_40dma
    
    # Incorporate Close Price Momentum
    momentum_30dma = df['close'].rolling(window=30).mean()
    adjusted_momentum = (momentum_30dma - df['close']) * daily_range_20dma
    
    # Adjusted Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    adjusted_momentum_ema = ema_10 - momentum_30dma
    
    # Filter for Recent Price Trend
    recent_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Introduce Volume Trend
    volume_sum_5 = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5
    integrated_momentum_factor = adjusted_momentum_ema * volume_ratio
    volume_trend_adjusted_factor = integrated_momentum_factor * volume_ratio
    
    # Final Factor Calculation
    final_factor = (
        0.55 * adjusted_intraday_movement + 
        0.225 * daily_range_20dma + 
        0.225 * adjusted_momentum
    )
    final_alpha_factor = (
        0.75 * final_factor + 
        0.25 * weighted_intraday_return
    )
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.6 * integrated_momentum_factor + 
        0.4 * volume_trend_adjusted_factor
    )
    
    return final_alpha_factor
