```python
def heuristics_v2(df):
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open-Close Price Difference
    open_close_difference = df['open'] - df['close']
    
    # Adjust High-to-Low Movement by Open-Close Price Difference
    adjusted_high_to_low = high_to_low_movement / open_close_difference
    
    # Calculate Previous Day's Volume Change
    prev_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume and Previous Day's Volume Change
    factor = adjusted_high_to_low * (df['volume'] * prev_volume_change)
    
    return factor
```
