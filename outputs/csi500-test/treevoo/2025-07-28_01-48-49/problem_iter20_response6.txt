```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['IntradayPriceMovement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    df['WeightedByVolumeAndCloseDiff'] = df['IntradayPriceMovement'] * (df['volume'] * close_diff)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    df['DailyRangeMA'] = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['ReversalSignal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['VolumeWeightedReturn'] = df['volume'] * df['ReversalSignal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_signal_weight = df['volume']
    df['WeightedIntradayReturnAndReversal'] = (intraday_return_weight * df['IntradayPriceMovement']) + (reversal_signal_weight * df['ReversalSignal'])
    
    # Combine Enhanced Factors
    df['CombinedEnhancedFactors'] = df['WeightedByVolumeAndCloseDiff'] * df['VolumeWeightedReturn']
    
    # Explore the Relationship Between Price Patterns and Future Returns
    df['IntradayRange'] = df['high'] - df['low']
    df['DailyMomentum'] = (df['close'] - df['open']) / df['open']
    
    # Investigate the Impact of Volume on Price Movements
    short_period = 5
    df['ClosingPriceToAvgVolumeRatio'] = df['close'] / df['volume'].rolling(window=short_period).sum()
    df['VolumeChangePercentage'] = df['volume'].pct_change()
    
    # Consider Multi-Day Price and Volume Trends
    df['MovingSumOfDailyReturns'] = df['close'].pct_change().rolling(window=20).sum()
    df['VolumeTrend'] = df['volume'].rolling(window=10).mean()
    
    # Combine Different Factors to Form a Composite Alpha Factor
    df['CompositeAlphaFactor'] = (0.3 * df['IntradayPriceMovement']) + (0.3 * df['DailyRangeMA']) + (0.4 * df['ReversalSignal'])
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (0.5 * df['CompositeAlphaFactor']) + (0.5 * df['WeightedIntradayReturnAndReversal'])
    
    return df['FinalAlphaFactor']
```
