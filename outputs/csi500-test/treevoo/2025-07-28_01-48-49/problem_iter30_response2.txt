```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index and the calculated alpha factor values.
    """
    # Momentum Indicator - Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Volatility Indicator - Average True Range (ATR) using high, low, and close
    tr = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Volume Indicator - Chaikin Money Flow (CMF)
    mf_multiplier = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    cmf = (mf_multiplier * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combining the indicators
    alpha_factor = roc + (cmf / atr)
    
    return alpha_factor
```
