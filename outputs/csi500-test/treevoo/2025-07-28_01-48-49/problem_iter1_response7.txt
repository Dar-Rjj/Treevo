```python
def heuristics_v2(data):
    # Calculate Intraday Return
    intraday_return = (data['high'] - data['low']) / data['low']
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (data['close'] - data['open']) / data['open']
    
    # Weights based on amount and volume
    intraday_return_weight = data['amount']
    reversal_signal_weight = data['volume']
    
    # Construct Hybrid Alpha
    hybrid_alpha = (intraday_return_weight * intraday_return) + (reversal_signal_weight * reversal_signal)
    
    return hybrid_alpha
```
