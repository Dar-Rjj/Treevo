```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Integrate Intraday Price Movement and Volume
    df['VolumeChange'] = df['volume'].pct_change()
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['5DayAvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    df['5DayMomentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = df['close'].pct_change()
    df['VWAP'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    df['CombinedFactor'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VWAP']
    
    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['CombinedFactor'] * df['VolumeRatio']
    
    # Utilize High and Low Prices for Volatility Measures
    df['DailyRange'] = df['high'] - df['low']
    df['RelativeRange'] = df['DailyRange'] / df['close']
    df['AverageDailyRange'] = df['DailyRange'].rolling(window=20).mean()
    df['VolatilitySpike'] = (df['DailyRange'] > 1.5 * df['AverageDailyRange']).astype(int)
    
    # Generate Complex Alpha Factors from Basic Components
    df['MomentumVolatilityScore'] = df['IntegratedMomentumFactor'] * (1 + df['VolatilitySpike'])
    df['ComprehensiveAlphaFactor'] = (df['5DayMomentum'] * df['VolumeTrendAdjustedFactor']) + (df['RelativeRange'] * df['VolumeRatio'])
    
    # Final Factor Calculation
    df['FinalAlphaFactor'] = (
        0.2 * df['RelativeRange'] +
        0.3 * df['5DayMomentum'] +
        0.2 * df['IntegratedMomentumFactor'] +
        0.3 * df['VolumeTrendAdjustedFactor']
    )
    
    return df['FinalAlphaFactor'].dropna()
```
