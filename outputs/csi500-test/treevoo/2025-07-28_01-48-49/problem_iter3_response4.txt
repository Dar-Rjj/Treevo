```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day Exponential Moving Average (EMA) of High-Low Spread
    ema_high_low_spread_10d = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate 20-Day Price Return
    price_return_20d = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Multiply 5-Day Average High-Low Spread by Momentum
    spread_momentum = avg_high_low_spread_5d * momentum
    
    # Add 10-Day EMA of High-Low Spread to 20-Day Price Return
    factor_value = ema_high_low_spread_10d + price_return_20d
    
    # Final factor value
    final_factor = spread_momentum + factor_value
    
    return final_factor
```
