```python
def heuristics_v2(df):
    # Calculate the relative position of the close price in the range of the day
    relative_close = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the ratio of the amount to the volume, which can be seen as a weighted average price
    avg_price = df['amount'] / df['volume']

    # Calculate the momentum of the weighted average price
    mom_avg_price = avg_price.pct_change()

    # Combine the two signals
    factor = relative_close * mom_avg_price.shift(1)

    return factor
```
