import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum = avg_high_low_spread * momentum_5d
    
    # Calculate Enhanced Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_movement = intraday_price_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = (weighted_intraday_movement * abs(df['volume'] * close_diff)).clip(lower=0)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_weight = df['volume']
    weighted_intraday_return_reversal = (intraday_return_weight * intraday_price_movement) + (reversal_weight * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factors = (intraday_momentum_score * intraday_volatility * volume_weighted_return)
    
    # Incorporate Close Price Momentum
    close_momentum = df['close'] / df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = close_momentum * intraday_volatility
    
    # Final Factor Calculation
    final_factor = (0.3 * intraday_momentum_score) + (0.3 * intraday_volatility) + (0.4 * adjusted_momentum)
    combined_final_factor = (final_factor * 0.5) + (weighted_intraday_return_reversal * 0.5)
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_final_factor = combined_final_factor * price_level_ratio
    
    # Final Factor Computation
    volume_trend_adjusted_factor = adjusted_final_factor * (df['volume'] / df['volume'].shift(1))
    
    return volume_trend_adjusted_factor
