import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous day's close to the current day's open
    df['overnight_move'] = df['open'] - df['close'].shift(1)
    
    # Calculate the daily trading range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the percentage of the overnight move relative to the daily range
    df['overnight_pct_of_range'] = df['overnight_move'].abs() / df['daily_range']
    
    # Calculate the percentage change in volume compared to the previous day
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate the weighted sum of overnight_pct_of_range and volume_change_pct
    # This is a simple heuristic that combines the relative importance of overnight moves and volume changes
    alpha_factor = df['overnight_pct_of_range'] * 0.7 + df['volume_change_pct'] * 0.3
    
    return alpha_factor
