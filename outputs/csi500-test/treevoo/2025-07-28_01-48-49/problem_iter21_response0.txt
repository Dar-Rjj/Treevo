```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Moving Average of Volume (20 days)
    moving_avg_vol = df['volume'].rolling(window=20).mean()
    
    # Current Day Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjusted Price Range by Volume Change
    adjusted_price_range = intraday_range * volume_change / moving_avg_vol
    
    # Price Level Ratio
    price_level_ratio = df['close'] / df['open']
    
    # Adjust the Divided Price Range by Price Level Ratio
    adjusted_range = adjusted_price_range * price_level_ratio
    
    # Close Price Momentum (10 days)
    momentum = df['close'].pct_change(periods=10)
    
    # 5-Day Average High-Low Spread
    high_low_spread = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Combine 5-Day Average High-Low Spread and Momentum
    combined_high_low_momentum = high_low_spread * momentum
    
    # Daily Range (High - Low)
    daily_range = df['high'] - df['low']
    
    # Average of Daily Ranges (20 days)
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Volume-Weighted Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Final Factor Calculation
    factor = (
        0.25 * adjusted_range +
        0.25 * momentum +
        0.25 * combined_high_low_momentum +
        0.15 * avg_daily_range +
        0.1 * volume_weighted_return
    )
    
    return factor
```
