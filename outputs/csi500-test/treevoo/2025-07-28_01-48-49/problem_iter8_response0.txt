```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['open_diff'] = df['open'].diff()
    df['intraday_movement'] = (df['high'] - df['low']) / df['open_diff']

    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'].diff()
    df['weighted_intraday'] = df['intraday_movement'] * df['volume'] * df['close_diff']

    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['volatility'] = df['daily_range'].rolling(window=20).mean()

    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']

    # Calculate Volume-Weighted Return
    df['volume_weighted_return'] = df['volume'] * df['reversal_signal']

    # Construct Weighted Intraday Return and Reversal Signal
    df['intraday_return_weight'] = df['amount']
    df['reversal_weight'] = df['volume']
    df['weighted_reversal_signal'] = (df['intraday_return_weight'] * df['intraday_movement']) + (df['reversal_weight'] * df['reversal_signal'])

    # Combine Enhanced Factors
    df['combined_factors'] = df['weighted_intraday'] * df['volatility'] * df['volume_weighted_return']

    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_factors'] = df['combined_factors'] / df['avg_volume']

    # Incorporate Close Price Momentum
    df['momentum'] = df['close'].rolling(window=10).mean()

    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['momentum'] * df['volatility']

    # Final Factor Calculation
    df['final_factor'] = (0.3 * df['intraday_movement']) + (0.3 * df['volatility']) + (0.4 * df['adjusted_momentum'])
    df['final_alpha_factor'] = (0.5 * df['final_factor']) + (0.5 * df['weighted_reversal_signal'])

    return df['final_alpha_factor']
```
