```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['open_diff'] = df['open'].diff()
    df['intraday_price_movement'] = (df['high'] - df['low']) / df['open_diff']
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'].diff()
    df['weighted_intraday_price_movement'] = df['intraday_price_movement'] * df['volume'] * df['close_diff']
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['volume_weighted_return'] = df['volume'] * df['reversal_signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['intraday_return_weight'] = df['amount']
    df['reversal_weight'] = df['volume']
    df['weighted_intraday_return_and_reversal'] = (df['intraday_return_weight'] * df['intraday_price_movement']) + (df['reversal_weight'] * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_factors'] = df['weighted_intraday_price_movement'] * df['volume_weighted_return'] * df['intraday_volatility']
    
    # Incorporate Close Price Momentum
    df['close_momentum'] = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['close_momentum'] * df['intraday_volatility']
    
    # Final Factor Calculation
    df['final_factor'] = (0.3 * df['weighted_intraday_price_movement']) + (0.3 * df['intraday_volatility']) + (0.4 * df['adjusted_momentum'])
    df['final_alpha_factor'] = (df['final_factor'] * 0.5) + (df['weighted_intraday_return_and_reversal'] * 0.5)
    
    return df['final_alpha_factor']
```
