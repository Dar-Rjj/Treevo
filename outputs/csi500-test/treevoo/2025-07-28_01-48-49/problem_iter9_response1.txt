```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['volume']
    
    # Calculate Momentum
    momentum = df['close'].ewm(span=10, adjust=False).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_adjustment_factor = df['open'] - df['open'].shift(1)
    
    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_to_low = high_to_low_movement / open_adjustment_factor
    
    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_adjusted_high_to_low = adjusted_high_to_low * (df['volume'] * close_difference)
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['volume']
    reversal_signal_weight = df['amount']
    weighted_intraday_return_and_reversal = (intraday_return_weight * daily_return) + (reversal_signal_weight * (-daily_return))
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum * 0.6) + (weighted_intraday_return_and_reversal * 0.4)
    
    # Filter for Recent Trend
    recent_trend = df['close'] - df['close'].rolling(window=20).mean()
    final_alpha_factor *= recent_trend
    
    return final_alpha_factor
```
