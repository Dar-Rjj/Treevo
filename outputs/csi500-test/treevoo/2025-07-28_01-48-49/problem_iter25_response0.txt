```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Determine Price Trend
    df['Upward_Trend'] = ((df['close'] > df['open']) & (df['high'] > df['low'])).astype(int)
    df['Downward_Trend'] = ((df['close'] < df['open']) & (df['high'] > df['low'])).astype(int) * -1
    df['Price_Trend'] = df['Upward_Trend'] + df['Downward_Trend']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Identify Volume Spikes
    df['Volume_Spike'] = (df['volume'] > 2 * df['volume'].rolling(window=5).mean().shift(1)).astype(int)
    
    # Evaluate Amount Traded
    df['AmountTraded'] = df['close'] * df['volume']
    df['Abnormal_Trading_Amount'] = (df['AmountTraded'] > 2 * df['AmountTraded'].rolling(window=5).mean().shift(1)).astype(int)
    
    # Aggregate Alpha Factors
    df['Alpha_Factor'] = df['Price_Trend'] + df['Volume_Spike'] + df['Abnormal_Trading_Amount']
    
    return df['Alpha_Factor']
```
