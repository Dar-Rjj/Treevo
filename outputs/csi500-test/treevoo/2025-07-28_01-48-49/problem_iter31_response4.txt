```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high = df['high']
    intraday_low = df['low']
    
    # Calculate Intraday Range
    intraday_range = intraday_high - intraday_low
    
    # Adjust by Volume Change and Average Volume
    volume_20_ma = df['volume'].rolling(window=20).mean()
    current_volume_change = df['volume'].diff()
    adjusted_intraday_range = (intraday_range / volume_20_ma).fillna(0)
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_intraday_range_by_price = adjusted_intraday_range * price_level_ratio
    
    # Incorporate Close Price Momentum
    close_momentum_10 = df['close'].pct_change(periods=10).fillna(0)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean().fillna(0)
    
    # Combine Factors
    combined_intraday_momentum = adjusted_intraday_range_by_price + close_momentum_10
    high_low_spread_momentum = (df['high'] - df['low']).pct_change(periods=10).fillna(0)
    combined_factor = combined_intraday_momentum * high_low_spread_momentum
    
    # Filter for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].rolling(window=5).mean().fillna(0)
    
    # Introduce Volume Trend
    volume_5_sum = df['volume'].rolling(window=5).sum().fillna(0)
    volume_ratio = df['volume'] / volume_5_sum
    combined_factor_with_volume_trend = combined_factor * volume_ratio
    
    # Adjust Momentum by Intraday Volatility
    integrated_momentum_factor = combined_factor_with_volume_trend * intraday_volatility
    
    # Calculate Dynamic Adjustment Factor
    volume_ema_n = df['volume'].ewm(span=20, adjust=False).mean().fillna(0)
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = integrated_momentum_factor * volume_ema_n
    
    # Calculate Open Price Impact
    relative_open_price_change = df['open'] / df['open'].shift(1).fillna(1)
    
    # Final Alpha Factor
    final_alpha_factor = 0.80 * dynamically_adjusted_momentum + 0.20 * relative_open_price_change
    
    # Enhance with Volume Trend
    volume_5_ma = df['volume'].rolling(window=5).mean().fillna(0)
    volume_trend = df['volume'] - volume_5_ma
    enhanced_final_alpha_factor = final_alpha_factor * volume_trend
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    alpha_factor_ema_n = enhanced_final_alpha_factor.ewm(span=20, adjust=False).mean().fillna(0)
    final_alpha_factor = enhanced_final_alpha_factor - alpha_factor_ema_n
    
    return final_alpha_factor
```
