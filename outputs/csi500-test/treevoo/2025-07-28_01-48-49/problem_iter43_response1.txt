```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    price_level_adjustment = df['close'] / df['open']
    adjusted_range = intraday_range * volume_change * price_level_adjustment
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = adjusted_range * volume_change
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    close_t = df['close']
    close_t_minus_5 = df['close'].shift(5)
    five_day_momentum = (close_t - close_t_minus_5) / close_t_minus_5
    high_low_spread_momentum = high_low_spread_5d * five_day_momentum
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    
    # Combine Factors with Volume-Adjusted Price Trend
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_weighted_average_price = (df['amount'] / df['volume']).shift(1)
    volume_adjusted_trend = (close_to_close_return * intraday_range) / volume_weighted_average_price
    
    # Introduce Volume Trend
    five_day_sum_volume = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / five_day_sum_volume
    volume_trend_adjusted_factor = (integrated_momentum_factor + volume_adjusted_trend) * volume_ratio
    
    # Utilize High and Low Prices for Volatility Measures
    daily_range = df['high'] - df['low']
    relative_range = daily_range / df['close']
    average_daily_range = daily_range.rolling(window=20).mean()
    volatility_spike = (daily_range > 1.5 * average_daily_range).astype(int)
    
    # Generate Complex Alpha Factors from Basic Components
    momentum_volatility_score = integrated_momentum_factor * (1 + volatility_spike)
    comprehensive_alpha_factor = (volume_adjusted_trend * volume_trend_adjusted_factor) + (relative_range * volume_ratio)
    
    # Final Factor Calculation
    adjusted_range_weight = 0.2
    five_day_momentum_weight = 0.3
    integrated_momentum_factor_weight = 0.2
    volume_trend_adjusted_factor_weight = 0.3
    final_factor = (
        adjusted_range * adjusted_range_weight +
        five_day_momentum * five_day_momentum_weight +
        integrated_momentum_factor * integrated_momentum_factor_weight +
        volume_trend_adjusted_factor * volume_trend_adjusted_factor_weight
    )
    
    return final_factor
```
