import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['IntradayMovement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['CloseDiff'] = df['close'] - df['close'].shift(1)
    df['WeightedIntradayMovement'] = df['IntradayMovement'] * df['volume'] * df['CloseDiff']
    
    # Integrate Intraday Volatility
    df['DailyRange'] = df['high'] - df['low']
    df['IntradayVolatility'] = df['DailyRange'].ewm(span=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['ReversalSignal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['VolumeWeightedReturn'] = df['volume'] * df['ReversalSignal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['WeightedIntradayReturn'] = (df['amount'] * df['IntradayMovement']) + (df['volume'] * df['ReversalSignal'])
    
    # Combine Enhanced Factors
    df['CombinedFactors'] = df['WeightedIntradayMovement'] + df['IntradayVolatility'] + df['VolumeWeightedReturn']
    
    # Adjust by Relative Volume
    df['RelativeVolume'] = df['volume'] / df['volume'].rolling(window=40).mean()
    df['AdjustedCombinedFactors'] = df['CombinedFactors'] / df['RelativeVolume']
    
    # Incorporate Close Price Momentum
    df['Momentum30Day'] = df['close'].rolling(window=30).mean()
    df['Momentum10DayEMA'] = df['close'].ewm(span=10).mean()
    df['Momentum'] = (df['Momentum10DayEMA'] - df['Momentum30Day']) * df['IntradayVolatility']
    
    # Construct Momentum Factor
    df['Momentum5Day'] = df['close'].pct_change(5)
    df['Momentum20Day'] = df['close'].pct_change(20)
    df['Momentum60Day'] = df['close'].pct_change(60)
    df['MomentumFactor'] = (df['Momentum5Day'] + df['Momentum20Day'] + df['Momentum60Day']) / 3
    
    # Design Volatility Measures
    df['LogReturns'] = np.log(df['close'] / df['close'].shift(1))
    df['Volatility5Day'] = df['LogReturns'].rolling(window=5).std()
    df['Volatility20Day'] = df['LogReturns'].rolling(window=20).std()
    df['Volatility60Day'] = df['LogReturns'].rolling(window=60).std()
    df['VolatilityMeasures'] = (df['Volatility5Day'] + df['Volatility20Day'] + df['Volatility60Day']) / 3
    
    # Incorporate Volume-Based Indicators
    df['VolumeRatio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Explore Price Action Patterns (simplified example: bullish engulfing pattern)
    df['BullishEngulfing'] = ((df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1)) & (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))).astype(int)
    df['PricePatternScore'] = df['BullishEngulfing'].rolling(window=10).sum()
    
    # Develop Trend-Following Signals
    df['TrendFollowingSignal'] = np.where(df['close'] > df['close'].rolling(window=20).mean(), 1, -1)
    
    # Create Liquidity-Related Factors
    df['Spread'] = df['high'] - df['low']
    
    # Examine Market Sentiment
    df['UpDays'] = (df['close'] > df['close'].shift(1)).astype(int)
    df['DownDays'] = (df['close'] < df['close'].shift(1)).astype(int)
    df['SentimentScore'] = (df['UpDays'] - df['DownDays']).rolling(window=20).sum()
