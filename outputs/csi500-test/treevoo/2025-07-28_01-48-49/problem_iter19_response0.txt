```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    moving_avg_vol = df['volume'].rolling(window=20).mean()
    vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_range = intraday_range * vol_change / moving_avg_vol
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_price_range = adjusted_range * price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).sum() / 5
    combined_high_low_spread = high_low_spread_5d * momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Calculate Volume Weighted Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Final Factor Calculation
    factor = (adjusted_price_range * 0.25 +
              momentum * 0.25 +
              combined_high_low_spread * 0.25 +
              avg_daily_range * 0.15 +
              volume_weighted_return * 0.1)
    
    return factor
```
