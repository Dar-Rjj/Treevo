import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    df['VolumeChange'] = df['volume'].pct_change()
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['AdjustedRange'] = df['IntradayRange'] * df['VolumeChange'] * df['PriceLevelRatio']
    
    # Close Price Momentum
    lookback_period = 10
    df['Momentum'] = (df['close'] / df['close'].shift(lookback_period)) - 1
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['5DayAvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    df['5DayCloseMomentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayCloseMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Calculate Volume Weighted Average Price
    df['VWAP'] = ((df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = df['close'].pct_change()
    df['CombinedFactor'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VWAP']
    
    # Filter for Recent Price Trend
    df['5DayMA'] = df['close'].rolling(window=5).mean()
    df['RecentPriceTrend'] = df['close'] - df['5DayMA']
    
    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['CombinedFactor'] * df['VolumeRatio']
    
    # Final Factor Calculation
    adjusted_range_weight = 0.2
    momentum_weight = 0.3
    integrated_momentum_weight = 0.2
    volume_trend_weight = 0.1
    rsi_weight = 0.1
    mfi_weight = 0.1
    
    df['FinalFactor'] = (adjusted_range_weight * df['AdjustedRange'] +
                         momentum_weight * df['Momentum'] +
                         integrated_momentum_weight * df['IntegratedMomentumFactor'] +
                         volume_trend_weight * df['VolumeTrendAdjustedFactor'])
    
    # Introduce Relative Strength Index (RSI) and Money Flow Index (MFI)
    rsi_lookback = 14
    mfi_lookback = 14
    
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_lookback).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_lookback).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_flow_sum = positive_flow.rolling(window=mfi_lookback).sum()
    negative_flow_sum = negative_flow.rolling(window=mfi_lookback).sum()
    money_ratio = positive_flow_sum / negative_flow_sum
    df['MFI'] = 100 - (100 / (1 + money_ratio))
    
    df['FinalFactor'] += (rsi_weight * df['RSI'] + mfi_weight * df['MFI'])
    
    return df['FinalFactor'].dropna()
