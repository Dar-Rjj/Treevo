def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    - pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the relative range of the stock price
    data['range'] = data['high'] - data['low']
    
    # Calculate the relative position of the closing price within the daily range
    data['relative_position'] = (data['close'] - data['low']) / data['range']
    
    # Calculate the ratio of the closing price to the opening price
    data['close_to_open_ratio'] = data['close'] / data['open']
    
    # Calculate the ratio of the trading amount to the volume
    data['amount_to_volume_ratio'] = data['amount'] / data['volume']
    
    # Combine the factors into a single alpha factor
    data['alpha_factor'] = (data['relative_position'] * data['close_to_open_ratio']) + data['amount_to_volume_ratio']
    
    return data['alpha_factor']
