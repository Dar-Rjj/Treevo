```python
def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Generate IntradayMomentumScore
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    df['IntradayMomentumScore'] = df['IntradayMomentumScore'].apply(lambda x: max(x, 0))
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['5DayAvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # 5-Day Momentum
    df['5DayMomentum'] = (df['open'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # HighLowSpreadMomentum
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayMomentum']
    
    # Finalize Alpha Factor
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    return df['IntegratedMomentumFactor']
```
