```python
import pandas as pd

def heuristics_v2(df):
    # 5-Day Average High-Low Spread
    high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # 5-Day Momentum
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    hl_momentum = high_low_spread * momentum_5d
    
    # Enhanced Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    intraday_weighted = intraday_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = intraday_weighted * abs(df['volume'] * close_diff)
    intraday_momentum_score = intraday_momentum_score.clip(lower=0)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount'] * intraday_movement
    weighted_intraday_return = (intraday_return_weight + (df['volume'] * reversal_signal))
    
    # Combine Enhanced Factors
    combined_factors = (0.3 * intraday_momentum_score) + (0.3 * intraday_volatility) + (0.4 * hl_momentum) * volume_weighted_return
    
    # Final Factor Calculation
    final_factor = (0.5 * combined_factors) + (0.5 * weighted_intraday_return)
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_factor_adjusted = final_factor * price_level_ratio
    
    # Enhance with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - avg_volume_10d
    final_factor_enhanced = final_factor_adjusted * volume_trend
    
    # Adjust by Average Volume
    avg_volume_15d = df['volume'].rolling(window=15).mean()
    final_factor_adjusted_by_volume = final_factor_enhanced / avg_volume_15d
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    ema_20d = final_factor_adjusted_by_volume.ewm(span=20, adjust=False).mean()
    final_alpha_factor = final_factor_adjusted_by_volume - ema_20d
    
    return final_alpha_factor
```
