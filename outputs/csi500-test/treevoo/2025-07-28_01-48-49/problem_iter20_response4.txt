import pandas as pd

def heuristics_v2(df):
    # 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # 5-Day Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum_product = avg_high_low_spread * momentum
    
    # Enhanced Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_price_movement = intraday_price_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = weighted_intraday_price_movement.abs()
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    close_open_reversal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * close_open_reversal
    
    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return = (df['amount'] * intraday_price_movement) + (df['volume'] * close_open_reversal)
    
    # Combine Enhanced Factors
    combined_factors = (intraday_momentum_score * 0.3) + (avg_daily_range * 0.3) + (momentum * 0.4)
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_combined_factors = combined_factors / avg_volume
    
    # Incorporate Close Price Momentum
    close_momentum = df['close'] / df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_close_momentum = close_momentum * avg_daily_range
    
    # Final Factor Calculation
    final_factor = (adjusted_combined_factors * 0.5) + (weighted_intraday_return * 0.5)
    
    # Introduce Volume Trend
    volume_sum_5_days = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5_days
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_factor_adjusted = final_factor * price_level_ratio
    
    # Final Factor Computation
    volume_trend_adjusted_factor = final_factor_adjusted * volume_ratio
    
    return volume_trend_adjusted_factor.dropna()
