```python
def heuristics_v2(df):
    # Calculate Price Range: Daily High - Daily Low
    df['PriceRange'] = df['high'] - df['low']
    
    # Examine Opening and Closing Gaps: Open - Close of Previous Day
    df['GapOpenClosePrevDay'] = df['open'] - df['close'].shift(1)
    
    # Analyze Volume-Price Interaction: (High + Low) / 2 * Volume
    df['AverageTransactionValue'] = ((df['high'] + df['low']) / 2) * df['volume']
    
    # Measure Trading Activity: Volume / (High - Low) if High != Low else 0
    df['VolumeToPriceVolatilityRatio'] = df.apply(lambda row: row['volume'] / (row['high'] - row['low']) if row['high'] != row['low'] else 0, axis=1)
    
    # Simple Moving Averages
    df['SMA5D'] = df['close'].rolling(window=5).mean()
    df['SMA20D'] = df['close'].rolling(window=20).mean()
    
    # Compare Short-Term and Long-Term Averages: SMA5D - SMA20D
    df['MomentumSignal'] = df['SMA5D'] - df['SMA20D']
    
    # Absolute Daily Price Movement: Abs(Close - Open)
    df['DailyPriceMovement'] = abs(df['close'] - df['open'])
    
    # Relative Daily Price Movement: (Close - Open) / Open if Open != 0 else 0
    df['RelativeDailyPriceMovement'] = df.apply(lambda row: (row['close'] - row['open']) / row['open'] if row['open'] != 0 else 0, axis=1)
    
    # Aggregate Multiple Factors
    df['CombinedAlphaFactor'] = (df['PriceRange'] + 
                                 df['GapOpenClosePrevDay'] + 
                                 df['VolumeToPriceVolatilityRatio'] + 
                                 df['MomentumSignal'] + 
                                 df['DailyPriceMovement'] + 
                                 df['RelativeDailyPriceMovement'])
    
    return df['CombinedAlphaFactor']
```
