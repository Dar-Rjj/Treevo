```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['intraday_price_movement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    df['weighted_intraday_price_movement'] = df['intraday_price_movement'] * df['volume'] * close_diff
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['volatility_20day_ma'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_return'] = df['amount'] * df['intraday_price_movement']
    df['weighted_reversal_signal'] = df['volume'] * df['reversal_signal']
    df['weighted_intraday_return_and_reversal'] = df['weighted_intraday_return'] + df['weighted_reversal_signal']
    
    # Combine Enhanced Factors
    df['combined_enhanced_factors'] = df['weighted_intraday_price_movement'] + df['volatility_20day_ma']
    df['volume_weighted_return'] = df['volume'] * (df['close'] - df['close'].shift(1))
    df['combined_factors'] = (df['combined_enhanced_factors'] * df['volume_weighted_return'])
    
    # Final Factor Calculation
    df['final_factor'] = (df['combined_factors'] * 0.6) + (df['weighted_intraday_return_and_reversal'] * 0.4)
    
    return df['final_factor']
```
