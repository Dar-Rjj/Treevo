import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate 5-Day Average Intraday Price Range
    avg_5day_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum using N-day Moving Average of Close price
    momentum = df['close'].pct_change(n)
    
    # Adjust Momentum by 5-Day Average Intraday Price Range
    adjusted_momentum = momentum * avg_5day_intraday_range
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum - adjusted_momentum.rolling(window=n).mean()

    return final_alpha_factor
