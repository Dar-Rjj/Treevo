```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return factors
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Explore price-based alpha factors
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Analyze trading activity with volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_close_ratio'] = df['volume'] / df['close']
    
    # Investigate momentum and trend following
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    df['short_ma_diff'] = df['close'] - df['short_ma']
    df['long_ma_diff'] = df['close'] - df['long_ma']
    
    # Examine volatility and risk
    df['volatility'] = df['daily_return'].rolling(window=10).std()
    
    # Consider breakout and reversal patterns
    n = 5
    df['breakout_high'] = (df['close'] > df['high'].rolling(window=n).max()).astype(int)
    df['breakout_low'] = (df['close'] < df['low'].rolling(window=n).min()).astype(int)
    
    # Evaluate the relationship between amount and volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    df['amount_volume_change'] = df['amount_volume_ratio'] - df['amount_volume_ratio'].shift(1)
    
    # Analyze the impact of gaps
    df['up_gap'] = (df['open'] > df['close'].shift(1)).astype(int)
    df['down_gap'] = (df['open'] < df['close'].shift(1)).astype(int)
    
    # Synthesize Alpha Factor Combining Intraday Dynamics, Volume-Adjusted Range, and Hybrid Momentum-Reversal
    df['intraday_price_movement'] = df['high'] - df['low']
    df['adj_intraday_price_movement'] = df['intraday_price_movement'] / (df['open'] - df['open'].shift(1))
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['weighted_adj_intraday_price_movement'] = df['adj_intraday_price_movement'] * df['volume'] * df['close_diff']
    
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=20).mean()
    
    df['close_open_reversal'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = df['volume'] * df['close_open_reversal']
    
    df['intraday_return_weight'] = df['amount']
    df['reversal_signal_weight'] = df['volume']
    df['weighted_intraday_return_and_reversal'] = (df['intraday_return_weight'] * df['weighted_adj_intraday_price_movement']) + (df['reversal_signal_weight'] * df['volume_weighted_return'])
    
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_factor'] = (df['weighted_adj_intraday_price_movement'] + df['avg_daily_range'] + df['volume_weighted_return']) / df['avg_volume']
    
    df['momentum'] = df['close'].rolling(window=10).mean()
    df['adjusted_momentum'] = df['momentum'] * df['volatility']
    
    df['final_alpha_factor'] = (0.3 * df['weighted_adj_intraday_price_movement'] + 0.3 * df['avg_daily_range'] + 0.4 * df['adjusted_momentum']) * 0.5 + df['weighted_intraday_return_and_reversal'] * 0.5
    
    return df['final_alpha_factor']
```
