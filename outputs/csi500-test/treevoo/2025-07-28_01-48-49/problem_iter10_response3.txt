def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = (intraday_range * volume_change).apply(lambda x: max(x, 0))
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    five_day_momentum = (df['open'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply 5-Day Average High-Low Spread by 5-Day Momentum
    high_low_spread_momentum = high_low_spread_5d * five_day_momentum
    
    # Add IntradayMomentumScore to HighLowSpreadMomentum
    integrated_momentum_factor = high_low_spread_momentum + intraday_momentum_score
    
    return integrated_momentum_factor
