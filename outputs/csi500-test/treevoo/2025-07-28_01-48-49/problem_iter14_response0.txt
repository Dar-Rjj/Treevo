```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Price Movement
    df['enhanced_intraday_price_movement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['intraday_weighted'] = df['enhanced_intraday_price_movement'] * (df['volume'] * df['close_diff'])
    
    # Generate IntradayMomentumScore
    df['intraday_momentum_score'] = (df['enhanced_intraday_price_movement'] * 
                                     abs(df['volume'] * df['close_diff']))
    df['intraday_momentum_score'] = df['intraday_momentum_score'].apply(lambda x: max(0, x))
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['volume_weighted_return'] = df['volume'] * df['reversal_signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_return'] = (df['amount'] * df['intraday_weighted']) + (df['volume'] * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_enhanced_factors'] = (df['intraday_momentum_score'] * df['volume_weighted_return']) * df['intraday_volatility']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_combined_factors'] = df['combined_enhanced_factors'] / df['avg_volume']
    
    # Incorporate Close Price Momentum
    df['close_momentum'] = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['close_momentum'] * df['intraday_volatility']
    
    # Final Factor Calculation
    df['final_alpha_factor'] = (0.3 * df['intraday_momentum_score'] + 
                                 0.3 * df['intraday_volatility'] + 
                                 0.4 * df['adjusted_momentum'])
    df['final_alpha_factor'] = (0.5 * df['final_alpha_factor'] + 
                                0.5 * df['weighted_intraday_return'])
    
    return df['final_alpha_factor']
```
