```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Calculate Volume Weighted Average Price
    vwap = (df[['open', 'high', 'low', 'close']].sum(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Factors: Multiply Close-to-Close Return by Intraday Range, then divide by VWAP
    combined_factor = (close_to_close_return * intraday_range) / vwap
    
    # Filter for Recent Price Trend: 5-Day Moving Average of Close Prices
    ma_5_day = df['close'].rolling(window=5).mean()
    recent_price_trend = df['close'] - ma_5_day
    
    # Introduce Volume Trend: 5-Day Sum of Volume
    volume_5_day_sum = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_5_day_sum
    
    # Final Factor: Multiply Combined Factor by Volume Ratio and adjust with recent price trend
    final_factor = combined_factor * volume_ratio + recent_price_trend
    
    return final_factor
```
