```python
def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Adjust by Average Volume
    lookback_period = 20
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['adjusted_range'] = df['price_range'] / df['avg_volume']
    
    # Incorporate Close Price Momentum
    momentum_lookback_period = 10
    df['momentum'] = df['close'].pct_change(momentum_lookback_period)
    
    # Final Factor Calculation
    adjusted_range_weight = 0.6
    momentum_weight = 0.4
    df['VARI'] = (df['adjusted_range'] * adjusted_range_weight) + (df['momentum'] * momentum_weight)
    
    return df['VARI']
```
