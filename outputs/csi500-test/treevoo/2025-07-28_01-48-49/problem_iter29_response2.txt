def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the intraday range as a percentage of the open price
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate the relative close position within the intraday range
    rel_close_pos = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the buying pressure as a ratio of volume to amount
    buying_pressure = df['volume'] / df['amount']
    
    # Combine the factors to create the final alpha factor
    alpha_factor = (intraday_range + rel_close_pos) * buying_pressure
    
    return alpha_factor
