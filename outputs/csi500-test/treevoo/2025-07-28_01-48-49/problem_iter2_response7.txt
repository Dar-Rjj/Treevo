import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['amount']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=5).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return = (df['amount'] * intraday_return) + (df['volume'] * reversal_signal)
    
    # Final Alpha Factor
    alpha_factor = (adjusted_momentum * 0.5) + (weighted_intraday_return * 0.5)
    
    return alpha_factor
