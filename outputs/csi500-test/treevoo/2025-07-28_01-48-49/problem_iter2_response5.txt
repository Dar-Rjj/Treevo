```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_difference = df['open'] - df['open'].shift(1)
    
    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_to_low = high_to_low_movement / open_difference
    
    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_adjusted_high_to_low = adjusted_high_to_low * (df['volume'] * close_difference)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = df['close'] / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Combine Enhanced Factors
    final_factor = weighted_adjusted_high_to_low * avg_daily_range * volume_weighted_return
    
    return final_factor
```
