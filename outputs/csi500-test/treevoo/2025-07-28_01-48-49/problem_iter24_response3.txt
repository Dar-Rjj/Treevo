```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_price_range = (intraday_price_movement * volume_change) / moving_average_volume
    
    # Apply Enhanced Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    smoothing_parameter = 1.5
    sigmoid_function = 1 / (1 + np.exp(-smoothing_parameter * (price_level_ratio - 1)))
    adjusted_range = adjusted_price_range * sigmoid_function
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    close_momentum = df['close'] / df['close'].shift(momentum_lookback) - 1
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_5_days = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread_10_days = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Combine 5-Day Average High-Low Spread and Momentum
    combined_spread_momentum = avg_high_low_spread_5_days * close_momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=5).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    highlowspreadmomentum = avg_high_low_spread_5_days * five_day_momentum
    
    # Combine Intraday and High-Low Spread Momentum
    integrated_momentum_factor = adjusted_range + highlowspreadmomentum
    
    # Calculate Volume Weighted Average Price
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Combine Factors with Volume-Adjusted Price Trend
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_factor = close_to_close_return * adjusted_range / vwap
    
    # Filter for Recent Price Trend
    ma_close_5_days = df['close'].rolling(window=5).mean()
    recent_price_trend = df['close'] - ma_close_5_days
    
    # Introduce Volume Trend
    sum_volume_5_days = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / sum_volume_5_days
    volume_trend_adjusted_factor = combined_factor * volume_ratio
    
    # Final Factor Calculation
    final_factor = (
        0.2 * adjusted_range +
        0.3 * close_momentum +
        0.3 * integrated_momentum_factor +
        0.2 * volume_trend_adjusted_factor
    )
    
    return final_factor
```
