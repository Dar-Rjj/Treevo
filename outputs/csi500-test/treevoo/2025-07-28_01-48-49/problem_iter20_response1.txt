```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['adj_intraday_move'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['weighted_adj_intraday_move'] = df['adj_intraday_move'] * df['volume'] * df['close_diff']
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['vol_weighted_return'] = df['volume'] * df['reversal_signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_return_reversal'] = (df['amount'] * df['weighted_adj_intraday_move']) + (df['volume'] * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_factors'] = (df['weighted_adj_intraday_move'] * df['avg_daily_range_20']) * df['vol_weighted_return']
    
    # Adjust by Average Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_combined_factors'] = df['combined_factors'] / df['avg_volume_20']
    
    # Incorporate Close Price Momentum
    df['close_momentum_10'] = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['close_momentum_10'] * df['avg_daily_range_20']
    
    # Final Factor Calculation
    df['final_factor'] = (0.3 * df['weighted_adj_intraday_move'] + 
                          0.3 * df['avg_daily_range_20'] + 
                          0.4 * df['adjusted_momentum'])
    df['alpha_factor'] = (df['final_factor'] * 0.5) + (df['weighted_intraday_return_reversal'] * 0.5)
    
    return df['alpha_factor']
```
