```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    df['5DayAvgIntradayRange'] = df['IntradayRange'].rolling(window=5).mean()
    
    # Calculate 5-Day Return
    df['5DayReturn'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Determine Intraday Volatility Adjusted Momentum
    df['IntradayVolAdjMomentum'] = df['5DayReturn'] * df['5DayAvgIntradayRange']
    
    # Compute Intraday Reversal Signal
    df['IntradayReversal'] = (df['close'] - df['open']) / df['5DayAvgIntradayRange']
    
    # Incorporate Volume Impact
    df['5DayCumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeWeight'] = df['volume'] / df['5DayCumVolume']
    
    # Final Alpha Factor Synthesis
    df['CombinedFactor'] = (df['IntradayReversal'] + df['IntradayVolAdjMomentum']) * df['VolumeWeight']
    
    # Generate Momentum Score
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['volume'].pct_change()
    
    # Generate High-Low Spread Momentum
    df['HighLowSpreadMomentum'] = df['5DayAvgIntradayRange'] * df['5DayReturn']
    
    # Combine Intraday and High-Low Spread Momentum
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = df['close'] / df['close'].shift(1) - 1
    df['VolumeWtdAvgPrice'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VolumeAdjPriceTrend'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VolumeWtdAvgPrice']
    
    # Filter for Recent Price Trend
    df['5DayMovAvgClose'] = df['close'].rolling(window=5).mean()
    df['RecentPriceTrend'] = df['close'] - df['5DayMovAvgClose']
    
    # Introduce Volume Trend
    df['VolumeRatio'] = df['volume'] / df['volume'].rolling(window=5).sum()
    df['VolumeTrendAdjustedFactor'] = df['CombinedFactor'] * df['VolumeRatio']
    
    # Incorporate Close Price Momentum
    df['ClosePriceMomentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['VolumeTrendAdjustedFactor'] + df['IntegratedMomentumFactor'] + 
                              df['VolumeAdjPriceTrend'] + df['RecentPriceTrend'] + df['ClosePriceMomentum'])
    
    return df['FinalAlphaFactor'].dropna()
```
