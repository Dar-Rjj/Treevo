```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_price_difference = df['open'].diff()
    open_adjustment_factor = open_price_difference
    
    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_to_low = high_to_low_movement / open_adjustment_factor
    
    # Weight by Volume and Close Difference
    close_difference = df['close'].diff()
    weighted_adjustment = adjusted_high_to_low * (df['volume'] * close_difference)
    
    return weighted_adjustment.dropna()
```
