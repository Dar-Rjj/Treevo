```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return factors
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Explore price-based alpha factors
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Analyze trading activity with volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_close_ratio'] = df['volume'] / df['close']
    
    # Investigate momentum and trend following
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    df['short_ma_diff'] = df['close'] - df['short_ma']
    df['long_ma_diff'] = df['close'] - df['long_ma']
    
    # Examine volatility and risk
    df['return_volatility'] = df['daily_return'].rolling(window=10).std()
    
    # Consider breakout and reversal patterns
    df['above_prev_high'] = df['close'] > df['high'].rolling(window=5).max()
    df['below_prev_low'] = df['close'] < df['low'].rolling(window=5).min()
    
    # Evaluate the relationship between amount and volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    df['amount_volume_ratio_change'] = df['amount_volume_ratio'] - df['amount_volume_ratio'].shift(1)
    
    # Analyze the impact of gaps
    df['up_gap'] = df['open'] > df['close'].shift(1)
    df['down_gap'] = df['open'] < df['close'].shift(1)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['daily_return'] +
        df['high_low_diff'] * 0.5 +
        df['close_open_ratio'] * 0.3 +
        df['volume_change'] * 0.7 +
        df['volume_close_ratio'] * 0.4 +
        df['short_ma_diff'] * 0.6 +
        df['long_ma_diff'] * 0.8 +
        df['return_volatility'] * -0.5 +
        df['above_prev_high'] * 0.9 +
        df['below_prev_low'] * -0.9 +
        df['amount_volume_ratio_change'] * 0.2 +
        df['up_gap'] * 0.8 +
        df['down_gap'] * -0.8
    )
    
    return alpha_factor
```
