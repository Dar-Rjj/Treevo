```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative spread between high and low prices
    relative_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative difference between close and open prices
    relative_difference = (df['close'] - df['open']) / df['close']
    
    # Calculate the volume-weighted amount
    volume_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the factor value by combining the above metrics
    factor_value = relative_spread * relative_difference * volume_weighted_amount
    
    return factor_value
```
