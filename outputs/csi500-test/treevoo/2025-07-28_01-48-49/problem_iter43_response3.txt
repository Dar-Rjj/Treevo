```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_range = intraday_range * volume_change * (df['close'] / df['open'])
    
    # Integrate Intraday Price Movement
    adjusted_intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    weighted_intraday_price_movement = adjusted_intraday_price_movement * df['volume'] * (df['close'] - df['close'].shift(1))
    
    # IntradayMomentumScore
    intraday_momentum_score = intraday_range * volume_change
    
    # 5-Day Average High-Low Spread
    high_low_spread_5day = (df['high'] - df['low']).rolling(window=5).mean()
    
    # 5-Day Momentum
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # HighLowSpreadMomentum
    high_low_spread_momentum = high_low_spread_5day * five_day_momentum
    
    # IntegratedMomentumFactor
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    
    # Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Weighted Intraday Return and Reversal Signal
    weighted_return_reversal = (df['amount'] * adjusted_intraday_price_movement) + (df['volume'] * reversal_signal)
    
    # Volume Trend
    five_day_sum_volume = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / five_day_sum_volume
    
    # VolumeTrendAdjustedFactor
    volume_trend_adjusted_factor = integrated_momentum_factor * volume_ratio
    
    # Final Factor Calculation
    final_factor = (
        0.2 * adjusted_range +
        0.3 * five_day_momentum +
        0.2 * integrated_momentum_factor +
        0.1 * volume_trend_adjusted_factor +
        0.2 * weighted_return_reversal
    )
    
    return final_factor
```
