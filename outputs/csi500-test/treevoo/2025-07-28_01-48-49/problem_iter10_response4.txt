```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['5Day_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=5).mean()

    # Calculate 5-Day Momentum
    df['5Day_Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply High-Low Spread and Momentum
    df['Spread_Momentum'] = df['5Day_High_Low_Spread'] * df['5Day_Momentum']

    # Calculate Close to Close Return
    df['Close_Return'] = df['close'] / df['close'].shift(1) - 1

    # Calculate Volume Weighted Average Price
    df['VWAP'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()

    # Filter for Recent Price Trend
    df['5Day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['Price_Trend'] = df['close'] - df['5Day_MA_Close']

    # Introduce Volume Trend
    df['5Day_Volume_Sum'] = df['volume'].rolling(window=5).sum()
    df['Volume_Ratio'] = df['volume'] / df['5Day_Volume_Sum']

    # Final Factor Computation
    df['Combined_Factor'] = (df['Close_Return'] * df['High_Low_Spread'] / df['VWAP']) * df['Volume_Ratio'] * df['Spread_Momentum']

    return df['Combined_Factor']
```
