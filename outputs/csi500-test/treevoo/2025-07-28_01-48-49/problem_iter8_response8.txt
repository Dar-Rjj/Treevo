import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_5_day_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_10_day_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['volume']
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Final Alpha Component 1
    alpha_component_1 = (avg_5_day_high_low_spread * adjusted_momentum) + \
                        (ema_10_day_high_low_spread + (df['close'] / df['close'].shift(20) - 1))
    
    # Final Alpha Component 2
    n_day_ma_adjusted_momentum = adjusted_momentum.rolling(window=20).mean()
    alpha_component_2 = adjusted_momentum - n_day_ma_adjusted_momentum
    
    # Combine both components to form the final alpha factor
    final_alpha = alpha_component_1 + alpha_component_2
    
    return final_alpha
