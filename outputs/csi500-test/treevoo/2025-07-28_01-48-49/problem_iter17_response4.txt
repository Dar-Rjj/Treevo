```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices using volume
    weighted_avg = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate the difference between close and open price
    diff_close_open = df['close'] - df['open']
    
    # Compute the factor as the sum of weighted average and the difference between close and open
    factor = (weighted_avg + diff_close_open) / df['amount']
    
    return factor
```
