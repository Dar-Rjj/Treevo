```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_difference = df['open'].diff()
    
    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_to_low = high_to_low_movement / open_difference
    
    # Weight by Volume and Close Difference
    close_difference = df['close'].diff()
    weighted_adjusted_high_to_low = adjusted_high_to_low * (df['volume'] * close_difference)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=5).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = df['close'] / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Calculate Volume Weighted Average Price
    vwap = ((df[['open', 'high', 'low', 'close']].sum(axis=1) / 4) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Factors
    combined_factors = (close_to_close_return * intraday_range) / vwap
    
    # Filter for Recent Trend
    recent_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = (weighted_adjusted_high_to_low + avg_daily_range + volume_weighted_return + combined_factors) * recent_trend
    
    return alpha_factor
```
