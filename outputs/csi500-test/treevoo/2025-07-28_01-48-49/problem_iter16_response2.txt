```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Current Day Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Volume-Adjusted Range
    adjusted_range = intraday_range * volume_change
    
    # Price Level Ratio
    price_level_ratio = df['close'] / df['open']
    
    # Adjusted Range * Price Level Ratio
    adjusted_range *= price_level_ratio
    
    # Close Price Momentum
    momentum_lookback_period = 10
    momentum = df['close'] - df['close'].shift(momentum_lookback_period)
    
    # Final Factor Calculation
    adjusted_range_weight = 0.6
    momentum_weight = 0.4
    final_factor = (adjusted_range * adjusted_range_weight) + (momentum * momentum_weight)
    
    return final_factor
```
