```python
import pandas as pd

def heuristics_v2(df, window=10):
    # Calculate High-Low Return
    high_low_return = df['high'] - df['low']
    
    # Calculate High-Low Momentum over a window
    high_low_momentum = high_low_return.rolling(window=window).sum()
    
    # Calculate Volume Weight over a window
    volume_weight = df['volume'].rolling(window=window).sum()
    
    # Combine High-Low Momentum with Volume Weight
    factor_values = high_low_momentum * volume_weight
    
    return factor_values
```
