```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low range and the close-open range
    hl_range = df['high'] - df['low']
    co_range = df['close'] - df['open']
    
    # Calculate the volume weighted by amount
    vol_weighted = df['volume'] * (df['amount'] / df['volume'].sum())
    
    # Calculate a signal based on the relative position of close to open, adjusted by the volume
    alpha_factor = (co_range / hl_range) * vol_weighted
    
    return alpha_factor
```
