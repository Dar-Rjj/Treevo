import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['HL_Spread'] = df['high'] - df['low']
    df['5Day_Avg_HL_Spread'] = df['HL_Spread'].rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    df['5Day_Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    df['Spread_Momentum'] = df['5Day_Avg_HL_Spread'] * df['5Day_Momentum']
    
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['AdjustedRange'] = df['IntradayRange'] * df['VolumeChange'] * df['PriceLevelRatio']
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['IntradayRange'] * abs((df['volume'] * (df['close'] - df['open'])))
    df['IntradayMomentumScore'] = df['IntradayMomentumScore'].apply(lambda x: max(x, 0))
    
    # Generate Intraday Volatility
    df['DailyRange'] = df['high'] - df['low']
    df['20Day_MA_DailyRange'] = df['DailyRange'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['ReversalSignal'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['VWAP'] = (df['volume'] * df['close']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['CombinedFactor'] = df['CloseToCloseReturn'] * df['IntradayRange'] / df['VWAP']
    
    # Introduce Volume Trend
    df['5Day_Sum_Volume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5Day_Sum_Volume']
    df['VolumeTrendAdjustedFactor'] = df['CombinedFactor'] * df['VolumeRatio']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['WeightedIntradayReturn'] = df['amount'] / df['volume']
    df['WeightedReversalSignal'] = df['volume']
    df['WeightedSignal'] = (df['WeightedIntradayReturn'] * df['IntradayReturn']) + (df['WeightedReversalSignal'] * df['ReversalSignal'])
    
    # Combine Enhanced Factors
    df['EnhancedFactor'] = (0.3 * df['IntradayMomentumScore']) + (0.3 * df['20Day_MA_DailyRange']) + (0.4 * df['Spread_Momentum'])
    
    # Final Factor Calculation
    df['FinalAlphaFactor'] = (df['EnhancedFactor'] * 0.5) + (df['WeightedSignal'] * 0.5)
    
    # Apply Price Level Adjustment
    df['PriceLevelAdjustedFinalFactor'] = df['FinalAlphaFactor'] * df['PriceLevelRatio']
    
    # Enhance with Volume Trend
    df['5Day_MA_Volume'] = df['volume'].rolling(window=5).mean()
    df['VolumeTrend'] = df['volume'] - df['5Day_MA_Volume']
    df['VolumeTrendAdjustedFinalFactor'] = df['PriceLevelAdjustedFinalFactor'] * df['VolumeTrend']
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    df['EMA_FinalAlphaFactor'] = df['FinalAlphaFactor'].ewm(span=10).mean()
    df['FinalAlphaFactor'] -= df['EMA_FinalAlphaFactor']
    
    # Introduce Volume Sustained Trend
    df['10Day_Sum_Volume'] = df['volume'].rolling(window=10).sum()
    df['VolumeSustainedTrend'] = df['volume'] / df['10Day_Sum_Volume']
    df
