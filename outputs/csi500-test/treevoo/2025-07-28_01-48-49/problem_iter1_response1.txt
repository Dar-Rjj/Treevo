def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Combine Factors
    combined_factor = (close_to_close_return * intraday_range) / vwap
    
    # Filter for Recent Trend
    recent_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = combined_factor + recent_trend
    
    return alpha_factor
