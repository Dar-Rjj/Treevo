def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the signed volume
    df['signed_volume'] = df['volume'] * df['daily_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate the weighted price
    df['weighted_price'] = (df['open'] + 2*df['high'] + 2*df['low'] + df['close']) / 6
    
    # Calculate the moving average of the weighted price
    df['ma_weighted_price'] = df['weighted_price'].rolling(window=5).mean()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['weighted_price'] - df['ma_weighted_price']) * df['signed_volume']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
