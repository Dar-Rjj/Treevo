```python
import pandas as pd

def heuristics_v2(df):
    # 5-Day Average High-Low Spread
    df['HighLowSpread'] = (df['high'] - df['low']).rolling(window=5).mean()

    # 5-Day Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply High-Low Spread and Momentum
    df['EnhancedMomentum'] = df['HighLowSpread'] * df['Momentum']

    # Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']

    # Adjust by Volume Change and Price Level
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['AdjustedRange'] = df['IntradayRange'] * df['VolumeChange'] * df['PriceLevelRatio']

    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    df['IntradayMomentumScore'] = df['IntradayMomentumScore'].abs()  # Ensure non-negative

    # Intraday Volatility
    df['DailyRange'] = df['high'] - df['low']
    df['IntradayVolatility'] = df['DailyRange'].rolling(window=20).mean()

    # Close-to-Open Reversal Signal
    df['CloseOpenReversal'] = (df['close'] - df['open']) / df['open']

    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['VolumeWeightedAvgPrice'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['CombinedFactor'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VolumeWeightedAvgPrice']

    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['CombinedFactor'] * df['VolumeRatio']

    # Construct Weighted Intraday Return and Reversal Signal
    df['IntradayReturnWeight'] = df['amount']
    df['ReversalSignalWeight'] = df['volume']
    df['WeightedIntradayReturn'] = (df['IntradayReturnWeight'] * df['IntradayRange']) + (df['ReversalSignalWeight'] * df['CloseOpenReversal'])

    # Combine Enhanced Factors
    df['CombinedFactors'] = (0.3 * df['IntradayMomentumScore']) + (0.3 * df['IntradayVolatility']) + (0.4 * df['EnhancedMomentum'])

    # Final Factor Calculation
    df['FinalAlphaFactor'] = (0.5 * df['CombinedFactors']) + (0.5 * df['WeightedIntradayReturn'])

    # Apply Price Level Adjustment
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'] * df['PriceLevelRatio']

    # Enhance with Volume Trend
    df['5DayMovingAverageVolume'] = df['volume'].rolling(window=5).mean()
    df['VolumeTrend'] = df['volume'] - df['5DayMovingAverageVolume']
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'] * df['VolumeTrend']

    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    df['EMA_FinalAlphaFactor'] = df['FinalAlphaFactor'].ewm(span=20, adjust=False).mean()
    df['FinalAlphaFactor'] -= df['EMA_FinalAlphaFactor']

    return df['FinalAlphaFactor']
```
