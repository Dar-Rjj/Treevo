import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Difference
    df['CloseToOpenDiff'] = df['close'].shift(-1) - df['open']
    
    # Apply Weight by Volume
    df['VolumeWeightedDiff'] = df['CloseToOpenDiff'] * df['volume']
    
    # Calculate Sum of Weighted Differences over a 5-day rolling window
    df['RollingWeightedSum'] = df['VolumeWeightedDiff'].rolling(window=5).sum()
    
    # Calculate Simple Moving Average of Weighted Sums over 10 days
    df['SMA_WeightedSum'] = df['RollingWeightedSum'].rolling(window=10).mean()
    
    # Calculate True Range
    df['TrueRange'] = df[['high' - 'low', 
                          ('high' - 'close').abs(), 
                          ('low' - 'close').abs()]].max(axis=1)
    
    # Calculate ATR over 14 days
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    
    # Normalize the Alpha Factor by Dividing by ATR
    df['AlphaFactor'] = df['SMA_WeightedSum'] / df['ATR']
    
    return df['AlphaFactor'].dropna()
