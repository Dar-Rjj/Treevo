import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Average Intraday Range
    intraday_range = df['high'] - df['low']
    avg_intraday_range_10d = intraday_range.rolling(window=10).mean()
    
    # Calculate 10-Day Return
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Determine Intraday Volatility Adjusted Momentum
    intraday_vol_adj_momentum = ten_day_return * avg_intraday_range_10d
    
    # Compute Intraday Reversal Signal
    intraday_reversal_signal = (df['close'] - df['open']) / avg_intraday_range_10d
    
    # Incorporate Volume Impact
    five_day_cumulative_volume = df['volume'].rolling(window=5).sum()
    volume_weight = df['volume'] / five_day_cumulative_volume
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (intraday_reversal_signal + intraday_vol_adj_momentum) * volume_weight
    
    return final_alpha_factor
