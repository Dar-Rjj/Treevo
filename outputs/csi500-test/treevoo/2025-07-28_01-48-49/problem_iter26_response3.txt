```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Determine price trend
    df['Price_Trend'] = 0
    df.loc[(df['close'] > df['open']) & (df['high'] > df['low']), 'Price_Trend'] = 1
    df.loc[(df['close'] < df['open']) & (df['high'] > df['low']), 'Price_Trend'] = -1
    
    # Calculate volume change and identify volume spikes
    df['Volume_Change'] = df['volume'].diff()
    avg_volume_5d = df['volume'].rolling(window=5).mean().shift(1)
    df['Volume_Spike'] = (df['volume'] > 2 * avg_volume_5d).astype(int)
    
    # Calculate daily amount traded
    df['AmountTraded'] = df['close'] * df['volume']
    
    # Determine abnormal trading amount
    avg_amount_traded_5d = df['AmountTraded'].rolling(window=5).mean().shift(1)
    df['Abnormal_Trading_Amount'] = (df['AmountTraded'] > 2 * avg_amount_traded_5d).astype(int)
    
    # Calculate short-term momentum
    df['Short_Term_Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate long-term reversal
    df['Long_Term_Reversal'] = df['close'].shift(20) - df['close']
    
    # Aggregate alpha factors
    df['Alpha_Factor'] = (df['Price_Trend'] + 
                          df['Volume_Spike'] + 
                          df['Abnormal_Trading_Amount'] + 
                          df['Short_Term_Momentum'] + 
                          df['Long_Term_Reversal'])
    
    return df['Alpha_Factor']
```
