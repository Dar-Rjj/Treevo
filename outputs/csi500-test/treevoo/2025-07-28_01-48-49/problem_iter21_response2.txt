import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Average Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    df['10DayAvgIntradayRange'] = df['IntradayRange'].rolling(window=10).mean()
    
    # Calculate 20-Day Return
    df['20DayReturn'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Determine Intraday Volatility Adjusted Momentum
    df['IntradayVolAdjMomentum'] = df['20DayReturn'] * df['10DayAvgIntradayRange']
    
    # Compute Intraday Reversal Signal
    df['IntradayReversalSignal'] = (df['close'] - df['open']) / df['10DayAvgIntradayRange']
    
    # Incorporate Volume Impact
    df['20DayCumulativeVolume'] = df['volume'].rolling(window=20).sum()
    df['VolumeWeight'] = df['volume'] / df['20DayCumulativeVolume']
    
    # Final Alpha Factor Synthesis
    df['AlphaFactor'] = (df['IntradayVolAdjMomentum'] + df['IntradayReversalSignal']) * df['VolumeWeight']
    
    return df['AlphaFactor']
