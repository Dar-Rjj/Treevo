```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and indexed by date.
                       
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the change in volume
    volume_change = df['volume'].diff()
    
    # Calculate the change in amount
    amount_change = df['amount'].diff()
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the volatility factor
    volatility = df['close'].rolling(window=10).std()
    
    # Combine factors to create the final alpha factor
    alpha_factor = (avg_price * volume_change * amount_change * momentum) / volatility
    
    return alpha_factor
```
