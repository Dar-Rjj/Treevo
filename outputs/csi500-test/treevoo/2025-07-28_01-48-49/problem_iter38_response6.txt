```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['Adjusted_Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    df['Weighted_Adjusted_Intraday_Price_Movement'] = df['Adjusted_Intraday_Price_Movement'] * (df['volume'] * close_diff)
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = df['Daily_Range'].ewm(span=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return_and_Reversal_Signal'] = (df['amount'] * df['Adjusted_Intraday_Price_Movement']) + (df['volume'] * df['Reversal_Signal'])
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = df['Weighted_Adjusted_Intraday_Price_Movement'] + df['Intraday_Volatility'] + df['Volume_Weighted_Return']
    
    # Adjust by Relative Volume
    df['Relative_Volume'] = df['volume'] / df['volume'].ewm(span=20).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['Relative_Volume']
    
    # Incorporate Close Price Momentum
    df['Momentum'] = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['Momentum'] / df['Intraday_Volatility']
    
    # Construct Momentum Factor
    df['Momentum_Factor'] = df['close'].pct_change(5) + df['close'].pct_change(20) + df['close'].pct_change(60)
    
    # Design Volatility Measures
    df['Volatility_Measures'] = df['close'].pct_change().rolling(5).std() + df['close'].pct_change().rolling(20).std() + df['close'].pct_change().rolling(60).std()
    
    # Incorporate Volume-Based Indicators
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Explore Price Action Patterns
    def candlestick_pattern_score(row):
        if row['open'] > row['close']:
            return -1  # Bearish
        elif row['open'] < row['close']:
            return 1  # Bullish
        else:
            return 0  # Doji
    df['Pattern_Score'] = df.apply(candlestick_pattern_score, axis=1)
    df['Price_Action_Pattern_Score'] = df['Pattern_Score'].rolling(window=20).sum()
    
    # Develop Trend-Following Signals
    df['Trend_Signal'] = df['close'] - df['close'].rolling(window=20).mean()
    
    # Create Liquidity-Related Factors
    df['Spread'] = df['high'] - df['low']
    df['Liquidity_Factor'] = df['volume'] / df['Spread']
    
    # Examine Market Sentiment
    df['Up_Down_Sentiment'] = (df['close'] > df['close'].shift(1)).rolling(window=20).sum() - (df['close'] < df['close'].shift(1)).rolling(window=20).sum()
    
    # Final Alpha Factor Calculation
    combined_adjusted_factors = (0.3 * df['Adjusted_Intraday_Price_Movement']) + (0.4 * df['Intraday_Volatility']) + (0.3 * df['Adjusted_Momentum'])
    weighted_intraday_return_and_reversal_signal = df['Weighted_Intraday_Return_and_Reversal_Signal']
    
    final_alpha_factor = (combined_adjusted_factors * 0.6) + (weighted_intraday_return_and_reversal_signal * 0.4)
    
    return final_alpha_factor
```
