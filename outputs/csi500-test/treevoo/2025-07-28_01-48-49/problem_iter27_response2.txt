```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the average true range over 14 days
    df['true_range'] = df[['high' - 'low', 
                           abs('high' - df['close'].shift(1)), 
                           abs('low' - df['close'].shift(1))]].max(axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate a modified on-balance volume (OBV) indicator
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Factor: (Price Change / Average True Range) * OBV
    factor = (df['price_change'] / df['average_true_range']) * df['obv']
    
    return factor
```
