```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_range = (intraday_range * volume_change) / volume_ma
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_price_range = adjusted_range * price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread_5 = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread_10 = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Combine 5-Day Average High-Low Spread and Momentum
    combined_high_low_momentum = avg_high_low_spread_5 * momentum
    
    # Integrate Intraday Volatility
    daily_ranges = df['high'] - df['low']
    avg_daily_range = daily_ranges.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Final Factor Calculation
    final_factor = (
        0.25 * adjusted_price_range + 
        0.25 * momentum + 
        0.25 * combined_high_low_momentum + 
        0.15 * avg_daily_range + 
        0.1 * volume_weighted_return
    )
    
    return final_factor
```
