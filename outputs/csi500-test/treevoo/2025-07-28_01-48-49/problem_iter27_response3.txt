import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Generate IntradayMomentumScore
    volume_change = df['volume'] - df['volume'].shift(1)
    intraday_momentum_score = intraday_range * volume_change
    
    # Finalize Alpha Factor Components
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Momentum
    five_day_momentum = (df['close'] / df['close'].shift(5)) - 1
    
    # Combine Intraday and High-Low Spread Momentum
    integrated_momentum_factor = (intraday_momentum_score + intraday_return) * avg_high_low_spread_5d
    
    # Calculate Volume Weighted Average Price
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    vwap /= df['volume']
    
    # Combine Factors with Volume-Adjusted Price Trend
    combined_factor = (close_to_close_return * intraday_range) / vwap
    
    # Introduce Volume Trend
    volume_sum_5d = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5d
    volume_trend_adjusted_factor = combined_factor * volume_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    close_price_momentum = (df['close'] / df['close'].shift(momentum_lookback)) - 1
    
    # Adjust Momentum by Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=5).mean()
    adjusted_momentum = close_price_momentum * avg_daily_range
    
    # Calculate Dynamic Adjustment Factor
    n_day_ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    dynamic_adjustment_factor = df['volume'] / n_day_ema_volume
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Open Price Impact
    relative_open_price_change = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate 5-Day Open Price Momentum
    open_price_momentum_5d = (df['open'] / df['open'].shift(5)) - 1
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.60 * dynamically_adjusted_momentum +
        0.20 * relative_open_price_change +
        0.20 * open_price_momentum_5d
    )
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    final_alpha_factor -= final_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
