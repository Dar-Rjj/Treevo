```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Next-Open Difference
    df['close_to_next_open'] = df['close'].shift(-1) - df['open']
    
    # Apply Weight by Volume to the Close-to-Next-Open Difference
    df['volume_weighted_diff'] = df['volume'] * df['close_to_next_open']
    
    # Calculate Sum of Volume-Weighted Differences
    weighted_sum = df['volume_weighted_diff'].rolling(window=5).sum()
    
    # Calculate Exponential Moving Average of Weighted Sums
    ema_weighted_sum = weighted_sum.ewm(span=10, adjust=False).mean()
    
    # Use Standard Deviation of Returns as a Proxy for Volatility
    df['returns'] = df['close'].pct_change()
    volatility = df['returns'].rolling(window=14).std()
    
    # Adjust for Volatility Impact
    factor_values = ema_weighted_sum / volatility
    
    return factor_values
```
