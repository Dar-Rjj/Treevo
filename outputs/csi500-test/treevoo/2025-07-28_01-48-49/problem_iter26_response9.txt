import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_high = df['high']
    intraday_low = df['low']
    
    # Calculate Intraday Range
    intraday_range = intraday_high - intraday_low
    
    # Adjust by Volume Change and Average Volume
    ma_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_intraday_range = (intraday_range * volume_change) / ma_volume
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_intraday_range *= price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    close_momentum = df['close'].pct_change(momentum_lookback)
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine 5-Day Average High-Low Spread and Momentum
    high_low_spread_momentum = avg_high_low_spread * close_momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Calculate Volume Weighted Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Calculate IntegratedMomentumFactor
    integrated_momentum_factor = adjusted_intraday_range + high_low_spread_momentum * close_momentum
    
    # Combine Factors with Volume-Adjusted Price Trend
    close_to_close_return = df['close'].pct_change()
    combined_factor = close_to_close_return * intraday_range / ma_volume
    
    # Filter for Recent Price Trend
    ma_close_5d = df['close'].rolling(window=5).mean()
    recent_price_trend = df['close'] - ma_close_5d
    
    # Introduce Volume Trend
    sum_volume_5d = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / sum_volume_5d
    volume_trend_adjusted_factor = combined_factor * volume_ratio
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * avg_daily_range
    
    # Calculate Dynamic Adjustment Factor
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * ema_volume
    
    # Calculate Open Price Impact
    relative_open_price_change = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (0.80 * dynamically_adjusted_momentum) + (0.20 * relative_open_price_change)
    
    # Enhance with Volume Trend
    ma_volume_5d = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] - ma_volume_5d
    enhanced_alpha_factor = final_alpha_factor * volume_trend
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    ema_final_alpha = enhanced_alpha_factor.ewm(span=20, adjust=False).mean()
    final_alpha_factor = enhanced_alpha_factor - ema_final_alpha
    
    return final_alpha_factor
