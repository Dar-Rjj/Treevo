import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_5d_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_10d_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply 5-Day Average High-Low Spread by Momentum
    weighted_avg_high_low_spread = avg_5d_high_low_spread * momentum
    
    # Calculate 20-Day Price Return
    price_return_20d = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Add 10-Day EMA of High-Low Spread to 20-Day Price Return
    factor_value = ema_10d_high_low_spread + price_return_20d
    
    return factor_value
