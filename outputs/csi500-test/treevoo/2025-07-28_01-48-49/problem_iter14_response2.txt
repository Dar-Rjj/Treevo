```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Calculate Moving Average of Volume
    lookback_period_vol = 20
    avg_volume = df['volume'].rolling(window=lookback_period_vol).mean()
    
    # Adjust Intraday Price Range by Average Volume
    adjusted_intraday_range = intraday_range / avg_volume

    # Calculate Close Price Momentum
    momentum_lookback = 10
    momentum = (df['close'] - df['close'].shift(momentum_lookback)) / df['close'].shift(momentum_lookback)

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()

    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()

    # Calculate Momentum (Close price today vs 5 days ago)
    close_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply 5-Day Average High-Low Spread by Momentum
    weighted_avg_high_low_spread = avg_high_low_spread_5d * close_momentum

    # Add 10-Day EMA of High-Low Spread to 20-Day Price Return
    price_return_20d = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    ema_plus_price_return = ema_high_low_spread + price_return_20d

    # Calculate Open Adjustment Factor
    open_adjustment = df['open'] - df['open'].shift(1)

    # Adjust High-to-Low Movement by Open Adjustment
    adjusted_high_low_movement = (df['high'] - df['low']) / open_adjustment

    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_high_low_movement = adjusted_high_low_movement * (df['volume'] * close_difference)

    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=lookback_period_vol).mean()

    # Incorporate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return

    # Calculate Open Price Impact
    relative_open_price_change = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)

    # Final Factor Calculation
    factor_value = (0.20 * adjusted_intraday_range +
                    0.20 * momentum +
                    0.20 * weighted_high_low_movement +
                    0.15 * avg_daily_range +
                    0.15 * volume_weighted_return +
                    0.10 * relative_open_price_change)

    return factor_value
```
