import pandas as pd

def heuristics_v2(df):
    # Compute daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Price-based alpha factors
    df['high_low_diff'] = df['high'] - df['low']
    df['close_open_ratio'] = df['close'] / df['open']

    # Volume analysis
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_close_ratio'] = df['volume'] / df['close']

    # Momentum and trend following
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    df['short_ma_diff'] = df['close'] - df['short_ma']
    df['long_ma_diff'] = df['close'] - df['long_ma']

    # Volatility and risk
    df['volatility'] = df['daily_return'].rolling(window=10).std()

    # Breakout and reversal patterns
    n = 5
    df['above_high'] = df['close'] > df['high'].rolling(window=n).max().shift(1)
    df['below_low'] = df['close'] < df['low'].rolling(window=n).min().shift(1)

    # Amount and volume relationship
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    df['amount_volume_ratio_change'] = df['amount_volume_ratio'] - df['amount_volume_ratio'].shift(1)

    # Gaps
    df['up_gap'] = df['open'] > df['close'].shift(1)
    df['down_gap'] = df['open'] < df['close'].shift(1)

    # Adjusted Intraday Price Movement
    df['adjusted_intraday_price_movement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))

    # Weight by Volume and Close Difference
    df['volume_weighted_return'] = df['volume'] * (df['close'] - df['close'].shift(1))
    df['weighted_intraday_price_movement'] = df['adjusted_intraday_price_movement'] * df['volume_weighted_return']

    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=20).mean()

    # Close-to-Open Reversal Signal
    df['close_to_open_reversal_signal'] = (df['close'] - df['open']) / df['open']

    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_return_reversal'] = (df['amount'] * df['adjusted_intraday_price_movement']) + (df['volume'] * df['close_to_open_reversal_signal'])

    # Combine Enhanced Factors
    df['combined_factors'] = (df['weighted_intraday_price_movement'] * df['volume_weighted_return']) + (df['intraday_volatility'] * df['volume_weighted_return'])

    # Adjusted Momentum
    df['momentum_10d_ma'] = df['close'] - df['close'].rolling(window=10).mean()
    df['adjusted_momentum'] = df['momentum_10d_ma'] * df['intraday_volatility']

    # Final Alpha Factor
    final_alpha_factor = (0.3 * df['adjusted_intraday_price_movement']) + (0.3 * df['intraday_volatility']) + (0.4 * df['adjusted_momentum']) * 0.5
    final_alpha_factor += (df['weighted_intraday_return_reversal'] * 0.5)

    return final_alpha_factor
