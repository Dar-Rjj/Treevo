import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_movement * volume_change
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d_avg = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor Components
    close_price = df['close']
    open_price = df['open']
    intraday_return = (close_price - open_price) / open_price
    five_day_momentum = (close_price - close_price.shift(5)) / close_price.shift(5)
    high_low_spread_momentum = high_low_spread_5d_avg * five_day_momentum
    
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=5).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * average_daily_range
    
    # Calculate Dynamic Adjustment Factor
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    dynamic_adjustment_factor = df['volume'] / volume_ema
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Open Price Impact
    relative_open_price_change = (open_price - open_price.shift(1)) / open_price.shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = 0.80 * dynamically_adjusted_momentum + 0.20 * relative_open_price_change
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    final_alpha_factor_ema = final_alpha_factor.ewm(span=5, adjust=False).mean()
    final_alpha_factor -= final_alpha_factor_ema
    
    return final_alpha_factor
