```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['amount']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=N).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate High-to-Low Movement
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate Open Adjustment Factor
    open_adj_factor = df['open'].diff()
    
    # Adjust High-to-Low Movement by Open Adjustment
    adj_high_to_low_movement = high_to_low_movement / open_adj_factor
    
    # Weight by Volume and Close Difference
    close_diff = df['close'].diff()
    weighted_adj_hl_movement = adj_high_to_low_movement * (df['volume'] * close_diff)
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_signal_weight = df['volume']
    weighted_intraday_return_reversal = (intraday_return_weight * daily_return) + (reversal_signal_weight * -adj_high_to_low_movement)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum * 0.5) + (weighted_intraday_return_reversal * 0.5)
    
    # Filter for Recent Trend
    recent_trend = df['close'] - df['close'].rolling(window=5).mean()
    final_alpha_factor = final_alpha_factor * (recent_trend > 0)

    return final_alpha_factor
```
