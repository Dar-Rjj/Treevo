import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_price_range / df['volume']
    
    # Calculate Weighted Momentum
    weighted_momentum = df['close'].ewm(span=N, adjust=False).mean()
    
    # Adjust Weighted Momentum by Intraday Volatility
    adjusted_weighted_momentum = weighted_momentum * intraday_volatility
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_weighted_momentum - adjusted_weighted_momentum.ewm(span=N, adjust=False).mean()
    
    return final_alpha_factor
