import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum = avg_high_low_spread_5d * momentum_5d
    
    # Calculate Enhanced Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_price_movement = intraday_price_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = weighted_intraday_price_movement.abs()
    intraday_momentum_score = intraday_momentum_score.clip(lower=0)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    close_open_reversal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * close_open_reversal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_signal_weight = df['volume']
    weighted_intraday_return_reversal = (intraday_return_weight * intraday_price_movement) + (reversal_signal_weight * close_open_reversal)
    
    # Combine Enhanced Factors
    combined_factors = (0.3 * intraday_momentum_score) + (0.3 * intraday_volatility) + (0.4 * spread_momentum)
    
    # Final Factor Calculation
    final_factor = (0.5 * combined_factors) + (0.5 * weighted_intraday_return_reversal)
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_factor_adjusted = final_factor * price_level_ratio
    
    # Enhance with Volume Trend
    volume_ma_5d = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] - volume_ma_5d
    final_factor_enhanced = final_factor_adjusted * volume_trend
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    ema_n = final_factor_enhanced.ewm(span=10, adjust=False).mean()
    final_alpha_factor = final_factor_enhanced - ema_n
    
    return final_alpha_factor
