```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Dynamics
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['AdjustedIntradayRange'] = df['IntradayRange'] * df['VolumeChange']
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['PriceAdjustedRange'] = df['AdjustedIntradayRange'] * df['PriceLevelRatio']
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['AdjustedIntradayRange'] * df['VolumeChange']
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['5DayAvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    df['5DayMomentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['VolumeWeightedAveragePrice'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VolumeAdjustedReturn'] = (df['close'] - df['close'].shift(1)) / df['VolumeWeightedAveragePrice']
    df['VolumePriceTrendFactor'] = df['VolumeAdjustedReturn'] * df['PriceAdjustedRange']
    
    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['VolumeRatio'] * (df['PriceAdjustedRange'] + df['IntegratedMomentumFactor'] + df['VolumePriceTrendFactor'])
    
    # Final Factor Calculation
    final_factor = (0.2 * df['PriceAdjustedRange'] +
                    0.3 * df['IntegratedMomentumFactor'] +
                    0.2 * df['VolumePriceTrendFactor'] +
                    0.3 * df['VolumeTrendAdjustedFactor'])
    
    return final_factor.dropna()
```
