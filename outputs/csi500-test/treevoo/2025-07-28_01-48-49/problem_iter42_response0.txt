```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, high, low, close) for each day
    avg_price = df[['open', 'high', 'low', 'close']].mean(axis=1)
    
    # Calculate the price range (high - low) for each day
    price_range = df['high'] - df['low']
    
    # Calculate a modified volume factor: (volume * (close - open) / (high - low))
    # This factor represents the volume weighted by the price change relative to the price range
    modified_volume = df['volume'] * (df['close'] - df['open']) / price_range
    
    # Calculate the factor as the ratio of the modified volume to the average price
    factor = modified_volume / avg_price
    
    return factor
```
