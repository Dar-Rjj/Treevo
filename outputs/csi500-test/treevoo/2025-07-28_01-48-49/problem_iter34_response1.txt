```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    df['VolumeChange'] = df['volume'].pct_change()
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['AdjustedRange'] = df['IntradayRange'] * df['VolumeChange'] * df['PriceLevelRatio']
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['AdjustedRange']
    
    # 5-Day Average High-Low Spread
    df['5DayAvgHighLowSpread'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # 5-Day Momentum
    df['5DayMomentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # HighLowSpreadMomentum
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayMomentum']
    
    # IntegratedMomentumFactor
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = df['close'].pct_change()
    df['VolumeWeightedAveragePrice'] = (df['amount'] / df['volume'])
    df['VolumeAdjustedPriceTrend'] = df['CloseToCloseReturn'] * df['IntradayRange'] / df['VolumeWeightedAveragePrice']
    
    # Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['VolumeAdjustedPriceTrend'] * df['VolumeRatio']
    
    # Final Factor Calculation
    final_factor = (
        0.2 * df['AdjustedRange'] +
        0.3 * df['5DayMomentum'] +
        0.2 * df['IntegratedMomentumFactor'] +
        0.3 * df['VolumeTrendAdjustedFactor']
    )
    
    return final_factor
```
