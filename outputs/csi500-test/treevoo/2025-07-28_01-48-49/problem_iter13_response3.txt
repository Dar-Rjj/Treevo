```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    df['IntradayRange'] = intraday_range
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    df['VolumeChange'] = volume_change
    
    # Generate Intraday Momentum Score
    intraday_momentum_score = intraday_range * volume_change
    df['IntradayMomentumScore'] = intraday_momentum_score
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    five_day_momentum = (df['close'] / df['close'].shift(5) - 1)
    
    # Generate HighLowSpreadMomentum
    high_low_spread_momentum = avg_high_low_spread * five_day_momentum
    df['HighLowSpreadMomentum'] = high_low_spread_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and High-Low Spread Momentum
    combined_intraday_momentum = intraday_return + intraday_momentum_score
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    df['IntegratedMomentumFactor'] = integrated_momentum_factor
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['volume']
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * intraday_volatility
    df['AdjustedMomentum'] = adjusted_momentum
    
    # Apply Dynamic Adjustment
    n_day_ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    dynamic_adjusted_momentum = adjusted_momentum * n_day_ema_volume
    n_day_ema_adjusted_momentum = dynamic_adjusted_momentum.ewm(span=5, adjust=False).mean()
    
    final_factor = dynamic_adjusted_momentum - n_day_ema_adjusted_momentum
    return final_factor
```
