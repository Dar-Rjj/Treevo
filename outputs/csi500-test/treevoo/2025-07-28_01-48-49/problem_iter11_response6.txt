```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Avg_High_Low_Spread_5D'] = df['High_Low_Spread'].rolling(window=5).mean()

    # Calculate 5-Day Momentum
    df['Momentum_5D'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply High-Low Spread and Momentum
    df['High_Low_Momentum'] = df['Avg_High_Low_Spread_5D'] * df['Momentum_5D']

    # Calculate Daily Return
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate N-day Exponential Moving Average of Close price
    df['EMA_Close'] = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate Intraday Price Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['Intraday_Range'] / df['volume']

    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['Momentum_5D'] * df['Intraday_Volatility']

    # Calculate Dynamic Adjustment Factor
    df['EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['Dynamic_Adjustment_Factor'] = df['volume'] / df['EMA_Volume']

    # Apply Dynamic Adjustment
    df['Adjusted_Momentum_Dynamic'] = df['Adjusted_Momentum'] * df['Dynamic_Adjustment_Factor']

    # Calculate Close to Close Return
    df['Close_to_Close_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Calculate Volume Weighted Average Price
    df['VWAP'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume'] / df['volume']

    # Combine Factors
    df['Combined_Factor'] = df['Close_to_Close_Return'] * df['Intraday_Range'] / df['VWAP']

    # Filter for Recent Price Trend
    df['5D_MA_Close'] = df['close'].rolling(window=5).mean()
    df['Recent_Price_Trend'] = df['close'] - df['5D_MA_Close']

    # Introduce Volume Trend
    df['5D_Sum_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Ratio'] = df['volume'] / df['5D_Sum_Volume']

    # Final Combined Factor Computation
    df['Final_Combined_Factor'] = df['Combined_Factor'] * df['Volume_Ratio'] * df['High_Low_Momentum']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Momentum_Dynamic'] - df['Adjusted_Momentum_Dynamic'].ewm(span=5, adjust=False).mean()) + df['Final_Combined_Factor']

    return df['Final_Alpha_Factor']
```
