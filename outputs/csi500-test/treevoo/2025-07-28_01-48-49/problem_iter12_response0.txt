import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = (intraday_range * abs(volume_change)).apply(lambda x: max(x, 0))
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 5-Day Close-Open Spread
    close_open_spread = (df['close'].shift(5) - df['open']) / df['close'].shift(5)
    
    # Finalize Alpha Factor
    high_low_spread_momentum = avg_high_low_spread * close_open_spread
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    
    return integrated_momentum_factor
