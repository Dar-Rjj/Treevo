```python
import pandas as pd

def heuristics_v2(df):
    # 5-Day Average High-Low Spread
    high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # 5-Day Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply High-Low Spread and Momentum
    spread_momentum = high_low_spread * momentum
    
    # Enhanced Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday = intraday_price_movement * (df['volume'] * close_diff)
    
    # Generate IntradayMomentumScore
    intraday_momentum_score = weighted_intraday * abs(df['volume'] * close_diff)
    intraday_momentum_score[intraday_momentum_score < 0] = 0
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_weight = df['volume']
    weighted_intraday_return = (intraday_return_weight * intraday_price_movement) + (reversal_weight * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factors = 0.3 * intraday_momentum_score + 0.3 * intraday_volatility + 0.4 * (spread_momentum * volume_weighted_return)
    
    # Final Factor Calculation
    final_factor = (0.5 * combined_factors) + (0.5 * weighted_intraday_return)
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_factor_adjusted = final_factor * price_level_ratio
    
    # Enhance with Volume Trend
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] - volume_ma_5
    final_factor_enhanced = final_factor_adjusted * volume_trend
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    ema_final_factor = final_factor_enhanced.ewm(span=5, adjust=False).mean()
    final_alpha_factor = final_factor_enhanced - ema_final_factor
    
    # Introduce Directional Bias
    close_ma_5 = df['close'].rolling(window=5).mean()
    directional_bias = (df['close'] > close_ma_5).astype(int).replace({True: 1, False: -1})
    final_alpha_factor = final_alpha_factor * directional_bias
    
    return final_alpha_factor
```
