```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low for each day
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the weighted amount based on the volume
    weighted_amount = df['amount'] * df['volume']
    
    # Calculate the ratio of close price to the average of high and low
    close_to_avg_ratio = df['close'] / avg_high_low
    
    # Calculate the momentum factor as the difference between the close price and open price
    momentum = df['close'] - df['open']
    
    # Final alpha factor is the product of the close-to-average ratio, weighted amount, and momentum
    alpha_factor = close_to_avg_ratio * weighted_amount * momentum
    
    return alpha_factor
```
