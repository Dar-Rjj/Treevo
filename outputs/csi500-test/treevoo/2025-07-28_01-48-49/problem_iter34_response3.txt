import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['Prev_Open'] = df['open'].shift(1)
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['Prev_Open']).fillna(0.0001)

    # Adjusted Intraday Price Movement
    df['Adjusted_Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['Prev_Open']).fillna(0.0001)

    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Weight by Volume and Close Difference
    df['Prev_Close'] = df['close'].shift(1)
    df['Volume_Close_Diff'] = df['volume'] * (df['close'] - df['Prev_Close'])

    # Adjust by Volume Change and Average Volume
    df['MA_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Intraday_Range_Adjusted'] = (df['Intraday_Range'] * df['Volume_Change']) / df['MA_Volume_20']

    # Apply Price Level Adjustment
    df['Price_Level_Ratio'] = df['close'] / df['open']
    df['Adjusted_Intraday_Range'] = df['Intraday_Range'] * df['Price_Level_Ratio']

    # Incorporate Close Price Momentum
    df['Momentum_10'] = df['close'] / df['close'].shift(10) - 1

    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range_20'] = df['Daily_Range'].rolling(window=20).mean()

    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Return_Reversal'] = (df['amount'] * df['Intraday_Price_Movement']) + df['Volume_Close_Diff']

    # Calculate IntegratedMomentumFactor
    df['CombinedIntradayMomentum'] = df['Intraday_Price_Movement'] * df['Momentum_10']
    df['HighLowSpreadMomentum'] = (df['high'] - df['low']) * df['Momentum_10']
    df['IntegratedMomentumFactor'] = df['CombinedIntradayMomentum'] * df['HighLowSpreadMomentum']

    # Combine Factors with Volume-Adjusted Price Trend
    df['Close_to_Close_Return'] = df['close'] / df['Prev_Close'] - 1
    df['Volume_Weighted_Avg_Price'] = (df['amount'] / df['volume'])
    df['Volume_Adjusted_Price_Trend'] = (df['Close_to_Close_Return'] * df['Intraday_Range']) / df['Volume_Weighted_Avg_Price']

    # Filter for Recent Price Trend
    df['5D_MA_Close'] = df['close'].rolling(window=5).mean()
    df['Recent_Price_Trend'] = df['close'] - df['5D_MA_Close']

    # Introduce Volume Trend
    df['5D_Sum_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Ratio'] = df['volume'] / df['5D_Sum_Volume']
    df['Combined_Factor'] = df['Volume_Adjusted_Price_Trend'] * df['Volume_Ratio']

    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['IntegratedMomentumFactor'] * (df['Daily_Range'] / df['Avg_Daily_Range_20'])

    # Calculate Dynamic Adjustment Factor
    df['EMA_Volume_N'] = df['volume'].ewm(span=20, adjust=False).mean()

    # Apply Dynamic Adjustment
    df['Dynamically_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (df['volume'] / df['EMA_Volume_N'])

    # Calculate Open Price Impact
    df['Relative_Open_Price_Change'] = (df['open'] - df['Prev_Open']) / df['Prev_Open'].fillna(0.0001)

    df['Final_Alpha_Factor'] = (
        0.3 * df['Adjusted_Intraday_Price_Movement'] +
        0.3 * (df['Daily_Range'] / df['Avg_Daily_Range_20'])
