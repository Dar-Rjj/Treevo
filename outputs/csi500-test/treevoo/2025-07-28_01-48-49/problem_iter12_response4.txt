```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['volume']
    
    # Calculate Momentum (N-day Exponential Moving Average of Close price)
    momentum = df['close'].ewm(span=n, adjust=False).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate Dynamic Adjustment Factor (N-day Exponential Moving Average of Volume)
    dynamic_adjustment_factor = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Reversal Signal
    ema_daily_return = daily_return.ewm(span=n, adjust=False).mean()
    reversal_signal = ema_daily_return - daily_return
    
    # Combine Adjusted Momentum and Reversal Signal
    combined_momentum = dynamically_adjusted_momentum * reversal_signal
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum - combined_momentum.ewm(span=n, adjust=False).mean()
    
    return final_alpha_factor
```
