import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Moving Average of Volume
    lookback_period = 20
    ma_volume = df['volume'].rolling(window=lookback_period).mean()
    
    # Current Day Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust by Volume Change and Price Level
    adjusted_range = intraday_range * volume_change
    price_level_ratio = df['close'] / df['open']
    adjusted_range_price_level = adjusted_range * price_level_ratio
    
    # Close Price Momentum
    momentum_lookback = 10
    momentum = df['close'].pct_change(momentum_lookback)
    
    # Final Factor Calculation
    adjusted_range_weight = 0.6
    momentum_weight = 0.4
    final_factor = (adjusted_range_price_level * adjusted_range_weight) + (momentum * momentum_weight)
    
    return final_factor
