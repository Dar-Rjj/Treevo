```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Range
    df['PriceRange'] = df['high'] - df['low']
    
    # Examine Opening and Closing Gaps
    df['GapOpenClosePrevDay'] = df['open'] - df['close'].shift(1)
    df['IntradayChange'] = df['close'] - df['open']
    
    # Analyze Volume-Price Interaction
    df['AverageTransactionValue'] = (df['high'] + df['low']) / 2 * df['volume']
    
    # Measure Trading Activity
    df['VolumeToPriceVolatilityRatio'] = df.apply(lambda row: row['volume'] / (row['high'] - row['low']) if row['high'] != row['low'] else 0, axis=1)
    
    # Simple Moving Averages
    df['SMA5D'] = df['close'].rolling(window=5).mean()
    df['SMA20D'] = df['close'].rolling(window=20).mean()
    
    # Compare Short-Term and Long-Term Averages
    df['MomentumSignal'] = df['SMA5D'] - df['SMA20D']
    
    # Absolute Daily Price Movement
    df['DailyPriceMovement'] = (df['close'] - df['open']).abs()
    
    # Relative Daily Price Movement
    df['RelativeDailyPriceMovement'] = df.apply(lambda row: (row['close'] - row['open']) / row['open'] if row['open'] != 0 else 0, axis=1)
    
    # Volatility Ratio
    df['VolatilityRatio'] = df.apply(lambda row: (row['high'] - row['low']) / row['close'] if row['close'] != 0 else 0, axis=1)
    
    # Combine all factors
    df['CombinedAlphaFactor'] = (df['PriceRange'] + df['GapOpenClosePrevDay'] + df['IntradayChange'] +
                                 df['VolumeToPriceVolatilityRatio'] + df['MomentumSignal'] +
                                 df['DailyPriceMovement'] + df['RelativeDailyPriceMovement'] +
                                 df['VolatilityRatio'])
    
    return df['CombinedAlphaFactor']
```
