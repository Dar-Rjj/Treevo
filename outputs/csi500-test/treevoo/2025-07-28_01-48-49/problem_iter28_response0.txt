import pandas as pd

def heuristics_v2(df):
    # Adjusted Intraday Price Movement
    df['Prev_Open'] = df['open'].shift(1)
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['Prev_Open'])
    
    # Weight by Volume and Close Difference
    df['Volume_Close_Diff'] = df['volume'] * (df['close'] - df['close'].shift(1))
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * df['Volume_Close_Diff']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Close-to-Open Reversal Signal
    df['Close_to_Open_Reversal'] = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Construct_Weighted_Return'] = (df['amount'] * df['Intraday_Price_Movement']) + (df['volume'] * df['Close_to_Open_Reversal'])
    
    # Combine Enhanced Factors
    df['Combined_Factors'] = (df['Weighted_Intraday_Price_Movement'] * df['Volume_Close_Diff']) + (df['Intraday_Volatility'] * df['Volume_Close_Diff'])
    
    # Adjusted Momentum
    df['Momentum'] = (df['close'] - df['close'].rolling(window=10).mean()) * df['Intraday_Volatility']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (0.3 * df['Intraday_Price_Movement'] + 0.3 * df['Intraday_Volatility'] + 0.4 * df['Momentum']) * 0.5 + (df['Construct_Weighted_Return'] * 0.5)
    
    return df['Alpha_Factor']
