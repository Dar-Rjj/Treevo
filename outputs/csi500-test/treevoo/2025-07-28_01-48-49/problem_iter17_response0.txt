import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_price_movement * volume_change
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5day_avg = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor Components
    close_price_today = df['close']
    open_price_today = df['open']
    intraday_return = (close_price_today - open_price_today) / open_price_today
    five_day_momentum = (close_price_today - close_price_today.shift(5)) / close_price_today.shift(5)
    high_low_spread_momentum = high_low_spread_5day_avg * five_day_momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    
    # Adjust by Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    adjusted_integrated_momentum = integrated_momentum_factor / moving_average_volume
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=20).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = adjusted_integrated_momentum * average_daily_range
    
    # Calculate Dynamic Adjustment Factor
    dynamic_adjustment_factor = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Open Price Impact
    relative_open_price_change = (open_price_today - open_price_today.shift(1)) / open_price_today.shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = 0.70 * dynamically_adjusted_momentum + 0.30 * relative_open_price_change
    final_alpha_factor = final_alpha_factor - final_alpha_factor.ewm(span=20, adjust=False).mean()
    
    return final_alpha_factor
