```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = current_volume - prev_volume
    
    # Integrate Intraday Price Movement and Volume
    intraday_range_non_neg = intraday_range.clip(lower=0)
    volume_change_non_neg = volume_change.clip(lower=0)
    intraday_momentum_score = intraday_range_non_neg * volume_change_non_neg
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor Components
    close_price = df['close']
    open_price = df['open']
    intraday_return = (close_price - open_price) / open_price
    close_price_5d_ago = close_price.shift(5)
    five_day_momentum = (close_price - close_price_5d_ago) / close_price_5d_ago
    high_low_spread_momentum = high_low_spread_5d * five_day_momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / current_volume
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * intraday_volatility
    
    # Calculate Dynamic Adjustment Factor
    ema_volume = df['volume'].ewm(span=5).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum / ema_volume
    
    # Final Alpha Factor
    n_day_ema = dynamically_adjusted_momentum.ewm(span=5).mean()
    final_alpha_factor = dynamically_adjusted_momentum - n_day_ema
    
    return final_alpha_factor
```
