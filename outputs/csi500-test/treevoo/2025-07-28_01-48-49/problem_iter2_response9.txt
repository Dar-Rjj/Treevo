```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Price Differences
    df['Daily_Range'] = df['high'] - df['low']
    df['Open_to_Open_Diff'] = df['open'].diff()

    # Aggregate Price Difference with Volume
    df['Volume_Weighted_Diff'] = df['Open_to_Open_Diff'] * df['volume']
    df['Sum_Weighted_Diffs'] = df['Volume_Weighted_Diff'].rolling(window=5).sum()

    # Enhance with Smoothed Moving Average Filter
    df['SMA_Weighted_Sums'] = df['Sum_Weighted_Diffs'].rolling(window=10).mean()
    
    # Calculate ATR for Volatility Adjustment
    df['True_Range'] = df[['high' - 'low', 'high' - df['close'].shift(), df['close'].shift() - 'low']].max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    df['Alpha_Factor_Vol_Adjusted'] = df['SMA_Weighted_Sums'] / df['ATR']

    # Combine with High-to-Low Movement Relative to Open
    df['High_to_Low_Movement'] = df['high'] - df['low']
    df['Open_Adjustment_Factor'] = df['open'].diff()
    df['Adjusted_High_to_Low_Movement'] = df['High_to_Low_Movement'] / df['Open_Adjustment_Factor']
    df['Close_Diff'] = df['close'].diff()
    df['Weighted_Adjusted_Movement'] = df['Adjusted_High_to_Low_Movement'] * (df['volume'] * df['Close_Diff'])

    # Final Composite Alpha Factor
    df['Final_Alpha_Factor'] = df['Alpha_Factor_Vol_Adjusted'] + df['Weighted_Adjusted_Movement']

    return df['Final_Alpha_Factor']
```
