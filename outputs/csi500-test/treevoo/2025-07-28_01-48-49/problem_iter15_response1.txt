def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Generate IntradayMomentumScore
    df['IntradayMomentumScore'] = (df['IntradayRange'] * df['VolumeChange'].abs()).apply(lambda x: max(x, 0))
    
    # Calculate 5-Day Average High-Low Spread
    df['AvgHighLowSpread_5Day'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate 5-Day Close-Open Spread
    df['CloseOpenSpread_5Day'] = (df['close'].shift(5) - df['open']) / df['close'].shift(5)
    
    # Calculate 10-Day Average High-Low Spread
    df['AvgHighLowSpread_10Day'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate 10-Day Close-Open Spread
    df['CloseOpenSpread_10Day'] = (df['close'].shift(10) - df['open']) / df['close'].shift(10)
    
    # Finalize Short-Term Alpha Factor
    df['HighLowSpreadMomentum_5Day'] = df['AvgHighLowSpread_5Day'] * df['CloseOpenSpread_5Day']
    df['IntegratedMomentumFactor_5Day'] = df['HighLowSpreadMomentum_5Day'] + df['IntradayMomentumScore']
    
    # Finalize Long-Term Alpha Factor
    df['HighLowSpreadMomentum_10Day'] = df['AvgHighLowSpread_10Day'] * df['CloseOpenSpread_10Day']
    df['IntegratedMomentumFactor_10Day'] = df['HighLowSpreadMomentum_10Day'] + df['IntradayMomentumScore']
    
    # Combine Short-Term and Long-Term Alpha Factors
    df['CombinedMomentumFactor'] = df['IntegratedMomentumFactor_5Day'] + df['IntegratedMomentumFactor_10Day']
    
    return df['CombinedMomentumFactor'].dropna()
