```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
        
    Returns:
        pd.Series: Series indexed by (date) representing the factor values.
    """
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume'])
    
    # Calculate the deviation from VWAP
    deviation_from_vwap = (df['close'] - vwap) / vwap
    
    # Combine the factors into a single alpha factor
    alpha_factor = (buying_pressure - selling_pressure) * deviation_from_vwap
    
    return alpha_factor
```
