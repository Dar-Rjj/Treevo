```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_price = df['high']
    low_price = df['low']
    intraday_range = (high_price - low_price).rename('IntradayRange')

    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = (current_volume - prev_volume).rename('VolumeChange')

    # Integrate Intraday Price Movement and Volume
    momentum_score = (intraday_range * volume_change).rename('MomentumScore')

    # Calculate Intraday Volatility Adjusted Momentum
    close_price = df['close']
    open_price = df['open']
    intraday_absolute_return = (close_price - open_price).abs().rename('IntradayAbsoluteReturn')
    
    # Ensure non-zero for division
    intraday_absolute_return = intraday_absolute_return.replace(0, 1e-6)

    # Finalize Alpha Factor
    intraday_momentum_with_volume_trend = (momentum_score / intraday_absolute_return).rename('IntradayMomentumWithVolumeTrend')
    
    return intraday_momentum_with_volume_trend
```
