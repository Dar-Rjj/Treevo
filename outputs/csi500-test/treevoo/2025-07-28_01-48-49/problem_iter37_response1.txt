import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * df['volume'] * df['Close_Difference']
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['20_day_Moving_Average_of_Daily_Ranges'] = df['Daily_Range'].rolling(window=20).mean()
    df['Intraday_Volatility'] = df['Daily_Range'] / df['20_day_Moving_Average_of_Daily_Ranges']
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return'] = df['volume'] * (df['close'] - df['open'])
    df['Weighted_Reversal_Signal'] = df['amount'] * df['Reversal_Signal']
    df['Combined_Weighted_Signal'] = df['Weighted_Intraday_Return'] + df['Weighted_Reversal_Signal']
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = (df['Weighted_Intraday_Price_Movement'] + 
                             df['Intraday_Volatility'] * df['volume'] * df['Close_Difference'])
    
    # Adjust by Average Volume
    df['40_day_Moving_Average_of_Volume'] = df['volume'].rolling(window=40).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['40_day_Moving_Average_of_Volume']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.6 * df['Weighted_Intraday_Price_Movement'] / df['40_day_Moving_Average_of_Volume']) + \
                         (0.2 * df['Intraday_Volatility']) + \
                         (0.2 * (df['volume'] * df['Close_Difference']) / df['40_day_Moving_Average_of_Volume'])
    df['Final_Alpha_Factor'] = (df['Final_Factor'] * 0.8) + (df['Combined_Weighted_Signal'] * 0.2)
    
    return df['Final_Alpha_Factor']
