```python
def heuristics_v2(data):
    """
    Generate an alpha factor based on the given market data.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    - pd.Series: A Series representing the alpha factor values indexed by date.
    """
    
    # Calculate price momentum using close prices
    momentum = (data['close'] / data['close'].shift(5)) - 1
    
    # Calculate the average volume over the last 5 days
    avg_volume = data['volume'].rolling(window=5).mean()
    
    # Calculate the percentage of amount traded relative to the average volume
    amount_to_volume_ratio = (data['amount'] / (data['volume'] * data['close'])) / avg_volume
    
    # Combine the factors: price momentum and trading activity intensity
    alpha_factor = (momentum + amount_to_volume_ratio) / 2
    
    return alpha_factor
```
