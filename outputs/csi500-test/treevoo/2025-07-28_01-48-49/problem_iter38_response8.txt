```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor that could help predict future stock returns.
    
    The function takes a pandas DataFrame with the following columns:
    - 'open': Opening price of the stock for the day
    - 'high': Highest price of the stock for the day
    - 'low': Lowest price of the stock for the day
    - 'close': Closing price of the stock for the day
    - 'amount': The total amount (price * volume) traded during the day
    - 'volume': The trading volume of the stock for the day
    
    The output is a pandas Series representing the factor values.
    """
    
    # Calculate the true range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    
    # Calculate the average true range over 14 days
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the ratio of the current true range to the average true range
    df['true_range_ratio'] = df['true_range'] / df['avg_true_range']
    
    # Calculate the directional movement
    df['directional_movement'] = df['close'].diff(1)
    
    # Calculate the positive and negative directional movement
    df['positive_dm'] = df['high'].diff(1).apply(lambda x: x if x > 0 else 0)
    df['negative_dm'] = -df['low'].diff(1).apply(lambda x: x if x < 0 else 0)
    
    # Calculate the positive and negative directional indicators
    df['positive_di'] = df['positive_dm'].rolling(window=14).sum() / df['avg_true_range']
    df['negative_di'] = df['negative_dm'].rolling(window=14).sum() / df['avg_true_range']
    
    # Calculate the difference between the positive and negative directional indicators
    df['di_diff'] = df['positive_di'] - df['negative_di']
    
    # Calculate the sum of the positive and negative directional indicators
    df['di_sum'] = df['positive_di'] + df['negative_di']
    
    # Calculate the directional movement index (DMI)
    df['dmi'] = df['di_diff'] / df['di_sum']
    
    # Combine the DMI with the true range ratio to form the final factor
    df['factor'] = df['dmi'] * df['true_range_ratio']
    
    return df['factor']
```
