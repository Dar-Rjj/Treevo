```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Moving Average of Volume
    lookback_period = 20
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    
    # Adjust by Average Volume
    adjusted_range = intraday_range / avg_volume
    
    # Incorporate Close Price Momentum
    momentum_lookback_period = 10
    momentum = df['close'].pct_change(momentum_lookback_period)
    
    # Final Factor Calculation
    factor_value = 0.6 * adjusted_range + 0.4 * momentum
    
    return factor_value
```
