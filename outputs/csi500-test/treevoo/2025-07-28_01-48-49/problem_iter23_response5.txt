import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    adjusted_intraday_range = (intraday_movement * volume_change) / df['volume_ma_20']
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_intraday_range *= price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    momentum = df['close'] / df['close'].shift(momentum_lookback) - 1
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine 5-Day Average High-Low Spread and Momentum
    high_low_spread_momentum = high_low_spread_5d * momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Calculate Volume Weighted Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Calculate IntegratedMomentumFactor
    integrated_momentum_factor = (adjusted_intraday_range + momentum) * high_low_spread_momentum
    
    # Combine Factors with Volume-Adjusted Price Trend
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    combined_factor = (close_to_close_return * intraday_movement) / df['volume_ma_20']
    
    # Filter for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].rolling(window=5).mean()
    
    # Introduce Volume Trend
    volume_sum_5d = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5d
    volume_trend_adjusted_factor = combined_factor * volume_ratio
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * intraday_volatility
    
    # Calculate Dynamic Adjustment Factor
    dynamic_adjustment_factor = df['volume'].ewm(span=20).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Open Price Impact
    relative_open_price_change = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = 0.80 * dynamically_adjusted_momentum + 0.20 * relative_open_price_change
    final_alpha_factor -= final_alpha_factor.ewm(span=20).mean()
    
    return final_alpha_factor
