import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_range.name = 'IntradayRange'

    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = (current_volume - prev_volume).clip(lower=0)
    volume_change.name = 'VolumeChange'

    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = (intraday_range * volume_change).clip(lower=0)
    intraday_momentum_score.name = 'IntradayMomentumScore'

    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()

    # Finalize Alpha Factor Components
    intraday_return = (df['close'] - df['open']) / df['open']
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    high_low_spread_momentum = avg_high_low_spread_5d * five_day_momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum

    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['volume']

    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * intraday_volatility

    # Apply Dynamic Adjustment
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    dynamic_adjusted_momentum = adjusted_momentum / ema_volume

    return dynamic_adjusted_momentum
