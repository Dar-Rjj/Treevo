import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_range.name = 'IntradayRange'
    
    # Adjust by Volume Change and Price Level
    volume_change = df['volume'] / df['volume'].shift(1)
    price_level_adjustment = df['close'] / df['open']
    adjusted_range = intraday_range * volume_change * price_level_adjustment
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_range * volume_change
    intraday_momentum_score.name = 'IntradayMomentumScore'
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    five_day_momentum.name = '5DayMomentum'
    
    high_low_spread_momentum = high_low_spread_5d * five_day_momentum
    high_low_spread_momentum.name = 'HighLowSpreadMomentum'
    
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    integrated_momentum_factor.name = 'IntegratedMomentumFactor'
    
    # Introduce Volume Trend
    five_day_volume_sum = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / five_day_volume_sum
    volume_trend_adjusted_factor = integrated_momentum_factor * volume_ratio
    volume_trend_adjusted_factor.name = 'VolumeTrendAdjustedFactor'
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * integrated_momentum_factor + 0.4 * volume_trend_adjusted_factor
    final_alpha_factor.name = 'FinalAlphaFactor'
    
    return final_alpha_factor
