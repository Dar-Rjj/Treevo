```python
def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high = df['high']
    low = df['low']
    intraday_range = (high - low).clip(lower=0).rename('IntradayRange')
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1).rename('PrevVolume')
    current_volume = df['volume'].rename('CurrentVolume')
    volume_change = (current_volume - prev_volume).clip(lower=0).rename('VolumeChange')
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = (intraday_range * volume_change).rename('IntradayMomentumScore')
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = (high - low)
    five_day_avg_high_low_spread = high_low_spread.rolling(window=5).mean().rename('FiveDayAvgHighLowSpread')
    
    # Calculate 5-Day Momentum
    close_today = df['close']
    close_5_days_ago = df['close'].shift(5)
    five_day_momentum = (close_today - close_5_days_ago) / close_5_days_ago
    
    # Multiply 5-Day Average High-Low Spread by 5-Day Momentum
    high_low_spread_momentum = (five_day_avg_high_low_spread * five_day_momentum).rename('HighLowSpreadMomentum')
    
    # Combine Intraday and High-Low Spread Momentum
    integrated_momentum_factor = (intraday_momentum_score + high_low_spread_momentum).rename('IntegratedMomentumFactor')
    
    return integrated_momentum_factor
```
