```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * df['volume'] * close_diff
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['10_day_Moving_Average_Range'] = df['Daily_Range'].rolling(window=10).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return'] = df['volume'] * df['Intraday_Price_Movement']
    df['Reversal_Signal_Weighted'] = df['amount'] * df['Reversal_Signal']
    df['Combined_Intraday_Return_Reversal'] = df['Weighted_Intraday_Return'] + df['Reversal_Signal_Weighted']
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = df['Weighted_Intraday_Price_Movement'] * df['10_day_Moving_Average_Range'] * df['Volume_Weighted_Return']
    
    # Adjust by Average Volume
    df['30_day_Average_Volume'] = df['volume'].rolling(window=30).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['30_day_Average_Volume']
    
    # Incorporate Close Price Momentum
    df['20_day_MA_Close'] = df['close'].rolling(window=20).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['Momentum_Adjusted'] = (df['close'] - df['20_day_MA_Close']) * df['10_day_Moving_Average_Range']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.6 * (0.4 * df['Weighted_Intraday_Price_Movement'] + 0.3 * df['10_day_Moving_Average_Range'] + 0.3 * df['Momentum_Adjusted'])) + \
                         (0.4 * df['Combined_Intraday_Return_Reversal'])
    
    return df['Final_Factor']
```
