import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Price
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) of Close Price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Rate of Change (ROC) of Close Price
    roc_close = df['close'].pct_change(periods=1)
    
    # Calculate Relative Strength Index (RSI) of Close Price
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_close = 100 - (100 / (1 + rs))
    
    # Calculate On-Balance Volume (OBV)
    obv = ((df['close'] - df['close'].shift(1)) / abs(df['close'] - df['close'].shift(1))).fillna(0) * df['volume']
    obv = obv.cumsum()
    
    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_money_flow = money_flow.where(df['close'] < df['close'].shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum())))
    
    # Calculate True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = tr.rolling(window=14).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate New Highs/New Lows
    new_highs = (df['high'] == df['high'].rolling(window=14).max()).astype(int)
    new_lows = (df['low'] == df['low'].rolling(window=14).min()).astype(int)
    
    # Calculate Directional Movement Index (DMI)
    up_move = df['high'].diff(1)
    down_move = -df['low'].diff(1)
    pos_dm = up_move.where(up_move > down_move, 0)
    neg_dm = down_move.where(down_move > up_move, 0)
    tr_sum = tr.rolling(window=14).sum()
    pos_dm_sum = pos_dm.rolling(window=14).sum()
    neg_dm_sum = neg_dm.rolling(window=14).sum()
    pos_di = 100 * (pos_dm_sum / tr_sum)
    neg_di = 100 * (neg_dm_sum / tr_sum)
    dx = 100 * (abs(pos_di - neg_di) / (pos_di + neg_di))
    adx = dx.rolling(window=14).mean()
    
    # Calculate Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate Cumulative Sum of Daily Returns
    daily_returns = df['close'].pct_change()
    cum_sum_daily_returns = daily_returns.cumsum()
    
    # Calculate Volume-Weighted Close
    volume_weighted_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine all indicators into a single factor
    factor = (
        sma_close + 
        ema_close + 
        roc_close + 
        rsi_close + 
        obv + 
        mfi + 
        atr + 
        high_low_range
