import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Close to Close Return
    df['CloseToCloseReturn'] = df['close'].pct_change()
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['AvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # Generate IntradayMomentumScore
    df['VolumeChange'] = df['volume'].pct_change()
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    
    # Finalize Alpha Factor Components
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    df['5DayMomentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['HighLowSpreadMomentum'] = df['AvgHighLowSpread'] * df['5DayMomentum']
    df['CombinedIntradayMomentum'] = df['IntradayMomentumScore'] + df['IntradayReturn']
    df['IntegratedMomentumFactor'] = df['CombinedIntradayMomentum'] * df['HighLowSpreadMomentum']
    
    # Calculate Volume Weighted Average Price
    df['VWAP'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['VWAP'] = df['VWAP'].cumsum() / df['volume'].cumsum()
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['VolAdjPriceTrend'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VWAP']
    
    # Filter for Recent Price Trend
    df['5DayMA'] = df['close'].rolling(window=5).mean()
    df['RecentPriceTrend'] = df['close'] - df['5DayMA']
    
    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['VolAdjPriceTrend'] * df['VolumeRatio']
    
    # Incorporate Close Price Momentum
    df['10DayMomentum'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Integrate Intraday Volatility
    df['DailyRange'] = df['high'] - df['low']
    df['AvgDailyRange'] = df['DailyRange'].rolling(window=5).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['AdjustedMomentum'] = df['IntegratedMomentumFactor'] * df['AvgDailyRange']
    
    # Calculate Dynamic Adjustment Factor
    df['EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Apply Dynamic Adjustment
    df['DynamicallyAdjustedMomentum'] = df['AdjustedMomentum'] * (df['volume'] / df['EMA_Volume'])
    
    # Calculate Open Price Impact
    df['RelativeOpenPriceChange'] = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Final Alpha Factor
    final_alpha = 0.80 * df['DynamicallyAdjustedMomentum'] + 0.20 * df['RelativeOpenPriceChange']
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    df['FinalAlpha'] = final_alpha - final_alpha.ewm(span=5, adjust=False).mean()
    
    return df['FinalAlpha']
