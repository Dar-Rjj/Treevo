```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_movement = intraday_movement * df['volume'] * close_diff
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return_reversal = (df['volume'] * intraday_movement) + (df['amount'] * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factor = (weighted_intraday_movement * volume_weighted_return) * intraday_volatility
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=40).mean()
    adjusted_combined_factor = combined_factor / avg_volume
    
    # Incorporate Close Price Momentum
    momentum = df['close'].rolling(window=30).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Final Factor Calculation
    final_alpha_factor = (0.5 * adjusted_combined_factor) + (0.25 * intraday_volatility) + (0.25 * adjusted_momentum)
    final_alpha_factor = (final_alpha_factor * 0.7) + (weighted_intraday_return_reversal * 0.3)
    
    return final_alpha_factor
```
