import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Average Volume
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    adjusted_range = intraday_range / avg_volume_20d
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=20).mean()
    
    # Adjust Momentum by Intraday Volatility
    intraday_volatility = intraday_range / df['amount']
    adjusted_momentum = momentum / intraday_volatility
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday and Reversal Signal
    intraday_return = (df['high'] - df['low']) / df['low']
    weighted_intraday_return = intraday_return * df['amount']
    weighted_reversal_signal = reversal_signal * df['volume']
    weighted_signals = (weighted_intraday_return + weighted_reversal_signal) / 2
    
    # Final Alpha Factor
    combined_adjusted_factors = (0.6 * adjusted_range) + (0.4 * adjusted_momentum)
    alpha_factor = (combined_adjusted_factors * 0.5) + (weighted_signals * 0.5)
    
    return alpha_factor
