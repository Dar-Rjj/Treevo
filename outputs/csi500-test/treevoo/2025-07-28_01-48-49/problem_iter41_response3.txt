```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_price_movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    df['weighted_intraday_movement'] = df['intraday_price_movement'] * df['volume'] * close_diff
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    weight = (df['volume'] + df['amount']) / 2
    df['weighted_signal'] = (weight * df['intraday_price_movement']) + (weight * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_factors'] = (df['weighted_intraday_movement'] + df['intraday_volatility']) * (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=35).mean()
    df['adjusted_combined_factors'] = df['combined_factors'] / df['avg_volume']
    
    # Incorporate Close Price Momentum
    df['close_momentum'] = df['close'].rolling(window=30).mean()
    df['adjusted_momentum'] = df['close_momentum'] * df['intraday_volatility']
    
    # Adjusted Price Momentum
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['momentum_adjustment'] = df['ema_10'] - df['close_momentum']
    
    # Final Factor Calculation
    df['final_alpha_factor'] = (0.5 * df['adjusted_combined_factors']) + (0.275 * df['intraday_volatility']) + (0.225 * df['momentum_adjustment'])
    final_factor = (df['final_alpha_factor'] * 0.75) + (df['weighted_signal'] * 0.25)
    
    return final_factor
```
