```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low price, then normalize it with the volume
    df['hl_diff_vol'] = (df['high'] - df['low']) / df['volume']
    
    # Calculate the percentage change in amount
    df['amount_change'] = df['amount'].pct_change()
    
    # Calculate the weighted sum of close change and amount change
    df['weighted_sum'] = (df['close_change'] * 0.6 + df['amount_change'] * 0.4)
    
    # Final factor: Combine the weighted sum with the high-low difference over volume
    factor = df['weighted_sum'] + df['hl_diff_vol']
    
    return factor.dropna()
```
