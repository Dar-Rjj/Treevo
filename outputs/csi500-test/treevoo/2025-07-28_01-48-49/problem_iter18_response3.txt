```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_price_movement * volume_change
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5day = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Finalize Alpha Factor Components
    close_today = df['close']
    open_today = df['open']
    
    # Calculate Intraday Return
    intraday_return = (close_today - open_today) / open_today
    
    # Calculate 5-Day Momentum
    close_5days_ago = df['close'].shift(5)
    five_day_momentum = (close_today - close_5days_ago) / close_5days_ago
    
    # Multiply 5-Day Average High-Low Spread by 5-Day Momentum
    high_low_spread_momentum = high_low_spread_5day * five_day_momentum
    
    # Combine Intraday and High-Low Spread Momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=5).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * average_daily_range
    
    # Calculate Dynamic Adjustment Factor
    n_day_ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    dynamic_adjustment_factor = df['volume'] / n_day_ema_volume
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Open Price Impact
    open_price_impact = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = 0.80 * dynamically_adjusted_momentum + 0.20 * open_price_impact
    
    # Subtract N-day Exponential Moving Average of Final Alpha Factor
    n_day_ema_final_alpha = final_alpha_factor.ewm(span=5, adjust=False).mean()
    final_alpha_factor -= n_day_ema_final_alpha
    
    return final_alpha_factor
```
