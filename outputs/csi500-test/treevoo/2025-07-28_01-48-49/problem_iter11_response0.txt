```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the relative strength, which is the ratio of the average price to the price range
    relative_strength = avg_price / price_range
    
    # Calculate the volume-to-amount ratio, which gives an indication of the average transaction size
    vol_to_amt_ratio = df['volume'] / df['amount']
    
    # The final alpha factor is a combination of the relative strength and the volume-to-amount ratio
    alpha_factor = relative_strength * vol_to_amt_ratio
    
    return alpha_factor
```
