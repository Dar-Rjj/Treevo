import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_intraday_movement = intraday_movement * (df['volume'] * close_diff)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_signal_weight = df['volume']
    weighted_intraday_and_reversal = (intraday_return_weight * weighted_intraday_movement) + (reversal_signal_weight * reversal_signal)
    
    # Combine Enhanced Factors
    combined_factor = (weighted_intraday_movement + avg_daily_range) * volume_weighted_return
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_combined_factor = combined_factor / avg_volume
    
    # Incorporate Close Price Momentum
    momentum = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * avg_daily_range
    
    # Final Alpha Factor
    final_alpha = (0.3 * weighted_intraday_movement + 0.3 * avg_daily_range + 0.4 * adjusted_momentum) * 0.5 + (weighted_intraday_and_reversal * 0.5)
    
    return final_alpha
