```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Next-Open Difference
    df['close_to_next_open'] = df['close'].shift(-1) - df['open'].shift(-1)
    
    # Apply Weight by Volume
    df['volume_weighted_diff'] = df['volume'] * df['close_to_next_open']
    
    # Calculate Sum of Weighted Differences over a 5-day rolling window
    df['sum_weighted_diff'] = df['volume_weighted_diff'].rolling(window=5).sum()
    
    # Calculate Exponential Moving Average of Weighted Sums with a smoothing factor of 10 days
    df['ema_weighted_sum'] = df['sum_weighted_diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate Standard Deviation of Returns over a 14-day period
    df['returns'] = df['close'].pct_change()
    df['std_returns'] = df['returns'].rolling(window=14).std()
    
    # Normalize the Alpha Factor by Dividing by Standard Deviation
    df['alpha_factor'] = df['ema_weighted_sum'] / df['std_returns']
    
    return df['alpha_factor']
```
