import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    df['5_day_avg_intraday_range'] = df['intraday_range'].rolling(window=5).mean()

    # Calculate 5-Day Return
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Determine Intraday Volatility Adjusted Momentum
    df['vol_adj_momentum'] = df['5_day_return'] * df['5_day_avg_intraday_range']

    # Compute Intraday Reversal Signal
    df['intraday_reversal_signal'] = (df['close'] - df['open']) / df['5_day_avg_intraday_range']

    # Incorporate Volume Impact
    df['5_day_cum_vol'] = df['volume'].rolling(window=5).sum()
    df['volume_weight'] = df['volume'] / df['5_day_cum_vol']

    # Final Alpha Factor Synthesis
    df['alpha_factor'] = (df['vol_adj_momentum'] + df['intraday_reversal_signal']) * df['volume_weight']

    return df['alpha_factor']
