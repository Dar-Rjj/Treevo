```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Movement'] = df['Intraday_Price_Movement'] * df['volume'] * df['Close_Difference']
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['20_day_MA_Range'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return'] = (df['volume'] * df['Intraday_Price_Movement']) + (df['amount'] * df['Reversal_Signal'])
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = df['Weighted_Intraday_Movement'] * df['20_day_MA_Range'] * df['Volume_Weighted_Return']
    
    # Adjust by Average Volume
    df['40_day_MA_Volume'] = df['volume'].rolling(window=40).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['40_day_MA_Volume']
    
    # Incorporate Close Price Momentum
    df['50_day_MA_Close'] = df['close'].rolling(window=50).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = (df['close'] - df['50_day_MA_Close']) * df['20_day_MA_Range']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.8 * df['Adjusted_Combined_Factor']) + (0.2 * df['Weighted_Intraday_Return'])
    final_factor = 0.6 * df['Intraday_Price_Movement'] + 0.2 * df['20_day_MA_Range'] + 0.2 * df['Adjusted_Momentum']
    final_factor = (0.8 * final_factor) + (0.2 * df['Weighted_Intraday_Return'])
    
    return final_factor
```
