```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open + close) / 2 for a more stable price representation
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the money flow using the average price and volume
    money_flow = avg_price * df['volume']
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate a weighted moving average of the daily return, using the money flow as weights
    weighted_moving_avg = (daily_return * money_flow).rolling(window=10).sum() / money_flow.rolling(window=10).sum()
    
    # Calculate the factor as the difference between the current day's daily return and the weighted moving average
    factor = daily_return - weighted_moving_avg
    
    return factor
```
