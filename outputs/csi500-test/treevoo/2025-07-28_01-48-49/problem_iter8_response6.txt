import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['volume']
    
    # Calculate Momentum (N-day Exponential Moving Average of Close price)
    n = 10  # N-day window
    momentum = df['close'].ewm(span=n, adjust=False).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate Dynamic Adjustment Factor (N-day Exponential Moving Average of Volume)
    dynamic_adjustment_factor = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    volume_change = df['volume'] - prev_volume
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_range * volume_change
    
    # Combine Dynamically Adjusted Momentum and IntradayMomentumScore
    combined_momentum = dynamically_adjusted_momentum + intraday_momentum_score
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5day = (df['high'] - df['low']).rolling(window=5).sum() / 5
    
    # Finalize 5-Day Momentum
    close_5_days_ago = df['close'].shift(5)
    five_day_momentum = (df['open'] - close_5_days_ago) / close_5_days_ago
    high_low_spread_momentum = high_low_spread_5day * five_day_momentum
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum + high_low_spread_momentum
    
    return final_alpha_factor
