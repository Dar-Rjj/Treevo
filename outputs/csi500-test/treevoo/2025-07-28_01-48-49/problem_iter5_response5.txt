```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Next-Open Difference
    df['close_to_next_open'] = df['close'].shift(-1) - df['open']
    
    # Adjust High-to-Low Movement by Open Adjustment
    df['open_difference'] = df['open'] - df['open'].shift(1)
    df['adjusted_high_to_low'] = df['daily_range'] / df['open_difference']
    
    # Weight by Volume and Close Difference
    df['close_difference'] = df['close'] - df['close'].shift(1)
    df['weighted_high_to_low'] = df['adjusted_high_to_low'] * (df['volume'] * df['close_difference'])
    
    # Aggregate Price Movement with Volume
    df['volume_weighted_close_to_next_open'] = df['volume'] * df['close_to_next_open']
    df['sum_weighted_diff'] = df['volume_weighted_close_to_next_open'].rolling(window=5).sum()
    
    # Enhance with Exponential Moving Average Filter
    smoothing_factor = 10
    df['ema_weighted_sum'] = df['sum_weighted_diff'].ewm(span=smoothing_factor, adjust=False).mean()
    
    # Adjust for Volatility Impact
    df['returns'] = df['close'].pct_change()
    df['volatility'] = df['returns'].rolling(window=14).std()
    df['normalized_ema'] = df['ema_weighted_sum'] / df['volatility']
    
    # Combine Enhanced Factors
    df['intraday_volatility'] = df['daily_range'].rolling(window=5).mean()
    df['volume_weighted_return'] = df['volume'] * (df['close'] - df['open']) / df['open']
    df['alpha_factor'] = df['weighted_high_to_low'] * df['intraday_volatility'] * df['volume_weighted_return']
    
    return df['alpha_factor']
```
