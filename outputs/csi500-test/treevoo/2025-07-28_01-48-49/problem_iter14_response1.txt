```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low price for the day
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the percentage of the daily range that the closing price is above the opening price
    df['close_above_open_pct'] = (df['close'] - df['open']) / df['daily_range']
    
    # Calculate the amount per volume traded
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['close_above_open_pct'] * df['amount_per_volume']
    
    return df['alpha_factor']
```
