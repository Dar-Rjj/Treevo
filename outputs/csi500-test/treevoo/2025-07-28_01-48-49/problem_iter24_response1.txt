import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    df['VolumeChange'] = df['volume'].pct_change()
    df['PriceLevelRatio'] = df['close'] / df['open']
    df['AdjustedRange'] = (df['IntradayRange'] * df['VolumeChange']) * df['PriceLevelRatio']
    
    # Close Price Momentum
    lookback_period = 10
    df['Momentum'] = df['close'].pct_change(lookback_period)
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    
    # Calculate 5-Day Average High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    df['5DayAvgHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    df['5DayMomentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['HighLowSpreadMomentum'] = df['5DayAvgHighLowSpread'] * df['5DayMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Calculate Volume Weighted Average Price
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP'] = vwap
    
    # Combine Factors with Volume-Adjusted Price Trend
    df['CloseToCloseReturn'] = df['close'].pct_change()
    df['VolumeAdjustedPriceTrend'] = df['CloseToCloseReturn'] * df['IntradayRange'] / df['VWAP']
    
    # Filter for Recent Price Trend
    df['5DayMA'] = df['close'].rolling(window=5).mean()
    df['RecentPriceTrend'] = df['close'] - df['5DayMA']
    
    # Introduce Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['VolumeAdjustedPriceTrend'] * df['VolumeRatio']
    
    # Final Factor Calculation
    final_factor = (
        0.2 * df['AdjustedRange'] +
        0.3 * df['Momentum'] +
        0.2 * df['IntegratedMomentumFactor'] +
        0.1 * df['VolumeTrendAdjustedFactor']
    )
    
    # Introduce Relative Strength Index (RSI) and Money Flow Index (MFI)
    def rsi(series, n=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def mfi(high, low, close, volume, n=14):
        typical_price = (high + low + close) / 3
        money_flow = typical_price * volume
        positive_flow = money_flow.copy()
        negative_flow = money_flow.copy()
        positive_flow[typical_price < typical_price.shift(1)] = 0
        negative_flow[typical_price > typical_price.shift(1)] = 0
        positive_mf = positive_flow.rolling(window=n).sum()
        negative_mf = negative_flow.rolling(window=n).sum()
        mfi = 100 - (100 / (1 + (positive_mf / negative_mf)))
        return mfi
    
    df['RSI'] = rsi(df['close'])
    df['MFI'] = mfi(df['high'], df['low'], df['close'], df['volume'])
    
    # Include RSI and MFI in Final Factor Calculation
    final_factor = (
        0.2 * df['AdjustedRange'] +
        0.3 * df['Momentum'] +
        0.2 * df['IntegratedMomentumFactor'] +
        0.1 * df['VolumeTrendAdjustedFactor'] +
        0.1 * df['RSI']
