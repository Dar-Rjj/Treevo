```python
def heuristics_v2(df):
    # Calculate the price movement from open to close
    price_movement = df['close'] - df['open']
    
    # Calculate the money flow using amount (value traded) and volume
    money_flow = df['amount'] / df['volume']
    
    # Calculate a modified on-balance volume indicator, using the price movement as a multiplier
    on_balance_volume = (price_movement / df['close']).fillna(0).cumsum()
    
    # Generate a factor that combines the on-balance volume and money flow
    factor = on_balance_volume * money_flow
    
    return factor
```
