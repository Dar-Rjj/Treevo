```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_price_move'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['weighted_intraday_move'] = df['intraday_price_move'] * df['volume'] * df['close_diff']
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['volume_weighted_return'] = df['volume'] * df['reversal_signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_and_reversal'] = (df['volume'] * df['intraday_price_move']) + (df['amount'] * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_factor'] = df['weighted_intraday_move'] * df['intraday_volatility'] * df['volume_weighted_return']
    
    # Adjust by Average Volume
    df['avg_volume_40d'] = df['volume'].rolling(window=40).mean()
    df['adjusted_combined_factor'] = df['combined_factor'] / df['avg_volume_40d']
    
    # Incorporate Close Price Momentum
    df['close_momentum_30d'] = df['close'].rolling(window=30).mean()
    df['momentum_adjusted_by_volatility'] = df['close_momentum_30d'] * df['intraday_volatility']
    
    # Adjusted Price Momentum
    df['ema_close_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['adjusted_price_momentum'] = df['ema_close_10d'] - df['close_momentum_30d']
    
    # Final Factor Calculation
    df['final_factor'] = (0.8 * (0.5 * df['adjusted_combined_factor'] + 0.25 * df['intraday_volatility'] + 0.25 * df['adjusted_price_momentum'])) + (0.2 * df['weighted_intraday_and_reversal'])
    
    return df['final_factor']
```
