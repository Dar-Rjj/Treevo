import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Average Volume
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    adjusted_range = intraday_range / avg_vol_20
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['amount']
    
    # Calculate Momentum
    momentum = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_weight = df['amount']
    reversal_weight = df['volume']
    weighted_intraday_return_reversal = (intraday_weight * intraday_return) + (reversal_weight * reversal_signal)
    
    # Combine Adjusted Range and Adjusted Momentum
    combined_adjusted = 0.6 * adjusted_range + 0.4 * adjusted_momentum
    
    # Final Alpha Factor
    final_factor = 0.5 * combined_adjusted + 0.5 * weighted_intraday_return_reversal
    
    return final_factor
