```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Return
    high_low_return = df['High'] - df['Low']
    
    # Calculate Intraday Return
    intraday_return = (df['High'] - df['Low']) / df['Low']
    
    # Calculate Close-to-Open Reversal Signal
    close_open_reversal = (df['Close'] - df['Open']) / df['Open']
    
    # Combine Intraday and Reversal Signals
    intraday_return_weight = df['Amount']
    reversal_signal_weight = df['Volume']
    combined_signal = (intraday_return * intraday_return_weight) + (close_open_reversal * reversal_signal_weight)
    
    # Calculate High-Low Momentum
    high_low_momentum = high_low_return.rolling(window=10).sum()
    
    # Calculate Volume Weight
    volume_weight = df['Volume'].rolling(window=10).sum()
    
    # Final Alpha Construction
    alpha = combined_signal + (high_low_momentum * volume_weight)
    
    return alpha
```
