```python
import pandas as pd

def heuristics_v2(df):
    """
    The function takes a pandas DataFrame with columns: open, high, low, close, amount, volume.
    It calculates a novel alpha factor based on the given market features.
    
    :param df: pandas DataFrame with (open, high, low, close, amount, volume) as columns and (date) as index.
    :return: pandas Series representing the factor values indexed by (date).
    """
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in closing price
    close_change = df['close'].pct_change()
    
    # Calculate the money flow ratio, which is the ratio of the product of close, volume, and price diff to the amount
    money_flow_ratio = (df['close'] * df['volume'] * price_diff) / df['amount']
    
    # Generate the alpha factor by combining the money flow ratio with the momentum represented by the close price change
    alpha_factor = money_flow_ratio * close_change.shift(1)
    
    return alpha_factor
```
