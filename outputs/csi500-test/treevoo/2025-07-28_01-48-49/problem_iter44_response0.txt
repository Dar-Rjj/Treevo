def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the upper shadow and lower shadow
    upper_shadow = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow = df[['open', 'close']].min(axis=1) - df['low']
    
    # Calculate the body of the candle
    body = abs(df['close'] - df['open'])
    
    # Calculate the ratio of the body to the price range
    body_to_range_ratio = body / price_range
    
    # Calculate the volume weighted amount
    volume_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the final factor
    factor = (body_to_range_ratio * volume_weighted_amount) * (upper_shadow - lower_shadow)
    
    return factor
