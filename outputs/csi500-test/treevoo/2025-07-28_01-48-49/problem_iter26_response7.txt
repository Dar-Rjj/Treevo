    df['Prev_Open'] = df['open'].shift(1)
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['Prev_Open'])

    df['Adjusted_Intraday_Price_Movement'] = df['Intraday_Price_Movement']

    df['Intraday_Range'] = df['high'] - df['low']

    df['Volume_Close_Diff'] = df['volume'] * (df['close'] - df['Prev_Close'])

    df['MA_Volume_20'] = df['volume'].rolling(window=20).mean()

    df['Price_Level_Ratio'] = df['close'] / df['open']
    df['Adjusted_Volume_Range'] = df['Volume_Adjusted_Range'] * df['Price_Level_Ratio']

    momentum_lookback = 10
    df['Momentum'] = df['close'] / df['close'].shift(momentum_lookback) - 1

    df['High_Low_Spread'] = df['high'] - df['low']
    df['Avg_High_Low_Spread_5'] = df['High_Low_Spread'].rolling(window=5).mean()

    df['Combined_High_Low_Spread_Momentum'] = df['Avg_High_Low_Spread_5'] * df['Momentum']
