```python
def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (2 * df['close'] - df['low'] - df['high']) / (df['high'] - df['low'])
    
    # Calculate the money flow
    df['money_flow'] = df['close'] * df['volume']
    
    # Calculate the factor as a combination of the above metrics
    df['factor'] = (df['range'] + df['buying_pressure'] * 10) * df['money_flow'] / df['amount']
    
    return df['factor']
```
