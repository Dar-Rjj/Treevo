```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['open']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mf_multiplier * df['volume']
    
    # Calculate the cumulative money flow volume
    cmf = mf_volume.rolling(window=14).sum()
    
    # Calculate the factor value
    factor_value = cmf / df['volume'].rolling(window=14).sum()
    
    return factor_value
```
