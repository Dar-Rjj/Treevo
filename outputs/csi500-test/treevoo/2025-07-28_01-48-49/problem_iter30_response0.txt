```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return factors
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Explore price-based alpha factors
    df['price_range'] = df['high'] - df['low']
    df['close_to_open_ratio'] = df['close'] / df['open']
    df['avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Analyze trading activity with volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_to_close_ratio'] = df['volume'] / df['close']
    df['volume_deviation'] = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Investigate momentum and trend following
    df['short_ma'] = df['close'].rolling(window=5).mean()
    df['long_ma'] = df['close'].rolling(window=20).mean()
    df['ma_diff_short'] = df['close'] - df['short_ma']
    df['ma_diff_long'] = df['close'] - df['long_ma']
    df['momentum_short'] = df['daily_return'].rolling(window=5).mean()
    
    # Examine volatility and risk
    df['std_daily_returns'] = df['daily_return'].rolling(window=10).std()
    df['abs_sum_daily_returns'] = df['daily_return'].rolling(window=10).apply(lambda x: abs(x).sum())
    
    # Consider breakout and reversal patterns
    df['above_high'] = df['close'] > df['high'].rolling(window=5).max()
    df['below_low'] = df['close'] < df['low'].rolling(window=5).min()
    df['above_short_ma_days'] = (df['close'] > df['short_ma']).rolling(window=5).sum()
    
    # Evaluate the relationship between amount and volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    df['amount_volume_change'] = df['amount_volume_ratio'] - df['amount_volume_ratio'].shift(1)
    df['amount_deviation'] = df['amount'] - df['amount'].rolling(window=5).mean()
    
    # Analyze the impact of gaps
    df['up_gap'] = df['open'] > df['close'].shift(1)
    df['down_gap'] = df['open'] < df['close'].shift(1)
    df['gap_size'] = (df['open'] - df['close'].shift(1)).abs()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['ma_diff_short'] + 
        df['momentum_short'] + 
        df['volume_to_close_ratio'] + 
        df['price_range'] + 
        df['std_daily_returns'] * -1 + 
        df['amount_volume_ratio'] + 
        df['gap_size'] * df['up_gap'].astype(int) - df['gap_size'] * df['down_gap'].astype(int)
    )
    
    return df['alpha_factor']
```
