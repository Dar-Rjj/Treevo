```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['avg_high_low_spread_5d'] = df['high_low_spread'].rolling(window=5).mean()

    # Calculate 5-Day Momentum
    df['momentum_5d'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply High-Low Spread and Momentum
    df['spread_momentum'] = df['avg_high_low_spread_5d'] * df['momentum_5d']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Close to Close Return
    df['close_to_close_return'] = df['close'] / df['close'].shift(1) - 1

    # Calculate Volume Weighted Average Price
    df['vwap'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['vwap'] = df['vwap'].rolling(window=1).sum() / df['volume'].rolling(window=1).sum()

    # Combine Factors
    df['combined_factor'] = df['close_to_close_return'] * df['intraday_range'] / df['vwap']

    # Filter for Recent Price Trend
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['price_trend_filter'] = df['close'] - df['5_day_ma_close']

    # Introduce Volume Trend
    df['5_day_sum_volume'] = df['volume'].rolling(window=5).sum()
    df['volume_ratio'] = df['volume'] / df['5_day_sum_volume']

    # Final Factor Computation
    df['final_factor'] = (df['combined_factor'] * df['volume_ratio'] * df['spread_momentum'])

    return df['final_factor'].dropna()
```
