```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Adjust by Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['VolumeAdjustedRange'] = df['IntradayRange'] * df['VolumeChange']
    
    # Close Price Momentum (lookback period of 10 days)
    df['ClosePriceMomentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Integrate Intraday Price Movement and Volume
    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']
    
    # Calculate 5-Day Average High-Low Spread
    df['AvgHighLowSpread'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # 5-Day Momentum
    df['5DayMomentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # High-Low Spread Momentum
    df['HighLowSpreadMomentum'] = df['AvgHighLowSpread'] * df['5DayMomentum']
    
    # Integrated Momentum Factor
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Volume-Adjusted Price Trend
    df['VWAP'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['VWAP'] = df['VWAP'].cumsum() / df['volume'].cumsum()
    df['CloseToCloseReturn'] = df['close'] / df['close'].shift(1) - 1
    df['VolumeAdjustedPriceTrend'] = df['CloseToCloseReturn'] * df['IntradayRange'] / df['VWAP']
    
    # Recent Price Trend
    df['5DayMA'] = df['close'].rolling(window=5).mean()
    df['RecentPriceTrend'] = df['close'] - df['5DayMA']
    
    # Volume Trend
    df['5DaySumVolume'] = df['volume'].rolling(window=5).sum()
    df['VolumeRatio'] = df['volume'] / df['5DaySumVolume']
    df['VolumeTrendAdjustedFactor'] = df['VolumeAdjustedPriceTrend'] * df['VolumeRatio']
    
    # Price-Volatility Interaction
    df['5DayStdDev'] = df['close'].rolling(window=5).std()
    df['PriceVolatilityScore'] = df['5DayStdDev'] * df['5DayMomentum']
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Final Factor Calculation
    weights = {
        'IntradayMomentumScore': 0.2,
        'ClosePriceMomentum': 0.3,
        'IntegratedMomentumFactor': 0.2,
        'VolumeTrendAdjustedFactor': 0.2,
        'PriceVolatilityScore': 0.1,
        'RSI': 0.05
    }
    
    alpha_factor = (
        df['IntradayMomentumScore'] * weights['IntradayMomentumScore'] +
        df['ClosePriceMomentum'] * weights['ClosePriceMomentum'] +
        df['IntegratedMomentumFactor'] * weights['IntegratedMomentumFactor'] +
        df['VolumeTrendAdjustedFactor'] * weights['VolumeTrendAdjustedFactor'] +
        df['PriceVolatilityScore'] * weights['PriceVolatilityScore'] +
        df['RSI'] * weights['RSI']
    )
    
    return alpha_factor
```
