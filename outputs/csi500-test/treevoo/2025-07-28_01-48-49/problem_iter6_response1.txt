```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_range = intraday_range.apply(lambda x: max(x, 0))  # Ensure non-negative
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = current_volume - prev_volume
    volume_change = volume_change.apply(lambda x: max(x, 0))  # Ensure non-negative
    
    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_range * volume_change
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    five_day_avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Finalize Alpha Factor
    five_day_momentum = (df['open'] - df['close'].shift(5)) / df['close'].shift(5)
    high_low_spread_momentum = five_day_avg_high_low_spread * five_day_momentum
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    
    return integrated_momentum_factor
```
