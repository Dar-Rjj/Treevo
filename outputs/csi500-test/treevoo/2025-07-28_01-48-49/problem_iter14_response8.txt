```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Average Volume
    lookback_period = 20
    volume_moving_avg = df['volume'].rolling(window=lookback_period).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_range = (intraday_range * volume_change) / volume_moving_avg
    
    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    adjusted_price_range = adjusted_range * price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback_period = 10
    momentum = (df['close'] / df['close'].shift(momentum_lookback_period)) - 1
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Combine 5-Day Average High-Low Spread and Momentum
    combined_high_low_spread = avg_high_low_spread * momentum
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=lookback_period).mean()
    
    # Incorporate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_open_return
    
    # Final Factor Calculation
    factor = (
        0.25 * adjusted_price_range +
        0.25 * momentum +
        0.2 * combined_high_low_spread +
        0.15 * avg_daily_range +
        0.15 * volume_weighted_return
    )
    
    return factor
```
