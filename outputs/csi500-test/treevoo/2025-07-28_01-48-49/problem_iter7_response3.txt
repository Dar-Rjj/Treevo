```python
import pandas as pd

def heuristics_v2(df, window=10):
    # Calculate High-Low Return
    high_low_return = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Reversal Signals with weights
    combined_signal = (intraday_return * df['amount']) + (reversal_signal * df['volume'])
    
    # Calculate High-Low Momentum over a window
    high_low_momentum = high_low_return.rolling(window=window).sum()
    
    # Calculate Volume Weight over a window
    volume_weight = df['volume'].rolling(window=window).sum()
    
    # Final Alpha Construction
    alpha = combined_signal + (high_low_momentum * volume_weight)
    
    return alpha
```
