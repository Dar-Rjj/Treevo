```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['intraday_price_movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['close_diff'] = df['close'] - df['close'].shift(1)
    df['weighted_intraday_price_movement'] = df['intraday_price_movement'] * df['volume'] * df['close_diff']
    
    # Integrate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['daily_range'].ewm(span=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['reversal_signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['volume_weighted_return'] = df['volume'] * df['reversal_signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['weighted_intraday_return'] = (df['amount'] * df['weighted_intraday_price_movement']) + (df['volume'] * df['reversal_signal'])
    
    # Combine Enhanced Factors
    df['combined_factors'] = df['weighted_intraday_price_movement'] + df['intraday_volatility'] + df['volume_weighted_return']
    
    # Adjust by Relative Volume
    df['relative_volume'] = df['volume'] / df['volume'].ewm(span=20).mean()
    df['adjusted_combined_factors'] = df['combined_factors'] / df['relative_volume']
    
    # Incorporate Close Price Momentum
    df['momentum'] = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['momentum'] / df['intraday_volatility']
    
    # Final Factor Calculation
    df['final_factor'] = 0.6 * (0.3 * df['weighted_intraday_price_movement'] + 
                                0.4 * df['intraday_volatility'] + 
                                0.3 * df['adjusted_momentum']) + \
                         0.4 * df['weighted_intraday_return']
    
    return df['final_factor']
```
