```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['Adjusted_Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))

    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Adjusted_Intraday_Price_Movement'] = df['Adjusted_Intraday_Price_Movement'] * df['volume'] * df['Close_Difference']

    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = df['Daily_Range'].rolling(window=20).mean()

    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']

    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Reversal_Signal'] = df['amount'] * df['Adjusted_Intraday_Price_Movement'] + df['volume'] * df['Reversal_Signal']

    # Combine Enhanced Factors
    df['Combined_Factor'] = df['Weighted_Adjusted_Intraday_Price_Movement'] * df['Intraday_Volatility'] * df['Close_Difference'] * df['volume']

    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['Average_Volume']

    # Incorporate Close Price Momentum
    df['Momentum'] = df['close'].rolling(window=10).mean()

    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['Momentum'] * df['Intraday_Volatility']

    # Final Factor Calculation
    df['Final_Alpha_Factor'] = 0.6 * ((0.4 * df['Weighted_Adjusted_Intraday_Price_Movement']) + 
                                       (0.3 * df['Intraday_Volatility']) + 
                                       (0.3 * df['Adjusted_Momentum'])) + \
                                0.4 * df['Weighted_Reversal_Signal']

    return df['Final_Alpha_Factor']
```
