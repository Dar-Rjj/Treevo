```python
def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['5Day_Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=5).mean()

    # Calculate 5-Day Momentum
    df['5Day_Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply the two factors
    df['Factor_Value'] = df['5Day_Avg_High_Low_Spread'] * df['5Day_Momentum']

    return df['Factor_Value']
```
