```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Average Intraday Range
    intraday_range = df['high'] - df['low']
    avg_intraday_range = intraday_range.rolling(window=10).mean()

    # Calculate 10-Day Return
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)

    # Determine Intraday Volatility Adjusted Momentum
    adjusted_momentum = ten_day_return * avg_intraday_range

    # Compute Intraday Reversal Signal
    intraday_reversal = (df['close'] - df['open']) / avg_intraday_range

    # Incorporate Volume Impact
    ten_day_cumulative_volume = df['volume'].rolling(window=10).sum()
    volume_weight = df['volume'] / ten_day_cumulative_volume

    # Final Alpha Factor Synthesis
    alpha_factor = (adjusted_momentum + intraday_reversal) * volume_weight

    return alpha_factor
```
