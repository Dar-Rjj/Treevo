```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close price to open price, high to low, and volume to amount
    close_open_ratio = df['close'] / df['open']
    high_low_ratio = df['high'] / df['low']
    vol_amount_ratio = df['volume'] / df['amount']

    # Compute the difference between high and low prices
    high_low_diff = df['high'] - df['low']

    # Calculate a weighted sum of the ratios and the difference
    factor = 0.4 * (close_open_ratio) + 0.3 * (high_low_ratio) + 0.2 * (vol_amount_ratio) + 0.1 * (high_low_diff)

    return factor
```
