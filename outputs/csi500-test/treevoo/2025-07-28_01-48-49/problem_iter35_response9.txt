```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Momentum Difference and Direction
    df['Momentum_Difference'] = df['SMA_5'] - df['SMA_20']
    df['Momentum_Direction'] = (df['SMA_5'] > df['SMA_20']).astype(int) * 2 - 1

    # Recent Price Changes
    df['Price_Change'] = df['close'] - df['open']
    df['Percentage_Change'] = (df['close'] - df['open']) / df['open']

    # Volume Analysis
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_20']
    df['Volume_Trend'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1

    # Volume Adjusted Metrics
    df['Volume_Adjusted_Price_Change'] = df['Price_Change'] * df['Volume_Ratio']
    df['Volume_Adjusted_Momentum'] = df['Momentum_Difference'] * df['Volume_Ratio']

    # Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Relative_Range'] = df['Daily_Range'] / df['close']

    # Historical Volatility
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()
    df['Volatility_Spike'] = (df['Daily_Range'] > 1.5 * df['Avg_Daily_Range']).astype(int)

    # Combined Indicators
    df['Momentum_Volatility_Score'] = df['Momentum_Difference'] * (1 + df['Volatility_Spike'])
    df['Comprehensive_Alpha_Factor'] = (df['Momentum_Direction'] * df['Volume_Adjusted_Price_Change']) + (df['Relative_Range'] * df['Volume_Ratio'])

    df['Multi_Indicator_Alpha'] = (df['Momentum_Difference'] * df['Volume_Ratio'] * df['Volatility_Spike']) + (df['Price_Change'] * df['Volume_Trend'])

    df['High_Low_Momentum'] = (df['high'] - df['low']) * df['Momentum_Direction']
    df['Volume_Adjusted_High_Low_Momentum'] = df['High_Low_Momentum'] * df['Volume_Ratio']

    # Compound Alpha Factor
    df['Compound_Alpha'] = (df['Momentum_Volatility_Score'] * df['Volume_Adjusted_High_Low_Momentum']) + (df['Comprehensive_Alpha_Factor'] * df['Multi_Indicator_Alpha'])

    # Dynamic Alpha Factor
    df['Volume_Weighted_Momentum'] = (df['SMA_5'] - df['SMA_20']) * df['Volume_Ratio']
    df['Dynamic_Alpha_Factor'] = (df['Volume_Weighted_Momentum'] * df['Momentum_Direction']) + (df['Relative_Range'] * df['Volume_Adjusted_Price_Change'])

    return df['Dynamic_Alpha_Factor']
```
