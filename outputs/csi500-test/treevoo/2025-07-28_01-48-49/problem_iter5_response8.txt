```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change
    # Calculate Moving Average of Volume
    lookback_period = 20
    df['volume_ma'] = df['volume'].rolling(window=lookback_period).mean()
    # Calculate Current Day Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    # Adjusted Intraday Range
    adjusted_range = intraday_range / df['volume_change']
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    df['momentum'] = (df['close'] / df['close'].shift(momentum_lookback) - 1) * 100
    
    # Calculate Volume Weighted Average Price
    df['vwap'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Combine Factors
    combined_factor = (df['close'] / df['close'].shift(1) - 1) * intraday_range / df['vwap']
    
    # Filter for Recent Price Trend
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    recent_trend = df['close'] - df['5_day_ma']
    
    # Introduce Volume Trend
    df['5_day_volume_sum'] = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / df['5_day_volume_sum']
    
    # Final Factor Calculation
    final_factor = 0.3 * adjusted_range + 0.3 * df['momentum'] + 0.4 * combined_factor * recent_trend * volume_ratio
    
    return final_factor
```
