```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Average of Daily Ranges
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Combine Intraday Volatility and Volume-Weighted Return
    combined_factor = avg_daily_range * volume_weighted_return
    
    return combined_factor
```
