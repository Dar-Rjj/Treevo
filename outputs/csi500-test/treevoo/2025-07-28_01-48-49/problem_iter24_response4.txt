def heuristics_v2(df):

    df['IntradayRange'] = df['high'] - df['low']

    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['PriceLevelRatio'] = df['close'] / df['open']

    lookback_period = 10
    df['Momentum'] = df['close'].pct_change(periods=lookback_period)

    df['IntradayMomentumScore'] = df['IntradayRange'] * df['VolumeChange']

    df['HighLowSpread'] = df['high'] - df['low']
    df['AverageHighLowSpread'] = df['HighLowSpread'].rolling(window=5).mean()

    df['Close_5D_Ago'] = df['close'].shift(5)
    df['5DayMomentum'] = (df['close'] - df['Close_5D_Ago']) / df['Close_5D_Ago']
    df['HighLowSpreadMomentum'] = df['AverageHighLowSpread'] * df['5DayMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayMomentumScore'] + df['HighLowSpreadMomentum']
    
    # Calculate Volume Weighted Average Price
    df['VWAP'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume'] / df['volume'].sum()

    df['CloseToCloseReturn'] = df['close'].pct_change()

        avg_loss = loss.rolling(window=period).mean()
        rs = avg_gain / avg_loss
