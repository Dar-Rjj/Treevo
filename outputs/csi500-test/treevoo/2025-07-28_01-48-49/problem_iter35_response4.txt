```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * df['volume'] * df['Close_Difference']
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return_and_Reversal_Signal'] = (df['volume'] * df['Intraday_Price_Movement']) + (df['amount'] * df['Reversal_Signal'])
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = (df['Weighted_Intraday_Price_Movement'] * df['Intraday_Volatility'] * df['Volume_Weighted_Return'])
    
    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=40).mean()
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] / df['Average_Volume']
    
    # Incorporate Close Price Momentum
    df['Momentum'] = df['close'].rolling(window=30).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['Momentum'] * df['Intraday_Volatility']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.5 * df['Weighted_Intraday_Price_Movement'] + 0.25 * df['Intraday_Volatility'] + 0.25 * df['Adjusted_Momentum'])
    df['Final_Factor'] = (df['Final_Factor'] * 0.7) + (df['Weighted_Intraday_Return_and_Reversal_Signal'] * 0.3)
    
    # IntegratedMomentumFactor
    df['5_Day_Average_High_Low_Spread'] = df['high'] - df['low']
    df['5_Day_Average_High_Low_Spread'] = df['5_Day_Average_High_Low_Spread'].rolling(window=5).mean()
    df['5_Day_Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['HighLowSpreadMomentum'] = df['5_Day_Average_High_Low_Spread'] * df['5_Day_Momentum']
    df['IntegratedMomentumFactor'] = df['Final_Factor'] + df['HighLowSpreadMomentum']
    
    # VolumePriceTrendFactor
    df['Volume_Adjusted_Return'] = (df['close'] - df['close'].shift(1)) / df['volume']
    df['VolumePriceTrendFactor'] = df['Volume_Adjusted_Return'] * df['PriceAdjustedRange']
    
    # VolumeTrendAdjustedFactor
    df['5_Day_Sum_of_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Ratio'] = df['volume'] / df['5_Day_Sum_of_Volume']
    df['VolumeTrendAdjustedFactor'] = df['Final_Factor'] * df['Volume_Ratio']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.2 * df['PriceAdjustedRange'] + 
                          0.3 * df['IntegratedMomentumFactor'] + 
                          0.2 * df['VolumePriceTrendFactor'] + 
                          0.3 * df['VolumeTrendAdjustedFactor'])
    
    return df['Final_Factor']
```
