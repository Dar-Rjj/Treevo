```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on close price
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the rolling mean of the daily return for a 5-day window, excluding the current day
    df['rolling_mean_5d'] = df['daily_return'].shift(1).rolling(window=5).mean()
    
    # Calculate the ratio of the current day's high to the previous day's low
    df['high_to_low_ratio'] = df['high'] / df['low'].shift(1)
    
    # Calculate the ratio of the current day's amount to the previous day's amount
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Alpha factor: (Rolling mean of daily returns) * (High to Low Ratio) * (Amount Ratio)
    df['alpha_factor'] = df['rolling_mean_5d'] * df['high_to_low_ratio'] * df['amount_ratio']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
