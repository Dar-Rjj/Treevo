```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']

    # Adjust by Volume Change and Average Volume
    lookback_period = 20
    df['volume_ma'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    adjusted_price_range = (intraday_price_movement * df['volume_change']) / df['volume_ma']

    # Apply Enhanced Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    smoothing_parameter = 1.5
    sigmoid_adjustment = 1 / (1 + np.exp(-smoothing_parameter * (price_level_ratio - 1)))
    adjusted_price_range *= sigmoid_adjustment

    # Incorporate Close Price Momentum
    momentum_lookback = 10
    momentum = (df['close'] - df['close'].shift(momentum_lookback)) / df['close'].shift(momentum_lookback)

    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()

    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()

    # Combine 5-Day Average High-Low Spread and Momentum
    high_low_spread_momentum = avg_high_low_spread * momentum

    # Integrate Intraday Volatility
    daily_range = high_low_spread
    avg_daily_range = daily_range.rolling(window=lookback_period).mean()

    # Incorporate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return

    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    close_5_days_ago = df['close'].shift(5)
    five_day_momentum = (df['close'] - close_5_days_ago) / close_5_days_ago
    high_low_spread_momentum = avg_high_low_spread * five_day_momentum

    # Combine Intraday and High-Low Spread Momentum
    integrated_momentum_factor = adjusted_price_range + high_low_spread_momentum

    # Calculate Volume Weighted Average Price
    vwap = (df[['open', 'high', 'low', 'close']].sum(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()

    # Combine Factors with Volume-Adjusted Price Trend
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_factor = (close_to_close_return * intraday_price_movement) / vwap

    # Filter for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].rolling(window=5).mean()

    # Introduce Volume Trend
    volume_sum_5_days = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / volume_sum_5_days
    volume_trend_adjusted_factor = combined_factor * volume_ratio

    # Final Factor Calculation
    final_factor = (
        0.2 * adjusted_price_range +
        0.3 * momentum +
        0.3 * integrated_momentum_factor +
        0.2 * volume_trend_adjusted_factor
    )

    return final_factor
```
