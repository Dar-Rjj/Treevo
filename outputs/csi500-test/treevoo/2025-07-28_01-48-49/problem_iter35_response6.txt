```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_intraday_price_movement = intraday_price_movement * (df['volume'] * close_difference)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    close_to_open_reversal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_reversal
    
    # Construct Weighted Intraday Return and Reversal Signal
    weighted_intraday_return = (df['volume'] * intraday_price_movement) + (df['amount'] * close_to_open_reversal)
    
    # Combine Enhanced Factors
    combined_factor = weighted_intraday_price_movement * volume_weighted_return * intraday_volatility
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=40).mean()
    adjusted_combined_factor = combined_factor / average_volume
    
    # Incorporate Close Price Momentum
    close_momentum = df['close'].rolling(window=30).mean()
    adjusted_momentum = close_momentum * intraday_volatility
    
    # Adjusted Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    adjusted_price_momentum = ema_10 - close_momentum
    
    # Final Factor Calculation
    final_factor = (0.55 * adjusted_combined_factor) + (0.225 * intraday_volatility) + (0.225 * adjusted_price_momentum)
    final_alpha_factor = (final_factor * 0.75) + (weighted_intraday_return * 0.25)
    
    return final_alpha_factor
```
