```python
import pandas as pd
import numpy as np
from pandas.core.window.rolling import Rolling

def heuristics_v2(df):
    # Calculate High-Low Return
    high_low_return = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Reversal Signals
    combined_signal = (intraday_return * df['amount']) + (reversal_signal * df['volume'])
    
    # Calculate High-Low Momentum
    high_low_momentum = high_low_return.rolling(window=10).sum()
    
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply 5-Day Average High-Low Spread by Momentum
    weighted_spread = avg_high_low_spread * momentum
    
    # Calculate Volume Weight
    volume_weight = df['volume'].rolling(window=10).sum()
    
    # Final Alpha Construction
    alpha = (combined_signal) + (high_low_momentum * volume_weight) + (weighted_spread) + (ema_high_low_spread)
    
    return alpha
```
