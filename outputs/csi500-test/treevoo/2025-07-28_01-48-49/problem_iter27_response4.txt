import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    daily_price_return = (df['close'] - df['open']) / df['open']
    
    # Evaluate Price Volatility
    price_volatility = (df['high'] - df['low']) / df['open']
    
    # Determine Price Trend
    strong_upward_trend = ((df['close'] > df['open']) & ((df['high'] - df['close']) < (df['open'] - df['low']))).astype(int)
    strong_downward_trend = ((df['close'] < df['open']) & ((df['high'] - df['open']) > (df['close'] - df['low']))).astype(int) * -1
    price_trend = strong_upward_trend + strong_downward_trend  # Combine upward and downward trends
    
    # Investigate Volume Dynamics
    daily_volume_change = df['volume'].diff()
    
    # Detect Volume Surges
    median_volume_past_5_days = df['volume'].rolling(window=5).median().shift(1)  # To avoid look-ahead bias
    volume_surge = (df['volume'] > 2 * median_volume_past_5_days).astype(int)
    
    # Integrate Price and Volume Indicators
    combine_price_return_volume_surge = daily_price_return * volume_surge
    combine_price_volatility_volume_surge = price_volatility * volume_surge
    
    # Assess Trading Amounts
    daily_trading_amount = df['close'] * df['volume']
    
    # Identify Significant Trading Amount
    median_trading_amount_past_5_days = daily_trading_amount.rolling(window=5).median().shift(1)  # To avoid look-ahead bias
    significant_trading_amount = (daily_trading_amount > 2 * median_trading_amount_past_5_days).astype(int)
    
    # Aggregate and Refine Alpha Factors
    alpha_factor = (price_trend + combine_price_return_volume_surge + 
                    combine_price_volatility_volume_surge + significant_trading_amount)
    
    return alpha_factor
