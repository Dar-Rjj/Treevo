```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Price-Based Indicators
    # Simple Moving Average (SMA) of Close Prices over 20 days
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) of Close Prices over 12 days
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    
    # Momentum Indicators
    # Relative Strength Index (RSI) using 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + RS))
    
    # Rate of Change (ROC) using 9-day period
    df['ROC_9'] = df['close'].pct_change(periods=9) * 100
    
    # Volume Weighted Indicators
    # Volume Weighted Average Price (VWAP) calculated on a daily basis
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Volume Oscillator, difference between 30-day and 10-day SMA of Volume
    df['SMA_Volume_30'] = df['volume'].rolling(window=30).mean()
    df['SMA_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Oscillator'] = df['SMA_Volume_30'] - df['SMA_Volume_10']
    
    # Volatility Indicators
    # Historical Volatility (HV) computed with close prices over the past 20 days
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['HV_20'] = df['Log_Returns'].rolling(window=20).std() * np.sqrt(252)
    
    # Bollinger Bands (BB) around 20-day SMA, with 2 standard deviations
    df['BB_Upper'] = df['SMA_20'] + 2 * df['close'].rolling(window=20).std()
    df['BB_Lower'] = df['SMA_20'] - 2 * df['close'].rolling(window=20).std()
    
    # Trend Following Indicators
    # Moving Average Convergence Divergence (MACD) using 12, 26, and 9 day periods
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD_Line'] = df['EMA_12'] - df['EMA_26']
    df['Signal_Line'] = df['MACD_Line'].ewm(span=9, adjust=False).mean()
    df['MACD_Histogram'] = df['MACD_Line'] - df['Signal_Line']
    
    # Custom Combined Indicators
    # Combination of EMA and RSI
    df['Undervaluation_Signal'] = ((df['EMA_12'] > df['EMA_26']) & (df['RSI_14'] < 30)).astype(int)
    
    # Integration of VWAP and ROC
    df['Bullish_Momentum_Signal'] = ((df['close'] > df['VWAP']) & (df['ROC_9'] > 0)).astype(int)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Undervaluation_Signal'] + df['Bullish_Momentum_Signal']
    
    return df['Alpha_Factor']
```
