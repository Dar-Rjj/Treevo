def heuristics_v2(data):
    # Calculate the weighted average of open and close prices using volume as weights
    data['weighted_avg'] = (data['open'] * data['volume'] + data['close'] * data['volume']) / (2 * data['volume'])
    
    # Calculate the difference between high and low prices
    data['high_low_diff'] = data['high'] - data['low']
    
    # Calculate the cumulative sum of the amount over the last 5 days
    data['amount_5d_sum'] = data['amount'].rolling(window=5).sum()
    
    # Generate the alpha factor by combining the above calculations
    data['alpha_factor'] = (data['weighted_avg'] * data['high_low_diff']) / data['amount_5d_sum']
    
    return data['alpha_factor']
