```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday and High-Low Movement
    df['IntradayRange'] = df['high'] - df['low']
    df['OpeningPriceDifference'] = df['open'] - df['open'].shift(1)
    df['AdjustedHighToLowMovement'] = df['IntradayRange'] / df['OpeningPriceDifference']

    # Integrate Volume-Weighted Momentum
    df['CloseDifference'] = df['close'] - df['close'].shift(1)
    df['VolumeWeightedMomentum'] = df['Volume'] * df['CloseDifference'] * df['AdjustedHighToLowMovement']

    # Combine Intraday and Reversal Signals
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    df['ReversalSignal'] = (df['close'] - df['open']) / df['open']
    df['CombinedSignal'] = (df['IntradayReturn'] * df['amount']) + (df['ReversalSignal'] * df['volume'])

    # Calculate High-Low Momentum
    df['HighLowMomentum'] = df['IntradayReturn'].rolling(window=10).sum()

    # Incorporate Intraday Volatility
    df['DailyRange'] = df['high'] - df['low']
    df['5DayAverageDailyRange'] = df['DailyRange'].rolling(window=5).mean()

    # Combine with 5-Day Average High-Low Spread
    df['5DayAverageHighLowSpread'] = df['IntradayRange'].rolling(window=5).sum() / 5

    # Finalize Combined Momentum Factors
    df['5DayMomentum'] = (df['open'] - df['close'].shift(5)) / df['close'].shift(5)
    df['HighLowSpreadMomentum'] = df['5DayAverageHighLowSpread'] * df['5DayMomentum']
    df['IntegratedMomentumFactor'] = df['IntradayRange'].clip(lower=0) + df['VolumeWeightedMomentum'].clip(lower=0)

    # Incorporate Volume-Weighted Return
    df['VolumeWeightedReturn'] = df['volume'] * ((df['close'] - df['open']) / df['open'])

    # Calculate Close to Close Return
    df['CloseToCloseReturn'] = df['close'] / df['close'].shift(1)

    # Combine Factors
    df['VWAP'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['CombinedFactors'] = (df['CloseToCloseReturn'] * df['IntradayRange']) / df['VWAP']

    # Filter for Recent Trend
    df['5DayMovingAverageClose'] = df['close'].rolling(window=5).mean()
    df['RecentTrend'] = df['close'] - df['5DayMovingAverageClose']

    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['IntegratedMomentumFactor'] +
        df['5DayAverageDailyRange'] +
        df['VolumeWeightedReturn'] +
        df['CombinedFactors']
    ) * df['RecentTrend'].clip(lower=0)

    return df['FinalAlphaFactor']
```
