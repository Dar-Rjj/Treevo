```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure using volume and price range
    buying_pressure = df['volume'] * (avg_price - df['low']) / price_range
    
    # Calculate the selling pressure using volume and price range
    selling_pressure = df['volume'] * (df['high'] - avg_price) / price_range
    
    # The alpha factor is the ratio of buying pressure to selling pressure
    alpha_factor = buying_pressure / selling_pressure
    
    return alpha_factor
```
