import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_range = intraday_range * volume_change
    
    # Incorporate Close Price Momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Final Factor Calculation
    factor = 0.6 * adjusted_range + 0.4 * momentum
    
    return factor
