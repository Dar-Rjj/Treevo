```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * (df['volume'] * df['Close_Difference'])
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = df['Daily_Range'].rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Close_to_Open_Reversal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Close_to_Open_Reversal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return_and_Reversal'] = (df['volume'] * df['Intraday_Price_Movement']) + (df['amount'] * df['Close_to_Open_Reversal'])
    
    # Combine Enhanced Factors
    df['Combined_Factors'] = df['Weighted_Intraday_Price_Movement'] * df['Volume_Weighted_Return']
    
    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=40).mean()
    df['Adjusted_Combined_Factors'] = df['Combined_Factors'] / df['Average_Volume']
    
    # Incorporate Close Price Momentum
    df['Momentum'] = df['close'].rolling(window=30).mean()
    
    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['Momentum'] * df['Intraday_Volatility']
    
    # Final Factor Calculation
    df['Final_Factor'] = (0.5 * df['Adjusted_Combined_Factors']) + (0.25 * df['Intraday_Volatility']) + (0.25 * df['Adjusted_Momentum'])
    df['Integrated_Factor'] = (df['Final_Factor'] * 0.7) + (df['Weighted_Intraday_Return_and_Reversal'] * 0.3)
    
    # Filter for Recent Price Trend
    df['5_Day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['Recent_Price_Trend'] = df['close'] - df['5_Day_MA_Close']
    
    # Introduce Volume Trend
    df['5_Day_Sum_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Ratio'] = df['volume'] / df['5_Day_Sum_Volume']
    df['VolumeTrendAdjustedFactor'] = df['Integrated_Factor'] * df['Volume_Ratio']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Integrated_Factor'] * 0.6) + (df['VolumeTrendAdjustedFactor'] * 0.4)
    
    return df['Final_Alpha_Factor']
```
