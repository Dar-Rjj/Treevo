def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, amount, volume.
        
    Returns:
        pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the ratio of the closing price to the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the ratio of the amount traded to the volume
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Combine the two ratios to create the alpha factor
    alpha_factor = close_to_avg_ratio * amount_to_volume_ratio
    
    return alpha_factor
