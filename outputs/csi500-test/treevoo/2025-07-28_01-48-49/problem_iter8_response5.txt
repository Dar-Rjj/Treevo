```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_range = intraday_range / avg_volume
    
    # Incorporate Close Price Momentum
    close_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread_5d = high_low_spread.rolling(window=5).mean()
    
    # Calculate 10-Day EMA of High-Low Spread
    ema_high_low_spread_10d = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Multiply 5-Day Average High-Low Spread by Momentum
    weighted_avg_high_low_spread = avg_high_low_spread_5d * momentum
    
    # Add 10-Day EMA of High-Low Spread to 20-Day Price Return
    price_return_20d = (df['close'] / df['close'].shift(20) - 1)
    combined_factor = ema_high_low_spread_10d + price_return_20d
    
    # Calculate Open Adjustment Factor
    open_adj_factor = df['open'] - df['open'].shift(1)
    
    # Adjust High-to-Low Movement by Open Adjustment
    adj_high_low_movement = (df['high'] - df['low']) / open_adj_factor
    
    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_high_low_movement = adj_high_low_movement * (df['volume'] * close_diff)
    
    # Integrate Intraday Volatility
    daily_ranges = df['high'] - df['low']
    avg_daily_ranges = daily_ranges.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Return
    volume_weighted_return = df['volume'] * close_to_open_return
    
    # Final Factor Calculation
    final_factor = (
        0.25 * adjusted_range +
        0.25 * close_momentum +
        0.25 * weighted_high_low_movement +
        0.15 * avg_daily_ranges +
        0.1 * volume_weighted_return
    ) + combined_factor
    
    return final_factor
```
