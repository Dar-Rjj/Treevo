```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term price momentum (5-day percentage change in close price)
    df['MOM_5'] = df['close'].pct_change(5)
    
    # Calculate long-term price momentum (20-day percentage change in close price)
    df['MOM_20'] = df['close'].pct_change(20)
    
    # Average high-low difference over the past 10 days
    df['HIGH_LOW_DIFF'] = df['high'] - df['low']
    df['AVG_HIGH_LOW_10'] = df['HIGH_LOW_DIFF'].rolling(window=10).mean()
    
    # Daily range divided by close price
    df['DAILY_RANGE_OVER_CLOSE'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the Average True Range (ATR) over the last 14 days
    df['TR'] = df[['high', 'low']].diff(axis=1).iloc[:, 0]
    df['TR'] = df[['high', 'low', 'close']].shift().apply(lambda x: max(x[0] - x[1], x[0] - x[2], x[1] - x[2]), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Percentage change in volume from one day to the next
    df['VOLUME_CHANGE'] = df['volume'].pct_change()
    
    # Volume to average volume over the past 20 days
    df['AVERAGE_VOLUME_20'] = df['volume'].rolling(window=20).mean()
    df['VOLUME_TO_AVERAGE_VOLUME_20'] = df['volume'] / df['AVERAGE_VOLUME_20']
    
    # Sum of positive and negative changes in close price over a 10-day period
    df['CLOSE_CHANGE'] = df['close'].diff()
    df['CUMULATIVE_CHANGE_SUM_10'] = df['CLOSE_CHANGE'].rolling(window=10).sum()
    
    # Amount to average amount over the past 10 days
    df['AVERAGE_AMOUNT_10'] = df['amount'].rolling(window=10).mean()
    df['AMOUNT_TO_AVERAGE_AMOUNT_10'] = df['amount'] / df['AVERAGE_AMOUNT_10']
    
    # Average gap between today's open and yesterday's close over the past 5 days
    df['GAP'] = df['open'] - df['close'].shift()
    df['AVERAGE_GAP_5'] = df['GAP'].rolling(window=5).mean()
    
    # Return the factor values as a pandas Series
    return df['MOM_5'].fillna(0) + df['MOM_20'].fillna(0) + df['AVG_HIGH_LOW_10'].fillna(0) + \
           df['DAILY_RANGE_OVER_CLOSE'].fillna(0) + df['ATR_14'].fillna(0) + df['VOLUME_CHANGE'].fillna(0) + \
           df['VOLUME_TO_AVERAGE_VOLUME_20'].fillna(0) + df['CUMULATIVE_CHANGE_SUM_10'].fillna(0) + \
           df['AMOUNT_TO_AVERAGE_AMOUNT_10'].fillna(0) + df['AVERAGE_GAP_5'].fillna(0)
```
