import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_range[intraday_range < 0] = 0  # Ensure non-negative

    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    current_volume = df['volume']
    volume_change = current_volume - prev_volume
    volume_change[volume_change < 0] = 0  # Ensure non-negative

    # Integrate Intraday Price Movement and Volume
    intraday_momentum_score = intraday_range * volume_change

    # Calculate 5-Day Average High-Low Spread
    high_low_spread_5d = (df['high'] - df['low']).rolling(window=5).mean()

    # Finalize Alpha Factor Components
    close_today = df['close']
    open_today = df['open']
    intraday_return = (close_today - open_today) / open_today
    close_5d_ago = df['close'].shift(5)
    five_day_momentum = (close_today - close_5d_ago) / close_5d_ago
    high_low_spread_momentum = high_low_spread_5d * five_day_momentum
    combined_intraday_momentum = intraday_momentum_score + intraday_return
    integrated_momentum_factor = combined_intraday_momentum * high_low_spread_momentum

    # Calculate Intraday Volatility
    intraday_volatility = intraday_range / df['volume']

    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = integrated_momentum_factor * intraday_volatility

    # Calculate Dynamic Adjustment Factor
    dynamic_adjustment_factor = df['volume'].ewm(span=5, adjust=False).mean()

    # Apply Dynamic Adjustment
    dynamically_adjusted_momentum = adjusted_momentum * dynamic_adjustment_factor

    # Final Alpha Factor
    final_alpha_factor = dynamically_adjusted_momentum - dynamically_adjusted_momentum.ewm(span=5, adjust=False).mean()

    return final_alpha_factor
