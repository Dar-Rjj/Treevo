```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['5Day_Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=5).mean()

    # Calculate 5-Day Momentum
    df['5Day_Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Multiply High-Low Spread and Momentum
    df['HL_Momentum_Product'] = df['5Day_Avg_High_Low_Spread'] * df['5Day_Momentum']

    # Calculate Enhanced Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))

    # Weight by Volume and Close Difference
    df['Close_Diff'] = df['close'] - df['close'].shift(1)
    df['Weighted_Intraday_Price_Movement'] = df['Intraday_Price_Movement'] * (df['volume'] * df['Close_Diff'])

    # Generate IntradayMomentumScore
    df['IntradayMomentumScore'] = df['Weighted_Intraday_Price_Movement'].abs()

    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['20Day_MA_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()

    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']

    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']

    # Construct Weighted Intraday Return and Reversal Signal
    df['Intraday_Return'] = df['Amount'] * (df['close'] - df['open'])
    df['Weighted_Intraday_Return_Reversal'] = (df['Amount'] * df['Intraday_Return']) + (df['volume'] * df['Reversal_Signal'])

    # Combine Enhanced Factors
    df['Combined_Factors'] = df['IntradayMomentumScore'] * df['20Day_MA_Daily_Range'] * df['Volume_Weighted_Return']

    # Incorporate Close Price Momentum
    df['30Day_MA_Close'] = df['close'].rolling(window=30).mean()

    # Adjust Momentum by Intraday Volatility
    df['Adjusted_Momentum'] = df['5Day_Momentum'] * df['20Day_MA_Daily_Range']

    # Final Factor Calculation
    df['Final_Factor'] = (0.5 * df['IntradayMomentumScore']) + (0.25 * df['20Day_MA_Daily_Range']) + (0.25 * df['Adjusted_Momentum'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Final_Factor'] * 0.7) + (df['Weighted_Intraday_Return_Reversal'] * 0.3)

    # Apply Price Level Adjustment
    df['Price_Level_Ratio'] = df['close'] / df['open']
    df['Final_Alpha_Factor_Adjusted'] = df['Final_Alpha_Factor'] * df['Price_Level_Ratio']

    # Multiply Final Alpha Factor by Volume Ratio
    df['40Day_MA_Volume'] = df['volume'].rolling(window=40).mean()
    df['VolumeTrendAdjustedFactor'] = df['Final_Alpha_Factor_Adjusted'] / df['40Day_MA_Volume']

    return df['VolumeTrendAdjustedFactor']
```
