```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Calculate 5-Day Average High-Low Spread
    avg_high_low_spread = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Generate Momentum Score
    volume_change = df['volume'].pct_change()
    intraday_momentum_score = intraday_range * volume_change
    
    # Finalize Alpha Factor for Intraday and High-Low Spread Momentum
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    high_low_spread_momentum = avg_high_low_spread * five_day_momentum
    integrated_momentum_factor = intraday_momentum_score + high_low_spread_momentum
    
    # Calculate Volume Weighted Average Price
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    vwap = vwap.cumsum() / df['volume'].cumsum()
    
    # Combine Factors with Volume-Adjusted Price Trend
    combined_factor = (close_to_close_return * intraday_range) / vwap
    
    # Filter for Recent Price Trend
    five_day_ma = df['close'].rolling(window=5).mean()
    recent_price_trend = df['close'] - five_day_ma
    
    # Introduce Volume Trend
    five_day_volume_sum = df['volume'].rolling(window=5).sum()
    volume_ratio = df['volume'] / five_day_volume_sum
    volume_trend_adjusted_factor = combined_factor * volume_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback = 10
    close_price_momentum = (df['close'] - df['close'].shift(momentum_lookback)) / df['close'].shift(momentum_lookback)
    
    # Final Alpha Factor
    final_alpha_factor = integrated_momentum_factor + volume_trend_adjusted_factor + close_price_momentum
    
    return final_alpha_factor
```
