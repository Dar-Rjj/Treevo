import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['daily_range'] = df['high'] - df['low']
    intraday_volatility = df['daily_range'].rolling(window=5).mean()

    # Calculate Dynamic Price Difference
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Aggregate Price Difference with Volume
    df['weighted_return'] = df['close_to_close_return'] * df['volume']
    sum_weighted_differences = df['weighted_return'].rolling(window=5).sum()
    
    # Enhance with Smoothed Moving Average Filter
    sma_weighted_sums = sum_weighted_differences.rolling(window=10).mean()
    alpha_factor = sma_weighted_sums / intraday_volatility
    
    # Combine Intraday Volatility and Volume-Weighted Return
    combined_factor = intraday_volatility * sum_weighted_differences
    
    # Introduce Volume and Price Trend Components
    df['5_day_moving_avg_close'] = df['close'].rolling(window=5).mean()
    df['price_trend'] = df['close'] - df['5_day_moving_avg_close']
    df['5_day_sum_volume'] = df['volume'].rolling(window=5).sum()
    df['volume_ratio'] = df['volume'] / df['5_day_sum_volume']
    final_factor = combined_factor * df['volume_ratio'] + df['price_trend']

    return final_factor
