import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    adjusted_intraday_price_movement = (df['high'] - df['low']) / (df['open'] - df['open'].shift(1))
    
    # Weight by Volume and Close Difference
    close_difference = df['close'] - df['close'].shift(1)
    weighted_intraday_price_movement = adjusted_intraday_price_movement * (df['volume'] * close_difference)
    
    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal
    
    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_signal_weight = df['volume']
    weighted_intraday_return_and_reversal = (intraday_return_weight * weighted_intraday_price_movement) + (reversal_signal_weight * volume_weighted_return)
    
    # Combine Enhanced Factors
    combined_factors = (weighted_intraday_price_movement + intraday_volatility + volume_weighted_return) * 0.6 + (weighted_intraday_return_and_reversal * 0.4)
    
    # Incorporate Close Price Momentum
    momentum = df['close'].rolling(window=10).mean()
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum * intraday_volatility
    
    # Final Factor Calculation
    final_factor = (0.4 * weighted_intraday_price_movement + 0.3 * intraday_volatility + 0.3 * adjusted_momentum) * 0.6 + (weighted_intraday_return_and_reversal * 0.4)
    
    return final_factor
