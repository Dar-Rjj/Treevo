```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume Change and Price Level
    lookback_period = 20
    ma_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_range = intraday_range * volume_change
    
    price_level_ratio = df['close'] / df['open']
    adjusted_range = adjusted_range * price_level_ratio
    
    # Incorporate Close Price Momentum
    momentum_lookback_period = 10
    momentum = df['close'] / df['close'].shift(momentum_lookback_period) - 1
    
    # Final Factor Calculation
    range_weight = 0.6
    momentum_weight = 0.4
    factor = range_weight * adjusted_range + momentum_weight * momentum
    
    return factor
```
