```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Compare Short-term and Long-term Momentum
    df['Momentum_Difference'] = df['SMA_5'] - df['SMA_20']
    df['Momentum_Direction'] = (df['SMA_5'] > df['SMA_20']).astype(int) * 2 - 1
    
    # Evaluate Recent Price Changes
    df['Price_Change'] = df['close'] - df['open']
    df['Percentage_Change'] = df['Price_Change'] / df['open']
    
    # Analyze Volume Patterns
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Volume_Trend'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Relative_Range'] = df['Daily_Range'] / df['close']
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=20).mean()
    df['Volatility_Spike'] = (df['Daily_Range'] > 1.5 * df['Average_Daily_Range']).astype(int)
    
    # Integrate Intraday Dynamics
    df['Adjusted_Intraday_Price_Movement'] = df['Daily_Range'] / (df['open'] - df['open'].shift(1))
    df['Weighted_Intraday_Price_Movement'] = df['Adjusted_Intraday_Price_Movement'] * (df['volume'] * df['Price_Change'])
    
    # Combine Volume and Price Data
    df['Volume_Adjusted_Price_Change'] = df['Price_Change'] * df['Volume_Ratio']
    df['Volume_Adjusted_Momentum'] = df['Momentum_Difference'] * df['Volume_Ratio']
    
    # Incorporate High and Low Prices into Momentum and Volume
    df['High_Low_Momentum'] = df['Daily_Range'] * df['Momentum_Direction']
    df['Volume_Adjusted_High_Low_Momentum'] = df['High_Low_Momentum'] * df['Volume_Ratio']
    
    # Generate a Compound Alpha Factor
    df['Momentum_Volatility_Score'] = df['Momentum_Difference'] * (1 + df['Volatility_Spike'])
    df['Comprehensive_Alpha_Factor'] = (df['Momentum_Direction'] * df['Volume_Adjusted_Price_Change']) + (df['Relative_Range'] * df['Volume_Ratio'])
    df['Multi_Indicator_Alpha'] = (df['Momentum_Difference'] * df['Volume_Ratio'] * df['Volatility_Spike']) + (df['Price_Change'] * df['Volume_Trend'])
    df['Compound_Alpha'] = (df['Momentum_Volatility_Score'] * df['Volume_Adjusted_High_Low_Momentum']) + (df['Comprehensive_Alpha_Factor'] * df['Multi_Indicator_Alpha'])
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    df['Weighted_Intraday_Return_and_Reversal'] = (df['amount'] * df['Adjusted_Intraday_Price_Movement']) + (df['volume'] * df['Reversal_Signal'])
    
    # Refine Alpha Factors with Enhanced Combination
    df['Combined_Enhanced_Factors'] = (df['Weighted_Intraday_Price_Movement'] + df['Daily_Range']) * (df['volume'] * df['Price_Change'])
    df['Final_Factor'] = (df['Combined_Enhanced_Factors'] * 0.6) + (df['Weighted_Intraday_Return_and_Reversal'] * 0.4)
    
    return df['Final_Factor']
```
