```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Moving Average of Volume
    lookback_period = 20
    df['volume_ma'] = df['volume'].rolling(window=lookback_period).mean()

    # Calculate Current Day Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Divide Price Range by Volume Change
    df['adjusted_range'] = df['price_range'] / df['volume_change']

    # Calculate Close Price Momentum
    momentum_lookback = 10
    df['momentum'] = df['close'] / df['close'].shift(momentum_lookback)

    # Combine Adjusted Range and Momentum
    adjusted_range_weight = 0.6
    momentum_weight = 0.4
    df['vwmri'] = (df['adjusted_range'] * adjusted_range_weight) + (df['momentum'] * momentum_weight)

    return df['vwmri']
```
