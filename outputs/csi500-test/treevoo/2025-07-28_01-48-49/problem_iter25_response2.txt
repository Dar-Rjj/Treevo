import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=5).mean()

    # Calculate 5-Day Momentum
    momentum = (df['close'] / df['close'].shift(5) - 1)

    # Multiply High-Low Spread and Momentum
    high_low_momentum = avg_high_low_spread * momentum

    # Calculate Enhanced Intraday Price Movement
    open_diff = df['open'] - df['open'].shift(1)
    enhanced_intraday_movement = (df['high'] - df['low']) / open_diff

    # Weight by Volume and Close Difference
    close_diff = df['close'] - df['close'].shift(1)
    weighted_movement = enhanced_intraday_movement * df['volume'] * close_diff

    # Generate IntradayMomentumScore
    intraday_momentum_score = weighted_movement.abs().clip(lower=0)

    # Integrate Intraday Volatility
    daily_range = df['high'] - df['low']
    intraday_volatility = daily_range.rolling(window=20).mean()

    # Incorporate Close-to-Open Reversal Signal
    reversal_signal = (df['close'] - df['open']) / df['open']

    # Calculate Volume-Weighted Return
    volume_weighted_return = df['volume'] * reversal_signal

    # Construct Weighted Intraday Return and Reversal Signal
    intraday_return_weight = df['amount']
    reversal_weight = df['volume']
    weighted_intraday_return = (intraday_return_weight * enhanced_intraday_movement) + (reversal_weight * reversal_signal)

    # Combine Enhanced Factors
    combined_factors = (0.3 * intraday_momentum_score) + (0.3 * intraday_volatility) + (0.4 * momentum * intraday_volatility)

    # Incorporate Close Price Momentum
    close_momentum = df['close'] / df['close'].rolling(window=10).mean() - 1

    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = close_momentum * intraday_volatility

    # Final Factor Calculation
    final_factor = (0.5 * combined_factors) + (0.5 * weighted_intraday_return)

    # Apply Price Level Adjustment
    price_level_ratio = df['close'] / df['open']
    final_alpha_factor = final_factor * price_level_ratio

    # Final Factor Computation
    volume_trend_adjusted_factor = final_alpha_factor * (df['volume'] / df['volume'].shift(1))

    return volume_trend_adjusted_factor
