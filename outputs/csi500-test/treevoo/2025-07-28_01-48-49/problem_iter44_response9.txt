import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Price Movement
    df['Adjusted_Intraday_Price_Movement'] = (df['high'] - df['low']) / (df['open'] - df['close'].shift(1))
    
    # Weight by Volume and Close Difference
    df['Close_Difference'] = df['close'] - df['close'].shift(1)
    df['Weighted_Adjusted_Intraday_Price_Movement'] = df['Adjusted_Intraday_Price_Movement'] * (df['volume'] * df['Close_Difference'])
    
    # Integrate Intraday Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = df['Daily_Range'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Close-to-Open Reversal Signal
    df['Reversal_Signal'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['volume'] * df['Reversal_Signal']
    
    # Construct Weighted Intraday Return and Reversal Signal
    df['Weighted_Intraday_Return'] = (df['amount'] * df['Adjusted_Intraday_Price_Movement']) + (df['volume'] * df['Reversal_Signal'])
    
    # Combine Enhanced Factors
    df['Combined_Factor'] = df['Weighted_Adjusted_Intraday_Price_Movement'] + df['Intraday_Volatility'] + df['Volume_Weighted_Return']
    
    # Adjust by Relative Volume
    df['Volume_EMA'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Relative_Volume'] = df['volume'] / df['Volume_EMA']
    df['Combined_Factor_Adjusted'] = df['Combined_Factor'] / df['Relative_Volume']
    
    # Incorporate Close Price Momentum
    df['30D_MA_Close'] = df['close'].rolling(window=30).mean()
    df['10D_EMA_Close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Momentum'] = df['10D_EMA_Close'] - df['30D_MA_Close']
    df['Adjusted_Momentum'] = df['Momentum'] * df['Intraday_Volatility']
    
    # Construct Momentum Factor
    df['5D_Pct_Change'] = df['close'].pct_change(5)
    df['20D_Pct_Change'] = df['close'].pct_change(20)
    df['60D_Pct_Change'] = df['close'].pct_change(60)
    df['Momentum_Factor'] = (df['5D_Pct_Change'] + df['20D_Pct_Change'] + df['60D_Pct_Change']) / 3
    
    # Design Volatility Measures
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['5D_Volatility'] = df['Log_Returns'].rolling(window=5).std()
    df['20D_Volatility'] = df['Log_Returns'].rolling(window=20).std()
    df['60D_Volatility'] = df['Log_Returns'].rolling(window=60).std()
    df['Volatility_Measures'] = (df['5D_Volatility'] + df['20D_Volatility'] + df['60D_Volatility']) / 3
    
    # Incorporate Volume-Based Indicators
    df['Average_Volume_20D'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_20D']
    
    # Explore Price Action Patterns
    def pattern_score(row):
        if row['open'] > row['close']:
            return 1  # Bearish
        elif row['open'] < row['close']:
            return -1  # Bullish
        else:
            return 0  # Doji
    df['Pattern_Score'] = df.apply(pattern_score, axis=1)
    
    # Develop Trend-Following Signals
    df['20D_SMA'] = df['close'].rolling(window=20).mean()
    df['Trend_Signal'] = np.where(df['close'] > df['20D_SMA'], 1, -1)
