```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA)
    df['SMA'] = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['SMA']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['AvgVolume'] = df['volume'].rolling(window=10).mean()
    df['RelativeVolume'] = df['volume'] / df['AvgVolume']
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']
    
    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    df['UpwardThrust'] = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else 0, axis=1)
    df['DownwardThrust'] = df.apply(lambda x: x['volume'] if x['close'] <= x['open'] else 0, axis=1)
    df['VolumeThrust'] = df['UpwardThrust'] - df['DownwardThrust']
    
    # Calculate Intraday Reversal
    df['IntradayReversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Reversal and Volume Change
    df['ReversalVolumeAdjustment'] = df['IntradayReversal'] * df['VolumeChange']
    
    # Calculate High-Low Price Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['VolumeAdjustedHighLowRange'] = df['HighLowRange'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['RollingAvgVolumeAdjustedHighLowRange'] = df['VolumeAdjustedHighLowRange'].rolling(window=20).mean()
    
    # Calculate Daily Log Return
    df['DailyLogReturn'] = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return
    df['RollingAvgDailyLogReturn'] = df['DailyLogReturn'].rolling(window=20).mean()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['PriceMomentum'] + 
        df['IntradayMomentum'] - 
        df['VolumeAdjustedSpread'] + 
        df['VolumeThrust'] - 
        df['ReversalVolumeAdjustment'] + 
        (df['RollingAvgVolumeAdjustedHighLowRange'] - df['RollingAvgDailyLogReturn'])
    )
    
    return df['FinalAlphaFactor']
```
