```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Momentum'] = (df['Log_Return'] * df['Volume_Change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['5_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (df['5_EMA'] > df['20_EMA']).astype(int) - (df['5_EMA'] < df['20_EMA']).astype(int)
    df['Integrated_Momentum'] += crossover_signal
    
    # Incorporate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Normalized_High_Low_Range'] = df['High_Low_Range'] / df['close']
    df['Integrated_Momentum'] *= df['Normalized_High_Low_Range']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['Upward_Thrust'] = 0.0
    df['Downward_Thrust'] = 0.0
    df.loc[df['close'] > df['open'], 'Upward_Thrust'] = df['volume'] * (df['close'] - df['open'])
    df.loc[df['close'] <= df['open'], 'Downward_Thrust'] = df['volume'] * (df['open'] - df['close'])
    df['Price_Range_Factor'] = (df['high'] - df['low']) / df['open']
    df['Enhanced_Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust'] + df['Price_Range_Factor']
    
    # Calculate Open-Close Momentum
    df['Open_Close_Momentum'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Price Range
    df['High_Low_Price_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Price_Range'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['Rolling_Avg_Volume_Adjusted_HL'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=20).mean()
    
    # Rolling Average of Daily Log Return
    df['Rolling_Avg_Log_Return'] = df['Log_Return'].rolling(window=20).mean()
    
    # Final Integration
    final_factor = (df['Integrated_Momentum'] + df['Intraday_Momentum'] + 
                    df['Enhanced_Volume_Thrust'] + df['Open_Close_Momentum'] + 
                    (df['Rolling_Avg_Volume_Adjusted_HL'] - df['Rolling_Avg_Log_Return']) + 
                    crossover_signal)
    
    return final_factor
```
