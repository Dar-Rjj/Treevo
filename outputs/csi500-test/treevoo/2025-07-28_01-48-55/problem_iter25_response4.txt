```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with a 10-day window
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(np.log(df['volume']))

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']

    # Enhance with Moving Average Crossover
    df['EMA_7'] = df['close'].ewm(span=7, adjust=False).mean()
    df['EMA_28'] = df['close'].ewm(span=28, adjust=False).mean()
    df['Crossover_Signal'] = 0
    df.loc[df['EMA_7'] > df['EMA_28'], 'Crossover_Signal'] = 1
    df.loc[df['EMA_7'] < df['EMA_28'], 'Crossover_Signal'] = -1
    df['Integrated_Momentum'] = df['Price_Momentum'] * df['Crossover_Signal']

    # Incorporate High-Low Range
    df['Integrated_Momentum'] *= df['High_Low_Range']

    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['open']

    # Calculate Enhanced Volume Thrust
    df['Upward_Thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['Downward_Thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['Enhanced_Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust'] + (df['high'] - df['low']) / df['close']

    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Average_Volume_20']
    df['Volume_Adjusted_Spread'] = df['Spread'] * np.sqrt(df['Relative_Volume'])

    # Final Integration
    df['Final_Integration'] = (df['Integrated_Momentum'] + df['Intraday_Momentum'] + df['Enhanced_Volume_Thrust']
                                - df['Volume_Adjusted_High_Low_Range'].rolling(window=20).mean()
                                + df['Price_Momentum'] - df['Volume_Adjusted_Spread'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Final_Integration']

    return df['Final_Alpha_Factor']
```
