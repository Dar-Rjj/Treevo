```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with a 10-day window and smoothing factor of 2/(N+1)
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) with exponentially increasing weights over a 15-day window
    weights = np.arange(1, 16)
    wma_weights = weights / weights.sum()
    df['WMA_15'] = df['close'].rolling(window=15).apply(lambda prices: (prices * wma_weights).sum(), raw=True)
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Integrated Momentum
    df['Integrated_Momentum'] = (df['close'] - df['WMA_15']) + df['Price_Momentum'].shift(1)
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_HL_Range'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    # Calculate Volume Adjusted Spread
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    df['Relative_Volume'] = df['volume'] / avg_volume_30
    df['Volume_Adjusted_Spread'] = (df['high'] - df['low']) * df['Relative_Volume']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Log_Return'] = (df['Daily_Log_Return'] * df['Volume_Change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (df['EMA_5'] > df['EMA_20']).astype(int) - (df['EMA_5'] < df['EMA_20']).astype(int)
    df['Enhanced_Crossover_Signal'] = df['Integrated_Momentum'] + crossover_signal
    
    # Calculate Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']) * (df['volume'] * (df['close'] - df['open']))
    downward_thrust = (df['close'] <= df['open']) * (df['volume'] * (df['open'] - df['close']))
    price_range_factor = (df['high'] - df['low']) / df['open']
    df['Enhanced_Volume_Thrust'] = (upward_thrust - downward_thrust) + price_range_factor
    
    # Incorporate Price-Volatility Interaction
    df['Historical_Volatility_10'] = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)
    df['Enhanced_VT_Volatility'] = df['Enhanced_Volume_Thrust'] * df['Historical_Volatility_10']
    
    # Rolling Average of Daily Log Return
    df['Rolling_Avg_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] +
        df['Integrated_Momentum'] -
        df['Volume_Adjusted_HL_Range'] -
        df['Volume_Adjusted_Spread'] +
        df['Integrated_Log_Return'] +
        df['Enhanced_Crossover_Signal'] +
        df['Enhanced_VT_Volatility'] -
        df['Rolling_Avg_Log_Return']
    )
    
    return df['Final_Alpha_Factor']
```
