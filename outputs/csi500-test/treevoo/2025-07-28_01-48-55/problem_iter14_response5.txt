```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)  # Linearly increasing weights
    df['WMA'] = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Calculate High-Low Price Range
    df['HL_Range'] = df['high'] - df['low']

    # Calculate Volume Adjusted High-Low Range
    df['Vol_Adj_HL_Range'] = df['HL_Range'] * np.log(df['volume'])

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA']

    # Calculate Relative Volume
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Avg_Volume_20']

    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Vol_Adj_Spread'] = df['Spread'] * df['Relative_Volume']

    # Calculate Momentum Factor
    window_size = 10
    df['Momentum_Factor'] = df['Vol_Adj_HL_Range'].rolling(window=window_size).sum()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Price_Momentum'] - df['Vol_Adj_Spread'] + df['Momentum_Factor']

    return df['Alpha_Factor']
```
