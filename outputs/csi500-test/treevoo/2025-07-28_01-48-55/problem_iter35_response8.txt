import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)

    # Exponential Moving Average (EMA)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()

    # High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Volume Adjusted High-Low Range
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    adjusted_high_low_range = high_low_range * np.sqrt(np.log(relative_volume))

    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + (df['close'] - df['close'].shift(1))

    # Price Momentum
    price_momentum = df['close'] - ema_10

    # Volume Adjusted Spread
    spread = high_low_range
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = spread * np.sqrt(relative_volume)

    # Enhanced Momentum Factor
    weights = np.arange(1, 6)
    adjusted_high_low_range_5 = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    enhanced_momentum_factor = (adjusted_high_low_range_5 * weights).sum(axis=1)

    # Simple Moving Averages (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()

    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Volume Shock Factor
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    volume_shock_factor = (df['volume'] - mean_volume) / std_volume

    # Volume Trend Factor
    volume_trend_factor = df['volume'].pct_change(periods=20)

    # Daily Price Range and Volatility Trends
    daily_price_range = high_low_range
    volatility_trend = daily_price_range.rolling(window=20).mean()

    # Breakout Indicator
    breakout_indicator = (df['high'] > df['high'].rolling(window=20).max()) | (df['low'] < df['low'].rolling(window=20).min())

    # Open-Close Ratio
    open_close_ratio = df['open'] / df['close']

    # Bullish/Bearish Candlestick Factor
    bullish_candle = (df['close'] > df['open']).rolling(window=10).sum()
    bearish_candle = (df['close'] < df['open']).rolling(window=10).sum()
    candlestick_factor = bullish_candle - bearish_candle

    # Liquidity Measure
    liquidity_measure = df['amount'].rolling(window=20).mean()

    # Amount-Weighted Price Movement
    price_change = df['close'] - df['close'].shift(1)
    amount_weighted_price_movement = (price_change * df['amount']).sum() / df['amount'].sum()

    # Regime-Specific Factors
    regime_specific_factors = (volatility_trend > volatility_trend.rolling(window=20).quantile(0.9)).astype(int)

    # Intraday Reversal Pattern
    intraday_reversal_pattern = (df['high'] - df['close']) - (df['open'] - df['low'])

    # Opening Gap Impact
    opening_gap_impact = df['open'] - df['close'].shift(1)

        price_momentum -

        rsi +
