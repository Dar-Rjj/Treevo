```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    
    # Exponential Moving Average (EMA) with smoothing factor of 2/(N+1) over a 10-day window
    n = 10
    alpha = 2 / (n + 1)
    df['ema_close_10'] = df['close'].ewm(alpha=alpha, adjust=False).mean()
    
    # High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * np.sqrt(np.log(df['volume']))
    
    # Composite Factor
    df['composite_factor'] = (df['intraday_reversal'] * df['volume_momentum']) + (df['close'] - df['close'].shift(1))
    
    # Price Momentum
    df['price_momentum'] = df['close'] - df['ema_close_10']
    
    # Volume Adjusted Spread
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    df['volume_adjusted_spread'] = df['high_low_range'] * np.sqrt(relative_volume)
    
    # Enhanced Momentum Factor
    adjusted_high_low_range = df['volume_adjusted_high_low_range'].rolling(window=5).apply(
        lambda x: np.sum(x * np.arange(1, len(x)+1) / len(x)), raw=True)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['composite_factor'] + df['price_momentum'] - df['volume_adjusted_spread'] + adjusted_high_low_range
    
    return df['final_alpha_factor']
```
