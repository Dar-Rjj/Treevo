```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with a 15-day window
    N = 15
    df['EMA_15'] = df['close'].ewm(span=N, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_15']
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Momentum'] = (df['Log_Return'] * np.sqrt(df['Volume_Change'])).rolling(window=14).sum()
    
    # Enhance with Moving Average Crossover
    df['EMA_7'] = df['close'].ewm(span=7, adjust=False).mean()
    df['EMA_28'] = df['close'].ewm(span=28, adjust=False).mean()
    crossover_signal = np.where(df['EMA_7'] > df['EMA_28'], 1, np.where(df['EMA_7'] < df['EMA_28'], -1, 0))
    df['Enhanced_Crossover_Signal'] = crossover_signal * 0.1  # Adjust the value as needed
    
    # Incorporate High-Low Range with Integrated Momentum
    df['Integrated_Momentum'] += df['High_Low_Range']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Enhanced Volume Thrust
    upward_thrust = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    downward_thrust = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['Enhanced_Volume_Thrust'] = upward_thrust - downward_thrust + (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Relative_Volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Spread'] = df['Spread'] * np.sqrt(df['Relative_Volume'])
    
    # Calculate Enhanced Momentum Factor
    weights = np.exp(np.linspace(0, 1, 6))
    df['Enhanced_Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=6).apply(lambda x: np.sum(x * weights), raw=True)
    
    # Final Integration
    df['Final_Integration'] = df['Integrated_Momentum'] + df['Intraday_Momentum'] + df['Enhanced_Volume_Thrust'] - \
                              df['Log_Return'].rolling(window=14).mean() + df['Volume_Adjusted_High_Low_Range'].rolling(window=14).mean() + \
                              df['Enhanced_Crossover_Signal']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Price_Momentum'] - df['Volume_Adjusted_Spread'] + df['Enhanced_Momentum_Factor'] + df['Final_Integration']
    
    return df['Final_Alpha_Factor'].dropna()
```
