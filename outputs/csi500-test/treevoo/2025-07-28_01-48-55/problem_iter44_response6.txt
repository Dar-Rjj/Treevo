```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving averages of daily returns
    df['ma_5_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['ma_20_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate 10-day EMA and Price Momentum
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10_close']
    
    # Calculate 15-day WMA and Integrated Momentum
    weights = np.arange(1, 16)
    df['wma_15_close'] = df['close'].rolling(window=15).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    df['integrated_momentum'] = df['close'] - df['wma_15_close'] + df['ma_20_daily_return']
    
    # Calculate Intraday Range and Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Upper and Lower Shadow Lengths and their Ratios
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Calculate 10-day Historical Volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['hist_volatility_10'] = df['log_return'].rolling(window=10).std()
    
    # Calculate Volume Change and Ratio to 20-day Moving Average
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate Log Return with Volume Change
    df['integrated_log_return'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = (df['volume'] * (df['close'] - df['open'])).where(df['close'] > df['open'], 0)
    df['downward_thrust'] = (df['volume'] * (df['open'] - df['close'])).where(df['close'] <= df['open'], 0)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust']
    
    # Calculate 5-day and 20-day EMAs for Crossover Signals
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Crossover Signal
    df['crossover_signal'] = np.where(df['ema_5_close'] > df['ema_20_close'], 1, -1)
    df['enhanced_crossover_signal'] = df['crossover_signal'] * df['integrated_momentum']
    
    # Construct Final Composite Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['intraday_range_ratio'] -
        df['hist_volatility_10'] +
        df['integrated_log_return'] +
        df['enhanced_crossover_signal'] +
        df['enhanced_volume_thrust'] +
        df['hist_volatility_10'] * df['enhanced_volume_thrust'] -
        df['log_return'].rolling(window=20).mean() +
        (df['high'] - df['low']) - (df['open'] - df['close'])
    )
    
    return df['alpha_factor']
```
