```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price Momentum
    close_prices = df['close']
    mean_close = close_prices.rolling(window=n).mean()
    price_momentum = close_prices - mean_close
    
    # Calculate Volume Breakout Indicator
    avg_volume = df['volume'].rolling(window=n).mean()
    breakout_indicator = (df['volume'] > avg_volume).astype(int)
    
    # Combine Momentum and Breakout
    interim_factor = 1 * price_momentum + 0.5 * breakout_indicator
    
    # Adjust Interim Factor by Average Volume
    avg_20_days_volume = df['volume'].rolling(window=20).mean()
    final_factor = (interim_factor * avg_20_days_volume) / df['volume']
    
    return final_factor
```
