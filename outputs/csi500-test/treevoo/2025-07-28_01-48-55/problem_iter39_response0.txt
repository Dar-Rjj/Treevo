```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMA and SMA
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()

    # Price Momentum
    df['Momentum'] = df['close'] - df['EMA_10']
    df['ROC_5'] = df['close'].pct_change(5)

    # Volume Adjusted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range_30'] = df['High_Low_Range'].rolling(window=30).mean()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Adj_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(np.log1p(df['Volume_Ratio']))

    # Enhanced Momentum Factor
    weights = np.exp(np.arange(7)) / np.sum(np.exp(np.arange(7)))
    df['Enhanced_Momentum'] = (df['Adj_High_Low_Range'].rolling(window=7).apply(lambda x: np.sum(weights * x), raw=True))

    # Daily Price Action and Volatility
    df['ATR_14'] = df[['high', 'low', 'close']].pct_change().abs().max(axis=1).rolling(window=14).mean() * df['close']
    df['Range_Ratio'] = df['High_Low_Range'] / df['High_Low_Range'].rolling(window=30).mean()
    df['Close_Position'] = (df['close'] - df['open']) / df['High_Low_Range']

    # Market Sentiment
    df['Up_Day'] = df['close'] > df['open']
    df['Down_Day'] = df['close'] < df['open']
    df['Sentiment_Score'] = df['Up_Day'].rolling(window=30).sum() - df['Down_Day'].rolling(window=30).sum()
    df['Overnight_Gap'] = df['open'] - df['close'].shift(1)

    # Trend Indicator
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Trend_Indicator'] = np.where(df['SMA_50'] > df['close'], -1, 1)

    # Incorporate Price Volatility
    df['Volatility_30'] = df['close'].rolling(window=30).std()

    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + RS))

    # On-Balance Volume (OBV)
    df['OBV'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'OBV'] += df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'OBV'] -= df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Momentum'] 
        - df['Adj_High_Low_Range'] 
        + df['Enhanced_Momentum'] 
        * df['Trend_Indicator'] 
        * df['Volatility_30'] 
        + np.where(df['RSI_14'] > 70, -1, np.where(df['RSI_14'] < 30, 1, 0)) 
        + df['OBV']
    )

    return df['Alpha_Factor']
```
