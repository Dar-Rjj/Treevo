```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Average Volume over 20 Days
    df['Average_Volume_20D'] = df['volume'].rolling(window=20).mean()
    
    # Adjust Momentum by Average Volume
    df['Momentum_Adjusted_by_Volume'] = (df['Price_Momentum'] * df['Average_Volume_20D']) / df['volume']
    
    # Calculate Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    df['Final_Alpha_Factor'] = df['Momentum_Adjusted_by_Volume'] / df['Price_Volatility']
    
    return df['Final_Alpha_Factor']
```
