```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    wma_window = 10
    weights = np.arange(1, wma_window + 1)
    df['WMA'] = df['close'].rolling(window=wma_window).apply(lambda prices: np.dot(prices, weights) / weights.sum(), raw=True)

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']

    # Calculate Daily Log Return
    df['LogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)

    # Integrate Log Return with Volume Change
    df['IntegratedMomentum'] = (df['LogReturn'] * df['VolumeChange']).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (ema_5 > ema_20).astype(int) - (ema_5 < ema_20).astype(int)
    df['IntegratedMomentum'] += crossover_signal * 0.01

    # Incorporate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    df['IntegratedMomentum'] *= df['HighLowRange']

    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    df['UpwardThrust'] = 0.0
    df['DownwardThrust'] = 0.0
    conditions = [
        (df['close'] > df['open']),
        (df['close'] <= df['open'])
    ]
    choices_up = [df['volume'] * (df['close'] - df['open']), 0.0]
    choices_down = [0.0, df['volume'] * (df['open'] - df['close'])]
    df['UpwardThrust'] = np.select(conditions, choices_up, default=0.0)
    df['DownwardThrust'] = np.select(conditions, choices_down, default=0.0)
    df['EnhancedVolumeThrust'] = (df['UpwardThrust'] - df['DownwardThrust']) + (df['high'] - df['low']) / df['open']

    # Final Integration
    df['FinalFactor'] = df['IntegratedMomentum'] + df['IntradayMomentum'] + df['EnhancedVolumeThrust']

    # Add Price Momentum and Subtract Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['RelativeVolume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']
    df['FinalAlphaFactor'] = df['FinalFactor'] + df['PriceMomentum'] - df['VolumeAdjustedSpread']

    return df['FinalAlphaFactor']
```
