```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the 20-day moving average of daily returns
    df['ma_daily_return_20'] = df['daily_return'].rolling(window=20).mean()
    
    # Intraday range
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Volume change from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend'] = np.where(df['volume_change'] > 0, 1, -1)
    
    # Ratio of volume to the 20-day moving average of volume
    df['ma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['ma_volume_20']
    
    # Composite factor by combining 20-day momentum of daily returns with 20-day moving average volume ratio
    df['composite_factor'] = 0.6 * df['ma_daily_return_20'] + 0.4 * df['volume_ratio']
    
    # Correlation between 20-day moving average of daily returns and 20-day moving average of volume
    df['ma_volume_20'] = df['volume'].rolling(window=20).mean()
    corr = df[['ma_daily_return_20', 'ma_volume_20']].corr().unstack().iloc[1]
    df['correlation_factor'] = corr
    
    # Combined price-volume factor
    df['combined_factor'] = 0.5 * df['ma_daily_return_20'] + 0.3 * df['volume_ratio'] + 0.2 * df['correlation_factor']
    
    # Weighted Moving Average (WMA) over a 10-day window
    weights = np.arange(1, 11)
    df['wma_10'] = df['close'].rolling(window=10).apply(lambda x: np.sum(weights * x) / np.sum(weights))
    
    # High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * np.log(df['volume'])
    
    # Price Momentum
    df['price_momentum'] = df['close'] - df['wma_10']
    
    # Volume Adjusted Spread
    df['spread'] = df['high'] - df['low']
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['volume_adjusted_spread'] = df['spread'] * df['relative_volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_adjusted_high_low_range'].rolling(window=20).sum()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['price_momentum'] - df['volume_adjusted_spread'] + df['momentum_factor']
    
    return df['final_alpha_factor']
```
