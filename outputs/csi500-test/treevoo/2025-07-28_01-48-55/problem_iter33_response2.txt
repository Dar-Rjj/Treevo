```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) and Simple Moving Average (SMA)
    N = 10
    df['SMA_10'] = df['close'].rolling(window=N).mean()
    df['EMA_10'] = df['close'].ewm(span=N, adjust=False).mean()
    
    # Calculate Price Momentum and Momentum Strength
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    df['ROC_5'] = df['close'].pct_change(periods=5)
    
    # Calculate Volume Adjusted High-Low Range and Trading Volume Patterns
    df['Range'] = df['high'] - df['low']
    df['Average_Range_30'] = df['Range'].rolling(window=30).mean()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Range'] = df['Range'] * np.sqrt(np.log(df['Volume_Ratio']))
    
    # Calculate Enhanced Momentum Factor
    window_size = 7
    weights = np.arange(1, window_size + 1)
    df['Enhanced_Momentum_Factor'] = df['Volume_Adjusted_Range'].rolling(window=window_size).apply(lambda x: np.sum(x * weights), raw=True)
    
    # Analyze Daily Price Action and Volatility
    df['ATR_14'] = df['close'].rolling(window=14).std()
    df['Normalized_Range'] = df['Range'] / df['Average_Range_30']
    df['Close_Open_Ratio'] = (df['close'] - df['open']) / df['Range']
    
    # Evaluate Market Sentiment
    df['Up_Day'] = (df['close'] > df['open']).astype(int)
    df['Down_Day'] = (df['close'] < df['open']).astype(int)
    df['Sentiment_Ratio'] = df['Up_Day'].rolling(window=30).sum() / df['Down_Day'].rolling(window=30).sum()
    df['Gap'] = df['open'] - df['close'].shift(1)
    
    # Trend Indicator
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Trend_Indicator'] = np.where(df['SMA_50'] > df['close'], -1, 1)
    
    # Incorporate Price Volatility
    df['Std_30'] = df['close'].rolling(window=30).std()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_Range'] 
        + df['Enhanced_Momentum_Factor'] 
        * df['Trend_Indicator'] 
        / df['Std_30']
    )
    
    return df['Final_Alpha_Factor']
```
