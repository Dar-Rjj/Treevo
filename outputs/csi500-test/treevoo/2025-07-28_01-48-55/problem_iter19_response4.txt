```python
def heuristics_v2(df):
    # Intraday and Price Composite
    intraday_price_composite = (df['high'] - df['low']) - (df['open'] - df['close']) + (df['close'] - df['open'])
    
    # Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Final Factor
    final_factor = (intraday_price_composite * volume_trend) - volume_trend
    
    return final_factor
```
