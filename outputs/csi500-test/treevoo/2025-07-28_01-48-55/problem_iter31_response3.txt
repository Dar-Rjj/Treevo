import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    volume_momentum = df['volume'].diff(1)
    
    # Calculate Exponential Moving Average (EMA)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    volume_adjusted_high_low_range = high_low_range * np.sqrt(np.log(relative_volume))
    
    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + df['close'].diff(1)
    
    # Price Momentum
    price_momentum = df['close'] - ema_10
    
    # Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / avg_volume_20
    volume_adjusted_spread = spread * np.sqrt(relative_volume)
    
    # Enhanced Momentum Factor
    weights = np.arange(1, 6)
    rolling_window = df['close'].rolling(window=5)
    enhanced_momentum_factor = (rolling_window.apply(lambda x: np.sum(weights * (x - x[0])) / np.sum(weights), raw=True))
    
    # Final Alpha Factor
    final_alpha_factor = (
        composite_factor 
        + price_momentum 
        - volume_adjusted_spread 
        + enhanced_momentum_factor
    )
    
    return final_alpha_factor
