```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with smoothing factor 0.2 over a 15-day window
    df['EMA'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()
    
    # Identify Crossover of 5-day and 30-day SMAs
    df['SMA_Crossover'] = np.where(df['SMA_5'] > df['SMA_30'], 1, -1)
    
    # Calculate Volume-Weighted High-Low Range
    df['VW_HL_Range'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    df['Sum_7_Days_Ranges'] = df['Daily_Range'].rolling(window=7).sum()
    
    # Calculate True Range
    df['Prev_Open'] = df['open'].shift(1)
    df['True_Range'] = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs().combine(
        (df['high'] - df['Prev_Open']).abs(), max
    ).combine(
        (df['low'] - df['Prev_Open']).abs(), max
    )
    
    # Calculate Average True Range (ATR) with a 10-day window
    df['ATR'] = df['True_Range'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume-Adjusted True Range
    df['Volume_Adjusted_TR'] = df['True_Range'] * np.log(df['volume'])
    
    # Calculate Momentum Factor
    df['Momentum_Factor'] = df['VW_HL_Range'].rolling(window=15).mean()
    
    # Analyze Volume for Market Strength
    df['Avg_Volume_7D'] = df['volume'].rolling(window=7).mean()
    df['Volume_Spike'] = np.where(df['volume'] > 1.2 * df['Avg_Volume_7D'], 1, 0)
    
    # Evaluate Buying or Selling Pressure
    df['Buy_Signal'] = np.where((df['close'] > df['open']) & (df['volume'] > df['Avg_Volume_7D']), 1, 0)
    df['Sell_Signal'] = np.where((df['close'] < df['open']) & (df['volume'] > df['Avg_Volume_7D']), -1, 0)
    
    # Measure Volume Spikes
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Spike_Alert'] = np.where(df['Volume_Change'] > 0.4, 1, 0)
    
    # Integrate Price and Volume Indicators
    df['Combined_Signal'] = 0.5 * df['Price_Momentum'] + 0.5 * (df['Volume_Spike_Alert'] * df['Momentum_Factor'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_TR'] 
        + df['Momentum_Factor'] 
        + df['SMA_Crossover'] 
        + df['Combined_Signal'] 
        - df['ATR']
    )
    
    return df['Alpha_Factor']
```
