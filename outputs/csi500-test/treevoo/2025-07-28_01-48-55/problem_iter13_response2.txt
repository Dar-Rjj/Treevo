```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Change
    volume_change = df['volume'].diff()
    
    # Composite Factor
    composite_factor = intraday_reversal * volume_change
    
    return composite_factor
```
