```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['integrated_momentum'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = 0
    df.loc[df['ema_5'] > df['ema_20'], 'crossover_signal'] = 1
    df.loc[df['ema_5'] < df['ema_20'], 'crossover_signal'] = -1
    df['integrated_momentum'] += df['crossover_signal']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['integrated_momentum'] *= df['high_low_range']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = 0
    df['downward_thrust'] = 0
    df.loc[df['close'] > df['open'], 'upward_thrust'] = df['volume'] * (df['close'] - df['open'])
    df.loc[df['close'] <= df['open'], 'downward_thrust'] = df['volume'] * (df['open'] - df['close'])
    df['price_range_factor'] = (df['high'] - df['low']) / df['open']
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + df['price_range_factor']
    
    # Incorporate Price-Volatility Interaction
    df['historical_volatility'] = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)
    df['enhanced_volume_thrust'] *= df['historical_volatility']
    
    # Final Integration
    df['final_factor'] = df['integrated_momentum'] + df['intraday_momentum'] + df['enhanced_volume_thrust']
    
    return df['final_factor']
```
