```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of closing price in the range
    close_pos = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Volume-weighted price
    vwp = (df['amount'] / df['volume'])

    # Calculate the relative difference between the volume-weighted price and the close price
    rel_diff = (vwp - df['close']) / df['close']

    # Compute a factor based on the interaction between the relative position and the relative difference
    factor = close_pos * rel_diff
    
    return factor
```
