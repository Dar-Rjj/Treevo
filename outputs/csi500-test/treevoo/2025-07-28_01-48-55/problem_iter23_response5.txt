import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    median_close_20 = df['close'].rolling(window=20).median()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    adjusted_price_momentum = (df['close'] - median_close_20) * avg_volume_20 / df['volume']
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Daily Volume Change
    daily_volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    intermediate_factor = 1.5 * adjusted_price_momentum + intraday_reversal * daily_volume_change
    
    # Calculate Exponential Moving Average (EMA)
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA)
    wma_weights = np.arange(1, 11)
    def wma(x):
        return np.nansum(wma_weights * x) / np.nansum(wma_weights)
    wma_close = df['close'].rolling(window=10).apply(wma, raw=True)
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Price Momentum
    price_momentum = df['close'] - ema_close
    
    # Calculate Additional Price Momentum
    additional_price_momentum = df['close'] - wma_close
    
    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (
        intermediate_factor + 
        price_momentum + 
        additional_price_momentum - 
        volume_adjusted_spread + 
        momentum_factor
    )
    
    return final_alpha_factor
