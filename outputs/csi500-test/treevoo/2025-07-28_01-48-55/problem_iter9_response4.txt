```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['SMA_10']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=10).mean()
    df['Relative_Volume'] = df['volume'] / avg_volume
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']

    # Calculate Range Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=10).mean()
    df['Range_Volatility'] = df['Average_Daily_Range']

    # Calculate Price Oscillator
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['Price_Oscillator'] = df['SMA_10'] - df['SMA_5']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_Spread'] 
        - df['Range_Volatility'] 
        + df['Price_Oscillator']
    )
    
    return df['Final_Alpha_Factor']
```
