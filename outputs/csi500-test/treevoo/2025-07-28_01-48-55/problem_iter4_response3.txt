```python
def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['high'] - df['low'] - (df['open'] - df['close'])
    
    # Measure Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate 5-day Moving Average of Volume
    moving_average_volume = df['volume'].rolling(window=5).mean()
    
    # Combine Intraday Reversal and Volume Change
    factor = (intraday_reversal * volume_change) / moving_average_volume
    
    return factor
```
