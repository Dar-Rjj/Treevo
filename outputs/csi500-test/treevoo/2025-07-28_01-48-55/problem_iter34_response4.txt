```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 20-day moving average of daily returns
    df['momentum_20d'] = df['daily_return'].rolling(window=20).mean()
    
    # Determine the intraday range and its ratio to previous day's close
    df['intraday_range'] = df['high'] - df['low']
    df['range_to_close_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Measure the upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Assess the change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute the ratio of volume to the 20-day moving average of volume
    df['volume_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20d_ma']
    
    # Create a composite factor by combining the 20-day momentum of daily returns with the 20-day moving average volume ratio
    df['composite_factor'] = 0.6 * df['momentum_20d'] + 0.4 * df['volume_ratio']
    
    # Calculate the correlation between 20-day moving average of daily returns and 20-day moving average of volume
    df['returns_20d_ma'] = df['daily_return'].rolling(window=20).mean()
    df['volume_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['corr_returns_volume'] = df['returns_20d_ma'].rolling(window=20).corr(df['volume_20d_ma'])
    
    # Create a combined price-volume factor
    df['combined_factor'] = 0.5 * df['momentum_20d'] + 0.3 * df['volume_ratio'] + 0.2 * df['corr_returns_volume']
    
    return df['combined_factor']
```
