```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_range = high_low_range * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range (20-day window)
    avg_volume_adjusted_range = volume_adjusted_range.rolling(window=20).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return (20-day window)
    avg_daily_log_return = daily_log_return.rolling(window=20).mean()
    
    # Combine Volume Adjusted High-Low Volatility and Momentum
    combined_metric = avg_volume_adjusted_range - avg_daily_log_return
    
    # Adjust Combined Metric by Intraday and Volume
    intraday_momentum = (df['close'] - df['close'].shift(20)) + (df['high'] - df['low']) / df['low']
    avg_20d_volume = df['volume'].rolling(window=20).mean()
    adjusted_combined_metric = (combined_metric * intraday_momentum) / df['volume'] * avg_20d_volume
    
    return adjusted_combined_metric
```
