def heuristics_v2(data):
    # Calculate Price Momentum
    price_momentum = data['close'] - data['close'].shift(20)
    
    # Adjust Momentum by Average Volume
    avg_volume_20_days = data['volume'].rolling(window=20).mean()
    adjusted_momentum = (price_momentum * avg_volume_20_days) / data['volume']
    
    # Incorporate Price Volatility
    price_volatility_20_days = data['close'].rolling(window=20).std()
    final_alpha_factor = adjusted_momentum / price_volatility_20_days
    
    return final_alpha_factor
