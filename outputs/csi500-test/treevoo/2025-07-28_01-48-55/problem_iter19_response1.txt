import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Price Momentum with WMA
    weights = np.arange(1, 11)
    close_prices = df['close'].rolling(window=10, min_periods=10).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)
    price_momentum = df['close'] - close_prices
    
    # Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Integrated Momentum
    integrated_momentum = (daily_log_return * volume_change).rolling(window=10, min_periods=10).sum()
    
    # Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (ema_5 > ema_20).astype(int) - (ema_5 < ema_20).astype(int)
    integrated_momentum += crossover_signal
    
    # Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']).astype(int) * (df['volume'] * (df['close'] - df['open']))
    downward_thrust = (df['close'] <= df['open']).astype(int) * (df['volume'] * (df['open'] - df['close']))
    price_range_factor = (df['high'] - df['low']) / df['open']
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + price_range_factor
    
    # Volume Adjusted Spread
    spread = df['high'] - df['low']
    average_volume = df['volume'].rolling(window=20, min_periods=20).mean()
    relative_volume = df['volume'] / average_volume
    volume_adjusted_spread = spread * relative_volume
    
    # Final Alpha Factor
    final_alpha_factor = (
        integrated_momentum + 
        intraday_momentum + 
        enhanced_volume_thrust + 
        price_momentum - 
        volume_adjusted_spread
    )
    
    return final_alpha_factor
