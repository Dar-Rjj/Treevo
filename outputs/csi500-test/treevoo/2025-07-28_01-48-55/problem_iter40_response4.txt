```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Exponential Moving Average (EMA) with smoothing factor 0.3 over a 15-day window
    ema = data['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Price Momentum
    price_momentum = data['open'] - ema
    
    # Calculate True Range
    true_range = data[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    
    # Calculate Volume-Adjusted True Range
    volume_adjusted_true_range = true_range * np.sqrt(data['volume'])
    
    # Calculate Volume-Weighted High-Low Range
    high_low_range = data['high'] - data['low']
    volume_weighted_high_low_range = (high_low_range * data['volume']) / data['volume'].rolling(window=7).mean()
    
    # Calculate Momentum Factor using a 7-day rolling window
    momentum_factor = volume_weighted_high_low_range.rolling(window=7).mean()
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum - volume_adjusted_true_range + momentum_factor
    
    return final_alpha_factor
```
