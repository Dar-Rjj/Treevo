```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])

    # Calculate Weighted Moving Average (WMA)
    def wma(values, window):
        weights = np.arange(1, window + 1)
        return values.rolling(window=window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    close_wma_10 = wma(df['close'], 10)

    # Calculate Price Momentum
    price_momentum = df['close'] - close_wma_10

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = spread * relative_volume

    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()

    # Calculate Weighted Price Change
    weighted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']

    # Calculate Range Volatility
    daily_range = df['high'] - df['low']
    range_volatility = daily_range.rolling(window=10).mean()

    # Calculate Price Oscillator
    sma_5 = df['close'].rolling(window=5).mean()
    price_oscillator = close_wma_10 - sma_5

    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)

    # Combined Intraday and Price Momentum
    combined_intraday_and_price_momentum = intraday_reversal + price_momentum

    # Intermediate Composite
    intermediate_composite = combined_intraday_and_price_momentum * volume_momentum

    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum
        - volume_adjusted_spread
        + momentum_factor
        + weighted_price_change
        - range_volatility
        + price_oscillator
        - intermediate_composite
    )

    return final_alpha_factor
```
