```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with smoothing factor 0.18 over a 15-day window
    ema = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = (df['high'] - df['low']) * np.sqrt(df['volume'])
    
    # Calculate Price Momentum
    price_momentum = df['open'] - ema
    
    # Calculate Volume-Adjusted True Range
    true_range = np.maximum(df['high'] - df['low'], 
                            np.abs(df['high'] - df['open'].shift(1)), 
                            np.abs(df['low'] - df['open'].shift(1)))
    volume_adjusted_true_range = true_range * (np.log(df['volume']))**2
    
    # Calculate Momentum Factor with a 20-day rolling window
    momentum_factor = volume_weighted_high_low_range.rolling(window=20).mean()
    
    # Calculate Price Change Ratio
    price_change_ratio = df['close'] / df['close'].shift(1)
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Integrate Momentum and Volume Change
    integrated_momentum = price_momentum * volume_change_ratio
    
    # Final Alpha Factor
    final_alpha_factor = (price_momentum 
                          - volume_adjusted_true_range 
                          + momentum_factor 
                          + integrated_momentum)
    
    return final_alpha_factor
```
