```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Simple Moving Average (SMA)
    sma = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - sma
    
    # Calculate Volume Adjusted Spread
    average_volume = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / average_volume
    spread = df['high'] - df['low']
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    upward_thrust = 0
    downward_thrust = 0
    for i in range(len(df)):
        if df.iloc[i]['close'] > df.iloc[i]['open']:
            upward_thrust += df.iloc[i]['volume']
        else:
            downward_thrust += df.iloc[i]['volume']
    volume_thrust = upward_thrust - downward_thrust
    
    # Final Alpha Factor
    combined_momentum = price_momentum + intraday_momentum
    final_alpha_factor = combined_momentum - volume_adjusted_spread + volume_thrust
    
    return final_alpha_factor
```
