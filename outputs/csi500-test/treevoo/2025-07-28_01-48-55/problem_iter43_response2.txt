```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the 20-day moving average of daily returns
    df['20_day_ma_returns'] = df['daily_return'].rolling(window=20).mean()
    
    # Compute WMA with linearly increasing weights over a 10-day window
    weights = list(range(1, 11))
    df['WMA_10_close'] = (df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / sum(weights), raw=True))
    
    # Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA_10_close']
    
    # Intraday range
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Volume change from previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # 20-day moving average of volume
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['20_day_avg_volume']
    
    # Composite factor combining 20-day momentum and 20-day moving average volume ratio
    df['composite_factor'] = 0.5 * df['20_day_ma_returns'] + 0.5 * df['volume_ratio']
    
    # High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * df['volume'].apply(lambda x: pd.np.log(x) if x > 0 else 0)
    
    # Relative Volume
    df['relative_volume'] = df['volume'] / df['20_day_avg_volume']
    
    # Volume Adjusted Spread
    df['volume_adjusted_spread'] = df['high_low_range'] * df['relative_volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['Price_Momentum'] - df['volume_adjusted_spread']
    
    return df['final_alpha_factor']
```
