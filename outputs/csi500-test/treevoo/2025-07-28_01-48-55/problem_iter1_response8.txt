import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Rolling Average of High-Low Difference with a 20-day window
    df['avg_high_low_diff'] = df['high_low_diff'].rolling(window=20).mean()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Rolling Average of Daily Log Return with a 20-day window
    df['avg_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Combine High-Low Volatility and Momentum
    df['alpha_factor'] = df['avg_high_low_diff'] - df['avg_log_return']
    
    return df['alpha_factor']
