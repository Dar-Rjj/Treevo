```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Comprehensive Price and Intraday Momentum
    close_t = df['close']
    close_t_minus_20 = df['close'].shift(20)
    high_t = df['high']
    low_t = df['low']
    price_momentum = (close_t - close_t_minus_20) + (high_t - low_t) / low_t
    
    # Adjust Momentum by Average Volume and Amount
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    adjusted_momentum = price_momentum * avg_volume * avg_amount / (df['volume'] * df['amount'])
    
    # Calculate High-Low Price Range
    high_low_range = high_t - low_t
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Simple Moving Average (SMA)
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum_sma = close_t - sma_10
    
    # Calculate Volume Adjusted Spread
    spread = high_t - low_t
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / avg_volume_10
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=20).sum()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_momentum + price_momentum_sma - volume_adjusted_spread + momentum_factor
    
    return final_alpha_factor
```
