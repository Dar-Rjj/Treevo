```python
def heuristics_v2(data):
    # Calculate Intraday Momentum
    data['intraday_momentum'] = (data['high'] - data['low']) / data['low']
    
    # Initialize Upward Thrust and Downward Thrust
    data['upward_thrust'] = 0
    data['downward_thrust'] = 0
    
    # Compute Upward Thrust
    data.loc[data['close'] > data['open'], 'upward_thrust'] = data['volume']
    
    # Compute Downward Thrust
    data.loc[data['close'] <= data['open'], 'downward_thrust'] = data['volume']
    
    # Subtract Downward Thrust from Upward Thrust
    data['volume_thrust'] = data['upward_thrust'] - data['downward_thrust']
    
    # Combine Intraday Momentum and Volume Thrust
    data['factor_value'] = data['intraday_momentum'] * data['volume_thrust']
    
    return data['factor_value']
```
