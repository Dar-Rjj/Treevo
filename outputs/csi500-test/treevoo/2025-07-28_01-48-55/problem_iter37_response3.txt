```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Short-term and medium-term momentum
    df['5_day_momentum'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_momentum'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA)
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Weighted Moving Average (WMA)
    weights = np.arange(1, 16)
    wma = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    df['Integrated_Momentum'] = df['close'] - wma + df['20_day_momentum']
    
    # Intraday range and volatility
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    df['Volume_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    df['10_day_volatility'] = df['close'].pct_change().rolling(window=10).std()
    
    # Volume and price interaction
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Integrated_Log_Return'] = (df['Daily_Log_Return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['Upward_Thrust'] = df.apply(lambda x: x['volume'] * (x['close'] - x['open']) if x['close'] > x['open'] else 0, axis=1)
    df['Downward_Thrust'] = df.apply(lambda x: x['volume'] * (x['open'] - x['close']) if x['close'] <= x['open'] else 0, axis=1)
    df['Enhanced_Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust']
    
    # Moving average crossover signals
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Crossover_Signal'] = np.where(df['EMA_5'] > df['EMA_20'], 1, -1)
    
    # Final composite alpha factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        + df['Integrated_Momentum'] 
        - df['Volume_Adjusted_High_Low_Range'] 
        - (df['intraday_range'] * df['volume_ratio'])
        + df['Integrated_Log_Return'] 
        + (df['Crossover_Signal'] * 0.5) 
        + (df['10_day_volatility'] * df['Enhanced_Volume_Thrust'])
    )
    
    return df['Final_Alpha_Factor'].dropna()
```
