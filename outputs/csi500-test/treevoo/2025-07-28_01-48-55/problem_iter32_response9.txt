```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with a 10-day window
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) with exponentially increasing weights over 15 days
    weights = np.arange(1, 16)
    df['WMA_15'] = df['close'].rolling(window=15).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Integrated Momentum
    df['Integrated_Momentum'] = (df['close'] - df['WMA_15']) + df['Price_Momentum']
    
    # Calculate Volume Adjusted High-Low Range
    df['High_Low_Range'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    # Calculate Volume Adjusted Spread
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume_30
    df['Volume_Adjusted_Spread'] = (df['high'] - df['low']) * relative_volume
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Log_Return'] = (df['Daily_Log_Return'] * df['Volume_Change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (df['EMA_5'] > df['EMA_20']).astype(int) - (df['EMA_5'] < df['EMA_20']).astype(int)
    df['Crossover_Signal'] = crossover_signal * 1  # Positive value for the signal
    
    # Calculate Enhanced Volume Thrust
    df['Upward_Thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['Downward_Thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['Price_Range_Factor'] = (df['high'] - df['low']) / df['open']
    df['Enhanced_Volume_Thrust'] = (df['Upward_Thrust'] - df['Downward_Thrust']) + df['Price_Range_Factor']
    
    # Rolling Average of Daily Log Return
    df['Rolling_Avg_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).mean()
    
    # Final Integration
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] +
        df['Integrated_Momentum'] +
        df['Enhanced_Volume_Thrust'] -
        df['High_Low_Range'] -
        df['Volume_Adjusted_Spread'] +
        df['Integrated_Log_Return'] +
        df['Crossover_Signal'] -
        df['Rolling_Avg_Log_Return']
    )
    
    return df['Final_Alpha_Factor']
```
