```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-Low Range by Multiplying with Logarithm of Volume
    volume_adjusted_high_low = high_low_range * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range (20-day window)
    rolling_avg_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=20).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Rolling Average of Daily Log Return (20-day window)
    rolling_avg_daily_log_return = daily_log_return.rolling(window=20).mean()
    
    # Combine Volume Adjusted High-Low Volatility and Momentum
    factor_values = rolling_avg_volume_adjusted_high_low - rolling_avg_daily_log_return
    
    return factor_values
```
