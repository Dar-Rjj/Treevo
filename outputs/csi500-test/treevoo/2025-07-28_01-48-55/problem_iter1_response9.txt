```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    df['mean_close'] = df['close'].rolling(window=n).mean()
    df['price_momentum'] = df['close'] - df['mean_close']
    
    # Calculate Volume Breakout Indicator
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_breakout'] = (df['volume'] > df['avg_volume']).astype(int)
    
    # Combine Momentum and Breakout
    df['factor'] = 1 * df['price_momentum'] + 0.5 * df['volume_breakout']
    
    return df['factor']
```
