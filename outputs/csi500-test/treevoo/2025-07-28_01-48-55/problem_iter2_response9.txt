```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window
    df['SMA'] = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['SMA']
    
    # Calculate High-Low Price Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['VolumeAdjustedHighLowRange'] = df['HighLowRange'] * np.log(df['volume'])
    
    # Calculate Rolling Momentum Factor with a 5-day window
    df['RollingMomentumFactor'] = df['VolumeAdjustedHighLowRange'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['PriceMomentum'] - df['VolumeAdjustedHighLowRange'] + df['RollingMomentumFactor']
    
    return df['FinalAlphaFactor'].dropna()
```
