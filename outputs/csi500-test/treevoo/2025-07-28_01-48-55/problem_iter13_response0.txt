import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def calculate_wma(prices, window=10):
        weights = list(range(1, window + 1))
        wma = prices.rolling(window=window).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
        return wma
    
    # Calculate Price Momentum
    def price_momentum(close, wma):
        return close - wma
    
    # Calculate Volume Adjusted Spread
    def volume_adjusted_spread(high, low, volume):
        spread = high - low
        avg_volume = volume.rolling(window=20).mean()
        relative_volume = volume / avg_volume
        return spread * relative_volume
    
    # Calculate Momentum Factor
    def momentum_factor(high, low, window=10):
        adjusted_high_low_range = high - low
        return adjusted_high_low_range.rolling(window=window).sum()
    
    # Compute WMA
    wma = calculate_wma(df['close'])
    
    # Calculate Price Momentum
    price_momentum = price_momentum(df['close'], wma)
    
    # Calculate Volume Adjusted Spread
    vas = volume_adjusted_spread(df['high'], df['low'], df['volume'])
    
    # Calculate Momentum Factor
    mom_factor = momentum_factor(df['high'], df['low'])
    
    # Final Alpha Factor
    alpha_factor = (price_momentum - vas + mom_factor).dropna()
    
    return alpha_factor
