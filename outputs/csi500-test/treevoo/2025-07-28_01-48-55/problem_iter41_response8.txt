```python
import pandas as pd

def heuristics_v2(df, window=20):
    # Compute the difference between today's close and today's open
    close_open_diff = df['close'] - df['open']
    
    # Sum the difference over a period to find trend
    trend_sum = close_open_diff.rolling(window).sum()
    
    # Identify the ratio of today's close to today's high
    close_high_ratio = df['close'] / df['high']
    
    # Track its change over time
    close_high_ratio_change = close_high_ratio.pct_change()
    
    # Analyze the gap from today's close to today's low
    close_low_gap = df['close'] - df['low']
    
    # Consider its average over a defined window
    close_low_gap_avg = close_low_gap.rolling(window).mean()
    
    # Calculate the cumulative sum of the differences between today's close and the previous day's close to detect momentum
    close_momentum = (df['close'] - df['close'].shift(1)).cumsum()
    
    # Relate changes in closing price with percentage change in volume
    close_vol_pct_change = df['close'].pct_change() * df['volume'].pct_change()
    
    # Measure the correlation between daily returns and volume
    daily_returns = df['close'].pct_change()
    vol_corr = daily_returns.rolling(window).corr(df['volume'].pct_change())
    
    # Analyze the ratio of today's volume to the moving average of volume over a specified period
    vol_ma_ratio = df['volume'] / df['volume'].rolling(window).mean()
    
    # Compare the current day’s amount traded with the previous day’s to assess market interest
    amount_interest = df['amount'] / df['amount'].shift(1)
    
    # Establish a factor based on the moving average of the amount traded relative to the closing price
    amount_close_ma_ratio = df['amount'].rolling(window).mean() / df['close']
    
    # Calculate the ratio of today's amount traded to the moving average of the amount traded over a specified period
    amount_ma_ratio = df['amount'] / df['amount'].rolling(window).mean()
    
    # Develop a factor that combines the trend of the closing price difference with the percentage change in volume
    combined_trend_vol = trend_sum * close_vol_pct_change
    
    # Create a factor that integrates the ratio of today's close to today's high with the correlation between daily returns and volume
    combined_close_high_vol_corr = close_high_ratio * vol_corr
    
    # Formulate a factor that synthesizes the gap from today's close to today's low with the ratio of today's volume to the moving average of volume
    combined_close_low_vol_ma = close_low_gap_avg * vol_ma_ratio
    
    # Construct a composite factor that merges the moving average of the amount traded relative to the closing price with the ratio of today's amount traded to the moving average of the amount traded
    composite_factor = amount_close_ma_ratio * amount_ma_ratio
    
    # Final alpha factor
    alpha_factor = 0.25 * (combined_trend_vol + combined_close_high_vol_corr + combined_close_low_vol_ma + composite_factor)
    
    return alpha_factor
```
