```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 16)  # Exponentially increasing weights
    df['WMA'] = df['close'].rolling(window=15).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']
    
    # Calculate Daily Log Return
    df['DailyLogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['IntegratedMomentum'] = (df['DailyLogReturn'] * df['VolumeChange']).rolling(window=15).sum()
    
    # Enhance with Moving Average Crossover
    df['7EMA'] = df['close'].ewm(span=7, adjust=False).mean()
    df['25EMA'] = df['close'].ewm(span=25, adjust=False).mean()
    df['CrossoverSignal'] = np.where(df['7EMA'] > df['25EMA'], 1, np.where(df['7EMA'] < df['25EMA'], -1, 0))
    df['EnhancedMomentum'] = df['IntegratedMomentum'] + df['CrossoverSignal']
    
    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['UpwardThrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['DownwardThrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['PriceRangeFactor'] = (df['high'] - df['low']) / df['open']
    df['EnhancedVolumeThrust'] = df['UpwardThrust'] - df['DownwardThrust'] + df['PriceRangeFactor']
    
    # Final Integration
    df['FinalIntegration'] = df['EnhancedMomentum'] + df['IntradayMomentum'] + df['EnhancedVolumeThrust']
    
    # Final Alpha Factor
    df['VolumeAdjustedSpread'] = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=30).mean())
    df['AlphaFactor'] = df['PriceMomentum'] + df['FinalIntegration'] - df['VolumeAdjustedSpread']
    
    return df['AlphaFactor']
```
