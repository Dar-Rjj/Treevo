import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    upward_thrust = 0
    downward_thrust = 0
    for i in range(len(df)):
        if df.iloc[i]['close'] > df.iloc[i]['open']:
            upward_thrust += df.iloc[i]['volume'] * (df.iloc[i]['close'] - df.iloc[i]['open'])
        else:
            downward_thrust += df.iloc[i]['volume'] * (df.iloc[i]['open'] - df.iloc[i]['close'])

    price_range_factor = (df['high'] - df['low']) / df['open']
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + price_range_factor

    # Calculate Accumulated Volume Impact
    accumulated_volume = 0
    for i in range(len(df)):
        if df.iloc[i]['close'] > df.iloc[i]['open']:
            accumulated_volume += df.iloc[i]['volume']
        else:
            accumulated_volume -= df.iloc[i]['volume']

    daily_volume_impact = accumulated_volume / len(df)

    # Combine Final Alpha Factor
    alpha_factor = intraday_momentum + enhanced_volume_thrust + daily_volume_impact
    return alpha_factor
