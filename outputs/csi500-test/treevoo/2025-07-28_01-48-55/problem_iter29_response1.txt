```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the money flow as the product of close, volume and amount
    money_flow = df['close'] * df['volume'] * df['amount']
    
    # Generate a signal based on the combination of these factors
    factor_value = close_change * high_low_ratio * money_flow
    
    return factor_value
```
