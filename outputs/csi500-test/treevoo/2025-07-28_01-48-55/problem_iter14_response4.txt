```python
def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Price Momentum
    price_momentum = df['close'] - df['open']
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combined Intraday and Price Momentum
    combined_intraday_price = intraday_reversal + price_momentum
    
    # Intermediate Composite
    intermediate_composite = combined_intraday_price * volume_change
    
    # Final Composite Factor
    final_composite_factor = intermediate_composite - volume_change
    
    return final_composite_factor
```
