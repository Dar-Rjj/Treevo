```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Simple Moving Average (SMA) over a 10-day window
    sma = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - sma
    
    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    average_volume = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / average_volume
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor_window = 10
    momentum_factor = volume_adjusted_high_low_range.rolling(window=momentum_factor_window).sum()
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum - volume_adjusted_spread + momentum_factor
    
    return final_alpha_factor
```
