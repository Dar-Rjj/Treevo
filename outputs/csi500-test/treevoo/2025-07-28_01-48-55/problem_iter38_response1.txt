def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']

    # Calculate the difference between close and open prices
    df['close_open_diff'] = df['close'] - df['open']

    # Calculate the ratio of volume to amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']

    # Calculate the cumulative sum of the daily return
    df['cumulative_return'] = df['daily_return'].cumsum()

    # Construct the alpha factor
    df['alpha_factor'] = (df['close_open_diff'] / df['high_low_diff']) * df['volume_amount_ratio'] * df['cumulative_return']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
