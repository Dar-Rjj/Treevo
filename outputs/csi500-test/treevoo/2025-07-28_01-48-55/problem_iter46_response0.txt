```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily returns based on close prices
    df['daily_return'] = df['close'].diff()
    
    # Create a factor that measures the strength of the opening action
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Form a factor using the formula: (close - low) / (high - low)
    df['close_position_in_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Create a factor that combines the daily return with the volume
    df['return_volume_factor'] = df['daily_return'] * df['volume']
    
    # Calculate the difference in volume from the previous day
    df['volume_change'] = df['volume'].diff()
    
    # Use the formula: (daily volume change) * (daily return)
    df['volume_change_return_factor'] = df['volume_change'] * df['daily_return']
    
    # Sum the daily returns over the past 5 days
    df['five_day_return_sum'] = df['daily_return'].rolling(window=5).sum()
    
    # Calculate the average of the past 5 days' close prices
    df['five_day_close_avg'] = df['close'].rolling(window=5).mean()
    
    # Create a factor by subtracting this average from the current close
    df['close_vs_five_day_avg'] = df['close'] - df['five_day_close_avg']
    
    # Calculate the standard deviation of the close prices over the past 20 days
    df['twenty_day_close_std'] = df['close'].rolling(window=20).std()
    
    # Create a z-score factor for the current close price
    df['close_z_score'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['twenty_day_close_std']
    
    # Aggregate different individual factors into a single score
    df['alpha_factor'] = (
        0.2 * df['daily_return'] +
        0.1 * df['open_close_ratio'] +
        0.1 * df['close_position_in_range'] +
        0.1 * df['return_volume_factor'] +
        0.1 * df['volume_change_return_factor'] +
        0.1 * df['five_day_return_sum'] +
        0.1 * df['close_vs_five_day_avg'] +
        0.2 * df['close_z_score']
    )
    
    return df['alpha_factor']
```
