```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['high'] - df['low']
    
    # Adjust for Opening Range
    adjusted_intraday_reversal = intraday_reversal - (df['open'] - df['close'])
    
    # Measure Volume Change
    volume_change = df['volume'].diff(1)
    
    # Combine Intraday Reversal and Volume Change
    factor_values = adjusted_intraday_reversal * volume_change
    
    return factor_values
```
