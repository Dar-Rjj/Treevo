def heuristics_v2(df):
    # Calculate the 5-day Simple Moving Average (SMA) of the closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Generate a Momentum Factor using the close price
    df['Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate the percentage change in volume from one day to the next
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Compute the average daily range over the past 5 days
    df['Range'] = df['high'] - df['low']
    df['Avg_Range_5'] = df['Range'].rolling(window=5).mean()
    
    # Define the composite factor as (Momentum / Average 5-Day Range) * Volume Change
    df['Composite_Factor'] = (df['Momentum'] / df['Avg_Range_5']) * df['Volume_Change']
    
    # Use the difference between the number of transactions at the bid and ask prices to measure order imbalance
    # This is a placeholder for the actual OFI calculation, which requires more detailed data
    df['OFI'] = (df['amount'] - df['amount'].shift(1)) / df['volume']
    
    # Formulate a High-Low Breakout Factor
    df['High_Breakout'] = df['high'] > df['high'].rolling(window=5).mean() + 2 * df['high'].rolling(window=5).std()
    df['Low_Breakout'] = df['low'] < df['low'].rolling(window=5).mean() - 2 * df['low'].rolling(window=5).std()
    
    # The final alpha factor is the Composite Factor
    return df['Composite_Factor']
