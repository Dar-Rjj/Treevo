```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    log_volume = np.log(df['volume'])
    adjusted_high_low_range = high_low_range * log_volume
    
    # Set rolling window size
    window_size = 20  # You can adjust the window size as needed
    
    # Calculate Momentum Factor using a rolling sum
    momentum_factor = adjusted_high_low_range.rolling(window=window_size).sum()
    
    return momentum_factor
```
