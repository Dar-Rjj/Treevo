```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price which gives an idea of the day's trend
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the percentage of the high that the close reached
    close_to_high = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow, indicating buying or selling pressure
    money_flow = (df['close'] - df['open']) * df['volume']
    
    # Heuristic: A combination of the day's trend and the money flow
    factor = close_open_ratio + (money_flow * close_to_high)
    
    return factor
```
