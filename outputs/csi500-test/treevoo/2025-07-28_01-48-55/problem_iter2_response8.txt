```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Combined Price Momentum
    df['mean_close'] = df['close'].rolling(window=3).mean()
    df['combined_price_momentum'] = (df['close'] - df['mean_close']) * 1
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = ((df['high'] - df['low']) / df['low']) * 0.5
    
    # Calculate Volume Thrust Indicator
    df['upward_thrust'] = 0
    df['downward_thrust'] = 0
    
    df.loc[df['close'] > df['open'], 'upward_thrust'] = df['volume']
    df.loc[df['close'] <= df['open'], 'downward_thrust'] = df['volume']
    
    df['volume_thrust_indicator'] = df['upward_thrust'] - df['downward_thrust']
    
    # Combine All Factors
    df['alpha_factor'] = df['combined_price_momentum'] + df['intraday_volatility'] + df['volume_thrust_indicator']
    
    return df['alpha_factor']
```
