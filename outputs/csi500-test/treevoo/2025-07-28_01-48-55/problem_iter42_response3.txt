```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns based on close prices
    df['daily_return'] = df['close'].diff()
    
    # Factor that measures the strength of the opening action
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Factor using the relative position of the close within the daily range
    df['close_range_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Factor that combines the daily return with the volume
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Calculate the difference in volume from the previous day
    df['volume_change'] = df['volume'].diff()
    
    # Factor combining volume change and daily return
    df['volume_change_return'] = df['volume_change'] * df['daily_return']
    
    # Sum the daily returns over the past 5 days as a momentum indicator
    df['momentum_5d'] = df['daily_return'].rolling(window=5).sum()
    
    # Factor by subtracting the average of the past 5 days' close prices from the current close
    df['deviation_from_mean'] = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the standard deviation of the close prices over the past 20 days
    df['std_20d'] = df['close'].rolling(window=20).std()
    
    # Z-score factor for the current close price
    df['z_score'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['std_20d']
    
    # Composite factor: Aggregate different individual factors into a single score
    df['alpha_factor'] = (
        0.2 * df['open_close_ratio'] +
        0.1 * df['close_range_position'] +
        0.1 * df['volume_weighted_return'] +
        0.1 * df['volume_change_return'] +
        0.2 * df['momentum_5d'] +
        0.2 * df['deviation_from_mean'] +
        0.1 * df['z_score']
    )
    
    return df['alpha_factor']
```
