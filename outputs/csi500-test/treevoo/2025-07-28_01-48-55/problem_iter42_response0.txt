```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 20-day and 50-day moving averages of daily returns
    df['20_day_momentum'] = df['daily_return'].rolling(window=20).mean()
    df['50_day_momentum'] = df['daily_return'].rolling(window=50).mean()
    
    # Calculate intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Ratio of intraday range to previous day's close
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Ratio of intraday range to the 20-day average intraday range
    df['20_day_avg_intraday_range'] = df['intraday_range'].rolling(window=20).mean()
    df['intraday_range_to_avg_ratio'] = df['intraday_range'] / df['20_day_avg_intraday_range']
    
    # Measure upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - np.maximum(df['open'], df['close'])
    df['lower_shadow'] = np.minimum(df['open'], df['close']) - df['low']
    
    # Ratio of upper shadow length to lower shadow length
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 20-day moving average of the shadow length ratio
    df['20_day_avg_shadow_ratio'] = df['shadow_ratio'].rolling(window=20).mean()
    
    # Change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # 20-day moving average of the volume change
    df['20_day_avg_volume_change'] = df['volume_change'].rolling(window=20).mean()
    
    # Ratio of volume to the 20-day moving average of volume
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_to_avg_ratio'] = df['volume'] / df['20_day_avg_volume']
    
    # Composite factor by combining 20-day momentum of daily returns with 20-day moving average volume ratio
    df['composite_factor'] = 0.5 * df['20_day_momentum'] + 0.5 * df['volume_to_avg_ratio']
    
    # Correlation between 20-day moving average of daily returns and 20-day moving average of volume
    df['20_day_corr_returns_volume'] = df['20_day_momentum'].rolling(window=20).corr(df['volume_to_avg_ratio'])
    
    # 50-day correlation for a longer-term view
    df['50_day_corr_returns_volume'] = df['50_day_momentum'].rolling(window=50).corr(df['volume_to_avg_ratio'])
    
    # Weighted Moving Average (WMA) over a 10-day window
    weights = np.arange(1, 11)
    df['wma_10'] = df['close'].rolling(window=10).apply(lambda x: np.sum(weights * x) / np.sum(weights))
    
    # High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * np.log(df['volume'])
    
    # Price Momentum
    df['price_momentum'] = df['close'] - df['wma_10']
    
    # Volume Adjusted Spread
    df['spread'] = df['high'] - df['low']
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['volume_adjusted_spread'] = df['spread'] * df['relative_volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_adjusted_high_low_range'].rolling(window=20).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] - df['volume_adjusted_spread'] + df['momentum_factor']
    
    return df['alpha_factor'].dropna()
```
