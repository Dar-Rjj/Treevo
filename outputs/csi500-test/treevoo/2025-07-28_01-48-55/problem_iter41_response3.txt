```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()

    # Compute 5-day and 20-day moving average of daily returns
    df['ma_5d_return'] = df['daily_return'].rolling(window=5).mean()
    df['ma_20d_return'] = df['daily_return'].rolling(window=20).mean()

    # Calculate Exponential Moving Average (EMA) with a 10-day window
    n = 10
    k = 2 / (n + 1)
    df['ema_10d'] = df['close'].ewm(alpha=k, adjust=False).mean()

    # Calculate Weighted Moving Average (WMA) with a 15-day window
    weights = np.arange(1, 16)
    df['wma_15d'] = df['close'].rolling(window=15).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)

    # Calculate Price Momentum and Integrated Momentum
    df['price_momentum'] = df['close'] - df['ema_10d']
    df['integrated_momentum'] = df['close'] - df['wma_15d'] + df['ma_20d_return']

    # Intraday price dynamics and volatility
    df['intraday_range_ratio'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']

    # 10-day Historical Volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_10d'] = df['log_return'].rolling(window=10).std() * np.sqrt(252)

    # Volume and its interaction with price
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['log_return_vol_adj'] = df['log_return'] * df['volume_change']
    df['accumulated_log_return_vol_adj'] = df['log_return_vol_adj'].rolling(window=10).sum()

    # Enhanced Volume Thrust
    df['upward_thrust'] = df.apply(lambda row: row['volume'] * (row['close'] - row['open']) if row['close'] > row['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda row: row['volume'] * (row['open'] - row['close']) if row['close'] <= row['open'] else 0, axis=1)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']

    # Moving Average Crossover Signals
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = (df['ema_5d'] > df['ema_20d']).astype(int) - (df['ema_5d'] < df['ema_20d']).astype(int)

    # Final composite alpha factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        (df['high'] - df['low']) * df['volume_ratio'] -
        (df['upper_shadow'] - df['lower_shadow']) * df['volume_ratio'] +
        df['accumulated_log_return_vol_adj'] +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] +
        df['volatility_10d'] * df['enhanced_volume_thrust']
    )

    # Apply 20-day rolling window to calculate the average of Daily Log Return
    df['rolling_avg_log_return'] = df['log_return'].rolling(window=20).mean()

    # Subtract the Rolling Average of Daily Log Return from the final alpha factor
    df['alpha_factor'] -= df['rolling_avg_log_return']

    return df['alpha_factor']
```
