```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price and Intraday Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    df['momentum_with_intraday'] = df['price_momentum'] + df['intraday_range']

    # Adjust Momentum by Average Volume
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['adjusted_momentum'] = (df['momentum_with_intraday'] * df['avg_volume_20_days']) / df['volume']

    return df['adjusted_momentum']
```
