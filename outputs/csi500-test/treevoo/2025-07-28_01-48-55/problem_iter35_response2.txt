```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Create a new feature based on the interaction between high-low difference and volume-amount ratio
    df['interaction_feature'] = df['high_low_diff'] * df['volume_amount_ratio']
    
    # Calculate the weighted sum of the interaction feature and the close price change
    df['weighted_sum'] = 0.5 * df['close_change'] + 0.5 * df['interaction_feature']
    
    # Return the factor values as a pandas Series
    return df['weighted_sum']
```
