```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window
    sma = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - sma
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum - volume_adjusted_high_low_range
    
    return final_alpha_factor
```
