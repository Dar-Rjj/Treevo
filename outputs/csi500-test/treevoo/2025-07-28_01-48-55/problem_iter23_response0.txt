import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Daily Log Return
    daily_log_return = np.log(df['close']).diff()

    # Volume Change
    volume_change = df['volume'].pct_change()

    # Integrated Momentum
    integrated_momentum = (daily_log_return * volume_change).rolling(window=15).sum()

    # Exponential Moving Averages
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    crossover_signal = (ema_7 > ema_21) * 1 - (ema_7 < ema_21) * 1
    integrated_momentum += crossover_signal

    # Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['open']

    # Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']) * df['volume'] * (df['close'] - df['open'])
    downward_thrust = (df['close'] <= df['open']) * df['volume'] * (df['open'] - df['close'])
    price_range_factor = (df['high'] - df['low']) / df['low']
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + price_range_factor

    # Adjusted Price Momentum
    median_close = df['close'].rolling(window=20).median()
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    adjusted_price_momentum = (df['close'] - median_close) * (avg_volume_20_days / df['volume'])

    # Volume Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    volume_adjusted_high_low_range = high_low_range * np.sqrt(np.log(df['volume']))

    # Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    volume_adjusted_spread = spread * relative_volume

    # Enhanced Momentum Factor
    weights = np.arange(1, 6)
    enhanced_momentum_factor = (volume_adjusted_high_low_range.rolling(window=5).apply(lambda x: np.sum(x * weights), raw=True))

    # Final Alpha Factor
    final_alpha_factor = (
        integrated_momentum + 
        intraday_momentum + 
        enhanced_volume_thrust + 
        adjusted_price_momentum + 
        enhanced_momentum_factor - 
        volume_adjusted_spread
    )

    return final_alpha_factor
