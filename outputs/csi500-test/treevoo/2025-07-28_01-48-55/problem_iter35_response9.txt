```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 20-day moving average of daily returns
    df['ma_20_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Intraday range
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Ratio of volume to the 20-day moving average of volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Weighted Moving Average (WMA) with linearly increasing weights
    weights = list(range(1, 11))
    df['wma_10'] = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
    
    # Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * (df['volume']).apply(lambda x: (x ** 0.5) if x > 0 else 0)
    
    # Price Momentum
    df['price_momentum'] = df['close'] - df['wma_10']
    
    # Volume Adjusted Spread
    df['spread'] = df['high'] - df['low']
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['volume_adjusted_spread'] = df['spread'] * (df['relative_volume']).apply(lambda x: (x ** 0.5) if x > 0 else 0)
    
    # Composite Factor
    df['composite_factor'] = (df['intraday_reversal'] * df['volume_momentum']) + (df['close'] - df['close'].shift(1))
    
    # Momentum Factor
    df['momentum_factor'] = df['volume_adjusted_high_low_range'].rolling(window=20).sum()
    
    # Enhanced Momentum
    df['enhanced_momentum'] = df['price_momentum'] + df['momentum_factor']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['composite_factor'] + df['enhanced_momentum'] - df['volume_adjusted_spread']
    
    return df['alpha_factor'].dropna()
```
