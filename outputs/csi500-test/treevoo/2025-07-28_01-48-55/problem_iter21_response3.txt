import pandas as pd

def heuristics_v2(df):
    # Short-Term Price Momentum
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    factor_1 = df['SMA_5'] - df['SMA_10']
    
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    factor_2 = df['SMA_20'] - df['SMA_50']
    
    # Long-Term Price Momentum
    df['SMA_100'] = df['close'].rolling(window=100).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    factor_3 = df['SMA_100'] - df['SMA_200']
    
    # Exponential Moving Average (EMA) Based Momentum
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    factor_4 = df['EMA_5'] - df['EMA_10']
    
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    factor_5 = df['EMA_20'] - df['EMA_50']
    
    # Volume Momentum
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    factor_6 = df['Volume_MA_5'] - df['Volume_MA_10']
    
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_MA_50'] = df['volume'].rolling(window=50).mean()
    factor_7 = df['Volume_MA_20'] - df['Volume_MA_50']
    
    # Volume-Weighted Price Changes
    df['VWAP_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWAP_10'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    factor_8 = df['VWAP_5'] - df['VWAP_10']
    
    df['VWAP_20'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWAP_50'] = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    factor_9 = df['VWAP_20'] - df['VWAP_50']
    
    # Price Breakout Indicators
    df['52_week_High'] = df['high'].rolling(window=252).max()
    df['52_week_Low'] = df['low'].rolling(window=252).min()
    factor_10 = (df['close'] - df['52_week_Low']) / (df['52_week_High'] - df['52_week_Low'])
    
    df['26_week_High'] = df['high'].rolling(window=126).max()
    df['26_week_Low'] = df['low'].rolling(window=126).min()
    factor_11 = (df['close'] - df['26_week_Low']) / (df['26_week_High'] - df['26_week_Low'])
    
    # Price Reversal Indicators (RSI)

        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
