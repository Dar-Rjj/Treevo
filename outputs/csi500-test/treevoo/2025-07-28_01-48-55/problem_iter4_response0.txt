```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-Low Range by Volume
    adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Momentum Factor using a rolling sum over a 10 day window
    momentum_factor = adjusted_high_low_range.rolling(window=10).sum()
    
    return momentum_factor
```
