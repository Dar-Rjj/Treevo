```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Volume Adjusted High-Low Range
    high_low_spread = df['high'] - df['low']
    log_volume = np.log(df['volume'])
    vol_adjusted_high_low_range = high_low_spread * np.sqrt(log_volume)
    
    # Enhanced Momentum Factor
    weights = np.arange(1, 6)  # Linearly increasing weights over a 5-day window
    adjusted_high_low_range = high_low_spread * np.sqrt(log_volume)
    rolling_window = adjusted_high_low_range.rolling(window=5)
    enhanced_momentum_factor = rolling_window.apply(lambda x: np.sum(x * weights), raw=True) / np.sum(weights)
    
    # Final Alpha Factor
    final_alpha_factor = (
        (intraday_reversal * volume_momentum) + 
        (df['close'] - df['close'].shift(1)) + 
        price_momentum - 
        vol_adjusted_high_low_range + 
        enhanced_momentum_factor
    )
    
    return final_alpha_factor
```
