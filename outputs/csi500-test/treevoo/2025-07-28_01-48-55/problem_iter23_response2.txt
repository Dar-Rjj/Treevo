```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on the close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range which is (high - low) / open
    df['relative_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume to amount ratio
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the weighted average of high and low prices, with weights being the volume
    df['volume_weighted_price'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Define the heuristic factor
    df['heuristic_factor'] = (
        df['daily_return'] * df['relative_range'] * 
        df['volume_to_amount_ratio'] * df['volume_weighted_price']
    )
    
    # Return the heuristic factor as a pandas Series
    return df['heuristic_factor'].dropna()
```
