```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day's close to the current day's open
    df['price_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate the intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative amount traded
    df['relative_amount'] = df['amount'] / df['volume']
    
    # Create a factor combining these three features
    df['alpha_factor'] = (df['price_change'] * df['intraday_volatility']) / df['relative_amount']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
