```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day Simple Moving Average (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Generate Momentum Factor
    df['Momentum'] = df['close'] - df['close'].shift(5)
    
    # Compute 10-day Weighted Moving Average (WMA) with linearly increasing weights
    weights = np.arange(1, 11)
    df['WMA_10'] = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA_10']
    
    # Volume Change Indicator
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Volume Momentum
    df['Volume_Momentum'] = df['volume'] - df['volume'].shift(1)
    
    # High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Average 5-Day Range
    df['Avg_5Day_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    
    # Volume Adjusted High-Low Range
    df['Volume_Adjusted_Range'] = df['High_Low_Range'] * np.sqrt(np.log(df['volume']))
    
    # Composite Factor
    df['Composite_Factor'] = (df['Momentum'] / df['Avg_5Day_Range']) * df['Volume_Change']
    
    # Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Add to Composite Factor
    df['Composite_Factor'] += (df['Intraday_Reversal'] * df['Volume_Momentum']) + (df['close'] - df['close'].shift(1))
    
    # Enhanced Momentum
    df['Enhanced_Momentum'] = df['Price_Momentum'] + df['Momentum'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Composite_Factor'] + df['Enhanced_Momentum'] - df['Volume_Adjusted_Range']
    
    return df['Alpha_Factor']
```
