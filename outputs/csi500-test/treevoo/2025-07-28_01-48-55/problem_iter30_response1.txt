```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute moving averages of daily returns
    df['5_day_momentum'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_momentum'] = df['daily_return'].rolling(window=20).mean()
    df['60_day_momentum'] = df['daily_return'].rolling(window=60).mean()
    
    # Examine intraday price dynamics
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio_close'] = df['intraday_range'] / df['close'].shift(1)
    df['avg_intraday_range_20'] = df['intraday_range'].rolling(window=20).mean()
    df['intraday_range_ratio_avg'] = df['intraday_range'] / df['avg_intraday_range_20']

    # Measure the upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    df['avg_upper_shadow_20'] = df['upper_shadow'].rolling(window=20).mean()
    df['avg_lower_shadow_20'] = df['lower_shadow'].rolling(window=20).mean()
    df['upper_shadow_ratio_avg'] = df['upper_shadow'] / df['avg_upper_shadow_20']
    df['lower_shadow_ratio_avg'] = df['lower_shadow'] / df['avg_lower_shadow_20']

    # Investigate volume for trading interest
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_to_avg_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['avg_volume_change_20'] = df['volume_change'].rolling(window=20).mean()

    # Integrate price and volume indicators
    df['composite_factor_20'] = 0.6 * df['20_day_momentum'] + 0.4 * df['volume_to_avg_ratio']
    df['composite_factor_60'] = 0.7 * df['60_day_momentum'] + 0.3 * df['volume_to_avg_ratio']

    return df[['5_day_momentum', '20_day_momentum', '60_day_momentum',
               'intraday_range_ratio_close', 'intraday_range_ratio_avg', 
               'shadow_ratio', 'upper_shadow_ratio_avg', 'lower_shadow_ratio_avg',
               'volume_change', 'volume_to_avg_ratio', 'avg_volume_change_20',
               'composite_factor_20', 'composite_factor_60']]
```
