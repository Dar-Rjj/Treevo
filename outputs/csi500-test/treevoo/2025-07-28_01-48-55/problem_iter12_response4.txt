```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Momentum'] = (df['Log_Return'] * df['Volume_Change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    df['Crossover_Signal'] = 0
    df.loc[df['EMA_5'] > df['EMA_20'], 'Crossover_Signal'] = 1
    df.loc[df['EMA_5'] < df['EMA_20'], 'Crossover_Signal'] = -1
    
    # Incorporate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Normalized_High_Low_Range'] = df['High_Low_Range'] / df['close']
    df['Integrated_Momentum'] *= df['Normalized_High_Low_Range']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['Upward_Thrust'] = (df['volume'] * (df['close'] - df['open'])).where(df['close'] > df['open'], 0)
    df['Downward_Thrust'] = (df['volume'] * (df['open'] - df['close'])).where(df['close'] <= df['open'], 0)
    df['Price_Range_Factor'] = (df['high'] - df['low']) / df['open']
    df['Enhanced_Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust'] + df['Price_Range_Factor']
    
    # Calculate High-Low Price Range
    df['High_Low_Price_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Price_Range'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['Rolling_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=20).mean()
    
    # Rolling Average of Daily Log Return
    df['Rolling_Daily_Log_Return'] = df['Log_Return'].rolling(window=20).mean()
    
    # Integrate with Previous Day's Close
    df['Integration_with_Prev_Close'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Final Integration
    df['Final_Alpha'] = (
        df['Integrated_Momentum'] + 
        df['Intraday_Momentum'] + 
        df['Enhanced_Volume_Thrust'] + 
        df['Crossover_Signal'] + 
        df['Integration_with_Prev_Close'] + 
        (df['Rolling_Volume_Adjusted_High_Low_Range'] - df['Rolling_Daily_Log_Return'])
    )
    
    return df['Final_Alpha']
```
