```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price and Intraday Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    df['Momentum'] = df['Price_Momentum'] + df['Intraday_Momentum']

    # Adjust Momentum by Average Volume and Amount
    df['Average_Volume_20D'] = df['volume'].rolling(window=20).mean()
    df['Average_Amount_20D'] = df['amount'].rolling(window=20).mean()

    df['Adjusted_Momentum'] = (df['Momentum'] * 
                               df['Average_Volume_20D'] * 
                               df['Average_Amount_20D']) / (df['volume'] * df['amount'])

    return df['Adjusted_Momentum'].dropna()
```
