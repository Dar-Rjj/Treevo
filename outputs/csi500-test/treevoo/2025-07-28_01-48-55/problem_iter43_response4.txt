```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving averages of daily returns
    df['5d_ma_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['20d_ma_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA)
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Weighted Moving Average (WMA)
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20d_ma_daily_return']
    
    # Intraday range
    df['intraday_range'] = (df['high'] - df['low'])
    df['range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Shadow lengths
    df['upper_shadow'] = (df['high'] - df[['open', 'close']].max(axis=1))
    df['lower_shadow'] = (df[['open', 'close']].min(axis=1) - df['low'])
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 10-day Historical Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['10d_volatility'] = df['log_returns'].rolling(window=10).std() * np.sqrt(252)
    
    # Volume analysis
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['vol_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['integrated_log_return'] = (df['daily_log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = df.apply(lambda row: row['volume'] * (row['close'] - row['open']) if row['close'] > row['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda row: row['volume'] * (row['open'] - row['close']) if row['close'] <= row['open'] else 0, axis=1)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust']
    
    # Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Moving average crossover signals
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, -1)
    
    # Construct final composite alpha factor
    df['final_alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['range_ratio'] -
        df['10d_volatility'] +
        df['integrated_log_return'] +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] +
        df['10d_volatility'] * df['enhanced_volume_thrust'] +
        df['intraday_reversal']
    )
    
    return df['final_alpha_factor']
```
