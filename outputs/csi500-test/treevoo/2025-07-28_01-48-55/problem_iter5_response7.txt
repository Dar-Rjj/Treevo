```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['SMA_10']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Relative_Volume'] = df['volume'] / df['Avg_Volume_10']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    df['Upward_Thrust'] = np.where(df['close'] > df['open'], df['volume'], 0)
    df['Downward_Thrust'] = np.where(df['close'] <= df['open'], df['volume'], 0)
    df['Upward_Thrust'] = df['Upward_Thrust'].rolling(window=10).sum()
    df['Downward_Thrust'] = df['Downward_Thrust'].rolling(window=10).sum()
    df['Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust']
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['Rolling_Avg_Volume_Adjusted_High_Low_Range'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=20).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return
    df['Rolling_Avg_Daily_Log_Return'] = df['Daily_Log_Return'].rolling(window=20).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] + 
        df['Intraday_Momentum'] - 
        df['Volume_Adjusted_Spread'] + 
        df['Volume_Thrust'] - 
        (df['Rolling_Avg_Volume_Adjusted_High_Low_Range'] - df['Rolling_Avg_Daily_Log_Return'])
    )
    
    return df['Final_Alpha_Factor']
```
