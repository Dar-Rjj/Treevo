```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 16)
    wma_15 = df['close'].rolling(window=15).apply(lambda prices: np.dot(prices, weights) / weights.sum(), raw=True)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - ema_10
    
    # Calculate Integrated Momentum
    integrated_momentum = (df['close'] - wma_15) + price_momentum
    
    # Calculate Volume Adjusted High-Low Range
    high_low_range_vol_adj = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    # Calculate Volume Adjusted Spread
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    rel_vol = df['volume'] / avg_vol_30
    vol_adj_spread = (df['high'] - df['low']) * rel_vol
    
    # Integrate Log Return with Volume Change
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    volume_change = df['volume'] / df['volume'].shift(1)
    log_ret_vol_change = daily_log_return * volume_change
    integrated_log_return = log_ret_vol_change.rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (ema_5 > ema_20).astype(int) - (ema_5 < ema_20).astype(int)
    enhanced_crossover = np.where(crossover_signal == 1, 1, np.where(crossover_signal == -1, -1, 0))
    enhanced_momentum = integrated_momentum + enhanced_crossover
    
    # Calculate Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']).astype(int) * (df['close'] - df['open']) * df['volume']
    downward_thrust = (df['close'] <= df['open']).astype(int) * (df['open'] - df['close']) * df['volume']
    price_range_factor = (df['high'] - df['low']) / df['open']
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + price_range_factor
    
    # Incorporate Price-Volatility Interaction
    historical_volatility_10 = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)
    price_volatility_interaction = historical_volatility_10 * enhanced_volume_thrust
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum +
        enhanced_momentum -
        high_low_range_vol_adj -
        vol_adj_spread +
        integrated_log_return +
        enhanced_crossover +
        enhanced_volume_thrust -
        daily_log_return.rolling(window=5).mean() +
        price_volatility_interaction
    )
    
    return final_alpha_factor
```
