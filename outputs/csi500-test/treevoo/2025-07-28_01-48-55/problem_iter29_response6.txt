```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price difference between high and low for the current day
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage of close price with respect to the price difference
    close_percent = (df['close'] - df['low']) / price_diff
    
    # Calculate the volume-weighted average price (VWAP) based on the day's close
    vwap = (df['amount'] / df['volume']) * (df['close'] / df['open'])
    
    # Calculate a smoothed VWAP using the exponential moving average (EMA)
    vwap_ema = vwap.ewm(span=5, adjust=False).mean()
    
    # Combine the close_percent and the smoothed VWAP to create the alpha factor
    alpha_factor = close_percent * vwap_ema
    
    return alpha_factor
```
