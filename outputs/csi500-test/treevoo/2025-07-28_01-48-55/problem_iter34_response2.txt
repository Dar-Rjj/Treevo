import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal: (High - Low) - (Open - Close)
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Weighted Moving Average (WMA) of Close prices over a 10-day window
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Volume Momentum: Today’s Volume - Yesterday’s Volume
    volume_momentum = df['volume'] - df['volume'].shift(1)

    # High-Low Range: High - Low
    high_low_range = df['high'] - df['low']

    # Volume Adjusted High-Low Range: Multiply Range by Square Root of Logarithm of Volume
    volume_adjusted_high_low_range = high_low_range * np.sqrt(np.log(df['volume']))

    # Price Momentum: Subtract WMA from Current Close price
    price_momentum = df['close'] - wma

    # Volume Adjusted Spread
    spread = high_low_range
    average_20_day_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_20_day_volume
    volume_adjusted_spread = spread * np.sqrt(relative_volume)

    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + (df['close'] - df['close'].shift(1))

    # Momentum Factor: Sum of Volume Adjusted High-Low Range over a Rolling Window
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()

    # Enhanced Momentum
    enhanced_momentum = price_momentum + momentum_factor

    # Final Alpha Factor
    final_alpha_factor = composite_factor + enhanced_momentum - volume_adjusted_spread

    return final_alpha_factor
