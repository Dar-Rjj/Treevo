import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = df['close'].apply(lambda x: np.log(x)) - df['close'].shift(1).apply(lambda x: np.log(x))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['integrated_momentum'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = 0
    df.loc[df['ema_5'] > df['ema_20'], 'crossover_signal'] = 1
    df.loc[df['ema_5'] < df['ema_20'], 'crossover_signal'] = -1
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['integrated_momentum'] *= df['high_low_range']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = 0
    df['downward_thrust'] = 0
    df.loc[df['close'] > df['open'], 'upward_thrust'] = df['volume'] * (df['close'] - df['open'])
    df.loc[df['close'] <= df['open'], 'downward_thrust'] = df['volume'] * (df['open'] - df['close'])
    df['price_range_factor'] = (df['high'] - df['low']) / df['open']
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + df['price_range_factor']
    
    # Calculate High-Low Price Range
    df['high_low_price_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_price_range'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['avg_volume_adjusted_high_low_range'] = df['volume_adjusted_high_low_range'].rolling(window=20).mean()
    
    # Rolling Average of Daily Log Return
    df['avg_daily_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Final Integration
    df['final_alpha'] = (
        df['integrated_momentum'] + 
        df['intraday_momentum'] + 
        df['enhanced_volume_thrust'] + 
        (df['avg_volume_adjusted_high_low_range'] - df['avg_daily_log_return']) + 
        df['crossover_signal']
    )
    
    return df['final_alpha']
