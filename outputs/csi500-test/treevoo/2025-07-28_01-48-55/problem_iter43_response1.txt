```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving Averages of Daily Returns
    df['5d_ma_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['20d_ma_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) and Price Momentum
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Weighted Moving Average (WMA) and Integrated Momentum
    wma_weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: (x * wma_weights).sum() / wma_weights.sum(), raw=True)
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20d_ma_daily_return']
    
    # Intraday Range and Ratio
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Shadow Lengths and Ratios
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Historical Volatility
    df['log_ret'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['10d_volatility'] = df['log_ret'].rolling(window=10).std()
    
    # Volume Change and Ratio
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrated Log Return with Volume Change
    df['integrated_log_return'] = (df['log_ret'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['downward_thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['enhanced_volume_thrust'] = df['upward_thrust'].rolling(window=10).sum() - df['downward_thrust'].rolling(window=10).sum()
    
    # Moving Average Crossover Signal
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_10'], 1, -1)
    
    # Final Composite Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['intraday_range_ratio'] -
        df['shadow_ratio'] +
        df['integrated_log_return'] +
        df['crossover_signal'] * df['enhanced_volume_thrust'] +
        df['10d_volatility'] * df['enhanced_volume_thrust'] -
        df['log_ret'].rolling(window=20).mean()
    )
    
    return df['alpha_factor']
```
