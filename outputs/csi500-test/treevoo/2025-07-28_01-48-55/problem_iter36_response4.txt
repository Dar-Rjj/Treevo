```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with smoothing factor 0.3 over a 15-day window
    df['EMA'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['Volume_Weighted_HL'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['open'] - df['EMA']
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x[0] - x[1], 
                                                                 abs(x[0] - df.shift(1)['close']),
                                                                 abs(x[1] - df.shift(1)['close'])), axis=1)
    
    # Calculate Volume-Adjusted True Range
    df['Volume_Adjusted_TR'] = df['True_Range'] * np.sqrt(df['volume'])
    
    # Calculate Momentum Factor using a 7-day rolling mean on Volume-Weighted High-Low Range
    df['Momentum_Factor'] = df['Volume_Weighted_HL'].rolling(window=7).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] - df['Volume_Adjusted_TR'] + df['Momentum_Factor'])
    
    return df['Final_Alpha_Factor']
```
