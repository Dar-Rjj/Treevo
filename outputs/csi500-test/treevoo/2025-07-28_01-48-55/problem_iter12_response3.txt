import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def calculate_wma(prices, window=15):
        weights = np.arange(1, window + 1)
        wma = prices.rolling(window=window).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
        return wma
    
    # Exponentially increasing weights
    def calculate_exp_wma(prices, window=15):
        weights = np.exp(np.arange(0, window))
        exp_wma = prices.rolling(window=window).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
        return exp_wma

    # Calculate Price Momentum
    def price_momentum(close, wma):
        return close - wma

    # Calculate Volume Adjusted Spread
    def volume_adjusted_spread(high, low, volume, avg_volume_window=30):
        spread = high - low
        relative_volume = volume / volume.rolling(window=avg_volume_window).mean()
        return spread * relative_volume

    # Calculate Trend Strength
    def trend_strength(close, wma, window=5):
        abs_diff = (close - close.shift(1)).abs()
        abs_diff_sum = abs_diff.rolling(window=window).sum()
        return wma - abs_diff_sum

    # Calculate the final alpha factor
    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    volume = df['volume']

    # Calculate WMA with exponentially increasing weights
    exp_wma = calculate_exp_wma(close_prices)

    # Calculate Price Momentum
    momentum = price_momentum(close_prices, exp_wma)

    # Calculate Volume Adjusted Spread
    vas = volume_adjusted_spread(high_prices, low_prices, volume)

    # Calculate Trend Strength
    ts = trend_strength(close_prices, exp_wma)

    # Final Alpha Factor
    alpha_factor = momentum - vas - ts

    return alpha_factor
