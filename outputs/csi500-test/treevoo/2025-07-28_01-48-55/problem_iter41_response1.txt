```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day simple moving averages of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Identify the crossover of 5-day and 20-day SMAs as a momentum indicator
    df['SMA_Crossover'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Compute the daily range (High - Low)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Sum up the last 10 days' ranges to identify recent market volatility
    df['Sum_10_Days_Range'] = df['Daily_Range'].rolling(window=10).sum()
    
    # Calculate average true range (ATR) over 14 days for volatility measurement
    df['TR'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].abs().max(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Create a 10-day average volume baseline
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Flagging days when volume exceeds the 10-day average by a certain threshold (e.g., 1.5 times)
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Avg_Volume_10']).astype(int)
    
    # Determine if close is above the open on high volume as an indication of strong buying interest
    df['Strong_Buying_Pressure'] = ((df['close'] > df['open']) & (df['volume'] > 1.5 * df['Avg_Volume_10'])).astype(int)
    
    # Detect days with a close below the open and high volume, signaling strong selling pressure
    df['Strong_Selling_Pressure'] = ((df['close'] < df['open']) & (df['volume'] > 1.5 * df['Avg_Volume_10'])).astype(int)
    
    # Measure percentage change in volume from one day to the next
    df['Volume_Change_Percent'] = df['volume'].pct_change() * 100
    
    # Alerting when daily volume change surpasses 50% increase from the previous day
    df['Volume_Spike_50'] = (df['Volume_Change_Percent'] > 50).astype(int)
    
    # Create a composite score that weights both momentum and volume signals
    df['Composite_Score'] = 0.7 * df['SMA_Crossover'] + 0.3 * df['Volume_Spike']
    
    # Construct a multi-factor model that considers momentum, volatility, and volume simultaneously
    df['Multi_Factor_Signal'] = (df['SMA_Crossover'] & (df['ATR_14'] > df['ATR_14'].median()) & df['Volume_Spike']).astype(int)
    
    # Refine the signal with logical conditions, such as requiring both a positive momentum and a significant volume spike
    df['Refined_Signal'] = (df['SMA_Crossover'] & df['Volume_Spike_50']).astype(int)
    
    # Include the impact of open and close prices on the overall trend
    df['Trend_Impact'] = (df['close'] > df['open']).astype(int)
    
    # Assess the consistency of the momentum and volatility signals over time
    df['Consistent_Momentum'] = (df['SMA_Crossover'].rolling(window=5).sum() >= 3).astype(int)
    df['Consistent_Volatility'] = (df['ATR_14'] > df['ATR_14'].rolling(window=10).median()).astype(int)
    
    # Final alpha factor
    df['Alpha_Factor'] = df['Refined_Signal'] * df['Trend_Impact'] * df['Consistent_Momentum'] * df['Consistent_Volatility']
    
    return df['Alpha_Factor']
```
