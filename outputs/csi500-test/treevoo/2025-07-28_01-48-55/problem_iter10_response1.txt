```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Calculate Volume Adjusted Spread
    average_volume_10 = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / average_volume_10
    spread = df['high'] - df['low']
    df['volume_adjusted_spread'] = spread * relative_volume
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    upward_thrust = (df['close'] > df['open']).astype(int) * df['volume']
    downward_thrust = (df['close'] <= df['open']).astype(int) * df['volume']
    df['volume_thrust'] = upward_thrust - downward_thrust
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['price_momentum'] + df['intraday_momentum'] 
                                - df['volume_adjusted_spread'] 
                                + df['volume_thrust'])
    
    return df['final_alpha_factor']
```
