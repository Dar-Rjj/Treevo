```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    
    # Calculate Weighted Moving Average (WMA) over a 10-day window
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - wma
    
    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    average_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume_20
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()
    
    # Calculate Weighted Price Change
    weighted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Range Volatility
    range_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate Price Oscillator
    sma_5 = df['close'].rolling(window=5).mean()
    price_oscillator = wma - sma_5
    
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Combined Intraday and Price Momentum
    combined_intraday_and_price_momentum = intraday_reversal + price_momentum
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Intermediate Composite
    intermediate_composite = combined_intraday_and_price_momentum * volume_change
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum 
        - volume_adjusted_spread 
        + momentum_factor 
        + weighted_price_change 
        - range_volatility 
        + price_oscillator 
        - intermediate_composite
    )
    
    return final_alpha_factor
```
