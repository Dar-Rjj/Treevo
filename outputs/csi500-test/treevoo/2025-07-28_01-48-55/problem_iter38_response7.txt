import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the percentage of the daily range that was closed above the open
    df['range_above_open'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted by the price change
    df['vol_weighted_change'] = df['volume'] * df['price_change']
    
    # Calculate the cumulative sum of the volume weighted price change
    df['cum_vol_weighted_change'] = df['vol_weighted_change'].cumsum()
    
    # Calculate the average trading amount over the last 5 days
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    
    # Heuristic alpha factor: Combine the components in a novel way
    df['alpha_factor'] = (df['range_above_open'] * df['avg_amount_5d'] + df['cum_vol_weighted_change']) / df['volume']
    
    return df['alpha_factor'].dropna()
