```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between close and open prices
    df['rel_diff'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the ratio of volume to amount
    df['vol_to_amt_ratio'] = df['volume'] / df['amount']
    
    # Create a weighted factor combining the relative price difference and the volume to amount ratio
    df['weighted_factor'] = df['rel_diff'] * df['vol_to_amt_ratio']
    
    # Return the series representing the factor values
    return df['weighted_factor']
```
