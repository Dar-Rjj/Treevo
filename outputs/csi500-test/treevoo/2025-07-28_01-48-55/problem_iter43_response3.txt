```python
import pandas as pd

def heuristics_v2(df):
    # Calculate price changes and trends
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    df['intraday_return'] = df['close'] / df['open'] - 1
    df['high_low_range'] = df['high'] - df['low']
    df['trend_direction'] = (df['close'] > df['close'].shift(1)).astype(int)
    
    # Analyze volume and liquidity
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume'] > 2 * df['volume'].shift(1)).astype(int)
    
    # Examine market sentiment
    df['relative_strength'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['market_sentiment'] = (df['close'] > df['open']).astype(int)
    
    # Combine multiple signals
    composite_score = (
        0.3 * df['daily_return'] +
        0.2 * df['intraday_return'] +
        0.1 * df['high_low_range'] +
        0.1 * df['trend_direction'] +
        0.1 * df['volume_change'] +
        0.1 * df['relative_strength'] +
        0.1 * df['market_sentiment']
    )
    
    # Evaluate momentum
    df['sma_5'] = df['daily_return'].rolling(window=5).mean()
    df['lma_20'] = df['daily_return'].rolling(window=20).mean()
    df['momentum'] = (df['sma_5'] > df['lma_20']).astype(int)
    
    # Analyze volatility
    df['daily_volatility'] = df['daily_return'].rolling(window=20).std()
    historical_threshold = df['daily_volatility'].median()
    df['high_volatility'] = (df['daily_volatility'] > historical_threshold).astype(int)
    
    # Assess trading activity
    df['atr_14'] = df[['high', 'low', 'close']].rolling(window=14).apply(
        lambda x: max(x[0] - x[1], x[0] - x[2], x[2] - x[1]), raw=True
    )
    df['increased_volatility'] = (df['atr_14'] > df['atr_14'].shift(1)).astype(int)
    
    # Final alpha factor
    final_alpha_factor = (
        0.2 * df['daily_return'] +
        0.2 * df['intraday_return'] +
        0.1 * df['high_low_range'] +
        0.1 * df['trend_direction'] +
        0.1 * df['volume_change'] +
        0.1 * df['relative_strength'] +
        0.1 * df['market_sentiment'] +
        0.1 * df['momentum'] +
        0.1 * df['high_volatility'] +
        0.1 * df['increased_volatility']
    )
    
    return final_alpha_factor
```
