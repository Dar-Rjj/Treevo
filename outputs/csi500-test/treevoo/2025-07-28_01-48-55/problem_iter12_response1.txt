```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Reversal Calculation
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Acceleration Calculation
    volume_acceleration = df['volume'].diff().diff()
    
    # Composite Factor
    composite_factor = intraday_reversal * volume_acceleration
    
    return composite_factor
```
