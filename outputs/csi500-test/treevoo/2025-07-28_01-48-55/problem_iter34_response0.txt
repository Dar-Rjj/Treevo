```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights over a 10-day window
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low = high_low_range * np.log(df['volume'])
    
    # Calculate Final Alpha Factor
    alpha_factor = df['close'] - wma + volume_adjusted_high_low
    
    return alpha_factor
```
