```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the difference between high and low price
    hl_diff = df['high'] - df['low']
    
    # Calculate the ratio of amount to volume
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Combine the features into a single alpha factor
    alpha_factor = (close_pct_change + hl_diff) * amount_volume_ratio
    
    return alpha_factor
```
