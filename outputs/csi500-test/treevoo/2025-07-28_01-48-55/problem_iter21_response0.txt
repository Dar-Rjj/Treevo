import numpy as np

def heuristics_v2(df):
    # Factor 1: 5-day SMA - 10-day SMA
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    factor_1 = df['5_day_SMA'] - df['10_day_SMA']
    
    # Factor 2: 20-day SMA - 50-day SMA
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    factor_2 = df['20_day_SMA'] - df['50_day_SMA']
    
    # Factor 3: 100-day SMA - 200-day SMA
    df['100_day_SMA'] = df['close'].rolling(window=100).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    factor_3 = df['100_day_SMA'] - df['200_day_SMA']
    
    # Factor 4: 5-day EMA - 10-day EMA
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    factor_4 = df['5_day_EMA'] - df['10_day_EMA']
    
    # Factor 5: 20-day EMA - 50-day EMA
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['50_day_EMA'] = df['close'].ewm(span=50, adjust=False).mean()
    factor_5 = df['20_day_EMA'] - df['50_day_EMA']
    
    # Factor 6: 5-day Volume MA - 10-day Volume MA
    df['5_day_Volume_MA'] = df['volume'].rolling(window=5).mean()
    df['10_day_Volume_MA'] = df['volume'].rolling(window=10).mean()
    factor_6 = df['5_day_Volume_MA'] - df['10_day_Volume_MA']
    
    # Factor 7: 20-day Volume MA - 50-day Volume MA
    df['20_day_Volume_MA'] = df['volume'].rolling(window=20).mean()
    df['50_day_Volume_MA'] = df['volume'].rolling(window=50).mean()
    factor_7 = df['20_day_Volume_MA'] - df['50_day_Volume_MA']
    
    # Factor 8: 5-day VWAP - 10-day VWAP
    df['5_day_VWAP'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['10_day_VWAP'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    factor_8 = df['5_day_VWAP'] - df['10_day_VWAP']
    
    # Factor 9: 20-day VWAP - 50-day VWAP
    df['20_day_VWAP'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['50_day_VWAP'] = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    factor_9 = df['20_day_VWAP'] - df['50_day_VWAP']
