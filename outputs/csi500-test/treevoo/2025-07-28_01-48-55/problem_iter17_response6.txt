```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponentially Weighted Moving Average (EWMA)
    ewma = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - ewma
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume_30
    volume_adjusted_spread = high_low_range * relative_volume
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume Thrust
    upward_thrust = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else 0, axis=1)
    downward_thrust = df.apply(lambda x: x['volume'] if x['close'] <= x['open'] else 0, axis=1)
    volume_thrust = upward_thrust - downward_thrust
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Trend Strength
    abs_diff = df['close'].diff().abs()
    trend_strength = ewma.rolling(window=5).mean() - abs_diff.rolling(window=5).sum()
    
    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum +
        intraday_momentum -
        volume_adjusted_spread +
        volume_thrust -
        trend_strength +
        momentum_factor
    )
    
    return final_alpha_factor
```
