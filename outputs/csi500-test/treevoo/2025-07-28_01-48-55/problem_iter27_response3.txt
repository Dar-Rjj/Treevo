```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume-Adjusted Spread
    average_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume_20
    spread = df['high'] - df['low']
    df['Volume_Adjusted_Spread'] = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor = df['Volume_Adjusted_High_Low_Range'].rolling(window=5).sum()
    
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].diff()
    
    # Calculate Price-Volatility Interaction
    df['Price_Volatility_Interaction'] = df['Daily_Price_Change'] * df['Daily_Volume_Change']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_Spread'] 
        + momentum_factor 
        + df['Price_Volatility_Interaction']
    )
    
    return df['Final_Alpha_Factor']
```
