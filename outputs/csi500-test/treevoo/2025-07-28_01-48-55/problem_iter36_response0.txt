import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving average of daily returns
    df['ma_5_day'] = df['daily_return'].rolling(window=5).mean()
    df['ma_20_day'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate EMA with a 10-day window
    df['ema_10_day'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate WMA with a 15-day window
    weights = np.arange(1, 16)
    df['wma_15_day'] = df['close'].rolling(window=15).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['ema_10_day']
    
    # Calculate Integrated Momentum
    df['integrated_momentum'] = df['close'] - df['wma_15_day'] + df['ma_20_day']
    
    # Intraday range
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Volume Adjusted High-Low Range
    df['vol_adj_high_low'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))

    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['squared_log_return'] = df['log_return'] ** 2

    # Change in volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20_day_ma']

    df['log_return_volume_change'] = df['log_return'] * df['volume_change']
