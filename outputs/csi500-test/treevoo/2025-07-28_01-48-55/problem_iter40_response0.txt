```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day simple moving averages of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Crossover of 5-day and 20-day SMAs as a momentum indicator
    df['Momentum_Signal'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Daily range (High - Low)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Summing up the last 10 days' ranges to identify recent market volatility
    df['Volatility_Signal'] = df['Daily_Range'].rolling(window=10).sum()
    
    # 10-day average volume baseline
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Flagging days when volume exceeds the 10-day average by a certain threshold
    df['Volume_Spike'] = (df['volume'] > df['Avg_Volume_10'] * 1.5).astype(int)
    
    # Close-to-open price and volume relationship
    df['Close_Above_Open_High_Volume'] = ((df['close'] > df['open']) & (df['volume'] > df['Avg_Volume_10'] * 1.5)).astype(int)
    df['Close_Below_Open_High_Volume'] = ((df['close'] < df['open']) & (df['volume'] > df['Avg_Volume_10'] * 1.5)).astype(int)
    
    # Percentage change in volume from one day to the next
    df['Volume_Percent_Change'] = df['volume'].pct_change()
    
    # Alerting when daily volume change surpasses 50% increase from the previous day
    df['Volume_Spike_Percent'] = (df['Volume_Percent_Change'] > 0.5).astype(int)
    
    # Composite score that weights both momentum and volume signals
    df['Composite_Score'] = 0.7 * df['Momentum_Signal'] + 0.3 * df['Volume_Spike']
    
    # Multi-factor model that considers momentum, volatility, and volume simultaneously
    df['Multi_Factor_Signal'] = (df['Momentum_Signal'] > 0) & (df['Volume_Spike_Percent'] > 0) & (df['Volatility_Signal'] > 0)
    
    # Final factor value
    df['Factor_Value'] = df['Composite_Score'] * df['Multi_Factor_Signal']
    
    return df['Factor_Value']
```
