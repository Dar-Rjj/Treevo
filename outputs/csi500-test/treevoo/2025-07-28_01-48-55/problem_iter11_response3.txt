```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Momentum
    base_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Range Contribution
    intraday_range_contribution = (df['high'] - df['low']) / df['low']
    combined_momentum = base_momentum + intraday_range_contribution
    
    # Adjust for Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = combined_momentum * avg_volume_20_days / df['volume']
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_moving_average = df['volume'].rolling(window=5).mean()
    combined_intraday_volume_trend = intraday_reversal * daily_volume_change / volume_moving_average
    
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Final Alpha Factor Components
    price_momentum = df['close'] - wma
    volume_adjusted_high_low_range = (df['high'] - df['low']) * np.log(df['volume'])
    rolling_momentum = volume_adjusted_high_low_range.rolling(window=10).sum()
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        price_momentum +
        volume_adjusted_momentum +
        combined_intraday_volume_trend -
        volume_adjusted_high_low_range +
        rolling_momentum
    )
    
    return final_alpha_factor
```
