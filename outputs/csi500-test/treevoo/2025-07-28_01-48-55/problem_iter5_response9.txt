```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Combined Price Momentum
    df['combined_momentum'] = df['close'] - df['close'].rolling(window=20).mean()
    
    # Adjust by Volume and Intraday Volatility
    df['average_volume'] = df['volume'].rolling(window=20).sum() / 20
    df['intraday_volatility'] = (df['high'] - df['low']) / df['low']
    df['adjusted_momentum'] = (df['combined_momentum'] * (df['average_volume'] / df['volume'])) / df['intraday_volatility']
    
    # Introduce Price Volatility and Volume Thrust Adjustment
    df['price_volatility'] = df['close'].rolling(window=20).std()
    df['upward_thrust'] = 0
    df['downward_thrust'] = 0
    
    for i in range(1, len(df)):
        if df.iloc[i]['close'] > df.iloc[i]['open']:
            df.loc[df.index[i], 'upward_thrust'] = df.iloc[i]['volume']
        else:
            df.loc[df.index[i], 'downward_thrust'] = df.iloc[i]['volume']
            
    df['volume_thrust'] = df['upward_thrust'] - df['downward_thrust']
    
    df['final_factor'] = (df['adjusted_momentum'] / df['price_volatility']) + df['volume_thrust']
    
    return df['final_factor']
```
