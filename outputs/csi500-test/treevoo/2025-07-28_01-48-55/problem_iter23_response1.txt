```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    df['WMA'] = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA']

    # Calculate Volume-Adjusted Spread
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    spread = df['high'] - df['low']
    df['Volume_Adjusted_Spread'] = spread * relative_volume

    # Calculate Momentum Factor
    df['Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=5).sum()

    # Calculate Price-Volatility Interaction
    daily_price_change = df['close'].diff()
    daily_volume_change = df['volume'].diff()
    df['Price_Volatility_Interaction'] = daily_price_change * daily_volume_change

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_Spread'] 
        + df['Momentum_Factor'] 
        + df['Price_Volatility_Interaction']
    )

    return df['Final_Alpha_Factor']
```
