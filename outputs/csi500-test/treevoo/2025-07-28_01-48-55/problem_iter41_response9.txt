```python
import pandas as pd

def heuristics_v2(df):
    # Compute the difference between today's close and today's open
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of today's close to today's high
    close_high_ratio = df['close'] / df['high']
    
    # Analyze the gap from today's close to today's low
    close_low_gap = df['close'] - df['low']
    
    # Relate changes in closing price with percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Measure the correlation between daily returns and volume
    daily_returns = df['close'].pct_change()
    return_volume_corr = daily_returns.rolling(window=20).corr(df['volume'])
    
    # Compare the current day’s amount traded with the previous day’s
    amount_change = df['amount'].pct_change()
    
    # Establish a factor based on the moving average of the amount traded relative to the closing price
    amount_moving_avg = df['amount'].rolling(window=20).mean() / df['close']
    
    # Combine the difference between today's close and open with the percentage change in volume
    composite_factor_1 = (close_open_diff + volume_change) / 2
    
    # Create a composite factor using the ratio of today's close to high and the correlation between returns and volume
    composite_factor_2 = (close_high_ratio + return_volume_corr) / 2
    
    # Develop a factor that incorporates the gap from today's close to low and the moving average of the amount traded
    composite_factor_3 = (close_low_gap + amount_moving_avg) / 2
    
    # Final alpha factor combining all the composite factors
    alpha_factor = (composite_factor_1 + composite_factor_2 + composite_factor_3) / 3
    
    return alpha_factor
```
