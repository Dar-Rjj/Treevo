```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['Close_Mean'] = df['close'].rolling(window=20).mean()
    df['Volume_Avg_20_Days'] = df['volume'].rolling(window=20).sum() / 20
    df['Adjusted_Price_Momentum'] = (df['close'] - df['Close_Mean']) * df['Volume_Avg_20_Days'] / df['volume']

    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Measure Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Calculate Volume Moving Average
    df['Volume_Moving_Average_5'] = df['volume'].rolling(window=5).mean()

    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    constant_factor = 1
    df['Factor_Value'] = (
        (constant_factor * df['Adjusted_Price_Momentum']) + 
        (df['Intraday_Reversal'] * df['Daily_Volume_Change'] / df['Volume_Moving_Average_5'])
    )

    # Drop intermediate columns
    df.drop(columns=['Close_Mean', 'Volume_Avg_20_Days', 'Adjusted_Price_Momentum', 'Intraday_Reversal',
                     'Daily_Volume_Change', 'Volume_Moving_Average_5'], inplace=True)

    return df['Factor_Value']
```
