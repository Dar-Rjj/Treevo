```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['high'] - df['low']
    intraday_reversal_adj = intraday_reversal - (df['open'] - df['close'])

    # Measure Volume Trend
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_ma = df['volume'].rolling(window=5).mean()

    # Combine Intraday Reversal and Volume Trend
    combined_factor = (intraday_reversal_adj * volume_change) / volume_ma

    return combined_factor
```
