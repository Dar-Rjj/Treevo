```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def wma(values, n):
        weights = np.arange(1, n+1)
        return values.rolling(window=n).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    close_10_wma = wma(df['close'], 10)

    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Calculate Volume Adjusted High-Low Range
    log_volume = np.log(df['volume'])
    volume_adjusted_high_low_range = high_low_range * log_volume

    # Calculate Price Momentum
    price_momentum = df['close'] - close_10_wma

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    average_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume_20
    volume_adjusted_spread = spread * relative_volume

    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()

    # Calculate Enhanced Momentum
    enhanced_momentum = price_momentum + momentum_factor

    # Final Alpha Factor
    final_alpha_factor = enhanced_momentum - volume_adjusted_spread

    return final_alpha_factor
```
