import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['Close_20_median'] = df['close'].rolling(window=20).median()
    df['Volume_20_avg'] = df['volume'].rolling(window=20).mean()
    df['Adjusted_Price_Momentum'] = (df['close'] - df['Close_20_median']) * (df['Volume_20_avg'] / df['volume'])

    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Measure Daily Volume Change
    df['Volume_Change'] = (df['volume'] / df['volume'].shift(1)) - 1

    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    df['Adjusted_Momentum_Weighted'] = df['Adjusted_Price_Momentum'] * 1.5
    df['Reversal_Volume_Adjusted'] = df['Intraday_Reversal'] * df['Volume_Change']
    df['Final_Factor'] = df['Adjusted_Momentum_Weighted'] + df['Reversal_Volume_Adjusted']

    return df['Final_Factor']
