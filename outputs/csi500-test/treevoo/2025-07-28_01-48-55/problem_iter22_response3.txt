import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights
    wma_weights = np.arange(1, 11)
    df['WMA'] = df['close'].rolling(window=10).apply(lambda x: np.average(x, weights=wma_weights), raw=True)
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA']
    
    # Calculate Spread
    df['Spread'] = df['high'] - df['low']
    
    # Calculate Relative Volume
    df['Avg_Volume_20d'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Avg_Volume_20d']
    
    # Calculate Volume Adjusted Spread
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Calculate Momentum Factor using Volume Adjusted High-Low Range
    df['Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=10).sum()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] 
                                 - df['Volume_Adjusted_Spread'] 
                                 + df['Momentum_Factor'])
    
    return df['Final_Alpha_Factor']
