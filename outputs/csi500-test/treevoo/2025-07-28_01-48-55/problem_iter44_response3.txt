```python
def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, defined as typical price * volume
    money_flow = typical_price * df['volume']
    
    # Calculate the money flow ratio, defined as today's money flow / yesterday's money flow
    money_flow_ratio = money_flow / money_flow.shift(1)
    
    # Calculate the price momentum, defined as today's close / yesterday's close
    price_momentum = df['close'] / df['close'].shift(1)
    
    # Calculate the alpha factor, defined as the product of money flow ratio and price momentum
    alpha_factor = money_flow_ratio * price_momentum
    
    return alpha_factor
```
