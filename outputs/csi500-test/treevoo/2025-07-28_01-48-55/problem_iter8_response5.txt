import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price and Intraday Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    df['adjusted_momentum'] = df['price_momentum'] + df['intraday_momentum']
    
    # Calculate Average Volume and Amount over 20 Days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['avg_amount_20'] = df['amount'].rolling(window=20).mean()
    
    # Adjust Momentum by Average Volume and Amount
    df['momentum_adjusted'] = (df['adjusted_momentum'] * df['avg_volume_20'] * df['avg_amount_20']) / (df['volume'] * df['amount'])
    
    # Calculate High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['vol_adj_high_low_range'] = df['high_low_range'] * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['vol_adj_high_low_range_20'] = df['vol_adj_high_low_range'].rolling(window=20).mean()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return
    df['log_return_20'] = df['log_return'].rolling(window=20).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['integrated_log_return'] = df['log_return'] * df['volume_change']
    df['integrated_log_return_10'] = df['integrated_log_return'].rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, np.where(df['ema_5'] < df['ema_20'], -1, 0))
    
    # Combine All Factors
    df['alpha_factor'] = (df['vol_adj_high_low_range_20'] - df['log_return_20']) + df['crossover_signal'] + df['momentum_adjusted']
    
    return df['alpha_factor']
