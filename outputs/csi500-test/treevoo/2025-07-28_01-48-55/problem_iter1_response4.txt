```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Average Volume over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Adjust Momentum by Average Volume and Current Volume
    adjusted_momentum = (price_momentum * avg_volume_20) / df['volume']
    
    return adjusted_momentum
```
