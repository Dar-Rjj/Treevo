```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving averages of daily returns
    df['5_day_momentum'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_momentum'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) with a 10-day window
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) with a 15-day window
    weights = np.arange(1, 16)
    df['WMA_15'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Integrate EMAs into momentum calculations
    df['price_momentum'] = df['close'] - df['EMA_10']
    df['integrated_momentum'] = df['close'] - df['WMA_15'] + df['20_day_momentum']
    
    # Determine the intraday range and its ratio to previous day's close
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Measure the upper and lower shadow lengths and their ratio
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Calculate Volume Adjusted High-Low Range
    df['volume_adjusted_range'] = df['intraday_range'] * np.sqrt(np.log(df['volume']))
    
    # Calculate 10-day Historical Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['10_day_volatility'] = df['log_returns'].rolling(window=10).std() * np.sqrt(252)
    
    # Assess the change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_change_flag'] = (df['volume_change'] > 0).astype(int)
    
    # Compute the ratio of volume to the 20-day moving average of volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate log return with volume change
    df['log_return_with_volume'] = (df['log_returns'] * df['volume_change']).rolling(window=10).sum()
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = df.apply(lambda x: x['volume'] * (x['close'] - x['open']) if x['close'] > x['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda x: x['volume'] * (x['open'] - x['close']) if x['close'] <= x['open'] else 0, axis=1)
    df['volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Enhance with moving average crossover signals
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = (df['EMA_5'] > df['EMA_20']).astype(int) - (df['EMA_5'] < df['EMA_20']).astype(int)
    
    # Construct final composite alpha factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['volume_adjusted_range'] -
        df['volume_ratio'] +
        df['log_return_with_volume'] +
        df['crossover_signal'] +
        df['volume_thrust'] +
        df['10_day_volatility'] * df['volume_thrust']
    )
    
    return df['alpha_factor']
```
