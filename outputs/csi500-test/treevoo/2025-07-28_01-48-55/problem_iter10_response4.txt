```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def calculate_wma(prices, window=10):
        weights = np.arange(1, window + 1)
        wma = prices.rolling(window=window).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
        return wma

    wma = calculate_wma(df['close'], window=10)

    # Calculate Price Momentum
    price_momentum = df['close'] - wma

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    volume_adjusted_spread = spread * relative_volume

    # Final Alpha Factor
    alpha_factor = price_momentum - volume_adjusted_spread

    return alpha_factor
```
