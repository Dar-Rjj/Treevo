```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['high'] - df['low']
    
    # Adjust for Opening Range
    adjusted_intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Price Momentum Factor
    sma_10 = df['close'].rolling(window=10).mean()
    price_momentum = df['close'] - sma_10
    
    # Volume-Adjusted Daily Range
    daily_range = df['high'] - df['low']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_daily_range = daily_range * (df['volume'] / avg_volume_10)
    
    # Range Volatility Measurement
    avg_daily_range_10 = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Synthesized Alpha Factor
    synthesized_alpha_factor = (
        (adjusted_intraday_reversal * volume_trend) + 
        price_momentum - 
        volume_adjusted_daily_range - 
        avg_daily_range_10
    )
    
    return synthesized_alpha_factor
```
