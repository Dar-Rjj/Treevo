```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['EMA'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume Adjusted Spread
    average_volume = df['volume'].rolling(window=15).mean()
    relative_volume = df['volume'] / average_volume
    spread = df['high'] - df['low']
    df['Volume_Adjusted_Spread'] = spread * relative_volume
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume Thrust
    df['Upward_Thrust'] = 0
    df['Downward_Thrust'] = 0
    df.loc[df['close'] > df['open'], 'Upward_Thrust'] = df['volume']
    df.loc[df['close'] <= df['open'], 'Downward_Thrust'] = df['volume']
    df['Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] + df['Intraday_Momentum'] - 
                                df['Volume_Adjusted_Spread'] + df['Volume_Thrust'])
    
    return df['Final_Alpha_Factor']
```
