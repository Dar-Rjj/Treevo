import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'].diff()
    
    # Calculate the daily volume change
    df['volume_change'] = df['volume'].diff()
    
    # Calculate the daily amount change
    df['amount_change'] = df['amount'].diff()
    
    # Calculate the rolling average of the price change over 5 days
    df['avg_price_change_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate the rolling sum of the volume change over 5 days
    df['sum_volume_change_5d'] = df['volume_change'].rolling(window=5).sum()
    
    # Calculate the rolling sum of the amount change over 5 days
    df['sum_amount_change_5d'] = df['amount_change'].rolling(window=5).sum()
    
    # Alpha factor: the ratio of the rolling sum of the volume change to the rolling sum of the amount change
    df['alpha_factor'] = df['sum_volume_change_5d'] / (df['sum_amount_change_5d'] + 1e-6) * df['avg_price_change_5d']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
