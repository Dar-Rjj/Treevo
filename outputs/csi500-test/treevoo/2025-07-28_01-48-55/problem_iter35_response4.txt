```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving Averages of Daily Returns
    df['5d_ma_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['20d_ma_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) on Close Prices
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Weighted Moving Average (WMA) on Close Prices
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: np.dot(x, weights)/weights.sum(), raw=True)
    
    # Price Momentum and Integrated Momentum
    df['price_momentum'] = df['close'] - df['ema_10']
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20d_ma_daily_return']
    
    # Intraday Range and Ratios
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Volume Adjusted High-Low Range
    df['vol_adj_high_low'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    # Historical Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['hist_volatility'] = log_returns.rolling(window=10).std() * np.sqrt(252)
    
    # Volume Change and Ratio
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Accumulate Product of Daily Log Return and Volume Change
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['accumulated_product'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['downward_thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Crossover Signal
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, np.where(df['ema_5'] < df['ema_20'], -1, 0))
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] +
        -df['vol_adj_high_low'] +
        -df['accumulated_product'] +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] +
        df['hist_volatility'] * df['enhanced_volume_thrust']
    )
    
    return df['alpha_factor']
```
