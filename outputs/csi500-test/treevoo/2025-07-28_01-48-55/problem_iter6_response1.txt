```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights over a 10-day window
    wma_weights = np.arange(1, 11)
    df['WMA'] = df['close'].rolling(window=10).apply(lambda prices: np.dot(prices, wma_weights)/wma_weights.sum(), raw=True)

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']

    # Calculate Spread
    df['Spread'] = df['high'] - df['low']

    # Calculate Relative Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['RelativeVolume'] = df['volume'] / avg_volume_20

    # Calculate Volume Adjusted Spread
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']

    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['PriceMomentum'] - df['VolumeAdjustedSpread']
    
    return df['FinalAlphaFactor']
```
