import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum: Close - Open
    intraday_momentum = df['close'] - df['open']
    
    # Volume Acceleration: Today’s Volume - Yesterday’s Volume
    volume_acceleration = df['volume'] - df['volume'].shift(1)
    
    # Price-Volume Interaction: Intraday Momentum * Volume Acceleration
    price_volume_interaction = intraday_momentum * volume_acceleration
    
    # Smooth: Moving Average of Price-Volume Interaction (5 days)
    smoothed_price_volume = price_volume_interaction.rolling(window=5).mean()
    
    # Enhanced Composite Factor: Smoothed Price-Volume Interaction + (Today’s Volume / Yesterday’s Volume)
    enhanced_composite = smoothed_price_volume + (df['volume'] / df['volume'].shift(1))
    
    return enhanced_composite
