```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['high'] - df['low']
    adjusted_intraday_reversal = intraday_reversal - (df['open'] - df['close'])
    
    # Measure Price Momentum
    price_momentum = df['close'] - df['open']
    
    # Combine Intraday Reversal and Price Momentum
    combined_momentum = adjusted_intraday_reversal + price_momentum
    
    # Calculate Volume Trend
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_moving_average = df['volume'].rolling(window=5).mean()
    
    # Adjust Combined Momentum by Volume
    adjusted_momentum = (combined_momentum * daily_volume_change) / volume_moving_average
    
    return adjusted_momentum
```
