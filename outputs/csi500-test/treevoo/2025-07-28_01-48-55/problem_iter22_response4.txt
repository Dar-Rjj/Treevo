```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    N = 10
    smoothing_factor = 2 / (N + 1)
    df['EMA'] = df['close'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']

    # Calculate Volume Adjusted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.sqrt(np.log(df['volume']))

    # Calculate Enhanced Momentum Factor
    weights = np.linspace(1, 5, 5)  # Linearly increasing weights
    df['Enhanced_Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=5).apply(
        lambda x: np.sum(weights * x), raw=True)

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] 
                                - df['Volume_Adjusted_High_Low_Range'] 
                                + df['Enhanced_Momentum_Factor'])

    return df['Final_Alpha_Factor'].dropna()
```
