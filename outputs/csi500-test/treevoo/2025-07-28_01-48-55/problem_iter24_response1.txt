```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the day's range
    df['day_range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['day_range']
    
    # Calculate the average transaction size
    df['avg_transaction_size'] = df['amount'] / df['volume']
    
    # Create a factor by combining the close position and average transaction size
    df['factor'] = df['close_position'] * df['avg_transaction_size']
    
    return df['factor']
```
