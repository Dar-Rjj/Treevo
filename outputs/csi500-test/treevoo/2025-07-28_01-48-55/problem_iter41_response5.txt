```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving averages of daily returns
    df['ma5_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['ma20_daily_return'] = df['daily_return'].rolling(window=20).mean()

    # Calculate Exponential Moving Average (EMA) over a 10-day window
    n_ema = 10
    k_ema = 2 / (n_ema + 1)
    df['ema10'] = df['close'].ewm(alpha=k_ema, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) over a 15-day window
    weights = np.arange(1, 16)
    wma_weights = weights / weights.sum()
    df['wma15'] = df['close'].rolling(window=15).apply(lambda prices: (prices * wma_weights).sum(), raw=True)
    
    # Integrate EMAs into momentum calculations
    df['price_momentum'] = df['close'] - df['ema10']
    df['integrated_momentum'] = df['close'] - df['wma15'] + df['ma20_daily_return']
    
    # Intraday range and shadow lengths
    df['intraday_range_ratio'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 10-day Historical Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['squared_log_returns'] = df['log_returns'] ** 2
    df['volatility_10'] = df['squared_log_returns'].rolling(window=10).mean().pow(1/2)
    
    # Volume and its interaction with price
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_to_avg_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['log_return_with_volume_change'] = df['daily_log_return'] * df['volume_change']
    df['accumulated_product'] = df['log_return_with_volume_change'].rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = (df['volume'] * (df['close'] - df['open'])).where(df['close'] > df['open'], 0)
    df['downward_thrust'] = (df['volume'] * (df['open'] - df['close'])).where(df['close'] <= df['open'], 0)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Moving average crossover signals
    df['ema5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema5'] > df['ema20'], 1, -1)
    
    # Final composite alpha factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['volume_to_avg_volume'] * (df['high'] - df['low']) / df['open'] -
        df['shadow_ratio'] +
        df['accumulated_product'] +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] * df['volatility_10'] -
        df['daily_log_return'].rolling(window=20).mean()
    )

    return df['alpha_factor']
```
