```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with exponential weights
    df['WMA'] = df['close'].ewm(span=15).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Integrated_Momentum'] = (df['Daily_Log_Return'] * df['Volume_Change']).rolling(window=15).sum()
    
    # Enhance with Moving Average Crossover
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_EMA'] = df['close'].ewm(span=20).mean()
    crossover_signal = np.where(df['5_day_SMA'] > df['20_day_EMA'], 1, 
                                np.where(df['5_day_SMA'] < df['20_day_EMA'], -1, 0))
    df['Enhanced_Crossover_Signal'] = df['Integrated_Momentum'] + (crossover_signal * 0.05)  # Example positive value
    
    # Incorporate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Integrated_Momentum_HL'] = df['Integrated_Momentum'] * df['High_Low_Range']
    
    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['Upward_Thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['Downward_Thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust'] + (df['high'] - df['low']) / df['open']
    
    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_HL_Range'] = df['High_Low_Range'] * np.sqrt(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    df['Rolling_Volume_Adjusted_HL_Range'] = df['Volume_Adjusted_HL_Range'].rolling(window=25).mean()
    
    # Rolling Average of Daily Log Return
    df['Rolling_Daily_Log_Return'] = df['Daily_Log_Return'].rolling(window=25).mean()
    
    # Final Integration
    df['Final_Integration'] = (df['Integrated_Momentum_HL'] + df['Intraday_Momentum'] + 
                               df['Volume_Thrust'] + 
                               (df['Rolling_Volume_Adjusted_HL_Range'] - df['Rolling_Daily_Log_Return']) + 
                               df['Enhanced_Crossover_Signal'])
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Average_Volume'] = df['volume'].rolling(window=25).mean()
    df['Relative_Volume'] = df['volume'] / df['Average_Volume']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] + df['Final_Integration'] - df['Volume_Adjusted_Spread'])
    
    return df['Final_Alpha_Factor']
```
