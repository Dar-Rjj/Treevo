```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily returns
    df['daily_return'] = df['close'].diff()
    
    # Open-Close Ratio Factor
    df['open_close_ratio'] = df['close'] / df['open']
    
    # Daily Price Range
    df['daily_price_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine Daily Return and Volume
    df['return_volume'] = df['daily_return'] * df['volume']
    
    # Examine Volume Change Over Time
    df['volume_change'] = df['volume'].diff()
    df['volume_change_return'] = df['volume_change'] * df['daily_return']
    
    # Calculate Exponential Moving Average (EMA)
    df['ema_15'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Simple Moving Averages (SMA)
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Crossover of 5-day and 30-day SMAs
    df['sma_crossover'] = df['sma_5'] > df['sma_30']
    
    # Volume-Weighted High-Low Range
    df['vwhl_range'] = (df['high'] - df['low']) * df['volume']
    
    # Sum Last 7 Days' Ranges
    df['sum_7_days_range'] = df['daily_price_range'].rolling(window=7).sum()
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].join(df['open'].shift(1)).apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['open']), abs(x['low'] - x['open'])), axis=1
    )
    
    # Calculate Average True Range (ATR)
    df['atr_10'] = df['true_range'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['ema_15']
    
    # Volume-Adjusted True Range
    df['vatr'] = df['true_range'] * np.log(df['volume'])
    
    # Momentum Factor
    df['momentum_factor'] = df['vwhl_range'].rolling(window=15).mean()
    
    # Analyze Volume for Market Strength
    df['avg_vol_7'] = df['volume'].rolling(window=7).mean()
    df['vol_strength'] = df['volume'] > df['avg_vol_7'] * 1.2
    
    # Evaluate Buying or Selling Pressure
    df['buying_pressure'] = (df['close'] > df['open']) & (df['volume'] > df['avg_vol_7'])
    df['selling_pressure'] = (df['close'] < df['open']) & (df['volume'] > df['avg_vol_7'])
    
    # Measure Volume Spikes
    df['volume_spike'] = df['volume_change'] > 0.4 * df['volume'].shift(1)
    
    # Integrate Price and Volume Indicators
    df['multi_factor_signal'] = 0.5 * df['price_momentum'] + 0.3 * df['momentum_factor'] + 0.2 * df['sma_crossover']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] 
        - df['vatr'] 
        + df['momentum_factor'] 
        + df['sma_crossover'] * 1.0 
        + df['volume_spike'] * 0.5 
        - df['atr_10']
    )
    
    return df['alpha_factor']
```
