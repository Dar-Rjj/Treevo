```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def wma(series, window=10):
        weights = np.arange(1, window+1)
        return series.rolling(window=window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    df['WMA'] = wma(df['close'])

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']

    # Calculate Daily Log Return
    df['LogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)

    # Integrate Log Return with Volume Change
    df['IntegratedMomentum'] = (df['LogReturn'] * df['VolumeChange']).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    df['EMA5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA20'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = np.where(df['EMA5'] > df['EMA20'], 1, np.where(df['EMA5'] < df['EMA20'], -1, 0))
    df['CrossoverSignal'] = crossover_signal
    df['IntegratedMomentum'] += df['CrossoverSignal'] * 0.1  # Adjust the positive value as needed

    # Incorporate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    df['NormalizedHighLowRange'] = df['HighLowRange'] / df['close']
    df['IntegratedMomentum'] *= df['NormalizedHighLowRange']

    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    df['UpwardThrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    df['DownwardThrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    df['PriceRangeFactor'] = (df['high'] - df['low']) / df['open']
    df['EnhancedVolumeThrust'] = (df['UpwardThrust'] - df['DownwardThrust']) + df['PriceRangeFactor']

    # Final Integration
    df['FinalFactor'] = df['IntegratedMomentum'] + df['IntradayMomentum'] + df['EnhancedVolumeThrust']

    # Add Price Momentum
    df['FinalFactor'] += df['PriceMomentum']

    # Subtract Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['AverageVolume'] = df['volume'].rolling(window=20).mean()
    df['RelativeVolume'] = df['volume'] / df['AverageVolume']
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']
    df['FinalFactor'] -= df['VolumeAdjustedSpread']

    return df['FinalFactor']
```
