```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_returns'] = df['close'].pct_change()
    
    # 20-day moving average of daily returns
    df['ma_20_returns'] = df['daily_returns'].rolling(window=20).mean()
    
    # 10-day Weighted Moving Average (WMA) of close prices
    weights = np.arange(1, 11)
    df['wma_10_close'] = df['close'].rolling(window=10).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Price Momentum
    df['price_momentum'] = df['close'] - df['wma_10_close']
    
    # Intraday range and ratio
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Change in volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume to 20-day moving average of volume ratio
    df['ma_20_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['ma_20_volume']
    
    # Composite factor
    df['composite_factor'] = df['ma_20_returns'] + df['volume_ratio']
    
    # Correlation between 20-day moving average of daily returns and 20-day moving average of volume
    df['correlation_returns_volume'] = df['daily_returns'].rolling(window=20).corr(df['volume'])
    
    # High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * np.log(df['volume'])
    
    # Relative Volume
    df['relative_volume'] = df['volume'] / df['ma_20_volume']
    
    # Volume Adjusted Spread
    df['volume_adjusted_spread'] = df['volume_adjusted_high_low_range'] * df['relative_volume']
    
    # Momentum Factor
    df['momentum_factor'] = df['high_low_range'].rolling(window=20).sum()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['price_momentum'] - df['volume_adjusted_spread'] + df['momentum_factor']
    
    return df['final_alpha_factor']
```
