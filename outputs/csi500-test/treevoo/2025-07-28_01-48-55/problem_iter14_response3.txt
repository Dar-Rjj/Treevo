```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_range = high_low_range * np.sqrt(df['volume'])
    
    # Calculate Weighted Moving Average (WMA) over a 10-day window
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - wma
    
    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_range.rolling(window=10).sum()
    
    # Calculate Weighted Price Change
    weighted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Range Volatility
    daily_range = df['high'] - df['low']
    range_volatility = daily_range.rolling(window=10).mean()
    
    # Calculate Price Oscillator
    sma_5 = df['close'].rolling(window=5).mean()
    price_oscillator = wma - sma_5
    
    # Final Alpha Factor
    final_alpha_factor = (price_momentum 
                          - volume_adjusted_spread 
                          + momentum_factor 
                          + weighted_price_change 
                          - range_volatility 
                          + price_oscillator)
    
    return final_alpha_factor
```
