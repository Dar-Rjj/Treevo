```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['EMA_15'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Identify Crossover of 5-day and 20-day SMAs
    df['SMA_Crossover'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    df['Vol_Weighted_HL'] = (df['high'] - df['low']) * np.sqrt(df['volume'])
    
    # Calculate Daily Range and Sum Last 10 Days' Ranges
    df['Daily_Range'] = df['high'] - df['low']
    df['Sum_Daily_Range_10'] = df['Daily_Range'].rolling(window=10).sum()
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'open']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1
    )
    
    # Calculate Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['open'] - df['EMA_15']
    
    # Calculate Volume-Adjusted True Range
    df['Volume_Adjusted_TR'] = df['True_Range'] * (np.log(df['volume']))**2
    
    # Calculate Momentum Factor
    df['Momentum_Factor'] = df['Vol_Weighted_HL'].rolling(window=20).mean()
    
    # Analyze Volume for Market Strength
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Avg_Volume_10']).astype(int)
    
    # Evaluate Buying or Selling Pressure
    df['High_Volume_Close_Above_Open'] = ((df['close'] > df['open']) & (df['volume'] > 1.5 * df['Avg_Volume_10'])).astype(int)
    df['High_Volume_Close_Below_Open'] = ((df['close'] < df['open']) & (df['volume'] > 1.5 * df['Avg_Volume_10'])).astype(int)
    
    # Measure Volume Spikes
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    df['Volume_Spike_Flag'] = (df['Daily_Volume_Change'] > 0.5).astype(int)
    
    # Integrate Price and Volume Indicators
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] 
                                - df['Volume_Adjusted_TR'] 
                                + df['Momentum_Factor'] 
                                + df['SMA_Crossover']
                                + df['Volume_Spike_Flag'] 
                                - df['ATR_14'])
    
    return df['Final_Alpha_Factor']
```
