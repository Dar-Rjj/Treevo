```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Compute 5-day and 20-day moving average of daily returns
    df['5D_MA_Return'] = df['Daily_Return'].rolling(window=5).mean()
    df['20D_MA_Return'] = df['Daily_Return'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) using Close prices over a 10-day window
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Weighted Moving Average (WMA) using Close prices over a 15-day window
    weights = np.arange(1, 16)
    df['WMA_15'] = df['close'].rolling(window=15).apply(lambda prices: np.dot(prices, weights)/weights.sum(), raw=True)
    
    # Integrated Momentum
    df['Integrated_Momentum'] = df['close'] - df['WMA_15'] + df['20D_MA_Return']
    
    # Intraday range
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_Ratio'] = df['Intraday_Range'] / df['close'].shift(1)
    
    # Upper and Lower Shadow Lengths
    df['Upper_Shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['Lower_Shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['Shadow_Ratio'] = df['Upper_Shadow'] / df['Lower_Shadow'].replace(0, np.nan)
    
    # 10-day Historical Volatility
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['10D_HV'] = df['Log_Return'].rolling(window=10).std() * np.sqrt(252)
    
    # Volume Analysis
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Vol_20D_MA'] = df['volume'].rolling(window=20).mean()
    df['Vol_to_Vol_20D_MA'] = df['volume'] / df['Vol_20D_MA']
    
    # Integrated Log Return with Volume Change
    df['Integrated_Log_Return'] = (df['Log_Return'] * df['Volume_Change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['Upward_Thrust'] = df.apply(lambda row: row['volume'] * (row['close'] - row['open']) if row['close'] > row['open'] else 0, axis=1)
    df['Downward_Thrust'] = df.apply(lambda row: row['volume'] * (row['open'] - row['close']) if row['close'] <= row['open'] else 0, axis=1)
    df['Enhanced_Volume_Thrust'] = df['Upward_Thrust'] - df['Downward_Thrust'] + (df['high'] - df['low']) / df['open']
    
    # 5-day and 20-day EMA Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Crossover_Signal'] = (df['EMA_5'] - df['EMA_20']).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Final Composite Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] +
        df['Integrated_Momentum'] -
        df['Intraday_Range_Ratio'] -
        df['10D_HV'] +
        df['Integrated_Log_Return'] +
        df['Crossover_Signal'] +
        df['Enhanced_Volume_Thrust'] +
        (df['10D_HV'] * df['Enhanced_Volume_Thrust']) -
        df['Log_Return'].rolling(window=20).mean() +
        (df['Upper_Shadow'] - df['Lower_Shadow'])
    )
    
    return df['Final_Alpha_Factor']
```
