```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Momentum
    df['Base_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Range Contribution
    df['Intraday_Range_Contribution'] = (df['high'] - df['low']) / df['low']
    
    # Adjust for Volume
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Combined_Momentum'] = (df['Base_Momentum'] + df['Intraday_Range_Contribution']) * df['Avg_Volume_20'] / df['volume']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Trend
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    
    # Combine Intraday Reversal and Volume Trend
    df['Intraday_Reversal_Volume_Trend'] = (df['Intraday_Reversal'] * df['Volume_Change']) / df['Volume_MA_5']
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_Close_10'] = df['close'].rolling(window=10).mean()
    
    # Final Alpha Factor Components
    df['Price_Momentum'] = df['close'] - df['SMA_Close_10']
    
    df['Volume_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * np.log(df['volume'])
    
    df['Rolling_Momentum'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=20).sum()
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] +
        df['Combined_Momentum'] +
        df['Intraday_Reversal_Volume_Trend'] -
        df['Volume_Adjusted_High_Low_Range'] +
        df['Rolling_Momentum']
    )
    
    return df['Final_Alpha_Factor']
```
