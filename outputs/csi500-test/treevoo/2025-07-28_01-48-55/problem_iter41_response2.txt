```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum from Close Prices
    df['10_day_momentum'] = df['close'].pct_change(10)
    df['20_day_momentum'] = df['close'].pct_change(20)
    df['momentum_diff'] = df['10_day_momentum'] - df['20_day_momentum']
    
    # Analyze High-Low Price Spread
    df['daily_range'] = df['high'] - df['low']
    df['30_day_avg_range'] = df['daily_range'].rolling(window=30).mean()
    df['range_above_avg'] = df['daily_range'] > df['30_day_avg_range']
    
    # Evaluate Open-Close Price Dynamics
    df['positive_day'] = df['close'] > df['open']
    df['5_day_positive_count'] = df['positive_day'].rolling(window=5).sum()
    df['positive_ratio'] = df['5_day_positive_count'] / 5
    
    # Assess Volume and Price Correlation
    df['VWAP'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    df['vwap_volume_corr'] = df['VWAP'].rolling(window=10).corr(df['volume'])
    df['positive_corr'] = df['vwap_volume_corr'] > 0
    
    # Investigate Amount and Volume Trends
    df['amount'] = df['close'] * df['volume']
    df['10_day_amount_ma'] = df['amount'].rolling(window=10).mean()
    df['amount_above_ma'] = df['amount'] > df['10_day_amount_ma']
    
    # Combine all the subthoughts into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_diff'] +
        (df['range_above_avg'] * 2 - 1) +
        df['positive_ratio'] +
        (df['positive_corr'] * 2 - 1) +
        (df['amount_above_ma'] * 2 - 1)
    )
    
    return df['alpha_factor']
```
