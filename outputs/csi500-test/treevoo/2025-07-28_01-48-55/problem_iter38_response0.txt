import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving average of daily returns
    df['5_day_ma_return'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_ma_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) with a 10-day window
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) with a 15-day window
    weights = np.arange(1, 16)
    df['wma_15'] = (df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True))
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Calculate Integrated Momentum
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20_day_ma_return']
    
    # Determine the intraday range
    df['intraday_range'] = df['high'] - df['low']
    df['range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Measure the upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Calculate Volume Adjusted High-Low Range
    df['volume_adjusted_high_low'] = df['intraday_range'] * np.sqrt(np.log(df['volume']))
    
    # Calculate 10-day Historical Volatility
    df['log_ret'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_ret'].rolling(window=10).std() * np.sqrt(252)
    
    # Assess the change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_change_sign'] = np.sign(df['volume_change'])
    
    # Compute the ratio of volume to the 20-day moving average of volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate log return with volume change
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['integrated_log_return'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = df.apply(lambda x: x['volume'] * (x['close'] - x['open']) if x['close'] > x['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda x: x['volume'] * (x['open'] - x['close']) if x['close'] <= x['open'] else 0, axis=1)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Calculate 5-day and 20-day Exponential Moving Averages
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()

    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, np.where(df['ema_5'] < df['ema_20'], -1, 0))
