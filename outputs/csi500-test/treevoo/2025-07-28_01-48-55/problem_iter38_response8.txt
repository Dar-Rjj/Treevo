```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close price to open price, which can capture the overall daily trend
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of high price to low price, which measures the volatility within a day
    high_to_low_ratio = df['high'] / df['low']
    
    # Calculate the ratio of volume to amount, which can indicate the average transaction size
    volume_to_amount_ratio = df['volume'] / df['amount']

    # Combine the three ratios into a single factor
    # The first term captures the overall trend, the second term captures the intra-day volatility,
    # and the third term adjusts for the average transaction size.
    factor = (close_to_open_ratio * high_to_low_ratio) / volume_to_amount_ratio
    
    return factor
```
