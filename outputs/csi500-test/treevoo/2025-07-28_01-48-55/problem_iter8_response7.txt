```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Change
    volume_change = df['volume'].diff()
    
    # Combine Intraday Reversal and Volume Change
    intraday_volume_adjusted = intraday_reversal * volume_change
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low = high_low_range * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range
    rolling_avg_high_low = volume_adjusted_high_low.rolling(window=20).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff()
    
    # Rolling Average of Daily Log Return
    rolling_avg_log_return = daily_log_return.rolling(window=20).mean()
    
    # Combine Volume Adjusted High-Low Volatility and Momentum
    combined_metric = rolling_avg_high_low - rolling_avg_log_return
    
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    intraday_momentum = (df['high'] - df['low']) / df['low']
    total_momentum = price_momentum + intraday_momentum
    
    # Calculate Average Volume over 20 Days
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    
    # Adjust Combined Metric by Intraday and Volume
    adjusted_combined_metric = combined_metric * total_momentum / df['volume'] * avg_volume_20_days
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = intraday_volume_adjusted + rolling_avg_high_low + rolling_avg_log_return + adjusted_combined_metric
    
    return final_alpha_factor
```
