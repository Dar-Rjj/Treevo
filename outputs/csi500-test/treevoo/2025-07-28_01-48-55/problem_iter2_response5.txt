def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    close = df['close']
    avg_close_20 = close.rolling(window=20).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    adjusted_price_momentum = (close - avg_close_20) * (df['volume'].rolling(window=20).sum() / 20) / df['volume']

    # Calculate Volume Breakout Indicator
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_breakout = (df['volume'] > avg_volume).astype(int)

    # Combine Adjusted Momentum and Breakout
    final_factor = (adjusted_price_momentum * 1) + (volume_breakout * 0.5)

    return final_factor
