def heuristics_v2(df):
    # Calculate Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Money Flow
    money_flow = typical_price * df['volume']
    
    # Initialize Positive and Negative Money Flow
    positive_money_flow = 0
    negative_money_flow = 0
    
    # Calculate Positive and Negative Money Flow
    for i in range(1, len(typical_price)):
        if typical_price[i] > typical_price[i-1]:
            positive_money_flow += money_flow[i]
        else:
            negative_money_flow += money_flow[i]
    
    # Calculate Money Flow Ratio
    if negative_money_flow == 0:
        money_flow_ratio = 100
    else:
        money_flow_ratio = positive_money_flow / negative_money_flow
    
    # Calculate MFI
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    return pd.Series(mfi, index=df.index)
