import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Base Momentum
    base_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Range Contribution
    intraday_range_contribution = (df['high'] - df['low']) / df['low']
    combined_momentum = base_momentum + intraday_range_contribution
    
    # Adjust for Volume
    average_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_momentum = (combined_momentum * average_volume) / df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Final Alpha Factor Components
    price_momentum = df['close'] - sma_10
    volume_adjusted_high_low_range = (df['high'] - df['low']) * np.log(df['volume'])
    rolling_momentum = volume_adjusted_high_low_range.rolling(window=20).sum()
    
    # Generate Final Alpha Factor
    final_alpha_factor = (
        price_momentum + 
        volume_adjusted_momentum - 
        volume_adjusted_high_low_range + 
        rolling_momentum
    )
    
    return final_alpha_factor
