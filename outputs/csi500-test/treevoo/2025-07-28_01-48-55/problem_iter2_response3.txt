```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    upward_thrust = np.where(df['close'] > df['open'], df['volume'], 0)
    downward_thrust = np.where(df['close'] <= df['open'], df['volume'], 0)
    volume_thrust = upward_thrust.sum() - downward_thrust.sum()
    
    # Calculate Volume Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    log_volume = np.log(df['volume'])
    adjusted_high_low_range = high_low_range * log_volume
    
    # Calculate Momentum Factor with a rolling window
    window_size = 5
    momentum_factor = adjusted_high_low_range.rolling(window=window_size).sum()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (intraday_momentum + volume_thrust + momentum_factor) / 3
    return alpha_factor
```
