```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low = high_low_range * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range (20-day)
    rolling_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=20).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return (20-day)
    rolling_daily_log_return = daily_log_return.rolling(window=20).mean()
    
    # Combine Volume Adjusted High-Low Volatility and Momentum
    combined_volatility_momentum = rolling_volume_adjusted_high_low - rolling_daily_log_return
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    integrated_momentum = (daily_log_return * volume_change).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    crossover_signal = np.where(ema_5 > ema_20, 1, np.where(ema_5 < ema_20, -1, 0))
    
    enhanced_integrated_momentum = integrated_momentum + (crossover_signal * 0.1)
    
    # Final Alpha Factor
    final_alpha_factor = combined_volatility_momentum + enhanced_integrated_momentum
    
    return final_alpha_factor
```
