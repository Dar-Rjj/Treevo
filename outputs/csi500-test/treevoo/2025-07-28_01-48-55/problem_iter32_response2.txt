import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Weighted Moving Average (WMA) of Close prices over a 10-day window
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + (df['close'] - df['close'].shift(1))
    
    # Price Momentum
    price_momentum = df['close'] - wma
    
    # Volume Adjusted Spread
    spread = df['high'] - df['low']
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = spread * np.sqrt(relative_volume)
    
    # Enhanced Momentum Factor
    adjusted_high_low_range = high_low_range * np.log(df['volume'])
    enhanced_momentum_factor = adjusted_high_low_range.rolling(window=5).apply(lambda x: (x * np.arange(1, 6)).sum() / 15, raw=True)
    
    # Final Alpha Factor
    final_alpha_factor = (composite_factor + 
                          price_momentum - 
                          volume_adjusted_spread + 
                          enhanced_momentum_factor)
    
    return final_alpha_factor
