import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume-Adjusted Spread
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Average_Volume_20']
    df['Spread'] = df['high'] - df['low']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Calculate Momentum Factor
    df['Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=5).sum()
    
    # Calculate Price-Volatility Interaction
    df['Daily_Price_Change'] = df['close'].diff()
    df['Daily_Volume_Change'] = df['volume'].diff()
    df['Price_Volatility_Interaction'] = df['Daily_Price_Change'] * df['Daily_Volume_Change']
    
    # Calculate Price Trend Strength
    df['Prev_Close'] = df['close'].shift(1)
    df['Prev_High'] = df['high'].shift(1)
    df['Prev_Low'] = df['low'].shift(1)
    df['True_Range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['Up_Move'] = df['high'] - df['Prev_High']
    df['Down_Move'] = df['Prev_Low'] - df['low']
    df['Positive_DM'] = np.where(df['Up_Move'] > df['Down_Move'], df['Up_Move'], 0)
    df['Negative_DM'] = np.where(df['Down_Move'] > df['Up_Move'], df['Down_Move'], 0)
    df['Positive_DM_Sum'] = df['Positive_DM'].rolling(window=14).sum()
    df['Negative_DM_Sum'] = df['Negative_DM'].rolling(window=14).sum()
    df['Directional_Index'] = 100 * (np.abs(df['Positive_DM_Sum'] - df['Negative_DM_Sum']) / (df['Positive_DM_Sum'] + df['Negative_DM_Sum']))
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] 
                                - df['Volume_Adjusted_Spread'] 
                                + df['Momentum_Factor'] 
                                + df['Price_Volatility_Interaction'] 
                                + df['Directional_Index'])
    
    return df['Final_Alpha_Factor']
