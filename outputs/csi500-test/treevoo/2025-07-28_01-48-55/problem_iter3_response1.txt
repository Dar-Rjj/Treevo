```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['integrated_momentum'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    crossover_signal = np.where(df['ema_5'] > df['ema_20'], 1, np.where(df['ema_5'] < df['ema_20'], -1, 0))
    df['crossover_signal'] = crossover_signal
    
    # Add or subtract a positive value based on the crossover signal
    df['integrated_momentum'] += df['crossover_signal'] * 0.1  # Positive value for enhancement
    
    # Incorporate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['normalized_high_low_range'] = df['high_low_range'] / df['close']
    
    # Integrate with Integrated Momentum
    df['alpha_factor'] = df['integrated_momentum'] * df['normalized_high_low_range']
    
    return df['alpha_factor']
```
