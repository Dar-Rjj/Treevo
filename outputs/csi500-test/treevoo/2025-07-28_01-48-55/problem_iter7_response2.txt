```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window using Close prices
    sma = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - sma
    
    # Calculate Volume-Adjusted Spread
    spread = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / avg_volume
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Range Volatility
    daily_range = df['high'] - df['low']
    range_volatility = daily_range.rolling(window=10).mean()
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum - volume_adjusted_spread - range_volatility
    
    return final_alpha_factor
```
