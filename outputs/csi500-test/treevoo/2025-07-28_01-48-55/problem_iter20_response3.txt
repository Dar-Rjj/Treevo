import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    volume_momentum = df['volume'].diff(1)
    
    # Exponential Moving Average (EMA) with smoothing factor of 2/(N+1)
    n = 10
    ema_close = df['close'].ewm(span=n, adjust=False).mean()
    
    # High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    adjusted_high_low_range = high_low_range * np.sqrt(np.log(df['volume']))
    
    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + df['close'].diff(1)
    
    # Price Momentum
    price_momentum = df['close'] - ema_close
    
    # Volume Adjusted Spread
    spread = df['high'] - df['low']
    average_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume
    adjusted_spread = spread * np.sqrt(relative_volume)
    
    # Enhanced Momentum Factor
    weights = np.arange(1, 6)  # Linearly increasing weights for the last 5 days
    enhanced_momentum_factor = (adjusted_high_low_range.rolling(window=5).apply(lambda x: np.sum(x * weights), raw=True))
    
    # Final Alpha Factor
    final_alpha_factor = (composite_factor
                          + price_momentum
                          - adjusted_spread
                          + enhanced_momentum_factor)
    
    return final_alpha_factor
