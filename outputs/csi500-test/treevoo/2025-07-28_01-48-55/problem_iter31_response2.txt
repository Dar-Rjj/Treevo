```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day EMA and SMA
    N = 10
    df['SMA_10'] = df['close'].rolling(window=N).mean()
    df['EMA_10'] = df['close'].ewm(span=N, adjust=False).mean()

    # Calculate Price Momentum and Momentum Strength
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    df['ROC_5'] = df['close'].pct_change(5)

    # Calculate Volume Adjusted High-Low Range and Trading Volume Patterns
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range_30'] = df['Daily_Range'].rolling(window=30).mean()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Adjusted_Range'] = df['Daily_Range'] * (np.log(df['Volume_Ratio']) ** (1/3))

    # Calculate Enhanced Momentum Factor
    weights = np.array([1.0 * (1.2 ** i) for i in range(7)])
    df['Enhanced_Momentum'] = df['Adjusted_Range'].rolling(window=7).apply(lambda x: np.dot(x, weights), raw=True)

    # Analyze Daily Price Action and Volatility
    df['ATR_14'] = df[['high', 'low', 'close']].shift().rolling(window=14).apply(
        lambda x: np.max(np.abs(x[1:] - x[:-1])), raw=True)
    df['Range_Ratio'] = df['Daily_Range'] / df['Avg_Daily_Range_30']
    df['Close_Position'] = (df['close'] - df['open']) / df['Daily_Range']

    # Evaluate Market Sentiment
    df['Up_Days'] = (df['close'] > df['open']).rolling(window=30).sum()
    df['Down_Days'] = (df['close'] < df['open']).rolling(window=30).sum()
    df['Sentiment_Gap'] = df['open'] - df['close'].shift(1)

    # Trend Indicator
    df['Trend_Indicator'] = np.where(df['close'] > df['SMA_10'].rolling(window=50).mean(), 1, -1)

    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Adjusted_Range'] 
        + df['Enhanced_Momentum'] 
        * df['Trend_Indicator']
    )

    return df['Alpha_Factor']
```
