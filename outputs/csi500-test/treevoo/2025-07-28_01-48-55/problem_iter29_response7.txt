```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving average of daily returns
    df['5d_ma_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['20d_ma_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate 10-day EMA and 15-day WMA
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(15).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Price Momentum and Integrated Momentum
    df['price_momentum'] = df['close'] - df['ema_10']
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20d_ma_daily_return']
    
    # Intraday range and shadow lengths
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = (df['high'] - df[['open', 'close']].max(axis=1))
    df['lower_shadow'] = (df[['open', 'close']].min(axis=1) - df['low'])
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Volume Adjusted High-Low Range
    df['volume_adjusted_high_low'] = (df['high'] - df['low']) * np.sqrt(np.log(df['volume']))
    
    # 10-day Historical Volatility
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['10d_volatility'] = df['log_return'].rolling(window=10).std() * np.sqrt(252)
    
    # Change in volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['vol_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['vol_ma_20']
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = df['volume'] * (df['close'] - df['open']).clip(lower=0)
    df['downward_thrust'] = df['volume'] * (df['open'] - df['close']).clip(lower=0)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Moving Average Crossover Signals
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, 0)
    df['crossover_signal'] = np.where(df['ema_5'] < df['ema_20'], -1, df['crossover_signal'])
    
    # Final Composite Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['volume_adjusted_high_low'] -
        (df['volume_adjusted_high_low'] * df['shadow_ratio']) +
        (df['log_return'] * df['volume_change']).rolling(window=10).sum() +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] +
        df['10d_volatility'] * df['enhanced_volume_thrust']
    )
    
    return df['alpha_factor']
```
