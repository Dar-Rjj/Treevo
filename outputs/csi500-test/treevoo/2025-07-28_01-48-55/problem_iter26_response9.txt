import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    money_flow_multiplier = (typical_price - df['open']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    money_flow_volume = money_flow_multiplier * df['volume']
    
    # Calculate the positive and negative money flow
    positive_money_flow = money_flow_volume.where(money_flow_multiplier > 0, 0).rolling(window=14).sum()
    negative_money_flow = abs(money_flow_volume.where(money_flow_multiplier < 0, 0).rolling(window=14).sum())
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_money_flow / negative_money_flow
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    return money_flow_index
