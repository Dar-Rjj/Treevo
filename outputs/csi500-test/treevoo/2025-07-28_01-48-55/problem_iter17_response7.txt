```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['Close_Median_20'] = df['close'].rolling(window=20).median()
    df['Volume_Avg_20'] = df['volume'].rolling(window=20).mean()
    df['Adjusted_Price_Momentum'] = (df['close'] - df['Close_Median_20']) * df['Volume_Avg_20'] / df['volume']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Daily Volume Change
    df['Volume_Change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    df['Combined_Factor'] = (df['Adjusted_Price_Momentum'] * 1.5) + (df['Intraday_Reversal'] * df['Volume_Change'])
    
    # Volume Trend Composite
    df['Volume_Trend_Composite'] = df['Intraday_Reversal'] * (df['volume'] - df['volume'].shift(1))
    
    # Synthesize Composite
    df['Final_Factor_Value'] = df['Combined_Factor'] * df['Volume_Trend_Composite']
    
    return df['Final_Factor_Value'].dropna()
```
