```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    median_close = df['close'].rolling(window=20).median()
    sum_volume_20 = df['volume'].rolling(window=20).sum() / 20
    adjusted_price_momentum = (df['close'] - median_close) * (sum_volume_20 / df['volume'])

    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Measure Daily Volume Change
    daily_volume_change = (df['volume'] / df['volume'].shift(1)) - 1

    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    intermediate_factor = (adjusted_price_momentum * 1.5) + (intraday_reversal * daily_volume_change)

    # Calculate Exponential Moving Average (EMA)
    ema = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)

    # Calculate Price Momentum
    price_momentum = df['close'] - ema

    # Calculate Additional Price Momentum
    additional_price_momentum = df['close'] - wma

    # Calculate Volume Adjusted Spread
    high_low_range = df['high'] - df['low']
    relative_volume = df['volume'] / (df['volume'].rolling(window=20).sum() / 20)
    volume_adjusted_spread = high_low_range * relative_volume

    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])

    # Calculate Momentum Factor
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()

    # Final Alpha Factor
    final_alpha_factor = (intermediate_factor + 
                          price_momentum + 
                          additional_price_momentum - 
                          volume_adjusted_spread + 
                          momentum_factor)

    return final_alpha_factor
```
