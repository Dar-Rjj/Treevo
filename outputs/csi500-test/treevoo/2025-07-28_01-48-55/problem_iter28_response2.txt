```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 5-day and 20-day moving average of daily returns
    df['ma_5_returns'] = df['daily_return'].rolling(window=5).mean()
    df['ma_20_returns'] = df['daily_return'].rolling(window=20).mean()
    
    # Determine intraday range and its ratio to previous day's close
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Measure upper and lower shadow lengths and their ratio
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Assess the change in volume from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute the ratio of volume to the 20-day moving average of volume
    df['ma_20_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['ma_20_volume']
    
    # Create a composite factor by combining 20-day momentum and 20-day moving average volume ratio
    # Assign weights to each component (e.g., 0.6 for returns, 0.4 for volume)
    df['composite_factor'] = 0.6 * df['ma_20_returns'] + 0.4 * df['volume_ratio']
    
    return df['composite_factor']
```
