```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of open, high, low, and close prices
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the relative strength as the ratio of closing price to the lowest price
    relative_strength = df['close'] / df['low']
    
    # Calculate the volume weighted amount
    volume_weighted_amount = df['amount'] * df['volume']
    
    # Alpha factor is a combination of the above factors
    alpha_factor = (avg_price / price_range) * relative_strength * volume_weighted_amount
    
    return alpha_factor
```
