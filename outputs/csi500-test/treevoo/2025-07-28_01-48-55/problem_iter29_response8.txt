```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) over a 10-day window
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = df['High_Low_Range'] * np.log(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Relative Volume
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Average_Volume_20']
    
    # Calculate Volume-Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Calculate Momentum Factor
    df['Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=5).sum()
    
    # Calculate Daily Price Change and Daily Volume Change
    df['Daily_Price_Change'] = df['close'].diff()
    df['Daily_Volume_Change'] = df['volume'].diff()
    
    # Calculate Price-Volatility Interaction
    df['Price_Volatility_Interaction'] = df['Daily_Price_Change'] * df['Daily_Volume_Change']
    
    # Calculate Price-Acceleration
    df['Price_Acceleration'] = df['close'] + df['close'].shift(2) - 2 * df['close'].shift(1)
    
    # Calculate Volume-Acceleration
    df['Volume_Acceleration'] = df['volume'] + df['volume'].shift(2) - 2 * df['volume'].shift(1)
    
    # Calculate Price-Volume Acceleration Interaction
    df['Price_Volume_Acceleration_Interaction'] = df['Price_Acceleration'] * df['Volume_Acceleration']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Price_Momentum'] 
                                 - df['Volume_Adjusted_Spread'] 
                                 + df['Momentum_Factor'] 
                                 + df['Price_Volatility_Interaction'] 
                                 + df['Price_Volume_Acceleration_Interaction'])

    return df['Final_Alpha_Factor']
```
