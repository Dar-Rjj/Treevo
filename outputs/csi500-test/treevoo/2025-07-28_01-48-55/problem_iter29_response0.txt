```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['Median_Close'] = df['close'].rolling(window=20).median()
    df['Sum_Volume_20'] = df['volume'].rolling(window=20).sum() / 20
    df['Adjusted_Momentum'] = (df['close'] - df['Median_Close']) * (df['Sum_Volume_20'] / df['volume'])
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Daily Volume Change
    df['Volume_Change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    df['Intermediate_Factor'] = (df['Adjusted_Momentum'] * 1.5) + (df['Intraday_Reversal'] * df['Volume_Change'])
    
    # Calculate Exponential Moving Average (EMA)
    df['EMA_10'] = df['close'].ewm(alpha=2/11, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Volume Adjusted Spread
    df['Volume_Adjusted_Spread'] = (df['high'] - df['low']) * (df['volume'] / df['Sum_Volume_20'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Intermediate_Factor'] + df['Price_Momentum'] - df['Volume_Adjusted_Spread']
    
    return df['Final_Alpha_Factor']
```
