```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])

    # Price Momentum
    price_momentum = df['close'] - df['open']

    # Volume Change
    volume_change = df['volume'].diff()

    # Combined Intraday and Price Momentum
    combined_momentum = intraday_reversal + price_momentum

    # Volume Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    log_volume = np.log(df['volume'])
    volume_adjusted_range = high_low_range * np.sqrt(log_volume)

    # Weighted Moving Average (WMA) with linearly increasing weights
    wma_weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: np.sum(wma_weights * x) / np.sum(wma_weights), raw=True)

    # Price Momentum (Adjusted)
    price_momentum_adj = df['close'] - wma

    # Volume Adjusted Spread
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    volume_adjusted_spread = high_low_range * np.sqrt(relative_volume)

    # Momentum Factor
    momentum_factor = volume_adjusted_range.rolling(window=10).apply(lambda x: np.sum(x * np.arange(1, 11)) / np.sum(np.arange(1, 11)), raw=True)

    # Weighted Price Change
    price_change = df['close'].diff()
    weighted_price_change = price_change * df['volume']

    # Range Volatility
    range_volatility = high_low_range.rolling(window=10).mean()

    # Price Oscillator
    sma_5 = df['close'].rolling(window=5).mean()
    price_oscillator = wma - sma_5

    # Composite Intermediate Factor
    composite_intermediate = combined_momentum + volume_adjusted_range - volume_change

    # Enhanced Momentum Factor
    enhanced_momentum_factor = volume_adjusted_range.rolling(window=10).apply(lambda x: np.sum(x * np.arange(1, 11)) / np.sum(np.arange(1, 11)), raw=True)

    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum_adj - 
        volume_adjusted_spread + 
        momentum_factor + 
        weighted_price_change - 
        range_volatility + 
        price_oscillator + 
        composite_intermediate + 
        enhanced_momentum_factor
    )

    return final_alpha_factor
```
