```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights
    w = np.arange(1, 11)
    df['WMA'] = df['close'].rolling(window=10).apply(lambda x: np.sum(w * x) / np.sum(w), raw=True)

    # Calculate Volume Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = (df['high'] - df['low']) * np.log(df['volume'])

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA']

    # Calculate Spread
    df['Spread'] = df['high'] - df['low']

    # Calculate Relative Volume
    df['Relative_Volume'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Calculate Volume Adjusted Spread
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']

    # Calculate Momentum Factor using a rolling window sum
    df['Momentum_Factor'] = df['Volume_Adjusted_High_Low_Range'].rolling(window=10).sum()

    # Calculate Open-Close Spread
    df['Open_Close_Spread'] = df['open'] - df['close']

    # Calculate Volume Adjusted Open-Close Spread
    df['Volume_Adjusted_Open_Close_Spread'] = df['Open_Close_Spread'] * df['Relative_Volume']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_Spread'] 
        + df['Momentum_Factor']
        - df['Volume_Adjusted_Open_Close_Spread']
    )

    return df['Final_Alpha_Factor']
```
