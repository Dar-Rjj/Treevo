```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with exponentially increasing weights
    weights = np.arange(1, 16)
    wma = df['close'].rolling(window=15).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)

    # Calculate Price Momentum
    price_momentum = df['close'] - wma

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume_30
    volume_adjusted_spread = spread * relative_volume

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)

    # Integrate Log Return with Volume Change
    integrated_log_return = (daily_log_return * volume_change).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (ema_5 > ema_20).astype(int) - (ema_5 < ema_20).astype(int)

    # Calculate Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']).astype(int) * (df['volume'] * (df['close'] - df['open']))
    downward_thrust = (df['close'] <= df['open']).astype(int) * (df['volume'] * (df['open'] - df['close']))
    price_range_factor = (df['high'] - df['low']) / df['open']
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + price_range_factor

    # Incorporate Price-Volatility Interaction
    historical_volatility = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)
    enhanced_volume_thrust_with_volatility = historical_volatility * enhanced_volume_thrust

    # Rolling Average of Daily Log Return
    rolling_avg_log_return = daily_log_return.rolling(window=20).mean()

    # Final Integration
    final_alpha_factor = (
        price_momentum +
        integrated_log_return +
        enhanced_volume_thrust_with_volatility -
        volume_adjusted_spread -
        rolling_avg_log_return +
        crossover_signal
    )

    return final_alpha_factor
```
