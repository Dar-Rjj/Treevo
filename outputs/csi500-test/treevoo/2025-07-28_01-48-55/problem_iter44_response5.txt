import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving averages of daily returns
    df['5_day_ma_return'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_ma_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) and Price Momentum
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Weighted Moving Average (WMA) and Integrated Momentum
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20_day_ma_return']
    
    # Intraday range and ratio
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 10-day Historical Volatility
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['10_day_volatility'] = df['log_return'].rolling(window=10).std()
    
    # Volume change and ratio to 20-day moving average
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrated Log Return
    df['integrated_log_return'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = (df['close'] > df['open']) * (df['volume'] * (df['close'] - df['open']))
    df['downward_thrust'] = (df['close'] <= df['open']) * (df['volume'] * (df['open'] - df['close']))
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust']
    
    # Trend of returns over multiple days
    df['5_day_sum_returns'] = df['daily_return'].rolling(window=5).sum()
    
    # Current day's performance compared to recent average
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['deviation_from_mean'] = df['close'] - df['5_day_avg_close']
    
    # Z-score factor
    df['20_day_std_close'] = df['close'].rolling(window=20).std()
    df['z_score'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['20_day_std_close']
    
    # EMA Crossover Signal
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = (df['ema_5'] > df['ema_20']) * 1 - (df['ema_5'] < df['ema_20']) * 1
    
    # Composite Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['intraday_range_ratio'] -
        df['shadow_ratio'] +
        df['integrated_log_return'] +
        df['enhanced_volume_thrust'] * df['10_day_volatility'] +
        df['crossover_signal'] +
        df['z_score']
    )
    
    # Apply a 20-day rolling window to calculate the average of Daily Log Return
    df['rolling_avg_log_return'] = df
