```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    upward_thrust = ((df['close'] > df['open']) * (df['volume'] * (df['close'] - df['open']))).sum()
    downward_thrust = ((df['close'] <= df['open']) * (df['volume'] * (df['open'] - df['close']))).sum()
    price_range_factor = (df['high'] - df['low']) / df['open']
    combined_factors = (upward_thrust - downward_thrust) + price_range_factor

    # Calculate Accumulated Volume Impact
    accumulated_volume = 0
    for i in range(len(df)):
        if df.iloc[i]['close'] > df.iloc[i]['open']:
            accumulated_volume += df.iloc[i]['volume']
        elif df.iloc[i]['close'] <= df.iloc[i]['open']:
            accumulated_volume -= df.iloc[i]['volume']
    daily_volume_impact = accumulated_volume / len(df)

    # Calculate Simple Moving Average (SMA)
    sma_10 = df['close'].rolling(window=10).mean()

    # Calculate Price Momentum
    price_momentum = df['close'] - sma_10

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    relative_volume = df['volume'] / avg_volume_10
    volume_adjusted_spread = spread * relative_volume

    # Measure Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)

    # Combine Final Alpha Factor
    final_alpha_factor = (intraday_momentum + combined_factors +
                          daily_volume_impact + price_momentum -
                          volume_adjusted_spread -
                          (intraday_momentum * volume_change))

    return final_alpha_factor
```
