```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    df['WMA'] = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']

    # Volume Adjusted High-Low Range
    df['VolumeAdjustedHL'] = (df['high'] - df['low']) * np.log(df['volume'])

    # Calculate Spread
    df['Spread'] = df['high'] - df['low']

    # Calculate Relative Volume
    df['AvgVolume20'] = df['volume'].rolling(window=20).mean()
    df['RelativeVolume'] = df['volume'] / df['AvgVolume20']

    # Volume Adjusted Spread
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']

    # Momentum Factor using Volume Adjusted High-Low Range
    df['MomentumFactor'] = df['VolumeAdjustedHL'].rolling(window=10).sum()

    # Calculate Open-Close Spread
    df['OC_Spread'] = df['close'] - df['open']

    # Volume Adjusted Open-Close Spread
    df['VolumeAdjustedOC'] = df['OC_Spread'] * df['RelativeVolume']

    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['PriceMomentum'] 
                              - df['VolumeAdjustedSpread'] 
                              + df['MomentumFactor'] 
                              - df['VolumeAdjustedOC'])

    return df['FinalAlphaFactor']
```
