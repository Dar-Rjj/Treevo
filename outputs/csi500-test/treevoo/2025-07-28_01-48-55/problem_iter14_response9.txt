```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponentially Weighted Moving Average (EWMA)
    ewma = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - ewma
    
    # Calculate Volume Adjusted Spread
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume_30
    spread = df['high'] - df['low']
    volume_adjusted_spread = spread * relative_volume
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume Thrust
    upward_thrust = (df['close'] > df['open']) * df['volume']
    downward_thrust = (df['close'] <= df['open']) * df['volume']
    volume_thrust = upward_thrust - downward_thrust
    
    # Calculate Trend Strength
    abs_diff = abs(df['close'].diff(1))
    sum_abs_diff = abs_diff.rolling(window=5).sum()
    trend_strength = ewma - sum_abs_diff
    
    # Final Alpha Factor
    alpha_factor = (
        price_momentum
        + intraday_momentum
        - volume_adjusted_spread
        + volume_thrust
        - trend_strength
    )
    
    return alpha_factor
```
