```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day simple moving averages of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Identify the crossover of 5-day and 20-day SMAs as a momentum indicator
    df['Momentum_Signal'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Compute the daily range (High - Low)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Sum up the last 10 days' ranges to identify recent market volatility
    df['Recent_Volatility'] = df['Daily_Range'].rolling(window=10).sum()
    
    # Calculate average true range (ATR) over 14 days for volatility measurement
    tr = pd.DataFrame(index=df.index)
    tr['H-L'] = df['high'] - df['low']
    tr['H-PC'] = abs(df['high'] - df['close'].shift(1))
    tr['L-PC'] = abs(df['low'] - df['close'].shift(1))
    tr['TR'] = tr[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    df['ATR_14'] = tr['TR'].rolling(window=14).mean()
    
    # Create a 10-day average volume baseline
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Flagging days when volume exceeds the 10-day average by a certain threshold
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Avg_Volume_10']).astype(int)
    
    # Determine if close is above the open on high volume as an indication of strong buying interest
    df['Strong_Buying'] = ((df['close'] > df['open']) & (df['volume'] > df['Avg_Volume_10'])).astype(int)
    
    # Detecting days with a close below the open and high volume, signaling strong selling pressure
    df['Strong_Selling'] = ((df['close'] < df['open']) & (df['volume'] > df['Avg_Volume_10'])).astype(int)
    
    # Measure percentage change in volume from one day to the next
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Alerting when daily volume change surpasses 50% increase from previous day
    df['Volume_Spike_Pct'] = (df['Volume_Change'] > 0.5).astype(int)
    
    # Creating a composite score that weights both momentum and volume signals
    df['Composite_Score'] = 0.7 * df['Momentum_Signal'] + 0.3 * df['Volume_Spike_Pct']
    
    return df['Composite_Score']
```
