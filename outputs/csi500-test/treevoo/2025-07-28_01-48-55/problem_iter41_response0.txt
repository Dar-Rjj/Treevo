```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 15-day EMA with smoothing factor 0.3
    df['EMA_15'] = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Price Momentum (Current Open - EMA)
    df['Price_Momentum'] = df['open'] - df['EMA_15']
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['True_Range'] = df['True_Range'].fillna(0)  # Fill any NaNs that might have been created
    
    # Calculate Volume-Adjusted True Range
    df['Volume_Adjusted_TR'] = df['True_Range'] * np.sqrt(df['volume'])
    
    # Calculate Volume-Weighted High-Low Range
    df['HL_Range'] = df['high'] - df['low']
    df['Volume_Weighted_HL_Range'] = df['HL_Range'] * df['volume']
    
    # Calculate Momentum Factor using a 7-day rolling window mean of Volume-Weighted High-Low Range
    df['Momentum_Factor'] = df['Volume_Weighted_HL_Range'].rolling(window=7).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] 
        - df['Volume_Adjusted_TR'] 
        + df['Momentum_Factor']
    )
    
    return df['Final_Alpha_Factor']
```
