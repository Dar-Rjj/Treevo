```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Price Change
    df['price_change'] = df['close'] - df['open']
    df['volume_weighted_price_change'] = df['price_change'] * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['adjusted_high_low_range'] = df['high_low_range'] * np.log(df['volume'])
    
    # Calculate Momentum Factor using a rolling window sum
    window_size = 5  # Set the desired window size
    df['momentum_factor'] = df['adjusted_high_low_range'].rolling(window=window_size).sum()
    
    # Return the momentum factor
    return df['momentum_factor']
```
