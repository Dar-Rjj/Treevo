```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    def calculate_wma(prices, window):
        weights = np.arange(1, window + 1)
        wma = prices.rolling(window=window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
        return wma
    
    # Calculate Price Momentum
    wma_10 = calculate_wma(df['close'], 10)
    price_momentum = df['close'] - wma_10
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Integrate Log Return with Volume Change
    integrated_momentum = (daily_log_return * volume_change).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = np.where(ema_5 > ema_20, 1, np.where(ema_5 < ema_20, -1, 0))
    integrated_momentum += crossover_signal
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    upward_thrust = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0)
    downward_thrust = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0)
    price_range_factor = (df['high'] - df['low']) / df['open']
    enhanced_volume_thrust = (upward_thrust - downward_thrust).rolling(window=10).sum() + price_range_factor
    
    # Final Integration
    final_integration = integrated_momentum + intraday_momentum + enhanced_volume_thrust
    
    # Final Alpha Factor
    volume_adjusted_spread = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=20).mean())
    final_alpha_factor = final_integration + price_momentum - volume_adjusted_spread
    
    return final_alpha_factor
```
