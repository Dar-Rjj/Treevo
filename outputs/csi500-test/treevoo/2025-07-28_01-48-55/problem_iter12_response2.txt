```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['close_momentum'] = df['close'] - df['close'].rolling(window=20).median()
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['adjusted_momentum'] = (df['close_momentum'] * df['avg_volume_20_days']) / df['volume']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Daily Volume Change
    df['daily_volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Calculate Volume Moving Average
    df['volume_ma_5_days'] = df['volume'].rolling(window=5).mean()
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    adjusted_momentum_factor = 1.5 * df['adjusted_momentum']
    intraday_reversal_factor = (df['intraday_reversal'] * df['daily_volume_change']) / df['volume_ma_5_days']
    final_factor = adjusted_momentum_factor + intraday_reversal_factor
    
    return final_factor
```
