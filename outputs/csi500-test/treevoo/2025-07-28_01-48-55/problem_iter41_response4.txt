import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving Averages of Daily Returns
    df['5_Day_MA_Returns'] = df['Daily_Return'].rolling(window=5).mean()
    df['20_Day_MA_Returns'] = df['Daily_Return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) on Close prices
    df['10_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Weighted Moving Average (WMA) on Close prices
    weights = np.arange(1, 16)
    df['15_WMA'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Price and Integrated Momentum
    df['Price_Momentum'] = df['close'] - df['10_EMA']
    df['Integrated_Momentum'] = df['close'] - df['15_WMA'] + df['20_Day_MA_Returns']
    
    # Intraday Range and Shadow Lengths
    df['Intraday_Range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['Upper_Shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['Lower_Shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['Shadow_Ratio'] = df['Upper_Shadow'] / df['Lower_Shadow']
    
    # Historical Volatility
    df['10_Historical_Volatility'] = df['high'] - df['low']
    df['10_HV_Sum'] = df['10_Historical_Volatility'].rolling(window=10).sum()
    
    # Volume and Its Interaction with Price
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_20_MA'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio'] = df['volume'] / df['Volume_20_MA']
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Threshold'] = (df['volume'] > df['Avg_Volume_10']).astype(int)
    df['Volume_Adjusted_High_Low'] = (df['high'] - df['low']) / df['open']
    df['Volume_Adjusted_Spread'] = (df['open'] - df['close']).abs() / df['open']
    
    # Enhanced Volume Thrust
    up_thrust = (df['close'] > df['open']) * (df['volume'] * (df['close'] - df['open']))
    down_thrust = (df['close'] <= df['open']) * (df['volume'] * (df['open'] - df['close']))
    df['Enhanced_Volume_Thrust'] = up_thrust - down_thrust + df['Volume_Adjusted_High_Low']
    
    # Moving Average Crossover Signals
    df['5_SMA'] = df['close'].rolling(window=5).mean()
    df['20_SMA'] = df['close'].rolling(window=20).mean()
    df['5_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (df['5_EMA'] > df['20_EMA']).astype(int) - (df['5_EMA'] < df['20_EMA']).astype(int)
    
    # Daily Log Return and Integrated Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Volume_Change_Pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Integrated_Log_Return'] = (df['Daily_Log_Return'] * df['Volume_Change_Pct']).rolling(window=10).sum()
    
    # Final Composite Alpha Factor
    final_factor = (
        df['Price_Momentum'] +
        df['Integrated_Momentum'] -
        df['Volume_Adjusted_High_Low'] -
        df['Volume_Adjusted_Spread'] +
        df['Integrated_Log_Return'] +
        crossover_signal
