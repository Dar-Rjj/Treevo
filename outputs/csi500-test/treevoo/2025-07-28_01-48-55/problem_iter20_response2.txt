```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponentially Weighted Moving Average (EWMA)
    ewma_15 = df['close'].ewm(span=15).mean()
    
    # Calculate Weighted Moving Average (WMA)
    wma_10 = df['close'].rolling(window=10).apply(lambda x: np.sum(x * np.arange(1, 11)) / np.sum(np.arange(1, 11)), raw=True)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - ewma_15
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_adjusted_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_30)
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume Thrust
    upward_thrust = df[df['close'] > df['open']]['volume'] * (df['close'] - df['open'])
    downward_thrust = df[df['close'] <= df['open']]['volume'] * (df['open'] - df['close'])
    volume_thrust = upward_thrust - downward_thrust
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    integrated_log_return = (daily_log_return * volume_change).rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5).mean()
    ema_20 = df['close'].ewm(span=20).mean()
    crossover_signal = (ema_5 > ema_20) * 1 - (ema_5 < ema_20) * 1
    
    # Calculate Trend Strength
    abs_diff = np.abs(df['close'] - df['close'].shift(1))
    trend_strength = abs_diff.rolling(window=5).sum() - ewma_15
    
    # Calculate Enhanced Volume Thrust
    open_price = df['open']
    close_price = df['close']
    high_price = df['high']
    low_price = df['low']
    volume = df['volume']
    
    upward_thrust_enhanced = (close_price > open_price) * (volume * (close_price - open_price))
    downward_thrust_enhanced = (close_price <= open_price) * (volume * (open_price - close_price))
    price_range_factor = (high_price - low_price) / open_price
    enhanced_volume_thrust = (upward_thrust_enhanced - downward_thrust_enhanced) + price_range_factor
    
    # Final Integration
    integrated_momentum = integrated_log_return + intraday_momentum
    final_integration = integrated_momentum + enhanced_volume_thrust
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum 
        + intraday_momentum 
        - volume_adjusted_spread 
        + volume_thrust 
        - trend_strength 
        + final_integration
    )
    
    return final_alpha_factor
```
