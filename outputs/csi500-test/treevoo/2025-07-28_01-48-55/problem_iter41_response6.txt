```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the relative range of the day's trading
    daily_range = df['high'] - df['low']
    open_to_close = abs(df['close'] - df['open'])
    
    # Calculate the buying and selling pressure
    buying_pressure = df['volume'] * (df['close'] - df['low'])
    selling_pressure = df['volume'] * (df['high'] - df['close'])
    
    # Calculate the net buying pressure
    net_buying_pressure = buying_pressure - selling_pressure
    
    # Combine the factors to create the final alpha factor
    factor_value = (daily_range / open_to_close) * net_buying_pressure
    
    return factor_value
```
