def heuristics_v2(data):
    # Calculate the typical price
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * data['volume']
    
    # Calculate the 10-day average money flow
    avg_money_flow_10 = money_flow.rolling(window=10).mean()
    
    # Calculate the 20-day average money flow
    avg_money_flow_20 = money_flow.rolling(window=20).mean()
    
    # Calculate the heuristic factor
    heuristics_factor = (avg_money_flow_10 - avg_money_flow_20) / avg_money_flow_20
    
    return heuristics_factor
