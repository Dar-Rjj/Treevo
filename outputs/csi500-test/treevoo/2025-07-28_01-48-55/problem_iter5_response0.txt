```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-Low Range by Multiplying with Logarithm of Volume
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Rolling Average of Volume Adjusted High-Low Range (20-day)
    rolling_avg_volume_adjusted_high_low = volume_adjusted_high_low_range.rolling(window=20).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Rolling Average of Daily Log Return (20-day)
    rolling_avg_daily_log_return = daily_log_return.rolling(window=20).mean()
    
    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change (10-day accumulation)
    integrated_momentum = (daily_log_return * volume_change).rolling(window=10).sum()
    
    # Calculate 5-day and 20-day Exponential Moving Averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine Crossover Signal
    crossover_signal = np.where(ema_5 > ema_20, 1, np.where(ema_5 < ema_20, -1, 0))
    
    # Combine All Factors
    factor = (rolling_avg_volume_adjusted_high_low - rolling_avg_daily_log_return 
              + integrated_momentum + crossover_signal)
    
    return factor
```
