```python
def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + price_momentum
    
    return composite_factor
```
