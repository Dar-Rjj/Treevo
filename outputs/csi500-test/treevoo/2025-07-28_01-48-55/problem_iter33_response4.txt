```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    df['EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Volume-Weighted High-Low Range
    df['Vol_Weighted_HL'] = (df['high'] - df['low']) * np.sqrt(df['volume'])
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(x[1] - df['close'].shift(1))), axis=1)
    
    # Calculate Volume-Adjusted True Range
    df['Vol_Adjusted_TR'] = df['True_Range'] * np.log(df['volume'])
    
    # Calculate Momentum Factor
    window_size = 10
    df['Momentum_Factor'] = df['Vol_Weighted_HL'].rolling(window=window_size).sum()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Price_Momentum'] - df['Vol_Adjusted_TR'] + df['Momentum_Factor']
    
    return df['Final_Alpha_Factor']
```
