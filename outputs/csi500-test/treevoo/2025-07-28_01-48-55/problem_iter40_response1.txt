```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the 5-day and 20-day moving average of daily returns
    df['ma_5d_return'] = df['daily_return'].rolling(window=5).mean()
    df['ma_20d_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) using Close prices over a 10-day window
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) using Close prices over a 15-day window
    weights = np.arange(1, 16)
    df['wma_15d'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Integrate EMAs into momentum calculations
    df['price_momentum'] = df['close'] - df['ema_10d']
    df['integrated_momentum'] = df['close'] - df['wma_15d'] + df['ma_20d_return']
    
    # Intraday price dynamics and volatility
    df['intraday_range_ratio'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    df['historical_volatility'] = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)
    
    # Volume and its interaction with price
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ma_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma_20d']
    
    # Daily log return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['integrated_log_return'] = df['log_return'] * df['volume_change']
    df['cumulative_integrated_log_return'] = df['integrated_log_return'].rolling(window=10).sum()
    
    # Enhanced volume thrust
    df['upward_thrust'] = df.apply(lambda x: x['volume'] * (x['close'] - x['open']) if x['close'] > x['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda x: x['volume'] * (x['open'] - x['close']) if x['close'] <= x['open'] else 0, axis=1)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Moving average crossover signals
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5d'] > df['ema_20d'], 1, -1)
    
    # Construct final composite alpha factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['intraday_range_ratio'] -
        df['shadow_ratio'] +
        df['cumulative_integrated_log_return'] +
        df['crossover_signal'] +
        df['enhanced_volume_thrust'] *
        df['historical_volatility'] -
        df['log_return'].rolling(window=20).mean()
    )
    
    return df['alpha_factor']
```
