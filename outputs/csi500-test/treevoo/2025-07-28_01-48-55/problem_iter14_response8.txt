```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with linearly increasing weights
    def wma(prices, window):
        weights = np.arange(1, window + 1)
        return prices.rolling(window=window).apply(lambda x: np.sum(x * weights) / np.sum(weights), raw=True)

    # Calculate Price Momentum
    df['WMA_10'] = wma(df['close'], 10)
    df['Price_Momentum'] = df['close'] - df['WMA_10']

    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Integrate Log Return with Volume Change
    df['Integrated_Momentum'] = (df['Log_Return'] * df['Volume_Change']).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = np.where(df['EMA_5'] > df['EMA_20'], 1, np.where(df['EMA_5'] < df['EMA_20'], -1, 0))
    df['Integrated_Momentum'] += crossover_signal

    # Incorporate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Integrated_Momentum'] *= df['High_Low_Range']

    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    upward_thrust = (df['close'] - df['open']).clip(lower=0) * df['volume']
    downward_thrust = (df['open'] - df['close']).clip(lower=0) * df['volume']
    price_range_factor = (df['high'] - df['low']) / df['open']
    df['Enhanced_Volume_Thrust'] = (upward_thrust - downward_thrust) + price_range_factor

    # Final Integration
    df['Final_Integration'] = df['Integrated_Momentum'] + df['Intraday_Momentum'] + df['Enhanced_Volume_Thrust']

    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Relative_Volume'] = df['volume'] / df['Average_Volume_20']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Final_Integration'] + df['Price_Momentum'] - df['Volume_Adjusted_Spread']

    return df['Final_Alpha_Factor']
```
