```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the 5-day and 20-day moving average of daily returns
    df['5_day_mavg_returns'] = df['daily_return'].rolling(window=5).mean()
    df['20_day_mavg_returns'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate Exponential Moving Average (EMA) using Close prices over a 10-day window
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Weighted Moving Average (WMA) using Close prices over a 15-day window
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Integrated Momentum
    df['price_momentum'] = df['close'] - df['ema_10']
    df['integrated_momentum'] = (df['close'] - df['wma_15']) + df['20_day_mavg_returns']
    
    # Intraday range and shadow lengths
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['upper_shadow'] = (df['high'] - df[['open', 'close']].max(axis=1))
    df['lower_shadow'] = (df[['open', 'close']].min(axis=1) - df['low'])
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 10-day Historical Volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['10_day_volatility'] = df['log_return'].rolling(window=10).std() * np.sqrt(252)
    
    # Volume analysis
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20_day_ma']
    df['volume_10_day_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_flag'] = (df['volume'] > df['volume_10_day_avg'] * 1.5).astype(int)
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = df.apply(lambda x: x['volume'] * (x['close'] - x['open']) if x['close'] > x['open'] else 0, axis=1)
    df['downward_thrust'] = df.apply(lambda x: x['volume'] * (x['open'] - x['close']) if x['close'] <= x['open'] else 0, axis=1)
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust'] + (df['high'] - df['low']) / df['open']
    
    # Moving Average Crossover Signals
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['crossover_signal'] = np.where(df['sma_5'] > df['sma_20'], 1, -1)
    
    # Construct Final Composite Alpha Factor
    df['final_alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        (df['high'] - df['low']) / df['open'] -
        (df['close'] - df['open']) / df['open'] * df['volume_ratio'] +
        df['log_return'] * df['crossover_signal'] +
        df['enhanced_volume_thrust'] * df['10_day_volatility'] -
        df['log_return'].rolling(window=20).mean()
    )
    
    return df['final_alpha_factor']
```
