```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Calculate Volume Weighted High-Low Range
    volume_weighted_range = high_low_range * np.sqrt(df['volume'])

    # Calculate Simple Moving Average (SMA)
    sma = df['close'].rolling(window=10).mean()

    # Calculate Price Momentum
    price_momentum = df['close'] - sma

    # Calculate Volume Adjusted Spread
    spread = df['high'] - df['low']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    volume_adjusted_spread = spread * relative_volume

    # Calculate Exponential Moving Average (EMA)
    ema = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate EMA Momentum
    ema_momentum = df['close'] - ema

    # Calculate Momentum Factor
    momentum_factor = volume_weighted_range.rolling(window=10).sum()

    # Final Alpha Factor
    final_alpha_factor = (price_momentum + 
                          ema_momentum - 
                          volume_adjusted_spread + 
                          momentum_factor)

    return final_alpha_factor
```
