```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['mean_close_20'] = df['close'].rolling(window=20).mean()
    df['sum_volume_20'] = df['volume'].rolling(window=20).sum()
    df['avg_volume_20'] = df['sum_volume_20'] / 20
    df['adjusted_price_momentum'] = (df['close'] - df['mean_close_20']) * df['avg_volume_20'] / df['volume']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Change
    df['moving_avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['combined_factor'] = df['adjusted_price_momentum'] * 1 + \
                            (df['intraday_reversal'] * df['volume_change'] / df['moving_avg_volume_5'])
    
    return df['combined_factor']
```
