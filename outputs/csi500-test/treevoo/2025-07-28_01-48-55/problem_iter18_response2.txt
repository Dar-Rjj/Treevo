```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA)
    N = 10
    alpha = 2 / (N + 1)
    df['EMA'] = df['close'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    df['volume_adj_high_low_range'] = df['high_low_range'] * np.sqrt(np.log(df['volume']))
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume Adjusted Spread
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20
    spread = df['high'] - df['low']
    df['volume_adj_spread'] = spread * np.sqrt(relative_volume)
    
    # Calculate Enhanced Momentum Factor
    weights = np.arange(1, 6)  # Linearly increasing weights for the last 5 days
    df['enhanced_momentum_factor'] = df['volume_adj_high_low_range'].rolling(window=5).apply(lambda x: np.dot(x, weights), raw=True)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['price_momentum'] 
                                - df['volume_adj_spread'] 
                                + df['enhanced_momentum_factor'])
    
    return df['final_alpha_factor']
```
