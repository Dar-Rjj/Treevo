import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].diff()
    
    # Calculate EMA of Close over 10 days
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.sqrt(np.log(df['volume']))
    
    # Composite Factor
    composite_factor = (intraday_reversal * volume_momentum) + (df['close'] - df['close'].shift(1))
    
    # Price Momentum
    price_momentum = df['close'] - ema_close_10
    
    # Volume Adjusted Spread
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = (df['high'] - df['low']) * np.sqrt(relative_volume)
    
    # Enhanced Momentum Factor
    weights = np.arange(1, 6)
    adjusted_high_low_range = high_low_range * np.sqrt(np.log(df['volume']))
    enhanced_momentum_factor = (adjusted_high_low_range.rolling(window=5).apply(lambda x: np.sum(weights * x), raw=True)) / np.sum(weights)
    
    # Final Alpha Factor
    final_alpha_factor = (composite_factor 
                          + price_momentum 
                          - volume_adjusted_spread 
                          + enhanced_momentum_factor)
    
    return final_alpha_factor
