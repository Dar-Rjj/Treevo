```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Calculate Price Momentum
    price_momentum = df['close'] - wma

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)

    # Integrate Log Return with Volume Change
    integrated_momentum = (daily_log_return * volume_change).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    crossover_signal = (ema_5 > ema_20) * 1 - (ema_5 < ema_20) * 1
    integrated_momentum += crossover_signal

    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    integrated_momentum *= high_low_range

    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    upward_thrust = (df['close'] > df['open']) * (df['volume'] * (df['close'] - df['open']))
    downward_thrust = (df['close'] <= df['open']) * (df['volume'] * (df['open'] - df['close']))
    enhanced_volume_thrust = (upward_thrust - downward_thrust) + (df['high'] - df['low']) / df['open']

    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = (df['high'] - df['low']) * np.log(df['volume'])

    # Rolling Average of Volume Adjusted High-Low Range
    rolling_avg_volume_adjusted_high_low = volume_adjusted_high_low_range.rolling(window=20).mean()

    # Rolling Average of Daily Log Return
    rolling_avg_daily_log_return = daily_log_return.rolling(window=20).mean()

    # Final Integration
    final_factor = (integrated_momentum + intraday_momentum + enhanced_volume_thrust +
                    (rolling_avg_volume_adjusted_high_low - rolling_avg_daily_log_return) +
                    crossover_signal)

    # Final Alpha Factor
    spread = df['high'] - df['low']
    average_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume
    volume_adjusted_spread = spread * relative_volume
    final_alpha_factor = final_factor + price_momentum - volume_adjusted_spread

    return final_alpha_factor
```
