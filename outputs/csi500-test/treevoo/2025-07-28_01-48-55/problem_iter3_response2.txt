```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Compute Upward Thrust
    upward_thrust = pd.Series(0, index=df.index)
    mask_up = df['close'] > df['open']
    upward_thrust[mask_up] = df['volume'][mask_up] * (df['close'][mask_up] - df['open'][mask_up])
    
    # Compute Downward Thrust
    downward_thrust = pd.Series(0, index=df.index)
    mask_down = df['close'] <= df['open']
    downward_thrust[mask_down] = df['volume'][mask_down] * (df['open'][mask_down] - df['close'][mask_down])
    
    # Compute Price Range Factor
    price_range_factor = (df['high'] - df['low']) / df['open']
    
    # Combine Factors
    volume_thrust = upward_thrust - downward_thrust
    combined_factor = volume_thrust + price_range_factor
    
    return combined_factor
```
