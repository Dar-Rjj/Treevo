```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Moving averages of daily returns
    df['5d_ma_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['20d_ma_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) and Price Momentum
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10']
    
    # Weighted Moving Average (WMA) and Integrated Momentum
    weights = np.arange(1, 16)
    df['wma_15'] = df['close'].rolling(window=15).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    df['integrated_momentum'] = df['close'] - df['wma_15'] + df['20d_ma_daily_return']
    
    # Intraday range and ratio
    df['intraday_range'] = df['high'] - df['low']
    df['range_ratio'] = df['intraday_range'] / df['close'].shift(1)
    
    # Upper and lower shadow lengths
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # 10-day Historical Volatility
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['10d_hist_volatility'] = df['log_return'].rolling(window=10).std() * np.sqrt(252)
    
    # Volume change and ratio to 20-day moving average
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20d']
    
    # Integrated Log Return
    df['integrated_log_return'] = (df['log_return'] * df['volume_change']).rolling(window=10).sum()
    
    # Enhanced Volume Thrust
    df['upward_thrust'] = np.where(df['close'] > df['open'], df['volume'] * (df['close'] - df['open']), 0).cumsum()
    df['downward_thrust'] = np.where(df['close'] <= df['open'], df['volume'] * (df['open'] - df['close']), 0).cumsum()
    df['enhanced_volume_thrust'] = df['upward_thrust'] - df['downward_thrust']
    
    # Intraday Reversal
    df['intraday_reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # 5-day and 20-day EMA for crossover signal
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = np.where(df['ema_5'] > df['ema_20'], 1, -1)
    
    # Final composite alpha factor
    df['alpha_factor'] = (
        df['price_momentum'] +
        df['integrated_momentum'] -
        df['range_ratio'] -
        df['10d_hist_volatility'] +
        df['integrated_log_return'] +
        df['crossover_signal'] * 1.0 +
        df['enhanced_volume_thrust'] +
        df['10d_hist_volatility'] * df['enhanced_volume_thrust'] +
        df['intraday_reversal'] +
        df['shadow_ratio'] * 0.5
    )
    
    return df['alpha_factor']
```
