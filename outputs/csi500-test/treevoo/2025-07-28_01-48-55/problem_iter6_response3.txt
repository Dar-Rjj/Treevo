```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) over a 10-day window
    df['EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Avg_Volume_10d'] = df['volume'].rolling(window=10).mean()
    df['Relative_Volume'] = df['volume'] / df['Avg_Volume_10d']
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']

    # Calculate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Thrust
    upward_thrust = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    downward_thrust = df.apply(lambda row: row['volume'] if row['close'] <= row['open'] else 0, axis=1)
    df['Volume_Thrust'] = upward_thrust - downward_thrust
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Price_Momentum'] + df['Intraday_Momentum'] - df['Volume_Adjusted_Spread'] + df['Volume_Thrust']
    
    return df['Final_Alpha_Factor']
```
