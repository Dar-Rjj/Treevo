```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, 11)  # Linearly increasing weights for 10 days
    wma = df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Calculate Volume Adjusted High-Low Range
    adjusted_high_low_range = high_low_range * np.log(df['volume'])

    # Calculate Price Momentum
    price_momentum = df['close'] - wma

    # Calculate Relative Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume_20

    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = high_low_range * relative_volume

    # Calculate Momentum Factor
    momentum_factor = adjusted_high_low_range.rolling(window=5).sum()

    # Final Alpha Factor
    alpha_factor = (
        price_momentum
        - volume_adjusted_spread
        + momentum_factor
    )

    return alpha_factor
```
