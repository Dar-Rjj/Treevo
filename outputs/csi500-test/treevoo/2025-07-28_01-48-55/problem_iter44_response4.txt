```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) with smoothing factor 0.2
    ema_15 = df['close'].ewm(span=15, adjust=False).mean()
    
    # Calculate Simple Moving Averages (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Identify Crossover of 5-day and 30-day SMAs
    crossover_signal = (sma_5 > sma_30).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_range = (df['high'] - df['low']) * df['volume']
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Sum Last 7 Days' Ranges
    sum_7_days_range = daily_range.rolling(window=7).sum()
    
    # Calculate True Range
    true_range = pd.concat([
        (df['high'] - df['low']).abs(),
        (df['high'] - df['close'].shift()).abs(),
        (df['low'] - df['close'].shift()).abs()
    ], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) with window size 14
    atr_14 = true_range.rolling(window=14).mean()
    
    # Calculate Price Momentum
    price_momentum = df['open'] - ema_15
    
    # Calculate Volume-Adjusted True Range
    volume_adjusted_tr = true_range * df['volume'] ** 0.5
    
    # Calculate Momentum Factor
    momentum_factor = volume_weighted_range.rolling(window=15).mean()
    
    # Analyze Volume for Market Strength
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_strength = (df['volume'] > 1.2 * avg_volume_10).astype(int)
    
    # Evaluate Buying or Selling Pressure
    buying_pressure = ((df['close'] > df['open']) & (df['volume'] > avg_volume_10)).astype(int)
    selling_pressure = ((df['close'] < df['open']) & (df['volume'] > avg_volume_10)).astype(int)
    
    # Measure Volume Spikes
    volume_spike = (df['volume'].pct_change() > 0.5).astype(int)
    
    # Integrate Price and Volume Indicators
    combined_signal = (price_momentum + momentum_factor + crossover_signal + volume_strength + volume_spike - volume_adjusted_tr)
    
    return combined_signal
```
