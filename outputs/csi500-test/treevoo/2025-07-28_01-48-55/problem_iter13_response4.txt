    df['momentum_vol'] = df['log_return'] * df['volume_change']
    df['integrated_momentum'] = df['momentum_vol'].rolling(window=10).sum()
    
    # Enhance with Moving Average Crossover
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = (df['ema_5'] > df['ema_20']).astype(int) - (df['ema_5'] < df['ema_20']).astype(int)
    df['integrated_momentum'] += df['crossover_signal']
    
    # Incorporate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['integrated_momentum'] *= df['high_low_range']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Enhanced Volume Thrust
    df['upward_thrust'] = (df['close'] > df['open']).astype(int) * (df['volume'] * (df['close'] - df['open']))
    df['downward_thrust'] = (df['close'] <= df['open']).astype(int) * (df['volume'] * (df['open'] - df['close']))
    df['price_range_factor'] = (df['high'] - df['low']) / df['open']
    df['enhanced_volume_thrust'] = (df['upward_thrust'] - df['downward_thrust']) + df['price_range_factor']
    
    # Calculate Open-Close Momentum
    df['open_close_momentum'] = (df['close'] - df['open']) / df['open']
    
    # Final Integration
    df['final_factor'] = df['integrated_momentum'] + df['intraday_momentum'] + df['enhanced_volume_thrust'] + df['open_close_momentum']
    
    return df['final_factor']
