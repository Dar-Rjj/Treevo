```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA) with exponentially increasing weights
    def wma(series, window):
        weights = np.arange(1, window + 1)
        return (series.rolling(window).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True))
    
    df['WMA_15'] = wma(df['close'], 15)
    
    # Calculate Adjusted Price Momentum
    df['Adj_Momentum'] = (df['close'] - df['close'].rolling(20).median()) * (df['volume'].rolling(20).sum() / 20) / df['volume']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['WMA_15']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['Relative_Volume'] = df['volume'] / df['volume'].rolling(30).mean()
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Relative_Volume']
    
    # Calculate Trend Strength
    df['Abs_Diff'] = abs(df['close'].diff())
    df['Sum_Abs_Diff'] = df['Abs_Diff'].rolling(5).sum()
    df['Trend_Strength'] = 5 * df['WMA_15'] - df['Sum_Abs_Diff']
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1)
    
    # Integrate Log Return with Volume Change
    df['Log_Return_Volume'] = df['Daily_Log_Return'] * df['Volume_Change']
    df['Accumulated_Log_Return_Volume'] = df['Log_Return_Volume'].rolling(10).sum()
    
    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Crossover_Signal'] = np.where(df['EMA_5'] > df['EMA_20'], 1, np.where(df['EMA_5'] < df['EMA_20'], -1, 0))
    
    # Incorporate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Normalized_High_Low_Range'] = df['High_Low_Range'] / df['close']
    df['Integrated_Momentum'] = df['Price_Momentum'] * df['Normalized_High_Low_Range']
    
    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Measure Daily Volume Change
    df['Daily_Volume_Change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Combine Adjusted Momentum, Intraday Reversal, and Volume Changes
    df['Combined_Factor'] = df['Adj_Momentum'] * 1.5 + df['Intraday_Reversal'] * df['Daily_Volume_Change']
    
    # Rolling Average of Daily Log Return
    df['Rolling_Avg_Log_Return'] = df['Daily_Log_Return'].rolling(20).mean()
    
    # Final Integration
    df['Final_Alpha_Factor'] = (df['Integrated_Momentum'] + df['Intraday_Reversal'] + 
                                 df['Accumulated_Log_Return_Volume'] - df['Volume_Adjusted_Spread'] - 
                                 df['Trend_Strength'] - df['Rolling_Avg_Log_Return'] + 
                                 df['Crossover_Signal'] + df['Combined_Factor'])
    
    return df['Final_Alpha_Factor']
```
