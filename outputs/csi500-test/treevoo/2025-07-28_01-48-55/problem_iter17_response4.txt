```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Price Momentum
    price_momentum = df['close'] - df['open']
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combined Intraday and Price Momentum
    combined_intraday_price_momentum = intraday_reversal + price_momentum
    
    # High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    
    # Weighted Moving Average (WMA)
    weights = np.arange(1, 11)
    wma = df['close'].rolling(window=10).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Price Momentum (Adjusted)
    price_momentum_adjusted = df['close'] - wma
    
    # Volume Adjusted Spread
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_adjusted_spread = high_low_range * relative_volume
    
    # Momentum Factor
    adjusted_high_low_range = high_low_range * np.sqrt(df['volume'])
    momentum_factor = adjusted_high_low_range.rolling(window=10).sum()
    
    # Weighted Price Change
    weighted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Range Volatility
    range_volatility = high_low_range.rolling(window=10).mean()
    
    # Price Oscillator
    sma_5 = df['close'].rolling(window=5).mean()
    price_oscillator = sma_5 - wma
    
    # Composite Intermediate Factor
    composite_intermediate_factor = combined_intraday_price_momentum + volume_adjusted_high_low_range - volume_change
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum_adjusted
        - volume_adjusted_spread
        + momentum_factor
        + weighted_price_change
        - range_volatility
        + price_oscillator
        + composite_intermediate_factor
    )
    
    return final_alpha_factor
```
