```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Moving Average (WMA)
    weights = [i + 1 for i in range(15)]
    df['WMA'] = df['close'].rolling(window=15).apply(lambda x: sum(w * p for w, p in zip(weights, x)) / sum(weights), raw=True)

    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['WMA']

    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['AvgVolume30'] = df['volume'].rolling(window=30).mean()
    df['RelativeVolume'] = df['volume'] / df['AvgVolume30']
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']

    # Calculate Trend Strength
    df['AbsDiffClose'] = df['close'].diff().abs()
    df['TrendStrength'] = df['AbsDiffClose'].rolling(window=5).sum() - df['WMA'].rolling(window=5).mean()

    # Calculate Daily Log Return
    df['DailyLogReturn'] = df['close'].pct_change()

    # Calculate Volume Change
    df['VolumeChange'] = df['volume'].diff()

    # Integrate Log Return with Volume Change
    df['IntegratedMomentum'] = (df['DailyLogReturn'] * df['VolumeChange']).rolling(window=10).sum()

    # Enhance with Moving Average Crossover
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['CrossoverSignal'] = (df['EMA_5'] > df['EMA_20']) * 1 - (df['EMA_5'] < df['EMA_20']) * 1
    df['EnhancedMomentum'] = df['IntegratedMomentum'] + df['CrossoverSignal']

    # Incorporate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    df['NormalizedHighLowRange'] = df['HighLowRange'] / df['close']
    df['EnhancedMomentum'] = df['EnhancedMomentum'] * df['NormalizedHighLowRange']

    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']

    # Calculate Enhanced Volume Thrust
    df['UpwardThrust'] = 0.0
    df['DownwardThrust'] = 0.0
    df.loc[df['close'] > df['open'], 'UpwardThrust'] = df['volume'] * (df['close'] - df['open'])
    df.loc[df['close'] <= df['open'], 'DownwardThrust'] = df['volume'] * (df['open'] - df['close'])
    df['PriceRangeFactor'] = (df['high'] - df['low']) / df['open']
    df['EnhancedVolumeThrust'] = df['UpwardThrust'] - df['DownwardThrust'] + df['PriceRangeFactor']

    # Rolling Average of Daily Log Return
    df['RollingAvgLogReturn'] = df['DailyLogReturn'].rolling(window=20).mean()

    # Final Integration
    df['FinalAlphaFactor'] = (
        df['IntradayMomentum'] +
        df['EnhancedVolumeThrust'] -
        df['VolumeAdjustedSpread'] -
        df['TrendStrength'] -
        df['RollingAvgLogReturn'] +
        df['EnhancedMomentum']
    )

    return df['FinalAlphaFactor']
```
