```python
def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Composite: Intraday Reversal * Volume Change
    composite = intraday_reversal * volume_change
    
    return composite
```
