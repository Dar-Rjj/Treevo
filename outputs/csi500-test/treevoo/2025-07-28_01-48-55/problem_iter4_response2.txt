```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']

    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])

    # Calculate Momentum Factor using a rolling sum
    momentum_factor = volume_adjusted_high_low_range.rolling(window=10).sum()

    return momentum_factor
```
