```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Reversal: (High - Low) - (Open - Close)
    intraday_reversal = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Price Momentum: Close - Open
    price_momentum = df['close'] - df['open']
    
    # Combine Intraday Reversal and Price Momentum
    combined_momentum = intraday_reversal + price_momentum
    
    # Volume Trend: Today’s Volume - Yesterday’s Volume
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Adjust Combined Momentum by Volume
    adjusted_momentum = combined_momentum * volume_trend
    
    return adjusted_momentum
```
