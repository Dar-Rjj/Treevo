```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) over a 10-day window
    df['SMA'] = df['close'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'] - df['SMA']
    
    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Spread
    df['Spread'] = df['high'] - df['low']
    df['AverageVolume'] = df['volume'].rolling(window=10).mean()
    df['RelativeVolume'] = df['volume'] / df['AverageVolume']
    df['VolumeAdjustedSpread'] = df['Spread'] * df['RelativeVolume']
    
    # Calculate Volume Thrust
    df['UpwardThrust'] = 0
    df['DownwardThrust'] = 0
    df.loc[df['close'] > df['open'], 'UpwardThrust'] = df['volume']
    df.loc[df['close'] <= df['open'], 'DownwardThrust'] = df['volume']
    df['VolumeThrust'] = df['UpwardThrust'] - df['DownwardThrust']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['PriceMomentum'] + 
                              df['IntradayMomentum'] - 
                              df['VolumeAdjustedSpread'] + 
                              df['VolumeThrust'])
    
    return df['FinalAlphaFactor']
```
