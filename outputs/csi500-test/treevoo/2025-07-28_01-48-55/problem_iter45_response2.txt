```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate simple price movements
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    df['opening_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Examine trading activity
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    
    # Analyze price patterns
    df['price_range'] = df['high'] - df['low']
    df['price_to_open_ratio'] = (df['close'] - df['open']) / df['open']
    df['price_to_close_ratio'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Investigate momentum
    n = 5  # Example window size
    df['sma_close_n'] = df['close'].rolling(window=n).mean()
    df['moving_average_difference'] = (df['sma_close_n'] - df['sma_close_n'].shift(1)) / df['sma_close_n'].shift(1)
    df['relative_strength'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    df['momentum_oscillator'] = (df['close'] - df['sma_close_n']) / df['sma_close_n']
    
    # Incorporate volatility
    df['volatility'] = df['daily_return'].rolling(window=n).std()
    df['range_volatility'] = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    
    # Consider volume-weighted indicators
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['volume_adjusted_return'] = (df['close'] - df['close'].shift(1)) * df['volume'] / df['volume'].shift(1)
    df['volume_weighted_momentum'] = (df['vwap'] - df['vwap'].shift(1)) / df['vwap'].shift(1)
    
    # Explore trend-following strategies
    m = 10  # Another example window size
    df['ema_close_n'] = df['close'].ewm(span=n, adjust=False).mean()
    df['ema_close_m'] = df['close'].ewm(span=m, adjust=False).mean()
    df['moving_average_crossover'] = np.sign(df['ema_close_n'] - df['ema_close_m'])
    df['ema_slope'] = (df['ema_close_n'] - df['ema_close_n'].shift(1)) / df['ema_close_n'].shift(1)
    df['trend_strength'] = (df['ema_close_n'] - df['ema_close_m']) / df['ema_close_m']
    
    # Combine multiple factors
    df['composite_factor'] = (df['daily_return'] + df['intraday_return'] + df['momentum_oscillator']) / 3
    df['factor_combination'] = (df['daily_return'] + df['intraday_return'] + df['momentum_oscillator'] + df['volume_weighted_momentum']) / 4
    
    # Select the final alpha factor
    alpha_factor = df['factor_combination']
    
    return alpha_factor
```
