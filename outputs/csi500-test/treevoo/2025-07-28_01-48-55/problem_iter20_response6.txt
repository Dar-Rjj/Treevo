```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the amount per volume, which is a measure of average transaction size
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Construct a novel alpha factor by combining the three calculated features
    # This factor aims to capture the momentum and volatility, as well as the transaction size
    df['alpha_factor'] = (df['close_change'] * df['high_low_ratio']) + df['amount_per_volume']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
