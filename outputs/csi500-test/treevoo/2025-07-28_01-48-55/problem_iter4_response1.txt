```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * np.log(df['volume'])
    
    # Calculate Momentum Factor using a rolling sum with a window size of 20
    window_size = 20
    momentum_factor = volume_adjusted_high_low_range.rolling(window=window_size).sum()
    
    return momentum_factor
```
