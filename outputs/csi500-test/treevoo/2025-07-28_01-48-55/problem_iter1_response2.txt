```python
def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    
    price_momentum = df['close'] - df['open']
    
    # Avoid division by zero
    high_low_range[high_low_range == 0] = 1e-10
    
    adjusted_momentum = price_momentum / high_low_range
    
    # Adjust by volume
    factor_values = adjusted_momentum * df['volume']
    
    return factor_values
```
