```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the percentage of the bar (close - open) / (high - low)
    df['bar_percentage'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the amount change from the previous day
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Combine the above factors into a single alpha factor
    df['alpha_factor'] = (df['bar_percentage'] * df['volume_change']) + (df['price_change'] * df['amount_change'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
