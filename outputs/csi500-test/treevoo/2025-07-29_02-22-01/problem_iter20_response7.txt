```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the ratio of close to open prices
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of volume to amount
    vol_amount_ratio = df['volume'] / df['amount']
    
    # Combine the three signals into a single alpha factor
    alpha_factor = (price_diff * close_open_ratio * vol_amount_ratio).rank(pct=True)
    
    return alpha_factor
```
