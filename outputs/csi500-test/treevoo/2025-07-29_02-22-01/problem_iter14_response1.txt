def heuristics_v2(data):
    # Calculate the relative difference between close and open prices
    rel_diff_close_open = (data['close'] - data['open']) / data['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (data['high'] + data['low']) / 2
    
    # Calculate the weighted amount by volume
    weighted_amount = data['amount'] * data['volume']
    
    # Generate a novel alpha factor
    alpha_factor = (rel_diff_close_open * avg_high_low) + weighted_amount
    
    return alpha_factor
