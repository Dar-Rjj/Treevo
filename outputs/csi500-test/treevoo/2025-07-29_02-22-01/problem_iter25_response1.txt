```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, min_periods=10).mean()
    long_ema = df['close'].ewm(span=50, min_periods=50).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = ((df['high'] - df['low']) / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    daily_intraday_range = df['high'] - df['low']
    volume_weighted_adjusted_return = (df['volume'] * daily_intraday_range * intraday_return) / df['volume'] * daily_intraday_range.rolling(window=20).sum()
    
    # Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = (short_ema - long_ema) * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume'].rolling(window=1).sum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Volume-Adjusted Metrics and Price Momentum
    volume_adjusted_metrics = (df['volume'] * (df['high'] - df['low'])).rolling(window=10).mean() + (df['volume'] * intraday_return).rolling(window=10).mean()
    
    # Final Alpha Factor
    alpha_factor = (
        price_momentum +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        combined_momentum_volatility +
        vwap +
        rsi
    )
    
    return alpha_factor
```
