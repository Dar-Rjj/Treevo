import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of the close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the ratio of the closing price to the average of the high and low prices
    df['close_avg_ratio'] = df['close'] / (df['high'] + df['low']) * 2
    
    # Compute the ratio of amount to volume to get a sense of the dollar volume per share
    df['dollar_volume_per_share'] = df['amount'] / df['volume']
    
    # Calculate the momentum factor by looking at the return over the last 5 days
    df['momentum_5_days'] = df['close'].pct_change(periods=5)
    
    # The alpha factor is a combination of the above calculated factors
    df['alpha_factor'] = (df['close_avg_ratio'] * df['momentum_5_days']) / df['dollar_volume_per_share']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
