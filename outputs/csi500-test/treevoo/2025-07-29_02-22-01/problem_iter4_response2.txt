```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Apply exponential weights to returns
    decay_factor = 0.94
    weighted_returns = df['daily_return'].ewm(alpha=1-decay_factor, adjust=False).mean()
    
    # Sum of weighted returns over a window
    window_size = 15
    momentum = weighted_returns.rolling(window=window_size).sum()
    
    # Aggregate volume of the past N days
    aggregate_volume = df['volume'].rolling(window=window_size).sum()
    
    # Volume adjusted momentum
    volume_adjusted_momentum = momentum / aggregate_volume
    
    # Output the factor values
    return volume_adjusted_momentum
```
