```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    df['High_Low_Spread'] = (df['high'] - df['low']) * df['volume']
    df['10Day_High_Low_Sum'] = df['High_Low_Spread'].rolling(window=10).sum()
    
    # Calculate Short-Term and Long-Term EMAs
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Short_Long_EMA_Diff'] = df['EMA_20'] - df['EMA_5']
    
    # Daily Price Change
    df['Daily_Price_Change'] = df['close'].pct_change()
    
    # Cumulative Daily Price Change
    df['Cumulative_Price_Change'] = df['Daily_Price_Change'].rolling(window=14).sum()
    
    # Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    df['Weighted_Cumulative_Price_Change'] = df['Cumulative_Price_Change'] * df['Average_Volume']
    
    # Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Volume Trend
    df['Volume_Trend'] = df['volume'].rolling(window=10).mean() - df['volume']
    
    # Assess Intraday Volume Impact
    df['Relative_Volume_Impact'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine Metrics for Volatility
    df['Volatility_Metric'] = (df['high'] - df['low']) * df['Volume_Trend'] + df['Daily_Price_Change']
    
    # Measure Sequential Momentum
    df['Sequential_Momentum'] = df['Volatility_Metric'].rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Integrate Intraday and Historical Effects
    df['Intraday_Historical_Effect'] = df['Intraday_Return'] * df['Relative_Volume_Impact'] * df['Sequential_Momentum']
    
    # Proportional Movement Index
    df['Proportional_Movement_Index'] = (df['Intraday_Return'] / df['close']) * df['Intraday_Return']
    
    # Volume-Weighted Adjusted Return
    df['Volume_Adjusted_Return'] = df['volume'] * df['Intraday_Return']
    
    # Volume Reversal Indicator
    df['Volume_Rate_of_Change'] = df['volume'].pct_change(periods=10)
    df['Smoothed_Volume_Rate_of_Change'] = df['Volume_Rate_of_Change'].rolling(window=5).mean()
    df['Volume_Reversal_Indicator'] = df['Weighted_Cumulative_Price_Change'] / (1 + df['Smoothed_Volume_Rate_of_Change'])
    
    # Volume Confirmation
    df['Daily_Volume_Change'] = df['volume'].diff()
    df['Average_Volume_Increase'] = df['Daily_Volume_Change'].rolling(window=10).apply(lambda x: sum(x[x > 0]) / x.count(), raw=True)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['10Day_High_Low_Sum'] + 
        df['Cumulative_Price_Change'] + 
        df['Weighted_Cumulative_Price_Change'] + 
        df['Proportional_Movement_Index'] + 
        df['Volume_Adjusted_Return'] + 
        df['Volume_Reversal_Indicator'] + 
        df['Intraday_Return'] + 
        df['Intraday_Historical_Effect'] + 
        df['Average_Volume_Increase'] * df['Weighted_Cumulative_Price_Change']
    )
    
    return df['Final_Alpha_Factor']
```
