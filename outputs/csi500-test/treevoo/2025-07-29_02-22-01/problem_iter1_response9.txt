```python
import pandas as pd

def heuristics_v2(df, period=5):
    # Calculate Intraday Movement Ratio
    intraday_movement_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] * df['close']
    
    # Combine Intraday Movement and Volume
    combined_intraday_mv = intraday_movement_ratio * intraday_volume_intensity
    
    # Calculate Cumulative Intraday Momentum over a specified period
    cumulative_intraday_momentum = combined_intraday_mv.rolling(window=period).sum()
    
    return cumulative_intraday_momentum
```
