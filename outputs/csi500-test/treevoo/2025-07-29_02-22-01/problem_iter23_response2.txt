```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=10).mean()

    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema

    # Daily Price Change
    daily_price_change = df['close'].diff()

    # Momentum Calculation (Sum of Daily Price Changes over the last N days)
    momentum = daily_price_change.rolling(window=10).sum()

    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_days = (daily_volume_change > 0).sum()
    avg_volume_increase = daily_volume_change[daily_volume_change > 0].sum() / positive_volume_days

    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change()
    smoothed_volume_roc = volume_rate_of_change.rolling(window=3).mean()
    volume_reversal_indicator = momentum * (1 / smoothed_volume_roc)

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    volume_weighted_adj_return = df['volume'] * intraday_return

    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return

    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']

    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()

    # Combine Volume-Adjusted Price Change and Price Momentum
    combined_momentum = ma_volume_adjusted_daily_price_change + price_momentum

    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=10).mean()

    # Weighted Combined Momentum
    weighted_combined_momentum = combined_momentum * average_volume

    # Incorporate High-Low Range
    daily_high_low_range = df['high'] - df['low']
    final_alpha_factor = weighted_combined_momentum / daily_high_low_range

    return final_alpha_factor
```
