import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_10_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = short_term_ema - long_term_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_volume_adjusted_high_low_spread_price_momentum = ma_10_volume_adjusted_high_low_spread + price_momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    intraday_price_range = df['high'] - df['low']
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Adjusted Intraday Momentum
    adjusted_intraday_momentum = intraday_return * np.sqrt(df['volume'])
    
    # Combine Factors
    final_factor = (combined_volume_adjusted_high_low_spread_price_momentum 
                    + proportional_movement_index 
                    + adjusted_intraday_momentum * np.log(df['volume']))
    
    return final_factor
