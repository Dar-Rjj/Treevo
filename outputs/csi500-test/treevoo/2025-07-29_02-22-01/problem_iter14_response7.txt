```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    momentum = sma_5 - sma_20

    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()

    # Enhance with Volume-Weighted Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Incorporate Volume Reversal Indicator
    volume_roc = df['volume'] / df['volume'].shift(1) - 1
    smoothed_volume_roc = volume_roc.rolling(window=5).mean()
    inverse_smoothed_volume_roc = 1 / smoothed_volume_roc

    # Adjust Final Alpha Factor
    adjusted_factor = (momentum / volatility + vwap) * inverse_smoothed_volume_roc

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    volume_weighted_return = df['volume'] * intraday_return

    # Calculate Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close']

    # Combine All Factors
    combined_factors = (proportional_movement_index + volume_weighted_return 
                        + momentum / volatility + vwap)

    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']

    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_10_vol_adj_price_change = volume_adjusted_price_change.rolling(window=10).mean()

    # Calculate Price Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = ema_20 - ema_5

    # Combine Volume-Adjusted Price Change and Price Momentum
    combined_price_momentum = ma_10_vol_adj_price_change + price_momentum

    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=20).mean()

    # Weighted Combined Momentum
    weighted_combined_momentum = combined_price_momentum * average_volume

    # Incorporate High-Low Range
    high_low_spread = df['high'] - df['low']
    high_low_spread_ratio = high_low_spread / df['close']

    # Final Alpha Factor
    final_alpha_factor = weighted_combined_momentum * high_low_spread_ratio

    return final_alpha_factor
```
