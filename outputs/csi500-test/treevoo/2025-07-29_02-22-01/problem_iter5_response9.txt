import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']

    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']

    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = (volume_ma - df['volume']).abs()

    # Assess Intraday Volume Impact
    avg_volume = df['volume'].rolling(window=10).mean()
    intraday_volume_impact = df['volume'] / avg_volume

    # Combine Metrics for Volatility
    combined_volatility_metric = (high_low_spread * weighted_volume_trend + daily_return).rolling(window=5).mean()

    # Integrate Intraday and Historical Effects
    integrated_effects = intraday_movement * intraday_volume_impact * combined_volatility_metric

    # Measure Sequential Momentum
    sequential_momentum = integrated_effects.rolling(window=5).sum()
    sequential_momentum = sequential_momentum.ewm(span=3, adjust=False).mean()

    # Final Alpha Factor
    # Calculate Price Momentum
    daily_price_change = df['close'].pct_change()
    price_momentum = daily_price_change.rolling(window=5).sum()

    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_days = (daily_volume_change > 0).astype(int)
    average_volume_increase = (daily_volume_change[daily_volume_change > 0]).rolling(window=5).mean()

    # Multiply Sequential Momentum by Integrated Intraday and Historical Effects
    final_alpha = sequential_momentum * integrated_effects

    # Adjust by Volume Confirmation
    final_alpha = final_alpha * (average_volume_increase > 0).replace({True: 1, False: 0.9})

    return final_alpha
