```python
import pandas as pd

def heuristics_v2(df, n_days=5):
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Daily Price Change over N Days
    cumulative_daily_price_change = daily_price_change.rolling(window=n_days).sum()
    
    # Calculate Average Volume over N Days
    average_volume = df['volume'].rolling(window=n_days).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    normalized_high_low_range = high_low_range / df['close']
    adjusted_cumulative_momentum = normalized_high_low_range * weighted_cumulative_momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * intraday_return
    
    # Combine Factors
    alpha_factor = adjusted_cumulative_momentum + proportional_movement_index + volume_weighted_intraday_return
    
    return alpha_factor
```
