import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement Ratio
    intraday_movement_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].diff()
    
    # Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['close']
    
    # Combine Intraday and Daily Momentum
    combined_momentum = intraday_movement_ratio * daily_return * intraday_volume_intensity
    
    # Weighted Moving Average of Combined Momentum
    decay_factor = 0.94
    window_size = 15
    wma_combined_momentum = combined_momentum.ewm(span=window_size, adjust=False).mean()
    
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = short_ema - long_ema
    
    # Calculate Intraday Momentum Swing
    intraday_price_range = df['high'] - df['low']
    intraday_return = df['close'] - df['open']
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    intraday_momentum_swing = proportional_movement_index + volume_weighted_adjusted_return
    
    # Combine Factors
    total_momentum = price_momentum + intraday_momentum_swing + wma_combined_momentum
    
    # Volume Adjusted Crossover Momentum
    n_days = 15
    aggregate_volume = df['volume'].rolling(window=n_days).sum()
    volume_adjusted_crossover_momentum = total_momentum / aggregate_volume
    
    return volume_adjusted_crossover_momentum
