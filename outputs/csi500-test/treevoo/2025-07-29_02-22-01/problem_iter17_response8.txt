```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Averages (EMA)
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Momentum Component
    momentum = ema_5 - ema_20

    # Calculate Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()

    # Volume-Adjusted Price and Momentum
    vwap = ((df['high'] + df['low'] + 2 * df['close']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()
    smoothed_vwap = vwap.ewm(span=10, adjust=False).mean()
    adjusted_momentum = momentum / smoothed_vwap

    # Incorporate Volume Reversal Indicator
    volume_rate_of_change = df['volume'] / df['volume'].shift(1) - 1
    smoothed_volume_roc = volume_rate_of_change.ewm(span=5, adjust=False).mean()
    final_alpha_factor = adjusted_momentum / smoothed_volume_roc

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Volume-Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return

    # Calculate Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['open']

    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']

    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).sum()

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_high_low_spread_momentum = moving_sum_high_low_spread * adjusted_momentum

    # Calculate Cumulative Daily Price Change
    cumulative_price_change = df['close'].pct_change().rolling(window=14).sum()

    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=14).mean()

    # Weighted Cumulative Daily Price Change
    weighted_cumulative_price_change = cumulative_price_change * average_volume

    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    normalized_high_low_range = high_low_range / df['close']
    adjusted_cumulative_momentum = normalized_high_low_range * cumulative_price_change

    # Combine All Factors
    alpha_factor = (proportional_movement_index +
                    volume_weighted_adjusted_return +
                    combined_high_low_spread_momentum +
                    weighted_cumulative_price_change +
                    adjusted_cumulative_momentum)

    return alpha_factor
```
