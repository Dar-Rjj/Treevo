```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Calculate Cumulative Daily Price Change over N days
    N = 5  # Set lookback window
    df['cumulative_daily_price_change'] = df['daily_price_change'].rolling(window=N).sum()

    # Calculate Average Volume over N days
    df['average_volume'] = df['volume'].rolling(window=N).mean()

    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_momentum_impact'] = df['cumulative_daily_price_change'] * df['average_volume']

    # Incorporate High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    df['normalized_high_low_range'] = df['daily_high_low_range'] / df['close']
    df['adjusted_cumulative_momentum'] = df['weighted_cumulative_momentum_impact'] * df['normalized_high_low_range']

    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['intraday_return']

    # Calculate Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_price_range'] / df['close']) * df['intraday_return']

    # Combine Factors
    df['final_alpha_factor'] = df['adjusted_cumulative_momentum'] + df['proportional_movement_index'] + df['volume_weighted_adjusted_return']

    return df['final_alpha_factor']
```
