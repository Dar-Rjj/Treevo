```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']

    # Define weights for the last 10 days, with higher weights for more recent days
    weights = np.arange(1, 11)  # Weights: 1, 2, 3, ..., 10
    weights = weights / weights.sum()  # Normalize to sum to 1

    # Calculate 10-Day Weighted Moving Sum of Volume-Adjusted High-Low Spread
    weighted_moving_sum = volume_adjusted_high_low_spread.rolling(window=10).apply(lambda x: (x * weights).sum(), raw=False)

    # Calculate Price Momentum using Short-Term and Long-Term EMA
    short_ema_span = 5
    long_ema_span = 20
    short_ema = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    long_ema = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    price_momentum = long_ema - short_ema

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    factor_values = weighted_moving_sum * price_momentum

    return factor_values
```
