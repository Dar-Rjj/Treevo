```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term and Long-Term EMA of Close Prices
    short_span = 10
    long_span = 50
    df['short_ema'] = df['close'].ewm(span=short_span, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=long_span, adjust=False).mean()

    # Price Momentum: Subtract Short-Term EMA from Long-Term EMA
    df['price_momentum'] = df['long_ema'] - df['short_ema']

    # Intraday Price Range: Subtract Low from High
    df['intraday_range'] = df['high'] - df['low']

    # Intraday Return: Subtract Open from Close
    df['intraday_return'] = df['close'] - df['open']

    # Volume Weighted Adjusted Return: Multiply Volume by Intraday Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['intraday_return']

    # Proportional Movement Index: Divide Intraday Price Range by Close, then multiply by Intraday Return
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']

    # Combine Factors
    df['alpha_factor'] = df['price_momentum'] + df['proportional_movement_index'] + df['volume_weighted_adjusted_return']

    return df['alpha_factor']
```
