```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    df['momentum'] = df['5_day_SMA'] - df['20_day_SMA']

    # Calculate Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()

    # Combine Momentum and Volatility
    df['momentum_volatility_adjusted'] = df['momentum'] * (1 / df['volatility'])

    # Enhance with Volume-Weighted Price
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    df['enhanced_momentum_volatility'] = df['momentum_volatility_adjusted'] + df['vwap']

    # Incorporate Volume Reversal Indicator
    df['volume_roc'] = df['volume'].pct_change(periods=5)
    df['smooth_volume_roc'] = df['volume_roc'].rolling(window=5).mean()
    df['adjusted_alpha'] = df['enhanced_momentum_volatility'] * (1 / df['smooth_volume_roc'])

    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    df['volume_weighted_return'] = df['volume'] * df['intraday_return']

    # Calculate Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']

    # Combine All Factors
    df['final_alpha'] = df['proportional_movement_index'] + \
                        (df['volume_weighted_return'] * np.log(df['volume'])) + \
                        df['adjusted_alpha']

    return df['final_alpha']
```
