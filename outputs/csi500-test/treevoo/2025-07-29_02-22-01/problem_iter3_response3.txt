```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5, short_window=12, long_window=26, signal_window=9):
    # Calculate Price Momentum
    df['Daily_Price_Change'] = df['close'].diff()
    df['Momentum'] = df['Daily_Price_Change'].rolling(window=N).sum()

    # Volume Confirmation
    df['Daily_Volume_Change'] = df['volume'].diff()
    positive_volume_changes = df['Daily_Volume_Change'].where(df['Daily_Volume_Change'] > 0, 0)
    df['Average_Positive_Volume_Increase'] = positive_volume_changes.rolling(window=M).sum() / M
    df['Count_Days_Positive_Volume'] = (df['Daily_Volume_Change'] > 0).rolling(window=M).sum()

    # Additional Price and Volume Indicators
    # MACD
    df['Short_EMA'] = df['close'].ewm(span=short_window, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=long_window, adjust=False).mean()
    df['MACD_Line'] = df['Short_EMA'] - df['Long_EMA']
    df['Signal_Line'] = df['MACD_Line'].ewm(span=signal_window, adjust=False).mean()

    # On-Balance Volume (OBV)
    df['OBV'] = np.nan
    df.loc[df.index[0], 'OBV'] = 0  # Initialize first OBV
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df.loc[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            df.loc[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] - df['volume'].iloc[i]
        else:
            df.loc[df.index[i], 'OBV'] = df['OBV'].iloc[i-1]

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Momentum'] + df['Average_Positive_Volume_Increase'] + 
                          (df['MACD_Line'] - df['Signal_Line']) + df['OBV'])

    return df['Alpha_Factor']
```
