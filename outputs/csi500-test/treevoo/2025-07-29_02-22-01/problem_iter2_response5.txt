```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low = high_low_spread * df['volume']

    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_10_day = volume_adjusted_high_low.rolling(window=10).sum()

    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = short_term_ema - long_term_ema

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    factor_values = moving_sum_10_day * price_momentum

    return factor_values
```
