```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Set Lookback Window (N) for Cumulative and Average Volume Calculations
    N = 20
    # Calculate Cumulative Daily Price Change
    df['cumulative_daily_price_change'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_daily_price_change'] = df['cumulative_daily_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['daily_high_low_range'] = df['high'] - df['low']
    df['momentum_impact_with_volatility'] = df['weighted_cumulative_daily_price_change'] * df['daily_high_low_range']
    
    # Calculate Short-Term and Long-Term EMA of Close Prices for Price Momentum
    short_span, long_span = 5, 30
    df['short_term_ema'] = df['close'].ewm(span=short_span, adjust=False).mean()
    df['long_term_ema'] = df['close'].ewm(span=long_span, adjust=False).mean()
    df['price_momentum'] = df['short_term_ema'] - df['long_term_ema']
    
    # Calculate Intraday Return and Volume-Weighted Adjusted Return for Intraday Momentum
    df['intraday_return'] = df['close'] - df['open']
    df['volume_weighted_adjusted_return'] = df['volume'] * df['intraday_return']
    
    # Synthesize Final Alpha Factor
    df['proportional_movement_index'] = (df['high'] - df['low']) / df['close'] * df['intraday_return']
    df['alpha_factor'] = df['momentum_impact_with_volatility'] + df['price_momentum'] + df['volume_weighted_adjusted_return'] + df['proportional_movement_index']
    
    return df['alpha_factor']
```
