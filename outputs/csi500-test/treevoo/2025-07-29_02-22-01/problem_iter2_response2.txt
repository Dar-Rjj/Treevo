```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    high_low_spread = df['high'] - df['low']
    intraday_reversal = high_low_spread / df['close']

    # Compute Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']

    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_10_day = volume_adjusted_high_low_spread.rolling(window=10).sum()

    # Calculate 1-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_1_day = volume_adjusted_high_low_spread.rolling(window=1).sum()

    # Calculate Difference from 10-Day to 1-Day Moving Sum
    volume_adjusted_momentum = moving_sum_10_day - moving_sum_1_day

    # Combine Factors
    alpha_factor = intraday_reversal + volume_adjusted_momentum

    return alpha_factor
```
