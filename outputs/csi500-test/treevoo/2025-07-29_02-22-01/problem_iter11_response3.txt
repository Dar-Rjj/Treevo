```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Volume Acceleration
    volume_weighted_average = df['volume'].rolling(window=10).mean()
    smoothed_volume_weighted_average = volume_weighted_average.ewm(span=10, adjust=False).mean()
    volume_acceleration = price_momentum * (1 / smoothed_volume_weighted_average)
    
    # High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_factor = moving_sum_high_low_spread * price_momentum
    
    # Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Cumulative Daily Price Change
    cumulative_daily_price_change = daily_price_change.rolling(window=14).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_daily_price_change = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    daily_high_low_range = df['high'] - df['low']
    normalized_high_low_range = daily_high_low_range / df['close']
    adjusted_cumulative_momentum = cumulative_daily_price_change * normalized_high_low_range
    
    # Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + adjusted_cumulative_momentum + proportional_movement_index + volume_weighted_intraday_return
    
    return final_alpha_factor
```
