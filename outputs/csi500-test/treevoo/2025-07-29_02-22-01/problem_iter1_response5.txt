```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Normalize by Close
    intraday_reversal = high_low_spread / df['close']
    
    # Compute Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Multiply by Volume
    volume_weighted_momentum = log_return * df['volume']
    
    # Combine Factors
    factor_values = intraday_reversal + volume_weighted_momentum
    
    return factor_values
```
