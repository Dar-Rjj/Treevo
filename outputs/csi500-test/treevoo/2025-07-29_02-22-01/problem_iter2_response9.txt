```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=14)
    smoothed_volume_roc = volume_rate_of_change.rolling(window=5).mean()
    inv_smoothed_volume_roc = 1 / smoothed_volume_roc
    volume_reversal_indicator = price_momentum * inv_smoothed_volume_roc
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Combine Factors
    combined_factors = proportional_movement_index + price_momentum + volume_reversal_indicator
    
    return combined_factors
```
