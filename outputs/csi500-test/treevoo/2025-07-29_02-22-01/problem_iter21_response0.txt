```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Price Momentum (Short-Term EMA - Long-Term EMA)
    price_momentum = short_term_ema - long_term_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_factor = ma_volume_adjusted_high_low_spread + price_momentum
    
    # Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Momentum Calculation (Sum of Daily Price Changes over the last N days, e.g., 10 days)
    momentum = daily_price_change.rolling(window=10).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_diff = daily_volume_change.where(daily_volume_change > 0, 0)
    avg_volume_increase = positive_volume_diff.rolling(window=10).sum() / (daily_volume_change > 0).rolling(window=10).sum()
    
    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_roc = volume_rate_of_change.rolling(window=5).mean()
    inverse_smoothed_volume_roc = 1 / smoothed_volume_roc
    volume_reversal_indicator = momentum * inverse_smoothed_volume_roc
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adj_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    combined_momentum = ma_volume_adjusted_daily_price_change + momentum
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Weighted Combined Momentum
    weighted_combined_momentum = combined_momentum * average_volume
    
    # Incorporate High-Low Range
    final_alpha_factor = weighted_combined_momentum / (df['high'] - df['low'])
    
    # Calculate Intraday Volatility (True Range)
    true_range = pd.DataFrame({
        'high-low': df['high'] - df['low'],
        'high-close_prev': abs(df['high'] - df['close'].shift(1)),
        'low-close_prev': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    
    # Calculate 10-Day Moving Average of Intraday Volatility
    ma_intraday_volatility = true_range.rolling(window=10).mean()
    
    # Incorporate Intraday Volatility into Final Alpha Factor
    final_adjusted_alpha_factor = final_alpha_factor / ma_intraday_volatility
    
    return final_adjusted_alpha_factor
```
