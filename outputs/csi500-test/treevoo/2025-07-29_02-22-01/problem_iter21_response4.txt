```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Momentum Swing
    intraday_return = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_adjusted_return = (df['volume'] * intraday_return) / avg_volume
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_avg_high_low_spread = (volume_adjusted_high_low_spread.rolling(window=10).sum() 
                                           / df['volume'].rolling(window=10).sum())
    
    # Calculate Volume-Adjusted Daily Price Change
    daily_price_change = df['close'].diff(1)
    volume_adjusted_daily_price_change = daily_price_change * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    moving_avg_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Proportional Movement Index
    proportional_movement_index = (high_low_spread / df['close']) * intraday_return
    
    # Combine Factors
    combined_factors = (price_momentum + 
                        proportional_movement_index + 
                        volume_weighted_adjusted_return)
    
    # Calculate High-Low Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility_ema = high_low_range.ewm(span=10, adjust=False).mean()
    high_low_volatility = high_low_volatility_ema / df['close']
    
    # Combine Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    combined_metrics = (volume_weighted_avg_high_low_spread + 
                       moving_avg_volume_adjusted_daily_price_change + 
                       price_momentum + 
                       high_low_volatility)
    
    # Incorporate High-Low Range
    final_factor = high_low_range * combined_metrics
    
    return final_factor
```
