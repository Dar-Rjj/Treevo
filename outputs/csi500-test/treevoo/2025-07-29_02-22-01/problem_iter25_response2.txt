```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low_spread'] = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    df['10_day_moving_sum_vol_adj_hl'] = df['volume_adjusted_high_low_spread'].rolling(window=10).sum()
    
    # Short-Term and Long-Term EMAs
    df['short_term_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Cumulative Daily Price Change over N Days (N=14)
    df['cumulative_daily_price_change'] = df['daily_price_change'].rolling(window=14).sum()
    
    # Average Volume over N Days (N=14)
    df['average_volume'] = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_daily_price_change'] * df['average_volume']
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    df['combined_vol_adj_hl_and_momentum'] = df['10_day_moving_sum_vol_adj_hl'] + df['cumulative_daily_price_change']
    
    # Volume-Weighted Adjusted Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_return'] = df['volume'] * df['intraday_return']
    
    # Volume Reversal Indicator
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=10)
    df['smoothed_vol_rate_of_change'] = df['volume_rate_of_change'].rolling(window=5).mean()
    df['volume_reversal_indicator'] = df['cumulative_daily_price_change'] / df['smoothed_vol_rate_of_change'].replace(0, np.nan)
    
    # Volume Confirmation
    df['daily_volume_change'] = df['volume'].diff()
    df['positive_volume_diff'] = df['daily_volume_change'].apply(lambda x: x if x > 0 else 0)
    df['positive_volume_days'] = df['daily_volume_change'].apply(lambda x: 1 if x > 0 else 0)
    df['avg_vol_increase_last_10_days'] = df['positive_volume_diff'].rolling(window=10).sum() / df['positive_volume_days'].rolling(window=10).sum().replace(0, np.nan)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['combined_vol_adj_hl_and_momentum'] +
        df['weighted_cumulative_price_change'] +
        df['volume_weighted_return'] +
        df['volume_reversal_indicator'] +
        df['avg_vol_increase_last_10_days']
    )
    
    return df['alpha_factor']
```
