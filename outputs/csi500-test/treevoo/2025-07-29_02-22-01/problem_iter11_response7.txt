```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = abs(volume_ma - df['volume'])
    
    # Assess Intraday Volume Impact
    average_volume = df['volume'].rolling(window=10).mean()
    volume_impact = df['volume'] / average_volume
    
    # Combine Metrics for Volatility
    combined_volatility_metric = high_low_spread * weighted_volume_trend + daily_return
    combined_volatility_sma = combined_volatility_metric.rolling(window=5).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_effects = (intraday_price_movement + volume_impact) * combined_volatility_sma
    
    # Measure Sequential Momentum
    sequential_momentum = integrated_effects.rolling(window=5).sum()
    smoothed_sequential_momentum = sequential_momentum.ewm(span=3, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_difference = short_ema - long_ema
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Final Alpha Factor
    price_momentum = df['close'].diff().rolling(window=5).sum()
    volume_change = df['volume'].diff()
    positive_volume_days = (volume_change > 0).sum()
    volume_confirmation = volume_change.rolling(window=10).apply(lambda x: x[x > 0].sum(), raw=True)
    
    final_alpha_factor = (
        smoothed_sequential_momentum * integrated_effects +
        ema_difference + 
        proportional_movement_index + 
        volume_weighted_adjusted_return
    )
    
    # Adjust by Volume Confirmation
    if volume_confirmation < 0:
        final_alpha_factor *= 0.9  # Reduce factor by 10% if volume confirmation is negative
    
    return final_alpha_factor
```
