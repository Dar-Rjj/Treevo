```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement Ratio
    df['intraday_movement_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['volume'] / df['close']
    
    # Combine Intraday and Daily Momentum
    df['combined_momentum'] = df['intraday_movement_ratio'] * df['daily_return'] * df['intraday_volume_intensity']
    
    # Weighted Moving Average of Combined Momentum
    decay_factor = 0.94
    window_size = 15
    weights = np.array([decay_factor**i for i in range(window_size)])
    df['wma_combined_momentum'] = df['combined_momentum'].rolling(window=window_size).apply(lambda x: np.sum(x * weights), raw=True)
    
    # Volume Adjusted Crossover Momentum
    n_days = 15
    df['agg_volume'] = df['volume'].rolling(window=n_days).sum()
    df['volume_adjusted_momentum'] = df['wma_combined_momentum'] / df['agg_volume']
    
    # Calculate Price Momentum
    short_span = 10
    long_span = 30
    df['short_ema_close'] = df['close'].ewm(span=short_span, adjust=False).mean()
    df['long_ema_close'] = df['close'].ewm(span=long_span, adjust=False).mean()
    df['price_momentum'] = df['short_ema_close'] - df['long_ema_close']
    
    # Volume Reversal Indicator
    roc_period = 10
    sma_period = 5
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=roc_period)
    df['smoothed_volume_roc'] = df['volume_rate_of_change'].rolling(window=sma_period).mean()
    df['adjusted_price_momentum'] = df['price_momentum'] / df['smoothed_volume_roc']
    
    return df['adjusted_price_momentum']
```
