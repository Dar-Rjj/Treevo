```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low_spread = high_low_spread.rolling(window=10).sum()
    
    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Daily Price Change
    daily_price_change = df['close'].pct_change()
    
    # Cumulative Daily Price Change over 14 days
    cumulative_daily_price_change = daily_price_change.rolling(window=14).sum()
    
    # Average Volume over 14 days
    average_volume = df['volume'].rolling(window=14).mean()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_momentum = moving_sum_high_low_spread + (short_term_ema - long_term_ema)
    
    # Volume-Weighted Adjusted Return
    volume_weighted_return = df['volume'] * (df['close'] / df['open'] - 1)
    
    # Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = (short_term_ema - long_term_ema) / (1 + smoothed_volume_rate_of_change)
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_days = (daily_volume_change > 0).astype(int)
    volume_increase_over_last_10_days = (daily_volume_change[daily_volume_change > 0]).rolling(window=10).sum() / positive_volume_days.rolling(window=10).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_momentum + 
        cumulative_daily_price_change + 
        volume_weighted_return + 
        volume_reversal_indicator + 
        (df['close'] / df['open'] - 1) + 
        volume_increase_over_last_10_days * cumulative_daily_price_change
    )
    
    return final_alpha_factor
```
