```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Daily Price Change
    df['vol_adj_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['vol_adj_ma_10'] = df['vol_adj_price_change'].rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = long_ema - short_ema
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    df['combined_momentum'] = df['vol_adj_ma_10'] + df['price_momentum']
    
    # Calculate Average Volume
    n_days = 30  # Set lookback window
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Weighted Combined Momentum
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['avg_volume']
    
    # Incorporate High-Low Range
    df['high_low_spread'] = df['high'] - df['low']
    df['high_low_ratio'] = df['high_low_spread'] / df['close']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_combined_momentum'] * df['high_low_ratio']
    
    return df['alpha_factor']
```
