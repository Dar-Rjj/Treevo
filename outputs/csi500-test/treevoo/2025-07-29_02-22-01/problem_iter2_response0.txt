```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema_span = 12
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    long_ema_span = 26
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['long_ema'] - df['short_ema']
    
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_return'] = df['volume'] * df['intraday_return']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']
    
    # Combine Factors
    df['combined_factor'] = df['price_momentum'] + df['proportional_movement_index'] + df['volume_weighted_return']
    
    return df['combined_factor']
```
