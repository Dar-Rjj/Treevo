```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / daily_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / daily_range
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the relative strength
    relative_strength = (df['close'] - vwap) / vwap
    
    # Calculate the heuristic factor
    heuristic_factor = (buying_pressure - selling_pressure) * relative_strength
    
    return heuristic_factor
```
