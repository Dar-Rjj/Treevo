```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Exponentially Weighted Moving Average (EWMA) of Log Returns
    ewma_log_returns = df['log_return'].ewm(alpha=0.06, adjust=False).mean()  # Decay factor 0.94 corresponds to alpha = 1 - 0.94 = 0.06
    
    # Volume Adjusted Momentum
    N = 15  # Define window size
    df['volume_rolling_sum'] = df['volume'].rolling(window=N).sum()
    volume_adjusted_momentum = ewma_log_returns / df['volume_rolling_sum']
    
    # Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Weighted High-Low Range
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    volume_weighted_high_low_range = df['volume_weighted_high_low_range'].rolling(window=N).sum()
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_momentum + volume_weighted_high_low_range
    
    return alpha_factor.dropna()
```
