```python
import pandas as pd

def heuristics_v2(df, short_span=10, long_span=30, m_days=10, n_days=20):
    # Compute Short-Term and Long-Term SMA of Close Prices
    df['SMA_short'] = df['close'].rolling(window=short_span).mean()
    df['SMA_long'] = df['close'].rolling(window=long_span).mean()
    
    # Subtract Short-Term SMA from Long-Term SMA
    df['price_momentum'] = df['SMA_long'] - df['SMA_short']
    
    # Daily Volume Change
    df['volume_change'] = df['volume'].diff(1)
    df['volume_change'] = df['volume_change'].fillna(0)  # Handle NaN for first day
    
    # Average Volume Increase over the Last M Days
    df['positive_volume'] = df['volume_change'].apply(lambda x: x if x > 0 else 0)
    df['volume_increase_avg'] = (df['positive_volume'].rolling(window=m_days).sum() / 
                                 (df['volume_change'] > 0).rolling(window=m_days).sum())
    df['volume_increase_avg'] = df['volume_increase_avg'].fillna(0)  # Handle NaN for initial days
    
    # Calculate Cumulative Daily Price Change
    df['daily_price_change'] = df['close'].pct_change()
    df['cumulative_price_change'] = df['daily_price_change'].rolling(window=n_days).sum()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_cumulative_price_change'] * df['high_low_range']
    
    return df['alpha_factor']
```
