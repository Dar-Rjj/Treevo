import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust by Open Price to get Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] - volume_10d_ma).abs()
    
    # Assess Intraday Volume Impact
    volume_weight = df['volume'] / volume_10d_ma
    
    # Combine Metrics for Volatility
    combined_metric = (high_low_spread * volume_trend) + daily_return
    combined_sma = combined_metric.rolling(window=5).mean()
    
    # Measure Sequential Momentum
    sequential_momentum = combined_sma.ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    price_momentum = df['close'].pct_change().rolling(window=10).sum()
    
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    volume_confirmation = (volume_increase * df['volume']).rolling(window=10).sum() / volume_increase.rolling(window=10).count()
    
    final_factor = sequential_momentum * price_momentum
    
    # Adjust by Volume Confirmation
    adjustment_factor = 0.9 if volume_confirmation < 0 else 1
    final_factor = final_factor * adjustment_factor
    
    return final_factor
