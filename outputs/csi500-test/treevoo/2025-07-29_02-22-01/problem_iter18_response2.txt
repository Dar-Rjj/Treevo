```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']

    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_10_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()

    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adj_hl_spread_price_momentum = ma_10_volume_adjusted_high_low_spread + price_momentum

    # Daily Price Change
    daily_price_change = df['close'].diff()

    # Momentum Calculation (Sum of Daily Price Changes over the last N days)
    momentum_calculation = daily_price_change.rolling(window=10).sum()

    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    average_volume_increase = (daily_volume_change[daily_volume_change > 0].rolling(window=10).sum() /
                               (daily_volume_change > 0).rolling(window=10).sum())

    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_vol_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = price_momentum / (smoothed_vol_rate_of_change + 1e-6)

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return

    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return

    # Combine Factors
    combined_factors = (combined_vol_adj_hl_spread_price_momentum +
                        proportional_movement_index +
                        volume_weighted_adjusted_return +
                        price_momentum +
                        momentum_calculation +
                        volume_reversal_indicator +
                        average_volume_increase * momentum_calculation)

    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = df['close'].diff() * df['volume']

    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_10_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()

    # Combine Volume-Adjusted Price Change and Price Momentum
    combined_vol_adj_price_change_price_momentum = ma_10_volume_adjusted_daily_price_change + price_momentum

    # Calculate Average Volume
    lookback_window = 10
    average_volume = df['volume'].rolling(window=lookback_window).mean()

    # Weighted Combined Momentum
    weighted_combined_momentum = combined_factors * average_volume

    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    final_alpha_factor = weighted_combined_momentum / (daily_high_low_spread + 1e-6)

    return final_alpha_factor
```
