```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term and Long-Term EMA of Close Prices
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    price_momentum = long_ema - short_ema
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Momentum Calculation over the last 14 days
    momentum = daily_price_change.rolling(window=14).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    volume_increase = daily_volume_change[daily_volume_change > 0]
    avg_volume_increase = volume_increase.rolling(window=10).sum() / (daily_volume_change > 0).rolling(window=10).sum()
    
    # Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_roc = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = price_momentum * (1 / smoothed_volume_roc)
    
    # High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_vol_adj_hl_spread = volume_adjusted_high_low_spread.rolling(window=10).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adj_hl_spread = moving_sum_vol_adj_hl_spread * price_momentum
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Combine Factors
    final_alpha_factor = (
        price_momentum +
        momentum +
        volume_reversal_indicator +
        avg_volume_increase * momentum +
        proportional_movement_index +
        combined_vol_adj_hl_spread +
        volume_weighted_adjusted_return
    )
    
    return final_alpha_factor
```
