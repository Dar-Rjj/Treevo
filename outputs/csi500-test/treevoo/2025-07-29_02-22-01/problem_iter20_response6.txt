```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']
    
    # Volume-Weighted Adjusted Return
    daily_volume_weighted_range = df['volume'] * df['intraday_range']
    df['vol_adjusted_return'] = (daily_volume_weighted_range * df['intraday_return']).rolling(window=20).sum() / daily_volume_weighted_range.rolling(window=20).sum()
    
    # Simple Moving Averages (SMA)
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Price Momentum
    df['price_momentum'] = df['sma_50'] - df['sma_10']
    
    # Momentum Difference
    df['momentum_diff'] = df['sma_50'] - df['sma_10']
    
    # Adjusted Intraday Return
    df['adj_intraday_return'] = (df['intraday_return'] * daily_volume_weighted_range).rolling(window=20).sum() / daily_volume_weighted_range.rolling(window=20).sum()
    
    # Momentum Component
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['momentum_component'] = df['sma_5'] - df['sma_20']
    
    # Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['momentum_component'] * (1 / df['volatility'])
    
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['price_momentum'] 
                          + df['proportional_movement_index'] 
                          + df['vol_adjusted_return'] 
                          + df['combined_momentum_volatility'] 
                          + df['vwap'] 
                          + df['rsi'])

    return df['alpha_factor']
```
