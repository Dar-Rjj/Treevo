import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_high_low_spread = (volume_adjusted_high_low_spread.rolling(window=10).sum() / 
                                       df['volume'].rolling(window=10).sum())
    
    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    moving_avg_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Combine Volume-Adjusted Metrics and Price Momentum
    combined_momentum = (volume_weighted_high_low_spread + 
                         moving_avg_volume_adjusted_price_change + 
                         price_momentum)
    
    # Calculate Average Volume
    lookback_window = 10
    average_volume = df['volume'].rolling(window=lookback_window).mean()
    
    # Weighted Combined Momentum
    weighted_combined_momentum = combined_momentum * average_volume
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    final_alpha_factor = daily_high_low_spread * weighted_combined_momentum
    
    return final_alpha_factor
