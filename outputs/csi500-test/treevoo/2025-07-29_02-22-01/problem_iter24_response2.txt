```python
import pandas as pd

def heuristics_v2(df, N=20, M=10):
    # Calculate the basic momentum over N days
    df['momentum'] = (df['close'] / df['close'].shift(N)) - 1
    
    # Calculate the Volume Weighted Average Price (VWAP) for each day in the lookback period
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=N).mean()
    
    # Compute the VWAP momentum
    df['vwap_momentum'] = (df['vwap'] / df['vwap'].shift(N)) - 1
    
    # Calculate the daily range (High - Low)
    df['range'] = df['high'] - df['low']
    
    # Compute the average true range (ATR) over N days
    df['atr'] = df['range'].rolling(window=N).mean()
    
    # Adjust the momentum and VWAP momentum by dividing by the ATR
    df['adjusted_momentum'] = df['momentum'] / df['atr']
    df['adjusted_vwap_momentum'] = df['vwap_momentum'] / df['atr']
    
    # Identify oversold/overbought conditions over a short window (e.g., 5 days)
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['reversal_score'] = (df['close'] - df['5_day_low']) / (df['5_day_high'] - df['5_day_low'])
    
    # Combine the reversal score with the adjusted VWAP momentum
    df['refined_alpha'] = df['adjusted_vwap_momentum'] * (1 - df['reversal_score'])
    
    # Observe the change in trading volume over the last N days versus the previous M days
    df['recent_volume_change'] = (df['volume'].rolling(window=N).mean() / df['volume'].shift(M).rolling(window=M).mean()) - 1
    
    # Define thresholds for "significant" changes in volume
    significant_increase = 0.20
    significant_decrease = -0.20
    
    # Adjust the overall alpha factor by incorporating a volume change multiplier
    df['volume_multiplier'] = 1.0
    df.loc[df['recent_volume_change'] > significant_increase, 'volume_multiplier'] = 1.5
    df.loc[df['recent_volume_change'] < significant_decrease, 'volume_multiplier'] = 0.5
    
    # Final alpha factor
    df['alpha_factor'] = df['refined_alpha'] * df['volume_multiplier']
    
    return df['alpha_factor']
```
