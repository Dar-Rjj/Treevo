```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = df['close'] / df['open'] - 1
    
    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = abs(volume_ma - df['volume'])
    
    # Assess Intraday Volume Impact
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    intraday_volume_impact = df['volume'] / avg_volume_10d
    
    # Combine Metrics for Volatility
    combined_volatility = high_low_spread * weighted_volume_trend + daily_return
    combined_volatility_sma = combined_volatility.rolling(window=5).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_effect = (intraday_movement + intraday_volume_impact) * combined_volatility_sma
    
    # Measure Sequential Momentum
    sequential_momentum = integrated_effect.rolling(window=5).sum()
    smoothed_sequential_momentum = sequential_momentum.ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change().rolling(window=5).sum()
    
    # Volume Confirmation
    volume_change = df['volume'].diff()
    positive_volume_days = (volume_change > 0).sum()
    total_volume_increase = volume_change[volume_change > 0].sum()
    volume_confirmation = total_volume_increase / positive_volume_days if positive_volume_days > 0 else 0
    
    # Integrate Price Momentum and Volume Confirmation
    final_alpha = smoothed_sequential_momentum * (1 + price_momentum)
    
    # Adjust by Volume Confirmation
    if volume_confirmation < 0:
        final_alpha = final_alpha * 0.9  # Reduce factor by a set percentage if volume confirmation is negative
    
    return final_alpha
```
