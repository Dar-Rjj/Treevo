```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema_span = 10
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    long_ema_span = 50
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    df['crossover_momentum'] = df['long_ema'] - df['short_ema']
    
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Momentum Calculation (Sum of Daily Price Changes over the last N days)
    N = 20
    df['momentum'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Volume Confirmation
    df['volume_change'] = df['volume'].diff()
    M = 10
    df['positive_volume_diff'] = df['volume_change'].clip(lower=0)
    df['volume_increase_days'] = (df['volume_change'] > 0).astype(int)
    df['average_volume_increase'] = df['positive_volume_diff'].rolling(window=M).sum() / df['volume_increase_days'].rolling(window=M).sum()
    
    # Volume Reversal Indicator
    volume_rate_of_change_period = 5
    df['volume_roc'] = df['volume'].pct_change(periods=volume_rate_of_change_period)
    df['smoothed_volume_roc'] = df['volume_roc'].rolling(window=3).mean()
    
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']
    
    # Volume-Weighted Adjusted Return
    df['volume_weighted_return'] = df['volume'] * df['intraday_return']
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['proportional_movement_index'] + df['volume_weighted_return']
    
    # Calculate Simple Moving Averages (SMA)
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    
    # Momentum Component
    df['momentum_component'] = df['5_day_sma'] - df['20_day_sma']
    
    # Volatility
    df['daily_log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['daily_log_returns'].rolling(window=20).std()
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['momentum_component'] * (1 / df['volatility'])
    
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Final Combined Intraday and Momentum Components
    df['final_intraday_momentum'] = df['combined_momentum_volatility'] + df['intraday_factors']
    
    # Calculate Cumulative Daily Price Change
    cumulative_window = 30
    df['cumulative_daily_price_change'] = df['daily_price_change'].rolling(window=cumulative_window).sum()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=cumulative_window).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_daily_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['final_alpha_factor'] = df['high_low_range'] * df['weighted_cumulative_price_change']
    
    return df['final_alpha_factor']
```
