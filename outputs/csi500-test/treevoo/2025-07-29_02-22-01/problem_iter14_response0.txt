import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = short_term_ema - long_term_ema

    # Daily Price Change
    daily_price_change = df['close'].diff()

    # Momentum Calculation
    N = 10
    momentum = daily_price_change.rolling(window=N).sum()

    # Volume Confirmation
    volume_change = df['volume'].diff()
    M = 10
    positive_volume_diff = volume_change.where(volume_change > 0, 0)
    days_with_positive_volume = (volume_change > 0).sum()
    average_volume_increase = positive_volume_diff.rolling(window=M).sum() / days_with_positive_volume

    # Calculate Cumulative Daily Price Change
    cumulative_price_change = daily_price_change.rolling(window=N).sum()

    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=N).mean()

    # Weighted Cumulative Daily Price Change
    weighted_cumulative_price_change = cumulative_price_change * average_volume

    # Incorporate High-Low Range
    high_low_range = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)
    high_low_enhanced_alpha = high_low_range * weighted_cumulative_price_change

    # Incorporate Open-Close Spread
    open_close_spread = abs(df['open'] - df['close'])
    open_close_enhanced_alpha = open_close_spread + high_low_enhanced_alpha

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = abs(df['volume'] - volume_ma)

    # Assess Intraday Volume Impact
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_weight = df['volume'] / avg_volume

    # Combine Metrics for Volatility
    combined_metric = (df['high'] - df['low']) * weighted_volume_trend + intraday_return
    combined_metric_sma = combined_metric.rolling(window=10).mean()

    # Integrate Intraday and Historical Effects
    integrated_effect = (intraday_return + volume_weight) * combined_metric_sma

    # Measure Sequential Momentum
    sequential_momentum = integrated_effect.rolling(window=5).sum()
    smoothed_sequential_momentum = sequential_momentum.ewm(span=3, adjust=False).mean()

    # Final Alpha Factor
    final_alpha = smoothed_sequential_momentum * open_close_enhanced_alpha
    final_alpha = final_alpha.where(average_volume_increase > 0, final_alpha * 0.9)

    return final_alpha
