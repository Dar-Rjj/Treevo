```python
import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate Price Momentum
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['short_ema'] - df['long_ema']
    
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Momentum Calculation
    df['momentum'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Volume Confirmation
    df['volume_change'] = df['volume'].diff()
    df['positive_volume_diff'] = df['volume_change'].apply(lambda x: x if x > 0 else 0)
    df['average_volume_increase'] = df['positive_volume_diff'].rolling(window=M).sum() / (df['volume_change'] > 0).rolling(window=M).sum()
    
    # Calculate Volume Reversal Indicator
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=M)
    df['smoothed_vol_roc'] = df['volume_rate_of_change'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = df['intraday_price_range'] / df['close'] * df['intraday_return']
    
    # Volume-Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['intraday_return']
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['proportional_movement_index'] + df['volume_weighted_adjusted_return']
    
    # Calculate Simple Moving Averages (SMA)
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    
    # Momentum Component
    df['momentum_component'] = df['5_day_sma'] - df['20_day_sma']
    
    # Volatility
    df['daily_log_returns'] = df['close'].apply(lambda x: np.log(x) - np.log(df['close'].shift(1)))
    df['volatility'] = df['daily_log_returns'].rolling(window=20).std()
    
    # Enhance with Volume-Weighted Price
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Incorporate Volume Reversal Indicator
    df['final_alpha_factor'] = (df['momentum_component'] + df['volatility'] + df['vwap']) / df['smoothed_vol_roc']
    
    # Calculate Cumulative Daily Price Change
    df['cumulative_price_change'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    df['cumulative_momentum_impact'] = df['cumulative_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_spread'] = df['high'] - df['low']
    df['high_low_spread_ratio'] = (df['high'] - df['low']) / df['close']
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['high_low_spread_ratio'] * df['cumulative_momentum_impact']
    
    return df['final_alpha_factor']
```
