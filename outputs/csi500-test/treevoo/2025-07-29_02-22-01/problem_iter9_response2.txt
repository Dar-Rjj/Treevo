```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_moving_average = (volume_adjusted_high_low_spread.rolling(window=10).sum() / 
                                      df['volume'].rolling(window=10).sum())
    
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Combine Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread and Price Momentum
    factor = volume_weighted_moving_average + price_momentum
    
    return factor
```
