```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['EMA_short'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_long'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Price_Momentum'] = df['EMA_long'] - df['EMA_short']
    
    # Calculate Intraday Momentum Swing
    df['Intraday_Return'] = df['close'] - df['open']
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Return'] = (df['Intraday_Return'] * df['volume']) / df['Avg_Volume_20']
    
    # Calculate Proportional Movement Index
    df['Intraday_Range'] = df['high'] - df['low']
    df['Proportional_Movement_Index'] = (df['Intraday_Range'] / df['close']) * df['Intraday_Return']
    
    # Calculate Volume-Weighted Adjusted Return
    df['Volume_Adjusted_Intraday_Range'] = df['volume'] * df['Intraday_Range']
    df['Volume_Adjusted_Return_20'] = (df['Volume_Adjusted_Intraday_Range'] * df['Intraday_Return']).rolling(window=20).sum() / df['Volume_Adjusted_Intraday_Range'].rolling(window=20).sum()
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_short'] = df['close'].rolling(window=10).mean()
    df['SMA_long'] = df['close'].rolling(window=50).mean()
    
    # Momentum Component
    df['Momentum_Component'] = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Volatility
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Volatility'] = df['Log_Returns'].rolling(window=20).std()
    
    # Combine Momentum and Volatility
    df['Combined_Momentum_Volatility'] = df['Momentum_Component'] * (1 / df['Volatility'])
    
    # Calculate Volume-Adjusted High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    df['Volume_Adjusted_High_Low_Spread_10'] = (df['Volume_Adjusted_High_Low_Spread'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Volume-Adjusted Daily Price Change
    df['Volume_Adjusted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['Volume_Adjusted_Price_Change_10'] = df['Volume_Adjusted_Price_Change'].rolling(window=10).mean()
    
    # Combine Factors
    df['Combined_Factor'] = (df['Price_Momentum'] 
                             + df['Proportional_Movement_Index'] 
                             + df['Volume_Adjusted_Return_20'] 
                             + df['Combined_Momentum_Volatility'] 
                             + df['Volume_Adjusted_High_Low_Spread_10'] 
                             + df['Volume_Adjusted_Price_Change_10'])
    
    # Incorporate High-Low Range
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Alpha_Factor'] = df['Combined_Factor'] * df['High_Low_Spread']
    
    return df['Alpha_Factor']
```
