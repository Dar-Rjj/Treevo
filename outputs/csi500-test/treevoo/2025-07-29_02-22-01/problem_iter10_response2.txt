```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Simple Moving Average of Close Prices
    short_window = 10
    df['SMA_Short'] = df['close'].rolling(window=short_window).mean()
    
    # Long-Term Simple Moving Average of Close Prices
    long_window = 50
    df['SMA_Long'] = df['close'].rolling(window=long_window).mean()
    
    # Price Momentum: Subtract Short-Term SMA from Long-Term SMA
    df['Price_Momentum'] = df['SMA_Long'] - df['SMA_Short']
    
    # Volume Weighted Average
    volume_weighted_avg = (df['close'] * df['volume']).rolling(window=short_window).mean() / df['volume'].rolling(window=short_window).mean()
    df['Volume_Weighted_Avg'] = volume_weighted_avg
    
    # Smoothed Volume Weighted Average
    smoothed_volume_weighted_avg = df['Volume_Weighted_Avg'].rolling(window=long_window).mean()
    df['Smoothed_Volume_Weighted_Avg'] = smoothed_volume_weighted_avg
    
    # Inverse of Smoothed Volume Weighted Average
    df['Inverse_Smoothed_VW_Avg'] = 1 / df['Smoothed_Volume_Weighted_Avg']
    
    # Multiply Price Momentum by Inverse of Smoothed Volume Weighted Average
    df['Price_Volume_Factor'] = df['Price_Momentum'] * df['Inverse_Smoothed_VW_Avg']
    
    return df['Price_Volume_Factor']
```
