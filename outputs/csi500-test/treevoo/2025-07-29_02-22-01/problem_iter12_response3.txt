```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()

    # Calculate Price Momentum by subtracting Short-Term EMA from Long-Term EMA
    df['price_momentum'] = df['ema_50'] - df['ema_10']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    average_volume = df['volume'].rolling(window=20).mean()
    df['volume_weighted_return'] = (df['intraday_return'] * df['volume']) / average_volume

    # Proportional Movement Index
    df['intraday_range'] = df['high'] - df['low']
    df['proportional_movement_index'] = (df['intraday_range'] / df['close']) * df['intraday_return']

    # Combine Factors
    df['combined_factor'] = df['price_momentum'] + df['volume_weighted_return'] + df['proportional_movement_index']

    return df['combined_factor']
```
