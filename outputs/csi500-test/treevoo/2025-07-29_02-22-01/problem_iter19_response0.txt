```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Trend
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_trend = volume_moving_avg - df['volume']
    
    # Assess Intraday Volume Impact
    volume_impact = df['volume'] / volume_moving_avg
    
    # Combine Metrics for Volatility
    combined_volatility = (high_low_spread * volume_trend) + daily_return
    
    # Measure Sequential Momentum
    sequential_momentum = combined_volatility.rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_effects = (intraday_movement + volume_impact) * combined_volatility
    
    # Final Alpha Factor
    # Calculate Price Momentum
    momentum = df['close'].diff().rolling(window=5).sum()
    
    # Volume Confirmation
    positive_volume_diff = (df['volume'] - df['volume'].shift(1)).clip(lower=0)
    positive_volume_days = (positive_volume_diff > 0).rolling(window=5).sum()
    volume_confirmation = positive_volume_diff.rolling(window=5).sum() / positive_volume_days
    
    # Multiply Sequential Momentum by Integrated Intraday and Historical Effects
    final_alpha = sequential_momentum * integrated_effects
    
    # Adjust by Volume Confirmation
    adjustment_factor = np.where(volume_confirmation > 0, 1, 0.9)
    final_alpha = final_alpha * adjustment_factor
    
    return final_alpha
```
