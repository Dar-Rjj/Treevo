```python
import pandas as pd

def heuristics_v2(df, short_ema_span=10, long_ema_span=30, n_days=5, m_days=5, roc_period=10, sma_span=3):
    # Calculate Price Momentum
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    df['price_momentum'] = df['short_ema'] - df['long_ema']
    
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Momentum Calculation
    df['momentum_calculation'] = df['daily_price_change'].rolling(window=n_days).sum()
    
    # Volume Confirmation
    df['daily_volume_change'] = df['volume'].diff()
    df['positive_volume_diff'] = df['daily_volume_change'].clip(lower=0)
    df['positive_volume_count'] = (df['daily_volume_change'] > 0).rolling(window=m_days).sum()
    df['average_volume_increase'] = df['positive_volume_diff'].rolling(window=m_days).sum() / df['positive_volume_count'].replace(0, 1)

    # Calculate Volume Reversal Indicator
    df['volume_roc'] = (df['volume'] - df['volume'].shift(roc_period)) / df['volume'].shift(roc_period) * 100
    df['smoothed_volume_roc'] = df['volume_roc'].rolling(window=sma_span).mean()
    df['inverse_smoothed_volume_roc'] = 1 / (df['smoothed_volume_roc'] + 1e-6)
    df['volume_reversal_indicator'] = df['price_momentum'] * df['inverse_smoothed_volume_roc']

    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_price_range'] / df['close']) * df['intraday_return']
    
    # Combine Factors
    df['alpha_factor'] = (df['proportional_movement_index'] 
                          + df['price_momentum'] 
                          + df['momentum_calculation'] 
                          + df['volume_reversal_indicator'] 
                          + (df['average_volume_increase'] * df['momentum_calculation']))

    return df['alpha_factor']
```
