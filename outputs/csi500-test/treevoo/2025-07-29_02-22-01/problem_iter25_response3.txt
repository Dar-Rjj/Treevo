```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Calculate Intraday Momentum Swing
    intraday_return = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_weighted_adj_return = (df['volume'] * intraday_return) / avg_volume_20
    
    # Calculate Proportional Movement Index
    intraday_range = df['high'] - df['low']
    proportional_movement_index = (intraday_range / df['close']) * intraday_return
    
    # Calculate Volume-Weighted Adjusted Return
    weighted_intraday_returns = (df['volume'] * intraday_range) * intraday_return
    sum_weighted_intraday_returns = df['volume'] * intraday_range
    sum_weighted_intraday_returns = sum_weighted_intraday_returns.rolling(window=20).sum()
    volume_weighted_adj_return = weighted_intraday_returns / sum_weighted_intraday_returns
    
    # Combine Momentum and Volatility
    momentum_component = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    combined_momentum_volatility = momentum_component * (1 / volatility)
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spreads = df['volume'] * (df['high'] - df['low'])
    volume_weighted_high_low_moving_avg = (volume_adjusted_high_low_spreads.rolling(window=10).sum()) / df['volume'].rolling(window=10).sum()
    
    # Calculate Volume-Adjusted Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    volume_adjusted_daily_price_change = daily_price_change * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_volume_adj_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Combine Factors
    combined_factors = (price_momentum + 
                        proportional_movement_index + 
                        volume_weighted_adj_return + 
                        combined_momentum_volatility + 
                        volume_weighted_high_low_moving_avg + 
                        ma_volume_adj_daily_price_change)
    
    # Incorporate High-Low Range
    high_low_spread = df['high'] - df['low']
    final_factor = combined_factors * high_low_spread
    
    return final_factor
```
