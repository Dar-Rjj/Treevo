```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Calculate Adjusted Intraday Return
    intraday_return = df['close'] - df['open']
    volume_weighted_range = df['volume'] * (df['high'] - df['low'])
    weighted_adjustment = (intraday_return * volume_weighted_range).rolling(window=20).sum() / volume_weighted_range.rolling(window=20).sum()
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    volume_adjusted_high_low_spread_ma = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    volume_adjusted_daily_price_change_ma = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Combine Factors
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * (df['close'] - df['open'])
    combined_factors = price_momentum + weighted_adjustment + proportional_movement_index
    
    # Calculate High-Low Volatility
    daily_high_low_range = df['high'] - df['low']
    high_low_volatility = (daily_high_low_range.ewm(span=10, adjust=False).mean()) / df['close']
    
    # Integrate Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    integrated_metrics = (volume_adjusted_high_low_spread_ma + 
                          volume_adjusted_daily_price_change_ma + 
                          price_momentum + 
                          high_low_volatility)
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    incorporated_factor = daily_high_low_spread * integrated_metrics
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_days = (daily_volume_change > 0).rolling(window=10).sum()
    average_volume_increase = (daily_volume_change[daily_volume_change > 0]).rolling(window=10).sum() / positive_volume_days
    
    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_roc = volume_rate_of_change.rolling(window=5).mean()
    inverse_smoothed_roc = 1 / smoothed_volume_roc
    volume_reversal_indicator = price_momentum * inverse_smoothed_roc
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * (df['close'] - df['open'])
    
    # Calculate Average Volume
    lookback_window = 10
    average_volume = df['volume'].rolling(window=lookback_window).mean()
    
    # Weighted Combined Momentum
    weighted_combined_momentum = combined_factors * average_volume
    
    # Adjust Final Alpha Factor with Intraday Volatility
    final_alpha_factor = (incorporated_factor + volume_reversal_indicator) * ((df['high'] - df['low']) / df['close'])
    
    return final_alpha_factor
```
