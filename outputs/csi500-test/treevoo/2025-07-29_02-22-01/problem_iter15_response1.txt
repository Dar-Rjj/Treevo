```python
import pandas as pd

def heuristics_v2(data):
    # Compute Short-Term and Long-Term SMA of Close Prices
    short_sma = data['close'].rolling(window=10).mean()
    long_sma = data['close'].rolling(window=50).mean()

    # Calculate Price Momentum
    price_momentum = long_sma - short_sma

    # Calculate Intraday Return
    intraday_return = data['close'] - data['open']

    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']

    # Calculate Volume and Range Weighted Adjusted Return
    volume_range_weight = (data['volume'] * intraday_range).rolling(window=20).sum()
    volume_range_adjusted_return = (data['volume'] * intraday_range * intraday_return) / volume_range_weight

    # Calculate Proportional Movement Index
    proportional_movement_index = (intraday_range / data['close']) * intraday_return

    # Combine Factors
    combined_factor = price_momentum + proportional_movement_index + volume_range_adjusted_return

    return combined_factor
```
