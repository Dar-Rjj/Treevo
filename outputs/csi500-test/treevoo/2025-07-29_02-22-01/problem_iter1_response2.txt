```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=20).mean()
    weighted_volume_trend = abs(volume_ma - df['volume'])
    
    # Combine Metrics
    combined_metric = high_low_spread * weighted_volume_trend + daily_return
    
    # Smoothing
    smoothed_factor = combined_metric.rolling(window=10).mean()
    
    return smoothed_factor
```
