```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Daily Price Change
    df['volume_adjusted_price_change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['volume_adjusted_10_day_ma'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term Exponential Moving Averages
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = short_ema - long_ema
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    df['combined_momentum'] = df['volume_adjusted_10_day_ma'] + df['price_momentum']
    
    # Calculate Average Volume over the last N days
    N = 20
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Combined Momentum
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_spread'] = df['high'] - df['low']
    df['final_alpha_factor'] = df['high_low_spread'] * df['weighted_combined_momentum']
    
    return df['final_alpha_factor']
```
