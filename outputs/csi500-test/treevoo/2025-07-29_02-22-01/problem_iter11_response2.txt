import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = abs(volume_ma - df['volume'])
    
    # Assess Intraday Volume Impact
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Combine Metrics for Volatility
    combined_metric = (high_low_spread * volume_trend + daily_return).rolling(window=5).mean()
    
    # Measure Sequential Momentum
    sequential_momentum = combined_metric.rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    price_momentum = (df['close'].diff() / df['close']).rolling(window=5).sum()
    volume_increase = (df['volume'] > df['volume'].shift()).astype(int)
    positive_volume_days = (df['volume'].diff() > 0).astype(int).rolling(window=5).sum()
    average_volume_increase = (df['volume'].diff() * (df['volume'].diff() > 0)).rolling(window=5).sum() / positive_volume_days
    volume_confirmation = (volume_increase * average_volume_increase).fillna(0)
    
    alpha_factor = sequential_momentum * (price_momentum + volume_confirmation)
    
    # Adjust by Volume Confirmation
    adjustment_factor = np.where(volume_confirmation > 0, 1, 0.8)
    final_alpha_factor = alpha_factor * adjustment_factor
    
    return final_alpha_factor
