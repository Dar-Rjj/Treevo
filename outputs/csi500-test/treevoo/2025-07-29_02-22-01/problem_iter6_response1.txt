```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Dynamic Volume Trend
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    dynamic_volume_trend = np.abs(df['volume'] - ema_volume)
    
    # Combine Metrics for Volatility
    volatility_metric = high_low_spread * dynamic_volume_trend + daily_return
    combined_metric = volatility_metric.rolling(window=5).mean()
    
    # Integrate Price Momentum
    price_momentum = (df['close'].diff().rolling(window=30).sum()).fillna(0)
    
    # Advanced Volume Confirmation
    volume_change = df['volume'].diff()
    weighted_volume_increase = (volume_change.where(volume_change > 0, 0) * np.arange(len(volume_change), 0, -1)).rolling(window=30).sum()
    positive_volume_days = (volume_change > 0).rolling(window=30).sum()
    
    advanced_volume_confirmation = weighted_volume_increase / positive_volume_days
    
    # Final Alpha Factor
    final_factor = (combined_metric * price_momentum)
    adjustment_percentage = 0.8  # Example adjustment percentage
    weak_confirmation = advanced_volume_confirmation < weighted_volume_increase.mean()
    final_factor[weak_confirmation] *= adjustment_percentage
    
    return final_factor
```
