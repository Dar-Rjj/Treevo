```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Averages (EMA)
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Momentum Component
    df['momentum'] = df['ema_5'] - df['ema_20']

    # Calculate Volatility
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()

    # Volume-Adjusted Price and Momentum
    df['vwap'] = (df['high'] + df['low'] + 2 * df['close']) / 4 * df['volume']
    df['smoothed_vwap'] = df['vwap'].ewm(span=10, adjust=False).mean()
    df['adjusted_momentum'] = df['momentum'] / df['smoothed_vwap']

    # Incorporate Volume Reversal Indicator
    df['volume_roc'] = df['volume'] / df['volume'].shift(1) - 1
    df['smoothed_vol_roc'] = df['volume_roc'].ewm(span=5, adjust=False).mean()

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Volume-Weighted Adjusted Return
    df['volume_weighted_return'] = df['volume'] * df['intraday_return']

    # Calculate Proportional Movement Index
    df['proportional_movement_index'] = (df['high'] - df['low']) / df['open']

    # Volume-Adjusted High-Low Spread
    df['volume_adjusted_hl_spread'] = (df['high'] - df['low']) * df['volume']

    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    df['moving_sum_hl_spread'] = df['volume_adjusted_hl_spread'].rolling(window=10).sum()

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    df['combined_momentum_hl_spread'] = df['moving_sum_hl_spread'] * df['adjusted_momentum']

    # Calculate Cumulative Daily Price Change
    df['cumulative_price_change'] = df['close'].pct_change().rolling(window=14).sum()

    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=14).mean()

    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_price_change'] * df['average_volume']

    # Incorporate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['normalized_high_low_range'] = df['high_low_range'] / df['close']

    # Combine All Factors
    df['alpha_factor'] = (
        df['proportional_movement_index'] +
        df['volume_weighted_return'] +
        df['combined_momentum_hl_spread'] +
        df['weighted_cumulative_price_change'] +
        df['normalized_high_low_range']
    )

    return df['alpha_factor']
```
