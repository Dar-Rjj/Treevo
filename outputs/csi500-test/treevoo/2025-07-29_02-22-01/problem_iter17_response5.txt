def heuristics_v2(df):
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted close
    volume_weighted_close = df['close'] * df['volume']
    
    # Combine the factors to create a novel alpha factor
    alpha_factor = (relative_range * buying_pressure) + volume_weighted_close
    
    return alpha_factor
