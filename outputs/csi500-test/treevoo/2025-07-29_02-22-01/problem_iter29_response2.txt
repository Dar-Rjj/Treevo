```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = short_ema - long_ema
    
    # Calculate Adjusted Intraday Return
    intraday_return = df['close'] - df['open']
    volume_weighted_range = df['volume'] * (df['high'] - df['low'])
    weighted_adjustment = (intraday_return * volume_weighted_range).rolling(window=20).sum() / volume_weighted_range.rolling(window=20).sum()
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    ma_high_low_spread = high_low_spread.rolling(window=10).mean()
    
    # Volume-Adjusted Daily Price Change
    daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    ma_daily_price_change = daily_price_change.rolling(window=10).mean()
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['open']
    
    # Volume-Weighted Adjusted Return
    intraday_range = df['high'] - df['low']
    volume_weighted_adj_return = (intraday_return * (df['volume'] * intraday_range)).rolling(window=20).sum() / (df['volume'] * intraday_range).rolling(window=20).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_high_low = ma_high_low_spread + price_momentum
    
    # Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Combined Momentum and Volatility
    momentum_and_volatility = price_momentum * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + 2*df['close']) / 4 * df['volume'] / df['volume']
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    positive_vol_increase = daily_volume_change[daily_volume_change > 0]
    avg_positive_vol_increase = positive_vol_increase.rolling(window=20).sum() / (positive_vol_increase > 0).rolling(window=20).sum()
    
    # High-Low Volatility
    high_low_range = df['high'] - df['low']
    ema_high_low_range = high_low_range.ewm(span=10, adjust=False).mean()
    normalized_high_low_vol = ema_high_low_range / df['close']
    
    # Integrate Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    integrated_metrics = ma_high_low_spread + ma_daily_price_change + price_momentum + normalized_high_low_vol
    
    # Incorporate High-Low Range
    final_factor = integrated_metrics * (df['high'] - df['low'])
    
    # Final Alpha Factor
    final_alpha_factor = final_factor + proportional_movement_index + volume_weighted_adj_return + momentum_and_volatility + vwap + ma_high_low_spread
    
    # Adjust by Volume Confirmation
    final_alpha_factor[avg_positive_vol_increase < 0] *= 0.9  # Reduce factor by 10% if volume confirmation is negative
    
    # Adjust Final Alpha Factor with Intraday Volatility
    final_alpha_factor = final_alpha_factor * (high_low_range / df['open'])
    
    return final_alpha_factor
```
