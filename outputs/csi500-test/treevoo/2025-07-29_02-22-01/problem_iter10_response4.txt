```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Final Factor
    final_factor = (price_momentum + 
                    proportional_movement_index + 
                    volume_weighted_adjusted_return)
    
    return final_factor
```
