```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Cumulative Daily Price Change
    cumulative_daily_price_change = daily_price_change.rolling(window=14).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum_impact = cumulative_daily_price_change * average_volume
    
    # High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    ten_day_sum_volume_adjusted_high_low = volume_adjusted_high_low_spread.rolling(window=10).sum()
    
    # Combined Volume-Adjusted High-Low and Price Momentum
    combined_volume_adjusted_high_low_and_price_momentum = ten_day_sum_volume_adjusted_high_low * price_momentum
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = price_momentum / (smoothed_volume_rate_of_change + 1e-6)  # Avoid division by zero
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_diff = daily_volume_change[daily_volume_change > 0].rolling(window=10).sum()
    count_positive_volume_days = (daily_volume_change > 0).rolling(window=10).sum()
    average_volume_increase = positive_volume_diff / count_positive_volume_days.replace(0, 1)  # Avoid division by zero
    
    # Combine Factors
    final_alpha_factor = (price_momentum + 
                          cumulative_daily_price_change + 
                          weighted_cumulative_momentum_impact + 
                          combined_volume_adjusted_high_low_and_price_momentum + 
                          intraday_return + 
                          proportional_movement_index + 
                          volume_weighted_adjusted_return + 
                          volume_reversal_indicator + 
                          average_volume_increase * cumulative_daily_price_change)
    
    return final_alpha_factor
```
