```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_hl'] = (df['high'] - df['low']) * df['volume']

    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    df['ma_10_vol_adj_hl'] = df['volume_adjusted_hl'].rolling(window=10).mean()

    # Calculate Short-Term and Long-Term EMAs
    df['ema_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['ema_20_close'] - df['ema_5_close']

    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    df['combined_momentum'] = df['ma_10_vol_adj_hl'] + df['price_momentum']

    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Momentum Calculation (Sum of Daily Price Changes over the last N days)
    N = 7
    df['momentum_calc'] = df['daily_price_change'].rolling(window=N).sum()

    # Volume Confirmation
    df['daily_volume_change'] = df['volume'].diff()
    M = 5
    positive_volume_changes = df['daily_volume_change'].apply(lambda x: max(x, 0))
    df['average_volume_increase'] = (positive_volume_changes.rolling(window=M).sum() / 
                                     (df['daily_volume_change'] > 0).rolling(window=M).sum()).fillna(0)

    # Calculate Volume Reversal Indicator
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=3)
    df['smoothed_vol_roc'] = df['volume_rate_of_change'].rolling(window=3).mean()
    df['inverse_smoothed_vol_roc'] = 1 / (df['smoothed_vol_roc'] + 1e-6)
    df['vol_reversal_indicator'] = df['price_momentum'] * df['inverse_smoothed_vol_roc']

    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    df['vwa_return'] = df['volume'] * df['intraday_return']

    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_return'] / df['close']) * df['intraday_return']

    # Calculate Volatility
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility'] = df['log_returns'].rolling(window=20).std()

    # Enhance with Volume-Weighted Price
    df['vwap'] = ((df['high'] + df['low'] + 2 * df['close']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()
    df['enhanced_momentum_vol'] = df['combined_momentum'] + df['vwap']

    # Incorporate Volume Reversal Indicator
    df['final_alpha'] = (df['combined_momentum'] + 
                         df['proportional_movement_index'] + 
                         df['vwa_return'] + 
                         df['price_momentum'] + 
                         df['momentum_calc'] + 
                         df['vol_reversal_indicator'] + 
                         df['average_volume_increase'] * df['momentum_calc'] + 
                         (df['price_momentum'] / (df['volatility'] + 1e-6)) + 
                         df['enhanced_momentum_vol'])

    return df['final_alpha']
```
