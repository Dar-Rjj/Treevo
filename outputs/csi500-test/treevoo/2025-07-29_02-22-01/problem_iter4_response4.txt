```python
import pandas as pd

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema_span = 12  # You can adjust the span
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()

    # Compute Long-Term EMA of Close Prices
    long_ema_span = 26  # You can adjust the span
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()

    # Calculate Price Momentum
    df['price_momentum'] = df['short_ema'] - df['long_ema']

    # Compute Volume Rate of Change
    volume_roc_period = 10  # You can adjust the period
    df['volume_roc'] = df['volume'].pct_change(periods=volume_roc_period)

    # Smooth Volume Rate of Change with a short-term SMA
    sma_volume_roc_period = 5  # You can adjust the period
    df['sma_volume_roc'] = df['volume_roc'].rolling(window=sma_volume_roc_period, min_periods=1).mean()

    # Calculate Volume Reversal Indicator
    df['volume_reversal_indicator'] = 1 / (1 + df['sma_volume_roc'])

    # Multiply Price Momentum by Inverse of Smoothed Volume Rate of Change
    df['alpha_factor'] = df['price_momentum'] * df['volume_reversal_indicator']

    return df['alpha_factor']
```
