```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_hl_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_volume_adjusted_hl = volume_adjusted_hl_spread.rolling(window=10).sum()
    
    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Component
    momentum_component = short_term_ema - long_term_ema
    
    # Calculate Volatility
    daily_log_returns = np.log(df['close']).diff()
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + 2 * df['close']) / 4) * df['volume']
    smoothed_vwap = vwap.ewm(span=10, adjust=False).mean()
    adjusted_momentum_signal = momentum_component / smoothed_vwap
    
    # Incorporate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    adjusted_alpha_factor = adjusted_momentum_signal * (1 / smoothed_volume_rate_of_change)
    
    # Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Cumulative Daily Price Change
    n = 14
    cumulative_daily_price_change = daily_price_change.rolling(window=n).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=n).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum_impact = cumulative_daily_price_change * average_volume
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_volume_adjusted_and_momentum = moving_sum_volume_adjusted_hl + momentum_component
    
    # Proportional Movement Index
    intraday_return = df['close'] - df['open']
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_increase = daily_volume_change[daily_volume_change > 0].rolling(window=10).sum()
    count_positive_volume_days = (daily_volume_change > 0).rolling(window=10).sum()
    average_volume_increase = positive_volume_increase / count_positive_volume_days.replace(0, np.nan)
    
    # Final Alpha Factor
    final_alpha_factor = (combined_volume_adjusted_and_momentum + 
                          cumulative_daily_price_change + 
                          weighted_cumulative_momentum_impact + 
                          proportional_movement_index + 
                          volume_weighted_adjusted_return + 
                          adjusted_alpha_factor + 
                          intraday_return + 
                          average_volume_increase * weighted_cumulative_momentum_impact)
    
    return final_alpha_factor
```
