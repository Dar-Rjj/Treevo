```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adj_return = df['volume'] * intraday_return
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Combine Factors
    intraday_momentum = proportional_movement_index
    weighted_adjustment = volume_weighted_adj_return * np.log(df['volume'])
    factor_values = intraday_momentum + weighted_adjustment
    
    return factor_values
```
