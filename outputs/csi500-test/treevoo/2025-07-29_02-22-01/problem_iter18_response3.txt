def heuristics_v2(df):
    # Calculate the price range for each day
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure as (close - low) / (high - low)
    df['buying_pressure'] = (df['close'] - df['low']) / df['price_range']
    
    # Calculate the selling pressure as (high - close) / (high - low)
    df['selling_pressure'] = (df['high'] - df['close']) / df['price_range']
    
    # Calculate the volume-weighted buying pressure
    df['volume_weighted_buying_pressure'] = df['buying_pressure'] * df['volume']
    
    # Calculate the volume-weighted selling pressure
    df['volume_weighted_selling_pressure'] = df['selling_pressure'] * df['volume']
    
    # Calculate the net volume-weighted pressure
    df['net_volume_weighted_pressure'] = df['volume_weighted_buying_pressure'] - df['volume_weighted_selling_pressure']
    
    # Calculate the factor value as the sum of the net volume-weighted pressure over a 5-day window
    df['factor_value'] = df['net_volume_weighted_pressure'].rolling(window=5).sum()
    
    return df['factor_value']
