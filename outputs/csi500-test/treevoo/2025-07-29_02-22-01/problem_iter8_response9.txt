```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term EMAs
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    price_momentum = long_ema - short_ema

    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Volume Weighted Adjusted Return
    volume_weighted_return = df['volume'] * intraday_return

    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return

    # Combine Factors
    combined_factor = price_momentum + proportional_movement_index + (volume_weighted_return * np.log(df['volume']))

    # Daily Price Change
    daily_price_change = df['close'].diff()

    # Cumulative Daily Price Change
    N = 30
    cumulative_daily_price_change = daily_price_change.rolling(window=N).sum()

    # Average Volume
    average_volume = df['volume'].rolling(window=N).mean()

    # Weighted Cumulative Daily Price Change
    weighted_cumulative_daily_price_change = cumulative_daily_price_change * average_volume

    # Incorporate High-Low Range
    high_low_range_normalized = (df['high'] - df['low']) / df['close']
    final_alpha_factor = combined_factor + (high_low_range_normalized * weighted_cumulative_daily_price_change)

    return final_alpha_factor
```
