```python
import pandas as pd

def heuristics(df):
    # Compute Short-Term (e.g., 10 days) SMA of Close Prices
    short_sma = df['close'].rolling(window=10).mean()
    
    # Compute Long-Term (e.g., 50 days) SMA of Close Prices
    long_sma = df['close'].rolling(window=50).mean()
    
    # Calculate Price Momentum by subtracting Short-Term SMA from Long-Term SMA
    price_momentum = long_sma - short_sma
    
    # Calculate Volume Weighted Average over a defined period (e.g., 20 days)
    volume_weighted_average = (df['volume'] * df['close']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Smooth Volume Weighted Average with medium-term (e.g., 30 days) EMA
    smoothed_volume_weighted_avg = volume_weighted_average.ewm(span=30, adjust=False).mean()
    
    # Invert the smoothed volume weighted average and multiply by Price Momentum
    inverse_smoothed_volume = 1 / smoothed_volume_weighted_avg
    factor_values = price_momentum * inverse_smoothed_volume
    
    return factor_values
```
