```python
import pandas as pd

def heuristics_v2(df, N=30, M=10):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Momentum Calculation
    df['momentum'] = df['price_change'].rolling(window=N).sum()
    
    # Volume Confirmation
    df['volume_change'] = df['volume'].diff()
    positive_volume_changes = df[df['volume_change'] > 0]['volume_change']
    df['avg_volume_increase'] = positive_volume_changes.rolling(window=M).sum() / df['volume_change'].gt(0).rolling(window=M).sum()

    # Calculate Cumulative Daily Price Change
    df['cumulative_price_change'] = df['price_change'].rolling(window=N).sum()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    df['cumulative_momentum_impact'] = df['cumulative_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)
    df['final_alpha_factor'] = df['cumulative_momentum_impact'] * df['high_low_range']
    
    # Incorporate Open-Close Spread
    df['open_close_spread'] = (df['open'] - df['close']).abs()
    df['enhanced_alpha_factor'] = df['final_alpha_factor'] + df['open_close_spread']
    
    return df['enhanced_alpha_factor']
```
