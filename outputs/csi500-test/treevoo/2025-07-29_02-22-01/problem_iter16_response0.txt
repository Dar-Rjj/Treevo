```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_10_volume_adjusted_high_low = volume_adjusted_high_low.rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = short_ema - long_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adj_high_low_price_mom = ma_10_volume_adjusted_high_low + price_momentum
    
    # Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Momentum Calculation
    momentum = daily_price_change.rolling(window=10).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_vol_days = (daily_volume_change > 0).astype(int)
    avg_vol_increase = (daily_volume_change[positive_vol_days == 1].sum()) / (positive_vol_days.sum())
    
    # Calculate Volume Reversal Indicator
    vol_rate_of_change = df['volume'].pct_change(periods=10)
    smoothed_vol_rate_of_change = vol_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = momentum * (1 / smoothed_vol_rate_of_change)
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Combine Factors
    final_factor = (
        combined_vol_adj_high_low_price_mom +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        price_momentum +
        momentum +
        volume_reversal_indicator +
        avg_vol_increase * momentum
    )
    
    return final_factor
```
