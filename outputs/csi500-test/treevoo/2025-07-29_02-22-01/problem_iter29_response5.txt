```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = short_ema - long_ema
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_range / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    daily_intraday_range = df['high'] - df['low']
    volume_weighted_adjusted_return = (df['volume'] * daily_intraday_range * intraday_return).rolling(window=20).mean() / (df['volume'] * daily_intraday_range).rolling(window=20).sum()
    
    # Simple Moving Averages (SMA)
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    
    # Momentum Component
    momentum_component = short_sma.rolling(window=5).mean() - short_sma.rolling(window=20).mean()
    
    # Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = momentum_component / volatility
    
    # Volume-Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_high_low_spread = (volume_adjusted_high_low_spread * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change_ma = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Combine Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    combined_factors = (
        volume_weighted_high_low_spread +
        volume_adjusted_daily_price_change_ma +
        price_momentum +
        (df['high'] - df['low']) / df['close']
    )
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        combined_momentum_volatility +
        vwap +
        rsi
    )
    
    return final_alpha_factor
```
