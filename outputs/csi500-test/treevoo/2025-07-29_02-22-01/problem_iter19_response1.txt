```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-day and 50-day Simple Moving Averages (SMA) of Close Prices
    sma_10 = df['close'].rolling(window=10).mean()
    sma_50 = df['close'].rolling(window=50).mean()

    # Calculate Momentum Difference
    momentum_difference = sma_10 - sma_50

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Calculate Daily Volume-Weighted Range
    volume_weighted_range = df['volume'] * (df['high'] - df['low'])

    # Calculate Weighted Adjustment
    weighted_adjustment_numerator = (intraday_return * volume_weighted_range).rolling(window=20).sum()
    weighted_adjustment_denominator = volume_weighted_range.rolling(window=20).sum()
    adjusted_intraday_return = weighted_adjustment_numerator / weighted_adjustment_denominator

    # Calculate Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * (df['close'] - df['open'])

    # Combine Factors
    factor_values = momentum_difference + adjusted_intraday_return + proportional_movement_index

    return factor_values
```
