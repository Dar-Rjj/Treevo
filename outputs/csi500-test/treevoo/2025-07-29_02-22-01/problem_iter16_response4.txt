```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Short-Term EMA (5 days) and Long-Term EMA (20 days) of Close Prices
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adj_and_momentum = ma_volume_adjusted_high_low_spread + price_momentum
    
    # Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Momentum Calculation: Sum of Daily Price Changes over the last N days (N=10)
    momentum_calculation = daily_price_change.rolling(window=10).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_diff = daily_volume_change[daily_volume_change > 0]
    avg_volume_increase = positive_volume_diff.sum() / (positive_volume_diff > 0).sum()
    
    # Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change()
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = price_momentum * (1 / smoothed_volume_rate_of_change)
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Combine all factors
    final_factor = (
        combined_vol_adj_and_momentum +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        price_momentum +
        momentum_calculation +
        volume_reversal_indicator +
        avg_volume_increase * momentum_calculation
    )
    
    return final_factor
```
