```python
import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Proportional Movement Index
    df['proportional_movement'] = (df['intraday_range'] / df['close']) * df['intraday_return']
    
    # Volume-Weighted Adjusted Return
    volume_weight = df['volume'] * df['intraday_range']
    df['volume_weighted_adjusted_return'] = (volume_weight * df['intraday_return']) / volume_weight.rolling(window=20).sum()
    
    # Short-Term and Long-Term SMA of Close Prices
    df['short_sma'] = df['close'].rolling(window=10).mean()
    df['long_sma'] = df['close'].rolling(window=50).mean()
    
    # Price Momentum
    df['price_momentum'] = df['short_sma'] - df['long_sma']
    
    # Momentum Component
    df['momentum_5day'] = df['close'].rolling(window=5).mean()
    df['momentum_20day'] = df['close'].rolling(window=20).mean()
    df['momentum_component'] = df['momentum_5day'] - df['momentum_20day']
    
    # Volatility
    df['daily_log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['daily_log_returns'].rolling(window=20).std()
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['momentum_component'] * (1 / df['volatility'])
    
    # Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    df['vwap'] = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['price_momentum'] 
                          + df['proportional_movement'] 
                          + df['volume_weighted_adjusted_return'] 
                          + df['combined_momentum_volatility'] 
                          + (df['close'] - df['vwap']) 
                          + df['rsi'])
    
    return df['alpha_factor']
```
