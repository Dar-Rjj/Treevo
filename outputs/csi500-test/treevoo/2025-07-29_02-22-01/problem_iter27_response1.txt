```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = short_term_ema - long_term_ema
    
    # Calculate Intraday Price Range and Intraday Return
    intraday_price_range = df['high'] - df['low']
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    intraday_range = df['high'] - df['low']
    volume_weighted_adjusted_return = (df['volume'] * intraday_range * intraday_return) / (df['volume'] * intraday_range).rolling(window=20).sum()
    
    # Simple Moving Averages (SMA)
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    
    # Momentum Component
    momentum_component = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = momentum_component * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    volume_weighted_high_low_spread = (volume_adjusted_high_low_spread.rolling(window=10).sum() / df['volume'].rolling(window=10).sum())
    
    # Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change_ma = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    high_low_range_component = daily_high_low_spread * combined_momentum_volatility
    
    # 10-Day Volume-Weighted Moving Average of Intraday Return
    volume_weighted_intraday_return = (intraday_return * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # 10-Day Moving Average of Volume-Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_intraday_return_ma = volume_adjusted_intraday_return.rolling(window=10).mean()
    
    # Incorporate Volume-Weighted Intraday Return
    volume_weighted_intraday_return_component = volume_adjusted_intraday_return_ma * (1 / volatility)
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        combined_momentum_volatility +
        vwap +
        rsi +
        volume_weighted_high_low_spread +
        volume_adjusted_daily_price_change_ma +
        high_low_range_component +
        volume_weighted_intraday_return_component
    )
    
    return final_alpha_factor
```
