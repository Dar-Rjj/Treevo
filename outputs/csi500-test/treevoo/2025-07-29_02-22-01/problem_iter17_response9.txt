```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    df['10_day_moving_avg_vol_adj_hl_spread'] = df['volume_adjusted_high_low_spread'].rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    df['short_ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['short_ema_close'] - df['long_ema_close']
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    df['combined_momentum'] = df['10_day_moving_avg_vol_adj_hl_spread'] + df['price_momentum']
    
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Momentum Calculation
    N = 10  # Example lookback period for momentum
    df['momentum'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Volume Confirmation
    df['daily_volume_change'] = df['volume'].diff()
    M = 10  # Example lookback period for volume confirmation
    positive_volume_changes = df[df['daily_volume_change'] > 0]['daily_volume_change'].rolling(window=M).sum()
    positive_volume_days = df[df['daily_volume_change'] > 0]['daily_volume_change'].rolling(window=M).count()
    df['average_volume_increase'] = positive_volume_changes / positive_volume_days
    
    # Calculate Volume Reversal Indicator
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=14)  # Example period
    df['smoothed_volume_rate_of_change'] = df['volume_rate_of_change'].rolling(window=3).mean()  # Example SMA period
    df['inverted_smoothed_volume_rate_of_change'] = 1 / (df['smoothed_volume_rate_of_change'] + 1e-6)  # Avoid division by zero
    df['volume_reversal_indicator'] = df['price_momentum'] * df['inverted_smoothed_volume_rate_of_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['intraday_return']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_return'] / df['close']) * df['intraday_return']
    
    # Calculate Volume-Adjusted Daily Price Change
    df['volume_adjusted_daily_price_change'] = df['close'].diff().shift(-1) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['10_day_moving_avg_vol_adj_price_change'] = df['volume_adjusted_daily_price_change'].rolling(window=10).mean()
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    df['combined_price_change_momentum'] = df['10_day_moving_avg_vol_adj_price_change'] + df['price_momentum']
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Combined Momentum
    df['weighted_combined_momentum'] = df['combined_price_change_momentum'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['final_alpha_factor'] = df['weighted_combined_momentum'] / df['high_low_spread']
    
    return df['final_alpha_factor']
```
