```python
import pandas as pd

def heuristics_v2(df, N=20, M=10):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Momentum Calculation: Sum of Daily Price Changes over the last N days
    df['price_momentum'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Average Volume Increase over the Last M Days
    df['positive_vol_diff'] = df['daily_volume_change'].apply(lambda x: x if x > 0 else 0)
    df['positive_vol_days'] = (df['daily_volume_change'] > 0).astype(int)
    df['avg_vol_increase'] = df['positive_vol_diff'].rolling(window=M).sum() / df['positive_vol_days'].rolling(window=M).sum()
    df['avg_vol_increase'].fillna(0, inplace=True)  # Handle division by zero
    
    # Calculate Cumulative Daily Price Change over N days
    df['cumulative_price_change'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Calculate Average Volume over N days
    df['average_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['final_alpha_factor'] = df['weighted_cumulative_price_change'] * df['high_low_range']
    
    return df['final_alpha_factor']
```
