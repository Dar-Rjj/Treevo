```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Proportional Movement Index
    proportional_movement_index = intraday_price_range / df['open']
    
    # Volume-Weighted Adjusted Return
    daily_intraday_range = df['high'] - df['low']
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_weighted_adjusted_return = (df['volume'] * daily_intraday_range * intraday_return) / (df['volume'] * daily_intraday_range).rolling(window=20).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_momentum = ma_volume_adjusted_high_low_spread + price_momentum
    
    # Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Combined Momentum and Volatility
    combined_momentum_volatility = price_momentum * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + 2 * df['close']) / 4
    vwap = (vwap * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_days = daily_volume_change[daily_volume_change > 0]
    average_volume_increase = positive_volume_days.sum() / (positive_volume_days > 0).sum()
    
    # Final Alpha Factor
    alpha_factor = (
        proportional_movement_index +
        volume_weighted_adjusted_return +
        combined_momentum_volatility +
        vwap +
        ma_volume_adjusted_high_low_spread
    )
    
    # Adjust by Volume Confirmation
    adjustment = np.where(average_volume_increase > 0, 1, 0.8)
    final_alpha_factor = alpha_factor * adjustment
    
    return final_alpha_factor
```
