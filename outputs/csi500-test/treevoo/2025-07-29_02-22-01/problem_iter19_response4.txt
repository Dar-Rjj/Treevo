```python
import pandas as pd
import numpy as np

def heuristics_v2(df, short_span=10, long_span=50, n_days=20):
    # Compute Short-Term and Long-Term EMA of Close Prices
    df['Short_EMA'] = df['close'].ewm(span=short_span, adjust=False).mean()
    df['Long_EMA'] = df['close'].ewm(span=long_span, adjust=False).mean()

    # Calculate Intraday Metrics
    df['Intraday_Price_Range'] = df['high'] - df['low']
    df['Intraday_Return'] = df['close'] - df['open']

    # Volume-Adjusted Intraday Return
    df['Volume_Adjusted_Intraday_Return'] = df['volume'] * df['Intraday_Return']

    # Proportional Movement Index
    df['Proportional_Movement_Index'] = (df['Intraday_Price_Range'] / df['close']) * df['Intraday_Return']

    # Combine Factors
    df['Price_Momentum'] = df['Short_EMA'] - df['Long_EMA']
    df['Intraday_Momentum'] = df['Proportional_Movement_Index']
    df['Weighted_Intraday_Adjustment'] = df['Volume_Adjusted_Intraday_Return'] * np.log(df['volume'])

    # Cumulative Price Change Over N Days
    df['Cumulative_Price_Change'] = df['close'].diff().rolling(window=n_days).sum()

    # Average Volume Over N Days
    df['Average_Volume'] = df['volume'].rolling(window=n_days).mean()

    # Weighted Cumulative Momentum
    df['Weighted_Cumulative_Momentum'] = df['Cumulative_Price_Change'] * df['Average_Volume']

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Price_Momentum'] + 
                          df['Intraday_Momentum'] + 
                          df['Weighted_Cumulative_Momentum'])

    return df['Alpha_Factor']
```
