```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement adjusted by Open Price
    intraday_price_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute 10-day Moving Average of Volume
    ma_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Subtract Current Volume from Moving Average and take absolute value
    weighted_volume_trend = (ma_volume_10 - df['volume']).abs()
    
    # Assess Intraday Volume Impact relative to average volume
    volume_impact = df['volume'] / ma_volume_10
    
    # Combine Metrics for Volatility
    combined_metrics_volatility = (high_low_spread * weighted_volume_trend) + daily_return
    combined_metrics_volatility = combined_metrics_volatility.rolling(window=5).mean()
    
    # Measure Sequential Momentum with 5-day sum and 3-day EMA smoothing
    sequential_momentum = combined_metrics_volatility.rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_intraday_historical = intraday_price_movement * volume_impact * combined_metrics_volatility
    
    # Calculate Price Momentum over the last N days
    N = 5
    momentum = (df['close'] - df['close'].shift(N)).rolling(window=N).sum()
    
    # Volume Confirmation over the last M days
    M = 10
    volume_confirmation = (df['volume'] - df['volume'].shift(1)).gt(0).rolling(window=M).sum() / M
    
    # Final Alpha Factor
    alpha_factor = sequential_momentum * integrated_intraday_historical
    alpha_factor = np.where(volume_confirmation > 0, alpha_factor, alpha_factor * 0.9)
    
    return alpha_factor
```
