```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Price_Momentum'] = df['EMA_10'] - df['EMA_50']

    # Calculate Intraday Price Range
    df['Intraday_Price_Range'] = df['high'] - df['low']

    # Calculate Proportional Movement Index
    df['Intraday_Return'] = df['close'] - df['open']
    df['Proportional_Movement_Index'] = (df['Intraday_Price_Range'] / df['close']) * df['Intraday_Return']

    # Volume-Weighted Adjusted Return
    df['Intraday_Range'] = df['high'] - df['low']
    df['Volume_Adjusted_Return'] = df['volume'] * df['Intraday_Range'] * df['Intraday_Return']
    df['Volume_Adjusted_Return_Sum'] = df['volume'] * df['Intraday_Range']
    df['Volume_Adjusted_Return_20D'] = df['Volume_Adjusted_Return'].rolling(window=20).sum()
    df['Volume_Adjusted_Return_Sum_20D'] = df['Volume_Adjusted_Return_Sum'].rolling(window=20).sum()
    df['Volume_Weighted_Adjusted_Return'] = df['Volume_Adjusted_Return'] / df['Volume_Adjusted_Return_Sum_20D']

    # Momentum Component
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Momentum_Component'] = df['SMA_5'] - df['SMA_20']

    # Volatility
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Volatility'] = df['Log_Returns'].rolling(window=20).std()

    # Combine Momentum and Volatility
    df['Combined_Momentum_Volatility'] = df['Momentum_Component'] * (1 / df['Volatility'])

    # Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume']

    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Volume-Adjusted High-Low Spread
    df['High_Low_Spread'] = (df['high'] - df['low']) * df['volume']

    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    df['VW_High_Low_Spread_10D'] = df['High_Low_Spread'].rolling(window=10).mean()

    # Volume-Adjusted Daily Price Change
    df['Daily_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']

    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['VW_Daily_Price_Change_10D'] = df['Daily_Price_Change'].rolling(window=10).mean()

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Price_Momentum'] +
        df['Proportional_Movement_Index'] +
        df['Volume_Weighted_Adjusted_Return'] +
        df['Combined_Momentum_Volatility'] +
        df['VWAP'] +
        df['RSI'] +
        df['VW_High_Low_Spread_10D'] +
        df['VW_Daily_Price_Change_10D']
    )

    return df['Final_Alpha_Factor'].dropna()
```
