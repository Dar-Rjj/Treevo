```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Daily Price Change
    df['Volume_Adjusted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['10D_Moving_Avg_Volume_Adjusted_Price_Change'] = df['Volume_Adjusted_Price_Change'].rolling(window=10).mean()
    
    # Calculate Short-Term EMA of Close Prices
    df['5D_EMA_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term EMA of Close Prices
    df['20D_EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['20D_EMA_Close'] - df['5D_EMA_Close']
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    df['Combined_Momentum'] = df['10D_Moving_Avg_Volume_Adjusted_Price_Change'] + df['Price_Momentum']
    
    # Calculate Average Volume over a lookback window of N days
    N = 10
    df['Average_Volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Combined Momentum
    df['Weighted_Combined_Momentum'] = df['Combined_Momentum'] * df['Average_Volume']
    
    # Calculate Daily High-Low Spread
    df['Daily_High_Low_Spread'] = df['high'] - df['low']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Weighted_Combined_Momentum'] / df['Daily_High_Low_Spread']
    
    return df['Final_Alpha_Factor']
```
