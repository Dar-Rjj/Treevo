```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum = volume_adjusted_spread.rolling(window=10).sum()
    
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_ema - short_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_factor = moving_sum * price_momentum
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Daily Price Change
    N = 10
    cumulative_price_change = daily_price_change.rolling(window=N).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum_impact = cumulative_price_change * average_volume
    
    # Incorporate High-Low Range
    daily_high_low_range = (df['high'] - df['low']) / df['close']
    adjusted_cumulative_momentum = daily_high_low_range * weighted_cumulative_momentum_impact
    
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Combine Factors
    final_alpha_factor = proportional_movement_index + volume_weighted_adjusted_return + adjusted_cumulative_momentum
    
    return final_alpha_factor
```
