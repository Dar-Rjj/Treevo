```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate Intraday Metrics
    intraday_price_range = df['high'] - df['low']
    intraday_return = df['close'] - df['open']
    
    # Volume-Adjusted Intraday Return
    volume_adjusted_intraday_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Combine Factors
    price_momentum = short_ema - long_ema
    intraday_momentum = proportional_movement_index
    weighted_intraday_adjustment = volume_adjusted_intraday_return * np.log(df['volume'])
    
    # Cumulative Price Change
    N = 10
    cumulative_price_change = df['close'].pct_change().rolling(window=N).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Momentum
    weighted_cumulative_momentum = cumulative_price_change * average_volume
    
    # Incorporate High-Low Impact
    high_low_impact = (df['high'] - df['low']) / df['close'] * weighted_cumulative_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum + 
        intraday_momentum + 
        high_low_impact
    )
    
    return final_alpha_factor
```
