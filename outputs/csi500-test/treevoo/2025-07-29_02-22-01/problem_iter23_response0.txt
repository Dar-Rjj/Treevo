```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices (10-day span)
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices (50-day span)
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    price_momentum = ema_long - ema_short
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume-Weighted Range
    volume_weighted_range = df['volume'] * (df['high'] - df['low'])
    
    # Weighted Adjustment
    weighted_adjustment = (intraday_return * volume_weighted_range).rolling(window=20).sum() / volume_weighted_range.rolling(window=20).sum()
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    vwma_volume_adjusted_high_low_spread = (volume_adjusted_high_low_spread * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    ma_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * (df['close'] - df['open'])
    
    # Combine Factors
    combined_factors = price_momentum + weighted_adjustment + proportional_movement_index
    
    # Calculate High-Low Volatility
    daily_high_low_range = df['high'] - df['low']
    high_low_volatility_ema = daily_high_low_range.ewm(span=10, adjust=False).mean()
    high_low_volatility = high_low_volatility_ema / df['close']
    
    # Integrate Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    integrated_factor = (vwma_volume_adjusted_high_low_spread + 
                         ma_volume_adjusted_daily_price_change + 
                         price_momentum + 
                         high_low_volatility)
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    final_alpha = daily_high_low_spread * integrated_factor
    
    return final_alpha
```
