def heuristics_v2(data):
    # Momentum Indicators
    data['SMA_5'] = data['close'].rolling(window=5).mean()
    data['SMA_20'] = data['close'].rolling(window=20).mean()
    data['EMA_5'] = data['close'].ewm(span=5, adjust=False).mean()
    data['EMA_20'] = data['close'].ewm(span=20, adjust=False).mean()
    data['Daily_Return'] = (data['close'] - data['close'].shift(1)) / data['close'].shift(1)
    data['Weekly_Return'] = (data['close'] - data['close'].shift(5)) / data['close'].shift(5)

    # Volatility Indicators
    data['True_Range'] = data[['high', 'low']].apply(lambda x: max(x['high'], x['low']), axis=1) - data['close'].shift(1)
    data['ATR_14'] = data['True_Range'].rolling(window=14).mean()
    data['Historical_Volatility_30'] = data['Daily_Return'].rolling(window=30).std()

    # Volume Factors
    data['Volume_Difference'] = data['volume'] - data['volume'].shift(1)
    data['Volume_Over_SMA_20'] = data['volume'] / data['volume'].rolling(window=20).mean()
    data['Volume_Relative_to_Close'] = (data['close'] - data['open']) / (data['volume'] * data['close'])

    # Trend Reversal and Breakout Indicators
    data['Upper_Band'] = data['SMA_20'] + 2 * data['close'].rolling(window=20).std()
    data['Lower_Band'] = data['SMA_20'] - 2 * data['close'].rolling(window=20).std()
    data['%B_Indicator'] = (data['close'] - data['Lower_Band']) / (data['Upper_Band'] - data['Lower_Band'])
    data['Upper_Channel'] = data['high'].rolling(window=20).max()
    data['Lower_Channel'] = data['low'].rolling(window=20).min()

    data['Pivot_Point'] = (data['high'] + data['low'] + data['close']) / 3
    data['R1'] = (2 * data['Pivot_Point']) - data['low']
    data['S1'] = (2 * data['Pivot_Point']) - data['high']

    data['S2'] = data['Pivot_Point'] - (data['R1'] - data['S1'])
    data['R3'] = (data['R2'] - data['S1']) + data['R1']
    data['S3'] = data['Pivot_Point'] - (data['R2'] - data['S2'])
