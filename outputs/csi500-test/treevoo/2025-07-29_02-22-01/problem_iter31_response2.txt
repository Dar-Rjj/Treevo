```python
def heuristics_v2(df):
    # Calculate the simple moving average (SMA) of closing prices
    sma = df['close'].rolling(window=10).mean()
    
    # Determine the difference between the closing price and SMA
    close_sma_diff = df['close'] - sma
    
    # Use the sign of the difference as a directional indicator
    sma_direction = close_sma_diff.apply(lambda x: 1 if x > 0 else -1)
    
    # Compute the range of daily price movement (high - low)
    daily_range = df['high'] - df['low']
    
    # Calculate the ratio of (close - open) to daily range
    close_open_ratio = (df['close'] - df['open']) / daily_range
    
    # Use the ratio to infer closing strength or weakness
    close_strength = close_open_ratio.apply(lambda x: 1 if x > 0.5 else (-1 if x < -0.5 else 0))
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Find the difference between closing price and VWAP
    close_vwap_diff = df['close'] - vwap
    
    # Use the sign of the difference to indicate undervalued or overvalued conditions
    vwap_indicator = close_vwap_diff.apply(lambda x: 1 if x > 0 else -1)
    
    # Compare today's volume to the average volume of the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Use the ratio to flag significant volume activity
    volume_activity = volume_ratio.apply(lambda x: 1 if x > 2 else (-1 if x < 0.5 else 0))
    
    # Calculate the ratio of amount to volume
    avg_trade_size = df['amount'] / df['volume']
    
    # Determine if large trades are bullish or bearish
    trade_size_signal = avg_trade_size.apply(lambda x: 1 if x > 100_000 else -1)
    
    # Combine all the signals into a single alpha factor
    alpha_factor = (sma_direction + close_strength + vwap_indicator + volume_activity + trade_size_signal) / 5
    
    return alpha_factor
```
