```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema_span = 10
    long_ema_span = 30
    
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    
    price_momentum = df['short_ema'] - df['long_ema']
    
    # Calculate Volume Reversal Indicator
    roc_period = 7
    sma_period = 5
    
    df['volume_roc'] = df['volume'].pct_change(periods=roc_period)
    df['smoothed_volume_roc'] = df['volume_roc'].rolling(window=sma_period, min_periods=sma_period).mean()
    
    volume_reversal_indicator = 1 / (df['smoothed_volume_roc'] + 1e-6)  # Adding small epsilon to avoid division by zero
    
    # Combine Price Momentum with Inverse of Smoothed Volume Rate of Change
    factor_values = price_momentum * volume_reversal_indicator
    
    return factor_values
```
