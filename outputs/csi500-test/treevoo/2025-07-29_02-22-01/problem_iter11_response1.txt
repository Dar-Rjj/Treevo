```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_10d_volume_adjusted_spread = volume_adjusted_spread.rolling(window=10).mean()
    
    # Calculate Short-Term and Long-Term EMAs
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adj_and_mom = ma_10d_volume_adjusted_spread + price_momentum
    
    # Daily Price Change
    daily_price_change = df['close'].diff()

    # Momentum Calculation: Sum of Daily Price Changes over the last N days (N=10)
    momentum = daily_price_change.rolling(window=10).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    positive_volume_diff = daily_volume_change[daily_volume_change > 0]
    average_volume_increase = positive_volume_diff.rolling(window=10).sum() / (daily_volume_change > 0).rolling(window=10).sum()
    
    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change()
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    inverted_smoothed_volume_rate_of_change = 1 / smoothed_volume_rate_of_change
    volume_reversal_indicator = price_momentum * inverted_smoothed_volume_rate_of_change
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Final Factor
    final_factor = (
        combined_vol_adj_and_mom +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        price_momentum +
        momentum +
        volume_reversal_indicator +
        average_volume_increase * momentum
    )
    
    return final_factor
```
