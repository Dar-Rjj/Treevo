import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Define the period for the moving sums
    period = 14
    
    # Calculate the positive and negative money flow over the period
    positive_flow_sum = positive_money_flow.rolling(window=period).sum()
    negative_flow_sum = negative_money_flow.rolling(window=period).sum()
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_flow_sum / negative_flow_sum
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    return money_flow_index
