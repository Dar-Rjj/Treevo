import pandas as pd

def heuristics_v2(data: pd.DataFrame, N: int = 20, M: int = 10) -> pd.Series:
    # Calculate the basic momentum over N days
    data['Momentum'] = (data['close'] / data['close'].shift(N)) - 1
    
    # Calculate VWAP for each day in the lookback period
    data['VWAP'] = (data['amount'] / data['volume']).rolling(window=N).mean()
    
    # Calculate the modified momentum using VWAP
    data['VWAP_Momentum'] = (data['VWAP'] / data['VWAP'].shift(N)) - 1
    
    # Calculate the daily range and ATR over N days
    data['Range'] = data['high'] - data['low']
    data['ATR'] = data['Range'].rolling(window=N).mean()
    
    # Adjust the VWAP momentum by dividing by the ATR
    data['Adjusted_VWAP_Momentum'] = data['VWAP_Momentum'] / data['ATR']
    
    # Introduce a Reversal Indicator based on short-term oversold/overbought conditions
    data['5_day_low'] = data['low'].rolling(window=5).min()
    data['5_day_high'] = data['high'].rolling(window=5).max()
    data['Reversal_Score'] = (data['close'] - data['5_day_low']) / (data['5_day_high'] - data['5_day_low']) - 0.5
    
    # Combine the reversal score with the adjusted VWAP momentum
    data['Combined_Factor'] = data['Adjusted_VWAP_Momentum'] + data['Reversal_Score']
    
    # Investigate the impact of trading volume changes
    data['Volume_Change'] = (data['volume'].rolling(window=N).mean() / data['volume'].rolling(window=M).mean()) - 1
    
    # Adjust the overall alpha factor by incorporating a volume change multiplier
    volume_multiplier = data['Volume_Change'].apply(lambda x: 1.2 if x > 0.2 else (0.8 if x < -0.2 else 1.0))
    data['Alpha_Factor'] = data['Combined_Factor'] * volume_multiplier
    
    # Return the alpha factor
    return data['Alpha_Factor']
