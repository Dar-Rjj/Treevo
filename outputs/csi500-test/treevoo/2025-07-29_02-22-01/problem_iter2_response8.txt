```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Return (Close - Open)
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = abs(volume_moving_avg - df['volume'])
    
    # Combine Metrics for Volatility
    combined_volatility_metric = high_low_spread * weighted_volume_trend + daily_return
    combined_volatility_sma = combined_volatility_metric.rolling(window=20).mean()
    
    # Integrate Price Momentum and Volume Confirmation
    price_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_changes = daily_volume_change[daily_volume_change > 0]
    volume_confirmation = positive_volume_changes.rolling(window=5).sum() / positive_volume_changes.rolling(window=5).count()
    
    # Final Alpha Factor
    alpha_factor = combined_volatility_sma * price_momentum
    alpha_factor = alpha_factor.where(volume_confirmation > 0, alpha_factor * 0.9)

    return alpha_factor
```
