```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term and Long-Term EMAs
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    ema_diff = short_ema - long_ema
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_range / df['close']) * intraday_return
    
    # Combine Factors
    price_momentum = ema_diff
    intraday_momentum = proportional_movement_index
    weighted_adjustment = volume_weighted_return * np.log(df['volume'])
    combined_factors = price_momentum + intraday_momentum + weighted_adjustment
    
    # Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Cumulative Daily Price Change
    N = 10
    cumulative_daily_price_change = daily_price_change.rolling(window=N).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_daily_price_change = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range_normalized = (df['high'] - df['low']) / df['close']
    final_alpha_factor = high_low_range_normalized * weighted_cumulative_daily_price_change + combined_factors
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted High-Low Spread
    ma_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=10).mean()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_vol_adjusted_momentum = ma_volume_adjusted_high_low_spread + price_momentum
    
    # Final Factor
    final_factor = combined_factors + weighted_cumulative_daily_price_change + combined_vol_adjusted_momentum
    
    return final_factor
```
