```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema_span = 10
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    long_ema_span = 30
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['short_ema'] - df['long_ema']
    
    # Volume Rate of Change
    volume_change_period = 10
    df['volume_roc'] = df['volume'].pct_change(periods=volume_change_period)
    
    # Smooth Volume Rate of Change
    roc_sma_span = 5
    df['smoothed_volume_roc'] = df['volume_roc'].rolling(window=roc_sma_span).mean()
    
    # Volume Reversal Indicator
    df['volume_reversal_indicator'] = df['price_momentum'] * (1 / (df['smoothed_volume_roc'] + 1e-6))
    
    # Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Proportional Movement Index
    df['proportional_movement_index'] = (df['intraday_price_range'] / df['close']) * df['intraday_return']
    
    # Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['volume'] / df['close']
    
    # Combine Intraday and Daily Momentum
    df['combined_momentum'] = (df['intraday_price_range'] * df['intraday_return']) * df['intraday_volume_intensity']
    
    # Weighted Moving Average of Combined Momentum
    decay_factor = 0.94
    window_size = 15
    df['wma_combined_momentum'] = df['combined_momentum'].ewm(alpha=(1 - decay_factor), adjust=False).mean()
    
    # Aggregate Volume of the Past N Days
    n_days = 15
    df['aggregate_volume'] = df['volume'].rolling(window=n_days).sum()
    
    # Volume Adjusted Crossover Momentum
    df['volume_adjusted_crossover_momentum'] = df['wma_combined_momentum'] / (df['aggregate_volume'] + 1e-6)
    
    # Combine Factors
    df['alpha_factor'] = (
        df['proportional_movement_index'] +
        df['price_momentum'] +
        df['volume_reversal_indicator'] +
        df['volume_adjusted_crossover_momentum']
    )
    
    return df['alpha_factor']
```
