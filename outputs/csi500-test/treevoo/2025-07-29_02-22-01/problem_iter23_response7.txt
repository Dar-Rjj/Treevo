import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum indicator (rate of change over n-periods)
    n = 10
    df['momentum'] = df['close'].pct_change(periods=n)
    
    # Calculate the daily range and its moving average
    df['range'] = df['high'] - df['low']
    df['range_ma'] = df['range'].rolling(window=20).mean()
    
    # Calculate the range position factor
    df['range_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Calculate the volume surge factor
    df['volume_surge'] = df['volume'] > df['volume'].rolling(window=20).mean() * 2
    
    # Calculate the up-volume and down-volume
    df['up_volume'] = (df['close'] > df['close'].shift(1)) * df['volume']
    df['down_volume'] = (df['close'] < df['close'].shift(1)) * df['volume']
    
    # Calculate the accumulation/distribution line
    df['accumulation_distribution'] = (df['close'] - df['low'] - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['accumulation_distribution'] = df['accumulation_distribution'].cumsum()
    
    # Combine factors to create hybrid alpha signals
    df['momentum_volatility'] = df['momentum'] * df['range_ma']
    
    # Calculate the price-volume anomaly factor
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    df['price_volume_anomaly'] = (df['price_change'] - df['volume_change']).abs()
    
    # Construct the liquidity adjusted return factor
    df['liquidity_adjusted_return'] = df['close'].pct_change() * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Final alpha factor
    df['alpha_factor'] = (
        df['momentum_volatility'] + 
        df['price_volume_anomaly'] + 
        df['liquidity_adjusted_return'] + 
        df['accumulation_distribution'].pct_change()
    )
    
    return df['alpha_factor']
