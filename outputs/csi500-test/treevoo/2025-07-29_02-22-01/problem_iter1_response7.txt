```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    ten_day_sum = volume_adjusted_spread.rolling(window=10).sum()
    
    # Calculate 1-Day Moving Sum of Volume-Adjusted High-Low Spread
    one_day_sum = volume_adjusted_spread.rolling(window=1).sum()
    
    # Calculate Difference from 10-Day to 1-Day Moving Sum
    factor = ten_day_sum - one_day_sum
    
    return factor
```
