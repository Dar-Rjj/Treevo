```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    df['momentum_signal'] = df['5_day_SMA'] - df['20_day_SMA']

    # Calculate Volatility
    df['daily_log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['daily_log_returns'].rolling(window=20).std()

    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['momentum_signal'] * (1 / df['volatility'])

    # Enhance with Volume-Weighted Price
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    df['enhanced_momentum_volatility'] = df['momentum_volatility'] + df['vwap']

    # Incorporate Volume Reversal Indicator
    df['volume_rate_of_change'] = df['volume'].pct_change(periods=20)
    df['smoothed_volume_rate_of_change'] = df['volume_rate_of_change'].rolling(window=5).mean()
    df['inverse_smoothed_volume_rate_of_change'] = 1 / df['smoothed_volume_rate_of_change']

    # Final Alpha Factor
    df['alpha_factor'] = df['enhanced_momentum_volatility'] * df['inverse_smoothed_volume_rate_of_change']

    return df['alpha_factor']
```
