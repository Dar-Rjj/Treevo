```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a heuristic approach.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
        pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the net buying pressure
    net_buying_pressure = buying_pressure - selling_pressure
    
    # Calculate the relative volume
    mean_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / mean_volume
    
    # Calculate the alpha factor
    alpha_factor = net_buying_pressure * relative_volume
    
    return alpha_factor
```
