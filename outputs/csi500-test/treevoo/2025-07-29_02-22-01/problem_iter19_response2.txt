```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices (10-day span)
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices (50-day span)
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Return (Close - Open)
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return (Volume * Intraday Return / 20-day average volume)
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_weighted_adjusted_return = (df['volume'] * intraday_return) / avg_volume_20
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_avg_high_low_spread = (
        (volume_adjusted_high_low_spread * df['volume']).rolling(window=10).sum() /
        df['volume'].rolling(window=10).sum()
    )
    
    # Calculate Volume-Adjusted Daily Price Change
    prev_close = df['close'].shift(1)
    daily_price_change = df['close'] - prev_close
    volume_adjusted_daily_price_change = daily_price_change * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    volume_weighted_avg_daily_price_change = (
        volume_adjusted_daily_price_change.rolling(window=10).mean()
    )
    
    # Proportional Movement Index (Intraday Price Range / Close * Intraday Return)
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * intraday_return
    
    # Combine Factors
    combined_factors = (
        price_momentum + 
        volume_weighted_adjusted_return +
        proportional_movement_index
    )
    
    # Combine Volume-Adjusted Metrics and Price Momentum
    combined_metrics = (
        volume_weighted_avg_high_low_spread +
        volume_weighted_avg_daily_price_change +
        price_momentum
    )
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    final_factor = daily_high_low_spread * combined_metrics
    
    return final_factor
```
