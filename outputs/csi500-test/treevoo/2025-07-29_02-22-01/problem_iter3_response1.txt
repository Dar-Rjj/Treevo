```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust by Open Price (Intraday Price Movement)
    intraday_price_movement = high_low_spread / df['open']

    # Determine Volume Weight
    volume_weight = df['volume']
    
    # Relative to Average Volume (10-day Moving Average)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    relative_volume = volume_weight / avg_volume_10d

    # Integrate Intraday Price and Volume Effects
    integrated_effect = intraday_price_movement * relative_volume

    # Measure Sequential Intraday Momentum
    accumulated_effect = integrated_effect.rolling(window=5).sum()

    # Apply Smoothing (3-day Exponential Moving Average)
    smoothed_momentum = accumulated_effect.ewm(span=3, adjust=False).mean()

    return smoothed_momentum
```
