```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    momentum = sma_5 - sma_20

    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()

    # Enhance with Volume-Weighted Price
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Volume Reversal Indicator
    volume_rate_of_change = df['volume'] / df['volume'].shift(1) - 1
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    inverse_smoothed_vol_rate = 1 / (1 + smoothed_volume_rate_of_change)

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close']

    # Combine All Factors
    combined_factor = (
        momentum * (1 / volatility) +
        vwap +
        volume_weighted_adjusted_return +
        proportional_movement_index
    ) * inverse_smoothed_vol_rate

    return combined_factor
```
