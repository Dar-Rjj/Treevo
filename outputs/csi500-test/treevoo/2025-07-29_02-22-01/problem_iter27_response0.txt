```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the intraday price range
    intraday_range = df['high'] - df['low']
    
    # Calculate the buying pressure as the difference between close and open, scaled by volume
    buying_pressure = (df['close'] - df['open']) * df['volume']
    
    # Calculate the traded amount to volume ratio
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (intraday_range + buying_pressure) * amount_to_volume_ratio
    
    return alpha_factor
```
