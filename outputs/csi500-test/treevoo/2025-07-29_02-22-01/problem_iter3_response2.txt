```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate 10-Day Exponentially Weighted Moving Average of Volume-Adjusted High-Low Spread
    ewma_10 = volume_adjusted_high_low_spread.ewm(span=10).mean()
    
    # Calculate 5-Day Exponentially Weighted Moving Average of Volume-Adjusted High-Low Spread
    ewma_5 = volume_adjusted_high_low_spread.ewm(span=5).mean()
    
    # Calculate Difference from 10-Day to 5-Day Exponentially Weighted Moving Averages
    factor = ewma_10 - ewma_5
    
    return factor
```
