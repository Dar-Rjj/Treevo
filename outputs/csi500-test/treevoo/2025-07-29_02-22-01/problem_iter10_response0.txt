```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term and Long-Term EMA of Close Prices
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Price Momentum
    price_momentum = long_ema - short_ema
    
    # Daily Price Change
    daily_price_change = df['close'].pct_change()
    
    # Momentum Calculation
    n_days = 10
    momentum_calculation = daily_price_change.rolling(window=n_days).sum()
    
    # Volume Confirmation
    volume_change = df['volume'].pct_change().fillna(0)
    m_days = 10
    positive_volume_changes = volume_change[volume_change > 0]
    average_volume_increase = (positive_volume_changes.rolling(window=m_days).sum() / 
                               positive_volume_changes.rolling(window=m_days).count()).fillna(0)
    
    # Volume Reversal Indicator
    volume_rate_of_change = (df['volume'] / df['volume'].shift(m_days) - 1) * 100
    smoothed_volume_roc = volume_rate_of_change.rolling(window=3).mean()
    volume_reversal_indicator = price_momentum / (1 / smoothed_volume_roc)
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Combine Factors
    combined_factor = (price_momentum + momentum_calculation + volume_reversal_indicator + 
                       average_volume_increase * momentum_calculation + proportional_movement_index)
    
    return combined_factor
```
