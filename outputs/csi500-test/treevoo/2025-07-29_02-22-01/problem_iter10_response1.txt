```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Price Momentum
    price_momentum = short_ema - long_ema
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_return = intraday_return * df['volume']
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * intraday_return
    
    # Combine Factors
    combined_factors = (price_momentum 
                        + proportional_movement_index 
                        + volume_weighted_return)
    
    return combined_factors
```
