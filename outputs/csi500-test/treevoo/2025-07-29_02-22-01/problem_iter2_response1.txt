```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    df['momentum'] = df['5_day_SMA'] - df['20_day_SMA']

    # Calculate Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()

    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['momentum'] * (1 / df['volatility'])

    # Enhance with Volume-Weighted Price
    df['vwap_price'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap'] = df['vwap_price'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['enhanced_momentum_volatility'] = df['momentum_volatility'] + df['vwap']

    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Compute Weighted Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    df['weighted_volume_trend'] = abs(df['volume_trend'])

    # Combine Metrics
    df['combined_metric'] = df['high_low_spread'] * df['weighted_volume_trend'] + df['enhanced_momentum_volatility']

    # Smoothing
    df['alpha_factor'] = df['combined_metric'].rolling(window=5).mean()

    return df['alpha_factor'].dropna()
```
