```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices (10-day span)
    df['short_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    # Compute Long-Term EMA of Close Prices (50-day span)
    df['long_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    # Subtract Short-Term EMA from Long-Term EMA
    df['price_momentum'] = df['long_ema'] - df['short_ema']
    
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adj_high_low'] = (df['high'] - df['low']) * df['volume']
    # 10-Day Moving Average of Volume-Adjusted High-Low Spread
    df['volume_adj_high_low_ma10'] = df['volume_adj_high_low'].rolling(window=10).mean()
    
    # Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    # Proportional Movement Index
    df['proportional_movement_index'] = df['intraday_range'] / df['open']
    
    # Volume-Weighted Adjusted Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_adjusted_return'] = (df['volume'] * df['intraday_range'] * df['intraday_return']).rolling(window=20).sum() / (df['volume'] * df['intraday_range']).rolling(window=20).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    df['combined_momentum_voladj'] = df['volume_adj_high_low_ma10'] + df['price_momentum']
    
    # Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Combined Momentum and Volatility
    df['combined_mom_vol'] = df['price_momentum'] / df['volatility']
    
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + 2 * df['close']) / 4) * df['volume']
    
    # Volume Confirmation
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_days'] = df['daily_volume_change'].apply(lambda x: 1 if x > 0 else 0)
    M = 20
    df['volume_confirmation'] = (df['daily_volume_change'].rolling(window=M).sum() / df['positive_volume_days'].rolling(window=M).sum()).fillna(0)
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['proportional_movement_index'] +
        df['volume_weighted_adjusted_return'] +
        df['combined_mom_vol'] +
        df['vwap'] +
        df['volume_adj_high_low_ma10']
    )
    
    # Adjust by Volume Confirmation
    adjustment_factor = 0.9  # Set a percentage to reduce the factor by
    df['alpha_factor'] = df.apply(
        lambda row: row['alpha_factor'] if row['volume_confirmation'] > 0 else row['alpha_factor'] * adjustment_factor,
        axis=1
    )
    
    return df['alpha_factor']
```
