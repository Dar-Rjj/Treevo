```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Daily Price Change
    df['Volume_Adjusted_Price_Change'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['MA_Volume_Adjusted_Price_Change_10'] = df['Volume_Adjusted_Price_Change'].rolling(window=10).mean()
    
    # Short-Term EMA of Close Prices
    df['EMA_Close_5'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Long-Term EMA of Close Prices
    df['EMA_Close_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Price Momentum
    df['Price_Momentum'] = df['EMA_Close_20'] - df['EMA_Close_5']
    
    # Combine Volume-Adjusted Price Change and Price Momentum
    df['Combined_Momentum'] = df['MA_Volume_Adjusted_Price_Change_10'] + df['Price_Momentum']
    
    # Calculate Average Volume
    df['Average_Volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Weighted Combined Momentum
    df['Weighted_Combined_Momentum'] = df['Combined_Momentum'] * df['Average_Volume_10']
    
    # Incorporate High-Low Range
    df['High_Low_Spread'] = df['high'] - df['low']
    df['High_Low_Ratio'] = df['High_Low_Spread'] / df['close']
    df['High_Low_Adjusted_Alpha'] = df['Weighted_Combined_Momentum'] * df['High_Low_Ratio']
    
    # Enhance with Open-Close Spread
    df['Open_Close_Spread'] = abs(df['open'] - df['close'])
    df['Open_Close_Enhanced_Alpha'] = df['High_Low_Adjusted_Alpha'] + df['Open_Close_Spread']
    
    # Intraday Price Movement
    df['Intraday_Price_Movement'] = (df['high'] - df['low']) / df['open']
    
    # Daily Return
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    df['Volume_Trend'] = df['volume'].rolling(window=10).mean() - df['volume']
    df['Weighted_Volume_Trend'] = df['Volume_Trend'].abs()
    
    # Assess Intraday Volume Impact
    df['Volume_Impact'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine Metrics for Volatility
    df['Combined_Volatility'] = (df['High_Low_Spread'] * df['Weighted_Volume_Trend'] + df['Daily_Return']).rolling(window=10).mean()
    
    # Integrate Intraday and Historical Effects
    df['Integrated_Effects'] = df['Intraday_Price_Movement'] * df['Volume_Impact'] * df['Combined_Volatility']
    
    # Measure Sequential Momentum
    df['Sequential_Momentum'] = df['Integrated_Effects'].rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Sequential_Momentum'] * df['Open_Close_Enhanced_Alpha']
    df['Volume_Confirmation'] = df['volume'] > df['volume'].shift(1)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].where(df['Volume_Confirmation'], df['Final_Alpha_Factor'] * 0.9)
    
    return df['Final_Alpha_Factor'].dropna()
```
