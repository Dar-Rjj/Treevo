```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, min_periods=10).mean()
    long_ema = df['close'].ewm(span=50, min_periods=50).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Momentum Swing
    intraday_return = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_adjusted_return = (df['volume'] * intraday_return) / avg_volume
    
    # Calculate Proportional Movement Index
    intraday_range = (df['high'] - df['low']) / df['close']
    proportional_movement_index = intraday_range * intraday_return
    
    # Calculate Volume-Weighted Adjusted Return
    volume_intraday_range = df['volume'] * (df['high'] - df['low'])
    volume_weighted_adjusted_return = (volume_intraday_range * intraday_return) / volume_intraday_range.rolling(window=20).sum()
    
    # Combine Momentum and Volatility
    momentum_component = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()
    combined_momentum_volatility = momentum_component / volatility
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_weighted_high_low_spread = (volume_adjusted_high_low_spread.rolling(window=10).sum() / df['volume'].rolling(window=10).sum())
    
    # Calculate Volume-Adjusted Daily Price Change
    volume_adjusted_daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    moving_avg_volume_adjusted_daily_price_change = volume_adjusted_daily_price_change.rolling(window=10).mean()
    
    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Factors
    alpha_factor = (
        price_momentum +
        proportional_movement_index +
        volume_weighted_adjusted_return +
        combined_momentum_volatility +
        volume_weighted_high_low_spread +
        moving_avg_volume_adjusted_daily_price_change +
        high_low_range
    )
    
    return alpha_factor
```
