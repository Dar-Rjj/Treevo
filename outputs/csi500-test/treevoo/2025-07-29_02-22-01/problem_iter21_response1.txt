```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the amount traded relative to the average volume
    amount_per_volume = df['amount'] / avg_volume_5d
    
    # Combine the two factors: price change and amount per volume
    alpha_factor = price_change * amount_per_volume
    
    return alpha_factor
```
