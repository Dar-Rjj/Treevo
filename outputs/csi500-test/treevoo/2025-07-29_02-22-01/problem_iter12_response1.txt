```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Compute Dynamic Volume Trend
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    dynamic_volume_trend = (df['volume'] - ema_volume).abs()
    
    # Combine Metrics for Volatility
    combined_volatility = high_low_spread * dynamic_volume_trend + daily_return
    sma_combined_volatility = combined_volatility.rolling(window=20).mean()
    
    # Integrate Price Momentum and Advanced Volume Confirmation
    daily_price_change = df['close'].diff()
    price_momentum = daily_price_change.rolling(window=10).sum()
    
    daily_volume_change = df['volume'].diff()
    weighted_volume_increase = ((daily_volume_change[daily_volume_change > 0] * np.arange(1, len(daily_volume_change) + 1)).rolling(window=5).sum() / np.arange(1, 6).sum())
    count_positive_volume = (daily_volume_change > 0).rolling(window=5).sum()
    
    advanced_volume_confirmation = (weighted_volume_increase / count_positive_volume).fillna(0)
    
    # Final Alpha Factor
    final_alpha = sma_combined_volatility * price_momentum
    final_alpha = final_alpha.where(advanced_volume_confirmation > 0, final_alpha * 0.9)
    
    # Calculate Cumulative Daily Price Change
    cumulative_price_change = daily_price_change.rolling(window=10).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum = cumulative_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range = df['high'] - df['low']
    normalized_high_low_range = high_low_range / df['close']
    adjusted_cumulative_momentum = normalized_high_low_range * weighted_cumulative_momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * intraday_return
    
    # Intraday Momentum Confirmation
    intraday_momentum = (df['close'] - df['open']) / df['open']
    intraday_confirmation = intraday_momentum > daily_return
    final_alpha = final_alpha.where(intraday_confirmation, final_alpha * 0.9)
    
    # Combine Factors
    final_alpha += proportional_movement_index + volume_weighted_intraday_return
    return final_alpha
```
