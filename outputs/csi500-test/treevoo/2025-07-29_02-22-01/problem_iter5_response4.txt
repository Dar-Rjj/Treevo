```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Swing Analysis
    intraday_price_range = df['high'] - df['low']
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    volume_weighted_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Combine Intraday Factors
    intraday_momentum_factor = intraday_return + proportional_movement_index + (volume_weighted_return * np.log(df['volume']))

    # Multi-Day Price and Volume Momentum Confirmation
    # Calculate Price Momentum over the last N days
    n_days = 5  # Example: 5 days for momentum
    price_change = df['close'].diff()
    price_momentum = price_change.rolling(window=n_days).sum()

    # Volume Confirmation over the Last M Days
    m_days = 5  # Example: 5 days for volume confirmation
    volume_change = df['volume'].diff()
    positive_volume_diff_sum = (volume_change[volume_change > 0]).rolling(window=m_days).sum()
    positive_volume_count = (volume_change > 0).rolling(window=m_days).sum()
    average_volume_increase = positive_volume_diff_sum / positive_volume_count.replace(0, np.nan)

    # Combine All Factors
    alpha_factor = intraday_momentum_factor + price_momentum + average_volume_increase

    return alpha_factor
```
