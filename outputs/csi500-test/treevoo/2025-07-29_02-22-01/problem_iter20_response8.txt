```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure: (close - low) / range
    df['buying_pressure'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the selling pressure: (high - close) / range
    df['selling_pressure'] = (df['high'] - df['close']) / df['range']
    
    # Calculate the volume-weighted buying and selling pressure
    df['volume_weighted_buying'] = df['buying_pressure'] * df['volume']
    df['volume_weighted_selling'] = df['selling_pressure'] * df['volume']
    
    # Calculate the net volume-weighted pressure
    df['net_volume_weighted_pressure'] = df['volume_weighted_buying'] - df['volume_weighted_selling']
    
    # Calculate the 10-day exponential moving average of the net volume-weighted pressure
    ema_period = 10
    df['ema_net_volume_weighted_pressure'] = df['net_volume_weighted_pressure'].ewm(span=ema_period, adjust=False).mean()
    
    # The final factor is the ratio of the current net volume-weighted pressure to its EMA
    df['alpha_factor'] = df['net_volume_weighted_pressure'] / df['ema_net_volume_weighted_pressure']
    
    return df['alpha_factor']
```
