```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Volume Acceleration
    smoothed_volume = df['volume'].ewm(span=10, adjust=False).mean()
    volume_acceleration = price_momentum / (smoothed_volume + 1e-8)
    
    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    moving_sum_high_low_spread = high_low_spread.rolling(window=10).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_factor_1 = moving_sum_high_low_spread * price_momentum
    
    # Calculate Cumulative Daily Price Change
    daily_price_change = df['close'].pct_change().fillna(0)
    cumulative_daily_price_change = daily_price_change.rolling(window=14).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_daily_price_change = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range = (df['high'] - df['low']) / df['close']
    adjusted_cumulative_momentum = high_low_range * weighted_cumulative_daily_price_change
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) * intraday_return
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Combine Factors
    final_alpha_factor = combined_factor_1 + proportional_movement_index + volume_weighted_intraday_return

    return final_alpha_factor
```
