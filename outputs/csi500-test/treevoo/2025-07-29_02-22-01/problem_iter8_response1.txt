import pandas as pd

def heuristics_v2(df, N=10, M=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Dynamic Volume Trend
    volume_ema = df['volume'].ewm(span=N).mean()
    dynamic_volume_trend = (df['volume'] - volume_ema).abs()
    
    # Combine Metrics for Volatility
    combined_volatility_metric = high_low_spread * dynamic_volume_trend + daily_return
    sma_combined_volatility_metric = combined_volatility_metric.rolling(window=N).mean()
    
    # Integrate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1)).rolling(window=N).sum()
    
    # Advanced Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    weighted_volume_increase = (volume_change.rolling(window=M).apply(lambda x: sum(x[x > 0] * (M - range(M)) / M), raw=False))
    positive_volume_days = (volume_change > 0).rolling(window=M).sum()
    
    advanced_volume_confirmation = weighted_volume_increase / positive_volume_days
    
    # Adjust by Advanced Volume Confirmation
    adjusted_volatility_metric = sma_combined_volatility_metric * price_momentum
    final_alpha_factor = adjusted_volatility_metric.where(advanced_volume_confirmation > 0, adjusted_volatility_metric * 0.9)
    
    # Calculate Cumulative Daily Price Change
    cumulative_daily_price_change = (df['close'] - df['close'].shift(1)).rolling(window=N).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_momentum_impact = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range = (df['high'] - df['low']) / df['close']
    adjusted_cumulative_momentum = weighted_cumulative_momentum_impact * high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * intraday_return
    
    # Proportional Movement Index
    proportional_movement_index = ((df['high'] - df['low']) / df['close']) * intraday_return
    
    # Combine Factors
    final_alpha_factor += proportional_movement_index
    final_alpha_factor += volume_weighted_intraday_return
    
    return final_alpha_factor.dropna()
