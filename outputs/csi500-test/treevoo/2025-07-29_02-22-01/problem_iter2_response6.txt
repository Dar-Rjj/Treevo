```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement Ratio
    intraday_movement_ratio = (df['high'] / df['low']) - 1
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['close']
    
    # Combine Intraday and Daily Momentum
    combined_momentum = (intraday_movement_ratio * daily_return) * intraday_volume_intensity
    
    # Weighted Moving Average of Combined Momentum
    window_size = 15
    decay_factor = 0.94
    weights = np.exp(np.arange(-1, -window_size-1, -1) * decay_factor)
    wma_combined_momentum = combined_momentum.rolling(window=window_size).apply(lambda x: np.nansum(x * weights), raw=True)
    
    # Aggregate Volume of the Past N Days
    n_days = 15
    aggregate_volume = df['volume'].rolling(window=n_days).sum()
    
    # Adjust Crossover Momentum by Volume
    volume_adjusted_crossover_momentum = wma_combined_momentum / aggregate_volume
    
    return volume_adjusted_crossover_momentum
```
