```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    moving_avg_volume = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = moving_avg_volume - df['volume']
    
    # Assess Intraday Volume Impact
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    intraday_volume_impact = df['volume'] / avg_volume_10d
    
    # Combine Metrics for Volatility
    combined_metrics_volatility = (high_low_spread * weighted_volume_trend) + daily_return
    
    # Integrate Intraday and Historical Effects
    integrated_effect = (intraday_price_movement * intraday_volume_impact) * combined_metrics_volatility
    
    # Measure Sequential Momentum
    sequential_momentum = integrated_effect.rolling(window=5).sum()
    sequential_momentum = sequential_momentum.ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    # Calculate Price Momentum
    N = 10
    momentum = (df['close'].diff().rolling(window=N).sum()) / df['close']
    
    # Volume Confirmation
    M = 10
    positive_volume_diff = (df['volume'].diff() > 0).astype(int)
    volume_confirmation = positive_volume_diff.rolling(window=M).sum()
    
    # Adjust by Volume Confirmation
    adjustment_factor = 0.9 if volume_confirmation < 0 else 1.0
    final_alpha = sequential_momentum * adjustment_factor
    
    return final_alpha
```
