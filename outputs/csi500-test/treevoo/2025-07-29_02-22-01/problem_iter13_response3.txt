```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Weighted Volume Trend
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = abs(volume_moving_avg - df['volume'])
    
    # Assess Intraday Volume Impact
    intraday_volume_impact = df['volume'] / volume_moving_avg
    
    # Combine Metrics for Volatility
    combined_volatility = (high_low_spread * weighted_volume_trend + daily_return).rolling(window=5).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_effect = (intraday_movement * intraday_volume_impact) * combined_volatility
    
    # Measure Sequential Momentum
    sequential_momentum = integrated_effect.rolling(window=5).sum().ewm(span=3).mean()
    
    # Calculate Price Momentum
    price_changes = df['close'].diff()
    N = 5
    momentum = price_changes.rolling(window=N).sum()
    
    # Volume Confirmation
    volume_diff = df['volume'].diff()
    M = 5
    positive_volume_sum = (volume_diff > 0).rolling(window=M).sum() * volume_diff
    volume_confirmation = positive_volume_sum / M
    
    # Final Alpha Factor
    alpha_factor = sequential_momentum * integrated_effect
    alpha_factor = alpha_factor.where(volume_confirmation >= 0, alpha_factor * 0.8)
    
    return alpha_factor
```
