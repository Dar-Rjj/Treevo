```python
import pandas as pd

def heuristics_v2(df, N=10, M=10):
    # Calculate Daily Price Change using Close price
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Price Momentum as the sum of daily price changes over the last N days
    df['price_momentum'] = df['daily_price_change'].rolling(window=N).sum()
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    
    # Calculate the number of days with positive volume increase in the last M days
    df['positive_volume_increase'] = (df['daily_volume_change'] > 0).astype(int)
    
    # Calculate the count of days with positive volume
    df['count_positive_vol_days'] = df['positive_volume_increase'].rolling(window=M).sum()
    
    # Calculate the sum of positive volume differences over the last M days
    df['sum_positive_vol_diff'] = df[df['daily_volume_change'] > 0]['daily_volume_change'].rolling(window=M).sum()
    
    # Calculate the average volume increase over the last M days
    df['average_volume_increase'] = df['sum_positive_vol_diff'] / df['count_positive_vol_days']
    
    # Fill NaN values with 0 for the average volume increase
    df['average_volume_increase'].fillna(0, inplace=True)
    
    # Combine Price Momentum and Volume Confirmation
    df['factor'] = df['price_momentum'] * df['average_volume_increase']
    
    return df['factor']
```
