```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Moving Averages (EMA)
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Momentum Component
    momentum = ema_5 - ema_20

    # Calculate Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()

    # Combine Momentum and Volatility
    momentum_volatility = momentum * (1 / volatility)

    # Enhance with Volume-Weighted Price
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap_sum = vwap.rolling(window=20).sum()
    volume_sum = df['volume'].rolling(window=20).sum()
    vwap_normalized = vwap_sum / volume_sum
    enhanced_momentum_volatility = momentum_volatility + vwap_normalized

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Compute Volume Trend
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    volume_trend = abs(df['volume'] - volume_ema)

    # Combine Metrics
    combined_metric = high_low_spread * volume_trend + enhanced_momentum_volatility

    # Smoothing
    final_factor = combined_metric.ewm(span=5, adjust=False).mean()

    return final_factor
```
