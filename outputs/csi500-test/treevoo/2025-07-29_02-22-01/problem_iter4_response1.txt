```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Cumulative Daily Price Change
    df['cumulative_daily_price_change'] = (df['daily_price_change'].rolling(window=N).sum()) / N
    
    # Calculate Average Volume
    df['average_volume'] = (df['volume'].rolling(window=N).sum()) / N
    
    # Weighted Cumulative Daily Price Change
    df['cumulative_momentum_impact'] = df['cumulative_daily_price_change'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['final_alpha_factor'] = df['cumulative_momentum_impact'] * df['high_low_range']
    
    return df['final_alpha_factor']
```
