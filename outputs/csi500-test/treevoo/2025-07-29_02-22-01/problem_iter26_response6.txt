import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema

    # Calculate Adjusted Intraday Return
    intraday_return = df['close'] - df['open']
    volume_weighted_range = df['volume'] * (df['high'] - df['low'])
    weighted_adjustment = (intraday_return * volume_weighted_range).rolling(window=20).sum() / volume_weighted_range.rolling(window=20).sum()

    # Calculate Volume-Adjusted High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    ma_high_low_spread = high_low_spread.rolling(window=10).mean()

    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    ma_daily_price_change = daily_price_change.rolling(window=10).mean()

    # Combine Factors
    proportional_movement_index = (df['high'] - df['low']) / df['close'] * (df['close'] - df['open'])
    combined_factors = price_momentum + weighted_adjustment + proportional_movement_index

    # Calculate High-Low Volatility
    daily_high_low_range = df['high'] - df['low']
    ema_high_low_range = daily_high_low_range.ewm(span=10, adjust=False).mean()
    high_low_volatility = ema_high_low_range / df['close']

    # Integrate Volume-Adjusted Metrics, Price Momentum, and High-Low Volatility
    integrated_metrics = ma_high_low_spread + ma_daily_price_change + price_momentum + high_low_volatility

    # Incorporate High-Low Range
    final_factor = (df['high'] - df['low']) * integrated_metrics

    # Momentum Calculation
    n_days = 10
    momentum = df['close'].diff().rolling(window=n_days).sum()

    # Volume Confirmation
    daily_volume_change = df['volume'].diff()
    m_days = 10
    positive_volume_diff = daily_volume_change[daily_volume_change > 0]
    average_volume_increase = positive_volume_diff.rolling(window=m_days).sum() / (daily_volume_change > 0).rolling(window=m_days).sum()

    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change()
    smoothed_volume_roc = volume_rate_of_change.rolling(window=5).mean()
    inverse_smoothed_volume_roc = 1 / smoothed_volume_roc
    volume_reversal_indicator = price_momentum * inverse_smoothed_volume_roc

    # Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * (df['close'] - df['open'])

    # Calculate Average Volume
    n = 10
    avg_volume = df['volume'].rolling(window=n).mean()

    # Weighted Combined Momentum
    weighted_combined_momentum = combined_factors * avg_volume

    # Adjust Final Alpha Factor with Intraday Volatility
    adjusted_final_alpha = final_factor * high_low_volatility

    return adjusted_final_alpha
