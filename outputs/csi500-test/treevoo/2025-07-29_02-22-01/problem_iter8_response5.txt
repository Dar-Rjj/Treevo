import pandas as pd
import numpy as np

def heuristics_v2(df, N=15, M=15, short_ema_span=10, long_ema_span=30, roc_period=10):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Momentum Calculation: Sum of Daily Price Changes over the last N days
    df['momentum'] = df['price_change'].rolling(window=N).sum()
    
    # Volume Confirmation
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    positive_volume_days = (df['volume_change'] > 0).rolling(window=M).sum()
    df['avg_volume_increase'] = df['volume_change'].where(df['volume_change'] > 0, 0).rolling(window=M).sum() / positive_volume_days
    
    # Cumulative Daily Price Change
    df['cumulative_price_change'] = df['price_change'].rolling(window=N).sum()
    
    # Average Volume
    df['avg_volume'] = df['volume'].rolling(window=N).mean()
    
    # Weighted Cumulative Daily Price Change
    df['weighted_cumulative_price_change'] = df['cumulative_price_change'] * df['avg_volume']
    
    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['adjusted_alpha_factor'] = df['weighted_cumulative_price_change'] * df['high_low_range']
    
    # Intraday Movement Ratio
    df['intraday_movement_ratio'] = df['high'] / df['low']
    
    # Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['volume'] / df['close']
    
    # Combine Intraday and Daily Momentum
    df['combined_momentum'] = df['intraday_movement_ratio'] * df['daily_return'] * df['intraday_volume_intensity']
    
    # Weighted Moving Average of Combined Momentum
    decay_factor = 0.94
    weights = np.array([decay_factor**i for i in range(N)])
    weights /= weights.sum()
    df['wma_combined_momentum'] = df['combined_momentum'].rolling(window=N).apply(lambda x: np.sum(weights * x), raw=True)
    
    # Volume Adjusted Crossover Momentum
    df['agg_volume'] = df['volume'].rolling(window=N).sum()
    df['volume_adjusted_momentum'] = df['wma_combined_momentum'] / df['agg_volume']
    
    # Short-Term EMA of Close Prices
    df['short_ema'] = df['close'].ewm(span=short_ema_span, adjust=False).mean()
    
    # Long-Term EMA of Close Prices
    df['long_ema'] = df['close'].ewm(span=long_ema_span, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    
    # Volume Rate of Change
    df['volume_roc'] = df['volume'].pct_change(periods=roc_period)
    
    # Smooth Volume Rate of Change
    df['smoothed_volume_roc'] = df['volume_roc'].rolling(window=short_ema_span).mean()
    
    # Adjust Final Alpha Factor by Smoothed Volume Rate of Change
    df['final_alpha_factor'] = df['adjusted_alpha_factor'] / df['smoothed_volume_roc']
    
    return df['final_alpha_factor'].dropna()
