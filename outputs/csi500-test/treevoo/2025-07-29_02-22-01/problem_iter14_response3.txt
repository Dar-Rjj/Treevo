```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term EMA of Close Prices (10-day span)
    df['short_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute Long-Term EMA of Close Prices (50-day span)
    df['long_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Subtract Short-Term EMA from Long-Term EMA
    df['price_momentum'] = df['long_ema'] - df['short_ema']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Weighted Adjusted Return
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['volume_weighted_adjusted_return'] = (df['volume'] * df['intraday_return']) / avg_volume_20
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Calculate 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    df['volume_weighted_ma_high_low_spread'] = (df['volume_adjusted_high_low_spread'].rolling(window=10).sum() / 
                                                df['volume'].rolling(window=10).sum())
    
    # Calculate Volume-Adjusted Daily Price Change
    df['previous_close'] = df['close'].shift(1)
    df['volume_adjusted_price_change'] = (df['close'] - df['previous_close']) * df['volume']
    
    # Calculate 10-Day Moving Average of Volume-Adjusted Daily Price Change
    df['ma_volume_adjusted_price_change'] = df['volume_adjusted_price_change'].rolling(window=10).mean()
    
    # Combine Factors
    df['combined_factors'] = df['price_momentum'] + df['volume_weighted_adjusted_return'] + \
                             (df['high_low_spread'] / df['close']) * df['intraday_return']
    
    # Combine Volume-Adjusted Metrics and Price Momentum
    df['combined_momentum'] = df['volume_weighted_ma_high_low_spread'] + df['ma_volume_adjusted_price_change'] + \
                             df['price_momentum']
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    
    # Weighted Combined Momentum
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['average_volume']
    
    # Incorporate High-Low Range
    df['final_alpha_factor'] = df['high_low_spread'] * df['weighted_combined_momentum']
    
    return df['final_alpha_factor']
```
