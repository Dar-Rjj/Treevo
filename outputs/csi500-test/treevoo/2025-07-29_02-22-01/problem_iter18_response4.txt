```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Price Movement
    intraday_price_movement = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Return
    daily_return = df['close'] / df['open'] - 1
    
    # Compute Weighted Volume Trend
    ma_volume_10 = df['volume'].rolling(window=10).mean()
    weighted_volume_trend = (ma_volume_10 - df['volume']).abs()
    
    # Assess Intraday Volume Impact
    relative_volume = df['volume'] / ma_volume_10
    
    # Combine Metrics for Volatility
    combined_metrics_volatility = (high_low_spread * weighted_volume_trend + daily_return).rolling(window=5).mean()
    
    # Measure Sequential Momentum
    sequential_momentum = intraday_price_movement.rolling(window=5).sum().ewm(span=3, adjust=False).mean()
    
    # Integrate Intraday and Historical Effects
    integrated_intraday_historical_effects = (intraday_price_movement * relative_volume) * combined_metrics_volatility
    
    # Final Alpha Factor: Integrate Price Momentum and Volume Confirmation
    momentum = df['close'].pct_change().rolling(window=10).sum()
    positive_volume_diff = (df['volume'].diff() > 0).rolling(window=10).sum()
    volume_confirmation = positive_volume_diff / 10
    final_alpha_factor = sequential_momentum * integrated_intraday_historical_effects
    final_alpha_factor = final_alpha_factor.where(volume_confirmation > 0, final_alpha_factor * 0.9)
    
    # Incorporate Trade Amount Analysis
    ma_amount_10 = df['amount'].rolling(window=10).mean()
    weighted_amount_trend = (ma_amount_10 - df['amount']).abs()
    
    # Assess Intraday Amount Impact
    relative_amount = df['amount'] / ma_amount_10
    
    # Combine Metrics for Liquidity
    combined_metrics_liquidity = (high_low_spread * weighted_amount_trend + daily_return).rolling(window=5).mean()
    
    # Final Alpha Factor with Liquidity
    final_alpha_factor *= combined_metrics_liquidity
    
    return final_alpha_factor
```
