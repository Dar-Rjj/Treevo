```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (intraday_price_range / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * intraday_return
    
    # Combine Intraday Factors
    combined_intraday_factors = proportional_movement_index + volume_weighted_adjusted_return
    
    # Calculate Simple Moving Averages (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Momentum Component
    momentum_component = sma_5 - sma_20
    
    # Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = momentum_component * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap /= df['volume'].rolling(window=1).sum()  # Normalize by total volume in the period
    
    # Final Combined Intraday and Momentum Components
    final_combined = combined_momentum_volatility + combined_intraday_factors
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_days = (daily_volume_change > 0).astype(int)
    average_volume_increase = (positive_volume_days * daily_volume_change).rolling(window=20).sum() / positive_volume_days.rolling(window=20).sum().replace(0, 1)
    
    # Calculate Cumulative Daily Price Change
    lookback_window = 20
    cumulative_daily_price_change = df['close'].pct_change().rolling(window=lookback_window).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=lookback_window).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_price_change = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range_normalized = (df['high'] - df['low']) / df['close']
    final_alpha_factor = high_low_range_normalized * weighted_cumulative_price_change
    
    return final_alpha_factor
```
