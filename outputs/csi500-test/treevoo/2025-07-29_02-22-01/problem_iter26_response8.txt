```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=10, min_periods=10).mean()
    long_ema = df['close'].ewm(span=50, min_periods=50).mean()
    price_momentum = long_ema - short_ema
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    intraday_range = df['high'] - df['low']
    proportional_movement_index = (intraday_range / df['close']) * intraday_return
    
    # Volume-Weighted Adjusted Return
    daily_intraday_range = df['high'] - df['low']
    volume_weighted_return = (df['volume'] * daily_intraday_range * intraday_return) / (df['volume'] * daily_intraday_range).rolling(window=20).sum()
    
    # Simple Moving Averages (SMA)
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    
    # Momentum Component
    momentum_component = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = momentum_component * (1 / volatility)
    
    # Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap.cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # 10-Day Volume-Weighted Moving Average of Volume-Adjusted High-Low Spread
    volume_adjusted_high_low = (df['high'] - df['low']) * df['volume']
    vwa_high_low_spread = (volume_adjusted_high_low.rolling(window=10).sum()) / (df['volume'].rolling(window=10).sum())
    
    # 10-Day Moving Average of Volume-Adjusted Daily Price Change
    volume_adjusted_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    moving_avg_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=10).mean()
    
    # Incorporate High-Low Range
    daily_high_low_spread = df['high'] - df['low']
    high_low_range_component = daily_high_low_spread * (price_momentum + proportional_movement_index + volume_weighted_return + combined_momentum_volatility + vwap + rsi + vwa_high_low_spread + moving_avg_volume_adjusted_price_change)
    
    # Final Alpha Factor
    final_alpha_factor = (
        price_momentum + 
        proportional_movement_index + 
        volume_weighted_return + 
        combined_momentum_volatility + 
        vwap + 
        rsi + 
        vwa_high_low_spread + 
        moving_avg_volume_adjusted_price_change + 
        high_low_range_component
    )
    
    return final_alpha_factor
```
