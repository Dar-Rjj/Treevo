```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = long_ema - short_ema
    
    # Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Momentum Calculation (Sum of Daily Price Changes over the last N days)
    momentum = daily_price_change.rolling(window=5).sum()
    
    # Volume Confirmation
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_increase = daily_volume_change[daily_volume_change > 0].rolling(window=5).sum() / (daily_volume_change > 0).rolling(window=5).sum()
    
    # Calculate Volume Reversal Indicator
    volume_rate_of_change = df['volume'].pct_change(periods=5)
    smoothed_volume_rate_of_change = volume_rate_of_change.rolling(window=5).mean()
    volume_reversal_indicator = price_momentum * (1 / smoothed_volume_rate_of_change)
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Proportional Movement Index
    proportional_movement_index = (intraday_return / df['close']) * intraday_return
    
    # Calculate Cumulative Daily Price Change
    cumulative_daily_price_change = daily_price_change.rolling(window=5).sum()
    
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_daily_price_change = cumulative_daily_price_change * average_volume
    
    # Incorporate High-Low Range
    high_low_range = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)
    final_alpha_factor = weighted_cumulative_daily_price_change * high_low_range
    
    # Incorporate Open-Close Spread
    open_close_spread = abs(df['open'] - df['close'])
    enhanced_alpha_factor = final_alpha_factor + open_close_spread
    
    return enhanced_alpha_factor
```
