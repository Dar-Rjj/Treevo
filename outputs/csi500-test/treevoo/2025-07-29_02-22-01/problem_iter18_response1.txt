import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    short_term_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=25, adjust=False).mean()
    price_momentum = long_term_ema - short_term_ema
    
    # Volume Acceleration
    smoothed_volume = df['volume'].ewm(span=5, adjust=False).mean()
    volume_acceleration = price_momentum * (1 / smoothed_volume)
    
    # Volume-Adjusted High-Low Spread
    vol_adj_high_low_spread = (df['high'] - df['low']) * df['volume'].pow(0.5)
    
    # 10-Day Moving Sum of Volume-Adjusted High-Low Spread
    ten_day_sum_vol_adj_spread = vol_adj_high_low_spread.rolling(window=10).sum()
    
    # Combine Volume-Adjusted High-Low Spread and Price Momentum
    combined_factor = ten_day_sum_vol_adj_spread * price_momentum
    
    # Cumulative Daily Price Change
    cumulative_price_change = df['close'].pct_change().rolling(window=14).sum()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=14).mean()
    
    # Weighted Cumulative Daily Price Change
    weighted_cumulative_price_change = cumulative_price_change * average_volume.pow(0.5)
    
    # Incorporate High-Low Range
    daily_high_low_range = df['high'] - df['low']
    normalized_high_low = daily_high_low_range / df['open']
    adjusted_cumulative_momentum = normalized_high_low * cumulative_price_change
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Proportional Movement Index
    proportional_movement_index = (df['high'] - df['low']) * intraday_return
    
    # Volume-Weighted Intraday Return
    vol_weighted_intraday_return = intraday_return * df['volume'].pow(0.5)
    
    # Final Alpha Factor
    final_alpha_factor = (combined_factor + 
                          proportional_movement_index + 
                          vol_weighted_intraday_return)
    
    return final_alpha_factor
