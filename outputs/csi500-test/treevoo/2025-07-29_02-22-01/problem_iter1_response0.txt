```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    # Calculate Momentum Component
    df['momentum'] = df['5_day_SMA'] - df['20_day_SMA']

    # Calculate Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()

    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['momentum'] * (1 / df['volatility'])

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    total_volume = df['volume'].rolling(window=20).sum()
    vwap = df['vwap'].rolling(window=20).sum() / total_volume
    df['vwap'] = vwap

    # Enhance with Volume-Weighted Price
    df['final_alpha_factor'] = df['combined_momentum_volatility'] + df['vwap']

    return df['final_alpha_factor']
```
