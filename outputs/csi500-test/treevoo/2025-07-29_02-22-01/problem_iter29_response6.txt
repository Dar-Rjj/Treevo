```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open + close) / 2
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the average of (high + low) / 2
    mid_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the average price and the midpoint price
    price_diff = avg_price - mid_price
    
    # Calculate the price range (high - low)
    price_range = df['high'] - df['low']
    
    # Calculate the factor value: (price_diff / price_range) * (volume / amount)
    factor_value = (price_diff / price_range) * (df['volume'] / df['amount'])
    
    return factor_value
```
