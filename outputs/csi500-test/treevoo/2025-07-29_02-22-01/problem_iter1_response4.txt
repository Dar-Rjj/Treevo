import pandas as pd

def heuristics_v2(data):
    # Calculate Daily Return
    data['daily_return'] = data['close'].diff()
    
    # Define decay factor and window size
    decay_factor = 0.94
    window_size = 15
    
    # Apply Exponential Weights to Returns
    weights = [decay_factor**i for i in range(window_size)]
    weighted_returns = data['daily_return'].rolling(window=window_size).apply(
        lambda x: (x * weights[:len(x)]).sum(), raw=True
    )
    
    # Aggregate Volume of the Past N Days
    aggregated_volume = data['volume'].rolling(window=window_size).sum()
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = weighted_returns / aggregated_volume
    
    return volume_adjusted_momentum.dropna()
