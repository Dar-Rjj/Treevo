```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Combine High-Low and Open-Close
    adjusted_opening_gap = (df['open'] - df['close'].shift(1))
    combined_high_low_open_close = intraday_range + adjusted_opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).apply(lambda x: max(x, 0))
    
    # Calculate Momentum Component
    momentum_component = (df['close'].pct_change().rolling(window=5).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10
    dynamic_coefficient = volume_change_rate.apply(lambda x: 1.7 if x > 2 else (-1.5 if x < -2 else 0.8))
    
    # Incorporate Price Action Strength
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_action_direction = (ma_5 > ma_20).astype(int) - (ma_5 < ma_20).astype(int)
    price_action_multiplier = momentum_component * price_action_direction
    
    # Integrate Exponential Price Trend
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    exponential_price_trend = df['close'] - ema_close
    
    # Combined Signal
    combined_signal = (intraday_return * df['volume'] + short_term_price_change + volume_adjusted_spread)
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_action_multiplier + combined_signal).apply(lambda x: max(x, 0))
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * dynamic_coefficient * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # Enhance Combined Momentum, Volume, and Price Factor
    opening_impact_factor = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    enhanced_factor = (price_action_multiplier * dynamic_coefficient * opening_impact_factor / df['volume'].rolling(window=20).mean()) * exponential_price_trend
    
    # Opening Impact and Gap Adjustment
    overnight_gap = df['open'] - df['close'].shift(1)
    enhanced_factor -= overnight_gap
    
    return enhanced_factor
```
