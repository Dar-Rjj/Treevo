```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Adjusted Intraday Range with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_intraday_range = intraday_range * scaled_volume

    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_score = volume_change_rate.apply(lambda x: 2.0 if x > 1 else -1.5)

    # Combined Momentum and Volume Factor
    n_days = 10
    sum_returns_n_days = df['close'].pct_change().rolling(window=n_days).sum()
    combined_momentum_volume_factor = sum_returns_n_days * close_to_range_ratio * volume_trend_score

    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    intermediate_factor = adjusted_intraday_range + (open_to_close_return * scaled_volume * sum_returns_n_days * volume_trend_score)

    # Price Momentum
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20d

    # Volume Trend
    sum_volume_20d = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / sum_volume_20d

    # Price Oscillator
    ema_7d = df['close'].ewm(span=7, adjust=False).mean()
    ema_14d = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7d - ema_14d

    # Integrated Composite Factor
    integrated_factor = intermediate_factor + price_oscillator

    # Smoothed Composite Factor
    sma_7d = integrated_factor.rolling(window=7).mean()
    
    # Additional Momentum Component
    ema_50d = df['close'].ewm(span=50, adjust=False).mean()
    additional_momentum = df['close'] - ema_50d

    # Enhanced Final Alpha Factor
    enhanced_alpha_factor = sma_7d + additional_momentum

    return enhanced_alpha_factor
```
