import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['low']
    
    # Calculate Volume-Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'].pct_change(periods=5)
    
    # Adjust for Volume Activity
    average_volume_20_days = df['volume'].rolling(window=20).mean().shift(1)
    short_term_volume_change = df['volume'].diff(periods=5)
    
    # Volume Trend Indicator
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Combine Intraday Signals with Price Momentum
    combined_signal = volume_adjusted_intraday_return + short_term_price_change
    
    # Final Factor Adjustment
    final_factor_adjustment = combined_signal * volume_trend_indicator
    
    # Final Alpha Factor
    final_alpha_factor = final_factor_adjustment / average_volume_20_days
    
    return final_alpha_factor
