```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'].diff(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    final_combined_factor = final_combined_factor.apply(lambda x: max(x, 0))
    
    # Volume-Adjusted Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5_days) * close_to_intraday_range_ratio
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + 
                       short_term_price_change + 
                       volume_weighted_high_low_spread)
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(df['volume'].pct_change(10) > 0, 3, -3)
    
    # Adaptive Volume Trend Coefficient
    volume_change_5_days = df['volume'] - df['volume'].shift(5)
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5_days / avg_volume_10_days
    adaptive_volume_trend_coefficient = np.where(volume_change_rate > 1.5, 2.0,
                                                 np.where(volume_change_rate < -1, -1.5, 0.5))
    volume_trend_score = volume_trend_indicator * adaptive_volume_trend_coefficient
    
    # Volume-Weighted Intraday Return (scaled by last 15 days' average)
    avg_volume_15_days = df['volume'].rolling(window=15).mean()
    scaled_volume = df['volume'] / avg_volume_15_days
    volume_weighted_intraday_return_scaled = (df['close'] - df['open']) * scaled_volume
    
    # Volume-Adjusted High-Low Spread (scaled by last 15 days' average)
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * scaled_volume
    
    # Final Combined Factor
    final_combined_factor = (volume_adjusted_momentum * volume_trend_score + 
                             volume_weighted_intraday_return_scaled + 
                             volume_adjusted_high_low_spread)
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_combined_factor - open_to_close_return
    
    return final_alpha_factor
```
