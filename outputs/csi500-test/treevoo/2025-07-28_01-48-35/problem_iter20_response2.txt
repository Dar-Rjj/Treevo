def heuristics_v2(df):
    """
    Generate an interpretable alpha factor based on a heuristic approach.
    
    The factor computes the weighted sum of percentage changes in close price and volume,
    where the weights are determined by the ratio of amount to volume, providing insight
    into the intensity of money flow relative to the activity.

    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A series representing the computed alpha factor.
    """
    
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the percentage change in volume
    pct_change_volume = df['volume'].pct_change()
    
    # Calculate the weight as the ratio of amount to volume
    weight = df['amount'] / df['volume']
    
    # Normalize the weight by its rolling 5-day mean to smooth out short-term noise
    weight_normalized = weight / weight.rolling(window=5).mean()
    
    # Compute the heuristic alpha factor
    heuristic_factor = (pct_change_close * weight_normalized) + (pct_change_volume * (1 - weight_normalized))
    
    return heuristic_factor
