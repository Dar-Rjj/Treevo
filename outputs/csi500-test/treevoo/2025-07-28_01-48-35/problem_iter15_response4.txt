```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_intraday_ratio
    
    # Apply Rolling Sum on Volume Adjusted Momentum with a window of 5 days
    rolling_sum_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()
    
    # Calculate Two-Day Moving Average
    two_day_moving_average = (df['close'].shift(1) + df['close']) / 2
    
    # Compute Daily Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Integrate with Two-Day Moving Average Return
    final_factor = rolling_sum_volume_adjusted_momentum * daily_return / two_day_moving_average
    
    return final_factor
```
