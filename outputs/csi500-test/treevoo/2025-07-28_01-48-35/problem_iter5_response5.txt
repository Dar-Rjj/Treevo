import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'].diff(5)

    # Adjust for Volume Activity
    avg_volume_20_days = df['volume'].rolling(window=20).mean().shift(1)
    short_term_volume_change = df['volume'].diff(5)
    
    # Volume Trend Indicator
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Volume Trend Score
    avg_volume_last_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_last_10_days
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 if x > 1 else -1.2 if x < 1 else 0)
    
    # Final VAPM Factor
    final_vapm_factor = (short_term_price_change * volume_trend_indicator) / avg_volume_20_days

    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_weighted_cro_return = close_to_open_return * df['volume']

    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_last_10_days = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread_last_10_days

    # Composite Alpha Factor
    combined_factor = final_vapm_factor * normalized_high_low_spread
    composite_factor = volume_weighted_cro_return * combined_factor

    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()

    # Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor
    return final_alpha_factor
