```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 if x > 1 else (-1.2 if x < 1 else 0))
    
    # Combined Momentum and Volume Factor
    n_days = 5
    momentum = df['close'].pct_change().rolling(window=n_days).sum()
    momentum_and_volume_factor = momentum * close_to_range_ratio * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    adjusted_momentum_and_volume_factor = momentum_and_volume_factor - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_to_open_return = close_to_open_return * scaled_volume
    intermediate_composite_factor = adjusted_intraday_range + volume_weighted_close_to_open_return * volume_trend_score
    
    # Price Momentum
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20d
    
    # Price Oscillator
    ema_5d = df['close'].ewm(span=5, adjust=False).mean()
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5d - ema_10d
    
    # Integrated Composite Factor
    integrated_composite_factor = intermediate_composite_factor + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite_factor = integrated_composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
