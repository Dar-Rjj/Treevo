import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio and adjust for Volume Activity
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, -1.5)
    
    # Combined Momentum and Volume Factor
    returns_sum = df['close'].pct_change().rolling(window=5).sum()
    combined_momentum_volume_factor = returns_sum * close_to_range_ratio * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    intermediate_composite_factor = adjusted_intraday_range + open_to_close_return * scaled_volume
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Volume Trend
    volume_sum_20d = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / volume_sum_20d
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite_factor = intermediate_composite_factor + price_oscillator
    
    # Smoothed Composite Factor
    sma_7 = integrated_composite_factor.rolling(window=7).mean()
    smoothed_composite_factor = sma_7
    
    # Additional Momentum Component
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    additional_momentum_component = df['close'] - ema_50
    
    # Enhanced Final Alpha Factor
    enhanced_final_alpha_factor = smoothed_composite_factor + additional_momentum_component
    
    return enhanced_final_alpha_factor
