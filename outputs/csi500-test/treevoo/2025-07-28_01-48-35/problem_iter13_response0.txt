```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'].diff(periods=5)
    
    # Adjust for Volume Activity
    volume_change = df['volume'].diff(periods=5)
    volume_trend_indicator = (volume_change > 0).astype(int) * 2 - 1  # +1 if true, -1 if false
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_open_return = (df['close'] - df['open']) * df['volume']
    
    # Combine Signals
    combined_signal = intraday_return + short_term_price_change + high_low_spread + close_open_return
    
    # Final Factor
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5d
    final_factor = combined_signal * scaled_volume * volume_trend_indicator

    return final_factor
```
