```python
import pandas as pd

def heuristics_v2(df, window=10, volume_spike_threshold=2.0):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_intraday_range_ratio'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['volume'] * df['close_to_intraday_range_ratio']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > volume_spike_threshold).astype(int)
    
    # Integrate Price Trend
    df['sma_close'] = df['close'].rolling(window=window).mean()
    df['price_trend'] = df['close'] - df['sma_close']
    
    # Combine Components
    df['combined_components'] = df['volume_adjusted_momentum'] + df['price_trend'] + df['volume_spike']
    
    # Apply Rolling Sum on Combined Components
    df['cumulative_sum'] = df['combined_components'].rolling(window=window).sum()
    
    return df['cumulative_sum']
```
