```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day to the current day
    df['price_change'] = df['close'].diff()
    
    # Calculate the signed volume, which is the volume adjusted for price direction
    df['signed_volume'] = df['volume'] * (df['close'] - df['open']).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate the buying pressure, which is a measure of how much more buying than selling is happening
    df['buying_pressure'] = (df['close'] - df['low']) / (df['high'] - df['low']) * df['volume']
    
    # Combine the three indicators into a single alpha factor
    df['alpha_factor'] = (df['price_change'] + df['signed_volume'] + df['buying_pressure'])

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
