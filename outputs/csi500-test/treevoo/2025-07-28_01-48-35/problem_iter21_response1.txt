```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    prev_close = df['close'].shift(1)
    opening_gap = df['open'] - prev_close
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = high_low_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = (df['close'] / prev_close).abs()
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume'] ** 0.5
    
    # Calculate Average Intraday Return over N days (e.g., 20 days)
    n_days = 20
    avg_intraday_return = intraday_return.rolling(window=n_days).mean()
    
    # Combine Initial Signals
    combined_initial_signals = avg_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_10d
    volume_trend_score = (volume_change_rate > 1.5) * 2.0 + (volume_change_rate < 0.8) * -1.5
    
    # Calculate Price Momentum
    avg_high_low_spread_20d = high_low_range.rolling(window=20).mean()
    normalized_high_low_spread = high_low_range / avg_high_low_spread_20d
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (final_combined_factor * normalized_high_low_spread * volume_trend_score +
                          combined_initial_signals)
    
    return final_alpha_factor
```
