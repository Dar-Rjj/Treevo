import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Intraday Return
    avg_vol_15 = df['volume'].rolling(window=15).mean()
    volume_weighted_intraday_return = (df['close'] - df['open']) / df['open'] * (df['volume'] / avg_vol_15)
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    adjusted_opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_and_open_close = intraday_range + adjusted_opening_gap
    
    # Measure Closing Behavior
    close_to_previous_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = combined_high_low_and_open_close * close_to_previous_close_return
    final_combined_factor = final_combined_factor.apply(lambda x: max(x, 0))
    
    # Volume-Adjusted Momentum
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_vol_5) * close_to_intraday_range_ratio
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (intraday_range) * (df['volume'] / avg_vol_15)
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=20).mean()
    
    # Combined Signal
    combined_signal = volume_weighted_intraday_return + short_term_price_change + volume_weighted_high_low_spread
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(df['volume'].diff(10) > 0, 3, -3)
    
    # Adaptive Volume Trend Coefficient
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_vol_10
    adaptive_coefficient = np.where(volume_change_rate > 1.5, 2.0, 
                                    np.where(volume_change_rate < -1, -1.5, 0.5))
    adaptive_volume_trend = volume_trend_indicator * adaptive_coefficient
    
    # Composite Factor
    composite_factor = final_combined_factor * price_momentum + combined_signal
    composite_factor = composite_factor.apply(lambda x: max(x, 0))
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=7).mean()
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (intraday_range) * (df['volume'] / avg_vol_15)
    
    # Final Combined Factor
    final_combined_factor = (composite_factor * adaptive_volume_trend) + volume_weighted_intraday_return + volume_adjusted_high_low_spread
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_combined_factor - open_to_close_return
    
    return final_alpha_factor
