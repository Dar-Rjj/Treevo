```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_c2o = close_to_open_return * df['volume']
    
    # 10-Day Exponential Moving Average of Close Prices
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    price_momentum = df['close'] - ema_10_close
    
    # 10-Day Sum of Volume
    sum_10_volume = df['volume'].rolling(window=10).sum()
    
    # Volume Trend
    volume_trend = df['volume'] / sum_10_volume
    
    # Combine Volume-Weighted Close-to-Open Return, Price Momentum, and Volume Trend
    composite_factor = volume_weighted_c2o * price_momentum * volume_trend
    
    # 5-Day Simple Moving Average (SMA) of Composite Factor
    sma_5_composite = composite_factor.rolling(window=5).mean()
    
    return sma_5_composite
```
