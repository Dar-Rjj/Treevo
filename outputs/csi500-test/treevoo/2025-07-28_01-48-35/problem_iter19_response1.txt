```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).apply(lambda x: max(x, 0))
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Composite Factor
    composite_factor = (volume_weighted_return * price_momentum) + final_combined_factor
    
    # Smooth the Composite Factor
    smoothed_composite = composite_factor.rolling(window=5).mean()
    
    # Calculate Volume Trend Score
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_10
    volume_trend_score = (volume_change_rate > 1.7) * 2.5 + (volume_change_rate < 0.7) * (-2.0)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite * volume_trend_score
    
    return final_alpha_factor
```
