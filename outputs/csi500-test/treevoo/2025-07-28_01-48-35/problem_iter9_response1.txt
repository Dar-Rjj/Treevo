```python
import pandas as pd

def heuristics_v2(df, sma_window=20, vol_spike_threshold=2.0, cumsum_window=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Detect Volume Spikes
    volume_change = df['volume'] / df['volume'].shift(1)
    volume_spike = (volume_change > vol_spike_threshold).astype(int)
    
    # Integrate Price Trend
    sma_close = df['close'].rolling(window=sma_window).mean()
    price_trend = df['close'] - sma_close
    
    # Combine and Cumulate
    combined_factor = volume_weighted_momentum + price_trend + volume_spike
    cumulative_factor = combined_factor.rolling(window=cumsum_window).sum()
    
    return cumulative_factor
```
