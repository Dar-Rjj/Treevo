import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Spread
    df['spread'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    df['volume_adjusted_spread'] = df['spread'] * df['volume']
    
    # Calculate Momentum Component
    df['momentum_component'] = df['close'].pct_change(N).rolling(window=N).sum()
    df['momentum_close_to_range'] = df['close_to_range_ratio'] * df['momentum_component']
    
    # Adjust for Volume Activity
    df['5_day_volume_change'] = df['volume'] - df['volume'].shift(5)
    df['volume_trend_indicator'] = df['5_day_volume_change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Average Intraday Return
    df['average_intraday_return'] = df['intraday_return'].rolling(window=N).mean()
    
    # Combine Initial Signals
    df['combined_initial_signals'] = df['average_intraday_return'] + df['volume_adjusted_spread']
    
    # Calculate Price Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['average_high_low_spread'] = df['high_low_spread'].rolling(window=10).mean()
    df['relative_high_low_spread'] = df['high_low_spread'] / df['high_low_spread'].shift(1)
    
    # Introduce Opening Impact Factor
    df['opening_impact_factor'] = df['close'] - df['open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    df['enhanced_combined_factor'] = (
        df['momentum_close_to_range'] * 
        df['volume_trend_indicator'] * 
        df['opening_impact_factor'] / 
        df['volume'].rolling(window=20).mean() * 
        df['volume_trend_indicator'] * 
        df['relative_high_low_spread']
    )
    
    return df['enhanced_combined_factor'].dropna()
