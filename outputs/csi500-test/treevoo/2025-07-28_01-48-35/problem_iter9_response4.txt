```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Advanced Price Momentum
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = (df['close'] - df['EMA_20']) / (df['open'] - df['low'])

    # Calculate Volume-Weighted Price Trend
    df['volume_weighted_price'] = df['close'] * df['volume']
    df['sum_volume_weighted_price_20'] = df['volume_weighted_price'].rolling(window=20).sum()
    df['sum_volume_20'] = df['volume'].rolling(window=20).sum()
    df['volume_weighted_price_trend'] = df['sum_volume_weighted_price_20'] / df['sum_volume_20']

    # Composite Factor
    df['composite_factor'] = df['price_momentum'] * df['volume_weighted_price_trend']
    
    # Smooth the Composite Factor
    df['smoothed_composite_factor'] = df['composite_factor'].ewm(span=10, adjust=False).mean()

    # Final Alpha Factor
    alpha_factor = df['smoothed_composite_factor']
    
    return alpha_factor
```
