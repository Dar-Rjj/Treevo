```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    momentum_component = (df['close'].diff().rolling(window=5).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10d
    volume_trend_score = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    ema_close_price = df['close'].ewm(span=5, adjust=False).mean()
    price_trend = df['close'] - ema_close_price
    
    # Combine Initial Signals
    combined_signal = (intraday_return.mean() + volume_adjusted_spread)
    
    # Introduce Opening Impact Factor
    opening_impact_factor = (df['open'] - df['close']).shift(1)  # Yesterday's Open-to-Close return

    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (momentum_component * volume_trend_score * opening_impact_factor 
                       / df['volume'].rolling(window=20).mean() * price_trend)
    
    # Subtract Overnight Gap
    overnight_gap = df['open'] - df['close'].shift(1)
    final_alpha_factor = enhanced_factor - overnight_gap
    
    return final_alpha_factor
```
