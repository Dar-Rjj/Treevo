import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = (df['high'] - df['low']) + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    final_combined_factor = np.abs(final_combined_factor)  # Ensure positive values
    
    # Volume-Adjusted Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5_days) * close_to_intraday_range_ratio
    
    # Cumulative Volume-Adjusted Momentum
    cumulative_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    volume_factor = df['volume'] / avg_volume_5_days
    volume_weighted_high_low_spread = (df['high'] - df['low']) * volume_factor
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + short_term_price_change + volume_weighted_high_low_spread) / 3
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(df['volume'].rolling(window=10).mean().diff() > 0, 2, -2)
    
    # Composite Factor
    composite_factor = final_combined_factor * price_momentum + combined_signal
    composite_factor = np.abs(composite_factor)  # Ensure positive values
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_indicator
    
    # Final Factor
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10_days
    final_factor = final_alpha_factor * scaled_volume
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    overnight_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_factor -= open_to_close_return - overnight_gap
    
    # Enhanced Combined Momentum, Volume, and Price Factor
    enhanced_combined_factor = volume_adjusted_momentum * price_momentum - open_to_close_return
    enhanced_combined_factor = np.abs(enhanced_combined_factor)  # Ensure positive values
    
    # Output as Final Alpha Factor
    alpha_factor = enhanced_combined_factor
    return alpha_factor
