```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_intraday_factor = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_intraday_factor * close_to_prev_close_return).apply(lambda x: max(x, 0))
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + 
                       short_term_price_change + 
                       volume_weighted_close_to_open_return)
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum + combined_signal).apply(lambda x: max(x, 0))
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Final Alpha Factor
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_trend_score = df['volume'] / avg_volume_10
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    return final_alpha_factor
```
