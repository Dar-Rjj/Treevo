```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
                       
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the relative change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate a signal based on the relationship between price difference and volume change
    signal = (price_diff * volume_change).rolling(window=5).mean()
    
    return signal
```
