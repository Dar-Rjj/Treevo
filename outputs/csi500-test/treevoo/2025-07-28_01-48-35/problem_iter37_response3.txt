```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    N = 10
    momentum_component = df['close'].pct_change().rolling(window=N).sum() * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10_days
    volume_trend_coefficient = volume_change_rate.apply(lambda x: 1.7 if x > 2 else (-1.5 if x < -2 else 0.8))
    
    # Calculate Volume-Weighted Intraday Return
    intraday_return = df['close'] - df['open']
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume-Adjusted High-Low Spread
    avg_volume_15_days = df['volume'].rolling(window=15).mean()
    scaled_volume = df['volume'] / avg_volume_15_days
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * scaled_volume
    
    # Detect Volume Spikes
    volume_change_daily = df['volume'] - df['volume'].shift(1)
    volume_spike = volume_change_daily.abs() > volume_change_daily.std()
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * scaled_volume
    
    # Intermediate Composite Factor
    intermediate_composite_factor = (volume_adjusted_high_low_spread + volume_weighted_intraday_return) * close_to_open_return * price_trend * scaled_volume
    
    # Integrated Composite Factor
    integrated_composite_factor = (price_trend + (df['close'] - df['close'].shift(1))) + volume_trend_coefficient * scaled_volume
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Enhanced Composite Factor with Oscillator
    enhanced_composite_factor = integrated_composite_factor + price_oscillator
    sma_5_enhanced = enhanced_composite_factor.rolling(window=5).mean()
    weighting_factor = df['volume'] / avg_volume_10_days
    enhanced_composite_factor = sma_5_enhanced * weighting_factor
    
    # Final Combined Factor
    final_combined_factor = (momentum_component * volume_trend_coefficient) + volume_weighted_intraday_return + volume_adjusted_high_low_spread
    
    return final_combined_factor
```
