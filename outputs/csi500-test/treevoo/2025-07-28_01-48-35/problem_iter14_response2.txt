import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = df['close'] / intraday_range
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_intraday_ratio
    
    # Detect Volume Spikes
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > volume_change.abs().mean()).astype(int)
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Combine Components
    combined_components = volume_adjusted_momentum + price_trend + volume_spike
    rolling_sum = combined_components.rolling(window=5).sum()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjusted High-Low Spread with Volume Trend
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / average_volume_10_days
    adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    
    # Intermediate Composite Factor
    intermediate_composite = adjusted_high_low_spread + volume_weighted_close_to_open * price_trend * scaled_volume
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite = integrated_composite.rolling(window=5).mean()
    
    return smoothed_composite
