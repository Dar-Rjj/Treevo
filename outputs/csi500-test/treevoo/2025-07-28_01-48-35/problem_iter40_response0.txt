```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Short-Term Price Change
    short_term_change = df['close'].diff(5)
    
    # Adjust for Opening Gap
    overnight_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_signal = (df['high'] - df['low']) + overnight_gap
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = (combined_signal * closing_behavior).clip(lower=0)
    
    # Calculate Momentum Component
    momentum = df['close'].pct_change().rolling(window=5).sum()
    momentum_component = momentum * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10_days
    dynamic_coefficient = volume_change_rate.apply(lambda x: 2.0 if x > 2 else -1.5 if x < -1 else 0.8)
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Volume-Weighted Intraday Return
    vwintraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Volume-Weighted High-Low Spread
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    vw_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)
    
    # Composite Factor
    composite_factor = (final_combined_factor * momentum_component) + vwintraday_return
    composite_factor = composite_factor.clip(lower=0)
    
    # Smooth the Composite Factor
    smoothed_composite = composite_factor.rolling(window=5).mean()
    
    # Volume Trend Indicator
    volume_trend_score = (df['volume'].rolling(window=10).sum() > 0).astype(int) * 4 - 2
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite * volume_trend_score
    
    # Opening Impact and Gap Adjustment
    opening_impact = final_alpha_factor - overnight_gap
    
    # Final Factor
    scaled_volume = df['volume'] / avg_volume_10_days
    final_factor = opening_impact * scaled_volume
    
    return final_factor
```
