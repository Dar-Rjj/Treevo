```python
def heuristics_v2(df):
    # Calculate the price movement from the open to close
    price_movement = df['close'] - df['open']
    
    # Calculate the total range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the volume-weighted price
    vwp = (df['close'] * df['volume']).rolling(window=5).mean() / df['volume'].rolling(window=5).mean()
    
    # Calculate the price change over a 5-day period
    price_change_5d = df['close'].pct_change(5)
    
    # Calculate the alpha factor
    alpha_factor = (price_movement / daily_range) * vwp * price_change_5d

    return alpha_factor
```
