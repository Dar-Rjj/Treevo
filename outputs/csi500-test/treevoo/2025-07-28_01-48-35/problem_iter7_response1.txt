```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Volume Weighted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_factor = df['volume'] / avg_volume_5d
    volume_weighted_momentum = close_to_range_ratio * volume_factor
    
    # Cumulative Sum of Volume Weighted Momentum
    cum_sum_momentum = volume_weighted_momentum.rolling(window=10).sum()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * (df['volume'] / avg_volume_5d)
    
    # Combine Factors
    combined_factor = cum_sum_momentum + volume_adjusted_high_low_spread
    
    return combined_factor
```
