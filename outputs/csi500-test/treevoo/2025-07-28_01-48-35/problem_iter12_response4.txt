```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_range_ratio
    
    # Apply Rolling Sum on Volume Adjusted Momentum
    rolling_sum = volume_adjusted_momentum.rolling(window=10).sum()
    
    # Integrate with Previous Day's Return
    previous_close = df['close'].shift(1)
    daily_return = (df['close'] - previous_close) / previous_close
    factor_value = rolling_sum * daily_return
    
    return factor_value
```
