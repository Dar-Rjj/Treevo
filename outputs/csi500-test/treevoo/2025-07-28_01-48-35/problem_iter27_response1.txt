```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_vol_10d
    adj_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_vol_10d
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, -1.5)
    
    # Combined Momentum and Volume Factor
    returns_n_days = df['close'].pct_change().rolling(window=20).sum()
    combined_momentum_volume = returns_n_days * close_to_range_ratio * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_volume - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = (df['close'] - df['open'])
    vol_weighted_close_to_open = close_to_open_return * scaled_volume
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    intermediate_composite = adj_intraday_range + (vol_weighted_close_to_open * price_momentum * volume_trend_score)
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    sma_7 = integrated_composite.rolling(window=7).mean()
    final_alpha_factor = sma_7
    
    # Additional Momentum Component
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    additional_momentum = df['close'] - ema_50
    enhanced_final_alpha_factor = final_alpha_factor + additional_momentum
    
    # High-Low Price Momentum
    high_low_ema_30 = (df['high'] - df['low']).ewm(span=30, adjust=False).mean()
    high_low_momentum = (df['high'] - df['low']) - high_low_ema_30
    further_enhanced_alpha_factor = enhanced_final_alpha_factor + high_low_momentum
    
    return further_enhanced_alpha_factor
```
