```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    prev_close = df['close'].shift(1)
    open_gap = df['open'] - prev_close
    
    # Combine High-Low and Open-Close
    combined_hl_open = high_low_range + open_gap
    
    # Measure Closing Behavior
    close_return = df['close'] / prev_close
    final_combined_factor = (combined_hl_open * close_return).abs()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - prev_close
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open']
    
    # Apply Volume and Open-to-Close Return Filter
    volume_filter = df['volume'] > 0
    return_filter = close_to_open_return > 0
    filtered_data = (volume_filter & return_filter)
    
    # Calculate Volume Trend Score
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 * x if x > 1 else -1.2 * x)
    
    # Calculate Price Momentum
    high_low_spread = high_low_range
    avg_high_low_spread = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread
    
    # Summarize Weighted Price Changes Over 10 Days
    weighted_price_changes = daily_price_change * filtered_data
    summarized_weighted_changes = weighted_price_changes.rolling(window=10).sum()
    
    # Combine All Factors
    final_alpha_factor = (final_combined_factor * normalized_high_low_spread * 
                          volume_trend_score * summarized_weighted_changes)
    
    return final_alpha_factor
```
