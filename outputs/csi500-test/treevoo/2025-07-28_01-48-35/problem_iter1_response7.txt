```python
def heuristics_v2(df, N=1000):
    # Calculate daily price change using close price
    df['price_change'] = df['close'].diff()
    
    # Apply volume filter to weight price changes
    df['weighted_price_change'] = df['price_change'] * (df['volume'] > N)
    
    # Summarize weighted price changes over 10 days
    df['factor_value'] = df['weighted_price_change'].rolling(window=10).sum()
    
    return df['factor_value']
```
