import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the average of (high - low) over 10 days
    df['avg_hl_10d'] = df['high'].sub(df['low']).rolling(window=10).mean()
    
    # Calculate the ratio of volume to amount
    df['volume_to_amount'] = df['volume'] / df['amount']
    
    # Generate a factor based on the interaction of the above features
    df['factor'] = df['close_change'] * df['avg_hl_10d'] * df['volume_to_amount']
    
    return df['factor']
