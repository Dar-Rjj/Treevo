```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range

    # Calculate Momentum Component
    N = 5  # Number of days for the momentum calculation
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio).shift(1)  # Shift to avoid look-ahead bias

    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, np.where(volume_change_rate < -1, -1.5, 0))

    # Integrate Price Trend
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_close

    # Combine Components
    combined_factor = (momentum_component * volume_trend_score) + price_trend

    # Opening Impact and Gap Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    overnight_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_factor -= open_to_close_return + overnight_gap

    return combined_factor
```
