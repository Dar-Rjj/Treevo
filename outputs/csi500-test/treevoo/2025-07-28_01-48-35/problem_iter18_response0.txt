```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_intraday_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Volume Adjusted Momentum
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_factor'] = df['volume'] / df['avg_volume_5d']
    df['volume_adjusted_momentum'] = df['close_to_intraday_ratio'] * df['volume_factor']
    df['cumulative_volume_adj_mom'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume_factor']
    
    # Calculate Price Momentum
    df['ema_close_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_close_20d']
    
    # Calculate Volume Trend
    df['sum_volume_30d'] = df['volume'].rolling(window=30).sum()
    df['volume_trend'] = df['volume'] / df['sum_volume_30d']
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Calculate Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    df['volume_adjusted_open_close_spread'] = df['open_close_spread'] * df['volume_factor']
    
    # Composite Factor
    df['composite_factor'] = (
        df['cumulative_volume_adj_mom'] +
        df['volume_adjusted_high_low_spread'] +
        df['price_momentum'] +
        df['volume_trend'] +
        df['obv'] +
        df['volume_adjusted_open_close_spread']
    )
    
    # Smooth the Composite Factor
    df['smoothed_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    
    return df['smoothed_composite_factor']
```
