```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute 20-Day Exponential Moving Average (EMA) of Close Prices
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Modified Price Momentum
    df['modified_price_momentum'] = (df['close'] - df['ema_20']) / (df['open'] - df['high'])

    # Compute 20-Day Sum of Volume-Weighted Close Prices
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['sum_volume_weighted_close_20'] = df['volume_weighted_close'].rolling(window=20).sum()

    # Compute 20-Day Sum of Volume
    df['sum_volume_20'] = df['volume'].rolling(window=20).sum()

    # Calculate Volume-Weighted Price Trend
    df['volume_weighted_price_trend'] = df['sum_volume_weighted_close_20'] / df['sum_volume_20']

    # Composite Factor
    df['composite_factor'] = df['modified_price_momentum'] * df['volume_weighted_price_trend']

    # Smooth the Composite Factor using 10-Day Simple Moving Average (SMA)
    df['smoothed_composite_factor'] = df['composite_factor'].rolling(window=10).mean()

    # Final Alpha Factor
    alpha_factor = df['smoothed_composite_factor']

    return alpha_factor
```
