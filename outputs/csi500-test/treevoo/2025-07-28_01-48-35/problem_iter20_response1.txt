```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the turnover amount per volume
    amount_per_volume = df['amount'] / df['volume']
    
    # Calculate the relative strength index (RSI)-like factor
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_like_factor = 100 - (100 / (1 + rs))
    
    # Combine the factors
    factor = close_to_avg_ratio * amount_per_volume * rsi_like_factor
    
    return factor
```
