```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = (df['High'] - df['Low']) / df['Low'] * df['Volume']
    
    # Short-Term Price Change
    df['Short_Term_Price_Change'] = df['Close'].diff(5)
    
    # Close to Intraday Range Ratio
    df['Intraday_Range'] = df['High'] - df['Low']
    df['Close_to_Intraday_Range_Ratio'] = df['Close'] / df['Intraday_Range']
    
    # Volume Adjusted Momentum
    avg_volume_5 = df['Volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = (df['Volume'] / avg_volume_5) * df['Close_to_Intraday_Range_Ratio']
    
    # Cumulative Volume Adjusted Momentum
    df['Cumulative_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    df['Volume_Weighted_High_Low_Spread'] = (df['High'] - df['Low']) * (df['Volume'] / avg_volume_5)
    
    # Price Momentum
    df['EMA_20'] = df['Close'].ewm(span=20, adjust=False).mean()
    df['Price_Momentum'] = df['Close'] - df['EMA_20']
    
    # Combined Signal
    df['Combined_Signal'] = df['Volume_Weighted_Intraday_Return'] + df['Short_Term_Price_Change'] + df['Volume_Weighted_High_Low_Spread']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['Close'] - df['Open']
    
    # Calculate Spread
    df['Spread'] = df['High'] - df['Low']
    
    # Calculate Volume Adjusted Spread
    df['Volume_Adjusted_Spread'] = df['Spread'] * df['Volume']
    
    # Calculate Momentum Component
    momentum_component = df['Close'].pct_change().rolling(window=5).sum()
    df['Momentum_Component'] = momentum_component * df['Close_to_Intraday_Range_Ratio']
    
    # Adjust for Volume Activity
    df['5_Day_Volume_Change'] = df['Volume'].diff(5)
    df['Volume_Trend_Indicator'] = np.where(df['5_Day_Volume_Change'] > 0, 1, -1)
    
    # Calculate Average Intraday Return
    df['Average_Intraday_Return'] = df['Intraday_Return'].rolling(window=5).mean()
    
    # Combine Initial Signals
    df['Combined_Initial_Signals'] = df['Average_Intraday_Return'] + df['Volume_Adjusted_Spread']
    
    # Calculate Price Momentum
    df['High_Low_Spread'] = df['High'] - df['Low']
    df['Average_High_Low_Spread_10_Days'] = df['High_Low_Spread'].rolling(window=10).mean()
    df['Relative_High_Low_Spread'] = df['High_Low_Spread'] / df['High_Low_Spread'].shift(1)
    
    # Introduce Opening Impact Factor
    df['Opening_Impact_Factor'] = df['Close'] - df['Open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    df['Enhanced_Combined_Factor'] = (df['Combined_Signal'] * df['Volume_Trend_Indicator'] * df['Opening_Impact_Factor']) / avg_volume_5 * (df['Volume_Trend_Indicator'] * df['Relative_High_Low_Spread'])
    
    # Output as Final Alpha Factor
    return df['Enhanced_Combined_Factor']
```
