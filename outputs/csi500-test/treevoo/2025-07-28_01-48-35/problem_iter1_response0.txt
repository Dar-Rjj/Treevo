```python
def heuristics_v2(df):
    # Calculate the High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume Adjustment Factor
    volume_5d_sum = df['volume'].rolling(window=5).sum()
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    volume_adjustment_factor = volume_5d_sum / volume_5d_avg
    
    # Calculate Adjusted High-Low Spread
    adjusted_high_low_spread = high_low_spread * volume_adjustment_factor
    
    # Calculate Close to Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Adjusted High-Low Spread with Close to Open Return
    combined_value = adjusted_high_low_spread + close_to_open_return
    
    # Final VHLCR Calculation
    vhclr = combined_value / df['close']
    
    return vhclr
```
