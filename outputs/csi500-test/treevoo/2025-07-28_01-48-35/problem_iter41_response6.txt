```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap_adjustment = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap_adjustment
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    final_combined_factor = np.maximum(final_combined_factor, 0)
    
    # Volume-Adjusted Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5_days) * close_to_intraday_range_ratio
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (intraday_range * (df['volume'] / avg_volume_5_days))
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Long-Term Price Momentum
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_momentum = df['close'] - ema_50
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + 
                       short_term_price_change + 
                       volume_weighted_high_low_spread)
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(df['volume'].diff(10) > 0, 3, -3)
    
    # Adaptive Volume Trend Coefficient
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10_days
    adaptive_coefficient = np.where(volume_change_rate > 1.5, 2.0, 
                                    np.where(volume_change_rate < -1, -1.5, 0.5))
    adaptive_volume_trend = volume_trend_indicator * adaptive_coefficient
    
    # Volume-Weighted Intraday Return
    scaled_volume = df['volume'] / df['volume'].rolling(window=15).mean()
    volume_weighted_intraday_return = (df['close'] - df['open']) * scaled_volume
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = intraday_range * scaled_volume
    
    # Final Combined Factor
    final_combined_factor = (final_combined_factor + 
                             volume_weighted_intraday_return + 
                             volume_adjusted_high_low_spread)
    
    # Volume-Weighted Price Momentum
    volume_weighted_price_momentum = (df['volume'] / avg_volume_5_days) * long_term_price_momentum
    
    # Adjusted Combined Signal
    enhanced_intraday_volatility = (intraday_range / df['open']) * df['volume']
    adjusted_combined_signal = (enhanced_intraday_volatility + 
                                short_term_price_change + 
                                volume_weighted_price_momentum)
    
    # Advanced Composite Factor
    advanced_composite_factor = (final_combined_factor * long_term_price_momentum + 
                                 adjusted_combined_signal)
    advanced_composite_factor = np.maximum(advanced_composite_factor, 0)
    
    # Smooth the Composite Factor
    smoothed_advanced_composite_factor = advanced_composite_factor.rolling(window=7).mean()
    
    # Final Alpha Factor with Enhanced Momentum
    final_alpha_factor = smoothed_advanced_composite_factor * adaptive_volume_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_alpha_factor - open_to_close_return
    
    return final_alpha_factor
```
