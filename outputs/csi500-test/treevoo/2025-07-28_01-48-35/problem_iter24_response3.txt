import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change (Close - Open)
    df['Daily_Price_Change'] = df['close'] - df['open']
    
    # Generate Factor: Sum of Daily Price Changes over a 10-day Window
    df['Sum_Daily_Price_Change_10D'] = df['Daily_Price_Change'].rolling(window=10).sum()
    
    # Generate Factor: Average of Daily Price Changes over a 20-day Window
    df['Avg_Daily_Price_Change_20D'] = df['Daily_Price_Change'].rolling(window=20).mean()
    
    # Evaluate Momentum by Considering High and Low Prices
    # Generate Factor: Difference between 10-day Moving Average of Highs and Lows
    df['MA_High_10D'] = df['high'].rolling(window=10).mean()
    df['MA_Low_10D'] = df['low'].rolling(window=10).mean()
    df['High_Low_Momentum'] = df['MA_High_10D'] - df['MA_Low_10D']
    
    # Generate Factor: Ratio of Current Day's High to 30-day Average High
    df['MA_High_30D'] = df['high'].rolling(window=30).mean()
    df['High_Ratio'] = df['high'] / df['MA_High_30D']
    
    # Analyze Daily Volume Change (Today's Volume - Yesterday's Volume)
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Generate Factor: Cumulative Volume Change over a 5-day Period
    df['Cum_Volume_Change_5D'] = df['Daily_Volume_Change'].rolling(window=5).sum()
    
    # Generate Factor: Count of Consecutive Days with Increasing Volume
    df['Volume_Increase'] = df['volume'] > df['volume'].shift(1)
    df['Consecutive_Volume_Increase'] = df['Volume_Increase'].groupby((df['Volume_Increase'] != df['Volume_Increase'].shift()).cumsum()).cumsum()
    
    # Generate Factor: Difference in Volume when Price is Up vs. Down over 20 days
    df['Price_Up'] = (df['close'] > df['close'].shift(1))
    df['Volume_Up'] = df['volume'][df['Price_Up']]
    df['Volume_Down'] = df['volume'][~df['Price_Up']]
    df['Vol_Diff_Up_Down'] = df['Volume_Up'].rolling(window=20).sum() - df['Volume_Down'].rolling(window=20).sum()
    
    # Generate Factor: Maximum Volume over the Past 10 Days Relative to 100-Day Average
    df['Max_Volume_10D'] = df['volume'].rolling(window=10).max()
    df['MA_Volume_100D'] = df['volume'].rolling(window=100).mean()
    df['Max_Vol_Relative_100D'] = df['Max_Volume_10D'] / df['MA_Volume_100D']
    
    # Generate Factor: Volume-Weighted Average Price (VWAP) over Last 10 Days
    df['VWAP_10D'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    df['Deviation_Close_VWAP_30D'] = df['close'] - df['VWAP_10D'].rolling(window=30).mean()
