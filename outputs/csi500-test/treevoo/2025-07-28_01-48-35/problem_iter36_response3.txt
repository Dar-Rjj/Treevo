```python
import pandas as pd

def heuristics(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10_days
    adj_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Compute 5-day Volume Change
    volume_change_5_days = df['volume'] - df['volume'].shift(5)
    
    # Compute Average Volume (Last 10 Days)
    avg_volume_10_days_vol_trend = df['volume'].rolling(window=10).mean()
    
    # Compute Volume Change Rate
    vol_change_rate = volume_change_5_days / avg_volume_10_days_vol_trend
    
    # Summarize Volume Trend
    vol_trend_score = vol_change_rate.apply(lambda x: 2.0 if x > 1 else -1.5 if x < 1 else 0)
    
    # Combined Momentum and Volume Factor
    sum_returns_n_days = df['close'].pct_change().rolling(window=20).sum()
    combined_momentum_volume = sum_returns_n_days * close_to_range_ratio * vol_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_volume - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = df['close'] - df['open']
    vol_weighted_close_to_open = close_to_open_return * scaled_volume
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    intermediate_composite = adj_intraday_range + (vol_weighted_close_to_open * price_momentum * vol_trend_score)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum_ema = df['close'] - ema_20
    
    # Volume Trend
    sum_volume_20_days = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / sum_volume_20_days
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    sma_7 = integrated_composite.rolling(window=7).mean()
    final_alpha_factor = sma_7
    
    # Additional Momentum Component
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    additional_momentum = df['close'] - ema_30
    
    # Enhance Combined Momentum, Volume, and Price Factor
    relative_high_low_spread = (df['high'] - df['low']) / (df['high'] - df['low']).rolling(window=10).mean().shift(1)
    enhanced_combined_factor = (combined_momentum_volume * volume_trend) / sum_volume_20_days * (volume_trend * relative_high_low_spread)
    
    # Final Alpha Factor
    final_alpha_factor = final_alpha_factor + enhanced_combined_factor - open_to_close_return
    
    return final_alpha_factor
```
