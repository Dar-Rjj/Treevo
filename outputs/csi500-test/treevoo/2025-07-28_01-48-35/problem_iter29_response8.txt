```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume
    dynamic_coefficient = (volume_change_rate > 2) * 1.7 + (volume_change_rate < -2) * -1.5
    dynamic_coefficient = dynamic_coefficient.fillna(0.8)
    volume_trend_score = dynamic_coefficient * volume_change_rate
    
    # Incorporate Price Action Strength
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    price_action_direction = (ma_5 > ma_20).astype(int) - (ma_5 < ma_20).astype(int)
    price_action_multiplier = price_action_direction * momentum_component
    
    # Combine Components
    final_factor = price_action_multiplier * volume_trend_score
    open_to_close_return = (df['open'] - df['close']) / df['close']
    final_factor += -open_to_close_return
    
    return final_factor
```
