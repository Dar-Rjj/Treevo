```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = df['close'] / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component (Sum Returns of Close prices for last N days)
    momentum_component = df['close'].pct_change().rolling(window=10).sum() * close_to_intraday_ratio
    
    # 5-day Volume Change
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # 10-day Average Volume
    average_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Volume Change Rate
    volume_change_rate = volume_change / average_volume_10
    
    # Apply Variable Coefficient to Volume Trend
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, np.where(volume_change_rate < -1, -1.5, 0))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Combine Initial Signals
    combined_signal = intraday_return.rolling(window=10).mean() + volume_adjusted_spread
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (momentum_component * volume_trend_score + (df['close'] - df['open'])) / df['volume'].rolling(window=20).mean() * price_trend
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = (df['open'] - df['close']) / df['close']
    overnight_gap = df['open'] - df['close'].shift(1)
    enhanced_factor -= open_to_close_return + overnight_gap
    
    # Introduce New Factor: Volume-Weighted Price Movement
    volume_weighted_price = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    volume_weighted_movement = (df['close'] - volume_weighted_price) / df['volume'].rolling(window=20).mean()
    
    # Final Enhancement with New Factor
    final_alpha_factor = enhanced_factor + volume_weighted_movement
    
    return final_alpha_factor
```
