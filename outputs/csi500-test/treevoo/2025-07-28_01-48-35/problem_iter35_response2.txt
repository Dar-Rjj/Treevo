import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['Close_to_Range_Ratio'] = df['close'] / df['Intraday_Range']
    
    # Short-Term Price Change
    df['Short_Term_Price_Change'] = df['close'].diff(5)
    
    # Adjust for Opening Gap
    df['Opening_Gap'] = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    df['High_Low_Open_Close'] = (df['high'] - df['low']) + df['Opening_Gap']
    
    # Measure Closing Behavior
    df['Closing_Behavior'] = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    df['Final_Combined_Factor'] = (df['High_Low_Open_Close'] * df['Closing_Behavior']).clip(lower=0)
    
    # Calculate Momentum Component
    df['Momentum'] = df['close'].pct_change().rolling(window=5).sum() * df['Close_to_Range_Ratio']
    
    # Adjust for Volume Activity
    df['Volume_Change_5D'] = df['volume'] - df['volume'].shift(5)
    df['Avg_Volume_10D'] = df['volume'].rolling(window=10).mean()
    df['Volume_Change_Rate'] = df['Volume_Change_5D'] / df['Avg_Volume_10D']
    df['Dynamic_Coefficient'] = df['Volume_Change_Rate'].apply(lambda x: 2.0 if x > 2 else (-1.5 if x < -1 else 0.8))
    
    # Integrate Price Trend
    df['EMA_20D'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend'] = df['close'] - df['EMA_20D']
    
    # Volume-Weighted Intraday Return
    df['VW_Intraday_Return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Volume-Weighted High-Low Spread
    df['VW_High_Low_Spread'] = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Composite Factor
    df['Composite_Factor'] = (df['Final_Combined_Factor'] * df['Momentum'] + df['VW_Intraday_Return']).clip(lower=0)
    
    # Smooth the Composite Factor
    df['Smoothed_Composite_Factor'] = df['Composite_Factor'].rolling(window=5).mean()
    
    # Volume Trend Indicator
    df['Volume_Trend_Indicator'] = (df['volume'].rolling(window=10).sum().diff(1) > 0).astype(int) * 2 - 2
    
    # Synthesize Final Alpha Factor
    df['Alpha_Factor'] = df['Smoothed_Composite_Factor'] * df['Volume_Trend_Indicator']
    
    # Opening Impact and Gap Adjustment
    df['Open_to_Close_Return'] = (df['close'] - df['open']) / df['open']
    df['Overnight_Gap'] = df['open'] - df['close'].shift(1)
    df['Alpha_Factor_Adjusted'] = df['Alpha_Factor'] - df['Open_to_Close_Return'] - df['Overnight_Gap']
    
    # Final Factor
    df['Scaled_Volume'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['Final_Alpha_Factor'] = df['Alpha_Factor_Adjusted'] * df['Scaled_Volume']
    
    return df['Final_Alpha_Factor']
