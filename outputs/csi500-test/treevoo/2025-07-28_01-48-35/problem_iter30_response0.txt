```python
import pandas as pd

def heuristics_v2(df):
    n = 10  # Lookback period for some of the calculations
    m = 3  # Recent average period
    
    # Sub-Thought 1.1: Momentum over a fixed period
    df['momentum'] = df['close'].diff(n)
    
    # Sub-Thought 1.2: Medium-term trend
    df['avg_close_n'] = df['close'].rolling(window=n).mean()
    df['trend'] = df['close'] - df['avg_close_n']
    
    # Sub-Thought 2.1: Overbought or oversold
    df['high_low_range'] = df['high'] - df['low']
    df['reversion'] = (df['close'] - df['low']) / df['high_low_range'] - 0.5
    
    # Sub-Thought 2.2: Difference between current close and recent average
    df['recent_avg'] = df['close'].rolling(window=m).mean()
    df['reversion_recent'] = (df['close'] - df['recent_avg']) / df['recent_avg']
    
    # Sub-Thought 3.1: Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['std_dev'] = df['daily_returns'].rolling(window=n).std()
    
    # Sub-Thought 3.2: Average True Range (ATR)
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], abs(row['high'] - row['close'].shift(1)), abs(row['low'] - row['close'].shift(1))), axis=1)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Sub-Thought 4.1: Relationship between price changes and volume
    df['volume_change'] = df['volume'].pct_change()
    df['price_change'] = df['close'].pct_change()
    df['vol_price_relation'] = df['volume_change'] * df['price_change']
    
    # Sub-Thought 4.2: Volume spike
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_spike'] = (df['volume'] > 2 * df['avg_volume']).astype(int)
    df['breakout'] = df['volume_spike'] * df['close'].diff()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['momentum'] + df['trend'] + df['reversion'] + df['reversion_recent'] + 
                          df['std_dev'] + df['atr'] + df['vol_price_relation'] + df['breakout'])
    
    return df['alpha_factor']
```
