```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    volume_spike = (volume_change > 0.2).astype(int)
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Combine Components
    combined_components = volume_weighted_return + price_trend + volume_spike
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_score = np.where(volume_change_rate > 1, 1.5, -1.2)
    
    # Intermediate Composite Factor
    intermediate_factor = combined_components + volume_weighted_return
    intermediate_factor *= volume_trend_score
    intermediate_factor += price_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    intermediate_alpha = intermediate_factor - open_to_close_return
    
    # Crossover Signals
    short_ma = df['close'].rolling(window=5).mean()
    long_ma = df['close'].rolling(window=20).mean()
    crossover_signal = (short_ma > long_ma).astype(int)
    
    # Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Volume Return
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_return = df['volume'] / avg_volume_20d
    
    # Open-Close Reversal
    next_day_open = df['open'].shift(-1)
    open_close_reversal = (next_day_open - df['close']).abs() / df['close']
    
    # Close-to-Open Gap
    close_to_open_gap = df['close'] - df['open'].shift(-1)
    
    # Average True Range (ATR)
    true_range = df[['high', 'low']].diff().abs().max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Trend Strength
    trend_strength = df['close'].rolling(window=20).apply(lambda x: np.polyfit(range(len(x)), x, 1)[0])
    
    # Final Alpha Factor
    smoothed_composite = intermediate_alpha.rolling(window=5).mean()
    
    # Calculate Momentum Component
    n_days = 10
    momentum_component = df['close'].pct_change().rolling(window=n_days).sum() * close_to_range_ratio
    
    # Calculate Volume-Adjusted High-Low Spread
    avg_volume_15d = df['volume'].rolling(window=15).mean()
    scaled_volume = df['volume'] / avg_volume_15d
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * scaled_volume
    
    # Final Combined Factor
    final_combined_factor = momentum_component * volume_trend_score + volume_weighted_return + volume_adjusted_high_low_spread
    
    return final_combined_factor
```
