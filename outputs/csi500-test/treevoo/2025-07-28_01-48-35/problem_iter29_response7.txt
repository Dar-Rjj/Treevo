import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).clip(lower=0)
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    final_combined_factor += volume_weighted_intraday_return
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Average Intraday Return
    avg_intraday_return = intraday_return.rolling(window=5).mean()
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=5).sum()
    final_combined_factor *= (close_to_range_ratio * momentum_component)
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=10, adjust=False).mean()
    
    # Composite Factor
    composite_factor = (volume_weighted_close_to_open_return * price_momentum) + final_combined_factor
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_indicator
    
    # Enhance Combined Momentum, Volume, and Price Factor
    avg_volume_22_days = df['volume'].rolling(window=22).mean()
    relative_high_low_spread = (df['high'] - df['low']) / df['low']
    enhanced_factor = (final_alpha_factor * volume_trend_indicator * (df['open'] - df['close'].shift(1))) / avg_volume_22_days * relative_high_low_spread
    
    # Combine Signals
    combined_signal = volume_weighted_intraday_return + short_term_price_change + spread
    
    # Final Factor
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10_days
    final_factor = combined_signal * scaled_volume * volume_trend_indicator
    
    return final_factor
