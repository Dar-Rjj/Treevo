```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_high_low_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_high_low_range + opening_gap
    
    # Measure Closing Behavior
    close_to_previous_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_high_low_open_close * close_to_previous_close_return).clip(lower=0)
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend Indicator
    volume_trend_indicator = (df['volume'].rolling(window=10).sum() > 0).apply(lambda x: 2 if x else -2)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Composite Factor
    composite_factor = (volume_weighted_close_to_open_return * price_momentum) + final_combined_factor
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_indicator
    
    # Combine Signals
    combined_signals = volume_weighted_intraday_return + short_term_price_change + high_low_spread
    
    # Scaled Volume
    scaled_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Final Factor
    final_factor = combined_signals * scaled_volume * volume_trend_indicator
    
    return final_factor
```
