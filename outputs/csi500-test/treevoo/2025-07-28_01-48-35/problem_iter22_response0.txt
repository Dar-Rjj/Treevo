```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    df['short_term_price_change'] = df['close'].diff(5)
    
    # Volume Trend Indicator
    df['volume_trend_indicator'] = df['volume'].diff(5).apply(lambda x: 1 if x > 0 else -1)
    
    # High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Long-Term Price Change
    df['long_term_price_change'] = df['close'].diff(20)

    # Combined Signal
    df['combined_signal'] = (df['volume_weighted_intraday_return'] + 
                             df['short_term_price_change'] + 
                             df['high_low_spread'] + 
                             df['long_term_price_change'])
    
    # Today's Volume / Average of Last 5 Days' Volume
    df['scaled_volume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Final Factor
    df['final_factor'] = (df['combined_signal'] * 
                           df['scaled_volume'] * 
                           df['volume_trend_indicator'] * 
                           df['long_term_price_change'])

    return df['final_factor']
```
