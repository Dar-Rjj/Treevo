import pandas as pd

def heuristics_v2(df):
    # Calculate +DM, -DM, and TR
    df['Hdiff'] = df['high'].diff()
    df['Ldiff'] = df['low'].shift().diff()
    df['+DM'] = df.apply(lambda x: x['Hdiff'] if (x['Hdiff'] > 0) and (x['Hdiff'] > x['Ldiff']) else 0, axis=1)
    df['-DM'] = df.apply(lambda x: x['Ldiff'] if (x['Ldiff'] > 0) and (x['Ldiff'] > x['Hdiff']) else 0, axis=1)
    df['TR'] = df[['high', 'close']].apply(lambda x: max(x['high'], x['close'].shift()) - min(x['low'], x['close'].shift()), axis=1)
    
    # Compute 14-day ADX
    df['+DMI14'] = df['+DM'].rolling(window=14).mean()
    df['-DMI14'] = df['-DM'].rolling(window=14).mean()
    df['TR14'] = df['TR'].rolling(window=14).mean()
    df['+DI14'] = (df['+DMI14'] / df['TR14']) * 100
    df['-DI14'] = (df['-DMI14'] / df['TR14']) * 100
    df['DX14'] = (abs(df['+DI14'] - df['-DI14']) / (df['+DI14'] + df['-DI14'])) * 100
    df['ADX14'] = df['DX14'].rolling(window=14).mean()

    # Calculate ROC over 12 days
    df['ROC12'] = df['close'].pct_change(periods=12) * 100

    # Bollinger Bands
    df['SMA20'] = df['close'].rolling(window=20).mean()
    df['STD20'] = df['close'].rolling(window=20).std()
    df['UpperBand'] = df['SMA20'] + (df['STD20'] * 2)
    df['LowerBand'] = df['SMA20'] - (df['STD20'] * 2)

    # On-Balance Volume (OBV)
    df['OBV'] = ((df['close'] - df['close'].shift()) / abs(df['close'] - df['close'].shift())) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low'] + df['close']) / 3
    df['CumulativeVolume'] = (df['volume'] * df['TypicalPrice']).cumsum()
    df['CumulativeVolumeSum'] = df['volume'].cumsum()
    df['VWAP'] = df['CumulativeVolume'] / df['CumulativeVolumeSum']

    # Pivot Points
    df['PivotPoint'] = (df['high'] + df['low'] + df['close']) / 3
    df['R1'] = (2 * df['PivotPoint']) - df['low']
    df['S1'] = (2 * df['PivotPoint']) - df['high']
    df['R2'] = df['PivotPoint'] + (df['high'] - df['low'])
    df['S2'] = df['PivotPoint'] - (df['high'] - df['low'])

    df['MF'] = df['TP'] * df['volume']
    df['PositiveMF'] = df['MF'].where(df['TP'] > df['TP'].shift(), 0)
