```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'].shift(-5) - df['close']
    
    # Calculate Volume Trend Indicator
    volume_trend = (df['volume'].rolling(window=5).sum().diff() > 0).astype(int) * 2 - 1
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_open_return = (df['close'] - df['open']) * df['volume']

    # Combine Signals
    combined_signal = intraday_return + short_term_price_change + high_low_spread

    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Indicators
    momentum_short_term = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    momentum_medium_term = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    momentum_long_term = (df['close'] - df['close'].shift(63)) / df['close'].shift(63)

    # Evaluate Volatility
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1)
    
    # Assess Relative Strength
    relative_strength = df['close'] / df['close'].shift(21)
    
    # Measure Money Flow
    money_flow_multiplier = (df['close'] - (df['high'] + df['low']) / 2) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    mfi = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Determine Trend Direction
    positive_dm = (df['high'] - df['high'].shift(1)).apply(lambda x: x if x > 0 and x > df['low'].shift(1) - df['low'] else 0)
    negative_dm = (df['low'].shift(1) - df['low']).apply(lambda x: x if x > 0 and x > df['high'] - df['high'].shift(1) else 0)
    di_positive = positive_dm.rolling(window=14).mean() / true_range.rolling(window=14).mean()
    di_negative = negative_dm.rolling(window=14).mean() / true_range.rolling(window=14).mean()
    directional_indicator = (di_positive - di_negative) / (di_positive + di_negative)
    
    # Combine Indicators
    composite_alpha_factor = (momentum_short_term + momentum_medium_term + momentum_long_term + relative_strength + mfi + directional_indicator) / 6
    
    # Evaluate Trading Volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=21).mean()
    
    # Integrate Price and Volume Analysis
    integrated_alpha_factor = composite_alpha_factor * volume_ratio
    final_alpha_factor = combined_signal * df['volume'] * volume_trend * integrated_alpha_factor

    return final_alpha_factor
```
