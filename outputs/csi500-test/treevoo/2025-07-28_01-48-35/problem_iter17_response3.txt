import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate 20-Day Exponential Moving Average (EMA) of Close Prices
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Advanced Price Momentum
    advanced_price_momentum = (df['close'] - ema_20) / (df['open'] - df['low'])
    
    # Combine with Close-to-Range Ratio
    combined_factor = close_to_range_ratio * advanced_price_momentum
    
    # Adjust for Volume Activity
    volume_20_sum = df['volume'].rolling(window=20).sum()
    volume_activity = df['volume'] / volume_20_sum
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(short_term_volume_change > 0, 1, -1)
    
    # Volume Trend Score
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_10_days
    volume_trend_score = np.where(volume_change_rate > 1, volume_change_rate * 1.5,
                                   np.where(volume_change_rate < 1, volume_change_rate * -1.2, 0))
    
    # Calculate Range Expansion
    true_range = df['high'] - df['low']
    atr_10 = true_range.rolling(window=10).mean()
    range_expansion = true_range / atr_10
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Composite Factor
    composite_factor = (combined_factor * volume_trend_score) / volume_20_sum
    composite_factor = composite_factor + range_expansion + obv
    smoothed_composite_factor = composite_factor.ewm(span=10, adjust=False).mean()
    
    # Opening Impact Adjustment
    open_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = smoothed_composite_factor - open_close_return
    
    return final_alpha_factor
