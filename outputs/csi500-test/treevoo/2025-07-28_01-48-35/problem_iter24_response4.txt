```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['vw_intraday_return'] = (df['High'] - df['Low']) / df['Low'] * df['Volume']

    # Short-Term Price Change
    df['short_term_price_change'] = df['Close'].shift(-5) - df['Close']

    # High-Low Spread
    df['high_low_spread'] = df['High'] - df['Low']

    # Daily Return
    df['daily_return'] = (df['Close'] - df['Open']) / df['Open']

    # Momentum Indicators
    df['short_term_momentum'] = (df['Close'] - df['Close'].shift(7)) / df['Close'].shift(7)
    df['medium_term_momentum'] = (df['Close'] - df['Close'].shift(21)) / df['Close'].shift(21)
    df['long_term_momentum'] = (df['Close'] - df['Close'].shift(63)) / df['Close'].shift(63)

    # True Range
    df['true_range'] = df[['High' - df['Low'], 
                            abs(df['High'] - df['Close'].shift(1)), 
                            abs(df['Low'] - df['Close'].shift(1))]].max(axis=1)

    # Relative Strength
    df['relative_strength'] = df['Close'] / df['Close'].shift(21)

    # Money Flow
    df['mf_multiplier'] = (df['Close'] - (df['High'] + df['Low']) / 2) / (df['High'] - df['Low'])
    df['mf_volume'] = df['mf_multiplier'] * df['Volume']
    df['mfi_14'] = df['mf_volume'].rolling(window=14).sum() / df['Volume'].rolling(window=14).sum()

    # Trend Direction
    df['+dm'] = df.apply(lambda x: x['High'] - df['High'].shift(1) if x['High'] > df['High'].shift(1) and (x['High'] - df['High'].shift(1)) > (df['Low'].shift(1) - x['Low']) else 0, axis=1)
    df['-dm'] = df.apply(lambda x: df['Low'].shift(1) - x['Low'] if df['Low'].shift(1) > x['Low'] and (df['Low'].shift(1) - x['Low']) > (x['High'] - df['High'].shift(1)) else 0, axis=1)

    # SMA of DMs
    df['+di'] = df['+dm'].rolling(window=14).mean() / df['true_range'].rolling(window=14).mean()
    df['-di'] = df['-dm'].rolling(window=14).mean() / df['true_range'].rolling(window=14).mean()

    # Volume Ratio
    df['volume_ratio'] = df['Volume'] / df['Volume'].rolling(window=21).mean()

    # Weighted Average of Momentum Indicators
    df['weighted_momentum'] = (df['short_term_momentum'] * 0.3 + df['medium_term_momentum'] * 0.5 + df['long_term_momentum'] * 0.2)

    # Price-Volatility Ratio
    df['price_volatility_ratio'] = (df['Close'] - df['Close'].shift(21)) / df['true_range']

    # Final Alpha Factor
    df['alpha_factor'] = (df['short_term_price_change'] + df['high_low_spread'] + df['daily_return'] + 
                          df['weighted_momentum'] + df['relative_strength'] + df['mfi_14'] + 
                          df[['+di', '-di']].max(axis=1) + df['price_volatility_ratio']) / 8 * df['volume_ratio']

    return df['alpha_factor']
```
