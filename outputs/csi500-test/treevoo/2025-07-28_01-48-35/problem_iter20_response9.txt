```python
import pandas as pd

def heuristics_v2(df, N=5, lookback_vol=10, lookback_avg=15):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change(N).sum() * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume = df['volume'].rolling(window=lookback_vol).mean()
    volume_change_rate = volume_change / avg_volume
    volume_trend_score = 1.5 if volume_change_rate > 1 else -1.2 if volume_change_rate < 1 else 0
    
    # Calculate Volume-Weighted Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(lookback_avg)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Adjusted High-Low Spread
    avg_volume_15 = df['volume'].rolling(window=lookback_avg).mean()
    scaled_volume = df['volume'] / avg_volume_15
    volume_adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Final Combined Factor
    final_factor = (momentum_component * volume_trend_score 
                    + volume_weighted_intraday_return 
                    + volume_adjusted_high_low_spread)
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_factor - open_to_close_return
    
    return final_alpha_factor
```
