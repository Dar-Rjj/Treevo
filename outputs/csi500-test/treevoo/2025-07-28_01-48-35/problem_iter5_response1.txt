```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Scale Volume by Last 5 Days' Average
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5d
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum (10-Day EMA)
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10d
    
    # Calculate Volume Trend
    sum_volume_10d = df['volume'].rolling(window=10).sum()
    volume_trend = df['volume'] / sum_volume_10d
    
    # Intermediate Composite Factor
    intermediate_composite_factor = (
        volume_adjusted_high_low_spread + 
        volume_weighted_close_to_open_return * price_momentum
    ) * volume_trend
    
    # Smoothed Composite Factor (5-Day SMA)
    smoothed_composite_factor = intermediate_composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
