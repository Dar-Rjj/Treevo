import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative range as (high - low) / (high + low)
    df['relative_range'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate the amount/volume ratio
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Combine the above factors into a single alpha factor
    df['alpha_factor'] = df['close_change'] * df['relative_range'] * df['amount_volume_ratio']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
