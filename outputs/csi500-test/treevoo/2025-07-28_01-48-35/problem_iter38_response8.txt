```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']

    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range

    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)

    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap

    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).apply(lambda x: max(x, 0))

    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']

    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)

    # Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']

    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20

    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + short_term_price_change + volume_weighted_close_to_open_return) / 3

    # Volume-Adjusted Momentum
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5) * close_to_range_ratio

    # Cumulative Volume-Adjusted Momentum
    cumulative_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()

    # Volume-Weighted High-Low Spread
    volume_factor = df['volume'] / avg_volume_5
    volume_weighted_high_low_spread = (df['high'] - df['low']) * volume_factor

    # Volume Trend Indicator
    volume_change_10 = df['volume'] - df['volume'].shift(10)
    volume_trend_indicator = volume_change_10.apply(lambda x: 2 if x > 0 else -2)

    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum + combined_signal).apply(lambda x: max(x, 0))
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()

    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_indicator

    # Calculate Daily Market Sentiment
    daily_market_sentiment = (df['close'] - df['open']).rolling(window=5).sum().apply(lambda x: x - x.rolling(window=10).mean())

    # Calculate Volatility Ratio
    volatility_ratio = df['volume'] / (df['close'] - df['open']).abs()
    volatility_ratio_diff = volatility_ratio - volatility_ratio.rolling(window=20).mean()

    # Calculate True Range
    true_range = (df['high'] - df['low']).rolling(window=30).mean()
    true_range_adjustment = (true_range - true_range.rolling(window=30).mean()) * np.sign(df['close'] - df['open'])

    # Final Enhanced Factor
    final_enhanced_factor = final_alpha_factor * daily_market_sentiment + volatility_ratio_diff + true_range_adjustment

    # Final Alpha Factor
    scaled_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    final_alpha_factor = final_enhanced_factor * scaled_volume

    return final_alpha_factor
```
