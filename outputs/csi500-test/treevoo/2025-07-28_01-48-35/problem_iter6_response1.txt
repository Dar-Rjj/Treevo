```python
import pandas as pd
import numpy as np

def heuristics_v2(df, trend_window=20, decay_factor=0.95, rolling_window=10):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=trend_window, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Adjust Momentum with Price Trend
    adjusted_momentum = volume_weighted_momentum + price_trend
    
    # Weighted Cumulative Sum
    weighted_cumsum = adjusted_momentum.rolling(window=rolling_window).apply(
        lambda x: np.sum(x * (decay_factor ** np.arange(len(x))[::-1])),
        raw=True
    )
    
    return weighted_cumsum
```
