```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    momentum_component = (df['close'].pct_change().rolling(window=N).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 if x > 1 else (-1.2 if x < 1 else 0))
    
    # Final Combined Factor
    combined_factor = momentum_component * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = combined_factor - open_to_close_return
    
    return final_alpha_factor
```
