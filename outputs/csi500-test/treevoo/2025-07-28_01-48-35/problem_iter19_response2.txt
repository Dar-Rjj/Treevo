```python
import pandas as pd

def heuristics_v2(df):
    # Calculate +DM, -DM, and TR
    df['H_minus_H1'] = df['high'].diff()
    df['L1_minus_L'] = -df['low'].shift(1) + df['low']
    
    df['+DM'] = df[['H_minus_H1', 0]].max(axis=1)
    df['-DM'] = df[['L1_minus_L', 0]].max(axis=1)
    
    df['+DM'] = df['+DM'].where(df['+DM'] > df['-DM'], 0)
    df['-DM'] = df['-DM'].where(df['-DM'] > df['+DM'], 0)
    
    df['TR'] = df[['high' - df['low'], 
                   (df['high'] - df['close'].shift(1)).abs(), 
                   (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    
    # Compute 14-day ADX
    n = 14
    df['+DM_14'] = df['+DM'].rolling(window=n).sum()
    df['-DM_14'] = df['-DM'].rolling(window=n).sum()
    df['TR_14'] = df['TR'].rolling(window=n).sum()
    
    df['+DI'] = 100 * (df['+DM_14'] / df['TR_14'])
    df['-DI'] = 100 * (df['-DM_14'] / df['TR_14'])
    
    df['DX'] = 100 * (df['+DI'] - df['-DI']).abs() / (df['+DI'] + df['-DI'])
    df['ADX'] = df['DX'].rolling(window=n).mean()
    
    # Calculate Rate of Change (ROC) over 12 days
    df['ROC_12'] = df['close'].pct_change(periods=12) * 100
    
    # Calculate Bollinger Bands
    sma = df['close'].rolling(window=20).mean()
    std = df['close'].rolling(window=20).std()
    df['BB_upper'] = sma + 2 * std
    df['BB_lower'] = sma - 2 * std
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['volume'] * (df['close'] > df['close'].shift(1)) - 
                 df['volume'] * (df['close'] < df['close'].shift(1))).cumsum()
    
    # Calculate Pivot Points
    df['PP'] = (df['high'] + df['low'] + df['close']) / 3
    df['R1'] = 2 * df['PP'] - df['low']
    df['S1'] = 2 * df['PP'] - df['high']
    
    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    n = 14
    positive_money_flow_14 = positive_money_flow.rolling(window=n).sum()
    negative_money_flow_14 = negative_money_flow.rolling(window=n).sum()
    
    df['MFI'] = 100 - 100 / (1 + (positive_money_flow_14 / negative_money_flow_14))
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['ADX'] + df['ROC_12'] + (df['close'] - df['BB_lower']) / (df['BB_upper'] - df['BB_lower']) + df['OBV'] + (df['close'] - df['S1']) / (df['R1'] - df['S1']) + df['MFI']) / 5
    
    return df['alpha_factor']
```
