```python
def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate the 10-day and 30-day exponential moving average of the close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()

    # Calculate the EMA crossover signal
    df['ema_crossover'] = df['ema_10'] - df['ema_30']

    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv -= (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = obv.cumsum()

    # Calculate the OBV rate of change over 10 days
    df['obv_roc_10'] = df['obv'].pct_change(periods=10)

    # Calculate the alpha factor as a combination of the EMA crossover and OBV ROC
    df['alpha_factor'] = df['ema_crossover'] + df['obv_roc_10']

    return df['alpha_factor']
```
