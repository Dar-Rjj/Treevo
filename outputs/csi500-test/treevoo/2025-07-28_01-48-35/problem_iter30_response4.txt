import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=10).sum()
    combined_momentum_close_to_range = momentum_component * close_to_intraday_range_ratio
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(7)
    average_volume_15_days = df['volume'].rolling(window=15).mean()
    volume_change_rate = short_term_volume_change / average_volume_15_days
    volume_trend_indicator = volume_change_rate.apply(lambda x: 2.5 if x > 1 else (-2.0 if x < -1 else 0.5))
    
    # Integrate Price Trend
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    price_trend = df['close'] - ema_12
    
    # Calculate Relative High-Low Spread
    high_low_spread = df['high'] - df['low']
    average_high_low_spread_15_days = high_low_spread.rolling(window=15).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Opening Impact Factor
    open_to_close_return = df['close'] - df['open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    combined_factor = combined_momentum_close_to_range * volume_trend_indicator
    combined_factor /= average_volume_15_days.rolling(window=25).mean()
    combined_factor *= volume_trend_indicator * relative_high_low_spread
    
    # Opening Impact and Gap Adjustment
    overnight_gap = df['open'] - df['close'].shift(1)
    final_factor = combined_factor - open_to_close_return - overnight_gap
    
    return final_factor
