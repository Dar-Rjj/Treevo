```python
def heuristics_v2(df):
    # Calculate the price change from the previous day's close to the current day's open
    price_change = df['open'] - df['close'].shift(1)
    
    # Calculate the buying pressure as the difference between amount and volume
    buying_pressure = df['amount'] - df['volume']
    
    # Calculate the relative range as (high - low) / close
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_change * buying_pressure) / relative_range
    
    return alpha_factor
```
