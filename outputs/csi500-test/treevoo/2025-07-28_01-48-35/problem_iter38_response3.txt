```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range (High - Low)
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio (Close / Intraday Range)
    df['close_to_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Momentum Component (Sum of Returns of Close prices last N days)
    df['returns'] = df['close'].pct_change()
    df['momentum'] = df['returns'].rolling(window=N).sum()
    df['momentum_component'] = df['momentum'] * df['close_to_range_ratio']
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume_change'] / df['avg_volume_10d']
    
    def adaptive_coefficient(vol_change_rate):
        if vol_change_rate > 1.5:
            return 2.0
        elif vol_change_rate < -1:
            return -1.5
        else:
            return 0.5
    
    df['volume_trend_score'] = df['volume_change_rate'].apply(adaptive_coefficient)
    
    # Calculate Volume-Weighted Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    df['scaled_volume'] = df['volume'] / df['volume'].rolling(window=15).mean()
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['scaled_volume']
    
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low_spread'] = df['intraday_range'] * df['scaled_volume']
    
    # Final Combined Factor
    df['final_factor'] = (df['momentum_component'] * df['volume_trend_score'] 
                          + df['volume_weighted_intraday_return'] 
                          + df['volume_adjusted_high_low_spread'])
    
    # Opening Impact Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['final_alpha_factor'] = df['final_factor'] - df['open_to_close_return']
    
    return df['final_alpha_factor']
```
