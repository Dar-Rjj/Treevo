```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Spread
    spread = df['high'] - df['low']

    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']

    # Calculate Average Intraday Return over N days
    average_intraday_return = intraday_return.rolling(window=N).mean()

    # Combine Signals
    alpha_factor = average_intraday_return + volume_adjusted_spread

    return alpha_factor
```
