```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Scale Volume by Last 5 Days' Average
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5d
    
    # Volume-Adjusted High-Low Spread
    vol_adj_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    vol_weighted_close_open_return = close_open_return * scaled_volume
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Volume Trend
    volume_trend = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intermediate Composite Factor
    intermediate_factor = vol_adj_high_low_spread + vol_weighted_close_open_return
    intermediate_factor *= price_momentum
    intermediate_factor *= volume_trend
    
    # Smoothed Composite Factor
    smoothed_factor = intermediate_factor.rolling(window=5).mean()
    
    return smoothed_factor
```
