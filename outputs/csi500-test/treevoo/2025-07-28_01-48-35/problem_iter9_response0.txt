```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return
    avg_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    initial_signals = avg_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume
    volume_trend_score = volume_change_rate.where(volume_change_rate > 1, 0) * 1.5 + \
                         volume_change_rate.where(volume_change_rate < 1, 0) * -1.2
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Price Volatility Factor
    daily_volatility = df['close'].rolling(window=N).std()
    avg_daily_volatility = daily_volatility.rolling(window=10).mean()
    relative_volatility = daily_volatility / daily_volatility.shift(1)
    
    # Synthesize Final Alpha Factor
    combined_momentum = volume_trend_score + relative_high_low_spread
    final_alpha_factor = initial_signals * combined_momentum * relative_volatility
    
    return final_alpha_factor
```
