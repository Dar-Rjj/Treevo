```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    initial_signal = average_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    avg_volume_last_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_last_10_days
    volume_trend_score = volume_change_rate.apply(lambda x: x * 1.5 if x > 1 else x * -1.2)
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_last_10_days = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread_last_10_days
    
    # Synthesize Final Alpha Factor
    alpha_factor = initial_signal * (volume_trend_score * normalized_high_low_spread)
    
    return alpha_factor
```
