```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    high_low_open_close = (df['high'] - df['low']) + opening_gap
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = high_low_open_close * closing_behavior
    final_combined_factor = final_combined_factor.apply(lambda x: max(x, 0))  # Ensure Positive Values
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=5).sum() * close_to_intraday_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10_days
    dynamic_coefficient = volume_change_rate.apply(lambda x: 2.0 if x > 2 else (-1.5 if x < -1 else 0.8))
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Volume-Weighted Intraday Return
    scaled_volume = df['volume'] / df['volume'].rolling(window=15).mean()
    volume_weighted_intraday_return = (df['close'] - df['open']) * scaled_volume
    
    # Volume-Adjusted High-Low Spread
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / average_volume_5_days)
    
    # Composite Factor
    composite_factor = (final_combined_factor * momentum_component + 
                        volume_weighted_intraday_return + 
                        volume_adjusted_high_low_spread)
    
    # Smooth the Composite Factor
    alpha_factor = composite_factor.rolling(window=5).mean()
    
    return alpha_factor
```
