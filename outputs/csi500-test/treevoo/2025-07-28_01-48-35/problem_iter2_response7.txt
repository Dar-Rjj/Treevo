import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = df['close'] / (intraday_range + 1e-8)
    
    # Calculate Volume Weighted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_scaled = df['volume'] / (avg_volume_5d + 1e-8)
    volume_weighted_momentum = close_to_intraday_ratio * volume_scaled
    
    # Cumulative Sum of Volume Weighted Momentum over a rolling window of past days
    cum_sum_volume_weighted_momentum = volume_weighted_momentum.rolling(window=20).sum()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Volume Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * volume_scaled
    
    # Combine Factors
    combined_factor = cum_sum_volume_weighted_momentum + volume_adjusted_high_low_spread
    
    return combined_factor
