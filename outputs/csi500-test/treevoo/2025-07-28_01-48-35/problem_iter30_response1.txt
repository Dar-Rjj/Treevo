```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_vol_10d
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Adjust for Volume Activity
    vol_change_5d = df['volume'] - df['volume'].shift(5)
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    vol_change_rate = vol_change_5d / avg_vol_10d
    volume_trend = vol_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < 1 else 0))
    
    # Combined Momentum and Volume Factor
    returns_n_days = df['close'].pct_change().rolling(window=5).sum()
    combined_momentum_vol_factor = returns_n_days * close_to_range_ratio * volume_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    intermediate_factor = adjusted_intraday_range + (open_to_close_return * scaled_volume)
    
    # Price Momentum
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20d
    
    # Volume Trend
    sum_vol_20d = df['volume'].rolling(window=20).sum()
    volume_trend_indicator = df['volume'] / sum_vol_20d
    
    # Price Oscillator
    ema_7d = df['close'].ewm(span=7, adjust=False).mean()
    ema_14d = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7d - ema_14d
    
    # Integrated Composite Factor
    integrated_composite = intermediate_factor + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite = integrated_composite.rolling(window=7).mean()
    
    # Additional Momentum Component
    ema_30d = df['close'].ewm(span=30, adjust=False).mean()
    additional_momentum = df['close'] - ema_30d
    
    # Enhance Combined Momentum, Volume, and Price Factor
    relative_high_low_spread = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    average_high_low_10d = (df['high'] - df['low']).rolling(window=10).mean()
    enhanced_factor = (combined_momentum_vol_factor * volume_trend_indicator) / (avg_vol_10d) * (volume_trend_indicator * relative_high_low_spread)
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_factor * smoothed_composite + open_to_close_return
    
    return final_alpha_factor
```
