```python
import pandas as pd

def heuristics_v2(df, n_days=10, n_avg_high_low=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    prev_close = df['close'].shift(1)
    opening_gap = df['open'] - prev_close
    
    # Combine High-Low and Open-Close
    combined_hl_gap = high_low_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = (df['close'] / prev_close).abs()
    final_combined_factor = combined_hl_gap * close_to_prev_close_return
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return
    avg_intraday_return = intraday_return.rolling(window=n_days).mean()
    
    # Combine Initial Signals
    initial_signals = avg_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume
    volume_trend_score = (volume_change_rate > 1.5) * 2.0 + (volume_change_rate < 0.8) * (-1.5)
    
    # Calculate Price Momentum
    high_low_spread = high_low_range
    avg_high_low_spread = high_low_spread.rolling(window=n_avg_high_low).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (final_combined_factor * normalized_high_low_spread * volume_trend_score) + initial_signals
    return final_alpha_factor
```
