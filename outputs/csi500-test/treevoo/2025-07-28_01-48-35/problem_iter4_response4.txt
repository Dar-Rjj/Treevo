```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    daily_returns = df['close'].pct_change()
    momentum_component = daily_returns.rolling(window=N).sum().shift(1)  # Shift to prevent lookahead bias
    
    # Combine with Close-to-Range Ratio
    combined_value = close_to_range_ratio * momentum_component
    
    # Subtract Opening Impact
    open_to_close_return = (df['close'] - df['open']) / df['open']
    factor_values = combined_value - open_to_close_return

    return factor_values
```
