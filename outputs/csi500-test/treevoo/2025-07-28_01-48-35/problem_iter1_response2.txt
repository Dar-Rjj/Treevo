import pandas as pd

def heuristics_v2(df):
    # Compute 10-Day Exponential Moving Average (EMA) of Close Prices
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract EMA from Current Close Price to get Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Compute 10-Day Sum of Volume
    df['Volume_10_Sum'] = df['volume'].rolling(window=10).sum()
    
    # Divide Current Volume by 10-Day Sum of Volume to get Volume Trend
    df['Volume_Trend'] = df['volume'] / df['Volume_10_Sum']
    
    # Multiply Price Momentum by Volume Trend to get the Composite Factor
    df['Composite_Factor'] = df['Price_Momentum'] * df['Volume_Trend']
    
    # Smooth the Composite Factor using a 5-Day Simple Moving Average (SMA)
    df['Alpha_Factor'] = df['Composite_Factor'].rolling(window=5).mean()
    
    # Return the Alpha Factor as a pandas Series indexed by (date)
    return df['Alpha_Factor']
