```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Change
    price_change = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = (volume_change > 0).astype(int) * 2 - 1  # +1 if positive, -1 if negative
    
    average_volume_20_days = df['volume'].rolling(window=20).mean().shift(1)
    final_vapm_factor = (price_change * volume_trend_indicator) / average_volume_20_days
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread_10_days = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / average_high_low_spread_10_days
    
    # Combine Volume and Price Momentum
    final_alpha_factor = final_vapm_factor * normalized_high_low_spread
    
    return final_alpha_factor
```
