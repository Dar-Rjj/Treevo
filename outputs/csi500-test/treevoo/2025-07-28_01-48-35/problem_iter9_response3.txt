```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['open']

    # Calculate Volume-Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * (df['volume'] ** 0.5)

    # Calculate Short-Term Price Change
    short_term_price_change = df['close'].diff(10)

    # Adjust for Volume Activity
    avg_volume_30_days = df['volume'].rolling(window=30).mean().shift(1)
    short_term_volume_change = df['volume'] - df['volume'].shift(10)
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Combine Intraday Signals with Price Momentum and High-Low Spread
    combined_signal = (volume_adjusted_intraday_return + 
                       short_term_price_change + 
                       high_low_spread)

    # Volume Adjusted Factor
    scaled_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    volume_adjusted_factor = combined_signal * scaled_volume

    # Final Factor Adjustment
    final_factor = volume_adjusted_factor * volume_trend_indicator

    # Final Alpha Factor
    alpha_factor = final_factor / avg_volume_30_days

    return alpha_factor
```
