```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Scale Volume by Last 10 Days' Average
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10
    
    # Adjusted High-Low Spread with Volume Trend
    adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    
    # Calculate Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Calculate Volume Trend
    volume_sum_20 = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / volume_sum_20
    
    # Intermediate Composite Factor
    intermediate_factor = (adjusted_high_low_spread + 
                           volume_weighted_close_to_open * price_momentum * volume_trend)
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Integrated Composite Factor
    integrated_factor = intermediate_factor + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_factor = integrated_factor.rolling(window=5).mean()
    
    return smoothed_factor
```
