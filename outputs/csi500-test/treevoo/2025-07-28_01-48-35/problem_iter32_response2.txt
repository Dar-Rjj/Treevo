```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor (Ensure Positive Values)
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).clip(lower=0)
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + short_term_price_change + volume_weighted_close_to_open)
    
    # Volume-Adjusted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5d) * close_to_range_ratio
    
    # Cumulative Volume-Adjusted Momentum (Sum over a rolling window of past 5 days)
    cumulative_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    volume_factor = df['volume'] / avg_volume_5d
    volume_weighted_high_low_spread = (df['high'] - df['low']) * volume_factor
    
    # Volume Trend Indicator
    volume_trend_score = (df['volume'].rolling(window=10).mean() > df['volume'].shift(10)).map({True: 2, False: -2})
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum + combined_signal).clip(lower=0)
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    # Final Factor
    scaled_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    final_alpha_factor *= scaled_volume
    
    return final_alpha_factor
```
