```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) * df['volume'] / df['low']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    final_combined_factor = final_combined_factor.where(final_combined_factor > 0, 0)
    
    # Volume-Adjusted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5d) * close_to_intraday_range_ratio
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (intraday_range * (df['volume'] / avg_volume_5d))
    
    # Price Momentum
    ema_20d = df['close'].ewm(span=20).mean()
    price_momentum = df['close'] - ema_20d
    
    # Combined Signal
    combined_signal = volume_weighted_intraday_return + short_term_price_change + volume_weighted_high_low_spread
    
    # Volume Trend Indicator
    volume_trend = (df['volume'].diff(10) > 0).astype(int) * 6 - 3
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum) + combined_signal
    composite_factor = composite_factor.where(composite_factor > 0, 0)
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=7).mean()
    
    # Adaptive Volume Trend Coefficient
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10d
    adaptive_coefficient = pd.Series(0, index=df.index)
    adaptive_coefficient[volume_change_rate > 1.5] = 2.0
    adaptive_coefficient[volume_change_rate < -1] = -1.5
    adaptive_coefficient[(volume_change_rate >= -1) & (volume_change_rate <= 1.5)] = 0.5
    adaptive_volume_trend = volume_trend * adaptive_coefficient
    
    # Volume-Weighted Intraday Return with Scaled Volume
    avg_volume_15d = df['volume'].rolling(window=15).mean()
    scaled_volume = df['volume'] / avg_volume_15d
    volume_weighted_intraday_return_scaled = (df['close'] - df['open']) * scaled_volume
    
    # Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = intraday_range * scaled_volume
    
    # Final Combined Factor with Volume Trend
    final_combined_factor_with_volume_trend = (price_momentum * adaptive_volume_trend) + volume_weighted_intraday_return_scaled + volume_adjusted_high_low_spread
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_combined_factor_with_volume_trend - open_to_close_return
    
    return final_alpha_factor
```
