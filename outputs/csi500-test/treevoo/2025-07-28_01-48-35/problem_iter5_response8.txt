```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    N = 10
    momentum_component = df['close'].pct_change().rolling(window=N).sum()
    
    # Combine with Close-to-Range Ratio
    combined_momentum = momentum_component * close_to_range_ratio
    
    # Adjust for Volume Activity
    avg_volume_20 = df['volume'].rolling(window=20).mean().shift(1)
    short_term_vol_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = short_term_vol_change.apply(lambda x: 1 if x > 0 else -1)
    
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_10
    volume_trend_score = (volume_change_rate > 1) * 1.5 + (volume_change_rate < 1) * -1.2
    
    # Final Combined Momentum and Volume Factor
    final_factor = (combined_momentum * volume_trend_score) / avg_volume_20
    
    # Opening Impact Adjustment
    open_to_close_return = (df['open'] - df['close']) / df['open']
    final_alpha_factor = final_factor - open_to_close_return
    
    return final_alpha_factor
```
