```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    scaled_vol = df['volume'] / avg_vol_10d
    adjusted_intraday_range = intraday_range * scaled_vol
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Activity
    vol_change_5d = df['volume'] - df['volume'].shift(5)
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    vol_change_rate = vol_change_5d / avg_vol_10d
    vol_trend_score = np.where(vol_change_rate > 1, 1.5, -1.2)
    
    # Combined Momentum and Volume Factor
    ret_close = df['close'].pct_change()
    sum_ret_close = ret_close.rolling(window=5).sum()
    combined_momentum_volume = sum_ret_close * close_to_range_ratio * vol_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_volume - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open = close_to_open_return * scaled_vol
    price_momentum = df['close'].ewm(span=20, adjust=False).mean()
    intermediate_composite = adjusted_intraday_range + volume_weighted_close_to_open * (df['close'] - price_momentum) * vol_trend_score
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume Trend
    vol_trend = df['volume'] / df['volume'].rolling(window=20).sum()
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite = integrated_composite.rolling(window=5).mean()
    
    return smoothed_composite
```
