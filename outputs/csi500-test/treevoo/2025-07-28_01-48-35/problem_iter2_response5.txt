```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = returns.rolling(window=N).sum()
    combined_momentum = close_to_range_ratio * momentum_component
    
    # Subtract Opening Impact
    open_to_close_return = (df['close'] - df['open']) / df['open']
    combined_momentum -= open_to_close_return
    
    # Calculate Volume Trend Score
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 * x if x > 1 else -1.2 * x)
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread
    
    # Combine All Components
    alpha_factor = (combined_momentum * normalized_high_low_spread * volume_trend_score)
    
    return alpha_factor
```
