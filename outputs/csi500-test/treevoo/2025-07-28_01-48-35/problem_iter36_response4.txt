import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    df['volume_adjusted_spread'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    df['momentum_component'] = df['close'].pct_change().rolling(window=5).sum()
    df['momentum_adjusted_close_to_range'] = df['close_to_range_ratio'] * df['momentum_component']
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume_change'] / df['avg_volume_10d']
    df['volume_trend_indicator'] = df['volume_change_rate'].apply(lambda x: 2.5 if x > 1 else (-2.0 if x < -1 else 0.5))
    
    # Integrate Price Trend
    df['ema_close_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_trend'] = df['close'] - df['ema_close_10d']
    
    # Calculate Relative High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['avg_high_low_spread_10d'] = df['high_low_spread'].rolling(window=10).mean()
    df['relative_high_low_spread'] = df['high_low_spread'] / df['avg_high_low_spread_10d'].shift(1)
    
    # Enhance Combined Momentum, Volume, and Price Factor
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['combined_momentum_volume_price'] = (df['momentum_adjusted_close_to_range'] * df['volume_trend_indicator']) / df['avg_volume_20d'] * df['relative_high_low_spread']
    
    # Opening Impact and Gap Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['overnight_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['opening_impact_factor'] = df['combined_momentum_volume_price'] - df['open_to_close_return'] - df['overnight_gap']
    
    return df['opening_impact_factor'].dropna()
