```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=5).sum()
    momentum_component = close_to_range_ratio * momentum_component
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = np.where(short_term_volume_change > 0, 1, -1)
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=5).mean()
    
    # Combine Initial Signals
    combined_signal = average_intraday_return + volume_adjusted_spread
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    ema_10_day = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10_day
    
    # Introduce Opening Impact Factor
    opening_impact_factor = df['close'] - df['open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    combined_momentum_volume_price = (combined_signal * close_to_range_ratio * volume_trend_indicator * opening_impact_factor) / df['volume'].rolling(window=20).mean()
    combined_momentum_volume_price *= (volume_trend_indicator * high_low_spread)
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend Score
    ten_day_volume_change = df['volume'] - df['volume'].shift(10)
    volume_trend_score = np.where(ten_day_volume_change > 0, 2, -2)
    
    # Composite Factor
    composite_factor = (volume_weighted_intraday_return * price_momentum) + combined_momentum_volume_price
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    # Combine Signals
    combined_signals = volume_weighted_intraday_return + short_term_price_change + high_low_spread
    
    # Final Factor
    scaled_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    final_factor = combined_signals * scaled_volume * volume_trend_indicator
    
    return final_factor
```
