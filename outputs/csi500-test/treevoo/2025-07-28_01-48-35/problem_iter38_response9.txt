```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / intraday_range
    
    # Short-Term Price Change
    short_term_price_change = df['close'].diff(5)
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    high_low_open_close = (df['high'] - df['low']) + opening_gap
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = (high_low_open_close * closing_behavior).clip(lower=0)
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Volume-Weighted High-Low Spread
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5d)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Volume Trend Indicator
    volume_trend = np.where(df['volume'].pct_change(10) > 0, 2, -2)
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum) + volume_weighted_intraday_return
    composite_factor = composite_factor.clip(lower=0)
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend
    
    # Calculate Daily Market Sentiment
    daily_market_sentiment = (df['close'] - df['open']).rolling(window=5).sum()
    daily_market_sentiment_sma = daily_market_sentiment.rolling(window=10).mean()
    daily_market_sentiment_adjusted = daily_market_sentiment - daily_market_sentiment_sma
    
    # Calculate Volatility Ratio
    volatility_ratio = df['volume'] / abs(df['close'] - df['open'])
    volatility_ratio_mean_20d = volatility_ratio.rolling(window=20).mean()
    volatility_ratio_diff = volatility_ratio - volatility_ratio_mean_20d
    
    # Calculate True Range
    true_range = df[['high', 'low', 'close']].join(df['close'].shift(1)).apply(
        lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1
    )
    true_range_mean_30d = true_range.rolling(window=30).mean()
    adjusted_true_range = (true_range - true_range_mean_30d) * np.sign(df['close'] - df['open'])
    
    # Final Enhanced Alpha Factor
    final_enhanced_alpha_factor = (
        final_alpha_factor * daily_market_sentiment +
        daily_market_sentiment_adjusted +
        volatility_ratio_diff +
        adjusted_true_range
    )
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_coefficient = np.where(volume_change_rate > 1, 2.5, np.where(volume_change_rate < -1, -2.0, 0.5))
    final_enhanced_alpha_factor = final_enhanced_alpha_factor * volume_trend_coefficient
    
    # Opening Impact Adjustment
    open_to_close_return = df['close'] - df['open']
    overnight_gap = df['open'] - df['close'].shift(1)
    opening_impact_factor = final_enhanced_alpha_factor - open_to_close_return - overnight_gap
    
    return opening_impact_factor
```
