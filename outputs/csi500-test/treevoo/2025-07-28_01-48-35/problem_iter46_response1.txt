```python
import pandas as pd

def heuristics_v2(df):
    # Momentum factor using closing prices
    short_term_return = df['close'].pct_change(5)
    long_term_return = df['close'].pct_change(20)
    momentum_factor = long_term_return - short_term_return

    # Analyze the relationship between trading volume and stock returns
    average_volume = df['volume'].rolling(window=30).mean()
    pct_change_volume = (df['volume'] / average_volume) - 1
    daily_return = df['close'].pct_change()
    volume_return_correlation = pct_change_volume.rolling(window=30).corr(daily_return)

    # Explore the volatility of the stock using the high and low prices
    true_range = pd.concat(
        [df['high'] - df['low'],
         (df['high'] - df['close'].shift(1)).abs(),
         (df['low'] - df['close'].shift(1)).abs()],
        axis=1
    ).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    atr_quartiles = pd.qcut(atr_14, 4, labels=False)
    atr_quartile_returns = df['close'].pct_change().groupby(atr_quartiles).mean()

    # Combine all factors into a single alpha factor
    alpha_factor = (momentum_factor + volume_return_correlation + atr_14).dropna()

    return alpha_factor
```
