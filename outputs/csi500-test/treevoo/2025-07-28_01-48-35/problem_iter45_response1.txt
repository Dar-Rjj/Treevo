```python
import pandas as pd

def heuristics_v2(df):
    # Compute 5-day and 20-day SMA
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate the difference between high and low prices to assess volatility
    df['HL_diff'] = df['high'] - df['low']
    df['Smoothed_Volatility'] = df['HL_diff'].rolling(window=10).mean()

    # Compute the 5-day cumulative volume and the 5-day cumulative price change
    df['Cum_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Cum_Price_Change_5'] = df['close'].diff().rolling(window=5).sum()
    df['Volume_to_Price_Change_Ratio'] = df['Cum_Volume_5'] / df['Cum_Price_Change_5'].replace(0, 1e-6)  # Avoid division by zero

    # Calculate the body of the candlestick (open - close)
    df['Candle_Body'] = df['close'] - df['open']
    df['Sum_Candle_Body_10'] = df['Candle_Body'].rolling(window=10).sum()

    # Generate the factor based on the combined signals
    df['SMA_Signal'] = (df['SMA_5'] > df['SMA_20']).astype(int) - (df['SMA_5'] < df['SMA_20']).astype(int)
    df['Volatility_Adjusted_SMA_Signal'] = df['SMA_Signal'] / (1 + df['Smoothed_Volatility'])
    df['Trend_Confirmation'] = (df['Volume_to_Price_Change_Ratio'].diff() > 0).astype(int) - (df['Volume_to_Price_Change_Ratio'].diff() < 0).astype(int)
    df['Sentiment_Signal'] = (df['Sum_Candle_Body_10'] > 0).astype(int) - (df['Sum_Candle_Body_10'] < 0).astype(int)

    # Combine all signals into a final alpha factor
    df['Alpha_Factor'] = (df['Volatility_Adjusted_SMA_Signal'] 
                          + df['Trend_Confirmation'] 
                          + df['Sentiment_Signal'])
    
    return df['Alpha_Factor']
```
