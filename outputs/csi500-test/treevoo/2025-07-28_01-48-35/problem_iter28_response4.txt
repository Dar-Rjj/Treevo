```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Scale Volume by Last 10 Days' Average
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Multiply Intraday Range by Scaled Volume
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Sum Returns of Close prices (last N days)
    N = 5
    close_returns = df['close'].pct_change().rolling(window=N).sum()
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Combined Momentum and Volume Factor
    combined_momentum_volume = close_returns * scaled_volume * close_to_range_ratio
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    
    # Intermediate Composite Factor
    intermediate_composite = adjusted_intraday_range + volume_weighted_close_to_open + combined_momentum_volume
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite = price_oscillator + intermediate_composite
    
    # Smoothed Composite Factor
    sma_7 = integrated_composite.rolling(window=7).mean()
    final_alpha_factor = sma_7
    
    # Additional Momentum Component
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    additional_momentum = df['close'] - ema_50
    
    # Enhanced Final Alpha Factor
    enhanced_alpha_factor = final_alpha_factor + additional_momentum
    
    return enhanced_alpha_factor
```
