```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum = (returns.rolling(window=N).sum() * close_to_range_ratio).fillna(0)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10d
    volume_trend_score = 1.5 * (volume_change_rate > 1) - 1.2 * (volume_change_rate < 1)
    
    # Detect Volume Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)).abs() > 2 * avg_volume_10d
    volume_spike_indicator = volume_spike.astype(int)
    
    # Integrate Price Trend
    sma_close = df['close'].rolling(window=20).mean()
    price_trend = df['close'] - sma_close
    
    # Combine Components
    factor = (momentum * volume_trend_score) + volume_spike_indicator + price_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_factor = factor - open_to_close_return
    
    return final_factor
```
