import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    adjusted_opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = (df['high'] - df['low']) + adjusted_opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    final_combined_factor = combined_high_low_open_close * close_to_prev_close_return
    final_combined_factor = final_combined_factor.clip(lower=0)
    
    # Volume Adjusted Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5_days) * close_to_intraday_range_ratio
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = volume_weighted_intraday_return + short_term_price_change + volume_weighted_high_low_spread
    
    # Volume Trend Indicator
    volume_trend_score = np.where(df['volume'].rolling(window=10).sum() > 0, 3, -3)
    
    # Composite Factor
    composite_factor = final_combined_factor * price_momentum + combined_signal
    composite_factor = composite_factor.clip(lower=0)
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=7).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    # Enhanced Intraday Volatility
    enhanced_intraday_volatility = (df['high'] - df['low']) / df['open'] * df['volume']
    
    # Long-Term Price Momentum
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_momentum = df['close'] - ema_50
    
    # Volume-Weighted Price Momentum
    volume_weighted_price_momentum = (df['volume'] / avg_volume_5_days) * long_term_price_momentum
    
    # Adjusted Combined Signal
    adjusted_combined_signal = enhanced_intraday_volatility + short_term_price_change + volume_weighted_price_momentum
    
    # Advanced Composite Factor
    advanced_composite_factor = final_combined_factor * long_term_price_momentum + adjusted_combined_signal
    advanced_composite_factor = advanced_composite_factor.clip(lower=0)
    
    # Final Alpha Factor with Enhanced Momentum
    final_alpha_factor_enhanced = smoothed_composite_factor * volume_trend_score
    
    return final_alpha_factor_enhanced
