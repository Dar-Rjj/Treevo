```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale Volume by Last 10 Days' Average
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_vol_10
    
    # Adjusted Intraday Range with Volume Trend
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Activity
    vol_change_5 = df['volume'] - df['volume'].shift(5)
    avg_vol_10_mean = df['volume'].rolling(window=10).mean()
    volume_change_rate = vol_change_5 / avg_vol_10_mean
    volume_trend = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < 1 else 0))
    
    # Combined Momentum and Volume Factor
    n_days = 10
    momentum_sum = df['close'].pct_change().rolling(window=n_days).sum()
    combined_momentum_volume = momentum_sum * close_to_range_ratio * volume_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    intermediate_composite = combined_momentum_volume - open_to_close_return
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Volume Trend
    sum_vol_20 = df['volume'].rolling(window=20).sum()
    volume_trend_indicator = df['volume'] / sum_vol_20
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    sma_7 = integrated_composite.rolling(window=7).mean()
    smoothed_composite = sma_7
    
    # Additional Momentum Component
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    additional_momentum = df['close'] - ema_30
    
    # Enhance Combined Momentum, Volume, and Price Factor
    high_low_spread = df['high'] - df['low']
    avg_high_low_10 = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / avg_high_low_10.shift(1)
    enhanced_factor = (combined_momentum_volume * volume_trend_indicator) / avg_vol_10 * relative_high_low_spread
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_factor + additional_momentum
    
    return final_alpha_factor
```
