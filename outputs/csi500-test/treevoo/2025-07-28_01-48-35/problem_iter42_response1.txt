```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = (df['high'] - df['low']) / df['low'] * df['volume']

    # Short-Term Price Change
    df['Short_Term_Price_Change'] = df['close'] - df['close'].shift(5)

    # Close to Intraday Range Ratio
    df['Close_to_Intraday_Range_Ratio'] = df['close'] / (df['high'] - df['low'])

    # Adjust for Opening Gap
    df['Opening_Gap'] = df['open'] - df['close'].shift(1)

    # Combine High-Low and Open-Close
    df['High_Low_Open_Close_Spread'] = (df['high'] - df['low']) + df['Opening_Gap']

    # Measure Closing Behavior
    df['Close_to_Prev_Close_Return'] = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    df['Final_Combined_Factor'] = df['High_Low_Open_Close_Spread'] * df['Close_to_Prev_Close_Return']

    # Volume-Weighted High-Low Spread
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    df['Volume_Weighted_High_Low_Spread'] = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)

    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Momentum'] = df['close'] - ema_20

    # Combined Signal
    df['Combined_Signal'] = (df['Volume_Weighted_Intraday_Return'] + 
                             df['Short_Term_Price_Change'] + 
                             df['Volume_Weighted_High_Low_Spread'])

    # Volume Trend Indicator
    volume_change_10_days = df['volume'].diff(periods=10)
    df['Volume_Trend_Score'] = volume_change_10_days.apply(lambda x: 3 if x > 0 else -3)

    # Composite Factor
    df['Composite_Factor'] = (df['Final_Combined_Factor'] * df['Price_Momentum']) + df['Combined_Signal']

    # Smooth the Composite Factor
    df['Smoothed_Composite_Factor'] = df['Composite_Factor'].rolling(window=7).mean()

    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Smoothed_Composite_Factor'] * df['Volume_Trend_Score']

    return df['Final_Alpha_Factor']
```
