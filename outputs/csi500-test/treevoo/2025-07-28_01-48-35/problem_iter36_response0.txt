import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open prices as a measure of daily momentum
    momentum = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the difference between the current day's close and the previous day's close
    df['prev_close'] = df['close'].shift(1)
    daily_return = df['close'] - df['prev_close']
    
    # Compute the average 5-day return to identify consistent trends
    five_day_return = df['close'].pct_change().rolling(window=5).mean()
    
    # Calculate the difference in volume from one day to the next
    df['volume_change'] = df['volume'].diff()
    volume_weighted_momentum = (df['close'] - df['open']) * (df['volume_change'] / df['volume'])
    
    # Calculate the ratio of amount to volume to infer average trade size
    avg_trade_size = df['amount'] / df['volume']
    amount_weighted_momentum = (df['close'] - df['open']) * avg_trade_size
    
    # Combine the factors to create a composite score
    composite_factor = (
        0.4 * momentum + 
        0.3 * five_day_return + 
        0.2 * volume_weighted_momentum + 
        0.1 * amount_weighted_momentum
    )
    
    return composite_factor.dropna()
