```python
import pandas as pd

def heuristics_v2(df):
    # 5-day Price Change
    df['price_change_5d'] = df['close'] - df['close'].shift(5)
    
    # 5-day Volume Change
    df['volume_change_5d'] = df['volume'] - df['volume'].shift(5)
    
    # Volume Trend Indicator
    df['volume_trend'] = (df['volume_change_5d'] > 0).astype(int) * 2 - 1  # +1 if True, -1 if False
    
    # Average Volume over 20 days (t-20 to t-1)
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean().shift(1)  # Exclude current day for average volume calculation
    
    # Final VAPM Factor
    df['VAPM'] = (df['price_change_5d'] * df['volume_trend']) / df['avg_volume_20d']
    
    return df['VAPM']
```
