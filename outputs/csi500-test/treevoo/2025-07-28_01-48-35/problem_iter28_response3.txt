```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio).fillna(0)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume
    
    # Apply Dynamic Coefficient to Volume Trend
    def dynamic_coefficient(x):
        if x > 2:
            return 1.7
        elif x < -2:
            return -1.5
        else:
            return 0.8
    volume_trend_score = volume_change_rate.apply(dynamic_coefficient)
    
    # Combine Components
    opening_return = (df['close'] - df['open']) / df['open']
    alpha_factor = (momentum_component * volume_trend_score) + opening_return
    
    return alpha_factor
```
