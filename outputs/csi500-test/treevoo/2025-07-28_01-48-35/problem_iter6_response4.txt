```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Exponential Moving Average (EMA) of Close Prices
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate 20-Day Sum of Volume
    df['Volume_20_Sum'] = df['volume'].rolling(window=20).sum()
    
    # Volume Trend
    df['Volume_Trend'] = df['volume'] / df['Volume_20_Sum']
    
    # Calculate 10-Day Average True Range (ATR)
    df['True_Range'] = df[['high', 'low', 'close']].join(
        df[['high', 'low', 'close']].shift(1)).apply(
        lambda x: max(x[0], x[3]) - min(x[1], x[4]), axis=1)
    df['ATR_10'] = df['True_Range'].rolling(window=10).mean()
    
    # Range Expansion
    df['Range_Expansion'] = df['True_Range'] / df['ATR_10']
    
    # On-Balance Volume (OBV)
    df['OBV'] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1] - df['volume'].iloc[i]
        else:
            df.at[df.index[i], 'OBV'] = df['OBV'].iloc[i-1]
    
    # Composite Factor
    df['Composite_Factor'] = (df['Price_Momentum'] 
                              + df['Volume_Trend'] 
                              + df['Range_Expansion'] 
                              + df['OBV'])
    
    # Smooth the Composite Factor with 5-Day Simple Moving Average (SMA)
    df['Smoothed_Composite_Factor'] = df['Composite_Factor'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = df['Smoothed_Composite_Factor']
    
    return alpha_factor
```
