```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    N = 5
    momentum_component = df['close'].pct_change().rolling(window=N).sum()
    momentum_component *= close_to_range_ratio
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    combined_signal = average_intraday_return + volume_adjusted_spread
    
    # Calculate Relative High-Low Spread
    high_low_spread = df['high'] - df['low']
    average_high_low_spread = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Opening Impact Factor
    opening_impact_factor = df['open'] - df['close']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_momentum = momentum_component * volume_trend_indicator
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    enhanced_momentum /= average_volume_20_days
    combined_volume_price_momentum = volume_trend_indicator * relative_high_low_spread
    final_factor = enhanced_momentum * combined_volume_price_momentum + opening_impact_factor
    
    return final_factor
```
