```python
def heuristics_v2(df):
    # Calculate the price change from previous close to current open
    price_change = df['open'] - df['close'].shift(1)
    
    # Calculate the buying pressure as (amount / volume) - (low + high) / 2
    buying_pressure = (df['amount'] / df['volume']) - (df['low'] + df['high']) / 2
    
    # Calculate the factor as a weighted sum of price change and buying pressure
    factor = 0.7 * price_change + 0.3 * buying_pressure
    
    return factor
```
