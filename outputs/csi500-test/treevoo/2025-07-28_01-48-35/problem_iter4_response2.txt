```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Scale Volume by Last 5 Days' Average
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5d
    
    # Volume Adjusted Factor
    volume_adjusted_factor = high_low_spread * scaled_volume
    
    # Combine Factors
    factor = volume_adjusted_factor + close_open_return
    
    return factor
```
