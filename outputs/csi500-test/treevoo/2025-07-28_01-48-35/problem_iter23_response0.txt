```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d_for_trend = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d_for_trend
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, -1.5)  # Fixed Coefficient
    
    # Combined Momentum and Volume Factor
    N = 5  # Example: Last 5 days
    returns_n_days = df['close'].pct_change(N).fillna(0)
    combined_momentum_volume = (returns_n_days + close_to_range_ratio) * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_volume - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    price_momentum_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - price_momentum_20d_ema
    volume_trend_20d_sum = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / volume_trend_20d_sum
    intermediate_composite = (adjusted_intraday_range + volume_weighted_close_to_open * price_momentum) * volume_trend
    
    # Price Oscillator
    ema_7d = df['close'].ewm(span=7, adjust=False).mean()
    ema_14d = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7d - ema_14d
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite = integrated_composite.rolling(window=7).mean()
    
    # Additional Momentum Component
    ema_50d = df['close'].ewm(span=50, adjust=False).mean()
    additional_momentum = df['close'] - ema_50d
    
    # Enhanced Final Alpha Factor
    enhanced_alpha_factor = smoothed_composite + additional_momentum
    return enhanced_alpha_factor
```
