```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume'])
    
    # Calculate the difference between VWAP and close price
    vwap_diff = vwap - df['close']
    
    # Calculate the spread between high and low prices
    spread = df['high'] - df['low']
    
    # Combine the factors using a heuristic formula
    factor = (price_change / spread) * (vwap_diff / df['close'])
    
    return factor
```
