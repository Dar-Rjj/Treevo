```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Apply Volume and Open-to-Close Return Filter
    df_filtered = df[(df['volume'] > N) & (df['close_to_open_return'] > 0)]
    
    # Summarize Weighted Price Changes Over 10 Days
    df['weighted_price_change'] = df_filtered['daily_price_change'] * df_filtered['volume'] * df_filtered['close_to_open_return']
    factor = df['weighted_price_change'].rolling(window=10).sum()
    
    return factor
```
