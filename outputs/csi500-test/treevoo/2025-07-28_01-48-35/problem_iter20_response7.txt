```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Calculate Short-Term Price Change
    df['short_term_price_change'] = df['close'].shift(-5) - df['close']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_intraday_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Volume Adjusted Momentum
    df['average_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_scaled'] = df['volume'] / df['average_volume_5_days']
    df['volume_adjusted_momentum'] = df['volume_scaled'] * df['close_to_intraday_range_ratio']
    
    # Cumulative Sum of Volume Adjusted Momentum
    df['cumulative_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=5).sum()
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    df['volume_factor'] = df['volume'] / df['average_volume_5_days']
    df['volume_weighted_high_low_spread'] = df['high_low_spread'] * df['volume_factor']
    
    # Calculate Price Momentum
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_20']
    
    # Combine Signals
    df['combined_signal'] = df['volume_weighted_intraday_return'] + df['short_term_price_change'] + df['high_low_spread']
    
    # Final Factor
    df['average_volume_10_days'] = df['volume'].rolling(window=10).mean()
    df['scaled_volume'] = df['volume'] / df['average_volume_10_days']
    df['final_factor'] = df['combined_signal'] * df['scaled_volume'] * df['cumulative_volume_adjusted_momentum'] * df['volume_weighted_high_low_spread'] * df['price_momentum']
    
    # Smooth the Composite Factor
    df['smoothed_final_factor'] = df['final_factor'].rolling(window=5).mean()
    
    return df['smoothed_final_factor'].dropna()
```
