```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio).fillna(0)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10_days
    volume_trend_score = volume_change_rate.apply(lambda x: 2.0 if x > 1.5 else (-1.5 if x < -1 else 0.5))
    
    # Calculate Volume-Weighted Intraday Return
    intraday_return = df['close'] - df['open']
    average_volume_15_days = df['volume'].rolling(window=15).mean()
    scaled_volume = df['volume'] / average_volume_15_days
    volume_weighted_intraday_return = intraday_return * scaled_volume
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = intraday_range * scaled_volume
    
    # Final Combined Factor
    final_factor = (momentum_component * volume_trend_score 
                    + volume_weighted_intraday_return 
                    + volume_adjusted_high_low_spread)
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_factor - open_to_close_return
    
    return final_alpha_factor
```
