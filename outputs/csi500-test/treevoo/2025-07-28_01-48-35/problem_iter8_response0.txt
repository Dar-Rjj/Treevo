import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    momentum_component = df['close'] - ema_10
    
    # Combine with Close-to-Range Ratio
    combined_momentum = close_to_range_ratio * momentum_component
    
    # Adjust for Volume Activity
    volume_20_sum = df['volume'].rolling(window=20).sum()
    volume_activity = df['volume'] / volume_20_sum
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = short_term_volume_change.apply(lambda x: 1 if x > 0 else -1)
    
    average_volume_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume_10
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 * x if x > 1 else -1.2 * x)
    
    # Calculate Range Expansion
    true_range = (df['high'] - df['low']).abs()
    atr_10 = true_range.rolling(window=10).mean()
    range_expansion = true_range / atr_10
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    obv_series = pd.Series(obv, index=df.index)
    
    # Final Combined Factor
    combined_factor = (combined_momentum * volume_trend_score) / volume_20_sum
    combined_factor = combined_factor * (range_expansion + obv_series)
    smoothed_combined_factor = combined_factor.rolling(window=5).mean()
    
    # Opening Impact Adjustment
    open_to_close_return = (df['open'] - df['close']) / df['open']
    final_alpha_factor = smoothed_combined_factor - open_to_close_return
    
    return final_alpha_factor
