```python
def heuristics_v2(df, N=5, M=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum = (returns.rolling(window=N).sum()) * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(N)
    avg_volume = df['volume'].rolling(window=M).mean()
    volume_change_rate = volume_change / avg_volume
    volume_trend_score = 1.5 * (volume_change_rate > 1) - 1.2 * (volume_change_rate < 1)
    
    # Integrate Price Trend
    moving_avg = df['close'].rolling(window=N).mean()
    price_trend = df['close'] - moving_avg
    
    # Combine Components
    combined_factor = (momentum * volume_trend_score) + price_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = combined_factor - open_to_close_return
    
    return final_alpha_factor
```
