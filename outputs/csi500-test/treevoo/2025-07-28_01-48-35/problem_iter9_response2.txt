```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return over N days
    avg_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    combined_signal = avg_intraday_return + volume_adjusted_spread
    
    # Calculate 10-Day Exponential Moving Average (EMA) of Close Prices
    ema_close = df['close'].ewm(span=N, adjust=False).mean()
    
    # Calculate Current Close Price - EMA
    close_ema_diff = df['close'] - ema_close
    
    # Calculate 10-Day Sum of Volume
    sum_volume_10_days = df['volume'].rolling(window=N).sum()
    
    # Calculate Volume Change Rate
    volume_change_rate = df['volume'] / sum_volume_10_days
    
    # Synthesize Combined Volume and Price Momentum
    composite_factor = volume_change_rate * close_ema_diff
    
    # Calculate 5-Day Simple Moving Average (SMA) of Composite Factor
    sma_composite_5_days = composite_factor.rolling(window=5).mean()
    
    # Introduce 10-Day Price Range
    price_range_10_days = df['high'].rolling(window=N).max() - df['low'].rolling(window=N).min()
    
    # Calculate 10-Day Average True Range (ATR)
    true_range = df['high'] - df['low']
    atr_10_days = true_range.rolling(window=N).mean()
    
    # Adjust Combined Signal with ATR
    adjusted_combined_signal = combined_signal / atr_10_days
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_signal * sma_composite_5_days
    
    return final_alpha_factor
```
