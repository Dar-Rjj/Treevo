```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Advanced Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    advanced_price_momentum = (df['close'] - ema_20) / (df['open'] - df['low'])
    
    # Combine with Close-to-Range Ratio
    combined_momentum = close_to_range_ratio * advanced_price_momentum
    
    # Adjust for Volume Activity
    volume_20_sum = df['volume'].rolling(window=20).sum()
    volume_trend = (df['volume'] / volume_20_sum)
    volume_change_5_day = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = np.where(volume_change_5_day > 0, 1, -1)
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume_10_days
    volume_trend_score = np.where(volume_change_rate > 1, 1.5 * volume_change_rate,
                                  np.where(volume_change_rate < 1, -1.2 * volume_change_rate, 0))
    
    # Calculate Range Expansion
    true_range = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
                            abs(df['low'] - df['close'].shift(1)))
    atr_10 = true_range.rolling(window=10).mean()
    range_expansion = true_range / atr_10
    
    # Calculate On-Balance Volume (OBV)
    obv = np.zeros_like(df['close'])
    obv[0] = 0
    for i in range(1, len(obv)):
        if df['close'][i] > df['close'][i-1]:
            obv[i] = obv[i-1] + df['volume'][i]
        elif df['close'][i] < df['close'][i-1]:
            obv[i] = obv[i-1] - df['volume'][i]
        else:
            obv[i] = obv[i-1]
    
    # Composite Factor
    composite_factor = (combined_momentum * volume_trend_score) / volume_20_sum
    composite_factor += range_expansion + obv
    smoothed_composite = composite_factor.ewm(span=10, adjust=False).mean()
    
    # Opening Impact Adjustment
    open_close_return = (df['open'] - df['close']) / df['open']
    final_alpha_factor = smoothed_composite - open_close_return
    
    return final_alpha_factor
```
