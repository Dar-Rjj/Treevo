```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_10d']
    df['adjusted_intraday_range'] = df['intraday_range'] * df['scaled_volume']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Adjust for Volume Activity
    df['volume_change_5d'] = df['volume'] - df['volume'].shift(5)
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume_change_5d'] / df['avg_volume_10d']
    df['volume_trend_score'] = df['volume_change_rate'].apply(lambda x: 2.0 if x > 1 else (-1.5 if x < 1 else 0))
    
    # Combined Momentum and Volume Factor
    N = 5  # Example value for N
    df['close_returns'] = df['close'].pct_change()
    df['momentum_factor'] = df['close_returns'].rolling(window=N).sum() * df['close_to_range_ratio'] * df['volume_trend_score']
    
    # Intermediate Composite Factor
    df['close_to_open_return'] = df['close'] - df['open']
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['scaled_volume']
    df['intermediate_composite_factor'] = (df['adjusted_intraday_range'] + df['volume_weighted_close_to_open']) * df['volume_trend_score']
    
    # Price Momentum
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_20']
    
    # Volume Trend
    df['volume_sum_20d'] = df['volume'].rolling(window=20).sum()
    df['volume_trend'] = df['volume'] / df['volume_sum_20d']
    
    # Price Oscillator
    df['ema_7'] = df['close'].ewm(span=7, adjust=False).mean()
    df['ema_14'] = df['close'].ewm(span=14, adjust=False).mean()
    df['price_oscillator'] = df['ema_7'] - df['ema_14']
    
    # Integrated Composite Factor
    df['integrated_composite_factor'] = df['intermediate_composite_factor'] + df['price_oscillator']
    
    # Smoothed Composite Factor
    df['smoothed_composite_factor'] = df['integrated_composite_factor'].rolling(window=7).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smoothed_composite_factor']
    
    return alpha_factor
```
