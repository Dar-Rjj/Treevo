```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    momentum_component = (df['close'].pct_change().rolling(window=N).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10_days
    adaptive_coefficient = volume_change_rate.apply(lambda x: 2.0 if x > 1.5 else (-1.5 if x < -1 else 0.5))
    
    # Volume-Weighted Intraday Return
    average_volume_15_days = df['volume'].rolling(window=15).mean()
    volume_weighted_intraday_return = (df['close'] - df['open']) * (df['volume'] / average_volume_15_days)
    
    # Calculate Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / average_volume_15_days)
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(N)
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = (
        momentum_component * adaptive_coefficient +
        volume_weighted_intraday_return +
        volume_adjusted_high_low_spread +
        short_term_price_change +
        (combined_high_low_open_close * closing_behavior) -
        (df['close'] - df['open'])
    )
    
    return final_combined_factor
```
