```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] / df['open'] - 1
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Calculate 10-Day EMA of Close Prices
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['EMA_10']
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Intraday Range and Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Scale Volume by Last 5 Days' Average
    df['Volume_Scaled'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Volume Adjusted Factor
    df['Volume_Adjusted_High_Low'] = df['High_Low_Spread'] * df['Volume_Scaled']
    df['Volume_Adjusted_Intraday'] = df['Intraday_Return'] * df['Volume_Scaled']
    
    # Combine Volume-Weighted Close-to-Open Return, Price Momentum, and Volume Trend
    df['Volume_Weighted_Composite'] = df['Volume_Weighted_Close_to_Open_Return'] * df['Price_Momentum'] * df['Volume_Trend']
    
    # 5-Day Simple Moving Average (SMA) of Composite Factor
    df['SMA_5_Composite'] = df['Volume_Weighted_Composite'].rolling(window=5).mean()
    
    # Combine Intraday and High-Low Signals
    df['Combined_Intraday_High_Low'] = df['Volume_Adjusted_High_Low'] + df['Volume_Adjusted_Intraday']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['SMA_5_Composite'] + df['Combined_Intraday_High_Low']) / df['volume'].rolling(window=20).mean()
    
    return df['Final_Alpha_Factor']
```
