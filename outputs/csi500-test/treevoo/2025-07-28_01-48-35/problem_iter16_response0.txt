```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    df['short_term_price_change'] = df['close'].diff(5)
    
    # Calculate Volume Trend Indicator
    df['volume_trend_indicator'] = (df['volume'].rolling(window=5).sum().diff() > 0).astype(int) * 2 - 1
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_close_to_open_return'] = (df['close'] - df['open']) * df['volume']

    # Combine Signals
    df['combined_signal'] = df['volume_weighted_intraday_return'] + df['short_term_price_change'] + df['high_low_spread']

    # Scaled Volume
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    df['scaled_volume'] = df['volume'] / average_volume_5_days

    # Final Factor
    df['final_factor'] = df['combined_signal'] * df['scaled_volume'] * df['volume_trend_indicator']

    return df['final_factor']
```
