```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    previous_close = df['close'].shift(1)
    opening_gap = df['open'] - previous_close
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = high_low_range + opening_gap
    
    # Measure Closing Behavior
    close_to_previous_close_return = df['close'] / previous_close
    final_combined_factor = combined_high_low_open_close * abs(close_to_previous_close_return)
    
    # Calculate Volume Trend Score
    average_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume_10d
    volume_trend_score = (volume_change_rate > 1.5) * 2.0 + (volume_change_rate < 0.8) * -1.5
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread_20d = high_low_spread.rolling(window=20).mean()
    normalized_high_low_spread = high_low_spread / average_high_low_spread_20d
    
    # Combine All Factors
    final_alpha_factor = final_combined_factor * normalized_high_low_spread * volume_trend_score
    
    return final_alpha_factor
```
