import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume-Weighted Intraday Return'] = (df['High'] - df['Low']) / df['Low'] * df['Volume']
    
    # Short-Term Price Change
    df['Short-Term Price Change'] = df['Close'] - df['Close'].shift(5)
    
    # High-Low Spread
    df['High-Low Spread'] = df['High'] - df['Low']
    
    # Daily Return
    df['Daily Return'] = (df['Close'] - df['Open']) / df['Open']
    
    # Short-term Momentum
    df['Short-term Momentum'] = (df['Close'] - df['Close'].shift(7)) / df['Close'].shift(7)
    
    # Medium-term Momentum
    df['Medium-term Momentum'] = (df['Close'] - df['Close'].shift(21)) / df['Close'].shift(21)
    
    # Long-term Momentum
    df['Long-term Momentum'] = (df['Close'] - df['Close'].shift(63)) / df['Close'].shift(63)
    
    # True Range
    df['True Range'] = df[['High' - 'Low', 
                            ('High' - df['Close'].shift(1)).abs(), 
                            ('Low' - df['Close'].shift(1)).abs()]].max(axis=1)
    
    # Relative Strength
    df['Relative Strength'] = df['Close'] / df['Close'].shift(21)
    
    # Money Flow Index
    money_flow_volume = (df['Close'] - df['Low']) - (df['High'] - df['Close']) * df['Volume'] / (df['High'] - df['Low'])
    df['Money Flow Index'] = money_flow_volume.rolling(window=14).sum() / df['Volume'].rolling(window=14).sum()
    
    # Positive Directional Movement
    df['Positive DM'] = (df['High'] - df['High'].shift(1)).apply(lambda x: max(x, 0)) - (df['Low'].shift(1) - df['Low']).apply(lambda x: min(x, 0))
    
    # Negative Directional Movement
    df['Negative DM'] = (df['Low'].shift(1) - df['Low']).apply(lambda x: max(x, 0)) - (df['High'] - df['High'].shift(1)).apply(lambda x: min(x, 0))
    
    # Directional Indicator
    df['ATR'] = df['True Range'].rolling(window=14).mean()
    df['+DI'] = df['Positive DM'].rolling(window=14).mean() / df['ATR']
    df['-DI'] = df['Negative DM'].rolling(window=14).mean() / df['ATR']
    
    # Volume Ratio
    df['SMA_Volume_21'] = df['Volume'].rolling(window=21).mean()
    df['Volume Ratio'] = df['Volume'] / df['SMA_Volume_21']
    
    # Weighted Average of Momentum Indicators
    df['Weighted Avg Momentum'] = df['Short-term Momentum'] * 0.3 + df['Medium-term Momentum'] * 0.5 + df['Long-term Momentum'] * 0.2
    
    # Price-Volatility Ratio
    df['Price-Volatility Ratio'] = (df['Close'] - df['Close'].shift(21)) / df['ATR']
    
    # Modified Volume-Weighted Intraday Return
    df['Modified VWIR'] = (df['High'] - df['Low']) / df['Open'] * df['Volume']
    
    # Cumulative Money Flow
    df['Cumulative MF'] = money_flow_volume.rolling(window=14).sum()
    
    # Relative High-Low
    df['Relative High-Low'] = (df['High'] - df['Low']) / df['Close']
    
    # Enhanced Short-Term Momentum
    df['Enhanced STM'] = (df['Close'] - df['Close'].shift(2)) / df['Close'].shift(2)
    
    # Volume-Adjusted Daily Return
    df['Volume-Adjusted DR'] = (df['Close'] - df['Open']) / df['Open'] * (df['Volume'] / df['SMA_Volume_21'])
    
    # Volatility-Adjusted Money Flow
    df['Volatility-Adjusted MF'] = df['Money Flow Index'] * (df['ATR'] / df['ATR'].rolling(window=21).mean
