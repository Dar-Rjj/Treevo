```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['10_day_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['10_day_ema']

    # Calculate Volume Trend
    df['20_day_sum_volume'] = df['volume'].rolling(window=20).sum()
    df['volume_trend'] = df['volume'] / df['20_day_sum_volume']

    # Calculate Range Expansion
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    df['10_day_atr'] = df['true_range'].rolling(window=10).mean()
    df['range_expansion'] = df['true_range'] / df['10_day_atr']

    # Composite Factor
    df['composite_factor'] = (df['price_momentum'] + 
                              df['volume_trend'] * 10 + 
                              df['range_expansion']) / 3
    
    # Smooth the Composite Factor
    df['alpha_factor'] = df['composite_factor'].rolling(window=5).mean()

    return df['alpha_factor']
```
