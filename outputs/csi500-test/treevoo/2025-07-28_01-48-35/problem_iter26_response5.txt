def heuristics_v2(data):
    # Calculate the price momentum
    data['price_momentum'] = (data['close'] - data['close'].shift(20)) / data['close'].shift(20)
    
    # Calculate the volume acceleration
    data['volume_acceleration'] = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    
    # Calculate the money flow
    data['money_flow'] = data['amount'] / data['volume']
    
    # Calculate the money flow ratio
    data['money_flow_ratio'] = data['money_flow'] / data['money_flow'].rolling(window=20).mean()
    
    # Calculate the final factor
    data['factor'] = (data['price_momentum'] * data['volume_acceleration'] * data['money_flow_ratio'])
    
    return data['factor'].dropna()
