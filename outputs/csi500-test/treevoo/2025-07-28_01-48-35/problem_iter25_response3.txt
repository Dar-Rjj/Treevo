```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume-Weighted Intraday Return'] = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Short-Term Price Change
    df['Short-Term Price Change'] = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    df['Close to Intraday Range Ratio'] = df['close'] / (df['high'] - df['low'])
    
    # Volume Adjusted Momentum
    df['Average Volume 5D'] = df['volume'].rolling(window=5).mean()
    df['Volume Adjusted Momentum'] = (df['volume'] / df['Average Volume 5D']) * df['Close to Intraday Range Ratio']
    
    # Cumulative Volume Adjusted Momentum
    df['Cumulative Volume Adjusted Momentum'] = df['Volume Adjusted Momentum'].rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    df['Volume Factor'] = df['volume'] / df['Average Volume 5D']
    df['Volume-Weighted High-Low Spread'] = (df['high'] - df['low']) * df['Volume Factor']
    
    # Price Momentum
    df['EMA 20D'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price Momentum'] = df['close'] - df['EMA 20D']
    
    # Combined Signal
    df['Combined Signal'] = df['Volume-Weighted Intraday Return'] + df['Short-Term Price Change'] + df['Volume-Weighted High-Low Spread']
    
    # Volume Trend Indicator
    df['Average Volume 10D'] = df['volume'].rolling(window=10).mean()
    df['Volume Trend Indicator'] = df['volume'] / df['Average Volume 10D']
    
    # Final Alpha Factor
    df['Composite Factor'] = df['Combined Signal'] * df['Volume Trend Indicator']
    df['Smoothed Composite Factor'] = df['Composite Factor'].rolling(window=5).mean()
    
    return df['Smoothed Composite Factor']
```
