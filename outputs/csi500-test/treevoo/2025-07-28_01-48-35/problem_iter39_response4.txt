```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'].diff(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = (df['high'] - df['low']) + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1) - 1
    
    # Final Combined Factor
    final_combined_factor = (combined_high_low_open_close * close_to_prev_close_return).clip(lower=0)
    
    # Volume-Adjusted Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5_days) * close_to_intraday_range_ratio
    
    # Cumulative Volume-Adjusted Momentum
    cumulative_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = (volume_weighted_intraday_return + 
                       short_term_price_change + 
                       volume_weighted_high_low_spread)
    
    # Volume Trend Indicator
    volume_trend_score = df['volume'].pct_change(10).apply(lambda x: 2 if x > 0 else -2)
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum + combined_signal).clip(lower=0)
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    # Final Factor
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10_days
    final_factor = final_alpha_factor * scaled_volume
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    overnight_gap = df['open'] - df['close'].shift(1)
    final_factor -= open_to_close_return - overnight_gap
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    intraday_volatility = intraday_volatility.clip(lower=0)
    
    # Long-Term Trend
    ema_60 = df['close'].ewm(span=60, adjust=False).mean()
    long_term_trend = df['close'] - ema_60
    
    # Combined Long-Term and Short-Term Momentum
    combined_momentum = (price_momentum + long_term_trend).clip(lower=0)
    
    # Enhanced Composite Factor
    enhanced_composite_factor = (composite_factor * combined_momentum).clip(lower=0)
    
    # Smoothed Enhanced Composite Factor
    smoothed_enhanced_composite_factor = enhanced_composite_factor.rolling(window=7).mean()
    
    # Final Enhanced Alpha Factor
    final_enhanced_alpha_factor = smoothed_enhanced_composite_factor * volume_trend_score
    
    # Final Enhanced Factor
    final_enhanced_factor = final_enhanced_alpha_factor * scaled_volume
    
    return final_enhanced_factor
```
