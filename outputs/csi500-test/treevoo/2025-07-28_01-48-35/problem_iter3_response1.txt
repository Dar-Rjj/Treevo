```python
import pandas as pd

def heuristics_v2(df):
    # Compute 20-Day Exponential Moving Average (EMA) of Close Prices
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Subtract EMA from Current Close Price to get Enhanced Price Momentum
    df['Enhanced_Price_Momentum'] = df['close'] - df['EMA_20']
    
    # Compute 5-Day Sum of Volume
    df['Sum_Volume_5D'] = df['volume'].rolling(window=5).sum()
    
    # Compute 15-Day Sum of Volume
    df['Sum_Volume_15D'] = df['volume'].rolling(window=15).sum()
    
    # Divide 5-Day Sum by 15-Day Sum of Volume to get Volume Acceleration
    df['Volume_Acceleration'] = df['Sum_Volume_5D'] / df['Sum_Volume_15D']
    
    # Multiply Enhanced Price Momentum by Volume Acceleration to get the Composite Factor
    df['Composite_Factor'] = df['Enhanced_Price_Momentum'] * df['Volume_Acceleration']
    
    # Smooth the Composite Factor with a 3-Day Simple Moving Average (SMA)
    df['Smoothed_Composite_Factor'] = df['Composite_Factor'].rolling(window=3).mean()
    
    # Final Alpha Factor
    alpha_factor = df['Smoothed_Composite_Factor']
    
    return alpha_factor
```
