```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum (10-Day EMA of Close Prices)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Integrate Short-Term Price and Volume Changes
    five_day_price_change = df['close'] - df['close'].shift(5)
    five_day_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = np.where(five_day_volume_change > 0, 1, -1)
    avg_volume_20 = df['volume'].rolling(window=20).mean().shift(1)
    final_vapm_factor = (five_day_price_change * volume_trend_indicator) / avg_volume_20
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Integrate Price Trend
    sma_close = df['close'].rolling(window=20).mean()
    price_trend = df['close'] - sma_close
    
    # Integrate Trading Volume Signal
    trading_volume_signal = np.where(df['volume'] > avg_volume_20, 1, -1)
    
    # Combine and Cumulate
    combined_momentum = volume_weighted_momentum + price_trend + final_vapm_factor + trading_volume_signal
    cumulative_combined_momentum = combined_momentum.rolling(window=20).sum()
    
    # Composite Factor
    composite_factor = volume_weighted_return * cumulative_combined_momentum
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
