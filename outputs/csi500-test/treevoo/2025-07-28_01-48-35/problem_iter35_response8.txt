```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_vol_10
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Compute 5-day Volume Change
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_vol_10
    volume_trend = np.where(volume_change_rate > 1, 2.0, -1.5)
    
    # Combined Momentum and Volume Factor
    returns_close = df['close'].pct_change().rolling(window=5).sum()
    combined_momentum_volume = returns_close * close_to_range_ratio * volume_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_volume - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = (df['close'] - df['open'])
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    intermediate_composite = adjusted_intraday_range + volume_weighted_close_to_open * price_momentum * volume_trend
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume Trend
    vol_trend_indicator = df['volume'] / df['volume'].rolling(window=20).sum()
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite = intermediate_composite + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite = integrated_composite.rolling(window=7).mean()
    
    # Additional Momentum Component
    additional_momentum = df['close'] - df['close'].ewm(span=30, adjust=False).mean()
    
    # Enhance Combined Momentum, Volume, and Price Factor
    relative_high_low_spread = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    enhanced_combined = (combined_momentum_volume * vol_trend_indicator) / avg_vol_10 * (vol_trend_indicator * relative_high_low_spread)
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_composite + additional_momentum + opening_impact_adjustment + enhanced_combined
    
    return final_alpha_factor
```
