```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Range and Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Scale Volume by Last 5 Days' Average
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5_days
    
    # Volume-Adjusted Factors
    volume_adjusted_high_low = high_low_spread * scaled_volume
    volume_adjusted_intraday = intraday_return * scaled_volume
    
    # Combine Intraday and High-Low Signals
    combined_signal = volume_adjusted_high_low + volume_adjusted_intraday
    
    # Final Alpha Factor
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    final_alpha_factor = combined_signal / avg_volume_20_days
    
    return final_alpha_factor
```
