```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    df['Volume-Weighted Intraday Return'] = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    df['Short-Term Price Change'] = df['close'].diff(5)
    
    # Calculate Volume Trend Indicator
    df['Volume Trend Indicator'] = (df['volume'].pct_change(5) > 0).astype(int) * 2 - 1
    
    # Calculate High-Low Spread
    df['High-Low Spread'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['Volume-Weighted Close-to-Open Return'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate Long-Term Price Change
    df['Long-Term Price Change'] = df['close'].diff(20)
    
    # Combine Signals
    df['Combined Signal'] = (
        df['Volume-Weighted Intraday Return'] +
        df['Short-Term Price Change'] +
        df['High-Low Spread'] +
        df['Long-Term Price Change']
    )
    
    # Final Factor
    df['Scaled Volume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['Final Factor'] = (
        df['Combined Signal'] * 
        df['Scaled Volume'] * 
        df['Volume Trend Indicator'] * 
        df['Long-Term Price Change']
    )
    
    return df['Final Factor']
```
