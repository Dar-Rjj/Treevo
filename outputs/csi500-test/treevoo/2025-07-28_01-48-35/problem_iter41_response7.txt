```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average of close prices over a 5-day window
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the simple moving average of close prices over a 10-day window
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the relative strength by comparing the 5-day SMA to the 10-day SMA
    df['Relative_Strength'] = df['SMA_5'] / df['SMA_10']
    
    # Calculate the average true range (ATR) over a 14-day window
    df['High-Low'] = df['high'] - df['low']
    df['High-Close'] = (df['high'] - df['close'].shift()).abs()
    df['Low-Close'] = (df['low'] - df['close'].shift()).abs()
    df['True_Range'] = df[['High-Low', 'High-Close', 'Low-Close']].max(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Factor: Relative Strength * ATR
    factor = df['Relative_Strength'] * df['ATR_14']
    
    return factor
```
