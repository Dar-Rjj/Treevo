def heuristics_v2(df):
    # Momentum-based Alpha Factors
    df['5d_SMA'] = df['close'].rolling(window=5).mean()
    df['10d_SMA'] = df['close'].rolling(window=10).mean()
    df['20d_SMA'] = df['close'].rolling(window=20).mean()
    df['50d_SMA'] = df['close'].rolling(window=50).mean()
    df['200d_SMA'] = df['close'].rolling(window=200).mean()

    df['5d_ROC'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10d_ROC'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['20d_ROC'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['50d_ROC'] = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    df['200d_ROC'] = (df['close'] - df['close'].shift(200)) / df['close'].shift(200)

    df['1m_Momentum'] = df['close'].pct_change(periods=21)
    df['3m_Momentum'] = df['close'].pct_change(periods=63)
    df['6m_Momentum'] = df['close'].pct_change(periods=126)
    df['1y_Momentum'] = df['close'].pct_change(periods=252)

    # Volume-based Alpha Factors
    df['VWAP'] = (df['amount'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['5d_VWAP'] = df['VWAP'].rolling(window=5).mean()
    df['10d_VWAP'] = df['VWAP'].rolling(window=10).mean()
    df['20d_VWAP'] = df['VWAP'].rolling(window=20).mean()
    df['50d_VWAP'] = df['VWAP'].rolling(window=50).mean()
    df['200d_VWAP'] = df['VWAP'].rolling(window=200).mean()

    df['5d_VROC'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['10d_VROC'] = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    df['20d_VROC'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    df['50d_VROC'] = (df['volume'] - df['volume'].shift(50)) / df['volume'].shift(50)
    df['200d_VROC'] = (df['volume'] - df['volume'].shift(200)) / df['volume'].shift(200)

    df['1m_Volume_Momentum'] = df['volume'].pct_change(periods=21)

    df['6m_Volume_Momentum'] = df['volume'].pct_change(periods=126)
    df['1y_Volume_Momentum'] = df['volume'].pct_change(periods=252)
