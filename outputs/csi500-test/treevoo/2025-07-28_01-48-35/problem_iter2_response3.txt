```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Volume Adjusted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_factor = df['volume'] / avg_volume_5d
    weighted_high_low_spread = high_low_spread * volume_factor
    
    # Combine Factors
    combined_factor = weighted_high_low_spread + close_open_return
    
    # Calculate Price Momentum
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10d
    
    # Calculate Volume Trend
    sum_volume_10d = df['volume'].rolling(window=10).sum()
    volume_trend = df['volume'] / sum_volume_10d
    
    # Composite Factor
    composite_factor = combined_factor * price_momentum * volume_trend
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
