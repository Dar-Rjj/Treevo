```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the weighted moving average of close price with volume as weight
    df['wma_close'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['close_change'] * df['high_low_diff'] * df['volume_amount_ratio'] * df['wma_close'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
