```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    N = 5
    momentum_component = (df['close'].pct_change().rolling(window=N).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    dynamic_coefficient = (volume_change_rate > 2).astype(int) * 1.7 + (volume_change_rate < 2).astype(int) * -1.5 + (volume_change_rate.between(-2, 2)).astype(int) * 0.8
    adjusted_momentum = momentum_component * dynamic_coefficient
    
    # Add Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    factor_with_opening_impact = adjusted_momentum - open_to_close_return
    
    # Detect Volume Spikes
    volume_change = df['volume'] - df['volume'].shift(1)
    significant_volume_change = (volume_change > volume_change.std()).astype(int)
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjusted High-Low Spread with Volume Trend
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open = close_to_open_return * scaled_volume
    
    # Intermediate Composite Factor
    intermediate_factor = (adjusted_high_low_spread + volume_weighted_close_to_open * price_trend) * scaled_volume
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Volume Trend Indicator
    volume_trend_indicator = (volume_change_5d > 0).astype(int) * 2 - 1
    
    # Integrated Composite Factor
    integrated_factor = (price_trend + short_term_price_change + volume_trend_indicator) * scaled_volume
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Enhanced Composite Factor with Oscillator
    enhanced_factor = integrated_factor + price_oscillator
    sma_5_enhanced_factor = enhanced_factor.rolling(window=5).mean()
    weighting_factor = df['volume'] / avg_volume_10d
    smoothed_factor = sma_5_enhanced_factor * weighting_factor
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_factor.rolling(window=5).mean()
    
    return final_alpha_factor
```
