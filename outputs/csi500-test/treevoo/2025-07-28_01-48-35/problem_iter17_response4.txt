```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_intraday_range_ratio
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    volume_spike = (volume_change > 0.5).astype(int)
    
    # Integrate Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_20
    
    # Combine Components
    combined_components = volume_weighted_intraday_return + volume_adjusted_momentum + price_trend + volume_spike
    
    # Apply Rolling Sum on Combined Components
    rolling_combined = combined_components.rolling(window=10).sum()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjusted High-Low Spread with Volume Trend
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10_days
    adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = close_to_open_return * scaled_volume
    
    # Intermediate Composite Factor
    intermediate_composite_factor = (adjusted_high_low_spread + volume_weighted_close_to_open_return) * price_trend * scaled_volume
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Volume Trend Indicator
    volume_trend_indicator = (df['volume'].rolling(window=5).sum() - df['volume'].shift(5).rolling(window=5).sum()).apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Signals
    combined_signal = volume_weighted_intraday_return + short_term_price_change + high_low_spread
    
    # Integrated Composite Factor
    integrated_composite_factor = combined_signal + price_trend + volume_trend_indicator * scaled_volume
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Integrated Composite Factor with Oscillator
    integrated_composite_factor_with_oscillator = integrated_composite_factor + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite_factor = integrated_composite_factor_with_oscillator.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = smoothed_composite_factor
    
    return alpha_factor
```
