```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    vol_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'].diff(5)
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Volume-Weighted High-Low Spread
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    vol_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5_days)
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Combined Signal
    combined_signal = vol_weighted_intraday_return + short_term_price_change + vol_weighted_high_low_spread
    
    # Volume Trend Indicator
    volume_change_10_days = df['volume'].pct_change(10)
    volume_trend_score = np.where(volume_change_10_days > 0, 3, -3)
    
    # Composite Factor
    composite_factor = (closing_behavior * price_momentum) + combined_signal
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=7).mean()
    
    # Incorporate Volume Acceleration
    ema_volume_5_days = df['volume'].ewm(span=5, adjust=False).mean()
    ema_volume_20_days = df['volume'].ewm(span=20, adjust=False).mean()
    volume_acceleration = ema_volume_5_days / ema_volume_20_days
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (smoothed_composite_factor * volume_trend_score) * volume_acceleration
    
    return final_alpha_factor
```
