```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_intraday_range_ratio
    
    # Integrate Short-Term Price and Volume Changes
    price_change_5d = df['close'].pct_change(5)
    volume_change_5d = df['volume'].pct_change(5)
    volume_trend_indicator = (volume_change_5d > 0).astype(int) * 2 - 1
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    final_vapm_factor = (price_change_5d * volume_trend_indicator) / avg_volume_20d
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Integrate Price Trend
    sma_20d = df['close'].rolling(window=20).mean()
    price_trend = df['close'] - sma_20d
    
    # Combine and Cumulate
    combined_momentum = volume_weighted_momentum + price_trend + final_vapm_factor
    cumulated_combined_momentum = combined_momentum.rolling(window=20).sum()
    
    # Composite Alpha Factor
    composite_factor = volume_weighted_close_to_open_return + cumulated_combined_momentum
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
