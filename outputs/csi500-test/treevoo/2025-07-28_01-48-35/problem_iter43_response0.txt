```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    df['short_term_price_change'] = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    df['close_to_intraday_range_ratio'] = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    df['combined_high_low_open_close'] = (df['high'] - df['low']) + (df['open'] - df['close'].shift(1))
    
    # Close-to-Previous Close Return
    df['close_to_previous_close_return'] = df['close'] / df['close'].shift(1)
    
    # Combined Factor
    df['combined_factor'] = df['combined_high_low_open_close'] * df['close_to_previous_close_return']
    
    # Volume-Adjusted Momentum
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = (df['volume'] / df['avg_volume_5_days']) * df['close_to_intraday_range_ratio']
    
    # Cumulative Volume-Adjusted Momentum
    df['cumulative_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    df['volume_weighted_high_low_spread'] = (df['volume'] / df['avg_volume_5_days']) * (df['high'] - df['low'])
    
    # Price Momentum
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_20']
    
    # Combined Signal
    df['combined_signal'] = df['volume_weighted_intraday_return'] + df['short_term_price_change'] + df['volume_weighted_high_low_spread']
    
    # Volume Trend Indicator
    df['volume_trend_indicator'] = (df['volume'] - df['volume'].shift(10)).apply(lambda x: 2 if x > 0 else -2)
    
    # Composite Factor
    df['composite_factor'] = (df['combined_factor'] * df['price_momentum']) + df['combined_signal']
    
    # Smooth the Composite Factor
    df['smoothed_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_composite_factor'] * df['volume_trend_indicator']
    
    # Final Factor
    df['final_factor'] = (df['volume'] / df['volume'].rolling(window=10).mean()) * df['final_alpha_factor']
    
    # Opening Impact and Gap Adjustment
    df['opening_impact_and_gap_adjustment'] = (df['open'] - df['close']) + (df['open'] - df['close'].shift(1))
    
    # Enhanced Combined Momentum, Volume, and Price Factor
    df['enhanced_combined_factor'] = (df['volume_adjusted_momentum'] + df['price_momentum'] - df['opening_impact_and_gap_adjustment']).apply(lambda x: max(x, 0))
    
    return df['enhanced_combined_factor']
```
