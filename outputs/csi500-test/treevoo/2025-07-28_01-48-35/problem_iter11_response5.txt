```python
def heuristics_v2(data):
    """
    Generate an alpha factor using a combination of price and volume data.
    
    Parameters:
        data (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
        pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the difference between high and low prices
    price_diff = data['high'] - data['low']
    
    # Calculate the volume weighted average price
    vwap = (data['amount'] / data['volume']).fillna(0)
    
    # Calculate the change in VWAP
    vwap_change = vwap.diff().fillna(0)
    
    # Combine the price difference and VWAP change
    factor = price_diff * vwap_change
    
    return factor
```
