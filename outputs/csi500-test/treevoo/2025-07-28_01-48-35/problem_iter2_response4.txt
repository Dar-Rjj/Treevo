```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Compute 10-Day Exponential Moving Average (EMA) of Close Prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    price_momentum = df['close'] - ema_10
    
    # Compute 10-Day Sum of Volume
    volume_sum_10 = df['volume'].rolling(window=10).sum()
    
    # Volume Trend
    volume_trend = df['volume'] / volume_sum_10
    
    # Composite Factor
    composite_factor = (volume_weighted_return * price_momentum) * volume_trend
    
    # Smooth the Composite Factor with 5-Day Simple Moving Average (SMA)
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
```
