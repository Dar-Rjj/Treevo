```python
def heuristics_v2(df):
    # Calculate the average of (open, high, low, close) for each day
    avg_price = df[['open', 'high', 'low', 'close']].mean(axis=1)
    
    # Calculate the total turnover by multiplying amount and volume
    total_turnover = df['amount'] * df['volume']
    
    # Compute the factor value as the ratio of average price to total turnover
    factor_values = avg_price / total_turnover
    
    return factor_values
```
