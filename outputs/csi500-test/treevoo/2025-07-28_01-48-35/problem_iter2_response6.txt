```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Opening Gap
    previous_close = df['close'].shift(1)
    opening_gap = df['open'] - previous_close
    
    # Combine High-Low and Open-Close
    combined_range = high_low_range + opening_gap
    
    # Measure Closing Behavior
    close_to_previous_close_return = df['close'] / previous_close
    final_combined_factor = combined_range * abs(close_to_previous_close_return)
    
    # Calculate Volume Trend Score
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 * x if x > 1 else -1.2 * x)
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / average_high_low_spread
    
    # Combine All Factors
    final_alpha_factor = final_combined_factor * normalized_high_low_spread * volume_trend_score
    
    return final_alpha_factor
```
