```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    close_to_intraday_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = (df['high'] - df['low']) + (df['open'] - df['close'].shift(1))
    
    # Measure Closing Behavior
    closing_behavior = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = (combined_high_low_open_close * closing_behavior).apply(lambda x: max(x, 0))
    
    # Volume Adjusted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = (df['volume'] / avg_volume_5d) * close_to_intraday_range_ratio
    
    # Cumulative Volume Adjusted Momentum
    cumulative_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).sum()
    
    # Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / avg_volume_5d)
    
    # Price Momentum
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20d
    
    # Combined Signal
    combined_signal = volume_weighted_intraday_return + short_term_price_change + volume_weighted_high_low_spread
    
    # Volume Trend Indicator
    volume_trend_indicator = np.where(df['volume'].diff(10) > 0, 2, -2)
    
    # Composite Factor
    composite_factor = (final_combined_factor * price_momentum + combined_signal).apply(lambda x: max(x, 0))
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_indicator
    
    # Final Factor
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    final_factor = (final_alpha_factor * (df['volume'] / avg_volume_10d))
    
    # Integrate Price Trend
    price_trend = df['close'] - ema_20d
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (cumulative_volume_adjusted_momentum * opening_gap / avg_volume_10d) * price_trend
    
    # Opening Impact and Gap Adjustment
    overnight_gap = df['open'] - df['close'].shift(1)
    enhanced_factor = enhanced_factor - overnight_gap
    
    # Output as Final Alpha Factor
    final_alpha_factor = enhanced_factor.apply(lambda x: max(x, 0))
    
    return final_alpha_factor
```
