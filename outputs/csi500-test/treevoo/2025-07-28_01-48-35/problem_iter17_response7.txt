```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjusted High-Low Spread with Volume Trend
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10
    adjusted_high_low_spread = high_low_spread * scaled_volume
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_open_return = close_open_return * scaled_volume
    
    # Calculate Price Momentum
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_momentum = df['close'] - ema_20
    
    # Calculate Volume Trend
    sum_volume_20 = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / sum_volume_20
    
    # Intermediate Composite Factor
    intermediate_composite_factor = (
        (adjusted_high_low_spread + volume_weighted_close_open_return) *
        volume_weighted_close_open_return * 
        price_momentum * 
        volume_trend
    )
    
    # Price Oscillator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_oscillator = ema_5 - ema_10
    
    # Integrated Composite Factor
    integrated_composite_factor = intermediate_composite_factor + price_oscillator
    
    # Enhanced Smoothed Composite Factor
    wma_weights = np.array([0.1, 0.2, 0.3, 0.2, 0.1])
    enhanced_smoothed_composite_factor = integrated_composite_factor.rolling(window=5).apply(
        lambda x: np.sum(wma_weights * x), raw=True
    )
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_smoothed_composite_factor
    
    return final_alpha_factor
```
