```python
import pandas as pd

def heuristics_v2(df):
    # Compute 10-Day Exponential Moving Average (EMA) of Close Prices
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Enhanced Price Momentum
    df['enhanced_momentum'] = (df['close'] - df['EMA_10']) / (df['high'] - df['low'])
    
    # Compute 10-Day Sum of Volume-Weighted Close Prices
    df['volume_weighted_close_sum_10'] = (df['close'] * df['volume']).rolling(window=10).sum()
    
    # Compute 10-Day Sum of Volume
    df['volume_sum_10'] = df['volume'].rolling(window=10).sum()
    
    # Divide 10-Day Sum of Volume-Weighted Close Prices by 10-Day Sum of Volume
    df['volume_weighted_price_trend'] = df['volume_weighted_close_sum_10'] / df['volume_sum_10']
    
    # Composite Factor
    df['composite_factor'] = df['enhanced_momentum'] * df['volume_weighted_price_trend']
    
    # Smooth the Composite Factor
    df['smoothed_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smoothed_composite_factor']
    
    return alpha_factor
```
