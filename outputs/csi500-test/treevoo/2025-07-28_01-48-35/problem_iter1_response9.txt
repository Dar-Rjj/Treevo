```python
import pandas as pd

def heuristics(data):
    # Compute Average Volume (Last 10 Days)
    avg_volume = data['volume'].rolling(window=10).mean()
    
    # Compute Volume Change Rate
    volume_change_rate = data['volume'] / avg_volume
    
    # Summarize Volume Trend
    volume_trend_score = volume_change_rate.copy()
    volume_trend_score[volume_change_rate > 1] *= 1.5
    volume_trend_score[volume_change_rate < 1] *= -1.2
    
    # Compute High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Compute Average High-Low Spread (Last 10 Days)
    avg_high_low_spread = high_low_spread.rolling(window=10).mean()
    
    # Normalize Daily High-Low Spread
    normalized_high_low_spread = high_low_spread / avg_high_low_spread
    
    # Combine Volume Trend and Price Momentum
    alpha_factor = volume_trend_score * normalized_high_low_spread
    
    return alpha_factor
```
