```python
import pandas as pd

def heuristics_v2(df, window_momentum=10, window_sma=5, volume_spike_threshold=2.0, window_cumsum=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_intraday_range_ratio
    
    # Apply Rolling Sum on Volume Adjusted Momentum
    rolling_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=window_momentum).sum()
    
    # Detect Volume Spikes
    volume_change = df['volume'] / df['volume'].shift(1)
    volume_spike_indicator = (volume_change > volume_spike_threshold).astype(int)
    
    # Integrate Price Trend
    sma_close = df['close'].rolling(window=window_sma).mean()
    price_trend = df['close'] - sma_close
    
    # Combine Volume Adjusted Momentum, Price Trend, and Volume Spike Indicator
    combined_components = rolling_volume_adjusted_momentum + price_trend + volume_spike_indicator
    
    # Cumulative Sum of Combined Components
    cumulative_sum = combined_components.rolling(window=window_cumsum).sum()
    
    return cumulative_sum
```
