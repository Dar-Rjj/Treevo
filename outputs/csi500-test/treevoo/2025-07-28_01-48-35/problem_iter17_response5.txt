```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Integrate Short-Term Price and Volume Changes
    five_day_price_change = df['close'] - df['close'].shift(5)
    five_day_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = (five_day_volume_change > 0).astype(int) * 2 - 1
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    vapm_factor = (five_day_price_change * volume_trend_indicator) / avg_volume_20
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Integrate Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    price_trend = df['close'] - sma_20
    
    # Combine and Cumulate
    combined_momentum = volume_weighted_momentum + price_trend + vapm_factor
    cumulated_momentum = combined_momentum.rolling(window=5).sum()
    
    # Enhanced Composite Factor
    enhanced_composite = volume_weighted_return * cumulated_momentum + (df['close'] - ema_10)
    
    # Smooth the Enhanced Composite Factor
    smoothed_enhanced_composite = enhanced_composite.rolling(window=5).mean()
    
    return smoothed_enhanced_composite
```
