import pandas as pd

def heuristics_v2(df, n=5, vol_trend_days=5, avg_vol_days=10, ema_span=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    momentum = df['close'].pct_change().rolling(window=n).sum() * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(vol_trend_days)
    avg_volume = df['volume'].rolling(window=avg_vol_days).mean()
    volume_change_rate = volume_change / avg_volume
    vol_trend_score = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    price_trend = df['close'] - df['close'].ewm(span=ema_span, adjust=False).mean()
    
    # Combine Initial Signals
    combined_signal = (intraday_return + volume_adjusted_spread) * momentum * close_to_range_ratio * vol_trend_score * price_trend
    
    # Opening Impact and Gap Adjustment
    overnight_gap = df['open'] - df['close'].shift(1)
    opening_impact = (df['close'] - df['open']) - overnight_gap
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (combined_signal * opening_impact) / df['volume'].rolling(window=20).mean()
    
    return enhanced_factor.dropna()
