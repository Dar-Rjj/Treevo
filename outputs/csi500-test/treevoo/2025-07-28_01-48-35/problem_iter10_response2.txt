import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10_day = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10_day
    
    # Composite Factor
    composite_factor = volume_weighted_close_to_open_return * price_momentum + volume_weighted_close_to_open_return.rolling(window=5).sum()
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    return smoothed_composite_factor
