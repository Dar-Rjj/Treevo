```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """

    # Calculate the average of open, high, low, and close prices
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4

    # Calculate the price range
    price_range = df['high'] - df['low']

    # Calculate the volume-weighted price
    vw_price = (df['close'] * df['volume']) / df['volume'].rolling(window=10).mean()

    # Calculate the momentum factor
    mom_factor = df['close'].pct_change(periods=21)

    # Calculate the amount/volume ratio
    av_ratio = df['amount'] / df['volume']

    # Combine the factors into a single alpha factor
    alpha_factor = (avg_price / df['close'].shift(1)) * (price_range / df['close'].shift(1)) * vw_price * mom_factor * av_ratio

    return alpha_factor
```
