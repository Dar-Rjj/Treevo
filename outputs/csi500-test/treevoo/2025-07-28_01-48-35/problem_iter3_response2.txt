```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Determine Price Trend
    ma_5_day = df['close'].rolling(window=5).mean()
    price_trend = df['close'] > ma_5_day
    
    # Calculate Volume Weighted Momentum (If Price Trend is Positive)
    volume_weighted_momentum = df['volume'] * close_to_range_ratio * price_trend
    
    # Cumulative Sum of Volume Weighted Momentum
    cumulative_momentum = volume_weighted_momentum.rolling(window=10).sum()
    
    return cumulative_momentum
```
