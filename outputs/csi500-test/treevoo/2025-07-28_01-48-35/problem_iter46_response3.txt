```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Short-Term Price Change
    df['Short_Term_Price_Change'] = df['close'] - df['close'].shift(5)
    
    # Adjust for Opening Gap
    df['Opening_Gap_Adjustment'] = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    df['High_Low_Open_Close_Combined'] = (df['high'] - df['low']) + df['Opening_Gap_Adjustment']
    
    # Measure Closing Behavior
    df['Close_to_Previous_Close_Return'] = df['close'] / df['close'].shift(1)
    df['Combined_Factor'] = df['High_Low_Open_Close_Combined'] * df['Close_to_Previous_Close_Return']
    
    # Volume-Adjusted Momentum
    df['Average_Volume_5_Days'] = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = (df['volume'] / df['Average_Volume_5_Days']) * (df['close'] / (df['high'] - df['low']))
    
    # Cumulative Volume-Adjusted Momentum
    df['Cumulative_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=5).sum()
    
    # Price Momentum
    df['20_Day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Momentum'] = df['close'] - df['20_Day_EMA']
    
    # Combined Signal
    df['Combined_Signal'] = df['Volume_Weighted_Intraday_Return'] + df['Short_Term_Price_Change'] + (df['high'] - df['low']) * df['volume']
    
    # Volume Trend Indicator
    df['10_Day_Volume_Change'] = df['volume'] - df['volume'].shift(10)
    df['Volume_Trend_Indicator'] = df['10_Day_Volume_Change'].apply(lambda x: 2 if x > 0 else -2)
    
    # Composite Factor
    df['Composite_Factor'] = (df['Combined_Factor'] * df['Price_Momentum']) + df['Combined_Signal']
    
    # Smooth the Composite Factor
    df['Smoothed_Composite_Factor'] = df['Composite_Factor'].rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Smoothed_Composite_Factor'] * df['Volume_Trend_Indicator']
    
    # Final Factor
    df['Average_Volume_10_Days'] = df['volume'].rolling(window=10).mean()
    df['Final_Factor'] = (df['volume'] / df['Average_Volume_10_Days']) * df['Final_Alpha_Factor']
    
    return df['Final_Factor'].dropna()
```
