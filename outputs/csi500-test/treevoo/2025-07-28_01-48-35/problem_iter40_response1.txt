```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['Volume-Weighted Intraday Return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    df['Short-Term Price Change'] = df['close'].diff(5)
    
    # Close to Intraday Range Ratio
    df['Intraday Range'] = df['high'] - df['low']
    df['Close to Intraday Range Ratio'] = df['close'] / df['Intraday Range']
    
    # Adjust for Opening Gap
    df['Opening Gap'] = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    df['Combined High-Low and Open-Close'] = df['Intraday Range'] + df['Opening Gap']
    
    # Measure Closing Behavior
    df['Close-to-Previous Close Return'] = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    df['Final Combined Factor'] = (df['Combined High-Low and Open-Close'] * df['Close-to-Previous Close Return']).clip(lower=0)
    
    # Volume-Adjusted Momentum
    df['Average Volume 5D'] = df['volume'].rolling(window=5).mean()
    df['Volume-Adjusted Momentum'] = df['volume'] / df['Average Volume 5D'] * df['Close to Intraday Range Ratio']
    
    # Cumulative Volume-Adjusted Momentum
    df['Cumulative Volume-Adjusted Momentum'] = df['Volume-Adjusted Momentum'].rolling(window=5).sum()
    
    # Price Momentum
    df['EMA 20D'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price Momentum'] = df['close'] - df['EMA 20D']
    
    # Combined Signal
    df['Combined Signal'] = df['Volume-Weighted Intraday Return'] + df['Short-Term Price Change'] + df['Cumulative Volume-Adjusted Momentum']
    
    # Composite Factor
    df['Composite Factor'] = (df['Final Combined Factor'] * df['Price Momentum'] + df['Combined Signal']).clip(lower=0)
    
    # Smooth the Composite Factor
    df['Smoothed Composite Factor'] = df['Composite Factor'].rolling(window=5).mean()
    
    # Volume Trend Indicator
    df['Volume Trend Score'] = np.where(df['volume'].diff(10) > 0, 2, -2)
    
    # Synthesize Final Alpha Factor
    df['Final Alpha Factor'] = df['Smoothed Composite Factor'] * df['Volume Trend Score']
    
    # Opening Impact and Gap Adjustment
    df['Open-to-Close Return'] = (df['close'] - df['open']) / df['open']
    df['Overnight Gap'] = df['open'] - df['close'].shift(1)
    
    # Final Factor
    df['Average Volume 10D'] = df['volume'].rolling(window=10).mean()
    df['Scaled Volume'] = df['volume'] / df['Average Volume 10D']
    
    df['Final Factor'] = (df['Final Alpha Factor'] - df['Open-to-Close Return'] - df['Overnight Gap']) * df['Scaled Volume']
    
    return df['Final Factor']
```
