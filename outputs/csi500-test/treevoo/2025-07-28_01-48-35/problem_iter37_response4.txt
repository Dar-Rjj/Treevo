```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Short-Term Price Change
    df['short_term_price_change'] = df['close'] - df['close'].shift(5)
    
    # Close to Intraday Range Ratio
    df['close_to_intraday_range_ratio'] = df['close'] / (df['high'] - df['low'])
    
    # Adjust for Opening Gap
    df['adjusted_opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    df['combined_high_low_open_close'] = (df['high'] - df['low']) + df['adjusted_opening_gap']
    
    # Measure Closing Behavior
    df['close_to_prev_close_return'] = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    df['final_combined_factor'] = df['combined_high_low_open_close'] * df['close_to_prev_close_return']
    df['final_combined_factor'] = df['final_combined_factor'].apply(lambda x: max(x, 0))
    
    # Volume Adjusted Momentum
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = (df['volume'] / avg_volume_5d) * df['close_to_intraday_range_ratio']
    
    # Volume-Weighted High-Low Spread
    df['volume_factor'] = df['volume'] / avg_volume_5d
    df['vol_weighted_high_low_spread'] = (df['high'] - df['low']) * df['volume_factor']
    
    # Price Momentum
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    df['price_momentum'] = df['close'] - ema_20d
    
    # Combined Signal
    df['combined_signal'] = df['vol_weighted_intraday_return'] + df['short_term_price_change'] + df['vol_weighted_high_low_spread']
    
    # Volume Trend Indicator
    volume_change_10d = df['volume'].diff(10)
    df['volume_trend_indicator'] = np.where(volume_change_10d > 0, 3, -3)
    
    # Composite Factor
    df['composite_factor'] = (df['final_combined_factor'] * df['price_momentum']) + df['combined_signal']
    df['composite_factor'] = df['composite_factor'].apply(lambda x: max(x, 0))
    
    # Smooth the Composite Factor
    df['smoothed_composite_factor'] = df['composite_factor'].rolling(window=7).mean()
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_composite_factor'] * df['volume_trend_indicator']
    
    return df['final_alpha_factor']
```
