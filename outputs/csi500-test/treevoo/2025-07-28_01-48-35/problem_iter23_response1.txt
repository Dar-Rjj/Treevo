```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    n_days = 5
    momentum_component = df['close'].pct_change().rolling(window=n_days).sum()
    momentum_component = momentum_component * close_to_range_ratio
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=n_days).mean()
    
    # Combine Initial Signals
    combined_signals = average_intraday_return + volume_adjusted_spread
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Opening Impact Factor
    open_to_close_return = df['close'] - df['open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (combined_signals * momentum_component * volume_trend_indicator * open_to_close_return 
                       / df['volume'].rolling(window=20).mean() * (volume_trend_indicator * relative_high_low_spread))
    
    return enhanced_factor
```
