import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike = (df['volume_change'] > df['volume_change'].abs().mean()).astype(int)
    
    # Integrate Price Trend
    price_trend = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine Components
    combined_components = volume_weighted_return + price_trend + volume_spike
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    average_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / average_volume_10d
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 if x > 1 else (-1.2 if x < -1 else 0))
    
    # Intermediate Composite Factor
    intermediate_factor = combined_components + volume_weighted_return * volume_trend_score + price_trend
    intermediate_factor -= (df['close'] - df['open']) / df['open']
    
    # Crossover Signals
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    crossover_signal = short_term_ma > long_term_ma
    
    # Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Volume Return
    average_volume_20d = df['volume'].rolling(window=20).mean()
    volume_return = df['volume'] / average_volume_20d
    
    # Open-Close Reversal
    open_close_reversal = (df['open'].shift(-1) - df['close']).abs() / df['close']
    
    # Close-to-Open Gap
    close_to_open_gap = df['close'] - df['open'].shift(-1)
    
    # Final Alpha Factor
    smoothed_intermediate_factor = intermediate_factor.rolling(window=5).mean()
    final_alpha_factor = smoothed_intermediate_factor + crossover_signal.astype(int) + daily_return + volume_return + open_close_reversal + close_to_open_gap
    
    return final_alpha_factor
