```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_intraday_ratio
    
    # Integrate Price Trend: Simple Moving Average of Close Price
    window = 20  # Define the window for SMA
    moving_avg = df['close'].rolling(window=window).mean()
    price_trend = df['close'] - moving_avg
    
    # Combine and Cumulate
    combined_factor = volume_weighted_momentum + price_trend
    cumulated_factor = combined_factor.rolling(window=window).sum()
    
    return cumulated_factor
```
