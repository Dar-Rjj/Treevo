```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Scale Volume by Last 5 Days' Average
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5d
    
    # Volume-Adjusted High-Low Spread
    vol_adj_high_low_spread = high_low_spread * scaled_volume
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    vol_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Price Momentum: 10-Day Exponential Moving Average (EMA) of Close Prices
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10d
    
    # Volume Trend: Current Volume divided by 10-Day Sum of Volume
    sum_volume_10d = df['volume'].rolling(window=10).sum()
    volume_trend = df['volume'] / sum_volume_10d
    
    # Intermediate Composite Factor
    intermediate_composite = (vol_adj_high_low_spread + vol_weighted_close_to_open) * price_momentum * volume_trend
    
    # Smoothed Composite Factor: 5-Day Simple Moving Average (SMA) of Intermediate Composite Factor
    smoothed_composite = intermediate_composite.rolling(window=5).mean()
    
    return smoothed_composite
```
