```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Momentum Component
    df['returns'] = df['close'].pct_change()
    df['momentum_component'] = df['returns'].rolling(window=N).sum() * df['close_to_range_ratio'].shift(1)
    
    # Adjust for Volume Activity
    df['volume_change_5d'] = df['volume'] - df['volume'].shift(5)
    df['average_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume_change_5d'] / df['average_volume_10d']
    df['volume_trend_score'] = df['volume_change_rate'].apply(lambda x: 2.0 if x > 2 else (-1.5 if x < -1 else 0.8))
    
    # Integrate Price Trend
    df['ema_close'] = df['close'].ewm(span=N, adjust=False).mean()
    df['price_trend'] = df['close'] - df['ema_close']
    
    # Combine Components
    df['combined_factor'] = df['momentum_component'] * df['volume_trend_score'] + df['price_trend']
    
    # Opening Impact and Gap Adjustment
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['overnight_gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['final_alpha_factor'] = df['combined_factor'] - df['open_to_close_return'] - df['overnight_gap']
    
    return df['final_alpha_factor']
```
