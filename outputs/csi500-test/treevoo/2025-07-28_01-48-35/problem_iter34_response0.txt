```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=N).sum()
    combined_momentum = close_to_range_ratio * momentum_component
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume_10d
    volume_trend_indicator = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Calculate Relative High-Low Spread
    high_low_spread = df['high'] - df['low']
    average_high_low_spread_10d = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / average_high_low_spread_10d.shift(1)
    
    # Enhance Combined Momentum, Volume, and Price Factor
    average_volume_20d = df['volume'].rolling(window=20).mean()
    combined_factor = (combined_momentum * volume_trend_indicator) / average_volume_20d
    combined_factor *= (volume_trend_indicator * relative_high_low_spread)
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = df['close'] - df['open']
    overnight_gap = df['open'] - df['close'].shift(1)
    opening_impact = open_to_close_return - overnight_gap
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + opening_impact
    
    return final_alpha_factor
```
