```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    df['short_term_price_change'] = df['close'].shift(-5) - df['close']
    
    # Calculate Volume Trend Indicator
    df['volume_trend_indicator'] = (df['volume'].diff(5) > 0).astype(int) * 2 - 1
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['volume_weighted_close_to_open_return'] = (df['close'] - df['open']) * df['volume']
    
    # Introduce New Factor: Average True Range (ATR)
    df['true_range'] = df.apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)[x.name]), 
                      abs(x['low'] - df['close'].shift(1)[x.name])),
        axis=1
    )
    df['atr_5_day'] = df['true_range'].rolling(window=5).mean()
    
    # Combine Signals
    df['combined_signal'] = (
        df['volume_weighted_intraday_return'] +
        df['short_term_price_change'] +
        df['high_low_spread'] +
        df['atr_5_day']
    )
    
    # Final Factor
    df['average_volume_5_day'] = df['volume'].rolling(window=5).mean()
    df['scaled_volume'] = df['volume'] / df['average_volume_5_day']
    df['final_factor'] = df['combined_signal'] * df['scaled_volume'] * df['volume_trend_indicator']
    
    return df['final_factor'].dropna()
```
