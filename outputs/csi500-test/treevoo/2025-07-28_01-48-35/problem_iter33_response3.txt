```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjusted Intraday Range with Volume Trend
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_volume_10d
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_score = np.where(volume_change_rate > 1, 2.0, -1.5)
    
    # Combined Momentum and Volume Factor
    n_days = 10
    returns_last_n_days = df['close'].pct_change().rolling(window=n_days).sum()
    combined_momentum_vol_factor = returns_last_n_days * close_to_range_ratio * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    opening_impact_adjustment = combined_momentum_vol_factor - open_to_close_return
    
    # Intermediate Composite Factor
    close_to_open_return = df['close'] - df['open']
    volume_weighted_close_to_open_return = close_to_open_return * scaled_volume
    price_momentum = df['close'] - df['close'].ewm(span=20).mean()
    intermediate_composite_factor = adjusted_intraday_range + volume_weighted_close_to_open_return * price_momentum * volume_trend_score
    
    # Price Momentum
    ema_20 = df['close'].ewm(span=20).mean()
    price_momentum = df['close'] - ema_20
    
    # Volume Trend
    volume_sum_20d = df['volume'].rolling(window=20).sum()
    volume_trend = df['volume'] / volume_sum_20d
    
    # Price Oscillator
    ema_7 = df['close'].ewm(span=7).mean()
    ema_14 = df['close'].ewm(span=14).mean()
    price_oscillator = ema_7 - ema_14
    
    # Integrated Composite Factor
    integrated_composite_factor = intermediate_composite_factor + price_oscillator
    
    # Smoothed Composite Factor
    smoothed_composite_factor = integrated_composite_factor.rolling(window=7).mean()
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor
    
    return final_alpha_factor
```
