```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume Adjusted Momentum
    average_volume_5d = df['volume'].rolling(window=5).mean()
    volume_factor = df['volume'] / average_volume_5d
    volume_adjusted_momentum = high_low_spread * volume_factor
    
    # Combine Factors
    factor_values = volume_adjusted_momentum + close_to_open_return
    
    return factor_values
```
