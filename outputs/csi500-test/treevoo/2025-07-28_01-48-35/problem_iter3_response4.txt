```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Integrate Price Trend
    moving_avg_window = 10  # Define the window size for the simple moving average
    sma_close = df['close'].rolling(window=moving_avg_window).mean()
    price_trend = df['close'] - sma_close
    
    # Combine Momentum and Trend
    combined_values = volume_weighted_momentum + price_trend
    
    # Cumulative Sum of Combined Values
    cumulative_sum_window = 5  # Define the rolling window size for cumulative sum
    cumulative_combined_values = combined_values.rolling(window=cumulative_sum_window).sum()
    
    return cumulative_combined_values
```
