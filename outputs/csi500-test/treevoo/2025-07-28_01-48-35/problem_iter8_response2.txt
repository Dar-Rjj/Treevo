```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day EMA of Close Prices
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract EMA from Current Close Price (Price Momentum)
    df['price_momentum'] = df['close'] - df['EMA_10']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Adjusted High-Low Range
    df['adjusted_high_low_range'] = (df['high'] - df['low']) - (df['open'] - df['close'].shift(1))
    
    # Combine High-Low and Open-Close
    df['combined_range'] = df['adjusted_high_low_range'] + df['intraday_return']
    
    # Measure Closing Behavior
    df['close_to_prev_close_return'] = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor (Ensure Positive Values by taking Absolute Value)
    df['final_combined_factor'] = df['combined_range'] * df['close_to_prev_close_return'].abs()
    
    # Calculate Spread
    df['spread'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    df['volume_adjusted_spread'] = df['spread'] * df['volume']
    
    # Calculate Average Intraday Return
    df['average_intraday_return'] = df['intraday_return'].rolling(window=10).mean()
    
    # Combine Initial Signals
    df['initial_signal'] = df['average_intraday_return'] + df['volume_adjusted_spread']
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] / df['volume'].rolling(window=20).sum()
    
    # Calculate Volume Trend Score
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume'] / avg_volume_10_days
    df['volume_trend_score'] = df['volume_change_rate'].apply(lambda x: 1.5 * x if x > 1 else -1.2 * x)
    
    # Calculate Price Momentum using High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    avg_high_low_spread_10_days = df['high_low_spread'].rolling(window=10).mean()
    df['relative_high_low_spread'] = df['high_low_spread'] / avg_high_low_spread_10_days.shift(1)
    
    # Calculate 10-Day Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr_10'] = df['true_range'].rolling(window=10).mean()
    df['range_expansion'] = df['true_range'] / df['atr_10']
    
    # Composite Factor
    df['composite_factor'] = df['price_momentum'] + df['volume_trend_score'] + df['range_expansion']
    
    # Combine with Final Combined Factor
    df['combined_factor'] = df['final_combined_factor'] * df['composite_factor']
    
    # Smooth the Combined Factor
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smoothed_combined_factor']
    
    return alpha_factor
```
