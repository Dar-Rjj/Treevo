```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high to low price
    high_to_low = df['high'] / df['low']
    
    # Calculate the ratio of amount to volume, which could be seen as an average price
    avg_price = df['amount'] / df['volume']
    
    # Construct a combined factor using the above calculated ratios
    factor_value = (close_to_open * 0.5) + (high_to_low * 0.3) + (avg_price * 0.2)
    
    return factor_value
```
