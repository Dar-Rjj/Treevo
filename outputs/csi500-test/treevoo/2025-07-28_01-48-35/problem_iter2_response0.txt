```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    initial_signals = average_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume
    volume_trend_score = (volume_change_rate > 1) * 1.5 * volume_change_rate + (volume_change_rate < 1) * -1.2 * volume_change_rate
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / average_high_low_spread
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = initial_signals * volume_trend_score * normalized_high_low_spread
    
    return final_alpha_factor
```
