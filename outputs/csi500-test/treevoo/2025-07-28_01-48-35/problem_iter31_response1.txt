```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    N = 5  # Number of days for momentum
    momentum_component = df['close'].pct_change().rolling(window=N).sum()
    combined_momentum = momentum_component * close_to_range_ratio
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    average_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = short_term_volume_change / average_volume_10d
    volume_trend_indicator = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (combined_momentum * volume_trend_indicator) / average_volume_10d
    relative_high_low_spread = volume_adjusted_spread / df['volume'].rolling(window=20).mean()
    final_factor = enhanced_factor * (volume_trend_indicator + relative_high_low_spread)
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = df['close'] - df['open']
    overnight_gap = df['open'] - df['close'].shift(1)
    opening_impact_factor = final_factor - open_to_close_return - overnight_gap
    
    return opening_impact_factor
```
