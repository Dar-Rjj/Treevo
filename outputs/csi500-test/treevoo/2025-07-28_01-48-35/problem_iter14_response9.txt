```python
import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel alpha factor based on the given market features.
    The factor aims to capture the momentum and volatility aspects of the stock, 
    which are believed to be predictive for future returns.

    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.

    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the range of the day as a measure of volatility
    daily_range = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = daily_return.diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors to create the final alpha factor
    alpha_factor = (daily_return * daily_range * rsi * avg_volume_5d)
    
    return alpha_factor
```
