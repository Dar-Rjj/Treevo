```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']

    # Calculate Volume Adjusted Spread
    high_low_spread = df['high'] - df['low']
    volume_adjusted_spread = high_low_spread * df['volume']

    # Calculate Average Intraday Return over N days
    df['average_intraday_return'] = df['intraday_return'].rolling(window=N).mean()

    # Calculate Close to Open Return
    close_to_open_return = df['close'] - df['open']

    # Combine Intraday Signals
    combined_signal = volume_adjusted_spread + close_to_open_return

    # Final IMVSCR Calculation
    df['IMVSCR'] = (df['average_intraday_return'] + combined_signal) / df['close']

    return df['IMVSCR']
```
