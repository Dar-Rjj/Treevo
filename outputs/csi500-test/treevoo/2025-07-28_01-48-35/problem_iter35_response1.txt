```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the range of high and low
    rel_pos = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the ratio of the amount traded to the volume
    amt_vol_ratio = df['amount'] / df['volume']
    
    # Calculate a smoothed version of the relative position
    rel_pos_smoothed = rel_pos.rolling(window=5).mean()
    
    # Calculate the change in the smoothed relative position
    rel_pos_change = rel_pos_smoothed.diff(1)
    
    # The final alpha factor is the product of the change in smoothed relative position and the amount-volume ratio
    alpha_factor = rel_pos_change * amt_vol_ratio
    
    return alpha_factor
```
