```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Momentum Component
    df['returns'] = df['close'].pct_change()
    df['momentum_component'] = df['returns'].rolling(window=N).sum() * df['close_to_range_ratio']
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    df['average_volume_10days'] = df['volume'].rolling(window=10).mean()
    df['volume_change_rate'] = df['volume_change'] / df['average_volume_10days']
    df['volume_trend_score'] = df['volume_change_rate'].apply(lambda x: 1.5 if x > 1 else (-1.2 if x < 1 else 0))
    
    # Calculate Volume-Weighted Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Volume-Adjusted High-Low Spread
    df['average_volume_15days'] = df['volume'].rolling(window=15).mean()
    df['scaled_volume'] = df['volume'] / df['average_volume_15days']
    df['volume_adjusted_high_low_spread'] = df['intraday_range'] * df['scaled_volume']
    
    # Final Combined Factor
    df['combined_factor'] = (df['momentum_component'] * df['volume_trend_score'] 
                             + df['volume_weighted_intraday_return'] 
                             + df['volume_adjusted_high_low_spread'])
    
    # Opening Impact Adjustment
    df['open_to_close_return'] = df['close'] - df['open']
    df['final_alpha_factor'] = df['combined_factor'] - df['open_to_close_return']
    
    return df['final_alpha_factor']
```
