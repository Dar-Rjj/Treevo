import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Weighted Adjusted Momentum
    volume_weighted_adjusted_momentum = df['volume'] * close_to_intraday_range_ratio
    
    # Detect Volume Spikes
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 2.0
    volume_spikes = (volume_change > volume_spike_threshold).astype(int)
    
    # Integrate Enhanced Price Momentum
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    enhanced_price_momentum = (df['close'] - ema_20_close) / (df['open'] - df['low'])
    
    # Calculate Volume-Weighted Price Trend
    volume_weighted_close = df['close'] * df['volume']
    volume_weighted_trend_20_sum = volume_weighted_close.rolling(window=20).sum()
    volume_20_sum = df['volume'].rolling(window=20).sum()
    volume_weighted_price_trend = volume_weighted_trend_20_sum / volume_20_sum
    
    # Combine and Cumulate
    combined_factor = volume_weighted_adjusted_momentum + enhanced_price_momentum + volume_weighted_price_trend
    cumulated_factor = combined_factor.rolling(window=10).sum()
    
    # Final Alpha Factor
    alpha_factor = cumulated_factor.ewm(span=10, adjust=False).mean()
    
    return alpha_factor
