```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = ((df['high'] - df['low']) / df['low']) * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend Indicator
    volume_trend_indicator = 2 * (df['volume'].rolling(window=10).sum().diff() > 0).astype(int) - 2
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = (df['close'] - df['open']) * df['volume']
    
    # Combine Signals
    combined_signal = (volume_weighted_intraday_return + 
                       short_term_price_change + 
                       high_low_spread)
    
    # Final Factor
    avg_10_days_volume = df['volume'].rolling(window=10).mean()
    scaled_volume = df['volume'] / avg_10_days_volume
    final_factor = combined_signal * (scaled_volume + volume_trend_indicator)
    
    return final_factor
```
