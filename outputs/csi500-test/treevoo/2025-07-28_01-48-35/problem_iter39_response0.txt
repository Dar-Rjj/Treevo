```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio).shift(1)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / average_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Calculate Volatility Component
    volatility = df['close'].rolling(window=10).std()
    average_close = df['close'].rolling(window=10).mean()
    normalized_volatility = volatility / average_close
    
    # Combine Components
    factor_value = (momentum_component * volume_trend_score + price_trend) * normalized_volatility
    
    # Opening Impact and Gap Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    overnight_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    factor_value -= open_to_close_return - overnight_gap
    
    return factor_value
```
