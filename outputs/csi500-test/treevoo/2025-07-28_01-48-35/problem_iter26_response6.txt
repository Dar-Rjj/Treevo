```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Momentum Component
    momentum_component = (df['close'].pct_change().rolling(window=n).sum()) * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10_days = df['volume'].rolling(window=m).mean()
    volume_change_rate = volume_change / avg_volume_10_days
    volume_trend_score = 1.5 if volume_change_rate > 1 else (-1.2 if volume_change_rate < 1 else 0)
    
    # Integrate Price Trend
    sma_5 = df['close'].rolling(window=n).mean()
    sma_10 = df['close'].rolling(window=m).mean()
    sma_20 = df['close'].rolling(window=p).mean()
    price_trend = df['close'] - sma_20
    
    # Combine Components
    combined_factor = momentum_component * volume_trend_score + price_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = combined_factor - open_to_close_return
    
    # Crossover Signals
    crossover_signal = (sma_5 > sma_20).astype(int) - (sma_5 < sma_20).astype(int)
    
    # Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Volume Return
    avg_volume_20_days = df['volume'].rolling(window=p).mean()
    volume_return = df['volume'] / avg_volume_20_days
    
    # Open-Close Reversal
    open_close_reversal = (df['open'].shift(-1) - df['close']).abs() > 0.02 * df['close']
    
    # Close-to-Open Gap
    close_to_open_gap = df['close'] - df['open'].shift(-1)
    
    # Average True Range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=m).mean()
    
    # Trend Strength
    trend_strength = (df['close'] - df['close'].shift(m)) / m
    
    # Final Alpha Factor
    final_alpha_factor += crossover_signal + daily_return + volume_return + open_close_reversal.astype(int) + close_to_open_gap + atr + trend_strength
    
    return final_alpha_factor
```
