```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price
    pct_change_avg_price = avg_price.pct_change()
    
    # Calculate the difference between close and open prices
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of volume to amount
    vol_to_amount_ratio = df['volume'] / df['amount']
    
    # Combine all the above indicators into a single factor
    factor = (pct_change_avg_price * close_open_diff * vol_to_amount_ratio).dropna()
    
    return factor
```
