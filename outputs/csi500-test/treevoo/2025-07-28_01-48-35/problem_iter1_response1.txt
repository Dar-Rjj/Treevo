```python
def heuristics_v2(data):
    # Calculate Short-Term Price Change
    short_term_price_change = data['close'].shift(5) - data['close']
    
    # Calculate Average Volume over 20 days ending at t-1
    average_volume_20 = data['volume'].rolling(window=20).mean().shift(1)
    
    # Calculate Short-Term Volume Change
    short_term_volume_change = data['volume'].shift(5) - data['volume']
    
    # Volume Trend Indicator
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Final VAPM Factor
    vapm_factor = (short_term_price_change * volume_trend_indicator) / average_volume_20
    
    return vapm_factor
```
