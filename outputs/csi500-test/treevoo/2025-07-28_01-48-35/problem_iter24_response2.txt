```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    returns = df['close'].pct_change()
    momentum_component = (returns.rolling(window=N).sum() * close_to_range_ratio)
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume
    volume_trend_score = volume_change_rate.apply(lambda x: 1.7 if x > 2 else (-1.5 if x < -2 else 0.8))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=N, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Combine Components
    combined_factor = (momentum_component * volume_trend_score) + price_trend
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = combined_factor - open_to_close_return
    
    return final_alpha_factor
```
