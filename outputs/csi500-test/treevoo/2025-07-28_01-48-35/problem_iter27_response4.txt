import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Momentum Component
    close_prices = df['close']
    momentum_component = close_prices.rolling(window=N).apply(lambda x: (x[-1] - x[0]) / x[0], raw=True)
    momentum_component *= close_to_range_ratio
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'].diff(periods=3)
    volume_trend_indicator = (short_term_volume_change > 0).astype(int) * 2 - 1
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=N).mean()
    
    # Combine Initial Signals
    initial_signals = average_intraday_return + volume_adjusted_spread
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    average_high_low_spread = high_low_spread.rolling(window=5).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Opening Impact Factor
    open_to_close_return = df['close'] - df['open']
    
    # Enhance Combined Momentum, Volume, and Price Factor
    enhanced_factor = (initial_signals * momentum_component * volume_trend_indicator * open_to_close_return
                       / df['volume'].rolling(window=10).mean() * (volume_trend_indicator * relative_high_low_spread))
    
    return enhanced_factor.dropna()
