```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine High-Low and Open-Close
    combined_hl_oc = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / df['close'].shift(1)
    
    # Final Combined Factor
    final_combined_factor = combined_hl_oc * close_to_prev_close_return
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['high'] - df['low']) / df['low'] * df['volume']
    
    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend Indicator
    volume_change_10d = df['volume'] - df['volume'].shift(10)
    volume_trend_score = np.where(volume_change_10d > 0, 2, -2)
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_open_return = df['close'] - df['open']
    volume_weighted_close_open_return = close_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10d
    
    # Composite Factor
    composite_factor = volume_weighted_close_open_return * price_momentum + final_combined_factor
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score
    
    # Adjust for Volume Activity
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change_5d / avg_volume_10d
    volume_trend_coefficient = np.where(volume_change_rate > 1, 2.0,
                                        np.where(volume_change_rate < -1, -1.5, 0))
    
    # Combine Components
    momentum_component = price_momentum * volume_trend_score
    price_trend = short_term_price_change
    final_alpha_factor_adjusted = (momentum_component +
                                   price_trend +
                                   volume_weighted_intraday_return +
                                   short_term_price_change +
                                   high_low_spread) * volume_trend_coefficient
    
    return final_alpha_factor_adjusted
```
