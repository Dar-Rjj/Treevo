import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Change
    price_change_5d = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Activity
    avg_volume_20d = df['volume'].rolling(window=20).mean().shift(1)
    volume_change_5d = df['volume'] - df['volume'].shift(5)
    
    # Volume Trend Indicator
    volume_trend_indicator = (volume_change_5d > 0).astype(int) * 2 - 1
    
    # Volume Trend Score
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / avg_volume_10d
    volume_trend_score = (volume_change_rate > 1) * 1.5 * volume_change_rate + (volume_change_rate < 1) * -1.2 * volume_change_rate
    
    # Final VAPM Factor
    vapm_factor = (price_change_5d * volume_trend_indicator) / avg_volume_20d
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_10d = high_low_spread.rolling(window=10).mean()
    normalized_high_low_spread = high_low_spread / avg_high_low_spread_10d
    
    # Combine Volume and Price Momentum
    final_alpha_factor = vapm_factor * normalized_high_low_spread
    
    return final_alpha_factor
