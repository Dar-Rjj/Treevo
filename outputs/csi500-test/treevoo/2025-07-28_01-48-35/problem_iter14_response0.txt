```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Spread
    df['spread'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    df['volume_adjusted_spread'] = df['spread'] * df['volume']
    
    # Calculate Momentum Component
    n_days = 10
    df['momentum_component'] = df['close'].pct_change().rolling(n_days).sum()
    df['momentum_close_to_range'] = df['close_to_range_ratio'] * df['momentum_component']
    
    # Adjust for Volume Activity
    avg_volume_20 = df['volume'].rolling(20).mean().shift(1)
    short_term_vol_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = short_term_vol_change.apply(lambda x: 1 if x > 0 else -1)
    avg_volume_10 = df['volume'].rolling(10).mean()
    volume_change_rate = df['volume'] / avg_volume_10
    volume_trend_score = (volume_change_rate > 1) * 1.5 + (volume_change_rate < 1) * -1.2
    
    # Calculate Average Intraday Return
    avg_intraday_return = df['intraday_return'].rolling(n_days).mean()
    
    # Combine Initial Signals
    combined_initial_signals = avg_intraday_return + df['volume_adjusted_spread']
    
    # Calculate Price Momentum
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_10 = high_low_spread.rolling(10).mean()
    relative_high_low_spread = high_low_spread / high_low_spread.shift(1)
    
    # Introduce Opening Impact Factor
    opening_impact_factor = df['open'] - df['close']
    adjusted_final_value = combined_initial_signals + opening_impact_factor
    
    # Enhance Combined Momentum, Volume, and Price Factor
    final_alpha_factor = (combined_initial_signals * df['momentum_close_to_range'] * volume_trend_score / avg_volume_20) * (volume_trend_score * relative_high_low_spread)
    
    return final_alpha_factor
```
