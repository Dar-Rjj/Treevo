```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given thought tree.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the daily price movement
    daily_movement = df['close'] - df['open']
    
    # Calculate the ratio of daily price movement to the daily range
    price_movement_ratio = daily_movement / daily_range
    
    # Calculate the ratio of the daily close to the high
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate the ratio of the daily close to the low
    close_to_low_ratio = df['close'] / df['low']
    
    # Calculate the average of the close-to-high and close-to-low ratios
    avg_close_ratio = (close_to_high_ratio + close_to_low_ratio) / 2
    
    # Calculate the final factor value
    factor_value = price_movement_ratio * avg_close_ratio * df['volume']
    
    return factor_value
```
