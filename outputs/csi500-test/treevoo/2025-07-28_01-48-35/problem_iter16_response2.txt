```python
def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_intraday_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * close_to_intraday_range_ratio
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].diff()
    volume_spike_indicator = df['volume_change'].abs() > df['volume_change'].rolling(window=10).std()
    
    # Integrate Price Trend
    simple_moving_average = df['close'].rolling(window=20).mean()
    price_trend = df['close'] - simple_moving_average
    
    # Combine Components
    combined_factor = (volume_adjusted_momentum 
                       + price_trend
                       + volume_spike_indicator.astype(int))
    
    final_alpha = combined_factor.rolling(window=5).sum()
    
    return final_alpha
```
