```python
import pandas as pd

def heuristics_v2(df, n_days=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=n_days).sum()
    momentum_component = momentum_component * close_to_range_ratio
    
    # Adjust for Volume Activity
    volume_change = df['volume'] - df['volume'].shift(5)
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change_rate = volume_change / avg_volume_10d
    volume_trend_indicator = volume_change_rate.apply(lambda x: 2.0 if x > 1 else (-1.5 if x < -1 else 0))
    
    # Integrate Price Trend
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_close
    
    # Calculate Relative High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_high_low_spread_10d = high_low_spread.rolling(window=10).mean()
    relative_high_low_spread = high_low_spread / avg_high_low_spread_10d.shift(1)
    
    # Introduce Opening Impact Factor
    open_close_return = df['close'] - df['open']
    overnight_gap = df['open'] - df['close'].shift(1)
    opening_impact_factor = open_close_return - overnight_gap
    
    # Enhance Combined Momentum, Volume, and Price Factor
    combined_momentum_price = momentum_component * volume_trend_indicator
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    enhanced_factor = combined_momentum_price / avg_volume_20d
    combined_volume_price_momentum = enhanced_factor * (volume_trend_indicator + relative_high_low_spread)
    
    # Add Volume Weighted Return
    volume_weighted_return = (intraday_return * df['volume']) / avg_volume_10d
    
    # Final Alpha Factor
    final_alpha_factor = (
        opening_impact_factor +
        volume_weighted_return +
        combined_volume_price_momentum
    )
    
    return final_alpha_factor
```
