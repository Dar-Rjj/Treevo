```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    close_to_range_ratio = df['close'] / intraday_range
    
    # Adjust for Opening Gap
    previous_close = df['close'].shift(1)
    opening_gap = df['open'] - previous_close
    
    # Combine High-Low and Open-Close
    combined_high_low_open_close = intraday_range + opening_gap
    
    # Measure Closing Behavior
    close_to_prev_close_return = df['close'] / previous_close
    final_combined_factor = combined_high_low_open_close * np.abs(close_to_prev_close_return)
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    volume_weighted_c_to_o_return = close_to_open_return * df['volume']
    
    # Calculate Price Momentum
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_10
    
    # Composite Factor
    composite_factor = volume_weighted_c_to_o_return * price_momentum + final_combined_factor
    rolling_sum_volume_weighted = volume_weighted_c_to_o_return.rolling(window=5).sum()
    composite_factor += rolling_sum_volume_weighted
    
    # Smooth the Composite Factor
    smoothed_composite_factor = composite_factor.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Spread
    spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted Spread
    volume_adjusted_spread = spread * df['volume']
    
    # Calculate Average Intraday Return
    average_intraday_return = intraday_return.rolling(window=10).mean()
    
    # Combine Initial Signals
    combined_initial_signals = average_intraday_return + volume_adjusted_spread
    
    # Calculate Volume Trend Score
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = df['volume'] / average_volume
    volume_trend_score = (volume_change_rate > 1.7) * 2.5 * (volume_change_rate - 1.7) + (volume_change_rate < 0.7) * -2.0 * (volume_change_rate - 0.7)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = smoothed_composite_factor * volume_trend_score + combined_initial_signals
    
    return final_alpha_factor
```
