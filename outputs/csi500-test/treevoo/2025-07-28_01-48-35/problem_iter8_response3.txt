```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Return
    intraday_range = df['high'] - df['low']
    intraday_return = intraday_range / df['low']
    volume_weighted_intraday_return = intraday_return * df['volume']

    # Calculate Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)

    # Adjust for Volume Activity
    avg_volume_20 = df['volume'].rolling(window=20).mean().shift(1)
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    volume_trend_indicator = np.where(short_term_volume_change > 0, 1, -1)

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']

    # Combine Intraday Signals with Price Momentum and High-Low Spread
    combined_signal = volume_weighted_intraday_return + short_term_price_change + high_low_spread

    # Volume-Weighted Factor
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    scaled_volume = df['volume'] / avg_volume_5
    volume_weighted_factor = combined_signal * scaled_volume

    # Final Factor Adjustment
    final_factor = volume_weighted_factor * volume_trend_indicator

    # Composite & Smoothing
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - ema_close
    composite_factor = final_factor * price_momentum * volume_trend_indicator

    # 5-Day Simple Moving Average (SMA) of Composite Factor
    sma_composite_factor = composite_factor.rolling(window=5).mean()

    return sma_composite_factor
```
