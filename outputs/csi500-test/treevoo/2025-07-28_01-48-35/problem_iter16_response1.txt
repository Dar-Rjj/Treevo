```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Close to Intraday Range Ratio
    df['close_to_intraday_range_ratio'] = df['close'] / df['intraday_range']
    
    # Calculate Volume Adjusted Momentum
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_scaled'] = df['volume'] / df['avg_vol_5d']
    df['volume_adjusted_momentum'] = df['volume_scaled'] * df['close_to_intraday_range_ratio']
    df['cumulative_volume_adjusted_momentum'] = df['volume_adjusted_momentum'].rolling(window=20).sum()
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['vol_adj_high_low_spread'] = df['high_low_spread'] * df['volume_scaled']
    
    # Calculate Price Momentum
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_10d']
    
    # Calculate Volume Trend
    df['vol_sum_20d'] = df['volume'].rolling(window=20).sum()
    df['volume_trend'] = df['volume'] / df['vol_sum_20d']
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = 0
    obv = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            obv += df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            obv -= df.loc[df.index[i], 'volume']
        df.loc[df.index[i], 'obv'] = obv
    
    # Composite Factor
    df['composite_factor'] = (df['cumulative_volume_adjusted_momentum'] 
                              + df['vol_adj_high_low_spread'] 
                              + df['price_momentum'] 
                              + df['volume_trend'] 
                              + df['obv'])
    
    # Smooth the Composite Factor
    df['sma_composite_5d'] = df['composite_factor'].rolling(window=5).mean()
    
    return df['sma_composite_5d']
```
