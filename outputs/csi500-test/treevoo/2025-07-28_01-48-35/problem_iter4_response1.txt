```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Price Momentum
    df['10_day_ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['10_day_ema_close']
    
    # Volume Trend
    df['10_day_sum_volume'] = df['volume'].rolling(window=10).sum()
    df['volume_trend'] = df['volume'] / df['10_day_sum_volume']
    
    # Composite & Smoothing
    df['composite_factor'] = df['price_momentum'] * df['volume_trend']
    df['smoothed_composite'] = df['composite_factor'].rolling(window=5).mean()
    
    return df['smoothed_composite']
```
