```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / intraday_range
    
    # Calculate Momentum Component
    momentum_component = df['close'].pct_change().rolling(window=N).sum()
    
    # Combine with Close-to-Range Ratio
    combined_momentum = close_to_range_ratio * momentum_component
    
    # Adjust for Volume Activity
    short_term_volume_change = df['volume'] - df['volume'].shift(5)
    average_volume = df['volume'].rolling(window=10).mean()
    volume_change_rate = short_term_volume_change / average_volume
    
    volume_trend_score = volume_change_rate.apply(lambda x: 1.5 * x if x > 1 else -1.2 * x if x < 1 else x)
    
    # Final Combined Momentum and Volume Factor
    final_factor = combined_momentum * volume_trend_score
    
    # Opening Impact Adjustment
    open_to_close_return = (df['close'] - df['open']) / df['open']
    final_alpha_factor = final_factor - open_to_close_return
    
    return final_alpha_factor
```
