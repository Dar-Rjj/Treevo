```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Adjusted Close-to-Open Return
    df['adj_clo_to_ope_return'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Calculate Rate of Change in Closing Price
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['crossover_5_20'] = df['sma_5'] > df['sma_20']
    df['crossover_10_50'] = df['sma_10'] > df['sma_50']
    
    # Examine Relationships Between Different Price Points
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Assess Volatility-Adjusted Returns
    df['volatility'] = df['close'].rolling(window=20).std()
    df['vol_adj_return'] = df['close'].pct_change() / df['volatility']
    
    # Measure Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    
    # Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['midpoint'] / df['open']
    
    # Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Close-to-Next Day Open Return
    df['close_to_next_open_return'] = df['vwap'].shift(-1) - df['open']
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_of_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=20).max()
    df['min_open'] = df['open'].rolling(window=20).min()
    
    # Adjusted Close-to-Open Envelope
    df['adj_max_close'] = df['close'].rolling(window=20).max()
    df['adj_min_open'] = df['open'].rolling(window=20).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()
    
    # Integrate Market Sentiment
    df['open_close_diff'] = df['close'] - df['open']
    df['positive_negative_diff'] = (df['open_close_diff'] > 0).astype(int)
    
    # Combine Factors
    df['combined_envelopes'] = (df['max_close'] - df['min_open']) * (df['adj_max_close'] - df['adj_min_open'])
    df['factor'] = (df['midpoint_to_open_ratio'] * df['weighted_sum_of_returns']) / df['combined_envelopes']
    df['factor'] += df['smoothed_volatility'] * df['positive_negative_diff']
    
    # Final Factor Adjustment
    df['final_factor'] = df['factor'] + df['adj_clo_to_ope_return']
    
    # Compute Cumulative Price Momentum
    df['ema_daily_price_change'] = df['daily_price_change'].ewm(span=20, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    return df['final_factor']
```
