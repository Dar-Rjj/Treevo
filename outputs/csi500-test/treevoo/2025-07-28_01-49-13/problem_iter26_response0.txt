```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
        
    Returns:
        pd.Series: A series of the generated alpha factor indexed by date.
    """
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    
    # Heuristic: High price range, increasing close price, and increasing volume indicate positive momentum
    factor = (price_range * close_change * volume_change).fillna(0)
    
    return factor
```
