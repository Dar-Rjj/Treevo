```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    cumulative_price_momentum = daily_price_momentum.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Metrics
    weighted_high_low_range = high_low_range * volume_momentum
    adjusted_metric = weighted_high_low_range + close_price_change
    
    # Combine Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Final Combination
    final_alpha = adjusted_metric + combined_cumulative_sums
    
    return final_alpha
```
