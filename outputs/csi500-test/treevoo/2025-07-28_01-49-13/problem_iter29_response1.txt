```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = midpoint / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_day_open_return = vwap.pct_change().shift(-1).fillna(0)
    
    # Weighted Sum of Returns by Volume
    weighted_sum_of_returns = close_to_next_day_open_return * df['volume']
    weighted_sum_of_returns = weighted_sum_of_returns.rolling(window=20).sum()
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] * df['volume'] / df['volume'].rolling(window=20).mean()
    max_adjusted_close = adjusted_close.rolling(window=20).max()
    min_adjusted_open = df['open'].rolling(window=20).min()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']).rolling(window=20).std()
    smoothed_volatility = daily_volatility.ewm(span=10).mean()
    
    # Combine Factors
    combined_envelopes = (max_close - min_open) * (max_adjusted_close - min_adjusted_open)
    combined_factor = (midpoint_to_open_ratio * weighted_sum_of_returns) / (combined_envelopes + smoothed_volatility)
    
    return combined_factor
```
