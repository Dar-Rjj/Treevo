```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] / df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Weight Intraday Range by Volume Momentum
    weighted_intraday_range = intraday_range * volume_momentum
    
    # Combine Metrics
    combined_metrics = weighted_intraday_range + adjusted_close_to_open_return
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_to_open_ratio = (df['high'] + df['low']) / 2 / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_day_open_return = df['close'] - df['open'].shift(-1)
    volume_weighted_momentum = (close_to_next_day_open_return * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.ewm(span=20, adjust=False).mean()
    
    # Compute Cumulative Volume Momentum
    cumulative_volume_momentum = volume_momentum.ewm(span=20, adjust=False).mean()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Momentum Metrics
    combined_momentum_metrics = cumulative_price_momentum * cumulative_volume_momentum
    
    # Combine Factors
    final_factor = (
        intraday_midpoint_to_open_ratio * volume_weighted_momentum * 
        combined_momentum_metrics * 
        (cumulative_price_change * cumulative_volume_change) * 
        combined_metrics
    )
    
    return final_factor.dropna()
```
