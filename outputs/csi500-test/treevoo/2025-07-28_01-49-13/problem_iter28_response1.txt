```python
import pandas as pd

def heuristics_v2(df):
    # Short-term SMA (5-day) and long-term SMA (20-day)
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    
    # SMA Crossover
    sma_crossover = (short_sma > long_sma).astype(int) - (short_sma < long_sma).astype(int)
    
    # Rate of Change (ROC) for close prices over 14 days
    roc = df['close'].pct_change(periods=14)
    
    # True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Average True Range (ATR) over 14 days
    atr = true_range.rolling(window=14).mean()
    
    # On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Chaikin Money Flow (CMF) over 20 days
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Support and Resistance Levels
    recent_high = df['high'].rolling(window=20).max()
    recent_low = df['low'].rolling(window=20).min()
    distance_to_support = df['close'] - recent_low
    distance_to_resistance = recent_high - df['close']
    
    # Gap Analysis
    positive_gap = (df['open'] > df['high'].shift(1)).astype(int)
    negative_gap = (df['open'] < df['low'].shift(1)).astype(int)
    
    # Intraday Price Movement
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_range = df['high'] - df['low']
    
    # Trend Following with DMI
    high_low_diff = df['high'] - df['low']
    high_close_diff = df['high'] - df['close'].shift(1)
    low_close_diff = df['low'] - df['close'].shift(1)
    pos_dm = (high_low_diff + high_close_diff).apply(lambda x: max(x, 0))
    neg_dm = (high_low_diff - low_close_diff).apply(lambda x: max(x, 0))
    di_plus = 100 * (pos_dm.rolling(window=14).sum() / true_range.rolling(window=14).sum())
    di_minus = 100 * (neg_dm.rolling(window=14).sum() / true_range.rolling(window=14).sum())
    dmi = di_plus - di_minus
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        sma_crossover + 
        roc + 
        atr + 
        obv + 
        cmf + 
        distance_to_support + 
        distance_to_resistance + 
        positive_gap + 
        negative_gap + 
        intraday_return + 
        intraday_range + 
        dmi
    )
    
    return alpha_factor
```
