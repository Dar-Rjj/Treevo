```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the close and open price
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate the difference between the high and low price
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of the close-open difference to the high-low difference
    df['price_range_ratio'] = df['close_open_diff'] / df['high_low_diff']
    
    # Calculate the ratio of the volume to the amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Heuristic alpha factor: multiply the price range ratio by the volume-amount ratio
    alpha_factor = df['price_range_ratio'] * df['volume_amount_ratio']
    
    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
```
