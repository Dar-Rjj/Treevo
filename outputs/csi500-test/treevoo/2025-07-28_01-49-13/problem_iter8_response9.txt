```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Open VWAP Momentum
    df['vwap_returns'] = df['vwap'].pct_change()
    df['close_to_open_vwap_momentum'] = df['vwap_returns'].rolling(window=2).sum()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['intraday_range'] * df['close_to_open_return']
    
    # Calculate Volume-Weighted Volatility
    df['volume_weighted_volatility'] = (df['intraday_range'] / df['volume']) * (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Combine Factors
    df['factor'] = (df['intraday_range_to_open_ratio'] * df['close_to_open_vwap_momentum'] * 
                    df['adjusted_close_to_open_return'] * df['volume_weighted_volatility'] * df['volume_change'])
    
    return df['factor'].dropna()
```
