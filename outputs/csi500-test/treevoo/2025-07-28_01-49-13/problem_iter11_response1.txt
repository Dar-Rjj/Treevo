import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Spread
    df['Volume_Weighted_HL'] = (df['high'] - df['low']) * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    ema_window = 10
    df['Volume_Weighted_EMA'] = df['Volume_Weighted_HL'].ewm(span=ema_window).mean()
    
    # Subtract Previous EMA from 5 days ago
    df['Volume_Weighted_EMA_Diff'] = df['Volume_Weighted_EMA'] - df['Volume_Weighted_EMA'].shift(5)
    
    # Identify Volume Spike
    volume_moving_avg_days = 30
    df['Volume_MA'] = df['volume'].rolling(window=volume_moving_avg_days).mean()
    df['Volume_STD'] = df['volume'].rolling(window=volume_moving_avg_days).std()
    df['Volume_Spike'] = (df['volume'] > (df['Volume_MA'] + 2 * df['Volume_STD']))
    
    # Calculate N-day Price Momentum
    n_days = 10
    df['N_day_Return'] = (df['close'] / df['close'].shift(n_days)) - 1
    
    # Calculate Intraday High-Low Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_Open_Return'] = (df['close'] / df['open']) - 1
    
    # High-Low Range Adjustment Factor and Close-Open Return Adjustment Factor
    df['High_Low_Range_Factor'] = df['Intraday_Range'] / df['close']
    df['Close_Open_Return_Factor'] = df['Close_Open_Return']
    
    # Combine Price Momentum, Volume Spike, and Intraday Volatility
    K = 1.5  # Amplification factor for volume spike
    df['Adjusted_Return'] = np.where(
        df['Volume_Spike'],
        df['N_day_Return'] * K,
        df['N_day_Return']
    )
    df['Adjusted_Return'] *= df['High_Low_Range_Factor'] * df['Close_Open_Return_Factor']
    
    # Compute Cumulative Sum of Price Momentum
    df['Cumulative_Price_Momentum'] = df['N_day_Return'].cumsum()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Volume Change
    df['Cumulative_Volume_Change'] = df['Daily_Volume_Change'].cumsum()
    
    # Combine Cumulative Sums
    df['Combined_Cumulative'] = df['Cumulative_Price_Momentum'] * df['Cumulative_Volume_Change']
    
    # Calculate Volume Weighted Adjusted Return
    df['Volume_Weighted_Adjusted_Return'] = df['volume'] * df['Adjusted_Return']
    
    # Calculate Moving Average of Volume Weighted Adjusted Return
    lookback_period = 20
    df['Moving_Avg_Adjusted_Return'] = df['Volume_Weighted_Adjusted_Return'].rolling(window=lookback_period).mean()
    
    # Calculate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Volume_Weighted_EMA_Diff'] * df['Volume_Weighted_Adjusted_Return'] / df['Moving_Avg_Adjusted_Return']
    
    return df['Final_Alpha_Factor']
