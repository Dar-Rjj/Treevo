```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Daily Momentum
    df['daily_momentum'] = df['high'] - df['low']
    
    # Combine Volume Change and Daily Momentum
    df['factor_value'] = df['volume_change'] * df['daily_momentum']
    
    return df['factor_value']
```
