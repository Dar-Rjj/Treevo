import pandas as pd

def heuristics_v2(df):
    # Evaluate Intraday Dynamics
    intraday_range = df['high'] - df['low']
    intraday_midpoint_to_open_ratio = ((df['high'] + df['low']) / 2) / df['open']
    adjusted_close_to_open_return = (df['close'] - df['open']) / intraday_range
    
    # Analyze Momentum
    rate_of_change_10d = df['close'].pct_change(10)
    rate_of_change_30d = df['close'].pct_change(30)
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    
    # Assess Volatility
    std_30d = df['close'].rolling(window=30).std()
    volatility_smoothed = std_30d.ewm(span=30, adjust=False).mean()
    
    # Combine Metrics
    daily_price_change = df['close'].diff()
    cumulative_price_change = daily_price_change.cumsum()
    vwap = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    adjusted_close_to_open_envelope = df['close'].rolling(window=10).max() - df['open'].rolling(window=10).min()
    midpoint_to_open_ratio_vwap = intraday_midpoint_to_open_ratio * vwap
    
    # Integrate Volume Analysis
    daily_volume_change = df['volume'].diff()
    vma_10 = df['volume'].rolling(window=10).mean()
    vma_30 = df['volume'].rolling(window=30).mean()
    volume_trend = (df['volume'] - vma_10) / vma_10
    
    # Final Factor Adjustment
    weighted_intraday_range = intraday_range * vwap
    adjusted_close_to_open_weighted = adjusted_close_to_open_return * weighted_intraday_range
    combined_intraday_factor = adjusted_close_to_open_weighted + midpoint_to_open_ratio_vwap
    cumulative_momentum = cumulative_price_change.ewm(span=10, adjust=False).mean()
    final_factor = combined_intraday_factor + cumulative_momentum
    
    return final_factor
