```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-Open Return
    combined_return = (intraday_range + close_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_return = combined_return * df['volume']
    
    # Calculate Price Change Momentum
    current_day_mid = (df['high'] + df['low']) / 2
    previous_day_mid = (df['high'].shift(1) + df['low'].shift(1)) / 2
    price_change_momentum = (current_day_mid - previous_day_mid) * intraday_range
    
    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = df['volume'].rolling(window=5).std() * 2
    volume_spike_indicator = (daily_volume_change > volume_spike_threshold).astype(int)
    
    # Final Factor Calculation
    final_factor = volume_weighted_return * price_change_momentum * volume_spike_indicator
    
    return final_factor
```
