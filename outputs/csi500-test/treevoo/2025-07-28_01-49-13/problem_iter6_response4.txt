```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']

    # Calculate VWAP Returns
    vwap_returns = vwap.pct_change()

    # Calculate Close-to-Open VWAP Momentum
    close_to_open_vwap_momentum = vwap_returns.rolling(window=2).sum()

    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=5).max()  # Adjust window size as needed
    min_open = df['open'].rolling(window=5).min()  # Adjust window size as needed

    # Combine Factors
    factor = intraday_range_to_open_ratio * close_to_open_vwap_momentum
    factor = factor * (max_close - min_open)

    return factor
```
