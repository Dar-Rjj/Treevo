```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate Price Momentum
    df['close_t'] = df['close']
    df['close_N_days_ago'] = df['close'].shift(N)
    df['N_day_return'] = (df['close_t'] / df['close_N_days_ago']) - 1
    
    # Identify Volume Spike
    df['volume_t'] = df['volume']
    df['vol_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['vol_std_dev'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = df['volume_t'] > (df['vol_moving_avg'] + 2 * df['vol_std_dev'])
    
    # Combine Price Momentum and Volume Spike
    df['amplified_return'] = df['N_day_return'] * K
    df['factor_value'] = df['N_day_return']
    df.loc[df['volume_spike'], 'factor_value'] = df['amplified_return']
    
    return df['factor_value']
```
