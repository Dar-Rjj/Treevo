import pandas as pd

def heuristics_v2(data):
    # Calculate Volume-Weighted High-Low Spread
    data['HL_Vol_Weighted'] = (data['high'] - data['low']) * data['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    ema_window = 10
    data['HL_Vol_EMA'] = data['HL_Vol_Weighted'].ewm(span=ema_window, adjust=False).mean()
    
    # Subtract Previous EMA (5 days ago)
    data['HL_Vol_EMA_Diff'] = data['HL_Vol_EMA'] - data['HL_Vol_EMA'].shift(5)
    
    # Simple Moving Average (SMA) of Close Prices
    sma_window = 20
    data['close_SMA'] = data['close'].rolling(window=sma_window).mean()
    
    # Calculate Price Momentum
    data['price_momentum'] = data['close'] - data['close_SMA']
    
    # Final Alpha Factor
    data['alpha_factor'] = data['price_momentum'] * data['HL_Vol_EMA_Diff']
    
    return data['alpha_factor'].dropna()
