```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow by multiplying the typical price with the volume
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flows
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the factor as a weighted sum of the money flow index and the high-low range
    factor = 0.6 * money_flow_index + 0.4 * high_low_range
    
    return factor
```
