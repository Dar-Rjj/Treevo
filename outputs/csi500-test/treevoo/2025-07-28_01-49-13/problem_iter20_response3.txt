```python
def heuristics_v2(df):
    # Calculate the body difference between close and open price
    body_diff = df['close'] - df['open']
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the amount traded relative to the daily range
    amount_relative_to_range = df['amount'] / daily_range
    
    # Combine the features into a single alpha factor
    alpha_factor = (body_diff / daily_range) * volume_change * amount_relative_to_range
    
    return alpha_factor
```
