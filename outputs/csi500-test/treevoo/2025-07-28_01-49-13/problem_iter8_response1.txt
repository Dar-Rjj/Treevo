```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5, ema_window=10):
    # Calculate Daily Price Momentum
    df['daily_momentum'] = df['close'].diff()

    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'] - df['close'].shift(N)

    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=M).mean()
    df['volume_std'] = df['volume'].rolling(window=M).std()
    df['is_volume_spike'] = (df['volume'] > (df['volume_ma'] + 2 * df['volume_std'])).astype(int)

    # Combine Price Momentum and Volume Spike
    df['adjusted_N_day_return'] = df['N_day_return'] * (K * df['is_volume_spike'] + 1 * (1 - df['is_volume_spike']))

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_open_return'] = df['close'] / df['open'] - 1

    # Combine Intraday High-Low Range and Close-Open Return
    df['combined_return'] = (df['intraday_range'] * df['close_open_return'])

    # Compute Cumulative Sum of Price Momentum
    df['cumulative_momentum'] = df['daily_momentum'].cumsum()

    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()

    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()

    # Combine Cumulative Sums
    df['combined_cumulative'] = df['cumulative_momentum'] * df['cumulative_volume_change']

    # Calculate Volume-Weighted Adjusted Return
    df['weighted_adjusted_return'] = df['volume'] * df['adjusted_N_day_return']

    # Calculate Moving Average of Volume-Weighted Adjusted Return
    df['ma_weighted_return'] = df['weighted_adjusted_return'].rolling(window=N).mean()

    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['weighted_adjusted_return'] / df['ma_weighted_return']

    # Calculate Exponential Moving Average (EMA) of Weighted High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['weighted_high_low_spread'] = df['high_low_spread'] * df['volume']
    df['ema_weighted_high_low_spread'] = df['weighted_high_low_spread'].ewm(span=ema_window, adjust=False).mean()

    return df['alpha_factor'], df['ema_weighted_high_low_spread']
```
