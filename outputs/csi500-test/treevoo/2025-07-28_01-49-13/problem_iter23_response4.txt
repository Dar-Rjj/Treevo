```python
def heuristics_v2(df):
    # Calculate the average price as (open + high + low + close) / 4
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the total trade value as amount * volume
    trade_value = df['amount'] * df['volume']
    
    # Calculate the weighted price as total trade value / volume
    weighted_price = trade_value / df['volume']
    
    # Calculate the relative weighted price difference from the average price
    rel_weighted_price_diff = (weighted_price - avg_price) / avg_price
    
    # Calculate the factor as the cumulative sum of the relative weighted price differences
    factor = rel_weighted_price_diff.cumsum()
    
    return factor
```
