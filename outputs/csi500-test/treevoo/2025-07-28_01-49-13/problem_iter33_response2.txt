```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_mid_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_open_return = (df['open'].shift(-1) - vwap) / vwap
    
    # Weighted Sum of Returns by Volume
    weighted_sum_returns = close_to_next_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low'])
    smoothed_volatility = daily_volatility.ewm(span=5).mean()
    
    # Combine Factors
    combined_factors = (intraday_mid_to_open_ratio * weighted_sum_returns) * (max_adj_close / min_open) * (smoothed_volatility / daily_volatility)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_next_open_return / intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * weighted_sum_returns
    adjusted_metric = weighted_intraday_range + adjusted_close_to_open_return
    
    # Final Factor Adjustment
    final_factor = combined_factors + adjusted_metric + cumulative_price_change
    
    # Compute Cumulative Price Momentum
    cumulative_momentum = daily_price_change.ewm(span=5).mean()
    
    # Incorporate Short-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Synthesize Composite Alpha Factor
    composite_alpha_factor = final_factor + cumulative_momentum + (short_term_ema - long_term_ema)
    
    # Introduce Trend Reversal Indicator
    trend_reversal_indicator = short_term_ema - long_term_ema
    composite_alpha_factor += trend_reversal_indicator
    
    return composite_alpha_factor
```
