```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=2, lookback=20):
    # Calculate Daily Price Momentum
    df['Daily_Price_Momentum'] = df['close'].diff()
    
    # Calculate N-day Price Momentum
    df['N_day_Return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['Volume_MA'] = df['volume'].rolling(window=M).mean()
    df['Volume_Std'] = df['volume'].rolling(window=M).std()
    df['Volume_Spike'] = (df['volume'] > (df['Volume_MA'] + 2 * df['Volume_Std'])).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['Combined_N_day_Return'] = np.where(df['Volume_Spike'] == 1, df['N_day_Return'] * K, df['N_day_Return'])
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday High-Low Range
    df['Adjusted_Close_to_Open_Return'] = df['Combined_N_day_Return'] * df['Intraday_High_Low_Range'] * df['Close_to_Open_Return']
    
    # Calculate Volume Momentum
    df['Volume_Momentum'] = df['volume'] - df['volume'].shift(1)
    
    # Weight Intraday High-Low Range by Volume Momentum
    df['Weighted_Intraday_High_Low_Range'] = df['Intraday_High_Low_Range'] * df['Volume_Momentum']
    
    # Combine Metrics
    df['Combined_Metrics'] = df['Adjusted_Close_to_Open_Return'] + df['Weighted_Intraday_High_Low_Range']
    
    # Compute Cumulative Sum of Price Momentum
    df['Cumulative_Price_Momentum'] = df['Daily_Price_Momentum'].cumsum()
    
    # Calculate Volume Weighted Adjusted Return
    df['Volume_Weighted_Adjusted_Return'] = df['volume'] * df['Combined_Metrics']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['EMA_Volume_Weighted_Adjusted_Return'] = df['Volume_Weighted_Adjusted_Return'].ewm(span=lookback, adjust=False).mean()
    
    # Final Combination
    df['Cumulative_Volume_Change'] = df['Volume_Momentum'].cumsum()
    df['Final_Combined_Factor'] = (df['Cumulative_Price_Momentum'] * df['Cumulative_Volume_Change']) + \
                                  (df['Volume_Weighted_Adjusted_Return'] / df['EMA_Volume_Weighted_Adjusted_Return'])
    
    return df['Final_Combined_Factor']
```
