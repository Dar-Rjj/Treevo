```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20, k=2.0, ema_window=10, lookback=20):
    # Calculate Daily Price Momentum
    df['Daily_Price_Momentum'] = df['close'].diff()
    
    # Calculate N-day Price Momentum
    df['N_day_Return'] = df['close'].pct_change(periods=n)
    
    # Identify Volume Spike
    df['Volume_Spike'] = (df['volume'] > (df['volume'].rolling(window=m).mean() + 2 * df['volume'].rolling(window=m).std()))
    
    # Combine Price Momentum and Volume Spike
    df['Enhanced_N_day_Return'] = df['N_day_Return'] * np.where(df['Volume_Spike'], k, 1)
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['Adjusted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * (df['Intraday_Range'])
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=lookback).mean()
    df['Volume_Weighted_Volatility'] = df['Intraday_Range'] / avg_volume
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    df['Combined_Adjusted_Return'] = df['Adjusted_Close_to_Open_Return'] * df['Volume_Weighted_Volatility']
    
    # Calculate Volume-Weighted High-Low Spread
    df['Weighted_High_Low_Spread'] = (df['high'] - df['low']) * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    df['EMA_Weighted_High_Low_Spread'] = df['Weighted_High_Low_Spread'].ewm(span=ema_window).mean()
    df['EMA_Diff'] = df['EMA_Weighted_High_Low_Spread'] - df['EMA_Weighted_High_Low_Spread'].shift(5)
    
    # Combine Metrics
    df['Combined_Metrics'] = df['Adjusted_Close_to_Open_Return'] + df['EMA_Diff']
    
    # Compute Cumulative Sum of Price Momentum
    df['Cumulative_Price_Momentum'] = df['Daily_Price_Momentum'].cumsum()
    
    # Calculate Volume Momentum
    df['Volume_Momentum'] = df['volume'].diff()
    
    # Weight Intraday High-Low Range by Volume Momentum
    df['Weighted_Intraday_Range'] = df['Intraday_Range'] * df['Volume_Momentum']
    
    # Calculate Volume-Weighted Adjusted Return
    df['Volume_Weighted_Adjusted_Return'] = df['Combined_Adjusted_Return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Adjusted Return
    df['EMA_Volume_Weighted_Adjusted_Return'] = df['Volume_Weighted_Adjusted_Return'].ewm(span=lookback).mean()
    
    # Final Combination
    df['Final_Combination'] = (
        df['Cumulative_Price_Momentum'] * df['Volume_Momentum'] +
        df['Combined_Metrics'] +
        df['EMA_Weighted_High_Low_Spread'] +
        df['EMA_Volume_Weighted_Adjusted_Return']
    )
    
    return df['Final_Combination']
```
