```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_day_open_return = vwap.shift(-1) / df['open'].shift(-1) - 1
    
    # Weighted Sum of Returns by Volume
    weighted_sum_of_returns = close_to_next_day_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']) / df['close'].shift(1)
    smoothed_volatility = daily_volatility.ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    combined_factors = (intraday_midpoint_to_open_ratio * weighted_sum_of_returns) * (max_adjusted_close / min_open) * (daily_volatility / smoothed_volatility)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_next_day_open_return / intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * weighted_sum_of_returns
    adjusted_combined_metrics = weighted_intraday_range + adjusted_close_to_open_return
    
    # Final Factor Adjustment
    final_factor = combined_factors * adjusted_combined_metrics
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.ewm(span=20, adjust=False).mean()
    
    # Compute Cumulative Volume Momentum
    cumulative_volume_momentum = daily_volume_change.ewm(span=20, adjust=False).mean()
    
    # Final Combination
    combined_momentum_metrics = cumulative_price_momentum * cumulative_volume_momentum
    final_alpha_factor = final_factor + combined_momentum_metrics
    
    return final_alpha_factor
```
