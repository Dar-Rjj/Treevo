```python
def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Volume Momentum
    volume_momentum = df['volume'].diff().fillna(0)

    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range

    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + adjusted_close_to_open_return

    # Calculate Daily Price Change
    daily_price_change = df['close'].diff().fillna(0)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()

    # Compute Cumulative Volume Change
    cumulative_volume_change = volume_momentum.cumsum()
    
    # Final Combination
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    final_factor = combined_cumulative_sums + combined_metrics
    
    return final_factor
```
