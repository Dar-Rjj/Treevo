```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the ratio of close to open price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of volume to amount
    vol_amount_ratio = df['volume'] / df['amount']
    
    # Combine the factors with weights
    factor_value = (0.4 * close_change + 0.3 * high_low_ratio + 
                    0.2 * close_open_ratio + 0.1 * vol_amount_ratio)
    
    return factor_value
```
