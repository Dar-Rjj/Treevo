```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = df['vwap'] / df['next_day_open'] - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    window = 20
    df['max_close'] = df['close'].rolling(window=window).max()
    df['min_open'] = df['open'].rolling(window=window).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['adjusted_close'] = df['close']
    df['max_adjusted_close'] = df['adjusted_close'].rolling(window=window).max()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=5, adjust=False).mean()
    
    # Combine Factors
    df['intraday_momentum_factor'] = df['intraday_midpoint_to_open_ratio'] * df['weighted_sum_returns']
    df['combined_envelope'] = (df['max_close'] + df['min_open'] + df['max_adjusted_close'] + df['min_open']) / 4
    df['adjusted_intraday_momentum'] = df['intraday_momentum_factor'] / df['combined_envelope'] * df['smoothed_volatility']
    
    # Calculate Short-Term Momentum
    short_term_window = 5
    df['avg_close_5d'] = df['close'].rolling(window=short_term_window).mean()
    df['stdev_close_5d'] = df['close'].rolling(window=short_term_window).std()
    df['short_term_momentum'] = (df['close'] - df['avg_close_5d']) / df['stdev_close_5d']
    
    # Incorporate Trading Volume
    df['avg_volume_5d'] = df['volume'].rolling(window=short_term_window).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    df['vol_adj_short_term_momentum'] = df['short_term_momentum'] * df['volume_ratio']
    
    # Consider Directional Movement
    df['directional_movement'] = 0
    bullish_condition = (df['high'] > df['high'].shift(1)) & (df['low'] > df['low'].shift(1))
    df.loc[bullish_condition, 'directional_movement'] = 1
    
    # Final Factor Combination
    df['final_factor'] = (
        df['adjusted_intraday_momentum'] +
        df['vol_adj_short_term_momentum'] +
        df['directional_movement']
    )
    
    return df['final_factor']
```
