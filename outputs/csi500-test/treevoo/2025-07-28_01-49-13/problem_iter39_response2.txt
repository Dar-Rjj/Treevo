import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change in Closing Price
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)

    # Evaluate Moving Average Crossovers
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['ma_crossover'] = df['sma_5'] > df['sma_20']

    # Assess Volatility-Adjusted Returns
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['vol_adj_return'] = df['close'].pct_change() / df['volatility_20']

    # Analyze Volume-Based Indicators
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['vol_ratio'] = df['volume'] / df['avg_vol_20']

    # Examine Relationships Between Different Price Points
    df['day_range'] = df['close'] - df['open']
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']

    # Calculate Intraday Metrics
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    df['close_to_open_return'] = df['close'] - df['open']
    df['adj_close_to_open_return'] = df['close_to_open_return'] / df['intraday_range']

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume'])

    # Calculate Close-to-Next Day Open Return over a Period
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = df['vwap'] - df['next_day_open']

    # Combine and Adjust Factors
    df['weighted_sum_returns'] = 0.5 * df['roc_5'] + 0.5 * df['roc_20']
    df['max_close'] = df['close'].rolling(window=20).max()
    df['min_open'] = df['open'].rolling(window=20).min()
    df['adjusted_factor'] = df['midpoint_to_open_ratio'] * df['weighted_sum_returns'] / (df['max_close'] - df['min_open'])

    # Introduce Intraday Momentum Indicator
    df['prev_intraday_midpoint'] = df['intraday_midpoint'].shift(1)
    df['intraday_momentum'] = df['intraday_midpoint'] - df['prev_intraday_midpoint']

    # Introduce Intraday Range to Volume Ratio
    df['intraday_range_to_vol_ratio'] = df['intraday_range'] / df['volume']

    # Final Adjustments
    df['final_adjusted_factor'] = df['midpoint_to_open_ratio'] + df['close_to_next_open_return'] / (df['max_close'] - df['min_open'])

    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()

    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['vol_ratio']
    df['combined_factor'] = df['adj_close_to_open_return'] + df['weighted_intraday_range']

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Compute Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()

    # Compute Cumulative Price Momentum
    df['cumulative_momentum'] = df['daily_price_change'].ewm(span=20, adjust=False).mean()

    # Introduce Intraday Momentum Indicator
    df['intraday_momentum'] = df['intraday_midpoint'] - df['intraday_midpoint'].shift(1)

    # Introduce Intraday Range to Volume Ratio
    df['intraday_range_to_vol_ratio'] = df['intraday_range'] / df['volume']
