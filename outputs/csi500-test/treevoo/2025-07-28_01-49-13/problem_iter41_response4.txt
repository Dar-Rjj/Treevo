```python
def heuristics_v2(df):
    # Calculate the weighted average of high and low prices
    weighted_avg = (df['high'] + 2 * df['low']) / 3
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff()
    
    # Calculate the trading intensity as a function of amount and volume
    trading_intensity = df['amount'] / df['volume']
    
    # The heuristic alpha factor is a combination of weighted average, price change, and trading intensity
    factor_value = (weighted_avg - df['close'].shift(1)) * price_change * trading_intensity
    
    return factor_value
```
