import numpy as np

    # Momentum-based factors
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_diff'] = df['EMA_5'] - df['EMA_20']
    df['ROC_5'] = df['close'].pct_change(5)
    df['ROC_10'] = df['close'].pct_change(10)

    df['volume_ratio_5'] = df['volume'] / df['volume'].rolling(window=5).median()
    df['volume_ratio_20'] = df['volume'] / df['volume'].rolling(window=20).median()
    df['volume_mom'] = df['volume'].diff()
    df['volume_ROC_5'] = df['volume'].pct_change(5)

    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    df['VWAP_5'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWAP_20'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWAP_close_diff_5'] = df['close'] - df['VWAP_5']
    df['VWAP_close_diff_20'] = df['close'] - df['VWAP_20']

    df['MAD_5'] = df['close'].rolling(window=5).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    df['MAD_20'] = df['close'].rolling(window=20).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    df['true_range'] = df[['high', 'low', df['close'].shift()].max(axis=1) - df[['low', df['close'].shift()]].min(axis=1)
