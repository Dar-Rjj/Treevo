```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Adjusted Close-to-Open Return
    adjusted_c_to_o_return = (df['close'] - df['open']) / intraday_range
    
    # Calculate Rate of Change in Closing Price
    roc_5 = df['close'].pct_change(5)
    roc_20 = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    crossover_5_20 = sma_5 - sma_20
    crossover_10_50 = sma_10 - sma_50
    
    # Examine Relationships Between Different Price Points
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    
    # Assess Volatility-Adjusted Returns
    volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_return = (df['close'] - df['close'].shift(1)) / volatility
    
    # Measure Volume Ratio
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Close-to-Next Day Open Return
    next_day_open = df['open'].shift(-1)
    close_to_next_open_return = (vwap - next_day_open) / vwap
    
    # Weighted Sum of Returns by Volume
    weighted_sum_returns = (close_to_next_open_return * df['volume']).rolling(window=5).mean()
    
    # Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Adjusted Close-to-Open Envelope
    max_adjusted_close = (df['close'] - df['open']).rolling(window=20).max()
    min_open_period = df['open'].rolling(window=20).min()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']) / df['close'].shift(1)
    smoothed_volatility = daily_volatility.ewm(span=20).mean()
    
    # Combine Factors
    combined_factor = (midpoint_to_open_ratio * weighted_sum_returns) * (max_close + min_open + max_adjusted_close + min_open_period) * (daily_volatility + smoothed_volatility)
    
    # Final Factor Adjustment
    final_factor = (midpoint_to_open_ratio * weighted_sum_returns) + (adjusted_c_to_o_return * intraday_range) + (max_adjusted_close - min_open_period)
    
    # Compute Cumulative Price Momentum
    price_momentum = daily_price_change.ewm(span=20).mean()
    
    return final_factor * price_momentum
```
