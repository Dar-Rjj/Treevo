```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on the close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range (high - low)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the average true range over 14 days
    df['true_range'] = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    df['avg_true_range_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the money flow, which is the typical price multiplied by the volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Calculate the 14-day money flow ratio
    df['positive_money_flow'] = df[df['daily_return'] > 0]['money_flow']
    df['negative_money_flow'] = df[df['daily_return'] < 0]['money_flow']
    df['money_ratio_14'] = df['positive_money_flow'].rolling(window=14).sum() / df['negative_money_flow'].rolling(window=14).sum()
    
    # Calculate the final alpha factor
    df['alpha_factor'] = (df['money_ratio_14'] - 1) * df['avg_true_range_14']
    
    # Return the alpha factor
    return df['alpha_factor']
```
