```python
def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'].diff()
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Cumulative Sums
    factor_values = cumulative_price_momentum * cumulative_volume_change
    
    return factor_values
```
