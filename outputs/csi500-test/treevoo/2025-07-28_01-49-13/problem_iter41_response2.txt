```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_to_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 5
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    
    # Calculate Rate of Change over a defined period
    df['roc'] = df['close'].pct_change(periods=short_period)
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: np.max([x[0] - x[1], 
                                                                 np.abs(x[0] - df['close'].shift(1)), 
                                                                 np.abs(x[1] - df['close'].shift(1))]), axis=1)
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).sum()
    
    # Measure Standard Deviation of Price Changes
    df['price_std'] = df['close'].diff().rolling(window=short_period).std()
    
    # Combine Indicators for Enhanced Alpha Factors
    df['close_to_next_open_return'] = df['vwap'].shift(-1) / df['open'].shift(-1) - 1
    df['weighted_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 20
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    df['adjusted_max_close'] = df['close'].rolling(window=envelope_period).max()
    df['adjusted_min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=short_period, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    
    # Final Factor Combination
    df['intraday_momentum_adjusted'] = df['midpoint_to_open_ratio'] * df['weighted_returns']
    df['combined_envelopes'] = (df['max_close'] - df['min_open']) + (df['adjusted_max_close'] - df['adjusted_min_open'])
    df['obv_trend_confirmation'] = df['obv'].pct_change() * df['close'].pct_change()
    
    # Develop factors based on combined metrics
    df['momentum_atr_volume_range'] = (df['momentum'] + df['atr'] + (df['volume'] / df['intraday_range'])) / 3
    df['adjusted_close_to_open_return'] = (df['vwap'] - df['open'].shift(-1)) / df['intraday_range']
    df['volume_weighted_volatility'] = df['volume'] * df['smoothed_volatility']
    
    # Integrate Crossover Alpha Factors
    df['intraday_momentum_price_volume_correlation'] = df['momentum'] * df['close'].rolling(window=short_period).corr(df['volume'])
    df['envelopes_volatility_factor'] = (df['max_close'] - df['min_open']) * df['smoothed_volatility']
    
    # Develop Composite Alpha Factors
    composite_alpha = (df['intraday_momentum_adjusted'] + 
                       df['combined_envelopes'] + 
                       df['intraday_momentum_price_volume_correlation'] + 
                       df['envelopes_volatility_factor'] + 
                       df['obv_trend_confirmation'] + 
                       df['momentum_atr_volume_range'] + 
                       df['adjusted_close_to_open_return'] + 
                       df['volume_weighted_volatility']) / 8
    
    return composite_alpha.dropna()
```
