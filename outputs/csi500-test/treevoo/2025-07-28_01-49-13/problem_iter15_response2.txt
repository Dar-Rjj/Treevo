```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] / df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + adjusted_close_to_open_return
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum using EMA
    cumulative_price_momentum = daily_price_change.ewm(span=10, adjust=False).mean()
    
    # Compute Cumulative Volume Momentum using EMA
    cumulative_volume_momentum = volume_momentum.ewm(span=10, adjust=False).mean()
    
    # Final Combination
    combined_momentum_metrics = cumulative_price_momentum * cumulative_volume_momentum
    final_alpha_factor = combined_metrics + combined_momentum_metrics
    
    return final_alpha_factor
```
