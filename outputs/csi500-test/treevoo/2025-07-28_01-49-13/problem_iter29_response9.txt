```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['Momentum_5'] = df['close'] - df['SMA_5']
    df['Rate_of_Change'] = df['close'].pct_change(periods=1)
    
    # Volatility Indicators
    df['True_Range'] = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close']), abs(x['low'] - x['close'])), axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['StdDev_14'] = df['close'].rolling(window=14).std()
    
    # Intraday Indicators
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Movement'] = df['close'] - df['open']
    df['Intraday_Midpoint'] = (df['high'] + df['low']) / 2
    df['Intraday_Midpoint_to_Open_Ratio'] = df['Intraday_Midpoint'] / df['open']
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Adjusted_Close_to_Open_Return'] = df['Close_to_Open_Return'] / df['Intraday_Range']
    
    # Volume Indicators
    df['OBV'] = ((df['close'] > df['close'].shift(1)) * df['volume']).cumsum() - ((df['close'] < df['close'].shift(1)) * df['volume']).cumsum()
    df['VWAP'] = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()
    df['Volume_Weighted_Volatility'] = df['Intraday_Range'] / df['volume']
    df['Intraday_Range_to_Volume_Ratio'] = df['Intraday_Range'] / df['volume']
    
    # Combined Indicators
    df['Momentum_Volatility_Ratio'] = df['Momentum_5'] / df['ATR_14']
    df['OBV_Trend_Confirmation'] = df['OBV'].pct_change(periods=1) * df['close'].pct_change(periods=1)
    df['Weighted_Sum'] = df['Momentum_5'] + df['ATR_14'] + df['Intraday_Range_to_Volume_Ratio']
    df['Close_to_Next_Day_Open_Return'] = (df['open'].shift(-1) - df['VWAP']) * df['volume']
    df['Max_Close'] = df['close'].rolling(window=10).max()
    df['Min_Open'] = df['open'].rolling(window=10).min()
    df['Close_to_Open_Envelope_Adjusted'] = (df['Max_Close'] - df['Min_Open']) * df['Intraday_Midpoint_to_Open_Ratio']
    
    # Final Alpha Factor
    alpha_factor = df['Close_to_Open_Envelope_Adjusted'] + df['Close_to_Next_Day_Open_Return'] + df['Weighted_Sum']
    
    return alpha_factor
```
