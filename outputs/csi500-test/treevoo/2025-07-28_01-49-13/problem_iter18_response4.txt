import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Volume-Weighted Volatility
    volume_sum_20_days = df['volume'].rolling(window=20).sum()
    volume_weighted_volatility = intraday_range / volume_sum_20_days
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    returns = vwap - next_day_open
    weighted_returns = returns * df['volume']
    
    # Calculate Close-to-Open Envelope
    max_close_20_days = df['close'].rolling(window=20).max()
    min_open_20_days = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close_to_open_return_20_days = adjusted_close_to_open_return.rolling(window=20).max()
    min_open_20_days_envelope = df['open'].rolling(window=20).min()
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_midpoint - df['open']
    
    # Combine Factors
    factor = intraday_midpoint_to_open_ratio * weighted_returns
    factor = factor * (max_close_20_days - min_open_20_days) / (max_adjusted_close_to_open_return_20_days - min_open_20_days_envelope)
    
    return factor.dropna()
