```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Spread
    weighted_spread = (df['high'] - df['low']) * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    ema_window = 10
    weighted_ema = weighted_spread.ewm(span=ema_window, adjust=False).mean()
    
    # Calculate the difference between current EMA and EMA from 5 days ago
    lag_days = 5
    ema_lag = weighted_ema.shift(lag_days)
    factor = weighted_ema - ema_lag
    
    return factor
```
