```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Momentum
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = (df['close'] - df['next_day_open']) / df['next_day_open']
    df['volume_weighted_momentum'] = (df['close_to_next_open_return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close_5d'] = df['close'].rolling(window=5).max()
    df['min_open_5d'] = df['open'].rolling(window=5).min()
    
    # Calculate Intraday Range to Open Ratio
    df['intraday_range_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume Momentum
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_momentum'] = (df['close_to_close_return'] * df['daily_volume_change']).rolling(window=5).sum()
    
    # Calculate High-Low Envelope
    df['max_high_5d'] = df['high'].rolling(window=5).max()
    df['min_low_5d'] = df['low'].rolling(window=5).min()
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = (df['midpoint_open_ratio'] * df['volume_weighted_momentum']) * (df['max_adjusted_close_5d'] / df['min_open_5d'])
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    df['adjusted_intraday_factors'] = (df['intraday_range_open_ratio'] * df['volume_momentum']) * (df['max_high_5d'] / df['min_low_5d'])
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Combine Final Intraday and Cumulative Factors
    df['final_alpha_factor'] = df['combined_intraday_factors'] * df['combined_cumulative_sums']
    
    return df['final_alpha_factor']
```
