```python
def heuristics_v2(df):
    # Calculate the weighted average of open and close prices using volume as weights
    df['weighted_avg'] = (df['open'] * df['volume'] + df['close'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between high and low prices
    df['price_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of the weighted average to the price difference
    df['factor'] = df['weighted_avg'] / df['price_diff']
    
    return df['factor']
```
