```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5, lookback=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    avg_volume = df['volume'].rolling(window=M).mean()
    std_volume = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (avg_volume + 2 * std_volume)).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['adjusted_return'] = df['N_day_return'] * np.where(df['volume_spike'] == 1, K, 1)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_to_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    avg_volume_20 = df['volume'].rolling(window=lookback).mean()
    df['volatility_weighted'] = df['intraday_range'] / avg_volume_20
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = df['vwap'] / df['next_day_open'] - 1
    df['weighted_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=N).max()
    df['min_open'] = df['open'].rolling(window=N).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=N).max()
    df['min_open'] = df['open'].rolling(window=N).min()
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume'].pct_change().fillna(0)
    df['combined_metrics'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Enhance Combined Metrics
    df['enhanced_combined_metrics'] = df['combined_metrics'] * np.where(df['volume_spike'] == 1, K, 1) * df['intraday_range'] * df['adjusted_close_to_open_return']
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'].diff()
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['volume_momentum'] = df['volume'].pct_change().fillna(0)
    df['cumulative_volume_change'] = df['volume_momentum'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['adjusted_return']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ema_adjusted_return'] = df['volume_weighted_adjusted_return'].ewm(span=lookback).mean()
    
    # Final Combination
    df['final_factor'] = df['enhanced_combined_metrics'] + df['combined_cumulative_sums']
    df['final_factor'] = df['final_factor'] * df['volatility_weighted'] * df['volume_momentum'] * (df['volume_weighted_adjusted_return'] / df['ema_adjusted_return'])
    
    return df['final_factor']
```
