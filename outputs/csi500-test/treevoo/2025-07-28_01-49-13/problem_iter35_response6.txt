```python
def heuristics_v2(data):
    # Calculate the change in close price
    data['close_change'] = data['close'].diff()
    
    # Calculate the average of high and low
    data['avg_high_low'] = (data['high'] + data['low']) / 2
    
    # Calculate the relative strength compared to the average of high and low
    data['rsi_like'] = data['close'] - data['avg_high_low']
    
    # Calculate the money flow factor
    data['money_flow'] = data['amount'] / data['volume']
    
    # Combine the factors into a final alpha factor
    data['alpha_factor'] = (data['close_change'] * data['rsi_like']) / data['money_flow']
    
    return data['alpha_factor']
```
