```python
import pandas as pd

def heuristics_v2(df):
    # Evaluate Intraday and Volume-Weighted Dynamics
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_midpoint_to_open_ratio'] = (df['high'] + df['low']) / 2 / df['open']
    df['vwap'] = df['amount'] / df['volume']
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    df['adj_close_to_open_return'] = df['close_to_open_return'] / df['intraday_range']
    df['vol_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_to_avg_vol_20d_ratio'] = df['volume'] / df['vol_20d_avg']

    # Analyze Trend and Momentum
    df['roc_5d'] = df['close'].pct_change(periods=5)
    df['roc_20d'] = df['close'].pct_change(periods=20)
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    df['sma_20d'] = df['close'].rolling(window=20).mean()
    df['sma_50d'] = df['close'].rolling(window=50).mean()
    df['sma_5d_vs_20d'] = df['sma_5d'] > df['sma_20d']
    df['sma_20d_vs_50d'] = df['sma_20d'] > df['sma_50d']
    df['day_range'] = df['close'] - df['open']
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']

    # Assess Volatility and Adjusted Returns
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=20).std()
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()

    # Combine Intraday and Cumulative Metrics
    df['daily_price_change'] = df['close'].diff()
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    df['close_to_next_open_return'] = df['vwap'].shift(-1) / df['open'] - 1
    df['weighted_returns_by_volume'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    df['adj_close_to_open_envelope'] = max_adj_close - min_open
    
    df['intraday_midpoint_weighted_returns'] = df['intraday_midpoint_to_open_ratio'] * df['weighted_returns_by_volume']

    # Final Factor Adjustment
    df['intraday_vol_weighted_momentum'] = df['intraday_range'] * df['volume_to_avg_vol_20d_ratio']
    df['adjusted_close_to_open_return'] = df['adj_close_to_open_return'] * df['weighted_returns_by_volume']
    df['combined_intraday_factor'] = df['intraday_vol_weighted_momentum'] + df['adjusted_close_to_open_return']
    df['cumulative_price_momentum'] = df['cumulative_price_change'].ewm(span=20, adjust=False).mean()
    df['final_alpha_factor'] = df['combined_intraday_factor'] + df['cumulative_price_momentum']

    return df['final_alpha_factor']
```
