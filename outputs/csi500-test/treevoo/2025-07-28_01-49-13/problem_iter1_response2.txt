```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    daily_return = df['close'].diff()
    
    # Generate Volume Weights
    total_volume_last_10_days = df['volume'].rolling(window=10).sum()
    volume_weights = df['volume'] / total_volume_last_10_days
    
    # Calculate Weighted Moving Average of Returns
    weighted_returns = daily_return.rolling(window=10).apply(
        lambda x: (x * volume_weights.loc[x.index]).sum(), raw=False)
    
    # Combine Momentum with Volume Sensitivity
    recent_volume = df['volume']
    factor_values = weighted_returns * recent_volume
    
    return factor_values
```
