def heuristics_v2(df):
    """
    Generate an alpha factor based on a heuristic approach.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the range of the day as a percentage of the open price
    daily_range_percent = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume weighted close price
    vw_close = df['close'] * df['volume']
    
    # Calculate the moving average of the volume weighted close price over 10 days
    vw_close_ma = vw_close.rolling(window=10).mean()
    
    # Heuristic alpha factor: Combine the above calculations
    alpha_factor = close_change + daily_range_percent + (vw_close / vw_close_ma - 1)
    
    return alpha_factor
