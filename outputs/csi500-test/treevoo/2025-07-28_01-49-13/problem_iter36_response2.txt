def heuristics_v2(df):
    # Momentum-based factors
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_5_20_diff'] = df['SMA_5'] - df['SMA_20']
    df['SMA_10_20_diff'] = df['SMA_10'] - df['SMA_20']
    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_10'] = df['close'].pct_change(periods=10)

    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_5'] = df['volume'] / df['avg_volume_5']
    df['volume_ratio_10'] = df['volume'] / df['avg_volume_10']
    df['volume_ratio_20'] = df['volume'] / df['avg_volume_20']
    df['volume_momentum'] = df['volume'].diff()
    df['volume_ROC_5'] = df['volume'].pct_change(periods=5)
    df['volume_ROC_10'] = df['volume'].pct_change(periods=10)

    # Price and volume combined factors
    df['OBV'] = (df['close'].diff() / df['close'].abs() * df['volume']).cumsum()
    df['VWAP_5'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    df['VWAP_10'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    df['VWAP_20'] = (df['amount'] / df['volume']).rolling(window=20).mean()
    df['close_VWAP_5_diff'] = df['close'] - df['VWAP_5']
    df['close_VWAP_10_diff'] = df['close'] - df['VWAP_10']
    df['close_VWAP_20_diff'] = df['close'] - df['VWAP_20']

    # Volatility-based factors
    df['std_5'] = df['close'].rolling(window=5).std()
    df['std_10'] = df['close'].rolling(window=10).std()
    df['std_20'] = df['close'].rolling(window=20).std()
    df['range'] = df['high'] - df['low']
    df['avg_range_5'] = df['range'].rolling(window=5).mean()
    df['avg_range_10'] = df['range'].rolling(window=10).mean()
    df['avg_range_20'] = df['range'].rolling(window=20).mean()

    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
