import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_mid_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'] - 1
    volume_weighted_momentum = (close_to_next_open_return * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    adj_close_to_open_envelope = max_adj_close / min_open
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_momentum = (close_to_close_return * daily_volume_change).rolling(window=20).sum() / daily_volume_change.rolling(window=20).sum()
    
    # Combine Intraday Factors
    combined_intraday_factors = (intraday_mid_to_open_ratio * volume_weighted_momentum) * adj_close_to_open_envelope
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    adjusted_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) * (df['high'] - df['low'])
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = (df['volume'] * (df['high'] - df['low']) * close_to_open_return)
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'] + df['low']).shift(1) / 2) * intraday_range
    
    # Detect Volume Spike
    volume_spike = (daily_volume_change > df['volume'].rolling(window=20).std())
    
    # Incorporate Volatility into Final Factor
    volatility = df['high'] - df['low']
    final_factor_volatility_adjusted = (combined_intraday_factors * adjusted_intraday_factors) / volatility
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.ewm(span=20, adjust=False).mean()
    
    # Compute Cumulative Volume Momentum
    cumulative_volume_momentum = volume_momentum.ewm(span=20, adjust=False).mean()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Momentum Metrics
    combined_momentum_metrics = cumulative_price_momentum * cumulative_volume_momentum
    
    # Combine Factors
    final_factor = (final_factor_volatility_adjusted * (cumulative_price_change * cumulative_volume_change))
    
    return final_factor
