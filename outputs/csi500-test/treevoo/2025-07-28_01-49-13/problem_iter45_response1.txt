```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Adjusted Close-to-Open Return
    adjusted_close_to_open = (df['close'] - df['open']) / intraday_range
    
    # Calculate Rate of Change in Closing Price
    roc_5 = df['close'].pct_change(5)
    roc_20 = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    ma_crossover_5_20 = ma_5 - ma_20
    ma_crossover_10_50 = ma_10 - ma_50
    
    # Examine Relationships Between Different Price Points
    close_to_high = df['close'] / df['high']
    close_to_low = df['close'] / df['low']
    
    # Assess Volatility-Adjusted Returns
    volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_returns = (df['close'].pct_change()) / volatility
    
    # Measure Volume Ratio
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Calculate Daily Price Change and Cumulative Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    cumulative_price_change = daily_price_change.cumsum()
    
    # Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Close-to-Next Day VWAP Return
    vwap_shift = vwap.shift(-1)
    close_to_next_day_vwap_return = (vwap_shift - df['close']) / vwap
    
    # Weighted Sum of Returns by Volume
    weighted_sum_returns = (close_to_next_day_vwap_return * df['volume']).rolling(window=20).mean()
    
    # Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] / df['close'].shift(1)
    max_adjusted_close = adjusted_close.rolling(window=20).max()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low'])
    smoothed_volatility = daily_volatility.ewm(span=20, adjust=False).mean()
    
    # Integrate Market Sentiment
    open_close_diff = df['close'] - df['open']
    positive_negative_open_close = (open_close_diff > 0).astype(int)
    
    # Combine Factors
    combined_envelopes = max_close - min_open
    factor = (midpoint_to_open_ratio * weighted_sum_returns) / combined_envelopes
    factor += (max_adjusted_close - min_open) / combined_envelopes
    factor += (volatility_adjusted_returns * smoothed_volatility)
    factor += positive_negative_open_close
    
    # Final Factor Adjustment
    factor *= (midpoint_to_open_ratio * weighted_sum_returns)
    factor += (adjusted_close_to_open * intraday_range)
    factor += positive_negative_open_close
    
    # Compute Cumulative Price Momentum
    ema_price_momentum = daily_price_change.ewm(span=20, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    final_factor = factor + ema_price_momentum + ema_volume
    return final_factor
```
