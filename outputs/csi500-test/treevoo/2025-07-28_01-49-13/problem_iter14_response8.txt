```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)

    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=M).mean()
    df['volume_std'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['volume_ma'] + 2 * df['volume_std'])).astype(int)

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_to_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1

    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']

    # Calculate Volume-Weighted Volatility
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_weighted_volatility'] = df['intraday_range'] / df['avg_volume']

    # Combine Adjusted Return and Volume-Weighted Volatility
    df['combined_adjusted_return_volatility'] = df['adjusted_close_to_open_return'] * df['volume_weighted_volatility']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']

    # Calculate Close-to-Next Day Open Return over a period
    df['close_to_next_day_open_return'] = (df['open'].shift(-1) / df['vwap']) - 1
    df['weighted_sum_returns'] = df['close_to_next_day_open_return'] * df['volume']

    # Calculate Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=N).max()
    df['min_open'] = df['open'].rolling(window=N).min()

    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'] * df['intraday_midpoint_to_open_ratio']
    df['max_adjusted_close'] = df['max_adjusted_close'].rolling(window=N).max()

    # Combine Metrics
    df['volume_momentum'] = df['volume'] / df['volume'].shift(1)
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume_momentum']
    df['combined_metrics'] = df['weighted_intraday_range'] + df['close_to_open_return']

    # Enhance Combined Metrics
    K = 2  # Amplification factor for volume spike
    df['enhanced_combined_metrics'] = df['N_day_return'] * (K * df['volume_spike'] + (1 - df['volume_spike']))
    df['enhanced_combined_metrics'] = df['enhanced_combined_metrics'] * df['intraday_range'] * df['adjusted_close_to_open_return']

    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)

    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()

    # Compute Cumulative Sum of Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['cumulative_volume_change'] = df['volume_change'].cumsum()

    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']

    # Final Combination
    df['volume_weighted_volatility_final'] = df['intraday_range'] / df['volume'] * df['amount'].rolling(window=20).mean()
    df['final_factor'] = df['enhanced_combined_metrics'] + df['combined_cumulative_sums'] * df['volume_weighted_volatility_final']

    return df['final_factor']
```
