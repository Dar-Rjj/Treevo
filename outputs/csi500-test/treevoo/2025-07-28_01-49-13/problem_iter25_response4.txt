```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_to_open_ratio'] = df['midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    df['close_to_next_day_open_return'] = df['vwap'].shift(-1) / df['open'] - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_of_returns'] = df['close_to_next_day_open_return'] * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=10).max()
    df['min_open'] = df['open'].rolling(window=10).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = (df['high'] - df['low']).abs()
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=10, adjust=False).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_midpoint_to_open_ratio'] * df['weighted_sum_of_returns']) / (df['max_adjusted_close'] - df['min_open'])
    df['combined_factor'] = df['combined_factor'] * (df['daily_volatility'] / df['smoothed_volatility'])
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_next_day_open_return'] / df['intraday_range']
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['weighted_sum_of_returns']
    df['adjusted_combined_factor'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Final Factor Adjustment
    df['final_factor'] = df['adjusted_combined_factor'] + df['cumulative_price_change']
    
    # Compute Cumulative Price Momentum
    df['cumulative_momentum'] = df['daily_price_change'].ewm(span=10, adjust=False).mean()
    
    # Generate the final interpretable alpha factor
    df['alpha_factor'] = df['final_factor'] + df['cumulative_momentum']
    
    return df['alpha_factor']
```
