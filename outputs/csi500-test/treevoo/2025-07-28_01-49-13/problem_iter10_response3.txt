```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range
    
    # Calculate Volume-Weighted Volatility
    average_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_volatility = intraday_range / average_volume
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    factor_values = adjusted_close_to_open_return * volume_weighted_volatility
    
    return factor_values
```
