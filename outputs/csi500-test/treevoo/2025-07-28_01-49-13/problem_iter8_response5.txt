```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = intraday_range * close_to_open_return
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + close_to_open_return
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum and Volume Change
    cumulative_price_momentum = daily_price_momentum.cumsum()
    cumulative_volume_change = volume_momentum.cumsum()
    
    # Combine Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Final Combination
    # Incorporate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / avg_volume) * avg_amount
    volume_weighted_volatility *= volume_momentum
    
    final_factor = combined_metrics + combined_cumulative_sums + volume_weighted_volatility
    
    return final_factor
```
