```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    midpoint = (df['high'] + df['low']) / 2
    intraday_mid_open_ratio = midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'].shift(-1) - 1
    volume_weighted_momentum = close_to_next_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=10).max()
    min_open = df['open'].rolling(window=10).min()
    adjusted_close_open_envelope = max_adjusted_close / min_open
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_momentum = volume_change * close_to_close_return
    
    # Combine Intraday Factors
    combined_intraday_factors = (intraday_mid_open_ratio * volume_weighted_momentum) / adjusted_close_open_envelope
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    high_low_envelope = df['high'].rolling(window=10).max() / df['low'].rolling(window=10).min()
    intraday_adjustment = (intraday_range_open_ratio * volume_momentum) / high_low_envelope
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] / df['open'] - 1
    
    # Combine High-Low Range and Close-Open Return
    combined_high_low_return = (intraday_range + close_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_return
    
    # Calculate Price Change Momentum
    price_change_momentum = (df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2
    price_change_momentum *= intraday_range
    
    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = daily_volume_change > daily_volume_change.rolling(window=10).std() * 2
    
    # Integrate Stochastic Oscillator into Final Factor
    period = 14
    lowest_low = df['low'].rolling(window=period).min()
    highest_high = df['high'].rolling(window=period).max()
    k_percent = (df['close'] - lowest_low) / (highest_high - lowest_low) * 100
    d_percent = k_percent.rolling(window=3).mean()
    
    # Integrate ATR into Final Factor
    true_range = (df['high'] - df['low']).abs().max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Generate Final Alpha Factor
    final_factor = (combined_intraday_factors * intraday_adjustment * volume_weighted_adjusted_return * price_change_momentum * d_percent * atr) * volume_spike
    
    return final_factor
```
