```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Volume Weighted Momentum
    close_to_next_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    volume_weighted_momentum = (close_to_next_open_return * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_momentum = (close_to_close_return * daily_volume_change).rolling(window=20).sum()

    # Combine Intraday Factors
    combined_intraday_factors = intraday_range * volume_weighted_momentum
    adjusted_combined_intraday_factors = combined_intraday_factors + volume_momentum

    # Detect Volume Spike
    volume_spike_threshold = 3
    daily_volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_spike_indicator = (daily_volume_change_percentage > volume_spike_threshold).astype(int)

    # Final Factor Calculation
    final_factor = adjusted_combined_intraday_factors * volume_momentum * volume_spike_indicator

    return final_factor
```
