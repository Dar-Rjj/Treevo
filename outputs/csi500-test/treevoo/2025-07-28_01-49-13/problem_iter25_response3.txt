```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate the intraday price movement
    df['IntradayPriceMovement'] = df['close'] - df['open']
    
    # Calculate the volume-range ratio
    df['VolumeRangeRatio'] = df['volume'] / df['DailyRange']
    
    # Calculate the rate of change over a defined period (Momentum)
    df['Momentum'] = df['close'].pct_change(periods=10)
    
    # Calculate the Moving Average Convergence Divergence (MACD) using close prices
    df['EMA12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD'] = df['EMA12'] - df['EMA26']
    
    # Calculate the average true range (ATR) over a specified time frame
    df['TR'] = df[['high', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Combine multiple indicators to create a composite factor
    df['CompositeFactor'] = 0.5 * df['Momentum'] + 0.3 * df['ATR'] + 0.2 * df['VolumeRangeRatio']
    
    return df['CompositeFactor']
```
