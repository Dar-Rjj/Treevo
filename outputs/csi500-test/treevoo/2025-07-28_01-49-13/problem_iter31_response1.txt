```python
import pandas as pd

def heuristics_v2(df):
    # Momentum-based alpha factors
    df['1d_price_change'] = df['close'].diff()
    df['5d_price_change'] = df['close'] - df['close'].shift(5)
    df['20d_price_change'] = df['close'] - df['close'].shift(20)

    df['1d_rate_of_change'] = df['1d_price_change'] / df['close'].shift(1)
    df['5d_rate_of_change'] = df['5d_price_change'] / df['close'].shift(5)
    df['20d_rate_of_change'] = df['20d_price_change'] / df['close'].shift(20)

    # Moving average crossovers
    df['5d_SMA'] = df['close'].rolling(window=5).mean()
    df['20d_SMA'] = df['close'].rolling(window=20).mean()
    df['50d_SMA'] = df['close'].rolling(window=50).mean()

    df['5_20_crossover'] = df['5d_SMA'] - df['20d_SMA']
    df['20_50_crossover'] = df['20d_SMA'] - df['50d_SMA']
    df['5_50_crossover'] = df['5d_SMA'] - df['50d_SMA']

    # Volatility-based alpha factors
    df['daily_log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['20d_volatility'] = df['daily_log_returns'].rolling(window=20).std()

    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    df['high_low_ratio'] = df['high_low_range'] / df['avg_high_low_range']

    # Volume-based alpha factors
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    df['1d_VROC'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['21d_VROC'] = (df['volume'] - df['volume'].shift(21)) / df['volume'].shift(21)

    # Pattern recognition alpha factors
    df['bullish_engulfing'] = ((df['open'] < df['close'].shift(1)) & 
                               (df['close'] > df['open'].shift(1))).astype(int)
    df['bearish_engulfing'] = ((df['open'] > df['close'].shift(1)) & 
                               (df['close'] < df['open'].shift(1))).astype(int)
    df['doji_pattern'] = (abs(df['open'] - df['close']) < 0.005 * (df['high'] - df['low'])).astype(int)
    df['hammer_pattern'] = (((df['close'] - df['low']) / (df['high'] - df['low']) > 0.66) & 
                            (df['low'] < df['open'].shift(1))).astype(int)

    # Combine all factors into a single factor
    df['alpha_factor'] = (df['1d_price_change'] + df['5d_price_change'] + df['20d_price_change'] + 
                          df['1d_rate_of_change'] + df['5d_rate_of_change'] + df['20d_rate_of_change'] + 
                          df['5_20_crossover'] + df['20_50_crossover'] + df['5_50_crossover'] + 
                          df['20d_volatility'] + df['high_low_ratio'] + 
                          df['OBV'] + df['1d_VROC'] + df['21d_VROC'] + 
                          df['bullish_engulfing'] - df['bearish_engulfing'] + 
                          df['doji_pattern'] + df['hammer_pattern'])

    return df['alpha_factor']
```
