```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Normalize By Total Volume
    df['total_volume'] = df['volume'].rolling(window=len(df), min_periods=1).sum()
    df['factor'] = df['combined_cumulative'] / df['total_volume']
    
    return df['factor']
```
