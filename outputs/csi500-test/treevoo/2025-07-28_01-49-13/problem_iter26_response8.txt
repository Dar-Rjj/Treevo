```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    df['close_to_next_day_open_return'] = df['vwap'].shift(-1) / df['open'].shift(-1) - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_day_open_return'] * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=20).max()
    df['min_open'] = df['open'].rolling(window=20).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = (df['high'] - df['low'])
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['combined_factors'] = (
        df['intraday_midpoint_open_ratio'] * 
        df['weighted_sum_returns'] * 
        (df['max_adjusted_close'] - df['min_open']) / df['intraday_range'] * 
        df['smoothed_volatility']
    )
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_next_day_open_return'] / df['intraday_range']
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['weighted_sum_returns']
    df['adjusted_combined_metrics'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Final Factor Adjustment
    df['final_factor'] = df['combined_factors'] + df['cumulative_price_change'] + df['adjusted_combined_metrics']
    
    # Compute Cumulative Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_change'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Short-Term Price Momentum
    df['short_term_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['short_term_momentum'] = df['short_term_ema'] - df['long_term_ema']
    
    # Synthesize Composite Alpha Factor
    df['composite_alpha_factor'] = (
        df['final_factor'] + 
        df['cumulative_price_momentum'] + 
        df['short_term_momentum']
    )
    
    return df['composite_alpha_factor'].dropna()
```
