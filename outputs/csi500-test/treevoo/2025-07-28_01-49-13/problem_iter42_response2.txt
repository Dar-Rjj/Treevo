```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving average (SMA) of close prices
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the slope of SMA to identify trends
    sma_slope = sma.diff()
    
    # Calculate exponential moving average (EMA) of close prices
    ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Examine the rate of change in EMA for short-term momentum
    ema_rate_of_change = ema.pct_change()
    
    # Determine the difference between EMA and SMA
    ema_sma_diff = ema - sma
    
    # Analyze if the difference can indicate market momentum
    ema_sma_diff_trend = ema_sma_diff.rolling(window=5).sum()
    
    # Analyze the high-low spread (High - Low)
    high_low_spread = df['high'] - df['low']
    
    # Assess volatility by comparing daily high-low spreads
    rolling_mean_high_low_spread = high_low_spread.rolling(window=20).mean()
    
    # Evaluate the open-close price change
    open_close_change = df['close'] - df['open']
    
    # Identify the frequency of consecutive up or down days
    consecutive_days = (open_close_change > 0).astype(int).diff().ne(0).cumsum()
    consecutive_days_count = open_close_change.groupby(consecutive_days).cumcount() + 1
    
    # Measure trading volume trends
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine multiple indicators for a composite factor
    composite_factor = (sma_slope + ema_rate_of_change + ema_sma_diff_trend + 
                        rolling_mean_high_low_spread + open_close_change + 
                        consecutive_days_count + volume_ratio)
    
    return composite_factor
```
