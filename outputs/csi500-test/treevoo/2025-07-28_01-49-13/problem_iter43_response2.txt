    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    df['20_day_AVG_Volume'] = df['volume'].rolling(window=20).mean()

    df['Volume_Ratio_20'] = df['volume'] / df['20_day_AVG_Volume']
    df['Volume_Change'] = df['volume'].diff()
    df['VROC_5'] = df['volume'].pct_change(5)

    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    df['VWAP_5'] = (df['volume'] * df['close']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWAP_20'] = (df['volume'] * df['close']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['Close_VWAP5_Diff'] = df['close'] - df['VWAP_5']
    df['Close_VWAP20_Diff'] = df['close'] - df['VWAP_20']

    df['STD_20'] = df['close'].rolling(window=20).std()
