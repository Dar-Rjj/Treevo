```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] / df['open'] - 1
    
    # Combine High-Low Range and Close-Open Return
    combined_return = (high_low_range + close_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_return = combined_return * df['volume']
    
    # Calculate Moving Average of Adjusted Return with a lookback period of 20 days
    moving_average = volume_weighted_return.rolling(window=20).mean()
    
    # Calculate Factor as Ratio of Adjusted Return to Moving Average
    factor = volume_weighted_return / moving_average
    
    return factor
```
