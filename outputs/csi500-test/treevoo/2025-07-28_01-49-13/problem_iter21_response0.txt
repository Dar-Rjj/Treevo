```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume Weighted Momentum
    df['close_to_next_open_return'] = (df['open'].shift(-1) / df['close']) - 1
    df['volume_weighted_momentum'] = (df['close_to_next_open_return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=5).max()
    df['min_open'] = df['open'].rolling(window=5).min()
    
    # Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Volume Momentum
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['close_to_close_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['volume_momentum'] = (df['close_to_close_return'] * df['daily_volume_change']).rolling(window=5).sum() / df['daily_volume_change'].rolling(window=5).sum()
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = (df['intraday_range_to_open_ratio'] * df['volume_weighted_momentum']) * (df['max_adjusted_close'] - df['min_open'])
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    df['high_envelope'] = df['high'].rolling(window=5).max()
    df['low_envelope'] = df['low'].rolling(window=5).min()
    df['adjusted_combined_intraday_factors'] = (df['intraday_range_to_open_ratio'] * df['volume_momentum']) * (df['high_envelope'] - df['low_envelope'])
    
    # Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['combined_intraday_factors']
    
    # Price Change Momentum
    df['price_change_momentum'] = (df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2
    df['price_change_momentum'] *= df['intraday_range']
    
    # Detect Volume Spike
    df['volume_spike_indicator'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > df['volume'].std() else 0)
    
    # Intraday Reversal Signal
    df['intraday_reversal'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    df['intraday_reversal_signal'] = df['intraday_reversal'].apply(lambda x: 0.8 if x > 0.5 else 1)
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_adjusted_return'] * df['price_change_momentum'] * df['volume_spike_indicator'] * df['intraday_reversal_signal']
    
    return df['final_factor']
```
