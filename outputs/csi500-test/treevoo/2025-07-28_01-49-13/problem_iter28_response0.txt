```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Momentum factor: sum of daily returns over a 30-day window
    df['momentum_30'] = df['daily_return'].rolling(window=30).sum()
    
    # Mean reversion factor: (close_t - average(close over a 30-day window)) / standard_deviation(close over a 30-day window)
    df['mean_reversion_30'] = (df['close'] - df['close'].rolling(window=30).mean()) / df['close'].rolling(window=30).std()
    
    # Volatility: standard deviation of daily returns over a 30-day window
    df['volatility_30'] = df['daily_return'].rolling(window=30).std()
    
    # Risk-adjusted return: (sum(daily return over a 30-day window)) / (standard deviation of daily returns over a 30-day window)
    df['risk_adjusted_return_30'] = df['momentum_30'] / df['volatility_30']
    
    # Daily volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Significant volume spike
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['significant_volume_spike'] = df['volume'] > 2 * df['avg_volume_30']
    
    # Impact on price when significant volume spike occurs
    df['price_impact_on_spike'] = df['daily_return'] * df['significant_volume_spike']
    
    # Combine factors to form the final alpha factor
    df['alpha_factor'] = df['momentum_30'] + df['mean_reversion_30'] + df['risk_adjusted_return_30'] + df['price_impact_on_spike']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
