```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']

    # Calculate Close-to-Close Return over a period
    close_returns = df['close'].pct_change()
    volume_momentum = (df['volume'] * close_returns).rolling(window=5).sum()

    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()

    # Combine Intraday Factors
    intraday_factors = (intraday_range_to_open_ratio * volume_momentum) / (max_high - min_low)

    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'].diff()

    # Calculate Daily Volume Change
    daily_volume_change = df['volume'].diff()

    # Compute Cumulative Sum of Price Momentum and Volume Change
    cum_price_momentum = daily_price_momentum.cumsum()
    cum_volume_change = daily_volume_change.cumsum()

    # Combine Cumulative Sums
    cumulative_sums = cum_price_momentum * cum_volume_change

    # Combine Final Factors
    final_factors = intraday_factors * cumulative_sums

    return final_factors
```
