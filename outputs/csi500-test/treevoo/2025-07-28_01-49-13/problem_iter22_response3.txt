def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_mid_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Calculate Volume-Weighted Volatility
    volume_weighted_volatility = intraday_range / df['volume'].sum()
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    close_to_next_day_open_return = vwap - next_day_open
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    close_to_open_envelope = max_close - min_open
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close_to_open_return = adjusted_close_to_open_return.rolling(window=20).max()
    adjusted_close_to_open_envelope = max_adjusted_close_to_open_return - min_open
    
    # Combine Factors
    factor = intraday_mid_open_ratio * close_to_next_day_open_return
    adjusted_factor = factor / (close_to_open_envelope + adjusted_close_to_open_envelope)
    
    return adjusted_factor
