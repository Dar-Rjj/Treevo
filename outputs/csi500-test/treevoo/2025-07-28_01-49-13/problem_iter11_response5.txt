```python
import pandas as pd

def heuristics_v2(df, n=5, m=20, k=1.5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_return = close_to_open_return * intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + adjusted_return
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = volume_momentum.cumsum()
    
    # Calculate N-day Price Return
    n_day_price_return = (df['close'] / df['close'].shift(n) - 1)
    
    # Identify Volume Spike
    moving_avg_volume = df['volume'].rolling(window=m).mean()
    std_volume = df['volume'].rolling(window=m).std()
    volume_spike = (df['volume'] > (moving_avg_volume + 2 * std_volume)).astype(int)
    
    # Adjust N-day Price Return for Volume Spike
    adjusted_n_day_return = n_day_price_return * (volume_spike * (k - 1) + 1)
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_return = adjusted_n_day_return.cumsum()
    
    # Combine Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Incorporate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / avg_volume) * avg_amount
    
    final_combination = combined_metrics + combined_cumulative_sums + (volume_weighted_volatility * volume_momentum)
    
    return final_combination
```
