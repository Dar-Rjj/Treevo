def heuristics_v2(df):
    """
    Generate an alpha factor using a combination of price and volume information.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the relative strength between close and open prices
    relative_strength = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Calculate the difference between the current close price and VWAP
    price_diff_vwap = df['close'] - vwap
    
    # Combine the relative strength and price difference from VWAP
    factor = relative_strength + price_diff_vwap
    
    return factor
