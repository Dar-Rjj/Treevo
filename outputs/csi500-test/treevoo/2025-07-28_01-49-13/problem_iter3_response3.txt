```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume-Weighted Close-to-Close Return
    close_returns = df['close'].pct_change()
    volume_weighted_returns = (close_returns * df['volume']).rolling(window=5).sum()  # Sum over a 5-day window
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()  # Max High over a period
    min_low = df['low'].rolling(window=5).min()  # Min Low over a period
    
    # Combine Factors
    factor = (intraday_range_to_open_ratio * volume_weighted_returns) / (max_high - min_low)
    
    return factor
```
