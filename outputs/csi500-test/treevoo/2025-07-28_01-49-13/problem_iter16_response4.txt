```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range
    
    # Calculate Volume-Weighted Volatility
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_weighted_volatility = intraday_range / avg_volume_20_days
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_return_volatility = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['amount'] / df['volume']).cumsum() / (1+df['volume']).cumsum()
    
    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    return_t_to_t1 = vwap - next_day_open
    weighted_sum_of_returns = (return_t_to_t1 * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    max_adjusted_close = (df['close'] * volume_weighted_volatility).rolling(window=20).max()
    min_adjusted_open = (df['open'] * volume_weighted_volatility).rolling(window=20).min()
    
    # Combine Factors
    factor = midpoint_to_open_ratio * weighted_sum_of_returns
    factor = factor * (max_close + min_open) / (max_adjusted_close + min_adjusted_open)
    
    return factor
```
