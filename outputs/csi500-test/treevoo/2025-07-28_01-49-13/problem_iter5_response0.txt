import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = intraday_range * close_to_open_return
    
    # Combine Metrics: Weight Intraday Range by Volume Momentum and add Close-to-Open Return
    combined_metrics = (intraday_range * volume_momentum) + close_to_open_return
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    cumulative_price_momentum = daily_price_momentum.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = volume_momentum.cumsum()
    
    # Combine Cumulative Sums: Multiply Cumulative Price Momentum by Cumulative Volume Change
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Final Combination: Add Combined Metrics to Combined Cumulative Sums
    final_factor = combined_metrics + combined_cumulative_sums
    
    return final_factor
