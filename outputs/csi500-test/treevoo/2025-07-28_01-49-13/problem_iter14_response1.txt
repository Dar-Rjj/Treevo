```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = intraday_range * close_to_open_return
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_change
    combined_metrics = weighted_intraday_range + adjusted_close_to_open_return
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Incorporate Volume Trend
    combined_metrics_with_volume_trend = combined_metrics * volume_trend
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = volume_change.cumsum()
    
    # Final Combination with Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    final_factor = combined_metrics_with_volume_trend + combined_cumulative_sums
    
    # Calculate Daily Price Return
    N = 5  # Example: 5-day return
    daily_price_return = (df['close'] / df['close'].shift(N)) - 1
    
    # Identify Volume Spike
    M = 20  # Example: 20-day moving average
    volume_spike = (df['volume'] > (df['volume'].rolling(window=M).mean() + 
                                    2 * df['volume'].rolling(window=M).std()))
    
    # Adjust Daily Price Return for Volume Spike
    K = 1.5  # Example constant factor
    adjusted_daily_price_return = daily_price_return.copy()
    adjusted_daily_price_return[volume_spike] = adjusted_daily_price_return[volume_spike] * K
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_price_return = adjusted_daily_price_return.cumsum()
    
    # Combine All Cumulative Sums
    final_factor += cumulative_adjusted_price_return * cumulative_volume_change
    
    return final_factor
```
