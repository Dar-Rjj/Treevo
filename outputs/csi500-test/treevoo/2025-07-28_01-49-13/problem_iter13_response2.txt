```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Close-to-Next Day Open Return
    close_to_next_open_return = vwap.shift(-1) / df['open'].shift(-1)
    
    # Close-to-Open Envelope
    period = 20  # Define the period for the envelope
    max_close_period = df['close'].rolling(window=period).max()
    min_open_period = df['open'].rolling(window=period).min()
    
    # Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] * (df['amount'] / df['volume'])
    max_adjusted_close_period = adjusted_close.rolling(window=period).max()
    
    # Combine Factors
    combined_factor = (midpoint_to_open_ratio * close_to_next_open_return) / (max_close_period - min_open_period)
    adjusted_combined_factor = (midpoint_to_open_ratio * close_to_next_open_return) / (max_adjusted_close_period - min_open_period)
    
    final_factor = combined_factor * adjusted_combined_factor
    
    return final_factor
```
