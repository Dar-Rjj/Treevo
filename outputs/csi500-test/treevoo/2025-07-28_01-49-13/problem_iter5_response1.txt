```python
def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = data['close'] - data['open']
    
    # Calculate Volume Change
    volume_change = data['volume'] - data['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = (close_to_open_return / intraday_range) * close_to_open_return
    
    # Calculate Volume-Weighted Volatility
    avg_volume = data['volume'].rolling(window=20).mean()
    avg_amount = data['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / avg_volume) * avg_amount
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    factor = adjusted_close_to_open_return * volume_weighted_volatility * volume_change
    
    return factor
```
