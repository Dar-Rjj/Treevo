import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum Indicators
    df['DailyMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Assess Volatility and Intraday Range
    df['DailyPriceRange'] = df['high'] - df['low']
    df['IntradayMidpoint'] = (df['high'] + df['low']) / 2
    df['IntradayMidpointToOpenRatio'] = df['IntradayMidpoint'] / df['open']
    df['IntradayReturnAdjustedByHLRange'] = (df['close'] - df['open']) / df['DailyPriceRange']
    
    # Investigate Volume and Trading Activity Levels
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['DollarVolume'] = df['amount'] * df['close']
    df['VWAP'] = df['amount'].cumsum() / df['volume'].cumsum()
    df['VolumeWeightedMomentum'] = (df['VWAP'] - df['open']) / df['VWAP']
    
    # Identify Technical Indicator Signals
    short_window = 10
    long_window = 30
    df['SMA_Short'] = df['close'].rolling(window=short_window).mean()
    df['SMA_Long'] = df['close'].rolling(window=long_window).mean()
    df['MovingAverageCrossover'] = df['SMA_Short'] - df['SMA_Long']
    
    def RSI(series, window):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=window, min_periods=1).mean()
        avg_loss = loss.rolling(window=window, min_periods=1).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    df['RSI'] = RSI(df['close'], 14)
    
    def EMA(series, span):
        return series.ewm(span=span, adjust=False).mean()
    
    short_term = 12
    long_term = 26
    df['EMA_Short'] = EMA(df['close'], short_term)
    df['EMA_Long'] = EMA(df['close'], long_term)
    df['MACD'] = df['EMA_Short'] - df['EMA_Long']
    
    # Examine Market Sentiment through Price, Volume, and OBV
    df['OBV'] = (df['volume'] * (df['close'] > df['close'].shift(1))).cumsum()
    df['PVT'] = (df['volume'] * (df['close'] - df['close'].shift(1)) / df['close'].shift(1)).cumsum()
    
    # Leverage Price, Volume, and Sentiment Correlation
    df['VolumeAdjustedIntradayMomentum'] = df['IntradayMidpoint'] * df['volume']
    df['PriceChangeMomentum'] = ((df['high'] + df['low']) / 2) - ((df['high'].shift(1) + df['low'].shift(1)) / 2)
    
    # Detect and Incorporate Volume Spikes and Reversals
    df['VolumeSpike'] = df['volume'] - df['volume'].shift(1)
    df['IntradayReversal'] = (df['high'] - df['open']) / df['DailyPriceRange']
    df['IntradayReversalAdjustment'] = df['IntradayReversal'].apply(lambda x: 0.8 if x > 0.5 else 1.0)
    
    # Generate Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['DailyMomentum'] * 
        df['IntradayMidpointToOpenRatio'] * 
        df['IntradayReturnAdjustedByHLRange'] * 
        df['VolumeChange'] * 
        df['VolumeWeightedMomentum'] * 
        df['MovingAverageCrossover'] * 
        df['RSI'] * 
        df['MACD'] * 
        df['OBV'] * 
        df['PVT'] * 
        df['VolumeAdjustedIntradayMomentum'] * 
        df['PriceChangeMomentum'] * 
        df['VolumeSpike'] * 
        df['IntradayReversalAdjustment']
    )
