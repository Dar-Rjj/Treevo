```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Rate of Change in Adjusted Closing Price
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)

    # Evaluate Exponential Moving Average Crossovers
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_crossover'] = df['ema_5'] - df['ema_20']

    # Assess Volatility-Adjusted Returns
    df['atr_20'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['vol_adj_return'] = (df['close'] - df['open']) / df['atr_20']

    # Analyze Volume-Based Indicators
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']

    # Examine Relationships Between Different Price Points
    df['close_open_diff'] = df['close'] - df['open']
    df['close_high_ratio'] = df['close'] / df['high']
    df['close_low_ratio'] = df['close'] / df['low']

    # Calculate Intraday Metrics
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_open_ratio'] = df['intraday_midpoint'] / df['open']
    df['close_open_return'] = df['close'] - df['open']
    df['adj_close_open_return'] = df['close_open_return'] / df['intraday_range']

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()

    # VWAP-to-Next Day Open Return over a Period
    df['vwap_next_day_open_return'] = (df['open'].shift(-1) - df['vwap']) / df['open']

    # Intraday Momentum Indicator
    df['intraday_momentum'] = df['intraday_midpoint'] - df['intraday_midpoint'].shift(1)

    # Intraday Range to Volume Ratio
    df['intraday_range_vol_ratio'] = df['intraday_range'] / df['volume']

    # Smoothed Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()

    # Combine and Adjust Factors
    df['weighted_returns'] = 0.5 * df['roc_5'] + 0.5 * df['roc_20']
    df['adjusted_factor'] = df['midpoint_open_ratio'] * df['weighted_returns'] * df['close_high_ratio'] * df['close_low_ratio']
    df['final_adjustment'] = df['adjusted_factor'] * (df['vwap_next_day_open_return'] + df['intraday_range_vol_ratio'])

    # Calculate Daily Adjusted Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)

    # Compute Cumulative Adjusted Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()

    # Compute Cumulative Adjusted Price Momentum
    df['cumulative_momentum'] = df['daily_price_change'].ewm(span=20, adjust=False).mean()

    # Final Factor Adjustment
    df['alpha_factor'] = df['final_adjustment'] * df['cumulative_momentum']

    return df['alpha_factor']
```
