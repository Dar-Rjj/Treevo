```python
import pandas as pd

def heuristics_v2(df):
    # Calculate rate of change in closing price
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)
    
    # Evaluate moving average crossovers
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_crossover_5_20'] = (df['sma_5'] > df['sma_20']).astype(int)
    df['sma_crossover_10_50'] = (df['sma_10'] > df['sma_50']).astype(int)
    
    # Assess volatility-adjusted returns
    df['std_20'] = df['close'].rolling(window=20).std()
    df['sharpe_20'] = (df['close'].pct_change(20) - 0.01) / df['std_20']
    
    # Analyze volume-based indicators
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['high_volume'] = (df['volume'] > df['avg_volume_20'] * 2).astype(int)
    df['low_volume'] = (df['volume'] < df['avg_volume_20'] * 0.5).astype(int)
    df['volume_change'] = df['volume'].pct_change()
    
    # Consider price patterns and support/resistance levels
    df['recent_high_20'] = df['close'].rolling(window=20).max()
    df['recent_low_20'] = df['close'].rolling(window=20).min()
    df['breakout_high'] = (df['close'] > df['recent_high_20']).astype(int)
    df['breakout_low'] = (df['close'] < df['recent_low_20']).astype(int)
    
    # Examine relationships between different price points
    df['day_range'] = df['close'] - df['open']
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    df['close_to_avg_ratio'] = df['close'] / ((df['high'] + df['low']) / 2)
    
    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (
        df['roc_5'] + 
        df['roc_20'] + 
        2 * df['sma_crossover_5_20'] + 
        2 * df['sma_crossover_10_50'] + 
        df['sharpe_20'] + 
        df['volume_ratio'] + 
        df['volume_change'] + 
        2 * df['breakout_high'] - 
        2 * df['breakout_low'] + 
        df['close_to_high_ratio'] + 
        df['close_to_low_ratio'] + 
        df['close_to_avg_ratio']
    )
    
    return df['alpha_factor']
```
