```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Exponential Moving Average of Price Momentum
    df['ema_price_momentum'] = df['price_momentum'].ewm(span=5, adjust=False).mean()
    
    # Compute Exponential Moving Average of Volume Change
    df['ema_volume_change'] = df['volume_change'].ewm(span=5, adjust=False).mean()
    
    # Combine EMAs
    factor = df['ema_price_momentum'] * df['ema_volume_change']
    
    return factor
```
