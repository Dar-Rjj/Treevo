```python
def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow
    money_flow = mf_multiplier * df['volume']
    
    # Calculate the cumulative money flow over the last 14 days
    cum_money_flow = money_flow.rolling(window=14).sum()
    
    # Calculate the cumulative volume over the last 14 days
    cum_volume = df['volume'].rolling(window=14).sum()
    
    # Calculate the Money Flow Index (MFI)
    mfi = cum_money_flow / cum_volume
    
    # Calculate the alpha factor
    alpha_factor = mfi - 0.5
    
    return alpha_factor
```
