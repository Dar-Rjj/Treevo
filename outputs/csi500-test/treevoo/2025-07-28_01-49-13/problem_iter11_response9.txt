import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate N-day Price Momentum
    df['N_day_return'] = (df['close'].shift(-N) / df['close']) - 1
    
    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=M).mean()
    df['volume_std'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['volume_ma'] + 2 * df['volume_std'])).astype(int)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close']) / df['close']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['intraday_range'] * df['close_to_open_return']
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume_momentum']
    df['combined_metrics'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Enhance Combined Metrics
    df['enhanced_combined_metrics'] = df['N_day_return'] + (df['volume_spike'] * (K - 1) * df['N_day_return'])
    df['enhanced_combined_metrics'] *= df['intraday_range']
    df['enhanced_combined_metrics'] *= df['adjusted_close_to_open_return']
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['volume_momentum'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Final Combination
    df['final_factor'] = df['enhanced_combined_metrics'] + df['combined_cumulative_sums']
    
    # Incorporate Volume-Weighted Volatility
    df['volatility'] = df['intraday_range'] / df['volume']
    df['volatility_20_avg'] = df['volatility'].rolling(window=20).mean()
    df['amount_20_avg'] = df['amount'].rolling(window=20).mean()
    df['volume_weighted_volatility'] = df['volatility_20_avg'] * df['amount_20_avg']
    df['final_factor'] *= df['volume_weighted_volatility']
    
    return df['final_factor']
