```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 5
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    )
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).mean()
    
    # Measure Standard Deviation of Price Changes
    std_period = 20
    df['price_change_std'] = df['close'].pct_change().rolling(window=std_period).std()
    
    # Calculate Close-to-Next Day Open Return
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['vwap']) / df['vwap']
    
    # Weighted Sum of Returns by Volume
    df['weighted_return'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 10
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    ema_alpha = 0.1
    df['smoothed_volatility'] = df['daily_volatility'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Final Factor Combination
    df['midpoint_weighted_return'] = df['midpoint_to_open_ratio'] * df['weighted_return']
    df['combined_envelope'] = (df['max_close'] + df['min_open']) / 2
    df['final_factor'] = (
        df['midpoint_weighted_return'] * 
        df['combined_envelope'] * 
        df['obv'].pct_change() * 
        df['momentum'] * 
        df['atr'] * 
        df['smoothed_volatility']
    )
    
    return df['final_factor']
```
