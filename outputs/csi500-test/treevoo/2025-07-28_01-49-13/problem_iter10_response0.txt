```python
import pandas as pd

def heuristics_v2(df, N=20, M=20, K=1.5, lookback=20):
    # Calculate Daily Price Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate N-day Price Momentum
    df['N_day_return'] = (df['close'] / df['close'].shift(N)) - 1
    
    # Identify Volume Spike
    vol_mean = df['volume'].rolling(window=M).mean()
    vol_std = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (vol_mean + 2 * vol_std)).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['combined_momentum'] = df['N_day_return'] * (K * df['volume_spike'] + (1 - df['volume_spike']))
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday High-Low Range and Close-Open Return
    df['combined_return'] = (df['intraday_range'] * df['close_open_return'])
    
    # Compute Cumulative Sum of Price Momentum
    df['cum_price_mom'] = df['daily_momentum'].cumsum()
    
    # Calculate Daily Volume Change
    df['daily_vol_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Volume Change
    df['cum_vol_change'] = df['daily_vol_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cum_sums'] = df['cum_price_mom'] * df['cum_vol_change']
    
    # Calculate Volume Weighted Adjusted Return
    df['vol_weighted_adj_return'] = df['combined_return'] * df['volume']
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['vol_weighted_adj_return'] / df['vol_weighted_adj_return'].rolling(window=lookback).mean()
    
    return df['alpha_factor']
```
