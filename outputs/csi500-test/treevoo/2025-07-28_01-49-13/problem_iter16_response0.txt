```python
import pandas as pd

def heuristics_v2(df, n_days=20, m_days=20, k_factor=1.5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    volume_20_day_ma = df['volume'].rolling(window=n_days).mean()
    volume_trend = df['volume'] - volume_20_day_ma
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_return = close_to_open_return * (intraday_range / df['volume'])
    
    # Calculate Volume-Weighted Volatility
    volume_weighted_volatility = (intraday_range / df['volume']).rolling(window=n_days).mean() * volume_change
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_volatility = adjusted_return * volume_weighted_volatility
    
    # Incorporate Volume Trend
    combined_volatility_with_trend = combined_volatility * volume_trend
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] / df['close'].shift(n_days) - 1
    
    # Identify Volume Spike
    volume_m_day_ma = df['volume'].rolling(window=m_days).mean()
    volume_std = df['volume'].rolling(window=m_days).std()
    volume_spike = (df['volume'] > (volume_m_day_ma + 2 * volume_std)).astype(int)
    
    # Adjust Daily Price Return for Volume Spike
    adjusted_daily_return = daily_price_return.where(volume_spike == 0, daily_price_return * k_factor)
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_return = adjusted_daily_return.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = volume_change.cumsum()
    
    # Combine Cumulative Sums
    factor = cumulative_adjusted_return * cumulative_volume_change
    
    return factor
```
