import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the daily turnover
    df['turnover'] = df['amount'] / df['volume']
    
    # Calculate the daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the factor: (daily return * daily range) / daily volatility
    df['factor'] = (df['daily_return'] * df['daily_range']) / df['daily_volatility']
    
    # Return the factor as a pandas Series
    return df['factor'].dropna()
