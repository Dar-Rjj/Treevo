import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Calculate Volume-Weighted Volatility
    rolling_volume_sum = df['volume'].rolling(window=20).sum()
    volume_weighted_volatility = intraday_range / rolling_volume_sum
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return_volatility = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['amount'] / df['volume']).rename('vwap')
    
    # Calculate Close-to-Next Day Open Return over a period
    df['next_day_open'] = df['open'].shift(-1)
    close_to_next_open_return = vwap - df['next_day_open']
    weighted_returns = close_to_next_open_return * df['volume']
    
    # Calculate Close-to-Open Envelope
    max_close_over_period = df['close'].rolling(window=20).max()
    min_open_over_period = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close_to_open_return = adjusted_close_to_open_return.rolling(window=20).max()
    min_open_over_period = df['open'].rolling(window=20).min()
    
    # Combine Factors
    combined_envelopes = (max_close_over_period - min_open_over_period) + (max_adjusted_close_to_open_return - min_open_over_period)
    combined_factors = midpoint_to_open_ratio * weighted_returns / combined_envelopes
    
    # Final Adjustments
    final_adjustment = combined_factors * (intraday_midpoint - intraday_midpoint.shift(1)) * (intraday_range / df['volume'])
    
    return final_adjustment
