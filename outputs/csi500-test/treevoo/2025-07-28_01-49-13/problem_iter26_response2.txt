```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume factor
    volume_factor = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the amount factor
    amount_factor = df['amount'] / df['amount'].rolling(window=20).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (buying_pressure - selling_pressure) * volume_factor * amount_factor
    
    return alpha_factor
```
