```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    mid_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_volatility = intraday_range / avg_volume
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    close_to_next_day_open_return = vwap - next_day_open
    weighted_sum_returns = close_to_next_day_open_return * df['volume']
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] * intraday_range
    max_adjusted_close = adjusted_close.rolling(window=20).max()
    
    # Combine Factors
    combined_factors = mid_to_open_ratio * weighted_sum_returns
    combined_envelopes = (max_close - min_open) * (max_adjusted_close - min_open)
    combined_factors_adjusted = combined_factors / combined_envelopes
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'].diff()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Final Combine
    final_factor = cumulative_price_change * cumulative_volume_change / combined_envelopes
    
    return final_factor
```
