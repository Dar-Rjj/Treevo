```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    df['volatility'] = df['intraday_range'] / df['volume']
    df['amount_20d_avg'] = df['amount'].rolling(window=20).mean()
    df['vol_weighted_volatility'] = df['volatility'] * df['amount_20d_avg']
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    df['combined_factor'] = df['adjusted_close_to_open_return'] * df['vol_weighted_volatility'] * df['volume_change']
    
    # Incorporate Volume Trend
    df['combined_factor_with_trend'] = df['combined_factor'] * df['volume_trend']
    
    # Calculate Daily Price Return
    n_days = 5
    df['daily_price_return'] = (df['close'] / df['close'].shift(n_days)) - 1
    
    # Identify Volume Spike
    m_days = 20
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=m_days).mean() + 
                          3 * df['volume'].rolling(window=m_days).std())
    
    # Adjust Daily Price Return for Volume Spike
    k = 1.5  # Example factor, can be adjusted
    df['adjusted_daily_price_return'] = df['daily_price_return'] * np.where(df['volume_spike'], k, 1)
    
    # Compute Cumulative Sum of Adjusted Price Return
    df['cumulative_adjusted_price_return'] = df['adjusted_daily_price_return'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['final_factor'] = df['cumulative_adjusted_price_return'] * df['cumulative_volume_change']
    
    return df['final_factor']
```
