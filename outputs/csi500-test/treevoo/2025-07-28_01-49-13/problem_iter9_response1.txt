```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-PreviousClose Return
    close_to_prev_close_return = df['close'].diff()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Adjust Close-to-PreviousClose Return by Intraday Range
    adjusted_return = close_to_prev_close_return * intraday_range
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / avg_volume) * avg_amount
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    factor_value = adjusted_return * volume_weighted_volatility * volume_change
    
    return factor_value
```
