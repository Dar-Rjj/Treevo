```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Adjusted Average Volume
    window = 20
    sum_volume = df['volume'].rolling(window=window).sum()
    adjusted_average_volume = sum_volume / window
    
    # Calculate Daily High-Low Ratio
    daily_high_low_ratio = df['high'] / df['low']
    
    # Combine to get the final factor
    factor = (close_to_open_return * intraday_range) / (adjusted_average_volume * daily_high_low_ratio)
    
    return factor
```
