import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_open_ratio = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Momentum
    df['next_day_open'] = df['open'].shift(-1)
    close_to_next_open_return = (df['next_day_open'] - df['close']) / df['close']
    volume_weighted_momentum = (close_to_next_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Factors
    adjusted_intraday_factor = intraday_midpoint_open_ratio * volume_weighted_momentum
    adj_intraday_factor_with_envelope = adjusted_intraday_factor * (max_adj_close / min_open)
    cumulative_momentum_factor = cumulative_price_change * cumulative_volume_change
    
    # Final Factor Adjustment
    final_factor = adj_intraday_factor_with_envelope + cumulative_momentum_factor
    
    return final_factor.dropna()
