```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_to_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Momentum
    df['close_to_next_open_return'] = df['close'].shift(-1) / df['open'].shift(-1) - 1
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    df['adjusted_close_to_open_envelope'] = max_adjusted_close / min_open
    
    # Calculate Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume Momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['close_to_close_return'] = df['close'] / df['close'].shift(1) - 1
    df['volume_momentum'] = df['volume_change'] * df['close_to_close_return']
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_midpoint_to_open_ratio'] * df['close_to_next_open_return'] * df['adjusted_close_to_open_envelope']
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    df['adjusted_intraday_factors'] = df['intraday_range_to_open_ratio'] * df['volume_momentum']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['close_to_open_return'] * (df['high'] - df['low'])
    
    # Calculate Price Change Momentum
    df['price_change_momentum'] = ((df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2) * df['intraday_range']
    
    # Detect Volume Spike
    df['volume_spike'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike_indicator'] = df['volume_spike'] > df['volume_spike'].quantile(0.9)
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_adjusted_return'] * df['price_change_momentum'] * df['volume_spike_indicator']
    
    return df['final_factor']
```
