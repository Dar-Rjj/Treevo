```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_open_return = vwap.shift(-1) / df['open'].shift(-1) - 1
    
    # Weighted Sum of Returns by Volume
    weighted_sum_of_returns = close_to_next_open_return * df['volume']
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=10).max()
    min_open = df['open'].rolling(window=10).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] / df['close'].iloc[0] * df['open'].iloc[0]
    max_adjusted_close = adjusted_close.rolling(window=10).max()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']).abs()
    smoothed_volatility = daily_volatility.ewm(span=10, adjust=False).mean()
    
    # Combine Factors
    combined_envelopes = (max_close / min_open) * (max_adjusted_close / min_open)
    factor = intraday_midpoint_open_ratio * weighted_sum_of_returns / combined_envelopes * smoothed_volatility
    
    return factor.dropna()
```
