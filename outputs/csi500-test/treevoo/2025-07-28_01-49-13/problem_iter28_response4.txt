```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['Momentum_Short_Term'] = df['close'] - df['SMA_5']
    
    # Volatility Indicators
    df['Previous_Close'] = df['close'].shift(1)
    df['True_Range'] = df[['high', 'low', 'Previous_Close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['Previous_Close']), abs(x['low'] - x['Previous_Close'])), axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).sum()
    
    # On-Balance Volume (OBV) Calculation
    df['Volume_Adjustment'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'Volume_Adjustment'] = df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'Volume_Adjustment'] = -df['volume']
    df['OBV'] = df['Volume_Adjustment'].cumsum()
    
    # Momentum-Volatility Ratio
    df['Momentum_Volatility_Ratio'] = df['Momentum_Short_Term'] / df['ATR_14']
    
    # OBV Trend Confirmation
    df['OBV_Trend'] = df['OBV'].diff()
    df['Close_Trend'] = df['close'].diff()
    df['Trend_Confirmation'] = (df['OBV_Trend'] * df['Close_Trend']).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Combined Alpha Factor
    alpha_factor = df['Momentum_Volatility_Ratio'] * df['Trend_Confirmation']
    
    return alpha_factor
```
