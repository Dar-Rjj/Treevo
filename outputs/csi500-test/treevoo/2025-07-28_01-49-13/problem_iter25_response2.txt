```python
import pandas as pd

def heuristics_v2(df):
    # Calculate rate of change in closing price
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)
    
    # Evaluate moving average crossovers
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['cross_5_20'] = df['sma_5'] - df['sma_20']
    df['cross_20_50'] = df['sma_20'] - df['sma_50']
    
    # Assess volatility-adjusted returns
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['adjusted_return_20'] = (df['close'] - df['close'].shift(20)) / df['volatility_20']
    
    # Analyze volume-based indicators
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    
    # Consider price patterns and support/resistance levels
    df['high_breakout'] = (df['close'] > df['high'].rolling(window=20).max()).astype(int)
    df['low_breakout'] = (df['close'] < df['low'].rolling(window=20).min()).astype(int)
    
    # Examine relationships between different price points
    df['day_range'] = df['close'] - df['open']
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Combine all the sub-thoughts into a single alpha factor
    alpha_factor = (
        df['roc_5'] + 
        df['roc_20'] + 
        df['cross_5_20'] + 
        df['cross_20_50'] + 
        df['adjusted_return_20'] + 
        df['volume_ratio'] + 
        df['high_breakout'] - 
        df['low_breakout'] + 
        df['day_range'] + 
        (df['close_to_high_ratio'] - df['close_to_low_ratio'])
    )
    
    return alpha_factor
```
