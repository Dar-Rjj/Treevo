```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_open_ratio = (df['high'] + df['low']) / 2 / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Adjust Close-to-Open Return by Intraday Range
    close_to_open_return = (df['close'] - df['open']) / intraday_range
    
    # Measure the ratio of today's volume to the average volume over the past 20 days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Calculate Rate of Change in Closing Price
    roc_5 = df['close'].pct_change(5)
    roc_20 = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    crossover_5_20 = sma_5 - sma_20
    crossover_20_50 = sma_20 - sma_50
    
    # Examine Relationships Between Different Price Points
    close_open_diff = df['close'] - df['open']
    close_high_ratio = df['close'] / df['high']
    close_low_ratio = df['close'] / df['low']
    
    # Use Standard Deviation Over Past 20 Days as Volatility Measure
    volatility = df['close'].rolling(window=20).std()
    
    # Identify Days with Exceptionally High or Low Volume Relative to the Recent Average
    high_volume_days = df['volume'] > (avg_volume_20 * 1.5)
    low_volume_days = df['volume'] < (avg_volume_20 * 0.5)
    
    # Integrate Volatility Factor
    daily_volatility = df['high'] - df['low']
    smoothed_volatility = daily_volatility.ewm(span=20, adjust=False).mean()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Calculate Close-to-Next Day Open Return
    close_next_day_open_return = vwap - df['open'].shift(-1)
    
    # Weighted Sum of Returns by Volume
    weighted_returns = close_next_day_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Multiply Intraday Midpoint to Open Ratio by Weighted Sum of Returns
    intraday_weighted_factor = intraday_midpoint_open_ratio * weighted_returns
    
    # Combine Intraday Range by Volume-Weighted Momentum
    intraday_range_momentum = intraday_range * volume_ratio
    
    # Add Adjusted Close-to-Open Return to the weighted Intraday Range
    adjusted_intraday_factor = intraday_range_momentum + close_to_open_return
    
    # Combine the adjusted intraday factor with the cumulative momentum factor
    combined_factor = adjusted_intraday_factor + cumulative_price_change
    
    # Generate the final interpretable alpha factor
    alpha_factor = combined_factor * (crossover_5_20 + crossover_20_50) * (close_high_ratio + close_low_ratio) * (daily_volatility / smoothed_volatility)
    
    # Compute Cumulative Price Momentum
    price_momentum = daily_price_change.ewm(span=20, adjust=False).mean()
    final_alpha_factor = alpha_factor + price_momentum
    
    return final_alpha_factor
```
