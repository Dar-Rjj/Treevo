```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low price
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure by comparing the close and open price with the daily range
    df['buying_pressure'] = 2 * (df['close'] - df['low']) + (df['high'] - df['close']) - (df['high'] - df['open'])
    
    # Calculate the average volume over a 5-day window
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Create an alpha factor that combines the price range, buying pressure, and relative volume
    df['alpha_factor'] = (df['price_range'] * df['buying_pressure']) / (df['avg_volume_5d'] + 1e-6)
    
    return df['alpha_factor']
```
