```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Volume-Weighted Volatility
    average_volume = df['volume'].rolling(window=20).mean()
    df['volume_weighted_volatility'] = df['intraday_range'] / average_volume
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    df['combined_adjusted_return_volatility'] = df['adjusted_close_to_open_return'] * df['volume_weighted_volatility']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    df['next_day_open'] = df['open'].shift(-1)
    df['vwap_to_next_day_open_return'] = df['vwap'] - df['next_day_open']
    df['weighted_returns'] = df['vwap_to_next_day_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] * df['intraday_range']
    max_adjusted_close = adjusted_close.rolling(window=20).max()
    
    # Combine Factors
    df['final_factor'] = (df['midpoint_to_open_ratio'] * df['weighted_returns']) / (max_close - min_open + max_adjusted_close - min_open)
    
    return df['final_factor']
```
