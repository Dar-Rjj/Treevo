import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, close) prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the money flow by multiplying the average price with the volume
    money_flow = avg_price * df['volume']
    
    # Calculate the typical price using (high, low, close)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow index (MFI) using the typical price and money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    money_flow_index = 100 - (100 / (1 + money_ratio))
    
    # Calculate the factor value
    factor_value = money_flow_index - typical_price.pct_change().rolling(window=14).mean()
    
    return factor_value
