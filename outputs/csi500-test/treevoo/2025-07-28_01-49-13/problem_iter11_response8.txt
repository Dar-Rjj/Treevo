```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Open VWAP Momentum
    df['vwap_momentum'] = (df['vwap'].shift(-1) - df['vwap']) / df['vwap']
    df['vwap_momentum'] = df['vwap_momentum'].fillna(0).cumsum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    df['adjusted_close_to_open_envelope'] = max_close / min_open
    
    # Calculate Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume-Weighted Momentum
    df['close_to_next_day_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['volume_weighted_momentum'] = df['close_to_next_day_open_return'] * df['volume']
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'].fillna(0).cumsum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    df['high_low_envelope'] = max_high / min_low
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = (df['midpoint_to_open_ratio'] * df['volume_weighted_momentum']) * df['vwap_momentum'] * df['adjusted_close_to_open_envelope']
    
    # Adjust by Intraday Range to Open Ratio and Volume Change
    df['range_volume_adjustment'] = df['intraday_range_to_open_ratio'] * df['volume_change']
    df['combined_intraday_factors'] = df['combined_intraday_factors'] * df['range_volume_adjustment'] * df['high_low_envelope']
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Combine Final Intraday and Cumulative Factors
    df['final_factor'] = df['combined_intraday_factors'] * df['combined_cumulative_sums']
    
    return df['final_factor']
```
