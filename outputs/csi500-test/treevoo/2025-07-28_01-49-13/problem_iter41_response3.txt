```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the buying pressure as (amount / volume)
    df['buying_pressure'] = df['amount'] / df['volume']
    
    # Calculate the weighted average of the above factors
    df['alpha_factor'] = (df['close_change'] * 0.4 + 
                          df['relative_range'] * 0.3 + 
                          df['buying_pressure'] * 0.3)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
