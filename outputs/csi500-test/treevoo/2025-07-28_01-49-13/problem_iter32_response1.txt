```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the high-low range
    close_rel_pos = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the relative amount traded compared to the average amount over the last 10 days
    avg_amount_10d = df['amount'].rolling(window=10).mean()
    rel_amount = df['amount'] / avg_amount_10d

    # Combine the two factors: close relative position and relative amount
    # This combination assumes that a high relative close with higher-than-usual volume is a positive signal
    alpha_factor = close_rel_pos * rel_amount

    return alpha_factor
```
