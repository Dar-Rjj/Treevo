```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate 20-day Moving Average of Volume
    df['volume_20ma'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume_20ma']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['avg_amount'] = df['amount'].rolling(window=20).mean()
    df['volume_weighted_volatility'] = (df['intraday_range'] / df['avg_volume']) * df['avg_amount']
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    df['combined_factor'] = df['adjusted_close_to_open_return'] * df['volume_weighted_volatility'] * df['volume_change']
    
    # Incorporate Volume Trend
    df['incorporated_volume_trend'] = df['combined_factor'] * df['volume_trend']
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'] - df['close'].shift(5)
    
    # Identify Volume Spike
    df['volume_20std'] = df['volume'].rolling(window=20).std()
    df['volume_spike'] = (df['volume'] > (df['volume_20ma'] + 3 * df['volume_20std'])).astype(int)
    
    # Adjust Daily Price Return for Volume Spike
    K = 1.5  # Example constant, can be adjusted
    df['adjusted_daily_price_return'] = df['daily_price_return'] * (K * df['volume_spike'] + (1 - df['volume_spike']))
    
    # Compute Cumulative Sum of Adjusted Price Return
    df['cumulative_adjusted_price_return'] = df['adjusted_daily_price_return'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sum'] = df['cumulative_adjusted_price_return'] * df['cumulative_volume_change']
    
    # Introduce Momentum Factor
    df['momentum_factor'] = df['close'] - df['close'].rolling(window=50).mean()
    
    # Incorporate Momentum Factor
    df['final_factor'] = df['adjusted_daily_price_return'] * df['momentum_factor']
    
    return df['final_factor']
```
