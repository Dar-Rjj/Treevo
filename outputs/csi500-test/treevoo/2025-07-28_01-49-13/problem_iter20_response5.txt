```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(-1) - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range
    
    # Calculate Volume-Weighted Volatility
    vol_weighted_volatility = (intraday_range / df['volume']).rolling(window=20).mean() * volume_change
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_close_to_open_return * vol_weighted_volatility
    
    # Incorporate Volume Trend
    combined_adjusted_return_with_trend = combined_adjusted_return * volume_trend
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] / df['close'].shift(5) - 1
    
    # Identify Volume Spike
    volume_spike = (df['volume'] > (df['volume'].rolling(window=20).mean() + 2 * df['volume'].rolling(window=20).std()))
    
    # Adjust Daily Price Return for Volume Spike
    k = 1.5  # Amplification factor
    adjusted_daily_price_return = daily_price_return.where(~volume_spike, daily_price_return * k)
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_price_return = adjusted_daily_price_return.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = volume_change.cumsum()
    
    # Combine Cumulative Sums
    combined_cumulative = cumulative_adjusted_price_return * cumulative_volume_change
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_to_open_ratio = (df['high'] + df['low']) / (2 * df['open'])
    
    # Calculate Close-to-Next Day Open Return over a period
    vwap = (df['amount'] / df['volume']).shift(1)
    close_to_next_day_open_return = (df['open'].shift(-1) - vwap) / vwap
    weighted_sum_of_returns = (close_to_next_day_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_adj_open = df['open'].rolling(window=5).min()
    
    # Combine Factors
    combined_factors = intraday_midpoint_to_open_ratio * weighted_sum_of_returns * (max_close - min_open) * (max_adj_close - min_adj_open)
    
    # Final Combine
    final_factor = combined_cumulative * combined_factors
    
    return final_factor
```
