```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'] - 1
    volume_weighted_momentum = df['volume'] * close_to_next_open_return
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    adjusted_close_to_open_envelope = max_adjusted_close - min_open
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    volume_diff = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_momentum = volume_diff * close_to_close_return
    
    # Combine Intraday Factors
    combined_intraday_factors = intraday_midpoint_to_open_ratio * volume_weighted_momentum
    combined_intraday_factors /= adjusted_close_to_open_envelope
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    max_high = df['high'].rolling(window=20).max()
    min_low = df['low'].rolling(window=20).min()
    high_low_envelope = max_high - min_low
    adjusted_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) / high_low_envelope
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Combine High-Low Range and Close-Open Return
    combined_high_low_range_and_return = intraday_range * close_to_open_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_range_and_return
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2) * intraday_range
    
    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_indicator = (daily_volume_change > daily_volume_change.rolling(window=20).std()).astype(int)
    
    # Final Factor Calculation
    final_factor = volume_weighted_adjusted_return * price_change_momentum * volume_spike_indicator
    
    # Integrate MACD into Final Factor
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_term_ema - long_term_ema
    final_factor *= macd
    
    # Integrate RSI into Final Factor
    gains = df['close'].diff().where(df['close'].diff() > 0, 0)
    losses = -df['close'].diff().where(df['close'].diff() < 0, 0)
    avg_gain = gains.rolling(window=14).mean()
    avg_loss = losses.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    final_factor *= rsi
    
    return final_factor
```
