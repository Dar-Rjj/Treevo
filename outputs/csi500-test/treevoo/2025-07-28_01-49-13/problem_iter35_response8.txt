import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change in Closing Price
    df['roc_5'] = df['close'].pct_change(periods=5)
    df['roc_20'] = df['close'].pct_change(periods=20)
    
    # Evaluate Moving Average Crossovers
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_crossover'] = df['sma_5'] - df['sma_20']
    
    # Assess Volatility-Adjusted Returns
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['vol_adj_return'] = (df['close'] / df['close'].shift(1) - 1) / df['volatility_20']
    
    # Analyze Volume-Based Indicators
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    
    # Examine Relationships Between Different Price Points
    df['day_range'] = df['close'] - df['open']
    df['close_to_high_ratio'] = df['close'] / df['high']
    df['close_to_low_ratio'] = df['close'] / df['low']
    
    # Intraday Metrics
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    df['close_to_open_return'] = df['close'] - df['open']
    df['adj_close_to_open_return'] = df['close_to_open_return'] / df['intraday_range']
    
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()
    
    # Close-to-Next Day Open Return
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_day_open_return'] = df['vwap'] - df['next_day_open']
    
    # Combine and Adjust Factors
    df['weighted_sum_returns'] = df['roc_5'] * 0.6 + df['roc_20'] * 0.4
    df['max_close_20'] = df['close'].rolling(window=20).max()
    df['min_open_20'] = df['open'].rolling(window=20).min()
    df['intraday_momentum'] = df['intraday_midpoint'] - df['intraday_midpoint'].shift(1)
    df['intraday_range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Final Adjustments
    df['factor_1'] = df['intraday_midpoint_to_open_ratio'] * df['weighted_sum_returns']
    df['factor_2'] = df['factor_1'] + df['close_to_next_day_open_return']
    df['factor_3'] = df['factor_2'] * (df['max_close_20'] / df['min_open_20'])
    df['factor_4'] = df['factor_3'] + df['intraday_momentum'] + df['intraday_range_to_volume_ratio']
    
    # Introduce Short-Term Momentum and Volume
    df['short_term_momentum'] = (df['close'] - df['close'].rolling(window=5).mean()) / df['close'].rolling(window=5).std()
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio_5'] = df['volume'] / df['avg_volume_5']
    df['combined_momentum_vol'] = df['short_term_momentum'] * df['volume_ratio_5']
    df['directional_movement'] = (df['high'] > df['high'].shift(1)) & (df['low'] > df['low'].shift(1)).astype(int)
    df['final_factor'] = df['factor_4'] * df['combined_momentum_vol'] * df['directional_movement']
    
    return df['final_factor'].dropna()
