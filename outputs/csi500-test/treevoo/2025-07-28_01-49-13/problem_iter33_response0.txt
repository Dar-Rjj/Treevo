```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Indicators
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ShortTermMomentum'] = df['close'] - df['EMA_5']
    df['Momentum'] = df['close'].pct_change(periods=14)
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD'] = df['EMA_12'] - df['EMA_26']

    # Volatility Indicators
    df['TrueRange'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    df['ATR_14'] = df['TrueRange'].rolling(window=14).mean()
    df['StdDev_PriceChange'] = df['close'].pct_change().rolling(window=20).std()

    # Intraday Indicators
    df['IntradayRange'] = df['high'] - df['low']
    df['IntradayMovement'] = df['close'] - df['open']
    df['IntradayMidpoint'] = (df['high'] + df['low']) / 2
    df['MidpointToOpenRatio'] = df['IntradayMidpoint'] / df['open']
    df['CloseToOpenReturn'] = df['close'] - df['open']
    df['AdjustedCloseToOpenReturn'] = df['CloseToOpenReturn'] / df['IntradayRange']

    # Volume Indicators
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VolumeWeightedVolatility'] = df['IntradayRange'] / df['volume'].cumsum()
    df['IntradayRangeToVolumeRatio'] = df['IntradayRange'] / df['volume']

    # Combined Alpha Factors
    df['MomentumVolatilityRatio'] = df['ShortTermMomentum'] / df['ATR_14']
    df['OBVTrendConfirmation'] = df['OBV'].diff() * df['close'].diff()

    # Combined Metrics
    df['CombinedMetrics'] = 0.4 * df['ShortTermMomentum'] + 0.3 * df['ATR_14'] + 0.3 * df['IntradayRangeToVolumeRatio']
    df['AdjustedCloseToOpenReturn_VWAP'] = df['AdjustedCloseToOpenReturn'] * df['volume'] / df['VWAP']

    # Close-to-Next Day Open Return
    df['CloseToNextDayOpenReturn'] = df['close'].shift(-1) - df['open']
    df['WeightedReturnsByVolume'] = (df['CloseToNextDayOpenReturn'] * df['volume']).rolling(window=10).sum()

    # Close-to-Open Envelope
    df['MaxClose'] = df['close'].rolling(window=20).max()
    df['MinOpen'] = df['open'].rolling(window=20).min()
    df['Envelope'] = (df['MaxClose'] - df['MinOpen']) / df['IntradayRange']

    # Final Adjustments
    df['FinalAlphaFactor'] = 0.5 * df['MidpointToOpenRatio'] + 0.5 * df['CloseToNextDayOpenReturn_VWAP']
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'] * df['Envelope']

    # Intraday Volatility
    df['IntradayVolatility'] = df['IntradayMovement'].rolling(window=20).std()
    df['FinalAlphaFactor_Adjusted'] = df['FinalAlphaFactor'] / df['IntradayVolatility']

    # Trading Range Factor
    df['TradingRange_20'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['NormalizedFinalAlphaFactor'] = df['FinalAlphaFactor_Adjusted'] / df['TradingRange_20']

    return df['NormalizedFinalAlphaFactor']
```
