```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['volume_MA_M'] = df['volume'].rolling(window=M).mean()
    df['volume_std_M'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > df['volume_MA_M'] + 2 * df['volume_std_M']).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['enhanced_N_day_return'] = df['N_day_return'] * (1 + (K - 1) * df['volume_spike'])
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_open_ratio'] = (df['high'] + df['low']) / (2 * df['open'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    df['average_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_weighted_volatility'] = df['intraday_range'] / df['average_volume_20']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate Close-to-Next Day Open Return over a period
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_day_open_return'] = (df['vwap'] - df['next_day_open']) / df['vwap']
    df['weighted_sum_returns'] = (df['close_to_next_day_open_return'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=N).max()
    df['min_open'] = df['open'].rolling(window=N).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=N).max()
    df['min_open'] = df['open'].rolling(window=N).min()
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume'].pct_change()
    df['combined_metrics'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Enhance Combined Metrics
    df['enhanced_combined_metrics'] = df['combined_metrics'] * (1 + (K - 1) * df['volume_spike'])
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['volume_change'] = df['volume'].diff()
    df['cumulative_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['enhanced_combined_metrics']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ema_volume_weighted_adjusted_return'] = df['volume_weighted_adjusted_return'].ewm(span=20).mean()
    
    # Final Combination
    df['final_factor'] = (df['enhanced_combined_metrics'] + df['combined_cumulative_sums']) * df['volume_weighted_volatility'] * df['volume_change'] * (df['volume_weighted_adjusted_return'] / df['ema_volume_weighted_adjusted_return'])
    
    return df['final_factor']
```
