```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day average closing price
    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between today's close and the 5-day average
    df['close_diff'] = df['close'] - df['avg_close_5d']
    
    # Calculate the standard deviation of the past 5 days' closing prices
    df['std_close_5d'] = df['close'].rolling(window=5).std()
    
    # Adjust the momentum signal by dividing by the standard deviation
    df['momentum_signal'] = df['close_diff'] / df['std_close_5d']
    
    # Calculate the ratio of todayâ€™s volume to the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Weigh the momentum signal by volume strength
    df['momentum_weighted_by_volume'] = df['momentum_signal'] * df['volume_ratio']
    
    # Check for bullish trend: today's high is higher than yesterday's AND today's low is higher
    df['bullish_trend'] = (df['high'] > df['high'].shift(1)) & (df['low'] > df['low'].shift(1))
    df['bullish_trend'] = df['bullish_trend'].astype(int)
    
    # Add the binary bullish trend to the overall factor
    df['alpha_factor'] = df['momentum_weighted_by_volume'] + df['bullish_trend']
    
    return df['alpha_factor']
```
