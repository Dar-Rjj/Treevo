import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] / df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = intraday_range * close_to_open_return
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + close_to_open_return
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Momentum
    cumulative_volume_momentum = volume_momentum.cumprod()
    
    # Final Combination
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_momentum
    final_factor = combined_metrics + combined_cumulative_sums
    
    return final_factor
