```python
import pandas as pd

def heuristics_v2(df, N=10, M=30, K=1.5, lookback=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['vol_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['vol_std'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['vol_moving_avg'] + 2 * df['vol_std'])).astype(int)
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Price Momentum, Volume Spike, and Intraday Volatility
    df['combined_return'] = df['N_day_return'] * (1 + (df['volume_spike'] * (K - 1)))
    df['combined_return'] = df['combined_return'] * df['intraday_range'] * df['close_to_open_return']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_return'] = df['combined_return'] * df['volume']
    
    # Calculate Moving Average of Volume Weighted Adjusted Return
    df['avg_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=lookback).mean()
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['avg_volume_weighted_return']
    
    return df['alpha_factor']
```
