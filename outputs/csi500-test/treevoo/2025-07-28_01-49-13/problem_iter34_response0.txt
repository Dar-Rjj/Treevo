```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_to_open_ratio = (df['high'] + df['low']) / 2 / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Adjust Close-to-Open Return by Intraday Range
    close_to_open_return = (df['close'] - df['open']) / intraday_range
    
    # Calculate Rate of Change in Closing Price
    roc_5 = df['close'].pct_change(5)
    roc_20 = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    crossover_5_20 = sma_5 > sma_20
    crossover_20_50 = sma_20 > sma_50
    
    # Examine Relationships Between Different Price Points
    close_to_open_diff = df['close'] - df['open']
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']
    
    # Assess Volatility-Adjusted Returns
    volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_return = (df['close'].pct_change()) / volatility.shift(1)
    
    # Measure the ratio of today's volume to the average volume over the past 20 days
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_open_return = vwap - df['open'].shift(-1)
    
    # Weighted Sum of Returns by Volume
    weighted_close_to_next_open_return = close_to_next_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Integrate Volatility Factor
    daily_volatility = (df['high'] - df['low']) / 2
    smooth_volatility = df['close'].ewm(span=20, adjust=False).std()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_range +
        intraday_midpoint_to_open_ratio +
        vwap +
        close_to_open_return +
        roc_5 + roc_20 +
        crossover_5_20 + crossover_20_50 +
        close_to_open_diff + close_to_high_ratio + close_to_low_ratio +
        volatility_adjusted_return +
        volume_ratio +
        daily_price_change + cumulative_price_change +
        close_to_next_open_return + weighted_close_to_next_open_return +
        max_adj_close + min_open +
        daily_volatility + smooth_volatility
    )
    
    return alpha_factor
```
