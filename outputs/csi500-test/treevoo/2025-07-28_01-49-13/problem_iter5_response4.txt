import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate N-day Price Momentum
    n_day_return = df['close'].pct_change(periods=N)
    
    # Identify Volume Spike
    volume_moving_avg = df['volume'].rolling(window=M).mean()
    volume_std = df['volume'].rolling(window=M).std()
    volume_spike = (df['volume'] > volume_moving_avg + 2 * volume_std).astype(int)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday Range and Close-to-Open Return
    combined_return = intraday_range * close_to_open_return
    
    # Adjusted Combined Return Based on Volume
    adjusted_n_day_return = n_day_return * (K * volume_spike + (1 - volume_spike))
    
    # Calculate Adjusted Average Volume
    avg_volume = df['volume'].rolling(window=20).sum() / 20
    daily_high_low_ratio = df['high'] / df['low']
    adjusted_avg_volume = avg_volume * daily_high_low_ratio
    
    # Multiply by Combined Intraday Range and Close-Open Return
    adjusted_combined_return = adjusted_n_day_return / adjusted_avg_volume * combined_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * adjusted_combined_return
    
    # Calculate Final Alpha Factor
    ma_volume_weighted_return = volume_weighted_adjusted_return.rolling(window=20).mean()
    final_alpha_factor = volume_weighted_adjusted_return / ma_volume_weighted_return
    
    return final_alpha_factor
