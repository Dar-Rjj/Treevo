import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 5
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    df['roc'] = df['close'].pct_change(periods=short_period)
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).sum() / atr_period
    
    # Measure Standard Deviation of Price Changes
    std_period = 20
    df['std_dev'] = df['close'].rolling(window=std_period).std()
    
    # Calculate Close-to-Next Day Open Return
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = (df['next_day_open'] / df['vwap']) - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 10
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['adjusted_max_close'] = (df['close'] / df['volume']).rolling(window=envelope_period).max()
    df['adjusted_min_open'] = (df['open'] / df['volume']).rolling(window=envelope_period).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = (df['high'] - df['low'])
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=std_period, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Final Factor Combination
    df['final_factor'] = (
        df['midpoint_to_open_ratio'] * df['weighted_sum_returns'] *
        (df['max_close'] - df['min_open']) * (df['adjusted_max_close'] - df['adjusted_min_open']) *
        df['obv'] * df['momentum'] * df['atr'] * df['daily_volatility'] * df['smoothed_volatility']
    )
    
    # Introduce Price-Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=std_period).corr(df['volume'])
    
    return df['final_factor']
