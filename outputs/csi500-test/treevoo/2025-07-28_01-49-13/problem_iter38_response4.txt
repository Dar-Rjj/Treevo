import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_open_ratio'] = df['midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 10
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    df['roc'] = df['close'].pct_change(periods=short_period)
    df['macd_line'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    df['macd'] = df['macd_line'] - df['signal_line']
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['atr'] = df['true_range'].rolling(window=short_period).sum()
    
    # Measure Standard Deviation of Price Changes
    df['price_change_std'] = df['close'].diff().rolling(window=short_period).std()
    
    # Combine Indicators for Enhanced Alpha Factors
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['vwap']) / df['vwap']
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 20
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['adj_close'] = df['close'] / df['open']
    df['max_adj_close'] = df['adj_close'].rolling(window=envelope_period).max()
    df['min_adj_open'] = df['min_open']
    
    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=short_period, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        else:
            obv.append(obv[-1] - df['volume'].iloc[i])
    df['obv'] = obv
    
    # Final Factor Combination
    df['factor'] = (
        df['intraday_midpoint_open_ratio'] * df['weighted_sum_returns'] +
        (df['max_close'] + df['min_open'] + df['max_adj_close'] + df['min_adj_open']) / 4 +
        df['obv'] * df['close'].pct_change().rolling(window=short_period).mean() +
        (df['momentum'] + df['atr'] + df['vwap'] * df['intraday_range']) / 3 +
        df['close_to_next_open_return'] * df['intraday_range'] +
        df['volume'] * df['smoothed_volatility']
    )
    
    # Introduce Price-Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=short_period).corr(df['volume'])
    
    return df['factor']
