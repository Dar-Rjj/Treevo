```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20, momentum_period=5):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-Open Return
    combined_return = intraday_range * close_open_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = combined_return * df['volume']
    
    # Calculate Moving Average of Adjusted Return
    moving_avg_adjusted_return = volume_weighted_adjusted_return.rolling(window=lookback).mean()
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    close_close_returns = df['close'].pct_change().fillna(0)
    volume_momentum = (close_close_returns * df['volume']).rolling(window=momentum_period).sum()
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=lookback).max()
    min_low = df['low'].rolling(window=lookback).min()
    
    # Introduce Trend Strength Indicator
    trend_strength = (df['close'].rolling(window=lookback).apply(lambda x: np.polyfit(range(len(x)), x, 1)[0]))
    
    # Combine Factors
    factor_ratio = volume_weighted_adjusted_return / moving_avg_adjusted_return
    factor_intraday_open = intraday_range_to_open_ratio * volume_momentum
    factor_envelope_adjustment = (max_high - min_low) / (max_high + min_low)
    final_factor = factor_ratio * factor_intraday_open * factor_envelope_adjustment * trend_strength
    
    return final_factor
```
