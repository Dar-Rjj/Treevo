def heuristics_v2(df):
    # Calculate the relative position of the closing price in the range for the day
    df['range'] = df['high'] - df['low']
    df['close_range_position'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the amount-volume ratio as a liquidity measure
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Calculate a smoothed version of the close range position using an exponential moving average
    df['smoothed_close_range_position'] = df['close_range_position'].ewm(span=5, adjust=False).mean()
    
    # Generate a factor based on the combination of the smoothed close range position and the liquidity measure
    df['factor'] = df['smoothed_close_range_position'] * df['amount_volume_ratio']
    
    # Return the alpha factor as a pandas Series
    return df['factor']
