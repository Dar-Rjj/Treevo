```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    
    # Calculate the intraday return from open to close
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume relative to the 5-day average
    volume_relative = df['volume'] / avg_volume_5d
    
    # Calculate the weighted price change
    weighted_price_change = (df['close'] - df['open']) * (df['volume'] / df['amount'])
    
    # Combine the factors into the final heuristic
    heuristic = (intraday_return + buying_pressure) * volume_relative * weighted_price_change
    
    return heuristic
```
