```python
import pandas as pd

def heuristics_v2(df):
    # Evaluate Intraday Dynamics
    intraday_range = df['high'] - df['low']
    intraday_midpoint_to_open_ratio = ((df['high'] + df['low']) / 2) / df['open']
    adjusted_close_to_open_return = (df['close'] - df['open']) / intraday_range
    
    # Analyze Momentum
    roc_10 = df['close'].pct_change(10)
    roc_30 = df['close'].pct_change(30)
    
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    
    close_to_high_ratio = df['close'] / df['high']
    close_to_low_ratio = df['close'] / df['low']

    # Assess Volatility
    daily_volatility = df['close'].pct_change().std()
    smooth_volatility = df['close'].pct_change().ewm(span=30, adjust=False).std()

    # Combine Metrics
    daily_price_change = df['close'].diff()
    cumulative_price_change = daily_price_change.cumsum()
    
    vwap = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    adjusted_close_to_open_envelope = (df['close'].rolling(window=10).max() - df['open'].rolling(window=10).min())
    
    weighted_intraday_range = intraday_range * vwap
    adjusted_close_to_open_plus_weighted_intraday = adjusted_close_to_open_return + weighted_intraday_range
    
    combined_momentum = cumulative_price_change + roc_30 + (sma_10 - sma_30) + (sma_30 - sma_60)
    
    ema_daily_price_change = daily_price_change.ewm(span=10, adjust=False).mean()
    cumulative_momentum = combined_momentum + ema_daily_price_change
    
    final_factor = adjusted_close_to_open_plus_weighted_intraday + cumulative_momentum
    
    return final_factor
```
