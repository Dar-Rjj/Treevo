import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow index (MFI) which is a volume-weighted relative strength index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Create a ratio of positive to total money flow
    positive_flow_sum = positive_flow.rolling(window=14).sum()
    negative_flow_sum = negative_flow.rolling(window=14).sum()
    money_ratio = positive_flow_sum / (positive_flow_sum + negative_flow_sum)
    
    # MFI is defined as 100 - (100 / (1 + money_ratio))
    mfi = 100 - (100 / (1 + money_ratio))

    # The alpha factor we use here is a variant where we incorporate the amount feature
    # Calculate a normalized amount by dividing by its rolling standard deviation
    normalized_amount = df['amount'] / df['amount'].rolling(window=14).std()
    
    # We combine MFI and our modified amount factor
    alpha_factor = mfi + normalized_amount
    
    return alpha_factor
