```python
def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = data['close'] - data['open']
    
    # Calculate Average Volume over 20 days
    average_volume = data['volume'].rolling(window=20).mean()
    
    # Normalize the Intraday Range by the Average Volume
    normalized_intraday_range = intraday_range / average_volume
    
    # Adjust the Close-to-Open Return by the Normalized Intraday Range
    adjusted_momentum = close_to_open_return * normalized_intraday_range
    
    return adjusted_momentum
```
