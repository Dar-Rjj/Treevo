```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] / df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    df['volume_weighted_volatility'] = df['intraday_range'] / df['volume'].sum()
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    df['combined_adjusted_return'] = df['adjusted_close_to_open_return'] * df['volume_weighted_volatility']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    df['close_to_next_day_open_return'] = df['vwap'].shift(-1) - df['open'].shift(1)
    
    # Calculate Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=10).max()
    df['min_open'] = df['open'].rolling(window=10).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close_to_open_return'] = df['adjusted_close_to_open_return'].rolling(window=10).max()
    df['min_open_envelope'] = df['open'].rolling(window=10).min()
    
    # Calculate Volume-Adjusted Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['intraday_midpoint'] * df['volume']
    
    # Combine Factors
    df['factor'] = df['midpoint_to_open_ratio'] * df['close_to_next_day_open_return']
    
    # Adjustment by Combined Envelopes
    df['factor'] = df['factor'] * df['max_close'] * df['min_open']

    return df['factor']
```
