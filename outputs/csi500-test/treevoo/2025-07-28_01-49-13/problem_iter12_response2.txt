```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_open_ratio = intraday_midpoint / df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']

    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    returns = (vwap - next_day_open) / vwap
    weighted_returns = returns * df['volume']
    weighted_sum_returns = weighted_returns.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()

    # Calculate Adjusted Close-to-Open Envelope
    adjusted_close = df['close'] * (df['volume'] / df['volume'].rolling(window=20).mean())
    max_adjusted_close = adjusted_close.rolling(window=20).max()

    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']) / df['close']
    smoothed_volatility = daily_volatility.ewm(span=20, adjust=False).mean()

    # Combine Factors
    factor = intraday_open_ratio * weighted_sum_returns
    factor *= (max_close - min_open) / (max_adjusted_close - min_open)
    factor /= smoothed_volatility

    return factor
```
