```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['volume_ma_M'] = df['volume'].rolling(window=M).mean()
    df['volume_std_M'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['volume_ma_M'] + 2 * df['volume_std_M'])).astype(int)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] - df['close']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['intraday_range']
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume_momentum']
    df['combined_metrics'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Enhance Combined Metrics
    df['enhanced_combined_metrics'] = df['N_day_return'] * (K * df['volume_spike'] + (1 - df['volume_spike']))
    df['enhanced_combined_metrics'] *= df['intraday_range']
    df['enhanced_combined_metrics'] *= df['adjusted_close_to_open_return']
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum and Volume Change
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    df['cumulative_volume_change'] = df['volume_momentum'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['combined_metrics']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ema_volume_weighted_adjusted_return'] = df['volume_weighted_adjusted_return'].ewm(span=20).mean()
    
    # Final Combination
    df['final_alpha_factor'] = (df['enhanced_combined_metrics'] + 
                                df['combined_cumulative_sums'] + 
                                df['volume_weighted_adjusted_return'])
    df['final_alpha_factor'] /= df['ema_volume_weighted_adjusted_return']
    
    return df['final_alpha_factor']
```
