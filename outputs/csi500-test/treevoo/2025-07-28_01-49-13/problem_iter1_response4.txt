```python
import pandas as pd

def heuristics_v2(df, period=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Close-to-Close Return over a period
    close_to_close_return = (df['close'].shift(-1) / df['close']) - 1
    sum_returns = close_to_close_return.rolling(window=period).sum()
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] * sum_returns
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=period).max()
    min_low = df['low'].rolling(window=period).min()
    high_low_envelope = max_high - min_low
    
    # Combine Factors
    factor = (intraday_range_to_open_ratio * volume_momentum) / high_low_envelope
    
    return factor
```
