```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, momentum_period=10):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-Open Return
    combined_return = (intraday_range + close_to_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = combined_return * df['volume']
    
    # Calculate Moving Average of Adjusted Return
    moving_avg_adjusted_return = volume_weighted_adjusted_return.rolling(window=lookback_period).mean()
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    close_returns = (df['close'].pct_change(periods=momentum_period)).fillna(0)
    volume_momentum = close_returns * df['volume']
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=lookback_period).max()
    min_low = df['low'].rolling(window=lookback_period).min()
    
    # Combine Factors
    factor = (volume_weighted_adjusted_return / moving_avg_adjusted_return) * \
             (intraday_range_to_open_ratio * volume_momentum) * \
             (max_high - min_low)
    
    return factor
```
