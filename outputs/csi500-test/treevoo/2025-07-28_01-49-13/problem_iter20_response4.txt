```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5, lookback_volatility=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=M).mean()
    df['volume_std'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['volume_ma'] + 2 * df['volume_std']))
    
    # Combine Price Momentum and Volume Spike
    df['combined_return'] = df['N_day_return']
    df.loc[df['volume_spike'], 'combined_return'] *= K
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_open_ratio'] = (df['high'] + df['low']) / (2 * df['open'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] / df['intraday_range']
    
    # Calculate Volume-Weighted Volatility
    df['avg_volume'] = df['volume'].rolling(window=lookback_volatility).mean()
    df['volume_weighted_volatility'] = df['intraday_range'] / df['avg_volume']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return over a period
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = (df['vwap'] - df['next_day_open']).fillna(0)
    df['weighted_sum_returns'] = (df['close_to_next_open_return'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Close-to-Open Envelope
    df['max_close_period'] = df['close'].rolling(window=20).max()
    df['min_open_period'] = df['open'].rolling(window=20).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adj_close_period'] = df['close'].rolling(window=20).max()
    df['min_open_period'] = df['open'].rolling(window=20).min()
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['combined_return']
    df['adjusted_combined_metric'] = df['weighted_intraday_range'] + df['adjusted_close_to_open_return']
    
    # Enhance Combined Metrics
    df['enhanced_combined_metric'] = df['adjusted_combined_metric'] * df['intraday_range'] * df['close_to_open_return']
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['cumulative_volume_change'] = df['volume_momentum'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['enhanced_combined_metric']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ema_volume_weighted_adjusted_return'] = df['volume_weighted_adjusted_return'].ewm(span=lookback_volatility).mean()
    
    # Final Combination
    df['final_factor'] = (df['enhanced_combined_metric'] + df['combined_cumulative_sums']) * df['volume_weighted_volatility'] * (df['volume_weighted_adjusted_return'] / df['ema_volume_weighted_adjusted_return'])
    
    return df['final_factor']
```
