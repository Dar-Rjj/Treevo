```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate Daily Price Return
    df['close_N_days_ago'] = df['close'].shift(N)
    df['N_day_return'] = (df['close'] / df['close_N_days_ago']) - 1

    # Identify Volume Spike
    df['volume_MA_M_days'] = df['volume'].rolling(window=M).mean()
    df['volume_STD_M_days'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (df['volume_MA_M_days'] + 2 * df['volume_STD_M_days'])).astype(int)

    # Adjust Daily Price Return for Volume Spike
    df['adjusted_N_day_return'] = df['N_day_return']
    df.loc[df['volume_spike'] == 1, 'adjusted_N_day_return'] *= K

    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)

    # Compute Cumulative Sum of Adjusted Price Return
    df['cumulative_adjusted_price_return'] = df['adjusted_N_day_return'].cumsum()

    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()

    # Combine Cumulative Sums
    df['alpha_factor'] = df['cumulative_adjusted_price_return'] * df['cumulative_volume_change']

    return df['alpha_factor']
```
