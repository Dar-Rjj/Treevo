import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()
    
    # Calculate Close-to-Next Day Open Return
    df['close_to_next_open_return'] = (df['vwap'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Weighted Sum of Returns by Volume
    df['weighted_return'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    df['max_close'] = df['close'].rolling(window=10).max()
    df['min_open'] = df['open'].rolling(window=10).min()
    
    # Introduce Daily Log Return
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    
    # Incorporate Williams %R for Momentum
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    df['williams_r'] = (highest_high - df['close']) / (highest_high - lowest_low)
    
    # Introduce True Range and Average True Range (ATR) for Volatility
    df['true_range'] = (df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1))
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Include Volume Weighted Moving Average (VWMA)
    df['vwma'] = (df['volume'] * df['close']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Calculate Money Flow Index (MFI) for Money Flow
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Combine Factors
    df['adjusted_intraday_factor'] = df['midpoint_open_ratio'] * df['weighted_return']
    df['composite_alpha'] = (
        0.3 * df['adjusted_intraday_factor'] +
        0.2 * df['daily_log_return'] +
        0.1 * (1 - df['williams_r']) +
        0.1 * (1 / df['atr']) +
        0.1 * (1 - (df['mfi'] / 100)) +
        0.2 * (df['close'] - df['vwma'])
    )
    
    # Create Composite Alpha Signal
    threshold = 0.05
    df['alpha_signal'] = df['composite_alpha'].apply(lambda x: 1 if x > threshold else (-1 if x < -threshold else 0))
    
    return df['alpha_signal']
