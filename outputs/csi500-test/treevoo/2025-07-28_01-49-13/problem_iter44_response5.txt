```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the signed volume, positive if the close is higher than the previous close, negative otherwise
    df['signed_volume'] = df['volume'] * (df['close'] > df['close'].shift()).astype(int) * 2 - df['volume']
    
    # Calculate a simple moving average of the high and low prices over a 10-day window
    df['sma_high_10'] = df['high'].rolling(window=10).mean()
    df['sma_low_10'] = df['low'].rolling(window=10).mean()
    
    # Calculate the ratio of the current close to the 10-day SMA of the high
    df['close_to_sma_high_10'] = df['close'] / df['sma_high_10']
    
    # Calculate the difference between the 10-day SMA of the high and the 10-day SMA of the low
    df['sma_high_low_diff_10'] = df['sma_high_10'] - df['sma_low_10']
    
    # The alpha factor is based on the combination of the price change, signed volume, and the ratio of close to 10-day SMA of the high
    alpha_factor = df['price_change'] * df['signed_volume'] * df['close_to_sma_high_10'] / df['sma_high_low_diff_10']
    
    return alpha_factor.dropna()
```
