```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Calculate Volume-Weighted Volatility
    window = 20
    volume_weighted_volatility = intraday_range.rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=window).mean()
    
    # Calculate Close-to-Next Day Open Return over a period
    next_day_open = df['open'].shift(-1)
    close_to_next_open_return = vwap - next_day_open
    weighted_sum_of_returns = close_to_next_open_return * df['volume'].rolling(window=window).mean()
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=window).max()
    min_open = df['open'].rolling(window=window).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close_to_open_return = adjusted_close_to_open_return.rolling(window=window).max()
    
    # Combine Factors
    factor = midpoint_to_open_ratio * weighted_sum_of_returns
    factor = factor * (max_close - min_open) * (max_adjusted_close_to_open_return - min_open)
    
    return factor
```
