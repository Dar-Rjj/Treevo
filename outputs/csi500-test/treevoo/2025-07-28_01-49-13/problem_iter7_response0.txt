```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate Daily Price Return
    close = df['close']
    n_day_return = close.pct_change(periods=N)
    
    # Identify Volume Spike
    volume = df['volume']
    vol_mavg = volume.rolling(window=M).mean()
    vol_std = volume.rolling(window=M).std()
    volume_spike = (volume > (vol_mavg + 2 * vol_std)).astype(int)
    
    # Adjust Daily Price Return for Volume Spike
    adjusted_return = n_day_return.copy()
    adjusted_return[volume_spike == 1] *= K
    
    # Compute Cumulative Sum of Adjusted Price Return
    cum_adjusted_return = adjusted_return.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    daily_vol_change = volume.diff()
    cum_vol_change = daily_vol_change.cumsum()
    
    # Combine Cumulative Sums
    combined_factor = cum_adjusted_return * cum_vol_change
    
    return combined_factor
```
