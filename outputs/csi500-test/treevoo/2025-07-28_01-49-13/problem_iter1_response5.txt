```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    volume_weighted_spread = high_low_spread * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    ema_window = 10
    ema_weighted_spread = volume_weighted_spread.ewm(span=ema_window, adjust=False).mean()
    
    # Momentum Indicator
    momentum_period = 5
    previous_ema = ema_weighted_spread.shift(momentum_period)
    momentum = ema_weighted_spread - previous_ema
    
    return momentum
```
