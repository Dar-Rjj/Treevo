```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low'])

    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']

    # Calculate Close-to-Next Day Open Return
    df['close_to_next_open_return'] = (df['vwap'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)

    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_open_return'] * df['volume']

    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=10).max()
    min_open = df['open'].rolling(window=10).min()

    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=10).max()
    min_adj_open = df['open'].rolling(window=10).min()

    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']).abs()
    smoothed_volatility = daily_volatility.ewm(span=10, adjust=False).mean()

    # Combine Factors
    combined_envelopes = (max_close - min_open) + (max_adj_close - min_adj_open)
    
    # Calculate Intraday Momentum
    intraday_momentum = ((df['high'] - df['low']) / df['open']) + ((df['close'] - df['open']) / df['open'])
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = ((df['high'] - df['low']) * df['volume']) + ((df['close'] - df['open']) * df['volume'])
    
    # Introduce Short-Term Reversal
    short_term_reversal = ((df['open'] - df['close']) / df['open']) + ((df['high'] - df['low']) / df['high'])
    weighted_short_term_reversal = short_term_reversal * df['volume']
    
    # Calculate Long-Term Trend
    long_term_trend_close = df['close'].rolling(window=50).mean()
    long_term_trend_vwap = df['vwap'].rolling(window=50).mean()
    
    # Adjust by Long-Term Trend
    adjusted_long_term_trend = (long_term_trend_close - long_term_trend_vwap) / long_term_trend_vwap
    
    # Final Alpha Factor
    alpha_factor = (
        intraday_momentum * df['weighted_sum_returns'] 
        + volume_weighted_momentum 
        + weighted_short_term_reversal 
        + adjusted_long_term_trend 
        - smoothed_volatility
    )
    
    alpha_factor = alpha_factor / combined_envelopes

    return alpha_factor
```
