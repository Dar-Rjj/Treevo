import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    The function takes as input a pandas DataFrame with (open, high, low, close, amount, volume) and
    produces a pandas Series representing the factor values.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume)
    
    Returns:
    pd.Series: Factor values indexed by date
    """
    
    # Calculate the relative strength index (RSI) of the close price
    def compute_rsi(series, period=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    # Calculate the average true range (ATR) of the prices
    def compute_atr(high, low, close, period=14):
        tr = pd.concat([
            (high - low),
            (high - close.shift()).abs(),
            (low - close.shift()).abs()
        ], axis=1).max(axis=1)
        atr = tr.rolling(window=period).mean()
        return atr
    
    # Calculate the RSI of the close price
    rsi_close = compute_rsi(df['close'])
    
    # Calculate the ATR of the prices
    atr = compute_atr(df['high'], df['low'], df['close'])
    
    # Calculate the factor value
    factor_value = (rsi_close - 50) * (atr / df['close']).shift(1)
    
    return factor_value
