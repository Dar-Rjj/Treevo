```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change of the close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of volume to amount, which can be seen as a proxy for the average price
    vol_to_amt_ratio = df['volume'] / df['amount']
    
    # Combine the above metrics into a single alpha factor
    # The idea is that a large price range might indicate volatility, 
    # while a positive close price change and a high volume-to-amount ratio might indicate strong buying pressure.
    alpha_factor = (price_range * close_change * vol_to_amt_ratio).shift(1)  # Shift to avoid lookahead bias
    
    return alpha_factor
```
