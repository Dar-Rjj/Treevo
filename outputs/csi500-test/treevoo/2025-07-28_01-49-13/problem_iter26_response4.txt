def heuristics_v2(df):
    """
    Generate an alpha factor that could help predict future stock returns.
    
    The function takes a pandas DataFrame with the following columns:
    - open: Opening price of the day
    - high: Highest price of the day
    - low: Lowest price of the day
    - close: Closing price of the day
    - amount: Total transaction amount for the day
    - volume: Total volume traded for the day
    
    The function returns a pandas Series with the index as (date) and the values as the factor.
    """
    
    # Calculate the range of the day
    range_day = df['high'] - df['low']
    
    # Calculate the percentage change from open to close
    pct_change_open_close = (df['close'] - df['open']) / df['open']
    
    # Calculate the average transaction size
    avg_transaction_size = df['amount'] / df['volume']
    
    # Calculate the factor: momentum weighted by the average transaction size
    factor = (pct_change_open_close * df['volume']) / avg_transaction_size
    
    return factor
