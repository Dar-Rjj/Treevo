```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adjusted_close_to_open_return'] = df['intraday_range'] * df['close_to_open_return']
    
    # Calculate Volume-Weighted Volatility
    df['volume_weighted_volatility'] = (df['intraday_range'] / df['volume']).rolling(window=20).mean() * df['volume_change']
    
    # Incorporate Volume Trend
    df['incorporated_volume_trend'] = df['volume_trend'] * df['volume_weighted_volatility']
    
    # Calculate Daily Price Return
    N = 5
    df['daily_price_return'] = df['close'] - df['close'].shift(N)
    
    # Identify Volume Spike
    M = 20
    df['volume_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['volume_std'] = df['volume'].rolling(window=M).std()
    df['volume_spike'] = df['volume'] > (df['volume_moving_avg'] + 2 * df['volume_std'])
    
    # Adjust Daily Price Return for Volume Spike
    K = 1.5
    df['adjusted_daily_price_return'] = np.where(df['volume_spike'], df['daily_price_return'] * K, df['daily_price_return'])
    
    # Compute Cumulative Sum of Adjusted Price Return
    df['cumulative_adjusted_price_return'] = df['adjusted_daily_price_return'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_factors'] = df['cumulative_adjusted_price_return'] * df['cumulative_volume_change']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    
    # Calculate Close-to-Next Day Open Return over a period
    df['vwap_next_day_open_return'] = (df['close'] - df['open'].shift(-1)) * df['volume']
    df['weighted_sum_returns'] = df['vwap_next_day_open_return'].rolling(window=20).sum()
    
    # Calculate Close-to-Open Envelope
    window = 20
    df['max_close'] = df['close'].rolling(window=window).max()
    df['min_open'] = df['open'].rolling(window=window).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['adjusted_close'] = df['close'] * df['intraday_midpoint_to_open_ratio']
    df['max_adjusted_close'] = df['adjusted_close'].rolling(window=window).max()
    
    # Combine Factors
    df['combined_envelopes'] = (df['max_close'] / df['min_open']) * (df['max_adjusted_close'] / df['min_open'])
    df['factor'] = df['intraday_midpoint_to_open_ratio'] * df['weighted_sum_returns'] * df['combined_envelopes'] * df['combined_cumulative_factors']
    
    return df['factor']
```
