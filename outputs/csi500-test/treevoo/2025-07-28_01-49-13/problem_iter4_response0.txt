```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    weighted_high_low_spread = high_low_spread * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread with a 10-day window
    ema_window = 10
    weighted_ema = weighted_high_low_spread.ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Momentum: Current EMA - EMA from 5 days ago
    momentum_window = 5
    momentum_indicator = weighted_ema - weighted_ema.shift(momentum_window)
    
    return momentum_indicator
```
