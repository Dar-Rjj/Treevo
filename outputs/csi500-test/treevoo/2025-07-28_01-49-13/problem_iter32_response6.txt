```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from open to close
    price_movement = df['close'] - df['open']
    
    # Calculate the buying pressure (amount / volume)
    buying_pressure = df['amount'] / df['volume']
    
    # Calculate the normalized buying pressure using the high and low of the day
    range_based_buying_pressure = buying_pressure * (df['high'] - df['low'])
    
    # Combine the price movement with the range-based buying pressure
    alpha_factor = price_movement + range_based_buying_pressure
    
    return alpha_factor
```
