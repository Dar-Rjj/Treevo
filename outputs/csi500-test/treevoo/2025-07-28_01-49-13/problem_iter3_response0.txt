```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate the 20-day average volume and amount
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    
    # Compute the final factor
    factor_value = (close_to_open_return * intraday_range) / (avg_volume * avg_amount)
    
    return factor_value
```
