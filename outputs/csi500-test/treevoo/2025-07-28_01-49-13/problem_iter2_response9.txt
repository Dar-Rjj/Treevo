```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-Open Return
    combined_return = (high_low_range + close_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_return = combined_return * df['volume']
    
    # Calculate Moving Average of Adjusted Return
    lookback_period = 20
    moving_average = volume_weighted_return.rolling(window=lookback_period).mean()
    
    # Calculate Price Change Momentum
    current_avg_price = (df['high'] + df['low']) / 2
    previous_avg_price = (df['high'].shift(1) + df['low'].shift(1)) / 2
    price_change_momentum = (current_avg_price - previous_avg_price) * high_low_range
    
    # Final Factor Calculation
    final_factor = (volume_weighted_return - moving_average) * price_change_momentum
    
    return final_factor
```
