```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['midpoint'] / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 5
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    df['roc'] = df['close'].pct_change(periods=short_period)
    df['macd_line'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], 
                                                               abs(x['high'] - df['close'].shift(1)), 
                                                               abs(x['low'] - df['close'].shift(1))), axis=1)
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).mean()
    
    # Measure Standard Deviation of Price Changes
    df['std_dev'] = df['close'].rolling(window=short_period).std()
    
    # Calculate Close-to-Next Day Open Return
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = df['vwap'] / df['next_day_open'] - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_return'] = df['close_to_next_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 10
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['adjusted_close'] = df['close'] * (df['volume'] / df['volume'].rolling(window=envelope_period).mean())
    df['max_adj_close'] = df['adjusted_close'].rolling(window=envelope_period).max()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = df['high'] - df['low']
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=envelope_period, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Final Factor Combination
    df['factor'] = (df['midpoint_to_open_ratio'] * df['weighted_return']) \
                   * (df['max_close'] - df['min_open']) \
                   * (df['max_adj_close'] - df['min_open']) \
                   * np.sign(df['obv'].diff()) \
                   * (df['momentum'] + df['atr'] + df['std_dev'] * df['volume'])
    
    # Integrate Volatility Factor
    df['final_factor'] = df['factor'] * (df['daily_volatility'] + df['smoothed_volatility'])
    
    return df['final_factor'].dropna()
```
