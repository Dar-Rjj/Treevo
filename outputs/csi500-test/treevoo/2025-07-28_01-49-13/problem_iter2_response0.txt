```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    df['cum_price_momentum'] = df['price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cum_volume_change'] = df['volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cum_sums'] = df['cum_price_momentum'] * df['cum_volume_change']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    df['range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume Momentum
    df['returns'] = (df['close'] / df['close'].shift(1)) - 1
    df['volume_momentum'] = df['volume'] * df['returns']
    
    # Calculate High-Low Envelope
    period = 5  # Example period, can be adjusted
    df['max_high'] = df['high'].rolling(window=period).max()
    df['min_low'] = df['low'].rolling(window=period).min()
    
    # Combine All Factors
    df['factor'] = (df['range_to_open_ratio'] * df['volume_momentum']) * (df['max_high'] / df['min_low'])
    df['factor'] *= df['combined_cum_sums']
    
    return df['factor']
```
