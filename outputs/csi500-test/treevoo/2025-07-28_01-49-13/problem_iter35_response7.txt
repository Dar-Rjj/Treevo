```python
import pandas as pd

def heuristics_v2(df):
    # Evaluate Intraday and Volume-Weighted Dynamics
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Midpoint_to_Open_Ratio'] = (df['high'] + df['low']) / 2 / df['open']
    df['VWAP'] = df['amount'] / df['volume']
    df['Adjusted_Close_to_Open_Return'] = (df['close'] - df['open']) / df['Intraday_Range']

    # Analyze Trend and Momentum
    df['ROC_5'] = df['close'].pct_change(5)
    df['ROC_20'] = df['close'].pct_change(20)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_Crossover_5_20'] = df['SMA_5'] > df['SMA_20']
    df['SMA_Crossover_20_50'] = df['SMA_20'] > df['SMA_50']
    df['Close_to_Open'] = df['close'] - df['open']
    df['Close_to_High'] = df['close'] / df['high']
    df['Close_to_Low'] = df['close'] / df['low']

    # Introduce Volatility and Volume Factors
    df['Volatility_Adjusted_Return'] = df['close'].pct_change() / df['close'].rolling(window=20).std()
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['High_Volume_Days'] = df['Volume_Ratio'] > 1.5
    df['Low_Volume_Days'] = df['Volume_Ratio'] < 0.5

    # Combine Intraday and Cumulative Metrics
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Cumulative_Price_Change'] = df['Daily_Price_Change'].cumsum()
    df['Close_to_Next_Open_Return'] = df['VWAP'] - df['open'].shift(-1)
    df['Weighted_Sum_of_Returns'] = df['Close_to_Next_Open_Return'] * df['volume']
    df['Max_Adjusted_Close'] = df['close'].rolling(window=20).max()
    df['Min_Open'] = df['open'].rolling(window=20).min()
    df['Daily_Volatility'] = (df['high'] - df['low']) / 2
    df['Smooth_Volatility'] = df['Daily_Volatility'].ewm(span=20, adjust=False).mean()

    # Final Factor Adjustment
    df['Factor'] = df['Intraday_Midpoint_to_Open_Ratio'] * df['Weighted_Sum_of_Returns']
    df['Factor'] = df['Factor'] * (df['Max_Adjusted_Close'] - df['Min_Open'])
    df['Factor'] += df['Adjusted_Close_to_Open_Return']

    # Compute Cumulative Price Momentum
    df['Cumulative_Price_Momentum'] = df['Daily_Price_Change'].ewm(span=20, adjust=False).mean()
    df['Cumulative_Volume'] = df['volume'].cumsum()
    df['Cumulative_Price_Momentum_Adjusted'] = df['Cumulative_Price_Momentum'] / df['Cumulative_Volume']

    return df['Factor']
```
