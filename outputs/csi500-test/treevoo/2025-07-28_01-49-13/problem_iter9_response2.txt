```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate 20-day Moving Average of Volume
    vol_20_day_ma = df['volume'].rolling(window=20).mean()
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - vol_20_day_ma
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = (intraday_range * close_to_open_return) / intraday_range.mean()
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / (avg_volume * avg_amount))
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    factor = adjusted_close_to_open_return * (volume_weighted_volatility * volume_change) * volume_trend
    
    return factor
```
