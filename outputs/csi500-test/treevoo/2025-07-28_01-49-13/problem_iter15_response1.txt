```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint_to_open_ratio = ((df['high'] + df['low']) / 2) / df['open']

    # Calculate Volume Weighted Momentum
    close_to_next_day_open_return = (df['close'].shift(-1) / df['close']).shift(1)
    volume_weighted_momentum = (close_to_next_day_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    adjusted_close_to_open_envelope = max_adjusted_close / min_open

    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']

    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = (df['close'] / df['close'].shift(1)).rolling(window=5).sum()
    volume_momentum = daily_volume_change * close_to_close_return

    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    high_low_envelope = max_high / min_low

    # Combine Intraday Factors
    combined_intraday_factors = (intraday_midpoint_to_open_ratio * volume_weighted_momentum) / adjusted_close_to_open_envelope

    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    adjusted_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) / high_low_envelope

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open']

    # Combine High-Low Range and Close-Open Return
    combined_high_low_range_and_close_open_return = intraday_range * close_to_open_return

    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_range_and_close_open_return

    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2) * intraday_range

    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_indicator = (daily_volume_change > daily_volume_change.rolling(window=5).std()).astype(int)

    # Final Factor Calculation
    final_factor = volume_weighted_adjusted_return * price_change_momentum * volume_spike_indicator

    # Calculate Moving Average Convergence Divergence (MACD)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_ema - long_ema

    # Integrate MACD into Final Factor
    final_factor = final_factor * macd

    # Combine Final Intraday and Cumulative Factors
    alpha_factor = combined_intraday_factors * adjusted_intraday_factors * final_factor

    return alpha_factor
```
