```python
def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = data['close'] - data['open']
    
    # Calculate Volume Change
    volume_change = data['volume'] - data['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_return = close_to_open_return * intraday_range
    
    # Combine Adjusted Return and Volume Change
    factor_values = adjusted_return * volume_change
    
    return factor_values
```
