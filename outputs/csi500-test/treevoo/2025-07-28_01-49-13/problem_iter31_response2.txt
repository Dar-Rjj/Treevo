import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    sma_short = df['close'].rolling(window=5).mean()
    short_term_momentum = df['close'] - sma_short
    
    # Volatility Indicators
    true_range = pd.concat([
        (df['high'] - df['low']),
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).sum()
    
    # On-Balance Volume (OBV) Calculation
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Momentum-Volatility Ratio
    momentum_volatility_ratio = short_term_momentum / atr
    
    # OBV Trend Confirmation
    close_trend = df['close'] - df['close'].shift(1)
    obv_trend = df['obv'] - df['obv'].shift(1)
    trend_confirmation = (close_trend * obv_trend > 0).astype(int)
    
    # Enhanced Alpha Factor
    alpha_factor = momentum_volatility_ratio * trend_confirmation
    
    return alpha_factor
