```python
import pandas as pd

def heuristics(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_open_ratio = intraday_range / df['open']

    # Calculate Volume-Weighted Close-to-Close Return
    volume_sum = df['volume'].rolling(window=5).sum()
    close_returns = (df['close'] / df['close'].shift(1)) - 1
    volume_weighted_returns = (close_returns * df['volume']) / volume_sum
    
    # Calculate Enhanced High-Low Envelope
    max_high = df['high'].rolling(window=20).max()
    min_low = df['low'].rolling(window=20).min()
    true_range = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    average_true_range = true_range.rolling(window=20).mean()
    high_envelope = max_high + average_true_range
    low_envelope = min_low - average_true_range
    
    # Combine Factors
    factor = (intraday_open_ratio * volume_weighted_returns) / ((high_envelope - low_envelope) / 2)
    
    return factor
```
