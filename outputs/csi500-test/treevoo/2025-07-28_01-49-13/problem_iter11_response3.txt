```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, K=1.5, ema_span=10):
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate N-day Price Momentum
    df['N_day_price_momentum'] = (df['close'] - df['close'].shift(N)) / df['close'].shift(N)
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Identify Volume Spike
    rolling_volume_mean = df['volume'].rolling(window=M).mean()
    rolling_volume_std = df['volume'].rolling(window=M).std()
    df['volume_spike'] = (df['volume'] > (rolling_volume_mean + 2 * rolling_volume_std)).astype(int)
    
    # Compute Cumulative Sum of Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_momentum'].cumsum()
    
    # Compute Cumulative Sum of Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative_sums'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust Combined Cumulative Sums for Volume Spikes
    df['adjusted_combined_cumulative_sums'] = df['combined_cumulative_sums'] * (1 + (df['volume_spike'] * (K - 1)))
    
    # Integrate Intraday Volatility and Open-Close Dynamics
    df['integrated_return'] = (df['adjusted_combined_cumulative_sums'] * df['intraday_range']) * df['close_to_open_return']
    
    # Calculate Volume Weighted Integrated Return
    df['volume_weighted_integrated_return'] = df['integrated_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume Weighted Integrated Return
    df['ema_volume_weighted_integrated_return'] = df['volume_weighted_integrated_return'].ewm(span=ema_span).mean()
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_integrated_return'] / df['ema_volume_weighted_integrated_return']
    
    return df['alpha_factor']
```
