import pandas as pd

def heuristics_v2(df, n=10, m=20, k=1.5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].diff()
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return * intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * volume_momentum
    combined_metrics = weighted_intraday_range + close_to_open_return
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Cumulative Price Momentum
    cumulative_price_momentum = daily_price_change.cumsum()
    
    # Compute Cumulative Volume Change
    cumulative_volume_change = volume_momentum.cumsum()
    
    # Calculate N-day Price Return
    n_day_price_return = df['close'].pct_change(n)
    
    # Identify Volume Spike
    volume_ma = df['volume'].rolling(window=m).mean()
    volume_std = df['volume'].rolling(window=m).std()
    volume_spike = df['volume'] > (volume_ma + 2 * volume_std)
    
    # Adjust N-day Price Return for Volume Spike
    adjusted_n_day_price_return = n_day_price_return.copy()
    adjusted_n_day_price_return[volume_spike] *= k
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_price_return = adjusted_n_day_price_return.cumsum()
    
    # Combine Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    volume_weighted_volatility = (intraday_range / avg_volume) * avg_amount
    
    # Adjust by Volume-Weighted Volatility
    adjusted_combined_metrics = combined_metrics * volume_weighted_volatility
    
    # Final Combination
    final_alpha_factor = adjusted_combined_metrics + combined_cumulative_sums
    
    return final_alpha_factor
