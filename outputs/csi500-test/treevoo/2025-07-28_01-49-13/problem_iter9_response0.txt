```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, K=1.5, lookback=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['vol_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['vol_std'] = df['volume'].rolling(window=M).std()
    df['is_vol_spike'] = (df['volume'] > (df['vol_moving_avg'] + 2 * df['vol_std'])).astype(int)
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Price Momentum, Volume Spike, and Intraday Volatility
    df['adjusted_return'] = df['N_day_return'] * np.where(df['is_vol_spike'], K, 1)
    df['combined_return'] = df['adjusted_return'] * df['intraday_range'] * df['close_to_open_return']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_return'] = df['combined_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ema_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=lookback, min_periods=lookback).mean()
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['ema_volume_weighted_return']
    
    return df['alpha_factor']
```
