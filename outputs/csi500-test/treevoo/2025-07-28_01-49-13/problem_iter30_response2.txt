```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_day_open_return = (vwap - df['open'].shift(-1)) / vwap
    
    # Weighted Sum of Returns by Volume
    weighted_sum_returns = close_to_next_day_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=10).max()
    min_open = df['open'].rolling(window=10).min()
    
    # Introduce Volatility Factor
    daily_volatility = (df['high'] - df['low']) / df['close']
    smoothed_volatility = daily_volatility.ewm(span=10).mean()
    
    # Combine Factors
    combined_factors = midpoint_to_open_ratio * weighted_sum_returns
    combined_factors = combined_factors * (max_adj_close - min_open) / (max_adj_close + min_open)
    combined_factors = combined_factors * (smoothed_volatility / daily_volatility)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_next_day_open_return / intraday_range
    
    # Combine Metrics
    weighted_intraday_range = intraday_range * weighted_sum_returns
    adjusted_close_to_open = adjusted_close_to_open_return + weighted_intraday_range
    
    # Final Factor Adjustment
    final_factor = combined_factors + adjusted_close_to_open + cumulative_price_change
    
    # Compute Cumulative Price Momentum
    cumulative_momentum = daily_price_change.ewm(span=10).mean()
    
    # Incorporate Short-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5).mean()
    long_term_ema = df['close'].ewm(span=20).mean()
    ema_difference = short_term_ema - long_term_ema
    
    # Synthesize Composite Alpha Factor
    composite_alpha_factor = final_factor + cumulative_momentum + ema_difference
    
    # Introduce Trend Reversal Indicator
    trend_reversal_indicator = ema_difference
    composite_alpha_factor = composite_alpha_factor * (1 + trend_reversal_indicator)
    
    return composite_alpha_factor
```
