```python
def heuristics_v2(df):
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the relative strength compared to the VWAP
    relative_strength = (df['close'] - vwap) / vwap
    
    # Combine the factors into a single alpha factor
    alpha_factor = (buying_pressure - selling_pressure) * relative_strength
    
    return alpha_factor
```
