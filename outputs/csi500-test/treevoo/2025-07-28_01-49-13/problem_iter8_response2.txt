```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Price Momentum
    df['cumulative_price_momentum'] = df['daily_price_change'].cumsum()
    
    # Compute Cumulative Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Calculate N-day Price Return
    df['N_day_return'] = (df['close'] / df['close'].shift(N)) - 1
    
    # Identify Volume Spike
    df['rolling_mean_volume'] = df['volume'].rolling(window=M).mean()
    df['rolling_std_volume'] = df['volume'].rolling(window=M).std()
    df['is_volume_spike'] = df['volume'] > (df['rolling_mean_volume'] + 2 * df['rolling_std_volume'])
    
    # Adjust N-day Price Return for Volume Spike
    df['adjusted_N_day_return'] = np.where(df['is_volume_spike'], df['N_day_return'] * K, df['N_day_return'])
    
    # Compute Cumulative Sum of Adjusted Price Return
    df['cumulative_adjusted_price_return'] = df['adjusted_N_day_return'].cumsum()
    
    # Combine Cumulative Sums
    df['combined_cumulative'] = df['cumulative_price_momentum'] * df['cumulative_volume_change']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_cumulative'] * df['cumulative_adjusted_price_return']
    
    return df['alpha_factor']
```
