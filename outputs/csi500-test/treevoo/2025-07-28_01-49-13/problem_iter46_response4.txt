import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] * 1000) / df['volume']  # Assuming amount is in thousands

    # Calculate Short-Term Momentum
    short_period = 5
    df['sma_short'] = df['close'].rolling(window=short_period).mean()

    df['true_range'] = df[['high' - 'low', 
                           abs(df['high'] - df['close'].shift(1)), 
                           abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).sum()

    # Measure Standard Deviation of Price Changes
    std_period = 20
    df['std_price_changes'] = df['close'].diff().rolling(window=std_period).std()

    # Close-to-Next Day Open Return
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_day_open_return'] = (df['next_day_open'] - df['vwap']) / df['vwap']

    # Weighted Sum of Returns by Volume
    df['weighted_returns'] = df['close_to_next_day_open_return'] * df['volume']

    # Close-to-Open Envelope
    envelope_period = 20
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()

    # Adjusted Close-to-Open Envelope
    df['adjusted_close'] = df['close'] * df['volume'] / df['volume'].rolling(window=envelope_period).mean()
    df['max_adjusted_close'] = df['adjusted_close'].rolling(window=envelope_period).max()

    # Daily Volatility
    df['daily_volatility'] = df['high'] - df['low']

    # Smooth Volatility
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=20, adjust=False).mean()

    # On-Balance Volume (OBV) Calculation
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()

    # Final Factor Combination
    df['intraday_momentum_weighted_returns'] = df['midpoint_to_open_ratio'] * df['weighted_returns']
    df['envelope_factor'] = (df['max_close'] + df['min_open'] + df['max_adjusted_close'] + df['min_open']) / 4
    df['volatility_factor'] = df['daily_volatility'] * df['smoothed_volatility']

    # OBV Trend Confirmation
    df['obv_trend'] = np.sign(df['obv'].diff())
    df['close_trend'] = np.sign(df['close'].diff())
    df['obv_trend_confirmation'] = (df['obv_trend'] == df['close_trend']).astype(int)

    # Composite Alpha Factors
    df['composite_alpha'] = (df['intraday_momentum_weighted_returns'] * 0.4 +
                             df['envelope_factor'] * 0.3 +
                             df['volatility_factor'] * 0.2 +
                             df['obv_trend_confirmation'] * 0.1)

    return df['composite_alpha'].dropna()
