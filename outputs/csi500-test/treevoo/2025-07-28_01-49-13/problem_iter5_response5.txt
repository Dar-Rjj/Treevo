```python
def heuristics_v2(data):
    # Calculate Daily Price Momentum
    daily_price_momentum = data['close'] - data['close'].shift(1)
    
    # Calculate Daily Volume Change
    daily_volume_change = data['volume'] - data['volume'].shift(1)
    
    # Compute Cumulative Sum of Price Momentum
    cumulative_price_momentum = daily_price_momentum.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Introduce Weighted Average of Price and Volume
    weighted_price_momentum = cumulative_price_momentum * 0.5
    weighted_volume_change = cumulative_volume_change * 0.5
    
    # Combine Weighted Sums
    combined_factor = weighted_price_momentum + weighted_volume_change
    
    return combined_factor
```
