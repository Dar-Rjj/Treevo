```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']

    # Calculate Volume Weighted Momentum
    close_returns = df['close'].shift(-1) / df['open'].shift(-1) - 1
    volume_weighted_momentum = (close_returns * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    adjusted_envelope = (max_adj_close - df['open']) / (max_adj_close - min_open)

    # Combine Factors
    factor = (midpoint_to_open_ratio * volume_weighted_momentum) * adjusted_envelope
    
    return factor
```
