```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    close_returns = df['close'].pct_change()
    volume_momentum = (close_returns * df['volume']).rolling(window=10).sum()
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=10).max()
    min_low = df['low'].rolling(window=10).min()
    high_low_envelope = max_high - min_low
    
    # Combine Intraday and Volume Factors
    combined_factor = intraday_range_to_open_ratio * volume_momentum / high_low_envelope
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    weighted_high_low_spread = high_low_spread * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    ema_window = 10
    weighted_ema = weighted_high_low_spread.ewm(span=ema_window, adjust=False).mean()
    
    # Momentum Indicator
    previous_ema = weighted_ema.shift(5)
    momentum_indicator = weighted_ema - previous_ema
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + momentum_indicator
    
    return final_alpha_factor
```
