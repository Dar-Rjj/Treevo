```python
import pandas as pd

def heuristics_v2(df, period=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Volume Weighted Momentum
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['volume_weighted_momentum'] = (df['close_to_next_open_return'] * df['volume']).rolling(window=period).sum() / df['volume'].rolling(window=period).sum()

    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=period).max()
    df['min_open'] = df['open'].rolling(window=period).min()

    # Calculate Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']

    # Calculate Volume Momentum
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_momentum'] = (df['close_to_close_return'] * df['daily_volume_change']).rolling(window=period).sum() / df['daily_volume_change'].rolling(window=period).sum()

    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_range_to_open_ratio'] * df['volume_weighted_momentum']
    df['combined_intraday_factors'] *= (df['max_adjusted_close'] - df['min_open'])

    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    df['high_low_envelope_max'] = df['high'].rolling(window=period).max()
    df['high_low_envelope_min'] = df['low'].rolling(window=period).min()
    df['combined_intraday_factors'] *= df['intraday_range_to_open_ratio'] * df['volume_momentum']
    df['combined_intraday_factors'] /= (df['high_low_envelope_max'] - df['high_low_envelope_min'])

    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['volume'] * df['combined_intraday_factors']

    # Calculate Price Change Momentum
    df['price_change_momentum'] = (df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2
    df['price_change_momentum'] *= df['intraday_range']

    # Detect Volume Spike
    volume_spike_threshold = df['volume'].rolling(window=period).mean() * 2
    df['volume_spike_indicator'] = (df['volume'] > volume_spike_threshold).astype(int)

    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_adjusted_return'] * df['price_change_momentum'] * df['volume_spike_indicator']

    return df['final_factor']
```
