import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=30, k_factor=1.5, ema_window=10):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    df['Weighted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    df['EMA_Weighted_High_Low_Spread'] = df['Weighted_High_Low_Spread'].ewm(span=ema_window).mean()
    
    # Calculate N-day Price Momentum
    df['N_Day_Return'] = df['close'].pct_change(periods=n_days)
    
    # Calculate Intraday Volatility
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    df['Close_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['Intraday_Volatility'] = df['Intraday_High_Low_Range'] + df['Close_Open_Return']
    
    # Identify Volume Spike
    df['Volume_Moving_Average'] = df['volume'].rolling(window=m_days).mean()
    df['Volume_Std'] = df['volume'].rolling(window=m_days).std()
    df['Volume_Spike'] = (df['volume'] > (df['Volume_Moving_Average'] + 2 * df['Volume_Std']))
    
    # Combine Price Momentum, Intraday Volatility, and Volume Spike
    df['Combined_Return'] = df['N_Day_Return'] * np.where(df['Volume_Spike'], k_factor, 1) * df['Intraday_Volatility']
    
    # Calculate Volume-Weighted Adjusted Return
    df['Volume_Weighted_Adjusted_Return'] = df['Combined_Return'] * df['volume']
    
    # Calculate EMA of Adjusted Return
    df['EMA_Adjusted_Return'] = df['Volume_Weighted_Adjusted_Return'].ewm(span=ema_window).mean()
    
    # Calculate Final Momentum Indicator
    df['Final_Momentum_Indicator'] = df['Volume_Weighted_Adjusted_Return'] - df['EMA_Adjusted_Return']
    
    return df['Final_Momentum_Indicator']
