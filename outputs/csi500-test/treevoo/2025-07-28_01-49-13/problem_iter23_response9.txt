```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'].shift(-1) - 1
    volume_weighted_momentum = (close_to_next_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_momentum = (close_to_close_return * daily_volume_change).rolling(window=5).sum()
    
    # Combine Intraday Factors
    intraday_factors = intraday_midpoint_open_ratio * volume_weighted_momentum
    adjusted_intraday_factors = intraday_factors * (max_adjusted_close / min_open)
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    high_low_envelope_max = df['high'].rolling(window=5).max()
    high_low_envelope_min = df['low'].rolling(window=5).min()
    adjusted_high_low_envelope = high_low_envelope_max / high_low_envelope_min
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Combine High-Low Range and Close-Open Return
    combined_high_low_close_open = intraday_range * close_to_open_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_close_open
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2) * intraday_range
    
    # Detect Volume Spike
    volume_spike = (df['volume'] - df['volume'].shift(1)) > df['volume'].rolling(window=5).std() * 2
    
    # Introduce New Factor: Intraday Reversal Signal
    intraday_reversal = (df['high'] - df['open']) / (df['high'] - df['low'])
    intraday_reversal_factor = np.where(intraday_reversal > 0.5, 0.8, 1.0)
    
    # Integrate MACD into Final Factor
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_ema - long_ema
    
    # Incorporate Volatility into Final Factor
    volatility = (df['high'] - df['low']) / df['low']
    
    # Final Factor Calculation
    final_factor = (volume_weighted_adjusted_return * price_change_momentum * volume_spike * intraday_reversal_factor * macd) / volatility
    
    return final_factor
```
