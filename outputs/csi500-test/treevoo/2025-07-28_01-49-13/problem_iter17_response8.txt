import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Momentum
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_open_return'] = (df['next_day_open'] - df['close']) / df['close']
    df['volume_weighted_momentum'] = (df['close_to_next_open_return'] * df['volume']).rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    df['adj_close_open_envelope'] = max_adj_close - min_open
    
    # Calculate Daily Price Change and Daily Volume Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Price and Volume Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Adjust Close-to-Open Return by Intraday Range
    df['adj_close_to_open_return'] = df['close_to_next_open_return'] / df['intraday_range']
    
    # Combine Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['volume_weighted_momentum']
    df['adjusted_factor'] = df['weighted_intraday_range'] + df['adj_close_to_open_return']
    
    # Compute Cumulative Price and Volume Momentum
    df['cumulative_price_momentum'] = df['daily_price_change'].ewm(span=5, adjust=False).mean()
    df['cumulative_volume_momentum'] = df['daily_volume_change'].ewm(span=5, adjust=False).mean()
    
    # Final Combination
    df['combined_momentum'] = df['cumulative_price_momentum'] * df['cumulative_volume_momentum']
    df['final_alpha_factor'] = df['adjusted_factor'] + df['combined_momentum']
    
    return df['final_alpha_factor'].dropna()
