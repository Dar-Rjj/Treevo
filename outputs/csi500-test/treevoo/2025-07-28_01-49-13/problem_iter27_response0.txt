```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price range
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate close to open return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate average volume over N days, N=10 for example
    df['Avg_Volume_10D'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Price Trend (VPT)
    df['VPT'] = (df['volume'] * ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))).cumsum()
    
    # Calculate daily transaction amount
    df['Transaction_Amount'] = df['close'] * df['volume']
    
    # Calculate change in transaction amount
    df['Change_in_Transaction_Amount'] = (df['Transaction_Amount'] - df['Transaction_Amount'].shift(1)) / df['Transaction_Amount'].shift(1)
    
    # Combine price and volume signals: (Price Signal) * (Volume Signal)
    df['Price_Signal'] = df['Close_to_Open_Return']  # Example of a price signal
    df['Volume_Signal'] = df['VPT']  # Example of a volume signal
    df['Composite_Factor'] = df['Price_Signal'] * df['Volume_Signal']
    
    # Return the composite factor as a pandas Series
    return df['Composite_Factor']
```
