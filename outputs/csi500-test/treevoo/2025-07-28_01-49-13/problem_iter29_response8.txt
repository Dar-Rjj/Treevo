```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series with the generated alpha factor, indexed by date.
    """
    # Calculate the relative difference between high and low prices
    rel_diff_high_low = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative difference between close and open prices
    rel_diff_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate the weighted volume using the amount
    weighted_volume = df['volume'] * df['amount']
    
    # Alpha factor: ratio of relative price differences to the weighted volume
    alpha_factor = (rel_diff_high_low + rel_diff_close_open) / weighted_volume
    
    return alpha_factor
```
