```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'].shift(-1)
    volume_weighted_momentum = df['volume'] * close_to_next_open_return
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    adjusted_close_to_open_envelope = max_adjusted_close / min_open
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1)
    volume_momentum = volume_change * close_to_close_return
    
    # Combine Intraday Factors
    combined_intraday_factors = (midpoint_to_open_ratio * volume_weighted_momentum) / adjusted_close_to_open_envelope
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    high_low_envelope = max_high / min_low
    adjusted_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) / high_low_envelope
    
    # Combine High-Low Range and Close-Open Return
    close_to_open_return = df['close'] / df['open']
    combined_high_low_and_close_open = intraday_range * close_to_open_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_and_close_open
    
    # Calculate Price Change Momentum
    price_change_momentum = (df['high'] + df['low']) / 2 - (df['high'] + df['low']).shift(1) / 2
    price_change_momentum *= intraday_range
    
    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (daily_volume_change > 0.5 * df['volume'].mean()).astype(int)
    
    # Final Factor Calculation
    final_factor = volume_weighted_adjusted_return * price_change_momentum * (1 + volume_spike)
    
    # Integrate MACD into Final Factor
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_term_ema - long_term_ema
    final_factor *= macd
    
    return final_factor
```
