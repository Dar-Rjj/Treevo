```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] - df['open']
    df['pct_change'] = df['close'] / df['open'] - 1
    
    # Use volume data
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Analyze price patterns
    df['up_day'] = (df['close'] > df['open']).astype(int)
    df['down_day'] = (df['close'] <= df['open']).astype(int)
    df['up_ratio'] = df['up_day'].rolling(window=5).sum() / 5
    
    # Consider high and low prices
    df['range'] = df['high'] - df['low']
    df['range_ratio'] = df['range'] / df['close']
    
    # Incorporate amount data
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_5d']
    
    # Introduce new factors
    df['ma_20d'] = df['close'].rolling(window=20).mean()
    df['deviation_from_ma'] = df['close'] - df['ma_20d']
    df['cumulative_return_1m'] = df['close'].pct_change(periods=20).shift(-20)
    df['volatility_10d'] = df['pct_change'].rolling(window=10).std()
    df['momentum_10d'] = df['close'] / df['close'].shift(10) - 1
    
    # Create a composite factor
    df['composite_factor'] = (
        0.2 * df['daily_return'] + 
        0.2 * df['pct_change'] + 
        0.1 * df['volume_ratio'] + 
        0.1 * df['up_ratio'] + 
        0.1 * df['range_ratio'] + 
        0.1 * df['amount_ratio'] + 
        0.1 * df['deviation_from_ma'] + 
        0.1 * df['cumulative_return_1m'] + 
        0.1 * df['volatility_10d'] + 
        0.1 * df['momentum_10d']
    )
    
    return df['composite_factor']
```
