import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_return = close_to_open_return * (intraday_range / intraday_range.max())
    
    # Calculate Volume-Weighted Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_amount = df['amount'].rolling(window=20).mean()
    vol_weighted_volatility = (intraday_range / (avg_volume + 1e-6)) * (avg_amount + 1e-6)
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return = adjusted_return * vol_weighted_volatility * volume_change
    
    # Calculate Daily Price Return
    n_day_return = df['close'] / df['close'].shift(20) - 1
    
    # Identify Volume Spike
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike = (df['volume'] > (moving_avg_volume + 2 * std_volume)).astype(int)
    
    # Adjust Daily Price Return for Volume Spike
    k = 1.5  # Amplification factor
    adjusted_n_day_return = n_day_return * (1 + (volume_spike * (k - 1)))
    
    # Compute Cumulative Sum of Adjusted Price Return
    cumulative_adjusted_return = adjusted_n_day_return.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = volume_change.cumsum()
    
    # Combine Cumulative Sums
    final_factor = cumulative_adjusted_return * cumulative_volume_change
    
    return final_factor
