```python
import pandas as pd

def heuristics_v2(df):
    # Calculate rate of change in closing price over the last 5 and 20 days
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)
    
    # Calculate simple moving averages
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Evaluate moving average crossovers
    df['ma_cross_5_20'] = df['sma_5'] - df['sma_20']
    df['ma_cross_20_50'] = df['sma_20'] - df['sma_50']
    
    # Assess volatility-adjusted returns using standard deviation over past 20 days
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['vol_adjusted_return'] = (df['close'] - df['close'].shift(1)) / df['volatility_20']
    
    # Analyze volume-based indicators
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['high_volume_day'] = (df['volume'] > df['avg_volume_20'] * 1.5).astype(int)
    df['low_volume_day'] = (df['volume'] < df['avg_volume_20'] * 0.5).astype(int)
    
    # Consider price patterns and support/resistance levels
    df['breakout_high'] = (df['high'] > df['high'].rolling(window=20).max()).astype(int)
    df['breakout_low'] = (df['low'] < df['low'].rolling(window=20).min()).astype(int)
    
    # Examine relationships between different price points
    df['day_range'] = df['close'] - df['open']
    df['close_to_high'] = df['close'] / df['high']
    df['close_to_low'] = df['close'] / df['low']
    df['high_to_low'] = df['high'] / df['low']
    df['close_to_prev_close'] = df['close'] / df['close'].shift(1)
    
    # Explore additional price point relationships
    df['high_to_sma_5'] = df['high'] / df['sma_5']
    df['low_to_sma_20'] = df['low'] / df['sma_20']
    df['close_to_sma_50'] = df['close'] / df['sma_50']
    df['sma_diff_5_20'] = df['sma_5'] - df['sma_20']
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (
        df['roc_5'] + 
        df['roc_20'] + 
        df['ma_cross_5_20'] + 
        df['ma_cross_20_50'] + 
        df['vol_adjusted_return'] + 
        df['volume_ratio'] + 
        df['high_volume_day'] + 
        df['low_volume_day'] + 
        df['breakout_high'] + 
        df['breakout_low'] + 
        df['day_range'] + 
        df['close_to_high'] + 
        df['close_to_low'] + 
        df['high_to_low'] + 
        df['close_to_prev_close'] + 
        df['high_to_sma_5'] + 
        df['low_to_sma_20'] + 
        df['close_to_sma_50'] + 
        df['sma_diff_5_20']
    )
    
    return df['alpha_factor']
```
