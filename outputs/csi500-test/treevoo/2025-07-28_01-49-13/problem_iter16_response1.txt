```python
import pandas as pd

def heuristics_v2(df, period=10):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    midpoint_to_open_ratio = intraday_midpoint / df['open']
    
    # Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Close-to-Next Day Open Return
    close_to_next_day_open_return = vwap.shift(-1) / df['open'].shift(-1) - 1
    
    # Close-to-Open Envelope
    max_close = df['close'].rolling(window=period).max()
    min_open = df['open'].rolling(window=period).min()
    
    # Combine Factors
    combined_factor = (midpoint_to_open_ratio * close_to_next_day_open_return) / (max_close - min_open)
    
    return combined_factor
```
