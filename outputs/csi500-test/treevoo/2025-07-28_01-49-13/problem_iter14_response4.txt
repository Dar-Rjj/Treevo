import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_mid_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Volume Weighted Momentum
    close_to_next_open_return = df['close'].shift(-1) / df['open'] - 1
    volume_weighted_momentum = (close_to_next_open_return * df['volume']).rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adj_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    volume_momentum = (close_to_close_return * daily_volume_change).rolling(window=5).sum()
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    
    # Combine Intraday Factors
    combined_intraday_factors = (intraday_mid_to_open_ratio * volume_weighted_momentum) * (max_adj_close - min_open)
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    adjusted_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) * (max_high - min_low)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Combine High-Low Range and Close-Open Return
    combined_high_low_close_open = intraday_range + close_to_open_return
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_high_low_close_open
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'] + df['low']) / 2).shift(1) * intraday_range
    
    # Detect Volume Spike
    volume_spike_indicator = (daily_volume_change > daily_volume_change.mean() + 2 * daily_volume_change.std()).astype(int)
    
    # Final Factor Calculation
    final_factor_calculation = volume_weighted_adjusted_return * price_change_momentum * volume_spike_indicator
    
    # Combine Final Intraday and Cumulative Factors
    final_alpha_factor = combined_intraday_factors * adjusted_intraday_factors * final_factor_calculation
    
    return final_alpha_factor
