```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from open to close
    df['price_change_open_close'] = df['close'] - df['open']
    
    # Calculate the percentage of the daily price range (high - low) that the close price covers
    df['close_range_percentage'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the ratio of the amount over volume, which can be interpreted as the average price per unit volume
    df['avg_price_per_vol'] = df['amount'] / df['volume']
    
    # Generate a simple signal based on whether the close is above the mid-point of the high-low range and the price change is positive
    df['signal'] = ((df['close_range_percentage'] > 0.5) & (df['price_change_open_close'] > 0)).astype(int)
    
    # Multiply the signal by the average price per unit volume to weight the signal by the price level
    alpha_factor = df['signal'] * df['avg_price_per_vol']
    
    return alpha_factor
```
