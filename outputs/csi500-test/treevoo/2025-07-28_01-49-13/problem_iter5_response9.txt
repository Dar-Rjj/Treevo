```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, K=1.5, lookback=20):
    # Calculate Daily Price Momentum
    df['Daily_Price_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate N-day Price Momentum
    df['N_day_Price_Momentum'] = (df['close'] - df['close'].shift(N)) / df['close'].shift(N)
    
    # Identify Volume Spike
    df['Vol_MA'] = df['volume'].rolling(window=M).mean()
    df['Vol_STD'] = df['volume'].rolling(window=M).std()
    df['Volume_Spike'] = (df['volume'] > (df['Vol_MA'] + 2 * df['Vol_STD'])).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['Adjusted_N_day_Return'] = df['N_day_Price_Momentum']
    df.loc[df['Volume_Spike'] == 1, 'Adjusted_N_day_Return'] *= K
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_Open_Return'] = df['close'] / df['open'] - 1
    
    # Combine Intraday High-Low Range and Close-Open Return
    df['Combined_Return'] = df['Adjusted_N_day_Return'] * (df['Intraday_High_Low_Range'] + df['Close_Open_Return'])
    
    # Compute Cumulative Sum of Price Momentum
    df['Cumulative_Price_Momentum'] = df['Daily_Price_Momentum'].cumsum()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Sum of Volume Change
    df['Cumulative_Volume_Change'] = df['Daily_Volume_Change'].cumsum()
    
    # Combine Cumulative Sums
    df['Combined_Cumulative'] = df['Cumulative_Price_Momentum'] * df['Cumulative_Volume_Change']
    
    # Calculate Volume Weighted Adjusted Return
    df['Volume_Weighted_Adjusted_Return'] = df['volume'] * df['Combined_Return']
    
    # Calculate Moving Average of Volume Weighted Adjusted Return
    df['MA_Volume_Weighted_Adjusted_Return'] = df['Volume_Weighted_Adjusted_Return'].rolling(window=lookback).mean()
    
    # Calculate Final Alpha Factor
    df['Alpha_Factor'] = df['Volume_Weighted_Adjusted_Return'] / df['MA_Volume_Weighted_Adjusted_Return']
    
    return df['Alpha_Factor']
```
