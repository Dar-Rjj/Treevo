```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Momentum
    df['next_day_open'] = df['open'].shift(-1)
    df['close_to_next_day_open_return'] = (df['next_day_open'] - df['close']) / df['close']
    df['volume_weighted_momentum'] = (df['close_to_next_day_open_return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adj_close'] = df['close'].rolling(window=5).max()
    df['min_open'] = df['open'].rolling(window=5).min()
    
    # Calculate Intraday Range to Open Ratio
    df['intraday_range_to_open_ratio'] = df['intraday_range'] / df['open']
    
    # Calculate Volume Momentum
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_momentum'] = (df['close_to_close_return'] * df['daily_volume_change']).rolling(window=5).sum() / df['daily_volume_change'].rolling(window=5).sum()
    
    # Combine Intraday Factors
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['intraday_midpoint_to_open_ratio'] = df['intraday_midpoint'] / df['open']
    df['intraday_factors_combined'] = (df['intraday_midpoint_to_open_ratio'] * df['volume_weighted_momentum']) * (df['max_adj_close'] / df['min_open'])
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    df['high_low_envelope'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    df['intraday_adjusted'] = (df['intraday_range_to_open_ratio'] * df['volume_momentum']) * df['high_low_envelope']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_adjusted_return'] = df['intraday_factors_combined'] * df['volume'] / df['volume'].rolling(window=5).sum()
    
    # Calculate Price Change Momentum
    df['mid_price_change'] = (df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2
    df['price_change_momentum'] = df['mid_price_change'] * df['intraday_range']
    
    # Detect Volume Spike
    df['volume_spike'] = np.where(df['daily_volume_change'] > df['volume'].rolling(window=5).std() * 2, 1, 0)
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_adjusted_return'] * df['price_change_momentum'] * df['volume_spike']
    
    return df['final_factor']
```
