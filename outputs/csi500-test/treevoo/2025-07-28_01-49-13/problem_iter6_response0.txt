```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-Open Return
    combined_return = (high_low_range + close_open_return) / 2
    
    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = combined_return * df['volume']
    
    # Calculate Moving Average of Adjusted Return
    lookback_period = 20
    moving_average = volume_weighted_adjusted_return.rolling(window=lookback_period).mean()
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df.shift(1)['high'] + df.shift(1)['low']) / 2) * high_low_range
    
    # Detect Volume Spike
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_indicator = (daily_volume_change > 2 * df['volume'].rolling(window=lookback_period).std()).astype(int)
    
    # Final Factor Calculation
    final_factor = (volume_weighted_adjusted_return - moving_average) * price_change_momentum * volume_spike_indicator
    
    return final_factor
```
