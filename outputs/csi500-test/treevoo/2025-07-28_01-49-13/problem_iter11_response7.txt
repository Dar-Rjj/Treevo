import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Intraday Midpoint to Open Ratio
    intraday_midpoint = (df['high'] + df['low']) / 2
    intraday_midpoint_to_open_ratio = intraday_midpoint / df['open']

    # Calculate Volume Weighted Momentum
    next_day_open = df['open'].shift(-1)
    close_to_next_day_open_return = (next_day_open - df['close']) / df['close']
    volume_weighted_momentum = (close_to_next_day_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    adjusted_close_to_open_envelope = (max_adjusted_close - min_open) / min_open

    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']

    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_momentum = (close_to_close_return * daily_volume_change).rolling(window=5).sum()

    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    high_low_envelope = (max_high - min_low) / min_low

    # Combine Intraday Factors
    combined_intraday_factors = (intraday_midpoint_to_open_ratio * volume_weighted_momentum) / adjusted_close_to_open_envelope

    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    intraday_range_to_open_ratio_and_volume_momentum = (intraday_range_to_open_ratio * volume_momentum) / high_low_envelope

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)

    # Combine High-Low Range and Close-Open Return
    combined_high_low_range_and_close_open_return = (intraday_range + close_to_open_return) / 2

    # Calculate Volume Weighted Adjusted Return
    volume_weighted_adjusted_return = (combined_high_low_range_and_close_open_return * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Price Change Momentum
    price_change_momentum = (intraday_midpoint - intraday_midpoint.shift(1)) * intraday_range

    # Detect Volume Spike
    volume_spike_indicator = (daily_volume_change > 0.5 * df['volume'].shift(1)).astype(int)

    # Final Factor Calculation
    final_factor_calculation = volume_weighted_adjusted_return * price_change_momentum * volume_spike_indicator

    # Combine Final Intraday and Cumulative Factors
    combined_final_intraday_and_cumulative_factors = (combined_intraday_factors * final_factor_calculation) * intraday_range_to_open_ratio_and_volume_momentum

    return combined_final_intraday_and_cumulative_factors
