```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving average (SMA) of close prices
    df['SMA'] = df['close'].rolling(window=20).mean()
    
    # Calculate exponential moving average (EMA) of close prices
    df['EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine the difference between EMA and SMA
    df['EMA_SMA_diff'] = df['EMA'] - df['SMA']
    
    # Analyze if the difference can indicate market momentum
    df['momentum'] = df['EMA_SMA_diff'].diff()
    
    # Analyze the high-low spread (High - Low)
    df['high_low_spread'] = df['high'] - df['low']
    
    # Assess volatility by comparing daily high-low spreads
    df['volatility'] = df['high_low_spread'].rolling(window=20).std()
    
    # Evaluate the open-close price change
    df['open_close_change'] = df['close'] - df['open']
    
    # Investigate patterns in the opening and closing prices over time
    df['price_trend'] = df['open_close_change'].rolling(window=20).sum()
    
    # Measure trading volume trends
    df['volume_trend'] = df['volume'].rolling(window=20).mean().pct_change()
    
    # Examine the relationship between volume changes and price movements
    df['volume_price_corr'] = df['close'].pct_change().rolling(window=20).corr(df['volume'].pct_change())
    
    # Combine multiple indicators for a composite factor
    df['composite_factor'] = 0.4 * df['momentum'] + 0.3 * df['volatility'] + 0.2 * df['price_trend'] + 0.1 * df['volume_trend']
    
    # Return the composite factor
    return df['composite_factor']
```
