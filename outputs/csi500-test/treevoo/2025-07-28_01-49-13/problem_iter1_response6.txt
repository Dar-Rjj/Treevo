```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change Momentum
    current_avg = (df['high'] + df['low']) / 2
    previous_avg = (df['high'].shift(1) + df['low'].shift(1)) / 2
    price_change_momentum = current_avg - previous_avg
    
    # Multiply by Range to get High-to-Low Momentum Spread
    high_to_low_momentum_spread = price_change_momentum * high_low_range
    
    return high_to_low_momentum_spread
```
