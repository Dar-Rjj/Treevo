import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, K=1.5, lookback=20):
    # Calculate N-day Price Momentum
    df['N_day_return'] = df['close'].pct_change(N)
    
    # Identify Volume Spike
    df['vol_moving_avg'] = df['volume'].rolling(window=M).mean()
    df['vol_std'] = df['volume'].rolling(window=M).std()
    df['is_volume_spike'] = (df['volume'] > (df['vol_moving_avg'] + 2 * df['vol_std']))
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Combine Price Momentum, Volume Spike, and Intraday Volatility
    df['combined_return'] = np.where(
        df['is_volume_spike'],
        df['N_day_return'] * K,
        df['N_day_return']
    )
    df['combined_return'] *= df['intraday_range']
    
    # Calculate Volume Weighted Adjusted Return
    df['volume_weighted_return'] = df['combined_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume Weighted Adjusted Return
    df['ewm_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=lookback, adjust=False).mean()
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['volume_weighted_return'] / df['ewm_volume_weighted_return']
    
    return df['alpha_factor']
