```python
def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted price change
    df['vwp_change'] = ((df['close'] - df['open']) * df['volume']) / df['volume'].rolling(window=10).mean()
    
    # Calculate the amount weighted return
    df['aw_return'] = (df['return'] * df['amount']) / df['amount'].rolling(window=10).mean()
    
    # Calculate the factor by combining the above components
    factor = 0.5 * df['range'] + 0.3 * df['vwp_change'] + 0.2 * df['aw_return']
    
    return factor.dropna()
```
