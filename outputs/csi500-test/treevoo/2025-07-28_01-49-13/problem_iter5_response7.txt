```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    daily_volume_change = df['volume'].diff()
    close_returns = df['close'].pct_change()
    volume_momentum = (daily_volume_change * close_returns).rolling(window=5).sum()
    
    # Calculate High-Low Envelope
    max_high = df['high'].rolling(window=5).max()
    min_low = df['low'].rolling(window=5).min()
    
    # Combine Intraday Factors
    combined_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) / (max_high - min_low)
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'].diff()
    
    # Compute Cumulative Sum of Price Momentum
    cumulative_price_momentum = daily_price_momentum.cumsum()
    
    # Compute Cumulative Sum of Volume Change
    cumulative_volume_change = daily_volume_change.cumsum()
    
    # Combine Cumulative Sums
    combined_cumulative_sums = cumulative_price_momentum * cumulative_volume_change
    
    # Combine Final Factors
    combined_final_factors = combined_intraday_factors * combined_cumulative_sums
    
    return combined_final_factors
```
