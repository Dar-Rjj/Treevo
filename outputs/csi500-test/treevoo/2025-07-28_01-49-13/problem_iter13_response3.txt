```python
def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Cumulative Price Change
    df['cumulative_price_change'] = df['daily_price_change'].cumsum()
    
    # Compute Cumulative Volume Change
    df['cumulative_volume_change'] = df['daily_volume_change'].cumsum()
    
    # Combine Cumulative Sums
    df['factor'] = df['cumulative_price_change'] * df['cumulative_volume_change']
    
    return df['factor']
```
