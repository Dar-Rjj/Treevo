import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_mid_to_open_ratio = (df['high'] + df['low']) / 2 / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Adjust Close-to-Open Return by Intraday Range
    close_to_open_adjusted = (df['close'] - df['open']) / intraday_range
    
    # Calculate Rate of Change in Closing Price
    roc_5 = df['close'].pct_change(5)
    roc_20 = df['close'].pct_change(20)
    
    # Evaluate Moving Average Crossovers
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    crossover_5_20 = sma_5 > sma_20
    crossover_20_50 = sma_20 > sma_50
    
    # Examine Relationships Between Different Price Points
    close_to_high = df['close'] / df['high']
    close_to_low = df['close'] / df['low']
    close_to_open_diff = df['close'] - df['open']
    
    # Assess Volatility-Adjusted Returns
    volatility = df['close'].rolling(window=20).std()
    vol_adjusted_return = df['close'].pct_change() / volatility
    
    # Measure the ratio of today's volume to the average volume over the past 20 days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    
    # Identify days with exceptionally high or low volume relative to the recent average
    high_volume_days = volume_ratio > 2
    low_volume_days = volume_ratio < 0.5
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Compute Cumulative Price Change
    cumulative_price_change = daily_price_change.cumsum()
    
    # Calculate Close-to-Next Day Open Return
    close_to_next_open_return = vwap - df['open'].shift(-1)
    
    # Weighted Sum of Returns by Volume
    weighted_sum_returns = close_to_next_open_return * df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Integrate Volatility Factor
    daily_volatility = intraday_range / 2
    smooth_volatility = daily_volatility.ewm(span=20).mean()
    
    # Calculate True Range and Average True Range (ATR)
    true_range = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift()), abs(x['low'] - df['close'].shift())), axis=1)
    atr = true_range.rolling(window=20).mean()
    
    # Calculate Short-Term Momentum
    short_term_sma = df['close'].rolling(window=5).mean()
    short_term_momentum = df['close'] - short_term_sma
    short_term_roc = df['close'].pct_change(5)
    macd_line = df['close'].ewm(span=12).mean() - df['close'].ewm(span=26).mean()
    signal_line = macd_line.ewm(span=9).mean()
    
    # On-Balance Volume (OBV) Calculation
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv_trend = obv.ewm(span=20).mean()
    
    # Final Factor Adjustment
    final_factor = (intraday_mid_to_open_ratio * weighted_sum_returns * (max_adjusted_close - min_open) + close_to_open_adjusted) / smooth_volatility
    
    # Compute Cumulative Price Momentum
    exp_moving_avg = df['close'].ewm(span=20).mean()
    cumulative_volume = df['volume'].cumsum()
    cumulative_price_momentum = (exp_moving_avg / cumulative_volume)
    
    return final_factor
