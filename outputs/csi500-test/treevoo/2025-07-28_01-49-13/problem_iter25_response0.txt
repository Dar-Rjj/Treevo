```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust Close-to-Open Return by Volume Change
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['volume_change']
    
    # Identify Volume Spike
    vol_moving_avg = df['volume'].rolling(window=20).mean()
    vol_std = df['volume'].rolling(window=20).std()
    df['volume_spike'] = (df['volume'] > (vol_moving_avg + 3 * vol_std)).astype(int)
    
    # Adjust Close-to-Open Return for Volume Spike
    K = 1.5  # Constant to amplify the return
    df.loc[df['volume_spike'] == 1, 'adjusted_close_to_open_return'] *= K
    
    # Compute Cumulative Sum of Adjusted Close-to-Open Return
    df['cumulative_adjusted_close_to_open_return'] = df['adjusted_close_to_open_return'].cumsum()
    
    # Introduce Enhanced Momentum Factor
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['enhanced_momentum_factor'] = df['close'] - df['50_day_ma'] - df['200_day_ma']
    
    # Incorporate Enhanced Momentum Factor
    df['final_alpha_factor'] = df['adjusted_close_to_open_return'] * df['enhanced_momentum_factor'] / 100  # Normalize by 100 to avoid excessively large values
    
    return df['final_alpha_factor']
```
