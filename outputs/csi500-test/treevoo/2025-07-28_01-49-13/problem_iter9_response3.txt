```python
import pandas as pd

def heuristics_v2(df, ema_window=10, ema_lookback=5, volume_spike_days=20, momentum_days=5, k_factor=1.5, adjusted_return_lookback=20):
    # Calculate Volume-Weighted High-Low Spread
    df['weighted_high_low'] = (df['high'] - df['low']) * df['volume']
    
    # Exponential Moving Average (EMA) of Weighted High-Low Spread
    df['weighted_ema'] = df['weighted_high_low'].ewm(span=ema_window).mean()
    df['previous_weighted_ema'] = df['weighted_ema'].shift(ema_lookback)
    df['current_weighted_ema'] = df['weighted_ema'] - df['previous_weighted_ema']
    
    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=volume_spike_days).mean()
    df['volume_std'] = df['volume'].rolling(window=volume_spike_days).std()
    df['volume_spike'] = (df['volume'] > (df['volume_ma'] + 2 * df['volume_std'])).astype(int)
    
    # Calculate N-day Price Momentum
    df['momentum'] = df['close'].pct_change(momentum_days)
    
    # Adjust for Volume Spike
    df['adjusted_momentum'] = df['momentum'] * (1 + (df['volume_spike'] * (k_factor - 1)))
    
    # Calculate Intraday High-Low Range and Close-to-Open Return
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday High-Low Range and Close-Open Return
    df['combined_return'] = df['adjusted_momentum'] * df['high_low_range'] * df['close_open_return']
    
    # Calculate Volume-Weighted Adjusted Return
    df['volume_weighted_return'] = df['combined_return'] * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Adjusted Return
    df['volume_weighted_return_ma'] = df['volume_weighted_return'].rolling(window=adjusted_return_lookback).mean()
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['current_weighted_ema'] * df['volume_weighted_return'] / df['volume_weighted_return_ma']
    
    return df['alpha_factor']
```
