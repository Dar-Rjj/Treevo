import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint to Open Ratio
    df['intraday_midpoint_open_ratio'] = ((df['high'] + df['low']) / 2) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = df['amount'] / df['volume']
    
    # Calculate Short-Term Momentum
    short_period = 10
    df['sma_close'] = df['close'].rolling(window=short_period).mean()
    df['momentum'] = df['close'] - df['sma_close']
    df['roc'] = df['close'].pct_change(periods=short_period)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    
    # Calculate True Range and Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1
    )
    atr_period = 14
    df['atr'] = df['true_range'].rolling(window=atr_period).sum() / atr_period
    
    # Measure Standard Deviation of Price Changes
    df['std_dev'] = df['close'].diff().rolling(window=atr_period).std()
    
    # Calculate Close-to-Next Day Open Return
    df['close_to_next_day_open_return'] = df['vwap'].shift(-1) / df['open'].shift(-1) - 1
    
    # Weighted Sum of Returns by Volume
    df['weighted_sum_returns'] = df['close_to_next_day_open_return'] * df['volume']
    
    # Calculate Close-to-Open Envelope
    envelope_period = 20
    df['max_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Calculate Adjusted Close-to-Open Envelope
    df['max_adjusted_close'] = df['close'].rolling(window=envelope_period).max()
    df['min_open'] = df['open'].rolling(window=envelope_period).min()
    
    # Introduce Volatility Factor
    df['daily_volatility'] = (df['high'] - df['low']).rolling(window=atr_period).mean()
    df['smoothed_volatility'] = df['daily_volatility'].ewm(span=atr_period, adjust=False).mean()
    
    # On-Balance Volume (OBV) Calculation
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Final Factor Combination
    df['factor'] = (
        df['intraday_midpoint_open_ratio'] * df['weighted_sum_returns'] * 
        (df['max_close'] - df['min_open']) * (df['max_adjusted_close'] - df['min_open']) * 
        (df['close'].pct_change() * df['obv'].pct_change()) * 
        (df['momentum'] + df['atr'] + (df['high'] - df['low']) / df['volume']) * 
        (df['vwap'].shift(-1) / df['open'].shift(-1) - 1) * df['smoothed_volatility']
    )
    
    return df['factor']
