import pandas as pd

def heuristics_v2(df):
    # Calculate daily price range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate close to open return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate average volume over N days (e.g., 10 days)
    n = 10
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Price Trend (VPT)
    df['vpt'] = (df['volume'] * ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))).cumsum()
    
    # Calculate daily transaction amount
    df['transaction_amount'] = df['close'] * df['volume']
    
    # Calculate change in transaction amount
    df['change_transaction_amount'] = (df['transaction_amount'] - df['transaction_amount'].shift(1)) / df['transaction_amount'].shift(1)
    
    # High-Low range relative to open
    df['high_low_range_rel_open'] = (df['high'] - df['low']) / df['open']
    
    # Open-adjusted volume
    df['open_adjusted_volume'] = df['volume'] * (df['open'] / df['close'])
    
    # Rolling sum of daily price ranges over N days
    df['rolling_sum_price_range'] = df['daily_price_range'].rolling(window=n).sum()
    
    # Rolling sum of volume over N days
    df['rolling_sum_volume'] = df['volume'].rolling(window=n).sum()
    
    # Create a composite alpha factor
    df['price_signal'] = df['close_to_open_return']
    df['volume_signal'] = df['vpt']
    df['amount_signal'] = df['change_transaction_amount']
    
    # Advanced composite factor
    df['composite_factor'] = df['price_signal'] * df['volume_signal'] * df['amount_signal']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    # Return the composite factor
    return df['composite_factor']
