import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Intraday Midpoint to Open Ratio
    intraday_mid_to_open_ratio = intraday_midpoint / df['open']
    
    # Calculate Close-to-Next Day Open Return over a period
    close_to_next_open_return = (df['open'].shift(-1) / df['close']) - 1
    volume_weighted_close_to_next_open_return = close_to_next_open_return * df['volume']
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = df['close'].pct_change() * df['volume']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Adjust Close-to-Open Return by Intraday Range
    adjusted_close_to_open_return = close_to_open_return / intraday_range
    
    # Calculate Volume-Weighted Volatility
    volume_weighted_volatility = intraday_range / df['volume'].rolling(window=20).sum()
    
    # Combine Adjusted Return and Volume-Weighted Volatility
    combined_adjusted_return_volatility = adjusted_close_to_open_return * volume_weighted_volatility
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = df['amount'] / df['volume']
    
    # Calculate Adjusted Close-to-Open Envelope
    max_adjusted_close = df['close'].rolling(window=20).max()
    min_open = df['open'].rolling(window=20).min()
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    close_to_close_return = df['close'].pct_change()
    volume_momentum = daily_volume_change * close_to_close_return.rolling(window=20).sum()
    
    # Combine Intraday Factors
    combined_intraday_factors = intraday_mid_to_open_ratio * volume_weighted_momentum
    combined_intraday_factors = combined_intraday_factors * (max_adjusted_close - min_open)
    
    # Adjust by Intraday Range to Open Ratio and Volume Momentum
    high_low_envelope = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    adjusted_combined_intraday_factors = (intraday_range_to_open_ratio * volume_momentum) * high_low_envelope
    
    # Calculate Volume-Weighted Adjusted Return
    volume_weighted_adjusted_return = df['volume'] * combined_intraday_factors
    
    # Calculate Price Change Momentum
    price_change_momentum = ((df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2) * intraday_range
    
    # Detect Volume Spike
    volume_spike_indicator = (daily_volume_change > 2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Final Factor Calculation
    final_factor = volume_weighted_adjusted_return * price_change_momentum * volume_spike_indicator
    
    return final_factor
