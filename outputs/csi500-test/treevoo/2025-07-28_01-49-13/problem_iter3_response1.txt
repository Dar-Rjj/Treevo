```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Range to Open Ratio
    intraday_range_to_open_ratio = intraday_range / df['open']
    
    # Calculate Volume Momentum
    close_to_open_return = df['close'].shift(-1) / df['open'].shift(-1) - 1
    volume_momentum = (close_to_open_return * df['volume']).rolling(window=5).sum()
    
    # Calculate Close-to-Open Envelope
    max_close = df['close'].rolling(window=5).max()
    min_open = df['open'].rolling(window=5).min()
    
    # Combine Factors
    factor = (intraday_range_to_open_ratio * volume_momentum) * (max_close / min_open)
    
    return factor
```
