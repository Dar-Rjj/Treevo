```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()

    # Combine Momentum Indicators
    df['combined_momentum'] = 0.55 * df['long_term_momentum'] + 0.45 * df['short_term_momentum']

    # Calculate Intraday Direction
    df['intraday_direction'] = df['close'] - df['open']

    # Calculate Relative Volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=21).mean()

    # Weight Intraday Direction by Relative Volume
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']

    # Apply Moving Averages to Intraday High-to-Low Percentage Change
    df['intraday_high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    fast_ema = df['intraday_high_low_pct_change'].ewm(span=12, adjust=False).mean()
    slow_ema = df['intraday_high_low_pct_change'].ewm(span=26, adjust=False).mean()

    # MACD Difference
    macd_diff = fast_ema - slow_ema

    # Combine Momentum Indicators with MACD
    df['momentum_macd'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum'] + macd_diff

    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)

    # Compute Combined Indicator
    df['final_factor'] = df['intraday_high_low_pct_change'] * df['volume_change_ratio']

    # Adjust for Market Volatility (Use Moving Standard Deviation of Daily Returns)
    df['daily_returns'] = df['close'].pct_change()
    df['volatility'] = df['daily_returns'].rolling(window=21).std()
    
    # Dynamic Threshold
    dynamic_threshold = 2 * df['volatility']
    
    # Apply Threshold to Identify Significant Movements
    df['final_factor'] = df['final_factor'].where(df['final_factor'].abs() > dynamic_threshold, 0)

    return df['final_factor']
```
