def heuristics_v2(df, n=20):
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_direction = df['close'] - df['open']
    
    # Calculate Weighted Volume Activity
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    relative_volume = df['volume'] / avg_volume_n_days
    
    # Combine Intraday Price and Volume Metrics
    weighted_intraday_range = intraday_range * relative_volume
    weighted_intraday_direction = intraday_direction * relative_volume
    combined_intraday_metrics = (weighted_intraday_range + weighted_intraday_direction) / 2
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Calculate Volume Confirmation
    avg_volume_past_20 = df['volume'].shift(1).rolling(window=n-1).mean()
    volume_ratio = df['volume'] / avg_volume_past_20
    volume_confirmation = volume_ratio.apply(lambda x: 1 if x > 1 else (-1 if x < 1 else 0))
    
    # Combine Momentum and Volume Confirmation
    combined_momentum_and_volume = price_momentum * volume_confirmation
    
    # Generate Final Alpha Factor
    final_alpha_factor = combined_intraday_metrics + combined_momentum_and_volume
    
    return final_alpha_factor
