```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Calculate 20-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    ema_std_dev = daily_returns.ewm(span=20, min_periods=20).std()
    
    # Apply Threshold to Identify Significant Movements
    dynamic_threshold = 3 * ema_std_dev
    significant_movement = combined_indicator > dynamic_threshold
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_returns = daily_returns.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, min_periods=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_ma_return = ma_log_returns * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Fast and Slow Exponential Moving Averages
    fast_ema = adjusted_ma_return.ewm(span=10, min_periods=10, adjust=False).mean()
    slow_ema = adjusted_ma_return.ewm(span=40, min_periods=40, adjust=False).mean()
    
    # Calculate Price Momentum
    close_sma = df['close'].rolling(window=20).mean()
    momentum_score = df['close'] - close_sma
    
    # Introduce Volatility-Adjusted Momentum
    ema_abs_log_returns = abs(daily_returns).ewm(span=20, min_periods=20, adjust=False).mean()
    volatility_adjusted_momentum = momentum_score / ema_abs_log_returns
    
    # Synthesize Final Indicator
    final_indicator = 0.3 * significant_movement + 0.4 * fast_ema + 0.3 * volatility_adjusted_momentum
    
    return final_indicator
```
