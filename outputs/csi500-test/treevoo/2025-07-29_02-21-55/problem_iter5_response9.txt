```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_mom'] = (df['close'] - df['close'].shift(20)) / df['range'].rolling(window=22).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_mom'] = (df['close'] - df['close'].shift(5)) / df['range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators
    df['combined_mom'] = 0.6 * df['long_term_mom'] + 0.4 * df['short_term_mom']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low'] * 100
    
    # Apply Moving Averages
    df['fast_ema'] = df['intraday_pct_change'].ewm(span=12, adjust=False).mean()
    df['slow_ema'] = df['intraday_pct_change'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['macd'] = df['fast_ema'] - df['slow_ema']
    df['combined_indicator'] = 0.5 * df['long_term_mom'] + 0.5 * df['short_term_mom'] + df['macd']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    threshold = 2 * df['volatility']
    
    df['significant_movement'] = (df['intraday_pct_change'] * df['volume_change_ratio']).apply(lambda x: 1 if x > threshold else 0)
    
    df['final_factor'] = df['combined_indicator'] * df['significant_movement']
    
    return df['final_factor'].dropna()
```
