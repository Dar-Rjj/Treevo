```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Intraday Volume Change
    volume_change = df['volume'].pct_change()
    
    # Apply Volume-Adjusted Intraday Momentum
    volume_adjusted_momentum = intraday_momentum * volume_change
    
    # Define Fast and Slow Exponential Moving Averages
    fast_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    slow_ema = volume_adjusted_momentum.ewm(span=30, adjust=False).mean()
    
    # Final alpha factor: difference between Fast EMA and Slow EMA
    alpha_factor = fast_ema - slow_ema
    
    return alpha_factor
```
