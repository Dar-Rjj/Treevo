import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.5 * df['long_term_momentuwm'] + 0.5 * df['short_term_momentum']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low'] * 100
    
    # Apply Moving Averages
    df['fast_ema'] = df['intraday_pct_change'].ewm(span=9, adjust=False).mean()
    df['slow_ema'] = df['intraday_pct_change'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['macd'] = df['fast_ema'] - df['slow_ema']
    df['momentum_macd'] = 0.4 * df['long_term_momentum'] + 0.6 * df['short_term_momentum'] + df['macd']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    avg_volume = df['volume'].mean()
    threshold = 1.5 * avg_volume
    df['significant_movement'] = (df['volume_change_ratio'] > threshold).astype(int) * df['intraday_pct_change']
    
    # Compute Combined Indicator
    df['alpha_factor'] = df['momentum_macd'] + df['significant_movement']
    
    return df['alpha_factor']
