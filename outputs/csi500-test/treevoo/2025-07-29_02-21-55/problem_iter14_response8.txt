```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the weighted amount based on the volume
    weighted_amount = df['amount'] * df['volume']
    
    # Calculate the factor as the ratio of the weighted amount to the average price
    factor = weighted_amount / avg_price
    
    return factor
```
