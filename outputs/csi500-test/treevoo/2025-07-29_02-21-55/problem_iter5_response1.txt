import pandas as pd

def heuristics_v2(df):
    # Compute Close Price Change
    df['close_price_change'] = df['close'].diff()
    
    # Compute Moving Average of Close Price Change (20 days)
    df['price_momentum'] = df['close_price_change'].rolling(window=20).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Define Volume Spike Threshold (2x recent average volume)
    recent_avg_volume = df['volume'].rolling(window=20).mean()
    df['volume_spike_threshold'] = 2 * recent_avg_volume
    df['volume_spike'] = df['volume_change'] > df['volume_spike_threshold']
    
    # Adjust Price Momentum by Volume Spike
    df['adjusted_price_momentum'] = df['price_momentum'].where(~df['volume_spike'], df['price_momentum'] * 2.0)
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Intraday Dynamics with Volume Ratio
    df['combined_intraday'] = df['intraday_hl_pct_change'] * df['volume_ratio']
    
    # Calculate Daily Log Return
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['moving_avg_log_return'] = df['daily_log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    df['exp_moving_avg_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    df['sma_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_return'] = (df['moving_avg_log_return'] * df['exp_moving_avg_volume']) / df['sma_volume']
    
    # Smooth the Adjusted Return (5-day Simple Moving Average)
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()
    
    # Calculate Intraday Price Movement
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_direction'] = df['close'] - df['open']
    
    # Calculate Weighted Volume Activity
    n = 20
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume']
    
    # Combine Intraday Price and Volume Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['relative_volume']
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']
    df['combined_metric'] = df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    # Adjust Combined Metric by Price Momentum and Volume Spike
    df['final_combined_metric'] = df['combined_metric'].where(~df['volume_spike'], df['combined_metric'] * 2.0)
    
    # Incorporate Smoothed Adjusted Return
    df['alpha_factor'] = df['final_combined_metric'] + df['smoothed_adjusted_return']
    
    return df['alpha_factor'].dropna()
