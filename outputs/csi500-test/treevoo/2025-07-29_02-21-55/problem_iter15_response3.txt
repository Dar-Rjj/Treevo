```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_low_pct = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_high_low_pct * df['volume_change_ratio']
    
    # Apply Threshold to Identify Significant Movements
    # Set a Dynamic Threshold
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    dynamic_threshold = 1 + 2 * std_dev_20
    
    significant_movements = (combined_indicator > dynamic_threshold) * combined_indicator
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Combine Momentum and Volume Indicators
    # Set a Dynamic Volume Threshold
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    dynamic_volume_threshold = 1.5 * avg_volume
    
    # Synthesize Final Indicator
    weight_momentum = 0.6
    weight_volume = 0.4
    final_indicator = (weight_momentum * momentum_score + 
                       weight_volume * significant_movements * (df['volume'] > dynamic_volume_threshold))
    
    return final_indicator
```
