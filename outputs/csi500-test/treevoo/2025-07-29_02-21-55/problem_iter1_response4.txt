import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Components
    open_to_high_return = (df['high'] - df['open']) / df['open']
    open_to_low_return = (df['low'] - df['open']) / df['open']
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Returns
    average_intraday_return = (open_to_high_return + open_to_low_return + close_to_open_return) / 3
    intraday_return_difference = (max(open_to_high_return, open_to_low_return, close_to_open_return) -
                                  min(open_to_high_return, open_to_low_return, close_to_open_return))
    
    # Volume Weighted Adjustments
    volume_weighted_intraday_average = df['volume'] * average_intraday_return
    volume_weighted_intraday_difference = df['volume'] * intraday_return_difference
    
    # Generate Final Alpha Factor
    final_alpha_factor = (volume_weighted_intraday_average - volume_weighted_intraday_difference)
    
    # Integrate with Previous Day's Alpha
    if 'previous_alpha' in df.columns:
        previous_alpha = df['previous_alpha'].shift(1)
        final_alpha_factor = 0.7 * final_alpha_factor + 0.3 * previous_alpha
    else:
        # If no previous alpha, use the current alpha
        final_alpha_factor = 0.7 * final_alpha_factor
    
    return final_alpha_factor
