```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_hl_change'] * df['volume_change_ratio']
    
    # Apply Dynamic Threshold to Identify Significant Movements
    df['daily_return'] = df['close'].pct_change()
    df['20_day_ma'] = df['daily_return'].rolling(window=20).mean()
    df['20_day_std'] = df['daily_return'].rolling(window=20).std()
    df['dynamic_threshold'] = df['20_day_ma'] + 2 * df['20_day_std']
    
    # Confirm with Price Momentum
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_score'] = df['close'] - df['ema_20']
    
    # Combine Momentum and Volume Indicators
    df['volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_threshold'] = 1.5 * df['volume_ema_20']
    df['final_indicator'] = (df['combined_indicator'] * 0.6 + 
                             df['momentum_score'] * 0.4) * (df['volume'] > df['volume_threshold'])
    
    return df['final_indicator']
```
