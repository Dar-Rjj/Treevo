```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Direction
    intraday_direction = df['close'] - df['open']
    
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Combine Intraday Price and High-to-Low Momentum
    combined_intraday_momentum = intraday_direction * intraday_momentum
    
    # Apply Moving Averages to Intraday High-to-Low Momentum
    fast_ema = combined_intraday_momentum.ewm(span=10, adjust=False).mean()
    slow_ema = combined_intraday_momentum.ewm(span=30, adjust=False).mean()
    
    # Adjust Combined Intraday Momentum by Long-Term Price Trend
    log_returns = np.log(df['close']).diff()
    long_term_trend = log_returns.rolling(window=20).mean()
    adjusted_momentum = combined_intraday_momentum * long_term_trend
    
    # Smooth the Adjusted Intraday Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Incorporate Volume Confirmation
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_confirmation = np.where(relative_volume > 1, 1, np.where(relative_volume < 1, -1, 0))
    
    # Incorporate Price Oscillation
    max_high_20d = df['high'].rolling(window=20).max()
    min_low_20d = df['low'].rolling(window=20).min()
    actual_range_20d = df['high'] - min_low_20d.shift(19)
    max_range_20d = max_high_20d - min_low_20d
    oscillation_score = np.where(actual_range_20d > 0.5 * max_range_20d, 1, 
                                 np.where(actual_range_20d < 0.2 * max_range_20d, -1, 0))
    
    # Final Alpha Factor
    final_factor = (smoothed_momentum * volume_confirmation) + oscillation_score
    
    return final_factor
```
