```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = data['high'] - data['low']
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = intraday_momentum.ewm(span=5, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(data['close'] / data['close'].shift(1))
    ma_log_returns = log_returns.rolling(window=20).mean()
    
    # Calculate Long-Term and Short-Term Momentum
    daily_range_avg_22 = (data['high'] - data['low']).rolling(window=22).mean()
    long_term_momentum = (data['close'] - data['close'].shift(20)) / daily_range_avg_22
    
    daily_range_avg_7 = (data['high'] - data['low']).rolling(window=7).mean()
    short_term_momentum = (data['close'] - data['close'].shift(5)) / daily_range_avg_7
    
    # Combine Momentum Indicators
    combined_momentum = (long_term_momentum * 0.6) + (short_term_momentum * 0.4)
    
    # Adjust Intraday Momentum by Combined Momentum and Volume
    adjusted_intraday = (fast_ema * ma_log_returns) / data['amount'].ewm(span=20, adjust=False).mean() - slow_ema
    
    # Incorporate Volume Spike
    volume_change_ratio = data['volume'] / data['volume'].shift(1)
    daily_returns = data['close'].pct_change()
    threshold = 1.5 * daily_returns.rolling(window=20).std()
    
    significant_movement = (volume_change_ratio > 1 + threshold) | (volume_change_ratio < 1 - threshold)
    significant_movement = significant_movement.astype(int)
    
    # Generate Final Alpha Factor
    weighted_intraday_range = (intraday_momentum * data['volume'] / data['volume'].mean()) * significant_movement
    weighted_intraday_direction = (adjusted_intraday * data['volume'] / data['volume'].mean()) * significant_movement
    weighted_combined_momentum = combined_momentum * 0.7
    
    alpha_factor = (weighted_intraday_range + weighted_intraday_direction + weighted_combined_momentum).dropna()
    
    return alpha_factor
```
