```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday_momentum'] = df['intraday_momentum'].ewm(span=12, adjust=False).mean()
    df['slow_ema_intraday_momentum'] = df['intraday_momentum'].ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd_diff'] = df['fast_ema_intraday_momentum'] - df['slow_ema_intraday_momentum']
    df['combined_momentum'] = df['weighted_momentum'] + df['macd_diff']
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['vol_ratio'] = np.where(df['volume'] > df['avg_vol_20'], 1, np.where(df['volume'] < df['avg_vol_20'], -1, 0))
    df['ema_vol_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Long-Term Price Trend
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ma_log_returns_20'] = df['log_returns'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday_momentum'] * df['ma_log_returns_20']) / df['ema_vol_20'] - df['slow_ema_intraday_momentum']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_alpha_factor'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    return df['smoothed_alpha_factor']
```
