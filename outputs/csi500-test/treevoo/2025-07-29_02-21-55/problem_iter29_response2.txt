```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'].pct_change()
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Calculate 30-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    ema_volatility = daily_returns.ewm(span=30, adjust=False).std()
    
    # Apply Threshold to Identify Significant Movements
    dynamic_threshold = 2 * ema_volatility
    significant_movements = (combined_indicator.abs() > dynamic_threshold)
    
    # Calculate Daily Log Return
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_returns = log_returns.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False, alpha=0.1).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_ma_return = ma_log_returns * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_ma_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_ma_return.ewm(span=30, adjust=False).mean()
    
    # Confirm with Price Momentum
    sma_close = df['close'].rolling(window=30).mean()
    momentum_score = df['close'] - sma_close
    
    # Synthesize Final Indicator
    final_indicator = (
        0.5 * significant_movements * combined_indicator +  # Higher weight to Intraday and Volume Indicator
        0.3 * momentum_score +  # Lower weight to Momentum Score
        0.2 * (df['open'] - df['close'].shift(1))  # Opening Gap Analysis
    )
    
    # Introduce Volatility-Adjusted Momentum
    ema_abs_log_returns = np.abs(log_returns).ewm(span=30, adjust=False).mean()
    volatility_adjusted_momentum = momentum_score / ema_abs_log_returns
    
    # Integrate all components
    final_indicator = (
        0.5 * significant_movements * combined_indicator +  # Higher weight to Intraday and Volume Indicator
        0.3 * volatility_adjusted_momentum +  # Adjusted Momentum Score
        0.2 * (df['open'] - df['close'].shift(1))  # Opening Gap Analysis
    )
    
    return final_indicator
```
