```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_high_low_pct_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    std_20_day = df['close'].pct_change().rolling(window=20).std()
    threshold = 2 * std_20_day
    significant_movements = combined_indicator > threshold
    
    # Calculate Intraday Price Movement
    intraday_range = df['high'] - df['low']
    intraday_direction = df['close'] - df['open']
    
    # Calculate Weighted Volume Activity
    avg_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / avg_volume
    
    # Combine Intraday Price and Volume Metrics
    weighted_intraday_range = intraday_range * relative_volume
    weighted_intraday_direction = intraday_direction * relative_volume
    combined_intraday_metric = weighted_intraday_range + weighted_intraday_direction
    
    # Apply Moving Averages to Intraday High-to-Low Momentum
    fast_ema = intraday_high_low_pct_change.ewm(span=10, adjust=False).mean()
    slow_ema = intraday_high_low_pct_change.ewm(span=20, adjust=False).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_20_log_returns = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_ma_return = ma_20_log_returns * (ema_20_volume / avg_volume)
    
    # Generate Final Alpha Factor
    smoothed_adjusted_return = adjusted_ma_return.rolling(window=5).mean()
    
    # Enhance with MACD Signal
    macd_line = fast_ema - slow_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    final_alpha_factor = macd_line - signal_line + smoothed_adjusted_return
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    price_momentum = df['close'] - sma_close
    volume_threshold = 1.5 * avg_volume
    
    # Synthesize Final Indicator
    momentum_and_volume_indicators = 0.5 * price_momentum + 0.5 * combined_intraday_metric * (df['volume'] > volume_threshold)
    
    return momentum_and_volume_indicators
```
