```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = df['high'] / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intra'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()
    df['slow_ema_intra'] = df['intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    macd = df['fast_ema_intra'] - df['slow_ema_intra']
    weighted_momentum = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['momentum_macd'] = weighted_momentum + macd
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['volume_ratio'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else (-1 if x < 1 else 0))
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Long-Term Price Trend
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['20d_log_returns_ma'] = df['log_returns'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_momentum'] = (df['fast_ema_intra'] * df['20d_log_returns_ma']) / df['ema_volume_20'] - df['slow_ema_intra']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adj_momentum'] = df['adjusted_momentum'].rolling(window=5).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['20d_log_returns'] = df['log_returns'].rolling(window=20).sum() / 20
    
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Apply Threshold to Identify Significant Movements
    df['std_dev_25'] = df['close'].rolling(window=25).std()
    threshold = 0.5
    df['significant_movement'] = (np.abs(df['volume_change_ratio']) > threshold) & (np.abs(df['log_returns']) > df['std_dev_25'])
    
    # Generate Final Alpha Factor
    df['weighted_intraday_range'] = df['daily_range'] * df['volume_ratio']
    df['weighted_intraday_direction'] = df['intraday_momentum'] * df['volume_ratio']
    df['final_alpha_factor'] = 0.7 * df['momentum_macd'] + df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    return df['final_alpha_factor']
```
