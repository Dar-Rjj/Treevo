```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted amount
    vwa = df['amount'] / df['volume']
    
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Generate the alpha factor using a combination of the above calculations
    alpha_factor = (avg_price * vwa) + price_diff
    
    # Return the alpha factor as a pandas Series
    return alpha_factor
```
