```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / daily_range.rolling(window=25).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / daily_range.rolling(window=10).mean()
    
    # Combine Momentum Indicators
    combined_momentum = 0.7 * long_term_momentum + 0.3 * short_term_momentum
    
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_high_low_pct_change * volume_change_ratio
    
    # Adjust for Market Volatility using Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=25).std()
    
    # Apply Threshold to Identify Significant Movements
    threshold = 0.02  # Fixed threshold
    significant_movements = combined_indicator * (combined_indicator > (threshold * volatility))
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * significant_movements
    
    return alpha_factor
```
