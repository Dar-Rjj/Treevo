```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / daily_range.rolling(window=25).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / daily_range.rolling(window=10).mean()
    
    # Combine Momentum Indicators
    combined_momentum = 0.7 * long_term_momentum + 0.3 * short_term_momentum
    
    # Calculate Intraday High-to-Low Percentage Change
    intraday_hl_pct_change = (df['high'] - df['low']) / df['low'] * 100
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_hl_pct_change * volume_ratio
    
    # Adjust for Market Volatility
    daily_returns = df['close'] - df['close'].shift(1)
    std_dev_daily_returns = daily_returns.rolling(window=20).std()
    adjusted_combined_indicator = combined_indicator * std_dev_daily_returns
    
    # Confirm with Price Momentum
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Final Alpha Factor
    # Combine Weighted Momentum, Adjusted Combined Indicator, and Momentum Score
    final_factor = combined_momentum + adjusted_combined_indicator + momentum_score
    
    # Synthesize Final Indicator
    volume_threshold = 1.5 * df['volume'].rolling(window=20).mean()
    final_factor = final_factor * (df['volume'] > volume_threshold)
    
    return final_factor
```
