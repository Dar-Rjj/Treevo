```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Apply Moving Averages
    fast_ema = df['intraday_hl_pct_change'].ewm(span=10, adjust=False).mean()
    slow_ema = df['intraday_hl_pct_change'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_change_ratio']
    df['threshold'] = 2 * df['close'].pct_change().rolling(window=20).std()
    df['significant_movement'] = (df['combined_indicator'] > df['threshold']).astype(int)
    
    # Calculate Long-Term Volume-Weighted Momentum
    volume_weighted_long = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=21).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Short-Term Volume-Weighted Momentum
    volume_weighted_short = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=6).sum() - df['close'].rolling(window=5).mean()
    
    # Combine Momentum Indicators with Enhanced MACD
    macd_diff = fast_ema - slow_ema
    combined_momentum = 0.6 * volume_weighted_long + 0.4 * volume_weighted_short + macd_diff
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_return = df['log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume and Amount
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    ema_amount = df['amount'].ewm(span=20, adjust=False).mean()
    
    # Calculate Simple Moving Average of Volume and Amount
    sma_volume = df['volume'].rolling(window=20).mean()
    sma_amount = df['amount'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume and Amount
    adjusted_return = ma_log_return * (ema_volume / sma_volume) * (ema_amount / sma_amount)
    
    # Smooth the Adjusted Return
    final_alpha_factor = adjusted_return.rolling(window=5).mean()
    
    return final_alpha_factor
```
