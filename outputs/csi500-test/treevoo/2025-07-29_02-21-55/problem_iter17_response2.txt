```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Calculate 20-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    ema_std = daily_returns.ewm(span=20).std()
    
    # Apply Threshold to Identify Significant Movements
    dynamic_threshold = 2 * ema_std
    significant_movements = combined_indicator > dynamic_threshold
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_returns = daily_returns.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = moving_avg_returns * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_return.ewm(span=30, adjust=False).mean()
    
    # Confirm with Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Synthesize Final Indicator
    final_indicator = (combined_indicator * significant_movements) + price_momentum
    final_indicator = final_indicator.ewm(span=10, adjust=False).mean()  # Apply EMA for smoothing
    
    return final_indicator
```
