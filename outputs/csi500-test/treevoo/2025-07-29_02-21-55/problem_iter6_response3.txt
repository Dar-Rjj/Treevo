import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Components
    open_to_high_return = (df['high'] - df['open']) / df['open']
    open_to_low_return = (df['low'] - df['open']) / df['open']
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Returns
    average_intraday_return = (open_to_high_return + open_to_low_return + close_to_open_return) / 3
    intraday_return_difference = (max(open_to_high_return, open_to_low_return, close_to_open_return) -
                                  min(open_to_high_return, open_to_low_return, close_to_open_return))
    
    # Volume Adjusted Intraday Returns
    volume_weighted_average = df['volume'] * average_intraday_return
    volume_weighted_difference = df['volume'] * intraday_return_difference
    
    # Generate Final Alpha Factor
    final_alpha = volume_weighted_average - volume_weighted_difference
    
    # Integrate with Previous Day's Alpha
    if 'previous_day_alpha' in df.columns:
        previous_day_alpha = df['previous_day_alpha']
    else:
        previous_day_alpha = 0.0
    
    decay_factor = 0.95
    weighted_sum = 0.7 * final_alpha + 0.3 * (decay_factor * previous_day_alpha)
    
    # Incorporate Volume Trend
    volume_trend = df['volume'].rolling(window=5).mean()
    final_alpha_adjusted = weighted_sum * volume_trend
    
    return pd.Series(final_alpha_adjusted, index=df.index)
