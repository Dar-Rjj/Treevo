```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_ratio']
    threshold = 1.5
    df['significant_movement'] = df['combined_indicator'].apply(lambda x: x if abs(x) > threshold else 0)
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['sma_close']
    
    volume_threshold = 1.5 * df['volume'].rolling(window=lookback_period).mean()
    df['volume_above_threshold'] = df['volume'] > volume_threshold
    
    # Synthesize Final Indicator
    momentum_weight = 0.5
    volume_weight = 0.5
    df['final_momentum_volume'] = (df['momentum_score'] * momentum_weight + 
                                   df['significant_movement'] * volume_weight * df['volume_above_threshold'])
    
    # Integrate Opening and Closing Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    positive_multiplier = 1.5
    negative_multiplier = -1.5
    df['gap_direction'] = df['price_gap'].apply(lambda x: x * positive_multiplier if x > 0 else x * negative_multiplier)
    
    # Combine All Indicators
    gap_weight = 0.3
    intraday_weight = 0.2
    final_weight = 0.5
    df['final_indicator'] = (df['gap_direction'] * gap_weight + 
                            df['intraday_hl_pct_change'] * intraday_weight + 
                            df['final_momentum_volume'] * final_weight)
    
    return df['final_indicator']
```
