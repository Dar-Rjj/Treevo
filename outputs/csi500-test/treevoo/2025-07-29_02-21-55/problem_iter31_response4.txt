```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's closing price and yesterday's closing price
    df['close_diff'] = df['close'].diff()
    
    # Label as Bullish or Bearish
    df['signal'] = 'Bullish'
    df.loc[df['close_diff'] < 0, 'signal'] = 'Bearish'
    
    # For bullish signals, calculate the ratio of close to open
    df['bullish_factor'] = df['close'] / df['open']
    
    # For bearish signals, calculate the ratio of open to close
    df['bearish_factor'] = df['open'] / df['close']
    
    # Calculate the distance between today's high and low prices
    df['price_range'] = df['high'] - df['low']
    
    # Define a threshold for high/low volatility (e.g., average range * 1.5)
    threshold = df['price_range'].mean() * 1.5
    
    # Label as High Volatility or Low Volatility
    df['volatility'] = 'Low Volatility'
    df.loc[df['price_range'] > threshold, 'volatility'] = 'High Volatility'
    
    # For high volatility days, calculate the percentage change from the opening price to the closing price
    df['high_volatility_change'] = (df['close'] - df['open']) / df['open']
    
    # For low volatility days, look at the position of the closing price relative to the day's high and low
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Look at the volume traded and compare to recent average
    df['volume_mean'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_mean']
    
    # When there's a large volume with an increase in the closing price
    df['volume_bullish'] = (df['volume_ratio'] > 2) & (df['close_diff'] > 0)
    
    # Conversely, a large volume with a decrease in the closing price
    df['volume_bearish'] = (df['volume_ratio'] > 2) & (df['close_diff'] < 0)
    
    # Consider the amount traded, not just volume
    df['amount_mean'] = df['amount'].rolling(window=20).mean()
    df['amount_ratio'] = df['amount'] / df['amount_mean']
    
    # An above-average amount with rising prices might suggest a robust upturn
    df['amount_bullish'] = (df['amount_ratio'] > 1) & (df['close_diff'] > 0)
    
    # For decreases, check if the price is also falling
    df['amount_bearish'] = (df['amount_ratio'] > 1) & (df['close_diff'] < 0)
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['bullish_factor'] * df['volume_bullish'] * df['amount_bullish'] +
        (1 - df['bearish_factor']) * df['volume_bearish'] * df['amount_bearish'] +
        df['high_volatility_change'] * (df['volatility'] == 'High Volatility') +
        df['close_position'] * (df['volatility'] == 'Low Volatility')
    )
    
    return df['alpha_factor']
```
