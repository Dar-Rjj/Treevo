```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_change_ratio
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    ema_std_20 = daily_returns.ewm(span=20).std()
    dynamic_threshold = 2 * ema_std_20
    significant_movements = (combined_indicator > dynamic_threshold) * combined_indicator
    
    # Confirm with Price Momentum
    lookback_period = 20
    close_ema_20 = df['close'].ewm(span=lookback_period).mean()
    momentum_score = df['close'] - close_ema_20
    
    # Combine Momentum and Volume Indicators
    volume_ema_20 = df['volume'].ewm(span=lookback_period).mean()
    volume_threshold = 1.5 * volume_ema_20
    significant_volume = (df['volume'] > volume_threshold) * significant_movements
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Enhance with Trend Confirmation
    def calculate_adx(high, low, close, lookback):
        plus_dm = high.diff()
        minus_dm = low.diff()
        tr = pd.Series(0, index=range(len(close)))
        tr[1:] = np.maximum(np.abs(high[1:] - low[1:]), 
                            np.maximum(np.abs(high[1:] - close[:-1]), 
                                       np.abs(low[1:] - close[:-1])))
        
        tr = tr.ewm(span=lookback, min_periods=lookback).mean()
        plus_di = 100 * (plus_dm.ewm(span=lookback, min_periods=lookback).mean() / tr)
        minus_di = 100 * (minus_dm.ewm(span=lookback, min_periods=lookback).mean() / tr)
        
        dx = 100 * np.abs(plus_di - minus_di) / (plus_di + minus_di)
        adx = dx.ewm(span=lookback, min_periods=lookback).mean()
        return adx
    
    adx = calculate_adx(df['high'], df['low'], df['close'], lookback_period)
    trend_confirmation = (adx > 25) * adx
    
    # Synthesize Final Indicator
    weights = {
        'significant_movements': 0.4,
        'momentum_score': 0.3,
        'opening_gap': 0.1,
        'trend_confirmation': 0.2
    }
    
    final_indicator = (weights['significant_movements'] * significant_movements +
                       weights['momentum_score'] * momentum_score +
                       weights['opening_gap'] * opening_gap +
                       weights['trend_confirmation'] * trend_confirmation)
    
    return final_indicator
```
