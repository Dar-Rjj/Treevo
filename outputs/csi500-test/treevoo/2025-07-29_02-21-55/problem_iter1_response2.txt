```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Close Price Change (e.g., 20 days)
    window = 20
    df['price_momentum'] = df['close_change'].rolling(window=window).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Define Volume Spike Threshold (e.g., 2x recent volume)
    threshold_multiplier = 2.0
    recent_volume = df['volume'].rolling(window=window).mean()
    df['volume_spike'] = (df['volume_change'] > threshold_multiplier * recent_volume).astype(int)
    
    # Combine Price Momentum and Volume Spike
    df['alpha_factor'] = df['price_momentum']
    high_factor = 2.0
    df.loc[df['volume_spike'] == 1, 'alpha_factor'] *= high_factor
    
    return df['alpha_factor']
```
