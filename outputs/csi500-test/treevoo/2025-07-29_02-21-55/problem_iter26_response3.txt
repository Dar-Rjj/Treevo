import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['Intraday_Pct_Change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Combined_Intraday_Volume'] = df['Intraday_Pct_Change'] * df['Volume_Ratio']
    
    # Confirm with Price Momentum
    df['Long_Term_Momentum'] = (df['close'] - df['close'].shift(20)) / df['Daily_Range'].rolling(window=20).mean()
    df['Short_Term_Momentum'] = (df['close'] - df['close'].shift(5)) / df['Daily_Range'].rolling(window=5).mean()
    df['Combined_Momentum'] = 0.7 * df['Long_Term_Momentum'] + 0.3 * df['Short_Term_Momentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['Fast_EMA_Intraday'] = df['Intraday_Pct_Change'].ewm(span=12, adjust=False).mean()
    df['Slow_EMA_Intraday'] = df['Intraday_Pct_Change'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['MACD'] = df['Fast_EMA_Intraday'] - df['Slow_EMA_Intraday']
    df['Weighted_Momentum'] = 0.4 * df['Long_Term_Momentum'] + 0.6 * df['Short_Term_Momentum']
    df['MACD_Momentum'] = df['MACD'] + df['Weighted_Momentum']
    
    # Incorporate Volume Confirmation
    df['Average_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio_Adjusted'] = np.where(df['volume'] / df['Average_Volume_20'] > 1.5, 1, 
                                           np.where(df['volume'] / df['Average_Volume_20'] < 0.5, -1, 0))
    
    # Adjust Combined Momentum by Volume
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['20_Day_Log_Returns_MA'] = df['Log_Returns'].rolling(window=20).mean()
    df['20_Day_EMA_Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Adjusted_Combined_Momentum'] = (df['Fast_EMA_Intraday'] * df['20_Day_Log_Returns_MA']) / df['20_Day_EMA_Volume'] - df['Slow_EMA_Intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['Smoothed_Adjusted_Momentum'] = df['Adjusted_Combined_Momentum'].rolling(window=5).mean()
    
    # Apply a Weighted Sum
    df['Weighted_Sum'] = 0.6 * df['Combined_Intraday_Volume'] + 0.4 * df['Combined_Momentum']
    
    # Synthesize Final Indicator
    df['Final_Indicator'] = df['Combined_Intraday_Volume'] * 0.7 + df['Combined_Momentum'] * 0.3
    df['Mkt_Volatility'] = df['close'].pct_change().rolling(window=30).std()
    df['Final_Indicator_Adjusted'] = df['Final_Indicator'] / df['Mkt_Volatility']
    
    return df['Final_Indicator_Adjusted']
