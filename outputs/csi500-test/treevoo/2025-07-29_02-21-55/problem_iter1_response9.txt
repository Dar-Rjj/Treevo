```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Price Momentum
    df['price_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate 20-day Average Volume
    df['avg_20_vol'] = df['volume'].rolling(window=20).mean().shift(1)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['avg_20_vol']
    
    # Set Volume Confirmation
    df['volume_confirmation'] = 0
    df.loc[df['volume_ratio'] > 1, 'volume_confirmation'] = 1
    df.loc[df['volume_ratio'] < 1, 'volume_confirmation'] = -1
    
    # Final Alpha Factor: Combine Price Momentum and Volume Confirmation
    df['alpha_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    return df['alpha_factor']
```
