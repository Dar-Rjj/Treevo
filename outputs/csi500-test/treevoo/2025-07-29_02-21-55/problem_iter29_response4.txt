```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_intraday_volume = intraday_pct_change * volume_ratio
    
    # Confirm with Price Momentum
    lookback_period = 20
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Synthesize Interim Indicator
    interim_indicator = (combined_intraday_volume * 0.5) + (momentum_score * 0.5)
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_indicator = (interim_indicator * 0.7) + (opening_gap * 0.3)
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    threshold = 2 * std_dev_20
    
    # Finalize Alpha Factor
    alpha_factor = final_indicator.where(final_indicator.abs() > threshold, 0)
    
    return alpha_factor
```
