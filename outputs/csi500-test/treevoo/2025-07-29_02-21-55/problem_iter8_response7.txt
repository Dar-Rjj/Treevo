```python
def heuristics_v2(df):
    # Calculate the average of (high - low) over the last 5 days
    avg_high_low = df['high'] - df['low']
    avg_high_low = avg_high_low.rolling(window=5).mean()
    
    # Calculate the difference between close and open
    diff_close_open = df['close'] - df['open']
    
    # Calculate the ratio of volume to amount
    vol_to_amount = df['volume'] / df['amount']
    
    # Combine the above signals into a single factor
    factor = (avg_high_low * diff_close_open) / vol_to_amount
    
    return factor
```
