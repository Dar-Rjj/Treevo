```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=25).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=10).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.7 * df['long_term_momentum'] + 0.3 * df['short_term_momentum']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate 25-Day Moving Standard Deviation of Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    df['25_day_std'] = df['daily_returns'].rolling(window=25).std()
    
    # Adjust for Market Volatility
    df['adjusted_intraday_hl'] = df['intraday_hl_pct_change'] * df['volume_change_ratio'] / df['25_day_std']
    
    # Apply Threshold to Identify Significant Movements
    fixed_threshold = 0.05
    df['significant_movement'] = df['adjusted_intraday_hl'] > fixed_threshold
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['significant_movement']
    
    return df['alpha_factor']
```
