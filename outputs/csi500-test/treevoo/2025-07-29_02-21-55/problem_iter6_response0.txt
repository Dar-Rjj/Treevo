```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Apply Moving Averages
    fast_ema = intraday_momentum.ewm(span=12, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Volume-Weighted Momentum
    long_term_vol_weighted_momentum = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    long_term_vol_weighted_momentum -= df['close'].rolling(window=20).mean()
    
    # Calculate Short-Term Volume-Weighted Momentum
    short_term_vol_weighted_momentum = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    short_term_vol_weighted_momentum -= df['close'].rolling(window=5).mean()
    
    # Combine Momentum Indicators with Enhanced MACD
    weighted_long_term_momentum = 0.6 * long_term_vol_weighted_momentum
    weighted_short_term_momentum = 0.4 * short_term_vol_weighted_momentum
    weighted_sum = weighted_long_term_momentum + weighted_short_term_momentum
    
    macd_line = fast_ema - slow_ema
    enhanced_macd = weighted_sum + macd_line
    
    return enhanced_macd
```
