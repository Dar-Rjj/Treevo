import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the given DataFrame.
    The factor is derived from the following logic:
    1. Compute the difference between the high and low prices for each day.
    2. Calculate the ratio of the close price to the open price.
    3. Determine the average trading volume over the last 5 days.
    4. Compute the weighted sum of the above three components.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # 1. Compute the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # 2. Calculate the ratio of the close price to the open price
    close_open_ratio = df['close'] / df['open']
    
    # 3. Determine the average trading volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # 4. Compute the weighted sum of the components
    factor_values = (0.4 * price_range) + (0.3 * close_open_ratio) + (0.3 * avg_volume_5d)
    
    return factor_values
