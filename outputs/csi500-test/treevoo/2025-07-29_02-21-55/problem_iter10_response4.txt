```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Direction
    intraday_direction = df['close'] - df['open']
    
    # Calculate Relative Volume
    average_volume = df['volume'].rolling(window=n).mean()
    relative_volume = df['volume'] / average_volume
    
    # Combine Intraday Price and Volume
    combined_metric = intraday_direction * relative_volume
    
    # Generate Final Alpha Factor
    alpha_factor = combined_metric
    
    return alpha_factor
```
