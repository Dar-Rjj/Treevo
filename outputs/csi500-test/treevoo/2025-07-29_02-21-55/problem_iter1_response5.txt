```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] / df['low']
    
    # Apply Moving Averages
    fast_ema = intraday_momentum.ewm(span=10, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=20, adjust=False).mean()
    
    # MACD of Intraday Momentum
    macd_intraday_momentum = fast_ema - slow_ema
    
    return macd_intraday_momentum
```
