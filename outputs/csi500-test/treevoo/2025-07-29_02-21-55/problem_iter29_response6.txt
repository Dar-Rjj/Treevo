```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['IntradayHLChange'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['VolumeRatio'] = df['volume'] / df['volume'].shift(1)
    df['CombinedIntradayVolume'] = df['IntradayHLChange'] * df['VolumeRatio']
    
    # Confirm with Price Momentum
    lookback_long = 20
    lookback_short = 5
    df['SMA_Close_20'] = df['close'].rolling(window=lookback_long).mean()
    df['LongTermMomentum'] = (df['close'] - df['close'].shift(lookback_long)) / df['DailyRange'].rolling(window=lookback_long).mean()
    df['ShortTermMomentum'] = (df['close'] - df['close'].shift(lookback_short)) / df['DailyRange'].rolling(window=lookback_short).mean()
    df['MomentumScore'] = df['close'] - df['SMA_Close_20']
    df['CombinedMomentum'] = 0.7 * df['LongTermMomentum'] + 0.3 * df['ShortTermMomentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['FastEMA_Intraday'] = df['IntradayHLChange'].ewm(span=12, adjust=False).mean()
    df['SlowEMA_Intraday'] = df['IntradayHLChange'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['WeightedMomentumSum'] = 0.4 * df['LongTermMomentum'] + 0.6 * df['ShortTermMomentum']
    df['MACD'] = df['FastEMA_Intraday'] - df['SlowEMA_Intraday']
    df['CombinedMomentumMACD'] = df['WeightedMomentumSum'] + df['MACD']
    
    # Incorporate Volume Confirmation
    df['AvgVolume_20'] = df['volume'].rolling(window=lookback_long).mean()
    df['VolumeRatio_20'] = df['volume'] / df['AvgVolume_20']
    df['VolumeConfirmation'] = np.where(df['VolumeRatio_20'] > 1.5, 1, np.where(df['VolumeRatio_20'] < 0.5, -1, 0))
    
    # Adjust Combined Momentum by Volume
    df['LogReturns'] = np.log(df['close'] / df['close'].shift(1))
    df['MA_LogReturns_20'] = df['LogReturns'].rolling(window=lookback_long).mean()
    df['EMA_Volume_20'] = df['volume'].ewm(span=lookback_long, adjust=False).mean()
    df['AdjustedCombinedMomentum'] = (df['FastEMA_Intraday'] * df['MA_LogReturns_20']) / df['EMA_Volume_20'] - df['SlowEMA_Intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['SmoothedAdjustedCombinedMomentum'] = df['AdjustedCombinedMomentum'].rolling(window=5).mean()
    
    # Apply a Weighted Sum
    weight_intraday_volume = 0.6
    weight_momentum_score = 0.4
    df['WeightedSum'] = weight_intraday_volume * df['CombinedIntradayVolume'] + weight_momentum_score * df['MomentumScore']
    
    # Synthesize Final Indicator
    final_weight_intraday_volume = 0.8
    final_weight_momentum_score = 0.2
    df['FinalIndicator'] = final_weight_intraday_volume * df['CombinedIntradayVolume'] + final_weight_momentum_score * df['MomentumScore']
    
    # Apply Threshold to Identify Significant Movements
    df['DailyReturn'] = df['close'].pct_change()
    df['StdDev_30'] = df['DailyReturn'].rolling(window=30).std()
    threshold = 2 * df['StdDev_30']
    df['FinalIndicator'] = np.where(np.abs(df['FinalIndicator']) > threshold, df['FinalIndicator'], 0)
    
    return df['FinalIndicator']
```
