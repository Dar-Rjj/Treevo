def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the average price for the day
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average price using volume
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the daily turnover
    df['turnover'] = df['amount'] / df['close']
    
    # Calculate the factor as a combination of price change, average price, and turnover
    df['factor'] = (df['price_change'] / df['avg_price']) * (df['turnover'] / df['vwap'])
    
    return df['factor']
