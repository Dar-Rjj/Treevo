```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_ratio']
    threshold = 1.5  # Set a fixed threshold
    df['significant_movement'] = df['combined_indicator'].apply(lambda x: x if abs(x) > threshold else 0)
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['sma_close']
    
    # Synthesize Final Indicator
    weight_combined = 0.6  # Weight for the combined intraday and volume indicator
    weight_momentum = 0.4  # Weight for the momentum score
    df['final_indicator'] = (weight_combined * df['significant_movement'] + 
                             weight_momentum * df['momentum_score'])
    
    return df['final_indicator']
```
