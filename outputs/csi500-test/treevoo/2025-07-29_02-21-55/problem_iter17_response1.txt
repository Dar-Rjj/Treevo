import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = df['high'] / df['low'] - 1
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = df['intraday_momentum'].ewm(span=12, adjust=False).mean()
    slow_ema = df['intraday_momentum'].ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    macd = fast_ema - slow_ema
    weighted_momentum = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    combined_momentum = weighted_momentum + macd
    
    # Incorporate Volume Confirmation
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    volume_indicator = np.where(volume_ratio > 1, 1, np.where(volume_ratio < 1, -1, 0))
    
    # Adjust Combined Momentum by Volume
    log_returns = np.log(df['close'] / df['close'].shift(1))
    ma_log_returns_20 = log_returns.rolling(window=20).mean()
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_momentum = (fast_ema * ma_log_returns_20 / ema_volume_20) - slow_ema
    
    # Smooth the Adjusted Combined Momentum
    smoothed_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Calculate Intraday High-to-Low Percentage Change
    intraday_percentage_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_percentage_change * volume_change_ratio
    
    # Dynamic Threshold for Significant Movements
    ema_std_20 = df['close'].ewm(span=20, adjust=False).std()
    threshold = 2 * ema_std_20
    significant_movement = np.where(combined_indicator > threshold, 1, np.where(combined_indicator < -threshold, -1, 0))
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close_20 = df['close'].rolling(window=lookback_period).mean()
    price_momentum_score = df['close'] - sma_close_20
    
    # Combine Momentum and Volume Indicators
    volume_threshold = 1.5 * df['volume'].rolling(window=lookback_period).mean()
    final_indicator = 0.5 * significant_movement + 0.5 * price_momentum_score
    
    # Ensure the Weighted Sum Reflects Both Momentum and Volume Significance
    final_factor = final_indicator * np.where(df['volume'] > volume_threshold, 1, 0)
    
    return final_factor.dropna()
