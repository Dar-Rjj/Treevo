```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, R=14, P=14, T=10, Q=30, S=10):
    # Momentum Indicators
    df['WMA'] = df['close'].rolling(window=N).apply(lambda x: np.average(x, weights=np.arange(1, len(x)+1)))
    df['Momentum_WMA'] = df['close'] - df['WMA']
    
    df['DEMA'] = 2 * df['close'].ewm(span=N, adjust=False).mean() - df['close'].ewm(span=N, adjust=False).mean().ewm(span=N, adjust=False).mean()
    df['Momentum_DEMA'] = df['close'] - df['DEMA']
    
    # Reversal Indicators
    df['Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Reversal_Signal'] = (df['close'] < df['low'].rolling(window=N).mean()).astype(int)
    
    # Volume-Based Indicators
    df['Sum_Volume_M'] = df['volume'].rolling(window=M).sum()
    df['OBV'] = (df['close'] > df['close'].shift(1)) * df['volume'] + (df['close'] <= df['close'].shift(1)) * -df['volume']
    df['OBV'] = df['OBV'].cumsum()
    df['Volume_Ratio_R'] = df['volume'] / df['volume'].rolling(window=R).sum()
    
    # Range-Based Indicators
    df['True_Range'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=P).mean()
    df['Relative_Range'] = df['ATR'] / (df['high'] - df['low'])
    
    # Open, High, Low, Close Combinations
    df['Intraday_Performance'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['Bullish_Days'] = ((df['close'] > df['open']) & (df['close'] > df['close'].shift(1))).astype(int)
    df['Trend_Strength_T'] = df['Intraday_Performance'].rolling(window=T).mean()
    
    # Amount Data for Indicators
    df['Total_Amount_Q'] = df['amount'].rolling(window=Q).sum()
    df['Amount_Volume_Ratio'] = df['amount'] / df['volume']
    df['Amount_Ratio_S'] = df['amount'] / df['amount'].rolling(window=S).sum()
    
    # Combine all alpha factors into a single series
    alpha_factors = (
        df['Momentum_WMA'] + 
        df['Momentum_DEMA'] + 
        df['Reversal_Signal'] + 
        df['OBV'] + 
        df['Volume_Ratio_R'] + 
        df['Relative_Range'] + 
        df['Trend_Strength_T'] + 
        df['Amount_Ratio_S']
    )
    
    return alpha_factors
```
