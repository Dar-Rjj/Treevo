```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 1.5
    market_volatility = df['close'].pct_change().rolling(window=30).std()
    significant_movement = (combined_indicator > threshold) & (combined_indicator > market_volatility)
    
    # Confirm with Price Momentum
    lookback_period = 30
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Combine Momentum and Volume Indicators
    dynamic_volume_threshold = 2.0 * df['volume'].rolling(window=lookback_period).mean()
    volume_significant = df['volume'] > dynamic_volume_threshold
    
    # Synthesize Final Indicator
    final_indicator = (0.7 * combined_indicator + 0.3 * momentum_score) * significant_movement * volume_significant
    
    return final_indicator
```
