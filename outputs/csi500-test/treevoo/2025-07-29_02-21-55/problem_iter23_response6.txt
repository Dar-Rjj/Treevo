```python
def heuristics_v2(df):
    # Calculate the relative strength compared to previous day
    df['relative_strength'] = df['close'].pct_change()
    
    # Calculate the buying pressure: (close - low) / (high - low)
    df['buying_pressure'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the selling pressure: (high - close) / (high - low)
    df['selling_pressure'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the amount/volume ratio
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Combine the factors
    factor = (df['relative_strength'] 
              + df['buying_pressure'] 
              - df['selling_pressure'] 
              + df['amount_volume_ratio'])
    
    return factor
```
