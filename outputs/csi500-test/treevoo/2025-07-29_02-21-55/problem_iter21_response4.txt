```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A series indexed by date representing the factor values.
    """
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the ratio of amount to volume
    amount_to_volume_ratio = df['amount'] / df['volume']
    
    # Combine the factors
    factor = (price_range * close_change) + amount_to_volume_ratio
    
    return factor
```
