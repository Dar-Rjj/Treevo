```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_pct_change'] * df['volume_ratio']
    threshold = 2.0  # Example threshold
    df['significant_movement'] = np.where(df['combined_indicator'] > threshold, 1, 0)
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_ma_return'] = (df['ma_log_return'] * df['ema_volume']) / df['sma_volume']
    
    # Apply Weighted Moving Averages
    fast_ema = df['adjusted_ma_return'].ewm(span=10, adjust=False).mean()
    slow_ema = df['adjusted_ma_return'].ewm(span=30, adjust=False).mean()
    df['fast_ema'] = fast_ema
    df['slow_ema'] = slow_ema
    
    # Confirm with Price Momentum
    df['sma_close'] = df['close'].rolling(window=20).mean()
    df['momentum_score'] = df['close'] - df['sma_close']
    
    # Synthesize Final Indicator
    # Combine Intraday and Volume Indicator
    weight_intraday = 0.3
    weight_momentum = 0.7
    df['final_indicator'] = (weight_intraday * df['combined_indicator']) + (weight_momentum * df['momentum_score'])
    
    # Introduce Volatility-Adjusted Momentum
    df['volatility_ema'] = df['log_return'].abs().ewm(span=20, adjust=False).mean()
    df['adjusted_momentum'] = df['momentum_score'] / df['volatility_ema']
    
    # Final weighted sum
    df['final_factor'] = (weight_intraday * df['combined_indicator']) + (weight_momentum * df['adjusted_momentum'])
    
    return df['final_factor']
```
