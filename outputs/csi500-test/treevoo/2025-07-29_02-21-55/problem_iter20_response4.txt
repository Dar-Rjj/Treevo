```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['IntradayHLChange'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['VolumeRatio'] = df['volume'] / df['volume'].shift(1)
    df['CombinedIntradayVolume'] = df['IntradayHLChange'] * df['VolumeRatio']
    
    # Confirm with Price Momentum
    df['LongTermMomentum'] = (df['close'] - df['close'].shift(20)) / df['DailyRange'].rolling(window=20).mean()
    df['ShortTermMomentum'] = (df['close'] - df['close'].shift(5)) / df['DailyRange'].rolling(window=5).mean()
    df['CombinedMomentum'] = 0.6 * df['LongTermMomentum'] + 0.4 * df['ShortTermMomentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['FastEMA_Intra'] = df['IntradayHLChange'].ewm(span=12, adjust=False).mean()
    df['SlowEMA_Intra'] = df['IntradayHLChange'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['MACD'] = df['FastEMA_Intra'] - df['SlowEMA_Intra']
    df['WeightedMomentum'] = 0.45 * df['LongTermMomentum'] + 0.55 * df['ShortTermMomentum']
    df['MACD_Combined'] = df['WeightedMomentum'] + df['MACD']
    
    # Incorporate Volume Confirmation
    df['AvgVolume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio'] = df['volume'] / df['AvgVolume_20']
    df['Volume_Confirmation'] = 0
    df.loc[df['Volume_Ratio'] > 1, 'Volume_Confirmation'] = 1
    df.loc[df['Volume_Ratio'] < 1, 'Volume_Confirmation'] = -1
    
    # Adjust Combined Momentum by Volume
    df['LogReturns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['MA_LogReturns_20'] = df['LogReturns'].rolling(window=20).mean()
    df['EMA_Volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['AdjustedCombinedMomentum'] = (df['FastEMA_Intra'] * df['MA_LogReturns_20']) / df['EMA_Volume_20'] - df['SlowEMA_Intra']
    
    # Smooth the Adjusted Combined Momentum
    df['SmoothedAdjustedMomentum'] = df['AdjustedCombinedMomentum'].rolling(window=5).mean()
    
    # Synthesize Final Indicator
    df['FinalIndicator'] = 0.5 * df['CombinedIntradayVolume'] + 0.5 * df['SmoothedAdjustedMomentum']
    
    return df['FinalIndicator'].dropna()
```
