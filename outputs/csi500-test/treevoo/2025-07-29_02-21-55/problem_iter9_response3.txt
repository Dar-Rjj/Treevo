```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=20).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.6 * df['long_term_momentum'] + 0.4 * df['short_term_momentum']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate 20-Day Moving Standard Deviation of Daily Returns
    df['daily_return'] = df['close'].pct_change()
    df['std_dev_20'] = df['daily_return'].rolling(window=20).std()
    
    # Compute Combined Indicator with Volume Spike
    df['combined_indicator'] = df['intraday_change'] * df['volume_ratio'] / df['std_dev_20']
    
    # Apply Threshold to Identify Significant Movements
    threshold = df['combined_indicator'].quantile(0.95)  # Dynamic threshold
    df['factor'] = df['combined_indicator'].apply(lambda x: x if x > threshold else 0)
    
    return df['factor'].dropna()
```
