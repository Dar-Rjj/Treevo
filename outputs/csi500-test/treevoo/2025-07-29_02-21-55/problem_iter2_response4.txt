```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Momentum
    df['IntradayMomentum'] = df['high'] - df['low']
    
    # Apply Moving Averages
    fast_period, slow_period = 12, 26
    df['FastEMA'] = df['IntradayMomentum'].ewm(span=fast_period, adjust=False).mean()
    df['SlowEMA'] = df['IntradayMomentum'].ewm(span=slow_period, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['LongTermMomentum'] = (df['close'] - df['close'].shift(20)) / df['DailyRange'].rolling(window=20).mean()
    
    # Calculate Short-Term Momentum
    df['ShortTermMomentum'] = (df['close'] - df['close'].shift(5)) / df['DailyRange'].rolling(window=5).mean()
    
    # Combine Momentum Indicators with MACD
    df['WeightedMomentum'] = 0.5 * df['LongTermMomentum'] + 0.5 * df['ShortTermMomentum']
    df['MACD'] = df['FastEMA'] - df['SlowEMA']
    
    # Final Factor
    df['AlphaFactor'] = df['WeightedMomentum'] + df['MACD']
    
    return df['AlphaFactor']
```
