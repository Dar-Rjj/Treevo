```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 1.5
    significant_movement = combined_indicator > threshold
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    market_volatility = daily_returns.rolling(window=20).std()
    adjusted_combined_indicator = combined_indicator / market_volatility
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Set a Volume Threshold
    average_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * average_volume
    significant_volume = df['volume'] > volume_threshold
    
    # Synthesize Final Indicator
    weight_combined = 0.6
    weight_momentum = 0.4
    final_indicator = (weight_combined * adjusted_combined_indicator + 
                       weight_momentum * momentum_score) * significant_movement * significant_volume
    
    return final_indicator
```
