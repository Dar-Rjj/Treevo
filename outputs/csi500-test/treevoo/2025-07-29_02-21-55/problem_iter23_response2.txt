```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_change_ratio
    
    # Adjust for Market Volatility using 20-day Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    market_volatility = daily_returns.rolling(window=20).std()
    significant_movements = (combined_indicator > 2 * market_volatility)
    
    # Confirm with Price Momentum
    lookback_period = 20
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Synthesize Final Indicator
    average_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * average_volume
    volume_above_threshold = df['volume'] > volume_threshold
    final_indicator = 0.6 * significant_movements * combined_indicator + 0.4 * momentum_score * volume_above_threshold
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_return = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = moving_avg_log_return * ema_volume / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    smoothed_return = adjusted_return.rolling(window=5).mean()
    final_alpha_factor = np.sqrt(smoothed_return)
    
    # Fast and Slow Exponential Moving Averages
    fast_ema = adjusted_return.ewm(span=12, adjust=False).mean()
    slow_ema = adjusted_return.ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term and Short-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / (df['high'] - df['low']).rolling(window=21).mean()
    short_term_momentum = (df['close'] - df['close'].shift(6)) / (df['high'] - df['low']).rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    weighted_momentum = 0.45 * long_term_momentum + 0.55 * short_term_momentum
    macd_diff = fast_ema - slow_ema
    final_momentum_macd = weighted_momentum + macd_diff
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_confirmation = np.where(volume_ratio > 1, 1, np.where(volume_ratio < 1, -1, 0))
    final_alpha_factor_with_volume = final_alpha_factor * volume_confirmation * 0.1 * df['volume'].ewm(span=20, adjust=False).mean()
    
    return final_alpha_factor_with_volume
```
