```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=22).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.6 * df['long_term_momentum'] + 0.4 * df['short_term_momentum']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    df['daily_returns'] = df['close'].pct_change()
    df['20_day_volatility'] = df['daily_returns'].rolling(window=20).std()
    
    # Set a Fixed Threshold and Adjust for Market Volatility
    threshold = 2
    df['alpha_factor'] = df['combined_indicator'] / df['20_day_volatility']
    df['alpha_factor'] = df['alpha_factor'].where(df['alpha_factor'] > threshold, 0)
    
    return df['alpha_factor']
```
