```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate the difference between today's closing price and yesterday's closing price
    df['price_diff'] = df['close'].diff()
    
    # Label as Bullish or Bearish
    df['sentiment'] = df['price_diff'].apply(lambda x: 'Bullish' if x > 0 else 'Bearish')
    
    # For bullish signals, calculate the ratio of close to open
    df['bullish_factor'] = df.apply(lambda row: row['close'] / row['open'] if row['sentiment'] == 'Bullish' else 0, axis=1)
    
    # For bearish signals, calculate the ratio of open to close
    df['bearish_factor'] = df.apply(lambda row: row['open'] / row['close'] if row['sentiment'] == 'Bearish' else 0, axis=1)
    
    # Calculate the distance between today's high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Define a threshold for high volatility (e.g., 5% of the average high-low difference)
    volatility_threshold = 0.05 * df['high_low_diff'].mean()
    
    # Label high and low volatility days
    df['volatility'] = df['high_low_diff'].apply(lambda x: 'High' if x > volatility_threshold else 'Low')
    
    # For high volatility days, calculate the percentage change from the opening price to the closing price
    df['high_vol_factor'] = df.apply(lambda row: (row['close'] - row['open']) / row['open'] if row['volatility'] == 'High' else 0, axis=1)
    
    # For low volatility days, look at the position of the closing price relative to the day's high and low
    df['low_vol_factor'] = df.apply(lambda row: (row['close'] - row['low']) / (row['high'] - row['low']) if row['volatility'] == 'Low' else 0, axis=1)
    
    # Look at the volume traded
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_sentiment'] = df.apply(lambda row: 'Strong' if row['volume'] > 1.5 * row['volume_ma'] else 'Weak', axis=1)
    
    # For strong volume sentiment, check if the closing price increased
    df['volume_price_factor'] = df.apply(lambda row: 1 if row['volume_sentiment'] == 'Strong' and row['price_diff'] > 0 else (-1 if row['volume_sentiment'] == 'Strong' and row['price_diff'] < 0 else 0), axis=1)
    
    # Consider the amount traded
    df['amount_ma'] = df['amount'].rolling(window=20).mean()
    df['amount_sentiment'] = df.apply(lambda row: 'Strong' if row['amount'] > 1.5 * row['amount_ma'] else 'Weak', axis=1)
    
    # For strong amount sentiment, check if the closing price increased
    df['amount_price_factor'] = df.apply(lambda row: 1 if row['amount_sentiment'] == 'Strong' and row['price_diff'] > 0 else (-1 if row['amount_sentiment'] == 'Strong' and row['price_diff'] < 0 else 0), axis=1)
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['bullish_factor'] + df['bearish_factor'] + 
                          df['high_vol_factor'] + df['low_vol_factor'] + 
                          df['volume_price_factor'] + df['amount_price_factor'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
