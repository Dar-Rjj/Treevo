```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']

    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_change_ratio

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_return = daily_log_return.rolling(window=20).mean()

    # Adjust Moving Average Return by Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    sma_volume = df['volume'].rolling(window=20).mean()
    adjusted_moving_avg_return = moving_avg_log_return * (ema_volume / sma_volume)

    # Apply Weighted Moving Averages
    fast_ema = adjusted_moving_avg_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_moving_avg_return.ewm(span=30, adjust=False).mean()

    # Integrate Intraday Volatility
    intraday_range = df['high'] - df['low']
    moving_avg_intraday_range = intraday_range.rolling(window=20).mean()

    # Combine Momentum, Volume, and Intraday Volatility
    alpha_factor = (adjusted_moving_avg_return * combined_indicator) / moving_avg_intraday_range

    return alpha_factor
```
