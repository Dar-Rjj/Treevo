```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_momentum * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 1.5
    market_volatility = df['close'].pct_change().rolling(window=20).std()
    significant_movements = combined_indicator > (threshold * market_volatility)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_return = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = moving_avg_return * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Smooth the Adjusted Return
    intermediate_alpha = adjusted_return.rolling(window=5).mean()
    
    # Calculate Long-Term Volume-Weighted Momentum
    long_term_momentum = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Short-Term Volume-Weighted Momentum
    short_term_momentum = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Combine Momentum Indicators with Enhanced MACD
    fast_ema = intraday_momentum.ewm(span=12, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=26, adjust=False).mean()
    macd = fast_ema - slow_ema
    enhanced_macd = 0.6 * long_term_momentum + 0.4 * short_term_momentum + macd
    
    # Final Alpha Factor
    final_alpha = 0.5 * intermediate_alpha + 0.5 * enhanced_macd
    
    return final_alpha
```
