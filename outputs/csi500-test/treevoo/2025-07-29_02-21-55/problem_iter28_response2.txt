```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'].pct_change() + 1
    df['combined_intraday_volume'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Confirm with Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=20).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    df['combined_momentum'] = 0.7 * df['long_term_momentum'] + 0.3 * df['short_term_momentum']
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = df['intraday_pct_change'].ewm(span=12, adjust=False).mean()
    slow_ema = df['intraday_pct_change'].ewm(span=26, adjust=False).mean()
    macd = fast_ema - slow_ema
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum_sum'] = 0.4 * df['long_term_momentum'] + 0.6 * df['short_term_momentum'] + macd
    
    # Incorporate Volume Confirmation
    df['volume_ratio_confirm'] = np.where(df['volume'] > df['volume'].rolling(window=20).mean() * 1.5, 1,
                                          np.where(df['volume'] < df['volume'].rolling(window=20).mean() * 0.5, -1, 0))
    
    # Adjust Combined Momentum by Volume
    log_returns = np.log(df['close']).diff()
    adjusted_momentum = (fast_ema * log_returns.rolling(window=20).mean()) / df['volume'].ewm(span=20, adjust=False).mean() - slow_ema
    
    # Smooth the Adjusted Combined Momentum
    smoothed_adjusted_momentum = adjusted_momentum.rolling(window=5).mean()
    
    # Synthesize Final Indicator
    final_indicator = (0.5 * df['combined_intraday_volume'] + 
                       0.3 * df['combined_momentum'] + 
                       0.2 * smoothed_adjusted_momentum)
    
    return final_indicator.dropna()
```
