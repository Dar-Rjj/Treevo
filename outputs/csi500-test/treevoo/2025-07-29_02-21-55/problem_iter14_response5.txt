import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_to_low = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_high_to_low * volume_change_ratio
    
    # Adjust for Market Volatility
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    moving_std = daily_log_returns.rolling(window=20).std()
    threshold = 2 * moving_std
    significant_movements = combined_indicator.abs() > threshold
    
    # Confirm with Price Momentum
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Set a Volume Threshold
    volume_threshold = 1.5 * df['volume'].rolling(window=20).mean()
    volume_above_threshold = df['volume'] > volume_threshold
    
    # Synthesize Final Indicator
    final_indicator = (combined_indicator * significant_movements) + (momentum_score * volume_above_threshold)
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_returns = daily_log_returns.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = (moving_avg_log_returns * ema_volume) / df['volume'].rolling(window=20).mean()
    
    # Smooth the Adjusted Return
    smoothed_return = adjusted_return.rolling(window=5).mean()
    alpha_factor = np.sqrt(smoothed_return)
    
    return alpha_factor
