import pandas as pd

def heuristics_v2(df):
    # Compute Close Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Close Price Change (e.g., 20 days)
    moving_avg_window = 20
    df['price_momentum'] = df['close_change'].rolling(window=moving_avg_window).mean()
    
    # Identify Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 2.0
    df['volume_spike'] = (df['volume_change'] > df['volume'].shift(1) * volume_spike_threshold).astype(int)
    
    # Combine Price Momentum and Volume Spike
    high_factor = 2.0
    df['adjusted_price_momentum'] = df['price_momentum'] * (1 + (df['volume_spike'] * (high_factor - 1)))
    
    # Calculate Intraday Price Movement
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_direction'] = df['close'] - df['open']
    
    # Calculate Weighted Volume Activity
    avg_volume_window = 20
    df['avg_volume'] = df['volume'].rolling(window=avg_volume_window).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume']
    
    # Combine Intraday Price and Volume Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['relative_volume']
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']
    df['combined_metric'] = df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    # Adjust Combined Metric by Price Momentum and Volume Spike
    df['final_alpha'] = df['combined_metric'] * (1 + (df['volume_spike'] * (high_factor - 1)))
    
    return df['final_alpha'].dropna()
