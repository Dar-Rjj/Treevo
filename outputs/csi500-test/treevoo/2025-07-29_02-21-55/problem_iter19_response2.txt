```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    ema_std = daily_returns.ewm(span=20, adjust=False).std()
    threshold = 1.5 * ema_std
    significant_movements = (np.abs(combined_indicator) > threshold).astype(int)
    combined_indicator = combined_indicator * significant_movements
    
    # Confirm with Price Momentum
    lookback_period = 20
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    price_momentum = df['close'] / ema_close - 1
    
    # Combine Momentum and Volume Indicators
    volume_threshold = 1.5 * df['volume'].ewm(span=lookback_period, adjust=False).mean()
    volume_signal = (df['volume'] > volume_threshold).astype(int)
    weighted_sum = 0.6 * combined_indicator + 0.4 * price_momentum
    final_indicator = weighted_sum * volume_signal
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_indicator = 0.5 * final_indicator + 0.5 * opening_gap
    
    return final_indicator
```
