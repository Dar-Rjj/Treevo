```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['SMA'] = df['close'].rolling(window=14).mean()
    df['Momentum_SMA'] = df['close'] - df['SMA']
    df['EMA'] = df['close'].ewm(span=14, adjust=False).mean()
    df['Momentum_EMA'] = df['close'] - df['EMA']

    # Reversal Indicators
    df['Pct_Change_Close'] = df['close'].pct_change()
    df['Reversal_Signal'] = (df['close'] < df['low'].rolling(window=7).min()).astype(int)

    # Volume-Based Indicators
    df['Vol_Sum_10'] = df['volume'].rolling(window=10).sum()
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()  # Start from an arbitrary initial value
    df['Vol_Ratio_5'] = df['volume'] / df['volume'].rolling(window=5).mean()

    # Range-Based Indicators
    df['True_Range'] = df[['high', 'close']].apply(lambda x: max(x['high'], x.shift(1)['close']) - min(x['low'], x.shift(1)['close']), axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    df['Relative_Range'] = df['ATR_14'] / (df['high'] - df['low'])

    # Open, High, Low, Close Combinations
    df['Intraday_Performance'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['Freq_Close_Above_Open_PrevClose'] = ((df['close'] > df['open']) & (df['close'] > df['close'].shift(1))).astype(int)
    df['Trend_Strength_10'] = df['Intraday_Performance'].rolling(window=10).mean()

    # Amount Data for Indicators
    df['Amount_Sum_10'] = df['amount'].rolling(window=10).sum()
    df['Avg_Trade_Size'] = df['amount'] / df['volume']
    df['Amount_Ratio_7'] = df['amount'] / df['amount'].rolling(window=7).mean()

    # Combine all the indicators into a single alpha factor
    df['Alpha_Factor'] = (
        df['Momentum_SMA'] + df['Momentum_EMA'] + 
        df['Pct_Change_Close'] + df['Reversal_Signal'] + 
        df['Vol_Sum_10'] + df['OBV'] + df['Vol_Ratio_5'] + 
        df['ATR_14'] + df['Relative_Range'] + 
        df['Intraday_Performance'] + df['Freq_Close_Above_Open_PrevClose'] + df['Trend_Strength_10'] + 
        df['Amount_Sum_10'] + df['Avg_Trade_Size'] + df['Amount_Ratio_7']
    )

    return df['Alpha_Factor'].copy()
```
