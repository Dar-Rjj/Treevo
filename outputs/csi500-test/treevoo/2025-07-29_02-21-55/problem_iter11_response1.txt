```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 20-day Return
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_momentum'].ewm(span=12, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_momentum'].ewm(span=26, adjust=False).mean()
    
    # Calculate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = np.where(df['volume'] > df['avg_volume_20'], 1, 
                                  np.where(df['volume'] < df['avg_volume_20'], -1, 0))
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Long-Term Price Trend
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ma_log_returns_20'] = df['log_returns'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_momentum'] = ((df['fast_ema_intraday'] * df['ma_log_returns_20']) / df['ema_volume_20']) - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['sma_adjusted_momentum_5'] = df['adjusted_momentum'].rolling(window=5).mean()
    
    # Combine Momentum Indicators with MACD
    df['daily_range'] = df['high'] - df['low']
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    df['weighted_momentum_sum'] = 0.55 * df['long_term_momentum'] + 0.45 * df['short_term_momentum']
    
    # Calculate MACD
    df['macd_line'] = df['fast_ema_intraday'] - df['slow_ema_intraday']
    
    # Final Indicator
    df['final_indicator'] = df['weighted_momentum_sum'] + df['macd_line']
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volatility'] = df['log_returns'].rolling(window=21).std()
    df['final_alpha_factor'] = df['final_indicator'] * df['volume_change_ratio'] / df['volatility']
    
    return df['final_alpha_factor']
```
