def heuristics_v2(df):
    # Calculate the difference between today's closing price and yesterday's closing price
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Label as Bullish or Bearish based on the close_diff
    df['trend_label'] = 'Bullish'
    df.loc[df['close_diff'] < 0, 'trend_label'] = 'Bearish'
    
    # For bullish signals, calculate the ratio of close to open
    df['bullish_factor'] = df['close'] / df['open']
    
    # For bearish signals, calculate the ratio of open to close
    df['bearish_factor'] = df['open'] / df['close']
    
    # Use the appropriate factor based on the trend label
    df['trend_factor'] = df.apply(lambda row: row['bullish_factor'] if row['trend_label'] == 'Bullish' else row['bearish_factor'], axis=1)
    
    # Calculate the distance between today's high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Define a threshold for high volatility
    vol_threshold = df['high_low_diff'].mean() * 1.5
    
    # Identify high and low volatility days
    df['volatility_label'] = 'Low'
    df.loc[df['high_low_diff'] > vol_threshold, 'volatility_label'] = 'High'
    
    # For high volatility days, calculate the percentage change from the opening price to the closing price
    df['high_vol_factor'] = (df['close'] - df['open']) / df['open']
    
    # For low volatility days, look at the position of the closing price relative to the day's high and low
    df['low_vol_factor'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Use the appropriate factor based on the volatility label
    df['volatility_factor'] = df.apply(lambda row: row['high_vol_factor'] if row['volatility_label'] == 'High' else row['low_vol_factor'], axis=1)
    
    # Look at the volume traded
    df['volume_avg'] = df['volume'].rolling(window=5).mean()
    df['volume_label'] = 'Normal'
    df.loc[df['volume'] > df['volume_avg'] * 1.5, 'volume_label'] = 'High'
    
    # For high volume days, check if the closing price increased
    df['high_volume_factor'] = df.apply(lambda row: 1 if (row['volume_label'] == 'High' and row['close'] > row['open']) else (-1 if (row['volume_label'] == 'High' and row['close'] < row['open']) else 0), axis=1)
    
    # Consider the amount traded
    df['amount_avg'] = df['amount'].rolling(window=5).mean()
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # An above-average amount with rising prices might suggest a robust upturn
    df['amount_factor'] = df.apply(lambda row: 1 if (row['amount'] > row['amount_avg'] and row['close'] > row['open']) else (-1 if (row['amount'] > row['amount_avg'] and row['close'] < row['open']) else 0), axis=1)
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['trend_factor'] + df['volatility_factor'] + df['high_volume_factor'] + df['amount_factor']
    
    return df['alpha_factor']
