```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 3
    std_dev_20 = df['close'].pct_change().rolling(window=20).std()
    significant_movement = (combined_indicator > threshold * std_dev_20).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_20_log_return = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = ma_20_log_return * (ema_20_volume / df['volume'].rolling(window=20).mean())
    
    # Smooth the Adjusted Return
    smoothed_adjusted_return = adjusted_return.rolling(window=5).mean()
    
    # Output the Final Alpha Factor
    final_alpha_factor = significant_movement * smoothed_adjusted_return
    return final_alpha_factor
```
