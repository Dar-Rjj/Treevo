import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day simple moving average (SMA) of close prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate 5-day and 20-day exponential moving average (EMA) of close prices
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the difference between the high and low prices over a 5-day period
    df['HL_diff'] = df['high'] - df['low']
    df['avg_HL_diff_5'] = df['HL_diff'].rolling(window=5).mean()
    
    # Calculate the accumulation/distribution line (A/D Line)
    df['MFM'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['MFM_vol'] = df['MFM'] * df['volume']
    df['AD_line'] = df['MFM_vol'].cumsum()
    
    # Calculate the on-balance volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['OBV'] = obv
    
    # Calculate the relative strength index (RSI) of close prices
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the momentum of close prices
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the rate of change (ROC) of close prices
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Calculate the Bollinger Bands
    df['BB_middle'] = df['close'].rolling(window=20).mean()
    df['BB_std'] = df['close'].rolling(window=20).std()
    df['BB_upper'] = df['BB_middle'] + 2 * df['BB_std']
    df['BB_lower'] = df['BB_middle'] - 2 * df['BB_std']
    
    # Calculate the average true range (ATR)
    tr = pd.DataFrame(index=df.index)
    tr['h_l'] = df['high'] - df['low']
    tr['h_pc'] = abs(df['high'] - df['close'].shift(1))
    tr['l_pc'] = abs(df['low'] - df['close'].shift(1))
    tr['tr'] = tr[['h_l', 'h_pc', 'l_pc']].max(axis=1)
    df['ATR_14'] = tr['tr'].rolling(window=14).mean()
