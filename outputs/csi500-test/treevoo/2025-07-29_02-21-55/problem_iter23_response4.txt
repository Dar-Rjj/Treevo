```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) manually
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(delta > 0, 0)
    negative_flow = raw_money_flow.where(delta < 0, 0)
    
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine RSI and MFI into a single alpha factor
    alpha_factor = (rsi - mfi) / (rsi + mfi)
    
    return alpha_factor
```
