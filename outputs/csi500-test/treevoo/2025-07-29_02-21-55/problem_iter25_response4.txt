```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_change_ratio
    significant_movements = (combined_indicator > 0.05)  # Set a fixed threshold
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Combine Momentum and Volume Indicators
    average_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * average_volume
    volume_spike = (df['volume'] > volume_threshold)
    
    # Synthesize Final Indicator
    weight_momentum = 0.6
    weight_volume = 0.4
    final_indicator = (weight_momentum * momentum_score + weight_volume * combined_indicator) * significant_movements * volume_spike
    
    return final_indicator
```
