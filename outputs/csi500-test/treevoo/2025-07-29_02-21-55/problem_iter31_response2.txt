```python
import pandas as pd

def heuristics_v2(df, n=10, m=14, p=20):
    # Momentum: difference between the current close and the n-periods ago close
    df['momentum'] = df['close'] - df['close'].shift(n)
    
    # ATR: Average True Range over m periods
    tr = pd.DataFrame()
    tr['h-l'] = df['high'] - df['low']
    tr['h-cp'] = abs(df['high'] - df['close'].shift(1))
    tr['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['atr'] = tr.max(axis=1).rolling(window=m).mean()
    
    # Daily bullish or bearish move
    df['daily_move'] = df['close'] - df['open']
    
    # Volume change from previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # OBV-like indicator
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Significant volume changes
    df['avg_volume'] = df['volume'].rolling(window=p).mean()
    df['significant_volume'] = (df['volume'] - df['avg_volume']) / df['avg_volume']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['momentum'] + df['atr'] + df['daily_move'] + df['volume_change'] + df['obv'] + df['significant_volume']).fillna(0)
    
    return df['alpha_factor']
```
