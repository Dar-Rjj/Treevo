```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_to_low_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_high_to_low_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 0.05
    significant_movements = combined_indicator[combined_indicator > threshold]
    
    # Confirm with Price Momentum
    lookback_period = 30
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Synthesize Final Indicator
    weight_intraday_volume = 0.7
    weight_momentum = 0.3
    final_indicator = (significant_movements * weight_intraday_volume) + (momentum_score * weight_momentum)
    
    return final_indicator
```
