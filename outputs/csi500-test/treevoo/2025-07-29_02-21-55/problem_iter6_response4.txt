```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = intraday_momentum.ewm(span=5, adjust=False).mean()  # Fast EMA with a shorter period
    slow_ema = intraday_momentum.ewm(span=20, adjust=False).mean()  # Slow EMA with a longer period
    
    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    ma_log_returns = log_returns.rolling(window=20).sum() / 20
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Standard Deviation of Daily Log Returns
    std_log_returns = log_returns.rolling(window=20).std()
    
    # Adjust Intraday Momentum by Volume, Long-Term Price Trend, and Volatility
    adjusted_intraday_momentum = (fast_ema * ma_log_returns) / ema_volume - slow_ema
    adjusted_intraday_momentum = adjusted_intraday_momentum * std_log_returns
    
    # Smooth the Adjusted Intraday Momentum
    final_alpha_factor = adjusted_intraday_momentum.rolling(window=5).mean()
    
    return final_alpha_factor
```
