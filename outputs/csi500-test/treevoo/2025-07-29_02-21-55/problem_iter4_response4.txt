```python
def heuristics_v2(df):
    # Calculate 20-day Price Return
    df['20_day_price_return'] = df['close'].pct_change(periods=20)
    
    # Determine Volume Ratio
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean().shift(1)
    df['volume_ratio'] = df['volume'] / df['avg_volume_20_days']
    
    # Set Volume Ratio to 1, -1, or 0 based on the conditions
    df['volume_signal'] = 0
    df.loc[df['volume_ratio'] > 1, 'volume_signal'] = 1
    df.loc[df['volume_ratio'] < 1, 'volume_signal'] = -1
    
    # Generate the factor by combining price return and volume signal
    df['alpha_factor'] = df['20_day_price_return'] * df['volume_signal']
    
    return df['alpha_factor']
```
