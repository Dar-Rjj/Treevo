```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average price
    wap = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate the relative spread: (high - low) / close
    rel_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate the price change: close - open
    price_change = df['close'] - df['open']
    
    # Calculate the volume change: current volume - previous volume
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate the factor: (wap * rel_spread * price_change) / volume_change
    factor = (wap * rel_spread * price_change) / (volume_change + 1e-6)  # Adding a small constant to avoid division by zero
    
    return factor
```
