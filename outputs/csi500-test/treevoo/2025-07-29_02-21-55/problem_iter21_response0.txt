```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor using a combination of price and volume data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A Series representing the factor values, indexed by (date).
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the price movement from open to close
    price_move = df['close'] - df['open']
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the distance between the closing price and the VWAP
    dist_to_vwap = df['close'] - vwap
    
    # Combine the factors into a single alpha factor
    alpha_factor = (daily_return + price_move + dist_to_vwap) / 3.0
    
    return alpha_factor
```
