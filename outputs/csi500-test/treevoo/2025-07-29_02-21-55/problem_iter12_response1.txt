```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    # Calculate 20-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    ema_std_20 = daily_returns.ewm(span=20, adjust=False).std()
    dynamic_threshold = 3 * ema_std_20
    significant_movements = combined_indicator.abs() > dynamic_threshold
    
    # Confirm with Price Momentum
    lookback_period = 20
    ema_close_20 = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close_20
    
    # Combine Momentum and Volume Indicators
    volume_ema_20 = df['volume'].ewm(span=lookback_period, adjust=False).mean()
    volume_threshold = 1.5 * volume_ema_20
    high_volume = df['volume'] > volume_threshold
    
    # Synthesize Final Indicator
    final_indicator = (combined_indicator * significant_movements * 0.5) + (momentum_score * high_volume * 0.5)
    
    return final_indicator
```
