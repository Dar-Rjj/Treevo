```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_change_ratio']
    
    # Apply Threshold to Identify Significant Movements
    fixed_threshold = 1.5
    df['daily_return'] = df['close'].pct_change()
    df['moving_std'] = df['daily_return'].rolling(window=20).std()
    df['threshold'] = fixed_threshold * df['moving_std']
    df['significant_movement'] = (df['combined_indicator'] > df['threshold']).astype(int)
    
    # Apply Moving Averages to Intraday High-to-Low Momentum
    fast_ema_period = 10
    slow_ema_period = 20
    df['fast_ema'] = df['intraday_hl_pct_change'].ewm(span=fast_ema_period, adjust=False).mean()
    df['slow_ema'] = df['intraday_hl_pct_change'].ewm(span=slow_ema_period, adjust=False).mean()
    
    # Calculate MACD
    df['macd'] = df['fast_ema'] - df['slow_ema']
    
    # Final alpha factor
    df['alpha_factor'] = df['macd'] * df['significant_movement']
    
    return df['alpha_factor']
```
