```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_intraday_volume'] = (df['high'] - df['low']) / df['low'] * df['volume_change_ratio']
    
    # Confirm with Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=20).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    df['combined_momentum'] = 0.6 * df['long_term_momentum'] + 0.4 * df['short_term_momentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema'] = df['daily_range'].ewm(span=10, adjust=False).mean()
    df['slow_ema'] = df['daily_range'].ewm(span=30, adjust=False).mean()
    
    # Incorporate Volume Confirmation
    df['volume_confirmation'] = df['volume_change_ratio'].apply(lambda x: 0 if x < 1.8 and x > 0.3 else x)
    
    # Adjust Combined Momentum by Volume
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['adjusted_combined_momentum'] = (df['fast_ema'] * df['log_returns'].rolling(window=20).mean()) / df['volume'].ewm(span=20, adjust=False).mean() - df['slow_ema']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adjusted_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Synthesize Final Indicator
    df['final_indicator'] = 0.4 * df['combined_intraday_volume'] + 0.6 * df['combined_momentum']
    
    return df['final_indicator'].dropna()
```
