```python
import pandas as pd

def heuristics_v2(df, M=10, P=14, Q=20):
    # Momentum Indicators
    df['SMA'] = df['close'].rolling(window=M).mean()
    df['Momentum_Indicator'] = df['close'] - df['SMA']
    
    # Reversal Indicators
    df['Reversal_Indicator'] = df['close'].pct_change()
    
    # Volume-Based Indicators
    df['Volume_Sum'] = df['volume'].rolling(window=M).sum()
    df['OBV'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] + (df['close'] < df['close'].shift(1)).astype(int) * -df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Range-Based Indicators
    df['True_Range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=P).mean()
    
    # Open, High, Low, Close Combinations
    df['Intraday_Performance'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['Bullish_Days'] = ((df['close'] > df['open']) & (df['close'] > df['close'].shift(1))).astype(int)
    df['Bullish_Days_Count'] = df['Bullish_Days'].rolling(window=M).sum()
    
    # Amount Data for Indicators
    df['Amount_Sum'] = df['amount'].rolling(window=Q).sum()
    df['Trade_Size_Ratio'] = df['amount'] / df['volume']
    
    # Alpha Factor
    alpha_factor = (
        df['Momentum_Indicator'] +
        df['Reversal_Indicator'] +
        (df['Volume_Sum'] / df['volume']) +
        df['OBV'] +
        df['ATR'] +
        df['Intraday_Performance'] +
        df['Bullish_Days_Count'] +
        df['Amount_Sum'] +
        df['Trade_Size_Ratio']
    )
    
    return alpha_factor
```
