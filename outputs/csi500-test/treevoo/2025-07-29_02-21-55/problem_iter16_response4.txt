```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_pct_change * volume_change_ratio
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_returns = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_ma_log_returns = (ma_log_returns * ema_volume) / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_ma_log_returns.ewm(span=5, adjust=False).mean()
    slow_ema = adjusted_ma_log_returns.ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = fast_ema - slow_ema + combined_indicator
    
    return alpha_factor
```
