```python
import pandas as pd

def heuristics_v2(df):
    # Determine the lookback period
    lookback_period = 20
    
    # Compute the Simple Moving Average (SMA) of Close prices
    df['SMA'] = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate the Momentum Score
    df['Momentum_Score'] = df['close'] - df['SMA']
    
    # Set a Volume Threshold (1.5 * Average Volume over the lookback period)
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * avg_volume
    
    # Assign a Volume Factor
    df['Volume_Factor'] = (df['volume'] > volume_threshold).astype(int)
    
    # Set an Amount Threshold (1.5 * Average Amount over the lookback period)
    avg_amount = df['amount'].rolling(window=lookback_period).mean()
    amount_threshold = 1.5 * avg_amount
    
    # Assign an Amount Factor
    df['Amount_Factor'] = (df['amount'] > amount_threshold).astype(int)
    
    # Combine Momentum, Volume, and Amount Factors
    df['Factor'] = df['Momentum_Score'] * df['Volume_Factor'] * df['Amount_Factor']
    
    return df['Factor']
```
