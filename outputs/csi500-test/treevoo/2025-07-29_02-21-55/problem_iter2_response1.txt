```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = intraday_momentum.ewm(span=5, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=20, adjust=False).mean()

    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(df['close']).diff().rolling(window=20).sum() / 20
    
    # Calculate 20-day Exponential Moving Average of Volume
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Intraday Momentum by Volume and Long-Term Price Trend
    adjusted_momentum = (fast_ema * log_returns) / volume_ema - slow_ema
    
    # Smooth the Adjusted Intraday Momentum
    final_alpha = adjusted_momentum.rolling(window=5).mean()
    
    return final_alpha
```
