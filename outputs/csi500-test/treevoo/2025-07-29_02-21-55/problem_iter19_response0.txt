```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pc'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike: Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_hl_pc'] * df['volume_change_ratio']
    
    # Apply Threshold to Identify Significant Movements
    df['daily_return'] = df['close'].pct_change()
    df['std_dev_20'] = df['daily_return'].ewm(span=20).std()
    threshold = df['std_dev_20'].iloc[-1]  # Dynamic threshold using the latest 20-day EMA std deviation
    
    df['combined_indicator_sig'] = df['combined_indicator'].apply(lambda x: x if abs(x) > threshold else 0)
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['ema_close_20'] = df['close'].ewm(span=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['ema_close_20']
    
    # Adjust by Intraday Volume Change
    df['volume_adj_momentum'] = df['momentum_score'] * df['volume_change_ratio']
    
    # Apply Moving Averages to Volume-Adjusted Intraday Momentum
    fast_ema_span = 5
    slow_ema_span = 10
    df['fast_ema_vol_adj_mom'] = df['volume_adj_momentum'].ewm(span=fast_ema_span).mean()
    df['slow_ema_vol_adj_mom'] = df['volume_adj_momentum'].ewm(span=slow_ema_span).mean()
    
    # Combine Momentum, Volume, and Long-Term Trend
    df['log_daily_return'] = np.log(df['close'] / df['close'].shift(1))
    df['avg_log_ret_20'] = df['log_daily_return'].rolling(window=lookback_period).mean()
    df['amount_ema_20'] = df['amount'].ewm(span=lookback_period).mean()
    df['combined_factor'] = (df['fast_ema_vol_adj_mom'] * df['avg_log_ret_20'] / df['amount_ema_20'] -
                             df['slow_ema_vol_adj_mom'] + df['close'].rolling(window=5).mean())
    
    # Smooth the Adjusted Intraday Momentum
    df['final_alpha_factor'] = df['combined_factor'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
