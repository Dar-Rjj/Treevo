def heuristics_v2(df):
    # Calculate Intraday Return Components
    open_to_high = df['high'] - df['open']
    open_to_low = df['open'] - df['low']
    close_to_open = df['close'] - df['open']
    
    # Combine Intraday Returns
    # Compute Average Intraday Return
    average_intraday_return = (open_to_high + open_to_low + close_to_open) / 3
    
    # Compute Intraday Return Difference
    max_intraday_return = df[['open_to_high', 'open_to_low', 'close_to_open']].max(axis=1)
    min_intraday_return = df[['open_to_high', 'open_to_low', 'close_to_open']].min(axis=1)
    intraday_return_difference = max_intraday_return - min_intraday_return
    
    # Volume Weighted Adjustments
    # Calculate Volume Weighted Intraday Average
    volume_weighted_intraday_average = df['volume'] * average_intraday_return
    
    # Generate Final Alpha Factor
    final_alpha_factor = volume_weighted_intraday_average - intraday_return_difference
    
    return final_alpha_factor
