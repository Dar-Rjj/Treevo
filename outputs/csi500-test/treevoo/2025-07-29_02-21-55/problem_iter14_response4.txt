```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Adjust by Intraday Volume Change
    volume_change = df['volume'].pct_change()
    volume_adjusted_intraday_momentum = intraday_momentum * (1 + volume_change)
    
    # Apply Exponential Moving Averages to Volume-Adjusted Intraday Momentum
    fast_ema = volume_adjusted_intraday_momentum.ewm(span=12, adjust=False).mean()
    slow_ema = volume_adjusted_intraday_momentum.ewm(span=26, adjust=False).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    ma_20_log_returns = daily_log_returns.rolling(window=20).mean()
    
    # Adjust Fast and Slow Exponential Moving Averages by Long-Term Price Trend
    adjusted_fast_ema = fast_ema * ma_20_log_returns
    adjusted_slow_ema = slow_ema * ma_20_log_returns
    adjusted_intraday_momentum = adjusted_fast_ema - adjusted_slow_ema
    
    # Smooth the Adjusted Intraday Momentum
    smoothed_intraday_momentum = adjusted_intraday_momentum.rolling(window=5).mean()
    
    # Combine with Long-Term and Short-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / (df['high'] - df['low']).rolling(window=21).mean()
    short_term_momentum = (df['close'] - df['close'].shift(6)) / (df['high'] - df['low']).rolling(window=7).mean()
    combined_momentum = 0.55 * long_term_momentum + 0.45 * short_term_momentum
    
    # Weight Intraday Direction by Relative Volume
    intraday_direction = df['close'] - df['open']
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    weighted_intraday_direction = intraday_direction * relative_volume
    
    # Apply Moving Averages to Intraday High-to-Low Percentage Change
    fast_ema_intraday_pct = (df['high'] / df['low'] - 1).ewm(span=12, adjust=False).mean()
    slow_ema_intraday_pct = (df['high'] / df['low'] - 1).ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    macd = fast_ema - slow_ema
    combined_momentum_macd = 0.45 * long_term_momentum + 0.55 * short_term_momentum + macd
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = (df['high'] / df['low'] - 1) * volume_change_ratio
    threshold = 2 * df['close'].pct_change().rolling(window=21).std()
    significant_movements = combined_indicator.apply(lambda x: x if abs(x) > threshold else 0)
    
    final_factor = combined_momentum_macd + smoothed_intraday_momentum + significant_movements
    
    return final_factor
```
