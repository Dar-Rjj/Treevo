```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_high_low_pct'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_high_low_pct'] * df['volume_ratio']
    
    # Calculate 20-Day Moving Standard Deviation of Daily Returns
    df['daily_return'] = df['close'].pct_change()
    df['rolling_std'] = df['daily_return'].rolling(window=20).std()
    
    # Apply Threshold to Identify Significant Movements
    threshold = 2.0  # Fixed threshold
    df['significant_movement'] = (df['combined_indicator'] > threshold * df['rolling_std'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_indicator'] * df['significant_movement'].astype(int)
    
    return df['alpha_factor']
```
