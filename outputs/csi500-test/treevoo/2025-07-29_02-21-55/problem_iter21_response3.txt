```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    ema_std_20 = daily_returns.ewm(span=20).std()
    dynamic_threshold = 2 * ema_std_20
    significant_movements = combined_indicator[combined_indicator > dynamic_threshold]
    
    # Confirm with Price Momentum
    ema_close_20 = df['close'].ewm(span=20).mean()
    momentum_score = df['close'] - ema_close_20
    adjusted_momentum = momentum_score * (df['volume'] / df['volume'].shift(1))
    
    # Apply Moving Averages to Volume-Adjusted Intraday Momentum
    fast_ema_momentum = adjusted_momentum.ewm(span=5).mean()
    slow_ema_momentum = adjusted_momentum.ewm(span=20).mean()
    
    # Combine Momentum, Volume, and Long-Term Trend
    log_returns = np.log(df['close'] / df['close'].shift(1))
    avg_log_returns_20 = log_returns.rolling(window=20).mean()
    avg_amount_20 = df['amount'].ewm(span=20).mean()
    combined_factor = (fast_ema_momentum * avg_log_returns_20) / avg_amount_20 - slow_ema_momentum + df['close'].rolling(window=5).mean()
    
    # Smooth the Adjusted Intraday Momentum
    smoothed_factor = combined_factor.rolling(window=5).mean()
    
    # Introduce Trading Range Factor
    trading_range = (df['high'] - df['low']) / df['open']
    ema_trading_range_5 = trading_range.ewm(span=5).mean()
    
    # Incorporate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust Combined Indicator
    weighting_factor = 0.8
    final_alpha = (smoothed_factor * rsi * weighting_factor).fillna(0)
    
    return final_alpha
```
