```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_change_ratio
    
    # Set a Dynamic Threshold Based on Historical Volatility
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    ema_volatility = daily_returns.ewm(span=20).std()
    threshold = 2 * ema_volatility
    significant_movement = (combined_indicator > threshold) | (combined_indicator < -threshold)
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_returns = daily_returns.rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_ma_return = ma_log_returns * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_ma_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_ma_return.ewm(span=30, adjust=False).mean()
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    volume_threshold = 1.5 * df['volume'].rolling(window=lookback_period).mean()
    momentum_volume_combined = (momentum_score * (df['volume'] > volume_threshold)) + (slow_ema - fast_ema)
    
    # Integrate Opening and Closing Price Gap
    price_gap = df['open'] - df['close'].shift(1)
    gap_direction_multiplier = np.where(price_gap > 0, 1, -1)
    
    # Combine All Indicators
    final_indicator = (
        0.4 * significant_movement.astype(int) * combined_indicator +
        0.3 * momentum_volume_combined +
        0.3 * (price_gap * gap_direction_multiplier)
    )
    
    return final_indicator
```
