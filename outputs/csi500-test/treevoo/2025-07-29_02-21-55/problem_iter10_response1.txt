```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['lt_momentum'] = (df['close'] - df['close'].shift(20)) / df['range'].rolling(window=22).mean()
    
    # Calculate Short-Term Momentum
    df['st_momentum'] = (df['close'] - df['close'].shift(5)) / df['range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.6 * df['lt_momentum'] + 0.4 * df['st_momentum']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_change'] = (df['high'] - df['low']) / df['low']
    
    # Apply Moving Averages
    fast_ema = df['intraday_hl_change'].ewm(span=12, adjust=False).mean()
    slow_ema = df['intraday_hl_change'].ewm(span=26, adjust=False).mean()
    macd = fast_ema - slow_ema
    
    # Combine Momentum Indicators with MACD
    df['momentum_macd'] = df['combined_momentum'] + macd
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    THRESHOLD = 1.5  # Set a fixed threshold to identify significant movements
    df['alpha_factor'] = df['momentum_macd'] * np.where(df['volume_ratio'] > THRESHOLD, df['volume_ratio'], 1)
    
    return df['alpha_factor']
```
