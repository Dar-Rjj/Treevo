```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema = intraday_momentum.ewm(span=5, adjust=False).mean()
    slow_ema = intraday_momentum.ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    ma_log_returns = log_returns.rolling(window=20).mean()
    
    # Calculate Long-Term and Short-Term Momentum
    daily_range = df['high'] - df['low']
    long_term_momentum = (df['close'] - df['close'].shift(20)) / daily_range.rolling(window=25).mean()
    short_term_momentum = (df['close'] - df['close'].shift(5)) / daily_range.rolling(window=10).mean()
    
    # Combine Momentum Indicators
    combined_momentum = (long_term_momentum * 0.7) + (short_term_momentum * 0.3)
    
    # Adjust Intraday Momentum by Combined Momentum and Volume
    adjusted_intraday_momentum = (fast_ema * ma_log_returns) / df['amount'].ewm(span=20, adjust=False).mean() - slow_ema
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    daily_returns = df['close'].pct_change()
    threshold = 2.0 * daily_returns.rolling(window=25).std()
    significant_volume_spike = (volume_change_ratio > threshold).astype(int)
    
    # Generate Final Alpha Factor
    weighted_intraday_range = intraday_momentum * significant_volume_spike
    weighted_intraday_direction = (intraday_momentum > 0).astype(int) * significant_volume_spike
    final_alpha_factor = (weighted_intraday_range + weighted_intraday_direction) + (combined_momentum * 0.7)
    
    return final_alpha_factor
```
