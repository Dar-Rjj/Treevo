```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_hl_pct'] * df['volume_ratio']
    
    # Adjust for Market Volatility
    df['returns'] = df['close'].pct_change()
    df['volatility'] = df['returns'].rolling(window=20).std()
    threshold = 1.5 * df['volatility']
    df['significant_movement'] = df['combined_indicator'].apply(lambda x: x if abs(x) > threshold else 0)
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['sma_close'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['sma_close']
    
    # Combine Momentum and Volume Indicators
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * avg_volume
    df['volume_confirm'] = df['volume'].apply(lambda x: 1 if x > volume_threshold else -1 if x < volume_threshold else 0)
    
    df['final_indicator'] = 0.6 * df['momentum_score'] + 0.4 * df['significant_movement'] * df['volume_confirm']
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['return_20d'] = df['close'] / df['close'].shift(20) - 1
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema'] = df['intraday_momentum'].ewm(span=12, adjust=False).mean()
    df['slow_ema'] = df['intraday_momentum'].ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentums'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd'] = df['fast_ema'] - df['slow_ema']
    df['combined_momentum_macd'] = df['weighted_momentums'] + df['macd']
    
    # Incorporate Volume Confirmation
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20d']
    df['volume_confirmation'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else -1 if x < 1 else 0)
    
    # Adjust Combined Momentum by Volume
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ma_log_returns'] = df['log_returns'].rolling(window=20).mean()
    df['ema_volume_20d'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_combined_momentum'] = (df['fast_ema'] * df['ma_log_returns']) / df['ema_volume_20d'] - df['slow_ema']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_factor'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    return df['smoothed_factor'].dropna()
```
