```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 2.0  # Fixed threshold
    moving_std = df['close'].pct_change().rolling(window=30).std()
    combined_indicator = combined_indicator * (combined_indicator > threshold * moving_std)
    
    # Confirm with Price Momentum
    lookback_period = 30
    sma = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma
    
    # Synthesize Final Indicator
    weight_intraday_volume = 0.7
    weight_momentum = 0.3
    final_indicator = (weight_intraday_volume * combined_indicator) + (weight_momentum * momentum_score)
    
    return final_indicator
```
