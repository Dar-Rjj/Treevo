```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    volatility = df['close'].pct_change().rolling(window=20).std()
    threshold = 2 * volatility
    df['significant_movement'] = df['combined_indicator'].where(df['combined_indicator'].abs() > threshold, 0)
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return_20'] = df['log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    df['adjusted_ma_return'] = df['ma_log_return_20'] * df['ema_volume_20'] / df['volume'].rolling(window=20).mean()
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['sma_close_20'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['sma_close_20']
    
    # Combine Momentum and Volume Indicators
    volume_threshold = 1.5 * df['volume'].rolling(window=lookback_period).mean()
    df['final_indicator'] = (df['significant_movement'] + df['momentum_score']).where(df['volume'] > volume_threshold, 0)
    
    # Apply Weighted Moving Averages
    fast_ema = df['adjusted_ma_return'].ewm(span=10, adjust=False).mean()
    slow_ema = df['adjusted_ma_return'].ewm(span=30, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (fast_ema - slow_ema) + df['final_indicator']
    
    return df['alpha_factor']
```
