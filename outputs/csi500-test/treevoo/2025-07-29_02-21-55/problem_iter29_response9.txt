```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Intraday and Volume Indicator
    df['combined_indicator'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    threshold = 3.0
    df['volatility'] = df['close'].rolling(window=30).std()
    df['combined_indicator'] = df['combined_indicator'] * (df['combined_indicator'].abs() > (threshold * df['volatility']))
    
    # Confirm with Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=20).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    df['momentum_combined'] = 0.65 * df['long_term_momentum'] + 0.35 * df['short_term_momentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema'] = df['intraday_pct_change'].ewm(span=12, adjust=False).mean()
    df['slow_ema'] = df['intraday_pct_change'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['macd_diff'] = df['fast_ema'] - df['slow_ema']
    df['weighted_momentum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd_combined'] = df['weighted_momentum'] + df['macd_diff']
    
    # Incorporate Volume Confirmation
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_adj'] = (df['volume'] / df['avg_volume_20_days']).apply(lambda x: 1.5 if x >= 1.5 else (-0.5 if x < 0.5 else 0))
    
    # Adjust Combined Momentum by Volume
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ma_log_returns_20'] = df['log_returns'].rolling(window=20).mean()
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_momentum'] = (df['fast_ema'] * df['ma_log_returns_20']) / df['ema_volume_20'] - df['slow_ema']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adjusted_momentum'] = df['adjusted_momentum'].rolling(window=5).mean()
    
    # Crossover Integration
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_close_roll_avg'] = df['open_close_return'].rolling(window=20).mean()
    df['trend_filter'] = df['close'] - df['close'].rolling(window=50).mean()
    
    # Combine with Volume Spikes
    df['volume_spike_threshold'] = (df['volume_ratio'] > 1.5) | (df['volume_ratio'] < 0.5)
    df['integrated_indicator'] = 0.5 * df['open_close_return'] + 0.5 * df['macd_combined']
    df['integrated_indicator'] = df['integrated_indicator'] * (df['trend_filter'] > 0)
    
    # Final Adjustment
    df['final_indicator'] = df['smoothed_adjusted_momentum'].rolling(window=5).mean()
    df['final_indicator'] = 0.7 * df['final_indicator'] + 0.3 * df['integrated_indicator']
    
    return df['final_indicator'].dropna()
```
