```python
def heuristics_v2(df):
    # Calculate 20-day return
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Calculate volume confirmation
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean().shift(1)
    df['vol_ratio'] = df['volume'] / df['avg_vol_20']
    conditions_vol = [
        (df['vol_ratio'] > 1),
        (df['vol_ratio'] < 1)
    ]
    choices_vol = [1, -1]
    df['vol_confirmation'] = np.select(conditions_vol, choices_vol, default=0)
    
    # Calculate price oscillation
    max_high_20 = df['high'].rolling(window=20).max()
    min_low_20 = df['low'].rolling(window=20).min()
    max_range_20 = max_high_20 - min_low_20
    
    actual_range_20 = df['high'] - df['low'].shift(19)
    
    df['oscillation_score'] = np.where(actual_range_20 > 0.5 * max_range_20, 1,
                                       np.where(actual_range_20 < 0.2 * max_range_20, -1, 0))
    
    # Combine all factors
    df['alpha_factor'] = df['20_day_return'] * df['vol_confirmation'] * df['oscillation_score']
    
    return df['alpha_factor']
```
