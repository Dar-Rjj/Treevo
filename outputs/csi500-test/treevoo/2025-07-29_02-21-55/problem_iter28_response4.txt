```python
import pandas as pd

def heuristics_v2(df, n=20, m=14, p=10):
    # Momentum: Difference between the current close and n-periods ago close
    momentum = df['close'] - df['close'].shift(n)
    
    # Average True Range (ATR) over m periods
    tr = df[['high', 'low', 'close']].join(df[['high', 'low', 'close']].shift(1))
    tr.columns = ['high', 'low', 'close', 'high_1', 'low_1', 'close_1']
    tr['h_l'] = tr['high'] - tr['low']
    tr['h_pc'] = abs(tr['high'] - tr['close_1'])
    tr['l_pc'] = abs(tr['low'] - tr['close_1'])
    tr['tr'] = tr[['h_l', 'h_pc', 'l_pc']].max(axis=1)
    atr = tr['tr'].rolling(window=m).mean()
    
    # Daily bullish or bearish move
    daily_move = df['close'] - df['open']
    
    # Volume change from the previous day
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # On-Balance Volume (OBV) type indicator
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Significant volume changes compared to a moving average of the past p days' volumes
    vol_ma = df['volume'].rolling(window=p).mean()
    significant_vol_change = (df['volume'] - vol_ma) / vol_ma
    
    # Combine all factors into a single DataFrame
    factor_df = pd.DataFrame({
        'momentum': momentum,
        'atr': atr,
        'daily_move': daily_move,
        'volume_change': volume_change,
        'obv': obv,
        'significant_vol_change': significant_vol_change
    })
    
    # Sum up all factors (heuristic combination)
    alpha_factor = factor_df.sum(axis=1)
    
    return alpha_factor
```
