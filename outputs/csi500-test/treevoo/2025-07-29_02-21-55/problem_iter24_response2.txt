def heuristics_v2(df):
    # Momentum Indicators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()

    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Volume-Based Indicators
    df['OBV'] = ((df['close'] > df['close'].shift(1)) * df['volume'] - (df['close'] < df['close'].shift(1)) * df['volume']).cumsum()
    
    df['PVT'] = (df['close'].pct_change() * df['volume']).cumsum()
    
    # Volatility Indicators
    df['TR'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    df['Bollinger_Middle'] = df['close'].rolling(window=20).mean()
    df['Bollinger_STD'] = df['close'].rolling(window=20).std()
    df['Bollinger_Upper'] = df['Bollinger_Middle'] + 2 * df['Bollinger_STD']
    df['Bollinger_Lower'] = df['Bollinger_Middle'] - 2 * df['Bollinger_STD']
    
    # Market Strength Indicators
    money_flow = (df['close'] - df['low']) - (df['high'] - df['close']) * df['volume']
    positive_money_flow = money_flow.where(money_flow > 0, 0)
    negative_money_flow = -money_flow.where(money_flow < 0, 0)
    mfr = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI_14'] = 100 - 100 / (1 + mfr)
