```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['Intraday_HL_Change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Combined_Indicator'] = df['Intraday_HL_Change'] * df['Volume_Ratio']
    
    # Set a Dynamic Threshold Based on Historical Volatility
    df['Daily_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['20D_EMSD'] = df['Daily_Return'].ewm(span=20, adjust=False).std()
    threshold = df['20D_EMSD'].mean()  # Example threshold, can be adjusted
    df['Significant_Movement'] = (df['Combined_Indicator'] > threshold).astype(int)
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['20D_MA_Log_Returns'] = df['Daily_Return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['20D_EMA_Volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['Adjusted_MA_Return'] = df['20D_MA_Log_Returns'] * (df['20D_EMA_Volume'] / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    df['Fast_EMA'] = df['Adjusted_MA_Return'].ewm(span=10, adjust=False).mean()
    df['Slow_EMA'] = df['Adjusted_MA_Return'].ewm(span=30, adjust=False).mean()
    
    # Confirm with Price Momentum
    df['Price_Momentum_Score'] = df['close'] - df['close'].rolling(window=20).mean()
    
    # Introduce Volatility-Adjusted Momentum
    df['20D_EMA_Abs_Log_Returns'] = np.abs(df['Daily_Return']).ewm(span=20, adjust=False).mean()
    df['Vol_Adjusted_Momentum'] = df['Price_Momentum_Score'] / df['20D_EMA_Abs_Log_Returns']
    
    # Synthesize Final Indicator
    final_indicator = (
        0.3 * df['Significant_Movement'] + 
        0.4 * df['Price_Momentum_Score'] + 
        0.3 * df['Vol_Adjusted_Momentum']
    )
    
    return final_indicator.dropna()
```
