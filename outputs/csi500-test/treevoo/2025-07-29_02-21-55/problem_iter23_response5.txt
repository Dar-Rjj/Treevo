```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_pct_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 1.5
    moving_std = df['close'].rolling(window=30).std()
    adjusted_threshold = threshold * moving_std
    significant_movements = combined_indicator > adjusted_threshold
    
    # Confirm with Price Momentum
    lookback_period = 30
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Synthesize Final Indicator
    weight_intraday_volume = 0.7
    weight_momentum = 0.3
    final_indicator = (weight_intraday_volume * significant_movements * combined_indicator) + \
                      (weight_momentum * momentum_score)
    
    return final_indicator
```
