def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['high_low_pct_change'] * df['volume_ratio']
    
    # Calculate Daily Log Return
    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x) if x > 0 else 0)
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_ma_return'] = (df['ma_log_return'] * df['ema_volume']) / df['volume'].rolling(window=20).mean()
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_momentum'].ewm(span=12, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_momentum'].ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_long_term_momentum'] = df['long_term_momentum'] * 0.45
    df['weighted_short_term_momentum'] = df['short_term_momentum'] * 0.55
    df['macd'] = df['fast_ema_intraday'] - df['slow_ema_intraday']
    df['combined_momentum'] = df['weighted_long_term_momentum'] + df['weighted_short_term_momentum'] + df['macd']
    
    # Incorporate Volume Confirmation
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    df['volume_confirmation'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else (-1 if x < 1 else 0))
    
    # Adjust Combined Momentum by Volume
    df['adjusted_combined_momentum'] = ((df['fast_ema_intraday'] * df['ma_log_return']) / df['ema_volume']) - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    return df['smoothed_momentum'].dropna()
