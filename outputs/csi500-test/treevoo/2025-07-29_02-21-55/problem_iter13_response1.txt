```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_change_ratio
    significant_movements = np.where(combined_indicator > 5, combined_indicator, 0)  # Set a fixed threshold of 5
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_return = daily_log_return.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    exp_moving_avg_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_return = (moving_avg_log_return * exp_moving_avg_volume) / df['volume'].rolling(window=20).mean()
    
    # Smooth the Adjusted Return
    smoothed_return = adjusted_return.rolling(window=5).mean()
    final_alpha_factor = np.sqrt(smoothed_return)
    
    return final_alpha_factor
```
