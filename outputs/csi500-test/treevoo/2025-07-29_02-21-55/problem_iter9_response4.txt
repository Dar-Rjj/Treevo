```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']

    # Calculate Long-Term Momentum
    df['Long_Term_Momentum'] = (df['close'] - df['close'].shift(21)) / df['Daily_Range'].rolling(window=21).mean()

    # Calculate Short-Term Momentum
    df['Short_Term_Momentum'] = (df['close'] - df['close'].shift(6)) / df['Daily_Range'].rolling(window=7).mean()

    # Combine Momentum Indicators
    df['Combined_Momentum'] = 0.55 * df['Long_Term_Momentum'] + 0.45 * df['Short_Term_Momentum']

    # Calculate Intraday High-to-Low Percentage Change
    df['Intraday_HL_Pct_Change'] = (df['high'] - df['low']) / df['low']

    # Apply Moving Averages
    fast_ema = df['Intraday_HL_Pct_Change'].ewm(span=12, adjust=False).mean()
    slow_ema = df['Intraday_HL_Pct_Change'].ewm(span=26, adjust=False).mean()

    # MACD Difference
    df['MACD'] = fast_ema - slow_ema

    # Combine Momentum Indicators with MACD
    df['Momentum_MACD'] = 0.45 * df['Long_Term_Momentum'] + 0.55 * df['Short_Term_Momentum'] + df['MACD']

    # Incorporate Volume Spike
    df['Volume_Change_Ratio'] = df['volume'] / df['volume'].shift(1)

    # Compute Combined Indicator
    df['Combined_Indicator'] = df['Intraday_HL_Pct_Change'] * df['Volume_Change_Ratio']

    # Adjust for Market Volatility
    df['21_Day_Volatility'] = df['close'].pct_change().rolling(window=21).std()
    df['Significant_Movement'] = np.where(df['Combined_Indicator'] > 2 * df['21_Day_Volatility'], 1, 0)

    # Final Factor
    df['Factor'] = df['Momentum_MACD'] * df['Significant_Movement']

    return df['Factor']
```
