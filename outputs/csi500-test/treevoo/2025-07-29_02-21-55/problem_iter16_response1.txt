```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Intraday Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_adjusted_momentum = intraday_momentum * volume_change
    
    # Apply Moving Averages to Volume-Adjusted Intraday Momentum
    fast_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    slow_ema = volume_adjusted_momentum.ewm(span=30, adjust=False).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    ma_log_returns = log_returns.rolling(window=20).mean()
    
    # Calculate 5-day Moving Average of Daily Close Price
    ma_close_5 = df['close'].rolling(window=5).mean()
    
    # Adjust Fast and Slow Exponential Moving Averages
    adjusted_ema = (fast_ema * ma_log_returns) - slow_ema + ma_close_5
    
    # Smooth the Adjusted Intraday Momentum
    final_alpha_factor = adjusted_ema.rolling(window=5).mean()
    
    return final_alpha_factor
```
