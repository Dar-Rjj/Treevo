```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10):
    # Calculate Volume-Weighted Moving Average (VWMA)
    df['VWMA'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Calculate Standard Deviation of VWMA
    df['VWMA_STD'] = df['VWMA'].rolling(window=n).std()
    
    # Calculate Mean of VWMA over n periods
    df['VWMA_MEAN'] = df['VWMA'].rolling(window=n).mean()
    
    # Calculate Normalized VWMA
    df['NORM_VWMA'] = (df['VWMA'] - df['VWMA_MEAN']) / df['VWMA_STD']
    
    # Calculate Momentum of VWMA
    df['MOM_VWMA'] = (df['VWMA'] - df['VWMA'].shift(n)) / df['VWMA'].shift(n)
    
    # Combine Normalized VWMA and Momentum
    df['COMBINED'] = (df['NORM_VWMA'] + df['MOM_VWMA']) / 2
    
    # Apply Exponential Moving Average (EMA) on the Combined Value
    df['ALPHA'] = df['COMBINED'].ewm(span=m, adjust=False).mean()
    
    return df['ALPHA']
```
