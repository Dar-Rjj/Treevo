```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_intraday_volume'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    threshold = 1.5
    df['significant_movement'] = df['combined_intraday_volume'] > threshold
    df['market_volatility'] = df['close'].pct_change().rolling(window=30).std()
    df['significant_movement_adj'] = df['combined_intraday_volume'] / df['market_volatility']
    
    # Confirm with Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=220).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    df['combined_momentum'] = 0.6 * df['long_term_momentum'] + 0.4 * df['short_term_momentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_pct_change'].ewm(span=12, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_pct_change'].ewm(span=26, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd_diff'] = df['fast_ema_intraday'] - df['slow_ema_intraday']
    df['combined_momentum_macd'] = df['weighted_momentum'] + df['macd_diff']
    
    # Incorporate Volume Confirmation
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_conf'] = np.where(df['volume'] > df['avg_volume_20'], 1, np.where(df['volume'] < df['avg_volume_20'], -1, 0))
    
    # Adjust Combined Momentum by Volume
    df['log_returns_20_ma'] = df['close'].pct_change().apply(np.log).rolling(window=20).mean()
    df['volume_20_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday'] * df['log_returns_20_ma']) / df['volume_20_ema'] - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adjusted_combined_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Synthesize Final Indicator
    final_indicator = (df['significant_movement_adj'] + df['combined_momentum_macd'] + df['smoothed_adjusted_combined_momentum']) / 3
    
    return final_indicator
```
