```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_indicator'] = df['intraday_hl_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    avg_return = daily_returns.rolling(window=20).mean()
    std_return = daily_returns.rolling(window=20).std()
    dynamic_threshold = avg_return + 2 * std_return
    df['significant_move'] = (df['combined_indicator'] > dynamic_threshold)
    
    # Confirm with Price Momentum
    lookback_period = 20
    df['sma'] = df['close'].rolling(window=lookback_period).mean()
    df['momentum_score'] = df['close'] - df['sma']
    
    # Synthesize Final Indicator
    weight_intraday_vol = 0.7
    weight_momentum = 0.3
    df['final_indicator'] = (weight_intraday_vol * df['combined_indicator'] + 
                             weight_momentum * df['momentum_score'])
    
    return df['final_indicator']
```
