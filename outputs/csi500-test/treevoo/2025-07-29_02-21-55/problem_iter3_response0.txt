```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return Components
    open_to_high = df['high'] / df['open'] - 1
    open_to_low = df['low'] / df['open'] - 1
    close_to_open = df['close'] / df['open'] - 1
    
    # Combine Intraday Returns
    avg_intraday_return = (open_to_high + open_to_low + close_to_open) / 3
    intraday_return_diff = (max(open_to_high, open_to_low, close_to_open) - 
                            min(open_to_high, open_to_low, close_to_open))
    
    # Volume Weighted Adjustments
    volume_weighted_avg = avg_intraday_return * df['volume']
    volume_weighted_diff = intraday_return_diff * df['volume']
    
    # Generate Final Alpha Factor
    alpha_factor = volume_weighted_avg - volume_weighted_diff
    alpha_factor = alpha_factor.ewm(alpha=0.7).mean()  # Exponential Moving Average with decay factor 0.95
    return alpha_factor
```

Note: The `max` and `min` functions used in the calculation of `intraday_return_diff` will not work directly on pandas Series. You need to use `df[['open_to_high', 'open_to_low', 'close_to_open']].max(axis=1)` and `df[['open_to_high', 'open_to_low', 'close_to_open']].min(axis=1)` instead. Additionally, the `ewm` function is applied with a specified alpha rather than a span or halflife. If you want to use a decay factor, you can adjust it accordingly.
