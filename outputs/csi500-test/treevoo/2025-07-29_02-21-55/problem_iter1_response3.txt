```python
import pandas as pd

def heuristics_v2(df):
    lookback_period = 20
    volume_threshold_factor = 1.5
    
    # Calculate the Simple Moving Average (SMA) of Close prices over the lookback period
    df['SMA'] = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate the Momentum Score
    df['Momentum_Score'] = df['close'] - df['SMA']
    
    # Calculate the average volume over the lookback period
    df['Average_Volume'] = df['volume'].rolling(window=lookback_period).mean()
    
    # Determine the Volume Threshold
    df['Volume_Threshold'] = volume_threshold_factor * df['Average_Volume']
    
    # Create a Volume Factor: 1 if the current volume is above the threshold, 0 otherwise
    df['Volume_Factor'] = (df['volume'] > df['Volume_Threshold']).astype(int)
    
    # Multiply the Momentum Score by the Volume Factor to get the final factor value
    df['Factor_Value'] = df['Momentum_Score'] * df['Volume_Factor']
    
    return df['Factor_Value']
```
