```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_percent_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_percent_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    # Set a Dynamic Threshold
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    threshold = 2 * volatility  # Adjust for market volatility
    significant_movements = combined_indicator.abs() > threshold.shift(1)
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Synthesize Final Indicator
    weighted_combined_indicator = 0.6 * combined_indicator  # Assign weight to combined indicator
    weighted_momentum_score = 0.4 * momentum_score  # Assign weight to momentum score
    final_indicator = significant_movements * (weighted_combined_indicator + weighted_momentum_score)
    
    return final_indicator
```
