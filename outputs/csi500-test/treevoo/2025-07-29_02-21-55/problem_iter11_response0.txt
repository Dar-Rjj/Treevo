```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / daily_range.rolling(window=25).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / daily_range.rolling(window=10).mean()
    
    # Combine Momentum Indicators
    combined_momentum = 0.7 * long_term_momentum + 0.3 * short_term_momentum
    
    # Calculate Intraday High-to-Low Percentage Change
    intraday_percentage_change = (df['high'] - df['low']) / df['low'] * 100
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_percentage_change * volume_ratio
    
    # Adjust for Market Volatility
    daily_returns = df['close'].pct_change()
    market_volatility = daily_returns.rolling(window=25).std()
    adjusted_combined_indicator = combined_indicator * market_volatility
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum + adjusted_combined_indicator
    
    return final_alpha_factor
```
