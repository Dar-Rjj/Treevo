```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_intraday_volume'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Confirm with Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=20).mean()
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=5).mean()
    df['combined_momentum'] = 0.7 * df['long_term_momentum'] + 0.3 * df['short_term_momentum']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_pct_change'].ewm(span=10, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_pct_change'].ewm(span=30, adjust=False).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum_sum'] = 0.4 * df['long_term_momentum'] + 0.6 * df['short_term_momentum']
    df['macd_diff'] = df['fast_ema_intraday'] - df['slow_ema_intraday']
    df['combined_momentum_macd'] = df['weighted_momentum_sum'] + df['macd_diff']
    
    # Incorporate Volume Confirmation
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_confirm'] = 0
    df.loc[df['volume_ratio'] > 1.5, 'volume_ratio_confirm'] = 1
    df.loc[df['volume_ratio'] < 0.5, 'volume_ratio_confirm'] = -1
    
    # Adjust Combined Momentum by Volume
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['ma_log_returns_20'] = df['log_returns'].rolling(window=20).mean()
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday'] * df['ma_log_returns_20']) / df['ema_volume_20'] - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Synthesize Final Indicator
    df['final_indicator'] = 0.5 * df['combined_intraday_volume'] + 0.5 * df['smoothed_momentum']
    
    return df['final_indicator'].dropna()
```
