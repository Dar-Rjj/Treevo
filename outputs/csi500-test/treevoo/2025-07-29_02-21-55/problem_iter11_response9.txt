```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    volatility = df['close'].pct_change().rolling(window=20).std()
    threshold = 2 * volatility
    significant_movements = combined_indicator.abs() > threshold
    
    # Confirm with Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / daily_range.rolling(window=20).mean()
    short_term_momentum = (df['close'] - df['close'].shift(5)) / daily_range.rolling(window=5).mean()
    momentum_score = 0.6 * long_term_momentum + 0.4 * short_term_momentum
    
    # Synthesize Final Indicator
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_threshold = 1.5 * avg_volume
    volume_significance = (df['volume'] > volume_threshold).astype(int)
    
    final_indicator = (combined_indicator * significant_movements) + (momentum_score * volume_significance)
    
    return final_indicator
```
