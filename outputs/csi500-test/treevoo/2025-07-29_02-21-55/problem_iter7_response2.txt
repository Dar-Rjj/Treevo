```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    
    # Adjust by Intraday Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = intraday_momentum * volume_change
    
    # Apply Weighted Moving Averages
    fast_ema = volume_adjusted_momentum.ewm(span=10, adjust=False).mean()
    slow_ema = volume_adjusted_momentum.ewm(span=30, adjust=False).mean()
    
    # Generate the alpha factor
    alpha_factor = fast_ema - slow_ema
    
    return alpha_factor
```
