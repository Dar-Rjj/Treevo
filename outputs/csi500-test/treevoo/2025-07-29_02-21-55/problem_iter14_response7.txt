```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from open to close and from high to low
    open_to_close = df['close'] - df['open']
    high_to_low = df['high'] - df['low']
    
    # Calculate the volume weighted by the amount
    volume_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the factor: (open_to_close + high_to_low) * volume_weighted_amount
    factor = (open_to_close + high_to_low) * volume_weighted_amount
    
    return factor
```
