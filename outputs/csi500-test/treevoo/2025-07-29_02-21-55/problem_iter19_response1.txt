```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_pct_change * volume_change_ratio
    significant_movements = (combined_indicator > 0.1).astype(int)  # Set a Fixed Threshold
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    ma_log_returns = daily_log_return.rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_ma_return = ma_log_returns * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_ma_return.ewm(span=12, adjust=False).mean()
    slow_ema = adjusted_ma_return.ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / ((df['high'] - df['low']).rolling(window=21).mean())
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(6)) / ((df['high'] - df['low']).rolling(window=7).mean())
    
    # Combine Momentum Indicators with MACD
    weighted_momentums = 0.45 * long_term_momentum + 0.55 * short_term_momentum
    macd_diff = fast_ema - slow_ema
    combined_factor = weighted_momentums + macd_diff
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    average_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / average_volume
    volume_trend = (volume_ratio > 1).astype(int) - (volume_ratio < 1).astype(int)
    final_factor = combined_factor * volume_trend
    
    return final_factor
```
