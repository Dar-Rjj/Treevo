```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_change_ratio
    significant_movements = combined_indicator.apply(lambda x: 1 if x > 0.05 else 0)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_return = daily_log_return.rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    exp_moving_avg_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_moving_avg_return = (moving_avg_log_return * exp_moving_avg_volume) / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_moving_avg_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_moving_avg_return.ewm(span=30, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = fast_ema - slow_ema
    
    return alpha_factor
```
