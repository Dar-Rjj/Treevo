```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20-day Return
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = df['high'] / df['low'] - 1
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd_diff'] = df['fast_ema_intraday'] - df['slow_ema_intraday']
    df['combined_momentum'] = df['weighted_momentum'] + df['macd_diff']
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = np.where(df['volume'] > df['avg_volume_20'], 1, np.where(df['volume'] < df['avg_volume_20'], -1, 0))
    df['20_day_ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Long-Term Price Trend
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['20_day_ma_log_returns'] = df['log_returns'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday'] * df['20_day_ma_log_returns']) / df['20_day_ema_volume'] - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adjusted_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['20_day_ma_log_returns'] = df['log_returns'].rolling(window=20).mean()
    
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Apply Threshold to Identify Significant Movements
    df['25_day_std_dev'] = df['close'].pct_change().rolling(window=25).std()
    df['significant_movement'] = np.where(np.abs(df['close'].pct_change()) > 2 * df['25_day_std_dev'], 1, 0)
    
    # Generate Final Alpha Factor
    df['weighted_intraday_range'] = df['daily_range'] * df['volume_ratio']
    df['weighted_intraday_direction'] = df['intraday_momentum'] * df['volume_ratio']
    df['final_alpha_factor'] = 0.7 * df['combined_momentum'] + df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    return df['final_alpha_factor']
```
