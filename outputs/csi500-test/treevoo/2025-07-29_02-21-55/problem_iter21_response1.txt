```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_change'] * df['volume_change_ratio']
    
    # Apply Threshold to Identify Significant Movements
    df['daily_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volatility'] = df['daily_return'].ewm(span=20, adjust=False).std()
    df['significant_movement'] = df['combined_indicator'] > 3 * df['volatility']
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_20_log_returns'] = df['daily_return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['ema_20_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_ma_return'] = df['ma_20_log_returns'] * df['ema_20_volume'] / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    df['fast_ema'] = df['adjusted_ma_return'].ewm(span=10, adjust=False).mean()
    df['slow_ema'] = df['adjusted_ma_return'].ewm(span=20, adjust=False).mean()
    
    # Confirm with Price Momentum
    df['sma_20_close'] = df['close'].rolling(window=20).mean()
    df['momentum_score'] = df['close'] - df['sma_20_close']
    
    # Introduce Volatility-Adjusted Momentum
    df['ema_abs_log_returns'] = df['daily_return'].abs().ewm(span=20, adjust=False).mean()
    df['volatility_adjusted_momentum'] = df['momentum_score'] / df['ema_abs_log_returns']
    
    # Synthesize Final Indicator
    df['final_indicator'] = 0.5 * df['combined_indicator'] + 0.5 * df['volatility_adjusted_momentum']
    
    return df['final_indicator']
```
