```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'].pct_change()
    df['combined_indicator'] = df['intraday_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    df['daily_return'] = df['close'].pct_change()
    df['ema_std_20'] = df['daily_return'].ewm(span=20).std()
    threshold = 2 * df['ema_std_20']
    df['significant_movement'] = df['combined_indicator'].abs() > threshold
    
    # Confirm with Price Momentum
    df['ema_close_20'] = df['close'].ewm(span=20).mean()
    df['momentum_score'] = df['close'] - df['ema_close_20']
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return_20'] = df['log_return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_ma_return'] = (df['ma_log_return_20'] * df['ema_volume_20']) / df['sma_volume_20']
    
    # Apply Moving Averages to Volume-Adjusted Intraday Momentum
    df['fast_ema_momentum'] = df['momentum_score'].ewm(span=5, adjust=False).mean()
    df['slow_ema_momentum'] = df['momentum_score'].ewm(span=20, adjust=False).mean()
    
    # Combine Momentum, Volume, and Long-Term Trend
    df['combined_factor'] = (df['fast_ema_momentum'] * df['adjusted_ma_return']) / df['amount'].ewm(span=20).mean()
    df['combined_factor'] -= df['slow_ema_momentum']
    df['combined_factor'] += df['close'].rolling(window=5).mean()
    
    # Smooth the Adjusted Intraday Momentum
    df['final_alpha_factor'] = df['combined_factor'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
