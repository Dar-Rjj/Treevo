def heuristics_v2(df):
    # Calculate the weighted average of open and close prices using volume as weights
    weighted_avg = (df['open'] * df['volume'] + df['close'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between high and low prices
    range_diff = df['high'] - df['low']
    
    # Calculate the factor value by dividing the weighted average by the range difference
    factor_value = weighted_avg / range_diff
    
    return factor_value
