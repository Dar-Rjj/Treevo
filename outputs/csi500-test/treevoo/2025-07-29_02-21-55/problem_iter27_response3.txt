```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    df['combined_indicator'] = df['intraday_high_low_pct_change'] * df['volume_ratio']
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    threshold = daily_returns.rolling(window=30).mean() + 2 * daily_returns.rolling(window=30).std()
    df['significant_movement'] = df['combined_indicator'] > threshold
    
    # Confirm with Price Momentum
    lookback_period = 25
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    df['momentum_score'] = df['close'] - ema_close
    
    # Synthesize Final Indicator
    weight_intraday_volume = 0.7
    weight_momentum = 0.3
    df['final_indicator'] = (weight_intraday_volume * df['combined_indicator']) + (weight_momentum * df['momentum_score'])
    
    return df['final_indicator']
```
