```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum and Intraday High-to-Low Momentum
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday_mom'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()
    df['slow_ema_intraday_mom'] = df['intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = np.where(df['volume'] > df['avg_volume_20_days'], 1, 
                                  np.where(df['volume'] < df['avg_volume_20_days'], -1, 0))
    df['20_day_ema_volume'] = df['volume'].ewm(alpha=0.1, adjust=False).mean()
    
    # Incorporate Price Oscillation and Long-Term Price Trend
    df['20_day_max_high'] = df['high'].rolling(window=20).max()
    df['20_day_min_low'] = df['low'].rolling(window=20).min()
    df['20_day_price_range_difference'] = (df['high'] - df['low'].shift(19)) - (df['20_day_max_high'] - df['20_day_min_low'])
    
    def oscillation_score(row):
        max_range = row['20_day_max_high'] - row['20_day_min_low']
        actual_range = row['high'] - row['low'].shift(19)
        if actual_range > 0.5 * max_range:
            return 1
        elif actual_range < 0.2 * max_range:
            return -1
        else:
            return 0
    
    df['oscillation_score'] = df.apply(oscillation_score, axis=1)
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['20_day_moving_avg_log_returns'] = df['log_returns'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_combined_momentum'] = ((df['fast_ema_intraday_mom'] * df['20_day_moving_avg_log_returns']) / 
                                        df['20_day_ema_volume']) - df['slow_ema_intraday_mom']
    
    # Smooth the Adjusted Combined Momentum
    df['final_alpha_factor'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
