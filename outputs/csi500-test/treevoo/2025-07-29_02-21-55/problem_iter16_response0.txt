```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    combined_intraday_volume = intraday_change * volume_ratio
    
    # Confirm with Price Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / (df['high'] - df['low']).rolling(window=20).mean()
    short_term_momentum = (df['close'] - df['close'].shift(5)) / (df['high'] - df['low']).rolling(window=5).mean()
    momentum_score = 0.6 * long_term_momentum + 0.4 * short_term_momentum
    
    # Synthesize Final Indicator
    average_volume = df['volume'].rolling(window=20).mean()
    volume_threshold = df['volume'] > 1.5 * average_volume
    final_indicator = (combined_intraday_volume * volume_threshold) + momentum_score
    
    return final_indicator
```
