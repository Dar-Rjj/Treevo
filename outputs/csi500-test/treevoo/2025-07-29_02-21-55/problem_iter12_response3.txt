```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_change_ratio
    
    # Set a Dynamic Threshold
    moving_avg_combined = combined_indicator.rolling(window=20).mean()
    moving_std_combined = combined_indicator.rolling(window=20).std()
    market_volatility = df['close'].rolling(window=20).std()
    dynamic_threshold = moving_avg_combined + 2 * moving_std_combined * market_volatility
    
    # Apply Threshold to Identify Significant Movements
    significant_movements = combined_indicator > dynamic_threshold
    
    # Calculate Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Set a Volume Threshold
    volume_threshold = 1.5 * df['volume'].rolling(window=20).mean()
    high_volume = df['volume'] > volume_threshold
    
    # Synthesize Final Indicator
    final_indicator = (combined_indicator * significant_movements) + (momentum_score * high_volume)
    
    # Assign Weights to Each Component
    weight_combined = 0.6
    weight_momentum = 0.4
    final_indicator = (weight_combined * combined_indicator * significant_movements) + (weight_momentum * momentum_score * high_volume)
    
    return final_indicator
```
