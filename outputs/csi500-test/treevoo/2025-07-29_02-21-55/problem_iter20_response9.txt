```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_change_ratio
    
    # Calculate 20-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    ems_daily_returns = daily_returns.ewm(span=20).std()
    
    # Apply Threshold to Identify Significant Movements
    dynamic_threshold = 2 * ems_daily_returns
    significant_movements = combined_indicator > dynamic_threshold
    
    # Calculate Price Momentum using 20-day EMA
    ema_20_close = df['close'].ewm(span=20).mean()
    price_momentum = df['close'] / ema_20_close - 1
    
    # Set a Volume Threshold (e.g., 1.5 * Exponential Moving Average Volume over the lookback period)
    ema_20_volume = df['volume'].ewm(span=20).mean()
    volume_threshold = 1.5 * ema_20_volume
    
    # Combine Momentum and Volume Indicators
    combined_score = (combined_indicator * significant_movements) + price_momentum
    combined_score = combined_score * (df['volume'] > volume_threshold)
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Synthesize Final Indicator with Weights
    final_indicator = (0.4 * combined_score) + (0.4 * price_momentum) + (0.2 * opening_gap)
    
    return final_indicator
```
