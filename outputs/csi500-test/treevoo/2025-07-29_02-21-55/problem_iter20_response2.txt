```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_high_low_pct'] = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    threshold = 2.0  # Dynamic threshold can be set based on historical volatility
    df['combined_indicator'] = df['intraday_high_low_pct'] * df['volume_change_ratio']
    df['significant_movement'] = df['combined_indicator'].apply(lambda x: 1 if abs(x) > threshold else 0)
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return_20'] = df['log_return'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['adjusted_ma_log_return'] = df['ma_log_return_20'] * df['ema_volume_20'] / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    df['fast_ema_adj_return'] = df['adjusted_ma_log_return'].ewm(span=5, adjust=False).mean()
    df['slow_ema_adj_return'] = df['adjusted_ma_log_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday_momentum'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()
    df['slow_ema_intraday_momentum'] = df['intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd'] = df['fast_ema_adj_return'] - df['slow_ema_adj_return']
    df['combined_momentum_macd'] = df['weighted_momentum'] + df['macd']
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    df['volume_ratio'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else (-1 if x < 1 else 0))
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday_momentum'] * df['ma_log_return_20']) / df['ema_volume_20'] - df['slow_ema_intraday_momentum']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_adj_combined_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    # Generate Final Alpha Factor
    df['weighted_intraday_range'] = df['intraday_momentum'] * df['volume_ratio']
    df['weighted_intraday_direction'] = df['intraday_high_low_pct'] * df['volume_ratio']
    df['final_alpha_factor'] = 0.7 * df['combined_momentum_macd'] + df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    return df['final_alpha_factor']
```
