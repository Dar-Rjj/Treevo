```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_pct_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_pct_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    # Calculate 20-Day Exponential Moving Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    ema_std_20 = daily_returns.ewm(span=20, adjust=False).std()
    dynamic_threshold = 3 * ema_std_20
    significant_movements = (combined_indicator > dynamic_threshold) | (combined_indicator < -dynamic_threshold)
    
    # Confirm with Price Momentum
    lookback_period = 20
    ema_close_20 = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close_20
    
    # Combine Momentum and Volume Indicators
    volume_threshold = 1.5 * df['volume'].ewm(span=lookback_period, adjust=False).mean()
    significant_volume = df['volume'] > volume_threshold
    final_indicator = (combined_indicator * significant_movements * significant_volume) + momentum_score
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Incorporate Opening Gap into Final Indicator
    # Assign Weights to Each Component
    weight_intraday_vol = 0.4
    weight_momentum = 0.3
    weight_opening_gap = 0.3
    final_indicator = (weight_intraday_vol * combined_indicator * significant_movements * significant_volume) + \
                      (weight_momentum * momentum_score) + \
                      (weight_opening_gap * opening_gap)
    
    return final_indicator
```
