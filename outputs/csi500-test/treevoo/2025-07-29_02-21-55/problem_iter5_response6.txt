```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day return (Price Momentum)
    df['20_day_return'] = df['close'].pct_change(periods=20)

    # Calculate Volume Ratio (Volume Confirmation)
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_20_days']
    df['volume_confirmation'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else -1 if x < 1 else 0)

    # Incorporate Price Oscillation
    df['high_20_max'] = df['high'].rolling(window=20).max()
    df['low_20_min'] = df['low'].rolling(window=20).min()
    df['price_range_20_days'] = df['high_20_max'] - df['low_20_min']
    df['actual_range_20_days'] = df['high'] - df['low'].shift(19)
    df['oscillation_score'] = df.apply(
        lambda row: 1 if row['actual_range_20_days'] > 0.5 * row['price_range_20_days'] else -1 if row['actual_range_20_days'] < 0.2 * row['price_range_20_days'] else 0,
        axis=1
    )

    # Identify Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike_threshold'] = 2 * df['avg_volume_20_days']
    df['volume_spike'] = (df['volume_change'] > df['volume_spike_threshold']).astype(int)

    # Combine Price Momentum and Volume Spike
    df['adjusted_momentum'] = df['20_day_return'] * (2.0 if df['volume_spike'] == 1 else 1.0)

    # Calculate Intraday Price Movement
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_direction'] = df['close'] - df['open']

    # Calculate Weighted Volume Activity
    df['relative_volume'] = df['volume'] / df['avg_volume_20_days']

    # Combine Intraday Price and Volume Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['relative_volume']
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']
    df['combined_intraday_metric'] = df['weighted_intraday_range'] + df['weighted_intraday_direction']

    # Adjust Combined Metric by Price Momentum and Volume Spike
    df['adjusted_combined_metric'] = df['combined_intraday_metric'] * (2.0 if df['volume_spike'] == 1 else 1.0)

    # Generate Final Alpha Factor
    df['alpha_factor'] = df['adjusted_combined_metric'] + df['adjusted_momentum'] + df['volume_confirmation'] + df['oscillation_score']

    return df['alpha_factor']
```
