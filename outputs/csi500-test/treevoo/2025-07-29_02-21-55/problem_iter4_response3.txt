```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close Price Change
    df['close_change'] = df['close'].diff()
    
    # Apply Simple Moving Average on Close Price Changes (e.g., 20 days)
    momentum_window = 20
    df['price_momentum'] = df['close_change'].rolling(window=momentum_window).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Define a Threshold as a Multiple of Recent Volume (e.g., 2x the average volume of the last 20 days)
    recent_volume_avg = df['volume'].rolling(window=momentum_window).mean()
    volume_threshold = 2 * recent_volume_avg
    
    # Identify Volume Spike
    df['volume_spike'] = df['volume_change'] > volume_threshold
    
    # Combine Price Momentum and Volume Spike
    high_factor = 2.0
    df['alpha_factor'] = df['price_momentum'].where(~df['volume_spike'], df['price_momentum'] * high_factor)
    
    return df['alpha_factor']
```
