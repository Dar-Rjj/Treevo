```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'].pct_change().fillna(0) + 1
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change().fillna(0)
    ema_std_20 = daily_returns.ewm(span=20).std()
    dynamic_threshold = 2 * ema_std_20
    significant_movements = (combined_indicator > dynamic_threshold).astype(int) * combined_indicator
    
    # Calculate Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Set a Dynamic Volume Threshold
    sma_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * sma_volume
    significant_volume = (df['volume'] > volume_threshold).astype(int) * significant_movements
    
    # Synthesize Final Indicator
    weight_momentum = 0.6
    weight_volume = 0.4
    final_indicator = (weight_momentum * momentum_score + weight_volume * significant_volume)
    
    return final_indicator
```
