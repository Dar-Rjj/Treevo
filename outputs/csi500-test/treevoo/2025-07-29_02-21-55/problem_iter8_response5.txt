```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    combined_indicator = intraday_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    volatility = df['close'].pct_change().rolling(window=20).std()
    significant_movement = (combined_indicator > 2 * volatility)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_return = daily_log_return.rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_moving_avg_return = moving_avg_return * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_moving_avg_return.ewm(span=5, adjust=False).mean()
    slow_ema = adjusted_moving_avg_return.ewm(span=20, adjust=False).mean()
    
    # Generate the final alpha factor
    alpha_factor = fast_ema - slow_ema
    
    return alpha_factor
```
