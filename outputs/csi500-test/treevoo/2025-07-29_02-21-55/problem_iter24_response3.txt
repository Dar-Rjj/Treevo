import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['Daily_Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['5_Day_Momentum'] = df['close'] / df['close'].shift(5) - 1
    
    # Volatility Indicators
    df['True_Range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], 
                      abs(x['high'] - df['close'].shift(1).iloc[x.name]), 
                      abs(x['low'] - df['close'].shift(1).iloc[x.name])), axis=1)
    df['Average_True_Range_5'] = df['True_Range'].rolling(window=5).mean()
    
    # Volume-based Alpha Factors
    df['Daily_Volume_Growth'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Raw_Money_Flow'] = df['Typical_Price'] * df['volume']
    df['Positive_Money_Flow'] = df.apply(lambda x: x['Raw_Money_Flow'] if x['close'] > df['close'].shift(1).iloc[x.name] else 0, axis=1)
    df['Negative_Money_Flow'] = df.apply(lambda x: x['Raw_Money_Flow'] if x['close'] < df['close'].shift(1).iloc[x.name] else 0, axis=1)
    
    # Combined Price and Volume Alpha Factors
    df['Accumulation_Distribution_Line'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['AD_Line_3Day_EMA'] = df['Accumulation_Distribution_Line'].ewm(span=3, adjust=False).mean()
    df['AD_Line_10Day_EMA'] = df['Accumulation_Distribution_Line'].ewm(span=10, adjust=False).mean()
    df['Chaikin_Oscillator'] = df['AD_Line_3Day_EMA'] - df['AD_Line_10Day_EMA']
    
    # High and Low Difference Factors
    df['Percentage_Price_Oscillator_%K'] = 100 * ((df['close'] - df['low'].rolling(window=14).min()) / 
                                                    (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()))
    df['Percentage_Price_Oscillator_%D'] = df['Percentage_Price_Oscillator_%K'].rolling(window=3).mean()
    
    # Gap Analysis Factors
    df['Opening_Gap_Size'] = abs(df['open'] - df['close'].shift(1))
    df['Normalized_Gap'] = df['Opening_Gap_Size'] / (df['high'].shift(1) - df['low'].shift(1))
    df['Close_Above_High_Previous_Day'] = df.apply(lambda x: 1 if x['close'] > df['high'].shift(1).iloc[x.name] else 0, axis=1)
    
    # Pattern Recognition
    df['Previous_Resistance_As_Support'] = df.apply(lambda x: 1 if x['low'] > df['high'].shift(5).iloc[x.name] else 0, axis=1)
    df['Hammer_Candle_Formation'] = df.apply(
        lambda x: 1 if (x['close'] - x['low']) / (x['high'] - x['low']) > 0.67 and (x['close'] - x['low']) / (x['high'] - x['low']) < 0.8 and
                  (x['close'] - x['open']) / (x['high'] - x['low']) > 0.1 and (x['close'] - x['open']) / (x['high'] - x['low']) < 0.4 else 0, axis=1)
    
    # Custom Interactions
    df['Relative_Close_to_Volume_Ratio'] = (df['close'] / df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    df['5_Day_MA'] = df['close'].rolling(window=5).mean()
    df['20_Day_MA'] = df['close'].rolling(window=20).mean()
    df['MA_Crossover_Interaction']
