```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate 20-day Return
    twenty_day_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Intraday High-to-Low Momentum
    intraday_momentum = (df['high'] / df['low']) - 1
    
    # Apply Moving Averages to Intraday Momentum
    fast_ema_intraday = intraday_momentum.ewm(span=12, adjust=False).mean()
    slow_ema_intraday = intraday_momentum.ewm(span=26, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(21)) / daily_range.rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(6)) / daily_range.rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    macd = fast_ema_intraday - slow_ema_intraday
    combined_momentum = 0.45 * long_term_momentum + 0.55 * short_term_momentum + macd
    
    # Incorporate Volume Confirmation
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    volume_confirmation = np.where(volume_ratio > 1, 1, np.where(volume_ratio < 1, -1, 0))
    
    # Adjust Combined Momentum by Volume
    log_returns = np.log(df['close'] / df['close'].shift(1))
    adjusted_combined_momentum = (fast_ema_intraday * log_returns.rolling(window=20).mean()) / (df['volume'].ewm(span=20, adjust=False).mean()) - slow_ema_intraday
    
    # Smooth the Adjusted Combined Momentum
    smoothed_momentum = adjusted_combined_momentum.rolling(window=5).mean() * volume_confirmation
    
    return smoothed_momentum
```
