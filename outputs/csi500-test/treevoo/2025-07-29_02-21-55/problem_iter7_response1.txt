```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_to_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_high_to_low_pct_change * volume_change_ratio
    
    # Apply Threshold to Identify Significant Movements
    fixed_threshold = 0.05
    daily_returns = df['close'].pct_change()
    moving_std = daily_returns.rolling(window=20).std()
    significant_movement = (combined_indicator > fixed_threshold) & (combined_indicator > moving_std)
    
    # Confirm with Price Momentum
    lookback_period = 20
    sma_close = df['close'].rolling(window=lookback_period).mean()
    momentum_score = df['close'] - sma_close
    
    # Combine Momentum and Volume Indicators
    weight_volume = 0.6
    weight_momentum = 0.4
    weighted_sum = (combined_indicator * weight_volume) + (momentum_score * weight_momentum)
    
    # Ensure the Weighted Sum Reflects Both Momentum and Volume Significance
    final_factor = weighted_sum.where(significant_movement, 0)
    
    return final_factor
```
