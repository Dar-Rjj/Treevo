```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close Price Change
    df['close_change'] = df['close'].diff()
    
    # Compute Moving Average of Close Price Change (20 days)
    df['price_momentum'] = df['close_change'].rolling(window=20).mean()
    
    # Identify Volume Spike
    df['volume_change'] = df['volume'].pct_change()
    spike_threshold = 2.0
    df['volume_spike'] = df['volume_change'] > spike_threshold
    
    # Adjust Price Momentum for Volume Spike
    high_factor = 2.0
    df['adjusted_price_momentum'] = df['price_momentum']
    df.loc[df['volume_spike'], 'adjusted_price_momentum'] *= high_factor
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_hl_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    significant_movement_threshold = 0.05
    df['combined_indicator'] = df['intraday_hl_pct_change'] * df['volume_ratio']
    df['significant_movement'] = df['combined_indicator'] > significant_movement_threshold
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return_20'] = df['log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    smoothing_factor = 0.1
    df['ema_volume_20'] = df['volume'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_ma_return'] = (df['ma_log_return_20'] * df['ema_volume_20']) / df['sma_volume_20']
    
    # Smooth the Adjusted Return (5-day Simple Moving Average)
    df['smoothed_adjusted_return'] = df['adjusted_ma_return'].rolling(window=5).mean()
    
    # Calculate Intraday Price Movement
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_direction'] = df['close'] - df['open']
    
    # Calculate Weighted Volume Activity
    n_days = 20
    df['average_volume'] = df['volume'].rolling(window=n_days).mean()
    df['relative_volume'] = df['volume'] / df['average_volume']
    
    # Combine Intraday Price and Volume Metrics
    df['weighted_intraday_range'] = df['intraday_range'] * df['relative_volume']
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']
    df['combined_metric'] = df['weighted_intraday_range'] + df['weighted_intraday_direction']
    
    # Adjust Combined Metric by Price Momentum and Volume Spike
    df['final_combined_metric'] = df['combined_metric']
    df.loc[df['volume_spike'], 'final_combined_metric'] *= high_factor
    
    # Incorporate Smoothed Adjusted Return
    df['final_alpha_factor'] = df['final_combined_metric'] + df['smoothed_adjusted_return']
    
    return df['final_alpha_factor']
```
