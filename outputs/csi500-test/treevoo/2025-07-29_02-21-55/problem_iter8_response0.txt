```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Momentum
    intraday_momentum = df['high'] / df['low']

    # Adjust by Intraday Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    volume_adjusted_momentum = intraday_momentum * volume_change

    # Apply Moving Averages to Volume-Adjusted Intraday Momentum
    fast_ema = volume_adjusted_momentum.ewm(span=5, adjust=False).mean()
    slow_ema = volume_adjusted_momentum.ewm(span=20, adjust=False).mean()

    # Calculate 20-day Moving Average of Daily Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    avg_log_returns = log_returns.rolling(window=20).mean()

    # Calculate 20-day Exponential Moving Average of Amount
    amount_ema = df['amount'].ewm(span=20, adjust=False).mean()

    # Adjust Fast and Slow Exponential Moving Averages by Amount and Long-Term Price Trend
    adjusted_fast_ema = (fast_ema * avg_log_returns) / amount_ema
    adjusted_slow_ema = slow_ema
    adjusted_intraday_momentum = adjusted_fast_ema - adjusted_slow_ema

    # Smooth the Adjusted Intraday Momentum
    final_alpha_factor = adjusted_intraday_momentum.rolling(window=5).mean()

    return final_alpha_factor
```
