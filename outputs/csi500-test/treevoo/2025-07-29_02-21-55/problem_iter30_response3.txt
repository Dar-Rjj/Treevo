```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 2.0
    moving_std = df['close'].pct_change().rolling(window=30).std()
    adjusted_threshold = threshold * moving_std
    significant_movements = (combined_indicator > adjusted_threshold)
    
    # Confirm with Price Momentum
    lookback_period = 30
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Synthesize Final Indicator
    recent_trend = df['close'].rolling(window=10).mean().diff().fillna(0) > 0
    weight_intraday_vol = 0.7 + 0.1 * (~recent_trend)
    weight_momentum = 0.3 - 0.1 * (~recent_trend)
    final_indicator = (weight_intraday_vol * significant_movements * combined_indicator +
                       weight_momentum * momentum_score)
    
    return final_indicator
```
