    df['Momentum_EMA'] = df['close'] - df['EMA']

    df['Reversal_Below_N_Low'] = (df['close'] < df['N_Low']).astype(int)

    gain = (delta.where(delta > 0, 0)).rolling(window=N).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=N).mean()
    df['RSI'] = 100 - (100 / (1 + (gain / loss)))

    df['Volume_Sum_M'] = df['volume'].rolling(window=M).sum()
