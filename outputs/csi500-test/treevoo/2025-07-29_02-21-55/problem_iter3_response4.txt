```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Simple Moving Average of Volume
    df['sma_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Amount
    df['ema_amount'] = df['amount'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Simple Moving Average of Amount
    df['sma_amount'] = df['amount'].rolling(window=20).mean()
    
    # Adjust Moving Average Return by Volume and Amount
    df['adjusted_return'] = (df['ma_log_return'] * df['ema_volume'] / df['sma_volume'] 
                             * df['ema_amount'] / df['sma_amount'])
    
    # Smooth the Adjusted Return with a 5-day Simple Moving Average
    df['final_factor'] = df['adjusted_return'].rolling(window=5).mean()
    
    return df['final_factor']
```
