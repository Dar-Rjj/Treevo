```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Incorporate Volume Spike
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    daily_returns = df['close'].pct_change()
    moving_avg_return = daily_returns.rolling(window=30).mean()
    moving_std_return = daily_returns.rolling(window=30).std()
    threshold = moving_avg_return + 2 * moving_std_return
    significant_movement = combined_indicator > threshold
    
    # Confirm with Price Momentum
    lookback_period = 25
    ema_close = df['close'].ewm(span=lookback_period, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Synthesize Final Indicator
    final_indicator = 0.7 * significant_movement * combined_indicator + 0.3 * momentum_score
    
    return final_indicator
```
