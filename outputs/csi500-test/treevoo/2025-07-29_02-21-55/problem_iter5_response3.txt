```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=22).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.6 * df['long_term_momentum'] + 0.4 * df['short_term_momentum']
    
    # Identify Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 2 * df['volume'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume_change'] > volume_spike_threshold).astype(int)
    
    # Incorporate Volume Spike into Momentum
    df['adjusted_momentum'] = df['combined_momentum'] * (2.0 if df['volume_spike'] == 1 else 1.0)
    
    # Calculate Intraday High-to-Low Percentage Change
    df['intraday_high_to_low_pct'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Direction
    df['intraday_direction'] = (df['close'] - df['open']) / df['open']
    
    # Relative Volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Adjust Combined Metric by Weighted Intraday Metrics
    df['weighted_intraday_high_to_low_pct'] = df['intraday_high_to_low_pct'] * df['relative_volume']
    df['weighted_intraday_direction'] = df['intraday_direction'] * df['relative_volume']
    df['intraday_metrics_sum'] = df['weighted_intraday_high_to_low_pct'] + df['weighted_intraday_direction']
    
    # Adjust Combined Metric by Momentum and Volume Spike
    df['final_alpha_factor'] = df['adjusted_momentum'] * (2.0 if df['volume_spike'] == 1 else 1.0) + df['intraday_metrics_sum']
    
    return df['final_alpha_factor']
```
