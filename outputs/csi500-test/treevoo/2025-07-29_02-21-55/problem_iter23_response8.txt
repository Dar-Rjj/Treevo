```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_high_low_pct = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_high_low_pct * volume_change_ratio
    
    # Set a Dynamic Threshold Based on Historical Volatility
    daily_returns = np.log(df['close'] / df['close'].shift(1))
    ema_volatility = daily_returns.ewm(span=20).std()
    threshold = 2 * ema_volatility
    significant_movement = combined_indicator[combined_indicator > threshold]
    
    # Calculate 20-day Moving Average of Daily Log Returns
    moving_avg_log_return = daily_returns.rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    adjusted_moving_avg_return = moving_avg_log_return * (ema_volume / df['volume'].rolling(window=20).mean())
    
    # Apply Weighted Moving Averages
    fast_ema = adjusted_moving_avg_return.ewm(span=10, adjust=False).mean()
    slow_ema = adjusted_moving_avg_return.ewm(span=30, adjust=False).mean()
    
    # Calculate Price Momentum
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    momentum_score = df['close'] - ema_close
    
    # Calculate 20-day Exponential Moving Average of Absolute Log Returns
    ema_abs_log_returns = np.abs(daily_returns).ewm(span=20, adjust=False).mean()
    volatility_adjusted_momentum = momentum_score / ema_abs_log_returns
    
    # Integrate Opening Gap Analysis
    opening_gap = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Synthesize Final Indicator
    final_indicator = (
        0.4 * significant_movement +
        0.3 * fast_ema +
        0.2 * slow_ema +
        0.1 * volatility_adjusted_momentum +
        0.1 * opening_gap
    )
    
    return final_indicator
```
