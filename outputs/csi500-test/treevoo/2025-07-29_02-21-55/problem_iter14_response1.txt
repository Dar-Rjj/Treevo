import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    df['high_low_pct_change'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    threshold = 1.5
    df['combined_indicator'] = df['high_low_pct_change'] * df['volume_ratio']
    df['significant_movement'] = (df['combined_indicator'] > threshold).astype(int)
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day Moving Average of Daily Log Returns
    df['ma_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Calculate 20-day Exponential Moving Average of Volume
    smoothing_factor = 0.1
    df['ema_volume'] = df['volume'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Adjust Moving Average Return by Volume
    df['adjusted_ma_return'] = df['ma_log_return'] * df['ema_volume'] / df['volume'].rolling(window=20).mean()
    
    # Apply Weighted Moving Averages
    fast_ema_window = 12
    slow_ema_window = 26
    df['fast_ema_adjusted'] = df['adjusted_ma_return'].ewm(span=fast_ema_window, adjust=False).mean()
    df['slow_ema_adjusted'] = df['adjusted_ma_return'].ewm(span=slow_ema_window, adjust=False).mean()
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Intraday High-to-Low Momentum
    df['intraday_momentum'] = (df['high'] - df['low'])
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_momentum'].ewm(span=fast_ema_window, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_momentum'].ewm(span=slow_ema_window, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['daily_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(6)) / df['daily_range'].rolling(window=7).mean()
    
    # Combine Momentum Indicators with MACD
    df['weighted_momentum_sum'] = 0.45 * df['long_term_momentum'] + 0.55 * df['short_term_momentum']
    df['macd'] = df['fast_ema_adjusted'] - df['slow_ema_adjusted']
    df['combined_momentum'] = df['weighted_momentum_sum'] + df['macd']
    
    # Incorporate Volume Confirmation and Long-Term Volume Trend
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio_trend'] = df['volume'] / df['avg_volume']
    df['volume_ratio_trend'] = df['volume_ratio_trend'].apply(lambda x: 0 if x == 1 else -1 if x < 1 else 1)
    
    # Incorporate Long-Term Price Trend
    df['long_term_price_trend'] = df['log_return'].rolling(window=20).mean()
    
    # Adjust Combined Momentum by Volume and Long-Term Trends
    df['adjusted_combined_momentum'] = (df['fast_ema_intraday'] * df['long_term_price_trend']) / df['ema_volume'] - df['slow_ema_intraday']
    
    # Smooth the Adjusted Combined Momentum
    df['smoothed_momentum'] = df['adjusted_combined_momentum'].rolling(window=5).mean()
    
    return df['smoothed_momentum']
