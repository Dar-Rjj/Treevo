```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['daily_range'].rolling(window=25).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['daily_range'].rolling(window=10).mean()
    
    # Combine Momentum Indicators
    df['combined_momentum'] = 0.7 * df['long_term_momentum'] + 0.3 * df['short_term_momentum']
    
    # Calculate Intraday High-to-Low Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Apply Moving Averages to Intraday Momentum
    df['fast_ema_intraday'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()
    df['slow_ema_intraday'] = df['intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Adjust Intraday Momentum by Combined Momentum and Volume
    df['daily_log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['adj_intraday_momentum'] = (df['fast_ema_intraday'] * df['daily_log_returns'].rolling(window=20).mean()) / df['amount'].ewm(span=20, adjust=False).mean() - df['slow_ema_intraday']
    
    # Incorporate Volume Spike
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    daily_returns = df['close'].pct_change()
    df['volatility'] = daily_returns.rolling(window=25).std()
    volume_spike = (df['volume_change_ratio'] > 1.5) & (df['volatility'] > 0.01)
    
    # Generate Final Alpha Factor
    df['weighted_intraday_range'] = df['intraday_momentum'] * (df['volume'] / df['volume'].rolling(window=20).mean())
    df['weighted_intraday_direction'] = df['adj_intraday_momentum'] * (df['volume'] / df['volume'].rolling(window=20).mean())
    df['final_alpha_factor'] = 0.7 * df['combined_momentum'] + 0.15 * df['weighted_intraday_range'] + 0.15 * df['weighted_intraday_direction']
    
    return df['final_alpha_factor']
```
