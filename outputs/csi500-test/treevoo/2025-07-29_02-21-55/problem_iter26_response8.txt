```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-to-Low Percentage Change
    intraday_change = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Compute Combined Indicator
    combined_indicator = intraday_change * volume_ratio
    
    # Apply Threshold to Identify Significant Movements
    threshold = 2.0  # Example fixed threshold
    significant_movements = combined_indicator.apply(lambda x: x if abs(x) > threshold else 0)
    
    # Determine the Lookback Period (e.g., 20 days)
    lookback_period = 20
    sma = df['close'].rolling(window=lookback_period).mean()
    
    # Calculate the Momentum Score
    momentum_score = df['close'] - sma
    
    # Set a Volume Threshold (e.g., 1.5 * Average Volume over the lookback period)
    average_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_threshold = 1.5 * average_volume
    
    # Synthesize Final Indicator
    final_indicator = (significant_movements + momentum_score) * (df['volume'] > volume_threshold)
    
    return final_indicator
```
