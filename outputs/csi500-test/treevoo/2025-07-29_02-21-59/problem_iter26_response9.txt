import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['Volume_Amount'] = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['Volume_Amount']
    
    # Integrate Intraday High-Low Movement
    df['Intraday_High_Low_Ratio'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust by Volume Change
    df['Adjusted_High_Low_Ratio'] = df['Intraday_High_Low_Ratio'] * (df['Volume_Change'] + 1)
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Combined Intraday Momentum
    df['Combined_Intraday_Momentum'] = df['Adjusted_High_Low_Ratio'] * df['Intraday_Return']
    
    # Adjust for Volume Volatility
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_STD_220'] = df['volume'].rolling(window=220).std()
    
    # Cumulative Log Return with VWAP Integration
    df['Log_Return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    df['Cumulative_Log_Return_20'] = df['Log_Return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    df['VWAP_MA_20'] = df['VWAP'].rolling(window=20).mean()
    
    # Weighted Intraday Momentum
    df['Weighted_Intraday_Momentum'] = df['Intraday_High_Low_Ratio'] * df['Intraday_Return']
    
    # PMI Score
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Sum_Daily_Price_Changes_20'] = df['Daily_Price_Change'].rolling(window=20).sum()
    df['PMI_Score'] = df['Sum_Daily_Price_Changes_20'] * df['VWAP']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['PMI_Score'] * df['Cumulative_Log_Return_20'] +
                                df['Weighted_Close_to_Open_Return'] +
                                df['Cumulative_Log_Return_20'] +
                                (df['Volume_Change'] + (df['amount'] - df['amount'].shift(1))) * df['Combined_Intraday_Momentum'])
    
    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Price_Oscillator'] = df['SMA_5'] - df['SMA_20']
    
    df['Final_Alpha_Factor'] += df['Price_Oscillator']
    
    return df['Final_Alpha_Factor']
