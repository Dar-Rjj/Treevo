```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted close
    vwap_like = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Heuristic alpha factor combining all components
    heuristic_factor = (buying_pressure * price_range) + vwap_like
    
    return heuristic_factor
```
