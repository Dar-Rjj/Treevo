import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['close']
    df['intraday_return'] = df['close'] / df['open'] - 1
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_adjusted_return'] = df['intraday_return'] * df['volume_change']
    df['volume_adjusted_intraday_momentum'] = df['volume_weighted_close_to_open_return'] + df['intraday_adjusted_return']

    # Adjust for Volume Volatility
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    df['volume_adjusted_intraday_momentum_vol_adj'] = df['volume_adjusted_intraday_momentum'] / df['volume_volatility']

    # Calculate Daily Log Return
    df['daily_log_return'] = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum()

    # Calculate VWAP and 20-Day Moving Average VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_20_day_ma'] = df['vwap'].rolling(window=20).mean()

    # Weighted Intraday Momentum
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['weighted_intraday_momentum'] = df['intraday_high_low_ratio'] * df['intraday_return']

    # Price Oscillator
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['5_day_sma'] - df['20_day_sma']

    # Combined and Enhanced Alpha Factor
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['combined_alpha_factor'] = (
        df['cumulative_log_return'] * df['vwap_20_day_ma'] +
        df['weighted_intraday_momentum'] +
        df['volume_change'] + df['amount_change'] + df['volume_volatility']
    ) * df['price_oscillator']

    return df['combined_alpha_factor']
