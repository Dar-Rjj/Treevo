```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['Intraday_High_Low'] = df['high'] - df['low']

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']

    # Weight by Volume and Amount Change
    df['Volume_Change'] = df['volume'].diff()
    df['Amount_Change'] = df['amount'].diff()

    # Combine Intraday Factors
    df['Combined_Intraday_Factor'] = (df['Intraday_Return'] + 
                                       df['Volume_Change'] +
                                       df['Amount_Change'] +
                                       df['Intraday_Volatility'])

    # Adjust Intraday Return
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * df['Combined_Intraday_Factor']

    # Calculate Volume Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Sum of Daily Price Changes over 20 days
    df['Sum_Daily_Price_Changes'] = df['Daily_Price_Change'].rolling(window=20).sum()

    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()

    # PMI Score
    df['PMI_Score'] = df['Sum_Daily_Price_Changes'] * df['VWAP']

    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))

    # Cumulative Sum of Daily Log Returns over 22 days
    df['Cumulative_Log_Returns'] = df['Daily_Log_Return'].rolling(window=22).sum()

    # Calculate 22-Day Momentum
    df['22Day_Momentum'] = (df['close'] - df['close'].shift(22)) / df['close'].shift(22)

    # Introduce New Factor: Intraday Range to Volume Ratio
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_to_Volume_Ratio'] = df['Intraday_Range'] / df['volume']

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['PMI_Score'] * df['Cumulative_Log_Returns'] +
                          df['Adjusted_Intraday_Return'] +
                          df['Volume_Weighted_Close_to_Open_Return'] +
                          df['22Day_Momentum'] +
                          df['Intraday_Range_to_Volume_Ratio'])

    return df['Alpha_Factor']
```
