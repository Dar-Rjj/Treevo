```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Combine Volume and Amount for weighting
    df['volume_amount_combined'] = df['volume'] * df['amount']
    
    # Weight close_to_open_return by volume and amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount_combined']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Intraday High-Low Movement by Volume Change
    df['adjusted_intraday_high_low_movement'] = df['intraday_high_low_movement'] * df['volume_change']
    
    # Calculate Intraday Open-Close Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Intraday High-Low Movement into the factor
    df['factor_with_intraday'] = df['weighted_close_to_open_return'] + df['adjusted_intraday_high_low_movement'] + df['intraday_return']
    
    # Calculate Volume Standard Deviation for 20 days
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_diff_from_ma'] = df['volume'] - df['volume_moving_average']
    df['volume_volatility'] = df['volume_diff_from_ma'].rolling(window=20).std().fillna(0)
    
    # Adjust Current Day's Factor for Volume Volatility
    df['final_factor'] = df['factor_with_intraday'] / (df['volume_volatility'] + 1e-6)  # Adding small constant to avoid division by zero
    
    return df['final_factor']
```
