```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between close and open prices
    factor1 = df['close'] - df['open']
    
    # Calculate percentage change from open to close
    factor2 = ((df['close'] - df['open']) / df['open']) * 100
    
    # Weight by Volume
    factor3 = (df['close'] - df['open']) * df['volume']
    
    # Measure intraday price range
    factor4 = df['high'] - df['low']
    
    # Assess upside potential
    factor5 = df['high'] - df['close']
    
    # Assess downside risk
    factor6 = df['open'] - df['low']
    
    # Intraday High-Low Ratio
    factor7 = (df['high'] - df['low']) / df['close']
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    factor8 = factor7 * df['volume_change']
    
    # Incorporate volume with price changes
    factor9 = (df['close'] - df['open']) * df['volume']
    
    # Calculate the ratio of volume to amount
    factor10 = df['volume'] / df['amount']
    
    # Compute the average trading price
    factor11 = df['amount'] / df['volume']
    
    # Examine the change in average trading price
    factor12 = factor11 - factor11.shift(1)
    
    # Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns over 20 days
    factor13 = df['log_return'].rolling(window=20).sum()
    
    # Daily Volume Weighted Average Price (VWAP)
    factor14 = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # 20-Day Moving Average VWAP
    factor15 = factor14.rolling(window=20).mean()
    
    # Multiply Cumulative Sum of Daily Log Returns by 20-Day Moving Average VWAP
    factor16 = factor13 * factor15
    
    # Weighted Intraday Momentum
    factor17 = (df['close'] - df['open']) * df['volume']
    
    # Volume Change + Amount Change
    factor18 = df['volume_change'] + (df['amount'] - df['amount'].shift(1))
    
    # Combined and Enhanced Alpha Factor
    factor19 = factor16 + (factor17 + factor18)
    
    # 5-Day Simple Moving Average (SMA) of Close price
    factor20 = df['close'].rolling(window=5).mean()
    
    # 20-Day Simple Moving Average (SMA) of Close price
    factor21 = df['close'].rolling(window=20).mean()
    
    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    factor22 = factor20 - factor21
    
    # Calculate Upward and Downward Price Movements
    df['up_movement'] = (df['close'] - df['close'].shift(1)) > 0
    df['down_movement'] = (df['close'] - df['close'].shift(1)) < 0
    
    # Calculate Average Up and Down Movements over 14 days
    avg_up = df['up_movement'].rolling(window=14).sum() * df['close'].diff().apply(lambda x: max(x, 0)).rolling(window=14).mean()
    avg_down = df['down_movement'].rolling(window=14).sum() * df['close'].diff().apply(lambda x: min(x, 0)).rolling(window=14).mean()
    
    # Calculate Relative Strength (RS)
    rs = avg_up / avg_down.abs()
    
    # Calculate RSI
    factor23 = 100 - (100 / (1 + rs))
    
    # Combine all factors into a single alpha factor
    alpha_factor = (factor1 + factor2 + factor3 + factor4 + factor5 + factor6 + factor8 + factor9 + 
                    factor10 + factor11 + factor12 + factor13 + factor14 + factor15 + factor16 + 
                    factor17 + factor18 + factor19 + factor20 + factor21 + factor22 + factor23)
    
    return alpha_factor
```
