```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 1-day momentum (today's close - yesterday's close)
    df['1_day_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate the 5-day momentum (today's close - close from 5 days ago)
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate the 5-day momentum change
    df['5_day_momentum_change'] = df['5_day_momentum'] - df['5_day_momentum'].shift(1)
    
    # Assign a score based on whether the 5-day momentum is increasing or decreasing
    df['momentum_score'] = df['5_day_momentum_change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate the volume-weighted momentum over the past 5 days
    df['volume_weighted_momentum'] = (df['close'] - df['close'].shift(5)) * df['volume']
    df['volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=5).sum()
    
    # Calculate the number of up days in the past 5 days
    df['up_day'] = (df['close'] > df['open']).astype(int)
    df['up_days_count'] = df['up_day'].rolling(window=5).sum()
    
    # Combine the momentum and up days count to create a sentiment-adjusted momentum factor
    df['sentiment_adjusted_momentum'] = df['1_day_momentum'] * df['up_days_count'] * df['momentum_score']
    
    # The final alpha factor
    alpha_factor = df['sentiment_adjusted_momentum']
    
    return alpha_factor
```
