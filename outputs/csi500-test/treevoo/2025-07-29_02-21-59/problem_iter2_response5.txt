```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    intraday_weighted_return = intraday_return * volume_change
    
    # Calculate Logarithmic Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns with 5-day SMA
    smoothed_log_returns = log_returns.rolling(window=5).mean()
    
    # Volume Shock Indicator
    std_volume_change = volume_change.rolling(window=5).std()
    volume_shock_indicator = (np.abs(volume_change) > 2 * std_volume_change).astype(int)
    
    # Adjusted Return based on Volume Shock
    shock_adjusted_return = smoothed_log_returns.where(volume_shock_indicator == 1, 0)
    
    # Combine Intraday Weighted Return and Shock Adjusted Return
    final_factor = intraday_weighted_return + shock_adjusted_return
    
    return final_factor
```
