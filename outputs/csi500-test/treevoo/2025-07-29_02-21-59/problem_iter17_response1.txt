```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume)
    
    Returns:
    pd.Series: A series of alpha factor values indexed by (date)
    """
    
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the price momentum
    price_momentum = df['close'].pct_change()
    
    # Calculate the volume momentum
    volume_momentum = df['volume'].pct_change()
    
    # Calculate the amount momentum
    amount_momentum = df['amount'].pct_change()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (relative_range + price_momentum + volume_momentum + amount_momentum) / 4
    
    return alpha_factor
```
