import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['Volume_Amount_Weight'] = (df['volume'] * df['amount'])
    df['Volume_Amount_Adjusted_Return'] = df['Close_to_Open_Return'] * df['Volume_Amount_Weight']
    
    # Intraday High-Low Movement with Volume Change
    df['Intraday_High_Low_Movement'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Change_Adjusted_Intraday'] = df['Intraday_High_Low_Movement'] * df['Volume_Change']
    
    # Adjust for Volume Volatility
    df['Volume_MA'] = df['volume'].rolling(window=20).mean()
    df['Volume_Diff_Squared'] = (df['volume'] - df['Volume_MA']) ** 2
    df['Volume_Volatility'] = (df['Volume_Diff_Squared'].rolling(window=20).sum()) ** 0.5
    
    # Integrate Previous Day's Close-to-Open and Intraday High-Low Movements
    df['Prev_Close_to_Open_Return'] = (df['open'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    df['Prev_Intraday_High_Low'] = (df['high'].shift(1) - df['low'].shift(1)) / df['close'].shift(2)
    df['Combined_Factor'] = df['Volume_Amount_Adjusted_Return'] + df['Prev_Close_to_Open_Return'] + df['Prev_Intraday_High_Low']
    
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Sum_Daily_Price_Changes'] = df['Daily_Price_Change'].rolling(window=20).sum()
    
    # Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['VWAP'] = df['VWAP'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # PMI Score
    df['PMI_Score'] = df['Sum_Daily_Price_Changes'] * df['VWAP']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Return
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * (df['Intraday_Volatility'] + df['Volume_Change'] + df['Amount_Change'])

    df['Alpha_Factor'] = df['PMI_Score'] * df['Cumulative_Log_Returns'] + df['Adjusted_Intraday_Return'] + df['Combined_Factor'] + df['Momentum_20_Day']
