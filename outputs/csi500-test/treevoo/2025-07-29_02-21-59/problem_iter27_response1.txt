import pandas as pd

def heuristics(df):
    # Volume-Adjusted Price Momentum
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    df['momentum_factor'] = df['volume_weighted_return'] + df['adjusted_intraday_return']
    
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_std_dev'] = df['volume'].rolling(window=20).std()
    df['volume_volatility'] = df['volume_std_dev'] / df['volume_moving_average']
    df['momentum_factor_adjusted'] = df['momentum_factor'] / df['volume_volatility']

    # Cumulative Log Return with VWAP Integration
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum()
    
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_20_day_ma'] = df['vwap'].rolling(window=20).mean()
    df['log_return_vwap'] = df['cumulative_log_return'] * df['vwap_20_day_ma']
    
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['weighted_log_return_vwap'] = df['log_return_vwap'] * (df['volume_change'] + df['amount_change'])

    # Combined and Enhanced Alpha Factor
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['5_day_sma'] - df['20_day_sma']
    
    df['upward_price_move'] = df['close'] - df['close'].shift(1)
    df['downward_price_move'] = df['close'].shift(1) - df['close']
    df['upward_move'] = df['upward_price_move'].apply(lambda x: max(x, 0))
    df['downward_move'] = df['downward_price_move'].apply(lambda x: max(x, 0))
    df['average_up'] = df['upward_move'].rolling(window=14).mean()
    df['average_down'] = df['downward_move'].rolling(window=14).mean()
    df['rs'] = df['average_up'] / df['average_down']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    df['alpha_factor'] = (
        df['weighted_log_return_vwap'] + 
        df['momentum_factor_adjusted'] * df['price_oscillator'] * df['rsi']
    )
    
    return df['alpha_factor']
