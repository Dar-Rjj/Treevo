import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_adjusted_return'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    df['total_volume_adjusted_return'] = df['volume_adjusted_return'] + df['volume_adjusted_intraday_return']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = ((df['volume'] - df['volume_moving_avg']).pow(2)).rolling(window=20).sum().apply(np.sqrt)
    df['adjusted_volume_alpha'] = df['total_volume_adjusted_return'] / df['volume_volatility']
    
    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum()
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['vwap_20_day_ma'] = df['vwap'].rolling(window=20).mean()
    df['weighted_intraday_momentum'] = df['intraday_high_low_ratio'] * df['intraday_return']
    
    # Combined Alpha Factor
    df['cumulative_vwap_product'] = df['cumulative_log_return'] * df['vwap_20_day_ma']
    df['combined_alpha_factor'] = df['cumulative_vwap_product'] + df['weighted_intraday_momentum']
    
    # Add volume and amount change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['combined_alpha_factor'] *= (df['volume_change'] + df['amount_change'] + df['intraday_high_low_ratio'])
    
    # Price Oscillator
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['5_day_sma'] - df['20_day_sma']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_alpha_factor'] + df['price_oscillator']
    
    return df['final_alpha_factor'].dropna()
