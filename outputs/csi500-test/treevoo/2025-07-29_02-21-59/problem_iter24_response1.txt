```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount'] = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_intraday_high_low_movement'] = df['intraday_high_low_movement'] + df['volume_change']
    
    # Calculate Volume Standard Deviation
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_std'] = ((df['volume'] - df['volume_ma'])**2).rolling(window=20).sum() / 20
    df['volume_std'] = np.sqrt(df['volume_std'])
    
    # Incorporate into Combined Alpha Factor
    df['combined_alpha_factor'] = df['weighted_close_to_open_return'] * (1 / df['volume_std']) + df['adjusted_intraday_high_low_movement']
    
    # Integrate Previous Day's Metrics
    df['prev_close_to_open_return'] = (df['open'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    df['prev_intraday_high_low_movement'] = (df['high'].shift(1) - df['low'].shift(1)) / df['close'].shift(2)
    df['prev_volume_std'] = ((df['volume'].shift(1) - df['volume'].shift(1).rolling(window=20).mean())**2).rolling(window=20).sum() / 20
    df['prev_volume_std'] = np.sqrt(df['prev_volume_std'])
    
    # Adjust Current Day's Factor
    df['final_alpha_factor'] = (df['weighted_close_to_open_return'] * (1 / df['volume_std']) + 
                                df['adjusted_intraday_high_low_movement'] +
                                df['prev_close_to_open_return'] * (1 / df['prev_volume_std']) +
                                df['prev_intraday_high_low_movement'])
    
    return df['final_alpha_factor'].dropna()
```
