```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using a combination of technical, volume-based, and price pattern indicators.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate Simple Moving Averages
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=30).mean()
    
    # Calculate SMA Crossover Signal
    sma_crossover = (sma_short > sma_long).astype(int)
    
    # Calculate Exponential Moving Average
    ema = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=12)
    
    # Calculate On-Balance Volume (OBV)
    obv = ((df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int)) * df['volume']
    obv = obv.cumsum()
    
    # Calculate Bollinger Bands
    bb_middle = df['close'].rolling(window=20).mean()
    bb_std = df['close'].rolling(window=20).std()
    bb_upper = bb_middle + 2 * bb_std
    bb_lower = bb_middle - 2 * bb_std
    
    # Calculate Donchian Channels
    upper_channel = df['high'].rolling(window=20).max()
    lower_channel = df['low'].rolling(window=20).min()
    middle_channel = (upper_channel + lower_channel) / 2
    
    # Calculate MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    # Calculate KDJ (Stochastic Oscillator)
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_line = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    d_line = k_line.rolling(window=3).mean()
    
    # Calculate Price-Volume Ratio
    price_volume_ratio = df['close'] / df['volume']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine Indicators
    combined_indicator = sma_crossover + (ema > df['close']).astype(int) + (roc > 0).astype(int) + (obv > obv.shift(1)).astype(int) + (df['close'] > bb_upper).astype(int) + (df['close'] < bb_lower).astype(int) + (df['close'] > upper_channel).astype(int) + (df['close'] < lower_channel).astype(int) + (macd_histogram > 0).astype(int) + (k_line > d_line).astype(int) + (price_volume_ratio > price_volume_ratio.shift(1)).astype(int) + (high_low_range > high_low_range.shift(1)).astype(int)
    
    return combined_indicator
```
