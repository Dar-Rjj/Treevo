```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between close and open prices
    factor1 = df['close'] - df['open']
    
    # Calculate percentage change from open to close
    factor2 = ((df['close'] - df['open']) / df['open']) * 100
    
    # Measure intraday price range
    factor3 = df['high'] - df['low']
    
    # Assess upside potential
    factor4 = df['high'] - df['close']
    
    # Assess downside risk
    factor5 = df['open'] - df['low']
    
    # Evaluate closing price momentum with volume
    factor6 = (df['close'] - df['open']) * df['volume']
    
    # Calculate the ratio of volume to amount
    factor7 = df['volume'] / df['amount']
    
    # Compute the average trading price
    avg_price = df['amount'] / df['volume']
    
    # Examine the change in average trading price
    avg_price_change = avg_price - avg_price.shift(1)
    
    # Combine all factors into a single DataFrame
    factors_df = pd.DataFrame({
        'factor1': factor1,
        'factor2': factor2,
        'factor3': factor3,
        'factor4': factor4,
        'factor5': factor5,
        'factor6': factor6,
        'factor7': factor7,
        'avg_price_change': avg_price_change
    })
    
    # Generate a combined alpha factor
    alpha_factor = factors_df.mean(axis=1)
    
    return alpha_factor
```
