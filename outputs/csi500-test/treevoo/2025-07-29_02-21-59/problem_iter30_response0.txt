import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Intraday Momentum
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_intraday_high_low_ratio'] = df['intraday_high_low_ratio'] * df['volume_change']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_std_dev'] = df['volume'].rolling(window=20).std()
    df['volume_adjustment_factor'] = df['volume_std_dev'] / df['volume_moving_avg']
    df['volume_adjusted_intraday_momentum'] = df['volume_weighted_close_to_open'] / df['volume_adjustment_factor']

    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x))
    df['cumulative_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    
    df['intraday_return'] = (df['close'] / df['open']) - 1
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_20_day_ema'] = df['vwap'].ewm(span=20, adjust=False).mean()
    
    # Combined Alpha Factor
    df['alpha_factor'] = (df['cumulative_log_returns'] * df['vwap_20_day_ema'] + 
                          df['volume_adjusted_intraday_momentum'] + 
                          df['weighted_intraday_return'])
    
    df['final_alpha_factor'] = df['alpha_factor'].ewm(span=20, adjust=False).mean().fillna(0)
    
    return df['final_alpha_factor']
