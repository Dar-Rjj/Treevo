```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Volume-Adjusted Momentum
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Diff'] = df['EMA_20'] - df['SMA_20']
    df['Volume_Adjusted_Momentum'] = df['Momentum_Diff'] * df['volume']
    df['Smoothed_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'].rolling(window=5).mean()

    # Compute Price-Volume Synchronization Measure
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    df['Prev_Day_Close_Open_Change'] = df['close'].shift(1) - df['open'].shift(1)
    df['Price_Volume_Synchronization'] = (df['volume'] * df['Intraday_High_Low_Range']) / df['Prev_Day_Close_Open_Change'].abs()

    # Combine Volume-Adjusted Momentum and Price-Volume Synchronization
    df['Combined_Factor'] = df['Smoothed_Volume_Adjusted_Momentum'] + df['Price_Volume_Synchronization']

    # Calculate Volume Acceleration
    df['Volume_Difference'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Acceleration_Sign'] = np.sign(df['Volume_Difference'])

    # Identify Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Price_Momentum_Sign'] = np.sign(df['Price_Change'])

    # Confirm Trend
    df['Positive_Trend'] = (df['Smoothed_Volume_Adjusted_Momentum'] > 0) & (df['Volume_Acceleration_Sign'] > 0) & (df['Price_Momentum_Sign'] > 0)
    df['Negative_Trend'] = (df['Smoothed_Volume_Adjusted_Momentum'] < 0) & (df['Volume_Acceleration_Sign'] < 0) & (df['Price_Momentum_Sign'] < 0)

    # Generate the alpha factor
    df['Alpha_Factor'] = df['Combined_Factor'].where(df['Positive_Trend'], -df['Combined_Factor'].where(df['Negative_Trend'], 0))

    return df['Alpha_Factor']
```
