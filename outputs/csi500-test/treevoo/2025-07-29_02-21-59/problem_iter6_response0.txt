```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_ratio = intraday_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    weighted_intraday_return = intraday_return * volume_change
    
    # Logarithmic Returns with Volume Shock
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns
    smoothed_log_returns = log_returns.rolling(window=5).mean()
    
    # Volume Shock Indicator
    volume_shock = np.abs(volume_change) > 2 * volume_change.rolling(window=5).mean()
    shock_adjusted_return = smoothed_log_returns.where(volume_shock, 0)
    
    # Combine Intraday Weighted Return and Shock Adjusted Return
    final_factor = weighted_intraday_return + shock_adjusted_return
    
    return final_factor
```
