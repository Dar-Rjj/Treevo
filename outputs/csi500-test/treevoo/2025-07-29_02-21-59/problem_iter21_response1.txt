```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount'] = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_movement'] = df['high'] - df['low']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Intraday High-Low Movement by Volume Change
    df['adjusted_intraday_high_low_movement'] = df['intraday_high_low_movement'] + df['volume_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Add to Volume and Amount Adjusted Close-to-Open Return
    df['combined_factor'] = df['weighted_close_to_open_return'] + df['intraday_return'] + df['adjusted_intraday_high_low_movement']
    
    # Calculate Volume Standard Deviation
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_diff'] = df['volume'] - df['volume_moving_avg']
    df['volume_diff_squared'] = df['volume_diff'] ** 2
    df['volume_std'] = df['volume_diff_squared'].rolling(window=20).sum().apply(lambda x: x ** 0.5)
    
    # Adjust for Volume Volatility
    df['volume_volatility_adjusted_factor'] = df['combined_factor'] / df['volume_std']
    
    # Integrate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Adjust Current Day's Factor
    df['integrated_close_to_open_return'] = df['weighted_close_to_open_return'] + df['prev_close_to_open_return']
    
    # Integrate Previous Day's Intraday High-Low Movement
    df['prev_intraday_high_low_movement'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Adjust Current Day's Factor
    df['integrated_intraday_high_low_movement'] = df['intraday_high_low_movement'] + df['prev_intraday_high_low_movement']
    
    # Integrate Previous Day's Volume Volatility
    df['prev_volume_volatility'] = df['volume_std'].shift(1)
    
    # Adjust Current Day's Factor
    df['final_factor'] = (df['integrated_close_to_open_return'] + 
                          df['integrated_intraday_high_low_movement']) / (df['volume_std'] + df['prev_volume_volatility'])
    
    return df['final_factor']
```
