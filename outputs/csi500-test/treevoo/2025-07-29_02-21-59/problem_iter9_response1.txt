import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Daily Close-to-Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Sum of Recent Returns
    recent_returns_sum = close_to_close_return.rolling(window=5).sum()

    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)

    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()

    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume

    # Calculate Amount Change
    amount_change = df['amount'] - df['amount'].shift(1)

    # Calculate Average Amount over Period
    average_amount = df['amount'].rolling(window=5).mean()

    # Ratio of Current Day's Amount to Average Amount
    amount_ratio = df['amount'] / average_amount

    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change

    # Weighted Amount Intraday Return
    weighted_amount_intraday_return = intraday_return * amount_change

    # Combined Weighted Momentum
    combined_weighted_momentum = (weighted_intraday_return + weighted_amount_intraday_return + recent_returns_sum) * volume_ratio * amount_ratio

    # Calculate Intraday High-Close Movement
    intraday_high_close = df['high'] - df['close']

    # Calculate Intraday Low-Close Movement
    intraday_low_close = df['low'] - df['close']

    # Adjusted Intraday Return
    adjusted_intraday_return = (df['close'] - df['open']) / df['open'] * (intraday_high_close + intraday_low_close)

    # Adjusted Weighted Intraday Return
    adjusted_weighted_intraday_return = adjusted_intraday_return * volume_change

    # Adjusted Weighted Amount Intraday Return
    adjusted_weighted_amount_intraday_return = adjusted_intraday_return * amount_change

    # Enhanced Combined Weighted Momentum
    enhanced_combined_weighted_momentum = (adjusted_weighted_intraday_return + adjusted_weighted_amount_intraday_return + recent_returns_sum) * volume_ratio * amount_ratio

    return enhanced_combined_weighted_momentum
