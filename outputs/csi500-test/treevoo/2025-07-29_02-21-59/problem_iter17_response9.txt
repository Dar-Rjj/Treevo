```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['Volume_Amount_Weight'] = (df['volume'] * df['amount']) / (df['volume'].mean() * df['amount'].mean())
    df['Volume_Adjusted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['Volume_Amount_Weight']
    
    # Integrate Intraday High-Low Movement
    df['Intraday_High_Low_Movement'] = (df['high'] - df['low']) / df['close']
    df['Intraday_High_Low_Ratio'] = (df['high'] - df['low']) / df['close']
    df['Adjusted_Intraday_Return'] = (df['close'] - df['open']) / df['open'] * df['Intraday_High_Low_Ratio']
    
    # Adjust by Volume and Amount Change
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1) - 1
    df['Amount_Change'] = df['amount'] / df['amount'].shift(1) - 1
    df['Adjusted_Volume_Change_Intraday_Return'] = df['Adjusted_Intraday_Return'] * df['Volume_Change']
    df['Adjusted_Amount_Change_Intraday_Return'] = df['Adjusted_Intraday_Return'] * df['Amount_Change']
    
    # Calculate Sum of Recent Returns
    df['Close_to_Close_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Sum_of_Recent_Returns'] = df['Close_to_Close_Return'].rolling(window=5).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['volume']
    
    # Calculate 20-Day Moving Average VWAP
    df['20_Day_MA_VWAP'] = df['VWAP'].rolling(window=20).mean()
    
    # Combined Weighted Momentum
    df['Combined_Weighted_Momentum'] = (df['Adjusted_Volume_Change_Intraday_Return'] +
                                        df['Adjusted_Amount_Change_Intraday_Return'] +
                                        df['Sum_of_Recent_Returns'])
    
    # Adjust for Volume Volatility
    df['Volume_Moving_Average'] = df['volume'].rolling(window=20).mean()
    df['Volume_Volatility'] = df['volume'].rolling(window=20).std()
    df['Volume_Ratio'] = (df['volume'] - df['Volume_Moving_Average']) / df['Volume_Volatility']
    
    # Enhanced Combined Weighted Momentum
    df['Enhanced_Combined_Weighted_Momentum'] = (df['Combined_Weighted_Momentum'] *
                                                 df['Volume_Change'] * df['Amount_Change'] *
                                                 df['Intraday_High_Low_Ratio'])
    
    # PMI Score
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    df['Sum_Daily_Price_Changes'] = df['Daily_Price_Change'].rolling(window=20).sum()
    df['PMI_Score'] = df['Sum_Daily_Price_Changes'] * df['VWAP']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Enhanced_Combined_Weighted_Momentum'] * df['20_Day_MA_VWAP'] +
                                df['Volume_Adjusted_Close_to_Open_Return'] +
                                df['Adjusted_Intraday_Return'] + df['PMI_Score'])
    
    return df['Final_Alpha_Factor']
```
