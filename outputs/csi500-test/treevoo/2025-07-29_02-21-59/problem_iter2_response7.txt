```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    lookback_window = 20
    sma = df['close'].rolling(window=lookback_window).mean()
    
    # Calculate Exponential Moving Average (EMA) of Close Prices
    ema = df['close'].ewm(span=lookback_window, adjust=False).mean()
    
    # Compute the Difference between EMA and SMA
    momentum_difference = ema - sma
    
    # Volume Adjust the Momentum
    volume_adjusted_momentum = momentum_difference * df['volume']
    
    # Smooth the adjusted momentum over a shorter period (e.g., 5 days)
    smoothed_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=5).mean()
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - df['volume'].shift(1)
    sign_volume_acceleration = volume_acceleration.apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Identify Price Momentum
    close_price_change = df['close'] - df['close'].shift(1)
    sign_price_momentum = close_price_change.apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Combine Volume-Adjusted Momentum and Volume-Price Confirmation
    combined_factor = pd.Series(index=df.index, data=0.0)
    combined_factor[(smoothed_volume_adjusted_momentum > 0) & (sign_volume_acceleration > 0) & (sign_price_momentum > 0)] = 1
    combined_factor[(smoothed_volume_adjusted_momentum < 0) & (sign_volume_acceleration < 0) & (sign_price_momentum < 0)] = -1
    
    return combined_factor
```
