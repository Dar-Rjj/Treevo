def heuristics_v2(data):
    # Calculate the daily return
    data['daily_return'] = data['close'].pct_change()

    # Calculate the average price for the day
    data['average_price'] = (data['high'] + data['low'] + data['close']) / 3

    # Calculate the money flow by multiplying the average price with the volume
    data['money_flow'] = data['average_price'] * data['volume']

    # Calculate the 14-day Money Flow Index (MFI)
    window = 14
    data['positive_money_flow'] = data['money_flow'].where(data['daily_return'] > 0, 0)
    data['negative_money_flow'] = data['money_flow'].where(data['daily_return'] < 0, 0)
    data['positive_money_flow_sum'] = data['positive_money_flow'].rolling(window=window).sum()
    data['negative_money_flow_sum'] = data['negative_money_flow'].rolling(window=window).sum()
    data['money_ratio'] = data['positive_money_flow_sum'] / data['negative_money_flow_sum']
    data['mfi'] = 100 - (100 / (1 + data['money_ratio']))

    # Calculate the 14-day Relative Strength Index (RSI)
    data['gain'] = data['daily_return'].apply(lambda x: x if x > 0 else 0)
    data['loss'] = data['daily_return'].apply(lambda x: -x if x < 0 else 0)
    data['average_gain'] = data['gain'].rolling(window=window).mean()
    data['average_loss'] = data['loss'].rolling(window=window).mean()
    data['rs'] = data['average_gain'] / data['average_loss']
    data['rsi'] = 100 - (100 / (1 + data['rs']))

    # Alpha factor: MFI - RSI
    alpha_factor = data['mfi'] - data['rsi']
    
    return alpha_factor.dropna()
