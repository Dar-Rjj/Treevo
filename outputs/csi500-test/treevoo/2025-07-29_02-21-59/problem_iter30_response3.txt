import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Amount Adjusted Intraday Momentum
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    volume_amount_weight = df['volume'] * df['amount']
    volume_adjusted_intraday_momentum = close_to_open_return * volume_amount_weight
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    adjusted_intraday_high_low = intraday_high_low_ratio * (volume_change + amount_change)
    
    # Adjust for Volume Volatility
    volume_ma = df['volume'].rolling(window=220).sum() / 20
    volume_volatility = (df['volume'] - volume_ma).pow(2).rolling(window=20).sum().pow(0.5)
    volume_volatility_adjusted = volume_adjusted_intraday_momentum / volume_volatility
    
    # Cumulative Momentum with VWAP Integration
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    cumulative_log_return = daily_log_return.rolling(window=20).sum()
    
    intraday_high_low_movement = (df['high'] - df['low']) / df['close']
    intraday_weighted_return = (df['close'] / df['open'] - 1) * (volume_change + amount_change)
    
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap_ema_20 = vwap.ewm(span=20, adjust=False).mean()
    
    # Combined Alpha Factor
    combined_factor = cumulative_log_return * vwap_ema_20
    combined_factor += volume_volatility_adjusted
    combined_factor += intraday_weighted_return
    
    # Include Volatility Component
    daily_volatility = df['close'].pct_change().rolling(window=20).std()
    volatility_component = daily_volatility * volume_adjusted_intraday_momentum
    combined_factor += volatility_component
    
    # Calculate Sum of Recent Returns
    recent_returns_sum = df['close'].pct_change().rolling(window=5).sum()
    recent_returns_adjusted = recent_returns_sum * volume_change * amount_change * intraday_high_low_ratio
    
    # Incorporate Intraday Range Adjustments
    intraday_range = df['high'] - df['low']
    intraday_range_adjusted = intraday_range * (df['volume'] + df['amount'])
    
    final_alpha_factor = combined_factor + recent_returns_adjusted + intraday_range_adjusted
    
    return final_alpha_factor.dropna()
