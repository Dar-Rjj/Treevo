def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['high_low_movement'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    df['high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Amount Change
    df['amount_change'] = df['amount'].pct_change()
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['intraday_return'] * df['volume_change'] * df['amount_change'] / df['intraday_volatility']
    
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of Recent Returns
    df['sum_recent_returns'] = df['daily_return'].rolling(window=5).sum()
    
    # Average Volume over Period
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Current Day's Volume to Average Volume Ratio
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Weighted Relative Momentum
    df['weighted_relative_momentum'] = df['sum_recent_returns'] * df['volume_ratio']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combined Volume-Adjusted Momentum
    df['combined_volume_adjusted_momentum'] = df['weighted_relative_momentum'] * df['close_to_open_return']

    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = ((df['volume'] - df['volume_moving_avg']).pow(2)).rolling(window=20).sum().pow(0.5)
    
    # Final Adjusted Momentum Score
    df['final_adjusted_momentum'] = df['combined_volume_adjusted_momentum'] / df['volume_volatility']
    
    # Adjust Intraday Return by Volume Change
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']

    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMI Score
    df['pmi_score'] = (df['close'] - df['close'].shift(1)).cumsum() * df['vwap']
    
    # Volume Weighted Close-to-Open Return
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
