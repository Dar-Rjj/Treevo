def heuristics_v2(df):
    # Calculate the weighted average of close and high prices using volume as weights
    weighted_avg = (df['close'] * df['volume'] + df['high'] * df['volume']) / (2 * df['volume'])
    
    # Determine the difference between the amount and the product of close and volume
    trading_imbalance = df['amount'] - (df['close'] * df['volume'])
    
    # Generate a factor by combining the weighted average with trading imbalance
    factor = weighted_avg + trading_imbalance
    
    return factor
