def heuristics_v2(df):
    # Calculate the price change from the previous day to the current day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average of high and low for the current day
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the money flow by multiplying the average price with the volume
    df['money_flow'] = df['avg_price'] * df['volume']
    
    # Calculate the money flow multiplier, which is the price change divided by the absolute price change
    df['mf_multiplier'] = df['price_change'] / df['price_change'].abs()
    
    # Calculate the money flow index, which is the money flow multiplied by the money flow multiplier
    df['mf_index'] = df['money_flow'] * df['mf_multiplier']
    
    # Calculate the 14-day sum of positive and negative money flow
    df['positive_mf_sum'] = df[df['mf_index'] > 0]['mf_index'].rolling(window=14).sum()
    df['negative_mf_sum'] = df[df['mf_index'] < 0]['mf_index'].abs().rolling(window=14).sum()
    
    # Calculate the Money Flow Ratio
    df['mf_ratio'] = df['positive_mf_sum'] / df['negative_mf_sum']
    
    # Calculate the Money Flow Index (MFI) as a factor
    df['mfi_factor'] = 100 - (100 / (1 + df['mf_ratio']))
    
    return df['mfi_factor']
