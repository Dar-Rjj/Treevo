import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # SMA Crossover
    df['SMA_Crossover'] = (df['SMA_5'] > df['SMA_10']).astype(int) - (df['SMA_5'] < df['SMA_10']).astype(int)
    
    # SMA Slope
    df['SMA_Slope'] = df['SMA_10'].diff()
    
    # Calculate Exponential Moving Averages
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # EMA Crossover
    df['EMA_Crossover'] = (df['EMA_5'] > df['EMA_10']).astype(int) - (df['EMA_5'] < df['EMA_10']).astype(int)
    
    # EMA Slope
    df['EMA_Slope'] = df['EMA_10'].diff()
    
    # Rate of Change (ROC)
    df['ROC'] = df['close'].pct_change(periods=1)
    
    # ROC Slope
    df['ROC_Slope'] = df['ROC'].diff()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # RSI Slope
    df['RSI_Slope'] = df['RSI'].diff()
    
    # On-Balance Volume (OBV)
    df['OBV'] = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0)) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # OBV Crossover with its moving average
    df['OBV_MA'] = df['OBV'].rolling(window=10).mean()
    df['OBV_Crossover'] = (df['OBV'] > df['OBV_MA']).astype(int) - (df['OBV'] < df['OBV_MA']).astype(int)
    
    # Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # CMF Crossover
    df['CMF_Crossover'] = (df['CMF'] > 0).astype(int) - (df['CMF'] < 0).astype(int)
    
    # Bollinger Bands
    df['BB_Middle'] = df['close'].rolling(window=20).mean()
    df['BB_STD'] = df['close'].rolling(window=20).std()
    df['BB_Upper'] = df['BB_Middle'] + 2 * df['BB_STD']
    df['BB_Lower'] = df['BB_Middle'] - 2 * df['BB_STD']
    df['BB_Bandwidth'] = df['BB_Upper'] - df['BB_Lower']
    df['BB_%B'] = (df['close'] - df['BB_Lower']) / (df['BB_Upper'] - df['BB_Lower'])
    
    # Donchian Channels
    df['Upper_Channel'] = df['high'].rolling(window=20).max()
    df['Lower_Channel'] = df['low'].rolling(window=20).min()
    df['Middle_Channel'] = (df['Upper_Channel'] + df['Lower_Channel']) / 2
    df['Donchian_Breakout'] = (df['close'] > df['Upper_Channel']).astype(int) - (df['close'] < df['Lower_Channel']).astype(int)
