```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['Volume_Amount_Combined'] = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    df['Volume_Amount_Adjusted_Return'] = df['Close_to_Open_Return'] * df['Volume_Amount_Combined']
    
    # Calculate Intraday High-Low Movement
    df['Intraday_High_Low_Movement'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    df['Intraday_High_Low_Ratio'] = df['Intraday_High_Low_Movement'] / df['close']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume Change
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * df['Volume_Change']
    
    # Add to Volume and Amount Adjusted Close-to-Open Return
    df['Combined_Return'] = df['Volume_Amount_Adjusted_Return'] + df['Adjusted_Intraday_Return']
    
    # Calculate Logarithmic Returns
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns using EMA with span of 5 days
    df['Smoothed_Log_Returns'] = df['Log_Returns'].ewm(span=5).mean()
    
    # Calculate Price Volatility
    df['High_Low_Diff'] = (df['high'] - df['low'] + df['high'].shift(1) - df['low'].shift(1)) / 2
    df['Price_Volatility'] = df['High_Low_Diff'].rolling(window=20).mean()
    
    # Combine Volume and Volatility
    df['Volume_Volatility_Combined'] = df['volume'] * df['Price_Volatility']
    
    # Multiply Smoothed Returns by Combined Weight
    df['Weighted_Smoothed_Log_Returns'] = df['Smoothed_Log_Returns'] * df['Volume_Volatility_Combined']
    
    # Calculate Volume Moving Average
    df['Volume_MA'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Standard Deviation of Volume
    df['Volume_Std'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['Volume_Volatility_Adjusted_Return'] = df['Combined_Return'] / df['Volume_Std']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Weighted_Smoothed_Log_Returns'] + df['Volume_Volatility_Adjusted_Return']
    
    return df['Alpha_Factor']
```
