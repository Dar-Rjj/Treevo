```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount_weight'] = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount_weight']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Intraday High-Low Movement by Volume Change
    df['adjusted_intraday_high_low_movement'] = df['intraday_high_low_movement'] * df['volume_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine with Intraday Return
    df['combined_intraday_movement'] = df['intraday_return'] + df['adjusted_intraday_high_low_movement']
    
    # Calculate Volume Standard Deviation
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = (df['volume'] - df['volume_moving_average'])
    df['squared_volume_deviation'] = df['volume_deviation'] ** 2
    df['sum_squared_volume_deviation'] = df['squared_volume_deviation'].rolling(window=20).sum()
    df['volume_standard_deviation'] = (df['sum_squared_volume_deviation'] / 20) ** 0.5
    
    # Adjust for Volume Volatility
    df['volume_adjusted_factor'] = df['weighted_close_to_open_return'] / df['volume_standard_deviation']
    
    # Integrate Previous Day's Close-to-Open Return
    df['prev_day_close_to_open_return'] = (df['open'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Adjust Current Day's Factor
    df['integrated_factor'] = df['volume_adjusted_factor'] + df['combined_intraday_movement'] + df['prev_day_close_to_open_return']
    
    return df['integrated_factor']
```
