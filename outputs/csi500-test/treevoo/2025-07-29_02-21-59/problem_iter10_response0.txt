```python
def heuristics_v2(df):
    # Calculate Intraday Movement
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Sum of Recent Returns (last 5 days)
    sum_recent_returns = daily_return.rolling(window=5).sum().shift()
    
    # Volume Change
    volume_change = df['volume'].pct_change()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Volume Ratio
    volume_ratio = df['volume'] / average_volume
    
    # Amount Change
    amount_change = df['amount'].pct_change()
    
    # Average Amount
    average_amount = df['amount'].rolling(window=5).mean()
    
    # Amount Ratio
    amount_ratio = df['amount'] / average_amount
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Weighted Amount Intraday Return
    weighted_amount_intraday_return = intraday_return * amount_change
    
    # Combined Weighted Momentum
    combined_weighted_momentum = (weighted_intraday_return + weighted_amount_intraday_return + sum_recent_returns) * volume_ratio * amount_ratio
    
    return combined_weighted_momentum
```
