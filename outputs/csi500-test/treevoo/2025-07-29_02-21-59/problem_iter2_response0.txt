```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day Close-to-Open Change
    previous_day_close_to_open_change = df['open'].shift(-1) - df['close']
    
    # Compute Volume-Weighted Breakout Measure
    volume_weighted_breakout_measure = (df['volume'] * intraday_high_low_spread) / previous_day_close_to_open_change
    
    # Smooth the Factor Over a Short Window, let's assume N=5 for a short window
    smoothed_factor = volume_weighted_breakout_measure.rolling(window=5).mean()
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Weight by Volume and Combine
    volume_weighted_combined_momentum = smoothed_factor + (df['volume'] * close_to_open_return)
    
    return volume_weighted_combined_momentum
```
