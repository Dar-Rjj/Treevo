import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_return_adjusted'] = (df['close'] - df['open']) / df['open'] * df['volume_change']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg_20'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = (df['close'] - df['close'].shift(1)).pct_change().fillna(0)
    df['cumulative_log_return_20'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Ratio
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume Change
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) / df['open'] * df['volume_change']
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    
    # Calculate 20-Day Exponential Moving Average VWAP
    df['vwap_ema_20'] = df['vwap'].ewm(span=20, adjust=False).mean()
    
    # Combined Alpha Factor
    df['cumulative_log_returns_ema_vwap'] = df['cumulative_log_return_20'] * df['vwap_ema_20']
    df['weighted_intraday_momentum'] = df['volume_weighted_return'] + df['intraday_return_adjusted']
    
    # Include Recent Trading Activity
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_spike_indicator'] = (df['volume_ema_5'] > df['volume_moving_avg_20']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['cumulative_log_returns_ema_vwap'] + df['weighted_intraday_momentum'] + df['volume_spike_indicator']
    
    return df['alpha_factor']
