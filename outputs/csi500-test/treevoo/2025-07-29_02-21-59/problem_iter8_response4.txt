```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Combine Volume and Amount
    df['volume_amount'] = (df['volume'] * df['amount']) ** 0.5  # Geometric mean for combining volume and amount
    
    # Weight by Volume and Amount
    df['volume_amount_weighted_co_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Calculate Intraday High-Low Movement
    df['intraday_hl_movement'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    df['intraday_hl_ratio'] = df['intraday_hl_movement'] / df['close']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Adjust Intraday Return by Volume Change
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Integrate with Volume and Amount Adjusted Close-to-Open Return
    df['integrated_factor'] = df['volume_amount_weighted_co_return'] + df['adjusted_intraday_return'] * df['intraday_hl_ratio']
    
    # Calculate Volume Moving Average
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Standard Deviation of Volume
    df['volume_std_dev'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['final_factor'] = df['integrated_factor'] / df['volume_std_dev']
    
    return df['final_factor'].dropna()
```
