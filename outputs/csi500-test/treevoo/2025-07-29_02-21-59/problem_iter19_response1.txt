import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    volume_amount_combined = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = close_to_open_return * volume_amount_combined
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = (df['high'] - df['low']) / df['close']
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_return = intraday_return * volume_change
    
    # Add to Volume and Amount Weighted Close-to-Open Return
    combined_return = weighted_close_to_open_return + adjusted_intraday_return
    
    # Include Volatility Component
    daily_volatility = df['close'].rolling(window=20).std()
    final_factor = combined_return * daily_volatility
    
    return final_factor
