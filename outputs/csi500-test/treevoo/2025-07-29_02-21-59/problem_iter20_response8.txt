```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    volume_change = df['volume'].pct_change()
    amount_change = df['amount'].pct_change()
    
    # Combine Intraday Factors
    combined_intraday_factors = intraday_return + volume_change + amount_change + intraday_volatility
    
    # Adjust Intraday Return
    adjusted_intraday_return = intraday_return * combined_intraday_factors
    
    # Calculate Volume Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Sum of Daily Price Changes over 20 days
    sum_daily_price_changes = daily_price_change.rolling(window=20).sum()
    
    # Calculate VWAP
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMI Score
    pmi_score = sum_daily_price_changes * vwap
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns over 20 days
    cumulative_sum_log_returns = daily_log_return.rolling(window=20).sum()
    
    # Calculate 20-Day Momentum
    momentum_20_day = df['close'] / df['close'].shift(20) - 1
    
    # Final Alpha Factor
    final_alpha_factor = (pmi_score * cumulative_sum_log_returns) + adjusted_intraday_return + volume_weighted_close_to_open_return + momentum_20_day
    
    return final_alpha_factor
```
