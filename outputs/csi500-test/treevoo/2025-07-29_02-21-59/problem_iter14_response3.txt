def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: Series with index as date representing the factor values
    """
    # Calculate the high-low price difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate the close-open price difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average transaction size (amount/volume)
    avg_transaction_size = df['amount'] / df['volume']
    
    # Construct the alpha factor
    alpha_factor = (close_open_diff / high_low_diff) * avg_transaction_size
    
    return alpha_factor
