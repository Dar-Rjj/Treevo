```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)

    # Sum of Daily Price Changes over 20 days
    df['Sum_Daily_Price_Change_20d'] = df['Daily_Price_Change'].rolling(window=20).sum()

    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # PMI Score
    df['PMI_Score'] = df['Sum_Daily_Price_Change_20d'] * df['VWAP']

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']

    # Adjust Intraday Return
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * (df['Intraday_Volatility'] + df['Volume_Change'] + df['Amount_Change'])

    # Calculate Volume Weighted Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))

    # Cumulative Sum of Daily Log Returns over 20 days
    df['Cumulative_Sum_Daily_Log_Return_20d'] = df['Daily_Log_Return'].rolling(window=20).sum()

    # Calculate 20-Day Momentum
    df['20_Day_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['PMI_Score'] * df['Cumulative_Sum_Daily_Log_Return_20d']) + \
                                df['Adjusted_Intraday_Return'] + \
                                df['Volume_Weighted_Close_to_Open_Return'] + \
                                df['20_Day_Momentum']
    
    return df['Final_Alpha_Factor']
```
