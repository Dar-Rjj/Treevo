import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume and Amount Weighted Close-to-Open Return.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                          indexed by date.
                          
    Returns:
        pd.Series: A Series of the calculated alpha factor, indexed by date.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount to create a weighting factor
    weight = df['volume'] * df['amount']
    
    # Weight the return by the combined volume and amount
    weighted_return = close_to_open_return * weight
    
    return weighted_return
