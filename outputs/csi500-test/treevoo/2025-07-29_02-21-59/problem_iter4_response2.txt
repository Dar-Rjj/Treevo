```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of daily returns over 20 days
    df['sum_daily_returns'] = df['daily_return'].rolling(window=20).sum()
    
    # Volume moving average over 20 days
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    
    # Subtract moving average from daily volumes
    df['volume_diff'] = df['volume'] - df['volume_moving_avg']
    
    # Square the differences
    df['squared_vol_diff'] = df['volume_diff'] ** 2
    
    # Sum of squared differences over 20 days
    df['sum_squared_vol_diff'] = df['squared_vol_diff'].rolling(window=20).sum()
    
    # Standard deviation of volume
    df['volatility_of_volume'] = np.sqrt(df['sum_squared_vol_diff'])
    
    # Adjust momentum for volume volatility
    df['volume_adjusted_momentum'] = df['sum_daily_returns'] / df['volatility_of_volume']
    
    return df['volume_adjusted_momentum'].dropna()
```
