```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    weighted_intraday_return = intraday_return * volume_change
    
    # Calculate Logarithmic Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns with 5-day SMA
    smoothed_log_returns = log_returns.rolling(window=5).mean()
    
    # Volume Shock Indicator
    std_volume_change = volume_change.rolling(window=5).std()
    volume_shock = (np.abs(volume_change) > 2 * std_volume_change).astype(int)
    
    # Adjusted Logarithmic Returns with Enhanced Volume Shock
    adjusted_smoothed_return = smoothed_log_returns * (intraday_high_low / df['close'])
    final_return = (volume_shock * adjusted_smoothed_return) + ((1 - volume_shock) * weighted_intraday_return)
    
    return final_return
```
