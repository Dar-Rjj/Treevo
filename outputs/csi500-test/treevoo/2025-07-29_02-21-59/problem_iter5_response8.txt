```python
def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'].pct_change()
    
    # Sum of Recent Returns (last 5 days)
    df['Sum_of_Recent_Returns'] = df['Close_to_Close_Return'].rolling(window=5).sum()
    
    # Calculate Average Volume over the last 5 days
    df['Average_Volume'] = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Weighted Relative Momentum
    df['Weighted_Relative_Momentum'] = df['Sum_of_Recent_Returns'] * df['Volume_Ratio']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combined Volume-Adjusted Momentum
    df['Combined_Adjusted_Momentum'] = df['Weighted_Relative_Momentum'] * df['Close_to_Open_Return']
    
    # Calculate Volume Moving Average over 20 days
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Standard Deviation of Volume
    df['Volatility'] = df['volume'].rolling(window=20).std()
    
    # Final Adjusted Momentum Score
    df['Final_Adjusted_Momentum_Score'] = df['Combined_Adjusted_Momentum'] / df['Volatility']
    
    return df['Final_Adjusted_Momentum_Score']
```
