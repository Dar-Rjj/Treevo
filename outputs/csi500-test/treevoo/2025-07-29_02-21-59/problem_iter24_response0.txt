```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Sum of Recent Returns
    recent_returns_sum = df['close'].pct_change().rolling(window=5).sum()
    
    # Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Amount Change
    amount_change = df['amount'] / df['amount'].shift(1) - 1
    
    # Average Volume and Amount
    avg_volume = df['volume'].rolling(window=5).mean()
    avg_amount = df['amount'].rolling(window=5).mean()
    
    # Volume and Amount Ratios
    volume_ratio = df['volume'] / avg_volume
    amount_ratio = df['amount'] / avg_amount
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    weighted_amount_intraday_return = intraday_return * amount_change
    
    # Combined Weighted Momentum
    combined_weighted_momentum = (weighted_intraday_return + weighted_amount_intraday_return + 
                                  recent_returns_sum) * volume_ratio * amount_ratio
    
    # Adjusted Logarithmic Returns with Enhanced Volume Shock
    log_returns = np.log(df['close'] / df['close'].shift(1))
    smoothed_log_returns = log_returns.rolling(window=5).mean()
    volume_shock = (df['volume'] - df['volume'].shift(1)).abs() > 2 * df['volume'].rolling(window=5).std()
    
    adjusted_returns = np.where(volume_shock, smoothed_log_returns * (df['high'] - df['low']) / df['close'], 
                                combined_weighted_momentum)
    
    # Volatility-Adjusted Returns
    realized_volatility = df['close'].rolling(window=5).std()
    vol_adjusted_returns = smoothed_log_returns / realized_volatility
    
    # Final Factor
    final_factor = adjusted_returns * vol_adjusted_returns
    
    return final_factor
```
