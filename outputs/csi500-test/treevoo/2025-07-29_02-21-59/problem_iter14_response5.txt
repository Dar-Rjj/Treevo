```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Combine Volume and Amount
    df['Volume_Amount_Combined'] = df['volume'] + df['amount']

    # Multiply by Close-to-Open Return
    df['Volume_Momentum_Adjusted'] = df['Volume_Amount_Combined'] * df['Close_to_Open_Return']

    # Integrate Intraday High-Low Movement
    df['Intraday_High_Low_Movement'] = df['high'] - df['low']
    df['Intraday_High_Low_Ratio'] = df['Intraday_High_Low_Movement'] / df['close']
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Volume Change and Amount Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)

    # Adjust by Volume and Amount Change
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * (df['Volume_Change'] + df['Amount_Change'])

    # Add to Volume and Amount Adjusted Close-to-Open Return
    df['Integrated_Factor'] = df['Volume_Momentum_Adjusted'] + df['Adjusted_Intraday_Return']

    # Adjust for Volume Volatility
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Volatility'] = (df['volume'] - df['Volume_MA_20']).rolling(window=20).std()

    # Cumulative Momentum with VWAP and Intraday High-Low Integration
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Cumulative_Sum_Log_Returns'] = df['Daily_Log_Return'].rolling(window=20).sum()

    # Intraday High-Low Ratio
    df['Intraday_High_Low_Ratio'] = df['Intraday_High_Low_Movement'] / df['close']

    # Daily Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    df['VWAP_MA_20'] = df['VWAP'].rolling(window=20).mean()

    # Weighted Intraday Momentum
    df['Weighted_Intraday_Momentum'] = df['Intraday_High_Low_Ratio'] * df['Intraday_Return']

    # Enhanced Combined Alpha Factor
    df['Enhanced_Alpha_Factor'] = (df['Cumulative_Sum_Log_Returns'] * df['VWAP_MA_20'] + 
                                   df['Weighted_Intraday_Momentum']) * \
                                  (df['Volume_Change'] + df['Amount_Change'] + df['Volume_Volatility'])

    return df['Enhanced_Alpha_Factor']
```
