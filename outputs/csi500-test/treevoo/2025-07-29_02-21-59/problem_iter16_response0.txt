import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_hl_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_hl_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    amount_change = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Combine Intraday Factors
    combined_intraday_factors = (intraday_return + volume_change + amount_change) * intraday_volatility
    
    # Calculate Daily Price Change
    daily_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum of Daily Price Changes
    sum_daily_price_changes = daily_price_change.rolling(window=20).sum()
    
    # Calculate VWAP
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMI Score
    pmi_score = sum_daily_price_changes * vwap
    
    # Calculate Volume Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Adjust Intraday Return by Volume Change
    adjusted_intraday_return = intraday_return * volume_change
    
    # Calculate Daily Log Return
    daily_log_return = (df['close'] - df['close'].shift(1)).apply(lambda x: 0 if x == 0 else x) / df['close'].shift(1)
    
    # Cumulative Sum of Daily Log Returns
    cumulative_sum_daily_log_returns = daily_log_return.rolling(window=20).sum()
    
    # Calculate 20-Day Moving Average VWAP
    moving_average_vwap = vwap.rolling(window=20).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (pmi_score * cumulative_sum_daily_log_returns) + adjusted_intraday_return + volume_weighted_close_to_open_return
    
    return final_alpha_factor
