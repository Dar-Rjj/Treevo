import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Intraday High-Low Movement
    intraday_high_low_movement = (df['high'] - df['low']) / df['low']
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Adjust Intraday Return by Volume Change
    adjusted_intraday_return = intraday_return * volume_change
    
    # Combine Adjustments
    alpha_factor = volume_weighted_close_to_open_return + adjusted_intraday_return
    
    return alpha_factor
