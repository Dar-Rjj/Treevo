```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Sum of Recent Returns (last 5 days)
    df['sum_recent_returns'] = df['daily_return'].rolling(window=5).sum()

    # Calculate Average Volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()

    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']

    # Weighted Relative Momentum
    df['weighted_momentum'] = df['sum_recent_returns'] * df['volume_ratio']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Combined Volume-Adjusted Momentum
    df['combined_momentum'] = df['weighted_momentum'] * df['close_to_open_return']

    # Adjust for Volume Volatility
    # Calculate Volume Moving Average over 20 days
    df['vol_ma_20d'] = df['volume'].rolling(window=20).mean()
    # Calculate Standard Deviation of Volume over 20 days
    df['vol_volatility'] = ((df['volume'] - df['vol_ma_20d']) ** 2).rolling(window=20).sum() ** 0.5

    # Final Adjusted Momentum Score
    df['final_momentum'] = df['combined_momentum'] / df['vol_volatility']

    # Intraday Movement Adjustment
    df['intraday_movement'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_final_momentum'] = df['final_momentum'] + df['intraday_return']

    # Calculate 5-Day Simple Moving Average (SMA) of Close price
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    # Calculate 20-Day Simple Moving Average (SMA) of Close price
    df['sma_20d'] = df['close'].rolling(window=20).mean()
    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    df['price_oscillator'] = df['sma_5d'] - df['sma_20d']

    # Enhanced PMVI Score
    df['enhanced_pmvi_score'] = df['adjusted_final_momentum'] * df['price_oscillator']

    return df['enhanced_pmvi_score']
```
