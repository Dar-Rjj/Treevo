```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Weight by Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Combine Intraday Factors
    combined_intraday_factors = (intraday_return * (intraday_volatility + volume_change))

    # Adjust Intraday Return
    adjusted_intraday_return = combined_intraday_factors * (intraday_volatility + volume_change)

    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Calculate Volume Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']

    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)

    # Sum of Daily Price Changes over 20 days
    sum_daily_price_changes = daily_price_change.rolling(window=20).sum()

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()

    # PMI Score
    pmi_score = sum_daily_price_changes * vwap

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Cumulative Sum of Daily Log Returns over 22 days
    cumulative_sum_daily_log_returns = daily_log_return.rolling(window=22).sum()

    # Calculate 20-Day Momentum
    momentum_20_day = df['close'] - df['close'].shift(20)

    # Introduce New Factor: Intraday Range to Volume Ratio
    intraday_range = df['high'] - df['low']
    intraday_range_volume_ratio = intraday_range / df['volume']

    # Final Alpha Factor
    final_alpha_factor = (
        pmi_score * cumulative_sum_daily_log_returns +
        adjusted_intraday_return +
        volume_weighted_close_to_open_return +
        momentum_20_day +
        intraday_range_volume_ratio
    )

    return final_alpha_factor
```
