```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(-1) / df['open'] - 1
    
    # Weight by Volume
    volume_weighted_momentum = close_to_open_return * df['volume']
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_intraday_return = (df['close'] / df['open'] - 1) * volume_change
    integrated_intraday_factor = intraday_high_low_ratio * adjusted_intraday_return
    
    # Adjust for Volume Volatility
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_std_220 = df['volume'].rolling(window=220).std()
    volume_volatility_adjustment = (df['volume'] - volume_ma_20) / volume_std_220
    volume_adjusted_momentum = volume_weighted_momentum / volume_volatility_adjustment
    
    # Compute Price-Volume Synchronization Measure
    intraday_range = df['high'] - df['low']
    prev_day_close_to_open_change = df['close'].shift(1) - df['open'].shift(1)
    price_volume_sync = (df['volume'] * intraday_range) / abs(prev_day_close_to_open_change)
    
    # Combine Volume-Adjusted Momentum and Price-Volume Synchronization
    combined_momentum = volume_adjusted_momentum + price_volume_sync
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Log Returns over 20 days
    aggregated_log_returns = daily_log_return.rolling(window=20).sum()
    
    # Weight by Volume Change
    weighted_intraday_return = (df['close'] / df['open'] - 1) * volume_change
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap = vwap / df['volume'].rolling(window=1).sum()  # Normalize by total volume of the day
    
    # Calculate 20-Day Moving Average VWAP
    vwap_ma_20 = vwap.rolling(window=20).mean()
    
    # Combine Cumulative Log Returns and VWAP
    cumulative_momentum_with_vwap = aggregated_log_returns * vwap_ma_20
    
    # Combined Alpha Factor
    final_alpha_factor = combined_momentum + weighted_intraday_return + cumulative_momentum_with_vwap
    
    # Identify and Confirm Trends
    today_volume_diff = df['volume'] - df['volume'].shift(1)
    sign_volume_change = np.sign(today_volume_diff)
    close_price_change = df['close'] - df['close'].shift(1)
    sign_price_change = np.sign(close_price_change)
    
    trend_confirmation = (sign_volume_change > 0) & (sign_price_change > 0) & (combined_momentum > 0)
    trend_confirmation |= (sign_volume_change < 0) & (sign_price_change < 0) & (combined_momentum < 0)
    
    # Apply trend confirmation
    final_alpha_factor[trend_confirmation] *= 1.5  # Boost factor where trend is confirmed
    
    return final_alpha_factor
```
