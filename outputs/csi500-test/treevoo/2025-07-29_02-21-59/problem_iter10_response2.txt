import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Volume-Adjusted Momentum
    sma_20 = df['close'].rolling(window=20).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    momentum_diff = ema_20 - sma_20
    adjusted_momentum = (momentum_diff * df['volume']).rolling(window=5).mean()

    # Compute Volume-Weighted Breakout Measure
    intraday_high_low_spread = df['high'] - df['low']
    previous_day_close_open_change = df['open'].shift(1) - df['close'].shift(1)
    volume_weighted_breakout = (df['volume'] * intraday_high_low_spread) / previous_day_close_open_change.abs()

    # Combine Volume-Adjusted Momentum and Volume-Weighted Breakout
    combined_factor = adjusted_momentum + volume_weighted_breakout

    # Calculate Volume Acceleration
    today_volume_difference = df['volume'] - df['volume'].shift(1)
    volume_acceleration_sign = np.sign(today_volume_difference)

    # Identify Price Momentum
    close_price_change = df['close'] - df['close'].shift(1)
    price_momentum_sign = np.sign(close_price_change)

    # Confirm Trend
    positive_trend = (adjusted_momentum > 0) & (volume_acceleration_sign > 0) & (price_momentum_sign > 0)
    negative_trend = (adjusted_momentum < 0) & (volume_acceleration_sign < 0) & (price_momentum_sign < 0)

    # Final factor value
    final_factor = np.where(positive_trend, combined_factor, np.where(negative_trend, -combined_factor, 0))

    return pd.Series(final_factor, index=df.index, name='heuristics_v2')
