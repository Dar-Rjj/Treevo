```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Combine Volume and Amount
    volume_amount = (df['volume'] + df['amount']) / 2
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = close_to_open_return * volume_amount
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Adjust by Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_intraday_return = intraday_return * volume_change
    
    # Calculate Daily Close-to-Close Return
    daily_close_to_close_return = df['close'] / df['close'].shift(1) - 1
    
    # Sum of Recent Returns
    sum_recent_returns = daily_close_to_close_return.rolling(window=5).sum()
    
    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Relative Momentum
    weighted_relative_momentum = sum_recent_returns * volume_ratio
    
    # Combined Volume-Adjusted Momentum
    combined_volume_adjusted_momentum = weighted_relative_momentum * close_to_open_return
    
    # Adjust for Volume Volatility
    volume_moving_average = df['volume'].rolling(window=20).mean()
    volume_std_dev = df['volume'].rolling(window=20).std()
    
    # Final Alpha Factor
    final_alpha_factor = (weighted_close_to_open_return + adjusted_intraday_return) * (combined_volume_adjusted_momentum / volume_std_dev)
    
    return final_alpha_factor
```
