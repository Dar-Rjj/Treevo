import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    volume_amount_weight = df['volume'] * df['amount']
    weighted_close_to_open_return = close_to_open_return * volume_amount_weight
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_return = intraday_return * volume_change
    
    # Combine with Volume and Amount Weighted Close-to-Open Return
    combined_return = weighted_close_to_open_return + adjusted_intraday_return
    
    # Include Volatility Component
    daily_volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_combined_return = combined_return * daily_volatility
    
    # Combine Crossover Components
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    crossover_momentum = short_term_ma - long_term_ma
    final_factor = volatility_adjusted_combined_return * crossover_momentum
    
    return final_factor
