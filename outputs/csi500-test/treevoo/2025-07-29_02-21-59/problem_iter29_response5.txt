```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume and Amount Adjusted Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    volume_amount_weight = (df['volume'] + df['amount']) / 2
    volume_adjusted_close_to_open = close_to_open_return * volume_amount_weight

    # Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    volume_change = df['volume'] - df['volume'].shift(1)
    intraday_adjusted = intraday_high_low_movement * (df['close'] / df['open'] - 1) * volume_change

    # Adjust for Volume Volatility
    volume_moving_avg = df['volume'].rolling(window=20).mean()
    volume_volatility = (df['volume'] - volume_moving_avg).pow(2).rolling(window=20).sum().apply(np.sqrt)

    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Weight by Volume and Amount Change
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)

    # Combine Intraday Factors
    intraday_return = df['close'] / df['open'] - 1
    combined_intraday = intraday_return * (intraday_volatility + volume_change + amount_change)

    # Adjust Intraday Return
    adjusted_intraday_return = combined_intraday * (intraday_volatility + volume_change + amount_change)

    # Calculate VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume']

    # 20-Day Exponential Moving Average VWAP
    ema_vwap = vwap.ewm(span=20, adjust=False).mean()

    # PMI Score
    pmi_score = (df['close'] - df['close'].shift(1)).cumsum() * vwap

    # Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Cumulative Sum of Daily Log Returns
    cum_sum_log_returns = daily_log_return.rolling(window=20).sum()

    # 20-Day Momentum
    momentum = df['close'] / df['close'].shift(20) - 1

    # Intraday Range to Volume Ratio
    intraday_range = df['high'] - df['low']
    intraday_range_volume_ratio = intraday_range / df['volume']

    # Combined Alpha Factor
    alpha_factor = (pmi_score * cum_sum_log_returns +
                    adjusted_intraday_return +
                    volume_adjusted_close_to_open +
                    momentum +
                    ema_vwap +
                    intraday_range_volume_ratio)

    return alpha_factor
```
