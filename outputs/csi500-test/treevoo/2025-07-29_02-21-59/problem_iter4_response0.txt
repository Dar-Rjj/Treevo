```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day Close-to-Open Change
    previous_day_close = df['close'].shift(1)
    todays_open = df['open']
    close_to_open_change = todays_open - previous_day_close
    
    # Compute Volume-Weighted Breakout Measure
    volume_weighted_breakout_measure = (df['volume'] * intraday_high_low_spread) / close_to_open_change
    
    # Smooth the Factor Over a Short Window
    smoothed_factor = volume_weighted_breakout_measure.rolling(window=N).mean()
    
    return smoothed_factor
```
