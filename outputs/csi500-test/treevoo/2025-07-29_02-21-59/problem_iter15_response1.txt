```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['IntradayHighLowMovement'] = df['high'] - df['low']
    
    # Calculate Adjusted Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    df['HighCloseMovement'] = (df['high'] - df['close']) / df['close']
    df['LowCloseMovement'] = (df['low'] - df['close']) / df['close']
    df['AdjustedIntradayReturn'] = df['IntradayReturn'] * (1 + df['HighCloseMovement'] - df['LowCloseMovement'])
    
    # Calculate Daily Close-to-Close Return
    df['PrevClose'] = df['close'].shift(1)
    df['CloseToCloseReturn'] = (df['close'] - df['PrevClose']) / df['PrevClose']
    
    # Sum of Recent Returns
    df['SumRecentReturns'] = df['CloseToCloseReturn'].rolling(window=5).sum()
    
    # Calculate Volume and Amount Adjusted Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    df['VolumeAmountWeight'] = df['volume'] * df['amount']
    df['VolumeAmountAdjustedReturn'] = df['CloseToOpenReturn'] * df['VolumeAmountWeight']
    
    # Calculate Intraday High-Low Ratio
    df['IntradayHighLowRatio'] = df['IntradayHighLowMovement'] / df['close']
    
    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average Volume over Period
    df['AverageVolume'] = df['volume'].rolling(window=5).mean()
    df['VolumeRatio'] = df['volume'] / df['AverageVolume']
    
    # Calculate Amount Change
    df['AmountChange'] = df['amount'] - df['amount'].shift(1)
    
    # Calculate Average Amount over Period
    df['AverageAmount'] = df['amount'].rolling(window=5).mean()
    df['AmountRatio'] = df['amount'] / df['AverageAmount']
    
    # Adjusted Weighted Intraday Return
    df['AdjustedWeightedIntradayReturn'] = df['AdjustedIntradayReturn'] * df['VolumeChange']
    
    # Adjusted Weighted Amount Intraday Return
    df['AdjustedWeightedAmountIntradayReturn'] = df['AdjustedIntradayReturn'] * df['AmountChange']
    
    # Combined Weighted Momentum
    df['CombinedWeightedMomentum'] = (
        df['AdjustedWeightedIntradayReturn'] +
        df['AdjustedWeightedAmountIntradayReturn'] +
        df['SumRecentReturns']
    ) * df['VolumeRatio'] * df['AmountRatio']
    
    # Adjust for Volume Volatility
    df['VolumeMovingAverage'] = df['volume'].rolling(window=20).mean()
    df['VolumeStdDev'] = df['volume'].rolling(window=20).std()
    
    # Enhanced Combined Weighted Momentum
    df['EnhancedCombinedWeightedMomentum'] = (
        df['CombinedWeightedMomentum'] * 
        df['IntradayHighLowRatio']
    )
    
    return df['EnhancedCombinedWeightedMomentum']
```
