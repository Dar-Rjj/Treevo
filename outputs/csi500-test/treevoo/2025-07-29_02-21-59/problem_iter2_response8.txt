import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Smoothing Logarithmic Returns with SMA
    df['smoothed_returns'] = df['log_returns'].rolling(window=5).mean()
    
    # Volume Shock Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_shock'] = (np.abs(df['volume_change']) > 2 * df['volume_change'].rolling(window=5).std()).astype(int)
    
    # Price Volatility
    df['price_volatility'] = (df['high'] - df['low']) + (df['high'].shift(1) - df['low'].shift(1))
    
    # Combine Volume and Volatility
    df['combined_weight'] = df['volume'] * df['price_volatility']
    
    # Multiply Smoothed Returns by Combined Weight
    df['weighted_smoothed_returns'] = df['smoothed_returns'] * df['combined_weight']
    df['final_factor'] = df.apply(lambda row: row['weighted_smoothed_returns'] if row['volume_shock'] == 1 else 0, axis=1)
    
    return df['final_factor'].dropna()
