```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_weighted_return'] = df['intraday_return'] * df['volume_change']
    
    # Calculate Logarithmic Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns
    df['smoothed_log_return'] = df['log_return'].rolling(window=5).mean()
    
    # Volume Shock Indicator
    vol_std = df['volume_change'].rolling(window=5).std() * 2
    df['vol_shock'] = 0
    df.loc[abs(df['volume_change']) > vol_std, 'vol_shock'] = 1
    
    # Combine Intraday Weighted Return and Shock Adjusted Return
    df['shock_adjusted_return'] = df['smoothed_log_return'] * df['vol_shock']
    df['combined_factor'] = df['intraday_weighted_return'] + df['shock_adjusted_return']
    
    # Return the combined factor
    return df['combined_factor'].fillna(0)
```
