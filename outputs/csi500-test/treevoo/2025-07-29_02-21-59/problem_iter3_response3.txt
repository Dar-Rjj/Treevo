```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns
    df['cum_sum_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume'])
    df['total_vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['cumulative_volume'] = df['volume'].rolling(window=1).sum()
    df['vwap'] = df['total_vwap'] / df['cumulative_volume']
    
    # Calculate 20-Day Moving Average VWAP
    df['ma_vwap_20'] = df['vwap'].rolling(window=20).mean()
    
    # PMI Score
    df['pmi_score'] = df['cum_sum_log_returns'] * df['ma_vwap_20']
    
    return df['pmi_score'].dropna()
```
