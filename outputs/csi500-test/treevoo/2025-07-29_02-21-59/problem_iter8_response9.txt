```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Sum of Recent Returns (last 5 days)
    df['sum_recent_returns'] = df['daily_return'].rolling(window=5).sum()

    # Calculate Average Volume over the last 5 days
    df['avg_volume'] = df['volume'].rolling(window=5).mean()

    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']

    # Weighted Relative Momentum
    df['weighted_relative_momentum'] = df['sum_recent_returns'] * df['volume_ratio']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1

    # Intraday High-Low Movement
    df['high_low_movement'] = df['high'] - df['low']

    # Intraday High-Low Ratio
    df['intraday_high_low_ratio'] = df['high_low_movement'] / df['close']

    # Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1

    # Volume Change
    df['volume_change'] = df['volume'].pct_change()

    # Adjust by Volume Change
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']

    # Combined Volume-Adjusted Momentum
    df['combined_volume_adjusted_momentum'] = df['weighted_relative_momentum'] * df['close_to_open_return']

    # Final Adjusted Momentum Score
    df['final_adjusted_momentum'] = df['combined_volume_adjusted_momentum'] * (df['volume'] + df['amount'])

    # Output the final adjusted momentum score
    return df['final_adjusted_momentum']
```
