```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['volume_amount_weight'] = (df['volume'] + df['amount']) / 2
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount_weight']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Adjust by Volume and Amount Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['adjusted_intraday_high_low_ratio'] = df['intraday_high_low_ratio'] * (df['volume_change'] + df['amount_change'])
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=220).sum() / 20
    df['volume_volatility'] = df['volume'].sub(df['volume_moving_avg']).pow(2).rolling(window=20).sum().pow(0.5)
    
    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['cumulative_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    df['vwap_ema_20'] = df['vwap'].ewm(span=20, adjust=False).mean()
    
    # Combined Alpha Factor
    df['combined_alpha'] = df['cumulative_log_returns'] * df['vwap_ema_20'] + df['weighted_close_to_open_return'] + df['adjusted_intraday_high_low_ratio']
    
    # Include Volatility Component
    df['daily_volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['volatility_component'] = df['daily_volatility'] * df['weighted_close_to_open_return']
    df['combined_alpha'] += df['volatility_component']
    
    # Calculate Sum of Recent Returns
    df['recent_returns_sum'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Adjust by Volume Ratios and Intraday High-Low Ratio
    df['final_alpha_factor'] = df['recent_returns_sum'] * df['volume_change'] * df['amount_change'] * df['intraday_high_low_ratio']
    df['final_alpha_factor'] += df['combined_alpha']
    
    return df['final_alpha_factor']
```
