import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Close-to-Close Return
    close_to_close_return = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Sum of Recent Returns
    sum_recent_returns = close_to_close_return.rolling(window=5).sum()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day Close-to-Open Change
    previous_day_close_to_open_change = df['open'] - df['close'].shift(1)
    
    # Compute Volume-Weighted Breakout Measure
    volume_weighted_breakout_measure = (df['volume'] * intraday_high_low_spread) / previous_day_close_to_open_change
    
    # Smooth the Factor Over a Short Window
    N = 3  # Example: smooth over 3 days
    smoothed_volume_weighted_breakout = volume_weighted_breakout_measure.rolling(window=N).mean()
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_change
    
    # Combined Weighted Momentum
    combined_weighted_momentum = (weighted_intraday_return + sum_recent_returns) * volume_ratio + smoothed_volume_weighted_breakout
    
    return combined_weighted_momentum
