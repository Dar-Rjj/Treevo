import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    volume_and_amount = df['volume'] * df['amount']
    weighted_return = close_to_open_return * volume_and_amount
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Adjust by Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_intraday_return = intraday_return * volume_change
    
    # Final Alpha Factor
    final_alpha_factor = weighted_return + adjusted_intraday_return
    
    return final_alpha_factor
