```python
def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Aggregate Daily Price Changes Over 20 Days
    df['aggregated_price_change'] = df['daily_price_change'].rolling(window=20).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # PMI Score
    df['pmi_score'] = df['aggregated_price_change'] * df['vwap']
    
    return df['pmi_score'].dropna()
```
