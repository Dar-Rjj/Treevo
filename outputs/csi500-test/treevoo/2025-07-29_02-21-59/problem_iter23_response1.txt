```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return based on the close price
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the average amount over the last 5 days
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the price range (high - low)
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the price movement (close - open)
    df['price_movement'] = df['close'] - df['open']
    
    # Alpha factor: Daily return weighted by the ratio of today's amount to the 5-day average amount
    alpha_factor = df['daily_return'] * (df['amount'] / df['avg_amount_5d'])
    
    # Return the alpha factor
    return alpha_factor.dropna()
```
