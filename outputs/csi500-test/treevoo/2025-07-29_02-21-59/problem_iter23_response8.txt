```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, high, low, close) prices
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the money flow by multiplying the average price with volume
    money_flow = avg_price * df['volume']
    
    # Calculate the change in money flow from the previous day
    money_flow_change = money_flow - money_flow.shift(1)
    
    # Calculate the factor as the ratio of money flow change to the amount
    factor = money_flow_change / df['amount']
    
    return factor
```
