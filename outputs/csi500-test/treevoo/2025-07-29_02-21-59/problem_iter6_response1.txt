```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff(1)
    
    # Calculate Amount Change
    amount_change = df['amount'].diff(1)
    
    # Multiply Intraday Return by (Volume Change + Amount Change + Intraday Volatility)
    factor = intraday_return * (volume_change + amount_change + intraday_volatility)
    
    return factor
```
