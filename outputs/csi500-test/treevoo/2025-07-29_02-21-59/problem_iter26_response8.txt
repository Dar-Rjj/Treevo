```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the factor values.
    """
    # Calculate the relative range of the day
    range_factor = (df['high'] - df['low']) / df['close']
    
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the buying pressure
    buying_pressure = df['volume'] * (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine the factors to create a composite alpha factor
    alpha_factor = (range_factor + price_change + buying_pressure) / 3
    
    return alpha_factor
```
