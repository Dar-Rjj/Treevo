```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    volume_change = df['volume'].diff().fillna(0)
    amount_change = df['amount'].diff().fillna(0)
    
    # Combine Intraday Factors
    combined_intraday_factors = intraday_return + volume_change + amount_change + intraday_volatility
    
    # Adjust Intraday Return
    adjusted_intraday_return = intraday_return * (intraday_volatility + volume_change + amount_change)
    
    # Calculate Volume Weighted Close-to-Open Return
    close_to_open_return = df['close'] / df.shift(1)['open'] - 1
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Sum of Daily Price Changes
    daily_price_changes = df['close'].diff().fillna(0).rolling(window=20).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMI Score
    pmi_score = daily_price_changes * vwap
    
    # Cumulative Sum of Daily Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1)).fillna(0)
    cumulative_log_returns = log_returns.rolling(window=20).sum()
    
    # Calculate 20-Day Momentum
    momentum_20d = df['close'] / df['close'].shift(20) - 1
    
    # Introduce New Factor: Intraday Range to Volume Ratio
    intraday_range = df['high'] - df['low']
    intraday_range_to_volume_ratio = intraday_range / df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (pmi_score * cumulative_log_returns +
                          adjusted_intraday_return +
                          volume_weighted_close_to_open_return +
                          momentum_20d +
                          intraday_range_to_volume_ratio)
    
    return final_alpha_factor
```
