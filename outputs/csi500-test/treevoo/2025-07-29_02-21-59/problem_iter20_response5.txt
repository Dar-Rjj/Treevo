import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Average (SMA) of Close Price
    df['SMA_Close'] = df['close'].rolling(window=14).mean()
    
    # SMA of High-Low Difference
    df['SMA_High_Low_Diff'] = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Exponential Moving Average (EMA) of Close Price
    df['EMA_Close'] = df['close'].ewm(span=14, adjust=False).mean()
    
    # EMA of Volume
    df['EMA_Volume'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # True Range
    df['True_Range'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    df['True_Range'] = df[['True_Range', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    
    # Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI'] = 100 - (100 / (1 + money_ratio))
    
    # On-Balance Volume (OBV)
    df['OBV'] = np.where(df['close'] > df['close'].shift(1), df['volume'], np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    
    # Bullish/Bearish Engulfing Pattern
    df['Engulfing_Pattern'] = np.where((df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1)), 1, 
                                       np.where((df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1)) & (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1)), -1, 0))
    
    # Hammer/Inverted Hammer Pattern
    df['Hammer_Inverted_Hammer'] = np.where((df['close'] - df['low']) > 2 * (df['high'] - df['close']), 1, 
                                             np.where((df['open'] - df['low']) > 2 * (df['high'] - df['open']), 1, 0))
    
    # Doji Pattern
    df['Doji_Pattern'] = np.where(abs(df['close'] - df['open']) / (df['high'] - df['low']) < 0.1, 1, 0)
    
    # Breakout Patterns
    df['Breakout_High'] = df['high'] > df['high'].rolling(window=20).max()
    df['Breakout_Low'] = df['low'] < df['low'].rolling(window=20).min()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Stochastic Oscillator
    high_14 = df['high'].rolling(window=14).max()
    low_14 = df['low'].rolling(window=14).min()
    df['%K'] = (df['close'] - low_14) / (high_14 - low_14) * 100
    df['%D'] = df['%K'].rolling(window=3).mean
