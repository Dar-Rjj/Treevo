```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    sma_close_20 = df['close'].rolling(window=20).mean()
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    momentum = ema_close_20 - sma_close_20
    volume_adjusted_momentum = momentum * df['volume']
    
    # Optionally, smooth the adjusted momentum over a 5-day window
    volume_adjusted_momentum_smoothed = volume_adjusted_momentum.rolling(window=5).mean()

    # Calculate Volume-Weighted Breakout Measure
    intraday_high_low_spread = df['high'] - df['low']
    close_to_open_change = df['open'].shift(-1) - df['close']
    volume_weighted_breakout = (intraday_high_low_spread * df['volume']) / close_to_open_change.shift(1)
    
    # Smooth the Volume-Weighted Breakout Measure Over a Short Window
    n_days = 5
    smoothed_volume_weighted_breakout = volume_weighted_breakout.rolling(window=n_days).mean()

    # Combine Volume-Adjusted Momentum and Smoothed Volume-Weighted Breakout
    combined_indicator = volume_adjusted_momentum_smoothed + smoothed_volume_weighted_breakout

    # Calculate Volume Acceleration
    volume_difference = df['volume'] - df['volume'].shift(1)
    sign_of_volume_change = np.sign(volume_difference)

    # Identify Price Momentum
    close_price_change = df['close'] - df['close'].shift(1)
    sign_of_price_change = np.sign(close_price_change)

    # Confirm Positive Trend
    positive_trend = (volume_adjusted_momentum_smoothed > 0) & (sign_of_volume_change > 0) & (sign_of_price_change > 0)

    # Confirm Negative Trend
    negative_trend = (volume_adjusted_momentum_smoothed < 0) & (sign_of_volume_change < 0) & (sign_of_price_change < 0)

    # Final alpha factor
    alpha_factor = pd.Series(np.where(positive_trend, combined_indicator, np.where(negative_trend, -combined_indicator, 0)), index=df.index)

    return alpha_factor
```
