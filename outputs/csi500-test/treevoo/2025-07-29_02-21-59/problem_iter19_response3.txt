import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    df['volume_weighted_momentum'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_intraday_high_low_ratio'] = df['intraday_high_low_ratio'] * df['volume_change']
    
    # Calculate Volume Moving Average and Standard Deviation
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_std_20'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['volume_volatility_adjusted'] = (df['volume'] - df['volume_ma_20']) / df['volume_std_20']
    
    # Volume-Adjusted Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['volume_weighted_momentum'] + df['adjusted_intraday_high_low_ratio'] * df['volume_volatility_adjusted']
    
    # Calculate Daily Log Return
    df['daily_log_return'] = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    
    # Aggregate Log Returns over 20 days
    df['cumulative_log_returns_20'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Ratio
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume Change
    df['weighted_intraday_high_low_ratio'] = df['intraday_high_low_ratio'] * df['volume_change']
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    df['vwap_ma_20'] = df['vwap'].rolling(window=20).mean()
    
    # Cumulative Momentum with VWAP Integration
    df['cumulative_momentum_with_vwap'] = df['cumulative_log_returns_20'] * df['vwap_ma_20']
    
    # Combined Alpha Factor
    df['alpha_factor'] = df['cumulative_momentum_with_vwap'] + df['volume_adjusted_intraday_momentum']
    
    return df['alpha_factor']
