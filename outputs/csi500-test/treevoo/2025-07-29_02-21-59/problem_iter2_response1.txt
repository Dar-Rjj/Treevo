import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_c2o_return = close_to_open_return * df['volume']
    
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] / df['open']) - 1
    
    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Combine with Intraday Return
    adjusted_intraday_return = intraday_return * volume_change
    
    # Integrate Intraday High-Low Movement
    factor_values = volume_weighted_c2o_return + adjusted_intraday_return
    
    return factor_values
