```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Intraday Momentum
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    intraday_return = df['close'] / df['open'] - 1
    intraday_momentum = intraday_high_low_ratio * volume_change + intraday_return
    volume_20d_ma = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_volatility = (df['volume'] - volume_20d_ma) / volume_std
    integrated_intraday_momentum = intraday_momentum / (1 + volume_volatility)

    # Cumulative Price and Volume Momentum
    daily_price_change = df['close'] - df['close'].shift(1)
    sum_daily_price_changes = daily_price_change.rolling(window=20).sum()
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    pmi_score = sum_daily_price_changes * vwap
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    cumulative_log_returns = daily_log_return.rolling(window=20).sum()
    momentum_20d = df['close'] / df['close'].shift(20) - 1
    
    # Final Alpha Factor
    alpha_factor = (pmi_score * cumulative_log_returns 
                    + integrated_intraday_momentum 
                    + cumulative_log_returns 
                    + momentum_20d)
    
    return alpha_factor
```
