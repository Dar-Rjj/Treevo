def heuristics_v2(df):
    # Volume-Adjusted Price Momentum
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    df['intraday_movement'] = df['high'] - df['low']

    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    df['weighted_intraday_momentum'] = df['volume_weighted_return'] + df['adjusted_intraday_return']

    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    df['cumulative_log_return'] = df['log_return'].rolling(window=20).sum()

    df['vwap_20_day_ma'] = df['vwap'].rolling(window=20).mean()
    
    df['volume_change_for_weight'] = df['volume'] - df['volume'].shift(1)
    df['amount_change_for_weight'] = df['amount'] - df['amount'].shift(1)
    df['weighted_cumulative_log_return'] = df['cumulative_log_return'] * df['vwap_20_day_ma'] * (df['volume_change_for_weight'] + df['amount_change_for_weight'])

    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['5_day_sma'] - df['20_day_sma']
    
    # Combined and Enhanced Alpha Factor
    df['enhanced_alpha_factor'] = df['weighted_cumulative_log_return'] + df['weighted_intraday_momentum'] * df['price_oscillator']
