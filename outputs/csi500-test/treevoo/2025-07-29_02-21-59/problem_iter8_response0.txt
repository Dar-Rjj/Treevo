```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Smoothing Logarithmic Returns using EMA
    df['smoothed_returns'] = df['log_returns'].ewm(span=5).mean()
    
    # Volume Shock Indicator
    df['volume_change'] = df['volume'].diff()
    df['volume_std'] = df['volume_change'].rolling(window=7).std()
    df['shock'] = (np.abs(df['volume_change']) > 3 * df['volume_std']).astype(int)
    
    # Integrate Intraday High-Low Movement
    df['intraday_movement'] = df['high'] - df['low']
    df['intraday_ratio'] = df['intraday_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Weight by Volume and Price Volatility
    df['price_volatility'] = (df['high'] - df['low'] + df['high'].shift(1) - df['low'].shift(1)) / 2
    df['combined_weight'] = df['volume'] * df['price_volatility']
    
    # Multiply Smoothed Returns by Combined Weight
    df['weighted_smoothed_returns'] = df['smoothed_returns'] * df['combined_weight'] * df['shock']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = np.sqrt(((df['volume'] - df['volume_moving_avg']) ** 2).rolling(window=20).sum())
    
    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_smoothed_returns'] / df['volume_volatility']
    
    return df['alpha_factor']
```
