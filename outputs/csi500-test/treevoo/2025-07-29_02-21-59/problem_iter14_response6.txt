import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Combine Volume and Amount
    volume_amount = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = close_to_open_return * volume_amount

    # Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Adjusted Intraday Return
    adjusted_intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume and Amount Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    amount_change = df['amount'] / df['amount'].shift(1) - 1
    
    # Sum of Recent Returns
    recent_returns = df['close'] / df['close'].shift(1) - 1
    sum_recent_returns = recent_returns.rolling(window=5).sum()
    
    # Calculate Daily VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume']
    
    # 20-Day Moving Average VWAP
    vwap_20_day_ma = vwap.rolling(window=20).mean()
    
    # Combined Weighted Momentum
    combined_weighted_momentum = (adjusted_intraday_return + sum_recent_returns) * (1 + volume_change) * (1 + amount_change)
    
    # Adjust for Volume Volatility
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_volatility = (df['volume'] - volume_ma).pow(2).rolling(window=20).sum().pow(0.5)
    
    # Enhanced Combined Weighted Momentum
    enhanced_combined_momentum = (weighted_close_to_open_return + combined_weighted_momentum) * (1 + volume_change) * (1 + amount_change) * (1 + intraday_high_low_ratio)
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_combined_momentum * vwap_20_day_ma + weighted_close_to_open_return
    
    return final_alpha_factor
