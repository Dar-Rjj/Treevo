import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Adjusted Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    adj_intraday_return = intraday_return * ((df['high'] - df['close']) + (df['close'] - df['low'])) / (df['high'] - df['low'])
    
    # Calculate Daily Close-to-Close Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Sum of Recent Returns
    sum_recent_returns = daily_return.rolling(window=5).sum()
    
    # Calculate Volume and Amount Adjusted Close-to-Open Return
    close_open_return = (df['close'] / df['open']) - 1
    vol_amount_weight = df['volume'] * df['amount']
    weighted_close_open_return = close_open_return * vol_amount_weight
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average Volume over Period
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / avg_volume
    
    # Calculate Amount Change
    amount_change = df['amount'] - df['amount'].shift(1)
    
    # Calculate Average Amount over Period
    avg_amount = df['amount'].rolling(window=5).mean()
    
    # Ratio of Current Day's Amount to Average Amount
    amount_ratio = df['amount'] / avg_amount
    
    # Adjusted Weighted Intraday Return
    adj_weighted_intraday_return = adj_intraday_return * volume_change
    
    # Adjusted Weighted Amount Intraday Return
    adj_weighted_amount_intraday_return = adj_intraday_return * amount_change
    
    # Combined Weighted Momentum
    combined_momentum = (adj_weighted_intraday_return + adj_weighted_amount_intraday_return + sum_recent_returns) * volume_ratio * amount_ratio
    
    # Adjust for Volume Volatility
    vol_moving_avg = df['volume'].rolling(window=20).mean()
    vol_std_dev = (df['volume'] - vol_moving_avg).pow(2).rolling(window=20).sum().apply('sqrt')
    
    # Enhanced Combined Weighted Momentum
    enhanced_combined_momentum = combined_momentum * volume_ratio * amount_ratio * intraday_high_low_ratio
    
    return enhanced_combined_momentum
