```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount_weight'] = (df['volume'] + df['amount']) / 2
    
    # Volume and Amount Adjusted Close-to-Open Return
    df['vol_amt_adj_close_to_open'] = df['close_to_open_return'] * df['volume_amount_weight']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_move'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust by Volume Change
    df['adj_intraday_high_low_move'] = df['intraday_high_low_move'] + df['volume_change']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Add to Volume and Amount Adjusted Close-to-Open Return
    df['combined_factor'] = df['vol_amt_adj_close_to_open'] + df['adj_intraday_high_low_move'] + df['intraday_return']
    
    # Calculate Volume Standard Deviation
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_diff'] = df['volume'] - df['volume_moving_avg']
    df['volume_std_dev'] = df['volume_diff'].rolling(window=20).std()

    # Adjust for Volume Volatility
    df['volatility_adjusted_factor'] = df['combined_factor'] / df['volume_std_dev']

    # Integrate Previous Day's Close-to-Open Return
    df['prev_day_close_to_open_return'] = (df['open'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Adjust Current Day's Factor
    df['adjusted_vol_amt_close_to_open'] = df['vol_amt_adj_close_to_open'] + df['prev_day_close_to_open_return']
    
    # Integrate Previous Day's Intraday High-Low Movement
    df['prev_day_intraday_high_low_move'] = (df['high'].shift(1) - df['low'].shift(1)) / df['low'].shift(1)
    
    # Final Integrated Alpha Factor
    df['integrated_alpha_factor'] = df['adjusted_vol_amt_close_to_open'] + df['adj_intraday_high_low_move'] + df['prev_day_intraday_high_low_move']
    
    return df['integrated_alpha_factor']
```
