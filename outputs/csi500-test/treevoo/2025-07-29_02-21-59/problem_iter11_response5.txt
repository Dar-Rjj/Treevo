import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    volume_amount_weight = (df['volume'] + df['amount']) / 2
    weighted_close_to_open_return = volume_amount_weight * close_to_open_return
    
    # Integrate Intraday High-Low Movement
    intraday_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_movement / df['close']
    intraday_return = df['close'] / df['open'] - 1
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    adjusted_intraday_return = intraday_return * (1 + volume_change)
    
    # Include Volatility Component
    daily_volatility = df['close'].rolling(window=5).std()
    volatility_adjusted_return = weighted_close_to_open_return * daily_volatility
    
    # Smooth Logarithmic Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    smoothed_log_returns = log_returns.ewm(span=5, adjust=False).mean()
    
    # Volume Shock Indicator
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock = (np.abs(daily_volume_change) > 3 * daily_volume_change.rolling(window=7).std()).astype(int)
    
    # Weight by Volume and Price Volatility
    price_volatility = (df['high'] - df['low'] + df['high'].shift(1) - df['low'].shift(1)) / 2
    combined_weight = (df['volume'] + df['volume'].shift(1) + df['volume'].shift(20)) * price_volatility
    
    # Final Alpha Factor
    alpha_factor = (weighted_close_to_open_return + adjusted_intraday_return + 
                    volatility_adjusted_return + smoothed_log_returns * combined_weight)
    alpha_factor = alpha_factor.where(volume_shock == 1, 0)
    
    # Adjust for Volume Volatility
    volume_moving_average = df['volume'].rolling(window=20).mean()
    volume_std = (df['volume'] - volume_moving_average).pow(2).rolling(window=20).sum().apply(np.sqrt)
    alpha_factor = alpha_factor / (1 + volume_std)
    
    return alpha_factor
