```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['high_low_movement'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Sum of Recent Returns
    df['sum_recent_returns'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Amount Change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Average Volume
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Average Amount
    df['avg_amount'] = df['amount'].rolling(window=5).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Amount Ratio
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Weighted Amount Intraday Return
    df['weighted_amount_intraday_return'] = df['intraday_return'] * df['amount_change']
    
    # Combined Weighted Momentum
    df['combined_weighted_momentum'] = (df['weighted_intraday_return'] + 
                                        df['weighted_amount_intraday_return'] + 
                                        df['sum_recent_returns']) * df['volume_ratio'] * df['amount_ratio']
    
    # Adjusted Logarithmic Returns with Enhanced Volume Shock
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns
    df['smoothed_log_returns'] = df['log_returns'].rolling(window=5).mean()
    
    # Volume Shock Indicator
    df['volume_shock'] = 0
    std_dev_vol = df['volume'].rolling(window=5).std()
    df.loc[np.abs(df['volume'] - df['volume'].shift(1)) > 2 * std_dev_vol, 'volume_shock'] = 1
    
    # Combine Intraday Weighted Return and Shock Adjusted Return
    df['adjusted_return'] = df['smoothed_log_returns'] * (df['high'] - df['low']) / df['close']
    df['final_return'] = df['combined_weighted_momentum']
    df.loc[df['volume_shock'] == 1, 'final_return'] = df['adjusted_return']
    
    return df['final_return']
```
