```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the relative return from open to close
    relative_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Alpha factor: combine relative return, VWAP, and distance from 5-day MA
    factor = relative_return + (vwap - ma_5) / ma_5
    
    return factor
```
