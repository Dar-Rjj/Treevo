```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return using Close price
    df['log_return'] = (df['close'].shift(0) - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate 20-Day Sum of Squared Daily Log Returns
    df['sum_sq_log_return_20d'] = df['log_return'].rolling(window=20).apply(lambda x: (x**2).sum(), raw=True)
    
    # Calculate 5-Day Simple Moving Average (SMA) of Close price
    df['sma_close_5d'] = df['close'].rolling(window=5).mean()
    
    # Calculate 20-Day Simple Moving Average (SMA) of Close price
    df['sma_close_20d'] = df['close'].rolling(window=20).mean()
    
    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    df['price_oscillator'] = df['sma_close_5d'] - df['sma_close_20d']
    
    # Calculate Daily Volume Weighted Average Price (VWAP) using High, Low, Close, and Volume
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMVI Score: Multiply 20-Day Sum of Squared Log Returns by VWAP and Price Oscillator
    df['pmvi_score'] = df['sum_sq_log_return_20d'] * df['vwap'] * df['price_oscillator']
    
    return df['pmvi_score']
```
