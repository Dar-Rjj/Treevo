```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Combine Volume and Amount
    df['volume_amount'] = (df['volume'] + df['amount']) / 2
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Calculate Intraday High-Low Movement
    df['intraday_high_low_movement'] = df['high'] - df['low']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Adjust Intraday High-Low Movement by Volume Change
    df['adjusted_intraday_high_low_movement'] = df['intraday_high_low_movement'] * df['volume_change']
    
    # Add to Volume and Amount Adjusted Close-to-Open Return
    df['combined_factor'] = df['weighted_close_to_open_return'] + df['adjusted_intraday_high_low_movement'] + df['intraday_return']
    
    # Calculate Volume Standard Deviation
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_diff'] = df['volume'] - df['volume_ma']
    df['volume_volatility'] = df['volume_diff'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['volume_adjusted_factor'] = df['combined_factor'] / (df['volume_volatility'] + 1e-6)
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'].shift(1) / df['close'].shift(2) - 1
    
    # Adjust Current Day's Factor
    df['current_day_factor'] = df['volume_adjusted_factor'] + df['prev_close_to_open_return']
    
    # Calculate Previous Day's Intraday High-Low Movement
    df['prev_intraday_high_low_movement'] = df['high'].shift(1) - df['low'].shift(1)
    
    # Adjust Current Day's Factor
    df['final_factor'] = df['current_day_factor'] + df['prev_intraday_high_low_movement']
    
    return df['final_factor']
```
