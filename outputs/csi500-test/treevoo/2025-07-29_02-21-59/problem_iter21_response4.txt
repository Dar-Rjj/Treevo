```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    df['ROC_5'] = ((df['close'] / df['close'].shift(5)) - 1) * 100
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['OBV'] = obv
    
    # Calculate Volume Surge
    df['Volume_Surge'] = (df['volume'] / df['volume'].rolling(window=20).mean()) - 1
    
    # Identify Doji pattern
    df['Doji'] = (abs(df['open'] - df['close']) / (df['high'] - df['low'])) < 0.1
    
    # Calculate Breakout High and Low
    df['Breakout_High'] = (df['close'] > df['high'].rolling(window=10).max()).astype(int)
    df['Breakout_Low'] = (df['close'] < df['low'].rolling(window=10).min()).astype(int)
    
    # Assuming EPS and P/E ratio are available in the DataFrame
    df['EPS_Trend'] = (df['eps'] / df['eps'].shift(1)) - 1
    df['P/E_Ratio_Change'] = df['pe_ratio'] - df['pe_ratio'].shift(1)
    
    # Combine and weight individual components to form a composite alpha factor
    df['Alpha_Factor'] = (
        0.3 * df['ROC_5'] +
        0.2 * df['OBV'] +
        0.1 * df['Volume_Surge'] +
        0.1 * df['Doji'] +
        0.1 * df['Breakout_High'] -
        0.1 * df['Breakout_Low'] +
        0.1 * df['EPS_Trend'] +
        0.1 * df['P/E_Ratio_Change']
    )
    
    return df['Alpha_Factor']
```
