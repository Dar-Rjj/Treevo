```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns (exclude negative values)
    df['positive_log_return'] = df['log_return'].apply(lambda x: x if x > 0 else 0)
    df['cumulative_log_return'] = df['positive_log_return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    df['ma_vwap'] = df['vwap'].rolling(window=20).mean()
    
    # PMI Score
    df['pmi_score'] = df['cumulative_log_return'] * df['ma_vwap']
    
    # Adjust for Recent Volatility
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['pmi_factor'] = df['pmi_score'] / df['volatility']
    
    # Final Alpha Factor
    alpha_factor = df['pmi_factor']
    
    return alpha_factor
```
