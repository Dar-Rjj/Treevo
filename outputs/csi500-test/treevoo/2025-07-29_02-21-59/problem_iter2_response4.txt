```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Weight by Volume Change
    weighted_intraday_return = intraday_return * volume_change
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Sum of Daily Price Changes over 20 days
    sum_daily_price_changes = daily_price_change.rolling(window=20).sum().abs()
    
    # Calculate Daily VWAP
    daily_vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Aggregate VWAP over 20 days
    vwap_20d = daily_vwap.rolling(window=20).mean()
    
    # Weighted Intraday Momentum
    weighted_intraday_momentum = intraday_high_low_ratio * intraday_return
    
    # Combined Alpha Factor
    combined_alpha_factor = (sum_daily_price_changes * vwap_20d) + weighted_intraday_momentum
    
    return combined_alpha_factor
```
