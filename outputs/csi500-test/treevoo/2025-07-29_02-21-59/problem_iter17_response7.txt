```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount'] = (df['volume'] + df['amount']) / 2
    
    # Weight by Volume and Amount
    df['vol_amount_adj_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Intraday Return by Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['adj_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Adjust by Volume and Amount Change
    df['amount_change'] = df['amount'] / df['amount'].shift(1)
    df['adj_intraday_return'] = df['adj_intraday_return'] * df['amount_change']
    
    # Calculate Sum of Recent Returns
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['sum_recent_returns'] = df['close_to_close_return'].rolling(window=5).sum()
    
    # Calculate Daily VWAP
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap'] /= df['volume']
    
    # Calculate 20-Day Moving Average VWAP
    df['ma_vwap_20'] = df['vwap'].rolling(window=20).mean()
    
    # Combined Weighted Momentum
    df['combined_momentum'] = df['intraday_return'] + df['adj_intraday_return'] + df['sum_recent_returns']
    df['combined_momentum'] *= df['volume_change'] * df['amount_change'] * df['intraday_high_low_ratio']
    
    # PMI Score
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=20).sum()
    df['pmi_score'] = df['sum_daily_price_changes'] * df['vwap']
    
    # Adjust for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_std'] = df['volume'] - df['volume_ma_20']
    df['volume_std'] = df['volume_std'].pow(2).rolling(window=20).sum().pow(0.5)
    
    # Enhanced Combined Weighted Momentum
    df['enhanced_combined_momentum'] = df['combined_momentum'] * df['volume_change'] * df['amount_change'] * df['intraday_high_low_ratio']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['enhanced_combined_momentum'] * df['ma_vwap_20'] + df['pmi_score'] + df['vol_amount_adj_close_to_open_return']
    
    return df['alpha_factor'].dropna()
```
