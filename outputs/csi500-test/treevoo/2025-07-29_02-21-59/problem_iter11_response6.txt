```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume and Amount
    volume_amount_weight = (df['volume'] + df['amount']) * close_to_open_return
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    intraday_return = (df['close'] / df['open']) - 1
    volume_change = df['volume'] / df['volume'].shift(1)
    adjusted_intraday_return = intraday_return * volume_change + volume_amount_weight
    
    # Calculate Daily Close-to-Close Return
    daily_close_to_close_return = df['close'] / df['close'].shift(1) - 1
    
    # Sum of Recent Returns
    recent_returns_sum = daily_close_to_close_return.rolling(window=5).sum()
    
    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Relative Momentum
    weighted_relative_momentum = recent_returns_sum * volume_ratio
    
    # Combined Volume-Adjusted Momentum
    combined_volume_adjusted_momentum = weighted_relative_momentum * close_to_open_return
    
    # Adjust for Volume Volatility
    volume_moving_average = df['volume'].rolling(window=20).mean()
    volume_volatility = np.sqrt(((df['volume'] - volume_moving_average) ** 2).rolling(window=20).sum())
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    volume_change = df['volume'] / df['volume'].shift(1)
    amount_change = df['amount'] / df['amount'].shift(1)
    
    # Final Alpha Factor
    final_alpha_factor = (
        (combined_volume_adjusted_momentum / volume_volatility) +
        (intraday_return * (volume_change + amount_change + intraday_volatility)) +
        (volume_amount_weight * (df['close'] / df['open'] - 1))
    )
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns
    cumulative_log_returns = daily_log_return.rolling(window=20).sum()
    
    # Calculate Daily VWAP
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    vwap_20_day_sma = vwap.rolling(window=20).mean()
    
    # PMI Score and Final Alpha Factor
    pmi_score = cumulative_log_returns * vwap_20_day_sma
    final_alpha_factor += pmi_score
    
    return final_alpha_factor
```
