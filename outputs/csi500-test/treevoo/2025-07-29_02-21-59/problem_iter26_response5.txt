def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Factors
    intraday_factor = intraday_return * volume_change
    
    # Calculate Daily Close-to-Close Return
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum of Recent Returns
    sum_recent_returns = close_to_close_return.rolling(window=5).sum()
    
    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Relative Momentum
    weighted_momentum = sum_recent_returns * volume_ratio
    
    # Adjust Intraday Return by Volume Change
    adjusted_intraday_return = intraday_return * volume_change
    
    # Calculate Volume and Amount Adjusted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_amount_weight = (df['volume'] + df['amount'])
    volume_amount_adjusted_return = close_to_open_return * volume_amount_weight
    
    # Intraday High-Low Movement
    high_low_movement = (df['high'] - df['low']) / df['low']
    high_low_adjusted_return = high_low_movement + volume_amount_adjusted_return
    
    # Calculate Volume Standard Deviation
    volume_std = df['volume'].rolling(window=20).std()
    
    # Adjust Current Day's Factor
    current_day_factor = volume_amount_adjusted_return + high_low_adjusted_return
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns
    cumulative_log_return = daily_log_return.rolling(window=20).sum()
    
    # Calculate 20-Day Momentum
    momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap = vwap.cumsum() / df['volume'].cumsum()
    
    # PMI Score
    pmi_score = sum_recent_returns * vwap
    
    # Adjust Intraday Return
    intraday_volatility = (df['high'] - df['low']) / df['close']
    adjusted_intraday = intraday_return * (intraday_volatility + volume_change + df['amount'].diff())
    
    # Final Alpha Factor
    final_alpha_factor = (pmi_score * cumulative_log_return) + adjusted_intraday + volume_amount_adjusted_return + momentum_20
    
    return final_alpha_factor
