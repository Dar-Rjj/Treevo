```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Sum of Daily Price Changes over 20 days
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=20).sum()
    
    # Calculate Daily VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Aggregate VWAP over 20 days
    df['vwap_20'] = df['vwap'].rolling(window=20).mean()
    
    # PMI Score
    df['pmi_score'] = df['sum_daily_price_changes'] * df['vwap_20']
    
    return df['pmi_score']
```
