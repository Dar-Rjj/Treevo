```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum of Recent Returns (last 5 days)
    df['sum_recent_returns'] = df['close_return'].rolling(window=5).sum()
    
    # Calculate Average Volume over Period (last 5 days)
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Calculate Price Volatility using High and Low prices
    df['high_low_diff'] = (df['high'] - df['low']) + (df['high'].shift(1) - df['low'].shift(1))
    
    # Combine Volume and Volatility
    df['combined_weight'] = df['volume_ratio'] * df['high_low_diff']
    
    # Weighted Relative Momentum
    df['weighted_momentum'] = df['sum_recent_returns'] * df['combined_weight']
    
    return df['weighted_momentum']
```
