```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['volume_amount_weight'] = (df['volume'] + df['amount']) / 2
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount_weight']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low'])
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    df['combined_return'] = df['weighted_close_to_open_return'] + df['adjusted_intraday_return']
    
    # Include Volatility Component
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    df['volatility_adjusted_return'] = df['weighted_close_to_open_return'] * df['daily_volatility']
    
    # Incorporate Price Trend Strength
    def adx(high, low, close, lookback):
        plus_dm = high.diff()
        minus_dm = low.diff()
        plus_dm[plus_dm < 0] = 0
        minus_dm[minus_dm > 0] = 0
        
        tr = pd.DataFrame(np.zeros_like(close), index=df.index)
        tr.columns = ['TR']
        for i in range(1, len(tr)):
            tr_i = max(high.iloc[i] - low.iloc[i], 
                       abs(high.iloc[i] - close.iloc[i-1]), 
                       abs(low.iloc[i] - close.iloc[i-1]))
            tr.iloc[i] = tr_i
            
        atr = tr.rolling(window=lookback).mean()
        
        plus_di = 100 * (plus_dm.ewm(alpha=1/lookback, min_periods=lookback).mean() / atr)
        minus_di = abs(100 * (minus_dm.ewm(alpha=1/lookback, min_periods=lookback).mean() / atr))
        
        dx = (plus_di - minus_di) / (plus_di + minus_di) * 100
        adx = dx.ewm(alpha=1/lookback, min_periods=lookback).mean()
        
        return adx
    
    df['adx'] = adx(df['high'], df['low'], df['close'], 14)
    df['trend_strength_adjusted_return'] = df['weighted_close_to_open_return'] * df['adx']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['combined_return'] 
                          + df['volatility_adjusted_return'] 
                          + df['trend_strength_adjusted_return'])
    
    return df['alpha_factor']
```
