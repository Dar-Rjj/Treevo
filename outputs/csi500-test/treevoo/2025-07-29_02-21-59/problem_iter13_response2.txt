```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns over 20 days
    df['cumulative_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    df['ma_vwap'] = df['vwap'].rolling(window=20).mean()
    
    # Calculate PMI Score
    df['pmi_score'] = df['cumulative_log_returns'] * df['ma_vwap']
    
    # Adjust for Recent Volatility
    df['volatility'] = df['daily_log_return'].rolling(window=20).std()
    df['pmi_factor'] = df['pmi_score'] / df['volatility']
    
    return df['pmi_factor']
```
