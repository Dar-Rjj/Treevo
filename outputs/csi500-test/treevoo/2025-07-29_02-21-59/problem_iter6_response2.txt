```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns with EMA
    df['smoothed_returns'] = df['log_returns'].ewm(span=5).mean()
    
    # Volume Shock Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_std = df['volume_change'].rolling(window=7).std()
    df['shock'] = (np.abs(df['volume_change']) > 3 * volume_std).astype(int)
    
    # Price Volatility
    df['high_low_diff'] = (df['high'] - df['low'] + df['high'].shift(1) - df['low'].shift(1)) / 2
    
    # Combine Volume and Price Volatility
    df['combined_weight'] = df['volume'] * df['high_low_diff']
    
    # Multiply Smoothed Returns by Combined Weight and Apply Shock Filter
    df['factor'] = df['smoothed_returns'] * df['combined_weight'] * df['shock']
    
    return df['factor']
```
