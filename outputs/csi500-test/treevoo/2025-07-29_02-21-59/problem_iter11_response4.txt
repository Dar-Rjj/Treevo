def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Adjusted Close-to-Open Return
    df['volume_adjusted_c_to_o'] = df['close_to_open_return'] * df['volume']
    
    # Intraday High-Low Movement
    df['intraday_high_low_move'] = df['high'] - df['low']
    
    # Intraday High-Low Ratio
    df['intraday_high_low_ratio'] = df['intraday_high_low_move'] / df['close']

    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Combine with Volume Adjusted Close-to-Open Return
    df['combined_volume_adjusted_return'] = df['volume_adjusted_c_to_o'] + df['adjusted_intraday_return']
    
    # Calculate Volume Moving Average and Standard Deviation
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['volatility_adjusted_return'] = df['combined_volume_adjusted_return'] / df['volume_volatility']
