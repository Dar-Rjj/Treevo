```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Intraday Momentum
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    intraday_high_low_movement = (df['high'] - df['low']) / df['close']
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_intraday_high_low_movement = intraday_high_low_movement * (1 + volume_change / df['volume'])
    
    volume_volatility = df['volume'].rolling(window=20).std()
    volume_volatility_adjustment = df['volume'] / volume_volatility
    
    integrated_intraday_momentum = (close_to_open_return + adjusted_intraday_high_low_movement) * volume_volatility_adjustment

    # Cumulative Price and Volume Momentum
    daily_price_change = df['close'] - df['close'].shift(1)
    sum_of_daily_price_changes = daily_price_change.rolling(window=20).sum()
    
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['volume']
    pmi_score = sum_of_daily_price_changes * vwap
    
    cumulative_sum_of_daily_log_returns = (np.log(df['close'] / df['close'].shift(1))).rolling(window=22).sum()
    
    twenty_two_day_momentum = df['close'] / df['close'].shift(22)

    # Final Alpha Factor
    final_alpha_factor = (pmi_score * cumulative_sum_of_daily_log_returns 
                          + integrated_intraday_momentum 
                          + cumulative_sum_of_daily_log_returns 
                          + twenty_two_day_momentum)
    
    return final_alpha_factor
```
