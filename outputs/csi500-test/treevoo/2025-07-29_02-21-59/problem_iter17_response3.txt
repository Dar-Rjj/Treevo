```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from open to close, and from high to low
    open_to_close_movement = df['close'] - df['open']
    high_to_low_movement = df['high'] - df['low']
    
    # Calculate the buying pressure, which is a measure of how much volume is needed to move prices
    buying_pressure = (df['amount'] / df['volume']) - (df['high'] + df['low']) / 2
    
    # The alpha factor is constructed by combining the price movement with the buying pressure
    # This factor tries to capture the momentum and volume pressure in the market
    alpha_factor = (open_to_close_movement / high_to_low_movement) * buying_pressure
    
    return alpha_factor
```
