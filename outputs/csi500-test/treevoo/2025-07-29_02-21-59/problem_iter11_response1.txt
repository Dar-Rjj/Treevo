```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range for each day
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure: (close - low) / (high - low)
    df['buying_pressure'] = (df['close'] - df['low']) / df['price_range']
    
    # Calculate the selling pressure: (high - close) / (high - low)
    df['selling_pressure'] = (df['high'] - df['close']) / df['price_range']
    
    # Calculate the volume-weighted buying pressure: sum(volume * buying_pressure)
    df['vw_buying_pressure'] = df['volume'] * df['buying_pressure']
    
    # Calculate the volume-weighted selling pressure: sum(volume * selling_pressure)
    df['vw_selling_pressure'] = df['volume'] * df['selling_pressure']
    
    # Calculate the net buying pressure: vw_buying_pressure - vw_selling_pressure
    df['net_buying_pressure'] = df['vw_buying_pressure'] - df['vw_selling_pressure']
    
    # Calculate the relative strength index (RSI) of the net buying pressure
    window = 14
    delta = df['net_buying_pressure'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    rs = gain / loss
    df['rsi_net_buying_pressure'] = 100 - (100 / (1 + rs))
    
    # The final alpha factor is the RSI of the net buying pressure
    return df['rsi_net_buying_pressure']
```
