def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)
    
    # Combine Intraday Factors
    factor = (
        intraday_return +
        volume_change +
        amount_change +
        intraday_volatility
    )
    
    return factor
