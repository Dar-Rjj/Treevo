```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of Recent Returns
    df['sum_recent_returns'] = df['daily_return'].rolling(window=5).sum()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Average Volume
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Amount Change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Average Amount
    df['average_amount'] = df['amount'].rolling(window=5).mean()
    
    # Amount Ratio
    df['amount_ratio'] = df['amount'] / df['average_amount']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Weighted Amount Intraday Return
    df['weighted_amount_intraday_return'] = df['intraday_return'] * df['amount_change']
    
    # Combined Weighted Momentum
    df['combined_weighted_momentum'] = (
        df['weighted_intraday_return'] + 
        df['weighted_amount_intraday_return'] + 
        df['sum_recent_returns']
    ) * df['volume_ratio'] * df['amount_ratio']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Volume and Amount Adjusted Close-to-Open Return
    df['volume_and_amount_adjusted_close_to_open_return'] = (
        df['close_to_open_return'] * (df['volume'] + df['amount'])
    )
    
    # Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['low']
    df['intraday_high_low_movement'] = df['intraday_high_low_movement'] * df['volume_change'] + df['intraday_return']
    
    # Adjust for Volume Volatility
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Final Adjusted Momentum Score
    df['final_adjusted_momentum_score'] = (
        df['combined_weighted_momentum'] / df['volume_volatility'] + 
        df['intraday_high_low_movement']
    )
    
    return df['final_adjusted_momentum_score']
```
