```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Combine Volume and Amount
    volume_amount = df['volume'] * df['amount']
    
    # Weight by Volume and Amount
    vol_adj_close_to_open_return = close_to_open_return * volume_amount
    
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjust by Volume Change
    adjusted_intraday_high_low_movement = intraday_high_low_movement / (1 + volume_change.abs())
    
    # Integrate Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['open'].shift(1) / df['close'].shift(2) - 1
    
    # Integrate Previous Day's Intraday High-Low Movement
    prev_intraday_high_low_movement = df['high'].shift(1) - df['low'].shift(1)
    
    # Combine all components
    factor = (vol_adj_close_to_open_return + 
              adjusted_intraday_high_low_movement - 
              prev_close_to_open_return - 
              prev_intraday_high_low_movement)
    
    return factor
```
