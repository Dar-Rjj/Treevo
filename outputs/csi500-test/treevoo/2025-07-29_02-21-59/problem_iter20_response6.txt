```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Amount
    df['volume_amount'] = df['volume'] * df['amount']
    
    # Volume and Amount Adjusted Close-to-Open Return
    df['vol_adj_close_to_open_return'] = df['close_to_open_return'] * df['volume_amount']
    
    # Intraday High-Low Movement
    df['high_low_movement'] = df['high'] - df['low']
    
    # Intraday High-Low Ratio
    df['high_low_ratio'] = df['high_low_movement'] / df['close']
    
    # Adjusted Intraday Return
    df['adjusted_intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume and Amount Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['vol_amount_change'] = df['volume_change'] + df['amount_change']
    
    # Daily Volatility
    df['daily_volatility'] = df['close'].rolling(window=20).std()
    
    # Volume and Amount Weighted Close-to-Open Return with Volatility
    df['vol_amount_weighted_close_to_open_with_volatility'] = df['vol_adj_close_to_open_return'] * df['daily_volatility']
    
    # Sum of Recent Returns
    df['recent_returns_sum'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Daily VWAP
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # 20-Day Moving Average VWAP
    df['ma_vwap_20'] = df['vwap'].rolling(window=20).mean()
    
    # Combined Weighted Momentum
    df['combined_weighted_momentum'] = (df['adjusted_intraday_return'] + 
                                        df['vol_adj_close_to_open_return'] + 
                                        df['recent_returns_sum']) * df['vol_amount_change']
    
    # Volume Volatility
    df['vol_moving_avg_20'] = df['volume'].rolling(window=20).mean()
    df['vol_std_20'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['enhanced_combined_weighted_momentum'] = (df['combined_weighted_momentum'] * 
                                                 df['volume_change'] * 
                                                 df['amount_change'] * 
                                                 df['high_low_ratio'])
    
    # PMI Score
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['sum_daily_price_change'] = df['daily_price_change'].rolling(window=20).sum()
    df['pmi_score'] = df['sum_daily_price_change'] * df['vwap']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['enhanced_combined_weighted_momentum'] * 
                                df['ma_vwap_20'] + 
                                df['vol_adj_close_to_open_return'] + 
                                df['adjusted_intraday_return'] + 
                                df['vol_amount_weighted_close_to_open_with_volatility'])
    
    return df['final_alpha_factor']
```
