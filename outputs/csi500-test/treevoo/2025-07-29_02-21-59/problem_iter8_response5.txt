import pandas as pd

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns using EMA
    smoothed_returns = log_returns.ewm(span=5, adjust=False).mean()
    
    # Volume Shock Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    std_7d = volume_change.rolling(window=7).std()
    shock_indicator = (np.abs(volume_change) > 3 * std_7d).astype(int)
    
    # Weight by Volume and Price Volatility
    price_volatility = (df['high'] - df['low'] + df['high'].shift(1) - df['low'].shift(1)) / 2
    combined_weight = df['volume'] * price_volatility
    
    # Multiply Smoothed Returns by Combined Weight
    weighted_smoothed_returns = smoothed_returns * combined_weight * shock_indicator
    
    # Volume Weighted Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Intraday Movement Adjustment
    intraday_movement = (df['high'] - df['low']) / df['low']
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_change = df['volume'] - df['volume'].shift(1)
    intraday_adjustment = intraday_return * volume_change
    
    # Combine Adjustments
    final_factor = weighted_smoothed_returns + volume_weighted_close_to_open + intraday_adjustment
    
    return final_factor
