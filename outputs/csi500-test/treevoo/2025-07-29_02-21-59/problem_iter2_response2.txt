```python
def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Sum of Recent Returns (last 5 days)
    df['sum_recent_returns'] = df['close_to_close_return'].rolling(window=5).sum()
    
    # Calculate Average Volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Weighted Relative Momentum
    df['weighted_relative_momentum'] = df['sum_recent_returns'] * df['volume_ratio']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combined Volume-Adjusted Momentum
    df['combined_momentum'] = df['weighted_relative_momentum'] * df['close_to_open_return']
    
    return df['combined_momentum'].dropna()
```
