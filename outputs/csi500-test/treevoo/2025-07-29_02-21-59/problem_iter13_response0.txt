import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_ratio = (df['high'] - df['low']) / df['close']
    
    # Adjust by Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    intraday_adjusted_return = ((df['close'] - df['open']) / df['open']) * volume_change
    
    # Combine with Intraday High-Low Ratio
    intraday_momentum = intraday_high_low_ratio * intraday_adjusted_return
    
    # Adjust for Volume Volatility
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_volatility = df['volume'].rolling(window=20).std()
    volume_adjusted_intraday_momentum = intraday_momentum / volume_volatility
    
    # Cumulative Momentum with VWAP Integration
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    cumulative_log_return = daily_log_return.rolling(window=20).sum()
    
    # Weight by Volume Change
    volume_weighted_intraday_return = ((df['close'] - df['open']) / df['open']) * volume_change
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    vwap_ma_20 = vwap.rolling(window=20).mean()
    
    # Combined Alpha Factor
    combined_alpha_factor = (cumulative_log_return * vwap_ma_20) + volume_adjusted_intraday_momentum
    
    return combined_alpha_factor
