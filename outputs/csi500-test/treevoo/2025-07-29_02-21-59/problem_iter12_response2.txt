import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(-1) / df['open'] - 1
    
    # Weight by Volume and Amount
    volume_amount_weight = (df['volume'] * df['amount']) / (df['volume'] + df['amount'])
    weighted_close_to_open_return = close_to_open_return * volume_amount_weight
    
    # Integrate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    intraday_return = df['close'] / df['open'] - 1
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    adjusted_intraday_return = intraday_return * (1 + volume_change)
    
    # Include Volatility Component
    daily_volatility = df['close'].pct_change().rolling(window=20).std()
    volatility_adjusted_return = weighted_close_to_open_return * daily_volatility
    
    # Incorporate Price Trend Strength
    def calculate_adx(high, low, close, lookback):
        plus_dm = high.diff()
        minus_dm = low.diff()
        plus_dm[plus_dm < 0] = 0
        minus_dm[minus_dm > 0] = 0
        
        tr = pd.DataFrame({'hl': high - low, 'hc': abs(high - close.shift()), 'lc': abs(low - close.shift())})
        tr['tr'] = tr.max(axis=1)
        
        atr = tr['tr'].rolling(lookback).mean()
        
        plus_di = 100 * (plus_dm.ewm(alpha=1/lookback).mean() / atr)
        minus_di = 100 * (minus_dm.abs().ewm(alpha=1/lookback).mean() / atr)
        
        dx = (abs(plus_di - minus_di) / (plus_di + minus_di)) * 100
        adx = dx.ewm(alpha=1/lookback).mean()
        
        return adx
    
    adx = calculate_adx(df['high'], df['low'], df['close'], 14)
    trend_strength_adjusted_return = weighted_close_to_open_return * adx
    
    # Final Alpha Factor
    final_alpha_factor = (weighted_close_to_open_return + 
                          adjusted_intraday_return + 
                          volatility_adjusted_return + 
                          trend_strength_adjusted_return)
    
    return final_alpha_factor
