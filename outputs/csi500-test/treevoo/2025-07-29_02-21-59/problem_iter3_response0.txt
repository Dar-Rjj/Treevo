```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Sum of Daily Price Changes (excluding negative values)
    df['positive_daily_change'] = df['daily_price_change'].apply(lambda x: x if x > 0 else 0)
    df['sum_positive_daily_change'] = df['positive_daily_change'].rolling(window=20).sum()
    
    # Calculate Daily VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    df['vwap_20'] = df['vwap'].rolling(window=20).mean()
    
    # PMI Score
    df['pmi_score'] = df['sum_positive_daily_change'] * df['vwap_20']
    
    # Adjust for Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['std_dev_20'] = df['daily_return'].rolling(window=20).std()
    df['pmi_factor'] = df['pmi_score'] / df['std_dev_20']
    
    return df['pmi_factor']
```
