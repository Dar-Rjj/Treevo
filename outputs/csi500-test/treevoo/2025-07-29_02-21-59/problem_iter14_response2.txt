```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()

    # Sum of Recent Returns
    df['sum_recent_returns'] = df['close_to_close_return'].rolling(window=5).sum()

    # Calculate Average Volume over Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()

    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']

    # Weighted Relative Momentum
    df['weighted_relative_momentum'] = df['sum_recent_returns'] * df['volume_ratio']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Combined Volume-Adjusted Momentum
    df['combined_vol_adj_momentum'] = df['weighted_relative_momentum'] * df['close_to_open_return']

    # Adjust for Volume Volatility
    df['volume_ma_20d'] = df['volume'].rolling(window=20).mean()
    df['vol_deviation'] = df['volume'] - df['volume_ma_20d']
    df['vol_dev_squared'] = df['vol_deviation'] ** 2
    df['vol_volatility'] = np.sqrt(df['vol_dev_squared'].rolling(window=20).sum())

    # Final Adjusted Momentum Score
    df['final_adj_momentum_score'] = df['combined_vol_adj_momentum'] / df['vol_volatility']

    # Intraday Movement Adjustment
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'].diff()
    df['intraday_adjustment'] = df['intraday_return'] * df['volume_change']

    # Final Adjusted Momentum Score with Intraday Adjustment
    df['final_adj_momentum_score'] = df['final_adj_momentum_score'] + df['intraday_adjustment']

    # Calculate 5-Day and 20-Day Simple Moving Average (SMA) of Close price
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    df['sma_20d'] = df['close'].rolling(window=20).mean()

    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    df['price_oscillator'] = df['sma_5d'] - df['sma_20d']

    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (3 * df['volume'])

    # Enhanced PMVI Score
    df['enhanced_pmvi_score'] = df['final_adj_momentum_score'] * df['vwap'] * df['price_oscillator']

    return df['enhanced_pmvi_score']
```
