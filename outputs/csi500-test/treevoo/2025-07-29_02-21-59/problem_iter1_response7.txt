```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns using Simple Moving Average (SMA) with a window of 5 days
    df['smoothed_returns'] = df['log_returns'].rolling(window=5).mean()
    
    # Volume Shock Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    # Calculate the standard deviation over a 5-day window for volume changes
    std_dev_vol = df['volume_change'].rolling(window=5).std()
    # Identify large changes in volume, assign 1 if shock, else 0
    df['volume_shock'] = np.where(np.abs(df['volume_change']) > 2 * std_dev_vol, 1, 0)
    
    # Multiply Smoothed Returns by Volume Shock Indicator
    df['factor'] = df['smoothed_returns'] * df['volume_shock']
    
    return df['factor']
```
