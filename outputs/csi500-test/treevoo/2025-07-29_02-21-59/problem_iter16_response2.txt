```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['high_low_ratio'] = (df['high'] - df['low']) / df['close']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_high_low_ratio'] = df['high_low_ratio'] * df['volume_change']

    # Adjust for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()

    # Volume-Adjusted Intraday Momentum
    df['volume_adjusted_momentum'] = (df['volume_weighted_return'] + df['adjusted_high_low_ratio']) / (df['volume_volatility'] + 1e-6)

    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Aggregate Log Returns over 20 days
    df['cumulative_log_return'] = df['log_return'].rolling(window=20).sum()
    
    # Weight by Volume Change
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Calculate Daily VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    df['vwap_ma_20'] = df['vwap'].rolling(window=20).mean()

    # Combined Alpha Factor
    df['combined_alpha'] = df['cumulative_log_return'] * df['vwap_ma_20'] + df['volume_adjusted_momentum']
    
    return df['combined_alpha']
```
