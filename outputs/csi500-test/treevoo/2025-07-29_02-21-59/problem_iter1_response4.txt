```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Sum of Daily Price Changes (using absolute value) over 20 days
    df['sum_daily_price_changes'] = df['daily_price_change'].abs().rolling(window=20).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap_daily'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Aggregate VWAP over 20 days
    df['vwap_20_day'] = df['vwap_daily'].rolling(window=20).mean()
    
    # PMI Score: Multiply Sum of Daily Price Changes by VWAP
    df['pmi_score'] = df['sum_daily_price_changes'] * df['vwap_20_day']
    
    return df['pmi_score']
```
