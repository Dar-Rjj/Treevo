```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return using Close price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Aggregate 20-Day Cumulative Sum of Daily Log Returns
    df['cum_log_return'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP) using High, Low, Close, and Volume
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # PMI Score: Multiply 20-Day Cumulative Log Returns by VWAP
    df['pmi_score'] = df['cum_log_return'] * df['vwap']
    
    return df['pmi_score']
```
