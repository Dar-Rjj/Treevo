```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Intraday Momentum
    df['close_to_open_return'] = df['open'] / df['close'].shift(1) - 1
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_intraday_high_low_ratio'] = df['intraday_high_low_ratio'] * df['volume_change']
    
    df['volume_moving_avg_20'] = df['volume'].rolling(window=20).mean()
    df['volume_std_20'] = df['volume'].rolling(window=20).std()
    df['volume_volatility_adjusted_momentum'] = df['volume_weighted_return'] / df['volume_std_20']

    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['cumulative_log_return_20'] = df['daily_log_return'].rolling(window=20).sum()
    
    df['intraday_return'] = df['close'] / df['open'] - 1
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_ema_20'] = df['vwap'].ewm(span=20, adjust=False).mean()

    # Combined Alpha Factor
    df['alpha_factor'] = (df['cumulative_log_return_20'] * df['vwap_ema_20']) + df['volume_volatility_adjusted_momentum']
    
    return df['alpha_factor']
```
