import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Intraday Momentum
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Adjust for Volume Volatility
    df['volume_adjusted_intraday_momentum'] = (df['volume_weighted_close_to_open'] 
                                               + df['intraday_high_low_ratio'] * df['volume_change'] 
                                               / df['volume_volatility'])
    
    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    df['cumulative_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    df['vwap'] = ((df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() 
                  / df['volume'].cumsum())
    df['vwap_ema'] = df['vwap'].ewm(span=20, adjust=False).mean()
    
    # Combined Alpha Factor
    df['combined_alpha_factor'] = (df['cumulative_log_returns'] * df['vwap_ema'] 
                                   + df['volume_adjusted_intraday_momentum'])
    
    return df['combined_alpha_factor']
