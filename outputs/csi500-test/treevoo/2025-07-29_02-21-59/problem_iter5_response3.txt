```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'] / df['close'].shift(1) - 1
    
    # Volume Adjusted Close-to-Open Return
    df['Volume_Adjusted_Close_to_Open'] = df['Close_to_Open_Return'] * df['volume']
    
    # Intraday High-Low Movement
    df['Intraday_High_Low_Movement'] = df['high'] - df['low']
    
    # Intraday High-Low Ratio
    df['Intraday_High_Low_Ratio'] = df['Intraday_High_Low_Movement'] / df['close']
    
    # Intraday Return
    df['Intraday_Return'] = df['close'] / df['open'] - 1
    
    # Volume Change
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Combine with Volume Adjusted Close-to-Open Return
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * (1 + df['Volume_Change'])
    df['Integrated_Factor'] = df['Volume_Adjusted_Close_to_Open'] + df['Adjusted_Intraday_Return']
    
    # Volume Moving Average
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    
    # Standard Deviation of Volume
    df['Volume_Volatility'] = df['volume'].rolling(window=20).std()
    
    # Final Alpha Factor adjusted for Volume Volatility
    df['Alpha_Factor'] = df['Integrated_Factor'] / df['Volume_Volatility']
    
    return df['Alpha_Factor']
```
