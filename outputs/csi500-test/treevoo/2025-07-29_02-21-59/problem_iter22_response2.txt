import pandas as pd

def heuristics_v2(df):
    # Volume-Adjusted Price Momentum
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_high_low_ratio'] = df['intraday_high_low_movement'] / df['close']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Adjust for Volume Volatility
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    df['volume_adjusted_price_momentum'] = (df['volume_weighted_close_to_open_return'] + 
                                            df['volume_adjusted_intraday_return']) / df['volume_volatility']

    # Cumulative Log Return with VWAP Integration
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: 0 if x <= 0 else np.log(x))
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum()
    
    df['volume_change_for_weighting'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_20_day_ma'] = df['vwap'].rolling(window=20).mean()
    
    df['weighted_intraday_momentum'] = df['intraday_high_low_ratio'] * df['intraday_return']
    
    # Combined and Enhanced Alpha Factor
    df['enhanced_alpha_factor'] = (df['cumulative_log_return'] * df['vwap_20_day_ma'] + 
                                   df['weighted_intraday_momentum']) * (df['volume_change_for_weighting'] + 
                                                                      df['amount_change'])
    
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['5_day_sma'] - df['20_day_sma']
    
    df['final_alpha_factor'] = df['enhanced_alpha_factor'] + df['price_oscillator']
    
    return df['final_alpha_factor'].dropna()
