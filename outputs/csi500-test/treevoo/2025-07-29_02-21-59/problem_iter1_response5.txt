def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_return = (df['close'].shift(1) / df['close'].shift(3)) - 1
    
    # Calculate Long-Term Price Momentum
    long_term_return = (df['close'].shift(1) / df['close'].shift(14)) - 1
    
    # Combine Short-Term and Long-Term Momentum
    avg_momentum = (short_term_return + long_term_return) / 2
    weighted_momentum = avg_momentum * abs(short_term_return)
    
    # Incorporate Volume Shock
    daily_volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    price_range = (df['high'] - df['low']) / df['close']
    
    # Combine Volume Shock and Momentum
    volume_shock_adjusted_momentum = weighted_momentum * daily_volume_change * (1 / price_range)
    
    # Final Alpha Factor
    alpha_factor = volume_shock_adjusted_momentum
    
    return alpha_factor
