```python
import pandas as pd

def heuristics_v2(df):
    lookback_window = 20
    short_smoothing_window = 5
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=lookback_window).mean()
    
    # Calculate Exponential Moving Average (EMA) of Close Prices
    ema_close = df['close'].ewm(span=lookback_window, adjust=False).mean()
    
    # Compute the Difference between EMA and SMA
    momentum_difference = ema_close - sma_close
    
    # Volume Adjust the Momentum
    volume_adjusted_momentum = momentum_difference * df['volume']
    
    # Optionally, apply a simple averaging or smoothing over a shorter period
    smoothed_volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=short_smoothing_window).mean()
    
    return smoothed_volume_adjusted_momentum
```
