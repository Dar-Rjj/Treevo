```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily close-to-close return
    df['close_to_close_return'] = df['close'].pct_change()

    # Sum of recent returns (last 5 days)
    df['sum_recent_returns'] = df['close_to_close_return'].rolling(window=5).sum()

    # Calculate average volume over the last 5 days
    df['average_volume'] = df['volume'].rolling(window=5).mean()

    # Ratio of current day's volume to average volume
    df['volume_ratio'] = df['volume'] / df['average_volume']

    # Weighted relative momentum
    df['weighted_relative_momentum'] = df['sum_recent_returns'] * df['volume_ratio']

    # Calculate close-to-open return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Combined volume-adjusted momentum
    df['combined_volume_adjusted_momentum'] = df['weighted_relative_momentum'] * df['close_to_open_return']

    # Adjust for volume volatility
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_average']
    df['squared_volume_deviation'] = df['volume_deviation'] ** 2
    df['volume_volatility'] = df['squared_volume_deviation'].rolling(window=20).sum().apply(lambda x: x ** 0.5)

    # Final adjusted momentum score
    df['final_adjusted_momentum_score'] = df['combined_volume_adjusted_momentum'] / df['volume_volatility']

    # Intraday movement adjustment
    df['intraday_high_low_movement'] = df['high'] - df['low']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_adjustment'] = df['intraday_return'] * df['volume_change']

    # Combine adjustments
    df['final_adjusted_momentum_with_intraday'] = df['final_adjusted_momentum_score'] + df['intraday_adjustment']

    return df['final_adjusted_momentum_with_intraday']
```
