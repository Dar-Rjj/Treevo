import pandas as pd

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    log_returns = df['close'].apply(lambda x: np.log(x / df['close'].shift(1).fillna(x)))
    
    # Smoothing Logarithmic Returns
    smoothed_log_returns = log_returns.rolling(window=5).mean()
    
    # Volume Shock Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_change_std = volume_change.rolling(window=5).std()
    shock_indicator = (abs(volume_change) > 2 * volume_change_std).astype(int)
    
    # Apply Shock to Smoothed Returns
    final_factor = shock_indicator * smoothed_log_returns
    
    return final_factor
