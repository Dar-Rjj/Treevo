```python
import pandas as pd

def heuristics_v2(df):
    # Compute Today's Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Determine Sign of Volume Change
    df['volume_acceleration'] = df['volume_diff'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Calculate Close Price Change
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Determine Sign of Price Change
    df['price_momentum'] = df['close_price_change'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Combine Volume and Price Momentum
    def confirm_trend(volume_accel, price_moment):
        if volume_accel == 1 and price_moment == 1:
            return 1  # Positive Trend
        elif volume_accel == -1 and price_moment == -1:
            return -1  # Negative Trend
        else:
            return 0  # No clear trend
    
    df['factor_value'] = df.apply(lambda x: confirm_trend(x['volume_acceleration'], x['price_momentum']), axis=1)
    
    return df['factor_value']
```
