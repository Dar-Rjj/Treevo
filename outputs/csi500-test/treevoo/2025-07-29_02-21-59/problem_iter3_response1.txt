```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_high_low_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_high_low_ratio = intraday_high_low_movement / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'].diff(1).fillna(0)
    
    # Calculate Amount Change
    amount_change = df['amount'].diff(1).fillna(0)
    
    # Weight by Volume and Amount Change
    factor = intraday_return * (volume_change + amount_change)
    
    return factor
```
