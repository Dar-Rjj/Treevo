```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['return'] = df['close'].pct_change()
    
    # Sum of Recent Returns (last 5 days)
    df['sum_recent_returns'] = df['return'].rolling(window=5).sum()
    
    # Calculate Average Volume over the last 5 days
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Weighted Relative Momentum
    df['weighted_momentum'] = df['sum_recent_returns'] * df['volume_ratio']
    
    return df['weighted_momentum']
```
