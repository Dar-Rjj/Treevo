import pandas as pd

def heuristics_v2(df):
    # Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume and Amount Combined
    volume_amount_combined = df['volume'] * df['amount']
    
    # Volume and Amount Weighted Close-to-Open Return
    weighted_close_to_open_return = close_to_open_return * volume_amount_combined
    
    # Intraday High-Low Movement
    intraday_high_low_movement = (df['high'] - df['low']) / df['low']
    
    # Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust Intraday High-Low Movement by Volume Change
    adjusted_intraday_high_low_movement = intraday_high_low_movement * (1 + volume_change)
    
    # Sum of Recent Close-to-Close Returns
    sum_recent_close_to_close_returns = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Ratio of Current Day's Volume to Average Volume
    average_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Relative Momentum
    weighted_relative_momentum = sum_recent_close_to_close_returns * volume_ratio
    
    # Combined Volume-Adjusted Momentum
    combined_volume_adjusted_momentum = weighted_relative_momentum * adjusted_intraday_high_low_movement
    
    # Volume Moving Average
    volume_moving_average = df['volume'].rolling(window=20).mean()
    
    # Standard Deviation of Volume
    volume_volatility = df['volume'].rolling(window=20).std()
    
    # Final Adjusted Momentum Score
    final_adjusted_momentum_score = combined_volume_adjusted_momentum / volume_volatility
    
    # Intraday Movement Adjustment
    intraday_movement_adjustment = intraday_return * volume_change
    final_adjusted_momentum_score += intraday_movement_adjustment
    
    return final_adjusted_momentum_score
