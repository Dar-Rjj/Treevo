```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['High_Low_Movement'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Calculate Amount Change
    df['Amount_Change'] = df['amount'].pct_change()
    
    # Combine Intraday Factors
    df['Combined_Intraday_Factor'] = (
        df['Intraday_Return'] + 
        df['Volume_Change'] + 
        df['Amount_Change'] + 
        df['Intraday_Volatility']
    )
    
    # Adjust Intraday Return
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] * df['Combined_Intraday_Factor']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Calculate Volume Weighted Close-to-Open Return
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Sum of Daily Price Changes
    df['Sum_Daily_Price_Changes'] = df['close'].diff().rolling(window=20).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * (df['volume'] / df['volume'].rolling(window=20).sum())
    
    # PMI Score
    df['PMI_Score'] = df['Sum_Daily_Price_Changes'] * df['VWAP']
    
    # Cumulative Sum of Daily Log Returns
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Cumulative_Log_Returns'] = df['Log_Returns'].rolling(window=20).sum()
    
    # Calculate 20-Day Momentum
    df['Momentum_20Day'] = df['close'] - df['close'].shift(20)
    
    # Introduce New Factor: Intraday Range to Volume Ratio
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Range_to_Volume_Ratio'] = df['Intraday_Range'] / df['volume']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['PMI_Score'] * df['Cumulative_Log_Returns'] +
        df['Adjusted_Intraday_Return'] +
        df['Volume_Weighted_Close_to_Open_Return'] +
        df['Momentum_20Day'] +
        df['Intraday_Range_to_Volume_Ratio']
    )
    
    return df['Final_Alpha_Factor']
```
