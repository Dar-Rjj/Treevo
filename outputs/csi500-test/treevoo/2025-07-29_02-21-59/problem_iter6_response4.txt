```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()

    # Sum of Recent Returns
    df['sum_recent_returns'] = df['close_to_close_return'].rolling(window=5).sum()

    # Calculate Average Volume over Period
    df['average_volume'] = df['volume'].rolling(window=5).mean()

    # Ratio of Current Day's Volume to Average Volume
    df['volume_ratio'] = df['volume'] / df['average_volume']

    # Calculate Price Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['price_volatility'] = (df['high_low_diff'] + df['high_low_diff'].shift(1))

    # Combine Volume and Volatility
    df['combined_weight'] = df['volume_ratio'] * df['price_volatility']

    # Incorporate Price Change Magnitude
    df['abs_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['sum_recent_abs_price_changes'] = df['abs_price_change'].rolling(window=5).sum()

    # Weighted Relative Momentum
    df['weighted_momentum'] = (df['sum_recent_returns'] * df['combined_weight']) + df['sum_recent_abs_price_changes']

    return df['weighted_momentum'].dropna()
```
