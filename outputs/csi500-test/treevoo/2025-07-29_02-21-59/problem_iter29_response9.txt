```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the weighted sum of these three features
    df['factor'] = (df['close_change'] * 0.5) + (df['high_low_diff'] * 0.3) + (df['volume_amount_ratio'] * 0.2)
    
    # Return the factor as a pandas Series
    return df['factor']
```
