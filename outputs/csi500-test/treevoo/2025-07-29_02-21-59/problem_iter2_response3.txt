```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    intraday_movement = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    intraday_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    weighted_intraday_return = intraday_return * volume_change
    
    # Calculate Daily Close-to-Close Return
    daily_close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum of Recent Returns
    recent_returns_sum = daily_close_to_close_return.rolling(window=5).sum()
    
    # Calculate Average Volume over Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Ratio of Current Day's Volume to Average Volume
    volume_ratio = df['volume'] / average_volume
    
    # Weighted Combined Momentum
    combined_momentum = (weighted_intraday_return + recent_returns_sum * volume_ratio) / 2
    
    return combined_momentum
```
