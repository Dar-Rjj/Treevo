```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns over 20 days
    df['cum_log_returns'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Ratio
    df['high_low_ratio'] = df['intraday_movement'] / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Weight Intraday Return by Volume Change
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_change']
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume'] / df['volume']
    
    # Calculate 20-Day Moving Average VWAP
    df['vwap_20_sma'] = df['vwap'].rolling(window=20).mean()
    
    # Weighted Intraday Momentum
    df['weighted_intraday_momentum'] = df['high_low_ratio'] * df['intraday_return']
    
    # Combined Alpha Factor
    df['alpha_factor'] = (df['cum_log_returns'] * df['vwap_20_sma']) + df['weighted_intraday_momentum']
    
    return df['alpha_factor']
```
