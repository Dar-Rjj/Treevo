```python
def heuristics_v2(df):
    # Calculate the difference between close and open prices
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate percentage change from open to close
    df['close_open_pct_change'] = ((df['close'] - df['open']) / df['open']) * 100
    
    # Incorporate previous day's close price
    df['close_prev_close_diff'] = df['close'].diff()
    
    # Measure intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Assess upside potential relative to previous close
    df['upside_potential'] = df['high'] - df['close'].shift(1)
    
    # Assess downside risk relative to previous close
    df['downside_risk'] = df['close'].shift(1) - df['low']
    
    # Evaluate closing price momentum with volume
    df['close_open_vol_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Evaluate change in volume
    df['volume_change'] = df['volume'].diff()
    
    # Compute the average trading price
    df['avg_price'] = df['amount'] / df['volume']
    
    # Examine the change in average trading price
    df['avg_price_change'] = df['avg_price'].diff()
    
    # Compare current day's high and low with previous day's range
    df['high_prev_high_diff'] = df['high'].diff()
    df['low_prev_low_diff'] = df['low'].diff()
    
    # Analyze the ratio of current day's volume to previous day's volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['close_open_diff'] + 
        df['close_open_pct_change'] + 
        df['close_prev_close_diff'] + 
        df['intraday_range'] + 
        df['upside_potential'] - 
        df['downside_risk'] + 
        df['close_open_vol_momentum'] + 
        df['volume_change'] + 
        df['avg_price_change'] + 
        df['high_prev_high_diff'] - 
        df['low_prev_low_diff'] + 
        df['volume_ratio']
    )
    
    return alpha_factor
```
