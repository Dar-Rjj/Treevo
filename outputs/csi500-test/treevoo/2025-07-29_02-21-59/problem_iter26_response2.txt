```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['volume_amount_combined'] = df['volume'] * df['amount']
    df['volume_amount_weighted_close_to_open'] = df['close_to_open_return'] * df['volume_amount_combined']
    
    # Intraday High-Low Movement
    df['intraday_high_low_movement'] = (df['high'] - df['low']) / df['low']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['adjusted_intraday_high_low'] = df['intraday_high_low_movement'] + df['intraday_return'] * df['volume_change']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_diff'] = df['volume'] - df['volume_moving_avg']
    df['volume_volatility'] = df['volume_diff'].rolling(window=20).std(ddof=0)  # ddof=0 for population std
    df['volatility_adjusted_factor'] = (df['volume_amount_weighted_close_to_open'] + df['adjusted_intraday_high_low']) / df['volume_volatility']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_return'] + df['volume_change'] + df['amount_change'] + df['intraday_volatility']
    
    # Adjust Intraday Return
    df['adjusted_intraday_return'] = df['intraday_return'] * (df['intraday_volatility'] + df['volume_change'] + df['amount_change'])
    
    # Calculate Volume Weighted Close-to-Open Return
    df['volume_weighted_close_to_open'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Sum of Daily Price Changes
    df['sum_daily_price_changes'] = df['daily_price_change'].rolling(window=20).sum()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['volume']
    df['vwap'] = df['vwap'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # PMI Score
    df['pmi_score'] = df['sum_daily_price_changes'] * df['vwap']
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Cumulative Sum of Daily Log Returns
    df['cum_log_returns'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate 20-Day Momentum
    df['momentum_20d'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Introduce New Factor: Intraday Range to Volume Ratio
    df['intraday_range'] = df['high'] - df['low']
    df['range_to_volume_ratio'] = df['intraday_range'] / df['volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['pmi_score'] * df['cum_log_returns'] + 
                          df['adjusted_intraday_return'] + 
                          df['volume_weighted_close_to_open'] + 
                          df['momentum_20d'] + 
                          df['range_to_volume_ratio'])
    
    return df['alpha_factor']
```
