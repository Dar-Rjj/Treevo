```python
def heuristics_v2(df):
    # Calculate the price movement from the previous close to the current open
    df['overnight_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate the intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure as the difference between the amount and the volume weighted by the close price
    df['buying_pressure'] = df['amount'] - df['volume'] * df['close']
    
    # Alpha factor: overnight return divided by the intraday range, multiplied by the buying pressure
    alpha_factor = (df['overnight_return'] / df['intraday_range']) * df['buying_pressure']
    
    # Return the alpha factor as a pandas Series
    return alpha_factor
```
