```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Movement
    df['high_low'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume and Amount Change
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    
    # Combine Intraday Factors
    df['combined_intraday_factor'] = df['intraday_return'] * (df['intraday_volatility'] + df['volume_change'] + df['amount_change'])
    
    # Adjust Intraday Return
    df['adjusted_intraday_return'] = df['intraday_return'] * (df['intraday_volatility'] + df['volume_change'] + df['amount_change'])
    
    # Volume-Adjusted Intraday Momentum
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    df['intraday_high_low_ratio'] = df['high_low'] / df['close']
    df['volume_adjusted_intraday_momentum'] = df['intraday_high_low_ratio'] * df['volume_change'] + df['adjusted_intraday_return']
    
    # Adjust for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_std'] = df['volume'].rolling(window=20).std()
    df['volume_volatility_adjusted'] = (df['volume'] - df['volume_moving_avg']) / df['volume_std']
    df['volume_adjusted_intraday_momentum'] = df['volume_adjusted_intraday_momentum'] * df['volume_volatility_adjusted']
    
    # Cumulative Momentum with VWAP Integration
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['cumulative_log_returns'] = df['daily_log_return'].rolling(window=20).sum()
    df['intraday_high_low_ratio'] = df['high_low'] / df['close']
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) / df['open'] * df['volume_change']
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['ema_vwap_20'] = df['vwap'].ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['cumulative_log_returns'] * df['ema_vwap_20'] + 
                                df['adjusted_intraday_return'] + 
                                df['volume_adjusted_intraday_momentum'])
    
    return df['final_alpha_factor'].dropna()
```
