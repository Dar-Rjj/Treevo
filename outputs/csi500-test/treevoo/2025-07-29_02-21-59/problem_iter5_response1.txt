```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Aggregate Daily Price Changes
    df['agg_price_change'] = df['price_change'].rolling(window=20).sum()
    
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Smoothing Logarithmic Returns with SMA
    df['smoothed_returns'] = df['log_returns'].rolling(window=5).mean()
    
    # Volume Shock Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_std = df['volume_change'].rolling(window=5).std()
    df['volume_shock'] = (np.abs(df['volume_change']) > 2 * volume_std).astype(int)
    
    # Multiply Smoothed Returns by Volume Shock Indicator
    df['shock_returns'] = df['smoothed_returns'] * df['volume_shock']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum()
    
    # PMI and Volume Shock Score
    df['pmi_score'] = df['agg_price_change'] * df['vwap']
    df['combined_score'] = df['pmi_score'] + df['shock_returns']
    
    return df['combined_score'].dropna()
```
