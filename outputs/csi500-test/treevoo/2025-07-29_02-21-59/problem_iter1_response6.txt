```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Momentum Factor: Sum of daily returns over 20 days
    df['momentum_20d'] = df['daily_return'].rolling(window=20).sum()
    
    # Volume Moving Average over 20 days
    df['vol_ma_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the standard deviation of volume
    df['vol_volatility'] = ((df['volume'] - df['vol_ma_20d']) ** 2).rolling(window=20).sum() ** 0.5
    
    # Adjust momentum for volume volatility
    df['vam_factor'] = df['momentum_20d'] / df['vol_volatility']
    
    return df['vam_factor'].dropna()
```
