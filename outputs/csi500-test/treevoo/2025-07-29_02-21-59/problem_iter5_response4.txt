```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Combine Volume and Amount
    combined_volume_amount = (df['volume'] + df['amount']) / 2
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Multiply by Weights
    weighted_factor = close_to_open_return * combined_volume_amount * close_open_spread
    
    return weighted_factor
```
