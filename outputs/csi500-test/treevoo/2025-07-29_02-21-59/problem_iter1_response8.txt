```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return using close price
    df['daily_return'] = df['close'].pct_change()

    # Sum of returns over a period (e.g., last 5 days)
    period = 5
    df['momentum_factor'] = df['daily_return'].rolling(window=period).sum()

    # Calculate price volatility using high and low prices
    df['price_volatility'] = (df['high'] - df['low']).rolling(window=period).sum()

    # Combine volume and volatility
    df['combined_weight'] = df['volume'] * df['price_volatility']

    # Calculate weighted average momentum
    df['weighted_momentum'] = df['momentum_factor'] / df['combined_weight']

    # Return the factor values
    return df['weighted_momentum'].dropna()
```
