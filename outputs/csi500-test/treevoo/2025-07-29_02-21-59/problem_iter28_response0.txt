import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Integrate Intraday High-Low Movement
    df['intraday_high_low_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Adjust by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_high_low_ratio'] * df['volume_change'] + df['intraday_return']
    
    # Adjust for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_std_20'] = df['volume'].rolling(window=20).std()
    df['volume_volatility_adjustment'] = (df['volume'] - df['volume_ma_20']) / df['volume_std_20']
    
    # Integrated Intraday Momentum with Volume Adjustment
    df['integrated_intraday_momentum'] = df['volume_weighted_return'] + df['intraday_momentum'] * df['volume_volatility_adjustment']
    
    # Calculate Daily Log Return
    df['daily_log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: 0 if pd.isna(x) else x).apply(np.log)
    
    # Cumulative Sum of Daily Log Returns
    df['cumulative_log_return'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Calculate Daily Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-Day Moving Average VWAP
    df['vwap_ma_20'] = df['vwap'].rolling(window=20).mean()
    
    # Calculate PMI Score
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['pmi_score'] = df['daily_price_change'].rolling(window=20).sum() * df['vwap_ma_20']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['pmi_score'] * df['cumulative_log_return'] +
                          df['integrated_intraday_momentum'] +
                          df['cumulative_log_return'])
    
    # Price Oscillator: Subtract 20-Day SMA from 5-Day SMA
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['price_oscillator'] = df['sma_5'] - df['sma_20']
    
    # Final Alpha Factor with Price Oscillator
    df['final_alpha_factor'] = df['alpha_factor'] + df['price_oscillator']
    
    return df['final_alpha_factor']
