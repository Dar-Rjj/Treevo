import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Adjusted Intraday Momentum
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    intraday_high_low_movement = df['high'] / df['low'] - 1
    intraday_return = df['close'] / df['open'] - 1
    
    volume_adjusted_intraday_momentum = (volume_weighted_close_to_open + intraday_high_low_movement + intraday_return) / 3

    # Combined Cumulative Log Return with VWAP Integration
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    cumulative_log_return = daily_log_return.rolling(window=20).sum()

    intraday_high_low_movement = df['high'] / df['low'] - 1
    intraday_return = df['close'] / df['open'] - 1
    
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_20_day_ma = vwap.rolling(window=20).mean()
    
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    price_oscillator = sma_5 - sma_20
    
    combined_factor = (volume_adjusted_intraday_momentum + cumulative_log_return + vwap_20_day_ma + price_oscillator) / 4
    return combined_factor
