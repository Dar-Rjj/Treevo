```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on price and volume data.
    
    Parameters:
    - df: pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    - pandas Series indexed by (date) representing the factor values
    """
    # Calculate the weighted average price using amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / df['volume']
    
    # Calculate the high-low price spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate the return of the weighted price
    df['weighted_return'] = df['weighted_price'].pct_change()
    
    # Calculate the momentum of the high-low spread
    df['momentum_high_low'] = df['high_low_spread'].pct_change(periods=5)
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = df['weighted_return'] * df['momentum_high_low']
    
    return df['alpha_factor']
```
