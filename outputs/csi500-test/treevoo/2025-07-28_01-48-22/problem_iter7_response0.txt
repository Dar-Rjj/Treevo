```python
import pandas as pd

def heuristics(data):
    # Volume Difference
    volume_diff = data['volume'].diff(1)
    
    # Avg High-Low Range (5 days)
    high_low_range = data['high'] - data['low']
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Close Change
    close_change = data['close'].diff(1)
    
    # Combine Factors
    factor = (volume_diff * 0.5) + (avg_high_low_range * 0.3) + close_change
    
    return factor
```
