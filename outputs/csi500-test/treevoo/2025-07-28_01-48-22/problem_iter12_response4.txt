```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']

    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(20)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Incorporate Volume Influence
    weighted_average_volume = (df['volume'].rolling(window=20).apply(lambda x: sum(x / (np.arange(1, 21))))).fillna(0)
    combined_momentum *= weighted_average_volume
    
    # Incorporate Advanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_daily_range = daily_high_low_range.ewm(span=15, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    adjusted_combined_momentum = combined_momentum * volatility_multiplier
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(3)
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=7).mean()
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(3)
    
    # Integrate Hybrid Factors
    hybrid_factors = 0.7 * volume_difference + 0.4 * avg_high_low_range + close_price_change
    
    # Final Factor
    final_factor = (adjusted_combined_momentum + hybrid_factors) * adjusted_combined_momentum
    
    return final_factor
```
