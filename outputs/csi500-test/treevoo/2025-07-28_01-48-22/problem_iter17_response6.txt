import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, which is the typical price multiplied by the volume traded
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-period negative and positive money flow
    df['positive_money_flow'] = money_flow.where(typical_price > typical_price.shift(1), 0)
    df['negative_money_flow'] = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Compute the money ratio
    n = 14
    positive_money_flow_14 = df['positive_money_flow'].rolling(window=n).sum()
    negative_money_flow_14 = df['negative_money_flow'].rolling(window=n).sum()
    money_ratio = positive_money_flow_14 / negative_money_flow_14
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_ratio))
    
    # Generate a novel alpha factor that looks at the smoothed change in the money flow index over time
    smooth_factor = money_flow_index.ewm(span=5, adjust=False).mean().pct_change(periods=1)
    
    return smooth_factor
