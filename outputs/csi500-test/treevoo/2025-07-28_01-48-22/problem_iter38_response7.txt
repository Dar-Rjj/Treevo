```python
import pandas as pd

def heuristics_v2(df):
    """
    Given a DataFrame with (open, high, low, close, amount, volume) prices,
    compute a novel and interpretable alpha factor.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the factor values.
    """
    # Calculate the average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the ratio of typical price to average of open and close
    price_ratio = typical_price / avg_open_close
    
    # Compute the difference between current and previous day's volume
    volume_diff = df['volume'].diff(1)
    
    # Compute the ratio of current day's amount to previous day's amount
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Heuristic factor: Combine the price ratio and volume diff, scaled by amount ratio
    heuristic_factor = price_ratio * volume_diff * amount_ratio
    
    return heuristic_factor
```
