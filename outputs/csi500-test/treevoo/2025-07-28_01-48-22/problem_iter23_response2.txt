```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Smooth Combined Momentum with 5-day SMA
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Weighted Average Volume over the last 15 days
    weights = 1 / np.arange(1, 16)
    weighted_avg_volume = (df['volume'].rolling(window=15).apply(lambda x: np.sum(x * weights), raw=True) / np.sum(weights))
    
    # Adjust Smoothed Momentum by Weighted Average Volume
    adjusted_momentum = smoothed_momentum * weighted_avg_volume
    
    # Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    # Adjust Combined Momentum by Advanced Volatility
    volatility_adjusted_momentum = adjusted_momentum * volatility_multiplier
    
    # Price Range and Volatility
    price_range_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Average High-Low Range for the past 5 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = 0.5 * volume_difference + 0.3 * avg_high_low_range + close_price_change
    
    # Adjusted Intraday Return
    adjusted_intraday_return = intraday_return * weighted_avg_volume
    
    # Final Factor
    final_factor = (volatility_adjusted_momentum + price_range_volatility + hybrid_factor) * adjusted_intraday_return
    
    return final_factor
```
