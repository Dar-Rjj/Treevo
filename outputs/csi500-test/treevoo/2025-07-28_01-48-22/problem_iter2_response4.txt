```python
import pandas as pd

def heuristics_v2(df, n_days=30, k=2):
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    df['prev_day_spread'] = df['intraday_spread'].shift(1)
    
    # Compare Intraday and Previous Day Spreads
    df['intraday_spread_ratio'] = df['intraday_spread'] / df['prev_day_spread']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Identify Volume Spikes
    df['volume_mean'] = df['volume'].rolling(window=n_days).mean()
    df['volume_std'] = df['volume'].rolling(window=n_days).std()
    df['volume_spike_indicator'] = (df['volume'] > df['volume_mean'] + k * df['volume_std']).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['multiplied_combined_return'] = df['combined_return'] * df['volume_spike_indicator']
    
    # Combine Signals for Alpha Factor
    weight_intraday_spread_ratio = 0.4
    weight_open_to_close_return = 0.3
    weight_multiplied_combined_return = 0.3
    
    df['alpha_factor'] = (weight_intraday_spread_ratio * df['intraday_spread_ratio'] +
                          weight_open_to_close_return * df['open_to_close_return'] +
                          weight_multiplied_combined_return * df['multiplied_combined_return'])
    
    return df['alpha_factor']
```
