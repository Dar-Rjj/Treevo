```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Combine Momentum Components
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Adjust Momentum
    df['volume_volatility'] = df['volume'].pct_change().rolling(window=20).std()
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = df['momentum_20'] / df['volume_volatility'] * df['average_volume']
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['high_low_over_avg'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_return'] + df['open_to_close_return']
    
    # Identify Volume Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=20).mean() + 
                          2 * df['volume'].rolling(window=20).std()).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['spike_adjusted_returns'] = df['combined_returns'] * df['volume_spike']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate VWAP
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['volume_to_vwap_ratio'] = df['volume'] / df['vwap']
    
    # Integrate Volume and Price Information
    df['integrated_info'] = (0.5 * df['volume_diff'] + 
                             0.3 * df['avg_high_low_range'] + 
                             (df['close'] - df['close'].shift(1)))
    
    # Advanced Volatility Adjustment
    df['ema_high_low_range'] = df['high_low_range'].ewm(span=10, adjust=False).mean()
    df['volatility_multiplier'] = 1 / (df['ema_high_low_range'] + 1)
    df['volatility_adjusted_momentum'] = df['smoothed_momentum'] * df['volatility_multiplier']
    
    # Final Factor
    df['final_factor'] = (df['volatility_adjusted_momentum'] + 
                          df['relative_strength'] + 
                          df['volume_diff'] + 
                          df['volume_to_vwap_ratio'] + 
                          df['integrated_info'] + 
                          (df['high_low_over_avg'] * df['volume_diff'] * 
                           df['volume_to_vwap_ratio'] * 
                           (df['close'] - df['close'].shift(1))))
    
    return df['final_factor']
```
