import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['ShortTermReturn'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['LongTermReturn'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    df['DailyLogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    df['20DayMomentum'] = df['DailyLogReturn'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['IntradayMomentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    df['HistoricalMomentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Combine Intraday and Historical Momentum Components
    df['CombinedMomentum'] = df['IntradayMomentum'] + df['HistoricalMomentum']
    
    # Smooth Combined Momentum
    df['SmoothedMomentum'] = df['CombinedMomentum'].ewm(span=7, adjust=False).mean()
    
    # 20-Day Volume Volatility
    df['VolumeVolatility'] = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    df['AverageVolume'] = df['volume'].rolling(window=10).mean()
    
    # Adjust Momentum
    df['AdjustedMomentum'] = df['20DayMomentum'] / df['VolumeVolatility'] * df['AverageVolume']

    df['RelativeStrength'] = df['ShortTermReturn'] - df['LongTermReturn']

    df['HighLowRange'] = df['high'] - df['low']

    df['AverageHighLowRange'] = df['HighLowRange'].rolling(window=5).mean()

    df['HighLowOverAverage'] = df['HighLowRange'] / df['AverageHighLowRange']

    df['OpenToCloseReturn'] = (df['close'] - df['open']) / df['open']

    df['CombinedReturn'] = df['IntradayReturn'] + df['OpenToCloseReturn']

    df['CombinedReturn_Spike'] = df['CombinedReturn'] * df['VolumeSpike']
