```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous close to the current open
    prev_close = df['close'].shift(1)
    open_change = (df['open'] - prev_close) / prev_close
    
    # Calculate the price range (high - low)
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure (amount/volume)
    buying_pressure = df['amount'] / df['volume']
    
    # Calculate the factor value
    # The idea is to combine the price movement, price range, and buying pressure
    # to create a heuristic that captures momentum, volatility, and volume
    factor_value = open_change + price_range + buying_pressure
    
    return factor_value
```
