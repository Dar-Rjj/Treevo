```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Weighted Average Volume
    weights = 1 / np.arange(1, 16)
    weighted_average_volume = (df['volume'].rolling(window=15).apply(lambda x: np.sum(x * weights), raw=True) 
                               / np.sum(weights))
    
    # Adjust Combined Momentum by Weighted Average Volume
    adjusted_combined_momentum = combined_momentum * weighted_average_volume
    
    # Advanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_daily_range = daily_high_low_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    # Adjust Combined Momentum by Advanced Volatility
    advanced_adjusted_combined_momentum = adjusted_combined_momentum * volatility_multiplier
    
    # Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Average High-Low Range for past 5 days
    average_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factors = 0.5 * volume_difference + 0.3 * average_high_low_range + close_price_change
    
    # Final Factor
    final_factor = (hybrid_factors * advanced_adjusted_combined_momentum).dropna()
    
    return final_factor
```
