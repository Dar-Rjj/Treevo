```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['crossover_signal'] = (df['short_ema'] > df['long_ema']).astype(int)

    # Volatility Measure
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()

    # Enhanced Price Dynamics
    df['alpha_factor_1'] = df['high'] - df['low']
    df['alpha_factor_2'] = df['close'] / df['open']
    df['alpha_factor_7'] = (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])

    # Volume Analysis Factors
    df['vwap'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    df['alpha_factor_3'] = df['volume'] / (df['high'] - df['low'])
    df['alpha_factor_4'] = df['amount'] / (df['high'] - df['low'])
    df['alpha_factor_8'] = (df['volume'] - df['volume'].shift(5)) / df['amount']

    # Market Strength and Sentiment
    gains = df['close'].pct_change().apply(lambda x: x if x > 0 else 0).rolling(window=14).mean()
    losses = -df['close'].pct_change().apply(lambda x: x if x < 0 else 0).rolling(window=14).mean()
    rs = gains / losses
    df['rsi'] = 100 - (100 / (1 + rs))

    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    money_flow_ratio = positive_flow / negative_flow
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))

    # Enhanced Momentum and Trend
    df['alpha_factor_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['alpha_factor_9'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['alpha_factor_6'] = ((df['open'] < df['close']) * 1 + (df['open'] > df['close']) * -1).fillna(0)
    df['alpha_factor_10'] = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))

    # Advanced Volatility Measures
    df['alpha_factor_11'] = (df['high'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    df['alpha_factor_12'] = (df['high'] - df['low']) / (df['high'].shift(5) - df['low'].shift(5))
    df['alpha_factor_13'] = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))

    return df['alpha_factor_1'], df['alpha_factor_2'], df['alpha_factor_3'], df['alpha_factor_4'], \
           df['alpha_factor_5'], df['alpha_factor_6'], df['alpha_factor_7'], df['alpha_factor_8'], \
           df['alpha_factor_9'], df['alpha_factor_10'], df['alpha_factor_11'], df['alpha_factor_12'], \
           df['alpha_factor_13']
```
