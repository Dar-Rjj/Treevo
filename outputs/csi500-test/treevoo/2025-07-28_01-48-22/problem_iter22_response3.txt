import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_close_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    k = 2  # Threshold for volume spike
    volume_spike = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_vol_spike = combined_return * volume_spike
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Calculate VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Volume-to-VWAP Ratio
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    close_change = df['close'] - df['close'].shift(1)
    
    # Combine Indicators
    intermediate_result = high_low_over_avg * volume_diff * volume_to_vwap_ratio * close_change
    
    # Weight by Volume and Volatility
    volume_avg = df['volume'].rolling(window=5).mean()
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    weighted_relative_strength = relative_strength * (volume_avg * volatility)
    
    # Final Alpha Factor
    alpha_factor = intermediate_result * weighted_relative_strength
    
    return alpha_factor
