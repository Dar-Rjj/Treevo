```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    df['prev_day_spread'] = df['intraday_spread'].shift(1)
    
    # Compare Intraday and Previous Day Spreads
    df['intraday_spread_ratio'] = df['intraday_spread'] / df['prev_day_spread']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # Example value for K
    df['volume_spike'] = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['multiplied_combined_return'] = df['open_to_close_return'] * df['volume_spike']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Average High-Low Range
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['high_low_over_avg'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Integrate Moving Averages
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_5_over_20'] = df['sma_5'] / df['sma_20']
    df['sma_20_over_50'] = df['sma_20'] / df['sma_50']
    
    # Combine Signals for Alpha Factor
    weights = {
        'intraday_spread_ratio': 0.2,
        'open_to_close_return': 0.1,
        'multiplied_combined_return': 0.1,
        'high_low_over_avg': 0.15,
        'volume_change': 0.1,
        'close_to_prev_close_return': 0.1,
        'sma_5_over_20': 0.1,
        'sma_20_over_50': 0.15
    }
    
    alpha_factor = sum(weights[component] * df[component] for component in weights)
    
    return alpha_factor
```
