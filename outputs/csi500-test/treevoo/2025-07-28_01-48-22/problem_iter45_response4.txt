```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    df['20_day_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Sum of Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=7).mean()
    
    # 20-Day Volume Volatility
    df['volume_volatility'] = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    
    # Average Daily Range
    df['average_daily_range'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Volatility Multiplier
    df['volatility_multiplier'] = 1 / (df['average_daily_range'] + 1)
    
    # Adjusted Smoothed Momentum
    df['adjusted_smoothed_momentum'] = (df['smoothed_momentum'] / df['volume_volatility'] * df['average_volume']) * df['volatility_multiplier']
    
    # Volume-Price Indicator
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['volume_price_indicator'] = 0.6 * df['volume_difference'] + 0.4 * (df['high'] - df['low']) + (df['close'] - df['close'].shift(1))
    
    # Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Volume Spikes
    df['volume_std_dev'] = df['volume'].rolling(window=20).std()
    df['volume_spike_indicator'] = (df['volume'] > (df['volume'].mean() + 2 * df['volume_std_dev'])).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    df['relative_strength_by_volume_spike'] = df['relative_strength'] * df['volume_spike_indicator']
    
    # High-Low Range Over Average
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_over_average'] = df['high_low_range'] / df['average_daily_range']
    
    # Final Factor
    df['final_factor'] = (
        0.5 * df['adjusted_smoothed_momentum'] +
        0.2 * df['volume_price_indicator'] +
        0.2 * df['relative_strength_by_volume_spike'] +
        0.1 * df['high_low_range_over_average']
    )
    
    return df['final_factor']
```
