```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    momentum_20 = daily_log_return.rolling(window=20).sum()
    
    # Combine Momentum Components
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust Momentum
    volume_volatility_20 = df['volume'].diff().rolling(window=20).std()
    average_volume = df['volume'].rolling(window=10).mean()
    adjusted_momentum = momentum_20 / volume_volatility_20 * average_volume
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    volume_spike_indicator = (df['volume'] > (df['volume'].rolling(window=20).mean() + 2 * df['volume'].rolling(window=20).std())).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    volume_spike_adjusted = combined_returns * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-to-VWAP Ratio
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    integrated_info = 0.5 * volume_difference + 0.3 * avg_high_low_range + (df['close'] - df['close'].shift(1))
    
    # Advanced Volatility Adjustment
    ema_high_low_range = high_low_range.ewm(span=10).mean()
    volatility_multiplier = 1 / (ema_high_low_range + 1)
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Final Factor
    final_factor = (
        adjusted_smoothed_momentum +
        relative_strength +
        volume_difference +
        vwap_ratio +
        (high_low_over_avg * volume_difference * vwap_ratio * (df['close'] - df['close'].shift(1)))
    )
    
    return final_factor
```
