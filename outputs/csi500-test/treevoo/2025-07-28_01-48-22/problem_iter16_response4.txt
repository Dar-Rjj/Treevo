import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_open_close_return = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    mean_volume = df['volume'].rolling(window=20).mean()
    k = 2  # Spike threshold
    volume_spike_indicator = (df['volume'] > (mean_volume + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike_indicator
    
    # Integrate Factors
    volume_difference = df['volume'].diff().fillna(0)
    integrated_factor = (0.5 * volume_difference + 
                         0.3 * (df['high'] - df['low']).mean() + 
                         (df['close'] - df['open']))
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).mean()
    volatility = ((df['close'] - df['open']) / df['open']).rolling(window=20).std()
    adjusted_integrated_factor = integrated_factor * (volume_average * volatility)
    
    return adjusted_integrated_factor
