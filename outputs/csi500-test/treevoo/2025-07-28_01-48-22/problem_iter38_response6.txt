    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['60_day_MA'] = df['close'].rolling(window=60).mean()

    df['5_20_momentum'] = df['5_day_MA'] - df['20_day_MA']

    df['5_day_volatility'] = df['log_returns'].rolling(window=5).std()
    df['20_day_volatility'] = df['log_returns'].rolling(window=20).std()
    df['60_day_volatility'] = df['log_returns'].rolling(window=60).std()

    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1

    df['5_day_volume_MA'] = df['volume'].rolling(window=5).mean()
    df['20_day_volume_MA'] = df['volume'].rolling(window=20).mean()
    df['60_day_volume_MA'] = df['volume'].rolling(window=60).mean()

    df['local_max'] = df['high'].rolling(window=20, min_periods=1).max()
    df['local_min'] = df['low'].rolling(window=20, min_periods=1).min()
