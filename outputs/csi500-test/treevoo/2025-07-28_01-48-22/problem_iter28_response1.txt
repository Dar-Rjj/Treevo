```python
import pandas as pd

def heuristics_v2(df):
    # Combined Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'].diff(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Volatility Multiplier
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (avg_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Volume-Price Indicator
    volume_price_indicator = 0.5 * df['volume'].diff() + 0.3 * (df['high'] - df['low']).rolling(window=5).mean() + df['close'].diff()
    
    # Short-Term and Long-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Combined Factors
    combined_factors = relative_strength * (df['volume'] - df['volume'].shift()).clip(lower=0) + \
                       0.3 * (df['close'] - df['open']) / df['open'] + \
                       df['close'].diff()
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_smoothed_momentum + volume_price_indicator + combined_factors
    
    return final_alpha_factor
```
