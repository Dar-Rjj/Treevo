```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday and Historical Momentum Components
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = (df['high'] - df['low']) * intraday_return * df['volume']
    
    historical_momentum = df['close'] - df['close'].shift(10)
    
    combined_momentum = intraday_momentum + historical_momentum
    
    # Incorporate Volume Influence
    weights = 1 / np.arange(1, 11)
    weighted_avg_volume = (df['volume'].rolling(window=10).apply(lambda x: (x * weights).sum(), raw=True) / weights.sum())
    
    momentum_adjusted_by_volume = combined_momentum * weighted_avg_volume
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    adjusted_momentum = momentum_adjusted_by_volume * volatility_multiplier
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Calculate Average True Range
    true_range = pd.DataFrame({
        'HL': df['high'] - df['low'],
        'HPC': (df['high'] - df['close'].shift()).abs(),
        'LPC': (df['low'] - df['close'].shift()).abs()
    }).max(axis=1)
    atr = true_range.rolling(window=5).mean()
    
    # Integrate Hybrid Factors
    hybrid_factor = (volume_trend * 0.7) + (atr * 0.4) + adjusted_momentum
    
    return hybrid_factor
```
