import pandas as pd

def heuristics_v2(df):
    # Leaf thought: Define alpha_factor_1 = high[t] - low[t]
    alpha_factor_1 = df['high'] - df['low']
    
    # Leaf thought: Define alpha_factor_2 = (close[t] - low[t]) / (high[t] - low[t])
    alpha_factor_2 = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Leaf thought: Define alpha_factor_3 = volume[t] / (high[t] - low[t])
    alpha_factor_3 = df['volume'] / (df['high'] - df['low'])
    
    # Leaf thought: Define alpha_factor_4 = amount[t] / (high[t] - low[t])
    alpha_factor_4 = df['amount'] / (df['high'] - df['low'])
    
    # Leaf thought: Define alpha_factor_5 = (close[t] - close[t-5]) / close[t-5]
    alpha_factor_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Leaf thought: Define alpha_factor_6 = (open[t] < close[t]) ? 1 : (open[t] > close[t]) ? -1 : 0
    alpha_factor_6 = (df['open'] < df['close']).astype(int) - (df['open'] > df['close']).astype(int)

    # Combine all factors into a single DataFrame
    combined_factors = pd.concat([alpha_factor_1, alpha_factor_2, alpha_factor_3, alpha_factor_4, alpha_factor_5, alpha_factor_6], axis=1)
    combined_factors.columns = ['alpha_factor_1', 'alpha_factor_2', 'alpha_factor_3', 'alpha_factor_4', 'alpha_factor_5', 'alpha_factor_6']

    # Calculate the final heuristic factor by taking the mean of all individual factors
    final_factor = combined_factors.mean(axis=1)

    return final_factor
