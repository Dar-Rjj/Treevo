```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log returns using close prices
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate simple returns using close prices
    df['simple_returns'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate moving averages using closing prices
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['ma_60'] = df['close'].rolling(window=60).mean()
    
    # Determine price momentum
    df['momentum_5_20'] = df['ma_5'] - df['ma_20']
    df['momentum_20_60'] = df['ma_20'] - df['ma_60']
    
    # Calculate rolling standard deviation of log returns
    df['std_log_returns_5'] = df['log_returns'].rolling(window=5).std()
    df['std_log_returns_20'] = df['log_returns'].rolling(window=20).std()
    df['std_log_returns_60'] = df['log_returns'].rolling(window=60).std()
    
    # Calculate daily volume change
    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Calculate moving averages of volume
    df['ma_volume_5'] = df['volume'].rolling(window=5).mean()
    df['ma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['ma_volume_60'] = df['volume'].rolling(window=60).mean()
    
    # Create a composite score with weighted sum of selected features
    weights = {
        'log_returns': 0.2,
        'simple_returns': 0.1,
        'momentum_5_20': 0.3,
        'momentum_20_60': 0.2,
        'std_log_returns_20': -0.1,
        'volume_change': 0.1
    }
    
    df['composite_score'] = (df[list(weights.keys())] * pd.Series(weights)).sum(axis=1)
    
    return df['composite_score']
```
