```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Compute Volume Average
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    
    # Weight by Volume and Volatility
    weighted_relative_strength = relative_strength * (volume_avg * volatility)
    
    # Incorporate Trading Range
    high_low_range = (df['high'] - df['low']) / df['low']
    avg_trading_range = high_low_range.rolling(window=5).mean()
    adjusted_relative_strength = weighted_relative_strength * (1 + avg_trading_range)
    
    # Integrate Amount
    amount_avg = df['amount'].rolling(window=5).mean()
    final_factor = adjusted_relative_strength * (amount_avg / volume_avg)
    
    # Calculate Daily Price Momentum
    daily_price_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate EMA of Volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = daily_price_momentum * ema_volume
    
    # Calculate Enhanced Price and Volume Interaction Signal
    enhanced_signal = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Final Alpha Factor
    alpha_factor = final_factor + volume_adjusted_momentum + enhanced_signal
    return alpha_factor
```
