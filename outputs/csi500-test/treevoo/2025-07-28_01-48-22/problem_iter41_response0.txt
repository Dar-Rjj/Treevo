import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of open and close prices using amount and volume
    weighted_avg = (df['open'] * df['amount'] + df['close'] * df['volume']) / (df['amount'] + df['volume'])
    
    # Calculate the difference between high and low prices, normalized by their average
    diff_high_low = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Generate the alpha factor by combining the two calculated values
    alpha_factor = weighted_avg * diff_high_low
    
    return alpha_factor
