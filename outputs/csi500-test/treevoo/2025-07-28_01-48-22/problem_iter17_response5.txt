import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_open_close_return = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    k = 2  # Threshold for identifying volume spikes
    volume_spike = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike
    
    # Integrate Factors
    volume_diff = df['volume'].diff().fillna(0)
    high_low_range = (df['high'] - df['low']).fillna(0)
    close_price_change = df['close'].diff().fillna(0)
    
    integrated_factor = (0.5 * volume_diff) + (0.3 * high_low_range) + close_price_change
    
    # Enhanced Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).mean()
    volatility = df['close'].pct_change().rolling(window=20).std().fillna(0)
    adjusted_integrated_factor = integrated_factor * (volume_average * volatility)
    
    # Integrate Momentum and Reversal Signals
    momentum_signal = df['close'].pct_change(periods=120)
    reversal_signal = df['close'].pct_change(periods=7)
    combined_momentum_reversal = momentum_signal - reversal_signal
    
    # Final Factor
    final_factor = adjusted_integrated_factor + combined_momentum_reversal
    
    return final_factor
