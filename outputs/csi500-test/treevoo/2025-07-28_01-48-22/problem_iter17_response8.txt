```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily log return
    df['log_return'] = df['close'].apply(lambda x: pd.Series(x).pct_change().dropna())
    
    # Compute 20-day momentum
    df['momentum_20'] = df['log_return'].rolling(window=22).sum()
    
    # Calculate short-term return (5 days)
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate long-term return (20 days)
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate relative strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Intraday momentum component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Historical momentum component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine momentum components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth combined momentum with SMA over 5 days
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Identify volume spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # arbitrary constant
    df['volume_spike'] = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Calculate 20-day volume volatility
    df['volume_pct_change'] = df['volume'].pct_change()
    df['volume_volatility'] = df['volume_pct_change'].rolling(window=20).std()
    
    # Calculate average volume
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Adjust momentum by volume
    df['adjusted_momentum'] = (df['momentum_20'] / df['volume_volatility']) * df['avg_volume']
    
    # Integrate factors
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['high_low_range'] = (df['high'] - df['low'])
    df['integrated_factor'] = 0.5 * df['volume_diff'] + 0.3 * df['high_low_range'] + (df['close'] - df['close'].shift(1))
    
    # Calculate price range and volatility
    df['price_range_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate volatility adjustment
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=10).mean()
    df['volatility_multiplier'] = 1 / (df['avg_daily_range'] + 1)
    
    # Adjusted integrated factor
    df['adjusted_integrated_factor'] = df['integrated_factor'] * df['volatility_multiplier']
    
    # Volume and volatility weight
    df['volume_avg'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = df['close'].pct_change()
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    df['final_factor'] = df['adjusted_integrated_factor'] * (df['volume_avg'] * df['volatility'])
    
    return df['final_factor']
```
