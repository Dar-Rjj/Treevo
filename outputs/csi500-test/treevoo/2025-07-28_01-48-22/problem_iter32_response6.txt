```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series indexed by date, representing the factor values.
    """
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the percentage change in average price
    avg_price_change = avg_price.pct_change()
    
    # Calculate the ratio of amount to volume as a proxy for average transaction size
    avg_transaction_size = df['amount'] / df['volume']
    
    # Calculate the percentage change in average transaction size
    avg_trans_size_change = avg_transaction_size.pct_change()
    
    # The heuristic alpha factor is the product of the two percentage changes
    alpha_factor = avg_price_change * avg_trans_size_change
    
    return alpha_factor
```
