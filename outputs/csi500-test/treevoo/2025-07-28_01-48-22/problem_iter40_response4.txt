```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['IntradayMomentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum
    df['HistoricalMomentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Combined Momentum
    df['CombinedMomentum'] = df['IntradayMomentum'] + df['HistoricalMomentum']
    
    # Calculate Smoothed Momentum
    df['SmoothedMomentum'] = df['CombinedMomentum'].rolling(window=5).mean()
    
    # Calculate Daily High-Low Range
    df['DailyHighLowRange'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Average Daily Range
    df['AverageDailyRange'] = df['DailyHighLowRange'].rolling(window=10).mean()
    
    # Calculate Adjusted Smoothed Momentum
    df['AdjustedSmoothedMomentum'] = df['SmoothedMomentum'] / (df['AverageDailyRange'] + 1)
    
    # Calculate Volume Change
    df['VolumeChange'] = 0.5 * (df['volume'] - df['volume'].shift(1))
    
    # Calculate Avg High-Low Range (5 days)
    df['AvgHighLowRange5'] = (df['high'] - df['low']) / df['close']
    df['AvgHighLowRange5'] = df['AvgHighLowRange5'].rolling(window=5).mean()
    
    # Calculate Volume-Price Factor
    df['VolumePriceFactor'] = df['VolumeChange'] + 0.3 * df['AvgHighLowRange5'] + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Short-Term Return
    df['ShortTermReturn'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['LongTermReturn'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['RelativeStrength'] = df['ShortTermReturn'] - df['LongTermReturn']
    
    # Calculate Multiplication with Relative Strength
    df['MultiplicationWithRelativeStrength'] = df['RelativeStrength'] * (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Close Price Change
    df['ClosePriceChange'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['AdjustedSmoothedMomentum'] + 
        df['VolumePriceFactor'] + 
        df['MultiplicationWithRelativeStrength'] + 
        df['ClosePriceChange']
    )
    
    return df['FinalAlphaFactor']
```
