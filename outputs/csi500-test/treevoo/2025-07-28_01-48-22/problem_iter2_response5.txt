import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return using Close price
    df['log_return'] = df['close'].apply(lambda x: np.log(x)) - df['close'].shift(1).apply(lambda x: np.log(x))
    
    # Compute 20-Day Momentum
    df['momentum_20d'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Volume Percentage Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Compute 20-Day Standard Deviation of Volume Changes
    df['volume_volatility'] = df['volume_change'].rolling(window=20).std()
    
    # Adjust Momentum by Inverse Volume Volatility
    df['adjusted_momentum'] = df['momentum_20d'] / df['volume_volatility']
    
    # Calculate Smoothed Volume using Simple Moving Average (SMA) over 5 days
    df['smoothed_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['adjusted_momentum'] * df['smoothed_volume']
    
    # Calculate Price and Volume Interaction Signal
    df['price_volume_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Final factor value
    factor_value = df['volume_adjusted_momentum'] + df['price_volume_signal']
    
    return factor_value.dropna()
