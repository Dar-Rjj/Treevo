```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close Price
    df['vol_weighted_close'] = df['close'] * df['volume']
    
    # Calculate 5-Day and 20-Day Volume-Weighted Close Price Moving Averages
    df['5_day_vol_weighted_close_ma'] = df['vol_weighted_close'].rolling(window=5).mean()
    df['20_day_vol_weighted_close_ma'] = df['vol_weighted_close'].rolling(window=20).mean()
    
    # Calculate Momentum
    df['momentum'] = df['5_day_vol_weighted_close_ma'] - df['20_day_vol_weighted_close_ma']
    
    # Calculate Volume Surge Indicator
    df['5_day_vol_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_surge_indicator'] = np.where(df['volume'] > 1.5 * df['5_day_vol_ma'], 1, 0)
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['20_day_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Volume Volatility
    df['vol_percentage_change'] = df['volume'].pct_change()
    df['vol_volatility'] = df['vol_percentage_change'].rolling(window=20).std()
    
    # Adjust 20-Day Momentum by Inverse Volume Volatility
    df['adjusted_20_day_momentum'] = df['20_day_momentum'] / (df['vol_volatility'] + 1e-8)  # small epsilon to avoid division by zero
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['momentum'] * df['volume_surge_indicator'] * df['adjusted_20_day_momentum'])
    
    return df['alpha_factor']
```
