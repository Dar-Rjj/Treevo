```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement within the day
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate the percentage change of the close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate a smoothed version of the price movement using a rolling window
    df['smoothed_price_movement'] = df['price_movement'].rolling(window=10).mean()
    
    # Calculate the cumulative sum of the volume
    df['cumulative_volume'] = df['volume'].cumsum()
    
    # Calculate the ratio of the smoothed price movement to the cumulative volume
    df['factor'] = df['smoothed_price_movement'] / df['cumulative_volume']
    
    return df['factor']
```
