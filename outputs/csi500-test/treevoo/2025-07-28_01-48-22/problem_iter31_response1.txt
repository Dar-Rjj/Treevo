```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].pct_change(5)
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    N = 20
    K = 2
    volume_std = df['volume'].rolling(window=N).std()
    volume_mean = df['volume'].rolling(window=N).mean()
    volume_spike = (df['volume'] > (volume_mean + K * volume_std)).astype(int)
    
    # Adjusted Integrated Factor
    adjusted_factor = relative_strength * volume_spike
    volume_avg = df['volume'].rolling(window=5).mean()
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    adjusted_factor *= (volume_avg * volatility)
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    daily_range_ema = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (daily_range_ema + 1)
    adjusted_factor *= volatility_multiplier
    
    # Integrate Hybrid Factors
    close_momentum = df['close'] - df['close'].rolling(window=21).mean() - df['close'].shift(1)
    adjusted_factor += close_momentum
    
    # Incorporate Trend Following Adjustment
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_adjustment = np.where(sma_10 > sma_30, 1.1, 0.9)
    adjusted_factor *= trend_adjustment
    
    # Include Volume and Price Correlation
    correlation = df[['volume', 'close']].rolling(window=10).corr().iloc[::2, :]['close']
    correlation_adjustment = np.where(correlation > 0, 1.05, 0.95)
    adjusted_factor *= correlation_adjustment
    
    return adjusted_factor
```
