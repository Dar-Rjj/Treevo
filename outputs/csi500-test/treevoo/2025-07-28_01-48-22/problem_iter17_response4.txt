import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    prev_typical_price = typical_price.shift(1)
    positive_money_flow = money_flow.where(typical_price > prev_typical_price, 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < prev_typical_price, 0).rolling(window=14).sum()
    
    # Calculate the money ratio
    money_ratio = positive_money_flow / negative_money_flow.abs()
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_ratio))
    
    # Calculate the factor value
    factor_value = (money_flow_index - money_flow_index.rolling(window=14).mean()) / money_flow_index.rolling(window=14).std()
    
    return factor_value
