```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day Simple Moving Average (SMA) of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Generate buy/sell signal based on today's close price and 5-day SMA
    df['SMA_Signal'] = (df['close'] > df['SMA_5']).astype(int) * 2 - 1
    
    # Calculate the percentage change in closing price from 10 days ago to today
    df['Momentum_10'] = df['close'].pct_change(periods=10)
    
    # Compute the difference between daily close and open prices
    df['Daily_Diff'] = df['close'] - df['open']
    
    # Create a cumulative sum of daily differences over the past 20 trading days
    df['Cumulative_Diff_20'] = df['Daily_Diff'].rolling(window=20).sum()
    
    # Calculate the 20-day average trading volume
    df['Volume_Avg_20'] = df['volume'].rolling(window=20).mean()
    
    # Flag days with significantly high volume (2 standard deviations above average)
    df['Volume_Anomaly'] = ((df['volume'] - df['Volume_Avg_20']) / df['Volume_Avg_20']).apply(lambda x: 1 if x > 2 else 0)
    
    # Combine volume anomaly with the sign of the day's return
    df['Volume_Return_Signal'] = df['Volume_Anomaly'] * (df['close'] - df['open']) / (df['open'] + df['close']) * 2
    
    # Calculate the daily range (high - low)
    df['Range'] = df['high'] - df['low']
    
    # Compute the 10-day moving average of ranges
    df['Range_MA_10'] = df['Range'].rolling(window=10).mean()
    
    # Calculate the relative range
    df['Relative_Range'] = df['Range'] / df['Range_MA_10']
    
    # Combine all signals into a single alpha factor
    df['Alpha_Factor'] = (df['SMA_Signal'] +
                          df['Momentum_10'] +
                          df['Cumulative_Diff_20'] +
                          df['Volume_Return_Signal'] +
                          df['Relative_Range'])
    
    return df['Alpha_Factor']
```
