import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate log and simple returns using close prices
    df['log_return'] = np.log(df['close']).diff()
    df['simple_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate moving averages of closing prices
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['60_day_MA'] = df['close'].rolling(window=60).mean()
    
    # Determine price momentum
    df['momentum_5_20'] = df['5_day_MA'] - df['20_day_MA']
    df['momentum_20_60'] = df['20_day_MA'] - df['60_day_MA']
    
    # Calculate rolling standard deviations of log returns
    df['log_return_std_5'] = df['log_return'].rolling(window=5).std()
    df['log_return_std_20'] = df['log_return'].rolling(window=20).std()
    df['log_return_std_60'] = df['log_return'].rolling(window=60).std()
    
    # Calculate Parkinson's volatility
    df['parkinson_vol'] = np.sqrt((1 / (4 * np.log(2) * 5)) * ((np.log(df['high'] / df['low'])) ** 2).rolling(window=5).sum())
    
    # Calculate daily volume change
    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Measure volume trends
    df['volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['volume_MA_60'] = df['volume'].rolling(window=60).mean()
    
    # Calculate volume-weighted average price
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP_5'] = df['VWAP'].rolling(window=5).sum()
    df['VWAP_20'] = df['VWAP'].rolling(window=20).sum()
    df['VWAP_60'] = df['VWAP'].rolling(window=60).sum()
    
    # Calculate true range
    df['true_range'] = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    df['ATR_5'] = df['true_range'].rolling(window=5).mean()
    df['ATR_20'] = df['true_range'].rolling(window=20).mean()
    df['ATR_60'] = df['true_range'].rolling(window=60).mean()
    
    # Calculate open to close difference and percentage change
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_pct'] = ((df['close'] - df['open']) / df['open']) * 100
    
    # Calculate intraday price range and percentage
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_pct'] = ((df['high'] - df['low']) / df['low']) * 100
    
    # Calculate volume-price correlation
    df['volume_price_corr_5'] = df['close'].rolling(window=5).corr(df['volume'])
    df['volume_price_corr_20'] = df['close'].rolling(window=20).corr(df['volume'])
    df['volume_price_corr_60'] = df['close'].rolling(window=60).corr(df['volume'])
    
    # Calculate daily amount change
    df['amount_change'] = (df['amount'] / df['amount'].shift(1)) - 1
    
    # Measure amount trends
    df['amount_MA_5'] = df['amount'].rolling(window=5).mean()
    df['amount_MA_20'] = df['amount'].rolling(window=20).mean()
    df['amount_MA_60'] = df['amount'].rolling(window=60).mean()
    
    # Calculate amount-to-volume ratio
    df['amount_to_volume_ratio'] = df['amount'] / df['volume']
