```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range for the past 5 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_returns_with_spike = combined_returns * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-to-VWAP Ratio
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    adjusted_close_change = (df['close'] - df['close'].shift(1)) * 0.2
    integrated_volume_price = 0.5 * volume_difference + 0.3 * avg_high_low_range + adjusted_close_change
    
    # Combine Indicators
    intermediate_result = high_low_over_avg * volume_difference * volume_to_vwap_ratio * (df['close'] / df['close'].shift(1) - 1)
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    adjusted_factor = (relative_strength + combined_returns_with_spike + integrated_volume_price + intermediate_result) * (volume_average * volatility)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor
    
    return final_alpha_factor
```
