```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative difference between high and low
    df['high_low_diff'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative difference between open and close
    df['open_close_diff'] = (df['open'] - df['close']) / df['close']
    
    # Volume weighted average price
    df['vwap'] = (df['amount'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Momentum factor: close price change over 5 days
    df['momentum_5d'] = df['close'].pct_change(5)
    
    # Volatility factor: standard deviation of daily returns over 5 days
    df['volatility_5d'] = df['close'].pct_change().rolling(window=5).std()
    
    # Combining the factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_change'] 
        + df['high_low_diff'] 
        + df['open_close_diff'] 
        + df['momentum_5d'] 
        - df['volatility_5d']
    )
    
    return df['alpha_factor'].dropna()
```
