```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (mean_volume + 2 * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_with_spike = combined_return * volume_spike_indicator
    
    # Weight by Volume and Volatility
    avg_volume = df['volume'].rolling(window=5).sum() / 5
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    weighted_relative_strength = relative_strength * (avg_volume * volatility)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_previous_close_return = df['close'].pct_change(periods=1)
    
    # Calculate Average High-Low Range
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume-to-VWAP Ratio
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    volume_weighted = volume_difference * 0.5
    range_weighted = avg_high_low_range * 0.3
    adjusted_close_change = close_to_previous_close_return
    
    integrated_info = volume_weighted + range_weighted + adjusted_close_change
    
    # Combine Indicators
    high_low_range_over_avg = high_low_range / avg_high_low_range
    intermediate_factor = high_low_range_over_avg * vwap_ratio * close_to_previous_close_return
    
    # Final Alpha Factor
    final_alpha_factor = (combined_return_with_spike + weighted_relative_strength + integrated_info + intermediate_factor)
    
    return final_alpha_factor
```
