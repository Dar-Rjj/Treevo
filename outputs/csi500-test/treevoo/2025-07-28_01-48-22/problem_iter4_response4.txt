```python
def heuristics_v2(df):
    # Calculate Volume-Weighted Close Price
    df['vw_close'] = df['close'] * df['volume']
    
    # Calculate 5-Day and 20-Day Volume-Weighted Close Price Moving Averages
    df['5d_vw_close_ma'] = df['vw_close'].rolling(window=5).mean()
    df['20d_vw_close_ma'] = df['vw_close'].rolling(window=20).mean()
    
    # Momentum: Subtract 20-Day Moving Average from 5-Day Moving Average
    df['momentum'] = df['5d_vw_close_ma'] - df['20d_vw_close_ma']
    
    # Volume Surge Indicator
    df['5d_volume_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = (df['volume'] > 1.5 * df['5d_volume_ma']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum'] * df['volume_surge']
    
    return df['alpha_factor']
```
