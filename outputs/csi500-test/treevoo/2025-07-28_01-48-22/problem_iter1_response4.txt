```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Volume Volatility
    df['volume_pct_change'] = df['volume'].pct_change()
    df['vol_volatility'] = df['volume_pct_change'].rolling(window=20).std()
    
    # Adjust Momentum by Inverse Volume Volatility
    df['adjusted_momentum'] = df['momentum_20'] / (df['vol_volatility'] + 1e-8)  # Small constant to avoid division by zero
    
    return df['adjusted_momentum']
```
