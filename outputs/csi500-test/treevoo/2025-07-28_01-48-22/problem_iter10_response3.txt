import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # Spike threshold in standard deviations
    volume_spike_indicator = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    adjusted_combined_return = combined_return * volume_spike_indicator
    
    # Integrate Volume and Price Information
    # Compute Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Compute Final Factor
    final_factor = (volume_difference * 0.5) + (avg_high_low_range * 0.3) + (df['close'] - df['close'].shift(1))
    
    return final_factor
