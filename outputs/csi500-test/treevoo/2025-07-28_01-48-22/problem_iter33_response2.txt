import pandas as pd

def heuristics_v2(df):
    # Simple Moving Average Crossover
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['SMA_50'] - df['SMA_200']

    # Exponential Moving Average Crossover
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['EMA_200'] = df['close'].ewm(span=200, adjust=False).mean()
    df['EMA_Crossover'] = df['EMA_50'] - df['EMA_200']

    # On-Balance Volume (OBV)
    df['OBV'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'OBV'] = df['volume']
    df.loc[df['close'] < df['close'].shift(1), 'OBV'] = -df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Volume-Weighted Average Price (VWAP)
    df['VWAP_daily'] = (df['close'] * df['volume']) / df['volume']
    df['VWAP_30'] = df['VWAP_daily'].rolling(window=30).mean()
    df['VWAP_Factor'] = (df['close'] - df['VWAP_30']) / df['VWAP_30']

    # True Range (TR)
    df['TR'] = df[['high', 'low', df['close'].shift(1)]].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()

    # Inside Bar Pattern
    df['Inside_Bar'] = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1)).astype(int)

    # Engulfing Pattern
    df['Engulfing'] = ((df['open'] < df['close']) & (df['open'].shift(1) > df['close'].shift(1)) 
                       & (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))).astype(int)

    # High-Low Ratio
    df['High_Low_Ratio'] = (df['high'] - df['low']) / df['low']

    # Close-Open Spread
    df['Close_Open_Spread'] = df['close'] - df['open']

    # Combine all factors into a single DataFrame
    alpha_factors = df[['SMA_Crossover', 'EMA_Crossover', 'OBV', 'VWAP_Factor', 'ATR_14', 'Inside_Bar', 'Engulfing', 'High_Low_Ratio', 'Close_Open_Spread']]
    return alpha_factors
