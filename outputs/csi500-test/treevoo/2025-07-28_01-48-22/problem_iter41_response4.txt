```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    momentum_20_day = daily_log_return.rolling(window=20).sum()
    
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Sum of Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    volume_volatility = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Average Daily Range
    avg_daily_range = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Volatility Multiplier
    volatility_multiplier = 1 / (avg_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = (smoothed_momentum / volume_volatility * avg_volume) * volatility_multiplier
    
    # Volume-Price Indicator
    volume_difference = df['volume'] - df['volume'].shift(1)
    volume_price_indicator = volume_difference * 0.5 + (df['high'] - df['low']).rolling(window=10).mean() * 0.3 + (df['close'] - df['close'].shift(1))
    
    # Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Volume Spikes
    volume_std_dev = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (df['volume'].rolling(window=20).mean() + 2 * volume_std_dev)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_by_volume_spike = relative_strength * volume_spike_indicator
    
    # High-Low Range Over Average
    high_low_range = df['high'] - df['low']
    high_low_range_over_avg = high_low_range / avg_daily_range
    
    # Final Factor
    final_factor = (adjusted_smoothed_momentum * 0.4 +
                    volume_price_indicator * 0.3 +
                    relative_strength_by_volume_spike * 0.2 +
                    high_low_range_over_avg * 0.1)
    
    return final_factor
```
