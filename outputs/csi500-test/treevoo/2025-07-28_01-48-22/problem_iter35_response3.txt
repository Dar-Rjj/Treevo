import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)

    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(20)

    # Daily Log Return
    daily_log_return = np.log(df['close']).diff()

    # 20-Day Momentum
    momentum_20d = daily_log_return.rolling(window=20).sum()

    # Relative Strength
    relative_strength = short_term_return - long_term_return

    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']

    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)

    # Sum of Components
    combined_momentum = intraday_momentum + historical_momentum

    # Smooth Combined Momentum
    smoothed_combined_momentum = combined_momentum.rolling(window=5).mean()

    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).mean()
    volatility = (df['close'] - df['open']) / df['open']
    adjusted_volatility = volatility.rolling(window=20).std()
    adjusted_integrated_factor = (smoothed_combined_momentum * volume_average * adjusted_volatility).fillna(0)

    # Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_combined_momentum = combined_momentum * volatility_multiplier

    # Volume-Price Indicator
    volume_price_indicator = 0.5 * (df['volume'] - df['volume'].shift(1)) + \
                             0.3 * (df['high'] - df['low']).rolling(window=5).mean() + \
                             (df['close'] - df['close'].shift(1))

    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)

    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_with_volume_spike = relative_strength * volume_spike_indicator

    # High-Low Range Over Average
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    high_low_range_over_avg = high_low_range / avg_high_low_range

    # Integrate Hybrid Factors
    close_price_momentum = df['close'] - df['close'].shift(10)
    trend_following_adjustment = (df['close'].rolling(window=10).mean() > df['close'].rolling(window=30).mean()).astype(int) * 0.2 + 0.9
    integrated_factors = 0.5 * volume_price_indicator + close_price_momentum * trend_following_adjustment

    # Include Volume and Price Correlation
    volume_close_corr = df[['volume', 'close']].rolling(window=10).corr().unstack().iloc[::2]['volume']
    correlation_adjustment = (volume_close_corr > 0).astype(int) * 0.1 + 0.95
    final_factor = integrated_factors * correlation_adjustment

    # Apply a Dynamic Weighting Scheme
    moving_average_final_factor = final_factor.rolling(window=10).mean()
    dynamic_weighting = (moving_average_final_factor.diff() > 0).astype(int) * 0.2 + 0.9

    final_factor = final_factor * dynamic_weighting

    return final_factor
