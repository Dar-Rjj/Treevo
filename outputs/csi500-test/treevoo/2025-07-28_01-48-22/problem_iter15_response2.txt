```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Weighted Average Volume
    weights = 1 / np.arange(1, 16)
    weighted_avg_volume = (df['volume'].rolling(window=15).apply(lambda x: np.sum(x * weights), raw=True) 
                           / np.sum(weights))
    
    # Incorporate Volume Influence
    combined_momentum *= weighted_avg_volume
    
    # Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    # Adjust Combined Momentum by Advanced Volatility
    adjusted_combined_momentum = combined_momentum * volatility_multiplier
    
    # Volume Momentum
    volume_diff = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # High-Low Volatility (5 days)
    high_low_std = (df['high'] - df['low']).rolling(window=5).std()
    
    # Close Price Momentum
    close_price_change = (df['close'] - df['close'].rolling(window=21).mean()) - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = (0.5 * volume_diff + 0.3 * high_low_std + close_price_change)
    
    # Trend Following Adjustment
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_following_adjustment = np.where(sma_10 > sma_30, 1.1, 0.9)
    
    # Final Factor
    final_factor = (hybrid_factor * trend_following_adjustment * adjusted_combined_momentum)
    
    return final_factor
```
