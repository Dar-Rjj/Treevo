```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_return * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Incorporate Volume Influence
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_influenced_momentum = combined_momentum * avg_volume_10_days
    
    # Adjust Combined Momentum by Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range_10_days = daily_range.rolling(window=10).sum() / 10
    volatility_multiplier = 1 / (avg_daily_range_10_days + 1)
    adjusted_momentum = volume_influenced_momentum * volatility_multiplier
    
    return adjusted_momentum
```
