```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Range Over Average
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Weight by Volume Trend
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_trend = volume_change / df['volume'].shift(1)
    
    # Introduce Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Combine Indicators
    indicator_product = high_low_over_avg * volume_trend
    combined_indicators = indicator_product + close_open_spread
    factor_value = combined_indicators * close_return
    
    return factor_value
```
