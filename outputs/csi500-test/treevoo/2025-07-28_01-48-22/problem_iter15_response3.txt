```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    k = 2  # Example value for K
    volume_spike_indicator = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_spike = combined_return * volume_spike_indicator
    
    # Integrate Volume and Price Information
    volume_diff = df['volume'] - df['volume'].shift(1)
    adjusted_volume_diff = volume_diff * 0.5
    average_high_low_range = (df['high'] - df['low']).rolling(window=5).mean() * 0.3
    adjusted_close_change = df['close'] - df['close'].shift(1)
    
    # Calculate High-Low Range Over Average
    high_low_range = df['high'] - df['low']
    average_high_low_range_5_days = (df['high'] - df['low']).rolling(window=5).mean()
    high_low_range_over_average = high_low_range / average_high_low_range_5_days
    
    # Calculate Volume-to-VWAP Ratio
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Combine Indicators
    intermediate_result = high_low_range_over_average * adjusted_volume_diff * volume_to_vwap_ratio * adjusted_close_change
    
    # Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    weighted_relative_strength = relative_strength * (volume_average * volatility)
    
    # Incorporate Price Movement Intensity
    price_movement_intensity = (df['close'] - df['close'].shift(1)).abs() / average_high_low_range_5_days
    final_factor_value = intermediate_result * price_movement_intensity * weighted_relative_strength
    
    return final_factor_value
```
