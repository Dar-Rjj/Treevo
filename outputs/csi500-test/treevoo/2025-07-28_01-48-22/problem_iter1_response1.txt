import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Daily_Price_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Smoothed Volume with 5-day SMA
    df['Smoothed_Volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = df['Daily_Price_Momentum'] * df['Smoothed_Volume']
    
    # Calculate Price and Volume Interaction Signal
    df['Price_Volume_Interaction'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Final alpha factor: Combine Volume-Adjusted Momentum and Price and Volume Interaction Signal
    df['Alpha_Factor'] = df['Volume_Adjusted_Momentum'] + df['Price_Volume_Interaction']
    
    return df['Alpha_Factor']
