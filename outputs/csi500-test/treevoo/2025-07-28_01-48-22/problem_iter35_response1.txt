```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['ShortTermReturn'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['LongTermReturn'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Combine Momentum Components
    df['IntradayMomentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['HistoricalMomentum'] = df['close'] - df['close'].shift(10)
    df['CombinedMomentum'] = df['IntradayMomentum'] + df['HistoricalMomentum']
    
    # Smooth Combined Momentum with EMA over 5 days
    df['SmoothedMomentum'] = df['CombinedMomentum'].ewm(span=5).mean()
    
    # 20-Day Volume Volatility
    df['VolumeVolatility'] = df['volume'].pct_change().rolling(window=20).std()
    
    # Average Volume
    df['AverageVolume'] = df['volume'].rolling(window=10).sum() / 10
    
    # Adjust Momentum
    df['AdjustedMomentum'] = df['SmoothedMomentum'] / df['VolumeVolatility'] * df['AverageVolume']
    
    # Calculate Relative Strength
    df['RelativeStrength'] = df['ShortTermReturn'] - df['LongTermReturn']
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    df['AvgHighLowRange'] = df['HighLowRange'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['HighLowOverAvg'] = df['HighLowRange'] / df['AvgHighLowRange']
    
    # Combine Intraday and Open-to-Close Returns
    df['IntradayReturn'] = df['high'] - df['low']
    df['OpenCloseReturn'] = df['close'] - df['open']
    df['CombinedReturn'] = df['IntradayReturn'] + df['OpenCloseReturn']
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    df['VolumeSpike'] = (df['volume'] > (mean_volume + 2 * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['CombinedReturnSpiked'] = df['CombinedReturn'] * df['VolumeSpike']
    
    # Calculate VWAP
    df['VWAP'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Integrate Volume and Price Information
    df['VolumeDiff'] = df['volume'] - df['volume'].shift(1)
    df['AdjCloseChange'] = df['close'] - df['close'].shift(1)
    df['CombinedAdjustments'] = (df['VolumeDiff'] * 0.6) + (df['AvgHighLowRange'] * 0.4) + df['AdjCloseChange']
    
    # Combine Indicators
    df['IntermediateResult'] = df['HighLowOverAvg'] * df['VolumeDiff'] * df['VWAP'] * (df['close'] - df['close'].shift(1))
    
    # Weight by Volume and Volatility
    df['VolumeAverage'] = df['volume'].rolling(window=5).sum() / 5
    df['DailyReturns'] = df['close'].pct_change()
    df['Volatility'] = df['DailyReturns'].rolling(window=20).std()
    df['FinalFactor'] = df['RelativeStrength'] * (df['VolumeAverage'] * df['Volatility'])
    
    return df['FinalFactor']
```
