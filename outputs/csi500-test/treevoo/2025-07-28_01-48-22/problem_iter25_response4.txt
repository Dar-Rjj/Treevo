```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Incorporate Volume Influence
    weighted_average_volume = df['volume'].rolling(window=15).apply(lambda x: sum(x / range(1, len(x) + 1)))
    volume_influence = smoothed_momentum * weighted_average_volume
    
    # Advanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_high_low_range = daily_high_low_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_high_low_range + 1)
    
    # Adjusted Momentum by Volatility
    adjusted_momentum = volume_influence * volatility_multiplier
    
    return adjusted_momentum
```
