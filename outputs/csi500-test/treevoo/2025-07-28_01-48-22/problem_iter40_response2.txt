```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_momentum'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combined Momentum
    df['combined_momentum'] = df['momentum_20'] + df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    df['volume_volatility'] = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    
    # Adjusted Smoothed Momentum
    df['adjusted_smoothed_momentum'] = (df['smoothed_momentum'] / df['volume_volatility']) * df['average_volume']
    
    # Volume-Price Indicator
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_close_change'] = df['close'] - df['close'].shift(1)
    df['avg_high_low_range'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['volume_price_indicator'] = (df['volume_difference'] * 0.5) + (df['avg_high_low_range'] * 0.3) + df['adjusted_close_change']
    
    # Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Volume Spikes
    df['volume_std_dev'] = df['volume'].rolling(window=20).std()
    mean_volume = df['volume'].rolling(window=20).mean()
    df['volume_spike_indicator'] = (df['volume'] > (mean_volume + 2 * df['volume_std_dev'])).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    df['rs_with_volume_spike'] = df['relative_strength'] * df['volume_spike_indicator']
    
    # High-Low Range Over Average
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_over_avg'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Final Factor
    df['final_factor'] = df['adjusted_smoothed_momentum'] + df['volume_price_indicator'] + df['rs_with_volume_spike'] + df['high_low_range_over_avg']
    
    return df['final_factor']
```
