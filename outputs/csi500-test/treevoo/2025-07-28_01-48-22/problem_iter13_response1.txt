import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return (5-day)
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return (20-day)
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    k = 2  # Standard deviation multiplier
    volume_spike_indicator = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_volume = relative_strength * volume_spike_indicator
    
    # Integrate Volume and Price Information
    volume_difference = df['volume'] - df['volume'].shift(1)
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Compute Final Factor
    final_factor = (volume_difference * 0.5) + (avg_high_low_range * 0.3) + (df['close'] - df['close'].shift(1))
    
    # Combine the relative strength adjusted by volume spikes and the integrated volume and price information
    factor_value = relative_strength_volume + final_factor
    
    return factor_value
