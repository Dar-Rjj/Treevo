```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['20_day_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Intraday Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Identify Volume Spikes
    vol_mean = df['volume'].rolling(window=20).mean()
    vol_std = df['volume'].rolling(window=20).std()
    df['volume_spike'] = (df['volume'] > (vol_mean + 2 * vol_std)).astype(int)
    
    # Calculate 20-Day Volume Volatility
    df['volume_volatility'] = df['volume'].pct_change().rolling(window=20).std()
    
    # Calculate Average Volume
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    
    # Adjust Momentum
    df['adjusted_momentum'] = df['20_day_momentum'] / df['volume_volatility'] * df['average_volume']
    
    # Integrate Factors
    df['volume_diff'] = (df['volume'] - df['volume'].shift(1)) * 0.5
    df['avg_high_low_range'] = (df['high'] - df['low']).rolling(window=10).mean() * 0.3
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Price Range and Volatility
    df['price_range_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    df['daily_high_low_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_high_low_range'].rolling(window=10).mean()
    df['volatility_multiplier'] = 1 / (df['avg_daily_range'] + 1)
    
    # Adjusted Integrated Factor
    df['integrated_factor'] = df['adjusted_momentum'] + df['volume_diff'] + df['avg_high_low_range'] + df['close_price_change'] + df['price_range_volatility']
    df['volume_average'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    df['adjusted_integrated_factor'] = df['integrated_factor'] * (df['volume_average'] * df['volatility'])
    
    return df['adjusted_integrated_factor']
```
