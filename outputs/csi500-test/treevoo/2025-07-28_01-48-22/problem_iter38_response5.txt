```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (2 * df['close'] - df['low'] - df['high']) / df['range']
    
    # Calculate the money flow
    df['money_flow'] = df['buying_pressure'] * df['volume'] * df['amount']
    
    # Calculate the factor by normalizing the money flow over a rolling window
    factor = df['money_flow'].rolling(window=10).mean()
    
    return factor
```
