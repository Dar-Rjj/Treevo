```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close price
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the money flow by multiplying the average price with the volume
    money_flow = avg_price * df['volume']
    
    # Calculate the money flow difference
    money_flow_diff = money_flow.diff()
    
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the factor: money flow difference divided by the high-low range
    # This factor reflects the change in money flow relative to the price range
    factor = money_flow_diff / high_low_range
    
    return factor
```
