```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_momentum = relative_strength * volume_spike
    
    # Integrate Factors
    volume_difference = df['volume'] - df['volume'].shift(1)
    integrated_factor = combined_momentum + 0.5 * volume_difference
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).sum() / 5
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    adjusted_integrated_factor = integrated_factor * (volume_average * volatility)
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, min_periods=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    final_factor = adjusted_integrated_factor * volatility_multiplier
    
    return final_factor
```
