```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price which is the average of high, low and close
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['open']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mf_multiplier * df['volume']
    
    # Calculate the positive and negative money flow
    pos_mf = mf_volume.where(mf_multiplier > 0, 0)
    neg_mf = mf_volume.where(mf_multiplier < 0, 0)
    
    # Calculate the 14-day positive and negative money flow
    n_days = 14
    pos_mf_14 = pos_mf.rolling(window=n_days).sum()
    neg_mf_14 = neg_mf.rolling(window=n_days).sum()
    
    # Calculate the money flow ratio
    mr = pos_mf_14 / neg_mf_14
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + mr))
    
    # Heuristic: If MFI is above 80, it is considered overbought, and if below 20, it is oversold
    # Generate a signal based on the MFI to predict future returns
    factor = (mfi < 20) * 1 + (mfi > 80) * (-1)
    
    return factor
```
