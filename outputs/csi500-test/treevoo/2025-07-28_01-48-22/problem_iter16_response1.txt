```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday and Historical Momentum Components
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_return * df['volume']

    historical_momentum = df['close'] - df['close'].shift(20)
    combined_momentum = intraday_momentum + historical_momentum

    # Incorporate Enhanced Volume Influence
    weighted_avg_volume = (df['volume'].rolling(window=20).sum() / 20) * np.arange(1, 21)
    volume_influence = combined_momentum * weighted_avg_volume

    # Incorporate Enhanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_daily_range = daily_high_low_range.ewm(span=15, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_momentum = volume_influence * volatility_multiplier

    # Calculate Volume Change Rate
    volume_change_rate = df['volume'] / df['volume'].shift(1)

    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=7).mean()

    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)

    # Integrate Hybrid Factors
    hybrid_factor = (volume_change_rate * 0.6) + (avg_high_low_range * 0.4) + close_price_change

    # Final Factor
    final_factor = hybrid_factor * adjusted_momentum

    return final_factor
```
