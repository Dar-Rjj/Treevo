```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']

    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)

    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum

    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()

    # Calculate 20-Day Volume Volatility
    volume_changes = df['volume'].pct_change()
    volume_volatility = volume_changes.rolling(window=20).std()

    # Calculate Weighted Average Volume
    weights = 1 / pd.Series(range(1, 16))
    weighted_avg_volume = (df['volume'].rolling(window=15).apply(lambda x: (x * weights).sum(), raw=False) / weights.sum())

    # Adjust Momentum
    adjusted_momentum = (smoothed_momentum / volume_volatility) * weighted_avg_volume

    # Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    advanced_adjusted_momentum = adjusted_momentum * volatility_multiplier

    # Price Range and Volatility
    price_range_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']

    # Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1

    # Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1

    # Relative Strength
    relative_strength = short_term_return - long_term_return

    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)

    # Multiply Combined Return by Volume Spike Indicator
    relative_strength_with_volume_spike = relative_strength * volume_spike

    # Integrate Hybrid Factors
    volume_difference = df['volume'] - df['volume'].shift(1)
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    close_price_change = df['close'] - df['close'].shift(1)

    integrated_factor = 0.5 * volume_difference + 0.3 * avg_high_low_range + close_price_change

    # Final Factor
    final_factor = integrated_factor * advanced_adjusted_momentum

    return final_factor
```
