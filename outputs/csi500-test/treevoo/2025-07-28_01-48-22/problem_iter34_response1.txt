```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    df['20_day_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_momentum'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Sum of Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Average Daily Range
    df['avg_daily_range'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Volatility Multiplier
    df['volatility_multiplier'] = 1 / (df['avg_daily_range'] + 1)
    
    # Adjusted Smoothed Momentum
    df['adjusted_smoothed_momentum'] = df['smoothed_momentum'] * df['volatility_multiplier']
    
    # Volume-Price Indicator
    df['volume_price_indicator'] = 0.5 * (df['volume'] - df['volume'].shift(1)) + \
                                   0.3 * (df['high'] - df['low']).rolling(window=5).mean() + \
                                   (df['close'] - df['close'].shift(1))
    
    # Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # Standard deviation threshold
    df['volume_spike_indicator'] = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    df['relative_strength_with_spike'] = df['relative_strength'] * df['volume_spike_indicator']
    
    # High-Low Range Over Average
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=10).mean()
    df['high_low_range_over_avg'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Final Factor
    df['final_factor'] = df['adjusted_smoothed_momentum'] + \
                         df['volume_price_indicator'] + \
                         df['relative_strength_with_spike'] + \
                         df['high_low_range_over_avg']
    
    return df['final_factor']
```
