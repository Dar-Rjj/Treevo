import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = df['close'].apply(lambda x: np.log(x) - np.log(x.shift(1)))
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday and Historical Momentum Components
    df['intraday_momentum'] = ((df['close'] - df['open']) / df['open']) * df['volume']
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['volume_volatility_20'] = df['volume'].pct_change().rolling(window=20).std()
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = (df['momentum_20'] / df['volume_volatility_20']) * df['average_volume']
    
    # Calculate Short-Term and Long-Term Returns
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Weight by Volume and Volatility
    df['volume_average'] = df['volume'].rolling(window=5).mean()
    df['volatility'] = df['log_return'].rolling(window=5).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_average'] * df['volatility'])
    
    # Incorporate Trading Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    df['average_trading_range'] = df['high_low_range'].rolling(window=5).mean()
    df['adjusted_relative_strength'] = df['weighted_relative_strength'] * (1 + df['average_trading_range'])
    
    # Integrate Amount
    df['amount_average'] = df['amount'].rolling(window=5).mean()
    df['final_factor'] = df['adjusted_relative_strength'] * (df['amount_average'] / df['volume_average'])
    
    # Calculate Enhanced Price and Volume Interaction Signal
    df['price_volume_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Combine Momentum, Volume, and Price Interaction
    df['alpha_factor'] = df['final_factor'] * df['price_volume_signal']
    
    return df['alpha_factor']
