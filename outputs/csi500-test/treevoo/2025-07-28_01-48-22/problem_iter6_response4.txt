```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    prev_day_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Compare Intraday and Previous Day Spreads
    spread_ratio = intraday_spread / prev_day_spread
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Intraday Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Combine Signals for Alpha Factor
    # Assign weights to each component
    weight_spread_ratio = 0.4
    weight_close_to_open_return = 0.3
    weight_volume_ratio = 0.3
    
    # Add the weighted components
    alpha_factor = (weight_spread_ratio * spread_ratio +
                    weight_close_to_open_return * close_to_open_return +
                    weight_volume_ratio * volume_ratio)
    
    return alpha_factor
```
