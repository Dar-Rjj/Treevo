```python
def heuristics_v2(df):
    # Calculate the price change from the previous day's close to the current open
    df['overnight_change'] = df['open'] - df['close'].shift(1)
    
    # Calculate the buying pressure as the difference between the amount and the volume times the close price
    df['buying_pressure'] = df['amount'] - (df['volume'] * df['close'])
    
    # Calculate the average of the overnight change over the last 5 days
    df['avg_overnight_change'] = df['overnight_change'].rolling(window=5).mean()
    
    # Calculate the momentum of the buying pressure over the last 10 days
    df['momentum_buying_pressure'] = df['buying_pressure'].rolling(window=10).sum()
    
    # Define the alpha factor as the product of the average overnight change and the momentum of the buying pressure
    alpha_factor = df['avg_overnight_change'] * df['momentum_buying_pressure']
    
    return alpha_factor
```
