import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['momentum_20d'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate 20-Day Volume Volatility
    df['volume_change'] = df['volume'].pct_change()
    df['vol_volatility_20d'] = df['volume_change'].rolling(window=20).std()
    
    # Adjust Momentum by Volume Volatility
    df['adjusted_momentum'] = df['momentum_20d'] / df['vol_volatility_20d']
    
    # Return the alpha factor
    return df['adjusted_momentum']
