```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Returns
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    momentum_20d = daily_log_return.rolling(window=20).sum()
    
    # Combine Momentum Components
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_combined_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust Momentum
    volume_volatility = df['volume'].diff().rolling(window=20).std()
    average_volume = df['volume'].rolling(window=10).mean()
    adjusted_momentum = momentum_20d / volume_volatility * average_volume
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_returns = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_spike = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Integrate Factors
    integrated_factors = (volume_diff * 0.5) + (intraday_returns * 0.3) + (df['close'] - df['close'].shift(1))
    
    # Price Range and Volatility
    price_range_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    avg_daily_high_low_range = (df['high'] - df['low']).rolling(window=10).mean()
    volatility_multiplier = 1 / (avg_daily_high_low_range + 1)
    
    # Final Integrated Alpha Factor
    final_factor = (adjusted_momentum * average_volume * volatility_multiplier + integrated_factors) * volume_spike * price_range_volatility
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    final_factor *= volume_average * volatility
    
    return final_factor
```
