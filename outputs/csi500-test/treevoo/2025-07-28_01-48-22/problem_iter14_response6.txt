```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['20_day_momentum'] = df['daily_log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Add Intraday and Historical Momentum Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    df['volume_volatility'] = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    
    # Adjust Momentum
    df['adjusted_momentum'] = (df['20_day_momentum'] / df['volume_volatility']) * df['average_volume']
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Calculate Intraday and Open-to-Close Returns
    df['intraday_close_open_return'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    df['volume_spike'] = (df['volume'] > (mean_volume + 2 * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['combined_return'] = df['relative_strength'] * df['volume_spike']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Factors
    df['integrated_factors'] = 0.5 * df['volume_diff'] + 0.3 * (df['high'] - df['low']).rolling(window=10).mean() + df['close'] - df['close'].shift(1)
    
    # Price Range and Volatility
    df['price_range_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    average_high_low_range = (df['high'] - df['low']).rolling(window=10).mean()
    volatility_multiplier = 1 / (average_high_low_range + 1)
    
    # Final Integrated Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_momentum'] * df['average_volume'] * volatility_multiplier +
                                df['integrated_factors'] * df['volume_spike'] * df['price_range_volatility'])
    
    # Compute Volume Average
    df['volume_avg'] = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    
    # Weight by Volume and Volatility
    df['final_alpha_factor'] = df['final_alpha_factor'] * (df['volume_avg'] * df['volatility'])

    return df['final_alpha_factor']
```
