```python
def heuristics_v2(data):
    # Calculate the change in close price
    data['close_change'] = data['close'].diff()
    
    # Calculate the average of high and low prices
    data['avg_price'] = (data['high'] + data['low']) / 2
    
    # Calculate the money flow as the product of typical price and volume
    data['money_flow'] = data['avg_price'] * data['volume']
    
    # Calculate the difference between the amount and the money flow
    data['flow_diff'] = data['amount'] - data['money_flow']
    
    # Calculate the weighted close price change by the flow difference
    data['weighted_close_change'] = data['close_change'] * data['flow_diff']
    
    # Generate the alpha factor
    alpha_factor = data['weighted_close_change'].rolling(window=10).mean()
    
    return alpha_factor
```
