```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    N = 20
    K = 2
    volume_mean = df['volume'].rolling(window=N).mean()
    volume_std = df['volume'].rolling(window=N).std()
    volume_spike = (df['volume'] > (volume_mean + K * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_spike = combined_return * volume_spike
    
    # Calculate Volume Difference
    volume_diff = df['volume'].diff()
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Volume and Price Information
    adjusted_close_change = 0.5 * volume_diff + 0.3 * avg_high_low_range + close_to_prev_close_return
    
    # Combine Indicators
    final_factor = combined_return_spike * (avg_high_low_range * volume_diff * close_to_prev_close_return)
    
    # Weight by Volume and Volatility
    volume_avg = df['volume'].rolling(window=5).mean()
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    weighted_final_factor = final_factor * (volume_avg * volatility)
    
    return weighted_final_factor
```
