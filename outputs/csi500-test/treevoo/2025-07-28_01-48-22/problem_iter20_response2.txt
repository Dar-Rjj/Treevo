```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the close location in the daily range
    close_location = (df['close'] - df['low']) / daily_range
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the factor
    factor = (df['close'] - vwap) * close_location
    
    return factor
```
