import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) based on the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the price change over the last 5 days
    price_change_5d = df['close'] / df['close'].shift(5) - 1
    
    # Construct the alpha factor
    alpha_factor = rsi * (df['volume'] / avg_volume) * price_change_5d
    
    return alpha_factor
