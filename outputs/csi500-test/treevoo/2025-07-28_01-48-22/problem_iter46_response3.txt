```python
import pandas as pd

def heuristics_v2(df):
    # Momentum and Reversal Alpha Factors
    df['12_day_ma'] = df['close'].rolling(window=12).mean()
    df['26_day_ma'] = df['close'].rolling(window=26).mean()
    df['price_momentum'] = df['12_day_ma'] - df['26_day_ma']

    daily_returns = df['close'].pct_change()
    df['volatility'] = daily_returns.rolling(window=20).std()

    df['10_day_volume_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_change'] = (df['volume'] - df['10_day_volume_ma']) / df['10_day_volume_ma']

    df['intraday_return'] = df['close'] - df['open']
    df['upper_shadow'] = df['high'] - df['close']
    df['lower_shadow'] = df['open'] - df['low']

    # Liquidity and Market Sentiment
    # Assuming 'amount' is the market value of traded shares, we can infer the number of outstanding shares.
    # This is a simplification for the purpose of this example.
    df['turnover_rate'] = df['volume'] / (df['amount'] / df['close'])
    df['5_day_turnover_ma'] = df['turnover_rate'].rolling(window=5).mean()

    df['bid_ask_spread'] = df['high'] - df['low']
    df['percentage_spread'] = df['bid_ask_spread'] / df['close']

    # Fundamental-based Alpha Factors
    # Assuming 'book_value_per_share', 'earnings_per_share', and 'annual_dividend' are in the DataFrame.
    # These would typically be provided by a fundamental data source.
    df['price_to_book_ratio'] = df['close'] / df['book_value_per_share']
    df['earnings_yield'] = df['earnings_per_share'] / df['close']
    df['dividend_yield'] = df['annual_dividend'] / df['close']

    # Combine factors into a single alpha factor
    df['alpha_factor'] = (df['price_momentum'] 
                          + 0.5 * df['volatility'] 
                          + 0.3 * df['volume_change'] 
                          + 0.2 * df['intraday_return'] 
                          + 0.1 * (df['upper_shadow'] - df['lower_shadow']) 
                          + 0.4 * df['turnover_rate'] 
                          + 0.1 * df['percentage_spread'] 
                          + 0.2 * (1 / df['price_to_book_ratio']) 
                          + 0.1 * df['earnings_yield'] 
                          + 0.1 * df['dividend_yield'])

    return df['alpha_factor'].dropna()
```
