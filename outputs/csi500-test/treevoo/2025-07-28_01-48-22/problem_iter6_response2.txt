```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = df['volume'] * intraday_return
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(20)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Incorporate Volume Influence
    ema_volume = df['volume'].ewm(alpha=0.2).mean()
    momentum_with_volume = combined_momentum * ema_volume
    
    # Calculate Daily True Range
    previous_close = df['close'].shift(1)
    high_low = df['high'] - df['low']
    high_pc = np.abs(df['high'] - previous_close)
    low_pc = np.abs(df['low'] - previous_close)
    daily_true_range = pd.concat([high_low, high_pc, low_pc], axis=1).max(axis=1)
    
    # Calculate Average True Range
    atr = daily_true_range.rolling(window=20).mean()
    
    # Calculate Volatility Multiplier
    volatility_multiplier = 1 / (atr + 1)
    
    # Adjust Combined Momentum by Volatility
    final_factor = momentum_with_volume * volatility_multiplier
    
    return final_factor
```
