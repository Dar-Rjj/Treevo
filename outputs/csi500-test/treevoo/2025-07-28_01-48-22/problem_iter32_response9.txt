```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator: Short-term SMA (5 days) and Long-term SMA (20 days)
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    crossover_signal = (short_sma > long_sma).astype(int)

    # Volatility Measure: True Range and Average True Range (14 days)
    true_range = pd.DataFrame({
        'H-L': df['high'] - df['low'],
        'H-PC': df['high'] - df['close'].shift(1),
        'L-PC': df['low'] - df['close'].shift(1)
    }).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Volume Analysis Factors: VWAP and Volume Ratio
    cumulative_volume_price = (df['volume'] * df['close']).cumsum()
    total_cumulative_volume = df['volume'].cumsum()
    vwap = cumulative_volume_price / total_cumulative_volume
    volume_ratio = df['volume'] / df['volume'].shift(1)

    # Market Strength and Sentiment: RSI
    up_gain = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    down_loss = df['close'].diff().apply(lambda x: abs(x) if x < 0 else 0)
    avg_gain = up_gain.rolling(window=14).mean()
    avg_loss = down_loss.rolling(window=14).mean()
    rs_value = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs_value))

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow[df['close'] > df['close'].shift(1)].fillna(0)
    negative_money_flow = raw_money_flow[df['close'] < df['close'].shift(1)].fillna(0)
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))

    # Combine all factors into a composite alpha factor
    alpha_factor = (crossover_signal * 0.2 +
                    (atr.pct_change() * 0.2) +
                    (vwap.pct_change() * 0.15) +
                    (volume_ratio * 0.15) +
                    (rsi * 0.1) +
                    (mfi * 0.2))
    
    return alpha_factor
```
