import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['12_day_MA'] = df['close'].rolling(window=12).mean()
    df['26_day_MA'] = df['close'].rolling(window=26).mean()
    df['MACD'] = df['12_day_MA'] - df['26_day_MA']
    
    df['ROC_10'] = (df['close'] / df['close'].shift(10) - 1) * 100
    
    # Identify Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Evaluate Volume Changes
    df['volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['volume_pct_change'] = (df['volume'] / df['volume_MA_10'] - 1) * 100
    
    # Incorporate Intraday Price Patterns
    df['intraday_return'] = df['close'] - df['open']
    df['upper_shadow'] = df['high'] - df['close']
    df['lower_shadow'] = df['open'] - df['low']
    df['shadow_ratio'] = df['upper_shadow'] / df['lower_shadow']
    
    # Investigate Liquidity and Market Sentiment
    # Assuming 'outstanding_shares' is a column in the DataFrame
    df['turnover_rate'] = df['volume'] / df['outstanding_shares']
    df['turnover_MA_5'] = df['turnover_rate'].rolling(window=5).mean()
    
    df['bid_ask_spread'] = df['high'] - df['low']
    df['spread_pct'] = (df['bid_ask_spread'] / df['close']) * 100
    
    df['signed_tick'] = df['close'].diff().apply(lambda x: 1 if x > 0 else -1)
    df['cum_signed_tick_30'] = df['signed_tick'].rolling(window=30).sum()
    
    # Develop Fundamental-based Alpha Factors
    # Assuming 'book_value_per_share', 'earnings_per_share', and 'annual_dividend' are columns in the DataFrame
    df['price_to_book'] = df['close'] / df['book_value_per_share']
    df['earnings_yield'] = df['earnings_per_share'] / df['close']
    df['dividend_yield'] = df['annual_dividend'] / df['close']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['MACD'] + 
        df['ROC_10'] +
        df['volatility_20'] + 
        df['ATR_14'] + 
        df['volume_pct_change'] + 
        df['intraday_return'] + 
        df['shadow_ratio'] + 
        df['turnover_MA_5'] + 
        df['spread_pct'] + 
        df['cum_signed_tick_30'] + 
        df['price_to_book'] + 
        df['earnings_yield'] + 
        df['dividend_yield']
    )
    
    return df['alpha_factor']

# Example usage:
