```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(2)
    
    # Avg True Range (5 days)
    high_low = df['high'] - df['low']
    close_prev_close = (df['close'] - df['close'].shift(1)).abs()
    true_range = pd.concat([high_low, close_prev_close], axis=1).max(axis=1)
    avg_true_range = true_range.rolling(window=5).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(2)
    
    # Combine Factors
    combined_factor = (volume_momentum * 0.5) + (avg_true_range * 0.3) + close_price_change
    
    return combined_factor
```
