```python
import pandas as pd

def heuristics_v2(df):
    # Enhanced Combined Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    enhanced_combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = enhanced_combined_momentum.rolling(window=5).mean()
    
    # Volatility Multiplier
    daily_high_low_range = (df['high'] - df['low']) / df['close']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Volume-Price Indicator
    volume_change = 0.5 * (df['volume'] - df['volume'].shift(1))
    avg_high_low_range_5_days = (df['high'] - df['low']).rolling(window=5).sum() / 5 / df['close']
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_price_indicator = volume_change + 0.3 * avg_high_low_range_5_days + intraday_return
    
    # Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Combined Factors
    volume_spike_indicator = (df['volume'] - df['volume'].shift(1)) > 0
    weighted_intraday_return = 0.3 * (df['close'] - df['open']) / df['open']
    close_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_factors = relative_strength * volume_spike_indicator + weighted_intraday_return + close_price_change
    
    # Novel Alpha Component
    intraday_volatility = (df['high'] - df['low']) / df['close']
    historical_volatility = df['close'].rolling(window=10).std()
    momentum_volatility_ratio = (df['close'] - df['close'].shift(1)) / historical_volatility
    volume_adjusted_momentum = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    novel_alpha_component = intraday_volatility + historical_volatility + momentum_volatility_ratio + volume_adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_smoothed_momentum + volume_price_indicator + combined_factors + novel_alpha_component
    
    return final_alpha_factor
```
