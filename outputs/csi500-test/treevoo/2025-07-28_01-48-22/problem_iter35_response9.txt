```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price in the range [low, high]
    df['close_range_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow by multiplying the typical price with the volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Calculate the weighted money flow over the last 5 days
    df['weighted_money_flow'] = df['money_flow'].rolling(window=5).mean()
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = df['close_range_pos'] * df['weighted_money_flow']
    
    return df['alpha_factor']
```
