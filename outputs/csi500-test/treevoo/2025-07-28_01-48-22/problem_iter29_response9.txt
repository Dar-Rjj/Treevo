import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between close and open price
    df['rel_diff_close_open'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative difference between high and low price
    df['rel_diff_high_low'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the volume weighted by amount
    df['volume_weighted_by_amount'] = df['volume'] * df['amount']
    
    # Define a novel alpha factor
    df['alpha_factor'] = (df['rel_diff_close_open'] + df['rel_diff_high_low']) * df['volume_weighted_by_amount']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
