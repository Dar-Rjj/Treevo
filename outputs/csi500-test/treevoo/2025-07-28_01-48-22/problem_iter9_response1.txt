```python
import pandas as pd

def heuristics(df):
    # Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # High-Low Volatility (5 days)
    df['high_low_diff'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_diff'].rolling(window=5).std()
    
    # Close Price Momentum
    df['close_ma_21'] = df['close'].rolling(window=21).mean()
    df['close_price_momentum'] = df['close'] - df['close_ma_21'] - df['close'].shift(1)
    
    # Combine Factors
    df['factor'] = (df['volume_momentum'] * 0.4 
                    + df['high_low_volatility'] * 0.3 
                    + df['close_price_momentum'] * 0.6)
    
    return df['factor']
```
