```python
import pandas as pd

def heuristics_v2(df):
    # Volume Difference
    volume_diff = df['volume'].diff(1)
    
    # Calculate High-Low Range for day t
    high_low_range = df['high'] - df['low']
    
    # Avg High-Low Range (5 days)
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Close Change
    close_change = df['close'].diff(1)
    
    # Weight by Volume Trend
    volume_trend = volume_diff / df['volume'].shift(1)
    
    # Integrate Volume and Price Information
    integrated_info = (volume_diff * 0.5) + (avg_high_low_range * 0.3) + close_change
    
    # Combine Indicators
    combined_indicators = high_low_range * volume_trend * (df['close'] / df['close'].shift(1))
    
    # Final Alpha Factor
    final_alpha_factor = integrated_info + combined_indicators
    
    return final_alpha_factor
```
