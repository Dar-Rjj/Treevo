```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday and Historical Momentum Components
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    historical_momentum = df['close'] - df['close'].shift(15)
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Incorporate Volume Influence
    weighted_avg_volume = (df['volume'].rolling(window=15).apply(lambda x: sum(x / (np.arange(1, len(x)+1)))))
    combined_momentum_with_volume = combined_momentum * weighted_avg_volume
    
    # Incorporate Advanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_daily_range = daily_high_low_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_combined_momentum = combined_momentum_with_volume * volatility_multiplier
    
    # Volume Momentum
    volume_difference = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # High-Low Volatility (5 days)
    high_low_volatility = (df['high'] - df['low']).rolling(window=5).std()
    
    # Close Price Momentum
    close_price_change = (df['close'] - df['close'].rolling(window=21).mean()) - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = (volume_difference * 0.5) + (high_low_volatility * 0.3) + close_price_change
    
    # Final Factor
    final_factor = (adjusted_combined_momentum * hybrid_factor)
    
    return final_factor
```
