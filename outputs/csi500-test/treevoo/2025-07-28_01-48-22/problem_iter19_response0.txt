```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Average Daily Range
    avg_daily_range = daily_range.rolling(window=10).sum() / 10
    
    # Volatility Multiplier
    volatility_multiplier = 1 / (avg_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Integrate Volume and Price Information
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_high_low_5d = (df['high'] - df['low']).rolling(window=5).mean()
    close_change = df['close'] - df['close'].shift(1)
    
    integrated_info = 0.5 * volume_change + 0.3 * avg_high_low_5d + close_change
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_smoothed_momentum + integrated_info
    
    return final_alpha_factor
```
