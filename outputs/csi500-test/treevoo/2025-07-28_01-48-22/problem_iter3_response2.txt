import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Compute Volume Average
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Average True Range
    true_range = df.apply(lambda x: max(x['high'] - x['low'], 
                                        abs(x['high'] - df['close'].shift(1)), 
                                        abs(x['low'] - df['close'].shift(1))), axis=1)
    average_true_range = true_range.rolling(window=20).mean()
    
    # Multiply Relative Strength by (Volume Average * Average True Range)
    weighted_relative_strength = relative_strength * (volume_avg * average_true_range)
    
    # Compute Volume Momentum
    volume_momentum = (df['volume'] / df['volume'].shift(20)) - 1
    
    # Final Alpha Factor
    alpha_factor = weighted_relative_strength * volume_momentum
    
    return alpha_factor
