def heuristics_v2(df):
    # Define alpha factors
    df['alpha_factor_1'] = df['high'] - df['low']
    df['alpha_factor_2'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['alpha_factor_3'] = df['volume'] / (df['high'] - df['low'])
    df['alpha_factor_4'] = df['amount'] / (df['high'] - df['low'])
    df['alpha_factor_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['alpha_factor_6'] = (df['open'] < df['close']).astype(int) - (df['open'] > df['close']).astype(int)
    df['alpha_factor_7'] = (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])
    df['alpha_factor_8'] = (df['volume'] - df['volume'].shift(5)) / df['amount']
    df['alpha_factor_9'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['alpha_factor_10'] = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))
    df['alpha_factor_11'] = (df['high'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    df['alpha_factor_12'] = (df['high'] - df['low']) / (df['high'].shift(5) - df['low'].shift(5))

    # Combine all alpha factors into a single factor
    alpha_factors = [f'alpha_factor_{i}' for i in range(1, 13)]
    combined_factor = df[alpha_factors].sum(axis=1)

    return combined_factor
