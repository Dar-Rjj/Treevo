```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum Indicators
    df['Intraday_Momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['Historical_Momentum'] = (df['close'] - df['close'].shift(15)) / df['close'].shift(15)
    df['Combined_Momentum'] = df['Intraday_Momentum'] + 0.8 * df['Historical_Momentum']
    
    # Smoothed Momentum
    df['Smoothed_Momentum'] = df['Combined_Momentum'].ewm(span=7).mean()
    
    # Volatility-Adjusted Momentum
    df['Daily_High_Low_Range'] = (df['high'] - df['low']) / df['close']
    df['Average_Daily_Range'] = df['Daily_High_Low_Range'].rolling(window=15).mean()
    df['Adjusted_Smoothed_Momentum'] = df['Smoothed_Momentum'] / (df['Average_Daily_Range'] + 0.5)
    
    # Volume-Price Indicator
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    df['Avg_High_Low_Range_7_days'] = df['Daily_High_Low_Range'].rolling(window=7).mean()
    df['Volume_Price_Factor'] = df['Volume_Change'] + 0.4 * df['Avg_High_Low_Range_7_days'] + (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    
    # Return-Based Strength
    df['Short_Term_Return'] = (df['close'] / df['close'].shift(7)) - 1
    df['Long_Term_Return'] = (df['close'] / df['close'].shift(30)) - 1
    df['Relative_Strength'] = df['Short_Term_Return'] - df['Long_Term_Return']
    
    # Combined Factors
    df['Weighted_Intraday_Return'] = 0.6 * (df['close'] - df['open']) / df['open']
    df['Volume_Spike_Indicator'] = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    df['Multiplication_with_Relative_Strength'] = df['Relative_Strength'] * df['Volume_Spike_Indicator']
    df['Add_Close_Price_Change'] = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Adjusted_Smoothed_Momentum'] + 
        df['Volume_Price_Factor'] + 
        df['Multiplication_with_Relative_Strength'] + 
        0.7 * df['Add_Close_Price_Change']
    )
    
    return df['Final_Alpha_Factor'].dropna()
```
