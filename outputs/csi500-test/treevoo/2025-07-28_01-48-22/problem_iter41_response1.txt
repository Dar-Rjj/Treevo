```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    momentum_20 = daily_log_return.rolling(window=20).sum()
    
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combined Momentum
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smooth_momentum = combined_momentum.rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Adjusted Smoothed Momentum
    adjusted_smooth_momentum = smooth_momentum / (volume_volatility * average_volume)
    
    # Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Adjusted Close Change
    adjusted_close_change = df['close'] - df['close'].shift(1)
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Average High-Low Range
    average_high_low_range = high_low_range.rolling(window=10).mean()
    
    # Volume-Price Indicator
    volume_price_indicator = (volume_difference * 0.5) + (average_high_low_range * 0.3) + adjusted_close_change
    
    # Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Volume Spikes
    volume_std_dev = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (df['volume'].rolling(window=20).mean() + 2 * volume_std_dev)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_spikes = relative_strength * volume_spike_indicator
    
    # High-Low Range Over Average
    high_low_over_average = high_low_range / average_high_low_range
    
    # Final Factor
    final_factor = (
        adjusted_smooth_momentum +
        volume_price_indicator +
        relative_strength_spikes +
        high_low_over_average
    )
    
    return final_factor
```
