```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust for Volatility
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=10).sum() / 10
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Integrate Volume and Price Information
    volume_price_info = 0.5 * (df['volume'] - df['volume'].shift(1)) + 0.3 * (daily_range.rolling(window=5).mean()) + (df['close'] - df['close'].shift(1))
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Combine Factors
    combined_factors = relative_strength * (df['volume'] - df['volume'].shift(1)).abs() + 0.3 * (df['close'] - df['open']) + df['close'].diff()
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).sum() / 5
    volatility = (df['close'] - df['open']) / df['open'].rolling(window=20).std()
    integrated_factor = (adjusted_smoothed_momentum + volume_price_info + combined_factors) * (volume_average * volatility)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_smoothed_momentum + volume_price_info + integrated_factor
    
    return final_alpha_factor
```
