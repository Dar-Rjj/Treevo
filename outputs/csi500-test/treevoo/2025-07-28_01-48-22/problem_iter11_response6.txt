```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_return * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum with 5-day SMA
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Calculate Average Volume over the last 10 days
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Incorporate Volume Influence
    volume_influenced_momentum = smoothed_momentum * average_volume
    
    # Calculate Price Range and Volatility
    price_range_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Combine Momentum, Volume, and Volatility
    momentum_adjusted = volume_influenced_momentum * volatility_multiplier
    
    # Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Avg High-Low Range (5 days)
    avg_high_low_range_5_days = high_low_range.rolling(window=5).mean()
    
    # Close Change
    close_change = df['close'] - df['close'].shift(1)
    
    # Weight by Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Integrate Volume and Price Information
    integrated_volume_price = (volume_difference * 0.5) + (avg_high_low_range_5_days * 0.3) + close_change
    
    # Combine Indicators
    indicator_combined = high_low_range * volume_trend * close_change
    
    # Final Alpha Factor
    final_alpha_factor = momentum_adjusted + integrated_volume_price + indicator_combined
    
    return final_alpha_factor
```
