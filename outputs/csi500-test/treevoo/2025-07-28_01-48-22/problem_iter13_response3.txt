```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Determine the trend
    df['trend'] = (df['SMA_10'] - df['Smma_50']) / df['SMA_50']
    
    # Calculate Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['ATR'] = df['true_range'].rolling(window=14).mean()
    
    # Normalize ATR to a range of -1 to 1
    df['volatility'] = (df['ATR'] - df['ATR'].min()) / (df['ATR'].max() - df['ATR'].min()) * 2 - 1
    
    # Calculate support and resistance levels
    df['highest_high'] = df['high'].rolling(window=20).max()
    df['lowest_low'] = df['low'].rolling(window=20).min()
    
    # Determine proximity to support and resistance
    df['support_resistance'] = (df['close'] - df['lowest_low']) / (df['highest_high'] - df['lowest_low'])
    
    # Calculate average volume over the past 30 days
    df['avg_volume'] = df['volume'].rolling(window=30).mean()
    
    # Compare current day's volume to average volume
    df['volume_interest'] = (df['volume'] - df['avg_volume']) / df['avg_volume']
    
    # Assign weights to each indicator
    weight_trend = 0.4
    weight_volatility = -0.2
    weight_support_resistance = 0.2
    weight_volume_interest = 0.2
    
    # Form the composite alpha factor
    df['alpha_factor'] = (weight_trend * df['trend'] +
                          weight_volatility * df['volatility'] +
                          weight_support_resistance * df['support_resistance'] +
                          weight_volume_interest * df['volume_interest'])
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
```
