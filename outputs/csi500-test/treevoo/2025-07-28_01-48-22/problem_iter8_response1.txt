```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['momentum_20_day'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday and Historical Momentum Components
    df['intraday_momentum'] = ((df['high'] - df['low']) / df['open']) * df['volume']
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum with 5-day SMA
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['volume_change_percent'] = df['volume'].pct_change()
    df['volume_volatility'] = df['volume_change_percent'].rolling(window=20).std()
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = (df['momentum_20_day'] / df['volume_volatility']) * df['average_volume']
    
    # Price Range and Volatility
    df['price_range_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    df['daily_high_low_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_high_low_range'].rolling(window=10).mean()
    df['volatility_multiplier'] = 1 / (df['avg_daily_range'] + 1)
    
    # Combine Momentum, Volume, and Volatility
    df['alpha_factor'] = df['smoothed_momentum'] * df['average_volume'] * df['volatility_multiplier']
    
    return df['alpha_factor']
```
