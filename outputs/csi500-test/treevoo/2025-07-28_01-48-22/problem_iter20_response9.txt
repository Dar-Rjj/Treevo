```python
def heuristics_v2(df):
    """
    Generate an alpha factor using a combination of price and volume data.
    
    Parameters:
    - df: A pandas DataFrame indexed by date with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    - A pandas Series representing the alpha factor values, indexed by date.
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the relative strength index (RSI) for close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the weighted sum of the three factors
    factor = (close_change * 0.5) + (volume_change * 0.3) + (rsi * 0.2)
    
    return factor
```
