def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average price using open, close, and volume
    weighted_avg_price = (df['open'] * df['volume'] + df['close'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the momentum factor as the difference between the weighted average price and the simple average price
    momentum_factor = weighted_avg_price - avg_price
    
    # Calculate the money flow ratio as the ratio of amount to volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # Generate the alpha factor by combining the momentum factor and the money flow ratio
    alpha_factor = momentum_factor * money_flow_ratio
    
    return alpha_factor
