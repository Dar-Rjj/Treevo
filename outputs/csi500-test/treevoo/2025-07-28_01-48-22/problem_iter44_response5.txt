```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day Momentum
    twenty_day_momentum = daily_log_return.rolling(window=20).sum()
    
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combined Momentum
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    volume_volatility = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = (smoothed_momentum / volume_volatility * average_volume)
    
    # Volume-Price Indicator
    volume_difference = df['volume'] - df['volume'].shift(1)
    average_high_low_range = (df['high'] - df['low']).rolling(window=5).mean() / df['close']
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_price_indicator = (volume_difference * 0.5) + (average_high_low_range * 0.3) + intraday_return
    
    # Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Volume Spikes
    volume_std_dev = df['volume'].rolling(window=20).std()
    mean_volume = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (mean_volume + 2 * volume_std_dev)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_volume_spike = relative_strength * volume_spike_indicator
    
    # High-Low Range Over Average
    high_low_range = df['high'] - df['low']
    average_high_low_range_10_days = (df['high'] - df['low']).rolling(window=10).mean() / df['close']
    high_low_range_over_average = high_low_range / average_high_low_range_10_days
    
    # Novel Alpha Component
    intraday_volatility = (df['high'] - df['low']) / df['close']
    historical_volatility = df['close'].rolling(window=10).std()
    momentum_volatility_ratio = (df['close'] - df['close'].shift(1)) / historical_volatility
    volume_adjusted_momentum = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    
    # Trend Following Factor
    ten_day_sma = df['close'].rolling(window=10).mean()
    twenty_day_sma = df['close'].rolling(window=20).mean()
    trend_signal = ten_day_sma - twenty_day_sma
    
    # Additional Novel Components
    intraday_momentum_adjusted = (df['close'] - df['open']) / df['open'] * (df['volume'] / df['volume'].shift(1))
    historical_momentum_adjusted = (ten_day_sma - df['close'].shift(10)) / df['close'].shift(10)
    composite_volatility = (intraday_volatility + historical_volatility) / 2
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    # Final Alpha Factor
    final_alpha_factor = (
        adjusted_smoothed_momentum +
        volume_price_indicator +
        relative_strength_volume_spike +
        high_low_range_over_average +
        (intraday_volatility + historical_volatility + momentum_volatility_ratio + volume_adjusted_momentum) +
        trend_signal +
        (intraday_momentum_adjusted + historical_momentum_adjusted + composite_volatility + volume_weighted_momentum)
    )
    
    return final_alpha_factor
```
