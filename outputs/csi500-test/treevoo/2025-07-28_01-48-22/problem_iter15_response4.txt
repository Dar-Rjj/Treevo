```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log return
    df['log_return'] = np.log(df['close']).diff()

    # Compute 20-day momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()

    # Intraday and Historical Momentum Components
    df['intraday_momentum'] = (df['high'] - df['low']) * ((df['close'] - df['open']) / df['open']) * df['volume']
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']

    # Smooth combined momentum with EMA over 5 days
    df['smoothed_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()

    # Volume influence
    df['volume_pct_change'] = df['volume'].pct_change()
    df['volume_volatility_20'] = df['volume_pct_change'].rolling(window=20).std()
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = (df['momentum_20'] / df['volume_volatility_20']) * df['avg_volume_10']

    # Short-term and Long-term returns
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']

    # Weight by volume and volatility
    df['volume_avg_5'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility_20'] = df['daily_returns'].rolling(window=20).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_avg_5'] * df['volatility_20'])

    # Incorporate trading range
    df['high_low_range'] = (df['high'].shift(1) - df['low'].shift(1)) / df['low'].shift(1)
    df['avg_trading_range_5'] = df['high_low_range'].rolling(window=5).mean()
    df['adjusted_relative_strength'] = df['weighted_relative_strength'] * (1 + df['avg_trading_range_5'])

    # Integrate amount
    df['amount_avg_5'] = df['amount'].rolling(window=5).mean()
    df['final_factor_adjustment'] = df['adjusted_relative_strength'] * (df['amount_avg_5'] / df['volume_avg_5'])

    # Enhanced price and volume interaction signal
    df['enhanced_pv_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)

    # Combine momentum, volume, and price interaction
    df['alpha_factor'] = df['smoothed_momentum'] * df['avg_volume_10'] * df['final_factor_adjustment'] * df['enhanced_pv_signal']

    return df['alpha_factor']
```
