```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    median_volume = df['volume'].rolling(window=20).median()
    K = 2  # Example threshold, can be adjusted
    volume_spike_indicator = (df['volume'] > (median_volume * K)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_with_spike = combined_return * volume_spike_indicator
    
    # Compute Volume Average
    volume_average = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    volatility = ((df['close'] - df['open']) / df['open']).rolling(window=20).std()
    
    # Weight by Volume and Volatility
    factor_values = relative_strength * (volume_average * volatility)
    
    return factor_values
```
