```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the open to the close
    price_movement = df['close'] - df['open']
    
    # Calculate the buying pressure using the amount and volume
    buying_pressure = (df['amount'] / df['volume']) - df['close']
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_movement + 2 * buying_pressure) / daily_range
    
    return alpha_factor
```
