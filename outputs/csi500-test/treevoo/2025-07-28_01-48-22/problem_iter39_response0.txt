import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'] / df['close'].shift(5) - 1
    
    # Calculate Long-Term Return
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Combine Momentum Components
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.ewm(span=7, adjust=False).mean()
    
    # Adjust Momentum
    volume_volatility = df['volume'].rolling(window=20).std()
    average_volume = df['volume'].rolling(window=10).sum() / 10
    adjusted_momentum = smoothed_momentum / volume_volatility * average_volume
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Combine Intraday and Open-to-Close Returns
    intraday_return = df['high'] - df['low']
    open_to_close_return = (df['close'] - df['open']) / df['open']
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # Threshold for spike detection
    volume_spike_indicator = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_with_spike = combined_return * volume_spike_indicator
    
    # Calculate VWAP
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Integrate Volume and Price Information
    volume_difference = df['volume'] - df['volume'].shift(1)
    adjusted_close_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_adjustments = (volume_difference * 0.6) + (avg_high_low_range * 0.4) + adjusted_close_change
    
    # Combine Indicators
    combined_indicators = high_low_over_avg * volume_difference * vwap * adjusted_close_change
    
    # Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).sum() / 5
    daily_returns = df['close'] / df['close'].shift(1) - 1
    volatility = daily_returns.rolling(window=20).std()
    weighted_relative_strength = relative_strength * volume_average * volatility
    
    # Final Alpha Factor
    alpha_factor = (adjusted_momentum + combined_return_with_spike + combined_indicators + weighted_relative_strength) / 4
    return alpha_factor
