```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate 5-Day and 20-Day Volume-Weighted Close Price Moving Averages
    df['5d_vol_weighted_ma'] = df['volume_weighted_close'].rolling(window=5).mean()
    df['20d_vol_weighted_ma'] = df['volume_weighted_close'].rolling(window=20).mean()
    
    # Calculate Momentum
    df['momentum'] = df['5d_vol_weighted_ma'] - df['20d_vol_weighted_ma']
    
    # Calculate Volume Surge Indicator
    df['5d_vol_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_surge_indicator'] = (df['volume'] > 1.5 * df['5d_vol_ma']).astype(int)
    
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    df['prev_day_spread'] = df['intraday_spread'].shift(1)
    
    # Calculate Intraday Spread Ratio
    df['intraday_spread_ratio'] = df['intraday_spread'] / df['prev_day_spread']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Signals for Alpha Factor
    combined_momentum_score = 0.6 * df['intraday_spread_ratio'] + 0.4 * df['close_to_open_return']
    
    # Final Alpha Factor
    df['alpha_factor'] = combined_momentum_score * df['volume_surge_indicator']
    
    return df['alpha_factor']
```
