```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_return = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike_indicator
    
    # Integrate Factors
    integrated_factors = 0.3 * intraday_return + combined_return
    
    # Incorporate Momentum and Reversal Signals
    momentum_signal = (df['close'] / df['close'].shift(120)) - 1
    reversal_signal = (df['close'] / df['close'].shift(7)) - 1
    combined_momentum_reversal = momentum_signal - reversal_signal
    integrated_factors += combined_momentum_reversal
    
    # Enhance with Additional Price Movement Indicators
    volatility = df['close'].rolling(window=20).std()
    range_ = df['high'] - df['low']
    
    # Calculate Average True Range (ATR) over N Days
    tr = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    atr = tr.rolling(window=14).mean()
    normalized_atr = atr / df['close']
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (
        0.3 * intraday_return +
        combined_return +
        volatility +
        range_ +
        normalized_atr
    )
    
    return final_alpha_factor
```
