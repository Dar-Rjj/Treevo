import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_returns = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Integrate Factors
    integrated_factor = (0.5 * volume_diff) + (0.3 * intraday_returns) + (df['close'] - df['close'].shift(1))
    
    # Adjusted Integrated Factor
    # Compute Volume Average
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    
    # Final Adjusted Integrated Factor
    adjusted_integrated_factor = integrated_factor * (volume_avg * volatility)
    
    return adjusted_integrated_factor
