```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum
    historical_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Combine Momentum Indicators
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Daily High-Low Range
    daily_high_low_range = (df['high'] - df['low']) / df['close']
    
    # Average Daily Range
    average_daily_range = daily_high_low_range.rolling(window=10).sum() / 10
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum / (average_daily_range + 1)
    
    # Volume-Price Indicator
    volume_change = 0.5 * (df['volume'] - df['volume'].shift(1))
    avg_high_low_range_5_days = (df['high'] - df['low']).rolling(window=5).sum() / (df['close'] * 5)
    volume_price_factor = volume_change + 0.3 * avg_high_low_range_5_days + (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Return-Based Strength
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    relative_strength = short_term_return - long_term_return
    
    # Volume Spikes
    volume_std_dev = df['volume'].rolling(window=10).std()
    volume_spike_indicator = (df['volume'] > (df['volume'].rolling(window=10).mean() + 2 * volume_std_dev)).astype(int)
    
    # Multiply Relative Strength by Volume Spike Indicator
    relative_strength_by_volume_spike = relative_strength * volume_spike_indicator
    
    # High-Low Range Over Average
    high_low_range = df['high'] - df['low']
    average_high_low_range = (df['high'] - df['low']).rolling(window=10).mean()
    high_low_range_over_average = high_low_range / average_high_low_range
    
    # Final Alpha Factor
    final_alpha_factor = (
        adjusted_smoothed_momentum * 0.4 +
        volume_price_factor * 0.3 +
        relative_strength_by_volume_spike * 0.2 +
        high_low_range_over_average * 0.1 +
        (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    )
    
    return final_alpha_factor
```
