```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: Series with index (date) representing the alpha factor values.
    """
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the percentage change in volume
    pct_change_volume = df['volume'].pct_change().fillna(0)
    
    # Calculate the range of the day
    day_range = df['high'] - df['low']
    
    # Calculate the weighted average of the day's range and the percentage change in volume
    weighted_avg = (day_range * 0.7) + (pct_change_volume * 0.3)
    
    # Calculate the final alpha factor
    alpha_factor = avg_price * weighted_avg
    
    return alpha_factor
```
