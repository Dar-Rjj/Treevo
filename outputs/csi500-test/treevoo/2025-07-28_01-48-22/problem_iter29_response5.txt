```python
def heuristics_v2(df):
    """
    Generate an alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate a combined factor using the above calculated components
    factor = (price_range * volume_change) / close_change.abs()
    
    return factor
```
