```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_return * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(20)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Enhanced Volume Influence
    weights = 1 / np.arange(1, 21)
    weighted_avg_volume = (df['volume'].rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True) / weights.sum())
    
    combined_momentum_vol = combined_momentum * weighted_avg_volume
    
    # Enhanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=15, min_periods=15, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    
    adjusted_combined_momentum = combined_momentum_vol * volatility_multiplier
    
    # Volume Change Rate
    volume_change_rate = df['volume'] / df['volume'].shift(1)
    
    # Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=7).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = 0.6 * volume_change_rate + 0.4 * avg_high_low_range + close_price_change
    
    # Final Factor
    final_factor = (hybrid_factor * adjusted_combined_momentum).dropna()
    
    return final_factor
```
