import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    # Calculate lagged daily return
    df['lagged_daily_return'] = (df['close'].shift(2) - df['close'].shift(3)) / df['close'].shift(3)
    
    # Combine current and lagged daily returns with weights
    df['composite_return_factor'] = 0.6 * df['daily_return'] + 0.4 * df['lagged_daily_return']
    
    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']
    # Calculate moving average of the high-low range over a 5-day window
    df['high_low_range_ma'] = df['high_low_range'].rolling(window=5).mean()
    # Calculate the ratio of the current high-low range to its moving average
    df['volatility_factor'] = df['high_low_range'] / df['high_low_range_ma']
    
    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Create a composite factor by combining the volume change with the daily return
    df['composite_volume_factor'] = 0.5 * df['daily_return'] + 0.5 * df['volume_change']
    
    # Combine the daily return, high-low range, and volume change in a multivariate model
    df['multivariate_factor'] = 0.4 * df['daily_return'] + 0.3 * df['high_low_range'] + 0.3 * df['volume_change']
    
    # Non-linear combination of factors
    df['nonlinear_factor'] = (df['daily_return'] * df['high_low_range']) + df['volume_change']**2
    
    # Return the nonlinear factor as the final alpha factor
    return df['nonlinear_factor']
