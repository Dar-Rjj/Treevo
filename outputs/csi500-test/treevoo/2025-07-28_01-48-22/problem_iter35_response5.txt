```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['crossover_signal'] = (df['SMA_5'] > df['SMA_20']).astype(int)

    # Volatility Measure
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    df['average_true_range_14'] = df['true_range'].rolling(window=14).mean()

    # Enhanced Price Dynamics
    df['alpha_factor_1'] = df['high'] - df['low']
    df['alpha_factor_2'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['alpha_factor_7'] = (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])

    # Volume Analysis Factors
    df['volume_weighted_average_price'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)

    # Volume and Amount Dynamics
    df['alpha_factor_3'] = df['volume'] / (df['high'] - df['low'])
    df['alpha_factor_4'] = df['amount'] / (df['high'] - df['low'])
    df['alpha_factor_8'] = (df['volume'] - df['volume'].shift(5)) / df['amount']

    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow[df['close'] > df['close'].shift(1)]
    negative_flow = raw_money_flow[df['close'] < df['close'].shift(1)]
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))

    # Enhanced Momentum and Trend
    df['alpha_factor_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['alpha_factor_9'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['alpha_factor_6'] = ((df['open'] < df['close']).astype(int) - (df['open'] > df['close']).astype(int))
    df['alpha_factor_10'] = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))

    # Advanced Volatility Measures
    df['alpha_factor_11'] = (df['high'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    df['alpha_factor_12'] = (df['high'] - df['low']) / (df['high'].shift(5) - df['low'].shift(5))

    # Return the factor values
    return df[['alpha_factor_1', 'alpha_factor_2', 'alpha_factor_3', 'alpha_factor_4',
               'alpha_factor_5', 'alpha_factor_6', 'alpha_factor_7', 'alpha_factor_8',
               'alpha_factor_9', 'alpha_factor_10', 'alpha_factor_11', 'alpha_factor_12']]
```
