```python
import pandas as pd

def heuristics_v2(df, k=2):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    prev_day_spread = df['high'].shift(1) - df['low'].shift(1)
    
    # Compare Intraday and Previous Day Spreads
    spread_ratio = intraday_spread / prev_day_spread
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Open-to-Close Return by Volume Spike Indicator
    multiplied_combined_return = open_to_close_return * volume_spike_indicator
    
    # Combine Signals for Alpha Factor
    alpha_factor = 0.4 * spread_ratio + 0.3 * open_to_close_return + 0.3 * multiplied_combined_return
    
    return alpha_factor
```
