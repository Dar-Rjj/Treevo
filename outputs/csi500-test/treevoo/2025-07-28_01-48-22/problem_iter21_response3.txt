import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_return = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    k = 2  # Standard deviation multiplier
    volume_spike = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = (short_term_return - long_term_return) * volume_spike
    
    # Integrate Factors
    volume_difference = (df['volume'] - df['volume'].rolling(window=7).mean())
    high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    close_price_change = df['close'] - df['close'].shift(1)
    
    integrated_factor = (
        0.5 * volume_difference +
        0.3 * high_low_range +
        close_price_change
    )
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).sum() / 5
    volatility = (df['close'] - df['open']).pct_change().rolling(window=20).std()
    adjusted_integrated_factor = integrated_factor * (volume_average * volatility)
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    advanced_volatility_adjusted = combined_return * (volume_average * volatility_multiplier)
    
    # Close Price Momentum
    close_momentum = df['close'] - df['close'].rolling(window=21).mean() - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    trend_following = (df['close'].rolling(window=10).mean() > df['close'].rolling(window=30).mean()).astype(int)
    trend_adjusted = trend_following.replace({1: 1.1, 0: 0.9})
    final_factor = (
        0.5 * volume_difference +
        0.3 * high_low_range +
        close_price_change
    ) * trend_adjusted
    
    # Include Volume and Price Correlation
    volume_close_corr = df['close'].rolling(window=10).corr(df['volume'])
    correlation_adjusted = volume_close_corr.apply(lambda x: 1.05 if x > 0 else 0.95)
    final_factor *= correlation_adjusted * advanced_volatility_adjusted
    
    return final_factor
