```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum Indicators
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = (df['close'] - df['close'].shift(15)) / df['close'].shift(15)
    combined_momentum = intraday_momentum + 0.8 * historical_momentum

    # Smoothed Momentum
    smoothed_momentum = combined_momentum.ewm(span=7, adjust=False).mean()

    # Volatility-Adjusted Momentum
    daily_high_low_range = (df['high'] - df['low']) / df['close']
    avg_daily_range = daily_high_low_range.rolling(window=15).mean()
    adjusted_smoothed_momentum = smoothed_momentum / (avg_daily_range + 0.5)

    # Volume-Price Indicator
    volume_change = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    avg_high_low_range_7d = (df['high'] - df['low']) / df['close']
    avg_high_low_range_7d = avg_high_low_range_7d.rolling(window=7).mean()
    volume_price_factor = volume_change + 0.4 * avg_high_low_range_7d + (df['close'] - df['close'].shift(2)) / df['close'].shift(2)

    # Return-Based Strength
    short_term_return = (df['close'] / df['close'].shift(7)) - 1
    long_term_return = (df['close'] / df['close'].shift(30)) - 1
    relative_strength = short_term_return - long_term_return

    # Final Alpha Factor
    alpha_factor = (
        adjusted_smoothed_momentum
        + volume_price_factor
        + relative_strength * (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
        + 0.7 * (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    )

    return alpha_factor
```
