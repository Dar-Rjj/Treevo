```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Smooth Daily Price Momentum
    df['Smoothed_Momentum'] = df['Daily_Momentum'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Momentum_Volume_Adjusted'] = df['Smoothed_Momentum'] * df['Average_Volume']
    
    # Calculate Price Range and Volatility
    df['Price_Range_Volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Calculate Daily High-Low Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate Average Daily Range
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=10).mean()
    
    # Calculate Volatility Multiplier
    df['Volatility_Multiplier'] = 1 / (df['Average_Daily_Range'] + 1)
    
    # Combine Momentum, Volume, and Volatility
    df['Alpha_Factor'] = df['Momentum_Volume_Adjusted'] * df['Volatility_Multiplier']
    
    return df['Alpha_Factor']
```
