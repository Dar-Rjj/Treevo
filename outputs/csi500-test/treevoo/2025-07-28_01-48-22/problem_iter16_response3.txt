```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(20)
    
    # Combined Momentum
    combined_momentum = intraday_momentum + historical_momentum
    
    # Enhanced Volume Influence
    weighted_volume = (df['volume'].rolling(window=20).sum() * (1 / np.arange(1, 21)).sum())
    combined_momentum *= weighted_volume
    
    # Enhanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=15).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    combined_momentum *= volatility_multiplier
    
    # Volume Change Rate
    volume_change_rate = df['volume'] / df['volume'].shift(1)
    
    # Average High-Low Range for past 7 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=7).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factors = (volume_change_rate * 0.6) + (avg_high_low_range * 0.4) + close_price_change
    
    # Final Factor
    final_factor = combined_momentum + hybrid_factors
    
    return final_factor
```
