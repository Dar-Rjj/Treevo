import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = intraday_return * df['volume']

    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)

    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum

    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()

    # Incorporate Volume Influence
    average_volume = df['volume'].rolling(window=10).mean()
    momentum_volume_adjusted = smoothed_momentum * average_volume

    # Calculate Price Range and Volatility
    price_range_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']

    # Incorporate Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)

    # Combine Momentum, Volume, and Volatility
    alpha_factor = momentum_volume_adjusted * volatility_multiplier

    return alpha_factor
