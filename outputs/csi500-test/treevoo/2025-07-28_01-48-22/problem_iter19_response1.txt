```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Incorporate Volume Influence
    weighted_volume = (df['volume'].rolling(window=15).apply(lambda x: sum(x / (np.arange(1, 16))))
                       / sum(1 / np.arange(1, 16)))
    momentum_with_volume = combined_momentum * weighted_volume
    
    # Calculate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    initial_ema = daily_range.rolling(window=10).mean().iloc[9]
    ema_daily_range.iloc[0] = initial_ema
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_momentum = momentum_with_volume * volatility_multiplier
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # High-Low Volatility (5 days)
    high_low_volatility = (df['high'] - df['low']).rolling(window=5).std()
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].rolling(window=21).mean() - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = (volume_momentum * 0.5) + (high_low_volatility * 0.3) + close_price_momentum
    
    # Trend Following Adjustment
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_adjustment = np.where(sma_10 > sma_30, 1.1, 0.9)
    final_hybrid_factor = hybrid_factor * trend_adjustment
    
    # Final Factor
    final_factor = final_hybrid_factor * adjusted_momentum
    
    return final_factor
```
