```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Compute Volume Average
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    
    # Weight by Volume and Volatility
    weighted_relative_strength = relative_strength * (volume_avg * volatility)
    
    # Incorporate Trading Range
    trading_range = (df['high'].shift(1) - df['low'].shift(1)) / df['low'].shift(1)
    avg_trading_range = trading_range.rolling(window=5).mean()
    adjusted_relative_strength = weighted_relative_strength * (1 + avg_trading_range)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    median_volume = df['volume'].rolling(window=20).median()
    volume_spike = (df['volume'] > median_volume * 1.5).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_spike = combined_return * volume_spike
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Weight by Volume-to-VWAP Ratio
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    volume_difference = 0.5 * (df['volume'] - volume_avg)
    range_weight = 0.3 * avg_high_low_range
    adjusted_close_change = close_to_prev_close_return
    
    integrated_info = volume_difference + range_weight + adjusted_close_change
    
    # Combine Indicators
    intermediate_factor = (high_low_range / avg_high_low_range) * volume_to_vwap_ratio * close_to_prev_close_return
    
    # Integrate Amount
    amount_avg = df['amount'].rolling(window=5).mean()
    final_adjusted_rs = adjusted_relative_strength * (amount_avg / volume_avg)
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_factor + final_adjusted_rs + combined_return_spike
    
    return final_alpha_factor
```
