import pandas as pd

def heuristics_v2(df):
    # Calculate short and long SMAs
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Determine SMA crossover signal
    df['SMA_Crossover'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Calculate Price Rate of Change (ROC) over 14 days
    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() / df['close'].abs()).fillna(0) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Calculate True Range (TR)
    df['TR'] = df[['high', 'low']].diff(axis=1).iloc[:, 0]
    df['TR'] = df[['TR', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    
    # Calculate Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Simple ensemble of multiple indicators
    df['Combined_Factor'] = (
        df['SMA_Crossover'] +
        df['ROC_14'] +
        df['OBV'].pct_change() +
        df['TR'] +
        (df['RSI_14'] > 70).astype(int) - (df['RSI_14'] < 30).astype(int)
    )
    
    return df['Combined_Factor']
