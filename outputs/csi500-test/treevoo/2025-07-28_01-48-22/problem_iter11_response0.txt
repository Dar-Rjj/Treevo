```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = df['volume'] * intraday_return
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(15)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum with 5-day SMA
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Incorporate Volume Influence
    n_days = 15
    weights = 1 / np.arange(1, n_days + 1)
    weighted_volume = (df['volume'].rolling(window=n_days).apply(lambda x: (x * weights).sum(), raw=True) /
                       weights.sum())
    volume_influence = smoothed_momentum * weighted_volume
    
    # Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, min_periods=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_momentum = volume_influence * volatility_multiplier
    
    # Price Range and Volatility
    price_volatility = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = 0.5 * volume_diff + 0.3 * avg_high_low_range + close_price_change
    
    # Final Factor
    final_factor = (adjusted_momentum + hybrid_factor) * price_volatility
    
    return final_factor
```
