```python
def heuristics_v2(df):
    # Calculate Volume Difference
    volume_difference = df['volume'].diff(1)
    
    # Calculate Average High-Low Range
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Adjusted Close Change
    close_change = df['close'].diff(1)
    
    # Integrate Volume and Price Information
    volume_component = volume_difference * 0.5
    range_component = avg_high_low_range * 0.3
    integrated_factor = volume_component + range_component + close_change
    
    return integrated_factor
```
