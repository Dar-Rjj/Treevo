```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Moving Averages
    df['5_day_vol_wt_close_ma'] = df['volume_weighted_close'].rolling(window=5).mean()
    df['20_day_vol_wt_close_ma'] = df['volume_weighted_close'].rolling(window=20).mean()
    
    # Momentum
    df['momentum'] = df['5_day_vol_wt_close_ma'] - df['20_day_vol_wt_close_ma']
    
    # Volume Surge
    df['5_day_volume_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = (df['volume'] > 1.5 * df['5_day_volume_ma']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum'] * df['volume_surge']
    
    return df['alpha_factor']
```
