```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Integrated Momentum Indicators
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = (df['close'].diff(10)) / df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.ewm(span=5).mean()
    
    # Volatility-Adjusted Momentum
    daily_high_low_range = (df['high'] - df['low']) / df['close']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    adjusted_smoothed_momentum = smoothed_momentum / (average_daily_range + 1)
    
    # Volume-Price Indicator
    volume_change = (df['volume'].pct_change())
    avg_high_low_range_5_days = daily_high_low_range.rolling(window=5).mean()
    volume_price_factor = volume_change + 0.3 * avg_high_low_range_5_days + (df['close'].pct_change())
    
    # Return-Based Strength
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    relative_strength = short_term_return - long_term_return
    
    # Combined Factors
    weighted_intraday_return = 0.4 * (df['close'] - df['open']) / df['open']
    volume_spike_indicator = (df['volume'].pct_change())
    multiplication_with_relative_strength = relative_strength * volume_spike_indicator
    add_close_price_change = (df['close'].pct_change())
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_smoothed_momentum + volume_price_factor + multiplication_with_relative_strength + 0.5 * add_close_price_change
    
    return final_alpha_factor
```
