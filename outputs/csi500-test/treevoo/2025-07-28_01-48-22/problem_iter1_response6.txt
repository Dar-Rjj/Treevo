```python
def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Incorporate Volume Influence
    average_volume = df['volume'].rolling(window=10).mean()
    momentum_volume = momentum * average_volume
    
    # Incorporate Volatility Adjustment
    daily_range = df['high'] - df['low']
    average_daily_range = daily_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)
    adjusted_momentum = momentum_volume * volatility_multiplier
    
    return adjusted_momentum
```
