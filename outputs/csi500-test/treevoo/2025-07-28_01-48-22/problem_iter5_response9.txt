```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    df['close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Average High-Low Range
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['high_low_range_over_avg'] = df['high_low_range'] / df['avg_high_low_range']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['normalized_volume_change'] = df['volume_change'] / df['volume'].shift(1)
    
    # Weight by Volume-to-VWAP Ratio
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['volume_to_vwap_ratio'] = df['volume'] / df['vwap']
    
    # Integrate Volume and Price Information
    df['volume_weighted'] = df['volume_diff'] * 0.5
    df['range_weighted'] = df['avg_high_low_range'] * 0.3
    df['price_movement_impact'] = df['close_return']
    
    # Combine Indicators
    df['intermediate_factor'] = (df['high_low_range_over_avg'] *
                                 df['normalized_volume_change'] *
                                 df['volume_to_vwap_ratio'] *
                                 df['close_return'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['intermediate_factor'] + df['volume_weighted'] + df['range_weighted'] + df['price_movement_impact']
    
    return df['final_alpha_factor'].dropna()
```
