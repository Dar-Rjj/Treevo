```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    k = 2  # threshold for standard deviation
    volume_spike = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Combine Factors
    combined_factor = relative_strength * volume_spike + (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).sum() / 5
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    adjusted_integrated_factor = combined_factor * (volume_average * volatility)
    
    # Calculate Combined Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust for Volatility
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).sum() / 10
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_integrated_factor + adjusted_smoothed_momentum
    
    return final_alpha_factor
```
