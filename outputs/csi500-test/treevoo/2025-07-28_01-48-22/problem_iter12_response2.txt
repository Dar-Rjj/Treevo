```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate 20-Day Volume Volatility
    df['volume_change'] = df['volume'].pct_change()
    df['volume_volatility'] = df['volume_change'].rolling(window=20).std()
    
    # Adjust 20-Day Momentum
    df['adjusted_momentum'] = df['momentum_20'] / df['volume_volatility']
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Weight by Volume and Volatility
    df['volume_average'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_average'] * df['volatility'])
    
    # Incorporate Trading Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    df['average_trading_range'] = df['high_low_range'].rolling(window=5).mean()
    df['adjusted_relative_strength'] = df['weighted_relative_strength'] * (1 + df['average_trading_range'])
    
    # Integrate Amount
    df['amount_average'] = df['amount'].rolling(window=5).mean()
    df['final_factor_adjustment'] = df['adjusted_relative_strength'] * (df['amount_average'] / df['volume_average'])
    
    # Calculate Daily Price Momentum
    df['daily_price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Exponential Moving Average (EMA) of Volume
    df['ema_volume_5'] = df['volume'].ewm(span=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['daily_price_momentum'] * df['ema_volume_5']
    
    # Calculate Enhanced Price and Volume Interaction Signal
    df['enhanced_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Adjust Relative Strength by Intraday Volatility
    df['final_alpha_factor'] = df['final_factor_adjustment'] * df['intraday_volatility']
    
    return df['final_alpha_factor']
```
