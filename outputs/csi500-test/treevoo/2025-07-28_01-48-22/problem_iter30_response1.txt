```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    k = 2  # Number of standard deviations
    volume_spike_indicator = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike_indicator
    
    # Integrate Factors
    volume_avg = df['volume'].rolling(window=5).sum() / 5
    volatility = (df['close'] - df['open']).div(df['open']).rolling(window=20).std()
    adjusted_factor = combined_return * volume_avg * volatility
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_factor *= volatility_multiplier
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # Close Price Momentum
    close_price_momentum = df['close'] - df['close'].rolling(window=21).mean() - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = 0.5 * volume_momentum + close_price_momentum
    
    # Incorporate Trend Following Adjustment
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_following_adjustment = np.where(sma_10 > sma_30, 1.1, 0.9)
    hybrid_factor *= trend_following_adjustment
    
    # Include Volume and Price Correlation
    correlation = df[['volume', 'close']].rolling(window=10).corr().unstack().iloc[::2, 1].reset_index(level=0, drop=True)
    hybrid_factor = np.where(correlation > 0, 1.05 * hybrid_factor, 0.95 * hybrid_factor)
    
    # Enhance with Price-Earnings Ratio
    pe_ratio = df['close'] / df['earnings_per_share']
    pe_ma_50 = pe_ratio.rolling(window=50).mean()
    pe_adjustment = np.where(pe_ratio < pe_ma_50, 1.05, 0.95)
    hybrid_factor *= pe_adjustment
    
    # Final Factor
    final_factor = adjusted_factor + hybrid_factor
    return final_factor
```
