def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the relative strength of the close price compared to the open, high, and low
    relative_strength = (df['close'] - df[['open', 'high', 'low']].min(axis=1)) / (df[['open', 'high', 'low']].max(axis=1) - df[['open', 'high', 'low']].min(axis=1))
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the momentum factor based on the VWAP
    momentum = (vwap - vwap.shift(5)) / vwap.shift(5)
    
    # Combine the relative strength and momentum factors
    factor = relative_strength * momentum
    
    return factor
