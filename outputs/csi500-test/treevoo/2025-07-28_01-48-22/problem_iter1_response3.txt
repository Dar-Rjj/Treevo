```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Volatility
    intraday_range_volatility = df['high'] - df['low']
    
    # Calculate Volume Scaled Intraday Movement
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Calculate Weighted Moving Average of Intraday Returns
    weighted_moving_average = intraday_return.rolling(window=5).mean()
    
    # Combine Indicators
    combined_indicator = (volume_scaled_intraday_movement - intraday_range_volatility + 
                          weighted_moving_average)
    
    return combined_indicator
```
