```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns for open, high, low, close, amount, and volume.
    
    Returns:
    pd.Series: A Series with (date) as index and the generated alpha factor values.
    """
    
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average price
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the normalized trading volume
    norm_volume = df['volume'] / df['amount']
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change()
    
    # Combine the factors into a single alpha factor
    alpha_factor = daily_range * norm_volume * momentum * avg_price
    
    return alpha_factor
```
