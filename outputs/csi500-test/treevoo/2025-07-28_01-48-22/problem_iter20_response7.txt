```python
def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with index as date and columns as 
      [open, high, low, close, amount, volume].
      
    Returns:
    - pd.Series: An alpha factor indexed by date.
    """
    # Calculate the price range
    price_range = data['high'] - data['low']
    
    # Calculate the buying pressure
    buying_pressure = (data['close'] - data['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (data['high'] - data['close']) / price_range
    
    # Calculate the volume weighted buying pressure
    volume_weighted_buying_pressure = buying_pressure * data['volume']
    
    # Calculate the volume weighted selling pressure
    volume_weighted_selling_pressure = selling_pressure * data['volume']
    
    # Calculate the net volume pressure
    net_volume_pressure = volume_weighted_buying_pressure - volume_weighted_selling_pressure
    
    # Calculate the smoothed net volume pressure
    smoothed_net_volume_pressure = net_volume_pressure.ewm(span=5).mean()
    
    # Calculate the alpha factor
    alpha_factor = smoothed_net_volume_pressure / data['volume']
    
    return alpha_factor
```
