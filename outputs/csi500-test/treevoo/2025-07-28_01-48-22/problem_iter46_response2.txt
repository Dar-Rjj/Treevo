import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['20d_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate 20-Day Volume Volatility
    df['volume_change'] = df['volume'].pct_change()
    df['20d_vol_volatility'] = df['volume_change'].rolling(window=20).std()
    
    # Adjust 20-Day Momentum by Volume Volatility
    df['adjusted_20d_momentum'] = df['20d_momentum'] / (df['20d_vol_volatility'] + 1e-8)
    
    # Calculate Relative Strength
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Weight by Volume and Volatility
    df['volume_avg'] = df['volume'].rolling(window=5).mean()
    df['volatility'] = ((df['close'] - df['open']) / df['open']).rolling(window=20).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_avg'] * df['volatility'])
    
    # Incorporate Trading Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    df['avg_trading_range'] = df['high_low_range'].rolling(window=5).mean()
    df['adjusted_weighted_relative_strength'] = df['weighted_relative_strength'] * (1 + df['avg_trading_range'])
    
    # Integrate Amount
    df['amount_avg'] = df['amount'].rolling(window=5).mean()
    df['final_factor'] = df['adjusted_weighted_relative_strength'] * (df['amount_avg'] / (df['volume_avg'] + 1e-8))
    
    # Calculate Volume-Adjusted Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['price_momentum'] * df['ema_volume']
    
    # Enhance with Intraday Indicators
    df['enhanced_pv_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    df['intraday_reversal'] = (df['high'] - df['low']) / (df['open'] - df['close'] + 1e-8)
    df['final_factor'] = df['final_factor'] * (1 + df['intraday_reversal'])
    
    # Incorporate Price and Volume Correlation
    df['pv_corr'] = df['close'].rolling(window=20).corr(df['volume'])
    df['final_factor'] = df['final_factor'] * (1 + df['pv_corr'])
    
    # Introduce Trend Following Component
    df['50d_sma'] = df['close'].rolling(window=50).mean()
    df['trend_following_signal'] = np.where(df['close'] > df['50d_sma'], 1.2, 0.8)
    df['final_factor'] = df['final_factor'] * df['trend_following_signal']
    
    return df['final_factor'].dropna()
