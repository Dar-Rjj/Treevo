```python
import pandas as pd

def heuristics_v2(df):
    # Alpha factor 1: Difference between high and low prices as an indicator of volatility
    alpha_factor_1 = df['high'] - df['low']
    
    # Alpha factor 2: Significance of closing price in relation to the day's range
    alpha_factor_2 = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Alpha factor 3: Measure the ratio of volume traded to the price change
    alpha_factor_3 = df['volume'] / (df['high'] - df['low'])
    
    # Alpha factor 4: Assess the financial intensity by calculating the trading value per unit of price movement
    alpha_factor_4 = df['amount'] / (df['high'] - df['low'])
    
    # Alpha factor 5: Determine the rate of change over a specified period
    alpha_factor_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Alpha factor 6: Analyze the consistency of the closing price being above or below the opening price
    alpha_factor_6 = (df['open'] < df['close']).astype(int) - (df['open'] > df['close']).astype(int)
    
    # Combine all factors into a single DataFrame
    combined_factors = pd.DataFrame({
        'alpha_factor_1': alpha_factor_1,
        'alpha_factor_2': alpha_factor_2,
        'alpha_factor_3': alpha_factor_3,
        'alpha_factor_4': alpha_factor_4,
        'alpha_factor_5': alpha_factor_5,
        'alpha_factor_6': alpha_factor_6
    })
    
    # Return the combined factors
    return combined_factors
```
