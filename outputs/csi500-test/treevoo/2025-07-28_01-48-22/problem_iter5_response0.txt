import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Average High-Low Range for past 5 days
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(4)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(19)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Compute Volume Average for past 5 days
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    
    # Weight by Volume and Volatility
    weighted_relative_strength = relative_strength * (volume_avg * volatility)
    
    # Incorporate Trading Range
    trading_range = (df['high'] - df['low']) / df['low']
    avg_trading_range = trading_range.rolling(window=5).mean()
    adjusted_relative_strength = weighted_relative_strength * (1 + avg_trading_range)
    
    # Weight by Volume-to-VWAP Ratio
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    volume_diff_weight = 0.5 * (df['volume'] - df['volume'].shift(1))
    range_weight = 0.3 * avg_high_low_range
    adjusted_close_change = close_return
    
    integrated_factor = (high_low_range / avg_high_low_range) * volume_to_vwap_ratio * close_return
    
    # Integrate Amount
    amount_avg = df['amount'].rolling(window=5).mean()
    final_adjusted_relative_strength = adjusted_relative_strength * (amount_avg / volume_avg)
    
    # Final Alpha Factor
    final_alpha_factor = integrated_factor + final_adjusted_relative_strength
    
    return final_alpha_factor
