```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Incorporate Volume Influence
    average_volume = df['volume'].rolling(window=10).mean()
    combined_momentum *= average_volume
    
    # Incorporate Enhanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    ema_daily_range = daily_high_low_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    combined_momentum_volatility_adjusted = combined_momentum * volatility_multiplier
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close_price = df['close'] * df['volume']
    
    # Calculate 5-Day and 20-Day Volume-Weighted Close Price Moving Averages
    vwc_5_day_ma = volume_weighted_close_price.rolling(window=5).mean()
    vwc_20_day_ma = volume_weighted_close_price.rolling(window=20).mean()
    
    # Calculate Momentum
    momentum = vwc_5_day_ma - vwc_20_day_ma
    
    # Calculate Volume Surge Indicator
    volume_5_day_ma = df['volume'].rolling(window=5).mean()
    volume_surge_indicator = (df['volume'] > 1.5 * volume_5_day_ma).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff()
    
    # Compute 20-Day Momentum
    twenty_day_momentum = daily_log_return.rolling(window=20).sum()
    
    # Calculate Volume Volatility
    volume_percentage_change = df['volume'].pct_change()
    volume_volatility = volume_percentage_change.rolling(window=20).std()
    
    # Adjust 20-Day Momentum by Inverse Volume Volatility
    adjusted_twenty_day_momentum = twenty_day_momentum / volume_volatility
    
    # Final Alpha Factor
    alpha_factor = (momentum * volume_surge_indicator) * adjusted_twenty_day_momentum
    return alpha_factor
```
