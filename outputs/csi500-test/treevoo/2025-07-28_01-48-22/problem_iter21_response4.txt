```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the average of high and low
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the relative difference between close and the average of high and low
    df['rel_diff_close_avg'] = (df['close'] - df['avg_high_low']) / df['avg_high_low']
    
    # Calculate the volume weighted amount
    df['volume_weighted_amount'] = df['amount'] * df['volume']
    
    # The alpha factor is a combination of these components
    alpha_factor = df['rel_diff_close_avg'] * df['close_change'] * df['volume_weighted_amount']
    
    return alpha_factor
```
