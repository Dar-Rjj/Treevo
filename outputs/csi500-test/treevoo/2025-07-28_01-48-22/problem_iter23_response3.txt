import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of close price with a 10-day window
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of close price with a 10-day window
    df['close_10d_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the z-score of the close price
    df['close_z_score'] = (df['close'] - df['close_10d_mean']) / df['close_10d_std']
    
    # Calculate the ratio of volume to its 10-day moving average
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Define a new factor based on the z-score of the close price and the volume ratio
    df['alpha_factor'] = (df['close_z_score'] * df['volume_ratio']) / (df['amount'] / df['volume'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
