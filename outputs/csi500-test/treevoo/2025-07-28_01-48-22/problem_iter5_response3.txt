```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_return = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Calculate Average High-Low Range over the past 5 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Change and normalize by previous day's volume
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Integrate Volume and Price Information
    weighted_volume_diff = volume_diff * 0.5
    weighted_avg_high_low_range = avg_high_low_range * 0.3
    adjusted_close_return = close_return
    
    # Combine Indicators
    intermediate_factor = (high_low_range / avg_high_low_range) * volume_change * close_return
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_factor + weighted_volume_diff + weighted_avg_high_low_range + adjusted_close_return
    
    return final_alpha_factor
```
