```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday and Open-to-Close Returns
    intraday_returns = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_spike = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike
    
    # Integrate Factors
    integrated_factors = combined_return + 0.3 * intraday_returns
    
    # Incorporate Momentum and Reversal Signals
    momentum_signal = (df['close'] / df['close'].shift(120)) - 1
    reversal_signal = (df['close'] / df['close'].shift(7)) - 1
    momentum_reversal = momentum_signal - reversal_signal
    
    # Add to the Integrated Factors
    integrated_factors += momentum_reversal
    
    # Enhance with Additional Price Pattern
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    moving_average_crossover = (sma_5 > sma_20).astype(int)
    
    # Multiply by the Integrated Factors
    enhanced_factors = integrated_factors * moving_average_crossover
    
    # Final Factor
    final_factor = integrated_factors + enhanced_factors
    
    return final_factor
```
