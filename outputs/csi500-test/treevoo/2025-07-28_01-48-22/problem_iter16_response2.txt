```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Average Volume
    average_volume = df['volume'].rolling(window=10).sum() / 10
    
    # Smoothed Momentum * Average Volume
    momentum_volume = smoothed_momentum * average_volume
    
    # Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Average Daily Range
    average_daily_range = daily_range.rolling(window=10).sum() / 10
    
    # Volatility Multiplier
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Combine Momentum, Volume, and Volatility
    momentum_volume_volatility = momentum_volume * volatility_multiplier
    
    # Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Integrate Volume and Price Information
    avg_high_low_5d = (df['high'] - df['low']).rolling(window=5).mean()
    integrate_info = 0.5 * (df['volume'] - df['volume'].shift(1)) + 0.3 * avg_high_low_5d + (df['close'] - df['close'].shift(1))
    
    # Final Alpha Factor
    alpha_factor = momentum_volume_volatility + integrate_info
    
    return alpha_factor
```
