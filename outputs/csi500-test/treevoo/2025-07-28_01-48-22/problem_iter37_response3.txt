```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day weighted moving average (WMA) of closing prices
    weights = [0.1, 0.2, 0.3, 0.2, 0.1]
    df['5_day_WMA'] = df['close'].rolling(window=5).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Generate a signal by comparing today's close price to the 5-day WMA
    df['wma_signal'] = df.apply(lambda row: 1 if row['close'] > row['5_day_WMA'] else -1, axis=1)
    
    # Calculate the combined percentage change in the closing price from 10 and 20 days ago to today
    df['pct_change_10'] = df['close'].pct_change(10)
    df['pct_change_20'] = df['close'].pct_change(20)
    df['combined_momentum'] = df['pct_change_10'] + df['pct_change_20']
    
    # Compute the cumulative sum of the difference between daily close and open prices over the past 20 trading days
    df['daily_diff'] = df['close'] - df['open']
    df['cumulative_sum'] = df['daily_diff'].rolling(window=20).sum()
    df['smoothed_cumulative_sum'] = df['cumulative_sum'].ewm(span=20).mean()
    
    # Calculate the 20-day average trading volume
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['std_vol_20'] = df['volume'].rolling(window=20).std()
    
    # Flag any day with volume significantly (e.g., 3 standard deviations) above the average
    df['vol_anomaly'] = df.apply(lambda row: 1 if row['volume'] > (row['avg_vol_20'] + 3 * row['std_vol_20']) else 0, axis=1)
    
    # Combine this volume anomaly with the sign of the day's return and the direction of the 5-day WMA
    df['return_sign'] = df['close'] - df['open']
    df['nuanced_signal'] = df.apply(lambda row: row['wma_signal'] * row['return_sign'] * row['vol_anomaly'], axis=1)
    
    # Calculate the range (high minus low) for each day
    df['range'] = df['high'] - df['low']
    
    # Compute the 10-day moving average of these ranges
    df['avg_range_10'] = df['range'].rolling(window=10).mean()
    
    # Interpret the relative range
    df['relative_range'] = df['range'] / df['avg_range_10']
    
    # Combine all signals into a single alpha factor
    df['alpha_factor'] = df['wma_signal'] + df['combined_momentum'] + df['smoothed_cumulative_sum'] + df['nuanced_signal'] + df['relative_range']
    
    return df['alpha_factor']
```
