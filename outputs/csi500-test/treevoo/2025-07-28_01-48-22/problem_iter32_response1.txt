def heuristics_v2(df):
    # Calculate the money flow multiplier
    mfm = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mfm * df['volume']
    
    # Calculate the cumulative money flow volume
    cmf = mf_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Calculate the relative strength index (RSI) for 14 periods
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the CMF and RSI to create the alpha factor
    alpha_factor = (cmf - 0.5) * rsi
    
    return alpha_factor
