```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    prev_day_spread = (df['high'].shift(1) - df['low'].shift(1))
    
    # Compare Intraday and Previous Day Spreads
    spread_ratio = intraday_spread / prev_day_spread
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_spread + close_to_open_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_with_spike = combined_return * volume_spike_indicator
    
    # Weighted Sum of Intraday Spread Ratio, Close-to-Open Return, and Relative Strength
    weighted_sum = 0.3 * spread_ratio + 0.2 * close_to_open_return + 0.5 * relative_strength
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Trend
    volume_trend = volume_diff / df['volume'].shift(1)
    
    # Integrate Volume and Price Information
    integrated_info = 0.5 * volume_diff + 0.3 * avg_high_low_range + (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    intermediate_factor = (high_low_range / avg_high_low_range) * volume_trend * close_to_prev_close_return
    
    # Final Alpha Factor
    final_alpha_factor = weighted_sum + integrated_info + intermediate_factor
    
    return final_alpha_factor
```
