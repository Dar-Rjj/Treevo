import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    # Calculate lagged daily return
    df['lagged_daily_return'] = (df['close'].shift(2) - df['close'].shift(3)) / df['close'].shift(3)
    # Combine current and lagged daily returns in a weighted sum
    df['composite_return'] = 0.5 * df['daily_return'] + 0.5 * df['lagged_daily_return']
    
    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']
    # Calculate moving average of high-low range over 5 days
    df['high_low_range_ma_5'] = df['high_low_range'].rolling(window=5).mean()
    # Use the ratio of the current high-low range to its moving average as a measure of volatility
    df['volatility_factor'] = df['high_low_range'] / df['high_low_range_ma_5']
    
    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    # Create a composite factor by combining the volume change with the daily return
    df['composite_volume_return'] = df['daily_return'] * df['volume_change']
    
    # Combine daily return, high-low range, and volume change in a multivariate model
    df['multivariate_factor'] = 0.3 * df['daily_return'] + 0.3 * df['high_low_range'] + 0.4 * df['volume_change']
    
    # Return the final alpha factor
    return df['multivariate_factor'].dropna()
