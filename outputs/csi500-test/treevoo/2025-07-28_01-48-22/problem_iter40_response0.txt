def heuristics_v2(df):
    # Momentum Indicator: Crossover Signal
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['crossover_signal'] = (df['SMA_5'] > df['SMA_20']).astype(int)

    # Volatility Measure: Average True Range (14 days)
    df['TR'] = df[['high', 'low']].apply(lambda x: max(x), axis=1) - df[['high', 'low']].apply(lambda x: min(x), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()

    # Enhanced Price Dynamics: Price Change Relative to Range
    df['alpha_factor_7'] = (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])

    # Volume Analysis Factors: Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)

    # Volume and Amount Dynamics: Volume to Price Range
    df['alpha_factor_3'] = df['volume'] / (df['high'] - df['low'])

    # Volume and Amount Dynamics: Volume Change Relative to Amount
    df['alpha_factor_8'] = (df['volume'] - df['volume'].shift(5)) / df['amount']

    # Market Strength and Sentiment: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Market Strength and Sentiment: Money Flow Index (MFI)
    tp = (df['high'] + df['low'] + df['close']) / 3
    mf = tp * df['volume']
    positive_mf = mf.where(df['close'] > df['close'].shift(1), 0)
    negative_mf = mf.where(df['close'] < df['close'].shift(1), 0)
    mfr = positive_mf.rolling(window=14).sum() / negative_mf.rolling(window=14).sum()
    df['MFI'] = 100 - (100 / (1 + mfr))

    # Enhanced Momentum and Trend: Close Price Momentum
    df['alpha_factor_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Enhanced Momentum and Trend: Trend Direction Indicator
    df['trend_direction_indicator'] = ((df['open'] < df['close']).astype(int) - (df['open'] > df['close']).astype(int))

    # Enhanced Momentum and Trend: Trend Strength Indicator
    df['alpha_factor_10'] = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))

    # Advanced Volatility Measures: Volatility Ratio
    df['alpha_factor_11'] = (df['high'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))

    # Advanced Volatility Measures: Range Volatility
    df['alpha_factor_12'] = (df['high'] - df['low']) / (df['high'].shift(5) - df['low'].shift(5))

    return df['alpha_factor_7'].fillna(0)
