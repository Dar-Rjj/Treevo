```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Average High-Low Range for the past 5 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-to-VWAP Ratio
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    volume_weighted_info = volume_diff * 0.5
    range_weighted_info = avg_high_low_range * 0.3
    price_movement_impact = close_to_prev_close_return
    
    # Combine Indicators
    intermediate_factor = (high_low_range / avg_high_low_range) * volume_to_vwap_ratio * close_to_prev_close_return
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_factor + volume_weighted_info + range_weighted_info + price_movement_impact
    
    return final_alpha_factor
```
