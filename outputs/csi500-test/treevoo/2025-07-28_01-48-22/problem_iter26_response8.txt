```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the close-open difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate the volume-weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Combine the three measures into a single alpha factor
    alpha_factor = (high_low_ratio + close_open_diff + volume_weighted_close) / 3
    
    return alpha_factor
```
