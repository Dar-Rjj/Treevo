import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    twenty_day_momentum = daily_log_return.rolling(window=20).sum()
    
    # Combine Momentum Components
    intraday_momentum = (df['close'] - df['open']) * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smooth_combined_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust Momentum
    volume_volatility = df['volume'].diff().rolling(window=20).std()
    average_volume = df['volume'].rolling(window=10).mean()
    adjusted_momentum = (twenty_day_momentum / volume_volatility) * average_volume
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range Over Average
    high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5, center=True).mean()
    high_low_over_avg = (df['high'] - df['low']) / avg_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (mean_volume + 2 * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_returns_spiked = combined_returns * volume_spike_indicator
    
    # Integrate Volume and Price Information
    volume_diff = df['volume'] - df['volume'].shift(1)
    integrated_info = (volume_diff * 0.5) + (avg_high_low_range * 0.3) + (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    combined_indicators = high_low_over_avg * volume_diff * ((df['high'] + df['low'] + df['close']) / 3) * (df['close'] - df['close'].shift(1))
    
    # Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    volatility = daily_returns.rolling(window=20).std()
    adjusted_relative_strength = relative_strength * (volume_average * volatility)
    
    # Final Factor Value
    final_factor = adjusted_relative_strength + integrated_info + combined_indicators
    
    # Incorporate Recent Volume Trend
    five_day_volume_trend = (df['volume'] / df['volume'].shift(5)) - 1
    final_factor *= five_day_volume_trend
    
    return final_factor.dropna()
