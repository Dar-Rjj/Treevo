```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Difference
    volume_diff = df['volume'].diff()
    
    # Calculate Average High-Low Range for the past 5 days
    high_low_range_avg = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Close Price Change
    close_price_change = df['close'].diff()
    
    # Integrate Factors
    factor = (0.5 * volume_diff) + (0.3 * high_low_range_avg) + close_price_change
    
    return factor
```
