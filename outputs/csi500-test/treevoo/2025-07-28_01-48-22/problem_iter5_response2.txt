import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday and Historical Momentum Components
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    df['Volume_Scaled_Intraday_Movement'] = df['Intraday_Return'] * df['volume']
    df['Historical_Momentum'] = df['close'] - df['close'].shift(10)
    df['Combined_Momentum'] = df['Volume_Scaled_Intraday_Movement'] + df['Historical_Momentum']

    # Smooth Combined Momentum with SMA over 5 days
    df['Smoothed_Momentum'] = df['Combined_Momentum'].rolling(window=5).mean()

    # Incorporate Volume Influence
    df['Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Smoothed_Momentum_Volume_Adjusted'] = df['Smoothed_Momentum'] * df['Average_Volume']

    # Calculate Price Range and Volatility
    df['Price_Range_Volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']

    # Incorporate Volatility Adjustment
    df['Daily_High_Low_Range'] = df['high'] - df['low']
    df['Average_Daily_Range'] = df['Daily_High_Low_Range'].rolling(window=10).mean()
    df['Volatility_Multiplier'] = 1 / (df['Average_Daily_Range'] + 1)

    # Combine Momentum, Volume, and Volatility
    df['Alpha_Factor'] = df['Smoothed_Momentum_Volume_Adjusted'] * df['Volatility_Multiplier']

    return df['Alpha_Factor']
