```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = df['close'].apply(lambda x: x if pd.isnull(x) else (x - df['close'].shift(1)) / df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['volume_change'] = df['volume'].pct_change()
    df['volume_volatility'] = df['volume_change'].rolling(window=20).std()
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = (df['momentum_20'] / df['volume_volatility']) * df['avg_volume']
    
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Weight by Volume and Volatility
    df['volume_avg'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_avg'] * df['volatility'])
    
    # Incorporate Trading Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    df['avg_trading_range'] = df['high_low_range'].rolling(window=5).mean()
    df['adjusted_relative_strength'] = df['weighted_relative_strength'] * (1 + df['avg_trading_range'])
    
    # Integrate Amount
    df['amount_avg'] = df['amount'].rolling(window=5).mean()
    df['final_adjusted_relative_strength'] = df['adjusted_relative_strength'] * (df['amount_avg'] / df['volume_avg'])
    
    # Enhanced Price and Volume Interaction Signal
    df['price_vol_signal'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Combine Momentum, Volume, and Price Interaction
    factor = df['smoothed_momentum'] * df['avg_volume'] * df['final_adjusted_relative_strength'] * df['price_vol_signal']
    
    return factor
```
