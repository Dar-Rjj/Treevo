```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Compute Volume Average
    volume_avg = df['volume'].rolling(window=5).mean()
    
    # Compute Volatility
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    
    # Weight by Volume and Volatility
    weighted_rs = relative_strength * (volume_avg * volatility)
    
    # Incorporate Trading Range
    high_low_range = (df['high'] - df['low']) / df['low']
    avg_trading_range = high_low_range.rolling(window=5).mean()
    adjusted_rs = weighted_rs * (1 + avg_trading_range)
    
    # Integrate Amount
    amount_avg = df['amount'].rolling(window=5).mean()
    final_factor = adjusted_rs * (amount_avg / volume_avg)
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # threshold for volume spike
    volume_spike_indicator = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_spiked = combined_return * volume_spike_indicator
    
    # Integrate Volume and Price Information
    volume_diff = 0.5 * (df['volume'] - df['volume'].shift(1))
    avg_high_low_range = 0.3 * (df['high'] - df['low']).rolling(window=5).mean()
    adjusted_close_change = df['close'] - df['close'].shift(1)
    enhanced_signal = volume_diff + avg_high_low_range + adjusted_close_change
    
    # Calculate Enhanced Price and Volume Interaction Signal
    interaction_signal = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Introduce Intraday Momentum Indicator
    intraday_return = (df['close'] / df['open']) - 1
    intraday_momentum = intraday_return - intraday_return.shift(5)
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    intraday_momentum_weighted = intraday_momentum * ema_volume
    
    # Final Alpha Factor
    alpha_factor = final_factor + combined_return_spiked + enhanced_signal + interaction_signal + intraday_momentum_weighted
    return alpha_factor
```
