```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Momentum Indicator
    short_term_momentum = df['close'] - df['close'].shift(1)
    
    # Long-Term Price Change
    long_term_price_change = (df['close'] / df['close'].shift(20)) - 1
    
    # Volume Analysis
    average_volume = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > (average_volume + 2 * df['volume'].rolling(window=5).std())).astype(int)
    
    # Intraday and Open-to-Close Movement
    intraday_movement = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Combined Momentum
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    relative_strength = short_term_return - long_term_price_change
    adjusted_momentum = relative_strength * volume_spike + short_term_momentum
    integrated_factor = adjusted_momentum + intraday_movement
    
    # Volatility and Stability Measures
    daily_log_returns_std = np.log(df['close']).diff().rolling(window=20).std()
    intraday_intensity = (df['high'] - df['low']) / df['close']
    combined_volatility_intensity = daily_log_returns_std * intraday_intensity
    
    # Adjusted Integrated Factor
    adjusted_integrated_factor = integrated_factor * (average_volume * combined_volatility_intensity)
    
    # Additional Momentum Components
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    smoothed_momentum = intraday_momentum.rolling(window=5).mean()
    
    # Adjust for Average Daily Range
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Integration of Volume and Price Information
    integrated_volume_price_info = 0.5 * (df['volume'] - df['volume'].shift(1)) + \
                                   0.3 * (df['high'] - df['low']).rolling(window=5).mean() + \
                                   (df['close'] - df['close'].shift(1))
    
    # Integration of Volume and Price Momentum
    integrated_volume_price_momentum = 0.4 * (df['volume'] - df['volume'].shift(1)) + \
                                       0.6 * (df['close'] - df['close'].shift(1))
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_integrated_factor + adjusted_smoothed_momentum + \
                         integrated_volume_price_info + integrated_volume_price_momentum
    
    return final_alpha_factor.dropna()
```
