import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day simple moving average of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Generate a signal based on comparison of today's close to the SMA
    df['SMA_Signal'] = (df['close'] > df['SMA_5']).astype(int) * 2 - 1
    
    # Calculate the percentage change in closing price from 10 days ago to today
    df['Momentum_10'] = df['close'].pct_change(periods=10)
    
    # Compute the difference between daily close and open prices
    df['Daily_Diff'] = df['close'] - df['open']
    
    # Create a cumulative sum of these differences over the past 20 trading days
    df['Cumulative_Diff_20'] = df['Daily_Diff'].rolling(window=20).sum()
    
    # Calculate the 20-day average trading volume
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Flag significant volume anomalies
    df['Volume_Anomaly'] = (df['volume'] > (df['Avg_Volume_20'] + 2 * df['volume'].rolling(window=20).std())).astype(int)
    
    # Combine volume anomaly with the sign of the day's return
    df['Volume_Return_Signal'] = df['Volume_Anomaly'] * (df['close'] - df['open']) / (df['close'] + df['open'])
    
    # Calculate the daily range (high - low)
    df['Range'] = df['high'] - df['low']
    
    # Compute the 10-day moving average of ranges
    df['Range_MA_10'] = df['Range'].rolling(window=10).mean()
    
    # Normalize the current day's range by its 10-day moving average
    df['Normalized_Range'] = df['Range'] / df['Range_MA_10']
    
    # Combining all the signals into one alpha factor
    df['Alpha_Factor'] = (
        df['SMA_Signal'] * 0.3 +
        df['Momentum_10'] * 0.2 +
        df['Cumulative_Diff_20'] * 0.1 +
        df['Volume_Return_Signal'] * 0.2 +
        df['Normalized_Range'] * 0.2
    )
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
