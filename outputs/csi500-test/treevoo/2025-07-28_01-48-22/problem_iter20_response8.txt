```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range and the real body of the candle
    df['price_range'] = df['high'] - df['low']
    df['real_body'] = abs(df['close'] - df['open'])

    # Calculate the upper and lower shadows
    df['upper_shadow'] = df['high'] - df['close'] if df['close'] > df['open'] else df['high'] - df['open']
    df['lower_shadow'] = df['open'] - df['low'] if df['close'] > df['open'] else df['close'] - df['low']

    # Calculate the volume-weighted price
    df['vwp'] = (df['amount'] / df['volume'])

    # Calculate the factor based on the ratio of the real body to the price range, adjusted by the volume-weighted price
    factor = (df['real_body'] / df['price_range']) * df['vwp']

    return factor
```
