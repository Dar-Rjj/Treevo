```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_spiked = combined_return * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate Close Price Change
    close_price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Factors
    integrated_factor = (0.5 * volume_difference) + (0.3 * avg_high_low_range) + close_price_change
    
    # Adjusted Integrated Factor
    adjusted_integrated_factor = integrated_factor + combined_return_spiked + relative_strength
    
    # Final Factor: Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    final_factor = adjusted_integrated_factor * (volume_average * volatility)
    
    return final_factor
```
