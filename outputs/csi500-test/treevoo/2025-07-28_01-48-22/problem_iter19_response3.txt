import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_avg = high_low_range / avg_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_spike_indicator = (df['volume'] > (volume_mean + 2 * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return_with_spike = combined_return * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Volume-to-VWAP Ratio
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    integrated_info = 0.5 * volume_difference + 0.3 * avg_high_low_range + (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    factor_value = (high_low_over_avg * volume_difference * volume_to_vwap_ratio * 
                    (df['close'] - df['close'].shift(1)))
    
    # Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).mean()
    volatility = df['close'].pct_change().rolling(window=20).std()
    weighted_factor = relative_strength * (volume_average * volatility)
    
    # Final Alpha Factor
    alpha_factor = factor_value * weighted_factor
    
    return alpha_factor
