```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: The generated alpha factor indexed by date.
    """
    # Calculate the difference between high and low price
    price_range = df['high'] - df['low']
    
    # Calculate the ratio of close to open price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of volume to amount
    volume_amount_ratio = df['volume'] / df['amount']
    
    # Combine the factors using a weighted sum
    alpha_factor = 0.5 * price_range + 0.3 * close_open_ratio + 0.2 * volume_amount_ratio
    
    return alpha_factor
```
