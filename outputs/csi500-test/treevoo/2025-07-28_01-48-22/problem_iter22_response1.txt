```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['LogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    df['Momentum_20D'] = df['LogReturn'].rolling(window=22).sum()
    
    # Calculate Short-Term Return
    df['ShortTermReturn'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['LongTermReturn'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['RelativeStrength'] = df['ShortTermReturn'] - df['LongTermReturn']
    
    # Intraday Momentum Component
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    df['IntradayMomentum'] = df['IntradayReturn'] * df['volume']
    
    # Historical Momentum Component
    df['HistoricalMomentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    df['CombinedMomentum'] = df['IntradayMomentum'] + df['HistoricalMomentum']
    
    # Smooth Combined Momentum
    df['SmoothedMomentum'] = df['CombinedMomentum'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['VolumeChange'] = df['volume'].pct_change()
    df['VolumeVolatility'] = df['VolumeChange'].rolling(window=20).std()
    df['AverageVolume'] = df['volume'].rolling(window=10).mean()
    df['AdjustedMomentum'] = (df['Momentum_20D'] / df['VolumeVolatility']) * df['AverageVolume']
    
    # Integrate Factors
    df['VolumeDifference'] = df['volume'] - df['volume'].shift(1)
    df['HighLowRange'] = (df['high'] - df['low'])
    df['ClosePriceChange'] = df['close'] - df['close'].shift(1)
    df['IntegratedFactor'] = 0.5 * df['VolumeDifference'] + 0.3 * df['HighLowRange'] + df['ClosePriceChange']
    
    # Calculate Price Range and Volatility
    df['PriceVolatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']
    
    # Incorporate Volatility Adjustment
    df['DailyRange'] = df['high'] - df['low']
    df['AverageRange'] = df['DailyRange'].rolling(window=10).mean()
    df['VolatilityMultiplier'] = 1 / (df['AverageRange'] + 1)
    
    # Adjusted Integrated Factor
    df['VolumeAverage'] = df['volume'].rolling(window=5).mean()
    df['DailyReturns'] = (df['close'] - df['open']) / df['open']
    df['Volatility'] = df['DailyReturns'].rolling(window=20).std()
    df['FinalFactor'] = df['IntegratedFactor'] * df['VolumeAverage'] * df['Volatility']
    
    return df['FinalFactor']
```
