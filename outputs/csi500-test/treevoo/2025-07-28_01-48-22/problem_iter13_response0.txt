```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()

    # Compute 20-Day Momentum
    df['momentum_20d'] = df['log_return'].rolling(window=20).sum()

    # Combine Momentum Components
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']

    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['combined_momentum'].rolling(window=5).mean()

    # Adjust Momentum
    df['volume_volatility_20d'] = df['volume'].pct_change().rolling(window=20).std()
    df['average_volume_10d'] = df['volume'].rolling(window=10).sum() / 10
    df['adjusted_momentum'] = df['momentum_20d'] / df['volume_volatility_20d'] * df['average_volume_10d']

    # Calculate Price Range and Volatility
    df['price_range_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open']) / df['volume']

    # Incorporate Volatility Adjustment
    df['average_high_low_range_10d'] = df['price_range_volatility'].rolling(window=10).mean()
    df['volatility_multiplier'] = 1 / (df['average_high_low_range_10d'] + 1)

    # Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_momentum'] * df['average_volume_10d'] * df['volatility_multiplier']

    return df['final_alpha_factor']
```
