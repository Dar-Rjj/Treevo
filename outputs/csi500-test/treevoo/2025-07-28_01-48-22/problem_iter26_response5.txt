import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    df['log_return'] = df['close'].apply(lambda x: pd.np.log(x)).diff()
    
    # Compute 20-Day Momentum
    df['momentum_20_day'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday Momentum Component
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Historical Momentum Component
    df['historical_momentum'] = df['close'] - df['close'].shift(10)
    
    # Combine Momentum Components
    df['combined_momentum'] = df['intraday_momentum'] + df['historical_momentum']
    
    # Smooth Combined Momentum
    df['smoothed_combined_momentum'] = df['combined_momentum'].rolling(window=5).mean()
    
    # 20-Day Volume Volatility
    df['volume_volatility_20_day'] = df['volume'].diff().rolling(window=20).std()
    
    # Average Volume
    df['average_volume_10_day'] = df['volume'].rolling(window=10).mean()
    
    # Adjusted Momentum
    df['adjusted_momentum'] = df['momentum_20_day'] / df['volume_volatility_20_day'] * df['average_volume_10_day']
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    df['average_high_low_range_5_day'] = df['high_low_range'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['high_low_range_over_average'] = df['high_low_range'] / df['average_high_low_range_5_day']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_intraday_returns'] = df['intraday_return'] + df['open_to_close_return']
    
    # Identify Volume Spikes
    df['volume_spike_indicator'] = (df['volume'] > (df['volume'].rolling(window=20).mean() + 2 * df['volume'].rolling(window=20).std())).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['combined_intraday_returns_with_spike'] = df['combined_intraday_returns'] * df['volume_spike_indicator']
    
    # Calculate Volume Difference
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate VWAP
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Volume-to-VWAP Ratio
    df['volume_to_vwap_ratio'] = df['volume'] / df['vwap']
    
    # Integrate Volume and Price Information
    df['integrated_volume_price_info'] = 0.5 * df['volume_difference'] + 0.3 * df['average_high_low_range_5_day'] + (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    df['final_factor_value'] = (df['high_low_range_over_average'] * df['volume_difference'] * 
                                df['volume_to_vwap_ratio'] * (df['close'] - df['close'].shift(1)))
    
    # Weight by Volume and Volatility
    df['volume_average_5_day'] = df['volume'].rolling(window=5).mean()
    df['volatility_20_day'] = df['close'].pct_change().rolling(window=20).std()
    df['weighted_relative_strength'] = df['relative_strength'] * (df['volume_average_5_day'] * df['volatility_20_day'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df
