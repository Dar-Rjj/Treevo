```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average High-Low Range for the past 5 days
    high_low_range = df['high'] - df['low']
    avg_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Calculate Adjusted Close Change
    close_change = df['close'] - df['close'].shift(1)
    
    # Integrate Volume and Price Information
    volume_weight = volume_diff * 0.5
    range_weight = avg_high_low_range * 0.3
    combined_factor = volume_weight + range_weight + close_change
    
    return combined_factor
```
