```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return using Close price
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute 20-Day Momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Volume Volatility
    df['volume_change'] = df['volume'].pct_change()
    df['volume_volatility'] = df['volume_change'].rolling(window=20).std()
    
    # Adjust Momentum by Inverse Volume Volatility
    df['adjusted_momentum'] = df['momentum_20'] / (df['volume_volatility'] + 1e-8)  # Avoid division by zero
    
    # Calculate Intraday Range Volatility
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Scaled Intraday Movement
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['volume_scaled_intraday'] = df['intraday_return'] * df['volume']
    
    # Calculate Weighted Moving Average of Intraday Returns
    df['wma_intraday_returns'] = df['intraday_return'].rolling(window=5).mean()
    
    # Combine Indicators
    df['alpha_factor'] = (df['volume_scaled_intraday'] - df['intraday_range']) + df['wma_intraday_returns'] + df['adjusted_momentum']
    
    return df['alpha_factor']
```
