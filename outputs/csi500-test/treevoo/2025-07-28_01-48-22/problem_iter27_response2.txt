import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute 20-Day Momentum
    momentum_20_day = daily_log_return.rolling(window=20).sum()
    
    # Combine Momentum Components
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    historical_momentum = df['close'] - df['close'].shift(10)
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smooth Combined Momentum
    smooth_combined_momentum = combined_momentum.ewm(span=5, adjust=False).mean()
    
    # Adjust Momentum
    volume_volatility_20_day = df['volume'].pct_change().rolling(window=20).std()
    average_volume = df['volume'].rolling(window=10).sum() / 10
    adjusted_momentum = (momentum_20_day / volume_volatility_20_day) * average_volume
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average High-Low Range
    avg_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_range_over_avg = high_low_range / avg_high_low_range
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_return + open_close_return
    
    # Identify Volume Spikes
    volume_spike_indicator = (df['volume'] > df['volume'].mean() + 2 * df['volume'].std()).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_returns_with_spike = combined_returns * volume_spike_indicator
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume-to-VWAP Ratio
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    integrated_info = (volume_difference * 0.6) + (avg_high_low_range * 0.4) + (df['close'] - df['close'].shift(1))
    
    # Combine Indicators
    final_factor = (high_low_range_over_avg * volume_difference * volume_to_vwap_ratio * (df['close'] - df['close'].shift(1)))
    
    # Weight by Volume and Volatility
    volume_average = df['volume'].rolling(window=5).mean()
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    weighted_final_factor = relative_strength * (volume_average * volatility)
    
    return final_factor * weighted_final_factor
