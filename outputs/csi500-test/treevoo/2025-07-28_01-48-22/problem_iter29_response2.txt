```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'] / df['close'].shift(5) - 1
    
    # Calculate Long-Term Return
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Identify Volume Spikes
    volume_std = df['volume'].rolling(window=20).std()
    volume_mean = df['volume'].rolling(window=20).mean()
    k = 2  # Tunable parameter
    volume_spike = (df['volume'] > (volume_mean + k * volume_std)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    combined_return = relative_strength * volume_spike
    
    # Integrate Factors
    volume_diff = df['volume'] - df['volume'].shift(1)
    integrated_factor = 0.5 * volume_diff + df['close'] - df['close'].shift(1)
    
    # Adjusted Integrated Factor
    volume_avg = df['volume'].rolling(window=5).sum() / 5
    daily_returns = (df['close'] - df['open']) / df['open']
    volatility = daily_returns.rolling(window=20).std()
    adjusted_factor = integrated_factor * (volume_avg * volatility)
    
    # Incorporate Advanced Volatility Adjustment
    daily_range = df['high'] - df['low']
    ema_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    advanced_adjusted_factor = adjusted_factor * volatility_multiplier
    
    # Volume Momentum
    volume_momentum = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # Close Price Momentum
    close_momentum = df['close'] - df['close'].rolling(window=21).mean() - df['close'].shift(1)
    
    # Integrate Hybrid Factors
    hybrid_factor = 0.5 * volume_momentum + close_momentum
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_adjustment = np.where(sma_10 > sma_30, 1.1, 0.9)
    hybrid_factor *= trend_adjustment
    
    # Include Volume and Price Correlation
    corr_vol_close = df['volume'].rolling(window=10).corr(df['close'])
    correlation_adjustment = np.where(corr_vol_close > 0, 1.05, 0.95)
    hybrid_factor *= correlation_adjustment
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(1)
    
    # Assess Directional Movement Over a Series of Days
    positive_changes = (df['close'] - df['close'].shift(1)).clip(lower=0).rolling(window=10).sum()
    negative_changes = (df['close'] - df['close'].shift(1)).clip(upper=0).rolling(window=10).sum()
    
    # Analyze Volume in Conjunction with Price Movements
    avg_volume = df['volume'].rolling(window=20).mean()
    vol_price_corr = df['volume'].rolling(window=10).corr(df['close'] / df['close'].shift(1) - 1)
    
    # Incorporate Volatility Measures
    log_returns = np.log(df['close'] / df['close'].shift(1))
    log_volatility = log_returns.rolling(window=20).std()
    high_low_ratio = (df['high'] - df['low']) / df['close']
    
    # Final Factor
    final_factor = adjusted_factor + hybrid_factor + short_term_momentum + \
                   (positive_changes - negative_changes) + (vol_price_corr * avg_volume) + \
                   (log_volatility + high_low_ratio)
    
    return final_factor
```
