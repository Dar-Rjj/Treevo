import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Return
    close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Average High-Low Range for past 5 days
    average_high_low_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    high_low_over_average = high_low_range / average_high_low_range
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Calculate VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Volume-to-VWAP Ratio
    volume_to_vwap_ratio = df['volume'] / vwap
    
    # Integrate Volume and Price Information
    integrated_volume_price = (volume_difference * 0.5) + (average_high_low_range * 0.3) + close_return
    
    # Combine Indicators
    combined_indicator = (high_low_over_average * volume_difference * volume_to_vwap_ratio * close_return)
    
    # Final Alpha Factor
    final_factor = combined_indicator + integrated_volume_price
    
    return final_factor.dropna()
