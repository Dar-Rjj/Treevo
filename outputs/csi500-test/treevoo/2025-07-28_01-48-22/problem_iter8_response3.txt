```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Spread
    df['prev_intraday_spread'] = df['intraday_spread'].shift(1)
    
    # Compare Intraday and Previous Day Spreads
    df['intraday_spread_ratio'] = df['intraday_spread'] / df['prev_intraday_spread']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=20).mean()
    std_volume = df['volume'].rolling(window=20).std()
    k = 2  # Example value for K
    df['volume_spike'] = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume_spike']
    
    # Calculate Close-to-Previous Close Return
    df['close_to_prev_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Average High-Low Range for past 5 days
    df['avg_high_low_range'] = df['intraday_spread'].rolling(window=5).mean()
    
    # Calculate High-Low Range Over Average
    df['high_low_range_over_avg'] = df['intraday_spread'] / df['avg_high_low_range']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume Change
    df['volume_change_ratio'] = df['volume_diff'] / df['volume'].shift(1)
    
    # Combine Signals for Alpha Factor
    weights = {
        'intraday_spread_ratio': 0.2,
        'open_to_close_return': 0.1,
        'weighted_open_to_close_return': 0.3,
        'high_low_range_over_avg': 0.1,
        'volume_change_ratio': 0.1,
        'close_to_prev_close_return': 0.2
    }
    
    df['alpha_factor'] = (
        df['intraday_spread_ratio'] * weights['intraday_spread_ratio'] +
        df['open_to_close_return'] * weights['open_to_close_return'] +
        df['weighted_open_to_close_return'] * weights['weighted_open_to_close_return'] +
        df['high_low_range_over_avg'] * weights['high_low_range_over_avg'] +
        df['volume_change_ratio'] * weights['volume_change_ratio'] +
        df['close_to_prev_close_return'] * weights['close_to_prev_close_return']
    )
    
    return df['alpha_factor']
```
