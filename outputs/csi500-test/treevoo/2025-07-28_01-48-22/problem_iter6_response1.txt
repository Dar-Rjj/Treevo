```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    df['relative_strength'] = df['short_term_return'] - df['long_term_return']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Identify Volume Spikes
    N = 20  # Number of days to calculate median volume
    K = 2  # Multiplier for median volume
    df['median_volume'] = df['volume'].rolling(window=N).median()
    df['volume_spike'] = (df['volume'] > df['median_volume'] * K).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    df['spike_adjusted_return'] = df['combined_return'] * df['volume_spike']
    
    # Weight by Volume and Volatility
    df['volume_average'] = df['volume'].rolling(window=5).mean()
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['volatility'] = df['daily_returns'].rolling(window=20).std()
    df['weighted_factor'] = df['relative_strength'] * (df['volume_average'] * df['volatility'])
    
    return df['weighted_factor']
```
