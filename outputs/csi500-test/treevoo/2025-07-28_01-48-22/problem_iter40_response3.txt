```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Generate a signal by comparing today's close price to the 5-day WMA
    weights = np.arange(1, 6)
    df['5d_WMA'] = (df['close'].rolling(window=5).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=False))
    df['wma_signal'] = np.where(df['close'] > df['5d_WMA'], 1, -1)

    # Calculate the combined percentage change in the closing price from 10 and 20 days ago to today
    df['10d_pc'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) * 100
    df['20d_pc'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * 100
    df['combined_momentum'] = df['10d_pc'] + df['20d_pc']

    # Compute the cumulative sum of the difference between daily close and open prices over the past 20 trading days
    df['price_diff'] = df['close'] - df['open']
    df['cumulative_sum'] = df['price_diff'].rolling(window=20).sum()
    # Introduce a smoothing factor to the cumulative sum
    df['smoothed_cumsum'] = df['cumulative_sum'].ewm(span=5).mean()

    # Flag any day with volume significantly (e.g., 3 standard deviations) above the 20-day average
    df['vol_20d_avg'] = df['volume'].rolling(window=20).mean()
    df['vol_20d_std'] = df['volume'].rolling(window=20).std()
    df['vol_anomaly'] = np.where(df['volume'] > df['vol_20d_avg'] + 3 * df['vol_20d_std'], 1, 0)

    # Combine this volume anomaly with the sign of the day's return and the direction of the 5-day WMA to generate a nuanced signal about the expected price movement
    df['return_sign'] = np.sign(df['close'] - df['close'].shift(1))
    df['nuanced_signal'] = df['vol_anomaly'] * df['return_sign'] * np.sign(df['5d_WMA'] - df['5d_WMA'].shift(1))

    # Calculate the range (high minus low) for each day
    df['range'] = df['high'] - df['low']
    # Compute the 10-day moving average of these ranges
    df['10d_range_avg'] = df['range'].rolling(window=10).mean()
    # Interpret the relative range
    df['relative_volatility'] = df['range'] / df['10d_range_avg']

    # Combine all the factors into a single alpha factor
    alpha_factor = (df['wma_signal'] 
                    + df['combined_momentum'] 
                    + df['smoothed_cumsum'] 
                    + df['nuanced_signal'] 
                    + df['relative_volatility'])

    return alpha_factor.dropna()
```
