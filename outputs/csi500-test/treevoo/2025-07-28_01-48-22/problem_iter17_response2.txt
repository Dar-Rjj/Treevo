```python
import pandas as pd

def heuristics_v2(df):
    # Moving Averages Crossovers
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['Short_vs_Long_MA'] = df['SMA_10'] - df['SMA_50']
    df['Mid_vs_Long_MA'] = df['SMA_30'] - df['SMA_200']
    
    # Momentum Indicators
    n_roc = 14  # You can adjust n_roc to any number of days for ROC calculation
    df['ROC'] = (df['close'] - df['close'].shift(n_roc)) / df['close'].shift(n_roc) * 100
    
    n_rsi = 14  # RSI lookback period, typically 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=n_rsi).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n_rsi).mean()
    df['RSI'] = gain / loss
    
    # On-Balance Volume (OBV)
    df['price_change'] = df['close'].diff()
    df['obv'] = ((df['price_change'] > 0).astype(int) - (df['price_change'] < 0).astype(int)) * df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)]
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)]
    MFR = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI'] = MFR / (1 + MFR)
    
    # Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low'] + df['close']) / 3
    tpv = df['TypicalPrice'] * df['volume']
    vwap = tpv.cumsum() / df['volume'].cumsum()
    df['VWAP'] = vwap
    
    # Chaikin Oscillator
    adl = ((2 * df['close'] - df['low'] - df['high']) / (df['high'] - df['low'])) * df['volume']
    adl = adl.fillna(0).cumsum()
    chaikin_oscillator = adl.ewm(span=3).mean() - adl.ewm(span=10).mean()
    df['ChaikinOscillator'] = chaikin_oscillator
    
    # Breakout and Breakdown
    df['Breakout_Above_High'] = (df['high'] > df['high'].rolling(window=20).max()).astype(int)
    df['Breakdown_Below_Low'] = (df['low'] < df['low'].rolling(window=20).min()).astype(int)
    
    # Custom Composite Factor: Integration of Momentum with Volume Support
    df['Obv_Change'] = df['obv'] - df['obv'].shift(14)
    df['Momentum_Volume_Support'] = df['ROC'] * (df['Obv_Change'] / df['volume'])
    
    # Complex Conditionals Involving Multiple Signals
    conditions = (df['ROC'] > 0) & (df['MFI'] < 0.8) & (df['Breakout_Above_High'] == 1)
    df['Buy_Signal'] = conditions.astype(int)
    
    # Return the final factor
    return df['Momentum_Volume_Support']
```
