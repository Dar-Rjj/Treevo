```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_momentum = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'].diff(10)
    
    # Combine Momentum Components
    combined_momentum = intraday_momentum + historical_momentum
    
    # Smoothed Momentum
    smoothed_momentum = combined_momentum.rolling(window=5).mean()
    
    # Adjust for Volatility
    daily_high_low_range = df['high'] - df['low']
    average_daily_range = daily_high_low_range.rolling(window=10).mean()
    volatility_multiplier = 1 / (average_daily_range + 1)
    
    # Adjusted Smoothed Momentum
    adjusted_smoothed_momentum = smoothed_momentum * volatility_multiplier
    
    # Integrate Volume and Price Information
    volume_change = (df['volume'] - df['volume'].shift(1))
    avg_high_low_5_days = (df['high'] - df['low']).rolling(window=5).mean()
    close_price_change = df['close'] - df['close'].shift(1)
    integrated_info = 0.5 * volume_change + 0.3 * avg_high_low_5_days + close_price_change
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Relative Strength
    relative_strength = short_term_return - long_term_return
    
    # Combine Factors
    inplay_factor = 0.3 * (intraday_momentum + close_price_change)
    combined_factors = relative_strength * volume_change + inplay_factor
    
    # Adjusted Integrated Factor
    volume_average = df['volume'].rolling(window=5).mean()
    volatility = ((df['close'] - df['open']) / df['open']).rolling(window=20).std()
    adjusted_integrated_factor = (adjusted_smoothed_momentum + integrated_info + combined_factors) * (volume_average * volatility)
    
    # Final Alpha Factor
    alpha_factor = adjusted_smoothed_momentum + integrated_info + adjusted_integrated_factor
    
    return alpha_factor
```
