```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_scaled_intraday_movement = intraday_return * df['volume']
    
    # Calculate Historical Momentum Component
    historical_momentum = df['close'] - df['close'].shift(20)
    
    # Combine Momentum Components
    combined_momentum = volume_scaled_intraday_movement + historical_momentum
    
    # Incorporate Volume Influence
    weighted_average_volume = (df['volume'].rolling(window=20).apply(lambda x: np.sum(x / np.arange(1, len(x) + 1)), raw=True) /
                               np.sum(1 / np.arange(1, 21)))
    momentum_with_volume = combined_momentum * weighted_average_volume
    
    # Incorporate Advanced Volatility Adjustment
    daily_high_low_range = df['high'] - df['low']
    ema_daily_range = daily_high_low_range.ewm(span=15, min_periods=15, adjust=False).mean()
    volatility_multiplier = 1 / (ema_daily_range + 1)
    adjusted_combined_momentum = momentum_with_volume * volatility_multiplier
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Calculate Average True Range
    true_range = np.maximum.reduce([df['high'] - df['low'], 
                                    abs(df['high'] - df['close'].shift(1)),
                                    abs(df['low'] - df['close'].shift(1))])
    atr_7_days = true_range.rolling(window=7).mean()
    
    # Integrate Hybrid Factors
    integrated_factor = (0.8 * volume_trend + 0.3 * atr_7_days + adjusted_combined_momentum)
    
    return integrated_factor
```
