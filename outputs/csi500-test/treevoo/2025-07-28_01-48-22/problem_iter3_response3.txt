```python
import pandas as pd

def heuristics(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Exponential Moving Average (EMA) of Volume over 5 days
    df['volume_ema'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = df['price_momentum'] * df['volume_ema']
    
    # Calculate Enhanced Price and Volume Interaction Signal
    df['enhanced_pv_interaction'] = (df['high'] - df['low']) * (df['close'] - df['open']) / (df['volume'] + 1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['enhanced_pv_interaction']
    
    return df['alpha_factor']
```
