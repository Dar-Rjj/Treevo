```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicator
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Crossover_Signal'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Signal by comparing today's close price to the 5-day WMA
    df['WMA_5'] = df['close'].rolling(window=5).apply(lambda x: (x * pd.Series(range(1,6))).sum() / 15, raw=True)
    alpha_factor_1 = (df['close'] > df['WMA_5']).astype(int).replace({True: 1, False: -1})
    
    # Combined Percentage Change in Close Price
    alpha_factor_2 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) + (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Price Change Relative to Range
    alpha_factor_7 = (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'])
    
    # Volume Ratio
    alpha_factor_3 = df['volume'] / (df['high'] - df['low'])
    
    # Cumulative Sum of Daily Close-Open Difference
    alpha_factor_4 = (df['close'] - df['open']).rolling(window=20).sum()
    
    # Volume Change Relative to Amount
    alpha_factor_8 = (df['volume'] - df['volume'].shift(5)) / df['amount']
    
    # Anomaly Detection in Volume
    vol_std = df['volume'].rolling(window=20).std()
    alpha_factor_9 = ((df['volume'] > 3 * vol_std) & (df['return'] > 0)).astype(int) - ((df['volume'] > 3 * vol_std) & (df['return'] < 0)).astype(int)
    
    # Close Price Momentum
    alpha_factor_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Trend Direction Indicator
    alpha_factor_6 = ((df['open'] < df['close']) * 1) + ((df['open'] > df['close']) * -1)
    
    # Trend Strength Indicator
    alpha_factor_10 = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))
    
    # Volatility Ratio
    alpha_factor_11 = (df['high'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    
    # Range Volatility
    alpha_factor_12 = (df['high'] - df['low']) / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    
    # Relative Range Measure
    range_sma = (df['high'] - df['low']).rolling(window=10).mean()
    alpha_factor_13 = (df['high'] - df['low']) / range_sma
    
    # Combine all factors into a single series
    result = (
        alpha_factor_1 + alpha_factor_2 + alpha_factor_7 + 
        alpha_factor_3 + alpha_factor_4 + alpha_factor_8 + 
        alpha_factor_9 + alpha_factor_5 + alpha_factor_6 + 
        alpha_factor_10 + alpha_factor_11 + alpha_factor_12 + 
        alpha_factor_13
    )
    
    return result
```
