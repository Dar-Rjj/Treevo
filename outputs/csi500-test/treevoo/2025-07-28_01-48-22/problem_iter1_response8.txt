```python
import pandas as pd

def heuristics_v2(df, n_days=20, k=2):
    # Calculate Intraday Return using High and Low Prices
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = intraday_return + open_to_close_return
    
    # Identify Volume Spikes
    mean_volume = df['volume'].rolling(window=n_days).mean()
    std_volume = df['volume'].rolling(window=n_days).std()
    volume_spike = (df['volume'] > (mean_volume + k * std_volume)).astype(int)
    
    # Multiply Combined Return by Volume Spike Indicator
    factor = combined_return * volume_spike
    
    return factor
```
