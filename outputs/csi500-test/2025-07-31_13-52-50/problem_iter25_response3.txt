import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()

    # Compute Volume Momentum Factor
    m = 20
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()

    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']

    # Adjust Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    small_constant = 1e-6
    df['adjusted_momentum'] = df['combined_momentum'] / (np.abs(df['volume_deviation']) + small_constant)

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vw_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']

    # Aggregate Indicators
    df['agg_breakout'] = df['vw_breakout_indicator']

    # Smooth Aggregate Breakout Indicators
    smoothing_factor = 0.1
    df['smoothed_breakout'] = df['agg_breakout'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Determine Daily Return Deviation from VWAP
    df['vwap'] = (df['volume'] * (df['high'] + df['low']) / 2).cumsum() / df['volume'].cumsum()
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['prev_return_deviation'] = df['return_deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['return_deviation_from_vwap'] > df['prev_return_deviation']) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Calculate Volume-Adjusted Intraday Return
    df['volume_adjusted_intraday_return'] = (df['close'] - df['open']) * df['volume']

    # Final Composite Factor Calculation
    df['composite_factor'] = df['smoothed_breakout'] + df['adjusted_momentum']
    df['composite_factor'] += df['volume_adjusted_intraday_return']
    df.loc[df['trend_reversal'], 'composite_factor'] += 1.0  # Assign higher score for stronger indication

    # Generate Alpha Factor
    window_size = 20
    df['alpha_factor'] = df['composite_factor'].rolling(window=window_size).mean()

    # Enhance Alpha Factor
    df['alpha_factor'] *= df['daily_return']

    # Adjust for Volatility
    df['volatility'] = (df['high'] - df['low']).rolling(window=n).std()
    df['alpha_factor'] /= (df['volatility'] + small_constant)

    return df['alpha_factor'].dropna()
