```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Adjust for Volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_return'] = (df['volume'] - df['volume_ma_20']) * df['intraday_return']

    # Incorporate Price Volatility
    df['close_std_15'] = df['close'].rolling(window=15).std()
    
    def volatility_adjuster(std):
        if std > 0.05:  # Threshold for high volatility, this can be adjusted
            return 1.8
        else:
            return 0.6
    
    df['volatility_adjustment'] = df['close_std_15'].apply(volatility_adjuster)
    df['final_factor'] = df['volume_adjusted_return'] * df['volatility_adjustment']

    return df['final_factor']
```
