```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Adjusted Momentum Indicator.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A series of the calculated factor values indexed by date.
    """
    # Parameters
    lookback_days = 20
    
    # Calculate Momentum Score
    close_price_change = df['close'] - df['close'].shift(lookback_days)
    momentum_score = close_price_change.rolling(window=lookback_days).sum()
    
    # Calculate Volume Moving Average
    volume_moving_avg = df['volume'].rolling(window=lookback_days).mean()
    
    # Calculate Volume Deviation
    volume_deviation = df['volume'] - volume_moving_avg
    
    # Normalize Momentum Score by Volume Deviation
    adjusted_momentum_score = momentum_score / (volume_deviation + 1e-6)  # Add small constant to avoid division by zero
    
    return adjusted_momentum_score
```
