```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['weighted_price_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['aggregated_breakout_ratios'] = df['weighted_price_breakout_ratio'].rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_return'] * df['volume']
    df['average_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weighted_momentum_score'] = df['volume_weighted_momentum'] / df['average_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum_breakout'] = df['volume_weighted_momentum_score'] + df['aggregated_breakout_ratios']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    df['trend_reversal_score'] = df['trend_reversal_potential'].astype(int) * 10  # Assign higher score for stronger indication
    
    # Compute Price Momentum Factor
    n_days = 10
    df['price_momentum_factor'] = df['daily_return'].rolling(window=n_days).sum()
    
    # Compute Volume Momentum Factor
    m_days = 10
    df['volume_momentum_factor'] = (df['volume'] - df['volume'].shift(1)).rolling(window=m_days).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum_factor'] + k2 * df['volume_momentum_factor']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Final Composite Factor Calculation
    df['final_factor'] = (df['trend_reversal_score'] + df['adjusted_combined_momentum'] + df['combined_momentum_breakout']).ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_factor']
```
