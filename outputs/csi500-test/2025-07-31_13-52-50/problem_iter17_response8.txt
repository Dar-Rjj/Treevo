```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['daily_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['daily_return_deviation'] = df['close'].diff()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compare Daily Return Deviation with VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    df['reversal_indicator'] = df['return_deviation_from_vwap'].diff().gt(0).astype(int)
    
    # Check for Volume Increase
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = df['volume'] > df['volume_ma_14']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust Intraday Return for Volume
    df['volume_ma_7'] = df['volume'].rolling(window=7).mean()
    df['volume_diff'] = df['volume'] - df['volume_ma_7']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_diff']
    
    # Incorporate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=7).std()
    df['volatility_adjusted_intraday_return'] = df['adjusted_intraday_return'] * np.where(df['price_volatility'] > df['price_volatility'].median(), 1.3, 0.7)
    
    # Compute Price Momentum Factor
    n = 7
    df['price_momentum'] = df['daily_return_deviation'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 7
    df['volume_momentum'] = df['volume_diff'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma_20']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_cross_up'] = (df['sma_5'] > df['sma_20']) & (df['sma_5'].shift(1) < df['sma_20'].shift(1))
    df['sma_cross_down'] = (df['sma_5'] < df['sma_20']) & (df['sma_5'].shift(1) > df['sma_20'].shift(1))
    df['momentum_shift'] = np.where(df['sma_cross_up'], 0.6, np.where(df['sma_cross_down'], -0.6, 0))
    df['final_momentum_score'] = df['adjusted_combined_momentum'] + df['momentum_shift']
    
    # Combine Indicators
    df['trend_reversal_score'] = (df['reversal_indicator'] & df['volume_increase']).astype(int) * 2
    df['composite_factor'] = df['trend_reversal_score'] + df['volatility_adjusted_intraday_return'] + df['final_momentum_score']
    
    return df['composite_factor']
```
