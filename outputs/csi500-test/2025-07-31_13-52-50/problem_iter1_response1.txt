import pandas as pd

def heuristics_v2(df, n=10, m=10, k1=1.0, k2=1.0):
    """
    Calculate a composite factor of price and volume momentum.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - n: number of days to look back for price momentum
    - m: number of days to look back for volume momentum
    - k1: weight for price momentum
    - k2: weight for volume momentum
    
    Returns:
    - pandas Series with the composite factor values indexed by date
    """
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate volume change
    df['volume_change'] = df['volume'].diff()
    
    # Compute price momentum factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute volume momentum factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine price and volume momentum
    df['composite_factor'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    return df['composite_factor'].dropna()
