```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Series of alpha factor values indexed by date.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    volume_weighted_typical_price = typical_price * df['volume']
    
    # Calculate the moving average of the volume-weighted typical price over a 10-day window
    ma_10 = volume_weighted_typical_price.rolling(window=10).mean()
    
    # Calculate the standard deviation of the volume-weighted typical price over a 10-day window
    std_10 = volume_weighted_typical_price.rolling(window=10).std()
    
    # Calculate the Z-score of the volume-weighted typical price
    z_score = (volume_weighted_typical_price - ma_10) / std_10
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the alpha factor
    alpha_factor = z_score * momentum
    
    return alpha_factor
```
