```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    close_prices = df['close']
    avg_close_14 = close_prices.rolling(window=14).mean()
    adjusted_price_momentum = close_prices - avg_close_14

    # Incorporate Volume Impact
    current_volume = df['volume']
    avg_volume_14 = current_volume.rolling(window=14).mean()
    volume_change_ratio = current_volume / avg_volume_14
    price_momentum_with_volume = adjusted_price_momentum * volume_change_ratio

    # Assess Daily Price Movement Intensity
    price_movement_intensity = df['high'] - df['low']

    # Calculate Volume-Adjusted Price Indicator
    volume_adjusted_price = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Determine Deviation from Volume-Weighted Average Price
    deviation_from_vwap = df['close'] - volume_adjusted_price

    # Identify Reversal or Continuation Potential
    prev_deviation = deviation_from_vwap.shift(1)
    reversal_potential = (deviation_from_vwap > prev_deviation) & (current_volume > avg_volume_14)

    # Final Score Calculation
    score = price_momentum_with_volume
    score[reversal_potential] = score[reversal_potential] * 1.5  # Assign higher score for stronger indication

    return score
```
