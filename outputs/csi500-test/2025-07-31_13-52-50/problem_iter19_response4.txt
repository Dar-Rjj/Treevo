import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=n, adjust=False).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = df['volume'] * (df['close'] - df['open'])
    df['volume_weighted_breakout_indicator'] = (df['price_breakout_ratio'] * df['volume'])

    # Aggregate Volume-Weighted Breakout Indicators
    df['aggregated_breakout_ratios'] = df['volume_weighted_breakout_indicator'].rolling(window=n).sum()

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_returns'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum'] / df['avg_volume']

    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_factor'] = df['volume_weighted_momentum_adj'] + df['aggregated_breakout_ratios']

    # Generate Final Alpha Factor
    final_alpha_factor = df['combined_factor'].ewm(span=n, adjust=False).mean()

    return final_alpha_factor
