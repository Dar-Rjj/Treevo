import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Volume-Weighted Breakout Ratios
    volume_weighted_breakout_ratios = price_breakout_ratio * df['volume']
    aggregate_volume_weighted_breakout = volume_weighted_breakout_ratios.rolling(window=10).sum()
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.9
    smoothed_factor = aggregate_volume_weighted_breakout.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return smoothed_factor
