import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    smoothed_returns = daily_returns.ewm(span=n, adjust=False).mean()
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = smoothed_returns * df['volume']
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weighted_momentum = volume_weighted_momentum / avg_volume
    
    # Weight Price Breakout Ratio by Volume
    volume_weighted_price_breakout_ratio = price_breakout_ratio * df['volume']
    
    # Aggregate Volume-Weighted Breakout Indicators
    aggregated_breakout_indicators = volume_weighted_breakout.sum()
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Indicators
    combined_factor = volume_weighted_momentum + aggregated_breakout_indicators
    
    return combined_factor
