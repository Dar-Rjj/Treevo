```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20  # Lookback period
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()

    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']

    # Combine Moving Average and Volume Weight
    df['volume_weighted_ma_return'] = df['ma_daily_return'] * df['volume_weight']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['agg_breakout'] = df['volume_weighted_breakout_indicator'].rolling(window=n).sum()
    df['weighted_price_breakout_ratio'] = (df['price_breakout_ratio'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Generate Alpha Factor
    df['alpha_factor'] = df['volume_weighted_ma_return'] + df['agg_breakout'] + df['weighted_price_breakout_ratio']
    alpha_window = 10  # Define window size for alpha factor moving average
    df['alpha_factor'] = df['alpha_factor'].rolling(window=alpha_window).mean()

    # Smooth with Exponential Moving Average
    ema_alpha = 0.3  # Define smoothing factor
    df['smoothed_alpha_factor'] = df['alpha_factor'].ewm(alpha=ema_alpha, adjust=False).mean()

    # Incorporate Daily Return
    df['enhanced_alpha_factor'] = df['smoothed_alpha_factor'] * df['daily_return']

    # Adjust for Volatility
    volatility_window = 20  # Define window size for volatility calculation
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=volatility_window).std().mean(axis=1)
    df['final_alpha_factor'] = df['enhanced_alpha_factor'] / df['volatility']

    return df['final_alpha_factor'].dropna()
```
