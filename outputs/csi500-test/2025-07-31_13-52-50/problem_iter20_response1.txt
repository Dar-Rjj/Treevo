```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = (df['volume'] - df['avg_volume']) * df['daily_return']
    
    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['combined_momentum'] = df['ema_daily_return'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['aggregated_breakout'] = df['price_breakout_ratio'] * df['volume'] + df['volume_breakout_indicator']
    
    # Smooth Aggregated Breakout Indicators
    window_size = 5
    df['smoothed_aggregated_breakout'] = df['aggregated_breakout'].rolling(window=window_size).mean()
    
    # Combine Momentum and Breakout Indicators
    df['combined_indicator'] = df['combined_momentum'] + df['smoothed_aggregated_breakout']
    
    # Calculate Volume-Weighted Combined Indicator
    df['volume_weighted_combined_indicator'] = df['combined_indicator'] * df['volume_weight']
    
    # Calculate Long-Term Trend
    long_term_alpha = 0.05
    df['long_term_trend'] = df['close'].ewm(alpha=long_term_alpha, adjust=False).mean()
    
    # Adjust for Trend
    df['trend_adjusted_indicator'] = df['volume_weighted_combined_indicator'] - df['long_term_trend']
    
    # Generate Final Alpha Factor
    final_alpha_smoothing_factor = 0.1
    df['final_alpha'] = df['trend_adjusted_indicator'].ewm(alpha=final_alpha_smoothing_factor, adjust=False).mean()
    
    # Integrate Reversal and Intraday Indicators
    df['daily_price_movement_range'] = df['high'] - df['low']
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['daily_return_deviation_from_vwap'] = df['close'] - df['vwap']
    df['is_reversal_potential'] = df['daily_return_deviation_from_vwap'] > df['daily_return_deviation_from_vwap'].shift(1)
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increased'] = df['volume'] > df['volume_ma_14']
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_difference'] = df['volume'] - df['volume_ma_20']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_difference']
    df['close_price_std_10'] = df['close'].rolling(window=10).std()
    df['adjusted_intraday_return'] = np.where(
        df['close_price_std_10'] > df['close_price_std_10'].median(),
        df['adjusted_intraday_return'] * 1.5,
        df['adjusted_intraday_return'] * 0.7
    )
    df['reversal_and_intraday_indicator'] = (
        df['is_reversal_potential'] & df['volume_increased']
    ) * df['adjusted_intraday_return']
    df['final_alpha'] = df['final_alpha'] * df['reversal_and_intraday_indicator']
    
    return df['final_alpha']
```
