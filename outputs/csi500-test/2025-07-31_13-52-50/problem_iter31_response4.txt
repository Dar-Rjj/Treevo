```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Diff'] = df['SMA_50'] - df['SMA_200']
    
    # Volume Indicators
    df['OBV'] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df['OBV'].iloc[i] = df['OBV'].iloc[i-1] + df['volume'].iloc[i]
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            df['OBV'].iloc[i] = df['OBV'].iloc[i-1] - df['volume'].iloc[i]
        else:
            df['OBV'].iloc[i] = df['OBV'].iloc[i-1]
    
    # Price Pattern Indicators
    df['IsDoji'] = (abs(df['open'] - df['close']) / (df['high'] - df['low'])) < 0.01
    
    # Volatility Indicators
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Market Sentiment Indicators
    df['TP'] = (df['high'] + df['low'] + df['close']) / 3
    df['MF'] = df['TP'] * df['volume']
    df['PositiveMF'] = df.apply(lambda x: x['MF'] if x['TP'] > x['TP'].shift(1) else 0, axis=1)
    df['NegativeMF'] = df.apply(lambda x: x['MF'] if x['TP'] < x['TP'].shift(1) else 0, axis=1)
    
    df['PositiveMF_Sum'] = df['PositiveMF'].rolling(window=14).sum()
    df['NegativeMF_Sum'] = df['NegativeMF'].rolling(window=14).sum()
    
    df['MFI'] = 100 - (100 / (1 + (df['PositiveMF_Sum'] / df['NegativeMF_Sum'])))
    
    # Combine all indicators into a single alpha factor
    df['AlphaFactor'] = (
        df['SMA_Diff'] + 
        df['OBV'].pct_change().fillna(0) + 
        df['IsDoji'].astype(int) + 
        df['ATR'] + 
        df['MFI']
    )
    
    return df['AlphaFactor']
```
