```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['deviation_from_vwap'] > df['deviation_from_vwap'].shift(1)).astype(int)
    
    # Check for Volume Increase
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['volume_ma_14']).astype(int)
    
    # Calculate 14-day Close Price Average
    df['close_ma_14'] = df['close'].rolling(window=14).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = df['close'] - df['close_ma_14']
    
    # Adjust for Volume Impact
    df['volume_avg_14'] = df['volume'].rolling(window=14).mean()
    df['volume_change_ratio'] = df['volume'] / df['volume_avg_14']
    df['adjusted_price_momentum'] = df['price_momentum'] * df['volume_change_ratio']
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    n = 20
    df['returns_ma_20'] = df['daily_returns'].rolling(window=n).mean()
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Combine Moving Average and Volume Weight
    df['combined_ma_volume'] = df['returns_ma_20'] * df['volume_weight']
    
    # Final Score Calculation
    df['final_score'] = (
        df['trend_reversal_potential'] + 
        df['volume_increase'] + 
        df['adjusted_price_momentum'] + 
        df['combined_ma_volume']
    )
    
    return df['final_score']
```
