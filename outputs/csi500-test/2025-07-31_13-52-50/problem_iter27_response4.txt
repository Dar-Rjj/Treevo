```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns (lookback 20 days, alpha=0.1)
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=20, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['vw_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']
    
    # Aggregate Indicators
    df['sum_vw_breakout_indicators'] = df['vw_breakout_indicator'].rolling(window=20).sum()
    df['weighted_breakout_ratio'] = (df['breakout_ratio'] * df['volume']).rolling(window=20).sum()
    
    # Calculate Volume-Weighted Momentum
    average_volume = df['volume'].rolling(window=20).mean()
    df['vw_momentum'] = df['ema_daily_returns'] * df['volume'] / average_volume
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum'] = df['vw_momentum'] + df['sum_vw_breakout_indicators']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap'] = vwap
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['reversal_potential'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_returns'].rolling(window=20).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=20).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum_score'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_deviation = df['volume'] - volume_ma
    small_constant = 1e-6
    volume_adjustment_factor = volume_deviation + small_constant
    df['adjusted_combined_momentum'] = df['combined_momentum_score'] / volume_adjustment_factor
    
    # Introduce Trend Following Component
    long_term_ma = df['close'].rolling(window=50).mean()
    short_term_ma = df['close'].rolling(window=20).mean()
    df['trend_signal'] = np.where(short_term_ma > long_term_ma, 1, -1)
    
    # Integrate Trend Signal into Alpha Factor
    df['trend_adjusted_momentum'] = df['adjusted_combined_momentum'] * df['trend_signal']
    
    # Final Composite Factor Calculation
    df['final_factor'] = df['combined_momentum'] * np.where(df['reversal_potential'], 1.5, 1)
    df['final_factor'] = df['final_factor'].ewm(span=20, adjust=False).mean()
    
    return df['final_factor']
```
