import pandas as pd

def heuristics_v2(df, n=10, m=10, k1=1.0, k2=1.0):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])
    
    # Incorporate Volume Impact
    df['average_volume_14'] = df['volume'].rolling(window=14).mean()
    df['volume_change_ratio'] = df['volume'] / df['average_volume_14']
    df['final_factor'] = df['combined_momentum'] * df['volume_change_ratio']
    
    return df['final_factor']
