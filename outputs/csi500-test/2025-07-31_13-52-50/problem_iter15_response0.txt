```python
import pandas as pd

def heuristics_v2(df, ma_window=20, ema_span=10, vol_window=20, trend_window=60):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Indicators
    aggregated_indicators = volume_weighted_breakout.rolling(window=ma_window).sum() / df['volume'].rolling(window=ma_window).sum()
    
    # Generate Alpha Factor
    alpha_factor = aggregated_indicators.ewm(span=ema_span, adjust=False).mean()
    
    # Enhance Alpha Factor
    # Incorporate Daily Return
    daily_return = df['close'].pct_change()
    alpha_factor *= daily_return
    
    # Adjust for Volatility
    volatility = (df['high'] - df['low']).rolling(window=vol_window).std()
    alpha_factor /= volatility
    
    # Introduce Trend Filter
    long_term_trend = df['close'].rolling(window=trend_window).mean()
    trend_alignment = (df['close'] > long_term_trend).astype(int)
    alpha_factor *= trend_alignment
    
    return alpha_factor
```
