```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Smoothing Intraday Momentum
    intraday_momentum = intraday_return.rolling(window=5).mean()
    weights = np.exp(np.linspace(0, -1, 5))
    intraday_momentum = intraday_return.rolling(window=5).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Evaluate Liquidity
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate Price Impact
    dollar_volume = df['volume'] * df['close']
    price_change = df['close'].diff().abs()
    price_impact = average_volume * price_change
    liquidity_evaluation = price_impact / 1000  # Scale by a constant
    
    # Combine Intraday Momentum and Liquidity
    alpha_factor = intraday_momentum + (intraday_momentum * liquidity_evaluation)
    
    # Normalize by Recent Volatility
    daily_returns = df['close'].pct_change()
    historical_volatility = daily_returns.rolling(window=20).std()
    alpha_factor = alpha_factor / historical_volatility
    
    # Finalize Alpha Factor
    alpha_factor = alpha_factor.clip(lower=0)
    alpha_factor = alpha_factor.where(alpha_factor > 0.01, 0)  # Apply threshold for stability
    
    return alpha_factor.dropna()
```
