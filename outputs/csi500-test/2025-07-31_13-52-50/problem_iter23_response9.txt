```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: Series with the calculated alpha factor values indexed by date.
    """
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative amount
    relative_amount = df['amount'] / df['volume']
    
    # Calculate the relative close change
    relative_close_change = (df['close'] - df['open']) / df['open']
    
    # Calculate the rolling mean of the relative range over 10 days
    rolling_mean_relative_range = relative_range.rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of the relative amount over 5 days
    rolling_std_relative_amount = relative_amount.rolling(window=5).std()
    
    # Calculate the final alpha factor
    alpha_factor = (relative_close_change 
                    + rolling_mean_relative_range 
                    - rolling_std_relative_amount)
    
    return alpha_factor
```
