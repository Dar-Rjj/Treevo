```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['close_14_day_avg'] = df['close'].rolling(window=14).mean()
    price_momentum = df['close_14_day_avg'] - df['close']
    
    # Incorporate Volume Impact
    df['volume_14_day_avg'] = df['volume'].rolling(window=14).mean()
    volume_change_ratio = df['volume'] / df['volume_14_day_avg']
    
    # Final factor
    factor = price_momentum * volume_change_ratio
    
    return factor
```
