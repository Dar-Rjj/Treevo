```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20  # Lookback period
    alpha = 0.1  # Smoothing factor
    df['smoothed_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['smoothed_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum_normalized'] = df['volume_weighted_momentum'] / df['avg_volume']
    
    # Aggregate Volume-Weighted Breakout Ratios
    df['volume_weighted_breakout'] = df['breakout_ratio'] * df['volume']
    df['aggregated_breakout_ratios'] = df['volume_weighted_breakout'].rolling(window=n).sum() / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['factor'] = df['volume_weighted_momentum_normalized'] + df['aggregated_breakout_ratios']
    
    return df['factor']
```
