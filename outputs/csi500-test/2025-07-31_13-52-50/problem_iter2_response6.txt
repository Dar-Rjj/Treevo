import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    aggregate_breakout_indicators = (price_breakout_ratio * df['volume']).rolling(window=10).sum()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_adjusted_intraday_return = (df['volume'] - volume_moving_avg) * intraday_return
    
    # Generate Final Alpha Factor
    final_alpha_factor = aggregate_breakout_indicators + volume_adjusted_intraday_return
    final_alpha_factor = final_alpha_factor.rolling(window=10).mean()
    
    return final_alpha_factor
