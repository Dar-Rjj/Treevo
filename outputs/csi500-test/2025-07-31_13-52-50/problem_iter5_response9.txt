```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Moving Average of Daily Returns
    lookback_period = 20
    df['moving_avg_return'] = df['daily_return'].rolling(window=lookback_period).mean()
    
    # Calculate Volume Weight
    df['average_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']
    
    # Combine Moving Average and Volume Weight
    df['vol_weighted_moving_avg_return'] = df['moving_avg_return'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['weighted_price_breakout'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    df['aggregate_breakout'] = df['weighted_price_breakout'].rolling(window=lookback_period).sum()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    vol_lookback = 10
    df['avg_volume_10d'] = df['volume'].rolling(window=vol_lookback).mean()
    df['vol_adjusted_intraday_return'] = (df['volume'] - df['avg_volume_10d']) * df['intraday_return']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['vol_weighted_moving_avg_return'] + 
                                df['aggregate_breakout'] + 
                                df['vol_adjusted_intraday_return'])
    
    # Smooth with Moving Average
    smoothing_window = 5
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].rolling(window=smoothing_window).mean()
    
    return df['final_alpha_factor_smoothed']
```
