```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation
    df['daily_return_deviation'] = df['close'].diff()
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1))
    
    # Check for Volume Increase
    df['volume_14_ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['volume_14_ma'])
    
    # Compute Price Momentum Factor
    n = 7  # Example: past 7 days
    df['price_momentum'] = df['daily_return_deviation'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 7  # Example: past 7 days
    df['volume_change'] = df['volume'].diff()
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5  # Weight for price momentum
    k2 = 0.5  # Weight for volume momentum
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_20_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_20_ma']
    small_constant = 1e-6
    df['volume_adjustment_factor'] = (df['volume_deviation'] + small_constant)
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout_indicator'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_from_vwap'] = df['close'] - df['vwap']
    
    # Incorporate Price Action Context
    df['price_change'] = df['close'] - df['open']
    df['price_change_sign'] = df['price_change'].apply(lambda x: 1 if x > 0 else -1)
    df['adjusted_aggregated_breakout'] = df['aggregated_breakout'] * df['price_change_sign']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_220_ma'] = df['volume'].rolling(window=220).mean()
    df['adjusted_intraday_return'] = (df['volume'] - df['volume_220_ma']) * df['intraday_return']
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20  # Example: 20 days
    alpha = 0.1  # Smoothing factor
    df['ema_daily_returns'] = df['daily_return_deviation'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Weight
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']
    
    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_ema'] = df['ema_daily_returns'] * df['volume_weight'] + df['adjusted_combined_momentum']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = (
        df['adjusted_ema'] * 
        (df['trend_reversal'] & df['volume_increase'] & (df['price_breakout_ratio'] > 0.5) & (df['adjusted_intraday_return'] > 0))
    )
    
    return df['alpha_factor']
```
