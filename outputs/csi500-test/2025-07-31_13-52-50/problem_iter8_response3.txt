import pandas as pd

def heuristics_v2(df, n=10, m=10, k1=0.5, k2=0.5, window_size=14):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['volume_weighted_breakout'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    df['aggregated_breakout'] = df['volume_weighted_breakout'].rolling(window=window_size).sum()
    
    # Smooth with Moving Average
    df['smoothed_aggregated_breakout'] = df['aggregated_breakout'].rolling(window=window_size).mean()
    
    # Combine Price and Volume Momentum
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Calculate Daily Price Movement Range
    df['daily_price_movement_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)).astype(int)
    df['volume_increase'] = (df['volume'] > df['volume'].rolling(window=14).mean()).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_diff'] = df['volume'] - df['volume'].rolling(window=10).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_diff']
    
    # Incorporate Price Volatility
    df['close_std'] = df['close'].rolling(window=5).std()
    df['adjusted_intraday_return'] = df['adjusted_intraday_return'] * (1.2 if df['close_std'].iloc[-1] > df['close_std'].median() else 0.8)
    
    # Combine Indicators
    df['composite_factor'] = df['trend_reversal_potential'] * df['volume_increase'] * df['adjusted_intraday_return']
    
    # Final Composite Factor Calculation
    df['final_composite_factor'] = df['composite_factor'] + df['combined_momentum']
    
    return df['final_composite_factor']
