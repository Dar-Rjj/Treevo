import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['volume_weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['aggregated_breakout'] = df['volume_weighted_breakout_ratio'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['normalized_volume_weighted_momentum'] = df['volume_weighted_momentum'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_factor'] = df['normalized_volume_weighted_momentum'] + df['aggregated_breakout']
    
    # Adjust for Trend
    long_term_trend_alpha = 0.05
    df['long_term_trend'] = df['close'].ewm(alpha=long_term_trend_alpha, adjust=False).mean()
    df['trend_adjusted_factor'] = df['combined_factor'] - df['long_term_trend']
    
    # Generate Final Alpha Factor
    final_smoothing_alpha = 0.05
    df['final_alpha_factor'] = df['trend_adjusted_factor'].ewm(alpha=final_smoothing_alpha, adjust=False).mean()
    
    return df['final_alpha_factor']
