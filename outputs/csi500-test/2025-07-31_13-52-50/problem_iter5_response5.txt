import pandas as pd

def heuristics_v2(df, n=20, m=14, k1=0.5, k2=0.3, k3=0.2):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation'].shift(1)
    df['trend_reversal'] = (df['deviation'] > df['prev_deviation']) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Combine Price and Volume Momentum
    df['weighted_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = 0
    df.loc[df['trend_reversal'], 'composite_factor'] += 1  # Higher score for trend reversal
    df['composite_factor'] += df['weighted_momentum']
    
    # Calculate Volume Weighted Moving Average (VWMA) Momentum
    df['daily_return_ma'] = df['daily_return'].rolling(window=n).mean()
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']
    df['vwma_momentum'] = df['daily_return_ma'] * df['volume_weight']
    
    # Combined Final Alpha Factor
    df['final_alpha_factor'] = df['composite_factor'] + k3 * df['vwma_momentum']
    
    return df['final_alpha_factor']
