```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ewma_returns'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 20
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregate_breakout'] = df['breakout_indicator'].rolling(window=n).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = df['deviation_from_vwap'].diff() > 0
    df['volume_increase'] = df['volume'] > df['volume'].rolling(window=14).mean()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.3
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_ewma'] = df['ewma_returns'] * df['volume_weight'] + df['combined_momentum']
    
    # Generate Alpha Factor
    alpha_factor_window = 10
    df['alpha_factor'] = (df['adjusted_ewma'] + df['aggregate_breakout'] + df['deviation_from_vwap']).ewm(span=alpha_factor_window, adjust=False).mean()
    
    # Incorporate Daily Return
    df['alpha_factor'] *= df['daily_return']
    
    # Adjust for Volatility
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    df['alpha_factor'] /= df['volatility']
    
    # Introduce Intraday Range Indicator
    df['intraday_range'] = df['high'] - df['low']
    intraday_range_ema_span = 20
    df['intraday_range_ema'] = df['intraday_range'].ewm(span=intraday_range_ema_span, adjust=False).mean()
    k3 = 0.2
    df['alpha_factor'] += k3 * df['intraday_range_ema']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = df['alpha_factor']
    df['composite_factor'] += np.where(df['trend_reversal_potential'] & df['volume_increase'], 1, 0) * df['combined_momentum']
    df['composite_factor'] = df['composite_factor'].ewm(span=alpha_factor_window, adjust=False).mean()
    
    return df['composite_factor']
```
