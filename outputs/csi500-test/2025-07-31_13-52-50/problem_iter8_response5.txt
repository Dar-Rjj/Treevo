```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    money_flow_multiplier = (typical_price - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    money_flow_volume = money_flow_multiplier * df['volume']
    
    # Calculate the 10-day exponential moving average of money flow volume
    ema_money_flow_volume = money_flow_volume.ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day rolling sum of volume
    rolling_volume_sum = df['volume'].rolling(window=10).sum()
    
    # Calculate the factor
    factor = ema_money_flow_volume / rolling_volume_sum
    
    return factor
```
