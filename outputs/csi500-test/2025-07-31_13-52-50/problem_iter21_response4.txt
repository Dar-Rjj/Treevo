import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['daily_return'] = df['close'].pct_change()
    df['return_deviation'] = df['close'].diff()
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['return_deviation'] > df['return_deviation'].shift(1))
    df['volume_14ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = df['volume'] > df['volume_14ma']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Compute Price Momentum Factor
    n = 7
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 7
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.6
    k2 = 0.4
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma_20']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Incorporate Price Volatility
    price_volatility = df['close'].rolling(window=7).std()
    df['intraday_return'] = df['close'].pct_change()
    df['adjusted_intraday_return'] = np.where(price_volatility > price_volatility.median(), 
                                              df['intraday_return'] * 1.5, 
                                              df['intraday_return'] * 0.8)
    
    # Incorporate Momentum Shift
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_shift'] = 0
    df.loc[df['ema_5'] > df['ema_20'], 'momentum_shift'] += 0.5
    df.loc[df['ema_5'] < df['ema_20'], 'momentum_shift'] -= 0.5
    
    # Combine Indicators
    df['composite_factor'] = 0
    df.loc[df['trend_reversal'] & df['volume_increase'], 'composite_factor'] = (
        0.5 * df['adjusted_intraday_return'] + 0.5 * df['adjusted_combined_momentum'] + df['momentum_shift']
    )
    
    # Final Composite Factor Calculation
    df['final_factor'] = df['composite_factor'].fillna(0)
    
    return df['final_factor']
