```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20  # Lookback period
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()

    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']

    # Combine Moving Average and Volume Weight
    df['volume_weighted_ma'] = df['ma_daily_return'] * df['volume_weight']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['sum_volume_weighted_breakout'] = df['volume_weighted_breakout'].rolling(window=n).sum()
    df['weighted_price_breakout'] = df['price_breakout_ratio'] * df['volume_weight']

    # Generate Alpha Factor
    alpha_factor = df['volume_weighted_ma'] + df['sum_volume_weighted_breakout'] + df['weighted_price_breakout']
    alpha_factor = alpha_factor.rolling(window=n).mean()

    # Enhance Alpha Factor
    df['enhanced_alpha_factor'] = df['daily_return'] * alpha_factor

    # Adjust for Volatility
    df['volatility'] = df[['high', 'low', 'close']].pct_change().rolling(window=n).std().mean(axis=1)
    df['alpha_factor'] = df['enhanced_alpha_factor'] / df['volatility']

    return df['alpha_factor']
```
