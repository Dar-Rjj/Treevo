```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['smoothed_returns'] = df['daily_returns'].ewm(span=n, adjust=False).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['smoothed_returns'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum_normalized'] = df['volume_weighted_momentum'] / df['avg_volume']

    # Weight Price Breakout Ratio by Volume
    df['weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']

    # Add Volume-Weighted Momentum to Aggregated Breakout Ratios
    df['alpha_factor'] = df['weighted_breakout_ratio'] + df['volume_weighted_momentum_normalized']

    return df['alpha_factor'].dropna()
```
