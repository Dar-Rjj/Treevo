```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Breakout Indicators
    aggregated_indicators = volume_weighted_breakout.rolling(window=20).sum()
    
    # Generate Alpha Factor
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change().fillna(0)
    
    # Compute Moving Average of Daily Returns
    moving_avg_daily_returns = daily_returns.rolling(window=20).mean()
    
    # Calculate Volume Weight
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Synthesize Final Alpha Factor
    alpha_factor = (aggregated_indicators.rolling(window=20).mean() * 
                    moving_avg_daily_returns * 
                    volume_weight)
    
    return alpha_factor
```
