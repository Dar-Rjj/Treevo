import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['daily_return_deviation'] = df['close'].diff()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compare Daily Return Deviation with VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    df['trend_reversal'] = df['deviation_from_vwap'].diff().apply(lambda x: 1 if x > 0 else 0)
    
    # Check for Volume Increase
    df['vol_14_ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['vol_14_ma']).astype(int)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['open']) / df['open']
    
    # Adjust Intraday Return for Volume
    df['vol_7_ma'] = df['volume'].rolling(window=7).mean()
    df['volume_diff'] = df['volume'] - df['vol_7_ma']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_diff']
    
    # Incorporate Price Volatility
    df['close_7_std'] = df['close'].rolling(window=7).std()
    df['volatility_adjusted_intraday_return'] = df['adjusted_intraday_return'] * \
        df['close_7_std'].apply(lambda x: 2.5 if x > df['close_7_std'].median() else 0.7)
    
    # Compute Price Momentum Factor
    n = 5
    df['price_momentum'] = df['daily_return_deviation'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 5
    df['volume_change'] = df['volume'].diff()
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['vol_20_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['vol_20_ma']
    df['adjusted_combined_momentum'] = df['combined_momentum'] / (df['volume_deviation'] + 1e-6)
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['momentum_shift'] = ((df['sma_5'] > df['sma_20']) & (df['sma_5'].shift(1) <= df['sma_20'].shift(1))).astype(int) * 1.5 - \
                           ((df['sma_5'] < df['sma_20']) & (df['sma_5'].shift(1) >= df['sma_20'].shift(1))).astype(int) * 1.5
    
    # Introduce Price and Volume Correlation
    df['correlation'] = df['close'].rolling(window=10).corr(df['volume'])
    df['correlation_adjusted_momentum'] = df['adjusted_combined_momentum'] * \
        df['correlation'].apply(lambda x: 1.2 if x > 0 else 0.8)
    
    # Enhance Intraday Dynamics
    df['intraday_price_range'] = df['high'] - df['open']
    df['normalized_intraday_price_range'] = df['intraday_price_range'] / df['close']
    df['intraday_volume_ratio'] = df['volume'] / df['vol_7_ma']
    
    # Combine Indicators
    df['trend_reversal_score'] = df['trend_reversal'] * df['volume_increase']
    df['final_factor'] = (
        df['trend_reversal_score'] * 1.0 +
        df['volatility_adjusted_intraday_return']
