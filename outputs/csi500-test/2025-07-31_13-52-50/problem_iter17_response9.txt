```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, k1=0.5, k2=0.5, smoothing_factor=0.2):
    # Calculate Daily Price Movement Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['DailyReturn'] = df['close'].diff()
    
    # Identify Trend Reversal Potential
    df['TrendReversal'] = (df['DailyReturn'] > df['DailyReturn'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Price Momentum Factor
    df['PriceMomentum'] = df['DailyReturn'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['VolumeChange'] = df['volume'].diff()
    df['VolumeMomentum'] = df['VolumeChange'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    df['CombinedMomentum'] = k1 * df['PriceMomentum'] + k2 * df['VolumeMomentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['VolumeMA20'] = df['volume'].rolling(window=20).mean()
    df['VolumeDeviation'] = df['volume'] - df['VolumeMA20']
    df['VolumeAdjustmentFactor'] = df['VolumeDeviation'] + 1e-6
    df['AdjustedMomentum'] = df['CombinedMomentum'] / df['VolumeAdjustmentFactor']
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['PriceBreakoutRatio'] = (df['high'] - df['open']) / df['HighLowRange']
    
    # Calculate Volume Breakout Indicator
    df['VolumeBreakoutIndicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['WeightedBreakoutRatio'] = df['PriceBreakoutRatio'] * df['VolumeBreakoutIndicator']
    
    # Incorporate Price Action Context
    df['PriceChange'] = df['close'] - df['open']
    df['PriceChangeSign'] = np.sign(df['PriceChange'])
    df['AdjustedBreakoutRatio'] = df['WeightedBreakoutRatio'] * df['PriceChangeSign']
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['VolumeMA20'] = df['volume'].rolling(window=20).mean()
    df['VolumeAdjustedIntradayReturn'] = (df['volume'] - df['VolumeMA20']) * df['IntradayReturn']
    
    # Incorporate Price Volatility
    df['CloseStdDev'] = df['close'].rolling(window=10).std()
    df['VolatilityAdjustedIntradayReturn'] = df['IntradayReturn'] * np.where(df['CloseStdDev'] > df['CloseStdDev'].median(), 1.5, 0.7)
    
    # Determine Daily Return Deviation from VWAP
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAPDeviation'] = df['close'] - df['VWAP']
    
    # Generate Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['AdjustedMomentum'] +
        df['AdjustedBreakoutRatio'] +
        df['VolatilityAdjustedIntradayReturn'] +
        df['VWAPDeviation']
    )
    
    # Smooth with Exponential Moving Average
    df['SmoothedAlphaFactor'] = df['FinalAlphaFactor'].ewm(alpha=smoothing_factor).mean()
    
    return df['SmoothedAlphaFactor']
```
