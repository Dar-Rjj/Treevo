```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day exponential moving average of the daily return
    df['ema_10'] = df['daily_return'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day volatility (standard deviation) of the daily return
    df['vol_10'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day cumulative sum of the volume
    df['cum_vol_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day moving average of the close price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day exponential moving average of the close price
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day range (high - low)
    df['range_10'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    
    # Calculate the 10-day mean of the range
    df['mean_range_10'] = df['range_10'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the range
    df['std_range_10'] = df['range_10'].rolling(window=10).std()
    
    # Calculate the factor as a combination of the above metrics
    df['factor'] = (df['ma_10'] + df['ema_10']) / (df['vol_10'] + 0.0001) * df['cum_vol_10'] * (df['ma_close_10'] - df['ema_close_10']) / (df['mean_range_10'] + df['std_range_10'] + 0.0001)
    
    return df['factor']
```
