```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout_indicator = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Indicators
    aggregated_indicators = volume_weighted_breakout_indicator.rolling(window=10).sum()
    
    # Generate Alpha Factor
    alpha_factor = aggregated_indicators.rolling(window=5).mean()
    
    return alpha_factor
```
