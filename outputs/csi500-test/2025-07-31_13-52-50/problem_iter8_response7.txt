import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Price Momentum Factor
    n_days = 5
    df['price_momentum'] = df['daily_return'].rolling(window=n_days).sum()
    
    # Compute Volume Momentum Factor
    m_days = 5
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_momentum'] = df['volume_change'].rolling(window=m_days).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.6
    k2 = 0.4
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])
    
    # Adjust Momentum for Volume Volatility
    lookback_days = 20
    df['vol_ma'] = df['volume'].rolling(window=lookback_days).mean()
    df['vol_deviation'] = df['volume'] - df['vol_ma']
    df['volume_adjustment_factor'] = (df['vol_deviation'] + 1e-6)  # Add a small constant to avoid division by zero
    df['adjusted_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = df['trend_reversal_potential'].astype(int) * 2 + df['adjusted_momentum']
    
    return df['composite_factor']
