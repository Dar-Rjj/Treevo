import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['daily_return'] = df['close'].diff()
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return'] > df['daily_return'].shift(1)).astype(int)
    
    # Check for Volume Increase
    df['volume_14_ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['volume_14_ma']).astype(int)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_from_vwap'] = df['close'] - df['vwap']
    
    # Compute Price Momentum Factor
    n = 5
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 5
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_20_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_20_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Incorporate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=7).std()
    df['intraday_return'] = df['close'] - df['open']
    df['intraday_return_adjusted'] = np.where(
        df['price_volatility'] > df['price_volatility'].median(),
        df['intraday_return'] * 1.3,
        df['intraday_return'] * 0.7
    )
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['momentum_shift'] = 0
    df.loc[df['sma_5'] > df['sma_20'], 'momentum_shift'] = 0.6
    df.loc[df['sma_5'] < df['sma_20'], 'momentum_shift'] = -0.6
    
    # Combine Indicators
    df['weighted_intraday_return'] = df['intraday_return_adjusted'] * (df['trend_reversal'] + df['volume_increase'])
    df['smoothed_score'] = df['weighted_intraday_return'].ewm(span=5, adjust=False).mean()
    
    # Final Composite Factor Calculation
    df['composite_factor'] = (
        df['adjusted_momentum'] +
        df['smoothed_score'] +
        df['momentum_shift']
    )
    
    return df['composite_factor']
