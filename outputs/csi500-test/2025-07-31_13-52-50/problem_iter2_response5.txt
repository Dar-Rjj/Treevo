import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal_potential'] = (df['deviation_from_vwap'] > df['prev_deviation']).astype(int)
    
    # Check for Volume Increase
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['volume_ma_14']).astype(int)
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    n = 20
    df['ma_daily_return'] = df['daily_return'].rolling(window=n, min_periods=1).mean()
    
    # Calculate Volume Weight
    df['average_volume'] = df['volume'].rolling(window=n, min_periods=1).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']
    
    # Combine Moving Average and Volume Weight
    df['combined_moving_average'] = df['ma_daily_return'] * df['volume_weight']
    
    # Final Score Calculation
    df['final_score'] = df['trend_reversal_potential'] * df['volume_increase'] * df['combined_moving_average']
    
    return df['final_score']
