```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    m = 20
    df['price_momentum'] = df['daily_return'].rolling(window=m).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['vwap_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_breakout'] = df['vwap_breakout_indicator'].rolling(window=20).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'].diff() > 0) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_ema'] = df['ema_daily_return'] * df['volume_weight'] + df['combined_momentum']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['adjusted_ema'] + df['aggregated_breakout'] + df['return_deviation_from_vwap']
    
    # Smooth with Exponential Moving Average
    df['smoothed_alpha_factor'] = df['alpha_factor'].ewm(span=20, adjust=False).mean()
    
    # Enhance Alpha Factor
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    df['enhanced_alpha_factor'] = (df['smoothed_alpha_factor'] * df['daily_return']) / df['volatility']
    
    # Final Composite Factor Calculation
    df['final_composite_factor'] = np.where(df['trend_reversal_potential'], df['enhanced_alpha_factor'] + df['combined_momentum'], df['enhanced_alpha_factor'])
    
    return df['final_composite_factor']
```
