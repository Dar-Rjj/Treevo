```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Indicators
    aggregated_indicators = volume_weighted_breakout.rolling(window=10).sum()
    
    # Generate Alpha Factor with Moving Average
    alpha_factor = aggregated_indicators.rolling(window=20).mean()
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.2
    alpha_factor = alpha_factor.ewm(alpha=smoothing_factor).mean()
    
    # Incorporate Daily Return
    daily_return = df['close'].pct_change()
    alpha_factor = alpha_factor * daily_return
    
    # Adjust for Volatility
    volatility = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    alpha_factor = alpha_factor / volatility
    
    return alpha_factor
```
