import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()

    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']

    # Identify Trend Reversal Potential
    df['return_deviation'] = df['daily_return'] - df['daily_return'].shift(1)
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['trend_reversal'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume_ma_14'])

    # Compute Price Momentum Factor
    n = 5
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()

    m = 5
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m).sum()

    # Combine Price and Volume Momentum
    k1 = 0.7
    k2 = 0.3
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']

    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma_20']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 0.01
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])

    # Aggregate Breakout Indicators
    df['weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
