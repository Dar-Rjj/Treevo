```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['prev_deviation']).astype(int)
    
    # Check for Volume Increase
    df['vol_moving_avg_5'] = df['volume'].rolling(window=5).mean()
    df['volume_increase'] = (df['volume'] > df['vol_moving_avg_5']).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['vol_moving_avg_7'] = df['volume'].rolling(window=7).mean()
    df['vol_adjustment'] = df['volume'] - df['vol_moving_avg_7']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['vol_adjustment']
    
    # Incorporate Price Volatility
    df['close_std_7'] = df['close'].rolling(window=7).std()
    df['volatility_factor'] = 1.3 if df['close_std_7'].iloc[-1] > df['close_std_7'].mean() else 0.7
    df['adjusted_intraday_return'] *= df['volatility_factor']
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['momentum_shift'] = 0
    df.loc[df['sma_5'] > df['sma_20'], 'momentum_shift'] = 0.6
    df.loc[df['sma_5'] < df['sma_20'], 'momentum_shift'] = -0.6
    df['adjusted_intraday_return'] += df['momentum_shift']
    
    # Combine Indicators
    df['combined_score'] = 0
    df['combined_score'] += df['trend_reversal'] * df['volume_increase'] * df['adjusted_intraday_return']
    df['combined_score'] = df['combined_score'].ewm(span=5, adjust=False).mean()  # Smoothing factor
    
    return df['combined_score']
```
