```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - df['volume_ma_20']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['volatility_adjusted_intraday_return'] = df['volume_adjusted_intraday_return'] * df.apply(lambda x: 1.5 if x['close_std_10'] > 0.5 else 0.7, axis=1)
    
    # Calculate Daily Price Movement Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['previous_daily_return_deviation'] = df['daily_return_deviation'].shift(1)
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['trend_reversal_potential'] = (df['daily_return_deviation'] > df['previous_daily_return_deviation']) & (df['volume'] > df['volume_ma_14'])
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    df['ma_daily_return_20'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine Moving Average and Volume Weight
    df['weighted_ma_daily_return'] = df['ma_daily_return_20'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['aggregate_breakout_indicators'] = df['price_breakout_ratio'] * df['volume']
    
    # Generate Intermediate Alpha Score
    df['intermediate_alpha_score'] = df.apply(lambda x: 1 if x['trend_reversal_potential'] else 0, axis=1) * (df['weighted_ma_daily_return'] + df['aggregate_breakout_indicators'] + df['volatility_adjusted_intraday_return'])
    
    # Smooth Final Alpha Factor
    df['final_alpha_factor'] = df['intermediate_alpha_score'].rolling(window=10).mean()
    
    return df['final_alpha_factor']
```
