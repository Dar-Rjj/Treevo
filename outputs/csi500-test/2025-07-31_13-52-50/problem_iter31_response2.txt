```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['smoothed_returns'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()

    # Compute Price Momentum Factor
    price_momentum_window = 20
    df['price_momentum'] = df['daily_return'].rolling(window=price_momentum_window).sum()

    # Compute Volume Momentum Factor
    volume_momentum_window = 20
    df['volume_momentum'] = df['volume_change'].rolling(window=volume_momentum_window).sum()

    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])

    # Adjust Momentum for Volume Volatility
    volume_ma_window = 20
    df['volume_ma'] = df['volume'].rolling(window=volume_ma_window).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    small_constant = 1e-6
    df['adjusted_momentum'] = df['combined_momentum'] / (df['volume_deviation'] + small_constant)

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vw_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']

    # Aggregate Indicators
    df['aggregate_breakout'] = df['vw_breakout_indicator']

    # Smooth Aggregate Breakout Indicators
    smoothing_factor = 0.1
    df['smoothed_aggregate_breakout'] = df['aggregate_breakout'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Determine Daily Return Deviation from VWAP
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Calculate Volume-Adjusted Intraday Return
    df['intraday_return_vol_adj'] = (df['close'] - df['open']) * df['volume']

    # Incorporate Price Action Context
    df['price_change'] = df['close'] - df['open']
    df['price_action_context'] = df['price_change'].apply(lambda x: 1 if x > 0 else -1)
    df['context_adjusted_breakout'] = df['smoothed_aggregate_breakout'] * df['price_action_context']

    # Final Composite Factor Calculation
    df['composite_factor'] = (
        df['trend_reversal_potential'] * 
        df['adjusted_momentum'] * 
        df['context_adjusted_breakout'] * 
        df['intraday_return_vol_adj']
    )

    # Generate Alpha Factor
    moving_avg_window = 20
    df['alpha_factor'] = df['composite_factor'].rolling(window=moving_avg_window).mean()

    # Enhance Alpha Factor
    df['enhanced_alpha_factor'] = df['alpha_factor'] * df['daily_return']

    return df['enhanced_alpha_factor'].dropna()
```
