```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['smoothed_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()

    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['price_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']

    # Aggregate Indicators
    df['sum_volume_weighted_breakout'] = df['volume_weighted_breakout'].rolling(window=lookback_period).sum()
    df['weighted_breakout_ratio'] = df['breakout_ratio'] * df['volume']
    df['sum_weighted_breakout_ratio'] = df['weighted_breakout_ratio'].rolling(window=lookback_period).sum()

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['smoothed_return'] * df['volume']
    df['average_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weighted_momentum_adj'] = df['volume_weighted_momentum'] / df['average_volume']

    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum'] = df['volume_weighted_momentum_adj'] + df['sum_weighted_breakout_ratio']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['trend_reversal_signal'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Compute Price Momentum Factor
    n_days = 5
    df['price_momentum'] = df['daily_return'].rolling(window=n_days).sum()

    # Compute Volume Momentum Factor
    m_days = 7
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m_days).sum()

    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum_score'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']

    # Adjust Combined Momentum for Volume Volatility
    volume_ma = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - volume_ma
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_combined_momentum_score'] = df['combined_momentum_score'] / df['volume_adjustment_factor']

    # Introduce Trend Following Component
    long_lookback = 50
    short_lookback = 20
    df['long_term_ma'] = df['close'].rolling(window=long_lookback).mean()
    df['short_term_ma'] = df['close'].rolling(window=short_lookback).mean()

    # Generate Trend Signal
    df['trend_signal'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Integrate Trend Signal into Alpha Factor
    df['integrated_alpha_factor'] = df['adjusted_combined_momentum_score'] * df['trend_signal']

    # Final Composite Factor Calculation
    df['final_alpha_factor'] = df['integrated_alpha_factor'] * (1 + 2 * df['trend_reversal_signal'])

    # Smooth with Exponential Moving Average
    smoothing_factor = 0.2
    df['final_smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=smoothing_factor, adjust=False).mean()

    return df['final_smoothed_alpha_factor']
```
