```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']) / df['volume'].rolling(window=1, min_periods=1).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & \
                                     (df['volume'] > df['volume'].rolling(window=5, min_periods=1).mean())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_adjustment'] = (df['volume'] - df['volume'].rolling(window=7, min_periods=1).mean()) * df['intraday_return']
    
    # Incorporate Price Volatility
    std_close_7d = df['close'].rolling(window=7, min_periods=1).std()
    df['volatility_adjusted_intraday_return'] = np.where(std_close_7d > std_close_7d.mean(), 
                                                         df['intraday_return'] * 1.5, 
                                                         df['intraday_return'] * 0.5)
    
    # Calculate EMAs
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Momentum Shift
    ema_cross_above = (df['ema_5'] > df['ema_20']) & (df['ema_5'].shift(1) <= df['ema_20'].shift(1))
    ema_cross_below = (df['ema_5'] < df['ema_20']) & (df['ema_5'].shift(1) >= df['ema_20'].shift(1))
    df['momentum_shift'] = np.where(ema_cross_above, 0.8, np.where(ema_cross_below, -0.8, 0))
    
    # Enhance Trend Reversal Detection
    df['ema_diff'] = df['ema_5'] - df['ema_20']
    df['trend_strength'] = np.where((df['ema_5'] > df['ema_20']) & (df['ema_diff'] < df['ema_diff'].shift(1)), -1,
                                    np.where((df['ema_5'] < df['ema_20']) & (df['ema_diff'] > df['ema_diff'].shift(1)), 1, 0))
    
    # Incorporate Price-Volume Divergence
    df['volume_ma_5'] = df['volume'].rolling(window=5, min_periods=1).mean()
    df['volume_ma_20'] = df['volume'].rolling(window=20, min_periods=1).mean()
    df['volume_divergence'] = df['volume_ma_5'] - df['volume_ma_20']
    volume_divergence_significant = (df['volume_ma_5'] < df['volume_ma_20']) & (df['volume_divergence'] > df['volume_divergence'].shift(1))
    df['divergence_adjusted_intraday_return'] = np.where(volume_divergence_significant, 
                                                         df['intraday_return'] * 1.2, 
                                                         df['intraday_return'] * 0.8)
    
    # Combine Indicators
    df['final_alpha_factor'] = np.where(df['trend_reversal_potential'], 
                                        df['volume_adjustment'] + df['momentum_shift'] + df['trend_strength'] * df['divergence_adjusted_intraday_return'], 
                                        0)
    
    # Apply a weighted average to balance the contributions
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
```
