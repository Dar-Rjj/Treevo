import pandas as pd

def heuristics_v2(df, n=10, m=10, k1=0.5, k2=0.5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['deviation_from_vwap'] > df['deviation_from_vwap'].shift(1)).astype(int)
    
    # Check for Volume Increase
    df['vol_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['vol_ma_14']).astype(int)
    
    # Combine Price and Volume Momentum
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = (
        df['trend_reversal_potential'] * df['volume_increase'] * 2 + 
        df['combined_momentum']
    )
    
    return df['composite_factor'].dropna()
