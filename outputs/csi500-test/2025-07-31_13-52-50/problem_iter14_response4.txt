import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['deviation'] > df['deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=20, adjust=False).mean()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']
    
    # Aggregate Volume-Weighted Breakout Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout'].rolling(window=20).sum()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_returns'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Indicators
    df['combined_indicators'] = df['volume_weighted_momentum'] + df['aggregated_breakout']
    
    # Smooth with Exponential Moving Average
    df['smoothed_combined_indicators'] = df['combined_indicators'].ewm(span=20, adjust=False).mean()
    
    # Adjust Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_momentum'] = df['volume_weighted_momentum'] / df['volume_adjustment_factor']
    
    # Final Score Calculation
    df['final_score'] = np.where(df['trend_reversal'], df['smoothed_combined_indicators'] * 1.5, df['smoothed_combined_indicators'])
    
    return df['final_score']
