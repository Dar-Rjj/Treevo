import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout_indicator = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Indicators
    sum_volume_weighted_breakout_indicators = volume_weighted_breakout_indicator.rolling(window=10, min_periods=1).sum()
    weighted_price_breakout_ratio = price_breakout_ratio * df['volume']
    sum_volume_weighted_breakout_ratios = weighted_price_breakout_ratio.rolling(window=10, min_periods=1).sum()
    
    # Generate Alpha Factor
    alpha_factor = (sum_volume_weighted_breakout_indicators / sum_volume_weighted_breakout_ratios).ewm(span=5, adjust=False).mean()
    
    return alpha_factor
