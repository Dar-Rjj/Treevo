import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided market data.
    The factor is designed to help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the close price
    df['std_10_close'] = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day normalized close price
    df['norm_close'] = (df['close'] - df['ma_10_close']) / df['std_10_close']
    
    # Calculate the 10-day moving average of the volume
    df['ma_10_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the volume
    df['std_10_volume'] = df['volume'].rolling(window=10).std()
    
    # Calculate the 10-day normalized volume
    df['norm_volume'] = (df['volume'] - df['ma_10_volume']) / df['std_10_volume']
    
    # Calculate the 10-day moving average of the amount
    df['ma_10_amount'] = df['amount'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the amount
    df['std_10_amount'] = df['amount'].rolling(window=10).std()
    
    # Calculate the 10-day normalized amount
    df['norm_amount'] = (df['amount'] - df['ma_10_amount']) / df['std_10_amount']
    
    # Calculate the factor as a weighted sum of the normalized close, volume, and amount
    df['factor'] = 0.5 * df['norm_close'] + 0.3 * df['norm_volume'] + 0.2 * df['norm_amount']
    
    # Drop the intermediate columns
    df.drop(columns=['return', 'ma_10_close', 'std_10_close', 'norm_close', 
                     'ma_10_volume', 'std_10_volume', 'norm_volume',
                     'ma_10_amount', 'std_10_amount', 'norm_amount'], inplace=True)
    
    return df['factor']
