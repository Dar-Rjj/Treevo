```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation
    df['daily_return'] = df['close'].pct_change()
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return'] > df['daily_return'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Moving Average of Daily Returns
    lookback = 20
    df['moving_avg_return'] = df['daily_return'].rolling(window=lookback).mean()
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=lookback).mean()
    
    # Combine Moving Average and Volume Weight
    df['weighted_moving_avg'] = df['moving_avg_return'] * df['volume_weight']
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=lookback).sum()
    
    # Compute Volume Momentum Factor
    m = 10
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma_20']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregate_breakout_indicator'] = df['volume_weighted_breakout'].rolling(window=lookback).sum() * df['volume_weight']
    
    # Smooth Aggregate Breakout Indicators
    smoothing_factor = 0.2
    df['smoothed_aggregate_breakout'] = df['aggregate_breakout_indicator'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Incorporate Price Action Context
    df['price_change'] = np.sign(df['close'] - df['open'])
    df['price_action_context'] = df['price_change'] * df['smoothed_aggregate_breakout']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_ma_220'] = df['volume'].rolling(window=220).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - df['volume_ma_220']) * df['intraday_return']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = 0.5 * df['trend_reversal'].astype(int) + 0.5 * df['price_action_context'] + df['volume_adjusted_intraday_return']
    
    # Enhance Alpha Factor
    df['alpha_factor'] = df['alpha_factor'] * df['daily_return']
    
    # Adjust for Volatility
    window_size = 20
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=window_size).std().mean(axis=1)
    df['normalized_alpha_factor'] = df['alpha_factor'] / df['volatility']
    
    return df['normalized_alpha_factor']
```
