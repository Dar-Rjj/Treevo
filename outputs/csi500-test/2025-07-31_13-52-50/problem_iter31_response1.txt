```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['vwb_indicator'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_indicators'] = df['vwb_indicator'].rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Momentum
    df['vwm'] = df['ema_daily_return'] * df['volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Indicators
    df['combined_momentum'] = df['vwm'] + df['aggregated_indicators']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['return_deviation'].diff() > 0).astype(int) * (df['volume'] > df['volume'].rolling(window=14).mean()).astype(int)
    
    # Compute Price and Volume Momentum
    n = 5
    m = 5
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum_score'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    small_constant = 1e-6
    df['volume_adjustment_factor'] = df['volume_deviation'] + small_constant
    df['adjusted_momentum_score'] = df['combined_momentum_score'] / df['volume_adjustment_factor']
    
    # Introduce Trend Following Component
    long_term_ma = 50
    short_term_ma = 20
    df['long_term_ma'] = df['close'].rolling(window=long_term_ma).mean()
    df['short_term_ma'] = df['close'].rolling(window=short_term_ma).mean()
    
    # Generate Trend Signal
    df['trend_signal'] = (df['short_term_ma'] > df['long_term_ma']).astype(int) * 2 - 1
    
    # Integrate Trend Signal into Alpha Factor
    df['trend_adjusted_momentum'] = df['adjusted_momentum_score'] * df['trend_signal']
    
    # Final Composite Factor Calculation
    smoothing_factor = 0.1
    df['final_alpha_factor'] = df['trend_adjusted_momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return df['final_alpha_factor']
```
