```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Breakout Indicators
    aggregated_breakout = volume_weighted_breakout.rolling(window=20).sum()
    
    # Apply Exponential Moving Average to Aggregated Indicators
    ema_aggregated_breakout = aggregated_breakout.ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Compute Weighted Moving Average of Daily Returns
    lookback_period = 20
    weights = np.arange(1, lookback_period + 1) / (lookback_period * (lookback_period + 1) / 2)
    weighted_returns = daily_returns.rolling(window=lookback_period).apply(lambda x: np.sum(weights * x), raw=True)
    
    # Synthesize Final Alpha Factor
    alpha_factor = ema_aggregated_breakout * weighted_returns
    
    return alpha_factor
```
