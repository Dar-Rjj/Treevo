import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['prev_deviation']) & (df['volume'] > df['volume'].rolling(window=5).mean())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_adjustment'] = (df['volume'] - df['volume'].rolling(window=7).mean()) * df['intraday_return']
    
    # Incorporate Price Volatility
    std_7d = df['close'].rolling(window=7).std()
    df['adjusted_intraday_return'] = df['intraday_return'] * 1.5 if std_7d.iloc[-1] > std_7d.mean() else df['intraday_return'] * 0.5
    
    # Incorporate Momentum Shift
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_shift'] = (df['ema_5'] > df['ema_20']) * 0.8 - (df['ema_5'] < df['ema_20']) * 0.8
    
    # Combine Indicators
    df['combined_indicator'] = (
        (df['trend_reversal'] * 1.0) +
        df['volume_adjustment'] +
        df['adjusted_intraday_return'] +
        df['momentum_shift']
    )
    
    # Apply to Predict Future Returns
    df['factor'] = df['combined_indicator'].ewm(span=3, adjust=False).mean()
    
    return df['factor']
