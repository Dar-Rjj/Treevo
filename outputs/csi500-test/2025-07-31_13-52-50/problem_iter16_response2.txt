```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, alpha=0.1, lookback_period=20):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    df['smoothed_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])

    # Aggregate Breakout Indicators
    df['weighted_price_breakout'] = df['price_breakout_ratio'] * df['volume']
    df['aggregated_breakout'] = df['weighted_price_breakout'].rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['smoothed_return'] * df['volume']
    df['average_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weighted_momentum_adjusted'] = df['volume_weighted_momentum'] / df['average_volume']

    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_factor'] = df['volume_weighted_momentum_adjusted'] + df['aggregated_breakout']

    # Generate Final Alpha Factor with Smoothing
    df['final_alpha_factor'] = df['combined_factor'].ewm(alpha=alpha, adjust=False).mean()

    return df['final_alpha_factor']
```
