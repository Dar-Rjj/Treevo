```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using open, high, low, close, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day EMA of the money flow
    ema_10_money_flow = money_flow.ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day EMA of the money flow
    ema_20_money_flow = money_flow.ewm(span=20, adjust=False).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = ema_10_money_flow / ema_20_money_flow
    
    # Calculate the 10-day EMA of the close price
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day EMA of the close price
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the close price ratio
    close_price_ratio = ema_10_close / ema_20_close
    
    # Combine the money flow ratio and close price ratio
    factor = money_flow_ratio * close_price_ratio
    
    return factor
```
