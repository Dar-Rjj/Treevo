```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']) / df['volume'].rolling(window=1).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['deviation_shift'] = df['deviation'].shift(1)
    df['trend_reversal'] = (df['deviation'] > df['deviation_shift'])
    
    # Check for Volume Increase
    df['volume_5d_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_increase'] = df['volume'] > df['volume_5d_ma']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_7d_ma'] = df['volume'].rolling(window=7).mean()
    df['adjusted_intraday_return'] = (df['volume'] - df['volume_7d_ma']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['close_7d_std'] = df['close'].rolling(window=7).std()
    df['volatility_adjustment'] = np.where(df['close_7d_std'] > df['close_7d_std'].mean(), 1.5, 0.5)
    df['adjusted_intraday_return'] *= df['volatility_adjustment']
    
    # Momentum Shift
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_shift'] = np.where(df['ema_5'] > df['ema_20'], 0.8, -0.8)
    
    # Combine Indicators
    df['combined_indicator'] = np.where((df['trend_reversal'] & df['volume_increase']), 
                                        df['adjusted_intraday_return'] + df['momentum_shift'], 
                                        df['adjusted_intraday_return'])
    
    # Apply smoothing factor
    df['smoothed_combined_indicator'] = df['combined_indicator'].rolling(window=3).mean()
    
    return df['smoothed_combined_indicator']
```
