import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    volume_weighted_breakout_ratios = price_breakout_ratio * df['volume']
    aggregate_breakout_indicators = volume_weighted_breakout_ratios.rolling(window=10).sum()
    
    # Generate Final Alpha Factor
    final_alpha_factor = aggregate_breakout_indicators.ewm(span=5, adjust=False).mean()
    
    return final_alpha_factor
