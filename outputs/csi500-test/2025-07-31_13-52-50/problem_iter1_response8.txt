import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Move
    intraday_move = df['close'] - df['open']
    
    # Compute Intraday Range Ratio
    intraday_range_ratio = intraday_range / intraday_move.abs()
    
    # Compute Intraday Momentum
    avg_daily_volume = df['volume'].rolling(window=5).mean()
    intraday_momentum = (intraday_move / avg_daily_volume)
    
    # Reverse Intraday Momentum
    smoothed_intraday_momentum = intraday_momentum.ewm(span=5, alpha=0.2).mean()
    reversed_intraday_momentum = -smoothed_intraday_momentum
    
    return reversed_intraday_momentum
