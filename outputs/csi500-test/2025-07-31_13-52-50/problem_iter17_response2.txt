```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout'] + (df['price_breakout_ratio'] * df['volume'])
    
    # Generate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum_normalized'] = df['volume_weighted_momentum'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_factor'] = df['volume_weighted_momentum_normalized'] + df['aggregated_breakout']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=n).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Enhance Alpha Factor
    df['alpha_factor'] = df['daily_return'] * df['adjusted_combined_momentum']
    
    # Adjust for Volatility
    window_size = 20
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=window_size).mean()
    df['alpha_factor'] /= df['volatility']
    
    # Final Composite Factor Calculation
    df['final_alpha_factor'] = df['combined_factor'] + (df['trend_reversal'] * 1.5)  # Assign higher score for stronger indication
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=n, adjust=False).mean()
    
    return df['final_alpha_factor']
```
