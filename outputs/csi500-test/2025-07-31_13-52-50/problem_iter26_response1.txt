```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()

    # Compute Volume Momentum Factor
    m = 20
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()

    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])

    # Adjust Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['adjusted_momentum'] = df['combined_momentum'] / (df['volume_deviation'] + 1e-6)

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vwb_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['aggregate_breakout'] = df['vwb_breakout'].rolling(window=n).sum()

    # Smooth Aggregate Breakout Indicators
    df['smoothed_aggregate_breakout'] = df['aggregate_breakout'].ewm(span=n, adjust=False).mean()

    # Determine Daily Return Deviation from VWAP
    df['vwap'] = (df['amount'] / df['volume'])
    df['deviation_from_vwap'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Calculate Volume-Adjusted Intraday Return
    df['volume_adjusted_intraday_return'] = (df['close'] - df['open']) * df['volume']

    # Final Composite Factor Calculation
    df['composite_factor'] = df['trend_reversal'] * df['smoothed_aggregate_breakout'] * df['volume_adjusted_intraday_return']

    # Generate Alpha Factor
    window_size = 20
    df['alpha_factor'] = df['composite_factor'].rolling(window=window_size).mean()

    # Enhance Alpha Factor
    df['enhanced_alpha_factor'] = df['alpha_factor'] * df['daily_return']

    # Adjust for Volatility
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1)
    df['normalized_alpha_factor'] = df['enhanced_alpha_factor'] / df['volatility']

    return df['normalized_alpha_factor'].dropna()
```
