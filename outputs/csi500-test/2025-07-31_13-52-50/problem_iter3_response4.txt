```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    volume_weighted_breakout = price_breakout_ratio * volume_breakout_indicator
    aggregated_breakout = volume_weighted_breakout.rolling(window=10).sum()
    
    # Generate Final Alpha Factor
    smoothing_factor = 0.2  # Define the smoothing factor
    final_alpha_factor = aggregated_breakout.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return final_alpha_factor
```
