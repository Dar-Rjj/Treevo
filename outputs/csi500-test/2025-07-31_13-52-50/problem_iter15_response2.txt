```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns (e.g., 20 days)
    n = 20
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()
    
    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    
    # Combine Moving Average and Volume Weight
    df['volume_weighted_ma'] = df['ma_daily_return'] * df['volume_weight']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['breakout_aggregate'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_adjustment'] = df['volume'] - df['volume'].rolling(window=10).mean()
    df['volume_adjusted_intraday_return'] = df['volume_adjustment'] * df['intraday_return']
    
    # Incorporate Price Volatility
    std_close = df['close'].rolling(window=10).std()
    df['volatility_adjustment'] = std_close.apply(lambda x: 1.5 if x > std_close.median() else 0.7)
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['volume_weighted_ma'] + 
                                df['breakout_aggregate'] + 
                                df['volume_adjusted_intraday_return']).rolling(window=n).mean()
    
    # Adjust Final Alpha Factor for Price Volatility
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['volatility_adjustment']
    
    return df['final_alpha_factor']
```
