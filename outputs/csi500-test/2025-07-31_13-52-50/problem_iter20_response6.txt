import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()

    # Compute Price Momentum Factor
    price_momentum = df['daily_return'].rolling(window=n).sum()
    k1 = 1.0  # Constant for price momentum

    # Compute Volume Momentum Factor
    m = 20
    volume_momentum = df['volume_change'].rolling(window=m).sum()
    k2 = 1.0  # Constant for volume momentum

    # Combine Price and Volume Momentum
    combined_momentum = k1 * price_momentum + k2 * volume_momentum

    # Adjust Momentum for Volume Volatility
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_deviation = df['volume'] - volume_ma
    volume_adjustment_factor = (volume_deviation + 1e-5) / volume_ma
    adjusted_momentum = combined_momentum / volume_adjustment_factor

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vwb_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['aggregated_vwb'] = df['vwb_indicator'].rolling(window=n).sum()

    # Smooth Aggregate Breakout Indicators
    smoothing_factor = 0.1
    df['smoothed_aggregated_vwb'] = df['aggregated_vwb'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Calculate Volume Weight
    average_volume = df['volume'].rolling(window=20).mean()
    df['volume_weight'] = df['volume'] / average_volume

    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_ema'] = df['ema_daily_return'] * df['volume_weight'] + adjusted_momentum

    # Final Composite Factor Calculation
    df['composite_factor'] = np.where(df['trend_reversal_potential'], 2 * df['adjusted_ema'], df['adjusted_ema'])

    # Generate Alpha Factor
    alpha_factor = (df['smoothed_aggregated_vwb'] + df['composite_factor']).rolling(window=n).mean()

    # Enhance Alpha Factor
    alpha_factor *= df['daily_return']
    
    # Adjust for Volatility
    volatility = df[['high', 'low', 'close']].std(axis=1, ddof=0).rolling(window=n).mean()
    alpha_factor /= volatility

    return alpha_factor.dropna()
