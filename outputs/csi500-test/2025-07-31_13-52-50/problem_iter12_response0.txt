import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    volume_weighted_breakout = price_breakout_ratio * volume_breakout_indicator
    
    # Incorporate Price Action Context
    price_change = df['close'] - df['open']
    price_action_sign = price_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Apply Price Action Context to Aggregate Breakout Indicators
    adjusted_breakout = volume_weighted_breakout * price_action_sign
    
    # Generate Final Alpha Factor
    alpha_factor = adjusted_breakout.ewm(span=10, adjust=False).mean()
    
    return alpha_factor
