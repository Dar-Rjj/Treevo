import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided DataFrame.
    The alpha factor is a combination of price and volume-based heuristics.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the computed alpha factor, indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor
    df['momentum'] = df['close'] / df['close'].shift(30) - 1
    
    # Calculate the volume factor
    df['volume_factor'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Calculate the range factor
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (df['return'] + df['momentum'] + 
                          (df['ma_10'] - df['ma_30']) + 
                          df['volume_factor'] * df['range'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
