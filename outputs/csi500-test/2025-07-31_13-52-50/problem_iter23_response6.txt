    df['daily_return'] = df['close'].pct_change()

    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['volume_weighted_ema'] = df['ema_daily_return'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['sum_volume_weighted_breakout'] = df['volume_weighted_breakout'].rolling(window=n).sum()
    df['weighted_price_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['sum_weighted_price_breakout_ratio'] = df['weighted_price_breakout_ratio'].rolling(window=n).sum()
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['volume_weighted_ema'] + df['sum_volume_weighted_breakout'] + df['sum_weighted_price_breakout_ratio']
    df['sma_alpha_factor'] = df['alpha_factor'].rolling(window=n).mean()
    
    # Smooth with Exponential Moving Average
    df['smoothed_alpha_factor'] = df['sma_alpha_factor'].ewm(span=n, adjust=False).mean()

    df['enhanced_alpha_factor'] = df['daily_return'] * df['smoothed_alpha_factor']
