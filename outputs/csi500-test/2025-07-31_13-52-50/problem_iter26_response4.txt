import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    
    # Calculate the 5-day moving average of the closing price
    df['close_5d_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20d_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 10-day sum of the volume
    df['volume_10d_sum'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 5-day rolling correlation between the close price and the volume
    df['corr_close_volume_5d'] = df['close'].rolling(window=5).corr(df['volume'])
    
    # Calculate the 10-day rolling mean of the amount
    df['amount_10d_mean'] = df['amount'].rolling(window=10).mean()
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day rolling mean of the daily return
    df['daily_return_5d_mean'] = df['daily_return'].rolling(window=5).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        df['close_5d_ma'] / df['close'] +
        df['close_20d_std'] / df['close'] +
        df['volume_10d_sum'] / df['volume'] +
        df['corr_close_volume_5d'] +
        df['amount_10d_mean'] / df['amount'] +
        df['daily_return_5d_mean']
    )
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
