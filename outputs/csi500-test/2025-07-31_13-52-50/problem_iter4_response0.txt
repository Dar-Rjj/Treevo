```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day Close Price Average
    df['14_day_close_avg'] = df['close'].rolling(window=14).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = df['close'] - df['14_day_close_avg']
    
    # Obtain current and 14-day average Volumes
    df['current_volume'] = df['volume']
    df['14_day_volume_avg'] = df['volume'].rolling(window=14).mean()
    
    # Calculate the Volume Change Ratio
    df['volume_change_ratio'] = df['current_volume'] / df['14_day_volume_avg']
    
    # Adjust for Volume Impact
    df['price_momentum_with_volume_impact'] = df['price_momentum'] * df['volume_change_ratio']
    
    return df['price_momentum_with_volume_impact']
```
