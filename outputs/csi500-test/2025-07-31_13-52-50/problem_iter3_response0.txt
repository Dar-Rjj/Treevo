```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adj'] = (df['volume'] - df['volume_ma_20']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['price_volatility_adjusted'] = df['volume_adj'] * df['close_std_10'].apply(
        lambda x: 1.5 if x > df['close_std_10'].median() else 0.7
    )
    
    # Final factor
    factor = df['price_volatility_adjusted']
    
    return factor
```
