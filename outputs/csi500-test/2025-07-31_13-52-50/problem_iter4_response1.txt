```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['returns'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20  # Lookback period
    df['moving_avg_returns'] = df['returns'].rolling(window=n, min_periods=1).mean()

    # Calculate Volume Weight
    df['average_volume'] = df['volume'].rolling(window=n, min_periods=1).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']

    # Combine Moving Average and Volume Weight
    df['vwma_momentum'] = df['moving_avg_returns'] * df['volume_weight']

    return df['vwma_momentum']
```
