```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    lookback_period = 20
    df['ma_daily_returns'] = df['daily_returns'].rolling(window=lookback_period).mean()
    
    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    
    # Combine Moving Average and Volume Weight
    df['volume_weighted_ma_returns'] = df['ma_daily_returns'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['aggregate_breakout'] = df['price_breakout_ratio'] * df['volume']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - df['volume_moving_avg']) * df['intraday_return']
    
    # Generate Intermediate Alpha Score
    df['intermediate_alpha_score'] = 0
    df.loc[df['trend_reversal'], 'intermediate_alpha_score'] += 1
    df['intermediate_alpha_score'] += df['volume_weighted_ma_returns']
    df['intermediate_alpha_score'] += df['aggregate_breakout']
    df['intermediate_alpha_score'] += df['volume_adjusted_intraday_return']
    
    # Smooth Final Alpha Factor
    smoothing_window = 5
    df['final_alpha_factor'] = df['intermediate_alpha_score'].rolling(window=smoothing_window).mean()
    
    return df['final_alpha_factor']
```
