```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Close_Volume'] = df['close'] * df['volume']
    df['VWAP'] = df['Close_Volume'].rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['Deviation'] = df['close'] - df['VWAP']
    
    # Identify Trend Reversal Potential
    df['Prev_Deviation'] = df['Deviation'].shift(1)
    df['Trend_Reversal'] = (df['Deviation'] > df['Prev_Deviation']) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Price Breakout Ratio
    df['Price_Breakout_Ratio'] = (df['high'] - df['open']) / df['Daily_Price_Range']
    
    # Calculate Volume Breakout Indicator
    df['Volume_Breakout_Indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Volume-Weighted Breakout Ratios
    df['Aggregated_Breakout_Ratios'] = df['Price_Breakout_Ratio'] * df['volume']
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.5
    df['Smoothed_Breakout_Ratios'] = df['Aggregated_Breakout_Ratios'].ewm(alpha=smoothing_factor, min_periods=14).mean()
    
    # Calculate Momentum Score
    lookback_days = 20
    df['Close_Price_Change'] = df['close'] - df['close'].shift(lookback_days)
    df['Momentum_Score'] = df['Close_Price_Change'].rolling(window=lookback_days).sum()
    
    # Adjust Momentum for Volume Volatility
    volume_lookback_days = 20
    df['Avg_Volume'] = df['volume'].rolling(window=volume_lookback_days).mean()
    df['Volume_Deviation'] = df['volume'] - df['Avg_Volume']
    df['Volume_Adjustment_Factor'] = (df['Volume_Deviation'] + 1e-6)  # Add small constant to avoid division by zero
    df['Adjusted_Momentum_Score'] = df['Momentum_Score'] / df['Volume_Adjustment_Factor']
    
    # Final Score Calculation
    df['Factor_Value'] = df.apply(
        lambda row: row['Adjusted_Momentum_Score'] * 1.5 if row['Trend_Reversal'] else row['Adjusted_Momentum_Score'], 
        axis=1
    )
    
    return df['Factor_Value']
```
