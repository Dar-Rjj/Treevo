```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_return = (df['volume'] - moving_avg_volume) * intraday_return
    
    # Incorporate Price Volatility
    std_close = df['close'].rolling(window=10).std()
    volatility_adjusted_return = volume_adjusted_return.where(
        std_close > std_close.mean(),
        other=volume_adjusted_return * 0.7
    ) * 1.5
    
    return volatility_adjusted_return
```
