```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20
    df['moving_avg_returns'] = df['daily_return'].rolling(window=n).mean()

    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']

    # Combine Moving Average and Volume Weight
    df['vol_weighted_moving_avg'] = df['moving_avg_returns'] * df['volume_weight']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vol_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['aggregated_breakout'] = df['vol_weighted_breakout'].rolling(window=n).sum() / df['volume_weight']

    # Generate Alpha Factor
    alpha_factor_window = 10
    df['alpha_factor'] = (df['vol_weighted_moving_avg'] + df['aggregated_breakout']).rolling(window=alpha_factor_window).mean()

    # Enhance Alpha Factor
    df['alpha_factor'] = df['alpha_factor'] * df['daily_return']

    # Adjust for Volatility
    volatility_window = 20
    df['volatility'] = df['high'].rolling(window=volatility_window).std() + df['low'].rolling(window=volatility_window).std() + df['close'].rolling(window=volatility_window).std()
    df['alpha_factor'] = df['alpha_factor'] / df['volatility']

    return df['alpha_factor']
```
