```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout_indicator = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Indicators
    aggregated_indicators = volume_weighted_breakout_indicator.rolling(window=10).sum()
    volume_weighted_breakout_ratios = price_breakout_ratio * df['volume']
    sum_volume_weighted_breakout_ratios = volume_weighted_breakout_ratios.rolling(window=10).sum()
    
    # Generate Alpha Factor
    alpha_factor = aggregated_indicators.ewm(span=5, adjust=False).mean()
    
    # Enhance Alpha Factor
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    alpha_factor = alpha_factor * daily_return
    
    # Adjust for Volatility
    volatility = df[['high', 'low', 'close']].std(axis=1).rolling(window=10).mean()
    alpha_factor = alpha_factor / volatility
    
    return alpha_factor
```
