```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].diff()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['smoothed_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / df['price_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']
    
    # Aggregate Indicators
    df['sum_volume_weighted_breakout'] = df['volume_weighted_breakout'].rolling(window=lookback_period).sum()
    df['weighted_breakout_ratio'] = (df['breakout_ratio'] * df['volume']).rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['smoothed_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_adjusted_momentum'] = df['volume_weighted_momentum'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['weighted_breakout_ratio']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Generate Trend Signal
    df['long_term_ma'] = df['close'].rolling(window=50).mean()
    df['short_term_ma'] = df['close'].rolling(window=20).mean()
    df['trend_signal'] = df.apply(lambda row: 1 if row['short_term_ma'] > row['long_term_ma'] else -1, axis=1)
    
    # Integrate Trend Signal into Alpha Factor
    df['trend_adjusted_momentum'] = df['combined_momentum'] * df['trend_signal']
    
    # Final Composite Factor Calculation
    df['final_factor'] = df['trend_adjusted_momentum']
    df['final_factor'] = df['final_factor'].ewm(alpha=alpha, adjust=False).mean()
    
    return df['final_factor']
```
