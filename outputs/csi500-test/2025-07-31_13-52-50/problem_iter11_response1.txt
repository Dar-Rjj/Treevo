```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_return'] = (df['volume'] - df['volume_ma_20']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['adjusted_intraday_return'] = df.apply(lambda row: row['intraday_return'] * 1.5 if row['close_std_10'] > 0 else row['intraday_return'] * 0.7, axis=1)
    
    # Calculate Daily Price Movement Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['volume_ma_14'] = df['volume'].rolling(window=14).mean()
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & (df['volume'] > df['volume_ma_14'])
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Volume-Weighted Breakout Ratios
    df['volume_weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['total_volume_weighted_breakout_ratio'] = df['volume_weighted_breakout_ratio'].cumsum()
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.2
    df['smoothed_breakout_ratio'] = df['volume_weighted_breakout_ratio'].ewm(alpha=smoothing_factor).mean()
    
    # Final Score Calculation
    df['final_score'] = df.apply(lambda row: 1 if row['trend_reversal'] and row['volume'] > row['volume_ma_14'] and row['smoothed_breakout_ratio'] > 0 and row['adjusted_intraday_return'] > 0 else 0, axis=1)
    
    return df['final_score']
```
