import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the relative range
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min())
    
    # Calculate the relative momentum
    df['momentum'] = df['close'] / df['close'].shift(1)
    df['relative_momentum'] = df['momentum'] / df['momentum'].rolling(window=20).mean()
    
    # Calculate the volume-amount ratio
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Combine the factors
    df['alpha_factor'] = (df['relative_range'] * df['relative_momentum'] * df['volume_amount_ratio']).ewm(span=20, adjust=False).mean()
    
    return df['alpha_factor']
