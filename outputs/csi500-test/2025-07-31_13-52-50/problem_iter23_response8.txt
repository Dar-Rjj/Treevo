```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=n, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout'].rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Generate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_returns'] * df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum_breakout'] = df['volume_weighted_momentum'] + df['aggregated_breakout']
    
    # Calculate Daily Price Movement Range
    df['daily_price_movement_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Price Momentum Factor
    n_days = 20
    df['price_momentum'] = df['daily_returns'].rolling(window=n_days).sum()
    
    # Compute Volume Momentum Factor
    m_days = 20
    df['volume_changes'] = df['volume'].pct_change()
    df['volume_momentum'] = df['volume_changes'].rolling(window=m_days).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_moving_average'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_average']
    df['volume_adjustment_factor'] = df['volume_deviation'] / (df['volume_deviation'].abs().mean() + 1e-6)
    df['adjusted_combined_momentum'] = df['combined_momentum'] / (df['volume_adjustment_factor'].abs() + 1e-6)
    
    # Enhance Alpha Factor
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    df['alpha_factor'] = (df['daily_returns'] * df['adjusted_combined_momentum']) / (df['volatility'] + 1e-6)
    
    # Final Composite Factor Calculation
    df['final_alpha_factor'] = df['combined_momentum_breakout'] + df['adjusted_combined_momentum'] + (df['trend_reversal_potential'] * 1.5)
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=n, adjust=False).mean()
    
    return df['final_alpha_factor']
```
