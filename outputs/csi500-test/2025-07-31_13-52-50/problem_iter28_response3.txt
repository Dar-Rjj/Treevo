```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return Deviation from Close
    df['daily_return_deviation'] = df['close'].diff()
    
    # Compare Daily Return Deviation with VWAP
    df['vwap_deviation'] = df['close'] - df['amount'] / df['volume']
    
    # Check for Volume Increase
    df['vol_14_day_ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['vol_14_day_ma']).astype(int)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['open']) / df['open']
    
    # Adjust Intraday Return for Volume
    df['vol_7_day_ma'] = df['volume'].rolling(window=7).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - df['vol_7_day_ma']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['close_std_7_day'] = df['close'].rolling(window=7).std()
    df['adjusted_intraday_return'] = df['volume_adjusted_intraday_return'] * 2.0
    df.loc[df['close_std_7_day'] < df['close_std_7_day'].median(), 'adjusted_intraday_return'] = df['volume_adjusted_intraday_return'] * 0.8
    
    # Compute Price Momentum Factor
    n_days = 10
    df['price_momentum'] = df['daily_return_deviation'].rolling(window=n_days).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.3
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_increase']
    
    # Adjust Combined Momentum for Volume Volatility
    df['vol_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['vol_20_day_ma']
    small_constant = 1e-6
    df['volume_adjustment_factor'] = df['volume_deviation'] + small_constant
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['momentum_shift'] = (df['sma_5'] > df['sma_20']).astype(int) - (df['sma_5'] < df['sma_20']).astype(int)
    df['final_momentum_score'] = df['adjusted_combined_momentum'] + df['momentum_shift']
    
    # Combine Indicators
    df['trend_reversal_signal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)).astype(int) * df['volume_increase']
    df['final_factor'] = (df['trend_reversal_signal'] * 2.0 + df['adjusted_intraday_return'] + df['final_momentum_score'])
    
    return df['final_factor']
```
