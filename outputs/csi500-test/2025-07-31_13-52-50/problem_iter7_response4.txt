```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_return_deviation'] = df['return_deviation'].shift(1)
    df['deviation_increase'] = df['return_deviation'] > df['prev_return_deviation']
    
    # Check for Volume Increase
    df['vol_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = df['volume'] > df['vol_ma_14']
    
    # Final Score Calculation
    df['reversal_score'] = 0
    df.loc[df['deviation_increase'] & df['volume_increase'], 'reversal_score'] = 1
    
    return df['reversal_score']
```
