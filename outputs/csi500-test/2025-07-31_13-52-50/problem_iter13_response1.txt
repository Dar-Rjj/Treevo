```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume-Weighted Breakout Indicator
    volume_weighted_breakout = (df['close'] - df['open']) * df['volume'] * price_breakout_ratio
    
    # Aggregate Breakout Indicators
    aggregated_breakout = volume_weighted_breakout.cumsum()
    
    # Apply Moving Average to Aggregated Indicators
    ma_window_size = 10
    ma_aggregated_breakout = aggregated_breakout.rolling(window=ma_window_size).mean()
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    returns_lookback_period = 20
    ma_daily_returns = daily_returns.rolling(window=returns_lookback_period).mean()
    
    # Synthesize Final Alpha Factor
    alpha_factor = ma_aggregated_breakout * ma_daily_returns
    
    return alpha_factor
```
