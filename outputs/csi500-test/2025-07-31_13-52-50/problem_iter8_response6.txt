```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / avg_volume
    
    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['momentum_factor'] = df['ema_daily_returns'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['aggregated_breakout'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    
    # Smooth Aggregated Breakout Indicators
    df['smoothed_aggregated_breakout'] = df['aggregated_breakout'].rolling(window=n).mean()
    
    # Combine Momentum and Breakout Indicators
    df['combined_indicator'] = df['momentum_factor'] + df['smoothed_aggregated_breakout']
    
    # Calculate Volume-Weighted Combined Indicator
    df['volume_weighted_combined_indicator'] = df['combined_indicator'] * df['volume_weight']
    
    # Calculate Daily Price Movement Range
    df['daily_price_movement_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap'] = vwap
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & \
                                     (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - avg_volume_10) * df['intraday_return']
    
    # Incorporate Price Volatility
    std_close_5 = df['close'].rolling(window=5).std()
    df['volatility_adjusted_intraday_return'] = np.where(
        std_close_5 > std_close_5.mean(),
        df['volume_adjusted_intraday_return'] * 1.2,
        df['volume_adjusted_intraday_return'] * 0.8
    )
    
    # Combine Reversal and Intraday Indicators
    df['reversal_intraday_indicator'] = np.where(
        df['trend_reversal_potential'],
        df['volatility_adjusted_intraday_return'],
        0
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_intraday_indicator'] * df['volume_weighted_combined_indicator']
    
    return df['final_alpha_factor']
```
