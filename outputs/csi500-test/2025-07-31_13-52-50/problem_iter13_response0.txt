import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['prev_deviation']) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Price Momentum Factor over the past 5 days
    df['price_momentum'] = df['daily_return'].rolling(window=5).sum()
    
    # Compute Volume Momentum Factor over the past 5 days
    df['volume_change'] = df['volume'].pct_change()
    df['volume_momentum'] = df['volume_change'].rolling(window=5).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.7
    k2 = 0.3
    df['combined_momentum'] = (k1 * df['price_momentum']) + (k2 * df['volume_momentum'])
    
    # Adjust Combined Momentum for Volume Volatility
    lookback_days = 20
    df['avg_volume'] = df['volume'].rolling(window=lookback_days).mean()
    df['volume_deviation'] = df['volume'] - df['avg_volume']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = df['adjusted_momentum'] + df['trend_reversal'].astype(int) * 0.5  # Assign higher score for stronger indication
    
    return df['composite_factor']
