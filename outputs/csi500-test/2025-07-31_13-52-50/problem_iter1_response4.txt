```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame, n: int = 20) -> pd.Series:
    # Calculate daily returns using close prices
    df['daily_return'] = df['close'].pct_change()
    
    # Compute moving average of daily returns over the lookback period
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()
    
    # Calculate volume weight
    avg_volume = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / avg_volume
    
    # Combine moving average and volume weight
    df['alpha_factor'] = df['ma_daily_return'] * df['volume_weight']
    
    return df['alpha_factor']
```
