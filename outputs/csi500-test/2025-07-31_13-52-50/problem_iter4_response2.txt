```python
import pandas as pd

def heuristics_v2(df):
    # Compute 12-day EMA of Close Price
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    
    # Compute 26-day EMA of Close Price
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate Daily Price Momentum
    price_momentum = ema_12 - ema_26
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate 14-day SMA of True Range
    tr_sma = true_range.rolling(window=14).mean()
    
    # Calculate 14-day Average Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=14).sum()
    
    # Adjust by Volume-Weighted Average True Range
    vol_weighted_tr = tr_sma * abs(avg_volume_change)
    
    # Final Alpha Factor
    alpha_factor = price_momentum / vol_weighted_tr
    
    return alpha_factor
```
