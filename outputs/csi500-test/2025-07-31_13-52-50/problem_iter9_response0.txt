import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['deviation_from_vwap'].shift(1)).astype(int)
    
    # Check for Volume Increase
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_increase'] = (df['volume'] > df['volume_ma_5']).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_ma_7'] = df['volume'].rolling(window=7).mean()
    df['volume_adjustment'] = (df['volume'] - df['volume_ma_7']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['std_close_7'] = df['close'].rolling(window=7).std()
    df['volatility_factor'] = df['std_close_7'].apply(lambda x: 1.4 if x > df['std_close_7'].quantile(0.75) else 0.6)
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volatility_factor']
    
    # Combine Indicators
    df['combined_score'] = (
        df['trend_reversal'] * df['volume_increase'] * 
        (df['volume_adjustment'] + df['adjusted_intraday_return'])
    )
    
    # Introduce a smoothing factor to the combined score
    df['smoothed_score'] = df['combined_score'].rolling(window=3).mean()
    
    # Use as a Factor for Stock Selection or Timing
    return df['smoothed_score'].dropna()
