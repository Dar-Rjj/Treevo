```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    n = 20
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()
    
    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Combine Moving Average and Volume Weight
    df['weighted_ma_daily_return'] = df['ma_daily_return'] * df['volume_weight']
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 20
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Momentum for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_avg']
    df['volume_adjustment_factor'] = df['volume_deviation'] / (df['volume_moving_avg'] + 1e-6)
    df['adjusted_combined_momentum'] = df['combined_momentum'] / (df['volume_adjustment_factor'] + 1e-6)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['vw_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregate_breakout'] = df['vw_breakout_indicator'].rolling(window=n).sum()
    
    # Smooth Aggregate Breakout Indicators
    alpha = 0.2
    df['smoothed_aggregate_breakout'] = df['aggregate_breakout'].ewm(alpha=alpha).mean()
    
    # Final Composite Factor Calculation
    df['composite_factor'] = np.where(df['trend_reversal'], 1, 0) * df['adjusted_combined_momentum'] * df['smoothed_aggregate_breakout']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['weighted_ma_daily_return'] + df['composite_factor']
    df['alpha_factor'] = df['alpha_factor'].rolling(window=n).mean()
    
    # Enhance Alpha Factor
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=n).mean()
    df['alpha_factor'] = df['alpha_factor'] * df['daily_return'] / df['volatility']
    
    return df['alpha_factor']
```
