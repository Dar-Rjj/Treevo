```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 10-day moving average of the typical price
    typical_price_ma_10 = typical_price.rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the volume
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the 10-day moving average of the amount
    amount_ma_10 = df['amount'].rolling(window=10).mean()
    
    # Calculate the 5-day rolling standard deviation of the typical price
    typical_price_std_5 = typical_price.rolling(window=5).std()
    
    # Calculate the alpha factor
    alpha_factor = (typical_price - typical_price_ma_10) / typical_price_std_5 * (volume_ma_5 / amount_ma_10)
    
    return alpha_factor
```
