```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20  # Lookback period
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()

    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']

    # Combine Moving Average and Volume Weight
    df['vol_weighted_ma_return'] = df['ma_daily_return'] * df['volume_weight']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vol_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['agg_vol_weighted_breakout'] = df['vol_weighted_breakout'].rolling(window=n).sum()
    df['agg_price_breakout_ratio'] = (df['price_breakout_ratio'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Generate Alpha Factor
    window_size = 10  # Define window size for SMA
    df['alpha_factor'] = (df['vol_weighted_ma_return'] + df['agg_vol_weighted_breakout'] + df['agg_price_breakout_ratio']).rolling(window=window_size).mean()

    # Smooth with Exponential Moving Average
    smoothing_factor = 0.5  # Define smoothing factor
    df['ema_alpha_factor'] = df['alpha_factor'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Enhance Alpha Factor
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=n).std().mean(axis=1)
    df['enhanced_alpha_factor'] = df['ema_alpha_factor'] * df['daily_return'] / df['volatility']

    # Introduce Trend Following Component
    m = 50  # Longer lookback period
    n = 20  # Shorter lookback period
    df['long_term_ma'] = df['close'].rolling(window=m).mean()
    df['short_term_ma'] = df['close'].rolling(window=n).mean()

    # Generate Trend Signal
    df['trend_signal'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Integrate Trend Signal into Alpha Factor
    df['final_alpha_factor'] = df['enhanced_alpha_factor'] * df['trend_signal']

    return df['final_alpha_factor']
```
