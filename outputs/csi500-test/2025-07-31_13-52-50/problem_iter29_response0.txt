```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor to help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day moving average to the 30-day moving average
    df['ma_ratio'] = df['ma_10_close'] / df['ma_30_close']
    
    # Calculate the difference between the high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the 10-day moving average of the high-low difference
    df['ma_10_high_low_diff'] = df['high_low_diff'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the high-low difference
    df['ma_30_high_low_diff'] = df['high_low_diff'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day moving average of the high-low difference to the 30-day moving average of the high-low difference
    df['ma_ratio_high_low_diff'] = df['ma_10_high_low_diff'] / df['ma_30_high_low_diff']
    
    # Calculate the 10-day rolling standard deviation of the closing price
    df['std_10_close'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day rolling standard deviation of the closing price
    df['std_30_close'] = df['close'].rolling(window=30).std()
    
    # Calculate the ratio of the 10-day rolling standard deviation to the 30-day rolling standard deviation
    df['std_ratio'] = df['std_10_close'] / df['std_30_close']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['ma_ratio'] + df['ma_ratio_high_low_diff'] + df['std_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
