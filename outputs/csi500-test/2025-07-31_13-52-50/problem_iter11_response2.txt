import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Momentum Score
    lookback_days = 20
    df['close_change'] = df['close'] - df['close'].shift(lookback_days)
    df['momentum_score'] = df['close_change'].rolling(window=lookback_days).sum()
    
    # Adjust Momentum for Volume Volatility
    volume_ma = df['volume'].rolling(window=lookback_days).mean()
    volume_deviation = df['volume'] - volume_ma
    volume_adjustment_factor = volume_deviation + 1e-6  # Add a small constant to avoid division by zero
    df['adjusted_momentum_score'] = df['momentum_score'] / volume_adjustment_factor
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_return'] = (df['volume'] - volume_ma_20) * df['intraday_return']
    
    # Incorporate Price Volatility
    close_std = df['close'].rolling(window=10).std()
    high_volatility = close_std > close_std.mean()
    df['adjusted_intraday_return'] = df['intraday_return'].where(~high_volatility, df['intraday_return'] * 1.5)
    df['adjusted_intraday_return'] = df['adjusted_intraday_return'].where(high_volatility, df['intraday_return'] * 0.7)
    
    # Final Score Calculation
    df['final_score'] = 0.0
    df.loc[df['trend_reversal'], 'final_score'] += 1.0
    df['final_score'] += df['adjusted_momentum_score'] * 0.5
    df['final_score'] += df['adjusted_intraday_return'] * 0.5
    
    return df['final_score']
