```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Price Momentum Factor
    n = 5  # Number of days for price momentum
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 5  # Number of days for volume momentum
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5  # Weight for price momentum
    k2 = 0.5  # Weight for volume momentum
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Momentum for Volume Volatility
    lookback_days = 20
    df['volume_ma'] = df['volume'].rolling(window=lookback_days).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6  # Small constant to avoid division by zero
    df['adjusted_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['weighted_breakout'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    df['aggregated_breakout'] = df['weighted_breakout'].rolling(window=5).sum()
    
    # Smooth Aggregate Breakout Indicators
    smoothing_factor = 0.2  # Example smoothing factor
    df['smoothed_breakout'] = df['aggregated_breakout'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Final Composite Factor Calculation
    df['composite_factor'] = 0
    df.loc[df['trend_reversal'], 'composite_factor'] += 2  # Higher score for trend reversal
    df['composite_factor'] += df['adjusted_momentum'] + df['smoothed_breakout']
    
    return df['composite_factor']
```
