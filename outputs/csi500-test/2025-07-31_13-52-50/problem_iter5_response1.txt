```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation_from_vwap'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['prev_deviation_from_vwap']) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Momentum Score
    lookback_days = 20
    df['close_change'] = df['close'] - df['close'].shift(lookback_days)
    df['momentum_score'] = df['close_change'].rolling(window=lookback_days).sum()
    
    # Adjust Momentum for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=lookback_days).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_avg']
    df['volume_adjustment_factor'] = (df['volume_deviation'] + 1e-6)  # Adding a small constant to avoid division by zero
    df['adjusted_momentum_score'] = df['momentum_score'] / df['volume_adjustment_factor']
    
    # Final Score Calculation
    df['final_score'] = df.apply(lambda row: row['adjusted_momentum_score'] * 1.5 if row['trend_reversal'] else row['adjusted_momentum_score'], axis=1)
    
    return df['final_score']
```
