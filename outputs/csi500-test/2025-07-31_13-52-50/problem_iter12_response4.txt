```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_returns'] = df['daily_returns'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    
    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['momentum_factor'] = df['ema_daily_returns'] * df['volume_weight']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['weighted_breakout'] = df['price_breakout_ratio'] * df['volume']
    df['aggregated_breakout'] = df['weighted_breakout'].rolling(window=n).sum()
    
    # Smooth Aggregated Breakout Indicators
    smoothing_window = 5
    df['smoothed_breakout'] = df['aggregated_breakout'].rolling(window=smoothing_window).mean()
    
    # Combine Momentum and Breakout Indicators
    df['combined_indicator'] = df['momentum_factor'] + df['smoothed_breakout']
    
    # Calculate Volume-Weighted Combined Indicator
    df['volume_weighted_combined'] = df['combined_indicator'] * df['volume_weight']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['prev_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal'] = (df['deviation_from_vwap'] > df['prev_deviation'])
    
    # Check for Volume Increase
    df['vol_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['vol_ma_14'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    df['vol_adjustment'] = (df['volume'] - df['vol_ma_10']) * df['intraday_return']
    
    # Incorporate Price Volatility
    df['std_close_5'] = df['close'].rolling(window=5).std()
    df['intraday_return_adjusted'] = np.where(df['std_close_5'] > df['std_close_5'].mean(), df['intraday_return'] * 1.2, df['intraday_return'] * 0.8)
    
    # Integrate Reversal and Intraday Indicators
    df['reversal_intraday'] = np.where(df['trend_reversal'] & df['volume_increase'], df['intraday_return_adjusted'], 0)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['reversal_intraday'] * df['volume_weighted_combined']
    
    return df['final_alpha_factor']
```
