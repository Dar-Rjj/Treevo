```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout'].rolling(window=n).sum()
    
    # Generate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum'] /= df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_factor'] = df['volume_weighted_momentum'] + df['aggregated_breakout']
    
    # Calculate Daily Price Movement Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = (df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume'].pct_change().rolling(window=14).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.5
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_avg']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 0.0001
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Enhance Alpha Factor
    df['alpha_factor'] = df['daily_return'] * df['adjusted_combined_momentum']
    
    # Adjust for Volatility
    window_size = 20
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=window_size).std(ddof=0).mean(axis=1)
    df['alpha_factor'] = df['alpha_factor'] / df['volatility']
    
    # Final Composite Factor Calculation
    df['final_alpha_factor'] = df['combined_factor'].ewm(span=n, adjust=False).mean()
    df.loc[df['trend_reversal_potential'], 'final_alpha_factor'] *= 1.2
    
    return df['final_alpha_factor']
```
