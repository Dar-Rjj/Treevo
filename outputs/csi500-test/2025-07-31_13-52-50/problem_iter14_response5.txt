```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Price Momentum Factor
    m = 20
    df['price_momentum'] = df['daily_return'].rolling(window=m).sum()

    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Determine Daily Return Deviation from VWAP
    df['return_deviation_vwap'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['trend_reversal'] = 0
    df.loc[(df['return_deviation_vwap'] > df['return_deviation_vwap'].shift(1)) & 
           (df['volume'] > df['volume'].rolling(window=14).mean()), 'trend_reversal'] = 1

    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']

    # Aggregate Breakout Indicators
    df['weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['aggregate_breakout_indicators'] = df['weighted_breakout_ratio'].rolling(window=m).sum()

    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']

    # Adjust Combined Momentum for Volume Volatility
    df['volume_moving_avg'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_avg']
    df['volume_adjustment_factor'] = (df['volume_deviation'] + 1e-6)  # Small constant to avoid division by zero
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']

    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    volume_weight = df['volume'] / df['volume'].rolling(window=n).mean()
    df['adjusted_ema_daily_return'] = df['ema_daily_return'] * volume_weight + df['adjusted_combined_momentum']

    # Final Composite Factor Calculation
    df['final_composite_factor'] = (
        df['trend_reversal'] * 2 +  # Higher score for trend reversal
        df['adjusted_ema_daily_return'] +
        df['aggregate_breakout_indicators'].ewm(span=m, adjust=False).mean()
    )

    return df['final_composite_factor']
```
