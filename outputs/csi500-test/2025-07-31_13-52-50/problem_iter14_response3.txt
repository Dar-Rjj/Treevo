```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['smoothed_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 20
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['aggregate_breakout'] = df['volume_weighted_breakout'].rolling(window=n).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)
    
    # Check for Volume Increase
    df['vol_ma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = df['volume'] > df['vol_ma_14']
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Calculate Volume Weight
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume_20']
    
    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_smoothed_return'] = df['smoothed_daily_return'] * df['volume_weight'] + df['combined_momentum']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['adjusted_smoothed_return'] + df['aggregate_breakout'] + df['return_deviation_from_vwap']
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.1
    df['smoothed_alpha_factor'] = df['alpha_factor'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Enhance Alpha Factor
    df['enhanced_alpha_factor'] = df['smoothed_alpha_factor'] * df['daily_return']
    
    # Calculate Volatility
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)
    
    # Normalize Alpha Factor by Volatility
    df['normalized_alpha_factor'] = df['enhanced_alpha_factor'] / df['volatility']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = (
        df['trend_reversal_potential'] & df['volume_increase']).astype(int) * df['combined_momentum'] + \
        df['aggregate_breakout'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return df['composite_factor']
```
