import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day Close Price Average
    df['14_day_close_avg'] = df['close'].rolling(window=14).mean()
    
    # Compute Price Momentum
    df['price_momentum'] = df['close'] - df['14_day_close_avg']
    
    # Adjust for Volume Impact
    df['current_volume'] = df['volume']
    df['14_day_volume_avg'] = df['volume'].rolling(window=14).mean()
    df['volume_change_ratio'] = df['current_volume'] / df['14_day_volume_avg']
    df['price_momentum_vol_adj'] = df['price_momentum'] * df['volume_change_ratio']
    
    # Adjust for Amount Impact
    df['current_amount'] = df['amount']
    df['14_day_amount_avg'] = df['amount'].rolling(window=14).mean()
    df['amount_change_ratio'] = df['current_amount'] / df['14_day_amount_avg']
    df['price_momentum_vol_amt_adj'] = df['price_momentum_vol_adj'] * df['amount_change_ratio']
    
    # Return the final factor as a Series
    return df['price_momentum_vol_amt_adj'].dropna()
