```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Set lookback period and smoothing factor
    n = 20
    alpha = 0.1

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    df['ema_daily_return'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()

    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']

    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['factor'] = df['ema_daily_return'] * df['volume_weight']

    return df['factor']
```
