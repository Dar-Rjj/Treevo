```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    price_breakout_ratio = (df['high'] - df['open']) / high_low_range
    
    # Calculate Volume Breakout Indicator
    volume_breakout_indicator = df['volume'] * (df['close'] - df['open'])
    
    # Weight Price Breakout Ratio by Volume
    volume_weighted_breakout = price_breakout_ratio * volume_breakout_indicator
    
    # Aggregate Breakout Indicators by summing the volume-weighted breakout ratios
    aggregate_breakout = volume_weighted_breakout.rolling(window=10).sum()
    
    # Smooth with Moving Average
    window_size = 20
    final_alpha_factor = aggregate_breakout.rolling(window=window_size).mean()
    
    return final_alpha_factor
```
