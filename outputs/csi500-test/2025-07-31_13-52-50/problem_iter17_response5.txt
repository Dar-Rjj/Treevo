```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Price Breakout Ratio
    df['breakout_ratio'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    alpha = 0.1
    df['ema_daily_returns'] = df['daily_returns'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['breakout_ratio']
    
    # Aggregate Volume-Weighted Breakout Indicators
    df['aggregated_breakout'] = df['volume_weighted_breakout'].cumsum()
    
    # Calculate Volume-Weighted Momentum
    lookback_period = 20
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weighted_momentum'] = df['ema_daily_returns'] * df['volume'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Indicators
    df['combined_indicators'] = df['volume_weighted_momentum'] + df['aggregated_breakout']
    
    # Smooth with Exponential Moving Average
    smoothing_factor = 0.1
    df['smoothed_combined_indicators'] = df['combined_indicators'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Adjust Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] / (df['volume_ma'] + 1e-6)
    df['adjusted_momentum'] = df['smoothed_combined_indicators'] / (1 + df['volume_adjustment_factor'])
    
    # Incorporate Price Volatility
    df['close_std'] = df['close'].rolling(window=10).std()
    df['volatility_adjustment'] = np.where(df['close_std'] > 0.5, 1.5, 0.7)
    df['final_alpha_factor'] = df['adjusted_momentum'] * df['volatility_adjustment']
    
    # Final Score Calculation
    df['final_score'] = np.where(df['trend_reversal'], 1.5 * df['final_alpha_factor'], df['final_alpha_factor'])
    
    # Apply Moving Average to Final Alpha Factor
    final_window_size = 5
    df['final_alpha_factor_smoothed'] = df['final_score'].rolling(window=final_window_size).mean()
    
    return df['final_alpha_factor_smoothed']
```
