```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Daily Returns
    n = 20
    df['moving_avg_returns'] = df['daily_return'].rolling(window=n).mean()

    # Calculate Volume Weight
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n).mean()

    # Combine Moving Average and Volume Weight
    df['vol_weighted_ma'] = df['moving_avg_returns'] * df['volume_weight']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vol_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Aggregate Indicators
    df['aggregated_breakout'] = df['vol_weighted_breakout'].rolling(window=n).sum()

    # Generate Alpha Factor
    df['alpha_factor'] = df['vol_weighted_ma'] + df['aggregated_breakout']
    alpha_factor_window = 10
    df['alpha_factor'] = df['alpha_factor'].rolling(window=alpha_factor_window).mean()

    # Smooth with Exponential Moving Average
    alpha_ema_span = 5
    df['alpha_factor'] = df['alpha_factor'].ewm(span=alpha_ema_span).mean()

    # Enhance Alpha Factor
    df['alpha_factor'] = df['alpha_factor'] * df['daily_return']

    # Calculate Volatility
    df['tr'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    volatility_window = 20
    df['volatility'] = df['tr'].rolling(window=volatility_window).std()
    df['alpha_factor'] = df['alpha_factor'] / df['volatility']

    # Introduce Trend Following Component
    m = 50
    n = 20
    df['long_term_ma'] = df['close'].rolling(window=m).mean()
    df['short_term_ma'] = df['close'].rolling(window=n).mean()
    df['trend_signal'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)

    # Integrate Trend Signal into Alpha Factor
    df['alpha_factor'] = df['alpha_factor'] * df['trend_signal']

    # Introduce Mean Reversion Component
    p = 20
    q = 20
    df['20_day_high'] = df['high'].rolling(window=p).max()
    df['20_day_low'] = df['low'].rolling(window=q).min()

    # Generate Mean Reversion Signal
    mean_reversion_threshold = 0.05
    df['mean_reversion_signal'] = np.where(df['close'] >= df['20_day_high'] * (1 - mean_reversion_threshold), -1,
                                           np.where(df['close'] <= df['20_day_low'] * (1 + mean_reversion_threshold), 1, 0))

    # Integrate Mean Reversion Signal into Alpha Factor
    df['alpha_factor'] = df['alpha_factor'] * df['mean_reversion_signal']

    return df['alpha_factor']
```
