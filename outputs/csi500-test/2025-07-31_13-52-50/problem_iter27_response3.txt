```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Determine Daily Return Deviation from Close
    df['daily_return'] = df['close'].diff()
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compare Daily Return Deviation with VWAP
    df['deviation'] = df['close'] - df['vwap']
    
    # Check for Volume Increase
    df['vol_14ma'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = df['volume'] > df['vol_14ma']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['open']) / df['open']
    
    # Adjust Intraday Return for Volume
    df['vol_7ma'] = df['volume'].rolling(window=7).mean()
    df['vol_diff'] = df['volume'] - df['vol_7ma']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['vol_diff']
    
    # Incorporate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=7).std()
    df['adjusted_intraday_return'] = df['adjusted_intraday_return'] * df['price_volatility'].apply(lambda x: 2.5 if x > df['price_volatility'].median() else 0.7)
    
    # Compute Price Momentum Factor
    n = 10
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    m = 10
    df['volume_change'] = df['volume'].diff()
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['vol_20ma'] = df['volume'].rolling(window=20).mean()
    df['vol_deviation'] = df['volume'] - df['vol_20ma']
    df['vol_adjustment_factor'] = df['vol_deviation'] + 1e-6
    df['adjusted_momentum'] = df['combined_momentum'] / df['vol_adjustment_factor']
    
    # Incorporate Momentum Shift
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_cross'] = (df['sma_5'] > df['sma_20']) & (df['sma_5'].shift(1) <= df['sma_20'].shift(1))
    df['sma_cross_below'] = (df['sma_5'] < df['sma_20']) & (df['sma_5'].shift(1) >= df['sma_20'].shift(1))
    df['adjusted_momentum'] = df['adjusted_momentum'] + df['sma_cross'] * 1.5 - df['sma_cross_below'] * 1.5
    
    # Introduce Price and Volume Correlation
    df['corr_10'] = df['close'].rolling(window=10).corr(df['volume'])
    df['adjusted_momentum'] = df['adjusted_momentum'] * df['corr_10'].apply(lambda x: 1.2 if x > 0 else 0.8)
    
    # Combine Indicators
    df['trend_reversal'] = (df['deviation'] > df['deviation'].shift(1)) & df['volume_increase']
    df['composite_factor'] = (
        0.3 * df['trend_reversal'].astype(int) +
        0.4 * df['adjusted_intraday_return'] +
        0.3 * df['adjusted_momentum']
    )
    
    return df['composite_factor']
```
