import pandas as pd

def heuristics_v2(df, lookback_period=20, ma_window=10):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Moving Average of Daily Returns
    df['ma_returns'] = df['daily_return'].rolling(window=lookback_period).mean()
    
    # Calculate Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    
    # Combine Moving Average and Volume Weight
    df['weighted_ma_returns'] = df['ma_returns'] * df['volume_weight']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['aggregate_breakout'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    df['volume_diff'] = df['volume'] - df['volume'].rolling(window=ma_window).mean()
    df['volume_adjusted_intraday_return'] = df['volume_diff'] * df['intraday_return']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['weighted_ma_returns'] + df['aggregate_breakout'] + df['volume_adjusted_intraday_return']
    df['final_alpha_factor'] = df['alpha_factor'].rolling(window=ma_window).mean()
    
    return df['final_alpha_factor'].dropna()
