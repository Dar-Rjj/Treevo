```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, k1=0.5, k2=0.5, smoothing_factor=0.2):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Compute Price Momentum Factor
    df['price_momentum'] = df['daily_return'].rolling(window=n).sum()
    
    # Compute Volume Momentum Factor
    df['volume_momentum'] = df['volume_change'].rolling(window=m).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['volume_weighted_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['aggregate_breakout_indicators'] = df['volume_weighted_breakout_ratio'].rolling(window=5).sum()
    
    # Determine Daily Return Deviation from VWAP
    df['deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['previous_deviation'] = df['deviation_from_vwap'].shift(1)
    df['trend_reversal_potential'] = (df['deviation_from_vwap'] > df['previous_deviation']).astype(int)
    
    # Check for Volume Increase
    df['volume_moving_average'] = df['volume'].rolling(window=14).mean()
    df['volume_increase'] = (df['volume'] > df['volume_moving_average']).astype(int)
    
    # Combine Price and Volume Momentum
    df['combined_momentum'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Final Composite Factor Calculation
    df['composite_factor'] = 0
    df.loc[(df['trend_reversal_potential'] == 1) & (df['volume_increase'] == 1), 'composite_factor'] += 10
    df['composite_factor'] += df['combined_momentum']
    df['composite_factor'] += df['aggregate_breakout_indicators'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return df['composite_factor']
```
