import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['Smoothed_Return'] = df['Daily_Return'].ewm(alpha=alpha, adjust=False).mean()

    # Determine Daily Price Movement Range
    df['Price_Range'] = df['high'] - df['low']

    # Identify Trend Reversal Potential
    df['Return_Deviation'] = df['Daily_Return'].abs().diff()
    df['Trend_Reversal'] = (df['Return_Deviation'] > df['Return_Deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Compute Price Momentum Factor
    n = 10
    df['Price_Momentum'] = df['Daily_Return'].rolling(window=n).sum()

    # Compute Volume Momentum Factor
    m = 10
    df['Volume_Change'] = df['volume'].diff()
    df['Volume_Momentum'] = df['Volume_Change'].rolling(window=m).sum()

    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['Combined_Momentum'] = k1 * df['Price_Momentum'] + k2 * df['Volume_Momentum']

    # Adjust Combined Momentum for Volume Volatility
    df['Volume_MA_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Volatility'] = df['volume'] - df['Volume_MA_20']
    df['Volume_Adjustment_Factor'] = df['Volume_Volatility'] + 0.0001
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] / df['Volume_Adjustment_Factor']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['Price_Breakout_Ratio'] = (df['high'] - df['open']) / df['High_Low_Range']

    # Calculate Volume Breakout Indicator
    df['Volume_Breakout_Indicator'] = df['volume'] * (df['close'] - df['open'])

    # Aggregate Breakout Indicators
    df['Weighted_Price_Breakout_Ratio'] = df['Price_Breakout_Ratio'] * df['volume']
    df['Aggregated_Breakout_Indicators'] = df['Weighted_Price_Breakout_Ratio'].rolling(window=10).sum()

    # Calculate Volume-Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['Smoothed_Return'] * df['volume']
    df['Average_Volume_Over_Lookback'] = df['volume'].rolling(window=lookback_period).mean()

    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['Combined_Breakout_and_Momentum'] = df['Volume_Weighted_Momentum'] + df['Aggregated_Breakout_Indicators']

    df['Price_Change'] = df['close'] - df['open']
    df['Price_Action_Sign'] = np.sign(df['Price_Change'])
    df['Adjusted_Aggregated_Breakout_Indicators'] = df['Aggregated_Breakout_Indicators'] * df['Price_Action_Sign']
