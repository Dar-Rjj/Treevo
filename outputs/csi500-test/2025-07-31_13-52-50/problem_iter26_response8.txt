```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume'.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day simple moving average of the close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day simple moving average of the close price
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day standard deviation of the close price
    df['std_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the close price
    df['std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the 10-day average volume
    df['avg_vol_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day average volume
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the 10-day price range
    df['price_range_10'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    
    # Calculate the 30-day price range
    df['price_range_30'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    
    # Calculate the 10-day average amount
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    
    # Calculate the 30-day average amount
    df['avg_amount_30'] = df['amount'].rolling(window=30).mean()
    
    # Calculate the factor
    df['factor'] = (df['sma_10'] / df['sma_30']) * (df['avg_vol_10'] / df['avg_vol_30']) * \
                   (df['price_range_10'] / df['price_range_30']) * (df['avg_amount_10'] / df['avg_amount_30'])
    
    # Return the factor as a pandas Series
    return df['factor']
```
