```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(periods=20)
    
    # Measure Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Analyze Volume Trends
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_change_20d = df['volume'].rolling(window=20).mean().pct_change(periods=20)
    
    # Combine Momentum and Volume Indicators
    combined_momentum_volume = short_term_momentum * avg_volume_5d
    
    # Assess High-Low Range Dynamics
    avg_high_5d = df['high'].rolling(window=5).mean()
    avg_low_5d = df['low'].rolling(window=5).mean()
    high_low_ratio = avg_high_5d / avg_low_5d
    
    # Evaluate Open-Close Price Behavior
    avg_open_5d = df['open'].rolling(window=5).mean()
    avg_close_5d = df['close'].rolling(window=5).mean()
    open_close_diff = avg_open_5d - avg_close_5d
    open_greater_than_close = (avg_open_5d > avg_close_5d).astype(int)
    
    # Analyze Volume and Price Correlation
    price_change_5d = df['close'].pct_change(periods=5)
    volume_change_5d = df['volume'].pct_change(periods=5)
    price_volume_corr = price_change_5d.rolling(window=5).corr(volume_change_5d)
    
    # Identify Reversal Patterns
    reversal_up = (short_term_momentum < 0) & (long_term_momentum > 0)
    reversal_down = (short_term_momentum > 0) & (long_term_momentum < 0)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        combined_momentum_volume +
        high_low_ratio +
        open_close_diff +
        open_greater_than_close +
        price_volume_corr +
        reversal_up.astype(int) -
        reversal_down.astype(int)
    )
    
    return alpha_factor
```
