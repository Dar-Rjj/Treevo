```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Score
    lookback_days = 20
    close_price_change = df['close'] - df['close'].shift(lookback_days)
    momentum_score = close_price_change.rolling(window=lookback_days).sum()

    # Adjust for Volume Volatility
    volume_ma = df['volume'].rolling(window=lookback_days).mean()
    volume_deviation = df['volume'] - volume_ma
    volume_adjustment_factor = (volume_deviation + 1e-6)  # Add a small constant to avoid division by zero
    adjusted_momentum_score = momentum_score / volume_adjustment_factor

    return adjusted_momentum_score
```
