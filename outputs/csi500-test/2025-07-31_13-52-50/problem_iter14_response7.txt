```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume Breakout Indicator
    df['volume_breakout_indicator'] = (df['close'] - df['open']) * df['volume']
    
    # Aggregate Breakout Indicators
    df['weighted_price_breakout_ratio'] = df['price_breakout_ratio'] * df['volume_breakout_indicator']
    df['aggregated_breakout_ratio'] = df['weighted_price_breakout_ratio'].sum()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['ema_daily_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_momentum_normalized'] = df['volume_weighted_momentum'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum_breakout'] = df['volume_weighted_momentum_normalized'] + df['aggregated_breakout_ratio']
    
    # Calculate Daily Price Movement Range
    df['daily_price_movement_range'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['daily_return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = df['daily_return_deviation'].diff() > 0
    df['volume_increase'] = df['volume'] > df['volume'].rolling(window=14).mean()
    
    # Compute Price Momentum Factor
    price_momentum_window = n
    df['price_momentum_factor'] = df['daily_return'].rolling(window=price_momentum_window).sum()
    
    # Compute Volume Momentum Factor
    volume_momentum_window = n
    df['volume_momentum_factor'] = df['volume'].pct_change().rolling(window=volume_momentum_window).sum()
    
    # Combine Price and Volume Momentum
    k1 = 0.5
    k2 = 0.5
    df['combined_momentum'] = k1 * df['price_momentum_factor'] + k2 * df['volume_momentum_factor']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_moving_average'] = df['volume'].rolling(window=n).mean()
    df['volume_deviation'] = df['volume'] - df['volume_moving_average']
    df['volume_adjustment_factor'] = (df['volume_deviation'] + 1e-6)  # Add a small constant to avoid division by zero
    df['adjusted_combined_momentum'] = df['combined_momentum'] / df['volume_adjustment_factor']
    
    # Final Composite Factor Calculation
    df['trend_reversal_score'] = (df['trend_reversal_potential'] & df['volume_increase']).astype(int) * 10
    df['final_composite_factor'] = df['combined_momentum_breakout'] + df['adjusted_combined_momentum'] + df['trend_reversal_score']
    
    # Apply Exponential Moving Average for Smoothing
    final_smoothing_factor = 0.1
    df['final_alpha_factor'] = df['final_composite_factor'].ewm(alpha=final_smoothing_factor, adjust=False).mean()
    
    return df['final_alpha_factor']
```
