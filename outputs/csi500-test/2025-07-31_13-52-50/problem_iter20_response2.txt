```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].diff()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['smoothed_returns'] = df['daily_returns'].ewm(span=lookback_period, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['sum_volume_weighted_breakout'] = df['volume_weighted_breakout'].rolling(window=lookback_period).sum()
    df['weighted_price_breakout_ratio'] = df['price_breakout_ratio'] * df['volume']
    df['sum_weighted_breakout_ratios'] = df['weighted_price_breakout_ratio'].rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Momentum
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    df['volume_weighted_momentum'] = df['smoothed_returns'] * df['volume'] / avg_volume
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['sum_weighted_breakout_ratios']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal'] = df['return_deviation'].diff() > 0
    df['volume_increase'] = df['volume'] > df['volume'].rolling(window=14).mean()
    df['reversal_score'] = (df['trend_reversal'] & df['volume_increase']).astype(int)
    
    # Compute Price Momentum Factor
    n_days = 20
    df['price_momentum'] = df['daily_returns'].rolling(window=n_days).sum()
    
    # Compute Volume Momentum Factor
    m_days = 20
    df['volume_change'] = df['volume'].diff()
    df['volume_momentum'] = df['volume_change'].rolling(window=m_days).sum()
    
    # Combine Price and Volume Momentum
    k1 = 1.0
    k2 = 1.0
    df['combined_momentum_score'] = k1 * df['price_momentum'] + k2 * df['volume_momentum']
    
    # Adjust Combined Momentum for Volume Volatility
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_deviation'] = df['volume'] - df['volume_ma']
    df['volume_adjustment_factor'] = df['volume_deviation'] + 1e-6
    df['adjusted_combined_momentum'] = df['combined_momentum_score'] / df['volume_adjustment_factor']
    
    # Introduce Trend Following Component
    long_lookback = 50
    short_lookback = 20
    df['long_ma'] = df['close'].rolling(window=long_lookback).mean()
    df['short_ma'] = df['close'].rolling(window=short_lookback).mean()
    
    # Generate Trend Signal
    df['trend_signal'] = (df['short_ma'] > df['long_ma']).astype(int) * 2 - 1
    
    # Integrate Trend Signal into Alpha Factor
    df['alpha_factor'] = df['adjusted_combined_momentum'] * df['trend_signal']
    
    # Final Composite Factor Calculation
    df['final_alpha_factor'] = df['combined_momentum'] + df['alpha_factor'] * df['reversal_score']
    
    # Apply Exponential Moving Average for Smoothing
    final_smoothing_factor = 0.1
    df['final_smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=final_smoothing_factor, adjust=False).mean()
    
    return df['final_smoothed_alpha_factor']
```
