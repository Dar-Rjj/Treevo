```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    lookback_period = 20
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=lookback_period, adjust=False).mean()
    
    # Determine Daily Price Movement Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Volume-Weighted Breakout Indicator
    df['vw_breakout_indicator'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']
    
    # Aggregate Indicators
    df['sum_vw_breakout'] = df['vw_breakout_indicator'].rolling(window=lookback_period).sum()
    df['weight_price_breakout_by_volume'] = df['price_breakout_ratio'] * df['volume']
    df['sum_weighted_breakout'] = df['weight_price_breakout_by_volume'].rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Momentum
    df['vw_momentum'] = df['ema_daily_return'] * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['vw_momentum_score'] = df['vw_momentum'] / df['avg_volume']
    
    # Combine Volume-Weighted Momentum and Aggregated Breakout Ratios
    df['combined_momentum'] = df['vw_momentum_score'] + df['sum_vw_breakout']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine Daily Return Deviation from VWAP
    df['return_deviation_from_vwap'] = df['close'] - df['vwap']
    
    # Identify Trend Reversal Potential
    df['trend_reversal_potential'] = df['return_deviation_from_vwap'] > df['return_deviation_from_vwap'].shift(1)
    
    # Check for Volume Increase
    df['volume_increase'] = df['volume'] > df['volume'].rolling(window=14).mean()
    df['reversal_signal'] = df['trend_reversal_potential'] & df['volume_increase']
    
    # Generate Trend Signal
    df['long_term_ma'] = df['close'].rolling(window=50).mean()
    df['short_term_ma'] = df['close'].rolling(window=20).mean()
    df['trend_signal'] = np.where(df['short_term_ma'] > df['long_term_ma'], 1, -1)
    
    # Integrate Trend Signal into Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['trend_signal']
    
    # Apply Exponential Moving Average to Final Composite Factor
    ema_alpha_factor = df['alpha_factor'].ewm(span=lookback_period, adjust=False).mean()
    
    return ema_alpha_factor
```
