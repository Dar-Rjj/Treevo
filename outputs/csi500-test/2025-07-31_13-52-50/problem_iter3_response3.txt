```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Smoothing Intraday Momentum
    # Rolling Mean of Intraday Returns
    intraday_momentum = df['intraday_return'].rolling(window=5).mean()
    
    # Apply Exponential Decay
    decay_factor = 0.9
    intraday_momentum_decay = intraday_momentum.ewm(alpha=decay_factor, adjust=False).mean()
    
    # Evaluate Liquidity
    # Calculate Average Volume
    average_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate Price Impact
    dollar_volume = df['volume'] * df['close']
    price_change = df['close'].diff().abs()
    price_impact = price_change / dollar_volume
    
    # Combine Liquidity and Price Impact
    liquidity = average_volume * price_impact
    liquidity_scaled = liquidity * 1e6  # Scale by a constant for better interpretation
    
    # Evaluate Volatility
    # Compute Historical Volatility
    daily_returns = df['close'].pct_change()
    historical_volatility = daily_returns.rolling(window=20).std()
    
    # Smoothing Volatility
    volatility_decay = historical_volatility.ewm(alpha=decay_factor, adjust=False).mean()
    
    # Combine Intraday Momentum, Liquidity, and Volatility
    alpha_factor = (intraday_momentum_decay * liquidity_scaled) / volatility_decay
    
    # Generate Alpha Factor
    # Ensure Non-Negative
    alpha_factor = alpha_factor.clip(lower=0)
    
    # Apply Threshold for Stability
    threshold = 1e-3
    alpha_factor = alpha_factor.where(alpha_factor > threshold, 0)
    
    return alpha_factor
```
