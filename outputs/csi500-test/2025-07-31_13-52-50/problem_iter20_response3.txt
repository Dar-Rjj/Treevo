```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 22
    alpha = 0.1
    df['ema_daily_return'] = df['daily_return'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Weight
    average_volume = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = (df['volume'] - average_volume) * df['daily_return']
    
    # Combine Exponentially Smoothed Moving Average and Volume Weight
    df['combined_momentum'] = df['ema_daily_return'] * df['volume_weight']
    
    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Volume-Weighted Breakout Indicator
    df['volume_weighted_breakout'] = df['volume'] * (df['close'] - df['open'])
    
    # Aggregate Breakout Indicators
    df['aggregated_breakout'] = df['volume'] * df['price_breakout_ratio']
    df['aggregated_breakout'] = df['aggregated_breakout'].rolling(window=n).sum()
    
    # Smooth Aggregated Breakout Indicators
    df['smoothed_aggregated_breakout'] = df['aggregated_breakout'].rolling(window=n).mean()
    
    # Combine Momentum and Breakout Indicators
    df['combined_indicator'] = df['smoothed_aggregated_breakout'] + df['combined_momentum']
    
    # Volume-Weighted Combined Indicator
    df['volume_weighted_combined_indicator'] = df['combined_indicator'] * df['volume_weight']
    
    # Determine Daily Return Deviation from VWAP
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=n).mean()
    df['daily_return_deviation'] = df['close'] - df['vwap']
    df['trend_reversal'] = (df['daily_return_deviation'] > df['daily_return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Adjust for Volume
    volume_ma = df['volume'].rolling(window=20).mean()
    df['adjusted_intraday_return'] = (df['volume'] - volume_ma) * df['intraday_return']
    
    # Incorporate Price Volatility
    std_close = df['close'].rolling(window=10).std()
    df['intraday_return_adjusted_vol'] = df['adjusted_intraday_return'] * np.where(std_close > std_close.mean(), 1.5, 0.7)
    
    # Integrate Reversal and Intraday Indicators
    df['integrated_indicator'] = df['trend_reversal'] * df['intraday_return_adjusted_vol']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_indicator'] * df['volume_weighted_combined_indicator']
    
    return df['final_alpha_factor']
```
