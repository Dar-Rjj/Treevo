```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()

    # Compute Exponentially Smoothed Moving Average of Daily Returns
    n = 20
    alpha = 0.1
    df['ewma_returns'] = df['daily_returns'].ewm(alpha=alpha, adjust=False).mean()

    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Price Momentum Factor
    price_momentum = df['daily_returns'].rolling(window=n).sum()
    k1 = 1.0  # Constant for price momentum

    # Compute Volume Momentum Factor
    m = 14
    volume_momentum = df['volume_change'].rolling(window=m).sum()
    k2 = 1.0  # Constant for volume momentum

    # Combine Price and Volume Momentum
    df['combined_momentum'] = (k1 * price_momentum) + (k2 * volume_momentum)

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Price Breakout Ratio
    df['price_breakout_ratio'] = (df['high'] - df['open']) / df['high_low_range']

    # Calculate Volume-Weighted Breakout Indicator
    df['vwap_breakout'] = (df['close'] - df['open']) * df['volume'] * df['price_breakout_ratio']

    # Determine Daily Return Deviation from VWAP
    df['vwap'] = (df['volume'] * df[['open', 'high', 'low', 'close']].mean(axis=1)).cumsum() / df['volume'].cumsum()
    df['return_deviation'] = df['close'] - df['vwap']

    # Identify Trend Reversal Potential
    df['reversal_potential'] = (df['return_deviation'] > df['return_deviation'].shift(1)) & (df['volume'] > df['volume'].rolling(window=14).mean())

    # Calculate Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / avg_volume

    # Adjust Exponentially Smoothed Moving Average with Volume Weight
    df['adjusted_ewma'] = df['ewma_returns'] * df['volume_weight'] + df['combined_momentum']

    # Generate Alpha Factor
    alpha_factor = df['adjusted_ewma'].rolling(window=7).mean()

    # Incorporate Aggregate Breakout Indicators
    breakout_indicators = df['vwap_breakout'].ewm(span=7, adjust=False).mean()
    df['composite_breakout'] = breakout_indicators

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Adjust for Volume
    vol_ma = df['volume'].rolling(window=7).mean()
    df['vol_adjusted_intraday'] = (df['volume'] - vol_ma) * df['intraday_return']

    # Incorporate Price Volatility
    std_close = df['close'].rolling(window=7).std()
    df['vol_adjusted_intraday'] = np.where(std_close > std_close.mean(), df['vol_adjusted_intraday'] * 1.5, df['vol_adjusted_intraday'] * 0.5)

    # Incorporate Momentum Shift
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    momentum_shift = np.where(ema_5 > ema_20, 0.8, -0.8)
    df['vol_adjusted_intraday'] += momentum_shift

    # Combine Indicators
    df['final_factor'] = alpha_factor + df['composite_breakout'] + df['vol_adjusted_intraday']
    df['final_factor'] = df['final_factor'].ewm(span=7, adjust=False).mean()

    return df['final_factor']
```
