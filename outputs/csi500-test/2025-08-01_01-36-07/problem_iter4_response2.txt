```python
import pandas as pd

def heuristics_v2(df, multi_day_window=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Volume-Adjusted Intraday Range
    volume_adjusted_intraday_range = intraday_range * df['volume']
    
    # Adjusted Intraday Range
    adjusted_intraday_range = volume_adjusted_intraday_range * close_to_open_ratio
    
    # Calculate Daily Return
    daily_return = df['close'].diff(1)
    
    # Sum Multi-Day Returns
    sum_multi_day_returns = daily_return.rolling(window=multi_day_window).sum()
    
    # Integrate Intraday and Multi-Day Factors
    integrated_factor = adjusted_intraday_range + sum_multi_day_returns
    
    # Final Alpha Factor
    alpha_factor = integrated_factor
    
    return alpha_factor
```
