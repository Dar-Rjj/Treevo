```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = (intraday_range * close_to_open_ratio) * df['volume']
    
    # Multi-Day Price Ratio
    multi_day_price_ratio = df['close'] / df['close'].shift(1)
    
    # Multi-Day Volume Ratio
    multi_day_volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Combine Intraday and Multi-Day Factors
    combined_factor = (volume_weighted_intraday_momentum + 
                       multi_day_price_ratio + 
                       multi_day_volume_ratio)
    
    return combined_factor
```
