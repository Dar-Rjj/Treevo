```python
import pandas as pd

def heuristics_v2(df, n=10, m=50):
    # Calculate Daily Price Movement
    df['Daily_Price_Movement'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Daily_Price_Movement'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth'] - df['Intraday_Volatility'])
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].mean()
    total_avg_volume = avg_volume * len(df)
    df['Final_Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (avg_volume / total_avg_volume)
    
    # Calculate Long-Term Trend
    df['Long_Term_Trend'] = df['close'].rolling(window=m).mean()
    
    # Combine Short-Term and Long-Term Momentum
    df['Combined_Momentum'] = df['Final_Volume_Adjusted_Momentum'] + df['Long_Term_Trend']
    
    # Incorporate Price Reversal Signal
    df['Price_Reversal'] = df['close'] - df['open']
    
    # Adjust Combined Momentum with Reversal Signal
    df['Adjusted_Combined_Momentum'] = df['Combined_Momentum'] + df['Price_Reversal']
    
    # Generate Alpha Factor
    alpha_factor = df['Adjusted_Combined_Momentum']
    
    return alpha_factor
```
