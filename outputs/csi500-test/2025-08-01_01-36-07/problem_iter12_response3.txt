```python
import pandas as pd

def heuristics_v2(df, n_days=10, vwap_window=5, ema_window=10):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n_days).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=vwap_window).sum() / df['volume'].rolling(window=vwap_window).sum()
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    average_volume = df['volume'].mean()
    total_average_volume = df['volume'].expanding().mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (average_volume / total_average_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5) - 1).fillna(0)
    df['10_day_return'] = (df['close'] / df['close'].shift(10) - 1).fillna(0)
    df['multi_day_return'] = (df['5_day_return'] + df['10_day_return']) / 2
    df['final_volume_adjusted_momentum'] += df['multi_day_return']
    
    # Incorporate Liquidity
    df['average_volume'] = df['volume'].rolling(window=n_days).mean()
    total_average_volume = df['volume'].expanding().mean()
    df['final_volume_adjusted_momentum'] *= (df['average_volume'] / total_average_volume)
    
    # Enhance with Dynamic Price and Volume Ratios
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    df['avg_price_to_volume_ratio'] = df['price_to_volume_ratio'].rolling(window=n_days).mean()
    df['final_volume_adjusted_momentum'] *= (df['avg_price_to_volume_ratio'] / df['price_to_volume_ratio'])
    
    # Consider Volatility Clustering
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=ema_window).mean()
    df['avg_intraday_volatility'] = df['intraday_volatility'].expanding().mean()
    df['final_volume_adjusted_momentum'] *= (df['ema_intraday_volatility'] / df['avg_intraday_volatility'])
    
    return df['final_volume_adjusted_momentum']
```
