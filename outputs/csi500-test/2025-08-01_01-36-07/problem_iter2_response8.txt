import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum (Sum of Daily Price Ranges over n days)
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    df['vol_weighted_momentum'] = df['raw_momentum'] * (avg_volume / total_avg_volume)
    
    # Incorporate Price Gap to get the final alpha factor
    df['alpha_factor'] = df['vol_weighted_momentum'] + df['price_gap']
    
    return df['alpha_factor']
