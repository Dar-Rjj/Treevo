```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, vol_window=5, ema_span=10):
    # Daily Price Dynamics
    df['Price Range'] = df['high'] - df['low']
    df['Price Gap'] = df['open'] - df['close']

    # Raw Momentum
    df['Daily Price Movement'] = df['close'].diff()
    df['Raw Momentum'] = df['Daily Price Movement'].rolling(window=n).sum() + \
                         df['Price Gap'].rolling(window=n).sum()

    # Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Market Depth
    df['Market Depth'] = df['amount'] / df['volume']

    # Intraday Volatility
    df['Intraday Volatility'] = df['high'] - df['low']

    # Adjusted Momentum
    df['Adjusted Momentum'] = df['Raw Momentum'] * (df['VWAP'] + df['Market Depth']) / df['Intraday Volatility']

    # Final Volume-Adjusted Momentum
    avg_vol = df['volume'].rolling(window=n).mean()
    total_avg_vol = df['volume'].mean()
    df['Volume-Adjusted Momentum'] = df['Adjusted Momentum'] * (avg_vol / total_avg_vol)

    # Multi-Day Momentum
    df['5-Day Return'] = df['close'].pct_change(5)
    df['10-Day Return'] = df['close'].pct_change(10)
    df['Multi-Day Momentum'] = (df['5-Day Return'] + df['10-Day Return']) / 2

    # Combine with Final Volume-Adjusted Momentum
    df['Combined Momentum'] = df['Volume-Adjusted Momentum'] + df['Multi-Day Momentum']

    # Liquidity Adjustment
    avg_vol_n_days = df['volume'].rolling(window=n).mean()
    total_avg_vol_n_days = df['volume'].mean()
    df['Liquidity Adjusted Momentum'] = df['Combined Momentum'] * (avg_vol_n_days / total_avg_vol_n_days)

    # Dynamic Price and Volume Ratios
    df['Price-to-Volume Ratio'] = df['close'] / df['volume']
    df['Moving Avg Price-to-Volume Ratio'] = df['Price-to-Volume Ratio'].rolling(window=n).mean()
    avg_price_to_volume_ratio = df['Price-to-Volume Ratio'].mean()
    df['Liquidity and Price Adjusted Momentum'] = df['Liquidity Adjusted Momentum'] * (df['Moving Avg Price-to-Volume Ratio'] / avg_price_to_volume_ratio)

    # Volatility Clustering
    df['EMA Intraday Volatility'] = df['Intraday Volatility'].ewm(span=ema_span).mean()
    avg_intraday_volatility = df['Intraday Volatility'].mean()
    df['Final Alpha Factor'] = df['Liquidity and Price Adjusted Momentum'] * (df['EMA Intraday Volatility'] / avg_intraday_volatility)

    return df['Final Alpha Factor']
```
