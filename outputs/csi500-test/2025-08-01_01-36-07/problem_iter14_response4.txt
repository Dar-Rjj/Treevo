```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Normalize the volume
    normalized_volume = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the amount weighted by the buying and selling pressures
    weighted_amount = df['amount'] * (buying_pressure - selling_pressure)
    
    # The final factor value is the moving average of the weighted amount, further scaled by the normalized volume
    factor = (weighted_amount.rolling(window=20).mean() * normalized_volume).dropna()
    
    return factor
```
