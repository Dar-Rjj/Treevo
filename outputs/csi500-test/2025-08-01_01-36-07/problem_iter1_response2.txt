```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters for EMA calculation
    short_span = 10
    long_span = 30
    
    # Compute EMAs
    df['EMA_short'] = df['close'].ewm(span=short_span, adjust=False).mean()
    df['EMA_long'] = df['close'].ewm(span=long_span, adjust=False).mean()
    
    # Generate EMA cross signals
    df['EMA_cross'] = np.where(df['EMA_short'] > df['EMA_long'], 1, -1)
    
    # Calculate day-to-day volume changes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Apply a weighted score to EMA cross signals using the absolute value of volume change
    df['weighted_signal'] = df['EMA_cross'] * df['volume_change'].abs()
    
    # Return the factor values
    return df['weighted_signal']
```
