import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Daily Price Movement
    daily_high_low = df['high'] - df['low']
    daily_open_close = df['open'] - df['close']
    
    # Raw Momentum
    raw_momentum = daily_high_low.rolling(window=n).sum()
    
    # Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Market Depth
    market_depth = df['amount'] / df['volume']
    
    # Intraday Volatility
    intraday_volatility = daily_high_low
    
    # Adjusted Momentum
    adjusted_momentum = (raw_momentum * (vwap + market_depth)) / intraday_volatility
    
    # Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    final_volume_adjusted_momentum = adjusted_momentum * (avg_volume / total_avg_volume)
    
    # Multi-Day Momentum
    five_day_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    combined_multi_day_momentum = (five_day_return + ten_day_return) / 2
    final_volume_adjusted_momentum += combined_multi_day_momentum
    
    # Liquidity
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    liquidity_factor = avg_volume_n_days / total_avg_volume
    final_volume_adjusted_momentum *= liquidity_factor
    
    # Dynamic Price and Volume Ratios
    price_to_volume_ratio = df['close'] / df['volume']
    moving_avg_price_to_volume_ratio = price_to_volume_ratio.rolling(window=n).mean()
    average_price_to_volume_ratio = price_to_volume_ratio.mean()
    dynamic_price_volume_ratio = moving_avg_price_to_volume_ratio / average_price_to_volume_ratio
    final_volume_adjusted_momentum *= dynamic_price_volume_ratio
    
    # Volatility Clustering
    ema_intraday_volatility = intraday_volatility.ewm(span=n, adjust=False).mean()
    average_intraday_volatility = intraday_volatility.mean()
    volatility_clustering = ema_intraday_volatility / average_intraday_volatility
    final_volume_adjusted_momentum *= volatility_clustering
    
    return final_volume_adjusted_momentum.dropna()
