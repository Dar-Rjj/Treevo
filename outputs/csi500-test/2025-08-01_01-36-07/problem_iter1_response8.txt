```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Calculate 5-Day Moving Average of Daily Returns
    df['5_day_return_ma'] = df['daily_return'].rolling(window=5).mean()
    
    # Adjust for Volume
    df['5_day_volume_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_return'] = df['5_day_return_ma'] * (1 / df['5_day_volume_ma'])
    
    # Extract the alpha factor
    alpha_factor = df['volume_adjusted_return']
    
    return alpha_factor
```
