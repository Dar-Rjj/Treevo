```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_price_range'] * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    df['adjusted_close_to_open_ratio'] = df['close_to_open_ratio'] * df['volume_weighted_intraday_range']
    
    # Calculate 5-Day Sum of Adjusted Close to Open Ratio
    df['5d_sum_adjusted_close_to_open'] = df['adjusted_close_to_open_ratio'].rolling(window=5).sum()
    
    # Calculate 5-Day Average Intraday Range
    df['5d_avg_intraday_range'] = df['intraday_price_range'].rolling(window=5).mean()
    
    # Calculate 5-Day Moving Average of Adjusted Close to Open Ratio
    df['5d_ma_adjusted_close_to_open'] = df['adjusted_close_to_open_ratio'].rolling(window=5).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate 5-Day Average Intraday Volatility
    df['5d_avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['5d_sum_adjusted_close_to_open'] - df['5d_ma_adjusted_close_to_open']) + df['5d_avg_intraday_range'] + df['5d_avg_intraday_volatility']
    
    return df['alpha_factor']
```
