```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Adjust Raw Momentum with VWAP and Market Depth
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth'])
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].rolling(window=n).mean().mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5d_return'] = df['close'].pct_change(periods=5)
    df['10d_return'] = df['close'].pct_change(periods=10)
    df['multi_day_return'] = (df['5d_return'] + df['10d_return']) / 2
    df['final_volume_adjusted_momentum'] += df['multi_day_return']
    
    # Incorporate Long-Term Momentum
    df['20d_return'] = df['close'].pct_change(periods=20)
    df['50d_return'] = df['close'].pct_change(periods=50)
    df['long_term_return'] = (df['20d_return'] + df['50d_return']) / 2
    df['final_volume_adjusted_momentum'] += df['long_term_return']
    
    # Incorporate Liquidity
    df['final_volume_adjusted_momentum'] *= (avg_volume / total_avg_volume)
    
    return df['final_volume_adjusted_momentum']
```
