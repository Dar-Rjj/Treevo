```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    volume_weighted_intraday_range = intraday_range * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    adjusted_close_to_open_ratio = close_to_open_ratio * volume_weighted_intraday_range
    
    # Sum Adjusted Close to Open Ratio over a 5-day window
    sum_adjusted_ratio = adjusted_close_to_open_ratio.rolling(window=5).sum()
    
    # Calculate the average 5-day Intraday Range over the same window
    avg_5day_intraday_range = intraday_range.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = sum_adjusted_ratio - avg_5day_intraday_range
    
    return alpha_factor
```
