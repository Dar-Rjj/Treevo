    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

def compute_rsi(close, n):
    delta = close.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(n).mean()
    roll_down = down.abs().rolling(n).mean()
    rs = roll_up / roll_down
    rsi = 100.0 - (100.0 / (1.0 + rs))
    return rsi
