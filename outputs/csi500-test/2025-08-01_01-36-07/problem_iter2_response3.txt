```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Price_Momentum'] = df['close'] - df['SMA_20']
    
    # Calculate Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Volatility'] = df['Daily_Returns'].rolling(window=20).std()
    
    # Calculate Volume Synchronization
    df['Volume_Difference'] = df['volume'].diff()
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adaptive Normalization
    df['Normalized_Price_Momentum'] = df['Price_Momentum'] / df['Price_Momentum'].rolling(window=20).std()
    df['Normalized_Volatility'] = df['Volatility'] / df['Volatility'].rolling(window=20).std()
    df['Normalized_Volume_Synchronization'] = df['Volume_Ratio'] / df['Volume_Ratio'].rolling(window=20).std()
    
    # Generate Alpha Factor
    df['Alpha_Factor'] = df['Normalized_Price_Momentum'] * df['Normalized_Volatility'] * df['Normalized_Volume_Synchronization']
    
    return df['Alpha_Factor']
```
