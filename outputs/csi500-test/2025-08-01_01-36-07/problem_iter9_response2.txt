import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, average_volume_days=20):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=average_volume_days).mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (df['volume'] / avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['multi_day_momentum'] = (df['5_day_return'] + df['10_day_return']) / 2
    
    # Combine Final Volume-Adjusted Momentum with Multi-Day Momentum
    df['final_momentum'] = df['final_volume_adjusted_momentum'] + df['multi_day_momentum']
    
    # Incorporate Liquidity
    total_avg_volume = df['volume'].rolling(window=average_volume_days).mean().mean()
    df['final_momentum'] = df['final_momentum'] * (avg_volume / total_avg_volume)
    
    # Return the final momentum factor
    return df['final_momentum'].dropna()
