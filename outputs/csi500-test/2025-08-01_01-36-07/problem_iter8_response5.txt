```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']

    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP and Market Depth
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth'])

    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].mean()
    total_avg_volume = df['volume'].sum()
    df['final_momentum'] = df['adjusted_momentum'] * (avg_volume / total_avg_volume)

    # Integrate Recent Momentum
    df['5d_close_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10d_close_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['recent_momentum'] = (df['5d_close_return'] + df['10d_close_return']) / 2

    # Combine Final Volume-Adjusted Momentum with Recent Momentum
    df['final_factor'] = df['final_momentum'] + df['recent_momentum']

    return df['final_factor']
```
