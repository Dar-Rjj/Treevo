```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Movement
    volume_weighted_intraday_movement = intraday_range * (df['volume'] * close_to_open_ratio)
    
    # Calculate Multi-Day Momentum
    daily_return = df['close'].pct_change()
    multi_day_momentum = daily_return.rolling(window=5).sum()
    
    # Integrate Intraday and Multi-Day Factors
    integrated_factor = volume_weighted_intraday_movement + multi_day_momentum
    
    return integrated_factor
```
