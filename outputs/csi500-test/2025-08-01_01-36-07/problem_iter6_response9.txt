```python
import pandas as pd

def heuristics_v2(df, momentum_days=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_open_ratio = df['close'] / df['open']
    
    # Weighted Difference
    weighted_difference = (intraday_range * close_open_ratio) * df['volume']
    
    # Calculate Multi-Day Momentum
    daily_return = df['close'] - df['close'].shift(1)
    multi_day_momentum = daily_return.rolling(window=momentum_days).sum()
    
    # Integrate Intraday and Multi-Day Factors
    combined_factor = weighted_difference + multi_day_momentum
    
    # Adjust for Liquidity
    liquidity_factor = (df['amount'] * 0.5) + df['volume']
    
    # Final Factor
    final_factor = combined_factor * liquidity_factor
    
    return final_factor
```
