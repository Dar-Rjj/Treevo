import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Generate a novel alpha factor combining price momentum, volume, and sector trends.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as dates.
    
    Returns:
    pd.Series: Series of the generated alpha factor indexed by date.
    """
    # Calculate daily log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate cumulative sum of log returns for the last 10 days
    df['cumulative_log_return'] = df['log_return'].rolling(window=10).sum()
    
    # Calculate average volume over the last 30 days
    df['avg_volume'] = df['volume'].rolling(window=30).mean()
    
    # Weighted momentum by average volume
    df['momentum_weighted_by_vol'] = df['cumulative_log_return'] * df['avg_volume']
    
    # Calculate historical volatility over the last 30 days
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Normalize momentum by volatility
    df['normalized_momentum'] = df['momentum_weighted_by_vol'] / df['volatility']
    
    # Assuming 'sector' is a column in the dataframe
    # If not, you would need to map each stock to its sector
    # For simplicity, let's assume all stocks are in the same sector
    # Calculate sector average log return
    df['sector_avg_log_return'] = df.groupby('sector')['log_return'].transform(lambda x: x.rolling(window=10).mean())
    
    # Adjust for sector trend
    df['alpha_factor'] = df['normalized_momentum'] - df['sector_avg_log_return']
    
    # Return the alpha factor as a series
    return df['alpha_factor'].dropna()
