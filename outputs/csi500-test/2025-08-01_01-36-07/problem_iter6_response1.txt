```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    volume_weighted_intraday_range = intraday_range * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    adjusted_close_to_open_ratio = close_to_open_ratio * volume_weighted_intraday_range
    
    # Calculate 5-Day Intraday Range Sum
    five_day_intraday_range_sum = intraday_range.rolling(window=5).sum()
    
    # Calculate 5-Day Intraday Range Average
    five_day_intraday_range_average = five_day_intraday_range_sum / 5
    
    # Calculate 5-Day Adjusted Close to Open Ratio Sum
    five_day_adjusted_close_to_open_ratio_sum = adjusted_close_to_open_ratio.rolling(window=5).sum()
    
    # Calculate Dynamic Momentum Factor
    dynamic_momentum_factor = five_day_adjusted_close_to_open_ratio_sum - five_day_intraday_range_average
    
    # Final Alpha Factor
    alpha_factor = dynamic_momentum_factor
    
    return alpha_factor
```
