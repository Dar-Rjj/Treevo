```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the Intraday Momentum Oscillator.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A series with the index as date and the values as the calculated factor.
    """
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Apply Range Adjustment
    range_adjustment = intraday_range * close_to_open_ratio
    
    # Weighted Difference
    weighted_difference = range_adjustment * df['volume']
    
    return weighted_difference
```
