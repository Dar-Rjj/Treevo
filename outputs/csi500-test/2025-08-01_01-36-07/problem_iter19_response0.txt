import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10):
    # Calculate Daily Price Movement
    df['daily_price_movement'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_movement'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    total_avg_volume = df['volume'].mean()
    df['final_vol_adj_momentum'] = df['adjusted_momentum'] * (df['volume'].rolling(window=n).mean() / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['multi_day_momentum'] = (df['5_day_return'] + df['10_day_return']) / 2
    
    # Incorporate Liquidity
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['final_vol_adj_momentum'] *= df['average_volume'] / total_avg_volume
    
    # Enhance with Dynamic Price and Volume Ratios
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    df['ma_price_to_volume_ratio'] = df['price_to_volume_ratio'].rolling(window=n).mean()
    df['final_vol_adj_momentum'] *= df['ma_price_to_volume_ratio'] / df['price_to_volume_ratio'].mean()
    
    # Consider Volatility Clustering
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=n, adjust=False).mean()
    df['final_vol_adj_momentum'] *= df['ema_intraday_volatility'] / df['intraday_volatility'].mean()
    
    # Incorporate Short-Term Reversal
    df['2_day_return'] = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    df['final_vol_adj_momentum'] *= (1 + df['2_day_return'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['final_vol_adj_momentum'] + df['multi_day_momentum']
    
    return df['final_alpha_factor'].dropna()
