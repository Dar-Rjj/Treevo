```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the price change (close - open) / open
    df['price_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume change percentage
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the amount change percentage
    df['amount_change'] = df['amount'].pct_change()
    
    # Heuristic 1: Relative Range and Price Change interaction
    df['heuristic_1'] = df['relative_range'] * df['price_change']
    
    # Heuristic 2: Volume and Amount change interaction
    df['heuristic_2'] = df['volume_change'] * df['amount_change']
    
    # Heuristic 3: Momentum factor using past returns
    df['heuristic_3'] = df['return'].rolling(window=5).mean()
    
    # Combine all heuristics into a single alpha factor
    df['alpha_factor'] = (df['heuristic_1'] + df['heuristic_2'] + df['heuristic_3']) / 3
    
    # Return the alpha factor
    return df['alpha_factor'].dropna()
```
