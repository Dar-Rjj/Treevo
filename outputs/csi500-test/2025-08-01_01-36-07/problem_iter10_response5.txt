```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Adjust Raw Momentum with VWAP
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * df['VWAP']
    
    # Calculate Final Volume-Adjusted Momentum
    total_volume = df['volume'].sum()
    df['Final_Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (df['volume'] / total_volume)
    
    # Integrate Multi-Day Momentum
    df['5_Day_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_Day_Return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['Combined_Return'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2
    df['Final_Volume_Adjusted_Momentum'] += df['Combined_Return']
    
    # Incorporate Liquidity
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = avg_volume.sum()
    df['Liquidity_Adjusted_Momentum'] = df['Final_Volume_Adjusted_Momentum'] * (avg_volume / total_avg_volume)
    
    return df['Liquidity_Adjusted_Momentum']
```
