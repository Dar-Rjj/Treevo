```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the relative price movement
    rel_price_movement = (df['close'] - df['open']) / avg_price
    
    # Calculate the liquidity factor
    liquidity_factor = df['volume'] / df['amount']
    
    # Calculate the volatility as the range divided by the average price
    volatility = (df['high'] - df['low']) / avg_price
    
    # Combine all factors to create the final alpha factor
    alpha_factor = rel_price_movement * liquidity_factor / (volatility + 1e-6)
    
    return alpha_factor
```
