import pandas as pd

def heuristics_v2(df, n=5, m=10):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].expanding().mean()
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5_Day_Return'] = (df['close'] / df['close'].shift(5) - 1).fillna(0)
    df['10_Day_Return'] = (df['close'] / df['close'].shift(10) - 1).fillna(0)
    df['Multi_Day_Return'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2
    df['Final_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Multi_Day_Return']
    
    # Incorporate Liquidity
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Liquidity_Adjusted_Momentum'] = df['Final_Volume_Adjusted_Momentum'] * (df['Average_Volume'] / total_avg_volume)
    
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Accumulated Price Change
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=m).sum()
    
    # Calculate Volume Derivative
    df['Volume_Derivative'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Market Trend
    df['20_MA'] = df['close'].rolling(window=20).mean()
    df['50_MA'] = df['close'].rolling(window=50).mean()
    df['Trend'] = (df['20_MA'] > df['50_MA']).astype(int) * 2 - 1  # Bullish: 1, Bearish: -1
    
    # Adjust Final Volume-Adjusted Momentum with Market Trend
    df['Trend_Adjusted_Momentum'] = df['Liquidity_Adjusted_Momentum'] * df['Trend']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Trend_Adjusted_Momentum'] + df['Accumulated_Price_Change'] + df['Volume_Derivative']
    
    return df['Alpha_Factor']
