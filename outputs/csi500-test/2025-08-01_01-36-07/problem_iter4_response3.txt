```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Weighted Difference with Range Adjustment
    weighted_difference = (intraday_range * close_to_open_ratio) * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['close'].shift(1)
    
    # Integrate Intraday Factors
    intraday_factors = weighted_difference + intraday_momentum
    
    return intraday_factors
```
