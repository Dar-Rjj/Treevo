```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    sma_10 = df['close'].rolling(window=10).mean()
    price_momentum = df['close'] - sma_10
    
    # Calculate Volume Synchronization
    volume_diff = df['volume'].diff()
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_synchronization = (volume_diff + volume_ratio) / 2  # Simple average for synchronization
    
    # Normalize Price Momentum
    normalized_price_momentum = price_momentum / price_momentum.abs().max()
    
    # Normalize Volume Synchronization
    normalized_volume_synchronization = volume_synchronization / volume_synchronization.abs().max()
    
    # Generate Alpha Factor
    alpha_factor = normalized_price_momentum * normalized_volume_synchronization
    
    return alpha_factor
```
