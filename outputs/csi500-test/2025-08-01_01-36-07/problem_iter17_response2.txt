import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].rolling(window=len(df)).mean()
    df['Final_Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5Day_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10Day_Return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['Multi_Day_Momentum'] = (df['5Day_Return'] + df['10Day_Return']) / 2
    df['Final_Volume_Adjusted_Momentum'] += df['Multi_Day_Momentum']
    
    # Incorporate Liquidity
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Total_Average_Volume'] = df['volume'].rolling(window=len(df)).mean()
    df['Liquidity_Adjusted_Momentum'] = df['Final_Volume_Adjusted_Momentum'] * (df['Average_Volume'] / df['Total_Average_Volume'])
    
    # Enhance with Dynamic Price and Volume Ratios
    df['Price_to_Volume_Ratio'] = df['close'] / df['volume']
    df['Moving_Avg_Price_to_Volume_Ratio'] = df['Price_to_Volume_Ratio'].rolling(window=n).mean()
    df['Average_Price_to_Volume_Ratio'] = df['Price_to_Volume_Ratio'].rolling(window=len(df)).mean()
    df['Dynamic_Adjusted_Momentum'] = df['Liquidity_Adjusted_Momentum'] * (df['Moving_Avg_Price_to_Volume_Ratio'] / df['Average_Price_to_Volume_Ratio'])
    
    # Consider Volatility Clustering
    df['EMA_Intraday_Volatility'] = df['Intraday_Volatility'].ewm(span=m, adjust=False).mean()
    df['Average_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=len(df)).mean()
    df['Final_Alpha_Factor'] = df['Dynamic_Adjusted_Momentum'] * (df['EMA_Intraday_Volatility'] / df['Average_Intraday_Volatility'])
    
    return df['Final_Alpha_Factor']
