```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Cumulative Sum of Log Returns for last 10 days
    df['cum_log_return'] = df['log_return'].rolling(window=10).sum()
    
    # Calculate Average Volume for last 30 days
    df['avg_volume'] = df['volume'].rolling(window=30).mean()
    
    # Weight the Momentum by Average Volume
    df['momentum_factor'] = df['cum_log_return'] * df['avg_volume']
    
    return df['momentum_factor']
```
