```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    df['adjusted_close_to_open_ratio'] = df['close_to_open_ratio'] * df['volume_weighted_intraday_range']
    
    # Calculate 5-Day Sum of Adjusted Close to Open Ratio
    df['5d_sum_adjusted_close_to_open_ratio'] = df['adjusted_close_to_open_ratio'].rolling(window=5).sum()
    
    # Calculate 5-Day Average Intraday Range
    df['5d_avg_intraday_range'] = df['intraday_range'].rolling(window=5).mean()
    
    # Calculate 5-Day Momentum
    df['5d_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Integrate 5-Day Momentum
    df['integrated_5d_momentum'] = (df['5d_sum_adjusted_close_to_open_ratio'] - df['5d_avg_intraday_range']) + df['5d_momentum']
    
    # Final Alpha Factor
    alpha_factor = df['integrated_5d_momentum']
    
    return alpha_factor
```
