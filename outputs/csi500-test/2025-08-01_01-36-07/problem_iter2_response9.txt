```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Intraday Momentum
    intraday_momentum = df['high'] - df['open']
    
    # Calculate Close-to-Previous Close Return
    close_to_prev_close_return = df['close'].diff()
    
    # Determine Volume-Weighted Momentum
    volume_weighted_momentum = intraday_momentum * intraday_range
    
    # Compute Adjusted Close-to-Previous Close Return
    adjusted_close_to_prev_close_return = close_to_prev_close_return / df['close'].shift(1)
    
    # Combine Factors
    combined_factors = intraday_range + volume_weighted_momentum + adjusted_close_to_prev_close_return
    
    # Final Factor
    final_factor = combined_factors / df['volume']
    final_factor[df['volume'] == 0] = 0  # Handle division by zero
    
    return final_factor
```
