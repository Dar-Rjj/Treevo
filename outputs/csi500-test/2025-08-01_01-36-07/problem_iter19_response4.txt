import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement
    daily_price_movement = df['high'] - df['low']
    
    # Calculate Price Gap
    price_gap = df['open'] - df['close']
    
    # Calculate Raw Momentum
    raw_momentum = daily_price_movement.rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    market_depth = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Raw Momentum
    adjusted_momentum = (raw_momentum * (vwap + market_depth) / intraday_volatility).fillna(0)
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].expanding().mean()
    final_vol_adj_momentum = adjusted_momentum * (avg_volume / total_avg_volume).fillna(0)
    
    # Integrate Multi-Day Momentum
    five_day_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    multi_day_momentum = (five_day_return + ten_day_return) / 2
    final_vol_adj_momentum += multi_day_momentum.fillna(0)
    
    # Incorporate Liquidity
    final_vol_adj_momentum *= (avg_volume / total_avg_volume).fillna(0)
    
    # Enhance with Dynamic Price and Volume Ratios
    price_to_volume_ratio = df['close'] / df['volume']
    ma_price_to_volume_ratio = price_to_volume_ratio.rolling(window=n).mean()
    avg_price_to_volume_ratio = price_to_volume_ratio.expanding().mean()
    final_vol_adj_momentum *= (ma_price_to_volume_ratio / avg_price_to_volume_ratio).fillna(0)
    
    # Consider Volatility Clustering
    ema_intraday_volatility = intraday_volatility.ewm(span=n, adjust=False).mean()
    avg_intraday_volatility = intraday_volatility.expanding().mean()
    final_vol_adj_momentum *= (ema_intraday_volatility / avg_intraday_volatility).fillna(0)
    
    # Incorporate Short-Term Reversal
    two_day_return = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    final_vol_adj_momentum *= (1 + two_day_return).fillna(0)
    
    return final_vol_adj_momentum
