```python
import pandas as pd

def heuristics_v2(df, n=5, long_period=20):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']

    # Calculate Raw Momentum
    raw_momentum = df['price_range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Incorporate Market Depth
    market_depth = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    adjusted_momentum = raw_momentum * (vwap + market_depth) / intraday_volatility

    # Calculate Final Volume-Adjusted Momentum
    moving_avg_volume = df['volume'].rolling(window=n).mean()
    final_vol_adj_momentum = adjusted_momentum * (df['volume'] / moving_avg_volume)

    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    multi_day_momentum = (df['5_day_return'] + df['10_day_return']) / 2
    final_vol_adj_momentum += multi_day_momentum

    # Incorporate Liquidity
    total_avg_volume = df['volume'].rolling(window=long_period).mean()
    final_factor = final_vol_adj_momentum * (moving_avg_volume / total_avg_volume)

    return final_factor
```
