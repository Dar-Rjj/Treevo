```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    df['vol_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    df['adj_close_to_open_ratio'] = df['close_to_open_ratio'] * df['vol_weighted_intraday_range']
    
    # Rolling 5-Day Sum of Adjusted Close to Open Ratio
    df['rolling_5d_sum_adj_close_to_open'] = df['adj_close_to_open_ratio'].rolling(window=5).sum()
    
    # Rolling 5-Day Average Intraday Range
    df['rolling_5d_avg_intraday_range'] = df['intraday_range'].rolling(window=5).mean()
    
    # Multi-Day Price Momentum
    df['multi_day_price_mom'] = df['close'].diff(periods=1)
    
    # Multi-Day Relative Strength
    df['multi_day_rel_strength'] = df['close'] / df['close'].shift(5)
    
    # Moving Average Convergence Divergence (MACD)
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    
    # Signal Line
    df['signal_line'] = df['macd'].ewm(span=9, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['rolling_5d_sum_adj_close_to_open'] - 
        df['rolling_5d_avg_intraday_range'] + 
        df['multi_day_price_mom'] + 
        df['multi_day_rel_strength'] + 
        df['macd'] - 
        df['signal_line']
    )
    
    return df['alpha_factor']
```
