```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Calculate Volume-Weighted Return
    df['prev_close'] = df['close'].shift(1)
    df['return'] = (df['close'] - df['prev_close']) / df['prev_close']
    df['volume_weighted_return'] = df['return'] * df['volume']
    
    # Combine Intraday Momentum and Volume-Weighted Return
    df['range_adjustment'] = df['intraday_range'] * df['close_to_open_ratio']
    df['combined_momentum'] = df['range_adjustment'] + df['volume_weighted_return']
    
    # Incorporate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['multi_day_momentum'] = (df['5_day_return'] + df['10_day_return']) / 2
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] + df['multi_day_momentum']
    
    return df['alpha_factor']
```
