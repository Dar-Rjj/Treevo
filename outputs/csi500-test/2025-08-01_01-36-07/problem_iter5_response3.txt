```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']

    # Calculate Volume-Weighted Momentum
    volume_weighted_momentum = intraday_range * close_to_open_ratio * df['volume']
    
    # Calculate Daily Volatility (using rolling 10 day standard deviation of close prices)
    daily_volatility = df['close'].rolling(window=10).std()
    
    # Calculate Volatility Adjusted Momentum
    volatility_adjusted_momentum = volume_weighted_momentum / daily_volatility
    
    # Replace NaN values with 0
    volatility_adjusted_momentum.fillna(0, inplace=True)
    
    return volatility_adjusted_momentum
```
