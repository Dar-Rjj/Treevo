```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement: High - Low
    daily_movement = df['high'] - df['low']
    
    # Calculate Price Gap: Open - Close
    price_gap = df['open'] - df['close']
    
    # Calculate Raw Momentum: Sum of (High - Low) over n days
    raw_momentum = daily_movement.rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth: Amount / Volume
    market_depth = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility: High - Low
    intraday_volatility = daily_movement
    
    # Adjust Raw Momentum
    adjusted_momentum = (raw_momentum * (vwap + market_depth) / intraday_volatility).fillna(0)
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    final_adjusted_momentum = adjusted_momentum * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    five_day_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    multi_day_return = (five_day_return + ten_day_return) / 2
    final_adjusted_momentum += multi_day_return
    
    # Incorporate Short-Term Reversal
    two_day_return = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    final_adjusted_momentum *= (1 + two_day_return)
    
    # Enhance with Dynamic Price and Volume Ratios
    price_to_volume_ratio = df['close'] / df['volume']
    moving_avg_price_to_volume = price_to_volume_ratio.rolling(window=n).mean()
    avg_price_to_volume = price_to_volume_ratio.mean()
    final_adjusted_momentum *= (moving_avg_price_to_volume / avg_price_to_volume)
    
    # Consider Volatility Clustering
    ema_intraday_volatility = intraday_volatility.ewm(span=n, adjust=False).mean()
    avg_intraday_volatility = intraday_volatility.mean()
    final_adjusted_momentum *= (ema_intraday_volatility / avg_intraday_volatility)
    
    # Emphasize Liquidity
    final_adjusted_momentum *= (avg_volume / total_avg_volume)
    
    return final_adjusted_momentum
```
