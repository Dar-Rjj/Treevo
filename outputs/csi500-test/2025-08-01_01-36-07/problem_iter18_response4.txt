```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5):
    # Calculate Daily Price Movement
    daily_price_movement = df['high'] - df['low']
    
    # Calculate Price Gap
    price_gap = df['open'] - df['close']
    
    # Calculate Raw Momentum
    raw_momentum = (daily_price_movement + price_gap).rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    market_depth = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    adjusted_momentum = raw_momentum * (vwap + market_depth) / intraday_volatility
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    final_volume_adjusted_momentum = adjusted_momentum * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    close_5d_return = (df['close'] / df['close'].shift(5) - 1).fillna(0)
    close_10d_return = (df['close'] / df['close'].shift(10) - 1).fillna(0)
    multi_day_momentum = (close_5d_return + close_10d_return) / 2
    final_volume_adjusted_momentum += multi_day_momentum
    
    # Incorporate Liquidity
    liquidity_adjustment = avg_volume / total_avg_volume
    final_volume_adjusted_momentum *= liquidity_adjustment
    
    # Enhance with Dynamic Price and Volume Ratios
    price_to_volume_ratio = df['close'] / df['volume']
    moving_avg_pvr = price_to_volume_ratio.rolling(window=n).mean()
    avg_pvr = price_to_volume_ratio.mean()
    final_volume_adjusted_momentum *= (moving_avg_pvr / avg_pvr)
    
    # Consider Volatility Clustering
    ema_intraday_volatility = intraday_volatility.ewm(span=n, adjust=False).mean()
    avg_intraday_volatility = intraday_volatility.mean()
    final_volume_adjusted_momentum *= (ema_intraday_volatility / avg_intraday_volatility)
    
    return final_volume_adjusted_momentum
```
