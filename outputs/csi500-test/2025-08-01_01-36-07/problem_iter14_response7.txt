```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['Raw_Momentum_Price_Range'] = df['Price_Range'].rolling(window=n).sum()
    df['Raw_Momentum_Price_Gap'] = df['Price_Gap'].rolling(window=n).sum()
    df['Raw_Momentum'] = df['Raw_Momentum_Price_Range'] + df['Raw_Momentum_Price_Gap']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['Close_Volume'] = df['close'] * df['volume']
    df['VWAP'] = df['Close_Volume'].rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum
    df['Adjusted_Momentum'] = (df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth'])) / df['Intraday_Volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = avg_volume.mean()
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Returns
    df['5_Day_Return'] = (df['close'] / df['close'].shift(5) - 1)
    df['10_Day_Return'] = (df['close'] / df['close'].shift(10) - 1)
    df['Combined_Returns'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2
    df['Final_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Combined_Returns']
    
    # Incorporate Liquidity
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    total_average_volume = df['Average_Volume'].mean()
    df['Final_Factor'] = df['Final_Momentum'] * (df['Average_Volume'] / total_average_volume)
    
    return df['Final_Factor']
```
