```python
import pandas as pd

def heuristics_v2(df, n=20, n_days_5=5, n_days_10=10):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum (Sum of Daily Price Movements over n days)
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']
    
    # Weight Adjusted Momentum by Volume
    average_volume = df['volume'].rolling(window=n).mean()
    total_average_volume = df['volume'].mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (average_volume / total_average_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(n_days_5)) / df['close'].shift(n_days_5)
    df['10_day_return'] = (df['close'] - df['close'].shift(n_days_10)) / df['close'].shift(n_days_10)
    df['multi_day_return'] = (df['5_day_return'] + df['10_day_return']) / 2
    
    df['final_volume_adjusted_momentum'] += df['multi_day_return']
    
    # Incorporate Liquidity
    df['average_volume_n_days'] = df['volume'].rolling(window=n).mean()
    df['final_volume_adjusted_momentum'] *= (df['average_volume_n_days'] / total_average_volume)
    
    return df['final_volume_adjusted_momentum'].dropna()
```
