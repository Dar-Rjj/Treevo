```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = df['close'].diff(5)
    
    # Calculate Volume Change Ratio
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_change_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Compute Weighted Momentum
    df['weighted_momentum'] = df['daily_momentum'] * df['volume_change_ratio']
    
    # Calculate Cumulative Momentum
    df['cumulative_momentum'] = df['weighted_momentum'].rolling(window=20).sum()
    
    # Incorporate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['adjusted_cumulative_momentum'] = df['cumulative_momentum'] + df['high_low_range']
    
    # Return the factor values
    return df['adjusted_cumulative_momentum']
```
