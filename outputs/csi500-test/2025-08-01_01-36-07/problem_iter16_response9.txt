import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the relative range of the day
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / df['close'].shift(1)
    
    # Calculate the relative change in close price
    df['relative_change'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the relative volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine the factors to create the alpha factor
    df['alpha_factor'] = df['relative_range'] * df['relative_change'] * df['relative_volume']
    
    return df['alpha_factor']
