```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']

    # Calculate Raw Momentum (Sum of Daily Price Movements over n days)
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP and Intraday Volatility
    df['adjusted_momentum'] = (df['raw_momentum'] * df['vwap'] / df['intraday_volatility']).fillna(0)

    # Calculate Final Volume-Adjusted Momentum
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].expanding().mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (avg_volume_n_days / total_avg_volume)

    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] / df['close'].shift(5) - 1).fillna(0)
    df['10_day_return'] = (df['close'] / df['close'].shift(10) - 1).fillna(0)
    df['multi_day_return'] = (df['5_day_return'] + df['10_day_return']) / 2

    # Final Alpha Factor
    df['alpha_factor'] = df['final_volume_adjusted_momentum'] + df['multi_day_return']
    
    return df['alpha_factor']
```
