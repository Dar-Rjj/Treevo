```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Calculate Volume-Weighted Return
    df['prev_close'] = df['close'].shift(1)
    df['return'] = (df['close'] - df['prev_close']) / df['prev_close']
    df['volume_weighted_return'] = df['return'] * df['volume']
    
    # Combine Intraday Momentum and Volume-Weighted Return
    df['range_adjustment'] = df['intraday_range'] * df['close_to_open_ratio']
    df['combined_momentum'] = df['volume_weighted_return'] + df['range_adjustment']
    
    # Simplified Multi-Day Momentum
    df['5_day_simple_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Incorporate 5-Day Simple Return
    df['final_factor'] = df['combined_momentum'] + df['5_day_simple_return']
    
    return df['final_factor'].dropna()
```
