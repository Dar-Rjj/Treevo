import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the ratio of positive to negative money flow
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Calculate the weighted moving average of the close price
    wma_close = df['close'].rolling(window=14, min_periods=1).apply(lambda x: (x * range(1, 15)).sum() / sum(range(1, 15)), raw=True)
    
    # Calculate the alpha factor
    alpha_factor = (mfi - 50) * (wma_close - df['close'])
    
    return alpha_factor
