```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Range
    volume_weighted_intraday_range = intraday_range * df['volume']
    
    # Calculate Adjusted Close to Open Ratio
    adjusted_close_to_open_ratio = close_to_open_ratio * volume_weighted_intraday_range
    
    # Rolling 5-Day Sum of Adjusted Close to Open Ratio
    rolling_sum_adjusted_close_to_open = adjusted_close_to_open_ratio.rolling(window=5).sum()
    
    # Rolling 5-Day Average Intraday Range
    rolling_avg_intraday_range = intraday_range.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = rolling_sum_adjusted_close_to_open - rolling_avg_intraday_range
    
    return alpha_factor
```
