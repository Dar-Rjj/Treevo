import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the 10-day and 30-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the momentum factor as the difference between the 10-day and 30-day moving averages
    df['momentum_factor'] = df['ma_10'] - df['ma_30']
    
    # Calculate the 10-day standard deviation of the close price for a volatility measure
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate the normalized volume as a liquidity measure
    df['normalized_volume'] = (df['volume'] - df['volume'].rolling(window=30).min()) / (df['volume'].rolling(window=30).max() - df['volume'].rolling(window=30).min())
    
    # Define the novel alpha factor as a combination of the momentum, volatility, and liquidity measures
    df['alpha_factor'] = (df['momentum_factor'] / df['volatility']) * df['normalized_volume']
    
    # Fill any NaN values with 0 to ensure the series is complete
    df['alpha_factor'] = df['alpha_factor'].fillna(0)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
