```python
def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Apply Range Adjustment
    range_adjusted = intraday_range * (close_to_open_ratio - 1)
    
    # Volume-Weighted Adjusted Intraday Measure
    volume_weighted_adjusted = range_adjusted * df['volume']
    
    return volume_weighted_adjusted
```
