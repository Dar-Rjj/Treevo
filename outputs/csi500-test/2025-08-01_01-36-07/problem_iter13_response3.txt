import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n_days).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()
    df['vwap'] = vwap
    
    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n_days).mean()
    total_avg_volume = df['volume'].mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = df['close'].pct_change(periods=5)
    df['10_day_return'] = df['close'].pct_change(periods=10)
    df['multi_day_momentum'] = (df['5_day_return'] + df['10_day_return']) / 2
    df['final_volume_adjusted_momentum'] += df['multi_day_momentum']
    
    # Incorporate Liquidity
    avg_volume_n_days = df['volume'].rolling(window=n_days).mean()
    df['final_volume_adjusted_momentum'] *= (avg_volume_n_days / total_avg_volume)
    
    # Enhance with Dynamic Price and Volume Ratios
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    moving_avg_price_to_volume_ratio = df['price_to_volume_ratio'].rolling(window=n_days).mean()
    avg_price_to_volume_ratio = df['price_to_volume_ratio'].mean()
    df['final_volume_adjusted_momentum'] *= (moving_avg_price_to_volume_ratio / avg_price_to_volume_ratio)
    
    # Consider Volatility Clustering
    ema_intraday_volatility = df['intraday_volatility'].ewm(span=n_days).mean()
    avg_intraday_volatility = df['intraday_volatility'].mean()
    df['final_volume_adjusted_momentum'] *= (ema_intraday_volatility / avg_intraday_volatility)
    
    return df['final_volume_adjusted_momentum']
