import pandas as pd

def heuristics_v2(df):
    # Compute 20-Day Simple Moving Average (SMA) of Close Prices
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20-Day SMA from Close Price
    df['long_term_momentum'] = df['close'] - df['20_day_SMA']
    
    # Compute 5-Day Simple Moving Average (SMA) of Close Prices
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    
    # Subtract 5-Day SMA from Close Price
    df['short_term_momentum'] = df['close'] - df['5_day_SMA']
    
    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']
    
    # Calculate Volume Difference
    df['volume_difference'] = df['volume'].diff()
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Momentum and Volume Synchronization
    df['alpha_factor'] = df['combined_momentum'] * df['volume_ratio']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
