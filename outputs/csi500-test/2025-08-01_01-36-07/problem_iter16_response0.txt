```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']
    
    # Calculate Final Volume-Adjusted Momentum
    avg_volume = df['volume'].mean()
    df['Final_Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (df['volume'] / avg_volume)
    
    # Integrate Multi-Day Momentum
    df['5_Day_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_Day_Return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['Multi_Day_Return'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2
    df['Final_Volume_Adjusted_Momentum'] += df['Multi_Day_Return']
    
    # Incorporate Sophisticated Volatility Measures
    df['EMA_Intraday_Volatility'] = df['Intraday_Volatility'].ewm(span=n, adjust=False).mean()
    df['Historical_Volatility'] = df['close'].pct_change().rolling(window=n).std()
    df['Final_Volume_Adjusted_Momentum'] *= df['EMA_Intraday_Volatility'] / df['Historical_Volatility']
    
    # Enhance with Dynamic Price and Volume Ratios
    df['Price_to_Volume_Ratio'] = df['close'] / df['volume']
    df['MA_Price_to_Volume_Ratio'] = df['Price_to_Volume_Ratio'].rolling(window=n).mean()
    df['Final_Volume_Adjusted_Momentum'] *= df['MA_Price_to_Volume_Ratio'] / df['Price_to_Volume_Ratio'].mean()
    
    # Incorporate Dynamic Liquidity Adjustment
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Average_Volume_Change'] = df['volume'].diff().abs().rolling(window=n).mean()
    df['Final_Volume_Adjusted_Momentum'] *= df['Average_Volume'] / (df['Average_Volume'] + df['Average_Volume_Change'])
    
    # Final Alpha Factor
    alpha_factor = df['Final_Volume_Adjusted_Momentum']
    
    return alpha_factor
```
