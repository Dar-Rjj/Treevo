```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative position of close price within the day's range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the ratio of volume to amount
    df['vol_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the weighted average of close_position and vol_amount_ratio
    # The idea is that if the close price is at the higher end of the day's range and the volume is high, it might indicate a strong buying pressure
    df['weighted_avg'] = 0.7 * df['close_position'] + 0.3 * df['vol_amount_ratio']
    
    # Calculate the exponential moving average of the weighted average
    df['ewm_weighted_avg'] = df['weighted_avg'].ewm(span=5).mean()
    
    # The final alpha factor is the difference between the current weighted average and its EMA
    # This can be interpreted as a momentum factor with a mean-reverting component
    alpha_factor = df['weighted_avg'] - df['ewm_weighted_avg']
    
    return alpha_factor
```
