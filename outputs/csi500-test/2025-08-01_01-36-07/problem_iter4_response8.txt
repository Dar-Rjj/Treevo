import pandas as pd

def heuristics_v2(df, n=10, m=30):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum() + df['price_gap'].rolling(window=n).sum()
    
    # Calculate Volume-Adjusted Momentum
    avg_volume = df['volume'].rolling(window=n).mean()
    total_avg_volume = avg_volume.mean()
    df['volume_adjusted_momentum'] = df['raw_momentum'] * (avg_volume / total_avg_volume)
    
    # Incorporate Multi-Day Momentum
    df['multi_day_momentum'] = df['close'].pct_change(periods=m)
    df['combined_momentum'] = df['volume_adjusted_momentum'] + df['multi_day_momentum']
    
    # Analyze Price Trend
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['trend'] = df.apply(lambda row: 'positive' if row['close'] > row['50_day_ma'] else 'negative', axis=1)
    
    # Generate Alpha Factor
    def generate_alpha_factor(row):
        if row['trend'] == 'positive':
            return row['volume_adjusted_momentum'] * row['multi_day_momentum']
        else:
            return row['volume_adjusted_momentum'] / row['multi_day_momentum']
    
    df['alpha_factor'] = df.apply(generate_alpha_factor, axis=1)
    
    return df['alpha_factor']
