```python
import pandas as pd

def heuristics_v2(df, momentum_days=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Apply Range Adjustment
    range_adjusted = intraday_range * close_to_open_ratio
    
    # Weighted Difference
    weighted_difference = range_adjusted * df['volume']
    
    # Calculate Multi-Day Momentum
    daily_return = df['close'].diff(1)
    multi_day_momentum = daily_return.rolling(window=momentum_days).sum()
    
    # Integrate Intraday and Multi-Day Factors
    alpha_factor = weighted_difference + multi_day_momentum
    
    return alpha_factor
```
