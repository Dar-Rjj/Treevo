```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flows
    delta = df['close'].diff()
    positive_flow = money_flow.where(delta > 0, 0).rolling(window=14).sum()
    negative_flow = money_flow.where(delta < 0, 0).rolling(window=14).sum().abs()
    
    # Calculate the money flow ratio
    money_ratio = positive_flow / negative_flow
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the accumulation/distribution line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl *= df['volume']
    adl = adl.cumsum()
    
    # Combine MFI and ADL into a single factor
    factor = (mfi - 50) * (adl / adl.rolling(window=14).std())
    
    return factor
```
