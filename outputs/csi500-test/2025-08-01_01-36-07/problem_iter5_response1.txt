```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Volume-Weighted Adjustment
    weighted_intraday_range = intraday_range * df['volume']
    
    # Combined Factor
    combined_factor = weighted_intraday_range * close_to_open_ratio
    
    return combined_factor
```
