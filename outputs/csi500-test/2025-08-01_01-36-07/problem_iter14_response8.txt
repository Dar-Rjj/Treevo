import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, multi_day_n=[5, 10], vol_cluster_n=20):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']

    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Incorporate Market Depth
    df['market_depth'] = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['adjusted_momentum'] = df['raw_momentum'] * (df['vwap'] + df['market_depth']) / df['intraday_volatility']

    # Calculate Final Volume-Adjusted Momentum
    average_volume = df['volume'].rolling(window=n).mean()
    total_average_volume = df['volume'].mean()
    df['final_volume_adjusted_momentum'] = df['adjusted_momentum'] * (average_volume / total_average_volume)

    # Integrate Multi-Day Momentum
    for days in multi_day_n:
        df[f'{days}_day_return'] = (df['close'] - df['close'].shift(days)) / df['close'].shift(days)
    df['multi_day_return'] = df[[f'{days}_day_return' for days in multi_day_n]].mean(axis=1)
    df['final_volume_adjusted_momentum'] += df['multi_day_return']

    # Incorporate Liquidity
    df['final_volume_adjusted_momentum'] *= (average_volume / total_average_volume)

    # Enhance with Dynamic Price and Volume Ratios
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    df['moving_avg_price_to_volume_ratio'] = df['price_to_volume_ratio'].rolling(window=n).mean()
    average_price_to_volume_ratio = df['price_to_volume_ratio'].mean()
    df['final_volume_adjusted_momentum'] *= (df['moving_avg_price_to_volume_ratio'] / average_price_to_volume_ratio)

    # Consider Volatility Clustering
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=vol_cluster_n, adjust=False).mean()
    average_intraday_volatility = df['intraday_volatility'].mean()
    df['final_volume_adjusted_momentum'] *= (df['ema_intraday_volatility'] / average_intraday_volatility)

    # Final Alpha Factor
    final_alpha_factor = df['final_volume_adjusted_momentum']
    
    return final_alpha_factor
