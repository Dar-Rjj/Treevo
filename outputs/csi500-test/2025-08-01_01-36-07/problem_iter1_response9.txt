```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range Return
    high_to_low_range_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Directional Movement
    open_to_close_direction = df['close'] - df['open']
    
    # Combine High-to-Low Range and Open-to-Close Direction
    combined_indicator = high_to_low_range_return + open_to_close_direction
    
    # Weight by Volume and Amount
    weighted_indicator = combined_indicator * (df['volume'] + 0.5 * df['amount'])
    
    return weighted_indicator
```
