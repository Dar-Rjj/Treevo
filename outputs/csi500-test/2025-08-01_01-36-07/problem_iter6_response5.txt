```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']

    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']

    # Calculate Final Volume-Adjusted Momentum
    average_volume = df['volume'].mean()
    total_average_volume = df['volume'].mean()
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (average_volume / total_average_volume)

    # Integrate Multi-Day Momentum
    df['5_Day_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_Day_Return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['Average_Return'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2

    # Combine with Final Volume-Adjusted Momentum
    df['Final_Factor'] = df['Volume_Adjusted_Momentum'] + df['Average_Return']

    return df['Final_Factor'].dropna()
```
