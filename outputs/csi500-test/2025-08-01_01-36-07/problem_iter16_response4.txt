import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow ratio for each day
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Create a column for the positive and negative money flow
    df['positive_money_flow'] = df['money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    
    # Calculate the money flow ratio
    window = 14  # Typical window size, can be adjusted
    df['positive_money_flow_sum'] = df['positive_money_flow'].rolling(window=window).sum()
    df['negative_money_flow_sum'] = df['negative_money_flow'].rolling(window=window).sum()
    df['money_flow_ratio'] = df['positive_money_flow_sum'] / df['negative_money_flow_sum']
    
    # Heuristic alpha factor: money flow momentum
    df['alpha_factor'] = 100 - (100 / (1 + df['money_flow_ratio']))
    
    return df['alpha_factor']
