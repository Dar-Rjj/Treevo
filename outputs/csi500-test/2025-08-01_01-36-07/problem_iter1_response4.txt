```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Open Momentum
    close_to_open_momentum = df['close'] - df['open']
    
    # Compute Intraday Range and Normalize by Close Price
    intraday_range = (df['high'] - df['low']) / df['close']
    adjusted_daily_momentum = close_to_open_momentum / (1 + intraday_range)
    
    # Compute Volume Weight with 5-day Moving Average
    volume_ma = df['volume'].rolling(window=5).mean()
    volume_weight = df['volume'] / volume_ma
    
    # Multiply by Adjusted Daily Momentum
    factor_values = adjusted_daily_momentum * volume_weight
    
    return factor_values
```
