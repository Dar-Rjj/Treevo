```python
import pandas as pd

def heuristics_v2(df, n_days=20):
    # Calculate Daily Price Movement
    daily_price_movement = df['high'] - df['low']
    
    # Calculate Raw Momentum
    raw_momentum = daily_price_movement.rolling(window=n_days).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    market_depth = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Raw Momentum
    adjusted_momentum = (raw_momentum * (vwap + market_depth) / intraday_volatility)
    
    # Calculate Final Volume-Adjusted Momentum
    total_avg_volume = df['volume'].mean()
    final_vol_adj_momentum = adjusted_momentum * (df['volume'].rolling(window=n_days).mean() / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    five_day_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    multi_day_momentum = (five_day_return + ten_day_return) / 2
    final_vol_adj_momentum += multi_day_momentum
    
    # Incorporate Liquidity
    avg_volume = df['volume'].rolling(window=n_days).mean()
    final_alpha_factor = final_vol_adj_momentum * (avg_volume / total_avg_volume)
    
    return final_alpha_factor
```
