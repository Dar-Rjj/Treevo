```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, vwap_days=30, multi_day_momentum_days=(5, 10), avg_volume_days=30):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']

    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n_days).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=vwap_days).sum() / df['volume'].rolling(window=vwap_days).sum()

    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']

    # Calculate Final Volume-Adjusted Momentum
    df['Final_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (df['volume'].rolling(window=avg_volume_days).mean() / df['volume'].rolling(window=avg_volume_days, min_periods=1).mean().mean())

    # Integrate Multi-Day Momentum
    df['5D_Return'] = (df['close'] / df['close'].shift(multi_day_momentum_days[0]) - 1)
    df['10D_Return'] = (df['close'] / df['close'].shift(multi_day_momentum_days[1]) - 1)
    df['Multi_Day_Momentum'] = (df['5D_Return'] + df['10D_Return']) / 2

    df['Final_Adjusted_Momentum'] += df['Multi_Day_Momentum']

    # Incorporate Liquidity
    df['Average_Volume'] = df['volume'].rolling(window=avg_volume_days).mean()
    df['Final_Adjusted_Momentum'] *= (df['Average_Volume'] / df['volume'].rolling(window=avg_volume_days, min_periods=1).mean().mean())

    # Enhance with Dynamic Price and Volume Ratios
    df['PVR'] = df['close'] / df['volume']
    df['MA_PVR'] = df['PVR'].rolling(window=avg_volume_days).mean()
    df['Final_Adjusted_Momentum'] *= (df['MA_PVR'] / df['PVR'].mean())

    # Consider Volatility Clustering
    df['EMA_Volatility'] = df['Intraday_Volatility'].ewm(span=avg_volume_days, adjust=False).mean()
    df['Final_Adjusted_Momentum'] *= (df['EMA_Volatility'] / df['Intraday_Volatility'].mean())

    # Final Alpha Factor
    alpha_factor = df['Final_Adjusted_Momentum'].dropna()

    return alpha_factor
```
