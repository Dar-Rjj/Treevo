```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_span=5):
    # Calculate Daily Price Movement
    daily_price_movement = df['high'] - df['low']
    
    # Calculate Price Gap
    price_gap = df['open'] - df['close']
    
    # Calculate Raw Momentum
    raw_momentum = daily_price_movement.rolling(window=n).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Incorporate Market Depth
    market_depth = df['amount'] / df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Raw Momentum
    adjusted_momentum = (raw_momentum * (vwap + market_depth) / intraday_volatility).fillna(0)
    
    # Calculate Final Volume-Adjusted Momentum
    total_avg_volume = df['volume'].mean()
    avg_volume = df['volume'].rolling(window=n).mean()
    final_volume_adjusted_momentum = adjusted_momentum * (avg_volume / total_avg_volume)
    
    # Integrate Multi-Day Momentum
    multi_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    final_volume_adjusted_momentum += multi_day_momentum
    
    # Incorporate Liquidity
    liquidity = avg_volume / total_avg_volume
    final_volume_adjusted_momentum *= liquidity
    
    # Consider Volatility Clustering
    ema_intraday_volatility = intraday_volatility.ewm(span=ema_span, adjust=False).mean()
    avg_intraday_volatility = intraday_volatility.mean()
    volatility_clustering = ema_intraday_volatility / avg_intraday_volatility
    final_volume_adjusted_momentum *= volatility_clustering
    
    return final_volume_adjusted_momentum
```
