```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = (intraday_range * close_to_open_ratio) * df['volume']
    
    # Multi-Day Price Momentum
    multi_day_price_momentum = df['close'].diff()
    
    # Multi-Day Relative Strength
    multi_day_relative_strength = df['close'] / df['close'].shift(5)
    
    # Multi-Day Volatility
    daily_return = df['close'].pct_change()
    multi_day_volatility = (daily_return ** 2).rolling(window=5).sum()
    
    # Multi-Day Volume Ratio
    multi_day_volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Combine Factors
    combined_factor = (
        volume_weighted_intraday_momentum +
        multi_day_price_momentum +
        multi_day_relative_strength -
        multi_day_volatility +
        multi_day_volume_ratio
    )
    
    return combined_factor
```
