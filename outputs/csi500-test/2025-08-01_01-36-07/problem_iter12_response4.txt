```python
import pandas as pd

def heuristics_v2(df, n_days=10, n_volume_days=30):
    # Calculate Daily Price Movement
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Gap
    df['price_gap'] = df['open'] - df['close']
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_range'].rolling(window=n_days).sum()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()
    
    # Adjust Raw Momentum with VWAP
    df['adjusted_momentum'] = df['raw_momentum'] * df['vwap']
    
    # Calculate Final Volume-Adjusted Momentum
    total_volume = df['volume'].rolling(window=n_days).sum()
    df['final_vol_adjusted_momentum'] = df['adjusted_momentum'] * (df['volume'] / total_volume)
    
    # Integrate Multi-Day Momentum
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['multi_day_return'] = (df['5_day_return'] + df['10_day_return']) / 2
    df['final_vol_adjusted_momentum'] += df['multi_day_return']
    
    # Incorporate Liquidity
    avg_volume = df['volume'].rolling(window=n_volume_days).mean()
    total_avg_volume = avg_volume.rolling(window=n_days).sum()
    df['final_factor'] = df['final_vol_adjusted_momentum'] * (avg_volume / total_avg_volume)
    
    return df['final_factor'].dropna()
```
