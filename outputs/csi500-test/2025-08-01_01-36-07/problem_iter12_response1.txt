```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average of close price for 5 days
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the exponential moving average of volume for 10 days
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Create a composite factor
    alpha_factor = (sma_5 * pct_change_close) / (ema_10_volume + price_range)
    
    return alpha_factor
```
