```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Daily Price Movement
    df['Price_Range'] = df['high'] - df['low']

    # Calculate Price Gap
    df['Price_Gap'] = df['open'] - df['close']

    # Calculate Raw Momentum (Sum of Daily Price Movements over n days)
    df['Raw_Momentum'] = df['Price_Range'].rolling(window=n).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Incorporate Market Depth
    df['Market_Depth'] = df['amount'] / df['volume']

    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']

    # Adjust Raw Momentum with VWAP, Market Depth, and Intraday Volatility
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (df['VWAP'] + df['Market_Depth']) / df['Intraday_Volatility']

    # Calculate Final Volume-Adjusted Momentum
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    total_avg_volume = df['volume'].mean()
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * (avg_volume_n_days / total_avg_volume)

    # Integrate Multi-Day Momentum
    df['5_Day_Return'] = df['close'].pct_change(periods=5)
    df['10_Day_Return'] = df['close'].pct_change(periods=10)
    df['Multi_Day_Momentum'] = (df['5_Day_Return'] + df['10_Day_Return']) / 2
    df['Final_Momentum'] = df['Volume_Adjusted_Momentum'] + df['Multi_Day_Momentum']

    # Integrate Long-Term Momentum
    df['20_Day_Return'] = df['close'].pct_change(periods=20)
    df['50_Day_Return'] = df['close'].pct_change(periods=50)
    df['Long_Term_Momentum'] = (df['20_Day_Return'] + df['50_Day_Return']) / 2
    df['Final_Momentum'] += df['Long_Term_Momentum']

    # Incorporate Liquidity
    df['Final_Momentum'] *= (avg_volume_n_days / total_avg_volume)

    return df['Final_Momentum']
```
