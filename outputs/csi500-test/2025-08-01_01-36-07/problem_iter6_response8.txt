```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Weighted Intraday Difference
    intraday_range_adjustment = intraday_range * close_to_open_ratio
    weighted_intraday_difference = intraday_range_adjustment * df['volume']
    
    # Calculate Multi-Day Momentum
    daily_return = df['close'].diff()
    multi_day_momentum = daily_return.rolling(window=5).sum()
    
    # Integrate Intraday and Multi-Day Factors
    integrated_factor = weighted_intraday_difference + multi_day_momentum
    
    # Adjust for Liquidity
    liquidity_adjustment = (df['amount'] * 0.5) + df['volume']
    
    # Final Factor
    final_factor = integrated_factor * liquidity_adjustment
    
    return final_factor
```
