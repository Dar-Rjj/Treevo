```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Average Intraday Return over 10 days
    df['avg_intraday_return_10'] = df['intraday_return'].rolling(window=10).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility_10'] = np.sqrt(df['squared_intraday_return'].rolling(window=10).sum())
    
    # Calculate Moving Average of Intraday Volatility
    df['ma_intraday_volatility_10'] = df['intraday_volatility_10'].rolling(window=10).mean()
    
    # Weight by Volume
    df['volume_weighted_volatility'] = df['ma_intraday_volatility_10'] * df['volume']
    
    # Incorporate Short-Term Price Momentum
    df['5_day_sma_close'] = df['close'].rolling(window=5).mean()
    df['short_term_momentum'] = df['close'] - df['5_day_sma_close']
    
    # Combine with Volume-Weighted Intraday Volatility
    df['combined_factor'] = df['short_term_momentum'] + df['volume_weighted_volatility']
    
    # Final Alpha Factor
    final_alpha_factor = df['combined_factor']
    
    return final_alpha_factor
```
