import pandas as pd

def heuristics_v2(df):
    # Compute Simple Moving Average of Close Price (SMA) for the past 14 days
    df['SMA_14'] = df['close'].rolling(window=14).mean()
    
    # Calculate Daily Price Momentum: Current Close Price - SMA
    df['Momentum'] = df['close'] - df['SMA_14']
    
    # Calculate High-Low Price Ratio: (High / Low) - 1
    df['High_Low_Ratio'] = (df['high'] / df['low']) - 1
    
    # Compute Rolling Sum of Volume over 5 days
    df['Volume_Sum_5'] = df['volume'].rolling(window=5).sum()
    
    # Adjust High-Low Ratio by Volume and Add Momentum
    df['Factor'] = (df['High_Low_Ratio'] * df['Volume_Sum_5']) + df['Momentum']
    
    # Drop rows with NaN values resulting from rolling calculations
    df.dropna(inplace=True)
    
    # Return the factor as a pandas Series
    return df['Factor']
