```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intra_return'] = df['intraday_return'] * df['volume']
    
    # Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_vol_weighted_intra_return'] = df['vol_weighted_intra_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=20).sum().apply(np.sqrt)
    
    # Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Combine Factors
    df['factor'] = df['ewma_vol_weighted_intra_return'] + df['sma_intraday_volatility']
    
    return df['factor']
```
