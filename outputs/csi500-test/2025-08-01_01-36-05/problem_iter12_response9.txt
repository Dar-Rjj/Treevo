```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Exponential Moving Average of Intraday Returns
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=20).sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume
    df['volume_weighted_ema_volatility'] = df['ema_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_ema_momentum'] = df['ema_intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_momentum'] = df['volume_weighted_ema_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['combined_factor'] = df['ema_intraday_volatility'] + df['ema_volume_weighted_momentum']
    
    # Final Adjustment
    df['final_adjustment'] = df['combined_factor'].diff().fillna(0)
    df['final_factor'] = df['combined_factor'] + df['final_adjustment']
    
    return df['final_factor']
```
