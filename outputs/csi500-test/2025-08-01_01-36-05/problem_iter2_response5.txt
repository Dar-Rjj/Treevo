import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Ratio
    intraday_price_ratio = (df['high'] / df['low']) - 1
    
    # Compute Rolling Sum of Volume over 5 days
    rolling_vol_sum = df['volume'].rolling(window=5).sum()
    
    # Compute Rolling Sum of (Price * Volume) over 5 days
    # Using close price for the calculation
    rolling_pv_sum = (df['close'] * df['volume']).rolling(window=5).sum()
    
    # Compute Volume-Weighted Average
    vwap = rolling_pv_sum / rolling_vol_sum
    
    # Multiply Intraday Price Ratio by Volume-Weighted Average
    factor_values = intraday_price_ratio * vwap
    
    return factor_values
