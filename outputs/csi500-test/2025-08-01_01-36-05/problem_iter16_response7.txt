```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_return_20d'] = df['squared_intraday_return'].rolling(window=20).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_intraday_return_20d'])
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility_sma'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(alpha=0.2).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'].ewm(alpha=0.1).mean()
    
    # Combine Factors
    df['final_factor'] = (
        df['ewma_volume_weighted_intraday_return'] +
        df['sma_intraday_volatility'] +
        df['recent_volatility_sma'] +
        df['volume_trend']
    )
    
    return df['final_factor']
```
