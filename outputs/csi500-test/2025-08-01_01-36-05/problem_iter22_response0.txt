import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Function to compute Adaptive Exponential Moving Average
    def aema(data, window, alpha=None):
        if alpha is None:
            alpha = 2 / (window + 1)
        ema = data.ewm(alpha=alpha).mean()
        return ema
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Return
    df['aema_volume_weighted_intraday_return'] = aema(df['volume_weighted_intraday_return'], 30)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].pow(2).rolling(window=30).sum().pow(0.5)
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=30).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility'] = df['intraday_return'].pow(2).rolling(window=10).mean().pow(0.5)
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['aema_volume_weighted_intraday_momentum'] = aema(df['volume_weighted_intraday_momentum'], 30)
    
    # Combine Factors
    final_factor = (
        df['aema_volume_weighted_intraday_return'] +
        df['sma_intraday_volatility'] +
        df['recent_volatility']
    )
    
    # Normalize Final Factor
    final_factor = (final_factor - final_factor.mean()) / final_factor.std()
    
    return final_factor
