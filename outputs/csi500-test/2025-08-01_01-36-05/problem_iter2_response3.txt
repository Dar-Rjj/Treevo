import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day Exponential Moving Average (EMA) of Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ema_10'] = df['log_returns'].ewm(span=10, adjust=False).mean()
    
    # Identify Long-Term Breakout Points
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    
    # Compute Long-Term Breakout Strength
    df['breakout_strength'] = (df['20_day_high'] - df['20_day_low']) * df['log_returns']
    
    # Volume-Weighted Final Factor
    df['vol_weighted_breakout'] = df['volume'] * df['breakout_strength']
    df['final_factor'] = df['vol_weighted_breakout'] + df['ema_10']
    
    return df['final_factor']
