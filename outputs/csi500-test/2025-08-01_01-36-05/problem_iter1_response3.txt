import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Incorporate Volume Impact
    volume_impact = intraday_range * df['volume']
    
    # Calculate Momentum Direction
    momentum_direction = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Apply Momentum Direction
    momentum_volume_impact = momentum_direction * volume_impact
    
    # Refine with Close-to-Previous Close Return
    df['previous_close'] = df['close'].shift(1)
    close_to_prev_close_return = (df['close'] - df['previous_close'])
    
    # Final Alpha Factor
    alpha_factor = close_to_prev_close_return * (df['close'] - df['open']) * intraday_range * df['volume']
    
    return alpha_factor
