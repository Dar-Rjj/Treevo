```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=20).sum().apply(np.sqrt)
    
    # Calculate Adaptive Volatility
    df['adaptive_volatility_20_days'] = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    df['adaptive_volatility_5_days'] = df['intraday_volatility'].ewm(span=5, adjust=False).mean()
    df['adaptive_volatility'] = df['adaptive_volatility_20_days'] + df['adaptive_volatility_5_days']

    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()

    # Combine Factors
    df['final_factor'] = (df['ema_volume_weighted_intraday_return'] / df['adaptive_volatility']) + df['ema_volume_weighted_intraday_momentum']

    return df['final_factor']
```
