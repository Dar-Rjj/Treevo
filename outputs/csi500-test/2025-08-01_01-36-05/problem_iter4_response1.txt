```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Multi-Period Price Momentum
    df['10_day_return'] = df['close'].pct_change(10)
    df['20_day_return'] = df['close'].pct_change(20)
    df['50_day_return'] = df['close'].pct_change(50)
    
    # Identify Breakout Points
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['breakout_strength'] = (df['10_day_high'] - df['10_day_low']) * df['10_day_return']
    
    # Calculate Adaptive Volatility
    log_returns = np.log(df['close']).diff()
    df['10_day_volatility'] = log_returns.rolling(window=10).std()
    df['20_day_volatility'] = log_returns.rolling(window=20).std()
    df['adaptive_volatility'] = 0.6 * df['10_day_volatility'] + 0.4 * df['20_day_volatility']
    
    # Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['volume'] * df['breakout_strength']
    
    # Incorporate Recent Price Trends
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['trend_strength'] = df['close'] - df['5_day_sma']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_factor'] - df['adaptive_volatility'] + df['trend_strength']
    
    return df['alpha_factor']
```
