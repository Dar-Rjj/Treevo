import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Adaptive Exponential Moving Average of Volume-Weighted Intraday Return
    def adaptive_ema(data, span):
        alpha = 2 / (1 + span)
        ema = data.ewm(alpha=alpha, adjust=False).mean()
        return ema
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=20).sum())
    
    # Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility_sma'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Adaptive Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['adaptive_ema_vol_weighted_momentum'] = adaptive_ema(df['vol_weighted_intraday_momentum'], 30)
    
    # Combine Factors
    df['final_factor'] = (
        adaptive_ema(df['vol_weighted_intraday_return'], 30) +
        df['sma_intraday_volatility'] +
        df['recent_volatility_sma']
    )
    
    return df['final_factor']
