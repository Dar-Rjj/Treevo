import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate EMA of Volume-Weighted Intraday Return (last 20 days)
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].pow(2).rolling(window=20).sum().pow(0.5)
    
    # Calculate SMA of Intraday Volatility (last 20 days)
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate EMA of Volume-Weighted Intraday Momentum (last 20 days)
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20).mean()
    
    # Adjust for Dynamic Volume Changes
    df['ema_volume'] = df['volume'].ewm(span=20).mean()
    df['volume_adjustment_factor'] = df['volume'] / df['ema_volume']
    
    # Combine Factors
    df['combined_factors'] = (df['ema_volume_weighted_intraday_return'] + 
                               df['sma_intraday_volatility'] + 
                               df['ema_volume_weighted_intraday_momentum']) * df['volume_adjustment_factor']
    
    return df['combined_factors']
