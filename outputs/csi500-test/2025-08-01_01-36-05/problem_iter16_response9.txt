import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].pow(2)
    df['intraday_volatility_sum'] = df['intraday_volatility'].rolling(window=30).sum()
    df['intraday_volatility_30d'] = df['intraday_volatility_sum'].pow(0.5)
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility_30d'].rolling(window=30).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility'] = df['intraday_volatility_30d'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=30, adjust=False).mean()
    
    # Dynamically Balance Factor Contributions
    # Assign Weights to Each Factor Based on Historical Performance
    weight_ema = 0.4  # Weight for EMA of Volume-Weighted Intraday Return
    weight_sma_volatility = 0.3  # Weight for SMA of Intraday Volatility
    weight_recent_volatility = 0.3  # Weight for Recent Volatility
    
    # Combine Weighted Factors to Form Final Factor
    df['final_factor'] = (
        weight_ema * df['ema_volume_weighted_intraday_return'] +
        weight_sma_volatility * df['sma_intraday_volatility'] +
        weight_recent_volatility * df['recent_volatility']
    )
    
    return df['final_factor']
