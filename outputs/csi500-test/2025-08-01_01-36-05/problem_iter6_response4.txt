```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust Intraday Return by Volume
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    
    # Calculate Average Volume-Adjusted Intraday Return
    avg_volume_adjusted_intraday_return = df['volume_adjusted_intraday_return'].mean()
    
    # Calculate Intraday Volatility
    df['squared_volume_adjusted_intraday_return'] = df['volume_adjusted_intraday_return'] ** 2
    intraday_volatility = np.sqrt(df['squared_volume_adjusted_intraday_return'].sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['squared_volume_adjusted_intraday_return'].ewm(span=20, adjust=False).mean().apply(np.sqrt)
    
    # Weight by Volume
    df['volume_weighted_ema_volatility'] = df['ema_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['volume_adjusted_intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['alpha_factor'] = df['ema_intraday_volatility'] + df['ema_volume_weighted_intraday_momentum']

    return df['alpha_factor']
```
