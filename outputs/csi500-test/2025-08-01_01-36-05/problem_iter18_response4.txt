```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=15, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=15).sum() ** 0.5
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=15).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=15, adjust=False).mean()
    
    # Integrate Liquidity Measures
    df['daily_volume_to_money'] = df['volume'] / df['amount']
    df['ewma_daily_volume_to_money'] = df['daily_volume_to_money'].ewm(span=15, adjust=False).mean()
    
    # Integrate Market Depth
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['daily_midpoint_difference'] = df['midpoint'].diff()
    df['ewma_daily_midpoint_difference'] = df['daily_midpoint_difference'].ewm(span=15, adjust=False).mean()
    
    # Dynamically Balance Factor Contributions
    # For simplicity, we assign equal weights to each factor
    weight_ewma_volume_weighted_intraday_return = 0.2
    weight_sma_intraday_volatility = 0.2
    weight_recent_volatility = 0.2
    weight_ewma_daily_volume_to_money = 0.2
    weight_ewma_daily_midpoint_difference = 0.2
    
    # Combine Factors
    df['final_factor'] = (
        weight_ewma_volume_weighted_intraday_return * df['ewma_volume_weighted_intraday_return'] +
        weight_sma_intraday_volatility * df['sma_intraday_volatility'] +
        weight_recent_volatility * df['recent_volatility'] +
        weight_ewma_daily_volume_to_money * df['ewma_daily_volume_to_money'] +
        weight_ewma_daily_midpoint_difference * df['ewma_daily_midpoint_difference']
    )
    
    return df['final_factor'].dropna()
```
