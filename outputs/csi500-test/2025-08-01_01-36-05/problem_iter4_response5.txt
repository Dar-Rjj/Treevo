import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Ratio
    intraday_price_ratio = (df['high'] / df['low']) - 1
    
    # Adjust by Volume-Weighted Average
    rolling_volume_sum = df['volume'].rolling(window=5).sum()
    rolling_close_volume_sum = (df['close'] * df['volume']).rolling(window=5).sum()
    volume_weighted_avg = rolling_close_volume_sum / rolling_volume_sum
    
    # Incorporate Price Momentum
    sma_14 = df['close'].rolling(window=14).mean()
    momentum = df['close'] - sma_14
    
    # Combine Intraday Price Ratio, Volume-Weighted Average, and Momentum
    factor_value = (intraday_price_ratio * volume_weighted_avg) + momentum
    
    return factor_value
