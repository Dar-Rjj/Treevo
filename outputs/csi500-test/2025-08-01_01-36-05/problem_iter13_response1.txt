```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Intraday Return
    intraday_return = data['high'] - data['low']
    
    # Calculate Volume-Weighted Intraday Return
    vol_weighted_intraday_return = intraday_return * data['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    ewma_vol_weighted_intraday_return = vol_weighted_intraday_return.ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = (intraday_return**2).rolling(window=20).sum().apply(np.sqrt)
    
    # Calculate Simple Moving Average of Intraday Volatility
    sma_intraday_volatility = intraday_volatility.rolling(window=20).mean()
    
    # Adjust for Recent Volatility
    recent_volatility = intraday_volatility.rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    vol_weighted_intraday_momentum = intraday_return * data['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    ewma_vol_weighted_intraday_momentum = vol_weighted_intraday_momentum.ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    factor = (
        ewma_vol_weighted_intraday_return + 
        sma_intraday_volatility + 
        recent_volatility + 
        ewma_vol_weighted_intraday_momentum
    )
    
    return factor
```
