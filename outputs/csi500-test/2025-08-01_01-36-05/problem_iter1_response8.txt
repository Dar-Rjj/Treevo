```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Subtract Closing Price from Open
    df['intraday_reversal'] = df['open'] - df['close']

    # Calculate Short-Term (e.g., 5-day) Logarithmic Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['short_term_momentum'] = df['log_returns'].rolling(window=5).mean()

    # Weigh Intraday Reversal by Volume
    df['weighted_intraday_reversal'] = df['intraday_reversal'] * df['volume']

    # Add Weighted Momentum, scaled by a fixed value
    fixed_value = 0.5
    df['weighted_momentum'] = fixed_value * df['short_term_momentum']

    # Combine Indicators
    df['alpha_factor'] = df['weighted_intraday_reversal'] + df['weighted_momentum']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor'].dropna()
```
