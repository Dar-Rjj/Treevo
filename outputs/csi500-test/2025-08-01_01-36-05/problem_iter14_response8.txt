```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Exponential Moving Average of Volume-Weighted Intraday Return (alpha=0.15, span=20)
    df['ema_vol_weighted_intraday_return'] = df['vol_weighted_intraday_return'].ewm(alpha=0.15, min_periods=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=20).sum().apply(np.sqrt)
    
    # Simple Moving Average of Intraday Volatility (window=20)
    df['sma_intraday_volatility_20'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Adjust for Recent Volatility (window=5)
    df['recent_volatility'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Exponential Moving Average of Volume-Weighted Intraday Momentum (alpha=0.15, span=20)
    df['ema_vol_weighted_intraday_momentum'] = df['vol_weighted_intraday_momentum'].ewm(alpha=0.15, min_periods=20).mean()
    
    # Incorporate Trade Amount
    df['amount_weighted_intraday_return'] = df['intraday_return'] * df['amount']
    df['ema_amount_weighted_intraday_return'] = df['amount_weighted_intraday_return'].ewm(alpha=0.15, min_periods=20).mean()
    
    df['amount_weighted_intraday_momentum'] = (df['close'] - df['open']) * df['amount']
    df['ema_amount_weighted_intraday_momentum'] = df['amount_weighted_intraday_momentum'].ewm(alpha=0.15, min_periods=20).mean()
    
    # Vary Lookback Periods
    df['sma_intraday_volatility_10'] = df['intraday_volatility'].rolling(window=10).mean()
    df['sma_intraday_volatility_30'] = df['intraday_volatility'].rolling(window=30).mean()
    
    # Experiment with Different Smoothing Techniques
    df['wma_vol_weighted_intraday_return_20'] = df['vol_weighted_intraday_return'].rolling(window=20, win_type='triang').mean()
    df['wma_vol_weighted_intraday_momentum_20'] = df['vol_weighted_intraday_momentum'].rolling(window=20, win_type='triang').mean()
    
    # Combine Factors
    df['final_factor'] = (
        df['ema_vol_weighted_intraday_return'] +
        df['sma_intraday_volatility_20'] +
        df['recent_volatility'] +
        df['sma_intraday_volatility_10'] +
        df['sma_intraday_volatility_30'] +
        df['ema_amount_weighted_intraday_return'] +
        df['ema_amount_weighted_intraday_momentum'] +
        df['wma_vol_weighted_intraday_return_20'] +
        df['wma_vol_weighted_intraday_momentum_20']
    )
    
    return df['final_factor']
```
