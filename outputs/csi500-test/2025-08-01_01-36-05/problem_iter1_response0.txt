```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day simple moving average (SMA) of Close price
    df['SMA_5_Close'] = df['close'].rolling(window=5).mean()
    
    # Compute short-term price momentum
    df['Price_Momentum'] = df['close'] - df['SMA_5_Close']
    
    # Calculate 20-day simple moving average of Volume
    df['SMA_20_Volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate 20-day standard deviation of Volume
    df['STD_20_Volume'] = df['volume'].rolling(window=20).std()
    
    # Determine if today's volume is more than 2 standard deviations above the 20-day SMA
    df['Volume_Shock'] = (df['volume'] > (df['SMA_20_Volume'] + 2 * df['STD_20_Volume']))
    
    # Combine Price Momentum with Volume Shock
    df['Combined_Factor'] = df.apply(
        lambda row: row['Price_Momentum'] * 2 if row['Volume_Shock'] else row['Price_Momentum'], 
        axis=1
    )
    
    return df['Combined_Factor']
```
