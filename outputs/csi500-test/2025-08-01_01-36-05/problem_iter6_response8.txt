```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Exponential Moving Average of Intraday Returns
    ema_intraday_return = df['intraday_return'].ewm(span=20, min_periods=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    intraday_volatility = np.sqrt(df['squared_intraday_return'].rolling(window=20, min_periods=20).sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    ema_intraday_volatility = intraday_volatility.ewm(span=20, min_periods=20).mean()
    
    # Weight by Volume
    volume_weighted_volatility = ema_intraday_volatility * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_momentum = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    ema_volume_weighted_momentum = volume_weighted_momentum.ewm(span=20, min_periods=20).mean()
    
    # Combine Factors
    factor_values = ema_intraday_volatility + ema_volume_weighted_momentum
    
    return factor_values
```
