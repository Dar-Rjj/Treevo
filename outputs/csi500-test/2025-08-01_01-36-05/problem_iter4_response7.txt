```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Average Intraday Return
    df['avg_intraday_return'] = df['intraday_return'].rolling(window=len(df), min_periods=1).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = np.sqrt((df['intraday_return'] ** 2).rolling(window=len(df), min_periods=1).sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Bid-Ask Spread
    df['bid_ask_spread'] = df['ask'] - df['bid']
    
    # Calculate Average Bid-Ask Spread
    df['avg_bid_ask_spread'] = df['bid_ask_spread'].rolling(window=len(df), min_periods=1).mean()
    
    # Combine Factors
    df['factor_value'] = (
        df['ema_intraday_volatility'] + 
        df['ema_volume_weighted_intraday_momentum'] + 
        df['avg_bid_ask_spread']
    )
    
    return df['factor_value']
```
