```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Average Volume-Weighted Intraday Return
    avg_volume_weighted_intraday_return = df['volume_weighted_intraday_return'].rolling(window=20).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'] ** 2
    intraday_volatility_sum = df['intraday_volatility'].sum()
    intraday_volatility = np.sqrt(intraday_volatility_sum)
    
    # Calculate Volume-Weighted Intraday Volatility
    df['volume_weighted_intraday_volatility'] = intraday_volatility * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Intraday Volatility
    moving_avg_volume_weighted_intraday_volatility = df['volume_weighted_intraday_volatility'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Intraday Momentum
    moving_avg_volume_weighted_intraday_momentum = df['volume_weighted_intraday_momentum'].rolling(window=20).mean()
    
    # Combine Factors
    combined_factor = (moving_avg_volume_weighted_intraday_volatility + 
                       moving_avg_volume_weighted_intraday_momentum)
    
    return combined_factor
```
