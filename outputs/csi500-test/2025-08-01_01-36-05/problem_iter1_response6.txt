```python
import pandas as pd
import numpy as np

def heuristics_v2(df, decay_factor=0.9, lookback_window=10):
    """
    Generate a novel and interpretable alpha factor based on dynamic momentum and reversal signal.
    
    Parameters:
    - df: pandas DataFrame with (date) as index and columns for open, high, low, close, amount, volume.
    - decay_factor: float, the decay factor for the exponential weighted moving average of intraday range.
    - lookback_window: int, the number of periods to use in the simple moving average for end-of-day price movement.
    
    Returns:
    - A pandas Series with (date) as index and the calculated alpha factor as values.
    """
    
    # Calculate Intraday Range Momentum
    df['DailyRange'] = df['high'] - df['low']
    df['IntradayMomentum'] = df['DailyRange'].ewm(alpha=(1-decay_factor), adjust=False).mean()
    
    # Incorporate End-of-Day Price Movement
    df['CloseToPrevCloseReturn'] = df['close'].pct_change()
    df['SmoothedReturn'] = df['CloseToPrevCloseReturn'].rolling(window=lookback_window).mean()
    
    # Combine Signals
    df['AlphaFactor'] = df['IntradayMomentum'] * df['SmoothedReturn']
    
    return df['AlphaFactor']
```
