```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return (last 30 days, alpha=0.1)
    df['ewm_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility (square root of sum of squared intraday returns over last 60 days)
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=60).sum() ** 0.5
    
    # Calculate Simple Moving Average of Intraday Volatility (last 60 days)
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=60).mean()
    
    # Adjust for Recent Volatility (SMA of last 10 days)
    df['sma_recent_volatility'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum (last 30 days, alpha=0.1)
    df['ewm_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=30, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = df['ewm_volume_weighted_intraday_return'] + df['sma_intraday_volatility'] + df['sma_recent_volatility']
    
    # Incorporate Liquidity and Relative Strength
    # Calculate Relative Strength
    df['close_returns'] = df['close'].pct_change()
    df['relative_strength'] = df['close_returns'].rolling(window=20).mean() / df['close_returns'].rolling(window=20).std()
    
    # Calculate Liquidity
    df['liquidity'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=20).std()
    
    # Adjust Final Factor
    df['adjusted_final_factor'] = df['final_factor'] * (df['relative_strength'] * df['liquidity'])
    
    return df['adjusted_final_factor']
```
