import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day return using close price
    df['10_day_return'] = df['close'].pct_change(periods=10)
    
    # Determine 10-day high and low
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    
    # Compute breakout strength
    df['breakout_strength'] = (df['10_day_high'] - df['10_day_low']) * df['10_day_return']
    
    # Volume adjusted factor
    df['volume_adjusted_factor'] = df['volume'] * df['breakout_strength']
    
    # Return the factor values as a pandas Series
    return df['volume_adjusted_factor'].dropna()
