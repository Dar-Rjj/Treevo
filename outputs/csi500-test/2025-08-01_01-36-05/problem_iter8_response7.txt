```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']

    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = (df['squared_intraday_return'].rolling(window=50).sum()) ** 0.5

    # Calculate Moving Average of Intraday Volatility
    df['ma_intraday_volatility'] = df['intraday_volatility'].rolling(window=50).mean()

    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']

    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']

    # Calculate Moving Average of Volume-Weighted Intraday Momentum
    df['ma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].rolling(window=50).mean()

    # Adjust for Dynamic Volume Changes
    df['ema_volume_50'] = df['volume'].ewm(span=50).mean()
    df['volume_adjustment_factor'] = df['volume'] / df['ema_volume_50']
    
    # Combine Factors
    df['combined_factors'] = (df['ma_intraday_volatility'] + df['ma_volume_weighted_intraday_momentum']) * df['volume_adjustment_factor']

    return df['combined_factors'].dropna()
```
