```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Exponential Moving Average of Intraday Returns
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=10).sum() ** 0.5
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=10, adjust=False).mean()
    
    # Weight by Volume
    df['vol_weighted_intraday_volatility'] = df['volume'] * df['ema_intraday_volatility']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = df['volume'] * df['ema_intraday_return']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_vol_weighted_intraday_momentum'] = df['vol_weighted_intraday_momentum'].ewm(span=10, adjust=False).mean()
    
    # Incorporate Trade Amount
    df['amount_adjusted_momentum'] = df['ema_vol_weighted_intraday_momentum'] / df['amount']
    
    # Adjust for Recent Volatility
    df['recent_ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=5, adjust=False).mean()
    
    # Combine Factors
    df['combined_factor'] = df['ema_intraday_volatility'] + df['amount_adjusted_momentum']
    
    # Final Adjustment
    df['final_factor'] = df['combined_factor'].diff() + df['combined_factor']
    
    return df['final_factor'].dropna()
```
