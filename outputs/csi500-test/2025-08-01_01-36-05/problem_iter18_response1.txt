```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Exponential Moving Average of Intraday Returns
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=5).sum().pow(0.5)
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=5, adjust=False).mean()
    
    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['ema_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['ema_intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=5, adjust=False).mean()
    
    # Incorporate Trade Amount
    df['adjusted_volume_weighted_intraday_momentum'] = df['ema_volume_weighted_intraday_momentum'] / df['amount']
    
    # Adjust for Recent Volatility
    df['recent_ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=3, adjust=False).mean()
    
    # Combine Factors
    df['combined_factor'] = df['ema_intraday_volatility'] + df['adjusted_volume_weighted_intraday_momentum']
    
    # Final Adjustment
    df['factor_difference'] = df['combined_factor'].diff()
    df['final_alpha_factor'] = df['combined_factor'] + df['factor_difference']
    
    return df['final_alpha_factor']
```
