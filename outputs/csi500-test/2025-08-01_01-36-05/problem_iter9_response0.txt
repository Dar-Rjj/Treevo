```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return (EMA)
    ema_volume_weighted_return = volume_weighted_intraday_return.ewm(span=20, min_periods=20).mean()
    
    # Calculate Intraday Volatility
    intraday_volatility = np.sqrt((intraday_return ** 2).rolling(window=20, min_periods=20).sum())
    
    # Calculate Moving Average of Intraday Volatility (SMA)
    sma_intraday_volatility = intraday_volatility.rolling(window=20, min_periods=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum (EMA)
    ema_volume_weighted_momentum = volume_weighted_intraday_momentum.ewm(span=20, min_periods=20).mean()
    
    # Combine Factors: EMA of Volume-Weighted Intraday Return and SMA of Intraday Volatility
    combined_factor = ema_volume_weighted_return - sma_intraday_volatility
    
    return combined_factor
```
