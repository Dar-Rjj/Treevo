```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust Intraday Return by Volume
    df['volume_adjusted_return'] = df['intraday_return'] / np.sqrt(df['volume'])
    
    # Calculate Average Volume-Adjusted Intraday Return
    df['average_volume_adjusted_return'] = df['volume_adjusted_return'].rolling(window=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_volume_adjusted_return'] = df['volume_adjusted_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_volume_adjusted_return'].rolling(window=20).sum())
    
    # Calculate Exponential Moving Average (EMA) of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume
    df['volume_weighted_ema_intraday_volatility'] = df['ema_intraday_volatility'] * np.sqrt(df['volume'])
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['volume_adjusted_return'] * np.sqrt(df['volume'])
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['combined_alpha_factor'] = df['ema_intraday_volatility'] + df['ema_volume_weighted_intraday_momentum']
    
    return df['combined_alpha_factor']
```
