```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Exponential Moving Average of Volume-Weighted Intraday Return (last 30 days, alpha=0.1)
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility (last 60 days)
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=60).sum())
    
    # Exponential Moving Average of Intraday Volatility (last 60 days, alpha=0.1)
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=60, adjust=False).mean()
    
    # Adjust for Recent Volatility (last 10 days, alpha=0.1)
    df['recent_volatility'] = df['intraday_volatility'].ewm(span=10, adjust=False).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Exponential Moving Average of Volume-Weighted Intraday Momentum (last 30 days, alpha=0.1)
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=30, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = (
        df['ema_volume_weighted_intraday_return'] 
        + df['ema_intraday_volatility'] 
        + df['recent_volatility']
        + df['ema_volume_weighted_intraday_momentum']
    )
    
    return df['final_factor']
```
