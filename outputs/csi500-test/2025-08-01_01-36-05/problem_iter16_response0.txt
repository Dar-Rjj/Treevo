```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Exponential Moving Average of Intraday Returns (EMA IR)
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=10, adjust=False).mean()

    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=10).sum())

    # Calculate EMA of Intraday Volatility (EMA IV)
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=10, adjust=False).mean()

    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['ema_intraday_volatility'] * df['volume']

    # Calculate Volume-Weighted Intraday Momentum (VWIM)
    df['volume_weighted_intraday_momentum'] = df['ema_intraday_return'] * df['volume']

    # Calculate EMA of VWIM
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=10, adjust=False).mean()

    # Incorporate Trade Amount
    df['adjusted_ema_volume_weighted_intraday_momentum'] = df['ema_volume_weighted_intraday_momentum'] / df['amount']

    # Adjust for Recent Volatility
    df['recent_ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=5, adjust=False).mean()

    # Combine Factors
    df['combined_factor'] = df['ema_intraday_volatility'] + df['adjusted_ema_volume_weighted_intraday_momentum']

    # Final Adjustment
    df['final_adjustment'] = (df['combined_factor'] - df['combined_factor'].shift(1)) + df['combined_factor']

    return df['final_adjustment']
```
