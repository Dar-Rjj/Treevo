```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_return'] = df['squared_intraday_return'].rolling(window=20).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_intraday_return'])
    
    # Calculate Simple Moving Average (SMA) of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate EMA of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['combined_alpha_factor'] = (df['ema_volume_weighted_intraday_return'] + df['ema_volume_weighted_intraday_momentum']) / df['sma_intraday_volatility']
    
    return df['combined_alpha_factor']
```
