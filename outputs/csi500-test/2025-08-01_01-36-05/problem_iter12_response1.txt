```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=5).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=5).sum())
    
    # Calculate Exponential Moving Average (EMA) of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=5).mean()
    
    # Calculate Market Sentiment
    df['market_sentiment'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Exponential Moving Average (EMA) of Market Sentiment
    df['ema_market_sentiment'] = df['market_sentiment'].ewm(span=5).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['ema_volume_weighted_intraday_return'] + 
                          df['ema_intraday_volatility'] + 
                          df['ema_volume_weighted_intraday_momentum'] + 
                          df['ema_market_sentiment'])
    
    return df['alpha_factor']
```
