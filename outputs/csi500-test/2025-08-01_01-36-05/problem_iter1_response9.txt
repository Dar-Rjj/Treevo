```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    high_low_ratio = df['high'] / df['low'] - 1
    
    # Compute Rolling Sum of Volume over 5 days
    rolling_volume_sum = df['volume'].rolling(window=5).sum()
    
    # Adjust the High-Low ratio by the rolling sum of volume
    adjusted_ratio = high_low_ratio * rolling_volume_sum
    
    return adjusted_ratio
```
