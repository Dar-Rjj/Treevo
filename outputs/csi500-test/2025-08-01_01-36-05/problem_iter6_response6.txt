```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']

    # Calculate Intraday Volatility
    df['squared_intraday_returns'] = df['intraday_return'] ** 2
    intraday_volatility = np.sqrt(df['squared_intraday_returns'].rolling(window=20, min_periods=1).sum())

    # Calculate Moving Average of Intraday Volatility
    df['moving_avg_intraday_volatility'] = intraday_volatility.rolling(window=20, min_periods=1).mean()

    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['squared_intraday_returns'] * df['volume']
    volume_weighted_intraday_volatility = np.sqrt(df['volume_weighted_intraday_volatility'].rolling(window=20, min_periods=1).sum())

    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']

    # Calculate Moving Average of Volume-Weighted Intraday Momentum
    df['moving_avg_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].rolling(window=20, min_periods=1).mean()

    # Combine Factors
    df['alpha_factor'] = (df['moving_avg_intraday_volatility'] + 
                          df['moving_avg_volume_weighted_intraday_momentum'])

    return df['alpha_factor']
```
