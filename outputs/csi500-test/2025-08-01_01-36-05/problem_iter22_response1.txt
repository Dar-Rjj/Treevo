import pandas as pd
import numpy as np

def adaptive_ema(data, span, alpha=None):
    if alpha is None:
        alpha = 2 / (span + 1)
    ema = data.ewm(alpha=alpha, adjust=False).mean()
    return ema

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Return
    span = 30
    recent_volatility = intraday_return.rolling(window=10).std()
    dynamic_alpha = 2 / (span + 1) * (1 + recent_volatility)
    ema_volume_weighted_return = adaptive_ema(volume_weighted_intraday_return, span, alpha=dynamic_alpha)
    
    # Calculate Intraday Volatility
    squared_intraday_return = intraday_return**2
    intraday_volatility = np.sqrt(squared_intraday_return.rolling(window=30).sum())
    
    # Calculate Simple Moving Average of Intraday Volatility
    sma_intraday_volatility = intraday_volatility.rolling(window=30).mean()
    
    # Adjust for Recent Volatility
    sma_recent_volatility = intraday_volatility.rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Momentum
    ema_volume_weighted_momentum = adaptive_ema(volume_weighted_intraday_momentum, span, alpha=dynamic_alpha)
    
    # Combine Factors
    final_factor = (ema_volume_weighted_return + sma_intraday_volatility - sma_recent_volatility + ema_volume_weighted_momentum)
    
    # Normalize Final Factor (if necessary)
    final_factor = (final_factor - final_factor.mean()) / final_factor.std()
    
    return final_factor
