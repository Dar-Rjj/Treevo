```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Ratio
    high_low_ratio = (df['high'] / df['low']) - 1
    
    # Adjust by Volume: Multiply the high-low ratio by the rolling sum of volume
    rolling_volume_sum = df['volume'].rolling(window=5).sum()
    adjusted_high_low_ratio = high_low_ratio * rolling_volume_sum
    
    # Incorporate Historical Price Trend: Compute 5-day Moving Average of Close prices
    moving_avg_5d = df['close'].rolling(window=5).mean()
    
    # Compute Deviation from 5-day Moving Average
    deviation_from_moving_avg = df['close'] - moving_avg_5d
    
    # Combine Adjusted High-Low Ratio and Historical Trend
    factor_values = adjusted_high_low_ratio * deviation_from_moving_avg
    
    return factor_values
```
