```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewa_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=50, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=40).sum().apply(np.sqrt)
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=40).mean()
    
    # Adjust for Recent Volatility
    df['recent_sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=15).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewa_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=50, adjust=False).mean()
    
    # Dynamically Balance Factor Contributions
    weight_ewa_return = 0.4
    weight_sma_volatility = 0.3
    weight_recent_volatility = 0.3
    
    df['final_factor'] = (
        weight_ewa_return * df['ewa_volume_weighted_intraday_return'] +
        weight_sma_volatility * df['sma_intraday_volatility'] +
        weight_recent_volatility * df['recent_sma_intraday_volatility']
    )
    
    return df['final_factor']
```
