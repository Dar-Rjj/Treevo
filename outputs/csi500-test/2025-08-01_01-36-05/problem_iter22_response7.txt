```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=80, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_return_80d'] = df['squared_intraday_return'].rolling(window=80).sum()
    df['intraday_volatility_80d'] = np.sqrt(df['sum_squared_intraday_return_80d'])
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility_80d'] = df['intraday_volatility_80d'].rolling(window=80).mean()
    
    # Adjust for Recent Volatility
    df['sma_recent_volatility_30d'] = df['intraday_volatility_80d'].rolling(window=30).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=80, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = (df['ewma_volume_weighted_intraday_return'] 
                          + df['sma_intraday_volatility_80d'] 
                          + df['sma_recent_volatility_30d'])
    
    return df['final_factor']
```
