```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Ratio
    intraday_price_ratio = (df['high'] / df['low']) - 1
    
    # Compute Rolling Sum of Volume over 5 days
    rolling_vol_sum = df['volume'].rolling(window=5).sum()
    
    # Compute Rolling Sum of (Price * Volume) over 5 days
    # Using the close price for this calculation
    rolling_price_vol_sum = (df['close'] * df['volume']).rolling(window=5).sum()
    
    # Compute Dynamic Volume-Weighted Average
    dynamic_vol_weighted_avg = rolling_price_vol_sum / rolling_vol_sum
    
    # Adjust Intraday Price Ratio by Dynamic Volume-Weighted Average
    adjusted_factor = intraday_price_ratio * dynamic_vol_weighted_avg
    
    return adjusted_factor
```
