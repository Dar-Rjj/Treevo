```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']

    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_vol_weighted_intraday_return'] = df['vol_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=70).sum())
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=70).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility'] = df['intraday_volatility'].rolling(window=15).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_vol_weighted_intraday_momentum'] = df['vol_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = (df['ema_vol_weighted_intraday_return'] 
                          + df['sma_intraday_volatility'] 
                          + df['recent_volatility'])
    
    # Drop intermediate columns if not needed
    final_factor = df['final_factor'].copy()
    return final_factor
```
