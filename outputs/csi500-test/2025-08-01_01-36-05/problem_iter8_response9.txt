```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Adjust Intraday Return by Volume
    df['volume_adjusted_intraday_return'] = df['intraday_return'] / df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Adjusted Intraday Return
    df['ema_vol_adj_intraday_return'] = df['volume_adjusted_intraday_return'].ewm(span=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_vol_adj_intraday_return'] = df['volume_adjusted_intraday_return'] ** 2
    df['sum_squared_vol_adj_intraday_return'] = df['squared_vol_adj_intraday_return'].rolling(window=20).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_vol_adj_intraday_return'])
    
    # Calculate Exponential Moving Average (EMA) of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=20).mean()
    
    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['ema_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['volume_adjusted_intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20).mean()
    
    # Combine Factors
    df['factor_value'] = df['ema_intraday_volatility'] + df['ema_volume_weighted_intraday_momentum']
    
    return df['factor_value']
```
