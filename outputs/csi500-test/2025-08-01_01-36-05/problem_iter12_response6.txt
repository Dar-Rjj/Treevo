import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = np.sqrt((df['intraday_return'] ** 2).rolling(window=20).sum())
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Integrate High-Frequency Data (This step is skipped in this example, but you can add it if you have high-frequency data)
    
    # Optimize Moving Average Periods (This step is skipped in this example, but you can experiment with different periods)
    
    # Consider Market Trends (Assume we have a market trend indicator in the DataFrame)
    df['market_trend'] = np.where(df['close'].pct_change() > 0, 1, 0)  # 1 for bullish, 0 for bearish
    
    # Adjust Factors Based on Market Trend
    df['adjusted_ema'] = df['ema_volume_weighted_intraday_return'] * (1 + df['market_trend'] * 0.1)
    df['adjusted_sma'] = df['sma_intraday_volatility'] * (1 - df['market_trend'] * 0.1)
    
    # Combine Factors
    df['final_factor'] = (df['adjusted_ema'] - df['adjusted_sma']) + df['recent_volatility']
    
    return df['final_factor']
