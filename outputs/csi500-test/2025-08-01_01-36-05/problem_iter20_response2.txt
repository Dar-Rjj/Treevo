import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_return_60d'] = df['squared_intraday_return'].rolling(window=60).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_intraday_return_60d'])
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility_60d'] = df['intraday_volatility'].rolling(window=60).mean()
    
    # Adjust for Recent Volatility
    df['sma_recent_volatility_10d'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=30, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = (df['ema_volume_weighted_intraday_return'] 
                          + df['sma_intraday_volatility_60d'] 
                          + df['sma_recent_volatility_10d'])
    
    # Normalize Final Factor
    df['normalized_final_factor'] = (df['final_factor'] - df['final_factor'].mean()) / df['final_factor'].std()
    
    return df['normalized_final_factor']
