```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_vol_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(alpha=0.15, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=30).sum().apply(np.sqrt)
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=30).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility_sma'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_vol_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(alpha=0.15, adjust=False).mean()
    
    # Dynamically Balance Factor Contributions
    ewma_factor = df['ewma_vol_weighted_intraday_return']
    sma_factor = df['sma_intraday_volatility']
    recent_vol_factor = df['recent_volatility_sma']
    
    # Assign weights to each factor based on historical performance
    weight_ewma = 0.4
    weight_sma = 0.3
    weight_recent_vol = 0.3
    
    # Combine Weighted Factors to Form Final Factor
    final_factor = (weight_ewma * ewma_factor) + (weight_sma * sma_factor) + (weight_recent_vol * recent_vol_factor)
    
    return final_factor
```
