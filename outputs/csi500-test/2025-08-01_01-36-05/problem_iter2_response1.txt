```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day return
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Determine 10-day high and low
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    
    # Compute breakout strength
    df['breakout_strength'] = (df['10_day_high'] - df['10_day_low']) * df['10_day_return']
    
    # Calculate 10-day standard deviation of log returns
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Volume-adjusted factor
    df['volume_adjusted_factor'] = df['volume'] * df['breakout_strength']
    
    # Calculate 5-day simple moving average
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    
    # Get trend strength
    df['trend_strength'] = df['close'] - df['5_day_sma']
    
    # Final alpha factor
    df['alpha_factor'] = df['volume_adjusted_factor'] - df['volatility'] + df['trend_strength']
    
    return df['alpha_factor'].dropna()
```
