```python
import pandas as pd
import numpy as np

def adaptive_ema(data, span, alpha):
    ema = data.ewm(span=span, adjust=False).mean()
    return ema * alpha + data * (1 - alpha)

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Return
    recent_volatility = df['intraday_return'].rolling(window=5).std().shift(1)
    alpha = 1 / (1 + (recent_volatility / df['intraday_return'].std()).fillna(1))
    df['adaptive_ema_vol_weighted_return'] = adaptive_ema(df['vol_weighted_intraday_return'], 30, alpha)
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_returns'] = df['squared_intraday_return'].rolling(window=15).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_intraday_returns'])
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=15).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility_sma'] = df['intraday_volatility'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Adaptive Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['adaptive_ema_vol_weighted_momentum'] = adaptive_ema(df['vol_weighted_intraday_momentum'], 30, alpha)
    
    # Combine Factors
    df['final_factor'] = (df['adaptive_ema_vol_weighted_return'] 
                          + df['sma_intraday_volatility'] 
                          - df['recent_volatility_sma'])

    return df['final_factor'].dropna()
```
