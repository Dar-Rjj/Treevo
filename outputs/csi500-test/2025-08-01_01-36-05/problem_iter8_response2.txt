```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Calculate Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']

    # Exponential Moving Average of Volume-Weighted Intraday Return (Last 20 Days)
    ema_volume_weighted_intraday_return = volume_weighted_intraday_return.ewm(span=20, adjust=False).mean()

    # Simple Moving Average of Volume-Weighted Intraday Return (Last 10 Days)
    sma_volume_weighted_intraday_return = volume_weighted_intraday_return.rolling(window=10).mean()

    # Calculate Intraday Volatility
    squared_intraday_return = intraday_return ** 2
    sum_squared_intraday_returns = squared_intraday_return.rolling(window=20).sum()
    intraday_volatility = np.sqrt(sum_squared_intraday_returns)

    # Simple Moving Average of Intraday Volatility (Last 20 Days)
    sma_intraday_volatility = intraday_volatility.rolling(window=20).mean()

    # Exponential Moving Average of Intraday Volatility (Last 10 Days)
    ema_intraday_volatility = intraday_volatility.ewm(span=10, adjust=False).mean()

    # Calculate Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = intraday_return * df['volume']

    # Exponential Moving Average of Volume-Weighted Intraday Momentum (Last 20 Days)
    ema_volume_weighted_intraday_momentum = volume_weighted_intraday_momentum.ewm(span=20, adjust=False).mean()

    # Simple Moving Average of Volume-Weighted Intraday Momentum (Last 10 Days)
    sma_volume_weighted_intraday_momentum = volume_weighted_intraday_momentum.rolling(window=10).mean()

    # Combine Factors
    combined_factor = (
        ema_volume_weighted_intraday_return
        + sma_volume_weighted_intraday_return
        + sma_intraday_volatility
        + ema_intraday_volatility
        + ema_volume_weighted_intraday_momentum
        + sma_volume_weighted_intraday_momentum
    )

    return combined_factor
```
