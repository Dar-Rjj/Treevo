```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=60, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=90).sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=90, adjust=False).mean()
    
    # Adjust for Recent Volatility
    df['recent_volatility_sma'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=60, adjust=False).mean()
    
    # Combine Factors
    df['final_factor'] = (df['ema_volume_weighted_intraday_return'] 
                          + df['ema_intraday_volatility'] 
                          + df['recent_volatility_sma'])
    
    # Normalize Final Factor (if necessary)
    df['normalized_final_factor'] = (df['final_factor'] - df['final_factor'].mean()) / df['final_factor'].std()
    
    return df['normalized_final_factor']
```
