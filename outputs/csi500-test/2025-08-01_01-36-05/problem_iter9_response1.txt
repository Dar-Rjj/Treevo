```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['sum_squared_intraday_returns'] = df['squared_intraday_return'].rolling(window=20).sum()
    df['intraday_volatility'] = np.sqrt(df['sum_squared_intraday_returns'])
    
    # Calculate Simple Moving Average of Intraday Volatility
    df['sma_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20).mean()
    
    # Combine Factors
    df['factor'] = (df['ewma_volume_weighted_intraday_return'] + 
                    df['sma_intraday_volatility'] + 
                    df['ewma_volume_weighted_intraday_momentum'])
    
    return df['factor']
```
