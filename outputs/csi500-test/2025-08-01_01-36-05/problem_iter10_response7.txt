```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ema_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=20).sum())
    
    # Calculate Adaptive Volatility
    df['ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    df['recent_ema_intraday_volatility'] = df['intraday_volatility'].ewm(span=5, adjust=False).mean()
    df['adaptive_volatility'] = df['ema_intraday_volatility'] + df['recent_ema_intraday_volatility']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Volume-Weighted Intraday Momentum Ratio
    df['volume_weighted_intraday_momentum_ratio'] = df['ema_volume_weighted_intraday_momentum'] / df['volume_weighted_intraday_momentum']
    
    # Calculate Adaptive Volatility Ratio
    df['adaptive_volatility_ratio'] = df['adaptive_volatility'] / df['intraday_volatility']
    
    # Combine Factors
    df['final_factor'] = (df['ema_volume_weighted_intraday_return'] 
                          + df['adaptive_volatility_ratio'] 
                          + df['volume_weighted_intraday_momentum_ratio'])
    
    return df['final_factor']
```
