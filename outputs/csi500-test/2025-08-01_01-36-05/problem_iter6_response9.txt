```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=20).sum().apply(np.sqrt)
    
    # Weight by Volume
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Incorporate Short-Term Price Momentum
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['price_momentum'] = df['close'] - df['5_day_sma']
    
    # Normalize Weighted Intraday Volatility by dividing by the sum of volumes over 20 days
    df['normalized_weighted_intraday_volatility'] = df['weighted_intraday_volatility'] / df['volume'].rolling(window=20).sum()
    
    # Combine Intraday Volatility and Momentum
    df['combined_factor'] = df['normalized_weighted_intraday_volatility'] + df['price_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] * df['volume']
    
    return df['alpha_factor'].dropna()
```
