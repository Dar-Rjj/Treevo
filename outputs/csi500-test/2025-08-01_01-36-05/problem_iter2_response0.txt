```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Incorporate Volume Impact
    volume_impact = intraday_range * df['volume']
    
    # Determine Momentum Direction
    momentum_direction = (df['close'] > df['open']).astype(int)
    momentum_direction[momentum_direction == 0] = -1
    
    # Apply Momentum Direction
    momentum_adjusted = (df['close'] - df['open']) * volume_impact * momentum_direction
    
    # Refine with Close-to-Previous Close Return
    df['previous_close'] = df['close'].shift(1)
    close_to_previous_close_return = df['close'] - df['previous_close']
    refined_factor = close_to_previous_close_return * momentum_adjusted
    
    return refined_factor
```
