import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    ema_vol_weighted_intraday_return = df['vol_weighted_intraday_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=30).sum() ** 0.5
    
    # Calculate Simple Moving Average of Intraday Volatility
    sma_intraday_volatility = df['intraday_volatility'].rolling(window=30).mean()
    
    # Adjust for Recent Volatility
    recent_sma_intraday_volatility = df['intraday_volatility'].rolling(window=10).mean()
    
    # Incorporate Price Trend
    df['daily_return'] = df['close'].diff()
    
    # Calculate Exponential Moving Average of Daily Return
    ema_daily_return = df['daily_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['vol_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    ema_vol_weighted_intraday_momentum = df['vol_weighted_intraday_momentum'].ewm(span=30, adjust=False).mean()
    
    # Combine Factors
    factor = (
        ema_vol_weighted_intraday_return +
        sma_intraday_volatility +
        recent_sma_intraday_volatility +
        ema_daily_return +
        ema_vol_weighted_intraday_momentum
    )
    
    return factor
