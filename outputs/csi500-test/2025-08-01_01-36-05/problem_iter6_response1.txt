```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    intraday_volatility = np.sqrt(df['squared_intraday_return'].rolling(window=20).sum())
    
    # Calculate Exponential Moving Average (EMA) of Intraday Volatility
    df['ema_intraday_volatility'] = intraday_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Weight by Volume
    df['volume_weighted_ema_intraday_volatility'] = df['ema_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate EMA of Volume-Weighted Intraday Momentum
    df['ema_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Combine Factors
    df['alpha_factor'] = df['ema_intraday_volatility'] + df['ema_volume_weighted_intraday_momentum']
    
    return df['alpha_factor']
```
