```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Simple Moving Average (SMA) of Log Returns over a 14-day window
    sma_window = 14
    df['sma_log_return'] = df['log_return'].rolling(window=sma_window).mean()
    
    # Calculate Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['sma_log_return'] * df['volume']
    
    # Sum the Volume-Weighted Momentum over a 20-day window
    momentum_window = 20
    df['volume_weighted_momentum_sum'] = df['volume_weighted_momentum'].rolling(window=momentum_window).sum()
    
    # Finalize the Alpha Factor
    alpha_factor = df['volume_weighted_momentum_sum']
    
    return alpha_factor
```
