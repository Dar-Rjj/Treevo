import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Average Intraday Return
    df['avg_intraday_return'] = df['intraday_return'].rolling(window=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = df['squared_intraday_return'].rolling(window=20).sum() ** 0.5
    
    # Calculate Moving Average of Intraday Volatility
    df['moving_avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Weight by Volume
    df['volume_weighted_intraday_volatility'] = df['moving_avg_intraday_volatility'] * df['volume']
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Moving Average of Volume-Weighted Intraday Momentum
    df['moving_avg_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].rolling(window=20).mean()
    
    # Combine Factors
    df['combined_short_term_factors'] = df['moving_avg_intraday_volatility'] + df['moving_avg_volume_weighted_intraday_momentum']
    
    # Incorporate Long-Term Trends
    df['long_term_moving_avg'] = df['close'].rolling(window=50).mean()
    df['long_term_volatility'] = df['intraday_return'].rolling(window=50).std()
    df['adjusted_long_term_trend'] = (df['close'] - df['long_term_moving_avg']) / df['long_term_volatility']
    
    # Combine with Short-Term Factors
    df['final_factor'] = df['combined_short_term_factors'] + df['adjusted_long_term_trend']
    
    return df['final_factor']
