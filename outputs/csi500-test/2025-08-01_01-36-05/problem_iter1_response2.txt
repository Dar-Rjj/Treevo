```python
import pandas as pd

def heuristics_v2(df):
    # Compute Simple Moving Average of Close Price (SMA) for the past 14 days
    df['SMA_14'] = df['close'].rolling(window=14).mean()
    
    # Calculate Daily Price Momentum: Subtract SMA from Current Close Price
    df['Price_Momentum'] = df['close'] - df['SMA_14']
    
    # Compute Standard Deviation of Daily Volume for the past 20 days
    df['Volume_Volatility'] = df['volume'].rolling(window=20).std()
    
    # Calculate Daily Volume Volatility: Divide Current Day Volume by Volume Volatility
    df['Volume_Factor'] = df['volume'] / df['Volume_Volatility']

    # Generate the factor value as a combination of Price Momentum and Volume Factor
    df['Factor_Value'] = df['Price_Momentum'] * df['Volume_Factor']
    
    return df['Factor_Value']
```
