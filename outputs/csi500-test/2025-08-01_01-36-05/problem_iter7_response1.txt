import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['ewma_volume_weighted_intraday_return'] = df['volume_weighted_intraday_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['intraday_return'] ** 2).rolling(window=20).sum().pow(1/2)
    
    # Calculate Adaptive Volatility
    ewma_intraday_volatility = df['intraday_volatility'].ewm(span=20, adjust=False).mean()
    recent_volatility = df['intraday_volatility'].ewm(span=5, adjust=False).mean()
    df['adaptive_volatility'] = (ewma_intraday_volatility + recent_volatility) / 2
    
    # Calculate Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['ewma_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['alpha_factor'] = df['ewma_volume_weighted_intraday_return'] + df['adaptive_volatility'] + df['ewma_volume_weighted_intraday_momentum']
    
    return df['alpha_factor']
