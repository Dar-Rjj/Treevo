```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Average Intraday Return over 20 days
    df['avg_intraday_return'] = df['intraday_return'].rolling(window=20).mean()
    
    # Calculate Intraday Volatility
    df['squared_intraday_return'] = df['intraday_return'] ** 2
    df['intraday_volatility'] = np.sqrt(df['squared_intraday_return'].rolling(window=20).sum())
    
    # Calculate Moving Average of Intraday Volatility
    df['moving_avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=20).mean()
    
    # Volume-Weighted Intraday Momentum
    df['volume_weighted_intraday_momentum'] = df['intraday_return'] * df['volume']
    
    # Moving Average of Volume-Weighted Intraday Momentum
    df['moving_avg_volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_momentum'].rolling(window=20).mean()
    
    # Combine Short-Term Factors
    df['short_term_factors'] = df['moving_avg_intraday_volatility'] + df['moving_avg_volume_weighted_intraday_momentum']
    
    # Incorporate Long-Term Trends
    # Calculate Long-Term Moving Average
    df['long_term_moving_avg'] = df['close'].rolling(window=50).mean()
    
    # Calculate Long-Term Volatility
    df['long_term_volatility'] = df['intraday_return'].rolling(window=50).std()
    
    # Adjust for Long-Term Trends
    df['adjusted_long_term_trend'] = (df['close'] - df['long_term_moving_avg']) / df['long_term_volatility']
    
    # Combine with Short-Term Factors
    df['combined_factors'] = df['short_term_factors'] + df['adjusted_long_term_trend']
    
    # Final Alpha Factor
    alpha_factor = df['combined_factors']
    
    return alpha_factor
```
