```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    df['EMA_Volume_Weighted_Intraday_Return'] = df['Volume_Weighted_Intraday_Return'].ewm(span=20, min_periods=20).mean()
    
    # Calculate Intraday Volatility
    df['Squared_Intraday_Returns'] = df['Intraday_Return'] ** 2
    df['Intraday_Volatility'] = np.sqrt(df['Squared_Intraday_Returns'].rolling(window=20, min_periods=20).sum())
    
    # Calculate Exponential Moving Average of Intraday Volatility
    df['EMA_Intraday_Volatility'] = df['Intraday_Volatility'].ewm(span=20, min_periods=20).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    df['Volume_Weighted_Intraday_Momentum'] = df['Intraday_Return'] * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    df['EMA_Volume_Weighted_Intraday_Momentum'] = df['Volume_Weighted_Intraday_Momentum'].ewm(span=20, min_periods=20).mean()
    
    # Combine Factors
    df['Combined_Factor'] = df['EMA_Intraday_Volatility'] + df['EMA_Volume_Weighted_Intraday_Momentum']
    
    return df['Combined_Factor']
```
