```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Volume-Weighted Intraday Return
    vol_weighted_intraday_return = intraday_return * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Return
    ewma_vol_weighted_intraday_return = vol_weighted_intraday_return.ewm(span=30, adjust=False).mean()
    
    # Calculate Intraday Volatility
    squared_intraday_return = intraday_return ** 2
    intraday_volatility = np.sqrt(squared_intraday_return.rolling(window=60).sum())
    
    # Calculate Simple Moving Average of Intraday Volatility
    sma_intraday_volatility = intraday_volatility.rolling(window=60).mean()
    
    # Adjust for Recent Volatility
    recent_volatility_sma = intraday_volatility.rolling(window=10).mean()
    
    # Calculate Volume-Weighted Intraday Momentum
    vol_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Calculate Exponential Moving Average of Volume-Weighted Intraday Momentum
    ewma_vol_weighted_intraday_momentum = vol_weighted_intraday_momentum.ewm(span=30, adjust=False).mean()
    
    # Combine Factors
    final_factor = (
        ewma_vol_weighted_intraday_return
        + sma_intraday_volatility
        + recent_volatility_sma
        + ewma_vol_weighted_intraday_momentum
    )
    
    return final_factor
```
