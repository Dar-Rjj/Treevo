```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = np.sqrt((intraday_return ** 2).rolling(window=20).sum())
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Moving Average of Volume-Weighted Intraday Return (Exponential)
    ema_volume_weighted_intraday_return = volume_weighted_intraday_return.ewm(span=20).mean()
    
    # Volume-Weighted Intraday Volatility
    volume_weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Moving Average of Volume-Weighted Intraday Volatility (Simple)
    sma_volume_weighted_intraday_volatility = volume_weighted_intraday_volatility.rolling(window=20).mean()
    
    # Combined Alpha Factor
    combined_alpha_factor = ema_volume_weighted_intraday_return / sma_volume_weighted_intraday_volatility
    
    return combined_alpha_factor
```
