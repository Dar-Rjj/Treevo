```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Simple Moving Average of Close Price (SMA) for the past 14 days
    df['SMA_14'] = df['close'].rolling(window=14).mean()
    
    # Calculate Daily Price Momentum
    df['Momentum'] = df['close'] - df['SMA_14']
    
    # Calculate High-Low Price Ratio
    df['High_Low_Ratio'] = (df['high'] / df['low']) - 1
    
    # Adjust by Volume and Momentum
    df['Rolling_Sum_Volume'] = df['volume'].rolling(window=5).sum()
    df['Adjusted_High_Low_Ratio'] = df['High_Low_Ratio'] * df['Rolling_Sum_Volume']
    
    # Incorporate Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    df['Normalized_Momentum'] = (df['close'] - df['SMA_14']) / df['Volatility']
    
    # Leverage Moving Averages
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_Comparison'] = np.where(df['close'] > df['EMA_5'], 1, -1)
    df['Adjusted_Momentum'] = df['Normalized_Momentum'] * df['EMA_Comparison']
    
    # Consider Volume-Weighted Price Change
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['VWAP_Comparison'] = np.where(df['close'] > df['VWAP'], 1, -1)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Momentum'] + df['Adjusted_High_Low_Ratio'] * df['VWAP_Comparison']
    
    return df['Alpha_Factor']
```
