```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a volume-weighted close-to-open return factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    
    Returns:
    pd.Series: Volume weighted close-to-open return factor indexed by (date).
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
