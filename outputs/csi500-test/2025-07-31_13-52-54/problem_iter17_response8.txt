```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Daily VWAP Change
    df['vwap_change'] = df['vwap'].diff()
    
    # Separate Positive and Negative Changes
    positive_changes = df['vwap_change'].where(df['vwap_change'] > 0, 0)
    negative_changes = -df['vwap_change'].where(df['vwap_change'] < 0, 0)
    
    # 14-Day Averages for RSI
    avg_positive = positive_changes.rolling(window=14).mean()
    avg_negative = negative_changes.rolling(window=14).mean()
    
    # Calculate RSI
    relative_strength = avg_positive / avg_negative
    rsi = 100 - (100 / (1 + relative_strength))
    df['rsi'] = rsi
    
    # Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_range = df['high'].shift(1) - df['low'].shift(1)
    df['intraday_range_growth'] = (today_range - prev_day_range) / prev_day_range
    
    # Volume-Weighted Moving Average (VWMA)
    n_days = 20  # You can adjust this parameter
    close_volume_product = df['close'] * df['volume']
    vwm_sum = close_volume_product.rolling(window=n_days).sum()
    total_volume = df['volume'].rolling(window=n_days).sum()
    df['vwma'] = vwm_sum / total_volume
    
    # Long-Term and Short-Term Volume-Weighted Returns
    daily_returns = df['close'].pct_change()
    long_term_days = 90
    short_term_days = 5
    
    # Long-Term VW-Return (Momentum Component)
    long_term_vw_return = (daily_returns.rolling(window=long_term_days).sum() * df['volume']).rolling(window=long_term_days).sum() / df['volume'].rolling(window=long_term_days).sum()
    
    # Short-Term VW-Return (Reversal Component)
    short_term_vw_return = (daily_returns.rolling(window=short_term_days).sum() * df['volume']).rolling(window=short_term_days).sum() / df['volume'].rolling(window=short_term_days).sum()
    
    # Adjusted VW-Return
    adjusted_vw_return = long_term_vw_return - short_term_vw_return
    df['adjusted_vw_return'] = adjusted_vw_return
    
    # Volatility-Adjusted Alpha Factor
    volatility = daily_returns.rolling(window=30).std()
    alpha_factor = (df['rsi'] + df['intraday_range_growth'] + df['vwma'] + df['adjusted_vw_return']) / volatility
    df['alpha_factor'] = alpha_factor
    
    return df['alpha_factor']
```
