```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the weighted high-low spread and volume-adjusted momentum.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series with the calculated alpha factor, indexed by date.
    """
    # Calculate the high-low spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate the average volume over the last 5 days
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    
    # Calculate the weighted high-low spread
    weighted_high_low_spread = high_low_spread * (df['volume'] / avg_volume_5_days)
    
    # Calculate the price momentum
    price_momentum = df['close'].diff()
    
    # Calculate the final alpha factor
    alpha_factor = price_momentum / weighted_high_low_spread
    
    return alpha_factor
```
