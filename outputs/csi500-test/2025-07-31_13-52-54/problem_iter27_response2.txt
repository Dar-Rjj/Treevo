import pandas as pd

def heuristics_v2(df, n=10, m=20):
    """
    Calculate the Price-Volume Momentum to Smoothed Average Ratio.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    - n: Number of days for the momentum calculation.
    - m: Number of days for the smoothed average calculation.
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate Price-Volume Momentum
    high_volume = df['high'] * df['volume']
    low_volume = df['low'] * df['volume']
    high_volume_sum = high_volume.rolling(window=n).sum()
    low_volume_sum = low_volume.rolling(window=n).sum()
    price_volume_momentum = high_volume_sum - low_volume_sum
    
    # Calculate Price-Volume Smoothed Average
    close_volume = df['close'] * df['volume']
    close_volume_sum = close_volume.rolling(window=m).sum()
    total_volume = df['volume'].rolling(window=m).sum()
    price_volume_smoothed_average = close_volume_sum / total_volume
    
    # Compute Momentum to Smoothed Average Ratio
    momentum_to_smoothed_ratio = price_volume_momentum / price_volume_smoothed_average
    
    return momentum_to_smoothed_ratio
