```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_days = 100
    df['long_term_vol_weighted_return'] = (df['daily_return'].rolling(window=long_term_days).sum() * 
                                           df['volume'].rolling(window=long_term_days).sum()) / df['volume'].rolling(window=long_term_days).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_days = 5
    df['short_term_vol_weighted_return'] = (df['daily_return'].rolling(window=short_term_days).sum() * 
                                            df['volume'].rolling(window=short_term_days).sum()) / df['volume'].rolling(window=short_term_days).sum()
    
    # Integrate High-Low Range into Volume-Weighted Returns
    df['high_low_range'] = df['high'] - df['low']
    df['vol_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    total_volume_last_5_days = df['volume'].rolling(window=short_term_days).sum()
    df['normalized_vol_weighted_high_low_range'] = df['vol_weighted_high_low_range'] / total_volume_last_5_days
    
    # Combine Momentum, Reversal, and High-Low Range Components
    df['alpha_factor'] = (df['long_term_vol_weighted_return'] - df['short_term_vol_weighted_return'] + 
                          df['normalized_vol_weighted_high_low_range'])
    
    return df['alpha_factor']
```
