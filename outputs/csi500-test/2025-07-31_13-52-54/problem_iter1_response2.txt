import pandas as pd

def heuristics_v2(df, n=10):
    # Compute Daily High-Low Range
    daily_range = df['high'] - df['low']
    
    # Compute Previous Day's High-Low Range
    prev_daily_range = df['high'].shift(1) - df['low'].shift(1)
    
    # Compute Intraday Range Growth
    intraday_range_growth = (daily_range - prev_daily_range) / prev_daily_range
    
    # Calculate Simple Moving Average
    sma_close = df['close'].rolling(window=n).mean()
    
    # Weight by Volume
    volume_weighted_close = df['close'] * df['volume']
    
    # Sum of weighted close prices and sum of volumes
    sum_weighted_close = volume_weighted_close.rolling(window=n).sum()
    sum_volume = df['volume'].rolling(window=n).sum()
    
    # Volume Weighted Moving Average
    vwma = sum_weighted_close / sum_volume
    
    # Combine Factors
    factor_values = intraday_range_growth * vwma
    
    return factor_values
