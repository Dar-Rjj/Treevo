```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_spread = high_low_spread * df['volume']
    volume_weighted_spread_sum = volume_weighted_spread.rolling(window=lookback_period).sum()
    
    # Calculate Momentum Component
    current_volume_weighted_spread = volume_weighted_spread
    past_avg_volume_weighted_spread = volume_weighted_spread.rolling(window=lookback_period).mean().shift(1)
    momentum = current_volume_weighted_spread - past_avg_volume_weighted_spread
    
    # Generate Final Factor
    factor = (volume_weighted_spread_sum + momentum) / df['close']
    
    # Incorporate Price Trend
    n_day_moving_avg_close = df['close'].rolling(window=lookback_period).mean()
    price_trend = df['close'] - n_day_moving_avg_close
    final_factor = factor * price_trend
    
    return final_factor
```
