```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Volume-Weighted Intraday Momentum Factor
    df['vwi_momentum'] = df['volume_weighted_intraday_return'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['vwi_reversal'] = -df['volume_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate Volume-Weighted Average Price
    df['vwap'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Compute Daily Change in Volume-Weighted Price
    df['vwap_change'] = df['vwap'].diff()
    
    # Separate Positive and Negative Changes
    positive_changes = df['vwap_change'].apply(lambda x: x if x > 0 else 0)
    negative_changes = df['vwap_change'].apply(lambda x: -x if x < 0 else 0)
    
    # Compute 14-Day Average of Positive Changes
    df['avg_positive_change'] = positive_changes.rolling(window=14).mean()
    
    # Compute 14-Day Average of Negative Changes
    df['avg_negative_change'] = negative_changes.rolling(window=14).mean()
    
    # Calculate RSI
    relative_strength = df['avg_positive_change'] / df['avg_negative_change']
    df['rsi'] = 100 - (100 / (1 + relative_strength))
    
    # Final Alpha Factor
    df['alpha_factor'] = df['vwi_momentum'] - df['vwi_reversal'] + df['rsi']
    
    return df['alpha_factor']
```
