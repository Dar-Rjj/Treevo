import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series representing the factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day exponential moving average of the daily return
    df['ema_10_daily_return'] = df['daily_return'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10_daily_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day mean of the volume
    df['ma_10_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day exponential moving average of the volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the volume
    df['std_10_volume'] = df['volume'].rolling(window=10).std()
    
    # Calculate the normalized daily return
    df['normalized_daily_return'] = (df['daily_return'] - df['ma_10_daily_return']) / df['std_10_daily_return']
    
    # Calculate the normalized volume
    df['normalized_volume'] = (df['volume'] - df['ma_10_volume']) / df['std_10_volume']
    
    # Calculate the alpha factor
    df['alpha_factor'] = df['normalized_daily_return'] * df['normalized_volume']
    
    return df['alpha_factor']
