```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day Simple Moving Average (SMA) of closing prices
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Generate a binary signal; 1 if the current close is above the SMA, 0 otherwise
    df['Trend_Signal'] = (df['close'] > df['SMA_50']).astype(int)
    
    # Calculate the difference between today’s high and the previous day’s high
    df['High_Momentum'] = df['high'] - df['high'].shift(1)
    
    # Calculate the difference between today’s low and the previous day’s low
    df['Low_Momentum'] = df['low'] - df['low'].shift(1)
    
    # Sum the differences to get a combined momentum score
    df['Momentum_Score'] = df['High_Momentum'] + df['Low_Momentum']
    
    # Create a simple threshold to classify stocks as having positive or negative momentum based on the score
    df['Momentum_Signal'] = (df['Momentum_Score'] > 0).astype(int)
    
    # Evaluate the percentage change in volume from the previous day
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Assign a value indicating strong buying (positive change) or selling (negative change) pressure
    df['Volume_Signal'] = df['Volume_Change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Multiply the return (close to close) by the volume traded
    df['Return'] = df['close'].pct_change()
    df['Intensity_Factor'] = df['Return'] * df['volume']
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    df['Vol_Weighted_Return_100'] = (df['Return'] * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    df['Vol_Weighted_Return_5'] = (df['Return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['close']
    
    # Intraday Momentum
    df['Intraday_Momentum'] = df['Intraday_Return'] * df['volume']
    df['Intraday_Momentum_5'] = df['Intraday_Momentum'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['Vol_Weighted_Intraday_Reversal'] = -df['Intraday_Momentum']
    df['Vol_Weighted_Intraday_Reversal_10'] = df['Vol_Weighted_Intraday_Reversal'].rolling(window=10).sum()
    
    # Price Momentum
    df['Price_Momentum_Close'] = df['close'].pct_change(periods=10)
    df['Price_Momentum_Open'] = df['open'].pct_change(periods=10)
    
    # Weighted High-Low Spread Adjusted for Volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    df['Weighted_High_Low_Spread'] = (df['high'] - df['low']) * (df['volume'] / avg_volume_5)
    
    # Divide Price Momentum by Weighted High-Low Spread Adjusted for Volume
    df['Price_Momentum_Divided'] = (df['Price_Momentum_Close'] + df['Price_Momentum_Open']) / df['Weighted_High_Low_Spread']
    
    # Combined Alpha Factor
    combined_alpha_factor = (
        (df['Vol_Weighted_Return_100'] - df['Vol_Weighted_Return_5']) +
        (df['Intraday_Momentum_5'] + df['Vol_Weighted_Intraday_Reversal_10'])
    ) * (df['Price_Momentum_Close'] + df['Price_Momentum_Open']) + df['Price_Momentum_Divided']
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.4 * combined_alpha_factor +
        0.3 * df['Trend_Signal'] +
        0.2 * df['Momentum_Signal'] +
        0.1 * df['Volume_Signal']
    )
    
    return final_alpha_factor.dropna()
```
