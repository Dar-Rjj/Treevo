import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    n = 50
    price_momentum = df['close'].pct_change(n)
    
    # Explore Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Adjusted Volatility
    volatility = (df['high'] - df['low']).rolling(window=n).mean()
    volume_adjusted_volatility = volatility * df['volume']
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].ne(0).astype(int)
    trade_frequency_adjusted_return = df['close'].pct_change(n) * trade_frequency.rolling(window=n).mean()
    
    # Combine Initial Factors
    combined_factors = (price_momentum + intraday_high_low_spread) * volume_adjusted_volatility + trade_frequency_adjusted_return
    
    # Develop trend following alpha factor using close prices
    sma_period = 50
    sma = df['close'].rolling(window=sma_period).mean()
    trend_signal = (df['close'] > sma).astype(int)
    
    # Explore combined momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum = (high_momentum + low_momentum) / 2
    combined_momentum_score = (combined_momentum > 0).astype(int)
    
    # Analyze volume patterns
    volume_change = df['volume'].pct_change()
    volume_sentiment = volume_change.apply(lambda x: 1 if x > 0 else 0)
    
    # Leverage price and volume for an intensity factor
    intensity_factor = df['close'].pct_change() * df['volume']
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_returns = (df['close'].pct_change() * df['volume']).rolling(window=200).sum()
    long_term_volume_weighted_return = long_term_returns / df['volume'].rolling(window=200).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_returns = (df['close'].pct_change() * df['volume']).rolling(window=10).sum()
    short_term_volume_weighted_return = short_term_returns / df['volume'].rolling(window=10).sum()
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Intraday Momentum
    intraday_momentum = (intraday_return * df['volume']).rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    volume_weighted_intraday_reversal = -(intraday_return * df['volume']).rolling(window=10).sum()
    
    # Combined Alpha Factor
    combined_alpha_factor = (long_term_volume_weighted_return - short_term_volume_weighted_return) + (intraday_momentum + volume_weighted_intraday_reversal) * price_momentum
    
    # Calculate Enhanced Volume-Weighted Volatility
    enhanced_volatility = (df['high'] - df['low']).rolling(window=n).mean() * df['volume']
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency_adjusted_return = df['close'].pct_change(n) * trade_frequency.rolling(window=n).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (price_momentum * enhanced_volatility) + trade_frequency_adjusted_return + combined_alpha_factor
    
    return final_alpha_factor
