import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum)
    df['Long_Term_Volume_Weighted_Return'] = (df['Daily_Return'] * df['volume']).rolling(window=120).sum() / df['volume'].rolling(window=120).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal)
    df['Short_Term_Volume_Weighted_Return'] = (df['Daily_Return'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Volume-Averaged Price
    df['Volume_Averaged_Price'] = (df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    
    # Volume-Averaged High-Low Spread Ratio
    df['Volume_Averaged_High_Low_Ratio'] = df['High_Low_Spread'] / df['Volume_Averaged_Price']
    
    # 50-day Simple Moving Average (SMA) of closing prices
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Generate a binary signal; 1 if the current close is above the SMA, -1 if below, and 0 if equal
    df['SMA_Binary_Signal'] = (df['close'] > df['SMA_50']).astype(int) - (df['close'] < df['SMA_50']).astype(int)
    
    # Difference between today’s high and the previous day’s high
    df['High_Diff'] = df['high'] - df['high'].shift(1)
    
    # Difference between today’s low and the previous day’s low
    df['Low_Diff'] = df['low'] - df['low'].shift(1)
    
    # Combined momentum score
    df['Combined_Momentum_Score'] = df['High_Diff'] + df['Low_Diff']
    
    # Simple threshold to classify stocks as having strong positive or negative momentum based on the score
    df['Momentum_Signal'] = (df['Combined_Momentum_Score'] > 0.5).astype(int) - (df['Combined_Momentum_Score'] < -0.5).astype(int)
    
    # Integrate the Volume-Averaged High-Low Spread Ratio with the trend following and momentum signals
    df['Integrated_Alpha_Factor'] = 0.5 * df['SMA_Binary_Signal'] + 0.3 * df['Momentum_Signal'] + 0.2 * df['Volume_Averaged_High_Low_Ratio']
    
    # Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['open'] * df['volume']
    df['Intraday_Momentum_Sum_7'] = df['Intraday_Momentum'].rolling(window=7).sum()
    
    # Volume-Weighted Intraday Reversal
    df['Intraday_Reversal'] = -(df['high'] - df['low']) / df['open'] * df['volume']
    df['Intraday_Reversal_Sum_15'] = df['Intraday_Reversal'].rolling(window=15).sum()
    
    # Combined Alpha Factor
    df['Combined_Alpha_Factor'] = (df['Long_Term_Volume_Weighted_Return'] - df['Short_Term_Volume_Weighted_Return']) + \
                                  (df['Intraday_Momentum_Sum_7'] + df['Intraday_Reversal_Sum_15'])
    
    # Adjusted Intraday High-Low Spread
    df['Adjusted_Intraday_High_Low_Spread'] = (df['high'] - df['low']) * df['volume'] + df['close']
    
    # Final Enhanced Alpha Factor
    df['Final_Enhanced_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Adjusted_Intraday_High_Low_Spread'] + df['Integrated_Alpha_Factor']
