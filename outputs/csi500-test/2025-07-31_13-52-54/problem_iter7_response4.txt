```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Intraday Volatility
    intraday_volatility = abs(df['high'] - df['low'])
    intraday_volatility_20d = intraday_volatility.rolling(window=20).sum()
    
    # Intraday Momentum
    intraday_momentum = intraday_return * df['volume']
    intraday_momentum_5d = intraday_momentum.rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    volume_weighted_reversal = -(intraday_return * df['volume'])
    volume_weighted_reversal_10d = volume_weighted_reversal.rolling(window=10).sum()
    
    # Price Momentum
    n_day_return_close = df['close'].pct_change(periods=20)
    n_day_return_open = df['open'].pct_change(periods=20)
    
    # Volume Adjusted Volatility
    volatility_n_days = (df['high'] - df['low']).rolling(window=20).std()
    volume_adjusted_volatility = volatility_n_days / df['volume']
    
    # Combined Factor
    combined_factor = (intraday_momentum_5d + volume_weighted_reversal_10d) * (n_day_return_close + n_day_return_open) / volume_adjusted_volatility
    
    return combined_factor
```
