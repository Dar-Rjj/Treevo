import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Average Price
    volume_average_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Volume-Averaged High-Low Spread Ratio
    spread_ratio = high_low_spread / volume_average_price
    
    # Calculate the 50-day Simple Moving Average (SMA) of closing prices
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Generate a binary signal; 1 if the current close is above the SMA, -1 if below, and 0 if equal
    sma_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Calculate the difference between today’s high and the previous day’s high
    high_diff = df['high'] - df['high'].shift(1)
    
    # Calculate the difference between today’s low and the previous day’s low
    low_diff = df['low'] - df['low'].shift(1)
    
    # Sum the differences to get a combined momentum score
    momentum_score = high_diff + low_diff
    
    # Create a simple threshold to classify stocks as having strong positive or negative momentum based on the score
    momentum_signal = (momentum_score > 0).astype(int) - (momentum_score < 0).astype(int)
    
    # Integrate the Volume-Averaged High-Low Spread Ratio with the trend following and momentum signals
    integrated_factor = 0.5 * spread_ratio + 0.3 * sma_signal + 0.2 * momentum_signal
    
    # Analyze volume patterns
    volume_change = df['volume'].pct_change()
    
    # Assign a value indicating strong buying (positive change) or selling (negative change) pressure
    volume_signal = (volume_change > 0).astype(int) - (volume_change < 0).astype(int)
    
    # Leverage price and volume for an intensity factor
    intensity_factor = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Use a moving average of this intensity factor over the past 20 days to smooth out short-term fluctuations
    intensity_moving_avg = intensity_factor.rolling(window=20).mean()
    
    # Combine the intensity factor with the integrated alpha factor
    final_factor = integrated_factor + intensity_moving_avg
    
    return final_factor
