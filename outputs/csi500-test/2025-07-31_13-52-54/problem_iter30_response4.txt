```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day EMA of the money flow
    ema_money_flow = money_flow.ewm(span=10).mean()
    
    # Calculate the 10-day EMA of the volume
    ema_volume = df['volume'].ewm(span=10).mean()
    
    # Calculate the Money Flow Index (MFI)
    mfi = ema_money_flow / ema_volume
    
    # Calculate the 10-day ATR (Average True Range)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=10).mean()
    
    # Calculate the final factor
    factor = (mfi - mfi.rolling(window=10).mean()) / atr
    
    return factor
```
