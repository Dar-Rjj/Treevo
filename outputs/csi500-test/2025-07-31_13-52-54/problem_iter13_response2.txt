import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(n)
    
    # Calculate Enhanced Volume-Weighted Volatility
    df['range'] = (df['high'] - df['low']).rolling(window=n).mean()
    df['volatility'] = df['range'] * df['volume']
    df['enhanced_volatility'] = df['volatility'].rolling(window=n).mean()
    
    # Calculate Trade Frequency Adjusted Return
    df['trade_frequency'] = df['volume'].rolling(window=n).count()
    df['trade_frequency_adjusted_return'] = (df['close'].pct_change(n) * df['trade_frequency'])
    
    # Combine Factors
    df['alpha_factor'] = (df['price_momentum'] * df['enhanced_volatility']) + df['trade_frequency_adjusted_return']
    
    return df['alpha_factor'].dropna()
