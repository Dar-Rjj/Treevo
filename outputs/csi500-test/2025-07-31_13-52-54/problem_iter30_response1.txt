import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Price
    volume_weighted_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Ratio
    volume_averaged_high_low_spread_ratio = high_low_spread / volume_weighted_price
    
    # Develop trend following alpha factor using close prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Explore momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum = high_momentum + low_momentum
    momentum_threshold = 0.5 * combined_momentum.abs().median()
    momentum_signal = (combined_momentum > momentum_threshold).astype(int) - (combined_momentum < -momentum_threshold).astype(int)
    
    # Analyze volume patterns
    volume_change = df['volume'].pct_change()
    volume_category = pd.cut(volume_change, bins=[-float('inf'), -0.1, 0.1, float('inf')], labels=[-1, 0, 1])
    
    # Leverage price and volume for an intensity factor
    intensity_factor = (df['close'].pct_change() * df['volume']).rolling(window=20).mean()
    
    # Calculate Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=20).sum()
    
    # Calculate Momentum Component
    avg_past_volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=20).mean()
    momentum_component = volume_weighted_high_low_spread - avg_past_volume_weighted_high_low_spread
    
    # Generate Final Factor
    final_factor = (momentum_component + volume_weighted_high_low_spread) / df['close']
    
    # Incorporate Price Trend
    n_day_sma = df['close'].rolling(window=20).mean()
    trend_component = (df['close'] - n_day_sma) * final_factor
    
    # Introduce High-Low Range Volatility
    high_low_range_volatility = high_low_spread.rolling(window=20).std()
    adjusted_final_factor = trend_component / (1 + high_low_range_volatility)
    
    # Synthesize Factors
    composite_alpha_factor = (0.4 * trend_signal + 0.3 * momentum_signal + 0.3 * intensity_factor) * adjusted_final_factor
    
    return composite_alpha_factor
