import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['high'] - df['open']
    high_close_to_open_high_ratio = high_close_diff / open_high_diff
    
    # Summarize Volume-Adjusted High-Low Spread Over Lookback Period
    cumulative_volume_adjusted_high_low_spread = volume_adjusted_high_low_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate Momentum Component
    current_day_value = volume_adjusted_high_low_spread
    past_n_days_average = (volume_adjusted_high_low_spread.rolling(window=lookback_period).sum() / total_volume)
    momentum_component = current_day_value - past_n_days_average
    
    # Combine Adjusted Spread and Ratio
    combined_spread_and_ratio = volume_adjusted_high_low_spread + high_close_to_open_high_ratio
    
    # Synthesize Final Factor
    final_factor = combined_spread_and_ratio + momentum_component
    
    return final_factor
