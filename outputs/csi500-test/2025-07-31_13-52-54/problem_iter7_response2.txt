```python
import pandas as pd

def heuristics_v2(df, N=10, M=20):
    # Calculate Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_range = df['high'].shift(1) - df['low'].shift(1)
    intraday_range_growth = (today_range - prev_day_range) / prev_day_range
    
    # Volume Weighted Moving Average
    close_prices = df['close'].rolling(window=N).mean()
    volumes = df['volume'].rolling(window=N).sum()
    weighted_close_prices = (df['close'] * df['volume']).rolling(window=N).sum()
    volume_weighted_moving_avg = weighted_close_prices / volumes
    
    # Price Momentum
    n_day_price_diff = df['close'] - df['close'].shift(N)
    m_day_price_diff = df['close'] - df['close'].shift(M)
    price_momentum = n_day_price_diff + m_day_price_diff
    
    # Volume Surge
    n_day_volume_change = df['volume'] - df['volume'].shift(N)
    m_day_volume_change = df['volume'] - df['volume'].shift(M)
    volume_surge = n_day_volume_change + m_day_volume_change
    adjusted_volume_surge = volume_surge.apply(lambda x: x * 0.8 if x < 0 else x)
    
    # Final Alpha Factor
    combined_factor = (intraday_range_growth 
                       + volume_weighted_moving_avg 
                       + price_momentum 
                       + adjusted_volume_surge)
    
    return combined_factor
```
