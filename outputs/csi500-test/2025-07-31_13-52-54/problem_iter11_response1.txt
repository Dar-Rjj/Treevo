```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted High-Low Spread Adjusted for Volume
    high_low_spread = df['high'] - df['low']
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * (df['volume'] / avg_volume_5d)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Divide Price Momentum by Weighted High-Low Spread Adjusted for Volume
    momentum_weighted_spread = price_momentum / weighted_high_low_spread
    
    # Generate Simple Moving Average (SMA) of Closing Prices
    sma_50d = df['close'].rolling(window=50).mean()
    
    # Determine Stock Position Relative to its SMA
    position_relative_to_sma = (df['close'] > sma_50d).astype(int)
    
    # Calculate High and Low Momentum
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum_score = high_momentum + low_momentum
    
    # Classify Stocks Based on Momentum Score
    momentum_classification = (combined_momentum_score > 0).astype(int)
    
    # Evaluate Volume Patterns
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Infer Trading Activity and Sentiment from Volume
    volume_intensity = volume_change * df['close'].pct_change()
    
    # Combine Factors into a Single Alpha Factor
    alpha_factor = (
        0.4 * momentum_weighted_spread +
        0.3 * position_relative_to_sma +
        0.2 * momentum_classification +
        0.1 * volume_intensity
    )
    
    return alpha_factor
```
