def heuristics(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Volume-Weighted Intraday Momentum Factor
    df['volume_weighted_intraday_momentum'] = df['volume_weighted_intraday_return'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['volume_weighted_intraday_reversal'] = -df['volume_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    df['high_close_open_high_ratio'] = (df['high'] - df['close']) / (df['open'] - df['high'])
    
    # Combine Spread and Ratio
    df['combined_spread_ratio'] = (df['high_low_spread'] * df['volume']) + df['high_close_open_high_ratio']
    
    # Volume-Weighted Intraday Momentum Intensity
    df['volume_weighted_intraday_momentum_intensity'] = df['combined_spread_ratio'] * df['volume']
    df['volume_weighted_intraday_momentum_intensity'] = df['volume_weighted_intraday_momentum_intensity'].rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price
    df['volume_weighted_price'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Daily Change in Volume-Weighted Price
    df['daily_change_volume_weighted_price'] = df['volume_weighted_price'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_changes'] = df['daily_change_volume_weighted_price'].apply(lambda x: max(x, 0))
    df['negative_changes'] = df['daily_change_volume_weighted_price'].apply(lambda x: -min(x, 0))
    
    # Compute 14-Day Average of Positive and Negative Changes
    df['avg_positive_changes'] = df['positive_changes'].rolling(window=14).mean()
    df['avg_negative_changes'] = df['negative_changes'].rolling(window=14).mean()
    
    # Calculate RSI
    df['relative_strength'] = df['avg_positive_changes'] / df['avg_negative_changes']
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
