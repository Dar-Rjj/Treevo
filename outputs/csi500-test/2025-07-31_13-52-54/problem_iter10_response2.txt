```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    high_open_diff = df['high'] - df['open']
    high_close_to_open_high_ratio = high_close_diff / high_open_diff
    
    # Adjust Intraday High-Low Spread by Volume
    volume_adjusted_spread = (high_low_spread * df['volume']) + df['open']
    
    # Combine Adjusted Spread and Ratio
    factor = volume_adjusted_spread + high_close_to_open_high_ratio
    
    return factor
```
