```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the intraday return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day MA to the 30-day MA
    ma_ratio = ma_10 / ma_30
    
    # Calculate the 10-day standard deviation of the closing price
    std_10 = df['close'].rolling(window=10).std()
    
    # Calculate the volatility normalized intraday return
    normalized_intraday_return = intraday_return / std_10
    
    # Calculate the factor as a combination of the normalized intraday return and the MA ratio
    factor = 0.5 * normalized_intraday_return + 0.5 * (ma_ratio - 1)
    
    return factor
```
