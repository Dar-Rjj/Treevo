```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff()
    
    # Calculate the factor: Divide Price Momentum by (Volume * High-Low Spread)
    factor = price_momentum / (df['volume'] * high_low_spread)
    
    return factor
```
