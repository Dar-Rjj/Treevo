```python
import pandas as pd

def heuristics(df, n_days=20, m_days=10):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(n_days)
    open_momentum = df['open'].pct_change(n_days)
    price_momentum = (close_momentum + open_momentum) / 2

    # Calculate Enhanced Volume Adjusted Volatility
    price_range = df['high'] - df['low']
    volatility = price_range.rolling(window=n_days).std()
    volume_adjusted_volatility = volatility / df['volume']

    # Calculate Volume Surge Indicator
    n_day_volume_change = df['volume'] - df['volume'].shift(n_days)
    m_day_volume_change = df['volume'] - df['volume'].shift(m_days)
    volume_surge = n_day_volume_change + m_day_volume_change

    # Combine Factors
    combined_factor = price_momentum * volume_adjusted_volatility
    adjusted_factor = combined_factor.where(volume_surge >= 0, combined_factor * 0.8)

    return adjusted_factor
```
