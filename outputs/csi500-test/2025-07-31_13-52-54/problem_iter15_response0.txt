```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(n))
    
    # Calculate Volatility using Average True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=n).mean()
    
    # Weight by Volume
    df['enhanced_volatility'] = df['average_true_range'] * np.log(df['volume'])
    
    # Calculate Trade Frequency Adjusted Return
    df['trade_frequency'] = df['volume']
    df['trade_frequency_adjusted_return'] = df['log_return'] / np.sqrt(df['trade_frequency'])
    
    # Combine Factors
    df['combined_factor'] = (df['log_return'] * df['enhanced_volatility']) + df['trade_frequency_adjusted_return']
    
    # Apply a nonlinear transformation to the sum
    df['alpha_factor'] = np.sqrt(df['combined_factor'])
    
    return df['alpha_factor']

```

This function `heuristics_v2` calculates a novel alpha factor by combining price momentum, enhanced volume-weighted volatility, and trade frequency-adjusted returns. The final factor is then transformed nonlinearly to generate the alpha factor.
