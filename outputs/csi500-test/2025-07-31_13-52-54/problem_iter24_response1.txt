```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=lookback).sum()
    
    # Calculate Momentum Component
    current_volume_weighted_spread = (high_low_spread * df['volume'])
    past_avg_volume_weighted_spread = (high_low_spread * df['volume']).rolling(window=lookback).mean()
    momentum_component = current_volume_weighted_spread - past_avg_volume_weighted_spread
    
    # Generate Final Factor
    combined_factor = (volume_weighted_high_low_spread + momentum_component) / df['close']
    
    # Incorporate Price Trend
    n_day_moving_avg_close = df['close'].rolling(window=lookback).mean()
    price_trend = df['close'] - n_day_moving_avg_close
    combined_factor *= price_trend
    
    # Calculate Price Momentum
    price_momentum = np.log(df['close']).diff(lookback)
    
    # Calculate Enhanced Volume-Weighted Volatility
    volatility = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=lookback).mean()
    enhanced_volatility = volatility * np.log(df['volume'])
    
    # Calculate Trade Frequency Adjusted Return
    daily_trade_frequency = df['volume'].ne(0).astype(int)
    trade_frequency_adjusted_return = (df['close'].pct_change(lookback) * np.sqrt(daily_trade_frequency.rolling(window=lookback).sum()))
    
    # Combine Factors
    combined_price_momentum_volatility = price_momentum * enhanced_volatility
    combined_factors = combined_price_momentum_volatility + trade_frequency_adjusted_return
    transformed_combined_factors = combined_factors ** 2
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor * transformed_combined_factors
    
    return final_alpha_factor
```
