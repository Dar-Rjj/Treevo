```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided
    DataFrame. The alpha factor is designed to predict future stock returns using current
    and past data.

    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as the index and columns representing
                       various market features (open, high, low, close, amount, volume).

    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """

    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate the 21-day moving average of the daily return
    df['ma_21'] = df['daily_return'].rolling(window=21).mean()

    # Calculate the 21-day exponential moving average of the close price
    df['ema_21'] = df['close'].ewm(span=21, adjust=False).mean()

    # Calculate the 21-day standard deviation of the close price
    df['std_21'] = df['close'].rolling(window=21).std()

    # Calculate the normalized volume (volume divided by its 21-day mean)
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=21).mean()

    # Calculate the 21-day moving average of the high-low range
    df['hl_range_21'] = df['high'] - df['low']
    df['ma_hl_range_21'] = df['hl_range_21'].rolling(window=21).mean()

    # Calculate the factor as a weighted sum of the above components
    factor = (
        0.3 * (df['close'] - df['ema_21']) / df['std_21'] +  # Momentum component
        0.2 * (df['norm_volume'] - 1) +  # Volume anomaly component
        0.2 * (df['hl_range_21'] - df['ma_hl_range_21']) / df['std_21'] +  # Volatility component
        0.3 * (df['daily_return'] - df['ma_21'])  # Mean reversion component
    )

    return factor
```
