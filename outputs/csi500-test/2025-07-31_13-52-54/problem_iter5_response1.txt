```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Intraday Return
    df['vol_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Volume-Weighted Intraday Momentum Factor
    df['vol_weighted_momentum'] = df['vol_weighted_intraday_return'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['vol_weighted_reversal'] = -df['vol_weighted_intraday_return'].rolling(window=10).sum()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    df['high_close_ratio'] = (df['high'] - df['close']) / (df['high'] - df['open'])
    
    # Combine Spread and Ratio
    df['weighted_spread'] = df['intraday_high_low_spread'] * df['volume']
    df['combined_intensity'] = df['weighted_spread'] + df['high_close_ratio']
    
    # Volume-Weighted Intraday Momentum Intensity
    df['vol_weighted_momentum_intensity'] = (df['combined_intensity'] * df['volume']).rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Daily Change in Volume-Weighted Price
    df['daily_change_vwap'] = df['vwap'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_change'] = df['daily_change_vwap'].apply(lambda x: x if x > 0 else 0)
    df['negative_change'] = df['daily_change_vwap'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # Compute 14-Day Average of Positive and Negative Changes
    df['avg_positive_change'] = df['positive_change'].rolling(window=14).mean()
    df['avg_negative_change'] = df['negative_change'].rolling(window=14).mean()
    
    # Calculate RSI
    df['relative_strength'] = df['avg_positive_change'] / df['avg_negative_change']
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['vol_weighted_momentum'] 
                          - df['vol_weighted_reversal'] 
                          + df['vol_weighted_momentum_intensity'] 
                          + df['rsi'])
    
    return df['alpha_factor']
```
