```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Intraday Return
    volume_weighted_intraday_return = intraday_return * df['volume']
    
    # Volume-Weighted Intraday Momentum Factor
    momentum_factor = volume_weighted_intraday_return.rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    reversal_factor = -volume_weighted_intraday_return.rolling(window=10).sum()
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['open'] - df['high']
    high_close_to_open_high_ratio = high_close_diff / open_high_diff
    
    # Combine Spread and Ratio
    combined_spread_ratio = (high_low_spread * df['volume']) + high_close_to_open_high_ratio
    
    # Volume-Weighted Intraday Momentum Intensity
    intensity_factor = combined_spread_ratio.rolling(window=5).sum()
    
    # Calculate Volume-Weighted Average Price
    volume_weighted_price = (df['close'] * df['volume']) / df['volume'].sum()
    
    # Compute Daily Change in Volume-Weighted Price
    daily_change = volume_weighted_price.diff()
    
    # Separate Positive and Negative Changes
    positive_changes = daily_change[daily_change > 0].fillna(0)
    negative_changes = -daily_change[daily_change < 0].fillna(0)
    
    # Compute 14-Day Average of Positive Changes
    avg_positive_changes = positive_changes.rolling(window=14).mean()
    
    # Compute 14-Day Average of Negative Changes
    avg_negative_changes = negative_changes.rolling(window=14).mean()
    
    # Calculate RSI
    relative_strength = avg_positive_changes / avg_negative_changes
    rsi = 100 - (100 / (1 + relative_strength))
    
    # Combine all factors into a single alpha factor
    alpha_factor = momentum_factor + reversal_factor + intensity_factor + rsi
    
    return alpha_factor
```
