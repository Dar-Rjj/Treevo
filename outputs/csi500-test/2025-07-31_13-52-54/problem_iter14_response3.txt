```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the money flow
    money_flow_std = money_flow.rolling(window=10).std()
    
    # Calculate the money flow z-score
    money_flow_zscore = (money_flow - money_flow_ma) / money_flow_std
    
    # Calculate the 5-day momentum of the close price
    momentum_5d = df['close'].pct_change(periods=5)
    
    # Combine the money flow z-score and momentum into a single alpha factor
    alpha_factor = (money_flow_zscore + momentum_5d) / 2
    
    return alpha_factor
```
