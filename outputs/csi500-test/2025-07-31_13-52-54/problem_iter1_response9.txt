```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(periods=n)
    open_momentum = df['open'].pct_change(periods=n)
    
    # Average the two momentum measures
    price_momentum = (close_momentum + open_momentum) / 2
    
    # Calculate Volume Adjusted Volatility
    price_range = df['high'] - df['low']
    volatility = price_range.rolling(window=n).std()
    volume_adjusted_volatility = volatility / df['volume']
    
    # Combine Factors
    alpha_factor = price_momentum * volume_adjusted_volatility
    
    return alpha_factor
```
