```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['high'] - df['open']
    high_close_open_high_ratio = high_close_diff / (open_high_diff.replace(0, 1))  # Avoid division by zero
    
    # Adjust Close Price by Volume
    volume_adjusted_close = df['close'] * df['volume']
    
    # Combine Spread, Ratio, and Volume-Adjusted Close
    weighted_spread = intraday_spread * df['volume']
    combined_factor = weighted_spread + high_close_open_high_ratio + volume_adjusted_close
    
    return combined_factor
```
