import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Long-Term Volume-Weighted Average Return (Momentum)
    long_term_vol_weighted_return = (df['daily_return'] * df['volume']).rolling(window=120).sum() / df['volume'].rolling(window=120).sum()

    # Short-Term Volume-Weighted Average Return (Reversal)
    short_term_vol_weighted_return = (df['daily_return'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()

    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['open'] * df['volume']
    intraday_momentum_7d = df['intraday_momentum'].rolling(window=7).sum()

    # Volume-Weighted Intraday Reversal
    df['intraday_momentum_reversal'] = -df['intraday_momentum']
    intraday_momentum_reversal_15d = df['intraday_momentum_reversal'].rolling(window=15).sum()

    # Combined Alpha Factor
    combined_alpha_factor = (long_term_vol_weighted_return - short_term_vol_weighted_return) + (intraday_momentum_7d + intraday_momentum_reversal_15d)

    # Adjusted Intraday High-Low Spread
    adjusted_intraday_high_low_spread = (df['high'] - df['low']) * df['volume'] + df['close']

    # Final Alpha Factor
    final_alpha_factor = combined_alpha_factor * adjusted_intraday_high_low_spread

    return final_alpha_factor
