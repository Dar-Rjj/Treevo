```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_returns = df['returns'].rolling(window=100).sum()
    long_term_volume_sum = df['volume'].rolling(window=100).sum()
    long_term_weighted_returns = (df['returns'] * df['volume']).rolling(window=100).sum() / long_term_volume_sum
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_returns = df['returns'].rolling(window=5).sum()
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    short_term_weighted_returns = (df['returns'] * df['volume']).rolling(window=5).sum() / short_term_volume_sum
    
    # Integrate High-Low Range into Volume-Weighted Returns
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low_range = high_low_range * df['volume']
    adjusted_volume_weighted_high_low_range = volume_weighted_high_low_range / short_term_volume_sum
    
    # Combine Momentum, Reversal, and High-Low Range Components
    alpha_factor = long_term_weighted_returns - short_term_weighted_returns + adjusted_volume_weighted_high_low_range
    
    return alpha_factor
```
