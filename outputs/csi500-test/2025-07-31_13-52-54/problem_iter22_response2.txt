```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Averaged Price
    volume_weighted_price_sum = (df['close'] * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    volume_averaged_price = volume_weighted_price_sum / total_volume
    
    # Compute Ratio
    ratio = high_low_spread / volume_averaged_price
    
    return ratio
```
