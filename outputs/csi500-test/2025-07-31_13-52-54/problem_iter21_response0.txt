```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=10, n_days=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = high_low_spread * df['volume']
    volume_weighted_high_low_spread_sum = volume_weighted_high_low_spread.rolling(window=lookback_period).sum()
    
    # Calculate Momentum Component
    current_volume_weighted_spread = volume_weighted_high_low_spread
    past_volume_weighted_spread_avg = volume_weighted_high_low_spread.rolling(window=lookback_period).mean()
    momentum_component = current_volume_weighted_spread - past_volume_weighted_spread_avg
    
    # Generate Final Factor
    combined_factor = (volume_weighted_high_low_spread_sum + momentum_component) / df['close']
    
    # Incorporate Price Trend
    n_day_moving_avg = df['close'].rolling(window=n_days).mean()
    price_trend = df['close'] - n_day_moving_avg
    factor_with_trend = combined_factor * price_trend
    
    # Calculate Price Momentum
    log_return = np.log(df['close'] / df['close'].shift(n_days))
    
    # Calculate Enhanced Volume-Weighted Volatility
    volatility = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=n_days).mean()
    enhanced_volatility = volatility * np.log(df['volume'])
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].ne(0).astype(int)
    trade_frequency_adjusted_return = (df['close'] / df['close'].shift(n_days) - 1) / np.cbrt(trade_frequency.rolling(window=n_days).sum())
    
    # Combine Factors
    combined_price_momentum_volatility = log_return * enhanced_volatility
    combined_factors = combined_price_momentum_volatility + trade_frequency_adjusted_return
    transformed_combined_factors = np.exp(combined_factors)
    
    # Final Alpha Factor
    final_alpha_factor = factor_with_trend * transformed_combined_factors
    
    return final_alpha_factor
```
