```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Price
    volume_weighted_price = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    
    # Compute Ratio: Volume-Averaged High-Low Spread Ratio
    vol_avg_high_low_ratio = high_low_spread / volume_weighted_price
    
    # Develop trend following alpha factor using close prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Explore momentum in high and low prices
    momentum_high = df['high'] - df['high'].shift(1)
    momentum_low = df['low'] - df['low'].shift(1)
    combined_momentum_score = momentum_high + momentum_low
    momentum_signal = (combined_momentum_score > 0).astype(int) - (combined_momentum_score < 0).astype(int)
    
    # Analyze volume patterns
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_intensity = (volume_change > 0.05).astype(int) - (volume_change < -0.05).astype(int)
    
    # Leverage price and volume for an intensity factor
    return_intensity = (df['close'] - df['close'].shift(1)) * df['volume']
    smoothed_intensity = return_intensity.rolling(window=20).mean()
    
    # Calculate Volume-Weighted High-Low Spread
    vol_weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    avg_vol_weighted_high_low_spread = vol_weighted_high_low_spread.rolling(window=50).mean()
    
    # Calculate Momentum Component
    momentum_component = vol_weighted_high_low_spread - avg_vol_weighted_high_low_spread
    
    # Generate Final Factor
    final_factor = (momentum_component + vol_weighted_high_low_spread) / df['close']
    
    # Incorporate Price Trend
    n_day_ma = df['close'].rolling(window=50).mean()
    price_trend = df['close'] - n_day_ma
    final_factor *= price_trend
    
    # Synthesize Factors
    composite_factor = (trend_signal + momentum_signal + smoothed_intensity) * vol_avg_high_low_ratio
    
    return composite_factor
```
