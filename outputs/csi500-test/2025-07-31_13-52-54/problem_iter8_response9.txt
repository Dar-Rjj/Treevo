```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel and interpretable alpha factor based on the given market features.
    The factor is designed to help predict future stock returns using only current and past data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with the following columns: open, high, low, close, amount, volume.
                       The index of the DataFrame is the date.
    
    Returns:
    pd.Series: A Series with the computed alpha factor values, indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume weighted price
    df['vwp'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the 21-day moving average of the volume weighted price
    df['vwp_21_ma'] = df['vwp'].rolling(window=21).mean()
    
    # Calculate the 21-day standard deviation of the volume weighted price
    df['vwp_21_std'] = df['vwp'].rolling(window=21).std()
    
    # Calculate the z-score of the volume weighted price
    df['vwp_z_score'] = (df['vwp'] - df['vwp_21_ma']) / df['vwp_21_std']
    
    # Calculate the 5-day momentum
    df['momentum_5'] = df['close'].pct_change(periods=5)
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['vwp_z_score'] * df['momentum_5']) + df['intraday_range']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
