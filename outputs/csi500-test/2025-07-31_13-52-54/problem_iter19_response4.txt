```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['CloseVolume'] = df['close'] * df['volume']
    df['VWAP'] = df['CloseVolume'].rolling(window=1).sum() / df['volume'].rolling(window=1).sum()

    # Daily VWAP Change
    df['VWAPChange'] = df['VWAP'] - df['VWAP'].shift(1)

    # Separate Positive and Negative Changes
    positive_changes = df['VWAPChange'].where(df['VWAPChange'] > 0, 0)
    negative_changes = -df['VWAPChange'].where(df['VWAPChange'] < 0, 0)

    # 14-Day Averages for RSI
    pos_avg = positive_changes.rolling(window=14).mean()
    neg_avg = negative_changes.rolling(window=14).mean()

    # Calculate RSI
    relative_strength = pos_avg / neg_avg
    rsi = 100 - (100 / (1 + relative_strength))

    # Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_range = df['high'].shift(1) - df['low'].shift(1)
    intraday_range_growth = (today_range - prev_day_range) / prev_day_range

    # Volume-Weighted Moving Average (VWMA)
    df['CloseVolume'] = df['close'] * df['volume']
    vwma = df['CloseVolume'].rolling(window=30).sum() / df['volume'].rolling(window=30).sum()

    # Long-Term and Short-Term Volume-Weighted Returns
    daily_returns = df['close'].pct_change()
    
    # Long-Term VW-Return (Momentum Component)
    long_term_vw_return = (daily_returns.rolling(window=90).sum() * df['volume'].rolling(window=90).sum()) / df['volume'].rolling(window=90).sum()
    
    # Short-Term VW-Return (Reversal Component)
    short_term_vw_return = (daily_returns.rolling(window=5).sum() * df['volume'].rolling(window=5).sum()) / df['volume'].rolling(window=5).sum()
    
    # Adjusted VW-Return
    adjusted_vw_return = long_term_vw_return - short_term_vw_return

    # Volatility-Adjusted Alpha Factor
    volatility = daily_returns.rolling(window=30).std()
    combined_factor = (rsi + intraday_range_growth + vwma + adjusted_vw_return) / volatility

    # Final Alpha Factor
    final_alpha_factor = combined_factor

    return final_alpha_factor
```
