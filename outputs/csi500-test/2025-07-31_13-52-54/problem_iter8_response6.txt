```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Intraday Volatility
    df['intraday_volatility'] = np.abs(df['high'] - df['low'])
    df['intraday_volatility_20d'] = df['intraday_volatility'].rolling(window=20).sum()
    
    # Intraday Range Growth
    df['today_range'] = df['high'] - df['low']
    df['prev_day_high'] = df['high'].shift(1)
    df['prev_day_low'] = df['low'].shift(1)
    df['prev_day_range'] = df['prev_day_high'] - df['prev_day_low']
    df['intraday_range_growth'] = (df['today_range'] - df['prev_day_range']) / df['prev_day_range']
    
    # Volume-Weighted Intraday Reversal
    df['volume_weighted_reversal'] = -(df['intraday_return'] * df['volume'])
    df['volume_weighted_reversal_10d'] = df['volume_weighted_reversal'].rolling(window=10).sum()
    
    # Price Momentum
    df['price_momentum_close'] = df['close'].pct_change(periods=5)
    df['price_momentum_open'] = df['open'].pct_change(periods=5)
    df['price_momentum'] = (df['price_momentum_close'] + df['price_momentum_open']) / 2
    
    # Volume Adjusted Volatility
    df['vol_adjusted_volatility'] = df['intraday_volatility'].rolling(window=5).std() / df['volume'].rolling(window=5).mean()
    
    # Volume Weighted Moving Average
    df['close_vol_weighted'] = df['close'] * df['volume']
    df['close_vol_sum'] = df['close_vol_weighted'].rolling(window=20).sum()
    df['volume_sum'] = df['volume'].rolling(window=20).sum()
    df['volume_weighted_ma'] = df['close_vol_sum'] / df['volume_sum']
    
    # Combined Factor
    df['combined_factor'] = (df['intraday_return'] + df['volume_weighted_reversal_10d']) * df['price_momentum'] / df['vol_adjusted_volatility']
    
    # Volume Surge
    df['n_day_volume_change'] = df['volume'] - df['volume'].shift(5)
    df['m_day_volume_change'] = df['volume'] - df['volume'].shift(10)
    df['volume_surge'] = df['n_day_volume_change'] + df['m_day_volume_change']
    df['adjusted_volume_surge'] = np.where(df['volume_surge'] < 0, df['volume_surge'] * 0.8, df['volume_surge'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['intraday_range_growth'] + df['volume_weighted_ma'] + df['combined_factor'] + df['adjusted_volume_surge']) / 4
    
    return df['final_alpha_factor']
```
