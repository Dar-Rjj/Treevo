import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Price
    volume_weighted_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Ratio: Volume-Averaged High-Low Spread Ratio
    volume_averaged_high_low_ratio = high_low_spread / volume_weighted_price
    
    # Develop trend following alpha factor using close prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Explore momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum_score = high_momentum + low_momentum
    momentum_threshold = 0.5 * (combined_momentum_score.abs().quantile(0.75))
    momentum_signal = (combined_momentum_score > momentum_threshold).astype(int) - (combined_momentum_score < -momentum_threshold).astype(int)
    
    # Analyze volume patterns
    volume_change = df['volume'].pct_change()
    strong_volume = (volume_change > 0.2).astype(int)
    moderate_volume = ((volume_change > 0.1) & (volume_change <= 0.2)).astype(int)
    weak_volume = (volume_change <= 0.1).astype(int)
    
    # Leverage price and volume for an intensity factor
    return_intensity = df['close'].pct_change() * df['volume']
    intensity_factor = return_intensity.rolling(window=20).mean()
    
    # Calculate Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    sum_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Momentum Component
    avg_past_volume_weighted_high_low_spread = sum_volume_weighted_high_low_spread.shift(1) / 20
    momentum_component = volume_weighted_high_low_spread - avg_past_volume_weighted_high_low_spread
    
    # Generate Intermediate Factor
    intermediate_factor = (momentum_component + volume_weighted_high_low_spread) / df['close']
    
    # Incorporate Price Trend
    n_day_ma = df['close'].rolling(window=20).mean()
    price_trend = (df['close'] - n_day_ma) * intermediate_factor
    
    # Synthesize Factors
    final_alpha_factor = (trend_signal + momentum_signal + intensity_factor) * volume_averaged_high_low_ratio
    
    return final_alpha_factor
