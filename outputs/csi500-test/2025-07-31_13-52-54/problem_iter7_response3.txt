import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Volume-Weighted Intraday Momentum Factor
    short_window = 5
    df['volumeweighted_intraday_momentum'] = df['volume_weighted_intraday_return'].rolling(window=short_window).sum()
    
    # Volume-Weighted Intraday Reversal
    long_window = 10
    df['volumeweighted_intraday_reversal'] = -df['volume_weighted_intraday_return'].rolling(window=long_window).sum()
    
    # Calculate Volume-Weighted Average Price
    df['volume_weighted_price'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Daily Change in Volume-Weighted Price
    df['daily_change_volume_weighted_price'] = df['volume_weighted_price'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_changes'] = df['daily_change_volume_weighted_price'].apply(lambda x: max(x, 0))
    df['negative_changes'] = df['daily_change_volume_weighted_price'].apply(lambda x: -min(x, 0))
    
    # Compute 14-Day Average of Positive Changes
    rsi_window = 14
    df['avg_positive_changes'] = df['positive_changes'].rolling(window=rsi_window).mean()
    
    # Compute 14-Day Average of Negative Changes
    df['avg_negative_changes'] = df['negative_changes'].rolling(window=rsi_window).mean()
    
    # Calculate RSI
    relative_strength = df['avg_positive_changes'] / df['avg_negative_changes']
    df['rsi'] = 100 - (100 / (1 + relative_strength))
    
    # Combine Momentum, Reversal, and RSI
    df['alpha_factor'] = df['volumeweighted_intraday_momentum'] - df['volumeweighted_intraday_reversal'] + df['rsi']
    
    return df['alpha_factor']
