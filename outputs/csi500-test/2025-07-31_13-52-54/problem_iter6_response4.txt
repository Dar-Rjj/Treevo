```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['high'] - df['open']
    high_close_to_open_high_ratio = high_close_diff / (open_high_diff + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Adjust Intraday High-Low Spread by Volume
    volume_adjusted_spread = (intraday_high_low_spread * df['volume']) + df['open']
    
    # Combine Adjusted Spread and Ratio
    alpha_factor = volume_adjusted_spread + high_close_to_open_high_ratio
    
    return alpha_factor
```
