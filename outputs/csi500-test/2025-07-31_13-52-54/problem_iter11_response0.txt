```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Price and Intraday Momentum
    price_momentum = df['close'].pct_change(periods=n)
    
    intraday_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Enhanced Volume-Adjusted Volatility
    volatility = (df['high'] - df['low']).rolling(window=n).std()
    volume_adjusted_volatility = volatility * df['volume']
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].rolling(window=n).mean()
    frequency_adjusted_return = (df['close'].pct_change(periods=n) * trade_frequency)
    
    # Calculate High-Close to Open-High Ratio
    high_close_ratio = (df['high'] - df['close']) / (df['high'] - df['open'])
    
    # Combine Factors
    combined_factor = (price_momentum + intraday_high_low_spread) * volume_adjusted_volatility + frequency_adjusted_return + high_close_ratio
    
    return combined_factor
```
