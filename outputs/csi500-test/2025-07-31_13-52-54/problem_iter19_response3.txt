```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, n_days=5):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    df['volume_weighted_spread'] = df['high_low_spread'] * df['volume']
    
    # Calculate the sum of volume-weighted high-low spreads over the lookback period
    df['sum_volume_weighted_spread'] = df['volume_weighted_spread'].rolling(window=lookback_period).sum()
    
    # Calculate Momentum Component
    df['current_volume_weighted_spread'] = df['volume_weighted_spread']
    df['avg_past_spread'] = df['volume_weighted_spread'].rolling(window=n_days).mean().shift(1)
    df['momentum'] = df['current_volume_weighted_spread'] - df['avg_past_spread']
    
    # Generate Final Factor
    df['final_factor'] = (df['sum_volume_weighted_spread'] + df['momentum']) / df['close']
    
    # Incorporate Price Trend
    df['n_day_moving_avg_close'] = df['close'].rolling(window=n_days).mean()
    df['price_trend'] = df['close'] - df['n_day_moving_avg_close']
    df['combined_factor'] = df['final_factor'] * df['price_trend']
    
    # Drop intermediate columns
    factor_values = df['combined_factor'].dropna()
    
    return factor_values
```
