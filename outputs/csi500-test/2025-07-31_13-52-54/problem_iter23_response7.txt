import pandas as pd

def heuristics_v2(df, lookback_period=20, n_days=10):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    df['Volume_Weighted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    df['Volume_Weighted_High_Low_Spread_Sum'] = df['Volume_Weighted_High_Low_Spread'].rolling(window=lookback_period).sum()
    
    # Calculate Volume-Averaged Close Price
    df['Volume_Averaged_Close'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Ratio
    df['Volume_Weighted_High_Low_Spread_Ratio'] = df['Volume_Weighted_High_Low_Spread_Sum'] / df['Volume_Averaged_Close']
    
    # Calculate Momentum Component
    df['Momentum_Component'] = df['Volume_Weighted_High_Low_Spread_Sum'] - df['Volume_Weighted_High_Low_Spread_Sum'].rolling(window=n_days).mean()
    
    # Generate Preliminary Factor
    df['Preliminary_Factor'] = df['Volume_Weighted_High_Low_Spread_Ratio'] + df['Momentum_Component']
    
    # Incorporate Price Trend
    df['n_Day_Moving_Average'] = df['close'].rolling(window=n_days).mean()
    df['Price_Trend'] = df['close'] - df['n_Day_Moving_Average']
    df['Preliminary_Factor_with_Trend'] = df['Preliminary_Factor'] * df['Price_Trend']
    
    # Calculate Volume-Averaged Price
    df['Volume_Averaged_Price'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Averaged High-Low Spread Ratio
    df['Volume_Averaged_High_Low_Spread_Ratio'] = df['High_Low_Spread'] / df['Volume_Averaged_Price']
    
    # Generate Final Combined Factor
    df['Combined_Factor'] = df['Preliminary_Factor_with_Trend'] * df['Volume_Averaged_High_Low_Spread_Ratio']
    
    return df['Combined_Factor']
