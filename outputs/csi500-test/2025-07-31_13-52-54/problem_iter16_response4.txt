```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum)
    df['long_term_vol_wgt_return'] = (df['daily_return'] * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal)
    df['short_term_vol_wgt_return'] = (df['daily_return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['close'] * df['volume']
    df['intraday_momentum_5d'] = df['intraday_momentum'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['vol_wgt_intraday_reversal'] = -df['intraday_momentum']
    df['vol_wgt_intraday_reversal_10d'] = df['vol_wgt_intraday_reversal'].rolling(window=10).sum()
    
    # Combined Alpha Factor
    df['combined_alpha_factor'] = (df['long_term_vol_wgt_return'] - df['short_term_vol_wgt_return']) + (df['intraday_momentum_5d'] + df['vol_wgt_intraday_reversal_10d'])
    
    # Adjusted Intraday High-Low Spread
    df['adjusted_intraday_high_low_spread'] = (df['high'] - df['low']) * df['volume'] + df['open']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['adjusted_intraday_high_low_spread']
    
    return df['final_alpha_factor'].dropna()
```
