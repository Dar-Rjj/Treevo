```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Volume-Weighted Volatility
    range_volatility = (df['high'] - df['low']).rolling(window=n).std()
    volume_weighted_volatility = range_volatility * df['volume']
    
    # Calculate Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_high = df['high'].shift(1)
    prev_day_low = df['low'].shift(1)
    prev_day_range = prev_day_high - prev_day_low
    intraday_range_growth = (today_range - prev_day_range) / prev_day_range
    
    # Volume Weighted Moving Average
    volume_weighted_close = df['close'] * df['volume']
    volume_sum = df['volume'].rolling(window=n).sum()
    vwma = volume_weighted_close.rolling(window=n).sum() / volume_sum
    
    # Calculate Amount Adjusted Return
    amount_adjusted_return = df['amount'].pct_change(n)
    
    # Combine Factors
    combined_factors = (close_momentum + open_momentum) * volume_weighted_volatility + amount_adjusted_return + intraday_range_growth + vwma
    
    # Calculate Price Momentum (Alternative)
    close_diff_n = df['close'] - df['close'].shift(n)
    close_diff_m = df['close'] - df['close'].shift(m)
    price_momentum_alt = close_diff_n + close_diff_m
    
    # Calculate Volume Surge
    volume_surge_n = df['volume'] - df['volume'].shift(n)
    volume_surge_m = df['volume'] - df['volume'].shift(m)
    volume_surge = volume_surge_n + volume_surge_m
    
    # Calculate Price Reversal
    high_low_range_n = df['high'] - df['low']
    high_low_range_m = df['high'].shift(m) - df['low'].shift(m)
    price_reversal = (high_low_range_n - high_low_range_m) / high_low_range_m
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * price_momentum_alt
    if volume_surge < 0:
        final_alpha_factor *= 0.8
    if price_reversal > 0:
        final_alpha_factor *= 1.2
    
    return final_alpha_factor
```
