import pandas as pd

def heuristics_v2(df, n=50):
    # Calculate 50-day Simple Moving Average (SMA) of closing prices
    df['SMA'] = df['close'].rolling(window=n).mean()
    
    # Generate a binary trend signal; 1 if the current close is above the SMA, 0 otherwise
    df['Trend_Signal'] = (df['close'] > df['SMA']).astype(int)
    
    # Calculate the difference between today’s high and the previous day’s high
    df['High_Diff'] = df['high'] - df['high'].shift(1)
    
    # Calculate the difference between today’s low and the previous day’s low
    df['Low_Diff'] = df['low'] - df['low'].shift(1)
    
    # Sum the differences to get a combined momentum score
    df['Combined_Momentum'] = df['High_Diff'] + df['Low_Diff']
    
    # Create a simple threshold to classify stocks as having positive or negative momentum based on the score
    df['Momentum_Signal'] = (df['Combined_Momentum'] > 0).astype(int)
    
    # Calculate the daily volume percentage change from the previous day
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * 100
    
    # Assign a value indicating strong buying (positive change) or selling (negative change) pressure
    df['Volume_Pressure'] = (df['Volume_Change'] > 0).astype(int)
    
    # Calculate n-day return using Close price
    df['Return_n_day'] = (df['close'] / df['close'].shift(n)) - 1
    
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Multiply (High - Low) by Volume
    df['Volume_Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Calculate range over n days using High and Low prices
    df['Range_n_days'] = df['high'].rolling(window=n).max() - df['low'].rolling(window=n).min()
    
    # Calculate Volume-Adjusted Volatility
    df['Volume_Adjusted_Volatility'] = df['Range_n_days'] * df['volume']
    
    # Count daily trades using Volume
    df['Trade_Frequency'] = df['volume'] / df['amount']
    
    # n-day return adjusted by trade frequency
    df['Adjusted_Return_n_day'] = df['Return_n_day'] * df['Trade_Frequency']
    
    # Combine all factors into a composite alpha factor
    df['Composite_Factor'] = (df['Trend_Signal'] + df['Momentum_Signal'] + df['Volume_Pressure']) * (1/3)
    
    # Use Volume-Adjusted High-Low Spread Ratio as a modifier to adjust the final alpha factor
    df['Volume_Adjusted_Ratio'] = df['Volume_Adjusted_High_Low_Spread'] / (df['close'] * df['volume'])
    
    # Final Composite Alpha Factor
    df['Final_Alpha_Factor'] = df['Composite_Factor'] * df['Volume_Adjusted_Ratio'] + df['Adjusted_Return_n_day']
    
    return df['Final_Alpha_Factor']
