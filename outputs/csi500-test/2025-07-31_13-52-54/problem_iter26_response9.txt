```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on (open, high, low, close, amount, volume) prices.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_ma_10 = money_flow.rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the typical price
    typical_price_ma_5 = typical_price.rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the typical price
    typical_price_ma_20 = typical_price.rolling(window=20).mean()
    
    # Calculate the ratio of the 5-day MA to the 20-day MA
    ma_ratio = typical_price_ma_5 / typical_price_ma_20
    
    # Calculate the factor as the product of the MA ratio and the normalized money flow
    factor = ma_ratio * (money_flow - money_flow_ma_10) / money_flow.std()
    
    return factor
```
