```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    df['Long_Term_VW_Return'] = (df['Daily_Return'] * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    df['Short_Term_VW_Return'] = (df['Daily_Return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # High-Close to Open-High Ratio
    df['High_Close_Ratio'] = (df['high'] - df['close']) / (df['high'] - df['open'])
    
    # Intraday Return
    df['Intraday_Return'] = df['Intraday_Spread'] / df['close']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = df['Intraday_Spread'].abs().rolling(window=20).sum()
    
    # Intraday Momentum
    df['Intraday_Momentum'] = (df['Intraday_Return'] * df['volume']).rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['Volume_Weighted_Reversal'] = (-df['Intraday_Return'] * df['volume']).rolling(window=10).sum()
    
    # Price Momentum
    df['Close_Price_Momentum'] = df['close'].pct_change(periods=10)
    df['Open_Price_Momentum'] = df['open'].pct_change(periods=10)
    
    # Volume Adjusted Volatility
    df['Vol_Adjusted_Volatility'] = df['Intraday_Spread'].rolling(window=10).std() / df['volume'].rolling(window=10).mean()
    
    # Combined Alpha Factor
    df['Combined_Alpha_Factor'] = (
        (df['Long_Term_VW_Return'] - df['Short_Term_VW_Return']) +
        (df['Intraday_Momentum'] + df['Volume_Weighted_Reversal'])
    ) * (df['Close_Price_Momentum'] + df['Open_Price_Momentum']) / df['Vol_Adjusted_Volatility']
    
    # Final Combined Adjusted Spread and Ratio
    df['Adjusted_Spread'] = df['Intraday_Spread'] * df['volume'] + df['open']
    df['Final_Combined_Adjusted_Spread'] = df['Adjusted_Spread'] + df['High_Close_Ratio']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Final_Combined_Adjusted_Spread']
    
    return df['Final_Alpha_Factor']
```
