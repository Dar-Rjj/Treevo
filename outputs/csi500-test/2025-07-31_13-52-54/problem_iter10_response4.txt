```python
import pandas as pd

def heuristics_v2(df):
    """
    Develop a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the 50-day Simple Moving Average (SMA) of closing prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int)
    
    # Calculate momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum = high_momentum + low_momentum
    momentum_signal = (combined_momentum > 0).astype(int)
    
    # Evaluate the percentage change in volume from the previous day
    volume_change = df['volume'].pct_change()
    volume_signal = (volume_change > 0).astype(int)
    
    # Leverage price and volume for an intensity factor
    close_return = df['close'].pct_change()
    intensity_factor = close_return * df['volume']
    
    # Combine all signals into a single alpha factor
    alpha_factor = (trend_signal + momentum_signal + volume_signal + intensity_factor).dropna()
    
    return alpha_factor
```
