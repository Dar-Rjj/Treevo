```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).rolling(window=1).sum() / df['volume'].rolling(window=1).sum()
    
    # Daily VWAP Change
    df['Daily_VWAP_Change'] = df['VWAP'] - df['VWAP'].shift(1)
    
    # Separate Positive and Negative Changes
    positive_changes = df['Daily_VWAP_Change'].apply(lambda x: x if x > 0 else 0)
    negative_changes = df['Daily_VWAP_Change'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # 14-Day Averages for RSI
    avg_positive_14 = positive_changes.rolling(window=14).mean()
    avg_negative_14 = negative_changes.rolling(window=14).mean()
    
    # Calculate RSI
    relative_strength = avg_positive_14 / avg_negative_14
    df['RSI'] = 100 - (100 / (1 + relative_strength))
    
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    df['Volume_Adjusted_High_Low_Spread'] = df['Intraday_High_Low_Spread'] * df['volume']
    
    # High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['open'] - df['high']
    df['High_Close_to_Open_High_Ratio'] = high_close_diff / open_high_diff
    
    # Daily Returns
    df['Daily_Returns'] = df['close'].pct_change()
    
    # Long-Term VW-Return (Momentum Component)
    long_term_returns = df['Daily_Returns'].rolling(window=90).sum() * df['volume'].rolling(window=90).sum()
    total_volume_90 = df['volume'].rolling(window=90).sum()
    df['Long_Term_VW_Return'] = long_term_returns / total_volume_90
    
    # Short-Term VW-Return (Reversal Component)
    short_term_returns = df['Daily_Returns'].rolling(window=5).sum() * df['volume'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    df['Short_Term_VW_Return'] = short_term_returns / total_volume_5
    
    # Adjusted VW-Return
    df['Adjusted_VW_Return'] = df['Long_Term_VW_Return'] - df['Short_Term_VW_Return']
    
    # Momentum Component
    current_day_momentum = (df['high'] - df['low']) * df['volume']
    past_n_days_momentum = ((df['high'] - df['low']) * df['volume']).rolling(window=90).sum() / df['volume'].rolling(window=90).sum()
    df['Momentum_Component'] = current_day_momentum - past_n_days_momentum
    
    # Price-Volume Imbalance Factor
    price_volume_imbalance = (df['close'] - df['open']) * df['volume']
    df['Price_Volume_Imbalance'] = price_volume_imbalance.rolling(window=90).sum()
    
    # Combine Volume-Adjusted High-Low Spread, High-Close to Open-High Ratio, and Price-Volume Imbalance
    df['Combined_Factor'] = df['Volume_Adjusted_High_Low_Spread'] + df['High_Close_to_Open_High_Ratio'] + df['Price_Volume_Imbalance']
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['RSI'] + df['Intraday_High_Low_Spread'] + df['Adjusted_VW_Return'] + 
                                df['Momentum_Component'] + df['Combined_Factor'])
    
    return df['Final_Alpha_Factor']
```
