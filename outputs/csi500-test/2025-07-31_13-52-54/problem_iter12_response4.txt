```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(periods=n)
    open_momentum = df['open'].pct_change(periods=n)
    price_momentum = (close_momentum + open_momentum) / 2

    # Calculate Enhanced Volume-Weighted Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=n).std()
    close_volatility = df['close'].rolling(window=n).std()
    volatility = (high_low_volatility + close_volatility) / 2
    volume_weight = df['volume'] / df['amount']
    trade_frequency = df['volume'] / df['amount']
    enhanced_volatility = volatility * (volume_weight * trade_frequency)

    # Calculate Trade Frequency Adjusted Return
    daily_trades = df['volume']
    trade_frequency_adjusted_return = (df['close'].pct_change(periods=n) * daily_trades.rolling(window=n).mean()) / daily_trades.rolling(window=n).mean()

    # Combine Factors
    alpha_factor = price_momentum * enhanced_volatility + trade_frequency_adjusted_return

    return alpha_factor
```
