```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Daily Change in Volume-Weighted Price
    df['vwap_change'] = df['vwap'].diff()
    
    # Separate Positive and Negative Changes
    df['positive_change'] = df['vwap_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_change'] = df['vwap_change'].apply(lambda x: -x if x < 0 else 0)
    
    # Compute 14-Day Averages of Positive and Negative Changes
    df['avg_positive_14'] = df['positive_change'].rolling(window=14).mean()
    df['avg_negative_14'] = df['negative_change'].rolling(window=14).mean()
    
    # Calculate RSI
    df['relative_strength'] = df['avg_positive_14'] / df['avg_negative_14']
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Calculate 21-Day Exponential Moving Average (EMA) of Volume-Weighted Price
    df['ema_21'] = df['vwap'].ewm(span=21, adjust=False).mean()
    
    # Compute Momentum Factor
    df['momentum_factor'] = (df['vwap'] - df['ema_21']) / df['ema_21'] * 100
    
    return df['momentum_factor']
```
