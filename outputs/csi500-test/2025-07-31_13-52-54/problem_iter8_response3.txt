```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * (df['volume'] / avg_volume_5_days)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Adjusted Momentum Factor
    adjusted_momentum_factor = price_momentum / weighted_high_low_spread
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_weighted_return = (daily_returns * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_weighted_return = (daily_returns * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Combine Adjusted Momentum and Reversal Components
    final_alpha_factor = (long_term_weighted_return - short_term_weighted_return) + adjusted_momentum_factor
    
    return final_alpha_factor
```
