```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Daily VWAP Change
    df['vwap_change'] = df['vwap'].diff()
    
    # Separate Positive and Negative Changes
    positive_changes = df[df['vwap_change'] > 0]['vwap_change']
    negative_changes = -df[df['vwap_change'] < 0]['vwap_change']
    
    # 14-Day Averages for RSI
    avg_gain = positive_changes.rolling(window=14).mean()
    avg_loss = negative_changes.rolling(window=14).mean()
    
    # Calculate RSI
    relative_strength = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + relative_strength))
    df['rsi'] = rsi
    
    # Intraday Range Growth
    df['today_range'] = df['high'] - df['low']
    df['prev_day_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['intraday_range_growth'] = (df['today_range'] - df['prev_day_range']) / df['prev_day_range']
    
    # Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Long-Term and Short-Term Volume-Weighted Returns
    df['weighted_return_90'] = df['daily_return'] * df['volume']
    long_term_vw_return = df['weighted_return_90'].rolling(window=90).sum() / df['volume'].rolling(window=90).sum()
    
    df['weighted_return_5'] = df['daily_return'] * df['volume']
    short_term_vw_return = df['weighted_return_5'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    df['adjusted_vw_return'] = long_term_vw_return - short_term_vw_return
    
    # Volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['rsi'] + df['intraday_range_growth'] + df['adjusted_vw_return']) / df['volatility']
    
    return df['alpha_factor']
```
