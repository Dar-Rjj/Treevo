import pandas as pd

def heuristics_v2(df, lookback_period=20, n_days=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=lookback_period).sum()
    
    # Calculate Volume-Averaged Close Price
    volume_averaged_close_price = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Weighted High-Low Spread Ratio
    volume_weighted_high_low_spread_ratio = volume_weighted_high_low_spread / volume_averaged_close_price
    
    # Calculate Momentum Component
    current_volume_weighted_high_low_spread = (high_low_spread * df['volume'])
    past_average_volume_weighted_high_low_spread = current_volume_weighted_high_low_spread.rolling(window=n_days).mean().shift(1)
    momentum_component = current_volume_weighted_high_low_spread - past_average_volume_weighted_high_low_spread
    
    # Generate Preliminary Factor
    preliminary_factor = volume_weighted_high_low_spread_ratio + momentum_component
    
    # Incorporate Price Trend
    n_day_moving_average_close = df['close'].rolling(window=n_days).mean()
    price_trend = df['close'] - n_day_moving_average_close
    price_trend_factor = preliminary_factor * price_trend
    
    # Calculate Volume-Average Price
    volume_averaged_price = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Averaged High-Low Spread Ratio
    volume_averaged_high_low_spread_ratio = high_low_spread / volume_averaged_price
    
    # Integrate Volume-Averaged High-Low Spread Ratio
    integrated_factor = price_trend_factor * volume_averaged_high_low_spread_ratio
    
    # Calculate Price-Volatility Interaction
    close_price_std = df['close'].rolling(window=lookback_period).std()
    price_volatility_interaction = (df['close'] - n_day_moving_average_close) / close_price_std
    final_combined_factor = integrated_factor * price_volatility_interaction
    
    return final_combined_factor
