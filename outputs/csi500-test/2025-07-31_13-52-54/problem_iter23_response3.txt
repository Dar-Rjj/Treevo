import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on historical price and volume data.
    The factor is designed to be predictive of future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume'
                       indexed by 'date'.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by 'date'.
    """
    # Calculate the 5-day moving average of the closing price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day sum of the volume
    df['sum_5_volume'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the 10-day standard deviation of the closing price
    df['std_10_close'] = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day exponential moving average of the high price
    df['ema_10_high'] = df['high'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day cumulative sum of the amount
    df['cumsum_20_amount'] = df['amount'].rolling(window=20).sum()
    
    # Alpha factor calculation
    # This factor combines the ratio of the current close to its 5-day moving average,
    # the ratio of the current volume to its 5-day sum, and the ratio of the 10-day standard deviation
    # to the 10-day EMA of the high price, scaled by the 20-day cumulative sum of the amount.
    df['alpha_factor'] = (df['close'] / df['ma_5_close']) * \
                         (df['volume'] / df['sum_5_volume']) * \
                         (df['std_10_close'] / df['ema_10_high']) * \
                         (1 / df['cumsum_20_amount'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
