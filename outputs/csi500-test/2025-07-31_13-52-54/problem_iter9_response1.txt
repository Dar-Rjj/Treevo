```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Compute Volume-Weighted High-Low Spread
    df['high_low_spread'] = (df['high'] - df['low']) * df['volume']
    
    # Summarize Over Lookback Period
    cumulative_high_low_spread = df['high_low_spread'].rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate Momentum
    current_day_high_low_spread = df['high_low_spread']
    avg_high_low_spread = cumulative_high_low_spread / total_volume
    momentum = current_day_high_low_spread - avg_high_low_spread
    
    # Incorporate Price Trend
    price_change = df['close'] - df['open']
    avg_price_change = price_change.rolling(window=lookback_period).mean()
    
    # Add Price Trend to Momentum
    factor = momentum + avg_price_change
    
    return factor
```
