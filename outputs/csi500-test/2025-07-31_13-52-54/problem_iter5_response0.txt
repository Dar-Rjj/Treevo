```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_returns = df['daily_return'].rolling(window=100).sum()
    long_term_volume = df['volume'].rolling(window=100).sum()
    long_term_vol_weighted_return = (df['daily_return'] * df['volume']).rolling(window=100).sum() / long_term_volume

    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_returns = df['daily_return'].rolling(window=5).sum()
    short_term_volume = df['volume'].rolling(window=5).sum()
    short_term_vol_weighted_return = (df['daily_return'] * df['volume']).rolling(window=5).sum() / short_term_volume

    # Integrate High-Low Range into Volume-Weighted Returns
    high_low_range = df['high'] - df['low']
    vol_weighted_high_low_range = (high_low_range * df['volume']) / short_term_volume

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['open'] - df['high']
    high_close_to_open_high_ratio = high_close_diff / open_high_diff

    # Adjust Close Price by Volume
    vol_adjusted_close = df['close'] * df['volume']

    # Combine Momentum, Reversal, and Intraday Components
    alpha_factor = (long_term_vol_weighted_return - short_term_vol_weighted_return +
                    vol_weighted_high_low_range + intraday_high_low_spread +
                    high_close_to_open_high_ratio + vol_adjusted_close)

    return alpha_factor
```
