```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Price
    volume_weighted_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Ratio: Volume-Averaged High-Low Spread Ratio
    volume_averaged_high_low_ratio = high_low_spread / volume_weighted_price
    
    # Develop trend following alpha factor using close prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Explore momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum_score = high_momentum + low_momentum
    momentum_threshold = 0.5 * combined_momentum_score.std()
    momentum_signal = (combined_momentum_score > momentum_threshold).astype(int) - (combined_momentum_score < -momentum_threshold).astype(int)
    
    # Analyze volume patterns
    volume_change = df['volume'].pct_change()
    volume_signal = (volume_change > 0.1).astype(int) - (volume_change < -0.1).astype(int)
    
    # Leverage price and volume for an intensity factor
    daily_return = df['close'].pct_change()
    intensity_factor = daily_return * df['volume']
    smoothed_intensity_factor = intensity_factor.rolling(window=20).mean()
    
    # Calculate Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    lookback_period = 20
    past_average_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=lookback_period).mean()
    momentum_component = volume_weighted_high_low_spread - past_average_volume_weighted_high_low_spread
    
    # Generate Final Factor
    final_factor = (momentum_component + volume_weighted_high_low_spread) / df['close']
    
    # Incorporate Price Trend
    price_trend = df['close'] - sma_50
    adjusted_final_factor = final_factor * price_trend
    
    # Synthesize Factors
    composite_alpha_factor = (trend_signal + momentum_signal + volume_signal + smoothed_intensity_factor) * volume_averaged_high_low_ratio
    return composite_alpha_factor
```
