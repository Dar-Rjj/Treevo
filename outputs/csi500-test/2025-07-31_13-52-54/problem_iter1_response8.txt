```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff()
    
    # Adjust by Volume and Spread
    volume_adjusted_spread = df['volume'] * high_low_spread
    adjusted_momentum = price_momentum / volume_adjusted_spread
    
    return adjusted_momentum
```
