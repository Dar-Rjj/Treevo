import pandas as pd

def heuristics_v2(df, lookback_period=20, n_days=10, m_days=5):
    # Calculate Volume-Weighted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_weighted_high_low_spread = high_low_spread * df['volume']
    
    # Compute Cumulative Volume-Weighted High-Low Spread
    cumulative_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    cumulative_volume_weighted_high_low_spread /= total_volume
    
    # Calculate Momentum Indicator
    current_day_spread = volume_weighted_high_low_spread
    past_average_spread = cumulative_volume_weighted_high_low_spread.shift(1)
    momentum_indicator = current_day_spread - past_average_spread
    
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change(n_days)
    
    # Calculate Volume-Weighted Volatility
    daily_range = df['high'] - df['low']
    volatility = daily_range.rolling(window=n_days).std()
    volume_weighted_volatility = volatility * df['volume']
    
    # Calculate Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_range = (df['high'].shift(1) - df['low'].shift(1))
    intraday_range_growth = (today_range - prev_day_range) / prev_day_range
    
    # Volume Weighted Moving Average
    close_prices = df['close']
    volume_weighted_close = close_prices * df['volume']
    volume_weighted_moving_average = volume_weighted_close.rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()
    
    # Combine Factors
    combined_factors = (price_momentum * volume_weighted_volatility + 
                        momentum_indicator + 
                        intraday_range_growth + 
                        volume_weighted_moving_average)
    
    # Calculate Volume Surge
    n_day_volume_change = df['volume'] - df['volume'].shift(n_days)
    m_day_volume_change = df['volume'] - df['volume'].shift(m_days)
    volume_surge = n_day_volume_change + m_day_volume_change
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors.copy()
    final_alpha_factor[volume_surge < 0] *= 0.8
    
    return final_alpha_factor
