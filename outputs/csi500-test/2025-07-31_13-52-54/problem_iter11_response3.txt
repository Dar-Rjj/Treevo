```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_returns = (df['Daily_Return'] * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_returns = (df['Daily_Return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['close']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).abs().rolling(window=20).sum()
    
    # Intraday Momentum
    df['Intraday_Momentum'] = (df['Intraday_Return'] * df['volume']).rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    df['Volume_Weighted_Intraday_Reversal'] = -(df['Intraday_Return'] * df['volume']).rolling(window=10).sum()
    
    # Price Momentum
    n_day_close_return = df['close'].pct_change(periods=20)
    n_day_open_return = df['open'].pct_change(periods=20)
    price_momentum = (n_day_close_return + n_day_open_return) / 2
    
    # Enhanced Volume-Weighted Volatility
    daily_volatility = (df['high'] - df['low']).abs()
    enhanced_volatility = daily_volatility.rolling(window=20).mean() * (df['volume'] / df['amount'])
    
    # Trade Frequency Adjusted Return
    trade_frequency = df['volume'].rolling(window=20).sum()
    trade_frequency_adjusted_return = (df['close'].pct_change(periods=20) * trade_frequency) / trade_frequency.rolling(window=20).max()
    
    # Combined Alpha Factor
    combined_alpha_factor = (
        (long_term_returns - short_term_returns) +
        (df['Intraday_Momentum'] + df['Volume_Weighted_Intraday_Reversal']) * 
        price_momentum / enhanced_volatility
    ) + trade_frequency_adjusted_return
    
    return combined_alpha_factor
```
