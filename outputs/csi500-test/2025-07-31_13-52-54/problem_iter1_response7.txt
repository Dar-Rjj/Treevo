```python
import pandas as pd

def heuristics_v2(data, N=20, M=60):
    # Calculate Price Momentum
    n_day_close_diff = data['close'] - data['close'].shift(N)
    m_day_close_diff = data['close'] - data['close'].shift(M)
    price_momentum = n_day_close_diff + m_day_close_diff
    
    # Calculate Volume Surge
    n_day_volume_change = data['volume'] - data['volume'].shift(N)
    m_day_volume_change = data['volume'] - data['volume'].shift(M)
    volume_surge = n_day_volume_change + m_day_volume_change
    
    # Combine Price Momentum and Volume Surge
    alpha_factor = price_momentum * volume_surge
    alpha_factor[volume_surge < 0] *= 0.8  # Apply penalty if Volume Surge is negative
    
    return alpha_factor
```
