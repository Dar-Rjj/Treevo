```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Enhanced Volume-Weighted Volatility
    volatility_range = (df['high'] - df['low']).rolling(window=n).std()
    volume_weighted_volatility = volatility_range * (df['volume'] / df['amount'])
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].rolling(window=n).mean()
    adjusted_return = (df['close'].pct_change().rolling(window=n).mean()) * trade_frequency
    
    # Combine Factors
    combined_factor = (close_momentum + open_momentum) * volume_weighted_volatility + adjusted_return
    
    return combined_factor
```
