```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['high'] - df['open']
    high_close_to_open_high_ratio = high_close_diff / open_high_diff
    
    # Weight by Volume
    weighted_spread = intraday_high_low_spread * df['volume']
    
    # Combine Spread and Ratio
    final_factor = weighted_spread + high_close_to_open_high_ratio
    
    return final_factor
```
