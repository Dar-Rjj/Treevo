import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Calculate Volume Momentum
    df['Volume_Momentum'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Calculate Price Range
    df['Price_Range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Average True Range
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['True_Range'].rolling(window=n).mean()
    for i in range(n, len(df)):
        df.loc[df.index[i], 'ATR'] = (df.loc[df.index[i-1], 'ATR'] * (n-1) + df.loc[df.index[i], 'True_Range']) / n
    
    # Calculate Price Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Price_Volatility'] = df['Daily_Return'].rolling(window=n).std()
    
    # Calculate Money Flow
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Money_Flow'] = (df['Typical_Price'] * df['volume']) - (df['Typical_Price'].shift(1) * df['volume'].shift(1))
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV'] + df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV'] - df.loc[df.index[i], 'volume']
        else:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV']
    
    # Identify Recent Highs and Lows
    df['Moving_Max_High'] = df['high'].rolling(window=n).max()
    df['Moving_Min_Low'] = df['low'].rolling(window=n).min()
    
    # Calculate Price Breakout
    df['Price_Breakout'] = (df['close'] - df['Moving_Min_Low']) / (df['Moving_Max_High'] - df['Moving_Min_Low'])
    
    # Calculate Price Reversal
    df['Price_Reversal'] = (df['close'] - df['Moving_Max_High']) / (df['Moving_Max_High'] - df['Moving_Min_Low'])
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Price_Momentum'] + 
        df['Volume_Momentum'] + 
        df['Price_Range'] + 
        df['ATR'] + 
        df['Price_Volatility'] + 
        df['Money_Flow'] + 
        df['OBV'] + 
        df['Price_Breakout'] + 
        df['Price_Reversal']
    )
    
    return df['Alpha_Factor'].dropna()
