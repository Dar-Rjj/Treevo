```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Average Price
    volume_weighted_price = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Compute Ratio
    high_low_spread_ratio = high_low_spread / volume_weighted_price
    
    return high_low_spread_ratio
```
