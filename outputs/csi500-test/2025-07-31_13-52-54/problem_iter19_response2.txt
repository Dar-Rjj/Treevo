```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages (SMA)
    df['SMA_10'] = df['close'].rolling(window=10, min_periods=1).mean()
    df['SMA_20'] = df['close'].rolling(window=20, min_periods=1).mean()
    
    # Determine if 10-day SMA is above 20-day SMA (Bullish Signal)
    df['SMA_Trend'] = (df['SMA_10'] > df['SMA_20']).astype(int)
    
    # Calculate rate of change (ROC) in close prices
    df['ROC_10'] = df['close'].pct_change(periods=10)
    df['ROC_20'] = df['close'].pct_change(periods=20)
    
    # Calculate volume momentum
    df['Volume_Momentum_10'] = df['volume'].pct_change(periods=10)
    df['Volume_Momentum_20'] = df['volume'].pct_change(periods=20)
    
    # Correlate volume changes with price changes
    df['Volume_Price_Correlation_10'] = df['close'].pct_change(periods=10).rolling(window=10, min_periods=1).corr(df['volume'].pct_change(periods=10))
    df['Volume_Price_Correlation_20'] = df['close'].pct_change(periods=20).rolling(window=20, min_periods=1).corr(df['volume'].pct_change(periods=20))
    
    # Identify bullish engulfing pattern
    df['Bullish_Engulfing'] = ((df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))).astype(int)
    
    # Identify bearish engulfing pattern
    df['Bearish_Engulfing'] = ((df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))).astype(int)
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['SMA_Trend'] + 
        df['ROC_10'] + 
        df['ROC_20'] + 
        df['Volume_Momentum_10'] + 
        df['Volume_Momentum_20'] + 
        df['Volume_Price_Correlation_10'] + 
        df['Volume_Price_Correlation_20'] + 
        df['Bullish_Engulfing'] - 
        df['Bearish_Engulfing']
    )
    
    return df['Alpha_Factor']
```
