import pandas as pd

def heuristics_v2(df, lookback_period=20, n_days=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low = (high_low_spread * df['volume']).rolling(window=lookback_period).sum()
    
    # Calculate Volume-Averaged Close Price
    volume_averaged_close = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Ratio
    volume_weighted_high_low_ratio = volume_weighted_high_low / volume_averaged_close
    
    # Calculate Momentum Component
    current_value = (high_low_spread * df['volume'])
    past_average = current_value.rolling(window=n_days).mean()
    momentum = current_value - past_average
    
    # Generate Preliminary Factor
    preliminary_factor = volume_weighted_high_low_ratio + momentum
    
    # Incorporate Price Trend
    n_day_moving_average = df['close'].rolling(window=n_days).mean()
    price_trend = (df['close'] - n_day_moving_average) * preliminary_factor
    
    # Calculate Volume-Average Price
    volume_averaged_price = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Averaged High-Low Spread Ratio
    volume_averaged_high_low_ratio = high_low_spread / volume_averaged_price
    
    # Integrate Volume-Averaged High-Low Spread Ratio
    integrated_factor = price_trend * volume_averaged_high_low_ratio
    
    # Calculate Price-Volatility Interaction
    close_price_std = df['close'].rolling(window=lookback_period).std()
    price_volatility_interaction = ((df['close'] - n_day_moving_average) / close_price_std) * integrated_factor
    
    # Generate Final Combined Factor
    final_combined_factor = price_volatility_interaction
    
    return final_combined_factor
