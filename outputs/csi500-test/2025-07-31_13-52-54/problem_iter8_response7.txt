```python
import pandas as pd

def heuristics_v2(df, N=20, M=50, lookback_days=100):
    # Calculate Price Momentum
    df['N_day_diff'] = df['close'].diff(N)
    df['M_day_diff'] = df['close'].diff(M)
    df['price_momentum'] = df['N_day_diff'] + df['M_day_diff']
    
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    df['volume_weighted_return_long'] = (df['daily_returns'].rolling(lookback_days).sum() * df['volume'].rolling(lookback_days).sum()) / df['volume'].rolling(lookback_days).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    df['volume_weighted_return_short'] = (df['daily_returns'].rolling(5).sum() * df['volume'].rolling(5).sum()) / df['volume'].rolling(5).sum()
    
    # Combine Momentum and Reversal Components
    df['combined_momentum_reversal'] = df['volume_weighted_return_long'] - df['volume_weighted_return_short']
    
    # Calculate Intraday Range Growth
    df['high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    df['intraday_range_growth'] = (df['high_low_range'] - df['prev_high_low_range']) / df['prev_high_low_range']
    
    # Compute Volume-Weighted High-Low Spread
    df['high_low_spread_vol'] = (df['high'] - df['low']) * df['volume']
    df['cum_high_low_spread_vol'] = df['high_low_spread_vol'].rolling(lookback_days).sum()
    df['total_volume'] = df['volume'].rolling(lookback_days).sum()
    df['avg_high_low_momentum'] = df['cum_high_low_spread_vol'] / df['total_volume']
    df['high_low_momentum'] = (df['high_low_spread_vol'] - df['avg_high_low_momentum']) / df['total_volume']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * df['combined_momentum_reversal'] * df['intraday_range_growth'] * df['high_low_momentum']
    
    # Apply penalty if volume surge is negative
    df['volume_surge'] = df['volume'] - df['volume'].shift(1)
    df['alpha_factor'] = df.apply(lambda x: x['alpha_factor'] * 0.8 if x['volume_surge'] < 0 else x['alpha_factor'], axis=1)
    
    return df['alpha_factor']
```
