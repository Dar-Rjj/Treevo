```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    N = 20
    M = 50
    long_term_window = 100
    short_term_window = 5

    # Calculate Price Momentum
    df['N_day_close_diff'] = df['close'].diff(N)
    df['M_day_close_diff'] = df['close'].diff(M)
    df['price_momentum'] = df['N_day_close_diff'] + df['M_day_close_diff']

    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Long-Term Volume-Weighted Average Return (Momentum Component)
    df['long_term_weighted_return'] = (df['daily_return'].rolling(window=long_term_window).sum() * 
                                       df['volume'].rolling(window=long_term_window).sum()) / \
                                      df['volume'].rolling(window=long_term_window).sum()

    # Short-Term Volume-Weighted Average Return (Reversal Component)
    df['short_term_weighted_return'] = (df['daily_return'].rolling(window=short_term_window).sum() * 
                                        df['volume'].rolling(window=short_term_window).sum()) / \
                                       df['volume'].rolling(window=short_term_window).sum()

    # Combine Momentum and Reversal Components
    df['combined_momentum_reversal'] = df['long_term_weighted_return'] - df['short_term_weighted_return']

    # Calculate Volume Surge
    df['N_day_volume_change'] = df['volume'].diff(N)
    df['M_day_volume_change'] = df['volume'].diff(M)
    df['volume_surge'] = df['N_day_volume_change'] + df['M_day_volume_change']

    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * df['combined_momentum_reversal']
    df['alpha_factor'] = df['alpha_factor'].where(df['volume_surge'] >= 0, df['alpha_factor'] * 0.8)

    return df['alpha_factor']
```
