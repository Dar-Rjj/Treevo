```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided DataFrame.
    The factor is calculated using a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the relative range
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the factor
    df['factor'] = df['close_pct_change'] * df['norm_volume'] * df['range']
    
    # Return the factor as a Series
    return df['factor']
```
