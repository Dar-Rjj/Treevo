```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided market data.
    The factor is a weighted combination of the following:
    - 10-day moving average of the close price
    - 5-day moving average of the volume
    - 10-day moving average of the daily return
    - 5-day moving average of the relative strength index (RSI)
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the 10-day moving average of the close price
    ma_close_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the volume
    ma_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    ma_daily_return_10 = daily_return.rolling(window=10).mean()
    
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 5-day moving average of the RSI
    ma_rsi_5 = rsi.rolling(window=5).mean()
    
    # Combine the factors into a single alpha factor
    factor = (
        0.4 * ma_close_10 +
        0.3 * ma_volume_5 +
        0.2 * ma_daily_return_10 +
        0.1 * ma_rsi_5
    )
    
    return factor
```
