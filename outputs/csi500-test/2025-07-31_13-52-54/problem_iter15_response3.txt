```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day Simple Moving Average (SMA) of closing prices
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Generate a binary signal; 1 if the current close is above the SMA, -1 if below, and 0 if equal
    df['trend_signal'] = 0
    df.loc[df['close'] > df['SMA_50'], 'trend_signal'] = 1
    df.loc[df['close'] < df['SMA_50'], 'trend_signal'] = -1
    
    # Calculate the difference between today’s high and the previous day’s high
    df['high_diff'] = df['high'] - df['high'].shift(1)
    
    # Calculate the difference between today’s low and the previous day’s low
    df['low_diff'] = df['low'] - df['low'].shift(1)
    
    # Sum the differences to get a combined momentum score
    df['momentum_score'] = df['high_diff'] + df['low_diff']
    
    # Create a simple threshold to classify stocks as having strong positive or negative momentum based on the score
    df['momentum_signal'] = 0
    df.loc[df['momentum_score'] > 0.5, 'momentum_signal'] = 1
    df.loc[df['momentum_score'] < -0.5, 'momentum_signal'] = -1
    
    # Calculate the daily volume percentage change from the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Assign a value indicating strong buying (positive change) or selling (negative change) pressure
    df['volume_signal'] = 0
    df.loc[df['volume_change'] > 0.1, 'volume_signal'] = 1
    df.loc[df['volume_change'] < -0.1, 'volume_signal'] = -1
    
    # Multiply the return (close to close) by the volume traded
    df['return'] = df['close'].pct_change()
    df['intensity'] = df['return'] * df['volume']
    
    # Use a moving average of this intensity factor over the past 20 days to smooth out short-term fluctuations
    df['intensity_ma_20'] = df['intensity'].rolling(window=20).mean()
    
    # Combine all the signals into a final alpha factor
    df['alpha_factor'] = df['trend_signal'] + df['momentum_signal'] + df['volume_signal'] + df['intensity_ma_20']
    
    return df['alpha_factor']
```
