```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Daily VWAP Change
    df['vwap_change'] = df['vwap'] - df['vwap'].shift(1)
    
    # Separate Positive and Negative Changes
    df['positive_changes'] = df['vwap_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_changes'] = df['vwap_change'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # 14-Day Averages for RSI
    df['avg_positive_14'] = df['positive_changes'].rolling(window=14).mean()
    df['avg_negative_14'] = df['negative_changes'].rolling(window=14).mean()
    
    # Calculate RSI
    df['relative_strength'] = df['avg_positive_14'] / df['avg_negative_14']
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # High-Close to Open-High Ratio
    df['high_close_diff'] = df['high'] - df['close']
    df['open_high_diff'] = df['open'] - df['high']
    df['high_close_open_high_ratio'] = df['high_close_diff'] / df['open_high_diff'].replace(0, 1e-6)
    
    # Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Return (Momentum Component)
    df['long_term_vw_return'] = (df['daily_returns'].rolling(window=90).sum() * df['volume'].rolling(window=90).sum()) / df['volume'].rolling(window=90).sum()
    
    # Short-Term Volume-Weighted Return (Reversal Component)
    df['short_term_vw_return'] = (df['daily_returns'].rolling(window=5).sum() * df['volume'].rolling(window=5).sum()) / df['volume'].rolling(window=5).sum()
    
    # Adjusted VW-Return
    df['adjusted_vw_return'] = df['long_term_vw_return'] - df['short_term_vw_return']
    
    # Momentum Component
    df['momentum_component'] = df['high_low_spread'] * df['volume'] - (df['high_low_spread'] * df['volume']).rolling(window=20).mean() / df['volume'].rolling(window=20).sum()
    
    # Price-Volume Imbalance Factor
    df['price_volume_imbalance'] = (df['close'] - df['open']) * df['volume']
    
    # Volume-Weighted Price-Volume Imbalance
    df['volume_weighted_price_volume_imbalance'] = (df['price_volume_imbalance'].rolling(window=20).sum()) / df['volume'].rolling(window=20).sum()
    
    # Combine Volume-Adjusted High-Low Spread, High-Close to Open-High Ratio, and Price-Volume Imbalance
    df['combined_factor'] = df['volume_adjusted_high_low_spread'] + df['high_close_open_high_ratio'] + df['volume_weighted_price_volume_imbalance']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['rsi'] + df['high_low_spread'] + df['adjusted_vw_return'] + df['momentum_component'] + df['combined_factor']
    
    return df['alpha_factor']
```
