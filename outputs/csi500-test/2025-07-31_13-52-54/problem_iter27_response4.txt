import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using open, high, low, close, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the rolling 10-day mean of the daily returns
    df['rolling_mean_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the rolling 10-day standard deviation of the daily returns
    df['rolling_std_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['rolling_mean_return'] * df['rsi'] / df['mfi']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
