import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Price
    volume_weighted_price = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Ratio
    high_low_spread_ratio = high_low_spread / volume_weighted_price
    
    # Develop trend following alpha factor using close prices
    sma_50 = df['close'].rolling(window=50).mean()
    trend_signal = (df['close'] > sma_50).astype(int) - (df['close'] < sma_50).astype(int)
    
    # Explore momentum in high and low prices
    high_momentum = df['high'] - df['high'].shift(1)
    low_momentum = df['low'] - df['low'].shift(1)
    combined_momentum = high_momentum + low_momentum
    momentum_score = combined_momentum.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Analyze volume patterns
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_intensity = (volume_change > 0.1).astype(int) - (volume_change < -0.1).astype(int)
    
    # Leverage price and volume for an intensity factor
    return_intensity = (df['close'] - df['close'].shift(1)) * df['volume']
    intensity_factor = return_intensity.rolling(window=20).mean()
    
    # Synthesize Factors
    composite_alpha = (
        high_low_spread_ratio + 
        trend_signal + 
        momentum_score + 
        intensity_factor
    )
    
    return composite_alpha
