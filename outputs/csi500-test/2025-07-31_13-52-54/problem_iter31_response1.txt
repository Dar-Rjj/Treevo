```python
def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price
    volume_weighted_avg_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Ratio
    factor = high_low_spread / volume_weighted_avg_price
    
    return factor
```
