import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate the weighted high-low spread
    df['High_Low_Spread'] = (df['high'] - df['low'])
    df['Volume_Weighted_Spread'] = df['High_Low_Spread'] * df['volume']
    
    # Compute cumulative volume-weighted high-low spread
    df['Cumulative_Volume_Weighted_Spread'] = df['Volume_Weighted_Spread'].rolling(window=lookback_period).sum()
    df['Total_Volume'] = df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate the average volume-weighted high-low spread over the lookback period
    df['Average_Volume_Weighted_Spread'] = df['Cumulative_Volume_Weighted_Spread'] / df['Total_Volume']
    
    # Calculate momentum indicator
    df['Current_Volume_Weighted_Spread'] = df['High_Low_Spread'] * df['volume']
    df['Past_Average_Volume_Weighted_Spread'] = df['Average_Volume_Weighted_Spread'].shift(1)
    df['Momentum'] = df['Current_Volume_Weighted_Spread'] - df['Past_Average_Volume_Weighted_Spread']
    
    # Output the Volume-Weighted High-Low Momentum Factor
    return df['Momentum']
