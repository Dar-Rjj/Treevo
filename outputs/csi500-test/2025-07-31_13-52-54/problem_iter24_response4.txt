```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-weighted high-low spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=20).sum()
    
    # Volume-weighted close price
    volume_weighted_close_price = (df['close'] * df['volume']).rolling(window=20).sum()
    
    # Compute the ratio
    factor_values = volume_weighted_high_low_spread / volume_weighted_close_price
    
    return factor_values
```
