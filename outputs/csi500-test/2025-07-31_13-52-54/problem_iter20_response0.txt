import pandas as pd

def heuristics_v2(df, lookback_period=50, sma_window=50, momentum_lookback=20):
    # High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Volume-Weighted High-Low Spread
    df['Volume_Weighted_High_Low_Spread'] = (df['high'] - df['low']) * df['volume']
    df['VW_HL_Spread_Sum'] = df['Volume_Weighted_High_Low_Spread'].rolling(window=lookback_period).sum()
    
    # 50-day Simple Moving Average (SMA) of closing prices
    df['SMA'] = df['close'].rolling(window=sma_window).mean()
    
    # Binary trend signal
    df['Trend_Signal'] = (df['close'] > df['SMA']).astype(int) - (df['close'] < df['SMA']).astype(int)
    
    # Momentum Score
    df['Momentum_Score'] = (df['high'] - df['high'].shift(1)) + (df['low'] - df['low'].shift(1))
    
    # Momentum Signal
    def momentum_signal(row):
        if row['Momentum_Score'] > 0.05:
            return 1
        elif row['Momentum_Score'] < -0.05:
            return -1
        else:
            return 0
    
    df['Momentum_Signal'] = df.apply(momentum_signal, axis=1)
    
    # Integrate Volume-Weighted High-Low Spread with the trend following and momentum signals
    df['Integrated_Alpha_Factor'] = 0.5 * df['Trend_Signal'] + 0.3 * df['Momentum_Signal'] + 0.2 * df['VW_HL_Spread_Sum']
    
    # Analyze volume patterns
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Assign a value indicating strong buying or selling pressure
    def volume_pressure(row):
        if row['Volume_Change'] > 0.1:
            return 1
        elif row['Volume_Change'] < -0.1:
            return -1
        else:
            return 0
    
    df['Volume_Pressure'] = df.apply(volume_pressure, axis=1)
    
    # Leverage price and volume for an intensity factor
    df['Intensity_Factor'] = (df['close'] - df['close'].shift(1)) * df['volume']
    df['Smoothed_Intensity_Factor'] = df['Intensity_Factor'].rolling(window=momentum_lookback).mean()
    
    # Combine the intensity factor with the integrated alpha factor
    df['Final_Enhanced_Alpha_Factor'] = df['Integrated_Alpha_Factor'] + df['Smoothed_Intensity_Factor']
    
    # Incorporate Price Trend
    df['n_Day_MA'] = df['close'].rolling(window=sma_window).mean()
    df['Price_Trend'] = df['close'] - df['n_Day_MA']
    df['Final_Composite_Factor'] = df['Final_Enhanced_Alpha_Factor'] * df['Price_Trend']
    
    # Volume-Weighted High-Low Spread as a modifier
    df['Composite_Alpha_Factor'] = df['Final_Composite_Factor'] + df['VW_HL_Spread_Sum']
    
    return df['Composite_Alpha_Factor']
