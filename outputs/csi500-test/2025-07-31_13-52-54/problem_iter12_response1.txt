```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    df['long_term_weighted_return'] = (df['daily_return'].rolling(window=100).sum() * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    df['short_term_weighted_return'] = (df['daily_return'].rolling(window=5).sum() * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Integrate High-Low Range into Volume-Weighted Returns
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['adjusted_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'] / df['volume'].rolling(window=5).sum()
    
    # Combine Momentum, Reversal, and High-Low Range Components
    df['intermediate_alpha_factor'] = df['long_term_weighted_return'] - df['short_term_weighted_return'] + df['adjusted_volume_weighted_high_low_range']
    
    # Calculate Price Momentum
    n_days = 20
    m_days = 60
    df['n_day_price_diff'] = df['close'] - df['close'].shift(n_days)
    df['m_day_price_diff'] = df['close'] - df['close'].shift(m_days)
    df['price_momentum'] = df['n_day_price_diff'] + df['m_day_price_diff']
    
    # Calculate Intraday Volatility
    df['today_high_low_range'] = df['high'] - df['low']
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    df['intraday_volatility'] = (df['today_high_low_range'] - df['prev_high_low_range']) / df['prev_high_low_range']
    
    # Calculate Volume Surge
    df['n_day_volume_change'] = df['volume'] - df['volume'].shift(n_days)
    df['m_day_volume_change'] = df['volume'] - df['volume'].shift(m_days)
    df['volume_surge'] = df['n_day_volume_change'] + df['m_day_volume_change']
    
    # Calculate Open-Close Momentum
    df['n_day_open_close_diff'] = df['open'] - df['close'].shift(n_days)
    df['m_day_open_close_diff'] = df['open'] - df['close'].shift(m_days)
    df['open_close_momentum'] = df['n_day_open_close_diff'] + df['m_day_open_close_diff']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['intermediate_alpha_factor'] * df['price_momentum'] * df['intraday_volatility'])
    df['final_alpha_factor'] = df['final_alpha_factor'] * (0.8 if df['volume_surge'] < 0 else 1) + df['open_close_momentum']
    
    return df['final_alpha_factor']
```
