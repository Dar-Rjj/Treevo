```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=lookback_period).sum()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close_price = (df['close'] * df['volume']).rolling(window=lookback_period).sum()
    
    # Compute Ratio
    ratio = volume_weighted_high_low_spread / volume_weighted_close_price
    
    # Calculate Momentum Component
    current_day_spread = (high_low_spread * df['volume'])
    past_average_spread = current_day_spread.rolling(window=lookback_period).mean()
    momentum_component = current_day_spread - past_average_spread
    
    # Generate Interim Factor
    interim_factor = (volume_weighted_high_low_spread + momentum_component) / volume_weighted_close_price
    
    # Incorporate Price Trend
    n_day_moving_average_close = df['close'].rolling(window=lookback_period).mean()
    price_trend = (df['close'] - n_day_moving_average_close) * interim_factor
    
    # Calculate Price Momentum
    close_log_return = np.log(df['close']).diff(lookback_period)
    
    # Calculate Enhanced Volume-Weighted Volatility
    log_volume = np.log(df['volume'])
    volatility = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=lookback_period).std()
    enhanced_volatility = volatility * log_volume
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].gt(0).rolling(window=lookback_period).sum()
    trade_frequency_adjusted_return = (df['close'].pct_change(lookback_period) * np.sqrt(trade_frequency))
    
    # Combine Factors
    combined_factor = (close_log_return * enhanced_volatility + trade_frequency_adjusted_return)**2
    
    # Final Alpha Factor
    final_alpha_factor = price_trend * combined_factor
    
    return final_alpha_factor
```
