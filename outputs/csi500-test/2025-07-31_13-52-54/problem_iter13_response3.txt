```python
import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Compute Volume-Weighted High-Low Spread
    high_low_vol = (df['high'] - df['low']) * df['volume']
    
    # Summarize Over Lookback Period
    cum_high_low_vol = high_low_vol.rolling(window=lookback_period).sum()
    total_vol = df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate Momentum
    current_day_hl_vol = (df['high'] - df['low']) * df['volume']
    avg_past_hl_vol = cum_high_low_vol / total_vol
    momentum = current_day_hl_vol - avg_past_hl_vol
    
    # Incorporate Price Trend
    price_change = df['close'] - df['open']
    factor = momentum + price_change
    
    return factor
```
