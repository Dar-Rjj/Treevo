def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_weighted_high_low_spread = high_low_spread * df['volume']
    
    # Sum of Volume-Weighted High-Low Spreads
    sum_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=20).sum()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close_price = (df['close'] * df['volume']).rolling(window=20).sum()
    
    # Total Volume
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Compute Ratio
    volume_weighted_ratio = sum_volume_weighted_high_low_spread / volume_weighted_close_price
    
    return volume_weighted_ratio
