```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Volume-Weighted Volatility
    daily_range = df['high'] - df['low']
    volatility = daily_range.rolling(window=n).std()
    volume_weighted_volatility = volatility * df['volume']
    
    # Calculate Intraday Range Growth
    today_range = df['high'] - df['low']
    prev_day_high = df['high'].shift(1)
    prev_day_low = df['low'].shift(1)
    prev_day_range = prev_day_high - prev_day_low
    intraday_range_growth = (today_range - prev_day_range) / prev_day_range
    
    # Volume Weighted Moving Average
    close_prices = df['close'].rolling(window=n).mean()
    volume_weights = df['volume'].rolling(window=n).mean()
    vwma = (df['close'] * df['volume']).rolling(window=n).sum() / volume_weights
    
    # Calculate Amount Adjusted Return
    amount_adjusted_return = df['amount'].pct_change(n)
    
    # Combine Factors
    combined_factors = (close_momentum + open_momentum) * volume_weighted_volatility + \
                       amount_adjusted_return + intraday_range_growth + vwma
    
    # Calculate Price Momentum (Alternative)
    n_day_diff = df['close'] - df['close'].shift(n)
    m_day_diff = df['close'] - df['close'].shift(m)
    price_momentum_alt = n_day_diff + m_day_diff
    
    # Calculate Volume Surge
    n_day_volume_change = df['volume'] - df['volume'].shift(n)
    m_day_volume_change = df['volume'] - df['volume'].shift(m)
    volume_surge = n_day_volume_change + m_day_volume_change
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * price_momentum_alt
    final_alpha_factor = final_alpha_factor.where(volume_surge >= 0, final_alpha_factor * 0.8)
    
    return final_alpha_factor
```
