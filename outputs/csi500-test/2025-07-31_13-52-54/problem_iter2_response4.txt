import pandas as pd

def heuristics_v2(df, lookback_period=20, short_window=5, long_window=10):
    # Calculate True Range
    df['High-Low'] = df['high'] - df['low']
    df['High-Close-Prev'] = abs(df['high'] - df['close'].shift(1))
    df['Low-Close-Prev'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['High-Low', 'High-Close-Prev', 'Low-Close-Prev']].max(axis=1)
    
    # Calculate Intraday Percentage Change
    df['Intraday_Percent_Change'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust for Volume and Intraday Price Change
    df['Intraday_Price_Change'] = df['close'] - df['open']
    df['Sum_Volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['Alpha_Factor_1'] = (df['Intraday_Price_Change'] / df['True_Range']) * df['Sum_Volume']
    
    # Compute Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']).abs().rolling(window=lookback_period).sum()
    
    # Intraday Momentum Factor
    df['Intraday_Return'] = df['close'] / df['open'] - 1
    df['Volume_Adjusted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    df['Intraday_Momentum'] = df['Volume_Adjusted_Intraday_Return'].rolling(window=short_window).sum()
    
    # Volume-Weighted Intraday Reversal
    df['Volume_Weighted_Intraday_Reversal'] = -df['Volume_Adjusted_Intraday_Return'].rolling(window=long_window).sum()
    
    # Generate the final alpha factor
    df['Alpha_Factor'] = (df['Alpha_Factor_1'] + df['Intraday_Momentum'] + df['Volume_Weighted_Intraday_Reversal']) / 3
    
    return df['Alpha_Factor']
