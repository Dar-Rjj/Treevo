```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price Momentum
    price_momentum = df['close'].pct_change(n)
    
    # Calculate Intraday High-Low Spread
    high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Calculate Volume-Adjusted Volatility
    volatility = (df['high'] - df['low']).rolling(window=n).std() * df['volume']
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].ne(0).astype(int).rolling(window=n).sum()
    trade_freq_adjusted_return = (df['close'].pct_change(n) / trade_frequency)
    
    # Combine Initial Factors
    initial_factors = (price_momentum + high_low_spread) * volatility + trade_freq_adjusted_return
    
    # Calculate 50-day Simple Moving Average (SMA) of closing prices
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Generate Binary Trend Signal
    trend_signal = (df['close'] > sma_50).astype(int)
    
    # Calculate Combined Momentum Score
    combined_momentum_score = (df['high'] - df['high'].shift(1)) - (df['low'] - df['low'].shift(1))
    
    # Classify Momentum
    momentum_class = (combined_momentum_score > 0).astype(int)
    
    # Calculate Daily Volume Percentage Change
    volume_pct_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Average Price
    volume_avg_price = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute Volume-Averaged High-Low Spread Ratio
    high_low_spread_ratio = high_low_spread / volume_avg_price
    
    # Calculate Volume-Weighted Return
    volume_weighted_return = (df['close'].pct_change() * df['volume'])
    
    # Combine All Factors into a Composite Alpha Factor
    composite_alpha_factor = (initial_factors + trend_signal + momentum_class + volume_weighted_return) * high_low_spread_ratio
    
    return composite_alpha_factor
```
