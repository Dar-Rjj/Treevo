```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Daily VWAP Change
    df['vwap_change'] = df['vwap'] - df['vwap'].shift(1)
    
    # Separate Positive and Negative Changes
    df['positive_changes'] = df['vwap_change'].apply(lambda x: x if x > 0 else 0)
    df['negative_changes'] = df['vwap_change'].apply(lambda x: -x if x < 0 else 0)
    
    # 14-Day Averages for RSI
    df['pos_avg_14'] = df['positive_changes'].rolling(window=14).mean()
    df['neg_avg_14'] = df['negative_changes'].rolling(window=14).mean()
    
    # Calculate RSI
    df['rs'] = df['pos_avg_14'] / df['neg_avg_14']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Adjusted High-Low Spread
    df['vol_adj_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # High-Close to Open-High Ratio
    df['high_close_diff'] = df['high'] - df['close']
    df['open_high_diff'] = df['high'] - df['open']
    df['high_close_to_open_high_ratio'] = df['high_close_diff'] / df['open_high_diff'].replace(0, 1)
    
    # Daily Returns
    df['daily_returns'] = df['close'] / df['close'].shift(1) - 1
    
    # Long-Term VW-Return (Momentum Component)
    df['long_term_vol_weighted_returns'] = (df['daily_returns'].rolling(window=90).sum() * df['volume'].rolling(window=90).sum()) / df['volume'].rolling(window=90).sum()
    
    # Short-Term VW-Return (Reversal Component)
    df['short_term_vol_weighted_returns'] = (df['daily_returns'].rolling(window=5).sum() * df['volume'].rolling(window=5).sum()) / df['volume'].rolling(window=5).sum()
    
    # Adjusted VW-Return
    df['adjusted_vw_return'] = df['long_term_vol_weighted_returns'] - df['short_term_vol_weighted_returns']
    
    # Momentum Component
    df['momentum_component'] = df['vol_adj_high_low_spread'] - df['vol_adj_high_low_spread'].rolling(window=90).mean()
    
    # Price-Volume Imbalance Factor
    df['price_volume_imbalance'] = (df['close'] - df['open']) * df['volume']
    df['price_volume_imbalance_sum'] = df['price_volume_imbalance'].rolling(window=90).sum()
    
    # Volume-Averaged High-Low Spread Ratio
    df['volume_averaged_price'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['high_low_spread_ratio'] = df['high_low_spread'] / df['volume_averaged_price']
    
    # Combine Volume-Adjusted High-Low Spread, High-Close to Open-High Ratio, and Price-Volume Imbalance
    df['combined_spread_ratio'] = df['vol_adj_high_low_spread'] + df['high_close_to_open_high_ratio'] + df['price_volume_imbalance_sum']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = (
        df['rsi'] 
        + df['intraday_high_low_spread'] 
        + df['adjusted_vw_return'] 
        + df['momentum_component'] 
        + df['combined_spread_ratio'] 
        + df['high_low_spread_ratio']
    )
    
    return df['alpha_factor']
```
