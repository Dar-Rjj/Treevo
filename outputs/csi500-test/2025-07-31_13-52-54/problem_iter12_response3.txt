import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['open'] - df['high']
    high_close_ratio = high_close_diff / (open_high_diff.replace(0, 1e-6))  # Avoid division by zero
    
    # Summarize Volume-Adjusted High-Low Spread Over Lookback Period
    cumulative_volume_adjusted_spread = volume_adjusted_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate Momentum Component
    current_day_value = volume_adjusted_spread
    past_n_days_average = (cumulative_volume_adjusted_spread / total_volume).shift(1)
    momentum_component = current_day_value - past_n_days_average
    
    # Introduce Price-Volume Imbalance Factor
    price_volume_imbalance = (df['close'] - df['open']) * df['volume']
    summarized_imbalance = price_volume_imbalance.rolling(window=lookback_period).sum()
    
    # Combine Adjusted Spread, Ratio, and Imbalance
    combined_factor = high_low_spread + high_close_ratio + price_volume_imbalance
    
    # Synthesize Final Factor
    final_factor = combined_factor + momentum_component
    
    return final_factor
