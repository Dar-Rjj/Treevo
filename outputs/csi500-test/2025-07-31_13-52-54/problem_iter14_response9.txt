import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: Alpha factor values indexed by date
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the relative range (High - Low) / Close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted price
    df['vwp'] = (df['amount'] / df['volume']).rolling(window=5, min_periods=1).mean()
    
    # Calculate the moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10, min_periods=1).mean()
    
    # Calculate the standard deviation of the close price
    df['std_10'] = df['close'].rolling(window=10, min_periods=1).std()
    
    # Calculate the z-score of the close price
    df['z_score'] = (df['close'] - df['ma_10']) / df['std_10']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_pct_change'] + df['relative_range'] + df['z_score']) * df['vwp']
    
    return df['alpha_factor']
