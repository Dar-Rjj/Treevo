```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).abs()
    intraday_volatility_20d = intraday_volatility.rolling(window=20).sum()
    
    # Intraday Momentum
    intraday_momentum = intraday_return * df['volume']
    intraday_momentum_5d = intraday_momentum.rolling(window=5).sum()
    
    # Volume-Weighted Intraday Reversal
    intraday_reversal = -(intraday_return * df['volume'])
    intraday_reversal_10d = intraday_reversal.rolling(window=10).sum()
    
    # Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Long-Term Volume-Weighted Average Return (Momentum Component)
    long_term_momentum = (daily_returns * df['volume']).rolling(window=100).sum() / df['volume'].rolling(window=100).sum()
    
    # Short-Term Volume-Weighted Average Return (Reversal Component)
    short_term_reversal = (daily_returns * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Volume Adjusted Volatility
    volatility = (df['high'] - df['low']).std()
    volume_adjusted_volatility = volatility / df['volume']
    
    # Combined Factor
    combined_factor = (intraday_momentum_5d + intraday_reversal_10d) * (long_term_momentum - short_term_reversal) / volume_adjusted_volatility
    
    return combined_factor
```
