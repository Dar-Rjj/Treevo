```python
import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate True Range
    high_low_diff = df['high'] - df['low']
    high_close_prev_diff = (df['high'] - df['close'].shift(1)).abs()
    low_close_prev_diff = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([high_low_diff, high_close_prev_diff, low_close_prev_diff], axis=1).max(axis=1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume and Intraday Price Change
    intraday_price_change = df['close'] - df['open']
    sum_of_volumes = df['volume'].rolling(window=lookback_period).sum()
    
    # Generate Alpha Factor
    alpha_factor = (intraday_price_change * true_range) * (price_momentum / high_low_spread) * sum_of_volumes
    
    return alpha_factor
```
