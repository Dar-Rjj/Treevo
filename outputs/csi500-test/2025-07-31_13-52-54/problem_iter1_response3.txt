```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate High-Close Difference (Previous Day)
    high_close_prev_diff = df['high'] - df['close'].shift(1)
    
    # Calculate Low-Close Difference (Previous Day)
    low_close_prev_diff = df['low'] - df['close'].shift(1)
    
    # Choose the Maximum of Differences
    true_range = pd.concat([high_low_diff, high_close_prev_diff, low_close_prev_diff], axis=1).max(axis=1)
    
    # Calculate Intraday Price Change
    intraday_price_change = df['close'] - df['open']
    
    # Sum of Volumes over a Lookback Period
    sum_of_volumes = df['volume'].rolling(window=lookback_period).sum()
    
    # Generate Alpha Factor
    alpha_factor = (intraday_price_change / true_range) * sum_of_volumes
    
    return alpha_factor
```
