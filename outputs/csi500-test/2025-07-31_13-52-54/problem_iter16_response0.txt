```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate short-term (5-day) and long-term (20-day) moving averages of log returns
    df['ma_5_log_return'] = df['log_return'].rolling(window=5).mean()
    df['ma_20_log_return'] = df['log_return'].rolling(window=20).mean()
    
    # Momentum factor: Difference between short-term and long-term moving averages
    df['momentum_factor'] = df['ma_5_log_return'] - df['ma_20_log_return']
    
    # Historical volatility: Standard deviation of log returns (20-day window)
    df['historical_volatility'] = df['log_return'].rolling(window=20).std()
    
    # Exponential weighted moving average of log returns (12-period)
    df['ewma_12_log_return'] = df['log_return'].ewm(span=12, adjust=False).mean()
    
    # Percentage change in volume
    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Moving average of volume (5-day and 20-day)
    df['ma_5_volume'] = df['volume'].rolling(window=5).mean()
    df['ma_20_volume'] = df['volume'].rolling(window=20).mean()
    
    # Volume momentum: Difference between short-term and long-term moving averages
    df['volume_momentum'] = df['ma_5_volume'] - df['ma_20_volume']
    
    # Dollar volume
    df['dollar_volume'] = df['close'] * df['volume']
    
    # Percentage change in dollar volume
    df['dollar_volume_change'] = (df['dollar_volume'] / df['dollar_volume'].shift(1)) - 1
    
    # Combined momentum: Weighted sum of price momentum, volume momentum, and dollar volume momentum
    combined_momentum = 0.5 * df['momentum_factor'] + 0.3 * df['volume_momentum'] + 0.2 * df['dollar_volume_change']
    
    # Trend strength: Directional Movement Index (DMI)
    df['high_low_diff'] = df['high'] - df['low']
    df['high_prev_close_diff'] = df['high'] - df['close'].shift(1)
    df['low_prev_close_diff'] = df['low'] - df['close'].shift(1)
    
    df['positive_dm'] = df[['high_low_diff', 'high_prev_close_diff']].max(axis=1)
    df['negative_dm'] = -df[['high_low_diff', 'low_prev_close_diff']].min(axis=1)
    
    df['tr'] = df[['high_low_diff', 'high_prev_close_diff', 'low_prev_close_diff']].max(axis=1)
    
    df['smoothed_positive_dm'] = df['positive_dm'].rolling(window=14).mean()
    df['smoothed_negative_dm'] = df['negative_dm'].rolling(window=14).mean()
    df['smoothed_tr'] = df['tr'].rolling(window=14).mean()
    
    df['positive_di'] = 100 * (df['smoothed_positive_dm'] / df['smoothed_tr'])
    df['negative_di'] = 100 * (df['smoothed_negative_dm'] / df['smoothed_tr'])
    
    df['dx'] = 100 * (abs(df['positive_di'] - df['negative_di']) / (df['positive_di'] + df['negative_di']))
    
    df['adx'] = df['dx'].rolling(window=14).mean()
    
    # Relative strength: Ratio of stock's close to benchmark's close (assuming a benchmark close column is provided)
    if 'benchmark_close' in df.columns:
        df['relative_strength'] = df['close'] / df['benchmark_close']
    
    # Return the combined momentum factor
    return combined_momentum
```
