```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(periods=n)
    open_momentum = df['open'].pct_change(periods=n)
    
    # Calculate Enhanced Volume-Weighted Volatility
    price_range = df['high'] - df['low']
    volatility = price_range.rolling(window=n).mean()
    volume_weighted_volatility = volatility * (df['volume'] * df['amount'])
    
    # Calculate Trade Frequency Adjusted Return
    daily_trades = df['volume'].ne(0).astype(int)
    trade_frequency = daily_trades.rolling(window=n).sum()
    trade_frequency_adjusted_return = (df['close'].pct_change() * trade_frequency).rolling(window=n).mean()
    
    # Combine Factors
    combined_factor = (close_momentum + open_momentum) * volume_weighted_volatility + trade_frequency_adjusted_return
    
    return combined_factor
```
