```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Calculate Volume-Weighted High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_weighted_high_low_spread = high_low_spread * df['volume']

    # Compute Cumulative Volume-Weighted High-Low Spread
    cumulative_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()

    # Calculate Momentum Indicator
    current_day_high_low_spread = volume_weighted_high_low_spread
    avg_past_high_low_spread = cumulative_volume_weighted_high_low_spread / total_volume
    momentum_indicator = current_day_high_low_spread - avg_past_high_low_spread

    # Calculate Price Momentum
    close_n_day_return = df['close'].pct_change(periods=lookback_period)
    open_n_day_return = df['open'].pct_change(periods=lookback_period)

    # Calculate Volume-Weighted Volatility
    price_range = df['high'] - df['low']
    volatility = price_range.rolling(window=lookback_period).std()
    volume_weighted_volatility = volatility * df['volume']

    # Calculate Amount Adjusted Return
    amount_adjusted_return = (df['amount'].pct_change(periods=lookback_period) * df['amount']).rolling(window=lookback_period).mean()

    # Combine Factors
    price_momentum_volatility = close_n_day_return * volume_weighted_volatility
    alpha_factor = (price_momentum_volatility + amount_adjusted_return + momentum_indicator).dropna()

    return alpha_factor
```
