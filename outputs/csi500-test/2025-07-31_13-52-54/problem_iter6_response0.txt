```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Weighted High-Low Spread
    high_low_spread = df['high'] - df['low']
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    weighted_high_low_spread = high_low_spread * (df['volume'] / avg_volume_5_days)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate the final factor
    factor = price_momentum / weighted_high_low_spread
    
    return factor
```
