```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5):
    # Calculate Price Momentum
    close_momentum = df['close'].pct_change(n)
    open_momentum = df['open'].pct_change(n)
    
    # Calculate Volume-Weighted Volatility
    price_range = df['high'] - df['low']
    volatility = price_range.rolling(window=n).std()
    volume_weighted_volatility = volatility * df['volume']
    
    # Calculate Amount Adjusted Return
    amount_adjusted_return = (df['close'] / df['amount']).pct_change(n)
    
    # Combine Factors
    combined_factor = (close_momentum + open_momentum) * volume_weighted_volatility + amount_adjusted_return
    
    return combined_factor
```
