```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate (High - Low) * Volume
    df['high_low_volume'] = (df['high'] - df['low']) * df['volume']
    
    # Cumulative (High - Low) * Volume over the lookback period
    cumulative_high_low_volume = df['high_low_volume'].rolling(window=lookback_period).sum()
    
    # Total Volume over the lookback period
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    
    # Average of Past n Days' (High - Low) * Volume / Total Volume
    average_high_low_spread = cumulative_high_low_volume / total_volume
    
    # Current Day's (High - Low) * Volume
    current_high_low_volume = df['high_low_volume']
    
    # Subtract average from current day's value to get the momentum
    momentum = current_high_low_volume - average_high_low_spread
    
    return momentum
```
