```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20, n_days=10):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    df['volume_weighted_high_low_spread'] = df['high_low_spread'] * df['volume']
    df['sum_volume_weighted_high_low_spread'] = df['volume_weighted_high_low_spread'].rolling(window=lookback).sum()
    
    # Calculate Momentum Component
    current_value = df['volume_weighted_high_low_spread']
    past_average = df['volume_weighted_high_low_spread'].rolling(window=n_days).mean()
    df['momentum_component'] = current_value - past_average
    
    # Generate Final Factor (Volume-Weighted High-Low Spread and Momentum)
    df['final_factor'] = (df['sum_volume_weighted_high_low_spread'] + df['momentum_component']) / df['close']
    
    # Incorporate Price Trend
    df['n_day_moving_avg_close'] = df['close'].rolling(window=n_days).mean()
    df['price_trend'] = df['close'] - df['n_day_moving_avg_close']
    df['final_factor'] *= df['price_trend']
    
    # Calculate Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(n_days))
    
    # Calculate Enhanced Volume-Weighted Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=n_days).mean()
    df['enhanced_volatility'] = df['average_true_range'] * np.log(df['volume'])
    
    # Calculate Trade Frequency Adjusted Return
    df['trade_frequency'] = df['volume']
    df['trade_frequency_adjusted_return'] = df['log_return'] / np.sqrt(df['trade_frequency'])
    
    # Combine Factors
    df['combined_factor'] = df['log_return'] * df['enhanced_volatility'] + df['trade_frequency_adjusted_return']
    df['nonlinear_transformed_factor'] = np.sqrt(np.abs(df['combined_factor']))
    
    # Final Alpha Factor
    df['alpha_factor'] = df['final_factor'] * df['nonlinear_transformed_factor']
    
    return df['alpha_factor']
```
