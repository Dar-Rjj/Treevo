```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-Averaged Price
    volume_averaged_price = (df['close'] * df['volume']).sum() / df['volume'].sum()
    
    # Compute Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume'])
    
    # Compute Cumulative Volume-Weighted High-Low Spread
    cumulative_volume_weighted_high_low_spread = volume_weighted_high_low_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    normalized_cumulative_spread = cumulative_volume_weighted_high_low_spread / total_volume
    
    # Calculate Momentum Component
    current_day_spread = volume_weighted_high_low_spread
    past_average_spread = volume_weighted_high_low_spread.rolling(window=lookback_period).mean().shift(1)
    momentum = current_day_spread - past_average_spread
    
    # Generate Final Factor
    final_factor = (normalized_cumulative_spread + momentum) / volume_averaged_price
    
    return final_factor
```
