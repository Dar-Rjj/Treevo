```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Daily VWAP Change
    df['VWAP_Change'] = df['VWAP'].diff()
    
    # Separate Positive and Negative Changes
    df['Positive_Change'] = df['VWAP_Change'].apply(lambda x: x if x > 0 else 0)
    df['Negative_Change'] = df['VWAP_Change'].apply(lambda x: -x if x < 0 else 0)
    
    # 14-Day Averages for RSI
    positive_avg = df['Positive_Change'].rolling(window=14).mean()
    negative_avg = df['Negative_Change'].rolling(window=14).mean()
    
    # Calculate RSI
    rs = positive_avg / negative_avg
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Intraday Range Growth
    df['Today_High_Low_Range'] = df['high'] - df['low']
    df['Prev_Day_High_Low_Range'] = df['high'].shift(1) - df['low'].shift(1)
    df['Intraday_Range_Growth'] = (df['Today_High_Low_Range'] - df['Prev_Day_High_Low_Range']) / df['Prev_Day_High_Low_Range']
    
    # Volume-Weighted Moving Average (VWMA)
    df['VWMA'] = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Daily Returns
    df['Daily_Return'] = df['close'].pct_change()
    
    # Long-Term VW-Return (Momentum Component)
    long_term_returns = (df['Daily_Return'] * df['volume']).rolling(window=90).sum() / df['volume'].rolling(window=90).sum()
    df['Long_Term_VW_Return'] = long_term_returns
    
    # Short-Term VW-Return (Reversal Component)
    short_term_returns = (df['Daily_Return'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['Short_Term_VW_Return'] = short_term_returns
    
    # Adjusted VW-Return
    df['Adjusted_VW_Return'] = df['Long_Term_VW_Return'] - df['Short_Term_VW_Return']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['RSI'] + df['Intraday_Range_Growth'] + df['VWMA'] + df['Adjusted_VW_Return']) / 4
    
    return df['Final_Alpha_Factor']
```
