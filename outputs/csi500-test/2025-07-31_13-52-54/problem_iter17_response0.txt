```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, sma_period=50):
    # Compute Volume-Weighted High-Low Spread
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    
    # Summarize Over Lookback Period
    df['cum_high_low_vol'] = df['high_low_vol'].rolling(window=n).sum()
    df['total_volume'] = df['volume'].rolling(window=n).sum()
    
    # Calculate Momentum
    df['avg_high_low_vol'] = df['cum_high_low_vol'] / df['total_volume']
    df['momentum'] = (df['high_low_vol'] - df['avg_high_low_vol'])
    
    # Incorporate Price Trend
    df['price_change'] = df['close'] - df['open']
    df['momentum'] += df['price_change']
    
    # Calculate Intraday Range Growth
    df['today_range'] = df['high'] - df['low']
    df['prev_day_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['intraday_range_growth'] = (df['today_range'] - df['prev_day_range']) / df['prev_day_range']
    
    # Volume Weighted Moving Average
    df['weighted_close'] = df['close'] * df['volume']
    df['vol_weighted_ma'] = df['weighted_close'].rolling(window=n).sum() / df['total_volume']
    
    # Combine Intraday Range Growth and Volume Weighted Moving Average
    df['range_growth_ma'] = df['intraday_range_growth'] + df['vol_weighted_ma']
    
    # Calculate Price Momentum
    df['n_day_return_close'] = df['close'].pct_change(n)
    df['n_day_return_open'] = df['open'].pct_change(n)
    df['price_momentum'] = (df['n_day_return_close'] + df['n_day_return_open']) / 2
    
    # Adjust for Volume Surge
    df['n_day_vol_change'] = df['volume'] - df['volume'].shift(n)
    df['m_day_vol_change'] = df['volume'] - df['volume'].shift(m)
    df['volume_surge'] = df['n_day_vol_change'] + df['m_day_vol_change']
    df['adjusted_volume_surge'] = df['volume_surge'].apply(lambda x: x * 0.8 if x < 0 else x)
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * df['momentum'] + df['intraday_range_growth'] + df['adjusted_volume_surge'] + df['momentum']
    
    # Calculate the Simple Moving Average (SMA) of closing prices
    df['sma'] = df['close'].rolling(window=sma_period).mean()
    
    # Determine the current stock's position relative to its SMA
    df['sma_signal'] = (df['close'] > df['sma']).astype(int)
    
    return df['alpha_factor']
```
