import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Moving Averages of Log Returns
    df['short_term_ma'] = df['log_returns'].rolling(window=5).mean()
    df['long_term_ma'] = df['log_returns'].rolling(window=20).mean()
    df['momentum'] = df['short_term_ma'] - df['long_term_ma']
    
    # Volatility Measures
    df['historical_volatility'] = df['log_returns'].rolling(window=20).std()
    df['ewma_volatility'] = df['log_returns'].ewm(span=12, adjust=False).mean()
    
    # Price Patterns
    df['bullish_engulfing'] = (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))
    df['bearish_engulfing'] = (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))
    df['inside_bar'] = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1))
    
    # Volume Data
    df['volume_change'] = (df['volume'] / df['volume'].shift(1) - 1)
    df['volume_short_term_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_long_term_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_momentum'] = df['volume_short_term_ma'] - df['volume_long_term_ma']
    df['volume_spike'] = (df['volume'] > 2 * df['volume'].shift(1))
    df['volume_divergence'] = (df['volume'] > df['volume'].shift(1)) & (df['close'] < df['close'].shift(1))
    
    # Amount Data
    df['dollar_volume'] = df['close'] * df['volume']
    df['dollar_volume_change'] = (df['dollar_volume'] / df['dollar_volume'].shift(1) - 1)
    df['dollar_volume_spike'] = (df['dollar_volume'] > 2 * df['dollar_volume'].shift(1))
    df['dollar_volume_divergence'] = (df['dollar_volume'] > df['dollar_volume'].shift(1)) & (df['close'] < df['close'].shift(1))
    
    # Composite Factors
    df['combined_momentum'] = 0.5 * df['momentum'] + 0.3 * df['volume_momentum'] + 0.2 * df['dollar_volume_change']
    
    # Trend Strength (DMI and ADX)
    df['dm_plus'] = np.where((df['high'] - df['high'].shift(1)) > (df['low'].shift(1) - df['low']), 
                             df['high'] - df['high'].shift(1), 0)
    df['dm_minus'] = np.where((df['low'].shift(1) - df['low']) > (df['high'] - df['high'].shift(1)), 
                              df['low'].shift(1) - df['low'], 0)
    df['tr'] = df[['high' - 'low', 'high' - 'close'].shift(1), 'close'].shift(1) - 'low']].max(axis=1)
    df['smoothed_dm_plus'] = df['dm_plus'].ewm(span=14, adjust=False).mean()
    df['smoothed_dm_minus'] = df['dm_minus'].ewm(span=14, adjust=False).mean()
    df['smoothed_tr'] = df['tr'].ewm(span=14, adjust=False).mean()
    df['di_plus'] = 100 * (df['smoothed_dm_plus'] / df['smoothed_tr'])
    df['di_minus'] = 100 * (df['smoothed_dm_minus'] / df['smoothed_tr'])
    df['dx'] = 100 * (np.abs(df['di_plus'] - df['di_minus']) / (df['di_plus'] + df['di_minus']))
    df['adx'] = df['dx'].ewm(span=14, adjust=False).mean()
