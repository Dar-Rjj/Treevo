```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Volume-Adjusted High-Low Spread
    volume_adjusted_spread = high_low_spread * df['volume']
    
    # Calculate High-Close to Open-High Ratio
    high_close_diff = df['high'] - df['close']
    open_high_diff = df['high'] - df['open']
    high_close_ratio = high_close_diff / open_high_diff.replace(0, 1)  # Avoid division by zero
    
    # Summarize Volume-Adjusted High-Low Spread Over Lookback Period
    cumulative_volume_adjusted_spread = volume_adjusted_spread.rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    avg_volume_adjusted_spread = cumulative_volume_adjusted_spread / total_volume.replace(0, 1)  # Avoid division by zero
    
    # Calculate Momentum Component
    momentum_component = volume_adjusted_spread - avg_volume_adjusted_spread
    
    # Introduce Price-Volume Imbalance Factor
    price_volume_imbalance = (df['close'] - df['open']) * df['volume']
    cumulative_price_volume_imbalance = price_volume_imbalance.rolling(window=lookback_period).sum()
    
    # Combine Adjusted Spread, Ratio, and Imbalance
    combined_factor = volume_adjusted_spread + high_close_ratio + cumulative_price_volume_imbalance
    
    # Synthesize Final Factor
    synthesized_final_factor = combined_factor + momentum_component
    
    # Calculate Enhanced Volume-Adjusted Volatility
    volatility = (df['high'] - df['low']).rolling(window=lookback_period).std()
    enhanced_volatility = volatility * df['volume']
    
    # Calculate Trade Frequency Adjusted Return
    trade_frequency = df['volume'].ne(0).astype(int).rolling(window=lookback_period).mean()
    n_day_return = (df['close'] / df['close'].shift(lookback_period)) - 1
    trade_frequency_adjusted_return = n_day_return * trade_frequency
    
    # Final Alpha Factor
    final_alpha_factor = (synthesized_final_factor * enhanced_volatility) + trade_frequency_adjusted_return
    
    return final_alpha_factor
```
