```python
import pandas as pd

def heuristics_v2(df, lookback=30):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Spread
    df['volume_weighted_high_low_spread'] = df['high_low_spread'] * df['volume']
    volume_weighted_high_low_spread_sum = df['volume_weighted_high_low_spread'].rolling(window=lookback).sum()
    total_volume = df['volume'].rolling(window=lookback).sum()
    volume_weighted_high_low_spread_ratio = volume_weighted_high_low_spread_sum / total_volume
    
    # Calculate Volume-Averaged Close Price
    df['volume_averaged_close'] = (df['close'] * df['volume']).rolling(window=lookback).sum() / total_volume
    
    # Compute Ratio
    volume_weighted_high_low_spread_ratio = volume_weighted_high_low_spread_sum / total_volume
    
    # Calculate Momentum Component
    current_volume_weighted_high_low_spread = df['volume_weighted_high_low_spread']
    past_avg_volume_weighted_high_low_spread = df['volume_weighted_high_low_spread'].rolling(window=lookback).mean()
    momentum_component = current_volume_weighted_high_low_spread - past_avg_volume_weighted_high_low_spread
    
    # Generate Preliminary Factor
    preliminary_factor = volume_weighted_high_low_spread_ratio + momentum_component
    
    # Incorporate Price Trend
    n_day_moving_average_close = df['close'].rolling(window=lookback).mean()
    price_trend = df['close'] - n_day_moving_average_close
    preliminary_factor_with_trend = preliminary_factor * price_trend
    
    # Calculate Volume-Average Price
    volume_averaged_price = (df['close'] * df['volume']).rolling(window=lookback).sum() / total_volume
    
    # Compute Volume-Averaged High-Low Spread Ratio
    volume_averaged_high_low_spread_ratio = df['high_low_spread'] / volume_averaged_price
    
    # Integrate Volume-Averaged High-Low Spread Ratio
    integrated_factor = preliminary_factor_with_trend * volume_averaged_high_low_spread_ratio
    
    # Calculate Price-Volatility Interaction
    close_price_std = df['close'].rolling(window=lookback).std()
    price_volatility_interaction = (df['close'] - n_day_moving_average_close) / close_price_std
    final_combined_factor = integrated_factor * price_volatility_interaction
    
    return final_combined_factor
```
