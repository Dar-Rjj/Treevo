```python
import pandas as pd

def heuristics_v2(df, N=10, M=20, volatility_threshold=1.5):
    # Calculate Price Momentum
    n_day_close_diff = df['close'] - df['close'].shift(N)
    m_day_close_diff = df['close'] - df['close'].shift(M)
    price_momentum = n_day_close_diff + m_day_close_diff
    
    # Calculate Volume Surge
    n_day_volume_change = df['volume'] - df['volume'].shift(N)
    m_day_volume_change = df['volume'] - df['volume'].shift(M)
    volume_surge = n_day_volume_change + m_day_volume_change
    
    # Apply penalty if Volume Surge is negative
    volume_surge_adjusted = volume_surge.apply(lambda x: x * 0.8 if x < 0 else x)
    
    # Calculate Volatility
    n_day_high_low_range = df['high'] - df['low']
    m_day_high_low_range = df['high'] - df['low']
    volatility = (n_day_high_low_range + m_day_high_low_range) / 2
    
    # Incorporate Volatility
    combined_value = price_momentum * volume_surge_adjusted
    adjusted_value = combined_value + volatility
    
    # Apply boost if Volatility is above a threshold
    final_alpha_factor = adjusted_value.apply(lambda x: x * 1.2 if x > volatility_threshold else x)
    
    return final_alpha_factor
```
