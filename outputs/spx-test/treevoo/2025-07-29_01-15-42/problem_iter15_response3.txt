```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    df['close_to_open_change'] = df['close'] - df['open']
    
    # Analyze Volume Trends
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Generate Volume-Adjusted Indicators
    df['volume_adjusted_price_range'] = df['daily_price_range'] / df['avg_volume_30']
    df['volume_adjusted_high_low_spread'] = (df['high'] - df['low']) * df['volume']
    df['volume_adjusted_close_open_change'] = df['close_to_open_change'] * df['volume']
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = ((df['high'] - df['low']) / df['open']).rolling(window=10).mean()
    
    # Define the threshold and median for realized volatility
    threshold = 0.05
    realized_vol_median = df['realized_volatility'].median()
    
    # Create the final alpha signal based on combined indicators
    alpha_signal = pd.Series(index=df.index)
    
    # Step through the DataFrame to generate the alpha factor
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'volume_adjusted_price_range'] > threshold and df.loc[df.index[i], 'realized_volatility'] < realized_vol_median:
            alpha_signal.loc[df.index[i]] = 1
        elif df.loc[df.index[i], 'volume_adjusted_price_range'] < -threshold and df.loc[df.index[i], 'realized_volatility'] < realized_vol_median:
            alpha_signal.loc[df.index[i]] = -1
        else:
            # Generate Preliminary Alpha Signal
            high_low_spread = df.loc[df.index[i], 'volume_adjusted_high_low_spread']
            close_open_change = df.loc[df.index[i], 'volume_adjusted_close_open_change']
            if high_low_spread > 0 and close_open_change > 0:
                prelim_alpha = 1
            elif high_low_spread < 0 and close_open_change < 0:
                prelim_alpha = -1
            else:
                prelim_alpha = 0
            
            # Calculate Momentum Score
            momentum_score = df.loc[df.index[i], 'volume_adjusted_price_range'] - df.loc[df.index[i-1], 'volume_adjusted_price_range']
            
            # Adjust the preliminary alpha based on the imbalance
            if prelim_alpha == 1:
                alpha_signal.loc[df.index[i]] = prelim_alpha + momentum_score
            elif prelim_alpha == -1:
                alpha_signal.loc[df.index[i]] = prelim_alpha - momentum_score
            else:
                alpha_signal.loc[df.index[i]] = momentum_score
    
    return alpha_signal
```
