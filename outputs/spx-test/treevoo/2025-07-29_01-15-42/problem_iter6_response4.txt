import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust Range by Volume
    adjusted_range = high_to_low_range / df['volume']
    
    # Incorporate Momentum Factor
    price_momentum = df['close'].pct_change(periods=12)
    final_factor = adjusted_range * price_momentum
    
    return final_factor
