```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of close prices over a 10-day window
    df['close_10_day_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of close prices over a 10-day window
    df['close_10_day_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the z-score of the close price
    df['close_z_score'] = (df['close'] - df['close_10_day_mean']) / df['close_10_day_std']
    
    # Calculate the difference between high and low, which is a measure of volatility
    df['high_low_diff'] = df['high'] - df['low']
    
    # Combine the z-score with the high-low difference to create a custom heuristic
    df['alpha_factor'] = df['close_z_score'] * df['high_low_diff']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
