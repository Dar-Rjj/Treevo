```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movements
    daily_price_range = df['high'] - df['low']
    
    # Analyze Volume Trends
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    volume_adjusted_price_range = daily_price_range / moving_avg_volume
    threshold = volume_adjusted_price_range.abs().mean()  # Use the mean of absolute values as a dynamic threshold
    
    # Generate Initial Alpha Signal
    alpha_signal = pd.Series(0, index=df.index)
    alpha_signal[volume_adjusted_price_range > threshold] = 1
    alpha_signal[volume_adjusted_price_range < -threshold] = -1
    
    # Incorporate Intraday Reversal with Volume Momentum
    range_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - range_midpoint
    volume_change = df['volume'].diff().fillna(0)  # Handle first day with no previous volume
    
    # Adjust Alpha Signal
    adjusted_alpha_signal = alpha_signal + (close_to_midpoint_deviation * volume_change)
    
    return adjusted_alpha_signal
```
