```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_prev'] = df['close'].shift(1)
    df['gains'] = df['close'] - df['close_prev']
    df['gains'] = df['gains'].apply(lambda x: max(x, 0))
    df['losses'] = df['close'] - df['close_prev']
    df['losses'] = df['losses'].apply(lambda x: -min(x, 0))
    
    df['smoothed_gains'] = df['gains'].rolling(window=14).mean()
    df['smoothed_losses'] = df['losses'].rolling(window=14).mean()
    
    df['rtsi'] = (df['smoothed_gains'] / (df['smoothed_gains'] + df['smoothed_losses'])) * 100
    
    # Calculate Daily Price Change and Smooth with Simple Moving Average (SMA)
    df['close_to_close_return'] = df['close'].diff()
    df['sma_return'] = df['close_to_close_return'].rolling(window=5).mean()
    
    # Adjust for Volume
    df['volume_sum'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = df['sma_return'] / df['volume_sum']
    
    # Combine Factors
    df['alpha_factor'] = df['price_range_ratio'] * df['rtsi'] * df['volume_adjusted_momentum']
    
    return df['alpha_factor']
```
