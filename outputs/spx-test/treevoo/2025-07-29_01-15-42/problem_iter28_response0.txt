```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility (Rolling 20-day standard deviation of log returns)
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Close-to-Open Movement
    close_to_open_movement = (df['close'] - df['open']) / df['open']
    
    # Smooth with Exponential Moving Average (EMA) on close-to-close returns
    ema_smoothed_returns = df['close'].pct_change().ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=5).sum()
    total_amount = df['amount'].rolling(window=5).sum()
    combined_total = total_volume + total_amount
    volume_adjusted_momentum = combined_momentum / combined_total
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'] - df['close'].shift(1)).clip(lower=0)
    losses = (df['close'].shift(1) - df['close']).clip(lower=0)
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * intraday_high_low_spread +
        0.1 * close_to_open_movement +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Volume Acceleration
    volume_acceleration = df['volume'].pct_change(10)
    
    # Final Factor
    final_factor = weighted_average * (1 + volume_acceleration)
    
    return final_factor
```
