```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday High-Low Spread
    intraday_spread = data['high'] - data['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = data['close'] - data['open']
    
    # Incorporate Volume Impact
    volume_impact = (intraday_spread + close_to_open_movement) * (data['volume'] / data['volume'].rolling(window=30).mean())
    
    # Calculate Momentum Score
    momentum_score = volume_impact - volume_impact.shift(1)
    
    return momentum_score
```
