```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the given market data.
    The alpha factor is based on the following heuristic:
    - If the closing price is higher than the opening price and the volume is above the 75th percentile, 
      then the stock is considered to be in a positive momentum and the factor value is set to the difference between high and low prices.
    - Otherwise, the stock is considered to have a negative or neutral momentum and the factor value is set to the difference between open and close prices.
    
    :param df: A pandas DataFrame with columns 'open', 'high', 'low', 'close', and 'volume'
    :return: A pandas Series representing the alpha factor values for each date
    """
    # Calculate the 75th percentile of the volume
    volume_75th_percentile = df['volume'].quantile(0.75)
    
    # Initialize an empty series to store the alpha factor values
    factor_values = pd.Series(index=df.index, dtype=float)
    
    # Calculate the alpha factor based on the defined heuristic
    for i in range(len(df)):
        if (df['close'].iloc[i] > df['open'].iloc[i]) and (df['volume'].iloc[i] > volume_75th_percentile):
            factor_values.iloc[i] = df['high'].iloc[i] - df['low'].iloc[i]
        else:
            factor_values.iloc[i] = df['open'].iloc[i] - df['close'].iloc[i]
    
    return factor_values
```
