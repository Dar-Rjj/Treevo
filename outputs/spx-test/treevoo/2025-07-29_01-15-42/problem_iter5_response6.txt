import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['high'] / df['low']
    
    # Compute Relative True Strength Index (RTSI)
    close_diff = df['close'].diff()
    gains = close_diff.where(close_diff > 0, 0).rolling(window=14).mean()
    losses = (-close_diff.where(close_diff < 0, 0)).rolling(window=14).mean()
    rtsi = (gains / (gains + losses)) * 100
    
    # Combine Intraday and Open-to-Close Returns
    open_to_close_return = df['open'] / df['close']
    weighted_intraday_return = intraday_return * 0.5
    weighted_open_to_close_return = open_to_close_return * 0.5
    combined_return = weighted_intraday_return + weighted_open_to_close_return
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std() / df['volume'].rolling(window=220).mean()
    
    # Combine Price and Volume Intensity
    price_volume_intensity = price_range_ratio * rtsi * volume_volatility
    
    # Calculate Daily Price Change
    close_to_close_return = df['close'].pct_change()
    sma_window_length = 10  # Example window length
    sma_smoothed_return = close_to_close_return.rolling(window=sma_window_length).mean()
    
    # Adjust for Volume
    volume_sum = df['volume'].rolling(window=sma_window_length).sum()
    volume_adjusted_momentum = sma_smoothed_return / volume_sum
    
    # Combine Intensity and Momentum Factors
    combined_factor = price_volume_intensity * volume_adjusted_momentum
    
    # Generate Final Alpha Factor
    final_alpha_factor = daily_price_range * rtsi * volume_adjusted_momentum
    
    return final_alpha_factor
