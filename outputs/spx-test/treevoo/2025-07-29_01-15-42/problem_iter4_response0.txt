```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, lookback_moving_avg=10):
    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Average True Range for Volatility
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=lookback_volatility).mean()

    # Determine Smoothing Factor based on ATR
    smoothing_factor = 2 / (1 + atr)

    # Apply Exponential Smoothing to Daily Returns
    smoothed_return = daily_return.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Moving Average of Returns
    moving_avg_return = daily_return.rolling(window=lookback_moving_avg).mean()

    # Compare Smoothed Daily Return with Moving Average of Returns
    trend_strength_signal = smoothed_return - moving_avg_return

    return trend_strength_signal
```
