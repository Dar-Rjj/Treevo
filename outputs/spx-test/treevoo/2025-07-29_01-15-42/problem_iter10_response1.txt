```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']

    # Compute Gains and Losses for RTSI
    df['close_shifted'] = df['close'].shift(1)
    df['diff'] = df['close'] - df['close_shifted']
    df['gains'] = np.where(df['diff'] > 0, df['diff'], 0)
    df['losses'] = np.where(df['diff'] < 0, -df['diff'], 0)
    
    # Smooth Gains and Losses using 14-period EMA
    df['ema_gains'] = df['gains'].ewm(span=14, adjust=False).mean()
    df['ema_losses'] = df['losses'].ewm(span=14, adjust=False).mean()
    
    # Calculate RTSI
    df['rtsi'] = (df['ema_gains'] / (df['ema_gains'] + df['ema_losses'])) * 100

    # Calculate Volume Momentum
    df['vol_20ma'] = df['volume'].rolling(window=20).mean()
    df['volume_momentum'] = df['volume'] - df['vol_20ma']

    # Combine Factors
    df['factor'] = (df['price_range_ratio'] * df['rtsi']) + df['volume_momentum']

    return df['factor']
```
