```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Smooth Returns with EMA (e.g., 5 days)
    df['ema_smoothed_return'] = df['close_to_close_return'].ewm(span=5).mean()
    
    # Summarize Volume and Amount over EMA period
    df['volume_ema_period'] = df['volume'].rolling(window=5).sum()
    df['amount_ema_period'] = df['amount'].rolling(window=5).sum()
    
    # Calculate Adjusted Momentum
    df['adjusted_momentum'] = df['ema_smoothed_return'] / (df['volume_ema_period'] + df['amount_ema_period'])
    
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(-1)
    df['close_to_open_return'] = df['close_to_open_return'].fillna(0)  # Handle NaN for the last row
    
    # Smooth Close-to-Open Return with EMA (e.g., 10 days)
    df['ema_smoothed_close_to_open'] = df['close_to_open_return'].ewm(span=10).mean()
    
    # Adjust for Volume and Amount
    df['volume_ema_period_close_to_open'] = df['volume'].rolling(window=10).sum()
    df['amount_ema_period_close_to_open'] = df['amount'].rolling(window=10).sum()
    df['adjusted_close_to_open'] = df['ema_smoothed_close_to_open'] / (df['volume_ema_period_close_to_open'] + df['amount_ema_period_close_to_open'])
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] / df['close'].shift(28)
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] / df['close'].shift(10)
    
    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']
    
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Relative True Strength Index (RTSI)
    df['gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['smoothed_gain'] = df['gain'].ewm(span=14).mean()
    df['smoothed_loss'] = df['loss'].ewm(span=14).mean()
    df['rtsi'] = (df['smoothed_gain'] / (df['smoothed_gain'] + df['smoothed_loss'])) * 100
    
    # Calculate Realized Volatility (20-day rolling window)
    df['realized_volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = df['close'] - df['open'].shift(-1)
    df['close_to_open_movement'] = df['close_to_open_movement'].fillna(0)  # Handle NaN for the last row
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.35,
        'high_low_spread': 0.2,
        'close_to_open_movement': 0.1,
        'price_range': 0.15,
        'rtsi': 0.1,
        'realized_volatility': 0.1
    }
    df['weighted_average'] = sum(df[col] * weight for col, weight in weights.items())
    
    # Adjust by Volume and Amount
    df['volume_amount_change_ratio'] = (df['volume'] + df['amount']) / (df['volume'].shift(1) + df['amount'].shift(1)) - 1
    df['volume_amount_change_ratio'] = df['volume_amount_change_ratio'].fillna(0)  # Handle NaN for the first row
    
    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['volume_amount_change_ratio']
    
    return df['alpha_factor']
```
