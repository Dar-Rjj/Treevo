```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Component
    sma_short = df['close'].rolling(window=5).mean()
    sma_long = df['close'].rolling(window=20).mean()
    momentum_diff = sma_short - sma_long
    trend_direction = (momentum_diff > 0).astype(int) - (momentum_diff < 0).astype(int)
    
    # Calculate Volume Component
    ema_short_vol = df['volume'].ewm(span=3, adjust=False).mean()
    ema_long_vol = df['volume'].ewm(span=10, adjust=False).mean()
    volume_diff = ema_short_vol - ema_long_vol
    volume_trend = (volume_diff > 0).astype(int) - (volume_diff < 0).astype(int)
    
    # Combine Momentum and Volume Components
    combined_signal = trend_direction * volume_trend
    
    # Generate Alpha Factor
    buy_signal = (combined_signal == 1).astype(int)
    sell_signal = (combined_signal == -1).astype(int)
    alpha_factor = buy_signal - sell_signal
    
    return alpha_factor
```
