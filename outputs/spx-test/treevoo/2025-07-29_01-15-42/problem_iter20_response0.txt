```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Smooth Returns with EMA (10 days)
    df['ema_close_to_open_return'] = df['close_to_open_return'].ewm(span=10).mean()
    
    # Adjust for Volume and Amount
    n_days = 10
    df['total_volume'] = df['volume'].rolling(window=n_days).sum()
    df['total_amount'] = df['amount'].rolling(window=n_days).sum()
    df['adjusted_momentum'] = df['ema_close_to_open_return'] / (df['total_volume'] + df['total_amount'])
    
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['close']
    
    # Calculate Relative True Strength Index (RTSI)
    df['gain'] = (df['close'].diff() > 0) * df['close'].diff()
    df['loss'] = (df['close'].diff() < 0) * (-df['close'].diff())
    df['smoothed_gain'] = df['gain'].ewm(span=14).mean()
    df['smoothed_loss'] = df['loss'].ewm(span=14).mean()
    df['rTSI'] = (df['smoothed_gain'] / (df['smoothed_gain'] + df['smoothed_loss'])) * 100
    
    # Calculate Intraday Volatility (ATR) over 14 days
    df['tr'] = df[['high' - 'low', (df['close'].shift(1) - df['high']).abs(), (df['close'].shift(1) - df['low']).abs()]].max(axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Calculate EMA Smoothed Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    df['ema_volume_change'] = df['volume_change'].ewm(span=5).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['price_range'] * df['rTSI']) + df['adjusted_momentum'] + df['atr'] + df['ema_volume_change']
    
    return df['combined_factor']
```
