```python
import pandas as pd

def heuristics_v2(df):
    # Calculate EMA Smoothed Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(-1)
    ema_close_to_open_return = df['close_to_open_return'].ewm(span=10).mean()

    # Adjust for Volume and Amount
    aggregated_volume = df['volume'].rolling(window=10).sum()
    aggregated_amount = df['amount'].rolling(window=10).sum()
    adjusted_momentum_score = ema_close_to_open_return / (aggregated_volume + aggregated_amount)

    # Calculate Price Range Ratio
    price_range_ratio = (df['high'] - df['low']) / df['close']

    # Compute RTSI
    gains = df['close'].diff().apply(lambda x: max(x, 0))
    losses = df['close'].diff().apply(lambda x: -min(x, 0))
    smoothed_gains = gains.ewm(span=14).mean()
    smoothed_losses = losses.ewm(span=14).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100

    # Calculate EMA Smoothed Daily Volume Change
    daily_volume_change = df['volume'].diff()
    ema_daily_volume_change = daily_volume_change.ewm(span=5).mean()

    # Combine Factors
    combined_factors = (price_range_ratio * rtsi) + adjusted_momentum_score + ema_daily_volume_change

    return combined_factors
```
