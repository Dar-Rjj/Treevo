import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index as 'date' and columns as ['open', 'high', 'low', 'close', 'volume'].
        
    Returns:
        pd.Series: A Series indexed by 'date' representing the factor values.
    """
    
    # Calculate the relative change in close price
    df['rel_close_change'] = df['close'].pct_change()
    
    # Calculate the range of the stock price for the day
    df['range'] = df['high'] - df['low']
    
    # Calculate the average true range over a 10-day period
    df['avg_true_range'] = df['range'].rolling(window=10).mean()
    
    # Calculate the ratio of the current day's range to the 10-day average true range
    df['range_ratio'] = df['range'] / df['avg_true_range']
    
    # Calculate the money flow, which is typical price multiplied by volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Calculate the 10-day cumulative money flow
    df['cum_money_flow'] = df['money_flow'].rolling(window=10).sum()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['rel_close_change'] * df['range_ratio']) * (df['money_flow'] / df['cum_money_flow'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
