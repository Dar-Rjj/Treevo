```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movements
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Analyze Volume Trends
    df['Moving_Average_Volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['Moving_Average_Volume']
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.5  # Example threshold, can be optimized
    conditions = [
        (df['Volume_Adjusted_Price_Range'] > threshold),
        (df['Volume_Adjusted_Price_Range'] < -threshold)
    ]
    choices = [1, -1]
    df['Preliminary_Alpha_Factor'] = pd.Series(0, index=df.index)  # Default value
    df['Preliminary_Alpha_Factor'] = pd.np.select(conditions, choices, default=0)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    df['Close_to_Open_Movement'] = df['close'] - df['open']
    
    # Weighted Average of Intraday Components
    weight_high_low = 0.7
    weight_close_open = 0.3
    df['Intraday_Component'] = (weight_high_low * df['Intraday_High_Low_Spread']) + (weight_close_open * df['Close_to_Open_Movement'])
    
    # Calculate Momentum Score
    df['Momentum_Score'] = df['Intraday_Component'].diff()
    
    # Generate Final Alpha Signal
    conditions_final = [
        (df['Preliminary_Alpha_Factor'] == 1),
        (df['Preliminary_Alpha_Factor'] == -1)
    ]
    choices_final = [df['Preliminary_Alpha_Factor'] + df['Momentum_Score'], df['Preliminary_Alpha_Factor'] - df['Momentum_Score']]
    df['Final_Alpha_Factor'] = pd.np.select(conditions_final, choices_final, default=df['Momentum_Score'])
    
    return df['Final_Alpha_Factor']
```
