```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].shift(28) - df['close']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].shift(10) - df['close']
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Close-to-Open Movement
    close_to_open_movement = df['open'] - df['close'].shift(1)
    
    # Calculate Daily Close-to-Close Return
    close_to_close_return = df['close'].diff()
    
    # Smooth with Exponential Moving Average (EMA)
    ema_5 = close_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount
    volume_sum = df['volume'].rolling(window=5).sum()
    amount_sum = df['amount'].rolling(window=5).sum()
    combined_volume_amount = volume_sum + amount_sum
    momentum_to_volume_amount_ratio = ema_5 / combined_volume_amount
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = np.where(df['close'].diff() > 0, df['close'].diff(), 0)
    losses = np.where(df['close'].diff() < 0, -df['close'].diff(), 0)
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * intraday_high_low_spread +
        0.1 * close_to_open_movement +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'].shift(10) - df['volume']
    final_factor = weighted_average * volume_acceleration * momentum_to_volume_amount_ratio
    
    return final_factor
```
