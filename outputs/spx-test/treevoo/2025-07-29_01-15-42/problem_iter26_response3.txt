```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change of close price from the previous day
    close_price_change = df['close'].pct_change()
    
    # Calculate the average volume over the last 10 days
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    
    # Calculate the alpha factor: (price range / close price) * (change in close price / average volume)
    alpha_factor = (price_range / df['close']) * (close_price_change / avg_volume_10_days)
    
    return alpha_factor
```
