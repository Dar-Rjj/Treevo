```python
import pandas as pd

def heuristics_v2(df):
    # Compute daily indicators
    df['daily_price_range'] = df['high'] - df['low']
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate moving averages
    ma_volume = df['volume'].rolling(window=20).mean()
    sma_period = 5  # Example SMA period, can be adjusted
    total_volume_sma = df['volume'].rolling(window=sma_period).sum()
    
    # Generate alpha signal based on enhanced price-volume relationship
    df['volume_adjusted_price_range'] = df['daily_price_range'] / ma_volume
    df['sma_daily_return'] = df['daily_return'].rolling(window=sma_period).mean()
    df['volume_adjusted_momentum'] = df['sma_daily_return'] / total_volume_sma
    
    # Calculate intraday and open-to-close returns
    df['intraday_return'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine intraday and open-to-close returns
    weighted_intraday_return = 0.5 * df['intraday_return']
    weighted_open_to_close_return = 0.5 * df['open_to_close_return']
    df['combined_returns'] = weighted_intraday_return + weighted_open_to_close_return
    
    # Dynamic threshold calculation
    std_dev = df['volume_adjusted_price_range'].rolling(window=20).std()
    dynamic_threshold = 2 * std_dev
    
    # Synthesize final alpha factor
    alpha_factor = pd.Series(index=df.index, dtype=float)
    for i in range(len(df)):
        if df['volume_adjusted_price_range'].iloc[i] > dynamic_threshold.iloc[i]:
            alpha_factor.iloc[i] = 1
        elif df['volume_adjusted_price_range'].iloc[i] < -dynamic_threshold.iloc[i]:
            alpha_factor.iloc[i] = -1
        else:
            alpha_factor.iloc[i] = df['volume_adjusted_momentum'].iloc[i] + df['combined_returns'].iloc[i]
    
    return alpha_factor
```
