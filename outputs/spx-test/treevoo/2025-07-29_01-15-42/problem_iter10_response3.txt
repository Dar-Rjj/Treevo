import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility (Close price)
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_high_low = high_low_range / avg_volume
    
    # Incorporate Volume Trend
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_momentum = high_low_range * volume_change
    
    # Compute Daily Price Range Ratio
    daily_price_range_ratio = high_low_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    close_diff = df['close'].diff()
    gains = close_diff.where(close_diff > 0, 0).rolling(window=14).mean()
    losses = (-close_diff.where(close_diff < 0, 0)).rolling(window=14).mean()
    rtsi = (gains / (gains + losses)) * 100
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Adjust for Volume
    summarized_volume = df['volume'].rolling(window=14).sum()
    volume_adjusted_momentum = daily_price_change / summarized_volume
    
    # Combine Factors
    combined_factor = (volume_adjusted_high_low * rtsi * volume_adjusted_momentum)
    
    return combined_factor
