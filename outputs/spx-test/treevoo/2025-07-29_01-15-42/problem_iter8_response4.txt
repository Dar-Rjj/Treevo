import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = df['close'] - df['open']
    
    # Compute Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Generate Adjusted Price-Volume Signal
    volume_adjusted_price_range = daily_price_range / df['volume']
    
    # Calculate Realized Volatility (using a rolling window of 20 days)
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Momentum Score
    momentum_score = volume_adjusted_price_range - volume_adjusted_price_range.shift(1)
    
    # Determine the threshold for the Volume-Adjusted Price Range
    threshold = np.median(volume_adjusted_price_range)
    
    # Initialize Alpha Factor
    alpha_factor = pd.Series(index=df.index, dtype=float)
    
    # Combine Indicators
    for i in range(1, len(df)):
        if (momentum_score[i] > 0 and volume_adjusted_price_range[i] > threshold) and realized_volatility[i] < np.median(realized_volatility):
            alpha_factor[i] = 1
        elif (momentum_score[i] < 0 and volume_adjusted_price_range[i] < -threshold) and realized_volatility[i] < np.median(realized_volatility):
            alpha_factor[i] = -1
        else:
            alpha_factor[i] = 0
    
    return alpha_factor
