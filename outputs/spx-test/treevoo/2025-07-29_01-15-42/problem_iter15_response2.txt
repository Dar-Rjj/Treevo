```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Smooth with Exponential Moving Average (EMA)
    ema_window = 10
    df['ema_smoothed_return'] = df['daily_return'].ewm(span=ema_window, adjust=False).mean()

    # Adjust for Volume and Amount
    df['total_volume'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount'] = df['amount'].rolling(window=ema_window).sum()
    df['combined_vol_amt'] = df['total_volume'] + df['total_amount']
    df['momentum_adj'] = df['ema_smoothed_return'] / df['combined_vol_amt']

    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']

    # Compute Relative True Strength Index (RTSI)
    df['gain_loss'] = df['close'].diff()
    df['gains'] = df['gain_loss'].apply(lambda x: max(x, 0))
    df['losses'] = df['gain_loss'].apply(lambda x: abs(min(x, 0)))
    df['smoothed_gains'] = df['gains'].ewm(span=14, adjust=False).mean()
    df['smoothed_losses'] = df['losses'].ewm(span=14, adjust=False).mean()
    df['rtsi'] = 100 * (df['smoothed_gains'] / (df['smoothed_gains'] + df['smoothed_losses']))

    # Calculate Intraday Volatility (ATR)
    atr_period = 14
    df['tr'] = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    df['atr'] = df['tr'].ewm(span=atr_period, adjust=False).mean()

    # Combine Factors
    df['combined_factor'] = (df['price_range_ratio'] * df['rtsi']) + df['momentum_adj']
    df['alpha_factor'] = df['combined_factor'] * (df['total_volume'] + df['total_amount']) / df['atr']

    return df['alpha_factor'].dropna()
```
