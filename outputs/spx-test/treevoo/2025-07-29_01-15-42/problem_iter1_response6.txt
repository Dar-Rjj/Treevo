```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(14)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(5)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(5)
    
    # Multiply Combined Momentum by Volume Acceleration
    factor_values = combined_momentum * volume_acceleration
    
    return factor_values
```
