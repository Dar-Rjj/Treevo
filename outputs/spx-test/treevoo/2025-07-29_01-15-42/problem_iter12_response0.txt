```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] / df['low']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_shifted'] = df['close'].shift(1)
    df['gain'] = df.apply(lambda x: x['close'] - x['close_shifted'] if x['close'] > x['close_shifted'] else 0, axis=1)
    df['loss'] = df.apply(lambda x: x['close_shifted'] - x['close'] if x['close'] < x['close_shifted'] else 0, axis=1)
    df['smoothed_gain'] = df['gain'].ewm(span=14).mean()
    df['smoothed_loss'] = df['loss'].ewm(span=14).mean()
    df['rtsi'] = (df['smoothed_gain'] / (df['smoothed_gain'] + df['smoothed_loss'])) * 100
    
    # Combine Intraday and Open-to-Close Returns
    df['open_to_close_return'] = df['open'] / df['close']
    df['weighted_intraday_return'] = df['intraday_return'] * 0.6
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * 0.4
    df['combined_return'] = df['weighted_intraday_return'] + df['weighted_open_to_close_return']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['open']
    
    # Calculate Volume Momentum
    df['volume_ma20'] = df['volume'].rolling(window=20).mean()
    df['volume_momentum'] = df['volume'] - df['volume_ma20']
    
    # Combine Price and Volume Intensity
    df['price_volume_intensity'] = (df['price_range_ratio'] * df['rtsi']) + df['volume_momentum']
    
    # Calculate Daily Price Change
    df['close_to_close_return'] = df['close'].diff()
    df['sma_5'] = df['close_to_close_return'].rolling(window=5).mean()
    
    # Adjust for Volume
    df['total_volume_5'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = df['sma_5'] / df['total_volume_5']
    
    # Combine Intensity and Momentum Factors
    df['combined_factor'] = df['price_volume_intensity'] * df['volume_adjusted_momentum']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['daily_price_range'] * df['rtsi']) + df['volume_adjusted_momentum']
    
    return df['final_alpha_factor']
```
