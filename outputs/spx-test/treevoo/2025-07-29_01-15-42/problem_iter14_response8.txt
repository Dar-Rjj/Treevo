```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_volume_signal = daily_price_change / moving_average_volume
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(periods=28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(periods=10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Average Volume
    adjusted_high_low_range = high_low_range / moving_average_volume
    
    # Weighted Average of Combined Components
    weighted_average = (combined_momentum * 0.4 +
                        high_low_range * 0.3 +
                        daily_price_change * 0.2 +
                        adjusted_high_low_range * 0.1)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(periods=10)
    
    # Final Alpha Signal Based on Combined Indicators
    alpha_factor = pd.Series(index=df.index, dtype=float)
    threshold = 0.01  # Example threshold value
    median_volatility = realized_volatility.median()
    
    for t in df.index:
        if adjusted_price_volume_signal.loc[t] > threshold and realized_volatility.loc[t] < median_volatility:
            alpha_factor.loc[t] = 1
        elif adjusted_price_volume_signal.loc[t] < -threshold and realized_volatility.loc[t] < median_volatility:
            alpha_factor.loc[t] = -1
        else:
            alpha_factor.loc[t] = weighted_average.loc[t] * volume_acceleration.loc[t]
    
    return alpha_factor
```
