```python
def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Range Midpoint Deviation
    range_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - range_midpoint

    # Calculate Volume and Amount Change
    volume_change = df['volume'] - df['volume'].shift(1)
    amount_change = df['amount'] - df['amount'].shift(1)

    # Combine Volume and Amount Trends with Close to Midpoint Deviation
    combined_factor = (close_to_midpoint_deviation * volume_change * amount_change) / (intraday_range + 1e-6)

    return combined_factor
```
