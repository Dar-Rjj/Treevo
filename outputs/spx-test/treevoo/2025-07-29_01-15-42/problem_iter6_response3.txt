```python
import pandas as pd

def heuristics_v2(df, sma_window):
    # Compute Close-to-Close Return
    close_returns = df['close'].pct_change()

    # Smooth with Simple Moving Average (SMA)
    smoothed_close_returns = close_returns.rolling(window=sma_window).mean()

    # Summarize Volume over the SMA period
    total_volume = df['volume'].rolling(window=sma_window).sum()

    # Summarize Amount over the SMA period
    total_amount = df['amount'].rolling(window=sma_window).sum()

    # Combine summarized volume and amount by summing them
    combined_total = total_volume + total_amount

    # Calculate Ratio of Momentum to Combined Volume and Amount
    adjusted_momentum = smoothed_close_returns / combined_total

    # Compute High-to-Low Range
    high_low_range = df['high'] - df['low']

    # Smooth with Simple Moving Average (SMA)
    smoothed_high_low_range = high_low_range.rolling(window=sma_window).mean()

    # Combine High-to-Low Range and Close-to-Close Return
    combined_momentum = smoothed_close_returns + smoothed_high_low_range

    # Calculate Ratio of Combined Momentum to Combined Volume and Amount
    adjusted_combined_momentum = combined_momentum / combined_total

    return adjusted_combined_momentum
```
