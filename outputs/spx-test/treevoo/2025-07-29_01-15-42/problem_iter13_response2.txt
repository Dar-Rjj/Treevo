```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute Moving Average Volume (Last 20 Days' Volumes)
    df['ma_volume'] = df['volume'].rolling(window=20).mean()
    
    # Generate Volume-Adjusted Price Range
    df['volume_adjusted_range'] = df['daily_price_range'] / df['ma_volume']
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Combine Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']
    
    # Calculate Intraday Price Range Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Close to Range Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - df['intraday_midpoint']
    
    # Incorporate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = df['close'] - df['open']
    
    # Weighted Average of Components
    df['weighted_average'] = (
        0.3 * df['combined_momentum'] +
        0.2 * df['daily_price_range'] +
        0.1 * df['close_to_open_movement'] +
        0.2 * df['volume_adjusted_range'] +
        0.2 * df['close_to_midpoint_deviation']
    )
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Generate Alpha Signal
    df['alpha_signal'] = df['weighted_average'] * df['volume_acceleration']
    
    # Final Alpha Factor
    threshold = 0.05
    df['alpha_factor'] = df['alpha_signal'].apply(lambda x: 1 * x if x > threshold else (-1 * x if x < -threshold else 0))
    
    return df['alpha_factor']
```
