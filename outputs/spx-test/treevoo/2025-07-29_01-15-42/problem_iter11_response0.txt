```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']

    # Calculate Close-to-Open Change
    df['Close_to_Open_Change'] = df['close'] - df['open']

    # Analyze Volume Trends
    df['Moving_Average_Volume'] = df['volume'].rolling(window=20).mean()

    # Generate Volume-Adjusted Indicators
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['Moving_Average_Volume']
    df['Volume_Adjusted_High_Low_Spread'] = df['high'] - df['low'] * df['volume']
    df['Volume_Adjusted_Close_to_Open_Change'] = df['Close_to_Open_Change'] * df['volume']

    # Calculate Realized Volatility
    df['Realized_Volatility'] = df['close'].pct_change().rolling(window=20).std()

    # Define the threshold and median for realized volatility
    threshold = 0.5
    median_volatility = df['Realized_Volatility'].median()

    # Initialize the alpha factor series
    df['Alpha_Factor'] = 0

    # Final Alpha Signal Based on Combined Indicators
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'Volume_Adjusted_Price_Range'] > threshold and df.loc[df.index[i], 'Realized_Volatility'] < median_volatility:
            df.loc[df.index[i], 'Alpha_Factor'] = 1
        elif df.loc[df.index[i], 'Volume_Adjusted_Price_Range'] < -threshold and df.loc[df.index[i], 'Realized_Volatility'] < median_volatility:
            df.loc[df.index[i], 'Alpha_Factor'] = -1
        else:
            # Generate Preliminary Alpha Signal
            if df.loc[df.index[i], 'Volume_Adjusted_High_Low_Spread'] > 0 and df.loc[df.index[i], 'Volume_Adjusted_Close_to_Open_Change'] > 0:
                preliminary_alpha = 1
            elif df.loc[df.index[i], 'Volume_Adjusted_High_Low_Spread'] < 0 and df.loc[df.index[i], 'Volume_Adjusted_Close_to_Open_Change'] < 0:
                preliminary_alpha = -1
            else:
                preliminary_alpha = 0

            # Calculate Momentum Score
            momentum_score = df.loc[df.index[i], 'Volume_Adjusted_Price_Range'] - df.loc[df.index[i-1], 'Volume_Adjusted_Price_Range']

            # Adjust the preliminary alpha based on momentum score
            if preliminary_alpha == 1:
                df.loc[df.index[i], 'Alpha_Factor'] = preliminary_alpha + momentum_score
            elif preliminary_alpha == -1:
                df.loc[df.index[i], 'Alpha_Factor'] = preliminary_alpha - momentum_score
            else:
                df.loc[df.index[i], 'Alpha_Factor'] = momentum_score

    return df['Alpha_Factor']
```
