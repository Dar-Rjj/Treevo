```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Smooth with Exponential Moving Average (EMA)
    ema_window = 10
    df['ema_return'] = df['daily_return'].ewm(span=ema_window, adjust=False).mean()

    # Adjust for Volume and Amount
    df['total_volume'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount'] = df['amount'].rolling(window=ema_window).sum()
    df['combined_vol_amount'] = df['total_volume'] + df['total_amount']
    df['adjusted_momentum'] = df['ema_return'] / df['combined_vol_amount']

    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']

    # Compute Relative True Strength Index (RTSI)
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    rtsi_window = 14
    df['ema_gain'] = df['gain'].ewm(span=rtsi_window, adjust=False).mean()
    df['ema_loss'] = df['loss'].ewm(span=rtsi_window, adjust=False).mean()
    df['rtsi'] = (df['ema_gain'] / (df['ema_gain'] + df['ema_loss'])) * 100

    # Calculate Daily Volatility
    df['daily_volatility'] = df['daily_return'].rolling(window=ema_window).std()

    # Combine Factors
    df['combined_factor'] = (df['price_range_ratio'] * df['rtsi'] + df['adjusted_momentum']) * df['adjusted_momentum'] / df['daily_volatility']

    # Drop rows with NaN values
    df.dropna(inplace=True)

    return df['combined_factor']
```
