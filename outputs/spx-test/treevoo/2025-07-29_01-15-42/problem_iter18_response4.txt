```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume (Last 20 Days)
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_vol_signal = daily_price_change / moving_avg_volume
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(periods=30)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility (Close price, rolling window of 15 days)
    realized_volatility = df['close'].rolling(window=15).std()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Average Volume (over the same rolling window of 15 days)
    avg_volume_15 = df['volume'].rolling(window=15).mean()
    adjusted_high_low_range = high_low_range / avg_volume_15
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.4 * combined_momentum +
        0.3 * high_low_range +
        0.2 * daily_price_change +
        0.1 * adjusted_high_low_range
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(periods=15)
    
    # Define Thresholds
    threshold = 0.02
    volatility_median = realized_volatility.median()
    
    # Final Alpha Signal Based on Combined Indicators
    alpha_factor = pd.Series(index=df.index, dtype=float)
    for t in df.index:
        if adjusted_price_vol_signal[t] > threshold and realized_volatility[t] < volatility_median:
            alpha_factor[t] = 1
        elif adjusted_price_vol_signal[t] < -threshold and realized_volatility[t] < volatility_median:
            alpha_factor[t] = -1
        else:
            alpha_factor[t] = weighted_average[t] * volume_acceleration[t]
    
    return alpha_factor
```
