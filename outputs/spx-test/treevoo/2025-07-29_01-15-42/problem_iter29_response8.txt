```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
        
    Returns:
        pd.Series: Alpha factor values indexed by date
    """
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the alpha factor
    alpha_factor = (avg_price * price_range) * volume_change
    
    return alpha_factor
```
