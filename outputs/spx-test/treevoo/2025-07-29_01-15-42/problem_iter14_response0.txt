import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] / df['close'].shift(28) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] / df['close'].shift(5) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=20).mean()
    adjusted_high_low_range = high_low_range / average_volume
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = (df['close'] - df['open']) / df['close']
    
    # Weighted Average of Combined Components
    weighted_average = (combined_momentum * 0.4 +
                        intraday_high_low_spread * 0.3 +
                        close_to_open_movement * 0.2 +
                        adjusted_high_low_range * 0.1)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] / df['volume'].shift(10) - 1
    
    # Adjust by Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.01
    positive_imbalance = (adjusted_high_low_range > threshold).astype(int)
    negative_imbalance = (adjusted_high_low_range < -threshold).astype(int)
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha_signal = positive_imbalance - negative_imbalance
    
    # Calculate Momentum Score
    momentum_score = weighted_average - weighted_average.shift(1)
    
    # Generate Final Alpha Signal
    final_alpha_signal = pd.Series(index=df.index, dtype=float)
    final_alpha_signal[positive_imbalance == 1] = momentum_score + preliminary_alpha_signal
    final_alpha_signal[negative_imbalance == 1] = -momentum_score + preliminary_alpha_signal
    final_alpha_signal[(positive_imbalance == 0) & (negative_imbalance == 0)] = 0
    
    return final_alpha_signal
