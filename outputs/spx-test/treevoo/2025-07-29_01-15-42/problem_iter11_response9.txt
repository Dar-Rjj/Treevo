```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] / df['close'].shift(14) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] / df['close'].shift(5) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust High-to-Low Range by Moving Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_high_low_range = high_low_range / moving_average_volume
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = df['close'] - df['open']
    
    # Weighted Average of Combined Components
    weights = [0.4, 0.2, 0.2, 0.2]
    weighted_average = (combined_momentum * weights[0] +
                        intraday_high_low_spread * weights[1] +
                        close_to_open_movement * weights[2] +
                        volume_adjusted_high_low_range * weights[3])
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] / df['volume'].shift(5)
    
    # Final Factor
    final_factor = weighted_average * volume_acceleration
    
    # Set Alpha Factor Value based on conditions
    threshold = 0.05
    median_realized_volatility = realized_volatility.median()
    alpha_factor = pd.Series(index=df.index, data=0)
    alpha_factor[(volume_adjusted_high_low_range > threshold) & (realized_volatility < median_realized_volatility)] = 1
    alpha_factor[(volume_adjusted_high_low_range < -threshold) & (realized_volatility < median_realized_volatility)] = -1
    
    return alpha_factor
```
