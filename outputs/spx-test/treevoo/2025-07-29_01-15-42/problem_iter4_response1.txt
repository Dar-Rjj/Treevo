```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Moving Average Volume (last 20 days)
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Volume-Adjusted Price Range
    volume_adjusted_price_range = daily_price_range / moving_average_volume
    
    # Define threshold
    threshold = 0.01  # Example threshold, can be adjusted based on empirical analysis
    
    # Generate Alpha Signal Based on Price-Volume Relationship
    alpha_factor = pd.Series(0, index=df.index)
    alpha_factor[volume_adjusted_price_range > threshold] = 1
    alpha_factor[volume_adjusted_price_range < -threshold] = -1
    
    return alpha_factor
```
