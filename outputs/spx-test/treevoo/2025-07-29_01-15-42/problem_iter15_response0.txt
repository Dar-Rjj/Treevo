```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of the close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Analyze the spread between high and low prices as a measure of volatility
    df['hl_spread'] = df['high'] - df['low']
    
    # Introduce a lagged term to evaluate the change in daily volatility over time
    df['hl_spread_change'] = df['hl_spread'].diff()
    
    # Consider the relationship between closing price and volume, such as Price-Volume Trend (PVT)
    df['pvt'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    df['pvt_ma5'] = df['pvt'].rolling(window=5).mean()
    df['pvt_diff'] = df['pvt'] - df['pvt_ma5']
    
    # Investigate the trend of the amount by comparing it with its own moving average
    df['amount_ma5'] = df['amount'].rolling(window=5).mean()
    df['amount_trend'] = df['amount'] - df['amount_ma5']
    
    # Create an alpha score based on the combination of the percentage change, PVT, and the amount trend
    df['alpha_score'] = 0.3 * df['close_pct_change'] + 0.4 * df['pvt_diff'] + 0.3 * df['amount_trend']
    
    # Incorporate intraday momentum by analyzing the difference between open and close prices
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Augment this factor with the relative position of the close price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine all factors into a final composite alpha factor
    df['composite_alpha'] = 0.3 * df['close_pct_change'] + 0.2 * df['pvt_diff'] + 0.2 * df['amount_trend'] + 0.2 * df['intraday_momentum'] + 0.1 * df['close_position']
    
    return df['composite_alpha'].dropna()
```
