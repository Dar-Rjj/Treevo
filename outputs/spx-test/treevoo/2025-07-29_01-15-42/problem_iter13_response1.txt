import pandas as pd

def heuristics_v2(df, threshold=0.05):
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA) - 5 days window
    df['sma_return'] = df['close_to_close_return'].rolling(window=5).mean()
    
    # Summarize Volume over the SMA period
    df['sum_volume'] = df['volume'].rolling(window=5).sum()
    
    # Summarize Amount over the SMA period
    df['sum_amount'] = df['amount'].rolling(window=5).sum()
    
    # Combine summarized volume and amount by summing them
    df['combined_vol_amt'] = df['sum_volume'] + df['sum_amount']
    
    # Calculate Ratio of Momentum to Combined Volume and Amount
    df['vol_amt_adjusted_momentum'] = df['sma_return'] / df['combined_vol_amt']
    
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['gain'] = (df['close'] - df['close'].shift(1)).apply(lambda x: max(x, 0))
    df['loss'] = (df['close'] - df['close'].shift(1)).apply(lambda x: max(-x, 0))
    df['ema_gain'] = df['gain'].ewm(span=14, adjust=False).mean()
    df['ema_loss'] = df['loss'].ewm(span=14, adjust=False).mean()
    df['rtsi'] = (df['ema_gain'] / (df['ema_gain'] + df['ema_loss'])) * 100
    
    # Calculate Volume Momentum
    df['volume_20d_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_momentum'] = df['volume'] - df['volume_20d_ma']
    
    # Combine Factors
    combined_factors = df['price_range_ratio'] * df['rtsi'] + df['volume_momentum']
    df['combined_factors'] = combined_factors * df['vol_amt_adjusted_momentum']
    
    # Generate Volume-Adjusted Price Range
    df['vol_adj_price_range'] = df['price_range'] / df['volume_20d_ma']
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(14)
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']
    
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = df['close'] - df['open']
    
    # Weighted Average of Momentum and Intraday Components
    weights = {'combined_momentum': 0.5, 'intraday_spread': 0.3, 'close_to_open_movement': 0.2}
    weighted_average = (df['combined_momentum'] * weights['combined_momentum'] +
                       df['intraday_spread'] * weights['intraday_spread'] +
                       df['close_to_open_movement'] * weights['close_to_open_movement'])
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'] - df['volume'].shift(5)
    
    # Generate Alpha Signal
    df['momentum_volume_adjusted_signal'] = weighted_average * df['volume_acceleration']
    
    # Final Alpha Factor
    alpha_factor = []
    for i in range(len(df)):
        if df['vol_adj_price_range'].iloc[i] > threshold:
            alpha_factor.append(1 * (df['momentum_volume_adjusted_signal'].iloc[i] * df['combined_factors'].iloc[i]))
        elif df['vol_adj_price_range'].iloc[i] < -threshold:
            alpha_factor.append(-1 * (df['momentum_volume_adjusted_signal'].iloc[i] * df['combined_factors'].iloc[i]))
        else:
            alpha_factor.append(0)
    
    return pd.Series(alpha_factor, index=df.index)
