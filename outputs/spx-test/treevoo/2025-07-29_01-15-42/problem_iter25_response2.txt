```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    long_term_momentum = (df['close'] / df['close'].shift(28)).fillna(1)
    short_term_momentum = (df['close'] / df['close'].shift(10)).fillna(1)
    combined_momentum = 0.5 * (long_term_momentum + short_term_momentum)
    
    # Calculate Realized Volatility
    log_returns = np.log(df['close']).diff()
    realized_volatility = log_returns.rolling(window=20).std().fillna(0)
    
    # Calculate Intraday High-Low Spread
    high_low_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_returns = (df['open'].shift(-1) - df['close']) / df['close']
    ema_close_to_open = close_to_open_returns.ewm(span=10, min_periods=10).mean().fillna(0)
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=10).sum()
    total_amount = (df['close'] * df['volume']).rolling(window=10).sum()
    adjusted_ema_return = ema_close_to_open / (total_volume + total_amount)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0) * df['close'].diff()
    losses = (df['close'].diff() < 0) * df['close'].diff().abs()
    smoothed_gains = gains.ewm(span=14, min_periods=14).mean().fillna(0)
    smoothed_losses = losses.ewm(span=14, min_periods=14).mean().fillna(0)
    rtsi = 100 * (smoothed_gains / (smoothed_gains + smoothed_losses))
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * high_low_spread +
        0.1 * adjusted_ema_return +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'] / df['volume'].shift(10)
    final_factor = weighted_average * volume_acceleration
    
    return final_factor
```
