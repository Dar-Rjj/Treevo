```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility over a 20-day rolling window
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust by Volume (High-to-Low Range divided by Average Volume over the same rolling window)
    avg_volume = df['volume'].rolling(window=20).mean()
    high_low_vol_adj = high_low_range / avg_volume
    
    # Calculate Close to Range Midpoint Deviation
    range_midpoint = (df['high'] + df['low']) / 2
    close_mid_dev = df['close'] - range_midpoint
    
    # Incorporate Volume Trend
    volume_change = df['volume'].diff()
    vol_adj_momentum = (volume_change * close_mid_dev) / df['volume'].shift(1)
    
    # Compute Daily Price Range Ratio
    daily_price_range_ratio = high_low_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0) * df['close'].diff()
    losses = (df['close'].diff() < 0) * abs(df['close'].diff())
    sma_gains = gains.rolling(window=14).mean()
    sma_losses = losses.rolling(window=14).mean()
    rtsi = (sma_gains / (sma_gains + sma_losses)) * 100
    
    # Calculate Daily Price Change
    daily_return = df['close'].diff()
    sma_daily_return = daily_return.rolling(window=5).mean()
    
    # Adjust for Volume
    total_volume = df['volume'].rolling(window=5).sum()
    vol_adj_momentum_score = sma_daily_return / total_volume
    
    # Combine Factors
    alpha_factor = (realized_volatility * rtsi * vol_adj_momentum_score)

    return alpha_factor
```
