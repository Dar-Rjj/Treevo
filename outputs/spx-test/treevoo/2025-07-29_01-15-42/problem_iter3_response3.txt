```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['gain'] = df['close'].diff()
    df['gain'] = df['gain'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = -df['close'].diff()
    df['loss'] = df['loss'].apply(lambda x: x if x > 0 else 0)
    
    # Smooth Gains and Losses
    df['smoothed_gain'] = df['gain'].ewm(span=14, adjust=False).mean()
    df['smoothed_loss'] = df['loss'].ewm(span=14, adjust=False).mean()
    
    # Calculate RTSI
    df['rtsi'] = (df['smoothed_gain'] / (df['smoothed_gain'] + df['smoothed_loss'])) * 100
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['price_range_ratio'] * df['rtsi']) + df['volume_momentum']
    
    return df['alpha_factor']
```
