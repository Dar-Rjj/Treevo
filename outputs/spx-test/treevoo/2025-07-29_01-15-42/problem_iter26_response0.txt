import pandas as pd

def heuristics_v2(df):
    # Sub-thought 1.1: Moving Average Crossover
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    ma_crossover = short_term_ma - long_term_ma
    
    # Sub-thought 1.2: Momentum Indicator
    momentum = df['close'] - df['close'].shift(10)
    
    # Sub-thought 2.1: On-Balance Volume (OBV)
    obv = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Sub-thought 2.2: Volume Oscillator
    short_term_avg_vol = df['volume'].rolling(window=5).mean()
    long_term_avg_vol = df['volume'].rolling(window=20).mean()
    volume_oscillator = (short_term_avg_vol / long_term_avg_vol) - 1
    
    # Sub-thought 3.1: Bollinger Bands
    bollinger_bands = (df['close'] - df['close'].rolling(window=20).mean()) / (2 * df['close'].rolling(window=20).std())
    
    # Sub-thought 3.2: ATR (Average True Range)
    true_range = df[['high', 'low']].diff().abs().max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Sub-thought 4.1: Inside Bar
    inside_bar = (df['high'] < df['high'].shift(1)) & (df['low'] > df['low'].shift(1)).astype(int)
    
    # Sub-thought 4.2: Outside Bar
    outside_bar = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1)).astype(int)
    
    # Sub-thought 5.1: Combine Trend and Volume
    combined_trend_volume = ma_crossover * volume_oscillator
    
    # Sub-thought 5.2: Combine Volatility and Momentum
    combined_volatility_momentum = bollinger_bands * momentum
    
    # Composite Alpha Factor
    composite_factor = (combined_trend_volume + combined_volatility_momentum) / 2
    return composite_factor
