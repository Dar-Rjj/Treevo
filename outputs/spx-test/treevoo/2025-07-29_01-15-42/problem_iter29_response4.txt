```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term and Short-Term Momentum
    df['Long_Term_Momentum'] = df['close'].pct_change(periods=28)
    df['Short_Term_Momentum'] = df['close'].pct_change(periods=10)
    df['Combined_Momentum'] = 0.7 * df['Long_Term_Momentum'] + 0.3 * df['Short_Term_Momentum']

    # Calculate Realized Volatility over a rolling window of 20 days
    df['Log_Returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['Realized_Volatility'] = df['Log_Returns'].rolling(window=20).std()

    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # EMA Smoothed Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'].shift(-1) / df['close']) - 1
    df['EMA_Smoothed_Close_to_Open_Return'] = df['Close_to_Open_Return'].ewm(span=10, adjust=False).mean()

    # Aggregate total volume and amount over EMA period
    df['Total_Volume'] = df['volume'].rolling(window=10).sum()
    df['Total_Amount'] = (df['close'] * df['volume']).rolling(window=10).sum()
    df['Volume_Amount_Adjusted_Return'] = df['EMA_Smoothed_Close_to_Open_Return'] / (df['Total_Volume'] + df['Total_Amount'])

    # Daily Price Range and Price Range Ratio
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['Price_Range_Ratio'] = df['Daily_Price_Range'] / df['close']

    # Relative True Strength Index (RTSI)
    df['Gains'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Losses'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['Smoothed_Gains'] = df['Gains'].ewm(span=14, adjust=False).mean()
    df['Smoothed_Losses'] = df['Losses'].ewm(span=14, adjust=False).mean()
    df['RTSI'] = 100 * (df['Smoothed_Gains'] / (df['Smoothed_Gains'] + df['Smoothed_Losses']))

    # Weighted Average of Combined Components
    df['Weighted_Average'] = (
        0.3 * df['Combined_Momentum'] +
        0.2 * df['Intraday_High_Low_Spread'] +
        0.1 * df['Volume_Amount_Adjusted_Return'] +
        0.1 * df['Realized_Volatility'] +
        0.15 * df['Price_Range_Ratio'] +
        0.15 * df['RTSI']
    )

    # Volume Change and Adjust for Volume and Amount Impact
    df['Volume_Change'] = df['volume'].pct_change()
    df['Final_Factor'] = df['Weighted_Average'] * df['Volume_Change'] / (df['Total_Volume'] + df['Total_Amount'])

    return df['Final_Factor'].dropna()
```
