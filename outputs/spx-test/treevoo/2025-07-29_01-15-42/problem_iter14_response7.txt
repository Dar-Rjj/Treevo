```python
def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate the 7-Day Average of Daily Price Changes
    avg_7day_price_change = daily_price_change.rolling(window=7).mean()
    
    # Calculate the 21-Day Average of Daily Price Changes
    avg_21day_price_change = daily_price_change.rolling(window=21).mean()
    
    # Combine Averages to Identify Long-Term Momentum
    long_term_momentum = (avg_21day_price_change - avg_7day_price_change)
    
    # Measure the Distance Between High and Low
    high_low_range = df['high'] - df['low']
    
    # Calculate the 5-Day Average High-Low Range
    avg_5day_high_low_range = high_low_range.rolling(window=5).mean()
    
    # Evaluate the Ratio of Today's Range to its 5-Day Average
    reversal_signal = (high_low_range / avg_5day_high_low_range) - 1
    
    # Compute Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the Moving Average Volume over 20 Days
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the Summarized Volume over 5 Days
    summarized_volume = df['volume'].rolling(window=5).sum()
    
    # Generate Volume-Adjusted Price Range
    volume_adjusted_price_range = high_low_range / moving_avg_volume
    
    # Smooth Daily Return with a 5-Day Simple Moving Average
    smoothed_daily_return = (df['close'] - df['close'].shift(1)).pct_change().rolling(window=5).mean()
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = smoothed_daily_return / summarized_volume
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Compute Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = 0.5 * intraday_return + 0.5 * open_to_close_return
    
    # Set threshold for volume-adjusted price range
    threshold = 0.1  # Example threshold, can be adjusted
    
    # Final Alpha Factor Based on Volume-Adjusted Price Range and Momentum
    alpha_factor = (volume_adjusted_momentum + combined_returns) / 2
    alpha_factor = alpha_factor.where((volume_adjusted_price_range <= threshold) & (volume_adjusted_price_range >= -threshold),
                                     1 * (volume_adjusted_price_range > threshold) - 1 * (volume_adjusted_price_range < -threshold))
    
    return alpha_factor
```
