```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low price to get the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the difference between close and open price to see if the stock closed higher or lower than it opened
    close_open_diff = df['close'] - df['open']
    
    # Calculate the weighted difference with volume, which indicates the momentum of price change
    weighted_diff = close_open_diff * df['volume']
    
    # Combine the daily range and weighted difference to create an alpha factor
    # This factor aims to capture both the volatility (range) and the momentum (weighted price change)
    factor_value = daily_range + weighted_diff
    
    return factor_value
```
