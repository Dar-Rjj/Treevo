```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].shift(30) - df['close']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].shift(10) - df['close']
    
    # Combine Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=25).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Close-to-Open Movement
    close_to_open_movement = df['open'] - df['close']
    
    # Calculate Daily Close-to-Close Return
    close_to_close_return = df['close'].diff()
    
    # Smooth with Exponential Moving Average (EMA)
    ema_smoothed_return = close_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount
    summarized_volume = df['volume'].rolling(window=5).sum()
    summarized_amount = df['amount'].rolling(window=5).sum()
    combined_volume_amount = summarized_volume + summarized_amount
    momentum_adjusted = ema_smoothed_return / combined_volume_amount
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0) * df['close'].diff()
    losses = (df['close'].diff() < 0) * df['close'].diff().abs()
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.35 * combined_momentum +
        0.15 * intraday_high_low_spread +
        0.10 * close_to_open_movement +
        0.10 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = (df['volume'] / df['volume'].shift(10))
    final_factor = weighted_average * volume_acceleration
    
    return final_factor
```
