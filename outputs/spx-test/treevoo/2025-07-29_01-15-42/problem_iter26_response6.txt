```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Smooth with EMA (10 days)
    ema_window = 10
    df['ema_close_to_open_return'] = df['close_to_open_return'].ewm(span=ema_window, min_periods=ema_window).mean()
    
    # Adjust for Volume and Amount
    df['total_volume'] = df['volume'].rolling(window=ema_window, min_periods=ema_window).sum()
    df['total_amount'] = df['amount'].rolling(window=ema_window, min_periods=ema_window).sum()
    df['adjusted_ema_return'] = df['ema_close_to_open_return'] / (df['total_volume'] + df['total_amount'])
    
    # Long-Term and Short-Term Momentum
    long_term_momentum = df['close'] / df['close'].shift(28)
    short_term_momentum = df['close'] / df['close'].shift(10)
    df['combined_momentum'] = long_term_momentum + short_term_momentum
    
    # Price Range Ratio
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Compute RTSI
    df['gains'] = df['close'].pct_change().apply(lambda x: max(x, 0))
    df['losses'] = df['close'].pct_change().apply(lambda x: -min(x, 0))
    df['ema_gains'] = df['gains'].ewm(span=14, min_periods=14).mean()
    df['ema_losses'] = df['losses'].ewm(span=14, min_periods=14).mean()
    df['rtpi'] = (df['ema_gains'] / (df['ema_gains'] + df['ema_losses'])) * 100
    
    # Realized Volatility
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20, min_periods=20).std() * np.sqrt(252)
    
    # Weighted Average of Combined Components
    df['weighted_average'] = (0.3 * df['combined_momentum'] 
                              + 0.2 * df['price_range_ratio'] 
                              + 0.1 * df['rtpi'] 
                              + 0.1 * df['realized_volatility'])
    
    # Dynamic Adjustment Factor
    df['ma_5'] = df['close'].rolling(window=5, min_periods=5).mean()
    df['ma_20'] = df['close'].rolling(window=20, min_periods=20).mean()
    df['dynamic_adjustment_factor'] = (df['ma_5'] - df['ma_20']) / df['ma_20']
    
    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['adjusted_ema_return'] * df['dynamic_adjustment_factor']
    
    return df['alpha_factor']
```
