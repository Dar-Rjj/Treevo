```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    long_term_momentum = df['close'] / df['close'].shift(28)
    short_term_momentum = df['close'] / df['close'].shift(7)
    combined_momentum = 0.5 * (long_term_momentum + short_term_momentum)
    
    # Calculate Realized Volatility
    daily_returns = df['close'].pct_change()
    realized_volatility = daily_returns.rolling(window=10).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_returns = (df['open'].shift(-1) - df['close']) / df['close']
    ema_smoothed_close_to_open_return = close_to_open_returns.ewm(span=10, min_periods=10).mean()
    
    # Adjust for Volume and Amount
    volume_ema = df['volume'].rolling(window=10).sum()
    amount_ema = df['amount'].rolling(window=10).sum()
    adjusted_ema_close_to_open = ema_smoothed_close_to_open_return / (volume_ema + amount_ema)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = np.where(daily_returns > 0, daily_returns, 0)
    losses = np.where(daily_returns < 0, -daily_returns, 0)
    avg_gain = gains.rolling(window=14).mean()
    avg_loss = losses.rolling(window=14).mean()
    rtsi = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.35 * combined_momentum +
        0.1 * intraday_high_low_spread +
        0.1 * adjusted_ema_close_to_open +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.2 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'] / df['volume'].shift(10)
    volume_change = df['volume'] / df['volume'].shift(1)
    final_factor = weighted_average * volume_acceleration * volume_change
    
    return final_factor
```
