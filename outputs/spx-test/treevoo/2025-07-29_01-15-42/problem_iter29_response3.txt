```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the short-term (5 days) and long-term (20 days) simple moving averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Compute the momentum indicator as the difference between the short-term and long-term SMAs
    df['Momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate the volume-weighted average price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Compute the difference between the close price and VWAP to identify overbought/oversold conditions
    df['Close_VWAP_Diff'] = df['close'] - df['VWAP']
    
    # Calculate the true range (TR) for each day
    df['Previous_Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['Previous_Close'])
    df['TR3'] = abs(df['low'] - df['Previous_Close'])
    df['TR'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Compute the average true range (ATR) over a 14-day period
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Create the alpha factor by dividing the momentum indicator by the ATR to account for volatility
    df['Alpha_Factor'] = df['Momentum'] / df['ATR']
    
    return df['Alpha_Factor']
```
