import pandas as pd
import numpy as np

def heuristics_v2(df, sma_window=5, rtsi_period=14, volume_ma_period=20, threshold=0.5):
    # Calculate Daily Price Change
    df['Close_Return'] = df['close'].pct_change()
    df['SMA_Return'] = df['Close_Return'].rolling(window=sma_window).mean()

    # Adjust for Volume and Amount
    df['Total_Volume_Sum'] = df['volume'].rolling(window=sma_window).sum()
    df['Total_Amount_Sum'] = df['amount'].rolling(window=sma_window).sum()
    df['Combined_VA'] = df['Total_Volume_Sum'] + df['Total_Amount_Sum']
    df['VA_Adjusted_Momentum'] = df['SMA_Return'] / df['Combined_VA']

    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['Price_Range_Ratio'] = df['Daily_Price_Range'] / df['open']

    # Compute Relative True Strength Index (RTSI)
    df['Gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Loss'] = df['close'].diff().apply(lambda x: abs(x) if x < 0 else 0)
    df['EMA_Gain'] = df['Gain'].ewm(span=rtsi_period, adjust=False).mean()
    df['EMA_Loss'] = df['Loss'].ewm(span=rtsi_period, adjust=False).mean()
    df['RTSI'] = (df['EMA_Gain'] / (df['EMA_Gain'] + df['EMA_Loss'])) * 100

    # Calculate Volume Momentum
    df['Volume_MA'] = df['volume'].rolling(window=volume_ma_period).mean()
    df['Volume_Momentum'] = df['volume'] - df['Volume_MA']

    # Combine Factors
    df['Combined_Factors'] = (df['Price_Range_Ratio'] * df['RTSI']) + df['Volume_Momentum']
    df['Combined_Factors'] *= df['VA_Adjusted_Momentum']

    # Generate Volume-Adjusted Price Range
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['Volume_MA']

    # Calculate Long-Term Price Momentum
    df['Long_Term_Momentum'] = df['close'].pct_change(periods=14)

    # Calculate Short-Term Price Momentum
    df['Short_Term_Momentum'] = df['close'].pct_change(periods=5)

    # Combine Long-Term and Short-Term Momentum
    df['Combined_Momentum'] = df['Long_Term_Momentum'] + df['Short_Term_Momentum']

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Movement
    df['Close_to_Open_Movement'] = df['close'] - df['open']

    # Weighted Average of Momentum and Intraday Components
    df['Weighted_Average'] = (df['Combined_Momentum'] * 0.4) + (df['Intraday_High_Low_Spread'] * 0.3) + (df['Close_to_Open_Movement'] * 0.3)

    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['volume'].diff(periods=5)

    # Generate Alpha Signal
    df['Momentum_Volume_Adjusted_Signal'] = df['Weighted_Average'] * df['Volume_Acceleration']

    # Final Alpha Factor
    def alpha_factor(row):
        if row['Volume_Adjusted_Price_Range'] > threshold:
            return 1 * (row['Momentum_Volume_Adjusted_Signal'] * row['Combined_Factors'])
        elif row['Volume_Adjusted_Price_Range'] < -threshold:
            return -1 * (row['Momentum_Volume_Adjusted_Signal'] * row['Combined_Factors'])
        else:
            return 0

    df['Alpha_Factor'] = df.apply(alpha_factor, axis=1)
    return df['Alpha_Factor']
