```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the range (open, high, low)
    df['close_range'] = (df['close'] - df[['open', 'high', 'low']].min(axis=1)) / (df[['open', 'high', 'low']].max(axis=1) - df[['open', 'high', 'low']].min(axis=1))
    
    # Calculate the price change momentum over the last 5 days
    df['momentum'] = df['close'].pct_change(periods=5)
    
    # Identify if the current day's close is higher than the previous 30-day moving average
    df['above_ma_30'] = (df['close'] > df['close'].rolling(window=30).mean()).astype(int)
    
    # Determine the accumulation/distribution line which helps in understanding buying and selling pressure
    df['money_flow_multiplier'] = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    df['money_flow_volume'] = df['money_flow_multiplier'] * df['volume']
    df['accumulation_distribution'] = df['money_flow_volume'].cumsum()
    
    # Create a final heuristic score by combining the above factors
    df['heuristic_score'] = df['close_range'] + df['momentum'] + df['above_ma_30'] + df['accumulation_distribution'].pct_change()
    
    return df['heuristic_score'].dropna()
```
