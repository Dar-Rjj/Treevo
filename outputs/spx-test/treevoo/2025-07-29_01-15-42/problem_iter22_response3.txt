```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']

    # Incorporate Volume Impact
    volume_impact_high_low = intraday_high_low_spread * df['volume']
    volume_impact_close_open = close_to_open_change * df['volume']

    # Analyze Volume Trends
    moving_avg_volume = df['volume'].rolling(window=20).mean()

    # Identify Positive and Negative Price-Volume Imbalances
    price_range = df['high'] - df['low']
    imbalance = price_range / moving_avg_volume
    threshold = 1.5
    positive_imbalance = (imbalance > threshold).astype(int)
    negative_imbalance = (imbalance < -threshold).astype(int)

    # Generate Preliminary Alpha Signal
    preliminary_alpha_signal = positive_imbalance - negative_imbalance

    # Calculate Long-Term and Short-Term Price Momentum
    long_term_momentum = df['close'].shift(28) - df['close']
    short_term_momentum = df['close'].shift(10) - df['close']
    combined_momentum = long_term_momentum + short_term_momentum

    # Adjust High-to-Low Range by Average Volume
    adjusted_high_low_range = (df['high'] - df['low']) / moving_avg_volume

    # Weighted Average of Combined Components
    weighted_average = (
        0.4 * combined_momentum +
        0.3 * volume_impact_high_low +
        0.2 * volume_impact_close_open +
        0.1 * adjusted_high_low_range
    )

    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()

    # Apply Volatility Filter
    volatility_filter = realized_volatility < realized_volatility.median()
    alpha_signal = preliminary_alpha_signal.where(volatility_filter, 0)

    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] / df['volume'].shift(10)

    # Generate Final Alpha Signal
    final_alpha_signal = pd.Series(index=df.index, dtype=float)
    for date in df.index:
        if preliminary_alpha_signal.loc[date] > 0 and volatility_filter.loc[date]:
            final_alpha_signal.loc[date] = combined_momentum.loc[date] + preliminary_alpha_signal.loc[date]
        elif preliminary_alpha_signal.loc[date] < 0 and volatility_filter.loc[date]:
            final_alpha_signal.loc[date] = combined_momentum.loc[date] + preliminary_alpha_signal.loc[date]
        elif preliminary_alpha_signal.loc[date] == 0 and volatility_filter.loc[date]:
            final_alpha_signal.loc[date] = combined_momentum.loc[date]
        else:
            final_alpha_signal.loc[date] = weighted_average.loc[date] * volume_acceleration.loc[date]

    return final_alpha_signal
```
