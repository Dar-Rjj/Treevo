```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_shifted'] = df['close'].shift(1)
    df['gains'] = df['close'] - df['close_shifted']
    df['losses'] = -df['gains']
    df.loc[df['gains'] < 0, 'gains'] = 0
    df.loc[df['losses'] < 0, 'losses'] = 0
    
    # Smooth Gains and Losses with 14-period Simple Moving Average
    df['smoothed_gains'] = df['gains'].rolling(window=14).mean()
    df['smoothed_losses'] = df['losses'].rolling(window=14).mean()
    
    # Calculate RTSI
    df['rTSI'] = (df['smoothed_gains'] / (df['smoothed_gains'] + df['smoothed_losses'])) * 100
    
    # Calculate Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    
    # Combine Factors
    df['alpha_factor'] = df['price_range_ratio'] * df['rTSI'] * df['volume_volatility']
    
    return df['alpha_factor']
```
