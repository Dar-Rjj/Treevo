import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust High-to-Low Range by Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_high_low_range = high_low_range / avg_volume
    
    # Calculate Intraday High-Low Spread
    intraday_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_open_movement = (df['close'] - df['open']) / df['open']
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.4 * combined_momentum +
        0.3 * intraday_spread +
        0.2 * close_open_movement +
        0.1 * adjusted_high_low_range
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(10)
    
    # Final Factor
    final_factor = weighted_average * volume_acceleration
    final_factor = final_factor * (df['volume'] / df['volume'].shift(1))
    
    return final_factor
