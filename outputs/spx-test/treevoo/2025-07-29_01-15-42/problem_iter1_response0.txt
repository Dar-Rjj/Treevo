import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    close_returns = df['close'].pct_change()
    realized_volatility = close_returns.rolling(window=20).std()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_factor = high_to_low_range / avg_volume
    
    # Final factor
    final_factor = (high_to_low_range * realized_volatility) / avg_volume
    
    return final_factor
