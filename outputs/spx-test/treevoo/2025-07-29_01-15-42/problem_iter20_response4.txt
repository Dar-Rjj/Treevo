```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Incorporate Volume Impact
    volume_adjusted_high_low_spread = intraday_high_low_spread * df['volume']
    volume_adjusted_daily_price_change = daily_price_change * df['volume']
    
    # Calculate Moving Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_volume_signal = volume_adjusted_daily_price_change / moving_average_volume
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std() * (252**0.5)
    
    # Compute Long-Term Price Momentum
    long_term_momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Compute Short-Term Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.005
    positive_imbalance = (volume_adjusted_daily_price_change > threshold) * 1
    negative_imbalance = (volume_adjusted_daily_price_change < -threshold) * -1
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha_signal = positive_imbalance + negative_imbalance
    preliminary_alpha_signal[preliminary_alpha_signal == 0] = 0
    
    # Calculate Momentum Score
    momentum_score = combined_momentum - combined_momentum.shift(1)
    
    # Apply Volatility Filter
    median_volatility = realized_volatility.median()
    alpha_signal = preliminary_alpha_signal.copy()
    alpha_signal[realized_volatility >= median_volatility] = 0
    
    # Adjust High-to-Low Range by Average Volume
    average_volume_15_days = df['volume'].rolling(window=15).mean()
    adjusted_high_low_range = (intraday_high_low_spread / average_volume_15_days)
    
    # Weighted Average of Combined Components
    weighted_combined = (
        combined_momentum * 0.4 +
        adjusted_high_low_range * 0.3 +
        daily_price_change * 0.2 +
        adjusted_high_low_range * 0.1
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] / df['volume'].shift(15))
    
    # Final Alpha Signal Based on Combined Indicators
    final_alpha_signal = pd.Series(index=df.index, data=0.0)
    final_alpha_signal[(preliminary_alpha_signal > 0) & (realized_volatility < median_volatility)] = (alpha_signal + momentum_score)[(preliminary_alpha_signal > 0) & (realized_volatility < median_volatility)]
    final_alpha_signal[(preliminary_alpha_signal < 0) & (realized_volatility < median_volatility)] = (alpha_signal - momentum_score)[(preliminary_alpha_signal < 0) & (realized_volatility < median_volatility)]
    final_alpha_signal[(preliminary_alpha_signal == 0) & (realized_volatility < median_volatility)] = momentum_score[(preliminary_alpha_signal == 0) & (realized_volatility < median_volatility)]
    final_alpha_signal[realized_volatility >= median_volatility] = weighted_combined[realized_volatility >= median_volatility] * volume_acceleration[realized_volatility >= median_volatility]
    
    return final_alpha_signal
```
