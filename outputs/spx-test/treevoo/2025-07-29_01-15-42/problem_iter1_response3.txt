def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate the Midpoint of the Intraday Range
    range_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Range Midpoint Deviation
    close_to_mid_deviation = df['close'] - range_midpoint
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Incorporate Volume Trend: Multiply by Close to Midpoint Deviation, weighted by Volume Change
    intraday_reversal_with_volume_momentum = close_to_mid_deviation * volume_change
    
    return intraday_reversal_with_volume_momentum
