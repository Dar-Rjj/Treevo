```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = df['open'] - df['close']
    
    # Weighted Average of Momentum and Intraday Components
    weights = [0.5, 0.25, 0.25]
    weighted_average = (combined_momentum * weights[0] +
                        intraday_high_low_spread * weights[1] +
                        close_to_open_movement * weights[2])
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff().clip(lower=0).fillna(0).rolling(window=14).sum()
             / 14)
    losses = (df['close'].diff().clip(upper=0).fillna(0).rolling(window=14).sum()
              / 14)
    rtsi = 100 - (100 / (1 + gains / abs(losses)))
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=30).std()
    
    # Combine Price and Volume Intensity
    intensity = price_range_ratio * rtsi * volume_volatility
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(10)
    
    # Multiply Weighted Average by Volume Acceleration
    weighted_average *= volume_acceleration
    
    # Adjust for Volume
    sma_volume = df['volume'].rolling(window=14).sum()
    adjusted_weighted_average = weighted_average / sma_volume
    
    # Combine Intensity and Momentum Factors
    alpha_factor = intensity * adjusted_weighted_average
    
    return alpha_factor
```
