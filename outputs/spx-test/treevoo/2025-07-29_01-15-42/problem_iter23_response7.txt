```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()

    # Smooth Returns with EMA (5 days)
    df['ema_smoothed_close_return'] = df['close_to_close_return'].ewm(span=5, adjust=False).mean()

    # Adjust for Volume and Amount over the EMA period
    df['volume_ema_period'] = df['volume'].rolling(window=5).sum()
    df['amount_ema_period'] = df['amount'].rolling(window=5).sum()
    df['adjusted_momentum'] = df['ema_smoothed_close_return'] / (df['volume_ema_period'] + df['amount_ema_period'])

    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(-1)

    # Smooth with EMA (10 days)
    df['ema_smoothed_close_open_return'] = df['close_to_open_return'].ewm(span=10, adjust=False).mean()

    # Adjust for Volume and Amount over the EMA period
    df['volume_ema_period_close_open'] = df['volume'].rolling(window=10).sum()
    df['amount_ema_period_close_open'] = df['amount'].rolling(window=10).sum()
    df['adjusted_momentum_close_open'] = df['ema_smoothed_close_open_return'] / (df['volume_ema_period_close_open'] + df['amount_ema_period_close_open'])

    # Calculate Long-Term Price Momentum (28 days)
    df['long_term_momentum'] = df['close'] / df['close'].shift(28)

    # Calculate Short-Term Price Momentum (10 days)
    df['short_term_momentum'] = df['close'] / df['close'].shift(10)

    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']

    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Relative True Strength Index (RTSI) (14 periods)
    df['gains'] = np.where(df['close'] > df['close'].shift(), df['close'] - df['close'].shift(), 0)
    df['losses'] = np.where(df['close'] < df['close'].shift(), df['close'].shift() - df['close'], 0)
    df['smoothed_gains'] = df['gains'].ewm(span=14, adjust=False).mean()
    df['smoothed_losses'] = df['losses'].ewm(span=14, adjust=False).mean()
    df['rtsi'] = (df['smoothed_gains'] / (df['smoothed_gains'] + df['smoothed_losses'])) * 100

    # Calculate Intraday Volatility (ATR) (14 periods)
    df['tr'] = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()

    # Calculate Realized Volatility (20 days)
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std()

    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = df['close'] - df['open']

    # Weighted Average of Combined Components
    df['weighted_average'] = (0.3 * df['combined_momentum'] +
                              0.2 * df['high_low_spread'] +
                              0.1 * df['close_to_open_movement'] +
                              0.2 * df['price_range'] +
                              0.1 * df['rtsi'] +
                              0.1 * df['realized_volatility'])

    # Adjust by Volume and Amount
    df['volume_adjustment'] = df['volume'] / df['volume'].shift()
    df['amount_adjustment'] = df['amount'] / df['amount'].shift()
    df['adjustment_factor'] = df['volume_adjustment'] + df['amount_adjustment']

    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['adjustment_factor']

    return df['alpha_factor']
```
