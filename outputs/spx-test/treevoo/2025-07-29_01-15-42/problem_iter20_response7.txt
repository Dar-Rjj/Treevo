```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous close to the current open
    df['prev_close_to_open'] = df['open'] - df['close'].shift(1)
    
    # Calculate the difference between the highest and lowest price of the day
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Construct an alpha factor that combines the three components
    # The idea is to capture the momentum (price direction) with volatility (high-low spread) and liquidity (volume change)
    df['alpha_factor'] = (df['prev_close_to_open'] / df['high_low_diff']) * df['volume_change']
    
    return df['alpha_factor']
```
