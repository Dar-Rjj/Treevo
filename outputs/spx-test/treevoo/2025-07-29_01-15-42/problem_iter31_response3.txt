```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Smooth Returns with EMA
    ema_window = 10
    df['ema_smoothed_return'] = df['daily_return'].ewm(span=ema_window, adjust=False).mean()

    # Adjust for Volume and Amount
    df['total_volume'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount'] = df['amount'].rolling(window=ema_window).sum()
    df['adjusted_momentum'] = df['ema_smoothed_return'] / (df['total_volume'] + df['total_amount'])

    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(28)) / df['close'].shift(28)

    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)

    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']

    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']

    # Calculate Relative True Strength Index (RTSI)
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rTSI'] = (df['avg_gain'] / (df['avg_gain'] + df['avg_loss'])) * 100

    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std()

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Weighted Average of Combined Components
    df['weighted_average'] = (
        0.35 * df['combined_momentum'] +
        0.2 * df['intraday_high_low_spread'] +
        0.15 * df['daily_price_range'] +
        0.1 * df['rTSI'] +
        0.1 * df['realized_volatility']
    )

    # Adjust by Volume and Amount
    df['volume_amount_change_ratio'] = (df['volume'] + df['amount']) / (df['volume'].shift(1) + df['amount'].shift(1))

    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['volume_amount_change_ratio']

    return df['alpha_factor']
```
